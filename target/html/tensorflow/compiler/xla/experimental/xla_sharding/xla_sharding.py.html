  30 class Sharding(object):
  31   """A class to support adding sharding attributes to Ops.
  32 
  33   Use the factory constructors and then call apply_to_tensor:
  34     Sharding.replicate().apply_to_tensor(tensor)
  35   """
  36 
  37   def __init__(self, proto=None):
  38     """Do not use this constructor; use the factory functions below."""
  39     self._proto = proto
  40 
  41   @classmethod
  42   def replicate(cls):
  43     """Returns a replicated sharding attribute.
  44 
  45     This causes an op to be computed in its entirety independently on all
  46     cores in the XLA device.
  47     """
  48     return Sharding(
  49         proto=xla_data_pb2.OpSharding(type=xla_data_pb2.OpSharding.REPLICATED))
  50 
  51   @classmethod
  52   def assign_device(cls, core):
  53     """Returns an AssignDevice sharding attribute.
  54 
  55     This causes an op to be computed in its entirety only on one core in
  56     the XLA device.
  57     Args:
  58       core: The core to assign this Op to.
  59     """
  60     return Sharding(
  61         proto=xla_data_pb2.OpSharding(
  62             type=xla_data_pb2.OpSharding.MAXIMAL,
  63             tile_assignment_dimensions=[1],
  64             tile_assignment_devices=[core]))
  65 
  66   @classmethod
  67   def tile(cls, tile_shape, tile_assignment):
  68     """Returns a Tiled sharding attribute.
  69 
  70     This causes an op to be partially computed on multiple cores in the
  71     XLA device.
  72 
  73     Args:
  74       tile_shape: A xla_shape.Shape describing the tile shape that each core
  75         will compute.
  76         The tile shape does not need to be divisible by the tile assignment.
  77       tile_assignment: An np.ndarray describing the topology of the tiling and
  78         which device will compute which part of the topology.
  79 
  80     Raises:
  81       TypeError: tile_assignment was not of np.array type or tile_shape was
  82          not of xla_shape.Shape type.
  83 
  84     TODO(jmolloy): This concept is nefarious and is not
  85     something we really want to expose to users (especially as the
  86     contract for tile_assignment is very strict).
  87     """
  88     if not isinstance(tile_assignment, _np.ndarray):
  89       raise TypeError('Tile assignment must be of type np.ndarray')
  90     if not isinstance(tile_shape, xla_shape.Shape):
  91       raise TypeError('Tile shape must be of type xla_shape.Shape')
  92     dims = list(tile_assignment.shape)
  93     flattened_devices = tile_assignment.reshape(-1, order='C')
  94     return Sharding(
  95         proto=xla_data_pb2.OpSharding(
  96             type=xla_data_pb2.OpSharding.OTHER,
  97             tile_shape=tile_shape.message,
  98             tile_assignment_dimensions=dims,
  99             tile_assignment_devices=list(flattened_devices)))
 100 
 101   @classmethod
 102   def split(cls, tensor, split_dimension, num_devices):
 103     """Returns a Sharding that splits a tensor across a dimension.
 104 
 105     This creates a Tiled attribute, similar to tile(), but easier to use for the
 106     common case of tiling a tensor N ways in one dimension.
 107 
 108     Args:
 109       tensor: A tf.Tensor to split.
 110       split_dimension: The dimension number to split.
 111       num_devices: The number of cores to split `tensor` over.
 112 
 113     Raises:
 114       ValueError: The tensor to split was smaller in the split dimension than
 115         the number of devices to split over.
 116     """
 117     tensor.shape.assert_is_fully_defined()
 118     shape = tensor.shape.as_list()
 119     if shape[split_dimension] < num_devices:
 120       raise ValueError('Split dimension was smaller than the required number '
 121                        'of splits: shape=%r, dimension=%r, num_devices=%r',
 122                        shape, split_dimension, num_devices)
 123 
 124     tile_shape = shape
 125     tile_shape[split_dimension] = int(
 126         math.ceil(tile_shape[split_dimension] / num_devices))
 127     tile_shape_proto = xla_data_pb2.Shape(
 128         element_type=xla_data_pb2.F32, dimensions=tile_shape)
 129 
 130     tile_assignment_dims = [1] * len(shape)
 131     tile_assignment_dims[split_dimension] = num_devices
 132 
 133     return Sharding(
 134         proto=xla_data_pb2.OpSharding(
 135             type=xla_data_pb2.OpSharding.OTHER,
 136             tile_shape=tile_shape_proto,
 137             tile_assignment_dimensions=tile_assignment_dims,
 138             tile_assignment_devices=range(num_devices)))
 139 
 140   def apply_to_tensor(self, tensor):
 141     """Applies this Sharding attribute to `tensor`."""
 142     if len(tensor.op.outputs) > 1:
 143       proto = self._get_or_create_tuple_proto(tensor.op)
 144       # We can't mutate an element of old_proto.tuple_shardings, so create
 145       # a new proto.
 146       tuple_shardings = list(proto.tuple_shardings)
 147       tuple_shardings[tensor.value_index] = self._proto
 148       proto = xla_data_pb2.OpSharding(
 149           type=xla_data_pb2.OpSharding.TUPLE, tuple_shardings=tuple_shardings)
 150     else:
 151       proto = self._proto
 152 
 153     attr_value = attr_value_pb2.AttrValue(s=proto.SerializeToString())
 154     # TODO(jmolloy): This need to be seriously revisited before declaring this
 155     # API available for public use.
 156     # pylint: disable=protected-access
 157     tensor.op._set_attr('_XlaSharding', attr_value)
 158 
 159   @property
 160   def proto(self):
 161     """Return the sharding protobuf of type xla_data_pb2.OpSharding."""
 162     return self._proto
 163 
 164   def _get_or_create_tuple_proto(self, op):
 165     try:
 166       attr = op.get_attr('_XlaSharding')
 167       proto = xla_data_pb2.OpSharding()
 168       proto.ParseFromString(attr)
 169       return proto
 170     except ValueError:
 171       return self._create_tuple_proto(op)
 172 
 173   def _create_tuple_proto(self, op):
 174     shardings = [
 175         xla_data_pb2.OpSharding(type=xla_data_pb2.OpSharding.REPLICATED)
 176         for _ in op.outputs
 177     ]
 178     return xla_data_pb2.OpSharding(
 179         type=xla_data_pb2.OpSharding.TUPLE, tuple_shardings=shardings)

 182 # Helpers for the above factory functions that allow easy application of
 183 # shardings, for example:

 187 def replicate(tensor):
 188   Sharding.replicate().apply_to_tensor(tensor)
 189   return tensor

 192 def assign_device(tensor, device):
 193   Sharding.assign_device(device).apply_to_tensor(tensor)
 194   return tensor

 197 def tile(tensor, tile_shape, tile_assignment):
 198   Sharding.tile(tile_shape, tile_assignment).apply_to_tensor(tensor)
 199   return tensor
