  27 class Shape(object):
  28   """Wraps a xla_data_pb2.Shape message with a convenient Python type.
  29 
  30   Provides direct access to the underlying xla_data_pb2.Shape message in the
  31   message attribute, along with accessor wrappers to the message's fields.
  32   Avoid direct access to .message unless interacting directly with protobuf APIs
  33   like CopyFrom. In other words, prefer hauling the shape around in a Shape, and
  34   only access .message when strictly required by the protobuf API.
  35   """
  36 
  37   def __init__(self, element_type, dimensions, layout=None):
  38     """Creates a new XLA Shape.
  39 
  40     Args:
  41       element_type: element type from xla_data_pb2.
  42       dimensions: sequence of dimensions sizes (integers), or sequence
  43         of Shapes in the case of a tuple, i.e. when element_type is
  44         TUPLE.
  45       layout: optional minor_to_major sequence for layout. If not given, the
  46         default major-to-minor layout is used.
  47 
  48     Raises:
  49       ValueError: if element_type is TUPLE but dimensions are not Shape objects.
  50     """
  51     self.message = xla_data_pb2.Shape()
  52     self.message.element_type = element_type
  53     if element_type == xla_data_pb2.TUPLE:
  54       if not all(isinstance(subshape, Shape) for subshape in dimensions):
  55         raise ValueError(
  56             'XLA tuple requires sequence of Shape objects as dimensions')
  57       self._tuple_shapes = tuple(dimensions)
  58       for component_shape in self._tuple_shapes:
  59         component_message = self.message.tuple_shapes.add()
  60         component_message.CopyFrom(component_shape.message)
  61     else:
  62       self.message.dimensions.extend(dimensions)
  63       if layout is None:
  64         layout = list(reversed(range(len(dimensions))))
  65       self.message.layout.format = xla_data_pb2.DENSE
  66       self.message.layout.minor_to_major.extend(layout)
  67 
  68   def element_type(self):
  69     return self.message.element_type
  70 
  71   def is_tuple(self):
  72     return self.element_type() == xla_data_pb2.TUPLE
  73 
  74   def dimensions(self):
  75     if self.is_tuple():
  76       raise ValueError('Tuple shape has no dimensions. Try tuple_shapes()?')
  77     return self.message.dimensions
  78 
  79   def tuple_shapes(self):
  80     """If this is a tuple, returns its sequence of constituent Shape objects.
  81 
  82     Returns:
  83       Tuple sub-shapes.
  84 
  85     Raises:
  86       ValueError: if this is not a tuple.
  87     """
  88     if not self.is_tuple():
  89       raise ValueError('tuple_shapes() called on a non-tuple shape')
  90     return self._tuple_shapes
  91 
  92   def layout(self):
  93     return self.message.layout
  94 
  95   @staticmethod
  96   def from_pyval(pyval):
  97     return CreateShapeFromNumpy(pyval)

 100 def _CreateShapeFromNumpy(ndarray):  # pylint: disable=invalid-name
 101   """Create a Shape from a given Numpy array.
 102 
 103   Args:
 104     ndarray: Numpy array.
 105 
 106   Returns:
 107     A Shape object.
 108   """
 109   element_type = types.MAP_DTYPE_TO_RECORD[str(ndarray.dtype)].primitive_type
 110   dimensions = ndarray.shape
 111 
 112   # Set the shape's layout based on the ordering of ndarray.
 113   # Numpy arrays come in two orders: Fortran (column-major) and C (row-major).
 114   if _np.isfortran(ndarray):
 115     # Column-major layout. This corresponds to a "dimension order is
 116     # minor-to-major" layout in XLA.
 117     layout = range(ndarray.ndim)
 118   else:
 119     # Row-major layout. This corresponds to a "dimension order is
 120     # major-to-minor" layout int XLA.
 121     layout = list(reversed(xrange(ndarray.ndim)))
 122 
 123   return Shape(element_type, dimensions, layout)

 126 def CreateShapeFromNumpy(value):  # pylint: disable=invalid-name
 127   """Create a Shape from a Numpy array or a nested tuple structure thereof.
 128 
 129   Args:
 130     value: Numpy array or (possibly nested) tuple structure that bottoms out in
 131       Numpy arrays.
 132 
 133   Returns:
 134     A Shape object.
 135   """
 136   if isinstance(value, tuple):
 137     return Shape(
 138         xla_data_pb2.TUPLE,
 139         [CreateShapeFromNumpy(component) for component in value])
 140   else:
 141     return _CreateShapeFromNumpy(value)

 144 def CreateShapeFromDtypeAndTuple(dtype, shape_tuple):  # pylint: disable=invalid-name
 145   """Create a shape from a Numpy dtype and a sequence of nonnegative integers.
 146 
 147   Args:
 148     dtype: a numpy dtype, e.g. np.dtype('int32').
 149     shape_tuple: a sequence of nonnegative integers.
 150 
 151   Returns:
 152     A Shape object.
 153   """
 154   element_type = types.MAP_DTYPE_TO_RECORD[str(dtype)].primitive_type
 155   return Shape(element_type, shape_tuple)
