  28 # Records corresponsence between a XLA primitive type and Python/Numpy types.
  29 #
  30 # primitive_type: value of type xla_data_pb2.PrimitiveType
  31 # numpy_dtype: corresponsing Numpy "dtype" (like np.float32)
  32 # literal_field_name: name of the field in the LiteralProto message elements
  33 # of this type go into.
  34 # literal_field_type: type of the field named 'literal_field_name'.
  35 #
  36 # TODO(eliben): figure out how to avoid knowing the extra Python type and the
  37 # astype cast when writing into Literals.
  38 TypeConversionRecord = collections.namedtuple('TypeConversionRecord', [
  39     'primitive_type', 'numpy_dtype', 'literal_field_name', 'literal_field_type'
  40 ])
  41

 124 # Maps from Numpy dtypes to TypeConversionRecord.
 125 # Note the conversion on the key. Numpy has a known issue wherein dtype hashing
 126 # doesn't work as expected (https://github.com/numpy/numpy/issues/7242). Thus,
 127 # when keying by dtype in this dict, we use the string form of dtypes.
 128 MAP_DTYPE_TO_RECORD = {
 129     str(_np.dtype(record.numpy_dtype)): record
 130     for record in MAP_XLA_TYPE_TO_RECORD.values()
 131 }
