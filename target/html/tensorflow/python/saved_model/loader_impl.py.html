  41 def _parse_saved_model(export_dir):
  42   """Reads the savedmodel.pb or savedmodel.pbtxt file containing `SavedModel`.
  43 
  44   Args:
  45     export_dir: Directory containing the SavedModel file.
  46 
  47   Returns:
  48     A `SavedModel` protocol buffer.
  49 
  50   Raises:
  51     IOError: If the file does not exist, or cannot be successfully parsed.
  52   """
  53   # Build the path to the SavedModel in pbtxt format.
  54   path_to_pbtxt = os.path.join(
  55       compat.as_bytes(export_dir),
  56       compat.as_bytes(constants.SAVED_MODEL_FILENAME_PBTXT))
  57   # Build the path to the SavedModel in pb format.
  58   path_to_pb = os.path.join(
  59       compat.as_bytes(export_dir),
  60       compat.as_bytes(constants.SAVED_MODEL_FILENAME_PB))
  61 
  62   # Parse the SavedModel protocol buffer.
  63   saved_model = saved_model_pb2.SavedModel()
  64   if file_io.file_exists(path_to_pb):
  65     try:
  66       file_content = file_io.FileIO(path_to_pb, "rb").read()
  67       saved_model.ParseFromString(file_content)
  68       return saved_model
  69     except message.DecodeError as e:
  70       raise IOError("Cannot parse file %s: %s." % (path_to_pb, str(e)))
  71   elif file_io.file_exists(path_to_pbtxt):
  72     try:
  73       file_content = file_io.FileIO(path_to_pbtxt, "rb").read()
  74       text_format.Merge(file_content.decode("utf-8"), saved_model)
  75       return saved_model
  76     except text_format.ParseError as e:
  77       raise IOError("Cannot parse file %s: %s." % (path_to_pbtxt, str(e)))
  78   else:
  79     raise IOError("SavedModel file does not exist at: %s/{%s|%s}" %
  80                   (export_dir,
  81                    constants.SAVED_MODEL_FILENAME_PBTXT,
  82                    constants.SAVED_MODEL_FILENAME_PB))

  85 def _get_asset_tensors(export_dir, meta_graph_def_to_load, import_scope=None):
  86   """Gets the asset tensors, if defined in the meta graph def to load.
  87 
  88   Args:
  89     export_dir: Directory where the SavedModel is located.
  90     meta_graph_def_to_load: The meta graph def from the SavedModel to be loaded.
  91     import_scope: Optional `string` -- if specified, prepend this followed by
  92         '/' to all returned asset tensor names.
  93 
  94   Returns:
  95     A dictionary of asset tensors, keyed by the name of the asset tensor. The
  96     value in the map corresponds to the absolute path of the asset file.
  97   """
  98   # Collection-def that may contain the assets key.
  99   collection_def = meta_graph_def_to_load.collection_def
 100 
 101   asset_tensor_dict = {}
 102   if constants.ASSETS_KEY in collection_def:
 103     # Location of the assets for SavedModel.
 104     assets_directory = os.path.join(
 105         compat.as_bytes(export_dir),
 106         compat.as_bytes(constants.ASSETS_DIRECTORY))
 107     assets_any_proto = collection_def[constants.ASSETS_KEY].any_list.value
 108     # Process each asset and add it to the asset tensor dictionary.
 109     for asset_any_proto in assets_any_proto:
 110       asset_proto = meta_graph_pb2.AssetFileDef()
 111       asset_any_proto.Unpack(asset_proto)
 112       tensor_name = asset_proto.tensor_info.name
 113       if import_scope:
 114         tensor_name = "%s/%s" % (import_scope, tensor_name)
 115       asset_tensor_dict[tensor_name] = os.path.join(
 116           compat.as_bytes(assets_directory),
 117           compat.as_bytes(asset_proto.filename))
 118   return asset_tensor_dict

 121 def _get_main_op_tensor(
 122     meta_graph_def_to_load, init_op_key=constants.MAIN_OP_KEY):
 123   """Gets the main op tensor, if one exists.
 124 
 125   Args:
 126     meta_graph_def_to_load: The meta graph def from the SavedModel to be loaded.
 127     init_op_key: name of collection to check; should be one of MAIN_OP_KEY
 128       or the deprecated LEGACY_INIT_OP_KEY
 129 
 130   Returns:
 131     The main op tensor, if it exists and `None` otherwise.
 132 
 133   Raises:
 134     RuntimeError: If the collection def corresponding to the main op key has
 135         other than exactly one tensor.
 136   """
 137   collection_def = meta_graph_def_to_load.collection_def
 138   main_op_tensor = None
 139   if init_op_key in collection_def:
 140     main_ops = collection_def[init_op_key].node_list.value
 141     if len(main_ops) != 1:
 142       raise RuntimeError("Expected exactly one SavedModel main op. "
 143                          "Found: {}".format(main_ops))
 144     main_op_tensor = ops.get_collection(init_op_key)[0]
 145   return main_op_tensor

 148 @tf_export("saved_model.maybe_saved_model_directory",
 149            "saved_model.loader.maybe_saved_model_directory")
 150 @deprecation.deprecated_endpoints(
 151     "saved_model.loader.maybe_saved_model_directory")
 152 def maybe_saved_model_directory(export_dir):
 153   """Checks whether the provided export directory could contain a SavedModel.
 154 
 155   Note that the method does not load any data by itself. If the method returns
 156   `false`, the export directory definitely does not contain a SavedModel. If the
 157   method returns `true`, the export directory may contain a SavedModel but
 158   provides no guarantee that it can be loaded.
 159 
 160   Args:
 161     export_dir: Absolute string path to possible export location. For example,
 162                 '/my/foo/model'.
 163 
 164   Returns:
 165     True if the export directory contains SavedModel files, False otherwise.
 166   """
 167   txt_path = os.path.join(export_dir, constants.SAVED_MODEL_FILENAME_PBTXT)
 168   pb_path = os.path.join(export_dir, constants.SAVED_MODEL_FILENAME_PB)
 169   return file_io.file_exists(txt_path) or file_io.file_exists(pb_path)

 172 @tf_export("saved_model.load", "saved_model.loader.load")
 173 def load(sess, tags, export_dir, import_scope=None, **saver_kwargs):
 174   """Loads the model from a SavedModel as specified by tags.
 175 
 176   Args:
 177     sess: The TensorFlow session to restore the variables.
 178     tags: Set of string tags to identify the required MetaGraphDef. These should
 179         correspond to the tags used when saving the variables using the
 180         SavedModel `save()` API.
 181     export_dir: Directory in which the SavedModel protocol buffer and variables
 182         to be loaded are located.
 183     import_scope: Optional `string` -- if specified, prepend this string
 184         followed by '/' to all loaded tensor names. This scope is applied to
 185         tensor instances loaded into the passed session, but it is *not* written
 186         through to the static `MetaGraphDef` protocol buffer that is returned.
 187     **saver_kwargs: Optional keyword arguments passed through to Saver.
 188 
 189   Returns:
 190     The `MetaGraphDef` protocol buffer loaded in the provided session. This
 191     can be used to further extract signature-defs, collection-defs, etc.
 192 
 193   Raises:
 194     RuntimeError: MetaGraphDef associated with the tags cannot be found.
 195   """
 196   loader = SavedModelLoader(export_dir)
 197   return loader.load(sess, tags, import_scope, **saver_kwargs)

 200 class SavedModelLoader(object):
 201   """Load graphs and restore variable values from a `SavedModel`."""
 202 
 203   def __init__(self, export_dir):
 204     """Creates a `SavedModelLoader`.
 205 
 206     Args:
 207       export_dir: Directory in which the SavedModel protocol buffer and
 208         variables to be loaded are located.
 209     """
 210     self._export_dir = export_dir
 211     self._variables_path = saved_model_utils.get_variables_path(export_dir)
 212     self._saved_model = _parse_saved_model(export_dir)
 213 
 214   @property
 215   def export_dir(self):
 216     """Directory containing the SavedModel."""
 217     return self._export_dir
 218 
 219   @property
 220   def variables_path(self):
 221     """Path to variable checkpoint files."""
 222     return self._variables_path
 223 
 224   @property
 225   def saved_model(self):
 226     """SavedModel object parsed from the export directory."""
 227     return self._saved_model
 228 
 229   def get_meta_graph_def_from_tags(self, tags):
 230     """Return MetaGraphDef with the exact specified tags.
 231 
 232     Args:
 233       tags: A list or set of string tags that identify the MetaGraphDef.
 234 
 235     Returns:
 236       MetaGraphDef with the same tags.
 237 
 238     Raises:
 239       RuntimeError: if no metagraphs were found with the associated tags.
 240     """
 241     found_match = False
 242     for meta_graph_def in self._saved_model.meta_graphs:
 243       if set(meta_graph_def.meta_info_def.tags) == set(tags):
 244         meta_graph_def_to_load = meta_graph_def
 245         found_match = True
 246         break
 247 
 248     if not found_match:
 249       raise RuntimeError(
 250           "MetaGraphDef associated with tags " + str(tags).strip("[]") +
 251           " could not be found in SavedModel. To inspect available tag-sets in"
 252           " the SavedModel, please use the SavedModel CLI: `saved_model_cli`"
 253       )
 254     return meta_graph_def_to_load
 255 
 256   def load_graph(self, graph, tags, import_scope=None, **saver_kwargs):
 257     """Load ops and nodes from SavedModel MetaGraph into graph.
 258 
 259     Args:
 260       graph: tf.Graph object.
 261       tags: a set of string tags identifying a MetaGraphDef.
 262       import_scope: Optional `string` -- if specified, prepend this string
 263         followed by '/' to all loaded tensor names. This scope is applied to
 264         tensor instances loaded into the passed session, but it is *not* written
 265         through to the static `MetaGraphDef` protocol buffer that is returned.
 266       **saver_kwargs: keyword arguments to pass to tf.train.import_meta_graph.
 267 
 268     Returns:
 269       A tuple of
 270         * Saver defined by the MetaGraph, which can be used to restore the
 271           variable values.
 272         * List of `Operation`/`Tensor` objects returned from
 273           `tf.import_graph_def` (may be `None`).
 274     """
 275     meta_graph_def = self.get_meta_graph_def_from_tags(tags)
 276     with graph.as_default():
 277       return tf_saver._import_meta_graph_with_return_elements(  # pylint: disable=protected-access
 278           meta_graph_def, import_scope=import_scope, **saver_kwargs)
 279 
 280   def restore_variables(self, sess, saver, import_scope=None):
 281     """Restore SavedModel variable values into the session.
 282 
 283     Args:
 284       sess: tf.Session to restore variable values.
 285       saver: a tf.train.Saver object. Can be None if there are no variables in
 286         graph. This may be the saver returned by the load_graph() function, or a
 287         default `tf.train.Saver()`.
 288       import_scope: Optional `string` -- if specified, prepend this string
 289         followed by '/' to all loaded tensor names. This scope is applied to
 290         tensor instances loaded into the passed session, but it is *not* written
 291         through to the static `MetaGraphDef` protocol buffer that is returned.
 292 
 293     Raises:
 294       ValueError: if no saver was passed to the saver argument, and there are
 295         variables in the graph.
 296     """
 297     with sess.graph.as_default():
 298       if (saver is None and
 299           not variables._all_saveable_objects(scope=import_scope)):  # pylint: disable=protected-access
 300         tf_logging.info("The specified SavedModel has no variables; no "
 301                         "checkpoints were restored.")
 302       elif isinstance(saver, tf_saver.Saver):
 303         saver.restore(sess, self._variables_path)
 304       else:
 305         raise ValueError(
 306             "No tf.train.Saver object was passed to the function "
 307             "SavedModelLoader.restore_variables. Since there are variables in "
 308             "the graph, a saver is required.")
 309 
 310   def run_init_ops(self, sess, tags, import_scope=None):
 311     """Run initialization ops defined in the `MetaGraphDef`.
 312 
 313     Args:
 314       sess: tf.Session to restore variable values.
 315       tags: a set of string tags identifying a MetaGraphDef.
 316       import_scope: Optional `string` -- if specified, prepend this string
 317         followed by '/' to all loaded tensor names. This scope is applied to
 318         tensor instances loaded into the passed session, but it is *not* written
 319         through to the static `MetaGraphDef` protocol buffer that is returned.
 320     """
 321     meta_graph_def = self.get_meta_graph_def_from_tags(tags)
 322     with sess.graph.as_default():
 323       # Get asset tensors, if any.
 324       asset_tensors_dictionary = _get_asset_tensors(
 325           self._export_dir, meta_graph_def, import_scope=import_scope)
 326 
 327       main_op_tensor = (
 328           _get_main_op_tensor(meta_graph_def, constants.MAIN_OP_KEY) or
 329           _get_main_op_tensor(meta_graph_def, constants.LEGACY_INIT_OP_KEY))
 330       if main_op_tensor is not None:
 331         sess.run(fetches=[main_op_tensor], feed_dict=asset_tensors_dictionary)
 332 
 333   def load(self, sess, tags, import_scope=None, **saver_kwargs):
 334     """Load the MetaGraphDef graph and restore variable values into the session.
 335 
 336     Args:
 337       sess: tf.Session to restore variable values.
 338       tags: a set of string tags identifying a MetaGraphDef.
 339       import_scope: Optional `string` -- if specified, prepend this string
 340         followed by '/' to all loaded tensor names. This scope is applied to
 341         tensor instances loaded into the passed session, but it is *not* written
 342         through to the static `MetaGraphDef` protocol buffer that is returned.
 343       **saver_kwargs: keyword arguments to pass to tf.train.import_meta_graph.
 344 
 345     Returns:
 346       `MetagraphDef` proto of the graph that was loaded.
 347     """
 348     with sess.graph.as_default():
 349       saver, _ = self.load_graph(sess.graph, tags, import_scope,
 350                                  **saver_kwargs)
 351       self.restore_variables(sess, saver, import_scope)
 352       self.run_init_ops(sess, tags, import_scope)
 353     return self.get_meta_graph_def_from_tags(tags)
