  37 @tf_export("saved_model.build_tensor_info",
  38            "saved_model.utils.build_tensor_info")
  39 @deprecation.deprecated_endpoints("saved_model.utils.build_tensor_info")
  40 def build_tensor_info(tensor):
  41   """Utility function to build TensorInfo proto.
  42 
  43   Args:
  44     tensor: Tensor or SparseTensor whose name, dtype and shape are used to
  45         build the TensorInfo. For SparseTensors, the names of the three
  46         constitutent Tensors are used.
  47 
  48   Returns:
  49     A TensorInfo protocol buffer constructed based on the supplied argument.
  50   """
  51   tensor_info = meta_graph_pb2.TensorInfo(
  52       dtype=dtypes.as_dtype(tensor.dtype).as_datatype_enum,
  53       tensor_shape=tensor.get_shape().as_proto())
  54   if isinstance(tensor, sparse_tensor.SparseTensor):
  55     tensor_info.coo_sparse.values_tensor_name = tensor.values.name
  56     tensor_info.coo_sparse.indices_tensor_name = tensor.indices.name
  57     tensor_info.coo_sparse.dense_shape_tensor_name = tensor.dense_shape.name
  58   else:
  59     tensor_info.name = tensor.name
  60   return tensor_info

  63 @tf_export("saved_model.get_tensor_from_tensor_info",
  64            "saved_model.utils.get_tensor_from_tensor_info")
  65 @deprecation.deprecated_endpoints(
  66     "saved_model.utils.get_tensor_from_tensor_info")
  67 def get_tensor_from_tensor_info(tensor_info, graph=None, import_scope=None):
  68   """Returns the Tensor or SparseTensor described by a TensorInfo proto.
  69 
  70   Args:
  71     tensor_info: A TensorInfo proto describing a Tensor or SparseTensor.
  72     graph: The tf.Graph in which tensors are looked up. If None, the
  73         current default graph is used.
  74     import_scope: If not None, names in `tensor_info` are prefixed with this
  75         string before lookup.
  76 
  77   Returns:
  78     The Tensor or SparseTensor in `graph` described by `tensor_info`.
  79 
  80   Raises:
  81     KeyError: If `tensor_info` does not correspond to a tensor in `graph`.
  82     ValueError: If `tensor_info` is malformed.
  83   """
  84   graph = graph or ops.get_default_graph()
  85   def _get_tensor(name):
  86     return graph.get_tensor_by_name(
  87         ops.prepend_name_scope(name, import_scope=import_scope))
  88   encoding = tensor_info.WhichOneof("encoding")
  89   if encoding == "name":
  90     return _get_tensor(tensor_info.name)
  91   elif encoding == "coo_sparse":
  92     return sparse_tensor.SparseTensor(
  93         _get_tensor(tensor_info.coo_sparse.indices_tensor_name),
  94         _get_tensor(tensor_info.coo_sparse.values_tensor_name),
  95         _get_tensor(tensor_info.coo_sparse.dense_shape_tensor_name))
  96   else:
  97     raise ValueError("Invalid TensorInfo.encoding: %s" % encoding)

 103 def get_or_create_variables_dir(export_dir):
 104   """Return variables sub-directory, or create one if it doesn't exist."""
 105   variables_dir = get_variables_dir(export_dir)
 106   if not file_io.file_exists(variables_dir):
 107     file_io.recursive_create_dir(variables_dir)
 108   return variables_dir
 109 
 110 
 111 def get_variables_dir(export_dir):
 112   """Return variables sub-directory in the SavedModel."""
 113   return os.path.join(
 114       compat.as_text(export_dir),
 115       compat.as_text(constants.VARIABLES_DIRECTORY))
 116 
 117 
 118 def get_variables_path(export_dir):
 119   """Return the variables path, used as the prefix for checkpoint files."""
 120   return os.path.join(
 121       compat.as_text(get_variables_dir(export_dir)),
 122       compat.as_text(constants.VARIABLES_FILENAME))
 123 
 124 
 125 def get_or_create_assets_dir(export_dir):
 126   """Return assets sub-directory, or create one if it doesn't exist."""
 127   assets_destination_dir = get_assets_dir(export_dir)
 128 
 129   if not file_io.file_exists(assets_destination_dir):
 130     file_io.recursive_create_dir(assets_destination_dir)
 131 
 132   return assets_destination_dir
 133 
 134 
 135 def get_assets_dir(export_dir):
 136   """Return path to asset directory in the SavedModel."""
 137   return os.path.join(
 138       compat.as_text(export_dir),
 139       compat.as_text(constants.ASSETS_DIRECTORY))
