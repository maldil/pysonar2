  43 @tf_export("saved_model.Builder",
  44            "saved_model.builder.SavedModelBuilder")
  45 @deprecated_endpoints("saved_model.builder.SavedModelBuilder")
  46 class SavedModelBuilder(object):
  47   """Builds the `SavedModel` protocol buffer and saves variables and assets.
  48 
  49   The `SavedModelBuilder` class provides functionality to build a `SavedModel`
  50   protocol buffer. Specifically, this allows multiple meta graphs to be saved as
  51   part of a single language-neutral `SavedModel`, while sharing variables and
  52   assets.
  53 
  54   To build a SavedModel, the first meta graph must be saved with variables.
  55   Subsequent meta graphs will simply be saved with their graph definitions. If
  56   assets need to be saved and written or copied to disk, they can be provided
  57   when the meta graph def is added. If multiple meta graph defs are associated
  58   an asset of the same name, only the first version is retained.
  59 
  60   Each meta graph added to the SavedModel must be annotated with tags. The tags
  61   provide a means to identify the specific meta graph to load and restore, along
  62   with the shared set of variables and assets.
  63 
  64   Typical usage for the `SavedModelBuilder`:
  65   ```python
  66   ...
  67   builder = tf.saved_model.Builder(export_dir)
  68 
  69   with tf.Session(graph=tf.Graph()) as sess:
  70     ...
  71     builder.add_meta_graph_and_variables(sess,
  72                                     ["foo-tag"],
  73                                     signature_def_map=foo_signatures,
  74                                     assets_collection=foo_assets)
  75   ...
  76 
  77   with tf.Session(graph=tf.Graph()) as sess:
  78     ...
  79     builder.add_meta_graph(["bar-tag", "baz-tag"])
  80   ...
  81 
  82   builder.save()
  83   ```
  84   """
  85 
  86   def __init__(self, export_dir):
  87     self._saved_model = saved_model_pb2.SavedModel()
  88     self._saved_model.saved_model_schema_version = (
  89         constants.SAVED_MODEL_SCHEMA_VERSION)
  90 
  91     self._export_dir = export_dir
  92     if file_io.file_exists(export_dir):
  93       raise AssertionError(
  94           "Export directory already exists. Please specify a different export "
  95           "directory: %s" % export_dir)
  96 
  97     file_io.recursive_create_dir(self._export_dir)
  98 
  99     # Boolean to track whether variables and assets corresponding to the
 100     # SavedModel have been saved. Specifically, the first meta graph to be added
 101     # MUST use the add_meta_graph_and_variables() API. Subsequent add operations
 102     # on the SavedModel MUST use the add_meta_graph() API which does not save
 103     # weights.
 104     self._has_saved_variables = False

 106   def _save_and_write_assets(self, assets_collection_to_add=None):
 107     """Saves asset to the meta graph and writes asset files to disk.
 108 
 109     Args:
 110       assets_collection_to_add: The collection where the asset paths are setup.
 111     """
 112     asset_filename_map = _maybe_save_assets(assets_collection_to_add)
 113 
 114     # Return if there are no assets to write.
 115     if not asset_filename_map:
 116       tf_logging.info("No assets to write.")
 117       return
 118 
 119     assets_destination_dir = saved_model_utils.get_or_create_assets_dir(
 120         self._export_dir)
 121 
 122     # Copy each asset from source path to destination path.
 123     for asset_basename, asset_source_filepath in asset_filename_map.items():
 124       asset_destination_filepath = os.path.join(
 125           compat.as_bytes(assets_destination_dir),
 126           compat.as_bytes(asset_basename))
 127 
 128       # Only copy the asset file to the destination if it does not already
 129       # exist. This is to ensure that an asset with the same name defined as
 130       # part of multiple graphs is only copied the first time.
 131       if not file_io.file_exists(asset_destination_filepath):
 132         file_io.copy(asset_source_filepath, asset_destination_filepath)
 133 
 134     tf_logging.info("Assets written to: %s",
 135                     compat.as_text(assets_destination_dir))

 137   def _maybe_add_main_op(self, main_op):
 138     """Adds main op to the SavedModel.
 139 
 140     Args:
 141       main_op: Main op to run as part of graph initialization. If None, no
 142         main op will be added to the graph.
 143 
 144     Raises:
 145       TypeError: if main op is provided but is not of type `Operation`.
 146       ValueError: if the Graph already contains an init op.
 147     """
 148     if main_op is None:
 149       return
 150 
 151     if not isinstance(main_op, ops.Operation):
 152       raise TypeError("main_op needs to be an Operation: %r" % main_op)
 153 
 154     # Validate that no other init ops have been added to this graph already.
 155     # We check main_op and legacy_init_op for thoroughness and explicitness.
 156     for init_op_key in (constants.MAIN_OP_KEY, constants.LEGACY_INIT_OP_KEY):
 157       if ops.get_collection(init_op_key):
 158         raise ValueError(
 159             "Graph already contains one or more main ops under the "
 160             "collection {}.".format(init_op_key))
 161 
 162     ops.add_to_collection(constants.MAIN_OP_KEY, main_op)
 163 
 164   def _add_train_op(self, train_op):
 165     """Add train op to the SavedModel.
 166 
 167     Note that this functionality is in development, and liable to be
 168     moved elsewhere.
 169 
 170     Args:
 171       train_op: Op or group of ops that are used for training. These are
 172         stored as a collection with key TRAIN_OP_KEY, but not executed.
 173 
 174     Raises:
 175       TypeError if Train op is not of type `Operation`.
 176     """
 177     if train_op is not None:
 178       if (not isinstance(train_op, ops.Tensor) and
 179           not isinstance(train_op, ops.Operation)):
 180         raise TypeError("train_op needs to be a Tensor or Op: %r" % train_op)
 181       ops.add_to_collection(constants.TRAIN_OP_KEY, train_op)

 183   def _tag_and_add_meta_graph(self, meta_graph_def, tags, signature_def_map):
 184     """Tags the meta graph def and adds it to the SavedModel.
 185 
 186     Tags the meta graph def with the supplied tags, adds signature defs to it if
 187     provided and appends the meta graph def to the SavedModel proto.
 188 
 189     Args:
 190       meta_graph_def: The meta graph def to add to the SavedModel.
 191       tags: The set of tags to annotate the meta graph def with.
 192       signature_def_map: The map of signature defs to be added to the meta graph
 193           def.
 194     """
 195     for tag in tags:
 196       meta_graph_def.meta_info_def.tags.append(tag)
 197 
 198     if signature_def_map is not None:
 199       for key in signature_def_map:
 200         meta_graph_def.signature_def[key].CopyFrom(signature_def_map[key])
 201 
 202     proto_meta_graph_def = self._saved_model.meta_graphs.add()
 203     proto_meta_graph_def.CopyFrom(meta_graph_def)
 204 
 205   def _validate_tensor_info(self, tensor_info):
 206     """Validates the `TensorInfo` proto.
 207 
 208     Checks if the `encoding` (`name` or `coo_sparse`) and `dtype` fields exist
 209     and are non-empty.
 210 
 211     Args:
 212       tensor_info: `TensorInfo` protocol buffer to validate.
 213 
 214     Raises:
 215       AssertionError: If the `name` or `dtype` fields of the supplied
 216           `TensorInfo` proto are not populated.
 217     """
 218     if tensor_info is None:
 219       raise AssertionError(
 220           "All TensorInfo protos used in the SignatureDefs must have the name "
 221           "and dtype fields set.")
 222     if tensor_info.WhichOneof("encoding") is None:
 223       # TODO(soergel) validate each of the fields of coo_sparse
 224       raise AssertionError(
 225           "All TensorInfo protos used in the SignatureDefs must have one of "
 226           "the 'encoding' fields (e.g., name or coo_sparse) set: %s"
 227           % tensor_info)
 228     if tensor_info.dtype is types_pb2.DT_INVALID:
 229       raise AssertionError(
 230           "All TensorInfo protos used in the SignatureDefs must have the dtype "
 231           "field set: %s" % tensor_info)
 232 
 233   def _validate_signature_def_map(self, signature_def_map):
 234     """Validates the `SignatureDef` entries in the signature def map.
 235 
 236     Validation of entries in the signature def map includes ensuring that the
 237     `name` and `dtype` fields of the TensorInfo protos of the `inputs` and
 238     `outputs` of each `SignatureDef` are populated.
 239 
 240     Args:
 241       signature_def_map: The map of signature defs to be validated.
 242     """
 243     if signature_def_map is not None:
 244       for signature_def_key in signature_def_map:
 245         signature_def = signature_def_map[signature_def_key]
 246         inputs = signature_def.inputs
 247         outputs = signature_def.outputs
 248         for inputs_key in inputs:
 249           self._validate_tensor_info(inputs[inputs_key])
 250         for outputs_key in outputs:
 251           self._validate_tensor_info(outputs[outputs_key])
 252 
 253   def _add_collections(
 254       self, assets_collection, main_op, train_op):
 255     """Add asset and op collections to be saved."""
 256     # Save asset files and write them to disk, if any.
 257     self._save_and_write_assets(assets_collection)
 258 
 259     self._maybe_add_main_op(main_op)
 260 
 261     self._add_train_op(train_op)

 263   def _maybe_create_saver(self, saver=None):
 264     """Creates a sharded saver if one does not already exist."""
 265     if not saver:
 266       # Initialize a saver to generate a sharded output for all saveables in the
 267       # current scope.
 268       saver = tf_saver.Saver(
 269           variables._all_saveable_objects(),  # pylint: disable=protected-access
 270           sharded=True,
 271           write_version=saver_pb2.SaverDef.V2,
 272           allow_empty=True)
 273     return saver

 275   @deprecated_args(None,
 276                    "Pass your op to the equivalent parameter main_op instead.",
 277                    "legacy_init_op")
 278   def add_meta_graph(self,
 279                      tags,
 280                      signature_def_map=None,
 281                      assets_collection=None,
 282                      legacy_init_op=None,
 283                      clear_devices=False,
 284                      main_op=None,
 285                      strip_default_attrs=False,
 286                      saver=None):
 287     # pylint: disable=line-too-long
 288     """Adds the current meta graph to the SavedModel.
 289 
 290     Creates a Saver in the current scope and uses the Saver to export the meta
 291     graph def. Invoking this API requires the `add_meta_graph_and_variables()`
 292     API to have been invoked before.
 293 
 294     Args:
 295       tags: The set of tags to annotate the meta graph def with.
 296       signature_def_map: The map of signature defs to be added to the meta graph
 297           def.
 298       assets_collection: Assets collection to be saved with SavedModel. Note
 299           that this collection should be a subset of the assets saved as part of
 300           the first meta graph in the SavedModel.
 301       legacy_init_op: Legacy support for op or group of ops to execute after the
 302           restore op upon a load. Deprecated; please use main_op instead.
 303       clear_devices: Set to true if the device info on the default graph should
 304           be cleared.
 305       main_op: Op or group of ops to execute when the graph is loaded. Note
 306           that when the main_op is specified it is run after the restore op at
 307           load-time.
 308       strip_default_attrs: Boolean. If `True`, default-valued attributes will be
 309         removed from the NodeDefs. For a detailed guide, see
 310         [Stripping Default-Valued Attributes](https://github.com/tensorflow/tensorflow/blob/master/tensorflow/python/saved_model/README.md#stripping-default-valued-attributes).
 311       saver: An instance of tf.train.Saver that will be used to export the
 312         metagraph. If None, a sharded Saver that restores all variables will
 313         be used.
 314 
 315     Raises:
 316       AssertionError: If the variables for the SavedModel have not been saved
 317           yet, or if the graph already contains one or more legacy init ops.
 318     """
 319     # pylint: enable=line-too-long
 320     if not self._has_saved_variables:
 321       raise AssertionError(
 322           "Graph state including variables and assets has not been saved yet. "
 323           "Please invoke `add_meta_graph_and_variables()` first.")
 324 
 325     # Validate the signature def map to ensure all included TensorInfos are
 326     # properly populated.
 327     self._validate_signature_def_map(signature_def_map)
 328 
 329     # legacy_init_op is deprecated, and going away in TF 2.0.
 330     # Re-mapping to main_op, as treatment is identical regardless.
 331     main_op = main_op or legacy_init_op
 332 
 333     # Add assets and ops
 334     self._add_collections(assets_collection, main_op, None)
 335 
 336     saver = self._maybe_create_saver(saver)
 337 
 338     # The graph almost certainly previously contained at least one Saver, and
 339     # possibly several (e.g. one for loading a pretrained embedding, and another
 340     # for the model weights).  Removing the preexisting ones was the
 341     # motivation for the clear_extraneous_savers option, but it turns out that
 342     # there are edge cases where that option breaks the graph.  Until that is
 343     # resolved, we just leave the option set to False for now.
 344     # TODO(soergel): Reinstate clear_extraneous_savers=True when possible.
 345     meta_graph_def = saver.export_meta_graph(
 346         clear_devices=clear_devices, strip_default_attrs=strip_default_attrs)
 347 
 348     # Tag the meta graph def and add it to the SavedModel.
 349     self._tag_and_add_meta_graph(meta_graph_def, tags, signature_def_map)
 350 
 351   @deprecated_args(None,
 352                    "Pass your op to the equivalent parameter main_op instead.",
 353                    "legacy_init_op")
 354   def add_meta_graph_and_variables(self,
 355                                    sess,
 356                                    tags,
 357                                    signature_def_map=None,
 358                                    assets_collection=None,
 359                                    legacy_init_op=None,
 360                                    clear_devices=False,
 361                                    main_op=None,
 362                                    strip_default_attrs=False,
 363                                    saver=None):
 364     # pylint: disable=line-too-long
 365     """Adds the current meta graph to the SavedModel and saves variables.
 366 
 367     Creates a Saver to save the variables from the provided session. Exports the
 368     corresponding meta graph def. This function assumes that the variables to be
 369     saved have been initialized. For a given `SavedModelBuilder`, this API must
 370     be called exactly once and for the first meta graph to save. For subsequent
 371     meta graph defs to be added, the `add_meta_graph()` API must be used.
 372 
 373     Args:
 374       sess: The TensorFlow session from which to save the meta graph and
 375         variables.
 376       tags: The set of tags with which to save the meta graph.
 377       signature_def_map: The map of signature def map to add to the meta graph
 378         def.
 379       assets_collection: Assets collection to be saved with SavedModel.
 380       legacy_init_op: Legacy support for op or group of ops to execute after the
 381           restore op upon a load. Deprecated; please use main_op instead.
 382       clear_devices: Set to true if the device info on the default graph should
 383           be cleared.
 384       main_op: Op or group of ops to execute when the graph is loaded. Note
 385           that when the main_op is specified it is run after the restore op at
 386           load-time.
 387       strip_default_attrs: Boolean. If `True`, default-valued attributes will be
 388         removed from the NodeDefs. For a detailed guide, see
 389         [Stripping Default-Valued Attributes](https://github.com/tensorflow/tensorflow/blob/master/tensorflow/python/saved_model/README.md#stripping-default-valued-attributes).
 390       saver: An instance of tf.train.Saver that will be used to export the
 391         metagraph and save variables. If None, a sharded Saver that restores
 392         all variables will be used.
 393 
 394     """
 395     # pylint: enable=line-too-long
 396     if self._has_saved_variables:
 397       raise AssertionError("Graph state including variables and assets has "
 398                            "already been saved. Please invoke "
 399                            "`add_meta_graph()` instead.")
 400 
 401     # Validate the signature def map to ensure all included TensorInfos are
 402     # properly populated.
 403     self._validate_signature_def_map(signature_def_map)
 404 
 405     # legacy_init_op is deprecated, and going away in TF 2.0.
 406     # Re-mapping to main_op, as treatment is identical regardless.
 407     main_op = main_op or legacy_init_op
 408 
 409     # Add assets and ops
 410     self._add_collections(assets_collection, main_op, None)
 411 
 412     saved_model_utils.get_or_create_variables_dir(self._export_dir)
 413     variables_path = saved_model_utils.get_variables_path(self._export_dir)
 414 
 415     saver = self._maybe_create_saver(saver)
 416 
 417     # Save the variables. Also, disable writing the checkpoint state proto. The
 418     # file is not used during SavedModel loading. In addition, since a
 419     # SavedModel can be copied or moved, this avoids the checkpoint state to
 420     # become outdated.
 421     saver.save(sess, variables_path, write_meta_graph=False, write_state=False)
 422 
 423     # Export the meta graph def.
 424 
 425     # The graph almost certainly previously contained at least one Saver, and
 426     # possibly several (e.g. one for loading a pretrained embedding, and another
 427     # for the model weights).  Removing the preexisting ones was the
 428     # motivation for the clear_extraneous_savers option, but it turns out that
 429     # there are edge cases where that option breaks the graph.  Until that is
 430     # resolved, we just leave the option set to False for now.
 431     # TODO(soergel): Reinstate clear_extraneous_savers=True when possible.
 432     meta_graph_def = saver.export_meta_graph(
 433         clear_devices=clear_devices, strip_default_attrs=strip_default_attrs)
 434 
 435     # Tag the meta graph def and add it to the SavedModel.
 436     self._tag_and_add_meta_graph(meta_graph_def, tags, signature_def_map)
 437 
 438     # Mark this instance of SavedModel as having saved variables, such that
 439     # subsequent attempts to save variables will fail.
 440     self._has_saved_variables = True
 441 
 442   def save(self, as_text=False):
 443     """Writes a `SavedModel` protocol buffer to disk.
 444 
 445     The function writes the SavedModel protocol buffer to the export directory
 446     in serialized format.
 447 
 448     Args:
 449       as_text: Writes the SavedModel protocol buffer in text format to disk.
 450 
 451     Returns:
 452       The path to which the SavedModel protocol buffer was written.
 453     """
 454     if not file_io.file_exists(self._export_dir):
 455       file_io.recursive_create_dir(self._export_dir)
 456 
 457     if as_text:
 458       path = os.path.join(
 459           compat.as_bytes(self._export_dir),
 460           compat.as_bytes(constants.SAVED_MODEL_FILENAME_PBTXT))
 461       file_io.write_string_to_file(path, str(self._saved_model))
 462     else:
 463       path = os.path.join(
 464           compat.as_bytes(self._export_dir),
 465           compat.as_bytes(constants.SAVED_MODEL_FILENAME_PB))
 466       file_io.write_string_to_file(path, self._saved_model.SerializeToString())
 467     tf_logging.info("SavedModel written to: %s", compat.as_text(path))
 468 
 469     return path

 472 def _maybe_save_assets(assets_collection_to_add=None):
 473   """Saves assets to the meta graph.
 474 
 475   Args:
 476     assets_collection_to_add: The collection where the asset paths are setup.
 477 
 478   Returns:
 479     A dict of asset basenames for saving to the original full path to the asset.
 480 
 481   Raises:
 482     ValueError: Indicating an invalid filepath tensor.
 483   """
 484   # Map of target file names to original filenames
 485   asset_filename_map = {}
 486 
 487   if assets_collection_to_add is None:
 488     tf_logging.info("No assets to save.")
 489     return asset_filename_map
 490 
 491   # Iterate over the supplied asset collection, build the `AssetFile` proto
 492   # and add them to the collection with key `constants.ASSETS_KEY`, in the
 493   # graph.
 494   for asset_tensor in assets_collection_to_add:
 495     asset_source_filepath = _asset_path_from_tensor(asset_tensor)
 496     if not asset_source_filepath:
 497       raise ValueError("Invalid asset filepath tensor %s" % asset_tensor)
 498 
 499     asset_filename = _get_asset_filename_to_add(
 500         asset_source_filepath, asset_filename_map)
 501 
 502     # Build `AssetFile` proto and add it to the asset collection in the graph.
 503     # Note that this should be done even when the file is a duplicate of an
 504     # already-added file, as the tensor reference should still exist.
 505     _add_asset_to_collection(asset_filename, asset_tensor)
 506 
 507     # In the cases where we are adding a duplicate, this will result in the
 508     # last of the filepaths being the one used for copying the file to the
 509     # SavedModel. Since the files in question are the same, it doesn't matter
 510     # either way.
 511     asset_filename_map[asset_filename] = asset_source_filepath
 512 
 513   tf_logging.info("Assets added to graph.")
 514   return asset_filename_map

 517 def _get_asset_filename_to_add(asset_filepath, asset_filename_map):
 518   """Get a unique basename to add to the SavedModel if this file is unseen.
 519 
 520   Assets come from users as full paths, and we save them out to the
 521   SavedModel as basenames. In some cases, the basenames collide. Here,
 522   we dedupe asset basenames by first checking if the file is the same,
 523   and, if different, generate and return an index-suffixed basename
 524   that can be used to add the asset to the SavedModel.
 525 
 526   Args:
 527     asset_filepath: the full path to the asset that is being saved
 528     asset_filename_map: a dict of filenames used for saving the asset in
 529       the SavedModel to full paths from which the filenames were derived.
 530 
 531   Returns:
 532     Uniquified filename string if the file is not a duplicate, or the original
 533     filename if the file has already been seen and saved.
 534   """
 535   asset_filename = os.path.basename(asset_filepath)
 536 
 537   if asset_filename not in asset_filename_map:
 538     # This is an unseen asset. Safe to add.
 539     return asset_filename
 540 
 541   other_asset_filepath = asset_filename_map[asset_filename]
 542   if other_asset_filepath == asset_filepath:
 543     # This is the same file, stored twice in the collection list. No need
 544     # to make unique.
 545     return asset_filename
 546 
 547   # Else, asset_filename is in the map, and the filepath is different. Dedupe.
 548   if not file_io.filecmp(asset_filepath, other_asset_filepath):
 549     # Files are different; dedupe filenames.
 550     return _get_unique_asset_filename(asset_filename, asset_filename_map)
 551 
 552   # Files are the same; don't make unique.
 553   return asset_filename
 554

 556 def _get_unique_asset_filename(asset_filename, asset_filename_map):
 557   i = 1
 558   unique_filename = asset_filename
 559   while unique_filename in asset_filename_map:
 560     unique_filename = compat.as_bytes("_").join(
 561         [compat.as_bytes(asset_filename), compat.as_bytes(str(i))])
 562     i += 1
 563   return unique_filename

 566 def _asset_path_from_tensor(path_tensor):
 567   """Returns the filepath value stored in constant `path_tensor`.
 568 
 569   Args:
 570     path_tensor: Tensor of a file-path.
 571 
 572   Returns:
 573     The string value i.e. path of the tensor, if valid.
 574 
 575   Raises:
 576     TypeError if tensor does not match expected op type, dtype or value.
 577   """
 578   if not isinstance(path_tensor, ops.Tensor):
 579     raise TypeError("Asset path tensor must be a Tensor.")
 580   if path_tensor.op.type != "Const":
 581     raise TypeError("Asset path tensor must be of type constant.")
 582   if path_tensor.dtype != dtypes.string:
 583     raise TypeError("Asset path tensor must be of dtype string.")
 584   str_values = path_tensor.op.get_attr("value").string_val
 585   if len(str_values) != 1:
 586     raise TypeError("Asset path tensor must be a scalar.")
 587   return str_values[0]
