  31 @tf_export('saved_model.build_signature_def',
  32            'saved_model.signature_def_utils.build_signature_def')
  33 @deprecation.deprecated_endpoints(
  34     'saved_model.signature_def_utils.build_signature_def')
  35 def build_signature_def(inputs=None, outputs=None, method_name=None):
  36   """Utility function to build a SignatureDef protocol buffer.
  37 
  38   Args:
  39     inputs: Inputs of the SignatureDef defined as a proto map of string to
  40         tensor info.
  41     outputs: Outputs of the SignatureDef defined as a proto map of string to
  42         tensor info.
  43     method_name: Method name of the SignatureDef as a string.
  44 
  45   Returns:
  46     A SignatureDef protocol buffer constructed based on the supplied arguments.
  47   """
  48   signature_def = meta_graph_pb2.SignatureDef()
  49   if inputs is not None:
  50     for item in inputs:
  51       signature_def.inputs[item].CopyFrom(inputs[item])
  52   if outputs is not None:
  53     for item in outputs:
  54       signature_def.outputs[item].CopyFrom(outputs[item])
  55   if method_name is not None:
  56     signature_def.method_name = method_name
  57   return signature_def

  60 @tf_export('saved_model.regression_signature_def',
  61            'saved_model.signature_def_utils.regression_signature_def')
  62 @deprecation.deprecated_endpoints(
  63     'saved_model.signature_def_utils.regression_signature_def')
  64 def regression_signature_def(examples, predictions):
  65   """Creates regression signature from given examples and predictions.
  66 
  67   This function produces signatures intended for use with the TensorFlow Serving
  68   Regress API (tensorflow_serving/apis/prediction_service.proto), and so
  69   constrains the input and output types to those allowed by TensorFlow Serving.
  70 
  71   Args:
  72     examples: A string `Tensor`, expected to accept serialized tf.Examples.
  73     predictions: A float `Tensor`.
  74 
  75   Returns:
  76     A regression-flavored signature_def.
  77 
  78   Raises:
  79     ValueError: If examples is `None`.
  80   """
  81   if examples is None:
  82     raise ValueError('Regression examples cannot be None.')
  83   if not isinstance(examples, ops.Tensor):
  84     raise ValueError('Regression examples must be a string Tensor.')
  85   if predictions is None:
  86     raise ValueError('Regression predictions cannot be None.')
  87 
  88   input_tensor_info = utils.build_tensor_info(examples)
  89   if input_tensor_info.dtype != types_pb2.DT_STRING:
  90     raise ValueError('Regression examples must be a string Tensor.')
  91   signature_inputs = {signature_constants.REGRESS_INPUTS: input_tensor_info}
  92 
  93   output_tensor_info = utils.build_tensor_info(predictions)
  94   if output_tensor_info.dtype != types_pb2.DT_FLOAT:
  95     raise ValueError('Regression output must be a float Tensor.')
  96   signature_outputs = {signature_constants.REGRESS_OUTPUTS: output_tensor_info}
  97 
  98   signature_def = build_signature_def(
  99       signature_inputs, signature_outputs,
 100       signature_constants.REGRESS_METHOD_NAME)
 101 
 102   return signature_def

 105 @tf_export('saved_model.classification_signature_def',
 106            'saved_model.signature_def_utils.classification_signature_def')
 107 @deprecation.deprecated_endpoints(
 108     'saved_model.signature_def_utils.classification_signature_def')
 109 def classification_signature_def(examples, classes, scores):
 110   """Creates classification signature from given examples and predictions.
 111 
 112   This function produces signatures intended for use with the TensorFlow Serving
 113   Classify API (tensorflow_serving/apis/prediction_service.proto), and so
 114   constrains the input and output types to those allowed by TensorFlow Serving.
 115 
 116   Args:
 117     examples: A string `Tensor`, expected to accept serialized tf.Examples.
 118     classes: A string `Tensor`.  Note that the ClassificationResponse message
 119       requires that class labels are strings, not integers or anything else.
 120     scores: a float `Tensor`.
 121 
 122   Returns:
 123     A classification-flavored signature_def.
 124 
 125   Raises:
 126     ValueError: If examples is `None`.
 127   """
 128   if examples is None:
 129     raise ValueError('Classification examples cannot be None.')
 130   if not isinstance(examples, ops.Tensor):
 131     raise ValueError('Classification examples must be a string Tensor.')
 132   if classes is None and scores is None:
 133     raise ValueError('Classification classes and scores cannot both be None.')
 134 
 135   input_tensor_info = utils.build_tensor_info(examples)
 136   if input_tensor_info.dtype != types_pb2.DT_STRING:
 137     raise ValueError('Classification examples must be a string Tensor.')
 138   signature_inputs = {signature_constants.CLASSIFY_INPUTS: input_tensor_info}
 139 
 140   signature_outputs = {}
 141   if classes is not None:
 142     classes_tensor_info = utils.build_tensor_info(classes)
 143     if classes_tensor_info.dtype != types_pb2.DT_STRING:
 144       raise ValueError('Classification classes must be a string Tensor.')
 145     signature_outputs[signature_constants.CLASSIFY_OUTPUT_CLASSES] = (
 146         classes_tensor_info)
 147   if scores is not None:
 148     scores_tensor_info = utils.build_tensor_info(scores)
 149     if scores_tensor_info.dtype != types_pb2.DT_FLOAT:
 150       raise ValueError('Classification scores must be a float Tensor.')
 151     signature_outputs[signature_constants.CLASSIFY_OUTPUT_SCORES] = (
 152         scores_tensor_info)
 153 
 154   signature_def = build_signature_def(
 155       signature_inputs, signature_outputs,
 156       signature_constants.CLASSIFY_METHOD_NAME)
 157 
 158   return signature_def

 161 @tf_export('saved_model.predict_signature_def',
 162            'saved_model.signature_def_utils.predict_signature_def')
 163 @deprecation.deprecated_endpoints(
 164     'saved_model.signature_def_utils.predict_signature_def')
 165 def predict_signature_def(inputs, outputs):
 166   """Creates prediction signature from given inputs and outputs.
 167 
 168   This function produces signatures intended for use with the TensorFlow Serving
 169   Predict API (tensorflow_serving/apis/prediction_service.proto). This API
 170   imposes no constraints on the input and output types.
 171 
 172   Args:
 173     inputs: dict of string to `Tensor`.
 174     outputs: dict of string to `Tensor`.
 175 
 176   Returns:
 177     A prediction-flavored signature_def.
 178 
 179   Raises:
 180     ValueError: If inputs or outputs is `None`.
 181   """
 182   if inputs is None or not inputs:
 183     raise ValueError('Prediction inputs cannot be None or empty.')
 184   if outputs is None or not outputs:
 185     raise ValueError('Prediction outputs cannot be None or empty.')
 186 
 187   signature_inputs = {key: utils.build_tensor_info(tensor)
 188                       for key, tensor in inputs.items()}
 189   signature_outputs = {key: utils.build_tensor_info(tensor)
 190                        for key, tensor in outputs.items()}
 191 
 192   signature_def = build_signature_def(
 193       signature_inputs, signature_outputs,
 194       signature_constants.PREDICT_METHOD_NAME)
 195 
 196   return signature_def

 213 def _supervised_signature_def(
 214     method_name, inputs, loss=None, predictions=None,
 215     metrics=None):
 216   """Creates a signature for training and eval data.
 217 
 218   This function produces signatures that describe the inputs and outputs
 219   of a supervised process, such as training or evaluation, that
 220   results in loss, metrics, and the like. Note that this function only requires
 221   inputs to be not None.
 222 
 223   Args:
 224     method_name: Method name of the SignatureDef as a string.
 225     inputs: dict of string to `Tensor`.
 226     loss: dict of string to `Tensor` representing computed loss.
 227     predictions: dict of string to `Tensor` representing the output predictions.
 228     metrics: dict of string to `Tensor` representing metric ops.
 229 
 230   Returns:
 231     A train- or eval-flavored signature_def.
 232 
 233   Raises:
 234     ValueError: If inputs or outputs is `None`.
 235   """
 236   if inputs is None or not inputs:
 237     raise ValueError('{} inputs cannot be None or empty.'.format(method_name))
 238 
 239   signature_inputs = {key: utils.build_tensor_info(tensor)
 240                       for key, tensor in inputs.items()}
 241 
 242   signature_outputs = {}
 243   for output_set in (loss, predictions, metrics):
 244     if output_set is not None:
 245       sig_out = {key: utils.build_tensor_info(tensor)
 246                  for key, tensor in output_set.items()}
 247       signature_outputs.update(sig_out)
 248 
 249   signature_def = build_signature_def(
 250       signature_inputs, signature_outputs, method_name)
 251 
 252   return signature_def

 255 @tf_export('saved_model.is_valid_signature',
 256            'saved_model.signature_def_utils.is_valid_signature')
 257 @deprecation.deprecated_endpoints(
 258     'saved_model.signature_def_utils.is_valid_signature')
 259 def is_valid_signature(signature_def):
 260   """Determine whether a SignatureDef can be served by TensorFlow Serving."""
 261   if signature_def is None:
 262     return False
 263   return (_is_valid_classification_signature(signature_def) or
 264           _is_valid_regression_signature(signature_def) or
 265           _is_valid_predict_signature(signature_def))
 266

 268 def _is_valid_predict_signature(signature_def):
 269   """Determine whether the argument is a servable 'predict' SignatureDef."""
 270   if signature_def.method_name != signature_constants.PREDICT_METHOD_NAME:
 271     return False
 272   if not signature_def.inputs.keys():
 273     return False
 274   if not signature_def.outputs.keys():
 275     return False
 276   return True
 277 
 278 
 279 def _is_valid_regression_signature(signature_def):
 280   """Determine whether the argument is a servable 'regress' SignatureDef."""
 281   if signature_def.method_name != signature_constants.REGRESS_METHOD_NAME:
 282     return False
 283 
 284   if (set(signature_def.inputs.keys())
 285       != set([signature_constants.REGRESS_INPUTS])):
 286     return False
 287   if (signature_def.inputs[signature_constants.REGRESS_INPUTS].dtype !=
 288       types_pb2.DT_STRING):
 289     return False
 290 
 291   if (set(signature_def.outputs.keys())
 292       != set([signature_constants.REGRESS_OUTPUTS])):
 293     return False
 294   if (signature_def.outputs[signature_constants.REGRESS_OUTPUTS].dtype !=
 295       types_pb2.DT_FLOAT):
 296     return False
 297 
 298   return True
 299

 301 def _is_valid_classification_signature(signature_def):
 302   """Determine whether the argument is a servable 'classify' SignatureDef."""
 303   if signature_def.method_name != signature_constants.CLASSIFY_METHOD_NAME:
 304     return False
 305 
 306   if (set(signature_def.inputs.keys())
 307       != set([signature_constants.CLASSIFY_INPUTS])):
 308     return False
 309   if (signature_def.inputs[signature_constants.CLASSIFY_INPUTS].dtype !=
 310       types_pb2.DT_STRING):
 311     return False
 312 
 313   allowed_outputs = set([signature_constants.CLASSIFY_OUTPUT_CLASSES,
 314                          signature_constants.CLASSIFY_OUTPUT_SCORES])
 315 
 316   if not signature_def.outputs.keys():
 317     return False
 318   if set(signature_def.outputs.keys()) - allowed_outputs:
 319     return False
 320   if (signature_constants.CLASSIFY_OUTPUT_CLASSES in signature_def.outputs
 321       and
 322       signature_def.outputs[signature_constants.CLASSIFY_OUTPUT_CLASSES].dtype
 323       != types_pb2.DT_STRING):
 324     return False
 325   if (signature_constants.CLASSIFY_OUTPUT_SCORES in signature_def.outputs
 326       and
 327       signature_def.outputs[signature_constants.CLASSIFY_OUTPUT_SCORES].dtype !=
 328       types_pb2.DT_FLOAT):
 329     return False
 330 
 331   return True
