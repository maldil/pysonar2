  56 class _PartitionInfo(object):
  57   """Holds partition info used by initializer functions.
  58   """
  59 
  60   def __init__(self, full_shape, var_offset):
  61     """Constructor.
  62 
  63     Args:
  64       full_shape: Tuple or list of `int` indicating the full combined shape
  65         of the partitioned variables.
  66       var_offset: Tuple or list of `int` specifying offset of this partition
  67         with respect to the full variable for each dimension.
  68 
  69     Raises:
  70       TypeError: If `full_shape` or `var_offset` is not a sequence.
  71       ValueError: If `full_shape` or `var_offset` differ in length. If
  72         `var_offset` exceeds `full_shape` in any dimension.
  73     """
  74     if not isinstance(full_shape, collections_lib.Sequence) or isinstance(
  75         full_shape, six.string_types):
  76       raise TypeError(
  77           "`full_shape` must be a sequence (like tuple or list) instead of " +
  78           type(full_shape).__name__)
  79 
  80     if not isinstance(var_offset, collections_lib.Sequence) or isinstance(
  81         var_offset, six.string_types):
  82       raise TypeError(
  83           "`var_offset` must be a sequence (like tuple or list) instead of " +
  84           type(var_offset).__name__)
  85 
  86     if len(var_offset) != len(full_shape):
  87       raise ValueError(
  88           "Expected equal length, but `var_offset` is of length {} while "
  89           "full_shape is of length {}.".format(
  90               len(var_offset), len(full_shape)))
  91 
  92     for i in xrange(len(full_shape)):
  93       offset = var_offset[i]
  94       shape = full_shape[i]
  95       if offset < 0 or offset >= shape:
  96         raise ValueError(
  97             "Expected 0 <= offset < shape but found offset={}, shape={} for "
  98             "var_offset={}, full_shape={}".format(offset, shape, var_offset,
  99                                                   full_shape))
 100 
 101     self._full_shape = full_shape
 102     self._var_offset = var_offset
 103 
 104   @property
 105   def full_shape(self):
 106     return self._full_shape
 107 
 108   @property
 109   def var_offset(self):
 110     return self._var_offset
 111 
 112   def single_offset(self, shape):
 113     """Returns the offset when the variable is partitioned in at most one dim.
 114 
 115     Args:
 116       shape: Tuple or list of `int` indicating the shape of one specific
 117         variable partition.
 118 
 119     Returns:
 120       `int` representing the offset in the dimension along which the variable is
 121        partitioned. Returns 0 if the variable is not being partitioned.
 122 
 123     Raises:
 124       ValueError: Depending on self.single_slice_dim().
 125     """
 126 
 127     single_slice_dim = self.single_slice_dim(shape)
 128     # If this variable is not being partitioned at all, single_slice_dim() could
 129     # return None.
 130     if single_slice_dim is None:
 131       return 0
 132     return self.var_offset[single_slice_dim]
 133 
 134   def single_slice_dim(self, shape):
 135     """Returns the slice dim when the variable is partitioned only in one dim.
 136 
 137     Args:
 138       shape: Tuple or list of `int` indicating the shape of one specific
 139         variable partition.
 140 
 141     Returns:
 142       `int` representing the dimension that the variable is partitioned in, or
 143       `None` if the variable doesn't seem to be partitioned at all.
 144 
 145     Raises:
 146       TypeError: If `shape` is not a sequence.
 147       ValueError: If `shape` is not the same length as `self.full_shape`. If
 148         the variable is partitioned in more than one dimension.
 149     """
 150     if not isinstance(shape, collections_lib.Sequence) or isinstance(
 151         shape, six.string_types):
 152       raise TypeError(
 153           "`shape` must be a sequence (like tuple or list) instead of " +
 154           type(shape).__name__)
 155 
 156     if len(shape) != len(self.full_shape):
 157       raise ValueError(
 158           "Expected equal length, but received shape={} of length {} while "
 159           "self.full_shape={} is of length {}.".format(shape, len(
 160               shape), self.full_shape, len(self.full_shape)))
 161 
 162     for i in xrange(len(shape)):
 163       if self.var_offset[i] + shape[i] > self.full_shape[i]:
 164         raise ValueError(
 165             "With self.var_offset={}, a partition of shape={} would exceed "
 166             "self.full_shape={} in dimension {}.".format(
 167                 self.var_offset, shape, self.full_shape, i))
 168 
 169     slice_dim = None
 170     for i in xrange(len(shape)):
 171       if shape[i] == self.full_shape[i]:
 172         continue
 173       if slice_dim is not None:
 174         raise ValueError(
 175             "Cannot use single_slice_dim() with shape={} and "
 176             "self.full_shape={} since slice dim could be either dimension {} "
 177             "or {}.".format(shape, self.full_shape, i, slice_dim))
 178       slice_dim = i
 179 
 180     return slice_dim

 212 @tf_export(v1=["enable_resource_variables"])
 213 def enable_resource_variables():
 214   """Creates resource variables by default.
 215 
 216   Resource variables are improved versions of TensorFlow variables with a
 217   well-defined memory model. Accessing a resource variable reads its value, and
 218   all ops which access a specific read value of the variable are guaranteed to
 219   see the same value for that tensor. Writes which happen after a read (by
 220   having a control or data dependency on the read) are guaranteed not to affect
 221   the value of the read tensor, and similarly writes which happen before a read
 222   are guaranteed to affect the value. No guarantees are made about unordered
 223   read/write pairs.
 224 
 225   Calling tf.enable_resource_variables() lets you opt-in to this TensorFlow 2.0
 226   feature.
 227   """
 228   global _DEFAULT_USE_RESOURCE
 229   _DEFAULT_USE_RESOURCE = True

 245 class _VariableStore(object):
 246   """Variable store that carries a number of named Variables.
 247 
 248   New variable names and new variables can be created; all stored
 249   variables are initialized with the initializer passed to __init__.
 250 
 251   Attributes:
 252     vars: a dictionary with string names (same as passed in GetVar) as keys
 253           and the corresponding TensorFlow Variables as values.
 254   """
 255 
 256   def __init__(self):
 257     """Create a variable store."""
 258     self._vars = {}  # A dictionary of the stored TensorFlow variables.
 259     self._partitioned_vars = {}  # A dict of the stored PartitionedVariables.
 260     self._store_eager_variables = False
 261 
 262   def get_variable(self,
 263                    name,
 264                    shape=None,
 265                    dtype=dtypes.float32,
 266                    initializer=None,
 267                    regularizer=None,
 268                    reuse=None,
 269                    trainable=None,
 270                    collections=None,
 271                    caching_device=None,
 272                    partitioner=None,
 273                    validate_shape=True,
 274                    use_resource=None,
 275                    custom_getter=None,
 276                    constraint=None,
 277                    synchronization=VariableSynchronization.AUTO,
 278                    aggregation=VariableAggregation.NONE):
 279     """Gets an existing variable with these parameters or create a new one.
 280 
 281     If a variable with the given name is already stored, we return the stored
 282     variable. Otherwise, we create a new one.
 283 
 284     Set `reuse` to `True` when you only want to reuse existing Variables.
 285     Set `reuse` to `False` when you only want to create new Variables.
 286     Set `reuse` to None (the default) or tf.AUTO_REUSE when you want
 287     variables to be created if they don't exist or returned if they do.
 288 
 289     If initializer is `None` (the default), the default initializer passed in
 290     the constructor is used. If that one is `None` too, we use a new
 291     `glorot_uniform_initializer`. If initializer is a Tensor, we use
 292     it as a value and derive the shape from the initializer.
 293 
 294     If a partitioner is provided, a `PartitionedVariable` is returned.
 295     Accessing this object as a `Tensor` returns the shards concatenated along
 296     the partition axis.
 297 
 298     Some useful partitioners are available.  See, e.g.,
 299     `variable_axis_size_partitioner` and `min_max_variable_partitioner`.
 300 
 301     Args:
 302       name: The name of the new or existing variable.
 303       shape: Shape of the new or existing variable.
 304       dtype: Type of the new or existing variable (defaults to `DT_FLOAT`).
 305       initializer: Initializer for the variable.
 306       regularizer: A (Tensor -> Tensor or None) function; the result of
 307         applying it on a newly created variable will be added to the collection
 308         GraphKeys.REGULARIZATION_LOSSES and can be used for regularization.
 309       reuse: a Boolean, None, or tf.AUTO_REUSE. Controls reuse or creation
 310         of variables. When eager execution is enabled  this argument is always
 311         forced to be False.
 312       trainable: If `True` also add the variable to the graph collection
 313         `GraphKeys.TRAINABLE_VARIABLES` (see `tf.Variable`).
 314         `trainable` defaults to `True` unless `synchronization` is
 315         set to `ON_READ`.
 316       collections: List of graph collections keys to add the `Variable` to.
 317         Defaults to `[GraphKeys.GLOBAL_VARIABLES]` (see `tf.Variable`).
 318       caching_device: Optional device string or function describing where the
 319         Variable should be cached for reading.  Defaults to the Variable's
 320         device.  If not `None`, caches on another device.  Typical use is to
 321         cache on the device where the Ops using the `Variable` reside, to
 322         deduplicate copying through `Switch` and other conditional statements.
 323       partitioner: Optional callable that accepts a fully defined `TensorShape`
 324         and dtype of the `Variable` to be created, and returns a list of
 325         partitions for each axis (currently only one axis can be partitioned).
 326       validate_shape: If False, allows the variable to be initialized with a
 327         value of unknown shape. If True, the default, the shape of initial_value
 328         must be known.
 329       use_resource: If False, creates a regular Variable. If True, creates
 330         instead an experimental ResourceVariable which has well-defined
 331         semantics. Defaults to False (will later change to True).
 332         When eager execution is enabled this argument is always forced to be
 333         true.
 334       custom_getter: Callable that takes as a first argument the true getter,
 335         and allows overwriting the internal get_variable method.
 336         The signature of `custom_getter` should match that of this method,
 337         but the most future-proof version will allow for changes:
 338         `def custom_getter(getter, *args, **kwargs)`.  Direct access to
 339         all `get_variable` parameters is also allowed:
 340         `def custom_getter(getter, name, *args, **kwargs)`.  A simple identity
 341         custom getter that simply creates variables with modified names is:
 342         ```python
 343         def custom_getter(getter, name, *args, **kwargs):
 344           return getter(name + '_suffix', *args, **kwargs)
 345         ```
 346       constraint: An optional projection function to be applied to the variable
 347         after being updated by an `Optimizer` (e.g. used to implement norm
 348         constraints or value constraints for layer weights). The function must
 349         take as input the unprojected Tensor representing the value of the
 350         variable and return the Tensor for the projected value
 351         (which must have the same shape). Constraints are not safe to
 352         use when doing asynchronous distributed training.
 353       synchronization: Indicates when a distributed a variable will be
 354         aggregated. Accepted values are constants defined in the class
 355         `tf.VariableSynchronization`. By default the synchronization is set to
 356         `AUTO` and the current `DistributionStrategy` chooses
 357         when to synchronize. If `synchronization` is set to `ON_READ`,
 358         `trainable` must not be set to `True`.
 359       aggregation: Indicates how a distributed variable will be aggregated.
 360         Accepted values are constants defined in the class
 361         `tf.VariableAggregation`.
 362 
 363     Returns:
 364       The created or existing `Variable` (or `PartitionedVariable`, if a
 365       partitioner was used).
 366 
 367     Raises:
 368       ValueError: when creating a new variable and shape is not declared,
 369         when reusing a variable and specifying a conflicting shape,
 370         or when violating reuse during variable creation.
 371       RuntimeError: when eager execution is enabled and not called from an
 372         EagerVariableStore.
 373     """
 374     if custom_getter is not None and not callable(custom_getter):
 375       raise ValueError(
 376           "Passed a custom_getter which is not callable: %s" % custom_getter)
 377 
 378     with ops.init_scope():
 379       if context.executing_eagerly():
 380         # Variable creation and initialization takes place in `init_scope`s;
 381         # as such, if an `init_scope` lifts us into the eager context, then we
 382         # need to use `ResourceVariable`s.
 383         use_resource = True
 384 
 385     # Note that it's fine to reuse eager variables whose initialization was
 386     # lifted from a function-building graph into the eager context (that's why
 387     # the following clause is not wrapped in an `init_scope`); lifted variables
 388     # are tracked by the graph's `VariableStore`.
 389     if context.executing_eagerly():
 390       if not self._store_eager_variables and reuse:
 391         raise RuntimeError(
 392             "When eager execution is enabled variable reuse is only supported"
 393             " when an EagerVariableStore is active. See the documentation on"
 394             " EagerVariableStore for example usage.")
 395       if self._store_eager_variables:
 396         reuse = AUTO_REUSE
 397 
 398     # If a *_ref type is passed in an error would be triggered further down the
 399     # stack. We prevent this using base_dtype to get a non-ref version of the
 400     # type, before doing anything else. When _ref types are removed in favor of
 401     # resources, this line can be removed.
 402     try:
 403       dtype = dtype.base_dtype
 404     except AttributeError:
 405       # .base_dtype not existing means that we will try and use the raw dtype
 406       # which was passed in - this might be a NumPy type which is valid.
 407       pass
 408 
 409     # This is the main logic of get_variable.  However, custom_getter
 410     # may override this logic.  So we save it as a callable and pass
 411     # it to custom_getter.
 412     # Note: the parameters of _true_getter, and their documentation, match
 413     # *exactly* item-for-item with the docstring of this method.
 414     def _true_getter(  # pylint: disable=missing-docstring
 415         name,
 416         shape=None,
 417         dtype=dtypes.float32,
 418         initializer=None,
 419         regularizer=None,
 420         reuse=None,
 421         trainable=None,
 422         collections=None,
 423         caching_device=None,
 424         partitioner=None,
 425         validate_shape=True,
 426         use_resource=None,
 427         constraint=None,
 428         synchronization=VariableSynchronization.AUTO,
 429         aggregation=VariableAggregation.NONE):
 430       is_scalar = (shape is not None
 431                    and isinstance(shape, collections_lib.Sequence)
 432                    and not shape)
 433       # Partitioned variable case
 434       if partitioner is not None and not is_scalar:
 435         if not callable(partitioner):
 436           raise ValueError(
 437               "Partitioner must be callable, but received: %s" % partitioner)
 438         with ops.name_scope(None):
 439           return self._get_partitioned_variable(name=name,
 440                                                 shape=shape,
 441                                                 dtype=dtype,
 442                                                 initializer=initializer,
 443                                                 regularizer=regularizer,
 444                                                 reuse=reuse,
 445                                                 trainable=trainable,
 446                                                 collections=collections,
 447                                                 caching_device=caching_device,
 448                                                 partitioner=partitioner,
 449                                                 validate_shape=validate_shape,
 450                                                 use_resource=use_resource,
 451                                                 constraint=constraint)
 452 
 453       # Special case for partitioned variable to allow reuse without having to
 454       # specify partitioner.
 455       if (reuse is True and partitioner is None
 456           and name in self._partitioned_vars):
 457         return self._get_partitioned_variable(name=name,
 458                                               shape=shape,
 459                                               dtype=dtype,
 460                                               initializer=initializer,
 461                                               regularizer=regularizer,
 462                                               reuse=reuse,
 463                                               trainable=trainable,
 464                                               collections=collections,
 465                                               caching_device=caching_device,
 466                                               partitioner=None,
 467                                               validate_shape=validate_shape,
 468                                               use_resource=use_resource,
 469                                               constraint=constraint)
 470 
 471       # Single variable case
 472       if "%s/part_0" % name in self._vars:
 473         raise ValueError(
 474             "No partitioner was provided, but a partitioned version of the "
 475             "variable was found: %s/part_0. Perhaps a variable of the same "
 476             "name was already created with partitioning?" % name)
 477 
 478       return self._get_single_variable(
 479           name=name,
 480           shape=shape,
 481           dtype=dtype,
 482           initializer=initializer,
 483           regularizer=regularizer,
 484           reuse=reuse,
 485           trainable=trainable,
 486           collections=collections,
 487           caching_device=caching_device,
 488           validate_shape=validate_shape,
 489           use_resource=use_resource,
 490           constraint=constraint,
 491           synchronization=synchronization,
 492           aggregation=aggregation)
 493 
 494     # Set trainable value based on synchronization value.
 495     trainable = _get_trainable_value(
 496         synchronization=synchronization, trainable=trainable)
 497 
 498     if custom_getter is not None:
 499       # Handle backwards compatibility with getter arguments that were added
 500       # to the API after users started writing custom getters.
 501       custom_getter_kwargs = {
 502           "getter": _true_getter,
 503           "name": name,
 504           "shape": shape,
 505           "dtype": dtype,
 506           "initializer": initializer,
 507           "regularizer": regularizer,
 508           "reuse": reuse,
 509           "trainable": trainable,
 510           "collections": collections,
 511           "caching_device": caching_device,
 512           "partitioner": partitioner,
 513           "validate_shape": validate_shape,
 514           "use_resource": use_resource,
 515           "synchronization": synchronization,
 516           "aggregation": aggregation,
 517       }
 518       # `fn_args` and `has_kwargs` can handle functions, `functools.partial`,
 519       # `lambda`.
 520       if ("constraint" in function_utils.fn_args(custom_getter) or
 521           function_utils.has_kwargs(custom_getter)):
 522         custom_getter_kwargs["constraint"] = constraint
 523       return custom_getter(**custom_getter_kwargs)
 524     else:
 525       return _true_getter(
 526           name,
 527           shape=shape,
 528           dtype=dtype,
 529           initializer=initializer,
 530           regularizer=regularizer,
 531           reuse=reuse,
 532           trainable=trainable,
 533           collections=collections,
 534           caching_device=caching_device,
 535           partitioner=partitioner,
 536           validate_shape=validate_shape,
 537           use_resource=use_resource,
 538           constraint=constraint,
 539           synchronization=synchronization,
 540           aggregation=aggregation)

 542   def _get_partitioned_variable(self,
 543                                 name,
 544                                 partitioner,
 545                                 shape=None,
 546                                 dtype=dtypes.float32,
 547                                 initializer=None,
 548                                 regularizer=None,
 549                                 reuse=None,
 550                                 trainable=None,
 551                                 collections=None,
 552                                 caching_device=None,
 553                                 validate_shape=True,
 554                                 use_resource=None,
 555                                 constraint=None):
 556     """Gets or creates a sharded variable list with these parameters.
 557 
 558     The `partitioner` must be a callable that accepts a fully defined
 559     `TensorShape` and returns a sequence of integers (the `partitions`).
 560     These integers describe how to partition the given sharded `Variable`
 561     along the given dimension.  That is, `partitions[1] = 3` means split
 562     the `Variable` into 3 shards along dimension 1.  Currently, sharding along
 563     only one axis is supported.
 564 
 565     If the list of variables with the given name (prefix) is already stored,
 566     we return the stored variables. Otherwise, we create a new one.
 567 
 568     Set `reuse` to `True` when you only want to reuse existing Variables.
 569     Set `reuse` to `False` when you only want to create new Variables.
 570     Set `reuse` to None (the default) or tf.AUTO_REUSE when you want
 571     variables to be created if they don't exist or returned if they do.
 572 
 573     If initializer is `None` (the default), the default initializer passed in
 574     the constructor is used. If that one is `None` too, we use a new
 575     `glorot_uniform_initializer`. If initializer is a Tensor, we use
 576     it as a value and derive the shape from the initializer.
 577 
 578     If the initializer is a callable, then it will be called for each
 579     shard.  Otherwise the initializer should match the shape of the entire
 580     sharded Variable, and it will be sliced accordingly for each shard.
 581 
 582     Some useful partitioners are available.  See, e.g.,
 583     `variable_axis_size_partitioner` and `min_max_variable_partitioner`.
 584 
 585     Args:
 586       name: the name of the new or existing sharded variable.
 587       partitioner: Optional callable that accepts a fully defined `TensorShape`
 588         and `dtype` of the Variable to be created, and returns a list of
 589         partitions for each axis (currently only one axis can be partitioned).
 590       shape: shape of the new or existing sharded variable.
 591       dtype: type of the new or existing sharded variable
 592         (defaults to `DT_FLOAT`).
 593       initializer: initializer for the sharded variable.
 594       regularizer: a (Tensor -> Tensor or None) function; the result of
 595         applying it on a newly created variable will be added to the collection
 596         GraphKeys.REGULARIZATION_LOSSES and can be used for regularization.
 597       reuse: a Boolean, None, or tf.AUTO_REUSE. Controls reuse or creation
 598         of variables.
 599       trainable: If `True` also add the variable to the graph collection
 600         `GraphKeys.TRAINABLE_VARIABLES` (see `tf.Variable`).
 601       collections: List of graph collections keys to add the Variable to.
 602         Defaults to `[GraphKeys.GLOBAL_VARIABLES]` (see `tf.Variable`).
 603       caching_device: Optional device string or function describing where the
 604         Variable should be cached for reading.  Defaults to the Variable's
 605         device.  If not `None`, caches on another device.  Typical use is to
 606         cache on the device where the Ops using the Variable reside, to
 607         deduplicate copying through `Switch` and other conditional statements.
 608       validate_shape: If False, allows the variable to be initialized with a
 609         value of unknown shape. If True, the default, the shape of initial_value
 610         must be known.
 611       use_resource: If False, creates a regular Variable. If True, creates an
 612         experimental ResourceVariable which has well-defined semantics. Defaults
 613         to False (will later change to True).
 614       constraint: An optional projection function to be applied to the variable
 615         after being updated by an `Optimizer` (e.g. used to implement norm
 616         constraints or value constraints for layer weights). The function must
 617         take as input the unprojected Tensor representing the value of the
 618         variable and return the Tensor for the projected value
 619         (which must have the same shape). Constraints are not safe to
 620         use when doing asynchronous distributed training.
 621 
 622     Returns:
 623       A `PartitionedVariable` object.
 624 
 625     Raises:
 626       ValueError: when creating a new variable and shape is not declared,
 627         when reusing a variable and specifying a conflicting shape,
 628         when violating reuse during variable creation, or if an existing
 629         sharded variable exists for the given name but with different sharding.
 630     """
 631     if context.executing_eagerly():
 632       raise NotImplementedError("Partitioned variables are not yet supported "
 633                                 "when eager execution is enabled.")
 634 
 635     initializing_from_value = initializer is not None and isinstance(
 636         initializer, ops.Tensor)
 637     reuse_without_partition = reuse and not partitioner
 638 
 639     if name in self._vars:
 640       raise ValueError(
 641           "A partitioner was provided, but an unpartitioned version of the "
 642           "variable was found: %s.  Perhaps a variable of the same name was "
 643           "already created without partitioning?" % name)
 644 
 645     shape = tensor_shape.as_shape(shape)
 646     if initializing_from_value:
 647       shape = shape.merge_with(initializer.get_shape())
 648 
 649     if not reuse_without_partition:
 650       if not shape.is_fully_defined():
 651         raise ValueError("Shape of a new partitioned variable (%s) must be "
 652                          "fully defined, but instead was %s." % (name, shape))
 653 
 654       if shape.ndims < 1:
 655         raise ValueError("A partitioned Variable must have rank at least 1, "
 656                          "shape: %s" % shape)
 657 
 658       partitions = partitioner(shape=shape, dtype=dtype)
 659 
 660       if not isinstance(partitions, collections_lib.Sequence):
 661         raise ValueError("Partitioner must return a sequence, but saw: %s"
 662                          % partitions)
 663 
 664       if len(partitions) != shape.ndims:
 665         raise ValueError(
 666             "Partitioner returned a partition list that does not match the "
 667             "Variable's rank: %s vs. %s" % (partitions, shape))
 668 
 669       if any([p < 1 for p in partitions]):
 670         raise ValueError(
 671             "Partitioner returned zero partitions for some axes: %s" %
 672             partitions)
 673 
 674     if name in self._partitioned_vars:
 675       if reuse is False:
 676         raise ValueError(
 677             "Partitioned variable with name %s already exists. Did you mean to "
 678             "set reuse=True or reuse=tf.AUTO_REUSE in VarScope?"
 679             % name)
 680 
 681       existing_var = self._partitioned_vars[name]
 682       if not shape.is_compatible_with(existing_var.get_shape()):
 683         raise ValueError(
 684             "Trying to reuse partitioned variable %s, but specified shape %s "
 685             "and found shape %s."
 686             % (name, shape, existing_var.get_shape()))
 687       if not dtype.is_compatible_with(existing_var.dtype):
 688         raise ValueError(
 689             "Trying to reuse partitioned variable %s, but specified dtype %s "
 690             "and found dtype %s."
 691             % (name, dtype.name, existing_var.dtype.name))
 692 
 693       # pylint: disable=protected-access
 694       if (not reuse_without_partition and
 695           existing_var._get_partitions() != partitions):
 696         raise ValueError(
 697             "Trying to reuse partitioned variable %s, but specified partitions "
 698             "%s and found partitions %s." %
 699             (name, partitions, existing_var._get_partitions()))
 700       # pylint: enable=protected-access
 701 
 702       return existing_var
 703 
 704     if reuse is True:
 705       raise ValueError("PartitionedVariable %s does not exist, or was not "
 706                        "created with tf.get_variable(). Did you mean to set "
 707                        "reuse=False or reuse=tf.AUTO_REUSE in VarScope?" % name)
 708 
 709     slice_dim, slice_shape = _compute_slice_dim_and_shape(
 710         shape.as_list(), partitions)
 711 
 712     vs = []
 713     num_slices = partitions[slice_dim]
 714     num_slices_with_excess = shape[slice_dim].value % num_slices
 715 
 716     slice_offset = [0] * shape.ndims
 717 
 718     if "%s/part_0" % name in self._vars:
 719       if "%s/part_%d" % (name, num_slices - 1) not in self._vars:
 720         raise ValueError(
 721             "Partitioner returned a different partitioning than what was "
 722             "already found.  Partitioner returned %d shards, and shard "
 723             "%s/part_0 was found, but %s/part_%d was not."
 724             % (num_slices, name, name, num_slices - 1))
 725       if "%s/part_%d" % (name, num_slices) in self._vars:
 726         raise ValueError(
 727             "Partitioner returned a different partitioning than what was "
 728             "already found.  Partitioner returned %d shards, and shard "
 729             "%s/part_0 was found, but so was the extra shard %s/part_%d."
 730             % (num_slices, name, name, num_slices))
 731 
 732     for i in xrange(num_slices):
 733       var_shape = slice_shape[:]
 734       var_offset = slice_offset[:]
 735       partition_info = _PartitionInfo(
 736           full_shape=shape.as_list(), var_offset=var_offset)
 737       if i < num_slices_with_excess:
 738         var_shape[slice_dim] += 1
 739       slice_offset[slice_dim] += var_shape[slice_dim]
 740 
 741       var_full_name = "%s/part_%d" % (name, i)
 742       with ops.name_scope(var_full_name + "/PartitionedInitializer"):
 743         # Create the tensor to initialize the variable with default value.
 744         if initializer is None:
 745           init, initializing_from_value = self._get_default_initializer(
 746               name=name, shape=shape, dtype=dtype)
 747           if initializing_from_value:
 748             init_shape = None
 749           else:
 750             init_shape = var_shape
 751         elif callable(initializer):
 752           init = initializer
 753           init_shape = var_shape
 754         elif isinstance(initializer, ops.Tensor):
 755           init = array_ops.slice(initializer, var_offset, var_shape)
 756           # Use the dtype of the given tensor.
 757           dtype = init.dtype.base_dtype
 758           init_shape = None
 759         else:
 760           init = ops.convert_to_tensor(initializer, dtype=dtype)
 761           init = array_ops.slice(init, var_offset, var_shape)
 762           init_shape = None
 763 
 764       with ops.name_scope(None):
 765         var = self._get_single_variable(
 766             name=var_full_name,
 767             shape=init_shape,
 768             dtype=dtype,
 769             initializer=init,
 770             partition_info=partition_info,
 771             regularizer=regularizer,
 772             reuse=reuse,
 773             trainable=trainable,
 774             collections=collections,
 775             caching_device=caching_device,
 776             validate_shape=validate_shape,
 777             use_resource=use_resource,
 778             constraint=constraint)
 779 
 780       # pylint: disable=protected-access
 781       var._set_save_slice_info(variables.Variable.SaveSliceInfo(
 782           name, shape.as_list(), var_offset, var_shape))
 783       vs.append(var)
 784       # pylint: enable=protected-access
 785 
 786       # pylint: disable=protected-access
 787     partitioned_var = variables.PartitionedVariable(name=name,
 788                                                     shape=shape,
 789                                                     dtype=dtype,
 790                                                     variable_list=vs,
 791                                                     partitions=partitions)
 792     # pylint: enable=protected-access
 793 
 794     self._partitioned_vars[name] = partitioned_var
 795     return partitioned_var

 797   def _get_single_variable(self,
 798                            name,
 799                            shape=None,
 800                            dtype=dtypes.float32,
 801                            initializer=None,
 802                            regularizer=None,
 803                            partition_info=None,
 804                            reuse=None,
 805                            trainable=None,
 806                            collections=None,
 807                            caching_device=None,
 808                            validate_shape=True,
 809                            use_resource=None,
 810                            constraint=None,
 811                            synchronization=VariableSynchronization.AUTO,
 812                            aggregation=VariableAggregation.NONE):
 813     """Get or create a single Variable (e.g. a shard or entire variable).
 814 
 815     See the documentation of get_variable above (ignore partitioning components)
 816     for details.
 817 
 818     Args:
 819       name: see get_variable.
 820       shape: see get_variable.
 821       dtype: see get_variable.
 822       initializer: see get_variable.
 823       regularizer: see get_variable.
 824       partition_info: _PartitionInfo object.
 825       reuse: see get_variable.
 826       trainable: see get_variable.
 827       collections: see get_variable.
 828       caching_device: see get_variable.
 829       validate_shape: see get_variable.
 830       use_resource: see get_variable.
 831       constraint: see get_variable.
 832       synchronization: see get_variable.
 833       aggregation: see get_variable.
 834 
 835     Returns:
 836       A Variable.  See documentation of get_variable above.
 837 
 838     Raises:
 839       ValueError: See documentation of get_variable above.
 840     """
 841     # Set to true if initializer is a constant.
 842     initializing_from_value = False
 843     if initializer is not None and not callable(initializer):
 844       initializing_from_value = True
 845     if shape is not None and initializing_from_value:
 846       raise ValueError("If initializer is a constant, do not specify shape.")
 847 
 848     dtype = dtypes.as_dtype(dtype)
 849     shape = tensor_shape.as_shape(shape)
 850 
 851     if name in self._vars:
 852       # Here we handle the case when returning an existing variable.
 853       if reuse is False:
 854         tb = self._vars[name].op.traceback[::-1]
 855         # Throw away internal tf entries and only take a few lines.
 856         tb = [x for x in tb if "tensorflow/python" not in x[0]][:3]
 857         raise ValueError("Variable %s already exists, disallowed."
 858                          " Did you mean to set reuse=True or "
 859                          "reuse=tf.AUTO_REUSE in VarScope? "
 860                          "Originally defined at:\n\n%s" % (
 861                              name, "".join(traceback.format_list(tb))))
 862       found_var = self._vars[name]
 863       if not shape.is_compatible_with(found_var.get_shape()):
 864         raise ValueError("Trying to share variable %s, but specified shape %s"
 865                          " and found shape %s." % (name, shape,
 866                                                    found_var.get_shape()))
 867       if not dtype.is_compatible_with(found_var.dtype):
 868         dtype_str = dtype.name
 869         found_type_str = found_var.dtype.name
 870         raise ValueError("Trying to share variable %s, but specified dtype %s"
 871                          " and found dtype %s." % (name, dtype_str,
 872                                                    found_type_str))
 873       return found_var
 874 
 875     # The code below handles only the case of creating a new variable.
 876     if reuse is True:
 877       raise ValueError("Variable %s does not exist, or was not created with "
 878                        "tf.get_variable(). Did you mean to set "
 879                        "reuse=tf.AUTO_REUSE in VarScope?" % name)
 880 
 881     # Create the tensor to initialize the variable with default value.
 882     if initializer is None:
 883       initializer, initializing_from_value = self._get_default_initializer(
 884           name=name, shape=shape, dtype=dtype)
 885     # Enter an init scope when creating the initializer.
 886     with ops.init_scope():
 887       if initializing_from_value:
 888         init_val = initializer
 889         variable_dtype = None
 890       else:
 891         # Instantiate initializer if provided initializer is a type object.
 892         if isinstance(initializer, type(init_ops.Initializer)):
 893           initializer = initializer(dtype=dtype)
 894         if shape and shape.is_fully_defined():
 895           init_val = lambda: initializer(  # pylint: disable=g-long-lambda
 896               shape.as_list(), dtype=dtype, partition_info=partition_info)
 897         elif not tf_inspect.getargspec(initializer).args:
 898           init_val = initializer
 899         else:
 900           raise ValueError("You can only pass an initializer function that "
 901                            "expects no arguments to its callable when the "
 902                            "shape is not fully defined. The given initializer "
 903                            "function expects the following args %s" %
 904                            tf_inspect.getargspec(initializer).args)
 905         variable_dtype = dtype.base_dtype
 906 
 907     # Create the variable.
 908     if use_resource is None:
 909       # Set the default value if unspecified.
 910       use_resource = _DEFAULT_USE_RESOURCE
 911     v = variables.VariableV1(
 912         initial_value=init_val,
 913         name=name,
 914         trainable=trainable,
 915         collections=collections,
 916         caching_device=caching_device,
 917         dtype=variable_dtype,
 918         validate_shape=validate_shape,
 919         constraint=constraint,
 920         use_resource=use_resource,
 921         synchronization=synchronization,
 922         aggregation=aggregation)
 923     if context.executing_eagerly() and self._store_eager_variables:
 924       if collections:
 925         ops.add_to_collections(collections, v)
 926       else:
 927         ops.add_to_collection(ops.GraphKeys.GLOBAL_VARIABLES, v)
 928       if trainable:
 929         ops.add_to_collection(ops.GraphKeys.TRAINABLE_VARIABLES, v)
 930 
 931     if not context.executing_eagerly() or self._store_eager_variables:
 932       # In eager mode we do not want to keep default references to Variable
 933       # objects as this will prevent their memory from being released.
 934       self._vars[name] = v
 935     logging.vlog(1, "Created variable %s with shape %s and init %s", v.name,
 936                  format(shape), initializer)
 937 
 938     # Run the regularizer if requested and save the resulting loss.
 939     if regularizer:
 940       with ops.colocate_with(v):
 941         with ops.name_scope(name + "/Regularizer/"):
 942           with ops.init_scope():
 943             loss = regularizer(v)
 944         if loss is not None:
 945           if context.executing_eagerly():
 946             v_name = "v_%s" % type(v)
 947             loss_name = "loss_%s" % type(loss)
 948           else:
 949             v_name = v.name
 950             loss_name = loss.name
 951           logging.vlog(1, "Applied regularizer to %s and added the result %s "
 952                        "to REGULARIZATION_LOSSES.", v_name, loss_name)
 953           ops.add_to_collection(ops.GraphKeys.REGULARIZATION_LOSSES, loss)
 954     return v

 957   def _get_default_initializer(self, name, shape=None, dtype=dtypes.float32):
 958     """Provide a default initializer and a corresponding value.
 959 
 960     Args:
 961       name: see get_variable.
 962       shape: see get_variable.
 963       dtype: see get_variable.
 964 
 965     Returns:
 966       initializer and initializing_from_value. See get_variable above.
 967 
 968     Raises:
 969       ValueError: When giving unsupported dtype.
 970     """
 971     del shape
 972     # If dtype is DT_FLOAT, provide a uniform unit scaling initializer
 973     if dtype.is_floating:
 974       initializer = init_ops.glorot_uniform_initializer()
 975       initializing_from_value = False
 976     # If dtype is DT_INT/DT_UINT, provide a default value `zero`
 977     # If dtype is DT_BOOL, provide a default value `FALSE`
 978     elif (dtype.is_integer or dtype.is_unsigned or dtype.is_bool
 979           or dtype == dtypes.string):
 980       initializer = init_ops.zeros_initializer()
 981       initializing_from_value = False
 982     # NOTES:Do we need to support for handling DT_STRING and DT_COMPLEX here?
 983     else:
 984       raise ValueError("An initializer for variable %s of %s is required"
 985                        % (name, dtype.base_dtype))
 986 
 987     return initializer, initializing_from_value

1952

1957

1960   see the [Variable Scope How To](https://tensorflow.org/guide/variables), here

1964 
1965   ```python

2358 def _compute_slice_dim_and_shape(full_shape, slicing):
2359   """Computes which dimension is being sliced and the typical slice shape."""
2360 
2361   slice_shape = [0] * len(full_shape)
2362   slice_dim = None
2363   for dim, num_slices in enumerate(slicing):
2364     dim_size = full_shape[dim]
2365     if num_slices <= 0 or dim_size < num_slices:
2366       raise ValueError("Cannot create %d slices for size %d. shape: %s, "
2367                        "slicing: %s" %
2368                        (num_slices, full_shape[dim], full_shape, slicing))
2369     if num_slices == 1:
2370       # Not slicing in this dimension.
2371       slice_shape[dim] = dim_size
2372     elif slice_dim is not None:
2373       # We only support slicing along one of the dimensions.
2374       raise ValueError("Can only slice a variable along one dimension: "
2375                        "shape: %s, slicing: %s" % (full_shape, slicing))
2376     else:
2377       # Note: We will add any extras onto the last slice, later.
2378       slice_dim = dim
2379       slice_shape[dim] = dim_size // num_slices
2380 
2381   # Degenerate case: If "slicing" was all ones, pretend we are slicing along
2382   # the first dimension.
2383   if slice_dim is None:
2384     slice_dim = 0
2385   return slice_dim, slice_shape
