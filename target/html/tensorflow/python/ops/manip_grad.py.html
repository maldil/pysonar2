  15 """Gradients for operators defined in manip_ops.py."""

  17 from __future__ import absolute_import
  18 from __future__ import division
  19 from __future__ import print_function
  20 
  21 from tensorflow.python.framework import ops
  22 from tensorflow.python.ops import manip_ops
  23 
  24 
  25 @ops.RegisterGradient("Roll")
  26 def _RollGrad(op, grad):
  27   # The gradient is just the roll reversed
  28   shift = op.inputs[1]
  29   axis = op.inputs[2]
  30   roll_grad = manip_ops.roll(grad, -shift, axis)
  31   return roll_grad, None, None
