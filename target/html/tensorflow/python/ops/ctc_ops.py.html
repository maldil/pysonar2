  32 @tf_export("nn.ctc_loss")
  33 def ctc_loss(labels, inputs, sequence_length,
  34              preprocess_collapse_repeated=False,
  35              ctc_merge_repeated=True,
  36              ignore_longer_outputs_than_inputs=False, time_major=True):
  37   """Computes the CTC (Connectionist Temporal Classification) Loss.
  38 
  39   This op implements the CTC loss as presented in the article:
  40 
  41   [A. Graves, S. Fernandez, F. Gomez, J. Schmidhuber.
  42   Connectionist Temporal Classification: Labeling Unsegmented Sequence Data
  43   with Recurrent Neural Networks. ICML 2006, Pittsburgh, USA,
  44   pp. 369-376.](http://www.cs.toronto.edu/~graves/icml_2006.pdf)
  45 
  46   Input requirements:
  47 
  48   ```
  49   sequence_length(b) <= time for all b
  50 
  51   max(labels.indices(labels.indices[:, 1] == b, 2))
  52     <= sequence_length(b) for all b.
  53   ```
  54 
  55   Notes:
  56 
  57   This class performs the softmax operation for you, so inputs should
  58   be e.g. linear projections of outputs by an LSTM.
  59 
  60   The `inputs` Tensor's innermost dimension size, `num_classes`, represents
  61   `num_labels + 1` classes, where num_labels is the number of true labels, and
  62   the largest value `(num_classes - 1)` is reserved for the blank label.
  63 
  64   For example, for a vocabulary containing 3 labels `[a, b, c]`,
  65   `num_classes = 4` and the labels indexing is `{a: 0, b: 1, c: 2, blank: 3}`.
  66 
  67   Regarding the arguments `preprocess_collapse_repeated` and
  68   `ctc_merge_repeated`:
  69 
  70   If `preprocess_collapse_repeated` is True, then a preprocessing step runs
  71   before loss calculation, wherein repeated labels passed to the loss
  72   are merged into single labels.  This is useful if the training labels come
  73   from, e.g., forced alignments and therefore have unnecessary repetitions.
  74 
  75   If `ctc_merge_repeated` is set False, then deep within the CTC calculation,
  76   repeated non-blank labels will not be merged and are interpreted
  77   as individual labels.  This is a simplified (non-standard) version of CTC.
  78 
  79   Here is a table of the (roughly) expected first order behavior:
  80 
  81   * `preprocess_collapse_repeated=False`, `ctc_merge_repeated=True`
  82 
  83     Classical CTC behavior: Outputs true repeated classes with blanks in
  84     between, and can also output repeated classes with no blanks in
  85     between that need to be collapsed by the decoder.
  86 
  87   * `preprocess_collapse_repeated=True`, `ctc_merge_repeated=False`
  88 
  89     Never learns to output repeated classes, as they are collapsed
  90     in the input labels before training.
  91 
  92   * `preprocess_collapse_repeated=False`, `ctc_merge_repeated=False`
  93 
  94     Outputs repeated classes with blanks in between, but generally does not
  95     require the decoder to collapse/merge repeated classes.
  96 
  97   * `preprocess_collapse_repeated=True`, `ctc_merge_repeated=True`
  98 
  99     Untested.  Very likely will not learn to output repeated classes.
 100 
 101   The `ignore_longer_outputs_than_inputs` option allows to specify the behavior
 102   of the CTCLoss when dealing with sequences that have longer outputs than
 103   inputs. If true, the CTCLoss will simply return zero gradient for those
 104   items, otherwise an InvalidArgument error is returned, stopping training.
 105 
 106   Args:
 107     labels: An `int32` `SparseTensor`.
 108       `labels.indices[i, :] == [b, t]` means `labels.values[i]` stores
 109       the id for (batch b, time t).
 110       `labels.values[i]` must take on values in `[0, num_labels)`.
 111       See `core/ops/ctc_ops.cc` for more details.
 112     inputs: 3-D `float` `Tensor`.
 113       If time_major == False, this will be a `Tensor` shaped:
 114         `[batch_size, max_time, num_classes]`.
 115       If time_major == True (default), this will be a `Tensor` shaped:
 116         `[max_time, batch_size, num_classes]`.
 117       The logits.
 118     sequence_length: 1-D `int32` vector, size `[batch_size]`.
 119       The sequence lengths.
 120     preprocess_collapse_repeated: Boolean.  Default: False.
 121       If True, repeated labels are collapsed prior to the CTC calculation.
 122     ctc_merge_repeated: Boolean.  Default: True.
 123     ignore_longer_outputs_than_inputs: Boolean. Default: False.
 124       If True, sequences with longer outputs than inputs will be ignored.
 125     time_major: The shape format of the `inputs` Tensors.
 126       If True, these `Tensors` must be shaped `[max_time, batch_size,
 127       num_classes]`.
 128       If False, these `Tensors` must be shaped `[batch_size, max_time,
 129       num_classes]`.
 130       Using `time_major = True` (default) is a bit more efficient because it
 131       avoids transposes at the beginning of the ctc_loss calculation.  However,
 132       most TensorFlow data is batch-major, so by this function also accepts
 133       inputs in batch-major form.
 134 
 135   Returns:
 136     A 1-D `float` `Tensor`, size `[batch]`, containing the negative log
 137       probabilities.
 138 
 139   Raises:
 140     TypeError: if labels is not a `SparseTensor`.
 141   """
 142   # The second, third, etc output tensors contain the gradients.  We use it in
 143   # _CTCLossGrad() below.
 144   if not isinstance(labels, sparse_tensor.SparseTensor):
 145     raise TypeError("Expected labels (first argument) to be a SparseTensor")
 146 
 147   # For internal calculations, we transpose to [time, batch, num_classes]
 148   if not time_major:
 149     inputs = array_ops.transpose(inputs, [1, 0, 2])  # (B,T,N) => (T,B,N)
 150 
 151   loss, _ = gen_ctc_ops.ctc_loss(
 152       inputs,
 153       labels.indices,
 154       labels.values,
 155       sequence_length,
 156       preprocess_collapse_repeated=preprocess_collapse_repeated,
 157       ctc_merge_repeated=ctc_merge_repeated,
 158       ignore_longer_outputs_than_inputs=ignore_longer_outputs_than_inputs)
 159 
 160   return loss

 163 # pylint: disable=unused-argument
 164 @ops.RegisterGradient("CTCLoss")
 165 def _CTCLossGrad(op, grad_loss, _):
 166   """The derivative provided by CTC Loss.
 167 
 168   Args:
 169      op: the CTCLoss op.
 170      grad_loss: The backprop for cost.
 171 
 172   Returns:
 173      The CTC Loss gradient.
 174   """
 175   # Outputs are: loss, grad
 176   #
 177   # Currently there is no way to take the second derivative of this op
 178   # due to the fused implementation's interaction with tf.gradients(),
 179   # so we make sure we prevent silently incorrect results by raising
 180   # an error if the second derivative is requested via prevent_gradient.
 181   grad_without_gradient = array_ops.prevent_gradient(
 182       op.outputs[1], message="Currently there is no way to take the second "
 183       " derivative of ctc_loss due to the fused implementation's interaction "
 184       " with tf.gradients()")
 185   # Return gradient for inputs and None for
 186   # labels_indices, labels_values and sequence_length
 187   return [_BroadcastMul(grad_loss, grad_without_gradient), None, None, None]

 190 @tf_export("nn.ctc_greedy_decoder")
 191 def ctc_greedy_decoder(inputs, sequence_length, merge_repeated=True):
 192   """Performs greedy decoding on the logits given in input (best path).
 193 
 194   Note: Regardless of the value of merge_repeated, if the maximum index of a
 195   given time and batch corresponds to the blank index `(num_classes - 1)`, no
 196   new element is emitted.
 197 
 198   If `merge_repeated` is `True`, merge repeated classes in output.
 199   This means that if consecutive logits' maximum indices are the same,
 200   only the first of these is emitted.  The sequence `A B B * B * B` (where '*'
 201   is the blank label) becomes
 202 
 203     * `A B B B` if `merge_repeated=True`.
 204     * `A B B B B` if `merge_repeated=False`.
 205 
 206   Args:
 207     inputs: 3-D `float` `Tensor` sized
 208       `[max_time, batch_size, num_classes]`.  The logits.
 209     sequence_length: 1-D `int32` vector containing sequence lengths,
 210       having size `[batch_size]`.
 211     merge_repeated: Boolean.  Default: True.
 212 
 213   Returns:
 214     A tuple `(decoded, neg_sum_logits)` where
 215     decoded: A single-element list. `decoded[0]`
 216       is an `SparseTensor` containing the decoded outputs s.t.:
 217       `decoded.indices`: Indices matrix `(total_decoded_outputs, 2)`.
 218         The rows store: `[batch, time]`.
 219       `decoded.values`: Values vector, size `(total_decoded_outputs)`.
 220         The vector stores the decoded classes.
 221       `decoded.dense_shape`: Shape vector, size `(2)`.
 222         The shape values are: `[batch_size, max_decoded_length]`
 223     neg_sum_logits: A `float` matrix `(batch_size x 1)` containing, for the
 224         sequence found, the negative of the sum of the greatest logit at each
 225         timeframe.
 226   """
 227   outputs = gen_ctc_ops.ctc_greedy_decoder(
 228       inputs, sequence_length, merge_repeated=merge_repeated)
 229   (decoded_ix, decoded_val, decoded_shape, log_probabilities) = outputs
 230   return ([sparse_tensor.SparseTensor(decoded_ix, decoded_val, decoded_shape)],
 231           log_probabilities)

 234 @tf_export("nn.ctc_beam_search_decoder")
 235 def ctc_beam_search_decoder(inputs, sequence_length, beam_width=100,
 236                             top_paths=1, merge_repeated=True):
 237   """Performs beam search decoding on the logits given in input.
 238 
 239   **Note** The `ctc_greedy_decoder` is a special case of the
 240   `ctc_beam_search_decoder` with `top_paths=1` and `beam_width=1` (but
 241   that decoder is faster for this special case).
 242 
 243   If `merge_repeated` is `True`, merge repeated classes in the output beams.
 244   This means that if consecutive entries in a beam are the same,
 245   only the first of these is emitted.  That is, when the sequence is
 246   `A B B * B * B` (where '*' is the blank label), the return value is:
 247 
 248     * `A B` if `merge_repeated = True`.
 249     * `A B B B` if `merge_repeated = False`.
 250 
 251   Args:
 252     inputs: 3-D `float` `Tensor`, size
 253       `[max_time x batch_size x num_classes]`.  The logits.
 254     sequence_length: 1-D `int32` vector containing sequence lengths,
 255       having size `[batch_size]`.
 256     beam_width: An int scalar >= 0 (beam search beam width).
 257     top_paths: An int scalar >= 0, <= beam_width (controls output size).
 258     merge_repeated: Boolean.  Default: True.
 259 
 260   Returns:
 261     A tuple `(decoded, log_probabilities)` where
 262     decoded: A list of length top_paths, where `decoded[j]`
 263       is a `SparseTensor` containing the decoded outputs:
 264       `decoded[j].indices`: Indices matrix `(total_decoded_outputs[j] x 2)`
 265         The rows store: [batch, time].
 266       `decoded[j].values`: Values vector, size `(total_decoded_outputs[j])`.
 267         The vector stores the decoded classes for beam j.
 268       `decoded[j].dense_shape`: Shape vector, size `(2)`.
 269         The shape values are: `[batch_size, max_decoded_length[j]]`.
 270     log_probability: A `float` matrix `(batch_size x top_paths)` containing
 271         sequence log-probabilities.
 272   """
 273 
 274   decoded_ixs, decoded_vals, decoded_shapes, log_probabilities = (
 275       gen_ctc_ops.ctc_beam_search_decoder(
 276           inputs, sequence_length, beam_width=beam_width, top_paths=top_paths,
 277           merge_repeated=merge_repeated))
 278 
 279   return (
 280       [sparse_tensor.SparseTensor(ix, val, shape) for (ix, val, shape)
 281        in zip(decoded_ixs, decoded_vals, decoded_shapes)],
 282       log_probabilities)
