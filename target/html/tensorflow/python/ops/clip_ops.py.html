  37 @tf_export("clip_by_value")
  38 def clip_by_value(t, clip_value_min, clip_value_max,
  39                   name=None):
  40   """Clips tensor values to a specified min and max.
  41 
  42   Given a tensor `t`, this operation returns a tensor of the same type and
  43   shape as `t` with its values clipped to `clip_value_min` and `clip_value_max`.
  44   Any values less than `clip_value_min` are set to `clip_value_min`. Any values
  45   greater than `clip_value_max` are set to `clip_value_max`.
  46 
  47   Note: `clip_value_min` needs to be smaller or equal to `clip_value_max` for
  48   correct results.
  49 
  50   Args:
  51     t: A `Tensor`.
  52     clip_value_min: A 0-D (scalar) `Tensor`, or a `Tensor` with the same shape
  53       as `t`. The minimum value to clip by.
  54     clip_value_max: A 0-D (scalar) `Tensor`, or a `Tensor` with the same shape
  55       as `t`. The maximum value to clip by.
  56     name: A name for the operation (optional).
  57 
  58   Returns:
  59     A clipped `Tensor`.
  60 
  61   Raises:
  62     ValueError: If the clip tensors would trigger array broadcasting
  63       that would make the returned tensor larger than the input.
  64   """
  65   with ops.name_scope(name, "clip_by_value",
  66                       [t, clip_value_min, clip_value_max]) as name:
  67     t = ops.convert_to_tensor(t, name="t")
  68 
  69     # Go through list of tensors, for each value in each tensor clip
  70     t_min = math_ops.minimum(t, clip_value_max)
  71     # Assert that the shape is compatible with the initial shape,
  72     # to prevent unintentional broadcasting.
  73     _ = t.shape.merge_with(t_min.shape)
  74 
  75     t_max = math_ops.maximum(t_min, clip_value_min, name=name)
  76     _ = t.shape.merge_with(t_max.shape)
  77 
  78   return t_max
  79   # TODO(scottzhu): switch to use new implmentation in 2 weeks.
  80   # return gen_math_ops.clip_by_value(
  81   #     t, clip_value_min, clip_value_max, name=name)

  85 # @ops.RegisterGradient("ClipByValue")
  86 def _clip_by_value_grad(op, grad):
  87   """Returns grad of clip_by_value."""
  88   x = op.inputs[0]
  89   y = op.inputs[1]
  90   z = op.inputs[2]
  91   gdtype = grad.dtype
  92   sx = array_ops.shape(x)
  93   sy = array_ops.shape(y)
  94   sz = array_ops.shape(z)
  95   gradshape = array_ops.shape(grad)
  96   zeros = array_ops.zeros(gradshape, gdtype)
  97   xymask = math_ops.less(x, y)
  98   xzmask = math_ops.greater(x, z)
  99   rx, ry = gen_array_ops.broadcast_gradient_args(sx, sy)
 100   rx, rz = gen_array_ops.broadcast_gradient_args(sx, sz)
 101   xgrad = array_ops.where(math_ops.logical_or(xymask, xzmask), zeros, grad)
 102   ygrad = array_ops.where(xymask, grad, zeros)
 103   zgrad = array_ops.where(xzmask, grad, zeros)
 104   gx = array_ops.reshape(math_ops.reduce_sum(xgrad, rx), sx)
 105   gy = array_ops.reshape(math_ops.reduce_sum(ygrad, ry), sy)
 106   gz = array_ops.reshape(math_ops.reduce_sum(zgrad, rz), sz)
 107   return (gx, gy, gz)

 110 @tf_export("clip_by_norm")
 111 def clip_by_norm(t, clip_norm, axes=None, name=None):
 112   """Clips tensor values to a maximum L2-norm.
 113 
 114   Given a tensor `t`, and a maximum clip value `clip_norm`, this operation
 115   normalizes `t` so that its L2-norm is less than or equal to `clip_norm`,
 116   along the dimensions given in `axes`. Specifically, in the default case
 117   where all dimensions are used for calculation, if the L2-norm of `t` is
 118   already less than or equal to `clip_norm`, then `t` is not modified. If
 119   the L2-norm is greater than `clip_norm`, then this operation returns a
 120   tensor of the same type and shape as `t` with its values set to:
 121 
 122   `t * clip_norm / l2norm(t)`
 123 
 124   In this case, the L2-norm of the output tensor is `clip_norm`.
 125 
 126   As another example, if `t` is a matrix and `axes == [1]`, then each row
 127   of the output will have L2-norm equal to `clip_norm`. If `axes == [0]`
 128   instead, each column of the output will be clipped.
 129 
 130   This operation is typically used to clip gradients before applying them with
 131   an optimizer.
 132 
 133   Args:
 134     t: A `Tensor`.
 135     clip_norm: A 0-D (scalar) `Tensor` > 0. A maximum clipping value.
 136     axes: A 1-D (vector) `Tensor` of type int32 containing the dimensions
 137       to use for computing the L2-norm. If `None` (the default), uses all
 138       dimensions.
 139     name: A name for the operation (optional).
 140 
 141   Returns:
 142     A clipped `Tensor`.
 143   """
 144   with ops.name_scope(name, "clip_by_norm", [t, clip_norm]) as name:
 145     t = ops.convert_to_tensor(t, name="t")
 146 
 147     # Calculate L2-norm, clip elements by ratio of clip_norm to L2-norm
 148     l2sum = math_ops.reduce_sum(t * t, axes, keepdims=True)
 149     pred = l2sum > 0
 150     # Two-tap tf.where trick to bypass NaN gradients
 151     l2sum_safe = array_ops.where(pred, l2sum, array_ops.ones_like(l2sum))
 152     l2norm = array_ops.where(pred, math_ops.sqrt(l2sum_safe), l2sum)
 153     intermediate = t * clip_norm
 154     # Assert that the shape is compatible with the initial shape,
 155     # to prevent unintentional broadcasting.
 156     _ = t.shape.merge_with(intermediate.shape)
 157     tclip = array_ops.identity(
 158         intermediate / math_ops.maximum(l2norm, clip_norm), name=name)
 159 
 160   return tclip

 163 @tf_export("linalg.global_norm", "global_norm")
 164 @deprecation.deprecated_endpoints("global_norm")
 165 def global_norm(t_list, name=None):
 166   """Computes the global norm of multiple tensors.
 167 
 168   Given a tuple or list of tensors `t_list`, this operation returns the
 169   global norm of the elements in all tensors in `t_list`. The global norm is
 170   computed as:
 171 
 172   `global_norm = sqrt(sum([l2norm(t)**2 for t in t_list]))`
 173 
 174   Any entries in `t_list` that are of type None are ignored.
 175 
 176   Args:
 177     t_list: A tuple or list of mixed `Tensors`, `IndexedSlices`, or None.
 178     name: A name for the operation (optional).
 179 
 180   Returns:
 181     A 0-D (scalar) `Tensor` of type `float`.
 182 
 183   Raises:
 184     TypeError: If `t_list` is not a sequence.
 185   """
 186   if (not isinstance(t_list, collections.Sequence)
 187       or isinstance(t_list, six.string_types)):
 188     raise TypeError("t_list should be a sequence")
 189   t_list = list(t_list)
 190   with ops.name_scope(name, "global_norm", t_list) as name:
 191     values = [
 192         ops.convert_to_tensor(
 193             t.values if isinstance(t, ops.IndexedSlices) else t,
 194             name="t_%d" % i)
 195         if t is not None else t
 196         for i, t in enumerate(t_list)]
 197     half_squared_norms = []
 198     for v in values:
 199       if v is not None:
 200         with ops.colocate_with(v):
 201           half_squared_norms.append(gen_nn_ops.l2_loss(v))
 202 
 203     half_squared_norm = math_ops.reduce_sum(array_ops.stack(half_squared_norms))
 204 
 205     norm = math_ops.sqrt(
 206         half_squared_norm *
 207         constant_op.constant(2.0, dtype=half_squared_norm.dtype),
 208         name="global_norm")
 209 
 210   return norm

 213 @tf_export("clip_by_global_norm")
 214 def clip_by_global_norm(t_list, clip_norm, use_norm=None, name=None):
 215   """Clips values of multiple tensors by the ratio of the sum of their norms.
 216 
 217   Given a tuple or list of tensors `t_list`, and a clipping ratio `clip_norm`,
 218   this operation returns a list of clipped tensors `list_clipped`
 219   and the global norm (`global_norm`) of all tensors in `t_list`. Optionally,
 220   if you've already computed the global norm for `t_list`, you can specify
 221   the global norm with `use_norm`.
 222 
 223   To perform the clipping, the values `t_list[i]` are set to:
 224 
 225       t_list[i] * clip_norm / max(global_norm, clip_norm)
 226 
 227   where:
 228 
 229       global_norm = sqrt(sum([l2norm(t)**2 for t in t_list]))
 230 
 231   If `clip_norm > global_norm` then the entries in `t_list` remain as they are,
 232   otherwise they're all shrunk by the global ratio.
 233 
 234   Any of the entries of `t_list` that are of type `None` are ignored.
 235 
 236   This is the correct way to perform gradient clipping (for example, see
 237   [Pascanu et al., 2012](http://arxiv.org/abs/1211.5063)
 238   ([pdf](http://arxiv.org/pdf/1211.5063.pdf))).
 239 
 240   However, it is slower than `clip_by_norm()` because all the parameters must be
 241   ready before the clipping operation can be performed.
 242 
 243   Args:
 244     t_list: A tuple or list of mixed `Tensors`, `IndexedSlices`, or None.
 245     clip_norm: A 0-D (scalar) `Tensor` > 0. The clipping ratio.
 246     use_norm: A 0-D (scalar) `Tensor` of type `float` (optional). The global
 247       norm to use. If not provided, `global_norm()` is used to compute the norm.
 248     name: A name for the operation (optional).
 249 
 250   Returns:
 251     list_clipped: A list of `Tensors` of the same type as `list_t`.
 252     global_norm: A 0-D (scalar) `Tensor` representing the global norm.
 253 
 254   Raises:
 255     TypeError: If `t_list` is not a sequence.
 256     InvalidArgumentError: If global norm is not finite.
 257   """
 258   if (not isinstance(t_list, collections.Sequence)
 259       or isinstance(t_list, six.string_types)):
 260     raise TypeError("t_list should be a sequence")
 261   t_list = list(t_list)
 262   if use_norm is None:
 263     use_norm = global_norm(t_list, name)
 264   use_norm = numerics.verify_tensor_all_finite(use_norm,
 265                                                "Found Inf or NaN global norm.")
 266 
 267   with ops.name_scope(name, "clip_by_global_norm",
 268                       t_list + [clip_norm]) as name:
 269     # Calculate L2-norm, clip elements by ratio of clip_norm to L2-norm
 270     scale = clip_norm * math_ops.minimum(
 271         1.0 / use_norm,
 272         constant_op.constant(1.0, dtype=use_norm.dtype) / clip_norm)
 273 
 274     values = [
 275         ops.convert_to_tensor(
 276             t.values if isinstance(t, ops.IndexedSlices) else t,
 277             name="t_%d" % i)
 278         if t is not None else t
 279         for i, t in enumerate(t_list)]
 280 
 281     values_clipped = []
 282     for i, v in enumerate(values):
 283       if v is None:
 284         values_clipped.append(None)
 285       else:
 286         with ops.colocate_with(v):
 287           values_clipped.append(
 288               array_ops.identity(v * scale, name="%s_%d" % (name, i)))
 289 
 290     list_clipped = [
 291         ops.IndexedSlices(c_v, t.indices, t.dense_shape)
 292         if isinstance(t, ops.IndexedSlices)
 293         else c_v
 294         for (c_v, t) in zip(values_clipped, t_list)]
 295 
 296   return list_clipped, use_norm

 299 @tf_export("clip_by_average_norm")
 300 def clip_by_average_norm(t, clip_norm, name=None):
 301   """Clips tensor values to a maximum average L2-norm.
 302 
 303   Given a tensor `t`, and a maximum clip value `clip_norm`, this operation
 304   normalizes `t` so that its average L2-norm is less than or equal to
 305   `clip_norm`. Specifically, if the average L2-norm is already less than or
 306   equal to `clip_norm`, then `t` is not modified. If the average L2-norm is
 307   greater than `clip_norm`, then this operation returns a tensor of the same
 308   type and shape as `t` with its values set to:
 309 
 310   `t * clip_norm / l2norm_avg(t)`
 311 
 312   In this case, the average L2-norm of the output tensor is `clip_norm`.
 313 
 314   This operation is typically used to clip gradients before applying them with
 315   an optimizer.
 316 
 317   Args:
 318     t: A `Tensor`.
 319     clip_norm: A 0-D (scalar) `Tensor` > 0. A maximum clipping value.
 320     name: A name for the operation (optional).
 321 
 322   Returns:
 323     A clipped `Tensor`.
 324   """
 325   with ops.name_scope(name, "clip_by_average_norm", [t, clip_norm]) as name:
 326     t = ops.convert_to_tensor(t, name="t")
 327 
 328     # Calculate L2-norm per element, clip elements by ratio of clip_norm to
 329     # L2-norm per element
 330     n_element = math_ops.cast(array_ops.size(t), dtypes.float32)
 331     l2norm_inv = math_ops.rsqrt(
 332         math_ops.reduce_sum(t * t, math_ops.range(array_ops.rank(t))))
 333     tclip = array_ops.identity(
 334         t * clip_norm * math_ops.minimum(
 335             l2norm_inv * n_element, constant_op.constant(1.0) / clip_norm),
 336         name=name)
 337 
 338   return tclip
