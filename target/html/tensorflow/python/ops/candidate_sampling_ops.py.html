  30 @tf_export('random.uniform_candidate_sampler', 'nn.uniform_candidate_sampler')
  31 @deprecation.deprecated_endpoints('nn.uniform_candidate_sampler')
  32 def uniform_candidate_sampler(true_classes, num_true, num_sampled, unique,
  33                               range_max, seed=None, name=None):
  34   """Samples a set of classes using a uniform base distribution.
  35 
  36   This operation randomly samples a tensor of sampled classes
  37   (`sampled_candidates`) from the range of integers `[0, range_max)`.
  38 
  39   The elements of `sampled_candidates` are drawn without replacement
  40   (if `unique=True`) or with replacement (if `unique=False`) from
  41   the base distribution.
  42 
  43   The base distribution for this operation is the uniform distribution
  44   over the range of integers `[0, range_max)`.
  45 
  46   In addition, this operation returns tensors `true_expected_count`
  47   and `sampled_expected_count` representing the number of times each
  48   of the target classes (`true_classes`) and the sampled
  49   classes (`sampled_candidates`) is expected to occur in an average
  50   tensor of sampled classes.  These values correspond to `Q(y|x)`
  51   defined in [this
  52   document](http://www.tensorflow.org/extras/candidate_sampling.pdf).
  53   If `unique=True`, then these are post-rejection probabilities and we
  54   compute them approximately.
  55 
  56   Args:
  57     true_classes: A `Tensor` of type `int64` and shape `[batch_size,
  58       num_true]`. The target classes.
  59     num_true: An `int`.  The number of target classes per training example.
  60     num_sampled: An `int`.  The number of classes to randomly sample. The
  61       `sampled_candidates` return value will have shape `[num_sampled]`. If
  62       `unique=True`, `num_sampled` must be less than or equal to `range_max`.
  63     unique: A `bool`. Determines whether all sampled classes in a batch are
  64       unique.
  65     range_max: An `int`. The number of possible classes.
  66     seed: An `int`. An operation-specific seed. Default is 0.
  67     name: A name for the operation (optional).
  68 
  69   Returns:
  70     sampled_candidates: A tensor of type `int64` and shape `[num_sampled]`.  The
  71       sampled classes, either with possible duplicates (`unique=False`) or all
  72       unique (`unique=True`). In either case, `sampled_candidates` is
  73       independent of the true classes.
  74     true_expected_count: A tensor of type `float`.  Same shape as
  75       `true_classes`. The expected counts under the sampling distribution
  76       of each of `true_classes`.
  77     sampled_expected_count: A tensor of type `float`. Same shape as
  78       `sampled_candidates`. The expected counts under the sampling distribution
  79       of each of `sampled_candidates`.
  80   """
  81   seed1, seed2 = random_seed.get_seed(seed)
  82   return gen_candidate_sampling_ops.uniform_candidate_sampler(
  83       true_classes, num_true, num_sampled, unique, range_max, seed=seed1,
  84       seed2=seed2, name=name)

  87 @tf_export('random.log_uniform_candidate_sampler',
  88            'nn.log_uniform_candidate_sampler')
  89 @deprecation.deprecated_endpoints('nn.log_uniform_candidate_sampler')
  90 def log_uniform_candidate_sampler(true_classes, num_true, num_sampled, unique,
  91                                   range_max, seed=None, name=None):
  92   """Samples a set of classes using a log-uniform (Zipfian) base distribution.
  93 
  94   This operation randomly samples a tensor of sampled classes
  95   (`sampled_candidates`) from the range of integers `[0, range_max)`.
  96 
  97   The elements of `sampled_candidates` are drawn without replacement
  98   (if `unique=True`) or with replacement (if `unique=False`) from
  99   the base distribution.
 100 
 101   The base distribution for this operation is an approximately log-uniform
 102   or Zipfian distribution:
 103 
 104   `P(class) = (log(class + 2) - log(class + 1)) / log(range_max + 1)`
 105 
 106   This sampler is useful when the target classes approximately follow such
 107   a distribution - for example, if the classes represent words in a lexicon
 108   sorted in decreasing order of frequency. If your classes are not ordered by
 109   decreasing frequency, do not use this op.
 110 
 111   In addition, this operation returns tensors `true_expected_count`
 112   and `sampled_expected_count` representing the number of times each
 113   of the target classes (`true_classes`) and the sampled
 114   classes (`sampled_candidates`) is expected to occur in an average
 115   tensor of sampled classes.  These values correspond to `Q(y|x)`
 116   defined in [this
 117   document](http://www.tensorflow.org/extras/candidate_sampling.pdf).
 118   If `unique=True`, then these are post-rejection probabilities and we
 119   compute them approximately.
 120 
 121   Args:
 122     true_classes: A `Tensor` of type `int64` and shape `[batch_size,
 123       num_true]`. The target classes.
 124     num_true: An `int`.  The number of target classes per training example.
 125     num_sampled: An `int`.  The number of classes to randomly sample.
 126     unique: A `bool`. Determines whether all sampled classes in a batch are
 127       unique.
 128     range_max: An `int`. The number of possible classes.
 129     seed: An `int`. An operation-specific seed. Default is 0.
 130     name: A name for the operation (optional).
 131 
 132   Returns:
 133     sampled_candidates: A tensor of type `int64` and shape `[num_sampled]`.
 134       The sampled classes.
 135     true_expected_count: A tensor of type `float`.  Same shape as
 136       `true_classes`. The expected counts under the sampling distribution
 137       of each of `true_classes`.
 138     sampled_expected_count: A tensor of type `float`. Same shape as
 139       `sampled_candidates`. The expected counts under the sampling distribution
 140       of each of `sampled_candidates`.
 141   """
 142   seed1, seed2 = random_seed.get_seed(seed)
 143   return gen_candidate_sampling_ops.log_uniform_candidate_sampler(
 144       true_classes, num_true, num_sampled, unique, range_max, seed=seed1,
 145       seed2=seed2, name=name)

 148 @tf_export('nn.learned_unigram_candidate_sampler')
 149 def learned_unigram_candidate_sampler(true_classes, num_true, num_sampled,
 150                                       unique, range_max, seed=None, name=None):
 151   """Samples a set of classes from a distribution learned during training.
 152 
 153   This operation randomly samples a tensor of sampled classes
 154   (`sampled_candidates`) from the range of integers `[0, range_max)`.
 155 
 156   The elements of `sampled_candidates` are drawn without replacement
 157   (if `unique=True`) or with replacement (if `unique=False`) from
 158   the base distribution.
 159 
 160   The base distribution for this operation is constructed on the fly
 161   during training.  It is a unigram distribution over the target
 162   classes seen so far during training.  Every integer in `[0, range_max)`
 163   begins with a weight of 1, and is incremented by 1 each time it is
 164   seen as a target class.  The base distribution is not saved to checkpoints,
 165   so it is reset when the model is reloaded.
 166 
 167   In addition, this operation returns tensors `true_expected_count`
 168   and `sampled_expected_count` representing the number of times each
 169   of the target classes (`true_classes`) and the sampled
 170   classes (`sampled_candidates`) is expected to occur in an average
 171   tensor of sampled classes.  These values correspond to `Q(y|x)`
 172   defined in [this
 173   document](http://www.tensorflow.org/extras/candidate_sampling.pdf).
 174   If `unique=True`, then these are post-rejection probabilities and we
 175   compute them approximately.
 176 
 177   Args:
 178     true_classes: A `Tensor` of type `int64` and shape `[batch_size,
 179       num_true]`. The target classes.
 180     num_true: An `int`.  The number of target classes per training example.
 181     num_sampled: An `int`.  The number of classes to randomly sample.
 182     unique: A `bool`. Determines whether all sampled classes in a batch are
 183       unique.
 184     range_max: An `int`. The number of possible classes.
 185     seed: An `int`. An operation-specific seed. Default is 0.
 186     name: A name for the operation (optional).
 187 
 188   Returns:
 189     sampled_candidates: A tensor of type `int64` and shape `[num_sampled]`.
 190       The sampled classes.
 191     true_expected_count: A tensor of type `float`.  Same shape as
 192       `true_classes`. The expected counts under the sampling distribution
 193       of each of `true_classes`.
 194     sampled_expected_count: A tensor of type `float`. Same shape as
 195       `sampled_candidates`. The expected counts under the sampling distribution
 196       of each of `sampled_candidates`.
 197 
 198   """
 199   seed1, seed2 = random_seed.get_seed(seed)
 200   return gen_candidate_sampling_ops.learned_unigram_candidate_sampler(
 201       true_classes, num_true, num_sampled, unique, range_max, seed=seed1,
 202       seed2=seed2, name=name)

 205 @tf_export('nn.fixed_unigram_candidate_sampler')
 206 def fixed_unigram_candidate_sampler(true_classes,
 207                                     num_true,
 208                                     num_sampled,
 209                                     unique,
 210                                     range_max,
 211                                     vocab_file='',
 212                                     distortion=1.0,
 213                                     num_reserved_ids=0,
 214                                     num_shards=1,
 215                                     shard=0,
 216                                     unigrams=(),
 217                                     seed=None,
 218                                     name=None):
 219   """Samples a set of classes using the provided (fixed) base distribution.
 220 
 221   This operation randomly samples a tensor of sampled classes
 222   (`sampled_candidates`) from the range of integers `[0, range_max)`.
 223 
 224   The elements of `sampled_candidates` are drawn without replacement
 225   (if `unique=True`) or with replacement (if `unique=False`) from
 226   the base distribution.
 227 
 228   The base distribution is read from a file or passed in as an
 229   in-memory array. There is also an option to skew the distribution by
 230   applying a distortion power to the weights.
 231 
 232   In addition, this operation returns tensors `true_expected_count`
 233   and `sampled_expected_count` representing the number of times each
 234   of the target classes (`true_classes`) and the sampled
 235   classes (`sampled_candidates`) is expected to occur in an average
 236   tensor of sampled classes.  These values correspond to `Q(y|x)`
 237   defined in [this
 238   document](http://www.tensorflow.org/extras/candidate_sampling.pdf).
 239   If `unique=True`, then these are post-rejection probabilities and we
 240   compute them approximately.
 241 
 242   Args:
 243     true_classes: A `Tensor` of type `int64` and shape `[batch_size,
 244       num_true]`. The target classes.
 245     num_true: An `int`.  The number of target classes per training example.
 246     num_sampled: An `int`.  The number of classes to randomly sample.
 247     unique: A `bool`. Determines whether all sampled classes in a batch are
 248       unique.
 249     range_max: An `int`. The number of possible classes.
 250     vocab_file: Each valid line in this file (which should have a CSV-like
 251       format) corresponds to a valid word ID. IDs are in sequential order,
 252       starting from num_reserved_ids. The last entry in each line is expected
 253       to be a value corresponding to the count or relative probability. Exactly
 254       one of `vocab_file` and `unigrams` needs to be passed to this operation.
 255     distortion: The distortion is used to skew the unigram probability
 256       distribution.  Each weight is first raised to the distortion's power
 257       before adding to the internal unigram distribution. As a result,
 258       `distortion = 1.0` gives regular unigram sampling (as defined by the vocab
 259       file), and `distortion = 0.0` gives a uniform distribution.
 260     num_reserved_ids: Optionally some reserved IDs can be added in the range
 261       `[0, num_reserved_ids)` by the users. One use case is that a special
 262       unknown word token is used as ID 0. These IDs will have a sampling
 263       probability of 0.
 264     num_shards: A sampler can be used to sample from a subset of the original
 265       range in order to speed up the whole computation through parallelism. This
 266       parameter (together with `shard`) indicates the number of partitions that
 267       are being used in the overall computation.
 268     shard: A sampler can be used to sample from a subset of the original range
 269       in order to speed up the whole computation through parallelism. This
 270       parameter (together with `num_shards`) indicates the particular partition
 271       number of the operation, when partitioning is being used.
 272     unigrams: A list of unigram counts or probabilities, one per ID in
 273       sequential order. Exactly one of `vocab_file` and `unigrams` should be
 274       passed to this operation.
 275     seed: An `int`. An operation-specific seed. Default is 0.
 276     name: A name for the operation (optional).
 277 
 278   Returns:
 279     sampled_candidates: A tensor of type `int64` and shape `[num_sampled]`.
 280       The sampled classes.
 281     true_expected_count: A tensor of type `float`.  Same shape as
 282       `true_classes`. The expected counts under the sampling distribution
 283       of each of `true_classes`.
 284     sampled_expected_count: A tensor of type `float`. Same shape as
 285       `sampled_candidates`. The expected counts under the sampling distribution
 286       of each of `sampled_candidates`.
 287 
 288   """
 289   seed1, seed2 = random_seed.get_seed(seed)
 290   return gen_candidate_sampling_ops.fixed_unigram_candidate_sampler(
 291       true_classes, num_true, num_sampled, unique, range_max,
 292       vocab_file=vocab_file, distortion=distortion,
 293       num_reserved_ids=num_reserved_ids, num_shards=num_shards, shard=shard,
 294       unigrams=unigrams, seed=seed1, seed2=seed2, name=name)
 295

 297 @tf_export('nn.all_candidate_sampler')
 298 def all_candidate_sampler(true_classes, num_true, num_sampled, unique,
 299                           seed=None, name=None):
 300   """Generate the set of all classes.
 301 
 302   Deterministically generates and returns the set of all possible classes.
 303   For testing purposes.  There is no need to use this, since you might as
 304   well use full softmax or full logistic regression.
 305 
 306   Args:
 307     true_classes: A `Tensor` of type `int64` and shape `[batch_size,
 308       num_true]`. The target classes.
 309     num_true: An `int`.  The number of target classes per training example.
 310     num_sampled: An `int`.  The number of possible classes.
 311     unique: A `bool`. Ignored.
 312       unique.
 313     seed: An `int`. An operation-specific seed. Default is 0.
 314     name: A name for the operation (optional).
 315 
 316   Returns:
 317     sampled_candidates: A tensor of type `int64` and shape `[num_sampled]`.
 318       This operation deterministically returns the entire range
 319       `[0, num_sampled]`.
 320     true_expected_count: A tensor of type `float`.  Same shape as
 321       `true_classes`. The expected counts under the sampling distribution
 322       of each of `true_classes`. All returned values are 1.0.
 323     sampled_expected_count: A tensor of type `float`. Same shape as
 324       `sampled_candidates`. The expected counts under the sampling distribution
 325       of each of `sampled_candidates`. All returned values are 1.0.
 326   """
 327   seed1, seed2 = random_seed.get_seed(seed)
 328   return gen_candidate_sampling_ops.all_candidate_sampler(
 329       true_classes, num_true, num_sampled, unique, seed=seed1, seed2=seed2,
 330       name=name)

 333 @tf_export('nn.compute_accidental_hits')
 334 def compute_accidental_hits(true_classes, sampled_candidates, num_true,
 335                             seed=None, name=None):
 336   """Compute the position ids in `sampled_candidates` matching `true_classes`.
 337 
 338   In Candidate Sampling, this operation facilitates virtually removing
 339   sampled classes which happen to match target classes.  This is done
 340   in Sampled Softmax and Sampled Logistic.
 341 
 342   See our [Candidate Sampling Algorithms
 343   Reference](http://www.tensorflow.org/extras/candidate_sampling.pdf).
 344 
 345   We presuppose that the `sampled_candidates` are unique.
 346 
 347   We call it an 'accidental hit' when one of the target classes
 348   matches one of the sampled classes.  This operation reports
 349   accidental hits as triples `(index, id, weight)`, where `index`
 350   represents the row number in `true_classes`, `id` represents the
 351   position in `sampled_candidates`, and weight is `-FLOAT_MAX`.
 352 
 353   The result of this op should be passed through a `sparse_to_dense`
 354   operation, then added to the logits of the sampled classes. This
 355   removes the contradictory effect of accidentally sampling the true
 356   target classes as noise classes for the same example.
 357 
 358   Args:
 359     true_classes: A `Tensor` of type `int64` and shape `[batch_size,
 360       num_true]`. The target classes.
 361     sampled_candidates: A tensor of type `int64` and shape `[num_sampled]`.
 362       The sampled_candidates output of CandidateSampler.
 363     num_true: An `int`.  The number of target classes per training example.
 364     seed: An `int`. An operation-specific seed. Default is 0.
 365     name: A name for the operation (optional).
 366 
 367   Returns:
 368     indices: A `Tensor` of type `int32` and shape `[num_accidental_hits]`.
 369       Values indicate rows in `true_classes`.
 370     ids: A `Tensor` of type `int64` and shape `[num_accidental_hits]`.
 371       Values indicate positions in `sampled_candidates`.
 372     weights: A `Tensor` of type `float` and shape `[num_accidental_hits]`.
 373       Each value is `-FLOAT_MAX`.
 374 
 375   """
 376   seed1, seed2 = random_seed.get_seed(seed)
 377   return gen_candidate_sampling_ops.compute_accidental_hits(
 378       true_classes, sampled_candidates, num_true, seed=seed1, seed2=seed2,
 379       name=name)
