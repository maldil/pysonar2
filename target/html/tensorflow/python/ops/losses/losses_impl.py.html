  36 @tf_export("losses.Reduction")
  37 class Reduction(object):
  38   """Types of loss reduction.
  39 
  40   Contains the following values:
  41   `NONE`: Un-reduced weighted losses with the same shape as input.
  42   `SUM`: Scalar sum of weighted losses.
  43   `MEAN`: Scalar `SUM` divided by sum of weights.
  44   `SUM_OVER_BATCH_SIZE`: Scalar `SUM` divided by number of elements in losses.
  45   `SUM_OVER_NONZERO_WEIGHTS`: Scalar `SUM` divided by number of non-zero
  46      weights.
  47   `SUM_BY_NONZERO_WEIGHTS`: Same as `SUM_OVER_NONZERO_WEIGHTS`.
  48   """
  49 
  50   NONE = "none"
  51 
  52   SUM = "weighted_sum"
  53 
  54   MEAN = "weighted_mean"
  55 
  56   SUM_OVER_BATCH_SIZE = "weighted_sum_over_batch_size"
  57 
  58   SUM_BY_NONZERO_WEIGHTS = "weighted_sum_by_nonzero_weights"
  59   SUM_OVER_NONZERO_WEIGHTS = SUM_BY_NONZERO_WEIGHTS
  60 
  61   @classmethod
  62   def all(cls):
  63     return (
  64         cls.NONE,
  65         cls.SUM,
  66         cls.MEAN,
  67         cls.SUM_OVER_BATCH_SIZE,
  68         cls.SUM_OVER_NONZERO_WEIGHTS,
  69         cls.SUM_BY_NONZERO_WEIGHTS)
  70 
  71   @classmethod
  72   def validate(cls, key):
  73     if key not in cls.all():
  74       raise ValueError("Invalid ReductionKey %s." % key)

  77 def _safe_div(numerator, denominator, name="value"):
  78   """Computes a safe divide which returns 0 if the denominator is zero.
  79 
  80   Note that the function contains an additional conditional check that is
  81   necessary for avoiding situations where the loss is zero causing NaNs to
  82   creep into the gradient computation.
  83 
  84   Args:
  85     numerator: An arbitrary `Tensor`.
  86     denominator: `Tensor` whose shape matches `numerator` and whose values are
  87       assumed to be non-negative.
  88     name: An optional name for the returned op.
  89 
  90   Returns:
  91     The element-wise value of the numerator divided by the denominator.
  92   """
  93   return array_ops.where(
  94       math_ops.greater(denominator, 0),
  95       math_ops.div(numerator, array_ops.where(
  96           math_ops.equal(denominator, 0),
  97           array_ops.ones_like(denominator), denominator)),
  98       array_ops.zeros_like(numerator),
  99       name=name)

 102 def _safe_mean(losses, num_present):
 103   """Computes a safe mean of the losses.
 104 
 105   Args:
 106     losses: `Tensor` whose elements contain individual loss measurements.
 107     num_present: The number of measurable elements in `losses`.
 108 
 109   Returns:
 110     A scalar representing the mean of `losses`. If `num_present` is zero,
 111       then zero is returned.
 112   """
 113   total_loss = math_ops.reduce_sum(losses)
 114   return _safe_div(total_loss, num_present)

 117 def _num_present(losses, weights, per_batch=False):
 118   """Computes the number of elements in the loss function induced by `weights`.
 119 
 120   A given weights tensor induces different numbers of usable elements in the
 121   `losses` tensor. The `weights` tensor is broadcast across `losses` for all
 122   possible dimensions. For example, if `losses` is a tensor of dimension
 123   `[4, 5, 6, 3]` and `weights` is a tensor of shape `[4, 5]`, then `weights` is,
 124   in effect, tiled to match the shape of `losses`. Following this effective
 125   tile, the total number of present elements is the number of non-zero weights.
 126 
 127   Args:
 128     losses: `Tensor` of shape `[batch_size, d1, ... dN]`.
 129     weights: `Tensor` of shape `[]`, `[batch_size]` or
 130       `[batch_size, d1, ... dK]`, where K < N.
 131     per_batch: Whether to return the number of elements per batch or as a sum
 132       total.
 133 
 134   Returns:
 135     The number of present (non-zero) elements in the losses tensor. If
 136       `per_batch` is `True`, the value is returned as a tensor of size
 137       `[batch_size]`. Otherwise, a single scalar tensor is returned.
 138   """
 139   if ((isinstance(weights, float) and weights != 0.0) or
 140       (context.executing_eagerly() and weights._rank() == 0  # pylint: disable=protected-access
 141        and not math_ops.equal(weights, 0.0))):
 142     return _num_elements(losses)
 143   with ops.name_scope(None, "num_present", (losses, weights)) as scope:
 144     weights = math_ops.to_float(weights)
 145     present = array_ops.where(
 146         math_ops.equal(weights, 0.0),
 147         array_ops.zeros_like(weights),
 148         array_ops.ones_like(weights))
 149     present = weights_broadcast_ops.broadcast_weights(present, losses)
 150     if per_batch:
 151       return math_ops.reduce_sum(
 152           present,
 153           axis=math_ops.range(1, array_ops.rank(present)),
 154           keepdims=True,
 155           name=scope)
 156     return math_ops.reduce_sum(present, name=scope)
 157

 159 def _num_elements(losses):
 160   """Computes the number of elements in `losses` tensor."""
 161   with ops.name_scope(None, "num_elements", values=[losses]) as scope:
 162     return math_ops.cast(array_ops.size(losses, name=scope), dtype=losses.dtype)

 165 @tf_export("losses.compute_weighted_loss")
 166 def compute_weighted_loss(
 167     losses, weights=1.0, scope=None, loss_collection=ops.GraphKeys.LOSSES,
 168     reduction=Reduction.SUM_BY_NONZERO_WEIGHTS):
 169   """Computes the weighted loss.
 170 
 171   Args:
 172     losses: `Tensor` of shape `[batch_size, d1, ... dN]`.
 173     weights: Optional `Tensor` whose rank is either 0, or the same rank as
 174       `losses`, and must be broadcastable to `losses` (i.e., all dimensions must
 175       be either `1`, or the same as the corresponding `losses` dimension).
 176     scope: the scope for the operations performed in computing the loss.
 177     loss_collection: the loss will be added to these collections.
 178     reduction: Type of reduction to apply to loss.
 179 
 180   Returns:
 181     Weighted loss `Tensor` of the same type as `losses`. If `reduction` is
 182     `NONE`, this has the same shape as `losses`; otherwise, it is scalar.
 183 
 184   Raises:
 185     ValueError: If `weights` is `None` or the shape is not compatible with
 186       `losses`, or if the number of dimensions (rank) of either `losses` or
 187       `weights` is missing.
 188 
 189   Note:
 190     When calculating the gradient of a weighted loss contributions from
 191     both `losses` and `weights` are considered. If your `weights` depend
 192     on some model parameters but you do not want this to affect the loss
 193     gradient, you need to apply `tf.stop_gradient` to `weights` before
 194     passing them to `compute_weighted_loss`.
 195 
 196   @compatibility(eager)
 197   The `loss_collection` argument is ignored when executing eagerly. Consider
 198   holding on to the return value or collecting losses via a `tf.keras.Model`.
 199   @end_compatibility
 200   """
 201   Reduction.validate(reduction)
 202   with ops.name_scope(scope, "weighted_loss", (losses, weights)):
 203     # Save the `reduction` argument for loss normalization when distributing
 204     # to multiple towers.
 205     # TODO(josh11b): Associate it with the returned op for more precision.
 206     ops.get_default_graph()._last_loss_reduction = reduction  # pylint: disable=protected-access
 207 
 208     with ops.control_dependencies((
 209         weights_broadcast_ops.assert_broadcastable(weights, losses),)):
 210       losses = ops.convert_to_tensor(losses)
 211       input_dtype = losses.dtype
 212       losses = math_ops.to_float(losses)
 213       weights = math_ops.to_float(weights)
 214       weighted_losses = math_ops.multiply(losses, weights)
 215       if reduction == Reduction.NONE:
 216         loss = weighted_losses
 217       else:
 218         loss = math_ops.reduce_sum(weighted_losses)
 219         if reduction == Reduction.MEAN:
 220           loss = _safe_mean(
 221               loss,
 222               math_ops.reduce_sum(array_ops.ones_like(losses) * weights))
 223         elif (reduction == Reduction.SUM_BY_NONZERO_WEIGHTS or
 224               reduction == Reduction.SUM_OVER_NONZERO_WEIGHTS):
 225           loss = _safe_mean(loss, _num_present(losses, weights))
 226         elif reduction == Reduction.SUM_OVER_BATCH_SIZE:
 227           loss = _safe_mean(loss, _num_elements(losses))
 228 
 229       # Convert the result back to the input type.
 230       loss = math_ops.cast(loss, input_dtype)
 231       util.add_loss(loss, loss_collection)
 232       return loss

 235 @tf_export("losses.absolute_difference")
 236 def absolute_difference(
 237     labels, predictions, weights=1.0, scope=None,
 238     loss_collection=ops.GraphKeys.LOSSES,
 239     reduction=Reduction.SUM_BY_NONZERO_WEIGHTS):
 240   """Adds an Absolute Difference loss to the training procedure.
 241 
 242   `weights` acts as a coefficient for the loss. If a scalar is provided, then
 243   the loss is simply scaled by the given value. If `weights` is a `Tensor` of
 244   shape `[batch_size]`, then the total loss for each sample of the batch is
 245   rescaled by the corresponding element in the `weights` vector. If the shape of
 246   `weights` matches the shape of `predictions`, then the loss of each
 247   measurable element of `predictions` is scaled by the corresponding value of
 248   `weights`.
 249 
 250   Args:
 251     labels: The ground truth output tensor, same dimensions as 'predictions'.
 252     predictions: The predicted outputs.
 253     weights: Optional `Tensor` whose rank is either 0, or the same rank as
 254       `labels`, and must be broadcastable to `labels` (i.e., all dimensions must
 255       be either `1`, or the same as the corresponding `losses` dimension).
 256     scope: The scope for the operations performed in computing the loss.
 257     loss_collection: collection to which this loss will be added.
 258     reduction: Type of reduction to apply to loss.
 259 
 260   Returns:
 261     Weighted loss float `Tensor`. If `reduction` is `NONE`, this has the same
 262     shape as `labels`; otherwise, it is scalar.
 263 
 264   Raises:
 265     ValueError: If the shape of `predictions` doesn't match that of
 266       `labels` or if the shape of `weights` is invalid or if `labels`
 267       or `predictions` is None.
 268 
 269   @compatibility(eager)
 270   The `loss_collection` argument is ignored when executing eagerly. Consider
 271   holding on to the return value or collecting losses via a `tf.keras.Model`.
 272   @end_compatibility
 273   """
 274   if labels is None:
 275     raise ValueError("labels must not be None.")
 276   if predictions is None:
 277     raise ValueError("predictions must not be None.")
 278   with ops.name_scope(scope, "absolute_difference",
 279                       (predictions, labels, weights)) as scope:
 280     predictions = math_ops.to_float(predictions)
 281     labels = math_ops.to_float(labels)
 282     predictions.get_shape().assert_is_compatible_with(labels.get_shape())
 283     losses = math_ops.abs(math_ops.subtract(predictions, labels))
 284     return compute_weighted_loss(
 285         losses, weights, scope, loss_collection, reduction=reduction)

 288 @tf_export("losses.cosine_distance")
 289 @deprecated_args(None, "dim is deprecated, use axis instead", "dim")
 290 def cosine_distance(
 291     labels, predictions, axis=None, weights=1.0, scope=None,
 292     loss_collection=ops.GraphKeys.LOSSES,
 293     reduction=Reduction.SUM_BY_NONZERO_WEIGHTS,
 294     dim=None):
 295   """Adds a cosine-distance loss to the training procedure.
 296 
 297   Note that the function assumes that `predictions` and `labels` are already
 298   unit-normalized.
 299 
 300   Args:
 301     labels: `Tensor` whose shape matches 'predictions'
 302     predictions: An arbitrary matrix.
 303     axis: The dimension along which the cosine distance is computed.
 304     weights: Optional `Tensor` whose rank is either 0, or the same rank as
 305       `labels`, and must be broadcastable to `labels` (i.e., all dimensions must
 306       be either `1`, or the same as the corresponding `losses` dimension).
 307     scope: The scope for the operations performed in computing the loss.
 308     loss_collection: collection to which this loss will be added.
 309     reduction: Type of reduction to apply to loss.
 310     dim: The old (deprecated) name for `axis`.
 311 
 312   Returns:
 313     Weighted loss float `Tensor`. If `reduction` is `NONE`, this has the same
 314     shape as `labels`; otherwise, it is scalar.
 315 
 316   Raises:
 317     ValueError: If `predictions` shape doesn't match `labels` shape, or
 318       `axis`, `labels`, `predictions` or `weights` is `None`.
 319 
 320   @compatibility(eager)
 321   The `loss_collection` argument is ignored when executing eagerly. Consider
 322   holding on to the return value or collecting losses via a `tf.keras.Model`.
 323   @end_compatibility
 324   """
 325   axis = deprecated_argument_lookup("axis", axis, "dim", dim)
 326   if axis is None:
 327     raise ValueError("You must specify 'axis'.")
 328   if labels is None:
 329     raise ValueError("labels must not be None.")
 330   if predictions is None:
 331     raise ValueError("predictions must not be None.")
 332   with ops.name_scope(scope, "cosine_distance_loss",
 333                       (predictions, labels, weights)) as scope:
 334     predictions = math_ops.to_float(predictions)
 335     labels = math_ops.to_float(labels)
 336     predictions.get_shape().assert_is_compatible_with(labels.get_shape())
 337 
 338     radial_diffs = math_ops.multiply(predictions, labels)
 339     losses = 1 - math_ops.reduce_sum(radial_diffs, axis=(axis,), keepdims=True)
 340     return compute_weighted_loss(
 341         losses, weights, scope, loss_collection, reduction=reduction)

 344 @tf_export("losses.hinge_loss")
 345 def hinge_loss(labels, logits, weights=1.0, scope=None,
 346                loss_collection=ops.GraphKeys.LOSSES,
 347                reduction=Reduction.SUM_BY_NONZERO_WEIGHTS):
 348   """Adds a hinge loss to the training procedure.
 349 
 350   Args:
 351     labels: The ground truth output tensor. Its shape should match the shape of
 352       logits. The values of the tensor are expected to be 0.0 or 1.0. Internally
 353       the {0,1} labels are converted to {-1,1} when calculating the hinge loss.
 354     logits: The logits, a float tensor. Note that logits are assumed to be
 355       unbounded and 0-centered. A value > 0 (resp. < 0) is considered a positive
 356       (resp. negative) binary prediction.
 357     weights: Optional `Tensor` whose rank is either 0, or the same rank as
 358       `labels`, and must be broadcastable to `labels` (i.e., all dimensions must
 359       be either `1`, or the same as the corresponding `losses` dimension).
 360     scope: The scope for the operations performed in computing the loss.
 361     loss_collection: collection to which the loss will be added.
 362     reduction: Type of reduction to apply to loss.
 363 
 364   Returns:
 365     Weighted loss float `Tensor`. If `reduction` is `NONE`, this has the same
 366     shape as `labels`; otherwise, it is scalar.
 367 
 368   Raises:
 369     ValueError: If the shapes of `logits` and `labels` don't match or
 370       if `labels` or `logits` is None.
 371 
 372   @compatibility(eager)
 373   The `loss_collection` argument is ignored when executing eagerly. Consider
 374   holding on to the return value or collecting losses via a `tf.keras.Model`.
 375   @end_compatibility
 376   """
 377   if labels is None:
 378     raise ValueError("labels must not be None.")
 379   if logits is None:
 380     raise ValueError("logits must not be None.")
 381   with ops.name_scope(scope, "hinge_loss", (logits, labels, weights)) as scope:
 382     logits = math_ops.to_float(logits)
 383     labels = math_ops.to_float(labels)
 384     logits.get_shape().assert_is_compatible_with(labels.get_shape())
 385     # We first need to convert binary labels to -1/1 labels (as floats).
 386     all_ones = array_ops.ones_like(labels)
 387     labels = math_ops.subtract(2 * labels, all_ones)
 388     losses = nn_ops.relu(
 389         math_ops.subtract(all_ones, math_ops.multiply(labels, logits)))
 390     return compute_weighted_loss(
 391         losses, weights, scope, loss_collection, reduction=reduction)

 394 @tf_export("losses.huber_loss")
 395 def huber_loss(labels, predictions, weights=1.0, delta=1.0, scope=None,
 396                loss_collection=ops.GraphKeys.LOSSES,
 397                reduction=Reduction.SUM_BY_NONZERO_WEIGHTS):
 398   """Adds a Huber Loss term to the training procedure.
 399 
 400   For each value x in `error=labels-predictions`, the following is calculated:
 401 
 402   ```
 403     0.5 * x^2                  if |x| <= d
 404     0.5 * d^2 + d * (|x| - d)  if |x| > d
 405   ```
 406 
 407   where d is `delta`.
 408 
 409   See: https://en.wikipedia.org/wiki/Huber_loss
 410 
 411   `weights` acts as a coefficient for the loss. If a scalar is provided, then
 412   the loss is simply scaled by the given value. If `weights` is a tensor of size
 413   `[batch_size]`, then the total loss for each sample of the batch is rescaled
 414   by the corresponding element in the `weights` vector. If the shape of
 415   `weights` matches the shape of `predictions`, then the loss of each
 416   measurable element of `predictions` is scaled by the corresponding value of
 417   `weights`.
 418 
 419   Args:
 420     labels: The ground truth output tensor, same dimensions as 'predictions'.
 421     predictions: The predicted outputs.
 422     weights: Optional `Tensor` whose rank is either 0, or the same rank as
 423       `labels`, and must be broadcastable to `labels` (i.e., all dimensions must
 424       be either `1`, or the same as the corresponding `losses` dimension).
 425     delta: `float`, the point where the huber loss function
 426       changes from a quadratic to linear.
 427     scope: The scope for the operations performed in computing the loss.
 428     loss_collection: collection to which the loss will be added.
 429     reduction: Type of reduction to apply to loss.
 430 
 431   Returns:
 432     Weighted loss float `Tensor`. If `reduction` is `NONE`, this has the same
 433     shape as `labels`; otherwise, it is scalar.
 434 
 435   Raises:
 436     ValueError: If the shape of `predictions` doesn't match that of `labels` or
 437       if the shape of `weights` is invalid.  Also if `labels` or
 438      `predictions` is None.
 439 
 440   @compatibility(eager)
 441   The `loss_collection` argument is ignored when executing eagerly. Consider
 442   holding on to the return value or collecting losses via a `tf.keras.Model`.
 443   @end_compatibility
 444   """
 445   if labels is None:
 446     raise ValueError("labels must not be None.")
 447   if predictions is None:
 448     raise ValueError("predictions must not be None.")
 449   with ops.name_scope(scope, "huber_loss",
 450                       (predictions, labels, weights)) as scope:
 451     predictions = math_ops.to_float(predictions)
 452     labels = math_ops.to_float(labels)
 453     predictions.get_shape().assert_is_compatible_with(labels.get_shape())
 454     error = math_ops.subtract(predictions, labels)
 455     abs_error = math_ops.abs(error)
 456     quadratic = math_ops.minimum(abs_error, delta)
 457     # The following expression is the same in value as
 458     # tf.maximum(abs_error - delta, 0), but importantly the gradient for the
 459     # expression when abs_error == delta is 0 (for tf.maximum it would be 1).
 460     # This is necessary to avoid doubling the gradient, since there is already a
 461     # nonzero contribution to the gradient from the quadratic term.
 462     linear = math_ops.subtract(abs_error, quadratic)
 463     losses = math_ops.add(
 464         math_ops.multiply(
 465             ops.convert_to_tensor(0.5, dtype=quadratic.dtype),
 466             math_ops.multiply(quadratic, quadratic)),
 467         math_ops.multiply(delta, linear))
 468     return compute_weighted_loss(
 469         losses, weights, scope, loss_collection, reduction=reduction)

 472 @tf_export("losses.log_loss")
 473 def log_loss(labels, predictions, weights=1.0, epsilon=1e-7, scope=None,
 474              loss_collection=ops.GraphKeys.LOSSES,
 475              reduction=Reduction.SUM_BY_NONZERO_WEIGHTS):
 476   """Adds a Log Loss term to the training procedure.
 477 
 478   `weights` acts as a coefficient for the loss. If a scalar is provided, then
 479   the loss is simply scaled by the given value. If `weights` is a tensor of size
 480   `[batch_size]`, then the total loss for each sample of the batch is rescaled
 481   by the corresponding element in the `weights` vector. If the shape of
 482   `weights` matches the shape of `predictions`, then the loss of each
 483   measurable element of `predictions` is scaled by the corresponding value of
 484   `weights`.
 485 
 486   Args:
 487     labels: The ground truth output tensor, same dimensions as 'predictions'.
 488     predictions: The predicted outputs.
 489     weights: Optional `Tensor` whose rank is either 0, or the same rank as
 490       `labels`, and must be broadcastable to `labels` (i.e., all dimensions must
 491       be either `1`, or the same as the corresponding `losses` dimension).
 492     epsilon: A small increment to add to avoid taking a log of zero.
 493     scope: The scope for the operations performed in computing the loss.
 494     loss_collection: collection to which the loss will be added.
 495     reduction: Type of reduction to apply to loss.
 496 
 497   Returns:
 498     Weighted loss float `Tensor`. If `reduction` is `NONE`, this has the same
 499     shape as `labels`; otherwise, it is scalar.
 500 
 501   Raises:
 502     ValueError: If the shape of `predictions` doesn't match that of `labels` or
 503       if the shape of `weights` is invalid.  Also if `labels` or `predictions`
 504       is None.
 505 
 506   @compatibility(eager)
 507   The `loss_collection` argument is ignored when executing eagerly. Consider
 508   holding on to the return value or collecting losses via a `tf.keras.Model`.
 509   @end_compatibility
 510   """
 511   if labels is None:
 512     raise ValueError("labels must not be None.")
 513   if predictions is None:
 514     raise ValueError("predictions must not be None.")
 515   with ops.name_scope(scope, "log_loss",
 516                       (predictions, labels, weights)) as scope:
 517     predictions = math_ops.to_float(predictions)
 518     labels = math_ops.to_float(labels)
 519     predictions.get_shape().assert_is_compatible_with(labels.get_shape())
 520     losses = -math_ops.multiply(
 521         labels,
 522         math_ops.log(predictions + epsilon)) - math_ops.multiply(
 523             (1 - labels), math_ops.log(1 - predictions + epsilon))
 524     return compute_weighted_loss(
 525         losses, weights, scope, loss_collection, reduction=reduction)

 528 # TODO(b/37208492): Add reduction arg.
 529 @tf_export("losses.mean_pairwise_squared_error")
 530 def mean_pairwise_squared_error(
 531     labels, predictions, weights=1.0, scope=None,
 532     loss_collection=ops.GraphKeys.LOSSES):
 533   """Adds a pairwise-errors-squared loss to the training procedure.
 534 
 535   Unlike `mean_squared_error`, which is a measure of the differences between
 536   corresponding elements of `predictions` and `labels`,
 537   `mean_pairwise_squared_error` is a measure of the differences between pairs of
 538   corresponding elements of `predictions` and `labels`.
 539 
 540   For example, if `labels`=[a, b, c] and `predictions`=[x, y, z], there are
 541   three pairs of differences are summed to compute the loss:
 542     loss = [ ((a-b) - (x-y)).^2 + ((a-c) - (x-z)).^2 + ((b-c) - (y-z)).^2 ] / 3
 543 
 544   Note that since the inputs are of shape `[batch_size, d0, ... dN]`, the
 545   corresponding pairs are computed within each batch sample but not across
 546   samples within a batch. For example, if `predictions` represents a batch of
 547   16 grayscale images of dimension [batch_size, 100, 200], then the set of pairs
 548   is drawn from each image, but not across images.
 549 
 550   `weights` acts as a coefficient for the loss. If a scalar is provided, then
 551   the loss is simply scaled by the given value. If `weights` is a tensor of size
 552   `[batch_size]`, then the total loss for each sample of the batch is rescaled
 553   by the corresponding element in the `weights` vector.
 554 
 555   Args:
 556     labels: The ground truth output tensor, whose shape must match the shape of
 557       `predictions`.
 558     predictions: The predicted outputs, a tensor of size
 559       `[batch_size, d0, .. dN]` where N+1 is the total number of dimensions in
 560       `predictions`.
 561     weights: Coefficients for the loss a scalar, a tensor of shape
 562       `[batch_size]` or a tensor whose shape matches `predictions`.
 563     scope: The scope for the operations performed in computing the loss.
 564     loss_collection: collection to which the loss will be added.
 565 
 566   Returns:
 567     A scalar `Tensor` that returns the weighted loss.
 568 
 569   Raises:
 570     ValueError: If the shape of `predictions` doesn't match that of `labels` or
 571       if the shape of `weights` is invalid.  Also if `labels` or `predictions`
 572       is None.
 573 
 574   @compatibility(eager)
 575   The `loss_collection` argument is ignored when executing eagerly. Consider
 576   holding on to the return value or collecting losses via a `tf.keras.Model`.
 577   @end_compatibility
 578   """
 579   if labels is None:
 580     raise ValueError("labels must not be None.")
 581   if predictions is None:
 582     raise ValueError("predictions must not be None.")
 583   with ops.name_scope(scope, "mean_pairwise_squared_error",
 584                       (predictions, labels, weights)) as scope:
 585     weights = math_ops.to_float(weights)
 586     labels = math_ops.to_float(labels)
 587     with ops.control_dependencies((
 588         weights_broadcast_ops.assert_broadcastable(weights, labels),)):
 589       predictions = math_ops.to_float(predictions)
 590       predictions.get_shape().assert_is_compatible_with(labels.get_shape())
 591 
 592       diffs = math_ops.subtract(predictions, labels)
 593 
 594       reduction_indices = math_ops.range(1, array_ops.rank(diffs))
 595 
 596       sum_squares_diff_per_batch = math_ops.reduce_sum(
 597           math_ops.square(diffs),
 598           reduction_indices=reduction_indices,
 599           keepdims=True)
 600       num_present_per_batch = _num_present(diffs, weights, per_batch=True)
 601 
 602       term1 = 2.0 * _safe_div(sum_squares_diff_per_batch,
 603                               num_present_per_batch - 1)
 604 
 605       sum_diff = math_ops.reduce_sum(
 606           diffs, reduction_indices=reduction_indices, keepdims=True)
 607       term2 = 2.0 * _safe_div(
 608           math_ops.square(sum_diff),
 609           math_ops.multiply(num_present_per_batch, num_present_per_batch - 1))
 610 
 611       weighted_losses = math_ops.multiply(term1 - term2, weights)
 612       loss = math_ops.reduce_sum(weighted_losses)
 613 
 614       mean_loss = array_ops.where(
 615           math_ops.reduce_sum(num_present_per_batch) > 0,
 616           loss,
 617           array_ops.zeros_like(loss),
 618           name="value")
 619       util.add_loss(mean_loss, loss_collection)
 620       return mean_loss

 740 @tf_export("losses.softmax_cross_entropy")
 741 def softmax_cross_entropy(
 742     onehot_labels, logits, weights=1.0, label_smoothing=0, scope=None,
 743     loss_collection=ops.GraphKeys.LOSSES,
 744     reduction=Reduction.SUM_BY_NONZERO_WEIGHTS):
 745   """Creates a cross-entropy loss using tf.nn.softmax_cross_entropy_with_logits_v2.
 746 
 747   `weights` acts as a coefficient for the loss. If a scalar is provided,
 748   then the loss is simply scaled by the given value. If `weights` is a
 749   tensor of shape `[batch_size]`, then the loss weights apply to each
 750   corresponding sample.
 751 
 752   If `label_smoothing` is nonzero, smooth the labels towards 1/num_classes:
 753       new_onehot_labels = onehot_labels * (1 - label_smoothing)
 754                           + label_smoothing / num_classes
 755 
 756   Note that `onehot_labels` and `logits` must have the same shape,
 757   e.g. `[batch_size, num_classes]`. The shape of `weights` must be
 758   broadcastable to loss, whose shape is decided by the shape of `logits`.
 759   In case the shape of `logits` is `[batch_size, num_classes]`, loss is
 760   a `Tensor` of shape `[batch_size]`.
 761 
 762   Args:
 763     onehot_labels: One-hot-encoded labels.
 764     logits: Logits outputs of the network.
 765     weights: Optional `Tensor` that is broadcastable to loss.
 766     label_smoothing: If greater than 0 then smooth the labels.
 767     scope: the scope for the operations performed in computing the loss.
 768     loss_collection: collection to which the loss will be added.
 769     reduction: Type of reduction to apply to loss.
 770 
 771   Returns:
 772     Weighted loss `Tensor` of the same type as `logits`. If `reduction` is
 773     `NONE`, this has shape `[batch_size]`; otherwise, it is scalar.
 774 
 775   Raises:
 776     ValueError: If the shape of `logits` doesn't match that of `onehot_labels`
 777       or if the shape of `weights` is invalid or if `weights` is None.  Also if
 778       `onehot_labels` or `logits` is None.
 779 
 780   @compatibility(eager)
 781   The `loss_collection` argument is ignored when executing eagerly. Consider
 782   holding on to the return value or collecting losses via a `tf.keras.Model`.
 783   @end_compatibility
 784   """
 785   if onehot_labels is None:
 786     raise ValueError("onehot_labels must not be None.")
 787   if logits is None:
 788     raise ValueError("logits must not be None.")
 789   with ops.name_scope(scope, "softmax_cross_entropy_loss",
 790                       (logits, onehot_labels, weights)) as scope:
 791     logits = ops.convert_to_tensor(logits)
 792     onehot_labels = math_ops.cast(onehot_labels, logits.dtype)
 793     logits.get_shape().assert_is_compatible_with(onehot_labels.get_shape())
 794 
 795     if label_smoothing > 0:
 796       num_classes = math_ops.cast(
 797           array_ops.shape(onehot_labels)[1], logits.dtype)
 798       smooth_positives = 1.0 - label_smoothing
 799       smooth_negatives = label_smoothing / num_classes
 800       onehot_labels = onehot_labels * smooth_positives + smooth_negatives
 801 
 802     onehot_labels = array_ops.stop_gradient(
 803         onehot_labels, name="labels_stop_gradient")
 804     losses = nn.softmax_cross_entropy_with_logits_v2(
 805         labels=onehot_labels, logits=logits, name="xentropy")
 806 
 807     return compute_weighted_loss(
 808         losses, weights, scope, loss_collection, reduction=reduction)

 811 # TODO(ptucker): Merge this with similar method in metrics_impl.
 812 def _remove_squeezable_dimensions(
 813     labels, predictions, weights=None, expected_rank_diff=0):
 814   """Internal version of _remove_squeezable_dimensions which handles weights.
 815 
 816   Squeezes `predictions` and `labels` if their ranks differ from expected by
 817   exactly 1.
 818   Squeezes `weights` if its rank is 1 more than the new rank of `predictions`
 819 
 820   This will use static shape if available. Otherwise, it will add graph
 821   operations, which could result in a performance hit.
 822 
 823   Args:
 824     labels: Label values, a `Tensor` whose dimensions match `predictions`.
 825     predictions: Predicted values, a `Tensor` of arbitrary dimensions.
 826     weights: Optional weight `Tensor`. It will be squeezed if it's not scalar,
 827       and its rank is 1 more than the new rank of `labels`.
 828     expected_rank_diff: Expected result of `rank(predictions) - rank(labels)`.
 829 
 830   Returns:
 831     Tuple of `predictions`, `labels` and `weights`, possibly with the last
 832     dimension squeezed.
 833   """
 834   labels, predictions = confusion_matrix.remove_squeezable_dimensions(
 835       labels, predictions, expected_rank_diff=expected_rank_diff)
 836 
 837   if weights is not None:
 838     weights = ops.convert_to_tensor(weights)
 839     labels_rank = labels.get_shape().ndims
 840     weights_shape = weights.get_shape()
 841     weights_rank = weights_shape.ndims
 842 
 843     if (labels_rank is not None) and (weights_rank is not None):
 844       # Use static rank.
 845       rank_diff = weights_rank - labels_rank
 846       if rank_diff == 1:
 847         weights = array_ops.squeeze(weights, [-1])
 848       return labels, predictions, weights
 849 
 850     # Use dynamic rank.
 851     rank_diff = array_ops.rank(weights) - array_ops.rank(labels)
 852     if (weights_rank is None) or (
 853         weights_rank > 0 and weights_shape.dims[-1].is_compatible_with(1)):
 854       weights = control_flow_ops.cond(
 855           math_ops.equal(1, rank_diff),
 856           lambda: array_ops.squeeze(weights, [-1]),
 857           lambda: weights)
 858 
 859   return labels, predictions, weights

 862 @tf_export("losses.sparse_softmax_cross_entropy")
 863 def sparse_softmax_cross_entropy(
 864     labels, logits, weights=1.0, scope=None,
 865     loss_collection=ops.GraphKeys.LOSSES,
 866     reduction=Reduction.SUM_BY_NONZERO_WEIGHTS):
 867   """Cross-entropy loss using `tf.nn.sparse_softmax_cross_entropy_with_logits`.
 868

 870   then the loss is simply scaled by the given value. If `weights` is a
 871   tensor of shape `[batch_size]`, then the loss weights apply to each
 872   corresponding sample.
 873 
 874   Args:
 875     labels: `Tensor` of shape `[d_0, d_1, ..., d_{r-1}]` (where `r` is rank of
 876       `labels` and result) and dtype `int32` or `int64`. Each entry in `labels`
 877       must be an index in `[0, num_classes)`. Other values will raise an
 878       exception when this op is run on CPU, and return `NaN` for corresponding
 879       loss and gradient rows on GPU.
 880     logits: Unscaled log probabilities of shape
 881       `[d_0, d_1, ..., d_{r-1}, num_classes]` and dtype `float16`, `float32` or
 882       `float64`.
 883     weights: Coefficients for the loss. This must be scalar or broadcastable to
 884       `labels` (i.e. same rank and each dimension is either 1 or the same).
 885     scope: the scope for the operations performed in computing the loss.
 886     loss_collection: collection to which the loss will be added.
 887     reduction: Type of reduction to apply to loss.
 888 
 889   Returns:
 890     Weighted loss `Tensor` of the same type as `logits`. If `reduction` is
 891     `NONE`, this has the same shape as `labels`; otherwise, it is scalar.
 892 
 893   Raises:
 894     ValueError: If the shapes of `logits`, `labels`, and `weights` are
 895       incompatible, or if any of them are None.
 896 
 897   @compatibility(eager)
 898   The `loss_collection` argument is ignored when executing eagerly. Consider
 899   holding on to the return value or collecting losses via a `tf.keras.Model`.
 900   @end_compatibility
 901   """
 902   if labels is None:
 903     raise ValueError("labels must not be None.")
 904   if logits is None:
 905     raise ValueError("logits must not be None.")
 906   with ops.name_scope(scope, "sparse_softmax_cross_entropy_loss",
 907                       (logits, labels, weights)) as scope:
 908     # As documented above in Args, labels contain class IDs and logits contains
 909     # 1 probability per class ID, so we expect rank(logits) - rank(labels) == 1;
 910     # therefore, expected_rank_diff=1.

 912         labels, logits, weights, expected_rank_diff=1)
 913     losses = nn.sparse_softmax_cross_entropy_with_logits(labels=labels,

 915                                                          name="xentropy")
 916     return compute_weighted_loss(
 917         losses, weights, scope, loss_collection, reduction=reduction)
