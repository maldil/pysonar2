  44 def _transpose_batch_time(x):
  45   """Transposes the batch and time dimensions of a Tensor.
  46 
  47   If the input tensor has rank < 2 it returns the original tensor. Retains as
  48   much of the static shape information as possible.
  49 
  50   Args:
  51     x: A Tensor.
  52 
  53   Returns:
  54     x transposed along the first two dimensions.
  55   """
  56   x_static_shape = x.get_shape()
  57   if x_static_shape.ndims is not None and x_static_shape.ndims < 2:
  58     return x
  59 
  60   x_rank = array_ops.rank(x)
  61   x_t = array_ops.transpose(
  62       x, array_ops.concat(
  63           ([1, 0], math_ops.range(2, x_rank)), axis=0))
  64   x_t.set_shape(
  65       tensor_shape.TensorShape([
  66           x_static_shape[1].value, x_static_shape[0].value
  67       ]).concatenate(x_static_shape[2:]))
  68   return x_t

  71 def _best_effort_input_batch_size(flat_input):
  72   """Get static input batch size if available, with fallback to the dynamic one.
  73 
  74   Args:
  75     flat_input: An iterable of time major input Tensors of shape
  76       `[max_time, batch_size, ...]`.
  77     All inputs should have compatible batch sizes.
  78 
  79   Returns:
  80     The batch size in Python integer if available, or a scalar Tensor otherwise.
  81 
  82   Raises:
  83     ValueError: if there is any input with an invalid shape.
  84   """
  85   for input_ in flat_input:
  86     shape = input_.shape
  87     if shape.ndims is None:
  88       continue
  89     if shape.ndims < 2:
  90       raise ValueError(
  91           "Expected input tensor %s to have rank at least 2" % input_)
  92     batch_size = shape[1].value
  93     if batch_size is not None:
  94       return batch_size
  95   # Fallback to the dynamic batch size of the first input.
  96   return array_ops.shape(flat_input[0])[1]

  99 def _infer_state_dtype(explicit_dtype, state):
 100   """Infer the dtype of an RNN state.
 101 
 102   Args:
 103     explicit_dtype: explicitly declared dtype or None.
 104     state: RNN's hidden state. Must be a Tensor or a nested iterable containing
 105       Tensors.
 106 
 107   Returns:
 108     dtype: inferred dtype of hidden state.
 109 
 110   Raises:
 111     ValueError: if `state` has heterogeneous dtypes or is empty.
 112   """
 113   if explicit_dtype is not None:
 114     return explicit_dtype
 115   elif nest.is_sequence(state):
 116     inferred_dtypes = [element.dtype for element in nest.flatten(state)]
 117     if not inferred_dtypes:
 118       raise ValueError("Unable to infer dtype from empty state.")
 119     all_same = all([x == inferred_dtypes[0] for x in inferred_dtypes])
 120     if not all_same:
 121       raise ValueError(
 122           "State has tensors of different inferred_dtypes. Unable to infer a "
 123           "single representative dtype.")
 124     return inferred_dtypes[0]
 125   else:
 126     return state.dtype

 136 def _should_cache():
 137   """Returns True if a default caching device should be set, otherwise False."""
 138   if context.executing_eagerly():
 139     return False
 140   # Don't set a caching device when running in a loop, since it is possible that
 141   # train steps could be wrapped in a tf.while_loop. In that scenario caching
 142   # prevents forward computations in loop iterations from re-reading the
 143   # updated weights.
 144   ctxt = ops.get_default_graph()._get_control_flow_context()  # pylint: disable=protected-access
 145   return control_flow_util.GetContainingWhileContext(ctxt) is None

 148 def _is_keras_rnn_cell(rnn_cell):
 149   """Check whether the cell is a Keras RNN cell.
 150 
 151   The Keras RNN cell accept the state as a list even the state is a single
 152   tensor, whereas the TF RNN cell does not wrap single state tensor in list.
 153   This behavior difference should be unified in future version.
 154 
 155   Args:
 156     rnn_cell: An RNN cell instance that either follow the Keras interface or TF
 157       RNN interface.
 158   Returns:
 159     Boolean, whether the cell is an Keras RNN cell.
 160   """
 161   # Cell type check is not strict enough since there are cells created by other
 162   # library like Deepmind that didn't inherit tf.nn.rnn_cell.RNNCell.
 163   # Keras cells never had zero_state method, which was from the original
 164   # interface from TF RNN cell.
 165   return (not isinstance(rnn_cell, rnn_cell_impl.RNNCell)
 166           and isinstance(rnn_cell, base_layer.Layer)
 167           and getattr(rnn_cell, "zero_state", None) is None)

 170 # pylint: disable=unused-argument
 171 def _rnn_step(
 172     time, sequence_length, min_sequence_length, max_sequence_length,
 173     zero_output, state, call_cell, state_size, skip_conditionals=False):
 174   """Calculate one step of a dynamic RNN minibatch.
 175 
 176   Returns an (output, state) pair conditioned on `sequence_length`.
 177   When skip_conditionals=False, the pseudocode is something like:
 178 
 179   if t >= max_sequence_length:
 180     return (zero_output, state)
 181   if t < min_sequence_length:
 182     return call_cell()
 183 
 184   # Selectively output zeros or output, old state or new state depending
 185   # on whether we've finished calculating each row.
 186   new_output, new_state = call_cell()
 187   final_output = np.vstack([
 188     zero_output if time >= sequence_length[r] else new_output_r
 189     for r, new_output_r in enumerate(new_output)
 190   ])
 191   final_state = np.vstack([
 192     state[r] if time >= sequence_length[r] else new_state_r
 193     for r, new_state_r in enumerate(new_state)
 194   ])
 195   return (final_output, final_state)
 196 
 197   Args:
 198     time: int32 `Tensor` scalar.
 199     sequence_length: int32 `Tensor` vector of size [batch_size].
 200     min_sequence_length: int32 `Tensor` scalar, min of sequence_length.
 201     max_sequence_length: int32 `Tensor` scalar, max of sequence_length.
 202     zero_output: `Tensor` vector of shape [output_size].
 203     state: Either a single `Tensor` matrix of shape `[batch_size, state_size]`,
 204       or a list/tuple of such tensors.
 205     call_cell: lambda returning tuple of (new_output, new_state) where
 206       new_output is a `Tensor` matrix of shape `[batch_size, output_size]`.
 207       new_state is a `Tensor` matrix of shape `[batch_size, state_size]`.
 208     state_size: The `cell.state_size` associated with the state.
 209     skip_conditionals: Python bool, whether to skip using the conditional
 210       calculations.  This is useful for `dynamic_rnn`, where the input tensor
 211       matches `max_sequence_length`, and using conditionals just slows
 212       everything down.
 213 
 214   Returns:
 215     A tuple of (`final_output`, `final_state`) as given by the pseudocode above:
 216       final_output is a `Tensor` matrix of shape [batch_size, output_size]
 217       final_state is either a single `Tensor` matrix, or a tuple of such
 218         matrices (matching length and shapes of input `state`).
 219 
 220   Raises:
 221     ValueError: If the cell returns a state tuple whose length does not match
 222       that returned by `state_size`.
 223   """
 224 
 225   # Convert state to a list for ease of use
 226   flat_state = nest.flatten(state)
 227   flat_zero_output = nest.flatten(zero_output)
 228 
 229   # Vector describing which batch entries are finished.
 230   copy_cond = time >= sequence_length
 231 
 232   def _copy_one_through(output, new_output):
 233     # TensorArray and scalar get passed through.
 234     if isinstance(output, tensor_array_ops.TensorArray):
 235       return new_output
 236     if output.shape.ndims == 0:
 237       return new_output
 238     # Otherwise propagate the old or the new value.
 239     with ops.colocate_with(new_output):
 240       return array_ops.where(copy_cond, output, new_output)
 241 
 242   def _copy_some_through(flat_new_output, flat_new_state):
 243     # Use broadcasting select to determine which values should get
 244     # the previous state & zero output, and which values should get
 245     # a calculated state & output.
 246     flat_new_output = [
 247         _copy_one_through(zero_output, new_output)
 248         for zero_output, new_output in zip(flat_zero_output, flat_new_output)]
 249     flat_new_state = [
 250         _copy_one_through(state, new_state)
 251         for state, new_state in zip(flat_state, flat_new_state)]
 252     return flat_new_output + flat_new_state
 253 
 254   def _maybe_copy_some_through():
 255     """Run RNN step.  Pass through either no or some past state."""
 256     new_output, new_state = call_cell()
 257 
 258     nest.assert_same_structure(state, new_state)
 259 
 260     flat_new_state = nest.flatten(new_state)
 261     flat_new_output = nest.flatten(new_output)
 262     return control_flow_ops.cond(
 263         # if t < min_seq_len: calculate and return everything
 264         time < min_sequence_length, lambda: flat_new_output + flat_new_state,
 265         # else copy some of it through
 266         lambda: _copy_some_through(flat_new_output, flat_new_state))
 267 
 268   # TODO(ebrevdo): skipping these conditionals may cause a slowdown,
 269   # but benefits from removing cond() and its gradient.  We should
 270   # profile with and without this switch here.
 271   if skip_conditionals:
 272     # Instead of using conditionals, perform the selective copy at all time
 273     # steps.  This is faster when max_seq_len is equal to the number of unrolls
 274     # (which is typical for dynamic_rnn).
 275     new_output, new_state = call_cell()
 276     nest.assert_same_structure(state, new_state)
 277     new_state = nest.flatten(new_state)
 278     new_output = nest.flatten(new_output)
 279     final_output_and_state = _copy_some_through(new_output, new_state)
 280   else:
 281     empty_update = lambda: flat_zero_output + flat_state
 282     final_output_and_state = control_flow_ops.cond(
 283         # if t >= max_seq_len: copy all state through, output zeros
 284         time >= max_sequence_length, empty_update,
 285         # otherwise calculation is required: copy some or all of it through
 286         _maybe_copy_some_through)
 287 
 288   if len(final_output_and_state) != len(flat_zero_output) + len(flat_state):
 289     raise ValueError("Internal error: state and output were not concatenated "
 290                      "correctly.")
 291   final_output = final_output_and_state[:len(flat_zero_output)]
 292   final_state = final_output_and_state[len(flat_zero_output):]
 293 
 294   for output, flat_output in zip(final_output, flat_zero_output):
 295     output.set_shape(flat_output.get_shape())
 296   for substate, flat_substate in zip(final_state, flat_state):
 297     if not isinstance(substate, tensor_array_ops.TensorArray):
 298       substate.set_shape(flat_substate.get_shape())
 299 
 300   final_output = nest.pack_sequence_as(
 301       structure=zero_output, flat_sequence=final_output)
 302   final_state = nest.pack_sequence_as(
 303       structure=state, flat_sequence=final_state)
 304 
 305   return final_output, final_state

 308 def _reverse_seq(input_seq, lengths):
 309   """Reverse a list of Tensors up to specified lengths.
 310 
 311   Args:
 312     input_seq: Sequence of seq_len tensors of dimension (batch_size, n_features)
 313                or nested tuples of tensors.
 314     lengths:   A `Tensor` of dimension batch_size, containing lengths for each
 315                sequence in the batch. If "None" is specified, simply reverses
 316                the list.
 317 
 318   Returns:
 319     time-reversed sequence
 320   """
 321   if lengths is None:
 322     return list(reversed(input_seq))
 323 
 324   flat_input_seq = tuple(nest.flatten(input_) for input_ in input_seq)
 325 
 326   flat_results = [[] for _ in range(len(input_seq))]
 327   for sequence in zip(*flat_input_seq):
 328     input_shape = tensor_shape.unknown_shape(
 329         ndims=sequence[0].get_shape().ndims)
 330     for input_ in sequence:
 331       input_shape.merge_with(input_.get_shape())
 332       input_.set_shape(input_shape)
 333 
 334     # Join into (time, batch_size, depth)
 335     s_joined = array_ops.stack(sequence)
 336 
 337     # Reverse along dimension 0
 338     s_reversed = array_ops.reverse_sequence(s_joined, lengths, 0, 1)
 339     # Split again into list
 340     result = array_ops.unstack(s_reversed)
 341     for r, flat_result in zip(result, flat_results):
 342       r.set_shape(input_shape)
 343       flat_result.append(r)
 344 
 345   results = [nest.pack_sequence_as(structure=input_, flat_sequence=flat_result)
 346              for input_, flat_result in zip(input_seq, flat_results)]
 347   return results

 350 @tf_export("nn.bidirectional_dynamic_rnn")
 351 def bidirectional_dynamic_rnn(cell_fw, cell_bw, inputs, sequence_length=None,
 352                               initial_state_fw=None, initial_state_bw=None,
 353                               dtype=None, parallel_iterations=None,
 354                               swap_memory=False, time_major=False, scope=None):
 355   """Creates a dynamic version of bidirectional recurrent neural network.
 356 
 357   Takes input and builds independent forward and backward RNNs. The input_size
 358   of forward and backward cell must match. The initial state for both directions
 359   is zero by default (but can be set optionally) and no intermediate states are
 360   ever returned -- the network is fully unrolled for the given (passed in)
 361   length(s) of the sequence(s) or completely unrolled if length(s) is not
 362   given.
 363 
 364   Args:
 365     cell_fw: An instance of RNNCell, to be used for forward direction.
 366     cell_bw: An instance of RNNCell, to be used for backward direction.
 367     inputs: The RNN inputs.
 368       If time_major == False (default), this must be a tensor of shape:
 369         `[batch_size, max_time, ...]`, or a nested tuple of such elements.
 370       If time_major == True, this must be a tensor of shape:
 371         `[max_time, batch_size, ...]`, or a nested tuple of such elements.
 372     sequence_length: (optional) An int32/int64 vector, size `[batch_size]`,
 373       containing the actual lengths for each of the sequences in the batch.
 374       If not provided, all batch entries are assumed to be full sequences; and
 375       time reversal is applied from time `0` to `max_time` for each sequence.
 376     initial_state_fw: (optional) An initial state for the forward RNN.
 377       This must be a tensor of appropriate type and shape
 378       `[batch_size, cell_fw.state_size]`.
 379       If `cell_fw.state_size` is a tuple, this should be a tuple of
 380       tensors having shapes `[batch_size, s] for s in cell_fw.state_size`.
 381     initial_state_bw: (optional) Same as for `initial_state_fw`, but using
 382       the corresponding properties of `cell_bw`.
 383     dtype: (optional) The data type for the initial states and expected output.
 384       Required if initial_states are not provided or RNN states have a
 385       heterogeneous dtype.
 386     parallel_iterations: (Default: 32).  The number of iterations to run in
 387       parallel.  Those operations which do not have any temporal dependency
 388       and can be run in parallel, will be.  This parameter trades off
 389       time for space.  Values >> 1 use more memory but take less time,
 390       while smaller values use less memory but computations take longer.
 391     swap_memory: Transparently swap the tensors produced in forward inference
 392       but needed for back prop from GPU to CPU.  This allows training RNNs
 393       which would typically not fit on a single GPU, with very minimal (or no)
 394       performance penalty.
 395     time_major: The shape format of the `inputs` and `outputs` Tensors.
 396       If true, these `Tensors` must be shaped `[max_time, batch_size, depth]`.
 397       If false, these `Tensors` must be shaped `[batch_size, max_time, depth]`.
 398       Using `time_major = True` is a bit more efficient because it avoids
 399       transposes at the beginning and end of the RNN calculation.  However,
 400       most TensorFlow data is batch-major, so by default this function
 401       accepts input and emits output in batch-major form.
 402     scope: VariableScope for the created subgraph; defaults to
 403       "bidirectional_rnn"
 404 
 405   Returns:
 406     A tuple (outputs, output_states) where:
 407       outputs: A tuple (output_fw, output_bw) containing the forward and
 408         the backward rnn output `Tensor`.
 409         If time_major == False (default),
 410           output_fw will be a `Tensor` shaped:
 411           `[batch_size, max_time, cell_fw.output_size]`
 412           and output_bw will be a `Tensor` shaped:
 413           `[batch_size, max_time, cell_bw.output_size]`.
 414         If time_major == True,
 415           output_fw will be a `Tensor` shaped:
 416           `[max_time, batch_size, cell_fw.output_size]`
 417           and output_bw will be a `Tensor` shaped:
 418           `[max_time, batch_size, cell_bw.output_size]`.
 419         It returns a tuple instead of a single concatenated `Tensor`, unlike
 420         in the `bidirectional_rnn`. If the concatenated one is preferred,
 421         the forward and backward outputs can be concatenated as
 422         `tf.concat(outputs, 2)`.
 423       output_states: A tuple (output_state_fw, output_state_bw) containing
 424         the forward and the backward final states of bidirectional rnn.
 425 
 426   Raises:
 427     TypeError: If `cell_fw` or `cell_bw` is not an instance of `RNNCell`.
 428   """
 429   rnn_cell_impl.assert_like_rnncell("cell_fw", cell_fw)
 430   rnn_cell_impl.assert_like_rnncell("cell_bw", cell_bw)
 431 
 432   with vs.variable_scope(scope or "bidirectional_rnn"):
 433     # Forward direction
 434     with vs.variable_scope("fw") as fw_scope:
 435       output_fw, output_state_fw = dynamic_rnn(
 436           cell=cell_fw, inputs=inputs, sequence_length=sequence_length,
 437           initial_state=initial_state_fw, dtype=dtype,
 438           parallel_iterations=parallel_iterations, swap_memory=swap_memory,
 439           time_major=time_major, scope=fw_scope)
 440 
 441     # Backward direction
 442     if not time_major:
 443       time_axis = 1
 444       batch_axis = 0
 445     else:
 446       time_axis = 0
 447       batch_axis = 1
 448 
 449     def _reverse(input_, seq_lengths, seq_axis, batch_axis):
 450       if seq_lengths is not None:
 451         return array_ops.reverse_sequence(
 452             input=input_, seq_lengths=seq_lengths,
 453             seq_axis=seq_axis, batch_axis=batch_axis)
 454       else:
 455         return array_ops.reverse(input_, axis=[seq_axis])
 456 
 457     with vs.variable_scope("bw") as bw_scope:
 458 
 459       def _map_reverse(inp):
 460         return _reverse(
 461             inp,
 462             seq_lengths=sequence_length,
 463             seq_axis=time_axis,
 464             batch_axis=batch_axis)
 465 
 466       inputs_reverse = nest.map_structure(_map_reverse, inputs)
 467       tmp, output_state_bw = dynamic_rnn(
 468           cell=cell_bw, inputs=inputs_reverse, sequence_length=sequence_length,
 469           initial_state=initial_state_bw, dtype=dtype,
 470           parallel_iterations=parallel_iterations, swap_memory=swap_memory,
 471           time_major=time_major, scope=bw_scope)
 472 
 473   output_bw = _reverse(
 474       tmp, seq_lengths=sequence_length,
 475       seq_axis=time_axis, batch_axis=batch_axis)
 476 
 477   outputs = (output_fw, output_bw)
 478   output_states = (output_state_fw, output_state_bw)
 479 
 480   return (outputs, output_states)

 483 @tf_export("nn.dynamic_rnn")
 484 def dynamic_rnn(cell, inputs, sequence_length=None, initial_state=None,
 485                 dtype=None, parallel_iterations=None, swap_memory=False,
 486                 time_major=False, scope=None):
 487   """Creates a recurrent neural network specified by RNNCell `cell`.
 488 
 489   Performs fully dynamic unrolling of `inputs`.
 490 
 491   Example:
 492 
 493   ```python
 494   # create a BasicRNNCell
 495   rnn_cell = tf.nn.rnn_cell.BasicRNNCell(hidden_size)
 496 
 497   # 'outputs' is a tensor of shape [batch_size, max_time, cell_state_size]
 498 
 499   # defining initial state
 500   initial_state = rnn_cell.zero_state(batch_size, dtype=tf.float32)
 501 
 502   # 'state' is a tensor of shape [batch_size, cell_state_size]
 503   outputs, state = tf.nn.dynamic_rnn(rnn_cell, input_data,
 504                                      initial_state=initial_state,
 505                                      dtype=tf.float32)
 506   ```
 507 
 508   ```python
 509   # create 2 LSTMCells
 510   rnn_layers = [tf.nn.rnn_cell.LSTMCell(size) for size in [128, 256]]
 511 
 512   # create a RNN cell composed sequentially of a number of RNNCells
 513   multi_rnn_cell = tf.nn.rnn_cell.MultiRNNCell(rnn_layers)
 514 
 515   # 'outputs' is a tensor of shape [batch_size, max_time, 256]
 516   # 'state' is a N-tuple where N is the number of LSTMCells containing a
 517   # tf.contrib.rnn.LSTMStateTuple for each cell
 518   outputs, state = tf.nn.dynamic_rnn(cell=multi_rnn_cell,
 519                                      inputs=data,
 520                                      dtype=tf.float32)
 521   ```
 522 
 523 
 524   Args:
 525     cell: An instance of RNNCell.
 526     inputs: The RNN inputs.
 527       If `time_major == False` (default), this must be a `Tensor` of shape:
 528         `[batch_size, max_time, ...]`, or a nested tuple of such
 529         elements.
 530       If `time_major == True`, this must be a `Tensor` of shape:
 531         `[max_time, batch_size, ...]`, or a nested tuple of such
 532         elements.
 533       This may also be a (possibly nested) tuple of Tensors satisfying
 534       this property.  The first two dimensions must match across all the inputs,
 535       but otherwise the ranks and other shape components may differ.
 536       In this case, input to `cell` at each time-step will replicate the
 537       structure of these tuples, except for the time dimension (from which the
 538       time is taken).
 539       The input to `cell` at each time step will be a `Tensor` or (possibly
 540       nested) tuple of Tensors each with dimensions `[batch_size, ...]`.
 541     sequence_length: (optional) An int32/int64 vector sized `[batch_size]`.
 542       Used to copy-through state and zero-out outputs when past a batch
 543       element's sequence length.  So it's more for performance than correctness.
 544     initial_state: (optional) An initial state for the RNN.
 545       If `cell.state_size` is an integer, this must be
 546       a `Tensor` of appropriate type and shape `[batch_size, cell.state_size]`.
 547       If `cell.state_size` is a tuple, this should be a tuple of
 548       tensors having shapes `[batch_size, s] for s in cell.state_size`.
 549     dtype: (optional) The data type for the initial state and expected output.
 550       Required if initial_state is not provided or RNN state has a heterogeneous
 551       dtype.
 552     parallel_iterations: (Default: 32).  The number of iterations to run in
 553       parallel.  Those operations which do not have any temporal dependency
 554       and can be run in parallel, will be.  This parameter trades off
 555       time for space.  Values >> 1 use more memory but take less time,
 556       while smaller values use less memory but computations take longer.
 557     swap_memory: Transparently swap the tensors produced in forward inference
 558       but needed for back prop from GPU to CPU.  This allows training RNNs
 559       which would typically not fit on a single GPU, with very minimal (or no)
 560       performance penalty.
 561     time_major: The shape format of the `inputs` and `outputs` Tensors.
 562       If true, these `Tensors` must be shaped `[max_time, batch_size, depth]`.
 563       If false, these `Tensors` must be shaped `[batch_size, max_time, depth]`.
 564       Using `time_major = True` is a bit more efficient because it avoids
 565       transposes at the beginning and end of the RNN calculation.  However,
 566       most TensorFlow data is batch-major, so by default this function
 567       accepts input and emits output in batch-major form.
 568     scope: VariableScope for the created subgraph; defaults to "rnn".
 569 
 570   Returns:
 571     A pair (outputs, state) where:
 572 
 573     outputs: The RNN output `Tensor`.
 574 
 575       If time_major == False (default), this will be a `Tensor` shaped:
 576         `[batch_size, max_time, cell.output_size]`.
 577 
 578       If time_major == True, this will be a `Tensor` shaped:
 579         `[max_time, batch_size, cell.output_size]`.
 580 
 581       Note, if `cell.output_size` is a (possibly nested) tuple of integers
 582       or `TensorShape` objects, then `outputs` will be a tuple having the
 583       same structure as `cell.output_size`, containing Tensors having shapes
 584       corresponding to the shape data in `cell.output_size`.
 585 
 586     state: The final state.  If `cell.state_size` is an int, this
 587       will be shaped `[batch_size, cell.state_size]`.  If it is a
 588       `TensorShape`, this will be shaped `[batch_size] + cell.state_size`.
 589       If it is a (possibly nested) tuple of ints or `TensorShape`, this will
 590       be a tuple having the corresponding shapes. If cells are `LSTMCells`
 591       `state` will be a tuple containing a `LSTMStateTuple` for each cell.
 592 
 593   Raises:
 594     TypeError: If `cell` is not an instance of RNNCell.
 595     ValueError: If inputs is None or an empty list.
 596   """
 597   rnn_cell_impl.assert_like_rnncell("cell", cell)
 598 
 599   with vs.variable_scope(scope or "rnn") as varscope:
 600     # Create a new scope in which the caching device is either
 601     # determined by the parent scope, or is set to place the cached
 602     # Variable using the same placement as for the rest of the RNN.
 603     if _should_cache():
 604       if varscope.caching_device is None:
 605         varscope.set_caching_device(lambda op: op.device)
 606 
 607     # By default, time_major==False and inputs are batch-major: shaped
 608     #   [batch, time, depth]
 609     # For internal calculations, we transpose to [time, batch, depth]
 610     flat_input = nest.flatten(inputs)
 611 
 612     if not time_major:
 613       # (B,T,D) => (T,B,D)
 614       flat_input = [ops.convert_to_tensor(input_) for input_ in flat_input]
 615       flat_input = tuple(_transpose_batch_time(input_) for input_ in flat_input)
 616 
 617     parallel_iterations = parallel_iterations or 32
 618     if sequence_length is not None:
 619       sequence_length = math_ops.to_int32(sequence_length)
 620       if sequence_length.get_shape().ndims not in (None, 1):
 621         raise ValueError(
 622             "sequence_length must be a vector of length batch_size, "
 623             "but saw shape: %s" % sequence_length.get_shape())
 624       sequence_length = array_ops.identity(  # Just to find it in the graph.
 625           sequence_length, name="sequence_length")
 626 
 627     batch_size = _best_effort_input_batch_size(flat_input)
 628 
 629     if initial_state is not None:
 630       state = initial_state
 631     else:
 632       if not dtype:
 633         raise ValueError("If there is no initial_state, you must give a dtype.")
 634       if getattr(cell, "get_initial_state", None) is not None:
 635         state = cell.get_initial_state(
 636             inputs=None, batch_size=batch_size, dtype=dtype)
 637       else:
 638         state = cell.zero_state(batch_size, dtype)
 639 
 640     def _assert_has_shape(x, shape):
 641       x_shape = array_ops.shape(x)
 642       packed_shape = array_ops.stack(shape)
 643       return control_flow_ops.Assert(
 644           math_ops.reduce_all(math_ops.equal(x_shape, packed_shape)),
 645           ["Expected shape for Tensor %s is " % x.name,
 646            packed_shape, " but saw shape: ", x_shape])
 647 
 648     if not context.executing_eagerly() and sequence_length is not None:
 649       # Perform some shape validation
 650       with ops.control_dependencies(
 651           [_assert_has_shape(sequence_length, [batch_size])]):
 652         sequence_length = array_ops.identity(
 653             sequence_length, name="CheckSeqLen")
 654 
 655     inputs = nest.pack_sequence_as(structure=inputs, flat_sequence=flat_input)
 656 
 657     (outputs, final_state) = _dynamic_rnn_loop(
 658         cell,
 659         inputs,
 660         state,
 661         parallel_iterations=parallel_iterations,
 662         swap_memory=swap_memory,
 663         sequence_length=sequence_length,
 664         dtype=dtype)
 665 
 666     # Outputs of _dynamic_rnn_loop are always shaped [time, batch, depth].
 667     # If we are performing batch-major calculations, transpose output back
 668     # to shape [batch, time, depth]
 669     if not time_major:
 670       # (T,B,D) => (B,T,D)
 671       outputs = nest.map_structure(_transpose_batch_time, outputs)
 672 
 673     return (outputs, final_state)

 676 def _dynamic_rnn_loop(cell,
 677                       inputs,
 678                       initial_state,
 679                       parallel_iterations,
 680                       swap_memory,
 681                       sequence_length=None,
 682                       dtype=None):
 683   """Internal implementation of Dynamic RNN.
 684 
 685   Args:
 686     cell: An instance of RNNCell.
 687     inputs: A `Tensor` of shape [time, batch_size, input_size], or a nested
 688       tuple of such elements.
 689     initial_state: A `Tensor` of shape `[batch_size, state_size]`, or if
 690       `cell.state_size` is a tuple, then this should be a tuple of
 691       tensors having shapes `[batch_size, s] for s in cell.state_size`.
 692     parallel_iterations: Positive Python int.
 693     swap_memory: A Python boolean
 694     sequence_length: (optional) An `int32` `Tensor` of shape [batch_size].
 695     dtype: (optional) Expected dtype of output. If not specified, inferred from
 696       initial_state.
 697 
 698   Returns:
 699     Tuple `(final_outputs, final_state)`.
 700     final_outputs:
 701       A `Tensor` of shape `[time, batch_size, cell.output_size]`.  If
 702       `cell.output_size` is a (possibly nested) tuple of ints or `TensorShape`
 703       objects, then this returns a (possibly nested) tuple of Tensors matching
 704       the corresponding shapes.
 705     final_state:
 706       A `Tensor`, or possibly nested tuple of Tensors, matching in length
 707       and shapes to `initial_state`.
 708 
 709   Raises:
 710     ValueError: If the input depth cannot be inferred via shape inference
 711       from the inputs.
 712     ValueError: If time_step is not the same for all the elements in the
 713       inputs.
 714     ValueError: If batch_size is not the same for all the elements in the
 715       inputs.
 716   """
 717   state = initial_state
 718   assert isinstance(parallel_iterations, int), "parallel_iterations must be int"
 719 
 720   state_size = cell.state_size
 721 
 722   flat_input = nest.flatten(inputs)
 723   flat_output_size = nest.flatten(cell.output_size)
 724 
 725   # Construct an initial output
 726   input_shape = array_ops.shape(flat_input[0])
 727   time_steps = input_shape[0]
 728   batch_size = _best_effort_input_batch_size(flat_input)
 729 
 730   inputs_got_shape = tuple(input_.get_shape().with_rank_at_least(3)
 731                            for input_ in flat_input)
 732 
 733   const_time_steps, const_batch_size = inputs_got_shape[0].as_list()[:2]
 734 
 735   for shape in inputs_got_shape:
 736     if not shape[2:].is_fully_defined():
 737       raise ValueError(
 738           "Input size (depth of inputs) must be accessible via shape inference,"
 739           " but saw value None.")
 740     got_time_steps = shape[0].value
 741     got_batch_size = shape[1].value
 742     if const_time_steps != got_time_steps:
 743       raise ValueError(
 744           "Time steps is not the same for all the elements in the input in a "
 745           "batch.")
 746     if const_batch_size != got_batch_size:
 747       raise ValueError(
 748           "Batch_size is not the same for all the elements in the input.")
 749 
 750   # Prepare dynamic conditional copying of state & output
 751   def _create_zero_arrays(size):
 752     size = _concat(batch_size, size)
 753     return array_ops.zeros(
 754         array_ops.stack(size), _infer_state_dtype(dtype, state))
 755 
 756   flat_zero_output = tuple(_create_zero_arrays(output)
 757                            for output in flat_output_size)
 758   zero_output = nest.pack_sequence_as(structure=cell.output_size,
 759                                       flat_sequence=flat_zero_output)
 760 
 761   if sequence_length is not None:
 762     min_sequence_length = math_ops.reduce_min(sequence_length)
 763     max_sequence_length = math_ops.reduce_max(sequence_length)
 764   else:
 765     max_sequence_length = time_steps
 766 
 767   time = array_ops.constant(0, dtype=dtypes.int32, name="time")
 768 
 769   with ops.name_scope("dynamic_rnn") as scope:
 770     base_name = scope
 771 
 772   def _create_ta(name, element_shape, dtype):
 773     return tensor_array_ops.TensorArray(dtype=dtype,
 774                                         size=time_steps,
 775                                         element_shape=element_shape,
 776                                         tensor_array_name=base_name + name)
 777 
 778   in_graph_mode = not context.executing_eagerly()
 779   if in_graph_mode:
 780     output_ta = tuple(
 781         _create_ta(
 782             "output_%d" % i,
 783             element_shape=(tensor_shape.TensorShape([const_batch_size])
 784                            .concatenate(
 785                                _maybe_tensor_shape_from_tensor(out_size))),
 786             dtype=_infer_state_dtype(dtype, state))
 787         for i, out_size in enumerate(flat_output_size))
 788     input_ta = tuple(
 789         _create_ta(
 790             "input_%d" % i,
 791             element_shape=flat_input_i.shape[1:],
 792             dtype=flat_input_i.dtype)
 793         for i, flat_input_i in enumerate(flat_input))
 794     input_ta = tuple(ta.unstack(input_)
 795                      for ta, input_ in zip(input_ta, flat_input))
 796   else:
 797     output_ta = tuple([0 for _ in range(time_steps.numpy())]
 798                       for i in range(len(flat_output_size)))
 799     input_ta = flat_input
 800 
 801   def _time_step(time, output_ta_t, state):
 802     """Take a time step of the dynamic RNN.
 803 
 804     Args:
 805       time: int32 scalar Tensor.
 806       output_ta_t: List of `TensorArray`s that represent the output.
 807       state: nested tuple of vector tensors that represent the state.
 808 
 809     Returns:
 810       The tuple (time + 1, output_ta_t with updated flow, new_state).
 811     """
 812 
 813     if in_graph_mode:
 814       input_t = tuple(ta.read(time) for ta in input_ta)
 815       # Restore some shape information
 816       for input_, shape in zip(input_t, inputs_got_shape):
 817         input_.set_shape(shape[1:])
 818     else:
 819       input_t = tuple(ta[time.numpy()] for ta in input_ta)
 820 
 821     input_t = nest.pack_sequence_as(structure=inputs, flat_sequence=input_t)
 822     # Keras RNN cells only accept state as list, even if it's a single tensor.
 823     is_keras_rnn_cell = _is_keras_rnn_cell(cell)
 824     if is_keras_rnn_cell and not nest.is_sequence(state):
 825       state = [state]
 826     call_cell = lambda: cell(input_t, state)
 827 
 828     if sequence_length is not None:
 829       (output, new_state) = _rnn_step(
 830           time=time,
 831           sequence_length=sequence_length,
 832           min_sequence_length=min_sequence_length,
 833           max_sequence_length=max_sequence_length,
 834           zero_output=zero_output,
 835           state=state,
 836           call_cell=call_cell,
 837           state_size=state_size,
 838           skip_conditionals=True)
 839     else:
 840       (output, new_state) = call_cell()
 841 
 842     # Keras cells always wrap state as list, even if it's a single tensor.
 843     if is_keras_rnn_cell and len(new_state) == 1:
 844       new_state = new_state[0]
 845     # Pack state if using state tuples
 846     output = nest.flatten(output)
 847 
 848     if in_graph_mode:
 849       output_ta_t = tuple(
 850           ta.write(time, out) for ta, out in zip(output_ta_t, output))
 851     else:
 852       for ta, out in zip(output_ta_t, output):
 853         ta[time.numpy()] = out
 854 
 855     return (time + 1, output_ta_t, new_state)
 856 
 857   if in_graph_mode:
 858     # Make sure that we run at least 1 step, if necessary, to ensure
 859     # the TensorArrays pick up the dynamic shape.
 860     loop_bound = math_ops.minimum(
 861         time_steps, math_ops.maximum(1, max_sequence_length))
 862   else:
 863     # Using max_sequence_length isn't currently supported in the Eager branch.
 864     loop_bound = time_steps
 865 
 866   _, output_final_ta, final_state = control_flow_ops.while_loop(
 867       cond=lambda time, *_: time < loop_bound,
 868       body=_time_step,
 869       loop_vars=(time, output_ta, state),
 870       parallel_iterations=parallel_iterations,
 871       maximum_iterations=time_steps,
 872       swap_memory=swap_memory)
 873 
 874   # Unpack final output if not using output tuples.
 875   if in_graph_mode:
 876     final_outputs = tuple(ta.stack() for ta in output_final_ta)
 877     # Restore some shape information
 878     for output, output_size in zip(final_outputs, flat_output_size):
 879       shape = _concat(
 880           [const_time_steps, const_batch_size], output_size, static=True)
 881       output.set_shape(shape)
 882   else:
 883     final_outputs = output_final_ta
 884 
 885   final_outputs = nest.pack_sequence_as(
 886       structure=cell.output_size, flat_sequence=final_outputs)
 887   if not in_graph_mode:
 888     final_outputs = nest.map_structure_up_to(
 889         cell.output_size, lambda x: array_ops.stack(x, axis=0), final_outputs)
 890 
 891   return (final_outputs, final_state)

 894 @tf_export("nn.raw_rnn")
 895 def raw_rnn(cell, loop_fn,
 896             parallel_iterations=None, swap_memory=False, scope=None):
 897   """Creates an `RNN` specified by RNNCell `cell` and loop function `loop_fn`.
 898 
 899   **NOTE: This method is still in testing, and the API may change.**
 900 
 901   This function is a more primitive version of `dynamic_rnn` that provides
 902   more direct access to the inputs each iteration.  It also provides more
 903   control over when to start and finish reading the sequence, and
 904   what to emit for the output.
 905 
 906   For example, it can be used to implement the dynamic decoder of a seq2seq
 907   model.
 908 
 909   Instead of working with `Tensor` objects, most operations work with
 910   `TensorArray` objects directly.
 911 
 912   The operation of `raw_rnn`, in pseudo-code, is basically the following:
 913 
 914   ```python
 915   time = tf.constant(0, dtype=tf.int32)
 916   (finished, next_input, initial_state, emit_structure, loop_state) = loop_fn(
 917       time=time, cell_output=None, cell_state=None, loop_state=None)
 918   emit_ta = TensorArray(dynamic_size=True, dtype=initial_state.dtype)
 919   state = initial_state
 920   while not all(finished):
 921     (output, cell_state) = cell(next_input, state)
 922     (next_finished, next_input, next_state, emit, loop_state) = loop_fn(
 923         time=time + 1, cell_output=output, cell_state=cell_state,
 924         loop_state=loop_state)
 925     # Emit zeros and copy forward state for minibatch entries that are finished.
 926     state = tf.where(finished, state, next_state)
 927     emit = tf.where(finished, tf.zeros_like(emit_structure), emit)
 928     emit_ta = emit_ta.write(time, emit)
 929     # If any new minibatch entries are marked as finished, mark these.
 930     finished = tf.logical_or(finished, next_finished)
 931     time += 1
 932   return (emit_ta, state, loop_state)
 933   ```
 934 
 935   with the additional properties that output and state may be (possibly nested)
 936   tuples, as determined by `cell.output_size` and `cell.state_size`, and
 937   as a result the final `state` and `emit_ta` may themselves be tuples.
 938 
 939   A simple implementation of `dynamic_rnn` via `raw_rnn` looks like this:
 940 
 941   ```python
 942   inputs = tf.placeholder(shape=(max_time, batch_size, input_depth),
 943                           dtype=tf.float32)
 944   sequence_length = tf.placeholder(shape=(batch_size,), dtype=tf.int32)
 945   inputs_ta = tf.TensorArray(dtype=tf.float32, size=max_time)
 946   inputs_ta = inputs_ta.unstack(inputs)
 947 
 948   cell = tf.contrib.rnn.LSTMCell(num_units)
 949 
 950   def loop_fn(time, cell_output, cell_state, loop_state):
 951     emit_output = cell_output  # == None for time == 0
 952     if cell_output is None:  # time == 0
 953       next_cell_state = cell.zero_state(batch_size, tf.float32)
 954     else:
 955       next_cell_state = cell_state
 956     elements_finished = (time >= sequence_length)
 957     finished = tf.reduce_all(elements_finished)
 958     next_input = tf.cond(
 959         finished,
 960         lambda: tf.zeros([batch_size, input_depth], dtype=tf.float32),
 961         lambda: inputs_ta.read(time))
 962     next_loop_state = None
 963     return (elements_finished, next_input, next_cell_state,
 964             emit_output, next_loop_state)
 965 
 966   outputs_ta, final_state, _ = raw_rnn(cell, loop_fn)
 967   outputs = outputs_ta.stack()
 968   ```
 969 
 970   Args:
 971     cell: An instance of RNNCell.
 972     loop_fn: A callable that takes inputs
 973       `(time, cell_output, cell_state, loop_state)`
 974       and returns the tuple
 975       `(finished, next_input, next_cell_state, emit_output, next_loop_state)`.
 976       Here `time` is an int32 scalar `Tensor`, `cell_output` is a
 977       `Tensor` or (possibly nested) tuple of tensors as determined by
 978       `cell.output_size`, and `cell_state` is a `Tensor`
 979       or (possibly nested) tuple of tensors, as determined by the `loop_fn`
 980       on its first call (and should match `cell.state_size`).
 981       The outputs are: `finished`, a boolean `Tensor` of
 982       shape `[batch_size]`, `next_input`: the next input to feed to `cell`,
 983       `next_cell_state`: the next state to feed to `cell`,
 984       and `emit_output`: the output to store for this iteration.
 985 
 986       Note that `emit_output` should be a `Tensor` or (possibly nested)
 987       tuple of tensors which is aggregated in the `emit_ta` inside the
 988       `while_loop`. For the first call to `loop_fn`, the `emit_output`
 989       corresponds to the `emit_structure` which is then used to determine the
 990       size of the `zero_tensor` for the `emit_ta` (defaults to
 991       `cell.output_size`). For the subsequent calls to the `loop_fn`, the
 992       `emit_output` corresponds to the actual output tensor
 993       that is to be aggregated in the `emit_ta`. The parameter `cell_state`
 994       and output `next_cell_state` may be either a single or (possibly nested)
 995       tuple of tensors.  The parameter `loop_state` and
 996       output `next_loop_state` may be either a single or (possibly nested) tuple
 997       of `Tensor` and `TensorArray` objects.  This last parameter
 998       may be ignored by `loop_fn` and the return value may be `None`.  If it
 999       is not `None`, then the `loop_state` will be propagated through the RNN
1000       loop, for use purely by `loop_fn` to keep track of its own state.
1001       The `next_loop_state` parameter returned may be `None`.
1002 
1003       The first call to `loop_fn` will be `time = 0`, `cell_output = None`,
1004       `cell_state = None`, and `loop_state = None`.  For this call:
1005       The `next_cell_state` value should be the value with which to initialize
1006       the cell's state.  It may be a final state from a previous RNN or it
1007       may be the output of `cell.zero_state()`.  It should be a
1008       (possibly nested) tuple structure of tensors.
1009       If `cell.state_size` is an integer, this must be
1010       a `Tensor` of appropriate type and shape `[batch_size, cell.state_size]`.
1011       If `cell.state_size` is a `TensorShape`, this must be a `Tensor` of
1012       appropriate type and shape `[batch_size] + cell.state_size`.
1013       If `cell.state_size` is a (possibly nested) tuple of ints or
1014       `TensorShape`, this will be a tuple having the corresponding shapes.
1015       The `emit_output` value may be either `None` or a (possibly nested)
1016       tuple structure of tensors, e.g.,
1017       `(tf.zeros(shape_0, dtype=dtype_0), tf.zeros(shape_1, dtype=dtype_1))`.
1018       If this first `emit_output` return value is `None`,
1019       then the `emit_ta` result of `raw_rnn` will have the same structure and
1020       dtypes as `cell.output_size`.  Otherwise `emit_ta` will have the same
1021       structure, shapes (prepended with a `batch_size` dimension), and dtypes
1022       as `emit_output`.  The actual values returned for `emit_output` at this
1023       initializing call are ignored.  Note, this emit structure must be
1024       consistent across all time steps.
1025 
1026     parallel_iterations: (Default: 32).  The number of iterations to run in
1027       parallel.  Those operations which do not have any temporal dependency
1028       and can be run in parallel, will be.  This parameter trades off
1029       time for space.  Values >> 1 use more memory but take less time,
1030       while smaller values use less memory but computations take longer.
1031     swap_memory: Transparently swap the tensors produced in forward inference
1032       but needed for back prop from GPU to CPU.  This allows training RNNs
1033       which would typically not fit on a single GPU, with very minimal (or no)
1034       performance penalty.
1035     scope: VariableScope for the created subgraph; defaults to "rnn".
1036 
1037   Returns:
1038     A tuple `(emit_ta, final_state, final_loop_state)` where:
1039 
1040     `emit_ta`: The RNN output `TensorArray`.
1041        If `loop_fn` returns a (possibly nested) set of Tensors for
1042        `emit_output` during initialization, (inputs `time = 0`,
1043        `cell_output = None`, and `loop_state = None`), then `emit_ta` will
1044        have the same structure, dtypes, and shapes as `emit_output` instead.
1045        If `loop_fn` returns `emit_output = None` during this call,
1046        the structure of `cell.output_size` is used:
1047        If `cell.output_size` is a (possibly nested) tuple of integers
1048        or `TensorShape` objects, then `emit_ta` will be a tuple having the
1049        same structure as `cell.output_size`, containing TensorArrays whose
1050        elements' shapes correspond to the shape data in `cell.output_size`.
1051 
1052     `final_state`: The final cell state.  If `cell.state_size` is an int, this
1053       will be shaped `[batch_size, cell.state_size]`.  If it is a
1054       `TensorShape`, this will be shaped `[batch_size] + cell.state_size`.
1055       If it is a (possibly nested) tuple of ints or `TensorShape`, this will
1056       be a tuple having the corresponding shapes.
1057 
1058     `final_loop_state`: The final loop state as returned by `loop_fn`.
1059 
1060   Raises:
1061     TypeError: If `cell` is not an instance of RNNCell, or `loop_fn` is not
1062       a `callable`.
1063   """
1064   rnn_cell_impl.assert_like_rnncell("cell", cell)
1065 
1066   if not callable(loop_fn):
1067     raise TypeError("loop_fn must be a callable")
1068 
1069   parallel_iterations = parallel_iterations or 32
1070 
1071   # Create a new scope in which the caching device is either
1072   # determined by the parent scope, or is set to place the cached
1073   # Variable using the same placement as for the rest of the RNN.
1074   with vs.variable_scope(scope or "rnn") as varscope:
1075     if _should_cache():
1076       if varscope.caching_device is None:

1144       Args:
1145         time: time scalar.
1146         elements_finished: batch-size vector.
1147         current_input: possibly nested tuple of input tensors.
1148         emit_ta: possibly nested tuple of output TensorArrays.
1149         state: possibly nested tuple of state tensors.
1150         loop_state: possibly nested tuple of loop state tensors.

1152       Returns:
1153         Tuple having the same size as Args but with updated values.
1154       """

1158       nest.assert_same_structure(cell.output_size, next_output)
1159

1162        next_loop_state) = loop_fn(
1163            next_time, next_output, cell_state, loop_state)
1164 
1165       nest.assert_same_structure(state, next_state)
1166       nest.assert_same_structure(current_input, next_input)
1167       nest.assert_same_structure(emit_ta, emit_output)
1168 
1169       # If loop_fn returns None for next_loop_state, just reuse the
1170       # previous one.
1171       loop_state = loop_state if next_loop_state is None else next_loop_state
1172 
1173       def _copy_some_through(current, candidate):
1174         """Copy some tensors through via array_ops.where."""
1175         def copy_fn(cur_i, cand_i):
1176           # TensorArray and scalar get passed through.
1177           if isinstance(cur_i, tensor_array_ops.TensorArray):
1178             return cand_i
1179           if cur_i.shape.ndims == 0:
1180             return cand_i
1181           # Otherwise propagate the old or the new value.
1182           with ops.colocate_with(cand_i):
1183             return array_ops.where(elements_finished, cur_i, cand_i)
1184         return nest.map_structure(copy_fn, current, candidate)
1185 
1186       emit_output = _copy_some_through(zero_emit, emit_output)
1187       next_state = _copy_some_through(state, next_state)
1188 
1189       emit_ta = nest.map_structure(
1190           lambda ta, emit: ta.write(time, emit), emit_ta, emit_output)
1191 
1192       elements_finished = math_ops.logical_or(elements_finished, next_finished)
1193 
1194       return (next_time, elements_finished, next_input,
1195               emit_ta, next_state, loop_state)
1196 
1197     returned = control_flow_ops.while_loop(
1198         condition, body, loop_vars=[
1199             time, elements_finished, next_input,
1200             emit_ta, state, loop_state],
1201         parallel_iterations=parallel_iterations,
1202         swap_memory=swap_memory)
1203 
1204     (emit_ta, final_state, final_loop_state) = returned[-3:]
1205 
1206     if init_loop_state is None:
1207       final_loop_state = None
1208 
1209     return (emit_ta, final_state, final_loop_state)

1212 @tf_export("nn.static_rnn")
1213 def static_rnn(cell,
1214                inputs,
1215                initial_state=None,
1216                dtype=None,
1217                sequence_length=None,
1218                scope=None):
1219   """Creates a recurrent neural network specified by RNNCell `cell`.
1220 
1221   The simplest form of RNN network generated is:
1222 
1223   ```python
1224     state = cell.zero_state(...)
1225     outputs = []
1226     for input_ in inputs:
1227       output, state = cell(input_, state)
1228       outputs.append(output)
1229     return (outputs, state)
1230   ```
1231   However, a few other options are available:
1232 
1233   An initial state can be provided.
1234   If the sequence_length vector is provided, dynamic calculation is performed.
1235   This method of calculation does not compute the RNN steps past the maximum
1236   sequence length of the minibatch (thus saving computational time),
1237   and properly propagates the state at an example's sequence length
1238   to the final state output.
1239 
1240   The dynamic calculation performed is, at time `t` for batch row `b`,
1241 
1242   ```python
1243     (output, state)(b, t) =
1244       (t >= sequence_length(b))
1245         ? (zeros(cell.output_size), states(b, sequence_length(b) - 1))
1246         : cell(input(b, t), state(b, t - 1))
1247   ```
1248 
1249   Args:
1250     cell: An instance of RNNCell.
1251     inputs: A length T list of inputs, each a `Tensor` of shape
1252       `[batch_size, input_size]`, or a nested tuple of such elements.
1253     initial_state: (optional) An initial state for the RNN.
1254       If `cell.state_size` is an integer, this must be
1255       a `Tensor` of appropriate type and shape `[batch_size, cell.state_size]`.
1256       If `cell.state_size` is a tuple, this should be a tuple of
1257       tensors having shapes `[batch_size, s] for s in cell.state_size`.
1258     dtype: (optional) The data type for the initial state and expected output.
1259       Required if initial_state is not provided or RNN state has a heterogeneous
1260       dtype.
1261     sequence_length: Specifies the length of each sequence in inputs.
1262       An int32 or int64 vector (tensor) size `[batch_size]`, values in `[0, T)`.
1263     scope: VariableScope for the created subgraph; defaults to "rnn".
1264 
1265   Returns:
1266     A pair (outputs, state) where:
1267 
1268     - outputs is a length T list of outputs (one for each input), or a nested
1269       tuple of such elements.
1270     - state is the final state
1271 
1272   Raises:
1273     TypeError: If `cell` is not an instance of RNNCell.
1274     ValueError: If `inputs` is `None` or an empty list, or if the input depth
1275       (column size) cannot be inferred from inputs via shape inference.
1276   """
1277   rnn_cell_impl.assert_like_rnncell("cell", cell)
1278   if not nest.is_sequence(inputs):
1279     raise TypeError("inputs must be a sequence")
1280   if not inputs:
1281     raise ValueError("inputs must not be empty")
1282 
1283   outputs = []
1284   # Create a new scope in which the caching device is either
1285   # determined by the parent scope, or is set to place the cached
1286   # Variable using the same placement as for the rest of the RNN.
1287   with vs.variable_scope(scope or "rnn") as varscope:
1288     if _should_cache():
1289       if varscope.caching_device is None:
1290         varscope.set_caching_device(lambda op: op.device)
1291 
1292     # Obtain the first sequence of the input
1293     first_input = inputs
1294     while nest.is_sequence(first_input):
1295       first_input = first_input[0]
1296 
1297     # Temporarily avoid EmbeddingWrapper and seq2seq badness
1298     # TODO(lukaszkaiser): remove EmbeddingWrapper
1299     if first_input.get_shape().ndims != 1:
1300 
1301       input_shape = first_input.get_shape().with_rank_at_least(2)
1302       fixed_batch_size = input_shape[0]
1303 
1304       flat_inputs = nest.flatten(inputs)
1305       for flat_input in flat_inputs:
1306         input_shape = flat_input.get_shape().with_rank_at_least(2)
1307         batch_size, input_size = input_shape[0], input_shape[1:]
1308         fixed_batch_size.merge_with(batch_size)
1309         for i, size in enumerate(input_size):
1310           if size.value is None:
1311             raise ValueError(
1312                 "Input size (dimension %d of inputs) must be accessible via "
1313                 "shape inference, but saw value None." % i)
1314     else:
1315       fixed_batch_size = first_input.get_shape().with_rank_at_least(1)[0]
1316 
1317     if fixed_batch_size.value:
1318       batch_size = fixed_batch_size.value
1319     else:
1320       batch_size = array_ops.shape(first_input)[0]
1321     if initial_state is not None:
1322       state = initial_state
1323     else:
1324       if not dtype:
1325         raise ValueError("If no initial_state is provided, "
1326                          "dtype must be specified")
1327       if getattr(cell, "get_initial_state", None) is not None:
1328         state = cell.get_initial_state(
1329             inputs=None, batch_size=batch_size, dtype=dtype)
1330       else:
1331         state = cell.zero_state(batch_size, dtype)
1332 
1333     if sequence_length is not None:  # Prepare variables
1334       sequence_length = ops.convert_to_tensor(
1335           sequence_length, name="sequence_length")
1336       if sequence_length.get_shape().ndims not in (None, 1):
1337         raise ValueError(
1338             "sequence_length must be a vector of length batch_size")
1339 
1340       def _create_zero_output(output_size):
1341         # convert int to TensorShape if necessary
1342         size = _concat(batch_size, output_size)
1343         output = array_ops.zeros(
1344             array_ops.stack(size), _infer_state_dtype(dtype, state))
1345         shape = _concat(fixed_batch_size.value, output_size, static=True)
1346         output.set_shape(tensor_shape.TensorShape(shape))
1347         return output
1348 
1349       output_size = cell.output_size
1350       flat_output_size = nest.flatten(output_size)
1351       flat_zero_output = tuple(
1352           _create_zero_output(size) for size in flat_output_size)
1353       zero_output = nest.pack_sequence_as(
1354           structure=output_size, flat_sequence=flat_zero_output)
1355 
1356       sequence_length = math_ops.to_int32(sequence_length)
1357       min_sequence_length = math_ops.reduce_min(sequence_length)
1358       max_sequence_length = math_ops.reduce_max(sequence_length)
1359 
1360     # Keras RNN cells only accept state as list, even if it's a single tensor.
1361     is_keras_rnn_cell = _is_keras_rnn_cell(cell)
1362     if is_keras_rnn_cell and not nest.is_sequence(state):
1363       state = [state]
1364     for time, input_ in enumerate(inputs):
1365       if time > 0:
1366         varscope.reuse_variables()
1367       # pylint: disable=cell-var-from-loop
1368       call_cell = lambda: cell(input_, state)
1369       # pylint: enable=cell-var-from-loop
1370       if sequence_length is not None:
1371         (output, state) = _rnn_step(
1372             time=time,
1373             sequence_length=sequence_length,
1374             min_sequence_length=min_sequence_length,
1375             max_sequence_length=max_sequence_length,
1376             zero_output=zero_output,
1377             state=state,
1378             call_cell=call_cell,
1379             state_size=cell.state_size)
1380       else:
1381         (output, state) = call_cell()
1382       outputs.append(output)
1383     # Keras RNN cells only return state as list, even if it's a single tensor.
1384     if is_keras_rnn_cell and len(state) == 1:
1385       state = state[0]
1386 
1387     return (outputs, state)

1483 def static_bidirectional_rnn(cell_fw,
1484                              cell_bw,
1485                              inputs,
1486                              initial_state_fw=None,
1487                              initial_state_bw=None,
1488                              dtype=None,

1491   """Creates a bidirectional recurrent neural network.
1492 
1493   Similar to the unidirectional case above (rnn) but takes input and builds
1494   independent forward and backward RNNs with the final forward and backward
1495   outputs depth-concatenated, such that the output will have the format
1496   [time][batch][cell_fw.output_size + cell_bw.output_size]. The input_size of
1497   forward and backward cell must match. The initial state for both directions
1498   is zero by default (but can be set optionally) and no intermediate states are
1499   ever returned -- the network is fully unrolled for the given (passed in)
1500   length(s) of the sequence(s) or completely unrolled if length(s) is not given.
1501 
1502   Args:
1503     cell_fw: An instance of RNNCell, to be used for forward direction.
1504     cell_bw: An instance of RNNCell, to be used for backward direction.
1505     inputs: A length T list of inputs, each a tensor of shape
1506       [batch_size, input_size], or a nested tuple of such elements.
1507     initial_state_fw: (optional) An initial state for the forward RNN.
1508       This must be a tensor of appropriate type and shape
1509       `[batch_size, cell_fw.state_size]`.
1510       If `cell_fw.state_size` is a tuple, this should be a tuple of
1511       tensors having shapes `[batch_size, s] for s in cell_fw.state_size`.
1512     initial_state_bw: (optional) Same as for `initial_state_fw`, but using
1513       the corresponding properties of `cell_bw`.
1514     dtype: (optional) The data type for the initial state.  Required if
1515       either of the initial states are not provided.
1516     sequence_length: (optional) An int32/int64 vector, size `[batch_size]`,
1517       containing the actual lengths for each of the sequences.
1518     scope: VariableScope for the created subgraph; defaults to
1519       "bidirectional_rnn"
1520 
1521   Returns:
1522     A tuple (outputs, output_state_fw, output_state_bw) where:
1523       outputs is a length `T` list of outputs (one for each input), which
1524         are depth-concatenated forward and backward outputs.
1525       output_state_fw is the final state of the forward rnn.
1526       output_state_bw is the final state of the backward rnn.
1527 
1528   Raises:
1529     TypeError: If `cell_fw` or `cell_bw` is not an instance of `RNNCell`.
1530     ValueError: If inputs is None or an empty list.
1531   """
1532   rnn_cell_impl.assert_like_rnncell("cell_fw", cell_fw)
1533   rnn_cell_impl.assert_like_rnncell("cell_bw", cell_bw)
1534   if not nest.is_sequence(inputs):
1535     raise TypeError("inputs must be a sequence")
1536   if not inputs:
1537     raise ValueError("inputs must not be empty")
1538 
1539   with vs.variable_scope(scope or "bidirectional_rnn"):
1540     # Forward direction
1541     with vs.variable_scope("fw") as fw_scope:
1542       output_fw, output_state_fw = static_rnn(
1543           cell_fw,
1544           inputs,
1545           initial_state_fw,
1546           dtype,
1547           sequence_length,
1548           scope=fw_scope)
1549 
1550     # Backward direction
1551     with vs.variable_scope("bw") as bw_scope:
1552       reversed_inputs = _reverse_seq(inputs, sequence_length)
1553       tmp, output_state_bw = static_rnn(
1554           cell_bw,
1555           reversed_inputs,
1556           initial_state_bw,
1557           dtype,
1558           sequence_length,
1559           scope=bw_scope)
1560 
1561   output_bw = _reverse_seq(tmp, sequence_length)
1562   # Concat each of the forward/backward outputs
1563   flat_output_fw = nest.flatten(output_fw)
1564   flat_output_bw = nest.flatten(output_bw)
1565 
1566   flat_outputs = tuple(
1567       array_ops.concat([fw, bw], 1)
1568       for fw, bw in zip(flat_output_fw, flat_output_bw))
1569 
1570   outputs = nest.pack_sequence_as(
1571       structure=output_fw, flat_sequence=flat_outputs)
1572 
1573   return (outputs, output_state_fw, output_state_bw)
