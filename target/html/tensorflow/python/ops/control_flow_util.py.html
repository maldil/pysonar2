  31 def IsInXLAContext(op):
  32   try:
  33     xla_compile = op.get_attr("_XlaCompile")
  34     if xla_compile: return True
  35   except ValueError:
  36     pass
  37   ctxt = op._get_control_flow_context()  # pylint: disable=protected-access
  38   return GetContainingXLAContext(ctxt) is not None

  43   return GetContainingWhileContext(ctxt) is not None

  51 def IsSwitch(op):
  52   """Return true if `op` is a Switch."""
  53   return op.type == "Switch" or op.type == "RefSwitch"
  54

  56 def IsMerge(op):
  57   """Return true if `op` is a Merge."""
  58   return op.type == "Merge" or op.type == "RefMerge"

  62   """Returns true if `op` is an Enter."""
  63   return op.type == "Enter" or op.type == "RefEnter"

  67   """Return true if `op` is an Exit."""

  71 def IsCondSwitch(op):
  72   """Return true if `op` is the Switch for a conditional."""
  73   if not IsSwitch(op):
  74     return False
  75   if not op.outputs:
  76     return False
  77   # Switch nodes are not part of the cond control flow context that they
  78   # represent, so consider the consumers of its outputs to determine if it is
  79   # cond switch or not. A switch is a cond switch iff all its consumers are in
  80   # cond contexts.
  81   is_cond_switch = True
  82   for o in op.outputs:
  83     for c in o.consumers():
  84       ctxt = c._get_control_flow_context()  # pylint: disable=protected-access
  85       if IsLoopEnter(c):
  86         ctxt = ctxt.outer_context
  87       is_cond_switch = is_cond_switch and (ctxt is not None and
  88                                            ctxt.IsCondContext())
  89   return is_cond_switch

  92 def IsCondMerge(op):
  93   """Return true if `op` is the Merge for a conditional."""
  94   if not IsMerge(op):
  95     return False
  96   if not op.inputs:
  97     return False
  98   # Merge nodes are not part of the cond control flow context that they
  99   # represent, so consider the inputs to the merge of to determine if it is
 100   # cond merge or not: A merge is a cond merge iff all its inputs are in
 101   # cond contexts.
 102   is_cond_merge = True
 103   for i in op.inputs:
 104     ctxt = GetOutputContext(i.op)
 105     is_cond_merge = is_cond_merge and ctxt is not None and ctxt.IsCondContext()
 106   return is_cond_merge

 109 def IsLoopSwitch(op):
 110   """Return true if `op` is the Switch for a while loop."""
 111   if IsSwitch(op):
 112     ctxt = op._get_control_flow_context()  # pylint: disable=protected-access
 113     return ctxt is not None and ctxt.IsWhileContext() and not IsCondSwitch(op)
 114   return False

 117 def IsLoopMerge(op):
 118   """Return true if `op` is the Merge for a while loop."""
 119   if IsMerge(op):
 120     ctxt = op._get_control_flow_context()  # pylint: disable=protected-access
 121     return ctxt is not None and ctxt.IsWhileContext() and not IsCondMerge(op)
 122   return False

 125 def IsLoopConstantEnter(op):
 126   """Return true iff op is a loop invariant."""
 127   return IsLoopEnter(op) and op.get_attr("is_constant")

 130 def GetLoopConstantEnter(value):
 131   """Return the enter op if we can infer `value` to be a loop invariant."""
 132   id_ops = {"Switch", "RefSwitch", "Identity", "RefIdentity"}
 133   op = value.op
 134   while op.type in id_ops:
 135     op = op.inputs[0].op
 136   return op if IsLoopConstantEnter(op) else None

 139 def GetOutputContext(op):
 140   """Return the control flow context for the output of an op."""
 141   ctxt = op._get_control_flow_context()  # pylint: disable=protected-access
 142   # Exit nodes usually have a control flow context, except in the case where the
 143   # exit node was imported via import_graph_def (in which case no nodes have
 144   # control flow contexts).
 145   if ctxt is not None and IsLoopExit(op):
 146     ctxt = ctxt.outer_context
 147   return ctxt

 150 def GetContainingWhileContext(ctxt, stop_ctxt=None):
 151   """Returns the first ancestor WhileContext of `ctxt`.
 152 
 153   Returns `ctxt` if `ctxt` is a WhileContext, or None if `ctxt` is not in a
 154   while loop.
 155 
 156   Args:
 157     ctxt: ControlFlowContext
 158     stop_ctxt: ControlFlowContext, optional. If provided, the search will end
 159       if it sees stop_ctxt.
 160 
 161   Returns:
 162     `ctxt` if `ctxt` is a WhileContext, the most nested WhileContext containing
 163     `ctxt`, or None if `ctxt` is not in a while loop.  If `stop_ctxt` is not
 164     `None`, this returns `ctxt` if it matches `stop_ctxt` in its traversal.
 165   """
 166   while ctxt:
 167     if ctxt.IsWhileContext() or ctxt == stop_ctxt: return ctxt
 168     ctxt = ctxt.outer_context
 169   return None

 172 def GetContainingXLAContext(ctxt):
 173   """Returns the first ancestor XLAContext of `ctxt`.
 174 
 175   Returns `ctxt` if `ctxt` is a XLAContext, or None if `ctxt` is not in a
 176   while loop.
 177 
 178   Args:
 179     ctxt: ControlFlowContext
 180 
 181   Returns:
 182     `ctxt` if `ctxt` is a XLAContext, the most nested XLAContext containing
 183     `ctxt`, or None if `ctxt` is not in a while loop.
 184   """
 185   while ctxt:
 186     if ctxt.IsXLAContext(): return ctxt
 187     ctxt = ctxt.outer_context
 188   return None

 191 def GetContainingCondContext(ctxt):
 192   """Returns the first ancestor CondContext of `ctxt`.
 193 
 194   Returns `ctxt` if `ctxt` is a CondContext, or None if `ctxt` is not in a cond.
 195 
 196   Args:
 197     ctxt: ControlFlowContext
 198 
 199   Returns:
 200     `ctxt` if `ctxt` is a CondContext, the most nested CondContext containing
 201     `ctxt`, or None if `ctxt` is not in a cond.
 202   """
 203   while ctxt:
 204     if ctxt.IsCondContext(): return ctxt
 205     ctxt = ctxt.outer_context
 206   return None

 209 def IsContainingContext(ctxt, maybe_containing_ctxt):
 210   """Returns true if `maybe_containing_ctxt` is or contains `ctxt`."""
 211   while ctxt is not maybe_containing_ctxt:
 212     if ctxt is None: return False
 213     ctxt = ctxt.outer_context
 214   return True

 217 def OpInContext(op, ctxt):
 218   return IsContainingContext(op._get_control_flow_context(), ctxt)  # pylint: disable=protected-access
 219 
 220 
 221 def TensorInContext(tensor, ctxt):
 222   return OpInContext(tensor.op, ctxt)

 225 def CheckInputFromValidContext(op, input_op):
 226   """Returns whether `input_op` can be used from `op`s context.
 227 
 228   Conceptually, only inputs from op's while context or any ancestor while
 229   context (including outside of any context) are valid. In practice, there are
 230   many other edge cases as well.
 231 
 232   Args:
 233     op: Operation
 234     input_op: Operation
 235 
 236   Raises:
 237     ValueError: if input_op is from an invalid context.
 238   """
 239   op_ctxt = op._get_control_flow_context()  # pylint: disable=protected-access
 240   input_ctxt = GetOutputContext(input_op)
 241   valid = False
 242 
 243   if not input_ctxt:
 244     # input_op isn't in a control flow context.
 245     valid = True
 246   elif op_ctxt is input_ctxt:
 247     # input_op is in the same context as op.
 248     valid = True
 249   else:
 250     while_ctxt = GetContainingWhileContext(op_ctxt)
 251     input_while_ctxt = GetContainingWhileContext(input_ctxt)
 252 
 253     if while_ctxt is None:
 254       if input_while_ctxt is None:
 255         # Neither op nor input_op is in a while loop, but one or both are in
 256         # conds. We allow this, although execution will fail if the branch
 257         # corresponding to input_op's cond context isn't taken.
 258         valid = True
 259       # Invalid if op isn't in a while loop and input_op is. Unless...
 260       if IsLoopEnter(op):
 261         # WhileContext._BuildLoop clears context for Enter nodes.
 262         valid = True
 263       if IsSwitch(op):
 264         # CondContext.AddValue clears context for Switch nodes.
 265         valid = True
 266     elif IsContainingContext(while_ctxt, input_while_ctxt):
 267       # input_op is in a while loop which contains op's while loop (or not in a
 268       # while loop at all).
 269       valid = True
 270     elif (while_ctxt.grad_state and
 271           IsContainingContext(while_ctxt.grad_state.forward_context,
 272                               input_while_ctxt)):
 273       # op is in a gradient context and input_op is in the associated forward
 274       # pass context or an ancestor thereof. This case is need to build while
 275       # loop gradients.
 276       # NOTE(skyewm): we theoretically also need this case for custom gradient
 277       # functions that close over tensors from ancestor contexts, but I haven't
 278       # verified this.
 279       valid = True
 280     elif (while_ctxt.grad_state and
 281           while_ctxt.grad_state.forward_context is
 282           input_while_ctxt._outer_context):  # pylint: disable=protected-access
 283       # op is in a gradient context and input_op is in a child of the associated
 284       # forward pass context. This case is needed for the gradients of while
 285       # loops with conds.
 286       valid = True
 287     elif (input_while_ctxt.grad_state and
 288           input_while_ctxt.grad_state.forward_context is while_ctxt):
 289       # input_op is in the gradient context of op's context. This case is needed
 290       # when the gradient of a while loop gradient is requested (this will
 291       # eventually fail unless there is a stop_gradient() or similar).
 292       valid = True
 293     elif (input_while_ctxt.grad_state and
 294           input_ctxt.grad_state.forward_context.grad_state and
 295           input_ctxt.grad_state.forward_context.grad_state.forward_context is
 296           while_ctxt):
 297       # input_op is in the grad grad context of op's context. This case is
 298       # needed when the gradient of a while loop gradient is requested (this
 299       # will eventually fail unless there is a stop_gradient() or similar).
 300       valid = True
 301 
 302   if not valid:
 303     if while_ctxt:
 304       error_msg = (
 305           "Cannot use '%s' as input to '%s' because they are in different while"
 306           " loops." % (op.name, input_op.name))
 307     else:
 308       error_msg = (
 309           "Cannot use '%s' as input to '%s' because '%s' is in a while loop."
 310           % (input_op.name, op.name, input_op.name))
 311 
 312     # Log the error message plus the relevant stack traces. The stacks may be
 313     # useful for debugging this error, but we don't want to raise an
 314     # unreadable exception.
 315     log_msg = error_msg
 316     log_msg += "\n\n%s while context: %s" % (op.name, while_ctxt)
 317     log_msg += "\n%s while context: %s" % (input_op.name, input_while_ctxt)
 318     log_msg += "\n\nTraceback for %s:\n%s\nTraceback for %s:\n%s\n" % (
 319         op.name, "".join(traceback.format_list(op.traceback)),
 320         input_op.name, "".join(traceback.format_list(input_op.traceback)))
 321     logging.info(log_msg)
 322     raise ValueError(error_msg + " See info log for more details.")
