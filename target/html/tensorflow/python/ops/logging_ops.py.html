 108 def _generate_placeholder_string(x, default_placeholder="{}"):
 109   """Generate and return a string that does not appear in `x`."""
 110   placeholder = default_placeholder
 111   rng = random.Random(5)
 112   while placeholder in x:
 113     placeholder = placeholder + str(rng.randint(0, 9))
 114   return placeholder

 123 @tf_export("print")
 124 def print_v2(*inputs, **kwargs):
 125   """Print the specified inputs.
 126 
 127   Returns an operator that prints the specified inputs to a desired
 128   output stream or logging level. The inputs may be dense or sparse Tensors,
 129   primitive python objects, data structures that contain Tensors, and printable
 130   python objects. Printed tensors will recursively show the first and last
 131   `summarize` elements of each dimension.
 132 
 133   With eager execution enabled and/or inside a `tf.contrib.eager.defun` this
 134   operator will automatically execute, and users only need to call `tf.print`
 135   without using the return value. When constructing graphs outside of a
 136   `tf.contrib.eager.defun`, one must either include the returned op
 137   in the input to `session.run`, or use the operator as a control dependency for
 138   executed ops by specifying `with tf.control_dependencies([print_op])`.
 139 
 140   @compatibility(python2)
 141   In python 2.7, make sure to import the following:
 142   `from __future__ import print_function`
 143   @end_compatibility
 144 
 145   Example:
 146     Single-input usage:
 147     ```python
 148     tf.enable_eager_execution()
 149     tensor = tf.range(10)
 150     tf.print(tensor, output_stream=sys.stderr)
 151     ```
 152     (This prints "[0 1 2 ... 7 8 9]" to sys.stderr)
 153 
 154     Multi-input usage:
 155     ```python
 156     tf.enable_eager_execution()
 157     tensor = tf.range(10)
 158     tf.print("tensors:", tensor, {2: tensor * 2}, output_stream=sys.stdout)
 159     ```
 160     (This prints "tensors: [0 1 2 ... 7 8 9] {2: [0 2 4 ... 14 16 18]}" to
 161     sys.stdout)
 162 
 163     Usage in a defun:
 164     ```python
 165     tf.enable_eager_execution()
 166 
 167     @tf.contrib.eager.defun
 168     def f():
 169         tensor = tf.range(10)
 170         tf.print(tensor, output_stream=sys.stderr)
 171         return tensor
 172 
 173     range_tensor = f()
 174     ```
 175     (This prints "[0 1 2 ... 7 8 9]" to sys.stderr)
 176 
 177     Usage when constructing graphs:
 178     ```python
 179     sess = tf.Session()
 180     with sess.as_default():
 181         tensor = tf.range(10)
 182         print_op = tf.print("tensors:", tensor, {2: tensor * 2},
 183                             output_stream=sys.stdout)
 184         with tf.control_dependencies([print_op]):
 185           tripled_tensor = tensor * 3
 186         sess.run(tripled_tensor)
 187     ```
 188     (This prints "tensors: [0 1 2 ... 7 8 9] {2: [0 2 4 ... 14 16 18]}" to
 189     sys.stdout)
 190 
 191   Note: This op is only partially compatible with Jupyter notebooks and colabs.
 192     Because it prints to the C++ standard out / standard error, this will go
 193     in the notebook kernel's console output, not in the notebook cell output.
 194 
 195   Args:
 196     *inputs: Positional arguments that are the inputs to print. Inputs in the
 197       printed output will be separated by spaces. Inputs may be python
 198       primitives, tensors, data structures such as dicts and lists that
 199       may contain tensors (with the data structures possibly nested in
 200       arbitrary ways), and printable python objects.
 201     output_stream: The output stream or logging level to print to. Defaults to
 202       sys.stderr, but sys.stdout, tf.logging.info, tf.logging.warning, and
 203       tf.logging.error are also supported.
 204     summarize: The first and last `summarize` elements within each dimension are
 205       recursively printed per Tensor. If None, then the first 3 and last 3
 206       elements of each dimension are printed for each tensor. If set to -1, it
 207       will print all elements of every tensor.
 208     name: A name for the operation (optional).
 209 
 210   Returns:
 211     A print operator that prints the specified inputs in the specified output
 212     stream or logging level.
 213 
 214   Raises:
 215     ValueError: If an unsupported output stream is specified.
 216   """
 217   # Because we are using arbitrary-length positional arguments, python 2
 218   # does not support explicitly specifying the keyword arguments in the
 219   # function definition. So, we manually get the keyword arguments w/ default
 220   # values here.
 221   output_stream = kwargs.pop("output_stream", sys.stderr)
 222   name = kwargs.pop("name", None)
 223   summarize = kwargs.pop("summarize", 3)
 224   if kwargs:
 225     raise ValueError("Unrecognized keyword arguments for tf.print: %s" % kwargs)
 226   format_name = None
 227   if name:
 228     format_name = name + "_format"
 229 
 230   # Match the C++ string constants representing the different output streams.
 231   # Keep this updated!
 232   output_stream_to_constant = {
 233       sys.stdout: "stdout",
 234       sys.stderr: "stderr",
 235       tf_logging.INFO: "log(info)",
 236       tf_logging.info: "log(info)",
 237       tf_logging.WARN: "log(warning)",
 238       tf_logging.warning: "log(warning)",
 239       tf_logging.warn: "log(warning)",
 240       tf_logging.ERROR: "log(error)",
 241       tf_logging.error: "log(error)",
 242   }
 243 
 244   output_stream_string = output_stream_to_constant.get(output_stream)
 245   if not output_stream_string:
 246     raise ValueError(
 247         "Unsupported output stream or logging level " +
 248         str(output_stream) + ". Supported streams are sys.stdout, "
 249                              "sys.stderr, tf.logging.info, "
 250                              "tf.logging.warning, tf.logging.error")
 251 
 252   # If we are only printing a single string scalar, there is no need to format
 253   if (len(inputs) == 1 and tensor_util.is_tensor(inputs[0])
 254       and (not isinstance(inputs[0], sparse_tensor.SparseTensor))
 255       and inputs[0].shape and (inputs[0].dtype == dtypes.string)):
 256     formatted_string = inputs[0]
 257   # Otherwise, we construct an appropriate template for the tensors we are
 258   # printing, and format the template using those tensors.
 259   else:
 260     # For each input to this print function, we extract any nested tensors,
 261     # and construct an appropriate template to format representing the
 262     # printed input.
 263     templates = []
 264     tensors = []
 265     tensor_free_structure = nest.map_structure(
 266         lambda x: "" if tensor_util.is_tensor(x) else x,
 267         inputs)
 268     tensor_free_template = " ".join(pprint.pformat(x)
 269                                     for x in tensor_free_structure)
 270     placeholder = _generate_placeholder_string(tensor_free_template)
 271 
 272     for input_ in inputs:
 273       placeholders = []
 274       # Use the nest utilities to flatten & process any nested elements in this
 275       # input. The placeholder for a tensor in the template should be the
 276       # placeholder string, and the placeholder for a non-tensor can just be
 277       # the printed value of the non-tensor itself.
 278       for x in nest.flatten(input_):
 279         # support sparse tensors
 280         if isinstance(x, sparse_tensor.SparseTensor):
 281           tensors.extend([x.indices, x.values, x.dense_shape])
 282           placeholders.append(
 283               "SparseTensor(indices={}, values={}, shape={})".format(
 284                   placeholder, placeholder, placeholder)
 285           )
 286         elif tensor_util.is_tensor(x):
 287           tensors.append(x)
 288           placeholders.append(placeholder)
 289         else:
 290           placeholders.append(x)
 291 
 292       if isinstance(input_, six.string_types):
 293         # If the current input to format/print is a normal string, that string
 294         # can act as the template.
 295         cur_template = input_
 296       else:
 297         # We pack the placeholders into a data structure that matches the
 298         # input data structure format, then format that data structure
 299         # into a string template.
 300         #
 301         # NOTE: We must use pprint.pformat here for building the template for
 302         # unordered data structures such as `dict`, because `str` doesn't
 303         # guarantee orderings, while pprint prints in sorted order. pprint
 304         # will match the ordering of `nest.flatten`.
 305         # This even works when nest.flatten reorders OrderedDicts, because
 306         # pprint is printing *after* the OrderedDicts have been reordered.
 307         cur_template = pprint.pformat(
 308             nest.pack_sequence_as(input_, placeholders))
 309       templates.append(cur_template)
 310 
 311     # We join the templates for the various inputs into a single larger
 312     # template. We also remove all quotes surrounding the placeholders, so that
 313     # the formatted/printed output will not contain quotes around tensors.
 314     # (example of where these quotes might appear: if we have added a
 315     # placeholder string into a list, then pretty-formatted that list)
 316     template = " ".join(templates)
 317     template = template.replace("'" + placeholder + "'", placeholder)
 318     formatted_string = string_ops.string_format(
 319         inputs=tensors, template=template, placeholder=placeholder,
 320         summarize=summarize,
 321         name=format_name)
 322 
 323   return gen_logging_ops.print_v2(formatted_string,
 324                                   output_stream=output_stream_string,
 325                                   name=name)
 326 # pylint: enable=g-doc-args

 386 def image_summary(tag, tensor, max_images=3, collections=None, name=None):
 387   # pylint: disable=line-too-long
 388   """Outputs a `Summary` protocol buffer with images.
 389 
 390   For an explanation of why this op was deprecated, and information on how to
 391   migrate, look ['here'](https://github.com/tensorflow/tensorflow/blob/master/tensorflow/contrib/deprecated/__init__.py)
 392 
 393   The summary has up to `max_images` summary values containing images. The
 394   images are built from `tensor` which must be 4-D with shape `[batch_size,
 395   height, width, channels]` and where `channels` can be:
 396 
 397   *  1: `tensor` is interpreted as Grayscale.
 398   *  3: `tensor` is interpreted as RGB.
 399   *  4: `tensor` is interpreted as RGBA.
 400 
 401   The images have the same number of channels as the input tensor. For float
 402   input, the values are normalized one image at a time to fit in the range
 403   `[0, 255]`.  `uint8` values are unchanged.  The op uses two different
 404   normalization algorithms:
 405 
 406   *  If the input values are all positive, they are rescaled so the largest one
 407      is 255.
 408 
 409   *  If any input value is negative, the values are shifted so input value 0.0
 410      is at 127.  They are then rescaled so that either the smallest value is 0,
 411      or the largest one is 255.
 412 
 413   The `tag` argument is a scalar `Tensor` of type `string`.  It is used to
 414   build the `tag` of the summary values:
 415 
 416   *  If `max_images` is 1, the summary value tag is '*tag*/image'.
 417   *  If `max_images` is greater than 1, the summary value tags are
 418      generated sequentially as '*tag*/image/0', '*tag*/image/1', etc.
 419 
 420   Args:
 421     tag: A scalar `Tensor` of type `string`. Used to build the `tag`
 422       of the summary values.
 423     tensor: A 4-D `uint8` or `float32` `Tensor` of shape `[batch_size, height,
 424       width, channels]` where `channels` is 1, 3, or 4.
 425     max_images: Max number of batch elements to generate images for.
 426     collections: Optional list of ops.GraphKeys.  The collections to add the
 427       summary to.  Defaults to [ops.GraphKeys.SUMMARIES]
 428     name: A name for the operation (optional).
 429 
 430   Returns:
 431     A scalar `Tensor` of type `string`. The serialized `Summary` protocol
 432     buffer.
 433   """
 434   with ops.name_scope(name, "ImageSummary", [tag, tensor]) as scope:
 435     val = gen_logging_ops.image_summary(
 436         tag=tag, tensor=tensor, max_images=max_images, name=scope)
 437     _Collect(val, collections, [ops.GraphKeys.SUMMARIES])
 438   return val

 557 def get_summary_op():
 558   """Returns a single Summary op that would run all summaries.
 559 
 560   Either existing one from `SUMMARY_OP` collection or merges all existing
 561   summaries.
 562 
 563   Returns:
 564     If no summaries were collected, returns None. Otherwise returns a scalar
 565     `Tensor` of type `string` containing the serialized `Summary` protocol
 566     buffer resulting from the merging.
 567   """
 568   summary_op = ops.get_collection(ops.GraphKeys.SUMMARY_OP)
 569   if summary_op is not None:
 570     if summary_op:
 571       summary_op = summary_op[0]
 572     else:
 573       summary_op = None
 574   if summary_op is None:
 575     summary_op = merge_all_summaries()
 576     if summary_op is not None:
 577       ops.add_to_collection(ops.GraphKeys.SUMMARY_OP, summary_op)
 578   return summary_op

 581 @deprecated(

 590   """Outputs a `Summary` protocol buffer with scalar values.
 591 
 592   This ops is deprecated. Please switch to tf.summary.scalar.
 593   For an explanation of why this op was deprecated, and information on how to
 594   migrate, look ['here'](https://github.com/tensorflow/tensorflow/blob/master/tensorflow/contrib/deprecated/__init__.py)
 595

 597   summary has a summary value for each tag-value pair in `tags` and `values`.
 598 
 599   Args:
 600     tags: A `string` `Tensor`.  Tags for the summaries.
 601     values: A real numeric Tensor.  Values for the summaries.
 602     collections: Optional list of graph collections keys. The new summary op is
 603       added to these collections. Defaults to `[GraphKeys.SUMMARIES]`.
 604     name: A name for the operation (optional).
 605 
 606   Returns:
 607     A scalar `Tensor` of type `string`. The serialized `Summary` protocol
 608     buffer.
 609   """
 610   with ops.name_scope(name, "ScalarSummary", [tags, values]) as scope:
 611     val = gen_logging_ops.scalar_summary(tags=tags, values=values, name=scope)
 612     _Collect(val, collections, [ops.GraphKeys.SUMMARIES])
 613   return val
