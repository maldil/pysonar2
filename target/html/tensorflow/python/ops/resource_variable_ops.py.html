  58 def eager_safe_variable_handle(shape, dtype, shared_name, name, graph_mode):
  59   """Creates a variable handle with information to do shape inference."""
  60   container = ops.get_default_graph()._container  # pylint: disable=protected-access
  61   if container is None:
  62     container = ""
  63   handle = gen_resource_variable_ops.var_handle_op(shape=shape, dtype=dtype,
  64                                                    shared_name=shared_name,
  65                                                    name=name,
  66                                                    container=container)
  67   if graph_mode:
  68     return handle
  69 
  70   # We do not want two distinct ResourceVariable objects for the same
  71   # underlying resource in the runtime.
  72   # When in eager mode, explicitly ensure so here. When in graph mode, it's
  73   # ensured by always generating different variable names.
  74   exists = gen_resource_variable_ops.var_is_initialized_op(handle)
  75   if exists:
  76     raise ValueError("variable object with name '%s' already created. Use "
  77                      "get_variable() if reuse is desired." %
  78                      shared_name)
  79   with context.graph_mode(), ops.Graph().as_default() as graph:
  80     h = gen_resource_variable_ops.var_handle_op(shape=shape, dtype=dtype,
  81                                                 shared_name=shared_name,
  82                                                 name=name,
  83                                                 container=container)
  84 
  85     # Tensor._handle_data contains information for the shape-inference code to
  86     # know the shape and dtype of the variable pointed to by a handle. Since
  87     # shape inference doesn't run in eager mode we copy this data here for when
  88     # the handle is captured by an eager mode function.
  89     # pylint: disable=protected-access
  90     if ops._USE_C_SHAPES:
  91       handle._handle_data = get_resource_handle_data(h)
  92     else:
  93       if h._handle_data is None:
  94         ops.set_shape_and_handle_data_for_outputs(h.op)
  95       handle._handle_data = h._handle_data
  96     # pylint: enable=protected-access
  97   # Clean up op->graph->op reference cycles.
  98   ops.dismantle_graph(graph)
  99   return handle

 114 class EagerResourceDeleter(object):
 115   """An object which cleans up a resource handle.
 116 
 117   An alternative to defining a __del__ method on an object. The intended use is
 118   that ResourceVariables or other objects with resource handles will maintain a
 119   single reference to this object. When the parent object is collected, this
 120   object will be too. Even if the parent object is part of a reference cycle,
 121   the cycle will be collectable.
 122   """
 123 
 124   def __init__(self, handle, handle_device):
 125     if not isinstance(handle, ops.Tensor):
 126       raise ValueError(
 127           ("Passed handle=%s to EagerResourceDeleter. Was expecting a handle "
 128            "Tensor." % (handle,)))
 129     self._handle = handle
 130     self._handle_device = handle_device
 131 
 132   def __del__(self):
 133     # Resources follow object-identity when executing eagerly, so it is safe to
 134     # delete the resource we have a handle to.
 135     try:
 136       # This resource was created in eager mode. However, this destructor may be
 137       # running in graph mode (especially during unit tests). To clean up
 138       # successfully, we switch back into eager mode temporarily.
 139       with context.eager_mode():
 140         with ops.device(self._handle_device):
 141           gen_resource_variable_ops.destroy_resource_op(
 142               self._handle, ignore_lookup_error=True)
 143     except TypeError:
 144       # Suppress some exceptions, mainly for the case when we're running on
 145       # module deletion. Things that can go wrong include the context module
 146       # already being unloaded, self._handle._handle_data no longer being
 147       # valid, and so on. Printing warnings in these cases is silly
 148       # (exceptions raised from __del__ are printed as warnings to stderr).
 149       pass  # 'NoneType' object is not callable when the handle has been
 150       # partially unloaded.
 151     except AttributeError:
 152       pass  # 'NoneType' object has no attribute 'eager_mode' when context has
 153       # been unloaded. Will catch other module unloads as well.
 154

 167 class ResourceVariable(variables.RefVariable):
 168   """Variable based on resource handles.
 169 
 170   See the [Variables How To](https://tensorflow.org/guide/variables)
 171   for a high level overview.
 172 
 173   A `ResourceVariable` allows you to maintain state across subsequent calls to
 174   session.run.
 175 
 176   The `ResourceVariable` constructor requires an initial value for the variable,
 177   which can be a `Tensor` of any type and shape. The initial value defines the
 178   type and shape of the variable. After construction, the type and shape of
 179   the variable are fixed. The value can be changed using one of the assign
 180   methods.
 181 
 182   Just like any `Tensor`, variables created with
 183   `tf.Variable(use_resource=True)` can be used as inputs for other Ops in the
 184   graph. Additionally, all the operators overloaded for the `Tensor` class are
 185   carried over to variables, so you can also add nodes to the graph by just
 186   doing arithmetic on variables.
 187 
 188   Unlike ref-based variable, a ResourceVariable has well-defined semantics. Each
 189   usage of a ResourceVariable in a TensorFlow graph adds a read_value operation
 190   to the graph. The Tensors returned by a read_value operation are guaranteed to
 191   see all modifications to the value of the variable which happen in any
 192   operation on which the read_value depends on (either directly, indirectly, or
 193   via a control dependency) and guaranteed to not see any modification to the
 194   value of the variable from operations that depend on the read_value operation.
 195   Updates from operations that have no dependency relationship to the read_value
 196   operation might or might not be visible to read_value.
 197 
 198   For example, if there is more than one assignment to a ResourceVariable in
 199   a single session.run call there is a well-defined value for each operation
 200   which uses the variable's value if the assignments and the read are connected
 201   by edges in the graph. Consider the following example, in which two writes
 202   can cause tf.Variable and tf.ResourceVariable to behave differently:
 203 
 204   ```python
 205   a = tf.Variable(1.0, use_resource=True)
 206   a.initializer.run()
 207 
 208   assign = a.assign(2.0)
 209   with tf.control_dependencies([assign]):
 210     b = a.read_value()
 211   with tf.control_dependencies([b]):
 212     other_assign = a.assign(3.0)
 213   with tf.control_dependencies([other_assign]):
 214     # Will print 2.0 because the value was read before other_assign ran. If
 215     # `a` was a tf.Variable instead, 2.0 or 3.0 could be printed.
 216     tf.Print(b, [b]).eval()
 217   ```
 218   """

 220   def __init__(self,
 221                initial_value=None,
 222                trainable=True,
 223                collections=None,
 224                validate_shape=True,
 225                caching_device=None,
 226                name=None,
 227                dtype=None,
 228                variable_def=None,
 229                import_scope=None,
 230                constraint=None):
 231     """Creates a variable.
 232 
 233     Args:
 234       initial_value: A `Tensor`, or Python object convertible to a `Tensor`,
 235         which is the initial value for the Variable. The initial value must have
 236         a shape specified unless `validate_shape` is set to False. Can also be a
 237         callable with no argument that returns the initial value when called.
 238         (Note that initializer functions from init_ops.py must first be bound
 239          to a shape before being used here.)
 240       trainable: If `True`, the default, also adds the variable to the graph
 241         collection `GraphKeys.TRAINABLE_VARIABLES`. This collection is used as
 242         the default list of variables to use by the `Optimizer` classes.
 243       collections: List of graph collections keys. The new variable is added to
 244         these collections. Defaults to `[GraphKeys.GLOBAL_VARIABLES]`.
 245       validate_shape: Ignored. Provided for compatibility with tf.Variable.
 246       caching_device: Optional device string or function describing where the
 247         Variable should be cached for reading.  Defaults to the Variable's
 248         device.  If not `None`, caches on another device.  Typical use is to
 249         cache on the device where the Ops using the Variable reside, to
 250         deduplicate copying through `Switch` and other conditional statements.
 251       name: Optional name for the variable. Defaults to `'Variable'` and gets
 252         uniquified automatically.
 253       dtype: If set, initial_value will be converted to the given type.
 254         If None, either the datatype will be kept (if initial_value is
 255         a Tensor) or float32 will be used (if it is a Python object convertible
 256         to a Tensor).
 257       variable_def: `VariableDef` protocol buffer. If not None, recreates the
 258         `ResourceVariable` object with its contents. `variable_def` and other
 259         arguments (except for import_scope) are mutually exclusive.
 260       import_scope: Optional `string`. Name scope to add to the
 261         ResourceVariable. Only used when `variable_def` is provided.
 262       constraint: An optional projection function to be applied to the variable
 263         after being updated by an `Optimizer` (e.g. used to implement norm
 264         constraints or value constraints for layer weights). The function must
 265         take as input the unprojected Tensor representing the value of the
 266         variable and return the Tensor for the projected value
 267         (which must have the same shape). Constraints are not safe to
 268         use when doing asynchronous distributed training.
 269 
 270     Raises:
 271       ValueError: If the initial value is not specified, or does not have a
 272         shape and `validate_shape` is `True`.
 273 
 274     @compatibility(eager)
 275     When Eager Execution is enabled, the default for the `collections` argument
 276     is `None`, which signifies that this `Variable` will not be added to any
 277     collections.
 278     @end_compatibility
 279     """
 280     if variable_def:
 281       if initial_value is not None:
 282         raise ValueError("variable_def and initial_value are mutually "
 283                          "exclusive.")
 284       if context.executing_eagerly():
 285         raise ValueError("Creating ResourceVariable from variable_def is "
 286                          "not supported when eager execution is enabled.")
 287       self._init_from_proto(variable_def, import_scope=import_scope)
 288     else:
 289       self._init_from_args(
 290           initial_value=initial_value,
 291           trainable=trainable,
 292           collections=collections,
 293           validate_shape=validate_shape,
 294           caching_device=caching_device,
 295           name=name,
 296           dtype=dtype,
 297           constraint=constraint)

 299   # pylint: disable=unused-argument
 300   def _init_from_args(self,
 301                       initial_value=None,
 302                       trainable=True,
 303                       collections=None,
 304                       validate_shape=True,
 305                       caching_device=None,
 306                       name=None,
 307                       dtype=None,
 308                       constraint=None):
 309     """Creates a variable.
 310 
 311     Args:
 312       initial_value: A `Tensor`, or Python object convertible to a `Tensor`,
 313         which is the initial value for the Variable. The initial value must have
 314         a shape specified unless `validate_shape` is set to False. Can also be a
 315         callable with no argument that returns the initial value when called.
 316         (Note that initializer functions from init_ops.py must first be bound
 317          to a shape before being used here.)
 318       trainable: If `True`, the default, also adds the variable to the graph
 319         collection `GraphKeys.TRAINABLE_VARIABLES`. This collection is used as
 320         the default list of variables to use by the `Optimizer` classes.
 321       collections: List of graph collections keys. The new variable is added to
 322         these collections. Defaults to `[GraphKeys.GLOBAL_VARIABLES]`.
 323       validate_shape: Ignored. Provided for compatibility with tf.Variable.
 324       caching_device: Optional device string or function describing where the
 325         Variable should be cached for reading.  Defaults to the Variable's
 326         device.  If not `None`, caches on another device.  Typical use is to
 327         cache on the device where the Ops using the Variable reside, to
 328         deduplicate copying through `Switch` and other conditional statements.
 329       name: Optional name for the variable. Defaults to `'Variable'` and gets
 330         uniquified automatically.
 331       dtype: If set, initial_value will be converted to the given type.
 332         If None, either the datatype will be kept (if initial_value is
 333        a Tensor) or float32 will be used (if it is a Python object convertible
 334        to a Tensor).
 335       constraint: An optional projection function to be applied to the variable
 336         after being updated by an `Optimizer` (e.g. used to implement norm
 337         constraints or value constraints for layer weights). The function must
 338         take as input the unprojected Tensor representing the value of the
 339         variable and return the Tensor for the projected value
 340         (which must have the same shape). Constraints are not safe to
 341         use when doing asynchronous distributed training.
 342 
 343     Raises:
 344       ValueError: If the initial value is not specified, or does not have a
 345         shape and `validate_shape` is `True`.
 346 
 347     @compatibility(eager)
 348     When Eager Execution is enabled, variables are never added to collections.
 349     It is not implicitly added to the `GLOBAL_VARIABLES` or
 350     `TRAINABLE_VARIABLES` collections, and the `collections` argument is
 351     ignored.
 352     @end_compatibility
 353     """
 354     if initial_value is None:
 355       raise ValueError("initial_value must be specified.")
 356     init_from_fn = callable(initial_value)
 357 
 358     if isinstance(initial_value, ops.Tensor) and hasattr(
 359         initial_value, "graph") and initial_value.graph.building_function:
 360       raise ValueError("Tensor-typed variable initializers must either be "
 361                        "wrapped in an init_scope or callable "
 362                        "(e.g., `tf.Variable(lambda : "
 363                        "tf.truncated_normal([10, 40]))`) when building "
 364                        "functions. Please file a feature request if this "
 365                        "restriction inconveniences you.")
 366 
 367     if collections is None:
 368       collections = [ops.GraphKeys.GLOBAL_VARIABLES]
 369     if not isinstance(collections, (list, tuple, set)):
 370       raise ValueError(
 371           "collections argument to Variable constructor must be a list, tuple, "
 372           "or set. Got %s of type %s" % (collections, type(collections)))
 373     if constraint is not None and not callable(constraint):
 374       raise ValueError("The `constraint` argument must be a callable.")
 375 
 376     if isinstance(initial_value, checkpointable.CheckpointInitialValue):
 377       self._maybe_initialize_checkpointable()
 378       self._update_uid = initial_value.checkpoint_position.restore_uid
 379       initial_value = initial_value.wrapped_value
 380 
 381     self._trainable = trainable
 382     if trainable and ops.GraphKeys.TRAINABLE_VARIABLES not in collections:
 383       collections = list(collections) + [ops.GraphKeys.TRAINABLE_VARIABLES]
 384     self._save_slice_info = None
 385     # Store the graph key so optimizers know how to only retrieve variables from
 386     # this graph.
 387     self._graph_key = ops.get_default_graph()._graph_key  # pylint: disable=protected-access
 388     with ops.init_scope():
 389       self._in_graph_mode = not context.executing_eagerly()
 390       with ops.name_scope(name, "Variable", []
 391                           if init_from_fn else [initial_value]) as name:
 392         # pylint: disable=protected-access
 393         handle_name = ops._name_from_scope_name(name)
 394         if self._in_graph_mode:
 395           shared_name = handle_name
 396         else:
 397           # When in eager mode use a uid for the shared_name, to prevent
 398           # accidental sharing.
 399           shared_name = "%s_%d" % (handle_name, ops.uid())
 400         # Use attr_scope and device(None) to simulate the behavior of
 401         # colocate_with when the variable we want to colocate with doesn't
 402         # yet exist.
 403         attr = attr_value_pb2.AttrValue(
 404             list=attr_value_pb2.AttrValue.ListValue(
 405                 s=[compat.as_bytes("loc:@%s" % handle_name)]))
 406         with ops.get_default_graph()._attr_scope({"_class": attr}):
 407           with ops.name_scope("Initializer"), ops.device(None):
 408             initial_value = ops.convert_to_tensor(
 409                 initial_value() if init_from_fn else initial_value,
 410                 name="initial_value", dtype=dtype)
 411           self._handle = eager_safe_variable_handle(
 412               shape=initial_value.get_shape(),
 413               dtype=initial_value.dtype.base_dtype,
 414               shared_name=shared_name,
 415               name=name,
 416               graph_mode=self._in_graph_mode)
 417         self._shape = initial_value.shape
 418         # pylint: disable=protected-access
 419         if (self._in_graph_mode and initial_value is not None and
 420             initial_value.op._get_control_flow_context() is not None):
 421           raise ValueError(
 422               "Initializer for variable %s is from inside a control-flow "
 423               "construct, such as a loop or conditional. When creating a "
 424               "variable inside a loop or conditional, use a lambda as the "
 425               "initializer." % name)
 426         # pylint: enable=protected-access
 427         self._unique_id = shared_name
 428         self._initial_value = initial_value if self._in_graph_mode else None
 429         self._handle_name = handle_name + ":0"
 430         self._dtype = initial_value.dtype.base_dtype
 431         self._constraint = constraint
 432 
 433         if self._in_graph_mode:
 434           with ops.name_scope("IsInitialized"):
 435             self._is_initialized_op = (
 436                 gen_resource_variable_ops.var_is_initialized_op(self._handle))
 437           if initial_value is not None:
 438             with ops.name_scope("Assign") as n, ops.colocate_with(self._handle):
 439               self._initializer_op = (
 440                   gen_resource_variable_ops.assign_variable_op(
 441                       self._handle,
 442                       self._try_guard_against_uninitialized_dependencies(
 443                           initial_value),
 444                       name=n))
 445           with ops.name_scope("Read"), ops.colocate_with(self._handle):
 446             # Manually assign reads to the handle's device to avoid log
 447             # messages.
 448             with ops.device(self._handle.device):
 449               value = self._read_variable_op()
 450             self._graph_element = value
 451             if caching_device is not None:
 452               # Variables may be created in a tf.device() or ops.colocate_with()
 453               # context. At the same time, users would expect caching device to
 454               # be independent of this context, and/or would not expect the
 455               # current device context to be merged with the caching device
 456               # spec.  Therefore we reset the colocation stack before creating
 457               # the cached value. Note that resetting the colocation stack will
 458               # also reset the device stack.
 459               with ops.colocate_with(None, ignore_existing=True):
 460                 with ops.device(caching_device):
 461                   self._cached_value = array_ops.identity(value)
 462             else:
 463               self._cached_value = None
 464         else:
 465           gen_resource_variable_ops.assign_variable_op(self._handle,
 466                                                        initial_value)
 467           self._is_initialized_op = None
 468           self._initializer_op = None
 469           self._graph_element = None
 470           if caching_device:
 471             with ops.device(caching_device):
 472               self._cached_value = self._read_variable_op()
 473           else:
 474             self._cached_value = None
 475         if not context.executing_eagerly():
 476           # Eager variables are only added to collections if they are part of an
 477           # eager variable store (otherwise in an interactive session they would
 478           # hog memory and cause OOM). This is done in ops/variable_scope.py.
 479           ops.add_to_collections(collections, self)
 480         elif ops.GraphKeys.GLOBAL_STEP in collections:
 481           ops.add_to_collections(ops.GraphKeys.GLOBAL_STEP, self)
 482 
 483     if not self._in_graph_mode:
 484       # After the handle has been created, set up a way to clean it up when
 485       # executing eagerly. We'll hold the only reference to the deleter, so that
 486       # when this object is garbage collected the deleter will be too. This
 487       # means ResourceVariables can be part of reference cycles without those
 488       # cycles being uncollectable, and means that no __del__ will be defined at
 489       # all in graph mode.
 490       self._handle_deleter = EagerResourceDeleter(
 491           handle=self._handle, handle_device=self._handle.device)
 492     self._cached_shape_as_list = None

 494   def _init_from_proto(self, variable_def, import_scope=None):
 495     """Initializes from `VariableDef` proto."""
 496     # Note that init_from_proto is currently not supported in Eager mode.
 497     assert not context.executing_eagerly()
 498     self._in_graph_mode = True
 499     assert isinstance(variable_def, variable_pb2.VariableDef)
 500     if not variable_def.is_resource:
 501       raise ValueError("Trying to restore Variable as ResourceVariable.")
 502 
 503     # Create from variable_def.
 504     g = ops.get_default_graph()
 505     self._handle = g.as_graph_element(
 506         ops.prepend_name_scope(
 507             variable_def.variable_name, import_scope=import_scope))
 508     self._shape = tensor_shape.TensorShape(
 509         self._handle.op.get_attr("shape"))
 510     self._handle_name = self._handle.name
 511     self._unique_id = self._handle_name
 512     self._initializer_op = g.as_graph_element(
 513         ops.prepend_name_scope(

 761   def to_proto(self, export_scope=None):
 762     """Converts a `ResourceVariable` to a `VariableDef` protocol buffer.
 763 
 764     Args:
 765       export_scope: Optional `string`. Name scope to remove.
 766

 768       RuntimeError: If run in EAGER mode.
 769

 772       in the specified name scope.
 773     """
 774     if context.executing_eagerly():
 775       raise RuntimeError("to_proto not supported in EAGER mode.")
 776     if export_scope is None or self.handle.name.startswith(export_scope):

 779                                                    export_scope)

1024   def scatter_nd_sub(self, indices, updates, name=None):
1025     """Applies sparse subtraction to individual values or slices in a Variable.
1026 
1027     `ref` is a `Tensor` with rank `P` and `indices` is a `Tensor` of rank `Q`.
1028 
1029     `indices` must be integer tensor, containing indices into `ref`.
1030     It must be shape `[d_0, ..., d_{Q-2}, K]` where `0 < K <= P`.
1031 
1032     The innermost dimension of `indices` (with length `K`) corresponds to
1033     indices into elements (if `K = P`) or slices (if `K < P`) along the `K`th
1034     dimension of `ref`.
1035 
1036     `updates` is `Tensor` of rank `Q-1+P-K` with shape:
1037 
1038     ```
1039     [d_0, ..., d_{Q-2}, ref.shape[K], ..., ref.shape[P-1]].
1040     ```
1041 
1042     For example, say we want to add 4 scattered elements to a rank-1 tensor to
1043     8 elements. In Python, that update would look like this:
1044 
1045     ```python
1046         ref = tf.Variable([1, 2, 3, 4, 5, 6, 7, 8])
1047         indices = tf.constant([[4], [3], [1] ,[7]])
1048         updates = tf.constant([9, 10, 11, 12])
1049         op = ref.scatter_nd_sub(indices, updates)
1050         with tf.Session() as sess:
1051           print sess.run(op)
1052     ```
1053 
1054     The resulting update to ref would look like this:
1055 
1056         [1, -9, 3, -6, -6, 6, 7, -4]
1057 
1058     See `tf.scatter_nd` for more details about how to make updates to
1059     slices.
1060 
1061     Args:
1062       indices: The indices to be used in the operation.
1063       updates: The values to be used in the operation.
1064       name: the name of the operation.
1065 
1066     Returns:
1067       A `Tensor` that will hold the new value of this variable after
1068       the scattered subtraction has completed.
1069 
1070     Raises:
1071       ValueError: if `sparse_delta` is not an `IndexedSlices`.
1072     """
1073     return self._lazy_read(gen_state_ops.resource_scatter_nd_sub(
1074         self.handle, indices, ops.convert_to_tensor(updates, self.dtype),
1075         name=name))

1130   def scatter_nd_update(self, indices, updates, name=None):
1131     """Applies sparse assignment to individual values or slices in a Variable.
1132 
1133     `ref` is a `Tensor` with rank `P` and `indices` is a `Tensor` of rank `Q`.
1134 
1135     `indices` must be integer tensor, containing indices into `ref`.
1136     It must be shape `[d_0, ..., d_{Q-2}, K]` where `0 < K <= P`.
1137 
1138     The innermost dimension of `indices` (with length `K`) corresponds to
1139     indices into elements (if `K = P`) or slices (if `K < P`) along the `K`th
1140     dimension of `ref`.
1141 
1142     `updates` is `Tensor` of rank `Q-1+P-K` with shape:
1143 
1144     ```
1145     [d_0, ..., d_{Q-2}, ref.shape[K], ..., ref.shape[P-1]].
1146     ```
1147 
1148     For example, say we want to add 4 scattered elements to a rank-1 tensor to
1149     8 elements. In Python, that update would look like this:
1150 
1151     ```python
1152         ref = tf.Variable([1, 2, 3, 4, 5, 6, 7, 8])
1153         indices = tf.constant([[4], [3], [1] ,[7]])
1154         updates = tf.constant([9, 10, 11, 12])
1155         op = ref.scatter_nd_update(indices, updates)
1156         with tf.Session() as sess:
1157           print sess.run(op)
1158     ```
1159 
1160     The resulting update to ref would look like this:
1161 
1162         [1, 11, 3, 10, 9, 6, 7, 12]
1163 
1164     See `tf.scatter_nd` for more details about how to make updates to
1165     slices.
1166 
1167     Args:
1168       indices: The indices to be used in the operation.
1169       updates: The values to be used in the operation.
1170       name: the name of the operation.
1171 
1172     Returns:
1173       A `Tensor` that will hold the new value of this variable after
1174       the scattered subtraction has completed.
1175 
1176     Raises:
1177       ValueError: if `sparse_delta` is not an `IndexedSlices`.
1178     """
1179     return self._lazy_read(gen_state_ops.resource_scatter_nd_update(
1180         self.handle, indices, ops.convert_to_tensor(updates, self.dtype),
1181         name=name))
