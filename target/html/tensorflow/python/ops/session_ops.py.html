  40 def encode_resource_handle(resource_handle):
  41   """Encode a ResourceHandle proto as custom numpy struct type."""
  42   return np.asarray(bytearray(resource_handle.SerializeToString()),
  43                     dtype=dtypes.np_resource)

  46 class TensorHandle(object):
  47   """Represents a handle for a live tensor in a session."""
  48 
  49   def __init__(self, handle, dtype, session):
  50     """Constructs a new tensor handle.
  51 
  52     A tensor handle for a persistent tensor is a python string
  53     that has the form of "tensor_name;unique_id;device_name".
  54 
  55     Args:
  56       handle: A tensor handle.
  57       dtype: The data type of the tensor represented by `handle`.
  58       session: The session in which the tensor is produced.
  59     """
  60     self._handle = compat.as_str_any(handle)
  61     self._resource_handle = None
  62     self._dtype = dtype
  63     self._session = session
  64     self._auto_gc_enabled = True
  65 
  66   def __del__(self):
  67     if self._auto_gc_enabled:
  68       self._session._register_dead_handle(self.handle)
  69 
  70   def __str__(self):
  71     return self._handle
  72 
  73   def _get_resource_handle(self):
  74     """The ResourceHandle representation of this handle."""
  75     if not self._resource_handle:
  76       self._resource_handle = resource_handle_pb2.ResourceHandleProto()
  77       self._resource_handle.device = self._handle.split(";")[-1]
  78       self._resource_handle.container = (
  79           pywrap_tensorflow_internal.TENSOR_HANDLE_KEY)
  80       self._resource_handle.name = self._handle
  81     return self._resource_handle
  82 
  83   def to_numpy_array(self):
  84     """Convert a TensorHandle object to a feedable numpy value.
  85 
  86     Returns:
  87       A numpy array of a custom struct type that can be used as a feed value
  88       to run().
  89     """
  90     return encode_resource_handle(self._get_resource_handle())
  91 
  92   @property
  93   def handle(self):
  94     """The string representation of this handle."""
  95     return self._handle
  96 
  97   def eval(self):
  98     """Return the value of the tensor represented by this handle."""
  99     if not self._auto_gc_enabled:
 100       raise TypeError("Persistent tensor %s may have already been deleted."
 101                       % self.handle)
 102     holder, reader = _get_handle_reader(self._session.graph, self._handle,
 103                                         self._dtype)
 104     return self._session.run(reader, feed_dict={holder: self._handle})
 105 
 106   def delete(self):
 107     """Force the deletion of this persistent tensor."""
 108     if not self._auto_gc_enabled:
 109       raise TypeError("Persistent tensor %s may have already been deleted."
 110                       % self.handle)
 111     self._auto_gc_enabled = False
 112     holder, deleter = _get_handle_deleter(self._session.graph, 0, self._handle)
 113     self._session.run(deleter, feed_dict={holder: self.handle})
 114 
 115   def get_raw_handle(self):
 116     """Return the raw handle of the tensor.
 117 
 118     Note that the method disables the automatic garbage collection of this
 119     persistent tensor. The caller is now responsible for managing the life
 120     time of the tensor.
 121     """
 122     self._auto_gc_enabled = False
 123     return self._handle
 124 
 125   @staticmethod
 126   def _get_device_name(handle):
 127     """The device name encoded in the handle."""
 128     handle_str = compat.as_str_any(handle)
 129     return pydev.canonical_name(handle_str.split(";")[-1])
 130 
 131   @staticmethod
 132   def _get_reader_key(handle):
 133     """The graph key for reader."""
 134     handle_parts = str(handle).split(";")
 135     return handle_parts[0] + ";" + handle_parts[-1]
 136 
 137   @staticmethod
 138   def _get_mover_key(feeder, handle):
 139     """The graph key for mover."""
 140     return feeder.op.name + ";" + TensorHandle._get_reader_key(handle)

 143 @tf_export("get_session_handle")
 144 def get_session_handle(data, name=None):
 145   """Return the handle of `data`.
 146 
 147   This is EXPERIMENTAL and subject to change.
 148 
 149   Keep `data` "in-place" in the runtime and create a handle that can be
 150   used to retrieve `data` in a subsequent run().
 151 
 152   Combined with `get_session_tensor`, we can keep a tensor produced in
 153   one run call in place, and use it as the input in a future run call.
 154 
 155   Args:
 156     data: A tensor to be stored in the session.
 157     name: Optional name prefix for the return tensor.
 158 
 159   Returns:
 160     A scalar string tensor representing a unique handle for `data`.
 161 
 162   Raises:
 163     TypeError: if `data` is not a Tensor.
 164 
 165   Example:
 166 
 167   ```python
 168   c = tf.multiply(a, b)
 169   h = tf.get_session_handle(c)
 170   h = sess.run(h)
 171 
 172   p, a = tf.get_session_tensor(h.handle, tf.float32)
 173   b = tf.multiply(a, 10)
 174   c = sess.run(b, feed_dict={p: h.handle})
 175   ```
 176 
 177   """
 178   if not isinstance(data, ops.Tensor):
 179     raise TypeError("`data` must be of type Tensor.")
 180 
 181   # Colocate this operation with data.
 182   with ops.colocate_with(data):
 183     return gen_data_flow_ops.get_session_handle(data, name=name)
 184

 186 @tf_export("get_session_tensor")
 187 def get_session_tensor(handle, dtype, name=None):
 188   """Get the tensor of type `dtype` by feeding a tensor handle.
 189 
 190   This is EXPERIMENTAL and subject to change.
 191 
 192   Get the value of the tensor from a tensor handle. The tensor
 193   is produced in a previous run() and stored in the state of the
 194   session.
 195 
 196   Args:
 197     handle: The string representation of a persistent tensor handle.
 198     dtype: The type of the output tensor.
 199     name: Optional name prefix for the return tensor.
 200 
 201   Returns:
 202     A pair of tensors. The first is a placeholder for feeding a
 203     tensor handle and the second is the tensor in the session state
 204     keyed by the tensor handle.
 205 
 206   Example:
 207 
 208   ```python
 209   c = tf.multiply(a, b)
 210   h = tf.get_session_handle(c)
 211   h = sess.run(h)
 212 
 213   p, a = tf.get_session_tensor(h.handle, tf.float32)
 214   b = tf.multiply(a, 10)
 215   c = sess.run(b, feed_dict={p: h.handle})
 216   ```
 217 
 218   """
 219   handle_device = TensorHandle._get_device_name(handle)
 220   with ops.device(handle_device):
 221     holder = array_ops.placeholder(dtypes.string)
 222     _register_handle_feeder(holder.graph, holder, dtype)
 223     tensor = gen_data_flow_ops.get_session_tensor(holder, dtype, name=name)
 224   return (holder, tensor)

 227 @tf_export("delete_session_tensor")
 228 def delete_session_tensor(handle, name=None):
 229   """Delete the tensor for the given tensor handle.
 230 
 231   This is EXPERIMENTAL and subject to change.
 232 
 233   Delete the tensor of a given tensor handle. The tensor is produced
 234   in a previous run() and stored in the state of the session.
 235 
 236   Args:
 237     handle: The string representation of a persistent tensor handle.
 238     name: Optional name prefix for the return tensor.
 239 
 240   Returns:
 241     A pair of graph elements. The first is a placeholder for feeding a
 242     tensor handle and the second is a deletion operation.
 243   """
 244   handle_device = TensorHandle._get_device_name(handle)
 245   with ops.device(handle_device):
 246     holder = array_ops.placeholder(dtypes.string)
 247     deleter = gen_data_flow_ops.delete_session_tensor(holder, name=name)
 248   return (holder, deleter)

 251 def _register_handle_feeder(graph, feeder, dtype):
 252   graph._handle_feeders[feeder.op.name] = dtype
 253 
 254 
 255 def _get_handle_feeder(graph, feeder):
 256   return graph._handle_feeders.get(feeder.op.name)

 259 def _get_handle_reader(graph, handle, dtype):
 260   """Return a read subgraph for this handle."""
 261   graph_key = TensorHandle._get_reader_key(handle)
 262   result = graph._handle_readers.get(graph_key)
 263   if result is None:
 264     # Create reader if we haven't done it.
 265     handle_device = TensorHandle._get_device_name(handle)
 266     with graph.as_default(), graph.device(handle_device):
 267       holder = array_ops.placeholder(dtypes.string)
 268       _register_handle_feeder(holder.graph, holder, dtype)
 269       reader = gen_data_flow_ops.get_session_tensor(holder, dtype)
 270     result = (holder, reader)
 271     graph._handle_readers[graph_key] = result
 272   return result
 273 
 274 
 275 def _get_handle_mover(graph, feeder, handle):
 276   """Return a move subgraph for this pair of feeder and handle."""
 277   dtype = _get_handle_feeder(graph, feeder)
 278   if dtype is None:
 279     return None
 280   handle_device = TensorHandle._get_device_name(handle)
 281   if feeder.op.device == handle_device:
 282     return None
 283   # Now we know we have to move the tensor.
 284   graph_key = TensorHandle._get_mover_key(feeder, handle)
 285   result = graph._handle_movers.get(graph_key)
 286   if result is None:
 287     # Create mover if we haven't done it.
 288     holder, reader = _get_handle_reader(graph, handle, dtype)
 289     with graph.as_default(), graph.device(feeder.op.device):
 290       mover = gen_data_flow_ops.get_session_handle(reader)
 291     result = (holder, mover)
 292     graph._handle_movers[graph_key] = result
 293   return result
 294 
 295 
 296 def _get_handle_deleter(graph, deleter_key, handle):
 297   """Return a deletion subgraph for this handle."""
 298   result = graph._handle_deleters.get(deleter_key)
 299   if result is None:
 300     # Create deleter if we haven't done it.
 301     handle_device = TensorHandle._get_device_name(handle)
 302     with graph.as_default(), graph.device(handle_device):
 303       holder = array_ops.placeholder(dtypes.string)
 304       deleter = gen_data_flow_ops.delete_session_tensor(holder)
 305     result = (holder, deleter)
 306     graph._handle_deleters[deleter_key] = result
 307   return result
