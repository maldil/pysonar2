  28 def jacobian(output, inputs, use_pfor=True):
  29   """Computes jacobian of `output` w.r.t. `inputs`.
  30 
  31   Args:
  32     output: A tensor.
  33     inputs: A tensor or a nested structure of tensor objects.
  34     use_pfor: If true, uses pfor for computing the jacobian. Else uses
  35       tf.while_loop.
  36 
  37   Returns:
  38     A tensor or a nested strucutre of tensors with the same structure as
  39     `inputs`. Each entry is the jacobian of `output` w.rt. to the corresponding
  40     value in `inputs`. If output has shape [y_1, ..., y_n] and inputs_i has
  41     shape [x_1, ..., x_m], the corresponding jacobian has shape
  42     [y_1, ..., y_n, x_1, ..., x_m].
  43   """
  44   flat_inputs = nest.flatten(inputs)
  45   output_tensor_shape = output.shape
  46   output_shape = array_ops.shape(output)
  47   output = array_ops.reshape(output, [-1])
  48 
  49   def loop_fn(i):
  50     y = array_ops.gather(output, i)
  51     return gradient_ops.gradients(y, flat_inputs)
  52 
  53   try:
  54     output_size = int(output.shape[0])
  55   except TypeError:
  56     output_size = array_ops.shape(output)[0]
  57 
  58   if use_pfor:
  59     pfor_outputs = control_flow_ops.pfor(loop_fn, output_size)
  60   else:
  61     pfor_outputs = control_flow_ops.for_loop(
  62         loop_fn, [output.dtype] * len(flat_inputs), output_size)
  63 
  64   for i, out in enumerate(pfor_outputs):
  65     if out is not None:
  66       new_shape = array_ops.concat(
  67           [output_shape, array_ops.shape(out)[1:]], axis=0)
  68       out = array_ops.reshape(out, new_shape)
  69       out.set_shape(output_tensor_shape.concatenate(flat_inputs[i].shape))
  70     pfor_outputs[i] = out
  71 
  72   return nest.pack_sequence_as(inputs, pfor_outputs)
  73

  75 def batch_jacobian(output, inp, use_pfor=True):
  76   """Computes and stacks jacobians of `output[i,...]` w.r.t. `input[i,...]`.
  77 
  78   e.g.
  79   x = tf.constant([[1, 2], [3, 4]], dtype=tf.float32)
  80   y = x * x
  81   jacobian = batch_jacobian(y, x)
  82   # => [[[2,  0], [0,  4]], [[6,  0], [0,  8]]]
  83 
  84   Args:
  85     output: A tensor with shape [b, y1, ..., y_n]. `output[i,...]` should
  86       only depend on `inp[i,...]`.
  87     inp: A tensor with shape [b, x1, ..., x_m]
  88     use_pfor: If true, uses pfor for computing the Jacobian. Else uses a
  89       tf.while_loop.
  90 
  91   Returns:
  92     A tensor `t` with shape [b, y_1, ..., y_n, x1, ..., x_m] where `t[i, ...]`
  93     is the jacobian of `output[i, ...]` w.r.t. `inp[i, ...]`, i.e. stacked
  94     per-example jacobians.
  95 
  96   Raises:
  97     ValueError: if first dimension of `output` and `inp` do not match.
  98   """
  99   output_shape = output.shape
 100   if not output_shape[0].is_compatible_with(inp.shape[0]):
 101     raise ValueError("Need first dimension of output shape (%s) and inp shape "
 102                      "(%s) to match." % (output.shape, inp.shape))
 103   if output_shape.is_fully_defined():
 104     batch_size = int(output_shape[0])
 105     output_row_size = output_shape.num_elements() // batch_size
 106   else:
 107     output_shape = array_ops.shape(output)
 108     batch_size = output_shape[0]
 109     output_row_size = array_ops.size(output) // batch_size
 110   inp_shape = array_ops.shape(inp)
 111   # Flatten output to 2-D.
 112   with ops.control_dependencies(
 113       [check_ops.assert_equal(batch_size, inp_shape[0])]):
 114     output = array_ops.reshape(output, [batch_size, output_row_size])
 115 
 116   def loop_fn(i):
 117     y = array_ops.gather(output, i, axis=1)
 118     return gradient_ops.gradients(y, inp)[0]
 119 
 120   if use_pfor:
 121     pfor_output = control_flow_ops.pfor(loop_fn, output_row_size)
 122   else:
 123     pfor_output = control_flow_ops.for_loop(loop_fn, output.dtype,
 124                                             output_row_size)
 125   if pfor_output is None:
 126     return None
 127   pfor_output = array_ops.reshape(pfor_output,
 128                                   [output_row_size, batch_size, -1])
 129   output = array_ops.transpose(pfor_output, [1, 0, 2])
 130   new_shape = array_ops.concat([output_shape, inp_shape[1:]], axis=0)
 131   return array_ops.reshape(output, new_shape)
