  29 def for_loop(loop_fn, loop_fn_dtypes, iters):
  30   """Runs `loop_fn` `iters` times and stacks the outputs.
  31 
  32 
  33   Runs `loop_fn` `iters` times, with input values from 0 to `iters - 1`, and
  34   stacks corresponding outputs of the different runs.
  35 
  36   Args:
  37     loop_fn: A function that takes an int32 scalar tf.Tensor object representing
  38       the iteration number, and returns a possibly nested structure of tensor
  39       objects. The shape of these outputs should not depend on the input.
  40     loop_fn_dtypes: dtypes for the outputs of loop_fn.
  41     iters: Number of iterations for which to run loop_fn.
  42 
  43   Returns:
  44     Returns a nested structure of stacked output tensor objects with the same
  45     nested structure as the output of `loop_fn`.
  46   """
  47 
  48   flat_loop_fn_dtypes = nest.flatten(loop_fn_dtypes)
  49   is_none_list = []
  50 
  51   def while_body(i, *ta_list):
  52     """Body of while loop."""
  53     fn_output = nest.flatten(loop_fn(i))
  54     if len(fn_output) != len(flat_loop_fn_dtypes):
  55       raise ValueError(
  56           "Number of expected outputs, %d, does not match the number of "
  57           "actual outputs, %d, from loop_fn" % (len(flat_loop_fn_dtypes),
  58                                                 len(fn_output)))
  59     outputs = []
  60     del is_none_list[:]
  61     is_none_list.extend([x is None for x in fn_output])
  62     for out, ta in zip(fn_output, ta_list):
  63       # TODO(agarwal): support returning Operation objects from loop_fn.
  64       if out is not None:
  65         ta = ta.write(i, array_ops.expand_dims(out, 0))
  66       outputs.append(ta)
  67     return tuple([i + 1] + outputs)
  68 
  69   ta_list = control_flow_ops.while_loop(
  70       lambda i, *ta: i < iters, while_body, [0] + [
  71           tensor_array_ops.TensorArray(dtype, iters)
  72           for dtype in flat_loop_fn_dtypes
  73       ])[1:]
  74 
  75   # TODO(rachelim): enable this for sparse tensors
  76 
  77   output = [None if is_none else ta.concat()
  78             for ta, is_none in zip(ta_list, is_none_list)]
  79   return nest.pack_sequence_as(loop_fn_dtypes, output)

  82 def pfor(loop_fn, iters):
  83   """Equivalent to running `loop_fn` `iters` times and stacking the outputs.
  84 
  85   `pfor` has functionality similar to `for_loop`, i.e. running `loop_fn` `iters`
  86   times, with input from 0 to `iters - 1`, and stacking corresponding output of
  87   each iteration. However the implementation does not use a tf.while_loop.
  88   Instead it adds new operations to the graph that collectively compute the same
  89   value as what running `loop_fn` in a loop would compute.
  90 
  91 
  92   This is an experimental feature and currently has a lot of limitations:
  93     - There should be no data depenendency between the different iterations. For
  94       example, a future iteration should not depend on a value or side-effect of
  95       a previous iteration.
  96     - Stateful kernels may mostly not be supported since these often imply a
  97       data dependency or ordering of the iterations. We do support a limited set
  98       of such stateful kernels though (like RandomFoo, Variable operations like
  99       reads, etc).
 100     - Conversion works only on a limited set of kernels for which a converter
 101       has been registered.
 102     - loop_fn cannot currently contain control flow operations like
 103       tf.while_loop or tf.cond.
 104     - `loop_fn` should return nested structure of Tensors or Operations. However
 105       if an Operation is returned, it should have zero outputs.
 106     - The shape and dtype of `loop_fn` outputs should not depend on the input
 107       to loop_fn.
 108 
 109   Args:
 110     loop_fn: A function that takes an int32 scalar tf.Tensor object representing
 111       the iteration number, and returns a possibly nested structure of Tensor or
 112       Operation objects.
 113     iters: Number of iterations for which to run loop_fn.
 114 
 115   Returns:
 116     Returns a nested structure of stacked tensor objects with the same nested
 117     structure as the output of `loop_fn`.
 118   """
 119   existing_ops = set(ops.get_default_graph().get_operations())
 120   with ops.name_scope("loop_body"):
 121     loop_var = array_ops.placeholder(dtypes.int32, shape=[])
 122     loop_fn_outputs = loop_fn(loop_var)
 123   new_ops = set(ops.get_default_graph().get_operations()) - existing_ops
 124   iters = ops.convert_to_tensor(iters)
 125   with ops.name_scope("pfor"):
 126     converter = PFor(loop_var, iters, new_ops)
 127     outputs = []
 128     for loop_fn_output in nest.flatten(loop_fn_outputs):
 129       outputs.append(converter.convert(loop_fn_output))
 130     return nest.pack_sequence_as(loop_fn_outputs, outputs)
