  37 @tf_export('math.lbeta', 'lbeta')
  38 @deprecation.deprecated_endpoints('lbeta')
  39 def lbeta(x, name=None):
  40   r"""Computes \\(ln(|Beta(x)|)\\), reducing along the last dimension.
  41 
  42   Given one-dimensional `z = [z_0,...,z_{K-1}]`, we define
  43 
  44   $$Beta(z) = \prod_j Gamma(z_j) / Gamma(\sum_j z_j)$$
  45 
  46   And for `n + 1` dimensional `x` with shape `[N1, ..., Nn, K]`, we define
  47   $$lbeta(x)[i1, ..., in] = Log(|Beta(x[i1, ..., in, :])|)$$.
  48 
  49   In other words, the last dimension is treated as the `z` vector.
  50 
  51   Note that if `z = [u, v]`, then
  52   \\(Beta(z) = int_0^1 t^{u-1} (1 - t)^{v-1} dt\\), which defines the
  53   traditional bivariate beta function.
  54 
  55   If the last dimension is empty, we follow the convention that the sum over
  56   the empty set is zero, and the product is one.
  57 
  58   Args:
  59     x: A rank `n + 1` `Tensor`, `n >= 0` with type `float`, or `double`.
  60     name: A name for the operation (optional).
  61 
  62   Returns:
  63     The logarithm of \\(|Beta(x)|\\) reducing along the last dimension.
  64   """
  65   # In the event that the last dimension has zero entries, we return -inf.
  66   # This is consistent with a convention that the sum over the empty set 0, and
  67   # the product is 1.
  68   # This is standard.  See https://en.wikipedia.org/wiki/Empty_set.
  69   with ops.name_scope(name, 'lbeta', [x]):
  70     x = ops.convert_to_tensor(x, name='x')
  71 
  72     # Note reduce_sum([]) = 0.
  73     log_prod_gamma_x = math_ops.reduce_sum(
  74         math_ops.lgamma(x), reduction_indices=[-1])
  75 
  76     # Note lgamma(0) = infinity, so if x = []
  77     # log_gamma_sum_x = lgamma(0) = infinity, and
  78     # log_prod_gamma_x = lgamma(1) = 0,
  79     # so result = -infinity
  80     sum_x = math_ops.reduce_sum(x, axis=[-1])
  81     log_gamma_sum_x = math_ops.lgamma(sum_x)
  82     result = log_prod_gamma_x - log_gamma_sum_x
  83 
  84     return result

  87 @tf_export('math.bessel_i0')
  88 def bessel_i0(x, name=None):
  89   """Computes the Bessel i0 function of `x` element-wise.
  90 
  91   Modified Bessel function of order 0.
  92 
  93   It is preferable to use the numerically stabler function `i0e(x)` instead.
  94 
  95   Args:
  96     x: A `Tensor` or `SparseTensor`. Must be one of the following types: `half`,
  97       `float32`, `float64`.
  98     name: A name for the operation (optional).
  99 
 100   Returns:
 101     A `Tensor` or `SparseTensor`, respectively. Has the same type as `x`.
 102 
 103   @compatibility(scipy)
 104   Equivalent to scipy.special.i0
 105   @end_compatibility
 106   """
 107   with ops.name_scope(name, 'bessel_i0', [x]):
 108     return math_ops.exp(math_ops.abs(x)) * math_ops.bessel_i0e(x)

 135 @tf_export('einsum', 'linalg.einsum')
 136 def einsum(equation, *inputs, **kwargs):
 137   """A generalized contraction between tensors of arbitrary dimension.
 138 
 139   This function returns a tensor whose elements are defined by `equation`,
 140   which is written in a shorthand form inspired by the Einstein summation
 141   convention.  As an example, consider multiplying two matrices
 142   A and B to form a matrix C.  The elements of C are given by:
 143 
 144   ```
 145     C[i,k] = sum_j A[i,j] * B[j,k]
 146   ```
 147 
 148   The corresponding `equation` is:
 149 
 150   ```
 151     ij,jk->ik
 152   ```
 153 
 154   In general, the `equation` is obtained from the more familiar element-wise
 155   equation by
 156     1. removing variable names, brackets, and commas,
 157     2. replacing "*" with ",",
 158     3. dropping summation signs, and
 159     4. moving the output to the right, and replacing "=" with "->".
 160 
 161   Many common operations can be expressed in this way.  For example:
 162 
 163   ```python
 164   # Matrix multiplication
 165   >>> einsum('ij,jk->ik', m0, m1)  # output[i,k] = sum_j m0[i,j] * m1[j, k]
 166 
 167   # Dot product
 168   >>> einsum('i,i->', u, v)  # output = sum_i u[i]*v[i]
 169 
 170   # Outer product
 171   >>> einsum('i,j->ij', u, v)  # output[i,j] = u[i]*v[j]
 172 
 173   # Transpose
 174   >>> einsum('ij->ji', m)  # output[j,i] = m[i,j]
 175 
 176   # Batch matrix multiplication
 177   >>> einsum('aij,ajk->aik', s, t)  # out[a,i,k] = sum_j s[a,i,j] * t[a, j, k]
 178   ```
 179 
 180   This function behaves like `numpy.einsum`, but does not support:
 181 
 182   * Ellipses (subscripts like `ij...,jk...->ik...`)
 183   * Subscripts where an axis appears more than once for a single input
 184     (e.g. `ijj,k->ik`).
 185   * Subscripts that are summed across multiple inputs (e.g., `ij,ij,jk->ik`).
 186 
 187   Args:
 188     equation: a `str` describing the contraction, in the same format as
 189       `numpy.einsum`.
 190     *inputs: the inputs to contract (each one a `Tensor`), whose shapes should
 191       be consistent with `equation`.
 192     name: A name for the operation (optional).
 193 
 194   Returns:
 195     The contracted `Tensor`, with shape determined by `equation`.
 196 
 197   Raises:
 198     ValueError: If
 199       - the format of `equation` is incorrect,
 200       - the number of inputs implied by `equation` does not match `len(inputs)`,
 201       - an axis appears in the output subscripts but not in any of the inputs,
 202       - the number of dimensions of an input differs from the number of
 203         indices in its subscript, or
 204       - the input shapes are inconsistent along a particular axis.
 205   """
 206   equation = equation.replace(' ', '')
 207 
 208   name = kwargs.pop('name', None)
 209   if kwargs:
 210     raise TypeError('invalid keyword arguments for this function: ' + ', '.join(
 211         [format(key) for key in sorted(list(kwargs.keys()))]))
 212   with ops.name_scope(name, 'einsum', [equation, inputs]) as name:
 213     if '...' in equation:
 214       raise ValueError('Subscripts with ellipses are not yet supported.')
 215 
 216     match = re.match('^([a-zA-Z,]+)(->[a-zA-Z]*)?$', equation)
 217     if not match:
 218       raise ValueError('Indices have incorrect format: %s' % equation)
 219 
 220     inputs = list(inputs)
 221     input_axis_labels = match.group(1).split(',')
 222 
 223     if len(inputs) != len(input_axis_labels):
 224       raise ValueError('Got %d arguments for equation "%s", expecting %d' %
 225                        (len(inputs), equation, len(input_axis_labels)))
 226 
 227     axis_labels = set(''.join(input_axis_labels))
 228     if match.group(2):
 229       output_axis_labels = match.group(2)[2:]
 230     else:
 231       # infer the output subscripts if not given, assume alphabetical order
 232       indices = ''.join(sorted(axis_labels))
 233       counts = {ax: 0 for ax in indices}
 234       for axes_ in input_axis_labels:
 235         for ax in axes_:
 236           counts[ax] += 1
 237 
 238       output_axis_labels = ''.join(
 239           sorted(ax for ax in indices if counts[ax] == 1))
 240 
 241     for a in axis_labels:
 242       input_count = sum(1 for s in input_axis_labels if a in s)
 243       if input_count > 2 and a not in output_axis_labels:
 244         logging.warn(
 245             'Falling back to exponential-space implementation of einsum()'
 246             ' because index "%s" is summed over more than two inputs.', a)
 247         return _exponential_space_einsum(equation, *inputs)
 248 
 249     temp = inputs[0]
 250     temp_axis_labels = input_axis_labels[0]
 251     for i in xrange(len(inputs) - 1):
 252       axes_to_sum = (
 253           set(temp_axis_labels) &
 254           set(input_axis_labels[i + 1]) - set(output_axis_labels))
 255       temp, temp_axis_labels = _einsum_reduction(
 256           temp, temp_axis_labels, inputs[i + 1], input_axis_labels[i + 1],
 257           axes_to_sum)
 258 
 259     missing_indices = set(temp_axis_labels) - set(output_axis_labels)
 260     if missing_indices:
 261       reduction_indices = [
 262           i for i, a in enumerate(temp_axis_labels)
 263           if a not in output_axis_labels
 264       ]
 265       temp = math_ops.reduce_sum(temp, reduction_indices=reduction_indices)
 266       temp_axis_labels = ''.join(
 267           a for a in temp_axis_labels if a in output_axis_labels)
 268 
 269     if sorted(temp_axis_labels) != sorted(output_axis_labels):
 270       raise ValueError('Invalid equation: %s' % equation)
 271 
 272     perm = [temp_axis_labels.index(a) for a in output_axis_labels]
 273     return _transpose_if_necessary(temp, perm)

 276 def _einsum_reduction(t0, t0_axis_labels, t1, t1_axis_labels, axes_to_sum):
 277   """Helper for einsum() that computes the result of a two-argument einsum().
 278 
 279   Args:
 280     t0: a `Tensor`
 281     t0_axis_labels: a string of axis labels.  This string's length must equal
 282       the rank of t0.
 283     t1: a `Tensor`
 284     t1_axis_labels: a string to axis labels.  This string's length must equal
 285       the rank of t1.
 286     axes_to_sum: set of labels of axes to be summed over
 287 
 288   Returns:
 289     A `Tensor` whose elements are obtained by summing, over all axes in
 290     `axes_to_sum`, the corresponding elements of `t0` and `t1`.
 291 
 292     For example, if t0_axis_labels == 'abijk', t1_axis_labels == 'acjkl', and
 293     axes_to_sum == {j,k}, this will return a tensor x where
 294 
 295       out[a,b,c,i,l] = sum_j sum_k t0[a,b,i,j,k] * t1[a,c,j,k,l]
 296 
 297   Raises:
 298     ValueError: if the rank of `t0` does not match the length of
 299       `t0_axis_labels`, or that of `t1` does not match the length of
 300       `t1_axis_labels`.
 301   """
 302   if len(t0_axis_labels) != len(t0.get_shape()):
 303     raise ValueError(
 304         'Tensor t0 of rank %d does not match einsum reduction of length %d' %
 305         (len(t0.get_shape()), len(t0_axis_labels)))
 306   if len(t1_axis_labels) != len(t1.get_shape()):
 307     raise ValueError(
 308         'Tensor t1 of rank %d does not match einsum reduction of length %d' %
 309         (len(t1.get_shape()), len(t1_axis_labels)))
 310 
 311   # This function computes the result of a two-argument einsum() using batch
 312   # matrix multiplication.  This involves
 313   # 1. transposing t0 and t1 so that axes are in the correct order for
 314   #    batch matrix multiplication, and
 315   # 2. reshaping t0 and t1 so that they are both of rank 3.
 316 
 317   # First, we divide axes into three groups:
 318   #  * "preserved" axes are present in both inputs and the output
 319   #  * "summed" axes are present in both inputs but not the output
 320   #  * "broadcast" axes are present in exactly one input and the output
 321   #
 322   # As an example, if the einsum is abijk,acjkl->abcil, then "a" is a
 323   # preserved axis, "b" and "c" are broadcast axes, and "j" and "k" are
 324   # summed axes.
 325   assert all(a in t0_axis_labels and a in t1_axis_labels for a in axes_to_sum)
 326   preserved_axes = (set(t0_axis_labels) & set(t1_axis_labels)) - axes_to_sum
 327   broadcast_axes = {}
 328   for i, sym_list in enumerate([t0_axis_labels, t1_axis_labels]):
 329     broadcast_axes[i] = set(sym_list) - preserved_axes - axes_to_sum
 330 
 331   # Reorder the axes so that:
 332   # 1. preserved axes come first in both inputs
 333   # 2. in input 0, broadcast axes come next, followed by summed axes
 334   # 3. in input 1, summed axes come next, followed by broadcast axes
 335   def sort_key(input_index, a):
 336     if a in preserved_axes:
 337       return (-1, a)
 338     elif ((input_index == 0 and a in broadcast_axes[0]) or
 339           (input_index == 1 and a in axes_to_sum)):
 340       return (0, a)
 341     else:
 342       return (1, a)
 343 
 344   axis_labels = [t0_axis_labels, t1_axis_labels]
 345   sorted_axes = [
 346       sorted(sym_list, key=lambda a: sort_key(i, a))
 347       for i, sym_list in enumerate(axis_labels)
 348   ]
 349   inputs = [t0, t1]
 350   for i, axes_str in enumerate(axis_labels):
 351     perm = [axes_str.find(a) for a in sorted_axes[i]]
 352     inputs[i] = _transpose_if_necessary(inputs[i], perm)
 353   t0, t1 = inputs
 354 
 355   if not axes_to_sum:
 356     # In the special case where there are no axes to sum over, reduce to mul()
 357     # rather than to batch matrix multiplication.
 358     for _ in broadcast_axes[1]:
 359       t0 = array_ops.expand_dims(t0, -1)
 360     for _ in broadcast_axes[0]:
 361       t1 = array_ops.expand_dims(t1, len(preserved_axes))
 362     product = math_ops.multiply(t0, t1)
 363     product_axes = sorted_axes[0] + sorted_axes[1][len(preserved_axes):]
 364     return product, ''.join(product_axes)
 365   else:
 366     # Reduce to matmul().
 367 
 368     # Reshape both inputs so as to combine multiple broadcast axes
 369     # into a single axis, and combine multiple summed axes into a
 370     # single axis.
 371 
 372     t0_shape = _get_shape(t0)
 373     num_broadcast_elements_t0 = _total_size(
 374         t0_shape[len(preserved_axes):-len(axes_to_sum)])
 375     num_summed_elements = _total_size(t0_shape[-len(axes_to_sum):])
 376     new_shape = (
 377         t0_shape[:len(preserved_axes)] +
 378         [num_broadcast_elements_t0, num_summed_elements])
 379     t0 = _reshape_if_necessary(t0, new_shape)
 380 
 381     t1_shape = _get_shape(t1)
 382     num_broadcast_elements_t1 = _total_size(
 383         t1_shape[len(preserved_axes) + len(axes_to_sum):])
 384     new_shape = (
 385         t1_shape[:len(preserved_axes)] +
 386         [num_summed_elements, num_broadcast_elements_t1])
 387     t1 = _reshape_if_necessary(t1, new_shape)
 388 
 389     product = math_ops.matmul(t0, t1)
 390 
 391     # Undo compaction of broadcast axes
 392     uncompacted_shape = (
 393         t0_shape[:len(preserved_axes) + len(broadcast_axes[0])] +
 394         t1_shape[len(t1_shape) - len(broadcast_axes[1]):])
 395     product = _reshape_if_necessary(product, uncompacted_shape)
 396 
 397     product_axes = (
 398         sorted_axes[0][:len(preserved_axes) + len(broadcast_axes[0])] +
 399         sorted_axes[1][len(sorted_axes[1]) - len(broadcast_axes[1]):])
 400 
 401     return product, ''.join(product_axes)

 404 def _transpose_if_necessary(tensor, perm):
 405   """Like transpose(), but avoids creating a new tensor if possible."""
 406   if perm != range(len(perm)):
 407     return array_ops.transpose(tensor, perm=perm)
 408   else:
 409     return tensor

 412 def _reshape_if_necessary(tensor, new_shape):
 413   """Like reshape(), but avoids creating a new tensor if possible."""
 414   # Accept None as an alias for -1 in new_shape.
 415   new_shape = tuple(-1 if x is None else x for x in new_shape)
 416   cur_shape = tuple(x.value for x in tensor.get_shape())
 417   if (len(new_shape) == len(cur_shape) and
 418       all(d0 == d1 or d1 == -1 for d0, d1 in zip(cur_shape, new_shape))):
 419     return tensor
 420   else:
 421     return array_ops.reshape(tensor, new_shape)

 424 def _get_shape(tensor):
 425   """Like get_shape().as_list(), but explicitly queries the shape of a tensor
 426   if necessary to ensure that the returned value contains no unknown value."""
 427 
 428   shape = tensor.get_shape().as_list()
 429   none_indices = [i for i, d in enumerate(shape) if d is None]
 430   if none_indices:
 431     # Query the shape if shape contains None values
 432     shape_tensor = array_ops.shape(tensor)
 433     for i in none_indices:
 434       shape[i] = shape_tensor[i]
 435   return shape

 438 def _total_size(shape_values):
 439   """Given list of tensor shape values, returns total size.
 440   If shape_values contains tensor values (which are results of
 441   array_ops.shape), then it returns a scalar tensor.
 442   If not, it returns an integer."""
 443 
 444   result = 1
 445   for val in shape_values:
 446     result *= val
 447   return result

 450 def _exponential_space_einsum(equation, *inputs):
 451   """Fallback implementation that supports summing an index over > 2 inputs."""
 452   if '...' in equation:
 453     raise ValueError('Subscripts with ellipses are not yet supported.')
 454 
 455   match = re.match('^([a-zA-Z,]+)(->[a-zA-Z]*)?$', equation)
 456   if not match:
 457     raise ValueError('Indices have incorrect format: %s' % equation)
 458 
 459   inputs = list(inputs)
 460   idx_in = match.group(1).split(',')
 461   idx_all = set(''.join(idx_in))
 462   indices = ''.join(sorted(idx_all))
 463 
 464   if match.group(2):
 465     idx_out = match.group(2)[2:]
 466 
 467   else:
 468     # infer the output subscripts if not given, assume alphabetical order
 469     counts = {ax: 0 for ax in indices}
 470     for axes_ in idx_in:
 471       for ax in axes_:
 472         counts[ax] += 1
 473 
 474     idx_out = ''.join(sorted(ax for ax in indices if counts[ax] == 1))
 475 
 476   if len(idx_in) != len(inputs):
 477     raise ValueError('Expected %d inputs but got %d' % (len(idx_in),
 478                                                         len(inputs)))
 479 
 480   missing_idx = set(idx_out).difference(idx_all)
 481   if missing_idx:
 482     raise ValueError('Unknown output axes: %s' % missing_idx)
 483 
 484   axis_order = {}
 485   for ax in indices:
 486     if ax not in idx_out:
 487       axis_order[ax] = len(axis_order)
 488   for ax in idx_out:
 489     axis_order[ax] = len(axis_order)
 490 
 491   # transpose inputs so axes are in order
 492   for i, (input_, axes_) in enumerate(zip(inputs, idx_in)):
 493     if input_.get_shape().ndims != len(axes_):
 494       raise ValueError(
 495           'Input %d with axes %s has incorrect' \
 496           ' number of dimensions (expected %d, got %d)' % (
 497               i, axes_, len(axes_), input_.get_shape().ndims
 498           )
 499       )
 500 
 501     sorted_idx = sorted(axes_, key=axis_order.get)
 502 
 503     if len(set(axes_)) != len(axes_):
 504       raise ValueError(
 505           'Subscript not supported: an axis appears more than once: %s' % axes_)
 506 
 507     if list(axes_) != sorted_idx:
 508       permuted = [axes_.find(ax) for ax in sorted_idx]
 509       inputs[i] = array_ops.transpose(input_, permuted)
 510       idx_in[i] = sorted_idx
 511 
 512   reduction_idx = []
 513   shapes = [[dim if dim else -1
 514              for dim in tensor.get_shape().as_list()]
 515             for tensor in inputs]
 516 
 517   # validate shapes for broadcasting
 518   for j, ax in enumerate(sorted(idx_all, key=axis_order.get)):
 519     dims = []
 520     for i, idx in enumerate(idx_in):
 521       if ax not in idx:
 522         shapes[i].insert(j, 1)
 523       else:
 524         dim = shapes[i][j]
 525         if isinstance(dim, int) and dim > 1:
 526           dims.append(dim)
 527 
 528     if len(set(dims)) > 1:
 529       raise ValueError('Dimension mismatch on axis: %s' % ax)
 530 
 531     if ax not in idx_out:
 532       reduction_idx.append(j)
 533 
 534   # reshape, multiply
 535   expanded_inputs = [
 536       array_ops.reshape(input_, shape) for input_, shape in zip(inputs, shapes)
 537   ]
 538   expanded_output = 1
 539   for input_ in expanded_inputs:
 540     expanded_output *= input_
 541 
 542   # contract
 543   return math_ops.reduce_sum(expanded_output, reduction_idx)
