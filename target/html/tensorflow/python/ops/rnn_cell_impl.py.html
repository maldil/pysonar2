  65 def assert_like_rnncell(cell_name, cell):
  66   """Raises a TypeError if cell is not like an RNNCell.
  67 
  68   NOTE: Do not rely on the error message (in particular in tests) which can be
  69   subject to change to increase readability. Use
  70   ASSERT_LIKE_RNNCELL_ERROR_REGEXP.
  71 
  72   Args:
  73     cell_name: A string to give a meaningful error referencing to the name
  74       of the functionargument.
  75     cell: The object which should behave like an RNNCell.
  76 
  77   Raises:
  78     TypeError: A human-friendly exception.
  79   """
  80   conditions = [
  81       hasattr(cell, "output_size"),
  82       hasattr(cell, "state_size"),
  83       hasattr(cell, "get_initial_state") or hasattr(cell, "zero_state"),
  84       callable(cell),
  85   ]
  86   errors = [
  87       "'output_size' property is missing",
  88       "'state_size' property is missing",
  89       "either 'zero_state' or 'get_initial_state' method is required",
  90       "is not callable"
  91   ]
  92 
  93   if not all(conditions):
  94 
  95     errors = [error for error, cond in zip(errors, conditions) if not cond]
  96     raise TypeError("The argument {!r} ({}) is not an RNNCell: {}.".format(
  97         cell_name, cell, ", ".join(errors)))

 100 def _concat(prefix, suffix, static=False):
 101   """Concat that enables int, Tensor, or TensorShape values.
 102 
 103   This function takes a size specification, which can be an integer, a
 104   TensorShape, or a Tensor, and converts it into a concatenated Tensor
 105   (if static = False) or a list of integers (if static = True).
 106 
 107   Args:
 108     prefix: The prefix; usually the batch size (and/or time step size).
 109       (TensorShape, int, or Tensor.)
 110     suffix: TensorShape, int, or Tensor.
 111     static: If `True`, return a python list with possibly unknown dimensions.
 112       Otherwise return a `Tensor`.
 113 
 114   Returns:
 115     shape: the concatenation of prefix and suffix.
 116 
 117   Raises:
 118     ValueError: if `suffix` is not a scalar or vector (or TensorShape).
 119     ValueError: if prefix or suffix was `None` and asked for dynamic
 120       Tensors out.
 121   """
 122   if isinstance(prefix, ops.Tensor):
 123     p = prefix
 124     p_static = tensor_util.constant_value(prefix)
 125     if p.shape.ndims == 0:
 126       p = array_ops.expand_dims(p, 0)
 127     elif p.shape.ndims != 1:
 128       raise ValueError("prefix tensor must be either a scalar or vector, "
 129                        "but saw tensor: %s" % p)
 130   else:
 131     p = tensor_shape.as_shape(prefix)
 132     p_static = p.as_list() if p.ndims is not None else None
 133     p = (constant_op.constant(p.as_list(), dtype=dtypes.int32)
 134          if p.is_fully_defined() else None)
 135   if isinstance(suffix, ops.Tensor):
 136     s = suffix
 137     s_static = tensor_util.constant_value(suffix)
 138     if s.shape.ndims == 0:
 139       s = array_ops.expand_dims(s, 0)
 140     elif s.shape.ndims != 1:
 141       raise ValueError("suffix tensor must be either a scalar or vector, "
 142                        "but saw tensor: %s" % s)
 143   else:
 144     s = tensor_shape.as_shape(suffix)
 145     s_static = s.as_list() if s.ndims is not None else None
 146     s = (constant_op.constant(s.as_list(), dtype=dtypes.int32)
 147          if s.is_fully_defined() else None)
 148 
 149   if static:
 150     shape = tensor_shape.as_shape(p_static).concatenate(s_static)
 151     shape = shape.as_list() if shape.ndims is not None else None
 152   else:
 153     if p is None or s is None:
 154       raise ValueError("Provided a prefix or suffix of None: %s and %s"
 155                        % (prefix, suffix))
 156     shape = array_ops.concat((p, s), 0)
 157   return shape

 160 def _zero_state_tensors(state_size, batch_size, dtype):
 161   """Create tensors of zeros based on state_size, batch_size, and dtype."""
 162   def get_state_shape(s):
 163     """Combine s with batch_size to get a proper tensor shape."""
 164     c = _concat(batch_size, s)
 165     size = array_ops.zeros(c, dtype=dtype)
 166     if not context.executing_eagerly():
 167       c_static = _concat(batch_size, s, static=True)
 168       size.set_shape(c_static)
 169     return size
 170   return nest.map_structure(get_state_shape, state_size)

 173 @tf_export("nn.rnn_cell.RNNCell")
 174 class RNNCell(base_layer.Layer):
 175   """Abstract object representing an RNN cell.
 176 
 177   Every `RNNCell` must have the properties below and implement `call` with
 178   the signature `(output, next_state) = call(input, state)`.  The optional
 179   third input argument, `scope`, is allowed for backwards compatibility
 180   purposes; but should be left off for new subclasses.
 181 
 182   This definition of cell differs from the definition used in the literature.
 183   In the literature, 'cell' refers to an object with a single scalar output.
 184   This definition refers to a horizontal array of such units.
 185 
 186   An RNN cell, in the most abstract setting, is anything that has
 187   a state and performs some operation that takes a matrix of inputs.
 188   This operation results in an output matrix with `self.output_size` columns.
 189   If `self.state_size` is an integer, this operation also results in a new
 190   state matrix with `self.state_size` columns.  If `self.state_size` is a
 191   (possibly nested tuple of) TensorShape object(s), then it should return a
 192   matching structure of Tensors having shape `[batch_size].concatenate(s)`
 193   for each `s` in `self.batch_size`.
 194   """
 195 
 196   def __init__(self, trainable=True, name=None, dtype=None, **kwargs):
 197     super(RNNCell, self).__init__(
 198         trainable=trainable, name=name, dtype=dtype, **kwargs)
 199     # Attribute that indicates whether the cell is a TF RNN cell, due the slight
 200     # difference between TF and Keras RNN cell.
 201     self._is_tf_rnn_cell = True
 202 
 203   def __call__(self, inputs, state, scope=None):
 204     """Run this RNN cell on inputs, starting from the given state.
 205 
 206     Args:
 207       inputs: `2-D` tensor with shape `[batch_size, input_size]`.
 208       state: if `self.state_size` is an integer, this should be a `2-D Tensor`
 209         with shape `[batch_size, self.state_size]`.  Otherwise, if
 210         `self.state_size` is a tuple of integers, this should be a tuple
 211         with shapes `[batch_size, s] for s in self.state_size`.
 212       scope: VariableScope for the created subgraph; defaults to class name.
 213 
 214     Returns:
 215       A pair containing:
 216 
 217       - Output: A `2-D` tensor with shape `[batch_size, self.output_size]`.
 218       - New state: Either a single `2-D` tensor, or a tuple of tensors matching
 219         the arity and shapes of `state`.
 220     """
 221     if scope is not None:
 222       with vs.variable_scope(scope,
 223                              custom_getter=self._rnn_get_variable) as scope:
 224         return super(RNNCell, self).__call__(inputs, state, scope=scope)
 225     else:
 226       scope_attrname = "rnncell_scope"
 227       scope = getattr(self, scope_attrname, None)
 228       if scope is None:
 229         scope = vs.variable_scope(vs.get_variable_scope(),
 230                                   custom_getter=self._rnn_get_variable)
 231         setattr(self, scope_attrname, scope)
 232       with scope:
 233         return super(RNNCell, self).__call__(inputs, state)
 234 
 235   def _rnn_get_variable(self, getter, *args, **kwargs):
 236     variable = getter(*args, **kwargs)
 237     if context.executing_eagerly():
 238       trainable = variable._trainable  # pylint: disable=protected-access
 239     else:
 240       trainable = (
 241           variable in tf_variables.trainable_variables() or
 242           (isinstance(variable, tf_variables.PartitionedVariable) and
 243            list(variable)[0] in tf_variables.trainable_variables()))
 244     if trainable and variable not in self._trainable_weights:
 245       self._trainable_weights.append(variable)
 246     elif not trainable and variable not in self._non_trainable_weights:
 247       self._non_trainable_weights.append(variable)
 248     return variable
 249 
 250   @property
 251   def state_size(self):
 252     """size(s) of state(s) used by this cell.
 253 
 254     It can be represented by an Integer, a TensorShape or a tuple of Integers
 255     or TensorShapes.
 256     """
 257     raise NotImplementedError("Abstract method")
 258 
 259   @property
 260   def output_size(self):
 261     """Integer or TensorShape: size of outputs produced by this cell."""
 262     raise NotImplementedError("Abstract method")
 263 
 264   def build(self, _):
 265     # This tells the parent Layer object that it's OK to call
 266     # self.add_variable() inside the call() method.
 267     pass
 268 
 269   def get_initial_state(self, inputs=None, batch_size=None, dtype=None):
 270     if inputs is not None:
 271       # Validate the given batch_size and dtype against inputs if provided.
 272       inputs = ops.convert_to_tensor(inputs, name="inputs")
 273       if batch_size is not None:
 274         if tensor_util.is_tensor(batch_size):
 275           static_batch_size = tensor_util.constant_value(
 276               batch_size, partial=True)
 277         else:
 278           static_batch_size = batch_size
 279         if inputs.shape[0].value != static_batch_size:
 280           raise ValueError(
 281               "batch size from input tensor is different from the "
 282               "input param. Input tensor batch: {}, batch_size: {}".format(
 283                   inputs.shape[0].value, batch_size))
 284 
 285       if dtype is not None and inputs.dtype != dtype:
 286         raise ValueError(
 287             "dtype from input tensor is different from the "
 288             "input param. Input tensor dtype: {}, dtype: {}".format(
 289                 inputs.dtype, dtype))
 290 
 291       batch_size = inputs.shape[0].value or array_ops.shape(inputs)[0]
 292       dtype = inputs.dtype
 293     if None in [batch_size, dtype]:
 294       raise ValueError(
 295           "batch_size and dtype cannot be None while constructing initial "
 296           "state: batch_size={}, dtype={}".format(batch_size, dtype))
 297     return self.zero_state(batch_size, dtype)
 298 
 299   def zero_state(self, batch_size, dtype):
 300     """Return zero-filled state tensor(s).
 301 
 302     Args:
 303       batch_size: int, float, or unit Tensor representing the batch size.
 304       dtype: the data type to use for the state.
 305 
 306     Returns:
 307       If `state_size` is an int or TensorShape, then the return value is a
 308       `N-D` tensor of shape `[batch_size, state_size]` filled with zeros.
 309 
 310       If `state_size` is a nested list or tuple, then the return value is
 311       a nested list or tuple (of the same structure) of `2-D` tensors with
 312       the shapes `[batch_size, s]` for each s in `state_size`.
 313     """
 314     # Try to use the last cached zero_state. This is done to avoid recreating
 315     # zeros, especially when eager execution is enabled.
 316     state_size = self.state_size
 317     is_eager = context.executing_eagerly()
 318     if is_eager and hasattr(self, "_last_zero_state"):
 319       (last_state_size, last_batch_size, last_dtype,
 320        last_output) = getattr(self, "_last_zero_state")
 321       if (last_batch_size == batch_size and
 322           last_dtype == dtype and
 323           last_state_size == state_size):
 324         return last_output
 325     with ops.name_scope(type(self).__name__ + "ZeroState", values=[batch_size]):
 326       output = _zero_state_tensors(state_size, batch_size, dtype)
 327     if is_eager:
 328       self._last_zero_state = (state_size, batch_size, dtype, output)
 329     return output

 332 class LayerRNNCell(RNNCell):
 333   """Subclass of RNNCells that act like proper `tf.Layer` objects.
 334 
 335   For backwards compatibility purposes, most `RNNCell` instances allow their
 336   `call` methods to instantiate variables via `tf.get_variable`.  The underlying
 337   variable scope thus keeps track of any variables, and returning cached
 338   versions.  This is atypical of `tf.layer` objects, which separate this
 339   part of layer building into a `build` method that is only called once.
 340 
 341   Here we provide a subclass for `RNNCell` objects that act exactly as
 342   `Layer` objects do.  They must provide a `build` method and their
 343   `call` methods do not access Variables `tf.get_variable`.
 344   """
 345 
 346   def __call__(self, inputs, state, scope=None, *args, **kwargs):
 347     """Run this RNN cell on inputs, starting from the given state.
 348 
 349     Args:
 350       inputs: `2-D` tensor with shape `[batch_size, input_size]`.
 351       state: if `self.state_size` is an integer, this should be a `2-D Tensor`
 352         with shape `[batch_size, self.state_size]`.  Otherwise, if
 353         `self.state_size` is a tuple of integers, this should be a tuple
 354         with shapes `[batch_size, s] for s in self.state_size`.
 355       scope: optional cell scope.
 356       *args: Additional positional arguments.
 357       **kwargs: Additional keyword arguments.
 358 
 359     Returns:
 360       A pair containing:
 361 
 362       - Output: A `2-D` tensor with shape `[batch_size, self.output_size]`.
 363       - New state: Either a single `2-D` tensor, or a tuple of tensors matching
 364         the arity and shapes of `state`.
 365     """
 366     # Bypass RNNCell's variable capturing semantics for LayerRNNCell.
 367     # Instead, it is up to subclasses to provide a proper build
 368     # method.  See the class docstring for more details.
 369     return base_layer.Layer.__call__(self, inputs, state, scope=scope,
 370                                      *args, **kwargs)

 373 @tf_export(v1=["nn.rnn_cell.BasicRNNCell"])
 374 class BasicRNNCell(LayerRNNCell):
 375   """The most basic RNN cell.
 376 
 377   Note that this cell is not optimized for performance. Please use
 378   `tf.contrib.cudnn_rnn.CudnnRNNTanh` for better performance on GPU.
 379 
 380   Args:
 381     num_units: int, The number of units in the RNN cell.
 382     activation: Nonlinearity to use.  Default: `tanh`. It could also be string
 383       that is within Keras activation function names.
 384     reuse: (optional) Python boolean describing whether to reuse variables
 385      in an existing scope.  If not `True`, and the existing scope already has
 386      the given variables, an error is raised.
 387     name: String, the name of the layer. Layers with the same name will
 388       share weights, but to avoid mistakes we require reuse=True in such
 389       cases.
 390     dtype: Default dtype of the layer (default of `None` means use the type
 391       of the first input). Required when `build` is called before `call`.
 392     **kwargs: Dict, keyword named properties for common layer attributes, like
 393       `trainable` etc when constructing the cell from configs of get_config().
 394   """
 395 
 396   @deprecated(None, "This class is equivalent as tf.keras.layers.SimpleRNNCell,"
 397                     " and will be replaced by that in Tensorflow 2.0.")
 398   def __init__(self,
 399                num_units,
 400                activation=None,
 401                reuse=None,
 402                name=None,
 403                dtype=None,
 404                **kwargs):
 405     super(BasicRNNCell, self).__init__(
 406         _reuse=reuse, name=name, dtype=dtype, **kwargs)
 407     if context.executing_eagerly() and context.num_gpus() > 0:
 408       logging.warn("%s: Note that this cell is not optimized for performance. "
 409                    "Please use tf.contrib.cudnn_rnn.CudnnRNNTanh for better "
 410                    "performance on GPU.", self)
 411 
 412     # Inputs must be 2-dimensional.
 413     self.input_spec = base_layer.InputSpec(ndim=2)
 414 
 415     self._num_units = num_units
 416     if activation:
 417       self._activation = activations.get(activation)
 418     else:
 419       self._activation = math_ops.tanh
 420 
 421   @property
 422   def state_size(self):
 423     return self._num_units
 424 
 425   @property
 426   def output_size(self):
 427     return self._num_units
 428 
 429   @tf_utils.shape_type_conversion
 430   def build(self, inputs_shape):
 431     if inputs_shape[-1] is None:
 432       raise ValueError("Expected inputs.shape[-1] to be known, saw shape: %s"
 433                        % str(inputs_shape))
 434 
 435     input_depth = inputs_shape[-1]
 436     self._kernel = self.add_variable(
 437         _WEIGHTS_VARIABLE_NAME,
 438         shape=[input_depth + self._num_units, self._num_units])
 439     self._bias = self.add_variable(
 440         _BIAS_VARIABLE_NAME,
 441         shape=[self._num_units],
 442         initializer=init_ops.zeros_initializer(dtype=self.dtype))
 443 
 444     self.built = True
 445 
 446   def call(self, inputs, state):
 447     """Most basic RNN: output = new_state = act(W * input + U * state + B)."""
 448 
 449     gate_inputs = math_ops.matmul(
 450         array_ops.concat([inputs, state], 1), self._kernel)
 451     gate_inputs = nn_ops.bias_add(gate_inputs, self._bias)
 452     output = self._activation(gate_inputs)
 453     return output, output
 454 
 455   def get_config(self):
 456     config = {
 457         "num_units": self._num_units,
 458         "activation": activations.serialize(self._activation),
 459         "reuse": self._reuse,
 460     }
 461     base_config = super(BasicRNNCell, self).get_config()
 462     return dict(list(base_config.items()) + list(config.items()))

 465 @tf_export("nn.rnn_cell.GRUCell")
 466 class GRUCell(LayerRNNCell):
 467   """Gated Recurrent Unit cell (cf. http://arxiv.org/abs/1406.1078).
 468 
 469   Note that this cell is not optimized for performance. Please use
 470   `tf.contrib.cudnn_rnn.CudnnGRU` for better performance on GPU, or
 471   `tf.contrib.rnn.GRUBlockCellV2` for better performance on CPU.
 472 
 473   Args:
 474     num_units: int, The number of units in the GRU cell.
 475     activation: Nonlinearity to use.  Default: `tanh`.
 476     reuse: (optional) Python boolean describing whether to reuse variables
 477      in an existing scope.  If not `True`, and the existing scope already has
 478      the given variables, an error is raised.
 479     kernel_initializer: (optional) The initializer to use for the weight and
 480     projection matrices.
 481     bias_initializer: (optional) The initializer to use for the bias.
 482     name: String, the name of the layer. Layers with the same name will
 483       share weights, but to avoid mistakes we require reuse=True in such
 484       cases.
 485     dtype: Default dtype of the layer (default of `None` means use the type
 486       of the first input). Required when `build` is called before `call`.
 487     **kwargs: Dict, keyword named properties for common layer attributes, like
 488       `trainable` etc when constructing the cell from configs of get_config().
 489   """
 490 
 491   def __init__(self,
 492                num_units,
 493                activation=None,
 494                reuse=None,
 495                kernel_initializer=None,
 496                bias_initializer=None,
 497                name=None,
 498                dtype=None,
 499                **kwargs):
 500     super(GRUCell, self).__init__(
 501         _reuse=reuse, name=name, dtype=dtype, **kwargs)
 502 
 503     if context.executing_eagerly() and context.num_gpus() > 0:
 504       logging.warn("%s: Note that this cell is not optimized for performance. "
 505                    "Please use tf.contrib.cudnn_rnn.CudnnGRU for better "
 506                    "performance on GPU.", self)
 507     # Inputs must be 2-dimensional.
 508     self.input_spec = base_layer.InputSpec(ndim=2)
 509 
 510     self._num_units = num_units
 511     if activation:
 512       self._activation = activations.get(activation)
 513     else:
 514       self._activation = math_ops.tanh
 515     self._kernel_initializer = initializers.get(kernel_initializer)
 516     self._bias_initializer = initializers.get(bias_initializer)
 517 
 518   @property
 519   def state_size(self):
 520     return self._num_units
 521 
 522   @property
 523   def output_size(self):
 524     return self._num_units
 525 
 526   @tf_utils.shape_type_conversion
 527   def build(self, inputs_shape):
 528     if inputs_shape[-1] is None:
 529       raise ValueError("Expected inputs.shape[-1] to be known, saw shape: %s"
 530                        % str(inputs_shape))
 531 
 532     input_depth = inputs_shape[-1]
 533     self._gate_kernel = self.add_variable(
 534         "gates/%s" % _WEIGHTS_VARIABLE_NAME,
 535         shape=[input_depth + self._num_units, 2 * self._num_units],
 536         initializer=self._kernel_initializer)
 537     self._gate_bias = self.add_variable(
 538         "gates/%s" % _BIAS_VARIABLE_NAME,
 539         shape=[2 * self._num_units],
 540         initializer=(
 541             self._bias_initializer
 542             if self._bias_initializer is not None
 543             else init_ops.constant_initializer(1.0, dtype=self.dtype)))
 544     self._candidate_kernel = self.add_variable(
 545         "candidate/%s" % _WEIGHTS_VARIABLE_NAME,
 546         shape=[input_depth + self._num_units, self._num_units],
 547         initializer=self._kernel_initializer)
 548     self._candidate_bias = self.add_variable(
 549         "candidate/%s" % _BIAS_VARIABLE_NAME,
 550         shape=[self._num_units],
 551         initializer=(
 552             self._bias_initializer
 553             if self._bias_initializer is not None
 554             else init_ops.zeros_initializer(dtype=self.dtype)))
 555 
 556     self.built = True
 557 
 558   def call(self, inputs, state):
 559     """Gated recurrent unit (GRU) with nunits cells."""
 560 
 561     gate_inputs = math_ops.matmul(
 562         array_ops.concat([inputs, state], 1), self._gate_kernel)
 563     gate_inputs = nn_ops.bias_add(gate_inputs, self._gate_bias)
 564 
 565     value = math_ops.sigmoid(gate_inputs)
 566     r, u = array_ops.split(value=value, num_or_size_splits=2, axis=1)
 567 
 568     r_state = r * state
 569 
 570     candidate = math_ops.matmul(
 571         array_ops.concat([inputs, r_state], 1), self._candidate_kernel)
 572     candidate = nn_ops.bias_add(candidate, self._candidate_bias)
 573 
 574     c = self._activation(candidate)
 575     new_h = u * state + (1 - u) * c
 576     return new_h, new_h

 582         "bias_initializer": initializers.serialize(self._bias_initializer),
 583         "activation": activations.serialize(self._activation),
 584         "reuse": self._reuse,

 741     # Parameters of gates are concatenated into one multiply for efficiency.
 742     if self._state_is_tuple:
 743       c, h = state
 744     else:

 748         array_ops.concat([inputs, h], 1), self._kernel)

 955   def call(self, inputs, state):
 956     """Run one step of LSTM.
 957 
 958     Args:
 959       inputs: input Tensor, must be 2-D, `[batch, input_size]`.
 960       state: if `state_is_tuple` is False, this must be a state Tensor,
 961         `2-D, [batch, state_size]`.  If `state_is_tuple` is True, this must be a
 962         tuple of state Tensors, both `2-D`, with column sizes `c_state` and
 963         `m_state`.
 964 
 965     Returns:
 966       A tuple containing:
 967 
 968       - A `2-D, [batch, output_dim]`, Tensor representing the output of the
 969         LSTM after reading `inputs` when previous state was `state`.
 970         Here output_dim is:
 971            num_proj if num_proj was set,
 972            num_units otherwise.
 973       - Tensor(s) representing the new state of LSTM after reading `inputs` when
 974         the previous state was `state`.  Same type and shape(s) as `state`.
 975 
 976     Raises:
 977       ValueError: If input size cannot be inferred from inputs via
 978         static shape inference.
 979     """
 980     num_proj = self._num_units if self._num_proj is None else self._num_proj
 981     sigmoid = math_ops.sigmoid
 982 
 983     if self._state_is_tuple:
 984       (c_prev, m_prev) = state
 985     else:
 986       c_prev = array_ops.slice(state, [0, 0], [-1, self._num_units])
 987       m_prev = array_ops.slice(state, [0, self._num_units], [-1, num_proj])
 988 
 989     input_size = inputs.get_shape().with_rank(2)[1]
 990     if input_size.value is None:
 991       raise ValueError("Could not infer input size from inputs.get_shape()[-1]")
 992 
 993     # i = input_gate, j = new_input, f = forget_gate, o = output_gate
 994     lstm_matrix = math_ops.matmul(
 995         array_ops.concat([inputs, m_prev], 1), self._kernel)
 996     lstm_matrix = nn_ops.bias_add(lstm_matrix, self._bias)
 997 
 998     i, j, f, o = array_ops.split(
 999         value=lstm_matrix, num_or_size_splits=4, axis=1)
1000     # Diagonal connections
1001     if self._use_peepholes:
1002       c = (sigmoid(f + self._forget_bias + self._w_f_diag * c_prev) * c_prev +
1003            sigmoid(i + self._w_i_diag * c_prev) * self._activation(j))
1004     else:
1005       c = (sigmoid(f + self._forget_bias) * c_prev + sigmoid(i) *
1006            self._activation(j))
1007 
1008     if self._cell_clip is not None:
1009       # pylint: disable=invalid-unary-operand-type
1010       c = clip_ops.clip_by_value(c, -self._cell_clip, self._cell_clip)
1011       # pylint: enable=invalid-unary-operand-type
1012     if self._use_peepholes:
1013       m = sigmoid(o + self._w_o_diag * c) * self._activation(c)
1014     else:
1015       m = sigmoid(o) * self._activation(c)
1016 
1017     if self._num_proj is not None:
1018       m = math_ops.matmul(m, self._proj_kernel)
1019 
1020       if self._proj_clip is not None:
1021         # pylint: disable=invalid-unary-operand-type
1022         m = clip_ops.clip_by_value(m, -self._proj_clip, self._proj_clip)
1023         # pylint: enable=invalid-unary-operand-type
1024 
1025     new_state = (LSTMStateTuple(c, m) if self._state_is_tuple else
1026                  array_ops.concat([c, m], 1))
1027     return m, new_state

1232   def _variational_recurrent_dropout_value(
1233       self, index, value, noise, keep_prob):
1234     """Performs dropout given the pre-calculated noise tensor."""
1235     # uniform [keep_prob, 1.0 + keep_prob)
1236     random_tensor = keep_prob + noise
1237 
1238     # 0. if [keep_prob, 1.0) and 1. if [1.0, 1.0 + keep_prob)
1239     binary_tensor = math_ops.floor(random_tensor)
1240     ret = math_ops.div(value, keep_prob) * binary_tensor
1241     ret.set_shape(value.get_shape())
1242     return ret

1244   def _dropout(self, values, salt_prefix, recurrent_noise, keep_prob,
1245                shallow_filtered_substructure=None):
1246     """Decides whether to perform standard dropout or recurrent dropout."""
1247 
1248     if shallow_filtered_substructure is None:
1249       # Put something so we traverse the entire structure; inside the
1250       # dropout function we check to see if leafs of this are bool or not.
1251       shallow_filtered_substructure = values
1252 
1253     if not self._variational_recurrent:
1254       def dropout(i, do_dropout, v):
1255         if not isinstance(do_dropout, bool) or do_dropout:
1256           return nn_ops.dropout(
1257               v, keep_prob=keep_prob, seed=self._gen_seed(salt_prefix, i))
1258         else:
1259           return v
1260       return _enumerated_map_structure_up_to(
1261           shallow_filtered_substructure, dropout,
1262           *[shallow_filtered_substructure, values])
1263     else:
1264       def dropout(i, do_dropout, v, n):
1265         if not isinstance(do_dropout, bool) or do_dropout:
1266           return self._variational_recurrent_dropout_value(i, v, n, keep_prob)
1267         else:
1268           return v
1269       return _enumerated_map_structure_up_to(
1270           shallow_filtered_substructure, dropout,
1271           *[shallow_filtered_substructure, values, recurrent_noise])

1278     if _should_dropout(self._input_keep_prob):
