  17 Useful reference for derivative formulas is
  18 An extended collection of matrix derivative results for forward and reverse
  19 mode algorithmic differentiation by Mike Giles:

  38 @ops.RegisterGradient("MatrixInverse")
  39 def _MatrixInverseGrad(op, grad):
  40   """Gradient for MatrixInverse."""
  41   ainv = op.outputs[0]
  42   return -math_ops.matmul(
  43       ainv, math_ops.matmul(grad, ainv, adjoint_b=True), adjoint_a=True)

  46 @ops.RegisterGradient("MatrixDeterminant")
  47 def _MatrixDeterminantGrad(op, grad):
  48   """Gradient for MatrixDeterminant."""
  49   a = op.inputs[0]
  50   c = op.outputs[0]
  51   a_adj_inv = linalg_ops.matrix_inverse(a, adjoint=True)
  52   multipliers = array_ops.reshape(grad * c,
  53                                   array_ops.concat([array_ops.shape(c), [1, 1]],
  54                                                    0))
  55   return multipliers * a_adj_inv

  58 @ops.RegisterGradient("LogMatrixDeterminant")
  59 def _LogMatrixDeterminantGrad(op, _, grad_b):
  60   """Gradient for LogMatrixDeterminant."""
  61   a = op.inputs[0]
  62   c = op.outputs[1]
  63   a_adj_inv = linalg_ops.matrix_inverse(a, adjoint=True)
  64   multipliers = array_ops.reshape(
  65       grad_b, array_ops.concat([array_ops.shape(c), [1, 1]], 0))
  66   return multipliers * a_adj_inv

  69 @ops.RegisterGradient("Cholesky")
  70 def _CholeskyGrad(op, grad):
  71   """Gradient for Cholesky."""
  72 
  73   # Gradient is l^{-H} @ ((l^{H} @ grad) * (tril(ones)-1/2*eye)) @ l^{-1}
  74   l = op.outputs[0]
  75   num_rows = array_ops.shape(l)[-1]
  76   batch_shape = array_ops.shape(l)[:-2]
  77   l_inverse = linalg_ops.matrix_triangular_solve(l,
  78                                                  linalg_ops.eye(
  79                                                      num_rows,
  80                                                      batch_shape=batch_shape,
  81                                                      dtype=l.dtype))
  82 
  83   middle = math_ops.matmul(l, grad, adjoint_a=True)
  84   middle = array_ops.matrix_set_diag(middle,
  85                                      0.5 * array_ops.matrix_diag_part(middle))
  86   middle = array_ops.matrix_band_part(middle, -1, 0)
  87 
  88   grad_a = math_ops.matmul(
  89       math_ops.matmul(l_inverse, middle, adjoint_a=True), l_inverse)
  90 
  91   grad_a += _linalg.adjoint(grad_a)
  92   return grad_a * 0.5

  95 @ops.RegisterGradient("Qr")
  96 def _QrGrad(op, dq, dr):
  97   """Gradient for Qr."""
  98   q, r = op.outputs
  99   if q.dtype.is_complex:
 100     raise NotImplementedError("QrGrad not implemented for dtype: %s" % q.dtype)
 101   if (r.shape.ndims is None or r.shape.as_list()[-2] is None or
 102       r.shape.as_list()[-1] is None):
 103     raise NotImplementedError("QrGrad not implemented with dynamic shapes.")
 104   if r.shape[-2].value != r.shape[-1].value:
 105     raise NotImplementedError("QrGrad not implemented when ncols > nrows "
 106                               "or full_matrices is true and ncols != nrows.")
 107 
 108   qdq = math_ops.matmul(q, dq, adjoint_a=True)
 109   qdq_ = qdq - _linalg.adjoint(qdq)
 110   rdr = math_ops.matmul(r, dr, adjoint_b=True)
 111   rdr_ = rdr - _linalg.adjoint(rdr)
 112   tril = array_ops.matrix_band_part(qdq_ + rdr_, -1, 0)
 113 
 114   def _TriangularSolve(x, r):
 115     """Equiv to matmul(x, adjoint(matrix_inverse(r))) if r is upper-tri."""
 116     return _linalg.adjoint(
 117         linalg_ops.matrix_triangular_solve(
 118             r, _linalg.adjoint(x), lower=False, adjoint=False))
 119 
 120   grad_a = math_ops.matmul(q, dr + _TriangularSolve(tril, r))
 121   grad_b = _TriangularSolve(dq - math_ops.matmul(q, qdq), r)
 122   return grad_a + grad_b

 125 @ops.RegisterGradient("MatrixSolve")
 126 def _MatrixSolveGrad(op, grad):
 127   """Gradient for MatrixSolve."""
 128   a = op.inputs[0]
 129   adjoint_a = op.get_attr("adjoint")
 130   c = op.outputs[0]
 131   grad_b = linalg_ops.matrix_solve(a, grad, adjoint=not adjoint_a)
 132   if adjoint_a:
 133     grad_a = -math_ops.matmul(c, grad_b, adjoint_b=True)
 134   else:
 135     grad_a = -math_ops.matmul(grad_b, c, adjoint_b=True)
 136   return (grad_a, grad_b)

 139 @ops.RegisterGradient("MatrixSolveLs")
 140 def _MatrixSolveLsGrad(op, grad):
 141   """Gradients for MatrixSolveLs."""
 142 
 143   # TODO(rmlarsen): The implementation could be more efficient:
 144   #   a) Output the Cholesky factorization from forward op instead of
 145   #      recomputing it here.
 146   #   b) Implement a symmetric rank-k update op instead of computing
 147   #      x*z + transpose(x*z). This pattern occurs other places in TensorFlow.
 148 
 149   def _Overdetermined(op, grad):
 150     """Gradients for the overdetermined case of MatrixSolveLs.
 151 
 152     This is the backprop for the solution to the normal equations of the first
 153     kind:
 154        X = F(A, B) = (A^T * A + lambda * I)^{-1} * A^T * B
 155     which solve the least squares problem
 156        min ||A * X - B||_F^2 + lambda ||X||_F^2.
 157     """
 158     a = op.inputs[0]
 159     b = op.inputs[1]
 160     x = op.outputs[0]
 161     l2_regularizer = math_ops.cast(op.inputs[2], a.dtype.base_dtype)
 162     # pylint: disable=protected-access
 163     chol = linalg_ops._RegularizedGramianCholesky(
 164         a, l2_regularizer=l2_regularizer, first_kind=True)
 165     # pylint: enable=protected-access
 166     # Temporary z = (A^T * A + lambda * I)^{-1} * grad.
 167     z = linalg_ops.cholesky_solve(chol, grad)
 168     xzt = math_ops.matmul(x, z, adjoint_b=True)
 169     zx_sym = xzt + array_ops.matrix_transpose(xzt)
 170     grad_a = -math_ops.matmul(a, zx_sym) + math_ops.matmul(b, z, adjoint_b=True)
 171     grad_b = math_ops.matmul(a, z)
 172     return (grad_a, grad_b, None)
 173 
 174   def _Underdetermined(op, grad):
 175     """Gradients for the underdetermined case of MatrixSolveLs.
 176 
 177     This is the backprop for the solution to the normal equations of the second
 178     kind:
 179       X = F(A, B) = A * (A*A^T + lambda*I)^{-1} * B
 180     that (for lambda=0) solve the least squares problem
 181       min ||X||_F subject to A*X = B.
 182     """
 183     a = op.inputs[0]
 184     b = op.inputs[1]
 185     l2_regularizer = math_ops.cast(op.inputs[2], a.dtype.base_dtype)
 186     # pylint: disable=protected-access
 187     chol = linalg_ops._RegularizedGramianCholesky(
 188         a, l2_regularizer=l2_regularizer, first_kind=False)
 189     # pylint: enable=protected-access
 190     grad_b = linalg_ops.cholesky_solve(chol, math_ops.matmul(a, grad))
 191     # Temporary tmp = (A * A^T + lambda * I)^{-1} * B.
 192     tmp = linalg_ops.cholesky_solve(chol, b)
 193     a1 = math_ops.matmul(tmp, a, adjoint_a=True)
 194     a1 = -math_ops.matmul(grad_b, a1)
 195     a2 = grad - math_ops.matmul(a, grad_b, adjoint_a=True)
 196     a2 = math_ops.matmul(tmp, a2, adjoint_b=True)
 197     grad_a = a1 + a2
 198     return (grad_a, grad_b, None)
 199 
 200   fast = op.get_attr("fast")
 201   if fast is False:
 202     raise ValueError("Gradient not defined for fast=False")
 203   matrix_shape = op.inputs[0].get_shape()[-2:]
 204   if matrix_shape.is_fully_defined():
 205     if matrix_shape[-2] >= matrix_shape[-1]:
 206       return _Overdetermined(op, grad)
 207     else:
 208       return _Underdetermined(op, grad)
 209   else:
 210     # We have to defer determining the shape to runtime and use
 211     # conditional execution of the appropriate graph.
 212     matrix_shape = array_ops.shape(op.inputs[0])[-2:]
 213     return control_flow_ops.cond(matrix_shape[-2] >= matrix_shape[-1],
 214                                  lambda: _Overdetermined(op, grad),
 215                                  lambda: _Underdetermined(op, grad))

 218 @ops.RegisterGradient("MatrixTriangularSolve")
 219 def _MatrixTriangularSolveGrad(op, grad):
 220   """Gradient for MatrixTriangularSolve."""
 221   a = op.inputs[0]
 222   adjoint_a = op.get_attr("adjoint")
 223   lower_a = op.get_attr("lower")
 224   c = op.outputs[0]
 225   grad_b = linalg_ops.matrix_triangular_solve(
 226       a, grad, lower=lower_a, adjoint=not adjoint_a)
 227   if adjoint_a:
 228     grad_a = -math_ops.matmul(c, grad_b, adjoint_b=True)
 229   else:
 230     grad_a = -math_ops.matmul(grad_b, c, adjoint_b=True)
 231   if lower_a:
 232     grad_a = array_ops.matrix_band_part(grad_a, -1, 0)
 233   else:
 234     grad_a = array_ops.matrix_band_part(grad_a, 0, -1)
 235   return (grad_a, grad_b)

 238 @ops.RegisterGradient("SelfAdjointEigV2")
 239 def _SelfAdjointEigV2Grad(op, grad_e, grad_v):
 240   """Gradient for SelfAdjointEigV2."""
 241   e = op.outputs[0]
 242   compute_v = op.get_attr("compute_v")
 243   # a = op.inputs[0], which satisfies
 244   # a[...,:,:] * v[...,:,i] = e[...,i] * v[...,i]
 245   with ops.control_dependencies([grad_e, grad_v]):
 246     if compute_v:
 247       v = op.outputs[1]
 248       # Construct the matrix f(i,j) = (i != j ? 1 / (e_i - e_j) : 0).
 249       # Notice that because of the term involving f, the gradient becomes
 250       # infinite (or NaN in practice) when eigenvalues are not unique.
 251       # Mathematically this should not be surprising, since for (k-fold)
 252       # degenerate eigenvalues, the corresponding eigenvectors are only defined
 253       # up to arbitrary rotation in a (k-dimensional) subspace.
 254       f = array_ops.matrix_set_diag(
 255           math_ops.reciprocal(
 256               array_ops.expand_dims(e, -2) - array_ops.expand_dims(e, -1)),
 257           array_ops.zeros_like(e))
 258       grad_a = math_ops.matmul(
 259           v,
 260           math_ops.matmul(
 261               array_ops.matrix_diag(grad_e) +
 262               f * math_ops.matmul(v, grad_v, adjoint_a=True),
 263               v,
 264               adjoint_b=True))
 265     else:
 266       _, v = linalg_ops.self_adjoint_eig(op.inputs[0])
 267       grad_a = math_ops.matmul(v,
 268                                math_ops.matmul(
 269                                    array_ops.matrix_diag(grad_e),
 270                                    v,
 271                                    adjoint_b=True))
 272     # The forward op only depends on the lower triangular part of a, so here we
 273     # symmetrize and take the lower triangle
 274     grad_a = array_ops.matrix_band_part(grad_a + _linalg.adjoint(grad_a), -1, 0)
 275     grad_a = array_ops.matrix_set_diag(grad_a,
 276                                        0.5 * array_ops.matrix_diag_part(grad_a))
 277     return grad_a

 280 @ops.RegisterGradient("Svd")
 281 def _SvdGrad(op, grad_s, grad_u, grad_v):
 282   """Gradient for the singular value decomposition."""
 283 
 284   # The derivation for the compute_uv=False case, and most of
 285   # the derivation for the full_matrices=True case, are in
 286   # Giles' paper (see reference at top of file).  A derivation for
 287   # the full_matrices=False case is available at
 288   # https://j-towns.github.io/papers/svd-derivative.pdf
 289   a = op.inputs[0]
 290   a_shape = a.get_shape().with_rank_at_least(2)
 291   grad_s_mat = array_ops.matrix_diag(grad_s)
 292 
 293   if not op.get_attr("compute_uv"):
 294     s, u, v = linalg_ops.svd(a, compute_uv=True)
 295     grad_a = math_ops.matmul(u, math_ops.matmul(grad_s_mat, v, adjoint_b=True))
 296     grad_a.set_shape(a_shape)
 297     return grad_a
 298 
 299   full_matrices = op.get_attr("full_matrices")
 300 
 301   # TODO(rmlarsen): Make this work with complex types.
 302   if a.dtype.is_complex:
 303     raise NotImplementedError(
 304         "SVD gradient is not implemented for complex types and "
 305         "compute_uv=True.")
 306   grad_u_shape = grad_u.get_shape().with_rank_at_least(2)
 307   grad_v_shape = grad_v.get_shape().with_rank_at_least(2)
 308   m = a_shape[-2].merge_with(grad_u_shape[-2])
 309   n = a_shape[-1].merge_with(grad_v_shape[-2])
 310   batch_shape = a_shape[:-2].merge_with(grad_u_shape[:-2]).merge_with(
 311       grad_v_shape[:-2])
 312   a_shape = batch_shape.concatenate([m, n])
 313 
 314   m = a_shape[-2].value
 315   n = a_shape[-1].value
 316   # TODO(rmlarsen): Make this work with placeholders.
 317   if m is None or n is None:
 318     raise NotImplementedError(
 319         "SVD gradient has not been implemented for input with unknown "
 320         "inner matrix shape.")
 321 
 322   s = op.outputs[0]
 323   u = op.outputs[1]
 324   v = op.outputs[2]
 325 
 326   use_adjoint = False
 327   if m > n:
 328     # Compute the gradient for A^H = V * S^T * U^H, and (implicitly) take the
 329     # Hermitian transpose of the gradient at the end.
 330     use_adjoint = True
 331     m, n = n, m
 332     u, v = v, u
 333     grad_u, grad_v = grad_v, grad_u
 334 
 335   with ops.control_dependencies([grad_s, grad_u, grad_v]):
 336     if full_matrices and abs(m - n) > 1:
 337       raise NotImplementedError(
 338           "svd gradient is not implemented for abs(m - n) > 1 "
 339           "when full_matrices is True")
 340     s_mat = array_ops.matrix_diag(s)
 341     s2 = math_ops.square(s)
 342 
 343     # NOTICE: Because of the term involving f, the gradient becomes
 344     # infinite (or NaN in practice) when singular values are not unique.
 345     # Mathematically this should not be surprising, since for (k-fold)
 346     # degenerate singular values, the corresponding singular vectors are
 347     # only defined up a (k-dimensional) subspace. In practice, this can
 348     # lead to numerical instability when singular values are close but not
 349     # exactly equal.
 350     f = array_ops.matrix_set_diag(
 351         math_ops.reciprocal(
 352             array_ops.expand_dims(s2, -2) - array_ops.expand_dims(s2, -1)),
 353         array_ops.zeros_like(s))
 354     s_inv_mat = array_ops.matrix_diag(math_ops.reciprocal(s))
 355 
 356     v1 = v[..., :, :m]
 357     grad_v1 = grad_v[..., :, :m]
 358 
 359     u_gu = math_ops.matmul(u, grad_u, adjoint_a=True)
 360     v_gv = math_ops.matmul(v1, grad_v1, adjoint_a=True)
 361 
 362     f_u = f * u_gu
 363     f_v = f * v_gv
 364 
 365     term1_nouv = (
 366         grad_s_mat + math_ops.matmul(f_u + _linalg.adjoint(f_u), s_mat) +
 367         math_ops.matmul(s_mat, f_v + _linalg.adjoint(f_v)))
 368 
 369     term1 = math_ops.matmul(u, math_ops.matmul(term1_nouv, v1, adjoint_b=True))
 370 
 371     if m == n:
 372       grad_a_before_transpose = term1
 373     else:
 374       gv1t = array_ops.matrix_transpose(grad_v1)
 375       gv1t_v1 = math_ops.matmul(gv1t, v1)
 376       term2_nous = gv1t - math_ops.matmul(gv1t_v1, v1, adjoint_b=True)
 377 
 378       if full_matrices:
 379         v2 = v[..., :, m:n]
 380         grad_v2 = grad_v[..., :, m:n]
 381 
 382         v1t_gv2 = math_ops.matmul(v1, grad_v2, adjoint_a=True)
 383         term2_nous -= math_ops.matmul(v1t_gv2, v2, adjoint_b=True)
 384 
 385       u_s_inv = math_ops.matmul(u, s_inv_mat)
 386       term2 = math_ops.matmul(u_s_inv, term2_nous)
 387 
 388       grad_a_before_transpose = term1 + term2
 389 
 390     if use_adjoint:
 391       grad_a = array_ops.matrix_transpose(grad_a_before_transpose)
 392     else:
 393       grad_a = grad_a_before_transpose
 394 
 395     grad_a.set_shape(a_shape)
 396     return grad_a
