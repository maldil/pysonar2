  27 @ops.RegisterGradient("ResizeNearestNeighbor")
  28 def _ResizeNearestNeighborGrad(op, grad):
  29   """The derivatives for nearest neighbor resizing.
  30 
  31   Args:
  32     op: The ResizeNearestNeighbor op.
  33     grad: The tensor representing the gradient w.r.t. the output.
  34 
  35   Returns:
  36     The gradients w.r.t. the input and the output.
  37   """
  38   image = op.inputs[0]
  39   if image.get_shape()[1:3].is_fully_defined():
  40     image_shape = image.get_shape()[1:3]
  41   else:
  42     image_shape = array_ops.shape(image)[1:3]
  43 
  44   grads = gen_image_ops.resize_nearest_neighbor_grad(
  45       grad,
  46       image_shape,
  47       align_corners=op.get_attr("align_corners"))
  48   return [grads, None]

  51 @ops.RegisterGradient("ResizeBilinear")

  55   Args:

  57     grad: The tensor representing the gradient w.r.t. the output.
  58 
  59   Returns:
  60     The gradients w.r.t. the input.
  61   """

  63       grad, op.inputs[0], align_corners=op.get_attr("align_corners"))
  64   return [grad0, None]

  67 @ops.RegisterGradient("ResizeBicubic")
  68 def _ResizeBicubicGrad(op, grad):
  69   """The derivatives for bicubic resizing.
  70 
  71   Args:
  72     op: The ResizeBicubic op.
  73     grad: The tensor representing the gradient w.r.t. the output.
  74 
  75   Returns:
  76     The gradients w.r.t. the input.
  77   """
  78   allowed_types = [dtypes.float32, dtypes.float64]
  79   grad0 = None
  80   if op.inputs[0].dtype in allowed_types:
  81     grad0 = gen_image_ops.resize_bicubic_grad(
  82         grad, op.inputs[0], align_corners=op.get_attr("align_corners"))
  83   return [grad0, None]

  86 @ops.RegisterGradient("CropAndResize")
  87 def _CropAndResizeGrad(op, grad):
  88   """The derivatives for crop_and_resize.
  89 
  90   We back-propagate to the image only when the input image tensor has floating
  91   point dtype but we always back-propagate to the input boxes tensor.
  92 
  93   Args:
  94     op: The CropAndResize op.
  95     grad: The tensor representing the gradient w.r.t. the output.
  96 
  97   Returns:
  98     The gradients w.r.t. the input image, boxes, as well as the always-None
  99     gradients w.r.t. box_ind and crop_size.
 100   """
 101   image = op.inputs[0]
 102   if image.get_shape().is_fully_defined():
 103     image_shape = image.get_shape().as_list()
 104   else:
 105     image_shape = array_ops.shape(image)
 106 
 107   allowed_types = [dtypes.float16, dtypes.float32, dtypes.float64]
 108   if op.inputs[0].dtype in allowed_types:
 109     # pylint: disable=protected-access
 110     grad0 = gen_image_ops.crop_and_resize_grad_image(
 111         grad, op.inputs[1], op.inputs[2], image_shape, T=op.get_attr("T"),
 112         method=op.get_attr("method"))
 113     # pylint: enable=protected-access
 114   else:
 115     grad0 = None
 116 
 117   # `grad0` is the gradient to the input image pixels and it
 118   # has been implemented for nearest neighbor and bilinear sampling
 119   # respectively. `grad1` is the gradient to the input crop boxes' coordinates.
 120   # When using nearest neighbor sampling, the gradient to crop boxes'
 121   # coordinates are not well defined. In practice, we still approximate
 122   # grad1 using the gradient derived from bilinear sampling.
 123   grad1 = gen_image_ops.crop_and_resize_grad_boxes(
 124       grad, op.inputs[0], op.inputs[1], op.inputs[2])
 125 
 126   return [grad0, grad1, None, None]
