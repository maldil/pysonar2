  41 def _RegularizedGramianCholesky(matrix, l2_regularizer, first_kind):
  42   r"""Computes Cholesky factorization of regularized gramian matrix.
  43 
  44   Below we will use the following notation for each pair of matrix and
  45   right-hand sides in the batch:
  46 
  47   `matrix`=\\(A \in \Re^{m \times n}\\),
  48   `output`=\\(C  \in \Re^{\min(m, n) \times \min(m,n)}\\),
  49   `l2_regularizer`=\\(\lambda\\).
  50 
  51   If `first_kind` is True, returns the Cholesky factorization \\(L\\) such that
  52   \\(L L^H =  A^H A + \lambda I\\).
  53   If `first_kind` is False, returns the Cholesky factorization \\(L\\) such that
  54   \\(L L^H =  A A^H + \lambda I\\).
  55 
  56   Args:
  57     matrix: `Tensor` of shape `[..., M, N]`.
  58     l2_regularizer: 0-D `double` `Tensor`. Ignored if `fast=False`.
  59     first_kind: bool. Controls what gramian matrix to factor.
  60   Returns:
  61     output: `Tensor` of shape `[..., min(M,N), min(M,N)]` whose inner-most 2
  62       dimensions contain the Cholesky factors \\(L\\) described above.
  63   """
  64 
  65   gramian = math_ops.matmul(
  66       matrix, matrix, adjoint_a=first_kind, adjoint_b=not first_kind)
  67   if isinstance(l2_regularizer, ops.Tensor) or l2_regularizer != 0:
  68     matrix_shape = array_ops.shape(matrix)
  69     batch_shape = matrix_shape[:-2]
  70     if first_kind:
  71       small_dim = matrix_shape[-1]
  72     else:
  73       small_dim = matrix_shape[-2]
  74     identity = eye(small_dim, batch_shape=batch_shape, dtype=matrix.dtype)
  75     small_dim_static = matrix.shape[-1 if first_kind else -2]
  76     identity.set_shape(
  77         matrix.shape[:-2].concatenate([small_dim_static, small_dim_static]))
  78     gramian += l2_regularizer * identity
  79   return gen_linalg_ops.cholesky(gramian)

  82 @tf_export('cholesky_solve', 'linalg.cholesky_solve')
  83 @deprecation.deprecated_endpoints('cholesky_solve')
  84 def cholesky_solve(chol, rhs, name=None):
  85   """Solves systems of linear eqns `A X = RHS`, given Cholesky factorizations.
  86 
  87   ```python
  88   # Solve 10 separate 2x2 linear systems:
  89   A = ... # shape 10 x 2 x 2
  90   RHS = ... # shape 10 x 2 x 1
  91   chol = tf.cholesky(A)  # shape 10 x 2 x 2
  92   X = tf.cholesky_solve(chol, RHS)  # shape 10 x 2 x 1
  93   # tf.matmul(A, X) ~ RHS
  94   X[3, :, 0]  # Solution to the linear system A[3, :, :] x = RHS[3, :, 0]
  95 
  96   # Solve five linear systems (K = 5) for every member of the length 10 batch.
  97   A = ... # shape 10 x 2 x 2
  98   RHS = ... # shape 10 x 2 x 5
  99   ...
 100   X[3, :, 2]  # Solution to the linear system A[3, :, :] x = RHS[3, :, 2]
 101   ```
 102 
 103   Args:
 104     chol:  A `Tensor`.  Must be `float32` or `float64`, shape is `[..., M, M]`.
 105       Cholesky factorization of `A`, e.g. `chol = tf.cholesky(A)`.
 106       For that reason, only the lower triangular parts (including the diagonal)
 107       of the last two dimensions of `chol` are used.  The strictly upper part is
 108       assumed to be zero and not accessed.
 109     rhs:  A `Tensor`, same type as `chol`, shape is `[..., M, K]`.
 110     name:  A name to give this `Op`.  Defaults to `cholesky_solve`.
 111 
 112   Returns:
 113     Solution to `A x = rhs`, shape `[..., M, K]`.
 114   """
 115   # To solve C C^* x = rhs, we
 116   # 1. Solve C y = rhs for y, thus y = C^* x
 117   # 2. Solve C^* x = y for x
 118   with ops.name_scope(name, 'cholesky_solve', [chol, rhs]):
 119     y = gen_linalg_ops.matrix_triangular_solve(
 120         chol, rhs, adjoint=False, lower=True)
 121     x = gen_linalg_ops.matrix_triangular_solve(
 122         chol, y, adjoint=True, lower=True)
 123     return x

 171 @tf_export('linalg.lstsq', 'matrix_solve_ls')
 172 @deprecation.deprecated_endpoints('matrix_solve_ls')
 173 def matrix_solve_ls(matrix, rhs, l2_regularizer=0.0, fast=True, name=None):
 174   r"""Solves one or more linear least-squares problems.
 175 
 176   `matrix` is a tensor of shape `[..., M, N]` whose inner-most 2 dimensions
 177   form `M`-by-`N` matrices. Rhs is a tensor of shape `[..., M, K]` whose
 178   inner-most 2 dimensions form `M`-by-`K` matrices.  The computed output is a
 179   `Tensor` of shape `[..., N, K]` whose inner-most 2 dimensions form `M`-by-`K`
 180   matrices that solve the equations
 181   `matrix[..., :, :] * output[..., :, :] = rhs[..., :, :]` in the least squares
 182   sense.
 183 
 184   Below we will use the following notation for each pair of matrix and
 185   right-hand sides in the batch:
 186 
 187   `matrix`=\\(A \in \Re^{m \times n}\\),
 188   `rhs`=\\(B  \in \Re^{m \times k}\\),
 189   `output`=\\(X  \in \Re^{n \times k}\\),
 190   `l2_regularizer`=\\(\lambda\\).
 191 
 192   If `fast` is `True`, then the solution is computed by solving the normal
 193   equations using Cholesky decomposition. Specifically, if \\(m \ge n\\) then
 194   \\(X = (A^T A + \lambda I)^{-1} A^T B\\), which solves the least-squares
 195   problem \\(X = \mathrm{argmin}_{Z \in \Re^{n \times k}} ||A Z - B||_F^2 +
 196   \lambda ||Z||_F^2\\). If \\(m \lt n\\) then `output` is computed as
 197   \\(X = A^T (A A^T + \lambda I)^{-1} B\\), which (for \\(\lambda = 0\\)) is
 198   the minimum-norm solution to the under-determined linear system, i.e.
 199   \\(X = \mathrm{argmin}_{Z \in \Re^{n \times k}} ||Z||_F^2 \\), subject to
 200   \\(A Z = B\\). Notice that the fast path is only numerically stable when
 201   \\(A\\) is numerically full rank and has a condition number
 202   \\(\mathrm{cond}(A) \lt \frac{1}{\sqrt{\epsilon_{mach}}}\\) or\\(\lambda\\)
 203   is sufficiently large.
 204 
 205   If `fast` is `False` an algorithm based on the numerically robust complete
 206   orthogonal decomposition is used. This computes the minimum-norm
 207   least-squares solution, even when \\(A\\) is rank deficient. This path is
 208   typically 6-7 times slower than the fast path. If `fast` is `False` then
 209   `l2_regularizer` is ignored.
 210 
 211   Args:
 212     matrix: `Tensor` of shape `[..., M, N]`.
 213     rhs: `Tensor` of shape `[..., M, K]`.
 214     l2_regularizer: 0-D `double` `Tensor`. Ignored if `fast=False`.
 215     fast: bool. Defaults to `True`.
 216     name: string, optional name of the operation.
 217 
 218   Returns:
 219     output: `Tensor` of shape `[..., N, K]` whose inner-most 2 dimensions form
 220       `M`-by-`K` matrices that solve the equations
 221       `matrix[..., :, :] * output[..., :, :] = rhs[..., :, :]` in the least
 222       squares sense.
 223 
 224   Raises:
 225     NotImplementedError: linalg.lstsq is currently disabled for complex128
 226     and l2_regularizer != 0 due to poor accuracy.
 227   """
 228 
 229   # pylint: disable=long-lambda
 230   def _use_composite_impl(fast, tensor_shape):
 231     """Determines whether to use the composite or specialized CPU kernel.
 232 
 233     When the total size of the tensor is larger than the cache size and the
 234     batch size is large compared to the smallest matrix dimension, then the
 235     composite implementation is inefficient since it has to read the entire
 236     tensor from memory multiple times. In this case we fall back to the
 237     original CPU kernel, which does all the computational steps on each
 238     matrix separately.
 239 
 240     Only fast mode is supported by the composite impl, so `False` is returned
 241     if `fast` is `False`.
 242 
 243     Args:
 244       fast: bool indicating if fast mode in the solver was requested.
 245       tensor_shape: The shape of the tensor.
 246 
 247     Returns:
 248       True if the composite impl should be used. False otherwise.
 249     """
 250     if fast is False:
 251       return False
 252     batch_shape = tensor_shape[:-2]
 253     matrix_shape = tensor_shape[-2:]
 254     if not tensor_shape.is_fully_defined():
 255       return True
 256     tensor_size = tensor_shape.num_elements() * matrix.dtype.size
 257     is_io_bound = batch_shape.num_elements() > np.min(matrix_shape)
 258     L2_CACHE_SIZE_GUESSTIMATE = 256000
 259     if tensor_size > L2_CACHE_SIZE_GUESSTIMATE and is_io_bound:
 260       return False
 261     else:
 262       return True
 263 
 264   def _overdetermined(matrix, rhs, l2_regularizer):
 265     """Computes (A^H*A + l2_regularizer)^{-1} * A^H * rhs."""
 266     chol = _RegularizedGramianCholesky(
 267         matrix, l2_regularizer=l2_regularizer, first_kind=True)
 268     return cholesky_solve(chol, math_ops.matmul(matrix, rhs, adjoint_a=True))
 269 
 270   def _underdetermined(matrix, rhs, l2_regularizer):
 271     """Computes A^H * (A*A^H + l2_regularizer)^{-1} * rhs."""
 272     chol = _RegularizedGramianCholesky(
 273         matrix, l2_regularizer=l2_regularizer, first_kind=False)
 274     return math_ops.matmul(matrix, cholesky_solve(chol, rhs), adjoint_a=True)
 275 
 276   def _composite_impl(matrix, rhs, l2_regularizer):
 277     """Composite implementation of matrix_solve_ls that supports GPU."""
 278     with ops.name_scope(name, 'matrix_solve_ls', [matrix, rhs, l2_regularizer]):
 279       matrix_shape = matrix.get_shape()[-2:]
 280       if matrix_shape.is_fully_defined():
 281         if matrix_shape[-2] >= matrix_shape[-1]:
 282           return _overdetermined(matrix, rhs, l2_regularizer)
 283         else:
 284           return _underdetermined(matrix, rhs, l2_regularizer)
 285       else:
 286         # We have to defer determining the shape to runtime and use
 287         # conditional execution of the appropriate graph.
 288         matrix_shape = array_ops.shape(matrix)[-2:]
 289         return control_flow_ops.cond(
 290             matrix_shape[-2] >= matrix_shape[-1],
 291             lambda: _overdetermined(matrix, rhs, l2_regularizer),
 292             lambda: _underdetermined(matrix, rhs, l2_regularizer))
 293 
 294   matrix = ops.convert_to_tensor(matrix, name='matrix')
 295   if matrix.dtype == dtypes.complex128 and l2_regularizer != 0:
 296     # TODO(rmlarsen): Investigate and fix accuracy bug.
 297     raise NotImplementedError('matrix_solve_ls is currently disabled for '
 298                               'complex128 and l2_regularizer != 0 due to '
 299                               'poor accuracy.')
 300   tensor_shape = matrix.get_shape()
 301   if _use_composite_impl(fast, tensor_shape):
 302     return _composite_impl(matrix, rhs, l2_regularizer)
 303   else:
 304     return gen_linalg_ops.matrix_solve_ls(
 305         matrix, rhs, l2_regularizer, fast=fast, name=name)

 354 @tf_export('svd', 'linalg.svd')
 355 @deprecation.deprecated_endpoints('svd')
 356 def svd(tensor, full_matrices=False, compute_uv=True, name=None):
 357   r"""Computes the singular value decompositions of one or more matrices.
 358 
 359   Computes the SVD of each inner matrix in `tensor` such that
 360   `tensor[..., :, :] = u[..., :, :] * diag(s[..., :, :]) *
 361    transpose(conj(v[..., :, :]))`
 362 
 363   ```python
 364   # a is a tensor.
 365   # s is a tensor of singular values.
 366   # u is a tensor of left singular vectors.
 367   # v is a tensor of right singular vectors.
 368   s, u, v = svd(a)
 369   s = svd(a, compute_uv=False)
 370   ```
 371 
 372   Args:
 373     tensor: `Tensor` of shape `[..., M, N]`. Let `P` be the minimum of `M` and
 374       `N`.
 375     full_matrices: If true, compute full-sized `u` and `v`. If false
 376       (the default), compute only the leading `P` singular vectors.
 377       Ignored if `compute_uv` is `False`.
 378     compute_uv: If `True` then left and right singular vectors will be
 379       computed and returned in `u` and `v`, respectively. Otherwise, only the
 380       singular values will be computed, which can be significantly faster.
 381     name: string, optional name of the operation.
 382 
 383   Returns:
 384     s: Singular values. Shape is `[..., P]`. The values are sorted in reverse
 385       order of magnitude, so s[..., 0] is the largest value, s[..., 1] is the
 386       second largest, etc.
 387     u: Left singular vectors. If `full_matrices` is `False` (default) then
 388       shape is `[..., M, P]`; if `full_matrices` is `True` then shape is
 389       `[..., M, M]`. Not returned if `compute_uv` is `False`.
 390     v: Right singular vectors. If `full_matrices` is `False` (default) then
 391       shape is `[..., N, P]`. If `full_matrices` is `True` then shape is
 392       `[..., N, N]`. Not returned if `compute_uv` is `False`.
 393 
 394   @compatibility(numpy)
 395   Mostly equivalent to numpy.linalg.svd, except that
 396     * The order of output  arguments here is `s`, `u`, `v` when `compute_uv` is
 397       `True`, as opposed to `u`, `s`, `v` for numpy.linalg.svd.
 398     * full_matrices is `False` by default as opposed to `True` for
 399        numpy.linalg.svd.
 400     * tf.linalg.svd uses the standard definition of the SVD
 401       \\(A = U \Sigma V^H\\), such that the left singular vectors of `a` are
 402       the columns of `u`, while the right singular vectors of `a` are the
 403       columns of `v`. On the other hand, numpy.linalg.svd returns the adjoint
 404       \\(V^H\\) as the third output argument.
 405   ```python
 406   import tensorflow as tf
 407   import numpy as np
 408   s, u, v = tf.linalg.svd(a)
 409   tf_a_approx = tf.matmul(u, tf.matmul(tf.linalg.diag(s), v, adjoint_b=True))
 410   u, s, v_adj = np.linalg.svd(a, full_matrices=False)
 411   np_a_approx = np.dot(u, np.dot(np.diag(s), v_adj))
 412   # tf_a_approx and np_a_approx should be numerically close.
 413   ```
 414   @end_compatibility
 415   """
 416   s, u, v = gen_linalg_ops.svd(
 417       tensor, compute_uv=compute_uv, full_matrices=full_matrices, name=name)
 418   if compute_uv:
 419     return math_ops.real(s), u, v
 420   else:
 421     return math_ops.real(s)

 424 # pylint: disable=redefined-builtin
 425 @tf_export('norm', 'linalg.norm')
 426 @deprecation.deprecated_args(
 427     None, 'keep_dims is deprecated, use keepdims instead', 'keep_dims')
 428 def norm(tensor,
 429          ord='euclidean',
 430          axis=None,
 431          keepdims=None,
 432          name=None,
 433          keep_dims=None):
 434   r"""Computes the norm of vectors, matrices, and tensors.
 435 
 436   This function can compute several different vector norms (the 1-norm, the
 437   Euclidean or 2-norm, the inf-norm, and in general the p-norm for p > 0) and
 438   matrix norms (Frobenius, 1-norm, 2-norm and inf-norm).
 439 
 440   Args:
 441     tensor: `Tensor` of types `float32`, `float64`, `complex64`, `complex128`
 442     ord: Order of the norm. Supported values are 'fro', 'euclidean',
 443       `1`, `2`, `np.inf` and any positive real number yielding the corresponding
 444       p-norm. Default is 'euclidean' which is equivalent to Frobenius norm if
 445       `tensor` is a matrix and equivalent to 2-norm for vectors.
 446       Some restrictions apply:
 447         a) The Frobenius norm `fro` is not defined for vectors,
 448         b) If axis is a 2-tuple (matrix norm), only 'euclidean', 'fro', `1`,
 449            `2`, `np.inf` are supported.
 450       See the description of `axis` on how to compute norms for a batch of
 451       vectors or matrices stored in a tensor.
 452     axis: If `axis` is `None` (the default), the input is considered a vector
 453       and a single vector norm is computed over the entire set of values in the
 454       tensor, i.e. `norm(tensor, ord=ord)` is equivalent to
 455       `norm(reshape(tensor, [-1]), ord=ord)`.
 456       If `axis` is a Python integer, the input is considered a batch of vectors,
 457       and `axis` determines the axis in `tensor` over which to compute vector
 458       norms.
 459       If `axis` is a 2-tuple of Python integers it is considered a batch of
 460       matrices and `axis` determines the axes in `tensor` over which to compute
 461       a matrix norm.
 462       Negative indices are supported. Example: If you are passing a tensor that
 463       can be either a matrix or a batch of matrices at runtime, pass
 464       `axis=[-2,-1]` instead of `axis=None` to make sure that matrix norms are
 465       computed.
 466     keepdims: If True, the axis indicated in `axis` are kept with size 1.
 467       Otherwise, the dimensions in `axis` are removed from the output shape.
 468     name: The name of the op.
 469     keep_dims: Deprecated alias for `keepdims`.
 470 
 471   Returns:
 472     output: A `Tensor` of the same type as tensor, containing the vector or
 473       matrix norms. If `keepdims` is True then the rank of output is equal to
 474       the rank of `tensor`. Otherwise, if `axis` is none the output is a scalar,
 475       if `axis` is an integer, the rank of `output` is one less than the rank
 476       of `tensor`, if `axis` is a 2-tuple the rank of `output` is two less
 477       than the rank of `tensor`.
 478 
 479   Raises:
 480     ValueError: If `ord` or `axis` is invalid.
 481 
 482   @compatibility(numpy)
 483   Mostly equivalent to numpy.linalg.norm.
 484   Not supported: ord <= 0, 2-norm for matrices, nuclear norm.
 485   Other differences:
 486     a) If axis is `None`, treats the flattened `tensor` as a vector
 487      regardless of rank.
 488     b) Explicitly supports 'euclidean' norm as the default, including for
 489      higher order tensors.
 490   @end_compatibility
 491   """
 492   keepdims = deprecation.deprecated_argument_lookup('keepdims', keepdims,
 493                                                     'keep_dims', keep_dims)
 494   if keepdims is None:
 495     keepdims = False
 496 
 497   is_matrix_norm = ((isinstance(axis, tuple) or isinstance(axis, list)) and
 498                     len(axis) == 2)
 499   if is_matrix_norm:
 500     axis = tuple(axis)
 501     if (not isinstance(axis[0], int) or not isinstance(axis[1], int) or
 502         axis[0] == axis[1]):
 503       raise ValueError(
 504           "'axis' must be None, an integer, or a tuple of 2 unique integers")
 505     supported_matrix_norms = ['euclidean', 'fro', 1, 2, np.inf]
 506     if ord not in supported_matrix_norms:
 507       raise ValueError("'ord' must be a supported matrix norm in %s, got %s" %
 508                        (supported_matrix_norms, ord))
 509   else:
 510     if not (isinstance(axis, int) or axis is None):
 511       raise ValueError(
 512           "'axis' must be None, an integer, or a tuple of 2 unique integers")
 513 
 514     supported_vector_norms = ['euclidean', 1, 2, np.inf]
 515     if (not np.isreal(ord) or ord <= 0) and ord not in supported_vector_norms:
 516       raise ValueError("'ord' must be a supported vector norm, got %s" % ord)
 517     if axis is not None:
 518       axis = (axis,)
 519 
 520   with ops.name_scope(name, 'norm', [tensor]):
 521     tensor = ops.convert_to_tensor(tensor)
 522 
 523     if ord in ['fro', 'euclidean', 2, 2.0]:
 524       if is_matrix_norm and ord in [2, 2.0]:
 525         rank = array_ops.rank(tensor)
 526         positive_axis = functional_ops.map_fn(
 527             lambda i: control_flow_ops.cond(i >= 0, lambda: i, lambda: i + rank),
 528             ops.convert_to_tensor(axis))
 529         axes = math_ops.range(rank)
 530         perm_before = array_ops.concat(
 531             [array_ops.setdiff1d(axes, positive_axis)[0], positive_axis],
 532             axis=0)
 533         perm_after = functional_ops.map_fn(
 534             lambda i: math_ops.cast(
 535                 array_ops.squeeze(
 536                     array_ops.where(math_ops.equal(perm_before, i))),
 537                 dtype=dtypes.int32), axes)
 538         permed = array_ops.transpose(tensor, perm=perm_before)
 539         matrix_2_norm = array_ops.expand_dims(
 540             math_ops.reduce_max(
 541                 math_ops.abs(gen_linalg_ops.svd(permed, compute_uv=False)[0]),
 542                 axis=-1,
 543                 keepdims=True),
 544             axis=-1)
 545         result = array_ops.transpose(matrix_2_norm, perm=perm_after)
 546       else:
 547         result = math_ops.sqrt(
 548             math_ops.reduce_sum(
 549                 tensor * math_ops.conj(tensor), axis, keepdims=True))
 550     else:
 551       result = math_ops.abs(tensor)
 552       if ord == 1:
 553         sum_axis = None if axis is None else axis[0]
 554         result = math_ops.reduce_sum(result, sum_axis, keepdims=True)
 555         if is_matrix_norm:
 556           result = math_ops.reduce_max(result, axis[-1], keepdims=True)
 557       elif ord == np.inf:
 558         if is_matrix_norm:
 559           result = math_ops.reduce_sum(result, axis[1], keepdims=True)
 560         max_axis = None if axis is None else axis[0]
 561         result = math_ops.reduce_max(result, max_axis, keepdims=True)
 562       else:
 563         # General p-norms (positive p only)
 564         result = math_ops.pow(
 565             math_ops.reduce_sum(math_ops.pow(result, ord), axis, keepdims=True),
 566             1.0 / ord)
 567     if not keepdims:
 568       result = array_ops.squeeze(result, axis)
 569     return result
