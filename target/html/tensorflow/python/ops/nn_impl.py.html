  41 @tf_export("nn.log_poisson_loss")
  42 def log_poisson_loss(targets, log_input, compute_full_loss=False, name=None):
  43   """Computes log Poisson loss given `log_input`.
  44 
  45   Gives the log-likelihood loss between the prediction and the target under the
  46   assumption that the target has a Poisson distribution.
  47   Caveat: By default, this is not the exact loss, but the loss minus a
  48     constant term [log(z!)]. That has no effect for optimization, but
  49     does not play well with relative loss comparisons. To compute an
  50     approximation of the log factorial term, specify
  51     compute_full_loss=True to enable Stirling's Approximation.
  52 
  53   For brevity, let `c = log(x) = log_input`, `z = targets`.  The log Poisson
  54   loss is
  55 
  56         -log(exp(-x) * (x^z) / z!)
  57       = -log(exp(-x) * (x^z)) + log(z!)
  58       ~ -log(exp(-x)) - log(x^z) [+ z * log(z) - z + 0.5 * log(2 * pi * z)]
  59           [ Note the second term is the Stirling's Approximation for log(z!).
  60             It is invariant to x and does not affect optimization, though
  61             important for correct relative loss comparisons. It is only
  62             computed when compute_full_loss == True. ]
  63       = x - z * log(x) [+ z * log(z) - z + 0.5 * log(2 * pi * z)]
  64       = exp(c) - z * c [+ z * log(z) - z + 0.5 * log(2 * pi * z)]
  65 
  66   Args:
  67     targets: A `Tensor` of the same type and shape as `log_input`.
  68     log_input: A `Tensor` of type `float32` or `float64`.
  69     compute_full_loss: whether to compute the full loss. If false, a constant
  70       term is dropped in favor of more efficient optimization.
  71     name: A name for the operation (optional).
  72 
  73   Returns:
  74     A `Tensor` of the same shape as `log_input` with the componentwise
  75     logistic losses.
  76 
  77   Raises:
  78     ValueError: If `log_input` and `targets` do not have the same shape.
  79   """
  80   with ops.name_scope(name, "log_poisson_loss", [log_input, targets]) as name:
  81     log_input = ops.convert_to_tensor(log_input, name="log_input")
  82     targets = ops.convert_to_tensor(targets, name="targets")
  83     try:
  84       targets.get_shape().merge_with(log_input.get_shape())
  85     except ValueError:
  86       raise ValueError(
  87           "log_input and targets must have the same shape (%s vs %s)" %
  88           (log_input.get_shape(), targets.get_shape()))
  89 
  90     result = math_ops.exp(log_input) - log_input * targets
  91     if compute_full_loss:
  92       # need to create constant tensors here so that their dtypes can be matched
  93       # to that of the targets.
  94       point_five = constant_op.constant(0.5, dtype=targets.dtype)
  95       two_pi = constant_op.constant(2 * math.pi, dtype=targets.dtype)
  96 
  97       stirling_approx = (targets * math_ops.log(targets)) - targets + (
  98           point_five * math_ops.log(two_pi * targets))
  99       zeros = array_ops.zeros_like(targets, dtype=targets.dtype)
 100       ones = array_ops.ones_like(targets, dtype=targets.dtype)
 101       cond = math_ops.logical_and(targets >= zeros, targets <= ones)
 102       result += array_ops.where(cond, zeros, stirling_approx)
 103     return result

 106 @tf_export("nn.sigmoid_cross_entropy_with_logits")
 107 def sigmoid_cross_entropy_with_logits(  # pylint: disable=invalid-name
 108     _sentinel=None,
 109     labels=None,
 110     logits=None,
 111     name=None):
 112   """Computes sigmoid cross entropy given `logits`.
 113 
 114   Measures the probability error in discrete classification tasks in which each
 115   class is independent and not mutually exclusive.  For instance, one could
 116   perform multilabel classification where a picture can contain both an elephant
 117   and a dog at the same time.
 118 
 119   For brevity, let `x = logits`, `z = labels`.  The logistic loss is
 120 
 121         z * -log(sigmoid(x)) + (1 - z) * -log(1 - sigmoid(x))
 122       = z * -log(1 / (1 + exp(-x))) + (1 - z) * -log(exp(-x) / (1 + exp(-x)))
 123       = z * log(1 + exp(-x)) + (1 - z) * (-log(exp(-x)) + log(1 + exp(-x)))
 124       = z * log(1 + exp(-x)) + (1 - z) * (x + log(1 + exp(-x))
 125       = (1 - z) * x + log(1 + exp(-x))
 126       = x - x * z + log(1 + exp(-x))
 127 
 128   For x < 0, to avoid overflow in exp(-x), we reformulate the above
 129 
 130         x - x * z + log(1 + exp(-x))
 131       = log(exp(x)) - x * z + log(1 + exp(-x))
 132       = - x * z + log(1 + exp(x))
 133 
 134   Hence, to ensure stability and avoid overflow, the implementation uses this
 135   equivalent formulation
 136 
 137       max(x, 0) - x * z + log(1 + exp(-abs(x)))
 138 
 139   `logits` and `labels` must have the same type and shape.
 140 
 141   Args:
 142     _sentinel: Used to prevent positional parameters. Internal, do not use.
 143     labels: A `Tensor` of the same type and shape as `logits`.
 144     logits: A `Tensor` of type `float32` or `float64`.
 145     name: A name for the operation (optional).
 146 
 147   Returns:
 148     A `Tensor` of the same shape as `logits` with the componentwise
 149     logistic losses.
 150 
 151   Raises:
 152     ValueError: If `logits` and `labels` do not have the same shape.
 153   """
 154   # pylint: disable=protected-access
 155   nn_ops._ensure_xent_args("sigmoid_cross_entropy_with_logits", _sentinel,
 156                            labels, logits)
 157   # pylint: enable=protected-access
 158 
 159   with ops.name_scope(name, "logistic_loss", [logits, labels]) as name:
 160     logits = ops.convert_to_tensor(logits, name="logits")
 161     labels = ops.convert_to_tensor(labels, name="labels")
 162     try:
 163       labels.get_shape().merge_with(logits.get_shape())
 164     except ValueError:
 165       raise ValueError("logits and labels must have the same shape (%s vs %s)" %
 166                        (logits.get_shape(), labels.get_shape()))
 167 
 168     # The logistic loss formula from above is
 169     #   x - x * z + log(1 + exp(-x))
 170     # For x < 0, a more numerically stable formula is
 171     #   -x * z + log(1 + exp(x))
 172     # Note that these two expressions can be combined into the following:
 173     #   max(x, 0) - x * z + log(1 + exp(-abs(x)))
 174     # To allow computing gradients at zero, we define custom versions of max and
 175     # abs functions.
 176     zeros = array_ops.zeros_like(logits, dtype=logits.dtype)
 177     cond = (logits >= zeros)
 178     relu_logits = array_ops.where(cond, logits, zeros)
 179     neg_abs_logits = array_ops.where(cond, -logits, logits)
 180     return math_ops.add(
 181         relu_logits - logits * labels,
 182         math_ops.log1p(math_ops.exp(neg_abs_logits)),
 183         name=name)

 186 @tf_export("nn.weighted_cross_entropy_with_logits")
 187 def weighted_cross_entropy_with_logits(targets, logits, pos_weight, name=None):
 188   """Computes a weighted cross entropy.
 189 
 190   This is like `sigmoid_cross_entropy_with_logits()` except that `pos_weight`,
 191   allows one to trade off recall and precision by up- or down-weighting the
 192   cost of a positive error relative to a negative error.
 193 
 194   The usual cross-entropy cost is defined as:
 195 
 196       targets * -log(sigmoid(logits)) +
 197           (1 - targets) * -log(1 - sigmoid(logits))
 198 
 199   A value `pos_weights > 1` decreases the false negative count, hence increasing
 200   the recall.
 201   Conversely setting `pos_weights < 1` decreases the false positive count and
 202   increases the precision.
 203   This can be seen from the fact that `pos_weight` is introduced as a
 204   multiplicative coefficient for the positive targets term
 205   in the loss expression:
 206 
 207       targets * -log(sigmoid(logits)) * pos_weight +
 208           (1 - targets) * -log(1 - sigmoid(logits))
 209 
 210   For brevity, let `x = logits`, `z = targets`, `q = pos_weight`.
 211   The loss is:
 212 
 213         qz * -log(sigmoid(x)) + (1 - z) * -log(1 - sigmoid(x))
 214       = qz * -log(1 / (1 + exp(-x))) + (1 - z) * -log(exp(-x) / (1 + exp(-x)))
 215       = qz * log(1 + exp(-x)) + (1 - z) * (-log(exp(-x)) + log(1 + exp(-x)))
 216       = qz * log(1 + exp(-x)) + (1 - z) * (x + log(1 + exp(-x))
 217       = (1 - z) * x + (qz +  1 - z) * log(1 + exp(-x))
 218       = (1 - z) * x + (1 + (q - 1) * z) * log(1 + exp(-x))
 219 
 220   Setting `l = (1 + (q - 1) * z)`, to ensure stability and avoid overflow,
 221   the implementation uses
 222 
 223       (1 - z) * x + l * (log(1 + exp(-abs(x))) + max(-x, 0))
 224 
 225   `logits` and `targets` must have the same type and shape.
 226 
 227   Args:
 228     targets: A `Tensor` of the same type and shape as `logits`.
 229     logits: A `Tensor` of type `float32` or `float64`.
 230     pos_weight: A coefficient to use on the positive examples.
 231     name: A name for the operation (optional).
 232 
 233   Returns:
 234     A `Tensor` of the same shape as `logits` with the componentwise
 235     weighted logistic losses.
 236 
 237   Raises:
 238     ValueError: If `logits` and `targets` do not have the same shape.
 239   """
 240   with ops.name_scope(name, "logistic_loss", [logits, targets]) as name:
 241     logits = ops.convert_to_tensor(logits, name="logits")
 242     targets = ops.convert_to_tensor(targets, name="targets")
 243     try:
 244       targets.get_shape().merge_with(logits.get_shape())
 245     except ValueError:
 246       raise ValueError(
 247           "logits and targets must have the same shape (%s vs %s)" %
 248           (logits.get_shape(), targets.get_shape()))
 249 
 250     # The logistic loss formula from above is
 251     #   (1 - z) * x + (1 + (q - 1) * z) * log(1 + exp(-x))
 252     # For x < 0, a more numerically stable formula is
 253     #   (1 - z) * x + (1 + (q - 1) * z) * log(1 + exp(x)) - l * x
 254     # To avoid branching, we use the combined version
 255     #   (1 - z) * x + l * (log(1 + exp(-abs(x))) + max(-x, 0))
 256     log_weight = 1 + (pos_weight - 1) * targets
 257     return math_ops.add(
 258         (1 - targets) * logits,
 259         log_weight * (math_ops.log1p(math_ops.exp(-math_ops.abs(logits))) +
 260                       nn_ops.relu(-logits)),
 261         name=name)

 264 @tf_export("nn.relu_layer")
 265 def relu_layer(x, weights, biases, name=None):
 266   """Computes Relu(x * weight + biases).
 267 
 268   Args:
 269     x: a 2D tensor.  Dimensions typically: batch, in_units
 270     weights: a 2D tensor.  Dimensions typically: in_units, out_units
 271     biases: a 1D tensor.  Dimensions: out_units
 272     name: A name for the operation (optional).  If not specified
 273       "nn_relu_layer" is used.
 274 
 275   Returns:
 276     A 2-D Tensor computing relu(matmul(x, weights) + biases).
 277     Dimensions typically: batch, out_units.
 278   """
 279   with ops.name_scope(name, "relu_layer", [x, weights, biases]) as name:
 280     x = ops.convert_to_tensor(x, name="x")
 281     weights = ops.convert_to_tensor(weights, name="weights")
 282     biases = ops.convert_to_tensor(biases, name="biases")
 283     xw_plus_b = nn_ops.bias_add(math_ops.matmul(x, weights), biases)
 284     return nn_ops.relu(xw_plus_b, name=name)

 292 @function.Defun(shape_func=_swish_shape, func_name="swish_grad", noinline=True)
 293 def _swish_grad(features, grad):
 294   """Gradient of Swish function defined below."""
 295   sigmoid_features = math_ops.sigmoid(features)
 296   activation_grad = (
 297       sigmoid_features * (1.0 + features * (1.0 - sigmoid_features)))
 298   return grad * activation_grad

 301 # Naively, x * tf.nn.sigmoid(x) requires keeping both x and sigmoid(x) around
 302 # for backprop, effectively doubling the tensor's memory consumption. We use a
 303 # @Defun decorator with noinline=True so that sigmoid(features) is re-computed
 304 # during backprop, and we can free the sigmoid(features) expression immediately
 305 # after use during the forward pass.
 306 @tf_export("nn.swish")
 307 @function.Defun(
 308     grad_func=_swish_grad,
 309     shape_func=_swish_shape,
 310     func_name="swish",
 311     noinline=True)
 312 def swish(features):
 313   # pylint: disable=g-doc-args
 314   """Computes the Swish activation function: `x * sigmoid(x)`.
 315 
 316   Source: "Searching for Activation Functions" (Ramachandran et al. 2017)
 317   https://arxiv.org/abs/1710.05941
 318 
 319   Args:
 320     features: A `Tensor` representing preactivation values.
 321     name: A name for the operation (optional).
 322 
 323   Returns:
 324     The activation value.
 325   """
 326   # pylint: enable=g-doc-args
 327   features = ops.convert_to_tensor(features, name="features")
 328   return features * math_ops.sigmoid(features)

 331 @tf_export("math.l2_normalize", "linalg.l2_normalize", "nn.l2_normalize")
 332 @deprecated_args(None, "dim is deprecated, use axis instead", "dim")
 333 def l2_normalize(x, axis=None, epsilon=1e-12, name=None, dim=None):
 334   """Normalizes along dimension `axis` using an L2 norm.
 335 
 336   For a 1-D tensor with `axis = 0`, computes
 337 
 338       output = x / sqrt(max(sum(x**2), epsilon))
 339 
 340   For `x` with more dimensions, independently normalizes each 1-D slice along
 341   dimension `axis`.
 342 
 343   Args:
 344     x: A `Tensor`.
 345     axis: Dimension along which to normalize.  A scalar or a vector of
 346       integers.
 347     epsilon: A lower bound value for the norm. Will use `sqrt(epsilon)` as the
 348       divisor if `norm < sqrt(epsilon)`.
 349     name: A name for this operation (optional).
 350     dim: Deprecated alias for axis.
 351 
 352   Returns:
 353     A `Tensor` with the same shape as `x`.
 354   """
 355   with ops.name_scope(name, "l2_normalize", [x]) as name:
 356     axis = deprecated_argument_lookup("axis", axis, "dim", dim)
 357     x = ops.convert_to_tensor(x, name="x")
 358     square_sum = math_ops.reduce_sum(math_ops.square(x), axis, keepdims=True)
 359     x_inv_norm = math_ops.rsqrt(math_ops.maximum(square_sum, epsilon))
 360     return math_ops.multiply(x, x_inv_norm, name=name)

 363 @tf_export("math.zero_fraction", "nn.zero_fraction")
 364 def zero_fraction(value, name=None):
 365   """Returns the fraction of zeros in `value`.
 366 
 367   If `value` is empty, the result is `nan`.
 368 
 369   This is useful in summaries to measure and report sparsity.  For example,
 370 
 371   ```python
 372       z = tf.nn.relu(...)
 373       summ = tf.summary.scalar('sparsity', tf.nn.zero_fraction(z))
 374   ```
 375 
 376   Args:
 377     value: A tensor of numeric type.
 378     name: A name for the operation (optional).
 379 
 380   Returns:
 381     The fraction of zeros in `value`, with type `float32`.
 382   """
 383   with ops.name_scope(name, "zero_fraction", [value]):
 384     value = ops.convert_to_tensor(value, name="value")
 385     zero = constant_op.constant(0, dtype=value.dtype, name="zero")
 386     return math_ops.reduce_mean(
 387         math_ops.cast(math_ops.equal(value, zero), dtypes.float32))

 391 @tf_export("nn.depthwise_conv2d")
 392 def depthwise_conv2d(input,
 393                      filter,
 394                      strides,
 395                      padding,
 396                      rate=None,
 397                      name=None,
 398                      data_format=None):
 399   """Depthwise 2-D convolution.
 400 
 401   Given a 4D input tensor ('NHWC' or 'NCHW' data formats)
 402   and a filter tensor of shape
 403   `[filter_height, filter_width, in_channels, channel_multiplier]`
 404   containing `in_channels` convolutional filters of depth 1, `depthwise_conv2d`
 405   applies a different filter to each input channel (expanding from 1 channel
 406   to `channel_multiplier` channels for each), then concatenates the results
 407   together.  The output has `in_channels * channel_multiplier` channels.
 408 
 409   In detail,
 410 
 411       output[b, i, j, k * channel_multiplier + q] = sum_{di, dj}
 412            filter[di, dj, k, q] * input[b, strides[1] * i + rate[0] * di,
 413                                            strides[2] * j + rate[1] * dj, k]
 414 
 415   Must have `strides[0] = strides[3] = 1`.  For the most common case of the
 416   same horizontal and vertical strides, `strides = [1, stride, stride, 1]`.
 417   If any value in `rate` is greater than 1, we perform atrous depthwise
 418   convolution, in which case all values in the `strides` tensor must be equal
 419   to 1.
 420 
 421   Args:
 422     input: 4-D with shape according to `data_format`.
 423     filter: 4-D with shape
 424       `[filter_height, filter_width, in_channels, channel_multiplier]`.
 425     strides: 1-D of size 4.  The stride of the sliding window for each
 426       dimension of `input`.
 427     padding: A string, either `'VALID'` or `'SAME'`. The padding algorithm.
 428       See the "returns" section of `tf.nn.convolution` for details.
 429     rate: 1-D of size 2. The dilation rate in which we sample input values
 430       across the `height` and `width` dimensions in atrous convolution. If it is
 431       greater than 1, then all values of strides must be 1.
 432     name: A name for this operation (optional).
 433     data_format: The data format for input. Either "NHWC" (default) or "NCHW".
 434 
 435   Returns:
 436     A 4-D `Tensor` with shape according to `data_format`.  E.g., for
 437     "NHWC" format, shape is
 438     `[batch, out_height, out_width, in_channels * channel_multiplier].`
 439   """
 440   with ops.name_scope(name, "depthwise", [input, filter]) as name:
 441     input = ops.convert_to_tensor(input, name="tensor_in")
 442     filter = ops.convert_to_tensor(filter, name="filter_in")
 443     if rate is None:
 444       rate = [1, 1]
 445 
 446     def op(input_converted, _, padding):
 447       return nn_ops.depthwise_conv2d_native(
 448           input=input_converted,
 449           filter=filter,
 450           strides=strides,
 451           padding=padding,
 452           data_format=data_format,
 453           name=name)
 454 
 455     return nn_ops.with_space_to_batch(
 456         input=input,
 457         filter_shape=array_ops.shape(filter),
 458         dilation_rate=rate,
 459         padding=padding,
 460         data_format=data_format,
 461         op=op)

 468 @tf_export("nn.separable_conv2d")
 469 def separable_conv2d(input,
 470                      depthwise_filter,
 471                      pointwise_filter,
 472                      strides,
 473                      padding,
 474                      rate=None,
 475                      name=None,
 476                      data_format=None):
 477   """2-D convolution with separable filters.
 478 
 479   Performs a depthwise convolution that acts separately on channels followed by
 480   a pointwise convolution that mixes channels.  Note that this is separability
 481   between dimensions `[1, 2]` and `3`, not spatial separability between
 482   dimensions `1` and `2`.
 483 
 484   In detail,
 485 
 486       output[b, i, j, k] = sum_{di, dj, q, r}
 487           input[b, strides[1] * i + di, strides[2] * j + dj, q] *
 488           depthwise_filter[di, dj, q, r] *
 489           pointwise_filter[0, 0, q * channel_multiplier + r, k]
 490 
 491   `strides` controls the strides for the depthwise convolution only, since
 492   the pointwise convolution has implicit strides of `[1, 1, 1, 1]`.  Must have
 493   `strides[0] = strides[3] = 1`.  For the most common case of the same
 494   horizontal and vertical strides, `strides = [1, stride, stride, 1]`.
 495   If any value in `rate` is greater than 1, we perform atrous depthwise
 496   convolution, in which case all values in the `strides` tensor must be equal
 497   to 1.
 498 
 499   Args:
 500     input: 4-D `Tensor` with shape according to `data_format`.
 501     depthwise_filter: 4-D `Tensor` with shape
 502       `[filter_height, filter_width, in_channels, channel_multiplier]`.
 503       Contains `in_channels` convolutional filters of depth 1.
 504     pointwise_filter: 4-D `Tensor` with shape
 505       `[1, 1, channel_multiplier * in_channels, out_channels]`.  Pointwise
 506       filter to mix channels after `depthwise_filter` has convolved spatially.
 507     strides: 1-D of size 4.  The strides for the depthwise convolution for
 508       each dimension of `input`.
 509     padding: A string, either `'VALID'` or `'SAME'`.  The padding algorithm.
 510       See the "returns" section of `tf.nn.convolution` for details.
 511     rate: 1-D of size 2. The dilation rate in which we sample input values
 512       across the `height` and `width` dimensions in atrous convolution. If it is
 513       greater than 1, then all values of strides must be 1.
 514     name: A name for this operation (optional).
 515     data_format: The data format for input. Either "NHWC" (default) or "NCHW".
 516 
 517   Returns:
 518     A 4-D `Tensor` with shape according to 'data_format'. For
 519       example, with data_format="NHWC", shape is [batch, out_height,
 520       out_width, out_channels].
 521   """
 522   with ops.name_scope(name, "separable_conv2d",
 523                       [input, depthwise_filter, pointwise_filter]) as name:
 524     input = ops.convert_to_tensor(input, name="tensor_in")
 525     depthwise_filter = ops.convert_to_tensor(
 526         depthwise_filter, name="depthwise_filter")
 527     pointwise_filter = ops.convert_to_tensor(
 528         pointwise_filter, name="pointwise_filter")
 529 
 530     pointwise_filter_shape = pointwise_filter.get_shape().with_rank(4)

 569 @tf_export("nn.sufficient_statistics")
 570 def sufficient_statistics(x, axes, shift=None, keep_dims=False, name=None):
 571   """Calculate the sufficient statistics for the mean and variance of `x`.
 572 
 573   These sufficient statistics are computed using the one pass algorithm on
 574   an input that's optionally shifted. See:
 575   https://en.wikipedia.org/wiki/Algorithms_for_calculating_variance#Computing_shifted_data
 576 
 577   Args:
 578     x: A `Tensor`.
 579     axes: Array of ints. Axes along which to compute mean and variance.
 580     shift: A `Tensor` containing the value by which to shift the data for
 581       numerical stability, or `None` if no shift is to be performed. A shift
 582       close to the true mean provides the most numerically stable results.
 583     keep_dims: produce statistics with the same dimensionality as the input.
 584     name: Name used to scope the operations that compute the sufficient stats.
 585 
 586   Returns:
 587     Four `Tensor` objects of the same type as `x`:
 588 
 589     * the count (number of elements to average over).
 590     * the (possibly shifted) sum of the elements in the array.
 591     * the (possibly shifted) sum of squares of the elements in the array.
 592     * the shift by which the mean must be corrected or None if `shift` is None.
 593   """
 594   axes = list(set(axes))
 595   with ops.name_scope(name, "sufficient_statistics", [x, shift]):
 596     x = ops.convert_to_tensor(x, name="x")
 597     x_shape = x.get_shape()
 598     if all(x_shape[d].value is not None for d in axes):
 599       counts = 1
 600       for d in axes:
 601         counts *= x_shape[d].value
 602       counts = constant_op.constant(counts, dtype=x.dtype)
 603     else:  # shape needs to be inferred at runtime.
 604       x_dims = array_ops.gather(
 605           math_ops.cast(array_ops.shape(x), x.dtype), axes)
 606       counts = math_ops.reduce_prod(x_dims, name="count")
 607     if shift is not None:
 608       shift = ops.convert_to_tensor(shift, name="shift")
 609       m_ss = math_ops.subtract(x, shift)
 610       v_ss = math_ops.squared_difference(x, shift)
 611     else:  # no shift.
 612       m_ss = x
 613       v_ss = math_ops.square(x)
 614     m_ss = math_ops.reduce_sum(m_ss, axes, keepdims=keep_dims, name="mean_ss")
 615     v_ss = math_ops.reduce_sum(v_ss, axes, keepdims=keep_dims, name="var_ss")
 616   return counts, m_ss, v_ss, shift

 619 @tf_export("nn.normalize_moments")
 620 def normalize_moments(counts, mean_ss, variance_ss, shift, name=None):
 621   """Calculate the mean and variance of based on the sufficient statistics.
 622 
 623   Args:
 624     counts: A `Tensor` containing the total count of the data (one value).
 625     mean_ss: A `Tensor` containing the mean sufficient statistics: the (possibly
 626       shifted) sum of the elements to average over.
 627     variance_ss: A `Tensor` containing the variance sufficient statistics: the
 628       (possibly shifted) squared sum of the data to compute the variance over.
 629     shift: A `Tensor` containing the value by which the data is shifted for
 630       numerical stability, or `None` if no shift was performed.
 631     name: Name used to scope the operations that compute the moments.
 632 
 633   Returns:
 634     Two `Tensor` objects: `mean` and `variance`.
 635   """
 636   with ops.name_scope(name, "normalize", [counts, mean_ss, variance_ss, shift]):
 637     divisor = math_ops.reciprocal(counts, name="divisor")
 638     if shift is not None:
 639       shifted_mean = math_ops.multiply(mean_ss, divisor, name="shifted_mean")
 640       mean = math_ops.add(shifted_mean, shift, name="mean")
 641     else:  # no shift.
 642       shifted_mean = math_ops.multiply(mean_ss, divisor, name="mean")
 643       mean = shifted_mean
 644     variance = math_ops.subtract(
 645         math_ops.multiply(variance_ss, divisor),
 646         math_ops.square(shifted_mean),
 647         name="variance")
 648   return (mean, variance)

 651 @tf_export("nn.moments")
 652 def moments(
 653     x,
 654     axes,
 655     shift=None,  # pylint: disable=unused-argument
 656     name=None,
 657     keep_dims=False):
 658   """Calculate the mean and variance of `x`.
 659 
 660   The mean and variance are calculated by aggregating the contents of `x`
 661   across `axes`.  If `x` is 1-D and `axes = [0]` this is just the mean
 662   and variance of a vector.
 663 
 664   Note: shift is currently not used; the true mean is computed and used.
 665 
 666   When using these moments for batch normalization (see
 667   `tf.nn.batch_normalization`):
 668 
 669    * for so-called "global normalization", used with convolutional filters with
 670      shape `[batch, height, width, depth]`, pass `axes=[0, 1, 2]`.
 671    * for simple batch normalization pass `axes=[0]` (batch only).
 672 
 673   Args:
 674     x: A `Tensor`.
 675     axes: Array of ints.  Axes along which to compute mean and
 676       variance.
 677     shift: Not used in the current implementation
 678     name: Name used to scope the operations that compute the moments.
 679     keep_dims: produce moments with the same dimensionality as the input.
 680 
 681   Returns:
 682     Two `Tensor` objects: `mean` and `variance`.
 683   """
 684   with ops.name_scope(name, "moments", [x, axes]):
 685     # The dynamic range of fp16 is too limited to support the collection of
 686     # sufficient statistics. As a workaround we simply perform the operations
 687     # on 32-bit floats before converting the mean and variance back to fp16
 688     y = math_ops.cast(x, dtypes.float32) if x.dtype == dtypes.float16 else x
 689     # Compute true mean while keeping the dims for proper broadcasting.
 690     mean = math_ops.reduce_mean(y, axes, keepdims=True, name="mean")
 691     # sample variance, not unbiased variance
 692     # Note: stop_gradient does not change the gradient that gets
 693     #       backpropagated to the mean from the variance calculation,
 694     #       because that gradient is zero
 695     variance = math_ops.reduce_mean(
 696         math_ops.squared_difference(y, array_ops.stop_gradient(mean)),
 697         axes,
 698         keepdims=True,
 699         name="variance")
 700     if not keep_dims:
 701       mean = array_ops.squeeze(mean, axes)
 702       variance = array_ops.squeeze(variance, axes)
 703     if x.dtype == dtypes.float16:
 704       return (math_ops.cast(mean, dtypes.float16),
 705               math_ops.cast(variance, dtypes.float16))
 706     else:
 707       return (mean, variance)

 710 @tf_export("nn.weighted_moments")
 711 def weighted_moments(x, axes, frequency_weights, name=None, keep_dims=False):
 712   """Returns the frequency-weighted mean and variance of `x`.
 713 
 714   Args:
 715     x: A tensor.
 716     axes: 1-d tensor of int32 values; these are the axes along which
 717       to compute mean and variance.
 718     frequency_weights: A tensor of positive weights which can be
 719       broadcast with x.
 720     name: Name used to scope the operation.
 721     keep_dims: Produce moments with the same dimensionality as the input.
 722 
 723   Returns:
 724     Two tensors: `weighted_mean` and `weighted_variance`.
 725   """
 726   with ops.name_scope(name, "weighted_moments", [x, frequency_weights, axes]):
 727     x = ops.convert_to_tensor(x, name="x")
 728     frequency_weights = ops.convert_to_tensor(
 729         frequency_weights, name="frequency_weights")
 730 
 731     # Unlike moments(), this just uses a simpler two-pass method.
 732 
 733     # See comment in moments() WRT precision; it applies here too.
 734     needs_cast = x.dtype == dtypes.float16
 735     if needs_cast:
 736       x = math_ops.cast(x, dtypes.float32)
 737 
 738     if frequency_weights.dtype != x.dtype:
 739       frequency_weights = math_ops.cast(frequency_weights, x.dtype)
 740 
 741     # Note that we use keep_dims=True for our reductions regardless of the arg;
 742     # this is so that the results remain broadcast-compatible with the inputs.
 743     weighted_input_sum = math_ops.reduce_sum(
 744         frequency_weights * x, axes, name="weighted_input_sum", keepdims=True)
 745 
 746     # The shape of the weights isn't necessarily the same as x's
 747     # shape, just broadcast-compatible with it -- so this expression
 748     # performs broadcasting to give a per-item weight, with the same
 749     # shape as (freqency_weights * x). This avoids having to reason
 750     # through all the broadcast logic to compute a correct
 751     # sum_of_weights.
 752     broadcasted_weights = frequency_weights + array_ops.zeros_like(x)
 753 
 754     sum_of_weights = math_ops.reduce_sum(
 755         broadcasted_weights, axes, name="sum_of_weights", keepdims=True)
 756 
 757     divisor = math_ops.reciprocal(sum_of_weights, name="inv_weight_sum")
 758 
 759     weighted_mean = math_ops.multiply(weighted_input_sum, divisor)
 760 
 761     # Have the weighted mean; now on to variance:
 762     weighted_distsq = math_ops.reduce_sum(
 763         frequency_weights * math_ops.squared_difference(x, weighted_mean),
 764         axes,
 765         name="weighted_distsq",
 766         keepdims=True)
 767 
 768     weighted_variance = math_ops.multiply(weighted_distsq, divisor)
 769 
 770     if not keep_dims:
 771       weighted_mean = array_ops.squeeze(weighted_mean, axis=axes)
 772       weighted_variance = array_ops.squeeze(
 773           weighted_variance, axis=axes)
 774 
 775     if needs_cast:
 776       weighted_mean = math_ops.cast(weighted_mean, dtypes.float16)
 777       weighted_variance = math_ops.cast(weighted_variance, dtypes.float16)
 778 
 779     return weighted_mean, weighted_variance

 782 @tf_export("nn.batch_normalization")
 783 def batch_normalization(x,
 784                         mean,
 785                         variance,
 786                         offset,
 787                         scale,
 788                         variance_epsilon,
 789                         name=None):
 790   r"""Batch normalization.
 791 
 792   As described in http://arxiv.org/abs/1502.03167.
 793   Normalizes a tensor by `mean` and `variance`, and applies (optionally) a
 794   `scale` \\(\gamma\\) to it, as well as an `offset` \\(\beta\\):
 795 
 796   \\(\frac{\gamma(x-\mu)}{\sigma}+\beta\\)
 797 
 798   `mean`, `variance`, `offset` and `scale` are all expected to be of one of two
 799   shapes:
 800 
 801     * In all generality, they can have the same number of dimensions as the
 802       input `x`, with identical sizes as `x` for the dimensions that are not
 803       normalized over (the 'depth' dimension(s)), and dimension 1 for the
 804       others which are being normalized over.
 805       `mean` and `variance` in this case would typically be the outputs of
 806       `tf.nn.moments(..., keep_dims=True)` during training, or running averages
 807       thereof during inference.
 808     * In the common case where the 'depth' dimension is the last dimension in
 809       the input tensor `x`, they may be one dimensional tensors of the same
 810       size as the 'depth' dimension.
 811       This is the case for example for the common `[batch, depth]` layout of
 812       fully-connected layers, and `[batch, height, width, depth]` for
 813       convolutions.
 814       `mean` and `variance` in this case would typically be the outputs of
 815       `tf.nn.moments(..., keep_dims=False)` during training, or running averages
 816       thereof during inference.
 817 
 818   Args:
 819     x: Input `Tensor` of arbitrary dimensionality.
 820     mean: A mean `Tensor`.
 821     variance: A variance `Tensor`.
 822     offset: An offset `Tensor`, often denoted \\(\beta\\) in equations, or
 823       None. If present, will be added to the normalized tensor.
 824     scale: A scale `Tensor`, often denoted \\(\gamma\\) in equations, or
 825       `None`. If present, the scale is applied to the normalized tensor.
 826     variance_epsilon: A small float number to avoid dividing by 0.
 827     name: A name for this operation (optional).
 828 
 829   Returns:
 830     the normalized, scaled, offset tensor.
 831   """
 832   with ops.name_scope(name, "batchnorm", [x, mean, variance, scale, offset]):
 833     inv = math_ops.rsqrt(variance + variance_epsilon)
 834     if scale is not None:
 835       inv *= scale
 836     # Note: tensorflow/contrib/quantize/python/fold_batch_norms.py depends on
 837     # the precise order of ops that are generated by the expression below.
 838     return x * math_ops.cast(inv, x.dtype) + math_ops.cast(
 839         offset - mean * inv if offset is not None else -mean * inv, x.dtype)

 964 def _compute_sampled_logits(weights,
 965                             biases,
 966                             labels,
 967                             inputs,
 968                             num_sampled,
 969                             num_classes,
 970                             num_true=1,
 971                             sampled_values=None,
 972                             subtract_log_q=True,
 973                             remove_accidental_hits=False,
 974                             partition_strategy="mod",
 975                             name=None,
 976                             seed=None):
 977   """Helper function for nce_loss and sampled_softmax_loss functions.
 978 
 979   Computes sampled output training logits and labels suitable for implementing
 980   e.g. noise-contrastive estimation (see nce_loss) or sampled softmax (see
 981   sampled_softmax_loss).
 982 
 983   Note: In the case where num_true > 1, we assign to each target class
 984   the target probability 1 / num_true so that the target probabilities
 985   sum to 1 per-example.
 986 
 987   Args:
 988     weights: A `Tensor` of shape `[num_classes, dim]`, or a list of `Tensor`
 989         objects whose concatenation along dimension 0 has shape
 990         `[num_classes, dim]`.  The (possibly-partitioned) class embeddings.
 991     biases: A `Tensor` of shape `[num_classes]`.  The (possibly-partitioned)
 992         class biases.
 993     labels: A `Tensor` of type `int64` and shape `[batch_size,
 994         num_true]`. The target classes.  Note that this format differs from
 995         the `labels` argument of `nn.softmax_cross_entropy_with_logits_v2`.
 996     inputs: A `Tensor` of shape `[batch_size, dim]`.  The forward
 997         activations of the input network.
 998     num_sampled: An `int`.  The number of classes to randomly sample per batch.
 999     num_classes: An `int`. The number of possible classes.
1000     num_true: An `int`.  The number of target classes per training example.
1001     sampled_values: a tuple of (`sampled_candidates`, `true_expected_count`,
1002         `sampled_expected_count`) returned by a `*_candidate_sampler` function.
1003         (if None, we default to `log_uniform_candidate_sampler`)
1004     subtract_log_q: A `bool`.  whether to subtract the log expected count of
1005         the labels in the sample to get the logits of the true labels.
1006         Default is True.  Turn off for Negative Sampling.
1007     remove_accidental_hits:  A `bool`.  whether to remove "accidental hits"
1008         where a sampled class equals one of the target classes.  Default is
1009         False.
1010     partition_strategy: A string specifying the partitioning strategy, relevant
1011         if `len(weights) > 1`. Currently `"div"` and `"mod"` are supported.
1012         Default is `"mod"`. See `tf.nn.embedding_lookup` for more details.
1013     name: A name for the operation (optional).
1014     seed: random seed for candidate sampling. Default to None, which doesn't set
1015         the op-level random seed for candidate sampling.
1016   Returns:
1017     out_logits: `Tensor` object with shape
1018         `[batch_size, num_true + num_sampled]`, for passing to either
1019         `nn.sigmoid_cross_entropy_with_logits` (NCE) or
1020         `nn.softmax_cross_entropy_with_logits_v2` (sampled softmax).
1021     out_labels: A Tensor object with the same shape as `out_logits`.
1022   """
1023 
1024   if isinstance(weights, variables.PartitionedVariable):
1025     weights = list(weights)
1026   if not isinstance(weights, list):
1027     weights = [weights]
1028 
1029   with ops.name_scope(name, "compute_sampled_logits",
1030                       weights + [biases, inputs, labels]):
1031     if labels.dtype != dtypes.int64:
1032       labels = math_ops.cast(labels, dtypes.int64)
1033     labels_flat = array_ops.reshape(labels, [-1])
1034 
1035     # Sample the negative labels.
1036     #   sampled shape: [num_sampled] tensor
1037     #   true_expected_count shape = [batch_size, 1] tensor
1038     #   sampled_expected_count shape = [num_sampled] tensor
1039     if sampled_values is None:
1040       sampled_values = candidate_sampling_ops.log_uniform_candidate_sampler(
1041           true_classes=labels,
1042           num_true=num_true,
1043           num_sampled=num_sampled,
1044           unique=True,
1045           range_max=num_classes,
1046           seed=seed)
1047     # NOTE: pylint cannot tell that 'sampled_values' is a sequence
1048     # pylint: disable=unpacking-non-sequence
1049     sampled, true_expected_count, sampled_expected_count = (
1050         array_ops.stop_gradient(s) for s in sampled_values)
1051     # pylint: enable=unpacking-non-sequence
1052     sampled = math_ops.cast(sampled, dtypes.int64)
1053 
1054     # labels_flat is a [batch_size * num_true] tensor
1055     # sampled is a [num_sampled] int tensor
1056     all_ids = array_ops.concat([labels_flat, sampled], 0)
1057 
1058     # Retrieve the true weights and the logits of the sampled weights.
1059 
1060     # weights shape is [num_classes, dim]
1061     all_w = embedding_ops.embedding_lookup(
1062         weights, all_ids, partition_strategy=partition_strategy)
1063 
1064     # true_w shape is [batch_size * num_true, dim]
1065     true_w = array_ops.slice(all_w, [0, 0],
1066                              array_ops.stack(
1067                                  [array_ops.shape(labels_flat)[0], -1]))
1068 
1069     sampled_w = array_ops.slice(
1070         all_w, array_ops.stack([array_ops.shape(labels_flat)[0], 0]), [-1, -1])
1071     # inputs has shape [batch_size, dim]
1072     # sampled_w has shape [num_sampled, dim]
1073     # Apply X*W', which yields [batch_size, num_sampled]
1074     sampled_logits = math_ops.matmul(inputs, sampled_w, transpose_b=True)
1075 
1076     # Retrieve the true and sampled biases, compute the true logits, and
1077     # add the biases to the true and sampled logits.
1078     all_b = embedding_ops.embedding_lookup(
1079         biases, all_ids, partition_strategy=partition_strategy)
1080     # true_b is a [batch_size * num_true] tensor
1081     # sampled_b is a [num_sampled] float tensor
1082     true_b = array_ops.slice(all_b, [0], array_ops.shape(labels_flat))
1083     sampled_b = array_ops.slice(all_b, array_ops.shape(labels_flat), [-1])
1084 
1085     # inputs shape is [batch_size, dim]
1086     # true_w shape is [batch_size * num_true, dim]
1087     # row_wise_dots is [batch_size, num_true, dim]
1088     dim = array_ops.shape(true_w)[1:2]
1089     new_true_w_shape = array_ops.concat([[-1, num_true], dim], 0)
1090     row_wise_dots = math_ops.multiply(
1091         array_ops.expand_dims(inputs, 1),
1092         array_ops.reshape(true_w, new_true_w_shape))
1093     # We want the row-wise dot plus biases which yields a
1094     # [batch_size, num_true] tensor of true_logits.
1095     dots_as_matrix = array_ops.reshape(row_wise_dots,
1096                                        array_ops.concat([[-1], dim], 0))
1097     true_logits = array_ops.reshape(_sum_rows(dots_as_matrix), [-1, num_true])
1098     true_b = array_ops.reshape(true_b, [-1, num_true])
1099     true_logits += true_b
1100     sampled_logits += sampled_b
1101 
1102     if remove_accidental_hits:
1103       acc_hits = candidate_sampling_ops.compute_accidental_hits(
1104           labels, sampled, num_true=num_true)
1105       acc_indices, acc_ids, acc_weights = acc_hits
1106 
1107       # This is how SparseToDense expects the indices.
1108       acc_indices_2d = array_ops.reshape(acc_indices, [-1, 1])
1109       acc_ids_2d_int32 = array_ops.reshape(
1110           math_ops.cast(acc_ids, dtypes.int32), [-1, 1])
1111       sparse_indices = array_ops.concat([acc_indices_2d, acc_ids_2d_int32], 1,
1112                                         "sparse_indices")
1113       # Create sampled_logits_shape = [batch_size, num_sampled]
1114       sampled_logits_shape = array_ops.concat(
1115           [array_ops.shape(labels)[:1],
1116            array_ops.expand_dims(num_sampled, 0)], 0)
1117       if sampled_logits.dtype != acc_weights.dtype:
1118         acc_weights = math_ops.cast(acc_weights, sampled_logits.dtype)
1119       sampled_logits += sparse_ops.sparse_to_dense(
1120           sparse_indices,
1121           sampled_logits_shape,
1122           acc_weights,
1123           default_value=0.0,
1124           validate_indices=False)
1125 
1126     if subtract_log_q:
1127       # Subtract log of Q(l), prior probability that l appears in sampled.
1128       true_logits -= math_ops.log(true_expected_count)
1129       sampled_logits -= math_ops.log(sampled_expected_count)
1130 
1131     # Construct output logits and labels. The true labels/logits start at col 0.
1132     out_logits = array_ops.concat([true_logits, sampled_logits], 1)
1133 
1134     # true_logits is a float tensor, ones_like(true_logits) is a float
1135     # tensor of ones. We then divide by num_true to ensure the per-example
1136     # labels sum to 1.0, i.e. form a proper probability distribution.
1137     out_labels = array_ops.concat([
1138         array_ops.ones_like(true_logits) / num_true,
1139         array_ops.zeros_like(sampled_logits)
1140     ], 1)
1141 
1142     return out_logits, out_labels

1145 @tf_export("nn.nce_loss")
1146 def nce_loss(weights,
1147              biases,
1148              labels,
1149              inputs,
1150              num_sampled,
1151              num_classes,
1152              num_true=1,
1153              sampled_values=None,
1154              remove_accidental_hits=False,
1155              partition_strategy="mod",
1156              name="nce_loss"):
1157   """Computes and returns the noise-contrastive estimation training loss.
1158 
1159   See [Noise-contrastive estimation: A new estimation principle for
1160   unnormalized statistical
1161   models](http://www.jmlr.org/proceedings/papers/v9/gutmann10a/gutmann10a.pdf).
1162   Also see our [Candidate Sampling Algorithms
1163   Reference](https://www.tensorflow.org/extras/candidate_sampling.pdf)
1164 
1165   A common use case is to use this method for training, and calculate the full
1166   sigmoid loss for evaluation or inference. In this case, you must set
1167   `partition_strategy="div"` for the two losses to be consistent, as in the
1168   following example:
1169 
1170   ```python
1171   if mode == "train":
1172     loss = tf.nn.nce_loss(
1173         weights=weights,
1174         biases=biases,
1175         labels=labels,
1176         inputs=inputs,
1177         ...,
1178         partition_strategy="div")
1179   elif mode == "eval":
1180     logits = tf.matmul(inputs, tf.transpose(weights))
1181     logits = tf.nn.bias_add(logits, biases)
1182     labels_one_hot = tf.one_hot(labels, n_classes)
1183     loss = tf.nn.sigmoid_cross_entropy_with_logits(
1184         labels=labels_one_hot,
1185         logits=logits)
1186     loss = tf.reduce_sum(loss, axis=1)
1187   ```
1188 
1189   Note: By default this uses a log-uniform (Zipfian) distribution for sampling,
1190   so your labels must be sorted in order of decreasing frequency to achieve
1191   good results.  For more details, see
1192   `tf.nn.log_uniform_candidate_sampler`.
1193 
1194   Note: In the case where `num_true` > 1, we assign to each target class
1195   the target probability 1 / `num_true` so that the target probabilities
1196   sum to 1 per-example.
1197 
1198   Note: It would be useful to allow a variable number of target classes per
1199   example.  We hope to provide this functionality in a future release.
1200   For now, if you have a variable number of target classes, you can pad them
1201   out to a constant number by either repeating them or by padding
1202   with an otherwise unused class.
1203 
1204   Args:
1205     weights: A `Tensor` of shape `[num_classes, dim]`, or a list of `Tensor`
1206         objects whose concatenation along dimension 0 has shape
1207         [num_classes, dim].  The (possibly-partitioned) class embeddings.
1208     biases: A `Tensor` of shape `[num_classes]`.  The class biases.
1209     labels: A `Tensor` of type `int64` and shape `[batch_size,
1210         num_true]`. The target classes.
1211     inputs: A `Tensor` of shape `[batch_size, dim]`.  The forward
1212         activations of the input network.
1213     num_sampled: An `int`.  The number of negative classes to randomly sample
1214         per batch. This single sample of negative classes is evaluated for each
1215         element in the batch.
1216     num_classes: An `int`. The number of possible classes.
1217     num_true: An `int`.  The number of target classes per training example.
1218     sampled_values: a tuple of (`sampled_candidates`, `true_expected_count`,
1219         `sampled_expected_count`) returned by a `*_candidate_sampler` function.
1220         (if None, we default to `log_uniform_candidate_sampler`)
1221     remove_accidental_hits:  A `bool`.  Whether to remove "accidental hits"
1222         where a sampled class equals one of the target classes.  If set to
1223         `True`, this is a "Sampled Logistic" loss instead of NCE, and we are
1224         learning to generate log-odds instead of log probabilities.  See
1225         our [Candidate Sampling Algorithms Reference]
1226         (https://www.tensorflow.org/extras/candidate_sampling.pdf).
1227         Default is False.
1228     partition_strategy: A string specifying the partitioning strategy, relevant
1229         if `len(weights) > 1`. Currently `"div"` and `"mod"` are supported.
1230         Default is `"mod"`. See `tf.nn.embedding_lookup` for more details.
1231     name: A name for the operation (optional).
1232 
1233   Returns:
1234     A `batch_size` 1-D tensor of per-example NCE losses.
1235   """
1236   logits, labels = _compute_sampled_logits(
1237       weights=weights,
1238       biases=biases,
1239       labels=labels,
1240       inputs=inputs,
1241       num_sampled=num_sampled,
1242       num_classes=num_classes,
1243       num_true=num_true,
1244       sampled_values=sampled_values,
1245       subtract_log_q=True,
1246       remove_accidental_hits=remove_accidental_hits,
1247       partition_strategy=partition_strategy,
1248       name=name)
1249   sampled_losses = sigmoid_cross_entropy_with_logits(
1250       labels=labels, logits=logits, name="sampled_losses")
1251   # sampled_losses is batch_size x {true_loss, sampled_losses...}
1252   # We sum out true and sampled losses.
1253   return _sum_rows(sampled_losses)

1335   """

1339       labels=labels,
1340       inputs=inputs,
