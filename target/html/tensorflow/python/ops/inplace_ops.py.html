  28 def _inplace_helper(x, i, v, op):
  29   """Applies an inplace op on (x, i, v).
  30 
  31   op is one of gen_array_ops.alias_inplace_update,
  32   gen_array_ops.alias_inplace_add, or gen_array_ops.alias_inplace_sub.
  33 
  34   If i is None, x and v must be the same shape. Computes
  35     x op v;
  36   If i is a scalar, x has a rank 1 higher than v's. Computes
  37     x[i, :] op v;
  38   Otherwise, x and v must have the same rank. Computes
  39     x[i, :] op v;
  40 
  41   Args:
  42     x: A Tensor.
  43     i: None, a scalar or a vector.
  44     v: A Tensor.
  45     op: alias_inplace_update, alias_inplace_add, or alias_inplace_sub.
  46 
  47   Returns:
  48     Returns x.
  49 
  50   """
  51   x = ops.convert_to_tensor(x)
  52   v = ops.convert_to_tensor(v, x.dtype)
  53   if i is None:
  54     # Full tensor.
  55     return array_ops.reshape(
  56         op(array_ops.reshape(x, [1, -1]), [0], array_ops.reshape(v, [1, -1])),
  57         array_ops.shape(x))
  58   i = math_ops.to_int32(i)
  59   if i.get_shape().ndims == 0:
  60     # Single 0-dim update.
  61     return op(x, array_ops.reshape(i, [1]), array_ops.expand_dims(v, 0))
  62   return op(x, i, v)

  65 def alias_inplace_update(x, i, v):
  66   """Applies an inplace update on input x at index i with value v. Aliases x.
  67 
  68   If i is None, x and v must be the same shape. Computes
  69     x = v;
  70   If i is a scalar, x has a rank 1 higher than v's. Computes
  71     x[i, :] = v;
  72   Otherwise, x and v must have the same rank. Computes
  73     x[i, :] = v;
  74 
  75   Args:
  76     x: A Tensor.
  77     i: None, a scalar or a vector.
  78     v: A Tensor.
  79 
  80   Returns:
  81     Returns x.
  82 
  83   """
  84   return _inplace_helper(x, i, v, gen_array_ops.inplace_update)

 131 def empty_like(x, init=None):
 132   """Returns a non-initialized tensor with the same shape and dtype as x.
 133 
 134   Args:
 135     x: A Tensor.
 136     init: Initialize the returned tensor with the default value of
 137       x.dtype(), if True. Otherwise, do not initialize. Defaults to
 138       None.
 139 
 140   Returns:
 141     A tensor y, whose dtype and shape are the same as those of x.
 142     y is guaranteed not to be an alias of x. Upon return, y may contain
 143     arbitrary data.
 144 
 145   """
 146   x = ops.convert_to_tensor(x)
 147   return gen_array_ops.empty(array_ops.shape(x), x.dtype, init=init)

 150 def inplace_update(x, i, v):
 151   """Applies an inplace update on input x at index i with value v.
 152 
 153   Note that this function is not actually inplace - it allocates
 154   a copy of x.  The utility is not avoiding memory copies but rather
 155   specifying a sparse update.
 156 
 157   If i is None, x and v must be the same shape. Computes
 158     y = x; y = v;
 159   If i is a scalar, x has a rank 1 higher than v's. Computes
 160     y = x; y[i, :] = v;
 161   Otherwise, x and v must have the same rank. Computes
 162     y = x; y[i, :] = v;
 163 
 164   Args:
 165     x: A Tensor.
 166     i: None, a scalar or a vector.
 167     v: A Tensor.
 168 
 169   Returns:
 170     Returns y, which is guaranteed not to be an alias of x.
 171 
 172   """
 173   return alias_inplace_update(gen_array_ops.deep_copy(x), i, v)
