  57 def should_record_summaries():
  58   """Returns boolean Tensor which is true if summaries should be recorded."""
  59   should_record_collection = ops.get_collection(_SHOULD_RECORD_SUMMARIES_NAME)
  60   if not should_record_collection:
  61     return False
  62   if len(should_record_collection) != 1:
  63     raise ValueError(
  64         "More than one tensor specified for whether summaries "
  65         "should be recorded: %s" % should_record_collection)
  66   return should_record_collection[0]

  71 def record_summaries_every_n_global_steps(n, global_step=None):
  72   """Sets the should_record_summaries Tensor to true if global_step % n == 0."""
  73   if global_step is None:
  74     global_step = training_util.get_or_create_global_step()
  75   collection_ref = ops.get_collection_ref(_SHOULD_RECORD_SUMMARIES_NAME)
  76   old = collection_ref[:]
  77   try:
  78     with ops.device("cpu:0"):
  79       collection_ref[:] = [math_ops.equal(global_step % n, 0)]
  80     yield
  81   finally:
  82     collection_ref[:] = old

 109 class SummaryWriter(object):
 110   """Encapsulates a stateful summary writer resource.
 111 
 112   See also:
 113   - `tf.contrib.summary.create_file_writer`
 114   - `tf.contrib.summary.create_db_writer`
 115   """
 116 
 117   def  __init__(self, resource, init_op_fn):
 118     self._resource = resource
 119     # TODO(nickfelt): cache constructed ops in graph mode
 120     self._init_op_fn = init_op_fn
 121     if context.executing_eagerly() and self._resource is not None:
 122       self._resource_deleter = resource_variable_ops.EagerResourceDeleter(
 123           handle=self._resource, handle_device="cpu:0")
 124 
 125   def set_as_default(self):
 126     """Enables this summary writer for the current thread."""
 127     context.context().summary_writer_resource = self._resource
 128 
 129   @tf_contextlib.contextmanager
 130   def as_default(self):
 131     """Enables summary writing within a `with` block."""
 132     if self._resource is None:
 133       yield self
 134     else:
 135       old = context.context().summary_writer_resource
 136       try:
 137         context.context().summary_writer_resource = self._resource
 138         yield self
 139         # Flushes the summary writer in eager mode or in graph functions, but
 140         # not in legacy graph mode (you're on your own there).
 141         with ops.device("cpu:0"):
 142           gen_summary_ops.flush_summary_writer(self._resource)
 143       finally:
 144         context.context().summary_writer_resource = old
 145 
 146 
 147   def init(self):
 148     """Operation to initialize the summary writer resource."""
 149     if self._resource is not None:
 150       return self._init_op_fn()
 151 
 152   def _flush(self):
 153     return _flush_fn(writer=self)
 154 
 155   def flush(self):
 156     """Operation to force the summary writer to flush any buffered data."""
 157     if self._resource is not None:
 158       return self._flush()
 159 
 160   def _close(self):
 161     with ops.control_dependencies([self.flush()]):
 162       with ops.device("cpu:0"):
 163         return gen_summary_ops.close_summary_writer(self._resource)
 164 
 165   def close(self):
 166     """Operation to flush and close the summary writer resource."""
 167     if self._resource is not None:
 168       return self._close()

 171 def initialize(
 172     graph=None,  # pylint: disable=redefined-outer-name
 173     session=None):
 174   """Initializes summary writing for graph execution mode.
 175 
 176   This helper method provides a higher-level alternative to using
 177   `tf.contrib.summary.summary_writer_initializer_op` and
 178   `tf.contrib.summary.graph`.
 179 
 180   Most users will also want to call `tf.train.create_global_step`
 181   which can happen before or after this function is called.
 182 
 183   Args:
 184     graph: A `tf.Graph` or `tf.GraphDef` to output to the writer.
 185       This function will not write the default graph by default. When
 186       writing to an event log file, the associated step will be zero.
 187     session: So this method can call `tf.Session.run`. This defaults
 188       to `tf.get_default_session`.
 189 
 190   Raises:
 191     RuntimeError: If  the current thread has no default
 192       `tf.contrib.summary.SummaryWriter`.
 193     ValueError: If session wasn't passed and no default session.
 194   """
 195   if context.executing_eagerly():
 196     return
 197   if context.context().summary_writer_resource is None:
 198     raise RuntimeError("No default tf.contrib.summary.SummaryWriter found")
 199   if session is None:
 200     session = ops.get_default_session()
 201     if session is None:
 202       raise ValueError("session must be passed if no default session exists")
 203   session.run(summary_writer_initializer_op())
 204   if graph is not None:
 205     data = _serialize_graph(graph)
 206     x = array_ops.placeholder(dtypes.string)
 207     session.run(_graph(x, 0), feed_dict={x: data})

 210 def create_file_writer(logdir,
 211                        max_queue=None,
 212                        flush_millis=None,
 213                        filename_suffix=None,
 214                        name=None):
 215   """Creates a summary file writer in the current context under the given name.
 216 
 217   Args:
 218     logdir: a string, or None. If a string, creates a summary file writer
 219      which writes to the directory named by the string. If None, returns
 220      a mock object which acts like a summary writer but does nothing,
 221      useful to use as a context manager.
 222     max_queue: the largest number of summaries to keep in a queue; will
 223      flush once the queue gets bigger than this. Defaults to 10.
 224     flush_millis: the largest interval between flushes. Defaults to 120,000.
 225     filename_suffix: optional suffix for the event file name. Defaults to `.v2`.
 226     name: Shared name for this SummaryWriter resource stored to default
 227       Graph. Defaults to the provided logdir prefixed with `logdir:`. Note: if a
 228       summary writer resource with this shared name already exists, the returned
 229       SummaryWriter wraps that resource and the other arguments have no effect.
 230 
 231   Returns:
 232     Either a summary writer or an empty object which can be used as a
 233     summary writer.
 234   """
 235   if logdir is None:
 236     return SummaryWriter(None, None)
 237   logdir = str(logdir)
 238   with ops.device("cpu:0"):
 239     if max_queue is None:
 240       max_queue = constant_op.constant(10)
 241     if flush_millis is None:
 242       flush_millis = constant_op.constant(2 * 60 * 1000)
 243     if filename_suffix is None:
 244       filename_suffix = constant_op.constant(".v2")
 245     if name is None:
 246       name = "logdir:" + logdir
 247     return _make_summary_writer(
 248         name,
 249         gen_summary_ops.create_summary_file_writer,
 250         logdir=logdir,
 251         max_queue=max_queue,
 252         flush_millis=flush_millis,
 253         filename_suffix=filename_suffix)

 256 def create_db_writer(db_uri,
 257                      experiment_name=None,
 258                      run_name=None,
 259                      user_name=None,
 260                      name=None):
 261   """Creates a summary database writer in the current context.
 262 
 263   This can be used to write tensors from the execution graph directly
 264   to a database. Only SQLite is supported right now. This function
 265   will create the schema if it doesn't exist. Entries in the Users,
 266   Experiments, and Runs tables will be created automatically if they
 267   don't already exist.
 268 
 269   Args:
 270     db_uri: For example "file:/tmp/foo.sqlite".
 271     experiment_name: Defaults to YYYY-MM-DD in local time if None.
 272       Empty string means the Run will not be associated with an
 273       Experiment. Can't contain ASCII control characters or <>. Case
 274       sensitive.
 275     run_name: Defaults to HH:MM:SS in local time if None. Empty string
 276       means a Tag will not be associated with any Run. Can't contain
 277       ASCII control characters or <>. Case sensitive.
 278     user_name: Defaults to system username if None. Empty means the
 279       Experiment will not be associated with a User. Must be valid as
 280       both a DNS label and Linux username.
 281     name: Shared name for this SummaryWriter resource stored to default
 282       `tf.Graph`.
 283 
 284   Returns:
 285     A `tf.contrib.summary.SummaryWriter` instance.
 286   """
 287   with ops.device("cpu:0"):
 288     if experiment_name is None:
 289       experiment_name = time.strftime("%Y-%m-%d", time.localtime(time.time()))
 290     if run_name is None:
 291       run_name = time.strftime("%H:%M:%S", time.localtime(time.time()))
 292     if user_name is None:
 293       user_name = getpass.getuser()
 294     experiment_name = _cleanse_string(
 295         "experiment_name", _EXPERIMENT_NAME_PATTERNS, experiment_name)
 296     run_name = _cleanse_string("run_name", _RUN_NAME_PATTERNS, run_name)
 297     user_name = _cleanse_string("user_name", _USER_NAME_PATTERNS, user_name)
 298     return _make_summary_writer(
 299         name,
 300         gen_summary_ops.create_summary_db_writer,
 301         db_uri=db_uri,
 302         experiment_name=experiment_name,
 303         run_name=run_name,
 304         user_name=user_name)

 307 def _make_summary_writer(name, factory, **kwargs):
 308   resource = gen_summary_ops.summary_writer(shared_name=name)
 309   init_op_fn = lambda: factory(resource, **kwargs)
 310   init_op = init_op_fn()
 311   if not context.executing_eagerly():
 312     # TODO(apassos): Consider doing this instead.
 313     #   ops.get_default_session().run(init_op)
 314     ops.add_to_collection(_SUMMARY_WRITER_INIT_COLLECTION_NAME, init_op)
 315   return SummaryWriter(resource, init_op_fn)

 318 def _cleanse_string(name, pattern, value):
 319   if isinstance(value, six.string_types) and pattern.search(value) is None:
 320     raise ValueError("%s (%s) must match %s" % (name, value, pattern.pattern))
 321   return ops.convert_to_tensor(value, dtypes.string)

 358 def summary_writer_function(name, tensor, function, family=None):
 359   """Helper function to write summaries.
 360 
 361   Args:
 362     name: name of the summary
 363     tensor: main tensor to form the summary
 364     function: function taking a tag and a scope which writes the summary
 365     family: optional, the summary's family
 366 
 367   Returns:
 368     The result of writing the summary.
 369   """
 370   name_scope = ops.get_name_scope()
 371   if name_scope:
 372     # Add a slash to allow reentering the name scope.
 373     name_scope += "/"
 374   def record():
 375     with ops.name_scope(name_scope), summary_op_util.summary_scope(
 376         name, family, values=[tensor]) as (tag, scope):
 377       with ops.control_dependencies([function(tag, scope)]):
 378         return constant_op.constant(True)
 379 
 380   if context.context().summary_writer_resource is None:
 381     return control_flow_ops.no_op()
 382   with ops.device("cpu:0"):
 383     op = smart_cond.smart_cond(
 384         should_record_summaries(), record, _nothing, name="")
 385     if not context.executing_eagerly():
 386       ops.add_to_collection(ops.GraphKeys._SUMMARY_COLLECTION, op)  # pylint: disable=protected-access
 387   return op

 390 def generic(name, tensor, metadata=None, family=None, step=None):
 391   """Writes a tensor summary if possible."""
 392 
 393   def function(tag, scope):
 394     if metadata is None:
 395       serialized_metadata = constant_op.constant("")
 396     elif hasattr(metadata, "SerializeToString"):
 397       serialized_metadata = constant_op.constant(metadata.SerializeToString())
 398     else:
 399       serialized_metadata = metadata
 400     # Note the identity to move the tensor to the CPU.
 401     return gen_summary_ops.write_summary(
 402         context.context().summary_writer_resource,
 403         _choose_step(step),
 404         array_ops.identity(tensor),
 405         tag,
 406         serialized_metadata,
 407         name=scope)
 408   return summary_writer_function(name, tensor, function, family=family)

 494 def graph(param, step=None, name=None):
 495   """Writes a TensorFlow graph to the summary interface.
 496 
 497   The graph summary is, strictly speaking, not a summary. Conditions
 498   like `tf.contrib.summary.never_record_summaries` do not apply. Only
 499   a single graph can be associated with a particular run. If multiple
 500   graphs are written, then only the last one will be considered by
 501   TensorBoard.
 502 
 503   When not using eager execution mode, the user should consider passing
 504   the `graph` parameter to `tf.contrib.summary.initialize` instead of
 505   calling this function. Otherwise special care needs to be taken when
 506   using the graph to record the graph.
 507 
 508   Args:
 509     param: A `tf.Tensor` containing a serialized graph proto. When
 510       eager execution is enabled, this function will automatically
 511       coerce `tf.Graph`, `tf.GraphDef`, and string types.
 512     step: The global step variable. This doesn't have useful semantics
 513       for graph summaries, but is used anyway, due to the structure of
 514       event log files. This defaults to the global step.
 515     name: A name for the operation (optional).
 516 
 517   Returns:
 518     The created `tf.Operation` or a `tf.no_op` if summary writing has
 519     not been enabled for this context.
 520 
 521   Raises:
 522     TypeError: If `param` isn't already a `tf.Tensor` in graph mode.
 523   """
 524   if not context.executing_eagerly() and not isinstance(param, ops.Tensor):
 525     raise TypeError("graph() needs a tf.Tensor (e.g. tf.placeholder) in graph "
 526                     "mode, but was: %s" % type(param))
 527   writer = context.context().summary_writer_resource
 528   if writer is None:
 529     return control_flow_ops.no_op()
 530   with ops.device("cpu:0"):
 531     if isinstance(param, (ops.Graph, graph_pb2.GraphDef)):
 532       tensor = ops.convert_to_tensor(_serialize_graph(param), dtypes.string)
 533     else:
 534       tensor = array_ops.identity(param)
 535     return gen_summary_ops.write_graph_summary(
 536         writer, _choose_step(step), tensor, name=name)
 537

 563 def flush(writer=None, name=None):
 564   """Forces summary writer to send any buffered data to storage.
 565 
 566   This operation blocks until that finishes.
 567 
 568   Args:
 569     writer: The `tf.contrib.summary.SummaryWriter` resource to flush.
 570       The thread default will be used if this parameter is None.
 571       Otherwise a `tf.no_op` is returned.
 572     name: A name for the operation (optional).
 573 
 574   Returns:
 575     The created `tf.Operation`.
 576   """
 577   if writer is None:
 578     writer = context.context().summary_writer_resource
 579     if writer is None:
 580       return control_flow_ops.no_op()
 581   else:
 582     if isinstance(writer, SummaryWriter):
 583       writer = writer._resource  # pylint: disable=protected-access
 584   with ops.device("cpu:0"):
 585     return gen_summary_ops.flush_summary_writer(writer, name=name)

 610 def _choose_step(step):
 611   if step is None:
 612     return training_util.get_or_create_global_step()
 613   if not isinstance(step, ops.Tensor):
 614     return ops.convert_to_tensor(step, dtypes.int64)
 615   return step
