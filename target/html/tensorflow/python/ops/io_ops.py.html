  42 def _save(filename, tensor_names, tensors, tensor_slices=None, name="save"):
  43   """Save a list of tensors to a file with given names.
  44 
  45   Example usage without slice info:
  46     Save("/foo/bar", ["w", "b"], [w, b])
  47 
  48   Example usage with slices:
  49     Save("/foo/bar", ["w", "w"], [slice0, slice1],
  50          tensor_slices=["4 10 0,2:-", "4 10 2,2:-"])
  51 
  52   Args:
  53     filename: the file name of the sstable.
  54     tensor_names: a list of strings.
  55     tensors: the list of tensors to be saved.
  56     tensor_slices: Optional list of strings to specify the shape and slices of
  57       a larger virtual tensor that each tensor is a part of.  If not specified
  58       each tensor is saved as a full slice.
  59     name: string.  Optional name for the op.
  60 
  61   Requires:
  62     The length of tensors should match the size of tensor_names and of
  63     tensor_slices.
  64 
  65   Returns:
  66     An Operation that saves the tensors.
  67   """
  68   if tensor_slices is None:
  69     return gen_io_ops.save(filename, tensor_names, tensors, name=name)
  70   else:
  71     return gen_io_ops.save_slices(filename, tensor_names, tensor_slices,
  72                                   tensors, name=name)

  75 def _restore_slice(file_pattern, tensor_name, shape_and_slice, tensor_type,
  76                    name="restore_slice", preferred_shard=-1):
  77   """Restore a tensor slice from a set of files with a given pattern.
  78 
  79   Example usage:
  80     RestoreSlice("/foo/bar-?????-of-?????", "w", "10 10 0,2:-", DT_FLOAT)
  81 
  82   Args:
  83     file_pattern: the file pattern used to match a set of checkpoint files.
  84     tensor_name: the name of the tensor to restore.
  85     shape_and_slice: the shape-and-slice spec of the slice.
  86     tensor_type: the type of the tensor to restore.
  87     name: string.  Optional name for the op.
  88     preferred_shard: Int. Optional shard to open first in the checkpoint file.
  89 
  90   Returns:
  91     A tensor of type "tensor_type".
  92   """
  93   base_type = dtypes.as_dtype(tensor_type).base_dtype
  94   return gen_io_ops.restore_slice(
  95       file_pattern, tensor_name, shape_and_slice, base_type,
  96       preferred_shard, name=name)

  99 @tf_export(v1=["ReaderBase"])
 100 class ReaderBase(object):
 101   """Base class for different Reader types, that produce a record every step.
 102 
 103   Conceptually, Readers convert string 'work units' into records (key,
 104   value pairs).  Typically the 'work units' are filenames and the
 105   records are extracted from the contents of those files.  We want a
 106   single record produced per step, but a work unit can correspond to
 107   many records.
 108 
 109   Therefore we introduce some decoupling using a queue.  The queue
 110   contains the work units and the Reader dequeues from the queue when
 111   it is asked to produce a record (via Read()) but it has finished the
 112   last work unit.
 113 
 114   @compatibility(eager)
 115   Readers are not compatible with eager execution. Instead, please
 116   use `tf.data` to get data into your model.
 117   @end_compatibility
 118   """
 119 
 120   def __init__(self, reader_ref, supports_serialize=False):
 121     """Creates a new ReaderBase.
 122 
 123     Args:
 124       reader_ref: The operation that implements the reader.
 125       supports_serialize: True if the reader implementation can
 126         serialize its state.
 127 
 128     Raises:
 129       RuntimeError: If eager execution is enabled.
 130     """
 131     if context.executing_eagerly():
 132       raise RuntimeError(
 133           "Readers are not supported when eager execution is enabled. "
 134           "Instead, please use tf.data to get data into your model.")
 135 
 136     self._reader_ref = reader_ref
 137     self._supports_serialize = supports_serialize
 138 
 139   @property
 140   def reader_ref(self):
 141     """Op that implements the reader."""
 142     return self._reader_ref
 143 
 144   def read(self, queue, name=None):
 145     """Returns the next record (key, value) pair produced by a reader.
 146 
 147     Will dequeue a work unit from queue if necessary (e.g. when the
 148     Reader needs to start reading from a new file since it has
 149     finished with the previous file).
 150 
 151     Args:
 152       queue: A Queue or a mutable string Tensor representing a handle
 153         to a Queue, with string work items.
 154       name: A name for the operation (optional).
 155 
 156     Returns:
 157       A tuple of Tensors (key, value).
 158       key: A string scalar Tensor.
 159       value: A string scalar Tensor.
 160     """
 161     if isinstance(queue, ops.Tensor):
 162       queue_ref = queue
 163     else:
 164       queue_ref = queue.queue_ref
 165     if self._reader_ref.dtype == dtypes.resource:
 166       return gen_io_ops.reader_read_v2(self._reader_ref, queue_ref, name=name)
 167     else:
 168       # For compatibility with pre-resource queues, create a ref(string) tensor
 169       # which can be looked up as the same queue by a resource manager.
 170       old_queue_op = gen_data_flow_ops.fake_queue(queue_ref)
 171       return gen_io_ops.reader_read(self._reader_ref, old_queue_op, name=name)
 172 
 173   def read_up_to(self, queue, num_records,  # pylint: disable=invalid-name
 174                  name=None):
 175     """Returns up to num_records (key, value) pairs produced by a reader.
 176 
 177     Will dequeue a work unit from queue if necessary (e.g., when the
 178     Reader needs to start reading from a new file since it has
 179     finished with the previous file).
 180     It may return less than num_records even before the last batch.
 181 
 182     Args:
 183       queue: A Queue or a mutable string Tensor representing a handle
 184         to a Queue, with string work items.
 185       num_records: Number of records to read.
 186       name: A name for the operation (optional).
 187 
 188     Returns:
 189       A tuple of Tensors (keys, values).
 190       keys: A 1-D string Tensor.
 191       values: A 1-D string Tensor.
 192     """
 193     if isinstance(queue, ops.Tensor):
 194       queue_ref = queue
 195     else:
 196       queue_ref = queue.queue_ref
 197     if self._reader_ref.dtype == dtypes.resource:
 198       return gen_io_ops.reader_read_up_to_v2(self._reader_ref,
 199                                              queue_ref,
 200                                              num_records,
 201                                              name=name)
 202     else:
 203       # For compatibility with pre-resource queues, create a ref(string) tensor
 204       # which can be looked up as the same queue by a resource manager.
 205       old_queue_op = gen_data_flow_ops.fake_queue(queue_ref)
 206       return gen_io_ops.reader_read_up_to(self._reader_ref,
 207                                           old_queue_op,
 208                                           num_records,
 209                                           name=name)
 210 
 211   def num_records_produced(self, name=None):
 212     """Returns the number of records this reader has produced.
 213 
 214     This is the same as the number of Read executions that have
 215     succeeded.
 216 
 217     Args:
 218       name: A name for the operation (optional).
 219 
 220     Returns:
 221       An int64 Tensor.
 222 
 223     """
 224     if self._reader_ref.dtype == dtypes.resource:
 225       return gen_io_ops.reader_num_records_produced_v2(self._reader_ref,
 226                                                        name=name)
 227     else:
 228       return gen_io_ops.reader_num_records_produced(self._reader_ref,
 229                                                     name=name)
 230 
 231   def num_work_units_completed(self, name=None):
 232     """Returns the number of work units this reader has finished processing.
 233 
 234     Args:
 235       name: A name for the operation (optional).
 236 
 237     Returns:
 238       An int64 Tensor.
 239     """
 240     if self._reader_ref.dtype == dtypes.resource:
 241       return gen_io_ops.reader_num_work_units_completed_v2(self._reader_ref,
 242                                                            name=name)
 243     else:
 244       return gen_io_ops.reader_num_work_units_completed(self._reader_ref,
 245                                                         name=name)
 246 
 247   def serialize_state(self, name=None):
 248     """Produce a string tensor that encodes the state of a reader.
 249 
 250     Not all Readers support being serialized, so this can produce an
 251     Unimplemented error.
 252 
 253     Args:
 254       name: A name for the operation (optional).
 255 
 256     Returns:
 257       A string Tensor.
 258     """
 259     if self._reader_ref.dtype == dtypes.resource:
 260       return gen_io_ops.reader_serialize_state_v2(self._reader_ref, name=name)
 261     else:
 262       return gen_io_ops.reader_serialize_state(self._reader_ref, name=name)
 263 
 264   def restore_state(self, state, name=None):
 265     """Restore a reader to a previously saved state.
 266 
 267     Not all Readers support being restored, so this can produce an
 268     Unimplemented error.
 269 
 270     Args:
 271       state: A string Tensor.
 272         Result of a SerializeState of a Reader with matching type.
 273       name: A name for the operation (optional).
 274 
 275     Returns:
 276       The created Operation.
 277     """
 278     if self._reader_ref.dtype == dtypes.resource:
 279       return gen_io_ops.reader_restore_state_v2(
 280           self._reader_ref, state, name=name)
 281     else:
 282       return gen_io_ops.reader_restore_state(self._reader_ref, state, name=name)
 283 
 284   @property
 285   def supports_serialize(self):
 286     """Whether the Reader implementation can serialize its state."""
 287     return self._supports_serialize
 288 
 289   def reset(self, name=None):
 290     """Restore a reader to its initial clean state.
 291 
 292     Args:
 293       name: A name for the operation (optional).
 294 
 295     Returns:
 296       The created Operation.
 297     """
 298     if self._reader_ref.dtype == dtypes.resource:
 299       return gen_io_ops.reader_reset_v2(self._reader_ref, name=name)
 300     else:
 301       return gen_io_ops.reader_reset(self._reader_ref, name=name)

 313 @tf_export(v1=["WholeFileReader"])
 314 class WholeFileReader(ReaderBase):
 315   """A Reader that outputs the entire contents of a file as a value.
 316 
 317   To use, enqueue filenames in a Queue.  The output of Read will
 318   be a filename (key) and the contents of that file (value).
 319 
 320   See ReaderBase for supported methods.
 321 
 322   @compatibility(eager)
 323   Readers are not compatible with eager execution. Instead, please
 324   use `tf.data` to get data into your model.
 325   @end_compatibility
 326   """
 327 
 328   @deprecation.deprecated(
 329       None, "Queue-based input pipelines have been replaced by `tf.data`. Use "
 330       "`tf.data.Dataset.map(tf.read_file)`.")
 331   def __init__(self, name=None):
 332     """Create a WholeFileReader.
 333 
 334     Args:
 335       name: A name for the operation (optional).
 336     """
 337     rr = gen_io_ops.whole_file_reader_v2(name=name)
 338     super(WholeFileReader, self).__init__(rr, supports_serialize=True)
 339 
 340 
 341 ops.NotDifferentiable("WholeFileReader")
