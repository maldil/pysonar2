  42 def _GetGradSource(op_or_tensor):
  43   """Identify which call to tf.gradients created this gradient op or tensor.
  44 
  45   TensorArray gradient calls use an accumulator TensorArray object.  If
  46   multiple gradients are calculated and run in the same session, the multiple
  47   gradient nodes may accidentally flow throuth the same accumulator TensorArray.
  48   This double counting breaks the TensorArray gradient flow.
  49 
  50   The solution is to identify which gradient call this particular
  51   TensorArray*Grad is being called in, by looking at the input gradient
  52   tensor's name, and create or lookup an accumulator gradient TensorArray
  53   associated with this specific call.  This solves any confusion and ensures
  54   different gradients from the same forward graph get their own accumulators.
  55 
  56   This function creates the unique label associated with the tf.gradients call
  57   that is used to create the gradient TensorArray.
  58 
  59   Args:
  60     op_or_tensor: `Tensor` or `Operation` which is an input to a
  61       TensorArray*Grad call.
  62 
  63   Returns:
  64     A python string, the unique label associated with this particular
  65     gradients calculation.
  66 
  67   Raises:
  68     ValueError: If not called within a gradients calculation.
  69   """
  70   name_tokens = op_or_tensor.name.split("/")
  71   grad_pos = [i for i, x in enumerate(name_tokens) if x.startswith("gradients")]
  72   if not grad_pos:
  73     raise ValueError(
  74         "Expected op/tensor name to start with gradients (excluding scope)"
  75         ", got: %s" % op_or_tensor.name)
  76   return "/".join(name_tokens[:grad_pos[-1] + 1])

  79 @ops.RegisterGradient("TensorArrayRead")
  80 @ops.RegisterGradient("TensorArrayReadV2")
  81 @ops.RegisterGradient("TensorArrayReadV3")
  82 def _TensorArrayReadGrad(op, grad):
  83   """Gradient for TensorArrayRead.
  84 
  85   Args:
  86     op: Forward TensorArrayRead op.
  87     grad: Gradient `Tensor` to TensorArrayRead.
  88 
  89   Returns:
  90     A flow `Tensor`, which can be used in control dependencies to
  91     force the write of `grad` to the gradient `TensorArray`.
  92   """
  93   # Note: the forward flow dependency in the call to grad() is necessary for
  94   # the case of dynamic sized TensorArrays.  When creating the gradient
  95   # TensorArray, the final size of the forward array must be known.
  96   # For this we need to wait until it has been created by depending on
  97   # the input flow of the original op.
  98   handle = op.inputs[0]
  99   index = op.inputs[1]
 100   flow = op.inputs[2]
 101   dtype = op.get_attr("dtype")
 102   grad_source = _GetGradSource(grad)
 103   g = (tensor_array_ops.TensorArray(dtype=dtype, handle=handle, flow=flow,
 104                                     colocate_with_first_write_call=False)
 105        .grad(source=grad_source, flow=flow))
 106   w_g = g.write(index, grad)
 107   return [None, None, w_g.flow]

 110 @ops.RegisterGradient("TensorArrayWrite")
 111 @ops.RegisterGradient("TensorArrayWriteV2")
 112 @ops.RegisterGradient("TensorArrayWriteV3")
 113 def _TensorArrayWriteGrad(op, flow):
 114   """Gradient for TensorArrayWrite.
 115 
 116   Args:
 117     op: Forward TensorArrayWrite op.
 118     flow: Gradient `Tensor` flow to TensorArrayWrite.
 119 
 120   Returns:
 121     A grad `Tensor`, the gradient created in an upstream ReadGrad or PackGrad.
 122   """
 123   # handle is the output store_handle of TensorArrayReadGrad or
 124   # the handle output of TensorArrayWriteGrad.  we must use this one.
 125   handle = op.inputs[0]
 126   index = op.inputs[1]
 127   dtype = op.get_attr("T")
 128   grad_source = _GetGradSource(flow)
 129   g = (tensor_array_ops.TensorArray(dtype=dtype, handle=handle, flow=flow,
 130                                     colocate_with_first_write_call=False)
 131        .grad(source=grad_source, flow=flow))
 132   grad = g.read(index)
 133   return [None, None, grad, flow]

 191 @ops.RegisterGradient("TensorArrayConcat")
 192 @ops.RegisterGradient("TensorArrayConcatV2")
 193 @ops.RegisterGradient("TensorArrayConcatV3")
 194 def _TensorArrayConcatGrad(op, grad, unused_lengths_grad):
 195   """Gradient for TensorArrayConcat.
 196 
 197   Args:
 198     op: Forward TensorArrayConcat op.
 199     grad: Gradient `Tensor` to TensorArrayConcat.
 200 
 201   Returns:
 202     A flow `Tensor`, which can be used in control dependencies to
 203     force the write of `grad` to the gradient `TensorArray`.
 204   """
 205   # Note: the forward flow dependency in the call to grad() is necessary for
 206   # the case of dynamic sized TensorArrays.  When creating the gradient
 207   # TensorArray, the final size of the forward array must be known.
 208   # For this we need to wait until it has been created by depending on
 209   # the input flow of the original op.
 210   handle = op.inputs[0]
 211   flow = op.inputs[1]
 212   lengths = op.outputs[1]
 213   dtype = op.get_attr("dtype")
 214   grad_source = _GetGradSource(grad)
 215   g = (tensor_array_ops.TensorArray(dtype=dtype, handle=handle, flow=flow,
 216                                     colocate_with_first_write_call=False)
 217        .grad(source=grad_source, flow=flow))
 218   u_g = g.split(grad, lengths=lengths)
 219   # handle, flow_in
 220   return [None, u_g.flow]

 223 @ops.RegisterGradient("TensorArraySplit")
 224 @ops.RegisterGradient("TensorArraySplitV2")
 225 @ops.RegisterGradient("TensorArraySplitV3")
 226 def _TensorArraySplitGrad(op, flow):
 227   """Gradient for TensorArraySplit.
 228 
 229   Args:
 230     op: Forward TensorArraySplit op.
 231     flow: Gradient `Tensor` flow to TensorArraySplit.
 232 
 233   Returns:
 234     A grad `Tensor`, the gradient created in upstream ReadGrads or PackGrad.
 235   """
 236   handle = op.inputs[0]
 237   dtype = op.get_attr("T")
 238   grad_source = _GetGradSource(flow)
 239   g = (tensor_array_ops.TensorArray(dtype=dtype, handle=handle, flow=flow,
 240                                     colocate_with_first_write_call=False)
 241        .grad(source=grad_source, flow=flow))
 242   grad = g.concat()
 243   # handle, value, lengths, flow_in
 244   return [None, grad, None, flow]
