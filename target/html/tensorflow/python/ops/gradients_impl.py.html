  73 def _IndexedSlicesToTensor(value, dtype=None, name=None, as_ref=False):
  74   """Converts an IndexedSlices object `value` to a Tensor.
  75 
  76   NOTE(mrry): This function is potentially expensive.
  77 
  78   Args:
  79     value: An ops.IndexedSlices object.
  80     dtype: The dtype of the Tensor to be returned.
  81     name: Optional name to use for the returned Tensor.
  82     as_ref: True if a ref is requested.
  83 
  84   Returns:
  85     A dense Tensor representing the values in the given IndexedSlices.
  86 
  87   Raises:
  88     ValueError: If the IndexedSlices does not have the same dtype.
  89   """
  90   _ = as_ref
  91   if dtype and not dtype.is_compatible_with(value.dtype):
  92     raise ValueError(
  93         "Tensor conversion requested dtype %s for IndexedSlices with dtype %s" %
  94         (dtype.name, value.dtype.name))
  95   if value.dense_shape is None:
  96     raise ValueError(
  97         "Tensor conversion requested for IndexedSlices without dense_shape: %s"
  98         % str(value))
  99   # TODO(mrry): Consider adding static shape information to
 100   # IndexedSlices, to avoid using numpy here.
 101   if not context.executing_eagerly():
 102     dense_shape_value = tensor_util.constant_value(value.dense_shape)
 103     if dense_shape_value is not None:
 104       num_elements = np.prod(dense_shape_value)
 105       if num_elements >= _LARGE_SPARSE_NUM_ELEMENTS:
 106         warnings.warn(
 107             "Converting sparse IndexedSlices to a dense Tensor with %d "
 108             "elements. This may consume a large amount of memory." %
 109             num_elements)
 110     else:
 111       warnings.warn(
 112           "Converting sparse IndexedSlices to a dense Tensor of unknown shape. "
 113           "This may consume a large amount of memory.")
 114   return math_ops.unsorted_segment_sum(
 115       value.values, value.indices, value.dense_shape[0], name=name)

 122 def _MarkReachedOps(from_ops, reached_ops, func_graphs):
 123   """Mark all ops reached from "from_ops".
 124 
 125   Args:
 126     from_ops: list of Operations.
 127     reached_ops: set of Operations.
 128     func_graphs: list of _function.FuncGraphs. This method will traverse through
 129       these functions if they capture from_ops or any reachable ops.
 130   """
 131   queue = collections.deque()
 132   queue.extend(from_ops)
 133   while queue:
 134     op = queue.popleft()
 135     if op not in reached_ops:
 136       reached_ops.add(op)
 137       for output in op.outputs:
 138         if _IsBackpropagatable(output):
 139           queue.extend(_Consumers(output, func_graphs))

 142 def _PendingCount(to_ops, from_ops, colocate_gradients_with_ops, func_graphs,
 143                   xs):
 144   """Initialize the pending count for ops between two lists of Operations.
 145 
 146   'pending_count[op]' indicates the number of backprop inputs
 147   to this operation.
 148 
 149   Args:
 150     to_ops: list of Operations.
 151     from_ops: list of Operations.
 152     colocate_gradients_with_ops: Python bool.  See docstring of gradients().
 153     func_graphs: list of _function.FuncGraphs. This method will traverse through
 154       these functions if they capture from_ops or any reachable ops. This is
 155       useful if to_ops occur in a function and from_ops are in an outer function
 156       or graph.
 157     xs: list of Tensors.
 158 
 159   Returns:
 160     A tuple containing: (1) the subset of to_ops reachable from from_ops by a
 161     path of zero or more backpropagatable tensors, (2) a mapping from operation
 162     to the number of backprop inputs to that op, and (3) a ControlFlowState
 163     object which is not None if the ops between from_ops and to_ops contain
 164     control flow loops.
 165   """
 166   # Mark reachable ops from from_ops.
 167   reached_ops = set()
 168   _MarkReachedOps(from_ops, reached_ops, func_graphs)
 169   # X in reached_ops iff X is reachable from from_ops by a path of zero or more
 170   # backpropagatable tensors.
 171 
 172   reachable_to_ops = set(op for op in to_ops if op in reached_ops)
 173 
 174   # Mark between ops.
 175   between_ops = set()
 176   between_op_list = []
 177   queue = collections.deque()
 178   queue.extend(to_ops)
 179   while queue:
 180     op = queue.popleft()
 181     # We are interested in this op.
 182     if op in reached_ops:
 183       between_ops.add(op)
 184       between_op_list.append(op)
 185       # Clear the boolean so we won't add the inputs again.
 186       reached_ops.remove(op)
 187       for inp in _NonEagerInputs(op, xs):
 188         queue.append(inp.op)
 189   # X in between_ops iff X is on a path of zero or more backpropagatable tensors
 190   # between from_ops and to_ops
 191 
 192   # 'loop_state' is None if there are no while loops.
 193   loop_state = control_flow_ops.MaybeCreateControlFlowState(
 194       between_op_list, between_ops, colocate_gradients_with_ops)
 195 
 196   # Initialize pending count for between ops.
 197   pending_count = collections.defaultdict(int)
 198   for op in between_op_list:
 199     for x in _NonEagerInputs(op, xs):
 200       if x.op in between_ops:
 201         pending_count[x.op] += 1
 202 
 203   return reachable_to_ops, pending_count, loop_state

 210 def _DefaultGradYs(grad_ys,
 211                    ys,
 212                    colocate_gradients_with_ops,
 213                    gradient_uid="__unsupported__"):
 214   """Fill in default values for grad_ys.
 215 
 216   Args:
 217     grad_ys: List of gradients, can contain None.
 218     ys: List of tensors.
 219     colocate_gradients_with_ops: If True, try colocating gradients with
 220       the corresponding op.
 221     gradient_uid: A unique identifier within the graph indicating
 222       which invocation of gradients is being executed. Used to cluster
 223       ops for compilation.
 224 
 225   Returns:
 226     A list of gradients to use, without None.
 227 
 228   Raises:
 229     ValueError: If sizes of gradients and inputs don't match
 230     TypeError: If type of any gradient is not valid for its input.
 231   """
 232   if len(grad_ys) != len(ys):
 233     raise ValueError("Passed %d grad_ys for %d ys" % (len(grad_ys), len(ys)))
 234   grad_ys = ops.convert_n_to_tensor_or_indexed_slices(grad_ys, name="grad_y")
 235   new_grad_ys = []
 236   for i in xrange(len(grad_ys)):
 237     grad_y = grad_ys[i]
 238     y = ys[i]
 239     with _maybe_colocate_with(y.op, gradient_uid, colocate_gradients_with_ops):
 240       if grad_y is None:
 241         if y.dtype.is_complex:
 242           raise TypeError(
 243               "Gradients of complex tensors must set grad_ys (y.dtype = %r)" %
 244               y.dtype)
 245         new_grad_ys.append(
 246             array_ops.fill(
 247                 array_ops.shape(y),
 248                 constant_op.constant(1, dtype=y.dtype, name="grad_ys_%d" % i)))
 249         continue
 250       if y.dtype.is_floating or y.dtype.is_integer:
 251         if not grad_y.dtype.is_floating and not grad_y.dtype.is_integer:
 252           raise TypeError(
 253               "Gradient type %s generated for real or "
 254               "integer-valued tensor %s with type %s must be "
 255               "real or integer" % (dtypes.as_dtype(grad_y.dtype).name, y,
 256                                    dtypes.as_dtype(y.dtype).name))
 257       elif y.dtype.is_complex:
 258         if not grad_y.dtype.is_complex:
 259           raise TypeError(
 260               "Gradient type %s generated for complex-valued "
 261               "tensor %s with type %s must be real" % (dtypes.as_dtype(
 262                   grad_y.dtype).name, y, dtypes.as_dtype(y.dtype).name))
 263       elif y.dtype == dtypes.variant:
 264         if grad_y.dtype != dtypes.variant:
 265           raise TypeError(
 266               "Gradient type %s generated for variant "
 267               "tensor %s with type %s must be variant" % (dtypes.as_dtype(
 268                   grad_y.dtype).name, y, dtypes.as_dtype(y.dtype).name))
 269       else:
 270         raise TypeError(
 271             "Tensor %s with type %s must be numeric "
 272             "to obtain a default gradient" % (y, dtypes.as_dtype(y.dtype).name))
 273       # Create a grad_y tensor in the name scope of the gradient.
 274       # Required for TensorArrays to identify which gradient call a
 275       # grad_y value is coming from.
 276       if isinstance(grad_y, ops.IndexedSlices):
 277         new_grad_ys.append(
 278             ops.IndexedSlices(
 279                 indices=(array_ops.identity(
 280                     grad_y.indices, name="grad_ys_%d_indices" % i)
 281                          if isinstance(grad_y.indices, ops.Tensor) else
 282                          grad_y.indices),
 283                 values=(array_ops.identity(
 284                     grad_y.values, name="grad_ys_%d_values" % i) if isinstance(
 285                         grad_y.values, ops.Tensor) else grad_y.values),
 286                 dense_shape=(array_ops.identity(
 287                     grad_y.dense_shape, name="grad_ys_%d_shape" % i)
 288                              if isinstance(grad_y.dense_shape, ops.Tensor) else
 289                              grad_y.dense_shape)))
 290       else:
 291         new_grad_ys.append(array_ops.identity(grad_y, name="grad_ys_%d" % i))
 292 
 293   return new_grad_ys

 311   """Verify that gradients are valid in number and type.
 312 
 313   Args:
 314     grads: List of generated gradients.
 315     op: Operation for which the gradients where generated.

 326 def _StopOps(from_ops, stop_gradient_ops, pending_count, xs):
 327   """The set of ops that terminate the gradient computation.
 328 
 329   This computes the frontier of the forward graph *before* which backprop
 330   should stop. Operations in the returned set will not be differentiated.
 331   This set is defined as the subset of `from_ops` containing ops that have
 332   no predecessor in `from_ops`. `pending_count` is the result of
 333   `_PendingCount(xs, from_ops)`. An 'op' has predecessors in `from_ops`
 334   iff pending_count[op] > 0.
 335 
 336   In addition, none of `stop_gradient_ops` will be differentiated.
 337 
 338   Args:
 339     from_ops: list of Operations.
 340     stop_gradient_ops: list of Operations never to backprop through.
 341     pending_count: mapping from operation to number of backprop inputs.
 342     xs: list of Tensors.
 343 
 344   Returns:
 345     The set of operations.
 346   """
 347   stop_ops = set()
 348   for op in from_ops:
 349     is_stop_op = True
 350     for inp in _NonEagerInputs(op, xs):
 351       if pending_count[inp.op] > 0:
 352         is_stop_op = False
 353         break
 354     if is_stop_op:
 355       stop_ops.add(op)
 356   stop_ops.update(op for op in stop_gradient_ops)
 357   return stop_ops

 374 def _SymGrad(op, out_grads):
 375   """Backprop through a function call node op given its outputs' gradients."""
 376   f_in = [x for x in op.inputs] + out_grads
 377   f_types = [x.dtype for x in op.inputs]
 378   f = attr_value_pb2.NameAttrList()
 379   if _IsPartitionedCall(op):
 380     f.name = op.get_attr("f").name
 381   else:
 382     f.name = op.type
 383   for k in op.node_def.attr:
 384     f.attr[k].CopyFrom(op.node_def.attr[k])
 385   # TODO(apassos) use a better dtype here
 386   in_grads = functional_ops.symbolic_gradient(
 387       input=f_in,
 388       Tout=[x if x != dtypes.resource else dtypes.float32 for x in f_types],
 389       f=f)
 390   return in_grads

 393 def _MaybeCompile(scope, op, func, grad_fn):
 394   """Compile the calculation in grad_fn if op was marked as compiled."""
 395   scope = scope.rstrip("/").replace("/", "_")
 396   if func is not None:
 397     xla_compile = func.definition.attr["_XlaCompile"].b
 398     xla_separate_compiled_gradients = func.definition.attr[
 399         "_XlaSeparateCompiledGradients"].b
 400     xla_scope = func.definition.attr["_XlaScope"].s.decode()
 401   else:
 402     try:
 403       xla_compile = op.get_attr("_XlaCompile")
 404       xla_separate_compiled_gradients = op.get_attr(
 405           "_XlaSeparateCompiledGradients")
 406       xla_scope = op.get_attr("_XlaScope").decode()
 407     except ValueError:
 408       return grad_fn()  # Exit early
 409 
 410   if not xla_compile:
 411     return grad_fn()  # Exit early
 412 
 413   # If the gradients are supposed to be compiled separately, we give them a
 414   # _XlaScope name that is based on the name_scope of the gradients.  Otherwise
 415   # they just inherit the existing _XlaScope name, which lets them be merged
 416   # together with the non-gradient computation.
 417   if xla_separate_compiled_gradients:
 418     xla_grad_scope = "%s_grad_%s" % (xla_scope, scope)
 419   else:
 420     xla_grad_scope = xla_scope
 421 
 422   attrs = {
 423       "_XlaCompile": attr_value_pb2.AttrValue(b=xla_compile),
 424       "_XlaScope": attr_value_pb2.AttrValue(s=xla_grad_scope.encode())
 425   }
 426   with ops.get_default_graph()._attr_scope(attrs):  # pylint: disable=protected-access
 427     return grad_fn()

 430 def _RaiseNoGradWrtInitialLoopValError(op, from_ops, xs):
 431   """Raises an error if we backprop through a loop var."""
 432   # Find the nearest 'to_op' reachable from 'op' to provide a more helpful error
 433   # message.
 434   target_op = None
 435   queue = collections.deque([op])
 436   visited = set()
 437   while queue:
 438     curr_op = queue.popleft()
 439     if curr_op in visited: continue
 440     visited.add(curr_op)
 441     if curr_op in from_ops:
 442       target_op = curr_op
 443       break
 444     queue.extend(t.op for t in _NonEagerInputs(curr_op, xs))
 445   assert target_op
 446   raise ValueError(
 447       "Cannot compute gradient inside while loop with respect to op '%s'. "
 448       "We do not support taking the gradient wrt or through the initial value "
 449       "of a loop variable. Gradients can be computed through loop invariants "
 450       "or wrt the input parameters to the loop body."
 451       % target_op.name)

 467 def _MaybeCaptured(t):
 468   """If t is a captured value placeholder, returns the original captured value.
 469 
 470   Args:
 471     t: Tensor
 472 
 473   Returns:
 474     A tensor, potentially from a different Graph/_function.FuncGraph.
 475   """
 476   # pylint: disable=protected-access
 477   if (not isinstance(t, ops.EagerTensor) and
 478       _IsFunction(t.op.graph) and t.op.type == "Placeholder"):
 479     for input_t, placeholder_t in _Captures(t.op.graph).items():
 480       if t == placeholder_t:
 481         return _MaybeCaptured(input_t)
 482   # pylint: enable=protected-access
 483   return t

 486 # TODO(skyewm): plumbing xs through everywhere is ugly, consider making
 487 # _GradientsHelper a class with xs as a member variable.
 488 def _NonEagerInputs(op, xs):
 489   """Returns the inputs of op, crossing closure boundaries where necessary.
 490 
 491   Does not return any captured EagerTensors, i.e., the number of tensors
 492   returned may be less than than the actual number of inputs.
 493 
 494   Args:
 495     op: Operation
 496     xs: list of Tensors we are differentiating w.r.t.
 497 
 498   Returns:
 499     A list of tensors. The tensors may be from multiple
 500     Graph/_function.FuncGraphs if op is in a _function.FuncGraph and has
 501     captured inputs.
 502   """
 503   if _IsFunction(op.graph):  # pylint: disable=protected-access
 504     inputs = []
 505     for t in op.inputs:
 506       # If we're differentiating w.r.t. `t`, do not attempt to traverse through
 507       # it to a captured value. The algorithm needs to "see" `t` in this case,
 508       # even if it's a function input for a captured value, whereas usually we'd
 509       # like to traverse through these closures as if the captured value was the
 510       # direct input to op.
 511       if t not in xs:
 512         t = _MaybeCaptured(t)
 513         # Skip captured eager inputs.
 514         if isinstance(t, ops.EagerTensor): continue
 515       inputs.append(t)
 516     return inputs
 517   else:
 518     return op.inputs

 521 def _Consumers(t, func_graphs):
 522   """Returns the consumers of t, crossing closure boundaries where necessary.
 523 
 524   Args:
 525     t: Tensor
 526     func_graphs: a list of _function.FuncGraphs that may have captured t.
 527 
 528   Returns:
 529     A list of tensors. The tensors will be from the current graph and/or
 530     func_graphs.
 531   """
 532   consumers = t.consumers()
 533   for func in func_graphs:
 534     for input_t, placeholder in _Captures(func).items():
 535       if input_t == t:
 536         consumers.extend(_Consumers(placeholder, func_graphs))
 537   return consumers

 540 @tf_export("gradients")
 541 def gradients(ys,
 542               xs,
 543               grad_ys=None,
 544               name="gradients",
 545               colocate_gradients_with_ops=False,
 546               gate_gradients=False,
 547               aggregation_method=None,
 548               stop_gradients=None):
 549   """Constructs symbolic derivatives of sum of `ys` w.r.t. x in `xs`.
 550 
 551   `ys` and `xs` are each a `Tensor` or a list of tensors.  `grad_ys`
 552   is a list of `Tensor`, holding the gradients received by the
 553   `ys`. The list must be the same length as `ys`.
 554 
 555   `gradients()` adds ops to the graph to output the derivatives of `ys` with
 556   respect to `xs`.  It returns a list of `Tensor` of length `len(xs)` where
 557   each tensor is the `sum(dy/dx)` for y in `ys`.
 558 
 559   `grad_ys` is a list of tensors of the same length as `ys` that holds
 560   the initial gradients for each y in `ys`.  When `grad_ys` is None,
 561   we fill in a tensor of '1's of the shape of y for each y in `ys`.  A
 562   user can provide their own initial `grad_ys` to compute the
 563   derivatives using a different initial gradient for each y (e.g., if
 564   one wanted to weight the gradient differently for each value in
 565   each y).
 566 
 567   `stop_gradients` is a `Tensor` or a list of tensors to be considered constant
 568   with respect to all `xs`. These tensors will not be backpropagated through,
 569   as though they had been explicitly disconnected using `stop_gradient`.  Among
 570   other things, this allows computation of partial derivatives as opposed to
 571   total derivatives. For example:
 572 
 573   ```python
 574   a = tf.constant(0.)
 575   b = 2 * a
 576   g = tf.gradients(a + b, [a, b], stop_gradients=[a, b])
 577   ```
 578 
 579   Here the partial derivatives `g` evaluate to `[1.0, 1.0]`, compared to the
 580   total derivatives `tf.gradients(a + b, [a, b])`, which take into account the
 581   influence of `a` on `b` and evaluate to `[3.0, 1.0]`.  Note that the above is
 582   equivalent to:
 583 
 584   ```python
 585   a = tf.stop_gradient(tf.constant(0.))
 586   b = tf.stop_gradient(2 * a)
 587   g = tf.gradients(a + b, [a, b])
 588   ```
 589 
 590   `stop_gradients` provides a way of stopping gradient after the graph has
 591   already been constructed, as compared to `tf.stop_gradient` which is used
 592   during graph construction.  When the two approaches are combined,
 593   backpropagation stops at both `tf.stop_gradient` nodes and nodes in
 594   `stop_gradients`, whichever is encountered first.
 595 
 596   All integer tensors are considered constant with respect to all `xs`, as if
 597   they were included in `stop_gradients`.
 598 
 599   Args:
 600     ys: A `Tensor` or list of tensors to be differentiated.
 601     xs: A `Tensor` or list of tensors to be used for differentiation.
 602     grad_ys: Optional. A `Tensor` or list of tensors the same size as
 603       `ys` and holding the gradients computed for each y in `ys`.
 604     name: Optional name to use for grouping all the gradient ops together.
 605       defaults to 'gradients'.
 606     colocate_gradients_with_ops: If True, try colocating gradients with
 607       the corresponding op.
 608     gate_gradients: If True, add a tuple around the gradients returned
 609       for an operations.  This avoids some race conditions.
 610     aggregation_method: Specifies the method used to combine gradient terms.
 611       Accepted values are constants defined in the class `AggregationMethod`.
 612     stop_gradients: Optional. A `Tensor` or list of tensors not to differentiate
 613       through.
 614 
 615   Returns:
 616     A list of `sum(dy/dx)` for each x in `xs`.
 617 
 618   Raises:
 619     LookupError: if one of the operations between `x` and `y` does not
 620       have a registered gradient function.
 621     ValueError: if the arguments are invalid.
 622     RuntimeError: if called in Eager mode.
 623 
 624   """
 625   # Creating the gradient graph for control flow mutates Operations.
 626   # _mutation_lock ensures a Session.run call cannot occur between creating and
 627   # mutating new ops.
 628   with ops.get_default_graph()._mutation_lock():  # pylint: disable=protected-access
 629     return _GradientsHelper(ys, xs, grad_ys, name, colocate_gradients_with_ops,
 630                             gate_gradients, aggregation_method, stop_gradients)
 631 
 632 
 633 def _GradientsHelper(ys,
 634                      xs,
 635                      grad_ys=None,
 636                      name="gradients",
 637                      colocate_gradients_with_ops=False,
 638                      gate_gradients=False,
 639                      aggregation_method=None,
 640                      stop_gradients=None,
 641                      src_graph=None):
 642   """Implementation of gradients()."""
 643   if context.executing_eagerly():
 644     raise RuntimeError("tf.gradients is not supported when eager execution "
 645                        "is enabled. Use tf.GradientTape instead.")
 646   if src_graph is None:
 647     src_graph = ops.get_default_graph()
 648 
 649   # If src_graph is a _FuncGraph (i.e. a function body), gather it and all
 650   # ancestor graphs. This is necessary for correctly handling captured values.
 651   func_graphs = []
 652   curr_graph = src_graph
 653   while _IsFunction(curr_graph):
 654     func_graphs.append(curr_graph)
 655     if isinstance(curr_graph, _function.FuncGraph):
 656       curr_graph = curr_graph.outer_graph
 657     else:
 658       assert isinstance(curr_graph, framework_function._FuncGraph)  # pylint: disable=protected-access
 659       curr_graph = curr_graph._outer_graph  # pylint: disable=protected-access
 660 
 661   ys = _AsList(ys)
 662   xs = _AsList(xs)
 663   stop_gradients = [] if stop_gradients is None else _AsList(stop_gradients)
 664   if grad_ys is None:
 665     grad_ys = [None] * len(ys)
 666   else:
 667     grad_ys = _AsList(grad_ys)
 668 
 669   with ops.name_scope(
 670       name, "gradients",
 671       list(ys) + list(xs) + list(stop_gradients) + list(grad_ys)) as grad_scope:
 672     # Get a uid for this call to gradients that can be used to help
 673     # cluster ops for compilation.
 674     gradient_uid = ops.get_default_graph().unique_name("uid")
 675     ys = ops.convert_n_to_tensor_or_indexed_slices(ys, name="y")
 676     xs = [
 677         x.handle if resource_variable_ops.is_resource_variable(x) else x
 678         for x in xs
 679     ]
 680     xs = ops.internal_convert_n_to_tensor_or_indexed_slices(
 681         xs, name="x", as_ref=True)
 682     grad_ys = _DefaultGradYs(grad_ys, ys, colocate_gradients_with_ops,
 683                              gradient_uid)
 684 
 685     # The approach we take here is as follows: Create a list of all ops in the
 686     # subgraph between the ys and xs.  Visit these ops in reverse order of ids
 687     # to ensure that when we visit an op the gradients w.r.t its outputs have
 688     # been collected.  Then aggregate these gradients if needed, call the op's
 689     # gradient function, and add the generated gradients to the gradients for
 690     # its input.
 691 
 692     # Initialize the pending count for ops in the connected subgraph from ys
 693     # to the xs.
 694     to_ops = [t.op for t in ys]
 695     from_ops = [t.op for t in xs]
 696     stop_gradient_ops = [t.op for t in stop_gradients]
 697     reachable_to_ops, pending_count, loop_state = _PendingCount(
 698         to_ops, from_ops, colocate_gradients_with_ops, func_graphs, xs)
 699 
 700     # Iterate over the collected ops.
 701     #
 702     # grads: op => list of gradients received on each output endpoint of the
 703     # op.  The gradients for each endpoint are initially collected as a list.
 704     # When it is time to call the op's gradient function, for each endpoint we
 705     # aggregate the list of received gradients into a Add() Operation if there
 706     # is more than one.
 707     grads = {}
 708 
 709     # Add the initial gradients for the ys.
 710     for y, grad_y in zip(ys, grad_ys):
 711       _SetGrad(grads, y, grad_y)
 712 
 713     # Initialize queue with to_ops.
 714     queue = collections.deque()
 715     # Add the ops in 'to_ops' into the queue.
 716     to_ops_set = set()
 717     for op in to_ops:
 718       # 'ready' handles the case where one output gradient relies on
 719       # another output's gradient.
 720       ready = (pending_count[op] == 0)
 721       if ready and op not in to_ops_set and op in reachable_to_ops:
 722         to_ops_set.add(op)
 723         queue.append(op)
 724 
 725     if loop_state:
 726       loop_exits = loop_state.ProcessUnusedLoopExits(pending_count, to_ops_set)
 727       for y in loop_exits:
 728         if _IsTrainable(y):
 729           _SetGrad(grads, y, loop_state.ZerosLikeForExit(y))
 730           queue.append(y.op)
 731 
 732     stop_ops = _StopOps(from_ops, stop_gradient_ops, pending_count, xs)
 733     while queue:
 734       # generate gradient subgraph for op.
 735       op = queue.popleft()
 736       with _maybe_colocate_with(op, gradient_uid, colocate_gradients_with_ops):
 737         if loop_state:
 738           loop_state.EnterGradWhileContext(op, before=True)
 739         out_grads = _AggregatedGrads(grads, op, gradient_uid, loop_state,
 740                                      aggregation_method)
 741         if loop_state:
 742           loop_state.ExitGradWhileContext(op, before=True)
 743 
 744         grad_fn = None
 745         func_call = None
 746         is_partitioned_call = _IsPartitionedCall(op)
 747         # pylint: disable=protected-access
 748         is_func_call = (
 749             src_graph._is_function(op.type) or is_partitioned_call)
 750         # pylint: enable=protected-access
 751         has_out_grads = any(isinstance(g, ops.Tensor) or g for g in out_grads)
 752         if has_out_grads and (op not in stop_ops):
 753           if is_func_call:
 754             if is_partitioned_call:
 755               func_call = src_graph._get_function(  # pylint: disable=protected-access
 756                   compat.as_bytes(op.get_attr("f").name))
 757             else:
 758               func_call = src_graph._get_function(op.type)  # pylint: disable=protected-access
 759             # Note that __defun is not set if the graph is
 760             # imported. If it's set, we prefer to access the original
 761             # defun.
 762             func_call = getattr(op, "__defun", func_call)
 763             grad_fn = func_call.python_grad_func
 764           else:
 765             # A grad_fn must be defined, either as a function or as None
 766             # for ops that do not have gradients.
 767             try:
 768               grad_fn = ops.get_gradient_function(op)
 769             except LookupError:
 770               raise LookupError(
 771                   "No gradient defined for operation '%s' (op type: %s)" %
 772                   (op.name, op.type))
 773         if loop_state:
 774           loop_state.EnterGradWhileContext(op, before=False)
 775 
 776         # NOTE(skyewm): We don't support computing gradients wrt a loop variable
 777         # unless it's within the context of a single iteration (i.e. the
 778         # gradient is wrt to the loop parameter in the body function, not wrt or
 779         # through the initial value). This means if we're in a while loop
 780         # context, we should never see a switch node from this context.
 781         # pylint: disable=protected-access
 782         if (control_flow_util.IsSwitch(op) and
 783             op._control_flow_context is not None and
 784             op._control_flow_context.IsWhileContext() and
 785             op._control_flow_context ==
 786             ops.get_default_graph()._get_control_flow_context()):
 787           _RaiseNoGradWrtInitialLoopValError(op, from_ops, xs)
 788         # pylint: enable=protected-access
 789 
 790         if (grad_fn or is_func_call) and has_out_grads:
 791           # NOTE: If _AggregatedGrads didn't compute a value for the i'th
 792           # output, it means that the cost does not depend on output[i],
 793           # therefore dC/doutput[i] is 0.
 794           for i, out_grad in enumerate(out_grads):
 795             if (not isinstance(out_grad, ops.Tensor) and not out_grad) and (
 796                 (not grad_fn and is_func_call) or _IsTrainable(op.outputs[i])):
 797               # Only trainable outputs or outputs for a function call that
 798               # will use SymbolicGradient get a zero gradient. Gradient
 799               # functions should ignore the gradient for other outputs.
 800               # TODO(apassos) gradients of resource handles might be an
 801               # issue here because of zeros.
 802               if loop_state:
 803                 out_grads[i] = loop_state.ZerosLike(op, i)
 804               else:
 805                 out_grads[i] = control_flow_ops.ZerosLikeOutsideLoop(op, i)
 806           with ops.name_scope(op.name + "_grad"):
 807             # pylint: disable=protected-access
 808             with src_graph._original_op(op):
 809               # pylint: enable=protected-access
 810               if grad_fn:
 811                 # If grad_fn was found, do not use SymbolicGradient even for
 812                 # functions.
 813                 in_grads = _MaybeCompile(grad_scope, op, func_call,
 814                                          lambda: grad_fn(op, *out_grads))
 815               else:
 816                 # For function call ops, we add a 'SymbolicGradient'
 817                 # node to the graph to compute gradients.
 818                 in_grads = _MaybeCompile(grad_scope, op, func_call,
 819                                          lambda: _SymGrad(op, out_grads))
 820               in_grads = _AsList(in_grads)
 821               _VerifyGeneratedGradients(in_grads, op)
 822               if gate_gradients and len([x for x in in_grads
 823                                          if x is not None]) > 1:
 824                 with ops.device(None):
 825                   with ops._colocate_with_for_gradient(  # pylint: disable=protected-access
 826                       None,
 827                       gradient_uid,
 828                       ignore_existing=True):
 829                     in_grads = control_flow_ops.tuple(in_grads)
 830           _LogOpGradients(op, out_grads, in_grads)
 831         else:
 832           # If no grad_fn is defined or none of out_grads is available,
 833           # just propagate a list of None backwards.
 834           in_grads = [None] * len(_NonEagerInputs(op, xs))
 835         for i, (t_in, in_grad) in enumerate(zip(_NonEagerInputs(op, xs),
 836                                                 in_grads)):
 837           if in_grad is not None:
 838             if (isinstance(in_grad, ops.Tensor) and
 839                 t_in.dtype != dtypes.resource):
 840               try:
 841                 in_grad.set_shape(t_in.get_shape())
 842               except ValueError:
 843                 raise ValueError(
 844                     "Incompatible shapes between op input and calculated "
 845                     "input gradient.  Forward operation: %s.  Input index: %d. "
 846                     "Original input shape: %s.  "
 847                     "Calculated input gradient shape: %s" %
 848                     (op.name, i, t_in.shape, in_grad.shape))
 849             _SetGrad(grads, t_in, in_grad)
 850         if loop_state:
 851           loop_state.ExitGradWhileContext(op, before=False)
 852 
 853       # Update pending count for the inputs of op and enqueue ready ops.
 854       _UpdatePendingAndEnqueueReady(grads, op, queue, pending_count, loop_state,
 855                                     xs)
 856 
 857   if loop_state:
 858     loop_state.PostProcessing()
 859   return [_GetGrad(grads, x) for x in xs]

 862 def _HasAnyNotNoneGrads(grads, op):
 863   """Return true iff op has real gradient."""
 864   out_grads = _GetGrads(grads, op)
 865   for out_grad in out_grads:
 866     if isinstance(out_grad, (ops.Tensor, ops.IndexedSlices)):
 867       return True
 868     if out_grad and isinstance(out_grad, collections.Sequence):
 869       if any([g is not None for g in out_grad]):
 870         return True
 871   return False

 874 def _UpdatePendingAndEnqueueReady(grads, op, queue, pending_count, loop_state,
 875                                   xs):
 876   """Update pending count for the inputs of op and enqueue ready ops."""
 877   for x in _NonEagerInputs(op, xs):
 878     pending_count[x.op] -= 1
 879     ready = (pending_count[x.op] == 0)
 880     if loop_state and not ready:
 881       ready = pending_count[x.op] > 0 and control_flow_util.IsLoopSwitch(x.op)
 882     if ready:
 883       if control_flow_util.IsLoopExit(x.op):
 884         # if x is an exit without real gradient, defer processing them.
 885         grad_state = loop_state.GetGradState(x.op, before=False)
 886         grad_state.deferred_exits.append(x)
 887         grad_state.pending_exits_count -= 1
 888         if grad_state.pending_exits_count == 0:
 889           # We now have all the exits so process them.
 890           has_not_none_grad = False
 891           for y in grad_state.deferred_exits:
 892             if _HasAnyNotNoneGrads(grads, y.op):
 893               has_not_none_grad = True
 894               queue.append(y.op)
 895             else:
 896               grad_state.unused_exits.append(y)
 897           if has_not_none_grad:
 898             # For an unused exit, if it has trainable outputs, backprop
 899             # a zero gradient. Otherwise, just ignore it.
 900             for y in grad_state.unused_exits:
 901               if _IsTrainable(y):
 902                 _SetGrad(grads, y, loop_state.ZerosLikeForExit(y))
 903               queue.append(y.op)
 904           else:
 905             # All exits are "unused" so use None as gradient.
 906             for y in grad_state.unused_exits:
 907               queue.append(y.op)
 908       else:
 909         queue.append(x.op)

 912 def _SetGrad(grads, t, grad):
 913   """Sets gradient "grad" in "grads" for tensor "t"."""
 914   op = t.op
 915   op_grads = grads.get(op)
 916   if not op_grads:
 917     op_grads = [[] for _ in xrange(len(op.outputs))]
 918     grads[op] = op_grads
 919   t_grads = op_grads[t.value_index]
 920   if isinstance(t_grads, list):
 921     t_grads.append(grad)
 922   else:
 923     assert control_flow_util.IsLoopSwitch(op)
 924     op_grads[t.value_index] = grad

 927 def _GetGrad(grads, t):
 928   """Gets gradient for tensor "t"."""
 929   op = t.op
 930   op_grads = grads.get(op)
 931   if not op_grads:
 932     return None
 933   t_grad = op_grads[t.value_index]
 934   assert not isinstance(
 935       t_grad, list), ("gradients list should have been aggregated by now.")
 936   return t_grad

 940   """Gets all gradients for op."""

 942     return grads[op]
 943   else:

 984 def _MultiDeviceAddN(tensor_list, gradient_uid):
 985   """Adds tensors from potentially multiple devices."""
 986   # Basic function structure comes from control_flow_ops.group().
 987   # Sort tensors according to their devices.
 988   tensors_on_device = collections.defaultdict(lambda: [])
 989   for tensor in tensor_list:
 990     tensors_on_device[tensor.device].append(tensor)
 991 
 992   # For each device, add the tensors on that device first.
 993   # Then gather the partial sums from multiple devices.
 994   # TODO(sjhwang): Create hierarchical aggregation tree as pbar's suggestion.
 995   # E.g., aggregate per GPU, then per task, and so on.
 996   summands = []
 997 
 998   def DeviceKey(dev):
 999     return "" if dev is None else dev
1000 
1001   for dev in sorted(six.iterkeys(tensors_on_device), key=DeviceKey):
1002     tensors = tensors_on_device[dev]
1003     with ops._colocate_with_for_gradient(  # pylint: disable=protected-access
1004         tensors[0].op,
1005         gradient_uid,
1006         ignore_existing=True):
1007       summands.append(math_ops.add_n(tensors))
1008 
1009   return math_ops.add_n(summands)

1012 @tf_export("AggregationMethod")
1013 class AggregationMethod(object):
1014   """A class listing aggregation methods used to combine gradients.
1015 
1016   Computing partial derivatives can require aggregating gradient
1017   contributions. This class lists the various methods that can
1018   be used to combine gradients in the graph:
1019 
1020   *  `ADD_N`: All of the gradient terms are summed as part of one
1021      operation using the "AddN" op. It has the property that all
1022      gradients must be ready before any aggregation is performed.
1023   *  `DEFAULT`: The system-chosen default aggregation method.
1024   """
1025   ADD_N = 0
1026   DEFAULT = ADD_N
1027   # The following are experimental and may not be supported in future releases.
1028   EXPERIMENTAL_TREE = 1
1029   EXPERIMENTAL_ACCUMULATE_N = 2
1030 
1031 
1032 def _AggregatedGrads(grads,
1033                      op,
1034                      gradient_uid,
1035                      loop_state,
1036                      aggregation_method=None):
1037   """Get the aggregated gradients for op.
1038 
1039   Args:
1040     grads: The map of memoized gradients.
1041     op: The op to get gradients for.
1042     gradient_uid: A unique identifier within the graph indicating
1043       which invocation of gradients is being executed. Used to cluster
1044       ops for compilation.
1045     loop_state: An object for maintaining the state of the while loops in the
1046                 graph. It is of type ControlFlowState. None if the graph
1047                 contains no while loops.
1048     aggregation_method: Specifies the method used to combine gradient terms.
1049       Accepted values are constants defined in the class `AggregationMethod`.
1050 
1051   Returns:
1052     A list of gradients, one per each output of `op`. If the gradients
1053       for a particular output is a list, this function aggregates it
1054       before returning.
1055 
1056   Raises:
1057     TypeError: if the incoming grads are not Tensors or IndexedSlices.
1058     ValueError: if the arguments are invalid.
1059 
1060   """
1061   if aggregation_method is None:
1062     aggregation_method = AggregationMethod.DEFAULT
1063   if aggregation_method not in [
1064       AggregationMethod.ADD_N, AggregationMethod.EXPERIMENTAL_TREE,
1065       AggregationMethod.EXPERIMENTAL_ACCUMULATE_N
1066   ]:
1067     raise ValueError(
1068         "Invalid aggregation_method specified %s." % aggregation_method)
1069   out_grads = _GetGrads(grads, op)
1070   for i, out_grad in enumerate(out_grads):
1071     if loop_state:
1072       if isinstance(out_grad, (ops.Tensor, ops.IndexedSlices)):
1073         assert control_flow_util.IsLoopSwitch(op)
1074         continue
1075     # Grads have to be Tensors or IndexedSlices
1076     if (isinstance(out_grad, collections.Sequence) and not all([
1077         isinstance(g, (ops.Tensor, ops.IndexedSlices))
1078         for g in out_grad
1079         if g is not None
1080     ])):
1081       raise TypeError("gradients have to be either all Tensors "
1082                       "or all IndexedSlices")
1083     # Aggregate multiple gradients, and convert [] to None.
1084     if out_grad:
1085       if len(out_grad) < 2:
1086         used = "nop"
1087         out_grads[i] = out_grad[0]
1088       elif all([isinstance(g, ops.Tensor) for g in out_grad if g is not None]):
1089         tensor_shape = _AccumulatorShape(out_grad)
1090         if (aggregation_method == AggregationMethod.EXPERIMENTAL_ACCUMULATE_N
1091             and len(out_grad) > 2 and tensor_shape.is_fully_defined()):
1092           # The benefit of using AccumulateN is that its inputs can be combined
1093           # in any order and this can allow the expression to be evaluated with
1094           # a smaller memory footprint.  When used with gpu_allocator_retry,
1095           # it is possible to compute a sum of terms which are much larger than
1096           # total GPU memory.
1097           # AccumulateN can currently only be used if we know the shape for
1098           # an accumulator variable.  If this is not known, or if we only have
1099           # 2 grads then we fall through to the "tree" case below.
1100           used = "accumulate_n"
1101           out_grads[i] = math_ops.accumulate_n(out_grad)
1102         elif aggregation_method in [
1103             AggregationMethod.EXPERIMENTAL_TREE,
1104             AggregationMethod.EXPERIMENTAL_ACCUMULATE_N
1105         ]:
1106           # Aggregate all gradients by doing pairwise sums: this may
1107           # reduce performance, but it can improve memory because the
1108           # gradients can be released earlier.
1109           #
1110           # TODO(vrv): Consider replacing this with a version of
1111           # tf.AddN() that eagerly frees its inputs as soon as they are
1112           # ready, so the order of this tree does not become a problem.
1113           used = "tree"
1114           with ops.name_scope(op.name + "_gradient_sum"):
1115             running_sum = out_grad[0]
1116             for grad in out_grad[1:]:
1117               running_sum = math_ops.add_n([running_sum, grad])
1118             out_grads[i] = running_sum
1119         else:
1120           used = "add_n"
1121           out_grads[i] = _MultiDeviceAddN(out_grad, gradient_uid)
1122         logging.vlog(2, "  _AggregatedGrads %d x %s using %s", len(out_grad),
1123                      tensor_shape, used)
1124       else:
1125         out_grads[i] = _AggregateIndexedSlicesGradients(out_grad)
1126     else:  # not out_grad
1127       # out_grads[i] is [], thus its aggregation is simply None.
1128       out_grads[i] = None
1129   return out_grads

1132 def _AggregateIndexedSlicesGradients(grads):
1133   """Aggregates gradients of type `IndexedSlices` by concatenation."""
1134   if len(grads) < 1:
1135     return None
1136   elif len(grads) == 1:
1137     return grads[0]
1138   else:
1139     grads = math_ops._as_indexed_slices_list(  # pylint: disable=protected-access
1140         [g for g in grads if g is not None])
1141     grads = [_HandleNestedIndexedSlices(x) for x in grads]  # pylint: disable=protected-access
1142     # Form IndexedSlices out of the concatenated values and indices.
1143     concat_grad = ops.IndexedSlices(
1144         array_ops.concat([x.values for x in grads], axis=0),
1145         array_ops.concat([x.indices for x in grads], axis=0),
1146         grads[0].dense_shape)
1147 
1148     return concat_grad

1151 # TODO(vrv): Make this available when we want to make it public.
1152 def _hessian_vector_product(ys, xs, v):
1153   """Multiply the Hessian of `ys` wrt `xs` by `v`.
1154 
1155   This is an efficient construction that uses a backprop-like approach
1156   to compute the product between the Hessian and another vector. The
1157   Hessian is usually too large to be explicitly computed or even
1158   represented, but this method allows us to at least multiply by it
1159   for the same big-O cost as backprop.
1160 
1161   Implicit Hessian-vector products are the main practical, scalable way
1162   of using second derivatives with neural networks. They allow us to
1163   do things like construct Krylov subspaces and approximate conjugate
1164   gradient descent.
1165 
1166   Example: if `y` = 1/2 `x`^T A `x`, then `hessian_vector_product(y,
1167   x, v)` will return an expression that evaluates to the same values
1168   as (A + A.T) `v`.
1169 
1170   Args:
1171     ys: A scalar value, or a tensor or list of tensors to be summed to
1172         yield a scalar.
1173     xs: A list of tensors that we should construct the Hessian over.
1174     v: A list of tensors, with the same shapes as xs, that we want to
1175        multiply by the Hessian.
1176 
1177   Returns:
1178     A list of tensors (or if the list would be length 1, a single tensor)
1179     containing the product between the Hessian and `v`.
1180 
1181   Raises:
1182     ValueError: `xs` and `v` have different length.
1183 
1184   """
1185 
1186   # Validate the input
1187   length = len(xs)
1188   if len(v) != length:
1189     raise ValueError("xs and v must have the same length.")
1190 
1191   # First backprop
1192   grads = gradients(ys, xs)
1193 
1194   assert len(grads) == length
1195   elemwise_products = [
1196       math_ops.multiply(grad_elem, array_ops.stop_gradient(v_elem))
1197       for grad_elem, v_elem in zip(grads, v)
1198       if grad_elem is not None
1199   ]
1200 
1201   # Second backprop
1202   return gradients(elemwise_products, xs)
1203

1205 @tf_export("hessians")
1206 def hessians(ys,
1207              xs,
1208              name="hessians",
1209              colocate_gradients_with_ops=False,
1210              gate_gradients=False,
1211              aggregation_method=None):
1212   """Constructs the Hessian of sum of `ys` with respect to `x` in `xs`.
1213 
1214   `hessians()` adds ops to the graph to output the Hessian matrix of `ys`
1215   with respect to `xs`.  It returns a list of `Tensor` of length `len(xs)`
1216   where each tensor is the Hessian of `sum(ys)`.
1217 
1218   The Hessian is a matrix of second-order partial derivatives of a scalar
1219   tensor (see https://en.wikipedia.org/wiki/Hessian_matrix for more details).
1220 
1221   Args:
1222     ys: A `Tensor` or list of tensors to be differentiated.
1223     xs: A `Tensor` or list of tensors to be used for differentiation.
1224     name: Optional name to use for grouping all the gradient ops together.
1225       defaults to 'hessians'.
1226     colocate_gradients_with_ops: See `gradients()` documentation for details.
1227     gate_gradients: See `gradients()` documentation for details.
1228     aggregation_method: See `gradients()` documentation for details.
1229 
1230   Returns:
1231     A list of Hessian matrices of `sum(ys)` for each `x` in `xs`.
1232 
1233   Raises:
1234     LookupError: if one of the operations between `xs` and `ys` does not
1235       have a registered gradient function.
1236   """
1237   xs = _AsList(xs)
1238   kwargs = {
1239       "colocate_gradients_with_ops": colocate_gradients_with_ops,
1240       "gate_gradients": gate_gradients,
1241       "aggregation_method": aggregation_method
1242   }
1243   # Compute first-order derivatives and iterate for each x in xs.
1244   hessians = []
1245   _gradients = gradients(ys, xs, **kwargs)
1246   for gradient, x in zip(_gradients, xs):
1247     # change shape to one-dimension without graph branching
1248     gradient = array_ops.reshape(gradient, [-1])
1249 
1250     # Declare an iterator and tensor array loop variables for the gradients.
1251     n = array_ops.size(x)
1252     loop_vars = [
1253         array_ops.constant(0, dtypes.int32),
1254         tensor_array_ops.TensorArray(x.dtype, n)
1255     ]
1256     # Iterate over all elements of the gradient and compute second order
1257     # derivatives.
1258     _, hessian = control_flow_ops.while_loop(
1259         lambda j, _: j < n,
1260         lambda j, result: (j + 1,
1261                            result.write(j, gradients(gradient[j], x)[0])),
1262         loop_vars
1263     )
1264 
1265     _shape = array_ops.shape(x)
1266     _reshaped_hessian = array_ops.reshape(hessian.stack(),
1267                                           array_ops.concat((_shape, _shape), 0))
1268     hessians.append(_reshaped_hessian)
1269   return hessians
