  79 def _check_table_dtypes(table, key_dtype, value_dtype):
  80   """Check that the given key_dtype and value_dtype matches the table dtypes.
  81 
  82   Args:
  83     table: The table to check types against to.
  84     key_dtype: The key data type to check.
  85     value_dtype: The value data type to check.
  86 
  87   Raises:
  88     TypeError: when 'key_dtype' or 'value_dtype' doesn't match the table data
  89       types.
  90   """
  91   if key_dtype.base_dtype != table.key_dtype:
  92     raise TypeError("Invalid key dtype, expected %s but got %s." %
  93                     (table.key_dtype, key_dtype))
  94   if value_dtype.base_dtype != table.value_dtype:
  95     raise TypeError("Invalid value dtype, expected %s but got %s." %
  96                     (table.value_dtype, value_dtype))

  99 class LookupInterface(object):
 100   """Represent a lookup table that persists across different steps."""
 101 
 102   def __init__(self, key_dtype, value_dtype, name):
 103     """Construct a lookup table interface.
 104 
 105     Args:
 106       key_dtype: The table key type.
 107       value_dtype: The table value type.
 108       name: A name for the operation (optional).
 109     """
 110     self._key_dtype = dtypes.as_dtype(key_dtype)
 111     self._value_dtype = dtypes.as_dtype(value_dtype)
 112     self._name = name
 113 
 114   @property
 115   def key_dtype(self):
 116     """The table key dtype."""
 117     return self._key_dtype
 118 
 119   @property
 120   def value_dtype(self):
 121     """The table value dtype."""
 122     return self._value_dtype
 123 
 124   @property
 125   def name(self):
 126     """The name of the table."""
 127     return self._name
 128 
 129   @property
 130   def init(self):
 131     """The table initialization op."""
 132     raise NotImplementedError
 133 
 134   def size(self, name=None):
 135     """Compute the number of elements in this table."""
 136     raise NotImplementedError
 137 
 138   def lookup(self, keys, name=None):
 139     """Looks up `keys` in a table, outputs the corresponding values."""
 140     raise NotImplementedError

 143 class InitializableLookupTableBase(LookupInterface):
 144   """Initializable lookup table interface.
 145 
 146   An initializable lookup tables persist across different steps.
 147   """
 148 
 149   def __init__(self, table_ref, default_value, initializer):
 150     """Construct a table object from a table reference.
 151 
 152     If requires a table initializer object (subclass of `TableInitializerBase`).
 153     It provides the table key and value types, as well as the op to initialize
 154     the table. The caller is responsible to execute the initialization op.
 155 
 156     Args:
 157       table_ref: The table reference, i.e. the output of the lookup table ops.
 158       default_value: The value to use if a key is missing in the table.
 159       initializer: The table initializer to use.
 160     """
 161     if context.executing_eagerly():
 162       name = context.context().scope_name
 163     else:
 164       name = table_ref.op.name.split("/")[-1]
 165     super(InitializableLookupTableBase,
 166           self).__init__(initializer.key_dtype, initializer.value_dtype,
 167                          name)
 168     self._table_ref = table_ref
 169     self._default_value = ops.convert_to_tensor(
 170         default_value, dtype=self._value_dtype)
 171     self._default_value.get_shape().merge_with(tensor_shape.scalar())
 172     self._init = initializer.initialize(self)
 173 
 174   @property
 175   def table_ref(self):
 176     """Get the underlying table reference."""
 177     return self._table_ref
 178 
 179   @property
 180   def default_value(self):
 181     """The default value of the table."""
 182     return self._default_value
 183 
 184   @property
 185   def init(self):
 186     """The table initialization op."""
 187     return self._init
 188 
 189   def size(self, name=None):
 190     """Compute the number of elements in this table.
 191 
 192     Args:
 193       name: A name for the operation (optional).
 194 
 195     Returns:
 196       A scalar tensor containing the number of elements in this table.
 197     """
 198     with ops.name_scope(name, "%s_Size" % self._name,
 199                         [self._table_ref]) as scope:
 200       return gen_lookup_ops.lookup_table_size_v2(self._table_ref, name=scope)
 201 
 202   def lookup(self, keys, name=None):
 203     """Looks up `keys` in a table, outputs the corresponding values.
 204 
 205     The `default_value` is used for keys not present in the table.
 206 
 207     Args:
 208       keys: Keys to look up. May be either a `SparseTensor` or dense `Tensor`.
 209       name: A name for the operation (optional).
 210 
 211     Returns:
 212       A `SparseTensor` if keys are sparse, otherwise a dense `Tensor`.
 213 
 214     Raises:
 215       TypeError: when `keys` or `default_value` doesn't match the table data
 216         types.
 217     """
 218     key_tensor = keys
 219     if isinstance(keys, sparse_tensor.SparseTensor):
 220       key_tensor = keys.values
 221 
 222     if keys.dtype.base_dtype != self._key_dtype:
 223       raise TypeError("Signature mismatch. Keys must be dtype %s, got %s." %
 224                       (self._key_dtype, keys.dtype))
 225 
 226     with ops.name_scope(name, "%s_Lookup" % self._name,
 227                         (self._table_ref, key_tensor,
 228                          self._default_value)) as scope:
 229       values = gen_lookup_ops.lookup_table_find_v2(
 230           self._table_ref, key_tensor, self._default_value, name=scope)
 231 
 232     values.set_shape(key_tensor.get_shape())
 233     if isinstance(keys, sparse_tensor.SparseTensor):
 234       return sparse_tensor.SparseTensor(keys.indices, values, keys.dense_shape)
 235     else:
 236       return values

 239 class HashTable(InitializableLookupTableBase):
 240   """A generic hash table implementation.
 241 
 242   Example usage:
 243 
 244   ```python
 245   table = tf.HashTable(
 246       tf.KeyValueTensorInitializer(keys, values), -1)
 247   out = table.lookup(input_tensor)
 248   table.init.run()
 249   print(out.eval())
 250   ```
 251   """
 252 
 253   def __init__(self, initializer, default_value, shared_name=None, name=None):
 254     """Creates a non-initialized `HashTable` object.
 255 
 256     Creates a table, the type of its keys and values are specified by the
 257     initializer.
 258     Before using the table you will have to initialize it. After initialization
 259     the table will be immutable.
 260 
 261     Args:
 262       initializer: The table initializer to use. See `HashTable` kernel for
 263         supported key and value types.
 264       default_value: The value to use if a key is missing in the table.
 265       shared_name: If non-empty, this table will be shared under
 266         the given name across multiple sessions.
 267       name: A name for the operation (optional).
 268 
 269     Returns:
 270       A `HashTable` object.
 271     """
 272     with ops.name_scope(name, "hash_table", (initializer,
 273                                              default_value)) as scope:
 274       table_ref = gen_lookup_ops.hash_table_v2(
 275           shared_name=shared_name,
 276           key_dtype=initializer.key_dtype,
 277           value_dtype=initializer.value_dtype,
 278           name=scope)
 279 
 280       super(HashTable, self).__init__(table_ref, default_value, initializer)
 281       self._value_shape = self._default_value.get_shape()
 282 
 283   def export(self, name=None):
 284     """Returns tensors of all keys and values in the table.
 285 
 286     Args:
 287       name: A name for the operation (optional).
 288 
 289     Returns:
 290       A pair of tensors with the first tensor containing all keys and the
 291         second tensors containing all values in the table.
 292     """
 293     with ops.name_scope(name, "%s_Export" % self._name,
 294                         [self._table_ref]) as name:
 295       with ops.colocate_with(self._table_ref):
 296         exported_keys, exported_values = gen_lookup_ops.lookup_table_export_v2(
 297             self._table_ref, self._key_dtype, self._value_dtype, name=name)
 298 
 299     exported_values.set_shape(exported_keys.get_shape().concatenate(
 300         self._value_shape))
 301     return exported_keys, exported_values

 332 class KeyValueTensorInitializer(TableInitializerBase):
 333   """Table initializers given `keys` and `values` tensors."""
 334 
 335   def __init__(self, keys, values, key_dtype=None, value_dtype=None, name=None):
 336     """Constructs a table initializer object based on keys and values tensors.
 337 
 338     Args:
 339       keys: The tensor for the keys.
 340       values: The tensor for the values.
 341       key_dtype: The `keys` data type. Used when `keys` is a python array.
 342       value_dtype: The `values` data type. Used when `values` is a python array.
 343       name: A name for the operation (optional).
 344     """
 345     with ops.name_scope(name, "key_value_init", [keys, values]) as scope:
 346       self._keys = ops.convert_to_tensor(keys, dtype=key_dtype, name="keys")
 347       self._values = ops.convert_to_tensor(
 348           values, dtype=value_dtype, name="values")
 349       self._name = scope
 350 
 351     super(KeyValueTensorInitializer, self).__init__(self._keys.dtype,
 352                                                     self._values.dtype)
 353 
 354   def initialize(self, table):
 355     """Initializes the given `table` with `keys` and `values` tensors.
 356 
 357     Args:
 358       table: The table to initialize.
 359 
 360     Returns:
 361       The operation that initializes the table.
 362 
 363     Raises:
 364       TypeError: when the keys and values data types do not match the table
 365       key and value data types.
 366     """
 367     _check_table_dtypes(table, self._keys.dtype, self._values.dtype)
 368     with ops.name_scope(
 369         self._name, values=(table.table_ref, self._keys,
 370                             self._values)) as scope:
 371       if context.executing_eagerly():
 372         # Ensure a unique name when eager execution is enabled to avoid spurious
 373         # sharing issues.
 374         scope += str(ops.uid())
 375       if fwd_compat.forward_compatible(2018, 9, 19):
 376         init_op = gen_lookup_ops.lookup_table_import_v2(
 377             table.table_ref, self._keys, self._values, name=scope)
 378       else:
 379         # To maintain forward compatibiltiy, use the old implementation.
 380         init_op = gen_lookup_ops.initialize_table_v2(
 381             table.table_ref, self._keys, self._values, name=scope)
 382     ops.add_to_collection(ops.GraphKeys.TABLE_INITIALIZERS, init_op)
 383     return init_op

 391 class TextFileInitializer(TableInitializerBase):
 392   """Table initializers from a text file.
 393 
 394   This initializer assigns one entry in the table for each line in the file.
 395 
 396   The key and value type of the table to initialize is given by `key_dtype` and
 397   `value_dtype`.
 398 
 399   The key and value content to get from each line is specified by
 400   the `key_index` and `value_index`.
 401 
 402   * `TextFileIndex.LINE_NUMBER` means use the line number starting from zero,
 403     expects data type int64.
 404   * `TextFileIndex.WHOLE_LINE` means use the whole line content, expects data
 405     type string.
 406   * A value `>=0` means use the index (starting at zero) of the split line based
 407       on `delimiter`.
 408 
 409   For example if we have a file with the following content:
 410 
 411   ```
 412   emerson 10
 413   lake 20
 414   palmer 30
 415   ```
 416 
 417   The following snippet initializes a table with the first column as keys and
 418   second column as values:
 419 
 420   * `emerson -> 10`
 421   * `lake -> 20`
 422   * `palmer -> 30`
 423 
 424   ```python
 425   table = tf.lookup.HashTable(tf.lookup.TextFileInitializer(
 426       "test.txt", tf.string, 0, tf.int64, 1, delimiter=" "), -1)
 427   ...
 428   table.init.run()
 429   ```
 430 
 431   Similarly to initialize the whole line as keys and the line number as values.
 432 
 433   * `emerson 10 -> 0`
 434   * `lake 20 -> 1`
 435   * `palmer 30 -> 2`
 436 
 437   ```python
 438   table = tf.lookup.HashTable(tf.lookup.TextFileInitializer(
 439       "test.txt", tf.string, tf.lookup.TextFileIndex.WHOLE_LINE,
 440       tf.int64, tf.lookup.TextFileIndex.LINE_NUMBER, delimiter=" "), -1)
 441   ...
 442   table.init.run()
 443   ```
 444   """
 445 
 446   def __init__(self,
 447                filename,
 448                key_dtype,
 449                key_index,
 450                value_dtype,
 451                value_index,
 452                vocab_size=None,
 453                delimiter="\t",
 454                name=None):
 455     """Constructs a table initializer object to populate from a text file.
 456 
 457     It generates one key-value pair per line. The type of table key and
 458     value are specified by `key_dtype` and `value_dtype`, respectively.
 459     Similarly the content of the key and value are specified by the key_index
 460     and value_index.
 461 
 462     - TextFileIndex.LINE_NUMBER means use the line number starting from zero,
 463       expects data type int64.
 464     - TextFileIndex.WHOLE_LINE means use the whole line content, expects data
 465       type string.
 466     - A value >=0 means use the index (starting at zero) of the split line based
 467       on `delimiter`.
 468 
 469     Args:
 470       filename: The filename of the text file to be used for initialization.
 471         The path must be accessible from wherever the graph is initialized
 472         (eg. trainer or eval workers). The filename may be a scalar `Tensor`.
 473       key_dtype: The `key` data type.
 474       key_index: the index that represents information of a line to get the
 475         table 'key' values from.
 476       value_dtype: The `value` data type.
 477       value_index: the index that represents information of a line to get the
 478         table 'value' values from.'
 479       vocab_size: The number of elements in the file, if known.
 480       delimiter: The delimiter to separate fields in a line.
 481       name: A name for the operation (optional).
 482 
 483     Raises:
 484       ValueError: when the filename is empty, or when the table key and value
 485       data types do not match the expected data types.
 486     """
 487     if not isinstance(filename, ops.Tensor) and not filename:
 488       raise ValueError("Filename required for %s." % name)
 489 
 490     key_dtype = dtypes.as_dtype(key_dtype)
 491     value_dtype = dtypes.as_dtype(value_dtype)
 492 
 493     if key_index < -2:
 494       raise ValueError("Invalid key index %s." % (key_index))
 495 
 496     if key_index == TextFileIndex.LINE_NUMBER and key_dtype != dtypes.int64:
 497       raise ValueError("Signature mismatch. Keys must be dtype %s, got %s." %
 498                        (dtypes.int64, key_dtype))
 499     if ((key_index == TextFileIndex.WHOLE_LINE) and
 500         (not key_dtype.is_integer) and (key_dtype != dtypes.string)):
 501       raise ValueError(
 502           "Signature mismatch. Keys must be integer or string, got %s." %
 503           key_dtype)
 504     if value_index < -2:
 505       raise ValueError("Invalid value index %s." % (value_index))
 506 
 507     if value_index == TextFileIndex.LINE_NUMBER and value_dtype != dtypes.int64:
 508       raise ValueError("Signature mismatch. Values must be dtype %s, got %s." %
 509                        (dtypes.int64, value_dtype))
 510     if value_index == TextFileIndex.WHOLE_LINE and value_dtype != dtypes.string:
 511       raise ValueError("Signature mismatch. Values must be dtype %s, got %s." %
 512                        (dtypes.string, value_dtype))
 513 
 514     if (vocab_size is not None) and (vocab_size <= 0):
 515       raise ValueError("Invalid vocab_size %s." % vocab_size)
 516 
 517     self._filename = filename
 518     self._key_index = key_index
 519     self._value_index = value_index
 520     self._vocab_size = vocab_size
 521     self._delimiter = delimiter
 522     self._name = name
 523 
 524     super(TextFileInitializer, self).__init__(key_dtype, value_dtype)
 525 
 526   def initialize(self, table):
 527     """Initializes the table from a text file.
 528 
 529     Args:
 530       table: The table to be initialized.
 531 
 532     Returns:
 533       The operation that initializes the table.
 534 
 535     Raises:
 536       TypeError: when the keys and values data types do not match the table
 537       key and value data types.
 538     """
 539     _check_table_dtypes(table, self.key_dtype, self.value_dtype)
 540     with ops.name_scope(self._name, "text_file_init",
 541                         (table.table_ref,)) as scope:
 542       filename = ops.convert_to_tensor(
 543           self._filename, dtypes.string, name="asset_filepath")
 544       init_op = gen_lookup_ops.initialize_table_from_text_file_v2(
 545           table.table_ref,
 546           filename,
 547           self._key_index,
 548           self._value_index,
 549           -1 if self._vocab_size is None else self._vocab_size,
 550           self._delimiter,
 551           name=scope)
 552     ops.add_to_collection(ops.GraphKeys.TABLE_INITIALIZERS, init_op)
 553     # If the filename tensor is anything other than a string constant (e.g., if
 554     # it is a placeholder) then it does not make sense to track it as an asset.
 555     if not context.executing_eagerly() and constant_op.is_constant(filename):
 556       ops.add_to_collection(ops.GraphKeys.ASSET_FILEPATHS, filename)
 557     return init_op

 626     The content of the key and value are specified by the key_index
 627     and value_index.
 628 
 629     - TextFileIndex.LINE_NUMBER means use the line number starting from zero,
 630       expects data type int64.
 631     - TextFileIndex.WHOLE_LINE means use the whole line content, expects data
 632       type string.
 633     - A value >=0 means use the index (starting at zero) of the split line based
 634       on `delimiter`.
 635 
 636     Args:
 637       filename: The filename of the text file to be used for initialization.
 638         The path must be accessible from wherever the graph is initialized
 639         (eg. trainer or eval workers). The filename may be a scalar `Tensor`.
 640       key_column_index: The column index from the text file to get the `key`
 641         values from. The default is to use the whole line content.
 642       value_column_index: The column index from the text file to get the `value`
 643         values from. The default is to use the line number, starting from zero.
 644       vocab_size: The number of elements in the file, if known.
 645       delimiter: The delimiter to separate fields in a line.
 646       name: Optional name for the op.
 647       key_dtype: The `key` data type.
 648 
 649     Raises:
 650       TypeError: when the filename is empty, or when the table key and value
 651       data types do not match the expected data types.
 652     """
 653     super(TextFileIdTableInitializer, self).__init__(
 654         filename,
 655         key_dtype,
 656         key_column_index,
 657         dtypes.int64,
 658         value_column_index,
 659         vocab_size=vocab_size,
 660         delimiter=delimiter,
 661         name=name)

 664 class HasherSpec(collections.namedtuple("HasherSpec", ["hasher", "key"])):
 665   """A structure for the spec of the hashing function to use for hash buckets.
 666 
 667   `hasher` is the name of the hashing function to use (eg. "fasthash",
 668   "stronghash").
 669   `key` is optional and specify the key to use for the hash function if
 670   supported, currently only used by a strong hash.
 671 
 672   Fields:
 673     hasher: The hasher name to use.
 674     key: The key to be used by the hashing function, if required.
 675   """
 676   __slots__ = ()
 677 
 678 
 679 FastHashSpec = HasherSpec("fasthash", None)  # pylint: disable=invalid-name

 682 class StrongHashSpec(HasherSpec):
 683   """A structure to specify a key of the strong keyed hash spec.
 684 
 685   The strong hash requires a `key`, which is a list of 2 unsigned integer
 686   numbers. These should be non-zero; random numbers generated from random.org
 687   would be a fine choice.
 688 
 689   Fields:
 690     key: The key to be used by the keyed hashing function.
 691   """
 692   __slots__ = ()
 693 
 694   def __new__(cls, key):
 695     if len(key) != 2:
 696       raise ValueError("key must have size 2, got %s." % len(key))
 697 
 698     if not isinstance(key[0], compat.integral_types) or not isinstance(
 699         key[1], compat.integral_types):
 700       raise TypeError("Invalid key %s. Must be unsigned integer values." % key)
 701 
 702     return super(cls, StrongHashSpec).__new__(cls, "stronghash", key)

 711 class IdTableWithHashBuckets(LookupInterface):
 712   """String to Id table wrapper that assigns out-of-vocabulary keys to buckets.
 713 
 714   For example, if an instance of `IdTableWithHashBuckets` is initialized with a
 715   string-to-id table that maps:
 716 
 717   * `emerson -> 0`
 718   * `lake -> 1`
 719   * `palmer -> 2`
 720 
 721   The `IdTableWithHashBuckets` object will performs the following mapping:
 722 
 723   * `emerson -> 0`
 724   * `lake -> 1`
 725   * `palmer -> 2`
 726   * `<other term> -> bucket_id`, where bucket_id will be between `3` and
 727   `3 + num_oov_buckets - 1`, calculated by:
 728   `hash(<term>) % num_oov_buckets + vocab_size`
 729 
 730   If input_tensor is `["emerson", "lake", "palmer", "king", "crimson"]`,
 731   the lookup result is `[0, 1, 2, 4, 7]`.
 732 
 733   If `table` is None, only out-of-vocabulary buckets are used.
 734 
 735   Example usage:
 736 
 737   ```python
 738   num_oov_buckets = 3
 739   input_tensor = tf.constant(["emerson", "lake", "palmer", "king", "crimnson"])
 740   table = tf.IdTableWithHashBuckets(
 741       tf.HashTable(tf.TextFileIdTableInitializer(filename), default_value),
 742       num_oov_buckets)
 743   out = table.lookup(input_tensor).
 744   table.init.run()
 745   print(out.eval())
 746   ```
 747 
 748   The hash function used for generating out-of-vocabulary buckets ID is handled
 749   by `hasher_spec`.
 750   """
 751 
 752   def __init__(self,
 753                table,
 754                num_oov_buckets,
 755                hasher_spec=FastHashSpec,
 756                name=None,
 757                key_dtype=None):
 758     """Construct a `IdTableWithHashBuckets` object.
 759 
 760     Args:
 761       table: Table that maps `tf.string` or `tf.int64` keys to `tf.int64` ids.
 762       num_oov_buckets: Number of buckets to use for out-of-vocabulary keys.
 763       hasher_spec: A `HasherSpec` to specify the hash function to use for
 764         assignation of out-of-vocabulary buckets  (optional).
 765       name: A name for the operation (optional).
 766       key_dtype: Data type of keys passed to `lookup`. Defaults to
 767         `table.key_dtype` if `table` is specified, otherwise `tf.string`.
 768         Must be string or integer, and must be castable to `table.key_dtype`.
 769 
 770     Raises:
 771       ValueError: when `table` in None and `num_oov_buckets` is not positive.
 772       TypeError: when `hasher_spec` is invalid.
 773     """
 774     # If a name ends with a '/' it is a "name scope", remove all trailing '/'
 775     # characters to use as table name.
 776     if name:
 777       name = name.rstrip("/")
 778     if table:
 779       if key_dtype is None:
 780         key_dtype = table.key_dtype
 781       supported_table_key_dtypes = (dtypes.int64, dtypes.string)
 782       if table.key_dtype not in supported_table_key_dtypes:
 783         raise TypeError("Invalid key dtype, expected one of %s, but got %s." %
 784                         (supported_table_key_dtypes, key_dtype))
 785       if table.key_dtype.is_integer != key_dtype.is_integer:
 786         raise TypeError("Invalid key dtype, expected %s but got %s." %
 787                         ("integer" if key_dtype.is_integer else "non-integer",
 788                          table.key_dtype))
 789       if table.value_dtype != dtypes.int64:
 790         raise TypeError("Invalid value dtype, expected %s but got %s." %
 791                         (dtypes.int64, table.value_dtype))
 792       self._table = table
 793       name = name or self._table.name
 794     else:
 795       if num_oov_buckets <= 0:
 796         raise ValueError("oov_buckets must be > 0 if no table is supplied.")
 797       key_dtype = dtypes.string if key_dtype is None else key_dtype
 798       self._table = None
 799       name = name or "hash_bucket"
 800     if (not key_dtype.is_integer) and (dtypes.string != key_dtype):
 801       raise TypeError(
 802           "Invalid key_dtype, expected integer or string, got %s." % key_dtype)
 803     self._num_oov_buckets = num_oov_buckets
 804 
 805     if not isinstance(hasher_spec, HasherSpec):
 806       raise TypeError(
 807           "hasher_spec must be of type HasherSpec, got %s" % hasher_spec)
 808     self._hasher_spec = hasher_spec
 809     super(IdTableWithHashBuckets, self).__init__(key_dtype, dtypes.int64,
 810                                                  name.split("/")[-1])
 811 
 812   @property
 813   def init(self):
 814     """The table initialization op."""
 815     if self._table:
 816       return self._table.init
 817     with ops.name_scope(None, "init"):
 818       return control_flow_ops.no_op()
 819 
 820   @property
 821   def table_ref(self):
 822     """Returns the table_ref of the underlying table, if one exists.
 823 
 824     Only use the table_ref directly if you know what you are doing. The
 825     table_ref does not have the "hash bucket" functionality, as that is provided
 826     by this class.
 827 
 828     One possible use of the table_ref is subtokenization, i.e. ops which
 829     dynamically decompose tokens into subtokens based on the contents of the
 830     table_ref.
 831 
 832     Returns:
 833       the underlying table_ref, or None if there is no underlying table
 834     """
 835     if self._table is not None:
 836       return self._table.table_ref
 837     return None

 839   def size(self, name=None):
 840     """Compute the number of elements in this table."""
 841     with ops.name_scope(name, "%s_Size" % self.name) as scope:
 842       if self._table:
 843         tsize = self._table.size(scope)
 844       else:
 845         tsize = ops.convert_to_tensor(0, dtype=dtypes.int64)
 846       return tsize + self._num_oov_buckets

 848   def _get_string_to_hash_bucket_fn(self, hasher_spec):
 849     """Returns the string_to_hash_bucket op to use based on `hasher_spec`."""
 850     if not isinstance(hasher_spec, HasherSpec):
 851       raise TypeError("hasher_spec must be of type HasherSpec %s" % hasher_spec)
 852     if hasher_spec.hasher == "fasthash":
 853       return string_ops.string_to_hash_bucket_fast
 854     if hasher_spec.hasher == "legacy":
 855       return string_ops.string_to_hash_bucket
 856     if hasher_spec.hasher == "stronghash":
 857       return functools.partial(
 858           string_ops.string_to_hash_bucket_strong, key=hasher_spec.key)
 859     raise ValueError("Unknown hasher %s" % hasher_spec.hasher)
 860 
 861   def lookup(self, keys, name=None):
 862     """Looks up `keys` in the table, outputs the corresponding values.
 863 
 864     It assigns out-of-vocabulary keys to buckets based in their hashes.
 865 
 866     Args:
 867       keys: Keys to look up. May be either a `SparseTensor` or dense `Tensor`.
 868       name: Optional name for the op.
 869 
 870     Returns:
 871       A `SparseTensor` if keys are sparse, otherwise a dense `Tensor`.
 872 
 873     Raises:
 874       TypeError: when `keys` doesn't match the table key data type.
 875     """
 876     if keys.dtype.base_dtype != self._key_dtype:
 877       raise TypeError("Signature mismatch. Keys must be dtype %s, got %s." %
 878                       (self._key_dtype, keys.dtype))
 879     values = keys
 880     if isinstance(keys, sparse_tensor.SparseTensor):
 881       values = keys.values
 882     if self._table and (self._table.key_dtype.base_dtype == dtypes.int64):
 883       values = math_ops.to_int64(values)
 884 
 885     if self._num_oov_buckets == 0:
 886       ids = self._table.lookup(values, name=name)
 887     else:
 888       # TODO(yleon): Consider moving this functionality to its own kernel.
 889       with ops.name_scope(name, "%s_Lookup" % self.name) as scope:
 890         str_to_hash_bucket = self._get_string_to_hash_bucket_fn(
 891             self._hasher_spec)
 892         buckets = str_to_hash_bucket(
 893             _as_string(values),
 894             num_buckets=self._num_oov_buckets,
 895             name="hash_bucket")
 896         if self._table:
 897           ids = self._table.lookup(values)
 898           buckets = math_ops.add(buckets, self._table.size())
 899           is_id_non_default = math_ops.not_equal(ids, self._table.default_value)
 900           ids = array_ops.where(is_id_non_default, ids, buckets, name=scope)
 901         else:
 902           ids = buckets
 903     if isinstance(keys, sparse_tensor.SparseTensor):
 904       return sparse_tensor.SparseTensor(keys.indices, ids, keys.dense_shape)
 905     return ids

 908 def index_table_from_file(vocabulary_file=None,
 909                           num_oov_buckets=0,
 910                           vocab_size=None,
 911                           default_value=-1,
 912                           hasher_spec=FastHashSpec,
 913                           key_dtype=dtypes.string,
 914                           name=None,
 915                           key_column_index=TextFileIndex.WHOLE_LINE,
 916                           value_column_index=TextFileIndex.LINE_NUMBER,
 917                           delimiter="\t"):
 918   """Returns a lookup table that converts a string tensor into int64 IDs.
 919 
 920   This operation constructs a lookup table to convert tensor of strings into
 921   int64 IDs. The mapping can be initialized from a vocabulary file specified in
 922   `vocabulary_file`, where the whole line is the key and the zero-based line
 923   number is the ID.
 924 
 925   Any lookup of an out-of-vocabulary token will return a bucket ID based on its
 926   hash if `num_oov_buckets` is greater than zero. Otherwise it is assigned the
 927   `default_value`.
 928   The bucket ID range is
 929   `[vocabulary size, vocabulary size + num_oov_buckets - 1]`.
 930 
 931   The underlying table must be initialized by calling
 932   `tf.tables_initializer.run()` or `table.init.run()` once.
 933 
 934   To specify multi-column vocabulary files, use key_column_index and
 935   value_column_index and delimiter.
 936 
 937   - TextFileIndex.LINE_NUMBER means use the line number starting from zero,
 938     expects data type int64.
 939   - TextFileIndex.WHOLE_LINE means use the whole line content, expects data
 940     type string.
 941   - A value >=0 means use the index (starting at zero) of the split line based
 942     on `delimiter`.
 943 
 944   Sample Usages:
 945 
 946   If we have a vocabulary file "test.txt" with the following content:
 947 
 948   ```
 949   emerson
 950   lake
 951   palmer
 952   ```
 953 
 954   ```python
 955   features = tf.constant(["emerson", "lake", "and", "palmer"])
 956   table = tf.lookup.index_table_from_file(
 957       vocabulary_file="test.txt", num_oov_buckets=1)
 958   ids = table.lookup(features)
 959   ...
 960   tf.tables_initializer().run()
 961 
 962   ids.eval()  ==> [0, 1, 3, 2]  # where 3 is the out-of-vocabulary bucket
 963   ```
 964 
 965   Args:
 966     vocabulary_file: The vocabulary filename, may be a constant scalar `Tensor`.
 967     num_oov_buckets: The number of out-of-vocabulary buckets.
 968     vocab_size: Number of the elements in the vocabulary, if known.
 969     default_value: The value to use for out-of-vocabulary feature values.
 970       Defaults to -1.
 971     hasher_spec: A `HasherSpec` to specify the hash function to use for
 972       assignation of out-of-vocabulary buckets.
 973     key_dtype: The `key` data type.
 974     name: A name for this op (optional).
 975     key_column_index: The column index from the text file to get the `key`
 976       values from. The default is to use the whole line content.
 977     value_column_index: The column index from the text file to get the `value`
 978       values from. The default is to use the line number, starting from zero.
 979     delimiter: The delimiter to separate fields in a line.
 980 
 981   Returns:
 982     The lookup table to map a `key_dtype` `Tensor` to index `int64` `Tensor`.
 983 
 984   Raises:
 985     ValueError: If `vocabulary_file` is not set.
 986     ValueError: If `num_oov_buckets` is negative or `vocab_size` is not greater
 987       than zero.
 988   """
 989   if vocabulary_file is None or (
 990       isinstance(vocabulary_file, six.string_types) and not vocabulary_file):
 991     raise ValueError("vocabulary_file must be specified and must not be empty.")
 992   if num_oov_buckets < 0:
 993     raise ValueError("num_oov_buckets must be greater or equal than 0, got %d."
 994                      % num_oov_buckets)
 995   if vocab_size is not None and vocab_size < 1:
 996     vocab_file_value = vocabulary_file
 997     if isinstance(vocabulary_file, ops.Tensor):
 998       vocab_file_value = tensor_util.constant_value(vocabulary_file) or "?"
 999     raise ValueError("vocab_size must be greater than 0, got %d. "
1000                      "vocabulary_file: %s" % (vocab_size, vocab_file_value))
1001   if (not key_dtype.is_integer) and (dtypes.string != key_dtype.base_dtype):
1002     raise TypeError("Only integer and string keys are supported.")
1003 
1004   with ops.name_scope(name, "string_to_index") as feat_to_id_scope:
1005     table = None
1006     shared_name = ""
1007     with ops.name_scope(None, "hash_table") as hash_table_scope:
1008       if vocab_size:
1009         # Keep the shared_name:
1010         # <table_type>_<filename>_<vocab_size>_<key_index>_<value_index>
1011         shared_name = "hash_table_%s_%d_%s_%s" % (vocabulary_file, vocab_size,
1012                                                   key_column_index,
1013                                                   value_column_index)
1014       else:
1015         # Keep the shared_name
1016         # <table_type>_<filename>_<key_index>_<value_index>
1017         shared_name = "hash_table_%s_%s_%s" % (vocabulary_file,
1018                                                key_column_index,
1019                                                value_column_index)
1020       init = TextFileIdTableInitializer(
1021           vocabulary_file,
1022           vocab_size=vocab_size,
1023           key_dtype=dtypes.int64 if key_dtype.is_integer else key_dtype,
1024           name="table_init",
1025           key_column_index=key_column_index,
1026           value_column_index=value_column_index,
1027           delimiter=delimiter)
1028 
1029       table = HashTable(
1030           init, default_value, shared_name=shared_name, name=hash_table_scope)
1031     if num_oov_buckets:
1032       table = IdTableWithHashBuckets(
1033           table,
1034           num_oov_buckets=num_oov_buckets,
1035           hasher_spec=hasher_spec,
1036           name=feat_to_id_scope,
1037           key_dtype=key_dtype)
1038 
1039     return table

1042 def index_table_from_tensor(vocabulary_list,
1043                             num_oov_buckets=0,
1044                             default_value=-1,
1045                             hasher_spec=FastHashSpec,
1046                             dtype=dtypes.string,
1047                             name=None):
1048   """Returns a lookup table that converts a string tensor into int64 IDs.
1049 
1050   This operation constructs a lookup table to convert tensor of strings into
1051   int64 IDs. The mapping can be initialized from a string `vocabulary_list` 1-D
1052   tensor where each element is a key and corresponding index within the tensor
1053   is the value.
1054 
1055   Any lookup of an out-of-vocabulary token will return a bucket ID based on its
1056   hash if `num_oov_buckets` is greater than zero. Otherwise it is assigned the
1057   `default_value`. The bucket ID range is
1058   `[vocabulary list size, vocabulary list size + num_oov_buckets - 1]`.
1059 
1060   The underlying table must be initialized by calling
1061   `tf.tables_initializer.run()` or `table.init.run()` once.
1062 
1063   Elements in `vocabulary_list` cannot have duplicates, otherwise when executing
1064   the table initializer op, it will throw a `FailedPreconditionError`.
1065 
1066   Sample Usages:
1067 
1068   ```python
1069   vocabulary_list = tf.constant(["emerson", "lake", "palmer"])
1070   table = tf.lookup.index_table_from_tensor(
1071       vocabulary_list=vocabulary_list, num_oov_buckets=1, default_value=-1)
1072   features = tf.constant(["emerson", "lake", "and", "palmer"])
1073   ids = table.lookup(features)
1074   ...
1075   tf.tables_initializer().run()
1076 
1077   ids.eval()  ==> [0, 1, 4, 2]
1078   ```
1079 
1080   Args:
1081     vocabulary_list: A 1-D `Tensor` that specifies the mapping of keys to
1082       indices. The type of this object must be castable to `dtype`.
1083     num_oov_buckets: The number of out-of-vocabulary buckets.
1084     default_value: The value to use for out-of-vocabulary feature values.
1085       Defaults to -1.
1086     hasher_spec: A `HasherSpec` to specify the hash function to use for
1087       assignment of out-of-vocabulary buckets.
1088     dtype: The type of values passed to `lookup`. Only string and integers are
1089       supported.
1090     name: A name for this op (optional).
1091 
1092   Returns:
1093     The lookup table to map an input `Tensor` to index `int64` `Tensor`.
1094 
1095   Raises:
1096     ValueError: If `vocabulary_list` is invalid.
1097     ValueError: If `num_oov_buckets` is negative.
1098   """
1099   if vocabulary_list is None:
1100     raise ValueError("vocabulary_list must be specified.")
1101 
1102   if num_oov_buckets < 0:
1103     raise ValueError("num_oov_buckets must be greater or equal than 0, got %d."
1104                      % num_oov_buckets)
1105 
1106   if (not dtype.is_integer) and (dtypes.string != dtype.base_dtype):
1107     raise TypeError("Only integer and string keys are supported.")
1108 
1109   with ops.name_scope(name, "string_to_index") as feat_to_id_scope:
1110     keys = ops.convert_to_tensor(vocabulary_list)
1111     if keys.dtype.is_integer != dtype.is_integer:
1112       raise ValueError("Expected %s, got %s." %
1113                        ("integer"
1114                         if dtype.is_integer else "non-integer", keys.dtype))
1115     if (not dtype.is_integer) and (keys.dtype.base_dtype != dtype):
1116       raise ValueError("Expected %s, got %s." % (dtype, keys.dtype))
1117     num_elements = array_ops.size(keys)
1118     values = math_ops.to_int64(math_ops.range(num_elements))
1119 
1120     shared_name = ""
1121     with ops.name_scope(None, "hash_table") as hash_table_scope:
1122       if context.executing_eagerly():
1123         # Ensure a unique name when eager execution is enabled to avoid spurious
1124         # sharing issues.
1125         shared_name += str(ops.uid())
1126       table_keys = math_ops.to_int64(keys) if keys.dtype.is_integer else keys
1127       init = KeyValueTensorInitializer(
1128           table_keys,
1129           values,
1130           table_keys.dtype.base_dtype,
1131           dtypes.int64,
1132           name="table_init")
1133       table = HashTable(
1134           init, default_value, shared_name=shared_name, name=hash_table_scope)
1135     if num_oov_buckets:
1136       table = IdTableWithHashBuckets(
1137           table,
1138           num_oov_buckets=num_oov_buckets,
1139           hasher_spec=hasher_spec,
1140           name=feat_to_id_scope,
1141           key_dtype=dtype)
1142 
1143     return table

1247 def index_to_string_table_from_tensor(vocabulary_list,
1248                                       default_value="UNK",
1249                                       name=None):
1250   """Returns a lookup table that maps a `Tensor` of indices into strings.
1251 
1252   This operation constructs a lookup table to map int64 indices into string
1253   values. The mapping is initialized from a string `vocabulary_list` 1-D
1254   `Tensor` where each element is a value and the corresponding index within the
1255   tensor is the key.
1256 
1257   Any input which does not have a corresponding index in 'vocabulary_list'
1258   (an out-of-vocabulary entry) is assigned the `default_value`
1259 
1260   The underlying table must be initialized by calling
1261   `tf.tables_initializer.run()` or `table.init.run()` once.
1262 
1263   Elements in `vocabulary_list` cannot have duplicates, otherwise when executing
1264   the table initializer op, it will throw a `FailedPreconditionError`.
1265 
1266   Sample Usages:
1267 
1268   ```python
1269   vocabulary_list = tf.constant(["emerson", "lake", "palmer"])
1270   indices = tf.constant([1, 5], tf.int64)
1271   table = tf.lookup.index_to_string_table_from_tensor(
1272       vocabulary_list, default_value="UNKNOWN")
1273   values = table.lookup(indices)
1274   ...
1275   tf.tables_initializer().run()
1276 
1277   values.eval() ==> ["lake", "UNKNOWN"]
1278   ```
1279 
1280   Args:
1281     vocabulary_list: A 1-D string `Tensor` that specifies the strings to map
1282       from indices.
1283     default_value: The value to use for out-of-vocabulary indices.
1284     name: A name for this op (optional).
1285 
1286   Returns:
1287     The lookup table to map a string values associated to a given index `int64`
1288     `Tensors`.
1289 
1290   Raises:
1291     ValueError: when `vocabulary_list` is not set.
1292   """
1293 
1294   if vocabulary_list is None:
1295     raise ValueError("vocabulary_list must be specified.")
1296 
1297   with ops.name_scope(name, "index_to_string") as scope:
1298     vocabulary_list = ops.convert_to_tensor(vocabulary_list, dtypes.string)
1299     num_elements = array_ops.size(vocabulary_list)
1300     keys = math_ops.to_int64(math_ops.range(num_elements))
1301 
1302     shared_name = ""
1303     init = KeyValueTensorInitializer(
1304         keys, vocabulary_list, dtypes.int64, dtypes.string, name="table_init")
1305     # TODO(yleon): Use a more effienct structure.
1306     return HashTable(init, default_value, shared_name=shared_name, name=scope)
