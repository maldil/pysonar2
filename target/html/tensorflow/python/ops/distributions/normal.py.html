  44 @tf_export("distributions.Normal")
  45 class Normal(distribution.Distribution):
  46   """The Normal distribution with location `loc` and `scale` parameters.
  47 
  48   #### Mathematical details
  49 
  50   The probability density function (pdf) is,
  51 
  52   ```none
  53   pdf(x; mu, sigma) = exp(-0.5 (x - mu)**2 / sigma**2) / Z
  54   Z = (2 pi sigma**2)**0.5
  55   ```
  56 
  57   where `loc = mu` is the mean, `scale = sigma` is the std. deviation, and, `Z`
  58   is the normalization constant.
  59 
  60   The Normal distribution is a member of the [location-scale family](
  61   https://en.wikipedia.org/wiki/Location-scale_family), i.e., it can be
  62   constructed as,
  63 
  64   ```none
  65   X ~ Normal(loc=0, scale=1)
  66   Y = loc + scale * X
  67   ```
  68 
  69   #### Examples
  70 
  71   Examples of initialization of one or a batch of distributions.
  72 
  73   ```python
  74   import tensorflow_probability as tfp
  75   tfd = tfp.distributions
  76 
  77   # Define a single scalar Normal distribution.
  78   dist = tfd.Normal(loc=0., scale=3.)
  79 
  80   # Evaluate the cdf at 1, returning a scalar.
  81   dist.cdf(1.)
  82 
  83   # Define a batch of two scalar valued Normals.
  84   # The first has mean 1 and standard deviation 11, the second 2 and 22.
  85   dist = tfd.Normal(loc=[1, 2.], scale=[11, 22.])
  86 
  87   # Evaluate the pdf of the first distribution on 0, and the second on 1.5,
  88   # returning a length two tensor.
  89   dist.prob([0, 1.5])
  90 
  91   # Get 3 samples, returning a 3 x 2 tensor.
  92   dist.sample([3])
  93   ```
  94 
  95   Arguments are broadcast when possible.
  96 
  97   ```python
  98   # Define a batch of two scalar valued Normals.
  99   # Both have mean 1, but different standard deviations.
 100   dist = tfd.Normal(loc=1., scale=[11, 22.])
 101 
 102   # Evaluate the pdf of both distributions on the same point, 3.0,
 103   # returning a length 2 tensor.
 104   dist.prob(3.0)
 105   ```
 106 
 107   """
 108 
 109   def __init__(self,
 110                loc,
 111                scale,
 112                validate_args=False,
 113                allow_nan_stats=True,
 114                name="Normal"):
 115     """Construct Normal distributions with mean and stddev `loc` and `scale`.
 116 
 117     The parameters `loc` and `scale` must be shaped in a way that supports
 118     broadcasting (e.g. `loc + scale` is a valid operation).
 119 
 120     Args:
 121       loc: Floating point tensor; the means of the distribution(s).
 122       scale: Floating point tensor; the stddevs of the distribution(s).
 123         Must contain only positive values.
 124       validate_args: Python `bool`, default `False`. When `True` distribution
 125         parameters are checked for validity despite possibly degrading runtime
 126         performance. When `False` invalid inputs may silently render incorrect
 127         outputs.
 128       allow_nan_stats: Python `bool`, default `True`. When `True`,
 129         statistics (e.g., mean, mode, variance) use the value "`NaN`" to
 130         indicate the result is undefined. When `False`, an exception is raised
 131         if one or more of the statistic's batch members are undefined.
 132       name: Python `str` name prefixed to Ops created by this class.
 133 
 134     Raises:
 135       TypeError: if `loc` and `scale` have different `dtype`.
 136     """
 137     parameters = dict(locals())
 138     with ops.name_scope(name, values=[loc, scale]) as name:
 139       with ops.control_dependencies([check_ops.assert_positive(scale)] if
 140                                     validate_args else []):
 141         self._loc = array_ops.identity(loc, name="loc")
 142         self._scale = array_ops.identity(scale, name="scale")
 143         check_ops.assert_same_float_dtype([self._loc, self._scale])
 144     super(Normal, self).__init__(
 145         dtype=self._scale.dtype,
 146         reparameterization_type=distribution.FULLY_REPARAMETERIZED,
 147         validate_args=validate_args,
 148         allow_nan_stats=allow_nan_stats,
 149         parameters=parameters,
 150         graph_parents=[self._loc, self._scale],
 151         name=name)
 152 
 153   @staticmethod
 154   def _param_shapes(sample_shape):
 155     return dict(
 156         zip(("loc", "scale"), ([ops.convert_to_tensor(
 157             sample_shape, dtype=dtypes.int32)] * 2)))
 158 
 159   @property
 160   def loc(self):
 161     """Distribution parameter for the mean."""
 162     return self._loc
 163 
 164   @property
 165   def scale(self):
 166     """Distribution parameter for standard deviation."""
 167     return self._scale
 168 
 169   def _batch_shape_tensor(self):
 170     return array_ops.broadcast_dynamic_shape(
 171         array_ops.shape(self.loc),
 172         array_ops.shape(self.scale))
 173 
 174   def _batch_shape(self):
 175     return array_ops.broadcast_static_shape(
 176         self.loc.get_shape(),
 177         self.scale.get_shape())
 178 
 179   def _event_shape_tensor(self):
 180     return constant_op.constant([], dtype=dtypes.int32)
 181 
 182   def _event_shape(self):
 183     return tensor_shape.scalar()
 184 
 185   def _sample_n(self, n, seed=None):
 186     shape = array_ops.concat([[n], self.batch_shape_tensor()], 0)
 187     sampled = random_ops.random_normal(
 188         shape=shape, mean=0., stddev=1., dtype=self.loc.dtype, seed=seed)
 189     return sampled * self.scale + self.loc
 190 
 191   def _log_prob(self, x):
 192     return self._log_unnormalized_prob(x) - self._log_normalization()
 193 
 194   def _log_cdf(self, x):
 195     return special_math.log_ndtr(self._z(x))
 196 
 197   def _cdf(self, x):
 198     return special_math.ndtr(self._z(x))
 199 
 200   def _log_survival_function(self, x):
 201     return special_math.log_ndtr(-self._z(x))
 202 
 203   def _survival_function(self, x):
 204     return special_math.ndtr(-self._z(x))
 205 
 206   def _log_unnormalized_prob(self, x):
 207     return -0.5 * math_ops.square(self._z(x))
 208 
 209   def _log_normalization(self):
 210     return 0.5 * math.log(2. * math.pi) + math_ops.log(self.scale)
 211 
 212   def _entropy(self):
 213     # Use broadcasting rules to calculate the full broadcast scale.
 214     scale = self.scale * array_ops.ones_like(self.loc)
 215     return 0.5 * math.log(2. * math.pi * math.e) + math_ops.log(scale)
 216 
 217   def _mean(self):
 218     return self.loc * array_ops.ones_like(self.scale)
 219 
 220   def _quantile(self, p):
 221     return self._inv_z(special_math.ndtri(p))
 222 
 223   def _stddev(self):
 224     return self.scale * array_ops.ones_like(self.loc)
 225 
 226   def _mode(self):
 227     return self._mean()
 228 
 229   def _z(self, x):
 230     """Standardize input `x` to a unit normal."""
 231     with ops.name_scope("standardize", values=[x]):
 232       return (x - self.loc) / self.scale
 233 
 234   def _inv_z(self, z):
 235     """Reconstruct input `x` from a its normalized version."""
 236     with ops.name_scope("reconstruct", values=[z]):
 237       return z * self.scale + self.loc

 240 class NormalWithSoftplusScale(Normal):
 241   """Normal with softplus applied to `scale`."""
 242 
 243   def __init__(self,
 244                loc,
 245                scale,
 246                validate_args=False,
 247                allow_nan_stats=True,
 248                name="NormalWithSoftplusScale"):
 249     parameters = dict(locals())
 250     with ops.name_scope(name, values=[scale]) as name:
 251       super(NormalWithSoftplusScale, self).__init__(
 252           loc=loc,
 253           scale=nn.softplus(scale, name="softplus_scale"),
 254           validate_args=validate_args,
 255           allow_nan_stats=allow_nan_stats,
 256           name=name)
 257     self._parameters = parameters

 260 @kullback_leibler.RegisterKL(Normal, Normal)
 261 def _kl_normal_normal(n_a, n_b, name=None):
 262   """Calculate the batched KL divergence KL(n_a || n_b) with n_a and n_b Normal.
 263 
 264   Args:
 265     n_a: instance of a Normal distribution object.
 266     n_b: instance of a Normal distribution object.
 267     name: (optional) Name to use for created operations.
 268       default is "kl_normal_normal".
 269 
 270   Returns:
 271     Batchwise KL(n_a || n_b)
 272   """
 273   with ops.name_scope(name, "kl_normal_normal", [n_a.loc, n_b.loc]):
 274     one = constant_op.constant(1, dtype=n_a.dtype)
 275     two = constant_op.constant(2, dtype=n_a.dtype)
 276     half = constant_op.constant(0.5, dtype=n_a.dtype)
 277     s_a_squared = math_ops.square(n_a.scale)
 278     s_b_squared = math_ops.square(n_b.scale)
 279     ratio = s_a_squared / s_b_squared
 280     return (math_ops.square(n_a.loc - n_b.loc) / (two * s_b_squared) +
 281             half * (ratio - one - math_ops.log(ratio)))
