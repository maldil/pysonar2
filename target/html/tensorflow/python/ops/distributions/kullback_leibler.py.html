  38 def _registered_kl(type_a, type_b):
  39   """Get the KL function registered for classes a and b."""
  40   hierarchy_a = tf_inspect.getmro(type_a)
  41   hierarchy_b = tf_inspect.getmro(type_b)
  42   dist_to_children = None
  43   kl_fn = None
  44   for mro_to_a, parent_a in enumerate(hierarchy_a):
  45     for mro_to_b, parent_b in enumerate(hierarchy_b):
  46       candidate_dist = mro_to_a + mro_to_b
  47       candidate_kl_fn = _DIVERGENCES.get((parent_a, parent_b), None)
  48       if not kl_fn or (candidate_kl_fn and candidate_dist < dist_to_children):
  49         dist_to_children = candidate_dist
  50         kl_fn = candidate_kl_fn
  51   return kl_fn

  54 @tf_export("distributions.kl_divergence")
  55 def kl_divergence(distribution_a, distribution_b,
  56                   allow_nan_stats=True, name=None):
  57   """Get the KL-divergence KL(distribution_a || distribution_b).
  58 
  59   If there is no KL method registered specifically for `type(distribution_a)`
  60   and `type(distribution_b)`, then the class hierarchies of these types are
  61   searched.
  62 
  63   If one KL method is registered between any pairs of classes in these two
  64   parent hierarchies, it is used.
  65 
  66   If more than one such registered method exists, the method whose registered
  67   classes have the shortest sum MRO paths to the input types is used.
  68 
  69   If more than one such shortest path exists, the first method
  70   identified in the search is used (favoring a shorter MRO distance to
  71   `type(distribution_a)`).
  72 
  73   Args:
  74     distribution_a: The first distribution.
  75     distribution_b: The second distribution.
  76     allow_nan_stats: Python `bool`, default `True`. When `True`,
  77       statistics (e.g., mean, mode, variance) use the value "`NaN`" to
  78       indicate the result is undefined. When `False`, an exception is raised
  79       if one or more of the statistic's batch members are undefined.
  80     name: Python `str` name prefixed to Ops created by this class.
  81 
  82   Returns:
  83     A Tensor with the batchwise KL-divergence between `distribution_a`
  84     and `distribution_b`.
  85 
  86   Raises:
  87     NotImplementedError: If no KL method is defined for distribution types
  88       of `distribution_a` and `distribution_b`.
  89   """
  90   kl_fn = _registered_kl(type(distribution_a), type(distribution_b))
  91   if kl_fn is None:
  92     raise NotImplementedError(
  93         "No KL(distribution_a || distribution_b) registered for distribution_a "
  94         "type %s and distribution_b type %s"
  95         % (type(distribution_a).__name__, type(distribution_b).__name__))
  96 
  97   with ops.name_scope("KullbackLeibler"):
  98     kl_t = kl_fn(distribution_a, distribution_b, name=name)
  99     if allow_nan_stats:
 100       return kl_t
 101 
 102     # Check KL for NaNs
 103     kl_t = array_ops.identity(kl_t, name="kl")
 104 
 105     with ops.control_dependencies([
 106         control_flow_ops.Assert(
 107             math_ops.logical_not(
 108                 math_ops.reduce_any(math_ops.is_nan(kl_t))),
 109             ["KL calculation between %s and %s returned NaN values "
 110              "(and was called with allow_nan_stats=False). Values:"
 111              % (distribution_a.name, distribution_b.name), kl_t])]):
 112       return array_ops.identity(kl_t, name="checked_kl")

 115 def cross_entropy(ref, other,
 116                   allow_nan_stats=True, name=None):
 117   """Computes the (Shannon) cross entropy.
 118 
 119   Denote two distributions by `P` (`ref`) and `Q` (`other`). Assuming `P, Q`
 120   are absolutely continuous with respect to one another and permit densities
 121   `p(x) dr(x)` and `q(x) dr(x)`, (Shanon) cross entropy is defined as:
 122 
 123   ```none
 124   H[P, Q] = E_p[-log q(X)] = -int_F p(x) log q(x) dr(x)
 125   ```
 126 
 127   where `F` denotes the support of the random variable `X ~ P`.
 128 
 129   Args:
 130     ref: `tfd.Distribution` instance.
 131     other: `tfd.Distribution` instance.
 132     allow_nan_stats: Python `bool`, default `True`. When `True`,
 133       statistics (e.g., mean, mode, variance) use the value "`NaN`" to
 134       indicate the result is undefined. When `False`, an exception is raised
 135       if one or more of the statistic's batch members are undefined.
 136     name: Python `str` prepended to names of ops created by this function.
 137 
 138   Returns:
 139     cross_entropy: `ref.dtype` `Tensor` with shape `[B1, ..., Bn]`
 140       representing `n` different calculations of (Shanon) cross entropy.
 141   """
 142   with ops.name_scope(name, "cross_entropy"):
 143     return ref.entropy() + kl_divergence(
 144         ref, other, allow_nan_stats=allow_nan_stats)

 147 @tf_export("distributions.RegisterKL")
 148 class RegisterKL(object):
 149   """Decorator to register a KL divergence implementation function.
 150 
 151   Usage:
 152 
 153   @distributions.RegisterKL(distributions.Normal, distributions.Normal)
 154   def _kl_normal_mvn(norm_a, norm_b):
 155     # Return KL(norm_a || norm_b)
 156   """
 157 
 158   def __init__(self, dist_cls_a, dist_cls_b):
 159     """Initialize the KL registrar.
 160 
 161     Args:
 162       dist_cls_a: the class of the first argument of the KL divergence.
 163       dist_cls_b: the class of the second argument of the KL divergence.
 164     """
 165     self._key = (dist_cls_a, dist_cls_b)
 166 
 167   def __call__(self, kl_fn):
 168     """Perform the KL registration.
 169 
 170     Args:
 171       kl_fn: The function to use for the KL divergence.
 172 
 173     Returns:
 174       kl_fn
 175 
 176     Raises:
 177       TypeError: if kl_fn is not a callable.
 178       ValueError: if a KL divergence function has already been registered for
 179         the given argument classes.
 180     """
 181     if not callable(kl_fn):
 182       raise TypeError("kl_fn must be callable, received: %s" % kl_fn)
 183     if self._key in _DIVERGENCES:
 184       raise ValueError("KL(%s || %s) has already been registered to: %s"
 185                        % (self._key[0].__name__, self._key[1].__name__,
 186                           _DIVERGENCES[self._key]))
 187     _DIVERGENCES[self._key] = kl_fn
 188     return kl_fn
