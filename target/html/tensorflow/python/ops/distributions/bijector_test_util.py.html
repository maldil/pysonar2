  28 def assert_finite(array):
  29   if not np.isfinite(array).all():
  30     raise AssertionError("array was not all finite. %s" % array[:15])

  33 def assert_strictly_increasing(array):
  34   np.testing.assert_array_less(0., np.diff(array))
  35 
  36 
  37 def assert_strictly_decreasing(array):
  38   np.testing.assert_array_less(np.diff(array), 0.)
  39 
  40 
  41 def assert_strictly_monotonic(array):
  42   if array[0] < array[-1]:
  43     assert_strictly_increasing(array)
  44   else:
  45     assert_strictly_decreasing(array)

  48 def assert_scalar_congruency(bijector,
  49                              lower_x,
  50                              upper_x,
  51                              n=int(10e3),
  52                              rtol=0.01,
  53                              sess=None):
  54   """Assert `bijector`'s forward/inverse/inverse_log_det_jacobian are congruent.
  55 
  56   We draw samples `X ~ U(lower_x, upper_x)`, then feed these through the
  57   `bijector` in order to check that:
  58 
  59   1. the forward is strictly monotonic.
  60   2. the forward/inverse methods are inverses of each other.
  61   3. the jacobian is the correct change of measure.
  62 
  63   This can only be used for a Bijector mapping open subsets of the real line
  64   to themselves.  This is due to the fact that this test compares the `prob`
  65   before/after transformation with the Lebesgue measure on the line.
  66 
  67   Args:
  68     bijector:  Instance of Bijector
  69     lower_x:  Python scalar.
  70     upper_x:  Python scalar.  Must have `lower_x < upper_x`, and both must be in
  71       the domain of the `bijector`.  The `bijector` should probably not produce
  72       huge variation in values in the interval `(lower_x, upper_x)`, or else
  73       the variance based check of the Jacobian will require small `rtol` or
  74       huge `n`.
  75     n:  Number of samples to draw for the checks.
  76     rtol:  Positive number.  Used for the Jacobian check.
  77     sess:  `tf.Session`.  Defaults to the default session.
  78 
  79   Raises:
  80     AssertionError:  If tests fail.
  81   """
  82   # Checks and defaults.
  83   if sess is None:
  84     sess = ops.get_default_session()
  85 
  86   # Should be monotonic over this interval
  87   ten_x_pts = np.linspace(lower_x, upper_x, num=10).astype(np.float32)
  88   if bijector.dtype is not None:
  89     ten_x_pts = ten_x_pts.astype(bijector.dtype.as_numpy_dtype)
  90   forward_on_10_pts = bijector.forward(ten_x_pts)
  91 
  92   # Set the lower/upper limits in the range of the bijector.
  93   lower_y, upper_y = sess.run(
  94       [bijector.forward(lower_x), bijector.forward(upper_x)])
  95   if upper_y < lower_y:  # If bijector.forward is a decreasing function.
  96     lower_y, upper_y = upper_y, lower_y
  97 
  98   # Uniform samples from the domain, range.
  99   uniform_x_samps = uniform_lib.Uniform(
 100       low=lower_x, high=upper_x).sample(n, seed=0)
 101   uniform_y_samps = uniform_lib.Uniform(
 102       low=lower_y, high=upper_y).sample(n, seed=1)
 103 
 104   # These compositions should be the identity.
 105   inverse_forward_x = bijector.inverse(bijector.forward(uniform_x_samps))
 106   forward_inverse_y = bijector.forward(bijector.inverse(uniform_y_samps))
 107 
 108   # For a < b, and transformation y = y(x),
 109   # (b - a) = \int_a^b dx = \int_{y(a)}^{y(b)} |dx/dy| dy
 110   # "change_measure_dy_dx" below is a Monte Carlo approximation to the right
 111   # hand side, which should then be close to the left, which is (b - a).
 112   # We assume event_ndims=0 because we assume scalar -> scalar. The log_det
 113   # methods will handle whether they expect event_ndims > 0.
 114   dy_dx = math_ops.exp(bijector.inverse_log_det_jacobian(
 115       uniform_y_samps, event_ndims=0))
 116   # E[|dx/dy|] under Uniform[lower_y, upper_y]
 117   # = \int_{y(a)}^{y(b)} |dx/dy| dP(u), where dP(u) is the uniform measure
 118   expectation_of_dy_dx_under_uniform = math_ops.reduce_mean(dy_dx)
 119   # dy = dP(u) * (upper_y - lower_y)
 120   change_measure_dy_dx = (
 121       (upper_y - lower_y) * expectation_of_dy_dx_under_uniform)
 122 
 123   # We'll also check that dy_dx = 1 / dx_dy.
 124   dx_dy = math_ops.exp(
 125       bijector.forward_log_det_jacobian(
 126           bijector.inverse(uniform_y_samps), event_ndims=0))
 127 
 128   [
 129       forward_on_10_pts_v,
 130       dy_dx_v,
 131       dx_dy_v,
 132       change_measure_dy_dx_v,
 133       uniform_x_samps_v,
 134       uniform_y_samps_v,
 135       inverse_forward_x_v,
 136       forward_inverse_y_v,
 137   ] = sess.run([
 138       forward_on_10_pts,
 139       dy_dx,
 140       dx_dy,
 141       change_measure_dy_dx,
 142       uniform_x_samps,
 143       uniform_y_samps,
 144       inverse_forward_x,
 145       forward_inverse_y,
 146   ])
 147 
 148   assert_strictly_monotonic(forward_on_10_pts_v)
 149   # Composition of forward/inverse should be the identity.
 150   np.testing.assert_allclose(
 151       inverse_forward_x_v, uniform_x_samps_v, atol=1e-5, rtol=1e-3)
 152   np.testing.assert_allclose(
 153       forward_inverse_y_v, uniform_y_samps_v, atol=1e-5, rtol=1e-3)
 154   # Change of measure should be correct.
 155   np.testing.assert_allclose(
 156       upper_x - lower_x, change_measure_dy_dx_v, atol=0, rtol=rtol)
 157   # Inverse Jacobian should be equivalent to the reciprocal of the forward
 158   # Jacobian.
 159   np.testing.assert_allclose(
 160       dy_dx_v, np.divide(1., dx_dy_v), atol=1e-5, rtol=1e-3)

 163 def assert_bijective_and_finite(
 164     bijector, x, y, event_ndims, atol=0, rtol=1e-5, sess=None):
 165   """Assert that forward/inverse (along with jacobians) are inverses and finite.
 166 
 167   It is recommended to use x and y values that are very very close to the edge
 168   of the Bijector's domain.
 169 
 170   Args:
 171     bijector:  A Bijector instance.
 172     x:  np.array of values in the domain of bijector.forward.
 173     y:  np.array of values in the domain of bijector.inverse.
 174     event_ndims: Integer describing the number of event dimensions this bijector
 175       operates on.
 176     atol:  Absolute tolerance.
 177     rtol:  Relative tolerance.
 178     sess:  TensorFlow session.  Defaults to the default session.
 179 
 180   Raises:
 181     AssertionError:  If tests fail.
 182   """
 183   sess = sess or ops.get_default_session()
 184 
 185   # These are the incoming points, but people often create a crazy range of
 186   # values for which these end up being bad, especially in 16bit.
 187   assert_finite(x)
 188   assert_finite(y)
 189 
 190   f_x = bijector.forward(x)
 191   g_y = bijector.inverse(y)
 192 
 193   [
 194       x_from_x,
 195       y_from_y,
 196       ildj_f_x,
 197       fldj_x,
 198       ildj_y,
 199       fldj_g_y,
 200       f_x_v,
 201       g_y_v,
 202   ] = sess.run([
 203       bijector.inverse(f_x),
 204       bijector.forward(g_y),
 205       bijector.inverse_log_det_jacobian(f_x, event_ndims=event_ndims),
 206       bijector.forward_log_det_jacobian(x, event_ndims=event_ndims),
 207       bijector.inverse_log_det_jacobian(y, event_ndims=event_ndims),
 208       bijector.forward_log_det_jacobian(g_y, event_ndims=event_ndims),
 209       f_x,
 210       g_y,
 211   ])
 212 
 213   assert_finite(x_from_x)
 214   assert_finite(y_from_y)
 215   assert_finite(ildj_f_x)
 216   assert_finite(fldj_x)
 217   assert_finite(ildj_y)
 218   assert_finite(fldj_g_y)
 219   assert_finite(f_x_v)
 220   assert_finite(g_y_v)
 221 
 222   np.testing.assert_allclose(x_from_x, x, atol=atol, rtol=rtol)
 223   np.testing.assert_allclose(y_from_y, y, atol=atol, rtol=rtol)
 224   np.testing.assert_allclose(-ildj_f_x, fldj_x, atol=atol, rtol=rtol)
 225   np.testing.assert_allclose(-ildj_y, fldj_g_y, atol=atol, rtol=rtol)
