  40 _multinomial_sample_note = """For each batch of counts, `value = [n_0, ...
  41 ,n_{k-1}]`, `P[value]` is the probability that after sampling `self.total_count`
  42 draws from this Multinomial distribution, the number of draws falling in class
  43 `j` is `n_j`. Since this definition is [exchangeable](
  44 https://en.wikipedia.org/wiki/Exchangeable_random_variables); different
  45 sequences have the same counts so the probability includes a combinatorial
  46 coefficient.
  47 
  48 Note: `value` must be a non-negative tensor with dtype `self.dtype`, have no
  49 fractional components, and such that
  50 `tf.reduce_sum(value, -1) = self.total_count`. Its shape must be broadcastable
  51 with `self.probs` and `self.total_count`."""
  52 
  53 
  54 @tf_export("distributions.Multinomial")
  55 class Multinomial(distribution.Distribution):
  56   """Multinomial distribution.
  57 
  58   This Multinomial distribution is parameterized by `probs`, a (batch of)
  59   length-`K` `prob` (probability) vectors (`K > 1`) such that
  60   `tf.reduce_sum(probs, -1) = 1`, and a `total_count` number of trials, i.e.,
  61   the number of trials per draw from the Multinomial. It is defined over a
  62   (batch of) length-`K` vector `counts` such that
  63   `tf.reduce_sum(counts, -1) = total_count`. The Multinomial is identically the
  64   Binomial distribution when `K = 2`.
  65 
  66   #### Mathematical Details
  67 
  68   The Multinomial is a distribution over `K`-class counts, i.e., a length-`K`
  69   vector of non-negative integer `counts = n = [n_0, ..., n_{K-1}]`.
  70 
  71   The probability mass function (pmf) is,
  72 
  73   ```none
  74   pmf(n; pi, N) = prod_j (pi_j)**n_j / Z
  75   Z = (prod_j n_j!) / N!
  76   ```
  77 
  78   where:
  79   * `probs = pi = [pi_0, ..., pi_{K-1}]`, `pi_j > 0`, `sum_j pi_j = 1`,
  80   * `total_count = N`, `N` a positive integer,
  81   * `Z` is the normalization constant, and,
  82   * `N!` denotes `N` factorial.
  83 
  84   Distribution parameters are automatically broadcast in all functions; see
  85   examples for details.
  86 
  87   #### Pitfalls
  88 
  89   The number of classes, `K`, must not exceed:
  90   - the largest integer representable by `self.dtype`, i.e.,
  91     `2**(mantissa_bits+1)` (IEE754),
  92   - the maximum `Tensor` index, i.e., `2**31-1`.
  93 
  94   In other words,
  95 
  96   ```python
  97   K <= min(2**31-1, {
  98     tf.float16: 2**11,
  99     tf.float32: 2**24,
 100     tf.float64: 2**53 }[param.dtype])
 101   ```
 102 
 103   Note: This condition is validated only when `self.validate_args = True`.
 104 
 105   #### Examples
 106 
 107   Create a 3-class distribution, with the 3rd class is most likely to be drawn,
 108   using logits.
 109 
 110   ```python
 111   logits = [-50., -43, 0]
 112   dist = Multinomial(total_count=4., logits=logits)
 113   ```
 114 
 115   Create a 3-class distribution, with the 3rd class is most likely to be drawn.
 116 
 117   ```python
 118   p = [.2, .3, .5]
 119   dist = Multinomial(total_count=4., probs=p)
 120   ```
 121 
 122   The distribution functions can be evaluated on counts.
 123 
 124   ```python
 125   # counts same shape as p.
 126   counts = [1., 0, 3]
 127   dist.prob(counts)  # Shape []
 128 
 129   # p will be broadcast to [[.2, .3, .5], [.2, .3, .5]] to match counts.
 130   counts = [[1., 2, 1], [2, 2, 0]]
 131   dist.prob(counts)  # Shape [2]
 132 
 133   # p will be broadcast to shape [5, 7, 3] to match counts.
 134   counts = [[...]]  # Shape [5, 7, 3]
 135   dist.prob(counts)  # Shape [5, 7]
 136   ```
 137 
 138   Create a 2-batch of 3-class distributions.
 139 
 140   ```python
 141   p = [[.1, .2, .7], [.3, .3, .4]]  # Shape [2, 3]
 142   dist = Multinomial(total_count=[4., 5], probs=p)
 143 
 144   counts = [[2., 1, 1], [3, 1, 1]]
 145   dist.prob(counts)  # Shape [2]
 146 
 147   dist.sample(5) # Shape [5, 2, 3]
 148   ```
 149   """
 150 
 151   def __init__(self,
 152                total_count,
 153                logits=None,
 154                probs=None,
 155                validate_args=False,
 156                allow_nan_stats=True,
 157                name="Multinomial"):
 158     """Initialize a batch of Multinomial distributions.
 159 
 160     Args:
 161       total_count: Non-negative floating point tensor with shape broadcastable
 162         to `[N1,..., Nm]` with `m >= 0`. Defines this as a batch of
 163         `N1 x ... x Nm` different Multinomial distributions. Its components
 164         should be equal to integer values.
 165       logits: Floating point tensor representing unnormalized log-probabilities
 166         of a positive event with shape broadcastable to
 167         `[N1,..., Nm, K]` `m >= 0`, and the same dtype as `total_count`. Defines
 168         this as a batch of `N1 x ... x Nm` different `K` class Multinomial
 169         distributions. Only one of `logits` or `probs` should be passed in.
 170       probs: Positive floating point tensor with shape broadcastable to
 171         `[N1,..., Nm, K]` `m >= 0` and same dtype as `total_count`. Defines
 172         this as a batch of `N1 x ... x Nm` different `K` class Multinomial
 173         distributions. `probs`'s components in the last portion of its shape
 174         should sum to `1`. Only one of `logits` or `probs` should be passed in.
 175       validate_args: Python `bool`, default `False`. When `True` distribution
 176         parameters are checked for validity despite possibly degrading runtime
 177         performance. When `False` invalid inputs may silently render incorrect
 178         outputs.
 179       allow_nan_stats: Python `bool`, default `True`. When `True`, statistics
 180         (e.g., mean, mode, variance) use the value "`NaN`" to indicate the
 181         result is undefined. When `False`, an exception is raised if one or
 182         more of the statistic's batch members are undefined.
 183       name: Python `str` name prefixed to Ops created by this class.
 184     """
 185     parameters = dict(locals())
 186     with ops.name_scope(name, values=[total_count, logits, probs]) as name:
 187       self._total_count = ops.convert_to_tensor(total_count, name="total_count")
 188       if validate_args:
 189         self._total_count = (
 190             distribution_util.embed_check_nonnegative_integer_form(
 191                 self._total_count))
 192       self._logits, self._probs = distribution_util.get_logits_and_probs(
 193           logits=logits,
 194           probs=probs,
 195           multidimensional=True,
 196           validate_args=validate_args,
 197           name=name)
 198       self._mean_val = self._total_count[..., array_ops.newaxis] * self._probs
 199     super(Multinomial, self).__init__(
 200         dtype=self._probs.dtype,
 201         reparameterization_type=distribution.NOT_REPARAMETERIZED,
 202         validate_args=validate_args,
 203         allow_nan_stats=allow_nan_stats,
 204         parameters=parameters,
 205         graph_parents=[self._total_count,
 206                        self._logits,
 207                        self._probs],
 208         name=name)
 209 
 210   @property
 211   def total_count(self):
 212     """Number of trials used to construct a sample."""
 213     return self._total_count
 214 
 215   @property
 216   def logits(self):
 217     """Vector of coordinatewise logits."""
 218     return self._logits
 219 
 220   @property
 221   def probs(self):
 222     """Probability of drawing a `1` in that coordinate."""
 223     return self._probs
 224 
 225   def _batch_shape_tensor(self):
 226     return array_ops.shape(self._mean_val)[:-1]
 227 
 228   def _batch_shape(self):
 229     return self._mean_val.get_shape().with_rank_at_least(1)[:-1]
 230 
 231   def _event_shape_tensor(self):
 232     return array_ops.shape(self._mean_val)[-1:]
 233 
 234   def _event_shape(self):
 235     return self._mean_val.get_shape().with_rank_at_least(1)[-1:]
 236 
 237   def _sample_n(self, n, seed=None):
 238     n_draws = math_ops.cast(self.total_count, dtype=dtypes.int32)
 239     k = self.event_shape_tensor()[0]
 240 
 241     # broadcast the total_count and logits to same shape
 242     n_draws = array_ops.ones_like(
 243         self.logits[..., 0], dtype=n_draws.dtype) * n_draws
 244     logits = array_ops.ones_like(
 245         n_draws[..., array_ops.newaxis], dtype=self.logits.dtype) * self.logits
 246 
 247     # flatten the total_count and logits
 248     flat_logits = array_ops.reshape(logits, [-1, k])  # [B1B2...Bm, k]
 249     flat_ndraws = n * array_ops.reshape(n_draws, [-1])  # [B1B2...Bm]
 250 
 251     # computes each total_count and logits situation by map_fn
 252     def _sample_single(args):
 253       logits, n_draw = args[0], args[1]  # [K], []
 254       x = random_ops.multinomial(logits[array_ops.newaxis, ...], n_draw,
 255                                  seed)  # [1, n*n_draw]
 256       x = array_ops.reshape(x, shape=[n, -1])  # [n, n_draw]
 257       x = math_ops.reduce_sum(array_ops.one_hot(x, depth=k), axis=-2)  # [n, k]
 258       return x
 259 
 260     x = functional_ops.map_fn(
 261         _sample_single, [flat_logits, flat_ndraws],
 262         dtype=self.dtype)  # [B1B2...Bm, n, k]
 263 
 264     # reshape the results to proper shape
 265     x = array_ops.transpose(x, perm=[1, 0, 2])
 266     final_shape = array_ops.concat([[n], self.batch_shape_tensor(), [k]], 0)
 267     x = array_ops.reshape(x, final_shape)  # [n, B1, B2,..., Bm, k]
 268     return x

 270   @distribution_util.AppendDocstring(_multinomial_sample_note)
 271   def _log_prob(self, counts):
 272     return self._log_unnormalized_prob(counts) - self._log_normalization(counts)
 273 
 274   def _log_unnormalized_prob(self, counts):
 275     counts = self._maybe_assert_valid_sample(counts)
 276     return math_ops.reduce_sum(counts * nn_ops.log_softmax(self.logits), -1)
 277 
 278   def _log_normalization(self, counts):
 279     counts = self._maybe_assert_valid_sample(counts)
 280     return -distribution_util.log_combinations(self.total_count, counts)
 281 
 282   def _mean(self):
 283     return array_ops.identity(self._mean_val)
 284 
 285   def _covariance(self):
 286     p = self.probs * array_ops.ones_like(
 287         self.total_count)[..., array_ops.newaxis]
 288     return array_ops.matrix_set_diag(
 289         -math_ops.matmul(self._mean_val[..., array_ops.newaxis],
 290                          p[..., array_ops.newaxis, :]),  # outer product
 291         self._variance())
 292 
 293   def _variance(self):
 294     p = self.probs * array_ops.ones_like(
 295         self.total_count)[..., array_ops.newaxis]
 296     return self._mean_val - self._mean_val * p
 297 
 298   def _maybe_assert_valid_sample(self, counts):
 299     """Check counts for proper shape, values, then return tensor version."""
 300     if not self.validate_args:
 301       return counts
 302     counts = distribution_util.embed_check_nonnegative_integer_form(counts)
 303     return control_flow_ops.with_dependencies([
 304         check_ops.assert_equal(
 305             self.total_count, math_ops.reduce_sum(counts, -1),
 306             message="counts must sum to `self.total_count`"),
 307     ], counts)
