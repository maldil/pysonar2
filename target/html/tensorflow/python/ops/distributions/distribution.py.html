  77 def _copy_fn(fn):
  78   """Create a deep copy of fn.
  79 
  80   Args:
  81     fn: a callable
  82 
  83   Returns:
  84     A `FunctionType`: a deep copy of fn.
  85 
  86   Raises:
  87     TypeError: if `fn` is not a callable.
  88   """
  89   if not callable(fn):
  90     raise TypeError("fn is not callable: %s" % fn)
  91   # The blessed way to copy a function. copy.deepcopy fails to create a
  92   # non-reference copy. Since:
  93   #   types.FunctionType == type(lambda: None),
  94   # and the docstring for the function type states:
  95   #
  96   #   function(code, globals[, name[, argdefs[, closure]]])
  97   #
  98   #   Create a function object from a code object and a dictionary.
  99   #   ...
 100   #
 101   # Here we can use this to create a new function with the old function's
 102   # code, globals, closure, etc.
 103   return types.FunctionType(
 104       code=fn.__code__, globals=fn.__globals__,
 105       name=fn.__name__, argdefs=fn.__defaults__,
 106       closure=fn.__closure__)

 109 def _update_docstring(old_str, append_str):
 110   """Update old_str by inserting append_str just before the "Args:" section."""
 111   old_str = old_str or ""
 112   old_str_lines = old_str.split("\n")
 113 
 114   # Step 0: Prepend spaces to all lines of append_str. This is
 115   # necessary for correct markdown generation.
 116   append_str = "\n".join("    %s" % line for line in append_str.split("\n"))
 117 
 118   # Step 1: Find mention of "Args":
 119   has_args_ix = [
 120       ix for ix, line in enumerate(old_str_lines)
 121       if line.strip().lower() == "args:"]
 122   if has_args_ix:
 123     final_args_ix = has_args_ix[-1]
 124     return ("\n".join(old_str_lines[:final_args_ix])
 125             + "\n\n" + append_str + "\n\n"
 126             + "\n".join(old_str_lines[final_args_ix:]))
 127   else:
 128     return old_str + "\n\n" + append_str

 145   def __new__(mcs, classname, baseclasses, attrs):
 146     """Control the creation of subclasses of the Distribution class.
 147 
 148     The main purpose of this method is to properly propagate docstrings
 149     from private Distribution methods, like `_log_prob`, into their
 150     public wrappers as inherited by the Distribution base class
 151     (e.g. `log_prob`).
 152 
 153     Args:
 154       classname: The name of the subclass being created.
 155       baseclasses: A tuple of parent classes.
 156       attrs: A dict mapping new attributes to their values.
 157 
 158     Returns:
 159       The class object.
 160 
 161     Raises:
 162       TypeError: If `Distribution` is not a subclass of `BaseDistribution`, or
 163         the new class is derived via multiple inheritance and the first
 164         parent class is not a subclass of `BaseDistribution`.
 165       AttributeError:  If `Distribution` does not implement e.g. `log_prob`.
 166       ValueError:  If a `Distribution` public method lacks a docstring.
 167     """
 168     if not baseclasses:  # Nothing to be done for Distribution
 169       raise TypeError("Expected non-empty baseclass. Does Distribution "
 170                       "not subclass _BaseDistribution?")
 171     which_base = [
 172         base for base in baseclasses
 173         if base == _BaseDistribution or issubclass(base, Distribution)]
 174     base = which_base[0]
 175     if base == _BaseDistribution:  # Nothing to be done for Distribution
 176       return abc.ABCMeta.__new__(mcs, classname, baseclasses, attrs)
 177     if not issubclass(base, Distribution):
 178       raise TypeError("First parent class declared for %s must be "
 179                       "Distribution, but saw '%s'" % (classname, base.__name__))
 180     for attr in _DISTRIBUTION_PUBLIC_METHOD_WRAPPERS:
 181       special_attr = "_%s" % attr
 182       class_attr_value = attrs.get(attr, None)
 183       if attr in attrs:
 184         # The method is being overridden, do not update its docstring
 185         continue
 186       base_attr_value = getattr(base, attr, None)
 187       if not base_attr_value:
 188         raise AttributeError(
 189             "Internal error: expected base class '%s' to implement method '%s'"
 190             % (base.__name__, attr))
 191       class_special_attr_value = attrs.get(special_attr, None)
 192       if class_special_attr_value is None:
 193         # No _special method available, no need to update the docstring.
 194         continue
 195       class_special_attr_docstring = tf_inspect.getdoc(class_special_attr_value)
 196       if not class_special_attr_docstring:
 197         # No docstring to append.
 198         continue
 199       class_attr_value = _copy_fn(base_attr_value)
 200       class_attr_docstring = tf_inspect.getdoc(base_attr_value)
 201       if class_attr_docstring is None:
 202         raise ValueError(
 203             "Expected base class fn to contain a docstring: %s.%s"
 204             % (base.__name__, attr))
 205       class_attr_value.__doc__ = _update_docstring(
 206           class_attr_value.__doc__,
 207           ("Additional documentation from `%s`:\n\n%s"
 208            % (classname, class_special_attr_docstring)))
 209       attrs[attr] = class_attr_value
 210 
 211     return abc.ABCMeta.__new__(mcs, classname, baseclasses, attrs)

 214 @tf_export("distributions.ReparameterizationType")
 215 class ReparameterizationType(object):
 216   """Instances of this class represent how sampling is reparameterized.
 217 
 218   Two static instances exist in the distributions library, signifying
 219   one of two possible properties for samples from a distribution:
 220 
 221   `FULLY_REPARAMETERIZED`: Samples from the distribution are fully
 222     reparameterized, and straight-through gradients are supported.
 223 
 224   `NOT_REPARAMETERIZED`: Samples from the distribution are not fully
 225     reparameterized, and straight-through gradients are either partially
 226     unsupported or are not supported at all. In this case, for purposes of
 227     e.g. RL or variational inference, it is generally safest to wrap the
 228     sample results in a `stop_gradients` call and use policy
 229     gradients / surrogate loss instead.
 230   """
 231 
 232   def __init__(self, rep_type):
 233     self._rep_type = rep_type
 234 
 235   def __repr__(self):
 236     return "<Reparameteriation Type: %s>" % self._rep_type
 237 
 238   def __eq__(self, other):
 239     """Determine if this `ReparameterizationType` is equal to another.
 240 
 241     Since RepaparameterizationType instances are constant static global
 242     instances, equality checks if two instances' id() values are equal.
 243 
 244     Args:
 245       other: Object to compare against.
 246 
 247     Returns:
 248       `self is other`.
 249     """
 250     return self is other

 269 @six.add_metaclass(_DistributionMeta)
 270 @tf_export("distributions.Distribution")
 271 class Distribution(_BaseDistribution):
 272   """A generic probability distribution base class.
 273 
 274   `Distribution` is a base class for constructing and organizing properties
 275   (e.g., mean, variance) of random variables (e.g, Bernoulli, Gaussian).
 276 
 277   #### Subclassing
 278 
 279   Subclasses are expected to implement a leading-underscore version of the
 280   same-named function. The argument signature should be identical except for
 281   the omission of `name="..."`. For example, to enable `log_prob(value,
 282   name="log_prob")` a subclass should implement `_log_prob(value)`.
 283 
 284   Subclasses can append to public-level docstrings by providing
 285   docstrings for their method specializations. For example:
 286 
 287   ```python
 288   @util.AppendDocstring("Some other details.")
 289   def _log_prob(self, value):
 290     ...
 291   ```
 292 
 293   would add the string "Some other details." to the `log_prob` function
 294   docstring. This is implemented as a simple decorator to avoid python
 295   linter complaining about missing Args/Returns/Raises sections in the
 296   partial docstrings.
 297 
 298   #### Broadcasting, batching, and shapes
 299 
 300   All distributions support batches of independent distributions of that type.
 301   The batch shape is determined by broadcasting together the parameters.
 302 
 303   The shape of arguments to `__init__`, `cdf`, `log_cdf`, `prob`, and
 304   `log_prob` reflect this broadcasting, as does the return value of `sample` and
 305   `sample_n`.
 306 
 307   `sample_n_shape = [n] + batch_shape + event_shape`, where `sample_n_shape` is
 308   the shape of the `Tensor` returned from `sample_n`, `n` is the number of
 309   samples, `batch_shape` defines how many independent distributions there are,
 310   and `event_shape` defines the shape of samples from each of those independent
 311   distributions. Samples are independent along the `batch_shape` dimensions, but
 312   not necessarily so along the `event_shape` dimensions (depending on the
 313   particulars of the underlying distribution).
 314 
 315   Using the `Uniform` distribution as an example:
 316 
 317   ```python
 318   minval = 3.0
 319   maxval = [[4.0, 6.0],
 320             [10.0, 12.0]]
 321 
 322   # Broadcasting:
 323   # This instance represents 4 Uniform distributions. Each has a lower bound at
 324   # 3.0 as the `minval` parameter was broadcasted to match `maxval`'s shape.
 325   u = Uniform(minval, maxval)
 326 
 327   # `event_shape` is `TensorShape([])`.
 328   event_shape = u.event_shape
 329   # `event_shape_t` is a `Tensor` which will evaluate to [].
 330   event_shape_t = u.event_shape_tensor()
 331 
 332   # Sampling returns a sample per distribution. `samples` has shape
 333   # [5, 2, 2], which is [n] + batch_shape + event_shape, where n=5,
 334   # batch_shape=[2, 2], and event_shape=[].
 335   samples = u.sample_n(5)
 336 
 337   # The broadcasting holds across methods. Here we use `cdf` as an example. The
 338   # same holds for `log_cdf` and the likelihood functions.
 339 
 340   # `cum_prob` has shape [2, 2] as the `value` argument was broadcasted to the
 341   # shape of the `Uniform` instance.
 342   cum_prob_broadcast = u.cdf(4.0)
 343 
 344   # `cum_prob`'s shape is [2, 2], one per distribution. No broadcasting
 345   # occurred.
 346   cum_prob_per_dist = u.cdf([[4.0, 5.0],
 347                              [6.0, 7.0]])
 348 
 349   # INVALID as the `value` argument is not broadcastable to the distribution's
 350   # shape.
 351   cum_prob_invalid = u.cdf([4.0, 5.0, 6.0])
 352   ```
 353 
 354   #### Shapes
 355 
 356   There are three important concepts associated with TensorFlow Distributions
 357   shapes:
 358   - Event shape describes the shape of a single draw from the distribution;
 359     it may be dependent across dimensions. For scalar distributions, the event
 360     shape is `[]`. For a 5-dimensional MultivariateNormal, the event shape is
 361     `[5]`.
 362   - Batch shape describes independent, not identically distributed draws, aka a
 363     "collection" or "bunch" of distributions.
 364   - Sample shape describes independent, identically distributed draws of batches
 365     from the distribution family.
 366 
 367   The event shape and the batch shape are properties of a Distribution object,
 368   whereas the sample shape is associated with a specific call to `sample` or
 369   `log_prob`.
 370 
 371   For detailed usage examples of TensorFlow Distributions shapes, see
 372   [this tutorial](
 373   https://github.com/tensorflow/probability/blob/master/tensorflow_probability/examples/jupyter_notebooks/Understanding_TensorFlow_Distributions_Shapes.ipynb)
 374 
 375   #### Parameter values leading to undefined statistics or distributions.
 376 
 377   Some distributions do not have well-defined statistics for all initialization
 378   parameter values. For example, the beta distribution is parameterized by
 379   positive real numbers `concentration1` and `concentration0`, and does not have
 380   well-defined mode if `concentration1 < 1` or `concentration0 < 1`.
 381 
 382   The user is given the option of raising an exception or returning `NaN`.
 383 
 384   ```python
 385   a = tf.exp(tf.matmul(logits, weights_a))
 386   b = tf.exp(tf.matmul(logits, weights_b))
 387 
 388   # Will raise exception if ANY batch member has a < 1 or b < 1.
 389   dist = distributions.beta(a, b, allow_nan_stats=False)
 390   mode = dist.mode().eval()
 391 
 392   # Will return NaN for batch members with either a < 1 or b < 1.
 393   dist = distributions.beta(a, b, allow_nan_stats=True)  # Default behavior
 394   mode = dist.mode().eval()
 395   ```
 396 
 397   In all cases, an exception is raised if *invalid* parameters are passed, e.g.
 398 
 399   ```python
 400   # Will raise an exception if any Op is run.
 401   negative_a = -1.0 * a  # beta distribution by definition has a > 0.
 402   dist = distributions.beta(negative_a, b, allow_nan_stats=True)
 403   dist.mean().eval()
 404   ```
 405 
 406   """
 407 
 408   def __init__(self,
 409                dtype,
 410                reparameterization_type,
 411                validate_args,
 412                allow_nan_stats,
 413                parameters=None,
 414                graph_parents=None,
 415                name=None):
 416     """Constructs the `Distribution`.
 417 
 418     **This is a private method for subclass use.**
 419 
 420     Args:
 421       dtype: The type of the event samples. `None` implies no type-enforcement.
 422       reparameterization_type: Instance of `ReparameterizationType`.
 423         If `distributions.FULLY_REPARAMETERIZED`, this
 424         `Distribution` can be reparameterized in terms of some standard
 425         distribution with a function whose Jacobian is constant for the support
 426         of the standard distribution. If `distributions.NOT_REPARAMETERIZED`,
 427         then no such reparameterization is available.
 428       validate_args: Python `bool`, default `False`. When `True` distribution
 429         parameters are checked for validity despite possibly degrading runtime
 430         performance. When `False` invalid inputs may silently render incorrect
 431         outputs.
 432       allow_nan_stats: Python `bool`, default `True`. When `True`, statistics
 433         (e.g., mean, mode, variance) use the value "`NaN`" to indicate the
 434         result is undefined. When `False`, an exception is raised if one or
 435         more of the statistic's batch members are undefined.
 436       parameters: Python `dict` of parameters used to instantiate this
 437         `Distribution`.
 438       graph_parents: Python `list` of graph prerequisites of this
 439         `Distribution`.
 440       name: Python `str` name prefixed to Ops created by this class. Default:
 441         subclass name.
 442 
 443     Raises:
 444       ValueError: if any member of graph_parents is `None` or not a `Tensor`.
 445     """
 446     graph_parents = [] if graph_parents is None else graph_parents
 447     for i, t in enumerate(graph_parents):
 448       if t is None or not tensor_util.is_tensor(t):
 449         raise ValueError("Graph parent item %d is not a Tensor; %s." % (i, t))
 450     if not name or name[-1] != "/":  # `name` is not a name scope
 451       non_unique_name = name or type(self).__name__
 452       with ops.name_scope(non_unique_name) as name:
 453         pass
 454     self._dtype = dtype
 455     self._reparameterization_type = reparameterization_type
 456     self._allow_nan_stats = allow_nan_stats
 457     self._validate_args = validate_args
 458     self._parameters = parameters or {}
 459     self._graph_parents = graph_parents
 460     self._name = name
 461

 467   def _parameters(self, value):

 470     Parameters are often created using locals(), so we need to clean out any

 473     Args:
 474       value: A dictionary of parameters to assign to the `_parameters` property.

 480   @classmethod
 481   def param_shapes(cls, sample_shape, name="DistributionParamShapes"):
 482     """Shapes of parameters given the desired shape of a call to `sample()`.
 483 
 484     This is a class method that describes what key/value arguments are required
 485     to instantiate the given `Distribution` so that a particular shape is
 486     returned for that instance's call to `sample()`.
 487 
 488     Subclasses should override class method `_param_shapes`.
 489 
 490     Args:
 491       sample_shape: `Tensor` or python list/tuple. Desired shape of a call to
 492         `sample()`.
 493       name: name to prepend ops with.
 494 
 495     Returns:
 496       `dict` of parameter name to `Tensor` shapes.
 497     """
 498     with ops.name_scope(name, values=[sample_shape]):
 499       return cls._param_shapes(sample_shape)

 501   @classmethod
 502   def param_static_shapes(cls, sample_shape):
 503     """param_shapes with static (i.e. `TensorShape`) shapes.
 504 
 505     This is a class method that describes what key/value arguments are required
 506     to instantiate the given `Distribution` so that a particular shape is
 507     returned for that instance's call to `sample()`. Assumes that the sample's
 508     shape is known statically.
 509 
 510     Subclasses should override class method `_param_shapes` to return
 511     constant-valued tensors when constant values are fed.
 512 
 513     Args:
 514       sample_shape: `TensorShape` or python list/tuple. Desired shape of a call
 515         to `sample()`.
 516 
 517     Returns:
 518       `dict` of parameter name to `TensorShape`.
 519 
 520     Raises:
 521       ValueError: if `sample_shape` is a `TensorShape` and is not fully defined.
 522     """
 523     if isinstance(sample_shape, tensor_shape.TensorShape):
 524       if not sample_shape.is_fully_defined():
 525         raise ValueError("TensorShape sample_shape must be fully defined")
 526       sample_shape = sample_shape.as_list()
 527 
 528     params = cls.param_shapes(sample_shape)
 529 
 530     static_params = {}
 531     for name, shape in params.items():
 532       static_shape = tensor_util.constant_value(shape)
 533       if static_shape is None:
 534         raise ValueError(
 535             "sample_shape must be a fully-defined TensorShape or list/tuple")
 536       static_params[name] = tensor_shape.TensorShape(static_shape)
 537 
 538     return static_params

1129   def cross_entropy(self, other, name="cross_entropy"):
1130     """Computes the (Shannon) cross entropy.
1131 
1132     Denote this distribution (`self`) by `P` and the `other` distribution by
1133     `Q`. Assuming `P, Q` are absolutely continuous with respect to
1134     one another and permit densities `p(x) dr(x)` and `q(x) dr(x)`, (Shanon)
1135     cross entropy is defined as:

1138     H[P, Q] = E_p[-log q(X)] = -int_F p(x) log q(x) dr(x)
1139     ```
1140 
1141     where `F` denotes the support of the random variable `X ~ P`.
1142 
1143     Args:

1145       name: Python `str` prepended to names of ops created by this function.
1146 
1147     Returns:

1149         representing `n` different calculations of (Shanon) cross entropy.
1150     """

1152       return self._cross_entropy(other)

1160 
1161     Denote this distribution (`self`) by `p` and the `other` distribution by
1162     `q`. Assuming `p, q` are absolutely continuous with respect to reference
1163     measure `r`, the KL divergence is defined as:
1164 
1165     ```none
1166     KL[p, q] = E_p[log(p(X)/q(X))]
1167              = -int_F p(x) log q(x) dr(x) + int_F p(x) log p(x) dr(x)
1168              = H[p, q] - H[p]
1169     ```
1170 
1171     where `F` denotes the support of the random variable `X ~ p`, `H[., .]`
1172     denotes (Shanon) cross entropy, and `H[.]` denotes (Shanon) entropy.
1173 
1174     Args:
1175       other: `tfp.distributions.Distribution` instance.
1176       name: Python `str` prepended to names of ops created by this function.
1177 
1178     Returns:
1179       kl_divergence: `self.dtype` `Tensor` with shape `[B1, ..., Bn]`

1224     x_static_val = tensor_util.constant_value(x)
1225     if x_static_val is None:
1226       prod = math_ops.reduce_prod(x)
1227     else:
1228       prod = np.prod(x_static_val, dtype=x.dtype.as_numpy_dtype())
1229 
1230     ndims = x.get_shape().ndims  # != sample_ndims
1231     if ndims is None:
1232       # Maybe expand_dims.
1233       ndims = array_ops.rank(x)
1234       expanded_shape = util.pick_vector(
1235           math_ops.equal(ndims, 0),
1236           np.array([1], dtype=np.int32), array_ops.shape(x))
1237       x = array_ops.reshape(x, expanded_shape)
1238     elif ndims == 0:
1239       # Definitely expand_dims.
1240       if x_static_val is not None:
1241         x = ops.convert_to_tensor(
1242             np.array([x_static_val], dtype=x.dtype.as_numpy_dtype()),
1243             name=name)
1244       else:
1245         x = array_ops.reshape(x, [1])
1246     elif ndims != 1:
1247       raise ValueError("Input is neither scalar nor vector.")
1248 
1249     return x, prod

1251   def _set_sample_static_shape(self, x, sample_shape):
1252     """Helper to `sample`; sets static shape info."""
1253     # Set shape hints.
1254     sample_shape = tensor_shape.TensorShape(
1255         tensor_util.constant_value(sample_shape))
1256 
1257     ndims = x.get_shape().ndims
1258     sample_ndims = sample_shape.ndims
1259     batch_ndims = self.batch_shape.ndims
1260     event_ndims = self.event_shape.ndims
1261 
1262     # Infer rank(x).
1263     if (ndims is None and
1264         sample_ndims is not None and
1265         batch_ndims is not None and
1266         event_ndims is not None):
1267       ndims = sample_ndims + batch_ndims + event_ndims
1268       x.set_shape([None] * ndims)
1269 
1270     # Infer sample shape.
1271     if ndims is not None and sample_ndims is not None:
1272       shape = sample_shape.concatenate([None]*(ndims - sample_ndims))
1273       x.set_shape(x.get_shape().merge_with(shape))
1274 
1275     # Infer event shape.
1276     if ndims is not None and event_ndims is not None:
1277       shape = tensor_shape.TensorShape(
1278           [None]*(ndims - event_ndims)).concatenate(self.event_shape)
1279       x.set_shape(x.get_shape().merge_with(shape))
1280 
1281     # Infer batch shape.
1282     if batch_ndims is not None:
1283       if ndims is not None:
1284         if sample_ndims is None and event_ndims is not None:
1285           sample_ndims = ndims - batch_ndims - event_ndims
1286         elif event_ndims is None and sample_ndims is not None:
1287           event_ndims = ndims - batch_ndims - sample_ndims
1288       if sample_ndims is not None and event_ndims is not None:
1289         shape = tensor_shape.TensorShape([None]*sample_ndims).concatenate(
1290             self.batch_shape).concatenate([None]*event_ndims)
1291         x.set_shape(x.get_shape().merge_with(shape))
1292 
1293     return x

1295   def _is_scalar_helper(self, static_shape, dynamic_shape_fn):
1296     """Implementation for `is_scalar_batch` and `is_scalar_event`."""
1297     if static_shape.ndims is not None:
1298       return static_shape.ndims == 0
1299     shape = dynamic_shape_fn()
1300     if (shape.get_shape().ndims is not None and
1301         shape.get_shape()[0].value is not None):
1302       # If the static_shape is correctly written then we should never execute
1303       # this branch. We keep it just in case there's some unimagined corner
1304       # case.
1305       return shape.get_shape().as_list() == [0]
1306     return math_ops.equal(array_ops.shape(shape)[0], 0)
