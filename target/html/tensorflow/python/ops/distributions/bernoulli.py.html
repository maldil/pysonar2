  34 @tf_export("distributions.Bernoulli")
  35 class Bernoulli(distribution.Distribution):
  36   """Bernoulli distribution.
  37 
  38   The Bernoulli distribution with `probs` parameter, i.e., the probability of a
  39   `1` outcome (vs a `0` outcome).
  40   """
  41 
  42   def __init__(self,
  43                logits=None,
  44                probs=None,
  45                dtype=dtypes.int32,
  46                validate_args=False,
  47                allow_nan_stats=True,
  48                name="Bernoulli"):
  49     """Construct Bernoulli distributions.
  50 
  51     Args:
  52       logits: An N-D `Tensor` representing the log-odds of a `1` event. Each
  53         entry in the `Tensor` parametrizes an independent Bernoulli distribution
  54         where the probability of an event is sigmoid(logits). Only one of
  55         `logits` or `probs` should be passed in.
  56       probs: An N-D `Tensor` representing the probability of a `1`
  57         event. Each entry in the `Tensor` parameterizes an independent
  58         Bernoulli distribution. Only one of `logits` or `probs` should be passed
  59         in.
  60       dtype: The type of the event samples. Default: `int32`.
  61       validate_args: Python `bool`, default `False`. When `True` distribution
  62         parameters are checked for validity despite possibly degrading runtime
  63         performance. When `False` invalid inputs may silently render incorrect
  64         outputs.
  65       allow_nan_stats: Python `bool`, default `True`. When `True`,
  66         statistics (e.g., mean, mode, variance) use the value "`NaN`" to
  67         indicate the result is undefined. When `False`, an exception is raised
  68         if one or more of the statistic's batch members are undefined.
  69       name: Python `str` name prefixed to Ops created by this class.
  70 
  71     Raises:
  72       ValueError: If p and logits are passed, or if neither are passed.
  73     """
  74     parameters = dict(locals())
  75     with ops.name_scope(name) as name:
  76       self._logits, self._probs = distribution_util.get_logits_and_probs(
  77           logits=logits,
  78           probs=probs,
  79           validate_args=validate_args,
  80           name=name)
  81     super(Bernoulli, self).__init__(
  82         dtype=dtype,
  83         reparameterization_type=distribution.NOT_REPARAMETERIZED,
  84         validate_args=validate_args,
  85         allow_nan_stats=allow_nan_stats,
  86         parameters=parameters,
  87         graph_parents=[self._logits, self._probs],
  88         name=name)
  89 
  90   @staticmethod
  91   def _param_shapes(sample_shape):
  92     return {"logits": ops.convert_to_tensor(sample_shape, dtype=dtypes.int32)}
  93 
  94   @property
  95   def logits(self):
  96     """Log-odds of a `1` outcome (vs `0`)."""
  97     return self._logits
  98 
  99   @property
 100   def probs(self):
 101     """Probability of a `1` outcome (vs `0`)."""
 102     return self._probs
 103 
 104   def _batch_shape_tensor(self):
 105     return array_ops.shape(self._logits)
 106 
 107   def _batch_shape(self):
 108     return self._logits.get_shape()
 109 
 110   def _event_shape_tensor(self):
 111     return array_ops.constant([], dtype=dtypes.int32)
 112 
 113   def _event_shape(self):
 114     return tensor_shape.scalar()
 115 
 116   def _sample_n(self, n, seed=None):
 117     new_shape = array_ops.concat([[n], self.batch_shape_tensor()], 0)
 118     uniform = random_ops.random_uniform(
 119         new_shape, seed=seed, dtype=self.probs.dtype)
 120     sample = math_ops.less(uniform, self.probs)
 121     return math_ops.cast(sample, self.dtype)
 122 
 123   def _log_prob(self, event):
 124     if self.validate_args:
 125       event = distribution_util.embed_check_integer_casting_closed(
 126           event, target_dtype=dtypes.bool)
 127 
 128     # TODO(jaana): The current sigmoid_cross_entropy_with_logits has
 129     # inconsistent behavior for logits = inf/-inf.
 130     event = math_ops.cast(event, self.logits.dtype)
 131     logits = self.logits
 132     # sigmoid_cross_entropy_with_logits doesn't broadcast shape,
 133     # so we do this here.
 134 
 135     def _broadcast(logits, event):
 136       return (array_ops.ones_like(event) * logits,
 137               array_ops.ones_like(logits) * event)
 138 
 139     if not (event.get_shape().is_fully_defined() and
 140             logits.get_shape().is_fully_defined() and
 141             event.get_shape() == logits.get_shape()):
 142       logits, event = _broadcast(logits, event)
 143     return -nn.sigmoid_cross_entropy_with_logits(labels=event, logits=logits)
 144 
 145   def _entropy(self):
 146     return (-self.logits * (math_ops.sigmoid(self.logits) - 1) +
 147             nn.softplus(-self.logits))
 148 
 149   def _mean(self):
 150     return array_ops.identity(self.probs)
 151 
 152   def _variance(self):
 153     return self._mean() * (1. - self.probs)
 154 
 155   def _mode(self):
 156     """Returns `1` if `prob > 0.5` and `0` otherwise."""
 157     return math_ops.cast(self.probs > 0.5, self.dtype)

 160 @kullback_leibler.RegisterKL(Bernoulli, Bernoulli)
 161 def _kl_bernoulli_bernoulli(a, b, name=None):
 162   """Calculate the batched KL divergence KL(a || b) with a and b Bernoulli.
 163 
 164   Args:
 165     a: instance of a Bernoulli distribution object.
 166     b: instance of a Bernoulli distribution object.
 167     name: (optional) Name to use for created operations.
 168       default is "kl_bernoulli_bernoulli".
 169 
 170   Returns:
 171     Batchwise KL(a || b)
 172   """
 173   with ops.name_scope(name, "kl_bernoulli_bernoulli",
 174                       values=[a.logits, b.logits]):
 175     delta_probs0 = nn.softplus(-b.logits) - nn.softplus(-a.logits)
 176     delta_probs1 = nn.softplus(b.logits) - nn.softplus(a.logits)
 177     return (math_ops.sigmoid(a.logits) * delta_probs0
 178             + math_ops.sigmoid(-a.logits) * delta_probs1)
