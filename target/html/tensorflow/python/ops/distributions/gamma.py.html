  45 @tf_export("distributions.Gamma")
  46 class Gamma(distribution.Distribution):
  47   """Gamma distribution.
  48 
  49   The Gamma distribution is defined over positive real numbers using
  50   parameters `concentration` (aka "alpha") and `rate` (aka "beta").
  51 
  52   #### Mathematical Details
  53 
  54   The probability density function (pdf) is,
  55 
  56   ```none
  57   pdf(x; alpha, beta, x > 0) = x**(alpha - 1) exp(-x beta) / Z
  58   Z = Gamma(alpha) beta**(-alpha)
  59   ```
  60 
  61   where:
  62 
  63   * `concentration = alpha`, `alpha > 0`,
  64   * `rate = beta`, `beta > 0`,
  65   * `Z` is the normalizing constant, and,
  66   * `Gamma` is the [gamma function](
  67     https://en.wikipedia.org/wiki/Gamma_function).
  68 
  69   The cumulative density function (cdf) is,
  70 
  71   ```none
  72   cdf(x; alpha, beta, x > 0) = GammaInc(alpha, beta x) / Gamma(alpha)
  73   ```
  74 
  75   where `GammaInc` is the [lower incomplete Gamma function](
  76   https://en.wikipedia.org/wiki/Incomplete_gamma_function).
  77 
  78   The parameters can be intuited via their relationship to mean and stddev,
  79 
  80   ```none
  81   concentration = alpha = (mean / stddev)**2
  82   rate = beta = mean / stddev**2 = concentration / mean
  83   ```
  84 
  85   Distribution parameters are automatically broadcast in all functions; see
  86   examples for details.
  87 
  88   Warning: The samples of this distribution are always non-negative. However,
  89   the samples that are smaller than `np.finfo(dtype).tiny` are rounded
  90   to this value, so it appears more often than it should.
  91   This should only be noticeable when the `concentration` is very small, or the
  92   `rate` is very large. See note in `tf.random_gamma` docstring.
  93 
  94   Samples of this distribution are reparameterized (pathwise differentiable).
  95   The derivatives are computed using the approach described in the paper
  96 
  97   [Michael Figurnov, Shakir Mohamed, Andriy Mnih.
  98   Implicit Reparameterization Gradients, 2018](https://arxiv.org/abs/1805.08498)
  99 
 100   #### Examples
 101 
 102   ```python
 103   import tensorflow_probability as tfp
 104   tfd = tfp.distributions
 105 
 106   dist = tfd.Gamma(concentration=3.0, rate=2.0)
 107   dist2 = tfd.Gamma(concentration=[3.0, 4.0], rate=[2.0, 3.0])
 108   ```
 109 
 110   Compute the gradients of samples w.r.t. the parameters:
 111 
 112   ```python
 113   concentration = tf.constant(3.0)
 114   rate = tf.constant(2.0)
 115   dist = tfd.Gamma(concentration, rate)
 116   samples = dist.sample(5)  # Shape [5]
 117   loss = tf.reduce_mean(tf.square(samples))  # Arbitrary loss function
 118   # Unbiased stochastic gradients of the loss function
 119   grads = tf.gradients(loss, [concentration, rate])
 120   ```
 121 
 122   """
 123 
 124   def __init__(self,
 125                concentration,
 126                rate,
 127                validate_args=False,
 128                allow_nan_stats=True,
 129                name="Gamma"):
 130     """Construct Gamma with `concentration` and `rate` parameters.
 131 
 132     The parameters `concentration` and `rate` must be shaped in a way that
 133     supports broadcasting (e.g. `concentration + rate` is a valid operation).
 134 
 135     Args:
 136       concentration: Floating point tensor, the concentration params of the
 137         distribution(s). Must contain only positive values.
 138       rate: Floating point tensor, the inverse scale params of the
 139         distribution(s). Must contain only positive values.
 140       validate_args: Python `bool`, default `False`. When `True` distribution
 141         parameters are checked for validity despite possibly degrading runtime
 142         performance. When `False` invalid inputs may silently render incorrect
 143         outputs.
 144       allow_nan_stats: Python `bool`, default `True`. When `True`, statistics
 145         (e.g., mean, mode, variance) use the value "`NaN`" to indicate the
 146         result is undefined. When `False`, an exception is raised if one or
 147         more of the statistic's batch members are undefined.
 148       name: Python `str` name prefixed to Ops created by this class.
 149 
 150     Raises:
 151       TypeError: if `concentration` and `rate` are different dtypes.
 152     """
 153     parameters = dict(locals())
 154     with ops.name_scope(name, values=[concentration, rate]) as name:
 155       with ops.control_dependencies([
 156           check_ops.assert_positive(concentration),
 157           check_ops.assert_positive(rate),
 158       ] if validate_args else []):
 159         self._concentration = array_ops.identity(
 160             concentration, name="concentration")
 161         self._rate = array_ops.identity(rate, name="rate")
 162         check_ops.assert_same_float_dtype(
 163             [self._concentration, self._rate])
 164     super(Gamma, self).__init__(
 165         dtype=self._concentration.dtype,
 166         validate_args=validate_args,
 167         allow_nan_stats=allow_nan_stats,
 168         reparameterization_type=distribution.FULLY_REPARAMETERIZED,
 169         parameters=parameters,
 170         graph_parents=[self._concentration,
 171                        self._rate],
 172         name=name)
 173 
 174   @staticmethod
 175   def _param_shapes(sample_shape):
 176     return dict(
 177         zip(("concentration", "rate"), ([ops.convert_to_tensor(
 178             sample_shape, dtype=dtypes.int32)] * 2)))
 179 
 180   @property
 181   def concentration(self):
 182     """Concentration parameter."""
 183     return self._concentration
 184 
 185   @property
 186   def rate(self):
 187     """Rate parameter."""
 188     return self._rate
 189 
 190   def _batch_shape_tensor(self):
 191     return array_ops.broadcast_dynamic_shape(
 192         array_ops.shape(self.concentration),
 193         array_ops.shape(self.rate))
 194 
 195   def _batch_shape(self):
 196     return array_ops.broadcast_static_shape(
 197         self.concentration.get_shape(),
 198         self.rate.get_shape())
 199 
 200   def _event_shape_tensor(self):
 201     return constant_op.constant([], dtype=dtypes.int32)
 202 
 203   def _event_shape(self):
 204     return tensor_shape.scalar()
 205 
 206   @distribution_util.AppendDocstring(
 207       """Note: See `tf.random_gamma` docstring for sampling details and
 208       caveats.""")
 209   def _sample_n(self, n, seed=None):
 210     return random_ops.random_gamma(
 211         shape=[n],
 212         alpha=self.concentration,
 213         beta=self.rate,
 214         dtype=self.dtype,
 215         seed=seed)
 216 
 217   def _log_prob(self, x):
 218     return self._log_unnormalized_prob(x) - self._log_normalization()
 219 
 220   def _cdf(self, x):
 221     x = self._maybe_assert_valid_sample(x)
 222     # Note that igamma returns the regularized incomplete gamma function,
 223     # which is what we want for the CDF.
 224     return math_ops.igamma(self.concentration, self.rate * x)
 225 
 226   def _log_unnormalized_prob(self, x):
 227     x = self._maybe_assert_valid_sample(x)
 228     return (self.concentration - 1.) * math_ops.log(x) - self.rate * x
 229 
 230   def _log_normalization(self):
 231     return (math_ops.lgamma(self.concentration)
 232             - self.concentration * math_ops.log(self.rate))
 233 
 234   def _entropy(self):
 235     return (self.concentration
 236             - math_ops.log(self.rate)
 237             + math_ops.lgamma(self.concentration)
 238             + ((1. - self.concentration) *
 239                math_ops.digamma(self.concentration)))
 240 
 241   def _mean(self):
 242     return self.concentration / self.rate
 243 
 244   def _variance(self):
 245     return self.concentration / math_ops.square(self.rate)
 246 
 247   def _stddev(self):
 248     return math_ops.sqrt(self.concentration) / self.rate
 249 
 250   @distribution_util.AppendDocstring(
 251       """The mode of a gamma distribution is `(shape - 1) / rate` when
 252       `shape > 1`, and `NaN` otherwise. If `self.allow_nan_stats` is `False`,
 253       an exception will be raised rather than returning `NaN`.""")
 254   def _mode(self):
 255     mode = (self.concentration - 1.) / self.rate
 256     if self.allow_nan_stats:
 257       nan = array_ops.fill(
 258           self.batch_shape_tensor(),
 259           np.array(np.nan, dtype=self.dtype.as_numpy_dtype()),
 260           name="nan")
 261       return array_ops.where(self.concentration > 1., mode, nan)
 262     else:
 263       return control_flow_ops.with_dependencies([
 264           check_ops.assert_less(
 265               array_ops.ones([], self.dtype),
 266               self.concentration,
 267               message="mode not defined when any concentration <= 1"),
 268           ], mode)
 269 
 270   def _maybe_assert_valid_sample(self, x):
 271     check_ops.assert_same_float_dtype(tensors=[x], dtype=self.dtype)
 272     if not self.validate_args:
 273       return x
 274     return control_flow_ops.with_dependencies([
 275         check_ops.assert_positive(x),
 276     ], x)

 279 class GammaWithSoftplusConcentrationRate(Gamma):
 280   """`Gamma` with softplus of `concentration` and `rate`."""
 281 
 282   def __init__(self,
 283                concentration,
 284                rate,
 285                validate_args=False,
 286                allow_nan_stats=True,
 287                name="GammaWithSoftplusConcentrationRate"):
 288     parameters = dict(locals())
 289     with ops.name_scope(name, values=[concentration, rate]) as name:
 290       super(GammaWithSoftplusConcentrationRate, self).__init__(
 291           concentration=nn.softplus(concentration,
 292                                     name="softplus_concentration"),
 293           rate=nn.softplus(rate, name="softplus_rate"),
 294           validate_args=validate_args,
 295           allow_nan_stats=allow_nan_stats,
 296           name=name)
 297     self._parameters = parameters

 300 @kullback_leibler.RegisterKL(Gamma, Gamma)
 301 def _kl_gamma_gamma(g0, g1, name=None):
 302   """Calculate the batched KL divergence KL(g0 || g1) with g0 and g1 Gamma.
 303 
 304   Args:
 305     g0: instance of a Gamma distribution object.
 306     g1: instance of a Gamma distribution object.
 307     name: (optional) Name to use for created operations.
 308       Default is "kl_gamma_gamma".
 309 
 310   Returns:
 311     kl_gamma_gamma: `Tensor`. The batchwise KL(g0 || g1).
 312   """
 313   with ops.name_scope(name, "kl_gamma_gamma", values=[
 314       g0.concentration, g0.rate, g1.concentration, g1.rate]):
 315     # Result from:
 316     #   http://www.fil.ion.ucl.ac.uk/~wpenny/publications/densities.ps
 317     # For derivation see:
 318     #   http://stats.stackexchange.com/questions/11646/kullback-leibler-divergence-between-two-gamma-distributions   pylint: disable=line-too-long
 319     return (((g0.concentration - g1.concentration)
 320              * math_ops.digamma(g0.concentration))
 321             + math_ops.lgamma(g1.concentration)
 322             - math_ops.lgamma(g0.concentration)
 323             + g1.concentration * math_ops.log(g0.rate)
 324             - g1.concentration * math_ops.log(g1.rate)
 325             + g0.concentration * (g1.rate / g0.rate - 1.))
