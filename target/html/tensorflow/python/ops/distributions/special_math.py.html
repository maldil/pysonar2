  93 # log_ndtr uses different functions over the ranges
  94 # (-infty, lower](lower, upper](upper, infty)
  95 # Lower bound values were chosen by examining where the support of ndtr
  96 # appears to be zero, relative to scipy's (which is always 64bit). They were
  97 # then made more conservative just to be safe. (Conservative means use the
  98 # expansion more than we probably need to.) See `NdtrTest` in
  99 # special_math_test.py.
 100 LOGNDTR_FLOAT64_LOWER = np.array(-20, np.float64)
 101 LOGNDTR_FLOAT32_LOWER = np.array(-10, np.float32)
 102 
 103 # Upper bound values were chosen by examining for which values of 'x'
 104 # Log[cdf(x)] is 0, after which point we need to use the approximation
 105 # Log[cdf(x)] = Log[1 - cdf(-x)] approx -cdf(-x). We chose a value slightly
 106 # conservative, meaning we use the approximation earlier than needed.
 107 LOGNDTR_FLOAT64_UPPER = np.array(8, np.float64)
 108 LOGNDTR_FLOAT32_UPPER = np.array(5, np.float32)

 111 def ndtr(x, name="ndtr"):
 112   """Normal distribution function.
 113 
 114   Returns the area under the Gaussian probability density function, integrated
 115   from minus infinity to x:
 116 
 117   ```
 118                     1       / x
 119      ndtr(x)  = ----------  |    exp(-0.5 t**2) dt
 120                 sqrt(2 pi)  /-inf
 121 
 122               = 0.5 (1 + erf(x / sqrt(2)))
 123               = 0.5 erfc(x / sqrt(2))
 124   ```
 125 
 126   Args:
 127     x: `Tensor` of type `float32`, `float64`.
 128     name: Python string. A name for the operation (default="ndtr").
 129 
 130   Returns:
 131     ndtr: `Tensor` with `dtype=x.dtype`.
 132 
 133   Raises:
 134     TypeError: if `x` is not floating-type.
 135   """
 136 
 137   with ops.name_scope(name, values=[x]):
 138     x = ops.convert_to_tensor(x, name="x")
 139     if x.dtype.as_numpy_dtype not in [np.float32, np.float64]:
 140       raise TypeError(
 141           "x.dtype=%s is not handled, see docstring for supported types."
 142           % x.dtype)
 143     return _ndtr(x)
 144 
 145 
 146 def _ndtr(x):
 147   """Implements ndtr core logic."""
 148   half_sqrt_2 = constant_op.constant(
 149       0.5 * np.sqrt(2.), dtype=x.dtype, name="half_sqrt_2")
 150   w = x * half_sqrt_2
 151   z = math_ops.abs(w)
 152   y = array_ops.where(math_ops.less(z, half_sqrt_2),
 153                       1. + math_ops.erf(w),
 154                       array_ops.where(math_ops.greater(w, 0.),
 155                                       2. - math_ops.erfc(z),
 156                                       math_ops.erfc(z)))
 157   return 0.5 * y

 160 def ndtri(p, name="ndtri"):
 161   """The inverse of the CDF of the Normal distribution function.
 162 
 163   Returns x such that the area under the pdf from minus infinity to x is equal
 164   to p.
 165 
 166   A piece-wise rational approximation is done for the function.
 167   This is a port of the implementation in netlib.
 168 
 169   Args:
 170     p: `Tensor` of type `float32`, `float64`.
 171     name: Python string. A name for the operation (default="ndtri").
 172 
 173   Returns:
 174     x: `Tensor` with `dtype=p.dtype`.
 175 
 176   Raises:
 177     TypeError: if `p` is not floating-type.
 178   """
 179 
 180   with ops.name_scope(name, values=[p]):
 181     p = ops.convert_to_tensor(p, name="p")
 182     if p.dtype.as_numpy_dtype not in [np.float32, np.float64]:
 183       raise TypeError(
 184           "p.dtype=%s is not handled, see docstring for supported types."
 185           % p.dtype)
 186     return _ndtri(p)

 189 def _ndtri(p):
 190   """Implements ndtri core logic."""
 191 
 192   # Constants used in piece-wise rational approximations. Taken from the cephes
 193   # library:
 194   # https://root.cern.ch/doc/v608/SpecFuncCephesInv_8cxx_source.html
 195   p0 = list(reversed([-5.99633501014107895267E1,
 196                       9.80010754185999661536E1,
 197                       -5.66762857469070293439E1,
 198                       1.39312609387279679503E1,
 199                       -1.23916583867381258016E0]))
 200   q0 = list(reversed([1.0,
 201                       1.95448858338141759834E0,
 202                       4.67627912898881538453E0,
 203                       8.63602421390890590575E1,
 204                       -2.25462687854119370527E2,
 205                       2.00260212380060660359E2,
 206                       -8.20372256168333339912E1,
 207                       1.59056225126211695515E1,
 208                       -1.18331621121330003142E0]))
 209   p1 = list(reversed([4.05544892305962419923E0,
 210                       3.15251094599893866154E1,
 211                       5.71628192246421288162E1,
 212                       4.40805073893200834700E1,
 213                       1.46849561928858024014E1,
 214                       2.18663306850790267539E0,
 215                       -1.40256079171354495875E-1,
 216                       -3.50424626827848203418E-2,
 217                       -8.57456785154685413611E-4]))
 218   q1 = list(reversed([1.0,
 219                       1.57799883256466749731E1,
 220                       4.53907635128879210584E1,
 221                       4.13172038254672030440E1,
 222                       1.50425385692907503408E1,
 223                       2.50464946208309415979E0,
 224                       -1.42182922854787788574E-1,
 225                       -3.80806407691578277194E-2,
 226                       -9.33259480895457427372E-4]))
 227   p2 = list(reversed([3.23774891776946035970E0,
 228                       6.91522889068984211695E0,
 229                       3.93881025292474443415E0,
 230                       1.33303460815807542389E0,
 231                       2.01485389549179081538E-1,
 232                       1.23716634817820021358E-2,
 233                       3.01581553508235416007E-4,
 234                       2.65806974686737550832E-6,
 235                       6.23974539184983293730E-9]))
 236   q2 = list(reversed([1.0,
 237                       6.02427039364742014255E0,
 238                       3.67983563856160859403E0,
 239                       1.37702099489081330271E0,
 240                       2.16236993594496635890E-1,
 241                       1.34204006088543189037E-2,
 242                       3.28014464682127739104E-4,
 243                       2.89247864745380683936E-6,
 244                       6.79019408009981274425E-9]))
 245 
 246   def _create_polynomial(var, coeffs):
 247     """Compute n_th order polynomial via Horner's method."""
 248     coeffs = np.array(coeffs, var.dtype.as_numpy_dtype)
 249     if not coeffs.size:
 250       return array_ops.zeros_like(var)
 251     return coeffs[0] + _create_polynomial(var, coeffs[1:]) * var
 252 
 253   maybe_complement_p = array_ops.where(p > -np.expm1(-2.), 1. - p, p)
 254   # Write in an arbitrary value in place of 0 for p since 0 will cause NaNs
 255   # later on. The result from the computation when p == 0 is not used so any
 256   # number that doesn't result in NaNs is fine.
 257   sanitized_mcp = array_ops.where(
 258       maybe_complement_p <= 0.,
 259       array_ops.fill(array_ops.shape(p), np.array(0.5, p.dtype.as_numpy_dtype)),
 260       maybe_complement_p)
 261 
 262   # Compute x for p > exp(-2): x/sqrt(2pi) = w + w**3 P0(w**2)/Q0(w**2).
 263   w = sanitized_mcp - 0.5
 264   ww = w ** 2
 265   x_for_big_p = w + w * ww * (_create_polynomial(ww, p0)
 266                               / _create_polynomial(ww, q0))
 267   x_for_big_p *= -np.sqrt(2. * np.pi)
 268 
 269   # Compute x for p <= exp(-2): x = z - log(z)/z - (1/z) P(1/z) / Q(1/z),
 270   # where z = sqrt(-2. * log(p)), and P/Q are chosen between two different
 271   # arrays based on whether p < exp(-32).
 272   z = math_ops.sqrt(-2. * math_ops.log(sanitized_mcp))
 273   first_term = z - math_ops.log(z) / z
 274   second_term_small_p = (
 275       _create_polynomial(1. / z, p2) /
 276       _create_polynomial(1. / z, q2) / z)
 277   second_term_otherwise = (
 278       _create_polynomial(1. / z, p1) /
 279       _create_polynomial(1. / z, q1) / z)
 280   x_for_small_p = first_term - second_term_small_p
 281   x_otherwise = first_term - second_term_otherwise
 282 
 283   x = array_ops.where(sanitized_mcp > np.exp(-2.),
 284                       x_for_big_p,
 285                       array_ops.where(z >= 8.0, x_for_small_p, x_otherwise))
 286 
 287   x = array_ops.where(p > 1. - np.exp(-2.), x, -x)
 288   infinity_scalar = constant_op.constant(np.inf, dtype=p.dtype)
 289   infinity = array_ops.fill(array_ops.shape(p), infinity_scalar)
 290   x_nan_replaced = array_ops.where(
 291       p <= 0.0, -infinity, array_ops.where(p >= 1.0, infinity, x))
 292   return x_nan_replaced

 295 def log_ndtr(x, series_order=3, name="log_ndtr"):
 296   """Log Normal distribution function.
 297 
 298   For details of the Normal distribution function see `ndtr`.
 299 
 300   This function calculates `(log o ndtr)(x)` by either calling `log(ndtr(x))` or
 301   using an asymptotic series. Specifically:
 302   - For `x > upper_segment`, use the approximation `-ndtr(-x)` based on
 303     `log(1-x) ~= -x, x << 1`.
 304   - For `lower_segment < x <= upper_segment`, use the existing `ndtr` technique
 305     and take a log.
 306   - For `x <= lower_segment`, we use the series approximation of erf to compute
 307     the log CDF directly.
 308 
 309   The `lower_segment` is set based on the precision of the input:
 310 
 311   ```
 312   lower_segment = { -20,  x.dtype=float64
 313                   { -10,  x.dtype=float32
 314   upper_segment = {   8,  x.dtype=float64
 315                   {   5,  x.dtype=float32
 316   ```
 317 
 318   When `x < lower_segment`, the `ndtr` asymptotic series approximation is:
 319 
 320   ```
 321      ndtr(x) = scale * (1 + sum) + R_N
 322      scale   = exp(-0.5 x**2) / (-x sqrt(2 pi))
 323      sum     = Sum{(-1)^n (2n-1)!! / (x**2)^n, n=1:N}
 324      R_N     = O(exp(-0.5 x**2) (2N+1)!! / |x|^{2N+3})
 325   ```
 326 
 327   where `(2n-1)!! = (2n-1) (2n-3) (2n-5) ...  (3) (1)` is a
 328   [double-factorial](https://en.wikipedia.org/wiki/Double_factorial).
 329 
 330 
 331   Args:
 332     x: `Tensor` of type `float32`, `float64`.
 333     series_order: Positive Python `integer`. Maximum depth to
 334       evaluate the asymptotic expansion. This is the `N` above.
 335     name: Python string. A name for the operation (default="log_ndtr").
 336 
 337   Returns:
 338     log_ndtr: `Tensor` with `dtype=x.dtype`.
 339 
 340   Raises:
 341     TypeError: if `x.dtype` is not handled.
 342     TypeError: if `series_order` is a not Python `integer.`
 343     ValueError:  if `series_order` is not in `[0, 30]`.
 344   """
 345   if not isinstance(series_order, int):
 346     raise TypeError("series_order must be a Python integer.")
 347   if series_order < 0:
 348     raise ValueError("series_order must be non-negative.")
 349   if series_order > 30:
 350     raise ValueError("series_order must be <= 30.")
 351 
 352   with ops.name_scope(name, values=[x]):
 353     x = ops.convert_to_tensor(x, name="x")
 354 
 355     if x.dtype.as_numpy_dtype == np.float64:
 356       lower_segment = LOGNDTR_FLOAT64_LOWER
 357       upper_segment = LOGNDTR_FLOAT64_UPPER
 358     elif x.dtype.as_numpy_dtype == np.float32:
 359       lower_segment = LOGNDTR_FLOAT32_LOWER
 360       upper_segment = LOGNDTR_FLOAT32_UPPER
 361     else:
 362       raise TypeError("x.dtype=%s is not supported." % x.dtype)
 363 
 364     # The basic idea here was ported from:
 365     #   https://root.cern.ch/doc/v608/SpecFuncCephesInv_8cxx_source.html
 366     # We copy the main idea, with a few changes
 367     # * For x >> 1, and X ~ Normal(0, 1),
 368     #     Log[P[X < x]] = Log[1 - P[X < -x]] approx -P[X < -x],
 369     #     which extends the range of validity of this function.
 370     # * We use one fixed series_order for all of 'x', rather than adaptive.
 371     # * Our docstring properly reflects that this is an asymptotic series, not a
 372     #   Taylor series. We also provided a correct bound on the remainder.
 373     # * We need to use the max/min in the _log_ndtr_lower arg to avoid nan when
 374     #   x=0. This happens even though the branch is unchosen because when x=0
 375     #   the gradient of a select involves the calculation 1*dy+0*(-inf)=nan
 376     #   regardless of whether dy is finite. Note that the minimum is a NOP if
 377     #   the branch is chosen.
 378     return array_ops.where(
 379         math_ops.greater(x, upper_segment),
 380         -_ndtr(-x),  # log(1-x) ~= -x, x << 1
 381         array_ops.where(math_ops.greater(x, lower_segment),
 382                         math_ops.log(_ndtr(math_ops.maximum(x, lower_segment))),
 383                         _log_ndtr_lower(math_ops.minimum(x, lower_segment),
 384                                         series_order)))
 385 
 386 
 387 def _log_ndtr_lower(x, series_order):
 388   """Asymptotic expansion version of `Log[cdf(x)]`, appropriate for `x<<-1`."""
 389   x_2 = math_ops.square(x)
 390   # Log of the term multiplying (1 + sum)
 391   log_scale = -0.5 * x_2 - math_ops.log(-x) - 0.5 * np.log(2. * np.pi)
 392   return log_scale + math_ops.log(_log_ndtr_asymptotic_series(x, series_order))
 393 
 394 
 395 def _log_ndtr_asymptotic_series(x, series_order):
 396   """Calculates the asymptotic series used in log_ndtr."""
 397   dtype = x.dtype.as_numpy_dtype
 398   if series_order <= 0:
 399     return np.array(1, dtype)
 400   x_2 = math_ops.square(x)
 401   even_sum = array_ops.zeros_like(x)
 402   odd_sum = array_ops.zeros_like(x)
 403   x_2n = x_2  # Start with x^{2*1} = x^{2*n} with n = 1.
 404   for n in range(1, series_order + 1):
 405     y = np.array(_double_factorial(2 * n - 1), dtype) / x_2n
 406     if n % 2:
 407       odd_sum += y
 408     else:
 409       even_sum += y
 410     x_2n *= x_2
 411   return 1. + even_sum - odd_sum

 414 def erfinv(x, name="erfinv"):
 415   """The inverse function for erf, the error function.
 416 
 417   Args:
 418     x: `Tensor` of type `float32`, `float64`.
 419     name: Python string. A name for the operation (default="erfinv").
 420 
 421   Returns:
 422     x: `Tensor` with `dtype=x.dtype`.
 423 
 424   Raises:
 425     TypeError: if `x` is not floating-type.
 426   """
 427 
 428   with ops.name_scope(name, values=[x]):
 429     x = ops.convert_to_tensor(x, name="x")
 430     if x.dtype.as_numpy_dtype not in [np.float32, np.float64]:
 431       raise TypeError(
 432           "x.dtype=%s is not handled, see docstring for supported types."
 433           % x.dtype)
 434     return ndtri((x + 1.0) / 2.0) / np.sqrt(2)

 437 def _double_factorial(n):
 438   """The double factorial function for small Python integer `n`."""
 439   return np.prod(np.arange(n, 1, -2))

 442 def log_cdf_laplace(x, name="log_cdf_laplace"):
 443   """Log Laplace distribution function.
 444 
 445   This function calculates `Log[L(x)]`, where `L(x)` is the cumulative
 446   distribution function of the Laplace distribution, i.e.
 447 
 448   ```L(x) := 0.5 * int_{-infty}^x e^{-|t|} dt```
 449 
 450   For numerical accuracy, `L(x)` is computed in different ways depending on `x`,
 451 
 452   ```
 453   x <= 0:
 454     Log[L(x)] = Log[0.5] + x, which is exact
 455 
 456   0 < x:
 457     Log[L(x)] = Log[1 - 0.5 * e^{-x}], which is exact
 458   ```
 459 
 460   Args:
 461     x: `Tensor` of type `float32`, `float64`.
 462     name: Python string. A name for the operation (default="log_ndtr").
 463 
 464   Returns:
 465     `Tensor` with `dtype=x.dtype`.
 466 
 467   Raises:
 468     TypeError: if `x.dtype` is not handled.
 469   """
 470 
 471   with ops.name_scope(name, values=[x]):
 472     x = ops.convert_to_tensor(x, name="x")
 473 
 474     # For x < 0, L(x) = 0.5 * exp{x} exactly, so Log[L(x)] = log(0.5) + x.
 475     lower_solution = -np.log(2.) + x
 476 
 477     # safe_exp_neg_x = exp{-x} for x > 0, but is
 478     # bounded above by 1, which avoids
 479     #   log[1 - 1] = -inf for x = log(1/2), AND
 480     #   exp{-x} --> inf, for x << -1
 481     safe_exp_neg_x = math_ops.exp(-math_ops.abs(x))
 482 
 483     # log1p(z) = log(1 + z) approx z for |z| << 1. This approxmation is used
 484     # internally by log1p, rather than being done explicitly here.
 485     upper_solution = math_ops.log1p(-0.5 * safe_exp_neg_x)
 486 
 487     return array_ops.where(x < 0., lower_solution, upper_solution)
