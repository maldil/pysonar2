  35 @tf_export("distributions.Uniform")
  36 class Uniform(distribution.Distribution):
  37   """Uniform distribution with `low` and `high` parameters.
  38 
  39   #### Mathematical Details
  40 
  41   The probability density function (pdf) is,
  42 
  43   ```none
  44   pdf(x; a, b) = I[a <= x < b] / Z
  45   Z = b - a
  46   ```
  47 
  48   where
  49 
  50   - `low = a`,
  51   - `high = b`,
  52   - `Z` is the normalizing constant, and
  53   - `I[predicate]` is the [indicator function](
  54     https://en.wikipedia.org/wiki/Indicator_function) for `predicate`.
  55 
  56   The parameters `low` and `high` must be shaped in a way that supports
  57   broadcasting (e.g., `high - low` is a valid operation).
  58 
  59   #### Examples
  60 
  61   ```python
  62   # Without broadcasting:
  63   u1 = Uniform(low=3.0, high=4.0)  # a single uniform distribution [3, 4]
  64   u2 = Uniform(low=[1.0, 2.0],
  65                high=[3.0, 4.0])  # 2 distributions [1, 3], [2, 4]
  66   u3 = Uniform(low=[[1.0, 2.0],
  67                     [3.0, 4.0]],
  68                high=[[1.5, 2.5],
  69                      [3.5, 4.5]])  # 4 distributions
  70   ```
  71 
  72   ```python
  73   # With broadcasting:
  74   u1 = Uniform(low=3.0, high=[5.0, 6.0, 7.0])  # 3 distributions
  75   ```
  76 
  77   """
  78 
  79   def __init__(self,
  80                low=0.,
  81                high=1.,
  82                validate_args=False,
  83                allow_nan_stats=True,
  84                name="Uniform"):
  85     """Initialize a batch of Uniform distributions.
  86 
  87     Args:
  88       low: Floating point tensor, lower boundary of the output interval. Must
  89         have `low < high`.
  90       high: Floating point tensor, upper boundary of the output interval. Must
  91         have `low < high`.
  92       validate_args: Python `bool`, default `False`. When `True` distribution
  93         parameters are checked for validity despite possibly degrading runtime
  94         performance. When `False` invalid inputs may silently render incorrect
  95         outputs.
  96       allow_nan_stats: Python `bool`, default `True`. When `True`, statistics
  97         (e.g., mean, mode, variance) use the value "`NaN`" to indicate the
  98         result is undefined. When `False`, an exception is raised if one or
  99         more of the statistic's batch members are undefined.
 100       name: Python `str` name prefixed to Ops created by this class.
 101 
 102     Raises:
 103       InvalidArgumentError: if `low >= high` and `validate_args=False`.
 104     """
 105     parameters = dict(locals())
 106     with ops.name_scope(name, values=[low, high]) as name:
 107       with ops.control_dependencies([
 108           check_ops.assert_less(
 109               low, high, message="uniform not defined when low >= high.")
 110       ] if validate_args else []):
 111         self._low = array_ops.identity(low, name="low")
 112         self._high = array_ops.identity(high, name="high")
 113         check_ops.assert_same_float_dtype([self._low, self._high])
 114     super(Uniform, self).__init__(
 115         dtype=self._low.dtype,
 116         reparameterization_type=distribution.FULLY_REPARAMETERIZED,
 117         validate_args=validate_args,
 118         allow_nan_stats=allow_nan_stats,
 119         parameters=parameters,
 120         graph_parents=[self._low,
 121                        self._high],
 122         name=name)
 123 
 124   @staticmethod
 125   def _param_shapes(sample_shape):
 126     return dict(
 127         zip(("low", "high"),
 128             ([ops.convert_to_tensor(sample_shape, dtype=dtypes.int32)] * 2)))
 129 
 130   @property
 131   def low(self):
 132     """Lower boundary of the output interval."""
 133     return self._low
 134 
 135   @property
 136   def high(self):
 137     """Upper boundary of the output interval."""
 138     return self._high
 139 
 140   def range(self, name="range"):
 141     """`high - low`."""
 142     with self._name_scope(name):
 143       return self.high - self.low
 144 
 145   def _batch_shape_tensor(self):
 146     return array_ops.broadcast_dynamic_shape(
 147         array_ops.shape(self.low),
 148         array_ops.shape(self.high))
 149 
 150   def _batch_shape(self):
 151     return array_ops.broadcast_static_shape(
 152         self.low.get_shape(),
 153         self.high.get_shape())
 154 
 155   def _event_shape_tensor(self):
 156     return constant_op.constant([], dtype=dtypes.int32)
 157 
 158   def _event_shape(self):
 159     return tensor_shape.scalar()
 160 
 161   def _sample_n(self, n, seed=None):
 162     shape = array_ops.concat([[n], self.batch_shape_tensor()], 0)
 163     samples = random_ops.random_uniform(shape=shape,
 164                                         dtype=self.dtype,
 165                                         seed=seed)
 166     return self.low + self.range() * samples
 167 
 168   def _prob(self, x):
 169     broadcasted_x = x * array_ops.ones(
 170         self.batch_shape_tensor(), dtype=x.dtype)
 171     return array_ops.where(
 172         math_ops.is_nan(broadcasted_x),
 173         broadcasted_x,
 174         array_ops.where(
 175             math_ops.logical_or(broadcasted_x < self.low,
 176                                 broadcasted_x >= self.high),
 177             array_ops.zeros_like(broadcasted_x),
 178             array_ops.ones_like(broadcasted_x) / self.range()))
 179 
 180   def _cdf(self, x):
 181     broadcast_shape = array_ops.broadcast_dynamic_shape(
 182         array_ops.shape(x), self.batch_shape_tensor())
 183     zeros = array_ops.zeros(broadcast_shape, dtype=self.dtype)
 184     ones = array_ops.ones(broadcast_shape, dtype=self.dtype)
 185     broadcasted_x = x * ones
 186     result_if_not_big = array_ops.where(
 187         x < self.low, zeros, (broadcasted_x - self.low) / self.range())
 188     return array_ops.where(x >= self.high, ones, result_if_not_big)
 189 
 190   def _entropy(self):
 191     return math_ops.log(self.range())
 192 
 193   def _mean(self):
 194     return (self.low + self.high) / 2.
 195 
 196   def _variance(self):
 197     return math_ops.square(self.range()) / 12.
 198 
 199   def _stddev(self):
 200     return self.range() / math.sqrt(12.)
