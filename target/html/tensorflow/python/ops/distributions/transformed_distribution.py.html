  44 def _static_value(x):
  45   """Returns the static value of a `Tensor` or `None`."""
  46   return tensor_util.constant_value(ops.convert_to_tensor(x))

  49 def _logical_and(*args):
  50   """Convenience function which attempts to statically `reduce_all`."""
  51   args_ = [_static_value(x) for x in args]
  52   if any(x is not None and not bool(x) for x in args_):
  53     return constant_op.constant(False)
  54   if all(x is not None and bool(x) for x in args_):
  55     return constant_op.constant(True)
  56   if len(args) == 2:
  57     return math_ops.logical_and(*args)
  58   return math_ops.reduce_all(args)
  59 
  60 
  61 def _logical_equal(x, y):
  62   """Convenience function which attempts to statically compute `x == y`."""
  63   x_ = _static_value(x)
  64   y_ = _static_value(y)
  65   if x_ is None or y_ is None:
  66     return math_ops.equal(x, y)
  67   return constant_op.constant(np.array_equal(x_, y_))
  68 
  69 
  70 def _logical_not(x):
  71   """Convenience function which attempts to statically apply `logical_not`."""
  72   x_ = _static_value(x)
  73   if x_ is None:
  74     return math_ops.logical_not(x)
  75   return constant_op.constant(np.logical_not(x_))

  78 def _concat_vectors(*args):
  79   """Convenience function which concatenates input vectors."""
  80   args_ = [_static_value(x) for x in args]
  81   if any(x_ is None for x_ in args_):
  82     return array_ops.concat(args, 0)
  83   return constant_op.constant([x_ for vec_ in args_ for x_ in vec_])

  86 def _pick_scalar_condition(pred, cond_true, cond_false):
  87   """Convenience function which chooses the condition based on the predicate."""
  88   # Note: This function is only valid if all of pred, cond_true, and cond_false
  89   # are scalars. This means its semantics are arguably more like tf.cond than
  90   # tf.select even though we use tf.select to implement it.
  91   pred_ = _static_value(pred)
  92   if pred_ is None:
  93     return array_ops.where(pred, cond_true, cond_false)
  94   return cond_true if pred_ else cond_false

  97 def _ones_like(x):
  98   """Convenience function attempts to statically construct `ones_like`."""
  99   # Should only be used for small vectors.
 100   if x.get_shape().is_fully_defined():
 101     return array_ops.ones(x.get_shape().as_list(), dtype=x.dtype)
 102   return array_ops.ones_like(x)

 105 def _ndims_from_shape(shape):
 106   """Returns `Tensor`'s `rank` implied by a `Tensor` shape."""
 107   if shape.get_shape().ndims not in (None, 1):
 108     raise ValueError("input is not a valid shape: not 1D")
 109   if not shape.dtype.is_integer:
 110     raise TypeError("input is not a valid shape: wrong dtype")
 111   if shape.get_shape().is_fully_defined():
 112     return constant_op.constant(shape.get_shape().as_list()[0])
 113   return array_ops.shape(shape)[0]
 114 
 115 
 116 def _is_scalar_from_shape(shape):
 117   """Returns `True` `Tensor` if `Tensor` shape implies a scalar."""
 118   return _logical_equal(_ndims_from_shape(shape), 0)

 121 class TransformedDistribution(distribution_lib.Distribution):
 122   """A Transformed Distribution.
 123 
 124   A `TransformedDistribution` models `p(y)` given a base distribution `p(x)`,
 125   and a deterministic, invertible, differentiable transform, `Y = g(X)`. The
 126   transform is typically an instance of the `Bijector` class and the base
 127   distribution is typically an instance of the `Distribution` class.
 128 
 129   A `Bijector` is expected to implement the following functions:
 130   - `forward`,
 131   - `inverse`,
 132   - `inverse_log_det_jacobian`.
 133   The semantics of these functions are outlined in the `Bijector` documentation.
 134 
 135   We now describe how a `TransformedDistribution` alters the input/outputs of a
 136   `Distribution` associated with a random variable (rv) `X`.
 137 
 138   Write `cdf(Y=y)` for an absolutely continuous cumulative distribution function
 139   of random variable `Y`; write the probability density function `pdf(Y=y) :=
 140   d^k / (dy_1,...,dy_k) cdf(Y=y)` for its derivative wrt to `Y` evaluated at
 141   `y`. Assume that `Y = g(X)` where `g` is a deterministic diffeomorphism,
 142   i.e., a non-random, continuous, differentiable, and invertible function.
 143   Write the inverse of `g` as `X = g^{-1}(Y)` and `(J o g)(x)` for the Jacobian
 144   of `g` evaluated at `x`.
 145 
 146   A `TransformedDistribution` implements the following operations:
 147 
 148     * `sample`
 149       Mathematically:   `Y = g(X)`
 150       Programmatically: `bijector.forward(distribution.sample(...))`
 151 
 152     * `log_prob`
 153       Mathematically:   `(log o pdf)(Y=y) = (log o pdf o g^{-1})(y)
 154                          + (log o abs o det o J o g^{-1})(y)`
 155       Programmatically: `(distribution.log_prob(bijector.inverse(y))
 156                          + bijector.inverse_log_det_jacobian(y))`
 157 
 158     * `log_cdf`
 159       Mathematically:   `(log o cdf)(Y=y) = (log o cdf o g^{-1})(y)`
 160       Programmatically: `distribution.log_cdf(bijector.inverse(x))`
 161 
 162     * and similarly for: `cdf`, `prob`, `log_survival_function`,
 163      `survival_function`.
 164 
 165   A simple example constructing a Log-Normal distribution from a Normal
 166   distribution:
 167 
 168   ```python
 169   ds = tf.contrib.distributions
 170   log_normal = ds.TransformedDistribution(
 171     distribution=ds.Normal(loc=0., scale=1.),
 172     bijector=ds.bijectors.Exp(),
 173     name="LogNormalTransformedDistribution")
 174   ```
 175 
 176   A `LogNormal` made from callables:
 177 
 178   ```python
 179   ds = tf.contrib.distributions
 180   log_normal = ds.TransformedDistribution(
 181     distribution=ds.Normal(loc=0., scale=1.),
 182     bijector=ds.bijectors.Inline(
 183       forward_fn=tf.exp,
 184       inverse_fn=tf.log,
 185       inverse_log_det_jacobian_fn=(
 186         lambda y: -tf.reduce_sum(tf.log(y), axis=-1)),
 187     name="LogNormalTransformedDistribution")
 188   ```
 189 
 190   Another example constructing a Normal from a StandardNormal:
 191 
 192   ```python
 193   ds = tf.contrib.distributions
 194   normal = ds.TransformedDistribution(
 195     distribution=ds.Normal(loc=0., scale=1.),
 196     bijector=ds.bijectors.Affine(
 197       shift=-1.,
 198       scale_identity_multiplier=2.)
 199     name="NormalTransformedDistribution")
 200   ```
 201 
 202   A `TransformedDistribution`'s batch- and event-shape are implied by the base
 203   distribution unless explicitly overridden by `batch_shape` or `event_shape`
 204   arguments. Specifying an overriding `batch_shape` (`event_shape`) is
 205   permitted only if the base distribution has scalar batch-shape (event-shape).
 206   The bijector is applied to the distribution as if the distribution possessed
 207   the overridden shape(s). The following example demonstrates how to construct a
 208   multivariate Normal as a `TransformedDistribution`.
 209 
 210   ```python
 211   ds = tf.contrib.distributions
 212   # We will create two MVNs with batch_shape = event_shape = 2.
 213   mean = [[-1., 0],      # batch:0
 214           [0., 1]]       # batch:1
 215   chol_cov = [[[1., 0],
 216                [0, 1]],  # batch:0
 217               [[1, 0],
 218                [2, 2]]]  # batch:1
 219   mvn1 = ds.TransformedDistribution(
 220       distribution=ds.Normal(loc=0., scale=1.),
 221       bijector=ds.bijectors.Affine(shift=mean, scale_tril=chol_cov),
 222       batch_shape=[2],  # Valid because base_distribution.batch_shape == [].
 223       event_shape=[2])  # Valid because base_distribution.event_shape == [].
 224   mvn2 = ds.MultivariateNormalTriL(loc=mean, scale_tril=chol_cov)
 225   # mvn1.log_prob(x) == mvn2.log_prob(x)
 226   ```
 227 
 228   """

 230   def __init__(self,
 231                distribution,
 232                bijector=None,
 233                batch_shape=None,
 234                event_shape=None,
 235                validate_args=False,
 236                name=None):
 237     """Construct a Transformed Distribution.
 238 
 239     Args:
 240       distribution: The base distribution instance to transform. Typically an
 241         instance of `Distribution`.
 242       bijector: The object responsible for calculating the transformation.
 243         Typically an instance of `Bijector`. `None` means `Identity()`.
 244       batch_shape: `integer` vector `Tensor` which overrides `distribution`
 245         `batch_shape`; valid only if `distribution.is_scalar_batch()`.
 246       event_shape: `integer` vector `Tensor` which overrides `distribution`
 247         `event_shape`; valid only if `distribution.is_scalar_event()`.
 248       validate_args: Python `bool`, default `False`. When `True` distribution
 249         parameters are checked for validity despite possibly degrading runtime
 250         performance. When `False` invalid inputs may silently render incorrect
 251         outputs.
 252       name: Python `str` name prefixed to Ops created by this class. Default:
 253         `bijector.name + distribution.name`.
 254     """
 255     parameters = dict(locals())
 256     name = name or (("" if bijector is None else bijector.name) +
 257                     distribution.name)
 258     with ops.name_scope(name, values=[event_shape, batch_shape]) as name:
 259       # For convenience we define some handy constants.
 260       self._zero = constant_op.constant(0, dtype=dtypes.int32, name="zero")
 261       self._empty = constant_op.constant([], dtype=dtypes.int32, name="empty")
 262 
 263       if bijector is None:
 264         bijector = identity_bijector.Identity(validate_args=validate_args)
 265 
 266       # We will keep track of a static and dynamic version of
 267       # self._is_{batch,event}_override. This way we can do more prior to graph
 268       # execution, including possibly raising Python exceptions.
 269 
 270       self._override_batch_shape = self._maybe_validate_shape_override(
 271           batch_shape, distribution.is_scalar_batch(), validate_args,
 272           "batch_shape")
 273       self._is_batch_override = _logical_not(_logical_equal(
 274           _ndims_from_shape(self._override_batch_shape), self._zero))
 275       self._is_maybe_batch_override = bool(
 276           tensor_util.constant_value(self._override_batch_shape) is None or
 277           tensor_util.constant_value(self._override_batch_shape).size != 0)
 278 
 279       self._override_event_shape = self._maybe_validate_shape_override(
 280           event_shape, distribution.is_scalar_event(), validate_args,
 281           "event_shape")
 282       self._is_event_override = _logical_not(_logical_equal(
 283           _ndims_from_shape(self._override_event_shape), self._zero))
 284       self._is_maybe_event_override = bool(
 285           tensor_util.constant_value(self._override_event_shape) is None or
 286           tensor_util.constant_value(self._override_event_shape).size != 0)
 287 
 288       # To convert a scalar distribution into a multivariate distribution we
 289       # will draw dims from the sample dims, which are otherwise iid. This is
 290       # easy to do except in the case that the base distribution has batch dims
 291       # and we're overriding event shape. When that case happens the event dims
 292       # will incorrectly be to the left of the batch dims. In this case we'll
 293       # cyclically permute left the new dims.
 294       self._needs_rotation = _logical_and(
 295           self._is_event_override,
 296           _logical_not(self._is_batch_override),
 297           _logical_not(distribution.is_scalar_batch()))
 298       override_event_ndims = _ndims_from_shape(self._override_event_shape)
 299       self._rotate_ndims = _pick_scalar_condition(
 300           self._needs_rotation, override_event_ndims, 0)
 301       # We'll be reducing the head dims (if at all), i.e., this will be []
 302       # if we don't need to reduce.
 303       self._reduce_event_indices = math_ops.range(
 304           self._rotate_ndims - override_event_ndims, self._rotate_ndims)
 305 
 306     self._distribution = distribution
 307     self._bijector = bijector
 308     super(TransformedDistribution, self).__init__(
 309         dtype=self._distribution.dtype,
 310         reparameterization_type=self._distribution.reparameterization_type,
 311         validate_args=validate_args,
 312         allow_nan_stats=self._distribution.allow_nan_stats,
 313         parameters=parameters,
 314         # We let TransformedDistribution access _graph_parents since this class
 315         # is more like a baseclass than derived.
 316         graph_parents=(distribution._graph_parents +  # pylint: disable=protected-access
 317                        bijector.graph_parents),
 318         name=name)
 319

 330   def _event_shape_tensor(self):
 331     return self.bijector.forward_event_shape_tensor(
 332         distribution_util.pick_vector(
 333             self._is_event_override,
 334             self._override_event_shape,
 335             self.distribution.event_shape_tensor()))

 337   def _event_shape(self):
 338     # If there's a chance that the event_shape has been overridden, we return
 339     # what we statically know about the `event_shape_override`. This works
 340     # because: `_is_maybe_event_override` means `static_override` is `None` or a
 341     # non-empty list, i.e., we don't statically know the `event_shape` or we do.
 342     #
 343     # Since the `bijector` may change the `event_shape`, we then forward what we
 344     # know to the bijector. This allows the `bijector` to have final say in the
 345     # `event_shape`.
 346     static_override = tensor_util.constant_value_as_shape(
 347         self._override_event_shape)
 348     return self.bijector.forward_event_shape(
 349         static_override
 350         if self._is_maybe_event_override
 351         else self.distribution.event_shape)
 352 
 353   def _batch_shape_tensor(self):
 354     return distribution_util.pick_vector(
 355         self._is_batch_override,
 356         self._override_batch_shape,
 357         self.distribution.batch_shape_tensor())
 358 
 359   def _batch_shape(self):
 360     # If there's a chance that the batch_shape has been overridden, we return
 361     # what we statically know about the `batch_shape_override`. This works
 362     # because: `_is_maybe_batch_override` means `static_override` is `None` or a
 363     # non-empty list, i.e., we don't statically know the `batch_shape` or we do.
 364     #
 365     # Notice that this implementation parallels the `_event_shape` except that
 366     # the `bijector` doesn't get to alter the `batch_shape`. Recall that
 367     # `batch_shape` is a property of a distribution while `event_shape` is
 368     # shared between both the `distribution` instance and the `bijector`.
 369     static_override = tensor_util.constant_value_as_shape(
 370         self._override_batch_shape)
 371     return (static_override
 372             if self._is_maybe_batch_override
 373             else self.distribution.batch_shape)
 374 
 375   def _sample_n(self, n, seed=None):
 376     sample_shape = _concat_vectors(
 377         distribution_util.pick_vector(self._needs_rotation, self._empty, [n]),
 378         self._override_batch_shape,
 379         self._override_event_shape,
 380         distribution_util.pick_vector(self._needs_rotation, [n], self._empty))
 381     x = self.distribution.sample(sample_shape=sample_shape, seed=seed)
 382     x = self._maybe_rotate_dims(x)
 383     # We'll apply the bijector in the `_call_sample_n` function.
 384     return x
 385 
 386   def _call_sample_n(self, sample_shape, seed, name, **kwargs):
 387     # We override `_call_sample_n` rather than `_sample_n` so we can ensure that
 388     # the result of `self.bijector.forward` is not modified (and thus caching
 389     # works).
 390     with self._name_scope(name, values=[sample_shape]):
 391       sample_shape = ops.convert_to_tensor(
 392           sample_shape, dtype=dtypes.int32, name="sample_shape")
 393       sample_shape, n = self._expand_sample_shape_to_vector(
 394           sample_shape, "sample_shape")
 395 
 396       # First, generate samples. We will possibly generate extra samples in the
 397       # event that we need to reinterpret the samples as part of the
 398       # event_shape.
 399       x = self._sample_n(n, seed, **kwargs)
 400 
 401       # Next, we reshape `x` into its final form. We do this prior to the call
 402       # to the bijector to ensure that the bijector caching works.
 403       batch_event_shape = array_ops.shape(x)[1:]
 404       final_shape = array_ops.concat([sample_shape, batch_event_shape], 0)
 405       x = array_ops.reshape(x, final_shape)
 406 
 407       # Finally, we apply the bijector's forward transformation. For caching to
 408       # work, it is imperative that this is the last modification to the
 409       # returned result.
 410       y = self.bijector.forward(x, **kwargs)
 411       y = self._set_sample_static_shape(y, sample_shape)
 412 
 413       return y

 415   def _log_prob(self, y):
 416     # For caching to work, it is imperative that the bijector is the first to
 417     # modify the input.
 418     x = self.bijector.inverse(y)
 419     event_ndims = self._maybe_get_static_event_ndims()
 420 
 421     ildj = self.bijector.inverse_log_det_jacobian(y, event_ndims=event_ndims)
 422     if self.bijector._is_injective:  # pylint: disable=protected-access
 423       return self._finish_log_prob_for_one_fiber(y, x, ildj, event_ndims)
 424 
 425     lp_on_fibers = [
 426         self._finish_log_prob_for_one_fiber(y, x_i, ildj_i, event_ndims)
 427         for x_i, ildj_i in zip(x, ildj)]
 428     return math_ops.reduce_logsumexp(array_ops.stack(lp_on_fibers), axis=0)
 429 
 430   def _finish_log_prob_for_one_fiber(self, y, x, ildj, event_ndims):
 431     """Finish computation of log_prob on one element of the inverse image."""
 432     x = self._maybe_rotate_dims(x, rotate_right=True)
 433     log_prob = self.distribution.log_prob(x)
 434     if self._is_maybe_event_override:
 435       log_prob = math_ops.reduce_sum(log_prob, self._reduce_event_indices)
 436     log_prob += math_ops.cast(ildj, log_prob.dtype)
 437     if self._is_maybe_event_override and isinstance(event_ndims, int):
 438       log_prob.set_shape(
 439           array_ops.broadcast_static_shape(
 440               y.get_shape().with_rank_at_least(1)[:-event_ndims],
 441               self.batch_shape))
 442     return log_prob
 443 
 444   def _prob(self, y):
 445     x = self.bijector.inverse(y)
 446     event_ndims = self._maybe_get_static_event_ndims()
 447     ildj = self.bijector.inverse_log_det_jacobian(y, event_ndims=event_ndims)
 448     if self.bijector._is_injective:  # pylint: disable=protected-access
 449       return self._finish_prob_for_one_fiber(y, x, ildj, event_ndims)
 450 
 451     prob_on_fibers = [
 452         self._finish_prob_for_one_fiber(y, x_i, ildj_i, event_ndims)
 453         for x_i, ildj_i in zip(x, ildj)]
 454     return sum(prob_on_fibers)
 455 
 456   def _finish_prob_for_one_fiber(self, y, x, ildj, event_ndims):
 457     """Finish computation of prob on one element of the inverse image."""
 458     x = self._maybe_rotate_dims(x, rotate_right=True)
 459     prob = self.distribution.prob(x)
 460     if self._is_maybe_event_override:
 461       prob = math_ops.reduce_prod(prob, self._reduce_event_indices)
 462     prob *= math_ops.exp(math_ops.cast(ildj, prob.dtype))
 463     if self._is_maybe_event_override and isinstance(event_ndims, int):
 464       prob.set_shape(
 465           array_ops.broadcast_static_shape(
 466               y.get_shape().with_rank_at_least(1)[:-event_ndims],
 467               self.batch_shape))
 468     return prob

 470   def _log_cdf(self, y):
 471     if self._is_maybe_event_override:
 472       raise NotImplementedError("log_cdf is not implemented when overriding "
 473                                 "event_shape")

 475       raise NotImplementedError("log_cdf is not implemented when "
 476                                 "bijector is not injective.")
 477     x = self.bijector.inverse(y)
 478     return self.distribution.log_cdf(x)

 510   def _quantile(self, value):
 511     if self._is_maybe_event_override:
 512       raise NotImplementedError("quantile is not implemented when overriding "
 513                                 "event_shape")
 514     if not self.bijector._is_injective:  # pylint: disable=protected-access
 515       raise NotImplementedError("quantile is not implemented when "
 516                                 "bijector is not injective.")
 517     # x_q is the "qth quantile" of X iff q = P[X <= x_q].  Now, since X =
 518     # g^{-1}(Y), q = P[X <= x_q] = P[g^{-1}(Y) <= x_q] = P[Y <= g(x_q)],
 519     # implies the qth quantile of Y is g(x_q).
 520     inv_cdf = self.distribution.quantile(value)
 521     return self.bijector.forward(inv_cdf)
 522 
 523   def _entropy(self):
 524     if not self.bijector.is_constant_jacobian:
 525       raise NotImplementedError("entropy is not implemented")
 526     if not self.bijector._is_injective:  # pylint: disable=protected-access
 527       raise NotImplementedError("entropy is not implemented when "
 528                                 "bijector is not injective.")
 529     # Suppose Y = g(X) where g is a diffeomorphism and X is a continuous rv. It
 530     # can be shown that:
 531     #   H[Y] = H[X] + E_X[(log o abs o det o J o g)(X)].
 532     # If is_constant_jacobian then:
 533     #   E_X[(log o abs o det o J o g)(X)] = (log o abs o det o J o g)(c)
 534     # where c can by anything.
 535     entropy = self.distribution.entropy()
 536     if self._is_maybe_event_override:
 537       # H[X] = sum_i H[X_i] if X_i are mutually independent.
 538       # This means that a reduce_sum is a simple rescaling.
 539       entropy *= math_ops.cast(math_ops.reduce_prod(self._override_event_shape),
 540                                dtype=entropy.dtype.base_dtype)
 541     if self._is_maybe_batch_override:
 542       new_shape = array_ops.concat([
 543           _ones_like(self._override_batch_shape),
 544           self.distribution.batch_shape_tensor()
 545       ], 0)
 546       entropy = array_ops.reshape(entropy, new_shape)
 547       multiples = array_ops.concat([
 548           self._override_batch_shape,
 549           _ones_like(self.distribution.batch_shape_tensor())
 550       ], 0)
 551       entropy = array_ops.tile(entropy, multiples)
 552     dummy = array_ops.zeros(
 553         shape=array_ops.concat(
 554             [self.batch_shape_tensor(), self.event_shape_tensor()],
 555             0),
 556         dtype=self.dtype)
 557     event_ndims = (self.event_shape.ndims if self.event_shape.ndims is not None
 558                    else array_ops.size(self.event_shape_tensor()))
 559     ildj = self.bijector.inverse_log_det_jacobian(
 560         dummy, event_ndims=event_ndims)
 561 
 562     entropy -= math_ops.cast(ildj, entropy.dtype)
 563     entropy.set_shape(self.batch_shape)
 564     return entropy
 565 
 566   def _maybe_validate_shape_override(self, override_shape, base_is_scalar,
 567                                      validate_args, name):
 568     """Helper to __init__ which ensures override batch/event_shape are valid."""
 569     if override_shape is None:
 570       override_shape = []
 571 
 572     override_shape = ops.convert_to_tensor(override_shape, dtype=dtypes.int32,
 573                                            name=name)
 574 
 575     if not override_shape.dtype.is_integer:
 576       raise TypeError("shape override must be an integer")
 577 
 578     override_is_scalar = _is_scalar_from_shape(override_shape)
 579     if tensor_util.constant_value(override_is_scalar):
 580       return self._empty
 581 
 582     dynamic_assertions = []
 583 
 584     if override_shape.get_shape().ndims is not None:
 585       if override_shape.get_shape().ndims != 1:
 586         raise ValueError("shape override must be a vector")
 587     elif validate_args:
 588       dynamic_assertions += [check_ops.assert_rank(
 589           override_shape, 1,
 590           message="shape override must be a vector")]
 591 
 592     if tensor_util.constant_value(override_shape) is not None:
 593       if any(s <= 0 for s in tensor_util.constant_value(override_shape)):
 594         raise ValueError("shape override must have positive elements")
 595     elif validate_args:
 596       dynamic_assertions += [check_ops.assert_positive(
 597           override_shape,
 598           message="shape override must have positive elements")]
 599 
 600     is_both_nonscalar = _logical_and(_logical_not(base_is_scalar),
 601                                      _logical_not(override_is_scalar))
 602     if tensor_util.constant_value(is_both_nonscalar) is not None:
 603       if tensor_util.constant_value(is_both_nonscalar):
 604         raise ValueError("base distribution not scalar")
 605     elif validate_args:
 606       dynamic_assertions += [check_ops.assert_equal(
 607           is_both_nonscalar, False,
 608           message="base distribution not scalar")]
 609 
 610     if not dynamic_assertions:
 611       return override_shape
 612     return control_flow_ops.with_dependencies(
 613         dynamic_assertions, override_shape)
 614 
 615   def _maybe_rotate_dims(self, x, rotate_right=False):
 616     """Helper which rolls left event_dims left or right event_dims right."""
 617     needs_rotation_const = tensor_util.constant_value(self._needs_rotation)
 618     if needs_rotation_const is not None and not needs_rotation_const:
 619       return x
 620     ndims = array_ops.rank(x)
 621     n = (ndims - self._rotate_ndims) if rotate_right else self._rotate_ndims
 622     return array_ops.transpose(
 623         x, _concat_vectors(math_ops.range(n, ndims), math_ops.range(0, n)))
 624 
 625   def _maybe_get_static_event_ndims(self):
 626     if self.event_shape.ndims is not None:
 627       return self.event_shape.ndims
 628 
 629     event_ndims = array_ops.size(self.event_shape_tensor())
 630     event_ndims_ = distribution_util.maybe_get_static_value(event_ndims)
 631 
 632     if event_ndims_ is not None:
 633       return event_ndims_
 634 
 635     return event_ndims
