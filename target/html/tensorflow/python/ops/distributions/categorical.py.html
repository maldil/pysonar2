  35 def _broadcast_cat_event_and_params(event, params, base_dtype):
  36   """Broadcasts the event or distribution parameters."""
  37   if event.dtype.is_integer:
  38     pass
  39   elif event.dtype.is_floating:
  40     # When `validate_args=True` we've already ensured int/float casting
  41     # is closed.
  42     event = math_ops.cast(event, dtype=dtypes.int32)
  43   else:
  44     raise TypeError("`value` should have integer `dtype` or "
  45                     "`self.dtype` ({})".format(base_dtype))
  46   shape_known_statically = (
  47       params.shape.ndims is not None and
  48       params.shape[:-1].is_fully_defined() and
  49       event.shape.is_fully_defined())
  50   if not shape_known_statically or params.shape[:-1] != event.shape:
  51     params *= array_ops.ones_like(event[..., array_ops.newaxis],
  52                                   dtype=params.dtype)
  53     params_shape = array_ops.shape(params)[:-1]
  54     event *= array_ops.ones(params_shape, dtype=event.dtype)
  55     if params.shape.ndims is not None:
  56       event.set_shape(tensor_shape.TensorShape(params.shape[:-1]))
  57 
  58   return event, params

  61 @tf_export("distributions.Categorical")
  62 class Categorical(distribution.Distribution):
  63   """Categorical distribution.
  64 
  65   The Categorical distribution is parameterized by either probabilities or
  66   log-probabilities of a set of `K` classes. It is defined over the integers
  67   `{0, 1, ..., K}`.
  68 
  69   The Categorical distribution is closely related to the `OneHotCategorical` and
  70   `Multinomial` distributions.  The Categorical distribution can be intuited as
  71   generating samples according to `argmax{ OneHotCategorical(probs) }` itself
  72   being identical to `argmax{ Multinomial(probs, total_count=1) }`.
  73 
  74   #### Mathematical Details
  75 
  76   The probability mass function (pmf) is,
  77 
  78   ```none
  79   pmf(k; pi) = prod_j pi_j**[k == j]
  80   ```
  81 
  82   #### Pitfalls
  83 
  84   The number of classes, `K`, must not exceed:
  85   - the largest integer representable by `self.dtype`, i.e.,
  86     `2**(mantissa_bits+1)` (IEEE 754),
  87   - the maximum `Tensor` index, i.e., `2**31-1`.
  88 
  89   In other words,
  90 
  91   ```python
  92   K <= min(2**31-1, {
  93     tf.float16: 2**11,
  94     tf.float32: 2**24,
  95     tf.float64: 2**53 }[param.dtype])
  96   ```
  97 
  98   Note: This condition is validated only when `self.validate_args = True`.
  99 
 100   #### Examples
 101 
 102   Creates a 3-class distribution with the 2nd class being most likely.
 103 
 104   ```python
 105   dist = Categorical(probs=[0.1, 0.5, 0.4])
 106   n = 1e4
 107   empirical_prob = tf.cast(
 108       tf.histogram_fixed_width(
 109         dist.sample(int(n)),
 110         [0., 2],
 111         nbins=3),
 112       dtype=tf.float32) / n
 113   # ==> array([ 0.1005,  0.5037,  0.3958], dtype=float32)
 114   ```
 115 
 116   Creates a 3-class distribution with the 2nd class being most likely.
 117   Parameterized by [logits](https://en.wikipedia.org/wiki/Logit) rather than
 118   probabilities.
 119 
 120   ```python
 121   dist = Categorical(logits=np.log([0.1, 0.5, 0.4])
 122   n = 1e4
 123   empirical_prob = tf.cast(
 124       tf.histogram_fixed_width(
 125         dist.sample(int(n)),
 126         [0., 2],
 127         nbins=3),
 128       dtype=tf.float32) / n
 129   # ==> array([0.1045,  0.5047, 0.3908], dtype=float32)
 130   ```
 131 
 132   Creates a 3-class distribution with the 3rd class being most likely.
 133   The distribution functions can be evaluated on counts.
 134 
 135   ```python
 136   # counts is a scalar.
 137   p = [0.1, 0.4, 0.5]
 138   dist = Categorical(probs=p)
 139   dist.prob(0)  # Shape []
 140 
 141   # p will be broadcast to [[0.1, 0.4, 0.5], [0.1, 0.4, 0.5]] to match counts.
 142   counts = [1, 0]
 143   dist.prob(counts)  # Shape [2]
 144 
 145   # p will be broadcast to shape [3, 5, 7, 3] to match counts.
 146   counts = [[...]] # Shape [5, 7, 3]
 147   dist.prob(counts)  # Shape [5, 7, 3]
 148   ```
 149 
 150   """

 152   def __init__(
 153       self,
 154       logits=None,
 155       probs=None,
 156       dtype=dtypes.int32,
 157       validate_args=False,
 158       allow_nan_stats=True,
 159       name="Categorical"):
 160     """Initialize Categorical distributions using class log-probabilities.
 161 
 162     Args:
 163       logits: An N-D `Tensor`, `N >= 1`, representing the log probabilities
 164         of a set of Categorical distributions. The first `N - 1` dimensions
 165         index into a batch of independent distributions and the last dimension
 166         represents a vector of logits for each class. Only one of `logits` or
 167         `probs` should be passed in.
 168       probs: An N-D `Tensor`, `N >= 1`, representing the probabilities
 169         of a set of Categorical distributions. The first `N - 1` dimensions
 170         index into a batch of independent distributions and the last dimension
 171         represents a vector of probabilities for each class. Only one of
 172         `logits` or `probs` should be passed in.
 173       dtype: The type of the event samples (default: int32).
 174       validate_args: Python `bool`, default `False`. When `True` distribution
 175         parameters are checked for validity despite possibly degrading runtime
 176         performance. When `False` invalid inputs may silently render incorrect
 177         outputs.
 178       allow_nan_stats: Python `bool`, default `True`. When `True`, statistics
 179         (e.g., mean, mode, variance) use the value "`NaN`" to indicate the
 180         result is undefined. When `False`, an exception is raised if one or
 181         more of the statistic's batch members are undefined.
 182       name: Python `str` name prefixed to Ops created by this class.
 183     """
 184     parameters = dict(locals())
 185     with ops.name_scope(name, values=[logits, probs]) as name:
 186       self._logits, self._probs = distribution_util.get_logits_and_probs(
 187           logits=logits,
 188           probs=probs,
 189           validate_args=validate_args,
 190           multidimensional=True,
 191           name=name)
 192 
 193       if validate_args:
 194         self._logits = distribution_util.embed_check_categorical_event_shape(
 195             self._logits)
 196 
 197       logits_shape_static = self._logits.get_shape().with_rank_at_least(1)
 198       if logits_shape_static.ndims is not None:
 199         self._batch_rank = ops.convert_to_tensor(
 200             logits_shape_static.ndims - 1,
 201             dtype=dtypes.int32,
 202             name="batch_rank")
 203       else:
 204         with ops.name_scope(name="batch_rank"):
 205           self._batch_rank = array_ops.rank(self._logits) - 1
 206 
 207       logits_shape = array_ops.shape(self._logits, name="logits_shape")
 208       if logits_shape_static[-1].value is not None:
 209         self._event_size = ops.convert_to_tensor(
 210             logits_shape_static[-1].value,
 211             dtype=dtypes.int32,
 212             name="event_size")
 213       else:
 214         with ops.name_scope(name="event_size"):
 215           self._event_size = logits_shape[self._batch_rank]
 216 
 217       if logits_shape_static[:-1].is_fully_defined():
 218         self._batch_shape_val = constant_op.constant(
 219             logits_shape_static[:-1].as_list(),
 220             dtype=dtypes.int32,
 221             name="batch_shape")
 222       else:
 223         with ops.name_scope(name="batch_shape"):
 224           self._batch_shape_val = logits_shape[:-1]
 225     super(Categorical, self).__init__(
 226         dtype=dtype,
 227         reparameterization_type=distribution.NOT_REPARAMETERIZED,
 228         validate_args=validate_args,
 229         allow_nan_stats=allow_nan_stats,
 230         parameters=parameters,
 231         graph_parents=[self._logits,
 232                        self._probs],
 233         name=name)
 234 
 235   @property
 236   def event_size(self):
 237     """Scalar `int32` tensor: the number of classes."""
 238     return self._event_size
 239 
 240   @property
 241   def logits(self):
 242     """Vector of coordinatewise logits."""
 243     return self._logits
 244 
 245   @property
 246   def probs(self):
 247     """Vector of coordinatewise probabilities."""
 248     return self._probs
 249 
 250   def _batch_shape_tensor(self):
 251     return array_ops.identity(self._batch_shape_val)
 252 
 253   def _batch_shape(self):
 254     return self.logits.get_shape()[:-1]
 255 
 256   def _event_shape_tensor(self):
 257     return constant_op.constant([], dtype=dtypes.int32)
 258 
 259   def _event_shape(self):
 260     return tensor_shape.scalar()
 261 
 262   def _sample_n(self, n, seed=None):
 263     if self.logits.get_shape().ndims == 2:
 264       logits_2d = self.logits
 265     else:
 266       logits_2d = array_ops.reshape(self.logits, [-1, self.event_size])
 267     sample_dtype = dtypes.int64 if self.dtype.size > 4 else dtypes.int32
 268     draws = random_ops.multinomial(
 269         logits_2d, n, seed=seed, output_dtype=sample_dtype)
 270     draws = array_ops.reshape(
 271         array_ops.transpose(draws),
 272         array_ops.concat([[n], self.batch_shape_tensor()], 0))
 273     return math_ops.cast(draws, self.dtype)
 274 
 275   def _cdf(self, k):
 276     k = ops.convert_to_tensor(k, name="k")
 277     if self.validate_args:
 278       k = distribution_util.embed_check_integer_casting_closed(
 279           k, target_dtype=dtypes.int32)
 280 
 281     k, probs = _broadcast_cat_event_and_params(
 282         k, self.probs, base_dtype=self.dtype.base_dtype)
 283 
 284     # batch-flatten everything in order to use `sequence_mask()`.
 285     batch_flattened_probs = array_ops.reshape(probs,
 286                                               (-1, self._event_size))
 287     batch_flattened_k = array_ops.reshape(k, [-1])
 288 
 289     to_sum_over = array_ops.where(
 290         array_ops.sequence_mask(batch_flattened_k, self._event_size),
 291         batch_flattened_probs,
 292         array_ops.zeros_like(batch_flattened_probs))
 293     batch_flattened_cdf = math_ops.reduce_sum(to_sum_over, axis=-1)
 294     # Reshape back to the shape of the argument.
 295     return array_ops.reshape(batch_flattened_cdf, array_ops.shape(k))
 296 
 297   def _log_prob(self, k):
 298     k = ops.convert_to_tensor(k, name="k")
 299     if self.validate_args:
 300       k = distribution_util.embed_check_integer_casting_closed(
 301           k, target_dtype=dtypes.int32)
 302     k, logits = _broadcast_cat_event_and_params(
 303         k, self.logits, base_dtype=self.dtype.base_dtype)
 304 
 305     return -nn_ops.sparse_softmax_cross_entropy_with_logits(labels=k,
 306                                                             logits=logits)
 307 
 308   def _entropy(self):
 309     return -math_ops.reduce_sum(
 310         nn_ops.log_softmax(self.logits) * self.probs, axis=-1)
 311 
 312   def _mode(self):
 313     ret = math_ops.argmax(self.logits, axis=self._batch_rank)
 314     ret = math_ops.cast(ret, self.dtype)
 315     ret.set_shape(self.batch_shape)
 316     return ret

 319 @kullback_leibler.RegisterKL(Categorical, Categorical)
 320 def _kl_categorical_categorical(a, b, name=None):
 321   """Calculate the batched KL divergence KL(a || b) with a and b Categorical.
 322 
 323   Args:
 324     a: instance of a Categorical distribution object.
 325     b: instance of a Categorical distribution object.
 326     name: (optional) Name to use for created operations.
 327       default is "kl_categorical_categorical".
 328 
 329   Returns:
 330     Batchwise KL(a || b)
 331   """
 332   with ops.name_scope(name, "kl_categorical_categorical",
 333                       values=[a.logits, b.logits]):
 334     # sum(probs log(probs / (1 - probs)))
 335     delta_log_probs1 = (nn_ops.log_softmax(a.logits) -
 336                         nn_ops.log_softmax(b.logits))
 337     return math_ops.reduce_sum(nn_ops.softmax(a.logits) * delta_log_probs1,
 338                                axis=-1)
