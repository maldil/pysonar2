  45 @tf_export("distributions.StudentT")
  46 class StudentT(distribution.Distribution):
  47   """Student's t-distribution.
  48 
  49   This distribution has parameters: degree of freedom `df`, location `loc`,
  50   and `scale`.
  51 
  52   #### Mathematical details
  53 
  54   The probability density function (pdf) is,
  55 
  56   ```none
  57   pdf(x; df, mu, sigma) = (1 + y**2 / df)**(-0.5 (df + 1)) / Z
  58   where,
  59   y = (x - mu) / sigma
  60   Z = abs(sigma) sqrt(df pi) Gamma(0.5 df) / Gamma(0.5 (df + 1))
  61   ```
  62 
  63   where:
  64   * `loc = mu`,
  65   * `scale = sigma`, and,
  66   * `Z` is the normalization constant, and,
  67   * `Gamma` is the [gamma function](
  68     https://en.wikipedia.org/wiki/Gamma_function).
  69 
  70   The StudentT distribution is a member of the [location-scale family](
  71   https://en.wikipedia.org/wiki/Location-scale_family), i.e., it can be
  72   constructed as,
  73 
  74   ```none
  75   X ~ StudentT(df, loc=0, scale=1)
  76   Y = loc + scale * X
  77   ```
  78 
  79   Notice that `scale` has semantics more similar to standard deviation than
  80   variance. However it is not actually the std. deviation; the Student's
  81   t-distribution std. dev. is `scale sqrt(df / (df - 2))` when `df > 2`.
  82 
  83   Samples of this distribution are reparameterized (pathwise differentiable).
  84   The derivatives are computed using the approach described in the paper
  85 
  86   [Michael Figurnov, Shakir Mohamed, Andriy Mnih.
  87   Implicit Reparameterization Gradients, 2018](https://arxiv.org/abs/1805.08498)
  88 
  89   #### Examples
  90 
  91   Examples of initialization of one or a batch of distributions.
  92 
  93   ```python
  94   import tensorflow_probability as tfp
  95   tfd = tfp.distributions
  96 
  97   # Define a single scalar Student t distribution.
  98   single_dist = tfd.StudentT(df=3)
  99 
 100   # Evaluate the pdf at 1, returning a scalar Tensor.
 101   single_dist.prob(1.)
 102 
 103   # Define a batch of two scalar valued Student t's.
 104   # The first has degrees of freedom 2, mean 1, and scale 11.
 105   # The second 3, 2 and 22.
 106   multi_dist = tfd.StudentT(df=[2, 3], loc=[1, 2.], scale=[11, 22.])
 107 
 108   # Evaluate the pdf of the first distribution on 0, and the second on 1.5,
 109   # returning a length two tensor.
 110   multi_dist.prob([0, 1.5])
 111 
 112   # Get 3 samples, returning a 3 x 2 tensor.
 113   multi_dist.sample(3)
 114   ```
 115 
 116   Arguments are broadcast when possible.
 117 
 118   ```python
 119   # Define a batch of two Student's t distributions.
 120   # Both have df 2 and mean 1, but different scales.
 121   dist = tfd.StudentT(df=2, loc=1, scale=[11, 22.])
 122 
 123   # Evaluate the pdf of both distributions on the same point, 3.0,
 124   # returning a length 2 tensor.
 125   dist.prob(3.0)
 126   ```
 127 
 128   Compute the gradients of samples w.r.t. the parameters:
 129 
 130   ```python
 131   df = tf.constant(2.0)
 132   loc = tf.constant(2.0)
 133   scale = tf.constant(11.0)
 134   dist = tfd.StudentT(df=df, loc=loc, scale=scale)
 135   samples = dist.sample(5)  # Shape [5]
 136   loss = tf.reduce_mean(tf.square(samples))  # Arbitrary loss function
 137   # Unbiased stochastic gradients of the loss function
 138   grads = tf.gradients(loss, [df, loc, scale])
 139   ```
 140 
 141   """
 142 
 143   def __init__(self,
 144                df,
 145                loc,
 146                scale,
 147                validate_args=False,
 148                allow_nan_stats=True,
 149                name="StudentT"):
 150     """Construct Student's t distributions.
 151 
 152     The distributions have degree of freedom `df`, mean `loc`, and scale
 153     `scale`.
 154 
 155     The parameters `df`, `loc`, and `scale` must be shaped in a way that
 156     supports broadcasting (e.g. `df + loc + scale` is a valid operation).
 157 
 158     Args:
 159       df: Floating-point `Tensor`. The degrees of freedom of the
 160         distribution(s). `df` must contain only positive values.
 161       loc: Floating-point `Tensor`. The mean(s) of the distribution(s).
 162       scale: Floating-point `Tensor`. The scaling factor(s) for the
 163         distribution(s). Note that `scale` is not technically the standard
 164         deviation of this distribution but has semantics more similar to
 165         standard deviation than variance.
 166       validate_args: Python `bool`, default `False`. When `True` distribution
 167         parameters are checked for validity despite possibly degrading runtime
 168         performance. When `False` invalid inputs may silently render incorrect
 169         outputs.
 170       allow_nan_stats: Python `bool`, default `True`. When `True`,
 171         statistics (e.g., mean, mode, variance) use the value "`NaN`" to
 172         indicate the result is undefined. When `False`, an exception is raised
 173         if one or more of the statistic's batch members are undefined.
 174       name: Python `str` name prefixed to Ops created by this class.
 175 
 176     Raises:
 177       TypeError: if loc and scale are different dtypes.
 178     """
 179     parameters = dict(locals())
 180     with ops.name_scope(name, values=[df, loc, scale]) as name:
 181       with ops.control_dependencies([check_ops.assert_positive(df)]
 182                                     if validate_args else []):
 183         self._df = array_ops.identity(df, name="df")
 184         self._loc = array_ops.identity(loc, name="loc")
 185         self._scale = array_ops.identity(scale, name="scale")
 186         check_ops.assert_same_float_dtype(
 187             (self._df, self._loc, self._scale))
 188     super(StudentT, self).__init__(
 189         dtype=self._scale.dtype,
 190         reparameterization_type=distribution.FULLY_REPARAMETERIZED,
 191         validate_args=validate_args,
 192         allow_nan_stats=allow_nan_stats,
 193         parameters=parameters,
 194         graph_parents=[self._df, self._loc, self._scale],
 195         name=name)
 196 
 197   @staticmethod
 198   def _param_shapes(sample_shape):
 199     return dict(
 200         zip(("df", "loc", "scale"), (
 201             [ops.convert_to_tensor(
 202                 sample_shape, dtype=dtypes.int32)] * 3)))
 203 
 204   @property
 205   def df(self):
 206     """Degrees of freedom in these Student's t distribution(s)."""
 207     return self._df

 212     return self._loc

 215   def scale(self):

 219   def _batch_shape_tensor(self):
 220     return array_ops.broadcast_dynamic_shape(
 221         array_ops.shape(self.df),
 222         array_ops.broadcast_dynamic_shape(
 223             array_ops.shape(self.loc), array_ops.shape(self.scale)))
 224 
 225   def _batch_shape(self):
 226     return array_ops.broadcast_static_shape(
 227         array_ops.broadcast_static_shape(self.df.get_shape(),
 228                                          self.loc.get_shape()),
 229         self.scale.get_shape())
 230 
 231   def _event_shape_tensor(self):
 232     return constant_op.constant([], dtype=math_ops.int32)
 233 
 234   def _event_shape(self):
 235     return tensor_shape.scalar()
 236 
 237   def _sample_n(self, n, seed=None):
 238     # The sampling method comes from the fact that if:
 239     #   X ~ Normal(0, 1)
 240     #   Z ~ Chi2(df)
 241     #   Y = X / sqrt(Z / df)
 242     # then:
 243     #   Y ~ StudentT(df).
 244     shape = array_ops.concat([[n], self.batch_shape_tensor()], 0)
 245     normal_sample = random_ops.random_normal(shape, dtype=self.dtype, seed=seed)
 246     df = self.df * array_ops.ones(self.batch_shape_tensor(), dtype=self.dtype)
 247     gamma_sample = random_ops.random_gamma(
 248         [n],
 249         0.5 * df,
 250         beta=0.5,
 251         dtype=self.dtype,
 252         seed=distribution_util.gen_new_seed(seed, salt="student_t"))
 253     samples = normal_sample * math_ops.rsqrt(gamma_sample / df)
 254     return samples * self.scale + self.loc  # Abs(scale) not wanted.
 255 
 256   def _log_prob(self, x):
 257     return self._log_unnormalized_prob(x) - self._log_normalization()
 258 
 259   def _log_unnormalized_prob(self, x):
 260     y = (x - self.loc) / self.scale  # Abs(scale) superfluous.
 261     return -0.5 * (self.df + 1.) * math_ops.log1p(y**2. / self.df)
 262 
 263   def _log_normalization(self):
 264     return (math_ops.log(math_ops.abs(self.scale)) +
 265             0.5 * math_ops.log(self.df) +
 266             0.5 * np.log(np.pi) +
 267             math_ops.lgamma(0.5 * self.df) -
 268             math_ops.lgamma(0.5 * (self.df + 1.)))
 269 
 270   def _cdf(self, x):
 271     # Take Abs(scale) to make subsequent where work correctly.
 272     y = (x - self.loc) / math_ops.abs(self.scale)
 273     x_t = self.df / (y**2. + self.df)
 274     neg_cdf = 0.5 * math_ops.betainc(0.5 * self.df, 0.5, x_t)
 275     return array_ops.where(math_ops.less(y, 0.), neg_cdf, 1. - neg_cdf)
 276 
 277   def _entropy(self):
 278     v = array_ops.ones(self.batch_shape_tensor(),
 279                        dtype=self.dtype)[..., array_ops.newaxis]
 280     u = v * self.df[..., array_ops.newaxis]
 281     beta_arg = array_ops.concat([u, v], -1) / 2.
 282     return (math_ops.log(math_ops.abs(self.scale)) +
 283             0.5 * math_ops.log(self.df) +
 284             special_math_ops.lbeta(beta_arg) +
 285             0.5 * (self.df + 1.) *
 286             (math_ops.digamma(0.5 * (self.df + 1.)) -
 287              math_ops.digamma(0.5 * self.df)))
 288 
 289   @distribution_util.AppendDocstring(
 290       """The mean of Student's T equals `loc` if `df > 1`, otherwise it is
 291       `NaN`. If `self.allow_nan_stats=True`, then an exception will be raised
 292       rather than returning `NaN`.""")
 293   def _mean(self):
 294     mean = self.loc * array_ops.ones(self.batch_shape_tensor(),
 295                                      dtype=self.dtype)
 296     if self.allow_nan_stats:
 297       nan = np.array(np.nan, dtype=self.dtype.as_numpy_dtype())
 298       return array_ops.where(
 299           math_ops.greater(
 300               self.df,
 301               array_ops.ones(self.batch_shape_tensor(), dtype=self.dtype)),
 302           mean,
 303           array_ops.fill(self.batch_shape_tensor(), nan, name="nan"))
 304     else:
 305       return control_flow_ops.with_dependencies(
 306           [
 307               check_ops.assert_less(
 308                   array_ops.ones([], dtype=self.dtype),
 309                   self.df,
 310                   message="mean not defined for components of df <= 1"),
 311           ],
 312           mean)
 313 
 314   @distribution_util.AppendDocstring("""
 315       The variance for Student's T equals
 316 
 317       ```
 318       df / (df - 2), when df > 2
 319       infinity, when 1 < df <= 2
 320       NaN, when df <= 1
 321       ```
 322       """)
 323   def _variance(self):
 324     # We need to put the tf.where inside the outer tf.where to ensure we never
 325     # hit a NaN in the gradient.
 326     denom = array_ops.where(math_ops.greater(self.df, 2.),
 327                             self.df - 2.,
 328                             array_ops.ones_like(self.df))
 329     # Abs(scale) superfluous.
 330     var = (array_ops.ones(self.batch_shape_tensor(), dtype=self.dtype) *
 331            math_ops.square(self.scale) * self.df / denom)
 332     # When 1 < df <= 2, variance is infinite.
 333     inf = np.array(np.inf, dtype=self.dtype.as_numpy_dtype())
 334     result_where_defined = array_ops.where(
 335         self.df > array_ops.fill(self.batch_shape_tensor(), 2.),
 336         var,
 337         array_ops.fill(self.batch_shape_tensor(), inf, name="inf"))
 338 
 339     if self.allow_nan_stats:
 340       nan = np.array(np.nan, dtype=self.dtype.as_numpy_dtype())
 341       return array_ops.where(
 342           math_ops.greater(
 343               self.df,
 344               array_ops.ones(self.batch_shape_tensor(), dtype=self.dtype)),
 345           result_where_defined,
 346           array_ops.fill(self.batch_shape_tensor(), nan, name="nan"))
 347     else:
 348       return control_flow_ops.with_dependencies(
 349           [
 350               check_ops.assert_less(
 351                   array_ops.ones([], dtype=self.dtype),
 352                   self.df,
 353                   message="variance not defined for components of df <= 1"),
 354           ],
 355           result_where_defined)
 356 
 357   def _mode(self):
 358     return array_ops.identity(self.loc)

 361 class StudentTWithAbsDfSoftplusScale(StudentT):
 362   """StudentT with `df = floor(abs(df))` and `scale = softplus(scale)`."""
 363 
 364   def __init__(self,
 365                df,
 366                loc,
 367                scale,
 368                validate_args=False,
 369                allow_nan_stats=True,
 370                name="StudentTWithAbsDfSoftplusScale"):
 371     parameters = dict(locals())
 372     with ops.name_scope(name, values=[df, scale]) as name:
 373       super(StudentTWithAbsDfSoftplusScale, self).__init__(
 374           df=math_ops.floor(math_ops.abs(df)),
 375           loc=loc,
 376           scale=nn.softplus(scale, name="softplus_scale"),
 377           validate_args=validate_args,
 378           allow_nan_stats=allow_nan_stats,
 379           name=name)
 380     self._parameters = parameters
