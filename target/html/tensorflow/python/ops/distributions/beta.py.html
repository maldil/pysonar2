  49 @tf_export("distributions.Beta")
  50 class Beta(distribution.Distribution):
  51   """Beta distribution.
  52 
  53   The Beta distribution is defined over the `(0, 1)` interval using parameters
  54   `concentration1` (aka "alpha") and `concentration0` (aka "beta").
  55 
  56   #### Mathematical Details
  57 
  58   The probability density function (pdf) is,
  59 
  60   ```none
  61   pdf(x; alpha, beta) = x**(alpha - 1) (1 - x)**(beta - 1) / Z
  62   Z = Gamma(alpha) Gamma(beta) / Gamma(alpha + beta)
  63   ```
  64 
  65   where:
  66 
  67   * `concentration1 = alpha`,
  68   * `concentration0 = beta`,
  69   * `Z` is the normalization constant, and,
  70   * `Gamma` is the [gamma function](
  71     https://en.wikipedia.org/wiki/Gamma_function).
  72 
  73   The concentration parameters represent mean total counts of a `1` or a `0`,
  74   i.e.,
  75 
  76   ```none
  77   concentration1 = alpha = mean * total_concentration
  78   concentration0 = beta  = (1. - mean) * total_concentration
  79   ```
  80 
  81   where `mean` in `(0, 1)` and `total_concentration` is a positive real number
  82   representing a mean `total_count = concentration1 + concentration0`.
  83 
  84   Distribution parameters are automatically broadcast in all functions; see
  85   examples for details.
  86 
  87   Warning: The samples can be zero due to finite precision.
  88   This happens more often when some of the concentrations are very small.
  89   Make sure to round the samples to `np.finfo(dtype).tiny` before computing the
  90   density.
  91 
  92   Samples of this distribution are reparameterized (pathwise differentiable).
  93   The derivatives are computed using the approach described in the paper
  94 
  95   [Michael Figurnov, Shakir Mohamed, Andriy Mnih.
  96   Implicit Reparameterization Gradients, 2018](https://arxiv.org/abs/1805.08498)
  97 
  98   #### Examples
  99 
 100   ```python
 101   import tensorflow_probability as tfp
 102   tfd = tfp.distributions
 103 
 104   # Create a batch of three Beta distributions.
 105   alpha = [1, 2, 3]
 106   beta = [1, 2, 3]
 107   dist = tfd.Beta(alpha, beta)
 108 
 109   dist.sample([4, 5])  # Shape [4, 5, 3]
 110 
 111   # `x` has three batch entries, each with two samples.
 112   x = [[.1, .4, .5],
 113        [.2, .3, .5]]
 114   # Calculate the probability of each pair of samples under the corresponding
 115   # distribution in `dist`.
 116   dist.prob(x)         # Shape [2, 3]
 117   ```
 118 
 119   ```python
 120   # Create batch_shape=[2, 3] via parameter broadcast:
 121   alpha = [[1.], [2]]      # Shape [2, 1]
 122   beta = [3., 4, 5]        # Shape [3]
 123   dist = tfd.Beta(alpha, beta)
 124 
 125   # alpha broadcast as: [[1., 1, 1,],
 126   #                      [2, 2, 2]]
 127   # beta broadcast as:  [[3., 4, 5],
 128   #                      [3, 4, 5]]
 129   # batch_Shape [2, 3]
 130   dist.sample([4, 5])  # Shape [4, 5, 2, 3]
 131 
 132   x = [.2, .3, .5]
 133   # x will be broadcast as [[.2, .3, .5],
 134   #                         [.2, .3, .5]],
 135   # thus matching batch_shape [2, 3].
 136   dist.prob(x)         # Shape [2, 3]
 137   ```
 138 
 139   Compute the gradients of samples w.r.t. the parameters:
 140 
 141   ```python
 142   alpha = tf.constant(1.0)
 143   beta = tf.constant(2.0)
 144   dist = tfd.Beta(alpha, beta)
 145   samples = dist.sample(5)  # Shape [5]
 146   loss = tf.reduce_mean(tf.square(samples))  # Arbitrary loss function
 147   # Unbiased stochastic gradients of the loss function
 148   grads = tf.gradients(loss, [alpha, beta])
 149   ```
 150 
 151   """
 152 
 153   def __init__(self,
 154                concentration1=None,
 155                concentration0=None,
 156                validate_args=False,
 157                allow_nan_stats=True,
 158                name="Beta"):
 159     """Initialize a batch of Beta distributions.
 160 
 161     Args:
 162       concentration1: Positive floating-point `Tensor` indicating mean
 163         number of successes; aka "alpha". Implies `self.dtype` and
 164         `self.batch_shape`, i.e.,
 165         `concentration1.shape = [N1, N2, ..., Nm] = self.batch_shape`.
 166       concentration0: Positive floating-point `Tensor` indicating mean
 167         number of failures; aka "beta". Otherwise has same semantics as
 168         `concentration1`.
 169       validate_args: Python `bool`, default `False`. When `True` distribution
 170         parameters are checked for validity despite possibly degrading runtime
 171         performance. When `False` invalid inputs may silently render incorrect
 172         outputs.
 173       allow_nan_stats: Python `bool`, default `True`. When `True`, statistics
 174         (e.g., mean, mode, variance) use the value "`NaN`" to indicate the
 175         result is undefined. When `False`, an exception is raised if one or
 176         more of the statistic's batch members are undefined.
 177       name: Python `str` name prefixed to Ops created by this class.
 178     """
 179     parameters = dict(locals())
 180     with ops.name_scope(name, values=[concentration1, concentration0]) as name:
 181       self._concentration1 = self._maybe_assert_valid_concentration(
 182           ops.convert_to_tensor(concentration1, name="concentration1"),
 183           validate_args)
 184       self._concentration0 = self._maybe_assert_valid_concentration(
 185           ops.convert_to_tensor(concentration0, name="concentration0"),
 186           validate_args)
 187       check_ops.assert_same_float_dtype([
 188           self._concentration1, self._concentration0])
 189       self._total_concentration = self._concentration1 + self._concentration0
 190     super(Beta, self).__init__(
 191         dtype=self._total_concentration.dtype,
 192         validate_args=validate_args,
 193         allow_nan_stats=allow_nan_stats,
 194         reparameterization_type=distribution.FULLY_REPARAMETERIZED,
 195         parameters=parameters,
 196         graph_parents=[self._concentration1,
 197                        self._concentration0,
 198                        self._total_concentration],
 199         name=name)
 200 
 201   @staticmethod
 202   def _param_shapes(sample_shape):
 203     return dict(zip(
 204         ["concentration1", "concentration0"],
 205         [ops.convert_to_tensor(sample_shape, dtype=dtypes.int32)] * 2))
 206 
 207   @property
 208   def concentration1(self):
 209     """Concentration parameter associated with a `1` outcome."""
 210     return self._concentration1
 211 
 212   @property
 213   def concentration0(self):
 214     """Concentration parameter associated with a `0` outcome."""
 215     return self._concentration0
 216 
 217   @property
 218   def total_concentration(self):
 219     """Sum of concentration parameters."""
 220     return self._total_concentration
 221 
 222   def _batch_shape_tensor(self):
 223     return array_ops.shape(self.total_concentration)
 224 
 225   def _batch_shape(self):
 226     return self.total_concentration.get_shape()
 227 
 228   def _event_shape_tensor(self):
 229     return constant_op.constant([], dtype=dtypes.int32)
 230 
 231   def _event_shape(self):
 232     return tensor_shape.scalar()
 233 
 234   def _sample_n(self, n, seed=None):
 235     expanded_concentration1 = array_ops.ones_like(
 236         self.total_concentration, dtype=self.dtype) * self.concentration1
 237     expanded_concentration0 = array_ops.ones_like(
 238         self.total_concentration, dtype=self.dtype) * self.concentration0
 239     gamma1_sample = random_ops.random_gamma(
 240         shape=[n],
 241         alpha=expanded_concentration1,
 242         dtype=self.dtype,
 243         seed=seed)
 244     gamma2_sample = random_ops.random_gamma(
 245         shape=[n],
 246         alpha=expanded_concentration0,
 247         dtype=self.dtype,
 248         seed=distribution_util.gen_new_seed(seed, "beta"))
 249     beta_sample = gamma1_sample / (gamma1_sample + gamma2_sample)
 250     return beta_sample

 252   @distribution_util.AppendDocstring(_beta_sample_note)
 253   def _log_prob(self, x):
 254     return self._log_unnormalized_prob(x) - self._log_normalization()
 255 
 256   @distribution_util.AppendDocstring(_beta_sample_note)
 257   def _prob(self, x):
 258     return math_ops.exp(self._log_prob(x))
 259 
 260   @distribution_util.AppendDocstring(_beta_sample_note)
 261   def _log_cdf(self, x):
 262     return math_ops.log(self._cdf(x))
 263 
 264   @distribution_util.AppendDocstring(_beta_sample_note)
 265   def _cdf(self, x):
 266     return math_ops.betainc(self.concentration1, self.concentration0, x)
 267 
 268   def _log_unnormalized_prob(self, x):
 269     x = self._maybe_assert_valid_sample(x)
 270     return ((self.concentration1 - 1.) * math_ops.log(x)
 271             + (self.concentration0 - 1.) * math_ops.log1p(-x))
 272 
 273   def _log_normalization(self):
 274     return (math_ops.lgamma(self.concentration1)
 275             + math_ops.lgamma(self.concentration0)
 276             - math_ops.lgamma(self.total_concentration))
 277 
 278   def _entropy(self):
 279     return (
 280         self._log_normalization()
 281         - (self.concentration1 - 1.) * math_ops.digamma(self.concentration1)
 282         - (self.concentration0 - 1.) * math_ops.digamma(self.concentration0)
 283         + ((self.total_concentration - 2.) *
 284            math_ops.digamma(self.total_concentration)))
 285 
 286   def _mean(self):
 287     return self._concentration1 / self._total_concentration
 288 
 289   def _variance(self):
 290     return self._mean() * (1. - self._mean()) / (1. + self.total_concentration)
 291 
 292   @distribution_util.AppendDocstring(
 293       """Note: The mode is undefined when `concentration1 <= 1` or
 294       `concentration0 <= 1`. If `self.allow_nan_stats` is `True`, `NaN`
 295       is used for undefined modes. If `self.allow_nan_stats` is `False` an
 296       exception is raised when one or more modes are undefined.""")
 297   def _mode(self):
 298     mode = (self.concentration1 - 1.) / (self.total_concentration - 2.)
 299     if self.allow_nan_stats:
 300       nan = array_ops.fill(
 301           self.batch_shape_tensor(),
 302           np.array(np.nan, dtype=self.dtype.as_numpy_dtype()),
 303           name="nan")
 304       is_defined = math_ops.logical_and(self.concentration1 > 1.,
 305                                         self.concentration0 > 1.)
 306       return array_ops.where(is_defined, mode, nan)
 307     return control_flow_ops.with_dependencies([
 308         check_ops.assert_less(
 309             array_ops.ones([], dtype=self.dtype),
 310             self.concentration1,
 311             message="Mode undefined for concentration1 <= 1."),
 312         check_ops.assert_less(
 313             array_ops.ones([], dtype=self.dtype),
 314             self.concentration0,
 315             message="Mode undefined for concentration0 <= 1.")
 316     ], mode)
 317 
 318   def _maybe_assert_valid_concentration(self, concentration, validate_args):
 319     """Checks the validity of a concentration parameter."""
 320     if not validate_args:
 321       return concentration
 322     return control_flow_ops.with_dependencies([
 323         check_ops.assert_positive(
 324             concentration,
 325             message="Concentration parameter must be positive."),
 326     ], concentration)
 327 
 328   def _maybe_assert_valid_sample(self, x):
 329     """Checks the validity of a sample."""
 330     if not self.validate_args:
 331       return x
 332     return control_flow_ops.with_dependencies([
 333         check_ops.assert_positive(x, message="sample must be positive"),
 334         check_ops.assert_less(
 335             x,
 336             array_ops.ones([], self.dtype),
 337             message="sample must be less than `1`."),
 338     ], x)

 341 class BetaWithSoftplusConcentration(Beta):
 342   """Beta with softplus transform of `concentration1` and `concentration0`."""
 343 
 344   def __init__(self,
 345                concentration1,
 346                concentration0,
 347                validate_args=False,
 348                allow_nan_stats=True,
 349                name="BetaWithSoftplusConcentration"):
 350     parameters = dict(locals())
 351     with ops.name_scope(name, values=[concentration1,
 352                                       concentration0]) as name:
 353       super(BetaWithSoftplusConcentration, self).__init__(
 354           concentration1=nn.softplus(concentration1,
 355                                      name="softplus_concentration1"),
 356           concentration0=nn.softplus(concentration0,
 357                                      name="softplus_concentration0"),
 358           validate_args=validate_args,
 359           allow_nan_stats=allow_nan_stats,
 360           name=name)
 361     self._parameters = parameters

 364 @kullback_leibler.RegisterKL(Beta, Beta)
 365 def _kl_beta_beta(d1, d2, name=None):
 366   """Calculate the batchwise KL divergence KL(d1 || d2) with d1 and d2 Beta.
 367 
 368   Args:
 369     d1: instance of a Beta distribution object.
 370     d2: instance of a Beta distribution object.
 371     name: (optional) Name to use for created operations.
 372       default is "kl_beta_beta".
 373 
 374   Returns:
 375     Batchwise KL(d1 || d2)
 376   """
 377   def delta(fn, is_property=True):
 378     fn1 = getattr(d1, fn)
 379     fn2 = getattr(d2, fn)
 380     return (fn2 - fn1) if is_property else (fn2() - fn1())
 381   with ops.name_scope(name, "kl_beta_beta", values=[
 382       d1.concentration1,
 383       d1.concentration0,
 384       d1.total_concentration,
 385       d2.concentration1,
 386       d2.concentration0,
 387       d2.total_concentration,
 388   ]):
 389     return (delta("_log_normalization", is_property=False)
 390             - math_ops.digamma(d1.concentration1) * delta("concentration1")
 391             - math_ops.digamma(d1.concentration0) * delta("concentration0")
 392             + (math_ops.digamma(d1.total_concentration)
 393                * delta("total_concentration")))
