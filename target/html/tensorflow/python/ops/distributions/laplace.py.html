  45 @tf_export("distributions.Laplace")
  46 class Laplace(distribution.Distribution):
  47   """The Laplace distribution with location `loc` and `scale` parameters.
  48 
  49   #### Mathematical details
  50 
  51   The probability density function (pdf) of this distribution is,
  52 
  53   ```none
  54   pdf(x; mu, sigma) = exp(-|x - mu| / sigma) / Z
  55   Z = 2 sigma
  56   ```
  57 
  58   where `loc = mu`, `scale = sigma`, and `Z` is the normalization constant.
  59 
  60   Note that the Laplace distribution can be thought of two exponential
  61   distributions spliced together "back-to-back."
  62 
  63   The Lpalce distribution is a member of the [location-scale family](
  64   https://en.wikipedia.org/wiki/Location-scale_family), i.e., it can be
  65   constructed as,
  66 
  67   ```none
  68   X ~ Laplace(loc=0, scale=1)
  69   Y = loc + scale * X
  70   ```
  71 
  72   """
  73 
  74   def __init__(self,
  75                loc,
  76                scale,
  77                validate_args=False,
  78                allow_nan_stats=True,
  79                name="Laplace"):
  80     """Construct Laplace distribution with parameters `loc` and `scale`.
  81 
  82     The parameters `loc` and `scale` must be shaped in a way that supports
  83     broadcasting (e.g., `loc / scale` is a valid operation).
  84 
  85     Args:
  86       loc: Floating point tensor which characterizes the location (center)
  87         of the distribution.
  88       scale: Positive floating point tensor which characterizes the spread of
  89         the distribution.
  90       validate_args: Python `bool`, default `False`. When `True` distribution
  91         parameters are checked for validity despite possibly degrading runtime
  92         performance. When `False` invalid inputs may silently render incorrect
  93         outputs.
  94       allow_nan_stats: Python `bool`, default `True`. When `True`,
  95         statistics (e.g., mean, mode, variance) use the value "`NaN`" to
  96         indicate the result is undefined. When `False`, an exception is raised
  97         if one or more of the statistic's batch members are undefined.
  98       name: Python `str` name prefixed to Ops created by this class.
  99 
 100     Raises:
 101       TypeError: if `loc` and `scale` are of different dtype.
 102     """
 103     parameters = dict(locals())
 104     with ops.name_scope(name, values=[loc, scale]) as name:
 105       with ops.control_dependencies([check_ops.assert_positive(scale)] if
 106                                     validate_args else []):
 107         self._loc = array_ops.identity(loc, name="loc")
 108         self._scale = array_ops.identity(scale, name="scale")
 109         check_ops.assert_same_float_dtype([self._loc, self._scale])
 110       super(Laplace, self).__init__(
 111           dtype=self._loc.dtype,
 112           reparameterization_type=distribution.FULLY_REPARAMETERIZED,
 113           validate_args=validate_args,
 114           allow_nan_stats=allow_nan_stats,
 115           parameters=parameters,
 116           graph_parents=[self._loc, self._scale],
 117           name=name)
 118 
 119   @staticmethod
 120   def _param_shapes(sample_shape):
 121     return dict(
 122         zip(("loc", "scale"), ([ops.convert_to_tensor(
 123             sample_shape, dtype=dtypes.int32)] * 2)))
 124 
 125   @property
 126   def loc(self):
 127     """Distribution parameter for the location."""
 128     return self._loc
 129 
 130   @property
 131   def scale(self):
 132     """Distribution parameter for scale."""
 133     return self._scale
 134 
 135   def _batch_shape_tensor(self):
 136     return array_ops.broadcast_dynamic_shape(
 137         array_ops.shape(self.loc), array_ops.shape(self.scale))
 138 
 139   def _batch_shape(self):
 140     return array_ops.broadcast_static_shape(
 141         self.loc.get_shape(), self.scale.get_shape())
 142 
 143   def _event_shape_tensor(self):
 144     return constant_op.constant([], dtype=dtypes.int32)
 145 
 146   def _event_shape(self):
 147     return tensor_shape.scalar()
 148 
 149   def _sample_n(self, n, seed=None):
 150     shape = array_ops.concat([[n], self.batch_shape_tensor()], 0)
 151     # Uniform variates must be sampled from the open-interval `(-1, 1)` rather
 152     # than `[-1, 1)`. In the case of `(0, 1)` we'd use
 153     # `np.finfo(self.dtype.as_numpy_dtype).tiny` because it is the smallest,
 154     # positive, "normal" number. However, the concept of subnormality exists
 155     # only at zero; here we need the smallest usable number larger than -1,
 156     # i.e., `-1 + eps/2`.
 157     uniform_samples = random_ops.random_uniform(
 158         shape=shape,
 159         minval=np.nextafter(self.dtype.as_numpy_dtype(-1.),
 160                             self.dtype.as_numpy_dtype(0.)),
 161         maxval=1.,
 162         dtype=self.dtype,
 163         seed=seed)
 164     return (self.loc - self.scale * math_ops.sign(uniform_samples) *
 165             math_ops.log1p(-math_ops.abs(uniform_samples)))
 166 
 167   def _log_prob(self, x):
 168     return self._log_unnormalized_prob(x) - self._log_normalization()
 169 
 170   def _prob(self, x):
 171     return math_ops.exp(self._log_prob(x))
 172 
 173   def _log_cdf(self, x):
 174     return special_math.log_cdf_laplace(self._z(x))
 175 
 176   def _log_survival_function(self, x):
 177     return special_math.log_cdf_laplace(-self._z(x))
 178 
 179   def _cdf(self, x):
 180     z = self._z(x)
 181     return (0.5 + 0.5 * math_ops.sign(z) *
 182             (1. - math_ops.exp(-math_ops.abs(z))))
 183 
 184   def _log_unnormalized_prob(self, x):
 185     return -math_ops.abs(self._z(x))
 186 
 187   def _log_normalization(self):
 188     return math.log(2.) + math_ops.log(self.scale)
 189 
 190   def _entropy(self):
 191     # Use broadcasting rules to calculate the full broadcast scale.
 192     scale = self.scale + array_ops.zeros_like(self.loc)
 193     return math.log(2.) + 1. + math_ops.log(scale)
 194 
 195   def _mean(self):
 196     return self.loc + array_ops.zeros_like(self.scale)
 197 
 198   def _stddev(self):
 199     return math.sqrt(2.) * self.scale + array_ops.zeros_like(self.loc)
 200 
 201   def _median(self):
 202     return self._mean()
 203 
 204   def _mode(self):
 205     return self._mean()
 206 
 207   def _z(self, x):
 208     return (x - self.loc) / self.scale

 211 class LaplaceWithSoftplusScale(Laplace):
 212   """Laplace with softplus applied to `scale`."""
 213 
 214   def __init__(self,
 215                loc,
 216                scale,
 217                validate_args=False,
 218                allow_nan_stats=True,
 219                name="LaplaceWithSoftplusScale"):
 220     parameters = dict(locals())
 221     with ops.name_scope(name, values=[loc, scale]) as name:
 222       super(LaplaceWithSoftplusScale, self).__init__(
 223           loc=loc,
 224           scale=nn.softplus(scale, name="softplus_scale"),
 225           validate_args=validate_args,
 226           allow_nan_stats=allow_nan_stats,
 227           name=name)
 228     self._parameters = parameters
