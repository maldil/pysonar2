  32 @tf_export('histogram_fixed_width_bins')
  33 def histogram_fixed_width_bins(values,
  34                                value_range,
  35                                nbins=100,
  36                                dtype=dtypes.int32,
  37                                name=None):
  38   """Bins the given values for use in a histogram.
  39 
  40   Given the tensor `values`, this operation returns a rank 1 `Tensor`
  41   representing the indices of a histogram into which each element
  42   of `values` would be binned. The bins are equal width and
  43   determined by the arguments `value_range` and `nbins`.
  44 
  45   Args:
  46     values:  Numeric `Tensor`.
  47     value_range:  Shape [2] `Tensor` of same `dtype` as `values`.
  48       values <= value_range[0] will be mapped to hist[0],
  49       values >= value_range[1] will be mapped to hist[-1].
  50     nbins:  Scalar `int32 Tensor`.  Number of histogram bins.
  51     dtype:  dtype for returned histogram.
  52     name:  A name for this operation (defaults to 'histogram_fixed_width').
  53 
  54   Returns:
  55     A `Tensor` holding the indices of the binned values whose shape matches
  56     `values`.
  57 
  58   Examples:
  59 
  60   ```python
  61   # Bins will be:  (-inf, 1), [1, 2), [2, 3), [3, 4), [4, inf)
  62   nbins = 5
  63   value_range = [0.0, 5.0]
  64   new_values = [-1.0, 0.0, 1.5, 2.0, 5.0, 15]
  65 
  66   with tf.get_default_session() as sess:
  67     indices = tf.histogram_fixed_width_bins(new_values, value_range, nbins=5)
  68     variables.global_variables_initializer().run()
  69     sess.run(indices) => [0, 0, 1, 2, 4]
  70   ```
  71   """
  72   with ops.name_scope(name, 'histogram_fixed_width_bins',
  73                       [values, value_range, nbins]):
  74     values = ops.convert_to_tensor(values, name='values')
  75     shape = array_ops.shape(values)
  76 
  77     values = array_ops.reshape(values, [-1])
  78     value_range = ops.convert_to_tensor(value_range, name='value_range')
  79     nbins = ops.convert_to_tensor(nbins, dtype=dtypes.int32, name='nbins')
  80     nbins_float = math_ops.cast(nbins, values.dtype)
  81 
  82     # Map tensor values that fall within value_range to [0, 1].
  83     scaled_values = math_ops.truediv(
  84         values - value_range[0],
  85         value_range[1] - value_range[0],
  86         name='scaled_values')
  87 
  88     # map tensor values within the open interval value_range to {0,.., nbins-1},
  89     # values outside the open interval will be zero or less, or nbins or more.
  90     indices = math_ops.floor(nbins_float * scaled_values, name='indices')
  91 
  92     # Clip edge cases (e.g. value = value_range[1]) or "outliers."
  93     indices = math_ops.cast(
  94         clip_ops.clip_by_value(indices, 0, nbins_float - 1), dtypes.int32)
  95     return array_ops.reshape(indices, shape)

  98 @tf_export('histogram_fixed_width')
  99 def histogram_fixed_width(values,
 100                           value_range,
 101                           nbins=100,
 102                           dtype=dtypes.int32,
 103                           name=None):
 104   """Return histogram of values.
 105 
 106   Given the tensor `values`, this operation returns a rank 1 histogram counting
 107   the number of entries in `values` that fell into every bin.  The bins are
 108   equal width and determined by the arguments `value_range` and `nbins`.
 109 
 110   Args:
 111     values:  Numeric `Tensor`.
 112     value_range:  Shape [2] `Tensor` of same `dtype` as `values`.
 113       values <= value_range[0] will be mapped to hist[0],
 114       values >= value_range[1] will be mapped to hist[-1].
 115     nbins:  Scalar `int32 Tensor`.  Number of histogram bins.
 116     dtype:  dtype for returned histogram.
 117     name:  A name for this operation (defaults to 'histogram_fixed_width').
 118 
 119   Returns:
 120     A 1-D `Tensor` holding histogram of values.
 121 
 122   Examples:
 123 
 124   ```python
 125   # Bins will be:  (-inf, 1), [1, 2), [2, 3), [3, 4), [4, inf)
 126   nbins = 5
 127   value_range = [0.0, 5.0]
 128   new_values = [-1.0, 0.0, 1.5, 2.0, 5.0, 15]
 129 
 130   with tf.get_default_session() as sess:
 131     hist = tf.histogram_fixed_width(new_values, value_range, nbins=5)
 132     variables.global_variables_initializer().run()
 133     sess.run(hist) => [2, 1, 1, 0, 2]
 134   ```
 135   """
 136   with ops.name_scope(name, 'histogram_fixed_width',
 137                       [values, value_range, nbins]) as name:
 138     # pylint: disable=protected-access
 139     return gen_math_ops._histogram_fixed_width(
 140         values, value_range, nbins, dtype=dtype, name=name)
 141     # pylint: enable=protected-access
