  33 def eye(num_rows,
  34         num_columns=None,
  35         batch_shape=None,
  36         dtype=dtypes.float32,
  37         name=None):
  38   """Construct an identity matrix, or a batch of matrices.
  39 
  40   See `linalg_ops.eye`.
  41   """
  42   with ops.name_scope(
  43       name, default_name='eye', values=[num_rows, num_columns, batch_shape]):
  44     is_square = num_columns is None
  45     batch_shape = [] if batch_shape is None else batch_shape
  46     num_columns = num_rows if num_columns is None else num_columns
  47     if isinstance(num_rows, ops.Tensor) or isinstance(
  48         num_columns, ops.Tensor) or isinstance(batch_shape, ops.Tensor):
  49       batch_shape = ops.convert_to_tensor(
  50           batch_shape, name='shape', dtype=dtypes.int32)
  51       diag_size = math_ops.minimum(num_rows, num_columns)
  52       diag_shape = array_ops.concat((batch_shape, [diag_size]), 0)
  53       if not is_square:
  54         shape = array_ops.concat((batch_shape, [num_rows, num_columns]), 0)
  55     else:
  56       if not isinstance(num_rows, compat.integral_types) or not isinstance(
  57           num_columns, compat.integral_types):
  58         raise TypeError(
  59             'num_rows and num_columns must be positive integer values.')
  60       batch_shape = [dim for dim in batch_shape]
  61       is_square = num_rows == num_columns
  62       diag_shape = batch_shape + [np.minimum(num_rows, num_columns)]
  63       if not is_square:
  64         shape = batch_shape + [num_rows, num_columns]
  65 
  66     diag_ones = array_ops.ones(diag_shape, dtype=dtype)
  67     if is_square:
  68       return array_ops.matrix_diag(diag_ones)
  69     else:
  70       zero_matrix = array_ops.zeros(shape, dtype=dtype)
  71       return array_ops.matrix_set_diag(zero_matrix, diag_ones)
