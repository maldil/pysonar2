  61   def __init__(self,
  62                spectrum,
  63                block_depth,
  64                input_output_dtype=_DTYPE_COMPLEX,
  65                is_non_singular=None,
  66                is_self_adjoint=None,
  67                is_positive_definite=None,
  68                is_square=True,
  69                name="LinearOperatorCirculant"):
  70     r"""Initialize an `_BaseLinearOperatorCirculant`.
  71 
  72     Args:
  73       spectrum:  Shape `[B1,...,Bb, N]` `Tensor`.  Allowed dtypes are
  74         `float32`, `complex64`.  Type can be different than `input_output_dtype`
  75       block_depth:  Python integer, either 1, 2, or 3.  Will be 1 for circulant,
  76         2 for block circulant, and 3 for nested block circulant.
  77       input_output_dtype: `dtype` for input/output.  Must be either
  78         `float32` or `complex64`.
  79       is_non_singular:  Expect that this operator is non-singular.
  80       is_self_adjoint:  Expect that this operator is equal to its hermitian
  81         transpose.  If `spectrum` is real, this will always be true.
  82       is_positive_definite:  Expect that this operator is positive definite,
  83         meaning the quadratic form `x^H A x` has positive real part for all
  84         nonzero `x`.  Note that we do not require the operator to be
  85         self-adjoint to be positive-definite.  See:
  86         https://en.wikipedia.org/wiki/Positive-definite_matrix\
  87             #Extension_for_non_symmetric_matrices
  88       is_square:  Expect that this operator acts like square [batch] matrices.
  89       name:  A name to prepend to all ops created by this class.
  90 
  91     Raises:
  92       ValueError:  If `block_depth` is not an allowed value.
  93       TypeError:  If `spectrum` is not an allowed type.
  94     """
  95 
  96     allowed_block_depths = [1, 2, 3]
  97 
  98     self._name = name
  99 
 100     if block_depth not in allowed_block_depths:
 101       raise ValueError("Expected block_depth to be in %s.  Found: %s." %
 102                        (allowed_block_depths, block_depth))
 103     self._block_depth = block_depth
 104 
 105     with ops.name_scope(name, values=[spectrum]):
 106       self._spectrum = self._check_spectrum_and_return_tensor(spectrum)
 107 
 108       # Check and auto-set hints.
 109       if not self.spectrum.dtype.is_complex:
 110         if is_self_adjoint is False:
 111           raise ValueError(
 112               "A real spectrum always corresponds to a self-adjoint operator.")
 113         is_self_adjoint = True
 114 
 115       if is_square is False:
 116         raise ValueError(
 117             "A [[nested] block] circulant operator is always square.")
 118       is_square = True
 119 
 120       # If spectrum.shape = [s0, s1, s2], and block_depth = 2,
 121       # block_shape = [s1, s2]
 122       s_shape = array_ops.shape(self.spectrum)
 123       self._block_shape_tensor = s_shape[-self.block_depth:]
 124 
 125       # Add common variants of spectrum to the graph.
 126       self._spectrum_complex = _to_complex(self.spectrum)
 127       self._abs_spectrum = math_ops.abs(self.spectrum)
 128       self._conj_spectrum = math_ops.conj(self._spectrum_complex)
 129 
 130       super(_BaseLinearOperatorCirculant, self).__init__(
 131           dtype=dtypes.as_dtype(input_output_dtype),
 132           graph_parents=[self.spectrum],
 133           is_non_singular=is_non_singular,
 134           is_self_adjoint=is_self_adjoint,
 135           is_positive_definite=is_positive_definite,
 136           is_square=is_square,
 137           name=name)

 139   def _check_spectrum_and_return_tensor(self, spectrum):
 140     """Static check of spectrum.  Then return `Tensor` version."""
 141     spectrum = ops.convert_to_tensor(spectrum, name="spectrum")
 142 
 143     allowed_dtypes = [dtypes.float32, dtypes.complex64]
 144     if spectrum.dtype not in allowed_dtypes:
 145       raise TypeError("Argument spectrum must have dtype in %s.  Found: %s" %
 146                       (allowed_dtypes, spectrum.dtype))
 147     if spectrum.get_shape().ndims is not None:
 148       if spectrum.get_shape().ndims < self.block_depth:
 149         raise ValueError(
 150             "Argument spectrum must have at least %d dimensions.  Found: %s" %
 151             (self.block_depth, spectrum))
 152     return spectrum

 154   @property
 155   def block_depth(self):
 156     """Depth of recursively defined circulant blocks defining this `Operator`.
 157 
 158     With `A` the dense representation of this `Operator`,
 159 
 160     `block_depth = 1` means `A` is symmetric circulant.  For example,
 161 
 162     ```
 163     A = |w z y x|
 164         |x w z y|
 165         |y x w z|
 166         |z y x w|
 167     ```
 168 
 169     `block_depth = 2` means `A` is block symmetric circulant with symemtric
 170     circulant blocks.  For example, with `W`, `X`, `Y`, `Z` symmetric circulant,
 171 
 172     ```
 173     A = |W Z Y X|
 174         |X W Z Y|
 175         |Y X W Z|
 176         |Z Y X W|
 177     ```
 178 
 179     `block_depth = 3` means `A` is block symmetric circulant with block
 180     symmetric circulant blocks.
 181 
 182     Returns:
 183       Python `integer`.
 184     """
 185     return self._block_depth

 188     """Shape of the block dimensions of `self.spectrum`."""
 189     return self._block_shape_tensor

 193     return self.spectrum.get_shape()[-self.block_depth:]

 199   def _vectorize_then_blockify(self, matrix):
 200     """Shape batch matrix to batch vector, then blockify trailing dimensions."""
 201     # Suppose
 202     #   matrix.shape = [m0, m1, m2, m3],
 203     # and matrix is a matrix because the final two dimensions are matrix dims.
 204     #   self.block_depth = 2,
 205     #   self.block_shape = [b0, b1]  (note b0 * b1 = m2).
 206     # We will reshape matrix to
 207     #   [m3, m0, m1, b0, b1].
 208 
 209     # Vectorize: Reshape to batch vector.
 210     #   [m0, m1, m2, m3] --> [m3, m0, m1, m2]
 211     # This is called "vectorize" because we have taken the final two matrix dims
 212     # and turned this into a size m3 batch of vectors.
 213     vec = distribution_util.rotate_transpose(matrix, shift=1)
 214 
 215     # Blockify: Blockfy trailing dimensions.
 216     #   [m3, m0, m1, m2] --> [m3, m0, m1, b0, b1]
 217     if (vec.get_shape().is_fully_defined() and
 218         self.block_shape.is_fully_defined()):
 219       # vec_leading_shape = [m3, m0, m1],
 220       # the parts of vec that will not be blockified.
 221       vec_leading_shape = vec.get_shape()[:-1]
 222       final_shape = vec_leading_shape.concatenate(self.block_shape)
 223     else:
 224       vec_leading_shape = array_ops.shape(vec)[:-1]
 225       final_shape = array_ops.concat(
 226           (vec_leading_shape, self.block_shape_tensor()), 0)
 227     return array_ops.reshape(vec, final_shape)
 228 
 229   def _unblockify_then_matricize(self, vec):
 230     """Flatten the block dimensions then reshape to a batch matrix."""
 231     # Suppose
 232     #   vec.shape = [v0, v1, v2, v3],
 233     #   self.block_depth = 2.
 234     # Then
 235     #   leading shape = [v0, v1]
 236     #   block shape = [v2, v3].
 237     # We will reshape vec to
 238     #   [v1, v2*v3, v0].
 239 
 240     # Un-blockify: Flatten block dimensions.  Reshape
 241     #   [v0, v1, v2, v3] --> [v0, v1, v2*v3].
 242     if vec.get_shape().is_fully_defined():
 243       # vec_shape = [v0, v1, v2, v3]
 244       vec_shape = vec.get_shape().as_list()
 245       # vec_leading_shape = [v0, v1]
 246       vec_leading_shape = vec_shape[:-self.block_depth]
 247       # vec_block_shape = [v2, v3]
 248       vec_block_shape = vec_shape[-self.block_depth:]
 249       # flat_shape = [v0, v1, v2*v3]
 250       flat_shape = vec_leading_shape + [np.prod(vec_block_shape)]
 251     else:
 252       vec_shape = array_ops.shape(vec)
 253       vec_leading_shape = vec_shape[:-self.block_depth]
 254       vec_block_shape = vec_shape[-self.block_depth:]
 255       flat_shape = array_ops.concat(
 256           (vec_leading_shape, [math_ops.reduce_prod(vec_block_shape)]), 0)
 257     vec_flat = array_ops.reshape(vec, flat_shape)
 258 
 259     # Matricize:  Reshape to batch matrix.
 260     #   [v0, v1, v2*v3] --> [v1, v2*v3, v0],
 261     # representing a shape [v1] batch of [v2*v3, v0] matrices.
 262     matrix = distribution_util.rotate_transpose(vec_flat, shift=-1)
 263     return matrix

 265   def _fft(self, x):
 266     """FFT along the last self.block_depth dimensions of x.
 267 
 268     Args:
 269       x: `Tensor` with floating or complex `dtype`.
 270         Should be in the form returned by self._vectorize_then_blockify.
 271 
 272     Returns:
 273       `Tensor` with `dtype` `complex64`.
 274     """
 275     x_complex = _to_complex(x)
 276     return _FFT_OP[self.block_depth](x_complex)
 277 
 278   def _ifft(self, x):
 279     """IFFT along the last self.block_depth dimensions of x.
 280 
 281     Args:
 282       x: `Tensor` with floating or complex dtype.  Should be in the form
 283         returned by self._vectorize_then_blockify.
 284 
 285     Returns:
 286       `Tensor` with `dtype` `complex64`.
 287     """
 288     x_complex = _to_complex(x)
 289     return _IFFT_OP[self.block_depth](x_complex)

 291   def convolution_kernel(self, name="convolution_kernel"):
 292     """Convolution kernel corresponding to `self.spectrum`.
 293 
 294     The `D` dimensional DFT of this kernel is the frequency domain spectrum of
 295     this operator.
 296 
 297     Args:
 298       name:  A name to give this `Op`.
 299 
 300     Returns:
 301       `Tensor` with `dtype` `self.dtype`.
 302     """
 303     with self._name_scope(name):
 304       h = self._ifft(self._spectrum_complex)
 305       return math_ops.cast(h, self.dtype)

 307   def _shape(self):
 308     s_shape = self._spectrum.get_shape()
 309     # Suppose spectrum.shape = [a, b, c, d]
 310     # block_depth = 2
 311     # Then:
 312     #   batch_shape = [a, b]
 313     #   N = c*d
 314     # and we want to return
 315     #   [a, b, c*d, c*d]
 316     batch_shape = s_shape[:-self.block_depth]
 317     # trailing_dims = [c, d]
 318     trailing_dims = s_shape[-self.block_depth:]
 319     if trailing_dims.is_fully_defined():
 320       n = np.prod(trailing_dims.as_list())
 321     else:
 322       n = None
 323     n_x_n = tensor_shape.TensorShape([n, n])
 324     return batch_shape.concatenate(n_x_n)
 325 
 326   def _shape_tensor(self):
 327     # See self.shape for explanation of steps
 328     s_shape = array_ops.shape(self._spectrum)
 329     batch_shape = s_shape[:-self.block_depth]
 330     trailing_dims = s_shape[-self.block_depth:]
 331     n = math_ops.reduce_prod(trailing_dims)
 332     n_x_n = [n, n]
 333     return array_ops.concat((batch_shape, n_x_n), 0)

 335   def assert_hermitian_spectrum(self, name="assert_hermitian_spectrum"):
 336     """Returns an `Op` that asserts this operator has Hermitian spectrum.
 337 
 338     This operator corresponds to a real-valued matrix if and only if its
 339     spectrum is Hermitian.
 340 
 341     Args:
 342       name:  A name to give this `Op`.
 343 
 344     Returns:
 345       An `Op` that asserts this operator has Hermitian spectrum.
 346     """
 347     eps = np.finfo(self.dtype.real_dtype.as_numpy_dtype).eps
 348     with self._name_scope(name):
 349       # Assume linear accumulation of error.
 350       max_err = eps * self.domain_dimension_tensor()
 351       imag_convolution_kernel = math_ops.imag(self.convolution_kernel())
 352       return check_ops.assert_less(
 353           math_ops.abs(imag_convolution_kernel),
 354           max_err,
 355           message="Spectrum was not Hermitian")
 356 
 357   def _assert_non_singular(self):
 358     return linear_operator_util.assert_no_entries_with_modulus_zero(
 359         self.spectrum,
 360         message="Singular operator:  Spectrum contained zero values.")
 361 
 362   def _assert_positive_definite(self):
 363     # This operator has the action  Ax = F^H D F x,
 364     # where D is the diagonal matrix with self.spectrum on the diag.  Therefore,
 365     # <x, Ax> = <Fx, DFx>,
 366     # Since F is bijective, the condition for positive definite is the same as
 367     # for a diagonal matrix, i.e. real part of spectrum is positive.
 368     message = (
 369         "Not positive definite:  Real part of spectrum was not all positive.")
 370     return check_ops.assert_positive(
 371         math_ops.real(self.spectrum), message=message)
 372 
 373   def _assert_self_adjoint(self):
 374     # Recall correspondence between symmetry and real transforms.  See docstring
 375     return linear_operator_util.assert_zero_imag_part(
 376         self.spectrum,
 377         message=(
 378             "Not self-adjoint:  The spectrum contained non-zero imaginary part."
 379         ))

 381   def _broadcast_batch_dims(self, x, spectrum):
 382     """Broadcast batch dims of batch matrix `x` and spectrum."""
 383     # spectrum.shape = batch_shape + block_shape
 384     # First make spectrum a batch matrix with
 385     #   spectrum.shape = batch_shape + [prod(block_shape), 1]
 386     spec_mat = array_ops.reshape(
 387         spectrum, array_ops.concat(
 388             (self.batch_shape_tensor(), [-1, 1]), axis=0))
 389     # Second, broadcast, possibly requiring an addition of array of zeros.
 390     x, spec_mat = linear_operator_util.broadcast_matrix_batch_dims((x,
 391                                                                     spec_mat))
 392     # Third, put the block shape back into spectrum.
 393     batch_shape = array_ops.shape(x)[:-2]
 394     spectrum = array_ops.reshape(
 395         spec_mat,
 396         array_ops.concat((batch_shape, self.block_shape_tensor()), axis=0))
 397 
 398     return x, spectrum

 400   def _matmul(self, x, adjoint=False, adjoint_arg=False):
 401     x = linalg.adjoint(x) if adjoint_arg else x
 402     # With F the matrix of a DFT, and F^{-1}, F^H the inverse and Hermitian
 403     # transpose, one can show that F^{-1} = F^{H} is the IDFT matrix.  Therefore
 404     # matmul(x) = F^{-1} diag(spectrum) F x,
 405     #           = F^{H} diag(spectrum) F x,
 406     # so that
 407     # matmul(x, adjoint=True) = F^{H} diag(conj(spectrum)) F x.
 408     spectrum = self._conj_spectrum if adjoint else self._spectrum_complex
 409 
 410     x, spectrum = self._broadcast_batch_dims(x, spectrum)
 411 
 412     x_vb = self._vectorize_then_blockify(x)
 413     fft_x_vb = self._fft(x_vb)
 414     block_vector_result = self._ifft(spectrum * fft_x_vb)
 415     y = self._unblockify_then_matricize(block_vector_result)
 416 
 417     return math_ops.cast(y, self.dtype)
 418 
 419   def _determinant(self):
 420     reduction_indices = [-(i + 1) for i in range(self.block_depth)]
 421     det = math_ops.reduce_prod(
 422         self.spectrum, reduction_indices=reduction_indices)
 423     return math_ops.cast(det, self.dtype)
 424 
 425   def _log_abs_determinant(self):
 426     reduction_indices = [-(i + 1) for i in range(self.block_depth)]
 427     lad = math_ops.reduce_sum(
 428         math_ops.log(self._abs_spectrum), reduction_indices=reduction_indices)
 429     return math_ops.cast(lad, self.dtype)
 430 
 431   def _solve(self, rhs, adjoint=False, adjoint_arg=False):
 432     rhs = linalg.adjoint(rhs) if adjoint_arg else rhs
 433     spectrum = self._conj_spectrum if adjoint else self._spectrum_complex
 434 
 435     rhs, spectrum = self._broadcast_batch_dims(rhs, spectrum)
 436 
 437     rhs_vb = self._vectorize_then_blockify(rhs)
 438     fft_rhs_vb = self._fft(rhs_vb)
 439     solution_vb = self._ifft(fft_rhs_vb / spectrum)
 440     x = self._unblockify_then_matricize(solution_vb)
 441     return math_ops.cast(x, self.dtype)
 442 
 443   def _diag_part(self):
 444     # Get ones in shape of diag, which is [B1,...,Bb, N]
 445     # Also get the size of the diag, "N".
 446     if self.shape.is_fully_defined():
 447       diag_shape = self.shape[:-1]
 448       diag_size = self.domain_dimension.value
 449     else:
 450       diag_shape = self.shape_tensor()[:-1]
 451       diag_size = self.domain_dimension_tensor()
 452     ones_diag = array_ops.ones(diag_shape, dtype=self.dtype)
 453 
 454     # As proved in comments in self._trace, the value on the diag is constant,
 455     # repeated N times.  This value is the trace divided by N.
 456 
 457     # The handling of self.shape = (0, 0) is tricky, and is the reason we choose
 458     # to compute trace and use that to compute diag_part, rather than computing
 459     # the value on the diagonal ("diag_value") directly.  Both result in a 0/0,
 460     # but in different places, and the current method gives the right result in
 461     # the end.
 462 
 463     # Here, if self.shape = (0, 0), then self.trace() = 0., and then
 464     # diag_value = 0. / 0. = NaN.
 465     diag_value = self.trace() / math_ops.cast(diag_size, self.dtype)
 466 
 467     # If self.shape = (0, 0), then ones_diag = [] (empty tensor), and then
 468     # the following line is NaN * [] = [], as needed.
 469     return diag_value[..., array_ops.newaxis] * ones_diag
 470 
 471   def _trace(self):
 472     # The diagonal of the [[nested] block] circulant operator is the mean of
 473     # the spectrum.
 474     # Proof:  For the [0,...,0] element, this follows from the IDFT formula.
 475     # Then the result follows since all diagonal elements are the same.
 476 
 477     # Therefore, the trace is the sum of the spectrum.
 478 
 479     # Get shape of diag along with the axis over which to reduce the spectrum.
 480     # We will reduce the spectrum over all block indices.
 481     if self.spectrum.get_shape().is_fully_defined():
 482       spec_rank = self.spectrum.get_shape().ndims
 483       axis = np.arange(spec_rank - self.block_depth, spec_rank, dtype=np.int32)
 484     else:
 485       spec_rank = array_ops.rank(self.spectrum)
 486       axis = math_ops.range(spec_rank - self.block_depth, spec_rank)
 487 
 488     # Real diag part "re_d".
 489     # Suppose spectrum.shape = [B1,...,Bb, N1, N2]
 490     # self.shape = [B1,...,Bb, N, N], with N1 * N2 = N.
 491     # re_d_value.shape = [B1,...,Bb]
 492     re_d_value = math_ops.reduce_sum(math_ops.real(self.spectrum), axis=axis)
 493 
 494     if not self.dtype.is_complex:
 495       return math_ops.cast(re_d_value, self.dtype)
 496 
 497     # Imaginary part, "im_d".
 498     if self.is_self_adjoint:
 499       im_d_value = 0.
 500     else:
 501       im_d_value = math_ops.reduce_sum(math_ops.imag(self.spectrum), axis=axis)
 502 
 503     return math_ops.cast(math_ops.complex(re_d_value, im_d_value), self.dtype)

 506 @tf_export("linalg.LinearOperatorCirculant")
 507 class LinearOperatorCirculant(_BaseLinearOperatorCirculant):
 508   """`LinearOperator` acting like a circulant matrix.
 509 
 510   This operator acts like a circulant matrix `A` with
 511   shape `[B1,...,Bb, N, N]` for some `b >= 0`.  The first `b` indices index a
 512   batch member.  For every batch index `(i1,...,ib)`, `A[i1,...,ib, : :]` is
 513   an `N x N` matrix.  This matrix `A` is not materialized, but for
 514   purposes of broadcasting this shape will be relevant.
 515 
 516   #### Description in terms of circulant matrices
 517 
 518   Circulant means the entries of `A` are generated by a single vector, the
 519   convolution kernel `h`: `A_{mn} := h_{m-n mod N}`.  With `h = [w, x, y, z]`,
 520 
 521   ```
 522   A = |w z y x|
 523       |x w z y|
 524       |y x w z|
 525       |z y x w|
 526   ```
 527 
 528   This means that the result of matrix multiplication `v = Au` has `Lth` column
 529   given circular convolution between `h` with the `Lth` column of `u`.
 530 
 531   See http://ee.stanford.edu/~gray/toeplitz.pdf
 532 
 533   #### Description in terms of the frequency spectrum
 534 
 535   There is an equivalent description in terms of the [batch] spectrum `H` and
 536   Fourier transforms.  Here we consider `A.shape = [N, N]` and ignore batch
 537   dimensions.  Define the discrete Fourier transform (DFT) and its inverse by
 538 
 539   ```
 540   DFT[ h[n] ] = H[k] := sum_{n = 0}^{N - 1} h_n e^{-i 2pi k n / N}
 541   IDFT[ H[k] ] = h[n] = N^{-1} sum_{k = 0}^{N - 1} H_k e^{i 2pi k n / N}
 542   ```
 543 
 544   From these definitions, we see that
 545 
 546   ```
 547   H[0] = sum_{n = 0}^{N - 1} h_n
 548   H[1] = "the first positive frequency"
 549   H[N - 1] = "the first negative frequency"
 550   ```
 551 
 552   Loosely speaking, with `*` element-wise multiplication, matrix multiplication
 553   is equal to the action of a Fourier multiplier: `A u = IDFT[ H * DFT[u] ]`.
 554   Precisely speaking, given `[N, R]` matrix `u`, let `DFT[u]` be the `[N, R]`
 555   matrix with `rth` column equal to the DFT of the `rth` column of `u`.
 556   Define the `IDFT` similarly.
 557   Matrix multiplication may be expressed columnwise:
 558 
 559   ```(A u)_r = IDFT[ H * (DFT[u])_r ]```
 560 
 561   #### Operator properties deduced from the spectrum.
 562 
 563   Letting `U` be the `kth` Euclidean basis vector, and `U = IDFT[u]`.
 564   The above formulas show that`A U = H_k * U`.  We conclude that the elements
 565   of `H` are the eigenvalues of this operator.   Therefore
 566 
 567   * This operator is positive definite if and only if `Real{H} > 0`.
 568 
 569   A general property of Fourier transforms is the correspondence between
 570   Hermitian functions and real valued transforms.
 571 
 572   Suppose `H.shape = [B1,...,Bb, N]`.  We say that `H` is a Hermitian spectrum
 573   if, with `%` meaning modulus division,
 574 
 575   ```H[..., n % N] = ComplexConjugate[ H[..., (-n) % N] ]```
 576 
 577   * This operator corresponds to a real matrix if and only if `H` is Hermitian.
 578   * This operator is self-adjoint if and only if `H` is real.
 579 
 580   See e.g. "Discrete-Time Signal Processing", Oppenheim and Schafer.
 581 
 582   #### Example of a self-adjoint positive definite operator
 583 
 584   ```python
 585   # spectrum is real ==> operator is self-adjoint
 586   # spectrum is positive ==> operator is positive definite
 587   spectrum = [6., 4, 2]
 588 
 589   operator = LinearOperatorCirculant(spectrum)
 590 
 591   # IFFT[spectrum]
 592   operator.convolution_kernel()
 593   ==> [4 + 0j, 1 + 0.58j, 1 - 0.58j]
 594 
 595   operator.to_dense()
 596   ==> [[4 + 0.0j, 1 - 0.6j, 1 + 0.6j],
 597        [1 + 0.6j, 4 + 0.0j, 1 - 0.6j],
 598        [1 - 0.6j, 1 + 0.6j, 4 + 0.0j]]
 599   ```
 600 
 601   #### Example of defining in terms of a real convolution kernel
 602 
 603   ```python

 607

 609   # spectrum is shape [3] ==> operator is shape [3, 3]
 610   # We force the input/output type to be real, which allows this to operate
 611   # like a real matrix.
 612   operator = LinearOperatorCirculant(spectrum, input_output_dtype=tf.float32)
 613 
 614   operator.to_dense()
 615   ==> [[ 1, 1, 2],
 616        [ 2, 1, 1],
 617        [ 1, 2, 1]]
 618   ```
