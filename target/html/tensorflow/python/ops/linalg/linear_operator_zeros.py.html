  42 @tf_export("linalg.LinearOperatorZeros")
  43 class LinearOperatorZeros(linear_operator.LinearOperator):
  44   """`LinearOperator` acting like a [batch] zero matrix.
  45 
  46   This operator acts like a [batch] zero matrix `A` with shape
  47   `[B1,...,Bb, N, M]` for some `b >= 0`.  The first `b` indices index a
  48   batch member.  For every batch index `(i1,...,ib)`, `A[i1,...,ib, : :]` is
  49   an `N x M` matrix.  This matrix `A` is not materialized, but for
  50   purposes of broadcasting this shape will be relevant.
  51 
  52   `LinearOperatorZeros` is initialized with `num_rows`, and optionally
  53   `num_columns, `batch_shape`, and `dtype` arguments.  If `num_columns` is
  54   `None`, then this operator will be initialized as a square matrix. If
  55   `batch_shape` is `None`, this operator efficiently passes through all
  56   arguments.  If `batch_shape` is provided, broadcasting may occur, which will
  57   require making copies.
  58 
  59   ```python
  60   # Create a 2 x 2 zero matrix.
  61   operator = LinearOperatorZero(num_rows=2, dtype=tf.float32)
  62 
  63   operator.to_dense()
  64   ==> [[0., 0.]
  65        [0., 0.]]
  66 
  67   operator.shape
  68   ==> [2, 2]
  69 
  70   operator.determinant()
  71   ==> 0.
  72 
  73   x = ... Shape [2, 4] Tensor
  74   operator.matmul(x)
  75   ==> Shape [2, 4] Tensor, same as x.
  76 
  77   # Create a 2-batch of 2x2 zero matrices
  78   operator = LinearOperatorZeros(num_rows=2, batch_shape=[2])
  79   operator.to_dense()
  80   ==> [[[0., 0.]
  81         [0., 0.]],
  82        [[0., 0.]
  83         [0., 0.]]]
  84 
  85   # Here, even though the operator has a batch shape, the input is the same as
  86   # the output, so x can be passed through without a copy.  The operator is able
  87   # to detect that no broadcast is necessary because both x and the operator
  88   # have statically defined shape.
  89   x = ... Shape [2, 2, 3]
  90   operator.matmul(x)
  91   ==> Shape [2, 2, 3] Tensor, same as tf.zeros_like(x)
  92 
  93   # Here the operator and x have different batch_shape, and are broadcast.
  94   # This requires a copy, since the output is different size than the input.
  95   x = ... Shape [1, 2, 3]
  96   operator.matmul(x)
  97   ==> Shape [2, 2, 3] Tensor, equal to tf.zeros_like([x, x])
  98   ```
  99 
 100   ### Shape compatibility
 101 
 102   This operator acts on [batch] matrix with compatible shape.
 103   `x` is a batch matrix with compatible shape for `matmul` and `solve` if
 104 
 105   ```
 106   operator.shape = [B1,...,Bb] + [N, M],  with b >= 0
 107   x.shape =   [C1,...,Cc] + [M, R],
 108   and [C1,...,Cc] broadcasts with [B1,...,Bb] to [D1,...,Dd]
 109   ```
 110 
 111   #### Matrix property hints
 112 
 113   This `LinearOperator` is initialized with boolean flags of the form `is_X`,
 114   for `X = non_singular, self_adjoint, positive_definite, square`.
 115   These have the following meaning:
 116 
 117   * If `is_X == True`, callers should expect the operator to have the
 118     property `X`.  This is a promise that should be fulfilled, but is *not* a
 119     runtime assert.  For example, finite floating point precision may result
 120     in these promises being violated.
 121   * If `is_X == False`, callers should expect the operator to not have `X`.
 122   * If `is_X == None` (the default), callers should have no expectation either
 123     way.
 124   """
 125 
 126   def __init__(self,
 127                num_rows,
 128                num_columns=None,
 129                batch_shape=None,
 130                dtype=None,
 131                is_non_singular=False,
 132                is_self_adjoint=True,
 133                is_positive_definite=False,
 134                is_square=True,
 135                assert_proper_shapes=False,
 136                name="LinearOperatorZeros"):
 137     r"""Initialize a `LinearOperatorZeros`.
 138 
 139     The `LinearOperatorZeros` is initialized with arguments defining `dtype`
 140     and shape.
 141 
 142     This operator is able to broadcast the leading (batch) dimensions, which
 143     sometimes requires copying data.  If `batch_shape` is `None`, the operator
 144     can take arguments of any batch shape without copying.  See examples.
 145 
 146     Args:
 147       num_rows:  Scalar non-negative integer `Tensor`.  Number of rows in the
 148         corresponding zero matrix.
 149       num_columns:  Scalar non-negative integer `Tensor`.  Number of columns in
 150         the corresponding zero matrix. If `None`, defaults to the value of
 151         `num_rows`.
 152       batch_shape:  Optional `1-D` integer `Tensor`.  The shape of the leading
 153         dimensions.  If `None`, this operator has no leading dimensions.
 154       dtype:  Data type of the matrix that this operator represents.
 155       is_non_singular:  Expect that this operator is non-singular.
 156       is_self_adjoint:  Expect that this operator is equal to its hermitian
 157         transpose.
 158       is_positive_definite:  Expect that this operator is positive definite,
 159         meaning the quadratic form `x^H A x` has positive real part for all
 160         nonzero `x`.  Note that we do not require the operator to be
 161         self-adjoint to be positive-definite.  See:
 162         https://en.wikipedia.org/wiki/Positive-definite_matrix#Extension_for_non-symmetric_matrices
 163       is_square:  Expect that this operator acts like square [batch] matrices.
 164       assert_proper_shapes:  Python `bool`.  If `False`, only perform static
 165         checks that initialization and method arguments have proper shape.
 166         If `True`, and static checks are inconclusive, add asserts to the graph.
 167       name: A name for this `LinearOperator`
 168 
 169     Raises:
 170       ValueError:  If `num_rows` is determined statically to be non-scalar, or
 171         negative.
 172       ValueError:  If `num_columns` is determined statically to be non-scalar,
 173         or negative.
 174       ValueError:  If `batch_shape` is determined statically to not be 1-D, or
 175         negative.
 176       ValueError:  If any of the following is not `True`:
 177         `{is_self_adjoint, is_non_singular, is_positive_definite}`.
 178     """
 179     dtype = dtype or dtypes.float32
 180     self._assert_proper_shapes = assert_proper_shapes
 181 
 182     with ops.name_scope(name):
 183       dtype = dtypes.as_dtype(dtype)
 184       if not is_self_adjoint and is_square:
 185         raise ValueError("A zero operator is always self adjoint.")
 186       if is_non_singular:
 187         raise ValueError("A zero operator is always singular.")
 188       if is_positive_definite:
 189         raise ValueError("A zero operator is always not positive-definite.")
 190 
 191       super(LinearOperatorZeros, self).__init__(
 192           dtype=dtype,
 193           is_non_singular=is_non_singular,
 194           is_self_adjoint=is_self_adjoint,
 195           is_positive_definite=is_positive_definite,
 196           is_square=is_square,
 197           name=name)
 198 
 199       self._num_rows = linear_operator_util.shape_tensor(
 200           num_rows, name="num_rows")
 201       self._num_rows_static = tensor_util.constant_value(self._num_rows)
 202 
 203       if num_columns is None:
 204         num_columns = num_rows
 205 
 206       self._num_columns = linear_operator_util.shape_tensor(
 207           num_columns, name="num_columns")
 208       self._num_columns_static = tensor_util.constant_value(self._num_columns)
 209 
 210       self._check_domain_range_possibly_add_asserts()
 211 
 212       if (self._num_rows_static is not None and
 213           self._num_columns_static is not None):
 214         if is_square and self._num_rows_static != self._num_columns_static:
 215           raise ValueError(
 216               "LinearOperatorZeros initialized as is_square=True, but got "
 217               "num_rows({}) != num_columns({})".format(
 218                   self._num_rows_static,
 219                   self._num_columns_static))
 220 
 221       if batch_shape is None:
 222         self._batch_shape_arg = None
 223       else:
 224         self._batch_shape_arg = linear_operator_util.shape_tensor(
 225             batch_shape, name="batch_shape_arg")
 226         self._batch_shape_static = tensor_util.constant_value(
 227             self._batch_shape_arg)
 228         self._check_batch_shape_possibly_add_asserts()
 229 
 230   def _shape(self):
 231     matrix_shape = tensor_shape.TensorShape((self._num_rows_static,
 232                                              self._num_columns_static))
 233     if self._batch_shape_arg is None:
 234       return matrix_shape
 235 
 236     batch_shape = tensor_shape.TensorShape(self._batch_shape_static)
 237     return batch_shape.concatenate(matrix_shape)
 238 
 239   def _shape_tensor(self):
 240     matrix_shape = array_ops.stack((self._num_rows, self._num_columns), axis=0)
 241     if self._batch_shape_arg is None:
 242       return matrix_shape
 243 
 244     return array_ops.concat((self._batch_shape_arg, matrix_shape), 0)
 245 
 246   def _assert_non_singular(self):
 247     raise errors.InvalidArgumentError(
 248         node_def=None, op=None, message="Zero operators are always "
 249         "non-invertible.")
 250 
 251   def _assert_positive_definite(self):
 252     raise errors.InvalidArgumentError(
 253         node_def=None, op=None, message="Zero operators are always "
 254         "non-positive definite.")
 255 
 256   def _assert_self_adjoint(self):
 257     return control_flow_ops.no_op("assert_self_adjoint")
 258 
 259   def _possibly_broadcast_batch_shape(self, x):
 260     """Return 'x', possibly after broadcasting the leading dimensions."""
 261     # If we have no batch shape, our batch shape broadcasts with everything!
 262     if self._batch_shape_arg is None:
 263       return x
 264 
 265     # Static attempt:
 266     #   If we determine that no broadcast is necessary, pass x through
 267     #   If we need a broadcast, add to an array of zeros.
 268     #
 269     # special_shape is the shape that, when broadcast with x's shape, will give
 270     # the correct broadcast_shape.  Note that
 271     #   We have already verified the second to last dimension of self.shape
 272     #   matches x's shape in assert_compatible_matrix_dimensions.
 273     #   Also, the final dimension of 'x' can have any shape.
 274     #   Therefore, the final two dimensions of special_shape are 1's.
 275     special_shape = self.batch_shape.concatenate([1, 1])
 276     bshape = array_ops.broadcast_static_shape(x.get_shape(), special_shape)
 277     if special_shape.is_fully_defined():
 278       # bshape.is_fully_defined iff special_shape.is_fully_defined.
 279       if bshape == x.get_shape():
 280         return x
 281       # Use the built in broadcasting of addition.
 282       zeros = array_ops.zeros(shape=special_shape, dtype=self.dtype)
 283       return x + zeros
 284 
 285     # Dynamic broadcast:
 286     #   Always add to an array of zeros, rather than using a "cond", since a
 287     #   cond would require copying data from GPU --> CPU.
 288     special_shape = array_ops.concat((self.batch_shape_tensor(), [1, 1]), 0)
 289     zeros = array_ops.zeros(shape=special_shape, dtype=self.dtype)
 290     return x + zeros
 291 
 292   def _matmul(self, x, adjoint=False, adjoint_arg=False):
 293     if self._assert_proper_shapes:
 294       x = linalg.adjoint(x) if adjoint_arg else x
 295       aps = linear_operator_util.assert_compatible_matrix_dimensions(self, x)
 296       x = control_flow_ops.with_dependencies([aps], x)
 297     if self.is_square:
 298       # Note that adjoint has no effect since this matrix is self-adjoint.
 299       if adjoint_arg:
 300         output_shape = array_ops.concat([
 301             array_ops.shape(x)[:-2],
 302             [array_ops.shape(x)[-1], array_ops.shape(x)[-2]]], axis=0)
 303       else:
 304         output_shape = array_ops.shape(x)
 305 
 306       return self._possibly_broadcast_batch_shape(
 307           array_ops.zeros(shape=output_shape, dtype=x.dtype))
 308 
 309     x_shape = array_ops.shape(x)
 310     n = self._num_columns if adjoint else self._num_rows
 311     m = x_shape[-2] if adjoint_arg else x_shape[-1]
 312 
 313     output_shape = array_ops.concat([x_shape[:-2], [n, m]], axis=0)
 314 
 315     zeros = array_ops.zeros(shape=output_shape, dtype=x.dtype)
 316     return self._possibly_broadcast_batch_shape(zeros)
 317 
 318   def _determinant(self):
 319     if self.batch_shape.is_fully_defined():
 320       return array_ops.zeros(shape=self.batch_shape, dtype=self.dtype)
 321     else:
 322       return array_ops.zeros(shape=self.batch_shape_tensor(), dtype=self.dtype)
 323 
 324   def _trace(self):
 325     # Get Tensor of all zeros of same shape as self.batch_shape.
 326     if self.batch_shape.is_fully_defined():
 327       return array_ops.zeros(shape=self.batch_shape, dtype=self.dtype)
 328     else:
 329       return array_ops.zeros(shape=self.batch_shape_tensor(), dtype=self.dtype)
 330 
 331   def _diag_part(self):
 332     return self._zeros_diag()
 333 
 334   def add_to_tensor(self, mat, name="add_to_tensor"):
 335     """Add matrix represented by this operator to `mat`.  Equiv to `I + mat`.
 336 
 337     Args:
 338       mat:  `Tensor` with same `dtype` and shape broadcastable to `self`.
 339       name:  A name to give this `Op`.
 340 
 341     Returns:
 342       A `Tensor` with broadcast shape and same `dtype` as `self`.
 343     """
 344     return self._possibly_broadcast_batch_shape(mat)

 346   def _check_domain_range_possibly_add_asserts(self):
 347     """Static check of init arg `num_rows`, possibly add asserts."""
 348     # Possibly add asserts.
 349     if self._assert_proper_shapes:
 350       self._num_rows = control_flow_ops.with_dependencies([
 351           check_ops.assert_rank(
 352               self._num_rows,
 353               0,
 354               message="Argument num_rows must be a 0-D Tensor."),
 355           check_ops.assert_non_negative(
 356               self._num_rows,
 357               message="Argument num_rows must be non-negative."),
 358       ], self._num_rows)
 359       self._num_columns = control_flow_ops.with_dependencies([
 360           check_ops.assert_rank(
 361               self._num_columns,
 362               0,
 363               message="Argument num_columns must be a 0-D Tensor."),
 364           check_ops.assert_non_negative(
 365               self._num_columns,
 366               message="Argument num_columns must be non-negative."),
 367       ], self._num_columns)
 368 
 369     # Static checks.
 370     if not self._num_rows.dtype.is_integer:
 371       raise TypeError("Argument num_rows must be integer type.  Found:"
 372                       " %s" % self._num_rows)
 373 
 374     if not self._num_columns.dtype.is_integer:
 375       raise TypeError("Argument num_columns must be integer type.  Found:"
 376                       " %s" % self._num_columns)
 377 
 378     num_rows_static = self._num_rows_static
 379     num_columns_static = self._num_columns_static
 380 
 381     if num_rows_static is not None:
 382       if num_rows_static.ndim != 0:
 383         raise ValueError("Argument num_rows must be a 0-D Tensor.  Found:"
 384                          " %s" % num_rows_static)
 385 
 386       if num_rows_static < 0:
 387         raise ValueError("Argument num_rows must be non-negative.  Found:"
 388                          " %s" % num_rows_static)
 389     if num_columns_static is not None:
 390       if num_columns_static.ndim != 0:
 391         raise ValueError("Argument num_columns must be a 0-D Tensor.  Found:"
 392                          " %s" % num_columns_static)
 393 
 394       if num_columns_static < 0:
 395         raise ValueError("Argument num_columns must be non-negative.  Found:"
 396                          " %s" % num_columns_static)
 397 
 398   def _check_batch_shape_possibly_add_asserts(self):
 399     """Static check of init arg `batch_shape`, possibly add asserts."""
 400     if self._batch_shape_arg is None:
 401       return
 402 
 403     # Possibly add asserts
 404     if self._assert_proper_shapes:
 405       self._batch_shape_arg = control_flow_ops.with_dependencies([
 406           check_ops.assert_rank(
 407               self._batch_shape_arg,
 408               1,
 409               message="Argument batch_shape must be a 1-D Tensor."),
 410           check_ops.assert_non_negative(
 411               self._batch_shape_arg,
 412               message="Argument batch_shape must be non-negative."),
 413       ], self._batch_shape_arg)
 414 
 415     # Static checks
 416     if not self._batch_shape_arg.dtype.is_integer:
 417       raise TypeError("Argument batch_shape must be integer type.  Found:"
 418                       " %s" % self._batch_shape_arg)
 419 
 420     if self._batch_shape_static is None:
 421       return  # Cannot do any other static checks.
 422 
 423     if self._batch_shape_static.ndim != 1:
 424       raise ValueError("Argument batch_shape must be a 1-D Tensor.  Found:"
 425                        " %s" % self._batch_shape_static)
 426 
 427     if np.any(self._batch_shape_static < 0):
 428       raise ValueError("Argument batch_shape must be non-negative.  Found:"
 429                        "%s" % self._batch_shape_static)

 431   def _min_matrix_dim(self):
 432     """Minimum of domain/range dimension, if statically available, else None."""
 433     domain_dim = self.domain_dimension.value
 434     range_dim = self.range_dimension.value
 435     if domain_dim is None or range_dim is None:
 436       return None
 437     return min(domain_dim, range_dim)
 438 
 439   def _min_matrix_dim_tensor(self):
 440     """Minimum of domain/range dimension, as a tensor."""
 441     return math_ops.reduce_min(self.shape_tensor()[-2:])
 442 
 443   def _zeros_diag(self):
 444     """Returns the diagonal of this operator as all zeros."""
 445     if self.shape.is_fully_defined():
 446       d_shape = self.batch_shape.concatenate([self._min_matrix_dim()])
 447     else:
 448       d_shape = array_ops.concat(
 449           [self.batch_shape_tensor(),
 450            [self._min_matrix_dim_tensor()]], axis=0)
 451 
 452     return array_ops.zeros(shape=d_shape, dtype=self.dtype)
