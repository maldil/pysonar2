  30 def assert_no_entries_with_modulus_zero(
  31     x, message=None, name="assert_no_entries_with_modulus_zero"):
  32   """Returns `Op` that asserts Tensor `x` has no entries with modulus zero.
  33 
  34   Args:
  35     x:  Numeric `Tensor`, real, integer, or complex.
  36     message:  A string message to prepend to failure message.
  37     name:  A name to give this `Op`.
  38 
  39   Returns:
  40     An `Op` that asserts `x` has no entries with modulus zero.
  41   """
  42   with ops.name_scope(name, values=[x]):
  43     x = ops.convert_to_tensor(x, name="x")
  44     dtype = x.dtype.base_dtype
  45     should_be_nonzero = math_ops.abs(x)
  46     zero = ops.convert_to_tensor(0, dtype=dtype.real_dtype)
  47     return check_ops.assert_less(zero, should_be_nonzero, message=message)

  50 def assert_zero_imag_part(x, message=None, name="assert_zero_imag_part"):
  51   """Returns `Op` that asserts Tensor `x` has no non-zero imaginary parts.
  52 
  53   Args:
  54     x:  Numeric `Tensor`, real, integer, or complex.
  55     message:  A string message to prepend to failure message.
  56     name:  A name to give this `Op`.
  57 
  58   Returns:
  59     An `Op` that asserts `x` has no entries with modulus zero.
  60   """
  61   with ops.name_scope(name, values=[x]):
  62     x = ops.convert_to_tensor(x, name="x")
  63     dtype = x.dtype.base_dtype
  64 
  65     if dtype.is_floating:
  66       return control_flow_ops.no_op()
  67 
  68     zero = ops.convert_to_tensor(0, dtype=dtype.real_dtype)
  69     return check_ops.assert_equal(zero, math_ops.imag(x), message=message)

  72 def assert_compatible_matrix_dimensions(operator, x):
  73   """Assert that an argument to solve/matmul has proper domain dimension.
  74 
  75   If `operator.shape[-2:] = [M, N]`, and `x.shape[-2:] = [Q, R]`, then
  76   `operator.matmul(x)` is defined only if `N = Q`.  This `Op` returns an
  77   `Assert` that "fires" if this is not the case.  Static checks are already
  78   done by the base class `LinearOperator`.
  79 
  80   Args:
  81     operator:  `LinearOperator`.
  82     x:  `Tensor`.
  83 
  84   Returns:
  85     `Assert` `Op`.
  86   """
  87   # Static checks are done in the base class.  Only tensor asserts here.
  88   assert_same_dd = check_ops.assert_equal(
  89       array_ops.shape(x)[-2],
  90       operator.domain_dimension_tensor(),
  91       message=("Incompatible matrix dimensions.  "
  92                "shape[-2] of argument to be the same as this operator"))
  93 
  94   return assert_same_dd

  97 def assert_is_batch_matrix(tensor):
  98   """Static assert that `tensor` has rank `2` or higher."""
  99   sh = tensor.get_shape()
 100   if sh.ndims is not None and sh.ndims < 2:
 101     raise ValueError(
 102         "Expected [batch] matrix to have at least two dimensions.  Found: "
 103         "%s" % tensor)

 106 def shape_tensor(shape, name=None):
 107   """Convert Tensor using default type, unless empty list or tuple."""
 108   # Works just like random_ops._ShapeTensor.
 109   if isinstance(shape, (tuple, list)) and not shape:
 110     dtype = dtypes.int32
 111   else:
 112     dtype = None
 113   return ops.convert_to_tensor(shape, dtype=dtype, name=name)

 122 def broadcast_matrix_batch_dims(batch_matrices, name=None):
 123   """Broadcast leading dimensions of zero or more [batch] matrices.
 124 
 125   Example broadcasting one batch dim of two simple matrices.
 126 
 127   ```python
 128   x = [[1, 2],
 129        [3, 4]]  # Shape [2, 2], no batch dims
 130 
 131   y = [[[1]]]   # Shape [1, 1, 1], 1 batch dim of shape [1]
 132 
 133   x_bc, y_bc = broadcast_matrix_batch_dims([x, y])
 134 
 135   x_bc
 136   ==> [[[1, 2],
 137         [3, 4]]]  # Shape [1, 2, 2], 1 batch dim of shape [1].
 138 
 139   y_bc
 140   ==> same as y
 141   ```
 142 
 143   Example broadcasting many batch dims
 144 
 145   ```python
 146   x = tf.random_normal(shape=(2, 3, 1, 4, 4))
 147   y = tf.random_normal(shape=(1, 3, 2, 5, 5))
 148   x_bc, y_bc = broadcast_matrix_batch_dims([x, y])
 149 
 150   x_bc.shape
 151   ==> (2, 3, 2, 4, 4)
 152 
 153   y_bc.shape
 154   ==> (2, 3, 2, 5, 5)
 155   ```
 156 
 157   Args:
 158     batch_matrices:  Iterable of `Tensor`s, each having two or more dimensions.
 159     name:  A string name to prepend to created ops.
 160 
 161   Returns:
 162     bcast_matrices: List of `Tensor`s, with `bcast_matricies[i]` containing
 163       the values from `batch_matrices[i]`, with possibly broadcast batch dims.
 164 
 165   Raises:
 166     ValueError:  If any input `Tensor` is statically determined to have less
 167       than two dimensions.
 168   """
 169   with ops.name_scope(
 170       name or "broadcast_matrix_batch_dims", values=batch_matrices):
 171     check_ops.assert_proper_iterable(batch_matrices)
 172     batch_matrices = list(batch_matrices)
 173 
 174     for i, mat in enumerate(batch_matrices):
 175       batch_matrices[i] = ops.convert_to_tensor(mat)
 176       assert_is_batch_matrix(batch_matrices[i])
 177 
 178     if len(batch_matrices) < 2:
 179       return batch_matrices
 180 
 181     # Try static broadcasting.
 182     # bcast_batch_shape is the broadcast batch shape of ALL matrices.
 183     # E.g. if batch_matrices = [x, y], with
 184     # x.shape =    [2, j, k]  (batch shape =    [2])
 185     # y.shape = [3, 1, l, m]  (batch shape = [3, 1])
 186     # ==> bcast_batch_shape = [3, 2]
 187     bcast_batch_shape = batch_matrices[0].get_shape()[:-2]
 188     for mat in batch_matrices[1:]:
 189       bcast_batch_shape = array_ops.broadcast_static_shape(
 190           bcast_batch_shape,
 191           mat.get_shape()[:-2])
 192     if bcast_batch_shape.is_fully_defined():
 193       # The [1, 1] at the end will broadcast with anything.
 194       bcast_shape = bcast_batch_shape.concatenate([1, 1])
 195       for i, mat in enumerate(batch_matrices):
 196         if mat.get_shape()[:-2] != bcast_batch_shape:
 197           batch_matrices[i] = _broadcast_to_shape(mat, bcast_shape)
 198       return batch_matrices
 199 
 200     # Since static didn't work, do dynamic, which always copies data.
 201     bcast_batch_shape = array_ops.shape(batch_matrices[0])[:-2]
 202     for mat in batch_matrices[1:]:
 203       bcast_batch_shape = array_ops.broadcast_dynamic_shape(
 204           bcast_batch_shape,
 205           array_ops.shape(mat)[:-2])
 206     bcast_shape = array_ops.concat([bcast_batch_shape, [1, 1]], axis=0)
 207     for i, mat in enumerate(batch_matrices):
 208       batch_matrices[i] = _broadcast_to_shape(mat, bcast_shape)
 209 
 210     return batch_matrices
 211

 213 def _broadcast_to_shape(x, shape):
 214   return x + array_ops.zeros(shape=shape, dtype=x.dtype)

 217 def cholesky_solve_with_broadcast(chol, rhs, name=None):
 218   """Solve systems of linear equations."""
 219   with ops.name_scope(name, "CholeskySolveWithBroadcast", [chol, rhs]):
 220     chol, rhs = broadcast_matrix_batch_dims([chol, rhs])
 221     return linalg_ops.cholesky_solve(chol, rhs)

 224 def matmul_with_broadcast(a,
 225                           b,
 226                           transpose_a=False,
 227                           transpose_b=False,
 228                           adjoint_a=False,
 229                           adjoint_b=False,
 230                           a_is_sparse=False,
 231                           b_is_sparse=False,
 232                           name=None):
 233   """Multiplies matrix `a` by matrix `b`, producing `a @ b`.
 234 
 235   Works identically to `tf.matmul`, but broadcasts batch dims
 236   of `a` and `b` (by replicating) if they are determined statically to be
 237   different, or if static shapes are not fully defined.  Thus, this may result
 238   in an inefficient replication of data.
 239 
 240   The inputs must be matrices (or tensors of rank > 2, representing batches of
 241   matrices).
 242 
 243   Both matrices must be of the same type. The supported types are:
 244   `float16`, `float32`, `float64`, `int32`, `complex64`, `complex128`.
 245 
 246   Either matrix can be transposed or adjointed (conjugated and transposed) on
 247   the fly by setting one of the corresponding flag to `True`. These are `False`
 248   by default.
 249 
 250   If one or both of the matrices contain a lot of zeros, a more efficient
 251   multiplication algorithm can be used by setting the corresponding
 252   `a_is_sparse` or `b_is_sparse` flag to `True`. These are `False` by default.
 253   This optimization is only available for plain matrices (rank-2 tensors) with
 254   datatypes `bfloat16` or `float32`.
 255 
 256   For example:
 257 
 258   ```python
 259   # A 2-batch of 3x4 matrices
 260   a = tf.random_normal(shape=(2, 3, 4))
 261 
 262   # A single 4x5 matrix
 263   b = tf.random_normal(shape=(4, 5))
 264 
 265   result = matmul_with_broadcast(a, b)
 266 
 267   result.shape
 268   ==> (2, 3, 5)
 269 
 270   result[0,...]
 271   ==> tf.matmul(a[0,...], b)
 272 
 273   result[1,...]
 274   ==> tf.matmul(a[1,...], b)
 275   ```
 276 
 277   Args:
 278     a: `Tensor` of type `float16`, `float32`, `float64`, `int32`, `complex64`,
 279       `complex128` and `rank > 1`.
 280     b: `Tensor` with same type as `a` having compatible matrix dimensions and
 281       broadcastable batch dimensions.
 282     transpose_a: If `True`, `a` is transposed before multiplication.
 283     transpose_b: If `True`, `b` is transposed before multiplication.
 284     adjoint_a: If `True`, `a` is conjugated and transposed before
 285       multiplication.
 286     adjoint_b: If `True`, `b` is conjugated and transposed before
 287       multiplication.
 288     a_is_sparse: If `True`, `a` is treated as a sparse matrix.
 289     b_is_sparse: If `True`, `b` is treated as a sparse matrix.
 290     name: Name for the operation (optional).
 291 
 292   Returns:
 293     A `Tensor` of the same type as `a` and `b` where each inner-most matrix is
 294     the product of the corresponding matrices in `a` and `b`, e.g. if all
 295     transpose or adjoint attributes are `False`:
 296 
 297     The leading shape of `output` is the result of broadcasting the leading
 298     dimensions of `a` and `b`.
 299 
 300     `output`[..., i, j] = sum_k (`a`[..., i, k] * `b`[..., k, j]),
 301     for all indices i, j.
 302 
 303     Note: This is matrix product, not element-wise product.
 304 
 305 
 306   Raises:
 307     ValueError: If transpose_a and adjoint_a, or transpose_b and adjoint_b
 308       are both set to True.
 309   """
 310   with ops.name_scope(name, "MatMulWithBroadcast", [a, b]):
 311     a, b = broadcast_matrix_batch_dims([a, b])
 312     return math_ops.matmul(
 313         a,
 314         b,
 315         transpose_a=transpose_a,
 316         transpose_b=transpose_b,
 317         adjoint_a=adjoint_a,
 318         adjoint_b=adjoint_b,
 319         a_is_sparse=a_is_sparse,
 320         b_is_sparse=b_is_sparse)

 323 def matrix_solve_with_broadcast(matrix, rhs, adjoint=False, name=None):
 324   """Solve systems of linear equations."""
 325   with ops.name_scope(name, "MatrixSolveWithBroadcast", [matrix, rhs]):
 326     matrix, rhs = broadcast_matrix_batch_dims([matrix, rhs])
 327     return linalg_ops.matrix_solve(matrix, rhs, adjoint=adjoint)

 330 def matrix_triangular_solve_with_broadcast(matrix,
 331                                            rhs,
 332                                            lower=True,
 333                                            adjoint=False,
 334                                            name=None):
 335   """Solves triangular systems of linear equations with by backsubstitution.
 336 
 337   Works identically to `tf.matrix_triangular_solve`, but broadcasts batch dims
 338   of `matrix` and `rhs` (by replicating) if they are determined statically to be
 339   different, or if static shapes are not fully defined.  Thus, this may result
 340   in an inefficient replication of data.
 341 
 342   Args:
 343     matrix: A Tensor. Must be one of the following types:
 344       `float64`, `float32`, `complex64`, `complex128`. Shape is `[..., M, M]`.
 345     rhs: A `Tensor`. Must have the same `dtype` as `matrix`.
 346       Shape is `[..., M, K]`.
 347     lower: An optional `bool`. Defaults to `True`. Indicates whether the
 348       innermost matrices in `matrix` are lower or upper triangular.
 349     adjoint: An optional `bool`. Defaults to `False`. Indicates whether to solve
 350       with matrix or its (block-wise) adjoint.
 351     name: A name for the operation (optional).
 352 
 353   Returns:
 354     `Tensor` with same `dtype` as `matrix` and shape `[..., M, K]`.
 355   """
 356   with ops.name_scope(name, "MatrixTriangularSolve", [matrix, rhs]):
 357     matrix, rhs = broadcast_matrix_batch_dims([matrix, rhs])
 358     return linalg_ops.matrix_triangular_solve(
 359         matrix,
 360         rhs,
 361         lower=lower,
 362         adjoint=adjoint)
