  42 class LinearOperator(object):
  43   """Base class defining a [batch of] linear operator[s].
  44 
  45   Subclasses of `LinearOperator` provide access to common methods on a
  46   (batch) matrix, without the need to materialize the matrix.  This allows:
  47 
  48   * Matrix free computations
  49   * Operators that take advantage of special structure, while providing a
  50     consistent API to users.
  51 
  52   #### Subclassing
  53 
  54   To enable a public method, subclasses should implement the leading-underscore
  55   version of the method.  The argument signature should be identical except for
  56   the omission of `name="..."`.  For example, to enable
  57   `matmul(x, adjoint=False, name="matmul")` a subclass should implement
  58   `_matmul(x, adjoint=False)`.
  59 
  60   #### Performance contract
  61 
  62   Subclasses should only implement the assert methods
  63   (e.g. `assert_non_singular`) if they can be done in less than `O(N^3)`
  64   time.
  65 
  66   Class docstrings should contain an explanation of computational complexity.
  67   Since this is a high-performance library, attention should be paid to detail,
  68   and explanations can include constants as well as Big-O notation.
  69 
  70   #### Shape compatibility
  71 
  72   `LinearOperator` subclasses should operate on a [batch] matrix with
  73   compatible shape.  Class docstrings should define what is meant by compatible
  74   shape.  Some subclasses may not support batching.
  75 
  76   Examples:
  77 
  78   `x` is a batch matrix with compatible shape for `matmul` if
  79 
  80   ```
  81   operator.shape = [B1,...,Bb] + [M, N],  b >= 0,
  82   x.shape =   [B1,...,Bb] + [N, R]
  83   ```
  84 
  85   `rhs` is a batch matrix with compatible shape for `solve` if
  86 
  87   ```
  88   operator.shape = [B1,...,Bb] + [M, N],  b >= 0,
  89   rhs.shape =   [B1,...,Bb] + [M, R]
  90   ```
  91 
  92   #### Example docstring for subclasses.
  93 
  94   This operator acts like a (batch) matrix `A` with shape
  95   `[B1,...,Bb, M, N]` for some `b >= 0`.  The first `b` indices index a
  96   batch member.  For every batch index `(i1,...,ib)`, `A[i1,...,ib, : :]` is
  97   an `m x n` matrix.  Again, this matrix `A` may not be materialized, but for
  98   purposes of identifying and working with compatible arguments the shape is
  99   relevant.
 100 
 101   Examples:
 102 
 103   ```python
 104   some_tensor = ... shape = ????
 105   operator = MyLinOp(some_tensor)
 106 
 107   operator.shape()
 108   ==> [2, 4, 4]
 109 
 110   operator.log_abs_determinant()
 111   ==> Shape [2] Tensor
 112 
 113   x = ... Shape [2, 4, 5] Tensor
 114 
 115   operator.matmul(x)
 116   ==> Shape [2, 4, 5] Tensor
 117   ```
 118 
 119   #### Shape compatibility
 120 
 121   This operator acts on batch matrices with compatible shape.
 122   FILL IN WHAT IS MEANT BY COMPATIBLE SHAPE
 123 
 124   #### Performance
 125 
 126   FILL THIS IN
 127 
 128   #### Matrix property hints
 129 
 130   This `LinearOperator` is initialized with boolean flags of the form `is_X`,
 131   for `X = non_singular, self_adjoint, positive_definite, square`.
 132   These have the following meaning:
 133 
 134   * If `is_X == True`, callers should expect the operator to have the
 135     property `X`.  This is a promise that should be fulfilled, but is *not* a
 136     runtime assert.  For example, finite floating point precision may result
 137     in these promises being violated.
 138   * If `is_X == False`, callers should expect the operator to not have `X`.
 139   * If `is_X == None` (the default), callers should have no expectation either
 140     way.
 141   """
 142   __metaclass__ = abc.ABCMeta

 144   def __init__(self,
 145                dtype,
 146                graph_parents=None,
 147                is_non_singular=None,
 148                is_self_adjoint=None,
 149                is_positive_definite=None,
 150                is_square=None,
 151                name=None):
 152     r"""Initialize the `LinearOperator`.
 153 
 154     **This is a private method for subclass use.**
 155     **Subclasses should copy-paste this `__init__` documentation.**
 156 
 157     Args:
 158       dtype: The type of the this `LinearOperator`.  Arguments to `matmul` and
 159         `solve` will have to be this type.
 160       graph_parents: Python list of graph prerequisites of this `LinearOperator`
 161         Typically tensors that are passed during initialization.
 162       is_non_singular:  Expect that this operator is non-singular.
 163       is_self_adjoint:  Expect that this operator is equal to its hermitian
 164         transpose.  If `dtype` is real, this is equivalent to being symmetric.
 165       is_positive_definite:  Expect that this operator is positive definite,
 166         meaning the quadratic form `x^H A x` has positive real part for all
 167         nonzero `x`.  Note that we do not require the operator to be
 168         self-adjoint to be positive-definite.  See:
 169         https://en.wikipedia.org/wiki/Positive-definite_matrix#Extension_for_non-symmetric_matrices
 170       is_square:  Expect that this operator acts like square [batch] matrices.
 171       name: A name for this `LinearOperator`.
 172 
 173     Raises:
 174       ValueError:  If any member of graph_parents is `None` or not a `Tensor`.
 175       ValueError:  If hints are set incorrectly.
 176     """
 177     # Check and auto-set flags.
 178     if is_positive_definite:
 179       if is_non_singular is False:
 180         raise ValueError("A positive definite matrix is always non-singular.")
 181       is_non_singular = True
 182 
 183     if is_non_singular:
 184       if is_square is False:
 185         raise ValueError("A non-singular matrix is always square.")
 186       is_square = True
 187 
 188     if is_self_adjoint:
 189       if is_square is False:
 190         raise ValueError("A self-adjoint matrix is always square.")
 191       is_square = True
 192 
 193     self._is_square_set_or_implied_by_hints = is_square
 194 
 195     graph_parents = [] if graph_parents is None else graph_parents
 196     for i, t in enumerate(graph_parents):
 197       if t is None or not tensor_util.is_tensor(t):
 198         raise ValueError("Graph parent item %d is not a Tensor; %s." % (i, t))
 199     self._dtype = dtype
 200     self._graph_parents = graph_parents
 201     self._is_non_singular = is_non_singular
 202     self._is_self_adjoint = is_self_adjoint
 203     self._is_positive_definite = is_positive_definite
 204     self._name = name or type(self).__name__
 205 
 206   @contextlib.contextmanager
 207   def _name_scope(self, name=None, values=None):
 208     """Helper function to standardize op scope."""
 209     with ops.name_scope(self.name):
 210       with ops.name_scope(
 211           name, values=((values or []) + self._graph_parents)) as scope:
 212         yield scope

 241   @property
 242   def is_square(self):
 243     """Return `True/False` depending on if this operator is square."""
 244     # Static checks done after __init__.  Why?  Because domain/range dimension
 245     # sometimes requires lots of work done in the derived class after init.
 246     auto_square_check = self.domain_dimension == self.range_dimension
 247     if self._is_square_set_or_implied_by_hints is False and auto_square_check:
 248       raise ValueError(
 249           "User set is_square hint to False, but the operator was square.")
 250     if self._is_square_set_or_implied_by_hints is None:
 251       return auto_square_check
 252 
 253     return self._is_square_set_or_implied_by_hints

 440   def _assert_non_singular(self):
 441     """Private default implementation of _assert_non_singular."""
 442     logging.warn(
 443         "Using (possibly slow) default implementation of assert_non_singular."
 444         "  Requires conversion to a dense matrix and O(N^3) operations.")
 445     if self._can_use_cholesky():
 446       return self.assert_positive_definite()
 447     else:
 448       singular_values = linalg_ops.svd(self.to_dense(), compute_uv=False)
 449       # TODO(langmore) Add .eig and .cond as methods.
 450       cond = (math_ops.reduce_max(singular_values, axis=-1) /
 451               math_ops.reduce_min(singular_values, axis=-1))
 452       return check_ops.assert_less(
 453           cond,
 454           self._max_condition_number_to_be_non_singular(),
 455           message="Singular matrix up to precision epsilon.")
 456 
 457   def _max_condition_number_to_be_non_singular(self):
 458     """Return the maximum condition number that we consider nonsingular."""
 459     with ops.name_scope("max_nonsingular_condition_number"):
 460       dtype_eps = np.finfo(self.dtype.as_numpy_dtype).eps
 461       eps = math_ops.cast(
 462           math_ops.reduce_max([
 463               100.,
 464               math_ops.cast(self.range_dimension_tensor(), self.dtype),
 465               math_ops.cast(self.domain_dimension_tensor(), self.dtype)
 466           ]), self.dtype) * dtype_eps
 467       return 1. / eps

 470     """Returns an `Op` that asserts this operator is non singular.

 472     This operator is considered non-singular if
 473 
 474     ```

 477     ```

 479     Args:
 480       name:  A string name to prepend to created ops.
 481 
 482     Returns:
 483       An `Assert` `Op`, that, when run, will raise an `InvalidArgumentError` if
 484         the operator is singular.
 485     """
 486     with self._name_scope(name):
 487       return self._assert_non_singular()

 489   def _assert_positive_definite(self):
 490     """Default implementation of _assert_positive_definite."""
 491     logging.warn(
 492         "Using (possibly slow) default implementation of "
 493         "assert_positive_definite."
 494         "  Requires conversion to a dense matrix and O(N^3) operations.")
 495     # If the operator is self-adjoint, then checking that
 496     # Cholesky decomposition succeeds + results in positive diag is necessary
 497     # and sufficient.
 498     if self.is_self_adjoint:
 499       return check_ops.assert_positive(
 500           array_ops.matrix_diag_part(linalg_ops.cholesky(self.to_dense())),
 501           message="Matrix was not positive definite.")
 502     # We have no generic check for positive definite.
 503     raise NotImplementedError("assert_positive_definite is not implemented.")

 513       name:  A name to give this `Op`.

 559   def matmul(self, x, adjoint=False, adjoint_arg=False, name="matmul"):
 560     """Transform [batch] matrix `x` with left multiplication:  `x --> Ax`.
 561 
 562     ```python
 563     # Make an operator acting like batch matrix A.  Assume A.shape = [..., M, N]
 564     operator = LinearOperator(...)
 565     operator.shape = [..., M, N]
 566 
 567     X = ... # shape [..., N, R], batch matrix, R > 0.
 568 
 569     Y = operator.matmul(X)
 570     Y.shape
 571     ==> [..., M, R]
 572 
 573     Y[..., :, r] = sum_j A[..., :, j] X[j, r]
 574     ```
 575 
 576     Args:
 577       x: `Tensor` with compatible shape and same `dtype` as `self`.
 578         See class docstring for definition of compatibility.
 579       adjoint: Python `bool`.  If `True`, left multiply by the adjoint: `A^H x`.
 580       adjoint_arg:  Python `bool`.  If `True`, compute `A x^H` where `x^H` is
 581         the hermitian transpose (transposition and complex conjugation).
 582       name:  A name for this `Op.
 583 
 584     Returns:
 585       A `Tensor` with shape `[..., M, R]` and same `dtype` as `self`.
 586     """
 587     with self._name_scope(name, values=[x]):
 588       x = ops.convert_to_tensor(x, name="x")
 589       self._check_input_dtype(x)
 590 
 591       self_dim = -2 if adjoint else -1
 592       arg_dim = -1 if adjoint_arg else -2
 593       self.shape[self_dim].assert_is_compatible_with(x.get_shape()[arg_dim])
 594 
 595       return self._matmul(x, adjoint=adjoint, adjoint_arg=adjoint_arg)
 596 
 597   def _matvec(self, x, adjoint=False):
 598     x_mat = array_ops.expand_dims(x, axis=-1)
 599     y_mat = self.matmul(x_mat, adjoint=adjoint)
 600     return array_ops.squeeze(y_mat, axis=-1)

 602   def matvec(self, x, adjoint=False, name="matvec"):
 603     """Transform [batch] vector `x` with left multiplication:  `x --> Ax`.
 604 
 605     ```python
 606     # Make an operator acting like batch matric A.  Assume A.shape = [..., M, N]
 607     operator = LinearOperator(...)
 608 
 609     X = ... # shape [..., N], batch vector
 610 
 611     Y = operator.matvec(X)
 612     Y.shape
 613     ==> [..., M]
 614 
 615     Y[..., :] = sum_j A[..., :, j] X[..., j]
 616     ```
 617 
 618     Args:
 619       x: `Tensor` with compatible shape and same `dtype` as `self`.
 620         `x` is treated as a [batch] vector meaning for every set of leading
 621         dimensions, the last dimension defines a vector.
 622         See class docstring for definition of compatibility.
 623       adjoint: Python `bool`.  If `True`, left multiply by the adjoint: `A^H x`.
 624       name:  A name for this `Op.
 625 
 626     Returns:
 627       A `Tensor` with shape `[..., M]` and same `dtype` as `self`.
 628     """
 629     with self._name_scope(name, values=[x]):
 630       x = ops.convert_to_tensor(x, name="x")
 631       self._check_input_dtype(x)
 632       self_dim = -2 if adjoint else -1
 633       self.shape[self_dim].assert_is_compatible_with(x.get_shape()[-1])
 634       return self._matvec(x, adjoint=adjoint)

 692   def _solve(self, rhs, adjoint=False, adjoint_arg=False):
 693     """Default implementation of _solve."""
 694     if self.is_square is False:
 695       raise NotImplementedError(
 696           "Solve is not yet implemented for non-square operators.")
 697     logging.warn(
 698         "Using (possibly slow) default implementation of solve."
 699         "  Requires conversion to a dense matrix and O(N^3) operations.")
 700     rhs = linalg.adjoint(rhs) if adjoint_arg else rhs
 701     if self._can_use_cholesky():
 702       return linear_operator_util.cholesky_solve_with_broadcast(
 703           linalg_ops.cholesky(self.to_dense()), rhs)
 704     return linear_operator_util.matrix_solve_with_broadcast(
 705         self.to_dense(), rhs, adjoint=adjoint)

 707   def solve(self, rhs, adjoint=False, adjoint_arg=False, name="solve"):
 708     """Solve (exact or approx) `R` (batch) systems of equations: `A X = rhs`.
 709 
 710     The returned `Tensor` will be close to an exact solution if `A` is well
 711     conditioned. Otherwise closeness will vary. See class docstring for details.
 712 
 713     Examples:
 714 
 715     ```python
 716     # Make an operator acting like batch matrix A.  Assume A.shape = [..., M, N]
 717     operator = LinearOperator(...)
 718     operator.shape = [..., M, N]
 719 
 720     # Solve R > 0 linear systems for every member of the batch.
 721     RHS = ... # shape [..., M, R]
 722 
 723     X = operator.solve(RHS)
 724     # X[..., :, r] is the solution to the r'th linear system
 725     # sum_j A[..., :, j] X[..., j, r] = RHS[..., :, r]
 726 
 727     operator.matmul(X)
 728     ==> RHS
 729     ```
 730 
 731     Args:
 732       rhs: `Tensor` with same `dtype` as this operator and compatible shape.
 733         `rhs` is treated like a [batch] matrix meaning for every set of leading
 734         dimensions, the last two dimensions defines a matrix.
 735         See class docstring for definition of compatibility.
 736       adjoint: Python `bool`.  If `True`, solve the system involving the adjoint
 737         of this `LinearOperator`:  `A^H X = rhs`.
 738       adjoint_arg:  Python `bool`.  If `True`, solve `A X = rhs^H` where `rhs^H`
 739         is the hermitian transpose (transposition and complex conjugation).
 740       name:  A name scope to use for ops added by this method.
 741 
 742     Returns:
 743       `Tensor` with shape `[...,N, R]` and same `dtype` as `rhs`.
 744 
 745     Raises:
 746       NotImplementedError:  If `self.is_non_singular` or `is_square` is False.
 747     """
 748     if self.is_non_singular is False:
 749       raise NotImplementedError(
 750           "Exact solve not implemented for an operator that is expected to "
 751           "be singular.")
 752     if self.is_square is False:
 753       raise NotImplementedError(
 754           "Exact solve not implemented for an operator that is expected to "
 755           "not be square.")
 756     with self._name_scope(name, values=[rhs]):
 757       rhs = ops.convert_to_tensor(rhs, name="rhs")
 758       self._check_input_dtype(rhs)
 759 
 760       self_dim = -1 if adjoint else -2
 761       arg_dim = -1 if adjoint_arg else -2
 762       self.shape[self_dim].assert_is_compatible_with(rhs.get_shape()[arg_dim])
 763 
 764       return self._solve(rhs, adjoint=adjoint, adjoint_arg=adjoint_arg)
 765 
 766   def _solvevec(self, rhs, adjoint=False):
 767     """Default implementation of _solvevec."""
 768     rhs_mat = array_ops.expand_dims(rhs, axis=-1)
 769     solution_mat = self.solve(rhs_mat, adjoint=adjoint)
 770     return array_ops.squeeze(solution_mat, axis=-1)

 772   def solvevec(self, rhs, adjoint=False, name="solve"):
 773     """Solve single equation with best effort: `A X = rhs`.
 774 
 775     The returned `Tensor` will be close to an exact solution if `A` is well
 776     conditioned. Otherwise closeness will vary. See class docstring for details.
 777 
 778     Examples:
 779 
 780     ```python
 781     # Make an operator acting like batch matrix A.  Assume A.shape = [..., M, N]
 782     operator = LinearOperator(...)
 783     operator.shape = [..., M, N]
 784 
 785     # Solve one linear system for every member of the batch.
 786     RHS = ... # shape [..., M]
 787 
 788     X = operator.solvevec(RHS)
 789     # X is the solution to the linear system
 790     # sum_j A[..., :, j] X[..., j] = RHS[..., :]
 791 
 792     operator.matvec(X)
 793     ==> RHS
 794     ```
 795 
 796     Args:
 797       rhs: `Tensor` with same `dtype` as this operator.
 798         `rhs` is treated like a [batch] vector meaning for every set of leading
 799         dimensions, the last dimension defines a vector.  See class docstring
 800         for definition of compatibility regarding batch dimensions.
 801       adjoint: Python `bool`.  If `True`, solve the system involving the adjoint
 802         of this `LinearOperator`:  `A^H X = rhs`.
 803       name:  A name scope to use for ops added by this method.
 804 
 805     Returns:
 806       `Tensor` with shape `[...,N]` and same `dtype` as `rhs`.
 807 
 808     Raises:
 809       NotImplementedError:  If `self.is_non_singular` or `is_square` is False.
 810     """
 811     with self._name_scope(name, values=[rhs]):
 812       rhs = ops.convert_to_tensor(rhs, name="rhs")
 813       self._check_input_dtype(rhs)
 814       self_dim = -1 if adjoint else -2
 815       self.shape[self_dim].assert_is_compatible_with(rhs.get_shape()[-1])
 816 
 817       return self._solvevec(rhs, adjoint=adjoint)

 819   def _to_dense(self):
 820     """Generic and often inefficient implementation.  Override often."""
 821     logging.warn("Using (possibly slow) default implementation of to_dense."
 822                  "  Converts by self.matmul(identity).")
 823     if self.batch_shape.is_fully_defined():
 824       batch_shape = self.batch_shape
 825     else:
 826       batch_shape = self.batch_shape_tensor()
 827 
 828     if self.domain_dimension.value is not None:
 829       n = self.domain_dimension.value
 830     else:
 831       n = self.domain_dimension_tensor()
 832 
 833     eye = linalg_ops.eye(num_rows=n, batch_shape=batch_shape, dtype=self.dtype)
 834     return self.matmul(eye)
