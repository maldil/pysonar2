  60 @tf_export('linalg.logdet')
  61 def logdet(matrix, name=None):
  62   """Computes log of the determinant of a hermitian positive definite matrix.
  63 
  64   ```python
  65   # Compute the determinant of a matrix while reducing the chance of over- or
  66   underflow:
  67   A = ... # shape 10 x 10
  68   det = tf.exp(tf.logdet(A))  # scalar
  69   ```
  70 
  71   Args:
  72     matrix:  A `Tensor`. Must be `float16`, `float32`, `float64`, `complex64`,
  73       or `complex128` with shape `[..., M, M]`.
  74     name:  A name to give this `Op`.  Defaults to `logdet`.
  75 
  76   Returns:
  77     The natural log of the determinant of `matrix`.
  78 
  79   @compatibility(numpy)
  80   Equivalent to numpy.linalg.slogdet, although no sign is returned since only
  81   hermitian positive definite matrices are supported.
  82   @end_compatibility
  83   """
  84   # This uses the property that the log det(A) = 2*sum(log(real(diag(C))))
  85   # where C is the cholesky decomposition of A.
  86   with ops.name_scope(name, 'logdet', [matrix]):
  87     chol = gen_linalg_ops.cholesky(matrix)
  88     return 2.0 * math_ops.reduce_sum(
  89         math_ops.log(math_ops.real(array_ops.matrix_diag_part(chol))),
  90         reduction_indices=[-1])

  93 @tf_export('linalg.adjoint')
  94 def adjoint(matrix, name=None):
  95   """Transposes the last two dimensions of and conjugates tensor `matrix`.
  96 
  97   For example:
  98 
  99   ```python
 100   x = tf.constant([[1 + 1j, 2 + 2j, 3 + 3j],
 101                    [4 + 4j, 5 + 5j, 6 + 6j]])
 102   tf.linalg.adjoint(x)  # [[1 - 1j, 4 - 4j],
 103                         #  [2 - 2j, 5 - 5j],
 104                         #  [3 - 3j, 6 - 6j]]
 105 
 106   Args:
 107     matrix:  A `Tensor`. Must be `float16`, `float32`, `float64`, `complex64`,
 108       or `complex128` with shape `[..., M, M]`.
 109     name:  A name to give this `Op` (optional).
 110 
 111   Returns:
 112     The adjoint (a.k.a. Hermitian transpose a.k.a. conjugate transpose) of
 113     matrix.
 114   """
 115   with ops.name_scope(name, 'adjoint', [matrix]):
 116     matrix = ops.convert_to_tensor(matrix, name='matrix')
 117     return array_ops.matrix_transpose(matrix, conjugate=True)

 120 # This section is ported nearly verbatim from Eigen's implementation:
 121 # https://eigen.tuxfamily.org/dox/unsupported/MatrixExponential_8h_source.html
 122 def _matrix_exp_pade3(matrix):
 123   """3rd-order Pade approximant for matrix exponential."""
 124   b = [120.0, 60.0, 12.0]
 125   b = [constant_op.constant(x, matrix.dtype) for x in b]
 126   ident = linalg_ops.eye(array_ops.shape(matrix)[-2],
 127                          batch_shape=array_ops.shape(matrix)[:-2],
 128                          dtype=matrix.dtype)
 129   matrix_2 = math_ops.matmul(matrix, matrix)
 130   tmp = matrix_2 + b[1] * ident
 131   matrix_u = math_ops.matmul(matrix, tmp)
 132   matrix_v = b[2] * matrix_2 + b[0] * ident
 133   return matrix_u, matrix_v

 136 def _matrix_exp_pade5(matrix):
 137   """5th-order Pade approximant for matrix exponential."""
 138   b = [30240.0, 15120.0, 3360.0, 420.0, 30.0]
 139   b = [constant_op.constant(x, matrix.dtype) for x in b]
 140   ident = linalg_ops.eye(array_ops.shape(matrix)[-2],
 141                          batch_shape=array_ops.shape(matrix)[:-2],
 142                          dtype=matrix.dtype)
 143   matrix_2 = math_ops.matmul(matrix, matrix)
 144   matrix_4 = math_ops.matmul(matrix_2, matrix_2)
 145   tmp = matrix_4 + b[3] * matrix_2 + b[1] * ident
 146   matrix_u = math_ops.matmul(matrix, tmp)
 147   matrix_v = b[4] * matrix_4 + b[2] * matrix_2 + b[0] * ident
 148   return matrix_u, matrix_v

 151 def _matrix_exp_pade7(matrix):
 152   """7th-order Pade approximant for matrix exponential."""
 153   b = [17297280.0, 8648640.0, 1995840.0, 277200.0, 25200.0, 1512.0, 56.0]
 154   b = [constant_op.constant(x, matrix.dtype) for x in b]
 155   ident = linalg_ops.eye(array_ops.shape(matrix)[-2],
 156                          batch_shape=array_ops.shape(matrix)[:-2],
 157                          dtype=matrix.dtype)
 158   matrix_2 = math_ops.matmul(matrix, matrix)
 159   matrix_4 = math_ops.matmul(matrix_2, matrix_2)
 160   matrix_6 = math_ops.matmul(matrix_4, matrix_2)
 161   tmp = matrix_6 + b[5] * matrix_4 + b[3] * matrix_2 + b[1] * ident
 162   matrix_u = math_ops.matmul(matrix, tmp)
 163   matrix_v = b[6] * matrix_6 + b[4] * matrix_4 + b[2] * matrix_2 + b[0] * ident
 164   return matrix_u, matrix_v

 167 def _matrix_exp_pade9(matrix):
 168   """9th-order Pade approximant for matrix exponential."""
 169   b = [
 170       17643225600.0, 8821612800.0, 2075673600.0, 302702400.0, 30270240.0,
 171       2162160.0, 110880.0, 3960.0, 90.0
 172   ]
 173   b = [constant_op.constant(x, matrix.dtype) for x in b]
 174   ident = linalg_ops.eye(array_ops.shape(matrix)[-2],
 175                          batch_shape=array_ops.shape(matrix)[:-2],
 176                          dtype=matrix.dtype)
 177   matrix_2 = math_ops.matmul(matrix, matrix)
 178   matrix_4 = math_ops.matmul(matrix_2, matrix_2)
 179   matrix_6 = math_ops.matmul(matrix_4, matrix_2)
 180   matrix_8 = math_ops.matmul(matrix_6, matrix_2)
 181   tmp = (
 182       matrix_8 + b[7] * matrix_6 + b[5] * matrix_4 + b[3] * matrix_2 +
 183       b[1] * ident)
 184   matrix_u = math_ops.matmul(matrix, tmp)
 185   matrix_v = (
 186       b[8] * matrix_8 + b[6] * matrix_6 + b[4] * matrix_4 + b[2] * matrix_2 +
 187       b[0] * ident)
 188   return matrix_u, matrix_v

 191 def _matrix_exp_pade13(matrix):
 192   """13th-order Pade approximant for matrix exponential."""
 193   b = [
 194       64764752532480000.0, 32382376266240000.0, 7771770303897600.0,
 195       1187353796428800.0, 129060195264000.0, 10559470521600.0, 670442572800.0,
 196       33522128640.0, 1323241920.0, 40840800.0, 960960.0, 16380.0, 182.0
 197   ]
 198   b = [constant_op.constant(x, matrix.dtype) for x in b]
 199   ident = linalg_ops.eye(array_ops.shape(matrix)[-2],
 200                          batch_shape=array_ops.shape(matrix)[:-2],
 201                          dtype=matrix.dtype)
 202   matrix_2 = math_ops.matmul(matrix, matrix)
 203   matrix_4 = math_ops.matmul(matrix_2, matrix_2)
 204   matrix_6 = math_ops.matmul(matrix_4, matrix_2)
 205   tmp_u = (
 206       math_ops.matmul(matrix_6,
 207                       matrix_6 + b[11] * matrix_4 + b[9] * matrix_2) +
 208       b[7] * matrix_6 + b[5] * matrix_4 + b[3] * matrix_2 + b[1] * ident)
 209   matrix_u = math_ops.matmul(matrix, tmp_u)
 210   tmp_v = b[12] * matrix_6 + b[10] * matrix_4 + b[8] * matrix_2
 211   matrix_v = (
 212       math_ops.matmul(matrix_6, tmp_v) + b[6] * matrix_6 + b[4] * matrix_4 +
 213       b[2] * matrix_2 + b[0] * ident)
 214   return matrix_u, matrix_v

 217 @tf_export('linalg.expm')
 218 def matrix_exponential(input, name=None):  # pylint: disable=redefined-builtin
 219   r"""Computes the matrix exponential of one or more square matrices.
 220 
 221   exp(A) = \sum_{n=0}^\infty A^n/n!
 222 
 223   The exponential is computed using a combination of the scaling and squaring
 224   method and the Pade approximation. Details can be found in:
 225   Nicholas J. Higham, "The scaling and squaring method for the matrix
 226   exponential revisited," SIAM J. Matrix Anal. Applic., 26:1179-1193, 2005.
 227 
 228   The input is a tensor of shape `[..., M, M]` whose inner-most 2 dimensions
 229   form square matrices. The output is a tensor of the same shape as the input
 230   containing the exponential for all input submatrices `[..., :, :]`.
 231 
 232   Args:
 233     input: A `Tensor`. Must be `float16`, `float32`, `float64`, `complex64`,
 234       or `complex128` with shape `[..., M, M]`.
 235     name:  A name to give this `Op` (optional).
 236 
 237   Returns:
 238     the matrix exponential of the input.
 239 
 240   Raises:
 241     ValueError: An unsupported type is provided as input.
 242 
 243   @compatibility(scipy)
 244   Equivalent to scipy.linalg.expm
 245   @end_compatibility
 246   """
 247   with ops.name_scope(name, 'matrix_exponential', [input]):
 248     matrix = ops.convert_to_tensor(input, name='input')
 249     if matrix.shape[-2:] == [0, 0]:
 250       return matrix
 251     batch_shape = matrix.shape[:-2]
 252     if not batch_shape.is_fully_defined():
 253       batch_shape = array_ops.shape(matrix)[:-2]
 254 
 255     # reshaping the batch makes the where statements work better
 256     matrix = array_ops.reshape(
 257         matrix, array_ops.concat(([-1], array_ops.shape(matrix)[-2:]), axis=0))
 258     l1_norm = math_ops.reduce_max(
 259         math_ops.reduce_sum(math_ops.abs(matrix),
 260                             axis=array_ops.size(array_ops.shape(matrix)) - 2),
 261         axis=-1)
 262     const = lambda x: constant_op.constant(x, l1_norm.dtype)
 263     def _nest_where(vals, cases):
 264       assert len(vals) == len(cases) - 1
 265       if len(vals) == 1:
 266         return array_ops.where(
 267             math_ops.less(l1_norm, const(vals[0])), cases[0], cases[1])
 268       else:
 269         return array_ops.where(
 270             math_ops.less(l1_norm, const(vals[0])), cases[0],
 271             _nest_where(vals[1:], cases[1:]))
 272 
 273     if matrix.dtype in [dtypes.float16, dtypes.float32, dtypes.complex64]:
 274       maxnorm = const(3.925724783138660)
 275       squarings = math_ops.maximum(
 276           math_ops.floor(
 277               math_ops.log(l1_norm / maxnorm) / math_ops.log(const(2.0))), 0)
 278       u3, v3 = _matrix_exp_pade3(matrix)
 279       u5, v5 = _matrix_exp_pade5(matrix)
 280       u7, v7 = _matrix_exp_pade7(
 281           matrix / math_ops.pow(
 282               constant_op.constant(2.0, dtype=matrix.dtype),
 283               math_ops.cast(squarings, matrix.dtype))[...,
 284                                                       array_ops.newaxis,
 285                                                       array_ops.newaxis])
 286       conds = (4.258730016922831e-001, 1.880152677804762e+000)
 287       u = _nest_where(conds, (u3, u5, u7))
 288       v = _nest_where(conds, (v3, v5, v7))
 289     elif matrix.dtype in [dtypes.float64, dtypes.complex128]:
 290       maxnorm = const(5.371920351148152)
 291       squarings = math_ops.maximum(
 292           math_ops.floor(
 293               math_ops.log(l1_norm / maxnorm) / math_ops.log(const(2.0))), 0)
 294       u3, v3 = _matrix_exp_pade3(matrix)
 295       u5, v5 = _matrix_exp_pade5(matrix)
 296       u7, v7 = _matrix_exp_pade7(matrix)
 297       u9, v9 = _matrix_exp_pade9(matrix)
 298       u13, v13 = _matrix_exp_pade13(
 299           matrix / math_ops.pow(
 300               constant_op.constant(2.0, dtype=matrix.dtype),
 301               math_ops.cast(squarings, matrix.dtype))[...,
 302                                                       array_ops.newaxis,
 303                                                       array_ops.newaxis])
 304       conds = (1.495585217958292e-002,
 305                2.539398330063230e-001,
 306                9.504178996162932e-001,
 307                2.097847961257068e+000)
 308       u = _nest_where(conds, (u3, u5, u7, u9, u13))
 309       v = _nest_where(conds, (v3, v5, v7, v9, v13))
 310     else:
 311       raise ValueError(
 312           'tf.linalg.expm does not support matrices of type %s' % matrix.dtype)
 313     numer = u + v
 314     denom = -u + v
 315     result = linalg_ops.matrix_solve(denom, numer)
 316     max_squarings = math_ops.reduce_max(squarings)
 317 
 318     i = const(0.0)
 319     c = lambda i, r: math_ops.less(i, max_squarings)
 320     def b(i, r):
 321       return i+1, array_ops.where(math_ops.less(i, squarings),
 322                                   math_ops.matmul(r, r), r)
 323     _, result = control_flow_ops.while_loop(c, b, [i, result])
 324     if not matrix.shape.is_fully_defined():
 325       return array_ops.reshape(
 326           result,
 327           array_ops.concat((batch_shape, array_ops.shape(result)[-2:]), axis=0))
 328     return array_ops.reshape(result, batch_shape.concatenate(result.shape[-2:]))
