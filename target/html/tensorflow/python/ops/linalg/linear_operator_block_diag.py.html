  37 @tf_export("linalg.LinearOperatorBlockDiag")
  38 class LinearOperatorBlockDiag(linear_operator.LinearOperator):
  39   """Combines one or more `LinearOperators` in to a Block Diagonal matrix.
  40 
  41   This operator combines one or more linear operators `[op1,...,opJ]`,
  42   building a new `LinearOperator`, whose underlying matrix representation is
  43   square and has each operator `opi` on the main diagonal, and zero's elsewhere.
  44 
  45   #### Shape compatibility
  46 
  47   If `opj` acts like a [batch] square matrix `Aj`, then `op_combined` acts like
  48   the [batch] square matrix formed by having each matrix `Aj` on the main
  49   diagonal.
  50 
  51 
  52   Each `opj` is required to represent a square matrix, and hence will have
  53   shape `batch_shape_j + [M_j, M_j]`.
  54 
  55   If `opj` has shape `batch_shape_j + [M_j, M_j]`, then the combined operator
  56   has shape `broadcast_batch_shape + [sum M_j, sum M_j]`, where
  57   `broadcast_batch_shape` is the mutual broadcast of `batch_shape_j`,
  58   `j = 1,...,J`, assuming the intermediate batch shapes broadcast.
  59   Even if the combined shape is well defined, the combined operator's
  60   methods may fail due to lack of broadcasting ability in the defining
  61   operators' methods.
  62 
  63   ```python
  64   # Create a 4 x 4 linear operator combined of two 2 x 2 operators.
  65   operator_1 = LinearOperatorFullMatrix([[1., 2.], [3., 4.]])
  66   operator_2 = LinearOperatorFullMatrix([[1., 0.], [0., 1.]])
  67   operator = LinearOperatorBlockDiag([operator_1, operator_2])
  68 
  69   operator.to_dense()
  70   ==> [[1., 2., 0., 0.],
  71        [3., 4., 0., 0.],
  72        [0., 0., 1., 0.],
  73        [0., 0., 0., 1.]]
  74 
  75   operator.shape
  76   ==> [4, 4]
  77 
  78   operator.log_abs_determinant()
  79   ==> scalar Tensor
  80 
  81   x1 = ... # Shape [2, 2] Tensor
  82   x2 = ... # Shape [2, 2] Tensor
  83   x = tf.concat([x1, x2], 0)  # Shape [2, 4] Tensor
  84   operator.matmul(x)
  85   ==> tf.concat([operator_1.matmul(x1), operator_2.matmul(x2)])
  86 
  87   # Create a [2, 3] batch of 4 x 4 linear operators.
  88   matrix_44 = tf.random_normal(shape=[2, 3, 4, 4])
  89   operator_44 = LinearOperatorFullMatrix(matrix)
  90 
  91   # Create a [1, 3] batch of 5 x 5 linear operators.
  92   matrix_55 = tf.random_normal(shape=[1, 3, 5, 5])
  93   operator_55 = LinearOperatorFullMatrix(matrix_55)
  94 
  95   # Combine to create a [2, 3] batch of 9 x 9 operators.
  96   operator_99 = LinearOperatorBlockDiag([operator_44, operator_55])
  97 
  98   # Create a shape [2, 3, 9] vector.
  99   x = tf.random_normal(shape=[2, 3, 9])
 100   operator_99.matmul(x)
 101   ==> Shape [2, 3, 9] Tensor
 102   ```
 103 
 104   #### Performance
 105 
 106   The performance of `LinearOperatorBlockDiag` on any operation is equal to
 107   the sum of the individual operators' operations.
 108 
 109 
 110   #### Matrix property hints
 111 
 112   This `LinearOperator` is initialized with boolean flags of the form `is_X`,
 113   for `X = non_singular, self_adjoint, positive_definite, square`.
 114   These have the following meaning:
 115 
 116   * If `is_X == True`, callers should expect the operator to have the
 117     property `X`.  This is a promise that should be fulfilled, but is *not* a
 118     runtime assert.  For example, finite floating point precision may result
 119     in these promises being violated.
 120   * If `is_X == False`, callers should expect the operator to not have `X`.
 121   * If `is_X == None` (the default), callers should have no expectation either
 122     way.
 123   """
 124 
 125   def __init__(self,
 126                operators,
 127                is_non_singular=None,
 128                is_self_adjoint=None,
 129                is_positive_definite=None,
 130                is_square=True,
 131                name=None):
 132     r"""Initialize a `LinearOperatorBlockDiag`.
 133 
 134     `LinearOperatorBlockDiag` is initialized with a list of operators
 135     `[op_1,...,op_J]`.
 136 
 137     Args:
 138       operators:  Iterable of `LinearOperator` objects, each with
 139         the same `dtype` and composable shape.
 140       is_non_singular:  Expect that this operator is non-singular.
 141       is_self_adjoint:  Expect that this operator is equal to its hermitian
 142         transpose.
 143       is_positive_definite:  Expect that this operator is positive definite,
 144         meaning the quadratic form `x^H A x` has positive real part for all
 145         nonzero `x`.  Note that we do not require the operator to be
 146         self-adjoint to be positive-definite.  See:
 147         https://en.wikipedia.org/wiki/Positive-definite_matrix#Extension_for_non-symmetric_matrices
 148       is_square:  Expect that this operator acts like square [batch] matrices.
 149         This is true by default, and will raise a `ValueError` otherwise.
 150       name: A name for this `LinearOperator`.  Default is the individual
 151         operators names joined with `_o_`.
 152 
 153     Raises:
 154       TypeError:  If all operators do not have the same `dtype`.
 155       ValueError:  If `operators` is empty or are non-square.
 156     """
 157     # Validate operators.
 158     check_ops.assert_proper_iterable(operators)
 159     operators = list(operators)
 160     if not operators:
 161       raise ValueError(
 162           "Expected a non-empty list of operators. Found: %s" % operators)
 163     self._operators = operators
 164 
 165     # Validate dtype.
 166     dtype = operators[0].dtype
 167     for operator in operators:
 168       if operator.dtype != dtype:
 169         name_type = (str((o.name, o.dtype)) for o in operators)
 170         raise TypeError(
 171             "Expected all operators to have the same dtype.  Found %s"
 172             % "   ".join(name_type))
 173 
 174     # Auto-set and check hints.
 175     if all(operator.is_non_singular for operator in operators):
 176       if is_non_singular is False:
 177         raise ValueError(
 178             "The direct sum of non-singular operators is always non-singular.")
 179       is_non_singular = True
 180 
 181     if all(operator.is_self_adjoint for operator in operators):
 182       if is_self_adjoint is False:
 183         raise ValueError(
 184             "The direct sum of self-adjoint operators is always self-adjoint.")
 185       is_self_adjoint = True
 186 
 187     if all(operator.is_positive_definite for operator in operators):
 188       if is_positive_definite is False:
 189         raise ValueError(
 190             "The direct sum of positive definite operators is always "
 191             "positive definite.")
 192       is_positive_definite = True
 193 
 194     if not (is_square and all(operator.is_square for operator in operators)):
 195       raise ValueError(
 196           "Can only represent a block diagonal of square matrices.")
 197 
 198     # Initialization.
 199     graph_parents = []
 200     for operator in operators:
 201       graph_parents.extend(operator.graph_parents)
 202 
 203     if name is None:
 204       # Using ds to mean direct sum.
 205       name = "_ds_".join(operator.name for operator in operators)
 206     with ops.name_scope(name, values=graph_parents):
 207       super(LinearOperatorBlockDiag, self).__init__(
 208           dtype=dtype,
 209           graph_parents=graph_parents,
 210           is_non_singular=is_non_singular,
 211           is_self_adjoint=is_self_adjoint,
 212           is_positive_definite=is_positive_definite,
 213           is_square=True,
 214           name=name)
 215 
 216   @property
 217   def operators(self):
 218     return self._operators
 219 
 220   def _shape(self):
 221     # Get final matrix shape.
 222     domain_dimension = self.operators[0].domain_dimension
 223     range_dimension = self.operators[0].range_dimension
 224     for operator in self.operators[1:]:
 225       domain_dimension += operator.domain_dimension
 226       range_dimension += operator.range_dimension
 227 
 228     matrix_shape = tensor_shape.TensorShape([domain_dimension, range_dimension])
 229 
 230     # Get broadcast batch shape.
 231     # broadcast_shape checks for compatibility.
 232     batch_shape = self.operators[0].batch_shape
 233     for operator in self.operators[1:]:
 234       batch_shape = common_shapes.broadcast_shape(
 235           batch_shape, operator.batch_shape)
 236 
 237     return batch_shape.concatenate(matrix_shape)
 238 
 239   def _shape_tensor(self):
 240     # Avoid messy broadcasting if possible.
 241     if self.shape.is_fully_defined():
 242       return ops.convert_to_tensor(
 243           self.shape.as_list(), dtype=dtypes.int32, name="shape")
 244 
 245     domain_dimension = self.operators[0].domain_dimension_tensor()
 246     range_dimension = self.operators[0].range_dimension_tensor()
 247     for operator in self.operators[1:]:
 248       domain_dimension += operator.domain_dimension_tensor()
 249       range_dimension += operator.range_dimension_tensor()
 250 
 251     matrix_shape = array_ops.stack([domain_dimension, range_dimension])
 252 
 253     # Dummy Tensor of zeros.  Will never be materialized.
 254     zeros = array_ops.zeros(shape=self.operators[0].batch_shape_tensor())
 255     for operator in self.operators[1:]:
 256       zeros += array_ops.zeros(shape=operator.batch_shape_tensor())
 257     batch_shape = array_ops.shape(zeros)
 258 
 259     return array_ops.concat((batch_shape, matrix_shape), 0)
 260 
 261   def _matmul(self, x, adjoint=False, adjoint_arg=False):
 262     split_dim = -1 if adjoint_arg else -2
 263     # Split input by rows normally, and otherwise columns.
 264     split_x = self._split_input_into_blocks(x, axis=split_dim)
 265 
 266     result_list = []
 267     for index, operator in enumerate(self.operators):
 268       result_list += [operator.matmul(
 269           split_x[index], adjoint=adjoint, adjoint_arg=adjoint_arg)]
 270     result_list = linear_operator_util.broadcast_matrix_batch_dims(
 271         result_list)
 272     return array_ops.concat(result_list, axis=-2)
 273 
 274   def _determinant(self):
 275     result = self.operators[0].determinant()
 276     for operator in self.operators[1:]:
 277       result *= operator.determinant()
 278     return result
 279 
 280   def _log_abs_determinant(self):
 281     result = self.operators[0].log_abs_determinant()
 282     for operator in self.operators[1:]:
 283       result += operator.log_abs_determinant()
 284     return result
 285 
 286   def _solve(self, rhs, adjoint=False, adjoint_arg=False):
 287     split_dim = -1 if adjoint_arg else -2
 288     # Split input by rows normally, and otherwise columns.
 289     split_rhs = self._split_input_into_blocks(rhs, axis=split_dim)
 290 
 291     solution_list = []
 292     for index, operator in enumerate(self.operators):
 293       solution_list += [operator.solve(
 294           split_rhs[index], adjoint=adjoint, adjoint_arg=adjoint_arg)]
 295 
 296     solution_list = linear_operator_util.broadcast_matrix_batch_dims(
 297         solution_list)
 298     return array_ops.concat(solution_list, axis=-2)
 299 
 300   def _diag_part(self):
 301     diag_list = []
 302     for operator in self.operators:
 303       # Extend the axis for broadcasting.
 304       diag_list += [operator.diag_part()[..., array_ops.newaxis]]
 305     diag_list = linear_operator_util.broadcast_matrix_batch_dims(diag_list)
 306     diagonal = array_ops.concat(diag_list, axis=-2)
 307     return array_ops.squeeze(diagonal, axis=-1)
 308 
 309   def _trace(self):
 310     result = self.operators[0].trace()
 311     for operator in self.operators[1:]:
 312       result += operator.trace()
 313     return result
 314 
 315   def _to_dense(self):
 316     num_cols = 0
 317     rows = []
 318     broadcasted_blocks = [operator.to_dense() for operator in self.operators]
 319     broadcasted_blocks = linear_operator_util.broadcast_matrix_batch_dims(
 320         broadcasted_blocks)
 321     for block in broadcasted_blocks:
 322       batch_row_shape = array_ops.shape(block)[:-1]
 323 
 324       zeros_to_pad_before_shape = array_ops.concat(
 325           [batch_row_shape, [num_cols]], axis=-1)
 326       zeros_to_pad_before = array_ops.zeros(
 327           shape=zeros_to_pad_before_shape, dtype=block.dtype)
 328       num_cols += array_ops.shape(block)[-1]
 329       zeros_to_pad_after_shape = array_ops.concat(
 330           [batch_row_shape,
 331            [self.domain_dimension_tensor() - num_cols]], axis=-1)
 332       zeros_to_pad_after = array_ops.zeros(
 333           shape=zeros_to_pad_after_shape, dtype=block.dtype)
 334 
 335       rows.append(array_ops.concat(
 336           [zeros_to_pad_before, block, zeros_to_pad_after], axis=-1))
 337 
 338     mat = array_ops.concat(rows, axis=-2)
 339     mat.set_shape(self.shape)
 340     return mat
 341 
 342   def _assert_non_singular(self):
 343     return control_flow_ops.group([
 344         operator.assert_non_singular() for operator in self.operators])
 345 
 346   def _assert_self_adjoint(self):
 347     return control_flow_ops.group([
 348         operator.assert_self_adjoint() for operator in self.operators])
 349 
 350   def _assert_positive_definite(self):
 351     return control_flow_ops.group([
 352         operator.assert_positive_definite() for operator in self.operators])
 353 
 354   def _split_input_into_blocks(self, x, axis=-1):
 355     """Split `x` into blocks matching `operators`'s `domain_dimension`.
 356 
 357     Specifically, if we have a block diagonal matrix, with block sizes
 358     `[M_j, M_j] j = 1..J`,  this method splits `x` on `axis` into `J`
 359     tensors, whose shape at `axis` is `M_j`.
 360 
 361     Args:
 362       x: `Tensor`. `x` is split into `J` tensors.
 363       axis: Python `Integer` representing the axis to split `x` on.
 364 
 365     Returns:
 366       A list of `Tensor`s.
 367     """
 368     block_sizes = []
 369     if self.shape.is_fully_defined():
 370       for operator in self.operators:
 371         block_sizes += [operator.domain_dimension.value]
 372     else:
 373       for operator in self.operators:
 374         block_sizes += [operator.domain_dimension_tensor()]
 375 
 376     return array_ops.split(x, block_sizes, axis=axis)
