  38 def _vec(x):
  39   """Stacks column of matrix to form a single column."""
  40   return array_ops.reshape(
  41       array_ops.matrix_transpose(x),
  42       array_ops.concat(
  43           [array_ops.shape(x)[:-2], [-1]], axis=0))
  44 
  45 
  46 def _unvec_by(y, num_col):
  47   """Unstack vector to form a matrix, with a specified amount of columns."""
  48   return array_ops.matrix_transpose(
  49       array_ops.reshape(
  50           y,
  51           array_ops.concat(
  52               [array_ops.shape(y)[:-1], [num_col, -1]], axis=0)))
  53 
  54 
  55 def _rotate_last_dim(x, rotate_right=False):
  56   """Rotate the last dimension either left or right."""
  57   ndims = array_ops.rank(x)
  58   if rotate_right:
  59     transpose_perm = array_ops.concat(
  60         [[ndims - 1], math_ops.range(0, ndims - 1)], axis=0)
  61   else:
  62     transpose_perm = array_ops.concat(
  63         [math_ops.range(1, ndims), [0]], axis=0)
  64   return array_ops.transpose(x, transpose_perm)

  67 @tf_export("linalg.LinearOperatorKronecker")
  68 class LinearOperatorKronecker(linear_operator.LinearOperator):
  69   """Kronecker product between two `LinearOperators`.
  70 
  71   This operator composes one or more linear operators `[op1,...,opJ]`,
  72   building a new `LinearOperator` representing the Kronecker product:
  73   `op1 x op2 x .. opJ` (we omit parentheses as the Kronecker product is
  74   associative).
  75 
  76   If `opj` has shape `batch_shape_j` + [M_j, N_j`, then the composed operator
  77   will have shape equal to `broadcast_batch_shape + [prod M_j, prod N_j]`,
  78   where the product is over all operators.
  79 
  80   ```python
  81   # Create a 4 x 4 linear operator composed of two 2 x 2 operators.
  82   operator_1 = LinearOperatorFullMatrix([[1., 2.], [3., 4.]])
  83   operator_2 = LinearOperatorFullMatrix([[1., 0.], [2., 1.]])
  84   operator = LinearOperatorKronecker([operator_1, operator_2])
  85 
  86   operator.to_dense()
  87   ==> [[1., 2., 0., 0.],
  88        [3., 4., 0., 0.],
  89        [2., 4., 1., 2.],
  90        [6., 8., 3., 4.]]
  91 
  92   operator.shape
  93   ==> [4, 4]
  94 
  95   operator.log_abs_determinant()
  96   ==> scalar Tensor
  97 
  98   x = ... Shape [4, 2] Tensor
  99   operator.matmul(x)
 100   ==> Shape [4, 2] Tensor
 101 
 102   # Create a [2, 3] batch of 4 x 5 linear operators.
 103   matrix_45 = tf.random_normal(shape=[2, 3, 4, 5])
 104   operator_45 = LinearOperatorFullMatrix(matrix)
 105 
 106   # Create a [2, 3] batch of 5 x 6 linear operators.
 107   matrix_56 = tf.random_normal(shape=[2, 3, 5, 6])
 108   operator_56 = LinearOperatorFullMatrix(matrix_56)
 109 
 110   # Compose to create a [2, 3] batch of 20 x 30 operators.
 111   operator_large = LinearOperatorKronecker([operator_45, operator_56])
 112 
 113   # Create a shape [2, 3, 20, 2] vector.
 114   x = tf.random_normal(shape=[2, 3, 6, 2])
 115   operator_large.matmul(x)
 116   ==> Shape [2, 3, 30, 2] Tensor
 117   ```
 118 
 119   #### Performance
 120 
 121   The performance of `LinearOperatorKronecker` on any operation is equal to
 122   the sum of the individual operators' operations.
 123 
 124   #### Matrix property hints
 125 
 126   This `LinearOperator` is initialized with boolean flags of the form `is_X`,
 127   for `X = non_singular, self_adjoint, positive_definite, square`.
 128   These have the following meaning:
 129 
 130   * If `is_X == True`, callers should expect the operator to have the
 131     property `X`.  This is a promise that should be fulfilled, but is *not* a
 132     runtime assert.  For example, finite floating point precision may result
 133     in these promises being violated.
 134   * If `is_X == False`, callers should expect the operator to not have `X`.
 135   * If `is_X == None` (the default), callers should have no expectation either
 136     way.
 137   """

 139   def __init__(self,
 140                operators,
 141                is_non_singular=None,
 142                is_self_adjoint=None,
 143                is_positive_definite=None,
 144                is_square=None,
 145                name=None):
 146     r"""Initialize a `LinearOperatorKronecker`.
 147 
 148     `LinearOperatorKronecker` is initialized with a list of operators
 149     `[op_1,...,op_J]`.
 150 
 151     Args:
 152       operators:  Iterable of `LinearOperator` objects, each with
 153         the same `dtype` and composable shape, representing the Kronecker
 154         factors.
 155       is_non_singular:  Expect that this operator is non-singular.
 156       is_self_adjoint:  Expect that this operator is equal to its hermitian
 157         transpose.
 158       is_positive_definite:  Expect that this operator is positive definite,
 159         meaning the quadratic form `x^H A x` has positive real part for all
 160         nonzero `x`.  Note that we do not require the operator to be
 161         self-adjoint to be positive-definite.  See:
 162         https://en.wikipedia.org/wiki/Positive-definite_matrix\
 163             #Extension_for_non_symmetric_matrices
 164       is_square:  Expect that this operator acts like square [batch] matrices.
 165       name: A name for this `LinearOperator`.  Default is the individual
 166         operators names joined with `_x_`.
 167 
 168     Raises:
 169       TypeError:  If all operators do not have the same `dtype`.
 170       ValueError:  If `operators` is empty.
 171     """
 172     # Validate operators.
 173     check_ops.assert_proper_iterable(operators)
 174     operators = list(operators)
 175     if not operators:
 176       raise ValueError(
 177           "Expected a list of >=1 operators. Found: %s" % operators)
 178     self._operators = operators
 179 
 180     # Validate dtype.
 181     dtype = operators[0].dtype
 182     for operator in operators:
 183       if operator.dtype != dtype:
 184         name_type = (str((o.name, o.dtype)) for o in operators)
 185         raise TypeError(
 186             "Expected all operators to have the same dtype.  Found %s"
 187             % "   ".join(name_type))
 188 
 189     # Auto-set and check hints.
 190     # A Kronecker product is invertible, if and only if all factors are
 191     # invertible.
 192     if all(operator.is_non_singular for operator in operators):
 193       if is_non_singular is False:
 194         raise ValueError(
 195             "The Kronecker product of non-singular operators is always "
 196             "non-singular.")
 197       is_non_singular = True
 198 
 199     if all(operator.is_self_adjoint for operator in operators):
 200       if is_self_adjoint is False:
 201         raise ValueError(
 202             "The Kronecker product of self-adjoint operators is always "
 203             "self-adjoint.")
 204       is_self_adjoint = True
 205 
 206     # The eigenvalues of a Kronecker product are equal to the products of eigen
 207     # values of the corresponding factors.
 208     if all(operator.is_positive_definite for operator in operators):
 209       if is_positive_definite is False:
 210         raise ValueError("The Kronecker product of positive-definite operators "
 211                          "is always positive-definite.")
 212       is_positive_definite = True
 213 
 214     # Initialization.
 215     graph_parents = []
 216     for operator in operators:
 217       graph_parents.extend(operator.graph_parents)
 218 
 219     if name is None:
 220       name = operators[0].name
 221       for operator in operators[1:]:
 222         name += "_x_" + operator.name
 223     with ops.name_scope(name, values=graph_parents):
 224       super(LinearOperatorKronecker, self).__init__(
 225           dtype=dtype,
 226           graph_parents=graph_parents,
 227           is_non_singular=is_non_singular,
 228           is_self_adjoint=is_self_adjoint,
 229           is_positive_definite=is_positive_definite,
 230           is_square=is_square,
 231           name=name)

 237   def _shape(self):
 238     # Get final matrix shape.
 239     domain_dimension = self.operators[0].domain_dimension
 240     for operator in self.operators[1:]:
 241       domain_dimension *= operator.domain_dimension
 242 
 243     range_dimension = self.operators[0].range_dimension
 244     for operator in self.operators[1:]:
 245       range_dimension *= operator.range_dimension
 246 
 247     matrix_shape = tensor_shape.TensorShape([
 248         range_dimension, domain_dimension])
 249 
 250     # Get broadcast batch shape.
 251     # broadcast_shape checks for compatibility.
 252     batch_shape = self.operators[0].batch_shape
 253     for operator in self.operators[1:]:
 254       batch_shape = common_shapes.broadcast_shape(
 255           batch_shape, operator.batch_shape)
 256 
 257     return batch_shape.concatenate(matrix_shape)

 259   def _shape_tensor(self):
 260     domain_dimension = self.operators[0].domain_dimension_tensor()
 261     for operator in self.operators[1:]:
 262       domain_dimension *= operator.domain_dimension_tensor()
 263 
 264     range_dimension = self.operators[0].range_dimension_tensor()
 265     for operator in self.operators[1:]:
 266       range_dimension *= operator.range_dimension_tensor()
 267 
 268     matrix_shape = [range_dimension, domain_dimension]
 269 
 270     # Get broadcast batch shape.
 271     # broadcast_shape checks for compatibility.
 272     batch_shape = self.operators[0].batch_shape_tensor()
 273     for operator in self.operators[1:]:
 274       batch_shape = array_ops.broadcast_dynamic_shape(
 275           batch_shape, operator.batch_shape_tensor())
 276 
 277     return array_ops.concat((batch_shape, matrix_shape), 0)

 279   def _matmul(self, x, adjoint=False, adjoint_arg=False):
 280     # Here we heavily rely on Roth's column Lemma [1]:
 281     # (A x B) * vec X = vec BXA^T,
 282     # where vec stacks all the columns of the matrix under each other. In our
 283     # case, x represents a batch of vec X (i.e. we think of x as a batch of
 284     # column vectors, rather than a matrix). Each member of the batch can be
 285     # reshaped to a matrix (hence we get a batch of matrices).
 286     # We can iteratively apply this lemma by noting that if B is a Kronecker
 287     # product, then we can apply the lemma again.
 288 
 289     # [1] W. E. Roth, "On direct product matrices,"
 290     # Bulletin of the American Mathematical Society, vol. 40, pp. 461-468,
 291     # 1934
 292 
 293     # Efficiency
 294 
 295     # Naively doing the Kronecker product, by calculating the dense matrix and
 296     # applying it will can take cubic time in  the size of domain_dimension
 297     # (assuming a square matrix). The other issue is that calculating the dense
 298     # matrix can be prohibitively expensive, in that it can take a large amount
 299     # of memory.
 300     #
 301     # This implementation avoids this memory blow up by only computing matmuls
 302     # with the factors. In this way, we don't have to realize the dense matrix.
 303     # In terms of complexity, if we have Kronecker Factors of size:
 304     # (n1, n1), (n2, n2), (n3, n3), ... (nJ, nJ), with N = \prod n_i, and we
 305     # have as input a [N, M] matrix, the naive approach would take O(N^2 M).
 306     # With this approach (ignoring reshaping of tensors and transposes for now),
 307     # the time complexity can be O(M * (\sum n_i) * N). There is also the
 308     # benefit of batched multiplication (In this example, the batch size is
 309     # roughly M * N) so this can be much faster. However, not factored in are
 310     # the costs of the several transposing of tensors, which can affect cache
 311     # behavior.
 312 
 313     # Below we document the shape manipulation for adjoint=False,
 314     # adjoint_arg=False, but the general case of different adjoints is still
 315     # handled.
 316 
 317     if adjoint_arg:
 318       x = linalg.adjoint(x)
 319 
 320     # Always add a batch dimension to enable broadcasting to work.
 321     batch_shape = array_ops.concat(
 322         [array_ops.ones_like(self.batch_shape_tensor()), [1, 1]], 0)
 323     x += array_ops.zeros(batch_shape, dtype=x.dtype.base_dtype)
 324 
 325     # x has shape [B, R, C], where B represent some number of batch dimensions,
 326     # R represents the number of rows, and C represents the number of columns.
 327     # In order to apply Roth's column lemma, we need to operate on a batch of
 328     # column vectors, so we reshape into a batch of column vectors. We put it
 329     # at the front to ensure that broadcasting between operators to the batch
 330     # dimensions B still works.
 331     output = _rotate_last_dim(x, rotate_right=True)
 332 
 333     # Also expand the shape to be [A, C, B, R]. The first dimension will be
 334     # used to accumulate dimensions from each operator matmul.
 335     output = output[array_ops.newaxis, ...]
 336 
 337     # In this loop, A is going to refer to the value of the accumulated
 338     # dimension. A = 1 at the start, and will end up being self.range_dimension.
 339     # V will refer to the last dimension. V = R at the start, and will end up
 340     # being 1 in the end.
 341     for operator in self.operators[:-1]:
 342       # Reshape output from [A, C, B, V] to be
 343       # [A, C, B, V / op.domain_dimension, op.domain_dimension]
 344       if adjoint:
 345         operator_dimension = operator.range_dimension_tensor()
 346       else:
 347         operator_dimension = operator.domain_dimension_tensor()
 348 
 349       output = _unvec_by(output, operator_dimension)
 350 
 351       # We are computing (XA^T) = (AX^T)^T.
 352       # output has [A, C, B, V / op.domain_dimension, op.domain_dimension],
 353       # which is being converted to:
 354       # [A, C, B, V / op.domain_dimension, op.range_dimension]
 355       output = array_ops.matrix_transpose(output)
 356       output = operator.matmul(output, adjoint=adjoint, adjoint_arg=False)
 357       output = array_ops.matrix_transpose(output)
 358       # Rearrange it to [A * op.range_dimension, C, B, V / op.domain_dimension]
 359       output = _rotate_last_dim(output, rotate_right=False)
 360       output = _vec(output)
 361       output = _rotate_last_dim(output, rotate_right=True)
 362 
 363     # After the loop, we will have
 364     # A = self.range_dimension / op[-1].range_dimension
 365     # V = op[-1].domain_dimension
 366 
 367     # We convert that using matvec to get:
 368     # [A, C, B, op[-1].range_dimension]
 369     output = self.operators[-1].matvec(output, adjoint=adjoint)
 370     # Rearrange shape to be [B1, ... Bn, self.range_dimension, C]
 371     output = _rotate_last_dim(output, rotate_right=False)
 372     output = _vec(output)
 373     output = _rotate_last_dim(output, rotate_right=False)
 374 
 375     if x.shape.is_fully_defined():
 376       column_dim = x.shape[-1]
 377       broadcast_batch_shape = common_shapes.broadcast_shape(
 378           x.shape[:-2], self.batch_shape)
 379       if adjoint:
 380         matrix_dimensions = [self.domain_dimension, column_dim]
 381       else:
 382         matrix_dimensions = [self.range_dimension, column_dim]
 383 
 384       output.set_shape(broadcast_batch_shape.concatenate(
 385           matrix_dimensions))
 386 
 387     return output

 389   def _determinant(self):
 390     # Note that we have |X1 x X2| = |X1| ** n * |X2| ** m, where X1 is an m x m
 391     # matrix, and X2 is an n x n matrix. We can iteratively apply this property
 392     # to get the determinant of |X1 x X2 x X3 ...|. If T is the product of the
 393     # domain dimension of all operators, then we have:
 394     # |X1 x X2 x X3 ...| =
 395     #    |X1| ** (T / m) * |X2 x X3 ... | ** m =
 396     #    |X1| ** (T / m) * |X2| ** (m * (T / m) / n) *  ... =
 397     #    |X1| ** (T / m) * |X2| ** (T / n) * | X3 x X4... | ** (m * n)
 398     #    And by doing induction we have product(|X_i| ** (T / dim(X_i))).
 399     total = self.domain_dimension_tensor()
 400     determinant = 1.
 401     for operator in self.operators:
 402       determinant *= operator.determinant() ** math_ops.cast(
 403           total / operator.domain_dimension_tensor(),
 404           dtype=operator.dtype)
 405     return determinant
 406 
 407   def _log_abs_determinant(self):
 408     # This will be sum((total / dim(x_i)) * log |X_i|)
 409     total = self.domain_dimension_tensor()
 410     log_abs_det = 0.
 411     for operator in self.operators:
 412       log_abs_det += operator.log_abs_determinant() * math_ops.cast(
 413           total / operator.domain_dimension_tensor(),
 414           dtype=operator.dtype)
 415     return log_abs_det

 417   def _trace(self):
 418     # tr(A x B) = tr(A) * tr(B)
 419     trace = 1.
 420     for operator in self.operators:
 421       trace *= operator.trace()
 422     return trace

 424   def _solve(self, rhs, adjoint=False, adjoint_arg=False):
 425     # Here we follow the same use of Roth's column lemma as in `matmul`, with
 426     # the key difference that we replace all `matmul` instances with `solve`.
 427     # This follows from the property that inv(A x B) = inv(A) x inv(B).
 428 
 429     # Below we document the shape manipulation for adjoint=False,
 430     # adjoint_arg=False, but the general case of different adjoints is still
 431     # handled.
 432 
 433     if adjoint_arg:
 434       rhs = linalg.adjoint(rhs)
 435 
 436     # Always add a batch dimension to enable broadcasting to work.
 437     batch_shape = array_ops.concat(
 438         [array_ops.ones_like(self.batch_shape_tensor()), [1, 1]], 0)
 439     rhs += array_ops.zeros(batch_shape, dtype=rhs.dtype.base_dtype)
 440 
 441     # rhs has shape [B, R, C], where B represent some number of batch
 442     # dimensions,
 443     # R represents the number of rows, and C represents the number of columns.
 444     # In order to apply Roth's column lemma, we need to operate on a batch of
 445     # column vectors, so we reshape into a batch of column vectors. We put it
 446     # at the front to ensure that broadcasting between operators to the batch
 447     # dimensions B still works.
 448     output = _rotate_last_dim(rhs, rotate_right=True)
 449 
 450     # Also expand the shape to be [A, C, B, R]. The first dimension will be
 451     # used to accumulate dimensions from each operator matmul.
 452     output = output[array_ops.newaxis, ...]
 453 
 454     # In this loop, A is going to refer to the value of the accumulated
 455     # dimension. A = 1 at the start, and will end up being self.range_dimension.
 456     # V will refer to the last dimension. V = R at the start, and will end up
 457     # being 1 in the end.
 458     for operator in self.operators[:-1]:
 459       # Reshape output from [A, C, B, V] to be
 460       # [A, C, B, V / op.domain_dimension, op.domain_dimension]
 461       if adjoint:
 462         operator_dimension = operator.range_dimension_tensor()
 463       else:
 464         operator_dimension = operator.domain_dimension_tensor()
 465 
 466       output = _unvec_by(output, operator_dimension)
 467 
 468       # We are computing (XA^-1^T) = (A^-1 X^T)^T.
 469       # output has [A, C, B, V / op.domain_dimension, op.domain_dimension],
 470       # which is being converted to:
 471       # [A, C, B, V / op.domain_dimension, op.range_dimension]
 472       output = array_ops.matrix_transpose(output)
 473       output = operator.solve(output, adjoint=adjoint, adjoint_arg=False)
 474       output = array_ops.matrix_transpose(output)
 475       # Rearrange it to [A * op.range_dimension, C, B, V / op.domain_dimension]
 476       output = _rotate_last_dim(output, rotate_right=False)
 477       output = _vec(output)
 478       output = _rotate_last_dim(output, rotate_right=True)
 479 
 480     # After the loop, we will have
 481     # A = self.range_dimension / op[-1].range_dimension
 482     # V = op[-1].domain_dimension
 483 
 484     # We convert that using matvec to get:
 485     # [A, C, B, op[-1].range_dimension]
 486     output = self.operators[-1].solvevec(output, adjoint=adjoint)
 487     # Rearrange shape to be [B1, ... Bn, self.range_dimension, C]
 488     output = _rotate_last_dim(output, rotate_right=False)
 489     output = _vec(output)
 490     output = _rotate_last_dim(output, rotate_right=False)
 491 
 492     if rhs.shape.is_fully_defined():
 493       column_dim = rhs.shape[-1]
 494       broadcast_batch_shape = common_shapes.broadcast_shape(
 495           rhs.shape[:-2], self.batch_shape)
 496       if adjoint:
 497         matrix_dimensions = [self.domain_dimension, column_dim]
 498       else:
 499         matrix_dimensions = [self.range_dimension, column_dim]
 500 
 501       output.set_shape(broadcast_batch_shape.concatenate(
 502           matrix_dimensions))
 503 
 504     return output
 505 
 506   def _diag_part(self):
 507     diag_part = self.operators[0].diag_part()
 508     for operator in self.operators[1:]:
 509       diag_part = diag_part[..., :, array_ops.newaxis]
 510       op_diag_part = operator.diag_part()[..., array_ops.newaxis, :]
 511       diag_part *= op_diag_part
 512       diag_part = array_ops.reshape(
 513           diag_part,
 514           shape=array_ops.concat(
 515               [array_ops.shape(diag_part)[:-2], [-1]], axis=0))
 516     if self.range_dimension > self.domain_dimension:
 517       diag_dimension = self.domain_dimension
 518     else:
 519       diag_dimension = self.range_dimension
 520     diag_part.set_shape(
 521         self.batch_shape.concatenate(diag_dimension))
 522     return diag_part
 523 
 524   def _to_dense(self):
 525     product = self.operators[0].to_dense()
 526     for operator in self.operators[1:]:
 527       # Product has shape [B, R1, 1, C1].
 528       product = product[
 529           ..., :, array_ops.newaxis, :, array_ops.newaxis]
 530       # Operator has shape [B, 1, R2, 1, C2].
 531       op_to_mul = operator.to_dense()[
 532           ..., array_ops.newaxis, :, array_ops.newaxis, :]
 533       # This is now [B, R1, R2, C1, C2].
 534       product *= op_to_mul
 535       # Now merge together dimensions to get [B, R1 * R2, C1 * C2].
 536       product = array_ops.reshape(
 537           product,
 538           shape=array_ops.concat(
 539               [array_ops.shape(product)[:-4],
 540                [array_ops.shape(product)[-4] * array_ops.shape(product)[-3],
 541                 array_ops.shape(product)[-2] * array_ops.shape(product)[-1]]
 542               ], axis=0))
 543     product.set_shape(self.shape)
 544     return product

 546   def _assert_non_singular(self):
 547     if all(operator.is_square for operator in self.operators):
 548       asserts = [operator.assert_non_singular() for operator in self.operators]
 549       return control_flow_ops.group(asserts)
 550     else:
 551       raise errors.InvalidArgumentError(
 552           node_def=None, op=None, message="All Kronecker factors must be "
 553           "square for the product to be invertible.")
 554 
 555   def _assert_self_adjoint(self):
 556     if all(operator.is_square for operator in self.operators):
 557       asserts = [operator.assert_self_adjoint() for operator in self.operators]
 558       return control_flow_ops.group(asserts)
 559     else:
 560       raise errors.InvalidArgumentError(
