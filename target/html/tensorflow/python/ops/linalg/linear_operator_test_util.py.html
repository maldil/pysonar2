  75   def assertAC(self, x, y):
  76     """Derived classes can set _atol, _rtol to get different tolerance."""
  77     dtype = dtypes.as_dtype(x.dtype)
  78     atol = self._atol[dtype]
  79     rtol = self._rtol[dtype]
  80     self.assertAllClose(x, y, atol=atol, rtol=rtol)

 231   def _test_matmul(self, with_batch):
 232     for use_placeholder in self._use_placeholder_options:
 233       for build_info in self._operator_build_infos:
 234         # If batch dimensions are omitted, but there are
 235         # no batch dimensions for the linear operator, then
 236         # skip the test case. This is already checked with
 237         # with_batch=True.
 238         if not with_batch and len(build_info.shape) <= 2:
 239           continue
 240         for dtype in self._dtypes_to_test:
 241           for adjoint in self._adjoint_options:
 242             for adjoint_arg in self._adjoint_arg_options:
 243               with self.session(graph=ops.Graph()) as sess:
 244                 sess.graph.seed = random_seed.DEFAULT_GRAPH_SEED
 245                 operator, mat = self._operator_and_matrix(
 246                     build_info, dtype, use_placeholder=use_placeholder)
 247                 x = self._make_x(
 248                     operator, adjoint=adjoint, with_batch=with_batch)
 249                 # If adjoint_arg, compute A X^H^H = A X.
 250                 if adjoint_arg:
 251                   op_matmul = operator.matmul(
 252                       linalg.adjoint(x),
 253                       adjoint=adjoint,
 254                       adjoint_arg=adjoint_arg)
 255                 else:
 256                   op_matmul = operator.matmul(x, adjoint=adjoint)
 257                 mat_matmul = linear_operator_util.matmul_with_broadcast(
 258                     mat, x, adjoint_a=adjoint)
 259                 if not use_placeholder:
 260                   self.assertAllEqual(op_matmul.get_shape(),
 261                                       mat_matmul.get_shape())
 262                 op_matmul_v, mat_matmul_v = sess.run(
 263                     [op_matmul, mat_matmul])

 381   @property
 382   def _operator_build_infos(self):
 383     build_info = OperatorBuildInfo
 384     # non-batch operators (n, n) and batch operators.
 385     return [
 386         build_info((0, 0)),
 387         build_info((1, 1)),
 388         build_info((1, 3, 3)),
 389         build_info((3, 4, 4)),
 390         build_info((2, 1, 4, 4))]

 392   def _make_rhs(self, operator, adjoint, with_batch=True):
 393     # This operator is square, so rhs and x will have same shape.
 394     # adjoint value makes no difference because the operator shape doesn't
 395     # change since it is square, but be pedantic.
 396     return self._make_x(operator, adjoint=not adjoint, with_batch=with_batch)
 397 
 398   def _make_x(self, operator, adjoint, with_batch=True):
 399     # Value of adjoint makes no difference because the operator is square.
 400     # Return the number of systems to solve, R, equal to 1 or 2.
 401     r = self._get_num_systems(operator)
 402     # If operator.shape = [B1,...,Bb, N, N] this returns a random matrix of
 403     # shape [B1,...,Bb, N, R], R = 1 or 2.
 404     if operator.shape.is_fully_defined():
 405       batch_shape = operator.batch_shape.as_list()
 406       n = operator.domain_dimension.value
 407       if with_batch:
 408         x_shape = batch_shape + [n, r]
 409       else:
 410         x_shape = [n, r]
 411     else:
 412       batch_shape = operator.batch_shape_tensor()
 413       n = operator.domain_dimension_tensor()
 414       if with_batch:
 415         x_shape = array_ops.concat((batch_shape, [n, r]), 0)
 416       else:
 417         x_shape = [n, r]
 418 
 419     return random_normal(x_shape, dtype=operator.dtype)

 421   def _get_num_systems(self, operator):
 422     """Get some number, either 1 or 2, depending on operator."""
 423     if operator.tensor_rank is None or operator.tensor_rank % 2:
 424       return 1
 425     else:
 426       return 2

 498 def random_positive_definite_matrix(shape, dtype, force_well_conditioned=False):
 499   """[batch] positive definite matrix.
 500 
 501   Args:
 502     shape:  `TensorShape` or Python list.  Shape of the returned matrix.
 503     dtype:  `TensorFlow` `dtype` or Python dtype.
 504     force_well_conditioned:  Python bool.  If `True`, returned matrix has
 505       eigenvalues with modulus in `(1, 4)`.  Otherwise, eigenvalues are
 506       chi-squared random variables.
 507 
 508   Returns:
 509     `Tensor` with desired shape and dtype.
 510   """
 511   dtype = dtypes.as_dtype(dtype)
 512   if not tensor_util.is_tensor(shape):
 513     shape = tensor_shape.TensorShape(shape)
 514     # Matrix must be square.
 515     shape[-1].assert_is_compatible_with(shape[-2])
 516 
 517   with ops.name_scope("random_positive_definite_matrix"):
 518     tril = random_tril_matrix(
 519         shape, dtype, force_well_conditioned=force_well_conditioned)
 520     return math_ops.matmul(tril, tril, adjoint_b=True)

 523 def random_tril_matrix(shape,
 524                        dtype,
 525                        force_well_conditioned=False,
 526                        remove_upper=True):
 527   """[batch] lower triangular matrix.
 528 
 529   Args:
 530     shape:  `TensorShape` or Python `list`.  Shape of the returned matrix.
 531     dtype:  `TensorFlow` `dtype` or Python dtype
 532     force_well_conditioned:  Python `bool`. If `True`, returned matrix will have
 533       eigenvalues with modulus in `(1, 2)`.  Otherwise, eigenvalues are unit
 534       normal random variables.
 535     remove_upper:  Python `bool`.
 536       If `True`, zero out the strictly upper triangle.
 537       If `False`, the lower triangle of returned matrix will have desired
 538       properties, but will not have the strictly upper triangle zero'd out.
 539 
 540   Returns:
 541     `Tensor` with desired shape and dtype.
 542   """
 543   with ops.name_scope("random_tril_matrix"):
 544     # Totally random matrix.  Has no nice properties.
 545     tril = random_normal(shape, dtype=dtype)
 546     if remove_upper:
 547       tril = array_ops.matrix_band_part(tril, -1, 0)
 548 
 549     # Create a diagonal with entries having modulus in [1, 2].
 550     if force_well_conditioned:
 551       maxval = ops.convert_to_tensor(np.sqrt(2.), dtype=dtype.real_dtype)
 552       diag = random_sign_uniform(
 553           shape[:-1], dtype=dtype, minval=1., maxval=maxval)
 554       tril = array_ops.matrix_set_diag(tril, diag)
 555 
 556     return tril

 559 def random_normal(shape, mean=0.0, stddev=1.0, dtype=dtypes.float32, seed=None):
 560   """Tensor with (possibly complex) Gaussian entries.
 561 
 562   Samples are distributed like
 563 
 564   ```
 565   N(mean, stddev^2), if dtype is real,
 566   X + iY,  where X, Y ~ N(mean, stddev^2) if dtype is complex.
 567   ```
 568 
 569   Args:
 570     shape:  `TensorShape` or Python list.  Shape of the returned tensor.
 571     mean:  `Tensor` giving mean of normal to sample from.
 572     stddev:  `Tensor` giving stdev of normal to sample from.
 573     dtype:  `TensorFlow` `dtype` or numpy dtype
 574     seed:  Python integer seed for the RNG.
 575 
 576   Returns:
 577     `Tensor` with desired shape and dtype.
 578   """
 579   dtype = dtypes.as_dtype(dtype)
 580 
 581   with ops.name_scope("random_normal"):
 582     samples = random_ops.random_normal(
 583         shape, mean=mean, stddev=stddev, dtype=dtype.real_dtype, seed=seed)
 584     if dtype.is_complex:
 585       if seed is not None:
 586         seed += 1234
 587       more_samples = random_ops.random_normal(
 588           shape, mean=mean, stddev=stddev, dtype=dtype.real_dtype, seed=seed)
 589       samples = math_ops.complex(samples, more_samples)
 590     return samples

 593 def random_uniform(shape,
 594                    minval=None,
 595                    maxval=None,
 596                    dtype=dtypes.float32,
 597                    seed=None):
 598   """Tensor with (possibly complex) Uniform entries.
 599 
 600   Samples are distributed like
 601 
 602   ```
 603   Uniform[minval, maxval], if dtype is real,
 604   X + iY,  where X, Y ~ Uniform[minval, maxval], if dtype is complex.
 605   ```
 606 
 607   Args:
 608     shape:  `TensorShape` or Python list.  Shape of the returned tensor.
 609     minval:  `0-D` `Tensor` giving the minimum values.
 610     maxval:  `0-D` `Tensor` giving the maximum values.
 611     dtype:  `TensorFlow` `dtype` or Python dtype
 612     seed:  Python integer seed for the RNG.
 613 
 614   Returns:
 615     `Tensor` with desired shape and dtype.
 616   """
 617   dtype = dtypes.as_dtype(dtype)
 618 
 619   with ops.name_scope("random_uniform"):
 620     samples = random_ops.random_uniform(
 621         shape, dtype=dtype.real_dtype, minval=minval, maxval=maxval, seed=seed)
 622     if dtype.is_complex:
 623       if seed is not None:
 624         seed += 12345
 625       more_samples = random_ops.random_uniform(
 626           shape,
 627           dtype=dtype.real_dtype,
 628           minval=minval,
 629           maxval=maxval,
 630           seed=seed)
 631       samples = math_ops.complex(samples, more_samples)
 632     return samples

 635 def random_sign_uniform(shape,
 636                         minval=None,
 637                         maxval=None,
 638                         dtype=dtypes.float32,
 639                         seed=None):
 640   """Tensor with (possibly complex) random entries from a "sign Uniform".
 641 
 642   Letting `Z` be a random variable equal to `-1` and `1` with equal probability,
 643   Samples from this `Op` are distributed like
 644 
 645   ```
 646   Z * X, where X ~ Uniform[minval, maxval], if dtype is real,
 647   Z * (X + iY),  where X, Y ~ Uniform[minval, maxval], if dtype is complex.
 648   ```
 649 
 650   Args:
 651     shape:  `TensorShape` or Python list.  Shape of the returned tensor.
 652     minval:  `0-D` `Tensor` giving the minimum values.
 653     maxval:  `0-D` `Tensor` giving the maximum values.
 654     dtype:  `TensorFlow` `dtype` or Python dtype
 655     seed:  Python integer seed for the RNG.
 656 
 657   Returns:
 658     `Tensor` with desired shape and dtype.
 659   """
 660   dtype = dtypes.as_dtype(dtype)
 661 
 662   with ops.name_scope("random_sign_uniform"):
 663     unsigned_samples = random_uniform(
 664         shape, minval=minval, maxval=maxval, dtype=dtype, seed=seed)
 665     if seed is not None:
 666       seed += 12
 667     signs = math_ops.sign(
 668         random_ops.random_uniform(shape, minval=-1., maxval=1., seed=seed))
 669     return unsigned_samples * math_ops.cast(signs, unsigned_samples.dtype)

 672 def random_normal_correlated_columns(shape,
 673                                      mean=0.0,
 674                                      stddev=1.0,
 675                                      dtype=dtypes.float32,
 676                                      eps=1e-4,
 677                                      seed=None):
 678   """Batch matrix with (possibly complex) Gaussian entries and correlated cols.
 679 
 680   Returns random batch matrix `A` with specified element-wise `mean`, `stddev`,
 681   living close to an embedded hyperplane.
 682 
 683   Suppose `shape[-2:] = (M, N)`.
 684 
 685   If `M < N`, `A` is a random `M x N` [batch] matrix with iid Gaussian entries.
 686 
 687   If `M >= N`, then the colums of `A` will be made almost dependent as follows:
 688 
 689   ```
 690   L = random normal N x N-1 matrix, mean = 0, stddev = 1 / sqrt(N - 1)
 691   B = random normal M x N-1 matrix, mean = 0, stddev = stddev.
 692 
 693   G = (L B^H)^H, a random normal M x N matrix, living on N-1 dim hyperplane
 694   E = a random normal M x N matrix, mean = 0, stddev = eps
 695   mu = a constant M x N matrix, equal to the argument "mean"
 696 
 697   A = G + E + mu
 698   ```
 699 
 700   Args:
 701     shape:  Python list of integers.
 702       Shape of the returned tensor.  Must be at least length two.
 703     mean:  `Tensor` giving mean of normal to sample from.
 704     stddev:  `Tensor` giving stdev of normal to sample from.
 705     dtype:  `TensorFlow` `dtype` or numpy dtype
 706     eps:  Distance each column is perturbed from the low-dimensional subspace.
 707     seed:  Python integer seed for the RNG.
 708 
 709   Returns:
 710     `Tensor` with desired shape and dtype.
 711 
 712   Raises:
 713     ValueError:  If `shape` is not at least length 2.
 714   """
 715   dtype = dtypes.as_dtype(dtype)
 716 
 717   if len(shape) < 2:
 718     raise ValueError(
 719         "Argument shape must be at least length 2.  Found: %s" % shape)
 720 
 721   # Shape is the final shape, e.g. [..., M, N]
 722   shape = list(shape)
 723   batch_shape = shape[:-2]
 724   m, n = shape[-2:]
 725 
 726   # If there is only one column, "they" are by definition correlated.
 727   if n < 2 or n < m:
 728     return random_normal(
 729         shape, mean=mean, stddev=stddev, dtype=dtype, seed=seed)
 730 
 731   # Shape of the matrix with only n - 1 columns that we will embed in higher
 732   # dimensional space.
 733   smaller_shape = batch_shape + [m, n - 1]
 734 
 735   # Shape of the embedding matrix, mapping batch matrices
 736   # from [..., N-1, M] to [..., N, M]
 737   embedding_mat_shape = batch_shape + [n, n - 1]
 738 
 739   # This stddev for the embedding_mat ensures final result has correct stddev.
 740   stddev_mat = 1 / np.sqrt(n - 1)
 741 
 742   with ops.name_scope("random_normal_correlated_columns"):
 743     smaller_mat = random_normal(
 744         smaller_shape, mean=0.0, stddev=stddev_mat, dtype=dtype, seed=seed)
 745 
 746     if seed is not None:
 747       seed += 1287
 748 
 749     embedding_mat = random_normal(embedding_mat_shape, dtype=dtype, seed=seed)
 750 
 751     embedded_t = math_ops.matmul(embedding_mat, smaller_mat, transpose_b=True)
 752     embedded = array_ops.matrix_transpose(embedded_t)
 753 
 754     mean_mat = array_ops.ones_like(embedded) * mean
 755 
 756     return embedded + random_normal(shape, stddev=eps, dtype=dtype) + mean_mat
