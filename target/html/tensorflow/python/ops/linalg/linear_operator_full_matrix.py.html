  31 @tf_export("linalg.LinearOperatorFullMatrix")
  32 class LinearOperatorFullMatrix(linear_operator.LinearOperator):
  33   """`LinearOperator` that wraps a [batch] matrix.
  34 
  35   This operator wraps a [batch] matrix `A` (which is a `Tensor`) with shape
  36   `[B1,...,Bb, M, N]` for some `b >= 0`.  The first `b` indices index a
  37   batch member.  For every batch index `(i1,...,ib)`, `A[i1,...,ib, : :]` is
  38   an `M x N` matrix.
  39 
  40   ```python
  41   # Create a 2 x 2 linear operator.
  42   matrix = [[1., 2.], [3., 4.]]
  43   operator = LinearOperatorFullMatrix(matrix)
  44 
  45   operator.to_dense()
  46   ==> [[1., 2.]
  47        [3., 4.]]
  48 
  49   operator.shape
  50   ==> [2, 2]
  51 
  52   operator.log_abs_determinant()
  53   ==> scalar Tensor
  54 
  55   x = ... Shape [2, 4] Tensor
  56   operator.matmul(x)
  57   ==> Shape [2, 4] Tensor
  58 
  59   # Create a [2, 3] batch of 4 x 4 linear operators.
  60   matrix = tf.random_normal(shape=[2, 3, 4, 4])
  61   operator = LinearOperatorFullMatrix(matrix)
  62   ```
  63 
  64   #### Shape compatibility
  65 
  66   This operator acts on [batch] matrix with compatible shape.
  67   `x` is a batch matrix with compatible shape for `matmul` and `solve` if
  68 
  69   ```
  70   operator.shape = [B1,...,Bb] + [M, N],  with b >= 0
  71   x.shape =        [B1,...,Bb] + [N, R],  with R >= 0.
  72   ```
  73 
  74   #### Performance
  75 
  76   `LinearOperatorFullMatrix` has exactly the same performance as would be
  77   achieved by using standard `TensorFlow` matrix ops.  Intelligent choices are
  78   made based on the following initialization hints.
  79 
  80   * If `dtype` is real, and `is_self_adjoint` and `is_positive_definite`, a
  81     Cholesky factorization is used for the determinant and solve.
  82 
  83   In all cases, suppose `operator` is a `LinearOperatorFullMatrix` of shape
  84   `[M, N]`, and `x.shape = [N, R]`.  Then
  85 
  86   * `operator.matmul(x)` is `O(M * N * R)`.
  87   * If `M=N`, `operator.solve(x)` is `O(N^3 * R)`.
  88   * If `M=N`, `operator.determinant()` is `O(N^3)`.
  89 
  90   If instead `operator` and `x` have shape `[B1,...,Bb, M, N]` and
  91   `[B1,...,Bb, N, R]`, every operation increases in complexity by `B1*...*Bb`.
  92 
  93   #### Matrix property hints
  94 
  95   This `LinearOperator` is initialized with boolean flags of the form `is_X`,
  96   for `X = non_singular, self_adjoint, positive_definite, square`.
  97   These have the following meaning:
  98 
  99   * If `is_X == True`, callers should expect the operator to have the
 100     property `X`.  This is a promise that should be fulfilled, but is *not* a
 101     runtime assert.  For example, finite floating point precision may result
 102     in these promises being violated.
 103   * If `is_X == False`, callers should expect the operator to not have `X`.
 104   * If `is_X == None` (the default), callers should have no expectation either
 105     way.
 106   """
 107 
 108   def __init__(self,
 109                matrix,
 110                is_non_singular=None,
 111                is_self_adjoint=None,
 112                is_positive_definite=None,
 113                is_square=None,
 114                name="LinearOperatorFullMatrix"):
 115     r"""Initialize a `LinearOperatorFullMatrix`.
 116 
 117     Args:
 118       matrix:  Shape `[B1,...,Bb, M, N]` with `b >= 0`, `M, N >= 0`.
 119         Allowed dtypes: `float16`, `float32`, `float64`, `complex64`,
 120         `complex128`.
 121       is_non_singular:  Expect that this operator is non-singular.
 122       is_self_adjoint:  Expect that this operator is equal to its hermitian
 123         transpose.
 124       is_positive_definite:  Expect that this operator is positive definite,
 125         meaning the quadratic form `x^H A x` has positive real part for all
 126         nonzero `x`.  Note that we do not require the operator to be
 127         self-adjoint to be positive-definite.  See:
 128         https://en.wikipedia.org/wiki/Positive-definite_matrix#Extension_for_non-symmetric_matrices
 129       is_square:  Expect that this operator acts like square [batch] matrices.
 130       name: A name for this `LinearOperator`.
 131 
 132     Raises:
 133       TypeError:  If `diag.dtype` is not an allowed type.
 134     """
 135 
 136     with ops.name_scope(name, values=[matrix]):
 137       self._matrix = ops.convert_to_tensor(matrix, name="matrix")
 138       self._check_matrix(self._matrix)
 139 
 140       super(LinearOperatorFullMatrix, self).__init__(
 141           dtype=self._matrix.dtype,
 142           graph_parents=[self._matrix],
 143           is_non_singular=is_non_singular,
 144           is_self_adjoint=is_self_adjoint,
 145           is_positive_definite=is_positive_definite,
 146           is_square=is_square,
 147           name=name)
 148 
 149   def _check_matrix(self, matrix):
 150     """Static check of the `matrix` argument."""
 151     allowed_dtypes = [
 152         dtypes.float16,
 153         dtypes.float32,
 154         dtypes.float64,
 155         dtypes.complex64,
 156         dtypes.complex128,
 157     ]
 158 
 159     matrix = ops.convert_to_tensor(matrix, name="matrix")
 160 
 161     dtype = matrix.dtype
 162     if dtype not in allowed_dtypes:
 163       raise TypeError(
 164           "Argument matrix must have dtype in %s.  Found: %s"
 165           % (allowed_dtypes, dtype))
 166 
 167     if matrix.get_shape().ndims is not None and matrix.get_shape().ndims < 2:
 168       raise ValueError(
 169           "Argument matrix must have at least 2 dimensions.  Found: %s"
 170           % matrix)
 171 
 172   def _shape(self):
 173     return self._matrix.get_shape()
 174 
 175   def _shape_tensor(self):
 176     return array_ops.shape(self._matrix)
 177 
 178   def _matmul(self, x, adjoint=False, adjoint_arg=False):
 179     return linear_operator_util.matmul_with_broadcast(
 180         self._matrix, x, adjoint_a=adjoint, adjoint_b=adjoint_arg)
 181 
 182   def _to_dense(self):
 183     return self._matrix
