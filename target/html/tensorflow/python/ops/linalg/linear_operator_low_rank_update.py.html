  37 @tf_export("linalg.LinearOperatorLowRankUpdate")
  38 class LinearOperatorLowRankUpdate(linear_operator.LinearOperator):
  39   """Perturb a `LinearOperator` with a rank `K` update.
  40 
  41   This operator acts like a [batch] matrix `A` with shape
  42   `[B1,...,Bb, M, N]` for some `b >= 0`.  The first `b` indices index a
  43   batch member.  For every batch index `(i1,...,ib)`, `A[i1,...,ib, : :]` is
  44   an `M x N` matrix.
  45 
  46   `LinearOperatorLowRankUpdate` represents `A = L + U D V^H`, where
  47 
  48   ```
  49   L, is a LinearOperator representing [batch] M x N matrices
  50   U, is a [batch] M x K matrix.  Typically K << M.
  51   D, is a [batch] K x K matrix.
  52   V, is a [batch] N x K matrix.  Typically K << N.
  53   V^H is the Hermitian transpose (adjoint) of V.
  54   ```
  55 
  56   If `M = N`, determinants and solves are done using the matrix determinant
  57   lemma and Woodbury identities, and thus require L and D to be non-singular.
  58 
  59   Solves and determinants will be attempted unless the "is_non_singular"
  60   property of L and D is False.
  61 
  62   In the event that L and D are positive-definite, and U = V, solves and
  63   determinants can be done using a Cholesky factorization.
  64 
  65   ```python
  66   # Create a 3 x 3 diagonal linear operator.
  67   diag_operator = LinearOperatorDiag(
  68       diag_update=[1., 2., 3.], is_non_singular=True, is_self_adjoint=True,
  69       is_positive_definite=True)
  70 
  71   # Perturb with a rank 2 perturbation
  72   operator = LinearOperatorLowRankUpdate(
  73       operator=diag_operator,
  74       u=[[1., 2.], [-1., 3.], [0., 0.]],
  75       diag_update=[11., 12.],
  76       v=[[1., 2.], [-1., 3.], [10., 10.]])
  77 
  78   operator.shape
  79   ==> [3, 3]
  80 
  81   operator.log_abs_determinant()
  82   ==> scalar Tensor
  83 
  84   x = ... Shape [3, 4] Tensor
  85   operator.matmul(x)
  86   ==> Shape [3, 4] Tensor
  87   ```
  88 
  89   ### Shape compatibility
  90 
  91   This operator acts on [batch] matrix with compatible shape.
  92   `x` is a batch matrix with compatible shape for `matmul` and `solve` if
  93 
  94   ```
  95   operator.shape = [B1,...,Bb] + [M, N],  with b >= 0
  96   x.shape =        [B1,...,Bb] + [N, R],  with R >= 0.
  97   ```
  98 
  99   ### Performance
 100 
 101   Suppose `operator` is a `LinearOperatorLowRankUpdate` of shape `[M, N]`,
 102   made from a rank `K` update of `base_operator` which performs `.matmul(x)` on
 103   `x` having `x.shape = [N, R]` with `O(L_matmul*N*R)` complexity (and similarly
 104   for `solve`, `determinant`.  Then, if `x.shape = [N, R]`,
 105 
 106   * `operator.matmul(x)` is `O(L_matmul*N*R + K*N*R)`
 107 
 108   and if `M = N`,
 109 
 110   * `operator.solve(x)` is `O(L_matmul*N*R + N*K*R + K^2*R + K^3)`
 111   * `operator.determinant()` is `O(L_determinant + L_solve*N*K + K^2*N + K^3)`
 112 
 113   If instead `operator` and `x` have shape `[B1,...,Bb, M, N]` and
 114   `[B1,...,Bb, N, R]`, every operation increases in complexity by `B1*...*Bb`.
 115 
 116   #### Matrix property hints
 117 
 118   This `LinearOperator` is initialized with boolean flags of the form `is_X`,
 119   for `X = non_singular`, `self_adjoint`, `positive_definite`,
 120   `diag_update_positive` and `square`. These have the following meaning:
 121 
 122   * If `is_X == True`, callers should expect the operator to have the
 123     property `X`.  This is a promise that should be fulfilled, but is *not* a
 124     runtime assert.  For example, finite floating point precision may result
 125     in these promises being violated.
 126   * If `is_X == False`, callers should expect the operator to not have `X`.
 127   * If `is_X == None` (the default), callers should have no expectation either
 128     way.
 129   """
 130 
 131   def __init__(self,
 132                base_operator,
 133                u,
 134                diag_update=None,
 135                v=None,
 136                is_diag_update_positive=None,
 137                is_non_singular=None,
 138                is_self_adjoint=None,
 139                is_positive_definite=None,
 140                is_square=None,
 141                name="LinearOperatorLowRankUpdate"):
 142     """Initialize a `LinearOperatorLowRankUpdate`.
 143 
 144     This creates a `LinearOperator` of the form `A = L + U D V^H`, with
 145     `L` a `LinearOperator`, `U, V` both [batch] matrices, and `D` a [batch]
 146     diagonal matrix.
 147 
 148     If `L` is non-singular, solves and determinants are available.
 149     Solves/determinants both involve a solve/determinant of a `K x K` system.
 150     In the event that L and D are self-adjoint positive-definite, and U = V,
 151     this can be done using a Cholesky factorization.  The user should set the
 152     `is_X` matrix property hints, which will trigger the appropriate code path.
 153 
 154     Args:
 155       base_operator:  Shape `[B1,...,Bb, M, N]`.
 156       u:  Shape `[B1,...,Bb, M, K]` `Tensor` of same `dtype` as `base_operator`.
 157         This is `U` above.
 158       diag_update:  Optional shape `[B1,...,Bb, K]` `Tensor` with same `dtype`
 159         as `base_operator`.  This is the diagonal of `D` above.
 160          Defaults to `D` being the identity operator.
 161       v:  Optional `Tensor` of same `dtype` as `u` and shape `[B1,...,Bb, N, K]`
 162          Defaults to `v = u`, in which case the perturbation is symmetric.
 163          If `M != N`, then `v` must be set since the perturbation is not square.
 164       is_diag_update_positive:  Python `bool`.
 165         If `True`, expect `diag_update > 0`.
 166       is_non_singular:  Expect that this operator is non-singular.
 167         Default is `None`, unless `is_positive_definite` is auto-set to be
 168         `True` (see below).
 169       is_self_adjoint:  Expect that this operator is equal to its hermitian
 170         transpose.  Default is `None`, unless `base_operator` is self-adjoint
 171         and `v = None` (meaning `u=v`), in which case this defaults to `True`.
 172       is_positive_definite:  Expect that this operator is positive definite.
 173         Default is `None`, unless `base_operator` is positive-definite
 174         `v = None` (meaning `u=v`), and `is_diag_update_positive`, in which case
 175         this defaults to `True`.
 176         Note that we say an operator is positive definite when the quadratic
 177         form `x^H A x` has positive real part for all nonzero `x`.
 178       is_square:  Expect that this operator acts like square [batch] matrices.
 179       name: A name for this `LinearOperator`.
 180 
 181     Raises:
 182       ValueError:  If `is_X` flags are set in an inconsistent way.
 183     """
 184     dtype = base_operator.dtype
 185 
 186     if diag_update is not None:
 187       if is_diag_update_positive and dtype.is_complex:
 188         logging.warn("Note: setting is_diag_update_positive with a complex "
 189                      "dtype means that diagonal is real and positive.")
 190 
 191     if diag_update is None:
 192       if is_diag_update_positive is False:
 193         raise ValueError(
 194             "Default diagonal is the identity, which is positive.  However, "
 195             "user set 'is_diag_update_positive' to False.")
 196       is_diag_update_positive = True
 197 
 198     # In this case, we can use a Cholesky decomposition to help us solve/det.
 199     self._use_cholesky = (
 200         base_operator.is_positive_definite and base_operator.is_self_adjoint
 201         and is_diag_update_positive
 202         and v is None)
 203 
 204     # Possibly auto-set some characteristic flags from None to True.
 205     # If the Flags were set (by the user) incorrectly to False, then raise.
 206     if base_operator.is_self_adjoint and v is None and not dtype.is_complex:
 207       if is_self_adjoint is False:
 208         raise ValueError(
 209             "A = L + UDU^H, with L self-adjoint and D real diagonal.  Since"
 210             " UDU^H is self-adjoint, this must be a self-adjoint operator.")
 211       is_self_adjoint = True
 212 
 213     # The condition for using a cholesky is sufficient for SPD, and
 214     # we no weaker choice of these hints leads to SPD.  Therefore,
 215     # the following line reads "if hints indicate SPD..."
 216     if self._use_cholesky:
 217       if (
 218           is_positive_definite is False
 219           or is_self_adjoint is False
 220           or is_non_singular is False):
 221         raise ValueError(
 222             "Arguments imply this is self-adjoint positive-definite operator.")
 223       is_positive_definite = True
 224       is_self_adjoint = True
 225 
 226     values = base_operator.graph_parents + [u, diag_update, v]
 227     with ops.name_scope(name, values=values):
 228 
 229       # Create U and V.
 230       self._u = ops.convert_to_tensor(u, name="u")
 231       if v is None:
 232         self._v = self._u
 233       else:
 234         self._v = ops.convert_to_tensor(v, name="v")
 235 
 236       if diag_update is None:
 237         self._diag_update = None
 238       else:
 239         self._diag_update = ops.convert_to_tensor(
 240             diag_update, name="diag_update")
 241 
 242       # Create base_operator L.
 243       self._base_operator = base_operator
 244       graph_parents = base_operator.graph_parents + [
 245           self.u, self._diag_update, self.v]
 246       graph_parents = [p for p in graph_parents if p is not None]
 247 
 248       super(LinearOperatorLowRankUpdate, self).__init__(
 249           dtype=self._base_operator.dtype,
 250           graph_parents=graph_parents,
 251           is_non_singular=is_non_singular,
 252           is_self_adjoint=is_self_adjoint,
 253           is_positive_definite=is_positive_definite,
 254           is_square=is_square,
 255           name=name)
 256 
 257       # Create the diagonal operator D.
 258       self._set_diag_operators(diag_update, is_diag_update_positive)
 259       self._is_diag_update_positive = is_diag_update_positive
 260 
 261       self._check_shapes()
 262 
 263       # Pre-compute the so-called "capacitance" matrix
 264       #   C := D^{-1} + V^H L^{-1} U
 265       self._capacitance = self._make_capacitance()
 266       if self._use_cholesky:
 267         self._chol_capacitance = linalg_ops.cholesky(self._capacitance)
 268 
 269   def _check_shapes(self):
 270     """Static check that shapes are compatible."""
 271     # Broadcast shape also checks that u and v are compatible.
 272     uv_shape = array_ops.broadcast_static_shape(
 273         self.u.get_shape(), self.v.get_shape())
 274 
 275     batch_shape = array_ops.broadcast_static_shape(
 276         self.base_operator.batch_shape, uv_shape[:-2])
 277 
 278     self.base_operator.domain_dimension.assert_is_compatible_with(
 279         uv_shape[-2])
 280 
 281     if self._diag_update is not None:
 282       uv_shape[-1].assert_is_compatible_with(self._diag_update.get_shape()[-1])
 283       array_ops.broadcast_static_shape(
 284           batch_shape, self._diag_update.get_shape()[:-1])
 285 
 286   def _set_diag_operators(self, diag_update, is_diag_update_positive):
 287     """Set attributes self._diag_update and self._diag_operator."""
 288     if diag_update is not None:
 289       self._diag_operator = linear_operator_diag.LinearOperatorDiag(
 290           self._diag_update, is_positive_definite=is_diag_update_positive)
 291       self._diag_inv_operator = linear_operator_diag.LinearOperatorDiag(
 292           1. / self._diag_update, is_positive_definite=is_diag_update_positive)
 293     else:
 294       if self.u.get_shape()[-1].value is not None:
 295         r = self.u.get_shape()[-1].value
 296       else:
 297         r = array_ops.shape(self.u)[-1]
 298       self._diag_operator = linear_operator_identity.LinearOperatorIdentity(
 299           num_rows=r, dtype=self.dtype)
 300       self._diag_inv_operator = self._diag_operator
 301

 332   def _shape(self):
 333     batch_shape = array_ops.broadcast_static_shape(
 334         self.base_operator.batch_shape,
 335         self.u.get_shape()[:-2])
 336     return batch_shape.concatenate(self.base_operator.shape[-2:])
 337 
 338   def _shape_tensor(self):
 339     batch_shape = array_ops.broadcast_dynamic_shape(
 340         self.base_operator.batch_shape_tensor(),
 341         array_ops.shape(self.u)[:-2])
 342     return array_ops.concat(
 343         [batch_shape, self.base_operator.shape_tensor()[-2:]], axis=0)

 345   def _matmul(self, x, adjoint=False, adjoint_arg=False):
 346     u = self.u
 347     v = self.v
 348     l = self.base_operator
 349     d = self.diag_operator
 350 
 351     leading_term = l.matmul(x, adjoint=adjoint, adjoint_arg=adjoint_arg)
 352 
 353     if adjoint:
 354       uh_x = linear_operator_util.matmul_with_broadcast(
 355           u, x, adjoint_a=True, adjoint_b=adjoint_arg)
 356       d_uh_x = d.matmul(uh_x, adjoint=adjoint)
 357       v_d_uh_x = linear_operator_util.matmul_with_broadcast(
 358           v, d_uh_x)
 359       return leading_term + v_d_uh_x
 360     else:
 361       vh_x = linear_operator_util.matmul_with_broadcast(
 362           v, x, adjoint_a=True, adjoint_b=adjoint_arg)
 363       d_vh_x = d.matmul(vh_x, adjoint=adjoint)
 364       u_d_vh_x = linear_operator_util.matmul_with_broadcast(u, d_vh_x)
 365       return leading_term + u_d_vh_x
 366 
 367   def _determinant(self):
 368     if self.is_positive_definite:
 369       return math_ops.exp(self.log_abs_determinant())
 370     # The matrix determinant lemma gives
 371     # https://en.wikipedia.org/wiki/Matrix_determinant_lemma
 372     #   det(L + UDV^H) = det(D^{-1} + V^H L^{-1} U) det(D) det(L)
 373     #                  = det(C) det(D) det(L)
 374     # where C is sometimes known as the capacitance matrix,
 375     #   C := D^{-1} + V^H L^{-1} U
 376     det_c = linalg_ops.matrix_determinant(self._capacitance)
 377     det_d = self.diag_operator.determinant()
 378     det_l = self.base_operator.determinant()
 379     return det_c * det_d * det_l
 380 
 381   def _log_abs_determinant(self):
 382     # Recall
 383     #   det(L + UDV^H) = det(D^{-1} + V^H L^{-1} U) det(D) det(L)
 384     #                  = det(C) det(D) det(L)
 385     log_abs_det_d = self.diag_operator.log_abs_determinant()
 386     log_abs_det_l = self.base_operator.log_abs_determinant()
 387 
 388     if self._use_cholesky:
 389       chol_cap_diag = array_ops.matrix_diag_part(self._chol_capacitance)
 390       log_abs_det_c = 2 * math_ops.reduce_sum(
 391           math_ops.log(chol_cap_diag), reduction_indices=[-1])
 392     else:
 393       det_c = linalg_ops.matrix_determinant(self._capacitance)
 394       log_abs_det_c = math_ops.log(math_ops.abs(det_c))
 395       if self.dtype.is_complex:
 396         log_abs_det_c = math_ops.cast(log_abs_det_c, dtype=self.dtype)
 397 
 398     return log_abs_det_c + log_abs_det_d + log_abs_det_l
 399 
 400   def _solve(self, rhs, adjoint=False, adjoint_arg=False):
 401     if self.base_operator.is_non_singular is False:
 402       raise ValueError(
 403           "Solve not implemented unless this is a perturbation of a "
 404           "non-singular LinearOperator.")
 405     # The Woodbury formula gives:
 406     # https://en.wikipedia.org/wiki/Woodbury_matrix_identity
 407     #   (L + UDV^H)^{-1}
 408     #   = L^{-1} - L^{-1} U (D^{-1} + V^H L^{-1} U)^{-1} V^H L^{-1}
 409     #   = L^{-1} - L^{-1} U C^{-1} V^H L^{-1}
 410     # where C is the capacitance matrix, C := D^{-1} + V^H L^{-1} U
 411     # Note also that, with ^{-H} being the inverse of the adjoint,
 412     #   (L + UDV^H)^{-H}
 413     #   = L^{-H} - L^{-H} V C^{-H} U^H L^{-H}
 414     l = self.base_operator
 415     if adjoint:
 416       v = self.u
 417       u = self.v
 418     else:
 419       v = self.v
 420       u = self.u
 421 
 422     # L^{-1} rhs
 423     linv_rhs = l.solve(rhs, adjoint=adjoint, adjoint_arg=adjoint_arg)
 424     # V^H L^{-1} rhs
 425     vh_linv_rhs = linear_operator_util.matmul_with_broadcast(
 426         v, linv_rhs, adjoint_a=True)
 427     # C^{-1} V^H L^{-1} rhs
 428     if self._use_cholesky:
 429       capinv_vh_linv_rhs = linear_operator_util.cholesky_solve_with_broadcast(
 430           self._chol_capacitance, vh_linv_rhs)
 431     else:
 432       capinv_vh_linv_rhs = linear_operator_util.matrix_solve_with_broadcast(
 433           self._capacitance, vh_linv_rhs, adjoint=adjoint)
 434     # U C^{-1} V^H M^{-1} rhs
 435     u_capinv_vh_linv_rhs = linear_operator_util.matmul_with_broadcast(
 436         u, capinv_vh_linv_rhs)
 437     # L^{-1} U C^{-1} V^H L^{-1} rhs
 438     linv_u_capinv_vh_linv_rhs = l.solve(u_capinv_vh_linv_rhs, adjoint=adjoint)
 439 
 440     # L^{-1} - L^{-1} U C^{-1} V^H L^{-1}
 441     return linv_rhs - linv_u_capinv_vh_linv_rhs
 442 
 443   def _make_capacitance(self):
 444     # C := D^{-1} + V^H L^{-1} U
 445     # which is sometimes known as the "capacitance" matrix.
 446 
 447     # L^{-1} U
 448     linv_u = self.base_operator.solve(self.u)
 449     # V^H L^{-1} U
 450     vh_linv_u = linear_operator_util.matmul_with_broadcast(
 451         self.v, linv_u, adjoint_a=True)
 452 
 453     # D^{-1} + V^H L^{-1} V
 454     capacitance = self._diag_inv_operator.add_to_tensor(vh_linv_u)
 455     return capacitance
