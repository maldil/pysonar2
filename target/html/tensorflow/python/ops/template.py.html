  40 @tf_export("make_template")
  41 def make_template(name_, func_, create_scope_now_=False, unique_name_=None,
  42                   custom_getter_=None, **kwargs):
  43   """Given an arbitrary function, wrap it so that it does variable sharing.
  44 
  45   This wraps `func_` in a Template and partially evaluates it. Templates are
  46   functions that create variables the first time they are called and reuse them
  47   thereafter. In order for `func_` to be compatible with a `Template` it must
  48   have the following properties:
  49 
  50   * The function should create all trainable variables and any variables that
  51      should be reused by calling `tf.get_variable`. If a trainable variable is
  52      created using `tf.Variable`, then a ValueError will be thrown. Variables
  53      that are intended to be locals can be created by specifying
  54      `tf.Variable(..., trainable=false)`.
  55   * The function may use variable scopes and other templates internally to
  56       create and reuse variables, but it shouldn't use `tf.global_variables` to
  57       capture variables that are defined outside of the scope of the function.
  58   * Internal scopes and variable names should not depend on any arguments that
  59       are not supplied to `make_template`. In general you will get a ValueError
  60       telling you that you are trying to reuse a variable that doesn't exist
  61       if you make a mistake.
  62 
  63   In the following example, both `z` and `w` will be scaled by the same `y`. It
  64   is important to note that if we didn't assign `scalar_name` and used a
  65   different name for z and w that a `ValueError` would be thrown because it
  66   couldn't reuse the variable.
  67 
  68   ```python
  69   def my_op(x, scalar_name):
  70     var1 = tf.get_variable(scalar_name,
  71                            shape=[],
  72                            initializer=tf.constant_initializer(1))
  73     return x * var1
  74 
  75   scale_by_y = tf.make_template('scale_by_y', my_op, scalar_name='y')
  76 
  77   z = scale_by_y(input1)
  78   w = scale_by_y(input2)
  79   ```
  80 
  81   As a safe-guard, the returned function will raise a `ValueError` after the
  82   first call if trainable variables are created by calling `tf.Variable`.
  83 
  84   If all of these are true, then 2 properties are enforced by the template:
  85 
  86   1. Calling the same template multiple times will share all non-local
  87       variables.
  88   2. Two different templates are guaranteed to be unique, unless you reenter the
  89       same variable scope as the initial definition of a template and redefine
  90       it. An examples of this exception:
  91 
  92   ```python
  93   def my_op(x, scalar_name):
  94     var1 = tf.get_variable(scalar_name,
  95                            shape=[],
  96                            initializer=tf.constant_initializer(1))
  97     return x * var1
  98 
  99   with tf.variable_scope('scope') as vs:
 100     scale_by_y = tf.make_template('scale_by_y', my_op, scalar_name='y')
 101     z = scale_by_y(input1)
 102     w = scale_by_y(input2)
 103 
 104   # Creates a template that reuses the variables above.
 105   with tf.variable_scope(vs, reuse=True):
 106     scale_by_y2 = tf.make_template('scale_by_y', my_op, scalar_name='y')
 107     z2 = scale_by_y2(input1)
 108     w2 = scale_by_y2(input2)
 109   ```
 110 
 111   Depending on the value of `create_scope_now_`, the full variable scope may be
 112   captured either at the time of first call or at the time of construction. If
 113   this option is set to True, then all Tensors created by repeated calls to the
 114   template will have an extra trailing _N+1 to their name, as the first time the
 115   scope is entered in the Template constructor no Tensors are created.
 116 
 117   Note: `name_`, `func_` and `create_scope_now_` have a trailing underscore to
 118   reduce the likelihood of collisions with kwargs.
 119 
 120   Args:
 121     name_: A name for the scope created by this template. If necessary, the name
 122       will be made unique by appending `_N` to the name.
 123     func_: The function to wrap.
 124     create_scope_now_: Boolean controlling whether the scope should be created
 125       when the template is constructed or when the template is called. Default
 126       is False, meaning the scope is created when the template is called.
 127     unique_name_: When used, it overrides name_ and is not made unique. If a
 128       template of the same scope/unique_name already exists and reuse is false,
 129       an error is raised. Defaults to None.
 130     custom_getter_: Optional custom getter for variables used in `func_`. See
 131       the `tf.get_variable` `custom_getter` documentation for
 132       more information.
 133     **kwargs: Keyword arguments to apply to `func_`.
 134 
 135   Returns:
 136     A function to encapsulate a set of variables which should be created once
 137     and reused. An enclosing scope will be created either when `make_template`
 138     is called or when the result is called, depending on the value of
 139     `create_scope_now_`. Regardless of the value, the first time the template
 140     is called it will enter the scope with no reuse, and call `func_` to create
 141     variables, which are guaranteed to be unique. All subsequent calls will
 142     re-enter the scope and reuse those variables.
 143 
 144   Raises:
 145     ValueError: if `name_` is None.
 146   """
 147   return make_template_internal(
 148       name_,
 149       func_,
 150       create_scope_now_,
 151       unique_name_,
 152       custom_getter_,
 153       create_graph_function_=False,
 154       **kwargs)
 155

 157 def make_template_internal(name_,
 158                            func_,
 159                            create_scope_now_=False,
 160                            unique_name_=None,
 161                            custom_getter_=None,
 162                            create_graph_function_=False,
 163                            **kwargs):
 164   """Make a template, optionally compiling func_ into a graph function.
 165 
 166   See `make_template` for full documentation.
 167 
 168   Args:
 169     name_: A name for the scope created by this template. If necessary, the name
 170       will be made unique by appending `_N` to the name.
 171     func_: The function to wrap.
 172     create_scope_now_: Boolean controlling whether the scope should be created
 173       when the template is constructed or when the template is called. Default
 174       is False, meaning the scope is created when the template is called.
 175     unique_name_: When used, it overrides name_ and is not made unique. If a
 176       template of the same scope/unique_name already exists and reuse is false,
 177       an error is raised. Defaults to None. If executing eagerly, must be None.
 178     custom_getter_: Optional custom getter for variables used in `func_`. See
 179       the `tf.get_variable` `custom_getter` documentation for
 180       more information.
 181     create_graph_function_: When True, `func_` will be executed as a graph
 182       function. This implies that `func_` must satisfy the properties that
 183       `function.defun` requires of functions: See the documentation of
 184       `function.defun` for details. When executing eagerly, setting this flag to
 185       True can improve performance. Regardless of whether eager execution is
 186       enabled, enabling this flag gives the caller access to graph-function
 187       semantics, i.e., accesses to variables are totally ordered and
 188       side-effecting ops are not pruned.
 189     **kwargs: Keyword arguments to apply to `func_`.
 190 
 191   Returns:
 192     A function to encapsulate a set of variables which should be created once
 193     and reused. An enclosing scope will be created either when `make_template`
 194     is called or when the result is called, depending on the value of
 195     `create_scope_now_`. Regardless of the value, the first time the template
 196     is called it will enter the scope with no reuse, and call `func_` to create
 197     variables, which are guaranteed to be unique. All subsequent calls will
 198     re-enter the scope and reuse those variables.
 199 
 200   Raises:
 201     ValueError: if `name_` is None.
 202     ValueError: if `unique_name_` is not None and eager execution is enabled.
 203   """
 204 
 205   if kwargs:
 206     func_ = tf_decorator.make_decorator(func_, functools.partial(
 207         func_, **kwargs))
 208   if context.executing_eagerly():
 209     if unique_name_ is not None:
 210       raise ValueError(
 211           "unique_name_ cannot be used when eager exeuction is enabled.")
 212     return EagerTemplate(
 213         name_,
 214         func_,
 215         create_scope_now=create_scope_now_,
 216         custom_getter=custom_getter_,
 217         create_graph_function=create_graph_function_)
 218   return Template(
 219       name_,
 220       func_,
 221       create_scope_now=create_scope_now_,
 222       unique_name=unique_name_,
 223       custom_getter=custom_getter_,
 224       create_graph_function=create_graph_function_)

 227 def _skip_common_stack_elements(stacktrace, base_case):
 228   """Skips items that the target stacktrace shares with the base stacktrace."""
 229   for i, (trace, base) in enumerate(zip(stacktrace, base_case)):
 230     if trace != base:
 231       return stacktrace[i:]
 232   return stacktrace[-1:]

 235 class Template(checkpointable.CheckpointableBase):
 236   """Wrap a function to aid in variable sharing.
 237 
 238   Templates are functions that create variables the first time they are called
 239   and reuse them thereafter. See `make_template` for full documentation.
 240 
 241   Note: By default, the full variable scope is captured at the time of first
 242   call. If `create_scope_now_` is passed as True to the constructor, the full
 243   scope will be captured there, but no variables will created until the first
 244   call.
 245   """
 246 
 247   def __init__(self, name, func, create_scope_now=False, unique_name=None,
 248                custom_getter=None, create_graph_function=False):
 249     """Creates a template for the given function.
 250 
 251     Args:
 252       name: A name for the scope created by this template. The
 253         name will be made unique by appending `_N` to the it (see how
 254         `tf.variable_scope` treats the `default_name` for details).
 255       func: The function to apply each time.
 256       create_scope_now: Whether to create the scope at Template construction
 257         time, rather than first call. Defaults to false. Creating the scope at
 258         construction time may be more convenient if the template is to passed
 259         through much lower level code, and you want to be sure of the scope
 260         name without knowing exactly where it will be first called. If set to
 261         True, the scope will be created in the constructor, and all subsequent
 262         times in `__call__`, leading to a trailing numeral being added to the
 263         names of all created Tensors. If set to False, the scope will be created
 264         at the first call location.
 265       unique_name: When used, it overrides `name` and is not made unique. If a
 266         template of the same scope/unique_name already exists and reuse is
 267         false, an error is raised. Defaults to None.
 268       custom_getter: optional custom getter to pass to `variable_scope()`
 269       create_graph_function: When True, `func` will be executed as a graph
 270         function. Enabling this flag gives the caller access to graph-function
 271         semantics, i.e., accesses to variables are totally ordered and
 272         side-effecting ops are not pruned.
 273 
 274     Raises:
 275       ValueError: if `name` is None.
 276     """
 277     if create_graph_function:
 278       self._func = function.defun(func)
 279     else:
 280       self._func = func
 281     self._stacktrace = traceback.format_stack()[:-2]
 282     self._name = name
 283     self._unique_name = unique_name
 284     self._custom_getter = custom_getter
 285     if name is None:
 286       raise ValueError("name cannot be None.")
 287     if create_scope_now:
 288       with variable_scope._pure_variable_scope(  # pylint:disable=protected-access
 289           (self._unique_name or
 290            variable_scope._get_unique_variable_scope(self._name)),  # pylint:disable=protected-access
 291           custom_getter=self._custom_getter) as vs:
 292         self._variable_scope = vs
 293     else:
 294       self._variable_scope = None
 295     # This variable keeps track of whether the template has been called yet,
 296     # which is not the same as whether the scope has been created.
 297     self._variables_created = False
 298 
 299   def _call_func(self, args, kwargs):
 300     try:
 301       vars_at_start = len(
 302           ops.get_collection_ref(ops.GraphKeys.GLOBAL_VARIABLES))
 303       trainable_at_start = len(
 304           ops.get_collection_ref(ops.GraphKeys.TRAINABLE_VARIABLES))
 305       if self._variables_created:
 306         result = self._func(*args, **kwargs)
 307       else:
 308         # The first time we run, restore variables if necessary (via
 309         # Checkpointable).
 310         with checkpointable_util.capture_dependencies(template=self):
 311           result = self._func(*args, **kwargs)
 312 
 313       if self._variables_created:
 314         # Variables were previously created, implying this is not the first
 315         # time the template has been called. Check to make sure that no new
 316         # trainable variables were created this time around.
 317         trainable_variables = ops.get_collection_ref(
 318             ops.GraphKeys.TRAINABLE_VARIABLES)
 319         # If a variable that we intend to train is created as a side effect
 320         # of creating a template, then that is almost certainly an error.
 321         if trainable_at_start != len(trainable_variables):
 322           raise ValueError("Trainable variable created when calling a template "
 323                            "after the first time, perhaps you used tf.Variable "
 324                            "when you meant tf.get_variable: %s" %
 325                            (trainable_variables[trainable_at_start:],))
 326 
 327         # Non-trainable tracking variables are a legitimate reason why a new
 328         # variable would be created, but it is a relatively advanced use-case,
 329         # so log it.
 330         variables = ops.get_collection_ref(ops.GraphKeys.GLOBAL_VARIABLES)
 331         if vars_at_start != len(variables):
 332           logging.info("New variables created when calling a template after "
 333                        "the first time, perhaps you used tf.Variable when you "
 334                        "meant tf.get_variable: %s",
 335                        variables[vars_at_start:])
 336       else:
 337         self._variables_created = True
 338       return result
 339     except Exception as exc:
 340       # Reraise the exception, but append the original definition to the
 341       # trace.
 342       args = exc.args
 343       if not args:
 344         arg0 = ""
 345       else:
 346         arg0 = args[0]
 347       trace = "".join(_skip_common_stack_elements(self._stacktrace,
 348                                                   traceback.format_stack()))
 349       arg0 = "%s\n\noriginally defined at:\n%s" % (arg0, trace)
 350       new_args = [arg0]
 351       new_args.extend(args[1:])
 352       exc.args = tuple(new_args)
 353       raise
 354 
 355   def __call__(self, *args, **kwargs):
 356     if self._variable_scope:
 357       # Only reuse variables if they were already created.
 358       with variable_scope.variable_scope(
 359           self._variable_scope, reuse=self._variables_created):
 360         return self._call_func(args, kwargs)
 361     else:
 362       # The scope was not created at construction time, so create it here.
 363       # Subsequent calls should reuse variables.
 364       with variable_scope.variable_scope(
 365           self._unique_name, self._name,
 366           custom_getter=self._custom_getter) as vs:
 367         self._variable_scope = vs
 368         return self._call_func(args, kwargs)

 370   @property
 371   def name(self):
 372     """Returns the name given to this Template."""
 373     return self._name
 374 
 375   @property
 376   def func(self):
 377     """Returns the func given to this Template."""
 378     return self._func
 379 
 380   @property
 381   def variable_scope(self):
 382     """Returns the variable scope object created by this Template."""
 383     return self._variable_scope
 384 
 385   @property
 386   def variable_scope_name(self):
 387     """Returns the variable scope name created by this Template."""
 388     if self._variable_scope:
 389       name = self._variable_scope.name
 390       # To prevent partial matches on the scope_name, we add '/' at the end.
 391       return name if name[-1] == "/" else name + "/"
 392 
 393   @property
 394   def variables(self):
 395     """Returns the list of global and local variables created by the Template.
 396     """
 397     return self.global_variables + self.local_variables
 398 
 399   @property
 400   def trainable_variables(self):
 401     """Returns the list of trainable variables created by the Template."""
 402     if self._variables_created:
 403       return ops.get_collection(ops.GraphKeys.TRAINABLE_VARIABLES,
 404                                 self.variable_scope_name)
 405     else:
 406       return []
 407 
 408   @property
 409   def non_trainable_variables(self):
 410     """Returns the list of non-trainable variables created by the Template."""
 411     # TODO(apassos) Make sure it matches Eager when using local variables.
 412     global_variables = self.global_variables
 413     trainable_variables = set(self.trainable_variables)
 414     return [x for x in global_variables if x not in trainable_variables]
 415 
 416   @property
 417   def global_variables(self):
 418     """Returns the list of global variables created by the Template."""
 419     if self._variables_created:
 420       return ops.get_collection(ops.GraphKeys.GLOBAL_VARIABLES,
 421                                 self.variable_scope_name)
 422     else:
 423       return []
 424 
 425   @property
 426   def local_variables(self):
 427     """Returns the list of global variables created by the Template."""
 428     if self._variables_created:
 429       return ops.get_collection(ops.GraphKeys.LOCAL_VARIABLES,
 430                                 self.variable_scope_name)
 431     else:
 432       return []
 433 
 434   @property
 435   def weights(self):
 436     """List of weights/variables created by the Template."""
 437     return self.variables
 438 
 439   @property
 440   def trainable_weights(self):
 441     """List of trainable weights/variables created by the Template."""
 442     return self.trainable_variables
 443 
 444   @property
 445   def non_trainable_weights(self):
 446     """List of non-trainable weights/variables created by the Template."""
 447     return self.non_trainable_variables
 448 
 449   @property
 450   @deprecated(
 451       "2017-02-21", "The .var_scope property is deprecated. Please change your "
 452       "code to use the .variable_scope property")
 453   def var_scope(self):
 454     """Returns the variable scope object created by this Template."""
 455     return self._variable_scope

 458 class _EagerTemplateVariableStore(object):
 459   """Wrapper around EagerVariableStore to support nesting EagerTemplates.
 460   """
 461 
 462   def __init__(self, variable_scope_name):
 463     self._variable_scope_name = variable_scope_name
 464     default = variable_scope._get_default_variable_store()  # pylint: disable=protected-access
 465     if default._store_eager_variables:  # pylint: disable=protected-access
 466       self._eager_variable_store = variable_scope.EagerVariableStore(default)
 467     else:
 468       self._eager_variable_store = variable_scope.EagerVariableStore()
 469 
 470   def set_variable_scope_name(self, variable_scope_name):
 471     self._variable_scope_name = variable_scope_name
 472 
 473   @tf_contextlib.contextmanager
 474   def as_default(self):
 475     try:
 476       with self._eager_variable_store.as_default():
 477         yield
 478     finally:
 479       # Each _EagerTemplateVariableStore object lives underneath a variable
 480       # scope (see EagerTemplate.__call__). This variable scope's subscopes are
 481       # closed when the EagerTemplate object returns from __call__. For
 482       # top-level _EagerTemplateVariableStore objects, the variable store to
 483       # which the variable scope is attached is different from the
 484       # EagerVariableStore; as such it is necessary to close its subscopes
 485       # here as well.
 486       if self._variable_scope_name is None:
 487         raise RuntimeError("A variable scope must be set before an "
 488                            "_EagerTemplateVariableStore object exits.")
 489       variable_scope.get_variable_scope_store().close_variable_subscopes(
 490           self._variable_scope_name)
 491 
 492   def _variables_in_scope(self, variable_list):
 493     if self._variable_scope_name is None:
 494       raise RuntimeError(
 495           "A variable scope must be set before variables can be accessed.")
 496     return [
 497         v for v in variable_list
 498         if v.name.startswith(self._variable_scope_name + "/")
 499     ]
 500 
 501   def variables(self):
 502     return self._variables_in_scope(self._eager_variable_store.variables())
 503 
 504   def trainable_variables(self):
 505     return self._variables_in_scope(
 506         self._eager_variable_store.trainable_variables())
 507 
 508   def non_trainable_variables(self):
 509     return self._variables_in_scope(
 510         self._eager_variable_store.non_trainable_variables())

 513 class EagerTemplate(Template):
 514   """Wrap a function to aid in variable sharing in Eager mode.
 515 
 516   Templates are functions that create variables the first time they are called
 517   and reuse them thereafter. See `make_template` for full documentation.
 518 
 519   Note: By default, the full variable scope is captured at the time of first
 520   call. If `create_scope_now` is passed as True to the constructor, the full
 521   scope will be captured there, but no variables will be created until the first
 522   call.
 523   """
 524 
 525   def __init__(self, name, func, create_scope_now=False, custom_getter=None,
 526                create_graph_function=False):
 527     """Creates a template for the given function.
 528 
 529     Args:
 530       name: A name for the scope created by this template. The
 531         name will be made unique by appending `_N` to the it (see how
 532         `tf.variable_scope` treats the `default_name` for details).
 533       func: The function to apply each time.
 534       create_scope_now: Whether to create the scope at Template construction
 535         time, rather than first call. Defaults to false. Creating the scope at
 536         construction time may be more convenient if the template is passed
 537         through much lower level code, and you want to be sure of the scope
 538         name without knowing exactly where it will be first called. If set to
 539         True, the scope will be created in the constructor, and all subsequent
 540         times in `__call__`, leading to a trailing numeral being added to the
 541         names of all created Tensors. If set to False, the scope will be created
 542         at the first call location.
 543       custom_getter: optional custom getter to pass to `variable_scope()`
 544       create_graph_function: When True, `func` will be executed as a graph
 545         function. Enabling this flag allows the caller to reap the performance
 546         benefits associated with executing graphs, at the cost of sacrificing
 547         debuggability; however, not all Python functions can be compiled into
 548         graph functions. See the documentation for `function.defun` for details.
 549 
 550     Raises:
 551       RuntimeError: if eager execution is not enabled.
 552     """
 553     if not context.executing_eagerly():
 554       raise RuntimeError(
 555           "{} objects can only be used when eager execution is enabled, use "
 556           "tf.Template for graph construction".
 557           format(type(self)))
 558     super(EagerTemplate, self).__init__(name, func, create_scope_now, None,
 559                                         custom_getter, create_graph_function)
 560     if self._variable_scope is not None:
 561       variable_scope_name = self._variable_scope.name
 562     else:
 563       # Defer setting the variable scope name until the variable scope
 564       # is created in __call__.
 565       variable_scope_name = None
 566     self._template_store = _EagerTemplateVariableStore(variable_scope_name)
 567     self._variable_scope_context_manager = None
 568 
 569   def _call_func(self, args, kwargs):
 570     try:
 571       vars_at_start = self._template_store.variables()
 572       trainable_at_start = self._template_store.trainable_variables()
 573       if self._variables_created:
 574         result = self._func(*args, **kwargs)
 575       else:
 576         # The first time we run, restore variables if necessary (via
 577         # Checkpointable).
 578         with checkpointable_util.capture_dependencies(template=self):
 579           result = self._func(*args, **kwargs)
 580 
 581       if self._variables_created:
 582         # Variables were previously created, implying this is not the first
 583         # time the template has been called. Check to make sure that no new
 584         # trainable variables were created this time around.
 585         trainable_variables = self._template_store.trainable_variables()
 586         # If a variable that we intend to train is created as a side effect
 587         # of creating a template, then that is almost certainly an error.
 588         if len(trainable_at_start) != len(trainable_variables):
 589           raise ValueError("Trainable variable created when calling a template "
 590                            "after the first time, perhaps you used tf.Variable "
 591                            "when you meant tf.get_variable: %s" %
 592                            list(set(trainable_variables) -
 593                                 set(trainable_at_start)))
 594 
 595         # Non-trainable tracking variables are a legitimate reason why a new
 596         # variable would be created, but it is a relatively advanced use-case,
 597         # so log it.
 598         variables = self._template_store.variables()
 599         if len(vars_at_start) != len(variables):
 600           logging.info("New variables created when calling a template after "
 601                        "the first time, perhaps you used tf.Variable when you "
 602                        "meant tf.get_variable: %s",
 603                        list(set(variables) - set(vars_at_start)))
 604       else:
 605         self._variables_created = True
 606       return result
 607     except Exception as exc:
 608       # Reraise the exception, but append the original definition to the
 609       # trace.
 610       args = exc.args
 611       if not args:
 612         arg0 = ""
 613       else:
 614         arg0 = args[0]
 615       trace = "".join(_skip_common_stack_elements(self._stacktrace,
 616                                                   traceback.format_stack()))
 617       arg0 = "%s\n\noriginally defined at:\n%s" % (arg0, trace)
 618       new_args = [arg0]
 619       new_args.extend(args[1:])
 620       exc.args = tuple(new_args)
 621       raise
 622 
 623   def __call__(self, *args, **kwargs):
 624     # In both branches below, the template store is installed as default after
 625     # the variable scope is opened in order to ensure that templates nested at
 626     # the same level correctly uniquify lower variable scope names.
 627     if self._variable_scope:
 628       # Create a cache for the variable scope context manager the first time
 629       # around so that we don't have to keep recreating it.
 630       if not self._variable_scope_context_manager:
 631         self._variable_scope_context_manager = variable_scope.variable_scope(
 632             self._variable_scope, reuse=variable_scope.AUTO_REUSE)
 633       with self._variable_scope_context_manager:
 634         with self._template_store.as_default():
 635           return self._call_func(args, kwargs)
 636     else:
 637       # The scope was not created at construction time, so create it here.
 638       # Subsequent calls should reuse variables.
 639       with variable_scope.variable_scope(
 640           self._unique_name, self._name,
 641           custom_getter=self._custom_getter) as vs:
 642         self._variable_scope = vs
 643         # Because the scope was not created at construction time, the template
 644         # store's variable scope name is unset; set it here.
 645         self._template_store.set_variable_scope_name(vs.name)
 646         with self._template_store.as_default():
 647           return self._call_func(args, kwargs)
