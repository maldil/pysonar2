  55 def while_loop(cond, body, loop_vars, name=None):
  56   """Like tf.while_loop, except emits a single While op."""
  57   if not name:
  58     name = "while"
  59 
  60   with ops.name_scope(name) as scope:
  61     with ops.name_scope(None):
  62       cond_name = _get_unique_name(("%scond" % scope).replace("/", "_"))
  63       body_name = _get_unique_name(("%sbody" % scope).replace("/", "_"))
  64 
  65     flattened_loop_vars = nest.flatten(loop_vars)
  66     num_outputs = len(flattened_loop_vars)
  67 
  68     # Add loop counter needed for computing gradients.
  69     flattened_loop_vars = [constant_op.constant(0., name="loop_counter")
  70                           ] + flattened_loop_vars
  71 
  72     # Build a `cond` wrapper that can handle the extra counter loop_var.
  73     def wrapped_cond(unused_loop_counter, *loop_vars):
  74       return cond(*loop_vars)
  75 
  76     cond_graph = function.func_graph_from_py_func(cond_name, wrapped_cond,
  77                                                   flattened_loop_vars, {})
  78 
  79     # Add external_captures of cond to the list of loop vars.
  80     # Note that external tensors will be treated as loop invariants, i.e.,
  81     # the value of that tensor in each iteration is the same as it was at the
  82     # beginning of the loop execution.
  83     flattened_loop_vars = flattened_loop_vars + cond_graph.external_captures
  84 
  85     def wrapped_body(loop_counter, *args):
  86       """Loop body augmented with counter update.
  87 
  88       Args:
  89         loop_counter: Loop counter which needs to be incremented in the body.
  90         *args: List of args
  91           args[:num_outputs] - Args for the original loop body.
  92           args[num_outputs:] - External captures of cond. These get passed
  93             through as is.
  94 
  95       Returns:
  96         A list of tensors the same length as args.
  97       """
  98       outputs = body(*args[:num_outputs])
  99       if not isinstance(outputs, collections.Sequence):
 100         outputs = [outputs]
 101 
 102       # Return the external_captures of cond_graph as is, i.e., treat them as
 103       # loop invariants.
 104       # TODO(srbs): Update lowering code to create _Enter nodes with
 105       # is_constant=True for inputs that are directly passed to outputs.
 106       return [loop_counter + 1] + list(outputs) + list(args[num_outputs:])
 107 
 108     body_graph = function.func_graph_from_py_func(body_name, wrapped_body,
 109                                                   flattened_loop_vars, {})
 110     # Add external captures of body to the list of loop vars.
 111     # Note that external tensors will be treated as loop invariants, i.e.,
 112     # the value of that tensor in each iteration is the same as it was at the
 113     # beginning of the loop execution.
 114     flattened_loop_vars = flattened_loop_vars + body_graph.external_captures
 115     # TODO(srbs): Update lowering code to create _Enter nodes with
 116     # is_constant=True for inputs that are directly passed to outputs.
 117     body_graph.outputs.extend(body_graph.internal_captures)
 118 
 119     # Capture `external_captures` of `body_graph` in `cond_graph` so that it
 120     # expects to receive those as arguments.
 121     # TODO(srbs): Dedup tensors that are captured in both the cond and body.
 122     # This logic already exists in cond_v2.
 123     with cond_graph.as_default():
 124       for external_capture in body_graph.external_captures:
 125         cond_graph.capture(external_capture)
 126 
 127     # Export all tensors in the loop body that may be needed for gradient
 128     # computation. We do this by accumulating the intermediate values in
 129     # TensorLists.
 130     intermediate_tensors = _get_intermediates(body_graph)
 131 
 132     for intermediate_tensor in intermediate_tensors:
 133       # TODO(srbs): Cache and re-use empty tensor lists.
 134       tensor_list = list_ops.empty_tensor_list(
 135           element_dtype=intermediate_tensor.dtype,
 136           element_shape=_get_tensor_convertible_shape(
 137               intermediate_tensor.shape))
 138       flattened_loop_vars.append(tensor_list)
 139       with cond_graph.as_default():
 140         # Add a placeholder to cond_graph's inputs corresponding to the
 141         # tensor_list.
 142         cond_graph.capture(tensor_list)
 143       with body_graph.as_default():
 144         # Push the intermediate tensor to the tensor list. This captures the
 145         # `tensor_list` as well.
 146         appended_tensor_list = list_ops.tensor_list_push_back(
 147             tensor_list,
 148             intermediate_tensor)
 149         # Add this modified tensor list to the list of outputs.
 150         body_graph.outputs.append(appended_tensor_list)
 151 
 152     outputs = gen_functional_ops._while(
 153         flattened_loop_vars,
 154         cond_v2._create_new_tf_function(cond_graph),
 155         cond_v2._create_new_tf_function(body_graph),
 156         name=scope)
 157 
 158     _copy_handle_data(body_graph.outputs, outputs)
 159     _maybe_set_lowering_attr(outputs[0].op)
 160 
 161   # First var is loop counter.
 162   if num_outputs == 1:
 163     return outputs[1]
 164   else:
 165     return nest.pack_sequence_as(loop_vars, outputs[1:1 + num_outputs])

 168 @ops.RegisterGradient("While")
 169 def _WhileGrad(op, *grads):  # pylint: disable=invalid-name
 170   """The gradient of a While op produced by while_loop."""
 171   body_graph = _get_body_graph(op)
 172 
 173   # Replace None gradients with zeros. This is needed because `grads` could have
 174   # None incoming gradients for the TensorLists. If we pass None's through, the
 175   # custom gradient of TensorListPopBack will create an EmptyTensorList inside
 176   # the FuncGraph which is undesirable.
 177   # TODO(b/80444525): There might be an issue with treating no gradient as zero
 178   # gradient in certain cases. Consider replacing None gradients with Zeros
 179   # for accumulators only.
 180   grads = [
 181       g if g is not None else array_ops.zeros_like(output)
 182       for g, output in zip(grads, op.outputs)
 183   ]
 184 
 185   body_grad_graph, args = _create_grad_func(
 186       body_graph, grads,
 187       _get_unique_name("%s_grad" % body_graph.name), op)
 188 
 189   intermediate_tensors = _get_intermediates(body_grad_graph)
 190 
 191   for intermediate_tensor in intermediate_tensors:
 192     tensor_list = list_ops.empty_tensor_list(
 193         element_dtype=intermediate_tensor.dtype,
 194         element_shape=_get_tensor_convertible_shape(intermediate_tensor.shape))
 195     with body_grad_graph.as_default():
 196       tensor_list_ph = body_grad_graph.capture(tensor_list, whitelisted=True)
 197       # Push the intermediate tensor to the tensor list.
 198       appended_tensor_list = list_ops.tensor_list_push_back(tensor_list_ph,
 199                                                             intermediate_tensor)
 200       # Add this modified tensor list to the list of outputs.
 201       body_grad_graph.outputs.append(appended_tensor_list)
 202 
 203   def grad_cond(counter, max_iters, *unused_args):
 204     return counter < max_iters
 205 
 206   loop_vars = args + body_grad_graph.external_captures
 207   cond_grad_graph = function.func_graph_from_py_func(
 208       _get_unique_name("%s_grad_cond" % op.name),
 209       grad_cond, loop_vars, {})
 210 
 211   assert len(loop_vars) == len(body_grad_graph.inputs)
 212   assert len(loop_vars) == len(body_grad_graph.outputs)
 213   assert len(loop_vars) == len(cond_grad_graph.inputs)
 214 
 215   outputs = gen_functional_ops._while(
 216       loop_vars,
 217       cond_v2._create_new_tf_function(cond_grad_graph),
 218       cond_v2._create_new_tf_function(body_grad_graph),
 219       name=_get_unique_name("%s_grad" % op.name))
 220 
 221   _copy_handle_data(body_grad_graph.outputs, outputs)
 222   _maybe_set_lowering_attr(outputs[0].op)
 223 
 224   # outputs[0] is the loop counter.
 225   # outputs[1] is the total number of loop iterations.
 226   return outputs[2:2 + len(op.inputs)]

 229 # TODO(srbs): Pull this into common utils for cond_v2 and while_v2.
 230 def _get_body_graph(while_op):
 231   """Returns `FuncGraph` for the while body.
 232 
 233   Args:
 234     while_op: The While Operation.
 235 
 236   Returns:
 237     `FuncGraph` for the while body.
 238   """
 239   extra_inputs = list(while_op.inputs)
 240   input_shapes = [t.shape for t in extra_inputs]
 241   func_name = while_op.get_attr("body").name
 242   fdef = while_op.graph._get_function(func_name).definition
 243   func_graph = function_def_to_graph.function_def_to_graph(fdef, input_shapes)
 244   func_graph._while = while_op
 245   return func_graph

 248 def _create_grad_func(func_graph, grads, name, while_op):
 249   """Builds and returns the gradient FuncGraph of `func_graph` and its args.
 250 
 251   The returned grad_func_graph must be called with the returned
 252   args + grad_func_graph.captures.
 253 
 254   Args:
 255     func_graph: FuncGraph for the forward body function.
 256     grads: The incoming grads for `func_graph`'s outputs.
 257     name: Name of the returned gradient function.
 258     while_op: The forward While op.
 259 
 260   Returns:
 261     2-tuple of (grad_func_graph, args).
 262   """
 263   assert len(func_graph.outputs) == len(grads)
 264 
 265   loop_counter = constant_op.constant(0.)
 266   # TODO(srbs): For nested while loops will need to lookup this value from
 267   # the accumulator of the enclosing while loop. For now use as is assuming
 268   # there is no nesting.
 269   num_iters_t = while_op.outputs[0]
 270 
 271   args = [loop_counter, num_iters_t] + grads
 272 
 273   # Note: The returned function does not have `args` in the list of
 274   # `external_captures`.
 275   grad_func_graph = function.func_graph_from_py_func(
 276       name,
 277       lambda *args: _grad_fn(func_graph, args),
 278       args, {},
 279       func_graph=_WhileBodyGradFuncGraph(name, func_graph))
 280 
 281   # Add the popped accumulators to the list of outputs.
 282   for internal_capture in grad_func_graph.internal_captures:
 283     grad_func_graph.outputs.append(
 284         grad_func_graph.popped_tensor_lists[internal_capture])
 285 
 286   return grad_func_graph, args

 289 def _grad_fn(func_graph, args):
 290   """Computes the gradient of `func_graph` in the current graph.
 291 
 292   This function builds the gradient graph of the corresponding forward-pass
 293   `func_graph` by differentiating `func_graph`'s outputs w.r.t. its inputs.
 294 
 295   Args:
 296     func_graph: function.FuncGraph. The corresponding forward-pass function.
 297     args: The input arguments. args[0] - Loop counter args[1] - Total number of
 298       iterations.
 299       args[2:] - Incoming gradients for `func_graph.outputs`.
 300 
 301   Returns:
 302     The output gradient Tensors.
 303   """
 304   xs = func_graph.inputs
 305   ys = func_graph.outputs
 306   grad_ys = args[2:]
 307 
 308   # Build the gradient graph. Note that this builds the gradient computation of
 309   # func_graph in the current graph, which requires capturing tensors from
 310   # func_graph. The captured func_graph tensors are resolved to external tensors
 311   # in _resolve_grad_inputs.
 312   # TODO(srbs): Mark GradientsHelper as public?
 313   grad_outs = gradients_impl._GradientsHelper(
 314       ys, xs, grad_ys=grad_ys, src_graph=func_graph)
 315 
 316   assert all([g is not None for g in grad_outs])
 317   counter = args[0]
 318   total_iters = args[1]
 319   return [counter + 1, total_iters] + grad_outs

 322 def _get_intermediates(func_graph):
 323   """Returns all tensors in `func_graph` that should be accumulated."""
 324   # We currently accumulate output tensors of most ops in the function and rely
 325   # on the pruning pass to get rid of the unused accumulators at runtime.
 326   # However, this can bloat the GraphDef and make debugging harder so we perform
 327   # some optimizations.
 328   #
 329   # Optimization we currently perform:
 330   # 1. We do not accumulate tensors which already have an accumulator
 331   #    in the loop body.
 332   # 2. We do not accumulate outputs of Identity nodes. When building the
 333   #    FuncGraph, we add an Identity node for each output (see
 334   #    `AutomaticControlDependencies.mark_as_return`). Accumulating outputs
 335   #    of all these nodes bloats the GraphDef quite a bit so we remove those.
 336   #    Since the gradient of an Identity node does not rely on its forward op's
 337   #    input this is safe to do.
 338   #
 339   # Other possible optimizations:
 340   # 1. Only accumulate tensors that will be required by the backward pass.
 341   #    This will require running the gradient pass and hence would increase the
 342   #    graph building time for the forward pass.
 343   # 2. Do not accumulate Const nodes created inside the loop body.
 344   # 3. Do not accumulate inputs that are passed as-is, e.g. loop invariants.
 345   # TODO(srbs): 2 and 3 may be hard optimizations for the runtime optimizer
 346   # since it requires knowledge of the while loop semantics. If so, consider
 347   # doing those here.
 348   intermediates = []
 349 
 350   for op in func_graph.get_operations():
 351     if op.type == "Identity":
 352       continue
 353     for o in op.outputs:
 354       if (o != func_graph.inputs[0] and  # Loop counter.
 355           _get_accumulator(o) is None):  # Has existing accumulator.
 356         intermediates.append(o)
 357   return intermediates

 360 def _get_accumulator(tensor):
 361   r"""Returns TensorList if any containing accumulated values of tensor.
 362 
 363   We try to find a pattern of the form:
 364 
 365      input_tl   tensor
 366         \        /
 367     (TensorListPushBack)
 368             |
 369         output_tl
 370 
 371   which satisfies the following conditions:
 372 
 373   1. input_tl must be in tensor.graph.inputs.
 374   2. output_tl or Identity(output_tl) must be in tensor.graph.outputs.
 375   3. tensor.graph.input_index(input_tl) == tensor.graph.output_index(output_t).
 376 
 377   output_tl or Identity(output_tl) (whichever is in tensor.graph.outputs) is
 378   returned if such a pattern is found else None is returned.
 379 
 380   Args:
 381     tensor: The Tensor to be accumulated.
 382 
 383   Returns:
 384     A variant tensor in the same graph as `tensor` or None if no accumulator is
 385     found.
 386   """
 387   assert isinstance(tensor.graph, function.FuncGraph)
 388 
 389   def get_func_graph_output(t):
 390     """Returns t or Identity(t) whichever exists in graph outputs else None."""
 391     if t in tensor.graph.outputs:
 392       return t
 393     # tf.defun adds an Identity for each output, check whether that is the case.
 394     identity_op = t.consumers()[0]
 395     if (identity_op.type == "Identity" and
 396         identity_op.outputs[0] in tensor.graph.outputs):
 397       return identity_op.outputs[0]
 398     return None
 399 
 400   for consumer in tensor.consumers():
 401     # Find the consumer that is a TensorListPushBack node whose TensorList input
 402     # is in the list of function inputs.
 403     if (consumer.type != "TensorListPushBack" or
 404         consumer.inputs[0] not in tensor.graph.inputs):
 405       continue
 406 
 407     output = get_func_graph_output(consumer.outputs[0])
 408     if output is None:
 409       # The TensorList output of `consumer` is not in the list of function
 410       # outputs.
 411       continue
 412 
 413     accum_input_idx = tensor.graph.inputs.index(consumer.inputs[0])
 414     accum_output_idx = tensor.graph.outputs.index(output)
 415     if accum_input_idx == accum_output_idx:
 416       return output
 417   return None

 434 class _WhileBodyGradFuncGraph(function.FuncGraph):
 435   """FuncGraph for the gradient function of the body of a While op.
 436 
 437   Contains the logic for capturing the tensors from the body of the forward
 438   While op which is as follows:
 439   1. Find the accumulator for that tensor.
 440   2. Capture the forward While op output tensor corresponding to the
 441      accumulator in this FuncGraph.
 442   3. Pop a value from the captured placeholder and use it as the captured value
 443      for the forward pass tensor.
 444 
 445   This only allows capturing tensors in the forward graph. A ValueError is
 446   raised if an attempt is made to capture a tensor not in the forward graph.
 447   To manually capture capture a tensor that is not in the forward graph, call
 448   `capture` with `whitelisted=True`.
 449 
 450   Note: The `captures` dict does not contain the forward tensor since it is not
 451   directly captured. It contains the accumulator corresponding to this forward
 452   tensor.
 453 
 454   Attributes:
 455     popped_tensor_lists: Dict from the captured accumulator placeholder to the
 456       TensorList obtained after popping the intermediate tensor from it. The
 457       values of this dict need to be added to the list of outputs.
 458   """
 459 
 460   def __init__(self, name, forward_graph):
 461     super(_WhileBodyGradFuncGraph, self).__init__(name)
 462     self.popped_tensor_lists = {}
 463     # FuncGraph for the body of the forward While op.
 464     self._forward_graph = forward_graph
 465     # Dict from forward intermediate tensor to the corresponding "popped" tensor
 466     # in this graph.
 467     self._indirect_captures = {}
 468     # Dict from forward graph tensor to the While op output corresponding to its
 469     # accumulator.
 470     self._tensor_to_accumulator = {}
 471 
 472   def capture(self, tensor, name=None, whitelisted=False):
 473     """Selectively captures external tensors.
 474 
 475     If `whitelisted` is False only allows capturing tensors in the
 476     `_forward_graph`.
 477 
 478     Args:
 479       tensor: Tensor. May be from this FuncGraph or a different graph.
 480       name: Optional name if a placeholder is created.
 481       whitelisted: If False (default), only allows capturing tensors from the
 482         forward graph.
 483 
 484     Returns:
 485       The placeholder in this graph for the tensor.
 486 
 487     Raises:
 488       ValueError: If attempting to capture an external tensor not in the forward
 489         graph with `whitelisted` set to False.
 490     """
 491     if (not whitelisted and tensor.graph is not self and
 492         tensor.graph != self._forward_graph):
 493       raise ValueError("Attempting to capture tensor", str(tensor),
 494                        " which is not in the forward graph but in ",
 495                        _graph_name(tensor.graph), ".")
 496     return super(_WhileBodyGradFuncGraph, self).capture(tensor, name)
 497 
 498   def _capture_helper(self, tensor, name):
 499     if tensor.graph is not self._forward_graph:
 500       return super(_WhileBodyGradFuncGraph, self)._capture_helper(tensor, name)
 501 
 502     captured_tensor = self._indirect_captures.get(tensor)
 503     if captured_tensor is not None:
 504       # For GradientTape housekeeping.
 505       assert self._tensor_to_accumulator[tensor] in self.captures
 506       super(_WhileBodyGradFuncGraph, self)._capture_helper(
 507           self._tensor_to_accumulator[tensor], name)
 508       return captured_tensor
 509 
 510     assert tensor not in self._tensor_to_accumulator
 511 
 512     accumulator = None
 513 
 514     # Find the TensorList that was used to accumulate the tensors of this
 515     # intermediate tensor.
 516     accumulator = _get_accumulator(tensor)
 517     if accumulator is None:
 518       raise ValueError("Reference to un-accumulated intermediate tensor: ",
 519                        tensor.name)
 520     assert accumulator.graph == self._forward_graph
 521     # Get the While op output corresponding to the accumulator.
 522     accumulator = self._forward_graph._while.outputs[self._forward_graph.outputs
 523                                                      .index(accumulator)]
 524 
 525     assert accumulator.graph == self._forward_graph.outer_graph
 526     self._tensor_to_accumulator[tensor] = accumulator
 527 
 528     # Capture the `accumulator`.
 529     accumulator_ph = super(_WhileBodyGradFuncGraph, self)._capture_helper(
 530         accumulator, name)
 531     new_tensor_list, captured_tensor = list_ops.tensor_list_pop_back(
 532         accumulator_ph, element_dtype=tensor.dtype)
 533     self._indirect_captures[tensor] = captured_tensor
 534     self.popped_tensor_lists[accumulator_ph] = new_tensor_list
 535     return captured_tensor

 543 # TODO(srbs): Move to common utils for cond_v2 and while_v2.
 544 def _maybe_set_lowering_attr(op):
 545   """Sets the flag to enable lowering on the `While` op if necessary.
 546 
 547   Lowering allows while_v2 to avoid some of the limitations of Functions,
 548   allowing users to specify devices & colocation inside of while_v2
 549   branches, and enabling non-strict evaluation & partial pruning of while_v2
 550   branches. This brings while_v2 closer to feature parity with
 551   tf.while_loop.
 552 
 553   However, we do not lower `While` in the XLA context because it is easier
 554   for XLA to apply its own optimizations when dealing with un-lowered
 555   `While` operators than with low-level control flow primitives.
 556 
 557   Args:
 558     op: The While op.
 559   """
 560   if not control_flow_util.IsInXLAContext(op):
 561     # pylint: disable=protected-access
 562     op._set_attr("_lower_using_switch_merge", attr_value_pb2.AttrValue(b=True))
 563     # pylint: enable=protected-access

 566 def _get_tensor_convertible_shape(shape):
 567   assert isinstance(shape, tensor_shape.TensorShape)
 568   if shape.is_fully_defined():
 569     return shape
 570   if not shape:  # Unknown shape.
 571     return -1
 572   # Partially defined shape.
 573   shape_list = shape.as_list()
 574   shape_list = [s if s is not None else -1 for s in shape_list]
 575   return ops.convert_to_tensor(shape_list)
