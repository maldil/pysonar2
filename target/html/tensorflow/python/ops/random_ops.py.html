  38 def _ShapeTensor(shape):
  39   """Convert to an int32 or int64 tensor, defaulting to int32 if empty."""
  40   if isinstance(shape, (tuple, list)) and not shape:
  41     dtype = dtypes.int32
  42   else:
  43     dtype = None
  44   return ops.convert_to_tensor(shape, dtype=dtype, name="shape")

  47 @tf_export("random.normal", "random_normal")
  48 def random_normal(shape,
  49                   mean=0.0,
  50                   stddev=1.0,
  51                   dtype=dtypes.float32,
  52                   seed=None,
  53                   name=None):
  54   """Outputs random values from a normal distribution.
  55 
  56   Args:
  57     shape: A 1-D integer Tensor or Python array. The shape of the output tensor.
  58     mean: A 0-D Tensor or Python value of type `dtype`. The mean of the normal
  59       distribution.
  60     stddev: A 0-D Tensor or Python value of type `dtype`. The standard deviation
  61       of the normal distribution.
  62     dtype: The type of the output.
  63     seed: A Python integer. Used to create a random seed for the distribution.
  64       See
  65       `tf.set_random_seed`
  66       for behavior.
  67     name: A name for the operation (optional).
  68 
  69   Returns:
  70     A tensor of the specified shape filled with random normal values.
  71   """
  72   with ops.name_scope(name, "random_normal", [shape, mean, stddev]) as name:
  73     shape_tensor = _ShapeTensor(shape)
  74     mean_tensor = ops.convert_to_tensor(mean, dtype=dtype, name="mean")
  75     stddev_tensor = ops.convert_to_tensor(stddev, dtype=dtype, name="stddev")
  76     seed1, seed2 = random_seed.get_seed(seed)
  77     rnd = gen_random_ops.random_standard_normal(
  78         shape_tensor, dtype, seed=seed1, seed2=seed2)
  79     mul = rnd * stddev_tensor
  80     value = math_ops.add(mul, mean_tensor, name=name)
  81     return value

  87 def parameterized_truncated_normal(shape,
  88                                    means=0.0,
  89                                    stddevs=1.0,
  90                                    minvals=-2.0,
  91                                    maxvals=2.0,
  92                                    dtype=dtypes.float32,
  93                                    seed=None,
  94                                    name=None):
  95   """Outputs random values from a truncated normal distribution.
  96 
  97   The generated values follow a normal distribution with specified mean and
  98   standard deviation, except that values whose magnitude is more than 2 standard
  99   deviations from the mean are dropped and re-picked.
 100 
 101   Args:
 102     shape: A 1-D integer Tensor or Python array. The shape of the output tensor.
 103     means: A 0-D Tensor or Python value of type `dtype`. The mean of the
 104       truncated normal distribution.
 105     stddevs: A 0-D Tensor or Python value of type `dtype`. The standard
 106       deviation of the truncated normal distribution.
 107     minvals: A 0-D Tensor or Python value of type `dtype`. The minimum value of
 108       the truncated normal distribution.
 109     maxvals: A 0-D Tensor or Python value of type `dtype`. The maximum value of
 110       the truncated normal distribution.
 111     dtype: The type of the output.
 112     seed: A Python integer. Used to create a random seed for the distribution.
 113       See
 114       `tf.set_random_seed`
 115       for behavior.
 116     name: A name for the operation (optional).
 117 
 118   Returns:
 119     A tensor of the specified shape filled with random truncated normal values.
 120   """
 121   with ops.name_scope(name, "parameterized_truncated_normal",
 122                       [shape, means, stddevs, minvals, maxvals]) as name:
 123     shape_tensor = _ShapeTensor(shape)
 124     means_tensor = ops.convert_to_tensor(means, dtype=dtype, name="means")
 125     stddevs_tensor = ops.convert_to_tensor(stddevs, dtype=dtype, name="stddevs")
 126     minvals_tensor = ops.convert_to_tensor(minvals, dtype=dtype, name="minvals")
 127     maxvals_tensor = ops.convert_to_tensor(maxvals, dtype=dtype, name="maxvals")
 128     seed1, seed2 = random_seed.get_seed(seed)
 129     rnd = gen_random_ops.parameterized_truncated_normal(
 130         shape_tensor,
 131         means_tensor,
 132         stddevs_tensor,
 133         minvals_tensor,
 134         maxvals_tensor,
 135         seed=seed1,
 136         seed2=seed2)
 137     return rnd

 185 @tf_export("random.uniform", "random_uniform")
 186 def random_uniform(shape,
 187                    minval=0,
 188                    maxval=None,
 189                    dtype=dtypes.float32,
 190                    seed=None,
 191                    name=None):
 192   """Outputs random values from a uniform distribution.
 193 
 194   The generated values follow a uniform distribution in the range
 195   `[minval, maxval)`. The lower bound `minval` is included in the range, while
 196   the upper bound `maxval` is excluded.
 197 
 198   For floats, the default range is `[0, 1)`.  For ints, at least `maxval` must
 199   be specified explicitly.
 200 
 201   In the integer case, the random integers are slightly biased unless
 202   `maxval - minval` is an exact power of two.  The bias is small for values of
 203   `maxval - minval` significantly smaller than the range of the output (either
 204   `2**32` or `2**64`).
 205 
 206   Args:
 207     shape: A 1-D integer Tensor or Python array. The shape of the output tensor.
 208     minval: A 0-D Tensor or Python value of type `dtype`. The lower bound on the
 209       range of random values to generate.  Defaults to 0.
 210     maxval: A 0-D Tensor or Python value of type `dtype`. The upper bound on
 211       the range of random values to generate.  Defaults to 1 if `dtype` is
 212       floating point.
 213     dtype: The type of the output: `float16`, `float32`, `float64`, `int32`,
 214       or `int64`.
 215     seed: A Python integer. Used to create a random seed for the distribution.
 216       See `tf.set_random_seed`
 217       for behavior.
 218     name: A name for the operation (optional).
 219 
 220   Returns:
 221     A tensor of the specified shape filled with random uniform values.
 222 
 223   Raises:
 224     ValueError: If `dtype` is integral and `maxval` is not specified.
 225   """
 226   dtype = dtypes.as_dtype(dtype)
 227   if dtype not in (dtypes.float16, dtypes.bfloat16, dtypes.float32,
 228                    dtypes.float64, dtypes.int32, dtypes.int64):
 229     raise ValueError("Invalid dtype %r" % dtype)
 230   if maxval is None:
 231     if dtype.is_integer:
 232       raise ValueError("Must specify maxval for integer dtype %r" % dtype)
 233     maxval = 1
 234   with ops.name_scope(name, "random_uniform", [shape, minval, maxval]) as name:
 235     shape = _ShapeTensor(shape)
 236     minval = ops.convert_to_tensor(minval, dtype=dtype, name="min")
 237     maxval = ops.convert_to_tensor(maxval, dtype=dtype, name="max")
 238     seed1, seed2 = random_seed.get_seed(seed)
 239     if dtype.is_integer:
 240       return gen_random_ops.random_uniform_int(
 241           shape, minval, maxval, seed=seed1, seed2=seed2, name=name)
 242     else:
 243       rnd = gen_random_ops.random_uniform(shape, dtype, seed=seed1, seed2=seed2)
 244       return math_ops.add(rnd * (maxval - minval), minval, name=name)

 250 @tf_export("random.shuffle", "random_shuffle")
 251 def random_shuffle(value, seed=None, name=None):
 252   """Randomly shuffles a tensor along its first dimension.
 253 
 254   The tensor is shuffled along dimension 0, such that each `value[j]` is mapped
 255   to one and only one `output[i]`. For example, a mapping that might occur for a
 256   3x2 tensor is:
 257 
 258   ```python
 259   [[1, 2],       [[5, 6],
 260    [3, 4],  ==>   [1, 2],
 261    [5, 6]]        [3, 4]]
 262   ```
 263 
 264   Args:
 265     value: A Tensor to be shuffled.
 266     seed: A Python integer. Used to create a random seed for the distribution.
 267       See
 268       `tf.set_random_seed`
 269       for behavior.
 270     name: A name for the operation (optional).
 271 
 272   Returns:
 273     A tensor of same shape and type as `value`, shuffled along its first
 274     dimension.
 275   """
 276   seed1, seed2 = random_seed.get_seed(seed)
 277   return gen_random_ops.random_shuffle(
 278       value, seed=seed1, seed2=seed2, name=name)

 281 @tf_export("image.random_crop", "random_crop")
 282 def random_crop(value, size, seed=None, name=None):
 283   """Randomly crops a tensor to a given size.
 284 
 285   Slices a shape `size` portion out of `value` at a uniformly chosen offset.
 286   Requires `value.shape >= size`.
 287 
 288   If a dimension should not be cropped, pass the full size of that dimension.
 289   For example, RGB images can be cropped with
 290   `size = [crop_height, crop_width, 3]`.
 291 
 292   Args:
 293     value: Input tensor to crop.
 294     size: 1-D tensor with size the rank of `value`.
 295     seed: Python integer. Used to create a random seed. See
 296       `tf.set_random_seed`
 297       for behavior.
 298     name: A name for this operation (optional).
 299 
 300   Returns:
 301     A cropped tensor of the same rank as `value` and shape `size`.
 302   """
 303   # TODO(shlens): Implement edge case to guarantee output size dimensions.
 304   # If size > value.shape, zero pad the result so that it always has shape
 305   # exactly size.
 306   with ops.name_scope(name, "random_crop", [value, size]) as name:
 307     value = ops.convert_to_tensor(value, name="value")
 308     size = ops.convert_to_tensor(size, dtype=dtypes.int32, name="size")
 309     shape = array_ops.shape(value)
 310     check = control_flow_ops.Assert(
 311         math_ops.reduce_all(shape >= size),
 312         ["Need value.shape >= size, got ", shape, size],
 313         summarize=1000)
 314     shape = control_flow_ops.with_dependencies([check], shape)
 315     limit = shape - size + 1
 316     offset = random_uniform(
 317         array_ops.shape(shape),
 318         dtype=size.dtype,
 319         maxval=size.dtype.max,
 320         seed=seed) % limit
 321     return array_ops.slice(value, offset, size, name=name)

 324 @tf_export("random.multinomial", "multinomial")
 325 def multinomial(logits, num_samples, seed=None, name=None, output_dtype=None):
 326   """Draws samples from a multinomial distribution.
 327 
 328   Example:
 329 
 330   ```python
 331   # samples has shape [1, 5], where each value is either 0 or 1 with equal
 332   # probability.
 333   samples = tf.multinomial(tf.log([[10., 10.]]), 5)
 334   ```
 335 
 336   Args:
 337     logits: 2-D Tensor with shape `[batch_size, num_classes]`.  Each slice
 338       `[i, :]` represents the unnormalized log-probabilities for all classes.
 339     num_samples: 0-D.  Number of independent samples to draw for each row slice.
 340     seed: A Python integer. Used to create a random seed for the distribution.
 341       See
 342       `tf.set_random_seed`
 343       for behavior.
 344     name: Optional name for the operation.
 345     output_dtype: integer type to use for the output. Defaults to int64.
 346 
 347   Returns:
 348     The drawn samples of shape `[batch_size, num_samples]`.
 349   """
 350   with ops.name_scope(name, "multinomial", [logits]):
 351     logits = ops.convert_to_tensor(logits, name="logits")
 352     seed1, seed2 = random_seed.get_seed(seed)
 353     return gen_random_ops.multinomial(
 354         logits, num_samples, seed=seed1, seed2=seed2, output_dtype=output_dtype)

 360 @tf_export("random.gamma", "random_gamma")
 361 @deprecation.deprecated_endpoints("random_gamma")
 362 def random_gamma(shape,
 363                  alpha,
 364                  beta=None,
 365                  dtype=dtypes.float32,
 366                  seed=None,
 367                  name=None):
 368   """Draws `shape` samples from each of the given Gamma distribution(s).
 369 
 370   `alpha` is the shape parameter describing the distribution(s), and `beta` is
 371   the inverse scale parameter(s).
 372 
 373   Note: Because internal calculations are done using `float64` and casting has
 374   `floor` semantics, we must manually map zero outcomes to the smallest
 375   possible positive floating-point value, i.e., `np.finfo(dtype).tiny`.  This
 376   means that `np.finfo(dtype).tiny` occurs more frequently than it otherwise
 377   should.  This bias can only happen for small values of `alpha`, i.e.,
 378   `alpha << 1` or large values of `beta`, i.e., `beta >> 1`.
 379 
 380   The samples are differentiable w.r.t. alpha and beta.
 381   The derivatives are computed using the approach described in the paper
 382 
 383   [Michael Figurnov, Shakir Mohamed, Andriy Mnih.
 384   Implicit Reparameterization Gradients, 2018](https://arxiv.org/abs/1805.08498)
 385 
 386   Example:
 387 
 388   ```python
 389   samples = tf.random_gamma([10], [0.5, 1.5])
 390   # samples has shape [10, 2], where each slice [:, 0] and [:, 1] represents
 391   # the samples drawn from each distribution
 392 
 393   samples = tf.random_gamma([7, 5], [0.5, 1.5])
 394   # samples has shape [7, 5, 2], where each slice [:, :, 0] and [:, :, 1]
 395   # represents the 7x5 samples drawn from each of the two distributions
 396 
 397   alpha = tf.constant([[1.],[3.],[5.]])
 398   beta = tf.constant([[3., 4.]])
 399   samples = tf.random_gamma([30], alpha=alpha, beta=beta)
 400   # samples has shape [30, 3, 2], with 30 samples each of 3x2 distributions.
 401 
 402   loss = tf.reduce_mean(tf.square(samples))
 403   dloss_dalpha, dloss_dbeta = tf.gradients(loss, [alpha, beta])
 404   # unbiased stochastic derivatives of the loss function
 405   alpha.shape == dloss_dalpha.shape  # True
 406   beta.shape == dloss_dbeta.shape  # True
 407   ```
 408 
 409   Args:
 410     shape: A 1-D integer Tensor or Python array. The shape of the output samples
 411       to be drawn per alpha/beta-parameterized distribution.
 412     alpha: A Tensor or Python value or N-D array of type `dtype`. `alpha`
 413       provides the shape parameter(s) describing the gamma distribution(s) to
 414       sample. Must be broadcastable with `beta`.
 415     beta: A Tensor or Python value or N-D array of type `dtype`. Defaults to 1.
 416       `beta` provides the inverse scale parameter(s) of the gamma
 417       distribution(s) to sample. Must be broadcastable with `alpha`.
 418     dtype: The type of alpha, beta, and the output: `float16`, `float32`, or
 419       `float64`.
 420     seed: A Python integer. Used to create a random seed for the distributions.
 421       See
 422       `tf.set_random_seed`
 423       for behavior.
 424     name: Optional name for the operation.
 425 
 426   Returns:
 427     samples: a `Tensor` of shape
 428       `tf.concat([shape, tf.shape(alpha + beta)], axis=0)` with values of type
 429       `dtype`.
 430   """
 431   with ops.name_scope(name, "random_gamma", [shape, alpha, beta]):
 432     shape = ops.convert_to_tensor(shape, name="shape", dtype=dtypes.int32)
 433     alpha = ops.convert_to_tensor(alpha, name="alpha", dtype=dtype)
 434     beta = ops.convert_to_tensor(
 435         beta if beta is not None else 1, name="beta", dtype=dtype)
 436     alpha_broadcast = alpha + array_ops.zeros_like(beta)
 437     seed1, seed2 = random_seed.get_seed(seed)
 438     return math_ops.maximum(
 439         np.finfo(dtype.as_numpy_dtype).tiny,
 440         gen_random_ops.random_gamma(
 441             shape, alpha_broadcast, seed=seed1, seed2=seed2) / beta)
 442

 444 @tf_export("random.poisson", "random_poisson")
 445 @deprecation.deprecated_endpoints("random_poisson")
 446 def random_poisson(lam, shape, dtype=dtypes.float32, seed=None, name=None):
 447   """Draws `shape` samples from each of the given Poisson distribution(s).
 448 
 449   `lam` is the rate parameter describing the distribution(s).
 450 
 451   Example:
 452 
 453   ```python
 454   samples = tf.random_poisson([0.5, 1.5], [10])
 455   # samples has shape [10, 2], where each slice [:, 0] and [:, 1] represents
 456   # the samples drawn from each distribution
 457 
 458   samples = tf.random_poisson([12.2, 3.3], [7, 5])
 459   # samples has shape [7, 5, 2], where each slice [:, :, 0] and [:, :, 1]
 460   # represents the 7x5 samples drawn from each of the two distributions
 461   ```
 462 
 463   Args:
 464     lam: A Tensor or Python value or N-D array of type `dtype`.
 465       `lam` provides the rate parameter(s) describing the poisson
 466       distribution(s) to sample.
 467     shape: A 1-D integer Tensor or Python array. The shape of the output samples
 468       to be drawn per "rate"-parameterized distribution.
 469     dtype: The type of the output: `float16`, `float32`, `float64`, `int32` or
 470       `int64`.
 471     seed: A Python integer. Used to create a random seed for the distributions.
 472       See
 473       `tf.set_random_seed`
 474       for behavior.
 475     name: Optional name for the operation.
 476 
 477   Returns:
 478     samples: a `Tensor` of shape `tf.concat([shape, tf.shape(lam)], axis=0)`
 479       with values of type `dtype`.
 480   """
 481   with ops.name_scope(name, "random_poisson", [lam, shape]):
 482     shape = ops.convert_to_tensor(shape, name="shape", dtype=dtypes.int32)
 483     seed1, seed2 = random_seed.get_seed(seed)
 484     return gen_random_ops.random_poisson_v2(
 485         shape, lam, dtype=dtype, seed=seed1, seed2=seed2)
