  45 class PruningMode(object):
  46   """Class for working with Pruning modes."""
  47   NO_PRUNING, PRE_PRUNING, POST_PRUNING = range(0, 3)
  48 
  49   _map = {'none': NO_PRUNING, 'pre': PRE_PRUNING, 'post': POST_PRUNING}
  50 
  51   @classmethod
  52   def from_str(cls, mode):
  53     if mode in cls._map:
  54       return cls._map[mode]
  55     else:
  56       raise ValueError('pruning_mode mode must be one of: {}'.format(', '.join(
  57           sorted(cls._map))))

  60 class _TreeEnsembleSavable(saver.BaseSaverBuilder.SaveableObject):
  61   """SaveableObject implementation for TreeEnsemble."""
  62 
  63   def __init__(self, resource_handle, create_op, name):
  64     """Creates a _TreeEnsembleSavable object.
  65 
  66     Args:
  67       resource_handle: handle to the decision tree ensemble variable.
  68       create_op: the op to initialize the variable.
  69       name: the name to save the tree ensemble variable under.
  70     """
  71     stamp_token, serialized = (
  72         gen_boosted_trees_ops.boosted_trees_serialize_ensemble(resource_handle))
  73     # slice_spec is useful for saving a slice from a variable.
  74     # It's not meaningful the tree ensemble variable. So we just pass an empty
  75     # value.
  76     slice_spec = ''
  77     specs = [
  78         saver.BaseSaverBuilder.SaveSpec(stamp_token, slice_spec,
  79                                         name + '_stamp'),
  80         saver.BaseSaverBuilder.SaveSpec(serialized, slice_spec,
  81                                         name + '_serialized'),
  82     ]
  83     super(_TreeEnsembleSavable, self).__init__(resource_handle, specs, name)
  84     self._resource_handle = resource_handle
  85     self._create_op = create_op
  86 
  87   def restore(self, restored_tensors, unused_restored_shapes):
  88     """Restores the associated tree ensemble from 'restored_tensors'.
  89 
  90     Args:
  91       restored_tensors: the tensors that were loaded from a checkpoint.
  92       unused_restored_shapes: the shapes this object should conform to after
  93         restore. Not meaningful for trees.
  94 
  95     Returns:
  96       The operation that restores the state of the tree ensemble variable.
  97     """
  98     with ops.control_dependencies([self._create_op]):
  99       return gen_boosted_trees_ops.boosted_trees_deserialize_ensemble(
 100           self._resource_handle,
 101           stamp_token=restored_tensors[0],
 102           tree_ensemble_serialized=restored_tensors[1])

 105 class TreeEnsemble(object):
 106   """Creates TreeEnsemble resource."""
 107 
 108   def __init__(self, name, stamp_token=0, is_local=False, serialized_proto=''):
 109     with ops.name_scope(name, 'TreeEnsemble') as name:
 110       self._resource_handle = (
 111           gen_boosted_trees_ops.boosted_trees_ensemble_resource_handle_op(
 112               container='', shared_name=name, name=name))
 113       create_op = gen_boosted_trees_ops.boosted_trees_create_ensemble(
 114           self.resource_handle,
 115           stamp_token,
 116           tree_ensemble_serialized=serialized_proto)
 117       is_initialized_op = (
 118           gen_boosted_trees_ops.is_boosted_trees_ensemble_initialized(
 119               self._resource_handle))
 120       # Adds the variable to the savable list.
 121       if not is_local:
 122         saveable = _TreeEnsembleSavable(self.resource_handle, create_op,
 123                                         self.resource_handle.name)
 124         ops.add_to_collection(ops.GraphKeys.SAVEABLE_OBJECTS, saveable)
 125       resources.register_resource(
 126           self.resource_handle,
 127           create_op,
 128           is_initialized_op,
 129           is_shared=not is_local)
 130 
 131   @property
 132   def resource_handle(self):
 133     return self._resource_handle
 134 
 135   def get_stamp_token(self):
 136     """Returns the current stamp token of the resource."""
 137     stamp_token, _, _, _, _ = (
 138         gen_boosted_trees_ops.boosted_trees_get_ensemble_states(
 139             self.resource_handle))
 140     return stamp_token
 141 
 142   def get_states(self):
 143     """Returns states of the tree ensemble.
 144 
 145     Returns:
 146       stamp_token, num_trees, num_finalized_trees, num_attempted_layers and
 147       range of the nodes in the latest layer.
 148     """
 149     (stamp_token, num_trees, num_finalized_trees, num_attempted_layers,
 150      nodes_range) = (
 151          gen_boosted_trees_ops.boosted_trees_get_ensemble_states(
 152              self.resource_handle))
 153     # Use identity to give names.
 154     return (array_ops.identity(stamp_token, name='stamp_token'),
 155             array_ops.identity(num_trees, name='num_trees'),
 156             array_ops.identity(num_finalized_trees, name='num_finalized_trees'),
 157             array_ops.identity(
 158                 num_attempted_layers, name='num_attempted_layers'),
 159             array_ops.identity(nodes_range, name='last_layer_nodes_range'))
 160 
 161   def serialize(self):
 162     """Serializes the ensemble into proto and returns the serialized proto.
 163 
 164     Returns:
 165       stamp_token: int64 scalar Tensor to denote the stamp of the resource.
 166       serialized_proto: string scalar Tensor of the serialized proto.
 167     """
 168     return gen_boosted_trees_ops.boosted_trees_serialize_ensemble(
 169         self.resource_handle)
 170 
 171   def deserialize(self, stamp_token, serialized_proto):
 172     """Deserialize the input proto and resets the ensemble from it.
 173 
 174     Args:
 175       stamp_token: int64 scalar Tensor to denote the stamp of the resource.
 176       serialized_proto: string scalar Tensor of the serialized proto.
 177 
 178     Returns:
 179       Operation (for dependencies).
 180     """
 181     return gen_boosted_trees_ops.boosted_trees_deserialize_ensemble(
 182         self.resource_handle, stamp_token, serialized_proto)
