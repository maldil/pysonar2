  36 @tf_export("custom_gradient")
  37 def custom_gradient(f):
  38   """Decorator to define a function with a custom gradient.
  39 
  40   This decorator allows fine grained control over the gradients of a sequence
  41   for operations.  This may be useful for multiple reasons, including providing
  42   a more efficient or numerically stable gradient for a sequence of operations.
  43 
  44   For example, consider the following function that commonly occurs in the
  45   computation of cross entropy and log likelihoods:
  46 
  47   ```python
  48   def log1pexp(x):
  49     return tf.log(1 + tf.exp(x))
  50   ```
  51 
  52   Due to numerical instability, the gradient this function evaluated at x=100 is
  53   NaN.  For example:
  54 
  55   ```python
  56   x = tf.constant(100.)
  57   y = log1pexp(x)
  58   dy = tf.gradients(y, x) # Will be NaN when evaluated.
  59   ```
  60 
  61   The gradient expression can be analytically simplified to provide numerical
  62   stability:
  63 
  64   ```python
  65   @tf.custom_gradient
  66   def log1pexp(x):
  67     e = tf.exp(x)
  68     def grad(dy):
  69       return dy * (1 - 1 / (1 + e))
  70     return tf.log(1 + e), grad
  71   ```
  72 
  73   With this definition, the gradient at x=100 will be correctly evaluated as
  74   1.0.
  75 
  76   See also `tf.RegisterGradient` which registers a gradient function for a
  77   primitive TensorFlow operation. `tf.custom_gradient` on the other hand allows
  78   for fine grained control over the gradient computation of a sequence of
  79   operations.
  80 
  81   Note that if the decorated function uses `Variable`s, the enclosing variable
  82   scope must be using `ResourceVariable`s.
  83 
  84   Args:
  85     f: function `f(*x)` that returns a tuple `(y, grad_fn)` where:
  86        - `x` is a sequence of `Tensor` inputs to the function.
  87        - `y` is a `Tensor` or sequence of `Tensor` outputs of applying
  88          TensorFlow operations in `f` to `x`.
  89        - `grad_fn` is a function with the signature `g(*grad_ys)` which returns
  90          a list of `Tensor`s - the derivatives of `Tensor`s in `y` with respect
  91          to the `Tensor`s in `x`.  `grad_ys` is a `Tensor` or sequence of
  92          `Tensor`s the same size as `y` holding the initial value gradients for
  93          each `Tensor` in `y`. If `f` uses `Variable`s (that are not part of the
  94          inputs), i.e. through `get_variable`, then `grad_fn` should have
  95          signature `g(*grad_ys, variables=None)`, where `variables` is a list of
  96          the `Variable`s, and return a 2-tuple `(grad_xs, grad_vars)`, where
  97          `grad_xs` is the same as above, and `grad_vars` is a `list<Tensor>`
  98          with the derivatives of `Tensor`s in `y` with respect to the variables
  99          (that is, grad_vars has one Tensor per variable in variables).
 100 
 101   Returns:
 102     A function `h(x)` which returns the same value as `f(x)[0]` and whose
 103     gradient (as calculated by `tf.gradients`) is determined by `f(x)[1]`.
 104   """
 105 
 106   def decorated(*args, **kwargs):
 107     """Decorated function with custom gradient."""
 108     if context.executing_eagerly():
 109       return _eager_mode_decorator(f, *args, **kwargs)
 110     else:
 111       return _graph_mode_decorator(f, *args, **kwargs)
 112 
 113   return tf_decorator.make_decorator(f, decorated)

 116 def _graph_mode_decorator(f, *args, **kwargs):
 117   """Implement custom gradient decorator for graph mode."""
 118   # TODO(rsepassi): Add support for kwargs
 119   if kwargs:
 120     raise ValueError(
 121         "The custom_gradient decorator currently supports keywords "
 122         "arguments only when eager execution is enabled.")
 123   name = "CustomGradient-%s" % ops.uid()
 124   args = [ops.convert_to_tensor(x) for x in args]
 125 
 126   # Checking global and local variables attempts to ensure that no non-resource
 127   # Variables are added to the graph.
 128   current_var_scope = variable_scope.get_variable_scope()
 129   before_vars = set(current_var_scope.global_variables() +
 130                     current_var_scope.local_variables())
 131   with backprop.GradientTape() as tape:
 132     result, grad_fn = f(*args)
 133   after_vars = set(current_var_scope.global_variables() +
 134                    current_var_scope.local_variables())
 135   new_vars = after_vars - before_vars
 136   for v in new_vars:
 137     if not isinstance(v, resource_variable_ops.ResourceVariable):
 138       raise TypeError(
 139           "All variables used by a function wrapped with @custom_gradient must "
 140           "be `ResourceVariable`s. Ensure that no `variable_scope` is created "
 141           "with `use_resource=False`.")
 142   # The variables that grad_fn needs to return gradients for are the set of
 143   # variables used that are *not* part of the inputs.
 144   variables = list(set(tape.watched_variables()) - set(args))
 145   grad_argspec = tf_inspect.getfullargspec(grad_fn)
 146   variables_in_signature = ("variables" in grad_argspec.args or
 147                             grad_argspec.varkw)
 148   if variables and not variables_in_signature:
 149     raise TypeError("If using @custom_gradient with a function that "
 150                     "uses variables, then grad_fn must accept a keyword "
 151                     "argument 'variables'.")
 152   if variables_in_signature and not variables:
 153     # User seems to intend to use variables but none were captured.
 154     if not variable_scope.get_variable_scope().use_resource:
 155       raise TypeError("If using @custom_gradient with a function that "
 156                       "uses variables, the enclosing variable scope must "
 157                       "have use_resource=True.")
 158     else:
 159       logging.warn("@custom_gradient grad_fn has 'variables' in signature, but "
 160                    "no ResourceVariables were used on the forward pass.")
 161   flat_result = nest.flatten(result)
 162   all_tensors = flat_result + args + variables
 163 
 164   @ops.RegisterGradient(name)
 165   def internal_grad_fn(unused_op, *result_grads):  # pylint: disable=unused-variable
 166     """Custom grad fn wrapper."""
 167     result_grads = result_grads[:len(flat_result)]
 168     if variables:
 169       input_grads, variable_grads = grad_fn(*result_grads, variables=variables)
 170       if len(variable_grads) != len(variables):
 171         raise ValueError("Must return gradient for each variable from "
 172                          "@custom_gradient grad_fn.")
 173     else:
 174       input_grads = grad_fn(*result_grads)
 175       variable_grads = []
 176 
 177     # Need to return one value per input to the IdentityN, so pad the
 178     # gradients of the inputs of the custom_gradient function with the
 179     # gradients of the outputs as well.
 180     input_grads = nest.flatten(input_grads)
 181     return ([None] * len(flat_result)) + input_grads + variable_grads
 182 
 183   with ops.get_default_graph().gradient_override_map({"IdentityN": name}):
 184     all_tensors = array_ops.identity_n(all_tensors)
 185   return nest.pack_sequence_as(
 186       structure=result, flat_sequence=all_tensors[:len(flat_result)])
 187

 189 def _eager_mode_decorator(f, *args, **kwargs):
 190   """Implement custom gradient decorator for eager mode."""
 191   with backprop.GradientTape() as tape:
 192     result, grad_fn = f(*args, **kwargs)
 193   all_inputs = list(args) + list(kwargs.values())
 194   # The variables that grad_fn needs to return gradients for are the set of
 195   # variables used that are *not* part of the inputs.
 196   variables = [v for v in set(tape.watched_variables()) if v not in all_inputs]
 197   grad_argspec = tf_inspect.getfullargspec(grad_fn)
 198   if (variables and ("variables" not in grad_argspec.args) and
 199       not grad_argspec.varkw):
 200     raise TypeError("If using @custom_gradient with a function that "
 201                     "uses variables, then grad_fn must accept a keyword "
 202                     "argument 'variables'.")
 203   flat_result = nest.flatten(result)
 204   # TODO(apassos) consider removing the identity below.
 205   flat_result = [gen_array_ops.identity(x) for x in flat_result]
 206 
 207   def actual_grad_fn(*result_grads):
 208     """Custom grad fn wrapper."""
 209     if variables:
 210       input_grads, variable_grads = grad_fn(*result_grads, variables=variables)
 211       if len(variable_grads) != len(variables):
 212         raise ValueError("Must return gradient for each variable from "
 213                          "@custom_gradient grad_fn.")
 214     else:
 215       input_grads = grad_fn(*result_grads)
 216       variable_grads = []
 217     return nest.flatten(input_grads) + variable_grads
 218 
 219   input_tensors = [ops.convert_to_tensor(x) for x
 220                    in list(args) + list(variables)]
 221   tape_lib.record_operation(f.__name__, flat_result, input_tensors,
 222                             actual_grad_fn)
 223   flat_result = list(flat_result)
 224   return nest.pack_sequence_as(result, flat_result)
