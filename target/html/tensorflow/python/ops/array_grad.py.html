  46 def _ConcatGradHelper(op, grad, start_value_index, end_value_index, dim_index):
  47   """Gradient for concat op.
  48 
  49   Args:
  50     op: An operation.
  51     grad: `Tensor` or `IndexedSlices` representing the gradients with respect
  52       to each output of the op.
  53     start_value_index: An integer index of the first value in the op.inputs.
  54     end_value_index: An integer index of the last value in the op.inputs.
  55     dim_index: An interger index of concat_dim or axis parameter in op.inputs.
  56 
  57   Returns:
  58     Tensors representing the partial gradients with respect to each input
  59     of the op.
  60 
  61   Raises:
  62     ValueError: if concat_dim/axis is not statically known.
  63   """
  64 
  65   def _CreateDenseMaskAndBegin(sizes, concat_dim):
  66     """Create variables for iteratively slicing a dense gradients tensor."""
  67     # Since shape is 1-D, shape_of_shape = [rank-of-inputs]
  68     shape_of_shape = array_ops.shape(sizes[0])
  69     # Make a vector of length equal to the input's dimensions,
  70     # with 0's everywhere and 1 in the concat dim position.
  71     # Note: Can't use sparse_to_dense since it isn't GPU-capable (for now)
  72     mask = array_ops.concat([
  73         array_ops.fill(array_ops.expand_dims(concat_dim, 0), 0), [1],
  74         array_ops.fill(shape_of_shape - concat_dim - 1, 0)
  75     ], 0)
  76     begin = array_ops.fill(shape_of_shape, 0)
  77     return mask, begin
  78 
  79   def _ExtractInputShapes(inputs):
  80     """Extract the shapes of a set of input tensors."""
  81     if context.executing_eagerly():
  82       return array_ops.shape_n(inputs)
  83     sizes = []
  84     fully_known = True
  85     for x in inputs:
  86       input_shape = array_ops.shape(x)
  87       if not isinstance(input_shape,
  88                         ops.Tensor) or input_shape.op.type != "Const":
  89         fully_known = False
  90         break
  91       sizes.append(input_shape)
  92 
  93     if fully_known:
  94       return sizes
  95     else:
  96       return array_ops.shape_n(inputs)
  97 
  98   # Degenerate concatenation, just return grad.
  99   if len(op.inputs) == 2:
 100     return grad + [None] if end_value_index <= dim_index else [None] + grad
 101 
 102   concat_dim = op.inputs[dim_index]
 103   input_values = op.inputs[start_value_index:end_value_index]
 104 
 105   out_grads = []
 106   if isinstance(grad, ops.Tensor):
 107     if context.executing_eagerly():
 108       # Using mod here for convenience since concat_dim is already verified
 109       # in concat implementation to be within the allowed [-rank, rank) range.
 110       non_neg_concat_dim = (
 111           concat_dim._numpy().item(0) % input_values[0]._rank())  # pylint: disable=protected-access
 112       # All inputs are guaranteed to be EagerTensors in eager mode
 113       sizes = pywrap_tensorflow.TFE_Py_TensorShapeSlice(input_values,
 114                                                         non_neg_concat_dim)
 115       out_grads = array_ops.split(grad, sizes, non_neg_concat_dim)
 116     else:
 117       if constant_op.is_constant(concat_dim):
 118         # If concat_dim is a constant defined in a different context,
 119         # then we duplicate it in the current context to avoid passing it
 120         # through an Enter node.
 121         # This is a small optimization in general, but it is required when
 122         # compiling with XLA, as XLA needs the concat input to be folded into a
 123         # constant.
 124         grad_context = control_flow_util.GetOutputContext(grad.op)
 125         dim_context = control_flow_util.GetOutputContext(concat_dim.op)
 126         if dim_context != grad_context:
 127           value = tensor_util.constant_value(concat_dim)
 128           concat_dim = constant_op.constant(value=value, dtype=concat_dim.dtype)
 129 
 130       # Using mod here for convenience since concat_dim is already verified
 131       # in concat implementation to be within the allowed [-rank, rank) range.
 132       non_neg_concat_dim = concat_dim % array_ops.rank(input_values[0])
 133 
 134       # Get the inputs' tensor shapes
 135       sizes = _ExtractInputShapes(input_values)
 136       # The magic number of 16 was found through benchmarking a range of sizes
 137       # on CPUs and a Maxwell TitanX.  A speedup was seen in a large majority of
 138       # cases when switching implementations at N=16, but it is possible that
 139       # there will be a small number of performance regressions.
 140       if len(sizes) > 16:
 141         # extract the size of each input along the concat dimension
 142         sizes = array_ops.squeeze(
 143             array_ops.slice(
 144                 array_ops.stack(sizes, axis=1), [non_neg_concat_dim, 0],
 145                 [1, -1]))
 146         out_grads = array_ops.split(grad, sizes, non_neg_concat_dim)
 147       else:
 148         offset = gen_array_ops.concat_offset(non_neg_concat_dim, sizes)
 149         for (begin, size) in zip(offset, sizes):
 150           out_grads.append(array_ops.slice(grad, begin, size))
 151   elif isinstance(grad, ops.IndexedSlices):
 152     # Using mod here for convenience since concat_dim is already verified
 153     # in concat implementation to be within the allowed [-rank, rank) range.
 154     non_neg_concat_dim = concat_dim % array_ops.rank(input_values[0])
 155     concat_dim_static = tensor_util.constant_value(concat_dim)
 156     if concat_dim_static is None:
 157       raise ValueError("Can only compute IndexedSlices gradient with "
 158                        "statically-known concat_dim")
 159     if concat_dim_static < 0:
 160       rank = tensor_util.constant_value(array_ops.rank(input_values[0]))
 161       if rank is None:
 162         raise ValueError("Can only compute IndexedSlices gradient with "
 163                          "negative concat_dim when first value rank is "
 164                          "statically-known.")
 165       concat_dim_static %= rank
 166     # Get the inputs' tensor shapes
 167     sizes = [array_ops.shape(x) for x in input_values]
 168     if concat_dim_static > 0:
 169       # IndexedSlices, non_neg_concat_dim > 0. Each input gets IndexedSlices
 170       # gradients with all the indices, but with grad.values sliced accordingly.
 171       # This is like the Tensor case, except shape(grad.values)[0] is not equal
 172       # to shape(sizes[i])[0], since only a subset of the dim-0 values are
 173       # stored.
 174       mask, begin = _CreateDenseMaskAndBegin(sizes, non_neg_concat_dim)
 175       for size in sizes:
 176         new_values = array_ops.slice(
 177             grad.values, begin,
 178             array_ops.concat([[-1], array_ops.slice(size, [1], [-1])], 0))
 179         out_grads.append(ops.IndexedSlices(new_values, grad.indices, size))
 180         # Lint complains begin = begin + ...
 181         begin = math_ops.add(begin, size * mask)
 182     else:
 183       # IndexedSlices, concat_dim == 0. Each input gets IndexedSlices gradients
 184       # only for the relevant indices.
 185       start = constant_op.constant(0, dtype=grad.indices.dtype)
 186       for size in sizes:
 187         size_concat_dim = array_ops.gather(size, non_neg_concat_dim)
 188         if size_concat_dim.dtype != grad.indices.dtype:
 189           size_concat_dim = math_ops.cast(
 190               size_concat_dim, dtype=grad.indices.dtype)
 191         end = start + size_concat_dim
 192         # Compute the 1-D Tensor of indices relevant for this input.
 193         indices_to_select = array_ops.squeeze(
 194             array_ops.where(
 195                 math_ops.logical_and(grad.indices >= start,
 196                                      grad.indices < end)),
 197             axis=[1])
 198         new_indices = array_ops.gather(grad.indices, indices_to_select) - start
 199         new_values = array_ops.gather(grad.values, indices_to_select)
 200         out_grads.append(ops.IndexedSlices(new_values, new_indices, size))
 201         start = end
 202   else:
 203     raise TypeError("Expected Tensor or IndexedSlices, got %s" % type(grad))
 204 
 205   return (out_grads + [None]
 206           if end_value_index <= dim_index else [None] + out_grads)

 228 @ops.RegisterGradient("Slice")
 229 def _SliceGrad(op, grad):
 230   """Gradient for Slice op."""
 231   # Create an Nx2 padding where the first column represents how many
 232   # zeros are to be prepended for each dimension, and the second
 233   # column indicates how many zeros are appended.
 234   #
 235   # The number of zeros to append is the shape of the input
 236   # elementwise-subtracted by both the begin vector and sizes vector.
 237   #
 238   # Some more reshaping is needed to assemble this tensor with the
 239   # right dimensions.
 240   input_vec = op.inputs[0]
 241   begin_vec = op.inputs[1]
 242   input_rank = array_ops.rank(input_vec)
 243   slice_size = array_ops.shape(op.outputs[0])
 244 
 245   shape = array_ops.stack([input_rank, 1])
 246   before_pad = array_ops.reshape(begin_vec, shape)
 247   after_pad = array_ops.reshape(
 248       array_ops.shape(input_vec) - slice_size - begin_vec, shape)
 249   paddings = array_ops.concat([before_pad, after_pad], 1)
 250   return array_ops.pad(grad, paddings), None, None

 253 @ops.RegisterGradient("StridedSlice")
 254 def _StridedSliceGrad(op, grad):
 255   """Gradient for StridedSlice op."""
 256   begin = op.inputs[1]
 257   end = op.inputs[2]
 258   strides = op.inputs[3]
 259   # StridedSliceGrad requires `x`, `begin`, `end` and `strides` to be of the
 260   # same dtype so we build a shape of the same type as other args.
 261   # Note that the choice of `begin` for specifying `out_type` is arbitrary.
 262   # We could choose any of {begin|end|strides}.dtype since they are required to
 263   # be the same.
 264   x = array_ops.shape(op.inputs[0], out_type=begin.dtype)
 265 
 266   return array_ops.strided_slice_grad(
 267       x,
 268       begin,
 269       end,
 270       strides,
 271       grad,
 272       begin_mask=op.get_attr("begin_mask"),
 273       end_mask=op.get_attr("end_mask"),
 274       ellipsis_mask=op.get_attr("ellipsis_mask"),
 275       new_axis_mask=op.get_attr("new_axis_mask"),
 276       shrink_axis_mask=op.get_attr("shrink_axis_mask")), None, None, None

 340 @ops.RegisterGradient("MatrixSetDiag")
 341 def _MatrixSetDiagGrad(op, grad):
 342   """Gradient for MatrixSetDiag."""
 343   input_shape = op.inputs[0].get_shape().merge_with(grad.get_shape())
 344   diag_shape = op.inputs[1].get_shape()
 345   batch_shape = input_shape[:-2].merge_with(diag_shape[:-1])
 346   matrix_shape = input_shape[-2:]
 347   if batch_shape.is_fully_defined() and matrix_shape.is_fully_defined():
 348     diag_shape = batch_shape.as_list() + [min(matrix_shape.as_list())]
 349   else:
 350     with ops.colocate_with(grad):
 351       grad_shape = array_ops.shape(grad)
 352       grad_rank = array_ops.rank(grad)
 353       batch_shape = array_ops.slice(grad_shape, [0], [grad_rank - 2])
 354       matrix_shape = array_ops.slice(grad_shape, [grad_rank - 2], [2])
 355       min_dim = math_ops.reduce_min(matrix_shape)
 356       diag_shape = array_ops.concat([batch_shape, [min_dim]], 0)
 357   grad_input = array_ops.matrix_set_diag(grad,
 358                                          array_ops.zeros(
 359                                              diag_shape, dtype=grad.dtype))
 360   grad_diag = array_ops.matrix_diag_part(grad)
 361   return (grad_input, grad_diag)

 390 @ops.RegisterGradient("Gather")
 391 def _GatherGrad(op, grad):
 392   """Gradient for Gather op."""
 393   # params can be large, so colocate the shape calculation with it.
 394   #
 395   # params can be very large for sparse model, array_ops.shape raises
 396   # exception on the Windows platform when any dimension is larger than
 397   # int32. params_shape is not used in optimizer apply_sparse gradients,
 398   # so it's fine to convert it back to int32 regardless of truncation.
 399   params = op.inputs[0]
 400   with ops.colocate_with(params):
 401     params_shape = array_ops.shape(params, out_type=ops.dtypes.int64)
 402     params_shape = math_ops.to_int32(params_shape)
 403 
 404   # Build appropriately shaped IndexedSlices
 405   indices = op.inputs[1]
 406   size = array_ops.expand_dims(array_ops.size(indices), 0)
 407   values_shape = array_ops.concat([size, params_shape[1:]], 0)
 408   values = array_ops.reshape(grad, values_shape)
 409   indices = array_ops.reshape(indices, size)
 410   return [ops.IndexedSlices(values, indices, params_shape), None]

 413 @ops.RegisterGradient("GatherV2")
 414 def _GatherV2Grad(op, grad):
 415   """Gradient for GatherV2 op."""
 416   # params can be large, so colocate the shape calculation with it.
 417   #
 418   # params can be very large for sparse model, array_ops.shape raises
 419   # exception on the Windows platform when any dimension is larger than
 420   # int32. params_shape is not used in optimizer apply_sparse gradients,
 421   # so it's fine to convert it back to int32 regardless of truncation.
 422   params = op.inputs[0]
 423   with ops.colocate_with(params):
 424     params_shape = array_ops.shape(params, out_type=ops.dtypes.int64)
 425     params_shape = math_ops.to_int32(params_shape)
 426 
 427   indices = op.inputs[1]
 428   indices_size = array_ops.expand_dims(array_ops.size(indices), 0)
 429   axis = op.inputs[2]
 430   axis_static = tensor_util.constant_value(axis)
 431 
 432   # For axis 0 gathers, build an appropriately shaped IndexedSlices.
 433   if axis_static == 0:
 434     if context.executing_eagerly():
 435       params_tail_shape = params_shape.cpu()[1:]
 436     else:
 437       params_tail_shape = params_shape[1:]
 438     values_shape = array_ops.concat([indices_size, params_tail_shape], 0)
 439     values = array_ops.reshape(grad, values_shape)
 440     indices = array_ops.reshape(indices, indices_size)
 441     return [ops.IndexedSlices(values, indices, params_shape), None, None]
 442 
 443   outer_shape = params_shape[:axis]
 444   outer_dims = array_ops.size(outer_shape)
 445   inner_shape = params_shape[axis:][1:]
 446   inner_dims = array_ops.size(inner_shape)
 447 
 448   outer_axes_indices = math_ops.range(outer_dims)
 449   inner_axes_indices = math_ops.range(outer_dims + 1,
 450                                       outer_dims + 1 + inner_dims)
 451 
 452   values_shape = array_ops.concat([outer_shape, indices_size, inner_shape], 0)
 453   values = array_ops.reshape(grad, values_shape)
 454   indices = array_ops.reshape(indices, indices_size)
 455 
 456   # We need to sum up every slice `values[..., i, ....]` corresponding to
 457   # `params[..., indices[i], ...]`. Since `unsorted_segment_sum` does not
 458   # support an axis parameter, we transpose the gather dimension to the front,
 459   # then use `unsorted_segment_sum` to build a
 460   # [gather_axis, outer_axes, inner_axes] tensor with all the gradients
 461   # affecting each index in `gather_axis` summed up.
 462   transpose_dims = array_ops.concat(
 463       [[outer_dims], outer_axes_indices, inner_axes_indices], 0)
 464   values_transpose = array_ops.transpose(values, transpose_dims)
 465   num_segments = params_shape[axis]
 466 
 467   params_grad = math_ops.unsorted_segment_sum(values_transpose, indices,
 468                                               num_segments)
 469 
 470   # Inverts the above transpose by moving dimension 0 back to its original
 471   # position.
 472   invert_transpose_dims = array_ops.concat(
 473       [outer_axes_indices + 1, [0], inner_axes_indices], 0)
 474   params_grad = array_ops.transpose(params_grad, invert_transpose_dims)
 475   return [params_grad, None, None]
 476

 547 @ops.RegisterGradient("ConjugateTranspose")
 548 def _ConjugateTransposeGrad(op, grad):
 549   """Returns conj(unshuffle(grad))."""

 566 @ops.RegisterGradient("Tile")
 567 def _TileGrad(op, grad):
 568   """Sum reduces grad along the tiled dimensions."""
 569   input_shape = array_ops.shape(op.inputs[0])
 570   # We interleave multiples and input_shape to get split_shape,
 571   # reshape grad to split_shape, and reduce along all even
 572   # dimensions (the tiled dimensions) to get the result
 573   # with shape input_shape.  For example
 574   #   input_shape = [20, 30, 40]
 575   #   multiples = [2, 3, 4]
 576   #   split_shape = [2, 20, 3, 30, 4, 40]
 577   #   axes = [0, 2, 4]
 578   split_shape = array_ops.reshape(
 579       array_ops.transpose(array_ops.stack([op.inputs[1], input_shape])), [-1])
 580   axes = math_ops.range(0, array_ops.size(split_shape), 2)
 581   # Sum reduces grad along the first dimension for IndexedSlices
 582   if isinstance(grad, ops.IndexedSlices):
 583     grad = math_ops.unsorted_segment_sum(
 584         grad.values,
 585         math_ops.mod(grad.indices, input_shape[0]),
 586         input_shape[0])
 587     split_shape = array_ops.concat([[1], split_shape[1:]], axis=0)
 588   input_grad = math_ops.reduce_sum(array_ops.reshape(grad, split_shape), axes)
 589   # Fix shape inference
 590   if not context.executing_eagerly():
 591     input_grad.set_shape(op.inputs[0].get_shape())
 592   return [input_grad, None]

 598 def _PadGrad(op, grad):
 599   """Gradient for Pad."""
 600   # Pad introduces values around the original tensor, so the gradient function
 601   # slices the original shape out of the gradient."""
 602   x = op.inputs[0]
 603   a = op.inputs[1]  # [Rank(x), 2]
 604   # Takes a slice of a. The 1st column. [Rank(x), 1].
 605   pad_before = array_ops.slice(a, [0, 0],
 606                                array_ops.stack([array_ops.rank(x), 1]))
 607   # Make it a 1-D tensor.
 608   begin = array_ops.reshape(pad_before, [-1])
 609   sizes = array_ops.shape(x)
 610   x_grad = array_ops.slice(grad, begin, sizes)
 611   if len(op.inputs) == 3:
 612     return x_grad, None, None
 613   else:
 614     return x_grad, None

 733 @ops.RegisterGradient("ExtractImagePatches")
 734 def _ExtractImagePatchesGrad(op, grad):
 735   batch_size, rows_in, cols_in, channels = [
 736       dim.value for dim in op.inputs[0].get_shape()
 737   ]
 738   input_bhwc = array_ops.shape(op.inputs[0])
 739   batch_size = input_bhwc[0]
 740   channels = input_bhwc[3]
 741 
 742   # Create indices matrix for input tensor.
 743   # Note that 0 is preserved for padding location,
 744   # so indices for input start from 1 to 1 + rows_in * cols_in.
 745   input_indices_num = 1 + rows_in * cols_in
 746   input_idx = array_ops.reshape(math_ops.range(1, input_indices_num,
 747                                                dtype=ops.dtypes.int64),
 748                                 (1, rows_in, cols_in, 1))
 749   input_idx_patched = gen_array_ops.extract_image_patches(
 750       input_idx,
 751       op.get_attr("ksizes"),
 752       op.get_attr("strides"),
 753       op.get_attr("rates"),
 754       op.get_attr("padding"))
 755 
 756   # Create indices matrix for output tensor.
 757   _, rows_out, cols_out, _ = [dim.value for dim in op.outputs[0].get_shape()]
 758   _, ksize_r, ksize_c, _ = op.get_attr("ksizes")
 759   # Indices for output start from 0.
 760   output_indices_num = rows_out * cols_out * ksize_r * ksize_c
 761   output_idx = array_ops.reshape(math_ops.range(output_indices_num,
 762                                                 dtype=ops.dtypes.int64),
 763                                  (1, rows_out, cols_out, ksize_r * ksize_c))
 764 
 765   # Construct mapping table for indices: (input -> output).
 766   idx_matrix = array_ops.concat(
 767       [array_ops.expand_dims(input_idx_patched, axis=-1),
 768        array_ops.expand_dims(output_idx, axis=-1)],
 769       axis=-1)
 770   idx_map = array_ops.reshape(idx_matrix, (-1, 2))
 771 
 772   sp_shape = (input_indices_num, output_indices_num)
 773   sp_mat_full = sparse_tensor.SparseTensor(
 774       idx_map,
 775       array_ops.ones([output_indices_num], dtype=grad.dtype),
 776       sp_shape)
 777   # Remove all padding locations [0, :].
 778   sp_mat = sparse_ops.sparse_slice(sp_mat_full,
 779                                    (1, 0),
 780                                    (input_indices_num - 1, output_indices_num))
 781 
 782   grad_expanded = array_ops.transpose(
 783       array_ops.reshape(
 784           grad, (batch_size, rows_out, cols_out, ksize_r, ksize_c, channels)),
 785       (1, 2, 3, 4, 0, 5))
 786   grad_flat = array_ops.reshape(grad_expanded, (-1, batch_size * channels))
 787 
 788   jac = sparse_ops.sparse_tensor_dense_matmul(sp_mat, grad_flat)
 789 
 790   grad_out = array_ops.reshape(jac, (rows_in, cols_in, batch_size, channels))
 791   grad_out = array_ops.transpose(grad_out, (2, 0, 1, 3))
 792 
 793   return [grad_out]

 810 @ops.RegisterGradient("BroadcastTo")
 811 def _BroadcastToGrad(op, grad):
 812   input_value = op.inputs[0]
 813   broadcast_shape = op.inputs[1]
 814   # Assign ids for each position in input_value.
 815   input_value_shape = array_ops.shape(input_value)
 816   input_value_size = array_ops.size(input_value)
 817   ids = array_ops.reshape(math_ops.range(input_value_size), input_value_shape)
 818   broadcast_ids = array_ops.broadcast_to(ids, broadcast_shape)
 819   # Group by ids and sum its gradients.
 820   grad_flatten = array_ops.reshape(grad, [-1])
 821   broadcast_ids_flatten = array_ops.reshape(broadcast_ids, [-1])
 822   updates_grad_flatten = math_ops.unsorted_segment_sum(grad_flatten,
 823                                                        broadcast_ids_flatten,
 824                                                        input_value_size)
 825   updates_grad = array_ops.reshape(updates_grad_flatten, input_value_shape)
 826   return [updates_grad, None]
