  59 @tf_export("io.SparseFeature", "SparseFeature")
  60 class SparseFeature(
  61     collections.namedtuple(
  62         "SparseFeature",
  63         ["index_key", "value_key", "dtype", "size", "already_sorted"])):
  64   """Configuration for parsing a sparse input feature from an `Example`.
  65 
  66   Note, preferably use `VarLenFeature` (possibly in combination with a
  67   `SequenceExample`) in order to parse out `SparseTensor`s instead of
  68   `SparseFeature` due to its simplicity.
  69 
  70   Closely mimicking the `SparseTensor` that will be obtained by parsing an
  71   `Example` with a `SparseFeature` config, a `SparseFeature` contains a
  72 
  73   * `value_key`: The name of key for a `Feature` in the `Example` whose parsed
  74     `Tensor` will be the resulting `SparseTensor.values`.
  75 
  76   * `index_key`: A list of names - one for each dimension in the resulting
  77     `SparseTensor` whose `indices[i][dim]` indicating the position of
  78     the `i`-th value in the `dim` dimension will be equal to the `i`-th value in
  79     the Feature with key named `index_key[dim]` in the `Example`.
  80 
  81   * `size`: A list of ints for the resulting `SparseTensor.dense_shape`.
  82 
  83   For example, we can represent the following 2D `SparseTensor`
  84 
  85   ```python
  86   SparseTensor(indices=[[3, 1], [20, 0]],
  87                values=[0.5, -1.0]
  88                dense_shape=[100, 3])
  89   ```
  90 
  91   with an `Example` input proto
  92 
  93   ```python
  94   features {
  95     feature { key: "val" value { float_list { value: [ 0.5, -1.0 ] } } }
  96     feature { key: "ix0" value { int64_list { value: [ 3, 20 ] } } }
  97     feature { key: "ix1" value { int64_list { value: [ 1, 0 ] } } }
  98   }
  99   ```
 100 
 101   and `SparseFeature` config with 2 `index_key`s
 102 
 103   ```python
 104   SparseFeature(index_key=["ix0", "ix1"],
 105                 value_key="val",
 106                 dtype=tf.float32,
 107                 size=[100, 3])
 108   ```
 109 
 110   Fields:
 111     index_key: A single string name or a list of string names of index features.
 112       For each key the underlying feature's type must be `int64` and its length
 113       must always match that of the `value_key` feature.
 114       To represent `SparseTensor`s with a `dense_shape` of `rank` higher than 1
 115       a list of length `rank` should be used.
 116     value_key: Name of value feature.  The underlying feature's type must
 117       be `dtype` and its length must always match that of all the `index_key`s'
 118       features.
 119     dtype: Data type of the `value_key` feature.
 120     size: A Python int or list thereof specifying the dense shape. Should be a
 121       list if and only if `index_key` is a list. In that case the list must be
 122       equal to the length of `index_key`. Each for each entry `i` all values in
 123       the `index_key`[i] feature must be in `[0, size[i])`.
 124     already_sorted: A Python boolean to specify whether the values in
 125       `value_key` are already sorted by their index position. If so skip
 126       sorting. False by default (optional).
 127   """
 128 
 129   def __new__(cls, index_key, value_key, dtype, size, already_sorted=False):
 130     return super(SparseFeature, cls).__new__(
 131         cls, index_key, value_key, dtype, size, already_sorted)

 188 def _features_to_raw_params(features, types):
 189   """Split feature tuples into raw params used by `gen_parsing_ops`.
 190 
 191   Args:
 192     features: A `dict` mapping feature keys to objects of a type in `types`.
 193     types: Type of features to allow, among `FixedLenFeature`, `VarLenFeature`,
 194       `SparseFeature`, and `FixedLenSequenceFeature`.
 195 
 196   Returns:
 197     Tuple of `sparse_keys`, `sparse_types`, `dense_keys`, `dense_types`,
 198       `dense_defaults`, `dense_shapes`.
 199 
 200   Raises:
 201     ValueError: if `features` contains an item not in `types`, or an invalid
 202         feature.
 203   """
 204   sparse_keys = []
 205   sparse_types = []
 206   dense_keys = []
 207   dense_types = []
 208   # When the graph is built twice, multiple dense_defaults in a normal dict
 209   # could come out in different orders. This will fail the _e2e_test which
 210   # expects exactly the same graph.
 211   # OrderedDict which preserves the order can solve the problem.
 212   dense_defaults = collections.OrderedDict()
 213   dense_shapes = []
 214   if features:
 215     # NOTE: We iterate over sorted keys to keep things deterministic.
 216     for key in sorted(features.keys()):
 217       feature = features[key]
 218       if isinstance(feature, VarLenFeature):
 219         if VarLenFeature not in types:
 220           raise ValueError("Unsupported VarLenFeature %s." % feature)
 221         if not feature.dtype:
 222           raise ValueError("Missing type for feature %s." % key)
 223         sparse_keys.append(key)
 224         sparse_types.append(feature.dtype)
 225       elif isinstance(feature, SparseFeature):
 226         if SparseFeature not in types:
 227           raise ValueError("Unsupported SparseFeature %s." % feature)
 228 
 229         if not feature.index_key:
 230           raise ValueError(
 231               "Missing index_key for SparseFeature %s." % feature)
 232         if not feature.value_key:
 233           raise ValueError(
 234               "Missing value_key for SparseFeature %s." % feature)
 235         if not feature.dtype:
 236           raise ValueError("Missing type for feature %s." % key)
 237         index_keys = feature.index_key
 238         if isinstance(index_keys, str):
 239           index_keys = [index_keys]
 240         elif len(index_keys) > 1:
 241           tf_logging.warning("SparseFeature is a complicated feature config "
 242                              "and should only be used after careful "
 243                              "consideration of VarLenFeature.")
 244         for index_key in sorted(index_keys):
 245           if index_key in sparse_keys:
 246             dtype = sparse_types[sparse_keys.index(index_key)]
 247             if dtype != dtypes.int64:
 248               raise ValueError("Conflicting type %s vs int64 for feature %s." %
 249                                (dtype, index_key))
 250           else:
 251             sparse_keys.append(index_key)
 252             sparse_types.append(dtypes.int64)
 253         if feature.value_key in sparse_keys:
 254           dtype = sparse_types[sparse_keys.index(feature.value_key)]
 255           if dtype != feature.dtype:
 256             raise ValueError("Conflicting type %s vs %s for feature %s." % (
 257                 dtype, feature.dtype, feature.value_key))
 258         else:
 259           sparse_keys.append(feature.value_key)
 260           sparse_types.append(feature.dtype)
 261       elif isinstance(feature, FixedLenFeature):
 262         if FixedLenFeature not in types:
 263           raise ValueError("Unsupported FixedLenFeature %s." % feature)
 264         if not feature.dtype:
 265           raise ValueError("Missing type for feature %s." % key)
 266         if feature.shape is None:
 267           raise ValueError("Missing shape for feature %s." % key)
 268         feature_tensor_shape = tensor_shape.as_shape(feature.shape)
 269         if (feature.shape and feature_tensor_shape.ndims and
 270             feature_tensor_shape.dims[0].value is None):
 271           raise ValueError("First dimension of shape for feature %s unknown. "
 272                            "Consider using FixedLenSequenceFeature." % key)
 273         if (feature.shape is not None and
 274             not feature_tensor_shape.is_fully_defined()):
 275           raise ValueError("All dimensions of shape for feature %s need to be "
 276                            "known but received %s." % (key, str(feature.shape)))
 277         dense_keys.append(key)
 278         dense_shapes.append(feature.shape)
 279         dense_types.append(feature.dtype)
 280         if feature.default_value is not None:
 281           dense_defaults[key] = feature.default_value
 282       elif isinstance(feature, FixedLenSequenceFeature):
 283         if FixedLenSequenceFeature not in types:
 284           raise ValueError("Unsupported FixedLenSequenceFeature %s." % feature)
 285         if not feature.dtype:
 286           raise ValueError("Missing type for feature %s." % key)
 287         if feature.shape is None:
 288           raise ValueError("Missing shape for feature %s." % key)
 289         dense_keys.append(key)
 290         dense_shapes.append(feature.shape)
 291         dense_types.append(feature.dtype)
 292         if feature.allow_missing:
 293           dense_defaults[key] = None
 294         if feature.default_value is not None:
 295           dense_defaults[key] = feature.default_value
 296       else:
 297         raise ValueError("Invalid feature %s:%s." % (key, feature))
 298   return (
 299       sparse_keys, sparse_types, dense_keys, dense_types, dense_defaults,
 300       dense_shapes)

 303 def _construct_sparse_tensors_for_sparse_features(features, tensor_dict):
 304   """Merges SparseTensors of indices and values of SparseFeatures.
 305 
 306   Constructs new dict based on `tensor_dict`. For `SparseFeatures` in the values
 307   of `features` expects their `index_key`s and `index_value`s to be present in
 308   `tensor_dict` mapping to `SparseTensor`s. Constructs a single `SparseTensor`
 309   from them, and adds it to the result with the key from `features`.
 310   Copies other keys and values from `tensor_dict` with keys present in
 311   `features`.
 312 
 313   Args:
 314     features: A `dict` mapping feature keys to `SparseFeature` values.
 315       Values of other types will be ignored.
 316     tensor_dict: A `dict` mapping feature keys to `Tensor` and `SparseTensor`
 317       values. Expected to contain keys of the `SparseFeature`s' `index_key`s and
 318       `value_key`s and mapping them to `SparseTensor`s.
 319   Returns:
 320     A `dict` mapping feature keys to `Tensor` and `SparseTensor` values. Similar
 321     to `tensor_dict` except each `SparseFeature`s in `features` results in a
 322     single `SparseTensor`.
 323   """
 324   tensor_dict = dict(tensor_dict)  # Do not modify argument passed in.
 325   # Construct SparseTensors for SparseFeatures.
 326   for key in sorted(features.keys()):
 327     feature = features[key]
 328     if isinstance(feature, SparseFeature):
 329       if isinstance(feature.index_key, str):
 330         sp_ids = tensor_dict[feature.index_key]
 331       else:
 332         sp_ids = [tensor_dict[index_key] for index_key in feature.index_key]
 333       sp_values = tensor_dict[feature.value_key]
 334       tensor_dict[key] = sparse_ops.sparse_merge(
 335           sp_ids,
 336           sp_values,
 337           vocab_size=feature.size,
 338           already_sorted=feature.already_sorted)
 339   # Remove tensors from dictionary that were only used to construct
 340   # SparseTensors for SparseFeature.
 341   for key in set(tensor_dict) - set(features):
 342     del tensor_dict[key]
 343   return tensor_dict

 347   if features:
 348     modified_features = dict(features)  # Create a copy to modify
 349     for key, feature in features.items():
 350       if isinstance(feature, FixedLenSequenceFeature):
 351         if not feature.allow_missing:
 352           raise ValueError("Unsupported: FixedLenSequenceFeature requires "

 354         modified_features[key] = FixedLenSequenceFeature(
 355             [None] + list(feature.shape),
 356             feature.dtype,
 357             feature.allow_missing,
 358             feature.default_value)
 359     return modified_features
 360   else:
 361     return features

 364 @tf_export("io.parse_example", "parse_example")
 365 def parse_example(serialized, features, name=None, example_names=None):
 366   # pylint: disable=line-too-long
 367   """Parses `Example` protos into a `dict` of tensors.
 368 
 369   Parses a number of serialized [`Example`](https://www.tensorflow.org/code/tensorflow/core/example/example.proto)
 370   protos given in `serialized`. We refer to `serialized` as a batch with
 371   `batch_size` many entries of individual `Example` protos.
 372 
 373   `example_names` may contain descriptive names for the corresponding serialized
 374   protos. These may be useful for debugging purposes, but they have no effect on
 375   the output. If not `None`, `example_names` must be the same length as
 376   `serialized`.
 377 
 378   This op parses serialized examples into a dictionary mapping keys to `Tensor`
 379   and `SparseTensor` objects. `features` is a dict from keys to `VarLenFeature`,
 380   `SparseFeature`, and `FixedLenFeature` objects. Each `VarLenFeature`
 381   and `SparseFeature` is mapped to a `SparseTensor`, and each
 382   `FixedLenFeature` is mapped to a `Tensor`.
 383 
 384   Each `VarLenFeature` maps to a `SparseTensor` of the specified type
 385   representing a ragged matrix. Its indices are `[batch, index]` where `batch`
 386   identifies the example in `serialized`, and `index` is the value's index in
 387   the list of values associated with that feature and example.
 388 
 389   Each `SparseFeature` maps to a `SparseTensor` of the specified type
 390   representing a Tensor of `dense_shape` `[batch_size] + SparseFeature.size`.
 391   Its `values` come from the feature in the examples with key `value_key`.
 392   A `values[i]` comes from a position `k` in the feature of an example at batch
 393   entry `batch`. This positional information is recorded in `indices[i]` as
 394   `[batch, index_0, index_1, ...]` where `index_j` is the `k-th` value of
 395   the feature in the example at with key `SparseFeature.index_key[j]`.
 396   In other words, we split the indices (except the first index indicating the
 397   batch entry) of a `SparseTensor` by dimension into different features of the
 398   `Example`. Due to its complexity a `VarLenFeature` should be preferred over a
 399   `SparseFeature` whenever possible.
 400 
 401   Each `FixedLenFeature` `df` maps to a `Tensor` of the specified type (or
 402   `tf.float32` if not specified) and shape `(serialized.size(),) + df.shape`.
 403 
 404   `FixedLenFeature` entries with a `default_value` are optional. With no default
 405   value, we will fail if that `Feature` is missing from any example in
 406   `serialized`.
 407 
 408   Each `FixedLenSequenceFeature` `df` maps to a `Tensor` of the specified type
 409   (or `tf.float32` if not specified) and shape
 410   `(serialized.size(), None) + df.shape`.
 411   All examples in `serialized` will be padded with `default_value` along the
 412   second dimension.
 413 
 414   Examples:
 415 
 416   For example, if one expects a `tf.float32` `VarLenFeature` `ft` and three
 417   serialized `Example`s are provided:
 418 
 419   ```
 420   serialized = [
 421     features
 422       { feature { key: "ft" value { float_list { value: [1.0, 2.0] } } } },
 423     features
 424       { feature []},
 425     features
 426       { feature { key: "ft" value { float_list { value: [3.0] } } }
 427   ]
 428   ```
 429 
 430   then the output will look like:
 431 
 432   ```python
 433   {"ft": SparseTensor(indices=[[0, 0], [0, 1], [2, 0]],
 434                       values=[1.0, 2.0, 3.0],
 435                       dense_shape=(3, 2)) }
 436   ```
 437 
 438   If instead a `FixedLenSequenceFeature` with `default_value = -1.0` and
 439   `shape=[]` is used then the output will look like:
 440 
 441   ```python
 442   {"ft": [[1.0, 2.0], [3.0, -1.0]]}
 443   ```
 444 
 445   Given two `Example` input protos in `serialized`:
 446 
 447   ```
 448   [
 449     features {
 450       feature { key: "kw" value { bytes_list { value: [ "knit", "big" ] } } }
 451       feature { key: "gps" value { float_list { value: [] } } }
 452     },
 453     features {
 454       feature { key: "kw" value { bytes_list { value: [ "emmy" ] } } }
 455       feature { key: "dank" value { int64_list { value: [ 42 ] } } }
 456       feature { key: "gps" value { } }
 457     }
 458   ]
 459   ```
 460 
 461   And arguments
 462 
 463   ```
 464   example_names: ["input0", "input1"],
 465   features: {
 466       "kw": VarLenFeature(tf.string),
 467       "dank": VarLenFeature(tf.int64),
 468       "gps": VarLenFeature(tf.float32),
 469   }
 470   ```
 471 
 472   Then the output is a dictionary:
 473 
 474   ```python
 475   {
 476     "kw": SparseTensor(
 477         indices=[[0, 0], [0, 1], [1, 0]],
 478         values=["knit", "big", "emmy"]
 479         dense_shape=[2, 2]),
 480     "dank": SparseTensor(
 481         indices=[[1, 0]],
 482         values=[42],
 483         dense_shape=[2, 1]),
 484     "gps": SparseTensor(
 485         indices=[],
 486         values=[],
 487         dense_shape=[2, 0]),
 488   }
 489   ```
 490 
 491   For dense results in two serialized `Example`s:
 492 
 493   ```
 494   [
 495     features {
 496       feature { key: "age" value { int64_list { value: [ 0 ] } } }
 497       feature { key: "gender" value { bytes_list { value: [ "f" ] } } }
 498      },
 499      features {
 500       feature { key: "age" value { int64_list { value: [] } } }
 501       feature { key: "gender" value { bytes_list { value: [ "f" ] } } }
 502     }
 503   ]
 504   ```
 505 
 506   We can use arguments:
 507 
 508   ```
 509   example_names: ["input0", "input1"],
 510   features: {
 511       "age": FixedLenFeature([], dtype=tf.int64, default_value=-1),
 512       "gender": FixedLenFeature([], dtype=tf.string),
 513   }
 514   ```
 515 
 516   And the expected output is:
 517 
 518   ```python
 519   {
 520     "age": [[0], [-1]],
 521     "gender": [["f"], ["f"]],
 522   }
 523   ```
 524 
 525   An alternative to `VarLenFeature` to obtain a `SparseTensor` is
 526   `SparseFeature`. For example, given two `Example` input protos in
 527   `serialized`:
 528 
 529   ```
 530   [
 531     features {
 532       feature { key: "val" value { float_list { value: [ 0.5, -1.0 ] } } }
 533       feature { key: "ix" value { int64_list { value: [ 3, 20 ] } } }
 534     },
 535     features {
 536       feature { key: "val" value { float_list { value: [ 0.0 ] } } }
 537       feature { key: "ix" value { int64_list { value: [ 42 ] } } }
 538     }
 539   ]
 540   ```
 541 
 542   And arguments
 543 
 544   ```
 545   example_names: ["input0", "input1"],
 546   features: {
 547       "sparse": SparseFeature(
 548           index_key="ix", value_key="val", dtype=tf.float32, size=100),
 549   }
 550   ```
 551 
 552   Then the output is a dictionary:
 553 
 554   ```python
 555   {
 556     "sparse": SparseTensor(
 557         indices=[[0, 3], [0, 20], [1, 42]],
 558         values=[0.5, -1.0, 0.0]
 559         dense_shape=[2, 100]),
 560   }
 561   ```
 562 
 563   Args:
 564     serialized: A vector (1-D Tensor) of strings, a batch of binary
 565       serialized `Example` protos.
 566     features: A `dict` mapping feature keys to `FixedLenFeature`,
 567       `VarLenFeature`, and `SparseFeature` values.
 568     name: A name for this operation (optional).
 569     example_names: A vector (1-D Tensor) of strings (optional), the names of
 570       the serialized protos in the batch.
 571 
 572   Returns:
 573     A `dict` mapping feature keys to `Tensor` and `SparseTensor` values.
 574 
 575   Raises:
 576     ValueError: if any feature is invalid.
 577   """
 578   if not features:
 579     raise ValueError("Missing: features was %s." % features)
 580   features = _prepend_none_dimension(features)
 581   (sparse_keys, sparse_types, dense_keys, dense_types, dense_defaults,
 582    dense_shapes) = _features_to_raw_params(
 583        features,
 584        [VarLenFeature, SparseFeature, FixedLenFeature, FixedLenSequenceFeature])
 585   outputs = _parse_example_raw(
 586       serialized, example_names, sparse_keys, sparse_types, dense_keys,
 587       dense_types, dense_defaults, dense_shapes, name)
 588   return _construct_sparse_tensors_for_sparse_features(features, outputs)

 659 def _process_raw_parameters(names, dense_defaults, sparse_keys, sparse_types,
 660                             dense_keys, dense_types, dense_shapes):
 661   """Process raw parameters to params used by `gen_parsing_ops`.
 662 
 663   Args:
 664     names: A vector (1-D Tensor) of strings (optional), the names of
 665       the serialized protos.
 666     dense_defaults: A dict mapping string keys to `Tensor`s.
 667       The keys of the dict must match the dense_keys of the feature.
 668     sparse_keys: A list of string keys in the examples' features.
 669       The results for these keys will be returned as `SparseTensor` objects.
 670     sparse_types: A list of `DTypes` of the same length as `sparse_keys`.
 671       Only `tf.float32` (`FloatList`), `tf.int64` (`Int64List`),
 672       and `tf.string` (`BytesList`) are supported.
 673     dense_keys: A list of string keys in the examples' features.
 674       The results for these keys will be returned as `Tensor`s
 675     dense_types: A list of DTypes of the same length as `dense_keys`.
 676       Only `tf.float32` (`FloatList`), `tf.int64` (`Int64List`),
 677       and `tf.string` (`BytesList`) are supported.
 678     dense_shapes: A list of tuples with the same length as `dense_keys`.
 679       The shape of the data for each dense feature referenced by `dense_keys`.
 680       Required for any input tensors identified by `dense_keys`.  Must be
 681       either fully defined, or may contain an unknown first dimension.
 682       An unknown first dimension means the feature is treated as having
 683       a variable number of blocks, and the output shape along this dimension
 684       is considered unknown at graph build time.  Padding is applied for
 685       minibatch elements smaller than the maximum number of blocks for the
 686       given feature along this dimension.
 687 
 688   Returns:
 689     Tuple of `names`, `dense_defaults_vec`, `sparse_keys`, `sparse_types`,
 690     `dense_keys`, `dense_shapes`.
 691 
 692   Raises:
 693     ValueError: If sparse and dense key sets intersect, or input lengths do not
 694       match up.
 695   """
 696   names = [] if names is None else names
 697   dense_defaults = collections.OrderedDict(
 698   ) if dense_defaults is None else dense_defaults
 699   sparse_keys = [] if sparse_keys is None else sparse_keys
 700   sparse_types = [] if sparse_types is None else sparse_types
 701   dense_keys = [] if dense_keys is None else dense_keys
 702   dense_types = [] if dense_types is None else dense_types
 703   dense_shapes = ([[]] * len(dense_keys)
 704                   if dense_shapes is None else dense_shapes)
 705 
 706   num_dense = len(dense_keys)
 707   num_sparse = len(sparse_keys)
 708 
 709   if len(dense_shapes) != num_dense:
 710     raise ValueError("len(dense_shapes) != len(dense_keys): %d vs. %d" %
 711                      (len(dense_shapes), num_dense))
 712   if len(dense_types) != num_dense:
 713     raise ValueError("len(dense_types) != len(num_dense): %d vs. %d" %
 714                      (len(dense_types), num_dense))
 715   if len(sparse_types) != num_sparse:
 716     raise ValueError("len(sparse_types) != len(sparse_keys): %d vs. %d" %
 717                      (len(sparse_types), num_sparse))
 718   if num_dense + num_sparse == 0:
 719     raise ValueError("Must provide at least one sparse key or dense key")
 720   if not set(dense_keys).isdisjoint(set(sparse_keys)):
 721     raise ValueError(
 722         "Dense and sparse keys must not intersect; intersection: %s" %
 723         set(dense_keys).intersection(set(sparse_keys)))
 724 
 725   # Convert dense_shapes to TensorShape object.
 726   dense_shapes = [tensor_shape.as_shape(shape) for shape in dense_shapes]
 727 
 728   dense_defaults_vec = []
 729   for i, key in enumerate(dense_keys):
 730     default_value = dense_defaults.get(key)
 731     dense_shape = dense_shapes[i]
 732     if (dense_shape.ndims is not None and dense_shape.ndims > 0 and
 733         dense_shape[0].value is None):
 734       # Variable stride dense shape, the default value should be a
 735       # scalar padding value
 736       if default_value is None:
 737         default_value = ops.convert_to_tensor(
 738             "" if dense_types[i] == dtypes.string else 0, dtype=dense_types[i])
 739       else:
 740         # Reshape to a scalar to ensure user gets an error if they
 741         # provide a tensor that's not intended to be a padding value
 742         # (0 or 2+ elements).
 743         key_name = "padding_" + re.sub("[^A-Za-z0-9_.\\-/]", "_", key)
 744         default_value = ops.convert_to_tensor(
 745             default_value, dtype=dense_types[i], name=key_name)
 746         default_value = array_ops.reshape(default_value, [])
 747     else:
 748       if default_value is None:
 749         default_value = constant_op.constant([], dtype=dense_types[i])
 750       elif not isinstance(default_value, ops.Tensor):
 751         key_name = "key_" + re.sub("[^A-Za-z0-9_.\\-/]", "_", key)
 752         default_value = ops.convert_to_tensor(
 753             default_value, dtype=dense_types[i], name=key_name)
 754         default_value = array_ops.reshape(default_value, dense_shape)
 755 
 756     dense_defaults_vec.append(default_value)
 757 
 758   # Finally, convert dense_shapes to TensorShapeProto
 759   dense_shapes_as_proto = [shape.as_proto() for shape in dense_shapes]
 760 
 761   return (names, dense_defaults_vec, sparse_keys, sparse_types, dense_keys,
 762           dense_shapes_as_proto, dense_shapes)

 915   mapping keys to `Tensor` and `SparseTensor` objects respectively.
 916   The first dictionary contains mappings for keys appearing in
 917   `context_features`, and the second dictionary contains mappings for keys
 918   appearing in `sequence_features`.
 919 
 920   At least one of `context_features` and `sequence_features` must be provided
 921   and non-empty.
 922 
 923   The `context_features` keys are associated with a `SequenceExample` as a
 924   whole, independent of time / frame.  In contrast, the `sequence_features` keys
 925   provide a way to access variable-length data within the `FeatureList` section
 926   of the `SequenceExample` proto.  While the shapes of `context_features` values
 927   are fixed with respect to frame, the frame dimension (the first dimension)
 928   of `sequence_features` values may vary between `SequenceExample` protos,
 929   and even between `feature_list` keys within the same `SequenceExample`.
 930 
 931   `context_features` contains `VarLenFeature` and `FixedLenFeature` objects.
 932   Each `VarLenFeature` is mapped to a `SparseTensor`, and each `FixedLenFeature`
 933   is mapped to a `Tensor`, of the specified type, shape, and default value.
 934 
 935   `sequence_features` contains `VarLenFeature` and `FixedLenSequenceFeature`
 936   objects. Each `VarLenFeature` is mapped to a `SparseTensor`, and each
 937   `FixedLenSequenceFeature` is mapped to a `Tensor`, each of the specified type.
 938   The shape will be `(B,T,) + df.dense_shape` for `FixedLenSequenceFeature`
 939   `df`, where `B` is the batch size, and `T` is the length of the associated
 940   `FeatureList` in the `SequenceExample`. For instance,
 941   `FixedLenSequenceFeature([])` yields a scalar 2-D `Tensor` of static shape
 942   `[None, None]` and dynamic shape `[B, T]`, while
 943   `FixedLenSequenceFeature([k])` (for `int k >= 1`) yields a 3-D matrix `Tensor`
 944   of static shape `[None, None, k]` and dynamic shape `[B, T, k]`.
 945

 947   means that the original per-example shapes of `VarLenFeature`s and
 948   `FixedLenSequenceFeature`s can be lost. To handle that situation, this op also
 949   provides dicts of shape tensors as part of the output. There is one dict for
 950   the context features, and one for the feature_list features. Context features
 951   of type `FixedLenFeature`s will not be present, since their shapes are already
 952   known by the caller. In situations where the input 'FixedLenFeature`s are of
 953   different lengths across examples, the shorter examples will be padded with
 954   default datatype values: 0 for numeric types, and the empty string for string

 958   vector.  Its indices are `[time, index]`, where `time` is the `FeatureList`

 961

 990   Raises:

 996    context_dense_types,
 997    context_dense_defaults, context_dense_shapes) = _features_to_raw_params(
 998        context_features, [VarLenFeature, FixedLenFeature])
 999   (feature_list_sparse_keys, feature_list_sparse_types, feature_list_dense_keys,

1630 def parse_single_example_v2(serialized, features, name=None):
1631   # pylint: disable=line-too-long
1632   """Parses an `Example` proto into a `dict` of tensors.
1633 
1634   Parses a serialized
1635   [`Example`](https://www.tensorflow.org/code/tensorflow/core/example/example.proto)
1636   proto given in `serialized`.
1637 
1638   This op parses serialized examples into a dictionary mapping keys to `Tensor`
1639   and `SparseTensor` objects. `features` is a dict from keys to `VarLenFeature`,
1640   `SparseFeature`, and `FixedLenFeature` objects. Each `VarLenFeature`
1641   and `SparseFeature` is mapped to a `SparseTensor`, and each
1642   `FixedLenFeature` is mapped to a `Tensor`.
1643 
1644   Each `VarLenFeature` maps to a `SparseTensor` of the specified type
1645   representing a ragged matrix. Its indices are `[index]` where
1646   `index` is the value's index in the list of values associated with
1647   that feature and example.
1648 
1649   Each `SparseFeature` maps to a `SparseTensor` of the specified type
1650   representing a Tensor of `dense_shape` `SparseFeature.size`.
1651   Its `values` come from the feature in the examples with key `value_key`.
1652   A `values[i]` comes from a position `k` in the feature of an example at batch
1653   entry `batch`. This positional information is recorded in `indices[i]` as
1654   `[batch, index_0, index_1, ...]` where `index_j` is the `k-th` value of
1655   the feature in the example at with key `SparseFeature.index_key[j]`.
1656   In other words, we split the indices (except the first index indicating the
1657   batch entry) of a `SparseTensor` by dimension into different features of the
1658   `Example`. Due to its complexity a `VarLenFeature` should be preferred over a
1659   `SparseFeature` whenever possible.
1660 
1661   Each `FixedLenFeature` `df` maps to a `Tensor` of the specified type (or
1662   `tf.float32` if not specified) and shape `df.shape`.
1663 
1664   `FixedLenFeature` entries with a `default_value` are optional. With no default
1665   value, we will fail if that `Feature` is missing from any example in
1666   `serialized`.
1667 
1668   Each `FixedLenSequenceFeature` `df` maps to a `Tensor` of the specified type
1669   (or `tf.float32` if not specified) and shape `(None,) + df.shape`.
1670 
1671   Args:
1672     serialized: A scalar (0-D Tensor) string, a serialized `Example` proto.
1673     features: A `dict` mapping feature keys to `FixedLenFeature`,
1674       `VarLenFeature`, and `SparseFeature` values.
1675     name: A name for this operation (optional).
1676 
1677   Returns:
1678     A `dict` mapping feature keys to `Tensor` and `SparseTensor` values.
1679 
1680   Raises:
1681     ValueError: if any feature is invalid.
1682   """
1683   if not features:
1684     raise ValueError("Missing: features was %s." % features)
1685   features = _prepend_none_dimension(features)
1686   (sparse_keys, sparse_types, dense_keys, dense_types,
1687    dense_defaults, dense_shapes) = _features_to_raw_params(
1688        features,
1689        [VarLenFeature, SparseFeature, FixedLenFeature, FixedLenSequenceFeature])
1690   outputs = _parse_single_example_v2_raw(serialized, sparse_keys, sparse_types,
1691                                          dense_keys, dense_types,
1692                                          dense_defaults, dense_shapes, name)
1693   return _construct_sparse_tensors_for_sparse_features(features, outputs)

1696 def _parse_single_example_v2_raw(serialized, sparse_keys, sparse_types,
1697                                  dense_keys, dense_types, dense_defaults,
1698                                  dense_shapes, name):
1699   """Parses `Example` protos.
1700 
1701   Args:
1702     serialized: A scalar (0-D Tensor) string, containing a binary
1703       serialized `Example` proto.
1704     sparse_keys: A list of string keys in the examples' features.
1705       The results for these keys will be returned as `SparseTensor` objects.
1706     sparse_types: A list of `DTypes` of the same length as `sparse_keys`.
1707       Only `tf.float32` (`FloatList`), `tf.int64` (`Int64List`),
1708       and `tf.string` (`BytesList`) are supported.
1709     dense_keys: A list of string keys in the examples' features.
1710       The results for these keys will be returned as `Tensor`s
1711     dense_types: A list of DTypes of the same length as `dense_keys`.
1712       Only `tf.float32` (`FloatList`), `tf.int64` (`Int64List`),
1713       and `tf.string` (`BytesList`) are supported.
1714     dense_defaults: A dict mapping string keys to `Tensor`s.
1715       The keys of the dict must match the dense_keys of the feature.
1716     dense_shapes: A list of tuples with the same length as `dense_keys`.
1717       The shape of the data for each dense feature referenced by `dense_keys`.
1718       Required for any input tensors identified by `dense_keys`.  Must be
1719       either fully defined, or may contain an unknown first dimension.
1720       An unknown first dimension means the feature is treated as having
1721       a variable number of blocks, and the output shape along this dimension
1722       is considered unknown at graph build time.  Padding is applied for
1723       minibatch elements smaller than the maximum number of blocks for the
1724       given feature along this dimension.
1725     name: A name for this operation (optional).
1726 
1727   Returns:
1728     A `dict` mapping keys to `Tensor`s and `SparseTensor`s.
1729 
1730   Raises:
1731     ValueError: If sparse and dense key sets intersect, or input lengths do not
1732       match up.
1733   """
1734   with ops.name_scope(name, "ParseSingleExample", [serialized]):
1735     serialized = ops.convert_to_tensor(serialized, name="serialized")
1736     dense_defaults = collections.OrderedDict(
1737     ) if dense_defaults is None else dense_defaults
1738     sparse_keys = [] if sparse_keys is None else sparse_keys
1739     sparse_types = [] if sparse_types is None else sparse_types
1740     dense_keys = [] if dense_keys is None else dense_keys
1741     dense_types = [] if dense_types is None else dense_types
1742     dense_shapes = ([[]] * len(dense_keys)
1743                     if dense_shapes is None else dense_shapes)
1744 
1745     num_dense = len(dense_keys)
1746     num_sparse = len(sparse_keys)
1747 
1748     if len(dense_shapes) != num_dense:
1749       raise ValueError("len(dense_shapes) != len(dense_keys): %d vs. %d" %
1750                        (len(dense_shapes), num_dense))
1751     if len(dense_types) != num_dense:
1752       raise ValueError("len(dense_types) != len(num_dense): %d vs. %d" %
1753                        (len(dense_types), num_dense))
1754     if len(sparse_types) != num_sparse:
1755       raise ValueError("len(sparse_types) != len(sparse_keys): %d vs. %d" %
1756                        (len(sparse_types), num_sparse))
1757     if num_dense + num_sparse == 0:
1758       raise ValueError("Must provide at least one sparse key or dense key")
1759     if not set(dense_keys).isdisjoint(set(sparse_keys)):
1760       raise ValueError(
1761           "Dense and sparse keys must not intersect; intersection: %s" %
1762           set(dense_keys).intersection(set(sparse_keys)))
1763 
1764     # Convert dense_shapes to TensorShape object.
1765     dense_shapes = [tensor_shape.as_shape(shape) for shape in dense_shapes]
1766 
1767     dense_defaults_vec = []
1768     for i, key in enumerate(dense_keys):
1769       default_value = dense_defaults.get(key)
1770       dense_shape = dense_shapes[i]
1771       if (dense_shape.ndims is not None and dense_shape.ndims > 0 and
1772           dense_shape[0].value is None):
1773         # Variable stride dense shape, the default value should be a
1774         # scalar padding value
1775         if default_value is None:
1776           default_value = ops.convert_to_tensor(
1777               "" if dense_types[i] == dtypes.string else 0,

1779         else:

1782           # (0 or 2+ elements).
1783           key_name = "padding_" + re.sub("[^A-Za-z0-9_.\\-/]", "_", key)
1784           default_value = ops.convert_to_tensor(
1785               default_value, dtype=dense_types[i], name=key_name)
1786           default_value = array_ops.reshape(default_value, [])
1787       else:
1788         if default_value is None:
1789           default_value = constant_op.constant([], dtype=dense_types[i])
1790         elif not isinstance(default_value, ops.Tensor):
1791           key_name = "key_" + re.sub("[^A-Za-z0-9_.\\-/]", "_", key)
1792           default_value = ops.convert_to_tensor(
1793               default_value, dtype=dense_types[i], name=key_name)
1794           default_value = array_ops.reshape(default_value, dense_shape)
1795 
1796       dense_defaults_vec.append(default_value)
1797 
1798     # Finally, convert dense_shapes to TensorShapeProto
1799     dense_shapes = [shape.as_proto() for shape in dense_shapes]
1800 
1801     outputs = gen_parsing_ops.parse_single_example(

1804         num_sparse=len(sparse_keys),

1806         sparse_types=sparse_types,
1807         dense_keys=dense_keys,
1808         dense_shapes=dense_shapes,
1809         name=name)

1812

1814         sparse_tensor.SparseTensor(ix, val, shape)
1815         for (ix, val,
1816              shape) in zip(sparse_indices, sparse_values, sparse_shapes)
1817     ]
