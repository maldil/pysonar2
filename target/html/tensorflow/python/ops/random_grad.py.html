  28 def add_leading_unit_dimensions(x, num_dimensions):
  29   new_shape = array_ops.concat(
  30       [array_ops.ones([num_dimensions], dtype=dtypes.int32),
  31        array_ops.shape(x)], axis=0)
  32   return array_ops.reshape(x, new_shape)

  35 @ops.RegisterGradient("RandomGamma")
  36 def _RandomGammaGrad(op, grad):  # pylint: disable=invalid-name
  37   """Returns the gradient of a Gamma sample w.r.t. alpha.
  38 
  39   The gradient is computed using implicit differentiation, see
  40   "Implicit Reparameterization Gradients" (https://arxiv.org/abs/1805.08498).
  41 
  42   Args:
  43     op: A `RandomGamma` operation. We assume that the inputs to the operation
  44       are `shape` and `alpha` tensors, and the output is the `sample` tensor.
  45     grad: The incoming gradient `dloss / dsample` of the same shape as
  46       `op.outputs[0]`.
  47 
  48   Returns:
  49     A `Tensor` with derivatives `dloss / dalpha`
  50   """
  51   shape = op.inputs[0]
  52   alpha = op.inputs[1]
  53   sample = op.outputs[0]
  54 
  55   with ops.control_dependencies([grad]):
  56     # Make the parameters alpha broadcastable with samples by appending
  57     # unit dimensions.
  58     num_sample_dimensions = array_ops.shape(shape)[0]
  59     alpha_broadcastable = add_leading_unit_dimensions(
  60         alpha, num_sample_dimensions)
  61     partial_a = gen_random_ops.random_gamma_grad(alpha_broadcastable, sample)
  62 
  63     # The first input is shape; the second input is alpha.
  64     return (None, math_ops.reduce_sum(
  65         grad * partial_a, axis=math_ops.range(num_sample_dimensions)))
