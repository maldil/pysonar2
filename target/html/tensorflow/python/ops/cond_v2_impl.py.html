  16 
  17 This is a version of cond that emits a single If op, as well as the gradient
  18 function for If ops produced by cond_v2. This will eventually replace the
  19 current tf.cond implementation once it reaches feature and performance parity.

  53 def cond_v2(pred, true_fn, false_fn, name="cond"):
  54   """Like tf.cond, except emits a single If op."""
  55   if not name:
  56     name = "cond"
  57 
  58   with ops.name_scope(name) as scope:
  59     with ops.name_scope(None):
  60       # Find the outer most graph for uniquing function names.
  61       # TODO(jpienaar): Make this work in eager mode.
  62       graph = ops.get_default_graph()
  63       while isinstance(graph, _function.FuncGraph):
  64         graph = graph.outer_graph
  65 
  66       true_name = graph.unique_name(("%strue" % scope).replace("/", "_"))
  67       false_name = graph.unique_name(("%sfalse" % scope).replace("/", "_"))
  68 
  69     true_graph = _function.func_graph_from_py_func(
  70         true_name, true_fn, [], {})
  71     false_graph = _function.func_graph_from_py_func(
  72         false_name, false_fn, [], {})
  73     _check_same_outputs(true_graph, false_graph)
  74 
  75     # Add inputs to true_graph and false_graph to make them match. Note that
  76     # this modifies true_graph and false_graph.
  77     cond_inputs = _make_inputs_match(true_graph, false_graph,
  78                                      true_graph.external_captures,
  79                                      false_graph.external_captures)
  80 
  81     # Add all intermediate tensors as function outputs so they're available for
  82     # the gradient computation.
  83 
  84     true_intermediates = _get_intermediates(true_graph)
  85     false_intermediates = _get_intermediates(false_graph)
  86 
  87     # Save the original number of outputs to return to the caller.
  88     num_cond_outputs = len(true_graph.outputs)
  89 
  90     # Make the number/type of new intermediate outputs match.
  91     extra_true_outputs, extra_false_outputs = _pad_params(
  92         true_graph, false_graph, true_intermediates, false_intermediates)
  93 
  94     true_graph.outputs.extend(extra_true_outputs)
  95     false_graph.outputs.extend(extra_false_outputs)
  96 
  97     # Create the If op.
  98     tensors = gen_functional_ops._if(  # pylint: disable=protected-access
  99         pred, cond_inputs, [t.dtype for t in true_graph.outputs],
 100         _create_new_tf_function(true_graph),
 101         _create_new_tf_function(false_graph),
 102         name=scope)
 103 
 104     # Set the flag to enable lowering on the `if` op if necessary
 105     # Lowering allows cond_v2 to avoid some of the limitations of Functions,
 106     # allowing users to specify devices & colocation inside of cond_v2 branches,
 107     # and enabling non-strict evaluation & partial pruning of cond_v2 branches.
 108     # This brings cond_v2 closer to feature parity with tf.cond.
 109     #
 110     # However, we do not lower `If` in the XLA context because it is easier for
 111     # XLA to apply its own optimizations when dealing with un-lowered `If`
 112     # operators than with lowered switch/merge control flow.
 113     #
 114     # TODO(b/110167197) this approach requires cond_v2 to have at least 1 output
 115     if_op = tensors[0].op
 116     if not control_flow_util.IsInXLAContext(if_op):
 117       # pylint: disable=protected-access
 118       if_op._set_attr("_lower_using_switch_merge",
 119                       attr_value_pb2.AttrValue(b=True))
 120       # pylint: enable=protected-access
 121 
 122     result = tuple(tensors[:num_cond_outputs])
 123     if len(result) == 1:
 124       return result[0]
 125     else:
 126       return result

 129 @ops.RegisterGradient("If")
 130 def _IfGrad(op, *grads):  # pylint: disable=invalid-name
 131   """The gradient of an If op produced by cond_v2."""
 132   true_graph, false_graph = _get_func_graphs(op)
 133   # Note: op.graph != ops.get_default_graph() when we are computing the gradient
 134   # of a nested cond.
 135   assert true_graph.outer_graph == op.graph
 136   assert false_graph.outer_graph == op.graph
 137 
 138   # Create grad functions that compute the gradient of the true/false forward
 139   # graphs. These functions will capture tensors from the forward pass
 140   # functions.
 141   true_grad_graph = _create_grad_func(
 142       true_graph, grads, _get_grad_fn_name(true_graph))
 143   false_grad_graph = _create_grad_func(
 144       false_graph, grads, _get_grad_fn_name(false_graph))
 145 
 146   assert ([t.dtype for t in true_grad_graph.outputs] ==
 147           [t.dtype for t in false_grad_graph.outputs])
 148 
 149   # Resolve references to forward graph tensors in grad graphs and ensure
 150   # they are in-scope, i.e., belong to one of outer graphs of the grad graph.
 151   true_grad_inputs = _resolve_grad_inputs(true_graph, true_grad_graph)
 152   false_grad_inputs = _resolve_grad_inputs(false_graph, false_grad_graph)
 153 
 154   # Make the inputs to true_grad_graph and false_grad_graph match. Note that
 155   # this modifies true_grad_graph and false_grad_graph.
 156   grad_inputs = _make_inputs_match(true_grad_graph, false_grad_graph,
 157                                    true_grad_inputs, false_grad_inputs)
 158 
 159   # Add all intermediate tensors as function outputs so they're available for
 160   # higher-order gradient computations.
 161 
 162   true_grad_intermediates = _get_intermediates(true_grad_graph)
 163   false_grad_intermediates = _get_intermediates(false_grad_graph)
 164 
 165   # Save the original number of gradient outputs to return.
 166   num_grad_outputs = len(true_grad_graph.outputs)
 167 
 168   # Make the number/type of new intermediate outputs match.
 169   extra_true_grad_outputs, extra_false_grad_outputs = _pad_params(
 170       true_grad_graph, false_grad_graph,
 171       true_grad_intermediates, false_grad_intermediates)
 172 
 173   true_grad_graph.outputs.extend(extra_true_grad_outputs)
 174   false_grad_graph.outputs.extend(extra_false_grad_outputs)
 175 
 176   # Create the gradient If op.
 177   tensors = gen_functional_ops._if(
 178       op.inputs[0], grad_inputs, [t.dtype for t in true_grad_graph.outputs],
 179       _create_new_tf_function(true_grad_graph),
 180       _create_new_tf_function(false_grad_graph))
 181 
 182   # The predicate has no gradient.
 183   return [None] + tensors[:num_grad_outputs]

 186 def _get_func_graphs(if_op):
 187   """Returns `FuncGraph`s for the input op branches.
 188 
 189   Args:
 190     if_op: The _If Operation.
 191 
 192   Returns:
 193     A 2-tuple of the `FuncGraph`s of the then_branch and else_branch.
 194   """
 195   def _get_func_graph_for_branch(branch_name):
 196     """Generates and returns a FuncGraph for the given branch."""
 197     inputs = if_op.inputs[1:]  # First input is pred.
 198     input_shapes = [t.shape for t in inputs]
 199     func_name = if_op.get_attr(branch_name).name
 200     fdef = if_op.graph._get_function(func_name).definition
 201     # `if_op.graph` may not be the same as `ops.get_default_graph()` e.g.
 202     # in the case of nested if ops or when the gradient is being computed
 203     # from inside a Defun. We build the `func_graph` with `if_op.graph` as its
 204     # `outer_graph`. This resembles how the `FuncGraph` was built in the
 205     # forward pass. We need this so that we can resolve references to tensors
 206     # in `func_graph` from its gradient graph in `_resolve_grad_inputs`.
 207     with if_op.graph.as_default():
 208       func_graph = _function_def_to_graph.function_def_to_graph(
 209           fdef, input_shapes)
 210     func_graph.captures = collections.OrderedDict(zip(inputs,
 211                                                       func_graph.inputs))
 212     # Set the if op so that the gradient code can use it.
 213     func_graph._if = if_op
 214     return func_graph
 215 
 216   return (_get_func_graph_for_branch("then_branch"),
 217           _get_func_graph_for_branch("else_branch"))

 220 def _grad_fn(func_graph, grads):
 221   """The gradient function for each conditional branch.
 222 
 223   This function builds the gradient graph of the corresponding forward-pass
 224   conditional branch in `func_graph`. This is done by differentiating
 225   func_graph's outputs w.r.t. its inputs.
 226 
 227   Args:
 228     func_graph: function.FuncGraph. The corresponding forward-pass function.
 229     grads: The list of input gradient Tensors.
 230 
 231   Returns:
 232     The output gradient Tensors.
 233   """
 234   # Filter out untrainable function outputs.
 235   # NOTE(skyewm): If we don't do this, the untrainable tensors can sometimes
 236   # cause _GradientsHelper to raise an exception (e.g. the implementation
 237   # doesn't expect 'ys' to contain boolean tensors).
 238   assert len(func_graph.outputs) == len(grads)
 239   ys = []
 240   grad_ys = []
 241   for y, grad_y in zip(func_graph.outputs, grads):
 242     if not _gradients_impl._IsTrainable(y):
 243       continue
 244     ys.append(y)
 245     grad_ys.append(grad_y)
 246 
 247   # Build the gradient graph. Note that this builds the gradient computation of
 248   # func_graph in the current graph, which requires capturing tensors from
 249   # func_graph. The captured func_graph tensors are resolved to external tensors
 250   # in _resolve_grad_inputs.
 251   result = _gradients_impl._GradientsHelper(
 252       ys, func_graph.inputs, grad_ys=grad_ys,
 253       src_graph=func_graph)
 254 
 255   # Functions can't return None; replace Nones with zero tensors.
 256   # TODO(b/80444525): don't return anything here and make _IfGrad return None if
 257   # both branches have zero gradient.
 258   for i in range(len(result)):
 259     if result[i] is None:
 260       result[i] = array_ops.zeros_like(func_graph.inputs[i])
 261 
 262   return result

 265 def _create_grad_func(func_graph, grads, name):
 266   """Returns the FuncGraph representation of _grad_fn."""
 267   return _function.func_graph_from_py_func(
 268       name, lambda: _grad_fn(func_graph, grads), [], {})

 271 def _resolve_grad_inputs(cond_graph, grad_graph):
 272   """Returns the tensors to pass as inputs to `grad_graph`.
 273 
 274   The `grad_graph` may have external references to
 275   1. Its outer graph containing the input gradients. These references are kept
 276      as is.
 277   2. Tensors in the forward pass graph. These tensors may not be "live"
 278      when the gradient is being computed. We replace such references by their
 279      corresponding tensor in the least common ancestor graph of `grad_graph` and
 280      `cond_graph`. Since we export intermediate tensors for all branch
 281      functions, this is always possible.
 282 
 283   Args:
 284     cond_graph: function.FuncGraph. The forward-pass function.
 285     grad_graph: function.FuncGraph. The gradients function.
 286 
 287   Returns:
 288     A list of inputs tensors to be passed to grad_graph.
 289   """
 290   new_inputs = []
 291 
 292   for t in grad_graph.external_captures:
 293     if t.graph != grad_graph.outer_graph:
 294       # `t` is a tensor in `cond_graph` or one of its ancestors. We bubble this
 295       # tensor to the least common ancestor of the `cond_graph` and
 296       # `grad_graph` so that it is "in-scope" for `grad_graph`.
 297       # TODO(srbs): `_is_ancestor` calls may be expensive. Compute the least
 298       # common ancestor once and re-use.
 299       assert _is_ancestor(cond_graph, t.graph)
 300       while not _is_ancestor(grad_graph, t.graph):
 301         assert isinstance(t.graph, _function.FuncGraph)
 302         if t in t.graph.internal_captures:
 303           # TODO(srbs): Consider building a map of internal_captures ->
 304           # external_captures instead of searching for `t` twice.
 305           t = t.graph.external_captures[t.graph.internal_captures.index(t)]
 306         else:
 307           # Note: All intermediate tensors are output by the If op.
 308           # TODO(srbs): .index() calls may be expensive. Optimize.
 309           t = t.graph._if.outputs[t.graph.outputs.index(t)]
 310       assert _is_ancestor(grad_graph, t.graph)
 311     new_inputs.append(t)
 312 
 313   return new_inputs

 316 def _create_new_tf_function(func_graph):
 317   """Converts func_graph to a TF_Function and adds it to the current graph.
 318 
 319   Args:
 320     func_graph: function.FuncGraph
 321 
 322   Returns:
 323     The name of the new TF_Function.
 324   """
 325   func = _function._EagerDefinedFunction(
 326       func_graph.name, func_graph, func_graph.inputs, func_graph.outputs, {})
 327   func.add_to_graph(func_graph.outer_graph)
 328   return func_graph.name

 331 def _get_intermediates(func_graph):
 332   """Returns all tensors in `func_graph` that aren't inputs or outputs."""
 333   intermediates = []
 334   for op in func_graph.get_operations():
 335     for t in op.outputs:
 336       if t in func_graph.inputs: continue
 337       if t in func_graph.outputs: continue
 338       intermediates.append(t)
 339   return intermediates

 342 def _separate_unique_inputs(true_inputs, false_inputs):
 343   """Separates tensors appearing only in true_inputs or false_inputs, or both.
 344 
 345   Args:
 346     true_inputs: list of Tensors
 347     false_inputs: list of Tensors
 348 
 349   Returns:
 350     Three lists of Tensors:
 351       1. The tensors that appear in both true_inputs and false_inputs
 352       2. The tensors that only appear in true_inputs
 353       3. The tensors that only appear in false_inputs
 354   """
 355   true_inputs = set(true_inputs)
 356   false_inputs = set(false_inputs)
 357 
 358   shared_inputs = true_inputs.intersection(false_inputs)
 359   true_only_inputs = true_inputs - false_inputs
 360   false_only_inputs = false_inputs - true_inputs
 361 
 362   return list(shared_inputs), list(true_only_inputs), list(false_only_inputs)

 365 def _pad_params(true_graph, false_graph, true_params, false_params):
 366   """Returns new param lists that have matching signatures.
 367 
 368   This is done by mirroring each param list in the other using dummy params.
 369   There is no merging of params.
 370 
 371   Args:
 372     true_graph: function.FuncGraph
 373     false_graph: function.FuncGraph
 374     true_params: a list of Tensors from true_graph
 375     false_params: a list of Tensors from false_graph
 376 
 377   Returns:
 378     A new list of Tensors in true_graph and a new list of Tensors in
 379     false_graph. The two lists have the same number of Tensors, with matching
 380     types and shapes across the lists.
 381   """
 382   new_true_params = (true_params +
 383                      _create_dummy_params(true_graph, false_params))
 384   new_false_inputs = (_create_dummy_params(false_graph, true_params)
 385                       + false_params)
 386   return new_true_params, new_false_inputs

 389 def _make_inputs_match(true_graph, false_graph, true_inputs, false_inputs):
 390   """Modifies true_graph and false_graph so they have the same input signature.
 391 
 392   This method reorders and/or adds parameters to true_graph and false_graph so
 393   they have the same input signature, and updates the 'inputs' and 'captured'
 394   fields of both graphs accordingly. It uses the input tensors from the outer
 395   graph to avoid duplicating shared arguments.
 396 
 397   Args:
 398     true_graph: function.FuncGraph
 399     false_graph: function.FuncGraph
 400     true_inputs: a list of Tensors in the outer graph. The inputs for
 401       true_graph.
 402     false_inputs: a list of Tensors in the outer graph. The inputs for
 403       false_graph.
 404 
 405   Returns:
 406     A new list of Tensors from the outer graph that are the new inputs for both
 407     true_graph and false_graph. This is a deduped version of true_inputs +
 408     false_inputs.
 409   """
 410   shared_inputs, true_only_inputs, false_only_inputs = _separate_unique_inputs(
 411       true_inputs, false_inputs)
 412 
 413   new_inputs = shared_inputs + true_only_inputs + false_only_inputs
 414 
 415   true_input_to_param = dict(zip(true_inputs, true_graph.inputs))
 416   false_input_to_param = dict(zip(false_inputs, false_graph.inputs))
 417 
 418   true_graph.inputs = (
 419       [true_input_to_param[t] for t in shared_inputs] +
 420       [true_input_to_param[t] for t in true_only_inputs] +
 421       _create_dummy_params(true_graph, false_only_inputs))
 422 
 423   false_graph.inputs = (
 424       [false_input_to_param[t] for t in shared_inputs] +
 425       _create_dummy_params(false_graph, true_only_inputs) +
 426       [false_input_to_param[t] for t in false_only_inputs])
 427 
 428   # Rewrite the FuncGraphs' state to reflect the new inputs.
 429   true_graph.captures = collections.OrderedDict(zip(new_inputs,
 430                                                     true_graph.inputs))
 431   false_graph.captures = collections.OrderedDict(zip(new_inputs,
 432                                                      false_graph.inputs))
 433 
 434   return new_inputs

 437 def _create_dummy_params(func_graph, template_tensors):
 438   """Creates tensors in func_graph to represent template_tensors.
 439 
 440   Args:
 441     func_graph: function.FuncGraph.
 442     template_tensors: a list of tensors in the outer graph.
 443 
 444   Returns:
 445     A list of tensors in func_graph.
 446   """
 447   with func_graph.as_default():
 448     return [gen_functional_ops.fake_param(dtype=t.dtype, shape=t.shape)
 449             for t in template_tensors]

 452 def _get_grad_fn_name(func_graph):
 453   """Returns a unique name to use for the grad function of `func_graph`.
 454 
 455   Ensures this name is unique in the entire hierarchy.
 456 
 457   Args:
 458     func_graph: The FuncGraph.
 459 
 460   Returns:
 461     A string, the name to use for the gradient function.
 462   """
 463   name = "%s_grad" % func_graph.name
 464   outer_most_graph = func_graph
 465   while isinstance(outer_most_graph, _function.FuncGraph):
 466     outer_most_graph = outer_most_graph.outer_graph
 467   return outer_most_graph.unique_name(name)

 470 def _check_same_outputs(true_graph, false_graph):
 471   """Raises an error if true_graph and false_graph have different outputs."""
 472   true_output_types = [t.dtype for t in true_graph.outputs]
 473   false_output_types = [t.dtype for t in false_graph.outputs]
 474   if (len(true_graph.outputs) != len(false_graph.outputs) or
 475       true_output_types != false_output_types):
 476     raise ValueError(
 477         "true_fn() and false_fn() must return the same number and type of "
 478         "arguments, got:\n"
 479         "  true_fn: %s\n"
 480         "  false_fn: %s" % (true_output_types, false_output_types))

 483 def _is_ancestor(graph, maybe_ancestor):
 484   if maybe_ancestor == graph:
 485     return True
 486   if isinstance(graph, _function.FuncGraph):
 487     return _is_ancestor(graph.outer_graph, maybe_ancestor)
 488   return False
