  80 def init_variable(v, init, name="init"):
  81   """Initializes variable with "init".
  82 
  83   This op does the following:
  84   if init is a Tensor, v = init
  85   if callable(init): v = init(VariableShape(v), v.dtype)
  86 
  87   Args:
  88     v: Variable to initialize
  89     init: Tensor to assign to v,
  90       Or an object convertible to Tensor e.g. nparray,
  91       Or an Initializer that generates a tensor given the shape and type of v.
  92       An "Initializer" is a callable that returns a tensor that "v" should be
  93       set to. It will be called as init(shape, dtype).
  94     name: Optional name for the op.
  95 
  96   Returns:
  97     The operation that initializes v.
  98   """
  99   with ops.name_scope(None, v.op.name + "/", [v, init]):
 100     with ops.name_scope(name) as scope:
 101       with ops.colocate_with(v):
 102         if callable(init):
 103           assert v.get_shape().is_fully_defined(), "Variable shape unknown."
 104           # TODO(mrry): Convert to v.shape when the property and
 105           # accessor are reconciled (and all initializers support
 106           # tf.TensorShape objects).
 107           value = init(v.get_shape().as_list(), v.dtype.base_dtype)
 108           value = ops.convert_to_tensor(value, name="value")
 109           return gen_state_ops.assign(v, value, name=scope)
 110         else:
 111           init = ops.convert_to_tensor(init, name="init")
 112           return gen_state_ops.assign(v, init, name=scope)

 248 @tf_export(v1=["scatter_update"])
 249 def scatter_update(ref, indices, updates, use_locking=True, name=None):
 250   # pylint: disable=line-too-long
 251   r"""Applies sparse updates to a variable reference.
 252 
 253   This operation computes
 254 
 255   ```python
 256       # Scalar indices
 257       ref[indices, ...] = updates[...]
 258 
 259       # Vector indices (for each i)
 260       ref[indices[i], ...] = updates[i, ...]
 261 
 262       # High rank indices (for each i, ..., j)
 263       ref[indices[i, ..., j], ...] = updates[i, ..., j, ...]
 264   ```
 265 
 266   This operation outputs `ref` after the update is done.
 267   This makes it easier to chain operations that need to use the reset value.
 268 
 269   If values in `ref` is to be updated more than once, because there are
 270   duplicate entries in `indices`, the order at which the updates happen
 271   for each value is undefined.
 272 
 273   Requires `updates.shape = indices.shape + ref.shape[1:]`.
 274 
 275   <div style="width:70%; margin:auto; margin-bottom:10px; margin-top:20px;">
 276   <img style="width:100%" src="https://www.tensorflow.org/images/ScatterUpdate.png" alt>
 277   </div>
 278 
 279   Args:
 280     ref: A `Variable`.
 281     indices: A `Tensor`. Must be one of the following types: `int32`, `int64`.
 282       A tensor of indices into the first dimension of `ref`.
 283     updates: A `Tensor`. Must have the same type as `ref`.
 284       A tensor of updated values to store in `ref`.
 285     use_locking: An optional `bool`. Defaults to `True`.
 286       If True, the assignment will be protected by a lock;
 287       otherwise the behavior is undefined, but may exhibit less contention.
 288     name: A name for the operation (optional).
 289 
 290   Returns:
 291     Same as `ref`.  Returned as a convenience for operations that want
 292     to use the updated values after the update is done.
 293   """
 294   if ref.dtype._is_ref_dtype:
 295     return gen_state_ops.scatter_update(ref, indices, updates,
 296                                         use_locking=use_locking, name=name)
 297   return ref._lazy_read(gen_resource_variable_ops.resource_scatter_update(  # pylint: disable=protected-access
 298       ref.handle, indices, ops.convert_to_tensor(updates, ref.dtype),
 299       name=name))

 302 @tf_export(v1=["scatter_nd_update"])
 303 def scatter_nd_update(ref, indices, updates, use_locking=True, name=None):
 304   r"""Applies sparse `updates` to individual values or slices in a Variable.
 305 
 306   `ref` is a `Tensor` with rank `P` and `indices` is a `Tensor` of rank `Q`.
 307 
 308   `indices` must be integer tensor, containing indices into `ref`.
 309   It must be shape `[d_0, ..., d_{Q-2}, K]` where `0 < K <= P`.
 310 
 311   The innermost dimension of `indices` (with length `K`) corresponds to
 312   indices into elements (if `K = P`) or slices (if `K < P`) along the `K`th
 313   dimension of `ref`.
 314 
 315   `updates` is `Tensor` of rank `Q-1+P-K` with shape:
 316 
 317   ```
 318   [d_0, ..., d_{Q-2}, ref.shape[K], ..., ref.shape[P-1]].
 319   ```
 320 
 321   For example, say we want to update 4 scattered elements to a rank-1 tensor to
 322   8 elements. In Python, that update would look like this:
 323 
 324   ```python
 325       ref = tf.Variable([1, 2, 3, 4, 5, 6, 7, 8])
 326       indices = tf.constant([[4], [3], [1] ,[7]])
 327       updates = tf.constant([9, 10, 11, 12])
 328       update = tf.scatter_nd_update(ref, indices, updates)
 329       with tf.Session() as sess:
 330         print sess.run(update)
 331   ```
 332

 334 
 335       [1, 11, 3, 10, 9, 6, 7, 12]
 336 
 337   See `tf.scatter_nd` for more details about how to make updates to
 338   slices.
 339 
 340   Args:
 341     ref: A Variable.
 342     indices: A `Tensor`. Must be one of the following types: `int32`, `int64`.
 343       A tensor of indices into ref.
 344     updates: A `Tensor`. Must have the same type as `ref`.
 345       A Tensor. Must have the same type as ref. A tensor of updated
 346       values to add to ref.
 347     use_locking: An optional `bool`. Defaults to `True`.
 348       An optional bool. Defaults to True. If True, the assignment will
 349       be protected by a lock; otherwise the behavior is undefined,
 350       but may exhibit less contention.
 351     name: A name for the operation (optional).
 352 
 353   Returns:
 354     The value of the variable after the update.
 355   """
 356   if ref.dtype._is_ref_dtype:
 357     return gen_state_ops.scatter_nd_update(
 358         ref, indices, updates, use_locking, name)
 359   return ref._lazy_read(gen_state_ops.resource_scatter_nd_update(  # pylint: disable=protected-access
 360       ref.handle, indices, ops.convert_to_tensor(updates, ref.dtype),
 361       name=name))

 598 @tf_export("batch_scatter_update")
 599 def batch_scatter_update(ref, indices, updates, use_locking=True, name=None):
 600   """Generalization of `tf.scatter_update` to axis different than 0.
 601 
 602   Analogous to `batch_gather`. This assumes that `ref`, `indices` and `updates`
 603   have a series of leading dimensions that are the same for all of them, and the
 604   updates are performed on the last dimension of indices. In other words, the
 605   dimensions should be the following:
 606 
 607   `num_prefix_dims = indices.ndims - 1`
 608   `batch_dim = num_prefix_dims + 1`
 609   `updates.shape = indices.shape + var.shape[batch_dim:]`
 610 
 611   where
 612 
 613   `updates.shape[:num_prefix_dims]`
 614   `== indices.shape[:num_prefix_dims]`
 615   `== var.shape[:num_prefix_dims]`
 616 
 617   And the operation performed can be expressed as:
 618 
 619   `var[i_1, ..., i_n, indices[i_1, ..., i_n, j]] = updates[i_1, ..., i_n, j]`
 620 
 621   When indices is a 1D tensor, this operation is equivalent to
 622   `tf.scatter_update`.
 623 
 624   To avoid this operation there would be 2 alternatives:
 625   1) Reshaping the variable by merging the first `ndims` dimensions. However,
 626      this is not possible because `tf.reshape` returns a Tensor, which we
 627      cannot use `tf.scatter_update` on.
 628   2) Looping over the first `ndims` of the variable and using
 629      `tf.scatter_update` on the subtensors that result of slicing the first
 630      dimension. This is a valid option for `ndims = 1`, but less efficient than
 631      this implementation.
 632 
 633   See also `tf.scatter_update` and `tf.scatter_nd_update`.
 634 
 635   Args:
 636     ref: `Variable` to scatter onto.
 637     indices: Tensor containing indices as described above.
 638     updates: Tensor of updates to apply to `ref`.
 639     use_locking: Boolean indicating whether to lock the writing operation.
 640     name: Optional scope name string.
 641 
 642   Returns:
 643     Ref to `variable` after it has been modified.
 644 
 645   Raises:
 646     ValueError: If the initial `ndims` of `ref`, `indices`, and `updates` are
 647         not the same.
 648   """
 649   with ops.name_scope(name):
 650     indices = ops.convert_to_tensor(indices, name="indices")
 651     indices_shape = array_ops.shape(indices)
 652     indices_dimensions = indices.get_shape().ndims
 653 
 654     if indices_dimensions is None:
 655       raise ValueError("batch_gather does not allow indices with unknown "
 656                        "shape.")
 657 
 658     nd_indices = array_ops.expand_dims(indices, axis=-1)
 659     nd_indices_list = []
 660 
 661     # Scatter ND requires indices to have an additional dimension, in which the
 662     # coordinates of the updated things are specified. For this to be adapted to
 663     # the scatter_update with several leading dimensions, we simply make use of
 664     # a tf.range for all the leading dimensions followed by concat of all the
 665     # coordinates we created with the original indices.
 666 
 667     # For example if indices.shape = [2, 3, 4], we should generate the following
 668     # indices for tf.scatter_nd_update:
 669     # nd_indices[:, :, 0] = [[0, 0, 0], [1, 1, 1]]
 670     # nd_indices[:, :, 1] = [[0, 1, 2], [0, 1, 2]]
 671     # nd_indices[:, :, 2] = indices
 672     for dimension in range(indices_dimensions - 1):
 673       # In this loop we generate the following for the example (one for each
 674       # iteration).
 675       # nd_indices[:, :, 0] = [[0, 0, 0], [1, 1, 1]]
 676       # nd_indices[:, :, 1] = [[0, 1, 2], [0, 1, 2]]
 677       # This is done at every iteration with a tf.range over the size of the
 678       # i-th dimension and using broadcasting over the desired shape.
 679       dimension_size = indices_shape[dimension]
 680       shape_to_broadcast = [1] * (indices_dimensions + 1)
 681       shape_to_broadcast[dimension] = dimension_size
 682       dimension_range = array_ops.reshape(
 683           gen_math_ops._range(0, dimension_size, 1), shape_to_broadcast)
 684       if dimension_range.dtype.base_dtype != nd_indices.dtype:
 685         dimension_range = gen_math_ops.cast(dimension_range, nd_indices.dtype)
 686       nd_indices_list.append(
 687           dimension_range * array_ops.ones_like(nd_indices))
 688     # Add the original indices at the end, as described above, and concat.
 689     nd_indices_list.append(nd_indices)
 690     final_indices = array_ops.concat(nd_indices_list, axis=-1)
 691     return scatter_nd_update(
 692         ref, final_indices, updates, use_locking=use_locking)
