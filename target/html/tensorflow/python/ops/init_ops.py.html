  52 @tf_export("keras.initializers.Initializer")
  53 class Initializer(object):
  54   """Initializer base class: all initializers inherit from this class.
  55   """
  56 
  57   def __call__(self, shape, dtype=None, partition_info=None):
  58     raise NotImplementedError
  59 
  60   def get_config(self):
  61     """Returns the configuration of the initializer as a JSON-serializable dict.
  62 
  63     Returns:
  64       A JSON-serializable Python dict.
  65     """
  66     return {}
  67 
  68   @classmethod
  69   def from_config(cls, config):
  70     """Instantiates an initializer from a configuration dictionary.
  71 
  72     Example:
  73 
  74     ```python
  75     initializer = RandomUniform(-1, 1)
  76     config = initializer.get_config()
  77     initializer = RandomUniform.from_config(config)
  78     ```
  79 
  80     Args:
  81       config: A Python dictionary.
  82         It will typically be the output of `get_config`.
  83 
  84     Returns:
  85       An Initializer instance.
  86     """
  87     return cls(**config)

 124 @tf_export("keras.initializers.Constant", "initializers.constant",
 125            "constant_initializer", "keras.initializers.constant")
 126 class Constant(Initializer):
 127   """Initializer that generates tensors with constant values.
 128 
 129   The resulting tensor is populated with values of type `dtype`, as
 130   specified by arguments `value` following the desired `shape` of the
 131   new tensor (see examples below).
 132 
 133   The argument `value` can be a constant value, or a list of values of type
 134   `dtype`. If `value` is a list, then the length of the list must be less
 135   than or equal to the number of elements implied by the desired shape of the
 136   tensor. In the case where the total number of elements in `value` is less
 137   than the number of elements required by the tensor shape, the last element
 138   in `value` will be used to fill the remaining entries. If the total number of
 139   elements in `value` is greater than the number of elements required by the
 140   tensor shape, the initializer will raise a `ValueError`.
 141 
 142   Args:
 143     value: A Python scalar, list or tuple of values, or a N-dimensional numpy
 144       array. All elements of the initialized variable will be set to the
 145       corresponding value in the `value` argument.
 146     dtype: The data type.
 147     verify_shape: Boolean that enables verification of the shape of `value`. If
 148       `True`, the initializer will throw an error if the shape of `value` is not
 149       compatible with the shape of the initialized tensor.
 150 
 151   Raises:
 152     TypeError: If the input `value` is not one of the expected types.
 153 
 154   Examples:
 155     The following example can be rewritten using a numpy.ndarray instead
 156     of the `value` list, even reshaped, as shown in the two commented lines
 157     below the `value` list initialization.
 158 
 159   ```python
 160     >>> import numpy as np
 161     >>> import tensorflow as tf
 162 
 163     >>> value = [0, 1, 2, 3, 4, 5, 6, 7]
 164     >>> # value = np.array(value)
 165     >>> # value = value.reshape([2, 4])
 166     >>> init = tf.constant_initializer(value)
 167 
 168     >>> print('fitting shape:')
 169     >>> with tf.Session():
 170     >>>   x = tf.get_variable('x', shape=[2, 4], initializer=init)
 171     >>>   x.initializer.run()
 172     >>>   print(x.eval())
 173 
 174     fitting shape:
 175     [[ 0.  1.  2.  3.]
 176      [ 4.  5.  6.  7.]]
 177 
 178     >>> print('larger shape:')
 179     >>> with tf.Session():
 180     >>>   x = tf.get_variable('x', shape=[3, 4], initializer=init)
 181     >>>   x.initializer.run()
 182     >>>   print(x.eval())
 183 
 184     larger shape:
 185     [[ 0.  1.  2.  3.]
 186      [ 4.  5.  6.  7.]
 187      [ 7.  7.  7.  7.]]
 188 
 189     >>> print('smaller shape:')
 190     >>> with tf.Session():
 191     >>>   x = tf.get_variable('x', shape=[2, 3], initializer=init)
 192 
 193     ValueError: Too many elements provided. Needed at most 6, but received 8
 194 
 195     >>> print('shape verification:')
 196     >>> init_verify = tf.constant_initializer(value, verify_shape=True)
 197     >>> with tf.Session():
 198     >>>   x = tf.get_variable('x', shape=[3, 4], initializer=init_verify)
 199 
 200     TypeError: Expected Tensor's shape: (3, 4), got (8,).
 201   ```
 202   """
 203 
 204   def __init__(self, value=0, dtype=dtypes.float32, verify_shape=False):
 205     if not (np.isscalar(value) or isinstance(value, (list, tuple, np.ndarray))):
 206       raise TypeError(
 207           "Invalid type for initial value: %s (expected Python scalar, list or "
 208           "tuple of values, or numpy.ndarray)." % type(value))
 209 
 210     self.value = value
 211     self.dtype = dtypes.as_dtype(dtype)
 212     self._verify_shape = verify_shape
 213 
 214   def __call__(self, shape, dtype=None, partition_info=None, verify_shape=None):
 215     if dtype is None:
 216       dtype = self.dtype
 217     if verify_shape is None:
 218       verify_shape = self._verify_shape
 219     return constant_op.constant(
 220         self.value, dtype=dtype, shape=shape, verify_shape=verify_shape)
 221 
 222   def get_config(self):
 223     # We don't include `verify_shape` for compatibility with Keras.
 224     # `verify_shape` should be passed as an argument to `__call__` rather
 225     # than as a constructor argument: conceptually it isn't a property
 226     # of the initializer.
 227     return {"value": self.value, "dtype": self.dtype.name}

 343 @tf_export("initializers.uniform_unit_scaling",
 344            "uniform_unit_scaling_initializer")
 345 @deprecation.deprecated_endpoints("uniform_unit_scaling_initializer")
 346 class UniformUnitScaling(Initializer):
 347   """Initializer that generates tensors without scaling variance.
 348 
 349   When initializing a deep network, it is in principle advantageous to keep
 350   the scale of the input variance constant, so it does not explode or diminish
 351   by reaching the final layer. If the input is `x` and the operation `x * W`,
 352   and we want to initialize `W` uniformly at random, we need to pick `W` from
 353 
 354       [-sqrt(3) / sqrt(dim), sqrt(3) / sqrt(dim)]
 355 
 356   to keep the scale intact, where `dim = W.shape[0]` (the size of the input).
 357   A similar calculation for convolutional networks gives an analogous result
 358   with `dim` equal to the product of the first 3 dimensions.  When
 359   nonlinearities are present, we need to multiply this by a constant `factor`.
 360   See [Sussillo et al., 2014](https://arxiv.org/abs/1412.6558)
 361   ([pdf](http://arxiv.org/pdf/1412.6558.pdf)) for deeper motivation, experiments
 362   and the calculation of constants. In section 2.3 there, the constants were
 363   numerically computed: for a linear layer it's 1.0, relu: ~1.43, tanh: ~1.15.
 364 
 365   Args:
 366     factor: Float.  A multiplicative factor by which the values will be scaled.
 367     seed: A Python integer. Used to create random seeds. See
 368       `tf.set_random_seed`
 369       for behavior.
 370     dtype: The data type. Only floating point types are supported.
 371   """
 372 
 373   @deprecated(None,
 374               "Use tf.initializers.variance_scaling instead with distribution="
 375               "uniform to get equivalent behavior.")
 376   def __init__(self, factor=1.0, seed=None, dtype=dtypes.float32):
 377     self.factor = factor
 378     self.seed = seed
 379     self.dtype = _assert_float_dtype(dtypes.as_dtype(dtype))
 380 
 381   def __call__(self, shape, dtype=None, partition_info=None):
 382     if dtype is None:
 383       dtype = self.dtype
 384     scale_shape = shape
 385     if partition_info is not None:
 386       scale_shape = partition_info.full_shape
 387 
 388     input_size = 1.0
 389     # Estimating input size is not possible to do perfectly, but we try.
 390     # The estimate, obtained by multiplying all dimensions but the last one,
 391     # is the right thing for matrix multiply and convolutions (see above).
 392     for dim in scale_shape[:-1]:
 393       input_size *= float(dim)
 394     # Avoid errors when initializing zero-size tensors.
 395     input_size = max(input_size, 1.0)
 396     max_val = math.sqrt(3 / input_size) * self.factor
 397     return random_ops.random_uniform(
 398         shape, -max_val, max_val, dtype, seed=self.seed)
 399 
 400   def get_config(self):
 401     return {"factor": self.factor, "seed": self.seed, "dtype": self.dtype.name}

 404 @tf_export("keras.initializers.VarianceScaling",
 405            "initializers.variance_scaling", "variance_scaling_initializer")
 406 @deprecation.deprecated_endpoints("variance_scaling_initializer")
 407 class VarianceScaling(Initializer):
 408   """Initializer capable of adapting its scale to the shape of weights tensors.
 409 
 410   With `distribution="truncated_normal" or "untruncated_normal"`,
 411   samples are drawn from a truncated/untruncated normal
 412   distribution with a mean of zero and a standard deviation (after truncation,
 413   if used) `stddev = sqrt(scale / n)`
 414   where n is:
 415     - number of input units in the weight tensor, if mode = "fan_in"
 416     - number of output units, if mode = "fan_out"
 417     - average of the numbers of input and output units, if mode = "fan_avg"
 418 
 419   With `distribution="uniform"`, samples are drawn from a uniform distribution
 420   within [-limit, limit], with `limit = sqrt(3 * scale / n)`.
 421 
 422   Args:
 423     scale: Scaling factor (positive float).
 424     mode: One of "fan_in", "fan_out", "fan_avg".
 425     distribution: Random distribution to use. One of "normal", "uniform".
 426     seed: A Python integer. Used to create random seeds. See
 427       `tf.set_random_seed`
 428       for behavior.
 429     dtype: The data type. Only floating point types are supported.
 430 
 431   Raises:
 432     ValueError: In case of an invalid value for the "scale", mode" or
 433       "distribution" arguments.
 434   """
 435 
 436   @deprecated_arg_values(
 437       None,
 438       "`normal` is a deprecated alias for `truncated_normal`",
 439       distribution="normal")
 440   def __init__(self,
 441                scale=1.0,
 442                mode="fan_in",
 443                distribution="truncated_normal",
 444                seed=None,
 445                dtype=dtypes.float32):
 446     if scale <= 0.:
 447       raise ValueError("`scale` must be positive float.")
 448     if mode not in {"fan_in", "fan_out", "fan_avg"}:
 449       raise ValueError("Invalid `mode` argument:", mode)
 450     distribution = distribution.lower()
 451     if distribution not in {"normal", "uniform",
 452                             "truncated_normal", "untruncated_normal"}:
 453       raise ValueError("Invalid `distribution` argument:", distribution)
 454     self.scale = scale
 455     self.mode = mode
 456     self.distribution = distribution
 457     self.seed = seed
 458     self.dtype = _assert_float_dtype(dtypes.as_dtype(dtype))
 459 
 460   def __call__(self, shape, dtype=None, partition_info=None):
 461     if dtype is None:
 462       dtype = self.dtype
 463     scale = self.scale
 464     scale_shape = shape
 465     if partition_info is not None:
 466       scale_shape = partition_info.full_shape
 467     fan_in, fan_out = _compute_fans(scale_shape)
 468     if self.mode == "fan_in":
 469       scale /= max(1., fan_in)
 470     elif self.mode == "fan_out":
 471       scale /= max(1., fan_out)
 472     else:
 473       scale /= max(1., (fan_in + fan_out) / 2.)
 474     if self.distribution == "normal" or self.distribution == "truncated_normal":
 475       # constant taken from scipy.stats.truncnorm.std(a=-2, b=2, loc=0., scale=1.)
 476       stddev = math.sqrt(scale) / .87962566103423978
 477       return random_ops.truncated_normal(
 478           shape, 0.0, stddev, dtype, seed=self.seed)
 479     elif self.distribution == "untruncated_normal":
 480       stddev = math.sqrt(scale)
 481       return random_ops.random_normal(
 482           shape, 0.0, stddev, dtype, seed=self.seed)
 483     else:
 484       limit = math.sqrt(3.0 * scale)
 485       return random_ops.random_uniform(
 486           shape, -limit, limit, dtype, seed=self.seed)
 487 
 488   def get_config(self):
 489     return {
 490         "scale": self.scale,
 491         "mode": self.mode,
 492         "distribution": self.distribution,
 493         "seed": self.seed,
 494         "dtype": self.dtype.name
 495     }

 498 @tf_export("keras.initializers.Orthogonal", "initializers.orthogonal",
 499            "orthogonal_initializer", "keras.initializers.orthogonal")
 500 @deprecation.deprecated_endpoints("orthogonal_initializer")
 501 class Orthogonal(Initializer):
 502   """Initializer that generates an orthogonal matrix.
 503 
 504   If the shape of the tensor to initialize is two-dimensional, it is initialized
 505   with an orthogonal matrix obtained from the QR decomposition of a matrix of
 506   random numbers drawn from a normal distribution.
 507   If the matrix has fewer rows than columns then the output will have orthogonal
 508   rows. Otherwise, the output will have orthogonal columns.
 509 
 510   If the shape of the tensor to initialize is more than two-dimensional,
 511   a matrix of shape `(shape[0] * ... * shape[n - 2], shape[n - 1])`
 512   is initialized, where `n` is the length of the shape vector.
 513   The matrix is subsequently reshaped to give a tensor of the desired shape.
 514 
 515   Args:
 516     gain: multiplicative factor to apply to the orthogonal matrix
 517     seed: A Python integer. Used to create random seeds. See
 518       `tf.set_random_seed`
 519       for behavior.
 520     dtype: The data type.
 521   """
 522 
 523   def __init__(self, gain=1.0, seed=None, dtype=dtypes.float32):
 524     self.gain = gain
 525     self.dtype = _assert_float_dtype(dtypes.as_dtype(dtype))
 526     self.seed = seed
 527 
 528   def __call__(self, shape, dtype=None, partition_info=None):
 529     if dtype is None:
 530       dtype = self.dtype
 531     # Check the shape
 532     if len(shape) < 2:
 533       raise ValueError("The tensor to initialize must be "
 534                        "at least two-dimensional")
 535     # Flatten the input shape with the last dimension remaining
 536     # its original shape so it works for conv2d
 537     num_rows = 1
 538     for dim in shape[:-1]:
 539       num_rows *= dim
 540     num_cols = shape[-1]
 541     flat_shape = (num_cols, num_rows) if num_rows < num_cols else (num_rows,
 542                                                                    num_cols)
 543 
 544     # Generate a random matrix
 545     a = random_ops.random_normal(flat_shape, dtype=dtype, seed=self.seed)
 546     # Compute the qr factorization
 547     q, r = gen_linalg_ops.qr(a, full_matrices=False)
 548     # Make Q uniform
 549     d = array_ops.diag_part(r)
 550     q *= math_ops.sign(d)
 551     if num_rows < num_cols:
 552       q = array_ops.matrix_transpose(q)
 553     return self.gain * array_ops.reshape(q, shape)
 554 
 555   def get_config(self):
 556     return {"gain": self.gain, "seed": self.seed, "dtype": self.dtype.name}

 559 class ConvolutionDeltaOrthogonal(Initializer):
 560   """Initializer that generates a delta orthogonal kernel for ConvNets.
 561 
 562   The shape of the tensor must have length 3, 4 or 5. The number of input
 563   filters must not exceed the number of output filters. The center pixels of the
 564   tensor form an orthogonal matrix. Other pixels are set to be zero. See
 565   algorithm 2 in [Xiao et al., 2018]: https://arxiv.org/abs/1806.05393
 566 
 567 
 568   Args:
 569     gain: Multiplicative factor to apply to the orthogonal matrix. Default is 1.
 570       The 2-norm of an input is multiplied by a factor of 'sqrt(gain)' after
 571       applying this convolution.
 572     seed: A Python integer. Used to create random seeds. See
 573       `tf.set_random_seed` for behavior.
 574     dtype: The data type.
 575   """
 576 
 577   def __init__(self, gain=1.0, seed=None, dtype=dtypes.float32):
 578     self.gain = gain
 579     self.dtype = _assert_float_dtype(dtypes.as_dtype(dtype))
 580     self.seed = seed
 581 
 582   def __call__(self, shape, dtype=None, partition_info=None):
 583     if dtype is None:
 584       dtype = self.dtype
 585     # Check the shape
 586     if len(shape) < 3 or len(shape) > 5:
 587       raise ValueError("The tensor to initialize must be at least "
 588                        "three-dimensional and at most five-dimensional")
 589 
 590     if shape[-2] > shape[-1]:
 591       raise ValueError("In_filters cannot be greater than out_filters.")
 592 
 593     # Generate a random matrix
 594     a = random_ops.random_normal([shape[-1], shape[-1]],
 595                                  dtype=dtype, seed=self.seed)
 596     # Compute the qr factorization
 597     q, r = gen_linalg_ops.qr(a, full_matrices=False)
 598     # Make Q uniform
 599     d = array_ops.diag_part(r)
 600     q *= math_ops.sign(d)
 601     q = q[:shape[-2], :]
 602     q *= math_ops.sqrt(math_ops.cast(self.gain, dtype=dtype))
 603     if len(shape) == 3:
 604       weight = array_ops.scatter_nd([[(shape[0]-1)//2]],
 605                                     array_ops.expand_dims(q, 0), shape)
 606     elif len(shape) == 4:
 607       weight = array_ops.scatter_nd([[(shape[0]-1)//2, (shape[1]-1)//2]],
 608                                     array_ops.expand_dims(q, 0), shape)
 609     else:
 610       weight = array_ops.scatter_nd([[(shape[0]-1)//2, (shape[1]-1)//2,
 611                                       (shape[2]-1)//2]],
 612                                     array_ops.expand_dims(q, 0), shape)
 613     return weight

 645   def _orthogonal_matrix(self, n):
 646     """Construct an n x n orthogonal matrix.
 647 
 648     Args:
 649       n: Dimension.
 650     Returns:
 651       A n x n orthogonal matrix.
 652     """
 653     a = random_ops.random_normal([n, n], dtype=self.dtype, seed=self.seed)
 654     if self.seed:
 655       self.seed += 1
 656     q, r = gen_linalg_ops.qr(a)
 657     d = array_ops.diag_part(r)
 658     # make q uniform
 659     q *= math_ops.sign(d)

 662   def _symmetric_projection(self, n):
 663     """Compute a n x n symmetric projection matrix.
 664

 666       n: Dimension.

 669     """
 670     q = self._orthogonal_matrix(n)
 671     # randomly zeroing out some columns
 672     mask = math_ops.cast(random_ops.random_normal([n], seed=self.seed) > 0,
 673                          self.dtype)
 674     if self.seed:
 675       self.seed += 1
 676     c = math_ops.multiply(q, mask)
 677     return math_ops.matmul(c, array_ops.matrix_transpose(c))

 728   def _block_orth(self, p1, p2):
 729     """Construct a 2 x 2 kernel. Used to construct orthgonal kernel.
 730 
 731     Args:
 732       p1: A symmetric projection matrix.
 733       p2: A symmetric projection matrix.
 734     Returns:
 735       A 2 x 2 kernel [[p1p2,         p1(1-p2)],
 736                       [(1-p1)p2, (1-p1)(1-p2)]].
 737     Raises:
 738       ValueError: If the dimensions of p1 and p2 are different.
 739     """
 740     if p1.shape.as_list() != p2.shape.as_list():
 741       raise ValueError("The dimension of the matrices must be the same.")
 742     n = p1.shape.as_list()[0]
 743     kernel2x2 = {}
 744     eye = linalg_ops_impl.eye(n, dtype=self.dtype)
 745     kernel2x2[0, 0] = math_ops.matmul(p1, p2)
 746     kernel2x2[0, 1] = math_ops.matmul(p1, (eye - p2))
 747     kernel2x2[1, 0] = math_ops.matmul((eye - p1), p2)
 748     kernel2x2[1, 1] = math_ops.matmul((eye - p1), (eye - p2))
 749 
 750     return kernel2x2

 752   def _matrix_conv(self, m1, m2):
 753     """Matrix convolution.
 754 
 755     Args:
 756       m1: A k x k dictionary, each element is a n x n matrix.
 757       m2: A l x l dictionary, each element is a n x n matrix.
 758 
 759     Returns:
 760       (k + l - 1) * (k + l - 1) dictionary each element is a n x n matrix.
 761     Raises:
 762       ValueError: if the entries of m1 and m2 are of different dimensions.
 763     """
 764 
 765     n = (m1[0, 0]).shape.as_list()[0]
 766     if n != (m2[0, 0]).shape.as_list()[0]:
 767       raise ValueError("The entries in matrices m1 and m2 "
 768                        "must have the same dimensions!")
 769     k = int(np.sqrt(len(m1)))
 770     l = int(np.sqrt(len(m2)))
 771     result = {}
 772     size = k + l - 1
 773     # Compute matrix convolution between m1 and m2.
 774     for i in range(size):
 775       for j in range(size):
 776         result[i, j] = array_ops.zeros([n, n], self.dtype)
 777         for index1 in range(min(k, i + 1)):
 778           for index2 in range(min(k, j + 1)):
 779             if (i - index1) < l and (j - index2) < l:
 780               result[i, j] += math_ops.matmul(m1[index1, index2],
 781                                               m2[i - index1, j - index2])
 782     return result

 784   def _orthogonal_kernel(self, ksize, cin, cout):
 785     """Construct orthogonal kernel for convolution.
 786 
 787     Args:
 788       ksize: Kernel size.
 789       cin: Number of input channels.
 790       cout: Number of output channels.
 791     Returns:
 792       An [ksize, ksize, cin, cout] orthogonal kernel.
 793     Raises:
 794       ValueError: If cin > cout.
 795     """
 796     if cin > cout:
 797       raise ValueError("The number of input channels cannot exceed "
 798                        "the number of output channels.")
 799     orth = self._orthogonal_matrix(cout)[0:cin, :]
 800     if ksize == 1:
 801       return array_ops.expand_dims(array_ops.expand_dims(orth, 0), 0)
 802 
 803     p = self._block_orth(self._symmetric_projection(cout),
 804                          self._symmetric_projection(cout))
 805     for _ in range(ksize - 2):
 806       temp = self._block_orth(self._symmetric_projection(cout),

 810       for j in range(ksize):
 811         p[i, j] = math_ops.matmul(orth, p[i, j])

1303 def _compute_fans(shape):
1304   """Computes the number of input and output units for a weight shape.
1305 
1306   Args:
1307     shape: Integer shape tuple or TF tensor shape.
1308 
1309   Returns:
1310     A tuple of scalars (fan_in, fan_out).
1311   """
1312   if len(shape) < 1:  # Just to avoid errors for constants.
1313     fan_in = fan_out = 1
1314   elif len(shape) == 1:
1315     fan_in = fan_out = shape[0]
1316   elif len(shape) == 2:
1317     fan_in = shape[0]
1318     fan_out = shape[1]
1319   else:
1320     # Assuming convolution kernels (2D, 3D, or more).
1321     # kernel shape: (..., input_depth, depth)
1322     receptive_field_size = 1.
1323     for dim in shape[:-2]:
1324       receptive_field_size *= dim
1325     fan_in = shape[-2] * receptive_field_size
1326     fan_out = shape[-1] * receptive_field_size
1327   return fan_in, fan_out
