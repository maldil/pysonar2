  51 def _non_atrous_convolution(
  52     input,  # pylint: disable=redefined-builtin
  53     filter,  # pylint: disable=redefined-builtin
  54     padding,
  55     data_format=None,  # pylint: disable=redefined-builtin
  56     strides=None,
  57     name=None):
  58   """Computes sums of N-D convolutions (actually cross correlation).
  59 
  60   It is required that 1 <= N <= 3.
  61 
  62   This is used to implement the more generic `convolution` function, which
  63   extends the interface of this function with a `dilation_rate` parameter.
  64 
  65   Args:
  66 
  67     input: Rank N+2 tensor of type T of shape
  68       `[batch_size] + input_spatial_shape + [in_channels]` if `data_format`
  69       does not start with `"NC"`, or
  70       `[batch_size, in_channels] + input_spatial_shape` if `data_format` starts
  71       with `"NC"`.
  72     filter: Rank N+2 tensor of type T of shape
  73       `filter_spatial_shape + [in_channels, out_channels]`.  Rank of either
  74       `input` or `filter` must be known.
  75     padding: Padding method to use, must be either "VALID" or "SAME".
  76     data_format: A string or None.  Specifies whether the channel dimension of
  77       the `input` and output is the last dimension (default, or if `data_format`
  78       does not start with "NC"), or the second dimension (if `data_format`
  79       starts with "NC").  For N=1, the valid values are "NWC" (default) and
  80       "NCW".  For N=2, the valid values are "NHWC" (default) and "NCHW".
  81       For N=3, the valid values are "NDHWC" (default) and "NCDHW".
  82     strides: Sequence of N positive integers, defaults to `[1] * N`.
  83     name: Name prefix to use.
  84 
  85   Returns:
  86     Rank N+2 tensor of type T of shape
  87     `[batch_size] + output_spatial_shape + [out_channels]`, where
  88     if padding == "SAME":
  89       output_spatial_shape = input_spatial_shape
  90     if padding == "VALID":
  91       output_spatial_shape = input_spatial_shape - filter_spatial_shape + 1.
  92 
  93   Raises:
  94     ValueError: if ranks are incompatible.
  95 
  96   """
  97   with ops.name_scope(name, "non_atrous_convolution", [input, filter]) as scope:
  98     input = ops.convert_to_tensor(input, name="input")  # pylint: disable=redefined-builtin
  99     input_shape = input.get_shape()
 100     filter = ops.convert_to_tensor(filter, name="filter")  # pylint: disable=redefined-builtin
 101     filter_shape = filter.get_shape()
 102     op = _NonAtrousConvolution(
 103         input_shape,
 104         filter_shape=filter_shape,
 105         padding=padding,
 106         data_format=data_format,
 107         strides=strides,
 108         name=scope)
 109     return op(input, filter)

 112 class _NonAtrousConvolution(object):
 113   """Helper class for _non_atrous_convolution.
 114 
 115   Note that this class assumes that shapes of input and filter passed to
 116   __call__ are compatible with input_shape and filter_shape passed to the
 117   constructor.
 118 
 119   Arguments:
 120     input_shape: static input shape, i.e. input.get_shape().
 121     filter_shape: static filter shape, i.e. filter.get_shape().
 122     padding: see _non_atrous_convolution.
 123     data_format: see _non_atrous_convolution.
 124     strides: see _non_atrous_convolution.
 125     name: see _non_atrous_convolution.
 126   """
 127 
 128   def __init__(
 129       self,
 130       input_shape,
 131       filter_shape,  # pylint: disable=redefined-builtin
 132       padding,
 133       data_format=None,
 134       strides=None,
 135       name=None):
 136     filter_shape = filter_shape.with_rank(input_shape.ndims)
 137     self.padding = padding
 138     self.name = name
 139     input_shape = input_shape.with_rank(filter_shape.ndims)
 140     if input_shape.ndims is None:
 141       raise ValueError("Rank of convolution must be known")
 142     if input_shape.ndims < 3 or input_shape.ndims > 5:
 143       raise ValueError(
 144           "`input` and `filter` must have rank at least 3 and at most 5")
 145     conv_dims = input_shape.ndims - 2
 146     if strides is None:
 147       strides = [1] * conv_dims
 148     elif len(strides) != conv_dims:
 149       raise ValueError("len(strides)=%d, but should be %d" % (len(strides),
 150                                                               conv_dims))
 151     if conv_dims == 1:
 152       # conv1d uses the 2-d data format names
 153       if data_format is None:
 154         data_format = "NWC"
 155       elif data_format not in {"NCW", "NWC", "NCHW", "NHWC"}:
 156         raise ValueError("data_format must be \"NWC\" or \"NCW\".")
 157       self.strides = strides[0]
 158       self.data_format = data_format
 159       self.conv_op = self._conv1d
 160     elif conv_dims == 2:
 161       if data_format is None or data_format == "NHWC":
 162         data_format = "NHWC"
 163         strides = [1] + list(strides) + [1]
 164       elif data_format == "NCHW":
 165         strides = [1, 1] + list(strides)
 166       else:
 167         raise ValueError("data_format must be \"NHWC\" or \"NCHW\".")
 168       self.strides = strides
 169       self.data_format = data_format
 170       self.conv_op = gen_nn_ops.conv2d
 171     elif conv_dims == 3:
 172       if data_format is None or data_format == "NDHWC":
 173         strides = [1] + list(strides) + [1]
 174       elif data_format == "NCDHW":
 175         strides = [1, 1] + list(strides)
 176       else:
 177         raise ValueError("data_format must be \"NDHWC\" or \"NCDHW\". Have: %s"
 178                          % data_format)
 179       self.strides = strides
 180       self.data_format = data_format
 181       self.conv_op = gen_nn_ops.conv3d
 182 
 183   # Note that we need this adapter since argument names for conv1d don't match
 184   # those for gen_nn_ops.conv2d and gen_nn_ops.conv3d.
 185   # pylint: disable=redefined-builtin
 186   def _conv1d(self, input, filter, strides, padding, data_format, name):
 187     return conv1d(
 188         value=input,
 189         filters=filter,
 190         stride=strides,
 191         padding=padding,
 192         data_format=data_format,
 193         name=name)
 194 
 195   # pylint: enable=redefined-builtin
 196 
 197   def __call__(self, inp, filter):  # pylint: disable=redefined-builtin
 198     return self.conv_op(
 199         input=inp,
 200         filter=filter,
 201         strides=self.strides,
 202         padding=self.padding,
 203         data_format=self.data_format,
 204         name=self.name)

 207 @tf_export("nn.with_space_to_batch")
 208 def with_space_to_batch(
 209     input,  # pylint: disable=redefined-builtin
 210     dilation_rate,
 211     padding,
 212     op,
 213     filter_shape=None,
 214     spatial_dims=None,
 215     data_format=None):
 216   """Performs `op` on the space-to-batch representation of `input`.
 217 
 218   This has the effect of transforming sliding window operations into the
 219   corresponding "atrous" operation in which the input is sampled at the
 220   specified `dilation_rate`.
 221 
 222   In the special case that `dilation_rate` is uniformly 1, this simply returns:
 223 
 224     op(input, num_spatial_dims, padding)
 225 
 226   Otherwise, it returns:
 227 
 228     batch_to_space_nd(
 229       op(space_to_batch_nd(input, adjusted_dilation_rate, adjusted_paddings),
 230          num_spatial_dims,
 231          "VALID")
 232       adjusted_dilation_rate,
 233       adjusted_crops),
 234 
 235   where:
 236 
 237     adjusted_dilation_rate is an int64 tensor of shape [max(spatial_dims)],
 238     adjusted_{paddings,crops} are int64 tensors of shape [max(spatial_dims), 2]
 239 
 240   defined as follows:
 241 
 242   We first define two int64 tensors `paddings` and `crops` of shape
 243   `[num_spatial_dims, 2]` based on the value of `padding` and the spatial
 244   dimensions of the `input`:
 245 
 246   If `padding = "VALID"`, then:
 247 
 248     paddings, crops = required_space_to_batch_paddings(
 249       input_shape[spatial_dims],
 250       dilation_rate)
 251 
 252   If `padding = "SAME"`, then:
 253 
 254     dilated_filter_shape =
 255       filter_shape + (filter_shape - 1) * (dilation_rate - 1)
 256 
 257     paddings, crops = required_space_to_batch_paddings(
 258       input_shape[spatial_dims],
 259       dilation_rate,
 260       [(dilated_filter_shape - 1) // 2,
 261        dilated_filter_shape - 1 - (dilated_filter_shape - 1) // 2])
 262 
 263   Because `space_to_batch_nd` and `batch_to_space_nd` assume that the spatial
 264   dimensions are contiguous starting at the second dimension, but the specified
 265   `spatial_dims` may not be, we must adjust `dilation_rate`, `paddings` and
 266   `crops` in order to be usable with these operations.  For a given dimension,
 267   if the block size is 1, and both the starting and ending padding and crop
 268   amounts are 0, then space_to_batch_nd effectively leaves that dimension alone,
 269   which is what is needed for dimensions not part of `spatial_dims`.
 270   Furthermore, `space_to_batch_nd` and `batch_to_space_nd` handle this case
 271   efficiently for any number of leading and trailing dimensions.
 272 
 273   For 0 <= i < len(spatial_dims), we assign:
 274 
 275     adjusted_dilation_rate[spatial_dims[i] - 1] = dilation_rate[i]
 276     adjusted_paddings[spatial_dims[i] - 1, :] = paddings[i, :]
 277     adjusted_crops[spatial_dims[i] - 1, :] = crops[i, :]
 278 
 279   All unassigned values of `adjusted_dilation_rate` default to 1, while all
 280   unassigned values of `adjusted_paddings` and `adjusted_crops` default to 0.
 281 
 282   Note in the case that `dilation_rate` is not uniformly 1, specifying "VALID"
 283   padding is equivalent to specifying `padding = "SAME"` with a filter_shape of
 284   `[1]*N`.
 285 
 286   Advanced usage. Note the following optimization: A sequence of
 287   `with_space_to_batch` operations with identical (not uniformly 1)
 288   `dilation_rate` parameters and "VALID" padding
 289 
 290     net = with_space_to_batch(net, dilation_rate, "VALID", op_1)
 291     ...
 292     net = with_space_to_batch(net, dilation_rate, "VALID", op_k)
 293 
 294   can be combined into a single `with_space_to_batch` operation as follows:
 295 
 296     def combined_op(converted_input, num_spatial_dims, _):
 297       result = op_1(converted_input, num_spatial_dims, "VALID")
 298       ...
 299       result = op_k(result, num_spatial_dims, "VALID")
 300 
 301     net = with_space_to_batch(net, dilation_rate, "VALID", combined_op)
 302 
 303   This eliminates the overhead of `k-1` calls to `space_to_batch_nd` and
 304   `batch_to_space_nd`.
 305 
 306   Similarly, a sequence of `with_space_to_batch` operations with identical (not
 307   uniformly 1) `dilation_rate` parameters, "SAME" padding, and odd filter
 308   dimensions
 309 
 310     net = with_space_to_batch(net, dilation_rate, "SAME", op_1, filter_shape_1)
 311     ...
 312     net = with_space_to_batch(net, dilation_rate, "SAME", op_k, filter_shape_k)
 313 
 314   can be combined into a single `with_space_to_batch` operation as follows:
 315 
 316     def combined_op(converted_input, num_spatial_dims, _):
 317       result = op_1(converted_input, num_spatial_dims, "SAME")
 318       ...
 319       result = op_k(result, num_spatial_dims, "SAME")
 320 
 321     net = with_space_to_batch(net, dilation_rate, "VALID", combined_op)
 322 
 323   Args:
 324     input: Tensor of rank > max(spatial_dims).
 325     dilation_rate: int32 Tensor of *known* shape [num_spatial_dims].
 326     padding: str constant equal to "VALID" or "SAME"
 327     op: Function that maps (input, num_spatial_dims, padding) -> output
 328     filter_shape: If padding = "SAME", specifies the shape of the convolution
 329       kernel/pooling window as an integer Tensor of shape [>=num_spatial_dims].
 330       If padding = "VALID", filter_shape is ignored and need not be specified.
 331     spatial_dims: Monotonically increasing sequence of `num_spatial_dims`
 332       integers (which are >= 1) specifying the spatial dimensions of `input`
 333       and output.  Defaults to: `range(1, num_spatial_dims+1)`.
 334     data_format: A string or None.  Specifies whether the channel dimension of
 335       the `input` and output is the last dimension (default, or if `data_format`
 336       does not start with "NC"), or the second dimension (if `data_format`
 337       starts with "NC").  For N=1, the valid values are "NWC" (default) and
 338       "NCW".  For N=2, the valid values are "NHWC" (default) and "NCHW".
 339       For N=3, the valid values are "NDHWC" (default) and "NCDHW".
 340 
 341   Returns:
 342     The output Tensor as described above, dimensions will vary based on the op
 343     provided.
 344 
 345   Raises:
 346     ValueError: if `padding` is invalid or the arguments are incompatible.
 347     ValueError: if `spatial_dims` are invalid.
 348 
 349   """
 350   input = ops.convert_to_tensor(input, name="input")  # pylint: disable=redefined-builtin
 351   input_shape = input.get_shape()
 352 
 353   def build_op(num_spatial_dims, padding):
 354     return lambda inp, _: op(inp, num_spatial_dims, padding)
 355 
 356   new_op = _WithSpaceToBatch(
 357       input_shape,
 358       dilation_rate,
 359       padding,
 360       build_op,
 361       filter_shape=filter_shape,
 362       spatial_dims=spatial_dims,
 363       data_format=data_format)
 364   return new_op(input, None)

 367 class _WithSpaceToBatch(object):
 368   """Helper class for with_space_to_batch.
 369 
 370   Note that this class assumes that shapes of input and filter passed to
 371   __call__ are compatible with input_shape and filter_shape passed to the
 372   constructor.
 373 
 374   Arguments
 375     input_shape: static shape of input. i.e. input.get_shape().
 376     dilation_rate: see with_space_to_batch
 377     padding: see with_space_to_batch
 378     build_op: Function that maps (num_spatial_dims, paddings) -> (function that
 379       maps (input, filter) -> output).
 380     filter_shape: see with_space_to_batch
 381     spatial_dims: see with_space_to_batch
 382     data_format: see with_space_to_batch
 383   """
 384 
 385   def __init__(self,
 386                input_shape,
 387                dilation_rate,
 388                padding,
 389                build_op,
 390                filter_shape=None,
 391                spatial_dims=None,
 392                data_format=None):
 393     """Helper class for _with_space_to_batch."""
 394     dilation_rate = ops.convert_to_tensor(
 395         dilation_rate, dtypes.int32, name="dilation_rate")
 396     try:
 397       rate_shape = dilation_rate.get_shape().with_rank(1)
 398     except ValueError:
 399       raise ValueError("rate must be rank 1")
 400 
 401     if not dilation_rate.get_shape().is_fully_defined():
 402       raise ValueError("rate must have known shape")
 403 
 404     num_spatial_dims = rate_shape[0].value
 405 
 406     if data_format is not None and data_format.startswith("NC"):
 407       starting_spatial_dim = 2
 408     else:
 409       starting_spatial_dim = 1
 410 
 411     if spatial_dims is None:
 412       spatial_dims = range(starting_spatial_dim,
 413                            num_spatial_dims + starting_spatial_dim)
 414     orig_spatial_dims = list(spatial_dims)
 415     spatial_dims = sorted(set(int(x) for x in orig_spatial_dims))
 416     if spatial_dims != orig_spatial_dims or any(x < 1 for x in spatial_dims):
 417       raise ValueError(
 418           "spatial_dims must be a montonically increasing sequence of positive "
 419           "integers")  # pylint: disable=line-too-long
 420 
 421     if data_format is not None and data_format.startswith("NC"):
 422       expected_input_rank = spatial_dims[-1]
 423     else:
 424       expected_input_rank = spatial_dims[-1] + 1
 425 
 426     try:
 427       input_shape.with_rank_at_least(expected_input_rank)
 428     except ValueError:
 429       ValueError("input tensor must have rank %d at least" %
 430                  (expected_input_rank))
 431 
 432     const_rate = tensor_util.constant_value(dilation_rate)
 433     rate_or_const_rate = dilation_rate
 434     if const_rate is not None:
 435       rate_or_const_rate = const_rate
 436       if np.any(const_rate < 1):
 437         raise ValueError("dilation_rate must be positive")
 438       if np.all(const_rate == 1):
 439         self.call = build_op(num_spatial_dims, padding)
 440         return
 441 
 442     # We have two padding contributions. The first is used for converting "SAME"
 443     # to "VALID". The second is required so that the height and width of the
 444     # zero-padded value tensor are multiples of rate.
 445 
 446     # Padding required to reduce to "VALID" convolution
 447     if padding == "SAME":
 448       if filter_shape is None:
 449         raise ValueError("filter_shape must be specified for SAME padding")
 450       filter_shape = ops.convert_to_tensor(filter_shape, name="filter_shape")
 451       const_filter_shape = tensor_util.constant_value(filter_shape)
 452       if const_filter_shape is not None:
 453         filter_shape = const_filter_shape
 454         self.base_paddings = _with_space_to_batch_base_paddings(
 455             const_filter_shape, num_spatial_dims, rate_or_const_rate)
 456       else:
 457         self.num_spatial_dims = num_spatial_dims
 458         self.rate_or_const_rate = rate_or_const_rate
 459         self.base_paddings = None
 460     elif padding == "VALID":
 461       self.base_paddings = np.zeros([num_spatial_dims, 2], np.int32)
 462     else:
 463       raise ValueError("Invalid padding method %r" % padding)
 464 
 465     self.input_shape = input_shape
 466     self.spatial_dims = spatial_dims
 467     self.dilation_rate = dilation_rate
 468     self.data_format = data_format
 469     self.op = build_op(num_spatial_dims, "VALID")
 470     self.call = self._with_space_to_batch_call
 471 
 472   def _with_space_to_batch_call(self, inp, filter):  # pylint: disable=redefined-builtin
 473     """Call functionality for with_space_to_batch."""
 474     # Handle input whose shape is unknown during graph creation.
 475     input_spatial_shape = None
 476     input_shape = self.input_shape
 477     spatial_dims = self.spatial_dims
 478     if input_shape.ndims is not None:
 479       input_shape_list = input_shape.as_list()
 480       input_spatial_shape = [input_shape_list[i] for i in spatial_dims]
 481     if input_spatial_shape is None or None in input_spatial_shape:
 482       input_shape_tensor = array_ops.shape(inp)
 483       input_spatial_shape = array_ops.stack(
 484           [input_shape_tensor[i] for i in spatial_dims])
 485 
 486     base_paddings = self.base_paddings
 487     if base_paddings is None:
 488       # base_paddings could not be computed at build time since static filter
 489       # shape was not fully defined.
 490       filter_shape = array_ops.shape(filter)
 491       base_paddings = _with_space_to_batch_base_paddings(
 492           filter_shape, self.num_spatial_dims, self.rate_or_const_rate)
 493     paddings, crops = array_ops.required_space_to_batch_paddings(
 494         input_shape=input_spatial_shape,
 495         base_paddings=base_paddings,
 496         block_shape=self.dilation_rate)
 497 
 498     dilation_rate = _with_space_to_batch_adjust(self.dilation_rate, 1,
 499                                                 spatial_dims)
 500     paddings = _with_space_to_batch_adjust(paddings, 0, spatial_dims)
 501     crops = _with_space_to_batch_adjust(crops, 0, spatial_dims)
 502     input_converted = array_ops.space_to_batch_nd(
 503         input=inp, block_shape=dilation_rate, paddings=paddings)
 504 
 505     result = self.op(input_converted, filter)
 506 
 507     result_converted = array_ops.batch_to_space_nd(
 508         input=result, block_shape=dilation_rate, crops=crops)
 509 
 510     # Recover channel information for output shape if channels are not last.
 511     if self.data_format is not None and self.data_format.startswith("NC"):
 512       if not result_converted.shape[1].value and filter is not None:
 513         output_shape = result_converted.shape.as_list()
 514         output_shape[1] = filter.shape[-1]
 515         result_converted.set_shape(output_shape)
 516 
 517     return result_converted
 518 
 519   def __call__(self, inp, filter):  # pylint: disable=redefined-builtin
 520     return self.call(inp, filter)

 523 def _with_space_to_batch_base_paddings(filter_shape, num_spatial_dims,
 524                                        rate_or_const_rate):
 525   """Helper function to compute base_paddings."""
 526   # Spatial dimensions of the filters and the upsampled filters in which we
 527   # introduce (rate - 1) zeros between consecutive filter values.
 528   filter_spatial_shape = filter_shape[:num_spatial_dims]
 529   dilated_filter_spatial_shape = (
 530       filter_spatial_shape + (filter_spatial_shape - 1) *
 531       (rate_or_const_rate - 1))
 532   pad_extra_shape = dilated_filter_spatial_shape - 1
 533 
 534   # When full_padding_shape is odd, we pad more at end, following the same
 535   # convention as conv2d.
 536   pad_extra_start = pad_extra_shape // 2
 537   pad_extra_end = pad_extra_shape - pad_extra_start
 538   base_paddings = array_ops.stack(
 539       [[pad_extra_start[i], pad_extra_end[i]] for i in range(num_spatial_dims)])
 540   return base_paddings

 543 def _with_space_to_batch_adjust(orig, fill_value, spatial_dims):
 544   """Returns an `adjusted` version of `orig` based on `spatial_dims`.
 545 
 546   Tensor of the same type as `orig` and with shape
 547   `[max(spatial_dims), ...]` where:
 548 
 549     adjusted[spatial_dims[i] - 1, ...] = orig[i, ...]
 550 
 551   for 0 <= i < len(spatial_dims), and
 552 
 553     adjusted[j, ...] = fill_value
 554 
 555   for j != spatial_dims[i] - 1 for some i.
 556 
 557   If `orig` is a constant value, then the result will be a constant value.
 558 
 559   Args:
 560     orig: Tensor of rank > max(spatial_dims).
 561     fill_value: Numpy scalar (of same data type as `orig) specifying the fill
 562       value for non-spatial dimensions.
 563     spatial_dims: See with_space_to_batch.
 564 
 565   Returns:
 566     `adjusted` tensor.
 567   """
 568   fill_dims = orig.get_shape().as_list()[1:]
 569   dtype = orig.dtype.as_numpy_dtype
 570   parts = []
 571   const_orig = tensor_util.constant_value(orig)
 572   const_or_orig = const_orig if const_orig is not None else orig
 573   prev_spatial_dim = 0
 574   i = 0
 575   while i < len(spatial_dims):
 576     start_i = i
 577     start_spatial_dim = spatial_dims[i]
 578     if start_spatial_dim > 1:
 579       # Fill in any gap from the previous spatial dimension (or dimension 1 if
 580       # this is the first spatial dimension) with `fill_value`.
 581       parts.append(
 582           np.full(
 583               [start_spatial_dim - 1 - prev_spatial_dim] + fill_dims,
 584               fill_value,
 585               dtype=dtype))
 586     # Find the largest value of i such that:
 587     #   [spatial_dims[start_i], ..., spatial_dims[i]]
 588     #     == [start_spatial_dim, ..., start_spatial_dim + i - start_i],
 589     # i.e. the end of a contiguous group of spatial dimensions.
 590     while (i + 1 < len(spatial_dims) and
 591            spatial_dims[i + 1] == spatial_dims[i] + 1):
 592       i += 1
 593     parts.append(const_or_orig[start_i:i + 1])
 594     prev_spatial_dim = spatial_dims[i]
 595     i += 1
 596   if const_orig is not None:
 597     return np.concatenate(parts)
 598   else:
 599     return array_ops.concat(parts, 0)

 602 def _get_strides_and_dilation_rate(num_spatial_dims, strides, dilation_rate):
 603   """Helper function for verifying strides and dilation_rate arguments.
 604 
 605   This is used by `convolution` and `pool`.
 606 
 607   Args:
 608     num_spatial_dims: int
 609     strides: Optional.  List of N ints >= 1.  Defaults to [1]*N.  If any value
 610       of strides is > 1, then all values of dilation_rate must be 1.
 611     dilation_rate: Optional.  List of N ints >= 1.  Defaults to [1]*N.  If any
 612       value of dilation_rate is > 1, then all values of strides must be 1.
 613 
 614   Returns:
 615     Normalized (strides, dilation_rate) as int32 numpy arrays of shape
 616     [num_spatial_dims].
 617 
 618   Raises:
 619     ValueError: if the parameters are invalid.
 620   """
 621   if dilation_rate is None:
 622     dilation_rate = [1] * num_spatial_dims
 623   elif len(dilation_rate) != num_spatial_dims:
 624     raise ValueError("len(dilation_rate)=%d but should be %d" %
 625                      (len(dilation_rate), num_spatial_dims))
 626   dilation_rate = np.array(dilation_rate, dtype=np.int32)
 627   if np.any(dilation_rate < 1):
 628     raise ValueError("all values of dilation_rate must be positive")
 629 
 630   if strides is None:
 631     strides = [1] * num_spatial_dims
 632   elif len(strides) != num_spatial_dims:
 633     raise ValueError("len(strides)=%d but should be %d" % (len(strides),
 634                                                            num_spatial_dims))
 635   strides = np.array(strides, dtype=np.int32)
 636   if np.any(strides < 1):
 637     raise ValueError("all values of strides must be positive")
 638 
 639   if np.any(strides > 1) and np.any(dilation_rate > 1):
 640     raise ValueError(
 641         "strides > 1 not supported in conjunction with dilation_rate > 1")
 642   return strides, dilation_rate

 645 @tf_export("nn.convolution")
 646 def convolution(
 647     input,  # pylint: disable=redefined-builtin
 648     filter,  # pylint: disable=redefined-builtin
 649     padding,
 650     strides=None,
 651     dilation_rate=None,
 652     name=None,
 653     data_format=None):
 654   # pylint: disable=line-too-long
 655   """Computes sums of N-D convolutions (actually cross-correlation).
 656 
 657   This also supports either output striding via the optional `strides` parameter
 658   or atrous convolution (also known as convolution with holes or dilated
 659   convolution, based on the French word "trous" meaning holes in English) via
 660   the optional `dilation_rate` parameter.  Currently, however, output striding
 661   is not supported for atrous convolutions.
 662 
 663   Specifically, in the case that `data_format` does not start with "NC", given
 664   a rank (N+2) `input` Tensor of shape
 665 
 666     [num_batches,
 667      input_spatial_shape[0],
 668      ...,
 669      input_spatial_shape[N-1],
 670      num_input_channels],
 671 
 672   a rank (N+2) `filter` Tensor of shape
 673 
 674     [spatial_filter_shape[0],
 675      ...,
 676      spatial_filter_shape[N-1],
 677      num_input_channels,
 678      num_output_channels],
 679 
 680   an optional `dilation_rate` tensor of shape [N] (defaulting to [1]*N)
 681   specifying the filter upsampling/input downsampling rate, and an optional list
 682   of N `strides` (defaulting [1]*N), this computes for each N-D spatial output
 683   position (x[0], ..., x[N-1]):
 684 
 685   ```
 686     output[b, x[0], ..., x[N-1], k] =
 687         sum_{z[0], ..., z[N-1], q}
 688             filter[z[0], ..., z[N-1], q, k] *
 689             padded_input[b,
 690                          x[0]*strides[0] + dilation_rate[0]*z[0],
 691                          ...,
 692                          x[N-1]*strides[N-1] + dilation_rate[N-1]*z[N-1],
 693                          q]
 694   ```
 695   where b is the index into the batch, k is the output channel number, q is the
 696   input channel number, and z is the N-D spatial offset within the filter. Here,
 697   `padded_input` is obtained by zero padding the input using an effective
 698   spatial filter shape of `(spatial_filter_shape-1) * dilation_rate + 1` and
 699   output striding `strides` as described in the
 700   [comment here](https://tensorflow.org/api_guides/python/nn#Convolution).
 701 
 702   In the case that `data_format` does start with `"NC"`, the `input` and output
 703   (but not the `filter`) are simply transposed as follows:
 704 
 705     convolution(input, data_format, **kwargs) =
 706       tf.transpose(convolution(tf.transpose(input, [0] + range(2,N+2) + [1]),
 707                                **kwargs),
 708                    [0, N+1] + range(1, N+1))
 709 
 710   It is required that 1 <= N <= 3.
 711 
 712   Args:
 713     input: An N-D `Tensor` of type `T`, of shape
 714       `[batch_size] + input_spatial_shape + [in_channels]` if data_format does
 715       not start with "NC" (default), or
 716       `[batch_size, in_channels] + input_spatial_shape` if data_format starts
 717       with "NC".
 718     filter: An N-D `Tensor` with the same type as `input` and shape
 719       `spatial_filter_shape + [in_channels, out_channels]`.
 720     padding: A string, either `"VALID"` or `"SAME"`. The padding algorithm.
 721     strides: Optional.  Sequence of N ints >= 1.  Specifies the output stride.
 722       Defaults to [1]*N.  If any value of strides is > 1, then all values of
 723       dilation_rate must be 1.
 724     dilation_rate: Optional.  Sequence of N ints >= 1.  Specifies the filter
 725       upsampling/input downsampling rate.  In the literature, the same parameter
 726       is sometimes called `input stride` or `dilation`.  The effective filter
 727       size used for the convolution will be `spatial_filter_shape +
 728       (spatial_filter_shape - 1) * (rate - 1)`, obtained by inserting
 729       (dilation_rate[i]-1) zeros between consecutive elements of the original
 730       filter in each spatial dimension i.  If any value of dilation_rate is > 1,
 731       then all values of strides must be 1.
 732     name: Optional name for the returned tensor.
 733     data_format: A string or None.  Specifies whether the channel dimension of
 734       the `input` and output is the last dimension (default, or if `data_format`
 735       does not start with "NC"), or the second dimension (if `data_format`
 736       starts with "NC").  For N=1, the valid values are "NWC" (default) and
 737       "NCW".  For N=2, the valid values are "NHWC" (default) and "NCHW".
 738       For N=3, the valid values are "NDHWC" (default) and "NCDHW".
 739 
 740   Returns:
 741     A `Tensor` with the same type as `input` of shape
 742 
 743         `[batch_size] + output_spatial_shape + [out_channels]`
 744 
 745     if data_format is None or does not start with "NC", or
 746 
 747         `[batch_size, out_channels] + output_spatial_shape`
 748 
 749     if data_format starts with "NC",
 750     where `output_spatial_shape` depends on the value of `padding`.
 751 
 752     If padding == "SAME":
 753       output_spatial_shape[i] = ceil(input_spatial_shape[i] / strides[i])
 754 
 755     If padding == "VALID":
 756       output_spatial_shape[i] =
 757         ceil((input_spatial_shape[i] -
 758               (spatial_filter_shape[i]-1) * dilation_rate[i])
 759              / strides[i]).
 760 
 761   Raises:
 762     ValueError: If input/output depth does not match `filter` shape, if padding
 763       is other than `"VALID"` or `"SAME"`, or if data_format is invalid.
 764 
 765   """
 766   # pylint: enable=line-too-long
 767   with ops.name_scope(name, "convolution", [input, filter]) as name:
 768     input = ops.convert_to_tensor(input, name="input")  # pylint: disable=redefined-builtin
 769     input_shape = input.get_shape()
 770     filter = ops.convert_to_tensor(filter, name="filter")  # pylint: disable=redefined-builtin
 771     filter_shape = filter.get_shape()
 772     op = Convolution(
 773         input_shape,
 774         filter_shape,
 775         padding,
 776         strides=strides,
 777         dilation_rate=dilation_rate,
 778         name=name,
 779         data_format=data_format)
 780     return op(input, filter)

 871 @tf_export("nn.pool")
 872 def pool(
 873     input,  # pylint: disable=redefined-builtin
 874     window_shape,
 875     pooling_type,
 876     padding,
 877     dilation_rate=None,
 878     strides=None,
 879     name=None,
 880     data_format=None):
 881   # pylint: disable=line-too-long
 882   """Performs an N-D pooling operation.
 883 
 884   In the case that `data_format` does not start with "NC", computes for
 885       0 <= b < batch_size,
 886       0 <= x[i] < output_spatial_shape[i],
 887       0 <= c < num_channels:
 888 
 889   ```
 890     output[b, x[0], ..., x[N-1], c] =
 891       REDUCE_{z[0], ..., z[N-1]}
 892         input[b,
 893               x[0] * strides[0] - pad_before[0] + dilation_rate[0]*z[0],
 894               ...
 895               x[N-1]*strides[N-1] - pad_before[N-1] + dilation_rate[N-1]*z[N-1],
 896               c],
 897   ```
 898 
 899   where the reduction function REDUCE depends on the value of `pooling_type`,
 900   and pad_before is defined based on the value of `padding` as described in
 901   the "returns" section of `tf.nn.convolution` for details.
 902   The reduction never includes out-of-bounds positions.
 903 
 904   In the case that `data_format` starts with `"NC"`, the `input` and output are
 905   simply transposed as follows:
 906 
 907   ```
 908     pool(input, data_format, **kwargs) =
 909       tf.transpose(pool(tf.transpose(input, [0] + range(2,N+2) + [1]),
 910                         **kwargs),
 911                    [0, N+1] + range(1, N+1))
 912   ```
 913 
 914   Args:
 915     input: Tensor of rank N+2, of shape
 916       `[batch_size] + input_spatial_shape + [num_channels]` if data_format does
 917       not start with "NC" (default), or
 918       `[batch_size, num_channels] + input_spatial_shape` if data_format starts
 919       with "NC".  Pooling happens over the spatial dimensions only.
 920     window_shape: Sequence of N ints >= 1.
 921     pooling_type: Specifies pooling operation, must be "AVG" or "MAX".
 922     padding: The padding algorithm, must be "SAME" or "VALID".
 923       See the "returns" section of `tf.nn.convolution` for details.
 924     dilation_rate: Optional.  Dilation rate.  List of N ints >= 1.
 925       Defaults to [1]*N.  If any value of dilation_rate is > 1, then all values
 926       of strides must be 1.
 927     strides: Optional.  Sequence of N ints >= 1.  Defaults to [1]*N.
 928       If any value of strides is > 1, then all values of dilation_rate must be
 929       1.
 930     name: Optional. Name of the op.
 931     data_format: A string or None.  Specifies whether the channel dimension of
 932       the `input` and output is the last dimension (default, or if `data_format`
 933       does not start with "NC"), or the second dimension (if `data_format`
 934       starts with "NC").  For N=1, the valid values are "NWC" (default) and
 935       "NCW".  For N=2, the valid values are "NHWC" (default) and "NCHW".
 936       For N=3, the valid values are "NDHWC" (default) and "NCDHW".
 937 
 938   Returns:
 939     Tensor of rank N+2, of shape
 940       [batch_size] + output_spatial_shape + [num_channels]
 941 
 942     if data_format is None or does not start with "NC", or
 943 
 944       [batch_size, num_channels] + output_spatial_shape
 945 
 946     if data_format starts with "NC",
 947     where `output_spatial_shape` depends on the value of padding:
 948 
 949     If padding = "SAME":
 950       output_spatial_shape[i] = ceil(input_spatial_shape[i] / strides[i])
 951 
 952     If padding = "VALID":
 953       output_spatial_shape[i] =
 954         ceil((input_spatial_shape[i] - (window_shape[i] - 1) * dilation_rate[i])
 955              / strides[i]).
 956 
 957   Raises:
 958     ValueError: if arguments are invalid.
 959 
 960   """
 961   # pylint: enable=line-too-long
 962   with ops.name_scope(name, "%s_pool" % (pooling_type.lower()),
 963                       [input]) as scope:
 964     input = ops.convert_to_tensor(input, name="input")  # pylint: disable=redefined-builtin
 965 
 966     num_spatial_dims = len(window_shape)
 967     if num_spatial_dims < 1 or num_spatial_dims > 3:
 968       raise ValueError("It is required that 1 <= num_spatial_dims <= 3.")
 969 
 970     input.get_shape().with_rank(num_spatial_dims + 2)
 971 
 972     strides, dilation_rate = _get_strides_and_dilation_rate(
 973         num_spatial_dims, strides, dilation_rate)
 974 
 975     if padding == "SAME" and np.any(dilation_rate > 1):
 976       raise ValueError(
 977           "pooling with SAME padding is not implemented for dilation_rate > 1")
 978 
 979     if np.any(strides > window_shape):
 980       raise ValueError(
 981           "strides > window_shape not supported due to inconsistency between "
 982           "CPU and GPU implementations")
 983 
 984     pooling_ops = {
 985         ("MAX", 1): max_pool,
 986         ("MAX", 2): max_pool,
 987         ("MAX", 3): max_pool3d,  # pylint: disable=undefined-variable
 988         ("AVG", 1): avg_pool,
 989         ("AVG", 2): avg_pool,
 990         ("AVG", 3): avg_pool3d,  # pylint: disable=undefined-variable
 991     }
 992     op_key = (pooling_type, num_spatial_dims)
 993     if op_key not in pooling_ops:
 994       raise ValueError("%d-D %s pooling is not supported." % (op_key[1],
 995                                                               op_key[0]))
 996 
 997     if data_format is None or not data_format.startswith("NC"):
 998       adjusted_window_shape = [1] + list(window_shape) + [1]
 999       adjusted_strides = [1] + list(strides) + [1]
1000       spatial_dims = range(1, num_spatial_dims + 1)
1001     else:
1002       adjusted_window_shape = [1, 1] + list(window_shape)
1003       adjusted_strides = [1, 1] + list(strides)
1004       spatial_dims = range(2, num_spatial_dims + 2)
1005 
1006     if num_spatial_dims == 1:
1007       if data_format is None or data_format == "NWC":
1008         data_format_kwargs = dict(data_format="NHWC")
1009       elif data_format == "NCW":
1010         data_format_kwargs = dict(data_format="NCHW")
1011       else:
1012         raise ValueError("data_format must be either \"NWC\" or \"NCW\".")
1013       adjusted_window_shape = [1] + adjusted_window_shape
1014       adjusted_strides = [1] + adjusted_strides
1015     else:
1016       data_format_kwargs = dict(data_format=data_format)
1017 
1018     def op(converted_input, _, converted_padding):  # pylint: disable=missing-docstring
1019       if num_spatial_dims == 1:
1020         converted_input = array_ops.expand_dims(converted_input,
1021                                                 spatial_dims[0])
1022       result = pooling_ops[op_key](
1023           converted_input,
1024           adjusted_window_shape,
1025           adjusted_strides,
1026           converted_padding,
1027           name=scope,
1028           **data_format_kwargs)
1029       if num_spatial_dims == 1:
1030         result = array_ops.squeeze(result, [spatial_dims[0]])
1031       return result
1032 
1033     return with_space_to_batch(
1034         input=input,
1035         dilation_rate=dilation_rate,
1036         padding=padding,
1037         op=op,
1038         spatial_dims=spatial_dims,
1039         filter_shape=window_shape)

1042 @tf_export("nn.atrous_conv2d")
1043 def atrous_conv2d(value, filters, rate, padding, name=None):
1044   """Atrous convolution (a.k.a. convolution with holes or dilated convolution).
1045 
1046   This function is a simpler wrapper around the more general
1047   `tf.nn.convolution`, and exists only for backwards compatibility. You can
1048   use `tf.nn.convolution` to perform 1-D, 2-D, or 3-D atrous convolution.
1049 
1050 
1051   Computes a 2-D atrous convolution, also known as convolution with holes or
1052   dilated convolution, given 4-D `value` and `filters` tensors. If the `rate`
1053   parameter is equal to one, it performs regular 2-D convolution. If the `rate`
1054   parameter is greater than one, it performs convolution with holes, sampling
1055   the input values every `rate` pixels in the `height` and `width` dimensions.
1056   This is equivalent to convolving the input with a set of upsampled filters,
1057   produced by inserting `rate - 1` zeros between two consecutive values of the
1058   filters along the `height` and `width` dimensions, hence the name atrous
1059   convolution or convolution with holes (the French word trous means holes in
1060   English).
1061 
1062   More specifically:
1063 
1064   ```
1065   output[batch, height, width, out_channel] =
1066       sum_{dheight, dwidth, in_channel} (
1067           filters[dheight, dwidth, in_channel, out_channel] *
1068           value[batch, height + rate*dheight, width + rate*dwidth, in_channel]
1069       )
1070   ```
1071 
1072   Atrous convolution allows us to explicitly control how densely to compute
1073   feature responses in fully convolutional networks. Used in conjunction with
1074   bilinear interpolation, it offers an alternative to `conv2d_transpose` in
1075   dense prediction tasks such as semantic image segmentation, optical flow
1076   computation, or depth estimation. It also allows us to effectively enlarge
1077   the field of view of filters without increasing the number of parameters or
1078   the amount of computation.
1079 
1080   For a description of atrous convolution and how it can be used for dense
1081   feature extraction, please see: [Semantic Image Segmentation with Deep
1082   Convolutional Nets and Fully Connected CRFs](http://arxiv.org/abs/1412.7062).
1083   The same operation is investigated further in [Multi-Scale Context Aggregation
1084   by Dilated Convolutions](http://arxiv.org/abs/1511.07122). Previous works
1085   that effectively use atrous convolution in different ways are, among others,
1086   [OverFeat: Integrated Recognition, Localization and Detection using
1087   Convolutional Networks](http://arxiv.org/abs/1312.6229) and [Fast Image
1088   Scanning with Deep Max-Pooling Convolutional Neural
1089   Networks](http://arxiv.org/abs/1302.1700).
1090   Atrous convolution is also closely related to the so-called noble identities
1091   in multi-rate signal processing.
1092 
1093   There are many different ways to implement atrous convolution (see the refs
1094   above). The implementation here reduces
1095 
1096   ```python
1097       atrous_conv2d(value, filters, rate, padding=padding)
1098   ```
1099 
1100   to the following three operations:
1101 
1102   ```python
1103       paddings = ...
1104       net = space_to_batch(value, paddings, block_size=rate)
1105       net = conv2d(net, filters, strides=[1, 1, 1, 1], padding="VALID")
1106       crops = ...
1107       net = batch_to_space(net, crops, block_size=rate)
1108   ```
1109 
1110   Advanced usage. Note the following optimization: A sequence of `atrous_conv2d`
1111   operations with identical `rate` parameters, 'SAME' `padding`, and filters
1112   with odd heights/ widths:
1113 
1114   ```python
1115       net = atrous_conv2d(net, filters1, rate, padding="SAME")
1116       net = atrous_conv2d(net, filters2, rate, padding="SAME")
1117       ...
1118       net = atrous_conv2d(net, filtersK, rate, padding="SAME")
1119   ```
1120 
1121   can be equivalently performed cheaper in terms of computation and memory as:
1122 
1123   ```python
1124       pad = ...  # padding so that the input dims are multiples of rate
1125       net = space_to_batch(net, paddings=pad, block_size=rate)
1126       net = conv2d(net, filters1, strides=[1, 1, 1, 1], padding="SAME")
1127       net = conv2d(net, filters2, strides=[1, 1, 1, 1], padding="SAME")
1128       ...
1129       net = conv2d(net, filtersK, strides=[1, 1, 1, 1], padding="SAME")
1130       net = batch_to_space(net, crops=pad, block_size=rate)
1131   ```
1132 
1133   because a pair of consecutive `space_to_batch` and `batch_to_space` ops with
1134   the same `block_size` cancel out when their respective `paddings` and `crops`
1135   inputs are identical.
1136 
1137   Args:
1138     value: A 4-D `Tensor` of type `float`. It needs to be in the default "NHWC"
1139       format. Its shape is `[batch, in_height, in_width, in_channels]`.
1140     filters: A 4-D `Tensor` with the same type as `value` and shape
1141       `[filter_height, filter_width, in_channels, out_channels]`. `filters`'
1142       `in_channels` dimension must match that of `value`. Atrous convolution is
1143       equivalent to standard convolution with upsampled filters with effective
1144       height `filter_height + (filter_height - 1) * (rate - 1)` and effective
1145       width `filter_width + (filter_width - 1) * (rate - 1)`, produced by
1146       inserting `rate - 1` zeros along consecutive elements across the
1147       `filters`' spatial dimensions.
1148     rate: A positive int32. The stride with which we sample input values across
1149       the `height` and `width` dimensions. Equivalently, the rate by which we
1150       upsample the filter values by inserting zeros across the `height` and
1151       `width` dimensions. In the literature, the same parameter is sometimes
1152       called `input stride` or `dilation`.
1153     padding: A string, either `'VALID'` or `'SAME'`. The padding algorithm.
1154     name: Optional name for the returned tensor.
1155 
1156   Returns:
1157     A `Tensor` with the same type as `value`.
1158     Output shape with `'VALID'` padding is:
1159 
1160         [batch, height - 2 * (filter_width - 1),
1161          width - 2 * (filter_height - 1), out_channels].
1162 
1163     Output shape with `'SAME'` padding is:
1164 
1165         [batch, height, width, out_channels].
1166 
1167   Raises:
1168     ValueError: If input/output depth does not match `filters`' shape, or if
1169       padding is other than `'VALID'` or `'SAME'`.
1170   """
1171   return convolution(
1172       input=value,
1173       filter=filters,
1174       padding=padding,
1175       dilation_rate=np.broadcast_to(rate, (2,)),
1176       name=name)

1257 @tf_export("nn.atrous_conv2d_transpose")
1258 def atrous_conv2d_transpose(value,
1259                             filters,
1260                             output_shape,
1261                             rate,
1262                             padding,
1263                             name=None):
1264   """The transpose of `atrous_conv2d`.
1265 
1266   This operation is sometimes called "deconvolution" after [Deconvolutional
1267   Networks](http://www.matthewzeiler.com/pubs/cvpr2010/cvpr2010.pdf), but is
1268   actually the transpose (gradient) of `atrous_conv2d` rather than an actual
1269   deconvolution.
1270 
1271   Args:
1272     value: A 4-D `Tensor` of type `float`. It needs to be in the default `NHWC`
1273       format. Its shape is `[batch, in_height, in_width, in_channels]`.
1274     filters: A 4-D `Tensor` with the same type as `value` and shape
1275       `[filter_height, filter_width, out_channels, in_channels]`. `filters`'
1276       `in_channels` dimension must match that of `value`. Atrous convolution is
1277       equivalent to standard convolution with upsampled filters with effective
1278       height `filter_height + (filter_height - 1) * (rate - 1)` and effective
1279       width `filter_width + (filter_width - 1) * (rate - 1)`, produced by
1280       inserting `rate - 1` zeros along consecutive elements across the
1281       `filters`' spatial dimensions.
1282     output_shape: A 1-D `Tensor` of shape representing the output shape of the
1283       deconvolution op.
1284     rate: A positive int32. The stride with which we sample input values across
1285       the `height` and `width` dimensions. Equivalently, the rate by which we
1286       upsample the filter values by inserting zeros across the `height` and
1287       `width` dimensions. In the literature, the same parameter is sometimes
1288       called `input stride` or `dilation`.
1289     padding: A string, either `'VALID'` or `'SAME'`. The padding algorithm.
1290     name: Optional name for the returned tensor.
1291 
1292   Returns:
1293     A `Tensor` with the same type as `value`.
1294 
1295   Raises:
1296     ValueError: If input/output depth does not match `filters`' shape, or if
1297       padding is other than `'VALID'` or `'SAME'`, or if the `rate` is less
1298       than one, or if the output_shape is not a tensor with 4 elements.
1299   """
1300   with ops.name_scope(name, "atrous_conv2d_transpose",
1301                       [value, filters, output_shape]) as name:
1302     value = ops.convert_to_tensor(value, name="value")
1303     filters = ops.convert_to_tensor(filters, name="filters")
1304     if not value.get_shape()[3].is_compatible_with(filters.get_shape()[3]):
1305       raise ValueError(
1306           "value's input channels does not match filters' input channels, "
1307           "{} != {}".format(value.get_shape()[3],
1308                             filters.get_shape()[3]))
1309     if rate < 1:
1310       raise ValueError("rate {} cannot be less than one".format(rate))
1311 
1312     if rate == 1:
1313       return conv2d_transpose(
1314           value,
1315           filters,
1316           output_shape,
1317           strides=[1, 1, 1, 1],
1318           padding=padding,
1319           data_format="NHWC")
1320 
1321     output_shape_ = ops.convert_to_tensor(output_shape, name="output_shape")
1322     if not output_shape_.get_shape().is_compatible_with(tensor_shape.vector(4)):
1323       raise ValueError("output_shape must have shape (4,), got {}".format(
1324           output_shape_.get_shape()))
1325 
1326     if isinstance(output_shape, (list, np.ndarray)):
1327       # output_shape's shape should be == [4] if reached this point.
1328       if not filters.get_shape()[2].is_compatible_with(output_shape[3]):
1329         raise ValueError(
1330             "output_shape does not match filter's output channels, "
1331             "{} != {}".format(output_shape[3],
1332                               filters.get_shape()[2]))
1333 
1334     # We have two padding contributions. The first is used for converting "SAME"
1335     # to "VALID". The second is required so that the height and width of the
1336     # zero-padded value tensor are multiples of rate.
1337 
1338     # Padding required to reduce to "VALID" convolution
1339     if padding == "SAME":
1340       # Handle filters whose shape is unknown during graph creation.
1341       if filters.get_shape().is_fully_defined():
1342         filter_shape = filters.get_shape().as_list()
1343       else:
1344         filter_shape = array_ops.shape(filters)
1345       filter_height, filter_width = filter_shape[0], filter_shape[1]
1346 
1347       # Spatial dimensions of the filters and the upsampled filters in which we
1348       # introduce (rate - 1) zeros between consecutive filter values.
1349       filter_height_up = filter_height + (filter_height - 1) * (rate - 1)
1350       filter_width_up = filter_width + (filter_width - 1) * (rate - 1)
1351 
1352       pad_height = filter_height_up - 1
1353       pad_width = filter_width_up - 1
1354 
1355       # When pad_height (pad_width) is odd, we pad more to bottom (right),
1356       # following the same convention as conv2d().
1357       pad_top = pad_height // 2
1358       pad_bottom = pad_height - pad_top
1359       pad_left = pad_width // 2
1360       pad_right = pad_width - pad_left
1361     elif padding == "VALID":
1362       pad_top = 0
1363       pad_bottom = 0
1364       pad_left = 0
1365       pad_right = 0
1366     else:
1367       raise ValueError("padding must be either VALID or SAME:"
1368                        " {}".format(padding))
1369 
1370     in_height = output_shape[1] + pad_top + pad_bottom
1371     in_width = output_shape[2] + pad_left + pad_right
1372 
1373     # More padding so that rate divides the height and width of the input.
1374     pad_bottom_extra = (rate - in_height % rate) % rate
1375     pad_right_extra = (rate - in_width % rate) % rate
1376 
1377     # The paddings argument to space_to_batch is just the extra padding
1378     # component.
1379     space_to_batch_pad = [[0, pad_bottom_extra], [0, pad_right_extra]]
1380 
1381     value = array_ops.space_to_batch(
1382         input=value, paddings=space_to_batch_pad, block_size=rate)
1383 
1384     input_sizes = [
1385         rate * rate * output_shape[0], (in_height + pad_bottom_extra) // rate,
1386         (in_width + pad_right_extra) // rate, output_shape[3]
1387     ]
1388 
1389     value = gen_nn_ops.conv2d_backprop_input(
1390         input_sizes=input_sizes,
1391         filter=filters,
1392         out_backprop=value,
1393         strides=[1, 1, 1, 1],
1394         padding="VALID",
1395         data_format="NHWC")
1396 
1397     # The crops argument to batch_to_space includes both padding components.
1398     batch_to_space_crop = [[pad_top, pad_bottom + pad_bottom_extra],
1399                            [pad_left, pad_right + pad_right_extra]]
1400 
1401     return array_ops.batch_to_space(
1402         input=value, crops=batch_to_space_crop, block_size=rate)

1634 def _softmax(logits, compute_op, dim=-1, name=None):
1635   """Helper function for softmax and log_softmax.
1636 
1637   It reshapes and transposes the input logits into a 2-D Tensor and then invokes
1638   the tf.nn._softmax or tf.nn._log_softmax function. The output would be
1639   transposed and reshaped back.
1640 
1641   Args:
1642     logits: A non-empty `Tensor`. Must be one of the following types: `half`,
1643       `float32`, `float64`.
1644     compute_op: Either gen_nn_ops.softmax or gen_nn_ops.log_softmax
1645     dim: The dimension softmax would be performed on. The default is -1 which
1646       indicates the last dimension.
1647     name: A name for the operation (optional).
1648 
1649   Returns:
1650     A `Tensor`. Has the same type as `logits`. Same shape as `logits`.
1651   Raises:
1652     InvalidArgumentError: if `logits` is empty or `dim` is beyond the last
1653       dimension of `logits`.
1654   """
1655 
1656   def _swap_axis(logits, dim_index, last_index, name=None):
1657     """Swaps logits's dim_index and last_index."""
1658     return array_ops.transpose(
1659         logits,
1660         array_ops.concat([
1661             math_ops.range(dim_index), [last_index],
1662             math_ops.range(dim_index + 1, last_index), [dim_index]
1663         ], 0),
1664         name=name)
1665 
1666   logits = ops.convert_to_tensor(logits)
1667 
1668   # We need its original shape for shape inference.
1669   shape = logits.get_shape()
1670   is_last_dim = (dim is -1) or (dim == shape.ndims - 1)
1671 
1672   if is_last_dim:
1673     return compute_op(logits, name=name)
1674 
1675   # If dim is not the last dimension, we have to do a transpose so that we can
1676   # still perform softmax on its last dimension.
1677 
1678   # Swap logits' dimension of dim and its last dimension.
1679   input_rank = array_ops.rank(logits)
1680   dim_axis = dim % shape.ndims
1681   logits = _swap_axis(logits, dim_axis, math_ops.subtract(input_rank, 1))
1682 
1683   # Do the actual softmax on its last dimension.
1684   output = compute_op(logits)
1685 
1686   output = _swap_axis(
1687       output, dim_axis, math_ops.subtract(input_rank, 1), name=name)
1688 
1689   # Make shape inference work since transpose may erase its static shape.
1690   output.set_shape(shape)
1691 
1692   return output

1870

1872     # shape.
1873     if not context.executing_eagerly(
1874     ) and shape is not None and shape.dims is not None:

2661     \\(targets_i\\) be the target class for example `i`,

2665 
2666   Args:

2669     targets: A `Tensor`. Must be one of the following types: `int32`, `int64`.
2670       A `batch_size` vector of class ids.
2671     k: An `int`. Number of top elements to look at for computing precision.
2672     name: A name for the operation (optional).
2673
