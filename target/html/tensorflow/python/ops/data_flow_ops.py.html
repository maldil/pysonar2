  48 def _as_type_list(dtypes):
  49   """Convert dtypes to a list of types."""
  50   assert dtypes is not None
  51   if not (isinstance(dtypes, list) or isinstance(dtypes, tuple)):
  52     # We have a single type.
  53     return [dtypes]
  54   else:
  55     # We have a list or tuple of types.
  56     return list(dtypes)

  59 def _as_shape_list(shapes,
  60                    dtypes,
  61                    unknown_dim_allowed=False,
  62                    unknown_rank_allowed=False):
  63   """Convert shapes to a list of tuples of int (or None)."""
  64   del dtypes
  65   if unknown_dim_allowed:
  66     if (not isinstance(shapes, collections.Sequence) or not shapes or
  67         any(shape is None or isinstance(shape, int) for shape in shapes)):
  68       raise ValueError(
  69           "When providing partial shapes, a list of shapes must be provided.")
  70   if shapes is None:
  71     return None
  72   if isinstance(shapes, tensor_shape.TensorShape):
  73     shapes = [shapes]
  74   if not isinstance(shapes, (tuple, list)):
  75     raise TypeError(
  76         "shapes must be a TensorShape or a list or tuple of TensorShapes.")
  77   if all(shape is None or isinstance(shape, int) for shape in shapes):
  78     # We have a single shape.
  79     shapes = [shapes]
  80   shapes = [tensor_shape.as_shape(shape) for shape in shapes]
  81   if not unknown_dim_allowed:
  82     if any([not shape.is_fully_defined() for shape in shapes]):
  83       raise ValueError("All shapes must be fully defined: %s" % shapes)
  84   if not unknown_rank_allowed:
  85     if any([shape.dims is None for shape in shapes]):
  86       raise ValueError("All shapes must have a defined rank: %s" % shapes)
  87 
  88   return shapes

  91 def _as_name_list(names, dtypes):
  92   if names is None:
  93     return None
  94   if not isinstance(names, (list, tuple)):
  95     names = [names]
  96   if len(names) != len(dtypes):
  97     raise ValueError("List of names must have the same length as the list "
  98                      "of dtypes")
  99   return list(names)

 102 def _shape_common(s1, s2):
 103   """The greatest lower bound (ordered by specificity) TensorShape."""
 104   s1 = tensor_shape.TensorShape(s1)
 105   s2 = tensor_shape.TensorShape(s2)
 106   if s1.ndims is None or s2.ndims is None or s1.ndims != s2.ndims:
 107     return tensor_shape.unknown_shape()
 108   d = [
 109       d1 if d1 is not None and d1 == d2 else None
 110       for (d1, d2) in zip(s1.as_list(), s2.as_list())
 111   ]
 112   return tensor_shape.TensorShape(d)

 116 @tf_export("io.QueueBase", "QueueBase")
 117 @deprecation.deprecated_endpoints("QueueBase")
 118 class QueueBase(object):
 119   """Base class for queue implementations.
 120 
 121   A queue is a TensorFlow data structure that stores tensors across
 122   multiple steps, and exposes operations that enqueue and dequeue
 123   tensors.
 124 
 125   Each queue element is a tuple of one or more tensors, where each
 126   tuple component has a static dtype, and may have a static shape. The
 127   queue implementations support versions of enqueue and dequeue that
 128   handle single elements, versions that support enqueuing and
 129   dequeuing a batch of elements at once.
 130 
 131   See `tf.FIFOQueue` and
 132   `tf.RandomShuffleQueue` for concrete
 133   implementations of this class, and instructions on how to create
 134   them.
 135   """
 136 
 137   def __init__(self, dtypes, shapes, names, queue_ref):
 138     """Constructs a queue object from a queue reference.
 139 
 140     The two optional lists, `shapes` and `names`, must be of the same length
 141     as `dtypes` if provided.  The values at a given index `i` indicate the
 142     shape and name to use for the corresponding queue component in `dtypes`.
 143 
 144     Args:
 145       dtypes:  A list of types.  The length of dtypes must equal the number
 146         of tensors in each element.
 147       shapes: Constraints on the shapes of tensors in an element:
 148         A list of shape tuples or None. This list is the same length
 149         as dtypes.  If the shape of any tensors in the element are constrained,
 150         all must be; shapes can be None if the shapes should not be constrained.
 151       names: Optional list of names.  If provided, the `enqueue()` and
 152         `dequeue()` methods will use dictionaries with these names as keys.
 153         Must be None or a list or tuple of the same length as `dtypes`.
 154       queue_ref: The queue reference, i.e. the output of the queue op.
 155 
 156     Raises:
 157       ValueError: If one of the arguments is invalid.
 158     """
 159     self._dtypes = dtypes
 160     if shapes is not None:
 161       if len(shapes) != len(dtypes):
 162         raise ValueError("Queue shapes must have the same length as dtypes")
 163       self._shapes = [tensor_shape.TensorShape(s) for s in shapes]
 164     else:
 165       self._shapes = [tensor_shape.unknown_shape() for _ in self._dtypes]
 166     if names is not None:
 167       if len(names) != len(dtypes):
 168         raise ValueError("Queue names must have the same length as dtypes")
 169       self._names = names
 170     else:
 171       self._names = None
 172     self._queue_ref = queue_ref
 173     if context.executing_eagerly():
 174       self._name = context.context().scope_name
 175       self._resource_deleter = resource_variable_ops.EagerResourceDeleter(
 176           queue_ref, None)
 177     else:
 178       self._name = self._queue_ref.op.name.split("/")[-1]

 180   @staticmethod
 181   def from_list(index, queues):
 182     """Create a queue using the queue reference from `queues[index]`.
 183 
 184     Args:
 185       index: An integer scalar tensor that determines the input that gets
 186         selected.
 187       queues: A list of `QueueBase` objects.
 188 
 189     Returns:
 190       A `QueueBase` object.
 191 
 192     Raises:
 193       TypeError: When `queues` is not a list of `QueueBase` objects,
 194         or when the data types of `queues` are not all the same.
 195     """
 196     if ((not queues) or (not isinstance(queues, list)) or
 197         (not all(isinstance(x, QueueBase) for x in queues))):
 198       raise TypeError("A list of queues expected")
 199 
 200     dtypes = queues[0].dtypes
 201     if not all([dtypes == q.dtypes for q in queues[1:]]):
 202       raise TypeError("Queues do not have matching component dtypes.")
 203 
 204     names = queues[0].names
 205     if not all([names == q.names for q in queues[1:]]):
 206       raise TypeError("Queues do not have matching component names.")
 207 
 208     queue_shapes = [q.shapes for q in queues]
 209     reduced_shapes = [
 210         six.moves.reduce(_shape_common, s) for s in zip(*queue_shapes)
 211     ]
 212 
 213     queue_refs = array_ops.stack([x.queue_ref for x in queues])
 214     selected_queue = array_ops.gather(queue_refs, index)
 215     return QueueBase(
 216         dtypes=dtypes,
 217         shapes=reduced_shapes,
 218         names=names,
 219         queue_ref=selected_queue)

 248   def _check_enqueue_dtypes(self, vals):
 249     """Validate and convert `vals` to a list of `Tensor`s.
 250 
 251     The `vals` argument can be a Tensor, a list or tuple of tensors, or a
 252     dictionary with tensor values.
 253 
 254     If it is a dictionary, the queue must have been constructed with a
 255     `names` attribute and the dictionary keys must match the queue names.
 256     If the queue was constructed with a `names` attribute, `vals` must
 257     be a dictionary.
 258 
 259     Args:
 260       vals: A tensor, a list or tuple of tensors, or a dictionary..
 261 
 262     Returns:
 263       A list of `Tensor` objects.
 264 
 265     Raises:
 266       ValueError: If `vals` is invalid.
 267     """
 268     if isinstance(vals, dict):
 269       if not self._names:
 270         raise ValueError("Queue must have names to enqueue a dictionary")
 271       if sorted(self._names, key=str) != sorted(vals.keys(), key=str):
 272         raise ValueError("Keys in dictionary to enqueue do not match "
 273                          "names of Queue.  Dictionary: (%s), Queue: (%s)" %
 274                          (sorted(vals.keys()), sorted(self._names)))
 275       # The order of values in `self._names` indicates the order in which the
 276       # tensors in the dictionary `vals` must be listed.
 277       vals = [vals[k] for k in self._names]
 278     else:
 279       if self._names:
 280         raise ValueError("You must enqueue a dictionary in a Queue with names")
 281       if not isinstance(vals, (list, tuple)):
 282         vals = [vals]
 283 
 284     tensors = []
 285     for i, (val, dtype) in enumerate(zip(vals, self._dtypes)):
 286       tensors.append(
 287           ops.convert_to_tensor(val, dtype=dtype, name="component_%d" % i))
 288 
 289     return tensors

 291   def _scope_vals(self, vals):
 292     """Return a list of values to pass to `name_scope()`.

 294     Args:
 295       vals: A tensor, a list or tuple of tensors, or a dictionary.

 301       return vals
 302     elif isinstance(vals, dict):

 304     else:

 307   def enqueue(self, vals, name=None):
 308     """Enqueues one element to this queue.
 309 
 310     If the queue is full when this operation executes, it will block
 311     until the element has been enqueued.
 312 
 313     At runtime, this operation may raise an error if the queue is
 314     `tf.QueueBase.close` before or during its execution. If the
 315     queue is closed before this operation runs,
 316     `tf.errors.CancelledError` will be raised. If this operation is
 317     blocked, and either (i) the queue is closed by a close operation
 318     with `cancel_pending_enqueues=True`, or (ii) the session is
 319     `tf.Session.close`,
 320     `tf.errors.CancelledError` will be raised.
 321 
 322     Args:
 323       vals: A tensor, a list or tuple of tensors, or a dictionary containing
 324         the values to enqueue.
 325       name: A name for the operation (optional).
 326 
 327     Returns:
 328       The operation that enqueues a new tuple of tensors to the queue.
 329     """
 330     with ops.name_scope(name, "%s_enqueue" % self._name,
 331                         self._scope_vals(vals)) as scope:
 332       vals = self._check_enqueue_dtypes(vals)
 333 
 334       # NOTE(mrry): Not using a shape function because we need access to
 335       # the `QueueBase` object.
 336       for val, shape in zip(vals, self._shapes):
 337         val.get_shape().assert_is_compatible_with(shape)
 338 
 339       if self._queue_ref.dtype == _dtypes.resource:
 340         return gen_data_flow_ops.queue_enqueue_v2(
 341             self._queue_ref, vals, name=scope)
 342       else:
 343         return gen_data_flow_ops.queue_enqueue(
 344             self._queue_ref, vals, name=scope)

 346   def enqueue_many(self, vals, name=None):
 347     """Enqueues zero or more elements to this queue.
 348 
 349     This operation slices each component tensor along the 0th dimension to
 350     make multiple queue elements. All of the tensors in `vals` must have the
 351     same size in the 0th dimension.
 352 
 353     If the queue is full when this operation executes, it will block
 354     until all of the elements have been enqueued.
 355 
 356     At runtime, this operation may raise an error if the queue is
 357     `tf.QueueBase.close` before or during its execution. If the
 358     queue is closed before this operation runs,
 359     `tf.errors.CancelledError` will be raised. If this operation is
 360     blocked, and either (i) the queue is closed by a close operation
 361     with `cancel_pending_enqueues=True`, or (ii) the session is
 362     `tf.Session.close`,
 363     `tf.errors.CancelledError` will be raised.
 364 
 365     Args:
 366       vals: A tensor, a list or tuple of tensors, or a dictionary
 367         from which the queue elements are taken.
 368       name: A name for the operation (optional).
 369 
 370     Returns:
 371       The operation that enqueues a batch of tuples of tensors to the queue.
 372     """
 373     with ops.name_scope(name, "%s_EnqueueMany" % self._name,
 374                         self._scope_vals(vals)) as scope:
 375       vals = self._check_enqueue_dtypes(vals)
 376 
 377       # NOTE(mrry): Not using a shape function because we need access to
 378       # the `QueueBase` object.
 379       batch_dim = vals[0].get_shape().with_rank_at_least(1)[0]
 380       for val, shape in zip(vals, self._shapes):
 381         batch_dim = batch_dim.merge_with(
 382             val.get_shape().with_rank_at_least(1)[0])
 383         val.get_shape()[1:].assert_is_compatible_with(shape)
 384 
 385       return gen_data_flow_ops.queue_enqueue_many_v2(
 386           self._queue_ref, vals, name=scope)

 388   def _dequeue_return_value(self, tensors):
 389     """Return the value to return from a dequeue op.
 390 
 391     If the queue has names, return a dictionary with the
 392     names as keys.  Otherwise return either a single tensor
 393     or a list of tensors depending on the length of `tensors`.
 394 
 395     Args:
 396       tensors: List of tensors from the dequeue op.
 397 
 398     Returns:
 399       A single tensor, a list of tensors, or a dictionary
 400       of tensors.
 401     """
 402     if self._names:
 403       # The returned values in `tensors` are in the same order as
 404       # the names in `self._names`.
 405       return {n: tensors[i] for i, n in enumerate(self._names)}
 406     elif len(tensors) == 1:
 407       return tensors[0]
 408     else:
 409       return tensors

 411   def dequeue(self, name=None):
 412     """Dequeues one element from this queue.
 413 
 414     If the queue is empty when this operation executes, it will block
 415     until there is an element to dequeue.
 416 
 417     At runtime, this operation may raise an error if the queue is
 418     `tf.QueueBase.close` before or during its execution. If the
 419     queue is closed, the queue is empty, and there are no pending
 420     enqueue operations that can fulfill this request,
 421     `tf.errors.OutOfRangeError` will be raised. If the session is
 422     `tf.Session.close`,
 423     `tf.errors.CancelledError` will be raised.
 424 
 425     Args:
 426       name: A name for the operation (optional).
 427 
 428     Returns:
 429       The tuple of tensors that was dequeued.
 430     """
 431     if name is None:
 432       name = "%s_Dequeue" % self._name
 433     if self._queue_ref.dtype == _dtypes.resource:
 434       ret = gen_data_flow_ops.queue_dequeue_v2(
 435           self._queue_ref, self._dtypes, name=name)
 436     else:
 437       ret = gen_data_flow_ops.queue_dequeue(
 438           self._queue_ref, self._dtypes, name=name)
 439 
 440     # NOTE(mrry): Not using a shape function because we need access to
 441     # the `QueueBase` object.
 442     if not context.executing_eagerly():
 443       op = ret[0].op
 444       for output, shape in zip(op.values(), self._shapes):
 445         output.set_shape(shape)
 446 
 447     return self._dequeue_return_value(ret)

 449   def dequeue_many(self, n, name=None):
 450     """Dequeues and concatenates `n` elements from this queue.
 451 
 452     This operation concatenates queue-element component tensors along
 453     the 0th dimension to make a single component tensor.  All of the
 454     components in the dequeued tuple will have size `n` in the 0th dimension.
 455 
 456     If the queue is closed and there are less than `n` elements left, then an
 457     `OutOfRange` exception is raised.
 458 
 459     At runtime, this operation may raise an error if the queue is
 460     `tf.QueueBase.close` before or during its execution. If the
 461     queue is closed, the queue contains fewer than `n` elements, and
 462     there are no pending enqueue operations that can fulfill this
 463     request, `tf.errors.OutOfRangeError` will be raised. If the
 464     session is `tf.Session.close`,
 465     `tf.errors.CancelledError` will be raised.
 466 
 467     Args:
 468       n: A scalar `Tensor` containing the number of elements to dequeue.
 469       name: A name for the operation (optional).
 470 
 471     Returns:
 472       The list of concatenated tensors that was dequeued.
 473     """
 474     if name is None:
 475       name = "%s_DequeueMany" % self._name
 476 
 477     ret = gen_data_flow_ops.queue_dequeue_many_v2(
 478         self._queue_ref, n=n, component_types=self._dtypes, name=name)
 479 
 480     # NOTE(mrry): Not using a shape function because we need access to
 481     # the Queue object.
 482     if not context.executing_eagerly():
 483       op = ret[0].op
 484       batch_dim = tensor_shape.Dimension(
 485           tensor_util.constant_value(op.inputs[1]))
 486       for output, shape in zip(op.values(), self._shapes):
 487         output.set_shape(
 488             tensor_shape.TensorShape([batch_dim]).concatenate(shape))
 489 
 490     return self._dequeue_return_value(ret)
 491 
 492   def dequeue_up_to(self, n, name=None):
 493     """Dequeues and concatenates `n` elements from this queue.
 494 
 495     **Note** This operation is not supported by all queues.  If a queue does not
 496     support DequeueUpTo, then a `tf.errors.UnimplementedError` is raised.
 497 
 498     This operation concatenates queue-element component tensors along
 499     the 0th dimension to make a single component tensor. If the queue
 500     has not been closed, all of the components in the dequeued tuple
 501     will have size `n` in the 0th dimension.
 502 
 503     If the queue is closed and there are more than `0` but fewer than
 504     `n` elements remaining, then instead of raising a
 505     `tf.errors.OutOfRangeError` like `tf.QueueBase.dequeue_many`,
 506     less than `n` elements are returned immediately.  If the queue is
 507     closed and there are `0` elements left in the queue, then a
 508     `tf.errors.OutOfRangeError` is raised just like in `dequeue_many`.
 509     Otherwise the behavior is identical to `dequeue_many`.
 510 
 511     Args:
 512       n: A scalar `Tensor` containing the number of elements to dequeue.
 513       name: A name for the operation (optional).
 514 
 515     Returns:
 516       The tuple of concatenated tensors that was dequeued.
 517     """
 518     if name is None:
 519       name = "%s_DequeueUpTo" % self._name
 520 
 521     ret = gen_data_flow_ops.queue_dequeue_up_to_v2(
 522         self._queue_ref, n=n, component_types=self._dtypes, name=name)
 523 
 524     # NOTE(mrry): Not using a shape function because we need access to
 525     # the Queue object.
 526     if not context.executing_eagerly():
 527       op = ret[0].op
 528       for output, shape in zip(op.values(), self._shapes):
 529         output.set_shape(tensor_shape.TensorShape([None]).concatenate(shape))
 530 
 531     return self._dequeue_return_value(ret)

 533   def close(self, cancel_pending_enqueues=False, name=None):
 534     """Closes this queue.
 535 
 536     This operation signals that no more elements will be enqueued in
 537     the given queue. Subsequent `enqueue` and `enqueue_many`
 538     operations will fail. Subsequent `dequeue` and `dequeue_many`
 539     operations will continue to succeed if sufficient elements remain
 540     in the queue. Subsequently dequeue and dequeue_many operations
 541     that would otherwise block waiting for more elements (if close
 542     hadn't been called) will now fail immediately.
 543 
 544     If `cancel_pending_enqueues` is `True`, all pending requests will also
 545     be canceled.
 546 
 547     Args:
 548       cancel_pending_enqueues: (Optional.) A boolean, defaulting to
 549         `False` (described above).
 550       name: A name for the operation (optional).
 551 
 552     Returns:
 553       The operation that closes the queue.
 554     """
 555     if name is None:
 556       name = "%s_Close" % self._name
 557     if self._queue_ref.dtype == _dtypes.resource:
 558       return gen_data_flow_ops.queue_close_v2(
 559           self._queue_ref,
 560           cancel_pending_enqueues=cancel_pending_enqueues,
 561           name=name)
 562     else:
 563       return gen_data_flow_ops.queue_close(
 564           self._queue_ref,
 565           cancel_pending_enqueues=cancel_pending_enqueues,
 566           name=name)

 898 class Barrier(object):
 899   """Represents a key-value map that persists across graph executions."""
 900 
 901   def __init__(self, types, shapes=None, shared_name=None, name="barrier"):
 902     """Creates a barrier that persists across different graph executions.
 903 
 904     A barrier represents a key-value map, where each key is a string, and
 905     each value is a tuple of tensors.
 906 
 907     At runtime, the barrier contains 'complete' and 'incomplete'
 908     elements. A complete element has defined tensors for all
 909     components of its value tuple, and may be accessed using
 910     take_many. An incomplete element has some undefined components in
 911     its value tuple, and may be updated using insert_many.
 912 
 913     The barrier call `take_many` outputs values in a particular order.
 914     First, it only outputs completed values.  Second, the order in which
 915     completed values are returned matches the order in which their very
 916     first component was inserted into the barrier.  So, for example, for this
 917     sequence of insertions and removals:
 918 
 919       barrier = Barrier((tf.string, tf.int32), shapes=((), ()))
 920       barrier.insert_many(0, keys=["k1", "k2"], values=["a", "b"]).run()
 921       barrier.insert_many(1, keys=["k1"], values=[1]).run()
 922       barrier.insert_many(0, keys=["k3"], values=["c"]).run()
 923       barrier.insert_many(1, keys=["k3"], values=[3]).run()
 924       barrier.insert_many(1, keys=["k2"], values=[2]).run()
 925 
 926       (indices, keys, values) = barrier.take_many(2)
 927       (indices_val, keys_val, values0_val, values1_val) =
 928          session.run([indices, keys, values[0], values[1]])
 929 
 930     The output will be (up to permutation of "k1" and "k2"):
 931 
 932       indices_val == (-2**63, -2**63)
 933       keys_val == ("k1", "k2")
 934       values0_val == ("a", "b")
 935       values1_val == (1, 2)
 936 
 937     Note the key "k2" was inserted into the barrier before "k3".  Even though
 938     "k3" was completed first, both are complete by the time
 939     take_many is called.  As a result, "k2" is prioritized and "k1" and "k2"
 940     are returned first.  "k3" remains in the barrier until the next execution
 941     of `take_many`.  Since "k1" and "k2" had their first insertions into
 942     the barrier together, their indices are the same (-2**63).  The index
 943     of "k3" will be -2**63 + 1, because it was the next new inserted key.
 944 
 945     Args:
 946       types: A single dtype or a tuple of dtypes, corresponding to the
 947         dtypes of the tensor elements that comprise a value in this barrier.
 948       shapes: Optional. Constraints on the shapes of tensors in the values:
 949         a single tensor shape tuple; a tuple of tensor shape tuples
 950         for each barrier-element tuple component; or None if the shape should
 951         not be constrained.
 952       shared_name: Optional. If non-empty, this barrier will be shared under
 953         the given name across multiple sessions.
 954       name: Optional name for the barrier op.
 955 
 956     Raises:
 957       ValueError: If one of the `shapes` indicate no elements.
 958     """
 959     self._types = _as_type_list(types)
 960 
 961     if shapes is not None:
 962       shapes = _as_shape_list(shapes, self._types)
 963       self._shapes = [tensor_shape.TensorShape(s) for s in shapes]
 964       for i, shape in enumerate(self._shapes):
 965         if shape.num_elements() == 0:
 966           raise ValueError("Empty tensors are not supported, but received "
 967                            "shape '%s' at index %d" % (shape, i))
 968     else:
 969       self._shapes = [tensor_shape.unknown_shape() for _ in self._types]
 970 
 971     self._barrier_ref = gen_data_flow_ops.barrier(
 972         component_types=self._types,
 973         shapes=self._shapes,
 974         shared_name=shared_name,
 975         name=name)
 976     if context.executing_eagerly():
 977       self._name = context.context().scope_name
 978     else:
 979       self._name = self._barrier_ref.op.name.split("/")[-1]

1049         the given element was inserted into the Barrier, starting with
1050         the value -2**63.  Note, this value is different from the
1051         index of the insert_many call for which the element was completed.
1052       "key" is a string tensor of length num_elements containing the keys.

1068     # the Barrier object.
1069     if not context.executing_eagerly():

1079         output.set_shape(

1544   def _check_put_dtypes(self, vals, indices=None):
1545     """Validate and convert `vals` to a list of `Tensor`s.
1546 
1547     The `vals` argument can be a Tensor, a list or tuple of tensors, or a
1548     dictionary with tensor values.
1549 
1550     If `vals` is a list, then the appropriate indices associated with the
1551     values must be provided.
1552 
1553     If it is a dictionary, the staging area must have been constructed with a
1554     `names` attribute and the dictionary keys must match the staging area names.
1555     `indices` will be inferred from the dictionary keys.
1556     If the staging area was constructed with a `names` attribute, `vals` must
1557     be a dictionary.
1558 
1559     Checks that the dtype and shape of each value matches that
1560     of the staging area.
1561 
1562     Args:
1563       vals: A tensor, a list or tuple of tensors, or a dictionary.
1564 
1565     Returns:
1566       A (tensors, indices) tuple where `tensors` is a list of `Tensor` objects
1567       and `indices` is a list of indices associed with the tensors.
1568 
1569     Raises:
1570       ValueError: If `vals` or `indices` is invalid.
1571     """
1572     if isinstance(vals, dict):
1573       if not self._names:
1574         raise ValueError(
1575             "Staging areas must have names to enqueue a dictionary")
1576       if not set(vals.keys()).issubset(self._names):
1577         raise ValueError("Keys in dictionary to put do not match names "
1578                          "of staging area. Dictionary: (%s), Queue: (%s)" %
1579                          (sorted(vals.keys()), sorted(self._names)))
1580       # The order of values in `self._names` indicates the order in which the
1581       # tensors in the dictionary `vals` must be listed.
1582       vals, indices, _ = zip(*[(vals[k], i, k)
1583                                for i, k in enumerate(self._names)
1584                                if k in vals])
1585     else:
1586       if self._names:
1587         raise ValueError("You must enqueue a dictionary in a staging area "
1588                          "with names")
1589 
1590       if indices is None:
1591         raise ValueError("Indices must be supplied when inserting a list "
1592                          "of tensors")
1593 
1594       if len(indices) != len(vals):
1595         raise ValueError("Number of indices '%s' doesn't match "
1596                          "number of values '%s'")
1597 
1598       if not isinstance(vals, (list, tuple)):
1599         vals = [vals]
1600         indices = [0]
1601 
1602     # Sanity check number of values
1603     if not len(vals) <= len(self._dtypes):
1604       raise ValueError("Unexpected number of inputs '%s' vs '%s'" %
1605                        (len(vals), len(self._dtypes)))
1606 
1607     tensors = []
1608 
1609     for val, i in zip(vals, indices):
1610       dtype, shape = self._dtypes[i], self._shapes[i]
1611       # Check dtype
1612       if val.dtype != dtype:
1613         raise ValueError("Datatypes do not match. '%s' != '%s'" %
1614                          (str(val.dtype), str(dtype)))
1615 
1616       # Check shape
1617       val.get_shape().assert_is_compatible_with(shape)
1618 
1619       tensors.append(
1620           ops.convert_to_tensor(val, dtype=dtype, name="component_%d" % i))
1621 
1622     return tensors, indices

1910 class MapStagingArea(BaseStagingArea):
1911   """A `MapStagingArea` is a TensorFlow data structure that stores tensors
1912   across multiple steps, and exposes operations that can put and get tensors.
1913 
1914   Each `MapStagingArea` element is a (key, value) pair.
1915   Only int64 keys are supported, other types should be
1916   hashed to produce a key.
1917   Values are a tuple of one or more tensors.
1918   Each tuple component has a static dtype,
1919   and may have a static shape.
1920 
1921   The capacity of a `MapStagingArea` may be bounded or unbounded.
1922   It supports multiple concurrent producers and consumers; and
1923   provides exactly-once delivery.
1924 
1925   Each value tuple of a `MapStagingArea` is a fixed-length tuple of tensors
1926   whose
1927   dtypes are described by `dtypes`, and whose shapes are optionally described
1928   by the `shapes` argument.
1929 
1930   If the `shapes` argument is specified, each component of a staging area
1931   element must have the respective fixed shape. If it is
1932   unspecified, different elements may have different shapes,
1933

1935 
1936    - put(key, values)
1937    - peek(key)         like dict.get(key)
1938    - get(key)          like dict.pop(key)
1939    - get(key=None)     like dict.popitem()
1940    - size()
1941    - clear()
1942

1944   get(key=None) will remove (key, value) pairs in increasing key order.
1945   Otherwise a hashtable
1946

1948   put(key, values) will block until space becomes available.
1949

1951   will block put(key, values) until space is available.
1952   This is mostly useful for limiting the number of tensors on
1953   devices such as GPUs.
1954

1956   (key, value) pair is not present in the staging area.
1957

1959   map until such time as all values associated with the key have
1960   been inserted. Once completed, this (key, value) pair will be
1961   inserted into the map. Data in the incomplete map

2295 class RecordInput(object):
2296   """RecordInput asynchronously reads and randomly yields TFRecords.
2297 
2298   A RecordInput Op will continuously read a batch of records asynchronously
2299   into a buffer of some fixed capacity. It can also asynchronously yield
2300   random records from this buffer.
2301 
2302   It will not start yielding until at least `buffer_size / 2` elements have been
2303   placed into the buffer so that sufficient randomization can take place.
2304 
2305   The order the files are read will be shifted each epoch by `shift_amount` so
2306   that the data is presented in a different order every epoch.
2307   """
2308 
2309   def __init__(self,
2310                file_pattern,
2311                batch_size=1,
2312                buffer_size=1,
2313                parallelism=1,
2314                shift_ratio=0,
2315                seed=0,
2316                name=None,
2317                batches=None,
2318                compression_type=None):
2319     """Constructs a RecordInput Op.
2320 
2321     Args:
2322       file_pattern: File path to the dataset, possibly containing wildcards.
2323         All matching files will be iterated over each epoch.
2324       batch_size: How many records to return at a time.
2325       buffer_size: The maximum number of records the buffer will contain.
2326       parallelism: How many reader threads to use for reading from files.
2327       shift_ratio: What percentage of the total number files to move the start
2328         file forward by each epoch.
2329       seed: Specify the random number seed used by generator that randomizes
2330         records.
2331       name: Optional name for the operation.
2332       batches: None by default, creating a single batch op. Otherwise specifies
2333         how many batches to create, which are returned as a list when
2334         `get_yield_op()` is called. An example use case is to split processing
2335         between devices on one computer.
2336       compression_type: The type of compression for the file. Currently ZLIB and
2337         GZIP are supported. Defaults to none.
2338 
2339     Raises:
2340       ValueError: If one of the arguments is invalid.
2341     """
2342     self._batch_size = batch_size
2343     if batches is not None:
2344       self._batch_size *= batches
2345     self._batches = batches
2346     self._file_pattern = file_pattern
2347     self._buffer_size = buffer_size
2348     self._parallelism = parallelism
2349     self._shift_ratio = shift_ratio
2350     self._seed = seed
2351     self._name = name
2352     self._compression_type = python_io.TFRecordCompressionType.NONE
2353     if compression_type is not None:
2354       self._compression_type = compression_type

2356   def get_yield_op(self):
2357     """Adds a node that yields a group of records every time it is executed.
2358     If RecordInput `batches` parameter is not None, it yields a list of
2359     record batches with the specified `batch_size`.
2360     """
2361     compression_type = python_io.TFRecordOptions.get_compression_type_string(
2362         python_io.TFRecordOptions(self._compression_type))
2363     records = gen_data_flow_ops.record_input(
2364         file_pattern=self._file_pattern,
2365         file_buffer_size=self._buffer_size,
2366         file_parallelism=self._parallelism,
2367         file_shuffle_shift_ratio=self._shift_ratio,
2368         batch_size=self._batch_size,
2369         file_random_seed=self._seed,
2370         compression_type=compression_type,
2371         name=self._name)
2372     if self._batches is None:
2373       return records
2374     else:
2375       with ops.name_scope(self._name):
2376         batch_list = [[] for _ in six.moves.range(self._batches)]
2377         records = array_ops.split(records, self._batch_size, 0)
2378         records = [array_ops.reshape(record, []) for record in records]
2379         for index, protobuf in zip(six.moves.range(len(records)), records):
2380           batch_index = index % self._batches
2381           batch_list[batch_index].append(protobuf)
2382         return batch_list
