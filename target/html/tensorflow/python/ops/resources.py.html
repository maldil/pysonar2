  34                                    ["handle", "create", "is_initialized"])

  37 def register_resource(handle, create_op, is_initialized_op, is_shared=True):
  38   """Registers a resource into the appropriate collections.
  39 
  40   This makes the resource findable in either the shared or local resources
  41   collection.
  42 
  43   Args:
  44    handle: op which returns a handle for the resource.
  45    create_op: op which initializes the resource.
  46    is_initialized_op: op which returns a scalar boolean tensor of whether
  47     the resource has been initialized.
  48    is_shared: if True, the resource gets added to the shared resource
  49     collection; otherwise it gets added to the local resource collection.
  50 
  51   """
  52   resource = _Resource(handle, create_op, is_initialized_op)
  53   if is_shared:
  54     ops.add_to_collection(ops.GraphKeys.RESOURCES, resource)
  55   else:
  56     ops.add_to_collection(ops.GraphKeys.LOCAL_RESOURCES, resource)
  57 
  58 
  59 def shared_resources():
  60   """Returns resources visible to all tasks in the cluster."""
  61   return ops.get_collection(ops.GraphKeys.RESOURCES)
  62 
  63 
  64 def local_resources():
  65   """Returns resources intended to be local to this session."""
  66   return ops.get_collection(ops.GraphKeys.LOCAL_RESOURCES)

  69 def report_uninitialized_resources(resource_list=None,
  70                                    name="report_uninitialized_resources"):
  71   """Returns the names of all uninitialized resources in resource_list.
  72 
  73   If the returned tensor is empty then all resources have been initialized.
  74 
  75   Args:
  76    resource_list: resources to check. If None, will use shared_resources() +
  77     local_resources().
  78    name: name for the resource-checking op.
  79 
  80   Returns:
  81    Tensor containing names of the handles of all resources which have not
  82    yet been initialized.
  83 
  84   """
  85   if resource_list is None:
  86     resource_list = shared_resources() + local_resources()
  87   with ops.name_scope(name):
  88     # Run all operations on CPU
  89     with ops.device("/cpu:0"):
  90       if not resource_list:
  91         # Return an empty tensor so we only need to check for returned tensor
  92         # size being 0 as an indication of model ready.
  93         return array_ops.constant([], dtype=dtypes.string)
  94       # Get a 1-D boolean tensor listing whether each resource is initialized.
  95       variables_mask = math_ops.logical_not(
  96           array_ops.stack([r.is_initialized for r in resource_list]))
  97       # Get a 1-D string tensor containing all the resource names.
  98       variable_names_tensor = array_ops.constant(
  99           [s.handle.name for s in resource_list])
 100       # Return a 1-D tensor containing all the names of uninitialized resources.
 101       return array_ops.boolean_mask(variable_names_tensor, variables_mask)

 104 @tf_should_use.should_use_result
 105 def initialize_resources(resource_list, name="init"):
 106   """Initializes the resources in the given list.
 107 
 108   Args:
 109    resource_list: list of resources to initialize.
 110    name: name of the initialization op.
 111 
 112   Returns:
 113    op responsible for initializing all resources.
 114   """
 115   if resource_list:
 116     return control_flow_ops.group(*[r.create for r in resource_list], name=name)
 117   return control_flow_ops.no_op(name=name)
