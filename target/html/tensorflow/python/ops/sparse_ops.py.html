  49 def _convert_to_sparse_tensor(sp_input):
  50   """Convert `sp_input` to `SparseTensor` and return it.
  51 
  52   Args:
  53     sp_input: `SparseTensor` or `SparseTensorValue`.
  54 
  55   Returns:
  56     `sp_input` converted to `SparseTensor`.
  57 
  58   Raises:
  59     ValueError: if `sp_input` is neither `SparseTensor` nor `SparseTensorValue`.
  60   """
  61   if isinstance(sp_input, sparse_tensor.SparseTensorValue):
  62     return sparse_tensor.SparseTensor.from_value(sp_input)
  63   if not isinstance(sp_input, sparse_tensor.SparseTensor):
  64     raise TypeError("Input must be a SparseTensor.")
  65   return sp_input

  89 def _make_int64_tensor(value, name):
  90   if isinstance(value, compat.integral_types):
  91     return ops.convert_to_tensor(value, name=name, dtype=dtypes.int64)
  92   if not isinstance(value, ops.Tensor):
  93     raise TypeError("{} must be an integer value".format(name))

  99 @tf_export("sparse.expand_dims")
 100 def sparse_expand_dims(sp_input, axis=None, name=None):
 101   """Inserts a dimension of 1 into a tensor's shape.
 102 
 103   Given a tensor `sp_input`, this operation inserts a dimension of 1 at the
 104   dimension index `axis` of `sp_input`'s shape. The dimension index `axis`
 105   starts at zero; if you specify a negative number for `axis` it is counted
 106   backwards from the end.
 107 
 108   Args:
 109     sp_input: A `SparseTensor`.
 110     axis: 0-D (scalar). Specifies the dimension index at which to expand the
 111       shape of `input`. Must be in the range `[-rank(sp_input) - 1,
 112       rank(sp_input)]`.
 113     name: The name of the output `SparseTensor`.
 114 
 115   Returns:
 116     A `SparseTensor` with the same data as `sp_input`, but its shape has an
 117     additional dimension of size 1 added.
 118   """
 119   rank = sp_input.dense_shape.get_shape()[0]
 120   axis = -1 if axis is None else axis
 121 
 122   with ops.name_scope(name, default_name="expand_dims", values=[sp_input]):
 123     if isinstance(axis, compat.integral_types):
 124       axis = ops.convert_to_tensor(axis, name="axis", dtype=dtypes.int32)
 125     elif not isinstance(axis, ops.Tensor):
 126       raise TypeError("axis must be an integer value in range [-rank(sp_input)"
 127                       " - 1, rank(sp_input)]")
 128 
 129     # Convert axis to a positive value if it is negative.
 130     axis = array_ops.where(axis >= 0, axis, axis + rank + 1)
 131 
 132     # Create the new column of indices for the sparse tensor by slicing
 133     # the indices and inserting a new column of indices for the new dimension.
 134     column_size = array_ops.shape(sp_input.indices)[0]
 135     new_index = array_ops.zeros([column_size, 1], dtype=dtypes.int64)
 136     indices_before = array_ops.slice(sp_input.indices, [0, 0], [-1, axis])
 137     indices_after = array_ops.slice(sp_input.indices, [0, axis], [-1, -1])
 138     indices = array_ops.concat(
 139         [indices_before, new_index, indices_after], axis=1)
 140 
 141     # Create the new dense shape by splicing the tensor [1] in the correct
 142     # dimension of the existing shape.
 143     shape_before = array_ops.slice(sp_input.dense_shape, [0], [axis])
 144     shape_after = array_ops.slice(sp_input.dense_shape, [axis], [-1])
 145     new_shape = ops.convert_to_tensor([1], name="new_shape", dtype=dtypes.int64)
 146     shape = array_ops.concat([shape_before, new_shape, shape_after], axis=0)
 147 
 148     # Create the output sparse tensor.
 149     return sparse_tensor.SparseTensor(
 150         indices=indices, values=sp_input.values, dense_shape=shape)

 153 @tf_export("sparse.eye")
 154 def sparse_eye(num_rows,
 155                num_columns=None,
 156                dtype=dtypes.float32,
 157                name=None):
 158   """Creates a two-dimensional sparse tensor with ones along the diagonal.
 159 
 160   Args:
 161     num_rows: Non-negative integer or `int32` scalar `tensor` giving the number
 162       of rows in the resulting matrix.
 163     num_columns: Optional non-negative integer or `int32` scalar `tensor` giving
 164       the number of columns in the resulting matrix. Defaults to `num_rows`.
 165     dtype: The type of element in the resulting `Tensor`.
 166     name: A name for this `Op`. Defaults to "eye".
 167 
 168   Returns:
 169     A `SparseTensor` of shape [num_rows, num_columns] with ones along the
 170     diagonal.
 171   """
 172   with ops.name_scope(name, default_name="eye", values=[num_rows, num_columns]):
 173     num_rows = _make_int64_tensor(num_rows, "num_rows")
 174     num_columns = num_rows if num_columns is None else _make_int64_tensor(
 175         num_columns, "num_columns")
 176 
 177     # Create the sparse tensor.
 178     diag_size = math_ops.minimum(num_rows, num_columns)
 179     diag_range = math_ops.range(diag_size, dtype=dtypes.int64)
 180 
 181     return sparse_tensor.SparseTensor(
 182         indices=array_ops.stack([diag_range, diag_range], axis=1),
 183         values=array_ops.ones(diag_size, dtype=dtype),
 184         dense_shape=[num_rows, num_columns])

 188 @tf_export("sparse.concat", "sparse_concat")
 189 @deprecation.deprecated_endpoints("sparse_concat")
 190 @deprecation.deprecated_args(
 191     None, "concat_dim is deprecated, use axis instead", "concat_dim")
 192 def sparse_concat(axis,
 193                   sp_inputs,
 194                   name=None,
 195                   expand_nonconcat_dim=False,
 196                   concat_dim=None):
 197   """Concatenates a list of `SparseTensor` along the specified dimension.
 198 
 199   Concatenation is with respect to the dense versions of each sparse input.
 200   It is assumed that each inputs is a `SparseTensor` whose elements are ordered
 201   along increasing dimension number.
 202 
 203   If expand_nonconcat_dim is False, all inputs' shapes must match, except for
 204   the concat dimension. If expand_nonconcat_dim is True, then inputs' shapes are
 205   allowed to vary among all inputs.
 206 
 207   The `indices`, `values`, and `shapes` lists must have the same length.
 208 
 209   If expand_nonconcat_dim is False, then the output shape is identical to the
 210   inputs', except along the concat dimension, where it is the sum of the inputs'
 211   sizes along that dimension.
 212 
 213   If expand_nonconcat_dim is True, then the output shape along the non-concat
 214   dimensions will be expand to be the largest among all inputs, and it is the
 215   sum of the inputs sizes along the concat dimension.
 216 
 217   The output elements will be resorted to preserve the sort order along
 218   increasing dimension number.
 219 
 220   This op runs in `O(M log M)` time, where `M` is the total number of non-empty
 221   values across all inputs. This is due to the need for an internal sort in
 222   order to concatenate efficiently across an arbitrary dimension.
 223 
 224   For example, if `axis = 1` and the inputs are
 225 
 226       sp_inputs[0]: shape = [2, 3]
 227       [0, 2]: "a"
 228       [1, 0]: "b"
 229       [1, 1]: "c"
 230 
 231       sp_inputs[1]: shape = [2, 4]
 232       [0, 1]: "d"
 233       [0, 2]: "e"
 234 
 235   then the output will be
 236 
 237       shape = [2, 7]
 238       [0, 2]: "a"
 239       [0, 4]: "d"
 240       [0, 5]: "e"
 241       [1, 0]: "b"
 242       [1, 1]: "c"
 243 
 244   Graphically this is equivalent to doing
 245 
 246       [    a] concat [  d e  ] = [    a   d e  ]
 247       [b c  ]        [       ]   [b c          ]
 248 
 249   Another example, if 'axis = 1' and the inputs are
 250 
 251       sp_inputs[0]: shape = [3, 3]
 252       [0, 2]: "a"
 253       [1, 0]: "b"
 254       [2, 1]: "c"
 255 
 256       sp_inputs[1]: shape = [2, 4]
 257       [0, 1]: "d"
 258       [0, 2]: "e"
 259 
 260   if expand_nonconcat_dim = False, this will result in an error. But if
 261   expand_nonconcat_dim = True, this will result in:
 262 
 263       shape = [3, 7]
 264       [0, 2]: "a"
 265       [0, 4]: "d"
 266       [0, 5]: "e"
 267       [1, 0]: "b"
 268       [2, 1]: "c"
 269 
 270   Graphically this is equivalent to doing
 271 
 272       [    a] concat [  d e  ] = [    a   d e  ]
 273       [b    ]        [       ]   [b            ]
 274       [  c  ]                    [  c          ]
 275 
 276 
 277   Args:
 278     axis: Dimension to concatenate along. Must be in range [-rank, rank),
 279       where rank is the number of dimensions in each input `SparseTensor`.
 280     sp_inputs: List of `SparseTensor` to concatenate.
 281     name: A name prefix for the returned tensors (optional).
 282     expand_nonconcat_dim: Whether to allow the expansion in the non-concat
 283       dimensions. Defaulted to False.
 284     concat_dim: The old (deprecated) name for axis.
 285 
 286   Returns:
 287     A `SparseTensor` with the concatenated output.
 288 
 289   Raises:
 290     TypeError: If `sp_inputs` is not a list of `SparseTensor`.
 291   """
 292   axis = deprecation.deprecated_argument_lookup("axis", axis, "concat_dim",
 293                                                 concat_dim)
 294   sp_inputs = _convert_to_sparse_tensors(sp_inputs)
 295 
 296   if len(sp_inputs) == 1:  # Degenerate case of one tensor.
 297     return sp_inputs[0]
 298 
 299   inds = [sp_input.indices for sp_input in sp_inputs]
 300   vals = [sp_input.values for sp_input in sp_inputs]
 301   shapes = [sp_input.dense_shape for sp_input in sp_inputs]
 302 
 303   if expand_nonconcat_dim:
 304     max_shape = math_ops.reduce_max(
 305         array_ops.concat(
 306             [array_ops.reshape(shape, [1, -1]) for shape in shapes], 0), 0)
 307     shapes = [
 308         array_ops.concat([
 309             max_shape[:axis], shape[-1:]
 310             if axis == -1 else shape[axis:axis + 1], []
 311             if axis == -1 else max_shape[axis + 1:]
 312         ], 0) for shape in shapes
 313     ]
 314 
 315   output_ind, output_val, output_shape = (
 316       gen_sparse_ops.sparse_concat(inds, vals, shapes, axis, name=name))
 317 
 318   return sparse_tensor.SparseTensor(output_ind, output_val, output_shape)

 321 @tf_export("sparse.add", "sparse_add")
 322 @deprecation.deprecated_endpoints("sparse_add")
 323 def sparse_add(a, b, thresh=0):
 324   """Adds two tensors, at least one of each is a `SparseTensor`.
 325 
 326   If one `SparseTensor` and one `Tensor` are passed in, returns a `Tensor`.  If
 327   both arguments are `SparseTensor`s, this returns a `SparseTensor`.  The order
 328   of arguments does not matter.  Use vanilla `tf.add()` for adding two dense
 329   `Tensor`s.
 330 
 331   The shapes of the two operands must match: broadcasting is not supported.
 332 
 333   The indices of any input `SparseTensor` are assumed ordered in standard
 334   lexicographic order.  If this is not the case, before this step run
 335   `SparseReorder` to restore index ordering.
 336 
 337   If both arguments are sparse, we perform "clipping" as follows.  By default,
 338   if two values sum to zero at some index, the output `SparseTensor` would still
 339   include that particular location in its index, storing a zero in the
 340   corresponding value slot.  To override this, callers can specify `thresh`,
 341   indicating that if the sum has a magnitude strictly smaller than `thresh`, its
 342   corresponding value and index would then not be included.  In particular,
 343   `thresh == 0.0` (default) means everything is kept and actual thresholding
 344   happens only for a positive value.
 345 
 346   For example, suppose the logical sum of two sparse operands is (densified):
 347 
 348       [       2]
 349       [.1     0]
 350       [ 6   -.2]
 351 
 352   Then,
 353 
 354       * `thresh == 0` (the default): all 5 index/value pairs will be returned.
 355       * `thresh == 0.11`: only .1 and 0 will vanish, and the remaining three
 356           index/value pairs will be returned.
 357       * `thresh == 0.21`: .1, 0, and -.2 will vanish.
 358 
 359   Args:
 360     a: The first operand; `SparseTensor` or `Tensor`.
 361     b: The second operand; `SparseTensor` or `Tensor`.  At least one operand
 362       must be sparse.
 363     thresh: A 0-D `Tensor`.  The magnitude threshold that determines if an
 364     output value/index pair takes space.  Its dtype should match that of the
 365     values if they are real; if the latter are complex64/complex128, then the
 366     dtype should be float32/float64, correspondingly.
 367 
 368   Returns:
 369     A `SparseTensor` or a `Tensor`, representing the sum.
 370 
 371   Raises:
 372     TypeError: If both `a` and `b` are `Tensor`s.  Use `tf.add()` instead.
 373   """
 374   sparse_classes = (sparse_tensor.SparseTensor, sparse_tensor.SparseTensorValue)
 375   if not any(isinstance(inp, sparse_classes) for inp in [a, b]):
 376     raise TypeError("At least one input should be SparseTensor; do you mean to"
 377                     " use tf.add()?")
 378 
 379   if all(isinstance(inp, sparse_classes) for inp in [a, b]):
 380     a = _convert_to_sparse_tensor(a)
 381     b = _convert_to_sparse_tensor(b)
 382     thresh = ops.convert_to_tensor(
 383         thresh, dtype=a.values.dtype.real_dtype.base_dtype, name="thresh")
 384     output_ind, output_val, output_shape = (
 385         gen_sparse_ops.sparse_add(a.indices, a.values, a.dense_shape,
 386                                   b.indices, b.values, b.dense_shape, thresh))
 387 
 388     # Attempt to get output_shape statically.
 389     a.get_shape().assert_is_compatible_with(b.get_shape())
 390     static_shape = array_ops.broadcast_static_shape(a.get_shape(),
 391                                                     b.get_shape())
 392     if static_shape.is_fully_defined():
 393       output_shape = static_shape.as_list()
 394 
 395     return sparse_tensor.SparseTensor(output_ind, output_val, output_shape)
 396   else:
 397     # swap to make `a` the SparseTensor.
 398     if isinstance(b, sparse_classes):
 399       a, b = b, a
 400     return gen_sparse_ops.sparse_tensor_dense_add(a.indices, a.values,
 401                                                   a.dense_shape, b)

 487 def _sparse_cross_internal(inputs,
 488                            hashed_output=False,
 489                            num_buckets=0,
 490                            hash_key=None,
 491                            name=None):
 492   """See gen_sparse_ops.sparse_cross."""
 493   if not isinstance(inputs, list):
 494     raise TypeError("Inputs must be a list")
 495   if not all(
 496       isinstance(i, sparse_tensor.SparseTensor) or isinstance(i, ops.Tensor)
 497       for i in inputs):
 498     raise TypeError("All inputs must be SparseTensors")
 499 
 500   sparse_inputs = [
 501       i for i in inputs if isinstance(i, sparse_tensor.SparseTensor)
 502   ]
 503   dense_inputs = [
 504       i for i in inputs if not isinstance(i, sparse_tensor.SparseTensor)
 505   ]
 506 
 507   indices = [sp_input.indices for sp_input in sparse_inputs]
 508   values = [sp_input.values for sp_input in sparse_inputs]
 509   shapes = [sp_input.dense_shape for sp_input in sparse_inputs]
 510   out_type = dtypes.int64 if hashed_output else dtypes.string
 511 
 512   internal_type = dtypes.string
 513   for i in range(len(values)):
 514     if values[i].dtype != dtypes.string:
 515       values[i] = math_ops.to_int64(values[i])
 516       internal_type = dtypes.int64
 517   for i in range(len(dense_inputs)):
 518     if dense_inputs[i].dtype != dtypes.string:
 519       dense_inputs[i] = math_ops.to_int64(dense_inputs[i])
 520       internal_type = dtypes.int64
 521 
 522   indices_out, values_out, shape_out = gen_sparse_ops.sparse_cross(
 523       indices=indices,
 524       values=values,
 525       shapes=shapes,
 526       dense_inputs=dense_inputs,
 527       hashed_output=hashed_output,
 528       num_buckets=num_buckets,
 529       hash_key=hash_key or _DEFAULT_HASH_KEY,
 530       out_type=out_type,
 531       internal_type=internal_type,
 532       name=name)
 533 
 534   return sparse_tensor.SparseTensor(indices_out, values_out, shape_out)

 537 def sparse_dense_cwise_add(sp_t, dense_t):
 538   """Adds up a SparseTensor and a dense Tensor, using these special rules:
 539 
 540   (1) Broadcasts the dense side to have the same shape as the sparse side, if
 541       eligible;
 542   (2) Then, only the dense values pointed to by the indices of the SparseTensor
 543       participate in the cwise addition.
 544 
 545   By the rules, the result is a logical SparseTensor with exactly the same
 546   indices and shape, but possibly with different non-zero values.  The output of
 547   this Op is the resultant non-zero values.
 548 
 549   Args:
 550     sp_t: the SparseTensor operand.
 551     dense_t: the dense Tensor operand; must have the same dtype and a
 552       broadcast-compatible shape as `sp_t`.
 553 
 554   Returns:
 555     output: the SparseTensor output.
 556   """
 557   result = gen_sparse_ops.sparse_dense_cwise_add(sp_t.indices, sp_t.values,
 558                                                  sp_t.dense_shape, dense_t)
 559   return sparse_tensor.SparseTensor(sp_t.indices, result, sp_t.dense_shape)

 631

 645       [8, 1]: e

 668         sp_input.indices, sp_input.dense_shape, shape, name=name)
 669

1235                  already_sorted=False):
1236   """Combines a batch of feature ids and values into a single `SparseTensor`.
1237 
1238   The most common use case for this function occurs when feature ids and
1239   their corresponding values are stored in `Example` protos on disk.
1240   `parse_example` will return a batch of ids and a batch of values, and this
1241   function joins them into a single logical `SparseTensor` for use in
1242   functions such as `sparse_tensor_dense_matmul`, `sparse_to_dense`, etc.
1243 
1244   The `SparseTensor` returned by this function has the following properties:
1245 
1246     - `indices` is equivalent to `sp_ids.indices` with the last
1247       dimension discarded and replaced with `sp_ids.values`.
1248     - `values` is simply `sp_values.values`.
1249     - If `sp_ids.dense_shape = [D0, D1, ..., Dn, K]`, then
1250       `output.shape = [D0, D1, ..., Dn, vocab_size]`.
1251

1253 
1254   ```python
1255     vector1 = [-3, 0, 0, 0, 0, 0]
1256     vector2 = [ 0, 1, 0, 4, 1, 0]
1257     vector3 = [ 5, 0, 0, 9, 0, 0]
1258   ```
1259

1261   only the feature ids (column number if the vectors are treated as a matrix)
1262   of the non-zero elements and the corresponding values:
1263

1265     examples = [Example(features={

1268                 Example(features={
1269                     "ids": Feature(int64_list=Int64List(value=[1, 4, 3])),
1270                     "values": Feature(float_list=FloatList(value=[1, 1, 4]))}),
1271                 Example(features={
1272                     "ids": Feature(int64_list=Int64List(value=[0, 3])),
1273                     "values": Feature(float_list=FloatList(value=[5, 9]))})]
1274   ```

1277   dictionary with entries for "ids" and "values". Passing those two objects
1278   to this function along with vocab_size=6, will produce a `SparseTensor` that
1279   sparsely represents all three instances. Namely, the `indices` property will

1281   first dimension is the row number in the matrix, i.e., the index within the
1282   batch, and the second dimension is the column number, i.e., the feature id);
1283   `values` will contain the actual values. `shape` will be the shape of the
1284   original matrix, i.e., (3, 6). For our example above, the output will be
1285   equal to:
1286 
1287   ```python
1288     SparseTensor(indices=[[0, 0], [1, 1], [1, 3], [1, 4], [2, 0], [2, 3]],

1291   ```

1294   both the sp_ids as well as the vocab_size.

1297       dimension discarded and concatenated with

1322       a `SparseTensor` nor a list thereof. Or if `vocab_size` is not a
1323       `Tensor` or a Python int and `sp_ids` is a `SparseTensor`. Or if

1325     ValueError: If `sp_ids` and `vocab_size` are lists of different lengths.
1326   """

1329     sp_ids = [sp_ids]
1330     if not (isinstance(vocab_size, ops.Tensor) or
1331             isinstance(vocab_size, numbers.Integral)):
1332       raise TypeError("vocab_size has to be a Tensor or Python int. Found %s" %
1333                       type(vocab_size))

1335   else:

1338                       "Found %s" % type(sp_ids))
1339     if not isinstance(vocab_size, collections.Iterable):
1340       raise TypeError("vocab_size has to be a list of Tensors or Python ints. "
1341                       "Found %s" % type(vocab_size))
1342     for dim in vocab_size:
1343       if not (isinstance(dim, ops.Tensor) or isinstance(dim, numbers.Integral)):

1363     indices_columns_to_preserve = sp_ids[0].indices[:, :-1]

1365 
1366     new_values = sp_values.values
1367     new_shape = array_ops.concat([sp_ids[0].dense_shape[:-1], vocab_size], 0)
1368 
1369     result = sparse_tensor.SparseTensor(new_indices, new_values, new_shape)
1370     if already_sorted:

2010   """Applies softmax to a batched N-D `SparseTensor`.
2011

2013   (where `N >= 2`), and with indices sorted in the canonical lexicographic
2014   order.
2015

2017   innermost logical submatrix with shape `[B, C]`, but with the catch that *the
2018   implicitly zero elements do not participate*.  Specifically, the algorithm is
2019   equivalent to:
2020 
2021     (1) Applies `tf.nn.softmax()` to a densified view of each innermost
2022         submatrix with shape `[B, C]`, along the size-C dimension;
2023     (2) Masks out the original implicitly-zero locations;
2024     (3) Renormalizes the remaining elements.
2025 
2026   Hence, the `SparseTensor` result has exactly the same non-zero indices and
2027   shape.
2028 
2029   Example:
2030

2032   # First batch:
2033   # [?   e.]
2034   # [1.  ? ]
2035   # Second batch:
2036   # [e   ? ]
2037   # [e   e ]
2038   shape = [2, 2, 2]  # 3-D SparseTensor
2039   values = np.asarray([[[0., np.e], [1., 0.]], [[np.e, 0.], [np.e, np.e]]])
2040   indices = np.vstack(np.where(values)).astype(np.int64).T
2041 
2042   result = tf.sparse_softmax(tf.SparseTensor(indices, values, shape))
2043   # ...returning a 3-D SparseTensor, equivalent to:
2044   # [?   1.]     [1    ?]
2045   # [1.  ? ] and [.5  .5]
2046   # where ? means implicitly zero.
2047   ```

2270                                        sparse_handles,
2271                                        rank=None,
2272                                        name=None):
2273   """Read `SparseTensors` from a `SparseTensorsMap` and concatenate them.
2274 
2275   The input `sparse_handles` must be a string matrix of shape `[N, 1]` where
2276   `N` is the minibatch size and the rows correspond to packed outputs of
2277   `add_sparse_to_tensors_map`.  The ranks of the original `SparseTensor` objects
2278   must all match.  When the final `SparseTensor` is created, it has rank one
2279   higher than the ranks of the incoming `SparseTensor` objects (they have been
2280   concatenated along a new row dimension).
2281 
2282   The output `SparseTensor` object's shape values for all dimensions but the
2283   first are the max across the input `SparseTensor` objects' shape values
2284   for the corresponding dimensions.  Its first shape value is `N`, the minibatch
2285   size.
2286 
2287   The input `SparseTensor` objects' indices are assumed ordered in
2288   standard lexicographic order.  If this is not the case, after this
2289   step run `sparse.reorder` to restore index ordering.

2292   original `SparseTensor` objects:
2293 
2294       index = [ 0]
2295               [10]
2296               [20]
2297       values = [1, 2, 3]

2302       index = [ 2]
2303               [10]
2304       values = [4, 5]
2305       shape = [30]

2308 
2309       index = [0  0]
2310               [0 10]
2311               [0 20]
