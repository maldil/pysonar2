  31 def _infer_fft_length_for_rfft(input_tensor, fft_rank):
  32   """Infers the `fft_length` argument for a `rank` RFFT from `input_tensor`."""
  33   # A TensorShape for the inner fft_rank dimensions.
  34   fft_shape = input_tensor.get_shape()[-fft_rank:]
  35 
  36   # If any dim is unknown, fall back to tensor-based math.
  37   if not fft_shape.is_fully_defined():
  38     return _array_ops.shape(input_tensor)[-fft_rank:]
  39 
  40   # Otherwise, return a constant.
  41   return _ops.convert_to_tensor(fft_shape.as_list(), _dtypes.int32)

  44 def _infer_fft_length_for_irfft(input_tensor, fft_rank):
  45   """Infers the `fft_length` argument for a `rank` IRFFT from `input_tensor`."""
  46   # A TensorShape for the inner fft_rank dimensions.
  47   fft_shape = input_tensor.get_shape()[-fft_rank:]
  48 
  49   # If any dim is unknown, fall back to tensor-based math.
  50   if not fft_shape.is_fully_defined():
  51     fft_length = _array_ops.unstack(_array_ops.shape(input_tensor)[-fft_rank:])
  52     fft_length[-1] = _math_ops.maximum(0, 2 * (fft_length[-1] - 1))
  53     return _array_ops.stack(fft_length)
  54 
  55   # Otherwise, return a constant.
  56   fft_length = fft_shape.as_list()
  57   if fft_length:
  58     fft_length[-1] = max(0, 2 * (fft_length[-1] - 1))
  59   return _ops.convert_to_tensor(fft_length, _dtypes.int32)

  62 def _maybe_pad_for_rfft(input_tensor, fft_rank, fft_length, is_reverse=False):
  63   """Pads `input_tensor` to `fft_length` on its inner-most `fft_rank` dims."""
  64   fft_shape = _tensor_util.constant_value_as_shape(fft_length)
  65 
  66   # Edge case: skip padding empty tensors.
  67   if (input_tensor.shape.ndims is not None and
  68       any(dim.value == 0 for dim in input_tensor.shape)):
  69     return input_tensor
  70 
  71   # If we know the shapes ahead of time, we can either skip or pre-compute the
  72   # appropriate paddings. Otherwise, fall back to computing paddings in
  73   # TensorFlow.
  74   if fft_shape.is_fully_defined() and input_tensor.shape.ndims is not None:
  75     # Slice the last FFT-rank dimensions from input_tensor's shape.
  76     input_fft_shape = input_tensor.shape[-fft_shape.ndims:]
  77 
  78     if input_fft_shape.is_fully_defined():
  79       # In reverse, we only pad the inner-most dimension to fft_length / 2 + 1.
  80       if is_reverse:
  81         fft_shape = fft_shape[:-1].concatenate(fft_shape[-1].value // 2 + 1)
  82 
  83       paddings = [[0, max(fft_dim.value - input_dim.value, 0)]
  84                   for fft_dim, input_dim in zip(fft_shape, input_fft_shape)]
  85       if any(pad > 0 for _, pad in paddings):
  86         outer_paddings = [[0, 0]] * max((input_tensor.shape.ndims -
  87                                          fft_shape.ndims), 0)
  88         return _array_ops.pad(input_tensor, outer_paddings + paddings)
  89       return input_tensor
  90 
  91   # If we can't determine the paddings ahead of time, then we have to pad. If
  92   # the paddings end up as zero, tf.pad has a special-case that does no work.
  93   input_rank = _array_ops.rank(input_tensor)
  94   input_fft_shape = _array_ops.shape(input_tensor)[-fft_rank:]
  95   outer_dims = _math_ops.maximum(0, input_rank - fft_rank)
  96   outer_paddings = _array_ops.zeros([outer_dims], fft_length.dtype)
  97   # In reverse, we only pad the inner-most dimension to fft_length / 2 + 1.
  98   if is_reverse:
  99     fft_length = _array_ops.concat([fft_length[:-1],
 100                                     fft_length[-1:] // 2 + 1], 0)
 101   fft_paddings = _math_ops.maximum(0, fft_length - input_fft_shape)
 102   paddings = _array_ops.concat([outer_paddings, fft_paddings], 0)
 103   paddings = _array_ops.stack([_array_ops.zeros_like(paddings), paddings],
 104                               axis=1)
 105   return _array_ops.pad(input_tensor, paddings)

 108 def _rfft_wrapper(fft_fn, fft_rank, default_name):
 109   """Wrapper around gen_spectral_ops.rfft* that infers fft_length argument."""
 110 
 111   def _rfft(input_tensor, fft_length=None, name=None):
 112     with _ops.name_scope(name, default_name,
 113                          [input_tensor, fft_length]) as name:
 114       input_tensor = _ops.convert_to_tensor(input_tensor, _dtypes.float32)
 115       input_tensor.shape.with_rank_at_least(fft_rank)
 116       if fft_length is None:
 117         fft_length = _infer_fft_length_for_rfft(input_tensor, fft_rank)
 118       else:
 119         fft_length = _ops.convert_to_tensor(fft_length, _dtypes.int32)
 120       input_tensor = _maybe_pad_for_rfft(input_tensor, fft_rank, fft_length)
 121       return fft_fn(input_tensor, fft_length, name)
 122   _rfft.__doc__ = fft_fn.__doc__
 123   return _rfft

 126 def _irfft_wrapper(ifft_fn, fft_rank, default_name):
 127   """Wrapper around gen_spectral_ops.irfft* that infers fft_length argument."""
 128 
 129   def _irfft(input_tensor, fft_length=None, name=None):
 130     with _ops.name_scope(name, default_name,
 131                          [input_tensor, fft_length]) as name:
 132       input_tensor = _ops.convert_to_tensor(input_tensor, _dtypes.complex64)
 133       input_tensor.shape.with_rank_at_least(fft_rank)
 134       if fft_length is None:
 135         fft_length = _infer_fft_length_for_irfft(input_tensor, fft_rank)
 136       else:
 137         fft_length = _ops.convert_to_tensor(fft_length, _dtypes.int32)
 138       input_tensor = _maybe_pad_for_rfft(input_tensor, fft_rank, fft_length,
 139                                          is_reverse=True)
 140       return ifft_fn(input_tensor, fft_length, name)
 141   _irfft.__doc__ = ifft_fn.__doc__
 142   return _irfft

 165 def _validate_dct_arguments(dct_type, n, axis, norm):
 166   if n is not None:
 167     raise NotImplementedError("The DCT length argument is not implemented.")
 168   if axis != -1:
 169     raise NotImplementedError("axis must be -1. Got: %s" % axis)
 170   if dct_type not in (2, 3):
 171     raise ValueError("Only Types II and III (I)DCT are supported.")
 172   if norm not in (None, "ortho"):
 173     raise ValueError(
 174         "Unknown normalization. Expected None or 'ortho', got: %s" % norm)

 178 @tf_export("spectral.dct")
 179 def dct(input, type=2, n=None, axis=-1, norm=None, name=None):  # pylint: disable=redefined-builtin
 180   """Computes the 1D [Discrete Cosine Transform (DCT)][dct] of `input`.
 181 
 182   Currently only Types II and III are supported. Type II is implemented using a
 183   length `2N` padded `tf.spectral.rfft`, as described here:
 184   https://dsp.stackexchange.com/a/10606. Type III is a fairly straightforward
 185   inverse of Type II (i.e. using a length `2N` padded `tf.spectral.irfft`).
 186 
 187   @compatibility(scipy)
 188   Equivalent to scipy.fftpack.dct for Type-II and Type-III DCT.
 189   https://docs.scipy.org/doc/scipy-0.14.0/reference/generated/scipy.fftpack.dct.html
 190   @end_compatibility
 191 
 192   Args:
 193     input: A `[..., samples]` `float32` `Tensor` containing the signals to
 194       take the DCT of.
 195     type: The DCT type to perform. Must be 2 or 3.
 196     n: For future expansion. The length of the transform. Must be `None`.
 197     axis: For future expansion. The axis to compute the DCT along. Must be `-1`.
 198     norm: The normalization to apply. `None` for no normalization or `'ortho'`
 199       for orthonormal normalization.
 200     name: An optional name for the operation.
 201 
 202   Returns:
 203     A `[..., samples]` `float32` `Tensor` containing the DCT of `input`.
 204 
 205   Raises:
 206     ValueError: If `type` is not `2` or `3`, `n` is not `None, `axis` is not
 207       `-1`, or `norm` is not `None` or `'ortho'`.
 208 
 209   [dct]: https://en.wikipedia.org/wiki/Discrete_cosine_transform
 210   """
 211   _validate_dct_arguments(type, n, axis, norm)
 212   with _ops.name_scope(name, "dct", [input]):
 213     # We use the RFFT to compute the DCT and TensorFlow only supports float32
 214     # for FFTs at the moment.
 215     input = _ops.convert_to_tensor(input, dtype=_dtypes.float32)
 216 
 217     axis_dim = input.shape[-1].value or _array_ops.shape(input)[-1]
 218     axis_dim_float = _math_ops.to_float(axis_dim)
 219     if type == 2:
 220       scale = 2.0 * _math_ops.exp(
 221           _math_ops.complex(
 222               0.0, -_math_ops.range(axis_dim_float) * _math.pi * 0.5 /
 223               axis_dim_float))
 224 
 225       # TODO(rjryan): Benchmark performance and memory usage of the various
 226       # approaches to computing a DCT via the RFFT.
 227       dct2 = _math_ops.real(
 228           rfft(input, fft_length=[2 * axis_dim])[..., :axis_dim] * scale)
 229 
 230       if norm == "ortho":
 231         n1 = 0.5 * _math_ops.rsqrt(axis_dim_float)
 232         n2 = n1 * _math_ops.sqrt(2.0)
 233         # Use tf.pad to make a vector of [n1, n2, n2, n2, ...].
 234         weights = _array_ops.pad(
 235             _array_ops.expand_dims(n1, 0), [[0, axis_dim - 1]],
 236             constant_values=n2)
 237         dct2 *= weights
 238 
 239       return dct2
 240 
 241     elif type == 3:
 242       if norm == "ortho":
 243         n1 = _math_ops.sqrt(axis_dim_float)
 244         n2 = n1 * _math_ops.sqrt(0.5)
 245         # Use tf.pad to make a vector of [n1, n2, n2, n2, ...].
 246         weights = _array_ops.pad(
 247             _array_ops.expand_dims(n1, 0), [[0, axis_dim - 1]],
 248             constant_values=n2)
 249         input *= weights
 250       else:
 251         input *= axis_dim_float
 252       scale = 2.0 * _math_ops.exp(
 253           _math_ops.complex(
 254               0.0,
 255               _math_ops.range(axis_dim_float) * _math.pi * 0.5 /
 256               axis_dim_float))
 257       dct3 = _math_ops.real(
 258           irfft(
 259               scale * _math_ops.complex(input, 0.0),
 260               fft_length=[2 * axis_dim]))[..., :axis_dim]
 261 
 262       return dct3

 267 def idct(input, type=2, n=None, axis=-1, norm=None, name=None):  # pylint: disable=redefined-builtin
 268   """Computes the 1D [Inverse Discrete Cosine Transform (DCT)][idct] of `input`.
 269 
 270   Currently only Types II and III are supported. Type III is the inverse of
 271   Type II, and vice versa.
 272 
 273   Note that you must re-normalize by 1/(2n) to obtain an inverse if `norm` is
 274   not `'ortho'`. That is:
 275   `signal == idct(dct(signal)) * 0.5 / signal.shape[-1]`.
 276   When `norm='ortho'`, we have:
 277   `signal == idct(dct(signal, norm='ortho'), norm='ortho')`.
 278 
 279   @compatibility(scipy)
 280   Equivalent to scipy.fftpack.idct for Type-II and Type-III DCT.
 281   https://docs.scipy.org/doc/scipy-0.14.0/reference/generated/scipy.fftpack.idct.html
 282   @end_compatibility
 283 
 284   Args:
 285     input: A `[..., samples]` `float32` `Tensor` containing the signals to take
 286       the DCT of.
 287     type: The IDCT type to perform. Must be 2 or 3.
 288     n: For future expansion. The length of the transform. Must be `None`.
 289     axis: For future expansion. The axis to compute the DCT along. Must be `-1`.
 290     norm: The normalization to apply. `None` for no normalization or `'ortho'`
 291       for orthonormal normalization.
 292     name: An optional name for the operation.
 293 
 294   Returns:
 295     A `[..., samples]` `float32` `Tensor` containing the IDCT of `input`.
 296 
 297   Raises:
 298     ValueError: If `type` is not `2` or `3`, `n` is not `None, `axis` is not
 299       `-1`, or `norm` is not `None` or `'ortho'`.
 300 
 301   [idct]:
 302   https://en.wikipedia.org/wiki/Discrete_cosine_transform#Inverse_transforms
 303   """
 304   _validate_dct_arguments(type, n, axis, norm)
 305   inverse_type = {2: 3, 3: 2}[type]
 306   return dct(input, type=inverse_type, n=n, axis=axis, norm=norm, name=name)
