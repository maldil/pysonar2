  31 @tf_export("debugging.assert_all_finite", "verify_tensor_all_finite")
  32 @deprecation.deprecated_endpoints("verify_tensor_all_finite")
  33 def verify_tensor_all_finite(t, msg, name=None):
  34   """Assert that the tensor does not contain any NaN's or Inf's.
  35 
  36   Args:
  37     t: Tensor to check.
  38     msg: Message to log on failure.
  39     name: A name for this operation (optional).
  40 
  41   Returns:
  42     Same tensor as `t`.
  43   """
  44   with ops.name_scope(name, "VerifyFinite", [t]) as name:
  45     t = ops.convert_to_tensor(t, name="t")
  46     with ops.colocate_with(t):
  47       verify_input = array_ops.check_numerics(t, message=msg)
  48       out = control_flow_ops.with_dependencies([verify_input], t)
  49   return out

  52 @tf_export("add_check_numerics_ops")
  53 def add_check_numerics_ops():
  54   """Connect a `check_numerics` to every floating point tensor.
  55 
  56   `check_numerics` operations themselves are added for each `half`, `float`,
  57   or `double` tensor in the graph. For all ops in the graph, the
  58   `check_numerics` op for all of its (`half`, `float`, or `double`) inputs
  59   is guaranteed to run before the `check_numerics` op on any of its outputs.
  60 
  61   Note: This API is not compatible with the use of `tf.cond` or
  62   `tf.while_loop`, and will raise a `ValueError` if you attempt to call it
  63   in such a graph.
  64 
  65   Returns:
  66     A `group` op depending on all `check_numerics` ops added.
  67 
  68   Raises:
  69     ValueError: If the graph contains any numeric operations in a control flow
  70       structure.
  71     RuntimeError: If called with eager execution enabled.
  72 
  73   @compatibility(eager)
  74   Not compatible with eager execution. To check for `Inf`s and `NaN`s under
  75   eager execution, call tfe.seterr(inf_or_nan='raise') once before executing
  76   the checked operations.
  77   @enc_compatibility
  78   """
  79   if context.executing_eagerly():
  80     raise RuntimeError(
  81         "add_check_numerics_ops() is not compatible with eager execution. "
  82         "To check for Inf's and NaN's under eager execution, call "
  83         "tfe.seterr(inf_or_nan='raise') once before executing the "
  84         "checked operations.")
  85 
  86   check_op = []
  87   # This code relies on the ordering of ops in get_operations().
  88   # The producer of a tensor always comes before that tensor's consumer in
  89   # this list. This is true because get_operations() returns ops in the order
  90   # added, and an op can only be added after its inputs are added.
  91   for op in ops.get_default_graph().get_operations():
  92     for output in op.outputs:
  93       if output.dtype in [dtypes.float16, dtypes.float32, dtypes.float64]:
  94         if op._get_control_flow_context() is not None:  # pylint: disable=protected-access
  95           raise ValueError("`tf.add_check_numerics_ops() is not compatible "
  96                            "with TensorFlow control flow operations such as "
  97                            "`tf.cond()` or `tf.while_loop()`.")
  98 
  99         message = op.name + ":" + str(output.value_index)
 100         with ops.control_dependencies(check_op):
 101           check_op = [array_ops.check_numerics(output, message=message)]
 102   return control_flow_ops.group(*check_op)
