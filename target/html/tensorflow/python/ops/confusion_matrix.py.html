  32 def remove_squeezable_dimensions(
  33     labels, predictions, expected_rank_diff=0, name=None):
  34   """Squeeze last dim if ranks differ from expected by exactly 1.
  35 
  36   In the common case where we expect shapes to match, `expected_rank_diff`
  37   defaults to 0, and we squeeze the last dimension of the larger rank if they
  38   differ by 1.
  39 
  40   But, for example, if `labels` contains class IDs and `predictions` contains 1
  41   probability per class, we expect `predictions` to have 1 more dimension than
  42   `labels`, so `expected_rank_diff` would be 1. In this case, we'd squeeze
  43   `labels` if `rank(predictions) - rank(labels) == 0`, and
  44   `predictions` if `rank(predictions) - rank(labels) == 2`.
  45 
  46   This will use static shape if available. Otherwise, it will add graph
  47   operations, which could result in a performance hit.
  48 
  49   Args:
  50     labels: Label values, a `Tensor` whose dimensions match `predictions`.
  51     predictions: Predicted values, a `Tensor` of arbitrary dimensions.
  52     expected_rank_diff: Expected result of `rank(predictions) - rank(labels)`.
  53     name: Name of the op.
  54 
  55   Returns:
  56     Tuple of `labels` and `predictions`, possibly with last dim squeezed.
  57   """
  58   with ops.name_scope(name, 'remove_squeezable_dimensions',
  59                       [labels, predictions]):
  60     predictions = ops.convert_to_tensor(predictions)
  61     labels = ops.convert_to_tensor(labels)
  62     predictions_shape = predictions.get_shape()
  63     predictions_rank = predictions_shape.ndims
  64     labels_shape = labels.get_shape()
  65     labels_rank = labels_shape.ndims
  66     if (labels_rank is not None) and (predictions_rank is not None):
  67       # Use static rank.
  68       rank_diff = predictions_rank - labels_rank
  69       if rank_diff == expected_rank_diff + 1:
  70         predictions = array_ops.squeeze(predictions, [-1])
  71       elif rank_diff == expected_rank_diff - 1:
  72         labels = array_ops.squeeze(labels, [-1])
  73       return labels, predictions
  74 
  75     # Use dynamic rank.
  76     rank_diff = array_ops.rank(predictions) - array_ops.rank(labels)
  77     if (predictions_rank is None) or (
  78         predictions_shape.dims[-1].is_compatible_with(1)):
  79       predictions = control_flow_ops.cond(
  80           math_ops.equal(expected_rank_diff + 1, rank_diff),
  81           lambda: array_ops.squeeze(predictions, [-1]),
  82           lambda: predictions)
  83     if (labels_rank is None) or (
  84         labels_shape.dims[-1].is_compatible_with(1)):
  85       labels = control_flow_ops.cond(
  86           math_ops.equal(expected_rank_diff - 1, rank_diff),
  87           lambda: array_ops.squeeze(labels, [-1]),
  88           lambda: labels)
  89     return labels, predictions
  90

  92 @tf_export('confusion_matrix')
  93 def confusion_matrix(labels, predictions, num_classes=None, dtype=dtypes.int32,
  94                      name=None, weights=None):
  95   """Computes the confusion matrix from predictions and labels.
  96 
  97   The matrix columns represent the prediction labels and the rows represent the
  98   real labels. The confusion matrix is always a 2-D array of shape `[n, n]`,
  99   where `n` is the number of valid labels for a given classification task. Both
 100   prediction and labels must be 1-D arrays of the same shape in order for this
 101   function to work.
 102 
 103   If `num_classes` is `None`, then `num_classes` will be set to one plus the
 104   maximum value in either predictions or labels. Class labels are expected to
 105   start at 0. For example, if `num_classes` is 3, then the possible labels
 106   would be `[0, 1, 2]`.
 107 
 108   If `weights` is not `None`, then each prediction contributes its
 109   corresponding weight to the total value of the confusion matrix cell.
 110 
 111   For example:
 112 
 113   ```python
 114     tf.confusion_matrix([1, 2, 4], [2, 2, 4]) ==>
 115         [[0 0 0 0 0]
 116          [0 0 1 0 0]
 117          [0 0 1 0 0]
 118          [0 0 0 0 0]
 119          [0 0 0 0 1]]
 120   ```
 121 
 122   Note that the possible labels are assumed to be `[0, 1, 2, 3, 4]`,
 123   resulting in a 5x5 confusion matrix.
 124 
 125   Args:
 126     labels: 1-D `Tensor` of real labels for the classification task.
 127     predictions: 1-D `Tensor` of predictions for a given classification.
 128     num_classes: The possible number of labels the classification task can
 129                  have. If this value is not provided, it will be calculated
 130                  using both predictions and labels array.
 131     dtype: Data type of the confusion matrix.
 132     name: Scope name.
 133     weights: An optional `Tensor` whose shape matches `predictions`.
 134 
 135   Returns:
 136     A `Tensor` of type `dtype` with shape `[n, n]` representing the confusion
 137     matrix, where `n` is the number of possible labels in the classification
 138     task.
 139 
 140   Raises:
 141     ValueError: If both predictions and labels are not 1-D vectors and have
 142       mismatched shapes, or if `weights` is not `None` and its shape doesn't
 143       match `predictions`.
 144   """
 145   with ops.name_scope(name, 'confusion_matrix',
 146                       (predictions, labels, num_classes, weights)) as name:
 147     labels, predictions = remove_squeezable_dimensions(
 148         ops.convert_to_tensor(labels, name='labels'),
 149         ops.convert_to_tensor(
 150             predictions, name='predictions'))
 151     predictions = math_ops.cast(predictions, dtypes.int64)
 152     labels = math_ops.cast(labels, dtypes.int64)
 153 
 154     # Sanity checks - underflow or overflow can cause memory corruption.
 155     labels = control_flow_ops.with_dependencies(
 156         [check_ops.assert_non_negative(
 157             labels, message='`labels` contains negative values')],
 158         labels)
 159     predictions = control_flow_ops.with_dependencies(
 160         [check_ops.assert_non_negative(
 161             predictions, message='`predictions` contains negative values')],
 162         predictions)
 163 
 164     if num_classes is None:
 165       num_classes = math_ops.maximum(math_ops.reduce_max(predictions),
 166                                      math_ops.reduce_max(labels)) + 1
 167     else:
 168       num_classes_int64 = math_ops.cast(num_classes, dtypes.int64)
 169       labels = control_flow_ops.with_dependencies(
 170           [check_ops.assert_less(
 171               labels, num_classes_int64, message='`labels` out of bound')],
 172           labels)
 173       predictions = control_flow_ops.with_dependencies(
 174           [check_ops.assert_less(
 175               predictions, num_classes_int64,
 176               message='`predictions` out of bound')],
 177           predictions)
 178 
 179     if weights is not None:
 180       predictions.get_shape().assert_is_compatible_with(weights.get_shape())
 181       weights = math_ops.cast(weights, dtype)
 182 
 183     shape = array_ops.stack([num_classes, num_classes])
 184     indices = array_ops.stack([labels, predictions], axis=1)
 185     values = (array_ops.ones_like(predictions, dtype)
 186               if weights is None else weights)
 187     cm_sparse = sparse_tensor.SparseTensor(
 188         indices=indices, values=values, dense_shape=math_ops.to_int64(shape))
 189     zero_matrix = array_ops.zeros(math_ops.to_int32(shape), dtype)
 190 
 191     return sparse_ops.sparse_add(zero_matrix, cm_sparse)
