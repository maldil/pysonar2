  49 def regex_full_match(input, pattern, name=None):
  50   r"""Match elements of `input` with regex `pattern`.
  51 
  52   Args:
  53     input: string `Tensor`, the source strings to process.
  54     pattern: string or scalar string `Tensor`, regular expression to use,
  55       see more details at https://github.com/google/re2/wiki/Syntax
  56     name: Name of the op.
  57 
  58   Returns:
  59     bool `Tensor` of the same shape as `input` with match results.
  60   """
  61   # TODO(b/112455102): Remove compat.forward_compatible once past the horizon.
  62   if not compat.forward_compatible(2018, 11, 10):
  63     return gen_string_ops.regex_full_match(
  64         input=input, pattern=pattern, name=name)
  65   if isinstance(pattern, util_compat.bytes_or_text_types):
  66     # When `pattern` is static through the life of the op we can
  67     # use a version which performs the expensive regex compilation once at
  68     # creation time.
  69     return gen_string_ops.static_regex_full_match(
  70         input=input, pattern=pattern, name=name)
  71   return gen_string_ops.regex_full_match(
  72       input=input, pattern=pattern, name=name)
  73 
  74 regex_full_match.__doc__ = gen_string_ops.regex_full_match.__doc__
  75 
  76 # Expose regex_full_match in strings namespace
  77 tf_export("strings.regex_full_match")(regex_full_match)

  80 def regex_replace(source, pattern, rewrite, replace_global=True):
  81   r"""Replace elements of `source` matching regex `pattern` with `rewrite`.
  82 
  83   Args:
  84     source: string `Tensor`, the source strings to process.
  85     pattern: string or scalar string `Tensor`, regular expression to use,
  86       see more details at https://github.com/google/re2/wiki/Syntax
  87     rewrite: string or scalar string `Tensor`, value to use in match
  88       replacement, supports backslash-escaped digits (\1 to \9) can be to insert
  89       text matching corresponding parenthesized group.
  90     replace_global: `bool`, if `True` replace all non-overlapping matches,
  91       else replace only the first match.
  92 
  93   Returns:
  94     string `Tensor` of the same shape as `source` with specified replacements.
  95   """
  96   if (isinstance(pattern, util_compat.bytes_or_text_types) and
  97       isinstance(rewrite, util_compat.bytes_or_text_types)):
  98     # When `pattern` and `rewrite` are static through the life of the op we can
  99     # use a version which performs the expensive regex compilation once at
 100     # creation time.
 101     return gen_string_ops.static_regex_replace(
 102         input=source, pattern=pattern,
 103         rewrite=rewrite, replace_global=replace_global)
 104   return gen_string_ops.regex_replace(
 105       input=source, pattern=pattern,
 106       rewrite=rewrite, replace_global=replace_global)

 109 @tf_export("strings.format")
 110 def string_format(template, inputs, placeholder="{}", summarize=3, name=None):
 111   r"""Formats a string template using a list of tensors.
 112 
 113   Formats a string template using a list of tensors, abbreviating tensors by
 114   only printing the first and last `summarize` elements of each dimension
 115   (recursively). If formatting only one tensor into a template, the tensor does
 116   not have to be wrapped in a list.
 117 
 118   Example:
 119     Formatting a single-tensor template:
 120     ```python
 121     sess = tf.Session()
 122     with sess.as_default():
 123         tensor = tf.range(10)
 124         formatted = tf.strings.format("tensor: {}, suffix", tensor)
 125         out = sess.run(formatted)
 126         expected = "tensor: [0 1 2 ... 7 8 9], suffix"
 127 
 128         assert(out.decode() == expected)
 129     ```
 130 
 131     Formatting a multi-tensor template:
 132     ```python
 133     sess = tf.Session()
 134     with sess.as_default():
 135         tensor_one = tf.reshape(tf.range(100), [10, 10])
 136         tensor_two = tf.range(10)
 137         formatted = tf.strings.format("first: {}, second: {}, suffix",
 138           (tensor_one, tensor_two))
 139 
 140         out = sess.run(formatted)
 141         expected = ("first: [[0 1 2 ... 7 8 9]\n"
 142               " [10 11 12 ... 17 18 19]\n"
 143               " [20 21 22 ... 27 28 29]\n"
 144               " ...\n"
 145               " [70 71 72 ... 77 78 79]\n"
 146               " [80 81 82 ... 87 88 89]\n"
 147               " [90 91 92 ... 97 98 99]], second: [0 1 2 ... 7 8 9], suffix")
 148 
 149         assert(out.decode() == expected)
 150     ```
 151 
 152   Args:
 153     template: A string template to format tensor values into.
 154     inputs: A list of `Tensor` objects, or a single Tensor.
 155       The list of tensors to format into the template string. If a solitary
 156       tensor is passed in, the input tensor will automatically be wrapped as a
 157       list.
 158     placeholder: An optional `string`. Defaults to `{}`.
 159       At each placeholder occurring in the template, a subsequent tensor
 160       will be inserted.
 161     summarize: An optional `int`. Defaults to `3`.
 162       When formatting the tensors, show the first and last `summarize`
 163       entries of each tensor dimension (recursively). If set to -1, all
 164       elements of the tensor will be shown.
 165     name: A name for the operation (optional).
 166 
 167   Returns:
 168     A scalar `Tensor` of type `string`.
 169 
 170   Raises:
 171     ValueError: if the number of placeholders does not match the number of
 172       inputs.
 173   """
 174   # If there is only one tensor to format, we will automatically wrap it in a
 175   # list to simplify the user experience
 176   if tensor_util.is_tensor(inputs):
 177     inputs = [inputs]
 178   if template.count(placeholder) != len(inputs):
 179     raise ValueError("%s placeholder(s) in template does not match %s tensor(s)"
 180                      " provided as input" % (template.count(placeholder),
 181                                              len(inputs)))
 182 
 183   return gen_string_ops.string_format(inputs,
 184                                       template=template,
 185                                       placeholder=placeholder,
 186                                       summarize=summarize,
 187                                       name=name)

 190 @tf_export("string_split")
 191 def string_split(source, delimiter=" ", skip_empty=True):  # pylint: disable=invalid-name
 192   """Split elements of `source` based on `delimiter` into a `SparseTensor`.
 193 
 194   Let N be the size of source (typically N will be the batch size). Split each
 195   element of `source` based on `delimiter` and return a `SparseTensor`
 196   containing the split tokens. Empty tokens are ignored.
 197 
 198   If `delimiter` is an empty string, each element of the `source` is split
 199   into individual strings, each containing one byte. (This includes splitting
 200   multibyte sequences of UTF-8.) If delimiter contains multiple bytes, it is
 201   treated as a set of delimiters with each considered a potential split point.
 202 
 203   For example:
 204   N = 2, source[0] is 'hello world' and source[1] is 'a b c', then the output
 205   will be
 206 
 207   st.indices = [0, 0;
 208                 0, 1;
 209                 1, 0;
 210                 1, 1;
 211                 1, 2]
 212   st.shape = [2, 3]
 213   st.values = ['hello', 'world', 'a', 'b', 'c']
 214 
 215   Args:
 216     source: `1-D` string `Tensor`, the strings to split.
 217     delimiter: `0-D` string `Tensor`, the delimiter character, the string should
 218       be length 0 or 1.
 219     skip_empty: A `bool`. If `True`, skip the empty strings from the result.
 220 
 221   Raises:
 222     ValueError: If delimiter is not a string.
 223 
 224   Returns:
 225     A `SparseTensor` of rank `2`, the strings split according to the delimiter.
 226     The first column of the indices corresponds to the row in `source` and the
 227     second column corresponds to the index of the split component in this row.
 228   """
 229   delimiter = ops.convert_to_tensor(delimiter, dtype=dtypes.string)
 230   source = ops.convert_to_tensor(source, dtype=dtypes.string)
 231 
 232   indices, values, shape = gen_string_ops.string_split(
 233       source, delimiter=delimiter, skip_empty=skip_empty)
 234   indices.set_shape([None, 2])
 235   values.set_shape([None])
 236   shape.set_shape([2])
 237   return sparse_tensor.SparseTensor(indices, values, shape)

 240 @tf_export("strings.split")
 241 def string_split_v2(source, sep=None, maxsplit=-1):
 242   """Split elements of `source` based on `sep` into a `SparseTensor`.
 243 
 244   Let N be the size of source (typically N will be the batch size). Split each
 245   element of `source` based on `sep` and return a `SparseTensor`
 246   containing the split tokens. Empty tokens are ignored.
 247 
 248   For example, N = 2, source[0] is 'hello world' and source[1] is 'a b c',
 249   then the output will be
 250 
 251   st.indices = [0, 0;
 252                 0, 1;
 253                 1, 0;
 254                 1, 1;
 255                 1, 2]
 256   st.shape = [2, 3]
 257   st.values = ['hello', 'world', 'a', 'b', 'c']
 258 
 259   If `sep` is given, consecutive delimiters are not grouped together and are
 260   deemed to delimit empty strings. For example, source of `"1<>2<><>3"` and
 261   sep of `"<>"` returns `["1", "2", "", "3"]`. If `sep` is None or an empty
 262   string, consecutive whitespace are regarded as a single separator, and the
 263   result will contain no empty strings at the startor end if the string has
 264   leading or trailing whitespace.
 265 
 266   Note that the above mentioned behavior matches python's str.split.
 267 
 268   Args:
 269     source: `1-D` string `Tensor`, the strings to split.
 270     sep: `0-D` string `Tensor`, the delimiter character.
 271     maxsplit: An `int`. If `maxsplit > 0`, limit of the split of the result.
 272 
 273   Raises:
 274     ValueError: If sep is not a string.
 275 
 276   Returns:
 277     A `SparseTensor` of rank `2`, the strings split according to the delimiter.
 278     The first column of the indices corresponds to the row in `source` and the
 279     second column corresponds to the index of the split component in this row.
 280   """
 281   if sep is None:
 282     sep = ""
 283   sep = ops.convert_to_tensor(sep, dtype=dtypes.string)
 284   source = ops.convert_to_tensor(source, dtype=dtypes.string)
 285 
 286   indices, values, shape = gen_string_ops.string_split_v2(
 287       source, sep=sep, maxsplit=maxsplit)
 288   indices.set_shape([None, 2])
 289   values.set_shape([None])
 290   shape.set_shape([2])
 291   return sparse_tensor.SparseTensor(indices, values, shape)

 294 def _reduce_join_reduction_dims(x, axis, reduction_indices):
 295   """Returns range(rank(x) - 1, 0, -1) if reduction_indices is None."""
 296   # TODO(aselle): Remove this after deprecation
 297   if reduction_indices is not None:
 298     if axis is not None:
 299       raise ValueError("Can't specify both 'axis' and 'reduction_indices'.")
 300     axis = reduction_indices
 301   if axis is not None:
 302     return axis
 303   else:
 304     # Fast path: avoid creating Rank and Range ops if ndims is known.
 305     if x.get_shape().ndims is not None:
 306       return constant_op.constant(
 307           np.arange(x.get_shape().ndims - 1, -1, -1), dtype=dtypes.int32)
 308 
 309     # Otherwise, we rely on Range and Rank to do the right thing at run-time.
 310     return math_ops.range(array_ops.rank(x) - 1, -1, -1)

 313 @tf_export("strings.reduce_join", "reduce_join")
 314 @deprecation.deprecated_endpoints("reduce_join")
 315 def reduce_join(inputs, axis=None,  # pylint: disable=missing-docstring
 316                 keep_dims=False,
 317                 separator="",
 318                 name=None,
 319                 reduction_indices=None):
 320   inputs_t = ops.convert_to_tensor(inputs)
 321   reduction_indices = _reduce_join_reduction_dims(
 322       inputs_t, axis, reduction_indices)
 323   return gen_string_ops.reduce_join(
 324       inputs=inputs_t,
 325       reduction_indices=reduction_indices,
 326       keep_dims=keep_dims,
 327       separator=separator,
 328       name=name)
