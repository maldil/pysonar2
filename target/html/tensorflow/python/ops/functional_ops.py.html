  50 @tf_export("foldl")
  51 def foldl(fn, elems, initializer=None, parallel_iterations=10, back_prop=True,
  52           swap_memory=False, name=None):
  53   """foldl on the list of tensors unpacked from `elems` on dimension 0.
  54 
  55   This foldl operator repeatedly applies the callable `fn` to a sequence
  56   of elements from first to last. The elements are made of the tensors
  57   unpacked from `elems` on dimension 0. The callable fn takes two tensors as
  58   arguments. The first argument is the accumulated value computed from the
  59   preceding invocation of fn. If `initializer` is None, `elems` must contain
  60   at least one element, and its first element is used as the initializer.
  61 
  62   Suppose that `elems` is unpacked into `values`, a list of tensors. The shape
  63   of the result tensor is fn(initializer, values[0]).shape`.
  64 
  65   This method also allows multi-arity `elems` and output of `fn`.  If `elems`
  66   is a (possibly nested) list or tuple of tensors, then each of these tensors
  67   must have a matching first (unpack) dimension.  The signature of `fn` may
  68   match the structure of `elems`.  That is, if `elems` is
  69   `(t1, [t2, t3, [t4, t5]])`, then an appropriate signature for `fn` is:
  70   `fn = lambda (t1, [t2, t3, [t4, t5]]):`.
  71 
  72   Args:
  73     fn: The callable to be performed.
  74     elems: A tensor or (possibly nested) sequence of tensors, each of which
  75       will be unpacked along their first dimension.  The nested sequence
  76       of the resulting slices will be the first argument to `fn`.
  77     initializer: (optional) A tensor or (possibly nested) sequence of tensors,
  78       as the initial value for the accumulator.
  79     parallel_iterations: (optional) The number of iterations allowed to run
  80       in parallel.
  81     back_prop: (optional) True enables support for back propagation.
  82     swap_memory: (optional) True enables GPU-CPU memory swapping.
  83     name: (optional) Name prefix for the returned tensors.
  84 
  85   Returns:
  86     A tensor or (possibly nested) sequence of tensors, resulting from applying
  87     `fn` consecutively to the list of tensors unpacked from `elems`, from first
  88     to last.
  89 
  90   Raises:
  91     TypeError: if `fn` is not callable.
  92 
  93   Example:
  94     ```python
  95     elems = tf.constant([1, 2, 3, 4, 5, 6])
  96     sum = foldl(lambda a, x: a + x, elems)
  97     # sum == 21
  98     ```
  99   """
 100   if not callable(fn):
 101     raise TypeError("fn must be callable.")
 102 
 103   def create_ta(elem):
 104     return tensor_array_ops.TensorArray(
 105         dtype=elem.dtype, size=n, dynamic_size=False,
 106         infer_shape=True).unstack(elem)
 107 
 108   in_graph_mode = not context.executing_eagerly()
 109   with ops.name_scope(name, "foldl", [elems]):
 110     # TODO(akshayka): Remove the in_graph_mode check once caching devices are
 111     # supported in Eager
 112     if in_graph_mode:
 113       # Any get_variable calls in fn will cache the first call locally
 114       # and not issue repeated network I/O requests for each iteration.
 115       varscope = vs.get_variable_scope()
 116       varscope_caching_device_was_none = False
 117       if varscope.caching_device is None:
 118         # TODO(ebrevdo): Change to using colocate_with here and in other
 119         # methods.
 120         varscope.set_caching_device(lambda op: op.device)
 121         varscope_caching_device_was_none = True
 122 
 123     # Convert elems to tensor array. n may be known statically.
 124     elems_flat = [
 125         ops.convert_to_tensor(elem, name="elem") for elem in nest.flatten(elems)
 126     ]
 127     n = elems_flat[0].shape[0].value or array_ops.shape(elems_flat[0])[0]
 128 
 129     elems_ta = nest.map_structure(create_ta, elems)
 130 
 131     if initializer is None:
 132       a = nest.map_structure(lambda elem: elem.read(0), elems_ta)
 133       i = constant_op.constant(1)
 134     else:
 135       a = initializer
 136       i = constant_op.constant(0)
 137 
 138     def compute(i, a):
 139       elem_i = nest.map_structure(lambda elem: elem.read(i), elems_ta)
 140       a = fn(a, elem_i)
 141       return [i + 1, a]
 142 
 143     _, r_a = control_flow_ops.while_loop(
 144         lambda i, a: i < n, compute, [i, a],
 145         parallel_iterations=parallel_iterations,
 146         back_prop=back_prop,
 147         swap_memory=swap_memory)
 148 
 149     # TODO(akshayka): Remove the in_graph_mode check once caching devices are
 150     # supported in Eager
 151     if in_graph_mode and varscope_caching_device_was_none:
 152       varscope.set_caching_device(None)
 153 
 154     return r_a

 157 @tf_export("foldr")
 158 def foldr(fn, elems, initializer=None, parallel_iterations=10, back_prop=True,
 159           swap_memory=False, name=None):
 160   """foldr on the list of tensors unpacked from `elems` on dimension 0.
 161 
 162   This foldr operator repeatedly applies the callable `fn` to a sequence
 163   of elements from last to first. The elements are made of the tensors
 164   unpacked from `elems`. The callable fn takes two tensors as arguments.
 165   The first argument is the accumulated value computed from the preceding
 166   invocation of fn. If `initializer` is None, `elems` must contain at least
 167   one element, and its first element is used as the initializer.
 168 
 169   Suppose that `elems` is unpacked into `values`, a list of tensors. The shape
 170   of the result tensor is `fn(initializer, values[0]).shape`.
 171 
 172   This method also allows multi-arity `elems` and output of `fn`.  If `elems`
 173   is a (possibly nested) list or tuple of tensors, then each of these tensors
 174   must have a matching first (unpack) dimension.  The signature of `fn` may
 175   match the structure of `elems`.  That is, if `elems` is
 176   `(t1, [t2, t3, [t4, t5]])`, then an appropriate signature for `fn` is:
 177   `fn = lambda (t1, [t2, t3, [t4, t5]]):`.
 178 
 179   Args:
 180     fn: The callable to be performed.
 181     elems: A tensor or (possibly nested) sequence of tensors, each of which
 182       will be unpacked along their first dimension.  The nested sequence
 183       of the resulting slices will be the first argument to `fn`.
 184     initializer: (optional) A tensor or (possibly nested) sequence of tensors,
 185       as the initial value for the accumulator.
 186     parallel_iterations: (optional) The number of iterations allowed to run
 187       in parallel.
 188     back_prop: (optional) True enables support for back propagation.
 189     swap_memory: (optional) True enables GPU-CPU memory swapping.
 190     name: (optional) Name prefix for the returned tensors.
 191 
 192   Returns:
 193     A tensor or (possibly nested) sequence of tensors, resulting from applying
 194     `fn` consecutively to the list of tensors unpacked from `elems`, from last
 195     to first.
 196 
 197   Raises:
 198     TypeError: if `fn` is not callable.
 199 
 200   Example:
 201     ```python
 202     elems = [1, 2, 3, 4, 5, 6]
 203     sum = foldr(lambda a, x: a + x, elems)
 204     # sum == 21
 205     ```
 206   """
 207   if not callable(fn):
 208     raise TypeError("fn must be callable.")
 209 
 210   def create_ta(elem):
 211     return tensor_array_ops.TensorArray(
 212         dtype=elem.dtype, size=n, dynamic_size=False,
 213         infer_shape=True).unstack(elem)
 214 
 215   in_graph_mode = not context.executing_eagerly()
 216   with ops.name_scope(name, "foldr", [elems]):
 217     # TODO(akshayka): Remove the in_graph_mode check once caching devices are
 218     # supported in Eager
 219     if in_graph_mode:
 220       # Any get_variable calls in fn will cache the first call locally and not
 221       # issue repeated network I/O requests for each iteration.
 222       varscope = vs.get_variable_scope()
 223       varscope_caching_device_was_none = False
 224       if varscope.caching_device is None:
 225         # TODO(ebrevdo): Change to using colocate_with here and in other
 226         # methods.
 227         varscope.set_caching_device(lambda op: op.device)
 228         varscope_caching_device_was_none = True
 229 
 230     # Convert elems to tensor array. n may be known statically.
 231     elems_flat = [
 232         ops.convert_to_tensor(elem, name="elem") for elem in nest.flatten(elems)
 233     ]
 234     n = elems_flat[0].shape[0].value or array_ops.shape(elems_flat[0])[0]
 235 
 236     elems_ta = nest.map_structure(create_ta, elems)
 237 
 238     if initializer is None:
 239       i = n - 1
 240       a = nest.map_structure(lambda elem: elem.read(i), elems_ta)
 241     else:
 242       i = n
 243       a = initializer
 244 
 245     def compute(i, a):
 246       i -= 1
 247       elem = nest.map_structure(lambda elem: elem.read(i), elems_ta)
 248       a_out = fn(a, elem)
 249       return [i, a_out]
 250 
 251     _, r_a = control_flow_ops.while_loop(
 252         lambda i, a: i > 0,
 253         compute, [i, a],
 254         parallel_iterations=parallel_iterations,
 255         back_prop=back_prop,
 256         swap_memory=swap_memory)
 257 
 258     # TODO(akshayka): Remove the in_graph_mode check once caching devices are
 259     # supported in Eager
 260     if in_graph_mode and varscope_caching_device_was_none:
 261       varscope.set_caching_device(None)
 262 
 263     return r_a

 266 @tf_export("map_fn")
 267 def map_fn(fn, elems, dtype=None, parallel_iterations=None, back_prop=True,
 268            swap_memory=False, infer_shape=True, name=None):
 269   """map on the list of tensors unpacked from `elems` on dimension 0.
 270 
 271   The simplest version of `map_fn` repeatedly applies the callable `fn` to a
 272   sequence of elements from first to last. The elements are made of the
 273   tensors unpacked from `elems`. `dtype` is the data type of the return
 274   value of `fn`. Users must provide `dtype` if it is different from
 275   the data type of `elems`.
 276 
 277   Suppose that `elems` is unpacked into `values`, a list of tensors. The shape
 278   of the result tensor is `[values.shape[0]] + fn(values[0]).shape`.
 279 
 280   This method also allows multi-arity `elems` and output of `fn`.  If `elems`
 281   is a (possibly nested) list or tuple of tensors, then each of these tensors
 282   must have a matching first (unpack) dimension.  The signature of `fn` may
 283   match the structure of `elems`.  That is, if `elems` is
 284   `(t1, [t2, t3, [t4, t5]])`, then an appropriate signature for `fn` is:
 285   `fn = lambda (t1, [t2, t3, [t4, t5]]):`.
 286 
 287   Furthermore, `fn` may emit a different structure than its input.  For example,
 288   `fn` may look like: `fn = lambda t1: return (t1 + 1, t1 - 1)`.  In this case,
 289   the `dtype` parameter is not optional: `dtype` must be a type or (possibly
 290   nested) tuple of types matching the output of `fn`.
 291 
 292   To apply a functional operation to the nonzero elements of a SparseTensor
 293   one of the following methods is recommended. First, if the function is
 294   expressible as TensorFlow ops, use
 295 
 296   ```python
 297     result = SparseTensor(input.indices, fn(input.values), input.dense_shape)
 298   ```
 299 
 300   If, however, the function is not expressible as a TensorFlow op, then use
 301 
 302   ```python
 303   result = SparseTensor(
 304     input.indices, map_fn(fn, input.values), input.dense_shape)
 305   ```
 306 
 307   instead.
 308 
 309   When executing eagerly, map_fn does not execute in parallel even if
 310   `parallel_iterations` is set to a value > 1. You can still get the
 311   performance benefits of running a function in parallel by using the
 312   `tf.contrib.eager.defun` decorator,
 313 
 314   ```python
 315   # Assume the function being used in map_fn is fn.
 316   # To ensure map_fn calls fn in parallel, use the defun decorator.
 317   @tf.contrib.eager.defun
 318   def func(tensor):
 319     return tf.map_fn(fn, tensor)
 320   ```
 321 
 322   Note that if you use the defun decorator, any non-TensorFlow Python code
 323   that you may have written in your function won't get executed. See
 324   `tf.contrib.eager.defun` for more details. The recommendation would be to
 325   debug without defun but switch to defun to get performance benefits of
 326   running map_fn in parallel.
 327 
 328   Args:
 329     fn: The callable to be performed.  It accepts one argument, which will
 330       have the same (possibly nested) structure as `elems`.  Its output
 331       must have the same structure as `dtype` if one is provided, otherwise
 332       it must have the same structure as `elems`.
 333     elems: A tensor or (possibly nested) sequence of tensors, each of which
 334       will be unpacked along their first dimension.  The nested sequence
 335       of the resulting slices will be applied to `fn`.
 336     dtype: (optional) The output type(s) of `fn`.  If `fn` returns a structure
 337       of Tensors differing from the structure of `elems`, then `dtype` is not
 338       optional and must have the same structure as the output of `fn`.
 339     parallel_iterations: (optional) The number of iterations allowed to run
 340       in parallel. When graph building, the default value is 10. While executing
 341       eagerly, the default value is set to 1.
 342     back_prop: (optional) True enables support for back propagation.
 343     swap_memory: (optional) True enables GPU-CPU memory swapping.
 344     infer_shape: (optional) False disables tests for consistent output shapes.
 345     name: (optional) Name prefix for the returned tensors.
 346 
 347   Returns:
 348     A tensor or (possibly nested) sequence of tensors.  Each tensor packs the
 349     results of applying `fn` to tensors unpacked from `elems` along the first
 350     dimension, from first to last.
 351 
 352   Raises:
 353     TypeError: if `fn` is not callable or the structure of the output of
 354       `fn` and `dtype` do not match, or if elems is a SparseTensor.
 355     ValueError: if the lengths of the output of `fn` and `dtype` do not match.
 356 
 357   Examples:
 358     ```python
 359     elems = np.array([1, 2, 3, 4, 5, 6])
 360     squares = map_fn(lambda x: x * x, elems)
 361     # squares == [1, 4, 9, 16, 25, 36]
 362     ```
 363 
 364     ```python
 365     elems = (np.array([1, 2, 3]), np.array([-1, 1, -1]))
 366     alternate = map_fn(lambda x: x[0] * x[1], elems, dtype=tf.int64)
 367     # alternate == [-1, 2, -3]
 368     ```
 369 
 370     ```python
 371     elems = np.array([1, 2, 3])
 372     alternates = map_fn(lambda x: (x, -x), elems, dtype=(tf.int64, tf.int64))
 373     # alternates[0] == [1, 2, 3]
 374     # alternates[1] == [-1, -2, -3]
 375     ```
 376   """
 377   if not callable(fn):
 378     raise TypeError("fn must be callable.")
 379 
 380   if isinstance(elems, sparse_tensor.SparseTensor):
 381     raise TypeError(
 382         "To perform a map on the values of a sparse tensor use either "
 383         " SparseTensor(input.indices, fn(input.values), input.dense_shape) or "
 384         " SparseTensor(input.indices, map_fn(fn, input.values), "
 385         "input.dense_shape)")
 386 
 387   in_graph_mode = not context.executing_eagerly()
 388   # Set the default number of parallel_iterations depending on graph/eager mode.
 389   if in_graph_mode and not parallel_iterations:
 390     parallel_iterations = 10
 391   elif not in_graph_mode and not parallel_iterations:
 392     parallel_iterations = 1
 393 
 394   if not in_graph_mode and parallel_iterations > 1:
 395     logging.log_first_n(logging.WARN, "Setting parallel_iterations > 1 has no "
 396                         "effect when executing eagerly. Consider calling map_fn"
 397                         " with tf.contrib.eager.defun to execute fn in "
 398                         "parallel.", 1)
 399     parallel_iterations = 1
 400 
 401   input_is_sequence = nest.is_sequence(elems)
 402   input_flatten = lambda x: nest.flatten(x) if input_is_sequence else [x]
 403   def input_pack(x):
 404     return nest.pack_sequence_as(elems, x) if input_is_sequence else x[0]
 405 
 406   if dtype is None:
 407     output_is_sequence = input_is_sequence
 408     output_flatten = input_flatten
 409     output_pack = input_pack
 410   else:
 411     output_is_sequence = nest.is_sequence(dtype)
 412     output_flatten = lambda x: nest.flatten(x) if output_is_sequence else [x]
 413     def output_pack(x):
 414       return (nest.pack_sequence_as(dtype, x)
 415               if output_is_sequence else x[0])
 416 
 417   elems_flat = input_flatten(elems)
 418 
 419   with ops.name_scope(name, "map", elems_flat):
 420     # TODO(akshayka): Remove the in_graph_mode check once caching devices are
 421     # supported in Eager
 422     if in_graph_mode:
 423       # Any get_variable calls in fn will cache the first call locally
 424       # and not issue repeated network I/O requests for each iteration.
 425       varscope = vs.get_variable_scope()
 426       varscope_caching_device_was_none = False
 427       if varscope.caching_device is None:
 428         # TODO(ebrevdo): Change to using colocate_with here and in other
 429         # methods.
 430         varscope.set_caching_device(lambda op: op.device)
 431         varscope_caching_device_was_none = True
 432 
 433     elems_flat = [
 434         ops.convert_to_tensor(elem, name="elem") for elem in elems_flat]
 435 
 436     dtype = dtype or input_pack([elem.dtype for elem in elems_flat])
 437     dtype_flat = output_flatten(dtype)
 438 
 439     # Convert elems to tensor array. n may be known statically.
 440     static_shape = elems_flat[0].shape
 441     if static_shape.ndims is not None and static_shape.ndims < 1:
 442       if len(elems_flat) == 1:
 443         raise ValueError("elems must be a 1+ dimensional Tensor, not a scalar")
 444       else:
 445         raise ValueError(
 446             "elements in elems must be 1+ dimensional Tensors, not scalars"
 447         )
 448     n = static_shape[0].value or array_ops.shape(elems_flat[0])[0]
 449 
 450     # TensorArrays are always flat
 451     elems_ta = [
 452         tensor_array_ops.TensorArray(dtype=elem.dtype, size=n,
 453                                      dynamic_size=False,
 454                                      infer_shape=True)
 455         for elem in elems_flat]
 456     # Unpack elements
 457     elems_ta = [
 458         elem_ta.unstack(elem) for elem_ta, elem in zip(elems_ta, elems_flat)]
 459 
 460     i = constant_op.constant(0)
 461 
 462     accs_ta = [
 463         tensor_array_ops.TensorArray(dtype=dt, size=n,
 464                                      dynamic_size=False,
 465                                      infer_shape=infer_shape)
 466         for dt in dtype_flat]
 467 
 468     def compute(i, tas):
 469       """The loop body of map_fn.
 470 
 471       Args:
 472         i: the loop counter
 473         tas: the flat TensorArray accumulator list
 474 
 475       Returns:
 476         (i + 1, tas): the updated counter + updated TensorArrays
 477 
 478       Raises:
 479         TypeError: if dtype and packed_fn_values structure do not match
 480         ValueType: if dtype and packed_fn_values lengths do not match
 481       """
 482       packed_values = input_pack([elem_ta.read(i) for elem_ta in elems_ta])
 483       packed_fn_values = fn(packed_values)
 484       nest.assert_same_structure(dtype or elems, packed_fn_values)
 485       flat_fn_values = output_flatten(packed_fn_values)
 486       tas = [ta.write(i, value) for (ta, value) in zip(tas, flat_fn_values)]
 487       return (i + 1, tas)
 488 
 489     _, r_a = control_flow_ops.while_loop(
 490         lambda i, _: i < n, compute, (i, accs_ta),
 491         parallel_iterations=parallel_iterations,
 492         back_prop=back_prop,
 493         swap_memory=swap_memory,
 494         maximum_iterations=n)
 495     results_flat = [r.stack() for r in r_a]
 496 
 497     n_static = elems_flat[0].get_shape().with_rank_at_least(1)[0]
 498     for elem in elems_flat[1:]:
 499       n_static.merge_with(elem.get_shape().with_rank_at_least(1)[0])
 500     for r in results_flat:
 501       r.set_shape(tensor_shape.TensorShape(n_static).concatenate(
 502           r.get_shape()[1:]))
 503 
 504     # TODO(akshayka): Remove the in_graph_mode check once caching devices are
 505     # supported in Eager
 506     if in_graph_mode and varscope_caching_device_was_none:
 507       varscope.set_caching_device(None)
 508 
 509     return output_pack(results_flat)
 510

 512 @tf_export("scan")
 513 def scan(fn, elems, initializer=None, parallel_iterations=10, back_prop=True,
 514          swap_memory=False, infer_shape=True, reverse=False, name=None):
 515   """scan on the list of tensors unpacked from `elems` on dimension 0.
 516 
 517   The simplest version of `scan` repeatedly applies the callable `fn` to a
 518   sequence of elements from first to last. The elements are made of the tensors
 519   unpacked from `elems` on dimension 0. The callable fn takes two tensors as
 520   arguments. The first argument is the accumulated value computed from the
 521   preceding invocation of fn. If `initializer` is None, `elems` must contain
 522   at least one element, and its first element is used as the initializer.
 523 
 524   Suppose that `elems` is unpacked into `values`, a list of tensors. The shape
 525   of the result tensor is `[len(values)] + fn(initializer, values[0]).shape`.
 526   If reverse=True, it's fn(initializer, values[-1]).shape.
 527 
 528   This method also allows multi-arity `elems` and accumulator.  If `elems`
 529   is a (possibly nested) list or tuple of tensors, then each of these tensors
 530   must have a matching first (unpack) dimension.  The second argument of
 531   `fn` must match the structure of `elems`.
 532 
 533   If no `initializer` is provided, the output structure and dtypes of `fn`
 534   are assumed to be the same as its input; and in this case, the first
 535   argument of `fn` must match the structure of `elems`.
 536 
 537   If an `initializer` is provided, then the output of `fn` must have the same
 538   structure as `initializer`; and the first argument of `fn` must match
 539   this structure.
 540 
 541   For example, if `elems` is `(t1, [t2, t3])` and `initializer` is
 542   `[i1, i2]` then an appropriate signature for `fn` in `python2` is:
 543   `fn = lambda (acc_p1, acc_p2), (t1, [t2, t3]):` and `fn` must return a list,
 544   `[acc_n1, acc_n2]`.  An alternative correct signature for `fn`, and the
 545    one that works in `python3`, is:
 546   `fn = lambda a, t:`, where `a` and `t` correspond to the input tuples.
 547 
 548   Args:
 549     fn: The callable to be performed.  It accepts two arguments.  The first
 550       will have the same structure as `initializer` if one is provided,
 551       otherwise it will have the same structure as `elems`.  The second
 552       will have the same (possibly nested) structure as `elems`.  Its output
 553       must have the same structure as `initializer` if one is provided,
 554       otherwise it must have the same structure as `elems`.
 555     elems: A tensor or (possibly nested) sequence of tensors, each of which
 556       will be unpacked along their first dimension.  The nested sequence
 557       of the resulting slices will be the first argument to `fn`.
 558     initializer: (optional) A tensor or (possibly nested) sequence of tensors,
 559       initial value for the accumulator, and the expected output type of `fn`.
 560     parallel_iterations: (optional) The number of iterations allowed to run
 561       in parallel.
 562     back_prop: (optional) True enables support for back propagation.
 563     swap_memory: (optional) True enables GPU-CPU memory swapping.
 564     infer_shape: (optional) False disables tests for consistent output shapes.
 565     reverse: (optional) True scans the tensor last to first (instead of first
 566       to last).
 567     name: (optional) Name prefix for the returned tensors.
 568 
 569   Returns:
 570     A tensor or (possibly nested) sequence of tensors.  Each tensor packs the
 571     results of applying `fn` to tensors unpacked from `elems` along the first
 572     dimension, and the previous accumulator value(s), from first to last (or
 573     last to first, if `reverse=True`).
 574 
 575   Raises:
 576     TypeError: if `fn` is not callable or the structure of the output of
 577       `fn` and `initializer` do not match.
 578     ValueError: if the lengths of the output of `fn` and `initializer`
 579       do not match.
 580 
 581   Examples:
 582     ```python
 583     elems = np.array([1, 2, 3, 4, 5, 6])
 584     sum = scan(lambda a, x: a + x, elems)
 585     # sum == [1, 3, 6, 10, 15, 21]
 586     sum = scan(lambda a, x: a + x, elems, reverse=True)
 587     # sum == [22, 21, 18, 15, 11, 6]
 588     ```
 589 
 590     ```python
 591     elems = np.array([1, 2, 3, 4, 5, 6])
 592     initializer = np.array(0)
 593     sum_one = scan(
 594         lambda a, x: x[0] - x[1] + a, (elems + 1, elems), initializer)
 595     # sum_one == [1, 2, 3, 4, 5, 6]
 596     ```
 597 
 598     ```python
 599     elems = np.array([1, 0, 0, 0, 0, 0])
 600     initializer = (np.array(0), np.array(1))
 601     fibonaccis = scan(lambda a, _: (a[1], a[0] + a[1]), elems, initializer)
 602     # fibonaccis == ([1, 1, 2, 3, 5, 8], [1, 2, 3, 5, 8, 13])
 603     ```
 604   """
 605   if not callable(fn):
 606     raise TypeError("fn must be callable.")
 607 
 608   input_is_sequence = nest.is_sequence(elems)
 609   input_flatten = lambda x: nest.flatten(x) if input_is_sequence else [x]
 610   def input_pack(x):
 611     return nest.pack_sequence_as(elems, x) if input_is_sequence else x[0]
 612 
 613   if initializer is None:
 614     output_is_sequence = input_is_sequence
 615     output_flatten = input_flatten
 616     output_pack = input_pack
 617   else:
 618     output_is_sequence = nest.is_sequence(initializer)
 619     output_flatten = lambda x: nest.flatten(x) if output_is_sequence else [x]
 620     def output_pack(x):
 621       return (nest.pack_sequence_as(initializer, x)
 622               if output_is_sequence else x[0])
 623 
 624   elems_flat = input_flatten(elems)
 625 
 626   in_graph_mode = not context.executing_eagerly()
 627   with ops.name_scope(name, "scan", elems_flat):
 628     # TODO(akshayka): Remove the in_graph_mode check once caching devices are
 629     # supported in Eager
 630     if in_graph_mode:
 631       # Any get_variable calls in fn will cache the first call locally
 632       # and not issue repeated network I/O requests for each iteration.
 633       varscope = vs.get_variable_scope()
 634       varscope_caching_device_was_none = False
 635       if varscope.caching_device is None:
 636         # TODO(ebrevdo): Change to using colocate_with here and in other
 637         # methods.
 638         varscope.set_caching_device(lambda op: op.device)
 639         varscope_caching_device_was_none = True
 640 
 641     # Convert elems to tensor array.
 642     elems_flat = [
 643         ops.convert_to_tensor(elem, name="elem") for elem in elems_flat]
 644 
 645     # Convert elems to tensor array. n may be known statically.
 646     n = elems_flat[0].shape[0].value or array_ops.shape(elems_flat[0])[0]
 647 
 648     # TensorArrays are always flat
 649     elems_ta = [
 650         tensor_array_ops.TensorArray(dtype=elem.dtype, size=n,
 651                                      dynamic_size=False,
 652                                      infer_shape=True)
 653         for elem in elems_flat]
 654     # Unpack elements
 655     elems_ta = [
 656         elem_ta.unstack(elem) for elem_ta, elem in zip(elems_ta, elems_flat)]
 657 
 658     if initializer is None:
 659       a_flat = [elem.read(n - 1 if reverse else 0) for elem in elems_ta]
 660       i = constant_op.constant(1)
 661     else:
 662       initializer_flat = output_flatten(initializer)
 663       a_flat = [ops.convert_to_tensor(init) for init in initializer_flat]
 664       i = constant_op.constant(0)
 665 
 666     # Create a tensor array to store the intermediate values.
 667     accs_ta = [
 668         tensor_array_ops.TensorArray(
 669             dtype=init.dtype, size=n,
 670             element_shape=init.shape if infer_shape else None,
 671             dynamic_size=False,
 672             infer_shape=infer_shape)
 673         for init in a_flat]
 674 
 675     if initializer is None:
 676       accs_ta = [acc_ta.write(n - 1 if reverse else 0, a)
 677                  for (acc_ta, a) in zip(accs_ta, a_flat)]
 678 
 679     def compute(i, a_flat, tas):
 680       """The loop body of scan.
 681 
 682       Args:
 683         i: the loop counter.
 684         a_flat: the accumulator value(s), flattened.
 685         tas: the output accumulator TensorArray(s), flattened.
 686 
 687       Returns:
 688         [i + 1, a_flat, tas]: the updated counter + new accumulator values +
 689           updated TensorArrays
 690 
 691       Raises:
 692         TypeError: if initializer and fn() output structure do not match
 693         ValueType: if initializer and fn() output lengths do not match
 694       """
 695       packed_elems = input_pack([elem_ta.read(i) for elem_ta in elems_ta])
 696       packed_a = output_pack(a_flat)
 697       a_out = fn(packed_a, packed_elems)
 698       nest.assert_same_structure(
 699           elems if initializer is None else initializer, a_out)
 700       flat_a_out = output_flatten(a_out)
 701       tas = [ta.write(i, value) for (ta, value) in zip(tas, flat_a_out)]
 702       if reverse:
 703         next_i = i - 1
 704       else:
 705         next_i = i + 1
 706       return (next_i, flat_a_out, tas)
 707 
 708     if reverse:
 709       initial_i = n - 1 - i
 710       condition = lambda i, _1, _2: i >= 0
 711     else:
 712       initial_i = i
 713       condition = lambda i, _1, _2: i < n
 714     _, _, r_a = control_flow_ops.while_loop(
 715         condition, compute, (initial_i, a_flat, accs_ta),
 716         parallel_iterations=parallel_iterations,
 717         back_prop=back_prop, swap_memory=swap_memory,
 718         maximum_iterations=n)
 719 
 720     results_flat = [r.stack() for r in r_a]
 721 
 722     n_static = elems_flat[0].get_shape().with_rank_at_least(1)[0]
 723     for elem in elems_flat[1:]:
 724       n_static.merge_with(elem.get_shape().with_rank_at_least(1)[0])
 725     for r in results_flat:
 726       r.set_shape(tensor_shape.TensorShape(n_static).concatenate(
 727           r.get_shape()[1:]))
 728 
 729     # TODO(akshayka): Remove the in_graph_mode check once caching devices are
 730     # supported in Eager
 731     if in_graph_mode and varscope_caching_device_was_none:
 732       varscope.set_caching_device(None)
 733 
 734     return output_pack(results_flat)

 836 # b/36459430
 837 #
 838 # Ideally, we do not need this rewrite For loop into a While loop.
 839 # However, today, if a While runs on GPU and the condition returns a
 840 # boolean, the While kernel crashes. Even if we fix the crash, the
 841 # bool needs to be copied between GPU and CPU. So, a for loop is much
 842 # preferred when running on GPU.
 843 #
 844 # On the other hand, For op has no directly XLA kernel. So, when we run
 845 # a for loop, we need to rewrite it using a While op.
 846 #
 847 # It should be possible and probably better to write a XLA C++ kernel
 848 # implementing the logic in _ForUsingWhile.
 849 def _ForUsingWhile(start,
 850                    limit,
 851                    delta,
 852                    inputs,
 853                    forbody,
 854                    name=None,
 855                    hostmem=None):
 856   """Helper to implement a For loop using a While."""
 857   # To support negative delta (e.g., range(100, 0, -3)), we iterate
 858   # over the range(n) and use iter * delta + start as the real
 859   # iteration index. (e.g., for i in range(34): iter = i * (-3) +
 860   # 100).
 861   d = math_ops.abs(delta)
 862   # XLA on TPUs doesn't support integer division
 863   n = math_ops.cast(
 864       math_ops.cast((math_ops.abs(limit - start) + d - 1), dtypes.float32) /
 865       math_ops.cast(d, dtypes.float32), dtypes.int32)
 866 
 867   # Carried loop variables ("extra_args") are implicitly added to the input list
 868   # of the WhileBody function. WhileCond does not call forbody, and so does not
 869   # depend on any of forbody's extra_args. Since WhileCond and WhileBody
 870   # must have identical inputs, we have to augment the cond signature to take
 871   # the same types as the carried loop variables.
 872   body_sig = [dtypes.int32] * 4 + list(forbody.declared_input_types)[1:]
 873   cond_sig = body_sig + [t.dtype for t in forbody.captured_inputs]
 874 
 875   cond_name = "%s_Cond" % forbody.name
 876 
 877   @function.Defun(*cond_sig, func_name=cond_name)
 878   def WhileCond(i, n, *args):
 879     del args
 880     return i < n
 881 
 882   body_name = "%s_Body" % forbody.name
 883 
 884   @function.Defun(*body_sig, func_name=body_name)
 885   def WhileBody(i, n, start, delta, *args):
 886     """A While wrapper for forbody that handles loop-carried captured inputs."""
 887     for_result = forbody(start + i * delta, *args)
 888     # Nullary functions return an Operation. Normal functions can't do this
 889     # because their return values are converted to Tensors.
 890     if isinstance(for_result, ops.Operation):
 891       for_result = ()
 892     # Unary functions return a single Tensor value.
 893     elif isinstance(for_result, ops.Tensor):
 894       for_result = (for_result,)
 895     extra_args = tuple(function.get_extra_args())
 896     return (i + 1, n, start, delta) + tuple(for_result) + extra_args
 897 
 898   if hostmem is not None:
 899     hostmem = [0, 1, 2, 3] + [(4 + _) for _ in hostmem]
 900   else:
 901     hostmem = [0, 1, 2, 3]
 902 
 903   results = While(
 904       input_=[0, n, start, delta] + inputs + WhileBody.captured_inputs,
 905       cond=WhileCond,
 906       body=WhileBody,
 907       name=name,
 908       hostmem=hostmem)
 909   # Slice off the loop-carried captured inputs.
 910   return list(results[4:len(results) - len(WhileBody.captured_inputs)])

 913 def For(start,
 914         limit,
 915         delta,
 916         inputs,
 917         body,
 918         name=None,
 919         hostmem=None,
 920         rewrite_with_while=None):
 921   r"""out = input; for i in range(start, limit, delta) out = body(i, out).
 922 
 923   Args:
 924     start: A `Tensor` of type `int32`.
 925     limit: A `Tensor` of type `int32`.
 926     delta: A `Tensor` of type `int32`.
 927     inputs: A list of `Tensor` objects.
 928       A list of input tensors whose types are T.
 929     body: A function takes a list of tensors and returns another
 930       list of tensors. Both lists have the same types as (int32, T...).
 931     name: A name for the operation (optional).
 932     hostmem: A list of integer. If i is in the list, inputs[i] is a
 933       host memory tensor. In other words, (i+1)-th argument of the body
 934       function is expecting a host memory.
 935     rewrite_with_while: If True, using While op to implement the For.
 936 
 937   Returns:
 938     A list of `Tensor` objects. Has the same type as `input`.
 939     A list of output tensors whose types are T.
 940   """
 941   if rewrite_with_while:
 942     return _ForUsingWhile(start, limit, delta, inputs, body, name, hostmem)
 943   if body.captured_inputs:
 944     wrapper_name = "%s_BodyWrapper" % body.name
 945 
 946     @function.Defun(*body.declared_input_types, func_name=wrapper_name)
 947     def BodyWrapper(*args):
 948       """A wrapper for body that handles loop-carried captured inputs."""
 949       body_result = body(*args)
 950       extra_args = tuple(function.get_extra_args())
 951       # Nullary functions return an Operation. Normal functions can't do this
 952       # because their return values are converted to Tensors.
 953       if isinstance(body_result, ops.Operation):
 954         return extra_args
 955       # Unary functions return a single Tensor value.
 956       elif not isinstance(body_result, tuple):
 957         return (body_result,) + extra_args
 958       # N-ary functions return a tuple of Tensors.
 959       else:
 960         return body_result + extra_args
 961 
 962     inputs += BodyWrapper.captured_inputs
 963     ret = gen_functional_ops._for(
 964         start, limit, delta, inputs, BodyWrapper, name=name)
 965     # Slice off the loop-carried captured inputs.
 966     ret = ret[:-len(BodyWrapper.captured_inputs)]
 967   else:
 968     ret = gen_functional_ops._for(start, limit, delta, inputs, body, name=name)
 969   if hostmem:
 970     num_for_params = 3  # start/limit/delta
 971 
 972     input_attr = attr_value_pb2.AttrValue()
 973     input_attr.list.i.extend([num_for_params + i for i in hostmem])
 974     ret[0].op._set_attr("_input_hostmem", input_attr)  # pylint: disable=protected-access
 975 
 976     output_attr = attr_value_pb2.AttrValue()
 977     output_attr.list.i.extend(hostmem)
 978     ret[0].op._set_attr("_output_hostmem", output_attr)  # pylint: disable=protected-access
 979   return ret
