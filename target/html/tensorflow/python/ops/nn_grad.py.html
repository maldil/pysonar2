 222 @ops.RegisterGradient("Softmax")
 223 def _SoftmaxGrad(op, grad_softmax):
 224   """The derivative of the softmax nonlinearity.
 225 
 226   We assume that probs is of shape [batch_size * dim]
 227   The formula for dsoftmax / dx = (diag(softmax) - softmax * softmax').
 228   This matrix is diagonal minus a rank one matrix, so it is easy to implement
 229   as follows:
 230 
 231     grad_x = grad_softmax * softmax - sum(grad_softmax * softmax) * softmax
 232 
 233   Args:
 234      op: the Softmax op.
 235      grad_softmax:  the tensor representing the gradient w.r.t. the
 236        softmax output.
 237 
 238   Returns:
 239      gradient w.r.t the input to the softmax
 240 
 241   """
 242   softmax = op.outputs[0]
 243   sum_channels = math_ops.reduce_sum(grad_softmax * softmax, -1, keepdims=True)
 244   return (grad_softmax - sum_channels) * softmax
 245

 247 @ops.RegisterGradient("LogSoftmax")
 248 def _LogSoftmaxGrad(op, grad):
 249   """The gradient for log_softmax.
 250 
 251       log_softmax = input - log(sum(exp(input))
 252       dlog_softmax/dinput = diag - softmax(input)
 253 
 254   Args:
 255     op: The log softmax op.
 256     grad: The tensor representing the gradient w.r.t. the output.
 257 
 258   Returns:
 259     The gradients w.r.t. the input.
 260   """
 261   softmax = math_ops.exp(op.outputs[0])
 262   return grad - math_ops.reduce_sum(grad, -1, keepdims=True) * softmax

 265 @ops.RegisterGradient("BiasAdd")
 266 def _BiasAddGrad(op, received_grad):
 267   """Return the gradients for the 2 inputs of bias_op.
 268 
 269   The first input of unused_bias_op is the tensor t, and its gradient is
 270   just the gradient the unused_bias_op received.
 271 
 272   The second input of unused_bias_op is the bias vector which has one fewer
 273   dimension than "received_grad" (the batch dimension.)  Its gradient is the
 274   received gradient Summed on the batch dimension, which is the first dimension.
 275 
 276   Args:
 277     op: The BiasOp for which we need to generate gradients.
 278     received_grad: Tensor.  The gradients passed to the BiasOp.
 279 
 280   Returns:
 281     Two tensors, the first one for the "tensor" input of the BiasOp,
 282     the second one for the "bias" input of the BiasOp.
 283   """
 284   try:
 285     data_format = op.get_attr("data_format")
 286   except ValueError:
 287     data_format = None
 288   return (received_grad,
 289           gen_nn_ops.bias_add_grad(
 290               out_backprop=received_grad, data_format=data_format))

 293 @ops.RegisterGradient("BiasAddGrad")
 294 def _BiasAddGradGrad(op, received_grad):
 295   """Gradient for the BiasAddGrad op.
 296 
 297   Args:
 298     op: BiasAddGrad op for which we are calculating gradients.
 299     received_grad: The gradients passed to the BiasAddGrad op.
 300 
 301   Returns:
 302     A single gradient Tensor for the input to BiasAddGrad (which
 303     is the gradient of the bias term in BiasAdd)
 304   """
 305 
 306   try:
 307     data_format = op.get_attr("data_format")
 308   except ValueError:
 309     data_format = None
 310 
 311   shape = array_ops.shape(op.inputs[0])
 312   rank = array_ops.rank(op.inputs[0])
 313   bias_shape = array_ops.shape(received_grad)
 314 
 315   if data_format == b"NCHW":
 316     expanded_shape = array_ops.concat([
 317         array_ops.ones_like(shape[:-3]), bias_shape,
 318         array_ops.ones_like(shape[-2:])
 319     ], 0)
 320     tile_mults = array_ops.concat([shape[:-3], [1], shape[-2:]], 0)
 321   else:
 322     expanded_shape = array_ops.concat(
 323         [array_ops.ones_like(shape[:-1]), bias_shape], 0)
 324     tile_mults = array_ops.concat([shape[:-1], [1]], 0)
 325 
 326   expanded_grad = array_ops.reshape(received_grad, expanded_shape)
 327   return array_ops.tile(expanded_grad, tile_mults)

 407 @ops.RegisterGradient("SoftplusGrad")
 408 def _SoftplusGradGrad(op, grad):
 409   # Let:
 410   #   y = tf.nn.softplus(x)
 411   #   dx = gen_nn_ops.softplus_grad(dy, x) = dy / (1 + exp(-x))
 412   # This op computes (ddy, d2x) from op.inputs == [dy, x] and grad == ddx.
 413   dy, x = op.inputs
 414   with ops.control_dependencies([grad]):
 415     ddy = gen_nn_ops.softplus_grad(grad, x)
 416     d2x = grad * dy / (math_ops.exp(-x) + 2.0 + math_ops.exp(x))
 417     return (ddy, d2x)

 432 def _BroadcastMul(vec, mat):
 433   """Multiply after broadcasting vec to match dimensions of mat.
 434 
 435   Args:
 436     vec: A 1-D tensor of dimension [D0]
 437     mat: A 2-D tensor of dimension [D0, D1]
 438 
 439   Returns:
 440     A tensor of dimension [D0, D1], the result of vec * mat
 441   """
 442   # Reshape vec to [D0, 1]
 443   vec = array_ops.expand_dims(vec, -1)
 444   return vec * mat

 447 @ops.RegisterGradient("SoftmaxCrossEntropyWithLogits")
 448 def _SoftmaxCrossEntropyWithLogitsGrad(op, grad_loss, grad_grad):
 449   """Gradient function for SoftmaxCrossEntropyWithLogits."""
 450   # grad_loss is the backprop for cost, and we multiply it with the gradients
 451   # (which is output[1])
 452   # grad_grad is the backprop for softmax gradient.
 453   #
 454   # Second derivative is just softmax derivative w.r.t. logits.
 455   softmax_grad = op.outputs[1]
 456   grad = _BroadcastMul(grad_loss, softmax_grad)
 457 
 458   def IsZero(g):
 459     # Some introspection to check if the gradient is feeding zeros
 460     if context.executing_eagerly():
 461       # TODO(apassos) add an efficient way to detect eager zeros here.
 462       return False
 463     if g.op.type in ("ZerosLike", "Zeros"):
 464       return True
 465     const_fill_value = tensor_util.constant_value(g)
 466     return const_fill_value is not None and (const_fill_value == 0).all()
 467 
 468   logits = op.inputs[0]
 469   if grad_grad is not None and not IsZero(grad_grad):
 470     softmax = nn_ops.softmax(logits)
 471 
 472     grad += ((grad_grad - array_ops.squeeze(
 473         math_ops.matmul(array_ops.expand_dims(grad_grad, 1),
 474                         array_ops.expand_dims(softmax, 2)),
 475         axis=1)) *
 476              softmax)
 477 
 478   return grad, _BroadcastMul(grad_loss, -nn_ops.log_softmax(logits))

 481 @ops.RegisterGradient("SparseSoftmaxCrossEntropyWithLogits")
 482 def _SparseSoftmaxCrossEntropyWithLogitsGrad(op, grad_0, _):
 483   """Gradient function for SparseSoftmaxCrossEntropyWithLogits."""
 484   # grad_0 is the backprop for cost, and we multiply it with the gradients
 485   # (which is output[1])
 486   # There is no gradient for the labels
 487   #
 488   # Currently there is no way to take the second derivative of this op
 489   # due to the fused implementation's interaction with tf.gradients(),
 490   # so we make sure we prevent silently incorrect results by raising
 491   # an error if the second derivative is requested via prevent_gradient.
 492   sparse_softmax_grad_without_gradient = array_ops.prevent_gradient(
 493       op.outputs[1],
 494       message="Currently there is no way to take the second "
 495       "derivative of sparse_softmax_cross_entropy_with_logits due to the fused "
 496       "implementation's interaction with tf.gradients()")
 497   return _BroadcastMul(grad_0, sparse_softmax_grad_without_gradient), None

 729 @ops.RegisterGradient("BatchNormWithGlobalNormalization")
 730 def _BatchNormWithGlobalNormalizationGrad(op, grad):
 731   """Return the gradients for the 5 inputs of BatchNormWithGlobalNormalization.
 732 
 733   We do not backprop anything for the mean and var intentionally as they are
 734   not being trained with backprop in the operation.
 735

 737     op: The BatchNormOp for which we need to generate gradients.
 738     grad: Tensor.  The gradients passed to the BatchNormOp.

 751       op.inputs[0], op.inputs[1], op.inputs[2], op.inputs[4], grad,
 752       op.get_attr("variance_epsilon"), op.get_attr("scale_after_normalization"))
 753   return dx, dm, dv, db, dg

 756 def _BaseFusedBatchNormGrad(op, use_v2, *grad):
 757   """Return the gradients for the 3 inputs of BatchNorm.
 758 
 759   Args:
 760     op: The BatchNormOp for which we need to compute gradients.
 761     use_v2: Boolean indicating whether to use the V2 version of the fused batch
 762             norm gradient.
 763     *grad: An argument list for tensors of gradients wrt the outputs
 764           with grad[0] as grad_y.
 765 
 766   Returns:
 767     grad_x: gradient for x, which is scale * rsqrt(variance + epsilon) *
 768             [grad_y - mean(grad_y) - (x - mean(x)) *
 769             mean(grad_y * (x - mean(x))) / (variance + epsilon)]
 770             in training mode; grad_y * scale * rsqrt(pop_variance + epsilon)
 771             in freeze mode.
 772 
 773     grad_scale: gradient for scale, which is sum(grad_y * (x - mean(x)) *
 774                 rsqrt(variance + epsilon)) in training mode;
 775                 sum(grad_y * (x - pop_mean) * rsqrt(pop_variance + epsilon))
 776                 in freeze mode.
 777 
 778     grad_offset: gradient for offset, which is sum(grad_y) in training mode;
 779                  sum(grad_y) in freeze mode.
 780   """
 781   x = op.inputs[0]
 782   grad_y = grad[0]
 783   scale = op.inputs[1]
 784   epsilon = op.get_attr("epsilon")
 785   data_format = op.get_attr("data_format")
 786   is_training = op.get_attr("is_training")
 787   grad_fun = (
 788       gen_nn_ops.fused_batch_norm_grad_v2
 789       if use_v2 else gen_nn_ops.fused_batch_norm_grad)
 790   if is_training:
 791     return grad_fun(
 792         grad_y,
 793         x,
 794         scale,
 795         op.outputs[3],
 796         op.outputs[4],
 797         epsilon=epsilon,
 798         data_format=data_format,
 799         is_training=is_training)
 800   else:
 801     pop_mean = op.inputs[3]
 802     pop_var = op.inputs[4]
 803     if data_format == b"NCHW":
 804       x = array_ops.transpose(x, [0, 2, 3, 1])
 805       grad_y = array_ops.transpose(grad_y, [0, 2, 3, 1])
 806     dx, dscale, doffset, _, _ = grad_fun(
 807         grad_y,
 808         x,
 809         scale,
 810         pop_mean,
 811         pop_var,
 812         epsilon=epsilon,
 813         data_format="NHWC",
 814         is_training=is_training)
 815     if data_format == b"NCHW":
 816       dx = array_ops.transpose(dx, [0, 3, 1, 2])
 817     return dx, dscale, doffset, None, None

 830 def _BatchNormGrad(grad_y,
 831                    x,
 832                    scale,
 833                    pop_mean,
 834                    pop_var,
 835                    epsilon,
 836                    data_format,
 837                    is_training=True):
 838   """Returns the gradients for the 3 inputs of BatchNorm.
 839 
 840   Args:
 841     grad_y: A `Tensor` of 4 dimensions for gradient for y.
 842     x: A `Tensor` of 4 dimensions for x.
 843     scale: A `Tensor` of 1 dimension for scaling.
 844     pop_mean: A `Tensor` of 1 dimension for the population mean. Only used when
 845       is_training=False.
 846     pop_var: A `Tensor` of 1 dimension for the population variance. Only used
 847       when is_training=False.
 848     epsilon: A small float number added to the variance of x.
 849     data_format: The data format for input. Either b"NHWC" or b"NCHW".
 850     is_training: A bool value to indicate the operation is for training
 851       (default)
 852         or inference.
 853 
 854   Returns:
 855     A tuple (grad_x, grad_scale, grad_offset), where grad_x is the gradient
 856     for x, grad_scale the gradient for scale, and grad_offset the gradient
 857     for offset.
 858   """
 859   x_dtype = x.dtype.base_dtype
 860   if x_dtype == dtypes.float16:
 861     # float16 math is too imprecise, so we do the batch norm gradient
 862     # computations in float32.
 863     x = math_ops.cast(x, dtypes.float32)
 864     grad_y = math_ops.cast(grad_y, dtypes.float32)
 865   if is_training:
 866     if data_format == b"NHWC":
 867       keepdims = False
 868       reduce_axis = [0, 1, 2]
 869     else:
 870       keepdims = True
 871       reduce_axis = [0, 2, 3]
 872       shape = [1, array_ops.size(scale), 1, 1]
 873       scale = array_ops.reshape(scale, shape)
 874     mean_grad_y = math_ops.reduce_mean(grad_y, reduce_axis, keepdims=keepdims)
 875     mean_x = math_ops.reduce_mean(x, reduce_axis, keepdims=keepdims)
 876     var_x = math_ops.reduce_mean(
 877         math_ops.squared_difference(x, array_ops.stop_gradient(mean_x)),
 878         reduce_axis,
 879         keepdims=keepdims)
 880     grad_y_offset = grad_y - mean_grad_y
 881     x_offset = x - mean_x
 882     mean = math_ops.reduce_mean(
 883         grad_y * x_offset, axis=reduce_axis, keepdims=keepdims)
 884     grad_x = scale * math_ops.rsqrt(var_x + epsilon) * (
 885         grad_y_offset - math_ops.reciprocal(var_x + epsilon) * mean * x_offset)
 886     grad_scale = math_ops.rsqrt(var_x + epsilon) * math_ops.reduce_sum(
 887         grad_y * x_offset, axis=reduce_axis, keepdims=keepdims)
 888     if data_format == b"NCHW":
 889       grad_scale = array_ops.squeeze(grad_scale)
 890     grad_offset = math_ops.reduce_sum(grad_y, axis=reduce_axis)
 891     return math_ops.cast(grad_x, x_dtype), grad_scale, grad_offset
 892   else:
 893     if data_format == b"NHWC":
 894       reduce_axis = [0, 1, 2]
 895     else:
 896       reduce_axis = [0, 2, 3]
 897       shape = [1, array_ops.size(pop_mean), 1, 1]
 898       pop_mean = array_ops.reshape(pop_mean, shape)
 899       pop_var = array_ops.reshape(pop_var, shape)
 900       scale = array_ops.reshape(scale, shape)
 901 
 902     grad_offset = math_ops.reduce_sum(grad_y, axis=reduce_axis)
 903     var_rsqrt = math_ops.rsqrt(pop_var + epsilon)
 904     grad_scale = math_ops.reduce_sum(
 905         grad_y * (x - pop_mean) * var_rsqrt, axis=reduce_axis)
 906     grad_x = grad_y * scale * var_rsqrt
 907     return math_ops.cast(grad_x, x_dtype), grad_scale, grad_offset

 963 @ops.RegisterGradient("TopK")
 964 @ops.RegisterGradient("TopKV2")
 965 def _TopKGrad(op, grad, _):
 966   """Return the gradients for TopK.
 967 
 968   Args:
 969     op: The TopKOp for which we need to generate gradients.
 970     grad: Tensor. The gradients passed to the TopKOp.
 971 
 972   Returns:
 973     A list of two tensors, the first being the gradient w.r.t to the input and
 974     TopK, and the second being the gradient w.r.t. to the indices (all zero).
 975   """
 976   in_shape = array_ops.shape(op.inputs[0])
 977   ind_shape = array_ops.shape(op.outputs[1])
 978 
 979   # int32 is not supported on GPU hence up-casting
 980   ind_lastdim = array_ops.gather(math_ops.cast(
 981       ind_shape, dtypes.int64), array_ops.size(ind_shape) - 1)
 982   # Flatten indices to 2D.
 983   ind_2d = array_ops.reshape(op.outputs[1], array_ops.stack([-1, ind_lastdim]))
 984 
 985   in_lastdim = array_ops.gather(math_ops.cast(
 986       in_shape, dtypes.int64), array_ops.size(in_shape) - 1)
 987   outerdim = array_ops.shape(ind_2d)[0]
 988   # Compute linear indices (flattened to 1D).
 989   ind = array_ops.reshape(ind_2d + math_ops.cast(array_ops.expand_dims(
 990       math_ops.range(0, math_ops.cast(outerdim, dtypes.int64)
 991                      * in_lastdim, in_lastdim), -1), dtypes.int32), [-1])
 992 
 993   # Substitute grad to appropriate locations and fill the rest with zeros,
 994   # finally reshaping it to the original input shape.
 995   return [
 996       array_ops.reshape(
 997           array_ops.scatter_nd(
 998               array_ops.expand_dims(ind, -1),
 999               array_ops.reshape(grad, [-1]),
1000               [math_ops.reduce_prod(in_shape)]
1001           ),
1002           in_shape),
1003       array_ops.zeros([], dtype=dtypes.int32)
1004   ]

1007 @ops.RegisterGradient("NthElement")
1008 def _NthElementGrad(op, grad):
1009   """Return the gradients for NthElement.
1010 
1011   Args:
1012     op: The NthElementOp for which we need to generate gradients.
1013     grad: Tensor. The gradients passed to the NthElementOp
1014 
1015   Returns:
1016     A list of two tensors, the first being the gradient w.r.t. the input,
1017     the second being the gradient w.r.t. the N (None).
1018   """
1019   input = op.inputs[0]  # pylint: disable=redefined-builtin
1020   output = op.outputs[0]
1021 
1022   # Compute the number of elements which equal to output in each reduction
1023   # dimension. If there are multiple elements then the gradient will be
1024   # divided between them.
1025   indicators = math_ops.cast(
1026       math_ops.equal(array_ops.expand_dims(output, -1), input), grad.dtype)
1027 
1028   grad = array_ops.expand_dims(grad, -1)
1029   num_selected = array_ops.expand_dims(math_ops.reduce_sum(indicators, -1), -1)
1030 
1031   return [math_ops.div(indicators, num_selected) * grad, None]
