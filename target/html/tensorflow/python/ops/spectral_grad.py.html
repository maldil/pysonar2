  29 def _FFTSizeForGrad(grad, rank):
  30   return math_ops.reduce_prod(array_ops.shape(grad)[-rank:])
  31 
  32 
  33 @ops.RegisterGradient("FFT")
  34 def _FFTGrad(_, grad):
  35   size = math_ops.cast(_FFTSizeForGrad(grad, 1), grad.dtype)
  36   return spectral_ops.ifft(grad) * size

  39 @ops.RegisterGradient("IFFT")
  40 def _IFFTGrad(_, grad):
  41   rsize = math_ops.cast(
  42       1. / math_ops.cast(_FFTSizeForGrad(grad, 1), grad.dtype.real_dtype),
  43       grad.dtype)
  44   return spectral_ops.fft(grad) * rsize

  75 def _RFFTGradHelper(rank, irfft_fn):
  76   """Returns a gradient function for an RFFT of the provided rank."""
  77   # Can't happen because we don't register a gradient for RFFT3D.
  78   assert rank in (1, 2), "Gradient for RFFT3D is not implemented."
  79 
  80   def _Grad(op, grad):
  81     """A gradient function for RFFT with the provided `rank` and `irfft_fn`."""
  82     fft_length = op.inputs[1]
  83     input_shape = array_ops.shape(op.inputs[0])
  84     is_even = math_ops.cast(1 - (fft_length[-1] % 2), dtypes.complex64)
  85 
  86     def _TileForBroadcasting(matrix, t):
  87       expanded = array_ops.reshape(
  88           matrix,
  89           array_ops.concat([
  90               array_ops.ones([array_ops.rank(t) - 2], dtypes.int32),
  91               array_ops.shape(matrix)
  92           ], 0))
  93       return array_ops.tile(
  94           expanded, array_ops.concat([array_ops.shape(t)[:-2], [1, 1]], 0))
  95 
  96     def _MaskMatrix(length):
  97       # TODO(rjryan): Speed up computation of twiddle factors using the
  98       # following recurrence relation and cache them across invocations of RFFT.
  99       #
 100       # t_n = exp(sqrt(-1) * pi * n^2 / line_len)
 101       # for n = 0, 1,..., line_len-1.
 102       # For n > 2, use t_n = t_{n-1}^2 / t_{n-2} * t_1^2
 103       a = array_ops.tile(
 104           array_ops.expand_dims(math_ops.range(length), 0), (length, 1))
 105       b = array_ops.transpose(a, [1, 0])
 106       return math_ops.exp(-2j * np.pi * math_ops.cast(a * b, dtypes.complex64) /
 107                           math_ops.cast(length, dtypes.complex64))
 108 
 109     def _YMMask(length):
 110       """A sequence of [1+0j, -1+0j, 1+0j, -1+0j, ...] with length `length`."""
 111       return math_ops.cast(1 - 2 * (math_ops.range(length) % 2),
 112                            dtypes.complex64)
 113 
 114     y0 = grad[..., 0:1]
 115     if rank == 1:
 116       ym = grad[..., -1:]
 117       extra_terms = y0 + is_even * ym * _YMMask(input_shape[-1])
 118     elif rank == 2:
 119       # Create a mask matrix for y0 and ym.
 120       base_mask = _MaskMatrix(input_shape[-2])
 121 
 122       # Tile base_mask to match y0 in shape so that we can batch-matmul the
 123       # inner 2 dimensions.
 124       tiled_mask = _TileForBroadcasting(base_mask, y0)
 125 
 126       y0_term = math_ops.matmul(tiled_mask, math_ops.conj(y0))
 127       extra_terms = y0_term
 128 
 129       ym = grad[..., -1:]
 130       ym_term = math_ops.matmul(tiled_mask, math_ops.conj(ym))
 131 
 132       inner_dim = input_shape[-1]
 133       ym_term = array_ops.tile(
 134           ym_term,
 135           array_ops.concat([
 136               array_ops.ones([array_ops.rank(grad) - 1], dtypes.int32),
 137               [inner_dim]
 138           ], 0)) * _YMMask(inner_dim)
 139 
 140       extra_terms += is_even * ym_term
 141 
 142     # The gradient of RFFT is the IRFFT of the incoming gradient times a scaling
 143     # factor, plus some additional terms to make up for the components dropped
 144     # due to Hermitian symmetry.
 145     input_size = math_ops.to_float(_FFTSizeForGrad(op.inputs[0], rank))
 146     irfft = irfft_fn(grad, fft_length)
 147     return 0.5 * (irfft * input_size + math_ops.real(extra_terms)), None
 148 
 149   return _Grad

 152 def _IRFFTGradHelper(rank, rfft_fn):
 153   """Returns a gradient function for an IRFFT of the provided rank."""
 154   # Can't happen because we don't register a gradient for IRFFT3D.
 155   assert rank in (1, 2), "Gradient for IRFFT3D is not implemented."
 156 
 157   def _Grad(op, grad):
 158     """A gradient function for IRFFT with the provided `rank` and `rfft_fn`."""
 159     # Generate a simple mask like [1.0, 2.0, ..., 2.0, 1.0] for even-length FFTs
 160     # and [1.0, 2.0, ..., 2.0] for odd-length FFTs. To reduce extra ops in the
 161     # graph we special-case the situation where the FFT length and last
 162     # dimension of the input are known at graph construction time.
 163     fft_length = op.inputs[1]
 164     is_odd = math_ops.mod(fft_length[-1], 2)
 165     input_last_dimension = array_ops.shape(op.inputs[0])[-1]
 166     mask = array_ops.concat(
 167         [[1.0], 2.0 * array_ops.ones([input_last_dimension - 2 + is_odd]),
 168          array_ops.ones([1 - is_odd])], 0)
 169 
 170     rsize = math_ops.reciprocal(math_ops.to_float(_FFTSizeForGrad(grad, rank)))
 171 
 172     # The gradient of IRFFT is the RFFT of the incoming gradient times a scaling
 173     # factor and a mask. The mask scales the gradient for the Hermitian
 174     # symmetric components of the RFFT by a factor of two, since these
 175     # components are de-duplicated in the RFFT.
 176     rfft = rfft_fn(grad, fft_length)
 177     return rfft * math_ops.cast(rsize * mask, dtypes.complex64), None
 178 
 179   return _Grad
