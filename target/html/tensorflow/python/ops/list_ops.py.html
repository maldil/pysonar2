  36 @ops.RegisterGradient("TensorListPushBack")
  37 def _PushBackGrad(op, dresult):
  38   return gen_list_ops.tensor_list_pop_back(
  39       dresult, element_dtype=op.get_attr("element_dtype"))

  42 @ops.RegisterGradient("TensorListPopBack")
  43 def _PopBackGrad(op, dlist, delement):
  44   if dlist is None:
  45     dlist = gen_list_ops.empty_tensor_list(
  46         element_dtype=delement.dtype,
  47         element_shape=gen_list_ops.tensor_list_element_shape(
  48             op.outputs[0], shape_type=dtypes.int32))
  49   return gen_list_ops.tensor_list_push_back(dlist, delement)

  52 @ops.RegisterGradient("TensorListStack")
  53 def _TensorListStackGrad(unused_op, dtensor):
  54   return gen_list_ops.tensor_list_from_tensor(dtensor,
  55                                               element_shape=dtensor.shape[1:])

  58 @ops.RegisterGradient("TensorListFromTensor")
  59 def _TensorListFromTensorGrad(op, dlist):
  60   """Gradient for TensorListFromTensor."""
  61   if op.inputs[0].shape[0].value is not None:
  62     num_elements = op.inputs[0].shape[0].value
  63   else:
  64     num_elements = None
  65   if dlist is None:
  66     dlist = gen_list_ops.empty_tensor_list(
  67         element_dtype=op.inputs[0].dtype,
  68         element_shape=gen_list_ops.tensor_list_element_shape(
  69             op.outputs[0], shape_type=dtypes.int32))
  70   tensor_grad = gen_list_ops.tensor_list_stack(
  71       dlist, element_dtype=op.inputs[0].dtype, num_elements=num_elements)
  72   shape_grad = None
  73   return tensor_grad, shape_grad

  76 @ops.RegisterGradient("TensorListGetItem")
  77 def _TensorListGetItemGrad(op, ditem):
  78   """Gradient for TensorListGetItem."""
  79   list_size = gen_list_ops.tensor_list_length(op.inputs[0])
  80   list_grad = gen_list_ops.tensor_list_set_item(
  81       gen_list_ops.tensor_list_reserve(
  82           gen_list_ops.tensor_list_element_shape(op.inputs[0],
  83                                                  shape_type=dtypes.int32),
  84           list_size, element_dtype=ditem.dtype),
  85       index=op.inputs[1],
  86       item=ditem)
  87   index_grad = None
  88   return list_grad, index_grad
  89

  91 @ops.RegisterGradient("TensorListSetItem")
  92 def _TensorListSetItemGrad(op, dlist):
  93   _, index, item = op.inputs
  94   list_grad = gen_list_ops.tensor_list_set_item(
  95       dlist, index=index, item=array_ops.zeros_like(item))
  96   index_grad = None
  97   element_grad = gen_list_ops.tensor_list_get_item(
  98       dlist, index, element_dtype=item.dtype)
  99   return list_grad, index_grad, element_grad

 102 @ops.RegisterGradient("TensorListGather")
 103 def _TensorListGatherGrad(op, dtensor):
 104   _, indices = op.inputs
 105   return gen_list_ops.tensor_list_scatter(
 106       tensor=dtensor, indices=indices,
 107       element_shape=ops.convert_to_tensor(-1, dtype=dtypes.int32)), None

 110 @ops.RegisterGradient("TensorListScatter")
 111 def _TensorListScatterGrad(op, dlist):
 112   t, indices, _ = op.inputs
 113   return gen_list_ops.tensor_list_gather(
 114       dlist, indices, element_dtype=t.dtype), None
