  61 @tf_export("VariableSynchronization")
  62 class VariableSynchronization(enum.Enum):
  63   """Indicates when a distributed variable will be synced.
  64 
  65   * `AUTO`: Indicates that the synchronization will be determined by the current
  66     `DistributionStrategy` (eg. With `MirroredStrategy` this would be
  67     `ON_WRITE`).
  68   * `NONE`: Indicates that there will only be one copy of the variable, so
  69     there is no need to sync.
  70   * `ON_WRITE`: Indicates that the variable will be updated across devices
  71     every time it is written.
  72   * `ON_READ`: Indicates that the variable will be aggregated across devices
  73     when it is read (eg. when checkpointing or when evaluating an op that uses
  74     the variable).
  75   """
  76   AUTO = 0
  77   NONE = 1
  78   ON_WRITE = 2
  79   ON_READ = 3

  82 @tf_export("VariableAggregation")
  83 class VariableAggregation(enum.Enum):
  84   """Indicates how a distributed variable will be aggregated.
  85 
  86   `tf.contrib.distribute.DistributionStrategy` distributes a model by making
  87   multiple copies (called "towers") acting data-parallel on different elements
  88   of the input batch. When performing some variable-update operation, say
  89   `var.assign_add(x)`, in a model, we need to resolve how to combine the
  90   different values for `x` computed in the different towers.
  91 
  92   * `NONE`: This is the default, giving an error if you use a
  93     variable-update operation with multiple towers.
  94   * `SUM`: Add the updates across towers.
  95   * `MEAN`: Take the arithmetic mean ("average") of the updates across towers.
  96   * `ONLY_FIRST_TOWER`: This is for when every tower is performing the same
  97     update, but we only want to perform the update once. Used, e.g., for the
  98     global step counter.
  99   """
 100   NONE = 0
 101   SUM = 1
 102   MEAN = 2
 103   ONLY_FIRST_TOWER = 3

 190 @tf_export(v2=["Variable"])
 191 class Variable(six.with_metaclass(VariableMetaclass,
 192                                   checkpointable.CheckpointableBase)):
 193   """See the [Variables Guide](https://tensorflow.org/guide/variables).
 194 
 195   A variable maintains state in the graph across calls to `run()`. You add a
 196   variable to the graph by constructing an instance of the class `Variable`.
 197 
 198   The `Variable()` constructor requires an initial value for the variable,
 199   which can be a `Tensor` of any type and shape. The initial value defines the
 200   type and shape of the variable. After construction, the type and shape of
 201   the variable are fixed. The value can be changed using one of the assign
 202   methods.
 203 
 204   If you want to change the shape of a variable later you have to use an
 205   `assign` Op with `validate_shape=False`.
 206 
 207   Just like any `Tensor`, variables created with `Variable()` can be used as
 208   inputs for other Ops in the graph. Additionally, all the operators
 209   overloaded for the `Tensor` class are carried over to variables, so you can
 210   also add nodes to the graph by just doing arithmetic on variables.
 211 
 212   ```python
 213   import tensorflow as tf
 214 
 215   # Create a variable.
 216   w = tf.Variable(<initial-value>, name=<optional-name>)
 217 
 218   # Use the variable in the graph like any Tensor.
 219   y = tf.matmul(w, ...another variable or tensor...)
 220 
 221   # The overloaded operators are available too.
 222   z = tf.sigmoid(w + y)
 223 
 224   # Assign a new value to the variable with `assign()` or a related method.
 225   w.assign(w + 1.0)
 226   w.assign_add(1.0)
 227   ```
 228 
 229   When you launch the graph, variables have to be explicitly initialized before
 230   you can run Ops that use their value. You can initialize a variable by
 231   running its *initializer op*, restoring the variable from a save file, or
 232   simply running an `assign` Op that assigns a value to the variable. In fact,
 233   the variable *initializer op* is just an `assign` Op that assigns the
 234   variable's initial value to the variable itself.
 235 
 236   ```python
 237   # Launch the graph in a session.
 238   with tf.Session() as sess:
 239       # Run the variable initializer.
 240       sess.run(w.initializer)
 241       # ...you now can run ops that use the value of 'w'...
 242   ```
 243 
 244   The most common initialization pattern is to use the convenience function
 245   `global_variables_initializer()` to add an Op to the graph that initializes
 246   all the variables. You then run that Op after launching the graph.
 247 
 248   ```python
 249   # Add an Op to initialize global variables.
 250   init_op = tf.global_variables_initializer()
 251 
 252   # Launch the graph in a session.
 253   with tf.Session() as sess:
 254       # Run the Op that initializes global variables.
 255       sess.run(init_op)
 256       # ...you can now run any Op that uses variable values...
 257   ```
 258 
 259   If you need to create a variable with an initial value dependent on another
 260   variable, use the other variable's `initialized_value()`. This ensures that
 261   variables are initialized in the right order.
 262 
 263   All variables are automatically collected in the graph where they are
 264   created. By default, the constructor adds the new variable to the graph
 265   collection `GraphKeys.GLOBAL_VARIABLES`. The convenience function
 266   `global_variables()` returns the contents of that collection.
 267 
 268   When building a machine learning model it is often convenient to distinguish
 269   between variables holding the trainable model parameters and other variables
 270   such as a `global step` variable used to count training steps. To make this
 271   easier, the variable constructor supports a `trainable=<bool>` parameter. If
 272   `True`, the new variable is also added to the graph collection
 273   `GraphKeys.TRAINABLE_VARIABLES`. The convenience function
 274   `trainable_variables()` returns the contents of this collection. The
 275   various `Optimizer` classes use this collection as the default list of
 276   variables to optimize.
 277 
 278   WARNING: tf.Variable objects by default have a non-intuitive memory model. A
 279   Variable is represented internally as a mutable Tensor which can
 280   non-deterministically alias other Tensors in a graph. The set of operations
 281   which consume a Variable and can lead to aliasing is undetermined and can
 282   change across TensorFlow versions. Avoid writing code which relies on the
 283   value of a Variable either changing or not changing as other operations
 284   happen. For example, using Variable objects or simple functions thereof as
 285   predicates in a `tf.cond` is dangerous and error-prone:
 286 
 287   ```
 288   v = tf.Variable(True)
 289   tf.cond(v, lambda: v.assign(False), my_false_fn)  # Note: this is broken.
 290   ```
 291 
 292   Here replacing adding `use_resource=True` when constructing the variable will
 293   fix any nondeterminism issues:
 294   ```
 295   v = tf.Variable(True, use_resource=True)
 296   tf.cond(v, lambda: v.assign(False), my_false_fn)
 297   ```
 298 
 299   To use the replacement for variables which does
 300   not have these issues:
 301 
 302   * Add `use_resource=True` when constructing `tf.Variable`;
 303   * Call `tf.get_variable_scope().set_use_resource(True)` inside a
 304     `tf.variable_scope` before the `tf.get_variable()` call.
 305   """
 306 
 307   def __init__(self,
 308                initial_value=None,
 309                trainable=True,
 310                validate_shape=True,
 311                caching_device=None,
 312                name=None,
 313                variable_def=None,
 314                dtype=None,
 315                import_scope=None,
 316                constraint=None,
 317                synchronization=VariableSynchronization.AUTO,
 318                aggregation=VariableAggregation.NONE):
 319     """Creates a new variable with value `initial_value`.
 320 
 321     The new variable is added to the graph collections listed in `collections`,
 322     which defaults to `[GraphKeys.GLOBAL_VARIABLES]`.
 323 
 324     If `trainable` is `True` the variable is also added to the graph collection
 325     `GraphKeys.TRAINABLE_VARIABLES`.
 326 
 327     This constructor creates both a `variable` Op and an `assign` Op to set the
 328     variable to its initial value.
 329 
 330     Args:
 331       initial_value: A `Tensor`, or Python object convertible to a `Tensor`,
 332         which is the initial value for the Variable. The initial value must have
 333         a shape specified unless `validate_shape` is set to False. Can also be a
 334         callable with no argument that returns the initial value when called. In
 335         that case, `dtype` must be specified. (Note that initializer functions
 336         from init_ops.py must first be bound to a shape before being used here.)
 337       trainable: If `True`, the default, GradientTapes automatically watch uses
 338         of this variable.
 339       validate_shape: If `False`, allows the variable to be initialized with a
 340         value of unknown shape. If `True`, the default, the shape of
 341         `initial_value` must be known.
 342       caching_device: Optional device string describing where the Variable
 343         should be cached for reading.  Defaults to the Variable's device.
 344         If not `None`, caches on another device.  Typical use is to cache
 345         on the device where the Ops using the Variable reside, to deduplicate
 346         copying through `Switch` and other conditional statements.
 347       name: Optional name for the variable. Defaults to `'Variable'` and gets
 348         uniquified automatically.
 349       variable_def: `VariableDef` protocol buffer. If not `None`, recreates
 350         the Variable object with its contents, referencing the variable's nodes
 351         in the graph, which must already exist. The graph is not changed.
 352         `variable_def` and the other arguments are mutually exclusive.
 353       dtype: If set, initial_value will be converted to the given type.
 354         If `None`, either the datatype will be kept (if `initial_value` is
 355         a Tensor), or `convert_to_tensor` will decide.
 356       import_scope: Optional `string`. Name scope to add to the
 357         `Variable.` Only used when initializing from protocol buffer.
 358       constraint: An optional projection function to be applied to the variable
 359         after being updated by an `Optimizer` (e.g. used to implement norm
 360         constraints or value constraints for layer weights). The function must
 361         take as input the unprojected Tensor representing the value of the
 362         variable and return the Tensor for the projected value
 363         (which must have the same shape). Constraints are not safe to
 364         use when doing asynchronous distributed training.
 365       synchronization: Indicates when a distributed a variable will be
 366         aggregated. Accepted values are constants defined in the class
 367         `tf.VariableSynchronization`. By default the synchronization is set to
 368         `AUTO` and the current `DistributionStrategy` chooses
 369         when to synchronize. If `synchronization` is set to `ON_READ`,
 370         `trainable` must not be set to `True`.
 371       aggregation: Indicates how a distributed variable will be aggregated.
 372         Accepted values are constants defined in the class
 373         `tf.VariableAggregation`.
 374 
 375     Raises:
 376       ValueError: If both `variable_def` and initial_value are specified.
 377       ValueError: If the initial value is not specified, or does not have a
 378         shape and `validate_shape` is `True`.
 379       RuntimeError: If eager execution is enabled.
 380     """
 381     raise NotImplementedError

1043   """See the [Variables Guide](https://tensorflow.org/guide/variables).
1044 
1045   A variable maintains state in the graph across calls to `run()`. You add a
1046   variable to the graph by constructing an instance of the class `Variable`.
1047 
1048   The `Variable()` constructor requires an initial value for the variable,
1049   which can be a `Tensor` of any type and shape. The initial value defines the
1050   type and shape of the variable. After construction, the type and shape of
1051   the variable are fixed. The value can be changed using one of the assign
1052   methods.
1053 
1054   If you want to change the shape of a variable later you have to use an
1055   `assign` Op with `validate_shape=False`.
1056 
1057   Just like any `Tensor`, variables created with `Variable()` can be used as
1058   inputs for other Ops in the graph. Additionally, all the operators
1059   overloaded for the `Tensor` class are carried over to variables, so you can
1060   also add nodes to the graph by just doing arithmetic on variables.
1061 
1062   ```python
1063   import tensorflow as tf
1064 
1065   # Create a variable.
1066   w = tf.Variable(<initial-value>, name=<optional-name>)
1067 
1068   # Use the variable in the graph like any Tensor.
1069   y = tf.matmul(w, ...another variable or tensor...)
1070 
1071   # The overloaded operators are available too.
1072   z = tf.sigmoid(w + y)
1073 
1074   # Assign a new value to the variable with `assign()` or a related method.
1075   w.assign(w + 1.0)
1076   w.assign_add(1.0)
1077   ```
1078 
1079   When you launch the graph, variables have to be explicitly initialized before
1080   you can run Ops that use their value. You can initialize a variable by
1081   running its *initializer op*, restoring the variable from a save file, or
1082   simply running an `assign` Op that assigns a value to the variable. In fact,
1083   the variable *initializer op* is just an `assign` Op that assigns the
1084   variable's initial value to the variable itself.
1085 
1086   ```python
1087   # Launch the graph in a session.
1088   with tf.Session() as sess:
1089       # Run the variable initializer.
1090       sess.run(w.initializer)
1091       # ...you now can run ops that use the value of 'w'...
1092   ```
1093 
1094   The most common initialization pattern is to use the convenience function
1095   `global_variables_initializer()` to add an Op to the graph that initializes
1096   all the variables. You then run that Op after launching the graph.
1097 
1098   ```python
1099   # Add an Op to initialize global variables.
1100   init_op = tf.global_variables_initializer()
1101 
1102   # Launch the graph in a session.
1103   with tf.Session() as sess:
1104       # Run the Op that initializes global variables.
1105       sess.run(init_op)
1106       # ...you can now run any Op that uses variable values...
1107   ```
1108 
1109   If you need to create a variable with an initial value dependent on another
1110   variable, use the other variable's `initialized_value()`. This ensures that
1111   variables are initialized in the right order.
1112 
1113   All variables are automatically collected in the graph where they are
1114   created. By default, the constructor adds the new variable to the graph
1115   collection `GraphKeys.GLOBAL_VARIABLES`. The convenience function
1116   `global_variables()` returns the contents of that collection.
1117 
1118   When building a machine learning model it is often convenient to distinguish
1119   between variables holding the trainable model parameters and other variables
1120   such as a `global step` variable used to count training steps. To make this
1121   easier, the variable constructor supports a `trainable=<bool>` parameter. If
1122   `True`, the new variable is also added to the graph collection
1123   `GraphKeys.TRAINABLE_VARIABLES`. The convenience function
1124   `trainable_variables()` returns the contents of this collection. The
1125   various `Optimizer` classes use this collection as the default list of
1126   variables to optimize.
1127 
1128   WARNING: tf.Variable objects by default have a non-intuitive memory model. A
1129   Variable is represented internally as a mutable Tensor which can
1130   non-deterministically alias other Tensors in a graph. The set of operations
1131   which consume a Variable and can lead to aliasing is undetermined and can
1132   change across TensorFlow versions. Avoid writing code which relies on the
1133   value of a Variable either changing or not changing as other operations
1134   happen. For example, using Variable objects or simple functions thereof as
1135   predicates in a `tf.cond` is dangerous and error-prone:
1136 
1137   ```
1138   v = tf.Variable(True)
1139   tf.cond(v, lambda: v.assign(False), my_false_fn)  # Note: this is broken.
1140   ```
1141 
1142   Here replacing adding `use_resource=True` when constructing the variable will
1143   fix any nondeterminism issues:
1144   ```
1145   v = tf.Variable(True, use_resource=True)
1146   tf.cond(v, lambda: v.assign(False), my_false_fn)
1147   ```
1148 
1149   To use the replacement for variables which does
1150   not have these issues:
1151 
1152   * Add `use_resource=True` when constructing `tf.Variable`;
1153   * Call `tf.get_variable_scope().set_use_resource(True)` inside a
1154     `tf.variable_scope` before the `tf.get_variable()` call.

2132   def _try_guard_against_uninitialized_dependencies(self, initial_value):
2133     """Attempt to guard against dependencies on uninitialized variables.
2134 
2135     Replace references to variables in `initial_value` with references to the
2136     variable's initialized values. The initialized values are essentially
2137     conditional TensorFlow graphs that return a variable's value if it is
2138     initialized or its `initial_value` if it hasn't been initialized. This
2139     replacement is done on a best effort basis:
2140 
2141     - If the `initial_value` graph contains cycles, we don't do any
2142       replacements for that graph.
2143     - If the variables that `initial_value` depends on are not present in the
2144       `GLOBAL_VARIABLES` or `LOCAL_VARIABLES` we don't replace them.
2145 
2146     In these cases, it is up to the caller to ensure that the `initial_value`
2147     graph uses initialized variables or that they guard access to variables
2148     using their `initialized_value` method.
2149 
2150     Args:
2151       initial_value: `Tensor`. The initial value.
2152     Returns:
2153       A `Tensor` suitable to initialize a variable.
2154     Raises:
2155       TypeError: If `initial_value` is not a `Tensor`.
2156     """
2157     if not isinstance(initial_value, ops.Tensor):
2158       raise TypeError("initial_value needs to be a Tensor: %s" % initial_value)
2159 
2160     # Don't modify initial_value if it contains any cyclic dependencies.
2161     def has_cycle(op, path):
2162       """Detect cycles in the dependencies of `initial_value`."""
2163       if op.name in path:
2164         return True
2165       path.add(op.name)
2166       for op_input in op.inputs:
2167         if has_cycle(op_input.op, path):
2168           return True
2169       for op_control_input in op.control_inputs:
2170         if has_cycle(op_control_input, path):
2171           return True
2172       path.remove(op.name)
2173       return False
2174     if has_cycle(initial_value.op, path=set()):
2175       return initial_value
2176 
2177     return self._safe_initial_value_from_tensor(initial_value, op_cache={})
2178 
2179   def _safe_initial_value_from_tensor(self, tensor, op_cache):
2180     """Replace dependencies on variables with their initialized values.
2181 
2182     Args:
2183       tensor: A `Tensor`. The tensor to replace.
2184       op_cache: A dict mapping operation names to `Operation`s. Used to memoize
2185         the results so as to avoid creating redundant operations.
2186     Returns:
