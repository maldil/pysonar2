  73 def _maybe_constant_value_string(t):
  74   if not isinstance(t, ops.Tensor):
  75     return str(t)
  76   const_t = tensor_util.constant_value(t)
  77   if const_t is not None:
  78     return str(const_t)
  79   return t

  82 def _assert_static(condition, data):
  83   """Raises a InvalidArgumentError with as much information as possible."""
  84   if not condition:
  85     data_static = [_maybe_constant_value_string(x) for x in data]
  86     raise errors.InvalidArgumentError(node_def=None, op=None,
  87                                       message='\n'.join(data_static))

  95 @tf_export('debugging.assert_proper_iterable', 'assert_proper_iterable')
  96 @deprecation.deprecated_endpoints('assert_proper_iterable')
  97 def assert_proper_iterable(values):
  98   """Static assert that values is a "proper" iterable.
  99 
 100   `Ops` that expect iterables of `Tensor` can call this to validate input.
 101   Useful since `Tensor`, `ndarray`, byte/text type are all iterables themselves.
 102 
 103   Args:
 104     values:  Object to be checked.
 105 
 106   Raises:
 107     TypeError:  If `values` is not iterable or is one of
 108       `Tensor`, `SparseTensor`, `np.array`, `tf.compat.bytes_or_text_types`.
 109   """
 110   unintentional_iterables = (
 111       (ops.Tensor, sparse_tensor.SparseTensor, np.ndarray)
 112       + compat.bytes_or_text_types
 113   )
 114   if isinstance(values, unintentional_iterables):
 115     raise TypeError(
 116         'Expected argument "values" to be a "proper" iterable.  Found: %s' %
 117         type(values))
 118 
 119   if not hasattr(values, '__iter__'):
 120     raise TypeError(
 121         'Expected argument "values" to be iterable.  Found: %s' % type(values))

 124 @tf_export('debugging.assert_negative', 'assert_negative')
 125 @deprecation.deprecated_endpoints('assert_negative')
 126 def assert_negative(x, data=None, summarize=None, message=None, name=None):
 127   """Assert the condition `x < 0` holds element-wise.
 128 
 129   Example of adding a dependency to an operation:
 130 
 131   ```python
 132   with tf.control_dependencies([tf.assert_negative(x)]):
 133     output = tf.reduce_sum(x)
 134   ```
 135 
 136   Negative means, for every element `x[i]` of `x`, we have `x[i] < 0`.
 137   If `x` is empty this is trivially satisfied.
 138 
 139   Args:
 140     x:  Numeric `Tensor`.
 141     data:  The tensors to print out if the condition is False.  Defaults to
 142       error message and first few entries of `x`.
 143     summarize: Print this many entries of each tensor.
 144     message: A string to prefix to the default message.
 145     name: A name for this operation (optional).  Defaults to "assert_negative".
 146 
 147   Returns:
 148     Op raising `InvalidArgumentError` unless `x` is all negative.
 149   """
 150   message = message or ''
 151   with ops.name_scope(name, 'assert_negative', [x, data]):
 152     x = ops.convert_to_tensor(x, name='x')
 153     if data is None:
 154       if context.executing_eagerly():
 155         name = _shape_and_dtype_str(x)
 156       else:
 157         name = x.name
 158       data = [
 159           message,
 160           'Condition x < 0 did not hold element-wise:',
 161           'x (%s) = ' % name, x]
 162     zero = ops.convert_to_tensor(0, dtype=x.dtype)
 163     return assert_less(x, zero, data=data, summarize=summarize)

 293 @tf_export('debugging.assert_equal', 'assert_equal')
 294 def assert_equal(x, y, data=None, summarize=None, message=None, name=None):
 295   """Assert the condition `x == y` holds element-wise.
 296 
 297   Example of adding a dependency to an operation:
 298 
 299   ```python
 300   with tf.control_dependencies([tf.assert_equal(x, y)]):
 301     output = tf.reduce_sum(x)
 302   ```
 303 
 304   This condition holds if for every pair of (possibly broadcast) elements
 305   `x[i]`, `y[i]`, we have `x[i] == y[i]`.
 306   If both `x` and `y` are empty, this is trivially satisfied.
 307 
 308   Args:
 309     x:  Numeric `Tensor`.
 310     y:  Numeric `Tensor`, same dtype as and broadcastable to `x`.
 311     data:  The tensors to print out if the condition is False.  Defaults to
 312       error message and first few entries of `x`, `y`.
 313     summarize: Print this many entries of each tensor.
 314     message: A string to prefix to the default message.
 315     name: A name for this operation (optional).  Defaults to "assert_equal".
 316 
 317   Returns:
 318     Op that raises `InvalidArgumentError` if `x == y` is False.
 319     @compatibility{eager} returns None
 320 
 321   Raises:
 322     InvalidArgumentError: if the check can be performed immediately and
 323       `x == y` is False. The check can be performed immediately during eager
 324       execution or if `x` and `y` are statically known.
 325   """
 326   message = message or ''
 327   with ops.name_scope(name, 'assert_equal', [x, y, data]):
 328     x = ops.convert_to_tensor(x, name='x')
 329     y = ops.convert_to_tensor(y, name='y')
 330 
 331     if context.executing_eagerly():
 332       eq = math_ops.equal(x, y)
 333       condition = math_ops.reduce_all(eq)
 334       if not condition:
 335         # Prepare a message with first elements of x and y.
 336         summary_msg = ''
 337         # Default to printing 3 elements like control_flow_ops.Assert (used
 338         # by graph mode) does.
 339         summarize = 3 if summarize is None else summarize
 340         if summarize:
 341           # reshape((-1,)) is the fastest way to get a flat array view.
 342           x_np = x.numpy().reshape((-1,))
 343           y_np = y.numpy().reshape((-1,))
 344           x_sum = min(x_np.size, summarize)
 345           y_sum = min(y_np.size, summarize)
 346           summary_msg = ('First %d elements of x:\n%s\n'
 347                          'First %d elements of y:\n%s\n' %
 348                          (x_sum, x_np[:x_sum],
 349                           y_sum, y_np[:y_sum]))
 350 
 351         index_and_values_str = ''
 352         if x.shape == y.shape and x.shape.as_list():
 353           # If the shapes of x and y are the same (and not scalars),
 354           # Get the values that actually differed and their indices.
 355           # If shapes are different this information is more confusing
 356           # than useful.
 357           mask = math_ops.logical_not(eq)
 358           indices = array_ops.where(mask)
 359           indices_np = indices.numpy()
 360           x_vals = array_ops.boolean_mask(x, mask)
 361           y_vals = array_ops.boolean_mask(y, mask)
 362           summarize = min(summarize, indices_np.shape[0])
 363           index_and_values_str = (
 364               'Indices of first %s different values:\n%s\n'
 365               'Corresponding x values:\n%s\n'
 366               'Corresponding y values:\n%s\n' %
 367               (summarize, indices_np[:summarize],
 368                x_vals.numpy().reshape((-1,))[:summarize],
 369                y_vals.numpy().reshape((-1,))[:summarize]))
 370 
 371         raise errors.InvalidArgumentError(
 372             node_def=None, op=None,
 373             message=('%s\nCondition x == y did not hold.\n%s%s' %
 374                      (message or '', index_and_values_str, summary_msg)))
 375       return
 376 
 377     if data is None:
 378       data = [
 379           message,
 380           'Condition x == y did not hold element-wise:',
 381           'x (%s) = ' % x.name, x,
 382           'y (%s) = ' % y.name, y
 383       ]
 384     condition = math_ops.reduce_all(math_ops.equal(x, y))
 385     x_static = tensor_util.constant_value(x)
 386     y_static = tensor_util.constant_value(y)
 387     if x_static is not None and y_static is not None:
 388       condition_static = (x_static == y_static).all()
 389       _assert_static(condition_static, data)
 390     return control_flow_ops.Assert(condition, data, summarize=summarize)

 445 @tf_export('debugging.assert_near', 'assert_near')
 446 @deprecation.deprecated_endpoints('assert_near')
 447 def assert_near(
 448     x, y, rtol=None, atol=None, data=None, summarize=None, message=None,
 449     name=None):
 450   """Assert the condition `x` and `y` are close element-wise.
 451 
 452   Example of adding a dependency to an operation:
 453 
 454   ```python
 455   with tf.control_dependencies([tf.assert_near(x, y)]):
 456     output = tf.reduce_sum(x)
 457   ```
 458 
 459   This condition holds if for every pair of (possibly broadcast) elements
 460   `x[i]`, `y[i]`, we have
 461 
 462   ```tf.abs(x[i] - y[i]) <= atol + rtol * tf.abs(y[i])```.
 463 
 464   If both `x` and `y` are empty, this is trivially satisfied.
 465 
 466   The default `atol` and `rtol` is `10 * eps`, where `eps` is the smallest
 467   representable positive number such that `1 + eps != eps`.  This is about
 468   `1.2e-6` in `32bit`, `2.22e-15` in `64bit`, and `0.00977` in `16bit`.
 469   See `numpy.finfo`.
 470 
 471   Args:
 472     x:  Float or complex `Tensor`.
 473     y:  Float or complex `Tensor`, same `dtype` as, and broadcastable to, `x`.
 474     rtol:  `Tensor`.  Same `dtype` as, and broadcastable to, `x`.
 475       The relative tolerance.  Default is `10 * eps`.
 476     atol:  `Tensor`.  Same `dtype` as, and broadcastable to, `x`.
 477       The absolute tolerance.  Default is `10 * eps`.
 478     data:  The tensors to print out if the condition is False.  Defaults to
 479       error message and first few entries of `x`, `y`.
 480     summarize: Print this many entries of each tensor.
 481     message: A string to prefix to the default message.
 482     name: A name for this operation (optional).  Defaults to "assert_near".
 483 
 484   Returns:
 485     Op that raises `InvalidArgumentError` if `x` and `y` are not close enough.
 486 
 487   @compatibility(numpy)
 488   Similar to `numpy.assert_allclose`, except tolerance depends on data type.
 489   This is due to the fact that `TensorFlow` is often used with `32bit`, `64bit`,
 490   and even `16bit` data.
 491   @end_compatibility
 492   """
 493   message = message or ''
 494   with ops.name_scope(name, 'assert_near', [x, y, rtol, atol, data]):
 495     x = ops.convert_to_tensor(x, name='x')
 496     y = ops.convert_to_tensor(y, name='y', dtype=x.dtype)
 497 
 498     eps = np.finfo(x.dtype.as_numpy_dtype).eps
 499     rtol = 10 * eps if rtol is None else rtol
 500     atol = 10 * eps if atol is None else atol
 501 
 502     rtol = ops.convert_to_tensor(rtol, name='rtol', dtype=x.dtype)
 503     atol = ops.convert_to_tensor(atol, name='atol', dtype=x.dtype)
 504 
 505     if context.executing_eagerly():
 506       x_name = _shape_and_dtype_str(x)
 507       y_name = _shape_and_dtype_str(y)
 508     else:
 509       x_name = x.name
 510       y_name = y.name
 511 
 512     if data is None:
 513       data = [
 514           message,
 515           'x and y not equal to tolerance rtol = %s, atol = %s' % (rtol, atol),
 516           'x (%s) = ' % x_name, x, 'y (%s) = ' % y_name, y
 517       ]
 518     tol = atol + rtol * math_ops.abs(y)
 519     diff = math_ops.abs(x - y)
 520     condition = math_ops.reduce_all(math_ops.less(diff, tol))
 521     return control_flow_ops.Assert(condition, data, summarize=summarize)

 720 def _assert_rank_condition(
 721     x, rank, static_condition, dynamic_condition, data, summarize):
 722   """Assert `x` has a rank that satisfies a given condition.
 723 
 724   Args:
 725     x:  Numeric `Tensor`.
 726     rank:  Scalar `Tensor`.
 727     static_condition:   A python function that takes `[actual_rank, given_rank]`
 728       and returns `True` if the condition is satisfied, `False` otherwise.
 729     dynamic_condition:  An `op` that takes [actual_rank, given_rank]
 730       and return `True` if the condition is satisfied, `False` otherwise.
 731     data:  The tensors to print out if the condition is false.  Defaults to
 732       error message and first few entries of `x`.
 733     summarize: Print this many entries of each tensor.
 734 
 735   Returns:
 736     Op raising `InvalidArgumentError` if `x` fails dynamic_condition.
 737 
 738   Raises:
 739     ValueError:  If static checks determine `x` fails static_condition.
 740   """
 741   assert_type(rank, dtypes.int32)
 742 
 743   # Attempt to statically defined rank.
 744   rank_static = tensor_util.constant_value(rank)
 745   if rank_static is not None:
 746     if rank_static.ndim != 0:
 747       raise ValueError('Rank must be a scalar.')
 748 
 749     x_rank_static = x.get_shape().ndims
 750     if x_rank_static is not None:
 751       if not static_condition(x_rank_static, rank_static):
 752         raise ValueError(
 753             'Static rank condition failed', x_rank_static, rank_static)
 754       return control_flow_ops.no_op(name='static_checks_determined_all_ok')

 901   if len(given_ranks) < 1:
 902     return ops.convert_to_tensor(False)
 903   result = math_ops.equal(given_ranks[0], actual_rank)

1096 def _get_diff_for_monotonic_comparison(x):
1097   """Gets the difference x[1:] - x[:-1]."""
1098   x = array_ops.reshape(x, [-1])
1099   if not is_numeric_tensor(x):
1100     raise TypeError('Expected x to be numeric, instead found: %s' % x)
1101 
1102   # If x has less than 2 elements, there is nothing to compare.  So return [].
1103   is_shorter_than_two = math_ops.less(array_ops.size(x), 2)
1104   short_result = lambda: ops.convert_to_tensor([], dtype=x.dtype)
1105 
1106   # With 2 or more elements, return x[1:] - x[:-1]
1107   s_len = array_ops.shape(x) - 1
1108   diff = lambda: array_ops.strided_slice(x, [1], [1] + s_len)- array_ops.strided_slice(x, [0], s_len)
1109   return control_flow_ops.cond(is_shorter_than_two, short_result, diff)

1119 @deprecation.deprecated_endpoints('is_non_decreasing')
1120 def is_non_decreasing(x, name=None):
1121   """Returns `True` if `x` is non-decreasing.
1122 
1123   Elements of `x` are compared in row-major order.  The tensor `[x[0],...]`
1124   is non-decreasing if for every adjacent pair we have `x[i] <= x[i+1]`.
1125   If `x` has less than two elements, it is trivially non-decreasing.
1126 
1127   See also:  `is_strictly_increasing`
1128

1132

1135 
1136   Raises:
1137     TypeError: if `x` is not a numeric tensor.
1138   """
1139   with ops.name_scope(name, 'is_non_decreasing', [x]):
1140     diff = _get_diff_for_monotonic_comparison(x)
1141     # When len(x) = 1, diff = [], less_equal = [], and reduce_all([]) = True.
1142     zero = ops.convert_to_tensor(0, dtype=diff.dtype)
1143     return math_ops.reduce_all(math_ops.less_equal(zero, diff))

1152   is strictly increasing if for every adjacent pair we have `x[i] < x[i+1]`.
1153   If `x` has less than two elements, it is trivially strictly increasing.

1158     x: Numeric `Tensor`.
1159     name: A name for this operation (optional).
1160       Defaults to "is_strictly_increasing"

1164 
1165   Raises:
1166     TypeError: if `x` is not a numeric tensor.
1167   """
1168   with ops.name_scope(name, 'is_strictly_increasing', [x]):
1169     diff = _get_diff_for_monotonic_comparison(x)
1170     # When len(x) = 1, diff = [], less = [], and reduce_all([]) = True.
1171     zero = ops.convert_to_tensor(0, dtype=diff.dtype)
1172     return math_ops.reduce_all(math_ops.less(zero, diff))

1175 def _assert_same_base_type(items, expected_type=None):
1176   r"""Asserts all items are of the same base type.
1177 
1178   Args:
1179     items: List of graph items (e.g., `Variable`, `Tensor`, `SparseTensor`,
1180         `Operation`, or `IndexedSlices`). Can include `None` elements, which
1181         will be ignored.
1182     expected_type: Expected type. If not specified, assert all items are
1183         of the same base type.
1184 
1185   Returns:
1186     Validated type, or none if neither expected_type nor items provided.
1187 
1188   Raises:
1189     ValueError: If any types do not match.
1190   """
1191   original_expected_type = expected_type
1192   mismatch = False
1193   for item in items:
1194     if item is not None:
1195       item_type = item.dtype.base_dtype
1196       if not expected_type:
1197         expected_type = item_type
1198       elif expected_type != item_type:
1199         mismatch = True
1200         break
1201   if mismatch:
1202     # Loop back through and build up an informative error message (this is very
1203     # slow, so we don't do it unless we found an error above).
1204     expected_type = original_expected_type
1205     original_item_str = None
1206     for item in items:
1207       if item is not None:
1208         item_type = item.dtype.base_dtype
1209         if not expected_type:
1210           expected_type = item_type
1211           original_item_str = item.name if hasattr(item, 'name') else str(item)
1212         elif expected_type != item_type:
1213           raise ValueError('%s, type=%s, must be of the same type (%s)%s.' % (
1214               item.name if hasattr(item, 'name') else str(item),
1215               item_type, expected_type,
1216               (' as %s' % original_item_str) if original_item_str else ''))
1217     return expected_type  # Should be unreachable
1218   else:
1219     return expected_type

1268 @tf_export('ensure_shape')
1269 def ensure_shape(x, shape, name=None):
1270   """Updates the shape of a tensor and checks at runtime that the shape holds.
1271 
1272   For example:
1273   ```python
1274   x = tf.placeholder(tf.int32)
1275   print(x.shape)
1276   ==> TensorShape(None)
1277   y = x * 2
1278   print(y.shape)
1279   ==> TensorShape(None)
1280 
1281   y = tf.ensure_shape(y, (None, 3, 3))
1282   print(y.shape)
1283   ==> TensorShape([Dimension(None), Dimension(3), Dimension(3)])
1284 
1285   with tf.Session() as sess:
1286     # Raises tf.errors.InvalidArgumentError, because the shape (3,) is not
1287     # compatible with the shape (None, 3, 3)
1288     sess.run(y, feed_dict={x: [1, 2, 3]})
1289 
1290   ```
1291 
1292   NOTE: This differs from `Tensor.set_shape` in that it sets the static shape
1293   of the resulting tensor and enforces it at runtime, raising an error if the
1294   tensor's runtime shape is incompatible with the specified shape.
1295   `Tensor.set_shape` sets the static shape of the tensor without enforcing it
1296   at runtime, which may result in inconsistencies between the statically-known
1297   shape of tensors and the runtime value of tensors.
1298 
1299   Args:
1300     x: A `Tensor`.
1301     shape: A `TensorShape` representing the shape of this tensor, a
1302       `TensorShapeProto`, a list, a tuple, or None.
1303     name: A name for this operation (optional). Defaults to "EnsureShape".
1304 
1305   Returns:
1306     A `Tensor`. Has the same type and contents as `x`. At runtime, raises a
1307     `tf.errors.InvalidArgumentError` if `shape` is incompatible with the shape
1308     of `x`.
1309   """
1310   if not isinstance(shape, tensor_shape.TensorShape):
1311     shape = tensor_shape.TensorShape(shape)
1312 
1313   return array_ops.ensure_shape(x, shape, name=name)
