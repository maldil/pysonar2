  43   """Graph-mode implementation of TensorArray.

  46   def __init__(self,
  47                dtype,
  48                size=None,
  49                dynamic_size=None,
  50                clear_after_read=None,
  51                tensor_array_name=None,
  52                handle=None,
  53                flow=None,
  54                infer_shape=True,
  55                element_shape=None,
  56                colocate_with_first_write_call=True,
  57                name=None):
  58     """Constructs a graph mode TensorArray.
  59 
  60     Args:
  61       dtype: (required) data type of the TensorArray.
  62       size: (optional) int32 scalar `Tensor`: the size of the TensorArray.
  63         Required if handle is not provided.
  64       dynamic_size: (optional) Python bool: If true, writes to the TensorArray
  65         can grow the TensorArray past its initial size.  Default: False.
  66       clear_after_read: Boolean (optional, default: True).  If True, clear
  67         TensorArray values after reading them.  This disables read-many
  68         semantics, but allows early release of memory.
  69       tensor_array_name: (optional) Python string: the name of the TensorArray.
  70         This is used when creating the TensorArray handle.  If this value is
  71         set, handle should be None.
  72       handle: (optional) A `Tensor` handle to an existing TensorArray.  If this
  73         is set, tensor_array_name should be None. Only supported in graph mode.
  74       flow: (optional) A float `Tensor` scalar coming from an existing
  75         `TensorArray.flow`. Only supported in graph mode.
  76       infer_shape: (optional, default: True) If True, shape inference
  77         is enabled.  In this case, all elements must have the same shape.
  78       element_shape: (optional, default: None) A `TensorShape` object specifying
  79         the shape constraints of each of the elements of the TensorArray.
  80         Need not be fully defined.
  81       colocate_with_first_write_call: If `True`, the TensorArray will be
  82         colocated on the same device as the Tensor used on its first write
  83         (write operations include `write`, `unstack`, and `split`).  If `False`,
  84         the TensorArray will be placed on the device determined by the
  85         device context available during its initialization.
  86       name: A name for the operation (optional).
  87 
  88     Raises:
  89       ValueError: if both handle and tensor_array_name are provided.
  90       TypeError: if handle is provided but is not a Tensor.
  91     """
  92     if handle is not None and tensor_array_name:
  93       raise ValueError(
  94           "Cannot construct with both handle and tensor_array_name")
  95     if handle is not None and not isinstance(handle, ops.Tensor):
  96       raise TypeError("Handle must be a Tensor")
  97     if handle is None and size is None:
  98       raise ValueError("Size must be provided if handle is not provided")
  99     if handle is not None and size is not None:
 100       raise ValueError("Cannot provide both a handle and size "
 101                        "at the same time")
 102     if handle is not None and element_shape is not None:
 103       raise ValueError("Cannot provide both a handle and element_shape "
 104                        "at the same time")
 105     if handle is not None and dynamic_size is not None:
 106       raise ValueError("Cannot provide both a handle and dynamic_size "
 107                        "at the same time")
 108     if handle is not None and clear_after_read is not None:
 109       raise ValueError("Cannot provide both a handle and clear_after_read "
 110                        "at the same time")
 111 
 112     if clear_after_read is None:
 113       clear_after_read = True
 114     dynamic_size = dynamic_size or False
 115 
 116     self._dtype = dtype
 117 
 118     # Used to keep track of what tensors the TensorArray should be
 119     # colocated with.  We choose to colocate the TensorArray with the
 120     # first tensor written to it.
 121     self._colocate_with_first_write_call = colocate_with_first_write_call
 122     if colocate_with_first_write_call:
 123       self._colocate_with = []
 124     else:
 125       self._colocate_with = None
 126 
 127     # Record the current static shape for the array elements. The element
 128     # shape is defined either by `element_shape` or the shape of the tensor
 129     # of the first write. If `infer_shape` is true, all writes checks for
 130     # shape equality.
 131     if element_shape is None:
 132       self._infer_shape = infer_shape
 133       self._element_shape = []
 134     else:
 135       self._infer_shape = True
 136       self._element_shape = [tensor_shape.TensorShape(element_shape)]
 137     with ops.name_scope(name, "TensorArray", [handle, size, flow]) as scope:
 138       if handle is not None:
 139         self._handle = handle
 140         if flow is None:
 141           raise ValueError("flow must not be None if handle is not None.")
 142         self._flow = flow
 143       else:
 144         # Construct the TensorArray with an empty device.  The first
 145         # write into the TensorArray from a Tensor with a set device
 146         # will retroactively set the device value of this op.
 147         def create():
 148           """Create the TensorArray op."""
 149           return gen_data_flow_ops.tensor_array_v3(
 150               dtype=dtype,
 151               size=size,
 152               element_shape=element_shape,
 153               identical_element_shapes=infer_shape,
 154               dynamic_size=dynamic_size,
 155               clear_after_read=clear_after_read,
 156               tensor_array_name=tensor_array_name,
 157               name=scope)
 158         if colocate_with_first_write_call:
 159           with ops.device(None), ops.colocate_with(None, ignore_existing=True):
 160             self._handle, self._flow = create()
 161         else:
 162           self._handle, self._flow = create()
 163

 176   def _merge_element_shape(self, shape):
 177     """Changes the element shape of the array given a shape to merge with.
 178 
 179     Args:
 180       shape: A `TensorShape` object to merge with.
 181 
 182     Raises:
 183       ValueError: if the provided shape is incompatible with the current
 184           element shape of the `TensorArray`.
 185     """
 186 
 187     if self._element_shape:
 188       if not shape.is_compatible_with(self._element_shape[0]):
 189         raise ValueError(
 190             "Inconsistent shapes: saw %s but expected %s "
 191             "(and infer_shape=True)" % (shape, self._element_shape[0]))
 192       self._element_shape[0] = self._element_shape[0].merge_with(shape)
 193     else:
 194       self._element_shape.append(shape)
 195 
 196   @contextlib.contextmanager
 197   def _maybe_colocate_with(self, value):
 198     """Colocate operations with an internal colocation group or `value`.
 199 
 200     Args:
 201       value: `Tensor`, the tensor to try to colocate with.
 202 
 203     Yields:
 204       Does not yield anything, but the new context is a colocation context.
 205 
 206     If no internal colocation group is set, colocate with `value` and set
 207     the internal colocation group to be value.
 208     """
 209     if not self._colocate_with_first_write_call:
 210       yield
 211     else:
 212       if not self._colocate_with:
 213         self._colocate_with.append(value)
 214       with ops.colocate_with(self._colocate_with[0]):
 215         yield

 217   def identity(self):
 218     """See TensorArray."""
 219     flow = array_ops.identity(self._flow)
 220     ta = TensorArray(
 221         dtype=self._dtype, handle=self._handle, flow=flow,
 222         infer_shape=self._infer_shape,
 223         colocate_with_first_write_call=self._colocate_with_first_write_call)
 224     ta._element_shape = self._element_shape
 225     ta._colocate_with = self._colocate_with
 226     return ta

 228   def grad(self, source, flow=None, name=None):
 229     """See TensorArray."""
 230     # tensor_array_grad requires a flow input when forward
 231     # TensorArrays are dynamically sized.  This forces the creation
 232     # of the grad TensorArray only once the final forward array's size
 233     # is fixed.
 234     if flow is None:
 235       flow = self.flow
 236     with ops.name_scope(name, "TensorArrayGrad", [self._handle]):
 237       with ops.colocate_with(self._handle):
 238         g_handle, unused_flow = gen_data_flow_ops.tensor_array_grad_v3(
 239             handle=self._handle, source=source, flow_in=flow, name=name)
 240         with ops.control_dependencies([g_handle]):
 241           flow = array_ops.identity(flow, name="gradient_flow")
 242         g = TensorArray(
 243             dtype=self._dtype,
 244             handle=g_handle,
 245             flow=flow,
 246             infer_shape=self._infer_shape,
 247             colocate_with_first_write_call=False)
 248         g._element_shape = self._element_shape
 249         return g

 251   def read(self, index, name=None):
 252     """See TensorArray."""
 253     value = gen_data_flow_ops.tensor_array_read_v3(
 254         handle=self._handle,
 255         index=index,
 256         flow_in=self._flow,
 257         dtype=self._dtype,
 258         name=name)
 259     if self._element_shape:
 260       value.set_shape(self._element_shape[0].dims)
 261     return value

 263   @tf_should_use.should_use_result
 264   def write(self, index, value, name=None):
 265     """See TensorArray."""
 266     with ops.name_scope(name, "TensorArrayWrite", [self._handle, index, value]):
 267       value = ops.convert_to_tensor(value, name="value")
 268       if self._infer_shape:
 269         self._merge_element_shape(value.shape)
 270       with self._maybe_colocate_with(value):
 271         flow_out = gen_data_flow_ops.tensor_array_write_v3(
 272             handle=self._handle,
 273             index=index,
 274             value=value,
 275             flow_in=self._flow,
 276             name=name)
 277       ta = TensorArray(
 278           dtype=self._dtype, handle=self._handle, flow=flow_out,
 279           colocate_with_first_write_call=self._colocate_with_first_write_call)
 280       ta._infer_shape = self._infer_shape
 281       ta._element_shape = self._element_shape
 282       ta._colocate_with = self._colocate_with
 283       return ta

 285   def stack(self, name=None):
 286     """See TensorArray."""
 287     with ops.colocate_with(self._handle):
 288       with ops.name_scope(name, "TensorArrayStack", [self._handle]):
 289         return self.gather(math_ops.range(0, self.size()), name=name)
 290 
 291   def gather(self, indices, name=None):
 292     """See TensorArray."""
 293     if self._element_shape:
 294       element_shape = self._element_shape[0]
 295     else:
 296       element_shape = tensor_shape.TensorShape(None)
 297     value = gen_data_flow_ops.tensor_array_gather_v3(
 298         handle=self._handle,
 299         indices=indices,
 300         flow_in=self._flow,
 301         dtype=self._dtype,
 302         name=name,
 303         element_shape=element_shape)
 304     if self._element_shape and self._element_shape[0].dims is not None:
 305       value.set_shape([None] + self._element_shape[0].dims)
 306     return value
 307 
 308   def concat(self, name=None):
 309     """See TensorArray."""
 310     if self._element_shape and self._element_shape[0].dims is not None:
 311       element_shape_except0 = (
 312           tensor_shape.TensorShape(self._element_shape[0].dims[1:]))
 313     else:
 314       element_shape_except0 = tensor_shape.TensorShape(None)
 315     value, _ = gen_data_flow_ops.tensor_array_concat_v3(
 316         handle=self._handle,
 317         flow_in=self._flow,
 318         dtype=self._dtype,
 319         name=name,
 320         element_shape_except0=element_shape_except0)
 321     if self._element_shape and self._element_shape[0].dims is not None:
 322       value.set_shape([None] + self._element_shape[0].dims[1:])
 323     return value
 324 
 325   @tf_should_use.should_use_result
 326   def unstack(self, value, name=None):
 327     """See TensorArray."""
 328     with ops.name_scope(name, "TensorArrayUnstack", [self._handle, value]):
 329       num_elements = array_ops.shape(value)[0]
 330       return self.scatter(
 331           indices=math_ops.range(0, num_elements), value=value, name=name)
 332 
 333   @tf_should_use.should_use_result
 334   def scatter(self, indices, value, name=None):
 335     """See TensorArray."""
 336     with ops.name_scope(name, "TensorArrayScatter",
 337                         [self._handle, value, indices]):
 338       value = ops.convert_to_tensor(value, name="value")
 339       if self._infer_shape and not context.executing_eagerly():
 340         self._merge_element_shape(value.shape[1:])
 341       with self._maybe_colocate_with(value):
 342         flow_out = gen_data_flow_ops.tensor_array_scatter_v3(
 343             handle=self._handle,
 344             indices=indices,
 345             value=value,
 346             flow_in=self._flow,
 347             name=name)
 348       ta = TensorArray(
 349           dtype=self._dtype, handle=self._handle, flow=flow_out,
 350           colocate_with_first_write_call=self._colocate_with_first_write_call)
 351       ta._infer_shape = self._infer_shape
 352       ta._element_shape = self._element_shape
 353       ta._colocate_with = self._colocate_with
 354       return ta

 356   @tf_should_use.should_use_result
 357   def split(self, value, lengths, name=None):
 358     """See TensorArray."""
 359     with ops.name_scope(name, "TensorArraySplit",
 360                         [self._handle, value, lengths]):
 361       value = ops.convert_to_tensor(value, name="value")
 362       with self._maybe_colocate_with(value):
 363         lengths_64 = math_ops.to_int64(lengths)
 364         if self._infer_shape and not context.executing_eagerly():
 365           clengths = tensor_util.constant_value(lengths_64)
 366           if value.shape.dims is not None:
 367             if clengths is not None and clengths.max() == clengths.min():
 368               self._merge_element_shape(
 369                   tensor_shape.TensorShape([clengths[0]]).concatenate(
 370                       value.shape[1:]))
 371         flow_out = gen_data_flow_ops.tensor_array_split_v3(
 372             handle=self._handle,
 373             value=value,
 374             lengths=lengths_64,
 375             flow_in=self._flow,
 376             name=name)
 377       ta = TensorArray(
 378           dtype=self._dtype, handle=self._handle, flow=flow_out,
 379           colocate_with_first_write_call=self._colocate_with_first_write_call)
 380       ta._infer_shape = self._infer_shape
 381       ta._element_shape = self._element_shape
 382       ta._colocate_with = self._colocate_with
 383       return ta

 399 class _EagerTensorArray(object):
 400   """Eager-compatible implementation of TensorArray.
 401   """
 402 
 403   def __init__(self,
 404                dtype,
 405                size=None,
 406                dynamic_size=None,
 407                clear_after_read=None,
 408                tensor_array_name=None,
 409                handle=None,
 410                flow=None,
 411                infer_shape=True,
 412                element_shape=None,
 413                colocate_with_first_write_call=True,
 414                name=None):
 415     """Constructs a TensorArray compatible with eager execution.
 416 
 417     Args:
 418       dtype: (required) data type of the TensorArray.
 419       size: (optional) int32 scalar `Tensor`: the size of the TensorArray.
 420         Required if handle is not provided.
 421       dynamic_size: (optional) Python bool: If true, writes to the TensorArray
 422         can grow the TensorArray past its initial size.  Default: False.
 423       clear_after_read: Boolean (optional, default: True).  If True, clear
 424         TensorArray values after reading them.  This disables read-many
 425         semantics, but allows early release of memory.
 426       tensor_array_name: unused.
 427       handle: unsupported.
 428       flow: unsupported.
 429       infer_shape: used for error checking, same semantics as TensorArray.
 430       element_shape: used for error checking, same semantics as TensorArray.
 431       colocate_with_first_write_call: unsupported.
 432       name: unsupported.
 433 
 434     Raises:
 435       ValueError: handle or flow are supplied, or if size is not supplied.
 436     """
 437 
 438     del (flow, tensor_array_name, name)  # Unused.
 439 
 440     if handle is not None:
 441       raise ValueError("TensorArray handles are not supported when eager "
 442                        "execution is enabled.")
 443     if size is None:
 444       raise ValueError("Size must be declared for TensorArrays when eager "
 445                        "execution is enabled.")
 446 
 447     # These attributes are not meaningful when eager is enabled, but some
 448     # library functions (e.g., those in control_flow_ops.py) access them to
 449     # create new tensor arrays; as such, we define them for the sake of
 450     # compatibility.
 451     self._handle = None
 452     # we assign a dummy value to _flow in case other code assumes it to be
 453     # a Tensor
 454     self._flow = constant_op.constant(0, dtype=dtypes.int32)
 455     self._infer_shape = infer_shape
 456     self._element_shape = element_shape
 457     self._colocate_with_first_write_call = colocate_with_first_write_call
 458 
 459     self._dtype = dtype
 460     self._dynamic_size = dynamic_size or False
 461     self._clear_after_read = (
 462         True if clear_after_read is None else clear_after_read)
 463     self._previously_read_indices = []
 464 
 465     if isinstance(size, ops.EagerTensor):
 466       size = size.numpy()
 467     self._tensor_array = [None for _ in range(size)]
 468 
 469   @property
 470   def flow(self):
 471     """For compatibility; flows are not meaningful when eager is enabled."""
 472     return self._flow
 473 
 474   @property
 475   def dtype(self):
 476     return self._dtype
 477 
 478   @property
 479   def handle(self):
 480     """For compatibility; handles are not meaningful when eager is enabled."""
 481     return self._handle
 482 
 483   def identity(self):
 484     """See TensorArray."""
 485     return self.parent()
 486 
 487   def grad(self, source, flow=None, name=None):
 488     raise NotImplementedError(
 489         "TensorArray.grad is not supported when executing eagerly; eager's "
 490         "gradient implementation does not use/need this function to compute "
 491         "gradients of operations that use TensorArrays.")
 492 
 493   def read(self, index, name=None):
 494     """See TensorArray."""
 495     del name  # not meaningful when executing eagerly.
 496 
 497     if isinstance(index, ops.EagerTensor):
 498       index = index.numpy()
 499 
 500     if index < 0:
 501       raise errors_impl.OutOfRangeError(
 502           None, None,
 503           "Reading from negative indices (index %d) is not allowed." % index)
 504 
 505     if index >= len(self._tensor_array):
 506       raise errors_impl.OutOfRangeError(
 507           None, None, "Tried to read from index %d but array size is: %d" %
 508           (index, len(self._tensor_array)))
 509 
 510     tensor = self._tensor_array[index]
 511     if tensor is None:
 512       if index in self._previously_read_indices:
 513         raise errors_impl.InvalidArgumentError(
 514             None, None,
 515             "Could not read index %d twice because it was cleared after "
 516             "a previous read (perhaps try setting clear_after_read = false?)" %
 517             index)
 518       else:
 519         tensor = self._maybe_zero(index)
 520 
 521     if self._clear_after_read:
 522       self._tensor_array[index] = None
 523       self._previously_read_indices.append(index)
 524     return tensor
 525 
 526   def _write(self, index, value):
 527     """Writes `value` into index named by `index`.
 528 
 529     Args:
 530       index: 0-D.  int32 scalar with the index to write to.
 531       value: N-D.  Tensor of type `dtype`.  The `Tensor` to write to `index`.
 532 
 533     Raises:
 534       errors_impl.InvalidArgumentError: `value` dtype does not match dtype.
 535       errors_impl.OutOfRangeError: `index` is out of bounds.
 536       ValueError: shape of `value` is not consistent with inferred shape.
 537     """
 538 
 539     if isinstance(index, ops.EagerTensor):
 540       index = index.numpy()
 541 
 542     if index < 0:
 543       raise errors_impl.OutOfRangeError(
 544           None, None,
 545           "Writing to negative indices (index %d) is not allowed." % index)
 546 
 547     size = len(self._tensor_array)
 548     if index >= size:
 549       if not self._dynamic_size:
 550         raise errors_impl.OutOfRangeError(
 551             None, None,
 552             "Tried to write to index %d but array is not resizeable and size "
 553             "is: %d" % (index, size))
 554       self._tensor_array.extend([None for _ in range(index - size + 1)])
 555 
 556     if not isinstance(value, ops.EagerTensor):
 557       value = ops.convert_to_tensor(value)
 558 
 559     if self._infer_shape:
 560       if self._element_shape is None:
 561         self._element_shape = value.shape
 562       elif self._element_shape != value.shape:
 563         raise ValueError("Incompatible shape for value (%s), expected (%s)" %
 564                          (value.shape.as_list(), self._element_shape.as_list()))
 565 
 566     if self._dtype != value.dtype:
 567       raise errors_impl.InvalidArgumentError(
 568           None, None,
 569           "TensorArray dtype is %s but Op is trying to write dtype %s" %
 570           (self._dtype.name, value.dtype.name))
 571     self._tensor_array[index] = value
 572 
 573   def write(self, index, value, name=None):
 574     """See TensorArray."""
 575     del name  # not meaningful when executing eagerly.
 576     self._write(index, value)
 577     return self.parent()

 579   def _maybe_zero(self, ix):
 580     val = self._tensor_array[ix]
 581     if val is None:
 582       val = self._tensor_array[ix] = array_ops.zeros(
 583           shape=self._element_shape, dtype=self._dtype)
 584     return val
 585 
 586   def stack(self, name=None):
 587     """See TensorArray."""
 588     if self._tensor_array:
 589       for ix in range(len(self._tensor_array)):
 590         self._maybe_zero(ix)
 591     return array_ops.stack(self._tensor_array, name=name)
 592 
 593   def gather(self, indices, name=None):
 594     """See TensorArray."""
 595     del name  # not meaningful when executing eagerly.
 596     return array_ops.stack([self._maybe_zero(i) for i in indices.numpy()])
 597 
 598   def concat(self, name=None):
 599     """See TensorArray."""
 600     try:
 601       return array_ops.concat(
 602           [self._maybe_zero(ix) for ix in range(len(self._tensor_array))],
 603           0, name=name)
 604     except errors_impl.OpError:
 605       # Reproduce a subset of the error-handling for graph-mode TensorArrays.
 606       shapes = [t.shape for t in self._tensor_array]
 607       ndims = [s.ndims for s in shapes]
 608       if 0 in ndims:
 609         idx = ndims.index(0)
 610         raise errors_impl.InvalidArgumentError(
 611             None, None, "Concat saw a scalar shape at index %d but requires "
 612             "at least vectors." % idx)
 613       else:
 614         raise
 615 
 616   def unstack(self, value, name=None):
 617     """See TensorArray."""
 618     tensors = array_ops.unstack(value, name=name)
 619     if len(tensors) > len(self._tensor_array) and not self._dynamic_size:
 620       raise ValueError(
 621           "Cannot unstack %d tensors into a TensorArray of static size %d" %
 622           (len(tensors), len(self._tensor_array)))
 623     self._tensor_array = tensors
 624     return self.parent()
 625 
 626   def scatter(self, indices, value, name=None):
 627     """See TensorArray."""
 628     del name  # not meaningful when executing eagerly.
 629     for index, val in zip(indices.numpy(), array_ops.unstack(value)):
 630       self._write(index, val)  # pylint: disable=protected-access
 631     return self.parent()
 632 
 633   def split(self, value, lengths, name=None):
 634     """See TensorArray."""
 635     # error checking to match graph-mode errors
 636     value = ops.convert_to_tensor(value)
 637     lengths = ops.convert_to_tensor(lengths)
 638     sum_lengths = math_ops.reduce_sum(lengths)
 639     if lengths.shape.ndims != 1:
 640       raise errors_impl.InvalidArgumentError(
 641           None, None, "Expected lengths to be a vector, received shape: %s" %
 642           lengths.shape.as_list())
 643     elif value.shape.ndims == 0:
 644       raise errors_impl.InvalidArgumentError(
 645           None, None, "Expected value to be at least a vector, "
 646           "but received shape: %s" % value.shape.as_list())
 647     elif sum_lengths.numpy() != value.shape.as_list()[0]:
 648       raise errors_impl.InvalidArgumentError(
 649           None, None, "Expected sum of lengths to be equal to "
 650           "values.shape[0], but sum of lengths is %d and "
 651           "value's shape is: %s " % (sum_lengths.numpy(),
 652                                      value.shape.as_list()))
 653     elif not self._dynamic_size and lengths.shape[0] != len(self._tensor_array):
 654       raise errors_impl.InvalidArgumentError(
 655           None, None, "TensorArray's size is not equal to the size of "
 656           "lengths (%d vs. %d), and the TensorArray is not marked as "
 657           "dynamically resizeable" % (len(self._tensor_array),
 658                                       lengths.shape[0]))
 659     else:
 660       self._tensor_array = array_ops.split(value, lengths, name=name)
 661       return self.parent()
 662 
 663   def size(self, name=None):
 664     """See TensorArray."""
 665     del name  # not meaningful when executing eagerly.
 666     return constant_op.constant(len(self._tensor_array))
 667 
 668   def close(self, name=None):
 669     del name  # not meaningful when executing eagerly.
 670     del self._tensor_array[:]

 673 # TensorArray is designed to hide an underlying implementation object
 674 # and as such accesses many of that object's hidden fields.
 675 # pylint: disable=protected-access
 676 @tf_export("TensorArray")
 677 class TensorArray(object):
 678   """Class wrapping dynamic-sized, per-time-step, write-once Tensor arrays.
 679 
 680   This class is meant to be used with dynamic iteration primitives such as
 681   `while_loop` and `map_fn`.  It supports gradient back-propagation via special
 682   "flow" control flow dependencies.
 683   """
 684 
 685   def __init__(self,
 686                dtype,
 687                size=None,
 688                dynamic_size=None,
 689                clear_after_read=None,
 690                tensor_array_name=None,
 691                handle=None,
 692                flow=None,
 693                infer_shape=True,
 694                element_shape=None,
 695                colocate_with_first_write_call=True,
 696                name=None):
 697     """Construct a new TensorArray or wrap an existing TensorArray handle.
 698 
 699     A note about the parameter `name`:
 700 
 701     The name of the `TensorArray` (even if passed in) is uniquified: each time
 702     a new `TensorArray` is created at runtime it is assigned its own name for
 703     the duration of the run.  This avoids name collisions if a `TensorArray`
 704     is created within a `while_loop`.
 705 
 706     Args:
 707       dtype: (required) data type of the TensorArray.
 708       size: (optional) int32 scalar `Tensor`: the size of the TensorArray.
 709         Required if handle is not provided.
 710       dynamic_size: (optional) Python bool: If true, writes to the TensorArray
 711         can grow the TensorArray past its initial size.  Default: False.
 712       clear_after_read: Boolean (optional, default: True).  If True, clear
 713         TensorArray values after reading them.  This disables read-many
 714         semantics, but allows early release of memory.
 715       tensor_array_name: (optional) Python string: the name of the TensorArray.
 716         This is used when creating the TensorArray handle.  If this value is
 717         set, handle should be None.
 718       handle: (optional) A `Tensor` handle to an existing TensorArray.  If this
 719         is set, tensor_array_name should be None. Only supported in graph mode.
 720       flow: (optional) A float `Tensor` scalar coming from an existing
 721         `TensorArray.flow`. Only supported in graph mode.
 722       infer_shape: (optional, default: True) If True, shape inference
 723         is enabled.  In this case, all elements must have the same shape.
 724       element_shape: (optional, default: None) A `TensorShape` object specifying
 725         the shape constraints of each of the elements of the TensorArray.
 726         Need not be fully defined.
 727       colocate_with_first_write_call: If `True`, the TensorArray will be
 728         colocated on the same device as the Tensor used on its first write
 729         (write operations include `write`, `unstack`, and `split`).  If `False`,
 730         the TensorArray will be placed on the device determined by the
 731         device context available during its initialization.
 732       name: A name for the operation (optional).
 733 
 734     Raises:
 735       ValueError: if both handle and tensor_array_name are provided.
 736       TypeError: if handle is provided but is not a Tensor.
 737     """
 738     if context.executing_eagerly():
 739       implementation = _EagerTensorArray
 740     else:
 741       implementation = _GraphTensorArray
 742 
 743     self._implementation = implementation(
 744         dtype,
 745         size=size,
 746         dynamic_size=dynamic_size,
 747         clear_after_read=clear_after_read,
 748         tensor_array_name=tensor_array_name,
 749         handle=handle,
 750         flow=flow,
 751         infer_shape=infer_shape,
 752         element_shape=element_shape,
 753         colocate_with_first_write_call=colocate_with_first_write_call,
 754         name=name)
 755 
 756     self._implementation.parent = weakref.ref(self)
