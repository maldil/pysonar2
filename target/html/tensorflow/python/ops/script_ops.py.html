  49 class EagerFunc(object):
  50   """A wrapper for a function owned by an EagerPyFunc."""
  51 
  52   def __init__(self, func, Tout, is_grad_func):
  53     """Constructs an EagerFunc.
  54 
  55     Args:
  56       func: The function to wrap.
  57       Tout: A list of datatypes for the output; an empty list if the output is
  58             None.
  59       is_grad_func: Whether this EagerFunc is the gradient of another
  60         EagerPyFunc.
  61     """
  62     self._func = func
  63     self._out_dtypes = Tout
  64     self._is_grad_func = is_grad_func
  65 
  66   def _convert(self, value, dtype):
  67     """Converts `value` to a tensor of type `dtype`, with error checking.
  68 
  69     Args:
  70       value: The tensor to convert.
  71       dtype: The desired dtype.
  72 
  73     Returns:
  74       A tensor of type `dtype`, or a zeros tensor if value is None and
  75       this function is in fact a grdient function.
  76 
  77     Raises:
  78       RuntimeError: if `value` is a variable.
  79     """
  80 
  81     if isinstance(value, resource_variable_ops.ResourceVariable):
  82       raise RuntimeError(
  83           "Attempting to return a variable from an eagerly executed py_func. "
  84           "Only numeric data structures like Tensors or NumPy arrays should "
  85           "be returned; to return the value of a variable, make sure to obtain "
  86           "the Tensor backing it by calling `.read_value()` on the variable in "
  87           "question: %s" % value)
  88     if value is None and self._is_grad_func:
  89       # Gradient functions may legitimately return a list that contains
  90       # both Tensors and Python Nones. Unfortuantely this breaks the
  91       # OpKernel, so for now we replace None objects with zeros, which is
  92       # mathematically correct but will prevent short-circuiting gradient
  93       # computations.
  94       #
  95       # TODO(akshayka): Make it possible to return a list of both Tensors and
  96       # Nones from an EagerPyFunc.
  97       return constant_op.constant(0.0, dtype=dtype)
  98     return ops.convert_to_tensor(value, dtype=dtype)
  99 
 100   def __call__(self, device, token, args):
 101     """Passes `args` to `self._func`, which is executed eagerly."""
 102 
 103     with context.eager_mode(), backprop.GradientTape() as tape:
 104       for tensor in args:
 105         tape.watch(tensor)
 106       ret = self._func(*args)
 107       # Use tf.identity to copy the returned tensors to device if neccesary.
 108       with ops.device(device):
 109         if isinstance(ret, (tuple, list)):
 110           outputs = [
 111               array_ops.identity(self._convert(x, dtype=dtype))
 112               for (x, dtype) in zip(ret, self._out_dtypes)
 113           ]
 114         elif ret is None:
 115           outputs = None
 116         else:
 117           outputs = array_ops.identity(
 118               self._convert(ret, dtype=self._out_dtypes[0]))
 119     tape_cache[compat.as_bytes(token)] = (tape, args, outputs)
 120     return outputs

 123 class FuncRegistry(object):
 124   """A helper class to keep track of registered py functions.
 125 
 126   FuncRegistry keeps a map from unique tokens (string) to python
 127   functions, which takes numpy arrays and outputs numpy arrays.
 128   """
 129 
 130   def __init__(self):
 131     self._lock = threading.Lock()
 132     self._unique_id = 0  # GUARDED_BY(self._lock)
 133     # Only store weakrefs to the functions. The strong reference is stored in
 134     # the graph.
 135     self._funcs = weakref.WeakValueDictionary()
 136 
 137   def insert(self, func):
 138     """Registers `func` and returns a unique token for this entry."""
 139     token = self._next_unique_token()
 140     # Store a weakref to the function
 141     self._funcs[token] = func
 142     return token
 143 
 144   def remove(self, token):
 145     """Removes the registered function corresponding to `token`."""
 146     self._funcs.pop(token, None)
 147 
 148   @staticmethod
 149   def _convert(value, dtype=None):
 150     """Converts an arg to numpy, avoiding dangerous string and unicode dtypes.
 151 
 152     Numpy pads with zeros when using string and unicode dtypes if different
 153     components of a tensor have different lengths.  This is bad: ignoring the
 154     padding is wrong for text data, and removing the padding is wrong for binary
 155     data.  To avoid this bug, we redo the conversion using an object dtype.
 156     Additionally, we convert unicode strings to (byte-)strings for
 157     compatibility.
 158 
 159     Args:
 160       value: Value to convert to a numpy array.
 161       dtype: (Optional.) Desired NumPy type for the returned value.
 162 
 163     Returns:
 164       A numpy array.
 165     """
 166     result = np.asarray(value, dtype=dtype, order="C")
 167     if result.dtype.char == "S" and result is not value:
 168       return np.asarray(value, order="C", dtype=object)
 169     elif result.dtype.char == "U" and result is not value:
 170       value = np.vectorize(lambda x: x.encode("utf8"))(value)
 171       return np.asarray(value, order="C", dtype=object)
 172     elif result.dtype.char == "U":
 173       return result.astype(np.bytes_)
 174     else:
 175       return result
 176 
 177   def __call__(self, token, device, args):
 178     """Calls the registered function for `token` with args.
 179 
 180     Args:
 181       token: A key into this `FuncRegistry` identifying which function to call.
 182       device: Name of the device on which outputs of `token`'s corresponding
 183         operation should be placed. Used iff the function registered for `token`
 184         is an EagerPyFunc.
 185       args: The arguments to pass to the function registered for `token`.
 186 
 187     Returns:
 188       The output of the function registered for `token`.
 189 
 190     Raises:
 191       ValueError: if no function is registered for `token`.
 192     """
 193     func = self._funcs.get(token, None)
 194     if func is None:
 195       raise ValueError("callback %s is not found" % token)
 196     if isinstance(func, EagerFunc):
 197       # NB: Different invocations of the same py_func will share the same
 198       # token, and the entries they stash in the tape_cache will collide.
 199       # In practice, when executing a graph, this should only happen if
 200       # the py_func is in a while_loop whose iterations are run in parallel
 201       # or if the graph is being driven by concurrent session.run() calls.
 202       #
 203       # TODO(akshayka): Key the tape cache in a thread-safe way.
 204       return func(device, token, args)
 205     else:
 206       ret = func(*args)
 207       # Strings seem to lead to a memory leak here if they're not wrapped in a
 208       # list.
 209       if isinstance(ret, six.binary_type):
 210         ret = [ret]
 211       # Ensures that we return either a single numpy array or a list of numpy
 212       # arrays.
 213       if isinstance(ret, (tuple, list)):
 214         return [self._convert(x) for x in ret]
 215       else:
 216         return self._convert(ret)
 217 
 218   def size(self):
 219     """Returns how many functions are currently registered."""
 220     return len(self._funcs)
 221 
 222   def _next_unique_token(self):
 223     """Returns a unique token."""
 224     with self._lock:
 225       uid = self._unique_id
 226       self._unique_id += 1
 227     return "pyfunc_%d" % uid

 235 def _internal_py_func(func,
 236                       inp,
 237                       Tout,
 238                       stateful=None,
 239                       eager=False,
 240                       is_grad_func=False,
 241                       name=None):
 242   """See documentation for py_func and eager_py_func."""
 243 
 244   is_list_or_tuple = False
 245   if isinstance(Tout, (list, tuple)):
 246     is_list_or_tuple = True
 247   else:
 248     Tout = [Tout]
 249 
 250   if eager:
 251     func = EagerFunc(func, Tout, is_grad_func)
 252 
 253   token = _py_funcs.insert(func)
 254   # We tie the registered function's lifetime with the current default graph,
 255   # i.e., when the current graph is destroyed, we remove its py funcs.
 256   graph = ops.get_default_graph()
 257 
 258   # pylint: disable=protected-access
 259   while isinstance(graph, function._FuncGraph):
 260     # If the py_func was declared inside a _FuncGraph, its lifetime should be
 261     # bound to that of the outer graph instead.
 262     graph = graph._outer_graph
 263 
 264   # TODO(zhifengc): Consider adding a Graph method to collect
 265   # `cleanup` objects in one of its member.
 266   if not hasattr(graph, "_py_funcs_used_in_graph"):
 267     graph._py_funcs_used_in_graph = []
 268 
 269   # Store a reference to the function in the graph to ensure it stays alive
 270   # as long as the graph lives. When the graph is destroyed, the function
 271   # is left to the garbage collector for destruction as well.
 272   graph._py_funcs_used_in_graph.append(func)
 273   # pylint: enable=protected-access
 274 
 275   if eager:
 276     result = gen_script_ops.eager_py_func(
 277         input=inp, token=token, Tout=Tout, name=name)
 278   else:
 279     if stateful:
 280       result = gen_script_ops.py_func(
 281           input=inp, token=token, Tout=Tout, name=name)
 282     else:
 283       result = gen_script_ops.py_func_stateless(
 284           input=inp, token=token, Tout=Tout, name=name)
 285   return result if is_list_or_tuple else result[0]

 289 @ops.RegisterGradient("EagerPyFunc")
 290 def _EagerPyFuncGrad(op, *dy):
 291   """Computes the gradient of an EagerPyFunc."""
 292 
 293   token = op.get_attr("token")
 294 
 295   def eagerly_executed_grad(*dy):
 296     tape, eager_inputs, eager_outputs = tape_cache.pop(compat.as_bytes(token))
 297     return tape.gradient(eager_outputs, eager_inputs, output_gradients=dy)
 298 
 299   with ops.control_dependencies(op.outputs):
 300     return _internal_py_func(
 301         func=eagerly_executed_grad,
 302         inp=dy,
 303         Tout=[tensor.dtype for tensor in op.inputs],
 304         eager=True,
 305         is_grad_func=True)

 308 def eager_py_func(func, inp, Tout, name=None):
 309   """Wraps a python function into a TensorFlow op that executes it eagerly.
 310 
 311   This function allows expressing computations in a TensorFlow graph as
 312   Python functions. In particular, it wraps a Python function `func`
 313   in a once-differentiable TensorFlow operation that executes it with eager
 314   exeuction enabled. As a consequence, `tf.contrib.eager.py_func` makes it
 315   possible to express control flow using Python constructs (`if`, `while`,
 316   `for`, etc.), instead of TensorFlow control flow constructs (`tf.cond`,
 317   `tf.while_loop`). For example, you might use `tf.contrib.eager.py_func` to
 318   implement the log huber function:
 319 
 320   ```python
 321   def log_huber(x, m):
 322     if tf.abs(x) <= m:
 323       return x**2
 324     else:
 325       return m**2 * (1 - 2 * tf.log(m) + tf.log(x**2))
 326 
 327   x = tf.placeholder(tf.float32)
 328   m = tf.placeholder(tf.float32)
 329 
 330   y = tf.contrib.eager.py_func(func=log_huber, inp=[x, m], Tout=tf.float32)
 331   dy_dx = tf.gradients(y, x)[0]
 332 
 333   with tf.Session() as sess:
 334     # The session executes `log_huber` eagerly. Given the feed values below,
 335     # it will take the first branch, so `y` evaluates to 1.0 and
 336     # `dy_dx` evaluates to 2.0.
 337     y, dy_dx = sess.run([y, dy_dx], feed_dict={x: 1.0, m: 2.0})
 338   ```
 339 
 340   You can also use `tf.contrib.eager.py_func` to debug your models at runtime
 341   using Python tools, i.e., you can isolate portions of your code that
 342   you want to debug, wrap them in Python functions and insert `pdb` tracepoints
 343   or print statements as desired, and wrap those functions in
 344   `tf.contrib.eager.py_func`.
 345 
 346   For more information on eager execution, see the
 347   [Eager guide](https://tensorflow.org/guide/eager).
 348 
 349   `tf.contrib.eager.py_func` is similar in spirit to `tf.py_func`, but unlike
 350   the latter, the former lets you use TensorFlow operations in the wrapped
 351   Python function. In particular, while `tf.py_func` only runs on CPUs and
 352   wraps functions that take NumPy arrays as inputs and return NumPy arrays as
 353   outputs, `tf.contrib.eager.py_func` can be placed on GPUs and wraps functions
 354   that take Tensors as inputs, execute TensorFlow operations in their bodies,
 355   and return Tensors as outputs.
 356 
 357   Like `tf.py_func`, `tf.contrib.eager.py_func` has the following limitations
 358   with respect to serialization and distribution:
 359 
 360   * The body of the function (i.e. `func`) will not be serialized in a
 361     `GraphDef`. Therefore, you should not use this function if you need to
 362     serialize your model and restore it in a different environment.
 363 
 364   * The operation must run in the same address space as the Python program
 365     that calls `tf.contrib.eager.py_func()`. If you are using distributed
 366     TensorFlow, you must run a `tf.train.Server` in the same process as the
 367     program that calls `tf.contrib.eager.py_func()` and you must pin the created
 368     operation to a device in that server (e.g. using `with tf.device():`).
 369 
 370 
 371   Args:
 372     func: A Python function which accepts a list of `Tensor` objects
 373       having element types that match the corresponding `tf.Tensor` objects
 374       in `inp` and returns a list of `Tensor` objects (or a single
 375       `Tensor`, or `None`) having element types that match the
 376       corresponding values in `Tout`.
 377     inp: A list of `Tensor` objects.
 378     Tout: A list or tuple of tensorflow data types or a single tensorflow data
 379       type if there is only one, indicating what `func` returns; an empty list
 380       if no value is returned (i.e., if the return value is `None`).
 381     name: A name for the operation (optional).
 382 
 383   Returns:
 384     A list of `Tensor` or a single `Tensor` which `func` computes; an empty list
 385     if `func` returns None.
 386   """
 387   return _internal_py_func(func=func, inp=inp, Tout=Tout, eager=True, name=name)
 388

 390 @tf_export("py_func")
 391 def py_func(func, inp, Tout, stateful=True, name=None):
 392   """Wraps a python function and uses it as a TensorFlow op.
 393 
 394   Given a python function `func`, which takes numpy arrays as its
 395   arguments and returns numpy arrays as its outputs, wrap this function as an
 396   operation in a TensorFlow graph. The following snippet constructs a simple
 397   TensorFlow graph that invokes the `np.sinh()` NumPy function as a operation
 398   in the graph:
 399 
 400   ```python
 401   def my_func(x):
 402     # x will be a numpy array with the contents of the placeholder below
 403     return np.sinh(x)
 404   input = tf.placeholder(tf.float32)
 405   y = tf.py_func(my_func, [input], tf.float32)
 406   ```
 407 
 408   **N.B.** The `tf.py_func()` operation has the following known limitations:
 409 
 410   * The body of the function (i.e. `func`) will not be serialized in a
 411     `GraphDef`. Therefore, you should not use this function if you need to
 412     serialize your model and restore it in a different environment.
 413 
 414   * The operation must run in the same address space as the Python program
 415     that calls `tf.py_func()`. If you are using distributed TensorFlow, you
 416     must run a `tf.train.Server` in the same process as the program that calls
 417     `tf.py_func()` and you must pin the created operation to a device in that
 418     server (e.g. using `with tf.device():`).
 419 
 420   Args:
 421     func: A Python function, which accepts `ndarray` objects as arguments and
 422       returns a list of `ndarray` objects (or a single `ndarray`). This function
 423       must accept as many arguments as there are tensors in `inp`, and these
 424       argument types will match the corresponding `tf.Tensor` objects
 425       in `inp`. The returns `ndarray`s must match the number and types defined
 426       `Tout`.
 427       Important Note: Input and output numpy `ndarray`s of `func` are not
 428       guaranteed to be copies. In some cases their underlying memory will be
 429       shared with the corresponding TensorFlow tensors.
 430       In-place modification or storing `func` input or return values in
 431       python datastructures without explicit (np.)copy
 432       can have non-deterministic consequences.
 433     inp: A list of `Tensor` objects.
 434     Tout: A list or tuple of tensorflow data types or a single tensorflow data
 435       type if there is only one, indicating what `func` returns.
 436     stateful: (Boolean.) If True, the function should be considered stateful.
 437       If a function is stateless, when given the same input it will return the
 438       same output and have no observable side effects. Optimizations such as
 439       common subexpression elimination are only performed on stateless
 440       operations.
 441     name: A name for the operation (optional).
 442 
 443   Returns:
 444     A list of `Tensor` or a single `Tensor` which `func` computes.
 445   """
 446   if context.executing_eagerly():
 447     result = func(*[x.numpy() for x in inp])
 448     result = nest.flatten(result)
 449 
 450     result = [x if x is None else ops.convert_to_tensor(x) for x in result]
 451     if len(result) == 1:
 452       # Mimic the automatic unwrapping in graph-mode py_func
 453       result, = result
 454     return result
 455 
 456   return _internal_py_func(
 457       func=func, inp=inp, Tout=Tout, stateful=stateful, eager=False, name=name)
 458
