  57 @tf_export('train.Scaffold')
  58 class Scaffold(object):
  59   """Structure to create or gather pieces commonly needed to train a model.
  60 
  61   When you build a model for training you usually need ops to initialize
  62   variables, a `Saver` to checkpoint them, an op to collect summaries for
  63   the visualizer, and so on.
  64 
  65   Various libraries built on top of the core TensorFlow library take care of
  66   creating some or all of these pieces and storing them in well known
  67   collections in the graph.  The `Scaffold` class helps pick these pieces from
  68   the graph collections, creating and adding them to the collections if needed.
  69 
  70   If you call the scaffold constructor without any arguments, it will pick
  71   pieces from the collections, creating default ones if needed when
  72   `scaffold.finalize()` is called.  You can pass arguments to the constructor to
  73   provide your own pieces.  Pieces that you pass to the constructor are not
  74   added to the graph collections.
  75 
  76   The following pieces are directly accessible as attributes of the `Scaffold`
  77   object:
  78 
  79   * `saver`: A `tf.train.Saver` object taking care of saving the variables.
  80     Picked from and stored into the `SAVERS` collection in the graph by default.
  81   * `init_op`: An op to run to initialize the variables.  Picked from and
  82     stored into the `INIT_OP` collection in the graph by default.
  83   * `ready_op`: An op to verify that the variables are initialized.  Picked
  84     from and stored into the `READY_OP` collection in the graph by default.
  85   * `ready_for_local_init_op`: An op to verify that global state has been
  86     initialized and it is alright to run `local_init_op`.  Picked from and
  87     stored into the `READY_FOR_LOCAL_INIT_OP` collection in the graph by
  88     default. This is needed when the initialization of local variables depends
  89     on the values of global variables.
  90   * `local_init_op`: An op to initialize the local variables.  Picked
  91     from and stored into the `LOCAL_INIT_OP` collection in the graph by default.
  92   * `summary_op`: An op to run and merge the summaries in the graph.  Picked
  93     from and stored into the `SUMMARY_OP` collection in the graph by default.
  94   * `global_step`: A tensor containing the global step counter.  Picked
  95     from and stored into the `GLOBAL_STEP` collection in the graph by default.
  96 
  97   You can also pass the following additional pieces to the constructor:
  98 
  99   * `init_feed_dict`: A session feed dictionary that should be used when
 100      running the init op.
 101   * `init_fn`: A callable to run after the init op to perform additional
 102     initializations.  The callable will be called as
 103     `init_fn(scaffold, session)`.
 104 
 105   """
 106 
 107   def __init__(self,
 108                init_op=None,
 109                init_feed_dict=None,
 110                init_fn=None,
 111                ready_op=None,
 112                ready_for_local_init_op=None,
 113                local_init_op=None,
 114                summary_op=None,
 115                saver=None,
 116                copy_from_scaffold=None):
 117     """Create a scaffold.
 118 
 119     Args:
 120       init_op: Optional op for initializing variables.
 121       init_feed_dict: Optional session feed dictionary to use when running the
 122         init_op.
 123       init_fn: Optional function to use to initialize the model after running
 124         the init_op.  Will be called as `init_fn(scaffold, session)`.
 125       ready_op: Optional op to verify that the variables are initialized.  Must
 126         return an empty 1D string tensor when the variables are initialized, or
 127         a non-empty 1D string tensor listing the names of the non-initialized
 128         variables.
 129       ready_for_local_init_op: Optional op to verify that the global variables
 130         are initialized and `local_init_op` can be run. Must return an empty
 131         1D string tensor when the global variables are initialized, or a
 132         non-empty 1D string tensor listing the names of the non-initialized
 133         global variables.
 134       local_init_op: Optional op to initialize local variables.
 135       summary_op: Optional op to gather all summaries.  Must return a scalar
 136         string tensor containing a serialized `Summary` proto.
 137       saver: Optional `tf.train.Saver` object to use to save and restore
 138         variables.
 139       copy_from_scaffold: Optional scaffold object to copy fields from. Its
 140         fields will be overwritten by the provided fields in this function.
 141     """
 142     if copy_from_scaffold is not None:
 143       if not isinstance(copy_from_scaffold, Scaffold):
 144         raise TypeError('copy_from_scaffold is not a Scaffold instance.')
 145       # We need _coalesce since Tensor is not converted to bool automatically,
 146       # so the common idiom of (a or b) does not work.
 147       coalesce = lambda a, b: a if a is not None else b
 148       init_op = coalesce(init_op, copy_from_scaffold.init_op)
 149       init_feed_dict = coalesce(init_feed_dict,
 150                                 copy_from_scaffold.init_feed_dict)
 151       # Use the original init_fn provided by the user to init the new Scaffold.
 152       init_fn = coalesce(init_fn, copy_from_scaffold._user_init_fn)  # pylint: disable=protected-access
 153       ready_op = coalesce(ready_op, copy_from_scaffold.ready_op)
 154       ready_for_local_init_op = coalesce(
 155           ready_for_local_init_op, copy_from_scaffold.ready_for_local_init_op)
 156       local_init_op = coalesce(local_init_op, copy_from_scaffold.local_init_op)
 157       summary_op = coalesce(summary_op, copy_from_scaffold.summary_op)
 158       saver = coalesce(saver, copy_from_scaffold.saver)
 159 
 160     # NOTE(touts): modifying the init function to be passed the scaffold is a
 161     # hack to make it easy to find the saver.  Is there a better way?
 162     self._user_init_fn = init_fn
 163     if init_fn:
 164       self._init_fn = lambda sess: init_fn(self, sess)
 165     else:
 166       self._init_fn = None
 167 
 168     self._init_op = init_op
 169     self._init_feed_dict = init_feed_dict
 170     self._ready_op = ready_op
 171     self._ready_for_local_init_op = ready_for_local_init_op
 172     self._local_init_op = local_init_op
 173     self._summary_op = summary_op
 174     self._saver = saver
 175 
 176   def finalize(self):
 177     """Creates operations if needed and finalizes the graph."""
 178     if self._init_op is None:
 179       def default_init_op():
 180         return control_flow_ops.group(
 181             variables.global_variables_initializer(),
 182             resources.initialize_resources(resources.shared_resources()))
 183       self._init_op = Scaffold.get_or_default(
 184           'init_op',
 185           ops.GraphKeys.INIT_OP,
 186           default_init_op)
 187     if self._ready_op is None:
 188       def default_ready_op():
 189         return array_ops.concat([
 190             variables.report_uninitialized_variables(),
 191             resources.report_uninitialized_resources()
 192         ], 0)
 193       self._ready_op = Scaffold.get_or_default(
 194           'ready_op', ops.GraphKeys.READY_OP,
 195           default_ready_op)
 196     if self._ready_for_local_init_op is None:
 197       def default_ready_for_local_init_op():
 198         return variables.report_uninitialized_variables(
 199             variables.global_variables())
 200       self._ready_for_local_init_op = Scaffold.get_or_default(
 201           'ready_for_local_init_op', ops.GraphKeys.READY_FOR_LOCAL_INIT_OP,
 202           default_ready_for_local_init_op)
 203     if self._local_init_op is None:
 204       self._local_init_op = Scaffold.get_or_default(
 205           'local_init_op', ops.GraphKeys.LOCAL_INIT_OP,
 206           Scaffold.default_local_init_op)
 207     if self._summary_op is None:
 208       self._summary_op = Scaffold.get_or_default('summary_op',
 209                                                  ops.GraphKeys.SUMMARY_OP,
 210                                                  summary.merge_all)
 211     # pylint: disable=g-long-lambda
 212     if self._saver is None:
 213       self._saver = training_saver._get_saver_or_default()  # pylint: disable=protected-access
 214     # pylint: enable=g-long-lambda
 215     self._saver.build()
 216 
 217     ops.get_default_graph().finalize()
 218     logging.info('Graph was finalized.')
 219     return self

 253   @staticmethod
 254   def get_or_default(arg_name, collection_key, default_constructor):
 255     """Get from cache or create a default operation."""
 256     elements = ops.get_collection(collection_key)
 257     if elements:
 258       if len(elements) > 1:
 259         raise RuntimeError('More than one item in the collection "%s". '
 260                            'Please indicate which one to use by passing it to '
 261                            'the tf.Scaffold constructor as:  '
 262                            'tf.Scaffold(%s=item to use)', collection_key,
 263                            arg_name)
 264       return elements[0]
 265     op = default_constructor()
 266     if op is not None:
 267       ops.add_to_collection(collection_key, op)
 268     return op

 270   @staticmethod
 271   def default_local_init_op():
 272     """Returns an op that groups the default local init ops.
 273 
 274     This op is used during session initialization when a Scaffold is
 275     initialized without specifying the local_init_op arg. It includes
 276     `tf.local_variables_initializer`, `tf.tables_initializer`, and also
 277     initializes local session resources.

 279     Returns:
 280       The default Scaffold local init op.

 282     return control_flow_ops.group(
 283         variables.local_variables_initializer(),
 284         lookup_ops.tables_initializer(),
 285         resources.initialize_resources(resources.local_resources()))

 288 def _create_monitored_session_with_worker_context(worker_context,  # pylint: disable=missing-docstring
 289                                                   scaffold,
 290                                                   checkpoint_dir=None,
 291                                                   hooks=None,
 292                                                   chief_only_hooks=None,
 293                                                   save_checkpoint_secs=None,
 294                                                   save_summaries_steps=None,
 295                                                   save_summaries_secs=None,
 296                                                   config=None,
 297                                                   stop_grace_period_secs=120,
 298                                                   log_step_count_steps=100,
 299                                                   max_wait_secs=7200,
 300                                                   save_checkpoint_steps=None,
 301                                                   summary_dir=None):
 302   all_hooks = []
 303   if hooks:
 304     all_hooks.extend(hooks)
 305   if chief_only_hooks and worker_context.is_chief:
 306     all_hooks.extend(chief_only_hooks)
 307 
 308   summary_dir = summary_dir or checkpoint_dir
 309   if summary_dir and worker_context.should_save_summary:
 310     if log_step_count_steps and log_step_count_steps > 0:
 311       all_hooks.append(
 312           basic_session_run_hooks.StepCounterHook(
 313               output_dir=summary_dir, every_n_steps=log_step_count_steps))
 314 
 315     if (save_summaries_steps and save_summaries_steps > 0) or (
 316         save_summaries_secs and save_summaries_secs > 0):
 317       all_hooks.append(
 318           basic_session_run_hooks.SummarySaverHook(
 319               scaffold=scaffold,
 320               save_steps=save_summaries_steps,
 321               save_secs=save_summaries_secs,
 322               output_dir=summary_dir))
 323 
 324   if checkpoint_dir and worker_context.should_checkpoint:
 325     if (save_checkpoint_secs and save_checkpoint_secs > 0) or (
 326         save_checkpoint_steps and save_checkpoint_steps > 0):
 327       all_hooks.append(
 328           basic_session_run_hooks.CheckpointSaverHook(
 329               checkpoint_dir,
 330               save_steps=save_checkpoint_steps,
 331               save_secs=save_checkpoint_secs,
 332               scaffold=scaffold))
 333 
 334   session_creator = worker_context.session_creator(
 335       scaffold,
 336       config=config,
 337       checkpoint_dir=checkpoint_dir,
 338       max_wait_secs=max_wait_secs)
 339   return MonitoredSession(
 340       session_creator=session_creator,
 341       hooks=all_hooks,
 342       stop_grace_period_secs=stop_grace_period_secs)

 345 @tf_export('train.MonitoredTrainingSession')
 346 def MonitoredTrainingSession(master='',  # pylint: disable=invalid-name
 347                              is_chief=True,
 348                              checkpoint_dir=None,
 349                              scaffold=None,
 350                              hooks=None,
 351                              chief_only_hooks=None,
 352                              save_checkpoint_secs=USE_DEFAULT,
 353                              save_summaries_steps=USE_DEFAULT,
 354                              save_summaries_secs=USE_DEFAULT,
 355                              config=None,
 356                              stop_grace_period_secs=120,
 357                              log_step_count_steps=100,
 358                              max_wait_secs=7200,
 359                              save_checkpoint_steps=USE_DEFAULT,
 360                              summary_dir=None):
 361   """Creates a `MonitoredSession` for training.
 362 
 363   For a chief, this utility sets proper session initializer/restorer. It also
 364   creates hooks related to checkpoint and summary saving. For workers, this
 365   utility sets proper session creator which waits for the chief to
 366   initialize/restore. Please check `tf.train.MonitoredSession` for more
 367   information.
 368 
 369 
 370   Args:
 371     master: `String` the TensorFlow master to use.
 372     is_chief: If `True`, it will take care of initialization and recovery the
 373       underlying TensorFlow session. If `False`, it will wait on a chief to
 374       initialize or recover the TensorFlow session.
 375     checkpoint_dir: A string.  Optional path to a directory where to restore
 376       variables.
 377     scaffold: A `Scaffold` used for gathering or building supportive ops. If
 378       not specified, a default one is created. It's used to finalize the graph.
 379     hooks: Optional list of `SessionRunHook` objects.
 380     chief_only_hooks: list of `SessionRunHook` objects. Activate these hooks if
 381       `is_chief==True`, ignore otherwise.
 382     save_checkpoint_secs: The frequency, in seconds, that a checkpoint is saved
 383       using a default checkpoint saver. If both `save_checkpoint_steps` and
 384       `save_checkpoint_secs` are set to `None`, then the default checkpoint
 385       saver isn't used. If both are provided, then only `save_checkpoint_secs`
 386       is used. Default 600.
 387     save_summaries_steps: The frequency, in number of global steps, that the
 388       summaries are written to disk using a default summary saver. If both
 389       `save_summaries_steps` and `save_summaries_secs` are set to `None`, then
 390       the default summary saver isn't used. Default 100.
 391     save_summaries_secs: The frequency, in secs, that the summaries are written
 392       to disk using a default summary saver.  If both `save_summaries_steps` and
 393       `save_summaries_secs` are set to `None`, then the default summary saver
 394       isn't used. Default not enabled.
 395     config: an instance of `tf.ConfigProto` proto used to configure the session.
 396       It's the `config` argument of constructor of `tf.Session`.
 397     stop_grace_period_secs: Number of seconds given to threads to stop after
 398       `close()` has been called.
 399     log_step_count_steps: The frequency, in number of global steps, that the
 400       global step/sec is logged.
 401     max_wait_secs: Maximum time workers should wait for the session to
 402       become available. This should be kept relatively short to help detect
 403       incorrect code, but sometimes may need to be increased if the chief takes
 404       a while to start up.
 405     save_checkpoint_steps: The frequency, in number of global steps, that a
 406       checkpoint is saved using a default checkpoint saver. If both
 407       `save_checkpoint_steps` and `save_checkpoint_secs` are set to `None`, then
 408       the default checkpoint saver isn't used. If both are provided, then only
 409       `save_checkpoint_secs` is used. Default not enabled.
 410     summary_dir: A string.  Optional path to a directory where to
 411       save summaries. If None, checkpoint_dir is used instead.
 412 
 413   Returns:
 414     A `MonitoredSession` object.
 415   """
 416   if save_summaries_steps == USE_DEFAULT and save_summaries_secs == USE_DEFAULT:
 417     save_summaries_steps = 100
 418     save_summaries_secs = None
 419   elif save_summaries_secs == USE_DEFAULT:
 420     save_summaries_secs = None
 421   elif save_summaries_steps == USE_DEFAULT:
 422     save_summaries_steps = None
 423 
 424   if (save_checkpoint_steps == USE_DEFAULT and
 425       save_checkpoint_secs == USE_DEFAULT):
 426     save_checkpoint_steps = None
 427     save_checkpoint_secs = 600
 428   elif save_checkpoint_secs == USE_DEFAULT:
 429     save_checkpoint_secs = None
 430   elif save_checkpoint_steps == USE_DEFAULT:
 431     save_checkpoint_steps = None
 432 
 433   scaffold = scaffold or Scaffold()
 434   worker_context = distribute_coordinator_context.get_current_worker_context()
 435 
 436   if worker_context:
 437     return _create_monitored_session_with_worker_context(
 438         worker_context,
 439         scaffold,
 440         checkpoint_dir=checkpoint_dir,
 441         hooks=hooks,
 442         chief_only_hooks=chief_only_hooks,
 443         save_checkpoint_secs=save_checkpoint_secs,
 444         save_summaries_steps=save_summaries_steps,
 445         save_summaries_secs=save_summaries_secs,
 446         config=config,
 447         stop_grace_period_secs=stop_grace_period_secs,
 448         log_step_count_steps=log_step_count_steps,
 449         max_wait_secs=max_wait_secs,
 450         save_checkpoint_steps=save_checkpoint_steps,
 451         summary_dir=summary_dir)
 452 
 453   if not is_chief:
 454     session_creator = WorkerSessionCreator(
 455         scaffold=scaffold,
 456         master=master,
 457         config=config,
 458         max_wait_secs=max_wait_secs)
 459     return MonitoredSession(
 460         session_creator=session_creator,
 461         hooks=hooks or [],
 462         stop_grace_period_secs=stop_grace_period_secs)
 463 
 464   all_hooks = []
 465   if chief_only_hooks:
 466     all_hooks.extend(chief_only_hooks)
 467   session_creator = ChiefSessionCreator(
 468       scaffold=scaffold,
 469       checkpoint_dir=checkpoint_dir,
 470       master=master,
 471       config=config)
 472 
 473   summary_dir = summary_dir or checkpoint_dir
 474   if summary_dir:
 475     if log_step_count_steps and log_step_count_steps > 0:
 476       all_hooks.append(
 477           basic_session_run_hooks.StepCounterHook(
 478               output_dir=summary_dir, every_n_steps=log_step_count_steps))
 479 
 480     if (save_summaries_steps and save_summaries_steps > 0) or (
 481         save_summaries_secs and save_summaries_secs > 0):
 482       all_hooks.append(
 483           basic_session_run_hooks.SummarySaverHook(
 484               scaffold=scaffold,
 485               save_steps=save_summaries_steps,
 486               save_secs=save_summaries_secs,
 487               output_dir=summary_dir))
 488 
 489   if checkpoint_dir:
 490     if (save_checkpoint_secs and save_checkpoint_secs > 0) or (
 491         save_checkpoint_steps and save_checkpoint_steps > 0):
 492       all_hooks.append(
 493           basic_session_run_hooks.CheckpointSaverHook(
 494               checkpoint_dir,
 495               save_steps=save_checkpoint_steps,
 496               save_secs=save_checkpoint_secs,
 497               scaffold=scaffold))
 498 
 499   if hooks:
 500     all_hooks.extend(hooks)
 501   return MonitoredSession(
 502       session_creator=session_creator,
 503       hooks=all_hooks,
 504       stop_grace_period_secs=stop_grace_period_secs)

 528

 673   def run_step_fn(self, step_fn):
 674     """Run ops using a step function.
 675 
 676     Args:
 677       step_fn: A function or a method with a single argument of type
 678         `StepContext`.  The function may use methods of the argument to
 679         perform computations with access to a raw session.
 680 
 681         The returned value of the `step_fn` will be returned from `run_step_fn`,
 682         unless a stop is requested.  In that case, the next `should_stop` call
 683         will return True.
 684 
 685         Example usage:
 686 
 687         ```python
 688            with tf.Graph().as_default():
 689              c = tf.placeholder(dtypes.float32)
 690              v = tf.add(c, 4.0)
 691              w = tf.add(c, 0.5)
 692 
 693              def step_fn(step_context):
 694                a = step_context.session.run(fetches=v, feed_dict={c: 0.5})
 695                if a <= 4.5:
 696                  step_context.request_stop()
 697                return step_context.run_with_hooks(fetches=w, feed_dict={c: 0.1})
 698 
 699              with tf.MonitoredSession() as session:
 700                while not session.should_stop():
 701                  a = session.run_step_fn(step_fn)
 702         ```
 703 
 704         Hooks interact with the `run_with_hooks()` call inside the `step_fn`
 705         as they do with a `MonitoredSession.run` call.
 706 
 707     Returns:
 708       Returns the returned value of `step_fn`.
 709 
 710     Raises:
 711       StopIteration: if `step_fn` has called `request_stop()`.  It may be
 712         caught by `with tf.MonitoredSession()` to close the session.
 713       ValueError: if `step_fn` doesn't have a single argument called
 714         `step_context`. It may also optionally have `self` for cases when it
 715         belongs to an object.
 716     """
 717     step_fn_arguments = function_utils.fn_args(step_fn)
 718     if step_fn_arguments != ('step_context',) and step_fn_arguments != (
 719         'self',
 720         'step_context',
 721     ):
 722       raise ValueError(
 723           '`step_fn` may either have one `step_context` argument, or'
 724           ' `self` and `step_context` arguments if it\'s an instance'
 725           ' method. Got {} instead.'.format(step_fn_arguments))
 726 
 727     # `self._sess` is either `_RecoverableSession` or a `_CoordinatedSession`.
 728     # Setting `run_with_hooks` to `None` will cause `run_with_hooks` to be
 729     # `_CoordinatedSession.run` downstream in either case. This allows
 730     # `_PREEMPTION_ERRORS` to propage from within `step_fn` to
 731     # `_RecoverableSession.run_step_fn`.
 732     return self._sess.run_step_fn(step_fn, self._tf_sess(), run_with_hooks=None)
 733 
 734   class StepContext(object):
 735     """Control flow instrument for the `step_fn` from `run_step_fn()`.
 736 
 737        Users of `step_fn` may perform `run()` calls without running hooks
 738        by accessing the `session`.  A `run()` call with hooks may be performed
 739        using `run_with_hooks()`.  Computation flow can be interrupted using
 740        `request_stop()`.
 741     """
 742 
 743     def __init__(self, session, run_with_hooks_fn):
 744       """Initializes the `step_context` argument for a `step_fn` invocation.
 745 
 746       Args:
 747         session: An instance of `tf.Session`.
 748         run_with_hooks_fn: A function for running fetches and hooks.
 749       """
 750       self._session = session
 751       self._run_with_hooks_fn = run_with_hooks_fn
 752 
 753     @property
 754     def session(self):
 755       return self._session
 756 
 757     def run_with_hooks(self, *args, **kwargs):
 758       """Same as `MonitoredSession.run`. Accepts the same arguments."""
 759       return self._run_with_hooks_fn(*args, **kwargs)
 760 
 761     def request_stop(self):
 762       """Exit the training loop by causing `should_stop()` to return `True`.
 763 
 764          Causes `step_fn` to exit by raising an exception.
 765 
 766       Raises:
 767         StopIteration
 768       """
 769       raise StopIteration('step_fn has requested the iterations to stop.')

 772     return self._sess is None or self._sess.should_stop()

 778     return self

 782       exception_type = None

 843   """Session-like object that handles initialization, recovery and hooks.
 844 
 845   Example usage:
 846 
 847   ```python
 848   saver_hook = CheckpointSaverHook(...)
 849   summary_hook = SummarySaverHook(...)
 850   with MonitoredSession(session_creator=ChiefSessionCreator(...),
 851                         hooks=[saver_hook, summary_hook]) as sess:
 852     while not sess.should_stop():
 853       sess.run(train_op)
 854   ```
 855 
 856   Initialization: At creation time the monitored session does following things
 857   in given order:
 858 
 859   * calls `hook.begin()` for each given hook
 860   * finalizes the graph via `scaffold.finalize()`
 861   * create session
 862   * initializes the model via initialization ops provided by `Scaffold`
 863   * restores variables if a checkpoint exists
 864   * launches queue runners
 865   * calls `hook.after_create_session()`
 866 
 867   Run: When `run()` is called, the monitored session does following things:
 868 
 869   * calls `hook.before_run()`
 870   * calls TensorFlow `session.run()` with merged fetches and feed_dict
 871   * calls `hook.after_run()`
 872   * returns result of `session.run()` asked by user
 873   * if `AbortedError` or `UnavailableError` occurs, it recovers or
 874     reinitializes the session before executing the run() call again
 875 
 876 
 877   Exit: At the `close()`, the monitored session does following things in order:
 878 
 879   * calls `hook.end()`
 880   * closes the queue runners and the session
 881   * suppresses `OutOfRange` error which indicates that all inputs have been
 882     processed if the monitored_session is used as a context
 883 
 884   How to set `tf.Session` arguments:
 885 
 886   * In most cases you can set session arguments as follows:
 887 
 888   ```python
 889   MonitoredSession(
 890     session_creator=ChiefSessionCreator(master=..., config=...))
 891   ```
 892 
 893   * In distributed setting for a non-chief worker, you can use following:
 894 
 895   ```python
 896   MonitoredSession(
 897     session_creator=WorkerSessionCreator(master=..., config=...))
 898   ```
 899 
 900   See `MonitoredTrainingSession` for an example usage based on chief or worker.
 901 
 902   Note: This is not a `tf.Session`. For example, it cannot do following:
 903 
 904   * it cannot be set as default session.
 905   * it cannot be sent to saver.save.
 906   * it cannot be sent to tf.train.start_queue_runners.
 907 
 908   Args:
 909     session_creator: A factory object to create session. Typically a
 910       `ChiefSessionCreator` which is the default one.
 911     hooks: An iterable of `SessionRunHook' objects.
 912 
 913   Returns:
 914     A MonitoredSession object.
 915   """
 916 
 917   def __init__(self, session_creator=None, hooks=None,
 918                stop_grace_period_secs=120):
 919     super(MonitoredSession, self).__init__(
 920         session_creator, hooks, should_recover=True,
 921         stop_grace_period_secs=stop_grace_period_secs)

1015 class _WrappedSession(object):
1016   """Wrapper around a `tf.Session`.
1017 
1018   This wrapper is used as a base class for various session wrappers
1019   that provide additional functionality such as monitoring, coordination,
1020   and recovery.
1021 
1022   In addition to the methods exported by `SessionInterface` the wrapper
1023   provides a method to check for stop and never raises exceptions from
1024   calls to `close()`.
1025   """
1026 
1027   def __init__(self, sess):
1028     """Creates a `_WrappedSession`.
1029 
1030     Args:
1031       sess: A `tf.Session` or `_WrappedSession` object.  The wrapped session.
1032     """
1033     self._sess = sess
1034     self._wrapped_is_stoppable = isinstance(self._sess, _WrappedSession)
1035 
1036   @property
1037   def graph(self):
1038     return self._sess.graph
1039 
1040   @property
1041   def sess_str(self):
1042     return self._sess.sess_str

1044   def should_stop(self):
1045     """Return true if this session should not be used anymore.
1046 
1047     Always return True if the session was closed.
1048 
1049     Returns:
1050       True if the session should stop, False otherwise.
1051     """
1052     if self._check_stop():
1053       return True
1054     if self._sess:
1055       return self._wrapped_is_stoppable and self._sess.should_stop()
1056     return True
1057 
1058   def _check_stop(self):
1059     """Hook for subclasses to provide their own stop condition.
1060 
1061     Returns:

1064     return False

1068       try:
1069         self._sess.close()
1070       except _PREEMPTION_ERRORS:
1071         pass
1072       finally:
1073         self._sess = None
1074 
1075   def run(self, *args, **kwargs):
1076     return self._sess.run(*args, **kwargs)
1077 
1078   def run_step_fn(self, step_fn, raw_session, run_with_hooks):
1079     # `_RecoverableSession` sets `run_with_hooks` to `_CoordinatedSession.run`.
1080     # It is `None` when called from `_CoordinatedSession`. In that case
1081     # `self.run` is `_CoordinatedSession.run`.
1082     run_with_hooks = run_with_hooks or self.run

1086 class _RecoverableSession(_WrappedSession):
1087   """A wrapped session that recreates a session upon certain kinds of errors.
1088 
1089   The constructor is passed a SessionCreator object, not a session.
1090 
1091   Calls to `run()` are delegated to the wrapped session.  If a call raises the
1092   exception `tf.errors.AbortedError` or `tf.errors.UnavailableError`, the
1093   wrapped session is closed, and a new one is created by calling the factory
1094   again.
1095   """
1096 
1097   def __init__(self, sess_creator):
1098     """Create a new `_RecoverableSession`.
1099 
1100     The value returned by calling `sess_creator.create_session()` will be the
1101     session wrapped by this recoverable session.
1102 
1103     Args:
1104       sess_creator: A 'SessionCreator' to be wrapped by recoverable.
1105     """
1106     self._sess_creator = sess_creator
1107     _WrappedSession.__init__(self, self._create_session())
1108 
1109   def _create_session(self):
1110     while True:
1111       try:
1112         return self._sess_creator.create_session()
1113       except _PREEMPTION_ERRORS as e:
1114         logging.info('An error was raised while a session was being created. '
1115                      'This may be due to a preemption of a connected worker '
1116                      'or parameter server. A new session will be created. '
1117                      'This error may also occur due to a gRPC failure caused '
1118                      'by high memory or network bandwidth usage in the '
1119                      'parameter servers. If this error occurs repeatedly, try '
1120                      'increasing the number of parameter servers assigned to '
1121                      'the job. Error: %s', e)
1122 
1123   def _check_stop(self):
1124     try:
1125       if self._sess:
1126         return self._sess._check_stop()  # pylint: disable=protected-access
1127       else:
1128         return True
1129     except _PREEMPTION_ERRORS as e:
1130       logging.info('An error was raised while considering whether the '
1131                    'session is complete. This may be due to a preemption in '
1132                    'a connected worker or parameter server. The current '
1133                    'session will be closed and a new session will be '
1134                    'created. This error may also occur due to a gRPC failure '
1135                    'caused by high memory or network bandwidth usage in the '
1136                    'parameter servers. If this error occurs repeatedly, try '
1137                    'increasing the number of parameter servers assigned to '
1138                    'the job. Error: %s', e)
1139       self.close()
1140       self._sess = self._create_session()
1141       # Since we have just recreated the session, the overall computation should
1142       # not stop:
1143       return False
1144     except Exception:  # pylint: disable=broad-except
1145       # `should_stop` should return True instead of raising an exception.
1146       return True
1147 
1148   def run(self, fetches, feed_dict=None, options=None, run_metadata=None):
1149     while True:
1150       try:
1151         if not self._sess:
1152           self._sess = self._create_session()
1153         return self._sess.run(fetches,
1154                               feed_dict=feed_dict,
1155                               options=options,
1156                               run_metadata=run_metadata)
1157       except _PREEMPTION_ERRORS as e:
1158         logging.info('An error was raised. This may be due to a preemption in '
1159                      'a connected worker or parameter server. The current '
1160                      'session will be closed and a new session will be '
1161                      'created. This error may also occur due to a gRPC failure '
1162                      'caused by high memory or network bandwidth usage in the '
1163                      'parameter servers. If this error occurs repeatedly, try '
1164                      'increasing the number of parameter servers assigned to '
1165                      'the job. Error: %s', e)
1166         self.close()
1167         self._sess = None

1172         if not self._sess:

1190 class _CoordinatedSession(_WrappedSession):
1191   """A wrapped session that works with a `tf.Coordinator`.
1192 
1193   Calls to `run()` are delegated to the wrapped session.  If a call
1194   raises an exception, the exception is reported to the coordinator.
1195 
1196   In addition, after each call to `run()` this session ask the coordinator if
1197   the session should stop.  In that case it will will join all the threads
1198   registered with the coordinator before returning.
1199 
1200   If the coordinator was requested to stop with an exception, that exception
1201   will be re-raised from the call to `run()`.
1202   """
1203 
1204   def __init__(self, sess, coord, stop_grace_period_secs=120):
1205     """Create a new `_CoordinatedSession`.
1206 
1207     Args:
1208       sess: A `tf.Session` object.  The wrapped session.
1209       coord: A `tf.train.Coordinator` object.
1210       stop_grace_period_secs: Number of seconds given to threads to stop after
1211         `close()` has been called.
1212     """
1213     _WrappedSession.__init__(self, sess)
1214     self._coord = coord
1215     self._stop_grace_period_secs = stop_grace_period_secs
1216 
1217   def _check_stop(self):
1218     # If the coordinator was asked to stop due to an exception, then it needs
1219     # to be propagated to this stack.
1220     self._coord.raise_requested_exception()
1221     # At this point, no exceptions are recorded in the coordinator.
1222     return self._coord.should_stop()
1223 
1224   def close(self):
1225     self._coord.request_stop()
1226     try:
1227       self._coord.join(
1228           stop_grace_period_secs=self._stop_grace_period_secs,
1229           ignore_live_threads=True)
1230     finally:
1231       try:
1232         _WrappedSession.close(self)
1233       except Exception:  # pylint: disable=broad-except
1234         # We intentionally suppress exceptions from the close() here since
1235         # useful exceptions are already reported by join().
1236         pass
1237 
1238   def run(self, *args, **kwargs):
1239     try:
1240       return self._sess.run(*args, **kwargs)
1241     except _PREEMPTION_ERRORS:
1242       raise
1243     except Exception:  # pylint: disable=broad-except
1244       # A non-preemption error could have been caused by a preemption error
1245       # in the coordinator. If this is the case, raise that exception instead,
1246       # since it's the root cause. Otherwise, stick to the `original_exc_info`.
1247       original_exc_info = sys.exc_info()
1248       try:
1249         self._coord.raise_requested_exception()
1250       except _PREEMPTION_ERRORS:
1251         raise
1252       except Exception:  # pylint: disable=broad-except
1253         raise six.reraise(*original_exc_info)
1254       else:
1255         raise six.reraise(*original_exc_info)
1256

1258 class _HookedSession(_WrappedSession):
1259   """A _WrappedSession that calls hooks during calls to run().
1260 
1261   The list of hooks to call is passed in the constructor.  Before each call
1262   to `run()` the session calls the `before_run()` method of the hooks, which
1263   can return additional ops or tensors to run.  These are added to the arguments
1264   of the call to `run()`.
1265 
1266   When the `run()` call finishes, the session calls the `after_run()` methods of
1267   the hooks, passing the values returned by the `run()` call corresponding to
1268   the ops and tensors that each hook requested.
1269 
1270   If any call to the hooks, requests stop via run_context the session will be
1271   marked as needing to stop and its `should_stop()` method will now return
1272   `True`.
1273   """
1274 
1275   def __init__(self, sess, hooks):
1276     """Initializes a _HookedSession object.
1277 
1278     Args:
1279       sess: A `tf.Session` or a `_WrappedSession` object.
1280       hooks: An iterable of `SessionRunHook' objects.
1281     """
1282 
1283     _WrappedSession.__init__(self, sess)
1284     self._hooks = hooks
1285     self._should_stop = False
1286 
1287   def _check_stop(self):
1288     """See base class."""
1289     return self._should_stop
1290 
1291   def run(self, fetches, feed_dict=None, options=None, run_metadata=None):
1292     """See base class."""
1293     if self.should_stop():
1294       raise RuntimeError('Run called even after should_stop requested.')
1295 
1296     actual_fetches = {'caller': fetches}
1297 
1298     run_context = session_run_hook.SessionRunContext(
1299         original_args=session_run_hook.SessionRunArgs(fetches, feed_dict),
1300         session=self._sess)
1301 
1302     options = options or config_pb2.RunOptions()
1303     feed_dict = self._call_hook_before_run(run_context, actual_fetches,
1304                                            feed_dict, options)
1305 
1306     # Do session run.
1307     run_metadata = run_metadata or config_pb2.RunMetadata()
1308     outputs = _WrappedSession.run(self,
1309                                   fetches=actual_fetches,
1310                                   feed_dict=feed_dict,
1311                                   options=options,
1312                                   run_metadata=run_metadata)
1313 
1314     for hook in self._hooks:
1315       hook.after_run(
1316           run_context,
1317           session_run_hook.SessionRunValues(
1318               results=outputs[hook] if hook in outputs else None,
1319               options=options,
1320               run_metadata=run_metadata))
1321     self._should_stop = self._should_stop or run_context.stop_requested
1322 
1323     return outputs['caller']
1324 
1325   def _call_hook_before_run(self, run_context, fetch_dict, user_feed_dict,
1326                             options):
1327     """Calls hooks.before_run and handles requests from hooks."""
1328     hook_feeds = {}
1329     for hook in self._hooks:
1330       request = hook.before_run(run_context)
1331       if request is not None:
1332         if request.fetches is not None:
1333           fetch_dict[hook] = request.fetches
1334         if request.feed_dict:
1335           self._raise_if_feeds_intersects(
1336               hook_feeds, request.feed_dict,
1337               'Same tensor is fed by two hooks.')
1338           hook_feeds.update(request.feed_dict)
1339         if request.options:
1340           self._merge_run_options(options, request.options)
1341 
1342     if not hook_feeds:
1343       return user_feed_dict
1344 
1345     if not user_feed_dict:
1346       return hook_feeds
1347 
1348     self._raise_if_feeds_intersects(
1349         user_feed_dict, hook_feeds,
1350         'Same tensor is fed by a SessionRunHook and user.')
1351     hook_feeds.update(user_feed_dict)
1352     return hook_feeds
1353 
1354   def _raise_if_feeds_intersects(self, feeds1, feeds2, message):
1355     intersection = set(feeds1.keys()) & set(feeds2.keys())
1356     if intersection:
1357       raise RuntimeError(message + ' Conflict(s): ' + str(list(intersection)))

1359   def _merge_run_options(self, options, incoming_options):
1360     """Merge two instances of RunOptions into the first one.
1361 
1362     During the merger, the numerical fields including trace_level,
1363     timeout_in_ms, inter_op_thread_pool are set to the larger one of the two.
1364     The boolean value is set to the logical OR of the two.
1365     debug_tensor_watch_opts of the original options is extended with that from
1366     the incoming one.
1367 
1368     Args:
1369       options: The options to merge into.
1370       incoming_options: The options to be merged into the first argument.
1371     """
1372     options.trace_level = max(options.trace_level, incoming_options.trace_level)
1373     options.timeout_in_ms = max(options.timeout_in_ms,
1374                                 incoming_options.timeout_in_ms)
1375     options.inter_op_thread_pool = max(options.inter_op_thread_pool,
1376                                        incoming_options.inter_op_thread_pool)
1377     options.output_partition_graphs = max(
1378         options.output_partition_graphs,
1379         incoming_options.output_partition_graphs)
1380 
1381     options.debug_options.debug_tensor_watch_opts.extend(
1382         incoming_options.debug_options.debug_tensor_watch_opts)
1383     options.debug_options.reset_disk_byte_usage = (
1384         options.debug_options.reset_disk_byte_usage or
1385         incoming_options.debug_options.reset_disk_byte_usage)
