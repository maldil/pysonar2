  16 """Standard functions for creating slots.
  17 
  18 A slot is a `Variable` created with the same shape as a primary variable or
  19 `Tensor`. A slot is always scoped in the namespace of the primary object and
  20 typically has the same device and type.
  21 
  22 Slots are typically used as accumulators to track values associated with
  23 the primary object:
  24 
  25 ```python
  26 # Optimizers can create a slot for each variable to track accumulators
  27 accumulators = {var : create_zeros_slot(var, "momentum") for var in vs}
  28 for var in vs:
  29   apply_momentum(var, accumulators[var], lr, grad, momentum_tensor)
  30 
  31 # Slots can also be used for moving averages
  32 mavg = create_slot(var, var.initialized_value(), "exponential_moving_avg")
  33 update_mavg = mavg.assign_sub((mavg - var) * (1 - decay))
  34 ```
  35 """

  51 def _create_slot_var(primary, val, scope, validate_shape, shape, dtype):
  52   """Helper function for creating a slot variable."""
  53 
  54   # TODO(lukaszkaiser): Consider allowing partitioners to be set in the current
  55   # scope.
  56   current_partitioner = variable_scope.get_variable_scope().partitioner
  57   variable_scope.get_variable_scope().set_partitioner(None)
  58   # When init from val instead of callable initializer, the shape is expected to
  59   # be None, not <unknown> or any fully defined shape.
  60   shape = shape if callable(val) else None
  61   slot = variable_scope.get_variable(
  62       scope, initializer=val, trainable=False,
  63       use_resource=resource_variable_ops.is_resource_variable(primary),
  64       shape=shape, dtype=dtype,
  65       validate_shape=validate_shape)
  66   variable_scope.get_variable_scope().set_partitioner(current_partitioner)
  67 
  68   # pylint: disable=protected-access
  69   if isinstance(primary, variables.Variable) and primary._save_slice_info:
  70     # Primary is a partitioned variable, so we need to also indicate that
  71     # the slot is a partitioned variable.  Slots have the same partitioning
  72     # as their primaries.
  73     # For examples when using AdamOptimizer in linear model, slot.name
  74     # here can be "linear//weights/Adam:0", while primary.op.name is
  75     # "linear//weight". We want to get 'Adam' as real_slot_name, so we
  76     # remove "'linear//weight' + '/'" and ':0'.
  77     real_slot_name = slot.name[len(primary.op.name + "/"):-2]
  78     slice_info = primary._save_slice_info
  79     slot._set_save_slice_info(variables.Variable.SaveSliceInfo(
  80         slice_info.full_name + "/" + real_slot_name,
  81         slice_info.full_shape[:],
  82         slice_info.var_offset[:],
  83         slice_info.var_shape[:]))
  84   # pylint: enable=protected-access
  85   return slot

  88 def create_slot(primary, val, name, colocate_with_primary=True):
  89   """Create a slot initialized to the given value.
  90 
  91   The type of the slot is determined by the given value.
  92 
  93   Args:
  94     primary: The primary `Variable` or `Tensor`.
  95     val: A `Tensor` specifying the initial value of the slot.
  96     name: Name to use for the slot variable.
  97     colocate_with_primary: Boolean.  If True the slot is located
  98       on the same device as `primary`.
  99 
 100   Returns:
 101     A `Variable` object.
 102   """
 103   # Scope the slot name in the namespace of the primary variable.
 104   # Set "primary.op.name + '/' + name" as default name, so the scope name of
 105   # optimizer can be shared when reuse is True. Meanwhile when reuse is False
 106   # and the same name has been previously used, the scope name will add '_N'
 107   # as suffix for unique identifications.
 108   validate_shape = val.get_shape().is_fully_defined()
 109   if context.executing_eagerly():
 110     prefix = primary._shared_name  # pylint: disable=protected-access
 111   else:
 112     prefix = primary.op.name
 113   with variable_scope.variable_scope(None, prefix + "/" + name):
 114     if colocate_with_primary:
 115       distribution_strategy = (
 116           distribution_strategy_context.get_distribution_strategy())
 117       with distribution_strategy.colocate_vars_with(primary):
 118         return _create_slot_var(primary, val, "", validate_shape, None, None)
 119     else:
 120       return _create_slot_var(primary, val, "", validate_shape, None, None)

 123 def create_slot_with_initializer(primary, initializer, shape, dtype, name,
 124                                  colocate_with_primary=True):
 125   """Creates a slot initialized using an `Initializer`.
 126 
 127   The type of the slot is determined by the given value.
 128 
 129   Args:
 130     primary: The primary `Variable` or `Tensor`.
 131     initializer: An `Initializer`.  The initial value of the slot.
 132     shape: Shape of the initial value of the slot.
 133     dtype: Type of the value of the slot.
 134     name: Name to use for the slot variable.
 135     colocate_with_primary: Boolean.  If True the slot is located
 136       on the same device as `primary`.
 137 
 138   Returns:
 139     A `Variable` object.
 140   """
 141   # Scope the slot name in the namespace of the primary variable.
 142   # Set "primary.op.name + '/' + name" as default name, so the scope name of
 143   # optimizer can be shared when reuse is True. Meanwhile when reuse is False
 144   # and the same name has been previously used, the scope name will add '_N'
 145   # as suffix for unique identifications.
 146   validate_shape = shape.is_fully_defined()
 147   if context.executing_eagerly():
 148     prefix = primary._shared_name  # pylint: disable=protected-access
 149   else:
 150     prefix = primary.op.name
 151   with variable_scope.variable_scope(None, prefix + "/" + name):
 152     if colocate_with_primary:
 153       distribution_strategy = (
 154           distribution_strategy_context.get_distribution_strategy())
 155       with distribution_strategy.colocate_vars_with(primary):
 156         return _create_slot_var(primary, initializer, "", validate_shape, shape,
 157                                 dtype)
 158     else:
 159       return _create_slot_var(primary, initializer, "", validate_shape, shape,
 160                               dtype)

 163 def create_zeros_slot(primary, name, dtype=None, colocate_with_primary=True):
 164   """Create a slot initialized to 0 with same shape as the primary object.
 165 
 166   Args:
 167     primary: The primary `Variable` or `Tensor`.
 168     name: Name to use for the slot variable.
 169     dtype: Type of the slot variable.  Defaults to the type of `primary`.
 170     colocate_with_primary: Boolean.  If True the slot is located
 171       on the same device as `primary`.
 172 
 173   Returns:
 174     A `Variable` object.
 175   """
 176   if dtype is None:
 177     dtype = primary.dtype
 178   slot_shape = primary.get_shape()
 179   if slot_shape.is_fully_defined():
 180     initializer = init_ops.zeros_initializer(dtype)
 181     return create_slot_with_initializer(
 182         primary, initializer, slot_shape, dtype, name,
 183         colocate_with_primary=colocate_with_primary)
 184   else:
 185     if isinstance(primary, variables.Variable):
 186       slot_shape = array_ops.shape(primary.initialized_value())
 187     else:
 188       slot_shape = array_ops.shape(primary)
 189     val = array_ops.zeros(slot_shape, dtype=dtype)
 190     return create_slot(primary, val, name,
 191                        colocate_with_primary=colocate_with_primary)
