  45 def get_filtered_grad_fn(grad_fn):
  46   # `distributed_context.join()` requires that its arguments are parallel
  47   # across threads, and in particular that `grads_and_vars` has the same
  48   # variables in the same order.
  49 
  50   # When computing gradients in eager mode with multiple threads, you
  51   # can get extra variables with a gradient of `None`. This happens when
  52   # those variables are accessed in another thread during the gradient
  53   # computation. To get a consistent set of variables, we filter out
  54   # those with `None` gradients.
  55   def filtered_grad_fn(*args, **kwargs):
  56     return [(g, v) for g, v in grad_fn(*args, **kwargs) if g is not None]
  57 
  58   return filtered_grad_fn

  61 def _deduplicate_indexed_slices(values, indices):
  62   """Sums `values` associated with any non-unique `indices`.
  63 
  64   Args:
  65     values: A `Tensor` with rank >= 1.
  66     indices: A one-dimensional integer `Tensor`, indexing into the first
  67       dimension of `values` (as in an IndexedSlices object).
  68   Returns:
  69     A tuple of (`summed_values`, `unique_indices`) where `unique_indices` is a
  70     de-duplicated version of `indices` and `summed_values` contains the sum of
  71     `values` slices associated with each unique index.
  72   """
  73   unique_indices, new_index_positions = array_ops.unique(indices)
  74   summed_values = math_ops.unsorted_segment_sum(
  75       values, new_index_positions,
  76       array_ops.shape(unique_indices)[0])
  77   return (summed_values, unique_indices)

  87 class _OptimizableVariable(object):
  88   """Interface for abstracting over variables in the optimizers."""
  89 
  90   @abc.abstractmethod
  91   def target(self):
  92     """Returns the optimization target for this variable."""
  93     raise NotImplementedError("Calling an abstract method.")
  94 
  95   @abc.abstractmethod
  96   def update_op(self, optimizer, g):
  97     """Returns the update ops for updating the variable."""
  98     raise NotImplementedError("Calling an abstract method.")

 101 class _RefVariableProcessor(_OptimizableVariable):
 102   """Processor for Variable."""
 103 
 104   def __init__(self, v):
 105     self._v = v
 106 
 107   def __str__(self):
 108     return "<_RefVariableProcessor(%s)>" % self._v
 109 
 110   def target(self):
 111     return self._v._ref()  # pylint: disable=protected-access
 112 
 113   def update_op(self, optimizer, g):
 114     if isinstance(g, ops.Tensor):
 115       update_op = optimizer._apply_dense(g, self._v)  # pylint: disable=protected-access
 116       if self._v.constraint is not None:
 117         with ops.control_dependencies([update_op]):
 118           return self._v.assign(self._v.constraint(self._v))
 119       else:
 120         return update_op
 121     else:
 122       assert isinstance(g, ops.IndexedSlices), ("Gradient ", g, " is neither a "
 123                                                 "tensor nor IndexedSlices.")
 124       if self._v.constraint is not None:
 125         raise RuntimeError(
 126             "Cannot use a constraint function on a sparse variable.")
 127       # pylint: disable=protected-access
 128       return optimizer._apply_sparse_duplicate_indices(g, self._v)

 150 class _DenseResourceVariableProcessor(_OptimizableVariable):
 151   """Processor for dense ResourceVariables."""
 152 
 153   def __init__(self, v):
 154     self._v = v
 155 
 156   def target(self):
 157     return self._v
 158 
 159   def update_op(self, optimizer, g):
 160     # pylint: disable=protected-access
 161     if isinstance(g, ops.IndexedSlices):
 162       if self._v.constraint is not None:
 163         raise RuntimeError(
 164             "Cannot use a constraint function on a sparse variable.")
 165       return optimizer._resource_apply_sparse_duplicate_indices(
 166           g.values, self._v, g.indices)
 167     update_op = optimizer._resource_apply_dense(g, self._v)
 168     if self._v.constraint is not None:
 169       with ops.control_dependencies([update_op]):
 170         return self._v.assign(self._v.constraint(self._v))
 171     else:
 172       return update_op

 175 class _TensorProcessor(_OptimizableVariable):
 176   """Processor for ordinary Tensors.
 177 
 178   Even though a Tensor can't really be updated, sometimes it is useful to
 179   compute the gradients with respect to a Tensor using the optimizer. Updating
 180   the Tensor is, of course, unsupported.
 181   """
 182 
 183   def __init__(self, v):
 184     self._v = v
 185 
 186   def target(self):
 187     return self._v
 188 
 189   def update_op(self, optimizer, g):
 190     raise NotImplementedError("Trying to update a Tensor ", self._v)
 191 
 192 
 193 def _get_processor(v):
 194   """The processor of v."""
 195   if context.executing_eagerly():
 196     if isinstance(v, ops.Tensor):
 197       return _TensorProcessor(v)
 198     else:
 199       return _DenseResourceVariableProcessor(v)
 200   if isinstance(
 201       v, resource_variable_ops.ResourceVariable) and not v._in_graph_mode:  # pylint: disable=protected-access
 202     # True if and only if `v` was initialized eagerly.
 203     return _DenseResourceVariableProcessor(v)
 204   if v.op.type == "VarHandleOp":
 205     return _DenseResourceVariableProcessor(v)
 206   if isinstance(v, variables.Variable):
 207     return _RefVariableProcessor(v)
 208   if isinstance(v, ops.Tensor):
 209     return _TensorProcessor(v)
 210   raise NotImplementedError("Trying to optimize unsupported type ", v)

 213 @tf_export("train.Optimizer")
 214 class Optimizer(
 215     # Optimizers inherit from CheckpointableBase rather than Checkpointable
 216     # since they do most of their dependency management themselves (slot
 217     # variables are special-cased, and non-slot variables are keyed to graphs).
 218     checkpointable.CheckpointableBase):
 219   """Base class for optimizers.
 220 
 221   This class defines the API to add Ops to train a model.  You never use this
 222   class directly, but instead instantiate one of its subclasses such as
 223   `GradientDescentOptimizer`, `AdagradOptimizer`, or `MomentumOptimizer`.
 224 
 225   ### Usage
 226 
 227   ```python
 228   # Create an optimizer with the desired parameters.
 229   opt = GradientDescentOptimizer(learning_rate=0.1)
 230   # Add Ops to the graph to minimize a cost by updating a list of variables.
 231   # "cost" is a Tensor, and the list of variables contains tf.Variable
 232   # objects.
 233   opt_op = opt.minimize(cost, var_list=<list of variables>)
 234   ```
 235 
 236   In the training program you will just have to run the returned Op.
 237 
 238   ```python
 239   # Execute opt_op to do one step of training:
 240   opt_op.run()
 241   ```
 242 
 243   ### Processing gradients before applying them.
 244 
 245   Calling `minimize()` takes care of both computing the gradients and
 246   applying them to the variables.  If you want to process the gradients
 247   before applying them you can instead use the optimizer in three steps:
 248 
 249   1.  Compute the gradients with `compute_gradients()`.
 250   2.  Process the gradients as you wish.
 251   3.  Apply the processed gradients with `apply_gradients()`.
 252 
 253   Example:
 254 
 255   ```python
 256   # Create an optimizer.
 257   opt = GradientDescentOptimizer(learning_rate=0.1)
 258 
 259   # Compute the gradients for a list of variables.
 260   grads_and_vars = opt.compute_gradients(loss, <list of variables>)
 261 
 262   # grads_and_vars is a list of tuples (gradient, variable).  Do whatever you
 263   # need to the 'gradient' part, for example cap them, etc.
 264   capped_grads_and_vars = [(MyCapper(gv[0]), gv[1]) for gv in grads_and_vars]
 265 
 266   # Ask the optimizer to apply the capped gradients.
 267   opt.apply_gradients(capped_grads_and_vars)
 268   ```
 269 
 270   ### Gating Gradients
 271 
 272   Both `minimize()` and `compute_gradients()` accept a `gate_gradients`
 273   argument that controls the degree of parallelism during the application of
 274   the gradients.
 275 
 276   The possible values are: `GATE_NONE`, `GATE_OP`, and `GATE_GRAPH`.
 277 
 278   <b>`GATE_NONE`</b>: Compute and apply gradients in parallel.  This provides
 279   the maximum parallelism in execution, at the cost of some non-reproducibility
 280   in the results.  For example the two gradients of `matmul` depend on the input
 281   values: With `GATE_NONE` one of the gradients could be applied to one of the
 282   inputs _before_ the other gradient is computed resulting in non-reproducible
 283   results.
 284 
 285   <b>`GATE_OP`</b>: For each Op, make sure all gradients are computed before
 286   they are used.  This prevents race conditions for Ops that generate gradients
 287   for multiple inputs where the gradients depend on the inputs.
 288 
 289   <b>`GATE_GRAPH`</b>: Make sure all gradients for all variables are computed
 290   before any one of them is used.  This provides the least parallelism but can
 291   be useful if you want to process all gradients before applying any of them.
 292 
 293   ### Slots
 294 
 295   Some optimizer subclasses, such as `MomentumOptimizer` and `AdagradOptimizer`
 296   allocate and manage additional variables associated with the variables to
 297   train.  These are called <i>Slots</i>.  Slots have names and you can ask the
 298   optimizer for the names of the slots that it uses.  Once you have a slot name
 299   you can ask the optimizer for the variable it created to hold the slot value.
 300 
 301   This can be useful if you want to log debug a training algorithm, report stats
 302   about the slots, etc.
 303   """
 304 
 305   # Values for gate_gradients.
 306   GATE_NONE = 0
 307   GATE_OP = 1
 308   GATE_GRAPH = 2
 309 
 310   def __init__(self, use_locking, name):
 311     """Create a new Optimizer.
 312 
 313     This must be called by the constructors of subclasses.
 314 
 315     Args:
 316       use_locking: Bool. If True apply use locks to prevent concurrent updates
 317         to variables.
 318       name: A non-empty string.  The name to use for accumulators created
 319         for the optimizer.
 320 
 321     Raises:
 322       ValueError: If name is malformed.
 323     """
 324     if not name:
 325       raise ValueError("Must specify the optimizer name")
 326     self._use_locking = use_locking
 327     self._name = name
 328     # Dictionary of slots.
 329     #  {slot_name :
 330     #      {_var_key(variable_to_train): slot_for_the_variable, ... },
 331     #   ... }
 332     self._slots = {}
 333     self._non_slot_dict = {}
 334     # For implementing Checkpointable. Stores information about how to restore
 335     # slot variables which have not yet been created
 336     # (checkpointable._CheckpointPosition objects).
 337     #  {slot_name :
 338     #      {_var_key(variable_to_train): [checkpoint_position, ... ], ... },
 339     #   ... }
 340     self._deferred_slot_restorations = {}
 341 
 342     # TODO(isaprykin): When using a DistributionStrategy, and when an
 343     # optimizer is created in each tower, it might be dangerous to
 344     # rely on some Optimer methods.  When such methods are called on a
 345     # per-tower optimizer, an exception needs to be thrown.  We do
 346     # allow creation per-tower optimizers however, because the
 347     # compute_gradients()->apply_gradients() sequence is safe.
 348 
 349   def get_name(self):
 350     return self._name

 352   def minimize(self, loss, global_step=None, var_list=None,
 353                gate_gradients=GATE_OP, aggregation_method=None,
 354                colocate_gradients_with_ops=False, name=None,
 355                grad_loss=None):
 356     """Add operations to minimize `loss` by updating `var_list`.
 357 
 358     This method simply combines calls `compute_gradients()` and
 359     `apply_gradients()`. If you want to process the gradient before applying
 360     them call `compute_gradients()` and `apply_gradients()` explicitly instead
 361     of using this function.
 362 
 363     Args:
 364       loss: A `Tensor` containing the value to minimize.
 365       global_step: Optional `Variable` to increment by one after the
 366         variables have been updated.
 367       var_list: Optional list or tuple of `Variable` objects to update to
 368         minimize `loss`.  Defaults to the list of variables collected in
 369         the graph under the key `GraphKeys.TRAINABLE_VARIABLES`.
 370       gate_gradients: How to gate the computation of gradients.  Can be
 371         `GATE_NONE`, `GATE_OP`, or  `GATE_GRAPH`.
 372       aggregation_method: Specifies the method used to combine gradient terms.
 373         Valid values are defined in the class `AggregationMethod`.
 374       colocate_gradients_with_ops: If True, try colocating gradients with
 375         the corresponding op.
 376       name: Optional name for the returned operation.
 377       grad_loss: Optional. A `Tensor` holding the gradient computed for `loss`.
 378 
 379     Returns:
 380       An Operation that updates the variables in `var_list`.  If `global_step`
 381       was not `None`, that operation also increments `global_step`.
 382 
 383     Raises:
 384       ValueError: If some of the variables are not `Variable` objects.
 385 
 386     @compatibility(eager)
 387     When eager execution is enabled, `loss` should be a Python function that
 388     takes no arguments and computes the value to be minimized. Minimization (and
 389     gradient computation) is done with respect to the elements of `var_list` if
 390     not None, else with respect to any trainable variables created during the
 391     execution of the `loss` function. `gate_gradients`, `aggregation_method`,
 392     `colocate_gradients_with_ops` and `grad_loss` are ignored when eager
 393     execution is enabled.
 394     @end_compatibility
 395     """
 396     grads_and_vars = self.compute_gradients(
 397         loss, var_list=var_list, gate_gradients=gate_gradients,
 398         aggregation_method=aggregation_method,
 399         colocate_gradients_with_ops=colocate_gradients_with_ops,
 400         grad_loss=grad_loss)
 401 
 402     vars_with_grad = [v for g, v in grads_and_vars if g is not None]
 403     if not vars_with_grad:
 404       raise ValueError(
 405           "No gradients provided for any variable, check your graph for ops"
 406           " that do not support gradients, between variables %s and loss %s." %
 407           ([str(v) for _, v in grads_and_vars], loss))
 408 
 409     return self.apply_gradients(grads_and_vars, global_step=global_step,
 410                                 name=name)

 412   def compute_gradients(self, loss, var_list=None,
 413                         gate_gradients=GATE_OP,
 414                         aggregation_method=None,
 415                         colocate_gradients_with_ops=False,
 416                         grad_loss=None):
 417     """Compute gradients of `loss` for the variables in `var_list`.
 418 
 419     This is the first part of `minimize()`.  It returns a list
 420     of (gradient, variable) pairs where "gradient" is the gradient
 421     for "variable".  Note that "gradient" can be a `Tensor`, an
 422     `IndexedSlices`, or `None` if there is no gradient for the
 423     given variable.
 424 
 425     Args:
 426       loss: A Tensor containing the value to minimize or a callable taking
 427         no arguments which returns the value to minimize. When eager execution
 428         is enabled it must be a callable.
 429       var_list: Optional list or tuple of `tf.Variable` to update to minimize
 430         `loss`.  Defaults to the list of variables collected in the graph
 431         under the key `GraphKeys.TRAINABLE_VARIABLES`.
 432       gate_gradients: How to gate the computation of gradients.  Can be
 433         `GATE_NONE`, `GATE_OP`, or `GATE_GRAPH`.
 434       aggregation_method: Specifies the method used to combine gradient terms.
 435         Valid values are defined in the class `AggregationMethod`.
 436       colocate_gradients_with_ops: If True, try colocating gradients with
 437         the corresponding op.
 438       grad_loss: Optional. A `Tensor` holding the gradient computed for `loss`.
 439 
 440     Returns:
 441       A list of (gradient, variable) pairs. Variable is always present, but
 442       gradient can be `None`.
 443 
 444     Raises:
 445       TypeError: If `var_list` contains anything else than `Variable` objects.
 446       ValueError: If some arguments are invalid.
 447       RuntimeError: If called with eager execution enabled and `loss` is
 448         not callable.
 449 
 450     @compatibility(eager)
 451     When eager execution is enabled, `gate_gradients`, `aggregation_method`,
 452     and `colocate_gradients_with_ops` are ignored.
 453     @end_compatibility
 454     """
 455     if callable(loss):
 456       with backprop.GradientTape() as tape:
 457         if var_list is not None:
 458           tape.watch(var_list)
 459         loss_value = loss()
 460 
 461         # Scale loss if using a "mean" loss reduction and multiple towers.
 462         # Have to be careful to call distribute_lib.get_loss_reduction()
 463         # *after* loss() is evaluated, so we know what loss reduction it uses.
 464         # TODO(josh11b): Test that we handle weight decay in a reasonable way.
 465         if (distribute_lib.get_loss_reduction() ==
 466             variable_scope.VariableAggregation.MEAN):
 467           num_towers = distribution_strategy_context.get_distribution_strategy(
 468           ).num_towers
 469           if num_towers > 1:
 470             loss_value *= (1. / num_towers)
 471 
 472       if var_list is None:
 473         var_list = tape.watched_variables()
 474       # TODO(jhseu): Figure out why GradientTape's gradients don't require loss
 475       # to be executed.
 476       with ops.control_dependencies([loss_value]):
 477         grads = tape.gradient(loss_value, var_list, grad_loss)
 478       return list(zip(grads, var_list))
 479 
 480     # Non-callable/Tensor loss case
 481     if context.executing_eagerly():
 482       raise RuntimeError(
 483           "`loss` passed to Optimizer.compute_gradients should "
 484           "be a function when eager execution is enabled.")
 485 
 486     # Scale loss if using a "mean" loss reduction and multiple towers.
 487     if (distribute_lib.get_loss_reduction() ==
 488         variable_scope.VariableAggregation.MEAN):
 489       num_towers = distribution_strategy_context.get_distribution_strategy(
 490       ).num_towers
 491       if num_towers > 1:
 492         loss *= (1. / num_towers)
 493 
 494     if gate_gradients not in [Optimizer.GATE_NONE, Optimizer.GATE_OP,
 495                               Optimizer.GATE_GRAPH]:
 496       raise ValueError("gate_gradients must be one of: Optimizer.GATE_NONE, "
 497                        "Optimizer.GATE_OP, Optimizer.GATE_GRAPH.  Not %s" %
 498                        gate_gradients)
 499     self._assert_valid_dtypes([loss])
 500     if grad_loss is not None:
 501       self._assert_valid_dtypes([grad_loss])
 502     if var_list is None:
 503       var_list = (
 504           variables.trainable_variables() +
 505           ops.get_collection(ops.GraphKeys.TRAINABLE_RESOURCE_VARIABLES))
 506     else:
 507       var_list = nest.flatten(var_list)
 508     # pylint: disable=protected-access
 509     var_list += ops.get_collection(ops.GraphKeys._STREAMING_MODEL_PORTS)
 510     # pylint: enable=protected-access
 511     processors = [_get_processor(v) for v in var_list]
 512     if not var_list:
 513       raise ValueError("No variables to optimize.")
 514     var_refs = [p.target() for p in processors]
 515     grads = gradients.gradients(
 516         loss, var_refs, grad_ys=grad_loss,
 517         gate_gradients=(gate_gradients == Optimizer.GATE_OP),
 518         aggregation_method=aggregation_method,
 519         colocate_gradients_with_ops=colocate_gradients_with_ops)
 520     if gate_gradients == Optimizer.GATE_GRAPH:
 521       grads = control_flow_ops.tuple(grads)
 522     grads_and_vars = list(zip(grads, var_list))
 523     self._assert_valid_dtypes(
 524         [v for g, v in grads_and_vars
 525          if g is not None and v.dtype != dtypes.resource])
 526     return grads_and_vars
 527 
 528   def apply_gradients(self, grads_and_vars, global_step=None, name=None):
 529     """Apply gradients to variables.
 530 
 531     This is the second part of `minimize()`. It returns an `Operation` that
 532     applies gradients.
 533 
 534     Args:
 535       grads_and_vars: List of (gradient, variable) pairs as returned by
 536         `compute_gradients()`.
 537       global_step: Optional `Variable` to increment by one after the
 538         variables have been updated.
 539       name: Optional name for the returned operation.  Default to the
 540         name passed to the `Optimizer` constructor.
 541 
 542     Returns:
 543       An `Operation` that applies the specified gradients. If `global_step`
 544       was not None, that operation also increments `global_step`.
 545 
 546     Raises:
 547       TypeError: If `grads_and_vars` is malformed.
 548       ValueError: If none of the variables have gradients.
 549       RuntimeError: If you should use `_distributed_apply()` instead.
 550     """
 551     # This is a default implementation of apply_gradients() that can be shared
 552     # by most optimizers.  It relies on the subclass implementing the following
 553     # methods: _create_slots(), _prepare(), _apply_dense(), and _apply_sparse().
 554 
 555     # Handle DistributionStrategy case.
 556     if distribution_strategy_context.get_cross_tower_context():
 557       raise RuntimeError("Use `_distributed_apply()` instead of "
 558                          "`apply_gradients()` in a cross-tower context.")
 559     # TODO(isaprykin): Get rid of `has_distribution_strategy()` check by
 560     # always calling _distributed_apply(), using the default distribution
 561     # as needed.
 562     if distribution_strategy_context.has_distribution_strategy():
 563       grads_and_vars = get_filtered_grad_fn(lambda: grads_and_vars)()
 564       return distribution_strategy_context.get_tower_context().merge_call(
 565           self._distributed_apply, grads_and_vars, global_step, name)
 566 
 567     # No DistributionStrategy case.
 568     grads_and_vars = tuple(grads_and_vars)  # Make sure repeat iteration works.
 569     if not grads_and_vars:
 570       raise ValueError("No variables provided.")
 571     converted_grads_and_vars = []
 572     for g, v in grads_and_vars:
 573       if g is not None:
 574         try:
 575           # Convert the grad to Tensor or IndexedSlices if necessary.
 576           g = ops.convert_to_tensor_or_indexed_slices(g)
 577         except TypeError:
 578           raise TypeError(
 579               "Gradient must be convertible to a Tensor"
 580               " or IndexedSlices, or None: %s" % g)
 581         if not isinstance(g, (ops.Tensor, ops.IndexedSlices)):
 582           raise TypeError(
 583               "Gradient must be a Tensor, IndexedSlices, or None: %s" % g)
 584       p = _get_processor(v)
 585       converted_grads_and_vars.append((g, v, p))
 586 
 587     converted_grads_and_vars = tuple(converted_grads_and_vars)
 588     var_list = [v for g, v, _ in converted_grads_and_vars if g is not None]
 589     if not var_list:
 590       raise ValueError("No gradients provided for any variable: %s." %
 591                        ([str(v) for _, v, _ in converted_grads_and_vars],))
 592     with ops.init_scope():
 593       self._create_slots(var_list)
 594     update_ops = []
 595     with ops.name_scope(name, self._name) as name:
 596       self._prepare()
 597       for grad, var, processor in converted_grads_and_vars:
 598         if grad is None:
 599           continue
 600         # We colocate all ops created in _apply_dense or _apply_sparse
 601         # on the same device as the variable.
 602         # TODO(apassos): figure out how to get the variable name here.
 603         if context.executing_eagerly() or isinstance(
 604             var,
 605             resource_variable_ops.ResourceVariable) and not var._in_graph_mode:  # pylint: disable=protected-access
 606           scope_name = ""
 607         else:
 608           scope_name = var.op.name
 609         with ops.name_scope("update_" + scope_name), ops.colocate_with(var):
 610           update_ops.append(processor.update_op(self, grad))
 611       if global_step is None:
 612         apply_updates = self._finish(update_ops, name)
 613       else:
 614         with ops.control_dependencies([self._finish(update_ops, "update")]):
 615           with ops.colocate_with(global_step):
 616             if isinstance(global_step, resource_variable_ops.ResourceVariable):
 617               # TODO(apassos): the implicit read in assign_add is slow; consider
 618               # making it less so.
 619               apply_updates = resource_variable_ops.assign_add_variable_op(
 620                   global_step.handle,
 621                   ops.convert_to_tensor(1, dtype=global_step.dtype),
 622                   name=name)
 623             else:
 624               apply_updates = state_ops.assign_add(global_step, 1, name=name)
 625 
 626       if not context.executing_eagerly():
 627         if isinstance(apply_updates, ops.Tensor):
 628           apply_updates = apply_updates.op
 629         train_op = ops.get_collection_ref(ops.GraphKeys.TRAIN_OP)
 630         if apply_updates not in train_op:
 631           train_op.append(apply_updates)
 632 
 633       return apply_updates
 634 
 635   def _distributed_apply(self,
 636                          distribution,
 637                          grads_and_vars,
 638                          global_step=None,
 639                          name=None):
 640     """A version of `apply_gradients` for cross-tower context.
 641 
 642     This is a version of `apply_gradients()` for when you are using a
 643     `DistributionStrategy` and are in a cross-tower context. If in a
 644     tower context, use `apply_gradients()` as normal.
 645 
 646     Args:
 647       distribution: A `DistributionStrategy` object.
 648       grads_and_vars: List of (gradient, variable) pairs as returned by
 649         `compute_gradients()`, and then aggregated across towers.
 650       global_step: Optional (mirrored) `Variable` to increment by one
 651         after the variables have been updated.
 652       name: Optional name for the returned operation.  Default to the
 653         name passed to the `Optimizer` constructor.
 654 
 655     Returns:
 656       An `Operation` that applies the specified gradients across all
 657       towers. If `global_step` was not None, that operation also
 658       increments `global_step`.
 659     """
 660     reduced_grads = distribution.batch_reduce(
 661         variable_scope.VariableAggregation.SUM, grads_and_vars)
 662     var_list = [v for _, v in grads_and_vars]
 663     grads_and_vars = zip(reduced_grads, var_list)
 664     # Note that this is called in a cross-tower context.
 665     self._create_slots(var_list)
 666 
 667     def update(v, g):
 668       """Apply gradients to a replica variable."""
 669       assert v is not None
 670 
 671       try:
 672         # Convert the grad to Tensor or IndexedSlices if necessary.
 673         g = ops.convert_to_tensor_or_indexed_slices(g)
 674       except TypeError:
 675         raise TypeError("Gradient must be convertible to a Tensor"
 676                         " or IndexedSlices, or None: %s" % g)
 677       if not isinstance(g, (ops.Tensor, ops.IndexedSlices)):
 678         raise TypeError(
 679             "Gradient must be a Tensor, IndexedSlices, or None: %s" % g)
 680       p = _get_processor(v)
 681 
 682       scope_name = "" if context.executing_eagerly() else v.op.name
 683       # device_policy is set because non-mirrored tensors will be read in
 684       # `update_op`. `_resource_apply_dense`, `lr_t`, `beta1_t` and `beta2_t`
 685       # is an example.
 686       with ops.name_scope("update_" + scope_name):
 687         return p.update_op(self, g)
 688 
 689     with ops.name_scope(name, self._name) as name:
 690       self._prepare()
 691 
 692       update_ops = [
 693           op
 694           for grad, var in grads_and_vars
 695           for op in distribution.update(var, update, grad, grouped=False)
 696       ]
 697 
 698       def finish(self, update_ops):
 699         return self._finish(update_ops, "update")
 700 
 701       non_slot_devices = distribution.non_slot_devices(var_list)
 702       finish_updates = distribution.update_non_slot(
 703           non_slot_devices, finish, self, update_ops, grouped=False)
 704       if global_step is None:
 705         apply_updates = distribution.group(finish_updates, name=name)
 706       else:
 707         with ops.control_dependencies(finish_updates):
 708           apply_updates = distribution.update(
 709               global_step, state_ops.assign_add, 1, name=name)
 710 
 711       if not context.executing_eagerly():
 712         if isinstance(apply_updates, ops.Tensor):
 713           apply_updates = apply_updates.op
 714         train_op = ops.get_collection_ref(ops.GraphKeys.TRAIN_OP)
 715         if apply_updates not in train_op:
 716           train_op.append(apply_updates)
 717 
 718       return apply_updates

 720   def get_slot(self, var, name):
 721     """Return a slot named `name` created for `var` by the Optimizer.
 722 
 723     Some `Optimizer` subclasses use additional variables.  For example
 724     `Momentum` and `Adagrad` use variables to accumulate updates.  This method
 725     gives access to these `Variable` objects if for some reason you need them.
 726 
 727     Use `get_slot_names()` to get the list of slot names created by the
 728     `Optimizer`.
 729 
 730     Args:
 731       var: A variable passed to `minimize()` or `apply_gradients()`.
 732       name: A string.
 733 
 734     Returns:
 735       The `Variable` for the slot if it was created, `None` otherwise.
 736     """
 737     # pylint: disable=protected-access
 738     named_slots = self._slots.get(name, None)
 739     if not named_slots:
 740       return None
 741 
 742     if hasattr(var, "_distributed_container"):
 743       # NOTE: If this isn't patched, then there is no `handle` in
 744       # `_resource_apply_dense`.
 745       distributed_container = var._distributed_container()
 746       assert distributed_container is not None
 747       if context.executing_eagerly():
 748         key = distributed_container._unique_id
 749       else:
 750         key = (distributed_container.graph, distributed_container._shared_name)
 751       # pylint: enable=protected-access
 752       mirrored_slot = named_slots.get(key, None)
 753       if mirrored_slot is None: return None
 754       return mirrored_slot.get(device=var.device)
 755 
 756     return named_slots.get(_var_key(var), None)

 768   def variables(self):
 769     """A list of variables which encode the current state of `Optimizer`.
 770 
 771     Includes slot variables and additional global variables created by the
 772     optimizer in the current default graph.
 773 
 774     Returns:
 775       A list of variables.
 776     """
 777     current_graph = ops.get_default_graph()
 778 
 779     def _from_current_graph(variable):
 780       if variable._in_graph_mode:  # pylint: disable=protected-access
 781         return variable.op.graph is current_graph
 782       else:
 783         # No variable.op in eager mode. We don't expect lots of eager graphs,
 784         # but behavior should be consistent with graph mode.
 785         return variable._graph_key == current_graph._graph_key  # pylint: disable=protected-access
 786 
 787     optimizer_variables = [v for v in self._non_slot_variables()
 788                            if _from_current_graph(v)]
 789     for _, variable_dict in self._slots.items():
 790       for _, slot_for_variable in variable_dict.items():
 791         if _from_current_graph(slot_for_variable):
 792           optimizer_variables.append(slot_for_variable)
 793     # Sort variables by name so that the return is deterministic.
 794     return sorted(optimizer_variables, key=lambda v: v.name)

 796   def _create_non_slot_variable(self, initial_value, name, colocate_with):
 797     """Add an extra variable, not associated with a slot."""
 798     # Recommendation: Use OptimizerV2 if your optimizer uses non-slot variables.
 799     eager = context.executing_eagerly()
 800     graph = None if eager else colocate_with.graph
 801 
 802     key = (name, graph)
 803     v = self._non_slot_dict.get(key, None)
 804     if v is None:
 805       self._maybe_initialize_checkpointable()
 806       distribution_strategy = (
 807           distribution_strategy_context.get_distribution_strategy())
 808       with distribution_strategy.colocate_vars_with(colocate_with):
 809         if eager:
 810           restored_initial_value = self._preload_simple_restoration(
 811               name=name, shape=None)
 812           if restored_initial_value is not None:
 813             initial_value = restored_initial_value
 814         v = variable_scope.variable(initial_value, name=name, trainable=False)
 815       # Restore this variable by name if necessary, but don't add a
 816       # Checkpointable dependency. Optimizers return the current graph's
 817       # non-slot variables from _checkpoint_dependencies explicitly rather
 818       # than unconditionally adding dependencies (since there may be multiple
 819       # non-slot variables with the same name in different graphs, trying to
 820       # save all of them would result in errors).
 821       self._handle_deferred_dependencies(name=name, checkpointable=v)
 822       self._non_slot_dict[key] = v
 823 
 824     return v

 826   @property
 827   def _checkpoint_dependencies(self):
 828     """From Checkpointable. Gather graph-specific non-slot variables to save."""
 829     current_graph_non_slot_variables = []
 830     current_graph_key = ops.get_default_graph()._graph_key  # pylint: disable=protected-access
 831     for (name, _), variable_object in sorted(self._non_slot_dict.items(),
 832                                              # Avoid comparing graphs
 833                                              key=lambda item: item[0][0]):
 834       if variable_object._graph_key == current_graph_key:  # pylint: disable=protected-access
 835         current_graph_non_slot_variables.append(
 836             checkpointable.CheckpointableReference(
 837                 name=name, ref=variable_object))
 838     return (super(Optimizer, self)._checkpoint_dependencies
 839             + current_graph_non_slot_variables)

 841   def _lookup_dependency(self, name):
 842     """From Checkpointable. Find a non-slot variable in the current graph."""
 843     unconditional = super(Optimizer, self)._lookup_dependency(name)
 844     if unconditional is not None:
 845       return unconditional
 846     graph = None if context.executing_eagerly() else ops.get_default_graph()
 847     return self._get_non_slot_variable(name, graph=graph)

 852       # This is a mirrored non-slot.  In order to enable code like `_finish`
 853       # to assign to a non-slot, return the current context replica.

 940   def _resource_apply_sparse_duplicate_indices(self, grad, handle, indices):
 941     """Add ops to apply sparse gradients to `handle`, with repeated indices.
 942 
 943     Optimizers which override this method must deal with repeated indices. See
 944     the docstring of `_apply_sparse_duplicate_indices` for details. By default
 945     the correct behavior, to sum non-unique indices and their associated
 946     gradients, is enforced by first pre-processing `grad` and `indices` and
 947     passing them on to `_resource_apply_sparse`. Optimizers which deal correctly
 948     with duplicate indices may instead override this method to avoid the
 949     overhead of summing.
 950 
 951     Args:
 952       grad: a `Tensor` representing the gradient for the affected indices.
 953       handle: a `Tensor` of dtype `resource` which points to the variable
 954        to be updated.
 955       indices: a `Tensor` of integral type representing the indices for
 956        which the gradient is nonzero. Indices may be repeated.
 957 
 958     Returns:
 959       An `Operation` which updates the value of the variable.
 960     """
 961     summed_grad, unique_indices = _deduplicate_indexed_slices(
 962         values=grad, indices=indices)
 963     return self._resource_apply_sparse(summed_grad, handle, unique_indices)

 985   def _apply_sparse_duplicate_indices(self, grad, var):
 986     """Add ops to apply sparse gradients to `var`, with repeated sparse indices.
 987 
 988     Optimizers which override this method must deal with IndexedSlices objects
 989     such as the following:
 990 
 991       IndexedSlicesValue(values=[1, 1], indices=[0, 0], dense_shape=[1])
 992 
 993     The correct interpretation is:
 994 
 995       IndexedSlicesValue(values=[2], indices=[0], dense_shape=[1])
 996 
 997     Many optimizers deal incorrectly with repeated indices when updating based
 998     on sparse gradients (e.g. summing squares rather than squaring the sum, or
 999     applying momentum terms multiple times). Adding first is always the correct
1000     behavior, so this is enforced here by reconstructing the IndexedSlices to
1001     have only unique indices, then calling _apply_sparse.
1002 
1003     Optimizers which deal correctly with repeated indices may instead override
1004     this method to avoid the overhead of summing indices.
1005 
1006     Args:
1007       grad: `IndexedSlices`.
1008       var: A `Variable` object.
1009 
1010     Returns:
1011       An `Operation`.
1012     """
1013     summed_values, unique_indices = _deduplicate_indexed_slices(
1014         values=grad.values, indices=grad.indices)
1015     gradient_no_duplicate_indices = ops.IndexedSlices(
1016         indices=unique_indices,
1017         values=summed_values,
1018         dense_shape=grad.dense_shape)
1019     return self._apply_sparse(gradient_no_duplicate_indices, var)

1150   def _restore_slot_variable(self, slot_name, variable, slot_variable):
1151     """Restore a newly created slot variable's value."""
1152     variable_key = _var_key(variable)
1153     deferred_restorations = self._deferred_slot_restorations.get(
1154         slot_name, {}).pop(variable_key, [])
1155     # Iterate over restores, highest restore UID first to minimize the number
1156     # of assignments.
1157     deferred_restorations.sort(key=lambda position: position.restore_uid,
1158                                reverse=True)
1159     for checkpoint_position in deferred_restorations:
1160       checkpoint_position.restore(slot_variable)

1162   def _create_or_restore_slot_variable(
1163       self, slot_variable_position, slot_name, variable):
1164     """Restore a slot variable's value, possibly creating it.
1165 
1166     Called when a variable which has an associated slot variable is created or
1167     restored. When executing eagerly, we create the slot variable with a
1168     restoring initializer.
1169 
1170     No new variables are created when graph building. Instead,
1171     _restore_slot_variable catches these after normal creation and adds restore
1172     ops to the graph. This method is nonetheless important when graph building
1173     for the case when a slot variable has already been created but `variable`
1174     has just been added to a dependency graph (causing us to realize that the
1175     slot variable needs to be restored).
1176 
1177     Args:
1178       slot_variable_position: A `checkpointable._CheckpointPosition` object
1179         indicating the slot variable `Checkpointable` object to be restored.
1180       slot_name: The name of this `Optimizer`'s slot to restore into.
1181       variable: The variable object this slot is being created for.
1182     """
1183     named_slots = self._slot_dict(slot_name)
1184     variable_key = _var_key(variable)
1185     slot_variable = named_slots.get(variable_key, None)
1186     if (slot_variable is None and context.executing_eagerly() and
1187         slot_variable_position.is_simple_variable()
1188         # Defer slot variable creation if there is an active variable creator
1189         # scope. Generally we'd like to eagerly create/restore slot variables
1190         # when possible, but this may mean that scopes intended to catch
1191         # `variable` also catch its eagerly created slot variable
1192         # unintentionally (specifically make_template would add a dependency on
1193         # a slot variable if not for this case). Deferring is mostly harmless
1194         # (aside from double initialization), and makes variable creator scopes
1195         # behave the same way they do when graph building.
1196         and not ops.get_default_graph()._variable_creator_stack):  # pylint: disable=protected-access
1197       initializer = checkpointable.CheckpointInitialValue(
1198           checkpoint_position=slot_variable_position)
1199       slot_variable = self._get_or_make_slot(
1200           var=variable,
1201           val=initializer,
1202           slot_name=slot_name,
1203           op_name=self._name)
1204       # Slot variables are not owned by any one object (because we don't want to
1205       # save the slot variable if the optimizer is saved without the non-slot
1206       # variable, or if the non-slot variable is saved without the optimizer;
1207       # it's a dependency hypergraph with edges of the form (optimizer, non-slot
1208       # variable, variable)). So we don't _track_ slot variables anywhere, and
1209       # instead special-case this dependency and otherwise pretend it's a normal
1210       # graph.
1211     if slot_variable is not None:
1212       # If we've either made this slot variable, or if we've pulled out an
1213       # existing slot variable, we should restore it.
1214       slot_variable_position.restore(slot_variable)
1215     else:
1216       # We didn't make the slot variable. Defer restoring until it gets created
1217       # normally. We keep a list rather than the one with the highest restore
1218       # UID in case slot variables have their own dependencies, in which case
1219       # those could differ between restores.
1220       self._deferred_slot_restorations.setdefault(
1221           slot_name, {}).setdefault(variable_key, []).append(
1222               slot_variable_position)
