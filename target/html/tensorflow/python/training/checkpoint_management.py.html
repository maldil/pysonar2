  58 @tf_export("train.generate_checkpoint_state_proto")
  59 def generate_checkpoint_state_proto(save_dir,
  60                                     model_checkpoint_path,
  61                                     all_model_checkpoint_paths=None,
  62                                     all_model_checkpoint_timestamps=None,
  63                                     last_preserved_timestamp=None):
  64   """Generates a checkpoint state proto.
  65 
  66   Args:
  67     save_dir: Directory where the model was saved.
  68     model_checkpoint_path: The checkpoint file.
  69     all_model_checkpoint_paths: List of strings.  Paths to all not-yet-deleted
  70       checkpoints, sorted from oldest to newest.  If this is a non-empty list,
  71       the last element must be equal to model_checkpoint_path.  These paths
  72       are also saved in the CheckpointState proto.
  73     all_model_checkpoint_timestamps: A list of floats, indicating the number of
  74       seconds since the Epoch when each checkpoint was generated.
  75     last_preserved_timestamp: A float, indicating the number of seconds since
  76       the Epoch when the last preserved checkpoint was written, e.g. due to a
  77       `keep_checkpoint_every_n_hours` parameter (see
  78       `tf.contrib.checkpoint.CheckpointManager` for an implementation).
  79   Returns:
  80     CheckpointState proto with model_checkpoint_path and
  81     all_model_checkpoint_paths updated to either absolute paths or
  82     relative paths to the current save_dir.
  83 
  84   Raises:
  85     ValueError: If `all_model_checkpoint_timestamps` was provided but its length
  86       does not match `all_model_checkpoint_paths`.
  87   """
  88   if all_model_checkpoint_paths is None:
  89     all_model_checkpoint_paths = []
  90 
  91   if (not all_model_checkpoint_paths or
  92       all_model_checkpoint_paths[-1] != model_checkpoint_path):
  93     logging.info("%s is not in all_model_checkpoint_paths. Manually adding it.",
  94                  model_checkpoint_path)
  95     all_model_checkpoint_paths.append(model_checkpoint_path)
  96 
  97   if (all_model_checkpoint_timestamps
  98       and (len(all_model_checkpoint_timestamps)
  99            != len(all_model_checkpoint_paths))):
 100     raise ValueError(
 101         ("Checkpoint timestamps, if provided, must match checkpoint paths (got "
 102          "paths %s and timestamps %s)")
 103         % (all_model_checkpoint_paths, all_model_checkpoint_timestamps))
 104 
 105   # Relative paths need to be rewritten to be relative to the "save_dir"
 106   # if model_checkpoint_path already contains "save_dir".
 107   if not os.path.isabs(save_dir):
 108     if not os.path.isabs(model_checkpoint_path):
 109       model_checkpoint_path = os.path.relpath(model_checkpoint_path, save_dir)
 110     for i in range(len(all_model_checkpoint_paths)):
 111       p = all_model_checkpoint_paths[i]
 112       if not os.path.isabs(p):
 113         all_model_checkpoint_paths[i] = os.path.relpath(p, save_dir)
 114 
 115   coord_checkpoint_proto = CheckpointState(
 116       model_checkpoint_path=model_checkpoint_path,
 117       all_model_checkpoint_paths=all_model_checkpoint_paths,
 118       all_model_checkpoint_timestamps=all_model_checkpoint_timestamps,
 119       last_preserved_timestamp=last_preserved_timestamp)
 120 
 121   return coord_checkpoint_proto

 166 def update_checkpoint_state_internal(save_dir,
 167                                      model_checkpoint_path,
 168                                      all_model_checkpoint_paths=None,
 169                                      latest_filename=None,
 170                                      save_relative_paths=False,
 171                                      all_model_checkpoint_timestamps=None,
 172                                      last_preserved_timestamp=None):
 173   """Updates the content of the 'checkpoint' file.
 174 
 175   This updates the checkpoint file containing a CheckpointState
 176   proto.
 177 
 178   Args:
 179     save_dir: Directory where the model was saved.
 180     model_checkpoint_path: The checkpoint file.
 181     all_model_checkpoint_paths: List of strings.  Paths to all not-yet-deleted
 182       checkpoints, sorted from oldest to newest.  If this is a non-empty list,
 183       the last element must be equal to model_checkpoint_path.  These paths
 184       are also saved in the CheckpointState proto.
 185     latest_filename: Optional name of the checkpoint file.  Default to
 186       'checkpoint'.
 187     save_relative_paths: If `True`, will write relative paths to the checkpoint
 188       state file.
 189     all_model_checkpoint_timestamps: Optional list of timestamps (floats,
 190       seconds since the Epoch) indicating when the checkpoints in
 191       `all_model_checkpoint_paths` were created.
 192     last_preserved_timestamp: A float, indicating the number of seconds since
 193       the Epoch when the last preserved checkpoint was written, e.g. due to a
 194       `keep_checkpoint_every_n_hours` parameter (see
 195       `tf.contrib.checkpoint.CheckpointManager` for an implementation).
 196 
 197   Raises:
 198     RuntimeError: If any of the model checkpoint paths conflict with the file
 199       containing CheckpointSate.
 200   """
 201   # Writes the "checkpoint" file for the coordinator for later restoration.
 202   coord_checkpoint_filename = _GetCheckpointFilename(save_dir, latest_filename)
 203   if save_relative_paths:
 204     if os.path.isabs(model_checkpoint_path):
 205       rel_model_checkpoint_path = os.path.relpath(
 206           model_checkpoint_path, save_dir)
 207     else:
 208       rel_model_checkpoint_path = model_checkpoint_path
 209     rel_all_model_checkpoint_paths = []
 210     for p in all_model_checkpoint_paths:
 211       if os.path.isabs(p):
 212         rel_all_model_checkpoint_paths.append(os.path.relpath(p, save_dir))
 213       else:
 214         rel_all_model_checkpoint_paths.append(p)
 215     ckpt = generate_checkpoint_state_proto(
 216         save_dir,
 217         rel_model_checkpoint_path,
 218         all_model_checkpoint_paths=rel_all_model_checkpoint_paths,
 219         all_model_checkpoint_timestamps=all_model_checkpoint_timestamps,
 220         last_preserved_timestamp=last_preserved_timestamp)
 221   else:
 222     ckpt = generate_checkpoint_state_proto(
 223         save_dir,
 224         model_checkpoint_path,
 225         all_model_checkpoint_paths=all_model_checkpoint_paths,
 226         all_model_checkpoint_timestamps=all_model_checkpoint_timestamps,
 227         last_preserved_timestamp=last_preserved_timestamp)
 228 
 229   if coord_checkpoint_filename == ckpt.model_checkpoint_path:
 230     raise RuntimeError("Save path '%s' conflicts with path used for "
 231                        "checkpoint state.  Please use a different save path." %
 232                        model_checkpoint_path)
 233 
 234   # Preventing potential read/write race condition by *atomically* writing to a
 235   # file.
 236   file_io.atomic_write_string_to_file(coord_checkpoint_filename,
 237                                       text_format.MessageToString(ckpt))

 240 @tf_export("train.get_checkpoint_state")
 241 def get_checkpoint_state(checkpoint_dir, latest_filename=None):
 242   """Returns CheckpointState proto from the "checkpoint" file.
 243 
 244   If the "checkpoint" file contains a valid CheckpointState
 245   proto, returns it.
 246 
 247   Args:
 248     checkpoint_dir: The directory of checkpoints.
 249     latest_filename: Optional name of the checkpoint file.  Default to
 250       'checkpoint'.
 251 
 252   Returns:
 253     A CheckpointState if the state was available, None
 254     otherwise.
 255 
 256   Raises:
 257     ValueError: if the checkpoint read doesn't have model_checkpoint_path set.
 258   """
 259   ckpt = None
 260   coord_checkpoint_filename = _GetCheckpointFilename(checkpoint_dir,
 261                                                      latest_filename)
 262   f = None
 263   try:
 264     # Check that the file exists before opening it to avoid
 265     # many lines of errors from colossus in the logs.
 266     if file_io.file_exists(coord_checkpoint_filename):
 267       file_content = file_io.read_file_to_string(
 268           coord_checkpoint_filename)
 269       ckpt = CheckpointState()
 270       text_format.Merge(file_content, ckpt)
 271       if not ckpt.model_checkpoint_path:
 272         raise ValueError("Invalid checkpoint state loaded from "
 273                          + checkpoint_dir)
 274       # For relative model_checkpoint_path and all_model_checkpoint_paths,
 275       # prepend checkpoint_dir.
 276       if not os.path.isabs(ckpt.model_checkpoint_path):
 277         ckpt.model_checkpoint_path = os.path.join(checkpoint_dir,
 278                                                   ckpt.model_checkpoint_path)
 279       for i in range(len(ckpt.all_model_checkpoint_paths)):
 280         p = ckpt.all_model_checkpoint_paths[i]
 281         if not os.path.isabs(p):
 282           ckpt.all_model_checkpoint_paths[i] = os.path.join(checkpoint_dir, p)
 283   except errors.OpError as e:
 284     # It's ok if the file cannot be read
 285     logging.warning("%s: %s", type(e).__name__, e)
 286     logging.warning("%s: Checkpoint ignored", coord_checkpoint_filename)
 287     return None
 288   except text_format.ParseError as e:
 289     logging.warning("%s: %s", type(e).__name__, e)
 290     logging.warning("%s: Checkpoint ignored", coord_checkpoint_filename)
 291     return None
 292   finally:
 293     if f:
 294       f.close()
 295   return ckpt

 298 def _prefix_to_checkpoint_path(prefix, format_version):
 299   """Returns the pathname of a checkpoint file, given the checkpoint prefix.
 300 
 301   For V1 checkpoint, simply returns the prefix itself (the data file).  For V2,
 302   returns the pathname to the index file.
 303 
 304   Args:
 305     prefix: a string, the prefix of a checkpoint.
 306     format_version: the checkpoint format version that corresponds to the
 307       prefix.
 308   Returns:
 309     The pathname of a checkpoint file, taking into account the checkpoint
 310       format version.
 311   """
 312   if format_version == saver_pb2.SaverDef.V2:
 313     return prefix + ".index"  # The index file identifies a checkpoint.
 314   return prefix  # Just the data file.

 317 @tf_export("train.latest_checkpoint")
 318 def latest_checkpoint(checkpoint_dir, latest_filename=None):
 319   """Finds the filename of latest saved checkpoint file.
 320 
 321   Args:
 322     checkpoint_dir: Directory where the variables were saved.
 323     latest_filename: Optional name for the protocol buffer file that
 324       contains the list of most recent checkpoint filenames.
 325       See the corresponding argument to `Saver.save()`.
 326 
 327   Returns:
 328     The full path to the latest checkpoint or `None` if no checkpoint was found.
 329   """
 330   # Pick the latest checkpoint based on checkpoint state.
 331   ckpt = get_checkpoint_state(checkpoint_dir, latest_filename)
 332   if ckpt and ckpt.model_checkpoint_path:
 333     # Look for either a V2 path or a V1 path, with priority for V2.
 334     v2_path = _prefix_to_checkpoint_path(ckpt.model_checkpoint_path,
 335                                          saver_pb2.SaverDef.V2)
 336     v1_path = _prefix_to_checkpoint_path(ckpt.model_checkpoint_path,
 337                                          saver_pb2.SaverDef.V1)
 338     if file_io.get_matching_files(v2_path) or file_io.get_matching_files(
 339         v1_path):
 340       return ckpt.model_checkpoint_path
 341     else:
 342       logging.error("Couldn't match files for checkpoint %s",
 343                     ckpt.model_checkpoint_path)
 344   return None

 347 @tf_export("train.checkpoint_exists")
 348 def checkpoint_exists(checkpoint_prefix):
 349   """Checks whether a V1 or V2 checkpoint exists with the specified prefix.
 350 
 351   This is the recommended way to check if a checkpoint exists, since it takes
 352   into account the naming difference between V1 and V2 formats.
 353 
 354   Args:
 355     checkpoint_prefix: the prefix of a V1 or V2 checkpoint, with V2 taking
 356       priority.  Typically the result of `Saver.save()` or that of
 357       `tf.train.latest_checkpoint()`, regardless of sharded/non-sharded or
 358       V1/V2.
 359   Returns:
 360     A bool, true iff a checkpoint referred to by `checkpoint_prefix` exists.
 361   """
 362   pathname = _prefix_to_checkpoint_path(checkpoint_prefix,
 363                                         saver_pb2.SaverDef.V2)
 364   if file_io.get_matching_files(pathname):
 365     return True
 366   elif file_io.get_matching_files(checkpoint_prefix):
 367     return True
 368   else:
 369     return False

 372 @tf_export("train.get_checkpoint_mtimes")
 373 def get_checkpoint_mtimes(checkpoint_prefixes):
 374   """Returns the mtimes (modification timestamps) of the checkpoints.
 375 
 376   Globs for the checkpoints pointed to by `checkpoint_prefixes`.  If the files
 377   exist, collect their mtime.  Both V2 and V1 checkpoints are considered, in
 378   that priority.
 379 
 380   This is the recommended way to get the mtimes, since it takes into account
 381   the naming difference between V1 and V2 formats.
 382 
 383   Args:
 384     checkpoint_prefixes: a list of checkpoint paths, typically the results of
 385       `Saver.save()` or those of `tf.train.latest_checkpoint()`, regardless of
 386       sharded/non-sharded or V1/V2.
 387   Returns:
 388     A list of mtimes (in microseconds) of the found checkpoints.
 389   """
 390   mtimes = []
 391 
 392   def match_maybe_append(pathname):
 393     fnames = file_io.get_matching_files(pathname)
 394     if fnames:
 395       mtimes.append(file_io.stat(fnames[0]).mtime_nsec / 1e9)
 396       return True
 397     return False
 398 
 399   for checkpoint_prefix in checkpoint_prefixes:
 400     # Tries V2's metadata file first.
 401     pathname = _prefix_to_checkpoint_path(checkpoint_prefix,
 402                                           saver_pb2.SaverDef.V2)
 403     if match_maybe_append(pathname):
 404       continue
 405     # Otherwise, tries V1, where the prefix is the complete pathname.
 406     match_maybe_append(checkpoint_prefix)
 407 
 408   return mtimes

 411 @tf_export("train.remove_checkpoint")
 412 def remove_checkpoint(checkpoint_prefix,
 413                       checkpoint_format_version=saver_pb2.SaverDef.V2,
 414                       meta_graph_suffix="meta"):
 415   """Removes a checkpoint given by `checkpoint_prefix`.
 416 
 417   Args:
 418     checkpoint_prefix: The prefix of a V1 or V2 checkpoint. Typically the result
 419       of `Saver.save()` or that of `tf.train.latest_checkpoint()`, regardless of
 420       sharded/non-sharded or V1/V2.
 421     checkpoint_format_version: `SaverDef.CheckpointFormatVersion`, defaults to
 422       `SaverDef.V2`.
 423     meta_graph_suffix: Suffix for `MetaGraphDef` file. Defaults to 'meta'.

 425   _delete_file_if_exists(
 426       meta_graph_filename(checkpoint_prefix, meta_graph_suffix))
 427   if checkpoint_format_version == saver_pb2.SaverDef.V2:
 428     # V2 has a metadata file and some data files.
 429     _delete_file_if_exists(checkpoint_prefix + ".index")
 430     _delete_file_if_exists(checkpoint_prefix + ".data-?????-of-?????")
 431   else:
 432     # V1, Legacy.  Exact match on the data file.
 433     _delete_file_if_exists(checkpoint_prefix)

 442 def meta_graph_filename(checkpoint_filename, meta_graph_suffix="meta"):
 443   """Returns the meta graph filename.
 444 
 445   Args:
 446     checkpoint_filename: Name of the checkpoint file.
 447     meta_graph_suffix: Suffix for `MetaGraphDef` file. Defaults to 'meta'.
 448 
 449   Returns:
 450     MetaGraph file name.
 451   """
 452   # If the checkpoint_filename is sharded, the checkpoint_filename could
 453   # be of format model.ckpt-step#-?????-of-shard#. For example,
 454   # model.ckpt-123456-?????-of-00005, or model.ckpt-123456-00001-of-00002.
 455   basename = re.sub(r"-[\d\?]+-of-\d+$", "", checkpoint_filename)
 456   suffixed_filename = ".".join([basename, meta_graph_suffix])
 457   return suffixed_filename

 460 # TODO(allenl): Allow tf.keras.Model instances in the constructor directly?
 461 class CheckpointManager(object):
 462   """Deletes old checkpoints.
 463 
 464   Example usage:
 465   ```python
 466   import tensorflow as tf
 467   checkpoint = tf.train.Checkpoint(optimizer=optimizer, model=model)
 468   manager = tf.contrib.checkpoint.CheckpointManager(
 469       checkpoint, directory="/tmp/model", max_to_keep=5)
 470   status = checkpoint.restore(manager.latest_checkpoint)
 471   while True:
 472     # train
 473     manager.save()
 474   ```
 475 
 476   `CheckpointManager` preserves its own state across instantiations (see the
 477   `__init__` documentation for details). Only one should be active in a
 478   particular directory at a time.
 479   """
 480 
 481   def __init__(self, checkpoint, directory,
 482                max_to_keep, keep_checkpoint_every_n_hours=None):
 483     """Configure a `CheckpointManager` for use in `directory`.
 484 
 485     If a `CheckpointManager` was previously used in `directory`, its
 486     state will be restored. This includes the list of managed checkpoints and
 487     the timestamp bookkeeping necessary to support
 488     `keep_checkpoint_every_n_hours`. The behavior of the new `CheckpointManager`
 489     will be the same as the previous `CheckpointManager`, including cleaning up
 490     existing checkpoints if appropriate.
 491 
 492     Checkpoints are only considered for deletion just after a new checkpoint has
 493     been added. At that point, `max_to_keep` checkpoints will remain in an
 494     "active set". Once a checkpoint is preserved by
 495     `keep_checkpoint_every_n_hours` it will not be deleted by this
 496     `CheckpointManager` or any future `CheckpointManager` instantiated in
 497     `directory` (regardless of the new setting of
 498     `keep_checkpoint_every_n_hours`). The `max_to_keep` checkpoints in the
 499     active set may be deleted by this `CheckpointManager` or a future
 500     `CheckpointManager` instantiated in `directory` (subject to its
 501     `max_to_keep` and `keep_checkpoint_every_n_hours` settings).
 502 
 503     Args:
 504       checkpoint: The `tf.train.Checkpoint` instance to save and manage
 505         checkpoints for.
 506       directory: The path to a directory in which to write checkpoints. A
 507         special file named "checkpoint" is also written to this directory (in a
 508         human-readable text format) which contains the state of the
 509         `CheckpointManager`.
 510       max_to_keep: An integer, the number of checkpoints to keep. Unless
 511         preserved by `keep_checkpoint_every_n_hours`, checkpoints will be
 512         deleted from the active set, oldest first, until only `max_to_keep`
 513         checkpoints remain. If `None`, no checkpoints are deleted and everything
 514         stays in the active set. Note that `max_to_keep=None` will keep all
 515         checkpoint paths in memory and in the checkpoint state protocol buffer
 516         on disk.
 517       keep_checkpoint_every_n_hours: Upon removal from the active set, a
 518         checkpoint will be preserved if it has been at least
 519         `keep_checkpoint_every_n_hours` since the last preserved checkpoint. The
 520         default setting of `None` does not preserve any checkpoints in this way.
 521 
 522     Raises:
 523       ValueError: If `max_to_keep` is not a positive integer.
 524     """
 525     self._checkpoint = checkpoint
 526     self._save_counter_assign = None
 527     if max_to_keep is not None and max_to_keep <= 0:
 528       raise ValueError(
 529           ("Expected a positive integer or `None` for `max_to_max_to_keep`, "
 530            "got %d.")
 531           % (max_to_keep,))
 532     self._max_to_keep = max_to_keep
 533     self._keep_checkpoint_every_n_hours = keep_checkpoint_every_n_hours
 534     self._directory = directory
 535     self._checkpoint_prefix = os.path.join(directory, "ckpt")
 536     recovered_state = get_checkpoint_state(directory)
 537     current_clock = time.time()
 538     self._maybe_delete = collections.OrderedDict()
 539     if recovered_state is None:
 540       self._latest_checkpoint = None
 541       # Set the clock back slightly to avoid race conditions when quckly
 542       # re-creating a CheckpointManager.
 543       self._last_preserved_timestamp = current_clock - 1.
 544     else:
 545       self._latest_checkpoint = recovered_state.model_checkpoint_path
 546       self._last_preserved_timestamp = recovered_state.last_preserved_timestamp
 547       if current_clock < self._last_preserved_timestamp:
 548         # Time seems to have reversed itself. In addition to this warning, we'll
 549         # min() saved checkpoint timestamps with the current time to ensure that
 550         # old checkpoints don't get deleted accidentally.
 551         logging.warning(
 552             ("time.time() returned a value %f seconds behind the last "
 553              "preserved checkpoint timestamp.")
 554             % (self._last_preserved_timestamp - current_clock,))
 555         self._last_preserved_timestamp = current_clock
 556       all_timestamps = recovered_state.all_model_checkpoint_timestamps
 557       all_paths = recovered_state.all_model_checkpoint_paths
 558       del recovered_state  # Uses modified values from now on
 559       if not all_timestamps:
 560         all_timestamps = [self._last_preserved_timestamp] * len(all_paths)
 561 
 562       for filename, timestamp in zip(all_paths, all_timestamps):
 563         timestamp = min(timestamp, current_clock)
 564         if timestamp > self._last_preserved_timestamp:
 565           self._maybe_delete[filename] = timestamp
 566 
 567   @property
 568   def latest_checkpoint(self):
 569     """The prefix of the most recent checkpoint in `directory`.
 570 
 571     Equivalent to `tf.train.latest_checkpoint(directory)` where `directory` is
 572     the constructor argument to `CheckpointManager`.
 573 
 574     Suitable for passing to `tf.train.Checkpoint.restore` to resume training.
 575 
 576     Returns:
 577       The checkpoint prefix. If there are no checkpoints, returns `None`.
 578     """
 579     return self._latest_checkpoint
 580 
 581   @property
 582   def checkpoints(self):
 583     """A list of managed checkpoints.
 584 
 585     Note that checkpoints saved due to `keep_checkpoint_every_n_hours` will not
 586     show up in this list (to avoid ever-growing filename lists).
 587 
 588     Returns:
 589       A list of filenames, sorted from oldest to newest.
 590     """
 591     return list(self._maybe_delete.keys())
 592 
 593   def _sweep(self):
 594     """Deletes or preserves managed checkpoints."""
 595     if not self._max_to_keep:
 596       # Does not update self._last_preserved_timestamp, since everything is kept
 597       # in the active set.
 598       return
 599     while len(self._maybe_delete) > self._max_to_keep:
 600       filename, timestamp = self._maybe_delete.popitem(last=False)
 601       # Even if we're keeping this checkpoint due to
 602       # keep_checkpoint_every_n_hours, we won't reference it to avoid
 603       # infinitely-growing CheckpointState protos.
 604       if (self._keep_checkpoint_every_n_hours
 605           and (timestamp - self._keep_checkpoint_every_n_hours * 3600.
 606                >= self._last_preserved_timestamp)):
 607         self._last_preserved_timestamp = timestamp
 608         continue
 609       remove_checkpoint(filename)
 610 
 611   def _record_state(self):
 612     """Saves the `CheckpointManager`'s state in `directory`."""
 613     filenames, timestamps = zip(*self._maybe_delete.items())
 614     update_checkpoint_state_internal(
 615         self._directory,
 616         model_checkpoint_path=self.latest_checkpoint,
 617         all_model_checkpoint_paths=filenames,
 618         all_model_checkpoint_timestamps=timestamps,
 619         last_preserved_timestamp=self._last_preserved_timestamp,
 620         save_relative_paths=True)
 621 
 622   @property
 623   def _prefix(self):
 624     """A common prefix for all checkpoints saved with this manager.
 625 
 626     For example, if `directory` (a constructor argument) were `"/tmp/tf-model"`,
 627     `prefix` would be `"/tmp/tf-model/ckpt"` and checkpoints would generally be
 628     numbered `"/tmp/tf-model/ckpt-1"`, `"/tmp/tf-model/ckpt-2"`, and so on. Each
 629     checkpoint has several associated files
 630     (e.g. `"/tmp/tf-model/ckpt-2.index"`).
 631 
 632     Returns:
 633       A string prefix.
 634     """
 635     return self._checkpoint_prefix
 636 
 637   def save(self, session=None, checkpoint_number=None):
 638     """Creates a new checkpoint and manages it.
 639 
 640     Args:
 641       session: The session to evaluate variables in. Ignored when executing
 642         eagerly. If not provided when graph building, the default session is
 643         used.
 644       checkpoint_number: An optional integer, or an integer-dtype `Variable` or
 645         `Tensor`, used to number the checkpoint. If `None` (default),
 646         checkpoints are numbered using `checkpoint.save_counter`. Even if
 647         `checkpoint_number` is provided, `save_counter` is still incremented. A
 648         user-provided `checkpoint_number` is not incremented even if it is a
 649         `Variable`.
 650 
 651     Returns:
 652       The path to the new checkpoint. It is also recorded in the `checkpoints`
 653       and `latest_checkpoint` properies.
 654     """
 655     # Save counter logic duplicated from tf.train.Checkpoint, soon to diverge
 656     # slightly with a custom numbering option.
 657     if context.executing_eagerly():
 658       save_counter = self._checkpoint.save_counter
 659       save_counter.assign_add(1)
 660     else:
 661       if session is None:
 662         session = ops.get_default_session()
 663 
 664       def _initializing_creator(next_creator, **kwargs):
 665         """Initialize the save counter if it has been newly created."""
 666         v = next_creator(**kwargs)
 667         session.run(v.initializer)
 668         return v
 669 
 670       with variable_scope.variable_creator_scope(_initializing_creator):
 671         save_counter = self._checkpoint.save_counter
 672       if self._save_counter_assign is None:
 673         self._save_counter_assign = save_counter.assign_add(1, read_value=False)
 674       session.run(self._save_counter_assign)
 675     if checkpoint_number is None:
 676       checkpoint_number = save_counter
 677     if not isinstance(checkpoint_number, compat.integral_types):
 678       checkpoint_number = training_util.global_step(
 679           sess=session, global_step_tensor=checkpoint_number)
 680     prefix = "%s-%d" % (self._prefix, checkpoint_number)
 681     save_path = self._checkpoint.write(prefix)
 682     timestamp = time.time()
 683     # If this is an overwritten checkpoint we were previously tracking, delete
 684     # and reinsert it to make sure it goes to the end of the queue.
 685     if save_path in self._maybe_delete:
 686       del self._maybe_delete[save_path]
 687     self._maybe_delete[save_path] = timestamp
 688     self._latest_checkpoint = save_path
 689     self._sweep()
 690     self._record_state()
 691     return save_path
