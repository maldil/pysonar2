  16 """Adadelta for TensorFlow."""
  17 from __future__ import absolute_import
  18 from __future__ import division
  19 from __future__ import print_function
  20 
  21 from tensorflow.python.framework import ops
  22 from tensorflow.python.ops import math_ops
  23 from tensorflow.python.training import optimizer
  24 from tensorflow.python.training import training_ops
  25 from tensorflow.python.util.tf_export import tf_export
  26 
  27 
  28 @tf_export("train.AdadeltaOptimizer")
  29 class AdadeltaOptimizer(optimizer.Optimizer):
  30   """Optimizer that implements the Adadelta algorithm.
  31 
  32   See [M. D. Zeiler](http://arxiv.org/abs/1212.5701)
  33   ([pdf](http://arxiv.org/pdf/1212.5701v1.pdf))
  34   """
  35 
  36   def __init__(self, learning_rate=0.001, rho=0.95, epsilon=1e-8,
  37                use_locking=False, name="Adadelta"):
  38     """Construct a new Adadelta optimizer.
  39 
  40     Args:
  41       learning_rate: A `Tensor` or a floating point value. The learning rate.
  42         To match the exact form in the original paper use 1.0.
  43       rho: A `Tensor` or a floating point value. The decay rate.
  44       epsilon: A `Tensor` or a floating point value.  A constant epsilon used
  45                to better conditioning the grad update.
  46       use_locking: If `True` use locks for update operations.
  47       name: Optional name prefix for the operations created when applying
  48         gradients.  Defaults to "Adadelta".
  49 
  50     @compatibility(eager)
  51     When eager execution is enabled, `learning_rate`, `rho`, and `epsilon` can
  52     each be a callable that takes no arguments and returns the actual value to
  53     use. This can be useful for changing these values across different
  54     invocations of optimizer functions.
  55     @end_compatibility
  56     """
  57     super(AdadeltaOptimizer, self).__init__(use_locking, name)
  58     self._lr = learning_rate
  59     self._rho = rho
  60     self._epsilon = epsilon
  61 
  62     # Tensor versions of the constructor arguments, created in _prepare().
  63     self._lr_t = None
  64     self._rho_t = None
  65     self._epsilon_t = None
  66 
  67   def _create_slots(self, var_list):
  68     for v in var_list:
  69       self._zeros_slot(v, "accum", self._name)
  70       self._zeros_slot(v, "accum_update", self._name)
  71 
  72   def _prepare(self):
  73     lr = self._call_if_callable(self._lr)
  74     rho = self._call_if_callable(self._rho)
  75     epsilon = self._call_if_callable(self._epsilon)
  76 
  77     self._lr_t = ops.convert_to_tensor(lr, name="lr")
  78     self._rho_t = ops.convert_to_tensor(rho, name="rho")
  79     self._epsilon_t = ops.convert_to_tensor(epsilon, name="epsilon")
  80 
  81   def _apply_dense(self, grad, var):
  82     accum = self.get_slot(var, "accum")
  83     accum_update = self.get_slot(var, "accum_update")
  84     return training_ops.apply_adadelta(
  85         var,
  86         accum,
  87         accum_update,
  88         math_ops.cast(self._lr_t, var.dtype.base_dtype),
  89         math_ops.cast(self._rho_t, var.dtype.base_dtype),
  90         math_ops.cast(self._epsilon_t, var.dtype.base_dtype),
  91         grad,
  92         use_locking=self._use_locking)
  93 
  94   def _resource_apply_dense(self, grad, var):
  95     accum = self.get_slot(var, "accum")
  96     accum_update = self.get_slot(var, "accum_update")
  97     return training_ops.resource_apply_adadelta(
  98         var.handle,
  99         accum.handle,
 100         accum_update.handle,
 101         math_ops.cast(self._lr_t, grad.dtype.base_dtype),
 102         math_ops.cast(self._rho_t, grad.dtype.base_dtype),
 103         math_ops.cast(self._epsilon_t, grad.dtype.base_dtype),
 104         grad,
 105         use_locking=self._use_locking)
 106 
 107   def _apply_sparse(self, grad, var):
 108     accum = self.get_slot(var, "accum")
 109     accum_update = self.get_slot(var, "accum_update")
 110     return training_ops.sparse_apply_adadelta(
 111         var,
 112         accum,
 113         accum_update,
 114         math_ops.cast(self._lr_t, var.dtype.base_dtype),
 115         math_ops.cast(self._rho_t, var.dtype.base_dtype),
 116         math_ops.cast(self._epsilon_t, var.dtype.base_dtype),
 117         grad.values,
 118         grad.indices,
 119         use_locking=self._use_locking)
 120 
 121   def _resource_apply_sparse(self, grad, var, indices):
 122     accum = self.get_slot(var, "accum")
 123     accum_update = self.get_slot(var, "accum_update")
 124     return training_ops.resource_sparse_apply_adadelta(
 125         var.handle,
 126         accum.handle,
 127         accum_update.handle,
 128         math_ops.cast(self._lr_t, grad.dtype),
 129         math_ops.cast(self._rho_t, grad.dtype),
 130         math_ops.cast(self._epsilon_t, grad.dtype),
 131         grad,
 132         indices,
 133         use_locking=self._use_locking)
