  32 @tf_export("train.exponential_decay", v1=[])
  33 def exponential_decay(learning_rate,
  34                       global_step,
  35                       decay_steps,
  36                       decay_rate,
  37                       staircase=False,
  38                       name=None):
  39   """Applies exponential decay to the learning rate.
  40 
  41   When training a model, it is often recommended to lower the learning rate as
  42   the training progresses.  This function applies an exponential decay function
  43   to a provided initial learning rate.  It requires a `global_step` value to
  44   compute the decayed learning rate.  You can just pass a TensorFlow variable
  45   that you increment at each training step.
  46 
  47   The function returns a no-arg function that produces the decayed learning
  48   rate. This can be useful for changing the learning rate value across
  49   different invocations of optimizer functions.
  50   It is computed as:
  51 
  52   ```python
  53   decayed_learning_rate = learning_rate *
  54                           decay_rate ^ (global_step / decay_steps)
  55   ```
  56 
  57   If the argument `staircase` is `True`, then `global_step / decay_steps` is an
  58   integer division and the decayed learning rate follows a staircase function.
  59 
  60   Example: decay every 100000 steps with a base of 0.96:
  61 
  62   ```python
  63   ...
  64   global_step = tf.Variable(0, trainable=False)
  65   starter_learning_rate = 0.1
  66   learning_rate_fn = tf.train.exponential_decay(starter_learning_rate,
  67                                                 global_step, 100000, 0.96,
  68                                                 staircase=True)
  69   # Passing global_step to minimize() will increment it at each step.
  70   learning_step = (
  71       tf.train.GradientDescentOptimizer(learning_rate_fn)
  72       .minimize(...my loss..., global_step=global_step)
  73   )
  74   ```
  75 
  76   Args:
  77     learning_rate: A scalar `float32` or `float64` `Tensor` or a
  78       Python number.  The initial learning rate.
  79     global_step: A scalar `int32` or `int64` `Tensor` or a Python number.
  80       Global step to use for the decay computation.  Must not be negative.
  81     decay_steps: A scalar `int32` or `int64` `Tensor` or a Python number.
  82       Must be positive.  See the decay computation above.
  83     decay_rate: A scalar `float32` or `float64` `Tensor` or a
  84       Python number.  The decay rate.
  85     staircase: Boolean.  If `True` decay the learning rate at discrete intervals
  86     name: String.  Optional name of the operation.  Defaults to
  87       'ExponentialDecay'.
  88 
  89   Returns:
  90     A no-arg function that outputs the decayed learning rate, a scalar `Tensor`
  91     of the same type as `learning_rate`.
  92 
  93   Raises:
  94     ValueError: if `global_step` is not supplied.
  95   """
  96   if global_step is None:
  97     raise ValueError("global_step is required for exponential_decay.")
  98   def decayed_lr(learning_rate, global_step, decay_steps, decay_rate,
  99                  staircase, name):
 100     """Helper to recompute learning rate; most helpful in eager-mode."""
 101     with ops.name_scope(
 102         name, "ExponentialDecay",
 103         [learning_rate, global_step, decay_steps, decay_rate]) as name:
 104       learning_rate = ops.convert_to_tensor(learning_rate, name="learning_rate")
 105       dtype = learning_rate.dtype
 106       decay_steps = math_ops.cast(decay_steps, dtype)
 107       decay_rate = math_ops.cast(decay_rate, dtype)
 108 
 109       global_step_recomp = math_ops.cast(global_step, dtype)
 110       p = global_step_recomp / decay_steps
 111       if staircase:
 112         p = math_ops.floor(p)
 113       return math_ops.multiply(
 114           learning_rate, math_ops.pow(decay_rate, p), name=name)
 115 
 116   return functools.partial(decayed_lr, learning_rate, global_step, decay_steps,
 117                            decay_rate, staircase, name)

 120 @tf_export("train.piecewise_constant", v1=[])
 121 def piecewise_constant(x, boundaries, values, name=None):
 122   """Piecewise constant from boundaries and interval values.
 123 
 124   This function returns a no-arg callable to compute the piecewise constant.
 125   This can be useful for changing the learning rate value across
 126   different invocations of optimizer functions.
 127 
 128   Example: use a learning rate that's 1.0 for the first 100001 steps, 0.5
 129     for the next 10000 steps, and 0.1 for any additional steps.
 130 
 131   ```python
 132   global_step = tf.Variable(0, trainable=False)
 133   boundaries = [100000, 110000]
 134   values = [1.0, 0.5, 0.1]
 135   learning_rate_fn = tf.train.piecewise_constant(global_step, boundaries,
 136     values)
 137   learning_rate = learning_rate_fn()
 138 
 139   # Later, whenever we perform an optimization step, we increment global_step.
 140   ```
 141 
 142   Args:
 143     x: A 0-D scalar `Tensor`. Must be one of the following types: `float32`,
 144       `float64`, `uint8`, `int8`, `int16`, `int32`, `int64`.
 145     boundaries: A list of `Tensor`s or `int`s or `float`s with strictly
 146       increasing entries, and with all elements having the same type as `x`.
 147     values: A list of `Tensor`s or `float`s or `int`s that specifies the values
 148       for the intervals defined by `boundaries`. It should have one more element
 149       than `boundaries`, and all elements should have the same type.
 150     name: A string. Optional name of the operation. Defaults to
 151       'PiecewiseConstant'.
 152 
 153   Returns:
 154     A no-arg function that outputs a 0-D Tensor. The output of the no-arg
 155     function is `values[0]` when `x <= boundaries[0]`,
 156     `values[1]` when `x > boundaries[0]` and `x <= boundaries[1]`, ...,
 157     and values[-1] when `x > boundaries[-1]`.
 158 
 159   Raises:
 160     ValueError: if types of `x` and `boundaries` do not match, or types of all
 161         `values` do not match or
 162         the number of elements in the lists does not match.
 163   """
 164   if len(boundaries) != len(values) - 1:
 165     raise ValueError(
 166         "The length of boundaries should be 1 less than the length of values")
 167   def decayed_lr(x, boundaries, values, name):
 168     """Helper to recompute learning rate; most helpful in eager-mode."""
 169     with ops.name_scope(name, "PiecewiseConstant",
 170                         [x, boundaries, values, name]) as name:
 171       boundaries = ops.convert_n_to_tensor(boundaries)
 172       values = ops.convert_n_to_tensor(values)
 173       x_recomp = ops.convert_to_tensor(x)
 174       # Avoid explicit conversion to x's dtype. This could result in faulty
 175       # comparisons, for example if floats are converted to integers.
 176       for i, b in enumerate(boundaries):
 177         if b.dtype.base_dtype != x_recomp.dtype.base_dtype:
 178           # We can promote int32 boundaries to int64 without loss of precision.
 179           # This covers the most common case where the user passes in boundaries
 180           # as an array of Python integers.
 181           if (b.dtype.base_dtype == dtypes.int32 and
 182               x_recomp.dtype.base_dtype == dtypes.int64):
 183             b = math_ops.cast(b, x_recomp.dtype.base_dtype)
 184             boundaries[i] = b
 185           else:
 186             raise ValueError(
 187                 "Boundaries (%s) must have the same dtype as x (%s)." %
 188                 (b.dtype.base_dtype, x_recomp.dtype.base_dtype))
 189       # TODO(rdipietro): Ensure that boundaries' elements strictly increases.
 190       for v in values[1:]:
 191         if v.dtype.base_dtype != values[0].dtype.base_dtype:
 192           raise ValueError(
 193               "Values must have elements all with the same dtype (%s vs %s)." %
 194               (values[0].dtype.base_dtype, v.dtype.base_dtype))
 195       pred_fn_pairs = []
 196       pred_fn_pairs.append((x_recomp <= boundaries[0], lambda: values[0]))
 197       pred_fn_pairs.append((x_recomp > boundaries[-1], lambda: values[-1]))
 198       for low, high, v in zip(boundaries[:-1], boundaries[1:], values[1:-1]):
 199         # Need to bind v here; can do this with lambda v=v: ...
 200         pred = (x_recomp > low) & (x_recomp <= high)
 201         pred_fn_pairs.append((pred, lambda v=v: v))
 202 
 203       # The default isn't needed here because our conditions are mutually
 204       # exclusive and exhaustive, but tf.case requires it.
 205       default = lambda: values[0]
 206       return control_flow_ops.case(pred_fn_pairs, default, exclusive=True)
 207 
 208   return functools.partial(decayed_lr, x, boundaries, values, name)

 211 @tf_export("train.polynomial_decay", v1=[])
 212 def polynomial_decay(learning_rate,
 213                      global_step,
 214                      decay_steps,
 215                      end_learning_rate=0.0001,
 216                      power=1.0,
 217                      cycle=False,
 218                      name=None):
 219   """Applies a polynomial decay to the learning rate.
 220 
 221   It is commonly observed that a monotonically decreasing learning rate, whose
 222   degree of change is carefully chosen, results in a better performing model.
 223   This function applies a polynomial decay function to a provided initial
 224   `learning_rate` to reach an `end_learning_rate` in the given `decay_steps`.
 225 
 226   It requires a `global_step` value to compute the decayed learning rate.  You
 227   can just pass a TensorFlow variable that you increment at each training step.
 228 
 229   The function returns a no-arg callable that outputs the decayed learning
 230   rate. This can be useful for changing the learning rate value across
 231   different invocations of optimizer functions. It is computed as:
 232 
 233   ```python
 234   global_step = min(global_step, decay_steps)
 235   decayed_learning_rate = (learning_rate - end_learning_rate) *
 236                           (1 - global_step / decay_steps) ^ (power) +
 237                           end_learning_rate
 238 
 239   ```
 240 
 241   If `cycle` is True then a multiple of `decay_steps` is used, the first one
 242   that is bigger than `global_steps`.
 243 
 244   ```python
 245   decay_steps = decay_steps * ceil(global_step / decay_steps)
 246   decayed_learning_rate_fn = (learning_rate - end_learning_rate) *
 247                           (1 - global_step / decay_steps) ^ (power) +
 248                           end_learning_rate
 249   decayed_learning_rate = decayed_learning_rate_fn()
 250 
 251   ```
 252 
 253   Example: decay from 0.1 to 0.01 in 10000 steps using sqrt (i.e. power=0.5):
 254 
 255   ```python
 256   ...
 257   global_step = tf.Variable(0, trainable=False)
 258   starter_learning_rate = 0.1
 259   end_learning_rate = 0.01
 260   decay_steps = 10000
 261   learning_rate_fn = tf.train.polynomial_decay(starter_learning_rate,
 262                                                global_step, decay_steps,
 263                                                end_learning_rate,
 264                                                power=0.5)
 265   # Passing global_step to minimize() will increment it at each step.
 266   learning_step = (
 267       tf.train.GradientDescentOptimizer(learning_rate_fn)
 268       .minimize(...my loss..., global_step=global_step)
 269   )
 270   ```
 271 
 272   Args:
 273     learning_rate: A scalar `float32` or `float64` `Tensor` or a
 274       Python number.  The initial learning rate.
 275     global_step: A scalar `int32` or `int64` `Tensor` or a Python number.
 276       Global step to use for the decay computation.  Must not be negative.
 277     decay_steps: A scalar `int32` or `int64` `Tensor` or a Python number.
 278       Must be positive.  See the decay computation above.
 279     end_learning_rate: A scalar `float32` or `float64` `Tensor` or a
 280       Python number.  The minimal end learning rate.
 281     power: A scalar `float32` or `float64` `Tensor` or a
 282       Python number.  The power of the polynomial. Defaults to linear, 1.0.
 283     cycle: A boolean, whether or not it should cycle beyond decay_steps.
 284     name: String.  Optional name of the operation. Defaults to
 285       'PolynomialDecay'.
 286 
 287   Returns:
 288     A no-arg function that outputs the decayed learning rate, a scalar `Tensor`
 289     of the same type as `learning_rate`.
 290 
 291   Raises:
 292     ValueError: if `global_step` is not supplied.
 293   """
 294   if global_step is None:
 295     raise ValueError("global_step is required for polynomial_decay.")
 296   def decayed_lr(learning_rate, global_step, decay_steps, end_learning_rate,
 297                  power, cycle, name):
 298     """Helper to recompute learning rate; most helpful in eager-mode."""
 299     with ops.name_scope(
 300         name, "PolynomialDecay",
 301         [learning_rate, global_step, decay_steps, end_learning_rate, power]
 302     ) as name:
 303       learning_rate = ops.convert_to_tensor(learning_rate, name="learning_rate")
 304       dtype = learning_rate.dtype
 305       end_learning_rate = math_ops.cast(end_learning_rate, dtype)
 306       power = math_ops.cast(power, dtype)
 307 
 308       global_step_recomp = math_ops.cast(global_step, dtype)
 309       decay_steps_recomp = math_ops.cast(decay_steps, dtype)
 310       if cycle:
 311         # Find the first multiple of decay_steps that is bigger than
 312         # global_step. If global_step is zero set the multiplier to 1
 313         multiplier = control_flow_ops.cond(
 314             math_ops.equal(global_step_recomp, 0), lambda: 1.0,
 315             lambda: math_ops.ceil(global_step_recomp / decay_steps))
 316         decay_steps_recomp = math_ops.multiply(decay_steps_recomp, multiplier)
 317       else:
 318         # Make sure that the global_step used is not bigger than decay_steps.
 319         global_step_recomp = math_ops.minimum(global_step_recomp, decay_steps)
 320 
 321       p = math_ops.div(global_step_recomp, decay_steps_recomp)
 322       return math_ops.add(
 323           math_ops.multiply(learning_rate - end_learning_rate,
 324                             math_ops.pow(1 - p, power)),
 325           end_learning_rate,
 326           name=name)
 327 
 328   return functools.partial(
 329       decayed_lr, learning_rate, global_step, decay_steps, end_learning_rate,
 330       power, cycle, name)

 333 @tf_export("train.natural_exp_decay", v1=[])
 334 def natural_exp_decay(learning_rate,
 335                       global_step,
 336                       decay_steps,
 337                       decay_rate,
 338                       staircase=False,
 339                       name=None):
 340   """Applies natural exponential decay to the initial learning rate.
 341 
 342   When training a model, it is often recommended to lower the learning rate as
 343   the training progresses.  This function applies an exponential decay function
 344   to a provided initial learning rate.  It requires an `global_step` value to
 345   compute the decayed learning rate.  You can just pass a TensorFlow variable
 346   that you increment at each training step.
 347 
 348   The function returns a no-arg callable that produces the decayed learning
 349   rate. This can be useful for changing the learning rate value across
 350   different invocations of optimizer functions. It is computed as:
 351 
 352   ```python
 353   decayed_learning_rate = learning_rate * exp(-decay_rate * global_step /
 354   decay_step)
 355   ```
 356 
 357   or, if `staircase` is `True`, as:
 358 
 359   ```python
 360   decayed_learning_rate = learning_rate * exp(-decay_rate * floor(global_step /
 361   decay_step))
 362   ```
 363 
 364   Example: decay exponentially with a base of 0.96:
 365 
 366   ```python
 367   ...
 368   global_step = tf.Variable(0, trainable=False)
 369   learning_rate = 0.1
 370   decay_steps = 5
 371   k = 0.5
 372   learning_rate_fn = tf.train.natural_exp_decay(learning_rate, global_step,
 373                                                 decay_steps, k)
 374 
 375   # Passing global_step to minimize() will increment it at each step.
 376   learning_step = (
 377       tf.train.GradientDescentOptimizer(learning_rate_fn)
 378       .minimize(...my loss..., global_step=global_step)
 379   )
 380   ```
 381 
 382   Args:
 383     learning_rate: A scalar `float32` or `float64` `Tensor` or a
 384       Python number.  The initial learning rate.
 385     global_step: A Python number.
 386       Global step to use for the decay computation.  Must not be negative.
 387     decay_steps: How often to apply decay.
 388     decay_rate: A Python number.  The decay rate.
 389     staircase: Whether to apply decay in a discrete staircase, as opposed to
 390       continuous, fashion.
 391     name: String.  Optional name of the operation.  Defaults to
 392       'ExponentialTimeDecay'.
 393 
 394   Returns:
 395     A no-arg function that outputs the decayed learning rate, a scalar `Tensor`
 396     of the same type as `learning_rate`.
 397 
 398   Raises:
 399     ValueError: if `global_step` is not supplied.
 400   """
 401   if global_step is None:
 402     raise ValueError("global_step is required for natural_exp_decay.")
 403   def decayed_lr(learning_rate, global_step, decay_steps, decay_rate, staircase,
 404                  name):
 405     """Helper to recompute learning rate; most helpful in eager-mode."""
 406     with ops.name_scope(name, "NaturalExpDecay",
 407                         [learning_rate, global_step, decay_rate]) as name:
 408       learning_rate = ops.convert_to_tensor(learning_rate, name="learning_rate")
 409       dtype = learning_rate.dtype
 410       decay_steps = math_ops.cast(decay_steps, dtype)
 411       decay_rate = math_ops.cast(decay_rate, dtype)
 412 
 413       global_step_recomp = math_ops.cast(global_step, dtype)
 414       p = global_step_recomp / decay_steps
 415       if staircase:
 416         p = math_ops.floor(p)
 417       exponent = math_ops.exp(
 418           math_ops.multiply(math_ops.negative(decay_rate), p))
 419       return math_ops.multiply(learning_rate, exponent, name=name)
 420 
 421   return functools.partial(decayed_lr, learning_rate, global_step, decay_steps,
 422                            decay_rate, staircase, name)

 428                        decay_steps,

 430                        staircase=False,
 431                        name=None):
 432   """Applies inverse time decay to the initial learning rate.
 433 
 434   When training a model, it is often recommended to lower the learning rate as

 465   decay_steps, decay_rate)
 466 
 467   # Passing global_step to minimize() will increment it at each step.

 471   )
 472   ```

 475     learning_rate: A scalar `float32` or `float64` `Tensor` or a
 476       Python number.  The initial learning rate.
 477     global_step: A Python number.
 478       Global step to use for the decay computation.  Must not be negative.
 479     decay_steps: How often to apply decay.
 480     decay_rate: A Python number.  The decay rate.
 481     staircase: Whether to apply decay in a discrete staircase, as opposed to
 482       continuous, fashion.
 483     name: String.  Optional name of the operation.  Defaults to
 484       'InverseTimeDecay'.
 485 
 486   Returns:
 487     A no-arg function that outputs the decayed learning rate, a scalar `Tensor`
 488     of the same type as `learning_rate`.
 489 
 490   Raises:
 491     ValueError: if `global_step` is not supplied.
 492   """
 493   if global_step is None:
 494     raise ValueError("global_step is required for inverse_time_decay.")
 495   def decayed_lr(learning_rate, global_step, decay_steps, decay_rate, staircase,
 496                  name):
 497     """Helper to recompute learning rate; most helpful in eager-mode."""
 498     with ops.name_scope(name, "InverseTimeDecay",
 499                         [learning_rate, global_step, decay_rate]) as name:
 500       learning_rate = ops.convert_to_tensor(learning_rate, name="learning_rate")
 501       dtype = learning_rate.dtype
 502       decay_steps = math_ops.cast(decay_steps, dtype)
 503       decay_rate = math_ops.cast(decay_rate, dtype)
 504 
 505       global_step_recomp = math_ops.cast(global_step, dtype)
 506       p = global_step_recomp / decay_steps
 507       if staircase:
 508         p = math_ops.floor(p)

 510       denom = math_ops.add(const, math_ops.multiply(decay_rate, p))
 511       return math_ops.div(learning_rate, denom, name=name)
 512 
 513   return functools.partial(decayed_lr, learning_rate, global_step, decay_steps,
 514                            decay_rate, staircase, name)

 517 @tf_export("train.cosine_decay", v1=[])
 518 def cosine_decay(learning_rate, global_step, decay_steps, alpha=0.0,
 519                  name=None):
 520   """Applies cosine decay to the learning rate.
 521 
 522   See [Loshchilov & Hutter, ICLR2016], SGDR: Stochastic Gradient Descent
 523   with Warm Restarts. https://arxiv.org/abs/1608.03983
 524 
 525   When training a model, it is often recommended to lower the learning rate as
 526   the training progresses.  This function applies a cosine decay function
 527   to a provided initial learning rate.  It requires a `global_step` value to
 528   compute the decayed learning rate.  You can just pass a TensorFlow variable
 529   that you increment at each training step.
 530 
 531   The function returns a no-arg callable that produces the decayed learning
 532   rate. This can be useful for changing the learning rate value across
 533   different invocations of optimizer functions. It is computed as:
 534 
 535   ```python
 536   global_step = min(global_step, decay_steps)
 537   cosine_decay = 0.5 * (1 + cos(pi * global_step / decay_steps))
 538   decayed = (1 - alpha) * cosine_decay + alpha
 539   decayed_learning_rate = learning_rate * decayed
 540   ```
 541 
 542   Example usage:
 543   ```python
 544   decay_steps = 1000
 545   lr_decayed_fn = tf.train.cosine_decay(learning_rate, global_step, decay_steps)
 546   ```
 547 
 548   Args:
 549     learning_rate: A scalar `float32` or `float64` Tensor or a Python number.
 550       The initial learning rate.
 551     global_step: A scalar `int32` or `int64` `Tensor` or a Python number.
 552       Global step to use for the decay computation.
 553     decay_steps: A scalar `int32` or `int64` `Tensor` or a Python number.
 554       Number of steps to decay over.
 555     alpha: A scalar `float32` or `float64` Tensor or a Python number.
 556       Minimum learning rate value as a fraction of learning_rate.
 557     name: String. Optional name of the operation.  Defaults to 'CosineDecay'.
 558   Returns:
 559     A no-arg function that outputs the decayed learning rate, a scalar `Tensor`
 560     of the same type as `learning_rate`.
 561   Raises:
 562     ValueError: if `global_step` is not supplied.
 563   """
 564   if global_step is None:
 565     raise ValueError("cosine decay requires global_step")
 566   def decayed_lr(learning_rate, global_step, decay_steps, alpha, name):
 567     """Helper to recompute learning rate; most helpful in eager-mode."""
 568     with ops.name_scope(name, "CosineDecay",
 569                         [learning_rate, global_step]) as name:

 587 @tf_export("train.cosine_decay_restarts", v1=[])
 588 def cosine_decay_restarts(learning_rate,
 589                           global_step,
 590                           first_decay_steps,
 591                           t_mul=2.0,
 592                           m_mul=1.0,
 593                           alpha=0.0,
 594                           name=None):
 595   """Applies cosine decay with restarts to the learning rate.
 596 
 597   See [Loshchilov & Hutter, ICLR2016], SGDR: Stochastic Gradient Descent
 598   with Warm Restarts. https://arxiv.org/abs/1608.03983
 599 
 600   When training a model, it is often recommended to lower the learning rate as
 601   the training progresses.  This function applies a cosine decay function with
 602   restarts to a provided initial learning rate.  It requires a `global_step`
 603   value to compute the decayed learning rate.  You can just pass a TensorFlow
 604   variable that you increment at each training step.
 605 
 606   The function returns a no-arg callable that produces the decayed learning
 607   rate while taking into account possible warm restarts. This can be useful for
 608   changing the learning rate value across different invocations of optimizer
 609   functions.
 610 
 611   The learning rate multiplier first decays
 612   from 1 to `alpha` for `first_decay_steps` steps. Then, a warm
 613   restart is performed. Each new warm restart runs for `t_mul` times more steps
 614   and with `m_mul` times smaller initial learning rate.
 615 
 616   Example usage:
 617   ```python
 618   first_decay_steps = 1000
 619   lr_decayed_fn = tf.train.cosine_decay_restarts(learning_rate, global_step,
 620                                      first_decay_steps)
 621   ```
 622 
 623   Args:
 624     learning_rate: A scalar `float32` or `float64` Tensor or a Python number.
 625       The initial learning rate.
 626     global_step: A scalar `int32` or `int64` `Tensor` or a Python number.
 627       Global step to use for the decay computation.
 628     first_decay_steps: A scalar `int32` or `int64` `Tensor` or a Python number.
 629       Number of steps to decay over.
 630     t_mul: A scalar `float32` or `float64` `Tensor` or a Python number.
 631       Used to derive the number of iterations in the i-th period
 632     m_mul: A scalar `float32` or `float64` `Tensor` or a Python number.
 633       Used to derive the initial learning rate of the i-th period:
 634     alpha: A scalar `float32` or `float64` Tensor or a Python number.
 635       Minimum learning rate value as a fraction of the learning_rate.
 636     name: String. Optional name of the operation.  Defaults to 'SGDRDecay'.
 637   Returns:
 638     A no-arg function that outputs the decayed learning rate, a scalar `Tensor`
 639     of the same type as `learning_rate`.
 640 
 641   Raises:
 642     ValueError: if `global_step` is not supplied.
 643   """
 644   if global_step is None:
 645     raise ValueError("cosine decay restarts requires global_step")
 646   def decayed_lr(learning_rate, global_step, first_decay_steps, t_mul, m_mul,
 647                  alpha, name):
 648     """Helper to recompute learning rate; most helpful in eager-mode."""
 649     with ops.name_scope(name, "SGDRDecay", [learning_rate, global_step]
 650                        ) as name:
 651       learning_rate = ops.convert_to_tensor(
 652           learning_rate, name="initial_learning_rate")
 653       dtype = learning_rate.dtype
 654       first_decay_steps = math_ops.cast(first_decay_steps, dtype)
 655       alpha = math_ops.cast(alpha, dtype)
 656       t_mul = math_ops.cast(t_mul, dtype)
 657       m_mul = math_ops.cast(m_mul, dtype)
 658 
 659       global_step_recomp = math_ops.cast(global_step, dtype)
 660       completed_fraction = global_step_recomp / first_decay_steps
 661 
 662       def compute_step(completed_fraction, geometric=False):
 663         """Helper for `cond` operation."""
 664         if geometric:
 665           i_restart = math_ops.floor(
 666               math_ops.log(1.0 - completed_fraction * (1.0 - t_mul)) /
 667               math_ops.log(t_mul))
 668 
 669           sum_r = (1.0 - t_mul**i_restart) / (1.0 - t_mul)
 670           completed_fraction = (completed_fraction - sum_r) / t_mul**i_restart
 671 
 672         else:
 673           i_restart = math_ops.floor(completed_fraction)
 674           completed_fraction -= i_restart
 675 
 676         return i_restart, completed_fraction
 677 
 678       i_restart, completed_fraction = control_flow_ops.cond(
 679           math_ops.equal(t_mul, 1.0),
 680           lambda: compute_step(completed_fraction, geometric=False),
 681           lambda: compute_step(completed_fraction, geometric=True))
 682 
 683       m_fac = m_mul**i_restart
 684       cosine_decayed = 0.5 * m_fac * (1.0 + math_ops.cos(
 685           constant_op.constant(math.pi) * completed_fraction))
 686       decayed = (1 - alpha) * cosine_decayed + alpha
 687 
 688       return math_ops.multiply(learning_rate, decayed, name=name)
 689 
 690   return functools.partial(decayed_lr, learning_rate, global_step,
 691                            first_decay_steps, t_mul, m_mul, alpha, name)

 694 @tf_export("train.linear_cosine_decay", v1=[])

 703

 705   https://arxiv.org/abs/1709.07417
 706 
 707   For the idea of warm starts here controlled by `num_periods`,
 708   see [Loshchilov & Hutter, ICLR2016] SGDR: Stochastic Gradient Descent
 709   with Warm Restarts. https://arxiv.org/abs/1608.03983
 710 
 711   Note that linear cosine decay is more aggressive than cosine decay and
 712   larger initial learning rates can typically be used.
 713

 722   different invocations of optimizer functions. It is computed as:

 734   ```python

 745     decay_steps: A scalar `int32` or `int64` `Tensor` or a Python number.

 747     num_periods: Number of periods in the cosine part of the decay.
 748       See computation above.
 749     alpha: See computation above.
 750     beta: See computation above.
 751     name: String.  Optional name of the operation.  Defaults to
 752       'LinearCosineDecay'.

 754     A no-arg function that outputs the decayed learning rate, a scalar `Tensor`

 756   Raises:
 757     ValueError: if `global_step` is not supplied.
 758   """

 760     raise ValueError("linear cosine decay requires global_step")
 761   def decayed_lr(learning_rate, global_step, decay_steps, num_periods, alpha,
 762                  beta, name):
 763     """Helper to recompute learning rate; most helpful in eager-mode."""
 764     with ops.name_scope(name, "LinearCosineDecay",

 768       decay_steps = math_ops.cast(decay_steps, dtype)
 769       num_periods = math_ops.cast(num_periods, dtype)

 773       global_step_recomp = math_ops.cast(global_step, dtype)

 788 @tf_export("train.noisy_linear_cosine_decay", v1=[])
 789 def noisy_linear_cosine_decay(learning_rate,
 790                               global_step,
 791                               decay_steps,
 792                               initial_variance=1.0,
 793                               variance_decay=0.55,
 794                               num_periods=0.5,
 795                               alpha=0.0,
 796                               beta=0.001,
 797                               name=None):
 798   """Applies noisy linear cosine decay to the learning rate.
 799 
 800   See [Bello et al., ICML2017] Neural Optimizer Search with RL.
 801   https://arxiv.org/abs/1709.07417
 802 
 803   For the idea of warm starts here controlled by `num_periods`,
 804   see [Loshchilov & Hutter, ICLR2016] SGDR: Stochastic Gradient Descent
 805   with Warm Restarts. https://arxiv.org/abs/1608.03983
 806 
 807   Note that linear cosine decay is more aggressive than cosine decay and
 808   larger initial learning rates can typically be used.
 809 
 810   When training a model, it is often recommended to lower the learning rate as
 811   the training progresses.  This function applies a noisy linear
 812   cosine decay function to a provided initial learning rate.
 813   It requires a `global_step` value to compute the decayed learning rate.
 814   You can just pass a TensorFlow variable that you increment at each
 815   training step.
 816 
 817   The function returns a no-arg callable that produces the decayed learning
 818   rate. This can be useful for changing the learning rate value across
 819   different invocations of optimizer functions. It is computed as:
 820 
 821   ```python
 822   global_step = min(global_step, decay_steps)
 823   linear_decay = (decay_steps - global_step) / decay_steps)
 824   cosine_decay = 0.5 * (
 825       1 + cos(pi * 2 * num_periods * global_step / decay_steps))
 826   decayed = (alpha + linear_decay + eps_t) * cosine_decay + beta
 827   decayed_learning_rate = learning_rate * decayed
 828   ```
 829   where eps_t is 0-centered gaussian noise with variance
 830   initial_variance / (1 + global_step) ** variance_decay
 831 
 832   Example usage:
 833   ```python
 834   decay_steps = 1000
 835   lr_decayed_fn = tf.train.noisy_linear_cosine_decay(learning_rate, global_step,
 836                                                      decay_steps)
 837   ```
 838 
 839   Args:
 840     learning_rate: A scalar `float32` or `float64` Tensor or a Python number.
 841       The initial learning rate.
 842     global_step: A scalar `int32` or `int64` `Tensor` or a Python number.
 843       Global step to use for the decay computation.
 844     decay_steps: A scalar `int32` or `int64` `Tensor` or a Python number.
 845       Number of steps to decay over.
 846     initial_variance: initial variance for the noise. See computation above.
 847     variance_decay: decay for the noise's variance. See computation above.
 848     num_periods: Number of periods in the cosine part of the decay.
 849       See computation above.
 850     alpha: See computation above.
 851     beta: See computation above.
 852     name: String.  Optional name of the operation.  Defaults to
 853       'NoisyLinearCosineDecay'.
 854   Returns:
 855     A no-arg function that outputs the decayed learning rate, a scalar `Tensor`
 856     of the same type as `learning_rate`.
 857   Raises:
 858     ValueError: if `global_step` is not supplied.
 859   """
 860   if global_step is None:
 861     raise ValueError("noisy linear cosine decay requires global_step")
 862   def decayed_lr(learning_rate, global_step, decay_steps, initial_variance,
 863                  variance_decay, num_periods, alpha, beta, name):
 864     """Helper to recompute learning rate; most helpful in eager-mode."""
 865     with ops.name_scope(name, "NoisyLinearCosineDecay",
 866                         [learning_rate, global_step]) as name:
 867       learning_rate = ops.convert_to_tensor(learning_rate, name="learning_rate")
 868       dtype = learning_rate.dtype
 869       decay_steps = math_ops.cast(decay_steps, dtype)
 870       initial_variance = math_ops.cast(initial_variance, dtype)
 871       variance_decay = math_ops.cast(variance_decay, dtype)
 872       num_periods = math_ops.cast(num_periods, dtype)
 873       alpha = math_ops.cast(alpha, dtype)
 874       beta = math_ops.cast(beta, dtype)
 875 
 876       global_step_recomp = math_ops.cast(global_step, dtype)
 877       global_step_recomp = math_ops.minimum(global_step_recomp, decay_steps)
 878       linear_decayed = (decay_steps - global_step_recomp) / decay_steps
 879       variance = initial_variance / (
 880           math_ops.pow(1.0 + global_step_recomp, variance_decay))
 881       std = math_ops.sqrt(variance)
 882       noisy_linear_decayed = (
 883           linear_decayed + random_ops.random_normal(
 884               linear_decayed.shape, stddev=std))
 885 
 886       completed_fraction = global_step_recomp / decay_steps
 887       fraction = 2.0 * num_periods * completed_fraction
 888       cosine_decayed = 0.5 * (
 889           1.0 + math_ops.cos(constant_op.constant(math.pi) * fraction))
 890       noisy_linear_cosine_decayed = (
 891           (alpha + noisy_linear_decayed) * cosine_decayed + beta)
 892 
 893       return math_ops.multiply(
 894           learning_rate, noisy_linear_cosine_decayed, name=name)
 895 
 896   return functools.partial(decayed_lr, learning_rate, global_step, decay_steps,
 897                            initial_variance, variance_decay, num_periods, alpha,
 898                            beta, name)
