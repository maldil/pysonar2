  28 @tf_export("train.FtrlOptimizer")
  29 class FtrlOptimizer(optimizer.Optimizer):
  30   """Optimizer that implements the FTRL algorithm.
  31 
  32   See this [paper](
  33   https://www.eecs.tufts.edu/~dsculley/papers/ad-click-prediction.pdf).
  34   This version has support for both online L2 (the L2 penalty given in the paper
  35   above) and shrinkage-type L2 (which is the addition of an L2 penalty to the
  36   loss function).
  37   """
  38 
  39   def __init__(self,
  40                learning_rate,
  41                learning_rate_power=-0.5,
  42                initial_accumulator_value=0.1,
  43                l1_regularization_strength=0.0,
  44                l2_regularization_strength=0.0,
  45                use_locking=False,
  46                name="Ftrl",
  47                accum_name=None,
  48                linear_name=None,
  49                l2_shrinkage_regularization_strength=0.0):
  50     r"""Construct a new FTRL optimizer.
  51 
  52     Args:
  53       learning_rate: A float value or a constant float `Tensor`.
  54       learning_rate_power: A float value, must be less or equal to zero.
  55       initial_accumulator_value: The starting value for accumulators.
  56         Only zero or positive values are allowed.
  57       l1_regularization_strength: A float value, must be greater than or
  58         equal to zero.
  59       l2_regularization_strength: A float value, must be greater than or
  60         equal to zero.
  61       use_locking: If `True` use locks for update operations.
  62       name: Optional name prefix for the operations created when applying
  63         gradients.  Defaults to "Ftrl".
  64       accum_name: The suffix for the variable that keeps the gradient squared
  65         accumulator.  If not present, defaults to name.
  66       linear_name: The suffix for the variable that keeps the linear gradient
  67         accumulator.  If not present, defaults to name + "_1".
  68       l2_shrinkage_regularization_strength: A float value, must be greater than
  69         or equal to zero. This differs from L2 above in that the L2 above is a
  70         stabilization penalty, whereas this L2 shrinkage is a magnitude penalty.
  71         The FTRL formulation can be written as:
  72         w_{t+1} = argmin_w(\hat{g}_{1:t}w + L1*||w||_1 + L2*||w||_2^2), where
  73         \hat{g} = g + (2*L2_shrinkage*w), and g is the gradient of the loss
  74         function w.r.t. the weights w.
  75         Specifically, in the absence of L1 regularization, it is equivalent to
  76         the following update rule:
  77         w_{t+1} = w_t - lr_t / (1 + 2*L2*lr_t) * g_t -
  78                   2*L2_shrinkage*lr_t / (1 + 2*L2*lr_t) * w_t
  79         where lr_t is the learning rate at t.
  80         When input is sparse shrinkage will only happen on the active weights.
  81 
  82     Raises:
  83       ValueError: If one of the arguments is invalid.
  84     """
  85     super(FtrlOptimizer, self).__init__(use_locking, name)
  86 
  87     if initial_accumulator_value < 0.0:
  88       raise ValueError(
  89           "initial_accumulator_value %f needs to be positive or zero" %
  90           initial_accumulator_value)
  91     if learning_rate_power > 0.0:
  92       raise ValueError("learning_rate_power %f needs to be negative or zero" %
  93                        learning_rate_power)
  94     if l1_regularization_strength < 0.0:
  95       raise ValueError(
  96           "l1_regularization_strength %f needs to be positive or zero" %
  97           l1_regularization_strength)
  98     if l2_regularization_strength < 0.0:
  99       raise ValueError(
 100           "l2_regularization_strength %f needs to be positive or zero" %
 101           l2_regularization_strength)
 102     if l2_shrinkage_regularization_strength < 0.0:
 103       raise ValueError(
 104           "l2_shrinkage_regularization_strength %f needs to be positive"
 105           " or zero" % l2_shrinkage_regularization_strength)
 106 
 107     self._learning_rate = learning_rate
 108     self._learning_rate_power = learning_rate_power
 109     self._initial_accumulator_value = initial_accumulator_value
 110     self._l1_regularization_strength = l1_regularization_strength
 111     self._l2_regularization_strength = l2_regularization_strength
 112     self._l2_shrinkage_regularization_strength = (
 113         l2_shrinkage_regularization_strength)
 114     self._learning_rate_tensor = None
 115     self._learning_rate_power_tensor = None
 116     self._l1_regularization_strength_tensor = None
 117     self._l2_regularization_strength_tensor = None
 118     self._l2_shrinkage_regularization_strength_tensor = None
 119     self._accum_name = accum_name
 120     self._linear_name = linear_name
 121 
 122   def _create_slots(self, var_list):
 123     # Create the "accum" and "linear" slots.
 124     for v in var_list:
 125       with ops.colocate_with(v):
 126         val = constant_op.constant(
 127             self._initial_accumulator_value, dtype=v.dtype, shape=v.get_shape())
 128         self._get_or_make_slot(v, val, "accum", self._accum_name or self._name)
 129         self._zeros_slot(v, "linear", self._linear_name or self._name)
 130 
 131   def _prepare(self):
 132     self._learning_rate_tensor = ops.convert_to_tensor(
 133         self._learning_rate, name="learning_rate")
 134     self._l1_regularization_strength_tensor = ops.convert_to_tensor(
 135         self._l1_regularization_strength, name="l1_regularization_strength")
 136     self._l2_regularization_strength_tensor = ops.convert_to_tensor(
 137         self._l2_regularization_strength, name="l2_regularization_strength")
 138     self._l2_shrinkage_regularization_strength_tensor = ops.convert_to_tensor(
 139         self._l2_shrinkage_regularization_strength,
 140         name="l2_shrinkage_regularization_strength")
 141     self._learning_rate_power_tensor = ops.convert_to_tensor(
 142         self._learning_rate_power, name="learning_rate_power")
 143 
 144   def _apply_dense(self, grad, var):
 145     accum = self.get_slot(var, "accum")
 146     linear = self.get_slot(var, "linear")
 147     if self._l2_shrinkage_regularization_strength <= 0.0:
 148       return training_ops.apply_ftrl(
 149           var,
 150           accum,
 151           linear,
 152           grad,
 153           math_ops.cast(self._learning_rate_tensor, var.dtype.base_dtype),
 154           math_ops.cast(self._l1_regularization_strength_tensor,
 155                         var.dtype.base_dtype),
 156           math_ops.cast(self._l2_regularization_strength_tensor,
 157                         var.dtype.base_dtype),
 158           math_ops.cast(self._learning_rate_power_tensor, var.dtype.base_dtype),
 159           use_locking=self._use_locking)
 160     else:
 161       return training_ops.apply_ftrl_v2(
 162           var,
 163           accum,
 164           linear,
 165           grad,
 166           math_ops.cast(self._learning_rate_tensor, var.dtype.base_dtype),
 167           math_ops.cast(self._l1_regularization_strength_tensor,
 168                         var.dtype.base_dtype),
 169           math_ops.cast(self._l2_regularization_strength_tensor,
 170                         var.dtype.base_dtype),
 171           math_ops.cast(self._l2_shrinkage_regularization_strength_tensor,
 172                         var.dtype.base_dtype),
 173           math_ops.cast(self._learning_rate_power_tensor, var.dtype.base_dtype),
 174           use_locking=self._use_locking)
 175
