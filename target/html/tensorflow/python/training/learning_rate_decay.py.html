  25 @tf_export(v1=["train.exponential_decay"])
  26 def exponential_decay(learning_rate,
  27                       global_step,
  28                       decay_steps,
  29                       decay_rate,
  30                       staircase=False,
  31                       name=None):
  32   """Applies exponential decay to the learning rate.
  33 
  34   When training a model, it is often recommended to lower the learning rate as
  35   the training progresses.  This function applies an exponential decay function
  36   to a provided initial learning rate.  It requires a `global_step` value to
  37   compute the decayed learning rate.  You can just pass a TensorFlow variable
  38   that you increment at each training step.
  39 
  40   The function returns the decayed learning rate.  It is computed as:
  41 
  42   ```python
  43   decayed_learning_rate = learning_rate *
  44                           decay_rate ^ (global_step / decay_steps)
  45   ```
  46 
  47   If the argument `staircase` is `True`, then `global_step / decay_steps` is an
  48   integer division and the decayed learning rate follows a staircase function.
  49 
  50   Example: decay every 100000 steps with a base of 0.96:
  51 
  52   ```python
  53   ...
  54   global_step = tf.Variable(0, trainable=False)
  55   starter_learning_rate = 0.1
  56   learning_rate = tf.train.exponential_decay(starter_learning_rate, global_step,
  57                                              100000, 0.96, staircase=True)
  58   # Passing global_step to minimize() will increment it at each step.
  59   learning_step = (
  60       tf.train.GradientDescentOptimizer(learning_rate)
  61       .minimize(...my loss..., global_step=global_step)
  62   )
  63   ```
  64 
  65   Args:
  66     learning_rate: A scalar `float32` or `float64` `Tensor` or a
  67       Python number.  The initial learning rate.
  68     global_step: A scalar `int32` or `int64` `Tensor` or a Python number.
  69       Global step to use for the decay computation.  Must not be negative.
  70     decay_steps: A scalar `int32` or `int64` `Tensor` or a Python number.
  71       Must be positive.  See the decay computation above.
  72     decay_rate: A scalar `float32` or `float64` `Tensor` or a
  73       Python number.  The decay rate.
  74     staircase: Boolean.  If `True` decay the learning rate at discrete intervals
  75     name: String.  Optional name of the operation.  Defaults to
  76       'ExponentialDecay'.
  77 
  78   Returns:
  79     A scalar `Tensor` of the same type as `learning_rate`.  The decayed
  80     learning rate.
  81 
  82   Raises:
  83     ValueError: if `global_step` is not supplied.
  84 
  85   @compatibility(eager)
  86   When eager execution is enabled, this function returns a function which in
  87   turn returns the decayed learning rate Tensor. This can be useful for changing
  88   the learning rate value across different invocations of optimizer functions.
  89   @end_compatibility
  90   """
  91   decayed_lr = learning_rate_decay_v2.exponential_decay(learning_rate,
  92                                                         global_step,
  93                                                         decay_steps,
  94                                                         decay_rate,
  95                                                         staircase=staircase,
  96                                                         name=name)
  97   if not context.executing_eagerly():
  98     decayed_lr = decayed_lr()
  99 
 100   return decayed_lr

 103 @tf_export(v1=["train.piecewise_constant"])
 104 def piecewise_constant(x, boundaries, values, name=None):
 105   """Piecewise constant from boundaries and interval values.
 106 
 107   Example: use a learning rate that's 1.0 for the first 100001 steps, 0.5
 108     for the next 10000 steps, and 0.1 for any additional steps.
 109 
 110   ```python
 111   global_step = tf.Variable(0, trainable=False)
 112   boundaries = [100000, 110000]
 113   values = [1.0, 0.5, 0.1]
 114   learning_rate = tf.train.piecewise_constant(global_step, boundaries, values)
 115 
 116   # Later, whenever we perform an optimization step, we increment global_step.
 117   ```
 118 
 119   Args:
 120     x: A 0-D scalar `Tensor`. Must be one of the following types: `float32`,
 121       `float64`, `uint8`, `int8`, `int16`, `int32`, `int64`.
 122     boundaries: A list of `Tensor`s or `int`s or `float`s with strictly
 123       increasing entries, and with all elements having the same type as `x`.
 124     values: A list of `Tensor`s or `float`s or `int`s that specifies the values
 125       for the intervals defined by `boundaries`. It should have one more element
 126       than `boundaries`, and all elements should have the same type.
 127     name: A string. Optional name of the operation. Defaults to
 128       'PiecewiseConstant'.
 129 
 130   Returns:
 131     A 0-D Tensor. Its value is `values[0]` when `x <= boundaries[0]`,
 132     `values[1]` when `x > boundaries[0]` and `x <= boundaries[1]`, ...,
 133     and values[-1] when `x > boundaries[-1]`.
 134 
 135   Raises:
 136     ValueError: if types of `x` and `boundaries` do not match, or types of all
 137         `values` do not match or
 138         the number of elements in the lists does not match.
 139 
 140   @compatibility(eager)
 141   When eager execution is enabled, this function returns a function which in
 142   turn returns the decayed learning rate Tensor. This can be useful for changing
 143   the learning rate value across different invocations of optimizer functions.
 144   @end_compatibility
 145   """
 146   decayed_lr = learning_rate_decay_v2.piecewise_constant(x, boundaries, values,
 147                                                          name=name)
 148   if not context.executing_eagerly():
 149     decayed_lr = decayed_lr()
 150 
 151   return decayed_lr

 154 @tf_export(v1=["train.polynomial_decay"])
 155 def polynomial_decay(learning_rate,
 156                      global_step,
 157                      decay_steps,
 158                      end_learning_rate=0.0001,
 159                      power=1.0,
 160                      cycle=False,
 161                      name=None):
 162   """Applies a polynomial decay to the learning rate.
 163 
 164   It is commonly observed that a monotonically decreasing learning rate, whose
 165   degree of change is carefully chosen, results in a better performing model.
 166   This function applies a polynomial decay function to a provided initial
 167   `learning_rate` to reach an `end_learning_rate` in the given `decay_steps`.
 168 
 169   It requires a `global_step` value to compute the decayed learning rate.  You
 170   can just pass a TensorFlow variable that you increment at each training step.
 171 
 172   The function returns the decayed learning rate.  It is computed as:
 173 
 174   ```python
 175   global_step = min(global_step, decay_steps)
 176   decayed_learning_rate = (learning_rate - end_learning_rate) *
 177                           (1 - global_step / decay_steps) ^ (power) +
 178                           end_learning_rate
 179 
 180   ```
 181 
 182   If `cycle` is True then a multiple of `decay_steps` is used, the first one
 183   that is bigger than `global_steps`.
 184 
 185   ```python
 186   decay_steps = decay_steps * ceil(global_step / decay_steps)
 187   decayed_learning_rate = (learning_rate - end_learning_rate) *
 188                           (1 - global_step / decay_steps) ^ (power) +
 189                           end_learning_rate
 190 
 191   ```
 192 
 193   Example: decay from 0.1 to 0.01 in 10000 steps using sqrt (i.e. power=0.5):
 194 
 195   ```python
 196   ...
 197   global_step = tf.Variable(0, trainable=False)
 198   starter_learning_rate = 0.1
 199   end_learning_rate = 0.01
 200   decay_steps = 10000
 201   learning_rate = tf.train.polynomial_decay(starter_learning_rate, global_step,
 202                                             decay_steps, end_learning_rate,
 203                                             power=0.5)
 204   # Passing global_step to minimize() will increment it at each step.
 205   learning_step = (
 206       tf.train.GradientDescentOptimizer(learning_rate)
 207       .minimize(...my loss..., global_step=global_step)
 208   )
 209   ```
 210 
 211   Args:
 212     learning_rate: A scalar `float32` or `float64` `Tensor` or a
 213       Python number.  The initial learning rate.
 214     global_step: A scalar `int32` or `int64` `Tensor` or a Python number.
 215       Global step to use for the decay computation.  Must not be negative.
 216     decay_steps: A scalar `int32` or `int64` `Tensor` or a Python number.
 217       Must be positive.  See the decay computation above.
 218     end_learning_rate: A scalar `float32` or `float64` `Tensor` or a
 219       Python number.  The minimal end learning rate.
 220     power: A scalar `float32` or `float64` `Tensor` or a
 221       Python number.  The power of the polynomial. Defaults to linear, 1.0.
 222     cycle: A boolean, whether or not it should cycle beyond decay_steps.
 223     name: String.  Optional name of the operation. Defaults to
 224       'PolynomialDecay'.
 225 
 226   Returns:
 227     A scalar `Tensor` of the same type as `learning_rate`.  The decayed
 228     learning rate.
 229 
 230   Raises:
 231     ValueError: if `global_step` is not supplied.
 232 
 233   @compatibility(eager)
 234   When eager execution is enabled, this function returns a function which in
 235   turn returns the decayed learning rate Tensor. This can be useful for changing
 236   the learning rate value across different invocations of optimizer functions.
 237   @end_compatibility
 238   """
 239   decayed_lr = learning_rate_decay_v2.polynomial_decay(
 240       learning_rate,
 241       global_step,
 242       decay_steps,
 243       end_learning_rate=end_learning_rate,
 244       power=power,
 245       cycle=cycle,
 246       name=name)
 247 
 248   if not context.executing_eagerly():
 249     decayed_lr = decayed_lr()
 250 
 251   return decayed_lr

 422 @tf_export(v1=["train.cosine_decay"])
 423 def cosine_decay(learning_rate, global_step, decay_steps, alpha=0.0, name=None):
 424   """Applies cosine decay to the learning rate.
 425 
 426   See [Loshchilov & Hutter, ICLR2016], SGDR: Stochastic Gradient Descent
 427   with Warm Restarts. https://arxiv.org/abs/1608.03983
 428 
 429   When training a model, it is often recommended to lower the learning rate as
 430   the training progresses.  This function applies a cosine decay function
 431   to a provided initial learning rate.  It requires a `global_step` value to
 432   compute the decayed learning rate.  You can just pass a TensorFlow variable
 433   that you increment at each training step.
 434 
 435   The function returns the decayed learning rate.  It is computed as:
 436   ```python
 437   global_step = min(global_step, decay_steps)
 438   cosine_decay = 0.5 * (1 + cos(pi * global_step / decay_steps))
 439   decayed = (1 - alpha) * cosine_decay + alpha
 440   decayed_learning_rate = learning_rate * decayed
 441   ```
 442 
 443   Example usage:
 444   ```python
 445   decay_steps = 1000
 446   lr_decayed = cosine_decay(learning_rate, global_step, decay_steps)
 447   ```
 448 
 449   Args:
 450     learning_rate: A scalar `float32` or `float64` Tensor or a Python number.
 451       The initial learning rate.
 452     global_step: A scalar `int32` or `int64` `Tensor` or a Python number.
 453       Global step to use for the decay computation.
 454     decay_steps: A scalar `int32` or `int64` `Tensor` or a Python number.
 455       Number of steps to decay over.
 456     alpha: A scalar `float32` or `float64` Tensor or a Python number.
 457       Minimum learning rate value as a fraction of learning_rate.
 458     name: String. Optional name of the operation.  Defaults to 'CosineDecay'.
 459   Returns:
 460     A scalar `Tensor` of the same type as `learning_rate`.  The decayed
 461     learning rate.
 462   Raises:
 463     ValueError: if `global_step` is not supplied.
 464 
 465   @compatibility(eager)
 466   When eager execution is enabled, this function returns a function which in
 467   turn returns the decayed learning rate Tensor. This can be useful for changing
 468   the learning rate value across different invocations of optimizer functions.
 469   @end_compatibility
 470   """
 471   decayed_lr = learning_rate_decay_v2.cosine_decay(
 472       learning_rate, global_step, decay_steps, alpha=alpha, name=name)
 473 
 474   if not context.executing_eagerly():
 475     decayed_lr = decayed_lr()
 476 
 477   return decayed_lr

 494   the training progresses.  This function applies a cosine decay function with

 514       The initial learning rate.
 515     global_step: A scalar `int32` or `int64` `Tensor` or a Python number.
 516       Global step to use for the decay computation.
 517     first_decay_steps: A scalar `int32` or `int64` `Tensor` or a Python number.
 518       Number of steps to decay over.
 519     t_mul: A scalar `float32` or `float64` `Tensor` or a Python number.
 520       Used to derive the number of iterations in the i-th period
 521     m_mul: A scalar `float32` or `float64` `Tensor` or a Python number.
 522       Used to derive the initial learning rate of the i-th period:
 523     alpha: A scalar `float32` or `float64` Tensor or a Python number.
 524       Minimum learning rate value as a fraction of the learning_rate.
 525     name: String. Optional name of the operation.  Defaults to 'SGDRDecay'.
 526   Returns:
 527     A scalar `Tensor` of the same type as `learning_rate`.  The decayed
 528     learning rate.
 529   Raises:
 530     ValueError: if `global_step` is not supplied.
 531 
 532   @compatibility(eager)
 533   When eager execution is enabled, this function returns a function which in
 534   turn returns the decayed learning rate Tensor. This can be useful for changing
 535   the learning rate value across different invocations of optimizer functions.
 536   @end_compatibility
 537   """
 538   decayed_lr = learning_rate_decay_v2.cosine_decay_restarts(
 539       learning_rate,
 540       global_step,
 541       first_decay_steps,
 542       t_mul=t_mul,
 543       m_mul=m_mul,
 544       alpha=alpha,
 545       name=name)
 546 
 547   if not context.executing_eagerly():
 548     decayed_lr = decayed_lr()
 549

 553 @tf_export(v1=["train.linear_cosine_decay"])
 554 def linear_cosine_decay(learning_rate,
 555                         global_step,
 556                         decay_steps,
 557                         num_periods=0.5,
 558                         alpha=0.0,
 559                         beta=0.001,
 560                         name=None):
 561   """Applies linear cosine decay to the learning rate.
 562 
 563   See [Bello et al., ICML2017] Neural Optimizer Search with RL.
 564   https://arxiv.org/abs/1709.07417
 565 
 566   For the idea of warm starts here controlled by `num_periods`,
 567   see [Loshchilov & Hutter, ICLR2016] SGDR: Stochastic Gradient Descent
 568   with Warm Restarts. https://arxiv.org/abs/1608.03983
 569 
 570   Note that linear cosine decay is more aggressive than cosine decay and
 571   larger initial learning rates can typically be used.
 572 
 573   When training a model, it is often recommended to lower the learning rate as
 574   the training progresses.  This function applies a linear cosine decay function
 575   to a provided initial learning rate.  It requires a `global_step` value to
 576   compute the decayed learning rate.  You can just pass a TensorFlow variable
 577   that you increment at each training step.
 578 
 579   The function returns the decayed learning rate.  It is computed as:
 580   ```python
 581   global_step = min(global_step, decay_steps)
 582   linear_decay = (decay_steps - global_step) / decay_steps)
 583   cosine_decay = 0.5 * (
 584       1 + cos(pi * 2 * num_periods * global_step / decay_steps))
 585   decayed = (alpha + linear_decay) * cosine_decay + beta
 586   decayed_learning_rate = learning_rate * decayed
 587   ```
 588 
 589   Example usage:
 590   ```python
 591   decay_steps = 1000
 592   lr_decayed = linear_cosine_decay(learning_rate, global_step, decay_steps)
 593   ```
 594 
 595   Args:
 596     learning_rate: A scalar `float32` or `float64` Tensor or a Python number.
 597       The initial learning rate.
 598     global_step: A scalar `int32` or `int64` `Tensor` or a Python number.
 599       Global step to use for the decay computation.
 600     decay_steps: A scalar `int32` or `int64` `Tensor` or a Python number.
 601       Number of steps to decay over.
 602     num_periods: Number of periods in the cosine part of the decay.
 603       See computation above.
 604     alpha: See computation above.
 605     beta: See computation above.
 606     name: String.  Optional name of the operation.  Defaults to
 607       'LinearCosineDecay'.
 608   Returns:
 609     A scalar `Tensor` of the same type as `learning_rate`.  The decayed
 610     learning rate.
 611   Raises:
 612     ValueError: if `global_step` is not supplied.
 613 
 614   @compatibility(eager)
 615   When eager execution is enabled, this function returns a function which in
 616   turn returns the decayed learning rate Tensor. This can be useful for changing
 617   the learning rate value across different invocations of optimizer functions.
 618   @end_compatibility
 619   """

 621       learning_rate,

 635 @tf_export(v1=["train.noisy_linear_cosine_decay"])
 636 def noisy_linear_cosine_decay(learning_rate,
 637                               global_step,
 638                               decay_steps,
 639                               initial_variance=1.0,
 640                               variance_decay=0.55,
 641                               num_periods=0.5,
 642                               alpha=0.0,
 643                               beta=0.001,
 644                               name=None):
 645   """Applies noisy linear cosine decay to the learning rate.
 646 
 647   See [Bello et al., ICML2017] Neural Optimizer Search with RL.
 648   https://arxiv.org/abs/1709.07417
 649 
 650   For the idea of warm starts here controlled by `num_periods`,
 651   see [Loshchilov & Hutter, ICLR2016] SGDR: Stochastic Gradient Descent
 652   with Warm Restarts. https://arxiv.org/abs/1608.03983
 653 
 654   Note that linear cosine decay is more aggressive than cosine decay and
 655   larger initial learning rates can typically be used.
 656 
 657   When training a model, it is often recommended to lower the learning rate as
 658   the training progresses.  This function applies a noisy linear
 659   cosine decay function to a provided initial learning rate.
 660   It requires a `global_step` value to compute the decayed learning rate.
 661   You can just pass a TensorFlow variable that you increment at each
 662   training step.
 663 
 664   The function returns the decayed learning rate.  It is computed as:
 665   ```python
 666   global_step = min(global_step, decay_steps)
 667   linear_decay = (decay_steps - global_step) / decay_steps)
 668   cosine_decay = 0.5 * (
 669       1 + cos(pi * 2 * num_periods * global_step / decay_steps))
 670   decayed = (alpha + linear_decay + eps_t) * cosine_decay + beta
 671   decayed_learning_rate = learning_rate * decayed
 672   ```
 673   where eps_t is 0-centered gaussian noise with variance
 674   initial_variance / (1 + global_step) ** variance_decay
 675 
 676   Example usage:
 677   ```python
 678   decay_steps = 1000
 679   lr_decayed = noisy_linear_cosine_decay(
 680     learning_rate, global_step, decay_steps)
 681   ```
 682 
 683   Args:
 684     learning_rate: A scalar `float32` or `float64` Tensor or a Python number.
 685       The initial learning rate.
 686     global_step: A scalar `int32` or `int64` `Tensor` or a Python number.
 687       Global step to use for the decay computation.
 688     decay_steps: A scalar `int32` or `int64` `Tensor` or a Python number.
 689       Number of steps to decay over.
 690     initial_variance: initial variance for the noise. See computation above.
 691     variance_decay: decay for the noise's variance. See computation above.
 692     num_periods: Number of periods in the cosine part of the decay.
 693       See computation above.
 694     alpha: See computation above.
 695     beta: See computation above.
 696     name: String.  Optional name of the operation.  Defaults to
 697       'NoisyLinearCosineDecay'.
 698   Returns:
 699     A scalar `Tensor` of the same type as `learning_rate`.  The decayed
 700     learning rate.
 701   Raises:
 702     ValueError: if `global_step` is not supplied.
 703 
 704   @compatibility(eager)
 705   When eager execution is enabled, this function returns a function which in
 706   turn returns the decayed learning rate Tensor. This can be useful for changing
 707   the learning rate value across different invocations of optimizer functions.
 708   @end_compatibility
 709   """
 710   decayed_lr = learning_rate_decay_v2.noisy_linear_cosine_decay(
 711       learning_rate, global_step,
 712       decay_steps,
 713       initial_variance=initial_variance,
 714       variance_decay=variance_decay,
 715       num_periods=num_periods,
 716       alpha=alpha,
 717       beta=beta,
 718       name=name)
 719
