  80 @tf_export(v1=["train.limit_epochs"])
  81 @deprecation.deprecated(
  82     None, "Queue-based input pipelines have been replaced by `tf.data`. Use "
  83     "`tf.data.Dataset.from_tensors(tensor).repeat(num_epochs)`.")
  84 def limit_epochs(tensor, num_epochs=None, name=None):
  85   """Returns tensor `num_epochs` times and then raises an `OutOfRange` error.
  86 
  87   Note: creates local counter `epochs`. Use `local_variables_initializer()` to
  88   initialize local variables.
  89 
  90   Args:
  91     tensor: Any `Tensor`.
  92     num_epochs: A positive integer (optional).  If specified, limits the number
  93       of steps the output tensor may be evaluated.
  94     name: A name for the operations (optional).
  95 
  96   Returns:
  97     tensor or `OutOfRange`.
  98 
  99   Raises:
 100     ValueError: if `num_epochs` is invalid.
 101   """
 102   if num_epochs is None:
 103     return tensor
 104   if num_epochs <= 0:
 105     raise ValueError("num_epochs must be > 0 not %d." % num_epochs)
 106   with ops.name_scope(name, "limit_epochs", [tensor]) as name:
 107     zero64 = constant_op.constant(0, dtype=dtypes.int64)
 108     epochs = vs.variable(
 109         zero64, name="epochs", trainable=False,
 110         collections=[ops.GraphKeys.LOCAL_VARIABLES])
 111     counter = epochs.count_up_to(num_epochs)
 112     with ops.control_dependencies([counter]):
 113       return array_ops.identity(tensor, name=name)

 116 @tf_export(v1=["train.input_producer"])
 117 @deprecation.deprecated(
 118     None, "Queue-based input pipelines have been replaced by `tf.data`. Use "
 119     "`tf.data.Dataset.from_tensor_slices(input_tensor).shuffle"
 120     "(tf.shape(input_tensor, out_type=tf.int64)[0]).repeat(num_epochs)`. If "
 121     "`shuffle=False`, omit the `.shuffle(...)`.")
 122 def input_producer(input_tensor,
 123                    element_shape=None,
 124                    num_epochs=None,
 125                    shuffle=True,
 126                    seed=None,
 127                    capacity=32,
 128                    shared_name=None,
 129                    summary_name=None,
 130                    name=None,
 131                    cancel_op=None):
 132   """Output the rows of `input_tensor` to a queue for an input pipeline.
 133 
 134   Note: if `num_epochs` is not `None`, this function creates local counter
 135   `epochs`. Use `local_variables_initializer()` to initialize local variables.
 136 
 137   Args:
 138     input_tensor: A tensor with the rows to produce. Must be at least
 139       one-dimensional. Must either have a fully-defined shape, or
 140       `element_shape` must be defined.
 141     element_shape: (Optional.) A `TensorShape` representing the shape of a
 142       row of `input_tensor`, if it cannot be inferred.
 143     num_epochs: (Optional.) An integer. If specified `input_producer` produces
 144       each row of `input_tensor` `num_epochs` times before generating an
 145       `OutOfRange` error. If not specified, `input_producer` can cycle through
 146       the rows of `input_tensor` an unlimited number of times.
 147     shuffle: (Optional.) A boolean. If true, the rows are randomly shuffled
 148       within each epoch.
 149     seed: (Optional.) An integer. The seed to use if `shuffle` is true.
 150     capacity: (Optional.) The capacity of the queue to be used for buffering
 151       the input.
 152     shared_name: (Optional.) If set, this queue will be shared under the given
 153       name across multiple sessions.
 154     summary_name: (Optional.) If set, a scalar summary for the current queue
 155       size will be generated, using this name as part of the tag.
 156     name: (Optional.) A name for queue.
 157     cancel_op: (Optional.) Cancel op for the queue
 158 
 159   Returns:
 160     A queue with the output rows.  A `QueueRunner` for the queue is
 161     added to the current `QUEUE_RUNNER` collection of the current
 162     graph.
 163 
 164   Raises:
 165     ValueError: If the shape of the input cannot be inferred from the arguments.
 166     RuntimeError: If called with eager execution enabled.
 167 
 168   @compatibility(eager)
 169   Input pipelines based on Queues are not supported when eager execution is
 170   enabled. Please use the `tf.data` API to ingest data under eager execution.
 171   @end_compatibility
 172   """
 173   if context.executing_eagerly():
 174     raise RuntimeError(
 175         "Input pipelines based on Queues are not supported when eager execution"
 176         " is enabled. Please use tf.data to ingest data into your model"
 177         " instead.")
 178   with ops.name_scope(name, "input_producer", [input_tensor]):
 179     input_tensor = ops.convert_to_tensor(input_tensor, name="input_tensor")
 180     element_shape = input_tensor.shape[1:].merge_with(element_shape)
 181     if not element_shape.is_fully_defined():
 182       raise ValueError("Either `input_tensor` must have a fully defined shape "
 183                        "or `element_shape` must be specified")
 184 
 185     if shuffle:
 186       input_tensor = random_ops.random_shuffle(input_tensor, seed=seed)
 187 
 188     input_tensor = limit_epochs(input_tensor, num_epochs)
 189 
 190     q = data_flow_ops.FIFOQueue(capacity=capacity,
 191                                 dtypes=[input_tensor.dtype.base_dtype],
 192                                 shapes=[element_shape],
 193                                 shared_name=shared_name, name=name)
 194     enq = q.enqueue_many([input_tensor])
 195     queue_runner.add_queue_runner(
 196         queue_runner.QueueRunner(
 197             q, [enq], cancel_op=cancel_op))
 198     if summary_name is not None:
 199       summary.scalar(summary_name,
 200                      math_ops.to_float(q.size()) * (1. / capacity))
 201     return q

 321 @tf_export(v1=["train.slice_input_producer"])
 322 @deprecation.deprecated(
 323     None, "Queue-based input pipelines have been replaced by `tf.data`. Use "
 324     "`tf.data.Dataset.from_tensor_slices(tuple(tensor_list)).shuffle"
 325     "(tf.shape(input_tensor, out_type=tf.int64)[0]).repeat(num_epochs)`. If "
 326     "`shuffle=False`, omit the `.shuffle(...)`.")
 327 def slice_input_producer(tensor_list, num_epochs=None, shuffle=True, seed=None,
 328                          capacity=32, shared_name=None, name=None):
 329   """Produces a slice of each `Tensor` in `tensor_list`.
 330 
 331   Implemented using a Queue -- a `QueueRunner` for the Queue
 332   is added to the current `Graph`'s `QUEUE_RUNNER` collection.
 333 
 334   Args:
 335     tensor_list: A list of `Tensor` objects. Every `Tensor` in
 336       `tensor_list` must have the same size in the first dimension.
 337     num_epochs: An integer (optional). If specified, `slice_input_producer`
 338       produces each slice `num_epochs` times before generating
 339       an `OutOfRange` error. If not specified, `slice_input_producer` can cycle
 340       through the slices an unlimited number of times.
 341     shuffle: Boolean. If true, the integers are randomly shuffled within each
 342       epoch.
 343     seed: An integer (optional). Seed used if shuffle == True.
 344     capacity: An integer. Sets the queue capacity.
 345     shared_name: (optional). If set, this queue will be shared under the given
 346       name across multiple sessions.
 347     name: A name for the operations (optional).
 348 
 349   Returns:
 350     A list of tensors, one for each element of `tensor_list`.  If the tensor
 351     in `tensor_list` has shape `[N, a, b, .., z]`, then the corresponding output
 352     tensor will have shape `[a, b, ..., z]`.
 353 
 354   Raises:
 355     ValueError: if `slice_input_producer` produces nothing from `tensor_list`.
 356 
 357   @compatibility(eager)
 358   Input pipelines based on Queues are not supported when eager execution is
 359   enabled. Please use the `tf.data` API to ingest data under eager execution.
 360   @end_compatibility
 361   """
 362   with ops.name_scope(name, "input_producer", tensor_list):
 363     tensor_list = ops.convert_n_to_tensor_or_indexed_slices(tensor_list)
 364     if not tensor_list:
 365       raise ValueError(
 366           "Expected at least one tensor in slice_input_producer().")
 367     range_size = array_ops.shape(tensor_list[0])[0]
 368     # TODO(josh11b): Add an assertion that the first dimension of
 369     # everything in TensorList matches. Maybe just check the inferred shapes?
 370     queue = range_input_producer(range_size, num_epochs=num_epochs,
 371                                  shuffle=shuffle, seed=seed, capacity=capacity,
 372                                  shared_name=shared_name)
 373     index = queue.dequeue()
 374     output = [array_ops.gather(t, index) for t in tensor_list]
 375     return output

 385 class _SparseMetaData(object):
 386   """Store information about the Tensor: Is it sparse?, map_op, and rank."""
 387 
 388   def __init__(self, sparse, map_op, rank):
 389     """Create the metadata.
 390 
 391     Args:
 392       sparse: Python boolean.
 393       map_op: The `Operation` that created the `SparseTensorsMap` in question.
 394         This Op contains information about the underlying Map object and the
 395         dtype of the original data.
 396       rank: The statically known rank of the `SparseTensor`.
 397     """
 398     self._sparse = sparse
 399     self._map_op = map_op
 400     self._rank = rank
 401 
 402   def __eq__(self, other):
 403     if self.sparse != other.sparse:
 404       return False
 405     if not self.sparse:
 406       return True
 407     # If map_ops are not the same, the data source is not the same.
 408     if (self.map_op is not None) != (other.map_op is not None):
 409       return False
 410     if self.map_op != other.map_op:
 411       return False
 412     if not self.rank.is_compatible_with(other.rank):
 413       return False
 414     return True
 415 
 416   def __ne__(self, other):
 417     return not self.__eq__(other)
 418 
 419   def __str__(self):
 420     return "[SparseMetaData(%s, %s, %s)]" % (self.sparse, self.map_op.name,
 421                                              self.rank)
 422 
 423   def merge_with(self, other):
 424     if self != other:
 425       raise ValueError("SparseMetaData objects are incompatible: %s vs. %s"
 426                        % (self, other))
 427     if self.sparse:
 428       self.rank.merge_with(other.rank)
 429     return self
 430 
 431   @property
 432   def map_op(self):
 433     return self._map_op
 434 
 435   @property
 436   def sparse(self):
 437     return self._sparse
 438 
 439   @property
 440   def rank(self):
 441     return self._rank

 451 def _as_tensor_list_list(tensors_list):
 452   if not tensors_list:
 453     raise ValueError("Expected at least one set of tensors")
 454   if isinstance(tensors_list[0], dict):
 455     expected_keys = set(tensors_list[0].keys())
 456     for tensors in tensors_list[1:]:
 457       if set(tensors.keys()) != expected_keys:
 458         raise ValueError("All dictionaries in tensors_list must have "
 459                          "the same keys")
 460     return [_as_tensor_list(tensors) for tensors in tensors_list]
 461   else:
 462     return tensors_list

 477 def _store_sparse_tensors(tensor_list, enqueue_many, keep_input,
 478                           shared_map_ops=None):
 479   """Store SparseTensors for feeding into batch, etc.
 480 
 481   If `shared_map_ops` is provided, the underlying `SparseTensorsMap` objects
 482   are reused (shared).  This argument is useful for, e.g., `batch_join`
 483   where multiple enqueue operations write to the same Queue component,
 484   and another (dequeue) thread reads from that same location and must then
 485   restore the associated `SparseTensor` objects.  In this case, the sparse
 486   restore must have a single `SparseTensorMap` from which to read out the
 487   handles; so a single `SparseTensorMap` must be shared for storing
 488   across the multiple enqueue operations.  This sharing is performed by
 489   calling `_store_sparse_tensors` the first time with `shared_map_ops=None`,
 490   and then in subsequent times with this value set to the list of `Operation`
 491   objects created in the first call.
 492 
 493   Args:
 494     tensor_list: List of `Tensor` and `SparseTensor` objects.
 495     enqueue_many: Python `Boolean`.
 496     keep_input: Must be a scalar bool Tensor (not a Python bool). If False,
 497       don't store.
 498     shared_map_ops: (optional) List of `Operation` objects from a previous
 499       call to `_store_sparse_tensors`.  If not `None`, the op types should be
 500       one of `AddSparseToTensorsMap` or `AddManySparseToTensorsMap` in the
 501       locations corresponding to `SparseTensors` in `tensor_list`.
 502 
 503   Returns:
 504     A tuple `(stored_list, sparse_info_list)` where `stored_list` is a list
 505     of `Tensor` objects (same length as `tensor_list`) and `sparse_info_list`
 506     is a list of the same length of `_SparseMetaData` objects.
 507   """
 508   maybe_shared_map_ops = shared_map_ops or [None] * len(tensor_list)
 509 
 510   def _sparse_meta_data(t, storing_op, map_op):
 511     if not isinstance(t, sparse_tensor.SparseTensor):
 512       return _SparseMetaData(False, None, None)
 513     rank = t.dense_shape.shape.with_rank(1)[0]
 514     if enqueue_many:
 515       rank -= 1
 516     # If a shared map_op was provided, use that. Otherwise use the name of
 517     # the operation used to store the SparseTensor.
 518     return _SparseMetaData(
 519         sparse=True, map_op=map_op or storing_op, rank=rank)
 520 
 521   def _maybe_store(t, shared_map_op):
 522     """Store Sparse tensor, if necessary."""
 523     if not isinstance(t, sparse_tensor.SparseTensor):
 524       return t
 525     map_op_name = shared_map_op.name if shared_map_op else None
 526     def _maybe_store_sparse(t, map_op_name, keep_input):
 527       """Conditionally store a single sparse Tensor."""
 528       return utils.smart_cond(
 529           keep_input,
 530           lambda: _store_sparse(t, shared_name=map_op_name),
 531           lambda: constant_op.constant(-1, dtypes.int64))
 532     def _maybe_store_many_sparse(t, map_op_name, keep_input):
 533       """Conditionally store multiple sparse Tensors."""
 534       out_tensor = utils.smart_cond(
 535           keep_input,
 536           lambda: _store_many_sparse(t, shared_name=map_op_name),
 537           lambda: -1 * array_ops.ones(array_ops.shape(t)[0:1], dtypes.int64))
 538       out_tensor.set_shape([None])  # necessary when t.ndims is unknown
 539       return out_tensor
 540     def _sparse_values_to_keep(t, keep_input):
 541       """Convert a per-row `keep_input` vector to a per-value one."""
 542       # Get the rows of every value in the sparse Tensor.
 543       row_values = t.indices[:, 0]
 544       # The value should be kept iff the row should be kept.
 545       return array_ops.gather(keep_input, row_values)
 546     if keep_input.shape.ndims == 1:
 547       t = sparse_ops.sparse_retain(t, _sparse_values_to_keep(t, keep_input))
 548       store_f = lambda t, name, _: _store_many_sparse(t, shared_name=name)
 549     elif enqueue_many:
 550       store_f = _maybe_store_many_sparse
 551     else:
 552       store_f = _maybe_store_sparse
 553     return store_f(t, map_op_name, keep_input)
 554 
 555   stored_list = [
 556       _maybe_store(t, shared_map_op) for t, shared_map_op
 557       in zip(tensor_list, maybe_shared_map_ops)]
 558   # Since the output of `_store{_many}_sparse is wrapped in a tf.cond `Merge`,
 559   # we can't just get the Op of the resulting tensor.
 560   def _sparse_op(stored):
 561     for input_tensor in stored.op.inputs:
 562       if input_tensor.op.type in ("AddSparseToTensorsMap",
 563                                   "AddManySparseToTensorsMap"):
 564         return input_tensor.op
 565     # If there was no sparse input, then the original stored Tensor wasn't
 566     # sparse and we can just return the original Tensor's Op.
 567     return stored.op
 568   sparse_info_list = [
 569       _sparse_meta_data(t, _sparse_op(stored), shared_map_op)
 570       for t, stored, shared_map_op
 571       in zip(tensor_list, stored_list, maybe_shared_map_ops)]
 572   # Expand dims of stored tensors by 1 for proper enqueue shape
 573   stored_list = [
 574       array_ops.expand_dims(s, [-1]) if s_info.sparse else s
 575       for s, s_info in zip(stored_list, sparse_info_list)]
 576   return stored_list, sparse_info_list

 579 def _store_sparse_tensors_join(tensor_list_list, enqueue_many, keep_input):
 580   """Store SparseTensors for feeding into batch_join, etc."""
 581   (s0, sparse_info_list) = _store_sparse_tensors(
 582       tensor_list_list[0], enqueue_many, keep_input)
 583   stored_list_list = [s0]
 584   for tensor_list in tensor_list_list[1:]:
 585     s, sparse_info_candidate = _store_sparse_tensors(
 586         tensor_list, enqueue_many, keep_input,
 587         [st.map_op for st in sparse_info_list])
 588     if sparse_info_list != sparse_info_candidate:
 589       raise ValueError("Inconsistent SparseTensors list: %s vs. %s"
 590                        % (tensor_list_list[0], tensor_list))

 592         info.merge_with(candidate)
 593         for (info, candidate) in zip(sparse_info_list, sparse_info_candidate)]
 594     stored_list_list.append(s)
 595 
 596   return (stored_list_list, sparse_info_list)

 599 def _restore_sparse_tensors(stored_list, sparse_info_list):
 600   """Restore SparseTensors after dequeue in batch, batch_join, etc."""
 601   received_sequence = isinstance(stored_list, collections.Sequence)
 602   if not received_sequence:
 603     stored_list = (stored_list,)
 604   tensors = [
 605       _restore_sparse(sparse_map_op=info.map_op,
 606                       sparse_handles=array_ops.squeeze(s, [1]),
 607                       rank=(info.rank + 1).value)
 608       if info.sparse else s
 609       for (s, info) in zip(stored_list, sparse_info_list)]
 610   has_st = any(isinstance(x, sparse_tensor.SparseTensor) for x in tensors)
 611   if has_st:
 612     t_values = [
 613         x.values if isinstance(x, sparse_tensor.SparseTensor)
 614         else x
 615         for x in tensors]
 616     with_deps = lambda x: control_flow_ops.with_dependencies(t_values, x)
 617     ensure_restore_tensors = [
 618         sparse_tensor.SparseTensor(indices=with_deps(x.indices),
 619                                    values=with_deps(x.values),
 620                                    dense_shape=with_deps(x.dense_shape))
 621         if isinstance(x, sparse_tensor.SparseTensor)
 622         else with_deps(x)
 623         for x in tensors]
 624   else:
 625     ensure_restore_tensors = tensors
 626   return ensure_restore_tensors if received_sequence else tensors[0]

 737 def _enqueue(queue, tensor_list, threads, enqueue_many, keep_input):
 738   """Enqueue `tensor_list` in `queue`."""

 748         keep_input,

 760            enqueue_many=False, shapes=None, dynamic_pad=False,

 932   The value returned by the function will be of the same type
 933   as `tensors`.
 934

 936   queue is added to the current `Graph`'s `QUEUE_RUNNER` collection.
 937

 939   example.  An input tensor with shape `[x, y, z]` will be output as a tensor

 943   examples, where the first dimension is indexed by example, and all members of
 944   `tensors` should have the same size in the first dimension.  If an input
 945   tensor has shape `[*, x, y, z]`, the output will have shape `[batch_size, x,
 946   y, z]`.  The `capacity` argument controls the how long the prefetching is
 947   allowed to grow the queues.
 948

 950   `tf.errors.OutOfRangeError` if the input queue is exhausted. If this
 951   operation is feeding another input queue, its queue runner will catch
 952   this exception, however, if this operation is used in your main thread
 953   you are responsible for catching this yourself.

 956   (i) the `shapes` argument is passed, or (ii) all of the tensors in

 961   tensors is known, but individual dimensions may have shape `None`.
 962   In this case, for each enqueue the dimensions with value `None`
 963   may have a variable length; upon dequeue, the output tensors will be padded
 964   on the right to the maximum shape of the tensors in the current minibatch.
 965   For numbers, this padding takes value 0.  For strings, this padding is
 966   the empty string.  See `PaddingFIFOQueue` for more info.
 967

 969   `batch_size` is returned when the queue is closed and there are not enough
