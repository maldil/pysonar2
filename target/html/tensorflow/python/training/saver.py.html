 108       spec = BaseSaverBuilder.SaveSpec(var, slice_spec, name, dtype=var.dtype)
 109       super(BaseSaverBuilder.VariableSaveable, self).__init__(var, [spec], name)

 112       restored_tensor = restored_tensors[0]
 113       if restored_shapes is not None:
 114         restored_tensor = array_ops.reshape(restored_tensor, restored_shapes[0])
 115       return state_ops.assign(
 116           self.op,
 117           restored_tensor,
 118           validate_shape=restored_shapes is None and

 121   class ResourceVariableSaveable(SaveableObject):
 122     """SaveableObject implementation that handles ResourceVariables."""
 123 
 124     def __init__(self, var, slice_spec, name):
 125       self._var_device = var.device
 126       self._var_shape = var.shape
 127       if isinstance(var, ops.Tensor):
 128         self.handle_op = var.op.inputs[0]
 129         tensor = var
 130       elif isinstance(var, resource_variable_ops.ResourceVariable):
 131 
 132         def _read_variable_closure(v):
 133           def f():
 134             with ops.device(v.device):
 135               x = v.read_value()
 136               # To allow variables placed on non-CPU devices to be checkpointed,
 137               # we copy them to CPU on the same machine first.
 138               with ops.device("/device:CPU:0"):
 139                 return array_ops.identity(x)
 140           return f
 141 
 142         self.handle_op = var.handle
 143         tensor = _read_variable_closure(var)
 144       else:
 145         raise ValueError(
 146             "Saveable is neither a resource variable nor a read operation."
 147             " Got: %s" % repr(var))
 148       spec = BaseSaverBuilder.SaveSpec(tensor, slice_spec, name,
 149                                        dtype=var.dtype)
 150       super(BaseSaverBuilder.ResourceVariableSaveable, self).__init__(
 151           var, [spec], name)
 152 
 153     def restore(self, restored_tensors, restored_shapes):
 154       restored_tensor = restored_tensors[0]
 155       if restored_shapes is not None:
 156         restored_tensor = array_ops.reshape(restored_tensor, restored_shapes[0])
 157       # Copy the restored tensor to the variable's device.
 158       with ops.device(self._var_device):
 159         restored_tensor = array_ops.identity(restored_tensor)
 160         return resource_variable_ops.shape_safe_assign_variable_handle(
 161             self.handle_op, self._var_shape, restored_tensor)

 166   def save_op(self, filename_tensor, saveables):
 167     """Create an Op to save 'saveables'.
 168 
 169     This is intended to be overridden by subclasses that want to generate
 170     different Ops.
 171 
 172     Args:
 173       filename_tensor: String Tensor.
 174       saveables: A list of BaseSaverBuilder.SaveableObject objects.
 175 
 176     Returns:
 177       An Operation that save the variables.
 178 
 179     Raises:
 180       RuntimeError: (implementation detail) if "self._write_version" is an
 181         unexpected value.
 182     """
 183     # pylint: disable=protected-access
 184     tensor_names = []
 185     tensors = []
 186     tensor_slices = []
 187     for saveable in saveables:
 188       for spec in saveable.specs:
 189         tensor_names.append(spec.name)
 190         tensors.append(spec.tensor)
 191         tensor_slices.append(spec.slice_spec)
 192     if self._write_version == saver_pb2.SaverDef.V1:
 193       return io_ops._save(
 194           filename=filename_tensor,
 195           tensor_names=tensor_names,
 196           tensors=tensors,
 197           tensor_slices=tensor_slices)
 198     elif self._write_version == saver_pb2.SaverDef.V2:
 199       # "filename_tensor" is interpreted *NOT AS A FILENAME*, but as a prefix
 200       # of a V2 checkpoint: e.g. "/fs/train/ckpt-<step>/tmp/worker<i>-<step>".
 201       return io_ops.save_v2(filename_tensor, tensor_names, tensor_slices,
 202                             tensors)
 203     else:
 204       raise RuntimeError("Unexpected write_version: " + self._write_version)

 206   def bulk_restore(self, filename_tensor, saveables, preferred_shard,
 207                    restore_sequentially):
 208     """Restore all tensors contained in saveables.
 209 
 210     By default, this issues separate calls to `restore_op` for each saveable.
 211     Subclasses may override to load multiple saveables in a single call.
 212 
 213     Args:
 214       filename_tensor: String Tensor.
 215       saveables: List of BaseSaverBuilder.SaveableObject objects.
 216       preferred_shard: Int.  Shard to open first when loading a sharded file.
 217       restore_sequentially: Unused.  Bool.  If true, each restore is sequential.
 218 
 219     Returns:
 220       A list of Tensors resulting from reading 'saveable' from
 221         'filename'.
 222 
 223     """
 224     del restore_sequentially
 225     all_tensors = []
 226     for saveable in saveables:
 227       with ops.device(_set_cpu0(saveable.device) if saveable.device else None):
 228         all_tensors.extend(
 229             self.restore_op(filename_tensor, saveable, preferred_shard))
 230     return all_tensors

 287   def _AddShardedSaveOpsForV2(self, checkpoint_prefix, per_device):
 288     """Add ops to save the params per shard, for the V2 format.
 289 
 290     Note that the sharded save procedure for the V2 format is different from
 291     V1: there is a special "merge" step that merges the small metadata produced
 292     from each device.
 293 
 294     Args:
 295       checkpoint_prefix: scalar String Tensor.  Interpreted *NOT AS A
 296         FILENAME*, but as a prefix of a V2 checkpoint;
 297       per_device: A list of (device, BaseSaverBuilder.VarToSave) pairs, as
 298         returned by _GroupByDevices().
 299 
 300     Returns:
 301       An op to save the variables, which, when evaluated, returns the prefix
 302         "<user-fed prefix>" only and does not include the sharded spec suffix.
 303     """
 304     # IMPLEMENTATION DETAILS: most clients should skip.
 305     #
 306     # Suffix for any well-formed "checkpoint_prefix", when sharded.
 307     # Transformations:
 308     # * Users pass in "save_path" in save() and restore().  Say "myckpt".
 309     # * checkpoint_prefix gets fed <save_path><_SHARDED_SUFFIX>.
 310     #
 311     # Example:
 312     #   During runtime, a temporary directory is first created, which contains
 313     #   files
 314     #
 315     #     <train dir>/myckpt_temp/
 316     #        part-?????-of-?????{.index, .data-00000-of-00001}
 317     #
 318     #   Before .save() finishes, they will be (hopefully, atomically) renamed to
 319     #
 320     #     <train dir>/
 321     #        myckpt{.index, .data-?????-of-?????}
 322     #
 323     # Users only need to interact with the user-specified prefix, which is
 324     # "<train dir>/myckpt" in this case.  Save() and Restore() work with the
 325     # prefix directly, instead of any physical pathname.  (On failure and
 326     # subsequent restore, an outdated and orphaned temporary directory can be
 327     # safely removed.)
 328     _SHARDED_SUFFIX = "_temp_%s/part" % uuid.uuid4().hex
 329     tmp_checkpoint_prefix = string_ops.string_join(
 330         [checkpoint_prefix, _SHARDED_SUFFIX])
 331 
 332     num_shards = len(per_device)
 333     sharded_saves = []
 334     sharded_prefixes = []
 335     num_shards_tensor = constant_op.constant(num_shards, name="num_shards")
 336     last_device = None
 337     for shard, (device, saveables) in enumerate(per_device):
 338       last_device = device
 339       with ops.device(_set_cpu0(device)):
 340         sharded_filename = self.sharded_filename(tmp_checkpoint_prefix, shard,
 341                                                  num_shards_tensor)
 342         sharded_prefixes.append(sharded_filename)
 343         sharded_saves.append(self._AddSaveOps(sharded_filename, saveables))
 344 
 345     with ops.control_dependencies([x.op for x in sharded_saves]):
 346       # Co-locates the merge step with the last device.
 347       with ops.device(_set_cpu0(last_device)):
 348         # V2 format write path consists of a metadata merge step.  Once merged,
 349         # attempts to delete the temporary directory, "<user-fed prefix>_temp".
 350         merge_step = gen_io_ops.merge_v2_checkpoints(
 351             sharded_prefixes, checkpoint_prefix, delete_old_dirs=True)
 352         with ops.control_dependencies([merge_step]):
 353           # Returns the prefix "<user-fed prefix>" only.  DOES NOT include the
 354           # sharded spec suffix.
 355           return array_ops.identity(checkpoint_prefix)
 356 
 357   def _AddShardedSaveOps(self, filename_tensor, per_device):
 358     """Add ops to save the params per shard.
 359 
 360     Args:
 361       filename_tensor: a scalar String Tensor.
 362       per_device: A list of (device, BaseSaverBuilder.SaveableObject) pairs, as
 363         returned by _GroupByDevices().
 364 
 365     Returns:
 366       An op to save the variables.
 367     """
 368     if self._write_version == saver_pb2.SaverDef.V2:
 369       return self._AddShardedSaveOpsForV2(filename_tensor, per_device)
 370 
 371     num_shards = len(per_device)
 372     sharded_saves = []
 373     num_shards_tensor = constant_op.constant(num_shards, name="num_shards")
 374     for shard, (device, saveables) in enumerate(per_device):
 375       with ops.device(device):
 376         sharded_filename = self.sharded_filename(filename_tensor, shard,
 377                                                  num_shards_tensor)
 378         sharded_saves.append(self._AddSaveOps(sharded_filename, saveables))
 379     # Return the sharded name for the save path.
 380     with ops.control_dependencies([x.op for x in sharded_saves]):
 381       return gen_io_ops.sharded_filespec(filename_tensor, num_shards_tensor)

 383   def _AddRestoreOps(self,
 384                      filename_tensor,
 385                      saveables,
 386                      restore_sequentially,
 387                      reshape,
 388                      preferred_shard=-1,
 389                      name="restore_all"):
 390     """Add operations to restore saveables.
 391 
 392     Args:
 393       filename_tensor: Tensor for the path of the file to load.
 394       saveables: A list of SaveableObject objects.
 395       restore_sequentially: True if we want to restore variables sequentially
 396         within a shard.
 397       reshape: True if we want to reshape loaded tensors to the shape of
 398         the corresponding variable.
 399       preferred_shard: Shard to open first when loading a sharded file.
 400       name: Name for the returned op.
 401 
 402     Returns:
 403       An Operation that restores the variables.
 404     """
 405     all_tensors = self.bulk_restore(filename_tensor, saveables, preferred_shard,
 406                                     restore_sequentially)
 407 
 408     assign_ops = []
 409     idx = 0
 410     # Load and optionally reshape on the CPU, as string tensors are not
 411     # available on the GPU.
 412     # TODO(touts): Re-enable restore on GPU when we can support annotating
 413     # string tensors as "HostMemory" inputs.
 414     for saveable in saveables:
 415       shapes = None
 416       if reshape:
 417         # Compute the shapes, let the restore op decide if and how to do
 418         # the reshape.
 419         shapes = []
 420         for spec in saveable.specs:
 421           v = spec.tensor
 422           shape = v.get_shape()
 423           if not shape.is_fully_defined():
 424             shape = array_ops.shape(v)
 425           shapes.append(shape)
 426       saveable_tensors = all_tensors[idx:idx + len(saveable.specs)]
 427       idx += len(saveable.specs)
 428       assign_ops.append(saveable.restore(saveable_tensors, shapes))
 429 
 430     # Create a Noop that has control dependencies from all the updates.
 431     return control_flow_ops.group(*assign_ops, name=name)

 433   def _AddShardedRestoreOps(self, filename_tensor, per_device,
 434                             restore_sequentially, reshape):
 435     """Add Ops to restore variables from multiple devices.
 436 
 437     Args:
 438       filename_tensor: Tensor for the path of the file to load.
 439       per_device: A list of (device, SaveableObject) pairs, as
 440         returned by _GroupByDevices().
 441       restore_sequentially: True if we want to restore variables sequentially
 442         within a shard.
 443       reshape: True if we want to reshape loaded tensors to the shape of
 444         the corresponding variable.
 445 
 446     Returns:
 447       An Operation that restores the variables.
 448     """
 449     sharded_restores = []
 450     for shard, (device, saveables) in enumerate(per_device):
 451       with ops.device(device):
 452         sharded_restores.append(
 453             self._AddRestoreOps(
 454                 filename_tensor,
 455                 saveables,
 456                 restore_sequentially,
 457                 reshape,
 458                 preferred_shard=shard,
 459                 name="restore_shard"))
 460     return control_flow_ops.group(*sharded_restores, name="restore_all")

 466   def _GroupByDevices(self, saveables):
 467     """Group Variable tensor slices per device.
 468 
 469     TODO(touts): Make sure that all the devices found are on different
 470     job/replica/task/cpu|gpu.  It would be bad if 2 were on the same device.
 471     It can happen if the devices are unspecified.
 472 
 473     Args:
 474       saveables: A list of BaseSaverBuilder.SaveableObject objects.
 475 
 476     Returns:
 477       A list of tuples: (device_name, BaseSaverBuilder.SaveableObject) tuples.
 478       The list is sorted by ascending device_name.
 479 
 480     Raises:
 481       ValueError: If the tensors of a saveable are on different devices.
 482     """
 483     per_device = collections.defaultdict(lambda: [])
 484     for saveable in saveables:
 485       canonical_device = set(
 486           pydev.canonical_name(spec.tensor.device) for spec in saveable.specs)
 487       if len(canonical_device) != 1:
 488         raise ValueError("All tensors of a saveable object must be "
 489                          "on the same device: %s" % saveable.name)
 490       per_device[canonical_device.pop()].append(saveable)
 491     return sorted(per_device.items(), key=lambda t: t[0])

 493   @staticmethod
 494   def OpListToDict(op_list, convert_variable_to_tensor=True):
 495     """Create a dictionary of names to operation lists.
 496 
 497     Args:
 498       op_list: A list, tuple, or set of Variables or SaveableObjects.
 499       convert_variable_to_tensor: Whether or not to convert single Variables
 500         with no slice info into Tensors.
 501 
 502     Returns:
 503       A dictionary of names to the operations that must be saved under
 504       that name.  Variables with save_slice_info are grouped together under the
 505       same key in no particular order.
 506 
 507     Raises:
 508       TypeError: If the type of op_list or its elements is not supported.
 509       ValueError: If at least two saveables share the same name.
 510     """
 511     if not isinstance(op_list, (list, tuple, set)):
 512       raise TypeError("Variables to save should be passed in a dict or a "
 513                       "list: %s" % op_list)
 514     # When ResourceVariables are converted to Tensors, read ops are added to the
 515     # graph. Sorting the op_list ensures that the resulting graph is always
 516     # constructed in a deterministic way:
 517     op_list = sorted(op_list, key=lambda x: x.name)
 518     names_to_saveables = {}
 519     # pylint: disable=protected-access
 520     for var in op_list:
 521       if isinstance(var, BaseSaverBuilder.SaveableObject):
 522         names_to_saveables[var.name] = var
 523       elif isinstance(var, variables.PartitionedVariable):
 524         if var.name in names_to_saveables:
 525           raise ValueError("At least two variables have the same name: %s" %
 526                            var.name)
 527         names_to_saveables[var.name] = var
 528       elif isinstance(var, variables.Variable) and var._save_slice_info:
 529         name = var._save_slice_info.full_name
 530         if name in names_to_saveables:
 531           if not isinstance(names_to_saveables[name], list):
 532             raise ValueError("Mixing slices and non-slices with the same name: "
 533                              "%s" % name)
 534           names_to_saveables[name].append(var)
 535         else:
 536           names_to_saveables[name] = [var]
 537       elif (isinstance(var, checkpointable.CheckpointableBase)
 538             and not isinstance(var, variables.Variable)):
 539         checkpointable_saveables = [
 540             (factory() if callable(factory) else factory)
 541             for factory in var._gather_saveables_for_checkpoint().values()]
 542         names_to_saveables.update(
 543             BaseSaverBuilder.OpListToDict(checkpointable_saveables))
 544       else:
 545         if context.executing_eagerly():
 546           if not isinstance(var, resource_variable_ops.ResourceVariable):
 547             raise ValueError(
 548                 "Can only save/restore ResourceVariables when eager execution "
 549                 "is enabled, type: %s." % type(var))
 550           set_var = names_to_saveables.setdefault(var._shared_name, var)
 551           if set_var is not var:
 552             raise ValueError(
 553                 ("Two different ResourceVariable objects with the same "
 554                  "shared_name '%s' were passed to the Saver. This likely means "
 555                  "that they were created in different Graphs or isolation "
 556                  "contexts, and may not be checkpointed together.") %
 557                 (var._shared_name,))
 558         else:
 559           if convert_variable_to_tensor:
 560             if isinstance(var, resource_variable_ops.ResourceVariable):
 561               var = var._graph_element  # pylint: disable=protected-access
 562             else:
 563               var = ops.internal_convert_to_tensor(var, as_ref=True)
 564             if not BaseSaverBuilder._IsVariable(var):
 565               raise TypeError("Variable to save is not a Variable: %s" % var)
 566           if var.op.type == "ReadVariableOp":
 567             name = var.op.inputs[0].op.name
 568           else:
 569             name = var.op.name
 570           if name in names_to_saveables:
 571             raise ValueError("At least two variables have the same name: %s" %
 572                              name)
 573           names_to_saveables[name] = var
 574 
 575       # pylint: enable=protected-access

 579   def SaveableObjectsForOp(op, name):
 580     """Create `SaveableObject`s from an operation.
 581 
 582     Args:
 583       op: A variable, operation, or SaveableObject to coerce into a
 584         SaveableObject.
 585       name: A string name for the SaveableObject.
 586 
 587     Yields:
 588       `SaveableObject`s which together save/restore `op`.
 589 
 590     Raises:
 591       TypeError: If `name` is not a string.
 592       ValueError: For operations with no known conversion to SaveableObject.
 593     """
 594     if not isinstance(name, six.string_types):
 595       raise TypeError(
 596           "names_to_saveables must be a dict mapping string names to "
 597           "checkpointable operations. Name is not a string: %s" % name)
 598     if isinstance(op, BaseSaverBuilder.SaveableObject):
 599       yield op
 600     elif isinstance(op, (list, tuple, variables.PartitionedVariable)):
 601       if isinstance(op, variables.PartitionedVariable):
 602         op = list(op)
 603       # A set of slices.
 604       slice_name = None
 605       # pylint: disable=protected-access
 606       for variable in op:
 607         if not isinstance(variable, variables.Variable):
 608           raise ValueError("Slices must all be Variables: %s" % variable)
 609         if not variable._save_slice_info:
 610           raise ValueError("Slices must all be slices: %s" % variable)
 611         if slice_name is None:
 612           slice_name = variable._save_slice_info.full_name
 613         elif slice_name != variable._save_slice_info.full_name:
 614           raise ValueError(
 615               "Slices must all be from the same tensor: %s != %s" %
 616               (slice_name, variable._save_slice_info.full_name))
 617         if variable.op.type in ["Variable", "VariableV2",
 618                                 "AutoReloadVariable"]:
 619           yield BaseSaverBuilder.VariableSaveable(
 620               variable, variable._save_slice_info.spec, name)
 621         else:
 622           yield BaseSaverBuilder.ResourceVariableSaveable(
 623               variable, variable._save_slice_info.spec, name)
 624       # pylint: enable=protected-access
 625     elif isinstance(op, checkpointable.CheckpointableBase) and not isinstance(
 626         op, variables.Variable):
 627       # pylint: disable=protected-access
 628       for attr, factory in op._gather_saveables_for_checkpoint().items():
 629         op = (factory(name + "_" + attr) if callable(factory) else factory)
 630         for op in BaseSaverBuilder.SaveableObjectsForOp(op, op.name):
 631           yield op
 632       # pylint: enable=protected-access
 633     else:
 634       # A variable or tensor.
 635       if context.executing_eagerly():
 636         if not isinstance(op, resource_variable_ops.ResourceVariable):
 637           raise ValueError("Can only save/restore ResourceVariable eager "
 638                            "mode is enabled, type: %s." % type(op))
 639         yield BaseSaverBuilder.ResourceVariableSaveable(op, "", name)
 640       else:
 641         if isinstance(op, resource_variable_ops.ResourceVariable):
 642           variable = op._graph_element  # pylint: disable=protected-access
 643         else:
 644           variable = ops.internal_convert_to_tensor(op, as_ref=True)
 645         if not BaseSaverBuilder._IsVariable(variable):
 646           raise TypeError("names_to_saveables must be a dict mapping string "
 647                           "names to Tensors/Variables. Not a variable: %s" %
 648                           variable)
 649         if variable.op.type in ["Variable", "VariableV2",
 650                                 "AutoReloadVariable"]:
 651           yield BaseSaverBuilder.VariableSaveable(variable, "", name)
 652         else:
 653           yield BaseSaverBuilder.ResourceVariableSaveable(
 654               variable, "", name)

 894   """Saves and restores variables.
 895 
 896   See [Variables](https://tensorflow.org/guide/variables)
 897   for an overview of variables, saving and restoring.
 898 
 899   The `Saver` class adds ops to save and restore variables to and from
 900   *checkpoints*.  It also provides convenience methods to run these ops.
 901 
 902   Checkpoints are binary files in a proprietary format which map variable names
 903   to tensor values.  The best way to examine the contents of a checkpoint is to
 904   load it using a `Saver`.
 905 
 906   Savers can automatically number checkpoint filenames with a provided counter.
 907   This lets you keep multiple checkpoints at different steps while training a
 908   model.  For example you can number the checkpoint filenames with the training
 909   step number.  To avoid filling up disks, savers manage checkpoint files
 910   automatically. For example, they can keep only the N most recent files, or
 911   one checkpoint for every N hours of training.
 912 
 913   You number checkpoint filenames by passing a value to the optional
 914   `global_step` argument to `save()`:
 915 
 916   ```python
 917   saver.save(sess, 'my-model', global_step=0) ==> filename: 'my-model-0'
 918   ...
 919   saver.save(sess, 'my-model', global_step=1000) ==> filename: 'my-model-1000'
 920   ```
 921 
 922   Additionally, optional arguments to the `Saver()` constructor let you control
 923   the proliferation of checkpoint files on disk:
 924 
 925   * `max_to_keep` indicates the maximum number of recent checkpoint files to
 926     keep.  As new files are created, older files are deleted.   If None or 0,
 927     no checkpoints are deleted from the filesystem but only the last one is
 928     kept in the `checkpoint` file.  Defaults to 5 (that is, the 5 most recent
 929     checkpoint files are kept.)
 930 
 931   * `keep_checkpoint_every_n_hours`: In addition to keeping the most recent
 932     `max_to_keep` checkpoint files, you might want to keep one checkpoint file
 933     for every N hours of training.  This can be useful if you want to later
 934     analyze how a model progressed during a long training session.  For
 935     example, passing `keep_checkpoint_every_n_hours=2` ensures that you keep
 936     one checkpoint file for every 2 hours of training.  The default value of
 937     10,000 hours effectively disables the feature.
 938 
 939   Note that you still have to call the `save()` method to save the model.
 940   Passing these arguments to the constructor will not save variables
 941   automatically for you.
 942 
 943   A training program that saves regularly looks like:
 944 
 945   ```python
 946   ...
 947   # Create a saver.
 948   saver = tf.train.Saver(...variables...)
 949   # Launch the graph and train, saving the model every 1,000 steps.
 950   sess = tf.Session()
 951   for step in xrange(1000000):
 952       sess.run(..training_op..)
 953       if step % 1000 == 0:
 954           # Append the step number to the checkpoint name:
 955           saver.save(sess, 'my-model', global_step=step)
 956   ```
 957 
 958   In addition to checkpoint files, savers keep a protocol buffer on disk with
 959   the list of recent checkpoints. This is used to manage numbered checkpoint
 960   files and by `latest_checkpoint()`, which makes it easy to discover the path
 961   to the most recent checkpoint. That protocol buffer is stored in a file named
 962   'checkpoint' next to the checkpoint files.
 963 
 964   If you create several savers, you can specify a different filename for the
 965   protocol buffer file in the call to `save()`.
 966   """

1194   def _RecordLastCheckpoint(self, latest_save_path):
1195     """Manages the list of the latest checkpoints."""
1196     if not self.saver_def.max_to_keep:
1197       return
1198     # Remove first from list if the same name was used before.
1199     for p in self._last_checkpoints:
1200       if latest_save_path == self._CheckpointFilename(p):
1201         self._last_checkpoints.remove(p)
1202     # Append new path to list
1203     self._last_checkpoints.append((latest_save_path, time.time()))
1204 
1205     # If more than max_to_keep, remove oldest.
1206     if len(self._last_checkpoints) > self.saver_def.max_to_keep:
1207       self._checkpoints_to_be_deleted.append(self._last_checkpoints.pop(0))

1209   def _MaybeDeleteOldCheckpoints(self, meta_graph_suffix="meta"):
1210     """Deletes old checkpoints if necessary.
1211 
1212     `self._checkpoints_to_be_deleted` is going to contain checkpoints that are
1213     over `max_to_keep`.  They are going to be deleted.  If
1214     `keep_checkpoint_every_n_hours` was specified, keep an additional checkpoint
1215     every `N` hours. For example, if `N` is 0.5, an additional checkpoint is
1216     kept for every 0.5 hours of training; if `N` is 10, an additional
1217     checkpoint is kept for every 10 hours of training.
1218 
1219     Args:
1220       meta_graph_suffix: Suffix for `MetaGraphDef` file. Defaults to 'meta'.
1221     """
1222     if self._checkpoints_to_be_deleted:
1223       p = self._checkpoints_to_be_deleted.pop(0)
1224       # Do not delete the file if we keep_checkpoint_every_n_hours is set and we
1225       # have reached N hours of training.
1226       should_keep = p[1] > self._next_checkpoint_time
1227       if should_keep:
1228         self._next_checkpoint_time += (
1229             self.saver_def.keep_checkpoint_every_n_hours * 3600)
1230         return
1231 
1232       # Otherwise delete the files.
1233       try:
1234         checkpoint_management.remove_checkpoint(
1235             self._CheckpointFilename(p), self.saver_def.version,
1236             meta_graph_suffix)
1237       except Exception as e:  # pylint: disable=broad-except
1238         logging.warning("Ignoring: %s", str(e))

1343   def save(self,
1344            sess,
1345            save_path,
1346            global_step=None,
1347            latest_filename=None,
1348            meta_graph_suffix="meta",
1349            write_meta_graph=True,
1350            write_state=True,
1351            strip_default_attrs=False):
1352     # pylint: disable=line-too-long
1353     """Saves variables.
1354 
1355     This method runs the ops added by the constructor for saving variables.
1356     It requires a session in which the graph was launched.  The variables to
1357     save must also have been initialized.
1358 
1359     The method returns the path prefix of the newly created checkpoint files.
1360     This string can be passed directly to a call to `restore()`.
1361 
1362     Args:
1363       sess: A Session to use to save the variables.
1364       save_path: String.  Prefix of filenames created for the checkpoint.
1365       global_step: If provided the global step number is appended to
1366         `save_path` to create the checkpoint filenames. The optional argument
1367         can be a `Tensor`, a `Tensor` name or an integer.
1368       latest_filename: Optional name for the protocol buffer file that will
1369         contains the list of most recent checkpoints.  That file,
1370         kept in the same directory as the checkpoint files, is automatically
1371         managed by the saver to keep track of recent checkpoints.  Defaults to
1372         'checkpoint'.
1373       meta_graph_suffix: Suffix for `MetaGraphDef` file. Defaults to 'meta'.
1374       write_meta_graph: `Boolean` indicating whether or not to write the meta
1375         graph file.
1376       write_state: `Boolean` indicating whether or not to write the
1377         `CheckpointStateProto`.
1378       strip_default_attrs: Boolean. If `True`, default-valued attributes will be
1379         removed from the NodeDefs. For a detailed guide, see
1380         [Stripping Default-Valued Attributes](https://github.com/tensorflow/tensorflow/blob/master/tensorflow/python/saved_model/README.md#stripping-default-valued-attributes).
1381 
1382     Returns:
1383       A string: path prefix used for the checkpoint files.  If the saver is
1384         sharded, this string ends with: '-?????-of-nnnnn' where 'nnnnn'
1385         is the number of shards created.
1386       If the saver is empty, returns None.
1387 
1388     Raises:
1389       TypeError: If `sess` is not a `Session`.
1390       ValueError: If `latest_filename` contains path components, or if it
1391         collides with `save_path`.
1392       RuntimeError: If save and restore ops weren't built.
1393     """
1394     # pylint: enable=line-too-long
1395     if not self._is_built and not context.executing_eagerly():
1396       raise RuntimeError(
1397           "`build()` should be called before save if defer_build==True")
1398     if latest_filename is None:
1399       latest_filename = "checkpoint"
1400     if self._write_version != saver_pb2.SaverDef.V2:
1401       logging.warning("*******************************************************")
1402       logging.warning("TensorFlow's V1 checkpoint format has been deprecated.")
1403       logging.warning("Consider switching to the more efficient V2 format:")
1404       logging.warning("   `tf.train.Saver(write_version=tf.train.SaverDef.V2)`")
1405       logging.warning("now on by default.")
1406       logging.warning("*******************************************************")
1407 
1408     if os.path.split(latest_filename)[0]:
1409       raise ValueError("'latest_filename' must not contain path components")
1410 
1411     if global_step is not None:
1412       if not isinstance(global_step, compat.integral_types):
1413         global_step = training_util.global_step(sess, global_step)
1414       checkpoint_file = "%s-%d" % (save_path, global_step)
1415       if self._pad_step_number:
1416         # Zero-pads the step numbers, so that they are sorted when listed.
1417         checkpoint_file = "%s-%s" % (save_path, "{:08d}".format(global_step))
1418     else:
1419       checkpoint_file = save_path
1420       if os.path.basename(
1421           save_path) == latest_filename and not self._sharded:
1422         # Guard against collision between data file and checkpoint state file.
1423         raise ValueError(
1424             "'latest_filename' collides with 'save_path': '%s' and '%s'" %
1425             (latest_filename, save_path))
1426 
1427     if (not context.executing_eagerly() and
1428         not isinstance(sess, session.SessionInterface)):
1429       raise TypeError("'sess' must be a Session; %s" % sess)
1430 
1431     save_path_parent = os.path.dirname(save_path)
1432     if not self._is_empty:
1433       try:
1434         if context.executing_eagerly():
1435           self._build_eager(
1436               checkpoint_file, build_save=True, build_restore=False)
1437           model_checkpoint_path = self.saver_def.save_tensor_name
1438         else:
1439           model_checkpoint_path = sess.run(
1440               self.saver_def.save_tensor_name,
1441               {self.saver_def.filename_tensor_name: checkpoint_file})
1442 
1443         model_checkpoint_path = compat.as_str(model_checkpoint_path)
1444         if write_state:
1445           self._RecordLastCheckpoint(model_checkpoint_path)
1446           checkpoint_management.update_checkpoint_state_internal(
1447               save_dir=save_path_parent,
1448               model_checkpoint_path=model_checkpoint_path,
1449               all_model_checkpoint_paths=self.last_checkpoints,
1450               latest_filename=latest_filename,
1451               save_relative_paths=self._save_relative_paths)
1452           self._MaybeDeleteOldCheckpoints(meta_graph_suffix=meta_graph_suffix)
1453       except (errors.FailedPreconditionError, errors.NotFoundError) as exc:
1454         if not gfile.IsDirectory(save_path_parent):
1455           exc = ValueError(
1456               "Parent directory of {} doesn't exist, can't save.".format(
1457                   save_path))
1458         raise exc
1459 
1460     if write_meta_graph:
1461       meta_graph_filename = checkpoint_management.meta_graph_filename(
1462           checkpoint_file, meta_graph_suffix=meta_graph_suffix)
1463       if not context.executing_eagerly():
1464         with sess.graph.as_default():
1465           self.export_meta_graph(
1466               meta_graph_filename, strip_default_attrs=strip_default_attrs)
1467 
1468     if self._is_empty:
1469       return None
1470     else:
1471       return model_checkpoint_path

1513   def restore(self, sess, save_path):
1514     """Restores previously saved variables.
1515 
1516     This method runs the ops added by the constructor for restoring variables.
1517     It requires a session in which the graph was launched.  The variables to
1518     restore do not have to have been initialized, as restoring is itself a way
1519     to initialize variables.
1520 
1521     The `save_path` argument is typically a value previously returned from a
1522     `save()` call, or a call to `latest_checkpoint()`.
1523 
1524     Args:
1525       sess: A `Session` to use to restore the parameters. None in eager mode.
1526       save_path: Path where parameters were previously saved.
1527 
1528     Raises:
1529       ValueError: If save_path is None or not a valid checkpoint.
1530     """
1531     if self._is_empty:
1532       return
1533     if save_path is None:
1534       raise ValueError("Can't load save_path when it is None.")
1535 
1536     if not checkpoint_management.checkpoint_exists(compat.as_text(save_path)):
1537       raise ValueError("The passed save_path is not a valid checkpoint: "
1538                        + compat.as_text(save_path))
1539 
1540     logging.info("Restoring parameters from %s", compat.as_text(save_path))
1541     try:
1542       if context.executing_eagerly():
1543         self._build_eager(save_path, build_save=False, build_restore=True)
1544       else:
1545         sess.run(self.saver_def.restore_op_name,
1546                  {self.saver_def.filename_tensor_name: save_path})
1547     except errors.NotFoundError as err:
1548       # There are three common conditions that might cause this error:
1549       # 0. The file is missing. We ignore here, as this is checked above.
1550       # 1. This is an object-based checkpoint trying name-based loading.
1551       # 2. The graph has been altered and a variable or other name is missing.
1552 
1553       # 1. The checkpoint would not be loaded successfully as is. Try to parse
1554       # it as an object-based checkpoint.
1555       try:
1556         names_to_keys = object_graph_key_mapping(save_path)
1557       except errors.NotFoundError:
1558         # 2. This is not an object-based checkpoint, which likely means there
1559         # is a graph mismatch. Re-raise the original error with
1560         # a helpful message (b/110263146)
1561         raise _wrap_restore_error_with_msg(
1562             err, "a Variable name or other graph key that is missing")
1563 
1564       # This is an object-based checkpoint. We'll print a warning and then do
1565       # the restore.
1566       logging.warning(
1567           "Restoring an object-based checkpoint using a name-based saver. This "
1568           "may be somewhat fragile, and will re-build the Saver. Instead, "
1569           "consider loading object-based checkpoints using "
1570           "tf.train.Checkpoint().")
1571       self._object_restore_saver = saver_from_object_based_checkpoint(
1572           checkpoint_path=save_path,
1573           var_list=self._var_list,
1574           builder=self._builder,
1575           names_to_keys=names_to_keys,
1576           cached_saver=self._object_restore_saver)
1577       self._object_restore_saver.restore(sess=sess, save_path=save_path)
1578     except errors.InvalidArgumentError as err:
1579       # There is a mismatch between the graph and the checkpoint being loaded.
1580       # We add a more reasonable error message here to help users (b/110263146)
1581       raise _wrap_restore_error_with_msg(
1582           err, "a mismatch between the current graph and the graph")

1601 
1602   This function takes a `MetaGraphDef` protocol buffer as input. If
1603   the argument is a file containing a `MetaGraphDef` protocol buffer ,
1604   it constructs a protocol buffer from the file content. The function
1605   then adds all the nodes from the `graph_def` field to the
1606   current graph, recreates all the collections, and returns a saver
1607   constructed from the `saver_def` field.
1608

1610 
1611   * Serialize a graph along with other Python objects such as `QueueRunner`,
1612     `Variable` into a `MetaGraphDef`.
1613 
1614   * Restart training from a saved graph and checkpoints.
1615 
1616   * Run inference from a saved graph and checkpoints.
1617 
1618   ```Python
1619   ...
1620   # Create a saver.
1621   saver = tf.train.Saver(...variables...)
1622   # Remember the training_op we want to run by adding it to a collection.
1623   tf.add_to_collection('train_op', train_op)
1624   sess = tf.Session()
1625   for step in xrange(1000000):
1626       sess.run(train_op)
1627       if step % 1000 == 0:
1628           # Saves checkpoint, which by default also exports a meta_graph
1629           # named 'my-model-global_step.meta'.
1630           saver.save(sess, 'my-model', global_step=step)
1631   ```
1632 
1633   Later we can continue training from this saved `meta_graph` without building
1634   the model from scratch.
1635 
1636   ```Python
1637   with tf.Session() as sess:
1638     new_saver = tf.train.import_meta_graph('my-save-dir/my-model-10000.meta')
1639     new_saver.restore(sess, 'my-save-dir/my-model-10000')
1640     # tf.get_collection() returns a list. In this example we only want the
1641     # first one.

1645   ```

1648   device assignments have not changed.

1651     meta_graph_or_file: `MetaGraphDef` protocol buffer or filename (including
1652       the path) containing a `MetaGraphDef`.
1653     clear_devices: Whether or not to clear the device field for an `Operation`
1654       or `Tensor` during import.

1656       initializing from protocol buffer.
1657     **kwargs: Optional keyed arguments.
1658 
1659   Returns:
1660     A saver constructed from `saver_def` in `MetaGraphDef` or None.
1661 
1662     A None value is returned if no variables exist in the `MetaGraphDef`
1663     (i.e., there are no variables to restore).

1841 def saver_from_object_based_checkpoint(
1842     checkpoint_path, var_list=None, builder=None, names_to_keys=None,
1843     cached_saver=None):
1844   """Return a `Saver` which reads from an object-based checkpoint.
1845 
1846   This function validates that all variables in the variables list are remapped
1847   in the object-based checkpoint (or `names_to_keys` dict if provided). A
1848   saver will be created with the list of remapped variables.
1849 
1850   The `cached_saver` argument allows the user to pass in a previously created
1851   saver, so multiple `saver.restore()` calls don't pollute the graph when graph
1852   building. This assumes that keys are consistent, meaning that the
1853     1) `checkpoint_path` checkpoint, and
1854     2) checkpoint used to create the `cached_saver`
1855   are the same type of object-based checkpoint. If this argument is set, this
1856   function will simply validate that all variables have been remapped by the
1857   checkpoint at `checkpoint_path`.
1858 
1859   Note that in general, `tf.train.Checkpoint` should be used to restore/save an
1860   object-based checkpoint.
1861 
1862   Args:
1863     checkpoint_path: string, path to object-based checkpoint
1864     var_list: list of `Variables` that appear in the checkpoint. If `None`,
1865       `var_list` will be set to all saveable objects.
1866     builder: a `BaseSaverBuilder` instance. If `None`, a new `BulkSaverBuilder`
1867       will be created.
1868     names_to_keys: dict mapping string tensor names to checkpooint keys. If
1869       `None`, this dict will be generated from the checkpoint file.
1870     cached_saver: Cached `Saver` object with remapped variables.
1871 
1872   Returns:
1873     `Saver` with remapped variables for reading from an object-based checkpoint.
1874 
1875   Raises:
1876     ValueError if the checkpoint provided is not an object-based checkpoint.
1877     NotFoundError: If one of the variables in `var_list` can not be found in the
1878       checkpoint. This could mean the checkpoint or `names_to_keys` mapping is
1879       missing the variable.
1880   """
1881   if names_to_keys is None:
1882     try:
1883       names_to_keys = object_graph_key_mapping(checkpoint_path)
1884     except errors.NotFoundError:
1885       raise ValueError("Checkpoint in %s not an object-based checkpoint."
1886                        % checkpoint_path)
1887   if var_list is None:
1888     var_list = variables._all_saveable_objects()  # pylint: disable=protected-access
1889   if builder is None:
1890     builder = BulkSaverBuilder()
1891 
1892   saveables = builder._ValidateAndSliceInputs(var_list)  # pylint: disable=protected-access
1893   for saveable in saveables:
1894     for spec in saveable.specs:
1895       if spec.name not in names_to_keys:
1896         raise errors.NotFoundError(
1897             None, None,
1898             message=("Attempting to load an object-based checkpoint using "
1899                      "variable names, but could not find %s in the "
1900                      "checkpoint.") % spec.name)
1901       spec.name = names_to_keys[spec.name]
1902 
1903   if cached_saver is None:
1904     return Saver(saveables)
1905   return cached_saver
