  86 def _require_cross_tower_context(distribution_strategy):
  87   """Verify in cross-tower context for `distribution_strategy`."""
  88   context = _get_per_thread_mode()
  89   if context.cross_tower_context is distribution_strategy: return
  90   # We have an error to report, figure out the right message.
  91   if context.distribution_strategy is not distribution_strategy:
  92     if (context.distribution_strategy is
  93         distribution_strategy_context._get_default_distribution_strategy()):  # pylint: disable=protected-access
  94       raise RuntimeError(
  95           'Need to be inside "with distribution_strategy.scope()" for %s' %
  96           (distribution_strategy,))
  97     else:
  98       raise RuntimeError(
  99           "Mixing different DistributionStrategy objects: %s is not %s" %
 100           (context.distribution_strategy, distribution_strategy))
 101   assert context.cross_tower_context is None
 102   raise RuntimeError("Method requires being in cross-tower context, use "
 103                      "get_tower_context().merge_call()")

 106 def require_tower_context(tower_ctx):
 107   """Verify in `tower_ctx` tower context."""
 108   context = _get_per_thread_mode()
 109   if context.tower_context is tower_ctx: return
 110   # We have an error to report, figure out the right message.
 111   if context.tower_context is None:
 112     raise RuntimeError("Need to be inside `call_for_each_tower()`")
 113   if context.distribution_strategy is tower_ctx.distribution_strategy:
 114     # Two different TowerContexts with the same DistributionStrategy.
 115     raise RuntimeError("Mismatching tower context.")
 116   raise RuntimeError(
 117       "Mismatching DistributionStrategy objects: %s is not %s." %
 118       (context.distribution_strategy, tower_ctx.distribution_strategy))

 194 class DistributionStrategy(object):
 195   """A list of devices with a state & compute distribution policy.
 196 
 197   See [tensorflow/contrib/distribute/README.md](
 198   https://www.tensorflow.org/code/tensorflow/contrib/distribute/README.md)
 199   for overview and examples.
 200 
 201   The intent is that you can write an algorithm in a stylized way and
 202   it will be usable with a variety of different `DistributionStrategy`
 203   implementations. Each descendant will implement a different strategy
 204   for distributing the algorithm across multiple devices/machines.
 205   Furthermore, these changes can be hidden inside the specific layers
 206   and other library classes that need special treatment to run in a
 207   distributed setting, so that most users' model definition code can
 208   run unchanged. The `DistributionStrategy` API works the same way
 209   with eager and graph execution.
 210 
 211   First let's introduce a few high-level concepts:
 212 
 213   * _Data parallelism_ is where we run multiple copies of the model
 214     on different slices of the input data. This is in contrast to
 215     _model parallelism_ where we divide up a single copy of a model
 216     across multiple devices.
 217     Note: we only support data parallelism for now, but
 218     hope to add support for model parallelism in the future.
 219   * A _tower_ is one copy of the model, running on one slice of the
 220     input data.
 221   * _Synchronous_, or more commonly _sync_, training is where the
 222     updates from each tower are aggregated together before updating
 223     the model variables. This is in contrast to _asynchronous_, or
 224     _async_ training, where each tower updates the model variables
 225     independently.
 226   * Furthermore you might run your computation on multiple devices
 227     on one machine (or "host"), or on multiple machines/hosts.
 228     If you are running on multiple machines, you might have a
 229     single master host that drives computation across all of them,
 230     or you might have multiple clients driving the computation
 231     asynchronously.
 232 
 233   To distribute an algorithm, we might use some of these ingredients:
 234 
 235   * Parameter servers: These are hosts that hold a single copy of
 236     parameters/variables. All towers that want to operate on a variable
 237     retrieve it at the beginning of a step and send an update to be
 238     applied at the end of the step. Can support either sync or async
 239     training.
 240   * Mirrored variables: These are variables that are copied to multiple
 241     devices, where we keep the copies in sync by applying the same
 242     updates to every copy. Normally would only be used with sync training.
 243   * Reductions and Allreduce: A _reduction_ is some method of
 244     aggregating multiple values into one value, like "sum" or
 245     "mean". If doing sync training, we will perform a reduction on the
 246     gradients to a parameter from all towers before applying the
 247     update. Allreduce is an algorithm for performing a reduction on
 248     values from multiple devices and making the result available on
 249     all of those devices.
 250   * In the future we will have support for TensorFlow's partitioned
 251     variables, where a single variable is split across multiple
 252     devices.
 253 
 254   We have then a few approaches we want to support:
 255 
 256   * Code written (as if) with no knowledge of class `DistributionStrategy`.
 257     This code should work as before, even if some of the layers, etc.
 258     used by that code are written to be distribution-aware. This is done
 259     by having a default `DistributionStrategy` that gives ordinary behavior,
 260     and by default being in a single tower context.
 261   * Ordinary model code that you want to run using a specific
 262     `DistributionStrategy`. This can be as simple as:
 263 
 264     ```
 265     with my_distribution.scope():
 266       iterator = my_distribution.distribute_dataset(
 267           dataset).make_one_shot_iterator()
 268       tower_train_ops = my_distribution.call_for_each_tower(
 269           tower_fn, iterator.get_next())
 270       train_op = tf.group(my_distribution.unwrap(tower_train_ops))
 271     ```
 272 
 273     This takes an ordinary `dataset` and `tower_fn` and runs it
 274     distributed using a particular `DistributionStrategy` in
 275     `my_distribution`. Any variables created in `tower_fn` are created
 276     using `my_distribution`'s policy, and library functions called by
 277     `tower_fn` can use the `get_tower_context()` API to get enhanced
 278     behavior in this case.
 279 
 280     You can also create an initializable iterator instead of a one-shot
 281     iterator. In that case, you will need to ensure that you initialize the
 282     iterator before calling get_next.
 283     ```
 284     iterator = my_distribution.distribute_dataset(
 285         dataset).make_initializable_iterator())
 286     session.run(iterator.initializer)
 287     ```
 288 
 289   * If you want to write a distributed algorithm, you may use any of
 290     the `DistributionStrategy` APIs inside a
 291     `with my_distribution.scope():` block of code.
 292 
 293   Lower-level concepts:
 294 
 295   * Wrapped values: In order to represent values parallel across devices
 296     (either towers or the devices associated with a particular value), we
 297     wrap them in a "PerDevice" or "Mirrored" object that contains a map
 298     from device to values. "PerDevice" is used when the value may be
 299     different across devices, and "Mirrored" when the value are the same.
 300   * Unwrapping and merging: Consider calling a function `fn` on
 301     multiple devices, like `call_for_each_tower(fn, w)` with an
 302     argument `w` that is a wrapped value. This means `w` will have a
 303     map taking tower device `d0` to `w0`, tower device `d1` to `w1`,
 304     etc. `call_for_each_tower()` unwraps `w` before calling `fn`, so
 305     it calls `fn(w0)` on `d0`, `fn(w1)` on `d1`, etc.  It then merges
 306     the return values from `fn()`, which can possibly result in
 307     wrapped values. For example, let's say `fn()` returns a tuple with
 308     three components: `(x, a, v0)` from tower 0, `(x, b, v1)` on tower 1,
 309     etc. If the first component is the same object `x` from every
 310     tower, then the first component of the merged result will also be
 311     `x`. If the second component is different (`a`, `b`, ...)  from
 312     each tower, then the merged value will have a wrapped map from
 313     tower device to the different values. If the third component is
 314     the members of a mirrored variable (`v` maps `d0` to `v0`, `d1` to
 315     `v1`, etc.), then the merged result will be that mirrored variable
 316     (`v`).
 317   * Tower context vs. Cross-tower context: _tower context_ is when we
 318     are in some function that is being called once for each tower.
 319     Otherwise we are in cross-tower context, which is useful for
 320     calling `DistributionStrategy` methods which operate across the
 321     towers (like `reduce()`). By default you start in a tower context
 322     (the default "single tower context") and then some methods can
 323     switch you back and forth, as described below.
 324   * Worker devices vs. parameter devices: Most tower computations will
 325     happen on worker devices. Since we don't yet support model
 326     parallelism, there will be one worker device per tower. When using
 327     parameter servers (see above), the set of devices holding
 328     variables may be different, otherwise the parameter devices might
 329     match the worker devices.
 330   * Non-slot devices are some subset of the parameter devices where we
 331     put all the non-slot variables. We need to ensure that all
 332     non-slot variables are allocated on the same device, or mirrored
 333     across the same set of devices. If you have some variable you want
 334     to colocate all the non-slot variables with, you can use
 335     `colocate_vars_with()` to get the remaining non-slot variables on
 336     the same device.  Otherwise you can use `non_slot_devices()` to
 337     pick a consistent set of devices to pass to both
 338     `colocate_vars_with()` and `update_non_slot()`.
 339 
 340   When using a `DistributionStrategy`, we have a new type dimension
 341   called _locality_ that says what values are compatible with which
 342   APIs:
 343 
 344   * T: different value for each tower (e.g. a PerDevice-wrapped value).
 345   * M: value is "mirrored" across towers, i.e. there are copies with the
 346     same value on each tower (e.g. a Mirrored-wrapped value).
 347   * V(`v`): value is "mirrored" across all the devices which have a
 348     copy of variable `v` (also a Mirrored-wrapped value, but over
 349     parameter devices instead of worker devices).
 350   * N: value is "mirrored" across all the "non-slot" devices
 351 
 352   Rules for methods with respect to locality and single-tower vs.
 353   cross-tower context:
 354 
 355   * `with d.scope()`: default single-tower context -> cross-tower context for
 356     `d`
 357   * `with d.colocate_vars_with(v)`: in tower/cross-tower context, variables
 358     will be created with locality V(`v`). That is, if we write
 359     `with d.colocate_vars_with(v1): v2 = tf.get_variable(...)`, then
 360     `v2` will have locality V(`v1`), i.e. locality V(`v2`) will equal
 361     V(`v1`).
 362   * `with d.colocate_vars_with(d.non_slot_devices(...))`: in
 363     tower/cross-tower context, variables will be created with locality N
 364   * `v = tf.get_variable(...)`: in tower/cross-tower context, creates
 365     a variable (which by definition will have locality V(`v`), though
 366     will match another locality if inside a `colocate_vars_with`
 367     scope).
 368   * `d.distribute_dataset(dataset).make_one_shot_iterator()`: in cross-tower
 369     context, produces an iterator with locality T
 370   * `d.broadcast(t)`: in cross-tower context, produces a value with locality M
 371   * `d.broadcast(t, v)`: in cross-tower context, produces a value with
 372     locality V(`v`)
 373   * `d.call_for_each_tower(fn, ...)`: in cross-tower context, runs
 374     `fn()` in a tower context (and so may call `get_tower_context()` and
 375     use its API, including `merge_call()` to get back to cross-tower
 376     context), once for each tower. May use values with locality T or
 377     M, and any variable.
 378   * `d.reduce(m, t, t)`: in cross-tower context, accepts t with locality T
 379     and produces a value with locality M.
 380   * `d.reduce(m, t, v)`: in cross-tower context, accepts t with
 381     locality T and produces a value with locality V(`v`).
 382   * `d.batch_reduce(m, [(t, v)]): see `d.reduce()`
 383   * `d.update(v, fn, ...)`: in cross-tower context, runs `fn()` once
 384     for each device `v` is copied to, all inputs should have locality
 385     V(`v`), output will have locality V(`v`) as well.
 386   * `d.update_non_slot(d.non_slot_devices(), fn)`: in cross-tower
 387     context, like `d.update()` except with locality N.
 388   * `d.read_var(v)`: Gets the (read-only) value of the variable `v` (on
 389     the device determined by the current device scope), aggregating
 390     across towers for tower-local variables. Frequently, this will be
 391     done automatically when using `v` in an expression or fetching it in
 392     a cross-tower context, but this function can be used to force that
 393     conversion happens at a particular point in time (for example, to
 394     add the result of the conversion to a graph collection).
 395 
 396   The standard pattern for updating variables is to:
 397 
 398   1. Wrap your input dataset in `d.distribute_dataset()` and create an iterator.
 399   2. Define each tower `d.call_for_each_tower()` up to the point of
 400      getting a list of gradient, variable pairs.
 401   3. Call `d.reduce(VariableAggregation.SUM, t, v)` or `d.batch_reduce()` to sum
 402      the gradients (with locality T) into values with locality V(`v`).
 403   4. Call `d.update(v)` for each variable to update its value.
 404 
 405   Steps 3 and 4 are done automatically by class `Optimizer` if you call
 406   its `apply_gradients` method in a tower context. Otherwise you can
 407   manually call its `_distributed_apply` method in a cross-tower context.
 408 
 409   Another thing you might want to do in the middle of your tower function
 410   is an all-reduce of some intermediate value, using `d.reduce()` or
 411   `d.batch_reduce()`. You simply provide the same tensor as the input and
 412   destination.
 413 
 414   Layers should expect to be called in a tower context, and can use
 415   the `get_tower_context()` function to get a `TowerContext` object. The
 416   `TowerContext` object has a `merge_call()` method for entering
 417   cross-tower context where you can use `reduce()` (or
 418   `batch_reduce()`) and then optionally `update()` to update state.
 419 
 420   You may use this API whether or not a `DistributionStrategy` is
 421   being used, since there is a default implementation of
 422   `TowerContext` and `DistributionStrategy`. Or you can use the
 423   `get_tower_context().is_single_tower` property to run different code
 424   in the distributed vs. single tower cases.
 425   """
 426

 441     """Returns a context manager selecting this DistributionStrategy as current.
 442 
 443     Inside a `with distribution_strategy.scope():` code block, this thread
 444     will use a variable creator set by `distribution_strategy`, and will
 445     enter its "cross-tower context".

 452       return _SameScopeAgainContext(self)

 455       _require_distribution_strategy_scope(self)
 456       kwargs["use_resource"] = True
 457       return self._create_variable(*args, **kwargs)
 458 
 459     def disable_partitioned_variables(getter, *args, **kwargs):
 460       if kwargs.pop("partitioner", None) is not None:
 461         tf_logging.log_first_n(
 462             tf_logging.WARN, "Partitioned variables are disabled when using "
 463             "DistributionStrategy.", 1)
 464       return getter(*args, **kwargs)

 467         self, variable_scope.variable_creator_scope(creator_with_resource_vars),

 469             variable_scope.get_variable_scope(),

 492   def colocate_vars_with(self, colocate_with_variable):
 493     """Scope that controls which devices variables will be created on.
 494 
 495     No operations should be added to the graph inside this scope, it
 496     should only be used when creating variables (some implementations
 497     work by changing variable creation, others work by using a
 498     tf.colocate_with() scope).
 499 
 500     This may only be used inside `self.scope()`.
 501 
 502     Example usage:
 503 
 504     ```
 505     with distribution_strategy.scope():
 506       var1 = tf.get_variable(...)
 507       with distribution_strategy.colocate_vars_with(v1):
 508         # var2 and var3 will be created on the same device(s) as var1
 509         var2 = tf.get_variable(...)
 510         var3 = tf.get_variable(...)
 511 
 512       def fn(v1, v2, v3):
 513         # operates on v1 from var1, v2 from var2, and v3 from var3
 514 
 515       # `fn` runs on every device `v1` is on, `v2` and `v3` will be there too.
 516       distribution_strategy.update(v1, fn, v2, v3)
 517     ```
 518 
 519     Args:
 520       colocate_with_variable: A created in `self.scope()`. Variables created
 521         while in the returned context manager will be on the same set of
 522         devices as `colocate_with_variable`.
 523

 674   def call_for_each_tower(self, fn, *args, **kwargs):
 675     """Run `fn` once per tower.
 676 
 677     `fn` may call `tf.get_tower_context()` to access methods such as
 678     `tower_id()` and `merge_call()`.
 679 
 680     `merge_call()` is used to communicate between the towers and
 681     re-enter the cross-tower context. All towers pause their execution
 682     having encountered a `merge_call()` call. After that the
 683     `merge_fn`-function is executed. Its results are then unwrapped and
 684     given back to each tower call. After that execution resumes until
 685     `fn` is complete or encounters another `merge_call()`.  Example:
 686 
 687     ```python
 688     # Called once in "cross-tower" context.
 689     def merge_fn(distribution, three_plus_tower_id):
 690       # sum the values across towers
 691       return sum(distribution.unwrap(three_plus_tower_id))
 692 
 693     # Called once per tower in `distribution`, in a "tower" context.
 694     def fn(three):
 695       tower_ctx = tf.get_tower_context()
 696       v = three + tower_ctx.tower_id
 697       # Computes the sum of the `v` values across all towers.
 698       s = tower_ctx.merge_call(merge_fn, v)
 699       return s + v
 700 
 701     with distribution.scope():
 702       # in "cross-tower" context
 703       ...
 704       merged_results = distribution.call_for_each_tower(fn, 3)
 705       # merged_results has the values from every tower execution of `fn`.
 706       print(distribution.unwrap(merged_results))  # Prints a list
 707     ```
 708 
 709     Args:
 710       fn: function to run (will be run once per tower).
 711       *args: positional arguments for `fn`
 712       **kwargs: keyword arguments for `fn`.
 713           `"run_concurrently"`: Boolean indicating whether executions of `fn`
 714              can be run concurrently (under eager execution only), defaults to
 715              `True`.
 716 
 717     Returns:
 718       Merged return value of `fn` across all towers.
 719     """
 720     _require_cross_tower_context(self)
 721     return self._call_for_each_tower(fn, *args, **kwargs)

 784   def update(self, var, fn, *args, **kwargs):
 785     """Run `fn` to update `var` using inputs mirrored to the same devices.
 786

 788     logic like:
 789 
 790     ```
 791     results = {}
 792     for device, v in var:
 793       with tf.device(device):

 810         specified, the return value will be unwrapped.

 818       for ensuring all elements are executed.

 990 # A note about the difference between the context managers
 991 # `TowerContext` (defined here) and `_CurrentDistributionContext`
 992 # (defined above) used by `DistributionStrategy.scope()`:
 993 #
 994 # * a TowerContext is only present during a `call_for_each_tower()`
 995 #   call (except during a `merge_run` call) and in such a scope it
 996 #   will be returned by calls to `get_tower_context()`.  Implementers of new
 997 #   DistributionStrategy descendants will frequently also need to
 998 #   define a descendant of TowerContext, and are responsible for
 999 #   entering and exiting this context.
1000 #
1001 # * DistributionStrategy.scope() sets up a variable_creator scope that
1002 #   changes variable creation calls (e.g. to make mirrored
1003 #   variables). This is intended as an outer scope that users enter once
1004 #   around their model creation and graph definition. There is no
1005 #   anticipated need to define descendants of _CurrentDistributionContext.
1006 #   It sets the current DistributionStrategy for purposes of
1007 #   `get_distribution_strategy()` and `has_distribution_strategy()`
1008 #   and switches the thread mode to a "cross-tower context".
1009 class TowerContext(object):
1010   """DistributionStrategy API inside a `call_for_each_tower()` call."""

1015         self)

1024   def merge_call(self, merge_fn, *args, **kwargs):
1025     """Merge args across towers and run `merge_fn` in a cross-tower context.
1026 
1027     This allows communication and coordination when there are multiple calls
1028     to a model function triggered by a call to
1029     `distribution.call_for_each_tower(model_fn, ...)`.
1030 
1031     See `MirroredDistribution.call_for_each_tower()` for an explanation.
1032 
1033     Otherwise, this is equivalent to:
1034 
1035     ```
1036     distribution = get_distribution_strategy()
1037     with cross-tower-context(distribution):
1038       return merge_fn(distribution, *args, **kwargs)
1039     ```
1040 
1041     Args:
1042       merge_fn: function that joins arguments from threads that are given as
1043         PerDevice. It accepts `DistributionStrategy` object as the first
1044         argument.
1045       *args: positional per-thread arguments for `merge_fn`
1046       **kwargs: keyword per-thread arguments for `merge_fn`.
1047 
1048     Returns:
1049       The return value of `merge_fn`, except for `PerDevice` values which are
1050       unpacked.
1051     """
1052     require_tower_context(self)
1053     return self._merge_call(merge_fn, *args, **kwargs)

1059             self._distribution_strategy))
1060     try:
1061       return merge_fn(self._distribution_strategy, *args, **kwargs)
1062     finally:
1063       _pop_per_thread_mode()
