  30 @tf_export("train.ProximalGradientDescentOptimizer")
  31 class ProximalGradientDescentOptimizer(optimizer.Optimizer):
  32   # pylint: disable=line-too-long
  33   """Optimizer that implements the proximal gradient descent algorithm.
  34 
  35   See this [paper](http://papers.nips.cc/paper/3793-efficient-learning-using-forward-backward-splitting.pdf).
  36   """
  37 
  38   def __init__(self, learning_rate, l1_regularization_strength=0.0,
  39                l2_regularization_strength=0.0, use_locking=False,
  40                name="ProximalGradientDescent"):
  41     """Construct a new proximal gradient descent optimizer.
  42 
  43     Args:
  44       learning_rate: A Tensor or a floating point value.  The learning
  45         rate to use.
  46       l1_regularization_strength: A float value, must be greater than or
  47         equal to zero.
  48       l2_regularization_strength: A float value, must be greater than or
  49         equal to zero.
  50       use_locking: If True use locks for update operations.
  51       name: Optional name prefix for the operations created when applying
  52         gradients. Defaults to "GradientDescent".
  53     """
  54     super(ProximalGradientDescentOptimizer, self).__init__(use_locking, name)
  55     self._learning_rate = learning_rate
  56     self._l1_regularization_strength = l1_regularization_strength
  57     self._l2_regularization_strength = l2_regularization_strength
  58     self._l1_regularization_strength_tensor = None
  59     self._l2_regularization_strength_tensor = None
  60 
  61   def _apply_dense(self, grad, var):
  62     return training_ops.apply_proximal_gradient_descent(
  63         var,
  64         self._learning_rate_tensor,
  65         self._l1_regularization_strength_tensor,
  66         self._l2_regularization_strength_tensor,
  67         grad,
  68         use_locking=self._use_locking).op
  69 
  70   def _resource_apply_dense(self, grad, var):
  71     return training_ops.resource_apply_proximal_gradient_descent(
  72         var.handle,
  73         self._learning_rate_tensor,
  74         self._l1_regularization_strength_tensor,
  75         self._l2_regularization_strength_tensor,
  76         grad,
  77         use_locking=self._use_locking)
  78 
  79   def _apply_sparse(self, grad, var):
  80     return training_ops.sparse_apply_proximal_gradient_descent(
  81         var,
  82         self._learning_rate_tensor,
  83         self._l1_regularization_strength_tensor,
  84         self._l2_regularization_strength_tensor,
  85         grad.values,
  86         grad.indices,
  87         use_locking=self._use_locking).op
  88 
  89   def _resource_apply_sparse(self, grad, var, indices):
  90     return training_ops.resource_sparse_apply_proximal_gradient_descent(
  91         var.handle,
  92         math_ops.cast(self._learning_rate_tensor, grad.dtype),
  93         math_ops.cast(self._l1_regularization_strength_tensor, grad.dtype),
  94         math_ops.cast(self._l2_regularization_strength_tensor, grad.dtype),
  95         grad,
  96         indices,
  97         use_locking=self._use_locking)
  98 
  99   def _prepare(self):
 100     self._learning_rate_tensor = ops.convert_to_tensor(self._learning_rate,
 101                                                        name="learning_rate")
 102     self._l1_regularization_strength_tensor = ops.convert_to_tensor(
 103         self._l1_regularization_strength, name="l1_regularization_strength")
 104     self._l2_regularization_strength_tensor = ops.convert_to_tensor(
 105         self._l2_regularization_strength, name="l2_regularization_strength")
