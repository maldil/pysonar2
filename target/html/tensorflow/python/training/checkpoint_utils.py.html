  42 @tf_export("train.load_checkpoint")
  43 def load_checkpoint(ckpt_dir_or_file):
  44   """Returns `CheckpointReader` for checkpoint found in `ckpt_dir_or_file`.
  45 
  46   If `ckpt_dir_or_file` resolves to a directory with multiple checkpoints,
  47   reader for the latest checkpoint is returned.
  48 
  49   Args:
  50     ckpt_dir_or_file: Directory with checkpoints file or path to checkpoint
  51       file.
  52 
  53   Returns:
  54     `CheckpointReader` object.
  55 
  56   Raises:
  57     ValueError: If `ckpt_dir_or_file` resolves to a directory with no
  58       checkpoints.
  59   """
  60   filename = _get_checkpoint_filename(ckpt_dir_or_file)
  61   if filename is None:
  62     raise ValueError("Couldn't find 'checkpoint' file or checkpoints in "
  63                      "given directory %s" % ckpt_dir_or_file)
  64   return pywrap_tensorflow.NewCheckpointReader(filename)

  67 @tf_export("train.load_variable")
  68 def load_variable(ckpt_dir_or_file, name):
  69   """Returns the tensor value of the given variable in the checkpoint.
  70 
  71   Args:
  72     ckpt_dir_or_file: Directory with checkpoints file or path to checkpoint.
  73     name: Name of the variable to return.
  74 
  75   Returns:
  76     A numpy `ndarray` with a copy of the value of this variable.
  77   """
  78   # TODO(b/29227106): Fix this in the right place and remove this.
  79   if name.endswith(":0"):
  80     name = name[:-2]
  81   reader = load_checkpoint(ckpt_dir_or_file)
  82   return reader.get_tensor(name)

  85 @tf_export("train.list_variables")
  86 def list_variables(ckpt_dir_or_file):
  87   """Returns list of all variables in the checkpoint.
  88 
  89   Args:
  90     ckpt_dir_or_file: Directory with checkpoints file or path to checkpoint.
  91 
  92   Returns:
  93     List of tuples `(name, shape)`.
  94   """
  95   reader = load_checkpoint(ckpt_dir_or_file)
  96   variable_map = reader.get_variable_to_shape_map()
  97   names = sorted(variable_map.keys())
  98   result = []
  99   for name in names:
 100     result.append((name, variable_map[name]))
 101   return result

 104 @tf_export("train.init_from_checkpoint")
 105 def init_from_checkpoint(ckpt_dir_or_file, assignment_map):
 106   """Initializes current variables with tensors loaded from given checkpoint.
 107 
 108   Note: This overrides default initialization ops of specified variables and
 109   redefines dtype.
 110 
 111   Assignment map supports following syntax:
 112 
 113   * `'checkpoint_scope_name/': 'scope_name/'` - will load all variables in
 114     current `scope_name` from `checkpoint_scope_name` with matching tensor
 115     names.
 116   * `'checkpoint_scope_name/some_other_variable': 'scope_name/variable_name'` -
 117     will initialize `scope_name/variable_name` variable
 118     from `checkpoint_scope_name/some_other_variable`.
 119   * `'scope_variable_name': variable` - will initialize given `tf.Variable`
 120     object with tensor 'scope_variable_name' from the checkpoint.
 121   * `'scope_variable_name': list(variable)` - will initialize list of
 122     partitioned variables with tensor 'scope_variable_name' from the checkpoint.
 123   * `'/': 'scope_name/'` - will load all variables in current `scope_name` from
 124     checkpoint's root (e.g. no scope).
 125 
 126   Supports loading into partitioned variables, which are represented as
 127   `'<variable>/part_<part #>'`.
 128 
 129   Example:
 130 
 131   ```python
 132 
 133   # Say, '/tmp/model.ckpt' has the following tensors:
 134   #  -- name='old_scope_1/var1', shape=[20, 2]
 135   #  -- name='old_scope_1/var2', shape=[50, 4]
 136   #  -- name='old_scope_2/var3', shape=[100, 100]
 137 
 138   # Create new model's variables
 139   with tf.variable_scope('new_scope_1'):
 140     var1 = tf.get_variable('var1', shape=[20, 2],
 141                            initializer=tf.zeros_initializer())
 142   with tf.variable_scope('new_scope_2'):
 143     var2 = tf.get_variable('var2', shape=[50, 4],
 144                            initializer=tf.zeros_initializer())
 145     # Partition into 5 variables along the first axis.
 146     var3 = tf.get_variable(name='var3', shape=[100, 100],
 147                            initializer=tf.zeros_initializer(),
 148                            partitioner=lambda shape, dtype: [5, 1])
 149 
 150   # Initialize all variables in `new_scope_1` from `old_scope_1`.
 151   init_from_checkpoint('/tmp/model.ckpt', {'old_scope_1/': 'new_scope_1'})
 152 
 153   # Use names to specify which variables to initialize from checkpoint.
 154   init_from_checkpoint('/tmp/model.ckpt',
 155                        {'old_scope_1/var1': 'new_scope_1/var1',
 156                         'old_scope_1/var2': 'new_scope_2/var2'})
 157 
 158   # Or use tf.Variable objects to identify what to initialize.
 159   init_from_checkpoint('/tmp/model.ckpt',
 160                        {'old_scope_1/var1': var1,
 161                         'old_scope_1/var2': var2})
 162 
 163   # Initialize partitioned variables using variable's name
 164   init_from_checkpoint('/tmp/model.ckpt',
 165                        {'old_scope_2/var3': 'new_scope_2/var3'})
 166 
 167   # Or specify the list of tf.Variable objects.
 168   init_from_checkpoint('/tmp/model.ckpt',
 169                        {'old_scope_2/var3': var3._get_variable_list()})
 170 
 171   ```
 172 
 173   Args:
 174     ckpt_dir_or_file: Directory with checkpoints file or path to checkpoint.
 175     assignment_map: Dict, where keys are names of the variables in the
 176       checkpoint and values are current variables or names of current variables
 177       (in default graph).
 178 
 179   Raises:
 180     tf.errors.OpError: If missing checkpoints or tensors in checkpoints.
 181     ValueError: If missing variables in current graph.
 182   """
 183   if distribution_strategy_context.get_cross_tower_context():
 184     _init_from_checkpoint(None, ckpt_dir_or_file, assignment_map)
 185   else:
 186     distribution_strategy_context.get_tower_context().merge_call(
 187         _init_from_checkpoint, ckpt_dir_or_file, assignment_map)
 188

 190 def _init_from_checkpoint(_, ckpt_dir_or_file, assignment_map):
 191   """See `init_from_checkpoint` for documentation."""
 192 
 193   ckpt_file = _get_checkpoint_filename(ckpt_dir_or_file)
 194   reader = load_checkpoint(ckpt_dir_or_file)
 195   variable_map = reader.get_variable_to_shape_map()
 196   for tensor_name_in_ckpt, current_var_or_name in sorted(
 197       six.iteritems(assignment_map)):
 198     var = None
 199     # Check if this is Variable object or list of Variable objects (in case of
 200     # partitioned variables).
 201     if _is_variable(current_var_or_name) or (
 202         isinstance(current_var_or_name, list)
 203         and all(_is_variable(v) for v in current_var_or_name)):
 204       var = current_var_or_name
 205     else:
 206       store_vars = vs._get_default_variable_store()._vars  # pylint:disable=protected-access
 207       # Check if this variable is in var_store.
 208       var = store_vars.get(current_var_or_name, None)
 209       # Also check if variable is partitioned as list.
 210       if var is None:
 211         var = _collect_partitioned_variable(current_var_or_name, store_vars)
 212     if var is not None:
 213       # If 1 to 1 mapping was provided, find variable in the checkpoint.
 214       if tensor_name_in_ckpt not in variable_map:
 215         raise ValueError("Tensor %s is not found in %s checkpoint %s" % (
 216             tensor_name_in_ckpt, ckpt_dir_or_file, variable_map
 217         ))
 218       if _is_variable(var):
 219         # Additional at-call-time checks.
 220         if not var.get_shape().is_compatible_with(
 221             variable_map[tensor_name_in_ckpt]):
 222           raise ValueError(
 223               "Shape of variable %s (%s) doesn't match with shape of "
 224               "tensor %s (%s) from checkpoint reader." % (
 225                   var.name, str(var.get_shape()),
 226                   tensor_name_in_ckpt, str(variable_map[tensor_name_in_ckpt])
 227               ))
 228         var_name = var.name
 229       else:
 230         var_name = ",".join([v.name for v in var])
 231       _set_variable_or_list_initializer(var, ckpt_file, tensor_name_in_ckpt)
 232       logging.debug("Initialize variable %s from checkpoint %s with %s",
 233                     var_name, ckpt_dir_or_file, tensor_name_in_ckpt)
 234     else:
 235       scopes = ""
 236       # TODO(vihanjain): Support list of 'current_var_or_name' here.
 237       if "/" in current_var_or_name:
 238         scopes = current_var_or_name[:current_var_or_name.rindex("/")]
 239       if not tensor_name_in_ckpt.endswith("/"):
 240         raise ValueError(
 241             "Assignment map with scope only name {} should map to scope only "
 242             "{}. Should be 'scope/': 'other_scope/'.".format(
 243                 scopes, tensor_name_in_ckpt))
 244       # If scope to scope mapping was provided, find all variables in the scope
 245       # and create variable to variable mapping.
 246       scope_variables = set()
 247       for var_name in store_vars:
 248         if not scopes or var_name.startswith(scopes + "/"):
 249           # Consume /part_ if partitioned variable.
 250           if "/part_" in var_name:
 251             var_name = var_name[:var_name.index("/part_")]
 252           scope_variables.add(var_name)
 253       for var_name in sorted(scope_variables):
 254         # Lookup name with specified prefix and suffix from current variable.
 255         # If tensor_name given is '/' (root), don't use it for full name.
 256         full_tensor_name = var_name[len(scopes):]
 257         if current_var_or_name != "/":
 258           full_tensor_name = full_tensor_name[1:]
 259         if tensor_name_in_ckpt != "/":
 260           full_tensor_name = tensor_name_in_ckpt + full_tensor_name
 261         # Remove trailing '/', if any, in the full_tensor_name
 262         if full_tensor_name.endswith("/"):
 263           full_tensor_name = full_tensor_name[:-1]
 264         if full_tensor_name not in variable_map:
 265           raise ValueError(
 266               "Tensor %s (%s in %s) is not found in %s checkpoint" % (
 267                   full_tensor_name, var_name[len(scopes) + 1:],
 268                   tensor_name_in_ckpt, ckpt_dir_or_file
 269               ))
 270         var = store_vars.get(var_name, None)
 271         if var is None:
 272           var = _collect_partitioned_variable(var_name, store_vars)
 273         _set_variable_or_list_initializer(var, ckpt_file, full_tensor_name)
 274         logging.debug("Initialize variable %s from checkpoint %s with %s",
 275                       var_name, ckpt_dir_or_file, full_tensor_name)

 278 def _get_checkpoint_filename(ckpt_dir_or_file):
 279   """Returns checkpoint filename given directory or specific checkpoint file."""
 280   if gfile.IsDirectory(ckpt_dir_or_file):
 281     return checkpoint_management.latest_checkpoint(ckpt_dir_or_file)
 282   return ckpt_dir_or_file

 285 def _set_checkpoint_initializer(variable,
 286                                 ckpt_file,
 287                                 tensor_name,
 288                                 slice_spec,
 289                                 name="checkpoint_initializer"):
 290   """Overrides given variable's initialization op.
 291 
 292   Sets variable initializer to assign op that initializes variable from tensor's
 293   value in the checkpoint.
 294 
 295   Args:
 296     variable: `tf.Variable` object.
 297     ckpt_file: string, full path of the checkpoint.
 298     tensor_name: Name of the tensor to load from the checkpoint.
 299     slice_spec: Slice specification for loading partitioned tensors.
 300     name: Name of the operation.
 301   """
 302   base_type = variable.dtype.base_dtype
 303   # Do not colocate with variable since RestoreV2 op only runs on CPU and
 304   # colocation will force variable (and other ops that colocate with variable)
 305   # to be on CPU as well. It is okay to place the variable's initializer op on
 306   # CPU since it will only be run once at the start.
 307   with ops.device(variable.device), ops.device("/cpu:0"):
 308     restore_op = io_ops.restore_v2(
 309         ckpt_file, [tensor_name], [slice_spec], [base_type], name=name)[0]
 310 
 311     names_to_saveables = saver.BaseSaverBuilder.OpListToDict([variable])
 312     saveable_objects = []
 313     for name, op in names_to_saveables.items():
 314       for s in saver.BaseSaverBuilder.SaveableObjectsForOp(op, name):
 315         saveable_objects.append(s)
 316 
 317     assert len(saveable_objects) == 1  # Should be only one variable.
 318     init_op = saveable_objects[0].restore([restore_op], restored_shapes=None)
 319 
 320     # pylint:disable=protected-access
 321     variable._initializer_op = init_op
 322     restore_op.set_shape(variable.shape)
 323     variable._initial_value = restore_op
 324     # pylint:enable=protected-access

 327 def _set_variable_or_list_initializer(variable_or_list, ckpt_file,
 328                                       tensor_name):
 329   """Overrides initialization op of given variable or list of variables.
 330 
 331   Calls `_set_checkpoint_initializer` for each variable in the given list of
 332   variables.
 333 
 334   Args:
 335     variable_or_list: `tf.Variable` object or a list of `tf.Variable` objects.
 336     ckpt_file: string, full path of the checkpoint.
 337     tensor_name: Name of the tensor to load from the checkpoint.
 338 
 339   Raises:
 340     ValueError: if all objects in `variable_or_list` are not partitions of the
 341       same large variable.
 342   """
 343   if isinstance(variable_or_list, (list, tuple)):
 344     # A set of slices.
 345     slice_name = None
 346     for v in variable_or_list:
 347       slice_info = v._save_slice_info  # pylint:disable=protected-access
 348       if slice_name is None:
 349         slice_name = slice_info.full_name
 350       elif slice_name != slice_info.full_name:
 351         raise ValueError("Slices must all be from the same tensor: %s != %s" %
 352                          (slice_name, slice_info.full_name))
 353       _set_checkpoint_initializer(v, ckpt_file, tensor_name, slice_info.spec)
 354   else:
 355     _set_checkpoint_initializer(variable_or_list, ckpt_file, tensor_name, "")

 358 def _is_variable(x):
 359   return (isinstance(x, variables.Variable) or
 360           resource_variable_ops.is_resource_variable(x))

 363 def _collect_partitioned_variable(name, all_vars):
 364   """Returns list of `tf.Variable` that comprise the partitioned variable."""
 365   if name + "/part_0" in all_vars:
 366     var = []
 367     i = 0
 368     while name + "/part_%d" % i in all_vars:
 369       var.append(all_vars[name + "/part_%d" % i])
 370       i += 1
 371     return var
 372   return None
