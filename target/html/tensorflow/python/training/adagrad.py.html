  22 from tensorflow.python.ops import array_ops
  23 from tensorflow.python.ops import gen_array_ops
  24 from tensorflow.python.ops import init_ops
  25 from tensorflow.python.ops import math_ops
  26 from tensorflow.python.training import optimizer
  27 from tensorflow.python.training import training_ops
  28 from tensorflow.python.util.tf_export import tf_export

  31 @tf_export("train.AdagradOptimizer")
  32 class AdagradOptimizer(optimizer.Optimizer):
  33   """Optimizer that implements the Adagrad algorithm.
  34 
  35   See this [paper](http://www.jmlr.org/papers/volume12/duchi11a/duchi11a.pdf)
  36   or this
  37   [intro](https://ppasupat.github.io/a9online/uploads/proximal_notes.pdf).
  38   """
  39 
  40   def __init__(self, learning_rate, initial_accumulator_value=0.1,
  41                use_locking=False, name="Adagrad"):
  42     """Construct a new Adagrad optimizer.
  43 
  44     Args:
  45       learning_rate: A `Tensor` or a floating point value.  The learning rate.
  46       initial_accumulator_value: A floating point value.
  47         Starting value for the accumulators, must be positive.
  48       use_locking: If `True` use locks for update operations.
  49       name: Optional name prefix for the operations created when applying
  50         gradients.  Defaults to "Adagrad".
  51 
  52     Raises:
  53       ValueError: If the `initial_accumulator_value` is invalid.
  54 
  55     @compatibility(eager)
  56     When eager execution is enabled, `learning_rate` can be a callable that
  57     takes no arguments and returns the actual value to use. This can be useful
  58     for changing these values across different invocations of optimizer
  59     functions.
  60     @end_compatibility
  61     """
  62     if initial_accumulator_value <= 0.0:
  63       raise ValueError("initial_accumulator_value must be positive: %s" %
  64                        initial_accumulator_value)
  65     super(AdagradOptimizer, self).__init__(use_locking, name)
  66     self._learning_rate = learning_rate
  67     self._initial_accumulator_value = initial_accumulator_value
  68     # Created in Initialize.
  69     self._learning_rate_tensor = None
  70 
  71   def _create_slots(self, var_list):
  72     for v in var_list:
  73       dtype = v.dtype.base_dtype
  74       if v.get_shape().is_fully_defined():
  75         init = init_ops.constant_initializer(self._initial_accumulator_value,
  76                                              dtype=dtype)
  77       else:
  78         init = self._init_constant_op(v, dtype)
  79       self._get_or_make_slot_with_initializer(v, init, v.get_shape(), dtype,
  80                                               "accumulator", self._name)
  81 
  82   def _init_constant_op(self, v, dtype):
  83     def init():
  84       # Use a Tensor instead of initializer if variable does not have
  85       # static shape.
  86       init_constant = gen_array_ops.fill(array_ops.shape(v),
  87                                          self._initial_accumulator_value)
  88       return math_ops.cast(init_constant, dtype)
  89     return init
  90 
  91   def _prepare(self):
  92     learning_rate = self._call_if_callable(self._learning_rate)
  93     self._learning_rate_tensor = ops.convert_to_tensor(
  94         learning_rate, name="learning_rate")
  95 
  96   def _apply_dense(self, grad, var):
  97     acc = self.get_slot(var, "accumulator")
  98     return training_ops.apply_adagrad(
  99         var,
 100         acc,
 101         math_ops.cast(self._learning_rate_tensor, var.dtype.base_dtype),
 102         grad,
 103         use_locking=self._use_locking)
 104 
 105   def _resource_apply_dense(self, grad, var):
 106     acc = self.get_slot(var, "accumulator")
 107     return training_ops.resource_apply_adagrad(
 108         var.handle,
 109         acc.handle,
 110         math_ops.cast(self._learning_rate_tensor, grad.dtype.base_dtype),
 111         grad,
 112         use_locking=self._use_locking)
 113 
 114   def _apply_sparse(self, grad, var):
 115     acc = self.get_slot(var, "accumulator")
 116     return training_ops.sparse_apply_adagrad(
 117         var,
 118         acc,
 119         math_ops.cast(self._learning_rate_tensor, var.dtype.base_dtype),
 120         grad.values,
 121         grad.indices,
 122         use_locking=self._use_locking)
 123 
 124   def _resource_apply_sparse(self, grad, var, indices):
 125     acc = self.get_slot(var, "accumulator")
 126     return training_ops.resource_sparse_apply_adagrad(
 127         var.handle,
 128         acc.handle,
 129         math_ops.cast(self._learning_rate_tensor, grad.dtype),
 130         grad,
 131         indices,
 132         use_locking=self._use_locking)
