  37 # Please note that the gradients from replicas are averaged instead of summed
  38 # (as in the old sync_replicas_optimizer) so you need to increase the learning
  39 # rate according to the number of replicas. This change is introduced to be
  40 # consistent with how gradients are aggregated (averaged) within a batch in a
  41 # replica.
  42 @tf_export("train.SyncReplicasOptimizer")
  43 class SyncReplicasOptimizer(optimizer.Optimizer):
  44   """Class to synchronize, aggregate gradients and pass them to the optimizer.
  45 
  46   In a typical asynchronous training environment, it's common to have some
  47   stale gradients. For example, with a N-replica asynchronous training,
  48   gradients will be applied to the variables N times independently. Depending
  49   on each replica's training speed, some gradients might be calculated from
  50   copies of the variable from several steps back (N-1 steps on average). This
  51   optimizer avoids stale gradients by collecting gradients from all replicas,
  52   averaging them, then applying them to the variables in one shot, after
  53   which replicas can fetch the new variables and continue.
  54 
  55   The following accumulators/queue are created:
  56 
  57   * N `gradient accumulators`, one per variable to train. Gradients are pushed
  58     to them and the chief worker will wait until enough gradients are collected
  59     and then average them before applying to variables. The accumulator will
  60     drop all stale gradients (more details in the accumulator op).
  61   * 1 `token` queue where the optimizer pushes the new global_step value after
  62     all variables are updated.
  63 
  64   The following local variable is created:
  65   * `sync_rep_local_step`, one per replica. Compared against the global_step in
  66     each accumulator to check for staleness of the gradients.
  67 
  68   The optimizer adds nodes to the graph to collect gradients and pause the
  69   trainers until variables are updated.
  70   For the Parameter Server job:
  71 
  72   1. An accumulator is created for each variable, and each replica pushes the
  73      gradients into the accumulators instead of directly applying them to the
  74      variables.
  75   2. Each accumulator averages once enough gradients (replicas_to_aggregate)
  76      have been accumulated.
  77   3. Apply the averaged gradients to the variables.
  78   4. Only after all variables have been updated, increment the global step.
  79   5. Only after step 4, pushes `global_step` in the `token_queue`, once for
  80      each worker replica. The workers can now fetch the global step, use it to
  81      update its local_step variable and start the next batch.
  82 
  83   For the replicas:
  84 
  85   1. Start a step: fetch variables and compute gradients.
  86   2. Once the gradients have been computed, push them into gradient
  87      accumulators. Each accumulator will check the staleness and drop the stale.
  88   3. After pushing all the gradients, dequeue an updated value of global_step
  89      from the token queue and record that step to its local_step variable. Note
  90      that this is effectively a barrier.
  91   4. Start the next batch.
  92 
  93   ### Usage
  94 
  95   ```python
  96   # Create any optimizer to update the variables, say a simple SGD:
  97   opt = GradientDescentOptimizer(learning_rate=0.1)
  98 
  99   # Wrap the optimizer with sync_replicas_optimizer with 50 replicas: at each
 100   # step the optimizer collects 50 gradients before applying to variables.
 101   # Note that if you want to have 2 backup replicas, you can change
 102   # total_num_replicas=52 and make sure this number matches how many physical
 103   # replicas you started in your job.
 104   opt = tf.train.SyncReplicasOptimizer(opt, replicas_to_aggregate=50,
 105                                  total_num_replicas=50)
 106 
 107   # Some models have startup_delays to help stabilize the model but when using
 108   # sync_replicas training, set it to 0.
 109 
 110   # Now you can call `minimize()` or `compute_gradients()` and
 111   # `apply_gradients()` normally
 112   training_op = opt.minimize(total_loss, global_step=self.global_step)
 113 
 114 
 115   # You can create the hook which handles initialization and queues.
 116   sync_replicas_hook = opt.make_session_run_hook(is_chief)
 117   ```
 118 
 119   In the training program, every worker will run the train_op as if not
 120   synchronized.
 121 
 122   ```python
 123   with training.MonitoredTrainingSession(
 124       master=workers[worker_id].target, is_chief=is_chief,
 125       hooks=[sync_replicas_hook]) as mon_sess:
 126     while not mon_sess.should_stop():
 127       mon_sess.run(training_op)
 128   ```
 129 
 130   To use SyncReplicasOptimizer with an `Estimator`, you need to send
 131   sync_replicas_hook while calling the fit.
 132   ```python
 133   my_estimator = DNNClassifier(..., optimizer=opt)
 134   my_estimator.fit(..., hooks=[sync_replicas_hook])
 135   ```
 136   """
 137 
 138   def __init__(self,
 139                opt,
 140                replicas_to_aggregate,
 141                total_num_replicas=None,
 142                variable_averages=None,
 143                variables_to_average=None,
 144                use_locking=False,
 145                name="sync_replicas"):
 146     """Construct a sync_replicas optimizer.
 147 
 148     Args:
 149       opt: The actual optimizer that will be used to compute and apply the
 150         gradients. Must be one of the Optimizer classes.
 151       replicas_to_aggregate: number of replicas to aggregate for each variable
 152         update.
 153       total_num_replicas: Total number of tasks/workers/replicas, could be
 154         different from replicas_to_aggregate.
 155         If total_num_replicas > replicas_to_aggregate: it is backup_replicas +
 156         replicas_to_aggregate.
 157         If total_num_replicas < replicas_to_aggregate: Replicas compute
 158         multiple batches per update to variables.
 159       variable_averages: Optional `ExponentialMovingAverage` object, used to
 160         maintain moving averages for the variables passed in
 161         `variables_to_average`.
 162       variables_to_average: a list of variables that need to be averaged. Only
 163         needed if variable_averages is passed in.
 164       use_locking: If True use locks for update operation.
 165       name: string. Optional name of the returned operation.
 166     """
 167     if total_num_replicas is None:
 168       total_num_replicas = replicas_to_aggregate
 169 
 170     super(SyncReplicasOptimizer, self).__init__(use_locking, name)
 171     logging.info(
 172         "SyncReplicasV2: replicas_to_aggregate=%s; total_num_replicas=%s",
 173         replicas_to_aggregate, total_num_replicas)
 174     self._opt = opt
 175     self._replicas_to_aggregate = replicas_to_aggregate
 176     self._gradients_applied = False
 177     self._variable_averages = variable_averages
 178     self._variables_to_average = variables_to_average
 179     self._total_num_replicas = total_num_replicas
 180     self._tokens_per_step = max(total_num_replicas, replicas_to_aggregate)
 181     self._global_step = None
 182     self._sync_token_queue = None
 183 
 184     # The synchronization op will be executed in a queue runner which should
 185     # only be executed by one of the replicas (usually the chief).
 186     self._chief_queue_runner = None
 187 
 188     # Remember which accumulator is on which device to set the initial step in
 189     # the accumulator to be global step. This list contains list of the
 190     # following format: (accumulator, device).
 191     self._accumulator_list = []
 192 
 193   def compute_gradients(self, *args, **kwargs):
 194     """Compute gradients of "loss" for the variables in "var_list".
 195 
 196     This simply wraps the compute_gradients() from the real optimizer. The
 197     gradients will be aggregated in the apply_gradients() so that user can
 198     modify the gradients like clipping with per replica global norm if needed.
 199     The global norm with aggregated gradients can be bad as one replica's huge
 200     gradients can hurt the gradients from other replicas.
 201 
 202     Args:
 203       *args: Arguments for compute_gradients().
 204       **kwargs: Keyword arguments for compute_gradients().
 205 
 206     Returns:
 207       A list of (gradient, variable) pairs.
 208     """
 209     return self._opt.compute_gradients(*args, **kwargs)
 210 
 211   def apply_gradients(self, grads_and_vars, global_step=None, name=None):
 212     """Apply gradients to variables.
 213 
 214     This contains most of the synchronization implementation and also wraps the
 215     apply_gradients() from the real optimizer.
 216 
 217     Args:
 218       grads_and_vars: List of (gradient, variable) pairs as returned by
 219         compute_gradients().
 220       global_step: Optional Variable to increment by one after the
 221         variables have been updated.
 222       name: Optional name for the returned operation.  Default to the
 223         name passed to the Optimizer constructor.
 224 
 225     Returns:
 226       train_op: The op to dequeue a token so the replicas can exit this batch
 227       and start the next one. This is executed by each replica.
 228 
 229     Raises:
 230       ValueError: If the grads_and_vars is empty.
 231       ValueError: If global step is not provided, the staleness cannot be
 232         checked.
 233     """
 234     if not grads_and_vars:
 235       raise ValueError("Must supply at least one variable")
 236 
 237     if global_step is None:
 238       raise ValueError("Global step is required to check staleness")
 239 
 240     self._global_step = global_step
 241     train_ops = []
 242     aggregated_grad = []
 243     var_list = []
 244 
 245     # local_anchor op will be placed on this worker task by default.
 246     local_anchor = control_flow_ops.no_op()
 247     # Colocating local_step variable prevents it being placed on the PS.
 248     with ops.colocate_with(local_anchor):
 249       self._local_step = variable_scope.variable(
 250           initial_value=0,
 251           trainable=False,
 252           collections=[ops.GraphKeys.LOCAL_VARIABLES],
 253           dtype=global_step.dtype.base_dtype,
 254           name="sync_rep_local_step")
 255 
 256     self.local_step_init_op = state_ops.assign(self._local_step, global_step)
 257     chief_init_ops = [self.local_step_init_op]
 258     self.ready_for_local_init_op = variables.report_uninitialized_variables(
 259         variables.global_variables())
 260 
 261     with ops.name_scope(None, self._name):
 262       for grad, var in grads_and_vars:
 263         var_list.append(var)
 264         with ops.device(var.device):
 265           # Dense gradients.
 266           if grad is None:
 267             aggregated_grad.append(None)  # pass-through.
 268             continue
 269           elif isinstance(grad, ops.Tensor):
 270             grad_accum = data_flow_ops.ConditionalAccumulator(
 271                 grad.dtype,
 272                 shape=var.get_shape(),
 273                 shared_name=var.name + "/grad_accum")
 274             train_ops.append(grad_accum.apply_grad(
 275                 grad, local_step=self._local_step))
 276             aggregated_grad.append(grad_accum.take_grad(
 277                 self._replicas_to_aggregate))
 278           else:
 279             if not isinstance(grad, ops.IndexedSlices):
 280               raise ValueError("Unknown grad type!")
 281             grad_accum = data_flow_ops.SparseConditionalAccumulator(
 282                 grad.dtype, shape=(), shared_name=var.name + "/grad_accum")
 283             train_ops.append(grad_accum.apply_indexed_slices_grad(
 284                 grad, local_step=self._local_step))
 285             aggregated_grad.append(grad_accum.take_indexed_slices_grad(
 286                 self._replicas_to_aggregate))
 287 
 288           self._accumulator_list.append((grad_accum, var.device))
 289 
 290       aggregated_grads_and_vars = zip(aggregated_grad, var_list)
 291 
 292       # sync_op will be assigned to the same device as the global step.
 293       with ops.device(global_step.device), ops.name_scope(""):
 294         update_op = self._opt.apply_gradients(aggregated_grads_and_vars,
 295                                               global_step)
 296 
 297       # Create token queue.
 298       with ops.device(global_step.device), ops.name_scope(""):
 299         sync_token_queue = (
 300             data_flow_ops.FIFOQueue(-1,
 301                                     global_step.dtype.base_dtype,
 302                                     shapes=(),
 303                                     name="sync_token_q",
 304                                     shared_name="sync_token_q"))
 305         self._sync_token_queue = sync_token_queue
 306 
 307         # dummy_queue is passed to the queue runner. Don't use the real queues
 308         # because the queue runner doesn't automatically reopen it once it
 309         # closed queues in PS devices.
 310         dummy_queue = (
 311             data_flow_ops.FIFOQueue(1,
 312                                     types_pb2.DT_INT32,
 313                                     shapes=(),
 314                                     name="dummy_queue",
 315                                     shared_name="dummy_queue"))
 316 
 317       with ops.device(global_step.device), ops.name_scope(""):
 318         # Replicas have to wait until they can get a token from the token queue.
 319         with ops.control_dependencies(train_ops):
 320           token = sync_token_queue.dequeue()
 321         train_op = state_ops.assign(self._local_step, token)
 322 
 323         with ops.control_dependencies([update_op]):
 324           # Sync_op needs to insert tokens to the token queue at the end of the
 325           # step so the replicas can fetch them to start the next step.
 326           tokens = array_ops.fill([self._tokens_per_step], global_step)
 327           sync_op = sync_token_queue.enqueue_many((tokens,))
 328 
 329         if self._variable_averages is not None:
 330           with ops.control_dependencies([sync_op]), ops.name_scope(""):
 331             sync_op = self._variable_averages.apply(
 332                 self._variables_to_average)
 333 
 334         self._chief_queue_runner = queue_runner.QueueRunner(dummy_queue,
 335                                                             [sync_op])
 336       for accum, dev in self._accumulator_list:
 337         with ops.device(dev):
 338           chief_init_ops.append(
 339               accum.set_global_step(
 340                   global_step, name="SetGlobalStep"))
 341       self.chief_init_op = control_flow_ops.group(*(chief_init_ops))
 342       self._gradients_applied = True
 343       return train_op
 344 
 345   def get_chief_queue_runner(self):
 346     """Returns the QueueRunner for the chief to execute.
 347 
 348     This includes the operations to synchronize replicas: aggregate gradients,
 349     apply to variables, increment global step, insert tokens to token queue.
 350 
 351     Note that this can only be called after calling apply_gradients() which
 352     actually generates this queuerunner.
 353 
 354     Returns:
 355       A `QueueRunner` for chief to execute.
 356 
 357     Raises:
 358       ValueError: If this is called before apply_gradients().
 359     """
 360     if self._gradients_applied is False:
 361       raise ValueError("Should be called after apply_gradients().")
 362 
 363     return self._chief_queue_runner

 404   def get_init_tokens_op(self, num_tokens=-1):
 405     """Returns the op to fill the sync_token_queue with the tokens.
 406 
 407     This is supposed to be executed in the beginning of the chief/sync thread
 408     so that even if the total_num_replicas is less than replicas_to_aggregate,
 409     the model can still proceed as the replicas can compute multiple steps per
 410     variable update. Make sure:
 411     `num_tokens >= replicas_to_aggregate - total_num_replicas`.
 412 
 413     Args:
 414       num_tokens: Number of tokens to add to the queue.
 415 
 416     Returns:
 417       An op for the chief/sync replica to fill the token queue.
 418 
 419     Raises:
 420       ValueError: If this is called before apply_gradients().
 421       ValueError: If num_tokens are smaller than replicas_to_aggregate -
 422         total_num_replicas.
 423     """
 424     if self._gradients_applied is False:
 425       raise ValueError(
 426           "get_init_tokens_op() should be called after apply_gradients().")
 427 
 428     tokens_needed = self._replicas_to_aggregate - self._total_num_replicas
 429     if num_tokens == -1:
 430       num_tokens = self._replicas_to_aggregate
 431     elif num_tokens < tokens_needed:
 432       raise ValueError(
 433           "Too few tokens to finish the first step: %d (given) vs %d (needed)" %
 434           (num_tokens, tokens_needed))
 435 
 436     if num_tokens > 0:
 437       with ops.device(self._global_step.device), ops.name_scope(""):
 438         tokens = array_ops.fill([num_tokens], self._global_step)
 439         init_tokens = self._sync_token_queue.enqueue_many((tokens,))
 440     else:
 441       init_tokens = control_flow_ops.no_op(name="no_init_tokens")
 442 
 443     return init_tokens

 445   def make_session_run_hook(self, is_chief, num_tokens=-1):
 446     """Creates a hook to handle SyncReplicasHook ops such as initialization."""
 447     return _SyncReplicasOptimizerHook(self, is_chief, num_tokens)

 450 class _SyncReplicasOptimizerHook(session_run_hook.SessionRunHook):
 451   """A SessionRunHook handles ops related to SyncReplicasOptimizer."""
 452 
 453   def __init__(self, sync_optimizer, is_chief, num_tokens):
 454     """Creates hook to handle SyncReplicasOptimizer initialization ops.
 455 
 456     Args:
 457       sync_optimizer: `SyncReplicasOptimizer` which this hook will initialize.
 458       is_chief: `Bool`, whether is this a chief replica or not.
 459       num_tokens: Number of tokens to add to the queue.
 460     """
 461     self._sync_optimizer = sync_optimizer
 462     self._is_chief = is_chief
 463     self._num_tokens = num_tokens
 464 
 465   def begin(self):
 466     if self._sync_optimizer._gradients_applied is False:  # pylint: disable=protected-access
 467       raise ValueError(
 468           "SyncReplicasOptimizer.apply_gradient should be called before using "
 469           "the hook.")
 470     if self._is_chief:
 471       self._local_init_op = self._sync_optimizer.chief_init_op
 472       self._ready_for_local_init_op = (
 473           self._sync_optimizer.ready_for_local_init_op)
 474       self._q_runner = self._sync_optimizer.get_chief_queue_runner()
 475       self._init_tokens_op = self._sync_optimizer.get_init_tokens_op(
 476           self._num_tokens)
 477     else:
 478       self._local_init_op = self._sync_optimizer.local_step_init_op
 479       self._ready_for_local_init_op = (
 480           self._sync_optimizer.ready_for_local_init_op)
 481       self._q_runner = None
 482       self._init_tokens_op = None
 483 
 484   def after_create_session(self, session, coord):
 485     """Runs SyncReplicasOptimizer initialization ops."""
 486     local_init_success, msg = session_manager._ready(  # pylint: disable=protected-access
 487         self._ready_for_local_init_op, session,
 488         "Model is not ready for SyncReplicasOptimizer local init.")
 489     if not local_init_success:
 490       raise RuntimeError(
 491           "Init operations did not make model ready for SyncReplicasOptimizer "
 492           "local_init. Init op: %s, error: %s" %
 493           (self._local_init_op.name, msg))
 494     session.run(self._local_init_op)
 495     if self._init_tokens_op is not None:
 496       session.run(self._init_tokens_op)
 497     if self._q_runner is not None:
 498       self._q_runner.create_threads(
 499           session, coord=coord, daemon=True, start=True)
