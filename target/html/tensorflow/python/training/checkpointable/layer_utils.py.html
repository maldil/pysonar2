  25 def is_layer(obj):
  26   """Implicit check for Layer-like objects."""
  27   # TODO(b/110718070): Replace with isinstance(obj, base_layer.Layer).
  28   return (hasattr(obj, "call")
  29           and hasattr(obj, "build")
  30           and hasattr(obj, "variables"))
  31 
  32 
  33 def has_weights(obj):
  34   """Implicit check for Layer-like objects."""
  35   # TODO(b/110718070): Replace with isinstance(obj, base_layer.Layer).
  36   return (hasattr(obj, "trainable_weights")
  37           and hasattr(obj, "non_trainable_weights"))
  38

  40 def filter_empty_layer_containers(layer_list):
  41   """Filter out empty Layer-like containers."""
  42   filtered = []
  43   for obj in layer_list:
  44     if is_layer(obj):
  45       filtered.append(obj)
  46     elif hasattr(obj, "layers"):
  47       # Checkpointable data structures will not show up in ".layers" lists, but
  48       # the layers they contain will.
  49       filtered.extend(obj.layers)
  50   return filtered

  53 def gather_trainable_weights(trainable, sub_layers, extra_variables):
  54   """Lists the trainable weights for an object with sub-layers.
  55 
  56   Args:
  57     trainable: Whether the object collecting the variables is trainable.
  58     sub_layers: A flat list of Layer objects owned by this object, to collect
  59       variables from.
  60     extra_variables: Any extra variables to include. Their `.trainable` property
  61       is used to categorize them.
  62 
  63   Returns:
  64     A list of collected trainable weights/variables.
  65   """
  66   if not trainable:
  67     return []
  68   weights = []
  69   for layer in sub_layers:
  70     weights += layer.trainable_weights
  71   trainable_extra_variables = [
  72       v for v in extra_variables if v.trainable]
  73   return weights + trainable_extra_variables

  76 def gather_non_trainable_weights(trainable, sub_layers, extra_variables):
  77   """Lists the non-trainable weights for an object with sub-layers.
  78 
  79   Args:
  80     trainable: Whether the object collecting the variables is trainable.
  81     sub_layers: A flat list of Layer objects owned by this object, to collect
  82       variables from.
  83     extra_variables: Any extra variables to include. Their `.trainable` property
  84       is used to categorize them.
  85 
  86   Returns:
  87     A list of collected non-trainable weights/variables.
  88   """
  89   trainable_extra_variables = []
  90   non_trainable_extra_variables = []
  91   for v in extra_variables:
  92     if v.trainable:
  93       trainable_extra_variables.append(v)
  94     else:
  95       non_trainable_extra_variables.append(v)
  96   weights = []
  97   for layer in sub_layers:
  98     weights += layer.non_trainable_weights
  99   if not trainable:
 100     trainable_weights = []
 101     for layer in sub_layers:
 102       trainable_weights += layer.trainable_weights
 103     return (trainable_weights + trainable_extra_variables
 104             + weights + non_trainable_extra_variables)
 105   return weights + non_trainable_extra_variables
