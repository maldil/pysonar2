  64 class CheckpointInitialValue(ops.Tensor):
  65   """Tensor wrapper for managing update UIDs in `Variables`.
  66 
  67   When supplied as an initial value, objects of this type let a `Variable`
  68   (`Variable`, `ResourceVariable`, etc.) know the UID of the restore the initial
  69   value came from. This allows deferred restorations to be sequenced in the
  70   order the user specified them, and lets us fall back on assignment if an
  71   initial value is not set (e.g. due to a custom getter interfering).
  72 
  73   See comments in _add_variable_with_custom_getter for more information about
  74   how `CheckpointInitialValue` is used.
  75   """
  76 
  77   def __init__(self, checkpoint_position, shape=None):
  78     self.wrapped_value = checkpoint_position.value_tensors()[
  79         VARIABLE_VALUE_KEY]
  80     if shape:
  81       # We need to set the static shape information on the initializer if
  82       # possible so we don't get a variable with an unknown shape.
  83       self.wrapped_value.set_shape(shape)
  84     self._checkpoint_position = checkpoint_position
  85 
  86   def __getattr__(self, attr):
  87     try:
  88       return getattr(self.wrapped_value, attr)
  89     except AttributeError:
  90       return self.__getattribute__(attr)
  91 
  92   @property
  93   def checkpoint_position(self):
  94     return self._checkpoint_position

 135 class PythonStringStateSaveable(PythonStateSaveable):
 136   """Saves Python state in a checkpoint."""
 137 
 138   def __init__(self, name, state_callback, restore_callback=None):
 139     """Configure saving.
 140 
 141     Args:
 142       name: The checkpoint key to write to.
 143       state_callback: A function taking no arguments which returns a
 144         string. This function is run every time a checkpoint is written.
 145       restore_callback: A function taking a Python string, used to restore
 146         state. Optional; defaults to doing nothing.
 147     """
 148     self._state_callback = state_callback
 149     self._restore_callback = restore_callback
 150     with ops.device("/cpu:0"):
 151       self._save_string = constant_op.constant("", dtype=dtypes.string)
 152     spec = saveable_object.SaveSpec(
 153         self._save_string, "", name, dtype=dtypes.string)
 154     super(PythonStringStateSaveable, self).__init__(
 155         self._save_string, [spec], name)
 156 
 157   def feed_dict_additions(self):
 158     """When running a graph, indicates fresh state to feed."""
 159     return {self._save_string: self._state_callback()}
 160 
 161   def freeze(self):
 162     """Create a frozen `SaveableObject` which saves the current state."""
 163     return NoRestoreSaveable(
 164         tensor=self._state_callback,
 165         dtype=dtypes.string,
 166         name=self.name)
 167 
 168   def python_restore(self, restored_strings):
 169     """Called to restore Python state."""
 170     if self._restore_callback:
 171       restored, = restored_strings
 172       self._restore_callback(restored)
 173 
 174   def restore(self, restored_tensors, restored_shapes):
 175     """Called to restore TensorFlow state (nothing to do)."""
 176     return control_flow_ops.no_op()

 179 class _CheckpointPosition(object):
 180   """Indicates a position within a `_Checkpoint`."""
 181 
 182   def __init__(self, checkpoint, proto_id):
 183     """Specify an object within a checkpoint.
 184 
 185     Args:
 186       checkpoint: A _Checkpoint object.
 187       proto_id: The index of this object in CheckpointableObjectGraph.nodes.
 188     """
 189     self._checkpoint = checkpoint
 190     self._proto_id = proto_id
 191 
 192   def restore(self, checkpointable):
 193     """Restore this value into `checkpointable`."""
 194     with ops.init_scope():
 195       if self.bind_object(checkpointable):
 196         # This object's correspondence with a checkpointed object is new, so
 197         # process deferred restorations for it and its dependencies.
 198         restore_ops = checkpointable._restore_from_checkpoint_position(self)  # pylint: disable=protected-access
 199         if restore_ops:
 200           self._checkpoint.new_restore_ops(restore_ops)
 201 
 202   def bind_object(self, checkpointable):
 203     """Set a checkpoint<->object correspondence and process slot variables.
 204 
 205     Args:
 206       checkpointable: The object to record a correspondence for.
 207     Returns:
 208       True if this is a new assignment, False if this object has already been
 209       mapped to a checkpointed `Object` proto.
 210     Raises:
 211       AssertionError: If another object is already bound to the `Object` proto.
 212     """
 213     checkpoint = self.checkpoint
 214     checkpoint.all_python_objects.add(checkpointable)
 215     current_assignment = checkpoint.object_by_proto_id.get(self._proto_id, None)
 216     if current_assignment is None:
 217       checkpoint.object_by_proto_id[self._proto_id] = checkpointable
 218       for deferred_slot_restoration in (
 219           checkpoint.deferred_slot_restorations.pop(self._proto_id, ())):
 220         checkpointable._create_or_restore_slot_variable(  # pylint: disable=protected-access
 221             slot_variable_position=_CheckpointPosition(
 222                 checkpoint=checkpoint,
 223                 proto_id=deferred_slot_restoration.slot_variable_id),
 224             variable=deferred_slot_restoration.original_variable,
 225             slot_name=deferred_slot_restoration.slot_name)
 226       for slot_restoration in checkpoint.slot_restorations.pop(
 227           self._proto_id, ()):
 228         optimizer_object = checkpoint.object_by_proto_id.get(
 229             slot_restoration.optimizer_id, None)
 230         if optimizer_object is None:
 231           # The optimizer has not yet been created or tracked. Record in the
 232           # checkpoint that the slot variables need to be restored when it is.
 233           checkpoint.deferred_slot_restorations.setdefault(
 234               slot_restoration.optimizer_id, []).append(
 235                   _DeferredSlotVariableRestoration(
 236                       original_variable=checkpointable,
 237                       slot_variable_id=slot_restoration.slot_variable_id,
 238                       slot_name=slot_restoration.slot_name))
 239         else:
 240           optimizer_object._create_or_restore_slot_variable(  # pylint: disable=protected-access
 241               slot_variable_position=_CheckpointPosition(
 242                   checkpoint=checkpoint,
 243                   proto_id=slot_restoration.slot_variable_id),
 244               variable=checkpointable,
 245               slot_name=slot_restoration.slot_name)
 246       return True  # New assignment
 247     else:
 248       # The object was already mapped for this checkpoint load, which means
 249       # we don't need to do anything besides check that the mapping is
 250       # consistent (if the dependency DAG is not a tree then there are
 251       # multiple paths to the same object).
 252       if current_assignment is not checkpointable:
 253         logging.warning(
 254             ("Inconsistent references when loading the checkpoint into this "
 255              "object graph. Either the Checkpointable object references in the "
 256              "Python program have changed in an incompatible way, or the "
 257              "checkpoint was generated in an incompatible program.\n\nTwo "
 258              "checkpoint references resolved to different objects (%s and %s).")
 259             % (current_assignment, checkpointable))
 260       return False  # Not a new assignment
 261 
 262   def is_simple_variable(self):
 263     """Determine whether this value is restorable with a Tensor initializer."""
 264     attributes = self.object_proto.attributes
 265     return (len(attributes) == 1
 266             and attributes[0].name == VARIABLE_VALUE_KEY
 267             and not self.object_proto.children)

 269   def value_tensors(self):
 270     """Create value `Tensor`s for this object's attributes.
 271 
 272     Does not require that the Python object has been created. Used for
 273     restore-on-create when executing eagerly.
 274 
 275     Returns:
 276       A dictionary mapping from object attribute names to `Tensor`s.
 277     """
 278     value_tensors = {}
 279     for serialized_tensor in self.object_proto.attributes:
 280       checkpoint_key = serialized_tensor.checkpoint_key
 281       dtype = self._checkpoint.dtype_map[checkpoint_key]
 282       base_type = dtype.base_dtype
 283       with ops.init_scope():
 284         with ops.device("/cpu:0"):
 285           # Run the restore itself on the CPU.
 286           value, = io_ops.restore_v2(
 287               prefix=self._checkpoint.save_path_tensor,
 288               tensor_names=[checkpoint_key],
 289               shape_and_slices=[""],
 290               dtypes=[base_type],
 291               name="%s_checkpoint_read" % (serialized_tensor.name,))
 292         # Copy the value to the current device if necessary.
 293         value_tensors[serialized_tensor.name] = array_ops.identity(value)
 294       return value_tensors

 296   def _gather_ops_or_named_saveables(self):
 297     """Looks up or creates SaveableObjects which don't have cached ops."""
 298     saveables = self.checkpointable._gather_saveables_for_checkpoint()  # pylint: disable=protected-access
 299     # Name saveables based on the name this object had when it was checkpointed.
 300     named_saveables = {}
 301     python_saveables = []
 302     existing_restore_ops = []
 303     for serialized_tensor in self.object_proto.attributes:
 304       if context.executing_eagerly():
 305         existing_op = None
 306       else:
 307         existing_op = self._checkpoint.restore_ops_by_name.get(
 308             serialized_tensor.checkpoint_key, None)
 309       if existing_op is not None:
 310         existing_restore_ops.append(existing_op)
 311         continue
 312 
 313       # Only if we don't have cached ops for this SaveableObject, we'll see if
 314       # the SaveableObject itself has been cached. If not, we'll make it, and
 315       # either way we'll extract new ops from it (or if it has Python state to
 316       # restore, we'll run that).
 317       if self._checkpoint.saveable_object_cache is None:
 318         # No SaveableObject caching when executing eagerly.
 319         saveable = None
 320       else:
 321         # If we've already created and cached a SaveableObject for this
 322         # attribute, we can re-use it to avoid re-creating some ops when graph
 323         # building.
 324         saveable_list = self._checkpoint.saveable_object_cache.get(
 325             self.checkpointable, {}).get(serialized_tensor.name, (None,))
 326         if len(saveable_list) == 1:
 327           # Almost every attribute will have exactly one SaveableObject.
 328           saveable, = saveable_list
 329         else:
 330           # Don't use cached SaveableObjects for partitioned variables, which is
 331           # the only case where we'd have a list of SaveableObjects. Op caching
 332           # will catch them.
 333           saveable = None
 334       if saveable is not None:
 335         # The name of this attribute has changed, so we need to re-generate
 336         # the SaveableObject.
 337         if serialized_tensor.checkpoint_key not in saveable.name:
 338           saveable = None
 339           del self._checkpoint.saveable_object_cache[self.checkpointable]
 340           break
 341       if saveable is None:
 342         # If there was no cached SaveableObject, we should check if the Python
 343         # object has the attribute.
 344         saveable_factory = saveables.get(serialized_tensor.name, None)
 345         if saveable_factory is None:
 346           # Purposefully does not throw an exception if attributes have been
 347           # added or deleted. Stores unused attributes so an exception can be
 348           # raised if the user decides to check that everything in the
 349           # checkpoint was loaded.
 350           self._checkpoint.unused_attributes.setdefault(
 351               self.checkpointable, []).append(serialized_tensor.name)
 352           continue
 353         if callable(saveable_factory):
 354           saveable = saveable_factory(name=serialized_tensor.checkpoint_key)
 355         else:
 356           saveable = saveable_factory
 357         if self._checkpoint.saveable_object_cache is not None:
 358           self._checkpoint.saveable_object_cache.setdefault(
 359               self.checkpointable, {})[serialized_tensor.name] = [saveable]
 360       if isinstance(saveable, PythonStateSaveable):
 361         python_saveables.append(saveable)
 362       else:
 363         named_saveables[serialized_tensor.checkpoint_key] = saveable
 364     return existing_restore_ops, named_saveables, python_saveables
 365 
 366   def restore_ops(self):
 367     """Create or fetch restore ops for this object's attributes.
 368 
 369     Requires that the `Checkpointable` Python object has been bound to an object
 370     ID in the checkpoint.
 371 
 372     Returns:
 373       A list of operations when graph building, or an empty list when executing
 374       eagerly.
 375     """
 376     (restore_ops,
 377      named_saveables,
 378      python_saveables) = self._gather_ops_or_named_saveables()
 379 
 380     # Eagerly run restorations for Python state.
 381     reader = pywrap_tensorflow.NewCheckpointReader(
 382         self._checkpoint.save_path_string)
 383     for saveable in python_saveables:
 384       spec_names = [spec.name for spec in saveable.specs]
 385       saveable.python_restore(
 386           [reader.get_tensor(name) for name in spec_names])
 387 
 388     # If we have new SaveableObjects, extract and cache restore ops.
 389     if named_saveables:
 390       validated_saveables = (
 391           self._checkpoint.builder._ValidateAndSliceInputs(named_saveables))  # pylint: disable=protected-access
 392       validated_names = set(saveable.name for saveable in validated_saveables)
 393       if set(named_saveables.keys()) != validated_names:
 394         raise AssertionError(
 395             ("Saveable keys changed when validating. Got back %s, was "
 396              "expecting %s") % (named_saveables.keys(), validated_names))
 397       all_tensors = self._checkpoint.builder.bulk_restore(
 398           filename_tensor=self._checkpoint.save_path_tensor,
 399           saveables=validated_saveables, preferred_shard=-1,
 400           restore_sequentially=False)
 401       saveable_index = 0
 402       for saveable in validated_saveables:
 403         num_specs = len(saveable.specs)
 404         saveable_tensors = all_tensors[
 405             saveable_index:saveable_index + num_specs]
 406         saveable_index += num_specs
 407         restore_op = saveable.restore(saveable_tensors, restored_shapes=None)
 408         if not context.executing_eagerly():
 409           assert saveable.name not in self._checkpoint.restore_ops_by_name
 410           self._checkpoint.restore_ops_by_name[saveable.name] = restore_op
 411           restore_ops.append(restore_op)
 412     return restore_ops

 454 def no_automatic_dependency_tracking(method):
 455   """Disables automatic dependency tracking on attribute assignment.
 456 
 457   Use to decorate any method of a Checkpointable object. Attribute assignment in
 458   that method will not add dependencies (also respected in Model). Harmless if
 459   used in a class which does not do automatic dependency tracking (which means
 460   it's safe to use in base classes which may have subclasses which also inherit
 461   from Checkpointable).
 462 
 463   Args:
 464     method: The method to decorate.
 465   Returns:
 466     A decorated method which sets and un-sets automatic dependency tracking for
 467     the object the method is called on (not thread safe).
 468   """
 469 
 470   def _method_wrapper(self, *args, **kwargs):
 471     previous_value = getattr(self, "_setattr_tracking", True)
 472     self._setattr_tracking = False  # pylint: disable=protected-access
 473     try:
 474       method(self, *args, **kwargs)
 475     finally:
 476       self._setattr_tracking = previous_value  # pylint: disable=protected-access
 477 
 478   return tf_decorator.make_decorator(
 479       target=method, decorator_func=_method_wrapper)

 482 class CheckpointableBase(object):
 483   """Base class for `Checkpointable` objects without automatic dependencies.
 484 
 485   This class has no __setattr__ override for performance reasons. Dependencies
 486   must be added explicitly. Unless attribute assignment is performance-critical,
 487   use `Checkpointable` instead. Use `CheckpointableBase` for `isinstance`
 488   checks.
 489   """
 490 
 491   # CheckpointableBase does not do automatic dependency tracking, but uses the
 492   # no_automatic_dependency_tracking decorator so it can avoid adding
 493   # dependencies if a subclass is Checkpointable / inherits from Model (both of
 494   # which have __setattr__ overrides).
 495   @no_automatic_dependency_tracking
 496   def _maybe_initialize_checkpointable(self):
 497     """Initialize dependency management.
 498 
 499     Not __init__, since most objects will forget to call it.
 500     """
 501     if hasattr(self, "_unconditional_checkpoint_dependencies"):
 502       # __init__ already called. This check means that we don't need
 503       # Checkpointable.__init__() in the constructor of every TensorFlow object.
 504       return
 505     # A list of CheckpointableReference objects. Some classes implementing
 506     # `Checkpointable`, notably `Optimizer`s, may override the
 507     # _checkpoint_dependencies property with conditional dependencies
 508     # (e.g. based on the current graph when saving).
 509     self._unconditional_checkpoint_dependencies = []
 510     # Maps names -> Checkpointable objects
 511     self._unconditional_dependency_names = {}
 512     # Restorations for other Checkpointable objects on which this object may
 513     # eventually depend. Maps local name -> _CheckpointPosition list. Optimizers
 514     # tack on conditional dependencies, and so need separate management of
 515     # deferred dependencies too.
 516     self._unconditional_deferred_dependencies = {}
 517     # The UID of the highest assignment to this object. Used to ensure that the
 518     # last requested assignment determines the final value of an object.
 519     if hasattr(self, "_update_uid"):
 520       raise AssertionError(
 521           "Internal error: the object had an update UID set before its "
 522           "initialization code was run.")
 523     self._update_uid = -1
 524     # When executing eagerly, holds a collection of _NameBasedRestoreCoordinator
 525     # instances, which should be checked when creating variables or other
 526     # saveables. These are passed on recursively to all dependencies, since
 527     # unlike object-based checkpoint restores we don't know which subgraph is
 528     # being restored in advance. This mechanism is only necessary for
 529     # restore-on-create when executing eagerly, and so is unused when graph
 530     # building.
 531     self._name_based_restores = set()
 532 
 533   def _no_dependency(self, value):
 534     """If automatic dependency tracking is enabled, ignores `value`."""
 535     return value
 536 
 537   def _name_based_attribute_restore(self, checkpoint):
 538     """Restore the object's attributes from a name-based checkpoint."""
 539     self._name_based_restores.add(checkpoint)
 540     if self._update_uid < checkpoint.restore_uid:
 541       checkpoint.eager_restore(self)
 542       self._update_uid = checkpoint.restore_uid
 543 
 544   @property
 545   def _checkpoint_dependencies(self):
 546     """All dependencies of this object.
 547 
 548     May be overridden to include conditional dependencies.
 549 
 550     Returns:
 551       A list of `CheckpointableReference` objects indicating named
 552       `Checkpointable` dependencies which should be saved along with this
 553       object.
 554     """
 555     return self._unconditional_checkpoint_dependencies
 556 
 557   @property
 558   def _deferred_dependencies(self):
 559     """A dictionary with deferred dependencies.
 560 
 561     Stores restorations for other Checkpointable objects on which this object
 562     may eventually depend. May be overridden by sub-classes (e.g. Optimizers use
 563     conditional dependencies based the current graph, and so need separate
 564     management of deferred dependencies too).
 565 
 566     Returns:
 567       A dictionary mapping from local name to a list of _CheckpointPosition
 568       objects.
 569     """
 570     return self._unconditional_deferred_dependencies
 571 
 572   def _lookup_dependency(self, name):
 573     """Look up a dependency by name.
 574 
 575     May be overridden to include conditional dependencies.
 576 
 577     Args:
 578       name: The local name of the dependency.
 579     Returns:
 580       A `Checkpointable` object, or `None` if no dependency by this name was
 581       found.
 582     """
 583     return self._unconditional_dependency_names.get(name, None)

 585   def _add_variable_with_custom_getter(
 586       self, name, shape=None, dtype=dtypes.float32,
 587       initializer=None, getter=None, overwrite=False,
 588       **kwargs_for_getter):
 589     """Restore-on-create for a variable be saved with this `Checkpointable`.
 590 
 591     If the user has requested that this object or another `Checkpointable` which
 592     depends on this object be restored from a checkpoint (deferred loading
 593     before variable object creation), `initializer` may be ignored and the value
 594     from the checkpoint used instead.
 595 
 596     Args:
 597       name: A name for the variable. Must be unique within this object.
 598       shape: The shape of the variable.
 599       dtype: The data type of the variable.
 600       initializer: The initializer to use. Ignored if there is a deferred
 601         restoration left over from a call to
 602         `_restore_from_checkpoint_position`.
 603       getter: The getter to wrap which actually fetches the variable.
 604       overwrite: If True, disables unique name and type checks.
 605       **kwargs_for_getter: Passed to the getter.
 606 
 607     Returns:
 608       The new variable object.
 609 
 610     Raises:
 611       ValueError: If the variable name is not unique.
 612     """
 613     self._maybe_initialize_checkpointable()
 614     with ops.init_scope():
 615       if context.executing_eagerly():
 616         # If this is a variable with a single Tensor stored in the checkpoint,
 617         # we can set that value as an initializer rather than initializing and
 618         # then assigning (when executing eagerly). This call returns None if
 619         # there is nothing to restore.
 620         checkpoint_initializer = self._preload_simple_restoration(
 621             name=name, shape=shape)
 622       else:
 623         checkpoint_initializer = None
 624       if (checkpoint_initializer is not None
 625           and not (
 626               isinstance(initializer, CheckpointInitialValue)
 627               and (initializer.restore_uid
 628                    > checkpoint_initializer.restore_uid))):
 629         # If multiple Checkpointable objects are "creating" the same variable
 630         # via the magic of custom getters, the one with the highest restore UID
 631         # (the one called last) has to make the final initializer. If another
 632         # custom getter interrupts this process by overwriting the initializer,
 633         # then we'll catch that when we call _track_checkpointable. So this is
 634         # "best effort" to set the initializer with the highest restore UID.
 635         initializer = checkpoint_initializer
 636         shape = None
 637     new_variable = getter(
 638         name=name, shape=shape, dtype=dtype, initializer=initializer,
 639         **kwargs_for_getter)
 640 
 641     # If we set an initializer and the variable processed it, tracking will not
 642     # assign again. It will add this variable to our dependencies, and if there
 643     # is a non-trivial restoration queued, it will handle that. This also
 644     # handles slot variables.
 645     if not overwrite or isinstance(new_variable, CheckpointableBase):
 646       return self._track_checkpointable(new_variable, name=name,
 647                                         overwrite=overwrite)
 648     else:
 649       # TODO(allenl): Some variable types are not yet supported. Remove this
 650       # fallback once all get_variable() return types are Checkpointable.
 651       return new_variable
 652 
 653   def _preload_simple_restoration(self, name, shape):
 654     """Return a dependency's value for restore-on-create.
 655 
 656     Note the restoration is not deleted; if for some reason preload is called
 657     and then not assigned to the variable (for example because a custom getter
 658     overrides the initializer), the assignment will still happen once the
 659     variable is tracked (determined based on checkpoint.restore_uid).
 660 
 661     Args:
 662       name: The object-local name of the dependency holding the variable's
 663         value.
 664       shape: The shape of the variable being loaded into.
 665     Returns:
 666       An callable for use as a variable's initializer/initial_value, or None if
 667       one should not be set (either because there was no variable with this name
 668       in the checkpoint or because it needs more complex deserialization). Any
 669       non-trivial deserialization will happen when the variable object is
 670       tracked.
 671     """
 672     deferred_dependencies_list = self._deferred_dependencies.get(name, ())
 673     if not deferred_dependencies_list:
 674       # Nothing to do; we don't have a restore for this dependency queued up.
 675       return
 676     for checkpoint_position in deferred_dependencies_list:
 677       if not checkpoint_position.is_simple_variable():
 678         # If _any_ pending restoration is too complicated to fit in an
 679         # initializer (because it has dependencies, or because there are
 680         # multiple Tensors to restore), bail and let the general tracking code
 681         # handle it.
 682         return None
 683     checkpoint_position = max(
 684         deferred_dependencies_list,
 685         key=lambda restore: restore.checkpoint.restore_uid)
 686     return CheckpointInitialValue(
 687         checkpoint_position=checkpoint_position, shape=shape)
 688 
 689   def _track_checkpointable(self, checkpointable, name, overwrite=False):
 690     """Declare a dependency on another `Checkpointable` object.
 691 
 692     Indicates that checkpoints for this object should include variables from
 693     `checkpointable`.
 694 
 695     Variables in a checkpoint are mapped to `Checkpointable`s based on the names
 696     provided when the checkpoint was written. To avoid breaking existing
 697     checkpoints when modifying a class, neither variable names nor dependency
 698     names (the names passed to `_track_checkpointable`) may change.
 699 
 700     Args:
 701       checkpointable: A `Checkpointable` which this object depends on.
 702       name: A local name for `checkpointable`, used for loading checkpoints into
 703         the correct objects.
 704       overwrite: Boolean, whether silently replacing dependencies is OK. Used
 705         for __setattr__, where throwing an error on attribute reassignment would
 706         be inappropriate.
 707 
 708     Returns:
 709       `checkpointable`, for convenience when declaring a dependency and
 710       assigning to a member variable in one statement.
 711 
 712     Raises:
 713       TypeError: If `checkpointable` does not inherit from `Checkpointable`.
 714       ValueError: If another object is already tracked by this name.
 715     """
 716     self._maybe_initialize_checkpointable()
 717     if not isinstance(checkpointable, CheckpointableBase):
 718       raise TypeError(
 719           ("Checkpointable._track_checkpointable() passed type %s, not a "
 720            "Checkpointable.") % (type(checkpointable),))
 721     new_reference = CheckpointableReference(name=name, ref=checkpointable)
 722     current_object = self._lookup_dependency(name)
 723     if (current_object is not None
 724         and current_object is not checkpointable):
 725       if not overwrite:
 726         raise ValueError(
 727             ("Called Checkpointable._track_checkpointable() with name='%s', "
 728              "but a Checkpointable with this name is already declared as a "
 729              "dependency. Names must be unique (or overwrite=True).") % (name,))
 730       # This is a weird thing to do, but we're not going to stop people from
 731       # using __setattr__.
 732       for index, (old_name, _) in enumerate(
 733           self._unconditional_checkpoint_dependencies):
 734         if name == old_name:
 735           self._unconditional_checkpoint_dependencies[index] = new_reference
 736     elif current_object is None:
 737       self._unconditional_checkpoint_dependencies.append(new_reference)
 738       self._handle_deferred_dependencies(
 739           name=name, checkpointable=checkpointable)
 740     self._unconditional_dependency_names[name] = checkpointable
 741     return checkpointable
 742 
 743   def _handle_deferred_dependencies(self, name, checkpointable):
 744     """Pop and load any deferred checkpoint restores into `checkpointable`.
 745 
 746     This method does not add a new dependency on `checkpointable`, but it does
 747     check if any outstanding/deferred dependencies have been queued waiting for
 748     this dependency to be added (matched based on `name`). If so,
 749     `checkpointable` and its dependencies are restored. The restorations are
 750     considered fulfilled and so are deleted.
 751 
 752     `_track_checkpointable` is more appropriate for adding a
 753     normal/unconditional dependency, and includes handling for deferred
 754     restorations. This method allows objects such as `Optimizer` to use the same
 755     restoration logic while managing conditional dependencies themselves, by
 756     overriding `_checkpoint_dependencies` and `_lookup_dependency` to change the
 757     object's dependencies based on the context it is saved/restored in (a single
 758     optimizer instance can have state associated with multiple graphs).
 759 
 760     Args:
 761       name: The name of the dependency within this object (`self`), used to
 762         match `checkpointable` with values saved in a checkpoint.
 763       checkpointable: The Checkpointable object to restore (inheriting from
 764         `CheckpointableBase`).
 765     """
 766     self._maybe_initialize_checkpointable()
 767     checkpointable._maybe_initialize_checkpointable()  # pylint: disable=protected-access
 768     deferred_dependencies_list = self._deferred_dependencies.pop(name, ())
 769     for checkpoint_position in sorted(
 770         deferred_dependencies_list,
 771         key=lambda restore: restore.checkpoint.restore_uid,
 772         reverse=True):
 773       checkpoint_position.restore(checkpointable)
 774 
 775     # Pass on any name-based restores queued in this object.
 776     for name_based_restore in sorted(
 777         self._name_based_restores,
 778         key=lambda checkpoint: checkpoint.restore_uid,
 779         reverse=True):
 780       checkpointable._name_based_attribute_restore(name_based_restore)  # pylint: disable=protected-access
 781 
 782   def _restore_from_checkpoint_position(self, checkpoint_position):
 783     """Restore this object and its dependencies (may be deferred)."""
 784     # Attempt a breadth-first traversal, since presumably the user has more
 785     # control over shorter paths. If we don't have all of the dependencies at
 786     # this point, the end result is not breadth-first (since other deferred
 787     # traversals will happen later).
 788     visit_queue = collections.deque([checkpoint_position])
 789     restore_ops = []
 790     while visit_queue:
 791       current_position = visit_queue.popleft()
 792       restore_ops.extend(nest.flatten(
 793           current_position.checkpointable  # pylint: disable=protected-access
 794           ._single_restoration_from_checkpoint_position(
 795               checkpoint_position=current_position,
 796               visit_queue=visit_queue)))
 797     return restore_ops
 798 
 799   def _single_restoration_from_checkpoint_position(
 800       self, checkpoint_position, visit_queue):
 801     """Restore this object, and either queue its dependencies or defer them."""
 802     self._maybe_initialize_checkpointable()
 803     checkpoint = checkpoint_position.checkpoint
 804     # If the UID of this restore is lower than our current update UID, we don't
 805     # need to actually restore the object. However, we should pass the
 806     # restoration on to our dependencies.
 807     if checkpoint.restore_uid > self._update_uid:
 808       restore_ops = checkpoint_position.restore_ops()
 809       self._update_uid = checkpoint.restore_uid
 810     else:
 811       restore_ops = ()
 812     for child in checkpoint_position.object_proto.children:
 813       child_position = _CheckpointPosition(
 814           checkpoint=checkpoint,
 815           proto_id=child.node_id)
 816       local_object = self._lookup_dependency(child.local_name)
 817       if local_object is None:
 818         # We don't yet have a dependency registered with this name. Save it
 819         # in case we do.
 820         self._deferred_dependencies.setdefault(child.local_name, []).append(
 821             child_position)
 822       else:
 823         if child_position.bind_object(checkpointable=local_object):
 824           # This object's correspondence is new, so dependencies need to be
 825           # visited. Delay doing it so that we get a breadth-first dependency
 826           # resolution order (shallowest paths first). The caller is responsible
 827           # for emptying visit_queue.
 828           visit_queue.append(child_position)
 829     return restore_ops

 831   def _gather_saveables_for_checkpoint(self):
 832     """Returns a dictionary of values to checkpoint with this object.
 833 
 834     Keys in the returned dictionary are local to this object and in a separate
 835     namespace from dependencies. Values may either be `SaveableObject` factories
 836     or variables easily converted to `SaveableObject`s (as in `tf.train.Saver`'s
 837     `var_list` constructor argument).
 838 
 839     `SaveableObjects` have a name set, which Checkpointable needs to generate
 840     itself. So rather than returning `SaveableObjects` directly, this method
 841     should return a dictionary of callables which take `name` arguments and
 842     return `SaveableObjects` with that name.
 843 
 844     If this object may also be passed to the global-name-based `tf.train.Saver`,
 845     the returned callables should have a default value for their name argument
 846     (i.e. be callable with no arguments).
 847 
 848     Returned values must be saved only by this object; if any value may be
 849     shared, it should instead be a dependency. For example, variable objects
 850     save their own values with the key `VARIABLE_VALUE_KEY`, but objects which
 851     reference variables simply add a dependency.
 852 
 853     Returns:
 854       The dictionary mapping attribute names to `SaveableObject` factories
 855       described above. For example:
 856       {VARIABLE_VALUE_KEY:
 857        lambda name="global_name_for_this_object":
 858        SaveableObject(name=name, ...)}
 859     """
 860     if not hasattr(self, "get_config"):
 861       return {}
 862     try:
 863       self.get_config()
 864     except NotImplementedError:
 865       return {}
 866     weak_self = weakref.ref(self)
 867     def _state_callback():
 868       dereferenced_self = weak_self()
 869       if dereferenced_self:
 870         return json.dumps(dereferenced_self,
 871                           default=serialization.get_json_type,
 872                           sort_keys=True).encode("utf8")
 873       else:
 874         return ""
 875     return {OBJECT_CONFIG_JSON_KEY: functools.partial(
 876         PythonStringStateSaveable,
 877         state_callback=_state_callback)}
