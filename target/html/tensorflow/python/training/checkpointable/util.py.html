  51 _ESCAPE_CHAR = "."  # For avoiding conflicts with user-specified names.

  53 # Keyword for identifying that the next bit of a checkpoint variable name is a
  54 # slot name. Checkpoint names for slot variables look like:
  55 #
  56 #   <path to variable>/<_OPTIMIZER_SLOTS_NAME>/<path to optimizer>/<slot name>
  57 #
  58 # Where <path to variable> is a full path from the checkpoint root to the
  59 # variable being slotted for.
  60 _OPTIMIZER_SLOTS_NAME = _ESCAPE_CHAR + "OPTIMIZER_SLOT"
  61 # Keyword for separating the path to an object from the name of an
  62 # attribute in checkpoint names. Used like:
  63 #   <path to variable>/<_OBJECT_ATTRIBUTES_NAME>/<name of attribute>
  64 _OBJECT_ATTRIBUTES_NAME = _ESCAPE_CHAR + "ATTRIBUTES"

  67 class _CheckpointRestoreCoordinator(object):
  68   """Holds the status of an object-based checkpoint load."""
  69 
  70   def __init__(self, object_graph_proto, save_path, save_path_tensor,
  71                restore_op_cache, saveable_object_cache):
  72     """Specify the checkpoint being loaded.
  73 
  74     Args:
  75       object_graph_proto: The CheckpointableObjectGraph protocol buffer
  76         associated with this checkpoint.
  77       save_path: A string, the path to the checkpoint, as returned by
  78         `tf.train.latest_checkpoint`.
  79       save_path_tensor: A string `Tensor` which contains or will be fed the save
  80         path.
  81       restore_op_cache: A dictionary shared between
  82         `_CheckpointRestoreCoordinator`s for the same Python objects, used to
  83         look up restore ops by name to avoid re-creating them across multiple
  84         `restore()` calls.
  85       saveable_object_cache: A mapping of checkpointable objects -> attribute
  86         names -> list(`SaveableObject`s), used when `SaveableObjects` must be
  87         referenced every restore (e.g. for Python state); otherwise they would
  88         create their own ops every restore.
  89     """
  90     self.builder = saver_lib.BulkSaverBuilder()
  91     self.object_graph_proto = object_graph_proto
  92     self.restore_uid = ops.uid()
  93     # Maps from objects to lists of attributes which were in the checkpoint but
  94     # not loaded into any object, for error checking.
  95     self.unused_attributes = weakref.WeakKeyDictionary()
  96     # Dictionary mapping from an id in the protocol buffer flat array to
  97     # Checkpointable Python objects. This mapping may be deferred if a
  98     # checkpoint is restored before all dependencies have been tracked. Uses
  99     # weak references so that partial restorations don't create reference cycles
 100     # (as objects with deferred dependencies will generally have references to
 101     # this object).
 102     self.object_by_proto_id = weakref.WeakValueDictionary()
 103     # A set of all Python objects we've seen as dependencies, even if we didn't
 104     # use them (for example because of inconsistent references when
 105     # loading). Used to make status assertions fail when loading checkpoints
 106     # that don't quite match.
 107     self.all_python_objects = _ObjectIdentityWeakSet()
 108     self.save_path_tensor = save_path_tensor
 109     self.save_path_string = save_path
 110     self.dtype_map = pywrap_tensorflow.NewCheckpointReader(
 111         save_path).get_variable_to_dtype_map()
 112     # A NewCheckpointReader for the most recent checkpoint, for streaming Python
 113     # state restoration.
 114     # When graph building, contains a list of ops to run to restore objects from
 115     # this checkpoint.
 116     self.restore_ops = []
 117     self.restore_ops_by_name = restore_op_cache
 118     self.saveable_object_cache = saveable_object_cache
 119     self.new_restore_ops_callback = None
 120     # A mapping from optimizer proto ids to lists of slot variables to be
 121     # restored when the optimizer is tracked. Only includes slot variables whose
 122     # regular variables have already been created, and only for optimizer
 123     # objects which have not yet been created/tracked.
 124     self.deferred_slot_restorations = {}
 125     # A mapping from variable proto ids to lists of slot variables to be
 126     # restored when the variable is created/tracked. These get shifted over to
 127     # deferred_slot_restorations if the optimizer hasn't been created when that
 128     # happens.
 129     self.slot_restorations = {}
 130     for node_index, node in enumerate(self.object_graph_proto.nodes):
 131       for slot_reference in node.slot_variables:
 132         # `node` refers to an `Optimizer`, since only these have slot variables.
 133         self.slot_restorations.setdefault(
 134             slot_reference.original_variable_node_id, []).append(
 135                 base._SlotVariableRestoration(  # pylint: disable=protected-access
 136                     optimizer_id=node_index,
 137                     slot_variable_id=slot_reference.slot_variable_node_id,
 138                     slot_name=slot_reference.slot_name))
 139 
 140   def new_restore_ops(self, new_ops):
 141     self.restore_ops.extend(new_ops)
 142     if self.new_restore_ops_callback:
 143       self.new_restore_ops_callback(new_ops)  # pylint: disable=not-callable

 146 class _NameBasedRestoreCoordinator(object):
 147   """Keeps the status of a name-based checkpoint restore."""
 148 
 149   def __init__(self, save_path, dtype_map=None):
 150     self.save_path = save_path
 151     self.dtype_map = dtype_map
 152     self.unused_attributes = weakref.WeakKeyDictionary()
 153     self.restore_uid = ops.uid()

 155   def globally_named_object_attributes(self, checkpointable):
 156     """Create globally named SaveableObjects from attributes.
 157 
 158     If an object's attribute has no global name specified (default construction
 159     for the SaveableObject factory), records the failure in
 160     `self.unused_attributes` (which can then be used to make status assertions
 161     fail; see `NameBasedSaverStatus`).
 162 
 163     Args:
 164       checkpointable: An object to save.
 165 
 166     Yields:
 167       SaveableObjects for `checkpointable`'s attributes.
 168     """
 169     for attribute_name, saveable_factory in (
 170         checkpointable._gather_saveables_for_checkpoint().items()):  # pylint: disable=protected-access
 171       if callable(saveable_factory):
 172         try:
 173           # This saveable object factory does not have a default name= argument,
 174           # which means there's no way to save/restore it using a name-based
 175           # checkpoint. Ignore the error now and make sure assert_consumed()
 176           # fails.
 177           saveable = saveable_factory()
 178         except TypeError:
 179           self.unused_attributes.setdefault(checkpointable, []).append(
 180               attribute_name)
 181           continue
 182       else:
 183         saveable = saveable_factory
 184       names_to_saveables = saver_lib.BaseSaverBuilder.OpListToDict(
 185           [saveable],
 186           convert_variable_to_tensor=False)
 187       for name, op in names_to_saveables.items():
 188         for saveable_object in saver_lib.BaseSaverBuilder.SaveableObjectsForOp(
 189             op=op, name=name):
 190           yield saveable_object
 191 
 192   def eager_restore(self, checkpointable):
 193     """Runs restore ops for `checkpointable`'s attributes."""
 194     # When graph building, we don't add any restore ops to the graph until
 195     # run_restore_ops/initialize_or_restore on the status object for name-based
 196     # checkpoints.
 197     assert context.executing_eagerly()
 198     for saveable in self.globally_named_object_attributes(
 199         checkpointable):
 200       restored_tensors = []
 201       tensor_missing = False
 202       for spec in saveable.specs:
 203         if spec.name in self.dtype_map:
 204           with ops.device("cpu:0"):
 205             restored, = io_ops.restore_v2(
 206                 prefix=self.save_path,
 207                 tensor_names=[spec.name],
 208                 shape_and_slices=[""],
 209                 dtypes=[self.dtype_map[spec.name]],
 210                 name="%s_checkpoint_read" % (spec.name,))
 211           restored_tensors.append(array_ops.identity(restored))
 212         else:
 213           tensor_missing = True
 214 
 215       if not tensor_missing:
 216         # Ignores values missing from the checkpoint, as with object-based
 217         # restore. Status assertions can be used to check exact matches,
 218         # although it's unlikely to ever happen for name-based checkpoints.
 219         saveable.restore(restored_tensors=restored_tensors,
 220                          restored_shapes=None)

 225 def _default_getter(name, shape, dtype, initializer=None,
 226                     partition_info=None, **kwargs):
 227   """A pared-down version of get_variable which does not reuse variables."""
 228   dtype = dtypes.as_dtype(dtype)
 229   shape_object = tensor_shape.as_shape(shape)
 230   with ops.init_scope():
 231     if initializer is None:
 232       initializer, initializing_from_value = (
 233           variable_scope._get_default_variable_store()._get_default_initializer(  # pylint: disable=protected-access
 234               name=name, shape=shape_object, dtype=dtype))
 235     else:
 236       initializing_from_value = not callable(initializer)
 237     # Same logic as get_variable
 238     variable_dtype = dtype.base_dtype
 239     if initializing_from_value:
 240       if shape is not None:
 241         raise ValueError("If initializer is a constant, do not specify shape.")
 242       initial_value = initializer
 243     else:
 244       # Instantiate initializer if provided initializer is a type object.
 245       if isinstance(initializer, type(init_ops.Initializer)):
 246         initializer = initializer(dtype=dtype)
 247       def initial_value():
 248         return initializer(
 249             shape_object.as_list(), dtype=dtype, partition_info=partition_info)
 250     return variables.VariableV1(
 251         initial_value=initial_value,
 252         name=name,
 253         dtype=variable_dtype,
 254         use_resource=True,
 255         **kwargs
 256     )

 267 def object_metadata(save_path):
 268   """Retrieves information about the objects in a checkpoint.
 269 
 270   Example usage:
 271 
 272   ```python
 273   object_graph = tf.contrib.checkpoint.object_metadata(
 274       tf.train.latest_checkpoint(checkpoint_directory))
 275   ckpt_variable_names = set()
 276   for node in object_graph.nodes:
 277     for attribute in node.attributes:
 278       ckpt_variable_names.add(attribute.full_name)
 279   ```
 280 
 281   Args:
 282     save_path: The path to the checkpoint, as returned by `save` or
 283       `tf.train.latest_checkpoint`.
 284   Returns:
 285     A parsed `tf.contrib.checkpoint.CheckpointableObjectGraph` protocol buffer.
 286   Raises:
 287     ValueError: If an object graph was not found in the checkpoint.
 288   """
 289   reader = pywrap_tensorflow.NewCheckpointReader(save_path)
 290   try:
 291     object_graph_string = reader.get_tensor(
 292         base.OBJECT_GRAPH_PROTO_KEY)
 293   except errors_impl.NotFoundError:
 294     raise ValueError(
 295         ('The specified checkpoint "%s" does not appear to be object-based (it '
 296          'is missing the key "%s"). Likely it was created with a name-based '
 297          'saver and does not contain an object dependency graph.') % (
 298              save_path, base.OBJECT_GRAPH_PROTO_KEY))
 299   object_graph_proto = (
 300       checkpointable_object_graph_pb2.CheckpointableObjectGraph())
 301   object_graph_proto.ParseFromString(object_graph_string)
 302   return object_graph_proto

 305 class _ObjectIdentityWrapper(object):
 306   """Wraps an object, mapping __eq__ on wrapper to "is" on wrapped.
 307 
 308   Since __eq__ is based on object identity, it's safe to also define __hash__
 309   based on object ids. This lets us add unhashable types like checkpointable
 310   _ListWrapper objects to object-identity collections.
 311   """
 312 
 313   def __init__(self, wrapped):
 314     self._wrapped = wrapped
 315 
 316   @property
 317   def unwrapped(self):
 318     return self._wrapped
 319 
 320   def __eq__(self, other):
 321     if isinstance(other, _ObjectIdentityWrapper):
 322       return self._wrapped is other._wrapped  # pylint: disable=protected-access
 323     return self._wrapped is other
 324 
 325   def __hash__(self):
 326     # Wrapper id() is also fine for weakrefs. In fact, we rely on
 327     # id(weakref.ref(a)) == id(weakref.ref(a)) and weakref.ref(a) is
 328     # weakref.ref(a) in _WeakObjectIdentityWrapper.
 329     return id(self._wrapped)
 330

 332 class _WeakObjectIdentityWrapper(_ObjectIdentityWrapper):
 333 
 334   def __init__(self, wrapped):
 335     super(_WeakObjectIdentityWrapper, self).__init__(weakref.ref(wrapped))
 336 
 337   @property
 338   def unwrapped(self):
 339     return self._wrapped()

 342 class _ObjectIdentityDictionary(collections.MutableMapping):
 343   """A mutable mapping data structure which compares using "is".
 344 
 345   This is necessary because we have checkpointable objects (_ListWrapper) which
 346   have behavior identical to built-in Python lists (including being unhashable
 347   and comparing based on the equality of their contents by default).
 348   """
 349 
 350   def __init__(self):
 351     self._storage = {}
 352 
 353   def _wrap_key(self, key):
 354     return _ObjectIdentityWrapper(key)
 355 
 356   def __getitem__(self, key):
 357     return self._storage[self._wrap_key(key)]
 358 
 359   def __setitem__(self, key, value):
 360     self._storage[self._wrap_key(key)] = value
 361 
 362   def __delitem__(self, key):
 363     del self._storage[self._wrap_key(key)]
 364 
 365   def __len__(self):
 366     return len(self._storage)
 367 
 368   def __iter__(self):
 369     for key in self._storage:
 370       yield key.unwrapped
 371

 373 class _ObjectIdentityWeakKeyDictionary(_ObjectIdentityDictionary):
 374   """Like weakref.WeakKeyDictionary, but compares objects with "is"."""
 375 
 376   def _wrap_key(self, key):
 377     return _WeakObjectIdentityWrapper(key)
 378 
 379   def __len__(self):
 380     # Iterate, discarding old weak refs
 381     return len(list(self._storage))
 382 
 383   def __iter__(self):
 384     keys = self._storage.keys()
 385     for key in keys:
 386       unwrapped = key.unwrapped
 387       if unwrapped is None:
 388         del self[key]
 389       else:
 390         yield unwrapped

 393 class _ObjectIdentitySet(collections.MutableSet):
 394   """Like the built-in set, but compares objects with "is"."""

 440 def _breadth_first_checkpointable_traversal(root_checkpointable):
 441   """Find shortest paths to all variables owned by dependencies of root."""
 442   bfs_sorted = []
 443   to_visit = collections.deque([root_checkpointable])
 444   path_to_root = _ObjectIdentityDictionary()
 445   path_to_root[root_checkpointable] = ()
 446   while to_visit:
 447     current_checkpointable = to_visit.popleft()
 448     if isinstance(current_checkpointable, tracking.NotCheckpointable):
 449       raise NotImplementedError(
 450           ("The object %s does not support object-based saving. File a feature "
 451            "request if this limitation bothers you. In the meantime, you can "
 452            "remove the dependency on this object and save everything else.")
 453           % (current_checkpointable,))
 454     current_checkpointable._maybe_initialize_checkpointable()  # pylint: disable=protected-access
 455     bfs_sorted.append(current_checkpointable)
 456     for child_checkpointable in (
 457         current_checkpointable._checkpoint_dependencies):  # pylint: disable=protected-access
 458       if child_checkpointable.ref not in path_to_root:
 459         path_to_root[child_checkpointable.ref] = (
 460             path_to_root[current_checkpointable] + (child_checkpointable,))
 461         to_visit.append(child_checkpointable.ref)
 462   return bfs_sorted, path_to_root

 465 def _escape_local_name(name):
 466   # We need to support slashes in local names for compatibility, since this
 467   # naming scheme is being patched in to things like Layer.add_variable where
 468   # slashes were previously accepted. We also want to use slashes to indicate
 469   # edges traversed to reach the variable, so we escape forward slashes in
 470   # names.
 471   return (name.replace(_ESCAPE_CHAR, _ESCAPE_CHAR + _ESCAPE_CHAR)
 472           .replace(r"/", _ESCAPE_CHAR + "S"))

 481 def _slot_variable_naming_for_optimizer(optimizer_path):
 482   """Make a function for naming slot variables in an optimizer."""
 483   # Name slot variables:
 484   #
 485   #   <variable name>/<_OPTIMIZER_SLOTS_NAME>/<optimizer path>/<slot name>
 486   #
 487   # where <variable name> is exactly the checkpoint name used for the original
 488   # variable, including the path from the checkpoint root and the local name in
 489   # the object which owns it. Note that we only save slot variables if the
 490   # variable it's slotting for is also being saved.
 491 
 492   optimizer_identifier = "/%s/%s/" % (_OPTIMIZER_SLOTS_NAME, optimizer_path)
 493 
 494   def _name_slot_variable(variable_path, slot_name):
 495     """With an optimizer specified, name a slot variable."""
 496     return (variable_path
 497             + optimizer_identifier
 498             + _escape_local_name(slot_name))
 499 
 500   return _name_slot_variable

 503 def _serialize_slot_variables(checkpointable_objects, node_ids, object_names):
 504   """Gather and name slot variables."""
 505   non_slot_objects = list(checkpointable_objects)
 506   slot_variables = _ObjectIdentityDictionary()
 507   for checkpointable in non_slot_objects:
 508     if isinstance(checkpointable, optimizer_lib.Optimizer):
 509       naming_scheme = _slot_variable_naming_for_optimizer(
 510           optimizer_path=object_names[checkpointable])
 511       slot_names = checkpointable.get_slot_names()
 512       for slot_name in slot_names:
 513         for original_variable_node_id, original_variable in enumerate(
 514             non_slot_objects):
 515           try:
 516             slot_variable = checkpointable.get_slot(
 517                 original_variable, slot_name)
 518           except AttributeError:
 519             slot_variable = None
 520           if slot_variable is None:
 521             continue
 522           slot_variable._maybe_initialize_checkpointable()  # pylint: disable=protected-access
 523           if slot_variable._checkpoint_dependencies:  # pylint: disable=protected-access
 524             # TODO(allenl): Gather dependencies of slot variables.
 525             raise NotImplementedError(
 526                 "Currently only variables with no dependencies can be saved as "
 527                 "slot variables. File a feature request if this limitation "
 528                 "bothers you.")
 529           if slot_variable in node_ids:
 530             raise NotImplementedError(
 531                 "A slot variable was re-used as a dependency of a "
 532                 "Checkpointable object. This is not currently allowed. File a "
 533                 "feature request if this limitation bothers you.")
 534           checkpoint_name = naming_scheme(
 535               variable_path=object_names[original_variable],
 536               slot_name=slot_name)
 537           object_names[slot_variable] = checkpoint_name
 538           slot_variable_node_id = len(checkpointable_objects)
 539           node_ids[slot_variable] = slot_variable_node_id
 540           checkpointable_objects.append(slot_variable)
 541           slot_variable_proto = (
 542               checkpointable_object_graph_pb2.CheckpointableObjectGraph
 543               .CheckpointableObject.SlotVariableReference(
 544                   slot_name=slot_name,
 545                   original_variable_node_id=original_variable_node_id,
 546                   slot_variable_node_id=slot_variable_node_id))
 547           slot_variables.setdefault(checkpointable, []).append(

 552 def _serialize_checkpointables(
 553     checkpointable_objects, node_ids, object_names, slot_variables,
 554     saveables_cache):
 555   """Name non-slot `Checkpointable`s and add them to `object_graph_proto`."""
 556   object_graph_proto = (
 557       checkpointable_object_graph_pb2.CheckpointableObjectGraph())
 558   named_saveables = []
 559   if saveables_cache is None:
 560     # No SaveableObject caching. Either we're executing eagerly, or building a
 561     # static save which is specialized to the current Python state.
 562     feed_additions = None
 563   else:
 564     # If we are caching SaveableObjects, we need to build up a feed_dict with
 565     # functions computing volatile Python state to be saved with the checkpoint.
 566     feed_additions = {}
 567   for checkpoint_id, checkpointable in enumerate(checkpointable_objects):
 568     assert node_ids[checkpointable] == checkpoint_id
 569     object_proto = object_graph_proto.nodes.add()
 570     object_proto.slot_variables.extend(slot_variables.get(checkpointable, ()))
 571     object_name = object_names[checkpointable]
 572     if saveables_cache is not None:
 573       cached_attributes = saveables_cache.setdefault(checkpointable, {})
 574     else:
 575       cached_attributes = None
 576     for name, saveable_factory in (
 577         checkpointable._gather_saveables_for_checkpoint().items()):  # pylint: disable=protected-access
 578       attribute = object_proto.attributes.add()
 579       attribute.name = name
 580       attribute.checkpoint_key = "%s/%s/%s" % (
 581           object_name, _OBJECT_ATTRIBUTES_NAME, _escape_local_name(name))
 582       if cached_attributes is None:
 583         saveables = None
 584       else:
 585         saveables = cached_attributes.get(name, None)
 586         if saveables is not None:
 587           for saveable in saveables:
 588             if attribute.checkpoint_key not in saveable.name:
 589               # The checkpoint key for this SaveableObject is different. We need
 590               # to re-create it.
 591               saveables = None
 592               del cached_attributes[name]
 593               break
 594       if saveables is None:
 595         if callable(saveable_factory):
 596           maybe_saveable = saveable_factory(name=attribute.checkpoint_key)
 597         else:
 598           maybe_saveable = saveable_factory
 599         if isinstance(maybe_saveable, saveable_object_lib.SaveableObject):
 600           saveables = (maybe_saveable,)
 601         else:
 602           # Figure out the name-based Saver's name for this variable. If it's
 603           # already a SaveableObject we'd just get the checkpoint key back, so
 604           # we leave full_name blank.
 605           saver_dict = saver_lib.BaseSaverBuilder.OpListToDict(
 606               [maybe_saveable], convert_variable_to_tensor=False)
 607           full_name, = saver_dict.keys()
 608           saveables = tuple(saver_lib.BaseSaverBuilder.SaveableObjectsForOp(
 609               op=maybe_saveable, name=attribute.checkpoint_key))
 610           for saveable in saveables:
 611             saveable.full_name = full_name
 612         for saveable in saveables:
 613           if attribute.checkpoint_key not in saveable.name:
 614             raise AssertionError(
 615                 ("The object %s produced a SaveableObject with name '%s' for "
 616                  "attribute '%s'. Expected a name containing '%s'.")
 617                 % (checkpointable, name, saveable.name,
 618                    attribute.checkpoint_key))
 619         if cached_attributes is not None:
 620           cached_attributes[name] = saveables
 621 
 622       for saveable in saveables:
 623         if hasattr(saveable, "full_name"):
 624           attribute.full_name = saveable.full_name
 625         if isinstance(saveable, base.PythonStateSaveable):
 626           if feed_additions is None:
 627             assert saveables_cache is None
 628             # If we're not caching saveables, then we're either executing
 629             # eagerly or building a static save/restore (e.g. for a
 630             # SavedModel). In either case, we should embed the current Python
 631             # state in the graph rather than relying on a feed dict.
 632             saveable = saveable.freeze()
 633           else:
 634             saveable_feed_dict = saveable.feed_dict_additions()
 635             for new_feed_key in saveable_feed_dict.keys():
 636               if new_feed_key in feed_additions:
 637                 raise AssertionError(
 638                     ("The object %s tried to feed a value for the Tensor %s "
 639                      "when saving, but another object is already feeding a "
 640                      "value.")
 641                     % (checkpointable, new_feed_key))
 642             feed_additions.update(saveable_feed_dict)
 643         named_saveables.append(saveable)
 644 
 645     for child in checkpointable._checkpoint_dependencies:  # pylint: disable=protected-access
 646       child_proto = object_proto.children.add()
 647       child_proto.node_id = node_ids[child.ref]
 648       child_proto.local_name = child.name
 649 
 650   return named_saveables, object_graph_proto, feed_additions

 653 def _serialize_object_graph(root_checkpointable, saveables_cache):
 654   """Determine checkpoint keys for variables and build a serialized graph.
 655 
 656   Non-slot variables are keyed based on a shortest path from the root saveable
 657   to the object which owns the variable (i.e. the one which called
 658   `Checkpointable._add_variable` to create it).
 659 
 660   Slot variables are keyed based on a shortest path to the variable being
 661   slotted for, a shortest path to their optimizer, and the slot name.
 662 
 663   Args:
 664     root_checkpointable: A `Checkpointable` object whose variables (including
 665       the variables of dependencies, recursively) should be saved.
 666     saveables_cache: A dictionary mapping `Checkpointable` objects -> attribute
 667       names -> SaveableObjects, used to avoid re-creating SaveableObjects when
 668       graph building.
 669 
 670   Returns:
 671     A tuple of (named_variables, object_graph_proto, feed_additions):
 672       named_variables: A dictionary mapping names to variable objects.
 673       object_graph_proto: A CheckpointableObjectGraph protocol buffer containing
 674         the serialized object graph and variable references.
 675       feed_additions: A dictionary mapping from Tensors to values which should
 676         be fed when saving.
 677 
 678   Raises:
 679     ValueError: If there are invalid characters in an optimizer's slot names.
 680   """
 681   checkpointable_objects, path_to_root = (
 682       _breadth_first_checkpointable_traversal(root_checkpointable))
 683   object_names = _ObjectIdentityDictionary()
 684   for obj, path in path_to_root.items():
 685     object_names[obj] = _object_prefix_from_path(path)
 686   node_ids = _ObjectIdentityDictionary()
 687   for node_id, node in enumerate(checkpointable_objects):
 688     node_ids[node] = node_id
 689   slot_variables = _serialize_slot_variables(
 690       checkpointable_objects=checkpointable_objects,
 691       node_ids=node_ids,
 692       object_names=object_names)
 693   return _serialize_checkpointables(
 694       checkpointable_objects=checkpointable_objects,
 695       node_ids=node_ids,
 696       object_names=object_names,
 697       slot_variables=slot_variables,
 698       saveables_cache=saveables_cache)

 756 @tf_contextlib.contextmanager
 757 def capture_dependencies(template):
 758   """Capture variables created within this scope as `Template` dependencies.
 759 
 760   Requires that `template.variable_scope` is active.
 761 
 762   This scope is intended as a compatibility measure, allowing a checkpointable
 763   object to add dependencies on variables created in a block of code which is
 764   not aware of object-based saving (and instead uses variable names
 765   heavily). This is how `Template` objects add dependencies on variables and
 766   sub-`Template`s. Where possible, use `tf.make_template` directly.
 767 
 768   Args:
 769     template: The `Template` object to register dependencies with.
 770 
 771   Yields:
 772     None (when used as a context manager).
 773   """
 774   name_prefix = template.variable_scope.name
 775 
 776   def _checkpointable_custom_creator(next_creator, name, initial_value,
 777                                      checkpointable_parent=None, **kwargs):
 778     """A variable creation hook which adds Checkpointable dependencies.
 779 
 780     Set for example during a `Template`'s first wrapped function
 781     execution. Ensures that (a) `template` depends on any checkpointable
 782     objects using their own `capture_dependencies` scope inside this scope which
 783     create variables, and (b) that any variables not in a more deeply nested
 784     scope are added as dependencies directly.
 785 
 786     The `checkpointable_parent` argument is passed between custom creators but
 787     ignored when the variable object itself is created. This argument indicates
 788     (if not `None`) that a more deeply nested scope has already added the
 789     variable as a dependency, and that parent scopes should add a dependency on
 790     that object rather than on the variable directly.
 791 
 792     Args:
 793       next_creator: See `variable_scope.variable_creator_scope`; the next
 794         creator in the chain.
 795       name: The (full, scope-influenced) name of the variable. The `name_prefix`
 796         itself is stripped for the purposes of object-based dependency tracking,
 797         but scopes opened within this scope are respected.
 798       initial_value: See `variable_scope.variable_creator_scope`. Taken
 799         explicitly so the argument can be re-named and used with
 800         `Checkpointable._add_variable_with_custom_getter`.
 801       checkpointable_parent: If not None, a more deeply nested checkpointable
 802         object and its name prefix which were passed to `capture_dependencies`
 803         to add a dependency on (rather than depending on the variable directly).
 804       **kwargs: Passed through to the next creator.
 805 
 806     Returns:
 807       The output of `next_creator`: the fetched/created variable object.
 808     """
 809     def _call_next_creator_renaming_initializer(initializer, **inner_kwargs):
 810       inner_kwargs.pop("name")  # Ignored; this is the scope-stripped name which
 811                                 # we don't want to propagate.
 812       return next_creator(
 813           initial_value=initializer,
 814           name=name,
 815           **inner_kwargs)
 816     if name is not None and name.startswith(name_prefix):
 817       scope_stripped_name = name[len(name_prefix) + 1:]
 818       if not checkpointable_parent:
 819         return template._add_variable_with_custom_getter(  # pylint: disable=protected-access
 820             initializer=initial_value,
 821             name=scope_stripped_name,
 822             getter=_call_next_creator_renaming_initializer,
 823             # Disable error checking for Checkpointable. Exceptions are instead
 824             # raised if necessary when the object-based saver tries to
 825             # save/restore the object.
 826             overwrite=True,
 827             checkpointable_parent=(template, name_prefix),
 828             **kwargs)
 829       else:
 830         parent_object, parent_name_prefix = checkpointable_parent
 831         template._track_checkpointable(  # pylint: disable=protected-access
 832             parent_object,
 833             name=parent_name_prefix[len(name_prefix) + 1:],
 834             overwrite=True)
 835     return next_creator(
 836         name=name, initial_value=initial_value,
 837         checkpointable_parent=(template, name_prefix), **kwargs)
 838 
 839   with variable_scope.variable_creator_scope(_checkpointable_custom_creator):
 840     yield

 892 class CheckpointLoadStatus(_LoadStatus):
 893   """Checks the status of checkpoint loading and manages restore ops.
 894 
 895   Returned from `Saver.restore`. Since `restore` may defer the loading of values
 896   in the checkpoint which don't yet have corresponding Python objects,
 897   `CheckpointLoadStatus` provides a callback to verify that checkpoint loading
 898   is complete (`assert_consumed`).
 899 
 900   When graph building, `restore` does not run restore ops itself since their
 901   creation may be deferred. The `run_restore_ops` method must be called once all
 902   Python objects with values to restore have been created and added to the
 903   dependency graph (this does not necessarily have to be the whole checkpoint;
 904   calling `run_restore_ops` while `assert_consumed` fails is supported and will
 905   partially restore the checkpoint).
 906 
 907   See `Saver.restore` for usage examples.
 908   """
 909 
 910   def __init__(self, checkpoint, feed_dict, root_checkpointable):
 911     self._checkpoint = checkpoint
 912     self._feed_dict = feed_dict
 913     self._root_checkpointable = root_checkpointable
 914 
 915   def assert_consumed(self):
 916     """Asserts that all objects in the checkpoint have been created/matched.
 917 
 918     Returns:
 919       `self` for chaining.
 920     Raises:
 921       AssertionError: If there are any Python objects in the dependency graph
 922         which have not been restored from this checkpoint or a later `restore`,
 923         or if there are any checkpointed values which have not been matched to
 924         Python objects.
 925     """
 926     self.assert_existing_objects_matched()
 927     for node_id, node in enumerate(self._checkpoint.object_graph_proto.nodes):
 928       checkpointable = self._checkpoint.object_by_proto_id.get(node_id, None)
 929       if checkpointable is None:
 930         raise AssertionError("Unresolved object in checkpoint: %s" % (node,))
 931     if self._checkpoint.slot_restorations:
 932       # Sanity check; this collection should be clear if everything has been
 933       # restored.
 934       raise AssertionError("Unresolved slot restorations: %s" % (
 935           self._checkpoint.slot_restorations,))
 936     if self._checkpoint.unused_attributes:
 937       raise AssertionError(
 938           ("Unused attributes in these objects (the attributes exist in the "
 939            "checkpoint but not in the objects): %s") % (
 940                self._checkpoint.unused_attributes.items(),))
 941     return self
 942 
 943   def assert_existing_objects_matched(self):
 944     """Asserts that checkpointable Python objects have been matched.
 945 
 946     Note that this is a weaker assertion than `assert_consumed`. It will only
 947     fail for existing Python objects which are (transitive) dependencies of the
 948     root object and which do not have an entry in the checkpoint.
 949 
 950     It will not fail, for example, if a `tf.keras.Layer` object has not yet been
 951     built and so has not created any `tf.Variable` objects.
 952 
 953     Returns:
 954       `self` for chaining.
 955 
 956     Raises:
 957       AssertionError: If a Python object exists in the transitive dependencies
 958         of the root object but does not have a value in the checkpoint.
 959     """
 960     for node_id, node in enumerate(self._checkpoint.object_graph_proto.nodes):
 961       checkpointable = self._checkpoint.object_by_proto_id.get(node_id, None)
 962       if (checkpointable is not None
 963           and checkpointable._update_uid < self._checkpoint.restore_uid):  # pylint: disable=protected-access
 964         raise AssertionError(
 965             "Object not assigned a value from checkpoint: %s" % (node,))
 966     for checkpointable_object in list_objects(self._root_checkpointable):
 967       self._checkpoint.all_python_objects.add(checkpointable_object)
 968     unused_python_objects = (
 969         _ObjectIdentitySet(self._checkpoint.all_python_objects)
 970         - _ObjectIdentitySet(self._checkpoint.object_by_proto_id.values()))
 971     if unused_python_objects:
 972       raise AssertionError(
 973           ("Some Python objects were not bound to checkpointed values, likely "
 974            "due to changes in the Python program: %s")
 975           % (list(unused_python_objects),))
 976     return self
 977 
 978   def run_restore_ops(self, session=None):
 979     """Run operations to restore objects in the dependency graph."""
 980     if context.executing_eagerly():
 981       return  # Run eagerly
 982     if session is None:
 983       session = ops.get_default_session()
 984     session.run(self._checkpoint.restore_ops, feed_dict=self._feed_dict)
 985 
 986   def initialize_or_restore(self, session=None):
 987     """Run operations to initialize or restore objects in the dependency graph.
 988 
 989     Any objects in the dependency graph which have initializers but are not in
 990     the checkpoint will have those initializers run, unless those variables are
 991     being restored by a later call to `tf.train.Checkpoint.restore()`.
 992 
 993     This method has a sibling in `InitializationOnlyStatus` which instead
 994     initializes variables. That type is returned if no checkpoint is specified
 995     in `Saver.restore`.
 996 
 997     Args:
 998       session: The session to run init/restore ops in. If `None`, uses the
 999         default session.
1000     """
1001     if context.executing_eagerly():
1002       return  # Initialization and restoration ops are run eagerly
1003     if session is None:
1004       session = ops.get_default_session()
1005     all_objects = list_objects(self._root_checkpointable)
1006     already_initialized_objects = _ObjectIdentitySet(
1007         self._checkpoint.object_by_proto_id.values())
1008     initializers_for_non_restored_variables = [
1009         c.initializer for c in all_objects
1010         if hasattr(c, "initializer")
1011         and c not in already_initialized_objects
1012         and (getattr(c, "_update_uid", self._checkpoint.restore_uid - 1)
1013              < self._checkpoint.restore_uid)]
1014     self.run_restore_ops(session=session)
1015     session.run(initializers_for_non_restored_variables)

1372   def restore(self, save_path):
1373     """Restore a training checkpoint.
1374 
1375     Restores `root_checkpointable` and any objects that it tracks
1376     (transitive). Either assigns values immediately if variables to restore have
1377     been created already, or defers restoration until the variables are
1378     created. Dependencies added to the `root_checkpointable` passed to the
1379     constructor after this call will be matched if they have a corresponding
1380     object in the checkpoint.
1381 
1382     When building a graph, restorations are added to the graph but not run.
1383 
1384     To disallow deferred loading, assert immediately that all checkpointed
1385     variables have been matched to variable objects:
1386 
1387     ```python
1388     saver = Saver(root)
1389     saver.restore(path).assert_consumed()
1390     ```
1391 
1392     An exception will be raised unless every object was matched and its
1393     variables already exist.
1394 
1395     When graph building, `assert_consumed()` indicates that all of the restore
1396     ops which will be created for this checkpoint have been created. They can be
1397     run via the `run_restore_ops()` function of the status object:
1398 
1399     ```python
1400     saver.restore(path).assert_consumed().run_restore_ops()
1401     ```
1402 
1403     If the checkpoint has not been consumed completely, then the list of restore
1404     ops will grow as more objects are added to the dependency graph.
1405 
1406     Name-based `tf.train.Saver` checkpoints can be loaded using this
1407     method. There is no deferred loading, and names are used to match
1408     variables. No restore ops are created/run until `run_restore_ops()` or
1409     `initialize_or_restore()` are called on the returned status object, even
1410     when executing eagerly. Re-encode name-based checkpoints using this
1411     object-based `Saver.save` as soon as possible.
1412 
1413     Args:
1414       save_path: The path to the checkpoint, as returned by `save` or
1415         `tf.train.latest_checkpoint`. If None (as when there is no latest
1416         checkpoint for `tf.train.latest_checkpoint` to return), returns an
1417         object which may run initializers for objects in the dependency
1418         graph. If the checkpoint was written by the name-based `tf.train.Saver`,
1419         names are used to match variables.
1420

1422       A load status object, which can be used to make assertions about the
1423       status of checkpoint restoration and run initialization/restore ops
1424       (of type `CheckpointLoadStatus`, or `InitializationOnlyStatus` if
1425       `save_path` is `None`).
1426 
1427       If `save_path` points to a name-based checkpoint, a `NameBasedSaverStatus`
1428       object is returned which runs restore ops from a name-based saver.
1429     """
1430     if save_path is None:
1431       return InitializationOnlyStatus(self._root_checkpointable, ops.uid())
1432     reader = pywrap_tensorflow.NewCheckpointReader(save_path)
1433     graph_building = not context.executing_eagerly()
1434     if graph_building:
1435       dtype_map = None
1436     else:
1437       dtype_map = reader.get_variable_to_dtype_map()
1438     try:
1439       object_graph_string = reader.get_tensor(
1440           base.OBJECT_GRAPH_PROTO_KEY)
1441     except errors_impl.NotFoundError:
1442       # The object graph proto does not exist in this checkpoint. Try the
1443       # name-based compatibility mode.
1444       restore_coordinator = _NameBasedRestoreCoordinator(
1445           save_path=save_path, dtype_map=dtype_map)
1446       if not graph_building:
1447         for existing_checkpointable in list_objects(self._root_checkpointable):
1448           # pylint: disable=protected-access
1449           existing_checkpointable._maybe_initialize_checkpointable()
1450           existing_checkpointable._name_based_restores.add(restore_coordinator)
1451           existing_checkpointable._name_based_attribute_restore(
1452               restore_coordinator)
1453           # pylint: enable=protected-access
1454       return NameBasedSaverStatus(
1455           restore_coordinator, root_checkpointable=self._root_checkpointable)
1456 
1457     if graph_building:
1458       if self._file_prefix_placeholder is None:
1459         with ops.device("/cpu:0"):
1460           self._file_prefix_placeholder = constant_op.constant("model")
1461       file_prefix_tensor = self._file_prefix_placeholder
1462       file_prefix_feed_dict = {self._file_prefix_placeholder: save_path}
1463     else:
1464       with ops.device("/cpu:0"):
1465         file_prefix_tensor = constant_op.constant(save_path)
1466       file_prefix_feed_dict = None
1467     object_graph_proto = (
1468         checkpointable_object_graph_pb2.CheckpointableObjectGraph())
1469     object_graph_proto.ParseFromString(object_graph_string)
1470     checkpoint = _CheckpointRestoreCoordinator(
1471         object_graph_proto=object_graph_proto,
1472         save_path=save_path,
1473         save_path_tensor=file_prefix_tensor,
1474         restore_op_cache=self._restore_op_cache,
1475         saveable_object_cache=self._saveable_object_cache)
1476     base._CheckpointPosition(  # pylint: disable=protected-access
1477         checkpoint=checkpoint, proto_id=0).restore(self._root_checkpointable)
1478     load_status = CheckpointLoadStatus(
1479         checkpoint,
1480         root_checkpointable=self._root_checkpointable,

1509 @tf_export("train.Checkpoint")
1510 class Checkpoint(tracking.Checkpointable):
1511   """Groups checkpointable objects, saving and restoring them.
1512 
1513   `Checkpoint`'s constructor accepts keyword arguments whose values are types
1514   that contain checkpointable state, such as `tf.train.Optimizer`
1515   implementations, `tf.Variable`, `tf.keras.Layer` implementations, or
1516   `tf.keras.Model` implementations. It saves these values with a checkpoint, and
1517   maintains a `save_counter` for numbering checkpoints.
1518 
1519   Example usage when graph building:
1520 
1521   ```python
1522   import tensorflow as tf
1523   import os
1524 
1525   checkpoint_directory = "/tmp/training_checkpoints"
1526   checkpoint_prefix = os.path.join(checkpoint_directory, "ckpt")
1527 
1528   checkpoint = tf.train.Checkpoint(optimizer=optimizer, model=model)
1529   status = checkpoint.restore(tf.train.latest_checkpoint(checkpoint_directory))
1530   train_op = optimizer.minimize( ... )
1531   status.assert_consumed()  # Optional sanity checks.
1532   with tf.Session() as session:
1533     # Use the Session to restore variables, or initialize them if
1534     # tf.train.latest_checkpoint returned None.
1535     status.initialize_or_restore(session)
1536     for _ in range(num_training_steps):
1537       session.run(train_op)
1538     checkpoint.save(file_prefix=checkpoint_prefix)
1539   ```
1540 
1541   Example usage with eager execution enabled:
1542 
1543   ```python
1544   import tensorflow as tf
1545   import os
1546 
1547   tf.enable_eager_execution()
1548 
1549   checkpoint_directory = "/tmp/training_checkpoints"
1550   checkpoint_prefix = os.path.join(checkpoint_directory, "ckpt")
1551

1553   status = checkpoint.restore(tf.train.latest_checkpoint(checkpoint_directory))
1554   for _ in range(num_training_steps):
1555     optimizer.minimize( ... )  # Variables will be restored on creation.
1556   status.assert_consumed()  # Optional sanity checks.
1557   checkpoint.save(file_prefix=checkpoint_prefix)

1560   `Checkpoint.save` and `Checkpoint.restore` write and read object-based
1561   checkpoints, in contrast to `tf.train.Saver` which writes and reads
1562   `variable.name` based checkpoints. Object-based checkpointing saves a graph of
1563   dependencies between Python objects (`Layer`s, `Optimizer`s, `Variable`s,
1564   etc.) with named edges, and this graph is used to match variables when
1565   restoring a checkpoint. It can be more robust to changes in the Python
1566   program, and helps to support restore-on-create for variables when executing
1567   eagerly. Prefer `tf.train.Checkpoint` over `tf.train.Saver` for new code.
1568 
1569   `Checkpoint` objects have dependencies on the objects passed as keyword
1570   arguments to their constructors, and each dependency is given a name that is
1571   identical to the name of the keyword argument for which it was created.
1572   TensorFlow classes like `Layer`s and `Optimizer`s will automatically add
1573   dependencies on their variables (e.g. "kernel" and "bias" for
1574   `tf.keras.layers.Dense`). Inheriting from `tf.keras.Model` makes managing
1575   dependencies easy in user-defined classes, since `Model` hooks into attribute
1576   assignment. For example:
1577 
1578   ```python
1579   class Regress(tf.keras.Model):
1580 
1581     def __init__(self):
1582       super(Regress, self).__init__()
1583       self.input_transform = tf.keras.layers.Dense(10)
1584       # ...
1585 
1586     def call(self, inputs):
1587       x = self.input_transform(inputs)
1588       # ...
1589   ```
1590 
1591   This `Model` has a dependency named "input_transform" on its `Dense` layer,
1592   which in turn depends on its variables. As a result, saving an instance of
1593   `Regress` using `tf.train.Checkpoint` will also save all the variables created
1594   by the `Dense` layer.
1595 
1596   Attributes:
1597     save_counter: Incremented when `save()` is called. Used to number
1598       checkpoints.
1599   """

1676     The saved checkpoint includes variables created by this object and any
1677     checkpointable objects it depends on at the time `Checkpoint.save()` is
1678     called.
1679 
1680     `save` is a basic convenience wrapper around the `write` method,
1681     sequentially numbering checkpoints using `save_counter` and updating the
1682     metadata used by `tf.train.latest_checkpoint`. More advanced checkpoint
1683     management, for example garbage collection and custom numbering, may be
1684     provided by other utilities which also wrap `write`
1685     (`tf.contrib.checkpoint.CheckpointManager` for example).
1686 
1687     Args:
1688       file_prefix: A prefix to use for the checkpoint filenames
1689         (/path/to/directory/and_a_prefix). Names are generated based on this
1690         prefix and `Checkpoint.save_counter`.
1691       session: The session to evaluate variables in. Ignored when executing
1692         eagerly. If not provided when graph building, the default session is
1693         used.
1694 
1695     Returns:
1696       The full path to the checkpoint.
1697     """
1698     graph_building = not context.executing_eagerly()
1699     if graph_building:
1700       if session is None:
1701         session = ops.get_default_session()
1702       if self._save_counter is None:
1703         # When graph building, if this is a new save counter variable then it
1704         # needs to be initialized before assign_add. This is only an issue if
1705         # restore() has not been called first.
1706         session.run(self.save_counter.initializer)
1707     if not graph_building or self._save_assign_op is None:
1708       with ops.colocate_with(self.save_counter):
1709         assign_op = self.save_counter.assign_add(1, read_value=True)
1710       if graph_building:
1711         self._save_assign_op = data_structures.NoDependency(assign_op)
1712     if graph_building:
1713       checkpoint_number = session.run(self._save_assign_op)
1714     else:
1715       checkpoint_number = assign_op.numpy()
1716     file_path = self.write("%s-%d" % (file_prefix, checkpoint_number),
1717                            session=session)
1718     checkpoint_management.update_checkpoint_state(
1719         save_dir=os.path.dirname(file_prefix),
1720         model_checkpoint_path=file_path,
1721         all_model_checkpoint_paths=[file_path])
1722     return file_path

1724   def restore(self, save_path):
1725     """Restore a training checkpoint.
1726 
1727     Restores this `Checkpoint` and any objects it depends on.
1728 
1729     When executing eagerly, either assigns values immediately if variables to
1730     restore have been created already, or defers restoration until the variables
1731     are created. Dependencies added after this call will be matched if they have
1732     a corresponding object in the checkpoint (the restore request will queue in
1733     any checkpointable object waiting for the expected dependency to be added).
1734 
1735     When graph building, restoration ops are added to the graph but not run
1736     immediately.
1737 
1738     To ensure that loading is complete and no more assignments will take place,
1739     use the `assert_consumed()` method of the status object returned by
1740     `restore`:
1741 
1742     ```python
1743     checkpoint = tf.train.Checkpoint( ... )
1744     checkpoint.restore(path).assert_consumed()
1745     ```
1746 
1747     An exception will be raised if any Python objects in the dependency graph
1748     were not found in the checkpoint, or if any checkpointed values do not have
1749     a matching Python object.
1750 
1751     When graph building, `assert_consumed()` indicates that all of the restore
1752     ops that will be created for this checkpoint have been created. They can be
1753     run via the `run_restore_ops()` method of the status object:
1754 
1755     ```python
1756     checkpoint.restore(path).assert_consumed().run_restore_ops()
1757     ```
1758 
1759     If the checkpoint has not been consumed completely, then the list of restore
1760     ops will grow as more objects are added to the dependency graph.
1761 
1762     Name-based `tf.train.Saver` checkpoints can be loaded using this
1763     method. Names are used to match variables. No restore ops are created/run
1764     until `run_restore_ops()` or `initialize_or_restore()` are called on the
1765     returned status object when graph building, but there is restore-on-creation
1766     when executing eagerly. Re-encode name-based checkpoints using
1767     `tf.train.Checkpoint.save` as soon as possible.
1768 
1769     Args:
1770       save_path: The path to the checkpoint, as returned by `save` or
1771         `tf.train.latest_checkpoint`. If None (as when there is no latest
1772         checkpoint for `tf.train.latest_checkpoint` to return), returns an
1773         object which may run initializers for objects in the dependency
1774         graph. If the checkpoint was written by the name-based `tf.train.Saver`,
1775         names are used to match variables.
1776 
1777     Returns:
1778       A load status object, which can be used to make assertions about the
1779       status of a checkpoint restoration and run initialization/restore ops.
1780 
1781       The returned status object has the following methods:
1782       - `assert_consumed()`:
1783           Raises an exception if any variables/objects are unmatched: either
1784           checkpointed values which don't have a matching Python object or
1785           Python objects in the dependency graph with no values in the
1786           checkpoint. This method returns the status object, and so may be
1787           chained with `initialize_or_restore` or `run_restore_ops`.
1788       -  `assert_existing_objects_matched()`:
1789           Raises an exception if any existing Python objects in the dependency
1790           graph are unmatched. Unlike `assert_consumed`, this assertion will
1791           pass if values in the checkpoint have no corresponding Python
1792           objects. For example a `tf.keras.Layer` object which has not yet been
1793           built, and so has not created any variables, will pass this assertion
1794           but fail `assert_consumed`. Useful when loading part of a larger
1795           checkpoint into a new Python program, e.g. a training checkpoint with
1796           a `tf.train.Optimizer` was saved but only the state required for
1797           inference is being loaded. This method returns the status object, and
1798           so may be chained with `initialize_or_restore` or `run_restore_ops`.
1799       - `initialize_or_restore(session=None)`:
1800           When graph building, runs variable initializers if `save_path` is
1801           `None`, but otherwise runs restore operations. If no `session` is
1802           explicitly specified, the default session is used. No effect when
1803           executing eagerly (variables are initialized or restored eagerly).
1804       - `run_restore_ops(session=None)`:
1805           When graph building, runs restore operations. If no `session` is
1806           explicitly specified, the default session is used. No effect when
1807           executing eagerly (restore operations are run eagerly). May only be
1808           called when `save_path` is not `None`.
1809     """
1810     status = self._saver.restore(save_path=save_path)
1811     # Create the save counter now so it gets initialized with other variables
1812     # when graph building. Creating it earlier would lead to double
1813     # initialization when executing eagerly.
1814     self._maybe_create_save_counter()
1815     return status
