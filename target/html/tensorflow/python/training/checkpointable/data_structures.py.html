  30 class NoDependency(object):
  31   """Allows attribute assignment to `Checkpointable` objects with no dependency.
  32 
  33   Example usage:
  34   ```python
  35   obj = Checkpointable()
  36   obj.has_dependency = tf.Variable(0., name="dep")
  37   obj.no_dependency = NoDependency(tf.Variable(1., name="nodep"))
  38   assert obj.no_dependency.name == "nodep:0"
  39   ```
  40 
  41   `obj` in this example has a dependency on the variable "dep", and both
  42   attributes contain un-wrapped `Variable` objects.
  43 
  44   `NoDependency` also works with `tf.keras.Model`, but only for checkpoint
  45   dependencies: wrapping a `Layer` in `NoDependency` will assign the (unwrapped)
  46   `Layer` to the attribute without a checkpoint dependency, but the `Model` will
  47   still track the `Layer` (so it will appear in `Model.layers`, and its
  48   variables will appear in `Model.variables`).
  49   """
  50 
  51   def __init__(self, value):
  52     self.value = value
  53 
  54 
  55 def _wrap_or_unwrap(value):
  56   """Wraps basic data structures, unwraps NoDependency objects."""
  57   if isinstance(value, NoDependency):
  58     return value.value
  59   if isinstance(value, base.CheckpointableBase):
  60     return value  # Skip conversion for already checkpointable objects.
  61   elif isinstance(value, dict):
  62     return _DictWrapper(value)
  63   elif isinstance(value, list):
  64     return _ListWrapper(value)
  65   else:
  66     return value
  67   # TODO(allenl): Handle other common data structures. Tuples will require
  68   # special casing (tuple subclasses are not weak referenceable, so replacement
  69   # with a wrapper that subclasses tuple on attribute assignment works poorly,
  70   # and replacement with a wrapper that isn't a tuple is also problematic),
  71   # probably a tree traversal where the leaves are non-tuples(/namedtuples) to
  72   # come up with names. Dictionaries should look like lists.
  73 
  74 
  75 def sticky_attribute_assignment(checkpointable, name, value):
  76   """Adds dependencies, generally called from __setattr__.
  77 
  78   This behavior is shared between Checkpointable and Model.
  79 
  80   Respects NoDependency indicators, but otherwise makes checkpointable objects
  81   out of common data structures and tracks objects by their attribute names.
  82 
  83   Args:
  84     checkpointable: The object to add dependencies to (generally the one having
  85       an attribute assigned).
  86     name: The attribute name being assigned.
  87     value: The value being assigned. Not necessarily a checkpointable object.
  88 
  89   Returns:
  90     The value which should be stored in the attribute (unwrapped from a
  91     NoDependency object if necessary).
  92   """
  93   if isinstance(value, NoDependency):
  94     add_dependency = False
  95   else:
  96     add_dependency = True
  97   value = _wrap_or_unwrap(value)
  98   if not add_dependency:
  99     return value
 100   if isinstance(value, base.CheckpointableBase):
 101     checkpointable._track_checkpointable(  # pylint: disable=protected-access
 102         value, name=name,
 103         # Allow the user to switch the Checkpointable which is tracked by this
 104         # name, since assigning a new variable to an attribute has
 105         # historically been fine (e.g. Adam did this).
 106         overwrite=True)
 107   return value

 110 class CheckpointableDataStructure(base.CheckpointableBase):
 111   """Base class for data structures which contain checkpointable objects."""
 112 
 113   def __init__(self):
 114     # An append-only ordered set
 115     self._layers = []
 116 
 117     self.trainable = True
 118     self._extra_variables = []
 119 
 120   def _track_value(self, value, name):
 121     """Add a dependency on `value`."""
 122     value = sticky_attribute_assignment(
 123         checkpointable=self, value=value, name=name)
 124     if isinstance(value, variables.Variable):
 125       self._extra_variables.append(value)
 126     if not isinstance(value, base.CheckpointableBase):
 127       raise ValueError(
 128           ("Only checkpointable objects (such as Layers or Optimizers) may be "
 129            "stored in a List object. Got %s, which does not inherit from "
 130            "CheckpointableBase.") % (value,))
 131     if (isinstance(value, CheckpointableDataStructure)
 132         or layer_utils.is_layer(value)
 133         or layer_utils.has_weights(value)):
 134       # Check for object-identity rather than with __eq__ to avoid
 135       # de-duplicating empty container types. Automatically generated list
 136       # wrappers keep things like "[] == []" true, which means "[] in [[]]" is
 137       # also true. This becomes not true once one of the lists is mutated.
 138       if not any((layer is value for layer in self._layers)):
 139         self._layers.append(value)
 140         if hasattr(value, "_use_resource_variables"):
 141           # In subclassed models, legacy layers (tf.layers) must always use
 142           # resource variables.
 143           value._use_resource_variables = True  # pylint: disable=protected-access
 144     return value
 145 
 146   @property
 147   def layers(self):
 148     return layer_utils.filter_empty_layer_containers(self._layers)
 149 
 150   @property
 151   def trainable_weights(self):
 152     return layer_utils.gather_trainable_weights(
 153         trainable=self.trainable,
 154         sub_layers=self._layers,
 155         extra_variables=self._extra_variables)
 156 
 157   @property
 158   def non_trainable_weights(self):
 159     return layer_utils.gather_non_trainable_weights(
 160         trainable=self.trainable,
 161         sub_layers=self._layers,
 162         extra_variables=self._extra_variables)
 163 
 164   @property
 165   def weights(self):
 166     return self.trainable_weights + self.non_trainable_weights
 167 
 168   @property
 169   def trainable_variables(self):
 170     return self.trainable_weights
 171 
 172   @property
 173   def non_trainable_variables(self):
 174     return self.non_trainable_weights
 175 
 176   @property
 177   def variables(self):
 178     return self.weights
 179 
 180   @property
 181   def updates(self):
 182     """Aggregate updates from any `Layer` instances."""
 183     # Updates and conditional losses are forwarded as-is rather than being
 184     # filtered based on inputs, since this is just a container and won't ever
 185     # have any inputs.
 186     aggregated = []
 187     for layer in self.layers:
 188       if hasattr(layer, "updates"):
 189         aggregated += layer.updates
 190     return aggregated
 191 
 192   @property
 193   def losses(self):
 194     """Aggregate losses from any `Layer` instances."""
 195     aggregated = []
 196     for layer in self.layers:
 197       if hasattr(layer, "losses"):
 198         aggregated += layer.losses
 199     return aggregated
 200 
 201   def __hash__(self):
 202     # Support object-identity hashing, so these structures can be used as keys
 203     # in sets/dicts.
 204     return id(self)
 205 
 206   def __eq__(self, other):
 207     # Similar to Tensors, checkpointable data structures use object-identity
 208     # equality to support set/dict membership.
 209     return self is other

 212 class List(CheckpointableDataStructure, collections.Sequence):
 213   """An append-only sequence type which is checkpointable.
 214 
 215   Maintains checkpoint dependencies on its contents (which must also be
 216   checkpointable), and forwards any `Layer` metadata such as updates and losses.
 217 
 218   Note that `List` is purely a container. It lets a `tf.keras.Model` or
 219   other checkpointable object know about its contents, but does not call any
 220   `Layer` instances which are added to it. To indicate a sequence of `Layer`
 221   instances which should be called sequentially, use `tf.keras.Sequential`.
 222 
 223   Example usage:
 224   ```python
 225   class HasList(tf.keras.Model):
 226 
 227     def __init__(self):
 228       super(HasList, self).__init__()
 229       self.layer_list = tf.contrib.checkpoint.List([layers.Dense(3)])
 230       self.layer_list.append(layers.Dense(4))
 231 
 232     def call(self, x):
 233       aggregation = 0.
 234       for l in self.layer_list:
 235         x = l(x)
 236         aggregation += tf.reduce_sum(x)
 237       return aggregation
 238   ```
 239 
 240   This kind of wrapping is necessary because `Checkpointable` objects do not
 241   (yet) deeply inspect regular Python data structures, so for example assigning
 242   a regular list (`self.layer_list = [layers.Dense(3)]`) does not create a
 243   checkpoint dependency and does not add the `Layer` instance's weights to its
 244   parent `Model`.
 245   """
 246 
 247   def __init__(self, *args, **kwargs):
 248     """Construct a new sequence. Arguments are passed to `list()`."""
 249     super(List, self).__init__()
 250     self._storage = self._make_storage(*args, **kwargs)
 251     for index, element in enumerate(self._storage):
 252       self._storage[index] = self._track_value(
 253           element, name=self._name_element(index))
 254 
 255   def __copy__(self):
 256     return type(self)(copy.copy(self._storage))
 257 
 258   def __deepcopy__(self, memo):
 259     return type(self)(copy.deepcopy(self._storage, memo))
 260 
 261   def _make_storage(self, *args, **kwargs):
 262     """Determines the backing storage (overridden in subclasses)."""
 263     return list(*args, **kwargs)
 264 
 265   def _name_element(self, index):
 266     return "%d" % (index,)
 267 
 268   def append(self, value):
 269     """Add a new checkpointable value."""
 270     value = self._track_value(value, self._name_element(len(self._storage)))
 271     self._storage.append(value)
 272 
 273   def extend(self, values):
 274     """Add a sequence of checkpointable values."""
 275     for value in values:
 276       self._storage.append(self._track_value(
 277           value, name=self._name_element(len(self._storage))))
 278 
 279   def __iadd__(self, values):
 280     self.extend(values)
 281     return self
 282 
 283   def __add__(self, other):
 284     if isinstance(other, List):
 285       return self.__class__(self._storage + other._storage)  # pylint: disable=protected-access
 286     else:
 287       return self.__class__(self._storage + other)
 288 
 289   def __radd__(self, other):
 290     return self + other
 291 
 292   def __getitem__(self, key):
 293     return self._storage[key]
 294 
 295   def __len__(self):
 296     return len(self._storage)
 297 
 298   def __repr__(self):
 299     return "List(%s)" % (repr(self._storage),)

 302 class _ListWrapper(List, collections.MutableSequence,
 303                    # Shadowed, but there for isinstance checks.
 304                    list):
 305   """Wraps the built-in `list` to support restore-on-create for variables.
 306 
 307   Unlike `List`, this sequence type is mutable in the same ways built-in lists
 308   are. Instead of throwing an error immediately like `List`, it records
 309   problematic mutations (e.g. assigning a new element to a position already
 310   occupied, meaning both elements get the same names at different times) and
 311   refuses to save.
 312 
 313   On assignment to an attribute of a Model or Checkpointable object, Python
 314   lists are replaced with _ListWrapper. Wrapping a list in a
 315   `tf.contrib.checkpoint.NoDependency` object prevents this.
 316   """
 317 
 318   def __init__(self, wrapped_list):
 319     """Construct a new list wrapper.
 320 
 321     Args:
 322       wrapped_list: The initial value of the data structure. A shallow copy may
 323         be maintained for error checking. `wrapped_list` itself should not be
 324         modified directly after constructing the `_ListWrapper`, and if changes
 325         are detected the `_ListWrapper` will throw an exception on save.
 326     """
 327     # Monotonic flags which indicate this object would not be restored properly,
 328     # and therefore should throw an error on save to avoid giving the impression
 329     # that restoring it will work.
 330     self._non_append_mutation = False
 331     self._external_modification = False
 332     super(_ListWrapper, self).__init__(wrapped_list)
 333     self._last_wrapped_list_snapshot = list(self._storage)
 334 
 335   # pylint: disable=protected-access
 336   def __copy__(self):
 337     copied = super(_ListWrapper, self).__copy__()
 338     copied._non_append_mutation = self._non_append_mutation
 339     copied._external_modification = self._external_modification
 340     return copied
 341 
 342   def __deepcopy__(self, memo):
 343     copied = super(_ListWrapper, self).__deepcopy__(memo)
 344     copied._non_append_mutation = self._non_append_mutation
 345     copied._external_modification = self._external_modification
 346     return copied
 347   # pylint: enable=protected-access
 348 
 349   def _make_storage(self, wrapped_list):
 350     """Use the user's original list for storage."""
 351     return wrapped_list
 352 
 353   def _check_external_modification(self):
 354     """Checks for any changes to the wrapped list not through the wrapper."""
 355     if self._external_modification or self._non_append_mutation:
 356       return
 357     if self._storage != self._last_wrapped_list_snapshot:
 358       self._external_modification = True
 359       self._last_wrapped_list_snapshot = None
 360 
 361   def _update_snapshot(self):
 362     """Acknowledges tracked changes to the wrapped list."""
 363     if self._external_modification or self._non_append_mutation:
 364       return
 365     self._last_wrapped_list_snapshot = list(self._storage)
 366 
 367   @property
 368   def _checkpoint_dependencies(self):
 369     self._check_external_modification()
 370     if self._non_append_mutation:
 371       raise ValueError(
 372           ("Unable to save the object %s (a list wrapper constructed to track "
 373            "checkpointable TensorFlow objects). A list element was replaced "
 374            "(__setitem__), deleted, or inserted. In order to support "
 375            "restoration on object creation, tracking is exclusively for "
 376            "append-only data structures.\n\nIf you don't need this list "
 377            "checkpointed, wrap it in a tf.contrib.checkpoint.NoDependency "
 378            "object; it will be automatically un-wrapped and subsequently "
 379            "ignored." % (self,)))
 380     if self._external_modification:
 381       raise ValueError(
 382           ("Unable to save the object %s (a list wrapper constructed to track "
 383            "checkpointable TensorFlow objects). The wrapped list was modified "
 384            "outside the wrapper (its final value was %s, its value when a "
 385            "checkpoint dependency was added was %s), which breaks restoration "
 386            "on object creation.\n\nIf you don't need this list checkpointed, "
 387            "wrap it in a tf.contrib.checkpoint.NoDependency object; it will be "
 388            "automatically un-wrapped and subsequently ignored." % (
 389                self, self._storage, self._last_wrapped_list_snapshot)))
 390     return super(_ListWrapper, self)._checkpoint_dependencies
 391 
 392   def __delitem__(self, key):
 393     self._non_append_mutation = True
 394     del self._storage[key]
 395 
 396   def __setitem__(self, key, value):
 397     self._non_append_mutation = True
 398     self._storage[key] = value
 399 
 400   def append(self, value):
 401     """Add a new checkpointable value."""
 402     self._check_external_modification()
 403     super(_ListWrapper, self).append(value)
 404     self._update_snapshot()
 405 
 406   def extend(self, values):
 407     """Add a sequence of checkpointable values."""
 408     self._check_external_modification()
 409     super(_ListWrapper, self).extend(values)
 410     self._update_snapshot()
 411 
 412   def __eq__(self, other):
 413     return self._storage == getattr(other, "_storage", other)
 414 
 415   def __ne__(self, other):
 416     return self._storage != getattr(other, "_storage", other)
 417 
 418   def __lt__(self, other):
 419     return self._storage < getattr(other, "_storage", other)
 420 
 421   def __le__(self, other):
 422     return self._storage <= getattr(other, "_storage", other)
 423 
 424   def __gt__(self, other):
 425     return self._storage > getattr(other, "_storage", other)
 426 
 427   def __ge__(self, other):
 428     return self._storage >= getattr(other, "_storage", other)
 429 
 430   def __hash__(self):
 431     # List wrappers need to compare like regular lists, and so like regular
 432     # lists they don't belong in hash tables.
 433     raise TypeError("unhashable type: 'ListWrapper'")
 434 
 435   def insert(self, index, obj):
 436     self._non_append_mutation = True
 437     self._storage.insert(index, obj)
 438 
 439   def _track_value(self, value, name):
 440     """Allows storage of non-checkpointable objects."""
 441     try:
 442       value = super(_ListWrapper, self)._track_value(value=value, name=name)
 443     except ValueError:
 444       # Even if this value isn't checkpointable, we need to make sure
 445       # NoDependency objects get unwrapped.
 446       value = sticky_attribute_assignment(
 447           checkpointable=self, value=value, name=name)
 448     return value
 449 
 450   def __repr__(self):
 451     return "ListWrapper(%s)" % (repr(self._storage),)

 454 class Mapping(CheckpointableDataStructure, collections.Mapping):
 455   """An append-only checkpointable mapping data structure with string keys.
 456 
 457   Maintains checkpoint dependencies on its contents (which must also be
 458   checkpointable), named based on its keys.
 459 
 460   Note that once a key has been added, it may not be deleted or replaced. If
 461   names may not be unique, see `tf.contrib.checkpoint.UniqueNameTracker`.
 462   """
 463 
 464   def __init__(self, *args, **kwargs):
 465     """Construct a new sequence. Arguments are passed to `dict()`."""
 466     super(Mapping, self).__init__()
 467     self._storage = self._make_storage(*args, **kwargs)
 468     self._storage.update(
 469         {key: self._track_value(
 470             value, name=self._name_element(key))
 471          for key, value in self._storage.items()})
 472 
 473   def __copy__(self):
 474     return type(self)(copy.copy(self._storage))
 475 
 476   def __deepcopy__(self, memo):
 477     return type(self)(copy.deepcopy(self._storage, memo))
 478 
 479   def _make_storage(self, *args, **kwargs):
 480     return dict(*args, **kwargs)
 481 
 482   def _name_element(self, key):
 483     if not isinstance(key, six.string_types):
 484       raise TypeError(
 485           "Mapping accepts only string keys, but got a key %s."
 486           % repr(key))
 487     return str(key)
 488 
 489   def __setitem__(self, key, value):
 490     name = self._name_element(key)
 491     value = self._track_value(value, name=name)
 492     current_value = self._storage.setdefault(key, value)
 493     if current_value is not value:
 494       raise ValueError(
 495           ("Mappings are an append-only data structure. Tried to overwrite the "
 496            "key '%s' with value %s, but it already contains %s")
 497           % (key, value, current_value))
 498 
 499   def update(self, *args, **kwargs):
 500     for key, value in dict(*args, **kwargs).items():
 501       self[key] = value
 502 
 503   def __getitem__(self, key):
 504     return self._storage[key]
 505 
 506   def __len__(self):
 507     return len(self._storage)
 508 
 509   def __repr__(self):
 510     return "Mapping(%s)" % (repr(self._storage),)
 511 
 512   def __iter__(self):
 513     return iter(self._storage)

 516 # Unlike _ListWrapper, having _DictWrapper inherit from dict and pass isinstance
 517 # checks seems infeasible. CPython will not call Python methods/properties on
 518 # dictionary subclasses when running e.g. {}.update(dict_subclass), and instead
 519 # collects elements directly from dict_subclass's C structs. So subclassing dict
 520 # implies that the storage has to be "self" (i.e. the C structs for the object
 521 # must be updated correctly), but we also need that storage to be the wrapped
 522 # dictionary to avoid synchronization bugs (un-tracked external modifications
 523 # should still show up when the dict is accessed through the wrapper). Monkey
 524 # patching all of the "wrapped" dict's methods instead of creating a wrapper
 525 # object is an option, but not a very attractive one (replacing methods without
 526 # creating reference cycles is difficult, and then dicts would need to be
 527 # special cased everywhere as being checkpointable).
 528 class _DictWrapper(Mapping, collections.MutableMapping):
 529   """Wraps built-in dicts to support restore-on-create for variables.
 530 
 531   _DictWrapper is to Mapping as _ListWrapper is to List. Unlike Mapping,
 532   _DictWrapper allows non-string keys and values and arbitrary mutations (delete
 533   keys, reassign values). Like _ListWrapper, these mutations mean that
 534   _DictWrapper will raise an exception on save.
 535   """
 536 
 537   def __new__(cls, *args):
 538     if len(args) == 1 and isinstance(args[0], dict):
 539       return super(_DictWrapper, cls).__new__(cls)
 540     else:
 541       # Allow construction from a sequence, e.g. for nest.pack_sequence_as. In
 542       # this case there's nothing to wrap, so we make a normal dictionary. Also
 543       # allows constructing empty instances of the _DictWrapper type, as Session
 544       # is wont to do (and again there's nothing to wrap, so a normal dictionary
 545       # makes more sense).
 546       return dict(*args)
 547 
 548   def __init__(self, wrapped_dict):
 549     self._non_string_key = False
 550     self._non_append_mutation = False
 551     self._external_modification = False
 552     super(_DictWrapper, self).__init__(wrapped_dict)
 553     self._update_snapshot()
 554 
 555   # pylint: disable=protected-access
 556   def __copy__(self):
 557     copied = super(_DictWrapper, self).__copy__()
 558     copied._non_append_mutation = self._non_append_mutation
 559     copied._external_modification = self._external_modification
 560     copied._non_string_key = self._non_string_key
 561     return copied
 562 
 563   def __deepcopy__(self, memo):
 564     copied = super(_DictWrapper, self).__deepcopy__(memo)
 565     copied._non_append_mutation = self._non_append_mutation
 566     copied._external_modification = self._external_modification
 567     copied._non_string_key = self._non_string_key
 568     return copied
 569   # pylint: enable=protected-access
 570 
 571   def _make_storage(self, wrapped_dict):
 572     """Re-use the wrapped dict for storage (to force them to be in sync)."""
 573     return wrapped_dict
 574 
 575   @property
 576   def _checkpoint_dependencies(self):
 577     """Check that the object is saveable before listing its dependencies."""
 578     self._check_external_modification()
 579     if self._non_string_key:
 580       raise ValueError(
 581           "Unable to save the object %s (a dictionary wrapper constructed "
 582           "automatically on attribute assignment). The wrapped dictionary "
 583           "contains a non-string key which maps to a checkpointable object or "
 584           "mutable data structure.\n\nIf you don't need this dictionary "
 585           "checkpointed, wrap it in a tf.contrib.checkpoint.NoDependency "
 586           "object; it will be automatically un-wrapped and subsequently "
 587           "ignored." % (self,))
 588     if self._non_append_mutation:
 589       raise ValueError(
 590           "Unable to save the object %s (a dictionary wrapper constructed "
 591           "automatically on attribute assignment). A key mapping to a "
 592           "checkpointable object was overwritten or deleted, which would "
 593           "cause problems for restoration.\n\nIf you don't need this "
 594           "dictionary checkpointed, wrap it in a "
 595           "tf.contrib.checkpoint.NoDependency object; it will be automatically "
 596           "un-wrapped and subsequently ignored." % (self,))
 597     if self._external_modification:
 598       raise ValueError(
 599           "Unable to save the object %s (a dictionary wrapper constructed "
 600           "automatically on attribute assignment). The wrapped dictionary was "
 601           "modified outside the wrapper (its final value was %s, its value "
 602           "when a checkpoint dependency was added was %s), which breaks "
 603           "restoration on object creation.\n\nIf you don't need this "
 604           "dictionary checkpointed, wrap it in a "
 605           "tf.contrib.checkpoint.NoDependency object; it will be automatically "
 606           "un-wrapped and subsequently ignored." % (
 607               self, self, self._last_wrapped_dict_snapshot))
 608     assert not self._dirty  # Any reason for dirtiness should have an exception.
 609     return super(_DictWrapper, self)._checkpoint_dependencies
 610 
 611   @property
 612   def _dirty(self):
 613     """Check if there has already been a mutation which prevents saving."""
 614     return (self._external_modification
 615             or self._non_append_mutation
 616             or self._non_string_key)
 617 
 618   def _check_external_modification(self):
 619     """Checks for any changes to the wrapped dict not through the wrapper."""
 620     if self._dirty:
 621       return
 622     if self != self._last_wrapped_dict_snapshot:
 623       self._external_modification = True
 624       self._last_wrapped_dict_snapshot = None
 625 
 626   def _update_snapshot(self):
 627     """Acknowledges tracked changes to the wrapped dict."""
 628     if self._dirty:
 629       return
 630     self._last_wrapped_dict_snapshot = dict(self)
 631 
 632   def _track_value(self, value, name):
 633     """Allows storage of non-checkpointable objects."""
 634     if isinstance(name, six.string_types):
 635       string_key = True
 636     else:
 637       name = "-non_string_key"
 638       string_key = False
 639     try:
 640       no_dependency = isinstance(value, NoDependency)
 641       value = super(_DictWrapper, self)._track_value(value=value, name=name)
 642       if not (string_key or no_dependency):
 643         # A non-string key maps to a checkpointable value. This data structure
 644         # is not saveable.
 645         self._non_string_key = True
 646       return value
 647     except ValueError:
 648       # Even if this value isn't checkpointable, we need to make sure
 649       # NoDependency objects get unwrapped.
 650       return sticky_attribute_assignment(
 651           checkpointable=self, value=value, name=name)
 652 
 653   def _name_element(self, key):
 654     """Don't throw errors for non-string keys."""
 655     if isinstance(key, six.string_types):
 656       return super(_DictWrapper, self)._name_element(key)
 657     else:
 658       return key
 659 
 660   def __setitem__(self, key, value):
 661     """Allow any modifications, but possibly mark the wrapper as unsaveable."""
 662     self._check_external_modification()
 663     no_dep = isinstance(value, NoDependency)
 664     if isinstance(key, six.string_types):
 665       existing_dependency = self._lookup_dependency(key)
 666       value = self._track_value(value, name=key)
 667     else:
 668       value = _wrap_or_unwrap(value)
 669       existing_dependency = None
 670       if not no_dep and isinstance(value, base.CheckpointableBase):
 671         # Non-string keys are OK as long as we have no reason to add a
 672         # dependency on the value (either because the value is not
 673         # checkpointable, or because it was wrapped in a NoDependency object).
 674         self._non_string_key = True
 675     current_value = self._storage.setdefault(key, value)
 676     if current_value is not value:
 677       if ((not no_dep and isinstance(value, base.CheckpointableBase))
 678           # We don't want to just check that the existing object is
 679           # checkpointable, since it may have been wrapped in a NoDependency
 680           # object.
 681           or existing_dependency is not None):
 682         # A checkpointable object was replaced under the same key; this means
 683         # that restoring would be error-prone, so we'll throw an exception on
 684         # save.
 685         self._non_append_mutation = True
 686       self._storage[key] = value
 687 
 688     self._update_snapshot()
 689 
 690   def __delitem__(self, key):
 691     self._check_external_modification()
 692     existing_value = self[key]
 693     if isinstance(existing_value, base.CheckpointableBase):
 694       # Deleting tracked checkpointable values means restoring is problematic,
 695       # so we'll throw an exception on save.
 696       self._non_append_mutation = True
 697     del self._storage[key]
 698     self._update_snapshot()
 699 
 700   def __repr__(self):
 701     return "DictWrapper(%s)" % (repr(self._storage),)
 702 
 703   def __hash__(self):
 704     raise TypeError("unhashable type: 'DictWrapper'")
 705 
 706   def __eq__(self, other):
 707     return self._storage == getattr(other, "_storage", other)
 708 
 709   def update(self, *args, **kwargs):
 710     for key, value in dict(*args, **kwargs).items():
 711       self[key] = value
