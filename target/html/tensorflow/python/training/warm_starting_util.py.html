  35 @tf_export("train.VocabInfo")
  36 class VocabInfo(
  37     collections.namedtuple("VocabInfo", [
  38         "new_vocab",
  39         "new_vocab_size",
  40         "num_oov_buckets",
  41         "old_vocab",
  42         "old_vocab_size",
  43         "backup_initializer",
  44         "axis",
  45     ])):
  46   """Vocabulary information for warm-starting.
  47 
  48   See `tf.estimator.WarmStartSettings` for examples of using
  49   VocabInfo to warm-start.
  50 
  51   Attributes:
  52     new_vocab: [Required] A path to the new vocabulary file (used with the
  53       model to be trained).
  54     new_vocab_size: [Required] An integer indicating how many entries of the new
  55       vocabulary will used in training.
  56     num_oov_buckets: [Required] An integer indicating how many OOV buckets are
  57       associated with the vocabulary.
  58     old_vocab: [Required] A path to the old vocabulary file (used with the
  59       checkpoint to be warm-started from).
  60     old_vocab_size: [Optional] An integer indicating how many entries of the old
  61       vocabulary were used in the creation of the checkpoint. If not provided,
  62       the entire old vocabulary will be used.
  63     backup_initializer: [Optional] A variable initializer used for variables
  64       corresponding to new vocabulary entries and OOV. If not provided, these
  65       entries will be zero-initialized.
  66     axis: [Optional] Denotes what axis the vocabulary corresponds to.  The
  67       default, 0, corresponds to the most common use case (embeddings or
  68       linear weights for binary classification / regression).  An axis of 1
  69       could be used for warm-starting output layers with class vocabularies.
  70 
  71       For example:
  72 
  73       embeddings_vocab_info = tf.VocabInfo(
  74           new_vocab='embeddings_vocab',
  75           new_vocab_size=100,
  76           num_oov_buckets=1,
  77           old_vocab='pretrained_embeddings_vocab',
  78           old_vocab_size=10000,
  79           backup_initializer=tf.truncated_normal_initializer(
  80               mean=0.0, stddev=(1 / math.sqrt(embedding_dim))),
  81           axis=0)
  82 
  83       softmax_output_layer_kernel_vocab_info = tf.VocabInfo(
  84           new_vocab='class_vocab',
  85           new_vocab_size=5,
  86           num_oov_buckets=0,  # No OOV for classes.
  87           old_vocab='old_class_vocab',
  88           old_vocab_size=8,
  89           backup_initializer=tf.glorot_uniform_initializer(),
  90           axis=1)
  91 
  92       softmax_output_layer_bias_vocab_info = tf.VocabInfo(
  93           new_vocab='class_vocab',
  94           new_vocab_size=5,
  95           num_oov_buckets=0,  # No OOV for classes.
  96           old_vocab='old_class_vocab',
  97           old_vocab_size=8,
  98           backup_initializer=tf.zeros_initializer(),
  99           axis=0)
 100 
 101       Currently, only axis=0 and axis=1 are supported.
 102   """
 103 
 104   def __new__(cls,
 105               new_vocab,
 106               new_vocab_size,
 107               num_oov_buckets,
 108               old_vocab,
 109               old_vocab_size=-1,
 110               backup_initializer=None,
 111               axis=0):
 112     if axis != 0 and axis != 1:
 113       raise ValueError("The only supported values for the axis argument are 0 "
 114                        "and 1.  Provided axis: {}".format(axis))
 115 
 116     return super(VocabInfo, cls).__new__(
 117         cls,
 118         new_vocab,
 119         new_vocab_size,
 120         num_oov_buckets,
 121         old_vocab,
 122         old_vocab_size,
 123         backup_initializer,
 124         axis,
 125     )

 128 def _infer_var_name(var):
 129   """Returns name of the `var`.
 130 
 131   Args:
 132     var: A list. The list can contain either of the following:
 133       (i) A single `Variable`
 134       (ii) A single `ResourceVariable`
 135       (iii) Multiple `Variable` objects which must be slices of the same larger
 136         variable.
 137       (iv) A single `PartitionedVariable`
 138 
 139   Returns:
 140     Name of the `var`
 141   """
 142   name_to_var_dict = saver.BaseSaverBuilder.OpListToDict(var)
 143   if len(name_to_var_dict) > 1:
 144     raise TypeError("`var` = %s passed as arg violates the constraints.  "
 145                     "name_to_var_dict = %s" % (var, name_to_var_dict))
 146   return list(name_to_var_dict.keys())[0]

 149 def _warm_start_var(var, prev_ckpt, prev_tensor_name=None):
 150   """Warm-starts given variable from `prev_tensor_name` tensor in `prev_ckpt`.
 151 
 152   Args:
 153     var: Current graph's variable that needs to be warm-started (initialized).
 154       Can be either of the following:
 155       (i) `Variable`
 156       (ii) `ResourceVariable`
 157       (iii) list of `Variable`: The list must contain slices of the same larger
 158         variable.
 159       (iv) `PartitionedVariable`
 160     prev_ckpt: A string specifying the directory with checkpoint file(s) or path
 161       to checkpoint. The given checkpoint must have tensor with name
 162       `prev_tensor_name` (if not None) or tensor with name same as given `var`.
 163     prev_tensor_name: Name of the tensor to lookup in provided `prev_ckpt`. If
 164       None, we lookup tensor with same name as given `var`.
 165   """
 166   if checkpoint_utils._is_variable(var):  # pylint: disable=protected-access
 167     current_var_name = _infer_var_name([var])
 168   elif (isinstance(var, list) and
 169         all(checkpoint_utils._is_variable(v) for v in var)):  # pylint: disable=protected-access
 170     current_var_name = _infer_var_name(var)
 171   elif isinstance(var, variables_lib.PartitionedVariable):
 172     current_var_name = _infer_var_name([var])
 173     var = var._get_variable_list()  # pylint: disable=protected-access
 174   else:
 175     raise TypeError(
 176         "var MUST be one of the following: a Variable, list of Variable or "
 177         "PartitionedVariable, but is {}".format(type(var)))
 178   if not prev_tensor_name:
 179     # Assume tensor name remains the same.
 180     prev_tensor_name = current_var_name
 181   checkpoint_utils.init_from_checkpoint(prev_ckpt, {prev_tensor_name: var})

 187 def _warm_start_var_with_vocab(var,
 188                                current_vocab_path,
 189                                current_vocab_size,
 190                                prev_ckpt,
 191                                prev_vocab_path,
 192                                previous_vocab_size=-1,
 193                                current_oov_buckets=0,
 194                                prev_tensor_name=None,
 195                                initializer=None,
 196                                axis=0):
 197   """Warm-starts given variable from `prev_tensor_name` tensor in `prev_ckpt`.
 198 
 199   Use this method when the `var` is backed by vocabulary. This method stitches
 200   the given `var` such that values corresponding to individual features in the
 201   vocabulary remain consistent irrespective of changing order of the features
 202   between old and new vocabularies.
 203 
 204   Args:
 205     var: Current graph's variable that needs to be warm-started (initialized).
 206       Can be either of the following:
 207       (i) `Variable`
 208       (ii) `ResourceVariable`
 209       (iii) list of `Variable`: The list must contain slices of the same larger
 210         variable.
 211       (iv) `PartitionedVariable`
 212     current_vocab_path: Path to the vocab file used for the given `var`.
 213     current_vocab_size: An `int` specifying the number of entries in the current
 214       vocab.
 215     prev_ckpt: A string specifying the directory with checkpoint file(s) or path
 216       to checkpoint. The given checkpoint must have tensor with name
 217       `prev_tensor_name` (if not None) or tensor with name same as given `var`.
 218     prev_vocab_path: Path to the vocab file used for the tensor in `prev_ckpt`.
 219     previous_vocab_size: If provided, will constrain previous vocab to the first
 220       `previous_vocab_size` entries.  -1 means use the entire previous vocab.
 221     current_oov_buckets: An `int` specifying the number of out-of-vocabulary
 222       buckets used for given `var`.
 223     prev_tensor_name: Name of the tensor to lookup in provided `prev_ckpt`. If
 224       None, we lookup tensor with same name as given `var`.
 225     initializer: Variable initializer to be used for missing entries.  If None,
 226       missing entries will be zero-initialized.
 227     axis: Axis of the variable that the provided vocabulary corresponds to.
 228 
 229   Raises:
 230     ValueError: If required args are not provided.
 231   """
 232   if not (current_vocab_path and current_vocab_size and prev_ckpt and
 233           prev_vocab_path):
 234     raise ValueError("Invalid args: Must provide all of [current_vocab_path, "
 235                      "current_vocab_size, prev_ckpt, prev_vocab_path}.")
 236   if checkpoint_utils._is_variable(var):
 237     var = [var]
 238   elif (isinstance(var, list) and
 239         all(checkpoint_utils._is_variable(v) for v in var)):
 240     var = var
 241   elif isinstance(var, variables_lib.PartitionedVariable):
 242     var = var._get_variable_list()
 243   else:
 244     raise TypeError(
 245         "var MUST be one of the following: a Variable, list of Variable or "
 246         "PartitionedVariable, but is {}".format(type(var)))
 247 
 248   if not prev_tensor_name:
 249     # Assume tensor name remains the same.
 250     prev_tensor_name = _infer_var_name(var)
 251 
 252   # TODO(eddz): Fix functionality for rank-1 Variables (like FC biases).
 253   total_v_first_axis = sum([v.get_shape().as_list()[0] for v in var])
 254   for v in var:
 255     v_shape = v.get_shape().as_list()
 256     slice_info = v._get_save_slice_info()
 257     partition_info = None
 258     if slice_info:
 259       partition_info = variable_scope._PartitionInfo(
 260           full_shape=slice_info.full_shape,
 261           var_offset=slice_info.var_offset)
 262 
 263     if axis == 0:
 264       new_row_vocab_size = current_vocab_size
 265       new_col_vocab_size = v_shape[1]
 266       old_row_vocab_size = previous_vocab_size
 267       old_row_vocab_file = prev_vocab_path
 268       new_row_vocab_file = current_vocab_path
 269       old_col_vocab_file = None
 270       new_col_vocab_file = None
 271       num_row_oov_buckets = current_oov_buckets
 272       num_col_oov_buckets = 0
 273     elif axis == 1:
 274       # Note that we must compute this value across all partitions, whereas
 275       # in the axis = 0 case, we can simply use v_shape[1] because we don't
 276       # allow partitioning across axis = 1.
 277       new_row_vocab_size = total_v_first_axis
 278       new_col_vocab_size = current_vocab_size
 279       old_row_vocab_size = -1
 280       old_row_vocab_file = None
 281       new_row_vocab_file = None
 282       old_col_vocab_file = prev_vocab_path
 283       new_col_vocab_file = current_vocab_path
 284       num_row_oov_buckets = 0
 285       num_col_oov_buckets = current_oov_buckets
 286     else:
 287       raise ValueError("The only supported values for the axis argument are 0 "
 288                        "and 1.  Provided axis: {}".format(axis))
 289 
 290     init = checkpoint_ops._load_and_remap_matrix_initializer(
 291         ckpt_path=checkpoint_utils._get_checkpoint_filename(prev_ckpt),
 292         old_tensor_name=prev_tensor_name,
 293         new_row_vocab_size=new_row_vocab_size,
 294         new_col_vocab_size=new_col_vocab_size,
 295         old_row_vocab_size=old_row_vocab_size,
 296         old_row_vocab_file=old_row_vocab_file,
 297         new_row_vocab_file=new_row_vocab_file,
 298         old_col_vocab_file=old_col_vocab_file,
 299         new_col_vocab_file=new_col_vocab_file,
 300         num_row_oov_buckets=num_row_oov_buckets,
 301         num_col_oov_buckets=num_col_oov_buckets,
 302         initializer=initializer)
 303     new_init_val = ops.convert_to_tensor(
 304         init(shape=v_shape, partition_info=partition_info))
 305     v._initializer_op = state_ops.assign(v, new_init_val)
 306 # pylint: enable=protected-access

 309 def _get_grouped_variables(vars_to_warm_start):
 310   """Collects and groups (possibly partitioned) variables into a dictionary.
 311 
 312   The variables can be provided explicitly through vars_to_warm_start, or they
 313   are retrieved from collections (see below).
 314 
 315   Args:
 316     vars_to_warm_start: One of the following:
 317 
 318       - A regular expression (string) that captures which variables to
 319         warm-start (see tf.get_collection).  This expression will only consider
 320         variables in the TRAINABLE_VARIABLES collection.
 321       - A list of Variables to warm-start.
 322       - A list of strings, each representing a full variable name to warm-start.
 323       - `None`, in which case only variables specified in
 324         `var_name_to_vocab_info` will be warm-started.
 325   Returns:
 326     A dictionary mapping variable names (strings) to lists of Variables.
 327   Raises:
 328     ValueError: If vars_to_warm_start is not a string, `None`, a list of
 329       `Variables`, or a list of strings.
 330   """
 331   if isinstance(vars_to_warm_start, str) or vars_to_warm_start is None:
 332     # Both vars_to_warm_start = '.*' and vars_to_warm_start = None will match
 333     # everything (in TRAINABLE_VARIABLES) here.
 334     list_of_vars = ops.get_collection(
 335         ops.GraphKeys.TRAINABLE_VARIABLES,
 336         scope=vars_to_warm_start)
 337   elif isinstance(vars_to_warm_start, list):
 338     if all([isinstance(v, str) for v in vars_to_warm_start]):
 339       list_of_vars = []
 340       for v in vars_to_warm_start:
 341         list_of_vars += ops.get_collection(ops.GraphKeys.GLOBAL_VARIABLES,
 342                                            scope=v)
 343     elif all([checkpoint_utils._is_variable(v) for v in vars_to_warm_start]):  # pylint: disable=protected-access
 344       list_of_vars = vars_to_warm_start
 345     else:
 346       raise ValueError("If `vars_to_warm_start` is a list, it must be all "
 347                        "`Variable` or all `str`.  Given types are {}".format(
 348                            [type(v) for v in vars_to_warm_start]))
 349   else:
 350     raise ValueError("`vars_to_warm_start must be a `list` or `str`.  Given "
 351                      "type is {}".format(type(vars_to_warm_start)))
 352   # We have to deal with partitioned variables, since get_collection flattens
 353   # out the list.
 354   grouped_variables = {}
 355   for v in list_of_vars:
 356     if not isinstance(v, list):
 357       var_name = _infer_var_name([v])
 358     else:
 359       var_name = _infer_var_name(v)
 360     grouped_variables.setdefault(var_name, []).append(v)
 361 
 362   return grouped_variables

 365 @tf_export("train.warm_start")
 366 def warm_start(ckpt_to_initialize_from,
 367                vars_to_warm_start=".*",
 368                var_name_to_vocab_info=None,
 369                var_name_to_prev_var_name=None):
 370   """Warm-starts a model using the given settings.
 371 
 372   If you are using a tf.estimator.Estimator, this will automatically be called
 373   during training.
 374 
 375   Args:
 376     ckpt_to_initialize_from: [Required] A string specifying the directory with
 377       checkpoint file(s) or path to checkpoint from which to warm-start the
 378       model parameters.
 379     vars_to_warm_start: [Optional] One of the following:
 380 
 381       - A regular expression (string) that captures which variables to
 382         warm-start (see tf.get_collection).  This expression will only consider
 383         variables in the TRAINABLE_VARIABLES collection.
 384       - A list of Variables to warm-start.
 385       - A list of strings, each representing a full variable name to warm-start.
 386       - `None`, in which case only variables specified in
 387         `var_name_to_vocab_info` will be warm-started.
 388 
 389       Defaults to `'.*'`, which warm-starts all variables in the
 390       TRAINABLE_VARIABLES collection.  Note that this excludes variables such as
 391       accumulators and moving statistics from batch norm.
 392     var_name_to_vocab_info: [Optional] Dict of variable names (strings) to
 393       VocabInfo. The variable names should be "full" variables, not the names
 394       of the partitions.  If not explicitly provided, the variable is assumed to
 395       have no vocabulary.
 396     var_name_to_prev_var_name: [Optional] Dict of variable names (strings) to
 397       name of the previously-trained variable in `ckpt_to_initialize_from`. If
 398       not explicitly provided, the name of the variable is assumed to be same
 399       between previous checkpoint and current model.
 400   Raises:
 401     ValueError: If the WarmStartSettings contains prev_var_name or VocabInfo
 402       configuration for variable names that are not used.  This is to ensure
 403       a stronger check for variable configuration than relying on users to
 404       examine the logs.
 405   """
 406   if var_name_to_vocab_info is None:
 407     var_name_to_vocab_info = {}
 408   if var_name_to_prev_var_name is None:
 409     var_name_to_prev_var_name = {}
 410   logging.info("Warm-starting from: %s", (ckpt_to_initialize_from,))
 411   grouped_variables = _get_grouped_variables(vars_to_warm_start)
 412 
 413   # Keep track of which var_names in var_name_to_prev_var_name and
 414   # var_name_to_vocab_info have been used.  Err on the safer side by throwing an
 415   # exception if any are unused by the end of the loop.  It is easy to misname
 416   # a variable during this configuration, in which case without this check, we
 417   # would fail to warm-start silently.
 418   prev_var_name_used = set()
 419   vocab_info_used = set()
 420 
 421   for var_name, variable in six.iteritems(grouped_variables):
 422     prev_var_name = var_name_to_prev_var_name.get(var_name)
 423     if prev_var_name:
 424       prev_var_name_used.add(var_name)
 425     vocab_info = var_name_to_vocab_info.get(var_name)
 426     if vocab_info:
 427       vocab_info_used.add(var_name)
 428       logging.info(
 429           "Warm-starting variable: {}; current_vocab: {} current_vocab_size: {}"
 430           " prev_vocab: {} prev_vocab_size: {} current_oov: {} prev_tensor: {}"
 431           " initializer: {}".format(
 432               var_name,
 433               vocab_info.new_vocab,
 434               vocab_info.new_vocab_size,
 435               vocab_info.old_vocab,
 436               (vocab_info.old_vocab_size if vocab_info.old_vocab_size > 0
 437                else "All"),
 438               vocab_info.num_oov_buckets,
 439               prev_var_name or "Unchanged",
 440               vocab_info.backup_initializer or "zero-initialized"))
 441       _warm_start_var_with_vocab(
 442           variable,
 443           current_vocab_path=vocab_info.new_vocab,
 444           current_vocab_size=vocab_info.new_vocab_size,
 445           prev_ckpt=ckpt_to_initialize_from,
 446           prev_vocab_path=vocab_info.old_vocab,
 447           previous_vocab_size=vocab_info.old_vocab_size,
 448           current_oov_buckets=vocab_info.num_oov_buckets,
 449           prev_tensor_name=prev_var_name,
 450           initializer=vocab_info.backup_initializer,
 451           axis=vocab_info.axis)
 452     else:
 453       # For the special value of vars_to_warm_start = None,
 454       # we only warm-start variables with explicitly specified vocabularies.
 455       if vars_to_warm_start:
 456         logging.info("Warm-starting variable: {}; prev_var_name: {}".format(
 457             var_name, prev_var_name or "Unchanged"))
 458         # Because we use a default empty list in grouped_variables, single
 459         # unpartitioned variables will be lists here, which we rectify in order
 460         # for init_from_checkpoint logic to work correctly.
 461         if len(variable) == 1:
 462           variable = variable[0]
 463         _warm_start_var(variable, ckpt_to_initialize_from, prev_var_name)
 464 
 465   prev_var_name_not_used = set(
 466       var_name_to_prev_var_name.keys()) - prev_var_name_used
 467   vocab_info_not_used = set(var_name_to_vocab_info.keys()) - vocab_info_used
 468 
 469   if prev_var_name_not_used:
 470     raise ValueError(
 471         "You provided the following variables in "
 472         "var_name_to_prev_var_name that were not used: "
 473         "{0}.  Perhaps you misspelled them?  Here is the list of viable "
 474         "variable names: {1}".format(prev_var_name_not_used,
 475                                      grouped_variables.keys()))
 476   if vocab_info_not_used:
 477     raise ValueError(
 478         "You provided the following variables in "
 479         "var_name_to_vocab_info that were not used: {0}. "
 480         " Perhaps you misspelled them?  Here is the list of viable variable "
 481         "names: {1}".format(vocab_info_not_used, grouped_variables.keys()))
