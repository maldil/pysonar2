  29 def _make_server_def(server_or_cluster_def, job_name, task_index, protocol,
  30                      config):
  31   """Creates a `tf.train.ServerDef` protocol buffer.
  32 
  33   Args:
  34     server_or_cluster_def: A `tf.train.ServerDef` or
  35       `tf.train.ClusterDef` protocol buffer, or a
  36       `tf.train.ClusterSpec` object, describing the server to be
  37       defined and/or the cluster of which it is a member.
  38     job_name: (Optional.) Specifies the name of the job of which the server
  39       is a member. Defaults to the value in `server_or_cluster_def`, if
  40       specified.
  41     task_index: (Optional.) Specifies the task index of the server in its job.
  42       Defaults to the value in `server_or_cluster_def`, if specified. Otherwise
  43       defaults to 0 if the server's job has only one task.
  44     protocol: (Optional.) Specifies the protocol to be used by the server.
  45       Acceptable values include `"grpc", "grpc+verbs"`. Defaults to the value
  46       in `server_or_cluster_def`, if specified. Otherwise defaults to `"grpc"`.
  47     config: (Options.) A `tf.ConfigProto` that specifies default configuration
  48       options for all sessions that run on this server.
  49 
  50   Returns:
  51     A `tf.train.ServerDef`.
  52 
  53   Raises:
  54     TypeError: If the arguments do not have the appropriate type.
  55     ValueError: If an argument is not specified and cannot be inferred.
  56   """
  57   server_def = tensorflow_server_pb2.ServerDef()
  58   if isinstance(server_or_cluster_def, tensorflow_server_pb2.ServerDef):
  59     server_def.MergeFrom(server_or_cluster_def)
  60     if job_name is not None:
  61       server_def.job_name = job_name
  62     if task_index is not None:
  63       server_def.task_index = task_index
  64     if protocol is not None:
  65       server_def.protocol = protocol
  66     if config is not None:
  67       server_def.default_session_config.MergeFrom(config)
  68   else:
  69     try:
  70       cluster_spec = ClusterSpec(server_or_cluster_def)
  71     except TypeError:
  72       raise TypeError("Could not convert `server_or_cluster_def` to a "
  73                       "`tf.train.ServerDef` or `tf.train.ClusterSpec`.")
  74     if job_name is None:
  75       if len(cluster_spec.jobs) == 1:
  76         job_name = cluster_spec.jobs[0]
  77       else:
  78         raise ValueError("Must specify an explicit `job_name`.")
  79     if task_index is None:
  80       task_indices = cluster_spec.task_indices(job_name)
  81       if len(task_indices) == 1:
  82         task_index = task_indices[0]
  83       else:
  84         raise ValueError("Must specify an explicit `task_index`.")
  85     if protocol is None:
  86       protocol = "grpc"
  87 
  88     server_def = tensorflow_server_pb2.ServerDef(
  89         cluster=cluster_spec.as_cluster_def(),
  90         job_name=job_name, task_index=task_index, protocol=protocol)
  91     if config is not None:
  92       server_def.default_session_config.MergeFrom(config)
  93   return server_def

  96 @tf_export("train.Server")
  97 class Server(object):
  98   """An in-process TensorFlow server, for use in distributed training.
  99 
 100   A `tf.train.Server` instance encapsulates a set of devices and a
 101   `tf.Session` target that
 102   can participate in distributed training. A server belongs to a
 103   cluster (specified by a `tf.train.ClusterSpec`), and
 104   corresponds to a particular task in a named job. The server can
 105   communicate with any other server in the same cluster.
 106   """
 107 
 108   def __init__(self,
 109                server_or_cluster_def,
 110                job_name=None,
 111                task_index=None,
 112                protocol=None,
 113                config=None,
 114                start=True):
 115     """Creates a new server with the given definition.
 116 
 117     The `job_name`, `task_index`, and `protocol` arguments are optional, and
 118     override any information provided in `server_or_cluster_def`.
 119 
 120     Args:
 121       server_or_cluster_def: A `tf.train.ServerDef` or
 122         `tf.train.ClusterDef` protocol buffer, or a
 123         `tf.train.ClusterSpec` object, describing the server to be
 124         created and/or the cluster of which it is a member.
 125       job_name: (Optional.) Specifies the name of the job of which the server
 126         is a member. Defaults to the value in `server_or_cluster_def`, if
 127         specified.
 128       task_index: (Optional.) Specifies the task index of the server in its
 129         job. Defaults to the value in `server_or_cluster_def`, if specified.
 130         Otherwise defaults to 0 if the server's job has only one task.
 131       protocol: (Optional.) Specifies the protocol to be used by the server.
 132         Acceptable values include `"grpc", "grpc+verbs"`. Defaults to the
 133         value in `server_or_cluster_def`, if specified. Otherwise defaults to
 134         `"grpc"`.
 135       config: (Options.) A `tf.ConfigProto` that specifies default
 136         configuration options for all sessions that run on this server.
 137       start: (Optional.) Boolean, indicating whether to start the server
 138         after creating it. Defaults to `True`.
 139 
 140     Raises:
 141       tf.errors.OpError: Or one of its subclasses if an error occurs while
 142         creating the TensorFlow server.
 143     """
 144     self._server_def = _make_server_def(server_or_cluster_def,
 145                                         job_name, task_index, protocol, config)
 146     with errors.raise_exception_on_not_ok_status() as status:
 147       self._server = pywrap_tensorflow.PyServer_New(
 148           self._server_def.SerializeToString(), status)
 149     if start:
 150       self.start()
 151 
 152   def start(self):
 153     """Starts this server.
 154 
 155     Raises:
 156       tf.errors.OpError: Or one of its subclasses if an error occurs while
 157         starting the TensorFlow server.
 158     """
 159     with errors.raise_exception_on_not_ok_status() as status:
 160       pywrap_tensorflow.PyServer_Start(self._server, status)
 161 
 162   def join(self):
 163     """Blocks until the server has shut down.
 164 
 165     This method currently blocks forever.
 166 
 167     Raises:
 168       tf.errors.OpError: Or one of its subclasses if an error occurs while
 169         joining the TensorFlow server.
 170     """
 171     with errors.raise_exception_on_not_ok_status() as status:
 172       pywrap_tensorflow.PyServer_Join(self._server, status)
 173 
 174   @property
 175   def server_def(self):
 176     """Returns the `tf.train.ServerDef` for this server.
 177 
 178     Returns:
 179       A `tf.train.ServerDef` protocol buffer that describes the configuration
 180       of this server.
 181     """
 182     return self._server_def

 184   @property
 185   def target(self):
 186     """Returns the target for a `tf.Session` to connect to this server.
 187 
 188     To create a
 189     `tf.Session` that
 190     connects to this server, use the following snippet:
 191 
 192     ```python
 193     server = tf.train.Server(...)
 194     with tf.Session(server.target):
 195       # ...
 196     ```
 197 
 198     Returns:
 199       A string containing a session target for this server.

 201     return self._server.target()

 203   @staticmethod
 204   def create_local_server(config=None, start=True):
 205     """Creates a new single-process cluster running on the local host.
 206 
 207     This method is a convenience wrapper for creating a
 208     `tf.train.Server` with a `tf.train.ServerDef` that specifies a
 209     single-process cluster containing a single task in a job called
 210     `"local"`.
 211 
 212     Args:
 213       config: (Options.) A `tf.ConfigProto` that specifies default
 214         configuration options for all sessions that run on this server.
 215       start: (Optional.) Boolean, indicating whether to start the server after
 216         creating it. Defaults to `True`.
 217 
 218     Returns:
 219       A local `tf.train.Server`.
 220     """
 221     # Specifying port 0 means that the OS will choose a free port for the
 222     # server.
 223     return Server({"local": ["localhost:0"]}, protocol="grpc", config=config,
 224                   start=start)

 227 @tf_export("train.ClusterSpec")
 228 class ClusterSpec(object):
 229   """Represents a cluster as a set of "tasks", organized into "jobs".
 230 
 231   A `tf.train.ClusterSpec` represents the set of processes that
 232   participate in a distributed TensorFlow computation. Every
 233   `tf.train.Server` is constructed in a particular cluster.
 234 
 235   To create a cluster with two jobs and five tasks, you specify the
 236   mapping from job names to lists of network addresses (typically
 237   hostname-port pairs).
 238 
 239   ```python
 240   cluster = tf.train.ClusterSpec({"worker": ["worker0.example.com:2222",
 241                                              "worker1.example.com:2222",
 242                                              "worker2.example.com:2222"],
 243                                   "ps": ["ps0.example.com:2222",
 244                                          "ps1.example.com:2222"]})
 245   ```
 246 
 247   Each job may also be specified as a sparse mapping from task indices
 248   to network addresses. This enables a server to be configured without
 249   needing to know the identity of (for example) all other worker
 250   tasks:
 251 
 252   ```python
 253   cluster = tf.train.ClusterSpec({"worker": {1: "worker1.example.com:2222"},
 254                                   "ps": ["ps0.example.com:2222",
 255                                          "ps1.example.com:2222"]})
 256   ```
 257   """
 258 
 259   def __init__(self, cluster):
 260     """Creates a `ClusterSpec`.
 261 
 262     Args:
 263       cluster: A dictionary mapping one or more job names to (i) a
 264         list of network addresses, or (ii) a dictionary mapping integer
 265         task indices to network addresses; or a `tf.train.ClusterDef`
 266         protocol buffer.
 267 
 268     Raises:
 269       TypeError: If `cluster` is not a dictionary mapping strings to lists
 270         of strings, and not a `tf.train.ClusterDef` protobuf.
 271     """
 272     if isinstance(cluster, dict):
 273       self._cluster_spec = {}
 274       for job_name, tasks in cluster.items():
 275         if isinstance(tasks, (list, tuple)):
 276           job_tasks = {i: task for i, task in enumerate(tasks)}
 277         elif isinstance(tasks, dict):
 278           job_tasks = {i: task for i, task in tasks.items()}
 279         else:
 280           raise TypeError("The tasks for job %r must be a list or a dictionary "
 281                           "from integers to strings." % job_name)
 282         self._cluster_spec[job_name] = job_tasks
 283       self._make_cluster_def()
 284     elif isinstance(cluster, cluster_pb2.ClusterDef):
 285       self._cluster_def = cluster
 286       self._cluster_spec = {}
 287       for job_def in self._cluster_def.job:
 288         self._cluster_spec[job_def.name] = {
 289             i: t for i, t in job_def.tasks.items()}
 290     elif isinstance(cluster, ClusterSpec):
 291       self._cluster_def = cluster_pb2.ClusterDef()
 292       self._cluster_def.MergeFrom(cluster.as_cluster_def())
 293       self._cluster_spec = {}
 294       for job_def in self._cluster_def.job:
 295         self._cluster_spec[job_def.name] = {
 296             i: t for i, t in job_def.tasks.items()}
 297     else:
 298       raise TypeError("`cluster` must be a dictionary mapping one or more "
 299                       "job names to lists of network addresses, or a "
 300                       "`ClusterDef` protocol buffer")
 301 
 302   def __nonzero__(self):
 303     return bool(self._cluster_spec)

 306   __bool__ = __nonzero__

 309     return self._cluster_spec == other

 312     return self._cluster_spec != other

 314   def __str__(self):
 315     key_values = self.as_dict()
 316     string_items = [
 317         repr(k) + ": " + repr(key_values[k]) for k in sorted(key_values)]
 318     return "ClusterSpec({" + ", ".join(string_items) + "})"
 319 
 320   def as_dict(self):
 321     """Returns a dictionary from job names to their tasks.
 322 
 323     For each job, if the task index space is dense, the corresponding
 324     value will be a list of network addresses; otherwise it will be a
 325     dictionary mapping (sparse) task indices to the corresponding
 326     addresses.
 327 
 328     Returns:
 329       A dictionary mapping job names to lists or dictionaries
 330       describing the tasks in those jobs.
 331     """
 332     ret = {}
 333     for job in self.jobs:
 334       task_indices = self.task_indices(job)
 335       if max(task_indices) + 1 == len(task_indices):
 336         # Return a list because the task indices are dense. This
 337         # matches the behavior of `as_dict()` before support for
 338         # sparse jobs was added.
 339         ret[job] = self.job_tasks(job)
 340       else:
 341         ret[job] = {i: self.task_address(job, i) for i in task_indices}
 342     return ret

 344   def as_cluster_def(self):
 345     """Returns a `tf.train.ClusterDef` protocol buffer based on this cluster."""
 346     return self._cluster_def

 348   @property
 349   def jobs(self):

 357   def num_tasks(self, job_name):
 358     """Returns the number of tasks defined in the given job.
 359 
 360     Args:
 361       job_name: The string name of a job in this cluster.
 362 
 363     Returns:
 364       The number of tasks defined in the given job.
 365 
 366     Raises:
 367       ValueError: If `job_name` does not name a job in this cluster.
 368     """
 369     try:
 370       job = self._cluster_spec[job_name]
 371     except KeyError:
 372       raise ValueError("No such job in cluster: %r" % job_name)
 373     return len(job)
 374 
 375   def task_indices(self, job_name):
 376     """Returns a list of valid task indices in the given job.
 377 
 378     Args:
 379       job_name: The string name of a job in this cluster.
 380 
 381     Returns:
 382       A list of valid task indices in the given job.
 383 
 384     Raises:
 385       ValueError: If `job_name` does not name a job in this cluster,
 386       or no task with index `task_index` is defined in that job.
 387     """
 388     try:
 389       job = self._cluster_spec[job_name]
 390     except KeyError:
 391       raise ValueError("No such job in cluster: %r" % job_name)
 392     return list(sorted(job.keys()))
 393 
 394   def task_address(self, job_name, task_index):
 395     """Returns the address of the given task in the given job.
 396 
 397     Args:
 398       job_name: The string name of a job in this cluster.
 399       task_index: A non-negative integer.
 400 
 401     Returns:
 402       The address of the given task in the given job.
 403 
 404     Raises:
 405       ValueError: If `job_name` does not name a job in this cluster,
 406       or no task with index `task_index` is defined in that job.
 407     """
 408     try:
 409       job = self._cluster_spec[job_name]
 410     except KeyError:
 411       raise ValueError("No such job in cluster: %r" % job_name)
 412     try:
 413       return job[task_index]
 414     except KeyError:
 415       raise ValueError("No task with index %r in job %r"
 416                        % (task_index, job_name))
 417 
 418   def job_tasks(self, job_name):
 419     """Returns a mapping from task ID to address in the given job.
 420 
 421     NOTE: For backwards compatibility, this method returns a list. If
 422     the given job was defined with a sparse set of task indices, the
 423     length of this list may not reflect the number of tasks defined in
 424     this job. Use the `tf.train.ClusterSpec.num_tasks` method
 425     to find the number of tasks defined in a particular job.
 426 
 427     Args:
 428       job_name: The string name of a job in this cluster.
 429 
 430     Returns:
 431       A list of task addresses, where the index in the list
 432       corresponds to the task index of each task. The list may contain
 433       `None` if the job was defined with a sparse set of task indices.
 434 
 435     Raises:
 436       ValueError: If `job_name` does not name a job in this cluster.
 437     """
 438     try:
 439       job = self._cluster_spec[job_name]
 440     except KeyError:
 441       raise ValueError("No such job in cluster: %r" % job_name)
 442     ret = [None for _ in range(max(job.keys()) + 1)]
 443     for i, task in job.items():
 444       ret[i] = task
 445     return ret

 447   def _make_cluster_def(self):
 448     """Creates a `tf.train.ClusterDef` based on the given `cluster_spec`.
 449 
 450     Raises:
 451       TypeError: If `cluster_spec` is not a dictionary mapping strings to lists
 452         of strings.
 453     """
 454     self._cluster_def = cluster_pb2.ClusterDef()
 455 
 456     # NOTE(mrry): Sort by job_name to produce deterministic protobufs.
 457     for job_name, tasks in sorted(self._cluster_spec.items()):
 458       try:
 459         job_name = compat.as_bytes(job_name)
 460       except TypeError:
 461         raise TypeError("Job name %r must be bytes or unicode" % job_name)
 462 
 463       job_def = self._cluster_def.job.add()
 464       job_def.name = job_name
 465 
 466       for i, task_address in sorted(tasks.items()):
 467         try:
 468           task_address = compat.as_bytes(task_address)
 469         except TypeError:
 470           raise TypeError(
 471               "Task address %r must be bytes or unicode" % task_address)
 472         job_def.tasks[i] = task_address
