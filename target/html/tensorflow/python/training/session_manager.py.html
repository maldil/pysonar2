  32 def _maybe_name(obj):
  33   """Returns object name if it has one, or a message otherwise.
  34 
  35   This is useful for names that apper in error messages.
  36   Args:
  37     obj: Object to get the name of.
  38   Returns:
  39     name, "None", or a "no name" message.
  40   """
  41   if obj is None:
  42     return "None"
  43   elif hasattr(obj, "name"):
  44     return obj.name
  45   else:
  46     return "<no name for %s>" % type(obj)

  49 @tf_export("train.SessionManager")
  50 class SessionManager(object):
  51   """Training helper that restores from checkpoint and creates session.
  52 
  53   This class is a small wrapper that takes care of session creation and
  54   checkpoint recovery. It also provides functions that to facilitate
  55   coordination among multiple training threads or processes.
  56 
  57   * Checkpointing trained variables as the training progresses.
  58   * Initializing variables on startup, restoring them from the most recent
  59     checkpoint after a crash, or wait for checkpoints to become available.
  60 
  61   ### Usage:
  62 
  63   ```python
  64   with tf.Graph().as_default():
  65      ...add operations to the graph...
  66     # Create a SessionManager that will checkpoint the model in '/tmp/mydir'.
  67     sm = SessionManager()
  68     sess = sm.prepare_session(master, init_op, saver, checkpoint_dir)
  69     # Use the session to train the graph.
  70     while True:
  71       sess.run(<my_train_op>)
  72   ```
  73 
  74   `prepare_session()` initializes or restores a model. It requires `init_op`
  75   and `saver` as an argument.
  76 
  77   A second process could wait for the model to be ready by doing the following:
  78 
  79   ```python
  80   with tf.Graph().as_default():
  81      ...add operations to the graph...
  82     # Create a SessionManager that will wait for the model to become ready.
  83     sm = SessionManager()
  84     sess = sm.wait_for_session(master)
  85     # Use the session to train the graph.
  86     while True:
  87       sess.run(<my_train_op>)
  88   ```
  89 
  90   `wait_for_session()` waits for a model to be initialized by other processes.
  91 
  92   """
  93 
  94   def __init__(self,
  95                local_init_op=None,
  96                ready_op=None,
  97                ready_for_local_init_op=None,
  98                graph=None,
  99                recovery_wait_secs=30,
 100                local_init_run_options=None):
 101     """Creates a SessionManager.
 102 
 103     The `local_init_op` is an `Operation` that is run always after a new session
 104     was created. If `None`, this step is skipped.
 105 
 106     The `ready_op` is an `Operation` used to check if the model is ready.  The
 107     model is considered ready if that operation returns an empty 1D string
 108     tensor. If the operation returns a non empty 1D string tensor, the elements
 109     are concatenated and used to indicate to the user why the model is not
 110     ready.
 111 
 112     The `ready_for_local_init_op` is an `Operation` used to check if the model
 113     is ready to run local_init_op.  The model is considered ready if that
 114     operation returns an empty 1D string tensor. If the operation returns a non
 115     empty 1D string tensor, the elements are concatenated and used to indicate
 116     to the user why the model is not ready.
 117 
 118     If `ready_op` is `None`, the model is not checked for readiness.
 119 
 120     `recovery_wait_secs` is the number of seconds between checks that
 121     the model is ready.  It is used by processes to wait for a model to
 122     be initialized or restored.  Defaults to 30 seconds.
 123 
 124     Args:
 125       local_init_op: An `Operation` run immediately after session creation.
 126          Usually used to initialize tables and local variables.
 127       ready_op: An `Operation` to check if the model is initialized.
 128       ready_for_local_init_op: An `Operation` to check if the model is ready
 129          to run local_init_op.
 130       graph: The `Graph` that the model will use.
 131       recovery_wait_secs: Seconds between checks for the model to be ready.
 132       local_init_run_options: RunOptions to be passed to session.run when
 133         executing the local_init_op.
 134 
 135     Raises:
 136       ValueError: If ready_for_local_init_op is not None but local_init_op is
 137         None
 138     """
 139     # Sets default values of arguments.
 140     if graph is None:
 141       graph = ops.get_default_graph()
 142     self._local_init_op = local_init_op
 143     self._ready_op = ready_op
 144     self._ready_for_local_init_op = ready_for_local_init_op
 145     self._graph = graph
 146     self._recovery_wait_secs = recovery_wait_secs
 147     self._target = None
 148     self._local_init_run_options = local_init_run_options
 149     if ready_for_local_init_op is not None and local_init_op is None:
 150       raise ValueError("If you pass a ready_for_local_init_op "
 151                        "you must also pass a local_init_op "
 152                        ", ready_for_local_init_op [%s]" %
 153                        ready_for_local_init_op)
 154 
 155   def _restore_checkpoint(self,
 156                           master,
 157                           saver=None,
 158                           checkpoint_dir=None,
 159                           checkpoint_filename_with_path=None,
 160                           wait_for_checkpoint=False,
 161                           max_wait_secs=7200,
 162                           config=None):
 163     """Creates a `Session`, and tries to restore a checkpoint.
 164 
 165 
 166     Args:
 167       master: `String` representation of the TensorFlow master to use.
 168       saver: A `Saver` object used to restore a model.
 169       checkpoint_dir: Path to the checkpoint files. The latest checkpoint in the
 170         dir will be used to restore.
 171       checkpoint_filename_with_path: Full file name path to the checkpoint file.
 172       wait_for_checkpoint: Whether to wait for checkpoint to become available.
 173       max_wait_secs: Maximum time to wait for checkpoints to become available.
 174       config: Optional `ConfigProto` proto used to configure the session.
 175 
 176     Returns:
 177       A pair (sess, is_restored) where 'is_restored' is `True` if
 178       the session could be restored, `False` otherwise.
 179 
 180     Raises:
 181       ValueError: If both checkpoint_dir and checkpoint_filename_with_path are
 182         set.
 183     """
 184     self._target = master
 185     sess = session.Session(self._target, graph=self._graph, config=config)
 186     # TODO(jhseu): Delete once tpu.initialize_system() goes away.
 187     initialize_ops = (
 188         distribution_strategy_context.get_distribution_strategy().initialize()
 189     )
 190     if initialize_ops:
 191       sess.run(initialize_ops)
 192 
 193     if checkpoint_dir and checkpoint_filename_with_path:
 194       raise ValueError("Can not provide both checkpoint_dir and "
 195                        "checkpoint_filename_with_path.")
 196     # If either saver or checkpoint_* is not specified, cannot restore. Just
 197     # return.
 198     if not saver or not (checkpoint_dir or checkpoint_filename_with_path):
 199       return sess, False
 200 
 201     if checkpoint_filename_with_path:
 202       saver.restore(sess, checkpoint_filename_with_path)
 203       return sess, True
 204 
 205     # Waits up until max_wait_secs for checkpoint to become available.
 206     wait_time = 0
 207     ckpt = checkpoint_management.get_checkpoint_state(checkpoint_dir)
 208     while not ckpt or not ckpt.model_checkpoint_path:
 209       if wait_for_checkpoint and wait_time < max_wait_secs:
 210         logging.info("Waiting for checkpoint to be available.")
 211         time.sleep(self._recovery_wait_secs)
 212         wait_time += self._recovery_wait_secs
 213         ckpt = checkpoint_management.get_checkpoint_state(checkpoint_dir)
 214       else:
 215         return sess, False
 216 
 217     # Loads the checkpoint.
 218     saver.restore(sess, ckpt.model_checkpoint_path)
 219     saver.recover_last_checkpoints(ckpt.all_model_checkpoint_paths)
 220     return sess, True
 221 
 222   def prepare_session(self,
 223                       master,
 224                       init_op=None,
 225                       saver=None,
 226                       checkpoint_dir=None,
 227                       checkpoint_filename_with_path=None,
 228                       wait_for_checkpoint=False,
 229                       max_wait_secs=7200,
 230                       config=None,
 231                       init_feed_dict=None,
 232                       init_fn=None):
 233     """Creates a `Session`. Makes sure the model is ready to be used.
 234 
 235     Creates a `Session` on 'master'. If a `saver` object is passed in, and
 236     `checkpoint_dir` points to a directory containing valid checkpoint
 237     files, then it will try to recover the model from checkpoint. If
 238     no checkpoint files are available, and `wait_for_checkpoint` is
 239     `True`, then the process would check every `recovery_wait_secs`,
 240     up to `max_wait_secs`, for recovery to succeed.
 241 
 242     If the model cannot be recovered successfully then it is initialized by
 243     running the `init_op` and calling `init_fn` if they are provided.
 244     The `local_init_op` is also run after init_op and init_fn, regardless of
 245     whether the model was recovered successfully, but only if
 246     `ready_for_local_init_op` passes.
 247 
 248     If the model is recovered from a checkpoint it is assumed that all
 249     global variables have been initialized, in particular neither `init_op`
 250     nor `init_fn` will be executed.
 251 
 252     It is an error if the model cannot be recovered and no `init_op`
 253     or `init_fn` or `local_init_op` are passed.
 254 
 255     Args:
 256       master: `String` representation of the TensorFlow master to use.
 257       init_op: Optional `Operation` used to initialize the model.
 258       saver: A `Saver` object used to restore a model.
 259       checkpoint_dir: Path to the checkpoint files. The latest checkpoint in the
 260         dir will be used to restore.
 261       checkpoint_filename_with_path: Full file name path to the checkpoint file.
 262       wait_for_checkpoint: Whether to wait for checkpoint to become available.
 263       max_wait_secs: Maximum time to wait for checkpoints to become available.
 264       config: Optional `ConfigProto` proto used to configure the session.
 265       init_feed_dict: Optional dictionary that maps `Tensor` objects to feed
 266         values.  This feed dictionary is passed to the session `run()` call when
 267         running the init op.
 268       init_fn: Optional callable used to initialize the model. Called after the
 269         optional `init_op` is called.  The callable must accept one argument,
 270         the session being initialized.
 271 
 272     Returns:
 273       A `Session` object that can be used to drive the model.
 274 
 275     Raises:
 276       RuntimeError: If the model cannot be initialized or recovered.
 277       ValueError: If both checkpoint_dir and checkpoint_filename_with_path are
 278         set.
 279     """
 280 
 281     sess, is_loaded_from_checkpoint = self._restore_checkpoint(
 282         master,
 283         saver,
 284         checkpoint_dir=checkpoint_dir,
 285         checkpoint_filename_with_path=checkpoint_filename_with_path,
 286         wait_for_checkpoint=wait_for_checkpoint,
 287         max_wait_secs=max_wait_secs,
 288         config=config)
 289     if not is_loaded_from_checkpoint:
 290       if init_op is None and not init_fn and self._local_init_op is None:
 291         raise RuntimeError("Model is not initialized and no init_op or "
 292                            "init_fn or local_init_op was given")
 293       if init_op is not None:
 294         sess.run(init_op, feed_dict=init_feed_dict)
 295       if init_fn:
 296         init_fn(sess)
 297 
 298     local_init_success, msg = self._try_run_local_init_op(sess)
 299     if not local_init_success:
 300       raise RuntimeError(
 301           "Init operations did not make model ready for local_init.  "
 302           "Init op: %s, init fn: %s, error: %s" % (_maybe_name(init_op),
 303                                                    init_fn,
 304                                                    msg))
 305 
 306     is_ready, msg = self._model_ready(sess)
 307     if not is_ready:
 308       raise RuntimeError(
 309           "Init operations did not make model ready.  "
 310           "Init op: %s, init fn: %s, local_init_op: %s, error: %s" %
 311           (_maybe_name(init_op), init_fn, self._local_init_op, msg))
 312     return sess
 313 
 314   def recover_session(self,
 315                       master,
 316                       saver=None,
 317                       checkpoint_dir=None,
 318                       checkpoint_filename_with_path=None,
 319                       wait_for_checkpoint=False,
 320                       max_wait_secs=7200,
 321                       config=None):
 322     """Creates a `Session`, recovering if possible.
 323 
 324     Creates a new session on 'master'.  If the session is not initialized
 325     and can be recovered from a checkpoint, recover it.
 326 
 327     Args:
 328       master: `String` representation of the TensorFlow master to use.
 329       saver: A `Saver` object used to restore a model.
 330       checkpoint_dir: Path to the checkpoint files. The latest checkpoint in the
 331         dir will be used to restore.
 332       checkpoint_filename_with_path: Full file name path to the checkpoint file.
 333       wait_for_checkpoint: Whether to wait for checkpoint to become available.
 334       max_wait_secs: Maximum time to wait for checkpoints to become available.
 335       config: Optional `ConfigProto` proto used to configure the session.
 336 
 337     Returns:
 338       A pair (sess, initialized) where 'initialized' is `True` if
 339       the session could be recovered and initialized, `False` otherwise.
 340 
 341     Raises:
 342       ValueError: If both checkpoint_dir and checkpoint_filename_with_path are
 343         set.
 344     """
 345 
 346     sess, is_loaded_from_checkpoint = self._restore_checkpoint(
 347         master,
 348         saver,
 349         checkpoint_dir=checkpoint_dir,
 350         checkpoint_filename_with_path=checkpoint_filename_with_path,
 351         wait_for_checkpoint=wait_for_checkpoint,
 352         max_wait_secs=max_wait_secs,
 353         config=config)
 354 
 355     # Always try to run local_init_op
 356     local_init_success, msg = self._try_run_local_init_op(sess)
 357 
 358     if not is_loaded_from_checkpoint:
 359       # Do not need to run checks for readiness
 360       return sess, False
 361 
 362     restoring_file = checkpoint_dir or checkpoint_filename_with_path
 363     if not local_init_success:
 364       logging.info(
 365           "Restoring model from %s did not make model ready for local init:"
 366           " %s", restoring_file, msg)
 367       return sess, False
 368 
 369     is_ready, msg = self._model_ready(sess)
 370     if not is_ready:
 371       logging.info("Restoring model from %s did not make model ready: %s",
 372                    restoring_file, msg)
 373       return sess, False
 374 
 375     logging.info("Restored model from %s", restoring_file)
 376     return sess, is_loaded_from_checkpoint
 377 
 378   def wait_for_session(self, master, config=None, max_wait_secs=float("Inf")):
 379     """Creates a new `Session` and waits for model to be ready.
 380 
 381     Creates a new `Session` on 'master'.  Waits for the model to be
 382     initialized or recovered from a checkpoint.  It's expected that
 383     another thread or process will make the model ready, and that this
 384     is intended to be used by threads/processes that participate in a
 385     distributed training configuration where a different thread/process
 386     is responsible for initializing or recovering the model being trained.
 387 
 388     NB: The amount of time this method waits for the session is bounded
 389     by max_wait_secs. By default, this function will wait indefinitely.
 390 
 391     Args:
 392       master: `String` representation of the TensorFlow master to use.
 393       config: Optional ConfigProto proto used to configure the session.
 394       max_wait_secs: Maximum time to wait for the session to become available.
 395 
 396     Returns:
 397       A `Session`. May be None if the operation exceeds the timeout
 398       specified by config.operation_timeout_in_ms.
 399 
 400     Raises:
 401       tf.DeadlineExceededError: if the session is not available after
 402         max_wait_secs.
 403     """
 404     self._target = master
 405 
 406     if max_wait_secs is None:
 407       max_wait_secs = float("Inf")
 408     timer = _CountDownTimer(max_wait_secs)
 409 
 410     while True:
 411       sess = session.Session(self._target, graph=self._graph, config=config)
 412       not_ready_msg = None
 413       not_ready_local_msg = None
 414       local_init_success, not_ready_local_msg = self._try_run_local_init_op(
 415           sess)
 416       if local_init_success:
 417         # Successful if local_init_op is None, or ready_for_local_init_op passes
 418         is_ready, not_ready_msg = self._model_ready(sess)
 419         if is_ready:
 420           return sess
 421 
 422       self._safe_close(sess)
 423 
 424       # Do we have enough time left to try again?
 425       remaining_ms_after_wait = (
 426           timer.secs_remaining() - self._recovery_wait_secs)
 427       if remaining_ms_after_wait < 0:
 428         raise errors.DeadlineExceededError(
 429             None, None,
 430             "Session was not ready after waiting %d secs." % (max_wait_secs,))
 431 
 432       logging.info("Waiting for model to be ready.  "
 433                    "Ready_for_local_init_op:  %s, ready: %s",
 434                    not_ready_local_msg, not_ready_msg)
 435       time.sleep(self._recovery_wait_secs)
 436 
 437   def _safe_close(self, sess):
 438     """Closes a session without raising an exception.
 439 
 440     Just like sess.close() but ignores exceptions.
 441 
 442     Args:
 443       sess: A `Session`.
 444     """
 445     # pylint: disable=broad-except
 446     try:
 447       sess.close()
 448     except Exception:
 449       # Intentionally not logging to avoid user complaints that
 450       # they get cryptic errors.  We really do not care that Close
 451       # fails.
 452       pass
 453     # pylint: enable=broad-except
 454 
 455   def _model_ready(self, sess):
 456     """Checks if the model is ready or not.
 457 
 458     Args:
 459       sess: A `Session`.
 460 
 461     Returns:
 462       A tuple (is_ready, msg), where is_ready is True if ready and False
 463       otherwise, and msg is `None` if the model is ready, a `String` with the
 464       reason why it is not ready otherwise.
 465     """
 466     return _ready(self._ready_op, sess, "Model not ready")
 467 
 468   def _model_ready_for_local_init(self, sess):
 469     """Checks if the model is ready to run local_init_op.
 470 
 471     Args:
 472       sess: A `Session`.
 473 
 474     Returns:
 475       A tuple (is_ready, msg), where is_ready is True if ready to run
 476       local_init_op and False otherwise, and msg is `None` if the model is
 477       ready to run local_init_op, a `String` with the reason why it is not ready
 478       otherwise.
 479     """
 480     return _ready(self._ready_for_local_init_op, sess,
 481                   "Model not ready for local init")
 482 
 483   def _try_run_local_init_op(self, sess):
 484     """Tries to run _local_init_op, if not None, and is ready for local init.
 485 
 486     Args:
 487       sess: A `Session`.
 488 
 489     Returns:
 490       A tuple (is_successful, msg), where is_successful is True if
 491       _local_init_op is None, or we ran _local_init_op, and False otherwise;
 492       and msg is a `String` with the reason why the model was not ready to run
 493       local init.
 494     """
 495     if self._local_init_op is not None:
 496       is_ready_for_local_init, msg = self._model_ready_for_local_init(sess)
 497       if is_ready_for_local_init:
 498         logging.info("Running local_init_op.")
 499         sess.run(self._local_init_op, options=self._local_init_run_options)
 500         logging.info("Done running local_init_op.")
 501         return True, None
 502       else:
 503         return False, msg
 504     return True, None

 507 def _ready(op, sess, msg):
 508   """Checks if the model is ready or not, as determined by op.
 509 
 510   Args:
 511     op: An op, either _ready_op or _ready_for_local_init_op, which defines the
 512       readiness of the model.
 513     sess: A `Session`.
 514     msg: A message to log to warning if not ready
 515 
 516   Returns:
 517     A tuple (is_ready, msg), where is_ready is True if ready and False
 518     otherwise, and msg is `None` if the model is ready, a `String` with the
 519     reason why it is not ready otherwise.
 520   """
 521   if op is None:
 522     return True, None
 523   else:
 524     try:
 525       ready_value = sess.run(op)
 526       # The model is considered ready if ready_op returns an empty 1-D tensor.
 527       # Also compare to `None` and dtype being int32 for backward
 528       # compatibility.
 529       if (ready_value is None or ready_value.dtype == np.int32 or
 530           ready_value.size == 0):
 531         return True, None
 532       else:
 533         # TODO(sherrym): If a custom ready_op returns other types of tensor,
 534         # or strings other than variable names, this message could be
 535         # confusing.
 536         non_initialized_varnames = ", ".join(
 537             [i.decode("utf-8") for i in ready_value])
 538         return False, "Variables not initialized: " + non_initialized_varnames
 539     except errors.FailedPreconditionError as e:
 540       if "uninitialized" not in str(e):
 541         logging.warning("%s : error [%s]", msg, str(e))
 542         raise e
 543       return False, str(e)

 546 class _CountDownTimer(object):
 547 
 548   def __init__(self, duration_secs):
 549     self._start_time_secs = time.time()
 550     self._duration_secs = duration_secs
 551 
 552   def secs_remaining(self):
 553     diff = self._duration_secs - (time.time() - self._start_time_secs)
 554     return max(0, diff)
