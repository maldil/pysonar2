  25 # There is a circular dependency between this and `distribute` module. So we
  26 # load it lazily to workaround this.
  27 distribute_lib = LazyLoader(
  28     "distribute_lib", globals(),

  31 # ------------------------------------------------------------------------------
  32 # Internal API for setting the current thread mode as being either in a
  33 # tower or cross-tower context for a particular distribution strategy.
  34 
  35 
  36 class _ThreadMode(object):
  37 
  38   def __init__(self, dist, cross, tower):
  39     self.distribution_strategy = dist
  40     self.cross_tower_context = cross
  41     self.tower_context = tower
  42 
  43 
  44 class _CrossTowerThreadMode(_ThreadMode):
  45 
  46   def __init__(self, distribution_strategy):
  47     _ThreadMode.__init__(
  48         self, distribution_strategy, distribution_strategy, None)
  49 
  50 
  51 class _InTowerThreadMode(_ThreadMode):
  52 
  53   def __init__(self, tower_ctx):
  54     _ThreadMode.__init__(
  55         self, tower_ctx.distribution_strategy, None, tower_ctx)
  56 
  57 
  58 def _push_per_thread_mode(context):
  59   ops.get_default_graph()._distribution_strategy_stack.append(context)  # pylint: disable=protected-access
  60 
  61 
  62 def _pop_per_thread_mode():
  63   ops.get_default_graph()._distribution_strategy_stack.pop(-1)  # pylint: disable=protected-access
  64 
  65 
  66 class _DefaultTowerThreadMode(_ThreadMode):
  67   """Type of default value returned by `_get_per_thread_mode()`.
  68 
  69   Used when the thread-local stack is empty.
  70   """
  71 
  72   def __init__(self):
  73     _ThreadMode.__init__(self, _get_default_distribution_strategy(), None,
  74                          _get_default_tower_context())
  75 
  76 
  77 def _get_per_thread_mode():
  78   try:
  79     return ops.get_default_graph()._distribution_strategy_stack[-1]  # pylint: disable=protected-access
  80   except (AttributeError, IndexError):
  81     return _get_default_tower_mode()
  82

  84 # ------------------------------------------------------------------------------
  85 # Public API for accessing the current thread mode

  88 def get_tower_context():
  89   """Returns the current TowerContext or None if in a cross-tower context.
  90 
  91   Note that execution:
  92 
  93   1. starts in the default (single-tower) tower context (this function
  94      will return the default TowerContext object);
  95   2. switches to cross-tower context (in which case this will return
  96      None) when entering a `with DistributionStrategy.scope():` block;
  97   3. switches to a (non-default) tower context inside
  98      `call_for_each_tower(fn, ...)`;
  99   4. if `fn` calls `get_tower_context()->merge_call(merge_fn, ...)`, then
 100      inside `merge_fn` you are back in the cross-tower context (and again
 101      this function will return None).
 102 
 103   Note that you can also go directly from step 1 to 4 to switch to a
 104   cross-tower context for the default `DistributionStrategy`. You may
 105   also switch from the cross-tower context of 4 to a tower context by
 106   calling `call_for_each_tower()`, jumping back to step 3.
 107 
 108   Most `DistributionStrategy` methods may only be executed in
 109   a cross-tower context, in a tower context you should use the
 110   `TowerContext` API instead.
 111 
 112   Returns:
 113     The current `TowerContext` object when in a tower context scope, else None.
 114 
 115     Exactly one of `get_tower_context()` and `get_cross_tower_context()`
 116     will return None in a particular block.
 117   """
 118   return _get_per_thread_mode().tower_context
 119 
 120 
 121 def get_cross_tower_context():
 122   """Returns the current DistributionStrategy if in a cross-tower context.
 123 
 124   Note that execution:
 125 
 126   1. starts in the default (single-tower) tower context;
 127   2. switches to cross-tower context when entering a
 128      `with DistributionStrategy.scope():` block;
 129   3. switches to a (non-default) tower context inside
 130      `call_for_each_tower(fn, ...)`;
 131   4. if `fn` calls `get_tower_context()->merge_call(merge_fn, ...)`, then
 132      inside `merge_fn` you are back in the cross-tower context.
 133 
 134   Note that you can also go directly from step 1 to 4 to switch to a
 135   cross-tower context for the default `DistributionStrategy`. You may
 136   also switch from the cross-tower context of 4 to a tower context by
 137   calling `call_for_each_tower()`, jumping back to step 3.
 138 
 139   Most `DistributionStrategy` methods may only be executed in
 140   a cross-tower context.
 141 
 142   Returns:
 143     Returns the current `DistributionStrategy` object in a cross-tower
 144     context, or None.
 145 
 146     Exactly one of `get_tower_context()` and `get_cross_tower_context()`
 147     will return None in a particular block.
 148   """
 149   return _get_per_thread_mode().cross_tower_context
 150 
 151 
 152 def get_distribution_strategy():
 153   """Returns the current `DistributionStrategy` object.
 154 
 155   Prefer to use `get_tower_context()` or `get_cross_tower_context()`
 156   instead when possible.
 157 
 158   Returns:
 159     A `DistributionStrategy` object. Inside a
 160     `with distribution_strategy.scope()` block, it returns
 161     `distribution_strategy`, otherwise it returns the default
 162     (single-tower) `DistributionStrategy` object.
 163   """
 164   return _get_per_thread_mode().distribution_strategy
 165 
 166 
 167 def has_distribution_strategy():
 168   """Return if there is a current non-default `DistributionStrategy`.
 169 
 170   Returns:
 171     True if inside a `with distribution_strategy.scope():`.
 172   """
 173   return get_distribution_strategy() is not _get_default_distribution_strategy()
 174

 176 # ------------------------------------------------------------------------------
 177 # Defaults that are used when no distribution strategy is explicitly created.
 178 # We create them lazily in a function so that we can workaround the circular
 179 # dependency on distribute_lib. See lazy loader at the top of this file.
 180 
 181 _defaults = {
 182     "distribution_strategy": None,
 183     "tower_context": None,
 184     "tower_mode": None
 185 }
 186 
 187 
 188 def _get_default_distribution_strategy():
 189   if _defaults["distribution_strategy"] is None:
 190     _defaults["distribution_strategy"] = (
 191         distribute_lib._DefaultDistributionStrategy())  # pylint: disable=protected-access
 192   return _defaults["distribution_strategy"]
 193 
 194 
 195 def _get_default_tower_context():
 196   if _defaults["tower_context"] is None:
 197     _defaults["tower_context"] = distribute_lib.TowerContext(
 198         _get_default_distribution_strategy(), tower_id=0)
 199   return _defaults["tower_context"]
 200 
 201 
 202 def _get_default_tower_mode():
 203   if _defaults["tower_mode"] is None:
 204     _defaults["tower_mode"] = _DefaultTowerThreadMode()
 205   return _defaults["tower_mode"]
