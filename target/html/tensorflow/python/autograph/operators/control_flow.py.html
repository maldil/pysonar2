  29 def for_stmt(iter_, extra_test, body, init_state):
  30   """Functional form of a for statement.
  31 
  32   The loop operates on a state, which includes all symbols that are
  33   variant across loop iterations, excluding the iterate as well as the
  34   variables local to the loop.
  35 
  36   For example, given the loop below that calculates the geometric and
  37   arithmetic means or some numbers:
  38 
  39     geo_mean = 1
  40     arith_mean = 0
  41     for i in range(n):
  42       a = numbers[i]
  43       geo_mean *= a
  44       arith_mean += a
  45 
  46   The state is represented by the variables geo_mean and arith_mean. The
  47   argument for initial_state may contain the tuple (1, 0), the body will
  48   include the arguments geo_mean and arith_mean and will return a tuple
  49   representing the new values for geo_mean and respectively arith_mean.
  50 
  51   Args:
  52     iter_: The entity being iterated over.
  53     extra_test: Callable with the state as arguments, and boolean return type.
  54         An additional loop condition.
  55     body: Callable with the iterate and the state as arguments, and
  56         state as return type. The actual loop body.
  57     init_state: Tuple containing the initial state.
  58 
  59   Returns:
  60     Tuple containing the final state.
  61   """
  62   if tensor_util.is_tensor(iter_):
  63     return _known_len_for_stmt(iter_, extra_test, body, init_state)
  64   elif isinstance(iter_, dataset_ops.Dataset):
  65     return _dataset_for_stmt(iter_, extra_test, body, init_state)
  66   else:
  67     return _py_for_stmt(iter_, extra_test, body, init_state)
  68

  70 def _py_for_stmt(iter_, extra_test, body, init_state):
  71   """Overload of for_stmt that executes a Python for loop."""
  72   state = init_state
  73   for target in iter_:
  74     if not extra_test(*state):
  75       break
  76     state = body(target, *state)
  77 
  78   # TODO(mdan): Remove this special case.
  79   if len(state) == 1:
  80     return state[0]
  81   return state
  82

  84 def _known_len_for_stmt(iter_, extra_test, body, init_state):
  85   """Overload of for_stmt that iterates over objects that admit a length."""
  86   n = py_builtins.len_(iter_)
  87 
  88   def while_body(iterate_index, *state):
  89     iterate = iter_[iterate_index]
  90     new_state = body(iterate, *state)
  91     return (iterate_index + 1,) + new_state
  92 
  93   def while_cond(iterate_index, *state):
  94     return gen_math_ops.logical_and(iterate_index < n, extra_test(*state))
  95 
  96   results = while_stmt(
  97       while_cond,
  98       while_body,
  99       init_state=(0,) + init_state,
 100       extra_deps=(iter_,),
 101       opts=dict(maximum_iterations=n))
 102   # Dropping the iteration index because it's not syntactically visible.
 103   results = results[1:]
 104 
 105   # TODO(mdan): Remove this special case.
 106   if len(results) == 1:
 107     return results[0]
 108   return results

 111 def _dataset_for_stmt(ds, extra_test, body, init_state):
 112   """Overload of for_stmt that iterates over TF Datasets."""
 113   # Because Datsets only expose get_next, in the style of Python iterators,
 114   # we are forced to unpack the loop as:
 115   #
 116   # epoch_number, iterate = ds.get_next()
 117   # while epoch_number < 2:
 118   #   <body>
 119   #   epoch_number, iterate = ds.get_next()
 120   epoch_numbers = dataset_ops.Dataset.range(2)
 121   def tag_with(ds, tag):
 122     return dataset_ops.Dataset.zip(
 123         (dataset_ops.Dataset.from_tensors(tag).repeat(), ds))
 124   ds_with_epoch = epoch_numbers.flat_map(lambda i: tag_with(ds, i))
 125 
 126   iterator = ds_with_epoch.make_initializable_iterator()
 127   with ops.control_dependencies((iterator.initializer,)):
 128     epoch_number, iterate = iterator.get_next()
 129 
 130     def while_body(epoch_number, iterate, *state):
 131       new_state = body(iterate, *state)
 132       epoch_number, iterate = iterator.get_next()
 133       return (epoch_number, iterate) + new_state
 134 
 135     def while_cond(epoch_number, iterate, *state):
 136       del iterate
 137       return gen_math_ops.logical_and(epoch_number < 1, extra_test(*state))
 138 
 139     results = while_stmt(
 140         while_cond,
 141         while_body,
 142         init_state=(epoch_number, iterate) + init_state,
 143         extra_deps=())
 144   # Dropping the epoch number and iterate because they are not syntactically
 145   # visible.
 146   results = results[2:]
 147 
 148   # TODO(mdan): Remove this special case.
 149   if len(results) == 1:
 150     return results[0]
 151   return results

 154 def while_stmt(test, body, init_state, extra_deps, opts=None):
 155   """Functional form of a while statement.
 156 
 157   The loop operates on a so-called state, which includes all symbols that are
 158   variant across loop iterations. In what follows we refer to state as either
 159   a tuple of entities that represent an actual state, or a list of arguments
 160   of the corresponding types.
 161 
 162   Args:
 163     test: Callable with the state as arguments, and boolean return type.
 164         The loop condition.
 165     body: Callable with the state as arguments, and state as return type.
 166         The actual loop body.
 167     init_state: Tuple containing the initial state.
 168     extra_deps: Tuple containing additional entities on which the loop may
 169         depend, such as loop invariants referenced by test. Used
 170         exclusively for dispatch control.
 171     opts: Optional dict of extra loop parameters.
 172 
 173   Returns:
 174     Tuple containing the final state.
 175   """
 176   # TODO(mdan): Consider adding a generic mechanism for dynamic dispatch.
 177   # That could be something as simple as a collection of dispatch rules, with
 178   # some prioritization.
 179   if any(tensor_util.is_tensor(v) for v in init_state + extra_deps):
 180     return _tf_while_stmt(test, body, init_state, opts)
 181   else:
 182     return _py_while_stmt(test, body, init_state, opts)
 183 
 184 
 185 def _tf_while_stmt(test, body, init_state, opts):
 186   """Overload of while_stmt that stages a TF while_stmt."""
 187   if opts is None:
 188     opts = {}
 189   return control_flow_ops.while_loop(test, body, init_state, **opts)
 190 
 191 
 192 def _py_while_stmt(test, body, init_state, opts):
 193   """Overload of while_stmt that executes a Python while loop."""
 194   del opts
 195   state = init_state
 196   while test(*state):
 197     state = body(*state)
 198   return state

 201 def if_stmt(cond, body, orelse):
 202   """Functional form of an if statement.
 203 
 204   Args:
 205     cond: Boolean.
 206     body: Callable with no arguments, and outputs of the positive (if) branch
 207         as return type.
 208     orelse: Callable with no arguments, and outputs of the negative (else)
 209         branch as return type.
 210 
 211   Returns:
 212     Tuple containing the statement outputs.
 213   """
 214   if tensor_util.is_tensor(cond):
 215     return tf_if_stmt(cond, body, orelse)
 216   else:
 217     return _py_if_stmt(cond, body, orelse)
 218 
 219 
 220 def tf_if_stmt(cond, body, orelse):
 221   """Overload of if_stmt that stages a TF cond."""
 222   return control_flow_ops.cond(cond, body, orelse)
 223 
 224 
 225 def _py_if_stmt(cond, body, orelse):
 226   """Overload of if_stmt that executes a Python if statement."""
 227   return body() if cond else orelse()
