  36 def new_list(iterable=None):
  37   """The list constructor.
  38 
  39   Args:
  40     iterable: Optional elements to fill the list with.
  41 
  42   Returns:
  43     A list-like object. The exact return value depends on the initial elements.
  44   """
  45   if iterable:
  46     elements = tuple(iterable)
  47   else:
  48     elements = ()
  49 
  50   if elements:
  51     # When the list contains elements, it is assumed to be a "Python" lvalue
  52     # list.
  53     return _py_list_new(elements)
  54   return tf_tensor_list_new(elements)

  57 def tf_tensor_array_new(elements, element_dtype=None, element_shape=None):
  58   """Overload of new_list that stages a Tensor list creation."""
  59   elements = tuple(ops.convert_to_tensor(el) for el in elements)
  60 
  61   all_dtypes = set(el.dtype for el in elements)
  62   if len(all_dtypes) == 1:
  63     inferred_dtype, = tuple(all_dtypes)
  64     if element_dtype is not None and element_dtype != inferred_dtype:
  65       raise ValueError(
  66           'incompatible dtype; specified: {}, inferred from {}: {}'.format(
  67               element_dtype, elements, inferred_dtype))
  68   elif len(all_dtypes) > 1:
  69     raise ValueError(
  70         'TensorArray requires all elements to have the same dtype:'
  71         ' {}'.format(elements))
  72   else:
  73     if element_dtype is None:
  74       raise ValueError('dtype is required to create an empty TensorArray')
  75 
  76   all_shapes = set(tuple(el.shape.as_list()) for el in elements)
  77   if len(all_shapes) == 1:
  78     inferred_shape, = tuple(all_shapes)
  79     if element_shape is not None and element_shape != inferred_shape:
  80       raise ValueError(
  81           'incompatible shape; specified: {}, inferred from {}: {}'.format(
  82               element_shape, elements, inferred_shape))
  83   elif len(all_shapes) > 1:
  84     raise ValueError(
  85         'TensorArray requires all elements to have the same shape:'
  86         ' {}'.format(elements))
  87     # TODO(mdan): We may want to allow different shapes with infer_shape=False.
  88   else:
  89     inferred_shape = None
  90 
  91   if element_dtype is None:
  92     element_dtype = inferred_dtype
  93   if element_shape is None:
  94     element_shape = inferred_shape
  95 
  96   l = tensor_array_ops.TensorArray(
  97       dtype=element_dtype,
  98       size=len(elements),
  99       dynamic_size=True,
 100       infer_shape=(element_shape is None),
 101       element_shape=element_shape)
 102   for i, el in enumerate(elements):
 103     l = l.write(i, el)
 104   return l

 107 def tf_tensor_list_new(elements, element_dtype=None, element_shape=None):
 108   """Overload of new_list that stages a Tensor list creation."""
 109   elements = tuple(ops.convert_to_tensor(el) for el in elements)
 110 
 111   all_dtypes = set(el.dtype for el in elements)
 112   if len(all_dtypes) == 1:
 113     inferred_dtype = tuple(all_dtypes)[0]
 114     if element_dtype is not None and element_dtype != inferred_dtype:
 115       raise ValueError(
 116           'incompatible dtype; specified: {}, inferred from {}: {}'.format(
 117               element_dtype, elements, inferred_dtype))
 118   else:
 119     # Heterogeneous lists are ok.
 120     if element_dtype is not None:
 121       raise ValueError(
 122           'specified dtype {} is inconsistent with that of elements {}'.format(
 123               element_dtype, elements))
 124     inferred_dtype = dtypes.variant
 125 
 126   all_shapes = set(tuple(el.shape.as_list()) for el in elements)
 127   if len(all_shapes) == 1:
 128     inferred_shape = array_ops.shape(elements[0])
 129     if element_shape is not None and element_shape != inferred_shape:
 130       raise ValueError(
 131           'incompatible shape; specified: {}, inferred from {}: {}'.format(
 132               element_shape, elements, inferred_shape))
 133   else:
 134     # Heterogeneous lists are ok.
 135     if element_shape is not None:
 136       raise ValueError(
 137           'specified shape {} is inconsistent with that of elements {}'.format(
 138               element_shape, elements))
 139     inferred_shape = constant_op.constant(-1)  # unknown shape, by convention
 140 
 141   if element_dtype is None:
 142     element_dtype = inferred_dtype
 143   if element_shape is None:
 144     element_shape = inferred_shape
 145 
 146   l = list_ops.empty_tensor_list(
 147       element_shape=element_shape, element_dtype=element_dtype)
 148   for el in elements:
 149     l = list_ops.tensor_list_push_back(l, el)
 150   return l

 158 def list_append(list_, x):
 159   """The list append function.
 160 
 161   Note: it is unspecified where list_ will be mutated or not. If list_ is
 162   a TensorFlow entity, it will not be typically mutated. If list_ is a plain
 163   list, it will be. In general, if the list is mutated then the return value
 164   should point to the original entity.
 165 
 166   Args:
 167     list_: An entity that supports append semantics.
 168     x: The element to append.
 169 
 170   Returns:
 171     Same as list_, after the append was performed.
 172 
 173   Raises:
 174     ValueError: if list_ is not of a known list-like type.
 175   """
 176   if isinstance(list_, tensor_array_ops.TensorArray):
 177     return _tf_tensorarray_append(list_, x)
 178   elif tensor_util.is_tensor(list_):
 179     if list_.dtype == dtypes.variant:
 180       return _tf_tensor_list_append(list_, x)
 181     else:
 182       raise ValueError(
 183           'tensor lists are expected to be Tensors with dtype=tf.variant,'
 184           ' instead found %s' % list_)
 185   else:
 186     return _py_list_append(list_, x)
 187 
 188 
 189 def _tf_tensor_list_append(list_, x):
 190   """Overload of list_append that stages a Tensor list write."""
 191   def empty_list_of_elements_like_x():
 192     tensor_x = ops.convert_to_tensor(x)
 193     return list_ops.empty_tensor_list(
 194         element_shape=array_ops.shape(tensor_x),
 195         element_dtype=tensor_x.dtype)
 196 
 197   list_ = control_flow_ops.cond(
 198       list_ops.tensor_list_length(list_) > 0,
 199       lambda: list_,
 200       empty_list_of_elements_like_x,
 201   )
 202   return list_ops.tensor_list_push_back(list_, x)
 203 
 204 
 205 def _tf_tensorarray_append(list_, x):
 206   """Overload of list_append that stages a TensorArray write."""
 207   return list_.write(list_.size(), x)

 210 def _py_list_append(list_, x):
 211   """Overload of list_append that executes a Python list append."""
 212   # Revert to the original call.
 213   list_.append(x)
 214   return list_

 222 def list_pop(list_, i, opts):
 223   """The list pop function.
 224 
 225   Note: it is unspecified where list_ will be mutated or not. If list_ is
 226   a TensorFlow entity, it will not be typically mutated. If list_ is a plain
 227   list, it will be. In general, if the list is mutated then the return value
 228   should point to the original entity.
 229 
 230   Args:
 231     list_: An entity that supports pop semantics.
 232     i: Optional index to pop from. May be None.
 233     opts: A ListPopOpts.
 234 
 235   Returns:
 236     Tuple (x, out_list_):
 237       out_list_: same as list_, after the removal was performed.
 238       x: the removed element value.
 239 
 240   Raises:
 241     ValueError: if list_ is not of a known list-like type or the operation is
 242     not supported for that type.
 243   """
 244   assert isinstance(opts, ListPopOpts)
 245 
 246   if isinstance(list_, tensor_array_ops.TensorArray):
 247     raise ValueError('TensorArray does not support item removal')
 248   elif tensor_util.is_tensor(list_):
 249     if list_.dtype == dtypes.variant:
 250       return _tf_tensor_list_pop(list_, i, opts)
 251     else:
 252       raise ValueError(
 253           'tensor lists are expected to be Tensors with dtype=tf.variant,'
 254           ' instead found %s' % list_)
 255   else:
 256     return _py_list_pop(list_, i)
 257 
 258 
 259 def _tf_tensor_list_pop(list_, i, opts):
 260   """Overload of list_pop that stages a Tensor list pop."""
 261   if i is not None:
 262     raise NotImplementedError('tensor lists only support removing from the end')
 263 
 264   if opts.element_dtype is None:
 265     raise ValueError('cannot pop from a list without knowing its element '
 266                      'type; use set_element_type to annotate it')
 267   if opts.element_shape is None:
 268     raise ValueError('cannot pop from a list without knowing its element '
 269                      'shape; use set_element_type to annotate it')
 270   list_out, x = list_ops.tensor_list_pop_back(
 271       list_, element_dtype=opts.element_dtype)
 272   x.set_shape(opts.element_shape)
 273   return list_out, x

 276 def _py_list_pop(list_, i):
 277   """Overload of list_pop that executes a Python list append."""
 278   if i is None:
 279     x = list_.pop()
 280   else:
 281     x = list_.pop(i)
 282   return list_, x

 292 def list_stack(list_, opts):
 293   """The list stack function.
 294 
 295   This does not have a direct correspondent in Python. The closest idiom to
 296   this is tf.append or np.stack. It's different from those in the sense that it
 297   accepts a Tensor list, rather than a list of tensors. It can also accept
 298   TensorArray. When the target is anything else, the dispatcher will rely on
 299   ctx.original_call for fallback.
 300 
 301   Args:
 302     list_: An entity that supports append semantics.
 303     opts: A ListStackOpts object.
 304 
 305   Returns:
 306     The output of the stack operation, typically a Tensor.
 307   """
 308   assert isinstance(opts, ListStackOpts)
 309 
 310   if isinstance(list_, tensor_array_ops.TensorArray):
 311     return _tf_tensorarray_stack(list_)
 312   elif tensor_util.is_tensor(list_):
 313     if list_.dtype == dtypes.variant:
 314       return _tf_tensor_list_stack(list_, opts)
 315     else:
 316       # No-op for primitive Tensor arguments.
 317       return list_
 318   else:
 319     return _py_list_stack(list_, opts)
 320 
 321 
 322 def _tf_tensorarray_stack(list_):
 323   """Overload of list_stack that stages a TensorArray stack."""
 324   return list_.stack()

 327 def _tf_tensor_list_stack(list_, opts):
 328   """Overload of list_stack that stages a Tensor list write."""
 329   if opts.element_dtype is None:
 330     raise ValueError('cannot stack a list without knowing its element type;'
 331                      ' use set_element_type to annotate it')
 332   return list_ops.tensor_list_stack(list_, element_dtype=opts.element_dtype)

 335 def _py_list_stack(list_, opts):
 336   """Overload of list_stack that executes a Python list append."""
 337   # Revert to the original call.
 338   return opts.original_call(list_)
