  33 class GetItemOpts(collections.namedtuple('GetItemOpts', ('element_dtype',))):
  34   pass

  37 def get_item(target, i, opts):
  38   """The slice read operator (i.e. __getitem__).
  39 
  40   Note: it is unspecified whether target will be mutated or not. In general,
  41   if target is mutable (like Python lists), it will be mutated.
  42 
  43   Args:
  44     target: An entity that supports getitem semantics.
  45     i: Index to read from.
  46     opts: A GetItemOpts object.
  47 
  48   Returns:
  49     The read element.
  50 
  51   Raises:
  52     ValueError: if target is not of a supported type.
  53   """
  54   assert isinstance(opts, GetItemOpts)
  55 
  56   if isinstance(target, tensor_array_ops.TensorArray):
  57     return _tf_tensorarray_get_item(target, i)
  58   elif tensor_util.is_tensor(target):
  59     if target.dtype == dtypes.variant:
  60       return _tf_tensor_list_get_item(target, i, opts)
  61     elif target.dtype == dtypes.string and target.shape.ndims == 0:
  62       return _tf_tensor_string_get_item(target, i)
  63     else:
  64       return _tf_tensor_get_item(target, i)
  65   else:
  66     return _py_get_item(target, i)
  67 
  68 
  69 def _tf_tensorarray_get_item(target, i):
  70   """Overload of get_item that stages a TensorArray read."""
  71   return target.read(i)
  72 
  73 
  74 def _tf_tensor_list_get_item(target, i, opts):
  75   """Overload of get_item that stages a Tensor list read."""
  76   if opts.element_dtype is None:
  77     raise ValueError('cannot retrieve from a list without knowing its '
  78                      'element type; use set_element_type to annotate it')
  79   x = list_ops.tensor_list_get_item(target, i, element_dtype=opts.element_dtype)
  80   return x
  81 
  82 
  83 def _tf_tensor_get_item(target, i):
  84   """Overload of get_item that stages a Tensor (not Tensor list) read."""
  85   return target[i]
  86 
  87 
  88 def _tf_tensor_string_get_item(target, i):
  89   """Overload of get_item that stages a Tensor string read."""
  90   x = gen_string_ops.substr(target, i, 1)
  91   return x
  92 
  93 
  94 def _py_get_item(target, i):
  95   """Overload of get_item that executes a Python list modification."""
  96   return target[i]

  99 def set_item(target, i, x):
 100   """The slice write operator (i.e. __setitem__).
 101 
 102   Note: it is unspecified whether target will be mutated or not. In general,
 103   if target is mutable (like Python lists), it will be mutated.
 104 
 105   Args:
 106     target: An entity that supports setitem semantics.
 107     i: Index to modify.
 108     x: The new element value.
 109 
 110   Returns:
 111     Same as target, after the update was performed.
 112 
 113   Raises:
 114     ValueError: if target is not of a supported type.
 115   """
 116   if isinstance(target, tensor_array_ops.TensorArray):
 117     return _tf_tensorarray_set_item(target, i, x)
 118   elif tensor_util.is_tensor(target):
 119     if target.dtype == dtypes.variant:
 120       return _tf_tensor_list_set_item(target, i, x)
 121     else:
 122       raise ValueError(
 123           'tensor lists are expected to be Tensors with dtype=tf.variant,'
 124           ' instead found %s' % target)
 125   else:
 126     return _py_set_item(target, i, x)
 127 
 128 
 129 def _tf_tensorarray_set_item(target, i, x):
 130   """Overload of set_item that stages a TensorArray write."""
 131   return target.write(i, x)
 132 
 133 
 134 def _tf_tensor_list_set_item(target, i, x):
 135   """Overload of set_item that stages a Tensor list update."""
 136   return list_ops.tensor_list_set_item(target, i, x)
 137 
 138 
 139 def _py_set_item(target, i, x):
 140   """Overload of set_item that executes a Python list modification."""
 141   target[i] = x
 142   return target
