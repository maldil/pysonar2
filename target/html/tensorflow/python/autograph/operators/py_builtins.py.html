  43 def overload_of(f):

  69 def _tf_float(x):
  70   # TODO(mdan): We shouldn't assume float32.
  71   if x.dtype == dtypes.string:
  72     return gen_parsing_ops.string_to_number(x, out_type=dtypes.float32)
  73   return math_ops.cast(x, dtype=dtypes.float32)
  74 
  75 
  76 def _py_float(x):
  77   return float(x)

  80 def int_(x=0, base=UNDEFINED):
  81   if tensor_util.is_tensor(x):
  82     return _tf_int(x, base)
  83   return _py_int(x, base)

  86 def _tf_int(x, base):
  87   if base not in (10, UNDEFINED):
  88     raise NotImplementedError('base {} not supported for int'.format(base))
  89 
  90   # TODO(mdan): We shouldn't assume int32.
  91   if x.dtype == dtypes.string:
  92     return gen_parsing_ops.string_to_number(x, out_type=dtypes.int32)
  93   return math_ops.cast(x, dtype=dtypes.int32)
  94 
  95 
  96 def _py_int(x, base):
  97   if base is UNDEFINED:
  98     return int(x)
  99   return int(x, base)

 102 def len_(s):
 103   if tensors.is_tensor_array(s):
 104     return _tf_tensor_array_len(s)
 105   elif tensors.is_tensor_list(s):
 106     return _tf_tensor_list_len(s)
 107   elif tensor_util.is_tensor(s):
 108     return _tf_tensor_len(s)
 109   return _py_len(s)
 110 
 111 
 112 def _tf_tensor_array_len(s):
 113   return s.size()
 114 
 115 
 116 def _tf_tensor_list_len(s):
 117   return list_ops.tensor_list_length(s)

 120 def _tf_tensor_len(s):
 121   """Overload of len_ for Tensor arguments."""
 122   # Statically shaped tensors: length is known ahead of time.
 123   if s.shape.ndims and s.shape[0].value is not None:
 124     return s.shape[0].value
 125 
 126   # Static shape of unknown dimensions: use dynamic shape but statically
 127   # chech that it's a scalar.
 128   shape = array_ops.shape(s)
 129 
 130   assert shape.shape, 'shape tensor of zero size? {}'.format(shape)
 131 
 132   if shape.shape[0] == 0:
 133     raise ValueError(
 134         'len requires a non-scalar tensor, got one of shape {}'.format(shape))
 135 
 136   if shape.shape[0].value is not None:
 137     return array_ops.shape(s)[0]
 138 
 139   # Fully dynamic shape: use ops.
 140   rank = array_ops.rank(s)
 141 
 142   def raise_zero_rank_error():
 143     msg = gen_string_ops.string_join(
 144         ['len requires non-zero rank, got ',
 145          gen_string_ops.as_string(rank)])
 146     with ops.control_dependencies([control_flow_ops.Assert(False, [msg])]):
 147       return constant_op.constant(0, dtype=dtypes.int32)
 148 
 149   return control_flow_ops.cond(rank > 0, lambda: array_ops.shape(s)[0],
 150                                raise_zero_rank_error)

 157 def print_(*objects, **kwargs):
 158   # Note: Python 2.6 doesn't support explicit keywords after starargs.
 159   unknown_kwargs = tuple(
 160       set(kwargs.keys()) - set(('sep', 'end', 'file', 'flush')))
 161   if unknown_kwargs:
 162     raise ValueError('invalid keyword arguments: {}'.format(unknown_kwargs))
 163 
 164   # TODO(mdan): use logging_ops.Print when py_func is not supported.
 165   return _tf_py_func_print(objects, kwargs)
 166 
 167 
 168 def _tf_py_func_print(objects, kwargs):
 169   """Overload of print_ as a py_func implementation."""
 170   override_kwargs = {k: v for k, v in kwargs.items() if v is not UNDEFINED}
 171   if 'flush' not in override_kwargs:
 172     # Defaulting to flushing the console in graph mode, which helps reduce
 173     # garbled output in IPython.
 174     override_kwargs['flush'] = True
 175 
 176   def print_wrapper(*vals):
 177     if six.PY3:
 178       # TensorFlow doesn't seem to generate Unicode when passing strings to
 179       # py_func. This causes the print to add a "b'" wrapper to the output,
 180       # which is probably never what you want.
 181       vals = tuple(
 182           v.decode('utf-8') if isinstance(v, bytes) else v for v in vals)
 183     six.print_(*vals, **override_kwargs)
 184 
 185   return py_func.wrap_py_func(
 186       print_wrapper, None, objects, use_dummy_return=True)

 189 def range_(start_or_stop, stop=UNDEFINED, step=UNDEFINED):
 190   if any(tensor_util.is_tensor(s) for s in (start_or_stop, stop, step)):
 191     return _tf_range(start_or_stop, stop, step)
 192   return _py_range(start_or_stop, stop, step)
 193 
 194 
 195 def _tf_range(start_or_stop, stop, step):
 196   # Note: for static inputs (e.g. constants), tf.range errors out at graph
 197   # construction time, instead of returning an empty tensor. Preventing the
 198   # graph construction error aligns the semantics with Python.
 199 
 200   # TODO(mdan): We should optimize this when a full tensor is not required.
 201   if step is not UNDEFINED:
 202     # TODO(mdan): Add argument coercion similar to other cases.
 203     return math_ops.range(start_or_stop, stop, step)
 204   if stop is not UNDEFINED:
 205     stop = math_ops.maximum(start_or_stop, stop)
 206     return math_ops.range(start_or_stop, stop)
 207   start_or_stop = math_ops.maximum(start_or_stop, 0)
 208   return math_ops.range(start_or_stop)
 209 
 210 
 211 def _py_range(start_or_stop, stop, step):
 212   if step is not UNDEFINED:
 213     return range(start_or_stop, stop, step)
 214   if stop is not UNDEFINED:
 215     return range(start_or_stop, stop)
 216   return range(start_or_stop)
