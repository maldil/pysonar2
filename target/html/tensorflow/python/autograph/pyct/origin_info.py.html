  32 class LineLocation(

  34   """Similar to Location, but without column information.

  36   Attributes:
  37     filename: Text
  38     lineno: int, 1-based
  39   """
  40   pass

  43 class Location(
  44     collections.namedtuple('Location', ('filename', 'lineno', 'col_offset'))):
  45   """Encodes code location information.
  46 
  47   Attributes:
  48     filename: Text
  49     lineno: int, 1-based
  50     col_offset: int
  51   """
  52 
  53   @property
  54   def line_loc(self):
  55     return LineLocation(self.filename, self.lineno)

  58 class OriginInfo(
  59     collections.namedtuple(
  60         'OriginInfo',
  61         ('loc', 'function_name', 'source_code_line', 'comment'))):
  62   """Container for information about the source code before conversion.
  63 
  64   Attributes:
  65     loc: Location
  66     function_name: Optional[Text]
  67     source_code_line: Text
  68     comment: Optional[Text]
  69   """
  70 
  71   def as_frame(self):
  72     """Returns a 4-tuple consistent with the return of traceback.extract_tb."""
  73     return (self.loc.filename, self.loc.lineno, self.function_name,
  74             self.source_code_line)

  78 def create_source_map(nodes, code, filename, indices_in_code):
  79   """Creates a source map between an annotated AST and the code it compiles to.
  80 
  81   Args:
  82     nodes: Iterable[ast.AST, ...]
  83     code: Text
  84     filename: Optional[Text]
  85     indices_in_code: Union[int, Iterable[int, ...]], the positions at which
  86         nodes appear in code. The parser always returns a module when parsing
  87         code. This argument indicates the position in that module's body at
  88         which the corresponding of node should appear.
  89 
  90   Returns:
  91     Dict[CodeLocation, OriginInfo], mapping locations in code to locations
  92     indicated by origin annotations in node.
  93   """
  94   reparsed_nodes = parser.parse_str(code)
  95   reparsed_nodes = [reparsed_nodes.body[i] for i in indices_in_code]
  96 
  97   resolve(reparsed_nodes, code)
  98   result = {}
  99 
 100   for before, after in ast_util.parallel_walk(nodes, reparsed_nodes):
 101     # Note: generated code might not be mapped back to its origin.
 102     # TODO(mdan): Generated code should always be mapped to something.
 103     origin_info = anno.getanno(before, anno.Basic.ORIGIN, default=None)
 104     final_info = anno.getanno(after, anno.Basic.ORIGIN, default=None)
 105     if origin_info is None or final_info is None:
 106       continue
 107 
 108     line_loc = LineLocation(filename, final_info.loc.lineno)
 109 
 110     existing_origin = result.get(line_loc)
 111     if existing_origin is not None:
 112       # Overlaps may exist because of child nodes, but almost never to
 113       # different line locations. Exception make decorated functions, where
 114       # both lines are mapped to the same line in the AST.
 115 
 116       # Line overlaps: keep bottom node.
 117       if existing_origin.loc.line_loc == origin_info.loc.line_loc:
 118         if existing_origin.loc.lineno >= origin_info.loc.lineno:
 119           continue
 120 
 121       # In case of overlaps, keep the leftmost node.
 122       if existing_origin.loc.col_offset <= origin_info.loc.col_offset:
 123         continue
 124 
 125     result[line_loc] = origin_info
 126 
 127   return result

 130 # TODO(znado): Consider refactoring this into a Visitor.
 131 # TODO(mdan): Does this work correctly with inner functions?
 132 def resolve(nodes, source, function=None):
 133   """Adds an origin information to all nodes inside the body of function.
 134 
 135   Args:
 136     nodes: Union[ast.AST, Iterable[ast.AST, ...]]
 137     source: Text, the source code string for the function whose body nodes will
 138       be annotated.
 139     function: Callable, the function that will have all nodes inside of it
 140       annotation with an OriginInfo annotation with key anno.Basic.ORIGIN.  If
 141       it is None then only the line numbers and column offset will be set in the
 142       annotation, with the rest of the information being None.
 143 
 144   Returns:
 145     A tuple of the AST node for function and a String containing its source
 146     code.
 147   """
 148   if not isinstance(nodes, (list, tuple)):
 149     nodes = (nodes,)
 150 
 151   if function:
 152     _, function_lineno = tf_inspect.getsourcelines(function)
 153     function_filepath = tf_inspect.getsourcefile(function)
 154   else:
 155     function_lineno = None
 156     function_filepath = None
 157 
 158   # TODO(mdan): Pull this to a separate utility.
 159   code_reader = six.StringIO(source)
 160   comment_map = {}
 161   for token in tokenize.generate_tokens(code_reader.readline):
 162     tok_type, tok_string, loc, _, _ = token
 163     srow, _ = loc
 164     if tok_type == tokenize.COMMENT:
 165       comment_map[srow] = tok_string.strip()[1:].strip()
 166 
 167   source_lines = source.split('\n')
 168   for node in nodes:
 169     for n in gast.walk(node):
 170       if not hasattr(n, 'lineno'):
 171         continue
 172 
 173       lineno_in_body = n.lineno
 174 
 175       source_code_line = source_lines[lineno_in_body - 1]
 176       if function:
 177         source_lineno = function_lineno + lineno_in_body
 178         function_name = function.__name__
 179       else:
 180         source_lineno = lineno_in_body
 181         function_name = None
 182 
 183       location = Location(function_filepath, source_lineno, n.col_offset)
 184       origin = OriginInfo(location, function_name,
 185                           source_code_line, comment_map.get(source_lineno))
 186       anno.setanno(n, anno.Basic.ORIGIN, origin)
