  15 """Conversion to A-normal form.
  16 
  17 The general idea of A-normal form is that every intermediate value is
  18 explicitly named with a variable.  For more, see
  19 https://en.wikipedia.org/wiki/A-normal_form.
  20 
  21 The specific converters used here are based on Python AST semantics as
  22 documented at https://greentreesnakes.readthedocs.io/en/latest/.
  23 """

  36 class DummyGensym(object):
  37   """A dumb gensym that suffixes a stem by sequential numbers from 1000."""
  38 
  39   def __init__(self, entity_info):
  40     del entity_info
  41     # A proper implementation needs to account for:
  42     #   * entity_info.namespace
  43     #   * all the symbols defined in the AST
  44     #   * the symbols generated so far
  45     self._idx = 0
  46 
  47   def new_name(self, stem='tmp'):
  48     self._idx += 1
  49     return stem + '_' + str(1000 + self._idx)

  52 class AnfTransformer(transformer.Base):
  53   """Performs the conversion to A-normal form (ANF)."""
  54 
  55   # The algorithm is a postorder recursive tree walk.  Any given node A may, in
  56   # general, require creation of a series B of Assign statements, which compute
  57   # and explicitly name the intermediate values needed to compute the value of
  58   # A.  If A was already a statement, it can be replaced with the sequence B +
  59   # [A].  If A was an expression, B needs to be propagated up the tree until a
  60   # statement is encountered.  Since the `ast.NodeTransformer` framework makes
  61   # no provision for subtraversals returning side information, this class
  62   # accumulates the sequence B in an instance variable.
  63 
  64   # The only other subtlety is that some Python statements (like `if`) have both
  65   # expression fields (`test`) and statement list fields (`body` and `orelse`).
  66   # Any additional assignments needed to name all the intermediate values in the
  67   # `test` can be prepended to the `if` node, but assignments produced by
  68   # processing the `body` and the `orelse` need to be kept together with them,
  69   # and not accidentally lifted out of the `if`.
  70 
  71   def __init__(self, entity_info, gensym_source=None):
  72     """Creates an ANF transformer.
  73 
  74     Args:
  75       entity_info: transformer.EntityInfo
  76       gensym_source: An optional object with the same interface as `DummyGensym`
  77         for generating unique names
  78     """
  79     super(AnfTransformer, self).__init__(entity_info)
  80     if gensym_source is None:
  81       self._gensym = DummyGensym(entity_info)
  82     else:
  83       self._gensym = gensym_source(entity_info)
  84     self._pending_statements = []
  85 
  86   def _consume_pending_statements(self):
  87     ans = self._pending_statements
  88     self._pending_statements = []
  89     return ans
  90 
  91   def _add_pending_statement(self, stmt):
  92     self._pending_statements.append(stmt)
  93 
  94   _trivial_nodes = (
  95       # Non-nodes that show up as AST fields
  96       bool, six.string_types,
  97       # Leaf nodes that are already in A-normal form
  98       gast.expr_context, gast.Name, gast.Num, gast.Str, gast.Bytes,
  99       gast.NameConstant, gast.Ellipsis,
 100       # Binary operators
 101       gast.Add, gast.Sub, gast.Mult, gast.Div, gast.Mod, gast.Pow, gast.LShift,
 102       gast.RShift, gast.BitOr, gast.BitXor, gast.BitAnd, gast.FloorDiv,
 103       # Unary operators
 104       gast.Invert, gast.Not, gast.UAdd, gast.USub,
 105       # Comparison operators
 106       gast.Eq, gast.NotEq, gast.Lt, gast.LtE, gast.Gt, gast.GtE,
 107       gast.Is, gast.IsNot, gast.In, gast.NotIn,
 108   )

 110   def _is_node_trivial(self, node):
 111     if node is None:
 112       return True
 113     elif isinstance(node, self._trivial_nodes):
 114       return True
 115     elif isinstance(node, gast.keyword):
 116       return self._is_node_trivial(node.value)
 117     elif isinstance(node, (gast.Starred, gast.withitem, gast.slice)):
 118       return self._are_children_trivial(node)
 119     return False
 120 
 121   def _are_children_trivial(self, node):
 122     for field in node._fields:
 123       if not field.startswith('__'):
 124         if not self._is_node_trivial(getattr(node, field)):
 125           return False
 126     return True
 127 
 128   def _ensure_node_is_trivial(self, node):
 129     if node is None:
 130       return node
 131     elif isinstance(node, self._trivial_nodes):
 132       return node
 133     elif isinstance(node, list):
 134       # If something's field was actually a list, e.g., variadic arguments.
 135       return [self._ensure_node_is_trivial(n) for n in node]
 136     elif isinstance(node, gast.keyword):
 137       node.value = self._ensure_node_is_trivial(node.value)
 138       return node
 139     elif isinstance(node, (gast.Starred, gast.withitem, gast.slice)):
 140       return self._ensure_fields_trivial(node)
 141     elif isinstance(node, gast.expr):
 142       temp_name = self._gensym.new_name()
 143       temp_assign = templates.replace(
 144           'temp_name = expr', temp_name=temp_name, expr=node)[0]
 145       self._add_pending_statement(temp_assign)
 146       answer = templates.replace('temp_name', temp_name=temp_name)[0]
 147       return answer
 148     else:
 149       raise ValueError('Do not know how to treat {}'.format(node))
 150 
 151   def _ensure_fields_trivial(self, node):
 152     for field in node._fields:
 153       if field.startswith('__'):
 154         continue
 155       setattr(node, field, self._ensure_node_is_trivial(getattr(node, field)))
 156     return node
 157 
 158   def _visit_strict_statement(self, node, trivialize_children=True):
 159     assert not self._pending_statements
 160     node = self.generic_visit(node)
 161     if trivialize_children:
 162       self._ensure_fields_trivial(node)
 163     results = self._consume_pending_statements()
 164     results.append(node)
 165     return results
 166 
 167   def _visit_strict_expression(self, node):
 168     node = self.generic_visit(node)
 169     self._ensure_fields_trivial(node)
 170     return node

 192   def visit_For(self, node):
 193     assert not self._pending_statements
 194     # It's important to visit node.iter first, because any statements created
 195     # thereby need to live outside the body.
 196     self.visit(node.iter)
 197     node.iter = self._ensure_node_is_trivial(node.iter)
 198     iter_stmts = self._consume_pending_statements()
 199     # This generic_visit will revisit node.iter, but that is both correct and
 200     # cheap because by this point node.iter is trivial.
 201     node = self.generic_visit(node)
 202     assert not self._pending_statements
 203     iter_stmts.append(node)
 204     return iter_stmts
 205 
 206   def visit_AsyncFor(self, node):
 207     if not self._are_children_trivial(node):
 208       msg = ('Nontrivial AsyncFor nodes not supported yet '
 209              '(need to think through the semantics).')
 210       raise ValueError(msg)
 211     return self.generic_visit(node)

 213   def visit_While(self, node):
 214     if not self._is_node_trivial(node.test):
 215       msg = ('While with nontrivial test not supported yet '
 216              '(need to avoid precomputing the test).')
 217       raise ValueError(msg)
 218     return self.generic_visit(node)
 219 
 220   def visit_If(self, node):
 221     assert not self._pending_statements
 222     # It's important to visit node.test first, because any statements created
 223     # thereby need to live outside the body.
 224     self.visit(node.test)
 225     node.test = self._ensure_node_is_trivial(node.test)
 226     condition_stmts = self._consume_pending_statements()
 227     # This generic_visit will revisit node.test, but that is both correct and
 228     # cheap because by this point node.test is trivial.
 229     node = self.generic_visit(node)
 230     assert not self._pending_statements
 231     condition_stmts.append(node)
 232     return condition_stmts
 233 
 234   def visit_With(self, node):
 235     assert not self._pending_statements
 236     # It's important to visit node.items first, because any statements created
 237     # thereby need to live outside the body.
 238     for item in node.items:
 239       self.visit(item)
 240     node.items = [self._ensure_node_is_trivial(n) for n in node.items]
 241     contexts_stmts = self._consume_pending_statements()
 242     # This generic_visit will revisit node.items, but that is both correct and
 243     # cheap because by this point node.items is trivial.
 244     node = self.generic_visit(node)
 245     assert not self._pending_statements
 246     contexts_stmts.append(node)
 247     return contexts_stmts

 409 def transform(node, entity_info, gensym_source=None):
 410   """Converts the given node to A-normal form (ANF).
 411 
 412   The general idea of A-normal form: https://en.wikipedia.org/wiki/A-normal_form
 413 
 414   The specific converters used here are based on Python AST semantics as
 415   documented at https://greentreesnakes.readthedocs.io/en/latest/.
 416 
 417   Args:
 418     node: The node to transform.
 419     entity_info: transformer.EntityInfo.  TODO(mdan): What information does this
 420       argument provide?
 421     gensym_source: An optional object with the same interface as `DummyGensym`
 422       for generating unique names.
 423   """
 424   return AnfTransformer(entity_info, gensym_source=gensym_source).visit(node)
