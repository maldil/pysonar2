  35 class ReplaceTransformer(gast.NodeTransformer):
  36   """Replace AST nodes."""
  37 
  38   def __init__(self, replacements):
  39     """Create a new ReplaceTransformer.
  40 
  41     Args:
  42       replacements: A mapping from placeholder names to (lists of) AST nodes
  43           that these placeholders will be replaced by.
  44     """
  45     self.replacements = replacements
  46     self.in_replacements = False
  47     self.preserved_annos = {
  48         anno.Basic.ORIGIN,
  49         anno.Basic.SKIP_PROCESSING,
  50         anno.Static.ORIG_DEFINITIONS,
  51     }
  52 
  53   def _prepare_replacement(self, replaced, key):
  54     """Prepares a replacement AST that's safe to swap in for a node.
  55 
  56     Args:
  57       replaced: ast.AST, the node being replaced
  58       key: Hashable, the key of the replacement AST
  59     Returns:
  60       ast.AST, the replacement AST
  61     """
  62     repl = self.replacements[key]
  63 
  64     new_nodes = ast_util.copy_clean(repl, preserve_annos=self.preserved_annos)
  65     if isinstance(new_nodes, gast.AST):
  66       new_nodes = [new_nodes]
  67 
  68     return new_nodes
  69 
  70   def visit_Expr(self, node):
  71     # When replacing a placeholder with an entire statement, the replacement
  72     # must stand on its own and not be wrapped in an Expr.
  73     new_value = self.visit(node.value)
  74     if new_value is node.value:
  75       return node
  76     return new_value
  77 
  78   def visit_keyword(self, node):
  79     if node.arg not in self.replacements:
  80       return self.generic_visit(node)
  81 
  82     repl = self._prepare_replacement(node, node.arg)
  83     if isinstance(repl, gast.keyword):
  84       return repl
  85     elif (repl and isinstance(repl, (list, tuple)) and
  86           all(isinstance(r, gast.keyword) for r in repl)):
  87       return repl
  88     # TODO(mdan): We may allow replacing with a string as well.
  89     # For example, if one wanted to replace foo with bar in foo=baz, then
  90     # we could allow changing just node arg, so that we end up with bar=baz.
  91     raise ValueError(
  92         'a keyword argument may only be replaced by another keyword or a '
  93         'non-empty list of keywords. Found: %s' % repl)
  94 
  95   def visit_FunctionDef(self, node):
  96     node = self.generic_visit(node)
  97     if node.name not in self.replacements:
  98       return node
  99 
 100     repl = self.replacements[node.name]
 101     if not isinstance(repl, (gast.Name, ast.Name)):
 102       raise ValueError(
 103           'a function name can only be replaced by a Name node. Found: %s' %
 104           repl)
 105     node.name = repl.id
 106     return node
 107 
 108   def _check_has_context(self, node):
 109     if not node.ctx:
 110       raise ValueError('node %s is missing ctx value' % node)
 111 
 112   # TODO(mdan): Rewrite _check and _set using a separate transformer.
 113   def _check_inner_children_have_context(self, node):
 114     if isinstance(node, gast.Attribute):
 115       self._check_inner_children_have_context(node.value)
 116       self._check_has_context(node)
 117     elif isinstance(node, (gast.Tuple, gast.List)):
 118       for e in node.elts:
 119         self._check_inner_children_have_context(e)
 120       self._check_has_context(node)
 121     elif isinstance(node, gast.Dict):
 122       for e in node.keys:
 123         self._check_inner_children_have_context(e)
 124       for e in node.values:
 125         self._check_inner_children_have_context(e)
 126     elif isinstance(node, gast.Index):
 127       self._check_inner_children_have_context(node.value)
 128     elif isinstance(node, gast.Subscript):
 129       self._check_inner_children_have_context(node.value)
 130       self._check_inner_children_have_context(node.slice)
 131     elif isinstance(node, gast.Slice):
 132       self._check_inner_children_have_context(node.lower)
 133       if node.upper:
 134         self._check_inner_children_have_context(node.upper)
 135       if node.step:
 136         self._check_inner_children_have_context(node.step)
 137     elif isinstance(node, gast.BinOp):
 138       self._check_inner_children_have_context(node.left)
 139       self._check_inner_children_have_context(node.right)
 140     elif isinstance(node, gast.UnaryOp):
 141       self._check_inner_children_have_context(node.operand)
 142     elif isinstance(node, gast.Name):
 143       self._check_has_context(node)
 144     elif isinstance(node, (gast.Str, gast.Num)):
 145       pass
 146     else:
 147       raise ValueError('unexpected node type "%s"' % node)
 148 
 149   def _set_inner_child_context(self, node, ctx):
 150     if isinstance(node, gast.Attribute):
 151       self._set_inner_child_context(node.value, gast.Load())
 152       node.ctx = ctx
 153     elif isinstance(node, (gast.Tuple, gast.List)):
 154       for e in node.elts:
 155         self._set_inner_child_context(e, ctx)
 156       node.ctx = ctx
 157     elif isinstance(node, gast.Name):
 158       node.ctx = ctx
 159     elif isinstance(node, gast.Call):
 160       self._set_inner_child_context(node.func, ctx)
 161       # We may be able to override these to Load(), but for now it's simpler
 162       # to just assert that they're set.
 163       for a in node.args:
 164         self._check_inner_children_have_context(a)
 165       for k in node.keywords:
 166         self._check_inner_children_have_context(k.value)
 167     elif isinstance(node, gast.Dict):
 168       # We may be able to override these to Load(), but for now it's simpler
 169       # to just assert that they're set.
 170       for e in node.keys:
 171         self._check_inner_children_have_context(e)
 172       for e in node.values:
 173         self._check_inner_children_have_context(e)
 174     elif isinstance(node, gast.Subscript):
 175       self._set_inner_child_context(node.value, ctx)
 176       self._check_inner_children_have_context(node.slice)
 177     elif isinstance(node, gast.BinOp):
 178       self._check_inner_children_have_context(node.left)
 179       self._check_inner_children_have_context(node.right)
 180     elif isinstance(node, gast.UnaryOp):
 181       self._check_inner_children_have_context(node.operand)
 182     elif isinstance(node, (gast.Str, gast.Num)):
 183       pass
 184     else:
 185       raise ValueError('unexpected node type "%s"' % node)

 187   def visit_Attribute(self, node):
 188     node = self.generic_visit(node)
 189     if node.attr not in self.replacements:
 190       return node
 191 
 192     repl = self.replacements[node.attr]
 193     if not isinstance(repl, gast.Name):
 194       raise ValueError(
 195           'An attribute can only be replaced by a Name node. Found: %s' % repl)
 196     node.attr = repl.id
 197     return node
 198 
 199   def visit_Name(self, node):
 200     if node.id not in self.replacements:
 201       return node
 202 
 203     new_nodes = self._prepare_replacement(node, node.id)
 204 
 205     # Preserve the target context.
 206     for n in new_nodes:
 207       if isinstance(n, (gast.Tuple, gast.List)):
 208         for e in n.elts:
 209           self._set_inner_child_context(e, node.ctx)
 210       if isinstance(n, gast.Attribute):
 211         # For attributes, the inner Name node receives the context, while the
 212         # outer ones have it set to Load.
 213         self._set_inner_child_context(n, node.ctx)
 214       else:
 215         n.ctx = node.ctx
 216 
 217     if len(new_nodes) == 1:
 218       new_nodes, = new_nodes
 219 
 220     return new_nodes

 223 def _convert_to_ast(n):
 224   """Converts from a known data type to AST."""
 225   if isinstance(n, str):
 226     # Note: the node will receive the ctx value from the template, see
 227     # ReplaceTransformer.visit_Name.
 228     return gast.Name(id=n, ctx=None, annotation=None)
 229   if isinstance(n, qual_names.QN):
 230     return n.ast()
 231   if isinstance(n, list):
 232     return [_convert_to_ast(e) for e in n]
 233   if isinstance(n, tuple):
 234     return tuple(_convert_to_ast(e) for e in n)
 235   return n

 238 def replace(template, **replacements):
 239   """Replaces placeholders in a Python template.
 240 
 241   AST Name and Tuple nodes always receive the context that inferred from
 242   the template. However, when replacing more complex nodes (that can potentially
 243   contain Name children), then the caller is responsible for setting the
 244   appropriate context.
 245 
 246   Args:
 247     template: A string representing Python code. Any symbol name can be used
 248         that appears in the template code can be used as placeholder.
 249     **replacements: A mapping from placeholder names to (lists of) AST nodes
 250         that these placeholders will be replaced by. String values are also
 251         supported as a shorthand for AST Name nodes with the respective ID.
 252 
 253   Returns:
 254     An AST node or list of AST nodes with the replacements made. If the
 255     template was a function, a list will be returned. If the template was a
 256     node, the same node will be returned. If the template was a string, an
 257     AST node will be returned (a `Module` node in the case of a multi-line
 258     string, an `Expr` node otherwise).
 259 
 260   Raises:
 261     ValueError: if the arguments are incorrect.
 262   """
 263   if not isinstance(template, str):
 264     raise ValueError('Expected string template, got %s' % type(template))
 265   tree = parser.parse_str(textwrap.dedent(template))
 266   for k in replacements:
 267     replacements[k] = _convert_to_ast(replacements[k])
 268   results = ReplaceTransformer(replacements).visit(tree).body
 269   if isinstance(results, list):
 270     return [qual_names.resolve(r) for r in results]
 271   return qual_names.resolve(results)

 274 def replace_as_expression(template, **replacements):
 275   """Variant of replace that generates expressions, instead of code blocks."""
 276   replacement = replace(template, **replacements)
 277   if len(replacement) != 1:
 278     raise ValueError(
 279         'single expression expected; for more general templates use replace')
 280   node = replacement[0]
 281   node = qual_names.resolve(node)
 282 
 283   if isinstance(node, gast.Expr):
 284     return node.value
 285   elif isinstance(node, gast.Name):
 286     return node
 287 
 288   raise ValueError(
 289       'the template is expected to generate an expression or a name node;'
 290       ' instead found %s' % node)
