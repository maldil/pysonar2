  29 class CleanCopier(object):
  30   """NodeTransformer-like visitor that copies an AST."""
  31 
  32   def __init__(self, preserve_annos):
  33     super(CleanCopier, self).__init__()
  34     self.preserve_annos = preserve_annos
  35 
  36   def copy(self, node):
  37     """Returns a deep copy of node (excluding some fields, see copy_clean)."""
  38 
  39     if isinstance(node, list):
  40       return [self.copy(n) for n in node]
  41     elif isinstance(node, tuple):
  42       return tuple(self.copy(n) for n in node)
  43     elif not isinstance(node, (gast.AST, ast.AST)):
  44       # Assuming everything that's not an AST, list or tuple is a value type
  45       # and may simply be assigned.
  46       return node
  47 
  48     assert isinstance(node, (gast.AST, ast.AST))
  49 
  50     new_fields = {}
  51     for f in node._fields:
  52       if not f.startswith('__') and hasattr(node, f):
  53         new_fields[f] = self.copy(getattr(node, f))
  54     new_node = type(node)(**new_fields)
  55 
  56     if self.preserve_annos:
  57       for k in self.preserve_annos:
  58         anno.copyanno(node, new_node, k)
  59     return new_node
  60 
  61 
  62 def copy_clean(node, preserve_annos=None):
  63   """Creates a deep copy of an AST.
  64 
  65   The copy will not include fields that are prefixed by '__', with the
  66   exception of user-specified annotations.
  67 
  68   Args:
  69     node: ast.AST
  70     preserve_annos: Optional[Set[Hashable]], annotation keys to include in the
  71         copy
  72   Returns:
  73     ast.AST
  74   """
  75   return CleanCopier(preserve_annos).copy(node)

  78 class SymbolRenamer(gast.NodeTransformer):
  79   """Transformer that can rename symbols to a simple names."""
  80 
  81   def __init__(self, name_map):
  82     self.name_map = name_map
  83 
  84   def _process(self, node):
  85     qn = anno.getanno(node, anno.Basic.QN)
  86     if qn in self.name_map:
  87       new_node = gast.Name(str(self.name_map[qn]), node.ctx, None)
  88       # All annotations get carried over.
  89       for k in anno.keys(node):
  90         anno.copyanno(node, new_node, k)
  91       return new_node
  92     return self.generic_visit(node)
  93 
  94   def visit_Name(self, node):
  95     return self._process(node)
  96 
  97   def visit_Attribute(self, node):
  98     if anno.hasanno(node, anno.Basic.QN):
  99       return self._process(node)
 100     # Attributes of dynamic objects will not have a QN.
 101     return self.generic_visit(node)

 104 def rename_symbols(node, name_map):
 105   """Renames symbols in an AST. Requires qual_names annotations."""
 106   renamer = SymbolRenamer(name_map)
 107   if isinstance(node, list):
 108     return [renamer.visit(n) for n in node]
 109   elif isinstance(node, tuple):
 110     return tuple(renamer.visit(n) for n in node)
 111   return renamer.visit(node)

 114 def keywords_to_dict(keywords):
 115   """Converts a list of ast.keyword objects to a dict."""
 116   keys = []
 117   values = []
 118   for kw in keywords:
 119     keys.append(gast.Str(kw.arg))
 120     values.append(kw.value)
 121   return gast.Dict(keys=keys, values=values)

 124 class PatternMatcher(gast.NodeVisitor):
 125   """Matches a node against a pattern represented by a node."""
 126 
 127   def __init__(self, pattern):
 128     self.pattern = pattern
 129     self.pattern_stack = []
 130     self.matches = True
 131 
 132   def compare_and_visit(self, node, pattern):
 133     self.pattern_stack.append(self.pattern)
 134     self.pattern = pattern
 135     self.generic_visit(node)
 136     self.pattern = self.pattern_stack.pop()
 137 
 138   def no_match(self):
 139     self.matches = False
 140     return False
 141 
 142   def is_wildcard(self, p):
 143     if isinstance(p, (list, tuple)) and len(p) == 1:
 144       p, = p
 145     if isinstance(p, gast.Name) and p.id == '_':
 146       return True
 147     if p == '_':
 148       return True
 149     return False
 150 
 151   def generic_visit(self, node):
 152     if not self.matches:
 153       return
 154 
 155     pattern = self.pattern
 156     for f in node._fields:
 157       if f.startswith('__'):
 158         continue
 159 
 160       if not hasattr(node, f):
 161         if hasattr(pattern, f) and getattr(pattern, f):
 162           return self.no_match()
 163         else:
 164           continue
 165       if not hasattr(pattern, f):
 166         return self.no_match()
 167 
 168       v = getattr(node, f)
 169       p = getattr(pattern, f)
 170 
 171       if self.is_wildcard(p):
 172         continue
 173       if isinstance(v, (list, tuple)):
 174         if not isinstance(p, (list, tuple)) or len(v) != len(p):
 175           return self.no_match()
 176         for v_item, p_item in zip(v, p):
 177           self.compare_and_visit(v_item, p_item)
 178       elif isinstance(v, (gast.AST, ast.AST)):
 179         if not isinstance(v, type(p)) and not isinstance(p, type(v)):
 180           return self.no_match()
 181         self.compare_and_visit(v, p)
 182       else:
 183         # Assume everything else is a value type.
 184         if v != p:
 185           return self.no_match()

 188 def matches(node, pattern):
 189   """Basic pattern matcher for AST.
 190 
 191   The pattern may contain wildcards represented by the symbol '_'. A node
 192   matches a pattern if for every node in the tree, either there is a node of
 193   the same type in pattern, or a Name node with id='_'.
 194 
 195   Args:
 196     node: ast.AST
 197     pattern: ast.AST
 198   Returns:
 199     bool
 200   """
 201   if isinstance(pattern, str):
 202     pattern = parser.parse_expression(pattern)
 203   matcher = PatternMatcher(pattern)
 204   matcher.visit(node)
 205   return matcher.matches

 209 def apply_to_single_assignments(targets, values, apply_fn):
 210   """Applies a function to each individual assignment.
 211 
 212   This function can process a possibly-unpacked (e.g. a, b = c, d) assignment.
 213   It tries to break down the unpacking if possible. In effect, it has the same
 214   effect as passing the assigned values in SSA form to apply_fn.
 215 
 216   Examples:
 217 
 218   The following will result in apply_fn(a, c), apply_fn(b, d):
 219 
 220       a, b = c, d
 221 
 222   The following will result in apply_fn(a, c[0]), apply_fn(b, c[1]):
 223 
 224       a, b = c
 225 
 226   The following will result in apply_fn(a, (b, c)):
 227 
 228       a = b, c
 229 
 230   It uses the visitor pattern to allow subclasses to process single
 231   assignments individually.
 232 
 233   Args:
 234     targets: Union[List[ast.AST, ...], Tuple[ast.AST, ...], ast.AST, should be
 235         used with the targets field of an ast.Assign node
 236     values: ast.AST
 237     apply_fn: Callable[[ast.AST, ast.AST], None], called with the
 238         respective nodes of each single assignment
 239   """
 240   if not isinstance(targets, (list, tuple)):
 241     targets = (targets,)
 242   for target in targets:
 243     if isinstance(target, (gast.Tuple, gast.List)):
 244       for i in range(len(target.elts)):
 245         target_el = target.elts[i]
 246         if isinstance(values, (gast.Tuple, gast.List)):
 247           value_el = values.elts[i]
 248         else:
 249           idx = parser.parse_expression(str(i))
 250           value_el = gast.Subscript(values, gast.Index(idx), ctx=gast.Load())
 251         apply_to_single_assignments(target_el, value_el, apply_fn)
 252     else:
 253       apply_fn(target, values)

 256 def parallel_walk(node, other):
 257   """Walks two ASTs in parallel.
 258 
 259   The two trees must have identical structure.
 260 
 261   Args:
 262     node: Union[ast.AST, Iterable[ast.AST]]
 263     other: Union[ast.AST, Iterable[ast.AST]]
 264   Yields:
 265     Tuple[ast.AST, ast.AST]
 266   Raises:
 267     ValueError: if the two trees don't have identical structure.
 268   """
 269   if isinstance(node, (list, tuple)):
 270     node_stack = list(node)
 271   else:
 272     node_stack = [node]
 273 
 274   if isinstance(other, (list, tuple)):
 275     other_stack = list(other)
 276   else:
 277     other_stack = [other]
 278 
 279   while node_stack and other_stack:
 280     assert len(node_stack) == len(other_stack)
 281     n = node_stack.pop()
 282     o = other_stack.pop()
 283 
 284     if (not isinstance(n, (ast.AST, gast.AST)) or
 285         not isinstance(o, (ast.AST, gast.AST)) or
 286         n.__class__.__name__ != o.__class__.__name__):
 287       raise ValueError('inconsistent nodes: {} and {}'.format(n, o))
 288 
 289     yield n, o
 290 
 291     for f in n._fields:
 292       n_child = getattr(n, f, None)
 293       o_child = getattr(o, f, None)
 294       if f.startswith('__') or n_child is None or o_child is None:
 295         continue
 296 
 297       if isinstance(n_child, (list, tuple)):
 298         if (not isinstance(o_child, (list, tuple)) or
 299             len(n_child) != len(o_child)):
 300           raise ValueError(
 301               'inconsistent values for field {}: {} and {}'.format(
 302                   f, n_child, o_child))
 303         node_stack.extend(n_child)
 304         other_stack.extend(o_child)
 305 
 306       elif isinstance(n_child, (gast.AST, ast.AST)):
 307         node_stack.append(n_child)
 308         other_stack.append(o_child)
 309 
 310       elif n_child != o_child:
 311         raise ValueError(
 312             'inconsistent values for field {}: {} and {}'.format(
 313                 f, n_child, o_child))
