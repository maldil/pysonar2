  15 """Control flow graph (CFG) structure for Python AST representation.
  16 
  17 The CFG is a digraph with edges representing valid control flow. Each
  18 node is associated with exactly one AST node, but not all AST nodes may have
  19 a corresponding CFG counterpart.
  20 
  21 Once built, the CFG itself is immutable, but the values it holds need not be;
  22 they are usually annotated with information extracted by walking the graph.
  23 """

  40 class Node(object):
  41   """A node in the CFG.
  42 
  43   Although new instances of this class are mutable, the objects that a user
  44   finds in the CFG are typically not.
  45 
  46   The nodes represent edges in the CFG graph, and maintain pointers to allow
  47   efficient walking in both forward and reverse order. The following property
  48   holds for all nodes: "child in node.next" iff "node in child.prev".
  49 
  50   Attributes:
  51     next: FrozenSet[Node, ...], the nodes that follow this node, in control
  52       flow order
  53     prev: FrozenSet[Node, ...], the nodes that precede this node, in reverse
  54       control flow order
  55     ast_node: ast.AST, the AST node corresponding to this CFG node
  56   """
  57 
  58   def __init__(self, next_, prev, ast_node):
  59     self.next = next_
  60     self.prev = prev
  61     self.ast_node = ast_node
  62 
  63   def freeze(self):
  64     self.next = frozenset(self.next)
  65     # Assumption: All CFG nodes have identical life spans, because the graph
  66     # owns them. Nodes should never be used outside the context of an existing
  67     # graph.
  68     self.prev = weakref.WeakSet(self.prev)
  69 
  70   def __repr__(self):
  71     if isinstance(self.ast_node, gast.FunctionDef):
  72       return 'def %s' % self.ast_node.name
  73     elif isinstance(self.ast_node, gast.withitem):
  74       return compiler.ast_to_source(self.ast_node.context_expr).strip()
  75     return compiler.ast_to_source(self.ast_node).strip()

  78 class Graph(
  79     collections.namedtuple(
  80         'Graph',
  81         ['entry', 'exit', 'error', 'index', 'stmt_prev', 'stmt_next'])):
  82   """A Control Flow Graph.
  83 
  84   The CFG maintains an index to allow looking up a CFG node by the AST node to
  85   which it is associated. The index can also be enumerated in top-down, depth
  86   first order.
  87 
  88   Walking the graph in forward or reverse order is supported by double
  89   parent-child links.
  90 
  91   Note: the error nodes are not wired to their corresponding finally guards,
  92   because these are shared, and wiring them would create a reverse path from
  93   normal control flow into the error nodes, which we want to avoid.
  94 
  95   The graph also maintains edges corresponding to higher level statements
  96   like for-else loops. A node is considered successor of a statement if there
  97   is an edge from a node that is lexically a child of that statement to a node
  98   that is not. Statement predecessors are analogously defined.
  99 
 100   Attributes:
 101     entry: Node, the entry node
 102     exit: FrozenSet[Node, ...], the exit nodes
 103     error: FrozenSet[Node, ...], nodes that exit due to an explicitly raised
 104         error (errors propagated from function calls are not accounted)
 105     index: Dict[ast.Node, Node], mapping AST nodes to the respective CFG
 106         node
 107     stmt_prev: Dict[ast.Node, FrozenSet[Node, ...]], mapping statement AST
 108         nodes to their predecessor CFG nodes
 109     stmt_next: Dict[ast.Node, FrozenSet[Node, ...]], mapping statement AST
 110         nodes to their successor CFG nodes
 111   """
 112 
 113   def __repr__(self):
 114     result = 'digraph CFG {\n'
 115     for node in self.index.values():
 116       result += '  %s [label="%s"];\n' % (id(node), node)
 117     for node in self.index.values():
 118       for next_ in node.next:
 119         result += '  %s -> %s;\n' % (id(node), id(next_))
 120     result += '}'
 121     return result

 131 class GraphVisitor(object):
 132   """Base class for a CFG visitors.
 133 
 134   This implementation is not thread safe.
 135 
 136   The visitor has some facilities to simplify dataflow analyses. In particular,
 137   it allows revisiting the nodes at the decision of the subclass. This can be
 138   used to visit the graph until the state reaches a fixed point.
 139 
 140   For more details on dataflow analysis, see
 141   https://www.seas.harvard.edu/courses/cs252/2011sp/slides/Lec02-Dataflow.pdf
 142 
 143   Note: the literature generally suggests visiting successor nodes only when the
 144   state of the current node changed, regardless of whether that successor has
 145   ever been visited. This implementation visits every successor at least once.
 146 
 147   Attributes:
 148     graph: Graph
 149     in_: Dict[Node, Any], stores node-keyed state during a visit
 150     out: Dict[Node, Any], stores node-keyed state during a visit
 151   """
 152 
 153   def __init__(self, graph):
 154     self.graph = graph
 155     self.reset()
 156 
 157   def init_state(self, node):
 158     """State initialization function. Optional to overload.
 159 
 160     An in/out state slot will be created for each node in the graph. Subclasses
 161     must overload this to control what that is initialized to.
 162 
 163     Args:
 164       node: Node
 165     """
 166     raise NotImplementedError('Subclasses must implement this.')
 167 
 168   # TODO(mdan): Rename to flow?
 169   def visit_node(self, node):
 170     """Visitor function.
 171 
 172     Args:
 173       node: Node
 174     Returns:
 175       bool, whether the node should be revisited; subclasses can visit every
 176           reachable node exactly once by always returning False
 177     """
 178     raise NotImplementedError('Subclasses must implement this.')
 179 
 180   def reset(self):
 181     self.in_ = {
 182         node: self.init_state(node) for node in self.graph.index.values()
 183     }
 184     self.out = {
 185         node: self.init_state(node) for node in self.graph.index.values()
 186     }
 187 
 188   def _visit_internal(self, mode):
 189     """Visits the CFG, depth-first."""
 190     assert mode in (_WalkMode.FORWARD, _WalkMode.REVERSE)
 191     if mode == _WalkMode.FORWARD:
 192       open_ = [self.graph.entry]
 193     elif mode == _WalkMode.REVERSE:
 194       open_ = list(self.graph.exit)
 195     closed = set()
 196 
 197     while open_:
 198       node = open_.pop(0)
 199       closed.add(node)
 200 
 201       should_revisit = self.visit_node(node)
 202 
 203       if mode == _WalkMode.FORWARD:
 204         children = node.next
 205       elif mode == _WalkMode.REVERSE:
 206         children = node.prev
 207 
 208       for next_ in children:
 209         if should_revisit or next_ not in closed:
 210           open_.append(next_)
 211 
 212   def visit_forward(self):
 213     self._visit_internal(_WalkMode.FORWARD)
 214 
 215   def visit_reverse(self):
 216     self._visit_internal(_WalkMode.REVERSE)

 219 class GraphBuilder(object):
 220   """Builder that constructs a CFG from a given AST.
 221 
 222   This GraphBuilder facilitates constructing the DAG that forms the CFG when
 223   nodes
 224   are supplied in lexical order (i.e., top-down, depth first). Under these
 225   conditions, it supports building patterns found in typical structured
 226   programs.
 227 
 228   This builder ignores the flow generated by exceptions, which are assumed to
 229   always be catastrophic and present purely for diagnostic purposes (e.g. to
 230   print debug information). Statements like raise and try/catch sections are
 231   allowed and will generate control flow edges, but ordinaty statements are
 232   assumed not to raise exceptions.
 233 
 234   Finally sections are also correctly interleaved between break/continue/return
 235   nodes and their subsequent statements.
 236 
 237   Important concepts:
 238    * nodes - nodes refer refer to CFG nodes; AST nodes are qualified explicitly
 239    * leaf set - since the graph is constructed gradually, a leaf set maintains
 240      the CFG nodes that will precede the node that the builder expects to
 241      receive next; when an ordinary node is added, it is connected to the
 242      existing leaves and it in turn becomes the new leaf
 243    * jump nodes - nodes that should generate edges other than what
 244      ordinary nodes would; these correspond to break, continue and return
 245      statements
 246    * sections - logical delimiters for subgraphs that require special
 247      edges; there are various types of nodes, each admitting various
 248      types of jump nodes; sections are identified by their corresponding AST
 249      node
 250   """
 251 
 252   # TODO(mdan): Perhaps detail this in a markdown doc.
 253   # TODO(mdan): Add exception support.
 254 
 255   def __init__(self, parent_ast_node):
 256     self.reset()
 257     self.parent = parent_ast_node
 258 
 259   def reset(self):
 260     """Resets the state of this factory."""
 261     self.head = None
 262     self.errors = set()
 263     self.node_index = {}
 264 
 265     # TODO(mdan): Too many primitives. Use classes.
 266     self.leaves = set()
 267 
 268     # Note: This mechanism requires that nodes are added in lexical order (top
 269     # to bottom, depth first).
 270     self.active_stmts = set()
 271     self.owners = {}  # type: Set[any]
 272     self.forward_edges = set()  # type: Tuple[Node, Node] # (from, to)
 273 
 274     self.finally_sections = {}
 275     # Dict values represent (entry, exits)
 276     self.finally_section_subgraphs = {
 277     }  # type: Dict[ast.AST, Tuple[Node, Set[Node]]]
 278     # Whether the guard section can be reached from the statement that precedes
 279     # it.
 280     self.finally_section_has_direct_flow = {}
 281     # Finally sections that await their first node.
 282     self.pending_finally_sections = set()
 283 
 284     # Exit jumps keyed by the section they affect.
 285     self.exits = {}
 286 
 287     # The entry of loop sections, keyed by the section.
 288     self.section_entry = {}
 289     # Continue jumps keyed by the section they affect.
 290     self.continues = {}
 291 
 292     # The entry of conditional sections, keyed by the section.
 293     self.cond_entry = {}
 294     # Lists of leaf nodes corresponding to each branch in the section.
 295     self.cond_leaves = {}
 296 
 297   def _connect_nodes(self, first, second):
 298     """Connects nodes to signify that control flows from first to second.
 299 
 300     Args:
 301       first: Union[Set[Node, ...], Node]
 302       second: Node
 303     """
 304     if isinstance(first, Node):
 305       first.next.add(second)
 306       second.prev.add(first)
 307       self.forward_edges.add((first, second))
 308     else:
 309       for node in first:
 310         self._connect_nodes(node, second)
 311 
 312   def _add_new_node(self, ast_node):
 313     """Grows the graph by adding a CFG node following the current leaves."""
 314     if ast_node is self.node_index:
 315       raise ValueError('%s added twice' % ast_node)
 316     # Assumption: All CFG nodes have identical life spans, because the graph
 317     # owns them. Nodes should never be used outside the context of an existing
 318     # graph.
 319     node = Node(next_=set(), prev=weakref.WeakSet(), ast_node=ast_node)
 320     self.node_index[ast_node] = node
 321     self.owners[node] = frozenset(self.active_stmts)
 322 
 323     if self.head is None:
 324       self.head = node
 325 
 326     for leaf in self.leaves:
 327       self._connect_nodes(leaf, node)
 328 
 329     # If any finally section awaits its first node, populate it.
 330     for section_id in self.pending_finally_sections:
 331       self.finally_section_subgraphs[section_id][0] = node
 332     self.pending_finally_sections = set()
 333 
 334     return node
 335 
 336   def begin_statement(self, stmt):
 337     """Marks the beginning of a statement.
 338 
 339     Args:
 340       stmt: Hashable, a key by which the statement can be identified in
 341           the CFG's stmt_prev and stmt_next attributes
 342     """
 343     self.active_stmts.add(stmt)
 344 
 345   def end_statement(self, stmt):
 346     """Marks the end of a statement.
 347 
 348     Args:
 349       stmt: Hashable, a key by which the statement can be identified in
 350           the CFG's stmt_prev and stmt_next attributes; must match a key
 351           previously passed to begin_statement.
 352     """
 353     self.active_stmts.remove(stmt)
 354 
 355   def add_ordinary_node(self, ast_node):
 356     """Grows the graph by adding an ordinary CFG node.
 357 
 358     Ordinary nodes are followed by the next node, in lexical order, that is,
 359     they become the new leaf set.
 360 
 361     Args:
 362       ast_node: ast.AST
 363     Returns:
 364       Node
 365     """
 366     node = self._add_new_node(ast_node)
 367     self.leaves = set((node,))
 368     return node
 369 
 370   def _add_jump_node(self, ast_node, guards):
 371     """Grows the graph by adding a jump node.
 372 
 373     Jump nodes are added to the current leaf set, and the leaf set becomes
 374     empty. If the jump node is the last in a cond section, then it may be added
 375     back to the leaf set by a separate mechanism.
 376 
 377     Args:
 378       ast_node: ast.AST
 379       guards: Tuple[ast.AST, ...], the finally sections active for this node
 380     Returns:
 381       Node
 382     """
 383     node = self._add_new_node(ast_node)
 384     self.leaves = set()
 385     # The guards themselves may not yet be complete, and will be wired later.
 386     self.finally_sections[node] = guards
 387     return node
 388 
 389   def _connect_jump_to_finally_sections(self, node):
 390     """Connects a jump node to the finally sections protecting it."""
 391     cursor = set((node,))
 392     for guard_section_id in self.finally_sections[node]:
 393       guard_begin, guard_ends = self.finally_section_subgraphs[guard_section_id]
 394       self._connect_nodes(cursor, guard_begin)
 395       cursor = guard_ends
 396     del self.finally_sections[node]
 397     # TODO(mdan): Should garbage-collect finally_section_subgraphs.
 398     return cursor
 399 
 400   def add_exit_node(self, ast_node, section_id, guards):
 401     """Grows the graph by adding an exit node.
 402 
 403     This node becomes an exit for the current section.
 404 
 405     Args:
 406       ast_node: ast.AST
 407       section_id: Hashable, the node for which ast_node should be considered
 408           to be an exit node
 409       guards: Tuple[ast.AST, ...], the finally sections that guard ast_node
 410     """
 411     node = self._add_jump_node(ast_node, guards)
 412     self.exits[section_id].add(node)
 413 
 414   def add_continue_node(self, ast_node, section_id, guards):
 415     """Grows the graph by adding a reentry node.
 416 
 417     This node causes control flow to go back to the loop section's entry.
 418 
 419     Args:
 420       ast_node: ast.AST
 421       section_id: Hashable, the node for which ast_node should be considered
 422           to be an exit node
 423       guards: Tuple[ast.AST, ...], the finally sections that guard ast_node
 424     """
 425     node = self._add_jump_node(ast_node, guards)
 426     self.continues[section_id].add(node)
 427 
 428   def add_error_node(self, ast_node, guards):
 429     """Grows the graph by adding an error node.
 430 
 431     This node becomes an exit for the entire graph.
 432 
 433     Args:
 434       ast_node: ast.AST
 435       guards: Tuple[ast.AST, ...], the finally sections that guard ast_node
 436     """
 437     node = self._add_jump_node(ast_node, guards)
 438     self.errors.add(node)
 439     self.leaves = set()
 440 
 441   def enter_section(self, section_id):
 442     """Enters a regular section.
 443 
 444     Regular sections admit exit jumps, which end the section.
 445 
 446     Args:
 447       section_id: Hashable, the same node that will be used in calls to the
 448           ast_node arg passed to add_exit_node
 449     """
 450     assert section_id not in self.exits
 451     self.exits[section_id] = set()
 452 
 453   def exit_section(self, section_id):
 454     """Exits a regular section."""
 455 
 456     # Exits are jump nodes, which may be protected.
 457     for exit_ in self.exits[section_id]:
 458       self.leaves |= self._connect_jump_to_finally_sections(exit_)
 459 
 460     del self.exits[section_id]
 461 
 462   def enter_loop_section(self, section_id, entry_node):
 463     """Enters a loop section.
 464 
 465     Loop sections define an entry node. The end of the section always flows back
 466     to the entry node. These admit continue jump nodes which also flow to the
 467     entry node.
 468 
 469     Args:
 470       section_id: Hashable, the same node that will be used in calls to the
 471           ast_node arg passed to add_continue_node
 472       entry_node: ast.AST, the entry node into the loop (e.g. the test node
 473           for while loops)
 474     """
 475     assert section_id not in self.section_entry
 476     assert section_id not in self.continues
 477     self.continues[section_id] = set()
 478     node = self.add_ordinary_node(entry_node)
 479     self.section_entry[section_id] = node
 480 
 481   def exit_loop_section(self, section_id):
 482     """Exits a loop section."""
 483     self._connect_nodes(self.leaves, self.section_entry[section_id])
 484 
 485     # continues are jump nodes, which may be protected.
 486     for reentry in self.continues[section_id]:
 487       guard_ends = self._connect_jump_to_finally_sections(reentry)
 488       self._connect_nodes(guard_ends, self.section_entry[section_id])
 489 
 490     # Loop nodes always loop back.
 491     self.leaves = set((self.section_entry[section_id],))
 492 
 493     del self.continues[section_id]
 494     del self.section_entry[section_id]
 495 
 496   def enter_cond_section(self, section_id):
 497     """Enters a conditional section.
 498 
 499     Conditional sections define an entry node, and one or more branches.
 500 
 501     Args:
 502       section_id: Hashable, the same node that will be used in calls to the
 503           section_id arg passed to new_cond_branch
 504     """
 505 
 506     assert section_id not in self.cond_entry
 507     assert section_id not in self.cond_leaves
 508     self.cond_leaves[section_id] = []
 509 
 510   def new_cond_branch(self, section_id):
 511     """Begins a new branch in a cond section."""
 512     assert section_id in self.cond_leaves
 513 
 514     if section_id in self.cond_entry:
 515       # Subsequent splits move back to the split point, and memorize the
 516       # current leaves.
 517       self.cond_leaves[section_id].append(self.leaves)
 518       self.leaves = self.cond_entry[section_id]
 519     else:
 520       # If this is the first time we split a section, just remember the split
 521       # point.
 522       self.cond_entry[section_id] = self.leaves
 523 
 524   def exit_cond_section(self, section_id):
 525     """Exits a conditional section."""
 526     for split in self.cond_leaves[section_id]:
 527       self.leaves |= split
 528     del self.cond_entry[section_id]
 529     del self.cond_leaves[section_id]
 530 
 531   def enter_finally_section(self, section_id):
 532     """Enters a finally section."""
 533     # TODO(mdan): This, not the caller, should track the active sections.
 534     self.finally_section_subgraphs[section_id] = [None, None]
 535     if self.leaves:
 536       self.finally_section_has_direct_flow[section_id] = True
 537     else:
 538       self.finally_section_has_direct_flow[section_id] = False
 539     self.pending_finally_sections.add(section_id)
 540 
 541   def exit_finally_section(self, section_id):
 542     """Exits a finally section."""
 543     assert section_id not in self.pending_finally_sections, 'Empty finally?'
 544     self.finally_section_subgraphs[section_id][1] = self.leaves
 545     # If the guard can only be reached by a jump, then it will not flow
 546     # into the statement that follows it.
 547     if not self.finally_section_has_direct_flow[section_id]:
 548       self.leaves = set()
 549     del self.finally_section_has_direct_flow[section_id]
 550 
 551   def build(self):
 552     """Returns the CFG accumulated so far and resets the builder.
 553 
 554     Returns:
 555       Graph
 556     """
 557     # Freeze the nodes.
 558     for node in self.node_index.values():
 559       node.freeze()
 560 
 561     # Build the statement edges.
 562     stmt_next = {}
 563     stmt_prev = {}
 564     for node, _ in self.forward_edges:
 565       for stmt in self.owners[node]:
 566         if stmt not in stmt_next:
 567           stmt_next[stmt] = set()
 568         if stmt not in stmt_prev:
 569           stmt_prev[stmt] = set()
 570     for first, second in self.forward_edges:
 571       stmts_exited = self.owners[first] - self.owners[second]
 572       for stmt in stmts_exited:
 573         stmt_next[stmt].add(second)
 574       stmts_entered = self.owners[second] - self.owners[first]
 575       for stmt in stmts_entered:
 576         stmt_prev[stmt].add(first)
 577     for stmt in stmt_next:
 578       stmt_next[stmt] = frozenset(stmt_next[stmt])
 579     for stmt in stmt_prev:
 580       stmt_prev[stmt] = frozenset(stmt_prev[stmt])
 581 
 582     # Construct the final graph object.
 583     result = Graph(
 584         entry=self.head,
 585         exit=self.leaves,
 586         error=self.errors,
 587         index=self.node_index,
 588         stmt_prev=stmt_prev,
 589         stmt_next=stmt_next)
 590 
 591     # Reset the state.
 592     self.reset()
 593 
 594     return result

 597 class AstToCfg(gast.NodeVisitor):
 598   """Converts an AST to CFGs.
 599 
 600   A separate CFG will be constructed for each function.
 601   """
 602 
 603   def __init__(self):
 604     super(AstToCfg, self).__init__()
 605 
 606     self.builder_stack = []
 607     self.builder = None
 608     self.cfgs = {}
 609 
 610     self.lexical_scopes = []
 611 
 612   def _enter_lexical_scope(self, node):
 613     self.lexical_scopes.append(node)
 614 
 615   def _exit_lexical_scope(self, node):
 616     leaving_node = self.lexical_scopes.pop()
 617     assert node == leaving_node
 618 
 619   def _get_enclosing_scopes(self, include, stop_at):
 620     included = []
 621     for node in reversed(self.lexical_scopes):
 622       if isinstance(node, include):
 623         included.append(node)
 624       if isinstance(node, stop_at):
 625         return node, included
 626     return None, included
 627 
 628   def _process_basic_statement(self, node):
 629     self.generic_visit(node)
 630     self.builder.add_ordinary_node(node)
 631 
 632   def _process_exit_statement(self, node, *exits_nodes_of_type):
 633     # Note: this is safe because we process functions separately.
 634     try_node, guards = self._get_enclosing_scopes(
 635         include=(gast.Try,),
 636         stop_at=tuple(exits_nodes_of_type),
 637     )
 638     if try_node is None:
 639       raise ValueError(
 640           '%s that is not enclosed by any of %s' % (node, exits_nodes_of_type))
 641     self.builder.add_exit_node(node, try_node, guards)
 642 
 643   def _process_continue_statement(self, node, *loops_to_nodes_of_type):
 644     # Note: this is safe because we process functions separately.
 645     try_node, guards = self._get_enclosing_scopes(
 646         include=(gast.Try,),
 647         stop_at=tuple(loops_to_nodes_of_type),
 648     )
 649     if try_node is None:
 650       raise ValueError('%s that is not enclosed by any of %s' %
 651                        (node, loops_to_nodes_of_type))
 652     self.builder.add_continue_node(node, try_node, guards)

 654   def visit_FunctionDef(self, node):
 655     # We also keep the FunctionDef node in the CFG. This allows us to determine
 656     # things like reaching definitions via closure. Note that the function body
 657     # will be stored in a separate graph, because function definitions are not
 658     # the same as function calls.
 659     if self.builder is not None:
 660       self.builder.add_ordinary_node(node)
 661 
 662     self.builder_stack.append(self.builder)
 663     self.builder = GraphBuilder(node)
 664 
 665     self._enter_lexical_scope(node)
 666     self.builder.enter_section(node)
 667 
 668     self._process_basic_statement(node.args)
 669     for stmt in node.body:
 670       self.visit(stmt)
 671 
 672     self.builder.exit_section(node)
 673     self._exit_lexical_scope(node)
 674 
 675     self.cfgs[node] = self.builder.build()
 676     self.builder = self.builder_stack.pop()
 677

 700   def visit_Raise(self, node):
 701     try_node, guards = self._get_enclosing_scopes(
 702         include=(gast.Try,),
 703         stop_at=(gast.FunctionDef,),

 716   def visit_If(self, node):
 717     # No need to track ifs as lexical scopes, for now.
 718     # Lexical scopes are generally tracked in order to be able to resolve the
 719     # targets of jump statements like break/continue/etc. Since there is no
 720     # statement that can interrupt a conditional, we don't need to track their
 721     # lexical scope. That may change in the future.
 722     self.builder.begin_statement(node)
 723 
 724     self.builder.enter_cond_section(node)
 725     self._process_basic_statement(node.test)
 726 
 727     self.builder.new_cond_branch(node)
 728     for stmt in node.body:
 729       self.visit(stmt)
 730 
 731     self.builder.new_cond_branch(node)
 732     for stmt in node.orelse:
 733       self.visit(stmt)
 734 
 735     self.builder.exit_cond_section(node)
 736     self.builder.end_statement(node)
 737 
 738   def visit_While(self, node):
 739     self.builder.begin_statement(node)
 740     self._enter_lexical_scope(node)
 741 
 742     self.builder.enter_section(node)
 743 
 744     self.builder.enter_loop_section(node, node.test)
 745     for stmt in node.body:
 746       self.visit(stmt)
 747     self.builder.exit_loop_section(node)
 748 
 749     # Note: although the orelse is technically part of the loop node,
 750     # the statements inside it don't affect the loop itself. For example, a
 751     # break in the loop's orelse will not affect the loop itself.
 752     self._exit_lexical_scope(node)
 753 
 754     for stmt in node.orelse:
 755       self.visit(stmt)
 756 
 757     self.builder.exit_section(node)
 758     self.builder.end_statement(node)
 759 
 760   def visit_For(self, node):
 761     self.builder.begin_statement(node)
 762     self._enter_lexical_scope(node)
 763 
 764     self.builder.enter_section(node)
 765 
 766     # TODO(mdan): Strictly speaking, this should be node.target + node.iter.
 767     # A blind dataflow analysis would have to process both node.target and
 768     # node.iter to properly process read and write access.
 769     self.builder.enter_loop_section(node, node.iter)
 770     for stmt in node.body:
 771       self.visit(stmt)
 772     self.builder.exit_loop_section(node)
 773 
 774     # Note: although the orelse is technically part of the loop node,
 775     # they don't count as loop bodies.  For example, a break in the loop's
 776     # orelse will affect the parent loop, not the current one.
 777     self._exit_lexical_scope(node)
 778 
 779     for stmt in node.orelse:
 780       self.visit(stmt)
 781 
 782     self.builder.exit_section(node)
 783     self.builder.end_statement(node)
 784 
 785   def visit_Break(self, node):
 786     self._process_exit_statement(node, gast.While, gast.For)
 787 
 788   def visit_Continue(self, node):
 789     self._process_continue_statement(node, gast.While, gast.For)
 790 
 791   def visit_Try(self, node):
 792     self._enter_lexical_scope(node)
 793 
 794     for stmt in node.body:
 795       self.visit(stmt)
 796     # Unlike loops, the orelse is a simple continuation of the body.
 797     for stmt in node.orelse:
 798       self.visit(stmt)
 799 
 800     if node.handlers:
 801       # TODO(mdan): Should we still support bare try/except? Might be confusing.
 802       raise NotImplementedError('exceptions are not yet supported')
 803 
 804     self._exit_lexical_scope(node)
 805 
 806     self.builder.enter_finally_section(node)
 807     for stmt in node.finalbody:
 808       self.visit(stmt)
 809     self.builder.exit_finally_section(node)
 810 
 811   def visit_With(self, node):
 812     # TODO(mdan): Mark the context manager's exit call as exit guard.
 813     for item in node.items:
 814       self._process_basic_statement(item)
 815     for stmt in node.body:
 816       self.visit(stmt)
