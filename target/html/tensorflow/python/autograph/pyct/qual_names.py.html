  15 """Utilities for manipulating qualified names.
  16 
  17 A qualified name is a uniform way to refer to simple (e.g. 'foo') and composite
  18 (e.g. 'foo.bar') syntactic symbols.
  19 
  20 This is *not* related to the __qualname__ attribute used by inspect, which
  21 refers to scopes.
  22 """

  36 class Symbol(collections.namedtuple('Symbol', ['name'])):
  37   """Represents a Python symbol."""

  40 class StringLiteral(collections.namedtuple('StringLiteral', ['value'])):
  41   """Represents a Python string literal."""
  42 
  43   def __str__(self):
  44     return '\'%s\'' % self.value
  45 
  46   def __repr__(self):
  47     return str(self)
  48 
  49 
  50 class NumberLiteral(collections.namedtuple('NumberLiteral', ['value'])):
  51   """Represents a Python numeric literal."""
  52 
  53   def __str__(self):
  54     return '%s' % self.value
  55 
  56   def __repr__(self):
  57     return str(self)

  60 # TODO(mdan): Use subclasses to remove the has_attr has_subscript booleans.
  61 class QN(object):
  62   """Represents a qualified name."""
  63 
  64   def __init__(self, base, attr=None, subscript=None):
  65     if attr is not None and subscript is not None:
  66       raise ValueError('A QN can only be either an attr or a subscript, not '
  67                        'both: attr={}, subscript={}.'.format(attr, subscript))
  68     self._has_attr = False
  69     self._has_subscript = False
  70 
  71     if attr is not None:
  72       if not isinstance(base, QN):
  73         raise ValueError(
  74             'for attribute QNs, base must be a QN; got instead "%s"' % base)
  75       if not isinstance(attr, str):
  76         raise ValueError('attr may only be a string; got instead "%s"' % attr)
  77       self._parent = base
  78       # TODO(mdan): Get rid of the tuple - it can only have 1 or 2 elements now.
  79       self.qn = (base, attr)
  80       self._has_attr = True
  81 
  82     elif subscript is not None:
  83       if not isinstance(base, QN):
  84         raise ValueError('For subscript QNs, base must be a QN.')
  85       self._parent = base
  86       self.qn = (base, subscript)
  87       self._has_subscript = True
  88 
  89     else:
  90       if not isinstance(base, (str, StringLiteral, NumberLiteral)):
  91         # TODO(mdan): Require Symbol instead of string.
  92         raise ValueError(
  93             'for simple QNs, base must be a string or a Literal object;'
  94             ' got instead "%s"' % type(base))
  95       assert '.' not in base and '[' not in base and ']' not in base
  96       self._parent = None
  97       self.qn = (base,)
  98 
  99   def is_symbol(self):
 100     return isinstance(self.qn[0], str)
 101 
 102   def is_composite(self):
 103     return len(self.qn) > 1
 104 
 105   def has_subscript(self):
 106     return self._has_subscript
 107 
 108   def has_attr(self):
 109     return self._has_attr
 110 
 111   @property
 112   def parent(self):
 113     if self._parent is None:
 114       raise ValueError('Cannot get parent of simple name "%s".' % self.qn[0])
 115     return self._parent
 116 
 117   @property
 118   def owner_set(self):
 119     """Returns all the symbols (simple or composite) that own this QN.
 120 
 121     In other words, if this symbol was modified, the symbols in the owner set
 122     may also be affected.
 123 
 124     Examples:
 125       'a.b[c.d]' has two owners, 'a' and 'a.b'
 126     """
 127     owners = set()
 128     if self.has_attr() or self.has_subscript():
 129       owners.add(self.parent)
 130       owners.update(self.parent.owner_set)
 131     return owners
 132 
 133   @property
 134   def support_set(self):
 135     """Returns the set of simple symbols that this QN relies on.
 136 
 137     This would be the smallest set of symbols necessary for the QN to
 138     statically resolve (assuming properties and index ranges are verified
 139     at runtime).
 140 
 141     Examples:
 142       'a.b' has only one support symbol, 'a'
 143       'a[i]' has two support symbols, 'a' and 'i'
 144     """
 145     # TODO(mdan): This might be the set of Name nodes in the AST. Track those?
 146     roots = set()
 147     if self.has_attr():
 148       roots.update(self.parent.support_set)
 149     elif self.has_subscript():
 150       roots.update(self.parent.support_set)
 151       roots.update(self.qn[1].support_set)
 152     else:
 153       roots.add(self)
 154     return roots
 155 
 156   def __hash__(self):
 157     return hash(self.qn + (self._has_attr, self._has_subscript))
 158 
 159   def __eq__(self, other):
 160     return (isinstance(other, QN) and self.qn == other.qn and
 161             self.has_subscript() == other.has_subscript() and
 162             self.has_attr() == other.has_attr())
 163 
 164   def __str__(self):
 165     if self.has_subscript():
 166       return str(self.qn[0]) + '[' + str(self.qn[1]) + ']'
 167     if self.has_attr():
 168       return '.'.join(map(str, self.qn))
 169     else:
 170       return str(self.qn[0])
 171 
 172   def __repr__(self):
 173     return str(self)
 174 
 175   def ssf(self):
 176     """Simple symbol form."""
 177     ssfs = [n.ssf() if isinstance(n, QN) else n for n in self.qn]
 178     ssf_string = ''
 179     for i in range(0, len(self.qn) - 1):
 180       if self.has_subscript():
 181         delimiter = '_sub_'
 182       else:
 183         delimiter = '_'
 184       ssf_string += ssfs[i] + delimiter
 185     return ssf_string + ssfs[-1]
 186 
 187   def ast(self):
 188     # The caller must adjust the context appropriately.
 189     if self.has_subscript():
 190       return gast.Subscript(self.parent.ast(), gast.Index(self.qn[-1].ast()),
 191                             None)
 192     if self.has_attr():
 193       return gast.Attribute(self.parent.ast(), self.qn[-1], None)
 194 
 195     base = self.qn[0]
 196     if isinstance(base, str):
 197       return gast.Name(base, None, None)
 198     elif isinstance(base, StringLiteral):
 199       return gast.Str(base.value)
 200     elif isinstance(base, NumberLiteral):
 201       return gast.Num(base.value)
 202     else:
 203       assert False, ('the constructor should prevent types other than '
 204                      'str, StringLiteral and NumberLiteral')

 207 class QnResolver(gast.NodeTransformer):
 208   """Annotates nodes with QN information.
 209 
 210   Note: Not using NodeAnnos to avoid circular dependencies.
 211   """
 212 
 213   def visit_Name(self, node):
 214     node = self.generic_visit(node)
 215     anno.setanno(node, anno.Basic.QN, QN(node.id))
 216     return node
 217 
 218   def visit_Attribute(self, node):
 219     node = self.generic_visit(node)
 220     if anno.hasanno(node.value, anno.Basic.QN):
 221       anno.setanno(node, anno.Basic.QN,
 222                    QN(anno.getanno(node.value, anno.Basic.QN), attr=node.attr))
 223     return node
 224 
 225   def visit_Subscript(self, node):
 226     # TODO(mdan): This may no longer apply if we overload getitem.
 227     node = self.generic_visit(node)
 228     s = node.slice
 229     if not isinstance(s, gast.Index):
 230       # TODO(mdan): Support range and multi-dimensional indices.
 231       # Continuing silently because some demos use these.
 232       return node
 233     if isinstance(s.value, gast.Num):
 234       subscript = QN(NumberLiteral(s.value.n))
 235     elif isinstance(s.value, gast.Str):
 236       subscript = QN(StringLiteral(s.value.s))
 237     else:
 238       # The index may be an expression, case in which a name doesn't make sense.
 239       if anno.hasanno(node.slice.value, anno.Basic.QN):
 240         subscript = anno.getanno(node.slice.value, anno.Basic.QN)
 241       else:
 242         return node
 243     if anno.hasanno(node.value, anno.Basic.QN):
 244       anno.setanno(node, anno.Basic.QN,
 245                    QN(anno.getanno(node.value, anno.Basic.QN),
 246                       subscript=subscript))
 247     return node
 248 
 249 
 250 def resolve(node):
 251   return QnResolver().visit(node)
 252 
 253 
 254 def from_str(qn_str):
 255   node = parser.parse_expression(qn_str)
 256   node = resolve(node)
 257   return anno.getanno(node, anno.Basic.QN)
