  32 def isbuiltin(f):
  33   # Note these return false for isinstance(f, types.BuiltinFunctionType) so we
  34   # need to specifically check for them.
  35   if f in (range, int, float):
  36     return True
  37   if isinstance(f, types.BuiltinFunctionType):
  38     return True
  39   if tf_inspect.isbuiltin(f):
  40     return True
  41   return False

  44 def getnamespace(f):
  45   """Returns the complete namespace of a function.
  46 
  47   Namespace is defined here as the mapping of all non-local variables to values.
  48   This includes the globals and the closure variables. Note that this captures
  49   the entire globals collection of the function, and may contain extra symbols
  50   that it does not actually use.
  51 
  52   Args:
  53     f: User defined function.
  54   Returns:
  55     A dict mapping symbol names to values.
  56   """
  57   namespace = dict(six.get_function_globals(f))
  58   closure = six.get_function_closure(f)
  59   freevars = six.get_function_code(f).co_freevars
  60   if freevars and closure:
  61     for name, cell in zip(freevars, closure):
  62       namespace[name] = cell.cell_contents
  63   return namespace

  66 def _get_unbound_function(m):
  67   # TODO(mdan): Figure out why six.get_unbound_function fails in some cases.
  68   # The failure case is for tf.keras.Model.
  69   if hasattr(m, 'im_func'):
  70     return m.im_func
  71   return m

  74 def getdefiningclass(m, owner_class):
  75   """Resolves the class (e.g. one of the superclasses) that defined a method."""
  76   # Normalize bound functions to their respective unbound versions.
  77   m = _get_unbound_function(m)
  78   for superclass in owner_class.__bases__:
  79     if hasattr(superclass, m.__name__):
  80       superclass_m = getattr(superclass, m.__name__)
  81       if _get_unbound_function(superclass_m) is m:
  82         return superclass
  83       elif hasattr(m, '__self__') and m.__self__ == owner_class:
  84         # Python 3 class methods only work this way it seems :S
  85         return superclass
  86   return owner_class

  89 def getmethodclass(m):
  90   """Resolves a function's owner, e.g. a method's class.
  91 
  92   Note that this returns the object that the function was retrieved from, not
  93   necessarily the class where it was defined.
  94 
  95   This function relies on Python stack frame support in the interpreter, and
  96   has the same limitations that inspect.currentframe.
  97 
  98   Limitations. This function will only work correctly if the owned class is
  99   visible in the caller's global or local variables.
 100 
 101   Args:
 102     m: A user defined function
 103 
 104   Returns:
 105     The class that this function was retrieved from, or None if the function
 106     is not an object or class method, or the class that owns the object or
 107     method is not visible to m.
 108 
 109   Raises:
 110     ValueError: if the class could not be resolved for any unexpected reason.
 111   """
 112 
 113   # Callable objects: return their own class.
 114   if (not hasattr(m, '__name__') and hasattr(m, '__class__') and
 115       hasattr(m, '__call__')):
 116     if isinstance(m.__class__, six.class_types):
 117       return m.__class__
 118 
 119   # Instance method and class methods: should be bound to a non-null "self".
 120   # If self is a class, then it's a class method.
 121   if hasattr(m, '__self__'):
 122     if m.__self__:
 123       if tf_inspect.isclass(m.__self__):
 124         return m.__self__
 125       return type(m.__self__)
 126 
 127   # Class, static and unbound methods: search all defined classes in any
 128   # namespace. This is inefficient but more robust method.
 129   owners = []
 130   caller_frame = tf_inspect.currentframe().f_back
 131   try:
 132     # TODO(mdan): This doesn't consider cell variables.
 133     # TODO(mdan): This won't work if the owner is hidden inside a container.
 134     # Cell variables may be pulled using co_freevars and the closure.
 135     for v in itertools.chain(caller_frame.f_locals.values(),
 136                              caller_frame.f_globals.values()):
 137       if hasattr(v, m.__name__):
 138         candidate = getattr(v, m.__name__)
 139         # Py2 methods may be bound or unbound, extract im_func to get the
 140         # underlying function.
 141         if hasattr(candidate, 'im_func'):
 142           candidate = candidate.im_func
 143         if hasattr(m, 'im_func'):
 144           m = m.im_func
 145         if candidate is m:
 146           owners.append(v)
 147   finally:
 148     del caller_frame
 149 
 150   if owners:
 151     if len(owners) == 1:
 152       return owners[0]
 153 
 154     # If multiple owners are found, and are not subclasses, raise an error.
 155     owner_types = tuple(o if tf_inspect.isclass(o) else type(o) for o in owners)
 156     for o in owner_types:
 157       if tf_inspect.isclass(o) and issubclass(o, tuple(owner_types)):
 158         return o
 159     raise ValueError('Found too many owners of %s: %s' % (m, owners))
 160 
 161   return None
