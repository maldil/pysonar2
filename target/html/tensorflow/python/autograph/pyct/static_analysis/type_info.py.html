  15 """Type resolution.
  16 
  17 This analyzer uses known live values to further infer object types. This
  18 may include for instance constructed objects and object member functions.
  19 
  20 In addition, the analyzer also handles user annotations made in the code (for
  21 example, the autograph.set_element_type function).
  22 
  23 Requires annotations generated by LiveValuesResolver.
  24 """

  56 class Scope(object):
  57   """Tracks symbol value references.
  58 
  59   Attributes:
  60     values: A dict mapping string to gast.Node, containing the value that was
  61         most recently assigned to the symbol.
  62   """
  63 
  64   def __init__(self, parent):
  65     """Create a new scope.
  66 
  67     Args:
  68       parent: A Scope or None.
  69     """
  70     self.parent = parent
  71     self.values = {}
  72 
  73   def __repr__(self):
  74     return 'Scope[%s]' % self.values.keys()
  75 
  76   def copy(self):
  77     s = Scope(self.parent)
  78     s.values = self.values.copy()
  79     return s
  80 
  81   def setval(self, name, value):
  82     self.values[name] = value
  83 
  84   def hasval(self, name):
  85     return (name in self.values or
  86             (self.parent is not None and self.parent.hasval(name)))
  87 
  88   def getval(self, name):
  89     if name in self.values:
  90       return self.values[name]
  91     if self.parent is not None:
  92       return self.parent.getval(name)
  93     raise KeyError(name)

  96 class TypeInfoResolver(transformer.Base):
  97   """Annotates symbols with type information where possible.
  98 
  99   Nodes currently annotated:
 100     * Call (helps detect class constructors)
 101     * Attribute (helps resolve object methods)
 102   """
 103 
 104   def __init__(self, context):
 105     super(TypeInfoResolver, self).__init__(context)
 106     self.scope = Scope(None)
 107 
 108   def visit_FunctionDef(self, node):
 109     self.scope = Scope(self.scope)
 110     node = self.generic_visit(node)
 111     self.scope = self.scope.parent
 112     return node
 113 
 114   def _visit_block(self, block):
 115     self.scope = Scope(self.scope)
 116     block = self.visit_block(block)
 117     self.scope = self.scope.parent
 118     return block
 119 
 120   def visit_For(self, node):
 121     self.generic_visit(node.target)
 122     self.generic_visit(node.iter)
 123     node.body = self._visit_block(node.body)
 124     node.orelse = self._visit_block(node.orelse)
 125     return node
 126 
 127   def visit_While(self, node):
 128     self.generic_visit(node.test)
 129     node.body = self._visit_block(node.body)
 130     node.orelse = self._visit_block(node.orelse)
 131     return node
 132 
 133   def visit_If(self, node):
 134     self.generic_visit(node.test)
 135     node.body = self._visit_block(node.body)
 136     node.orelse = self._visit_block(node.orelse)
 137     return node
 138 
 139   def _process_function_arg(self, arg_node):
 140     qn = anno.getanno(arg_node, anno.Basic.QN)
 141     arg_name = str(qn)
 142     self.scope.setval(qn, arg_node)
 143     if (len(self.enclosing_entities) == 1 and
 144         arg_name in self.entity_info.arg_types):
 145       # Forge a node to hold the type information, so that method calls on
 146       # it can resolve the type.
 147       type_string, type_obj = self.entity_info.arg_types[arg_name]
 148       anno.setanno(arg_node, 'type', type_obj)
 149       anno.setanno(arg_node, 'type_fqn', tuple(type_string.split('.')))
 150 
 151   def visit_arg(self, node):
 152     self._process_function_arg(node.arg)
 153     return node
 154 
 155   def visit_Name(self, node):
 156     self.generic_visit(node)
 157     if isinstance(node.ctx, gast.Param):
 158       self._process_function_arg(node)
 159     elif isinstance(node.ctx, gast.Load):
 160       qn = anno.getanno(node, anno.Basic.QN)
 161       if self.scope.hasval(qn):
 162         # E.g. if we had
 163         # a = b
 164         # then for future references to `a` we should have definition = `b`
 165         definition = self.scope.getval(qn)
 166         anno.copyanno(definition, node, 'type')
 167         anno.copyanno(definition, node, 'type_fqn')
 168 
 169         # TODO(mdan): Remove this when the directives module is in.
 170         anno.copyanno(definition, node, 'element_type')
 171         anno.copyanno(definition, node, 'element_shape')
 172     return node
 173 
 174   def _process_variable_assignment(self, target, value):
 175     # Constructors
 176     if isinstance(value, gast.Call):
 177       func = value.func
 178       if anno.hasanno(func, 'live_val'):
 179         func_obj = anno.getanno(func, 'live_val')
 180         if tf_inspect.isclass(func_obj):
 181           anno.setanno(value, 'is_constructor', True)
 182           anno.setanno(value, 'type', func_obj)
 183           anno.setanno(value, 'type_fqn', anno.getanno(func, 'fqn'))
 184           # TODO(mdan): Raise an error if constructor has side effects.
 185           # We can have a whitelist of no-side-effects constructors.
 186           # We can also step inside the constructor and further analyze.
 187 
 188     if isinstance(target, (gast.Name, gast.Attribute)):
 189       target_symbol = anno.getanno(target, anno.Basic.QN)
 190       self.scope.setval(target_symbol, value)
 191     elif isinstance(target, gast.Subscript):
 192       pass
 193     else:
 194       raise ValueError('assignment target has unknown type: %s' % target)
 195 
 196   def visit_With(self, node):
 197     for item in node.items:
 198       if item.optional_vars is not None:
 199         ast_util.apply_to_single_assignments((item.optional_vars,),
 200                                              item.context_expr,
 201                                              self._process_variable_assignment)
 202     self.generic_visit(node)
 203     return node
 204 
 205   def visit_Assign(self, node):
 206     self.generic_visit(node)
 207     ast_util.apply_to_single_assignments(node.targets, node.value,
 208                                          self._process_variable_assignment)
 209     return node
 210 
 211 
 212 def resolve(node, context):
 213   return TypeInfoResolver(context).visit(node)
