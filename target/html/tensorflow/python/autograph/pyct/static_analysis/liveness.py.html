  15 """Live variable analysis.
  16 
  17 This analysis attaches a set containing the live symbols that are live at the
  18 exit of control flow statements.

  20 Requires activity analysis.
  21 """

  35 class Analyzer(cfg.GraphVisitor):
  36   """CFG visitor that performs liveness analysis at statement level."""
  37 
  38   def __init__(self, graph):
  39     super(Analyzer, self).__init__(graph)
  40     # This allows communicating that nodes generate extra symbols,
  41     # e.g. those that a function definition closes over.
  42     self.extra_gen = {}
  43 
  44   def init_state(self, _):
  45     return set()
  46 
  47   def visit_node(self, node):
  48     prev_live_in = self.in_[node]
  49 
  50     if anno.hasanno(node.ast_node, anno.Static.SCOPE):
  51       node_scope = anno.getanno(node.ast_node, anno.Static.SCOPE)
  52 
  53       gen = node_scope.used | self.extra_gen.get(node.ast_node, frozenset())
  54       # TODO(mdan): verify whether composites' parents need to be added.
  55       # E.g. if x.y is live whether x needs to be added. Theoretically the
  56       # activity analysis should have both so that wouldn't be needed.
  57       kill = node_scope.modified
  58 
  59       live_out = set()
  60       for n in node.next:
  61         live_out |= self.in_[n]
  62       live_in = gen | (live_out - kill)
  63 
  64     else:
  65       # Nodes that don't have a scope annotation are assumed not to touch any
  66       # symbols.
  67       # This Name node below is a literal name, e.g. False
  68       assert isinstance(node.ast_node,
  69                         (gast.Name, gast.Continue, gast.Break)), type(
  70                             node.ast_node)
  71       live_in = prev_live_in
  72       live_out = live_in
  73 
  74     self.in_[node] = live_in
  75     self.out[node] = live_out
  76 
  77     # TODO(mdan): Move this to the superclass?
  78     return prev_live_in != live_in
  79

  81 class WholeTreeAnalyzer(transformer.Base):
  82   """Runs liveness analysis on each of the functions defined in the AST.
  83 
  84   If a function defined other local functions, those will have separate CFGs.
  85   However, dataflow analysis needs to tie up these CFGs to properly emulate the
  86   effect of closures. In the case of liveness, the parent function's live
  87   variables must account for the variables that are live at the entry of each
  88   subfunction. For example:
  89 
  90     def foo():
  91       # baz is live here
  92       def bar():
  93         print(baz)
  94 
  95   This analyzer runs liveness analysis on each individual function, accounting
  96   for the effect above.
  97   """
  98 
  99   def __init__(self, source_info, graphs):
 100     super(WholeTreeAnalyzer, self).__init__(source_info)
 101     self.graphs = graphs
 102     self.current_analyzer = None
 103     self.analyzers = {}
 104 
 105   def visit_FunctionDef(self, node):
 106     parent_analyzer = self.current_analyzer
 107     subgraph = self.graphs[node]
 108 
 109     # Postorder tree processing makes this a bit complicated:
 110     #  1. construct an analyzer object and put it on stack
 111     #  2. recursively walk the subtree; this will initialize the analyzer's
 112     #     in_ state properly (done in a block below)
 113     #  3. run the final analysis
 114     analyzer = Analyzer(subgraph)
 115     self.current_analyzer = analyzer
 116     node = self.generic_visit(node)
 117     analyzer.visit_reverse()
 118 
 119     if parent_analyzer is not None:
 120       # Wire the state between the two subgraphs' analyzers.
 121       child_in_state = analyzer.in_[subgraph.entry]
 122       # Exception: symbols modified in the child function are local to it
 123       body_scope = anno.getanno(node, annos.NodeAnno.BODY_SCOPE)
 124       for qn in body_scope.modified:
 125         # Note: a function modifying the symbol doesn't make that symbol
 126         # live at the function's entry. In fact when that happens it is
 127         # probably a case of undefined assignment, like this:
 128         #
 129         #   bar = 0
 130         #   def foo():
 131         #     print(bar)  # bar is undefined here!
 132         #     bar = 1
 133         #
 134         # Hence we use discard and not remove below.
 135         child_in_state.discard(qn)
 136       parent_analyzer.extra_gen[node] = frozenset(child_in_state,)
 137 
 138     self.analyzers[node] = analyzer
 139     self.current_analyzer = parent_analyzer
 140     return node
 141 
 142   def visit_nonlocal(self, node):
 143     raise NotImplementedError()
 144 
 145   def visit_global(self, node):
 146     raise NotImplementedError()

 149 class Annotator(transformer.Base):
 150   """AST visitor that annotates each control flow block with live symbols."""
 151 
 152   # Note: additional nodes may be added as needed.
 153 
 154   def __init__(self, source_info, cross_function_analyzer):
 155     super(Annotator, self).__init__(source_info)
 156     self.cross_function_analyzer = cross_function_analyzer
 157     self.current_analyzer = None
 158 
 159   def visit_FunctionDef(self, node):
 160     parent_analyzer = self.current_analyzer
 161     self.current_analyzer = self.cross_function_analyzer.analyzers[node]
 162 
 163     node = self.generic_visit(node)
 164     self.current_analyzer = parent_analyzer
 165     return node
 166 
 167   def _aggregate_successors_live_in(self, node):
 168     successors = self.current_analyzer.graph.stmt_next[node]
 169     node_live_out = set()
 170     for s in successors:
 171       node_live_out.update(self.current_analyzer.in_[s])
 172     anno.setanno(node, anno.Static.LIVE_VARS_OUT, frozenset(node_live_out))
 173     node = self.generic_visit(node)
 174     return node
 175 
 176   def visit_If(self, node):
 177     return self._aggregate_successors_live_in(node)
 178 
 179   def visit_For(self, node):
 180     return self._aggregate_successors_live_in(node)
 181 
 182   def visit_While(self, node):
 183     return self._aggregate_successors_live_in(node)

 186 def resolve(node, source_info, graphs):
 187   """Resolves the live symbols at the exit of control flow statements.
 188 
 189   Args:
 190     node: ast.AST
 191     source_info: transformer.SourceInfo
 192     graphs: Dict[ast.FunctionDef, cfg.Graph]
 193   Returns:
 194     ast.AST
 195   """
 196   cross_function_analyzer = WholeTreeAnalyzer(source_info, graphs)
 197   node = cross_function_analyzer.visit(node)
 198   visitor = Annotator(source_info, cross_function_analyzer)
 199   node = visitor.visit(node)
 200   return node
