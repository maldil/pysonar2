  15 """Reaching definition analysis.
  16 
  17 This analysis attaches a set of a Definition objects to each symbol, one
  18 for each distinct definition that may reach it. The Definition objects are
  19 mutable and may be used by subsequent analyses to further annotate data like
  20 static type and value information.
  21 The analysis also attaches the set of the symbols defined at the entry of
  22 control flow statements.
  23 
  24 Requires activity analysis.
  25 """

  39 class Definition(object):
  40   """Definition objects describe a unique definition of a variable.
  41 
  42   Subclasses of this may be used by passing an appropriate factory function to
  43   resolve.
  44 
  45   Attributes:
  46     param_of: Optional[ast.AST]
  47   """
  48 
  49   def __init__(self):
  50     self.param_of = None
  51 
  52   def __repr__(self):
  53     return '%s[%d]' % (self.__class__.__name__, id(self))
  54

  56 class _NodeState(object):
  57   """Abstraction for the state of the CFG walk for reaching definition analysis.
  58 
  59   This is a value type. Only implements the strictly necessary operators.
  60 
  61   Attributes:
  62     value: Dict[qual_names.QN, Set[Definition, ...]], the defined symbols and
  63         their possible definitions
  64   """
  65 
  66   def __init__(self, init_from=None):
  67     if init_from:
  68       if isinstance(init_from, _NodeState):
  69         self.value = {
  70             s: set(other_infos) for s, other_infos in init_from.value.items()
  71         }
  72       elif isinstance(init_from, dict):
  73         self.value = {s: set((init_from[s],)) for s in init_from}
  74       else:
  75         assert False, init_from
  76     else:
  77       self.value = {}
  78 
  79   def __eq__(self, other):
  80     if frozenset(self.value.keys()) != frozenset(other.value.keys()):
  81       return False
  82     ret = all(self.value[s] == other.value[s] for s in self.value)
  83     return ret
  84 
  85   def __ne__(self, other):
  86     return not self.__eq__(other)
  87 
  88   def __or__(self, other):
  89     assert isinstance(other, _NodeState)
  90     result = _NodeState(self)
  91     for s, other_infos in other.value.items():
  92       if s in result.value:
  93         result.value[s].update(other_infos)
  94       else:
  95         result.value[s] = set(other_infos)
  96     return result
  97 
  98   def __sub__(self, other):
  99     assert isinstance(other, set)
 100     result = _NodeState(self)
 101     for s in other:
 102       result.value.pop(s, None)
 103     return result
 104 
 105   def __repr__(self):
 106     return 'NodeState[%s]=%s' % (id(self), repr(self.value))

 109 class Analyzer(cfg.GraphVisitor):
 110   """CFG visitor that determines reaching definitions at statement level."""
 111 
 112   def __init__(self, graph, definition_factory):
 113     self._definition_factory = definition_factory
 114     super(Analyzer, self).__init__(graph)
 115     # This allows communicating that nodes have extra reaching definitions,
 116     # e.g. those that a function closes over.
 117     self.extra_in = {}
 118 
 119     self.gen_map = {}
 120 
 121   def init_state(self, _):
 122     return _NodeState()
 123 
 124   def visit_node(self, node):
 125     prev_defs_out = self.out[node]
 126 
 127     defs_in = _NodeState(self.extra_in.get(node.ast_node, None))
 128     for n in node.prev:
 129       defs_in |= self.out[n]
 130 
 131     if anno.hasanno(node.ast_node, anno.Static.SCOPE):
 132       node_scope = anno.getanno(node.ast_node, anno.Static.SCOPE)
 133       # The definition objects created by each node must be singletons because
 134       # their ids are used in equality checks.
 135       if node not in self.gen_map:
 136         node_symbols = {}
 137         for s in node_scope.modified:
 138           def_ = self._definition_factory()
 139           if s in node_scope.params:
 140             def_.param_of = node_scope.params[s]
 141           node_symbols[s] = def_
 142         self.gen_map[node] = _NodeState(node_symbols)
 143 
 144       gen = self.gen_map[node]
 145       kill = node_scope.modified
 146       defs_out = gen | (defs_in - kill)
 147 
 148     else:
 149       # Nodes that don't have a scope annotation are assumed not to touch any
 150       # symbols.
 151       # This Name node below is a literal name, e.g. False
 152       # This can also happen if activity.py forgot to annotate the node with a
 153       # scope object.
 154       assert isinstance(
 155           node.ast_node,
 156           (gast.Name, gast.Break, gast.Continue, gast.Raise)), (node.ast_node,
 157                                                                 node)
 158       defs_out = defs_in
 159 
 160     self.in_[node] = defs_in
 161     self.out[node] = defs_out
 162 
 163     # TODO(mdan): Move this to the superclass?
 164     return prev_defs_out != defs_out

 167 class TreeAnnotator(transformer.Base):
 168   """AST visitor that annotates each symbol name with its reaching definitions.
 169 
 170   Simultaneously, the visitor runs the dataflow analysis on each function node,
 171   accounting for the effect of closures. For example:
 172 
 173     def foo():
 174       bar = 1
 175       def baz():
 176         # bar = 1 reaches here
 177   """
 178 
 179   def __init__(self, source_info, graphs, definition_factory):
 180     super(TreeAnnotator, self).__init__(source_info)
 181     self.definition_factory = definition_factory
 182     self.graphs = graphs
 183     self.current_analyzer = None
 184     self.current_cfg_node = None
 185 
 186   def visit_FunctionDef(self, node):
 187     parent_analyzer = self.current_analyzer
 188     subgraph = self.graphs[node]
 189 
 190     # Preorder tree processing:
 191     #  1. if this is a child function, the parent was already analyzed and it
 192     #     has the proper state value for the subgraph's entry
 193     #  2. analyze the current function body
 194     #  2. recursively walk the subtree; child functions will be processed
 195     analyzer = Analyzer(subgraph, self.definition_factory)
 196     if parent_analyzer is not None:
 197       # Wire the state between the two subgraphs' analyzers.
 198       parent_out_state = parent_analyzer.out[parent_analyzer.graph.index[node]]
 199       # Exception: symbols modified in the child function are local to it
 200       body_scope = anno.getanno(node, annos.NodeAnno.BODY_SCOPE)
 201       parent_out_state -= body_scope.modified
 202       analyzer.extra_in[node.args] = parent_out_state
 203 
 204     # Complete the analysis for the local function and annotate its body.
 205     analyzer.visit_forward()
 206 
 207     # Recursively process any remaining subfunctions.
 208     self.current_analyzer = analyzer
 209     # Note: not visiting name, decorator_list and returns because they don't
 210     # apply to this anlysis.
 211     # TODO(mdan): Should we still process the function name?
 212     node.args = self.visit(node.args)
 213     node.body = self.visit_block(node.body)
 214     self.current_analyzer = parent_analyzer
 215 
 216     return node
 217 
 218   def visit_nonlocal(self, node):
 219     raise NotImplementedError()
 220 
 221   def visit_global(self, node):
 222     raise NotImplementedError()
 223 
 224   def visit_Name(self, node):
 225     if self.current_analyzer is None:
 226       # Names may appear outside function defs - for example in class
 227       # definitions.
 228       return node
 229 
 230     analyzer = self.current_analyzer
 231     cfg_node = self.current_cfg_node
 232 
 233     assert cfg_node is not None, 'name node outside of any statement?'
 234 
 235     qn = anno.getanno(node, anno.Basic.QN)
 236     if isinstance(node.ctx, gast.Load):
 237       anno.setanno(node, anno.Static.DEFINITIONS,
 238                    tuple(analyzer.in_[cfg_node].value.get(qn, ())))
 239     else:
 240       anno.setanno(node, anno.Static.DEFINITIONS,
 241                    tuple(analyzer.out[cfg_node].value.get(qn, ())))
 242 
 243     return node
 244 
 245   def _aggregate_predecessors_defined_in(self, node):
 246     preds = self.current_analyzer.graph.stmt_prev[node]
 247     node_defined_in = set()
 248     for p in preds:
 249       node_defined_in |= set(self.current_analyzer.out[p].value.keys())
 250     anno.setanno(node, anno.Static.DEFINED_VARS_IN, frozenset(node_defined_in))
 251 
 252   def visit_If(self, node):
 253     self._aggregate_predecessors_defined_in(node)
 254     return self.generic_visit(node)
 255 
 256   def visit_For(self, node):
 257     self._aggregate_predecessors_defined_in(node)
 258 
 259     # Manually accounting for the shortcoming described in
 260     # cfg.AstToCfg.visit_For.
 261     parent = self.current_cfg_node
 262     self.current_cfg_node = self.current_analyzer.graph.index[node.iter]
 263     node.target = self.visit(node.target)
 264     self.current_cfg_node = parent
 265 
 266     node.iter = self.visit(node.iter)
 267     node.body = self.visit_block(node.body)
 268     node.orelse = self.visit_block(node.orelse)
 269 
 270     return node
 271 
 272   def visit_While(self, node):
 273     self._aggregate_predecessors_defined_in(node)
 274     return self.generic_visit(node)
 275 
 276   def visit(self, node):
 277     parent = self.current_cfg_node
 278 
 279     if (self.current_analyzer is not None and
 280         node in self.current_analyzer.graph.index):
 281       self.current_cfg_node = self.current_analyzer.graph.index[node]
 282     node = super(TreeAnnotator, self).visit(node)
 283 
 284     self.current_cfg_node = parent
 285     return node

 288 def resolve(node, source_info, graphs, definition_factory):
 289   """Resolves reaching definitions for each symbol.
 290 
 291   Args:
 292     node: ast.AST
 293     source_info: transformer.SourceInfo
 294     graphs: Dict[ast.FunctionDef, cfg.Graph]
 295     definition_factory: Callable[[], Definition]
 296   Returns:
 297     ast.AST
 298   """
 299   visitor = TreeAnnotator(source_info, graphs, definition_factory)
 300   node = visitor.visit(node)
 301   return node
