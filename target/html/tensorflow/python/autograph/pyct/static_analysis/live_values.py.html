  15 """Live value resolution.
  16 
  17 Live values are extracted from the known execution context.
  18 
  19 Requires activity and reaching definitions analyses.
  20 """
  21 
  22 from __future__ import absolute_import
  23 from __future__ import division
  24 from __future__ import print_function
  25 
  26 import gast
  27 
  28 from tensorflow.python.autograph.pyct import anno
  29 from tensorflow.python.autograph.pyct import transformer
  30 from tensorflow.python.autograph.pyct.static_analysis.annos import NodeAnno
  31 
  32 
  33 # TODO(aqj): Do we need this? Do other builtins fail in similar ways
  34 # See b/114389775 for a related bug in pyct
  35 # These symbols are legal in Python, but don't appear in the namespace.
  36 _SPECIAL_SYMBOLS = {'range': range, 'print': print}
  37 
  38 
  39 class LiveValueResolver(transformer.Base):
  40   """Annotates nodes with live values."""
  41 
  42   def __init__(self, context, literals):
  43     super(LiveValueResolver, self).__init__(context)
  44     self.literals = literals
  45 
  46   def visit_ClassDef(self, node):
  47     self.generic_visit(node)
  48     anno.setanno(node, 'live_val', self.entity_info.namespace[node.name])
  49     return node
  50 
  51   def visit_Name(self, node):
  52     self.generic_visit(node)
  53     if isinstance(node.ctx, gast.Load):
  54       defs = anno.getanno(node, anno.Static.DEFINITIONS, ())
  55 
  56       is_defined = bool(defs)
  57       has_single_def = len(defs) == 1
  58 
  59       if not is_defined:
  60         if node.id in self.literals:
  61           anno.setanno(node, 'live_val', self.literals[node.id])
  62         elif node.id in self.entity_info.namespace:
  63           obj = self.entity_info.namespace[node.id]
  64           anno.setanno(node, 'live_val', obj)
  65           if hasattr(obj, '__name__'):
  66             anno.setanno(node, 'fqn', (obj.__name__,))
  67           elif hasattr(obj, '__class__'):
  68             obj_class = obj.__class__
  69             anno.setanno(node, 'fqn',
  70                          (obj_class.__module__, obj_class.__name__))
  71           else:
  72             # If the symbol value is for example a primitive, then it will not
  73             # have a name.
  74             pass
  75         elif node.id in _SPECIAL_SYMBOLS:
  76           # Note: if the user redefined any of these symbols, then they would
  77           # be visible in the namespace and we would never reach this branch.
  78           anno.setanno(node, 'live_val', _SPECIAL_SYMBOLS[node.id])
  79         else:
  80           pass
  81           # TODO(mdan): Should we raise an error here?
  82           # Can encounter this when:
  83           #  * a symbol truly lacks reference
  84           #  * a symbol is new, like the new name of a function we just renamed.
  85       else:
  86         pass
  87         # TODO(mdan): Attempt to trace its value through the local chain.
  88         # TODO(mdan): Use type annotations as fallback.
  89 
  90       if has_single_def:
  91         def_, = defs
  92         # Note: param_of is a weakref.
  93         if def_.param_of and def_.param_of() is self.enclosing_entities[0]:
  94           if node.id in self.entity_info.arg_values:
  95             obj = self.entity_info.arg_values[node.id]
  96             anno.setanno(node, 'live_val', obj)
  97             anno.setanno(node, 'fqn', (obj.__class__.__name__,))
  98     return node
  99 
 100   def visit_Attribute(self, node):
 101     self.generic_visit(node)
 102     if anno.hasanno(node.value, 'live_val'):
 103       assert anno.hasanno(node.value, 'fqn')
 104       parent_object = anno.getanno(node.value, 'live_val')
 105 
 106       anno.setanno(node, 'parent_type', type(parent_object))
 107       anno.setanno(node, 'fqn', anno.getanno(node.value, 'fqn') + (node.attr,))
 108       if hasattr(parent_object, node.attr):
 109         # This can happen when the attribute's creation and use depend on the
 110         # same static condition, for example:
 111         #
 112         #  if cond:
 113         #    foo.bar = baz
 114         #  if cond:
 115         #    x = foo.bar
 116         #
 117         anno.setanno(node, 'live_val', getattr(parent_object, node.attr))
 118 
 119     # TODO(mdan): Investigate the role built-in annotations can play here.
 120     elif anno.hasanno(node.value, 'type'):
 121       parent_type = anno.getanno(node.value, 'type')
 122       if hasattr(parent_type, node.attr):
 123         # This should hold for static members like methods.
 124         # This would not hold for dynamic members like function attributes.
 125         # For the dynamic case, we simply leave the node without an annotation,
 126         # and let downstream consumers figure out what to do.
 127         anno.setanno(node, 'parent_type', parent_type)
 128         anno.setanno(node, 'live_val', getattr(parent_type, node.attr))
 129         anno.setanno(node, 'fqn',
 130                      anno.getanno(node.value, 'type_fqn') + (node.attr,))
 131     elif isinstance(node.value, gast.Name):
 132       stem_name = node.value
 133       # All nonlocal symbols should be fully resolved.
 134       assert anno.hasanno(stem_name, NodeAnno.IS_LOCAL), stem_name
 135       # TODO(mdan): Figure out what to do when calling attribute on local object
 136       # Maybe just leave as-is?
 137     return node
 138 
 139 
 140 def resolve(node, context, literals):
 141   return LiveValueResolver(context, literals).visit(node)
