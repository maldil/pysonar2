  38 class Scope(object):
  39   """Encloses local symbol definition and usage information.
  40 
  41   This can track for instance whether a symbol is modified in the current scope.
  42   Note that scopes do not necessarily align with Python's scopes. For example,
  43   the body of an if statement may be considered a separate scope.
  44 
  45   Attributes:
  46     modified: identifiers modified in this scope
  47     created: identifiers created in this scope
  48     used: identifiers referenced in this scope
  49   """
  50 
  51   def __init__(self, parent, isolated=True, add_unknown_symbols=False):
  52     """Create a new scope.
  53 
  54     Args:
  55       parent: A Scope or None.
  56       isolated: Whether the scope is isolated, that is, whether variables
  57           created in this scope should be visible to the parent scope.
  58       add_unknown_symbols: Whether to handle attributed and subscripts
  59           without having first seen the base name.
  60           E.g., analyzing the statement 'x.y = z' without first having seen 'x'.
  61     """
  62     self.isolated = isolated
  63     self.parent = parent
  64     self.add_unknown_symbols = add_unknown_symbols
  65     self.modified = set()
  66     # TODO(mdan): Completely remove this.
  67     self.created = set()
  68     self.used = set()
  69     self.params = {}
  70     self.returned = set()
  71 
  72   # TODO(mdan): Rename to `locals`
  73   @property
  74   def referenced(self):
  75     if not self.isolated and self.parent is not None:
  76       return self.used | self.parent.referenced
  77     return self.used
  78 
  79   def __repr__(self):
  80     return 'Scope{r=%s, c=%s, w=%s}' % (tuple(self.used), tuple(self.created),
  81                                         tuple(self.modified))
  82 
  83   def copy_from(self, other):
  84     """Recursively copies the contents of this scope from another scope."""
  85     if (self.parent is None) != (other.parent is None):
  86       raise ValueError('cannot copy scopes of different structures')
  87     if other.parent is not None:
  88       self.parent.copy_from(other.parent)
  89     self.isolated = other.isolated
  90     self.modified = copy.copy(other.modified)
  91     self.created = copy.copy(other.created)
  92     self.used = copy.copy(other.used)
  93     self.params = copy.copy(other.params)
  94     self.returned = copy.copy(other.returned)
  95 
  96   @classmethod
  97   def copy_of(cls, other):
  98     if other.parent is not None:
  99       parent = cls.copy_of(other.parent)
 100     else:
 101       parent = None
 102     new_copy = cls(parent)
 103     new_copy.copy_from(other)
 104     return new_copy
 105 
 106   def merge_from(self, other):
 107     if (self.parent is None) != (other.parent is None):
 108       raise ValueError('cannot merge scopes of different structures')
 109     if other.parent is not None:
 110       self.parent.merge_from(other.parent)
 111     self.modified |= other.modified
 112     self.created |= other.created
 113     self.used |= other.used
 114     self.params.update(other.params)
 115     self.returned |= other.returned
 116 
 117   def has(self, name):
 118     if name in self.modified:
 119       return True
 120     elif self.parent is not None:
 121       return self.parent.has(name)
 122     return False
 123 
 124   def mark_read(self, name):
 125     self.used.add(name)
 126     if self.parent is not None and name not in self.created:
 127       self.parent.mark_read(name)
 128 
 129   def mark_param(self, name, owner):
 130     # Assumption: all AST nodes have the same life span. This lets us use
 131     # a weak reference to mark the connection between a symbol node and the
 132     # function node whose argument that symbol is.
 133     self.params[name] = weakref.ref(owner)
 134 
 135   def mark_creation(self, name, writes_create_symbol=False):
 136     """Mark a qualified name as created."""
 137     if name.is_composite():
 138       parent = name.parent
 139       if not writes_create_symbol:
 140         return
 141       else:
 142         if not self.has(parent):
 143           if self.add_unknown_symbols:
 144             self.mark_read(parent)
 145           else:
 146             raise ValueError('Unknown symbol "%s".' % parent)
 147     self.created.add(name)
 148 
 149   def mark_write(self, name):
 150     """Marks the given symbol as modified in the current scope."""
 151     self.modified.add(name)
 152     if self.isolated:
 153       self.mark_creation(name)
 154     else:
 155       if self.parent is None:
 156         self.mark_creation(name)
 157       else:
 158         if not self.parent.has(name):
 159           self.mark_creation(name)
 160         self.parent.mark_write(name)
 161 
 162   def mark_returned(self, name):
 163     self.returned.add(name)
 164     if not self.isolated and self.parent is not None:
 165       self.parent.mark_returned(name)

 168 class ActivityAnalyzer(transformer.Base):
 169   """Annotates nodes with local scope information.
 170 
 171   See Scope.
 172 
 173   The use of this class requires that qual_names.resolve() has been called on
 174   the node. This class will ignore nodes have not been
 175   annotated with their qualified names.
 176   """
 177 
 178   def __init__(self, context, parent_scope=None, add_unknown_symbols=False):
 179     super(ActivityAnalyzer, self).__init__(context)
 180     self.scope = Scope(parent_scope, None, add_unknown_symbols)
 181     self._in_return_statement = False
 182     self._in_aug_assign = False
 183 
 184   @property
 185   def _in_constructor(self):
 186     if len(self.enclosing_entities) > 1:
 187       innermost = self.enclosing_entities[-1]
 188       parent = self.enclosing_entities[-2]
 189       return isinstance(parent, gast.ClassDef) and innermost.name == '__init__'
 190     return False
 191 
 192   def _node_sets_self_attribute(self, node):
 193     if anno.hasanno(node, anno.Basic.QN):
 194       qn = anno.getanno(node, anno.Basic.QN)
 195       # TODO(mdan): The 'self' argument is not guaranteed to be called 'self'.
 196       if qn.has_attr and qn.parent.qn == ('self',):
 197         return True
 198     return False
 199 
 200   def _track_symbol(self,
 201                     node,
 202                     composite_writes_alter_parent=False,
 203                     writes_create_symbol=False):
 204     # A QN may be missing when we have an attribute (or subscript) on a function
 205     # call. Example: a().b
 206     if not anno.hasanno(node, anno.Basic.QN):
 207       return
 208     qn = anno.getanno(node, anno.Basic.QN)
 209 
 210     if isinstance(node.ctx, gast.Store):
 211       self.scope.mark_write(qn)
 212       if qn.is_composite and composite_writes_alter_parent:
 213         self.scope.mark_write(qn.parent)
 214       if writes_create_symbol:
 215         self.scope.mark_creation(qn, writes_create_symbol=True)
 216       if self._in_aug_assign:
 217         self.scope.mark_read(qn)
 218     elif isinstance(node.ctx, gast.Load):
 219       self.scope.mark_read(qn)
 220     elif isinstance(node.ctx, gast.Param):
 221       # Param contexts appear in function defs, so they have the meaning of
 222       # defining a variable.
 223       self.scope.mark_write(qn)
 224       self.scope.mark_param(qn, self.enclosing_entities[-1])
 225     else:
 226       raise ValueError('Unknown context %s for node %s.' % (type(node.ctx), qn))
 227 
 228     anno.setanno(node, NodeAnno.IS_LOCAL, self.scope.has(qn))
 229 
 230     if self._in_return_statement:
 231       self.scope.mark_returned(qn)
 232 
 233   def _enter_scope(self, isolated):
 234     self.scope = Scope(self.scope, isolated=isolated)
 235 
 236   def _exit_scope(self):
 237     self.scope = self.scope.parent
 238 
 239   def _process_statement(self, node):
 240     self._enter_scope(False)
 241     node = self.generic_visit(node)
 242     anno.setanno(node, anno.Static.SCOPE, self.scope)
 243     self._exit_scope()
 244     return node
 245 
 246   def visit_Expr(self, node):
 247     return self._process_statement(node)
 248 
 249   def visit_Return(self, node):
 250     self._in_return_statement = True
 251     node = self._process_statement(node)
 252     self._in_return_statement = False
 253     return node
 254 
 255   def visit_Assign(self, node):
 256     return self._process_statement(node)
 257 
 258   def visit_AugAssign(self, node):
 259     # Special rules for AugAssign. In Assign, the target is only written,
 260     # but in AugAssig (e.g. a += b), the target is both read and written.
 261     self._in_aug_assign = True
 262     node = self._process_statement(node)
 263     self._in_aug_assign = False
 264     return node
 265 
 266   def visit_Name(self, node):
 267     node = self.generic_visit(node)
 268     self._track_symbol(node)
 269     return node
 270 
 271   def visit_Attribute(self, node):
 272     node = self.generic_visit(node)
 273     if self._in_constructor and self._node_sets_self_attribute(node):
 274       self._track_symbol(
 275           node, composite_writes_alter_parent=True, writes_create_symbol=True)
 276     else:
 277       self._track_symbol(node)
 278     return node
 279 
 280   def visit_Subscript(self, node):
 281     node = self.generic_visit(node)
 282     # Subscript writes (e.g. a[b] = "value") are considered to modify
 283     # both the element itself (a[b]) and its parent (a).
 284     self._track_symbol(node)
 285     return node
 286 
 287   def visit_Print(self, node):
 288     self._enter_scope(False)
 289     node.values = self.visit_block(node.values)
 290     anno.setanno(node, anno.Static.SCOPE, self.scope)
 291     anno.setanno(node, NodeAnno.ARGS_SCOPE, self.scope)
 292     self._exit_scope()
 293     return node
 294 
 295   def visit_Assert(self, node):
 296     return self._process_statement(node)
 297 
 298   def visit_Call(self, node):
 299     self._enter_scope(False)
 300     node.args = self.visit_block(node.args)
 301     node.keywords = self.visit_block(node.keywords)
 302     # TODO(mdan): Account starargs, kwargs
 303     anno.setanno(node, NodeAnno.ARGS_SCOPE, self.scope)
 304     self._exit_scope()
 305     node.func = self.visit(node.func)
 306     return node
 307 
 308   def _process_block_node(self, node, block, scope_name):
 309     self._enter_scope(False)
 310     block = self.visit_block(block)
 311     anno.setanno(node, scope_name, self.scope)
 312     self._exit_scope()
 313     return node
 314 
 315   def _process_parallel_blocks(self, parent, children):
 316     # Because the scopes are not isolated, processing any child block
 317     # modifies the parent state causing the other child blocks to be
 318     # processed incorrectly. So we need to checkpoint the parent scope so that
 319     # each child sees the same context.
 320     before_parent = Scope.copy_of(self.scope)
 321     after_children = []
 322     for child, scope_name in children:
 323       self.scope.copy_from(before_parent)
 324       parent = self._process_block_node(parent, child, scope_name)
 325       after_child = Scope.copy_of(self.scope)
 326       after_children.append(after_child)
 327     for after_child in after_children:
 328       self.scope.merge_from(after_child)
 329     return parent
 330 
 331   def visit_arguments(self, node):
 332     return self._process_statement(node)
 333 
 334   def visit_FunctionDef(self, node):
 335     # The FunctionDef node itself has a Scope object that tracks the creation
 336     # of its name, along with the usage of any decorator accompany it.
 337     self._enter_scope(False)
 338     node.decorator_list = self.visit_block(node.decorator_list)
 339     self.scope.mark_write(qual_names.QN(node.name))
 340     anno.setanno(node, anno.Static.SCOPE, self.scope)
 341     self._exit_scope()
 342 
 343     # A separate Scope tracks the actual function definition.
 344     self._enter_scope(True)
 345     node.args = self.visit(node.args)
 346 
 347     # Track the body separately. This is for compatibility reasons, it may not
 348     # be strictly needed.
 349     self._enter_scope(False)
 350     node.body = self.visit_block(node.body)
 351     anno.setanno(node, NodeAnno.BODY_SCOPE, self.scope)
 352     self._exit_scope()
 353 
 354     self._exit_scope()
 355     return node
 356 
 357   def visit_With(self, node):
 358     self._enter_scope(False)
 359     node = self.generic_visit(node)
 360     anno.setanno(node, NodeAnno.BODY_SCOPE, self.scope)
 361     self._exit_scope()
 362     return node
 363 
 364   def visit_withitem(self, node):
 365     return self._process_statement(node)
 366 
 367   def visit_If(self, node):
 368     self._enter_scope(False)
 369     node.test = self.visit(node.test)
 370     anno.setanno(node, NodeAnno.COND_SCOPE, self.scope)
 371     anno.setanno(node.test, anno.Static.SCOPE, self.scope)
 372     self._exit_scope()
 373     node = self._process_parallel_blocks(node,
 374                                          ((node.body, NodeAnno.BODY_SCOPE),
 375                                           (node.orelse, NodeAnno.ORELSE_SCOPE)))
 376     return node
 377 
 378   def visit_For(self, node):
 379     self._enter_scope(False)
 380     node.target = self.visit(node.target)
 381     node.iter = self.visit(node.iter)
 382     anno.setanno(node.iter, anno.Static.SCOPE, self.scope)
 383     self._exit_scope()
 384     node = self._process_parallel_blocks(node,
 385                                          ((node.body, NodeAnno.BODY_SCOPE),
 386                                           (node.orelse, NodeAnno.ORELSE_SCOPE)))
 387     return node
 388 
 389   def visit_While(self, node):
 390     self._enter_scope(False)
 391     node.test = self.visit(node.test)
 392     anno.setanno(node, NodeAnno.COND_SCOPE, self.scope)
 393     anno.setanno(node.test, anno.Static.SCOPE, self.scope)
 394     self._exit_scope()
 395     node = self._process_parallel_blocks(node,
 396                                          ((node.body, NodeAnno.BODY_SCOPE),
 397                                           (node.orelse, NodeAnno.ORELSE_SCOPE)))
 398     return node
 399

 401 def resolve(node, context, parent_scope=None):
 402   return ActivityAnalyzer(context, parent_scope).visit(node)
