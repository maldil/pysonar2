  99 def keys(node, field_name='___pyct_anno'):
 100   if not hasattr(node, field_name):
 101     return frozenset()
 102   return frozenset(getattr(node, field_name).keys())
 103 
 104 
 105 def getanno(node, key, default=FAIL, field_name='___pyct_anno'):
 106   if (default is FAIL or (hasattr(node, field_name) and
 107                           (key in getattr(node, field_name)))):
 108     return getattr(node, field_name)[key]
 109   else:
 110     return default
 111 
 112 
 113 def hasanno(node, key, field_name='___pyct_anno'):
 114   return hasattr(node, field_name) and key in getattr(node, field_name)
 115 
 116 
 117 def setanno(node, key, value, field_name='___pyct_anno'):
 118   annotations = getattr(node, field_name, {})
 119   setattr(node, field_name, annotations)
 120   annotations[key] = value
 121 
 122   # So that the annotations survive gast_to_ast() and ast_to_gast()
 123   if field_name not in node._fields:
 124     node._fields += (field_name,)
 125 
 126 
 127 def delanno(node, key, field_name='___pyct_anno'):
 128   annotations = getattr(node, field_name)
 129   del annotations[key]
 130   if not annotations:
 131     delattr(node, field_name)
 132     node._fields = tuple(f for f in node._fields if f != field_name)
 133 
 134 
 135 def copyanno(from_node, to_node, key, field_name='___pyct_anno'):
 136   if hasanno(from_node, key, field_name=field_name):
 137     setanno(
 138         to_node,
 139         key,
 140         getanno(from_node, key, field_name=field_name),
 141         field_name=field_name)
 142

 144 def dup(node, copy_map, field_name='___pyct_anno'):
 145   """Recursively copies annotations in an AST tree.
 146 
 147   Args:
 148     node: ast.AST
 149     copy_map: Dict[Hashable, Hashable], maps a source anno key to a destination
 150         key. All annotations with the source key will be copied to identical
 151         annotations with the destination key.
 152     field_name: str
 153   """
 154   for n in gast.walk(node):
 155     for k in copy_map:
 156       if hasanno(n, k, field_name):
 157         setanno(n, copy_map[k], getanno(n, k, field_name), field_name)
