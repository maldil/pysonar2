  36 class EntityInfo(object):
  37   """Contains information about a Python entity. Immutable.
  38 
  39   Examples of entities include functions and classes.
  40 
  41   Attributes:
  42     source_code: The entity's source code.
  43     source_file: The entity's source file.
  44     namespace: Dict[str, ], containing symbols visible to the entity
  45         (excluding parameters).
  46     arg_values: dict[str->*], containing parameter values, if known.
  47     arg_types: dict[str->*], containing parameter types, if known.
  48     owner_type: The surrounding class type of the function, if present.
  49   """
  50 
  51   # TODO(mdan): Remove the default and update tests.
  52   def __init__(self, source_code, source_file, namespace, arg_values, arg_types,
  53                owner_type):
  54     self.source_code = source_code
  55     self.source_file = source_file
  56     self.namespace = namespace
  57     self.arg_values = {} if arg_values is None else arg_values
  58     self.arg_types = {} if arg_types is None else arg_types
  59     self.owner_type = owner_type

  62 class _StateStack(object):
  63   """Typed stack abstraction.
  64 
  65   This class provides syntactic sugar for a stack of objects of known
  66   type. It allows accessing attributes of the object at the top of the stack
  67   directly against this object, which allows for very terse syntax.
  68 
  69   For example, this code:
  70 
  71     stack = _StateStack(Foo)
  72     stack.enter()
  73     stack.bar
  74 
  75   Is equivalent to:
  76 
  77     stack = []
  78     stack.append(Foo())
  79     foo = stack[-1]
  80     foo.bar
  81 
  82   See _State for more on how this is used.
  83 
  84   Attributes:
  85     type: Any, the type of objects that this stack holds
  86     level: int, the current stack depth
  87     value: Any, the instance of the object at the top of the stack
  88   """
  89 
  90   def __init__(self, type_):
  91     # Because we override __setattr__, we need to attach these attributes using
  92     # the superclass' setattr.
  93     object.__setattr__(self, 'type', type_)
  94     object.__setattr__(self, '_stack', [])
  95     self.enter()
  96 
  97   def enter(self):
  98     self._stack.append(self.type())
  99 
 100   def exit(self):
 101     return self._stack.pop()
 102 
 103   @property
 104   def level(self):
 105     return len(self._stack)
 106 
 107   @property
 108   def value(self):
 109     return self._stack[-1]
 110 
 111   def __getattr__(self, key):
 112     return getattr(self._stack[-1], key)
 113 
 114   def __setattr__(self, key, value):
 115     setattr(self._stack[-1], key, value)

 118 class _State(object):
 119   """Supporting class for nested scope variable space for converter.Base.
 120 
 121   This structure offers syntactic sugar over a dict of stacks of objects
 122   of known type. These structures are useful to keep state during AST walks.
 123   Multiple different scopes can be tracked in parallel. For example:
 124 
 125     s = _State()
 126 
 127     s[foo].enter()
 128     s[bar].enter()  # this will not affect s[foo]
 129 
 130   Element access has special semantics:
 131     * keys are a data type
 132     * element values are _StateStack(type=key) objects
 133     * missing elements are automatically added, similarly to defaultdict
 134 
 135   For example, the following block :
 136 
 137     _State s
 138     s[Foo]
 139 
 140   Is equivalent to:
 141 
 142     s = {}
 143     if Foo not in s:
 144       s[Foo] = Foo()
 145     s[Foo]
 146 
 147   See Base for how it's used.
 148   """
 149 
 150   def __init__(self):
 151     self._value = {}
 152 
 153   def __getitem__(self, key):
 154     if key not in self._value:
 155       self._value[key] = _StateStack(key)
 156     return self._value[key]

 159 class Base(gast.NodeTransformer):
 160   """Base class for general-purpose code transformers transformers.
 161 
 162   This is an extension of ast.NodeTransformer that provides a few additional
 163   functions, like state tracking within the scope of arbitrary node, helpers
 164   for processing code blocks, debugging, mapping of transformed code to
 165   original code, and others.
 166 
 167   Scope-local state tracking: to keep state across nodes, at the level of
 168   (possibly nested) scopes, use enter/exit_local_scope and set/get_local.
 169   You must call enter/exit_local_scope manually, but the transformer detects
 170   when they are not properly paired.
 171 
 172   The transformer allows keeping state across calls to visit_* that is local to
 173   arbitrary nodes and their descendants, using the self.state attribute.
 174   Multiple independent scopes are allowed and automatically constructed.
 175 
 176   For example, to keep track of the If node that encloses any Name node, one can
 177   write:
 178 
 179     class FooType(object):
 180 
 181       def __init__(self):
 182         self.foo_property = None
 183 
 184     class DummyTransformer(Base):
 185 
 186       def visit_If(self, node):
 187         self.state[FooType].enter()
 188         self.state[FooType].foo_property = node
 189 
 190       def visit_Name(self, node):
 191         self.state[FooType].foo_property  # will hold the innermost enclosing if
 192   """
 193 
 194   # TODO(mdan): Document all extra features.
 195 
 196   def __init__(self, entity_info):
 197     """Initialize the transformer. Subclasses should call this.
 198 
 199     Args:
 200       entity_info: An EntityInfo object.
 201     """
 202     self._lineno = 0
 203     self._col_offset = 0
 204     self.entity_info = entity_info
 205     self._enclosing_entities = []
 206 
 207     # A stack that allows keeping mutable, scope-local state where scopes may be
 208     # nested. For example, it can be used to track the usage of break
 209     # statements in each loop, where loops may be nested.
 210     self._local_scope_state = []
 211     self.enter_local_scope()
 212 
 213     # Allows scoping of local variables to keep state across calls to visit_*
 214     # methods. Multiple scope hierchies may exist and are keyed by tag. A scope
 215     # is valid at one or more nodes and all its children. Scopes created in
 216     # child nodes supersede their parent. Scopes are isolated from one another.
 217     self.state = _State()
 218 
 219   @property
 220   def enclosing_entities(self):
 221     return tuple(self._enclosing_entities)
 222 
 223   @property
 224   def local_scope_level(self):
 225     return len(self._local_scope_state)
 226 
 227   def enter_local_scope(self, inherit=None):
 228     """Deprecated. Use self.state instead.
 229 
 230     Marks entry into a new local scope.
 231 
 232     Args:
 233       inherit: Optional enumerable of variable names to copy from the
 234           parent scope.
 235     """
 236     scope_entered = {}
 237     if inherit:
 238       this_scope = self._local_scope_state[-1]
 239       for name in inherit:
 240         if name in this_scope:
 241           scope_entered[name] = this_scope[name]
 242     self._local_scope_state.append(scope_entered)
 243 
 244   def exit_local_scope(self, keep=None):
 245     """Deprecated. Use self.state instead.
 246 
 247     Marks exit from the current local scope.
 248 
 249     Args:
 250       keep: Optional enumerable of variable names to copy into the
 251           parent scope.
 252     Returns:
 253       A dict containing the scope that has just been exited.
 254     """
 255     scope_left = self._local_scope_state.pop()
 256     if keep:
 257       this_scope = self._local_scope_state[-1]
 258       for name in keep:
 259         if name in scope_left:
 260           this_scope[name] = scope_left[name]
 261     return scope_left
 262 
 263   def set_local(self, name, value):
 264     """Deprecated. Use self.state instead."""
 265     self._local_scope_state[-1][name] = value
 266 
 267   def get_local(self, name, default=None):
 268     """Deprecated. Use self.state instead."""
 269     return self._local_scope_state[-1].get(name, default)
 270 
 271   def debug_print(self, node):
 272     """Helper method useful for debugging."""
 273     if __debug__:
 274       print(pretty_printer.fmt(node))
 275     return node

 277   def visit_block(self, nodes, before_visit=None, after_visit=None):
 278     """A more powerful version of generic_visit for statement blocks.
 279 
 280     An example of a block is the body of an if statement.
 281 
 282     This function allows specifying a postprocessing callback (the
 283     after_visit argument) argument which can be used to move nodes to a new
 284     destination. This is done by after_visit by returning a non-null
 285     second return value, e.g. return new_node, new_destination.
 286 
 287     For example, a transformer could perform the following move:
 288 
 289         foo()
 290         bar()
 291         baz()
 292 
 293         foo()
 294         if cond:
 295           bar()
 296           baz()
 297 
 298     The above could be done with a postprocessor of this kind:
 299 
 300         def after_visit(node):
 301           if node_is_function_call(bar):
 302             new_container_node = build_cond()
 303             new_container_node.body.append(node)
 304             return new_container_node, new_container_node.body
 305           else:
 306             # Once we set a new destination, all subsequent items will be
 307             # moved to it, so we don't need to explicitly handle baz.
 308             return node, None
 309 
 310     Args:
 311       nodes: enumerable of AST node objects
 312       before_visit: optional callable that is called before visiting each item
 313           in nodes
 314       after_visit: optional callable that takes in an AST node and
 315           returns a tuple (new_node, new_destination). It is called after
 316           visiting each item in nodes. Is used in the same was as the
 317           visit_* methods: new_node will replace the node; if not None,
 318           new_destination must be a list, and subsequent nodes will be placed
 319           in this list instead of the list returned by visit_block.
 320     Returns:
 321       A list of AST node objects containing the transformed items fron nodes,
 322       except those nodes that have been relocated using after_visit.
 323     """
 324     results = []
 325     node_destination = results
 326     for node in nodes:
 327       if before_visit:
 328         # TODO(mdan): We can modify node here too, if ever needed.
 329         before_visit()
 330 
 331       replacement = self.visit(node)
 332 
 333       if after_visit and replacement:
 334         replacement, new_destination = after_visit(replacement)
 335       else:
 336         new_destination = None
 337 
 338       if replacement:
 339         if isinstance(replacement, (list, tuple)):
 340           node_destination.extend(replacement)
 341         else:
 342           node_destination.append(replacement)
 343 
 344       # Allow the postprocessor to reroute the remaining nodes to a new list.
 345       if new_destination is not None:
 346         node_destination = new_destination
 347     return results

 350   def apply_to_single_assignments(self, targets, values, apply_fn):
 351     """Applies a function to each individual assignment.
 352 
 353     This function can process a possibly-unpacked (e.g. a, b = c, d) assignment.
 354     It tries to break down the unpacking if possible. In effect, it has the same
 355     effect as passing the assigned values in SSA form to apply_fn.
 356 
 357     Examples:
 358 
 359     The following will result in apply_fn(a, c), apply_fn(b, d):
 360 
 361         a, b = c, d
 362 
 363     The following will result in apply_fn(a, c[0]), apply_fn(b, c[1]):
 364 
 365         a, b = c
 366 
 367     The following will result in apply_fn(a, (b, c)):
 368 
 369         a = b, c
 370 
 371     It uses the visitor pattern to allow subclasses to process single
 372     assignments individually.
 373 
 374     Args:
 375       targets: list, tuple of or individual AST node. Should be used with the
 376           targets field of an ast.Assign node.
 377       values: an AST node.
 378       apply_fn: a function of a single argument, which will be called with the
 379           respective nodes of each single assignment. The signature is
 380           apply_fn(target, value), no return value.
 381     """
 382     if not isinstance(targets, (list, tuple)):
 383       targets = (targets,)
 384     for target in targets:
 385       if isinstance(target, (gast.Tuple, gast.List)):
 386         for i in range(len(target.elts)):
 387           target_el = target.elts[i]
 388           if isinstance(values, (gast.Tuple, gast.List)):
 389             value_el = values.elts[i]
 390           else:
 391             value_el = gast.Subscript(values, gast.Index(i), ctx=gast.Store())
 392           self.apply_to_single_assignments(target_el, value_el, apply_fn)
 393       else:
 394         # TODO(mdan): Look into allowing to rewrite the AST here.
 395         apply_fn(target, values)

 397   def _get_source(self, node):
 398     try:
 399       source, _ = compiler.ast_to_source(node)
 400       return source
 401     # pylint: disable=broad-except
 402     # This function is used for error reporting.  If an exception occurs here,
 403     # it should be suppressed, in favor of emitting as informative a message
 404     # about the original error as possible.
 405     except Exception:
 406       return '<could not convert AST to source>'
 407 
 408   def visit(self, node):
 409     if not isinstance(node, gast.AST):
 410       # This is not that uncommon a mistake: various node bodies are lists, for
 411       # example, posing a land mine for transformers that need to recursively
 412       # call `visit`.  The error needs to be raised before the exception handler
 413       # below is installed, because said handler will mess up if `node` is not,
 414       # in fact, a node.
 415       msg = (
 416           'invalid value for "node": expected "ast.AST", got "{}"; to'
 417           ' visit lists of nodes, use "visit_block" instead').format(type(node))
 418       raise ValueError(msg)
 419 
 420     source_code = self.entity_info.source_code
 421     source_file = self.entity_info.source_file
 422     did_enter_function = False
 423     local_scope_size_at_entry = len(self._local_scope_state)
 424     processing_expr_node = False
 425 
 426     try:
 427       if isinstance(node, (gast.FunctionDef, gast.ClassDef, gast.Lambda)):
 428         did_enter_function = True
 429       elif isinstance(node, gast.Expr):
 430         processing_expr_node = True
 431 
 432       if did_enter_function:
 433         self._enclosing_entities.append(node)
 434 
 435       if source_code and hasattr(node, 'lineno'):
 436         self._lineno = node.lineno
 437         self._col_offset = node.col_offset
 438 
 439       if processing_expr_node:
 440         entry_expr_value = node.value
 441 
 442       if not anno.hasanno(node, anno.Basic.SKIP_PROCESSING):
 443         result = super(Base, self).visit(node)
 444 
 445       # Adjust for consistency: replacing the value of an Expr with
 446       # an Assign node removes the need for the Expr node.
 447       if processing_expr_node:
 448         if isinstance(result, gast.Expr) and result.value != entry_expr_value:
 449           # When the replacement is a list, it is assumed that the list came
 450           # from a template that contained a number of statements, which
 451           # themselves are standalone and don't require an enclosing Expr.
 452           if isinstance(result.value,
 453                         (list, tuple, gast.Assign, gast.AugAssign)):
 454             result = result.value
 455 
 456       # On exception, the local scope integrity is not guaranteed.
 457       if did_enter_function:
 458         self._enclosing_entities.pop()
 459 
 460       if local_scope_size_at_entry != len(self._local_scope_state):
 461         raise AssertionError(
 462             'Inconsistent local scope stack. Before entering node %s, the'
 463             ' stack had length %d, after exit it has length %d. This'
 464             ' indicates enter_local_scope and exit_local_scope are not'
 465             ' well paired.' % (
 466                 node,
 467                 local_scope_size_at_entry,
 468                 len(self._local_scope_state)
 469             ))
 470       return result
 471 
 472     except (ValueError, AttributeError, KeyError, NotImplementedError) as e:
 473       msg = '%s: %s\nOffending source:\n%s\n\nOccurred at node:\n%s' % (
 474           e.__class__.__name__, str(e), self._get_source(node),
 475           pretty_printer.fmt(node, color=False))
 476       if source_code:
 477         line = source_code.splitlines()[self._lineno - 1]
 478       else:
 479         line = '<no source available>'
 480       # TODO(mdan): Avoid the printing of the original exception.
 481       # In other words, we need to find how to suppress the "During handling
 482       # of the above exception, another exception occurred" message.
 483       six.reraise(AutographParseError,
 484                   AutographParseError(
 485                       msg,
 486                       (source_file, self._lineno, self._col_offset + 1, line)),
 487                   sys.exc_info()[2])
