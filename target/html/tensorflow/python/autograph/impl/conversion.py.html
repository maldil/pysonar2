  60 def is_whitelisted_for_graph(o):
  61   """Check whether an entity is whitelisted for use in graph mode.
  62 
  63   Examples of whitelisted entities include all members of the tensorflow
  64   package.
  65 
  66   Args:
  67     o: A Python entity.
  68   Returns:
  69     Boolean
  70   """
  71   m = tf_inspect.getmodule(o)
  72   for prefix, in config.DEFAULT_UNCOMPILED_MODULES:
  73     if m.__name__.startswith(prefix):
  74       return True
  75   if hasattr(o, 'autograph_info__'):
  76     return True
  77   return False

  80 def entity_to_graph(o, program_ctx, arg_values, arg_types):
  81   """Compile a Python entity into equivalent TensorFlow.
  82 
  83   The function will also recursively compile all the entities that `o`
  84   references, updating `dependency_cache`.
  85 
  86   This function is reentrant, and relies on dependency_cache to avoid
  87   generating duplicate code.
  88 
  89   Args:
  90     o: A Python entity.
  91     program_ctx: A ProgramContext object.
  92     arg_values: A dict containing value hints for symbols like function
  93         parameters.
  94     arg_types: A dict containing type hints for symbols like function
  95         parameters.
  96 
  97   Returns:
  98     A tuple (ast, new_name, namespace):
  99         * ast: An AST representing an entity with interface equivalent to `o`,
 100             but which when executed it creates TF a graph.
 101         * new_name: The symbol name under which the new entity can be found.
 102         * namespace: A dict mapping all symbols visible to the converted entity,
 103             keyed by their symbol name.
 104 
 105   Raises:
 106     ValueError: if the entity type is not supported.
 107   """
 108   if tf_inspect.isclass(o):
 109     node, name, ns = class_to_graph(o, program_ctx)
 110   elif tf_inspect.isfunction(o):
 111     # TODO(mdan): This is not a reliable mechanism.
 112     # The most reliable way is to check the source code, the AST will contain
 113     # a Lambda node instead of a FunctionDef
 114     if o.__name__ == '<lambda>':
 115       raise NotImplementedError(
 116           'lambda functions are not yet supported; declare the function'
 117           ' using def instead: %s' % o)
 118     else:
 119       node, name, ns = function_to_graph(o, program_ctx, arg_values, arg_types)
 120   elif tf_inspect.ismethod(o):
 121     node, name, ns = function_to_graph(o, program_ctx, arg_values, arg_types)
 122   # TODO(mdan,yashkatariya): Remove when object conversion is implemented.
 123   elif hasattr(o, '__class__'):
 124     raise NotImplementedError(
 125         'Object conversion is not yet supported. If you are '
 126         'trying to convert code that uses an existing object, '
 127         'try including the creation of that object in the '
 128         'conversion. For example, instead of converting the method '
 129         'of a class, try converting the entire class instead. '
 130         'See https://github.com/tensorflow/tensorflow/blob/master/tensorflow/'
 131         'contrib/autograph/README.md#using-the-functional-api '
 132         'for more information.')
 133   else:
 134     raise ValueError(
 135         'Entity "%s" has unsupported type "%s". Only functions and classes are '
 136         'supported for now.' % (o, type(o)))
 137 
 138   # TODO(mdan): This is temporary. it should be created using a converter.
 139   # TODO(mdan): The attribute should be added with a helper, not directly.
 140   # The helper can ensure there are no collisions.
 141   template = '''
 142       entity.autograph_info__ = {}
 143   '''
 144   node.extend(templates.replace(template, entity=name))
 145 
 146   program_ctx.add_to_cache(o, node)
 147 
 148   if program_ctx.recursive:
 149     while True:
 150       candidate = None
 151       for obj in program_ctx.name_map.keys():
 152         if obj not in program_ctx.dependency_cache:
 153           candidate = obj
 154           break
 155       if candidate is None:
 156         break
 157       if (hasattr(candidate, 'im_class') and
 158           getattr(candidate, 'im_class') not in program_ctx.partial_types):
 159         # Class members are converted with their objects, unless they're
 160         # only converted partially.
 161         continue
 162       entity_to_graph(candidate, program_ctx, {}, {})
 163 
 164   return node, name, ns

 167 def class_to_graph(c, program_ctx):
 168   """Specialization of `entity_to_graph` for classes."""
 169   converted_members = {}
 170   method_filter = lambda m: tf_inspect.isfunction(m) or tf_inspect.ismethod(m)
 171   members = tf_inspect.getmembers(c, predicate=method_filter)
 172   if not members:
 173     raise ValueError('Cannot convert %s: it has no member methods.' % c)
 174 
 175   class_namespace = {}
 176   for _, m in members:
 177     # Only convert the members that are directly defined by the class.
 178     if inspect_utils.getdefiningclass(m, c) is not c:
 179       continue
 180     node, _, namespace = function_to_graph(
 181         m,
 182         program_ctx=program_ctx,
 183         arg_values={},
 184         arg_types={'self': (c.__name__, c)},
 185         owner_type=c,
 186         rewrite_errors=False)
 187     if class_namespace is None:
 188       class_namespace = namespace
 189     else:
 190       class_namespace.update(namespace)
 191     converted_members[m] = node[0]
 192   namer = program_ctx.new_namer(class_namespace)
 193   class_name = namer.compiled_class_name(c.__name__, c)
 194 
 195   # TODO(mdan): This needs to be explained more thoroughly.
 196   # Process any base classes: if the superclass if of a whitelisted type, an
 197   # absolute import line is generated. Otherwise, it is marked for conversion
 198   # (as a side effect of the call to namer.compiled_class_name() followed by
 199   # program_ctx.update_name_map(namer)).
 200   output_nodes = []
 201   renames = {}
 202   base_names = []
 203   for base in c.__bases__:
 204     if isinstance(object, base):
 205       base_names.append('object')
 206       continue
 207     if is_whitelisted_for_graph(base):
 208       alias = namer.new_symbol(base.__name__, ())
 209       output_nodes.append(
 210           gast.ImportFrom(
 211               module=base.__module__,
 212               names=[gast.alias(name=base.__name__, asname=alias)],
 213               level=0))
 214     else:
 215       # This will trigger a conversion into a class with this name.
 216       alias = namer.compiled_class_name(base.__name__, base)
 217     base_names.append(alias)
 218     renames[qual_names.QN(base.__name__)] = qual_names.QN(alias)
 219   program_ctx.update_name_map(namer)
 220 
 221   # Generate the definition of the converted class.
 222   bases = [gast.Name(n, gast.Load(), None) for n in base_names]
 223   class_def = gast.ClassDef(
 224       class_name,
 225       bases=bases,
 226       keywords=[],
 227       body=list(converted_members.values()),
 228       decorator_list=[])
 229   # Make a final pass to replace references to the class or its base classes.
 230   # Most commonly, this occurs when making super().__init__() calls.
 231   # TODO(mdan): Making direct references to superclass' superclass will fail.
 232   class_def = qual_names.resolve(class_def)
 233   renames[qual_names.QN(c.__name__)] = qual_names.QN(class_name)
 234   class_def = ast_util.rename_symbols(class_def, renames)
 235 
 236   output_nodes.append(class_def)
 237 
 238   return output_nodes, class_name, class_namespace

 241 def _add_reserved_symbol(namespace, name, entity):
 242   if name not in namespace:
 243     namespace[name] = entity
 244   elif namespace[name] != entity:
 245     raise ValueError('The name "%s" is reserved and may not be used.' % name)

 251 def _add_self_references(namespace, autograph_module):
 252   """Adds namespace references to the module that exposes the api itself."""
 253   global ag_internal
 254   if ag_internal is None:
 255     # Craft a module that exposes parts of the external API as well as certain
 256     # internal modules.
 257     ag_internal = imp.new_module('autograph')
 258     ag_internal.converted_call = autograph_module.converted_call
 259     ag_internal.ConversionOptions = autograph_module.ConversionOptions
 260     ag_internal.utils = utils
 261     ag_internal.function_scope = function_wrapping.function_scope
 262     ag_internal.rewrite_graph_construction_error = (
 263         errors.rewrite_graph_construction_error)
 264     # TODO(mdan): Add safeguards against name clashes.
 265     # We don't want to create a submodule because we want the operators to be
 266     # accessible as ag__.<operator>
 267     ag_internal.__dict__.update(operators.__dict__)
 268 
 269   _add_reserved_symbol(namespace, 'ag__', ag_internal)

 272 def function_to_graph(f,
 273                       program_ctx,
 274                       arg_values,
 275                       arg_types,
 276                       owner_type=None,
 277                       rewrite_errors=True):
 278   """Specialization of `entity_to_graph` for callable functions."""
 279 
 280   node, source = parser.parse_entity(f)
 281   node = node.body[0]
 282   origin_info.resolve(node, source, f)
 283   namespace = inspect_utils.getnamespace(f)
 284   _add_self_references(namespace, program_ctx.autograph_module)
 285   namer = program_ctx.new_namer(namespace)
 286 
 287   entity_info = transformer.EntityInfo(
 288       source_code=source,
 289       source_file='<fragment>',
 290       namespace=namespace,
 291       arg_values=arg_values,
 292       arg_types=arg_types,
 293       owner_type=owner_type)
 294   context = converter.EntityContext(namer, entity_info, program_ctx)
 295   node = node_to_graph(node, context, rewrite_errors=rewrite_errors)
 296 
 297   # TODO(mdan): This somewhat duplicates the call rename logic in call_trees.py
 298   new_name, did_rename = namer.compiled_function_name(f.__name__, f, owner_type)
 299   if not did_rename:
 300     new_name = f.__name__
 301     if node.name != f.__name__:
 302       raise NotImplementedError('Strange corner case. Send us offending code!')
 303   node.name = new_name
 304 
 305   program_ctx.update_name_map(namer)
 306   # TODO(mdan): Use this at compilation.
 307 
 308   return [node], new_name, namespace

 311 def node_to_graph(node, context, rewrite_errors=True):
 312   """Convert Python code to equivalent TF graph mode code.
 313 
 314   Args:
 315     node: AST, the code to convert.
 316     context: converter.EntityContext
 317     rewrite_errors: Boolean, whether or not to rewrite the error traceback.
 318 
 319   Returns:
 320     A tuple (node, deps):
 321         * node: A Python ast node, representing the converted code.
 322         * deps: A set of strings, the fully qualified names of entity
 323             dependencies that this node has.
 324   """
 325   # TODO(mdan): Insert list_comprehensions somewhere.
 326 
 327   node = converter.standard_analysis(node, context, is_initial=True)
 328   # Past this point, line numbers are no longer accurate so we ignore the
 329   # source.
 330   # TODO(mdan): Is it feasible to reconstruct intermediate source code?
 331   context.info.source_code = None
 332 
 333   node = converter.apply_(node, context, decorators)
 334   node = converter.apply_(node, context, directives)
 335   node = converter.apply_(node, context, break_statements)
 336   node = converter.apply_(node, context, asserts)
 337   # Note: sequencing continue canonicalization before for loop one avoids
 338   # dealing with the extra loop increment operation that the for
 339   # canonicalization creates.
 340   node = converter.apply_(node, context, continue_statements)
 341   context.info.namespace['len'] = len
 342   node = converter.apply_(node, context, return_statements)
 343   node = converter.apply_(node, context, lists)
 344   node = converter.apply_(node, context, slices)
 345   node = converter.apply_(node, context, builtin_functions)
 346   node = converter.apply_(node, context, call_trees)
 347   node = converter.apply_(node, context, control_flow)
 348   node = converter.apply_(node, context, conditional_expressions)
 349   node = converter.apply_(node, context, logical_expressions)
 350   node = converter.apply_(node, context, side_effect_guards)
 351   node = converter.apply_(node, context, function_scopes)
 352   if rewrite_errors:
 353     node = converter.apply_(node, context, error_handlers)
 354   return node
