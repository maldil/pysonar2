  42 class ConversionOptions(
  43     collections.namedtuple('ConversionOptions',
  44                            ('recursive', 'verbose', 'strip_decorators',
  45                             'force_conversion', 'arg_types'))):
  46   """Container for conversion flags.
  47 
  48   Attributes:
  49     recursive: bool, whether to recursively convert any user functions or
  50         classes that the converted function may use.
  51     verbose: bool, whether to log the compiled code.
  52     strip_decorators: Tuple[Callable], contains decorators that should be in
  53         excluded from the compiled output. By default, when converting a
  54         function before the decorators are applied, the compiled output will
  55         include those decorators.
  56     force_conversion: bool, whether to force convertinng the target entity.
  57         When force_conversion is turned off, the converter may decide to
  58         return the function as-is.
  59     arg_types: Optional[Dict[Text, Type]], type hints for symbols including
  60         function arguments.
  61   """
  62 
  63   @classmethod
  64   def new(cls,
  65           recursive=False,
  66           verbose=False,
  67           strip_decorators=None,
  68           force_conversion=False,
  69           arg_types=None):
  70     return cls(recursive=recursive,
  71                verbose=verbose,
  72                strip_decorators=strip_decorators or (),
  73                force_conversion=force_conversion,
  74                arg_types=arg_types or {})
  75

  77 # TODO(mdan): This should behave like to_graph (e.g. convert statically).
  78 def convert(recursive=False, verbose=False):
  79   """Decorator that compiles a function to use TensorFlow ops.
  80 
  81   The decorator is dynamic - it recompiles the target whenever the decorated
  82   function is called. This means the parameter values are known at conversion.
  83   It also means that repeated calls with different types of parameters will be
  84   correctly processed.
  85 
  86   Args:
  87     recursive: bool, whether to recursively convert any functions or classes
  88         that the converted function may use.
  89     verbose: bool, whether to output the compiled code in the logs.
  90 
  91   Returns:
  92     Callable, a decorator that converts the given function into an equivalent
  93     function that uses TensorFlow ops.
  94   """
  95   def decorator(f):
  96     """Decorator implementation."""
  97 
  98     @functools.wraps(f)
  99     def wrapper(*args, **kwargs):
 100       return converted_call(
 101           f,
 102           ConversionOptions.new(
 103               recursive=recursive,
 104               verbose=verbose,
 105               force_conversion=True,
 106           ), *args, **kwargs)
 107 
 108     wrapper = tf_decorator.make_decorator(f, wrapper)
 109 
 110     # Sometimes the decorator is just desugared, making it impossible to detect.
 111     # This attribute makes detection easier.
 112     setattr(wrapper, '__pyct_is_compile_decorator', True)
 113     return wrapper
 114 
 115   return decorator

 118 class RunMode(Enum):
 119   """Specifies the way a converted function or method should be executed in TF.
 120 
 121   The enum values have the following semantics:
 122 
 123    * GRAPH: Call this function directly, as-is. This is suitable for functions
 124        that were already designed for TF graphs and contain ops.
 125    * PY_FUNC: Wrap this function into a py_func op. This is suitable for code
 126        that will only run correctly in Python, for example code that renders
 127        to the display, reads keyboard input, etc.
 128   """
 129   GRAPH = 1
 130   PY_FUNC = 2
 131

 133 def do_not_convert(run_as=RunMode.GRAPH, return_dtypes=None):
 134   """Decorator that suppresses the conversion of a function.
 135 
 136   See also: docs/pyfunc_dtypes.md
 137 
 138   Args:
 139     run_as: RunMode, specifies how to use the function in TensorFlow.
 140     return_dtypes: Optional[Iterable[
 141         Union[tf.DType, utils.py_func.MatchDType]]], the return data types of
 142         the converted function, if run_as is RunMode.PY_FUNC. Ignored otherwise.
 143         May be set to None if the function has no return values.
 144 
 145   Returns:
 146     Callable, a decorator that wraps the original function.
 147   """
 148 
 149   def decorator(f):
 150     """Decorator implementation."""
 151 
 152     @functools.wraps(f)
 153     def graph_wrapper(*args, **kwargs):
 154       return f(*args, **kwargs)
 155 
 156     @functools.wraps(f)
 157     def py_func_wrapper(*args, **kwargs):
 158       if kwargs:
 159         raise NotImplementedError('RunMode.PY_FUNC does not yet support kwargs')
 160       # TODO(mdan): Add support for kwargs.
 161       return py_func.wrap_py_func(
 162           f, return_dtypes, args, kwargs, use_dummy_return=not return_dtypes)
 163 
 164     if run_as == RunMode.GRAPH:
 165       wrapper = graph_wrapper
 166     elif run_as == RunMode.PY_FUNC:
 167       wrapper = py_func_wrapper
 168     else:
 169       raise ValueError('unknown value for run_as: %s' % run_as)
 170 
 171     # Sometimes the decorator is just desugared, making it impossible to detect.
 172     # This attribute makes detection easier.
 173     setattr(wrapper, '__pyct_is_compile_decorator', True)
 174     return wrapper
 175 
 176   return decorator

 179 # TODO(mdan): Move to a private, undocumented module.
 180 def converted_call(f, options, *args, **kwargs):
 181   """Compiles a function call inline. For internal use only."""
 182   # TODO(mdan): This needs cleanup.
 183   # In particular, we may want to avoid renaming functions altogether.
 184   if not options.force_conversion and conversion.is_whitelisted_for_graph(f):
 185     return f(*args, **kwargs)
 186 
 187   unknown_arg_value = object()  # Sentinel for arguments of unknown value
 188 
 189   if inspect_utils.isbuiltin(f):
 190     return py_builtins.overload_of(f)(*args, **kwargs)
 191 
 192   if tf_inspect.isfunction(f) or tf_inspect.ismethod(f):
 193     # Regular functions
 194     target_entity = f
 195     arg_map_target = f
 196     effective_args = args
 197     f_class = inspect_utils.getmethodclass(f)
 198 
 199     if f_class is not None:
 200       partial_types = (f_class,)
 201     else:
 202       partial_types = ()
 203 
 204   elif tf_inspect.isclass(f):
 205     # Constructors
 206     target_entity = f
 207     arg_map_target = f.__init__
 208     effective_args = args
 209     partial_types = ()
 210 
 211   elif hasattr(f, '__call__') and hasattr(f, '__class__'):
 212     # Callable objects
 213     target_entity = f.__call__
 214     arg_map_target = f.__call__
 215     effective_args = (f,) + args
 216     partial_types = (f.__class__,)
 217 
 218   else:
 219     NotImplementedError('unknown callable type "%s"' % type(f))
 220 
 221   arg_values = tf_inspect.getcallargs(arg_map_target, *args, **kwargs)
 222   for name, arg in arg_values.items():
 223     if arg is unknown_arg_value:
 224       continue
 225     arg_class = arg.__class__
 226     # If arg_value_hints specifies any name, use that instead.
 227     if name not in options.arg_types:
 228       options.arg_types[name] = (arg_class.__name__, arg_class)
 229 
 230   # When called from within a decorator, this is the only indication that
 231   # the function is a method - it appears that the decorator is applied
 232   # before the method is bound.
 233   if not partial_types:
 234     if 'self' in arg_values:
 235       if tf_inspect.isclass(arg_values['self'].__class__):
 236         partial_types = (arg_values['self'].__class__,)
 237     elif 'cls' in arg_values:
 238       if tf_inspect.isclass(arg_values['cls']):
 239         partial_types = (arg_values['cls'],)
 240 
 241   converted_f = to_graph(
 242       target_entity,
 243       recursive=options.recursive,
 244       verbose=options.verbose,
 245       arg_values=arg_values,
 246       arg_types=options.arg_types,
 247       partial_types=partial_types,
 248       strip_decorators=options.strip_decorators)
 249   return converted_f(*effective_args, **kwargs)

 255 def to_graph(e,
 256              recursive=True,
 257              verbose=False,
 258              arg_values=None,
 259              arg_types=None,
 260              partial_types=None,
 261              strip_decorators=None):
 262   """Converts a Python entity into equivalent code that uses TensorFlow ops.
 263 
 264   Supported Python entities include:
 265     * functions
 266     * classes
 267 
 268   Classes are converted by converting all their methods into a new class.
 269 
 270   Args:
 271     e: Union[Callable, Type], the Python entity to convert.
 272     recursive: bool, whether to recursively convert any functions that the
 273         converted function may call.
 274     verbose: bool, whether to output the compiled code in the logs.
 275     arg_values: Optional[Dict[Text, Any]], value hints for symbols including
 276         function arguments.
 277     arg_types: Optional[Dict[Text, Type]], type hints for symbols including
 278         function arguments.
 279     partial_types: Set[Type], reserved for internal use.
 280     strip_decorators: Tuple[Callable], same as
 281         ConversionOptions.strip_decorators.
 282 
 283   Returns:
 284     Union[Callable, Type], the converted entity, which is the same kind as e
 285     (that is, a function is e is a function, a class if e is a class, etc.) but
 286     its code has been converted to use TF ops.
 287 
 288   Raises:
 289     ValueError: If the entity could not be converted.
 290   """
 291   if strip_decorators is None:
 292     strip_decorators = ()
 293   strip_decorators += (convert, do_not_convert, converted_call)
 294 
 295   program_ctx = converter.ProgramContext(
 296       recursive=recursive,
 297       autograph_decorators=strip_decorators,
 298       partial_types=partial_types,
 299       autograph_module=tf_inspect.getmodule(to_graph),
 300       uncompiled_modules=config.DEFAULT_UNCOMPILED_MODULES)
 301   _, name, namespace = conversion.entity_to_graph(e, program_ctx, arg_values,
 302                                                   arg_types)
 303 
 304   nodes = []
 305   for dep in reversed(program_ctx.conversion_order):
 306     nodes.extend(program_ctx.dependency_cache[dep])
 307 
 308   compiled_module, compiled_src = compiler.ast_to_object(
 309       nodes,
 310       source_prefix=program_ctx.required_imports,
 311       include_source_map=True)
 312 
 313   # The compiled code should see everything the entry entity saw.
 314   # TODO(mdan): This might not work well if the call tree spans modules?
 315   for key, val in namespace.items():
 316     # Avoid overwriting entities that have been transformed.
 317     if key not in compiled_module.__dict__:
 318       compiled_module.__dict__[key] = val
 319   compiled = getattr(compiled_module, name)
 320 
 321   # Need this so the source_mapping attribute is available for the context
 322   # manager to access for runtime errors.
 323   #
 324   # Note that compiler.ast_to_object attaches the source map 'ag_source_map__'
 325   # symbol to the compiled module.
 326   # TODO(mdan): Record this statically in the generated code.
 327   # TODO(mdan): Rename this attribute to 'autograph_info__'
 328   source_map_attribute_name = 'ag_source_map'
 329   if getattr(compiled, source_map_attribute_name, None) is not None:
 330     raise ValueError('cannot convert %s because is has an attribute '
 331                      '"%s", which is reserved for AutoGraph.' %
 332                      (compiled, source_map_attribute_name))
 333   setattr(compiled, source_map_attribute_name,
 334           compiled_module.__dict__['ag_source_map__'])
 335 
 336   if verbose:
 337     logging.info('Compiled output of %s:\n\n%s\n', e, compiled_src)
 338 
 339   return compiled

 342 def to_code(e,
 343             recursive=True,
 344             arg_values=None,
 345             arg_types=None,
 346             partial_types=None,
 347             indentation='  '):
 348   """Returns the equivalent code that uses TensorFlow ops.
 349 
 350   Also see: `to_graph`, `convert`
 351 
 352   Args:
 353     e: Union[Callable, Type], the Python entity to convert.
 354     recursive: bool, whether to recursively convert any functions that the
 355         converted function may call.
 356     arg_values: Optional[Dict[Text, Any]], value hints for symbols including
 357         function arguments.
 358     arg_types: Optional[Dict[Text, Type]], type hints for symbols including
 359         function arguments.
 360     partial_types: Set[Type], reserved for internal use.
 361     indentation: Text, when to use for each level of indentation.
 362 
 363   Returns:
 364     Text, the converted code.
 365   """
 366   program_ctx = converter.ProgramContext(
 367       recursive=recursive,
 368       autograph_decorators=(convert, do_not_convert, converted_call),
 369       partial_types=partial_types,
 370       autograph_module=tf_inspect.getmodule(to_graph),
 371       uncompiled_modules=config.DEFAULT_UNCOMPILED_MODULES)
 372   conversion.entity_to_graph(e, program_ctx, arg_values, arg_types)
 373 
 374   code = '\n'.join(
 375       compiler.ast_to_source(program_ctx.dependency_cache[dep], indentation)
 376       for dep in reversed(program_ctx.conversion_order))
 377 
 378   return program_ctx.required_imports + '\n\n' + code
