  26 def dynamic_list_append(target, element):
  27   """Converts a list append call inline."""
  28   if isinstance(target, tensor_array_ops.TensorArray):
  29     return target.write(target.size(), element)
  30   # TODO(mdan): What's the right way to check this?
  31   # TODO(mdan): We may not need this branch.
  32   # It may be possible to use TensorList alone if the loop body will not
  33   # require wrapping it, although we'd have to think about an autoboxing
  34   # mechanism for lists received as parameter.
  35   if isinstance(target, ops.Tensor):
  36     return list_ops.tensor_list_push_back(target, element)
  37 
  38   # Python targets (including TensorList): fallback to their original append.
  39   target.append(element)
  40   return target

  43 class TensorList(object):
  44   """Tensor list wrapper API-compatible with Python built-in list."""
  45 
  46   def __init__(self, shape, dtype):
  47     self.dtype = dtype
  48     self.shape = shape
  49     self.clear()
  50 
  51   def append(self, value):
  52     self.list_ = list_ops.tensor_list_push_back(self.list_, value)
  53 
  54   def pop(self):
  55     self.list_, value = list_ops.tensor_list_pop_back(self.list_, self.dtype)
  56     return value
  57 
  58   def clear(self):
  59     self.list_ = list_ops.empty_tensor_list(self.shape, self.dtype)
  60 
  61   def count(self):
  62     return list_ops.tensor_list_length(self.list_)
  63 
  64   def __getitem__(self, key):
  65     return list_ops.tensor_list_get_item(self.list_, key, self.dtype)
  66 
  67   def __setitem__(self, key, value):
  68     self.list_ = list_ops.tensor_list_set_item(self.list_, key, value)
