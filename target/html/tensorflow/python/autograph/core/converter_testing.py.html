  44 # TODO(mdan): We might be able to use the real namer here.
  45 class FakeNamer(object):
  46   """A fake namer that uses a global counter to generate unique names."""
  47 
  48   def __init__(self):
  49     self.i = 0
  50 
  51   def new_symbol(self, name_root, used):
  52     while True:
  53       self.i += 1
  54       name = '%s%d' % (name_root, self.i)
  55       if name not in used:
  56         return name
  57 
  58   def compiled_function_name(self,
  59                              original_fqn,
  60                              live_entity=None,
  61                              owner_type=None):
  62     del live_entity
  63     if owner_type is not None:
  64       return None, False
  65     return ('renamed_%s' % '_'.join(original_fqn)), True
  66

  68 class FakeNoRenameNamer(FakeNamer):
  69 
  70   def compiled_function_name(self, original_fqn, **_):
  71     return str(original_fqn), False

  74 class TestCase(test.TestCase):
  75   """Base class for unit tests in this module. Contains relevant utilities."""
  76 
  77   @contextlib.contextmanager
  78   def assertPrints(self, expected_result):
  79     try:
  80       out_capturer = six.StringIO()
  81       sys.stdout = out_capturer
  82       yield
  83       self.assertEqual(out_capturer.getvalue(), expected_result)
  84     finally:
  85       sys.stdout = sys.__stdout__
  86 
  87   @contextlib.contextmanager
  88   def compiled(self, node, namespace, *symbols):
  89     source = None
  90 
  91     self.dynamic_calls = []
  92     def converted_call(*args):
  93       """Mock version of api.converted_call."""
  94       self.dynamic_calls.append(args)
  95       return 7
  96 
  97     class ConversionOptions(object):
  98       """Mock version of api.ConversionOptions."""
  99 
 100       def __init__(self, recursive):
 101         self.recursive = recursive
 102 
 103       @classmethod
 104       def new(cls, recursive):
 105         cls(recursive)
 106 
 107     try:
 108       result, source = compiler.ast_to_object(node, include_source_map=True)
 109 
 110       result.tf = self.make_fake_mod('fake_tf', *symbols)
 111       fake_ag = self.make_fake_mod('fake_ag', converted_call, ConversionOptions)
 112       fake_ag.__dict__.update(operators.__dict__)
 113       fake_ag.__dict__['utils'] = utils
 114       fake_ag.__dict__['rewrite_graph_construction_error'] = (
 115           errors.rewrite_graph_construction_error)
 116       fake_ag.__dict__['function_scope'] = function_wrapping.function_scope
 117       result.__dict__['ag__'] = fake_ag
 118       for k, v in namespace.items():
 119         result.__dict__[k] = v
 120       yield result
 121     except Exception:  # pylint:disable=broad-except
 122       if source is None:
 123         print('Offending AST:\n%s' % pretty_printer.fmt(node, color=False))
 124       else:
 125         print('Offending compiled code:\n%s' % source)
 126       raise

 128   @contextlib.contextmanager
 129   def converted(self, entity, converter_module, namespace, *tf_symbols):
 130     node, ctx = self.prepare(entity, namespace)
 131     node = converter_module.transform(node, ctx)
 132     with self.compiled(node, namespace, *tf_symbols) as result:
 133       yield result
 134 
 135   def make_fake_mod(self, name, *symbols):
 136     fake_mod = imp.new_module(name)
 137     for s in symbols:
 138       if hasattr(s, '__name__'):
 139         setattr(fake_mod, s.__name__, s)
 140       elif hasattr(s, 'name'):
 141         # This is a bit of a hack, but works for things like tf.int32
 142         setattr(fake_mod, s.name, s)
 143       else:
 144         raise ValueError('can not attach %s - what should be its name?' % s)
 145     return fake_mod
 146 
 147   def attach_namespace(self, module, **ns):
 148     for k, v in ns.items():
 149       setattr(module, k, v)
 150 
 151   def prepare(self,
 152               test_fn,
 153               namespace,
 154               namer=None,
 155               arg_types=None,
 156               owner_type=None,
 157               recursive=True,
 158               autograph_decorators=()):
 159     node, source = parser.parse_entity(test_fn)
 160     node = node.body[0]
 161     if namer is None:
 162       namer = FakeNamer()
 163     program_ctx = converter.ProgramContext(
 164         recursive=recursive,
 165         autograph_decorators=autograph_decorators,
 166         partial_types=None,
 167         autograph_module=None,
 168         uncompiled_modules=config.DEFAULT_UNCOMPILED_MODULES)
 169     entity_info = transformer.EntityInfo(
 170         source_code=source,
 171         source_file='<fragment>',
 172         namespace=namespace,
 173         arg_values=None,
 174         arg_types=arg_types,
 175         owner_type=owner_type)
 176     ctx = converter.EntityContext(namer, entity_info, program_ctx)
 177     node = converter.standard_analysis(node, ctx, is_initial=True)
 178     return node, ctx
