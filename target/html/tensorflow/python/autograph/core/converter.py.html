  15 """Converter construction support.
  16 
  17 This module contains a base class for all converters, as well as supporting
  18 structures. These structures are referred to as contexts.
  19 
  20 The class hierarchy is as follows:
  21 
  22     <your converter>
  23       [extends] converter.Base
  24         [extends] transformer.Base
  25             [extends] gast.nodeTransformer
  26           [uses] transfomer.SourceInfo
  27         [uses] converter.EntityContext
  28           [uses] converter.ProgramContext
  29           [uses] transfomer.SourceInfo
  30 
  31 converter.Base is a specialization of transformer.Base for AutoGraph. It's a
  32 very lightweight subclass that adds a `ctx` attribute holding the corresponding
  33 EntityContext object (see below). Note that converters are not reusable, and
  34 `visit` will raise an error if called more than once.
  35 
  36 converter.EntityContext contains mutable state associated with an entity that
  37 the converter processes.
  38 
  39 converter.ProgramContext contains mutable state across related entities. For
  40 example, when converting several functions that call one another, the
  41 ProgramContext should be shared across these entities.
  42 
  43 Below is the overal flow at conversion:
  44 
  45     program_ctx = ProgramContext(<entities to convert>, <global settings>, ...)
  46     while <program_ctx has more entities to convert>:
  47       entity, source_info = <get next entity from program_ctx>
  48       entity_ctx = EntityContext(program_ctx, source_info)
  49       for <each ConverterClass>:
  50         converter = ConverterClass(entity_ctx)
  51 
  52         # May update entity_ctx and program_ctx
  53         entity = converter.visit(entity)
  54 
  55       <add entity's dependencies to program_ctx>
  56 
  57 Note that pyct contains a small number of transformers used for static analysis.
  58 These implement transformer.Base, rather than converter.Base, to avoid a
  59 dependency on AutoGraph.
  60 """

  89 class ProgramContext(object):
  90   """ProgramContext keeps track of converting function hierarchies.
  91 
  92   This object is mutable, and is updated during conversion. Not thread safe.
  93 
  94   Attributes:
  95     recursive: bool, whether to recursively convert any functions that the
  96         decorator function may call.
  97     autograph_decorators: Tuple[Callable, ...], decorator functions that belong
  98         to AutoGraph. These require special treatment.
  99     dependency_cache: Dict[Any, ast.AST], the original entities mapped to their
 100         converted AST
 101     additional_imports: Set[Any], additional entities which for any reason
 102         cannot be attached after loading and need to be explicitly imported
 103         in the generated code
 104     name_map: Dict[str, str], map of original entity name to the name of
 105         their converted counterparts
 106     autograph_module: Module, a reference to the autograph module. This
 107         needs to be specified by the caller to avoid circular dependencies.
 108     uncompiled_modules: Set[Tuple[str, ...]], with each tuple representing the
 109         fully qualified name of a package containing functions that will not be
 110         compiled.
 111     required_imports: str, containing an import statement on each line. These
 112         are all the imports necessary for the compiled code to run, in addition
 113         to the closures of each entity, which are attached dynamically.
 114   """
 115 
 116   def __init__(
 117       self,
 118       recursive,
 119       autograph_decorators,
 120       partial_types,
 121       autograph_module,
 122       uncompiled_modules,
 123   ):
 124     self.recursive = recursive
 125     self.autograph_decorators = autograph_decorators
 126     self.partial_types = partial_types if partial_types else ()
 127     self.autograph_module = autograph_module
 128     self.uncompiled_modules = uncompiled_modules
 129 
 130     self.conversion_order = []
 131     self.dependency_cache = {}
 132     self.additional_imports = set()
 133     self.name_map = {}
 134 
 135   @property
 136   def required_imports(self):
 137     """Returns a block containing all imports required by the converted code."""
 138     # TODO(mdan): Check that these don't clobber one another.
 139     return '\n'.join(config.COMPILED_IMPORT_STATEMENTS +
 140                      tuple(self.additional_imports))
 141 
 142   def new_namer(self, namespace):
 143     return naming.Namer(namespace, self.recursive, self.name_map,
 144                         self.partial_types)
 145 
 146   def update_name_map(self, namer):
 147     """Updates renamed_calls based on the recent activity from the namer.
 148 
 149     Whenever we convert a new entity, any references to other entities are being
 150     renamed to match their soon-to-be-converted counterparts. The namer keeps
 151     track of these renames. When conversion is complete, we copy those renames
 152     so that when those referenced entities are being converted, their new name
 153     matches.
 154 
 155     Args:
 156       namer: naming.Namer
 157 
 158     Raises:
 159       ValueError: when an entity was renamed twice and to different names.
 160     """
 161     # TODO(mdan): Have call_trees do this directly.
 162     # This is done so indirectly, via the namer, for historic reasons. But
 163     # now we can have the converter that does the rename record the new name
 164     # as well and skip this step altogether.
 165     for o, name in namer.renamed_calls.items():
 166       if o in self.name_map:
 167         if self.name_map[o] != name:
 168           raise ValueError(
 169               'Calls to %s were converted using multiple names (%s). This is '
 170               'possible when an entity with one of these names already '
 171               'existed. To fix, avoid using any of these names.' %
 172               (o, (name, self.name_map[o])))
 173       else:
 174         self.name_map[o] = name
 175 
 176   def add_to_cache(self, original_entity, converted_ast):
 177     self.conversion_order.append(original_entity)
 178     self.dependency_cache[original_entity] = converted_ast

 198 class Base(transformer.Base):
 199   """All converters should inherit from this class.
 200 
 201   Attributes:
 202     ctx: EntityContext
 203   """
 204 
 205   def __init__(self, ctx):
 206     super(Base, self).__init__(ctx.info)
 207     self.ctx = ctx  # Keeping this short because it's used frequently.
 208 
 209     self._used = False
 210     self._ast_depth = 0
 211 
 212   def get_definition_directive(self, node, directive, arg, default):
 213     """Returns the unique directive for a symbol, or a default if none exist.
 214 
 215     See lang/directives.py for details on directives.
 216 
 217     Args:
 218       node: ast.AST
 219       directive: Callable[..., Any]
 220       arg: str
 221       default: Any
 222 
 223     Raises:
 224       ValueError: if conflicting annotations have been found
 225     """
 226     defs = anno.getanno(node, anno.Static.ORIG_DEFINITIONS, ())
 227     if not defs:
 228       return default
 229 
 230     # TODO(mdan): Simplify this.
 231     arg_values = []
 232     for def_ in defs:
 233       if (directive not in def_.directives or
 234           arg not in def_.directives[directive]):
 235         continue
 236       arg_value = def_.directives[directive][arg]
 237       for prev_value in arg_values:
 238         if not ast_util.matches(arg_value, prev_value):
 239           qn = anno.getanno(node, anno.Basic.QN)
 240           raise ValueError('%s has ambiguous annotations for %s(%s): %s, %s' %
 241                            (qn, directive.__name__, arg,
 242                             compiler.ast_to_source(arg_value).strip(),
 243                             compiler.ast_to_source(prev_value).strip()))
 244       arg_values.append(arg_value)
 245 
 246     if not arg_values:
 247       return default
 248 
 249     arg_value, = arg_values
 250     return arg_value
 251 
 252   def visit(self, node):
 253     if not self._ast_depth:
 254       if self._used:
 255         raise ValueError('converter objects cannot be reused')
 256       self._used = True
 257 
 258     self._ast_depth += 1
 259     try:
 260       return super(Base, self).visit(node)
 261     finally:
 262       self._ast_depth -= 1

 281 def standard_analysis(node, context, is_initial=False):
 282   """Performs a complete static analysis of the given code.
 283 
 284   Args:
 285     node: ast.AST
 286     context: converter.EntityContext
 287     is_initial: bool, whether this is the initial analysis done on the input
 288         source code
 289 
 290   Returns:
 291     ast.AST, same as node, with the static analysis annotations added
 292   """
 293   # TODO(mdan): Clear static analysis here.
 294   # TODO(mdan): Consider not running all analyses every time.
 295   # TODO(mdan): Don't return a node because it's modified by reference.
 296   graphs = cfg.build(node)
 297   node = qual_names.resolve(node)
 298   node = activity.resolve(node, context.info, None)
 299   node = reaching_definitions.resolve(node, context.info, graphs, AnnotatedDef)
 300   node = liveness.resolve(node, context.info, graphs)
 301   node = live_values.resolve(node, context.info, config.PYTHON_LITERALS)
 302   node = type_info.resolve(node, context.info)
 303   # This second call allows resolving first-order class attributes.
 304   node = live_values.resolve(node, context.info, config.PYTHON_LITERALS)
 305   if is_initial:
 306     anno.dup(
 307         node,
 308         {
 309             anno.Static.DEFINITIONS: anno.Static.ORIG_DEFINITIONS,
 310         },
 311     )
 312   return node

 315 def apply_(node, context, converter_module):
 316   """Applies a converter to an AST.
 317 
 318   Args:
 319     node: ast.AST
 320     context: converter.EntityContext
 321     converter_module: converter.Base
 322 
 323   Returns:
 324     ast.AST, the result of applying converter to node
 325   """
 326   node = standard_analysis(node, context)
 327   node = converter_module.transform(node, context)
 328   return node
