  15 """Canonicalizes continue statements by de-sugaring into a control boolean."""
  16 
  17 from __future__ import absolute_import
  18 from __future__ import division
  19 from __future__ import print_function
  20 
  21 from tensorflow.python.autograph.core import converter
  22 from tensorflow.python.autograph.pyct import anno
  23 from tensorflow.python.autograph.pyct import templates
  24 from tensorflow.python.autograph.pyct.static_analysis.annos import NodeAnno
  25 
  26 
  27 # Tags for local state.
  28 CONTROL_VAR_NAME = 'control_var_name'
  29 CONTINUE_USED = 'continue_used'
  30 GUARD_CREATED = 'guard_created'
  31 CREATE_GUARD_NEXT = 'create_guard_next'
  32 
  33 
  34 class ContinueCanonicalizationTransformer(converter.Base):
  35   """Canonicalizes continue statements into additional conditionals."""
  36 
  37   def visit_Continue(self, node):
  38     self.set_local(CONTINUE_USED, True)
  39     template = """
  40       var_name = tf.constant(True)
  41     """
  42     return templates.replace(
  43         template, var_name=self.get_local(CONTROL_VAR_NAME))
  44 
  45   def _postprocess_statement(self, node):
  46     # Example of how the state machine below works:
  47     #
  48     #   1| stmt           # State: CONTINUE_USED = False
  49     #    |                # Action: none
  50     #   2| if cond:
  51     #   3|   continue     # State: CONTINUE_USED = True,
  52     #    |                #        GUARD_CREATED = False,
  53     #    |                #        CREATE_GUARD_NEXT = False
  54     #    |                # Action: set CREATE_GUARD_NEXT = True
  55     #   4| stmt           # State: CONTINUE_USED = True,
  56     #    |                #        GUARD_CREATED = False,
  57     #    |                #        CREATE_GUARD_NEXT = True
  58     #    |                # Action: create `if not continue_used`,
  59     #    |                #         set GUARD_CREATED = True
  60     #   5| stmt           # State: CONTINUE_USED = True, GUARD_CREATED = True
  61     #    |                # Action: none (will be wrapped under previously
  62     #    |                #         created if node)
  63 
  64     if self.get_local(CONTINUE_USED, False):
  65       if self.get_local(GUARD_CREATED, False):
  66         return node, None
  67 
  68       elif not self.get_local(CREATE_GUARD_NEXT, False):
  69         self.set_local(CREATE_GUARD_NEXT, True)
  70         return node, None
  71 
  72       else:
  73         self.set_local(GUARD_CREATED, True)
  74         template = """
  75           if not var_name:
  76             original_node
  77         """
  78         cond, = templates.replace(
  79             template,
  80             var_name=self.get_local(CONTROL_VAR_NAME),
  81             original_node=node)
  82         return cond, cond.body
  83     return node, None
  84 
  85   def _visit_loop_body(self, node, nodes):
  86     self.enter_local_scope()
  87     scope = anno.getanno(node, NodeAnno.BODY_SCOPE)
  88     continue_var = self.ctx.namer.new_symbol('continue_', scope.referenced)
  89     self.set_local(CONTROL_VAR_NAME, continue_var)
  90 
  91     nodes = self.visit_block(nodes, after_visit=self._postprocess_statement)
  92 
  93     if self.get_local(CONTINUE_USED, False):
  94       template = """
  95         var_name = tf.constant(False)
  96       """
  97       control_var_init = templates.replace(template, var_name=continue_var)
  98       nodes = control_var_init + nodes
  99 
 100     self.exit_local_scope()
 101     return nodes
 102 
 103   def _visit_non_loop_body(self, nodes):
 104     self.enter_local_scope(inherit=(CONTROL_VAR_NAME,))
 105     nodes = self.visit_block(nodes, after_visit=self._postprocess_statement)
 106     continue_used = self.get_local(CONTINUE_USED, False)
 107     self.exit_local_scope(keep=(CONTINUE_USED,))
 108     return nodes, continue_used
 109 
 110   def visit_While(self, node):
 111     node.test = self.visit(node.test)
 112     node.body = self._visit_loop_body(node, node.body)
 113     # A continue in the else clause applies to the containing scope.
 114     node.orelse, _ = self._visit_non_loop_body(node.orelse)
 115     return node
 116 
 117   def visit_For(self, node):
 118     node.target = self.generic_visit(node.target)
 119     node.iter = self.generic_visit(node.iter)
 120     node.body = self._visit_loop_body(node, node.body)
 121     # A continue in the else clause applies to the containing scope.
 122     node.orelse, _ = self._visit_non_loop_body(node.orelse)
 123     return node
 124 
 125   def visit_If(self, node):
 126     node.test = self.generic_visit(node.test)
 127     node.body, continue_used_body = self._visit_non_loop_body(node.body)
 128     node.orelse, continue_used_orelse = self._visit_non_loop_body(node.orelse)
 129     self.set_local(CONTINUE_USED, continue_used_body or continue_used_orelse)
 130     return node
 131 
 132   def visit_With(self, node):
 133     node.items = self.visit_block(node.items)
 134     node.body, _ = self._visit_non_loop_body(node.body)
 135     return node
 136 
 137 
 138 def transform(node, ctx):
 139   return ContinueCanonicalizationTransformer(ctx).visit(node)
