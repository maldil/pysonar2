  15 """Converter for list operations.
  16 
  17 This includes converting Python lists to TensorArray/TensorList.
  18 """

  43 # Tags for local state.
  44 POP_USES = 'pop_uses'

  47 class ListTransformer(converter.Base):
  48   """Converts lists and related operations to their TF counterpart."""
  49 
  50   def visit_List(self, node):
  51     node = self.generic_visit(node)
  52     template = """
  53       ag__.new_list(elements)
  54     """
  55     return templates.replace_as_expression(template, elements=node)
  56 
  57   def _replace_append_call(self, node):
  58     assert len(node.args) == 1
  59     assert isinstance(node.func, gast.Attribute)
  60     template = """
  61       target = ag__.list_append(target, element)
  62     """
  63     return templates.replace(
  64         template,
  65         target=node.func.value,
  66         element=node.args[0])
  67 
  68   def _replace_pop_call(self, node):
  69     # Expressions that use pop() are converted to a statement + expression.
  70     #
  71     # For example:
  72     #
  73     #   print(target.pop())
  74     #
  75     # ... is converted to:
  76     #
  77     #   target, target_pop = ag__.list_pop(target)
  78     #   print(target_pop)
  79     #
  80     # Here, we just generate the variable name and swap it in,
  81     # and _generate_pop_operation will handle the rest.
  82     #
  83     # Multiple uses of pop() are allowed:
  84     #
  85     #   print(tartget.pop(), target.pop())
  86     #   print(tartget.pop().pop())
  87     #
  88     assert isinstance(node.func, gast.Attribute)
  89     scope = anno.getanno(node, NodeAnno.ARGS_SCOPE)
  90     target_node = node.func.value
  91 
  92     # Attempt to use a related name if one exists. Otherwise use something
  93     # generic.
  94     if anno.hasanno(target_node, anno.Basic.QN):
  95       target_name = anno.getanno(target_node, anno.Basic.QN).ssf()
  96     else:
  97       target_name = 'list_'
  98     pop_var_name = self.ctx.namer.new_symbol(target_name, scope.referenced)
  99 
 100     pop_uses = self.get_local(POP_USES, [])
 101     pop_uses.append((node, pop_var_name))
 102     self.set_local(POP_USES, pop_uses)
 103 
 104     return templates.replace_as_expression('var_name', var_name=pop_var_name)
 105 
 106   def _replace_stack_call(self, node):
 107     assert len(node.args) == 1
 108     dtype = self.get_definition_directive(
 109         node.args[0],
 110         directives.set_element_type,
 111         'dtype',
 112         default=templates.replace_as_expression('None'))
 113     template = """
 114       ag__.list_stack(
 115           target,
 116           opts=ag__.ListStackOpts(
 117               element_dtype=dtype,
 118               original_call=orig_call))
 119     """
 120     return templates.replace_as_expression(
 121         template,
 122         dtype=dtype,
 123         target=node.args[0],
 124         orig_call=node.func)
 125 
 126   def visit_Call(self, node):
 127     node = self.generic_visit(node)
 128 
 129     # TODO(mdan): This is insufficient if target is a function argument.
 130     # In the case of function arguments, we need to add the list to the
 131     # function's return value, because it is being modified.
 132     # TODO(mdan): Checking just the name is brittle, can it be improved?
 133     if isinstance(node.func, gast.Attribute):
 134       func_name = node.func.attr
 135       if func_name == 'append' and (len(node.args) == 1):
 136         node = self._replace_append_call(node)
 137       elif func_name == 'pop' and (len(node.args) <= 1):
 138         node = self._replace_pop_call(node)
 139       elif (func_name == 'stack' and (len(node.args) == 1) and
 140             (not node.keywords or node.keywords[0].arg == 'strict')):
 141         # This avoids false positives with keyword args.
 142         # TODO(mdan): handle kwargs properly.
 143         node = self._replace_stack_call(node)
 144 
 145     return node
 146 
 147   def _generate_pop_operation(self, original_call_node, pop_var_name):
 148     assert isinstance(original_call_node.func, gast.Attribute)
 149 
 150     if original_call_node.args:
 151       pop_element = original_call_node.args[0]
 152     else:
 153       pop_element = parser.parse_expression('None')
 154 
 155     # The call will be something like "target.pop()", and the dtype is hooked to
 156     # target, hence the func.value.
 157     # TODO(mdan): For lists of lists, this won't work.
 158     # The reason why it won't work is because it's unclear how to annotate
 159     # the list as a "list of lists with a certain element type" when using
 160     # operations like `l.pop().pop()`.
 161     dtype = self.get_definition_directive(
 162         original_call_node.func.value,
 163         directives.set_element_type,
 164         'dtype',
 165         default=templates.replace_as_expression('None'))
 166     shape = self.get_definition_directive(
 167         original_call_node.func.value,
 168         directives.set_element_type,
 169         'shape',
 170         default=templates.replace_as_expression('None'))
 171 
 172     template = """
 173       target, pop_var_name = ag__.list_pop(
 174           target, element,
 175           opts=ag__.ListPopOpts(element_dtype=dtype, element_shape=shape))
 176     """
 177     return templates.replace(
 178         template,
 179         target=original_call_node.func.value,
 180         pop_var_name=pop_var_name,
 181         element=pop_element,
 182         dtype=dtype,
 183         shape=shape)
 184 
 185   def _postprocess_statement(self, node):
 186     """Inserts any separate pop() calls that node may use."""
 187     pop_uses = self.get_local(POP_USES, None)
 188     if pop_uses:
 189       replacements = []
 190       for original_call_node, pop_var_name in pop_uses:
 191         replacements.extend(
 192             self._generate_pop_operation(original_call_node, pop_var_name))
 193       replacements.append(node)
 194       node = replacements
 195     self.exit_local_scope()
 196     return node, None
 197 
 198   # TODO(mdan): Should we have a generic visit_block instead?
 199   # Right now it feels that a visit_block would add too much magic that's
 200   # hard to follow.
 201 
 202   def _visit_and_process_block(self, block):
 203     return self.visit_block(
 204         block,
 205         before_visit=self.enter_local_scope,
 206         after_visit=self._postprocess_statement)
 207 
 208   def visit_FunctionDef(self, node):
 209     node.args = self.generic_visit(node.args)
 210     node.decorator_list = self.visit_block(node.decorator_list)
 211     node.body = self._visit_and_process_block(node.body)
 212     return node
 213 
 214   def visit_For(self, node):
 215     node.target = self.visit(node.target)
 216     node.body = self._visit_and_process_block(node.body)
 217     node.orelse = self._visit_and_process_block(node.orelse)
 218     return node
 219 
 220   def visit_While(self, node):
 221     node.test = self.visit(node.test)
 222     node.body = self._visit_and_process_block(node.body)
 223     node.orelse = self._visit_and_process_block(node.orelse)
 224     return node
 225 
 226   def visit_If(self, node):
 227     node.test = self.visit(node.test)
 228     node.body = self._visit_and_process_block(node.body)
 229     node.orelse = self._visit_and_process_block(node.orelse)
 230     return node
 231 
 232   def visit_With(self, node):
 233     node.items = self.visit_block(node.items)
 234     node.body = self._visit_and_process_block(node.body)
 235     return node
 236 
 237 
 238 def transform(node, ctx):
 239   return ListTransformer(ctx).visit(node)
