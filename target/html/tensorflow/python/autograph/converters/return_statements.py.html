  31 class BodyVisitor(converter.Base):
  32   """Walks breadth- or depth-first the list-of-nodes bodies of AST nodes."""
  33 
  34   def __init__(self, ctx, depth_first=False):
  35     super(BodyVisitor, self).__init__(ctx)
  36     self.depth_first = depth_first
  37     self.changes_made = False
  38 
  39   def visit_nodelist(self, nodelist):
  40     for node in nodelist:
  41       if isinstance(node, list):
  42         node = self.visit_nodelist(node)
  43       else:
  44         node = self.generic_visit(node)
  45     return nodelist
  46 
  47   def visit_If(self, node):
  48     if self.depth_first:
  49       node = self.generic_visit(node)
  50     node.body = self.visit_nodelist(node.body)
  51     node.orelse = self.visit_nodelist(node.orelse)
  52     if not self.depth_first:
  53       node = self.generic_visit(node)
  54     return node
  55 
  56   def visit_For(self, node):
  57     if self.depth_first:
  58       node = self.generic_visit(node)
  59     node.body = self.visit_nodelist(node.body)
  60     node.orelse = self.visit_nodelist(node.orelse)
  61     if not self.depth_first:
  62       node = self.generic_visit(node)
  63     return node
  64 
  65   def visit_While(self, node):
  66     if self.depth_first:
  67       node = self.generic_visit(node)
  68     node.body = self.visit_nodelist(node.body)
  69     node.orelse = self.visit_nodelist(node.orelse)
  70     if not self.depth_first:
  71       node = self.generic_visit(node)
  72     return node
  73 
  74   def visit_Try(self, node):
  75     if self.depth_first:
  76       node = self.generic_visit(node)
  77     node.body = self.visit_nodelist(node.body)
  78     node.orelse = self.visit_nodelist(node.orelse)
  79     node.finalbody = self.visit_nodelist(node.finalbody)
  80     for i in range(len(node.handlers)):
  81       node.handlers[i].body = self.visit_nodelist(node.handlers[i].body)
  82     if not self.depth_first:
  83       node = self.generic_visit(node)
  84     return node
  85 
  86   def visit_With(self, node):
  87     if self.depth_first:
  88       node = self.generic_visit(node)
  89     node.body = self.visit_nodelist(node.body)
  90     if not self.depth_first:
  91       node = self.generic_visit(node)
  92     return node
  93 
  94   def visit_FunctionDef(self, node):
  95     if self.depth_first:
  96       node = self.generic_visit(node)
  97     node.body = self.visit_nodelist(node.body)
  98     self.generic_visit(node)
  99     if not self.depth_first:
 100       node = self.generic_visit(node)
 101     return node

 104 class FoldElse(BodyVisitor):
 105 
 106   def visit_nodelist(self, nodelist):
 107     for i in range(len(nodelist)):
 108       node = nodelist[i]
 109       if isinstance(node, gast.If):
 110         true_branch_returns = isinstance(node.body[-1], gast.Return)
 111         false_branch_returns = len(node.orelse) and isinstance(
 112             node.orelse[-1], gast.Return)
 113         # If the last node in the if body is a return,
 114         # then every line after this if statement effectively
 115         # belongs in the else.
 116         if true_branch_returns and not false_branch_returns:
 117           for j in range(i + 1, len(nodelist)):
 118             nodelist[i].orelse.append(ast_util.copy_clean(nodelist[j]))
 119           if nodelist[i + 1:]:
 120             self.changes_made = True
 121           return nodelist[:i + 1]
 122         elif not true_branch_returns and false_branch_returns:
 123           for j in range(i + 1, len(nodelist)):
 124             nodelist[i].body.append(ast_util.copy_clean(nodelist[j]))
 125           if nodelist[i + 1:]:
 126             self.changes_made = True
 127           return nodelist[:i + 1]
 128         elif true_branch_returns and false_branch_returns:
 129           if nodelist[i + 1:]:
 130             raise ValueError(
 131                 'Unreachable code after conditional where both branches return.'
 132             )
 133           return nodelist
 134       elif isinstance(node, gast.Return) and nodelist[i + 1:]:
 135         raise ValueError(
 136             'Cannot have statements after a return in the same basic block')
 137     return nodelist

 140 def contains_return(node):
 141   for n in gast.walk(node):
 142     if isinstance(n, gast.Return):
 143       return True
 144   return False

 147 class LiftReturn(converter.Base):
 148   """Move return statements out of If and With blocks."""
 149 
 150   def __init__(self, ctx):
 151     super(LiftReturn, self).__init__(ctx)
 152     self.changes_made = False
 153     self.common_return_name = None
 154 
 155   def visit_If(self, node):
 156     # Depth-first traversal of if statements
 157     node = self.generic_visit(node)
 158 
 159     # We check if both branches return, and if so, lift the return out of the
 160     # conditional. We don't enforce that the true and false branches either
 161     # both return or both do not, because FoldElse might move a return
 162     # into a branch after this transform completes. FoldElse and LiftReturn
 163     # are alternately run until the code reaches a fixed point.
 164     true_branch_returns = isinstance(node.body[-1], gast.Return)
 165     false_branch_returns = len(node.orelse) and isinstance(
 166         node.orelse[-1], gast.Return)
 167     if true_branch_returns and false_branch_returns:
 168       node.body[-1] = templates.replace(
 169           'a = b', a=self.common_return_name, b=node.body[-1].value)[0]
 170       node.orelse[-1] = templates.replace(
 171           'a = b', a=self.common_return_name, b=node.orelse[-1].value)[0]
 172       return_node = templates.replace('return a', a=self.common_return_name)[0]
 173       self.changes_made = True
 174       return [node, return_node]
 175     else:
 176       return node
 177 
 178   def visit_With(self, node):
 179     # Depth-first traversal of syntax
 180     node = self.generic_visit(node)
 181 
 182     # If the with statement returns, lift the return
 183     if isinstance(node.body[-1], gast.Return):
 184       node.body[-1] = templates.replace(
 185           'a = b', a=self.common_return_name, b=node.body[-1].value)[0]
 186       return_node = templates.replace('return a', a=self.common_return_name)[0]
 187       node = self.generic_visit(node)
 188       self.changes_made = True
 189       return [node, return_node]
 190     else:
 191       return node
 192 
 193   def visit_FunctionDef(self, node):
 194     # Ensure we're doing depth-first traversal
 195     last_return_name = self.common_return_name
 196     body_scope = anno.getanno(node, NodeAnno.BODY_SCOPE)
 197     referenced_names = body_scope.referenced
 198     self.common_return_name = self.ctx.namer.new_symbol('return_',
 199                                                         referenced_names)
 200     node = self.generic_visit(node)
 201     self.common_return_name = last_return_name
 202     return node

 205 class DetectReturnInUnsupportedControlFlow(gast.NodeVisitor):
 206   """Throws an error if code returns inside loops or try/except."""
 207 
 208   # First, throw an error if we detect a return statement in a loop.
 209   # TODO(alexbw): we need to learn to handle returns inside a loop,
 210   # but don't currently have the TF constructs to do so (need something
 211   # that looks vaguely like a goto).
 212 
 213   def __init__(self):
 214     self.cant_return = False
 215     self.function_level = 0
 216     super(DetectReturnInUnsupportedControlFlow, self).__init__()
 217 
 218   def visit_While(self, node):
 219     self.cant_return = True
 220     self.generic_visit(node)
 221     self.cant_return = False
 222 
 223   def visit_For(self, node):
 224     self.cant_return = True
 225     self.generic_visit(node)
 226     self.cant_return = False
 227 
 228   def visit_Try(self, node):
 229     self.cant_return = True
 230     self.generic_visit(node)
 231     self.cant_return = False
 232 
 233   def visit_FunctionDef(self, node):
 234     if not self.function_level:
 235       self.function_level += 1
 236       self.generic_visit(node)
 237       self.function_level -= 1
 238 
 239   def visit_Return(self, node):
 240     if self.cant_return:
 241       raise ValueError(
 242           '`return` statements are not supported in loops. '
 243           'Try assigning to a variable in the while loop, and returning '
 244           'outside of the loop')
 245

 247 class DetectReturnInConditional(gast.NodeVisitor):
 248   """Assert that no return statements are present in conditionals."""
 249 
 250   def __init__(self):
 251     self.cant_return = False
 252     self.function_level = 0
 253     super(DetectReturnInConditional, self).__init__()
 254 
 255   def visit_If(self, node):
 256     self.cant_return = True
 257     self.generic_visit(node)
 258     self.cant_return = False
 259 
 260   def visit_FunctionDef(self, node):
 261     if not self.function_level:
 262       self.function_level += 1
 263       self.generic_visit(node)
 264       self.function_level -= 1
 265 
 266   def visit_Return(self, node):
 267     if self.cant_return:
 268       raise ValueError(
 269           'After transforms, a conditional contained a `return `statement, '
 270           'which is not allowed. This is a bug, and should not happen.')
 271 
 272 
 273 class DetectReturnInFunctionDef(gast.NodeVisitor):
 274 
 275   def visit_FunctionDef(self, node):
 276     self.generic_visit(node)
 277     if not contains_return(node):
 278       raise ValueError(
 279           'Each function definition should contain at least one return.')

 282 def transform(node, ctx):
 283   """Ensure a function has only a single return.
 284 
 285   This transforms an AST node with multiple returns successively into containing
 286   only a single return node.
 287   There are a few restrictions on what we can handle:
 288    - An AST being transformed must contain at least one return.
 289    - No returns allowed in loops. We have to know the type of the return value,
 290    and we currently don't have either a type inference system to discover it,
 291    nor do we have a mechanism for late type binding in TensorFlow.
 292    - After all transformations are finished, a Return node is not allowed inside
 293    control flow. If we were unable to move a return outside of control flow,
 294    this is an error.
 295 
 296   Args:
 297      node: ast.AST
 298      ctx: converter.EntityContext
 299 
 300   Returns:
 301      new_node: an AST with a single return value
 302 
 303   Raises:
 304     ValueError: if the AST is structured so that we can't perform the
 305    transform.
 306   """
 307   # Make sure that the function has at least one return statement
 308   # TODO(alexbw): turning off this assertion for now --
 309   # we need to not require this in e.g. class constructors.
 310   # DetectReturnInFunctionDef().visit(node)
 311 
 312   # Make sure there's no returns in unsupported locations (loops, try/except)
 313   DetectReturnInUnsupportedControlFlow().visit(node)
 314 
 315   while True:
 316 
 317     # Try to lift all returns out of if statements and with blocks
 318     lr = LiftReturn(ctx)
 319     node = lr.visit(node)
 320     changes_made = lr.changes_made
 321     fe = FoldElse(ctx)
 322     node = fe.visit(node)
 323     changes_made = changes_made or fe.changes_made
 324 
 325     if not changes_made:
 326       break
 327 
 328   # Make sure we've scrubbed all returns from conditionals
 329   DetectReturnInConditional().visit(node)
 330 
 331   return node
