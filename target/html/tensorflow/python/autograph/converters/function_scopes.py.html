  15 """Wraps the body of a converted function with auxiliary constructs."""
  16 
  17 from __future__ import absolute_import
  18 from __future__ import division
  19 from __future__ import print_function
  20 
  21 import gast
  22 
  23 from tensorflow.python.autograph.core import converter
  24 from tensorflow.python.autograph.pyct import templates
  25 
  26 
  27 class FunctionBodyTransformer(converter.Base):
  28   """Wraps function bodies around autograph-specific boilerplate."""
  29 
  30   def _name_for_current_scope(self):
  31     innermost = self.enclosing_entities[-1]
  32     if len(self.enclosing_entities) > 1:
  33       parent = self.enclosing_entities[-2]
  34       if isinstance(parent, gast.ClassDef):
  35         # Methods also take the name of their class.
  36         name = '%s/%s' % (parent.name, innermost.name)
  37       else:
  38         name = innermost.name
  39     else:
  40       name = innermost.name
  41 
  42     # Sanitize the name.
  43     # See https://www.tensorflow.org/api_docs/python/tf/Graph#name_scope
  44     # TensorFlow doesn't like leading underscores at the top level.
  45     while name[0] == '_':
  46       name = name[1:]
  47     return name
  48 
  49   def visit_FunctionDef(self, node):
  50     node = self.generic_visit(node)
  51 
  52     final_body = []
  53     indented_body = node.body
  54     if node.body:
  55       first_statement = node.body[0]
  56       # Skip the docstring, if any.
  57       if (isinstance(first_statement, gast.Expr) and
  58           isinstance(first_statement.value, gast.Str)):
  59         indented_body = indented_body[1:]
  60         final_body.append(first_statement)
  61 
  62     template = """
  63       with ag__.function_scope(scope_name):
  64         body
  65     """
  66     scoped_body = templates.replace(
  67         template,
  68         scope_name=gast.Str(self._name_for_current_scope()),
  69         body=indented_body)
  70     final_body.extend(scoped_body)
  71     node.body = final_body
  72     return node
  73 
  74 
  75 def transform(node, ctx):
  76   return FunctionBodyTransformer(ctx).visit(node)
