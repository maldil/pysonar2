  15 """Adds guards against function calls with side effects.
  16 
  17 Only standalone calls are guarded.
  18 
  19 WARNING: This mechanism is incomplete. Particularly, it only guards the
  20 arguments passed to functions, and does not account for indirectly modified
  21 state.
  22 
  23 Example:
  24   y = tf.layers.dense(x)       # Creates TF variable 'foo'
  25   loss = loss(y)
  26   opt.minimize(loss)           # indirectly affects 'foo'
  27   z = tf.get_variable('foo')   # Indirectly affects `loss` and 'foo'
  28   # Here, `loss` can be guarded. But `z` cannot.
  29 
  30 # TODO(mdan): We should probably define a safe mode where we guard everything.
  31 """
  32

  47 class SymbolNamer(object):
  48   """Describes the interface for SideEffectGuardTransformer's namer."""
  49 
  50   def new_symbol(self, name_root, reserved_locals):
  51     """Generate a new unique function_name.
  52 
  53     Args:
  54       name_root: String, used as stem in the new name.
  55       reserved_locals: Set(string), additional local symbols that are reserved.
  56     Returns:
  57       String.
  58     """
  59     raise NotImplementedError()
  60 
  61 
  62 class SideEffectGuardTransformer(converter.Base):
  63   """Adds control dependencies to functions with side effects."""
  64 
  65   def _visit_and_reindent(self, nodes):
  66     new_nodes = []
  67     current_dest = new_nodes
  68     alias_map = {}
  69     reindent_requested = False
  70     for n in nodes:
  71       n = self.visit(n)
  72       # NOTE: the order in which these statements execute is important; in
  73       # particular, watch out for ending up with cycles in the AST.
  74       if alias_map:
  75         n = ast_util.rename_symbols(n, alias_map)
  76       if isinstance(n, (list, tuple)):
  77         current_dest.extend(n)
  78       else:
  79         current_dest.append(n)
  80       if anno.hasanno(n, anno.Basic.INDENT_BLOCK_REMAINDER):
  81         reindent_requested = True
  82         new_dest, new_alias_map = anno.getanno(
  83             n, anno.Basic.INDENT_BLOCK_REMAINDER)
  84         anno.delanno(n, anno.Basic.INDENT_BLOCK_REMAINDER)
  85         new_alias_map.update(alias_map)
  86         alias_map = new_alias_map
  87         current_dest = new_dest
  88     if reindent_requested and not current_dest:
  89       # TODO(mdan): There may still be something that could be done.
  90       raise ValueError('Unable to insert statement into the computation flow: '
  91                        'it is not followed by any computation which '
  92                        'the statement could gate.')
  93     return new_nodes
  94 
  95   def visit_FunctionDef(self, node):
  96     node.body = self._visit_and_reindent(node.body)
  97     return node
  98 
  99   def visit_With(self, node):
 100     node.body = self._visit_and_reindent(node.body)
 101     return node
 102 
 103   def visit_If(self, node):
 104     node.body = self._visit_and_reindent(node.body)
 105     node.orelse = self._visit_and_reindent(node.orelse)
 106     return node
 107 
 108   def visit_While(self, node):
 109     node.body = self._visit_and_reindent(node.body)
 110     node.orelse = self._visit_and_reindent(node.orelse)
 111     return node
 112 
 113   def visit_Expr(self, node):
 114     self.generic_visit(node)
 115     if isinstance(node.value, gast.Call):
 116       # Patterns of single function calls, like:
 117       #   opt.minimize(loss)
 118       # or:
 119       #   tf.py_func(...)
 120 
 121       # First, attempt to gate future evaluation of args. If that's not
 122       # possible, gate all remaining statements (and that may fail too, see
 123       # _visit_and_reindent.
 124       args_scope = anno.getanno(node.value, NodeAnno.ARGS_SCOPE)
 125       # NOTE: We can't guard object attributes because they may not be writable.
 126       # In addition, avoid renaming well-known names.
 127       # TODO(mdan): Move these names into config.
 128       unguarded_names = (qual_names.QN('self'), qual_names.QN('tf'))
 129       guarded_args = tuple(s for s in args_scope.used
 130                            if not s.is_composite() and s not in unguarded_names)
 131 
 132       # TODO(mdan): Include all arguments which depended on guarded_args too.
 133       # For example, the following will still cause a race:
 134       #   tf.assign(a, a + 1)
 135       #   b = a + 1
 136       #   tf.assign(a, a + 1)  # Control deps here should include `b`
 137       #   c = b + 1
 138       # Or maybe we should just raise an "unsafe assign" error?
 139 
 140       if guarded_args:
 141         # The aliases may need new names to avoid incorrectly making them local.
 142         # TODO(mdan): This is brutal. It will even rename modules - any fix?
 143         need_alias = tuple(
 144             s for s in guarded_args if s not in args_scope.parent.modified)
 145         aliased_new_names = tuple(
 146             qual_names.QN(
 147                 self.ctx.namer.new_symbol(
 148                     s.ssf(), args_scope.parent.referenced)) for s in need_alias)
 149         alias_map = dict(zip(need_alias, aliased_new_names))
 150         if len(guarded_args) == 1:
 151           s, = guarded_args
 152           aliased_guarded_args = alias_map.get(s, s)
 153         else:
 154           aliased_guarded_args = gast.Tuple(
 155               [alias_map.get(s, s).ast() for s in guarded_args], None)
 156 
 157         template = """
 158           with ag__.utils.control_dependency_on_returns(call):
 159             aliased_guarded_args = ag__.utils.alias_tensors(guarded_args)
 160         """
 161         control_deps_guard = templates.replace(
 162             template,
 163             call=node.value,
 164             aliased_guarded_args=aliased_guarded_args,
 165             guarded_args=guarded_args)[-1]
 166       else:
 167         alias_map = {}
 168 
 169         template = """
 170           with ag__.utils.control_dependency_on_returns(call):
 171             pass
 172         """
 173         control_deps_guard = templates.replace(template, call=node.value)[-1]
 174         control_deps_guard.body = []
 175 
 176       node = control_deps_guard
 177       anno.setanno(node, anno.Basic.INDENT_BLOCK_REMAINDER,
 178                    (node.body, alias_map))
 179     return node
 180 
 181 
 182 def transform(node, ctx):
 183   return SideEffectGuardTransformer(ctx).visit(node)
