  34   def new_symbol(self, name_root, reserved_locals):
  35     """Generate a new unique symbol.
  36 
  37     Args:
  38       name_root: String, used as stem in the new name.
  39       reserved_locals: Set(string), additional local symbols that are reserved
  40           and which should not be used.
  41     Returns:
  42       String.
  43     """
  44     raise NotImplementedError()

  48   """Transforms control flow structures like loops an conditionals."""

  50   def _create_cond_branch(self, body_name, aliased_orig_names,
  51                           aliased_new_names, body, returns):
  52     if aliased_orig_names:
  53       template = """
  54         def body_name():
  55           aliased_new_names, = aliased_orig_names,
  56           body
  57           return (returns,)
  58       """
  59       return templates.replace(
  60           template,
  61           body_name=body_name,
  62           body=body,
  63           aliased_orig_names=aliased_orig_names,
  64           aliased_new_names=aliased_new_names,
  65           returns=returns)
  66     else:
  67       template = """
  68         def body_name():
  69           body
  70           return (returns,)
  71       """
  72       return templates.replace(
  73           template, body_name=body_name, body=body, returns=returns)

  75   def _create_cond_expr(self, results, test, body_name, orelse_name):
  76     if results is not None:
  77       template = """
  78         results = ag__.utils.run_cond(test, body_name, orelse_name)
  79       """
  80       return templates.replace(
  81           template,
  82           test=test,
  83           results=results,
  84           body_name=body_name,
  85           orelse_name=orelse_name)
  86     else:
  87       template = """
  88         ag__.utils.run_cond(test, body_name, orelse_name)
  89       """
  90       return templates.replace(
  91           template, test=test, body_name=body_name, orelse_name=orelse_name)

  93   def _fmt_symbol_list(self, symbol_set):
  94     if not symbol_set:
  95       return 'no variables'
  96     return ', '.join(map(str, symbol_set))

  98   def _validate_no_live_vars_created(self, node):
  99     body_scope = anno.getanno(node, annos.NodeAnno.BODY_SCOPE)
 100     live_vars_out = anno.getanno(node, anno.Static.LIVE_VARS_OUT)
 101     live_vars_created_in_body = live_vars_out & body_scope.created
 102     if live_vars_created_in_body:
 103       raise ValueError(
 104           'The following variables are created inside the loop and used later:'
 105           '\n%s\n'
 106           'Variables must be declared outside loops because loops may not'
 107           ' necessarily execute.' % self._fmt_symbol_list(
 108               live_vars_created_in_body))

 110   def visit_If(self, node):
 111     node = self.generic_visit(node)
 112 
 113     body_scope = anno.getanno(node, annos.NodeAnno.BODY_SCOPE)
 114     orelse_scope = anno.getanno(node, annos.NodeAnno.ORELSE_SCOPE)
 115     defined_in = anno.getanno(node, anno.Static.DEFINED_VARS_IN)
 116     live_out = anno.getanno(node, anno.Static.LIVE_VARS_OUT)
 117 
 118     modified_in_cond = body_scope.modified | orelse_scope.modified
 119     returned_from_cond = set()
 120     for s in modified_in_cond:
 121       if s in live_out:
 122         returned_from_cond.add(s)
 123       elif s.is_composite():
 124         # Special treatment for compound objects: if any of their owner entities
 125         # are live, then they are outputs as well.
 126         if any(owner in live_out for owner in s.owner_set):
 127           returned_from_cond.add(s)
 128 
 129     need_alias_in_body = body_scope.modified & defined_in
 130     need_alias_in_orelse = orelse_scope.modified & defined_in
 131 
 132     created_in_body = body_scope.modified & returned_from_cond - defined_in
 133     created_in_orelse = orelse_scope.modified & returned_from_cond - defined_in
 134 
 135     if created_in_body != created_in_orelse:
 136       raise ValueError(
 137           'if statement may not initialize all variables: the true branch'
 138           ' creates %s, while the false branch creates %s. Make sure all'
 139           ' these variables are initialized either in both'
 140           ' branches or before the if statement.' %
 141           (self._fmt_symbol_list(created_in_body),
 142            self._fmt_symbol_list(created_in_orelse)))
 143 
 144     # Alias the closure variables inside the conditional functions, to allow
 145     # the functions access to the respective variables.
 146     # We will alias variables independently for body and orelse scope,
 147     # because different branches might write different variables.
 148     aliased_body_orig_names = tuple(need_alias_in_body)
 149     aliased_orelse_orig_names = tuple(need_alias_in_orelse)
 150     aliased_body_new_names = tuple(
 151         self.ctx.namer.new_symbol(s.ssf(), body_scope.referenced)
 152         for s in aliased_body_orig_names)
 153     aliased_orelse_new_names = tuple(
 154         self.ctx.namer.new_symbol(s.ssf(), orelse_scope.referenced)
 155         for s in aliased_orelse_orig_names)
 156 
 157     alias_body_map = dict(zip(aliased_body_orig_names, aliased_body_new_names))
 158     alias_orelse_map = dict(
 159         zip(aliased_orelse_orig_names, aliased_orelse_new_names))
 160 
 161     node_body = ast_util.rename_symbols(node.body, alias_body_map)
 162     node_orelse = ast_util.rename_symbols(node.orelse, alias_orelse_map)
 163 
 164     returned_from_cond = tuple(returned_from_cond)
 165     if returned_from_cond:
 166       if len(returned_from_cond) == 1:
 167         # TODO(mdan): Move this quirk into the operator implementation.
 168         cond_results = returned_from_cond[0]
 169       else:
 170         cond_results = gast.Tuple([s.ast() for s in returned_from_cond], None)
 171 
 172       returned_from_body = tuple(
 173           alias_body_map[s] if s in need_alias_in_body else s
 174           for s in returned_from_cond)
 175       returned_from_orelse = tuple(
 176           alias_orelse_map[s] if s in need_alias_in_orelse else s
 177           for s in returned_from_cond)
 178 
 179     else:
 180       # When the cond would return no value, we leave the cond called without
 181       # results. That in turn should trigger the side effect guards. The
 182       # branch functions will return a dummy value that ensures cond
 183       # actually has some return value as well.
 184       cond_results = None
 185       # TODO(mdan): This doesn't belong here; it's specific to the operator.
 186       returned_from_body = templates.replace_as_expression('tf.constant(1)')
 187       returned_from_orelse = templates.replace_as_expression('tf.constant(1)')
 188 
 189     body_name = self.ctx.namer.new_symbol('if_true', body_scope.referenced)
 190     orelse_name = self.ctx.namer.new_symbol('if_false', orelse_scope.referenced)
 191 
 192     body_def = self._create_cond_branch(
 193         body_name,
 194         aliased_orig_names=aliased_body_orig_names,
 195         aliased_new_names=aliased_body_new_names,
 196         body=node_body,
 197         returns=returned_from_body)
 198     orelse_def = self._create_cond_branch(
 199         orelse_name,
 200         aliased_orig_names=aliased_orelse_orig_names,
 201         aliased_new_names=aliased_orelse_new_names,
 202         body=node_orelse,
 203         returns=returned_from_orelse)
 204     cond_expr = self._create_cond_expr(cond_results, node.test, body_name,
 205                                        orelse_name)
 206 
 207     return body_def + orelse_def + cond_expr

 209   def visit_While(self, node):
 210     self.generic_visit(node)
 211 
 212     self._validate_no_live_vars_created(node)
 213 
 214     body_scope = anno.getanno(node, annos.NodeAnno.BODY_SCOPE)
 215     body_closure = body_scope.modified - body_scope.created
 216     all_referenced = body_scope.referenced
 217 
 218     cond_scope = anno.getanno(node, annos.NodeAnno.COND_SCOPE)
 219     cond_closure = set()
 220     for s in cond_scope.used:
 221       for root in s.support_set:
 222         if root not in body_scope.created:
 223           cond_closure.add(root)
 224 
 225     state = list(body_closure)
 226     if not state:
 227       # TODO(mdan): Implement this properly.
 228       # To complete this statement, we need to check whether any variable
 229       # created inside the body scope is used before being modified outside the
 230       # scope. This should be done during activity analysis, and in general
 231       # should cover the case where variables may not be initialized.
 232       raise ValueError('cannot convert while loop: no outputs')
 233 
 234     state_ssf = [
 235         self.ctx.namer.new_symbol(s.ssf(), all_referenced) for s in state
 236     ]
 237     ssf_map = {
 238         name: ssf
 239         for name, ssf in zip(state, state_ssf)
 240         if str(name) != ssf
 241     }
 242 
 243     if len(state) == 1:
 244       state = state[0]
 245       state_ssf = state_ssf[0]
 246       state_ast_tuple = state
 247     else:
 248       state_ast_tuple = gast.Tuple([n.ast() for n in state], None)
 249 
 250     node_body = ast_util.rename_symbols(node.body, ssf_map)
 251     test = ast_util.rename_symbols(node.test, ssf_map)
 252 
 253     # TODO(b/113118541) investigate the need-for and correctness-of extra_deps
 254     template = """
 255       def test_name(state_ssf):
 256         return test
 257       def body_name(state_ssf):
 258         body
 259         return state_ssf,
 260       state_ast_tuple = ag__.while_stmt(
 261           test_name, body_name, (state,), (extra_deps,))
 262     """
 263     node = templates.replace(
 264         template,
 265         state=state,
 266         state_ssf=state_ssf,
 267         state_ast_tuple=state_ast_tuple,
 268         test_name=self.ctx.namer.new_symbol('loop_test', body_scope.referenced),
 269         test=test,
 270         body_name=self.ctx.namer.new_symbol('loop_body', body_scope.referenced),
 271         body=node_body,
 272         extra_deps=tuple(s.ast() for s in cond_closure),
 273     )
 274 
 275     return node
 276 
 277   def visit_For(self, node):
 278     self.generic_visit(node)
 279 
 280     self._validate_no_live_vars_created(node)
 281 
 282     body_scope = anno.getanno(node, annos.NodeAnno.BODY_SCOPE)
 283     body_closure = body_scope.modified - body_scope.created
 284     all_referenced = body_scope.referenced
 285 
 286     state = list(body_closure)
 287 
 288     state_ssf = [
 289         self.ctx.namer.new_symbol(s.ssf(), all_referenced) for s in state
 290     ]
 291     ssf_map = {
 292         name: ssf
 293         for name, ssf in zip(state, state_ssf)
 294         if str(name) != ssf
 295     }
 296 
 297     if len(state) == 1:
 298       state = state[0]
 299       state_ssf = state_ssf[0]
 300       state_ast_tuple = state
 301     else:
 302       state_ast_tuple = gast.Tuple([n.ast() for n in state], None)
 303 
 304     node_body = ast_util.rename_symbols(node.body, ssf_map)
 305     if anno.hasanno(node, 'extra_test'):
 306       extra_test = anno.getanno(node, 'extra_test')
 307       extra_test = ast_util.rename_symbols(extra_test, ssf_map)
 308     else:
 309       extra_test = parser.parse_expression('True')
 310 
 311     template = """
 312       def extra_test_name(state_ssf):
 313         return extra_test_expr
 314       def body_name(loop_vars, state_ssf):
 315         # Workaround for PEP-3113
 316         iterate = loop_vars
 317         body
 318         return state_ssf,
 319       state_ast_tuple = ag__.for_stmt(
 320           iter_, extra_test_name, body_name, (state,))
 321     """
 322     node = templates.replace(
 323         template,
 324         state=state,
 325         state_ssf=state_ssf,
 326         state_ast_tuple=state_ast_tuple,
 327         iter_=node.iter,
 328         iterate=node.target,
 329         extra_test_name=self.ctx.namer.new_symbol('extra_test', all_referenced),
 330         extra_test_expr=extra_test,
 331         body_name=self.ctx.namer.new_symbol('loop_body', all_referenced),
 332         body=node_body)
 333 
 334     return node

 337 def transform(node, ctx):
 338   node = ControlFlowTransformer(ctx).visit(node)
 339   return node
