  15 """Converter for logical expressions.
  16 
  17 e.g. `a and b -> tf.logical_and(a, b)`. This is not done automatically in TF.
  18 """
  19 
  20 from __future__ import absolute_import
  21 from __future__ import division
  22 from __future__ import print_function
  23 
  24 import gast
  25 
  26 from tensorflow.python.autograph.core import converter
  27 from tensorflow.python.autograph.pyct import anno
  28 from tensorflow.python.autograph.pyct import parser
  29 from tensorflow.python.autograph.pyct import templates
  30 
  31 
  32 # TODO(mdan): Properly extrack boolean ops according to lazy eval rules.
  33 # Note that this isn't completely safe either, because tensors may have control
  34 # dependencies.
  35 # Note that for loops that should be done after the loop was converted to
  36 # tf.while_loop so that the expanded conditionals are properly scoped.
  37 
  38 # Used to signal that an operand is safe for non-lazy evaluation.
  39 SAFE_BOOLEAN_OPERAND = 'SAFE_BOOLEAN_OPERAND'
  40 
  41 
  42 class LogicalExpressionTransformer(converter.Base):
  43   """Converts logical expressions to corresponding TF calls."""
  44 
  45   def __init__(self, ctx):
  46     super(LogicalExpressionTransformer, self).__init__(ctx)
  47     # TODO(mdan): Look into replacing with bitwise operators instead.
  48     # TODO(mdan): Skip replacing if the function is trivial.
  49     self.op_mapping = {
  50         gast.And: 'tf.logical_and',
  51         gast.Eq: 'tf.equal',
  52         gast.Gt: 'tf.greater',
  53         gast.GtE: 'tf.greater_equal',
  54         gast.Lt: 'tf.less',
  55         gast.LtE: 'tf.less_equal',
  56         gast.Not: 'tf.logical_not',
  57         gast.NotEq: 'tf.not_equal',
  58         gast.Or: 'tf.logical_or',
  59         gast.USub: 'tf.negative',
  60     }
  61 
  62   def _expect_simple_symbol(self, operand):
  63     if isinstance(operand, gast.Name):
  64       return
  65     if anno.hasanno(operand, SAFE_BOOLEAN_OPERAND):
  66       return
  67     raise NotImplementedError(
  68         'only simple local variables are supported in logical and compound '
  69         'comparison expressions; for example, we support "a or b" but not '
  70         '"a.x or b"; for a workaround, assign the expression to a local '
  71         'variable and use that instead, for example "tmp = a.x", "tmp or b"')
  72 
  73   def _has_matching_func(self, operator):
  74     op_type = type(operator)
  75     return op_type in self.op_mapping
  76 
  77   def _matching_func(self, operator):
  78     op_type = type(operator)
  79     return self.op_mapping[op_type]
  80 
  81   def _as_function(self, func_name, args):
  82     template = """
  83       func_name(args)
  84     """
  85     replacement = templates.replace_as_expression(
  86         template, func_name=parser.parse_expression(func_name), args=args)
  87     anno.setanno(replacement, SAFE_BOOLEAN_OPERAND, True)
  88     return replacement
  89 
  90   def visit_Compare(self, node):
  91     node = self.generic_visit(node)
  92 
  93     if not all(self._has_matching_func(op) for op in node.ops):
  94       if len(node.ops) == 1:
  95         # Basic expressions are safe to leave as they are.
  96         return node
  97       else:
  98         raise NotImplementedError(
  99             'compound expression with at least one unsupported '
 100             'operator: {}'.format(node.ops))
 101 
 102     ops_and_comps = list(zip(node.ops, node.comparators))
 103     left = node.left
 104     op_tree = None
 105 
 106     # Repeated comparisons are converted to conjunctions:
 107     #   a < b < c   ->   a < b and b < c
 108     while ops_and_comps:
 109       op, right = ops_and_comps.pop(0)
 110       binary_comparison = self._as_function(
 111           self._matching_func(op), (left, right))
 112       if isinstance(left, gast.Name) and isinstance(right, gast.Name):
 113         anno.setanno(binary_comparison, SAFE_BOOLEAN_OPERAND, True)
 114       if op_tree:
 115         self._expect_simple_symbol(right)
 116         op_tree = self._as_function('tf.logical_and',
 117                                     (binary_comparison, op_tree))
 118       else:
 119         op_tree = binary_comparison
 120       left = right
 121     assert op_tree is not None
 122     return op_tree
 123 
 124   def visit_UnaryOp(self, node):
 125     node = self.generic_visit(node)
 126     return self._as_function(self._matching_func(node.op), node.operand)
 127 
 128   def visit_BoolOp(self, node):
 129     node = self.generic_visit(node)
 130     node_values = node.values
 131     right = node.values.pop()
 132     self._expect_simple_symbol(right)
 133     while node_values:
 134       left = node_values.pop()
 135       self._expect_simple_symbol(left)
 136       right = self._as_function(self._matching_func(node.op), (left, right))
 137     return right
 138 
 139 
 140 def transform(node, ctx):
 141   return LogicalExpressionTransformer(ctx).visit(node)
