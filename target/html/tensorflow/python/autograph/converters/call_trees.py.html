  51 class FunctionNamer(object):
  52   """Describes the interface for CallTreeTransformer's namer."""
  53 
  54   def compiled_function_name(self,
  55                              original_fqn,
  56                              live_entity=None,
  57                              owner_type=None):
  58     """Generate the name corresponding to the compiled version of a function.
  59 
  60     Args:
  61       original_fqn: string or tuple(string)
  62       live_entity: Callable, the actual target function, if known.
  63       owner_type: Optional object. If present, it indicates that the function is
  64           a member of the given type.
  65     Returns:
  66       string, bool
  67     """
  68     raise NotImplementedError()
  69 
  70   def compiled_class_name(self, original_fqn, live_entity=None):
  71     """Generate the name corresponding to the compiled version of a class.
  72 
  73     Args:
  74       original_fqn: string or tuple(string)
  75       live_entity: The actual target class, if known.
  76     Returns:
  77       string
  78     """
  79     raise NotImplementedError()

  85 class CallTreeTransformer(converter.Base):
  86   """Transforms the call tree by renaming transformed symbols."""
  87 
  88   def _resolve_name(self, node):
  89     """Used to resolve decorator info."""
  90     if isinstance(node, gast.Call):
  91       return self._resolve_name(node.func)
  92     if isinstance(node, gast.Name):
  93       return self.ctx.namespace.get(node.id)
  94     if isinstance(node, gast.Attribute):
  95       parent = self._resolve_name(node.value)
  96       if parent is not None:
  97         return getattr(parent, node.attr)
  98       return None
  99     raise ValueError(node)
 100 
 101   def _try_resolve_target(self, node):
 102     """Works for methods of objects of known type."""
 103     if anno.hasanno(node, 'live_val'):
 104       return anno.getanno(node, 'live_val')
 105     if isinstance(node, gast.Attribute) and anno.hasanno(node, 'type'):
 106       owner_type = anno.getanno(node, 'type')
 107       if hasattr(owner_type, node.attr):
 108         return getattr(owner_type, node.attr)
 109       else:
 110         raise ValueError('Type "%s" has not attribute "%s". Is it dynamic?' %
 111                          (owner_type, node.attr))
 112     return None
 113 
 114   def _function_is_compilable(self, target_entity):
 115     """Determines whether an entity can be compiled at all."""
 116     # TODO(mdan): This is just a placeholder. Implement.
 117     return not inspect_utils.isbuiltin(target_entity)
 118 
 119   def _should_compile(self, node, fqn):
 120     """Determines whether an entity should be compiled in the context."""
 121     # TODO(mdan): Needs cleanup. We should remove the use of fqn altogether.
 122     module_name = fqn[0]
 123     for mod in self.ctx.program.uncompiled_modules:
 124       if module_name.startswith(mod[0] + '.'):
 125         return False
 126 
 127     for i in range(1, len(fqn)):
 128       if fqn[:i] in self.ctx.program.uncompiled_modules:
 129         return False
 130 
 131     # Check for local decorations
 132     if anno.hasanno(node, 'graph_ready'):
 133       return False
 134 
 135     # The decorators themselves are not to be converted.
 136     # If present, the decorators should appear as static functions.
 137     target_entity = self._try_resolve_target(node.func)
 138     if target_entity is not None:
 139       # This attribute is set by the decorator itself.
 140       # TODO(mdan): This may not play nicely with other wrapping decorators.
 141       if hasattr(target_entity, '__pyct_is_compile_decorator'):
 142         return False
 143 
 144       if target_entity in self.ctx.program.autograph_decorators:
 145         return False
 146 
 147       # Inspect the target function decorators. If any include a @convert
 148       # or @graph_ready annotation, then they must be called as they are.
 149       # TODO(mdan): This may be quite heavy.
 150       # To parse and re-analyze each function for every call site could be quite
 151       # wasteful. Maybe we could cache the parsed AST?
 152       try:
 153         target_node, _ = parser.parse_entity(target_entity)
 154         target_node = target_node.body[0]
 155       except TypeError:
 156         # Functions whose source we cannot access are compilable (e.g. wrapped
 157         # to py_func).
 158         return True
 159 
 160       for dec in target_node.decorator_list:
 161         decorator_fn = self._resolve_name(dec)
 162         if (decorator_fn is not None and
 163             decorator_fn in self.ctx.program.autograph_decorators):
 164           return False
 165 
 166     return True
 167 
 168   def _rename_compilable_function(self, node):
 169     assert anno.hasanno(node.func, 'live_val')
 170     assert anno.hasanno(node.func, 'fqn')
 171     target_entity = anno.getanno(node.func, 'live_val')
 172     target_fqn = anno.getanno(node.func, 'fqn')
 173 
 174     if not self._should_compile(node, target_fqn):
 175       return node
 176 
 177     if anno.hasanno(node, 'is_constructor'):
 178       new_name = self.ctx.namer.compiled_class_name(
 179           target_fqn, live_entity=target_entity)
 180       do_rename = True
 181     else:
 182       if anno.hasanno(node.func, 'parent_type'):
 183         owner_type = anno.getanno(node.func, 'parent_type')
 184       else:
 185         # Fallback - not reliable.
 186         owner_type = inspect_utils.getmethodclass(target_entity)
 187       new_name, do_rename = self.ctx.namer.compiled_function_name(
 188           target_fqn, live_entity=target_entity, owner_type=owner_type)
 189 
 190     if do_rename:
 191       if target_entity is not None:
 192         if tf_inspect.ismethod(target_entity):
 193           # The renaming process will transform it into a regular function.
 194           # TODO(mdan): Is this complete? How does it work with nested members?
 195           node.args = [node.func.value] + node.args
 196       node.func = templates.replace('func_name', func_name=new_name)[0]
 197     return node
 198 
 199   def _wrap_to_py_func_no_return(self, node):
 200     # TODO(mdan): Properly handle varargs, etc.
 201     template = """
 202       ag__.utils.wrap_py_func(func, None, (args,), kwargs, True)
 203     """
 204     return templates.replace(
 205         template,
 206         func=node.func,
 207         args=node.args,
 208         kwargs=ast_util.keywords_to_dict(node.keywords))
 209 
 210   def _wrap_to_py_func_single_return(self, node, dtype):
 211     # TODO(mdan): Properly handle varargs, etc.
 212     template = """
 213       ag__.utils.wrap_py_func(func, dtype, (args,), kwargs, False)
 214     """
 215     return templates.replace_as_expression(
 216         template,
 217         func=node.func,
 218         dtype=parser.parse_expression(dtype),
 219         args=node.args,
 220         kwargs=ast_util.keywords_to_dict(node.keywords))
 221 
 222   def _insert_dynamic_conversion(self, node):
 223     """Inlines a dynamic conversion for a dynamic function."""
 224     # TODO(mdan): Pass information on the statically compiled functions.
 225     # Having access to the statically compiled functions can help avoid
 226     # unnecessary compilation.
 227     # For example, this would lead to function `a` being compiled twice:
 228     #
 229     #   def a():
 230     #     v = b
 231     #     b()
 232     #   def b():
 233     #     a()
 234     #
 235     # This is really a problem with recursive calls, which currently can
 236     # only be gated by a static condition, and should be rare.
 237     # TODO(mdan): It probably makes sense to use dynamic conversion every time.
 238     # Before we could convert all the time though, we'd need a reasonable
 239     # caching mechanism.
 240     template = """
 241       ag__.converted_call(
 242           func,
 243           ag__.ConversionOptions.new(recursive=recursive_val),
 244           args)
 245     """
 246     call_expr = templates.replace(
 247         template,
 248         func=node.func,
 249         recursive_val=parser.parse_expression(str(self.ctx.program.recursive)),
 250         args=node.args)
 251     new_call = call_expr[0].value
 252     # TODO(mdan): Improve the template mechanism to better support this.
 253     new_call.keywords = node.keywords
 254     return new_call
 255 
 256   def visit_Expr(self, node):
 257     if isinstance(node.value, gast.Call):
 258       if anno.hasanno(node.value.func, 'live_val'):
 259         target_entity = anno.getanno(node.value.func, 'live_val')
 260         if not self._function_is_compilable(target_entity):
 261           if anno.hasanno(node.value.func, 'fqn'):
 262             target_fqn = anno.getanno(node.value.func, 'fqn')
 263             if not self._should_compile(node.value, target_fqn):
 264               return node
 265             node = self._wrap_to_py_func_no_return(node.value)
 266             return node
 267       # Only the case of py_func with no return value is special.
 268       # Everything else is processed by visit_Call.
 269       self.visit(node.value)
 270     else:
 271       self.generic_visit(node)
 272     return node
 273 
 274   def visit_Call(self, node):
 275     # If the function call is wrapped by one of the marker decorators,
 276     # consider it graph ready.
 277     if anno.hasanno(node.func, 'live_val'):
 278       target_entity = anno.getanno(node.func, 'live_val')
 279       if target_entity in self.ctx.program.autograph_decorators:
 280         if len(node.args) < 1:
 281           raise ValueError(
 282               'Found call to decorator function "%s", but it had no arguments. '
 283               'A decorator needs at least one positional argument.' %
 284               target_entity)
 285         anno.setanno(node.args[0], 'graph_ready', True)
 286 
 287     self.generic_visit(node)
 288     if anno.hasanno(node.func, 'live_val'):
 289       target_entity = anno.getanno(node.func, 'live_val')
 290       if anno.hasanno(node.func, 'fqn'):
 291         target_fqn = anno.getanno(node.func, 'fqn')
 292       else:
 293         target_fqn = None
 294       if self._function_is_compilable(target_entity):
 295         node = self._rename_compilable_function(node)
 296       elif target_fqn and target_fqn in KNOWN_NUMPY_FUNCTIONS:
 297         # TODO(mdan): Should we replace these with equivalent TF ops instead?
 298         node = self._wrap_to_py_func_single_return(
 299             node, KNOWN_NUMPY_FUNCTIONS[target_fqn].dtype)
 300       else:
 301         raise NotImplementedError(
 302             'py_func with return values (unknown function)')
 303     else:
 304       if anno.hasanno(node.func, anno.Basic.QN):
 305         # Special-case a few builtins that otherwise go undetected. This
 306         # normally doesn't pose a problem, but the dict built-in doesn't
 307         # work with inspect.getargspec which is required for dynamic functions.
 308         # Note: expecting this is resilient to aliasing (e.g.
 309         # dict = an_evil_dict), because in those cases the regular mechanisms
 310         # process a simple user function.
 311         qn = anno.getanno(node.func, anno.Basic.QN)
 312         # Add items to this list as needed.
 313         if str(qn) in ('dict',):
 314           return node
 315 
 316       if ast_util.matches(node, 'super(_)'):
 317         # super() calls are preserved. The class conversion mechanism will
 318         # ensure that they return the correct value.
 319         return node
 320 
 321       if self.ctx.program.recursive:
 322         node = self._insert_dynamic_conversion(node)
 323     return node
 324

 326 def transform(node, ctx):
 327   """Transform function call to the compiled counterparts.
 328 
 329   Args:
 330     node: AST
 331     ctx: EntityContext
 332   Returns:
 333     A tuple (node, new_names):
 334         node: The transformed AST
 335         new_names: set(string), containing any newly-generated names
 336   """
 337   return CallTreeTransformer(ctx).visit(node)
