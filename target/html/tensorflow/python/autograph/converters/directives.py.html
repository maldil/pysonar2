  15 """Handles directives.
  16 
  17 This converter removes the directive functions from the code and moves the
  18 information they specify into AST annotations. It is a specialized form of
  19 static analysis, one that is specific to AutoGraph.
  20 """

  33 ENCLOSING_LOOP = 'enclosing_loop'
  34 
  35 
  36 def _map_args(call_node, function):
  37   """Maps AST call nodes to the actual function's arguments.
  38 
  39   Args:
  40     call_node: ast.Call
  41     function: Callable[..., Any], the actual function matching call_node
  42   Returns:
  43     Dict[Text, ast.AST], mapping each of the function's argument names to
  44     the respective AST node.
  45   Raises:
  46       ValueError: if the default arguments are not correctly set
  47   """
  48   args = call_node.args
  49   kwds = {kwd.arg: kwd.value for kwd in call_node.keywords}
  50   call_args = tf_inspect.getcallargs(function, *args, **kwds)
  51 
  52   # Keyword arguments not specified in kwds will be mapped to their defaults,
  53   # which are Python values. Since we don't currently have a way to transform
  54   # those into AST references, we simply remove them. By convention, directives
  55   # use UNSPECIFIED as default value for for optional arguments. No other
  56   # defaults should be present.
  57   unexpected_defaults = []
  58   for k in call_args:
  59     if (k not in kwds
  60         and call_args[k] not in args
  61         and call_args[k] is not directives.UNSPECIFIED):
  62       unexpected_defaults.append(k)
  63   if unexpected_defaults:
  64     raise ValueError('Unexpected keyword argument values, %s, for function %s'
  65                      % (zip(unexpected_defaults,
  66                             [call_args[k] for k in unexpected_defaults]),
  67                         function))
  68   return {k: v for k, v in call_args.items() if v is not directives.UNSPECIFIED}

  71 class DirectivesTransformer(converter.Base):
  72   """Parses compiler directives and converts them into AST annotations."""
  73 
  74   def _process_symbol_directive(self, call_node, directive):
  75     if len(call_node.args) < 1:
  76       raise ValueError('"%s" requires a positional first argument'
  77                        ' as the target' % directive.__name__)
  78     target = call_node.args[0]
  79     defs = anno.getanno(target, anno.Static.ORIG_DEFINITIONS)
  80     for def_ in defs:
  81       def_.directives[directive] = _map_args(call_node, directive)
  82     return call_node
  83 
  84   def _process_statement_directive(self, call_node, directive):
  85     if self.local_scope_level < 1:
  86       raise ValueError(
  87           '"%s" must be used inside a statement' % directive.__name__)
  88     target = self.get_local(ENCLOSING_LOOP)
  89     node_anno = anno.getanno(target, converter.AgAnno.DIRECTIVES, {})
  90     node_anno[directive] = _map_args(call_node, directive)
  91     anno.setanno(target, converter.AgAnno.DIRECTIVES, node_anno)
  92     return call_node
  93 
  94   def visit_Expr(self, node):
  95     if isinstance(node.value, gast.Call):
  96       call_node = node.value
  97       if anno.hasanno(call_node.func, 'live_val'):
  98         live_val = anno.getanno(call_node.func, 'live_val')
  99 
 100         if live_val is directives.set_element_type:
 101           call_node = self._process_symbol_directive(call_node, live_val)
 102         elif live_val is directives.set_loop_options:
 103           call_node = self._process_statement_directive(call_node, live_val)
 104         else:
 105           return self.generic_visit(node)
 106 
 107         return None  # Directive calls are not output in the generated code.
 108     return self.generic_visit(node)
 109 
 110   # TODO(mdan): This will be insufficient for other control flow.
 111   # That means that if we ever have a directive that affects things other than
 112   # loops, we'll need support for parallel scopes, or have multiple converters.
 113   def _track_and_visit_loop(self, node):
 114     self.enter_local_scope()
 115     self.set_local(ENCLOSING_LOOP, node)
 116     node = self.generic_visit(node)
 117     self.exit_local_scope()
 118     return node
 119 
 120   def visit_While(self, node):
 121     return self._track_and_visit_loop(node)
 122 
 123   def visit_For(self, node):
 124     return self._track_and_visit_loop(node)
 125 
 126 
 127 def transform(node, ctx):
 128   return DirectivesTransformer(ctx).visit(node)
