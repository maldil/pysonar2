  34 
  35   def __init__(self):
  36     self.scope = None

  39 class ConditionalExpressionTransformer(converter.Base):
  40   """Converts contitional expressions to functional form."""
  41 
  42   def _postprocess_statement(self, node):
  43     """Inserts any separate functions that node may use."""
  44     replacements = []
  45     for def_node in self.state[_FunctionDefs].nodes:
  46       replacements.extend(def_node)
  47     replacements.append(node)
  48     node = replacements
  49     # The corresponding enter is called by self.visit_block (see _process_block)
  50     self.state[_FunctionDefs].exit()
  51     return node, None
  52 
  53   def _create_branch(self, expr, name_stem):
  54     scope = self.state[_Statement].scope
  55     name = self.ctx.namer.new_symbol(name_stem, scope.referenced)
  56     template = """
  57       def name():
  58         return expr,
  59     """
  60     node = templates.replace(template, name=name, expr=expr)
  61     self.state[_FunctionDefs].nodes.append(node)
  62     return name
  63 
  64   def visit_IfExp(self, node):
  65     if anno.hasanno(node.test, anno.Basic.QN):
  66       name_root = anno.getanno(node.test, anno.Basic.QN).ssf()
  67     else:
  68       name_root = 'ifexp'
  69 
  70     true_fn_name = self._create_branch(node.body, '%s_true' % name_root)
  71     false_fn_name = self._create_branch(node.orelse, '%s_false' % name_root)
  72 
  73     return templates.replace_as_expression(
  74         'ag__.utils.run_cond(test, true_fn_name, false_fn_name)',
  75         test=node.test,
  76         true_fn_name=true_fn_name,
  77         false_fn_name=false_fn_name)
  78 
  79   def _process_block(self, scope, block):
  80     self.state[_Statement].enter()
  81     self.state[_Statement].scope = scope
  82     block = self.visit_block(
  83         block,
  84         before_visit=self.state[_FunctionDefs].enter,
  85         after_visit=self._postprocess_statement)
  86     self.state[_Statement].exit()
  87     return block
  88 
  89   def visit_FunctionDef(self, node):
  90     node.args = self.generic_visit(node.args)
  91     node.decorator_list = self.visit_block(node.decorator_list)
  92     node.body = self._process_block(
  93         anno.getanno(node, anno.Static.SCOPE), node.body)
  94     return node
  95 
  96   def visit_For(self, node):
  97     node.target = self.visit(node.target)
  98     node.body = self._process_block(
  99         anno.getanno(node, NodeAnno.BODY_SCOPE), node.body)
 100     node.orelse = self._process_block(
 101         anno.getanno(node, NodeAnno.ORELSE_SCOPE), node.orelse)
 102     return node
 103 
 104   def visit_While(self, node):
 105     node.test = self.visit(node.test)
 106     node.body = self._process_block(
 107         anno.getanno(node, NodeAnno.BODY_SCOPE), node.body)
 108     node.orelse = self._process_block(
 109         anno.getanno(node, NodeAnno.ORELSE_SCOPE), node.orelse)
 110     return node
 111 
 112   def visit_If(self, node):
 113     node.test = self.visit(node.test)
 114     node.body = self._process_block(
 115         anno.getanno(node, NodeAnno.BODY_SCOPE), node.body)
 116     node.orelse = self._process_block(
 117         anno.getanno(node, NodeAnno.ORELSE_SCOPE), node.orelse)
 118     return node
 119 
 120   def visit_With(self, node):
 121     node.items = self.visit_block(node.items)
 122     node.body = self._process_block(
 123         anno.getanno(node, NodeAnno.BODY_SCOPE), node.body)
 124     return node
 125 
 126 
 127 def transform(node, ctx):
 128   node = ConditionalExpressionTransformer(ctx).visit(node)
 129   return node
