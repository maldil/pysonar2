  15 """Handles decorators.
  16 
  17 Note: this module only deals with functions whose decorators are still recorded
  18 in the AST. This does not always happen. See the unit test for an example.
  19 """
  20 
  21 from __future__ import absolute_import
  22 from __future__ import division
  23 from __future__ import print_function
  24 
  25 import gast
  26 
  27 from tensorflow.python.autograph.core import converter
  28 from tensorflow.python.autograph.pyct import anno
  29 from tensorflow.python.util import tf_inspect
  30 
  31 
  32 class DecoratorsTransformer(converter.Base):
  33   """Converts or removes decorators."""
  34 
  35   def visit_FunctionDef(self, node):
  36     self.generic_visit(node)
  37     kept_decorators = []
  38     for dec in node.decorator_list:
  39       if isinstance(dec, gast.Call):
  40         dec_func = dec.func
  41       else:
  42         dec_func = dec
  43 
  44       # Special cases.
  45       # TODO(mdan): Is there any way we can treat these more generically?
  46       # We may want to forego using decorators altogether if we can't
  47       # properly support them.
  48       if isinstance(dec_func, gast.Name) and dec_func.id in ('classmethod',):
  49         # Assumption: decorators are only visible in the AST when converting
  50         # a function inline (via another decorator).
  51         # In that case, the converted function is no longer part of the
  52         # original object that it was declared into.
  53         # This is currently verified by tests.
  54         continue
  55 
  56       original_dec = anno.getanno(dec_func, anno.Basic.QN)
  57       dec_value = anno.getanno(dec_func, 'live_val')
  58 
  59       if dec_value in self.ctx.program.autograph_decorators:
  60         # AutoGraph decorators do not need to be preserved.
  61         continue
  62 
  63       # When using foo.bar.baz, we only really need to grab foo and import
  64       # that.
  65       dec_support_node = dec_func
  66       while isinstance(dec_support_node, gast.Attribute):
  67         dec_support_node = dec_support_node.value
  68 
  69       if not anno.hasanno(dec_support_node, 'live_val'):
  70         raise ValueError(
  71             'could not resolve symbol "%s" when looking up decorator "%s"' %
  72             (anno.getanno(dec_support_node, anno.Basic.QN), original_dec))
  73 
  74       dec_support = anno.getanno(dec_support_node, 'live_val')
  75       # The tuple contains:
  76       #  * the AST that represents the decorator
  77       #  * the entity supporting the decorator (i.e., what we need to import)
  78       #  * the name of the module that needs to be imported for this decorator
  79       #    to properly resolve.
  80       # Examples:
  81       #  for foo.bar, the tuple is (<ast>, <module foo>, 'foo')
  82       #  for baz, the tuple is (<ast>, <module baz.__module__>, 'baz')
  83       kept_decorators.append((dec, dec_support,
  84                               anno.getanno(dec_support_node, anno.Basic.QN)))
  85 
  86     for _, dec_support, name in kept_decorators:
  87       if tf_inspect.ismodule(dec_support):
  88         self.ctx.program.additional_imports.add(
  89             'import %s as %s' % (dec_support.__name__, name))
  90       else:
  91         if dec_support.__module__ == '__main__':
  92           raise ValueError(
  93               'decorator "%s" was not allowed because it is declared '
  94               'in the module "%s". To fix this, declare it in a separate '
  95               'module that we can import it from.' % (dec_support,
  96                                                       dec_support.__module__))
  97         self.ctx.program.additional_imports.add(
  98             'from %s import %s' % (dec_support.__module__, name))
  99 
 100     node.decorator_list = [dec for dec, _, _ in kept_decorators]
 101     return node
 102 
 103 
 104 def transform(node, ctx):
 105   return DecoratorsTransformer(ctx).visit(node)
