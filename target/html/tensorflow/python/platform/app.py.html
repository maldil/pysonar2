  28 def _usage(shorthelp):
  29   """Writes __main__'s docstring to stdout with some help text.
  30 
  31   Args:
  32     shorthelp: bool, if True, prints only flags from the main module,
  33         rather than all flags.
  34   """
  35   doc = _sys.modules['__main__'].__doc__
  36   if not doc:
  37     doc = '\nUSAGE: %s [flags]\n' % _sys.argv[0]
  38     doc = flags.text_wrap(doc, indent='       ', firstline_indent='')
  39   else:
  40     # Replace all '%s' with sys.argv[0], and all '%%' with '%'.
  41     num_specifiers = doc.count('%') - 2 * doc.count('%%')
  42     try:
  43       doc %= (_sys.argv[0],) * num_specifiers
  44     except (OverflowError, TypeError, ValueError):
  45       # Just display the docstring as-is.
  46       pass
  47   if shorthelp:
  48     flag_str = flags.FLAGS.main_module_help()
  49   else:
  50     flag_str = str(flags.FLAGS)
  51   try:
  52     _sys.stdout.write(doc)
  53     if flag_str:
  54       _sys.stdout.write('\nflags:\n')
  55       _sys.stdout.write(flag_str)
  56     _sys.stdout.write('\n')
  57   except IOError as e:
  58     # We avoid printing a huge backtrace if we get EPIPE, because
  59     # "foo.par --help | less" is a frequent use case.
  60     if e.errno != _errno.EPIPE:
  61       raise

  64 class _HelpFlag(flags.BooleanFlag):
  65   """Special boolean flag that displays usage and raises SystemExit."""
  66   NAME = 'help'
  67   SHORT_NAME = 'h'
  68 
  69   def __init__(self):
  70     super(_HelpFlag, self).__init__(
  71         self.NAME, False, 'show this help', short_name=self.SHORT_NAME)
  72 
  73   def parse(self, arg):
  74     if arg:
  75       _usage(shorthelp=True)
  76       print()
  77       print('Try --helpfull to get a list of all flags.')
  78       _sys.exit(1)
  79 
  80 
  81 class _HelpshortFlag(_HelpFlag):
  82   """--helpshort is an alias for --help."""
  83   NAME = 'helpshort'
  84   SHORT_NAME = None
  85 
  86 
  87 class _HelpfullFlag(flags.BooleanFlag):
  88   """Display help for flags in main module and all dependent modules."""
  89 
  90   def __init__(self):
  91     super(_HelpfullFlag, self).__init__('helpfull', False, 'show full help')
  92 
  93   def parse(self, arg):
  94     if arg:
  95       _usage(shorthelp=False)
  96       _sys.exit(1)

  99 _define_help_flags_called = False

 102 def _define_help_flags():
 103   global _define_help_flags_called
 104   if not _define_help_flags_called:
 105     flags.DEFINE_flag(_HelpFlag())
 106     flags.DEFINE_flag(_HelpfullFlag())
 107     flags.DEFINE_flag(_HelpshortFlag())
 108     _define_help_flags_called = True

 111 @tf_export('app.run')
 112 def run(main=None, argv=None):
 113   """Runs the program with an optional 'main' function and 'argv' list."""
 114 
 115   # Define help flags.
 116   _define_help_flags()
 117 
 118   # Parse known flags.
 119   argv = flags.FLAGS(_sys.argv if argv is None else argv, known_only=True)
 120 
 121   main = main or _sys.modules['__main__'].main
 122 
 123   # Call the main function, passing through any arguments
 124   # to the final program.
 125   _sys.exit(main(argv))
