  53 def g_main(argv):
  54   """Delegate to unittest.main after redefining testLoader."""
  55   if 'TEST_SHARD_STATUS_FILE' in os.environ:
  56     try:
  57       f = None
  58       try:
  59         f = open(os.environ['TEST_SHARD_STATUS_FILE'], 'w')
  60         f.write('')
  61       except IOError:
  62         sys.stderr.write('Error opening TEST_SHARD_STATUS_FILE (%s). Exiting.'
  63                          % os.environ['TEST_SHARD_STATUS_FILE'])
  64         sys.exit(1)
  65     finally:
  66       if f is not None: f.close()
  67 
  68   if ('TEST_TOTAL_SHARDS' not in os.environ or
  69       'TEST_SHARD_INDEX' not in os.environ):
  70     return unittest_main(argv=argv)
  71 
  72   total_shards = int(os.environ['TEST_TOTAL_SHARDS'])
  73   shard_index = int(os.environ['TEST_SHARD_INDEX'])
  74   base_loader = TestLoader()
  75 
  76   delegate_get_names = base_loader.getTestCaseNames
  77   bucket_iterator = itertools.cycle(range(total_shards))
  78 
  79   def getShardedTestCaseNames(testCaseClass):
  80     filtered_names = []
  81     for testcase in sorted(delegate_get_names(testCaseClass)):
  82       bucket = next(bucket_iterator)
  83       if bucket == shard_index:
  84         filtered_names.append(testcase)
  85     return filtered_names
  86 
  87   # Override getTestCaseNames
  88   base_loader.getTestCaseNames = getShardedTestCaseNames
  89 
  90   unittest_main(argv=argv, testLoader=base_loader)

  93 # Redefine main to allow running benchmarks
  94 def main(argv=None):  # pylint: disable=function-redefined
  95   def main_wrapper():
  96     args = argv
  97     if args is None:
  98       args = sys.argv
  99     return app.run(main=g_main, argv=args)
 100   benchmark.benchmarks_main(true_main=main_wrapper)

 103 def GetTempDir():
 104   """Return a temporary directory for tests to use."""
 105   global _googletest_temp_dir
 106   if not _googletest_temp_dir:
 107     first_frame = tf_inspect.stack()[-1][0]
 108     temp_dir = os.path.join(tempfile.gettempdir(),
 109                             os.path.basename(tf_inspect.getfile(first_frame)))
 110     temp_dir = tempfile.mkdtemp(prefix=temp_dir.rstrip('.py'))
 111 
 112     def delete_temp_dir(dirname=temp_dir):
 113       try:
 114         file_io.delete_recursively(dirname)
 115       except errors.OpError as e:
 116         logging.error('Error removing %s: %s', dirname, e)
 117 
 118     atexit.register(delete_temp_dir)
 119     _googletest_temp_dir = temp_dir
 120 
 121   return _googletest_temp_dir

 142 @tf_export('test.StubOutForTesting')
 143 class StubOutForTesting(object):
 144   """Support class for stubbing methods out for unit testing.
 145 
 146   Sample Usage:
 147 
 148   You want os.path.exists() to always return true during testing.
 149 
 150      stubs = StubOutForTesting()
 151      stubs.Set(os.path, 'exists', lambda x: 1)
 152        ...
 153      stubs.CleanUp()
 154 
 155   The above changes os.path.exists into a lambda that returns 1.  Once
 156   the ... part of the code finishes, the CleanUp() looks up the old
 157   value of os.path.exists and restores it.
 158   """
 159 
 160   def __init__(self):
 161     self.cache = []
 162     self.stubs = []
 163 
 164   def __del__(self):
 165     """Do not rely on the destructor to undo your stubs.
 166 
 167     You cannot guarantee exactly when the destructor will get called without
 168     relying on implementation details of a Python VM that may change.
 169     """
 170     self.CleanUp()
 171 
 172   # __enter__ and __exit__ allow use as a context manager.
 173   def __enter__(self):
 174     return self
 175 
 176   def __exit__(self, unused_exc_type, unused_exc_value, unused_tb):
 177     self.CleanUp()
 178 
 179   def CleanUp(self):
 180     """Undoes all SmartSet() & Set() calls, restoring original definitions."""
 181     self.SmartUnsetAll()
 182     self.UnsetAll()
 183 
 184   def SmartSet(self, obj, attr_name, new_attr):
 185     """Replace obj.attr_name with new_attr.
 186 
 187     This method is smart and works at the module, class, and instance level
 188     while preserving proper inheritance. It will not stub out C types however
 189     unless that has been explicitly allowed by the type.
 190 
 191     This method supports the case where attr_name is a staticmethod or a
 192     classmethod of obj.
 193 
 194     Notes:
 195       - If obj is an instance, then it is its class that will actually be
 196         stubbed. Note that the method Set() does not do that: if obj is
 197         an instance, it (and not its class) will be stubbed.
 198       - The stubbing is using the builtin getattr and setattr. So, the __get__
 199         and __set__ will be called when stubbing (TODO: A better idea would
 200         probably be to manipulate obj.__dict__ instead of getattr() and
 201         setattr()).
 202 
 203     Args:
 204       obj: The object whose attributes we want to modify.
 205       attr_name: The name of the attribute to modify.
 206       new_attr: The new value for the attribute.
 207 
 208     Raises:
 209       AttributeError: If the attribute cannot be found.
 210     """
 211     _, obj = tf_decorator.unwrap(obj)
 212     if (tf_inspect.ismodule(obj) or
 213         (not tf_inspect.isclass(obj) and attr_name in obj.__dict__)):
 214       orig_obj = obj
 215       orig_attr = getattr(obj, attr_name)
 216     else:
 217       if not tf_inspect.isclass(obj):
 218         mro = list(tf_inspect.getmro(obj.__class__))
 219       else:
 220         mro = list(tf_inspect.getmro(obj))
 221 
 222       mro.reverse()
 223 
 224       orig_attr = None
 225       found_attr = False
 226 
 227       for cls in mro:
 228         try:
 229           orig_obj = cls
 230           orig_attr = getattr(obj, attr_name)
 231           found_attr = True
 232         except AttributeError:
 233           continue
 234 
 235       if not found_attr:
 236         raise AttributeError('Attribute not found.')
 237 
 238     # Calling getattr() on a staticmethod transforms it to a 'normal' function.
 239     # We need to ensure that we put it back as a staticmethod.
 240     old_attribute = obj.__dict__.get(attr_name)
 241     if old_attribute is not None and isinstance(old_attribute, staticmethod):
 242       orig_attr = staticmethod(orig_attr)
 243 
 244     self.stubs.append((orig_obj, attr_name, orig_attr))
 245     setattr(orig_obj, attr_name, new_attr)
 246 
 247   def SmartUnsetAll(self):
 248     """Reverses SmartSet() calls, restoring things to original definitions.
 249 
 250     This method is automatically called when the StubOutForTesting()
 251     object is deleted; there is no need to call it explicitly.
 252 
 253     It is okay to call SmartUnsetAll() repeatedly, as later calls have
 254     no effect if no SmartSet() calls have been made.
 255     """
 256     for args in reversed(self.stubs):
 257       setattr(*args)
 258 
 259     self.stubs = []
 260 
 261   def Set(self, parent, child_name, new_child):
 262     """In parent, replace child_name's old definition with new_child.
 263 
 264     The parent could be a module when the child is a function at
 265     module scope.  Or the parent could be a class when a class' method
 266     is being replaced.  The named child is set to new_child, while the
 267     prior definition is saved away for later, when UnsetAll() is
 268     called.
 269 
 270     This method supports the case where child_name is a staticmethod or a
 271     classmethod of parent.
 272 
 273     Args:
 274       parent: The context in which the attribute child_name is to be changed.
 275       child_name: The name of the attribute to change.
 276       new_child: The new value of the attribute.
 277     """
 278     old_child = getattr(parent, child_name)
 279 
 280     old_attribute = parent.__dict__.get(child_name)
 281     if old_attribute is not None and isinstance(old_attribute, staticmethod):
 282       old_child = staticmethod(old_child)
 283 
 284     self.cache.append((parent, old_child, child_name))
 285     setattr(parent, child_name, new_child)
 286 
 287   def UnsetAll(self):
 288     """Reverses Set() calls, restoring things to their original definitions.
 289 
 290     This method is automatically called when the StubOutForTesting()
 291     object is deleted; there is no need to call it explicitly.
 292 
 293     It is okay to call UnsetAll() repeatedly, as later calls have no
 294     effect if no Set() calls have been made.
 295     """
 296     # Undo calls to Set() in reverse order, in case Set() was called on the
 297     # same arguments repeatedly (want the original call to be last one undone)
 298     for (parent, old_child, child_name) in reversed(self.cache):
 299       setattr(parent, child_name, old_child)
 300     self.cache = []
