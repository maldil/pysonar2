  48 def _global_report_benchmark(
  49     name, iters=None, cpu_time=None, wall_time=None,
  50     throughput=None, extras=None):
  51   """Method for recording a benchmark directly.
  52 
  53   Args:
  54     name: The BenchmarkEntry name.
  55     iters: (optional) How many iterations were run
  56     cpu_time: (optional) Total cpu time in seconds
  57     wall_time: (optional) Total wall time in seconds
  58     throughput: (optional) Throughput (in MB/s)
  59     extras: (optional) Dict mapping string keys to additional benchmark info.
  60 
  61   Raises:
  62     TypeError: if extras is not a dict.
  63     IOError: if the benchmark output file already exists.
  64   """
  65   if extras is not None:
  66     if not isinstance(extras, dict):
  67       raise TypeError("extras must be a dict")
  68 
  69   logging.info("Benchmark [%s] iters: %d, wall_time: %g, cpu_time: %g,"
  70                "throughput: %g %s", name, iters if iters is not None else -1,
  71                wall_time if wall_time is not None else -1, cpu_time if
  72                cpu_time is not None else -1, throughput if
  73                throughput is not None else -1, str(extras) if extras else "")
  74 
  75   entries = test_log_pb2.BenchmarkEntries()
  76   entry = entries.entry.add()
  77   entry.name = name
  78   if iters is not None:
  79     entry.iters = iters
  80   if cpu_time is not None:
  81     entry.cpu_time = cpu_time
  82   if wall_time is not None:
  83     entry.wall_time = wall_time
  84   if throughput is not None:
  85     entry.throughput = throughput
  86   if extras is not None:
  87     for (k, v) in extras.items():
  88       if isinstance(v, numbers.Number):
  89         entry.extras[k].double_value = v
  90       else:
  91         entry.extras[k].string_value = str(v)
  92 
  93   test_env = os.environ.get(TEST_REPORTER_TEST_ENV, None)
  94   if test_env is None:
  95     # Reporting was not requested, just print the proto
  96     print(str(entries))
  97     return
  98 
  99   serialized_entry = entries.SerializeToString()
 100 
 101   mangled_name = name.replace("/", "__")
 102   output_path = "%s%s" % (test_env, mangled_name)
 103   if gfile.Exists(output_path):
 104     raise IOError("File already exists: %s" % output_path)
 105   with gfile.GFile(output_path, "wb") as out:
 106     out.write(serialized_entry)

 109 class _BenchmarkRegistrar(type):
 110   """The Benchmark class registrar.  Used by abstract Benchmark class."""
 111 
 112   def __new__(mcs, clsname, base, attrs):
 113     newclass = super(mcs, _BenchmarkRegistrar).__new__(
 114         mcs, clsname, base, attrs)
 115     if not newclass.is_abstract():
 116       GLOBAL_BENCHMARK_REGISTRY.add(newclass)
 117     return newclass

 120 class Benchmark(six.with_metaclass(_BenchmarkRegistrar, object)):
 121   """Abstract class that provides helper functions for running benchmarks.
 122 
 123   Any class subclassing this one is immediately registered in the global
 124   benchmark registry.
 125 
 126   Only methods whose names start with the word "benchmark" will be run during
 127   benchmarking.
 128   """
 129 
 130   @classmethod
 131   def is_abstract(cls):
 132     # mro: (_BenchmarkRegistrar, Benchmark) means this is Benchmark
 133     return len(cls.mro()) <= 2
 134 
 135   def _get_name(self, overwrite_name=None):
 136     """Returns full name of class and method calling report_benchmark."""
 137 
 138     # Find the caller method (outermost Benchmark class)
 139     stack = tf_inspect.stack()
 140     calling_class = None
 141     name = None
 142     for frame in stack[::-1]:
 143       f_locals = frame[0].f_locals
 144       f_self = f_locals.get("self", None)
 145       if isinstance(f_self, Benchmark):
 146         calling_class = f_self  # Get the outermost stack Benchmark call
 147         name = frame[3]  # Get the method name
 148         break
 149     if calling_class is None:
 150       raise ValueError("Unable to determine calling Benchmark class.")
 151 
 152     # Use the method name, or overwrite_name is provided.
 153     name = overwrite_name or name
 154     # Prefix the name with the class name.
 155     class_name = type(calling_class).__name__
 156     name = "%s.%s" % (class_name, name)
 157     return name
 158 
 159   def report_benchmark(
 160       self,
 161       iters=None,
 162       cpu_time=None,
 163       wall_time=None,
 164       throughput=None,
 165       extras=None,
 166       name=None):
 167     """Report a benchmark.
 168 
 169     Args:
 170       iters: (optional) How many iterations were run
 171       cpu_time: (optional) median or mean cpu time in seconds.
 172       wall_time: (optional) median or mean wall time in seconds.
 173       throughput: (optional) Throughput (in MB/s)
 174       extras: (optional) Dict mapping string keys to additional benchmark info.
 175         Values may be either floats or values that are convertible to strings.
 176       name: (optional) Override the BenchmarkEntry name with `name`.
 177         Otherwise it is inferred from the top-level method name.
 178     """
 179     name = self._get_name(overwrite_name=name)
 180     _global_report_benchmark(
 181         name=name, iters=iters, cpu_time=cpu_time, wall_time=wall_time,
 182         throughput=throughput, extras=extras)

 185 @tf_export("test.Benchmark")
 186 class TensorFlowBenchmark(Benchmark):
 187   """Abstract class that provides helpers for TensorFlow benchmarks."""
 188 
 189   @classmethod
 190   def is_abstract(cls):
 191     # mro: (_BenchmarkRegistrar, Benchmark, TensorFlowBenchmark) means
 192     # this is TensorFlowBenchmark.
 193     return len(cls.mro()) <= 3
 194 
 195   def run_op_benchmark(self,
 196                        sess,
 197                        op_or_tensor,
 198                        feed_dict=None,
 199                        burn_iters=2,
 200                        min_iters=10,
 201                        store_trace=False,
 202                        store_memory_usage=True,
 203                        name=None,
 204                        extras=None,
 205                        mbs=0):
 206     """Run an op or tensor in the given session.  Report the results.
 207 
 208     Args:
 209       sess: `Session` object to use for timing.
 210       op_or_tensor: `Operation` or `Tensor` to benchmark.
 211       feed_dict: A `dict` of values to feed for each op iteration (see the
 212         `feed_dict` parameter of `Session.run`).
 213       burn_iters: Number of burn-in iterations to run.
 214       min_iters: Minimum number of iterations to use for timing.
 215       store_trace: Boolean, whether to run an extra untimed iteration and
 216         store the trace of iteration in returned extras.
 217         The trace will be stored as a string in Google Chrome trace format
 218         in the extras field "full_trace_chrome_format". Note that trace
 219         will not be stored in test_log_pb2.TestResults proto.
 220       store_memory_usage: Boolean, whether to run an extra untimed iteration,
 221         calculate memory usage, and store that in extras fields.
 222       name: (optional) Override the BenchmarkEntry name with `name`.
 223         Otherwise it is inferred from the top-level method name.
 224       extras: (optional) Dict mapping string keys to additional benchmark info.
 225         Values may be either floats or values that are convertible to strings.
 226       mbs: (optional) The number of megabytes moved by this op, used to
 227         calculate the ops throughput.
 228 
 229     Returns:
 230       A `dict` containing the key-value pairs that were passed to
 231       `report_benchmark`. If `store_trace` option is used, then
 232       `full_chrome_trace_format` will be included in return dictionary even
 233       though it is not passed to `report_benchmark` with `extras`.
 234     """
 235     for _ in range(burn_iters):
 236       sess.run(op_or_tensor, feed_dict=feed_dict)
 237 
 238     deltas = [None] * min_iters
 239 
 240     for i in range(min_iters):
 241       start_time = time.time()
 242       sess.run(op_or_tensor, feed_dict=feed_dict)
 243       end_time = time.time()
 244       delta = end_time - start_time
 245       deltas[i] = delta
 246 
 247     extras = extras if extras is not None else {}
 248     unreported_extras = {}
 249     if store_trace or store_memory_usage:
 250       run_options = config_pb2.RunOptions(
 251           trace_level=config_pb2.RunOptions.FULL_TRACE)
 252       run_metadata = config_pb2.RunMetadata()
 253       sess.run(op_or_tensor, feed_dict=feed_dict,
 254                options=run_options, run_metadata=run_metadata)
 255       tl = timeline.Timeline(run_metadata.step_stats)
 256 
 257       if store_trace:
 258         unreported_extras["full_trace_chrome_format"] = (
 259             tl.generate_chrome_trace_format())
 260 
 261       if store_memory_usage:
 262         step_stats_analysis = tl.analyze_step_stats(show_memory=True)
 263         allocator_maximums = step_stats_analysis.allocator_maximums
 264         for k, v in allocator_maximums.items():
 265           extras["allocator_maximum_num_bytes_%s" % k] = v.num_bytes
 266 
 267     def _median(x):
 268       if not x:
 269         return -1
 270       s = sorted(x)
 271       l = len(x)
 272       lm1 = l - 1
 273       return (s[l//2] + s[lm1//2]) / 2.0
 274 
 275     median_delta = _median(deltas)
 276 
 277     benchmark_values = {
 278         "iters": min_iters,
 279         "wall_time": median_delta,
 280         "extras": extras,
 281         "name": name,
 282         "throughput": mbs / median_delta
 283     }
 284     self.report_benchmark(**benchmark_values)
 285     benchmark_values["extras"].update(unreported_extras)
 286     return benchmark_values

 289 def _run_benchmarks(regex):
 290   """Run benchmarks that match regex `regex`.
 291 
 292   This function goes through the global benchmark registry, and matches
 293   benchmark class and method names of the form
 294   `module.name.BenchmarkClass.benchmarkMethod` to the given regex.
 295   If a method matches, it is run.
 296 
 297   Args:
 298     regex: The string regular expression to match Benchmark classes against.
 299   """
 300   registry = list(GLOBAL_BENCHMARK_REGISTRY)
 301 
 302   # Match benchmarks in registry against regex
 303   for benchmark in registry:
 304     benchmark_name = "%s.%s" % (benchmark.__module__, benchmark.__name__)
 305     attrs = dir(benchmark)
 306     # Don't instantiate the benchmark class unless necessary
 307     benchmark_instance = None
 308 
 309     for attr in attrs:
 310       if not attr.startswith("benchmark"):
 311         continue
 312       candidate_benchmark_fn = getattr(benchmark, attr)
 313       if not callable(candidate_benchmark_fn):
 314         continue
 315       full_benchmark_name = "%s.%s" % (benchmark_name, attr)
 316       if regex == "all" or re.search(regex, full_benchmark_name):
 317         # Instantiate the class if it hasn't been instantiated
 318         benchmark_instance = benchmark_instance or benchmark()
 319         # Get the method tied to the class
 320         instance_benchmark_fn = getattr(benchmark_instance, attr)
 321         # Call the instance method
 322         instance_benchmark_fn()

 325 def benchmarks_main(true_main, argv=None):
 326   """Run benchmarks as declared in argv.
 327 
 328   Args:
 329     true_main: True main function to run if benchmarks are not requested.
 330     argv: the command line arguments (if None, uses sys.argv).
 331   """
 332   if argv is None:
 333     argv = sys.argv
 334   found_arg = [arg for arg in argv
 335                if arg.startswith("--benchmarks=")
 336                or arg.startswith("-benchmarks=")]
 337   if found_arg:
 338     # Remove --benchmarks arg from sys.argv
 339     argv.remove(found_arg[0])
 340 
 341     regex = found_arg[0].split("=")[1]
 342     app.run(lambda _: _run_benchmarks(regex), argv=argv)
 343   else:
 344     true_main()
