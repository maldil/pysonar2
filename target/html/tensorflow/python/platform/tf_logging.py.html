  40 # Don't use this directly. Use _get_logger() instead.
  41 _logger = None
  42 _logger_lock = threading.Lock()
  43 
  44 
  45 def _get_logger():
  46   global _logger
  47 
  48   # Use double-checked locking to avoid taking lock unnecessarily.
  49   if _logger:
  50     return _logger
  51 
  52   _logger_lock.acquire()
  53 
  54   try:
  55     if _logger:
  56       return _logger
  57 
  58     # Scope the TensorFlow logger to not conflict with users' loggers.
  59     logger = _logging.getLogger('tensorflow')
  60 
  61     # Don't further configure the TensorFlow logger if the root logger is
  62     # already configured. This prevents double logging in those cases.
  63     if not _logging.getLogger().handlers:
  64       # Determine whether we are in an interactive environment
  65       _interactive = False
  66       try:
  67         # This is only defined in interactive shells.
  68         if _sys.ps1: _interactive = True
  69       except AttributeError:
  70         # Even now, we may be in an interactive shell with `python -i`.
  71         _interactive = _sys.flags.interactive
  72 
  73       # If we are in an interactive environment (like Jupyter), set loglevel
  74       # to INFO and pipe the output to stdout.
  75       if _interactive:
  76         logger.setLevel(INFO)
  77         _logging_target = _sys.stdout
  78       else:
  79         _logging_target = _sys.stderr
  80 
  81       # Add the output handler.
  82       _handler = _logging.StreamHandler(_logging_target)
  83       _handler.setFormatter(_logging.Formatter(_logging.BASIC_FORMAT, None))
  84       logger.addHandler(_handler)
  85 
  86     _logger = logger
  87     return _logger
  88 
  89   finally:
  90     _logger_lock.release()
  91

 162 def _GetNextLogCountPerToken(token):
 163   """Wrapper for _log_counter_per_token.
 164 
 165   Args:
 166     token: The token for which to look up the count.
 167 
 168   Returns:
 169     The number of times this function has been called with
 170     *token* as an argument (starting at 0)
 171   """
 172   global _log_counter_per_token  # pylint: disable=global-variable-not-assigned
 173   _log_counter_per_token[token] = 1 + _log_counter_per_token.get(token, -1)
 174   return _log_counter_per_token[token]

 177 @tf_export('logging.log_every_n')
 178 def log_every_n(level, msg, n, *args):
 179   """Log 'msg % args' at level 'level' once per 'n' times.
 180 
 181   Logs the 1st call, (N+1)st call, (2N+1)st call,  etc.
 182   Not threadsafe.
 183 
 184   Args:
 185     level: The level at which to log.
 186     msg: The message to be logged.
 187     n: The number of times this should be called before it is logged.
 188     *args: The args to be substituted into the msg.
 189   """
 190   count = _GetNextLogCountPerToken(_GetFileAndLine())
 191   log_if(level, msg, not (count % n), *args)
 192

 194 @tf_export('logging.log_first_n')
 195 def log_first_n(level, msg, n, *args):  # pylint: disable=g-bad-name
 196   """Log 'msg % args' at level 'level' only first 'n' times.
 197 
 198   Not threadsafe.
 199 
 200   Args:
 201     level: The level at which to log.
 202     msg: The message to be logged.
 203     n: The number of times this should be called before it is logged.
 204     *args: The args to be substituted into the msg.
 205   """
 206   count = _GetNextLogCountPerToken(_GetFileAndLine())
 207   log_if(level, msg, count < n, *args)

 210 @tf_export('logging.log_if')
 211 def log_if(level, msg, condition, *args):
 212   """Log 'msg % args' at level 'level' only if condition is fulfilled."""
 213   if condition:
 214     vlog(level, msg, *args)

 217 def _GetFileAndLine():
 218   """Returns (filename, linenumber) for the stack frame."""
 219   # Use sys._getframe().  This avoids creating a traceback object.
 220   # pylint: disable=protected-access
 221   f = _sys._getframe()
 222   # pylint: enable=protected-access
 223   our_file = f.f_code.co_filename
 224   f = f.f_back
 225   while f:
 226     code = f.f_code
 227     if code.co_filename != our_file:
 228       return (code.co_filename, f.f_lineno)
 229     f = f.f_back
 230   return ('<unknown>', 0)

 233 def google2_log_prefix(level, timestamp=None, file_and_line=None):
 234   """Assemble a logline prefix using the google2 format."""
 235   # pylint: disable=global-variable-not-assigned
 236   global _level_names
 237   # pylint: enable=global-variable-not-assigned
 238 
 239   # Record current time
 240   now = timestamp or _time.time()
 241   now_tuple = _time.localtime(now)
 242   now_microsecond = int(1e6 * (now % 1.0))
 243 
 244   (filename, line) = file_and_line or _GetFileAndLine()
 245   basename = _os.path.basename(filename)
 246 
 247   # Severity string
 248   severity = 'I'
 249   if level in _level_names:
 250     severity = _level_names[level][0]
 251 
 252   s = '%c%02d%02d %02d:%02d:%02d.%06d %5d %s:%d] ' % (
 253       severity,
 254       now_tuple[1],  # month
 255       now_tuple[2],  # day
 256       now_tuple[3],  # hour
 257       now_tuple[4],  # min
 258       now_tuple[5],  # sec
 259       now_microsecond,
 260       _get_thread_id(),
 261       basename,
 262       line)
 263 
 264   return s

 279 def _get_thread_id():
 280   """Get id of current thread, suitable for logging as an unsigned quantity."""
 281   # pylint: disable=protected-access
 282   thread_id = six.moves._thread.get_ident()
 283   # pylint:enable=protected-access
 284   return thread_id & _THREAD_ID_MASK
