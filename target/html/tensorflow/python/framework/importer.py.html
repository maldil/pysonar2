  35 def _IsControlInput(input_name):
  36   # Expected format: '^operation_name' (control input).
  37   return input_name.startswith('^')

  40 def _ParseTensorName(tensor_name):
  41   """Parses a tensor name into an operation name and output index.
  42 
  43   This function will canonicalize tensor names as follows:
  44 
  45   * "foo:0"       -> ("foo", 0)
  46   * "foo:7"       -> ("foo", 7)
  47   * "foo"         -> ("foo", 0)
  48   * "foo:bar:baz" -> ValueError
  49 
  50   Args:
  51     tensor_name: The name of a tensor.
  52 
  53   Returns:
  54     A tuple containing the operation name, and the output index.
  55 
  56   Raises:
  57     ValueError: If `tensor_name' cannot be interpreted as the name of a tensor.
  58   """
  59   components = tensor_name.split(':')
  60   if len(components) == 2:
  61     # Expected format: 'operation_name:output_index'.
  62     try:
  63       output_index = int(components[1])
  64     except ValueError:
  65       raise ValueError('Cannot convert %r to a tensor name.' % (tensor_name,))
  66     return components[0], output_index
  67   elif len(components) == 1:
  68     # Expected format: 'operation_name' (implicit 0th output).
  69     return components[0], 0
  70   else:
  71     raise ValueError('Cannot convert %r to a tensor name.' % (tensor_name,))

  75 def _MaybeDevice(device):
  76   """Applies the given device only if device is not None or empty."""
  77   if device:
  78     with ops.device(device):
  79       yield
  80   else:
  81     yield

  84 def _ProcessGraphDefParam(graph_def, op_dict):
  85   """Type-checks and possibly canonicalizes `graph_def`."""
  86   if not isinstance(graph_def, graph_pb2.GraphDef):
  87     # `graph_def` could be a dynamically-created message, so try a duck-typed
  88     # approach
  89     try:
  90       old_graph_def = graph_def
  91       graph_def = graph_pb2.GraphDef()
  92       graph_def.MergeFrom(old_graph_def)
  93     except TypeError:
  94       raise TypeError('graph_def must be a GraphDef proto.')
  95   else:
  96     # If we're using the graph_def provided by the caller, modify graph_def
  97     # in-place to add attr defaults to the NodeDefs (this is visible to the
  98     # caller).
  99     # NOTE(skyewm): this is undocumented behavior that at least meta_graph.py
 100     # depends on. It might make sense to move this to meta_graph.py and have
 101     # import_graph_def not modify the graph_def argument (we'd have to make sure
 102     # this doesn't break anything else.)
 103     for node in graph_def.node:
 104       if node.op not in op_dict:
 105         # Assume unrecognized ops are functions for now. TF_ImportGraphDef will
 106         # report an error if the op is actually missing.
 107         continue
 108       op_def = op_dict[node.op]
 109       _SetDefaultAttrValues(node, op_def)
 110 
 111   return graph_def

 114 def _ProcessInputMapParam(input_map):
 115   """Type-checks and possibly canonicalizes `input_map`."""
 116   if input_map is None:
 117     input_map = {}
 118   else:
 119     if not (isinstance(input_map, dict) and all(
 120         isinstance(k, compat.bytes_or_text_types) for k in input_map.keys())):
 121       raise TypeError('input_map must be a dictionary mapping strings to '
 122                       'Tensor objects.')
 123   return input_map

 126 def _ProcessReturnElementsParam(return_elements):
 127   """Type-checks and possibly canonicalizes `return_elements`."""
 128   if return_elements is None:
 129     return None
 130   if not all(
 131       isinstance(x, compat.bytes_or_text_types) for x in return_elements):
 132     raise TypeError('return_elements must be a list of strings.')
 133   return tuple(compat.as_str(x) for x in return_elements)

 136 def _FindAttrInOpDef(attr_name, op_def):
 137   for attr_def in op_def.attr:
 138     if attr_name == attr_def.name:
 139       return attr_def
 140   return None

 143 def _RemoveDefaultAttrs(op_dict, producer_op_list, graph_def):
 144   """Removes unknown default attrs according to `producer_op_list`.
 145 
 146   Removes any unknown attrs in `graph_def` (i.e. attrs that do not appear in
 147   the OpDefs in `op_dict`) that have a default value in `producer_op_list`.
 148 
 149   Args:
 150     op_dict: dict mapping operation name to OpDef.
 151     producer_op_list: OpList proto.
 152     graph_def: GraphDef proto
 153   """
 154   producer_op_dict = {op.name: op for op in producer_op_list.op}
 155   for node in graph_def.node:
 156     # Remove any default attr values that aren't in op_def.
 157     if node.op in producer_op_dict:
 158       op_def = op_dict[node.op]
 159       producer_op_def = producer_op_dict[node.op]
 160       # We make a copy of node.attr to iterate through since we may modify
 161       # node.attr inside the loop.
 162       for key in list(node.attr):
 163         if _FindAttrInOpDef(key, op_def) is None:
 164           # No attr_def in consumer, look in producer.
 165           attr_def = _FindAttrInOpDef(key, producer_op_def)
 166           if (attr_def and attr_def.HasField('default_value') and
 167               node.attr[key] == attr_def.default_value):
 168             # Unknown attr had default value in producer, delete it so it can be
 169             # understood by consumer.
 170             del node.attr[key]

 173 def _ConvertInputMapValues(name, input_map):
 174   """Ensures all input map values are tensors.
 175 
 176   This should be called from inside the import name scope.
 177 
 178   Args:
 179     name: the `name` argument passed to import_graph_def
 180     input_map: the `input_map` argument passed to import_graph_def.
 181 
 182   Returns:
 183     An possibly-updated version of `input_map`.
 184 
 185   Raises:
 186     ValueError: if input map values cannot be converted due to empty name scope.
 187   """
 188   if not all(isinstance(v, ops.Tensor) for v in input_map.values()):
 189     if name == '':  # pylint: disable=g-explicit-bool-comparison
 190       raise ValueError(
 191           'tf.import_graph_def() requires a non-empty `name` if `input_map` '
 192           'contains non-Tensor values. Try calling tf.convert_to_tensor() on '
 193           '`input_map` values before calling tf.import_graph_def().')
 194     with ops.name_scope('_inputs'):
 195       input_map = {k: ops.convert_to_tensor(v) for k, v in input_map.items()}
 196   return input_map

 199 def _PopulateTFImportGraphDefOptions(options, prefix, input_map,
 200                                      return_elements):
 201   """Populates the TF_ImportGraphDefOptions `options`."""
 202   c_api.TF_ImportGraphDefOptionsSetPrefix(options, prefix)
 203   c_api.TF_ImportGraphDefOptionsSetUniquifyNames(options, True)
 204 
 205   for input_src, input_dst in input_map.items():
 206     input_src = compat.as_str(input_src)
 207     if input_src.startswith('^'):
 208       src_name = compat.as_str(input_src[1:])
 209       dst_op = input_dst._as_tf_output().oper  # pylint: disable=protected-access
 210       c_api.TF_ImportGraphDefOptionsRemapControlDependency(
 211           options, src_name, dst_op)
 212     else:
 213       src_name, src_idx = _ParseTensorName(input_src)
 214       src_name = compat.as_str(src_name)
 215       dst_output = input_dst._as_tf_output()  # pylint: disable=protected-access
 216       c_api.TF_ImportGraphDefOptionsAddInputMapping(options, src_name, src_idx,
 217                                                     dst_output)
 218   for name in return_elements or []:
 219     if ':' in name:
 220       op_name, index = _ParseTensorName(name)
 221       op_name = compat.as_str(op_name)
 222       c_api.TF_ImportGraphDefOptionsAddReturnOutput(options, op_name, index)
 223     else:
 224       c_api.TF_ImportGraphDefOptionsAddReturnOperation(options,
 225                                                        compat.as_str(name))

 228 def _ProcessNewOps(graph):
 229   """Processes the newly-added TF_Operations in `graph`."""
 230   # Maps from a node to the names of the ops it's colocated with, if colocation
 231   # is specified in the attributes.
 232   colocation_pairs = {}
 233 
 234   for new_op in graph._add_new_tf_operations(compute_devices=False):  # pylint: disable=protected-access
 235     original_device = new_op.device
 236     new_op._set_device('')  # pylint: disable=protected-access
 237     colocation_names = _GetColocationNames(new_op)
 238     if colocation_names:
 239       colocation_pairs[new_op] = colocation_names
 240       # Don't set a device for this op, since colocation constraints override
 241       # device functions and the original device. Note that this op's device may
 242       # still be set by the loop below.
 243       # TODO(skyewm): why does it override the original device?
 244     else:
 245       with _MaybeDevice(original_device):
 246         graph._apply_device_functions(new_op)  # pylint: disable=protected-access
 247 
 248   # The following loop populates the device field of ops that are colocated
 249   # with another op.  This is implied by the colocation attribute, but we
 250   # propagate the device field for completeness.
 251   for op, coloc_op_list in colocation_pairs.items():
 252     coloc_device = None
 253     # Find any device in the list of colocated ops that have a device, if it
 254     # exists.  We assume that if multiple ops have devices, they refer to the
 255     # same device.  Otherwise, a runtime error will occur since the colocation
 256     # property cannot be guaranteed.
 257     #
 258     # One possible improvement is to try to check for compatibility of all
 259     # devices in this list at import time here, which would require
 260     # implementing a compatibility function for device specs in python.
 261     for coloc_op_name in coloc_op_list:
 262       try:
 263         coloc_op = graph._get_operation_by_name_unsafe(coloc_op_name)  # pylint: disable=protected-access
 264       except KeyError:
 265         raise ValueError('Specified colocation to an op that '
 266                          'does not exist during import: %s in %s' %
 267                          (coloc_op_name, op.name))
 268       if coloc_op.device:
 269         coloc_device = pydev.DeviceSpec.from_string(coloc_op.device)
 270         break
 271     if coloc_device:
 272       op._set_device(coloc_device)  # pylint: disable=protected-access

 275 def _GetColocationNames(op):
 276   """Returns names of the ops that `op` should be colocated with."""
 277   colocation_names = []
 278   try:
 279     class_values = op.get_attr('_class')
 280   except ValueError:
 281     # No _class attr
 282     return
 283   for val in class_values:
 284     val = compat.as_str(val)
 285     if val.startswith('loc:@'):
 286       colocation_node_name = val[len('loc:@'):]
 287       if colocation_node_name != op.name:
 288         colocation_names.append(colocation_node_name)
 289   return colocation_names

 292 def _GatherReturnElements(requested_return_elements, graph, results):
 293   """Returns the requested return elements from results.
 294 
 295   Args:
 296     requested_return_elements: list of strings of operation and tensor names
 297     graph: Graph
 298     results: wrapped TF_ImportGraphDefResults
 299 
 300   Returns:
 301     list of `Operation` and/or `Tensor` objects
 302   """
 303   return_outputs = c_api.TF_ImportGraphDefResultsReturnOutputs(results)
 304   return_opers = c_api.TF_ImportGraphDefResultsReturnOperations(results)
 305 
 306   combined_return_elements = []
 307   outputs_idx = 0
 308   opers_idx = 0
 309   for name in requested_return_elements:
 310     if ':' in name:
 311       combined_return_elements.append(
 312           graph._get_tensor_by_tf_output(return_outputs[outputs_idx]))  # pylint: disable=protected-access
 313       outputs_idx += 1
 314     else:
 315       combined_return_elements.append(
 316           graph._get_operation_by_tf_operation(return_opers[opers_idx]))  # pylint: disable=protected-access
 317       opers_idx += 1
 318   return combined_return_elements

 321 def _SetDefaultAttrValues(node_def, op_def):
 322   """Set any default attr values in `node_def` that aren't present."""
 323   assert node_def.op == op_def.name
 324   for attr_def in op_def.attr:
 325     key = attr_def.name
 326     if attr_def.HasField('default_value'):
 327       value = node_def.attr[key]
 328       if value is None or value.WhichOneof('value') is None:
 329         node_def.attr[key].CopyFrom(attr_def.default_value)

 332 @tf_export('graph_util.import_graph_def', 'import_graph_def')
 333 @deprecated_args(None, 'Please file an issue at '
 334                  'https://github.com/tensorflow/tensorflow/issues if you depend'
 335                  ' on this feature.', 'op_dict')
 336 def import_graph_def(graph_def,
 337                      input_map=None,
 338                      return_elements=None,
 339                      name=None,
 340                      op_dict=None,
 341                      producer_op_list=None):
 342   """Imports the graph from `graph_def` into the current default `Graph`.
 343 
 344   This function provides a way to import a serialized TensorFlow
 345   [`GraphDef`](https://www.tensorflow.org/code/tensorflow/core/framework/graph.proto)
 346   protocol buffer, and extract individual objects in the `GraphDef` as
 347   `tf.Tensor` and `tf.Operation` objects. Once extracted,
 348   these objects are placed into the current default `Graph`. See
 349   `tf.Graph.as_graph_def` for a way to create a `GraphDef`
 350   proto.
 351 
 352   Args:
 353     graph_def: A `GraphDef` proto containing operations to be imported into
 354       the default graph.
 355     input_map: A dictionary mapping input names (as strings) in `graph_def`
 356       to `Tensor` objects. The values of the named input tensors in the
 357       imported graph will be re-mapped to the respective `Tensor` values.
 358     return_elements: A list of strings containing operation names in
 359       `graph_def` that will be returned as `Operation` objects; and/or
 360       tensor names in `graph_def` that will be returned as `Tensor` objects.
 361     name: (Optional.) A prefix that will be prepended to the names in
 362       `graph_def`. Note that this does not apply to imported function names.
 363       Defaults to `"import"`.
 364     op_dict: (Optional.) Deprecated, do not use.
 365     producer_op_list: (Optional.) An `OpList` proto with the (possibly stripped)
 366       list of `OpDef`s used by the producer of the graph. If provided,
 367       unrecognized attrs for ops in `graph_def` that have their default value
 368       according to `producer_op_list` will be removed. This will allow some more
 369       `GraphDef`s produced by later binaries to be accepted by earlier binaries.
 370 
 371   Returns:
 372     A list of `Operation` and/or `Tensor` objects from the imported graph,
 373     corresponding to the names in `return_elements`.
 374 
 375   Raises:
 376     TypeError: If `graph_def` is not a `GraphDef` proto,
 377       `input_map` is not a dictionary mapping strings to `Tensor` objects,
 378       or `return_elements` is not a list of strings.
 379     ValueError: If `input_map`, or `return_elements` contains names that
 380       do not appear in `graph_def`, or `graph_def` is not well-formed (e.g.
 381       it refers to an unknown tensor).
 382   """
 383   op_dict = op_def_registry.get_registered_ops()
 384 
 385   graph_def = _ProcessGraphDefParam(graph_def, op_dict)
 386   input_map = _ProcessInputMapParam(input_map)
 387   return_elements = _ProcessReturnElementsParam(return_elements)
 388 
 389   if producer_op_list is not None:
 390     # TODO(skyewm): make a copy of graph_def so we're not mutating the argument?
 391     _RemoveDefaultAttrs(op_dict, producer_op_list, graph_def)
 392 
 393   graph = ops.get_default_graph()
 394   with ops.name_scope(name, 'import', input_map.values()) as scope:
 395     # Save unique prefix generated by name_scope
 396     if scope:
 397       assert scope.endswith('/')
 398       prefix = scope[:-1]
 399     else:
 400       prefix = ''
 401 
 402     # Generate any input map tensors inside name scope
 403     input_map = _ConvertInputMapValues(name, input_map)
 404 
 405   scoped_options = c_api_util.ScopedTFImportGraphDefOptions()
 406   options = scoped_options.options
 407   _PopulateTFImportGraphDefOptions(options, prefix, input_map,
 408                                    return_elements)
 409 
 410   # _ProcessNewOps mutates the new operations. _mutation_lock ensures a
 411   # Session.run call cannot occur between creating the TF_Operations in the
 412   # TF_GraphImportGraphDefWithResults call and mutating the them in
 413   # _ProcessNewOps.
 414   with graph._mutation_lock():  # pylint: disable=protected-access
 415     with c_api_util.tf_buffer(graph_def.SerializeToString()) as serialized:
 416       try:
 417         results = c_api.TF_GraphImportGraphDefWithResults(
 418             graph._c_graph, serialized, options)  # pylint: disable=protected-access
 419         results = c_api_util.ScopedTFImportGraphDefResults(results)
 420       except errors.InvalidArgumentError as e:
 421         # Convert to ValueError for backwards compatibility.
 422         raise ValueError(str(e))
 423 
 424     # Create _DefinedFunctions for any imported functions.
 425     #
 426     # We do this by creating _DefinedFunctions directly from `graph_def`, and
 427     # adding them to `graph`. Adding an existing function to a TF_Graph is a
 428     # no-op, so this only has the effect of updating the Python state (usually
 429     # _DefinedFunction.add_to_graph also adds the function to the TF_Graph).
 430     #
 431     # TODO(skyewm): fetch the TF_Functions directly from the TF_Graph
 432     # TODO(skyewm): avoid sending serialized FunctionDefs back to the TF_Graph
 433     # TODO(b/74620627): move this after _ProcessNewOps outside the lock once
 434     # _USE_C_SHAPES is removed.
 435     if graph_def.library and graph_def.library.function:
 436       # pylint: disable=protected-access
 437       functions = function._from_library(graph_def.library)
 438       for f in functions:
 439         f.add_to_graph(graph)
 440       # pylint: enable=protected-access
 441 
 442     _ProcessNewOps(graph)
 443 
 444   # Treat input mappings that don't appear in the graph as an error, because
 445   # they are likely to be due to a typo.
 446   missing_unused_input_keys = (
 447       c_api.TF_ImportGraphDefResultsMissingUnusedInputMappings_wrapper(
 448           results.results))
 449   if missing_unused_input_keys:
 450     missing_unused_input_keys = [
 451         compat.as_str(s) for s in missing_unused_input_keys
 452     ]
 453     raise ValueError(
 454         'Attempted to map inputs that were not found in graph_def: [%s]' %
 455         ', '.join(missing_unused_input_keys))
 456 
 457   if return_elements is None:
 458     return None
 459   else:
 460     return _GatherReturnElements(return_elements, graph, results.results)
