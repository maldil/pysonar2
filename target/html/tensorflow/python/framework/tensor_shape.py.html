  26 @tf_export("Dimension")
  27 class Dimension(object):
  28   """Represents the value of one dimension in a TensorShape."""
  29 
  30   def __init__(self, value):
  31     """Creates a new Dimension with the given value."""
  32     if value is None:
  33       self._value = None
  34     elif isinstance(value, dtypes.DType):
  35       raise TypeError("Cannot convert %s to Dimension" % value)
  36     else:
  37       self._value = int(value)
  38       if (not isinstance(value, compat.bytes_or_text_types) and
  39           self._value != value):
  40         raise ValueError("Ambiguous dimension: %s" % value)
  41       if self._value < 0:
  42         raise ValueError("Dimension %d must be >= 0" % self._value)
  43 
  44   def __repr__(self):
  45     return "Dimension(%s)" % repr(self._value)
  46 
  47   def __str__(self):
  48     value = self._value
  49     return "?" if value is None else str(value)
  50 
  51   def __eq__(self, other):
  52     """Returns true if `other` has the same known value as this Dimension."""
  53     try:
  54       other = as_dimension(other)
  55     except (TypeError, ValueError):
  56       return NotImplemented
  57     if self._value is None or other.value is None:
  58       return None
  59     return self._value == other.value
  60 
  61   def __ne__(self, other):
  62     """Returns true if `other` has a different known value from `self`."""
  63     try:
  64       other = as_dimension(other)
  65     except (TypeError, ValueError):
  66       return NotImplemented
  67     if self._value is None or other.value is None:
  68       return None
  69     return self._value != other.value
  70 
  71   def __int__(self):
  72     return self._value
  73 
  74   # This is needed for Windows.
  75   # See https://github.com/tensorflow/tensorflow/pull/9780
  76   def __long__(self):
  77     return self._value
  78 
  79   def __index__(self):
  80     # Allow use in Python 3 range
  81     return self._value

  83   @property
  84   def value(self):
  85     """The value of this dimension, or None if it is unknown."""
  86     return self._value

  88   def is_compatible_with(self, other):
  89     """Returns true if `other` is compatible with this Dimension.
  90 
  91     Two known Dimensions are compatible if they have the same value.
  92     An unknown Dimension is compatible with all other Dimensions.
  93 
  94     Args:
  95       other: Another Dimension.
  96 
  97     Returns:
  98       True if this Dimension and `other` are compatible.
  99     """
 100     other = as_dimension(other)
 101     return (self._value is None or other.value is None or
 102             self._value == other.value)
 103 
 104   def assert_is_compatible_with(self, other):
 105     """Raises an exception if `other` is not compatible with this Dimension.
 106 
 107     Args:
 108       other: Another Dimension.
 109 
 110     Raises:
 111       ValueError: If `self` and `other` are not compatible (see
 112         is_compatible_with).
 113     """
 114     if not self.is_compatible_with(other):
 115       raise ValueError("Dimensions %s and %s are not compatible" % (self,
 116                                                                     other))

 118   def merge_with(self, other):
 119     """Returns a Dimension that combines the information in `self` and `other`.
 120 
 121     Dimensions are combined as follows:
 122 
 123     ```python
 124     tf.Dimension(n)   .merge_with(tf.Dimension(n))    == tf.Dimension(n)
 125     tf.Dimension(n)   .merge_with(tf.Dimension(None)) == tf.Dimension(n)
 126     tf.Dimension(None).merge_with(tf.Dimension(n))    == tf.Dimension(n)
 127     tf.Dimension(None).merge_with(tf.Dimension(None)) == tf.Dimension(None)
 128     tf.Dimension(n)   .merge_with(tf.Dimension(m))  # raises ValueError for n != m
 129     ```
 130 
 131     Args:
 132       other: Another Dimension.
 133 
 134     Returns:
 135       A Dimension containing the combined information of `self` and
 136       `other`.
 137 
 138     Raises:
 139       ValueError: If `self` and `other` are not compatible (see
 140         is_compatible_with).
 141     """
 142     other = as_dimension(other)
 143     self.assert_is_compatible_with(other)
 144     if self._value is None:
 145       return Dimension(other.value)
 146     else:
 147       return Dimension(self._value)

 149   def __add__(self, other):
 150     """Returns the sum of `self` and `other`.
 151 
 152     Dimensions are summed as follows:
 153 
 154     ```python
 155     tf.Dimension(m)    + tf.Dimension(n)    == tf.Dimension(m + n)
 156     tf.Dimension(m)    + tf.Dimension(None) == tf.Dimension(None)
 157     tf.Dimension(None) + tf.Dimension(n)    == tf.Dimension(None)
 158     tf.Dimension(None) + tf.Dimension(None) == tf.Dimension(None)
 159     ```
 160 
 161     Args:
 162       other: Another Dimension, or a value accepted by `as_dimension`.
 163 
 164     Returns:
 165       A Dimension whose value is the sum of `self` and `other`.
 166     """
 167     other = as_dimension(other)
 168     if self._value is None or other.value is None:
 169       return Dimension(None)
 170     else:
 171       return Dimension(self._value + other.value)
 172

 466 def as_dimension(value):
 467   """Converts the given value to a Dimension.
 468 
 469   A Dimension input will be returned unmodified.
 470   An input of `None` will be converted to an unknown Dimension.
 471   An integer input will be converted to a Dimension with that value.
 472 
 473   Args:
 474     value: The value to be converted.
 475 
 476   Returns:
 477     A Dimension corresponding to the given value.
 478   """
 479   if isinstance(value, Dimension):
 480     return value
 481   else:
 482     return Dimension(value)

 485 @tf_export("TensorShape")
 486 class TensorShape(object):
 487   """Represents the shape of a `Tensor`.
 488 
 489   A `TensorShape` represents a possibly-partial shape specification for a
 490   `Tensor`. It may be one of the following:
 491 
 492   * *Fully-known shape:* has a known number of dimensions and a known size
 493     for each dimension. e.g. `TensorShape([16, 256])`
 494   * *Partially-known shape:* has a known number of dimensions, and an unknown
 495     size for one or more dimension. e.g. `TensorShape([None, 256])`
 496   * *Unknown shape:* has an unknown number of dimensions, and an unknown
 497     size in all dimensions. e.g. `TensorShape(None)`
 498 
 499   If a tensor is produced by an operation of type `"Foo"`, its shape
 500   may be inferred if there is a registered shape function for
 501   `"Foo"`. See [Shape
 502   functions](https://tensorflow.org/extend/adding_an_op#shape_functions_in_c)
 503   for details of shape functions and how to register them. Alternatively,
 504   the shape may be set explicitly using `tf.Tensor.set_shape`.
 505   """
 506 
 507   def __init__(self, dims):
 508     """Creates a new TensorShape with the given dimensions.
 509 
 510     Args:
 511       dims: A list of Dimensions, or None if the shape is unspecified.
 512         DEPRECATED: A single integer is treated as a singleton list.
 513 
 514     Raises:
 515       TypeError: If dims cannot be converted to a list of dimensions.
 516     """
 517     # TODO(irving): Eliminate the single integer special case.
 518     if dims is None:
 519       self._dims = None
 520     elif isinstance(dims, compat.bytes_or_text_types):
 521       raise TypeError("A string has ambiguous TensorShape, please wrap in a "
 522                       "list or convert to an int: %s" % dims)
 523     elif isinstance(dims, tensor_shape_pb2.TensorShapeProto):
 524       if dims.unknown_rank:
 525         self._dims = None
 526       else:
 527         self._dims = [
 528             # Protos store variable-size dimensions as -1
 529             as_dimension(dim.size if dim.size != -1 else None)
 530             for dim in dims.dim
 531         ]
 532     elif isinstance(dims, TensorShape):
 533       self._dims = dims.dims
 534     else:
 535       try:
 536         dims_iter = iter(dims)
 537       except TypeError:
 538         # Treat as a singleton dimension
 539         self._dims = [as_dimension(dims)]
 540       else:
 541         # Got a list of dimensions
 542         self._dims = [as_dimension(d) for d in dims_iter]
 543     self._ndims = None
 544 
 545   def __repr__(self):
 546     return "TensorShape(%r)" % self._dims
 547 
 548   def __str__(self):
 549     if self.ndims is None:
 550       return "<unknown>"
 551     elif self.ndims == 1:
 552       return "(%s,)" % self._dims[0]
 553     else:
 554       return "(%s)" % ", ".join(str(d) for d in self._dims)
 555 
 556   @property
 557   def dims(self):
 558     """Returns a list of Dimensions, or None if the shape is unspecified."""
 559     return self._dims
 560 
 561   @dims.setter
 562   def dims(self, dims):
 563     self._dims = dims
 564     self._ndims = None
 565 
 566   @property
 567   def ndims(self):
 568     """Returns the rank of this shape, or None if it is unspecified."""
 569     if self._dims is None:
 570       return None
 571     else:
 572       if self._ndims is None:
 573         self._ndims = len(self._dims)
 574       return self._ndims
 575 
 576   def __len__(self):
 577     """Returns the rank of this shape, or raises ValueError if unspecified."""
 578     if self._dims is None:
 579       raise ValueError("Cannot take the length of Shape with unknown rank.")
 580     return self.ndims
 581 
 582   def __bool__(self):
 583     """Returns True if this shape contains non-zero information."""
 584     return self._dims is not None
 585 
 586   # Python 3 wants __bool__, Python 2.7 wants __nonzero__
 587   __nonzero__ = __bool__
 588 
 589   def __iter__(self):
 590     """Returns `self.dims` if the rank is known, otherwise raises ValueError."""
 591     if self._dims is None:
 592       raise ValueError("Cannot iterate over a shape with unknown rank.")
 593     else:
 594       return iter(self._dims)
 595 
 596   def __getitem__(self, key):
 597     """Returns the value of a dimension or a shape, depending on the key.
 598 
 599     Args:
 600       key: If `key` is an integer, returns the dimension at that index;
 601         otherwise if `key` is a slice, returns a TensorShape whose
 602         dimensions are those selected by the slice from `self`.
 603 
 604     Returns:
 605       A dimension if `key` is an integer, or a `TensorShape` if `key` is a
 606       slice.
 607 
 608     Raises:
 609       ValueError: If `key` is a slice and `self` is completely unknown and
 610         the step is set.
 611     """
 612     if self._dims is not None:
 613       if isinstance(key, slice):
 614         return TensorShape(self._dims[key])
 615       else:
 616         return self._dims[key]
 617     else:
 618       if isinstance(key, slice):
 619         start = key.start if key.start is not None else 0
 620         stop = key.stop
 621 
 622         if key.step is not None:
 623           # TODO(mrry): Handle these maybe.
 624           raise ValueError("Steps are not yet handled")
 625         if stop is None:
 626           # NOTE(mrry): This implies that TensorShape(None) is compatible with
 627           # TensorShape(None)[1:], which is obviously not true. It would be
 628           # possible to track the number of dimensions symbolically,
 629           # and perhaps we should do that.
 630           return unknown_shape()
 631         elif start < 0 or stop < 0:
 632           # TODO(mrry): Handle this better, as it will be useful for handling
 633           # suffixes of otherwise unknown shapes.
 634           return unknown_shape()
 635         else:
 636           return unknown_shape(ndims=stop - start)
 637       else:
 638         return Dimension(None)
 639 
 640   def num_elements(self):
 641     """Returns the total number of elements, or none for incomplete shapes."""
 642     if self.is_fully_defined():
 643       size = 1
 644       for dim in self._dims:
 645         size *= dim.value
 646       return size
 647     else:
 648       return None
 649 
 650   def merge_with(self, other):
 651     """Returns a `TensorShape` combining the information in `self` and `other`.
 652 
 653     The dimensions in `self` and `other` are merged elementwise,
 654     according to the rules defined for `Dimension.merge_with()`.
 655 
 656     Args:
 657       other: Another `TensorShape`.
 658 
 659     Returns:
 660       A `TensorShape` containing the combined information of `self` and
 661       `other`.
 662 
 663     Raises:
 664       ValueError: If `self` and `other` are not compatible.
 665     """
 666     other = as_shape(other)
 667     if self._dims is None:
 668       return other
 669     else:
 670       try:
 671         self.assert_same_rank(other)
 672         new_dims = []
 673         for i, dim in enumerate(self._dims):
 674           new_dims.append(dim.merge_with(other[i]))
 675         return TensorShape(new_dims)
 676       except ValueError:
 677         raise ValueError("Shapes %s and %s are not compatible" % (self, other))
 678 
 679   def concatenate(self, other):
 680     """Returns the concatenation of the dimension in `self` and `other`.
 681 
 682     *N.B.* If either `self` or `other` is completely unknown,
 683     concatenation will discard information about the other shape. In
 684     future, we might support concatenation that preserves this
 685     information for use with slicing.
 686 
 687     Args:
 688       other: Another `TensorShape`.
 689 
 690     Returns:
 691       A `TensorShape` whose dimensions are the concatenation of the
 692       dimensions in `self` and `other`.
 693     """
 694     # TODO(mrry): Handle the case where we concatenate a known shape with a
 695     # completely unknown shape, so that we can use the partial information.
 696     other = as_shape(other)
 697     if self._dims is None or other.dims is None:
 698       return unknown_shape()
 699     else:
 700       return TensorShape(self._dims + other.dims)

 788   def is_compatible_with(self, other):
 789     """Returns True iff `self` is compatible with `other`.
 790 
 791     Two possibly-partially-defined shapes are compatible if there
 792     exists a fully-defined shape that both shapes can represent. Thus,
 793     compatibility allows the shape inference code to reason about
 794     partially-defined shapes. For example:
 795 
 796     * TensorShape(None) is compatible with all shapes.
 797 
 798     * TensorShape([None, None]) is compatible with all two-dimensional
 799       shapes, such as TensorShape([32, 784]), and also TensorShape(None). It is
 800       not compatible with, for example, TensorShape([None]) or
 801       TensorShape([None, None, None]).
 802 
 803     * TensorShape([32, None]) is compatible with all two-dimensional shapes
 804       with size 32 in the 0th dimension, and also TensorShape([None, None])
 805       and TensorShape(None). It is not compatible with, for example,
 806       TensorShape([32]), TensorShape([32, None, 1]) or TensorShape([64, None]).
 807 
 808     * TensorShape([32, 784]) is compatible with itself, and also
 809       TensorShape([32, None]), TensorShape([None, 784]), TensorShape([None,
 810       None]) and TensorShape(None). It is not compatible with, for example,
 811       TensorShape([32, 1, 784]) or TensorShape([None]).
 812 
 813     The compatibility relation is reflexive and symmetric, but not
 814     transitive. For example, TensorShape([32, 784]) is compatible with
 815     TensorShape(None), and TensorShape(None) is compatible with
 816     TensorShape([4, 4]), but TensorShape([32, 784]) is not compatible with
 817     TensorShape([4, 4]).
 818 
 819     Args:
 820       other: Another TensorShape.
 821 
 822     Returns:
 823       True iff `self` is compatible with `other`.
 824 
 825     """
 826     other = as_shape(other)
 827     if self._dims is not None and other.dims is not None:
 828       if self.ndims != other.ndims:
 829         return False
 830       for x_dim, y_dim in zip(self._dims, other.dims):
 831         if not x_dim.is_compatible_with(y_dim):
 832           return False
 833     return True
 834 
 835   def assert_is_compatible_with(self, other):
 836     """Raises exception if `self` and `other` do not represent the same shape.
 837 
 838     This method can be used to assert that there exists a shape that both
 839     `self` and `other` represent.
 840 
 841     Args:
 842       other: Another TensorShape.
 843 
 844     Raises:
 845       ValueError: If `self` and `other` do not represent the same shape.
 846     """
 847     if not self.is_compatible_with(other):
 848       raise ValueError("Shapes %s and %s are incompatible" % (self, other))
 849 
 850   def most_specific_compatible_shape(self, other):
 851     """Returns the most specific TensorShape compatible with `self` and `other`.
 852 
 853     * TensorShape([None, 1]) is the most specific TensorShape compatible with
 854       both TensorShape([2, 1]) and TensorShape([5, 1]). Note that
 855       TensorShape(None) is also compatible with above mentioned TensorShapes.
 856 
 857     * TensorShape([1, 2, 3]) is the most specific TensorShape compatible with
 858       both TensorShape([1, 2, 3]) and TensorShape([1, 2, 3]). There are more
 859       less specific TensorShapes compatible with above mentioned TensorShapes,
 860       e.g. TensorShape([1, 2, None]), TensorShape(None).
 861 
 862     Args:
 863       other: Another `TensorShape`.
 864 
 865     Returns:
 866       A `TensorShape` which is the most specific compatible shape of `self`
 867       and `other`.
 868     """
 869 
 870     other = as_shape(other)
 871     if self._dims is None or other.dims is None or self.ndims != other.ndims:
 872       return unknown_shape()
 873 
 874     dims = [(Dimension(None))] * self.ndims
 875     for i, (d1, d2) in enumerate(zip(self._dims, other.dims)):
 876       if d1 is not None and d2 is not None and d1 == d2:
 877         dims[i] = d1
 878     return TensorShape(dims)

 880   def is_fully_defined(self):
 881     """Returns True iff `self` is fully defined in every dimension."""
 882     return (self._dims is not None and all(dim.value is not None
 883                                            for dim in self._dims))
