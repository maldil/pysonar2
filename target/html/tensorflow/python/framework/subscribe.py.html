  31 def _recursive_apply(tensors, apply_fn):
  32   """Helper method to recursively apply a function to structure of tensors.
  33 
  34   The structure of the tensors should take the form similar to fetches in
  35   `tf.Session` and includes single `Tensor`, `list`, nested `list`, `tuple`,
  36   `namedtuple`, or `dict`.
  37 
  38   Args:
  39     tensors: Single `Tensor`, `list`, nested `list, `tuple`,
  40       `namedtuple`, or `dict`.
  41     apply_fn: Function to apply to each `Tensor` and should return a `Tensor`.
  42   Returns:
  43     Returns the modified tensors with the same structure.
  44   Raises:
  45     `TypeError` if undefined type in the tensors structure.
  46   """
  47   tensors_type = type(tensors)
  48   if tensors_type is ops.Tensor:
  49     return apply_fn(tensors)
  50   elif isinstance(tensors, variables.Variable):
  51     return apply_fn(tensors.value())
  52   elif isinstance(tensors, (list, tuple)):
  53     tensors = [_recursive_apply(t, apply_fn) for t in tensors]
  54     if tensors_type is list:
  55       return list(tensors)
  56     elif tensors_type is tuple:
  57       return tuple(tensors)
  58     return tensors_type(*tensors)  # collections.namedtuple
  59   elif tensors_type is dict:
  60     return dict([(k, _recursive_apply(v, apply_fn))
  61                  for k, v in tensors.items()])
  62   else:
  63     raise TypeError('_recursive_apply argument %r has invalid type %r' %
  64                     (tensors, tensors_type))

  67 class _ControlOutputCache(object):
  68   """Helper class to manage calculating and caching control_outputs in graph."""
  69 
  70   def __init__(self):
  71     self.cache = {}
  72 
  73   def calc_control_outputs(self, graph):
  74     """Returns the map of control_outputs for a given graph.
  75 
  76     Args:
  77       graph: The graph to parse.
  78     Returns:
  79       A map of the control outputs.
  80     """
  81     control_outputs = {}
  82     for op in graph.get_operations():
  83       for control_input in op.control_inputs:
  84         if control_input not in control_outputs:
  85           control_outputs[control_input] = set()
  86         control_outputs[control_input].add(op)
  87     return control_outputs
  88 
  89   def get_control_outputs(self, op):
  90     """Return the control outputs for a given op.
  91 
  92     Args:
  93       op: The op to fetch control outputs for.
  94     Returns:
  95       Iterable of control output ops.
  96     """
  97     if op.graph not in self.cache:
  98       control_outputs = self.calc_control_outputs(op.graph)
  99       self.cache[op.graph] = control_outputs
 100     else:
 101       control_outputs = self.cache[op.graph]
 102     return control_outputs.get(op, [])

 105 def _subscribe_new(tensor, side_effects, control_cache):
 106   """Helper method that subscribes a single tensor to a list of side_effects.
 107 
 108   Args:
 109     tensor: `tf.Tensor`
 110     side_effects: List of side_effect functions see subscribe for details.
 111     control_cache: `_ControlOutputCache` helper to get control_outputs faster.
 112   Returns:
 113     The modified replacement to the passed in tensor which triggers the side
 114     effects.
 115   """
 116   update_input = []
 117   for consumer_op in list(tensor.consumers()):  # explicit copy
 118     update_input.append((consumer_op, list(consumer_op.inputs).index(tensor)))
 119 
 120   update_control_input = control_cache.get_control_outputs(tensor.op)
 121 
 122   # Trailing slash on name scope to replace the scope.
 123   name_scope = tensor.op.name + '/subscription/'
 124   with ops.name_scope(name_scope):
 125     outs = []
 126     for s in side_effects:
 127       outs += s(tensor)
 128 
 129     with ops.control_dependencies(outs):
 130       out = array_ops.identity(tensor)
 131 
 132   for consumer_op, index in update_input:
 133     consumer_op._update_input(index, out)  # pylint: disable=protected-access
 134 
 135   for consumer_op in update_control_input:
 136     # If an op has more than one output and two or more of its output tensors
 137     # are subscribed at the same time, we remove the control dependency from
 138     # the original op only once and we add the dependencies to all the
 139     # new identities.
 140     new_control_inputs = consumer_op.control_inputs
 141     if tensor.op in new_control_inputs:
 142       new_control_inputs.remove(tensor.op)
 143     new_control_inputs.append(out.op)
 144     # pylint: disable=protected-access
 145     consumer_op._remove_all_control_inputs()
 146     consumer_op._add_control_inputs(new_control_inputs)
 147     # pylint: enable=protected-access
 148   return out

 151 def _subscribe_extend(tensor, side_effects):
 152   """Helper method to extend the list of side_effects for a subscribed tensor.
 153 
 154   Args:
 155     tensor: A `tf.Tensor` as returned by subscribe().
 156     side_effects: List of side_effect functions, see subscribe for details.
 157   Returns:
 158     The given subscribed tensor (for API consistency).
 159   """
 160   assert len(tensor.op.inputs) == 1, 'Op {} must only have one input'.format(
 161       tensor.op.name)
 162   source_tensor = tensor.op.inputs[0]
 163 
 164   # Build the side effect graphs and add their outputs to the list of control
 165   # dependencies for the subscribed tensor.
 166   outs = []
 167   name_scope = source_tensor.op.name + '/subscription/'
 168   with ops.name_scope(name_scope):
 169     for s in side_effects:
 170       outs += s(source_tensor)
 171 
 172   out_ops = [out.op if isinstance(out, ops.Tensor) else out for out in outs]
 173   tensor.op._add_control_inputs(out_ops)  # pylint: disable=protected-access
 174 
 175   return tensor

 178 def _is_subscribed_identity(tensor):
 179   """Checks if the given tensor is an identity op returned by `subscribe()`.
 180 
 181   Args:
 182     tensor: A `tf.Tensor` to check.
 183   Returns:
 184     True if the given tensor matches the criteria for subscription identies:
 185     its op type is `Identity`, its name matches the name of its input and
 186     conforms to the convention for subscribed nodes.
 187     False otherwise.
 188   """
 189   # Subscribed tensor are assumed to be identity ops.
 190   if tensor.op.type != 'Identity':
 191     return False
 192 
 193   # Check that the tensor name matches the convention in place for identity ops
 194   # created by subscribe().
 195   match = re.match(
 196       r'(?P<prefix_name>^.*?)/subscription/Identity[^/]+', tensor.name)
 197   if match is None or len(match.groups()) != 1:
 198     return False
 199   prefix_name = match.group('prefix_name')
 200 
 201   # Get a reference to the source tensor and check that it has a matching name.
 202   assert len(tensor.op.inputs) == 1, 'Op {} must only have one input'.format(
 203       tensor.op.name)
 204   source_tensor = tensor.op.inputs[0]
 205   if prefix_name != source_tensor.op.name:
 206     return False
 207 
 208   return True

 211 def _subscribe(tensor, side_effects, control_cache):
 212   """Helper method that subscribes a single tensor to a list of side_effects.
 213 
 214   This method will check if the given tensor has already been subscribed or if
 215   it's a tensor returned by a previous call to `subscribe()` and, if so, will
 216   reuse the existing identity op, appending the given side effects to the list
 217   of existing ones.
 218 
 219   Args:
 220     tensor: The `tf.Tensor` to be subscribed.
 221     side_effects: List of side_effect functions, see subscribe for details.
 222     control_cache: `_ControlOutputCache` helper to get control_outputs faster.
 223   Returns:
 224     The modified replacement to the passed in tensor which triggers the side
 225     effects or the given tensor, if it was already been subscribed.
 226   """
 227   # Check if the given tensor has a numpy compatible type (see dtypes.py).
 228   # If not, we cannot subscribe it, so we just return the original tensor.
 229   if not tensor.dtype.is_numpy_compatible:
 230     logging.debug(('Tensor {} has an un-supported {} type and cannot be '
 231                    'subscribed.').format(tensor.name, tensor.dtype))
 232     return tensor
 233 
 234   if _is_subscribed_identity(tensor):
 235     return _subscribe_extend(tensor, side_effects)
 236 
 237   # Check if the given tensor has already been subscribed by inspecting its
 238   # outputs.
 239   name_scope = tensor.op.name + '/subscription/Identity'
 240   consumers = tensor.consumers()
 241   matching_ops = [op for op in consumers if op.name.startswith(name_scope)]
 242   assert len(matching_ops) <= 1, ('Op {} must only have one subscription '
 243                                   'op connected to it').format(tensor.op.name)
 244   if len(matching_ops) == 1:
 245     candidate_tensor = matching_ops[0].outputs[0]
 246     if _is_subscribed_identity(candidate_tensor):
 247       return _subscribe_extend(candidate_tensor, side_effects)
 248 
 249   return _subscribe_new(tensor, side_effects, control_cache)

 252 @contextlib.contextmanager
 253 def _preserve_control_flow_context(tensor):
 254   """Preserve the control flow context for the given tensor.
 255 
 256   Sets the graph context to the tensor's context so that side effect ops are
 257   added under the same context.
 258 
 259   This is needed when subscribing to tensors defined within a conditional
 260   block or a while loop. In these cases we need that the side-effect ops
 261   are created within the same control flow context as that of the tensor
 262   they are attached to.
 263 
 264   Args:
 265     tensor: tensor whose context should be preserved.
 266 
 267   Yields:
 268     None
 269   """
 270 
 271   # pylint: disable=protected-access
 272   context = tensor.op._get_control_flow_context()
 273   # pylint: enable=protected-access
 274   if context:
 275     context.Enter()
 276   try:
 277     yield
 278   finally:
 279     if context:
 280       context.Exit()

 283 def _scoped_subscribe(tensor, side_effects, control_cache):
 284   """Helper method that subscribes a single tensor to a list of side_effects.
 285 
 286   This is a thin wrapper around `_subscribe` and ensures that the side effect
 287   ops are added within the same device and control flow context of the
 288   subscribed tensor.
 289 
 290   Args:
 291     tensor: The `tf.Tensor` to be subscribed.
 292     side_effects: List of side_effect functions, see subscribe for details.
 293     control_cache: `_ControlOutputCache` helper to get control_outputs faster.
 294   Returns:
 295     The modified replacement to the passed in tensor which triggers the side
 296     effects or the given tensor, if it was already been subscribed.
 297   """
 298 
 299   with ops.device(tensor.device):
 300     with _preserve_control_flow_context(tensor):
 301       return _subscribe(tensor, side_effects, control_cache)

 304 def subscribe(tensors, side_effects):
 305   """Subscribe to a tensor.
 306 
 307   This method will attach side effect graphs to a given set
 308   of tensors. Set of tensors follows from session.run and supports
 309   single `Tensor`, `list`, nested `list`, `tuple`, `namedtuple`, or `dict`. It
 310   returns the tensors in the same passed in structure, but as clones with
 311   side effects applied. The supplied side effect graphs are specified
 312   as a constructor function which takes the target tensor and
 313   constructs a side effect graph and returns a list of ops that should
 314   be control dependencies on fetching the tensor. It will append
 315   'subscription' to the name scope of the tensor for every node in
 316   the side effect graph. These control dependencies are what trigger
 317   the side effects. Subscribe will construct the additions to your
 318   graph and return the created identity tensor downstream of the control
 319   dependencies. Use these tensors as you would normally in the rest of
 320   your tensorflow code. If a given tensor has already been subscribed or a
 321   tensor returned by a call to subscribe is passed, the previously created
 322   identity tensor will be reused and the side effect graphs will be added to
 323   the existing ones.
 324 
 325   Args:
 326     tensors: `Tensor` or set of tensors to subscribe to. Set of tensors format
 327       follows from `Session.run` and supports single `Tensor`, `list`, nested
 328       `list`, `tuple`, `namedtuple`, or `dict`.
 329     side_effects: Function(s) that takes a `Tensor`, construct a subgraph, and
 330       return a nonempty list of control dependencies. This can be a single
 331       function or list of functions.
 332   Returns:
 333     Subscribed tensors, which are identity copies of the passed in tensors
 334       in the same passed in structure, but the graph has been modified
 335       such that these are downstream of the control dependencies for
 336       the side effect graphs. Use these functionally equivalent tensors
 337       instead of the passed in tensors for further construction or running.
 338   """
 339   if not hasattr(side_effects, '__iter__'):
 340     side_effects = [side_effects]
 341 
 342   control_outputs = _ControlOutputCache()
 343   result = _recursive_apply(
 344       tensors, lambda t: _scoped_subscribe(t, side_effects, control_outputs))
 345   return result
