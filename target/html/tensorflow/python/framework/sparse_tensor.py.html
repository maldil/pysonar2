  36 @tf_export("sparse.SparseTensor", "SparseTensor")
  37 class SparseTensor(_TensorLike):
  38   """Represents a sparse tensor.
  39 
  40   TensorFlow represents a sparse tensor as three separate dense tensors:
  41   `indices`, `values`, and `dense_shape`.  In Python, the three tensors are
  42   collected into a `SparseTensor` class for ease of use.  If you have separate
  43   `indices`, `values`, and `dense_shape` tensors, wrap them in a `SparseTensor`
  44   object before passing to the ops below.
  45 
  46   Concretely, the sparse tensor `SparseTensor(indices, values, dense_shape)`
  47   comprises the following components, where `N` and `ndims` are the number
  48   of values and number of dimensions in the `SparseTensor`, respectively:
  49 
  50   * `indices`: A 2-D int64 tensor of dense_shape `[N, ndims]`, which specifies
  51     the indices of the elements in the sparse tensor that contain nonzero
  52     values (elements are zero-indexed). For example, `indices=[[1,3], [2,4]]`
  53     specifies that the elements with indexes of [1,3] and [2,4] have
  54     nonzero values.
  55 
  56   * `values`: A 1-D tensor of any type and dense_shape `[N]`, which supplies the
  57     values for each element in `indices`. For example, given
  58     `indices=[[1,3], [2,4]]`, the parameter `values=[18, 3.6]` specifies
  59     that element [1,3] of the sparse tensor has a value of 18, and element
  60     [2,4] of the tensor has a value of 3.6.
  61 
  62   * `dense_shape`: A 1-D int64 tensor of dense_shape `[ndims]`, which specifies
  63     the dense_shape of the sparse tensor. Takes a list indicating the number of
  64     elements in each dimension. For example, `dense_shape=[3,6]` specifies a
  65     two-dimensional 3x6 tensor, `dense_shape=[2,3,4]` specifies a
  66     three-dimensional 2x3x4 tensor, and `dense_shape=[9]` specifies a
  67     one-dimensional tensor with 9 elements.
  68 
  69   The corresponding dense tensor satisfies:
  70 
  71   ```python
  72   dense.shape = dense_shape
  73   dense[tuple(indices[i])] = values[i]
  74   ```
  75 
  76   By convention, `indices` should be sorted in row-major order (or equivalently
  77   lexicographic order on the tuples `indices[i]`). This is not enforced when
  78   `SparseTensor` objects are constructed, but most ops assume correct ordering.
  79   If the ordering of sparse tensor `st` is wrong, a fixed version can be
  80   obtained by calling `tf.sparse_reorder(st)`.
  81 
  82   Example: The sparse tensor
  83 
  84   ```python
  85   SparseTensor(indices=[[0, 0], [1, 2]], values=[1, 2], dense_shape=[3, 4])
  86   ```
  87 
  88   represents the dense tensor
  89 
  90   ```python
  91   [[1, 0, 0, 0]
  92    [0, 0, 2, 0]
  93    [0, 0, 0, 0]]
  94   ```
  95   """
  96 
  97   @classmethod
  98   def from_value(cls, sparse_tensor_value):
  99     if not is_sparse(sparse_tensor_value):
 100       raise TypeError("Neither a SparseTensor nor SparseTensorValue: %s." %
 101                       sparse_tensor_value)
 102     return SparseTensor(
 103         indices=sparse_tensor_value.indices,
 104         values=sparse_tensor_value.values,
 105         dense_shape=sparse_tensor_value.dense_shape)
 106 
 107   def __init__(self, indices, values, dense_shape):
 108     """Creates a `SparseTensor`.
 109 
 110     Args:
 111       indices: A 2-D int64 tensor of shape `[N, ndims]`.
 112       values: A 1-D tensor of any type and shape `[N]`.
 113       dense_shape: A 1-D int64 tensor of shape `[ndims]`.
 114 
 115     """
 116     with ops.name_scope(None, "SparseTensor",
 117                         [indices, values, dense_shape]):
 118       indices = ops.convert_to_tensor(
 119           indices, name="indices", dtype=dtypes.int64)
 120       # Always pass as_ref=True because we want to be able to update
 121       # values later if it is a VariableOp.
 122       # TODO(touts): Consider adding mutable_values() when 'values'
 123       # is a VariableOp and updating users of SparseTensor.
 124       values = ops.internal_convert_to_tensor(
 125           values, name="values", as_ref=True)
 126       dense_shape = ops.convert_to_tensor(
 127           dense_shape, name="dense_shape", dtype=dtypes.int64)
 128     self._indices = indices
 129     self._values = values
 130     self._dense_shape = dense_shape
 131 
 132     indices_shape = indices.get_shape().with_rank(2)
 133     values_shape = values.get_shape().with_rank(1)
 134     dense_shape_shape = dense_shape.get_shape().with_rank(1)
 135 
 136     # Assert number of rows in indices match the number of elements in values.
 137     indices_shape[0].merge_with(values_shape[0])
 138     # Assert number of columns in indices matches the number of elements in
 139     # dense_shape.
 140     indices_shape[1].merge_with(dense_shape_shape[0])
 141 
 142   def get_shape(self):
 143     """Get the `TensorShape` representing the shape of the dense tensor.
 144 
 145     Returns:
 146       A `TensorShape` object.
 147     """
 148     return tensor_util.constant_value_as_shape(self._dense_shape)
 149 
 150   @property
 151   def indices(self):
 152     """The indices of non-zero values in the represented dense tensor.
 153 
 154     Returns:
 155       A 2-D Tensor of int64 with dense_shape `[N, ndims]`, where `N` is the
 156         number of non-zero values in the tensor, and `ndims` is the rank.
 157     """
 158     return self._indices
 159 
 160   @property
 161   def values(self):
 162     """The non-zero values in the represented dense tensor.
 163 
 164     Returns:
 165       A 1-D Tensor of any data type.
 166     """
 167     return self._values
 168 
 169   @property
 170   def op(self):
 171     """The `Operation` that produces `values` as an output."""
 172     return self.values.op
 173 
 174   @property
 175   def dtype(self):
 176     """The `DType` of elements in this tensor."""
 177     return self._values.dtype
 178 
 179   @property
 180   def dense_shape(self):
 181     """A 1-D Tensor of int64 representing the shape of the dense tensor."""
 182     return self._dense_shape
 183 
 184   @property
 185   def shape(self):
 186     """Get the `TensorShape` representing the shape of the dense tensor.
 187 
 188     Returns:
 189       A `TensorShape` object.
 190     """
 191     return tensor_util.constant_value_as_shape(self._dense_shape)
 192 
 193   @property
 194   def graph(self):
 195     """The `Graph` that contains the index, value, and dense_shape tensors."""
 196     return self._indices.graph
 197 
 198   def consumers(self):
 199     """Returns a list of `Operation`s that consume this `SparseTensor`.
 200 
 201     Returns:
 202       A list of `Operation`s.
 203     """
 204     values_consumers = set(self._values.consumers())
 205     indices_consumers = set(self._indices.consumers())
 206     dense_shape_consumers = set(self._dense_shape.consumers())
 207     return list(values_consumers \
 208                 .union(indices_consumers, dense_shape_consumers))
 209 
 210   def __str__(self):
 211     return "SparseTensor(indices=%s, values=%s, dense_shape=%s)" % (
 212         self._indices, self._values, self._dense_shape)
 213 
 214   def eval(self, feed_dict=None, session=None):
 215     """Evaluates this sparse tensor in a `Session`.
 216 
 217     Calling this method will execute all preceding operations that
 218     produce the inputs needed for the operation that produces this
 219     tensor.
 220 
 221     *N.B.* Before invoking `SparseTensor.eval()`, its graph must have been
 222     launched in a session, and either a default session must be
 223     available, or `session` must be specified explicitly.
 224 
 225     Args:
 226       feed_dict: A dictionary that maps `Tensor` objects to feed values.
 227         See `tf.Session.run` for a
 228         description of the valid feed values.
 229       session: (Optional.) The `Session` to be used to evaluate this sparse
 230         tensor. If none, the default session will be used.
 231 
 232     Returns:
 233       A `SparseTensorValue` object.
 234     """
 235     indices, values, dense_shape = _eval_using_default_session(
 236         [self.indices, self.values, self.dense_shape], feed_dict, self.graph,
 237         session)
 238     return SparseTensorValue(indices, values, dense_shape)
 239 
 240   @staticmethod
 241   def _override_operator(operator, func):
 242     _override_helper(SparseTensor, operator, func)
 243

 251 @tf_export("convert_to_tensor_or_sparse_tensor")
 252 def convert_to_tensor_or_sparse_tensor(value, dtype=None, name=None):
 253   """Converts value to a `SparseTensor` or `Tensor`.
 254 
 255   Args:
 256     value: A `SparseTensor`, `SparseTensorValue`, or an object whose type has a
 257       registered `Tensor` conversion function.
 258     dtype: Optional element type for the returned tensor. If missing, the
 259       type is inferred from the type of `value`.
 260     name: Optional name to use if a new `Tensor` is created.
 261 
 262   Returns:
 263     A `SparseTensor` or `Tensor` based on `value`.
 264 
 265   Raises:
 266     RuntimeError: If result type is incompatible with `dtype`.
 267   """
 268   if dtype is not None:
 269     dtype = dtypes.as_dtype(dtype)
 270   if isinstance(value, SparseTensorValue):
 271     value = SparseTensor.from_value(value)
 272   if isinstance(value, SparseTensor):
 273     if dtype and not dtype.is_compatible_with(value.dtype):
 274       raise RuntimeError(
 275           "Sparse dtype: requested = %s, actual = %s" % (
 276               dtype.name, value.dtype.name))
 277     return value
 278   return ops.internal_convert_to_tensor(
 279       value, dtype=dtype, name=name)

 282 def is_sparse(x):
 283   """Check whether `x` is sparse.
 284 
 285   Check whether an object is a `tf.SparseTensor` or `tf.SparseTensorValue`.
 286 
 287   Args:
 288     x: A python object to check.
 289 
 290   Returns:
 291     `True` iff `x` is a `tf.SparseTensor` or `tf.SparseTensorValue`.
 292   """
 293   return isinstance(x, (SparseTensor, SparseTensorValue))
