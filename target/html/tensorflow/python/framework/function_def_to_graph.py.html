  36 def function_def_to_graph(fdef, input_shapes=None):
  37   """Converts a FunctionDef to a function.FuncGraph (sub-class Graph).
  38 
  39   The returned FuncGraph's `name`, `inputs` and `outputs` fields will be set.
  40   The input tensors are represented as placeholders.
  41 
  42   Note: `FuncGraph.inputs` and `FuncGraph.captures` are not set and may be set
  43   by the caller.
  44 
  45   Args:
  46     fdef: FunctionDef.
  47     input_shapes: Optional. A list of TensorShape objects of the shapes of
  48       function inputs. If specified, its length must match length of
  49       `fdef.signature.input_arg`. If a shape is None, the corresponding input
  50       placeholder will have unknown shape.
  51 
  52   Returns:
  53     A FuncGraph.
  54   """
  55   func_graph = function.FuncGraph(fdef.signature.name)
  56   graph_def, nested_to_flat_tensor_name = function_def_to_graph_def(
  57       fdef, input_shapes)
  58 
  59   with func_graph.as_default():
  60     # Add all function nodes to the graph.
  61     importer.import_graph_def(graph_def, name="")
  62 
  63     # Initialize fields specific to FuncGraph.
  64 
  65     # inputs
  66     input_tensor_names = [
  67         nested_to_flat_tensor_name[arg.name] for arg in fdef.signature.input_arg
  68     ]
  69     func_graph.inputs = [
  70         func_graph.get_tensor_by_name(name) for name in input_tensor_names
  71     ]
  72 
  73     # outputs
  74     output_tensor_names = [
  75         nested_to_flat_tensor_name[fdef.ret[arg.name]]
  76         for arg in fdef.signature.output_arg
  77     ]
  78     func_graph.outputs = [
  79         func_graph.get_tensor_by_name(name) for name in output_tensor_names
  80     ]
  81 
  82   return func_graph

  85 def function_def_to_graph_def(fdef, input_shapes=None):
  86   """Convert a FunctionDef to a GraphDef.
  87 
  88   Steps:
  89   1. Creates placeholder nodes corresponding to inputs in
  90      `FunctionDef.signature.input_arg`.
  91   2. Adds NodeDefs in `FunctionDef.node_def` to `GraphDef.node`.
  92   3. Renames inputs of all nodes to use the convention of GraphDef instead of
  93      FunctionDef. See comment on `FunctionDef.node_def` on how the tensor naming
  94      in FunctionDefs is different from GraphDefs.
  95 
  96   Args:
  97     fdef: FunctionDef.
  98     input_shapes: Optional. A list of TensorShape objects of the shapes of
  99       function inputs. If specified, its length must match length of
 100       `fdef.signature.input_arg`. If a shape is None, the corresponding input
 101       placeholder will have unknown shape.
 102 
 103   Returns:
 104     A tuple of (GraphDef, dict<string, string>). The dict contains a mapping
 105     from nested tensor names (in FunctionDef) to flattened names (in GraphDef).
 106 
 107   Raises:
 108     ValueError: If the length of input_shapes does not match the number of
 109       input_args or if the FunctionDef is invalid.
 110   """
 111   graph_def = graph_pb2.GraphDef()
 112   graph_def.versions.CopyFrom(
 113       versions_pb2.VersionDef(
 114           producer=versions.GRAPH_DEF_VERSION,
 115           min_consumer=versions.GRAPH_DEF_VERSION_MIN_CONSUMER))
 116 
 117   # Copy *all* functions from outer graph to `graph_def` so that both direct
 118   # and indirect references are safely handled.
 119   ops.get_default_graph()._copy_functions_to_graph_def(graph_def, 0)  # pylint: disable=protected-access
 120 
 121   if input_shapes and len(input_shapes) != len(fdef.signature.input_arg):
 122     raise ValueError("Length of input_shapes must match the number of " +
 123                      "input_args. len(input_shapes): {} len(input_arg): {}".
 124                      format(len(input_shapes), len(fdef.signature.input_arg)))
 125 
 126   # 1. Create placeholders for input nodes.
 127   for i, arg_def in enumerate(fdef.signature.input_arg):
 128     node_def = graph_def.node.add()
 129     node_def.name = arg_def.name
 130     node_def.op = "Placeholder"
 131     node_def.attr["dtype"].type = arg_def.type
 132     if input_shapes and input_shapes[i] is not None:
 133       node_def.attr["shape"].shape.CopyFrom(input_shapes[i].as_proto())
 134 
 135   # 2. Copy all body NodeDefs to the GraphDef.
 136   graph_def.node.extend(fdef.node_def)
 137 
 138   # 3. Perform the renaming.
 139 
 140   # Build the tensor name mapping then flatten the tensor names.
 141   # See comment on `FunctionDef.node_def` on how the tensor naming in
 142   # FunctionDefs is different from GraphDefs.
 143   nested_to_flat_tensor_name = {}
 144 
 145   for arg_def in fdef.signature.input_arg:
 146     nested_to_flat_tensor_name[arg_def.name] = "{}:0".format(arg_def.name)
 147     control_name = "^" + arg_def.name
 148     nested_to_flat_tensor_name[control_name] = control_name
 149 
 150   for node_def in fdef.node_def:
 151     op_def = ops.get_default_graph()._get_op_def(node_def.op)  # pylint: disable=protected-access
 152 
 153     for attr in op_def.attr:
 154       if attr.type == "func":
 155         fname = node_def.attr[attr.name].func.name
 156         if not ops.get_default_graph()._is_function(fname):  # pylint: disable=protected-access
 157           raise ValueError("%s function not found." % fname)
 158       elif attr.type == "list(func)":
 159         for fn in node_def.attr[attr.name].list.func:
 160           fname = fn.name
 161           if not ops.get_default_graph()._is_function(fname):  # pylint: disable=protected-access
 162             raise ValueError("%s function not found." % fname)
 163 
 164     # Iterate over output_args in op_def to build the map.
 165     # Index of the output tensor in the flattened list of *all* output
 166     # tensors of the op.
 167     flattened_index = 0
 168     for arg_def in op_def.output_arg:
 169       num_args = _get_num_args(arg_def, node_def)
 170       for i in range(num_args):
 171         # Map tensor names from "node_name:output_arg_name:index" to
 172         # "node_name:flattened_index".
 173         nested_name = "{}:{}:{}".format(node_def.name, arg_def.name, i)
 174         flat_name = "{}:{}".format(node_def.name, flattened_index)
 175         nested_to_flat_tensor_name[nested_name] = flat_name
 176         flattened_index += 1
 177       control_name = "^" + node_def.name
 178       nested_to_flat_tensor_name[control_name] = control_name
 179 
 180   # Update inputs of all nodes in graph.
 181   for node_def in graph_def.node:
 182     for i in range(len(node_def.input)):
 183       node_def.input[i] = nested_to_flat_tensor_name[node_def.input[i]]
 184 
 185   return graph_def, nested_to_flat_tensor_name

 188 # Based on implementation in core/framework/node_def_util.cc::ComputeArgRange.
 189 def _get_num_args(arg_def, node_def):
 190   if arg_def.number_attr:
 191     return node_def.attr[arg_def.number_attr].i
 192   elif arg_def.type_list_attr:
 193     return len(node_def.attr[arg_def.type_list_attr].list.type)
 194   elif arg_def.type_attr or arg_def.type != types_pb2.DT_INVALID:
 195     return 1
 196   else:
 197     raise ValueError("Invalid arg_def:\n\n{}".format(str(arg_def)))
