  55 def _node_def(from_node_def, export_scope, unbound_inputs, clear_devices=False):
  56   """Create a `NodeDef` proto with export_scope stripped.
  57 
  58   Args:
  59     from_node_def: A `node_def_pb2.NodeDef` protocol buffer.
  60     export_scope: A `string` representing the name scope to remove.
  61     unbound_inputs: An array of unbound input names if they exist.
  62     clear_devices: Boolean which controls whether to clear device information
  63       from node_def. Default false.
  64 
  65   Returns:
  66     A `node_def_pb2.NodeDef` protocol buffer.
  67   """
  68   node_def = copy.deepcopy(from_node_def)
  69   for i, v in enumerate(node_def.input):
  70     if (export_scope and
  71         not node_def.input[i].lstrip("^").startswith(export_scope)):
  72       # Adds "$unbound_inputs_" prefix to the unbound name so they are easily
  73       # identifiable.
  74       node_def.input[i] = re.sub(r"([\^]|^)(.*)",
  75                                  r"\1" + _UNBOUND_INPUT_PREFIX + r"\2",
  76                                  compat.as_str(v))
  77       unbound_inputs.append(node_def.input[i])
  78     else:
  79       node_def.input[i] = ops.strip_name_scope(v, export_scope)
  80   node_def.name = compat.as_bytes(
  81       ops.strip_name_scope(from_node_def.name, export_scope))
  82   for k, v in six.iteritems(from_node_def.attr):
  83     if k == "_class":
  84       new_s = [compat.as_bytes(
  85           ops.strip_name_scope(s, export_scope)) for s in v.list.s
  86                if not export_scope or
  87                compat.as_str(s).split("@")[1].startswith(export_scope)]
  88       node_def.attr[k].CopyFrom(attr_value_pb2.AttrValue(
  89           list=attr_value_pb2.AttrValue.ListValue(s=new_s)))
  90     elif node_def.op in ("Enter", "RefEnter") and k == "frame_name":
  91       if not export_scope or compat.as_str(v.s).startswith(export_scope):
  92         new_s = compat.as_bytes(ops.strip_name_scope(v.s, export_scope))
  93       node_def.attr[k].CopyFrom(attr_value_pb2.AttrValue(s=new_s))
  94     else:
  95       node_def.attr[k].CopyFrom(v)
  96 
  97   if clear_devices:
  98     node_def.device = ""
  99 
 100   return node_def

 103 def _read_file(filename):
 104   """Reads a file containing `GraphDef` and returns the protocol buffer.
 105 
 106   Args:
 107     filename: `graph_def` filename including the path.
 108 
 109   Returns:
 110     A `GraphDef` protocol buffer.
 111 
 112   Raises:
 113     IOError: If the file doesn't exist, or cannot be successfully parsed.
 114   """
 115   graph_def = graph_pb2.GraphDef()
 116   if not file_io.file_exists(filename):
 117     raise IOError("File %s does not exist." % filename)
 118   # First try to read it as a binary file.
 119   file_content = file_io.FileIO(filename, "rb").read()
 120   try:
 121     graph_def.ParseFromString(file_content)
 122     return graph_def
 123   except Exception:  # pylint: disable=broad-except
 124     pass
 125 
 126   # Next try to read it as a text file.
 127   try:
 128     text_format.Merge(file_content, graph_def)
 129   except text_format.ParseError as e:
 130     raise IOError("Cannot parse file %s: %s." % (filename, str(e)))
 131 
 132   return graph_def

 135 def ops_used_by_graph_def(graph_def):
 136   """Collect the list of ops used by a graph.
 137 
 138   Does not validate that the ops are all registered.
 139 
 140   Args:
 141     graph_def: A `GraphDef` proto, as from `graph.as_graph_def()`.
 142 
 143   Returns:
 144     A list of strings, each naming an op used by the graph.
 145   """
 146   # Map function names to definitions
 147   name_to_function = {}
 148   for fun in graph_def.library.function:
 149     name_to_function[fun.signature.name] = fun
 150 
 151   # Collect the list of op names.  Since functions can reference functions, we
 152   # need a recursive traversal.
 153   used_ops = set()  # Includes both primitive ops and functions
 154   functions_to_process = []  # A subset of used_ops
 155 
 156   def mark_op_as_used(op):
 157     if op not in used_ops and op in name_to_function:
 158       functions_to_process.append(name_to_function[op])
 159     used_ops.add(op)
 160 
 161   for node in graph_def.node:
 162     mark_op_as_used(node.op)
 163   while functions_to_process:
 164     fun = functions_to_process.pop()
 165     for node in fun.node_def:
 166       mark_op_as_used(node.op)
 167 
 168   return [op for op in used_ops if op not in name_to_function]

 171 def stripped_op_list_for_graph(graph_def):
 172   """Collect the stripped OpDefs for ops used by a graph.
 173 
 174   This function computes the `stripped_op_list` field of `MetaGraphDef` and
 175   similar protos.  The result can be communicated from the producer to the
 176   consumer, which can then use the C++ function
 177   `RemoveNewDefaultAttrsFromGraphDef` to improve forwards compatibility.
 178 
 179   Args:
 180     graph_def: A `GraphDef` proto, as from `graph.as_graph_def()`.
 181 
 182   Returns:
 183     An `OpList` of ops used by the graph.
 184 
 185   Raises:
 186     ValueError: If an unregistered op is used.
 187   """
 188   # This is the Python equivalent of StrippedOpListForGraph in C++.
 189   # Unfortunately, since the Python op registry can differ from that in C++, we
 190   # can't remove the duplication using swig (at least naively).
 191   # TODO(irving): Support taking graphs directly.
 192 
 193   used_ops = ops_used_by_graph_def(graph_def)
 194 
 195   # Verify that all used ops are registered.
 196   registered_ops = op_def_registry.get_registered_ops()
 197   # These internal ops used by functions are not registered, so we need to
 198   # whitelist them.  # TODO(irving): Do something better here.
 199   op_whitelist = ("_Arg", "_Retval", "_ListToArray", "_ArrayToList")
 200   for op in used_ops:
 201     if op not in registered_ops and op not in op_whitelist:
 202       raise ValueError("Op %s is used by the graph, but is not registered" % op)
 203 
 204   # Build the stripped op list in sorted order
 205   return op_def_pb2.OpList(op=[registered_ops[op] for op in sorted(used_ops)
 206                                if op in registered_ops])

 209 def _get_kind_name(item):
 210   """Returns the kind name in CollectionDef.
 211 
 212   Args:
 213     item: A data item.
 214 
 215   Returns:
 216     The string representation of the kind in CollectionDef.
 217   """
 218   if isinstance(item, (six.string_types, six.binary_type)):
 219     kind = "bytes_list"
 220   elif isinstance(item, six.integer_types):
 221     kind = "int64_list"
 222   elif isinstance(item, float):
 223     kind = "float_list"
 224   elif isinstance(item, Any):
 225     kind = "any_list"
 226   else:
 227     kind = "node_list"
 228   return kind

 239 def _op_name(tensor_name):
 240   """Extract the Op name from a Tensor name.
 241 
 242   The Op name is everything before a colon, if present,
 243   not including any ^ prefix denoting a control dependency.
 244 
 245   Args:
 246     tensor_name: the full name of a Tensor in the graph.
 247   Returns:
 248     The name of the Op of which the given Tensor is an output.
 249   Raises:
 250     ValueError: if tensor_name is None or empty.
 251   """
 252   if not tensor_name:
 253     raise ValueError("Tensor name cannot be empty or None.")
 254 
 255   # Control dependency inputs start with ^.
 256   if tensor_name.startswith("^"):
 257     tensor_name = tensor_name[1:]
 258   if ":" in tensor_name:
 259     op_name, _ = tensor_name.split(":")
 260     return op_name
 261   return tensor_name

 264 def _get_scope(node_name):
 265   """Extract the scope name from a node name.
 266 
 267   The scope name is everything before the final slash,
 268   not including any ^ prefix denoting a control dependency.
 269 
 270   Args:
 271     node_name: the full name of an Op or a Tensor in the graph.
 272   Returns:
 273     The deepest named scope containing the node.
 274   Raises:
 275     ValueError: if tensor_name is None or empty
 276   """
 277   if not node_name:
 278     raise ValueError("Node name cannot be empty or None.")
 279 
 280   # Control dependency inputs start with ^.
 281   if node_name.startswith("^"):
 282     node_name = node_name[1:]
 283   if "/" in node_name:
 284     scope, _ = node_name.rsplit("/", 1)
 285     return scope
 286 
 287   return ""

 290 def _find_extraneous_saver_nodes(graph_def, saver_def):
 291   """Identifies any nodes in the graph_def related to unused Savers.
 292 
 293   This approach assumes that each Saver is cleanly isolated in its own name
 294   scope, so we need only identify the scopes associated with extraneous Savers
 295   and return all the nodes in those scopes.
 296 
 297   Args:
 298     graph_def: a GraphDef proto to evaluate.
 299     saver_def: a SaverDef proto referencing Save/Restore ops to be retained.
 300   Returns:
 301     An iterable of node names that may be safely omitted.
 302   """
 303   # TODO(soergel): confirm that the assumption of scope isolation is valid.
 304   # If not, we need to walk up the graph from any restore_all nodes, and walk
 305   # down the graph from any Save/Restore nodes.  I drafted that approach too,
 306   # but it seems unnecessarily complex given the name scope solution.
 307 
 308   # load the graph DAG in minimal form, without initializing a full Graph object
 309   nodes = {node_def.name:
 310            (set([_op_name(x) for x in node_def.input]), node_def.op)
 311            for node_def in graph_def.node}
 312 
 313   retain_scope_save = None
 314   retain_scope_restore = None
 315   # It's possible to have no saver if the graph has no Variables
 316   if saver_def is not None:
 317     save_op_name = _op_name(saver_def.save_tensor_name)
 318     restore_op_name = _op_name(saver_def.restore_op_name)
 319 
 320     # The save and restore scopes should always be the same, but if they differ
 321     # for some reason, we retain them both to be safe.
 322     retain_scope_restore = _get_scope(restore_op_name) + "/"
 323     retain_scope_save = _get_scope(save_op_name) + "/"
 324 
 325   all_saver_node_names = set([name for name, (_, op) in nodes.items()
 326                               if op in SAVE_AND_RESTORE_OPS])
 327 
 328   all_saver_scopes = (set([_get_scope(x) for x in all_saver_node_names])
 329                       - all_saver_node_names)
 330   all_saver_scopes = set([x + "/" for x in all_saver_scopes])
 331 
 332   extraneous_scopes = all_saver_scopes - set([retain_scope_save,
 333                                               retain_scope_restore])
 334 
 335   extraneous_node_names = set()
 336   for name, _ in nodes.items():
 337     for extraneous_scope in extraneous_scopes:
 338       if name.startswith(extraneous_scope):
 339         extraneous_node_names.add(name)
 340         break
 341 
 342   return extraneous_node_names

 345 def _should_include_node(node_or_node_name, export_scope, exclude_nodes):
 346   """Returns `True` if a node should be included.
 347 
 348   Args:
 349     node_or_node_name: A node or `string` node name.
 350     export_scope: `string`. Name scope under which to extract the subgraph. The
 351       scope name will be stripped from the node definitions for easy import
 352       later into new name scopes.
 353     exclude_nodes: An iterable of nodes or `string` node names to omit from the
 354       export, or None.  Note no sanity-checking is done, so this list must be
 355       carefully constructed to avoid producing an invalid graph.
 356 
 357   Returns:
 358     `True` if the node should be included.
 359   """
 360   if not isinstance(node_or_node_name, six.string_types):
 361     try:
 362       node_name = node_or_node_name.name
 363     except AttributeError:
 364       # Keep the object that we don't know how to process.
 365       return True
 366   else:
 367     node_name = node_or_node_name
 368 
 369   if exclude_nodes and (node_or_node_name in exclude_nodes
 370                         or node_name in exclude_nodes):
 371     return False
 372 
 373   return (node_name.startswith(_UNBOUND_INPUT_PREFIX) or
 374           (not export_scope or node_name.startswith(export_scope)))

 377 def add_collection_def(meta_graph_def, key, graph=None,
 378                        export_scope=None, exclude_nodes=None,
 379                        override_contents=None):
 380   """Adds a collection to MetaGraphDef protocol buffer.
 381 
 382   Args:
 383     meta_graph_def: MetaGraphDef protocol buffer.
 384     key: One of the GraphKeys or user-defined string.
 385     graph: The `Graph` from which to get collections.
 386     export_scope: Optional `string`. Name scope to remove.
 387     exclude_nodes: An iterable of nodes or `string` node names to omit from the
 388       collection, or None.
 389     override_contents: An iterable of values to place in the collection,
 390       ignoring the current values (if set).
 391   """
 392   if graph and not isinstance(graph, ops.Graph):
 393     raise TypeError("graph must be of type Graph, not %s", type(graph))
 394 
 395   if not isinstance(key, six.string_types) and not isinstance(key, bytes):
 396     logging.warning("Only collections with string type keys will be "
 397                     "serialized. This key has %s", type(key))
 398     return
 399 
 400   # Sets graph to default graph if it's not passed in.
 401   graph = graph or ops.get_default_graph()
 402 
 403   if override_contents:
 404     collection_list = override_contents
 405   else:
 406     collection_list = graph.get_collection(key)
 407 
 408   # Remove nodes that should not be exported from the collection list.
 409   collection_list = [x for x in collection_list if
 410                      _should_include_node(x, export_scope, exclude_nodes)]
 411   if not collection_list:
 412     return
 413 
 414   try:
 415     col_def = meta_graph_def.collection_def[key]
 416     to_proto = ops.get_to_proto_function(key)
 417     proto_type = ops.get_collection_proto_type(key)
 418     if to_proto:
 419       kind = "bytes_list"
 420       for x in collection_list:
 421         # Additional type check to make sure the returned proto is indeed
 422         # what we expect.
 423         proto = to_proto(x, export_scope=export_scope)
 424         if proto:
 425           assert isinstance(proto, proto_type)
 426           getattr(col_def, kind).value.append(proto.SerializeToString())
 427     else:
 428       kind = _get_kind_name(collection_list[0])
 429       if kind == "node_list":
 430         for x in collection_list:
 431           if not export_scope or x.name.startswith(export_scope):
 432             getattr(col_def, kind).value.append(
 433                 ops.strip_name_scope(x.name, export_scope))
 434       elif kind == "bytes_list":
 435         # NOTE(opensource): This force conversion is to work around the fact
 436         # that Python3 distinguishes between bytes and strings.
 437         getattr(col_def, kind).value.extend(
 438             [compat.as_bytes(x) for x in collection_list])
 439       else:
 440         getattr(col_def, kind).value.extend([x for x in collection_list])
 441   except Exception as e:  # pylint: disable=broad-except
 442     logging.warning("Issue encountered when serializing %s.\n"
 443                     "Type is unsupported, or the types of the items don't "
 444                     "match field type in CollectionDef. Note this is a warning "
 445                     "and probably safe to ignore.\n%s", key, str(e))
 446     if key in meta_graph_def.collection_def:
 447       del meta_graph_def.collection_def[key]
 448     return

 451 def _is_default_attr_value(op_def, attr_name, attr_value):
 452   """Checks if given attribute matches the default value in the op def."""
 453   for attr_def in op_def.attr:
 454     if attr_def.name == attr_name:
 455       if not attr_def.HasField("default_value"):
 456         return False
 457       # pywrap_tensorflow.EqualAttrValueWrapper returns an empty string
 458       # if both arguments represent an equivalent AttrValue instance.
 459       return not pywrap_tensorflow.EqualAttrValueWrapper(
 460           attr_value.SerializeToString(),
 461           attr_def.default_value.SerializeToString())
 462   return False

 465 def _strip_graph_default_valued_attrs(meta_graph_def):
 466   """Strips default valued attributes for node defs in given MetaGraphDef.
 467 
 468   This method also sets `meta_info_def.stripped_default_attrs` in the given
 469   `MetaGraphDef` proto to True.
 470 
 471   Args:
 472     meta_graph_def: `MetaGraphDef` protocol buffer
 473 
 474   Returns:
 475     None.
 476   """
 477   # Map function op names to their function definitions.
 478   op_name_to_function = {}
 479   for function_def in meta_graph_def.graph_def.library.function:
 480     op_name_to_function[function_def.signature.name] = function_def
 481 
 482   # Get all registered ops.
 483   registered_ops = op_def_registry.get_registered_ops()
 484 
 485   def _strip_node_default_valued_attrs(node_def):
 486     """Removes default valued attributes from a single node def."""
 487     if node_def.op in op_name_to_function or node_def.op not in registered_ops:
 488       return
 489     op_def = registered_ops[node_def.op]
 490 
 491     attrs_to_strip = set()
 492     for attr_name, attr_value in node_def.attr.items():
 493       if _is_default_attr_value(op_def, attr_name, attr_value):
 494         attrs_to_strip.add(attr_name)
 495 
 496     for attr in attrs_to_strip:
 497       del node_def.attr[attr]
 498 
 499   # Process all NodeDef instances in graph_def.
 500   for node_def in meta_graph_def.graph_def.node:
 501     _strip_node_default_valued_attrs(node_def)
 502 
 503   # Process all NodeDef instances in graph_def.library.function.
 504   for function_def in meta_graph_def.graph_def.library.function:
 505     for function_node_def in function_def.node_def:
 506       _strip_node_default_valued_attrs(function_node_def)
 507 
 508   # Tell consumers of this graph that default valued attrs have been stripped.
 509   meta_graph_def.meta_info_def.stripped_default_attrs = True
 510

 512 def create_meta_graph_def(meta_info_def=None,
 513                           graph_def=None,
 514                           saver_def=None,
 515                           collection_list=None,
 516                           graph=None,
 517                           export_scope=None,
 518                           exclude_nodes=None,
 519                           clear_extraneous_savers=False,
 520                           strip_default_attrs=False):
 521   # pylint: disable=line-too-long
 522   """Construct and returns a `MetaGraphDef` protocol buffer.
 523 
 524   Args:
 525     meta_info_def: `MetaInfoDef` protocol buffer.
 526     graph_def: `GraphDef` protocol buffer.
 527     saver_def: `SaverDef` protocol buffer.
 528     collection_list: List of string keys to collect.
 529     graph: The `Graph` to create `MetaGraphDef` out of.
 530     export_scope: Optional `string`. Name scope to remove.
 531     exclude_nodes: An iterable of nodes or `string` node names to omit from all
 532       collection, or None.
 533     clear_extraneous_savers: Remove any preexisting SaverDefs from the SAVERS
 534         collection.  Note this method does not alter the graph, so any
 535         extraneous Save/Restore ops should have been removed already, as needed.
 536     strip_default_attrs: Boolean. If `True`, default-valued attributes will be
 537         removed from the NodeDefs. For a detailed guide, see
 538         [Stripping Default-Valued Attributes](https://github.com/tensorflow/tensorflow/blob/master/tensorflow/python/saved_model/README.md#stripping-default-valued-attributes).
 539 
 540   Returns:
 541     MetaGraphDef protocol buffer.
 542 
 543   Raises:
 544     TypeError: If the arguments are not of the correct proto buffer type.
 545   """
 546   # pylint: enable=line-too-long
 547   # Type check.
 548   if graph and not isinstance(graph, ops.Graph):
 549     raise TypeError("graph must be of type Graph, not %s", type(graph))
 550   if meta_info_def and not isinstance(meta_info_def,
 551                                       meta_graph_pb2.MetaGraphDef.MetaInfoDef):
 552     raise TypeError("meta_info_def must be of type MetaInfoDef, not %s",
 553                     type(meta_info_def))
 554   if graph_def and not isinstance(graph_def, graph_pb2.GraphDef):
 555     raise TypeError("graph_def must be of type GraphDef, not %s",
 556                     type(graph_def))
 557   if saver_def and not isinstance(saver_def, saver_pb2.SaverDef):
 558     raise TypeError("saver_def must be of type SaverDef, not %s",
 559                     type(saver_def))
 560 
 561   # Sets graph to default graph if it's not passed in.
 562   graph = graph or ops.get_default_graph()
 563 
 564   # Creates a MetaGraphDef proto.
 565   meta_graph_def = meta_graph_pb2.MetaGraphDef()
 566   # Adds meta_info_def.
 567   if not meta_info_def:
 568     meta_info_def = meta_graph_pb2.MetaGraphDef.MetaInfoDef()
 569 
 570   # Set the tf version strings to the current tf build.
 571   meta_info_def.tensorflow_version = versions.__version__
 572   meta_info_def.tensorflow_git_version = versions.__git_version__
 573   meta_graph_def.meta_info_def.MergeFrom(meta_info_def)
 574 
 575   # Adds graph_def or the default.
 576   if not graph_def:
 577     meta_graph_def.graph_def.MergeFrom(graph.as_graph_def(add_shapes=True))
 578   else:
 579     meta_graph_def.graph_def.MergeFrom(graph_def)
 580 
 581   # Fills in meta_info_def.stripped_op_list using the ops from graph_def.
 582   # pylint: disable=g-explicit-length-test
 583   if len(meta_graph_def.meta_info_def.stripped_op_list.op) == 0:
 584     meta_graph_def.meta_info_def.stripped_op_list.MergeFrom(
 585         stripped_op_list_for_graph(meta_graph_def.graph_def))
 586   # pylint: enable=g-explicit-length-test
 587 
 588   # Strip default valued attributes in graph_def.
 589   if strip_default_attrs:
 590     _strip_graph_default_valued_attrs(meta_graph_def)
 591 
 592   # Adds saver_def.
 593   if saver_def:
 594     meta_graph_def.saver_def.MergeFrom(saver_def)
 595 
 596   # Adds collection_list.
 597   if collection_list is not None:
 598     clist = collection_list
 599   else:
 600     clist = graph.get_all_collection_keys()
 601 
 602   for ctype in clist:
 603     if clear_extraneous_savers and ctype == ops.GraphKeys.SAVERS:
 604       # Avoid importing Saver here
 605       from_proto = ops.get_from_proto_function(ctype)
 606       add_collection_def(meta_graph_def, ctype,
 607                          graph=graph,
 608                          export_scope=export_scope,
 609                          exclude_nodes=exclude_nodes,
 610                          override_contents=[from_proto(saver_def)])
 611     else:
 612       add_collection_def(meta_graph_def, ctype,
 613                          graph=graph,
 614                          export_scope=export_scope,
 615                          exclude_nodes=exclude_nodes)
 616   return meta_graph_def

 619 def read_meta_graph_file(filename):
 620   """Reads a file containing `MetaGraphDef` and returns the protocol buffer.
 621 
 622   Args:
 623     filename: `meta_graph_def` filename including the path.
 624 
 625   Returns:
 626     A `MetaGraphDef` protocol buffer.
 627 
 628   Raises:
 629     IOError: If the file doesn't exist, or cannot be successfully parsed.
 630   """
 631   meta_graph_def = meta_graph_pb2.MetaGraphDef()
 632   if not file_io.file_exists(filename):
 633     raise IOError("File %s does not exist." % filename)
 634   # First try to read it as a binary file.
 635   file_content = file_io.FileIO(filename, "rb").read()
 636   try:
 637     meta_graph_def.ParseFromString(file_content)
 638     return meta_graph_def
 639   except Exception:  # pylint: disable=broad-except
 640     pass
 641 
 642   # Next try to read it as a text file.
 643   try:
 644     text_format.Merge(file_content.decode("utf-8"), meta_graph_def)
 645   except text_format.ParseError as e:
 646     raise IOError("Cannot parse file %s: %s." % (filename, str(e)))
 647 
 648   return meta_graph_def

 651 def import_scoped_meta_graph(meta_graph_or_file,
 652                              clear_devices=False,
 653                              graph=None,
 654                              import_scope=None,
 655                              input_map=None,
 656                              unbound_inputs_col_name="unbound_inputs",
 657                              restore_collections_predicate=(lambda key: True)):
 658   """Recreates a `Graph` saved in a `MetaGraphDef` proto.
 659 
 660   This function takes a `MetaGraphDef` protocol buffer as input. If
 661   the argument is a file containing a `MetaGraphDef` protocol buffer ,
 662   it constructs a protocol buffer from the file content. The function
 663   then adds all the nodes from the `graph_def` field to the
 664   current graph, recreates the desired collections, and returns a dictionary of
 665   all the Variables imported into the name scope.
 666 
 667   In combination with `export_scoped_meta_graph()`, this function can be used to
 668 
 669   * Serialize a graph along with other Python objects such as `QueueRunner`,
 670     `Variable` into a `MetaGraphDef`.
 671 
 672   * Restart training from a saved graph and checkpoints.
 673 
 674   * Run inference from a saved graph and checkpoints.
 675 
 676   Args:
 677     meta_graph_or_file: `MetaGraphDef` protocol buffer or filename (including
 678       the path) containing a `MetaGraphDef`.
 679     clear_devices: Boolean which controls whether to clear device information
 680       from graph_def. Default false.
 681     graph: The `Graph` to import into. If `None`, use the default graph.
 682     import_scope: Optional `string`. Name scope into which to import the
 683       subgraph. If `None`, the graph is imported to the root name scope.
 684     input_map: A dictionary mapping input names (as strings) in `graph_def` to
 685       `Tensor` objects. The values of the named input tensors in the imported
 686       graph will be re-mapped to the respective `Tensor` values.
 687     unbound_inputs_col_name: Collection name for looking up unbound inputs.
 688     restore_collections_predicate: a predicate on collection names. A collection
 689       named c (i.e whose key is c) will be restored iff
 690       1) `restore_collections_predicate(c)` is True, and
 691       2) `c != unbound_inputs_col_name`.
 692 
 693   Returns:
 694     A dictionary of all the `Variables` imported into the name scope.
 695 
 696   Raises:
 697     ValueError: If the graph_def contains unbound inputs.
 698   """
 699   return import_scoped_meta_graph_with_return_elements(
 700       meta_graph_or_file, clear_devices, graph, import_scope, input_map,
 701       unbound_inputs_col_name, restore_collections_predicate)[0]
 702 
 703 
 704 def import_scoped_meta_graph_with_return_elements(
 705     meta_graph_or_file,
 706     clear_devices=False,
 707     graph=None,
 708     import_scope=None,
 709     input_map=None,
 710     unbound_inputs_col_name="unbound_inputs",
 711     restore_collections_predicate=(lambda key: True),
 712     return_elements=None):
 713   """Imports graph from `MetaGraphDef` and returns vars and return elements.
 714 
 715   This function takes a `MetaGraphDef` protocol buffer as input. If
 716   the argument is a file containing a `MetaGraphDef` protocol buffer ,
 717   it constructs a protocol buffer from the file content. The function
 718   then adds all the nodes from the `graph_def` field to the
 719   current graph, recreates the desired collections, and returns a dictionary of
 720   all the Variables imported into the name scope.
 721 
 722   In combination with `export_scoped_meta_graph()`, this function can be used to
 723 
 724   * Serialize a graph along with other Python objects such as `QueueRunner`,
 725     `Variable` into a `MetaGraphDef`.
 726 
 727   * Restart training from a saved graph and checkpoints.
 728 
 729   * Run inference from a saved graph and checkpoints.
 730 
 731   Args:
 732     meta_graph_or_file: `MetaGraphDef` protocol buffer or filename (including
 733       the path) containing a `MetaGraphDef`.
 734     clear_devices: Boolean which controls whether to clear device information
 735       from graph_def. Default false.
 736     graph: The `Graph` to import into. If `None`, use the default graph.
 737     import_scope: Optional `string`. Name scope into which to import the
 738       subgraph. If `None`, the graph is imported to the root name scope.
 739     input_map: A dictionary mapping input names (as strings) in `graph_def` to
 740       `Tensor` objects. The values of the named input tensors in the imported
 741       graph will be re-mapped to the respective `Tensor` values.
 742     unbound_inputs_col_name: Collection name for looking up unbound inputs.
 743     restore_collections_predicate: a predicate on collection names. A collection
 744       named c (i.e whose key is c) will be restored iff
 745       1) `restore_collections_predicate(c)` is True, and
 746       2) `c != unbound_inputs_col_name`.
 747     return_elements:  A list of strings containing operation names in the
 748       `MetaGraphDef` that will be returned as `Operation` objects; and/or
 749       tensor names in `MetaGraphDef` that will be returned as `Tensor` objects.
 750 
 751   Returns:
 752     A tuple of (
 753       dictionary of all the `Variables` imported into the name scope,
 754       list of `Operation` or `Tensor` objects from the `return_elements` list).
 755 
 756   Raises:
 757     ValueError: If the graph_def contains unbound inputs.
 758 
 759   """
 760   if context.executing_eagerly():
 761     raise ValueError("Exporting/importing meta graphs is not supported when "
 762                      "eager execution is enabled.")
 763   if isinstance(meta_graph_or_file, meta_graph_pb2.MetaGraphDef):
 764     meta_graph_def = meta_graph_or_file
 765   else:
 766     meta_graph_def = read_meta_graph_file(meta_graph_or_file)
 767 
 768   if unbound_inputs_col_name:
 769     for key, col_def in meta_graph_def.collection_def.items():
 770       if key == unbound_inputs_col_name:
 771         kind = col_def.WhichOneof("kind")
 772         field = getattr(col_def, kind)
 773         if field.value and (
 774             not input_map or
 775             sorted([compat.as_str(v) for v in field.value]) !=
 776             sorted(input_map)):
 777           raise ValueError("Graph contains unbound inputs: %s. Must "
 778                            "provide these inputs through input_map." %
 779                            ",".join([compat.as_str(v) for v in field.value
 780                                      if not input_map or v not in input_map]))
 781         break
 782 
 783   # Sets graph to default graph if it's not passed in.
 784   graph = graph or ops.get_default_graph()
 785 
 786   # Gathers the list of nodes we are interested in.
 787   with graph.as_default():
 788     producer_op_list = None
 789     if meta_graph_def.meta_info_def.HasField("stripped_op_list"):
 790       producer_op_list = meta_graph_def.meta_info_def.stripped_op_list
 791     input_graph_def = meta_graph_def.graph_def
 792     # Remove all the explicit device specifications for this node. This helps to
 793     # make the graph more portable.
 794     if clear_devices:
 795       for node in input_graph_def.node:
 796         node.device = ""
 797 
 798     scope_to_prepend_to_names = graph.unique_name(
 799         import_scope or "", mark_as_used=False)
 800 
 801     imported_return_elements = importer.import_graph_def(
 802         input_graph_def,
 803         name=(import_scope or scope_to_prepend_to_names),
 804         input_map=input_map,
 805         producer_op_list=producer_op_list,
 806         return_elements=return_elements)
 807 
 808     # Restores all the other collections.
 809     variable_objects = {}
 810     for key, col_def in sorted(meta_graph_def.collection_def.items()):
 811       # Don't add unbound_inputs to the new graph.
 812       if key == unbound_inputs_col_name:
 813         continue
 814       if not restore_collections_predicate(key):
 815         continue
 816 
 817       kind = col_def.WhichOneof("kind")
 818       if kind is None:
 819         logging.error("Cannot identify data type for collection %s. Skipping.",
 820                       key)
 821         continue
 822       from_proto = ops.get_from_proto_function(key)
 823       if from_proto and kind == "bytes_list":
 824         proto_type = ops.get_collection_proto_type(key)
 825         if key in ops.GraphKeys._VARIABLE_COLLECTIONS:  # pylint: disable=protected-access
 826           for value in col_def.bytes_list.value:
 827             variable = variable_objects.get(value, None)
 828             if variable is None:
 829               proto = proto_type()
 830               proto.ParseFromString(value)
 831               variable = from_proto(
 832                   proto, import_scope=scope_to_prepend_to_names)
 833               variable_objects[value] = variable
 834             graph.add_to_collection(key, variable)
 835         else:
 836           for value in col_def.bytes_list.value:
 837             proto = proto_type()
 838             proto.ParseFromString(value)
 839             graph.add_to_collection(
 840                 key, from_proto(
 841                     proto, import_scope=scope_to_prepend_to_names))
 842       else:
 843         field = getattr(col_def, kind)
 844         if key in _COMPAT_COLLECTION_LIST:
 845           logging.warning(
 846               "The saved meta_graph is possibly from an older release:\n"
 847               "'%s' collection should be of type 'byte_list', but instead "
 848               "is of type '%s'.", key, kind)
 849         if kind == "node_list":
 850           for value in field.value:
 851             col_op = graph.as_graph_element(
 852                 ops.prepend_name_scope(value, scope_to_prepend_to_names))
 853             graph.add_to_collection(key, col_op)
 854         elif kind == "int64_list":
 855           # NOTE(opensource): This force conversion is to work around the fact
 856           # that Python2 distinguishes between int and long, while Python3 has
 857           # only int.
 858           for value in field.value:
 859             graph.add_to_collection(key, int(value))
 860         else:
 861           for value in field.value:
 862             graph.add_to_collection(
 863                 key, ops.prepend_name_scope(value, scope_to_prepend_to_names))
 864 
 865     var_list = {}
 866     variables = graph.get_collection(ops.GraphKeys.GLOBAL_VARIABLES,
 867                                      scope=scope_to_prepend_to_names)
 868     for v in variables:
 869       var_list[ops.strip_name_scope(v.name, scope_to_prepend_to_names)] = v
 870 
 871   return var_list, imported_return_elements
 872

 874 def export_scoped_meta_graph(filename=None,
 875                              graph_def=None,
 876                              graph=None,
 877                              export_scope=None,
 878                              as_text=False,
 879                              unbound_inputs_col_name="unbound_inputs",
 880                              clear_devices=False,
 881                              saver_def=None,
 882                              clear_extraneous_savers=False,
 883                              strip_default_attrs=False,
 884                              **kwargs):
 885   """Returns `MetaGraphDef` proto. Optionally writes it to filename.
 886 
 887   This function exports the graph, saver, and collection objects into
 888   `MetaGraphDef` protocol buffer with the intention of it being imported
 889   at a later time or location to restart training, run inference, or be
 890   a subgraph.
 891 
 892   Args:
 893     filename: Optional filename including the path for writing the
 894       generated `MetaGraphDef` protocol buffer.
 895     graph_def: `GraphDef` protocol buffer.
 896     graph: The `Graph` to export. If `None`, use the default graph.
 897     export_scope: Optional `string`. Name scope under which to extract
 898       the subgraph. The scope name will be stripped from the node definitions
 899       for easy import later into new name scopes. If `None`, the whole graph
 900       is exported.
 901     as_text: If `True`, writes the `MetaGraphDef` as an ASCII proto.
 902     unbound_inputs_col_name: Optional `string`. If provided, a string collection
 903       with the given name will be added to the returned `MetaGraphDef`,
 904       containing the names of tensors that must be remapped when importing the
 905       `MetaGraphDef`.
 906     clear_devices: Boolean which controls whether to clear device information
 907       before exporting the graph.
 908     saver_def: `SaverDef` protocol buffer.
 909     clear_extraneous_savers: Remove any Saver-related information from the
 910         graph (both Save/Restore ops and SaverDefs) that are not associated
 911         with the provided SaverDef.
 912     strip_default_attrs: Set to true if default valued attributes must be
 913         removed while exporting the GraphDef.
 914     **kwargs: Optional keyed arguments, including meta_info_def and
 915         collection_list.
 916 
 917   Returns:
 918     A `MetaGraphDef` proto and dictionary of `Variables` in the exported
 919     name scope.
 920 
 921   Raises:
 922     ValueError: When the `GraphDef` is larger than 2GB.
 923   """
 924   if context.executing_eagerly():
 925     raise ValueError("Exporting/importing meta graphs is not supported when "
 926                      "Eager Execution is enabled.")
 927   graph = graph or ops.get_default_graph()
 928 
 929   exclude_nodes = None
 930   unbound_inputs = []
 931   if export_scope or clear_extraneous_savers or clear_devices:
 932     if graph_def:
 933       new_graph_def = graph_pb2.GraphDef()
 934       new_graph_def.versions.CopyFrom(graph_def.versions)
 935       new_graph_def.library.CopyFrom(graph_def.library)
 936 
 937       if clear_extraneous_savers:
 938         exclude_nodes = _find_extraneous_saver_nodes(graph_def, saver_def)
 939 
 940       for node_def in graph_def.node:
 941         if _should_include_node(node_def.name, export_scope, exclude_nodes):
 942           new_node_def = _node_def(node_def, export_scope, unbound_inputs,
 943                                    clear_devices=clear_devices)
 944           new_graph_def.node.extend([new_node_def])
 945       graph_def = new_graph_def
 946     else:
 947       # Only do this complicated work if we want to remove a name scope.
 948       graph_def = graph_pb2.GraphDef()
 949       # pylint: disable=protected-access
 950       graph_def.versions.CopyFrom(graph.graph_def_versions)
 951       bytesize = 0
 952 
 953       if clear_extraneous_savers:
 954         exclude_nodes = _find_extraneous_saver_nodes(graph.as_graph_def(),
 955                                                      saver_def)
 956 
 957       for key in sorted(graph._nodes_by_id):
 958         if _should_include_node(graph._nodes_by_id[key].name,
 959                                 export_scope,
 960                                 exclude_nodes):
 961           value = graph._nodes_by_id[key]
 962           # pylint: enable=protected-access
 963           node_def = _node_def(value.node_def, export_scope, unbound_inputs,
 964                                clear_devices=clear_devices)
 965           graph_def.node.extend([node_def])
 966           if value.outputs:
 967             assert "_output_shapes" not in graph_def.node[-1].attr
 968             graph_def.node[-1].attr["_output_shapes"].list.shape.extend([
 969                 output.get_shape().as_proto() for output in value.outputs])
 970           bytesize += value.node_def.ByteSize()
 971           if bytesize >= (1 << 31) or bytesize < 0:
 972             raise ValueError("GraphDef cannot be larger than 2GB.")
 973 
 974       graph._copy_functions_to_graph_def(graph_def, bytesize)  # pylint: disable=protected-access
 975 
 976     # It's possible that not all the inputs are in the export_scope.
 977     # If we would like such information included in the exported meta_graph,
 978     # add them to a special unbound_inputs collection.
 979     if unbound_inputs_col_name:
 980       # Clears the unbound_inputs collections.
 981       graph.clear_collection(unbound_inputs_col_name)
 982       for k in unbound_inputs:
 983         graph.add_to_collection(unbound_inputs_col_name, k)
 984 
 985   var_list = {}
 986   variables = graph.get_collection(ops.GraphKeys.GLOBAL_VARIABLES,
 987                                    scope=export_scope)
 988   for v in variables:
 989     if _should_include_node(v, export_scope, exclude_nodes):
 990       var_list[ops.strip_name_scope(v.name, export_scope)] = v
 991 
 992   scoped_meta_graph_def = create_meta_graph_def(
 993       graph_def=graph_def,
 994       graph=graph,
 995       export_scope=export_scope,
 996       exclude_nodes=exclude_nodes,
 997       clear_extraneous_savers=clear_extraneous_savers,
 998       saver_def=saver_def,
 999       strip_default_attrs=strip_default_attrs,
1000       **kwargs)
1001 
1002   if filename:
1003     graph_io.write_graph(
1004         scoped_meta_graph_def,
1005         os.path.dirname(filename),
1006         os.path.basename(filename),
1007         as_text=as_text)
1008 
1009   return scoped_meta_graph_def, var_list

1012 def copy_scoped_meta_graph(from_scope, to_scope,
1013                            from_graph=None, to_graph=None):
1014   """Copies a sub-meta_graph from one scope to another.
1015 
1016   Args:
1017     from_scope: `String` name scope containing the subgraph to be copied.
1018     to_scope: `String` name scope under which the copied subgraph will reside.
1019     from_graph: Optional `Graph` from which to copy the subgraph. If `None`, the
1020       default graph is use.
1021     to_graph: Optional `Graph` to which to copy the subgraph. If `None`, the
1022       default graph is used.
1023 
1024   Returns:
1025     A dictionary of `Variables` that has been copied into `to_scope`.
1026 
1027   Raises:
1028     ValueError: If `from_scope` and `to_scope` are the same while
1029       `from_graph` and `to_graph` are also the same.
1030   """
1031   from_graph = from_graph or ops.get_default_graph()
1032   to_graph = to_graph or ops.get_default_graph()
1033 
1034   if from_graph == to_graph and from_scope == to_scope:
1035     raise ValueError("'from_scope' and 'to_scope' need to be different "
1036                      "when performing copy in the same graph.")
1037 
1038   orig_meta_graph, var_list = export_scoped_meta_graph(
1039       export_scope=from_scope, graph=from_graph)
1040   var_list = import_scoped_meta_graph(orig_meta_graph,
1041                                       graph=to_graph,
1042                                       import_scope=to_scope)
1043   return var_list
