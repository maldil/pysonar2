  37 @tf_export('load_op_library')
  38 def load_op_library(library_filename):
  39   """Loads a TensorFlow plugin, containing custom ops and kernels.
  40 
  41   Pass "library_filename" to a platform-specific mechanism for dynamically
  42   loading a library. The rules for determining the exact location of the
  43   library are platform-specific and are not documented here. When the
  44   library is loaded, ops and kernels registered in the library via the
  45   `REGISTER_*` macros are made available in the TensorFlow process. Note
  46   that ops with the same name as an existing op are rejected and not
  47   registered with the process.
  48 
  49   Args:
  50     library_filename: Path to the plugin.
  51       Relative or absolute filesystem path to a dynamic library file.
  52 
  53   Returns:
  54     A python module containing the Python wrappers for Ops defined in
  55     the plugin.
  56 
  57   Raises:
  58     RuntimeError: when unable to load the library or get the python wrappers.
  59   """
  60   lib_handle = py_tf.TF_LoadLibrary(library_filename)
  61 
  62   op_list_str = py_tf.TF_GetOpList(lib_handle)
  63   op_list = op_def_pb2.OpList()
  64   op_list.ParseFromString(compat.as_bytes(op_list_str))
  65   wrappers = py_tf.GetPythonWrappers(op_list_str)
  66 
  67   # Delete the library handle to release any memory held in C
  68   # that are no longer needed.
  69   py_tf.TF_DeleteLibraryHandle(lib_handle)
  70 
  71   # Get a unique name for the module.
  72   module_name = hashlib.md5(wrappers).hexdigest()
  73   if module_name in sys.modules:
  74     return sys.modules[module_name]
  75   module = imp.new_module(module_name)
  76   # pylint: disable=exec-used
  77   exec(wrappers, module.__dict__)
  78   # Stash away the library handle for making calls into the dynamic library.
  79   module.LIB_HANDLE = lib_handle
  80   # OpDefs of the list of ops defined in the library.
  81   module.OP_LIST = op_list
  82   sys.modules[module_name] = module
  83   return module

  86 @tf_export('load_file_system_library')
  87 def load_file_system_library(library_filename):
  88   """Loads a TensorFlow plugin, containing file system implementation.
  89 
  90   Pass `library_filename` to a platform-specific mechanism for dynamically
  91   loading a library. The rules for determining the exact location of the
  92   library are platform-specific and are not documented here.
  93 
  94   Args:
  95     library_filename: Path to the plugin.
  96       Relative or absolute filesystem path to a dynamic library file.
  97 
  98   Returns:
  99     None.
 100 
 101   Raises:
 102     RuntimeError: when unable to load the library.
 103   """
 104   py_tf.TF_LoadLibrary(library_filename)

 107 def _is_shared_object(filename):
 108   """Check the file to see if it is a shared object, only using extension."""
 109   if platform.system() == 'Linux':
 110     if filename.endswith('.so'):
 111       return True
 112     else:
 113       index = filename.rfind('.so.')
 114       if index == -1:
 115         return False
 116       else:
 117         # A shared object with the API version in filename
 118         return filename[index + 4].isdecimal()
 119   elif platform.system() == 'Darwin':
 120     return filename.endswith('.dylib')
 121   elif platform.system() == 'Windows':
 122     return filename.endswith('.dll')
 123   else:
 124     return False

 127 @tf_export('load_library')
 128 def load_library(library_location):
 129   """Loads a TensorFlow plugin.
 130 
 131   "library_location" can be a path to a specific shared object, or a folder.
 132   If it is a folder, all sahred objects that are named "libtfkernel*" will be
 133   loaded. When the library is loaded, kernels registered in the library via the
 134   `REGISTER_*` macros are made available in the TensorFlow process.
 135 
 136   Args:
 137     library_location: Path to the plugin or the folder of plugins.
 138       Relative or absolute filesystem path to a dynamic library file or folder.
 139 
 140   Returns:
 141     None
 142 
 143   Raises:
 144     OSError: When the file to be loaded is not found.
 145     RuntimeError: when unable to load the library.
 146   """
 147   if file_io.file_exists(library_location):
 148     if file_io.is_directory(library_location):
 149       directory_contents = file_io.list_directory(library_location)
 150 
 151       kernel_libraries = [
 152           os.path.join(library_location, f) for f in directory_contents
 153           if _is_shared_object(f)]
 154     else:
 155       kernel_libraries = [library_location]
 156 
 157     for lib in kernel_libraries:
 158       py_tf.TF_LoadLibrary(lib)
 159 
 160   else:
 161     raise OSError(
 162         errno.ENOENT,
 163         'The file or folder to load kernel libraries from does not exist.',
 164         library_location)
