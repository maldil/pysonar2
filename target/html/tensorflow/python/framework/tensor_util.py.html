 185 def GetFromNumpyDTypeDict(dtype_dict, dtype):
 186   # NOTE: dtype_dict.get(dtype) always returns None.
 187   for key, val in six.iteritems(dtype_dict):
 188     if key == dtype:
 189       return val
 190   return None

 193 def GetNumpyAppendFn(dtype):
 194   # numpy dtype for strings are variable length. We can not compare
 195   # dtype with a single constant (np.string does not exist) to decide
 196   # dtype is a "string" type. We need to compare the dtype.type to be
 197   # sure it's a string type.
 198   if dtype.type == np.string_ or dtype.type == np.unicode_:
 199     if _FAST_TENSOR_UTIL_AVAILABLE:
 200       return fast_tensor_util.AppendObjectArrayToTensorProto
 201     else:
 202       return SlowAppendObjectArrayToTensorProto
 203   return GetFromNumpyDTypeDict(_NP_TO_APPEND_FN, dtype)

 218 def _GetDenseDimensions(list_of_lists):
 219   """Returns the inferred dense dimensions of a list of lists."""
 220   if not isinstance(list_of_lists, (list, tuple)):
 221     return []
 222   elif not list_of_lists:
 223     return [0]
 224   else:
 225     return [len(list_of_lists)] + _GetDenseDimensions(list_of_lists[0])

 228 def _FlattenToStrings(nested_strings):
 229   if isinstance(nested_strings, (list, tuple)):
 230     for inner in nested_strings:
 231       for flattened_string in _FlattenToStrings(inner):
 232         yield flattened_string
 233   else:
 234     yield nested_strings

 253 def _FirstNotNone(l):
 254   for x in l:
 255     if x is not None:
 256       if isinstance(x, ops.Tensor):
 257         return _Message("list containing Tensors")
 258       else:
 259         return x
 260   return None

 263 def _NotNone(v):
 264   if v is None:
 265     return _Message("None")
 266   else:
 267     return v

 270 def _FilterTuple(v):
 271   if not isinstance(v, (list, tuple)):
 272     return v
 273   if isinstance(v, tuple):
 274     if not any(isinstance(x, (list, tuple)) for x in v):
 275       return None
 276   if isinstance(v, list):
 277     if not any(isinstance(x, (list, tuple)) for x in v):
 278       return _FirstNotNone(
 279           [None if isinstance(x, (list, tuple)) else x for x in v])
 280   return _FirstNotNone([_FilterTuple(x) for x in v])
 281 
 282 
 283 def _FilterInt(v):
 284   if isinstance(v, (list, tuple)):
 285     return _FirstNotNone([_FilterInt(x) for x in v])
 286   return None if isinstance(
 287       v, (compat.integral_types, tensor_shape.Dimension)) else _NotNone(v)
 288 
 289 
 290 def _FilterFloat(v):
 291   if isinstance(v, (list, tuple)):
 292     return _FirstNotNone([_FilterFloat(x) for x in v])
 293   return None if isinstance(v, compat.real_types) else _NotNone(v)
 294 
 295 
 296 def _FilterComplex(v):
 297   if isinstance(v, (list, tuple)):
 298     return _FirstNotNone([_FilterComplex(x) for x in v])
 299   return None if isinstance(v, compat.complex_types) else _NotNone(v)
 300

 302 def _FilterStr(v):
 303   if isinstance(v, (list, tuple)):
 304     return _FirstNotNone([_FilterStr(x) for x in v])
 305   if isinstance(v, compat.bytes_or_text_types):
 306     return None
 307   else:
 308     return _NotNone(v)

 311 def _FilterBool(v):
 312   if isinstance(v, (list, tuple)):

 345 def _AssertCompatible(values, dtype):
 346   fn_list = _TF_TO_IS_OK.get(dtype, [_FilterNotTensor])
 347   mismatch = _FirstNotNone([fn(values) for fn in fn_list])
 348   if mismatch is not None:
 349     if dtype is None:
 350       raise TypeError("List of Tensors when single Tensor expected")
 351     else:
 352       raise TypeError("Expected %s, got %s of type '%s' instead." %
 353                       (dtype.name, repr(mismatch), type(mismatch).__name__))

 356 @tf_export("make_tensor_proto")
 357 def make_tensor_proto(values, dtype=None, shape=None, verify_shape=False):
 358   """Create a TensorProto.
 359 
 360   Args:
 361     values:         Values to put in the TensorProto.
 362     dtype:          Optional tensor_pb2 DataType value.
 363     shape:          List of integers representing the dimensions of tensor.
 364     verify_shape:   Boolean that enables verification of a shape of values.
 365 
 366   Returns:
 367     A `TensorProto`. Depending on the type, it may contain data in the
 368     "tensor_content" attribute, which is not directly useful to Python programs.
 369     To access the values you should convert the proto back to a numpy ndarray
 370     with `tf.make_ndarray(proto)`.
 371 
 372     If `values` is a `TensorProto`, it is immediately returned; `dtype` and
 373     `shape` are ignored.
 374 
 375   Raises:
 376     TypeError:  if unsupported types are provided.
 377     ValueError: if arguments have inappropriate values or if verify_shape is
 378      True and shape of values is not equals to a shape from the argument.
 379 
 380   make_tensor_proto accepts "values" of a python scalar, a python list, a
 381   numpy ndarray, or a numpy scalar.
 382 
 383   If "values" is a python scalar or a python list, make_tensor_proto
 384   first convert it to numpy ndarray. If dtype is None, the
 385   conversion tries its best to infer the right numpy data
 386   type. Otherwise, the resulting numpy array has a compatible data
 387   type with the given dtype.
 388 
 389   In either case above, the numpy ndarray (either the caller provided
 390   or the auto converted) must have the compatible type with dtype.
 391 
 392   make_tensor_proto then converts the numpy array to a tensor proto.
 393 
 394   If "shape" is None, the resulting tensor proto represents the numpy
 395   array precisely.
 396 
 397   Otherwise, "shape" specifies the tensor's shape and the numpy array
 398   can not have more elements than what "shape" specifies.
 399 
 400   """
 401   if isinstance(values, tensor_pb2.TensorProto):
 402     return values
 403 
 404   if dtype:
 405     dtype = dtypes.as_dtype(dtype)
 406 
 407   is_quantized = (
 408       dtype in [
 409           dtypes.qint8, dtypes.quint8, dtypes.qint16, dtypes.quint16,
 410           dtypes.qint32
 411       ])
 412 
 413   # We first convert value to a numpy array or scalar.
 414   if isinstance(values, (np.ndarray, np.generic)):
 415     if dtype:
 416       nparray = values.astype(dtype.as_numpy_dtype)
 417     else:
 418       nparray = values
 419   elif callable(getattr(values, "__array__", None)) or isinstance(
 420       getattr(values, "__array_interface__", None), dict):
 421     # If a class has the __array__ method, or __array_interface__ dict, then it
 422     # is possible to convert to numpy array.
 423     nparray = np.asarray(values, dtype=dtype)
 424 
 425     # This is the preferred way to create an array from the object, so replace
 426     # the `values` with the array so that _FlattenToStrings is not run.
 427     values = nparray
 428   else:
 429     if values is None:
 430       raise ValueError("None values not supported.")
 431     # if dtype is provided, forces numpy array to be the type
 432     # provided if possible.
 433     if dtype and dtype.is_numpy_compatible:
 434       np_dt = dtype.as_numpy_dtype
 435     else:
 436       np_dt = None
 437     # If shape is None, numpy.prod returns None when dtype is not set, but raises
 438     # exception when dtype is set to np.int64
 439     if shape is not None and np.prod(shape, dtype=np.int64) == 0:
 440       nparray = np.empty(shape, dtype=np_dt)
 441     else:
 442       _AssertCompatible(values, dtype)
 443       nparray = np.array(values, dtype=np_dt)
 444       # check to them.
 445       # We need to pass in quantized values as tuples, so don't apply the shape
 446       if (list(nparray.shape) != _GetDenseDimensions(values) and
 447           not is_quantized):
 448         raise ValueError("""Argument must be a dense tensor: %s"""
 449                          """ - got shape %s, but wanted %s.""" %
 450                          (values, list(nparray.shape),
 451                           _GetDenseDimensions(values)))
 452 
 453     # python/numpy default float type is float64. We prefer float32 instead.
 454     if (nparray.dtype == np.float64) and dtype is None:
 455       nparray = nparray.astype(np.float32)
 456     # python/numpy default int type is int64. We prefer int32 instead.
 457     elif (nparray.dtype == np.int64) and dtype is None:
 458       downcasted_array = nparray.astype(np.int32)
 459       # Do not down cast if it leads to precision loss.
 460       if np.array_equal(downcasted_array, nparray):
 461         nparray = downcasted_array
 462 
 463   # if dtype is provided, it must be compatible with what numpy
 464   # conversion says.
 465   numpy_dtype = dtypes.as_dtype(nparray.dtype)
 466   if numpy_dtype is None:
 467     raise TypeError("Unrecognized data type: %s" % nparray.dtype)
 468 
 469   # If dtype was specified and is a quantized type, we convert
 470   # numpy_dtype back into the quantized version.
 471   if is_quantized:
 472     numpy_dtype = dtype
 473 
 474   if dtype is not None and (not hasattr(dtype, "base_dtype") or
 475                             dtype.base_dtype != numpy_dtype.base_dtype):
 476     raise TypeError("Incompatible types: %s vs. %s. Value is %s" %
 477                     (dtype, nparray.dtype, values))
 478 
 479   # If shape is not given, get the shape from the numpy array.
 480   if shape is None:
 481     shape = nparray.shape
 482     is_same_size = True
 483     shape_size = nparray.size
 484   else:
 485     shape = [int(dim) for dim in shape]
 486     shape_size = np.prod(shape, dtype=np.int64)
 487     is_same_size = shape_size == nparray.size
 488 
 489     if verify_shape:
 490       if not nparray.shape == tuple(shape):
 491         raise TypeError("Expected Tensor's shape: %s, got %s." %
 492                         (tuple(shape), nparray.shape))
 493 
 494     if nparray.size > shape_size:
 495       raise ValueError(
 496           "Too many elements provided. Needed at most %d, but received %d" %
 497           (shape_size, nparray.size))
 498 
 499   tensor_proto = tensor_pb2.TensorProto(
 500       dtype=numpy_dtype.as_datatype_enum,
 501       tensor_shape=tensor_shape.as_shape(shape).as_proto())
 502 
 503   if is_same_size and numpy_dtype in _TENSOR_CONTENT_TYPES and shape_size > 1:
 504     if nparray.size * nparray.itemsize >= (1 << 31):
 505       raise ValueError(
 506           "Cannot create a tensor proto whose content is larger than 2GB.")
 507     tensor_proto.tensor_content = nparray.tostring()
 508     return tensor_proto
 509 
 510   # If we were not given values as a numpy array, compute the proto_values
 511   # from the given values directly, to avoid numpy trimming nulls from the
 512   # strings. Since values could be a list of strings, or a multi-dimensional
 513   # list of lists that might or might not correspond to the given shape,
 514   # we flatten it conservatively.
 515   if numpy_dtype == dtypes.string and not isinstance(values, np.ndarray):
 516     proto_values = _FlattenToStrings(values)
 517 
 518     # At this point, values may be a list of objects that we could not
 519     # identify a common type for (hence it was inferred as
 520     # np.object/dtypes.string).  If we are unable to convert it to a
 521     # string, we raise a more helpful error message.
 522     #
 523     # Ideally, we'd be able to convert the elements of the list to a
 524     # common type, but this type inference requires some thinking and
 525     # so we defer it for now.
 526     try:
 527       str_values = [compat.as_bytes(x) for x in proto_values]
 528     except TypeError:
 529       raise TypeError("Failed to convert object of type %s to Tensor. "
 530                       "Contents: %s. Consider casting elements to a "
 531                       "supported type." % (type(values), values))
 532     tensor_proto.string_val.extend(str_values)
 533     return tensor_proto
 534 
 535   # TensorFlow expects C order (a.k.a., eigen row major).
 536   proto_values = nparray.ravel()
 537 
 538   append_fn = GetNumpyAppendFn(proto_values.dtype)
 539   if append_fn is None:
 540     raise TypeError(
 541         "Element type not supported in TensorProto: %s" % numpy_dtype.name)
 542   append_fn(tensor_proto, proto_values)
 543 
 544   return tensor_proto

 547 @tf_export("make_ndarray")
 548 def MakeNdarray(tensor):
 549   """Create a numpy ndarray from a tensor.
 550 
 551   Create a numpy ndarray with the same shape and data as the tensor.
 552 
 553   Args:
 554     tensor: A TensorProto.
 555 
 556   Returns:
 557     A numpy array with the tensor contents.
 558 
 559   Raises:
 560     TypeError: if tensor has unsupported type.
 561 
 562   """
 563   shape = [d.size for d in tensor.tensor_shape.dim]
 564   num_elements = np.prod(shape, dtype=np.int64)
 565   tensor_dtype = dtypes.as_dtype(tensor.dtype)
 566   dtype = tensor_dtype.as_numpy_dtype
 567 
 568   if tensor.tensor_content:
 569     return (np.frombuffer(tensor.tensor_content, dtype=dtype).copy()
 570             .reshape(shape))
 571   elif tensor_dtype == dtypes.float16 or tensor_dtype == dtypes.bfloat16:
 572     # the half_val field of the TensorProto stores the binary representation
 573     # of the fp16: we need to reinterpret this as a proper float16
 574     if len(tensor.half_val) == 1:
 575       tmp = np.array(tensor.half_val[0], dtype=np.uint16)
 576       tmp.dtype = tensor_dtype.as_numpy_dtype
 577       return np.repeat(tmp, num_elements).reshape(shape)
 578     else:
 579       tmp = np.fromiter(tensor.half_val, dtype=np.uint16)
 580       tmp.dtype = tensor_dtype.as_numpy_dtype
 581       return tmp.reshape(shape)
 582   elif tensor_dtype == dtypes.float32:
 583     if len(tensor.float_val) == 1:
 584       return np.repeat(
 585           np.array(tensor.float_val[0], dtype=dtype),
 586           num_elements).reshape(shape)
 587     else:
 588       return np.fromiter(tensor.float_val, dtype=dtype).reshape(shape)
 589   elif tensor_dtype == dtypes.float64:
 590     if len(tensor.double_val) == 1:
 591       return np.repeat(
 592           np.array(tensor.double_val[0], dtype=dtype),
 593           num_elements).reshape(shape)
 594     else:
 595       return np.fromiter(tensor.double_val, dtype=dtype).reshape(shape)
 596   elif tensor_dtype in [
 597       dtypes.int32, dtypes.uint8, dtypes.uint16, dtypes.int16, dtypes.int8,
 598       dtypes.qint32, dtypes.quint8, dtypes.qint8, dtypes.qint16, dtypes.quint16
 599   ]:
 600     if len(tensor.int_val) == 1:
 601       return np.repeat(np.array(tensor.int_val[0], dtype=dtype),
 602                        num_elements).reshape(shape)
 603     else:
 604       return np.fromiter(tensor.int_val, dtype=dtype).reshape(shape)
 605   elif tensor_dtype == dtypes.int64:
 606     if len(tensor.int64_val) == 1:
 607       return np.repeat(
 608           np.array(tensor.int64_val[0], dtype=dtype),
 609           num_elements).reshape(shape)
 610     else:
 611       return np.fromiter(tensor.int64_val, dtype=dtype).reshape(shape)
 612   elif tensor_dtype == dtypes.string:
 613     if len(tensor.string_val) == 1:
 614       return np.repeat(
 615           np.array(tensor.string_val[0], dtype=dtype),
 616           num_elements).reshape(shape)
 617     else:
 618       return np.array(
 619           [x for x in tensor.string_val], dtype=dtype).reshape(shape)
 620   elif tensor_dtype == dtypes.complex64:
 621     it = iter(tensor.scomplex_val)
 622     if len(tensor.scomplex_val) == 2:
 623       return np.repeat(
 624           np.array(
 625               complex(tensor.scomplex_val[0], tensor.scomplex_val[1]),
 626               dtype=dtype), num_elements).reshape(shape)
 627     else:
 628       return np.array(
 629           [complex(x[0], x[1]) for x in zip(it, it)],
 630           dtype=dtype).reshape(shape)
 631   elif tensor_dtype == dtypes.complex128:
 632     it = iter(tensor.dcomplex_val)
 633     if len(tensor.dcomplex_val) == 2:
 634       return np.repeat(
 635           np.array(
 636               complex(tensor.dcomplex_val[0], tensor.dcomplex_val[1]),
 637               dtype=dtype), num_elements).reshape(shape)
 638     else:
 639       return np.array(
 640           [complex(x[0], x[1]) for x in zip(it, it)],
 641           dtype=dtype).reshape(shape)
 642   elif tensor_dtype == dtypes.bool:
 643     if len(tensor.bool_val) == 1:
 644       return np.repeat(np.array(tensor.bool_val[0], dtype=dtype),
 645                        num_elements).reshape(shape)
 646     else:
 647       return np.fromiter(tensor.bool_val, dtype=dtype).reshape(shape)
 648   else:
 649     raise TypeError("Unsupported tensor type: %s" % tensor.dtype)

 652 def ShapeEquals(tensor_proto, shape):
 653   """Returns True if "tensor_proto" has the given "shape".
 654 
 655   Args:
 656     tensor_proto: A TensorProto.
 657     shape: A tensor shape, expressed as a TensorShape, list, or tuple.
 658 
 659   Returns:
 660     True if "tensor_proto" has the given "shape", otherwise False.
 661 
 662   Raises:
 663     TypeError: If "tensor_proto" is not a TensorProto, or shape is not a
 664       TensorShape, list, or tuple.
 665   """
 666   if not isinstance(tensor_proto, tensor_pb2.TensorProto):
 667     raise TypeError("tensor_proto is not a tensor_pb2.TensorProto object")
 668   if isinstance(shape, tensor_shape_pb2.TensorShapeProto):
 669     shape = [d.size for d in shape.dim]
 670   elif not isinstance(shape, (list, tuple)):
 671     raise TypeError("shape is not a list or tuple")
 672   tensor_shape_list = [d.size for d in tensor_proto.tensor_shape.dim]
 673   return all(x == y for x, y in zip(tensor_shape_list, shape))

 676 def _ConstantValue(tensor, partial):
 677   # TODO(touts): Support Variables?
 678   if not isinstance(tensor, ops.Tensor):
 679     raise TypeError("tensor is not a Tensor")
 680   if tensor.op.type == "Const":
 681     return MakeNdarray(tensor.op.get_attr("value"))
 682   elif tensor.op.type == "Shape":
 683     input_shape = tensor.op.inputs[0].get_shape()
 684     if input_shape.is_fully_defined():
 685       return np.array(
 686           [dim.value for dim in input_shape.dims],
 687           dtype=tensor.dtype.as_numpy_dtype)
 688     else:
 689       return None
 690   elif tensor.op.type == "Size":
 691     input_shape = tensor.op.inputs[0].get_shape()
 692     if input_shape.is_fully_defined():
 693       return np.prod([dim.value for dim in input_shape.dims], dtype=np.int32)
 694     else:
 695       return None
 696   elif tensor.op.type == "Rank":
 697     input_shape = tensor.op.inputs[0].get_shape()
 698     if input_shape.ndims is not None:
 699       return np.ndarray(
 700           shape=(),
 701           buffer=np.array([input_shape.ndims], dtype=np.int32),
 702           dtype=np.int32)
 703     else:
 704       return None
 705   elif tensor.op.type == "Range":
 706     start = constant_value(tensor.op.inputs[0])
 707     if start is None:
 708       return None
 709     limit = constant_value(tensor.op.inputs[1])
 710     if limit is None:
 711       return None
 712     delta = constant_value(tensor.op.inputs[2])
 713     if delta is None:
 714       return None
 715     return np.arange(start, limit, delta, dtype=tensor.dtype.as_numpy_dtype)
 716   elif tensor.op.type == "Cast":
 717     pre_cast = constant_value(tensor.op.inputs[0])
 718     if pre_cast is None:
 719       return None
 720     cast_dtype = dtypes.as_dtype(tensor.op.get_attr("DstT"))
 721     return pre_cast.astype(cast_dtype.as_numpy_dtype)
 722   elif tensor.op.type == "Concat":
 723     dim = constant_value(tensor.op.inputs[0])
 724     if dim is None:
 725       return None
 726     values = []
 727     for x in tensor.op.inputs[1:]:
 728       value = constant_value(x)
 729       if value is None:
 730         return None
 731       values.append(value)
 732     return np.concatenate(values, axis=dim)
 733   elif tensor.op.type == "ConcatV2":
 734     dim = constant_value(tensor.op.inputs[-1])
 735     if dim is None:
 736       return None
 737     values = []
 738     for x in tensor.op.inputs[:-1]:
 739       value = constant_value(x)
 740       if value is None:
 741         return None
 742       values.append(value)
 743     return np.concatenate(values, axis=dim)
 744   elif tensor.op.type == "Pack":
 745     values = []
 746     # Some imported GraphDefs have Pack ops with zero inputs. Those are invalid
 747     # and shouldn't be produced, but to deal sensibly with them here we check
 748     # and return None.
 749     if not tensor.op.inputs:
 750       return None
 751     # We can't handle axis != 0 Packs at the moment.
 752     if tensor.op.get_attr("axis") != 0:
 753       return None
 754     for x in tensor.op.inputs:
 755       value = constant_value(x, partial)
 756       if value is None and not partial:
 757         return None
 758       values.append(value)
 759     return np.array(values)
 760   elif tensor.op.type == "Fill":
 761     fill_shape = tensor.shape
 762     fill_value = constant_value(tensor.op.inputs[1])
 763     if fill_shape.is_fully_defined() and fill_value is not None:
 764       return np.full(fill_shape.as_list(), fill_value, dtype=fill_value.dtype)
 765     else:
 766       return None
 767   elif tensor.op.type == "Equal":
 768     value1 = constant_value(tensor.op.inputs[0])
 769     if value1 is None:
 770       return None
 771     value2 = constant_value(tensor.op.inputs[1])
 772     if value2 is None:
 773       return None
 774     return np.equal(value1, value2)
 775   elif tensor.op.type == "NotEqual":
 776     value1 = constant_value(tensor.op.inputs[0])
 777     if value1 is None:
 778       return None
 779     value2 = constant_value(tensor.op.inputs[1])
 780     if value2 is None:
 781       return None
 782     return np.not_equal(value1, value2)
 783   else:
 784     return None

 787 def constant_value(tensor, partial=False):  # pylint: disable=invalid-name
 788   """Returns the constant value of the given tensor, if efficiently calculable.
 789 
 790   This function attempts to partially evaluate the given tensor, and
 791   returns its value as a numpy ndarray if this succeeds.
 792 
 793   TODO(mrry): Consider whether this function should use a registration
 794   mechanism like gradients and ShapeFunctions, so that it is easily
 795   extensible.
 796 
 797   NOTE: If `constant_value(tensor)` returns a non-`None` result, it will no
 798   longer be possible to feed a different value for `tensor`. This allows the
 799   result of this function to influence the graph that is constructed, and
 800   permits static shape optimizations.
 801 
 802   Args:
 803     tensor: The Tensor to be evaluated.
 804     partial: If True, the returned numpy array is allowed to have partially
 805       evaluated values. Values that can't be evaluated will be None.
 806 
 807   Returns:
 808     A numpy ndarray containing the constant value of the given `tensor`,
 809     or None if it cannot be calculated.
 810 
 811   Raises:
 812     TypeError: if tensor is not an ops.Tensor.
 813   """
 814   if isinstance(tensor, ops.EagerTensor):
 815     return tensor.numpy()
 816   ret = _ConstantValue(tensor, partial)
 817   if ret is not None:
 818     # The caller may now depend on the constant value of `tensor`, so we
 819     # conservatively prevent it from being fed.
 820     tensor.graph.prevent_feeding(tensor)
 821   return ret

 824 def constant_value_as_shape(tensor):  # pylint: disable=invalid-name
 825   """A version of `constant_value()` that returns a `TensorShape`.
 826 
 827   This version should be used when a constant tensor value is
 828   interpreted as a (possibly partial) shape, e.g. in the shape
 829   function for `tf.reshape()`. By explicitly requesting a
 830   `TensorShape` as the return value, it is possible to represent
 831   unknown dimensions; by contrast, `constant_value()` is
 832   all-or-nothing.
 833 
 834   Args:
 835     tensor: The rank-0 or rank-1 Tensor to be evaluated.
 836 
 837   Returns:
 838     A `TensorShape` based on the constant value of the given `tensor`.
 839 
 840   Raises:
 841     ValueError: If the shape is rank-0 and is not statically known to be -1.
 842   """
 843   if isinstance(tensor, ops.EagerTensor):
 844     return tensor_shape.as_shape(
 845         [dim if dim != -1 else None for dim in tensor.numpy()])
 846 
 847   if tensor.get_shape().ndims == 0:
 848     value = constant_value(tensor)
 849     if value is None:
 850       raise ValueError(
 851           "Received a scalar with unknown value as shape; require a statically "
 852           "known scalar with value '-1' to describe an unknown shape.")
 853     if value != -1:
 854       raise ValueError(
 855           "Received a scalar value '%s' as shape; require a statically known "
 856           "scalar with value '-1' to describe an unknown shape." % value)
 857     return tensor_shape.unknown_shape()
 858 
 859   shape = tensor.get_shape().with_rank(1)
 860   if shape == [0]:
 861     return tensor_shape.scalar()
 862   elif tensor.op.type == "Shape":
 863     return tensor.op.inputs[0].get_shape()
 864   elif tensor.op.type == "Pack":
 865     ret = tensor_shape.scalar()  # Empty list.
 866     # Since we expect rank 1 inputs, Pack's axis must be zero, otherwise it
 867     # would not be rank 1.
 868     assert tensor.op.get_attr("axis") == 0
 869     for pack_input in tensor.op.inputs:
 870       # `pack_input` must be a scalar. Attempt to evaluate it, and append it
 871       # to `ret`.
 872       pack_input_val = constant_value(pack_input)
 873       if pack_input_val is None or pack_input_val < 0:
 874         new_dim = tensor_shape.Dimension(None)
 875       else:
 876         new_dim = tensor_shape.Dimension(pack_input_val)
 877       ret = ret.concatenate([new_dim])
 878     return ret
 879   elif tensor.op.type == "Concat":
 880     # We assume that `tensor.op.inputs[0]` evaluates to 0, as this is
 881     # the only legal value when concatenating vectors, and it will
 882     # have been checked by a previous shape function.
 883     ret = tensor_shape.scalar()  # Empty list.
 884     for concat_input in tensor.op.inputs[1:]:
 885       # `concat_input` must be a vector. Attempt to evaluate it as a shape,
 886       # and concatenate it with `ret`.
 887       ret = ret.concatenate(constant_value_as_shape(concat_input))
 888     return ret
 889   elif tensor.op.type == "ConcatV2":
 890     # We assume that `tensor.op.inputs[-1]` evaluates to 0, as this is
 891     # the only legal value when concatenating vectors, and it will
 892     # have been checked by a previous shape function.
 893     ret = tensor_shape.scalar()  # Empty list.
 894     for concat_input in tensor.op.inputs[:-1]:
 895       # `concat_input` must be a vector. Attempt to evaluate it as a shape,
 896       # and concatenate it with `ret`.
 897       ret = ret.concatenate(constant_value_as_shape(concat_input))
 898     return ret
 899   elif tensor.op.type == "StridedSlice":
 900     try:
 901       begin = constant_value(tensor.op.inputs[1])
 902       end = constant_value(tensor.op.inputs[2])
 903       strides = constant_value(tensor.op.inputs[3])
 904       if begin is not None and end is not None and strides is not None:
 905         begin = begin[0]
 906         end = end[0]
 907         strides = strides[0]
 908         begin_mask = tensor.op.get_attr("begin_mask")
 909         if begin_mask == 1:
 910           begin = None
 911         end_mask = tensor.op.get_attr("end_mask")
 912         if end_mask == 1:
 913           end = None
 914 
 915         ellipsis_mask = tensor.op.get_attr("ellipsis_mask")
 916         new_axis_mask = tensor.op.get_attr("new_axis_mask")
 917         shrink_axis_mask = tensor.op.get_attr("shrink_axis_mask")
 918         valid_attributes = (not ellipsis_mask and not new_axis_mask and
 919                             not shrink_axis_mask and (not begin_mask or
 920                                                       (begin_mask == 1)) and
 921                             (not end_mask or (end_mask == 1)))
 922         if valid_attributes:  # additional inputs not supported
 923           prev = constant_value_as_shape(tensor.op.inputs[0])
 924           prev = prev[begin:end:strides]
 925           ret = tensor_shape.TensorShape(prev)
 926           return ret
 927 
 928     except ValueError:  # Could come from get_attr or slicing prev.
 929       pass
 930     except TypeError:  # Could come from slicing prev.
 931       pass
 932 
 933   ret = tensor_shape.unknown_shape(shape[0].value)
 934   value = constant_value(tensor)
 935   if value is not None:
 936     ret = ret.merge_with(
 937         tensor_shape.TensorShape([d if d >= 0 else None for d in value]))
 938   return ret

 941 def is_tensor(x):  # pylint: disable=invalid-name
 942   """Check whether `x` is of tensor type.
 943 
 944   Check whether an object is a tensor. This check is equivalent to calling
 945   `isinstance(x, (tf.Tensor, tf.SparseTensor, tf.Variable))` and also checks
 946   if all the component variables of a MirroredVariable or a TowerLocalVariable
 947   are tensors.
 948 
 949   Args:
 950     x: A python object to check.
 951 
 952   Returns:
 953     `True` if `x` is a tensor, `False` if not.
 954   """
 955   return (isinstance(x, ops._TensorLike) or ops.is_dense_tensor_like(x) or  # pylint: disable=protected-access
 956           (hasattr(x, "is_tensor_like") and x.is_tensor_like))
