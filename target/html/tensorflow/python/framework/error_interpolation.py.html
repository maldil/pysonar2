  34 _NAME_REGEX = r"[A-Za-z0-9.][A-Za-z0-9_.\-/]*?"
  35 _TAG_REGEX = r"{{{{({name}) ({name})}}}}".format(name=_NAME_REGEX)
  36 _INTERPOLATION_REGEX = r"^(.*?)({tag})".format(tag=_TAG_REGEX)
  37 _INTERPOLATION_PATTERN = re.compile(_INTERPOLATION_REGEX, re.DOTALL)
  38 
  39 _ParseTag = collections.namedtuple("_ParseTag", ["type", "name"])
  40 
  41 _BAD_FILE_SUBSTRINGS = [
  42     os.path.join("tensorflow", "python"),
  43     "<embedded",
  44 ]

  47 def _parse_message(message):
  48   """Parses the message.
  49 
  50   Splits the message into separators and tags. Tags are named tuples
  51   representing the string {{type name}} and they are separated by
  52   separators. For example, in "123{{node Foo}}456{{node Bar}}789", there are
  53   two tags and three separators. The separators are the numeric characters.
  54 
  55   Args:
  56     message: String to parse
  57 
  58   Returns:
  59     (list of separator strings, list of _ParseTags).
  60 
  61     For example, if message is "123{{node Foo}}456" then this function
  62     returns (["123", "456"], [_ParseTag("node", "Foo")])
  63   """
  64   seps = []
  65   tags = []
  66   pos = 0
  67   while pos < len(message):
  68     match = re.match(_INTERPOLATION_PATTERN, message[pos:])
  69     if match:
  70       seps.append(match.group(1))
  71       tags.append(_ParseTag(match.group(3), match.group(4)))
  72       pos += match.end()
  73     else:
  74       break
  75   seps.append(message[pos:])
  76   return seps, tags

  79 def _compute_device_summary_from_list(name, device_assignment_list, prefix=""):
  80   """Return a summary of an op's device function stack.
  81 
  82   Args:
  83     name: The name of the op.
  84     device_assignment_list: The op._device_assignments list.
  85     prefix:  An optional string prefix used before each line of the multi-
  86         line string returned by this function.
  87 
  88   Returns:
  89     A multi-line string similar to:
  90         Device assignments active during op 'foo' creation:
  91           with tf.device(/cpu:0): <test_1.py:27>
  92           with tf.device(some_func<foo.py, 123>): <test_2.py:38>
  93     The first line will have no padding to its left by default.  Subsequent
  94     lines will have two spaces of left-padding.  Use the prefix argument
  95     to increase indentation.
  96   """
  97   if not device_assignment_list:
  98     message = "No device assignments were active during op '%s' creation."
  99     message %= name
 100     return prefix + message
 101 
 102   str_list = []
 103   str_list.append(
 104       "%sDevice assignments active during op '%s' creation:" % (prefix, name))
 105 
 106   for traceable_obj in device_assignment_list:
 107     location_summary = "<{file}:{line}>".format(
 108         file=traceable_obj.filename, line=traceable_obj.lineno)
 109     subs = {
 110         "prefix": prefix,
 111         "indent": "  ",
 112         "dev_name": traceable_obj.obj,
 113         "loc": location_summary,
 114     }
 115     str_list.append(
 116         "{prefix}{indent}with tf.device({dev_name}): {loc}".format(**subs))
 117 
 118   return "\n".join(str_list)

 128 def _compute_colocation_summary_from_dict(name, colocation_dict, prefix=""):
 129   """Return a summary of an op's colocation stack.
 130 
 131   Args:
 132     name: The op name.
 133     colocation_dict: The op._colocation_dict.
 134     prefix:  An optional string prefix used before each line of the multi-
 135         line string returned by this function.
 136 
 137   Returns:
 138     A multi-line string similar to:
 139         Node-device colocations active during op creation:
 140           with tf.colocate_with(test_node_1): <test_1.py:27>
 141           with tf.colocate_with(test_node_2): <test_2.py:38>
 142     The first line will have no padding to its left by default.  Subsequent
 143     lines will have two spaces of left-padding.  Use the prefix argument
 144     to increase indentation.
 145   """
 146   if not colocation_dict:
 147     message = "No node-device colocations were active during op '%s' creation."
 148     message %= name
 149     return prefix + message
 150 
 151   str_list = []
 152   str_list.append("%sNode-device colocations active during op '%s' creation:" %
 153                   (prefix, name))
 154 
 155   for coloc_name, location in colocation_dict.items():
 156     location_summary = "<{file}:{line}>".format(
 157         file=location.filename, line=location.lineno)
 158     subs = {
 159         "prefix": prefix,
 160         "indent": "  ",
 161         "name": coloc_name,
 162         "loc": location_summary,
 163     }
 164     str_list.append(
 165         "{prefix}{indent}with tf.colocate_with({name}): {loc}".format(**subs))
 166 
 167   return "\n".join(str_list)

 178 def _find_index_of_defining_frame_for_op(op):
 179   """Return index in op._traceback with first 'useful' frame.
 180 
 181   This method reads through the stack stored in op._traceback looking for the
 182   innermost frame which (hopefully) belongs to the caller.  It accomplishes this
 183   by rejecting frames whose filename appears to come from TensorFlow (see
 184   error_interpolation._BAD_FILE_SUBSTRINGS for the list of rejected substrings).
 185 
 186   Args:
 187     op: the Operation object for which we would like to find the defining
 188         location.
 189 
 190   Returns:
 191     Integer index into op._traceback where the first non-TF file was found
 192     (innermost to outermost), or 0 (for the outermost stack frame) if all files
 193     came from TensorFlow.
 194   """
 195   # pylint: disable=protected-access
 196   # Index 0 of tf_traceback is the outermost frame.
 197   tf_traceback = tf_stack.convert_stack(op._traceback)
 198   size = len(tf_traceback)
 199   # pylint: enable=protected-access
 200   filenames = [frame[tf_stack.TB_FILENAME] for frame in tf_traceback]
 201   # We process the filenames from the innermost frame to outermost.
 202   for idx, filename in enumerate(reversed(filenames)):
 203     contains_bad_substrings = [ss in filename for ss in _BAD_FILE_SUBSTRINGS]
 204     if not any(contains_bad_substrings):
 205       return size - idx - 1
 206   return 0

 209 def _get_defining_frame_from_op(op):
 210   """Find and return stack frame where op was defined."""
 211   frame_index = _find_index_of_defining_frame_for_op(op)
 212   # pylint: disable=protected-access
 213   frame = op._traceback[frame_index]
 214   # pylint: enable=protected-access
 215   return frame

 218 def compute_field_dict(op):
 219   """Return a dictionary mapping interpolation tokens to values.
 220 
 221   Args:
 222     op: op.Operation object having a _traceback member.
 223 
 224   Returns:
 225     A dictionary mapping string tokens to string values.  The keys are shown
 226     below along with example values.
 227     {
 228       "file": "tool_utils.py",
 229       "line": "124",
 230       "defined_at": " (defined at tool_utils.py:124)",
 231       "colocations":
 232           '''Node-device colocations active during op creation:
 233                with tf.colocate_with(test_node_1): <test_1.py:27>
 234                with tf.colocate_with(test_node_2): <test_2.py:38>'''
 235       "devices":
 236           '''Device assignments active during op 'foo' creation:
 237                with tf.device(/cpu:0): <test_1.py:27>
 238                with tf.device(some_func<foo.py, 123>): <test_2.py:38>'''
 239       "devs_and_colocs": A concatenation of colocations and devices, e.g.
 240           '''Node-device colocations active during op creation:
 241                with tf.colocate_with(test_node_1): <test_1.py:27>
 242                with tf.colocate_with(test_node_2): <test_2.py:38>'''
 243              Device assignments active during op 'foo' creation:
 244                with tf.device(/cpu:0): <test_1.py:27>
 245                with tf.device(some_func<foo.py, 123>): <test_2.py:38>'''
 246     }
 247   """
 248   frame = _get_defining_frame_from_op(op)
 249   filename = frame[tf_stack.TB_FILENAME]
 250   lineno = frame[tf_stack.TB_LINENO]
 251   defined_at = " (defined at %s:%d)" % (filename, lineno)
 252   colocation_summary = _compute_colocation_summary_from_op(op)
 253   device_summary = _compute_device_assignment_summary_from_op(op)
 254   combined_summary = "\n".join([colocation_summary, device_summary])
 255 
 256   field_dict = {
 257       "file": filename,
 258       "line": lineno,
 259       "defined_at": defined_at,
 260       "colocations": colocation_summary,
 261       "devices": device_summary,
 262       "devs_and_colocs": combined_summary,
 263   }
 264   return field_dict

 267 def interpolate(error_message, graph):
 268   """Interpolates an error message.
 269 
 270   The error message can contain tags of the form ^^type:name^^ which will
 271   be replaced.
 272 
 273   Args:
 274     error_message: A string to interpolate.
 275     graph: ops.Graph object containing all nodes referenced in the error
 276         message.
 277 
 278   Returns:
 279     The string with tags of the form {{type name}} interpolated.
 280   """
 281   seps, tags = _parse_message(error_message)
 282   subs = []
 283   end_msg = ""
 284 
 285   for t in tags:
 286     try:
 287       op = graph.get_operation_by_name(t.name)
 288     except KeyError:
 289       op = None
 290 
 291     msg = "{{%s %s}}" % (t.type, t.name)
 292     if op is not None:
 293       field_dict = compute_field_dict(op)
 294       if t.type == "node":
 295         msg = "node %s%s " % (t.name, field_dict["defined_at"])
 296       elif t.type == "colocation_node":
 297         msg = "node %s%s having device %s " % (t.name, field_dict["defined_at"],
 298                                                field_dict["devices"])
 299         end_msg += "\n\n" + field_dict["devs_and_colocs"]
 300     subs.append(msg)
 301   subs.append(end_msg)
 302 
 303   return "".join(
 304       itertools.chain(*six.moves.zip_longest(seps, subs, fillvalue="")))
