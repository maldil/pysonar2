  29 @tf_export("dtypes.DType", "DType")
  30 class DType(object):
  31   """Represents the type of the elements in a `Tensor`.
  32 
  33   The following `DType` objects are defined:
  34 
  35   * `tf.float16`: 16-bit half-precision floating-point.
  36   * `tf.float32`: 32-bit single-precision floating-point.
  37   * `tf.float64`: 64-bit double-precision floating-point.
  38   * `tf.bfloat16`: 16-bit truncated floating-point.
  39   * `tf.complex64`: 64-bit single-precision complex.
  40   * `tf.complex128`: 128-bit double-precision complex.
  41   * `tf.int8`: 8-bit signed integer.
  42   * `tf.uint8`: 8-bit unsigned integer.
  43   * `tf.uint16`: 16-bit unsigned integer.
  44   * `tf.uint32`: 32-bit unsigned integer.
  45   * `tf.uint64`: 64-bit unsigned integer.
  46   * `tf.int16`: 16-bit signed integer.
  47   * `tf.int32`: 32-bit signed integer.
  48   * `tf.int64`: 64-bit signed integer.
  49   * `tf.bool`: Boolean.
  50   * `tf.string`: String.
  51   * `tf.qint8`: Quantized 8-bit signed integer.
  52   * `tf.quint8`: Quantized 8-bit unsigned integer.
  53   * `tf.qint16`: Quantized 16-bit signed integer.
  54   * `tf.quint16`: Quantized 16-bit unsigned integer.
  55   * `tf.qint32`: Quantized 32-bit signed integer.
  56   * `tf.resource`: Handle to a mutable resource.
  57   * `tf.variant`: Values of arbitrary types.
  58 
  59   In addition, variants of these types with the `_ref` suffix are
  60   defined for reference-typed tensors.
  61 
  62   The `tf.as_dtype()` function converts numpy types and string type
  63   names to a `DType` object.
  64   """
  65 
  66   def __init__(self, type_enum):
  67     """Creates a new `DataType`.
  68 
  69     NOTE(mrry): In normal circumstances, you should not need to
  70     construct a `DataType` object directly. Instead, use the
  71     `tf.as_dtype()` function.
  72 
  73     Args:
  74       type_enum: A `types_pb2.DataType` enum value.
  75 
  76     Raises:
  77       TypeError: If `type_enum` is not a value `types_pb2.DataType`.
  78 
  79     """
  80     # TODO(mrry): Make the necessary changes (using __new__) to ensure
  81     # that calling this returns one of the interned values.
  82     type_enum = int(type_enum)
  83     if (type_enum not in types_pb2.DataType.values() or
  84         type_enum == types_pb2.DT_INVALID):
  85       raise TypeError(
  86           "type_enum is not a valid types_pb2.DataType: %s" % type_enum)
  87     self._type_enum = type_enum
  88 
  89   @property
  90   def _is_ref_dtype(self):
  91     """Returns `True` if this `DType` represents a reference type."""
  92     return self._type_enum > 100
  93 
  94   @property
  95   def _as_ref(self):
  96     """Returns a reference `DType` based on this `DType`."""
  97     if self._is_ref_dtype:
  98       return self
  99     else:
 100       return _INTERN_TABLE[self._type_enum + 100]
 101 
 102   @property
 103   def base_dtype(self):
 104     """Returns a non-reference `DType` based on this `DType`."""
 105     if self._is_ref_dtype:
 106       return _INTERN_TABLE[self._type_enum - 100]
 107     else:
 108       return self
 109 
 110   @property
 111   def real_dtype(self):
 112     """Returns the dtype correspond to this dtype's real part."""
 113     base = self.base_dtype
 114     if base == complex64:
 115       return float32
 116     elif base == complex128:
 117       return float64
 118     else:
 119       return self
 120 
 121   @property
 122   def is_numpy_compatible(self):
 123     return self._type_enum not in _NUMPY_INCOMPATIBLE
 124 
 125   @property
 126   def as_numpy_dtype(self):
 127     """Returns a `numpy.dtype` based on this `DType`."""
 128     return _TF_TO_NP[self._type_enum]
 129 
 130   @property
 131   def as_datatype_enum(self):
 132     """Returns a `types_pb2.DataType` enum value based on this `DType`."""
 133     return self._type_enum
 134 
 135   @property
 136   def is_bool(self):
 137     """Returns whether this is a boolean data type"""
 138     return self.base_dtype == bool
 139 
 140   @property
 141   def is_integer(self):
 142     """Returns whether this is a (non-quantized) integer type."""
 143     return (self.is_numpy_compatible and not self.is_quantized and
 144             np.issubdtype(self.as_numpy_dtype, np.integer))
 145 
 146   @property
 147   def is_floating(self):
 148     """Returns whether this is a (non-quantized, real) floating point type."""
 149     return ((self.is_numpy_compatible and
 150              np.issubdtype(self.as_numpy_dtype, np.floating)) or
 151             self.base_dtype == bfloat16)
 152 
 153   @property
 154   def is_complex(self):
 155     """Returns whether this is a complex floating point type."""
 156     return self.base_dtype in (complex64, complex128)
 157 
 158   @property
 159   def is_quantized(self):
 160     """Returns whether this is a quantized data type."""
 161     return self.base_dtype in _QUANTIZED_DTYPES_NO_REF
 162 
 163   @property
 164   def is_unsigned(self):
 165     """Returns whether this type is unsigned.
 166 
 167     Non-numeric, unordered, and quantized types are not considered unsigned, and
 168     this function returns `False`.
 169 
 170     Returns:
 171       Whether a `DType` is unsigned.
 172     """
 173     try:
 174       return self.min == 0
 175     except TypeError:
 176       return False
 177 
 178   @property
 179   def min(self):
 180     """Returns the minimum representable value in this data type.
 181 
 182     Raises:
 183       TypeError: if this is a non-numeric, unordered, or quantized type.
 184 
 185     """
 186     if (self.is_quantized or
 187         self.base_dtype in (bool, string, complex64, complex128)):
 188       raise TypeError("Cannot find minimum value of %s." % self)
 189 
 190     # there is no simple way to get the min value of a dtype, we have to check
 191     # float and int types separately
 192     try:
 193       return np.finfo(self.as_numpy_dtype()).min
 194     except:  # bare except as possible raises by finfo not documented
 195       try:
 196         return np.iinfo(self.as_numpy_dtype()).min
 197       except:
 198         if self.base_dtype == bfloat16:
 199           return _np_bfloat16(float.fromhex("-0x1.FEp127"))
 200         raise TypeError("Cannot find minimum value of %s." % self)
 201 
 202   @property
 203   def max(self):
 204     """Returns the maximum representable value in this data type.
 205 
 206     Raises:
 207       TypeError: if this is a non-numeric, unordered, or quantized type.
 208 
 209     """
 210     if (self.is_quantized or
 211         self.base_dtype in (bool, string, complex64, complex128)):
 212       raise TypeError("Cannot find maximum value of %s." % self)
 213 
 214     # there is no simple way to get the max value of a dtype, we have to check
 215     # float and int types separately
 216     try:
 217       return np.finfo(self.as_numpy_dtype()).max
 218     except:  # bare except as possible raises by finfo not documented
 219       try:
 220         return np.iinfo(self.as_numpy_dtype()).max
 221       except:
 222         if self.base_dtype == bfloat16:
 223           return _np_bfloat16(float.fromhex("0x1.FEp127"))
 224         raise TypeError("Cannot find maximum value of %s." % self)
 225 
 226   @property
 227   def limits(self, clip_negative=True):
 228     """Return intensity limits, i.e. (min, max) tuple, of the dtype.
 229     Args:
 230       clip_negative : bool, optional
 231           If True, clip the negative range (i.e. return 0 for min intensity)
 232           even if the image dtype allows negative values.
 233     Returns
 234       min, max : tuple
 235         Lower and upper intensity limits.
 236     """
 237     min, max = dtype_range[self.as_numpy_dtype]  # pylint: disable=redefined-builtin
 238     if clip_negative:
 239       min = 0  # pylint: disable=redefined-builtin
 240     return min, max
 241 
 242   def is_compatible_with(self, other):
 243     """Returns True if the `other` DType will be converted to this DType.
 244 
 245     The conversion rules are as follows:
 246 
 247     ```python
 248     DType(T)       .is_compatible_with(DType(T))        == True
 249     DType(T)       .is_compatible_with(DType(T).as_ref) == True
 250     DType(T).as_ref.is_compatible_with(DType(T))        == False
 251     DType(T).as_ref.is_compatible_with(DType(T).as_ref) == True
 252     ```
 253 
 254     Args:
 255       other: A `DType` (or object that may be converted to a `DType`).
 256 
 257     Returns:
 258       True if a Tensor of the `other` `DType` will be implicitly converted to
 259       this `DType`.
 260     """
 261     other = as_dtype(other)
 262     return self._type_enum in (other.as_datatype_enum,
 263                                other.base_dtype.as_datatype_enum)
 264 
 265   def __eq__(self, other):
 266     """Returns True iff this DType refers to the same type as `other`."""
 267     if other is None:
 268       return False
 269     try:
 270       dtype = as_dtype(other).as_datatype_enum
 271       return self._type_enum == dtype  # pylint: disable=protected-access
 272     except TypeError:
 273       return False
 274 
 275   def __ne__(self, other):
 276     """Returns True iff self != other."""
 277     return not self.__eq__(other)
 278 
 279   @property
 280   def name(self):
 281     """Returns the string name for this `DType`."""
 282     return _TYPE_TO_STRING[self._type_enum]
 283 
 284   def __int__(self):
 285     return self._type_enum
 286 
 287   def __str__(self):
 288     return "<dtype: %r>" % self.name
 289 
 290   def __repr__(self):
 291     return "tf." + self.name
 292 
 293   def __hash__(self):
 294     return self._type_enum
 295 
 296   def __reduce__(self):
 297     return as_dtype, (self.name,)
 298 
 299   @property
 300   def size(self):
 301     if (self._type_enum == types_pb2.DT_VARIANT or
 302         self._type_enum == types_pb2.DT_RESOURCE):
 303       return 1
 304     return np.dtype(self.as_numpy_dtype).itemsize

 661 @tf_export("dtypes.as_dtype", "as_dtype")
 662 def as_dtype(type_value):
 663   """Converts the given `type_value` to a `DType`.
 664 
 665   Args:
 666     type_value: A value that can be converted to a `tf.DType` object. This may
 667       currently be a `tf.DType` object, a [`DataType`
 668       enum](https://www.tensorflow.org/code/tensorflow/core/framework/types.proto),
 669       a string type name, or a `numpy.dtype`.
 670 
 671   Returns:
 672     A `DType` corresponding to `type_value`.
 673 
 674   Raises:
 675     TypeError: If `type_value` cannot be converted to a `DType`.
 676   """
 677   if isinstance(type_value, DType):
 678     return type_value
 679 
 680   try:
 681     return _INTERN_TABLE[type_value]
 682   except KeyError:
 683     pass
 684 
 685   try:
 686     return _STRING_TO_TF[type_value]
 687   except KeyError:
 688     pass
 689 
 690   try:
 691     return _PYTHON_TO_TF[type_value]
 692   except KeyError:
 693     pass
 694 
 695   if isinstance(type_value, np.dtype):
 696     # The numpy dtype for strings is variable length. We can not compare
 697     # dtype with a single constant (np.string does not exist) to decide
 698     # dtype is a "string" type. We need to compare the dtype.type to be
 699     # sure it's a string type.
 700     if type_value.type == np.string_ or type_value.type == np.unicode_:
 701       return string
 702 
 703   if isinstance(type_value, (type, np.dtype)):
 704     for key, val in _NP_TO_TF:
 705       try:
 706         if key == type_value:
 707           return val
 708       except TypeError as e:
 709         raise TypeError("Cannot convert {} to a dtype. {}".format(
 710             type_value, e))
 711 
 712   raise TypeError("Cannot convert value %r to a TensorFlow DType." % type_value)
