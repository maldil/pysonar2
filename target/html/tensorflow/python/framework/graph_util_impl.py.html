  53 @tf_export("graph_util.must_run_on_cpu")
  54 def must_run_on_cpu(node, pin_variables_on_cpu=False):
  55   """Returns True if the given node_def must run on CPU, otherwise False.
  56 
  57   Args:
  58     node: The node to be assigned to a device. Could be either an ops.Operation
  59       or NodeDef.
  60     pin_variables_on_cpu: If True, this function will return False if node_def
  61       represents a variable-related op.
  62 
  63   Returns:
  64     True if the given node must run on CPU, otherwise False.
  65   """
  66 
  67   if isinstance(node, ops.Operation):
  68     node_def = node.node_def
  69   else:
  70     assert isinstance(node, node_def_pb2.NodeDef)
  71     node_def = node
  72 
  73   # If the op is a variable-related op, should we pin it on CPU?
  74   if pin_variables_on_cpu and _is_variable_op(node_def.op):
  75     return True
  76 
  77   # Constant operations producing a string or int32 must run on CPU.
  78   if node_def.op == "Const":
  79     # Get the value of the 'dtype' attr
  80     dtype = node_def.attr["dtype"].type
  81     if dtype == dtypes.string or dtype == dtypes.int32:
  82       return True
  83 
  84   if node_def.op in ["DynamicStitch", "ParallelDynamicStitch"]:
  85     dtype = node_def.attr["T"].type
  86     if dtype == dtypes.int32:
  87       # DynamicStitch on GPU only works for int32 values.
  88       return True
  89 
  90   if node_def.op in ["Cast"]:
  91     dtype = node_def.attr["SrcT"].type
  92     if dtype == dtypes.int32:
  93       # Cast on GPU does not works for int32 values.
  94       return True
  95   return False

 105 def _node_name(n):
 106   if n.startswith("^"):
 107     return n[1:]
 108   else:
 109     return n.split(":")[0]

 112 def _extract_graph_summary(graph_def):
 113   """Extracts useful information from the graph and returns them."""
 114   name_to_input_name = {}  # Keyed by the dest node name.
 115   name_to_node = {}  # Keyed by node name.
 116 
 117   # Keeps track of node sequences. It is important to still output the
 118   # operations in the original order.
 119   name_to_seq_num = {}  # Keyed by node name.
 120   seq = 0
 121   for node in graph_def.node:
 122     n = _node_name(node.name)
 123     name_to_node[n] = node
 124     name_to_input_name[n] = [_node_name(x) for x in node.input]
 125     name_to_seq_num[n] = seq
 126     seq += 1
 127   return name_to_input_name, name_to_node, name_to_seq_num

 130 def _assert_nodes_are_present(name_to_node, nodes):
 131   """Assert that nodes are present in the graph."""
 132   for d in nodes:
 133     assert d in name_to_node, "%s is not in graph" % d

 136 def _bfs_for_reachable_nodes(target_nodes, name_to_input_name):
 137   """Breadth first search for reachable nodes from target nodes."""
 138   nodes_to_keep = set()
 139   # Breadth first search to find all the nodes that we should keep.
 140   next_to_visit = target_nodes[:]
 141   while next_to_visit:
 142     n = next_to_visit[0]
 143     del next_to_visit[0]
 144     if n in nodes_to_keep:
 145       # Already visited this node.
 146       continue
 147     nodes_to_keep.add(n)
 148     next_to_visit += name_to_input_name[n]
 149   return nodes_to_keep

 152 @tf_export("graph_util.extract_sub_graph")
 153 def extract_sub_graph(graph_def, dest_nodes):
 154   """Extract the subgraph that can reach any of the nodes in 'dest_nodes'.
 155 
 156   Args:
 157     graph_def: A graph_pb2.GraphDef proto.
 158     dest_nodes: A list of strings specifying the destination node names.
 159   Returns:
 160     The GraphDef of the sub-graph.
 161 
 162   Raises:
 163     TypeError: If 'graph_def' is not a graph_pb2.GraphDef proto.
 164   """
 165 
 166   if not isinstance(graph_def, graph_pb2.GraphDef):
 167     raise TypeError("graph_def must be a graph_pb2.GraphDef proto.")
 168 
 169   if isinstance(dest_nodes, six.string_types):
 170     raise TypeError("dest_nodes must be a list.")
 171 
 172   name_to_input_name, name_to_node, name_to_seq_num = _extract_graph_summary(
 173       graph_def)
 174   _assert_nodes_are_present(name_to_node, dest_nodes)
 175 
 176   nodes_to_keep = _bfs_for_reachable_nodes(dest_nodes, name_to_input_name)
 177 
 178   nodes_to_keep_list = sorted(
 179       list(nodes_to_keep), key=lambda n: name_to_seq_num[n])
 180   # Now construct the output GraphDef
 181   out = graph_pb2.GraphDef()
 182   for n in nodes_to_keep_list:
 183     out.node.extend([copy.deepcopy(name_to_node[n])])
 184   out.library.CopyFrom(graph_def.library)
 185   out.versions.CopyFrom(graph_def.versions)
 186 
 187   return out

 190 @tf_export("graph_util.tensor_shape_from_node_def_name")
 191 def tensor_shape_from_node_def_name(graph, input_name):
 192   """Convenience function to get a shape from a NodeDef's input string."""
 193   # To get a tensor, the name must be in the form <input>:<port>, for example
 194   # 'Mul:0'. The GraphDef input strings don't always have the port specified
 195   # though, so if there isn't a colon we need to add a default ':0' to the end.
 196   if ":" not in input_name:
 197     canonical_name = input_name + ":0"
 198   else:
 199     canonical_name = input_name
 200   tensor = graph.get_tensor_by_name(canonical_name)
 201   shape = tensor.get_shape()
 202   return shape

 205 @tf_export("graph_util.convert_variables_to_constants")
 206 def convert_variables_to_constants(sess,
 207                                    input_graph_def,
 208                                    output_node_names,
 209                                    variable_names_whitelist=None,
 210                                    variable_names_blacklist=None):
 211   """Replaces all the variables in a graph with constants of the same values.
 212 
 213   If you have a trained graph containing Variable ops, it can be convenient to
 214   convert them all to Const ops holding the same values. This makes it possible
 215   to describe the network fully with a single GraphDef file, and allows the
 216   removal of a lot of ops related to loading and saving the variables.
 217 
 218   Args:
 219     sess: Active TensorFlow session containing the variables.
 220     input_graph_def: GraphDef object holding the network.
 221     output_node_names: List of name strings for the result nodes of the graph.
 222     variable_names_whitelist: The set of variable names to convert (by default,
 223                               all variables are converted).
 224     variable_names_blacklist: The set of variable names to omit converting
 225                               to constants.
 226 
 227   Returns:
 228     GraphDef containing a simplified version of the original.
 229   """
 230   # This graph only includes the nodes needed to evaluate the output nodes, and
 231   # removes unneeded nodes like those involved in saving and assignment.
 232   inference_graph = extract_sub_graph(input_graph_def, output_node_names)
 233 
 234   found_variables = {}
 235   variable_names = []
 236   variable_dict_names = []
 237   for node in inference_graph.node:
 238     if node.op in ["Variable", "VariableV2", "VarHandleOp"]:
 239       variable_name = node.name
 240       if ((variable_names_whitelist is not None and
 241            variable_name not in variable_names_whitelist) or
 242           (variable_names_blacklist is not None and
 243            variable_name in variable_names_blacklist)):
 244         continue
 245       variable_dict_names.append(variable_name)
 246       if node.op == "VarHandleOp":
 247         variable_names.append(variable_name + "/Read/ReadVariableOp:0")
 248       else:
 249         variable_names.append(variable_name + ":0")
 250   if variable_names:
 251     returned_variables = sess.run(variable_names)
 252   else:
 253     returned_variables = []
 254   found_variables = dict(zip(variable_dict_names, returned_variables))
 255   logging.info("Froze %d variables.", len(returned_variables))
 256 
 257   output_graph_def = graph_pb2.GraphDef()
 258   how_many_converted = 0
 259   for input_node in inference_graph.node:
 260     output_node = node_def_pb2.NodeDef()
 261     if input_node.name in found_variables:
 262       output_node.op = "Const"
 263       output_node.name = input_node.name
 264       dtype = input_node.attr["dtype"]
 265       data = found_variables[input_node.name]
 266       output_node.attr["dtype"].CopyFrom(dtype)
 267       output_node.attr["value"].CopyFrom(
 268           attr_value_pb2.AttrValue(
 269               tensor=tensor_util.make_tensor_proto(
 270                   data, dtype=dtype.type, shape=data.shape)))
 271       how_many_converted += 1
 272     elif input_node.op == "ReadVariableOp" and (
 273         input_node.input[0] in found_variables):
 274       # The preceding branch converts all VarHandleOps of ResourceVariables to
 275       # constants, so we need to convert the associated ReadVariableOps to
 276       # Identity ops.
 277       output_node.op = "Identity"
 278       output_node.name = input_node.name
 279       output_node.input.extend([input_node.input[0]])
 280       output_node.attr["T"].CopyFrom(input_node.attr["dtype"])
 281       if "_class" in input_node.attr:
 282         output_node.attr["_class"].CopyFrom(input_node.attr["_class"])
 283     else:
 284       output_node.CopyFrom(input_node)
 285     output_graph_def.node.extend([output_node])
 286 
 287   output_graph_def.library.CopyFrom(inference_graph.library)
 288   logging.info("Converted %d variables to const ops.", how_many_converted)
 289   return output_graph_def

 292 @tf_export("graph_util.remove_training_nodes")
 293 def remove_training_nodes(input_graph, protected_nodes=None):
 294   """Prunes out nodes that aren't needed for inference.
 295 
 296   There are nodes like Identity and CheckNumerics that are only useful
 297   during training, and can be removed in graphs that will be used for
 298   nothing but inference. Here we identify and remove them, returning an
 299   equivalent graph. To be specific, CheckNumerics nodes are always removed, and
 300   Identity nodes that aren't involved in control edges are spliced out so that
 301   their input and outputs are directly connected.
 302 
 303   Args:
 304     input_graph: Model to analyze and prune.
 305     protected_nodes: An optional list of names of nodes to be kept
 306       unconditionally. This is for example useful to preserve Identity output
 307       nodes.
 308 
 309   Returns:
 310     A list of nodes with the unnecessary ones removed.
 311   """
 312   if not protected_nodes:
 313     protected_nodes = []
 314 
 315   types_to_remove = {"CheckNumerics": True}
 316 
 317   input_nodes = input_graph.node
 318   names_to_remove = {}
 319   for node in input_nodes:
 320     if node.op in types_to_remove and node.name not in protected_nodes:
 321       names_to_remove[node.name] = True
 322 
 323   nodes_after_removal = []
 324   for node in input_nodes:
 325     if node.name in names_to_remove:
 326       continue
 327     new_node = node_def_pb2.NodeDef()
 328     new_node.CopyFrom(node)
 329     input_before_removal = node.input
 330     del new_node.input[:]
 331     for full_input_name in input_before_removal:
 332       input_name = re.sub(r"^\^", "", full_input_name)
 333       if input_name in names_to_remove:
 334         continue
 335       new_node.input.append(full_input_name)
 336     nodes_after_removal.append(new_node)
 337 
 338   types_to_splice = {"Identity": True}
 339   names_to_splice = {}
 340   for node in nodes_after_removal:
 341     if node.op in types_to_splice and node.name not in protected_nodes:
 342       # We don't want to remove nodes that have control edge inputs, because
 343       # they might be involved in subtle dependency issues that removing them
 344       # will jeopardize.
 345       has_control_edge = False
 346       for input_name in node.input:
 347         if re.match(r"^\^", input_name):
 348           has_control_edge = True
 349       if not has_control_edge:
 350         names_to_splice[node.name] = node.input[0]
 351 
 352   nodes_after_splicing = []
 353   for node in nodes_after_removal:
 354     if node.name in names_to_splice:
 355       continue
 356     new_node = node_def_pb2.NodeDef()
 357     new_node.CopyFrom(node)
 358     input_before_removal = node.input
 359     del new_node.input[:]
 360     for full_input_name in input_before_removal:
 361       input_name = re.sub(r"^\^", "", full_input_name)
 362       while input_name in names_to_splice:
 363         full_input_name = names_to_splice[input_name]
 364         input_name = re.sub(r"^\^", "", full_input_name)
 365       new_node.input.append(full_input_name)
 366     nodes_after_splicing.append(new_node)
 367 
 368   output_graph = graph_pb2.GraphDef()
 369   output_graph.node.extend(nodes_after_splicing)
 370   return output_graph
