  29 class TensorSpec(object):
  30   """Describes a tf.Tensor.
  31 
  32   A TensorSpec allows an API to describe the Tensors that it accepts or
  33   returns, before that Tensor exists. This allows dynamic and flexible graph
  34   construction and configuration.
  35   """
  36 
  37   __slots__ = ["_shape", "_shape_tuple", "_dtype", "_name"]
  38 
  39   def __init__(self, shape, dtype, name=None):
  40     """Creates a TensorSpec.
  41 
  42     Args:
  43       shape: Value convertible to `tf.TensorShape`. The shape of the tensor.
  44       dtype: Value convertible to `tf.DType`. The type of the tensor values.
  45       name: Optional name for the Tensor.
  46 
  47     Raises:
  48       TypeError: If shape is not convertible to a `tf.TensorShape`, or dtype is
  49         not convertible to a `tf.DType`.
  50     """
  51     self._shape = tensor_shape.TensorShape(shape)
  52     try:
  53       self._shape_tuple = tuple(self.shape.as_list())
  54     except ValueError:
  55       self._shape_tuple = None
  56     self._dtype = dtypes.as_dtype(dtype)
  57     self._name = name
  58 
  59   @classmethod
  60   def from_spec(cls, spec, name=None):
  61     return cls(spec.shape, spec.dtype, name or spec.name)
  62 
  63   @classmethod
  64   def from_tensor(cls, tensor, name=None):
  65     if isinstance(tensor, ops.EagerTensor):
  66       return TensorSpec(tensor.shape, tensor.dtype, name)
  67     elif isinstance(tensor, ops.Tensor):
  68       return TensorSpec(tensor.shape, tensor.dtype, name or tensor.op.name)
  69     else:
  70       raise ValueError("`tensor` should be a tf.Tensor")
  71 
  72   @classmethod
  73   def is_bounded(cls):
  74     del cls
  75     return False
  76 
  77   @property
  78   def shape(self):
  79     """Returns the `TensorShape` that represents the shape of the tensor."""
  80     return self._shape
  81 
  82   @property
  83   def dtype(self):
  84     """Returns the `dtype` of elements in the tensor."""
  85     return self._dtype
  86 
  87   @property
  88   def name(self):
  89     """Returns the name of the described tensor."""
  90     return self._name
  91 
  92   @property
  93   def is_discrete(self):
  94     """Whether spec is discrete."""
  95     return self.dtype.is_integer
  96 
  97   @property
  98   def is_continuous(self):
  99     """Whether spec is continuous."""
 100     return self.dtype.is_floating
 101 
 102   def is_compatible_with(self, spec_or_tensor):
 103     """True if the shape and dtype of `spec_or_tensor` are compatible."""
 104     return (self._dtype.is_compatible_with(spec_or_tensor.dtype) and
 105             self._shape.is_compatible_with(spec_or_tensor.shape))
 106 
 107   def __repr__(self):
 108     return "TensorSpec(shape={}, dtype={}, name={})".format(
 109         self.shape, repr(self.dtype), repr(self.name))
 110 
 111   def __hash__(self):
 112     return hash((self._shape_tuple, self.dtype))
 113 
 114   def __eq__(self, other):
 115     return self.shape == other.shape and self.dtype == other.dtype
 116 
 117   def __ne__(self, other):
 118     return not self == other
 119 
 120   def __reduce__(self):
 121     return TensorSpec, (self._shape, self._dtype, self._name)

 124 class BoundedTensorSpec(TensorSpec):
 125   """A `TensorSpec` that specifies minimum and maximum values.
 126 
 127   Example usage:
 128   ```python
 129   spec = tensor_spec.BoundedTensorSpec((1, 2, 3), tf.float32, 0, (5, 5, 5))
 130   tf_minimum = tf.convert_to_tensor(spec.minimum, dtype=spec.dtype)
 131   tf_maximum = tf.convert_to_tensor(spec.maximum, dtype=spec.dtype)
 132   ```
 133 
 134   Bounds are meant to be inclusive. This is especially important for
 135   integer types. The following spec will be satisfied by tensors
 136   with values in the set {0, 1, 2}:
 137   ```python
 138   spec = tensor_spec.BoundedTensorSpec((3, 5), tf.int32, 0, 2)
 139   ```
 140   """
 141 
 142   __slots__ = ("_minimum", "_maximum")
 143 
 144   def __init__(self, shape, dtype, minimum, maximum, name=None):
 145     """Initializes a new `BoundedTensorSpec`.
 146 
 147     Args:
 148       shape: Value convertible to `tf.TensorShape`. The shape of the tensor.
 149       dtype: Value convertible to `tf.DType`. The type of the tensor values.
 150       minimum: Number or sequence specifying the minimum element bounds
 151         (inclusive). Must be broadcastable to `shape`.
 152       maximum: Number or sequence specifying the maximum element bounds
 153         (inclusive). Must be broadcastable to `shape`.
 154       name: Optional string containing a semantic name for the corresponding
 155         array. Defaults to `None`.
 156 
 157     Raises:
 158       ValueError: If `minimum` or `maximum` are not provided or not
 159         broadcastable to `shape`.
 160       TypeError: If the shape is not an iterable or if the `dtype` is an invalid
 161         numpy dtype.
 162     """
 163     super(BoundedTensorSpec, self).__init__(shape, dtype, name)
 164 
 165     if minimum is None or maximum is None:
 166       raise ValueError("minimum and maximum must be provided; but saw "
 167                        "'%s' and '%s'" % (minimum, maximum))
 168 
 169     try:
 170       minimum_shape = np.shape(minimum)
 171       common_shapes.broadcast_shape(
 172           tensor_shape.TensorShape(minimum_shape), self.shape)
 173     except ValueError as exception:
 174       raise ValueError("minimum is not compatible with shape. "
 175                        "Message: {!r}.".format(exception))
 176 
 177     try:
 178       maximum_shape = np.shape(maximum)
 179       common_shapes.broadcast_shape(
 180           tensor_shape.TensorShape(maximum_shape), self.shape)
 181     except ValueError as exception:
 182       raise ValueError("maximum is not compatible with shape. "
 183                        "Message: {!r}.".format(exception))
 184 
 185     self._minimum = np.array(minimum, dtype=self.dtype.as_numpy_dtype())
 186     self._minimum.setflags(write=False)
 187 
 188     self._maximum = np.array(maximum, dtype=self.dtype.as_numpy_dtype())
 189     self._maximum.setflags(write=False)
 190 
 191   @classmethod
 192   def is_bounded(cls):
 193     del cls
 194     return True
 195 
 196   @classmethod
 197   def from_spec(cls, spec):
 198     dtype = dtypes.as_dtype(spec.dtype)
 199     minimum = getattr(spec, "minimum", dtype.min)
 200     maximum = getattr(spec, "maximum", dtype.max)
 201     return BoundedTensorSpec(spec.shape, dtype, minimum, maximum, spec.name)
 202 
 203   @property
 204   def minimum(self):
 205     """Returns a NumPy array specifying the minimum bounds (inclusive)."""
 206     return self._minimum
 207 
 208   @property
 209   def maximum(self):
 210     """Returns a NumPy array specifying the maximum bounds (inclusive)."""
 211     return self._maximum
 212 
 213   def __repr__(self):
 214     s = "BoundedTensorSpec(shape={}, dtype={}, name={}, minimum={}, maximum={})"
 215     return s.format(self.shape, repr(self.dtype), repr(self.name),
 216                     repr(self.minimum), repr(self.maximum))
 217 
 218   def __eq__(self, other):
 219     tensor_spec_eq = super(BoundedTensorSpec, self).__eq__(other)
 220     return (tensor_spec_eq and np.allclose(self.minimum, other.minimum) and
 221             np.allclose(self.maximum, other.maximum))
 222 
 223   def __reduce__(self):
 224     return BoundedTensorSpec, (self._shape, self._dtype, self._minimum,
 225                                self._maximum, self._name)
