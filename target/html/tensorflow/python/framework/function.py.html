  44 class Defun(object):
  45   """Decorator used to define TensorFlow functions.
  46 
  47   Use this decorator to make a Python function usable directly as a TensorFlow
  48   function.
  49 
  50   The decorated function must add ops to the default graph and return zero or
  51   more `Tensor` objects.  Call the decorator with named arguments, one for each
  52   argument of the function to decorate, with the expected type of the argument
  53   as value.
  54 
  55   For example if the function to decorate accepts two `tf.float32` arguments
  56   named `x` and `y`, call the decorator with:
  57 
  58       @Defun(tf.float32, tf.float32)
  59       def foo(x, y):
  60         ...
  61 
  62   When you call the decorated function it will add `call` ops to the
  63   default graph and adds the definition of the function into the
  64   default graph. Because the addition of the function into the graph
  65   is deferred, the decorator can be used anywhere in the program.
  66 
  67   Any variables created inside of the function are hoisted into the outer graph.
  68   Note that the variables are created in the variable scope that was active
  69   during the first call to the function. Subsequent function calls will refer to
  70   the same set of variables.
  71 
  72   Definitions of functions in a graph are frozen as soon as the graph is used to
  73   create a session. However, new functions and new calls to existing functions
  74   may be added to the graph, with the new functions themselves becoming
  75   immediately frozen.
  76 
  77   Example, but also see the [How To on functions](link_needed).
  78 
  79   ```python
  80   # Defining the function.
  81   @tf.Defun(tf.float32, tf.float32)
  82   def MyFunc(x, y):
  83     return x + y, x - y
  84 
  85   # Building the graph.
  86   a = tf.constant([1.0])
  87   b = tf.constant([2.0])
  88   c, d = MyFunc(a, b, name='mycall')
  89   ```
  90   """
  91 
  92   def __init__(self, *input_types, **kwargs):
  93     """Create a `Defun` decorator.
  94 
  95     Args:
  96       *input_types: A list of `tf.DType`
  97       **kwargs: Optional keyword arguments, including
  98          func_name - (optional).  A python string, the name to use to
  99            declare this `Function` in the graph.
 100 
 101          grad_func - (optional).  A function implementing the gradient
 102            of the function-to-register.  This is must be a
 103            `_DefinedFunction` object. The gradient
 104            function must satisfy the criterion defined in
 105            function.proto:GradientDef.
 106 
 107          python_grad_func - (optional).  A function implementing the
 108            gradient of the function python-side. This function must
 109            take the current op and the gradients w.r.t. its outputs,
 110            and return the gradients w.r.t. the inputs. That is it must
 111            implement the interface expected by `tf.RegisterGradient`).
 112            This will be called by tf.gradients to add the gradient ops
 113            to the graph. At most one of grad_func and python_grad_func
 114            can be specified.
 115 
 116          out_names = (optional). A list of strings, one per output
 117            tensor.
 118 
 119          shape_func - (optional). A function taking the op and returning a list
 120            of static shapes to set for the function's outputs.
 121     """
 122     self._input_types = input_types
 123     self._func_name = kwargs.pop("func_name", None)
 124     self._grad_func = kwargs.pop("grad_func", None)
 125     self._python_grad_func = kwargs.pop("python_grad_func", None)
 126     self._out_names = kwargs.pop("out_names", None)
 127     self._extra_kwargs = kwargs
 128 
 129   def __call__(self, func):
 130     # Various sanity checks on the callable func.
 131     if not callable(func):
 132       raise ValueError("func %s must be callable" % func)
 133 
 134     # Func should not use kwargs and defaults.
 135     argspec = tf_inspect.getargspec(func)
 136     if argspec.keywords or argspec.defaults:
 137       raise ValueError("Functions with argument defaults or keyword "
 138                        "arguments are not supported.")
 139 
 140     # Computes how many arguments 'func' has.
 141     min_args = len(argspec.args)
 142     max_args = min_args
 143     if argspec.varargs:
 144       max_args = 1000000
 145     argnames = argspec.args
 146     if tf_inspect.ismethod(func):
 147       # 1st argument is the "class" type.
 148       min_args -= 1
 149       argnames = argnames[1:]
 150 
 151     if self._input_types:
 152       # If Defun is given a list of types for the inputs, the number
 153       # of input types should be compatible with 'func'.
 154       num = len(self._input_types)
 155       if num < min_args or num > max_args:
 156         raise ValueError(
 157             "The function has fewer arguments than the number of specified "
 158             "input types.")
 159       return _DefinedFunction(
 160           func,
 161           argnames,
 162           self._input_types,
 163           self._func_name,
 164           self._grad_func,
 165           self._python_grad_func,
 166           out_names=self._out_names,
 167           **self._extra_kwargs)
 168 
 169     # 'func' expects no arguments and input types is an empty list.
 170     if min_args == 0 and max_args == 0:
 171       return _DefinedFunction(
 172           func, [], [],
 173           self._func_name,
 174           self._grad_func,
 175           self._python_grad_func,
 176           out_names=self._out_names,
 177           **self._extra_kwargs)
 178 
 179     # Input types are unknown. It's an overloaded function and hence
 180     # its definition needs to be deferred until it's called.
 181     return _OverloadedFunction(
 182         func,
 183         argnames,
 184         self._func_name,
 185         self._grad_func,
 186         self._python_grad_func,
 187         out_names=self._out_names,
 188         **self._extra_kwargs)

 191 class _DefinedFunction(object):
 192   """_DefinedFunction encapsulates a function definition and its properties.
 193 
 194   Attributes:
 195     name: The function name.
 196     definition: The definition of this function. A FunctionDef proto.
 197     grad_func_name: If not None, the name of this function's gradient function.
 198     python_grad_func: A python callable implementing the gradient of
 199       the function python-side.
 200   """
 201 
 202   def __init__(self,
 203                func,
 204                argnames,
 205                input_types,
 206                func_name=None,
 207                grad_func=None,
 208                python_grad_func=None,
 209                out_names=None,
 210                shape_func=None,
 211                capture_by_value=False,
 212                **kwargs):
 213     """Creates _DefinedFunction.
 214 
 215     Args:
 216       func:  A python callable which constructs a tf function body.
 217       argnames: A list of strings for function argument names.
 218       input_types: The function's argument types. Can be a tuple, list of
 219         tf data types.
 220       func_name: The function name. Defaults to None, in which derives from
 221         'func'.
 222       grad_func: This function's gradient function, if not None. Defaults
 223         to None.
 224       python_grad_func: A python callable implementing the gradient of
 225         the function python-side.
 226       out_names: An optional list of strings for the function return value
 227         names.
 228       shape_func: An optional function mapping an op to a list of static
 229         output shapes.
 230       capture_by_value: Boolean (defaults to False). If True, captured values
 231         will be copied into the function body.
 232       **kwargs: The keyword arguments. **kwargs is passed to every call
 233         site of this function.
 234 
 235     Raises:
 236       ValueError: The function definition is invalid.
 237 
 238     """
 239     self._func = func
 240     self._input_types = input_types
 241     self._func_name = func_name
 242     self._grad_func = grad_func
 243     self._python_grad_func = python_grad_func
 244     self._out_names = out_names
 245     self._shape_func = shape_func
 246     self._capture_by_value = capture_by_value
 247     self._extra_kwargs = kwargs
 248     # Constructed only when C API is disabled, lazily
 249     self._definition = None
 250     # Constructed only when C API is enabled, lazily
 251     self._c_func = None
 252     self._sub_functions = dict()  # Constructed with _definition or _c_func
 253     # pylint: disable=protected-access
 254     device_funcs = ops.get_default_graph()._device_functions_outer_to_inner
 255     # pylint: enable=protected-access
 256 
 257     # Get the innermost device if possbile.
 258     self._caller_device = device_funcs[-1] if device_funcs else None
 259 
 260     # Cached OpDef for this function. When C API is enabled, this is
 261     # the only part of FunctionDef that we cache in Python. When C API
 262     # is disabled the whole _definition is available and this is simply
 263     # another reference to _definition.signature
 264     self._op_def = None
 265 
 266     assert isinstance(input_types, (list, tuple))
 267     self._arg_types = input_types
 268     self._arg_names = [argnames[i] if i < len(argnames) else ("arg%d" % i)
 269                        for i in range(len(input_types))]
 270 
 271   @property
 272   def name(self):
 273     """Function name."""
 274     self._create_definition_if_needed()
 275     return self._func_name
 276 
 277   @property
 278   def definition(self):
 279     """Function definition proto."""
 280     self._create_definition_if_needed()
 281     if self._c_func:
 282       with c_api_util.tf_buffer() as buf:
 283         c_api.TF_FunctionToFunctionDef(self._c_func.func, buf)
 284         fdef = function_pb2.FunctionDef()
 285         proto_data = c_api.TF_GetBuffer(buf)
 286         fdef.ParseFromString(compat.as_bytes(proto_data))
 287       return fdef
 288     return self._definition
 289 
 290   @property
 291   def _signature(self):
 292     self._create_definition_if_needed()
 293     return self._op_def

 332   def _create_definition_if_needed(self):
 333     """Creates the function definition if it's not created yet."""
 334     with context.graph_mode():
 335       self._create_definition_if_needed_impl()
 336 
 337   def _create_definition_if_needed_impl(self):
 338     """This is not what you want, see _create_definition_if_needed."""
 339     if self._definition is not None or self._c_func is not None:
 340       return
 341 
 342     temp_graph = func_graph_from_py_func(
 343         self._func, self._arg_names, self._arg_types, self._func_name,
 344         self._capture_by_value, self._caller_device)
 345 
 346     self._extra_inputs = temp_graph.extra_inputs
 347     # pylint: disable=protected-access
 348     self._sub_functions = temp_graph._functions
 349     # pylint: enable=protected-access
 350 
 351     # Extra kwargs are treated as attrs on the function def.
 352     if self._func_name:
 353       base_func_name = self._func_name
 354     else:
 355       base_func_name = function_utils.get_func_name(self._func)
 356       if self._grad_func:
 357         base_func_name += ("_%s" % self._grad_func.name)
 358     kwargs_attr = _parse_kwargs_as_attrs(base_func_name, **self._extra_kwargs)
 359 
 360     if not temp_graph._c_graph:  # pylint: disable=protected-access
 361       # Build the FunctionDef
 362       self._definition = graph_to_function_def.graph_to_function_def(
 363           temp_graph,
 364           temp_graph.get_operations(),
 365           temp_graph.inputs,
 366           temp_graph.outputs,
 367           out_names=self._out_names)
 368 
 369       for k in kwargs_attr:
 370         self._definition.attr[k].CopyFrom(kwargs_attr[k])
 371 
 372       # Hash the definition and its dependencies.
 373       self._hash_str = self._create_hash_str(
 374           self._definition.signature.input_arg,
 375           self._definition.signature.output_arg, self._definition.node_def)
 376 
 377       # Finally, we decide the function name to use.  If not specified,
 378       # make up something which is almost certainly unique (but deterministic).
 379       if not self._func_name:
 380         self._func_name = "_".join([base_func_name, self._hash_str])
 381       self._definition.signature.name = self._func_name
 382       if self._func.__doc__:
 383         self._definition.signature.description = self._func.__doc__
 384 
 385       self._op_def = self._definition.signature
 386     else:  # C API is enabled
 387       output_names = ([compat.as_bytes(x) for x in self._out_names]
 388                       if self._out_names else [])
 389       description = self._func.__doc__ or None
 390       # pylint: disable=protected-access
 391       c_func = c_api.TF_GraphToFunction_wrapper(
 392           temp_graph._c_graph,
 393           base_func_name,
 394           self._func_name is None,  # append_hash_to_fn_name
 395           None,  # opers
 396           [t._as_tf_output() for t in temp_graph.inputs],
 397           [t._as_tf_output() for t in temp_graph.outputs],
 398           output_names,
 399           None,  # opts
 400           description)
 401       self._c_func = c_api_util.ScopedTFFunction(c_func)
 402       # pylint: enable=protected-access
 403       self._set_c_attrs(kwargs_attr)
 404 
 405       # Set cached fields: _op_def and _func_name (if not already set)
 406       self._op_def = self.definition.signature
 407       if self._func_name:
 408         assert self._func_name == self._op_def.name
 409       else:
 410         self._func_name = compat.as_str(self._op_def.name)
 411 
 412     self._stateful_ops = [(op.name, op.type)
 413                           for op in temp_graph.get_operations()
 414                           if op.op_def.is_stateful]
 415 
 416   def _set_c_attrs(self, attrs):
 417     """Sets `attrs` as attributes of self._c_func.
 418 
 419     Requires that self._c_func is not None.
 420 
 421     Args:
 422       attrs: a dictionary from attribute name to attribute proto value
 423     """
 424     for name, attr_value in attrs.items():
 425       serialized = attr_value.SerializeToString()
 426       # TODO(skyewm): this creates and deletes a new TF_Status for every attr.
 427       # It might be worth creating a convenient way to re-use the same status.
 428       c_api.TF_FunctionSetAttrValueProto(self._c_func.func, compat.as_str(name),
 429                                          serialized)

 431   def _create_hash_str(self, input_arg, output_arg, node_def):
 432     """Creates an 8-character string unique to this input.
 433 
 434     Args:
 435       input_arg: the input_arg field of an OpDef
 436                  (e.g. self._definition.signature.input_arg)
 437       output_arg: the output_arg field of an OpDef
 438                  (e.g. self._definition.signature.output_arg)
 439       node_def: the node_def field of a FunctionDef
 440                 (e.g. self._definition.node_def)
 441 
 442     Returns:
 443       The unique string for this input
 444     """
 445     hasher = hashlib.sha1()
 446 
 447     def update_num(n):
 448       hasher.update(compat.as_bytes("%x" % n))
 449 
 450     def update_str(s):
 451       update_num(len(s))
 452       hasher.update(compat.as_bytes(s))
 453 
 454     def update_strs(slist):
 455       update_num(len(slist))
 456       for s in slist:
 457         update_str(s)
 458 
 459     for adef in input_arg:
 460       update_str(adef.SerializeToString())
 461 
 462     for adef in output_arg:
 463       update_str(adef.SerializeToString())
 464 
 465     for n in sorted(node_def, key=lambda n: n.name):
 466       update_str(n.name)
 467       update_str(n.op)
 468       update_strs(n.input)
 469       update_num(len(n.attr))
 470       # NOTE: protobuf map serialization does not guarantee ordering.
 471       for k in sorted(n.attr):
 472         update_str(k)
 473         update_str(n.attr[k].SerializeToString())
 474 
 475     return hasher.hexdigest()[:8]
 476 
 477   def add_to_graph(self, g):
 478     """Adds this function into the graph g."""
 479     self._create_definition_if_needed()
 480 
 481     # Adds this function into 'g'.
 482     # pylint: disable=protected-access
 483     if context.executing_eagerly():
 484       context.context().add_function_def(self.definition)
 485     else:
 486       g._add_function(self)
 487     # pylint: enable=protected-access
 488 
 489     # Ensures related sub-routines are defined in 'g', too.
 490     for f in self._sub_functions.values():
 491       f.add_to_graph(g)
 492 
 493     # Adds its gradient function, too.
 494     if self._grad_func:
 495       self._grad_func.add_to_graph(g)
 496 
 497   def __call__(self, *args, **kwargs):
 498     self.add_to_graph(ops.get_default_graph())
 499     args = [ops.convert_to_tensor(_) for _ in args] + self._extra_inputs
 500     ret, op = _call(self._signature, *args, **kwargs)
 501 
 502     # Set a hidden attr in 'op' so that gradients_impl can refer back
 503     # to this _DefinedFunction instance to access python_grad_func.
 504     assert isinstance(op, ops.Operation)
 505     setattr(op, "__defun", self)
 506 
 507     if self._shape_func is not None:
 508       shapes = self._shape_func(op)
 509       if len(shapes) != len(op.outputs):
 510         raise ValueError("shape_func produced %d shapes for %d outputs" %
 511                          (len(shapes), len(op.outputs)))
 512       for (t, shape) in zip(op.outputs, shapes):
 513         t.set_shape(shape)
 514     return ret

 517 class _OverloadedFunction(object):
 518   """_OverloadedFunction encapsulates an overloaded function.
 519 
 520   _OverloadedFunction maintains a mapping from input types to
 521   instantiated _DefinedFunction in self._overload.
 522 
 523   """
 524 
 525   def __init__(self,
 526                func,
 527                argnames,
 528                func_name=None,
 529                grad_func=None,
 530                python_grad_func=None,
 531                out_names=None,
 532                **kwargs):
 533     """Creates _DefinedFunction.
 534 
 535     Args:
 536       func:  A python callable which constructs a tf function body.
 537       argnames: A list of strings for function argument names.
 538       func_name: The function name. Defaults to None, in which derives from
 539         'func'.
 540       grad_func: This function's gradient function, if not None. Defaults
 541         to None.
 542       python_grad_func: A python callable implementing the gradient of
 543         the function python-side.
 544       out_names: A list of strings for the function return value names.
 545       **kwargs: The keyword arguments. **kwargs is passed to every call
 546         site of this function.
 547 
 548     Raises:
 549       ValueError: The function definition is invalid.
 550 
 551     """
 552     self._func = func
 553     self._argnames = argnames
 554     self._func_name = func_name
 555     assert grad_func is None or isinstance(grad_func, _OverloadedFunction)
 556     self._grad_func = grad_func
 557     self._python_grad_func = python_grad_func
 558     self._out_names = out_names
 559     self._extra_kwargs = kwargs
 560     self._overload = {}
 561 
 562   def instantiate(self, input_types):
 563     """Instantiate this function given input argument types.
 564 
 565     Args:
 566       input_types: A list of data types for the inputs.
 567 
 568     Returns:
 569       _DefinedFunction for the given input types.
 570 
 571     """
 572     # Stringify the type list.
 573     key = _type_list_to_str(input_types)
 574     defined = self._overload.get(key)
 575     if not defined:
 576       # If not defined yet, define the function given the input types.
 577       name = self._func_name
 578       if name is not None:
 579         name = "_".join([name, key])
 580       defined = _DefinedFunction(
 581           self._func,
 582           self._argnames,
 583           input_types,
 584           name,
 585           None,
 586           self._python_grad_func,
 587           out_names=self._out_names,
 588           **self._extra_kwargs)
 589       _ = defined.name  # Fully instantiate the function definition.
 590       if self._grad_func:
 591         # If _grad_func is given, it is another
 592         # _OverloadedFunction. We need to instantiate it with the
 593         # right input types.
 594         output_types = [
 595             dtypes.DType(_.type) for _ in defined._signature.output_arg  # pylint: disable=protected-access
 596         ]
 597         # pylint: disable=protected-access
 598         defined._grad_func = self._grad_func.instantiate(input_types +
 599                                                          output_types)
 600         # pylint: enable=protected-access
 601       self._overload[key] = defined
 602     return defined
 603 
 604   def __call__(self, *args, **kwargs):
 605     input_types = []
 606     args = list(args)
 607     for (i, x) in enumerate(args):
 608       x = ops.convert_to_tensor(x)
 609       if not isinstance(x, ops.Tensor):
 610         raise ValueError("Expect a Tensor but get ", x)
 611       input_types.append(x.dtype)
 612       args[i] = x
 613     return self.instantiate(input_types)(*args, **kwargs)

 616 class _FuncGraph(ops.Graph):
 617   """A helper for constructing a function.
 618 
 619   _FuncGraph overrides ops.Graph's create_op() so that we can keep
 620   track of all inputs into every op created inside the function.  If
 621   any input is from other graphs, we keep track of it in self.capture
 622   and substitute the input with a place holder.
 623 
 624   Each captured input's corresponding place holder is converted into a
 625   function argument and the caller passes in the captured tensor.
 626   """
 627 
 628   def __init__(self, name, capture_by_value, *args, **kwargs):
 629     super(_FuncGraph, self).__init__(*args, **kwargs)
 630     self._capture_by_value = capture_by_value
 631     self._building_function = True
 632     self._outer_graph = ops.get_default_graph()
 633     self._vscope = vs.get_variable_scope()
 634     self._old_custom_getter = self._vscope.custom_getter
 635 
 636     # The name of the function.
 637     self.name = name
 638     # Placeholder tensors representing the inputs to this function. The tensors
 639     # are in this _FuncGraph.
 640     self.inputs = []
 641     # Tensors that will be returned this function. The tensors are in this
 642     # _FuncGraph.
 643     self.outputs = []
 644     # Maps external tensor -> internal tensor (e.g. input placeholder).
 645     self._captured = {}
 646     # The external tensors that have been captured as inputs and must be passed
 647     # to this function (empty if capturing by value, otherwise these are the
 648     # keys of _captured).
 649     self.extra_inputs = []
 650     # Input placeholders that been added for captured values (empty if capturing
 651     # by value).
 652     self.extra_args = []
 653     # Captured variables.
 654     # TODO(skyewm): is this needed?
 655     self.extra_vars = []

 657   # pylint: disable=g-doc-return-or-yield
 658 
 659   @tf_contextlib.contextmanager
 660   def container(self, container_name):
 661     """Returns a context manager that specifies the resource container to use.
 662 
 663     Overridden from `tf.Graph` to update both the init_scope container
 664     and the present inner container. This is necessary to make sure setting
 665     containers applies correctly both to created variables and to stateful
 666     ops.
 667 
 668     Args:
 669       container_name: container name string.
 670 
 671     Returns:
 672       A context manager for defining resource containers for stateful ops,
 673         yields the container name.
 674     """
 675     original_container = self._container
 676     # pylint: disable=protected-access
 677     with ops.init_scope():
 678       original_init_container = ops.get_default_graph()._container
 679     try:
 680       self._container = container_name
 681       with ops.init_scope():
 682         ops.get_default_graph()._container = container_name
 683       yield self._container
 684     finally:
 685       self._container = original_container
 686       with ops.init_scope():
 687         ops.get_default_graph()._container = original_init_container
 688     # pylint: enable=protected-access

 692   def getvar(
 693       self,
 694       getter,
 695       name,
 696       shape=None,
 697       dtype=None,
 698       initializer=None,
 699       reuse=None,
 700       trainable=True,
 701       collections=None,  # pylint: disable=redefined-outer-name
 702       use_resource=None,
 703       **kwargs):
 704     """A custom variable getter."""
 705     # Here, we switch the default graph to the outer graph and ask the
 706     # variable scope in which the function is defined to give us the
 707     # variable. The variable is stashed in extra_vars and returned to
 708     # the caller.
 709     #
 710     # We capture these variables so that the variable definition is
 711     # hoisted upward to the outer most graph.
 712     with self._outer_graph.as_default():
 713       # pylint: disable=protected-access
 714       var = self._vscope.get_variable(
 715           vs._get_default_variable_store(),
 716           name,
 717           shape=shape,
 718           dtype=dtype,
 719           initializer=initializer,
 720           reuse=reuse,
 721           trainable=trainable,
 722           collections=collections,
 723           use_resource=use_resource)
 724       self.extra_vars.append(var)
 725       if isinstance(var, resource_variable_ops.ResourceVariable):
 726         # For resource-based variables read the variable outside the function
 727         # and pass in the value. This ensures that the function is pure and
 728         # differentiable. TODO(apassos) this may have performance problems if
 729         # the function will only do embedding lookups on the variable.
 730         return var.value()
 731       return var

 733   def create_op(self, op_type, inputs, data_types, **kwargs):
 734     for i, x in enumerate(inputs):
 735       if isinstance(x, ops.EagerTensor) or x.graph is not self:
 736         inputs[i] = self.capture(x)
 737     return super(_FuncGraph, self).create_op(op_type, inputs, data_types,
 738                                              **kwargs)

 740   def capture(self, tensor, name=None):
 741     """Adds the given tensor to this graph and returns the captured tensor."""
 742     if tensor in self._captured:
 743       # Captured already.
 744       return self._captured[tensor]
 745     elif self._capture_by_value:
 746       return self._add_tensor_and_parents(tensor)
 747     else:
 748       return self._capture_tensor_as_extra_input(tensor, name)
 749 
 750   def _capture_tensor_as_extra_input(self, tensor, name=None):
 751     # Substitute with a placeholder.
 752     self.extra_inputs.append(tensor)
 753     # Hoist the new input placeholder out of any control flow context
 754     # we're currently in.
 755     with ops.control_dependencies(None):
 756       ph = array_ops.placeholder(
 757           tensor.dtype, shape=tensor.get_shape(), name=name)
 758     # pylint: disable=protected-access
 759     if ops._USE_C_SHAPES:
 760       if isinstance(tensor, ops.EagerTensor):
 761         handle_data = tensor._handle_data
 762         if handle_data:
 763           handle_data = handle_data.SerializeToString()
 764       else:
 765         handle_data = c_api.GetHandleShapeAndType(tensor.graph._c_graph,
 766                                                   tensor._as_tf_output())
 767 
 768       if handle_data:
 769         c_api.SetHandleShapeAndType(ph.graph._c_graph, ph._as_tf_output(),
 770                                     compat.as_bytes(handle_data))
 771     else:
 772       ph._handle_data = tensor._handle_data
 773     # pylint: enable=protected-access
 774     self.inputs.append(ph)
 775     self._captured[tensor] = ph
 776     self.extra_args.append(ph)
 777     if _is_guaranteed_const(tensor):
 778       with ops.control_dependencies(None):
 779         return array_ops.guarantee_const(ph)
 780     else:
 781       return ph
 782 
 783   def _add_tensor_and_parents(self, tensor):
 784     op = self._add_op_and_parents(tensor.op)
 785     return op.outputs[tensor.value_index]
 786 
 787   def _add_op_and_parents(self, op):
 788     # pylint: disable=protected-access
 789     op_def = graph_to_function_def._get_op_def(op)
 790     # pylint: enable=protected-access
 791     if op_def.is_stateful:
 792       raise ValueError("Cannot capture a stateful node (name:%s, type:%s) "
 793                        "by value." % (op.name, op.type))
 794     elif op.type in ("Placeholder", "PlaceholderV2"):
 795       raise ValueError("Cannot capture a placeholder (name:%s, type:%s) "
 796                        "by value." % (op.name, op.type))
 797 
 798     captured_inputs = [self._add_tensor_and_parents(x) for x in op.inputs]
 799 
 800     captured_op = self.create_op(
 801         op.type,
 802         captured_inputs, [o.dtype for o in op.outputs],
 803         name=op.name,
 804         attrs=op.node_def.attr,
 805         op_def=op_def)
 806 
 807     for t, captured_t in zip(op.outputs, captured_op.outputs):
 808       self._captured[t] = captured_t
 809 
 810     return captured_op

 813 def func_graph_from_py_func(func, arg_names, arg_types, name=None,
 814                             capture_by_value=False, device=None,
 815                             colocation_stack=None, container=None,
 816                             collections_ref=None, arg_shapes=None):
 817   """Returns a _FuncGraph generated from `func`.
 818 
 819   Args:
 820     func: A Python callable which constructs a TF function body. The arguments
 821       must correspond to `arg_types`. Returns a value or list/tuple of values.
 822       No returned value can be None.
 823     arg_names: A sequence of strings for the function argument names.
 824     arg_types: A sequence of the function's argument types.
 825     name: The function name. If None, the name is derived from `func`.
 826     capture_by_value: boolean. If True, captured values will be copied into the
 827       function body.
 828     device: device name or function.
 829     colocation_stack: A colocation stack (list) the _FuncGraph should use.
 830     container: A container name the _FuncGraph should start with.
 831     collections_ref: A reference to a collections dict the _FuncGraph should
 832       use internally.
 833     arg_shapes: A sequence of the function's argument shapes.
 834 
 835   Returns:
 836     A _FuncGraph.
 837 
 838   Raises:
 839     ValueError: if func returns None.
 840   """
 841   if not name:
 842     name = function_utils.get_func_name(func)
 843   func_graph = _FuncGraph(name, capture_by_value)
 844 
 845   with func_graph.as_default(), ops.device(device):
 846     # pylint: disable=protected-access
 847     if collections_ref is not None:
 848       func_graph._collections = collections_ref
 849     if container is not None:
 850       func_graph._container = container
 851     if colocation_stack is not None:
 852       func_graph._colocation_stack = colocation_stack
 853     # pylint: enable=protected-access
 854 
 855     if arg_shapes is None:
 856       arg_shapes = [None] * len(arg_types)
 857 
 858     # Create placeholders for the function arguments.
 859     for (argname, argtype, argshape) in zip(arg_names, arg_types, arg_shapes):
 860       argholder = array_ops.placeholder(argtype, shape=argshape, name=argname)
 861       func_graph.inputs.append(argholder)
 862     # Call func and gather the output tensors.
 863     with vs.variable_scope("", custom_getter=func_graph.getvar):
 864       outputs = func(*func_graph.inputs)
 865 
 866     # There is no way of distinguishing between a function not returning
 867     # anything and a function returning None in Python.
 868     # We need to allow the former and ideally want to forbid the latter as
 869     # it is most likely user error.
 870     # TODO(iga): Consider adding a @NoOutput decorator on top of @Defun to
 871     # allow users to explicitly mark the function as not returning anything.
 872     # For now, we allow a single None return and interpret it as a function
 873     # with no output.
 874     if outputs is None:
 875       outputs = []
 876     else:
 877       # If func only returned one value, make it a tuple.
 878       if not isinstance(outputs, (list, tuple)):
 879         outputs = (outputs,)
 880       if any([_ is None for _ in outputs]):
 881         raise ValueError("Function can not return None.")
 882     # Ensures each output is a Tensor in the function graph.
 883     outputs = [ops.convert_to_tensor(t) for t in outputs]
 884     outputs = [func_graph.capture(t) if t.graph is not func_graph else t
 885                for t in outputs]
 886     func_graph.outputs = outputs
 887   return func_graph

 890 def _is_guaranteed_const(tensor):
 891   """Determines whether `tensor` is guaranteed to be a constant.
 892 
 893   A tensor is guaranteed to be a constant if either it was produced by
 894   a `GuaranteeConst` op or if all of its children are guaranteed to be
 895   constants.
 896 
 897   Args:
 898     tensor: The tensor for which to determine const-ness.
 899 
 900   Returns:
 901     True if `tensor` is guaranteed to be a constant, False otherwise.
 902   """
 903 
 904   if isinstance(tensor, ops.EagerTensor):
 905     return False
 906 
 907   class Work(object):
 908 
 909     def __init__(self, op, leaving):
 910       self.op = op
 911       self.leaving = leaving
 912 
 913   is_guaranteed_const = lambda op: op.node_def.op == "GuaranteeConst"
 914   constants = set([])
 915   def all_inputs_const(op):
 916     # If all inputs of an op are guaranteed constants, then we can infer that
 917     # the op produces a constant as well.
 918     return op.inputs and all(inp.op in constants for inp in op.inputs)
 919 
 920   visited = set([])
 921   stack = [Work(tensor.op, leaving=False)]
 922   while stack:
 923     work = stack.pop()
 924     if work.leaving:
 925       if all_inputs_const(work.op):
 926         constants.add(work.op)
 927       continue
 928     visited.add(work.op)
 929     if is_guaranteed_const(work.op):
 930       constants.add(work.op)
 931       continue
 932 
 933     # This op will be revisited after all its inputs are checked for const-ness.
 934     stack.append(Work(work.op, leaving=True))
 935     for inp in work.op.inputs:
 936       if inp.op not in visited:
 937         stack.append(Work(inp.op, leaving=False))
 938   return tensor.op in constants

 941 def _call(sig, *inputs, **kwargs):
 942   """Adds a node calling a function.
 943 
 944   This adds a `call` op to the default graph that calls the function
 945   of signature `sig`, passing the tensors in `inputs` as arguments.
 946   It returns the outputs of the call, which are one or more tensors.
 947 
 948   `sig` is OpDefArg.a `_DefinedFunction` object.
 949 
 950   You can pass an optional keyword parameter `name=string` to name the
 951   added operation.
 952 
 953   You can pass an optional keyword parameter `noinline=True|False` to
 954   instruct the runtime not to inline the function body into the call
 955   site.
 956 
 957   Args:
 958     sig: OpDefArg. The signature of the function.
 959     *inputs: arguments to the function.
 960     **kwargs: Optional keyword arguments.  Can only contain 'name' or
 961         'noinline'.
 962 
 963   Returns:
 964      A 2-element tuple. First element: a Tensor if the function returns a single
 965      value; a list of Tensors if the function returns multiple value; the
 966      Operation if the function returns no values. Second element: the Operation.
 967 
 968   Raises:
 969     ValueError: if the arguments are invalid.
 970   """
 971   if len(inputs) != len(sig.input_arg):
 972     raise ValueError("Expected number of arguments: %d, received: %d" % (len(
 973         sig.input_arg), len(inputs)))
 974   name = kwargs.pop("name", None)
 975   g = ops.get_default_graph()
 976   func_name = sig.name
 977   attrs = _parse_kwargs_as_attrs(func_name, **kwargs)
 978   output_types = [dtypes.DType(x.type) for x in sig.output_arg]
 979   with ops.name_scope(name, func_name, inputs) as name:
 980     op = g.create_op(
 981         func_name,
 982         list(inputs),
 983         output_types,
 984         name=name,
 985         attrs=attrs,
 986         op_def=sig,
 987         compute_shapes=False)
 988   if op.outputs:
 989     if len(op.outputs) == 1:
 990       ret = op.outputs[0]
 991     else:
 992       ret = tuple(op.outputs)
 993   else:
 994     ret = op
 995   return ret, op

 998 def _from_definition(fdef, grad_func=None):
 999   """Creates a _DefinedFunction initialized from a FunctionDef proto.
1000 
1001   Args:
1002     fdef: a FunctionDef
1003     grad_func: a _DefinedFunction or None
1004 
1005   Returns:
1006     A _DefinedFunction representing fdef
1007   """
1008   # TODO(iga): This method does major surgery on _DefinedFunction.
1009   # Make it a named constructor using @classmethod of _DefinedFunction.
1010 
1011   # The Python callable is only needed to create a FunctionDef. Since we have
1012   # the FunctionDef here, we don't need to set _DefinedFunction._func (nor do we
1013   # have access to such a callable here).
1014   func = None
1015   argnames = [arg.name for arg in fdef.signature.input_arg]
1016   input_types = tuple(
1017       dtypes.as_dtype(arg.type) for arg in fdef.signature.input_arg)
1018   func_name = fdef.signature.name
1019   # Note: FunctionDefs do not include python gradient functions, so if the
1020   # original _DefinedFunction included one it will not be reflected here.
1021   python_grad_func = None
1022   out_names = [arg.name for arg in fdef.signature.output_arg]
1023   result = _DefinedFunction(func, argnames, input_types, func_name, grad_func,
1024                             python_grad_func, out_names)
1025   # pylint: disable=protected-access
1026   serialized = fdef.SerializeToString()
1027   c_func = c_api.TF_FunctionImportFunctionDef(serialized)
1028   result._c_func = c_api_util.ScopedTFFunction(c_func)
1029   result._extra_inputs = []
1030   # pylint: enable=protected-access
1031 
1032   return result

1035 def _from_library(lib):
1036   """Creates _DefinedFunctions initialized from a FunctionDefLibrary proto.
1037 
1038   This method handles assigning the correct gradient functions to each
1039   function.
1040 
1041   Args:
1042     lib: a FunctionDefLibrary
1043 
1044   Returns:
1045     A list of _DefinedFunctions
1046 
1047   Raises:
1048     ValueError: `lib` is invalid
1049   """
1050   if not lib.function and not lib.gradient:
1051     return []
1052 
1053   # function name -> FunctionDef proto
1054   funcs = {fdef.signature.name: fdef for fdef in lib.function}
1055 
1056   # Validate that all references function names have function defs
1057   for g in lib.gradient:
1058     if g.function_name not in funcs:
1059       raise ValueError("FunctionDefLibrary missing '%s' FunctionDef\n%s" %
1060                        (g.function_name, str(lib)))
1061     if g.gradient_func not in funcs:
1062       raise ValueError("FunctionDefLibrary missing '%s' FunctionDef\n%s" %
1063                        (g.gradient_func, str(lib)))
1064 
1065   # function name -> gradient function name
1066   func_to_grad = collections.defaultdict(lambda: None)
1067   # gradient function name -> names of functions having that grad function
1068   grad_to_funcs = collections.defaultdict(list)
1069 
1070   for gdef in lib.gradient:
1071     func_to_grad[gdef.function_name] = gdef.gradient_func
1072     grad_to_funcs[gdef.gradient_func].append(gdef.function_name)
1073 
1074   # Start with functions without gradients
1075   ready = [
1076       fdef for fdef in lib.function if func_to_grad[fdef.signature.name] is None
1077   ]
1078   if not ready:
1079     raise ValueError(
1080         "FunctionDefLibrary contains cyclic gradient functions!\n" + str(lib))
1081   # function name -> _DefinedFunction
1082   initialized = {}
1083 
1084   while ready:
1085     fdef = ready.pop()
1086     name = fdef.signature.name
1087 
1088     grad = initialized.get(func_to_grad[name])
1089     if func_to_grad[name]:
1090       assert grad
1091     defined_func = _from_definition(fdef, grad_func=grad)
1092     initialized[name] = defined_func
1093 
1094     ready.extend(funcs[f] for f in grad_to_funcs[name])
1095 
1096   return initialized.values()

1099 def _get_experimental_kwarg_as_attr(attr_name, value):
1100   """Creates an AttrValue for a python object."""
1101   if isinstance(value, bool):
1102     return attr_value_pb2.AttrValue(b=value)
1103   elif isinstance(value, int):
1104     return attr_value_pb2.AttrValue(i=value)
1105   elif isinstance(value, float):
1106     return attr_value_pb2.AttrValue(f=value)
1107   elif isinstance(value, str):
1108     return attr_value_pb2.AttrValue(s=compat.as_bytes(value))
1109   else:
1110     raise ValueError("Unsupported attribute type for %s with type %s" %
1111                      (attr_name, type(value)))

1114 def _parse_kwargs_as_attrs(func_name, **kwargs):
1115   """Parses **kwargs into a node's attributes."""
1116   attrs = {}
1117 
1118   noinline = kwargs.pop("noinline", None)
1119   if noinline is not None:
1120     attrs["_noinline"] = attr_value_pb2.AttrValue(b=bool(noinline))
1121 
1122   compiled = kwargs.pop("compiled", None)
1123   separate_compiled_gradients = kwargs.pop("separate_compiled_gradients", None)
1124   if compiled is not None:
1125     attrs["_XlaCompile"] = attr_value_pb2.AttrValue(b=bool(compiled))
1126     attrs["_XlaSeparateCompiledGradients"] = attr_value_pb2.AttrValue(
1127         b=bool(separate_compiled_gradients))
1128     # Forward _XlaScope from enclosing context (if set), otherwise create new.
1129     # pylint: disable=protected-access
1130     if "_XlaScope" in ops.get_default_graph()._attr_scope_map:
1131       attrs["_XlaScope"] = ops.get_default_graph()._attr_scope_map["_XlaScope"]
1132     else:
1133       attrs["_XlaScope"] = attr_value_pb2.AttrValue(
1134           s=("function_%s" % func_name).encode())
1135     # pylint: enable=protected-access
1136 
1137   kwargs_keys = list(kwargs.keys())
1138   for key in kwargs_keys:
1139     if key.startswith("experimental_"):
1140       attrs[key] = _get_experimental_kwarg_as_attr(key, kwargs[key])
1141       del kwargs[key]
1142 
1143   if kwargs:
1144     raise ValueError("Unknown keyword arguments: %s" % kwargs.keys())
1145   return attrs
