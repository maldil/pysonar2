  33 @tf_export("errors.OpError", "OpError")
  34 @deprecation.deprecated_endpoints("OpError")
  35 class OpError(Exception):
  36   """A generic error that is raised when TensorFlow execution fails.
  37 
  38   Whenever possible, the session will raise a more specific subclass
  39   of `OpError` from the `tf.errors` module.
  40   """
  41 
  42   def __init__(self, node_def, op, message, error_code):
  43     """Creates a new `OpError` indicating that a particular op failed.
  44 
  45     Args:
  46       node_def: The `node_def_pb2.NodeDef` proto representing the op that
  47         failed, if known; otherwise None.
  48       op: The `ops.Operation` that failed, if known; otherwise None.
  49       message: The message string describing the failure.
  50       error_code: The `error_codes_pb2.Code` describing the error.
  51     """
  52     super(OpError, self).__init__()
  53     self._node_def = node_def
  54     self._op = op
  55     self._message = message
  56     self._error_code = error_code
  57 
  58   def __reduce__(self):
  59     # Allow the subclasses to accept less arguments in their __init__.
  60     init_argspec = tf_inspect.getargspec(self.__class__.__init__)
  61     args = tuple(getattr(self, arg) for arg in init_argspec.args[1:])
  62     return self.__class__, args
  63 
  64   @property
  65   def message(self):
  66     """The error message that describes the error."""
  67     return self._message
  68 
  69   @property
  70   def op(self):
  71     """The operation that failed, if known.
  72 
  73     *N.B.* If the failed op was synthesized at runtime, e.g. a `Send`
  74     or `Recv` op, there will be no corresponding
  75     `tf.Operation`
  76     object.  In that case, this will return `None`, and you should
  77     instead use the `tf.errors.OpError.node_def` to
  78     discover information about the op.
  79 
  80     Returns:
  81       The `Operation` that failed, or None.
  82     """
  83     return self._op
  84 
  85   @property
  86   def error_code(self):
  87     """The integer error code that describes the error."""
  88     return self._error_code
  89 
  90   @property
  91   def node_def(self):
  92     """The `NodeDef` proto representing the op that failed."""
  93     return self._node_def
  94 
  95   def __str__(self):
  96     if self._op is not None:
  97       output = ["%s\n\nCaused by op %r, defined at:\n" % (self.message,
  98                                                           self._op.name,)]
  99       curr_traceback_list = traceback.format_list(self._op.traceback)
 100       output.extend(curr_traceback_list)
 101       # pylint: disable=protected-access
 102       original_op = self._op._original_op
 103       # pylint: enable=protected-access
 104       while original_op is not None:
 105         output.append(
 106             "\n...which was originally created as op %r, defined at:\n"
 107             % (original_op.name,))
 108         prev_traceback_list = curr_traceback_list
 109         curr_traceback_list = traceback.format_list(original_op.traceback)
 110 
 111         # Attempt to elide large common subsequences of the subsequent
 112         # stack traces.
 113         #
 114         # TODO(mrry): Consider computing the actual longest common subsequence.
 115         is_eliding = False
 116         elide_count = 0
 117         last_elided_line = None
 118         for line, line_in_prev in zip(curr_traceback_list, prev_traceback_list):
 119           if line == line_in_prev:
 120             if is_eliding:
 121               elide_count += 1
 122               last_elided_line = line
 123             else:
 124               output.append(line)
 125               is_eliding = True
 126               elide_count = 0
 127           else:
 128             if is_eliding:
 129               if elide_count > 0:
 130                 output.extend(
 131                     ["[elided %d identical lines from previous traceback]\n"
 132                      % (elide_count - 1,), last_elided_line])
 133               is_eliding = False
 134             output.extend(line)
 135 
 136         # pylint: disable=protected-access
 137         original_op = original_op._original_op
 138         # pylint: enable=protected-access
 139       output.append("\n%s (see above for traceback): %s\n" %
 140                     (type(self).__name__, self.message))
 141       return "".join(output)
 142     else:
 143       return self.message

 501 def _make_specific_exception(node_def, op, message, error_code):
 502   try:
 503     exc_type = exception_type_from_error_code(error_code)
 504     return exc_type(node_def, op, message)
 505   except KeyError:
 506     warnings.warn("Unknown error code: %d" % error_code)
 507     return UnknownError(node_def, op, message, error_code)

 510 # Named like a function for backwards compatibility with the
 511 # @tf_contextlib.contextmanager version, which was switched to a class to avoid
 512 # some object creation overhead.
 513 # TODO(b/77295559): expand use of TF_Status* SWIG typemap and deprecate this.
 514 @tf_export("errors.raise_exception_on_not_ok_status")  # pylint: disable=invalid-name
 515 class raise_exception_on_not_ok_status(object):
 516   """Context manager to check for C API status."""
 517 
 518   def __enter__(self):
 519     self.status = c_api_util.ScopedTFStatus()
 520     return self.status.status
 521 
 522   def __exit__(self, type_arg, value_arg, traceback_arg):
 523     try:
 524       if c_api.TF_GetCode(self.status.status) != 0:
 525         raise _make_specific_exception(
 526             None, None,
 527             compat.as_text(c_api.TF_Message(self.status.status)),
 528             c_api.TF_GetCode(self.status.status))
 529     # Delete the underlying status object from memory otherwise it stays alive
 530     # as there is a reference to status from this from the traceback due to
 531     # raise.
 532     finally:
 533       del self.status
 534     return False  # False values do not suppress exceptions
