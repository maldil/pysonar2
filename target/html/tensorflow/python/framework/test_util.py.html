  89 def assert_ops_in_graph(expected_ops, graph):
  90   """Assert all expected operations are found.
  91 
  92   Args:
  93     expected_ops: `dict<string, string>` of op name to op type.
  94     graph: Graph to check.
  95   Returns:
  96     `dict<string, node>` of node name to node.
  97 
  98   Raises:
  99     ValueError: If the expected ops are not present in the graph.
 100   """
 101   actual_ops = {}
 102   gd = graph.as_graph_def()
 103   for node in gd.node:
 104     if node.name in expected_ops:
 105       if expected_ops[node.name] != node.op:
 106         raise ValueError("Expected op for node %s is different. %s vs %s" %
 107                          (node.name, expected_ops[node.name], node.op))
 108       actual_ops[node.name] = node
 109   if set(expected_ops.keys()) != set(actual_ops.keys()):
 110     raise ValueError("Not all expected ops are present. Expected %s, found %s" %
 111                      (expected_ops.keys(), actual_ops.keys()))
 112   return actual_ops

 115 @tf_export("test.assert_equal_graph_def")
 116 def assert_equal_graph_def(actual, expected, checkpoint_v2=False):
 117   """Asserts that two `GraphDef`s are (mostly) the same.
 118 
 119   Compares two `GraphDef` protos for equality, ignoring versions and ordering of
 120   nodes, attrs, and control inputs.  Node names are used to match up nodes
 121   between the graphs, so the naming of nodes must be consistent.
 122 
 123   Args:
 124     actual: The `GraphDef` we have.
 125     expected: The `GraphDef` we expected.
 126     checkpoint_v2: boolean determining whether to ignore randomized attribute
 127         values that appear in V2 checkpoints.
 128 
 129   Raises:
 130     AssertionError: If the `GraphDef`s do not match.
 131     TypeError: If either argument is not a `GraphDef`.
 132   """
 133   if not isinstance(actual, graph_pb2.GraphDef):
 134     raise TypeError(
 135         "Expected tf.GraphDef for actual, got %s" % type(actual).__name__)
 136   if not isinstance(expected, graph_pb2.GraphDef):
 137     raise TypeError(
 138         "Expected tf.GraphDef for expected, got %s" % type(expected).__name__)
 139 
 140   if checkpoint_v2:
 141     _strip_checkpoint_v2_randomized(actual)
 142     _strip_checkpoint_v2_randomized(expected)
 143 
 144   diff = pywrap_tensorflow.EqualGraphDefWrapper(actual.SerializeToString(),
 145                                                 expected.SerializeToString())
 146   if diff:
 147     raise AssertionError(compat.as_str(diff))

 150 def assert_meta_graph_protos_equal(tester, a, b):
 151   """Compares MetaGraphDefs `a` and `b` in unit test class `tester`."""
 152   # Carefully check the collection_defs
 153   tester.assertEqual(set(a.collection_def), set(b.collection_def))
 154   collection_keys = a.collection_def.keys()
 155   for k in collection_keys:
 156     a_value = a.collection_def[k]
 157     b_value = b.collection_def[k]
 158     proto_type = ops.get_collection_proto_type(k)
 159     if proto_type:
 160       a_proto = proto_type()
 161       b_proto = proto_type()
 162       # Number of entries in the collections is the same
 163       tester.assertEqual(
 164           len(a_value.bytes_list.value), len(b_value.bytes_list.value))
 165       for (a_value_item, b_value_item) in zip(a_value.bytes_list.value,
 166                                               b_value.bytes_list.value):
 167         a_proto.ParseFromString(a_value_item)
 168         b_proto.ParseFromString(b_value_item)
 169         tester.assertProtoEquals(a_proto, b_proto)
 170     else:
 171       tester.assertEquals(a_value, b_value)
 172   # Compared the fields directly, remove their raw values from the
 173   # proto comparison below.
 174   a.ClearField("collection_def")
 175   b.ClearField("collection_def")
 176 
 177   # Check the graph_defs.
 178   assert_equal_graph_def(a.graph_def, b.graph_def, checkpoint_v2=True)
 179   # Check graph_def versions (ignored by assert_equal_graph_def).
 180   tester.assertProtoEquals(a.graph_def.versions, b.graph_def.versions)
 181   # Compared the fields directly, remove their raw values from the
 182   # proto comparison below.
 183   a.ClearField("graph_def")
 184   b.ClearField("graph_def")
 185 
 186   tester.assertProtoEquals(a, b)

 194 def _strip_checkpoint_v2_randomized(graph_def):
 195   for node in graph_def.node:
 196     delete_keys = []
 197     for attr_key in node.attr:
 198       attr_tensor_value = node.attr[attr_key].tensor
 199       if attr_tensor_value and len(attr_tensor_value.string_val) == 1:
 200         attr_tensor_string_value = attr_tensor_value.string_val[0]
 201         if (attr_tensor_string_value and
 202             re.match(_SHARDED_SAVE_OP_PATTERN, str(attr_tensor_string_value))):
 203           delete_keys.append(attr_key)
 204     for attr_key in delete_keys:
 205       del node.attr[attr_key]

 224 def NHWCToNCHW(input_tensor):
 225   """Converts the input from the NHWC format to NCHW.
 226 
 227   Args:
 228     input_tensor: a 4- or 5-D tensor, or an array representing shape
 229 
 230   Returns:
 231     converted tensor or shape array
 232   """
 233   # tensor dim -> new axis order
 234   new_axes = {4: [0, 3, 1, 2], 5: [0, 4, 1, 2, 3]}
 235   if isinstance(input_tensor, ops.Tensor):
 236     ndims = input_tensor.shape.ndims
 237     return array_ops.transpose(input_tensor, new_axes[ndims])
 238   else:
 239     ndims = len(input_tensor)
 240     return [input_tensor[a] for a in new_axes[ndims]]

 243 def NHWCToNCHW_VECT_C(input_shape_or_tensor):
 244   """Transforms the input from the NHWC layout to NCHW_VECT_C layout.
 245 
 246   Note: Does not include quantization or type conversion steps, which should
 247   be applied afterwards.
 248 
 249   Args:
 250     input_shape_or_tensor: a 4- or 5-D tensor, or an array representing shape
 251 
 252   Returns:
 253     tensor or shape array transformed into NCHW_VECT_C
 254 
 255   Raises:
 256     ValueError: if last dimension of `input_shape_or_tensor` is not evenly
 257         divisible by 4.
 258   """
 259   permutations = {5: [0, 3, 1, 2, 4], 6: [0, 4, 1, 2, 3, 5]}
 260   is_tensor = isinstance(input_shape_or_tensor, ops.Tensor)
 261   temp_shape = (
 262       input_shape_or_tensor.shape.as_list()
 263       if is_tensor else input_shape_or_tensor)
 264   if temp_shape[-1] % 4 != 0:
 265     raise ValueError(
 266         "Last dimension of input must be evenly divisible by 4 to convert to "
 267         "NCHW_VECT_C.")
 268   temp_shape[-1] //= 4
 269   temp_shape.append(4)
 270   permutation = permutations[len(temp_shape)]
 271   if is_tensor:
 272     t = array_ops.reshape(input_shape_or_tensor, temp_shape)
 273     return array_ops.transpose(t, permutation)
 274   else:
 275     return [temp_shape[a] for a in permutation]

 278 def NCHW_VECT_CToNHWC(input_shape_or_tensor):
 279   """Transforms the input from the NCHW_VECT_C layout to NHWC layout.
 280 
 281   Note: Does not include de-quantization or type conversion steps, which should
 282   be applied beforehand.
 283 
 284   Args:
 285     input_shape_or_tensor: a 5- or 6-D tensor, or an array representing shape
 286 
 287   Returns:
 288     tensor or shape array transformed into NHWC
 289 
 290   Raises:
 291     ValueError: if last dimension of `input_shape_or_tensor` is not 4.
 292   """
 293   permutations = {5: [0, 2, 3, 1, 4], 6: [0, 2, 3, 4, 1, 5]}
 294   is_tensor = isinstance(input_shape_or_tensor, ops.Tensor)
 295   input_shape = (
 296       input_shape_or_tensor.shape.as_list()
 297       if is_tensor else input_shape_or_tensor)
 298   if input_shape[-1] != 4:
 299     raise ValueError("Last dimension of NCHW_VECT_C must be 4.")
 300   permutation = permutations[len(input_shape)]
 301   nhwc_shape = [input_shape[a] for a in permutation[:-1]]
 302   nhwc_shape[-1] *= input_shape[-1]
 303   if is_tensor:
 304     t = array_ops.transpose(input_shape_or_tensor, permutation)
 305     return array_ops.reshape(t, nhwc_shape)
 306   else:
 307     return nhwc_shape

 329 def skip_if(condition):
 330   """Skips the decorated function if condition is or evaluates to True.
 331 
 332   Args:
 333     condition: Either an expression that can be used in "if not condition"
 334                statement, or a callable whose result should be a boolean.
 335   Returns:
 336     The wrapped function
 337   """
 338 
 339   def real_skip_if(fn):
 340 
 341     def wrapper(*args, **kwargs):
 342       if callable(condition):
 343         skip = condition()
 344       else:
 345         skip = condition
 346       if not skip:
 347         fn(*args, **kwargs)
 348 
 349     return wrapper
 350 
 351   return real_skip_if

 381 def with_c_shapes(cls):
 382   """Adds methods that call original methods but with C API shapes enabled.
 383 
 384   Note this enables C shapes in new methods after running the test class's
 385   setup method.
 386 
 387   Args:
 388     cls: class to decorate
 389 
 390   Returns:
 391     cls with new test methods added
 392   """
 393   # If C shapes are already enabled, don't do anything. Some tests break if the
 394   # same test is run twice, so this allows us to turn on the C shapes by default
 395   # without breaking these tests.
 396   if ops._USE_C_SHAPES:
 397     return cls
 398 
 399   for name, value in cls.__dict__.copy().items():
 400     if callable(value) and name.startswith("test"):
 401       setattr(cls, name + "WithCShapes", enable_c_shapes(value))
 402   return cls

 435 def with_control_flow_v2(cls):
 436   """Adds methods that call original methods with WhileV2 and CondV2 enabled.
 437 
 438   Note this enables CondV2 and WhileV2 in new methods after running the test
 439   class's setup method.
 440 
 441   In addition to this, callers must import the while_v2 module in order to set

 443

 445   @disable_control_flow_v2 decorator), the v2 function is not generated for it.
 446

 448 
 449   @test_util.with_control_flow_v2
 450   class ControlFlowTest(test.TestCase):
 451 
 452     def testEnabledForV2(self):
 453       ...
 454 
 455     @test_util.disable_control_flow_v2("b/xyzabc")

 460   class ControlFlowTest(test.TestCase):
 461 
 462     def testEnabledForV2(self):
 463       ...
 464 
 465     def testEnabledForV2WithControlFlowV2(self):
 466       // Enable V2 flags.
 467       testEnabledForV2(self)
 468       // Restore V2 flags.
 469 
 470     def testDisabledForV2(self):
 471       ...
 472 
 473   Args:
 474     cls: class to decorate
 475 
 476   Returns:
 477     cls with new test methods added
 478   """
 479   if control_flow_ops.ENABLE_WHILE_V2 and control_flow_ops.ENABLE_COND_V2:
 480     return cls
 481 
 482   for name, value in cls.__dict__.copy().items():
 483     if (callable(value) and name.startswith("test") and
 484         not getattr(value, "_disable_control_flow_v2", False)):
 485       setattr(cls, name + "WithControlFlowV2", enable_control_flow_v2(value))
 486   return cls

 506 def assert_no_new_pyobjects_executing_eagerly(f):
 507   """Decorator for asserting that no new Python objects persist after a test.
 508 
 509   Runs the test multiple times executing eagerly, first as a warmup and then to
 510   let objects accumulate. The warmup helps ignore caches which do not grow as
 511   the test is run repeatedly.
 512 
 513   Useful for checking that there are no missing Py_DECREFs in the C exercised by
 514   a bit of Python.
 515   """
 516 
 517   def decorator(self, **kwargs):
 518     """Warms up, gets an object count, runs the test, checks for new objects."""
 519     with context.eager_mode():
 520       gc.disable()
 521       # Run the test 2 times as warmup, in an attempt to fill up caches, which
 522       # should not grow as the test is run repeatedly below.
 523       #
 524       # TODO(b/117156879): Running warmup twice is black magic; we have seen
 525       # tests that fail with 1 warmup run, and pass with 2, on various versions
 526       # of python2.7.x.
 527       for _ in range(2):
 528         f(self, **kwargs)
 529       gc.collect()
 530       previous_count = len(gc.get_objects())
 531       if ops.has_default_graph():
 532         collection_sizes_before = {
 533             collection: len(ops.get_collection(collection))
 534             for collection in ops.get_default_graph().collections
 535         }
 536       for _ in range(3):
 537         f(self, **kwargs)
 538       # Note that gc.get_objects misses anything that isn't subject to garbage
 539       # collection (C types). Collections are a common source of leaks, so we
 540       # test for collection sizes explicitly.
 541       if ops.has_default_graph():
 542         for collection_key in ops.get_default_graph().collections:
 543           collection = ops.get_collection(collection_key)
 544           size_before = collection_sizes_before.get(collection_key, 0)
 545           if len(collection) > size_before:
 546             raise AssertionError(
 547                 ("Collection %s increased in size from "
 548                  "%d to %d (current items %s).") %
 549                 (collection_key, size_before, len(collection), collection))
 550           # Make sure our collection checks don't show up as leaked memory by
 551           # removing references to temporary variables.
 552           del collection
 553           del collection_key
 554           del size_before
 555         del collection_sizes_before
 556       gc.collect()
 557       # There should be no new Python objects hanging around.
 558       new_count = len(gc.get_objects())
 559       # In some cases (specifacally on MacOS), new_count is somehow
 560       # smaller than previous_count.
 561       # Using plain assert because not all classes using this decorator
 562       # have assertLessEqual
 563       assert new_count <= previous_count, (
 564           "new_count(%d) is not less than or equal to previous_count(%d)" %
 565           (new_count, previous_count))
 566       gc.enable()
 567 
 568   return decorator

 571 def assert_no_new_tensors(f):
 572   """Decorator for asserting that no new Tensors persist after a test.
 573 
 574   Mainly useful for checking that code using the Python C API has correctly
 575   manipulated reference counts.
 576 
 577   Clears the caches that it knows about, runs the garbage collector, then checks
 578   that there are no Tensor or Tensor-like objects still around. This includes
 579   Tensors to which something still has a reference (e.g. from missing
 580   Py_DECREFs) and uncollectable cycles (i.e. Python reference cycles where one
 581   of the objects has __del__ defined).
 582 
 583   Args:
 584     f: The test case to run.
 585   Returns:
 586     The decorated test case.
 587   """
 588 
 589   def decorator(self, **kwargs):
 590     """Finds existing Tensors, runs the test, checks for new Tensors."""
 591 
 592     def _is_tensorflow_object(obj):
 593       try:
 594         return isinstance(obj,
 595                           (ops.Tensor, variables.Variable,
 596                            tensor_shape.Dimension, tensor_shape.TensorShape))
 597       except ReferenceError:
 598         # If the object no longer exists, we don't care about it.
 599         return False
 600 
 601     tensors_before = set(

 691 def _combine_named_parameters(**kwargs):
 692   """Generate combinations based on its keyword arguments.
 693 
 694   Two sets of returned combinations can be concatenated using +.  Their product
 695   can be computed using `times()`.
 696 
 697   Args:
 698     **kwargs: keyword arguments of form `option=[possibilities, ...]`
 699          or `option=the_only_possibility`.
 700 
 701   Returns:
 702     a list of dictionaries for each combination. Keys in the dictionaries are
 703     the keyword argument names.  Each key has one value - one of the
 704     corresponding keyword argument values.
 705   """
 706   if not kwargs:
 707     return [OrderedDict()]
 708 
 709   sort_by_key = lambda k: k[0][0]
 710   kwargs = OrderedDict(sorted(kwargs.items(), key=sort_by_key))
 711   first = list(kwargs.items())[0]
 712 
 713   rest = dict(list(kwargs.items())[1:])
 714   rest_combined = _combine_named_parameters(**rest)
 715 
 716   key = first[0]
 717   values = first[1]
 718   if not isinstance(values, list):
 719     values = [values]
 720 
 721   combinations = [
 722       OrderedDict(sorted(list(combined.items()) + [(key, v)], key=sort_by_key))
 723       for v in values
 724       for combined in rest_combined
 725   ]
 726   return combinations

 818       tests may create garbage for legitimate reasons (e.g. they define a class

 821       collected elsewhere in the unit test file will not work). Additionally,
 822       checks that nothing still has a reference to Tensors that the test
 823       allocated.

 826     once by constructing and executing a graph in a session and once with

 831     if tf_inspect.isclass(f):

 833           "`run_test_in_graph_and_eager_modes` only supports test methods. "
 834           "Did you mean to use `run_all_tests_in_graph_and_eager_modes`?")

 837       try:

 840             f(self, **kwargs)
 841       except unittest.case.SkipTest:
 842         pass
 843 
 844       def run_eagerly(self, **kwargs):
 845         if not use_gpu:
 846           with ops.device("/device:CPU:0"):
 847             f(self, **kwargs)
 848         else:
 849           f(self, **kwargs)
 850 
 851       if assert_no_eager_garbage:
 852         ops.reset_default_graph()

 854             assert_no_garbage_created(run_eagerly))
 855

 857         # This decorator runs the wrapped test twice.
 858         # Reset the test environment between runs.
 859         self.tearDown()

 862       # better isolation.

 873     return decorator(func)

1453   def _assertArrayLikeAllClose(self, a, b, rtol=1e-6, atol=1e-6, msg=None):
1454     a = self._GetNdArray(a)
1455     b = self._GetNdArray(b)
1456     # When the array rank is small, print its contents. Numpy array printing is
1457     # implemented using inefficient recursion so prints can cause tests to
1458     # time out.

1460       shape_mismatch_msg = ("Shape mismatch: expected %s, got %s with contents "

1464                                                                      b.shape)

1466 
1467     msgs = [msg]
1468     if not np.allclose(a, b, rtol=rtol, atol=atol):
1469       # Adds more details to np.testing.assert_allclose.
1470       #
1471       # NOTE: numpy.allclose (and numpy.testing.assert_allclose)
1472       # checks whether two arrays are element-wise equal within a
1473       # tolerance. The relative difference (rtol * abs(b)) and the
1474       # absolute difference atol are added together to compare against
1475       # the absolute difference between a and b.  Here, we want to
1476       # tell user which elements violate such conditions.

1478           np.abs(a - b) > atol + rtol * np.abs(b),

1483         msgs.append("not close where = {}".format(np.where(cond)))
1484       else:
1485         # np.where is broken for scalars
1486         x, y = a, b
1487       msgs.append("not close lhs = {}".format(x))
1488       msgs.append("not close rhs = {}".format(y))
1489       msgs.append("not close dif = {}".format(np.abs(x - y)))
1490       msgs.append("not close tol = {}".format(atol + rtol * np.abs(y)))
1491       msgs.append("dtype = {}, shape = {}".format(a.dtype, a.shape))
1492       # TODO(xpan): There seems to be a bug:
1493       # tensorflow/compiler/tests:binary_ops_test pass with float32
1494       # nan even though the equal_nan is False by default internally.
1495       np.testing.assert_allclose(
1496           a, b, rtol=rtol, atol=atol, err_msg="\n".join(msgs), equal_nan=True)

1812         np.less_equal(target, lower_bound)
1813         if open_lower_bound else np.less(target, lower_bound))

1815         violations,
1816         np.greater_equal(target, upper_bound)
1817         if open_upper_bound else np.greater(target, upper_bound))
