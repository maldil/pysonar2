  46 class CoordinatorMode(object):
  47   """Specify how distribute coordinator runs."""
  48   # The default mode where distribute coordinator will run as a standalone
  49   # client and connects to remote servers for training.  Each remote server can
  50   # use the distribute coordinator binary with task_type set correctly which
  51   # will then turn into standard servers.
  52   STANDALONE_CLIENT = "standalone_client"
  53 
  54   # The distribute coordinator runs on each worker. It will run a standard
  55   # server on each worker and optionally run the `worker_fn` that is configured
  56   # to talk to its standard server.
  57   INDEPENDENT_WORKER = "independent_worker"

  60 class _Barrier(object):
  61   """A reusable barrier class for worker synchronization."""
  62 
  63   def __init__(self, num_participants):
  64     """Initializes the barrier object.
  65 
  66     Args:
  67       num_participants: an integer which is the expected number of calls of
  68         `wait` pass to through this barrier.
  69     """
  70     self._num_participants = num_participants
  71     self._counter = 0
  72     self._flag = False
  73     self._local_sense = threading.local()
  74     self._lock = threading.Lock()
  75     self._condition = threading.Condition()
  76 
  77   def wait(self):
  78     """Waits until all other callers reach the same wait call."""
  79     if not hasattr(self._local_sense, "value"):
  80       self._local_sense.value = False
  81     self._local_sense.value = not self._flag
  82     with self._lock:
  83       self._counter += 1
  84       if self._counter == self._num_participants:
  85         self._counter = 0
  86         self._flag = self._local_sense.value
  87     with self._condition:
  88       while self._flag != self._local_sense.value:
  89         self._condition.wait()
  90       self._condition.notify_all()

  93 def _get_num_workers(cluster_spec):
  94   """Gets number of workers including chief."""
  95   if not cluster_spec:
  96     return 0
  97   return len(cluster_spec.as_dict().get(_TaskType.WORKER, [])) + len(
  98       cluster_spec.as_dict().get(_TaskType.CHIEF, []))

 101 class _WorkerContext(object):
 102   """The worker context class.
 103 
 104   This context object provides configuration information for each task. One
 105   context manager with a worker context object will be created per
 106   invocation to the `worker_fn` where `get_current_worker_context` can be called
 107   to access the worker context object.
 108   """
 109 
 110   def __init__(self,
 111                strategy,
 112                cluster_spec,
 113                task_type,
 114                task_id,
 115                session_config=None,
 116                rpc_layer="grpc",
 117                worker_barrier=None):
 118     """Initialize the worker context object.
 119 
 120     Args:
 121       strategy: a `DistributionStrategy` object.
 122       cluster_spec: a ClusterSpec object. It can be empty or None in the local
 123         training case.
 124       task_type: a string indicating the role of the corresponding task, such as
 125         "worker" or "ps". It can be None if it is local training or in-graph
 126         replicated training.
 127       task_id: an integer indicating id of the corresponding task. It can be
 128         None if it is local training or in-graph replicated training.
 129       session_config: an optional `tf.ConfigProto` object.
 130       rpc_layer: optional string specifying the RPC protocol for communication
 131         with worker masters. If None or empty, hosts in the `cluster_spec` will
 132         be used directly.
 133       worker_barrier: optional, the barrier object for worker synchronization.
 134     """
 135     self._strategy = strategy
 136     self._cluster_spec = cluster_spec
 137     self._task_type = task_type
 138     self._task_id = task_id
 139     self._session_config = session_config
 140     self._worker_barrier = worker_barrier
 141     self._rpc_layer = rpc_layer
 142     self._master_target = self._get_master_target()
 143     self._num_workers = _get_num_workers(cluster_spec)
 144     self._is_chief_node = self._is_chief()
 145 
 146   def _debug_message(self):
 147     if self._cluster_spec:
 148       return "[cluster_spec: %r, task_type: %r, task_id: %r]" % (
 149           self._cluster_spec, self.task_type, self.task_id)
 150     else:
 151       return "[local]"
 152 
 153   def __enter__(self):
 154     old_context = distribute_coordinator_context.get_current_worker_context()
 155     if old_context:
 156       raise ValueError(
 157           "You cannot run distribute coordinator in a `worker_fn`.\t" +
 158           self._debug_message())
 159     # pylint: disable=protected-access
 160     distribute_coordinator_context._worker_context.current = self
 161 
 162   def __exit__(self, unused_exception_type, unused_exception_value,
 163                unused_traceback):
 164     # pylint: disable=protected-access
 165     distribute_coordinator_context._worker_context.current = None
 166 
 167   def _get_master_target(self):
 168     """Return the master target for a task."""
 169     # If cluster_spec is None or empty, we use local master.
 170     if not self._cluster_spec:
 171       return ""
 172 
 173     # If task_type is None, then it is in-graph replicated training. In this
 174     # case we use the chief or first worker's master target.
 175     if not self._task_type:
 176       if _TaskType.CHIEF in self._cluster_spec.jobs:
 177         task_type = _TaskType.CHIEF
 178         task_id = 0
 179       else:
 180         assert _TaskType.WORKER in self._cluster_spec.jobs
 181         task_type = _TaskType.WORKER
 182         task_id = 0
 183     else:
 184       task_type = self._task_type
 185       task_id = self._task_id
 186 
 187     prefix = ""
 188     if self._rpc_layer:
 189       prefix = self._rpc_layer + "://"
 190     return prefix + self._cluster_spec.job_tasks(task_type)[task_id or 0]
 191 
 192   def _is_chief(self):
 193     """Return whether the task is the chief worker."""
 194     if (not self._cluster_spec or
 195         self._task_type in [_TaskType.CHIEF, _TaskType.EVALUATOR, None]):
 196       return True
 197 
 198     # If not local and chief not in the cluster_spec, use the first worker as
 199     # chief.
 200     if (_TaskType.CHIEF not in self._cluster_spec.jobs and
 201         self._task_type == _TaskType.WORKER and self._task_id == 0):
 202       return True
 203     return False

 205   def wait_for_other_workers(self):
 206     """Waits for other workers to reach the same call to this method.
 207 
 208     Raises:
 209       ValueError: if `worker_barrier` is not passed to the __init__ method.
 210     """
 211     if not self._worker_barrier:
 212       raise ValueError("`worker_barrier is not set in the worker context.` \t" +
 213                        self._debug_message())
 214     self._worker_barrier.wait()

 216   def session_creator(self,
 217                       scaffold=None,
 218                       config=None,
 219                       checkpoint_dir=None,
 220                       checkpoint_filename_with_path=None,
 221                       max_wait_secs=7200):
 222     """Returns a session creator.
 223 
 224     The returned session creator will be configured with the correct master
 225     target and session configs. It will also run either init ops or ready ops
 226     by querying the `strategy` object when `create_session` is called on it.
 227 
 228     Args:
 229       scaffold: A `Scaffold` used for gathering or building supportive ops. If
 230         not specified a default one is created. It's used to finalize the graph.
 231       config: `ConfigProto` proto used to configure the session.
 232       checkpoint_dir: A string. Optional path to a directory where to restore
 233         variables.
 234       checkpoint_filename_with_path: Full file name path to the checkpoint file.
 235         Only one of `checkpoint_dir` or `checkpoint_filename_with_path` can be
 236         specified.
 237       max_wait_secs: Maximum time to wait for the session to become available.
 238 
 239     Returns:
 240       a descendant of SessionCreator.
 241     """
 242     if config:
 243       session_config = copy.deepcopy(config)
 244       session_config.MergeFrom(self._session_config)
 245     else:
 246       session_config = self._session_config
 247 
 248     if not self._strategy or self._strategy.should_init:
 249       logging.info("Creating chief session creator with config: %r", config)
 250       return monitored_session.ChiefSessionCreator(
 251           scaffold,
 252           master=self.master_target,
 253           config=session_config,
 254           checkpoint_dir=checkpoint_dir,
 255           checkpoint_filename_with_path=checkpoint_filename_with_path)
 256     else:
 257       logging.info("Creating worker session creator with config: %r", config)
 258       return monitored_session.WorkerSessionCreator(
 259           scaffold,
 260           master=self.master_target,
 261           config=session_config,
 262           max_wait_secs=max_wait_secs)

 315 def _run_single_worker(worker_fn,
 316                        strategy,
 317                        cluster_spec,
 318                        task_type,
 319                        task_id,
 320                        session_config,
 321                        rpc_layer="",
 322                        worker_barrier=None):
 323   """Runs a single worker by calling `worker_fn` under context."""
 324   session_config = copy.deepcopy(session_config)
 325   strategy = copy.deepcopy(strategy)
 326   # If there is an EVALUATOR task, we run single-machine eval on that task.
 327   if task_type == _TaskType.EVALUATOR:
 328     # It is possible to not have a strategy object for EVALUATOR task.
 329     if strategy:
 330       strategy.configure(session_config)
 331   else:
 332     assert strategy
 333     strategy.configure(session_config, cluster_spec, task_type, task_id)
 334 
 335   context = _WorkerContext(
 336       strategy,
 337       cluster_spec,
 338       task_type,
 339       task_id,
 340       session_config=session_config,
 341       rpc_layer=rpc_layer,
 342       worker_barrier=worker_barrier)
 343   with context:
 344     worker_fn(strategy)

 347 def _split_cluster_for_evaluator(cluster_spec, task_type):
 348   """Split the cluster for evaluator since it needn't talk to other tasks."""
 349   # Splitting the cluster is important to prevent the evaluator from talking to
 350   # other tasks in the cluster. Since we allow evaluator not to use
 351   # distribution strategies and as a result ops in the evalauator task may have
 352   # unspecified devices. Those ops may end up on other tasks if we don't split
 353   # the cluster.
 354   new_cluster_spec = multi_worker_util.normalize_cluster_spec(
 355       cluster_spec).as_dict()
 356   if task_type == _TaskType.EVALUATOR:
 357     assert _TaskType.EVALUATOR in new_cluster_spec
 358     new_cluster_spec = {
 359         _TaskType.EVALUATOR: new_cluster_spec[_TaskType.EVALUATOR]
 360     }
 361   else:
 362     new_cluster_spec.pop(_TaskType.EVALUATOR, None)
 363   return multi_worker_util.normalize_cluster_spec(new_cluster_spec)

 366 def _run_std_server(cluster_spec=None,
 367                     task_type=None,
 368                     task_id=None,
 369                     session_config=None,
 370                     rpc_layer=None,
 371                     environment=None):
 372   """Runs a standard server."""
 373   assert cluster_spec
 374   target = cluster_spec.task_address(task_type, task_id)
 375   if rpc_layer:
 376     target = rpc_layer + "://" + target

 378   class _FakeServer(object):
 379     """A fake server that runs a master session."""
 380 
 381     def start(self):
 382       # A tensorflow server starts when a remote session is created.
 383       logging.info(
 384           "Creating a remote session to start a TensorFlow server, "
 385           "target = %r, session_config=%r", target, session_config)
 386       session.Session(target=target, config=session_config)
 387 
 388     def join(self):
 389       while True:
 390         time.sleep(5)
 391 
 392   if environment == "google":
 393     server = _FakeServer()
 394     server.start()
 395     return server
 396   else:
 397     if session_config:
 398       logging.info(
 399           "Starting standard TensorFlow server, target = %r, session_config= "
 400           "%r", target, session_config)
 401     else:
 402       logging.info("Starting standard TensorFlow server, target = %r", target)
 403     cluster_spec = _split_cluster_for_evaluator(cluster_spec, task_type)
 404     server = server_lib.Server(
 405         cluster_spec,
 406         job_name=task_type,
 407         task_index=task_id,
 408         config=session_config,
 409         protocol=rpc_layer)
 410     server.start()
 411     return server

 414 def _run_between_graph_client(worker_fn, strategy, eval_fn, eval_strategy,
 415                               cluster_spec, session_config, rpc_layer):
 416   """Runs a standalone client for between-graph replication."""
 417   eval_thread = None
 418   if _TaskType.EVALUATOR in cluster_spec.jobs:
 419     eval_thread = threading.Thread(
 420         target=_run_single_worker,
 421         args=(eval_fn, eval_strategy, cluster_spec, _TaskType.EVALUATOR, 0,
 422               session_config),
 423         kwargs={
 424             "rpc_layer": rpc_layer,
 425         })
 426     eval_thread.start()
 427 
 428   threads = []
 429   worker_barrier = _Barrier(_get_num_workers(cluster_spec))
 430   for task_type in [_TaskType.CHIEF, _TaskType.WORKER]:
 431     for task_id in range(len(cluster_spec.as_dict().get(task_type, []))):
 432       t = threading.Thread(
 433           target=_run_single_worker,
 434           args=(worker_fn, strategy, cluster_spec, task_type, task_id,
 435                 session_config),
 436           kwargs={
 437               "rpc_layer": rpc_layer,
 438               "worker_barrier": worker_barrier
 439           })
 440       t.start()
 441       threads.append(t)
 442 
 443   # TODO(yuefengz): wrap threads into thread coordinator?
 444   for t in threads:
 445     t.join()
 446 
 447   # TODO(yuefengz): is it necessary to join eval thread?
 448   if eval_thread:
 449     eval_thread.join()

 452 def _run_in_graph_client(worker_fn, strategy, eval_fn, eval_strategy,
 453                          cluster_spec, session_config, rpc_layer):
 454   """Runs a standalone client for in-graph replication."""
 455   eval_thread = None
 456   if _TaskType.EVALUATOR in cluster_spec.jobs:
 457     eval_thread = threading.Thread(
 458         target=_run_single_worker,
 459         args=(eval_fn, eval_strategy, cluster_spec, _TaskType.EVALUATOR, 0,
 460               session_config),
 461         kwargs={
 462             "rpc_layer": rpc_layer,
 463         })
 464     eval_thread.start()
 465 
 466   _run_single_worker(
 467       worker_fn,
 468       strategy,
 469       cluster_spec,
 470       None,
 471       None,
 472       session_config,
 473       rpc_layer=rpc_layer)
 474   if eval_thread:
 475     eval_thread.join()

 478 def _configure_session_config_for_std_servers(
 479     strategy, eval_strategy, session_config, cluster_spec, task_type, task_id):
 480   # pylint: disable=g-doc-args
 481   """Call strategy's `configure` to mutate the session_config.
 482 
 483   The session_config is currently needed as default config for a TensorFlow
 484   server. In the future, we should be able to remove this method and only pass
 485   the session config to a client session.
 486   """
 487   if task_type == _TaskType.EVALUATOR:
 488     if eval_strategy:
 489       eval_strategy.configure(session_config=session_config)
 490   else:
 491     # The strategy may be shared in standalone client mode.
 492     strategy = copy.deepcopy(strategy)
 493     strategy.configure(
 494         session_config=session_config,
 495         cluster_spec=cluster_spec,
 496         task_type=task_type,
 497         task_id=task_id)
 498   # Remove the device filters specific to the strategy, so that the
 499   # TensorFlow server brought up with one strategy can be used by other
 500   # strategies. The device filters can be set in the client side as well.
 501   del session_config.device_filters[:]

 504 def run_standard_tensorflow_server(session_config=None):
 505   """Starts a standard TensorFlow server.
 506 
 507   This method parses configurations from "TF_CONFIG" environment variable and
 508   starts a TensorFlow server. The "TF_CONFIG" is typically a json string and
 509   must have information of the cluster and the role of the server in the
 510   cluster. One example is:
 511 
 512   TF_CONFIG='{
 513       "cluster": {
 514           "worker": ["host1:2222", "host2:2222", "host3:2222"],
 515           "ps": ["host4:2222", "host5:2222"]
 516       },
 517       "task": {"type": "worker", "index": 1}
 518   }'
 519 
 520   This "TF_CONFIG" specifies there are 3 workers and 2 ps tasks in the cluster
 521   and the current role is worker 1.
 522 
 523   Valid task types are "chief", "worker", "ps" and "evaluator" and you can have
 524   at most one "chief" and at most one "evaluator".
 525 
 526   An optional key-value can be specified is "rpc_layer". The default value is
 527   "grpc".
 528 
 529   Args:
 530     session_config: an optional `tf.ConfigProto` object. Users can pass in
 531       the session config object to configure server-local devices.
 532 
 533   Returns:
 534     a `tf.train.Server` object which has already been started.
 535 
 536   Raises:
 537     ValueError: if the "TF_CONFIG" environment is not complete.
 538   """
 539   tf_config = json.loads(os.environ.get("TF_CONFIG", "{}"))
 540   if "cluster" not in tf_config:
 541     raise ValueError("\"cluster\" is not found in TF_CONFIG.")
 542   cluster_spec = multi_worker_util.normalize_cluster_spec(tf_config["cluster"])
 543   if "task" not in tf_config:
 544     raise ValueError("\"task\" is not found in TF_CONFIG.")
 545   task_env = tf_config["task"]
 546   if "type" not in task_env:
 547     raise ValueError(
 548         "\"task_type\" is not found in the `task` part of TF_CONFIG.")
 549   task_type = task_env["type"]
 550   task_id = int(task_env.get("index", 0))
 551 
 552   rpc_layer = tf_config.get("rpc_layer", "grpc")
 553 
 554   session_config = session_config or config_pb2.ConfigProto()
 555   # Set the collective group leader for collective ops to initialize collective
 556   # ops when server starts.
 557   if "chief" in cluster_spec.jobs:
 558     session_config.experimental.collective_group_leader = (
 559         "/job:chief/replica:0/task:0")
 560   else:
 561     if "worker" not in cluster_spec.jobs:
 562       raise ValueError(
 563           "You must have `chief` or `worker` jobs in the `cluster_spec`.")
 564     session_config.experimental.collective_group_leader = (
 565         "/job:worker/replica:0/task:0")
 566 
 567   server = _run_std_server(
 568       cluster_spec=cluster_spec,
 569       task_type=task_type,
 570       task_id=task_id,
 571       session_config=session_config,
 572       rpc_layer=rpc_layer)
 573   server.start()
 574   return server

 577 # TODO(yuefengz): propagate cluster_spec in the STANDALONE_CLIENT mode.

 579 # is the special task when we support cluster_spec propagation.
 580 def run_distribute_coordinator(worker_fn,
 581                                strategy,
 582                                eval_fn=None,
 583                                eval_strategy=None,
 584                                mode=CoordinatorMode.STANDALONE_CLIENT,
 585                                cluster_spec=None,
 586                                task_type=None,
 587                                task_id=None,
 588                                session_config=None,
 589                                rpc_layer="grpc"):
 590   """Runs the coordinator for distributed TensorFlow.
 591 
 592   This function runs a split coordinator for distributed TensorFlow in its
 593   default mode, i.e the STANDALONE_CLIENT mode. Given a `cluster_spec`
 594   specifying server addresses and their roles in a cluster, this coordinator
 595   will figure out how to set them up, give the underlying function the right
 596   targets for master sessions via a scope object and coordinate their training.
 597   The cluster consisting of standard servers needs to be brought up either with
 598   the standard server binary or with a binary running distribute coordinator
 599   with `task_type` set to non-client type which will then turn into standard
 600   servers.
 601 
 602   In addition to be the distribute coordinator, this is also the source of
 603   configurations for each job in the distributed training. As there are multiple
 604   ways to configure a distributed TensorFlow cluster, its context object
 605   provides these configurations so that users or higher-level APIs don't have to
 606   figure out the configuration for each job by themselves.
 607 
 608   In the between-graph replicated training, this coordinator will create
 609   multiple threads and each calls the `worker_fn` which is supposed to create
 610   its own graph and connect to one worker master given by its context object. In
 611   the in-graph replicated training, it has only one thread calling this
 612   `worker_fn`.
 613 
 614   Another mode is the INDEPENDENT_WORKER mode where each server runs a
 615   distribute coordinator which will start a standard server and optionally runs
 616   `worker_fn` depending whether it is between-graph training or in-graph
 617   replicated training.
 618 
 619   The `strategy` object is expected to be a DistributionStrategy object which
 620   has implemented methods needed by distributed coordinator such as
 621   `configure(session_config, cluster_spec, task_type, task_id)` which configures
 622   the strategy object for a specific task and `should_init` property which
 623   instructs the distribute coordinator whether to run init ops for a task. The
 624   distribute coordinator will make a copy of the `strategy` object, call its
 625   `configure` method and pass it to `worker_fn` as an argument.
 626 
 627   The `worker_fn` defines the training logic and is called under a its own
 628   worker context which can be accessed to via `get_current_worker_context`. A
 629   worker context provides access to configurations for each task, e.g. the
 630   task_type, task_id, master target and so on. Since `worker_fn` will be called
 631   in a thread and possibly multiple times, caller should be careful when it
 632   accesses global data. For example, it is unsafe to define flags in a
 633   `worker_fn` or to define different environment variables for different
 634   `worker_fn`s.
 635 
 636   The `worker_fn` for the between-graph replication is defined as if there is
 637   only one worker corresponding to the `worker_fn` and possibly ps jobs. For
 638   example, when training with parameter servers, it assigns variables to
 639   parameter servers and all other operations to that worker. In the in-graph
 640   replication case, the `worker_fn` has to define operations for all worker
 641   jobs. Using a distribution strategy can simplify the `worker_fn` by not having
 642   to worry about the replication and device assignment of variables and
 643   operations.
 644 
 645   This method is intended to be invoked by high-level APIs so that users don't
 646   have to explictly call it to run this coordinator. For those who don't use
 647   high-level APIs, to change a program to use this coordinator, wrap everything
 648   in a the program after global data definitions such as commandline flag
 649   definition into the `worker_fn` and get task-specific configurations from
 650   the worker context.
 651 
 652   The `cluster_spec` can be either passed by the argument or parsed from the
 653   "TF_CONFIG" envrionment variable. Example of a TF_CONFIG:
 654   ```
 655     cluster = {'chief': ['host0:2222'],
 656                'ps': ['host1:2222', 'host2:2222'],
 657                'worker': ['host3:2222', 'host4:2222', 'host5:2222']}
 658     os.environ['TF_CONFIG'] = json.dumps({'cluster': cluster})
 659   ```
 660 
 661   If `cluster_spec` is not given in any format, it becomes local training and
 662   this coordinator will connect to a local session.
 663 
 664   For evaluation, if "evaluator" exists in the cluster_spec, a separate thread
 665   will be created to call `eval_fn` with its `task_type` set to "evaluator". If
 666   `eval_fn` is not defined, fall back to `worker_fn`. This implies that
 667   evaluation will be done on a single machine if there is an "evaluator" task.
 668   If "evaluator" doesn't exit in the cluster_spec, it entirely depends on the
 669   `worker_fn` for how to do evaluation.
 670 
 671   Args:
 672     worker_fn: the function to be called. The function should accept a
 673       `strategy` object and will be given access to a context object via a
 674       context manager scope.
 675     strategy: a DistributionStrategy object which specifying whether it should
 676       run between-graph replicated training or not, whether to run init ops,
 677       etc. This object will also be configured given `session_config`,
 678       `cluster_spec`, `task_type` and `task_id`.
 679     eval_fn: optional function for "evaluator" task. If `eval_fn` is not passed
 680       in but a "evaluator" task found in the `cluster_spec`, the `worker_fn`
 681       will be used for this task.
 682     eval_strategy: optional DistributionStrategy object for "evaluator" task.
 683     mode: in which mode this distribute coordinator runs.
 684     cluster_spec: a dict, ClusterDef or ClusterSpec specifying servers and roles
 685       in a cluster. If not set or empty, fall back to local training.
 686     task_type: the current task type, optional if this is a client.
 687     task_id: the current task id, optional if this is a client.
 688     session_config: an optional `tf.ConfigProto` object which will be passed
 689       to `strategy`'s `configure` method and used to create a session.
 690     rpc_layer: optional string, the protocol for RPC, e.g. "grpc".
 691 
 692   Raises:
 693     ValueError: if `cluster_spec` is supplied but not a dict or a ClusterDef or
 694       a ClusterSpec.
 695   """
 696   tf_config = json.loads(os.environ.get("TF_CONFIG", "{}"))
 697   if not cluster_spec:
 698     cluster_spec = tf_config.get("cluster", {})
 699     task_env = tf_config.get("task", {})
 700     if task_env:
 701       task_type = task_env.get("type", task_type)
 702       task_id = int(task_env.get("index", task_id))
 703 
 704   if cluster_spec:
 705     cluster_spec = multi_worker_util.normalize_cluster_spec(cluster_spec)
 706     # TODO(yuefengz): validate cluster_spec.
 707 
 708   rpc_layer = tf_config.get("rpc_layer", rpc_layer)
 709   environment = tf_config.get("environment", None)
 710 
 711   # Setting the session config is necessary for some strategies such
 712   # CollectiveAllReduceStrategy.
 713   session_config = session_config or config_pb2.ConfigProto(
 714       allow_soft_placement=True)
 715 
 716   if cluster_spec:
 717     logging.info(
 718         "Running Distribute Coordinator with mode = %r, cluster_spec = %r, "
 719         "task_type = %r, task_id = %r, environment = %r, rpc_layer = %r", mode,
 720         cluster_spec.as_dict(), task_type, task_id, environment, rpc_layer)
 721 
 722   if not cluster_spec:
 723     # `mode` is ignored in the local case.
 724     logging.info("Running local Distribute Coordinator.")
 725     _run_single_worker(worker_fn, strategy, None, None, None, session_config,
 726                        rpc_layer)
 727     if eval_fn:
 728       _run_single_worker(eval_fn, eval_strategy, None, None, None,
 729                          session_config, rpc_layer)
 730     else:
 731       logging.warning("Skipped evaluation since `eval_fn` is not passed in.")
 732   elif mode == CoordinatorMode.STANDALONE_CLIENT:
 733     if not eval_fn:
 734       logging.warning("`eval_fn` is not passed in. The `worker_fn` will be "
 735                       "used if an \"evaluator\" task exists in the cluster.")
 736     eval_fn = eval_fn or worker_fn
 737     if not eval_strategy:
 738       logging.warning("`eval_strategy` is not passed in. No distribution "
 739                       "strategy will be used for evaluation.")
 740 
 741     # The client must know the cluster but servers in the cluster don't have to
 742     # know the client.
 743     if task_type in [_TaskType.CLIENT, None]:
 744       if strategy.between_graph:
 745         _run_between_graph_client(worker_fn, strategy, eval_fn, eval_strategy,
 746                                   cluster_spec, session_config, rpc_layer)
 747       else:
 748         _run_in_graph_client(worker_fn, strategy, eval_fn, eval_strategy,
 749                              cluster_spec, session_config, rpc_layer)
 750     else:
 751       # If not a client job, run the standard server.
 752       _configure_session_config_for_std_servers(strategy, eval_strategy,
 753                                                 session_config, cluster_spec,
 754                                                 task_type, task_id)
 755       server = _run_std_server(
 756           cluster_spec=cluster_spec,
 757           task_type=task_type,
 758           task_id=task_id,
 759           session_config=session_config,
 760           rpc_layer=rpc_layer,
 761           environment=environment)
 762       server.join()
 763   else:
 764     if mode != CoordinatorMode.INDEPENDENT_WORKER:
 765       raise ValueError("Unexpected coordinator mode: %r" % mode)
 766 
 767     if not eval_fn:
 768       logging.warning("`eval_fn` is not passed in. The `worker_fn` will be "
 769                       "used if an \"evaluator\" task exists in the cluster.")
 770     eval_fn = eval_fn or worker_fn
 771     if not eval_strategy:
 772       logging.warning("`eval_strategy` is not passed in. No distribution "
 773                       "strategy will be used for evaluation.")
 774 
 775     # Every one starts a standard server, get session config from `configure`
 776     # method.
 777     _configure_session_config_for_std_servers(strategy, eval_strategy,
 778                                               session_config, cluster_spec,
 779                                               task_type, task_id)
 780     server = _run_std_server(
 781         cluster_spec=cluster_spec,
 782         task_type=task_type,
 783         task_id=task_id,
 784         session_config=session_config,
 785         rpc_layer=rpc_layer,
 786         environment=environment)
 787 
 788     if task_type in [_TaskType.CHIEF, _TaskType.WORKER]:
 789       if strategy.between_graph:
 790         # All jobs run `worker_fn` if between-graph.
 791         _run_single_worker(worker_fn, strategy, cluster_spec, task_type,
 792                            task_id, session_config, rpc_layer)
 793       else:
 794         # Only one node runs `worker_fn` if in-graph.
 795         context = _WorkerContext(strategy, cluster_spec, task_type, task_id)
 796         if context.is_chief:
 797           _run_single_worker(worker_fn, strategy, cluster_spec, None, None,
 798                              session_config, rpc_layer)
 799         else:
 800           server.join()
 801     elif task_type == _TaskType.EVALUATOR:
 802       _run_single_worker(eval_fn, eval_strategy, cluster_spec, task_type,
 803                          task_id, session_config, rpc_layer)
 804     else:
 805       if task_type != _TaskType.PS:
 806         raise ValueError("Unexpected task_type: %r" % task_type)
 807       server.join()
