  39 def _count_ps(cluster_spec):
  40   """Counts the number of parameter servers in cluster_spec."""
  41   if not cluster_spec:
  42     raise RuntimeError(
  43         'Internal error: `_count_ps` does not expect empty cluster_spec.')
  44 
  45   return len(cluster_spec.as_dict().get(PS, []))
  46 
  47 
  48 def _count_worker(cluster_spec, chief_task_type):
  49   """Counts the number of workers (including chief) in cluster_spec."""
  50   if not cluster_spec:
  51     raise RuntimeError(
  52         'Internal error: `_count_worker` does not expect empty cluster_spec.')
  53 
  54   return (len(cluster_spec.as_dict().get(WORKER, [])) + len(
  55       cluster_spec.as_dict().get(chief_task_type, [])))

  58 def _get_global_id(cluster_spec, task_type, task_id, chief_task_type):
  59   """Returns the global id of the given task type in a cluster."""
  60   if not task_type:
  61     return 0
  62 
  63   # Sort task names in cluster by "chief"/"master", "evaluator", "worker"
  64   # and "ps". More details can be found at the documentation of
  65   # `tf.estimator.RunConfig.global_id_in_cluster`.
  66   task_type_ordered_list = []
  67   if chief_task_type in cluster_spec.jobs:
  68     task_type_ordered_list = [chief_task_type]
  69   task_type_ordered_list.extend([
  70       t for t in sorted(cluster_spec.jobs) if t != chief_task_type and t != PS
  71   ])
  72   if PS in cluster_spec.jobs:
  73     task_type_ordered_list.append(PS)
  74 
  75   # Find the right gloabl_id for current task.
  76   next_global_id = 0
  77   for t in task_type_ordered_list:
  78     if t == task_type:
  79       return next_global_id + task_id
  80     # `cluster_spec.job_tasks` returns all task addresses of type `t`.
  81     next_global_id += len(cluster_spec.job_tasks(t))
  82 
  83   # It is unexpected that it passes through all task_types in
  84   # `task_type_ordered_list`.
  85   raise RuntimeError('Internal Error: `task_type` ({}) is not in '
  86                      'cluster_spec ({}).'.format(task_type, cluster_spec))

  89 def _init_run_config_from_worker_context(config, worker_context):
  90   """Initializes run config from distribute coordinator's worker context."""
  91 
  92   # pylint: disable=protected-access
  93   config._service = None
  94   config._cluster_spec = worker_context.cluster_spec
  95   config._task_type = worker_context.task_type
  96   config._task_id = worker_context.task_id
  97   config._evaluation_master = worker_context.master_target
  98   config._master = worker_context.master_target
  99   config._is_chief = worker_context.is_chief
 100 
 101   if config._cluster_spec:
 102     # Distributed mode.
 103     if config._task_type != EVALUATOR:
 104 
 105       config._num_ps_replicas = _count_ps(config._cluster_spec)
 106       config._num_worker_replicas = _count_worker(
 107           config._cluster_spec, chief_task_type=CHIEF)
 108       config._global_id_in_cluster = _get_global_id(
 109           config._cluster_spec,
 110           config._task_type,
 111           config._task_id,
 112           chief_task_type=CHIEF)
 113     else:
 114       # Evaluator task should not be aware of the other tasks.
 115       config._cluster_spec = server_lib.ClusterSpec({})
 116       config._num_ps_replicas = 0
 117       config._num_worker_replicas = 0
 118       config._global_id_in_cluster = None  # undefined
 119   else:
 120     # Local mode.
 121     config._global_id_in_cluster = 0
 122     config._num_ps_replicas = 0
 123     config._num_worker_replicas = 1

 126 def init_run_config(config, tf_config):
 127   """Initializes RunConfig for distribution strategies."""
 128   # pylint: disable=protected-access
 129   if (config._experimental_distribute and
 130       config._experimental_distribute.train_distribute):
 131     if config._train_distribute:
 132       raise ValueError('Either `train_distribute` or'
 133                        '`experimental_distribute.train_distribute` can be set.')
 134     config._train_distribute = config._experimental_distribute.train_distribute
 135 
 136   if (config._experimental_distribute and
 137       config._experimental_distribute.eval_distribute):
 138     if config._eval_distribute:
 139       raise ValueError('Either `eval_distribute` or'
 140                        '`experimental_distribute.eval_distribute` can be set.')
 141     config._eval_distribute = config._experimental_distribute.eval_distribute
 142 
 143   cluster_spec = server_lib.ClusterSpec(tf_config.get('cluster', {}))
 144   config._init_distributed_setting_from_environment_var({})
 145 
 146   # Use distribute coordinator with STANDALONE_CLIENT mode if
 147   # `experimental_distribute.remote_cluster` is set.
 148   if (config._train_distribute and config._experimental_distribute and
 149       config._experimental_distribute.remote_cluster):
 150     if cluster_spec:
 151       raise ValueError('Cannot set both "cluster_spec" of TF_CONFIG and '
 152                        '`experimental_distribute.remote_cluster`')
 153     config._distribute_coordinator_mode = dc.CoordinatorMode.STANDALONE_CLIENT
 154     config._cluster_spec = config._experimental_distribute.remote_cluster
 155     logging.info('RunConfig initialized for Distribute Coordinator with '
 156                  'STANDALONE_CLIENT mode')
 157     return
 158 
 159   # Don't use distribute coordinator if it is local training or cluster has a
 160   # MASTER job or `train_distribute` is not specifed.
 161   if (not tf_config or 'master' in cluster_spec.jobs or
 162       not config._train_distribute):
 163     config._distribute_coordinator_mode = None
 164     config._init_distributed_setting_from_environment_var(tf_config)
 165     config._maybe_overwrite_session_config_for_distributed_training()
 166     logging.info('Not using Distribute Coordinator.')
 167     return
 168 
 169   # Use distribute coordinator with INDEPENDENT_WORKER mode otherwise.
 170   assert tf_config
 171 
 172   # Set the cluster_spec only since the distributed setting will come from
 173   # distribute coordinator.
 174   config._cluster_spec = cluster_spec
 175   config._distribute_coordinator_mode = dc.CoordinatorMode.INDEPENDENT_WORKER
 176   logging.info('RunConfig initialized for Distribute Coordinator with '
 177                'INDEPENDENT_WORKER mode')

 180 def should_run_distribute_coordinator(config):
 181   """Checks the config to see whether to run distribute coordinator."""
 182   # pylint: disable=protected-access
 183   if (not hasattr(config, '_distribute_coordinator_mode') or
 184       config._distribute_coordinator_mode is None):
 185     logging.info('Not using Distribute Coordinator.')
 186     return False
 187   if (not isinstance(config._distribute_coordinator_mode, six.string_types) or
 188       config._distribute_coordinator_mode not in [
 189           dc.CoordinatorMode.STANDALONE_CLIENT,
 190           dc.CoordinatorMode.INDEPENDENT_WORKER
 191       ]):
 192     logging.warning('Unexpected distribute_coordinator_mode: %r',
 193                     config._distribute_coordinator_mode)
 194     return False
 195   if not config.cluster_spec:
 196     logging.warning('Running `train_and_evaluate` locally, ignoring '
 197                     '`experimental_distribute_coordinator_mode`.')
 198     return False
 199   return True

 202 def train_and_evaluate(estimator, train_spec, eval_spec, executor_cls):
 203   """Run distribute coordinator for Estimator's `train_and_evaluate`.
 204 
 205   Args:
 206     estimator: An `Estimator` instance to train and evaluate.
 207     train_spec: A `TrainSpec` instance to specify the training specification.
 208     eval_spec: A `EvalSpec` instance to specify the evaluation and export
 209       specification.
 210     executor_cls: the evaluation executor class of Estimator.
 211 
 212   Raises:
 213     ValueError: if `distribute_coordinator_mode` is None in RunConfig.
 214   """
 215   run_config = estimator.config
 216   if not run_config._distribute_coordinator_mode:  # pylint: disable=protected-access
 217     raise ValueError(
 218         'Distribute coordinator mode is not specified in `RunConfig`.')
 219 
 220   def _worker_fn(strategy):
 221     """Function for worker task."""
 222     local_estimator = copy.deepcopy(estimator)
 223     # pylint: disable=protected-access
 224     local_estimator._config._train_distribute = strategy
 225     context = dc_context.get_current_worker_context()
 226     _init_run_config_from_worker_context(local_estimator._config, context)
 227     logging.info('Updated config: %s', str(vars(local_estimator._config)))
 228     local_estimator._train_distribution = strategy
 229     # pylint: enable=protected-access
 230 
 231     # In the standalone client, we don't need to run hooks on all threads
 232     # because logging hooks on all threads may be too much on the screen; also
 233     # tensor passed to one hook can only be fetched with the graph where the
 234     # tensor is defined. Other hooks such as checkpointing hooks will added by
 235     # MonitoredTrainingSession.
 236     # TODO(yuefengz): Is there a hook that does need to run on all threads in
 237     # standalone client mode?
 238     if (run_config._distribute_coordinator_mode ==  # pylint: disable=protected-access
 239         dc.CoordinatorMode.INDEPENDENT_WORKER or context.is_chief):
 240       hooks = list(train_spec.hooks)
 241     else:
 242       hooks = []
 243     local_estimator.train(
 244         input_fn=train_spec.input_fn,
 245         max_steps=train_spec.max_steps,
 246         hooks=hooks)
 247 
 248   def _eval_fn(strategy):
 249     """Function for evaluator task."""
 250     local_estimator = copy.deepcopy(estimator)
 251     # pylint: disable=protected-access
 252     local_estimator._config._eval_distribute = strategy
 253     _init_run_config_from_worker_context(
 254         local_estimator._config, dc_context.get_current_worker_context())
 255     logging.info('Updated config: %s', str(vars(local_estimator._config)))
 256     local_estimator._eval_distribution = strategy
 257 
 258     executor = executor_cls(local_estimator, train_spec, eval_spec)
 259     executor._start_continuous_evaluation()
 260     # pylint: enable=protected-access
 261 
 262   # pylint: disable=protected-access
 263   if (run_config._distribute_coordinator_mode ==
 264       dc.CoordinatorMode.STANDALONE_CLIENT):
 265     cluster_spec = run_config.cluster_spec
 266     assert cluster_spec
 267   else:
 268     # The cluster_spec comes from TF_CONFIG environment variable if it is
 269     # INDEPENDENT_WORKER mode.
 270     cluster_spec = None
 271 
 272   dc.run_distribute_coordinator(
 273       _worker_fn,
 274       run_config.train_distribute,
 275       _eval_fn,
 276       run_config.eval_distribute,
 277       mode=run_config._distribute_coordinator_mode,
 278       cluster_spec=cluster_spec,
 279       session_config=run_config.session_config)
