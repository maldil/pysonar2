  33 class _PerDeviceGenerator(dataset_ops.Dataset):
  34   """A `dummy` generator dataset."""
  35 
  36   def __init__(self, shard_num, multi_device_iterator_resource, incarnation_id,
  37                source_device, target_device, output_shapes, output_types,
  38                output_classes):
  39     self._target_device = target_device
  40     self._output_types = output_types
  41     self._output_shapes = output_shapes
  42     self._output_classes = output_classes
  43     self._flat_output_shapes = nest.flatten(
  44         sparse.as_dense_shapes(self._output_shapes, self._output_classes))
  45     self._flat_output_types = nest.flatten(
  46         sparse.as_dense_types(self._output_types, self._output_classes))
  47 
  48     multi_device_iterator_string_handle = (
  49         gen_dataset_ops.multi_device_iterator_to_string_handle(
  50             multi_device_iterator_resource))
  51 
  52     @function.Defun()
  53     def _init_func():
  54       return multi_device_iterator_string_handle
  55 
  56     @function.Defun()
  57     def _remote_init_func():
  58       return functional_ops.remote_call(
  59           target=source_device,
  60           args=_init_func.captured_inputs,
  61           Tout=[dtypes.string],
  62           f=_init_func)
  63 
  64     self._init_func = _remote_init_func
  65     self._init_captured_args = _remote_init_func.captured_inputs
  66 
  67     @function.Defun(dtypes.string)
  68     def _next_func(string_handle):
  69       multi_device_iterator = (
  70           gen_dataset_ops.multi_device_iterator_from_string_handle(
  71               string_handle=string_handle,
  72               output_types=self._flat_output_types,
  73               output_shapes=self._flat_output_shapes))
  74       return gen_dataset_ops.multi_device_iterator_get_next_from_shard(
  75           multi_device_iterator=multi_device_iterator,
  76           shard_num=shard_num,
  77           incarnation_id=incarnation_id,
  78           output_types=self._flat_output_types,
  79           output_shapes=self._flat_output_shapes)
  80 
  81     @function.Defun(dtypes.string)
  82     def _remote_next_func(string_handle):
  83       return functional_ops.remote_call(
  84           target=source_device,
  85           args=[string_handle] + _next_func.captured_inputs,
  86           Tout=self._flat_output_types,
  87           f=_next_func)
  88 
  89     self._next_func = _remote_next_func
  90     self._next_captured_args = _remote_next_func.captured_inputs
  91 
  92     @function.Defun(dtypes.string)
  93     def _finalize_func(unused_string_handle):
  94       return array_ops.constant(0, dtypes.int64)
  95 
  96     @function.Defun(dtypes.string)
  97     def _remote_finalize_func(string_handle):
  98       return functional_ops.remote_call(
  99           target=source_device,
 100           args=[string_handle] + _finalize_func.captured_inputs,
 101           Tout=[dtypes.int64],
 102           f=_finalize_func)
 103 
 104     self._finalize_func = _remote_finalize_func
 105     self._finalize_captured_args = _remote_finalize_func.captured_inputs
 106 
 107   def _as_variant_tensor(self):
 108     with ops.device(self._target_device):
 109       return gen_dataset_ops.generator_dataset(
 110           self._init_captured_args,
 111           self._next_captured_args,
 112           self._finalize_captured_args,
 113           init_func=self._init_func,
 114           next_func=self._next_func,
 115           finalize_func=self._finalize_func,
 116           output_types=self._flat_output_types,
 117           output_shapes=self._flat_output_shapes)
 118 
 119   def _inputs(self):
 120     # TODO(b/116506223): Determine which datasets should be used as inputs here.
 121     return []
 122 
 123   @property
 124   def output_types(self):
 125     return self._output_types
 126 
 127   @property
 128   def output_shapes(self):
 129     return self._output_shapes
 130 
 131   @property
 132   def output_classes(self):
 133     return self._output_classes

 136 class MultiDeviceIterator(object):
 137   """An iterator over multiple devices.
 138 
 139   @compatibility(eager)
 140   MultiDeviceIterator isn't currently supported in Eager mode but support is
 141   coming soon.
 142   @end_compatibility
 143   """
 144 
 145   def __init__(self,
 146                dataset,
 147                devices,
 148                max_buffer_size=1,
 149                prefetch_buffer_size=1,
 150                source_device="/cpu:0"):
 151     """Constructs a MultiDeviceIterator.
 152 
 153     Args:
 154       dataset: The input dataset to be iterated over.
 155       devices: The list of devices to fetch data to.
 156       max_buffer_size: Maximum size of the host side per device buffer to keep.
 157       prefetch_buffer_size: if > 1, then we setup a buffer on each device
 158         to prefetch into.
 159       source_device: The host device to place the `dataset` on.
 160 
 161     Raises:
 162       RuntimeError: If run in Eager mode.
 163     """
 164     if context.executing_eagerly():
 165       # TODO(rohanj): Fix this. Tracking bug: b/116467184
 166       raise RuntimeError("MultiDeviceIterator is not currently supported in "
 167                          "Eager mode.")
 168     self._dataset = dataset
 169     self._devices = devices
 170     self._source_device = source_device
 171     self._source_device_tensor = ops.convert_to_tensor(source_device)
 172 
 173     self._flat_output_shapes = nest.flatten(
 174         sparse.as_dense_shapes(self._dataset.output_shapes,
 175                                self._dataset.output_classes))
 176     self._flat_output_types = nest.flatten(
 177         sparse.as_dense_types(self._dataset.output_types,
 178                               self._dataset.output_classes))
 179 
 180     # Create the MultiDeviceIterator.
 181     with ops.device(self._source_device):
 182       self._multi_device_iterator_resource = (
 183           gen_dataset_ops.multi_device_iterator(
 184               devices=self._devices,
 185               shared_name="",
 186               container="",
 187               output_types=self._flat_output_types,
 188               output_shapes=self._flat_output_shapes))
 189 
 190       # The incarnation ID is used to ensure consistency between the per-device
 191       # iterators and the multi-device iterator.
 192       self._incarnation_id = gen_dataset_ops.multi_device_iterator_init(
 193           self._dataset._as_variant_tensor(),  # pylint: disable=protected-access
 194           self._multi_device_iterator_resource,
 195           max_buffer_size=max_buffer_size)
 196 
 197     # TODO(rohanj): Explore the possibility of the MultiDeviceIterator to
 198     # initialize the device side of the pipeline. This would allow the
 199     # MultiDeviceIterator to choose, for example, to move some transformations
 200     # into the device side from its input. It might be useful in rewriting.
 201     # Create the per device iterators.
 202     self._device_iterators = []
 203     i = 0
 204     for device in self._devices:
 205       ds = _PerDeviceGenerator(
 206           i, self._multi_device_iterator_resource, self._incarnation_id,
 207           self._source_device_tensor, device, self._dataset.output_shapes,
 208           self._dataset.output_types, self._dataset.output_classes)
 209       if prefetch_buffer_size > 0:
 210         ds = ds.prefetch(prefetch_buffer_size)
 211       with ops.device(device):
 212         self._device_iterators.append(ds.make_initializable_iterator())
 213       i += 1
 214 
 215     device_iterator_initializers = [
 216         iterator.initializer for iterator in self._device_iterators
 217     ]
 218     self._initializer = control_flow_ops.group(*device_iterator_initializers)
 219 
 220   def get_next(self):
 221     result = []
 222     i = 0
 223     for device in self._devices:
 224       with ops.device(device):
 225         result.append(self._device_iterators[i].get_next())
 226       i += 1
 227     return result
 228 
 229   @property
 230   def initializer(self):
 231     return self._initializer
