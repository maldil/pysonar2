  40 # NOTE(mrry): It is legitimate to call `Iterator.get_next()` multiple
  41 # times, e.g. when you are distributing different elements to multiple
  42 # devices in a single step. However, a common pitfall arises when
  43 # users call `Iterator.get_next()` in each iteration of their training
  44 # loop. `Iterator.get_next()` adds ops to the graph, and executing
  45 # each op allocates resources (including threads); as a consequence,
  46 # invoking it in every iteration of a training loop causes slowdown
  47 # and eventual resource exhaustion. To guard against this outcome, we
  48 # log a warning when the number of uses crosses a threshold of suspicion.
  49 GET_NEXT_CALL_WARNING_THRESHOLD = 32
  50 
  51 GET_NEXT_CALL_WARNING_MESSAGE = (
  52     "An unusually high number of `Iterator.get_next()` calls was detected. "
  53     "This often indicates that `Iterator.get_next()` is being called inside "
  54     "a training loop, which will cause gradual slowdown and eventual resource "
  55     "exhaustion. If this is the case, restructure your code to call "
  56     "`next_element = iterator.get_next()` once outside the loop, and use "
  57     "`next_element` as the input to some computation that is invoked inside "
  58     "the loop.")

  75   def __init__(self, iterator_resource, initializer, output_types,
  76                output_shapes, output_classes):
  77     """Creates a new iterator from the given iterator resource.
  78 
  79     Note: Most users will not call this initializer directly, and will
  80     instead use `Dataset.make_initializable_iterator()` or
  81     `Dataset.make_one_shot_iterator()`.
  82 
  83     Args:
  84       iterator_resource: A `tf.resource` scalar `tf.Tensor` representing the
  85         iterator.
  86       initializer: A `tf.Operation` that should be run to initialize this
  87         iterator.
  88       output_types: A nested structure of `tf.DType` objects corresponding to
  89         each component of an element of this iterator.
  90       output_shapes: A nested structure of `tf.TensorShape` objects
  91         corresponding to each component of an element of this iterator.
  92       output_classes: A nested structure of Python `type` objects corresponding
  93         to each component of an element of this iterator.
  94     """
  95     self._iterator_resource = iterator_resource
  96     self._initializer = initializer
  97     self._output_classes = output_classes
  98     self._output_types = output_types
  99     self._output_shapes = output_shapes
 100     self._string_handle = gen_dataset_ops.iterator_to_string_handle(
 101         self._iterator_resource)
 102     self._get_next_call_count = 0
 103     ops.add_to_collection(GLOBAL_ITERATORS, self._iterator_resource)

 105   @staticmethod
 106   def from_structure(output_types,
 107                      output_shapes=None,
 108                      shared_name=None,
 109                      output_classes=None):
 110     """Creates a new, uninitialized `Iterator` with the given structure.
 111 
 112     This iterator-constructing method can be used to create an iterator that
 113     is reusable with many different datasets.
 114 
 115     The returned iterator is not bound to a particular dataset, and it has
 116     no `initializer`. To initialize the iterator, run the operation returned by
 117     `Iterator.make_initializer(dataset)`.
 118 
 119     The following is an example
 120 
 121     ```python
 122     iterator = Iterator.from_structure(tf.int64, tf.TensorShape([]))
 123 
 124     dataset_range = Dataset.range(10)
 125     range_initializer = iterator.make_initializer(dataset_range)
 126 
 127     dataset_evens = dataset_range.filter(lambda x: x % 2 == 0)
 128     evens_initializer = iterator.make_initializer(dataset_evens)
 129 
 130     # Define a model based on the iterator; in this example, the model_fn
 131     # is expected to take scalar tf.int64 Tensors as input (see
 132     # the definition of 'iterator' above).
 133     prediction, loss = model_fn(iterator.get_next())
 134 
 135     # Train for `num_epochs`, where for each epoch, we first iterate over
 136     # dataset_range, and then iterate over dataset_evens.
 137     for _ in range(num_epochs):
 138       # Initialize the iterator to `dataset_range`
 139       sess.run(range_initializer)
 140       while True:
 141         try:
 142           pred, loss_val = sess.run([prediction, loss])
 143         except tf.errors.OutOfRangeError:
 144           break
 145 
 146       # Initialize the iterator to `dataset_evens`
 147       sess.run(evens_initializer)
 148       while True:
 149         try:
 150           pred, loss_val = sess.run([prediction, loss])
 151         except tf.errors.OutOfRangeError:
 152           break
 153     ```
 154 
 155     Args:
 156       output_types: A nested structure of `tf.DType` objects corresponding to
 157         each component of an element of this dataset.
 158       output_shapes: (Optional.) A nested structure of `tf.TensorShape` objects
 159         corresponding to each component of an element of this dataset. If
 160         omitted, each component will have an unconstrainted shape.
 161       shared_name: (Optional.) If non-empty, this iterator will be shared under
 162         the given name across multiple sessions that share the same devices
 163         (e.g. when using a remote server).
 164       output_classes: (Optional.) A nested structure of Python `type` objects
 165         corresponding to each component of an element of this iterator. If
 166         omitted, each component is assumed to be of type `tf.Tensor`.
 167 
 168     Returns:
 169       An `Iterator`.
 170 
 171     Raises:
 172       TypeError: If the structures of `output_shapes` and `output_types` are
 173         not the same.
 174     """
 175     output_types = nest.map_structure(dtypes.as_dtype, output_types)
 176     if output_shapes is None:
 177       output_shapes = nest.map_structure(
 178           lambda _: tensor_shape.TensorShape(None), output_types)
 179     else:
 180       output_shapes = nest.map_structure_up_to(
 181           output_types, tensor_shape.as_shape, output_shapes)
 182     if output_classes is None:
 183       output_classes = nest.map_structure(lambda _: ops.Tensor, output_types)
 184     nest.assert_same_structure(output_types, output_shapes)
 185     if shared_name is None:
 186       shared_name = ""
 187     if compat.forward_compatible(2018, 8, 3):
 188       if _device_stack_is_empty():
 189         with ops.device("/cpu:0"):
 190           iterator_resource = gen_dataset_ops.iterator_v2(
 191               container="",
 192               shared_name=shared_name,
 193               output_types=nest.flatten(
 194                   sparse.as_dense_types(output_types, output_classes)),
 195               output_shapes=nest.flatten(
 196                   sparse.as_dense_shapes(output_shapes, output_classes)))
 197       else:
 198         iterator_resource = gen_dataset_ops.iterator_v2(
 199             container="",
 200             shared_name=shared_name,
 201             output_types=nest.flatten(
 202                 sparse.as_dense_types(output_types, output_classes)),
 203             output_shapes=nest.flatten(
 204                 sparse.as_dense_shapes(output_shapes, output_classes)))
 205     else:
 206       iterator_resource = gen_dataset_ops.iterator(
 207           container="",
 208           shared_name=shared_name,
 209           output_types=nest.flatten(
 210               sparse.as_dense_types(output_types, output_classes)),
 211           output_shapes=nest.flatten(
 212               sparse.as_dense_shapes(output_shapes, output_classes)))
 213     return Iterator(iterator_resource, None, output_types, output_shapes,
 214                     output_classes)

 216   @staticmethod
 217   def from_string_handle(string_handle,
 218                          output_types,
 219                          output_shapes=None,
 220                          output_classes=None):
 221     """Creates a new, uninitialized `Iterator` based on the given handle.
 222 
 223     This method allows you to define a "feedable" iterator where you can choose
 224     between concrete iterators by feeding a value in a `tf.Session.run` call.
 225     In that case, `string_handle` would be a `tf.placeholder`, and you would
 226     feed it with the value of `tf.data.Iterator.string_handle` in each step.
 227 
 228     For example, if you had two iterators that marked the current position in
 229     a training dataset and a test dataset, you could choose which to use in
 230     each step as follows:
 231 
 232     ```python
 233     train_iterator = tf.data.Dataset(...).make_one_shot_iterator()
 234     train_iterator_handle = sess.run(train_iterator.string_handle())
 235 
 236     test_iterator = tf.data.Dataset(...).make_one_shot_iterator()
 237     test_iterator_handle = sess.run(test_iterator.string_handle())
 238 
 239     handle = tf.placeholder(tf.string, shape=[])
 240     iterator = tf.data.Iterator.from_string_handle(
 241         handle, train_iterator.output_types)
 242 
 243     next_element = iterator.get_next()
 244     loss = f(next_element)
 245 
 246     train_loss = sess.run(loss, feed_dict={handle: train_iterator_handle})
 247     test_loss = sess.run(loss, feed_dict={handle: test_iterator_handle})
 248     ```
 249 
 250     Args:
 251       string_handle: A scalar `tf.Tensor` of type `tf.string` that evaluates
 252         to a handle produced by the `Iterator.string_handle()` method.
 253       output_types: A nested structure of `tf.DType` objects corresponding to
 254         each component of an element of this dataset.
 255       output_shapes: (Optional.) A nested structure of `tf.TensorShape` objects
 256         corresponding to each component of an element of this dataset. If
 257         omitted, each component will have an unconstrainted shape.
 258       output_classes: (Optional.) A nested structure of Python `type` objects
 259         corresponding to each component of an element of this iterator. If
 260         omitted, each component is assumed to be of type `tf.Tensor`.
 261 
 262     Returns:
 263       An `Iterator`.
 264     """
 265     output_types = nest.map_structure(dtypes.as_dtype, output_types)
 266     if output_shapes is None:
 267       output_shapes = nest.map_structure(
 268           lambda _: tensor_shape.TensorShape(None), output_types)
 269     else:
 270       output_shapes = nest.map_structure_up_to(
 271           output_types, tensor_shape.as_shape, output_shapes)
 272     if output_classes is None:
 273       output_classes = nest.map_structure(lambda _: ops.Tensor, output_types)
 274     nest.assert_same_structure(output_types, output_shapes)
 275     string_handle = ops.convert_to_tensor(string_handle, dtype=dtypes.string)
 276     if compat.forward_compatible(2018, 8, 3):
 277       if _device_stack_is_empty():
 278         with ops.device("/cpu:0"):
 279           iterator_resource = gen_dataset_ops.iterator_from_string_handle_v2(
 280               string_handle,
 281               output_types=nest.flatten(
 282                   sparse.as_dense_types(output_types, output_classes)),
 283               output_shapes=nest.flatten(
 284                   sparse.as_dense_shapes(output_shapes, output_classes)))
 285       else:
 286         iterator_resource = gen_dataset_ops.iterator_from_string_handle_v2(
 287             string_handle,
 288             output_types=nest.flatten(
 289                 sparse.as_dense_types(output_types, output_classes)),
 290             output_shapes=nest.flatten(
 291                 sparse.as_dense_shapes(output_shapes, output_classes)))
 292     else:
 293       iterator_resource = gen_dataset_ops.iterator_from_string_handle(
 294           string_handle,
 295           output_types=nest.flatten(
 296               sparse.as_dense_types(output_types, output_classes)),
 297           output_shapes=nest.flatten(
 298               sparse.as_dense_shapes(output_shapes, output_classes)))
 299     return Iterator(iterator_resource, None, output_types, output_shapes,
 300                     output_classes)

 304     """A `tf.Operation` that should be run to initialize this iterator.
 305 
 306     Returns:
 307       A `tf.Operation` that should be run to initialize this iterator

 309     Raises:
 310       ValueError: If this iterator initializes itself automatically.

 312     if self._initializer is not None:
 313       return self._initializer

 315       # TODO(mrry): Consider whether one-shot iterators should have

 319   def make_initializer(self, dataset, name=None):
 320     """Returns a `tf.Operation` that initializes this iterator on `dataset`.
 321 
 322     Args:
 323       dataset: A `Dataset` with compatible structure to this iterator.
 324       name: (Optional.) A name for the created operation.
 325 
 326     Returns:
 327       A `tf.Operation` that can be run to initialize this iterator on the given
 328       `dataset`.
 329 
 330     Raises:
 331       TypeError: If `dataset` and this iterator do not have a compatible
 332         element structure.
 333     """
 334     with ops.name_scope(name, "make_initializer") as name:
 335       nest.assert_same_structure(self._output_types, dataset.output_types)
 336       nest.assert_same_structure(self._output_shapes, dataset.output_shapes)
 337       for iterator_class, dataset_class in zip(
 338           nest.flatten(self._output_classes),
 339           nest.flatten(dataset.output_classes)):
 340         if iterator_class is not dataset_class:
 341           raise TypeError(
 342               "Expected output classes %r but got dataset with output class %r."
 343               % (self._output_classes, dataset.output_classes))
 344       for iterator_dtype, dataset_dtype in zip(
 345           nest.flatten(self._output_types), nest.flatten(dataset.output_types)):
 346         if iterator_dtype != dataset_dtype:
 347           raise TypeError(
 348               "Expected output types %r but got dataset with output types %r." %
 349               (self._output_types, dataset.output_types))
 350       for iterator_shape, dataset_shape in zip(
 351           nest.flatten(self._output_shapes), nest.flatten(
 352               dataset.output_shapes)):
 353         if not iterator_shape.is_compatible_with(dataset_shape):
 354           raise TypeError("Expected output shapes compatible with %r but got "
 355                           "dataset with output shapes %r." %
 356                           (self._output_shapes, dataset.output_shapes))
 357     with ops.colocate_with(self._iterator_resource):
 358       return gen_dataset_ops.make_iterator(
 359           dataset._as_variant_tensor(), self._iterator_resource, name=name)  # pylint: disable=protected-access

 361   def get_next(self, name=None):
 362     """Returns a nested structure of `tf.Tensor`s representing the next element.
 363 
 364     In graph mode, you should typically call this method *once* and use its
 365     result as the input to another computation. A typical loop will then call
 366     `tf.Session.run` on the result of that computation. The loop will terminate
 367     when the `Iterator.get_next()` operation raises
 368     `tf.errors.OutOfRangeError`. The following skeleton shows how to use
 369     this method when building a training loop:
 370 
 371     ```python
 372     dataset = ...  # A `tf.data.Dataset` object.
 373     iterator = dataset.make_initializable_iterator()
 374     next_element = iterator.get_next()
 375 
 376     # Build a TensorFlow graph that does something with each element.
 377     loss = model_function(next_element)
 378     optimizer = ...  # A `tf.train.Optimizer` object.
 379     train_op = optimizer.minimize(loss)
 380 
 381     with tf.Session() as sess:
 382       try:
 383         while True:
 384           sess.run(train_op)
 385       except tf.errors.OutOfRangeError:
 386         pass
 387     ```
 388 
 389     NOTE: It is legitimate to call `Iterator.get_next()` multiple times, e.g.
 390     when you are distributing different elements to multiple devices in a single
 391     step. However, a common pitfall arises when users call `Iterator.get_next()`
 392     in each iteration of their training loop. `Iterator.get_next()` adds ops to
 393     the graph, and executing each op allocates resources (including threads); as
 394     a consequence, invoking it in every iteration of a training loop causes
 395     slowdown and eventual resource exhaustion. To guard against this outcome, we
 396     log a warning when the number of uses crosses a fixed threshold of
 397     suspiciousness.
 398 
 399     Args:
 400       name: (Optional.) A name for the created operation.
 401 
 402     Returns:
 403       A nested structure of `tf.Tensor` objects.
 404     """
 405     self._get_next_call_count += 1
 406     if self._get_next_call_count > GET_NEXT_CALL_WARNING_THRESHOLD:
 407       warnings.warn(GET_NEXT_CALL_WARNING_MESSAGE)
 408 
 409     return sparse.deserialize_sparse_tensors(
 410         nest.pack_sequence_as(self._output_types,
 411                               gen_dataset_ops.iterator_get_next(
 412                                   self._iterator_resource,
 413                                   output_types=nest.flatten(
 414                                       sparse.as_dense_types(
 415                                           self._output_types,
 416                                           self._output_classes)),
 417                                   output_shapes=nest.flatten(
 418                                       sparse.as_dense_shapes(
 419                                           self._output_shapes,
 420                                           self._output_classes)),
 421                                   name=name)), self._output_types,
 422         self._output_shapes, self._output_classes)
 423

 491 class EagerIterator(checkpointable.CheckpointableBase):
 492   """An iterator producing tf.Tensor objects from a tf.data.Dataset."""
 493 
 494   def __init__(self, dataset):
 495     """Creates a new iterator over the given dataset.
 496 
 497     For example:
 498     ```python
 499     dataset = tf.data.Dataset.range(4)
 500     for x in Iterator(dataset):
 501       print(x)
 502     ```
 503 
 504     Tensors produced will be placed on the device on which this iterator object
 505     was created.
 506 
 507     Args:
 508       dataset: A `tf.data.Dataset` object.
 509 
 510     Raises:
 511       RuntimeError: When invoked without eager execution enabled.
 512     """
 513 
 514     if not context.executing_eagerly():
 515       raise RuntimeError(
 516           "{} objects can only be used when eager execution is enabled, use "
 517           "tf.data.Dataset.make_initializable_iterator or "
 518           "tf.data.Dataset.make_one_shot_iterator for graph construction".
 519           format(type(self)))
 520     self._device = context.context().device_name
 521     with ops.device("/cpu:0"):
 522       ds_variant = dataset._as_variant_tensor()  # pylint: disable=protected-access
 523       self._output_classes = dataset.output_classes
 524       self._output_types = dataset.output_types
 525       self._output_shapes = dataset.output_shapes
 526       self._flat_output_types = nest.flatten(
 527           sparse.as_dense_types(self._output_types, self._output_classes))
 528       self._flat_output_shapes = nest.flatten(
 529           sparse.as_dense_shapes(self._output_shapes, self._output_classes))
 530       with ops.colocate_with(ds_variant):
 531         self._resource = gen_dataset_ops.anonymous_iterator(
 532             output_types=self._flat_output_types,
 533             output_shapes=self._flat_output_shapes)
 534         gen_dataset_ops.make_iterator(ds_variant, self._resource)
 535         # Delete the resource when this object is deleted
 536         self._resource_deleter = resource_variable_ops.EagerResourceDeleter(
 537             handle=self._resource, handle_device=self._device)
 538 
 539   def __iter__(self):
 540     return self
 541 
 542   def __next__(self):  # For Python 3 compatibility
 543     return self.next()

 545   def _next_internal(self):
 546     """Returns a nested structure of `tf.Tensor`s containing the next element.
 547     """
 548     # This runs in sync mode as iterators use an error status to communicate
 549     # that there is no more data to iterate over.
 550     # TODO(b/77291417): Fix
 551     with context.execution_mode(context.SYNC):
 552       with ops.device(self._device):
 553         # TODO(ashankar): Consider removing this ops.device() contextmanager
 554         # and instead mimic ops placement in graphs: Operations on resource
 555         # handles execute on the same device as where the resource is placed.
 556         # NOTE(mrry): Here we use the "_sync" variant of `iterator_get_next`
 557         # because in eager mode this code will run synchronously on the calling
 558         # thread. Therefore we do not need to make a defensive context switch
 559         # to a background thread, and can achieve a small constant performance
 560         # boost by invoking the iterator synchronously.
 561         ret = gen_dataset_ops.iterator_get_next_sync(
 562             self._resource,
 563             output_types=self._flat_output_types,
 564             output_shapes=self._flat_output_shapes)
 565 
 566       return sparse.deserialize_sparse_tensors(
 567           nest.pack_sequence_as(self._output_types, ret), self._output_types,
 568           self._output_shapes, self._output_classes)
 569 
 570   def next(self):
 571     """Returns a nested structure of `tf.Tensor`s containing the next element.
 572     """
 573     try:
 574       return self._next_internal()
 575     except errors.OutOfRangeError:
 576       raise StopIteration
