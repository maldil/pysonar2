  29 class Optional(object):
  30   """Wraps a nested structure of tensors that may/may not be present at runtime.
  31 
  32   An `Optional` can represent the result of an operation that may fail as a
  33   value, rather than raising an exception and halting execution. For example,
  34   `tf.data.experimental.get_next_as_optional` returns an `Optional` that either
  35   contains the next value from a `tf.data.Iterator` if one exists, or a "none"
  36   value that indicates the end of the sequence has been reached.
  37   """
  38 
  39   @abc.abstractmethod
  40   def has_value(self, name=None):
  41     """Returns a tensor that evaluates to `True` if this optional has a value.
  42 
  43     Args:
  44       name: (Optional.) A name for the created operation.
  45 
  46     Returns:
  47       A scalar `tf.Tensor` of type `tf.bool`.
  48     """
  49     raise NotImplementedError("Optional.has_value()")
  50 
  51   @abc.abstractmethod
  52   def get_value(self, name=None):
  53     """Returns a nested structure of values wrapped by this optional.
  54 
  55     If this optional does not have a value (i.e. `self.has_value()` evaluates
  56     to `False`), this operation will raise `tf.errors.InvalidArgumentError`
  57     at runtime.
  58 
  59     Args:
  60       name: (Optional.) A name for the created operation.
  61 
  62     Returns:
  63       A nested structure of `tf.Tensor` and/or `tf.SparseTensor` objects.
  64     """
  65     raise NotImplementedError("Optional.get_value()")
  66 
  67   @abc.abstractproperty
  68   def value_structure(self):
  69     """The structure of the components of this optional.
  70 
  71     Returns:
  72       A `Structure` object representing the structure of the components of this
  73         optional.
  74     """
  75     raise NotImplementedError("Optional.value_structure")
  76 
  77   @staticmethod
  78   def from_value(value):
  79     """Returns an `Optional` that wraps the given value.
  80 
  81     Args:
  82       value: A nested structure of `tf.Tensor` and/or `tf.SparseTensor` objects.
  83 
  84     Returns:
  85       An `Optional` that wraps `value`.
  86     """
  87     with ops.name_scope("optional") as scope:
  88       with ops.name_scope("value"):
  89         value_structure = structure.Structure.from_value(value)
  90         encoded_value = value_structure._to_tensor_list(value)  # pylint: disable=protected-access
  91 
  92     return _OptionalImpl(
  93         gen_dataset_ops.optional_from_value(encoded_value, name=scope),
  94         value_structure)
  95 
  96   @staticmethod
  97   def none_from_structure(value_structure):
  98     """Returns an `Optional` that has no value.
  99 
 100     NOTE: This method takes an argument that defines the structure of the value
 101     that would be contained in the returned `Optional` if it had a value.
 102 
 103     Args:
 104       value_structure: A `Structure` object representing the structure of the
 105         components of this optional.
 106 
 107     Returns:
 108       An `Optional` that has no value.
 109     """
 110     return _OptionalImpl(gen_dataset_ops.optional_none(), value_structure)

 113 class _OptionalImpl(Optional):
 114   """Concrete implementation of `tf.data.experimental.Optional`.
 115 
 116   NOTE(mrry): This implementation is kept private, to avoid defining
 117   `Optional.__init__()` in the public API.
 118   """
 119 
 120   def __init__(self, variant_tensor, value_structure):
 121     self._variant_tensor = variant_tensor
 122     self._value_structure = value_structure
 123 
 124   def has_value(self, name=None):
 125     return gen_dataset_ops.optional_has_value(self._variant_tensor, name=name)
 126 
 127   def get_value(self, name=None):
 128     # TODO(b/110122868): Consolidate the restructuring logic with similar logic
 129     # in `Iterator.get_next()` and `StructuredFunctionWrapper`.
 130     with ops.name_scope(name, "OptionalGetValue",
 131                         [self._variant_tensor]) as scope:
 132       # pylint: disable=protected-access
 133       return self._value_structure._from_tensor_list(
 134           gen_dataset_ops.optional_get_value(
 135               self._variant_tensor,
 136               name=scope,
 137               output_types=self._value_structure._flat_types,
 138               output_shapes=self._value_structure._flat_shapes))
 139 
 140   @property
 141   def value_structure(self):
 142     return self._value_structure
 143

 145 class OptionalStructure(structure.Structure):
 146   """Represents an optional potentially containing a structured value."""
 147 
 148   def __init__(self, value_structure):
 149     self._value_structure = value_structure
 150 
 151   @property
 152   def _flat_shapes(self):
 153     return [tensor_shape.scalar()]
 154 
 155   @property
 156   def _flat_types(self):
 157     return [dtypes.variant]
 158 
 159   def is_compatible_with(self, other):
 160     # pylint: disable=protected-access
 161     return (isinstance(other, OptionalStructure) and
 162             self._value_structure.is_compatible_with(other._value_structure))
 163 
 164   def _to_tensor_list(self, value):
 165     return [value._variant_tensor]  # pylint: disable=protected-access
 166 
 167   def _from_tensor_list(self, flat_value):
 168     if (len(flat_value) != 1 or flat_value[0].dtype != dtypes.variant or
 169         not flat_value[0].shape.is_compatible_with(tensor_shape.scalar())):
 170       raise ValueError(
 171           "OptionalStructure corresponds to a single tf.variant scalar.")
 172     # pylint: disable=protected-access
 173     return _OptionalImpl(flat_value[0], self._value_structure)
 174 
 175   @staticmethod
 176   def from_value(value):
 177     return OptionalStructure(value.value_structure)
