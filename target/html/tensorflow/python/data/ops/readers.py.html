  39     """Creates a `TextLineDataset`.
  40 
  41     Args:
  42       filenames: A `tf.string` tensor containing one or more filenames.
  43       compression_type: (Optional.) A `tf.string` scalar evaluating to one of
  44         `""` (no compression), `"ZLIB"`, or `"GZIP"`.
  45       buffer_size: (Optional.) A `tf.int64` scalar denoting the number of bytes
  46         to buffer. A value of 0 results in the default buffering values chosen
  47         based on the compression type.

  50     self._filenames = ops.convert_to_tensor(
  51         filenames, dtype=dtypes.string, name="filenames")

  53         "compression_type",
  54         compression_type,
  55         argument_default="",
  56         argument_dtype=dtypes.string)
  57     self._buffer_size = convert.optional_param_to_tensor(
  58         "buffer_size", buffer_size, _DEFAULT_READER_BUFFER_SIZE_BYTES)
  59 
  60   def _as_variant_tensor(self):
  61     return gen_dataset_ops.text_line_dataset(
  62         self._filenames, self._compression_type, self._buffer_size)

 127 class ParallelInterleaveDataset(dataset_ops.InterleaveDataset):
 128   """A `Dataset` that maps a function over its input and flattens the result."""
 129 
 130   def __init__(self, input_dataset, map_func, cycle_length, block_length,
 131                sloppy, buffer_output_elements, prefetch_input_elements):
 132     """See `tf.data.experimental.parallel_interleave()` for details."""
 133     super(ParallelInterleaveDataset, self).__init__(input_dataset, map_func,
 134                                                     cycle_length, block_length)
 135     self._sloppy = ops.convert_to_tensor(
 136         sloppy, dtype=dtypes.bool, name="sloppy")
 137     self._buffer_output_elements = convert.optional_param_to_tensor(
 138         "buffer_output_elements",
 139         buffer_output_elements,
 140         argument_default=2 * block_length)
 141     self._prefetch_input_elements = convert.optional_param_to_tensor(
 142         "prefetch_input_elements",
 143         prefetch_input_elements,
 144         argument_default=2 * cycle_length)
 145 
 146   def _as_variant_tensor(self):
 147     # pylint: disable=protected-access
 148     return gen_dataset_ops.parallel_interleave_dataset(
 149         self._input_dataset._as_variant_tensor(),
 150         self._map_func.captured_inputs,
 151         self._cycle_length,
 152         self._block_length,
 153         self._sloppy,
 154         self._buffer_output_elements,
 155         self._prefetch_input_elements,
 156         f=self._map_func,
 157         **dataset_ops.flat_structure(self))
 158     # pylint: enable=protected-access
 159 
 160   def _transformation_name(self):
 161     return "tf.data.experimental.parallel_interleave()"

 164 @tf_export("data.TFRecordDataset")
 165 class TFRecordDataset(dataset_ops.Dataset):
 166   """A `Dataset` comprising records from one or more TFRecord files."""
 167 
 168   def __init__(self, filenames, compression_type=None, buffer_size=None,
 169                num_parallel_reads=None):
 170     """Creates a `TFRecordDataset` to read for one or more TFRecord files.
 171 
 172     NOTE: The `num_parallel_reads` argument can be used to improve performance
 173     when reading from a remote filesystem.
 174 
 175     Args:
 176       filenames: A `tf.string` tensor or `tf.data.Dataset` containing one or
 177         more filenames.
 178       compression_type: (Optional.) A `tf.string` scalar evaluating to one of
 179         `""` (no compression), `"ZLIB"`, or `"GZIP"`.
 180       buffer_size: (Optional.) A `tf.int64` scalar representing the number of
 181         bytes in the read buffer. 0 means no buffering.
 182       num_parallel_reads: (Optional.) A `tf.int64` scalar representing the
 183         number of files to read in parallel. Defaults to reading files
 184         sequentially.
 185 
 186     Raises:
 187       TypeError: If any argument does not have the expected type.
 188       ValueError: If any argument does not have the expected shape.
 189     """
 190     super(TFRecordDataset, self).__init__()
 191     if isinstance(filenames, dataset_ops.Dataset):
 192       if filenames.output_types != dtypes.string:
 193         raise TypeError(
 194             "`filenames` must be a `tf.data.Dataset` of `tf.string` elements.")
 195       if not filenames.output_shapes.is_compatible_with(tensor_shape.scalar()):
 196         raise ValueError(
 197             "`filenames` must be a `tf.data.Dataset` of scalar `tf.string` "
 198             "elements.")
 199     else:
 200       filenames = ops.convert_to_tensor(filenames, dtype=dtypes.string)
 201       filenames = array_ops.reshape(filenames, [-1], name="flat_filenames")
 202       filenames = dataset_ops.Dataset.from_tensor_slices(filenames)
 203 
 204     self._filenames = filenames
 205     self._compression_type = compression_type
 206     self._buffer_size = buffer_size
 207     self._num_parallel_reads = num_parallel_reads
 208 
 209     def read_one_file(filename):
 210       return _TFRecordDataset(filename, compression_type, buffer_size)
 211 
 212     if num_parallel_reads is None:
 213       self._impl = filenames.flat_map(read_one_file)
 214     else:
 215       self._impl = ParallelInterleaveDataset(
 216           filenames, read_one_file, cycle_length=num_parallel_reads,
 217           block_length=1, sloppy=False, buffer_output_elements=None,
 218           prefetch_input_elements=None)
 219 
 220   def _clone(self,
 221              filenames=None,
 222              compression_type=None,
 223              buffer_size=None,
 224              num_parallel_reads=None):
 225     return TFRecordDataset(filenames or self._filenames,
 226                            compression_type or self._compression_type,
 227                            buffer_size or self._buffer_size,
 228                            num_parallel_reads or self._num_parallel_reads)
 229 
 230   def _as_variant_tensor(self):
 231     return self._impl._as_variant_tensor()  # pylint: disable=protected-access
 232 
 233   def _inputs(self):
 234     return self._impl._inputs()  # pylint: disable=protected-access
 235 
 236   @property
 237   def output_classes(self):
 238     return self._impl.output_classes
 239 
 240   @property
 241   def output_shapes(self):
 242     return self._impl.output_shapes
 243 
 244   @property
 245   def output_types(self):
 246     return self._impl.output_types

 249 @tf_export("data.FixedLengthRecordDataset")
 250 class FixedLengthRecordDataset(dataset_ops.Dataset):
 251   """A `Dataset` of fixed-length records from one or more binary files."""
 252 
 253   def __init__(self,
 254                filenames,
 255                record_bytes,
 256                header_bytes=None,
 257                footer_bytes=None,
 258                buffer_size=None):
 259     """Creates a `FixedLengthRecordDataset`.
 260 
 261     Args:
 262       filenames: A `tf.string` tensor containing one or more filenames.
 263       record_bytes: A `tf.int64` scalar representing the number of bytes in
 264         each record.
 265       header_bytes: (Optional.) A `tf.int64` scalar representing the number of
 266         bytes to skip at the start of a file.
 267       footer_bytes: (Optional.) A `tf.int64` scalar representing the number of
 268         bytes to ignore at the end of a file.
 269       buffer_size: (Optional.) A `tf.int64` scalar representing the number of
 270         bytes to buffer when reading.
 271     """
 272     super(FixedLengthRecordDataset, self).__init__()
 273     self._filenames = ops.convert_to_tensor(
 274         filenames, dtype=dtypes.string, name="filenames")
 275     self._record_bytes = ops.convert_to_tensor(
 276         record_bytes, dtype=dtypes.int64, name="record_bytes")
 277 
 278     self._header_bytes = convert.optional_param_to_tensor(
 279         "header_bytes", header_bytes)
 280     self._footer_bytes = convert.optional_param_to_tensor(
 281         "footer_bytes", footer_bytes)
 282     self._buffer_size = convert.optional_param_to_tensor(
 283         "buffer_size", buffer_size, _DEFAULT_READER_BUFFER_SIZE_BYTES)
 284 
 285   def _as_variant_tensor(self):
 286     return gen_dataset_ops.fixed_length_record_dataset(
 287         self._filenames, self._header_bytes, self._record_bytes,
 288         self._footer_bytes, self._buffer_size)
 289 
 290   def _inputs(self):
 291     return []
 292 
 293   @property
 294   def output_classes(self):
 295     return ops.Tensor
 296 
 297   @property
 298   def output_shapes(self):
 299     return tensor_shape.scalar()
 300 
 301   @property
 302   def output_types(self):
 303     return dtypes.string
