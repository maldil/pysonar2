  27 _uid_counter = 0
  28 _uid_lock = threading.Lock()
  29 
  30 
  31 def _generate_shared_name(prefix):
  32   with _uid_lock:
  33     global _uid_counter
  34     uid = _uid_counter
  35     _uid_counter += 1
  36   return "{}{}".format(prefix, uid)

  41 class PrivateThreadPool(object):
  42   """A stateful resource that represents a private thread pool."""
  43 
  44   def __init__(self, num_threads, display_name=None,
  45                max_intra_op_parallelism=1):
  46     """Creates a `PrivateThreadPool` with the given number of threads."""
  47     if context.executing_eagerly():
  48       shared_name = _generate_shared_name("privatethreadpool")
  49       self._resource = ged_ops.experimental_thread_pool_handle(
  50           num_threads=num_threads,
  51           max_intra_op_parallelism=max_intra_op_parallelism,
  52           display_name=display_name,
  53           shared_name=shared_name)
  54       self._resource_deleter = resource_variable_ops.EagerResourceDeleter(
  55           handle=self._resource, handle_device=context.context().device_name)
  56     else:
  57       self._resource = ged_ops.experimental_thread_pool_handle(
  58           num_threads=num_threads,
  59           max_intra_op_parallelism=max_intra_op_parallelism,
  60           display_name=display_name)

  63 class _ThreadPoolDataset(dataset_ops.UnaryDataset):
  64   """A `Dataset` that acts as an identity, and sets a custom threadpool."""
  65 
  66   def __init__(self, input_dataset, thread_pool):
  67     super(_ThreadPoolDataset, self).__init__(input_dataset)
  68     self._input_dataset = input_dataset
  69     self._thread_pool = thread_pool
  70 
  71   def _as_variant_tensor(self):
  72     return ged_ops.experimental_thread_pool_dataset(
  73         self._input_dataset._as_variant_tensor(),  # pylint: disable=protected-access
  74         self._thread_pool._resource,  # pylint: disable=protected-access
  75         **dataset_ops.flat_structure(self))
  76 
  77   @property
  78   def output_shapes(self):
  79     return self._input_dataset.output_shapes
  80 
  81   @property
  82   def output_types(self):
  83     return self._input_dataset.output_types
  84 
  85   @property
  86   def output_classes(self):
  87     return self._input_dataset.output_classes

  90 # TODO(b/73383364): Properly export in the `tf.data.experimental` API when
  91 # stable or make private / remove.
  92 def override_threadpool(dataset, thread_pool):
  93   """Returns a new dataset that uses the given thread pool for its operations.
  94 
  95   Args:
  96     dataset: A `tf.data.Dataset` object.
  97     thread_pool: A `PrivateThreadPool` object.
  98 
  99   Returns:
 100     A dataset containing the same values as `dataset`, but which uses
 101     `thread_pool` to compute any of its parallel operations (such as
 102     `tf.data.Dataset.map`).
 103   """
 104   return _ThreadPoolDataset(dataset, thread_pool)
