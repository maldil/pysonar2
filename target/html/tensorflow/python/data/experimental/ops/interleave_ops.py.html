  34 @tf_export("data.experimental.parallel_interleave")
  35 def parallel_interleave(map_func,
  36                         cycle_length,
  37                         block_length=1,
  38                         sloppy=False,
  39                         buffer_output_elements=None,
  40                         prefetch_input_elements=None):
  41   """A parallel version of the `Dataset.interleave()` transformation.
  42 
  43   `parallel_interleave()` maps `map_func` across its input to produce nested
  44   datasets, and outputs their elements interleaved. Unlike
  45   `tf.data.Dataset.interleave`, it gets elements from `cycle_length` nested
  46   datasets in parallel, which increases the throughput, especially in the
  47   presence of stragglers. Furthermore, the `sloppy` argument can be used to
  48   improve performance, by relaxing the requirement that the outputs are produced
  49   in a deterministic order, and allowing the implementation to skip over nested
  50   datasets whose elements are not readily available when requested.
  51 
  52   Example usage:
  53 
  54   ```python
  55   # Preprocess 4 files concurrently.
  56   filenames = tf.data.Dataset.list_files("/path/to/data/train*.tfrecords")
  57   dataset = filenames.apply(
  58       tf.data.experimental.parallel_interleave(
  59           lambda filename: tf.data.TFRecordDataset(filename),
  60           cycle_length=4))
  61   ```
  62 
  63   WARNING: If `sloppy` is `True`, the order of produced elements is not
  64   deterministic.
  65 
  66   Args:
  67     map_func: A function mapping a nested structure of tensors to a `Dataset`.
  68     cycle_length: The number of input `Dataset`s to interleave from in parallel.
  69     block_length: The number of consecutive elements to pull from an input
  70       `Dataset` before advancing to the next input `Dataset`.
  71     sloppy: If false, elements are produced in deterministic order. Otherwise,
  72       the implementation is allowed, for the sake of expediency, to produce
  73       elements in a non-deterministic order.
  74     buffer_output_elements: The number of elements each iterator being
  75       interleaved should buffer (similar to the `.prefetch()` transformation for
  76       each interleaved iterator).
  77     prefetch_input_elements: The number of input elements to transform to
  78       iterators before they are needed for interleaving.
  79 
  80   Returns:
  81     A `Dataset` transformation function, which can be passed to
  82     `tf.data.Dataset.apply`.
  83   """
  84   def _apply_fn(dataset):
  85     return readers.ParallelInterleaveDataset(
  86         dataset, map_func, cycle_length, block_length, sloppy,
  87         buffer_output_elements, prefetch_input_elements)
  88 
  89   return _apply_fn

  92 class _DirectedInterleaveDataset(dataset_ops.Dataset):
  93   """A substitute for `Dataset.interleave()` on a fixed list of datasets."""
  94 
  95   def __init__(self, selector_input, data_inputs):
  96     self._selector_input = selector_input
  97     self._data_inputs = list(data_inputs)
  98 
  99     for data_input in data_inputs[1:]:
 100       if (data_input.output_types != data_inputs[0].output_types or
 101           data_input.output_classes != data_inputs[0].output_classes):
 102         raise TypeError("All datasets must have the same type and class.")
 103 
 104   def _as_variant_tensor(self):
 105     # pylint: disable=protected-access
 106     return (
 107         gen_experimental_dataset_ops.experimental_directed_interleave_dataset(
 108             self._selector_input._as_variant_tensor(), [
 109                 data_input._as_variant_tensor()
 110                 for data_input in self._data_inputs
 111             ], **dataset_ops.flat_structure(self)))
 112     # pylint: enable=protected-access
 113 
 114   def _inputs(self):
 115     return [self._selector_input] + self._data_inputs
 116 
 117   @property
 118   def output_classes(self):
 119     return self._data_inputs[0].output_classes
 120 
 121   @property
 122   def output_shapes(self):
 123     ret = self._data_inputs[0].output_shapes
 124     for data_input in self._data_inputs[1:]:
 125       ret = nest.pack_sequence_as(ret, [
 126           ts1.most_specific_compatible_shape(ts2) for (ts1, ts2) in zip(
 127               nest.flatten(ret), nest.flatten(data_input.output_shapes))
 128       ])
 129     return ret
 130 
 131   @property
 132   def output_types(self):
 133     return self._data_inputs[0].output_types

 136 @tf_export("data.experimental.sample_from_datasets")
 137 def sample_from_datasets(datasets, weights=None, seed=None):
 138   """Samples elements at random from the datasets in `datasets`.
 139 
 140   Args:
 141     datasets: A list of `tf.data.Dataset` objects with compatible structure.
 142     weights: (Optional.) A list of `len(datasets)` floating-point values where
 143       `weights[i]` represents the probability with which an element should be
 144       sampled from `datasets[i]`, or a `tf.data.Dataset` object where each
 145       element is such a list. Defaults to a uniform distribution across
 146       `datasets`.
 147     seed: (Optional.) A `tf.int64` scalar `tf.Tensor`, representing the
 148       random seed that will be used to create the distribution. See
 149       `tf.set_random_seed` for behavior.
 150 
 151   Returns:
 152     A dataset that interleaves elements from `datasets` at random, according to
 153     `weights` if provided, otherwise with uniform probability.
 154 
 155   Raises:
 156     TypeError: If the `datasets` or `weights` arguments have the wrong type.
 157     ValueError: If the `weights` argument is specified and does not match the
 158       length of the `datasets` element.
 159   """
 160   num_datasets = len(datasets)
 161   if not isinstance(weights, dataset_ops.Dataset):
 162     if weights is None:
 163       # Select inputs with uniform probability.
 164       logits = [[1.0] * num_datasets]
 165 
 166     else:
 167       # Use the given `weights` as the probability of choosing the respective
 168       # input.
 169       weights = ops.convert_to_tensor(weights, name="weights")
 170       if weights.dtype not in (dtypes.float32, dtypes.float64):
 171         raise TypeError("`weights` must be convertible to a tensor of "
 172                         "`tf.float32` or `tf.float64` elements.")
 173       if not weights.shape.is_compatible_with([num_datasets]):
 174         raise ValueError(
 175             "`weights` must be a vector of length `len(datasets)`.")
 176 
 177       # The `stateless_multinomial()` op expects log-probabilities, as opposed
 178       # to weights.
 179       logits = array_ops.expand_dims(math_ops.log(weights, name="logits"), 0)
 180 
 181     # NOTE(mrry): We only specialize when `weights` is not a `Dataset`. When it
 182     # is a `Dataset`, it is possible that evaluating it has a side effect the
 183     # user depends on.
 184     if len(datasets) == 1:
 185       return datasets[0]
 186 
 187     def select_dataset_constant_logits(seed):
 188       return array_ops.squeeze(
 189           gen_stateless_random_ops.stateless_multinomial(logits, 1, seed=seed),
 190           axis=[0, 1])
 191 
 192     selector_input = dataset_ops.MapDataset(
 193         random_ops.RandomDataset(seed).batch(2),
 194         select_dataset_constant_logits,
 195         use_inter_op_parallelism=False)
 196 
 197   else:
 198     # Use each element of the given `weights` dataset as the probability of
 199     # choosing the respective input.
 200 
 201     # The `stateless_multinomial()` op expects log-probabilities, as opposed to
 202     # weights.
 203     logits_ds = weights.map(lambda *p: math_ops.log(p, name="logits"))
 204 
 205     def select_dataset_varying_logits(logits, seed):
 206       return array_ops.squeeze(
 207           gen_stateless_random_ops.stateless_multinomial(logits, 1, seed=seed),
 208           axis=[0, 1])
 209 
 210     logits_and_seeds = dataset_ops.Dataset.zip(
 211         (logits_ds, random_ops.RandomDataset(seed).batch(2)))
 212     selector_input = dataset_ops.MapDataset(
 213         logits_and_seeds,
 214         select_dataset_varying_logits,
 215         use_inter_op_parallelism=False)
 216 
 217   return _DirectedInterleaveDataset(selector_input, datasets)

 220 @tf_export("data.experimental.choose_from_datasets")
 221 def choose_from_datasets(datasets, choice_dataset):
 222   """Creates a dataset that deterministically chooses elements from `datasets`.
 223 
 224   For example, given the following datasets:
 225 
 226   ```python
 227   datasets = [tf.data.Dataset.from_tensors("foo").repeat(),
 228               tf.data.Dataset.from_tensors("bar").repeat(),
 229               tf.data.Dataset.from_tensors("baz").repeat()]
 230 
 231   # Define a dataset containing `[0, 1, 2, 0, 1, 2, 0, 1, 2]`.
 232   choice_dataset = tf.data.Dataset.range(3).repeat(3)
 233 
 234   result = tf.data.experimental.choose_from_datasets(datasets, choice_dataset)
 235   ```
 236 
 237   The elements of `result` will be:
 238 
 239   ```
 240   "foo", "bar", "baz", "foo", "bar", "baz", "foo", "bar", "baz"
 241   ```
 242 
 243   Args:
 244     datasets: A list of `tf.data.Dataset` objects with compatible structure.
 245     choice_dataset: A `tf.data.Dataset` of scalar `tf.int64` tensors between
 246       `0` and `len(datasets) - 1`.
 247 
 248   Returns:
 249     A dataset that interleaves elements from `datasets` according to the values
 250     of `choice_dataset`.
 251 
 252   Raises:
 253     TypeError: If the `datasets` or `choice_dataset` arguments have the wrong
 254       type.
 255   """
 256   if not (choice_dataset.output_types == dtypes.int64
 257           and choice_dataset.output_shapes.is_compatible_with(
 258               tensor_shape.scalar())
 259           and choice_dataset.output_classes == ops.Tensor):
 260     raise TypeError("`choice_dataset` must be a dataset of scalar "
 261                     "`tf.int64` tensors.")
 262   return _DirectedInterleaveDataset(choice_dataset, datasets)
