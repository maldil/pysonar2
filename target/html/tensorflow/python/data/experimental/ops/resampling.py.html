  37 @tf_export("data.experimental.rejection_resample")
  38 def rejection_resample(class_func, target_dist, initial_dist=None, seed=None):
  39   """A transformation that resamples a dataset to achieve a target distribution.
  40 
  41   **NOTE** Resampling is performed via rejection sampling; some fraction
  42   of the input values will be dropped.
  43 
  44   Args:
  45     class_func: A function mapping an element of the input dataset to a scalar
  46       `tf.int32` tensor. Values should be in `[0, num_classes)`.
  47     target_dist: A floating point type tensor, shaped `[num_classes]`.
  48     initial_dist: (Optional.)  A floating point type tensor, shaped
  49       `[num_classes]`.  If not provided, the true class distribution is
  50       estimated live in a streaming fashion.
  51     seed: (Optional.) Python integer seed for the resampler.
  52 
  53   Returns:
  54     A `Dataset` transformation function, which can be passed to
  55     `tf.data.Dataset.apply`.
  56   """
  57   def _apply_fn(dataset):
  58     """Function from `Dataset` to `Dataset` that applies the transformation."""
  59     target_dist_t = ops.convert_to_tensor(target_dist, name="target_dist")
  60     class_values_ds = dataset.map(class_func)
  61 
  62     # Get initial distribution.
  63     if initial_dist is not None:
  64       initial_dist_t = ops.convert_to_tensor(initial_dist, name="initial_dist")
  65       acceptance_dist, prob_of_original = (
  66           _calculate_acceptance_probs_with_mixing(initial_dist_t,
  67                                                   target_dist_t))
  68       initial_dist_ds = dataset_ops.Dataset.from_tensors(
  69           initial_dist_t).repeat()
  70       acceptance_dist_ds = dataset_ops.Dataset.from_tensors(
  71           acceptance_dist).repeat()
  72       prob_of_original_ds = dataset_ops.Dataset.from_tensors(
  73           prob_of_original).repeat()
  74     else:
  75       initial_dist_ds = _estimate_initial_dist_ds(
  76           target_dist_t, class_values_ds)
  77       acceptance_and_original_prob_ds = initial_dist_ds.map(
  78           lambda initial: _calculate_acceptance_probs_with_mixing(  # pylint: disable=g-long-lambda
  79               initial, target_dist_t))
  80       acceptance_dist_ds = acceptance_and_original_prob_ds.map(
  81           lambda accept_prob, _: accept_prob)
  82       prob_of_original_ds = acceptance_and_original_prob_ds.map(
  83           lambda _, prob_original: prob_original)
  84     filtered_ds = _filter_ds(dataset, acceptance_dist_ds, initial_dist_ds,
  85                              class_values_ds, seed)
  86     # Prefetch filtered dataset for speed.
  87     filtered_ds = filtered_ds.prefetch(3)
  88 
  89     prob_original_static = _get_prob_original_static(
  90         initial_dist_t, target_dist_t) if initial_dist is not None else None
  91     if prob_original_static == 1:
  92       return dataset_ops.Dataset.zip((class_values_ds, dataset))
  93     elif prob_original_static == 0:
  94       return filtered_ds
  95     else:
  96       return interleave_ops.sample_from_datasets(
  97           [dataset_ops.Dataset.zip((class_values_ds, dataset)), filtered_ds],
  98           weights=prob_of_original_ds.map(lambda prob: [(prob, 1.0 - prob)]),
  99           seed=seed)
 100 
 101   return _apply_fn

 104 def _get_prob_original_static(initial_dist_t, target_dist_t):
 105   """Returns the static probability of sampling from the original.
 106 
 107   `tensor_util.constant_value(prob_of_original)` returns `None` if it encounters
 108   an Op that it isn't defined for. We have some custom logic to avoid this.
 109 
 110   Args:
 111     initial_dist_t: A tensor of the initial distribution.
 112     target_dist_t: A tensor of the target distribution.
 113 
 114   Returns:
 115     The probability of sampling from the original distribution as a constant,
 116     if it is a constant, or `None`.
 117   """
 118   init_static = tensor_util.constant_value(initial_dist_t)
 119   target_static = tensor_util.constant_value(target_dist_t)
 120 
 121   if init_static is None or target_static is None:
 122     return None
 123   else:
 124     return np.min(target_static / init_static)

 127 def _filter_ds(dataset, acceptance_dist_ds, initial_dist_ds, class_values_ds,
 128                seed):
 129   """Filters a dataset based on per-class acceptance probabilities.
 130 
 131   Args:
 132     dataset: The dataset to be filtered.
 133     acceptance_dist_ds: A dataset of acceptance probabilities.
 134     initial_dist_ds: A dataset of the initial probability distribution, given or
 135         estimated.
 136     class_values_ds: A dataset of the corresponding classes.
 137     seed: (Optional.) Python integer seed for the resampler.
 138 
 139   Returns:
 140     A dataset of (class value, data) after filtering.
 141   """
 142   def maybe_warn_on_large_rejection(accept_dist, initial_dist):
 143     proportion_rejected = math_ops.reduce_sum((1 - accept_dist) * initial_dist)
 144     return control_flow_ops.cond(
 145         math_ops.less(proportion_rejected, .5),
 146         lambda: accept_dist,
 147         lambda: logging_ops.Print(  # pylint: disable=g-long-lambda
 148             accept_dist, [proportion_rejected, initial_dist, accept_dist],
 149             message="Proportion of examples rejected by sampler is high: ",
 150             summarize=100,
 151             first_n=10))
 152 
 153   acceptance_dist_ds = (dataset_ops.Dataset.zip((acceptance_dist_ds,
 154                                                  initial_dist_ds))
 155                         .map(maybe_warn_on_large_rejection))
 156 
 157   def _gather_and_copy(class_val, acceptance_prob, data):
 158     return class_val, array_ops.gather(acceptance_prob, class_val), data
 159 
 160   current_probabilities_and_class_and_data_ds = dataset_ops.Dataset.zip(
 161       (class_values_ds, acceptance_dist_ds, dataset)).map(_gather_and_copy)
 162   filtered_ds = (
 163       current_probabilities_and_class_and_data_ds
 164       .filter(lambda _1, p, _2: random_ops.random_uniform([], seed=seed) < p))
 165   return filtered_ds.map(lambda class_value, _, data: (class_value, data))

 168 def _estimate_initial_dist_ds(
 169     target_dist_t, class_values_ds, dist_estimation_batch_size=32,
 170     smoothing_constant=10):
 171   num_classes = (target_dist_t.shape[0].value or
 172                  array_ops.shape(target_dist_t)[0])
 173   initial_examples_per_class_seen = array_ops.fill(
 174       [num_classes], np.int64(smoothing_constant))
 175 
 176   def update_estimate_and_tile(num_examples_per_class_seen, c):
 177     updated_examples_per_class_seen, dist = _estimate_data_distribution(
 178         c, num_examples_per_class_seen)
 179     tiled_dist = array_ops.tile(
 180         array_ops.expand_dims(dist, 0), [dist_estimation_batch_size, 1])
 181     return updated_examples_per_class_seen, tiled_dist
 182 
 183   initial_dist_ds = (class_values_ds.batch(dist_estimation_batch_size)
 184                      .apply(scan_ops.scan(initial_examples_per_class_seen,
 185                                           update_estimate_and_tile))
 186                      .apply(batching.unbatch()))
 187 
 188   return initial_dist_ds

 191 def _get_target_to_initial_ratio(initial_probs, target_probs):
 192   # Add tiny to initial_probs to avoid divide by zero.
 193   denom = (initial_probs + np.finfo(initial_probs.dtype.as_numpy_dtype).tiny)
 194   return target_probs / denom

 197 def _estimate_data_distribution(c, num_examples_per_class_seen):
 198   """Estimate data distribution as labels are seen.
 199 
 200   Args:
 201     c: The class labels.  Type `int32`, shape `[batch_size]`.
 202     num_examples_per_class_seen: Type `int64`, shape `[num_classes]`,
 203       containing counts.
 204 
 205   Returns:
 206     num_examples_per_lass_seen: Updated counts.  Type `int64`, shape
 207       `[num_classes]`.
 208     dist: The updated distribution.  Type `float32`, shape `[num_classes]`.
 209   """
 210   num_classes = num_examples_per_class_seen.get_shape()[0].value
 211   # Update the class-count based on what labels are seen in batch.
 212   num_examples_per_class_seen = math_ops.add(
 213       num_examples_per_class_seen, math_ops.reduce_sum(
 214           array_ops.one_hot(c, num_classes, dtype=dtypes.int64), 0))
 215   init_prob_estimate = math_ops.truediv(
 216       num_examples_per_class_seen,
 217       math_ops.reduce_sum(num_examples_per_class_seen))
 218   dist = math_ops.cast(init_prob_estimate, dtypes.float32)
 219   return num_examples_per_class_seen, dist

 222 def _calculate_acceptance_probs_with_mixing(initial_probs, target_probs):
 223   """Calculates the acceptance probabilities and mixing ratio.
 224 
 225   In this case, we assume that we can *either* sample from the original data
 226   distribution with probability `m`, or sample from a reshaped distribution
 227   that comes from rejection sampling on the original distribution. This
 228   rejection sampling is done on a per-class basis, with `a_i` representing the
 229   probability of accepting data from class `i`.
 230 
 231   This method is based on solving the following analysis for the reshaped
 232   distribution:
 233 
 234   Let F be the probability of a rejection (on any example).
 235   Let p_i be the proportion of examples in the data in class i (init_probs)
 236   Let a_i is the rate the rejection sampler should *accept* class i
 237   Let t_i is the target proportion in the minibatches for class i (target_probs)
 238 
 239   ```
 240   F = sum_i(p_i * (1-a_i))
 241     = 1 - sum_i(p_i * a_i)     using sum_i(p_i) = 1
 242   ```
 243 
 244   An example with class `i` will be accepted if `k` rejections occur, then an
 245   example with class `i` is seen by the rejector, and it is accepted. This can
 246   be written as follows:
 247 
 248   ```
 249   t_i = sum_k=0^inf(F^k * p_i * a_i)
 250       = p_i * a_j / (1 - F)    using geometric series identity, since 0 <= F < 1
 251       = p_i * a_i / sum_j(p_j * a_j)        using F from above
 252   ```
 253 
 254   Note that the following constraints hold:
 255   ```
 256   0 <= p_i <= 1, sum_i(p_i) = 1
 257   0 <= a_i <= 1
 258   0 <= t_i <= 1, sum_i(t_i) = 1
 259   ```
 260 
 261   A solution for a_i in terms of the other variables is the following:
 262     ```a_i = (t_i / p_i) / max_i[t_i / p_i]```
 263 
 264   If we try to minimize the amount of data rejected, we get the following:
 265 
 266   M_max = max_i [ t_i / p_i ]
 267   M_min = min_i [ t_i / p_i ]
 268 
 269   The desired probability of accepting data if it comes from class `i`:
 270 
 271   a_i = (t_i/p_i - m) / (M_max - m)
 272 
 273   The desired probability of pulling a data element from the original dataset,
 274   rather than the filtered one:
 275 
 276   m = M_min
 277 
 278   Args:
 279     initial_probs: A Tensor of the initial probability distribution, given or
 280       estimated.
 281     target_probs: A Tensor of the corresponding classes.
 282 
 283   Returns:
 284     (A 1D Tensor with the per-class acceptance probabilities, the desired
 285     probability of pull from the original distribution.)
 286   """
 287   ratio_l = _get_target_to_initial_ratio(initial_probs, target_probs)
 288   max_ratio = math_ops.reduce_max(ratio_l)
 289   min_ratio = math_ops.reduce_min(ratio_l)
 290 
 291   # Target prob to sample from original distribution.
 292   m = min_ratio
 293 
 294   # TODO(joelshor): Simplify fraction, if possible.
 295   a_i = (ratio_l - m) / (max_ratio - m)
 296   return a_i, m
