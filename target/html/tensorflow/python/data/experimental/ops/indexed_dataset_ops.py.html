  31 class MaterializedIndexedDataset(object):
  32   """MaterializedIndexedDataset is highly experimental!
  33   """
  34 
  35   def __init__(self, materialized_resource, materializer, output_classes,
  36                output_types, output_shapes):
  37     self._materialized_resource = materialized_resource
  38     self._materializer = materializer
  39     self._output_classes = output_classes
  40     self._output_types = output_types
  41     self._output_shapes = output_shapes
  42 
  43   @property
  44   def initializer(self):
  45     if self._materializer is not None:
  46       return self._materializer
  47     raise ValueError("MaterializedDataset does not have a materializer")
  48 
  49   def get(self, index):
  50     """Get retrieves a value (or set of values) from the IndexedDataset.
  51 
  52     Args:
  53       index: A uint64 scalar or vector tensor with the indices to retrieve.
  54 
  55     Returns:
  56       A tensor containing the values corresponding to `index`.
  57     """
  58     # TODO(saeta): nest.pack_sequence_as(...)
  59     return ged_ops.experimental_indexed_dataset_get(
  60         self._materialized_resource,
  61         index,
  62         output_types=nest.flatten(
  63             sparse.as_dense_types(self._output_types, self._output_classes)),
  64         output_shapes=nest.flatten(
  65             sparse.as_dense_types(self._output_shapes, self._output_classes)))

  68 class IndexedDataset(dataset_ops.Dataset):
  69   """IndexedDataset is highly experimental!
  70   """
  71 
  72   def __init__(self):
  73     pass
  74 
  75   def materialize(self, shared_name=None, container=None):
  76     """Materialize creates a MaterializedIndexedDataset.
  77 
  78     IndexedDatasets can be combined through operations such as TBD. Therefore,
  79     they are only materialized when absolutely required.
  80 
  81     Args:
  82       shared_name: a string for the shared name to use for the resource.
  83       container: a string for the container to store the resource.
  84 
  85     Returns:
  86       A MaterializedIndexedDataset.
  87     """
  88     if container is None:
  89       container = ""
  90     if shared_name is None:
  91       shared_name = ""
  92     materialized_resource = (
  93         ged_ops.experimental_materialized_index_dataset_handle(
  94             container=container,
  95             shared_name=shared_name,
  96             output_types=nest.flatten(
  97                 sparse.as_dense_types(self.output_types, self.output_classes)),
  98             output_shapes=nest.flatten(
  99                 sparse.as_dense_types(self.output_shapes,
 100                                       self.output_classes))))
 101 
 102     with ops.colocate_with(materialized_resource):
 103       materializer = ged_ops.experimental_indexed_dataset_materialize(
 104           self._as_variant_tensor(), materialized_resource)
 105     return MaterializedIndexedDataset(materialized_resource, materializer,
 106                                       self.output_classes, self.output_types,
 107                                       self.output_shapes)

 109   @abc.abstractproperty
 110   def output_types(self):
 111     """Returns the type of each component of an element of this IndexedDataset.
 112 
 113     Returns:
 114       A nested structure of `tf.DType` objects corresponding to each component
 115       of an element of this IndexedDataset.
 116     """
 117     raise NotImplementedError("IndexedDataset.output_types")
 118 
 119   @abc.abstractproperty
 120   def output_classes(self):
 121     """Returns the class of each component of an element of this IndexedDataset.
 122 
 123     The expected values are `tf.Tensor` and `tf.SparseTensor`.
 124 
 125     Returns:
 126       A nested structure of Python `type` objects corresponding to each
 127       component of an element of this IndexedDataset.
 128     """
 129     raise NotImplementedError("IndexedDataset.output_classes")
 130 
 131   @abc.abstractproperty
 132   def output_shapes(self):
 133     """Returns the shape of each component of an element of this IndexedDataset.
 134 
 135     Returns:
 136       A nested structure of `tf.TensorShape` objects corresponding to each
 137       component of an element of this IndexedDataset.
 138     """
 139     raise NotImplementedError("IndexedDataset.output_shapes")
 140 
 141   @abc.abstractmethod
 142   def _as_variant_tensor(self):
 143     """Creates a `tf.variant` `tf.Tensor` representing this IndexedDataset.
 144 
 145     Returns:
 146       A scalar `tf.Tensor` of `tf.variant` type, which represents this
 147       IndexedDataset.
 148     """
 149     raise NotImplementedError("IndexedDataset._as_variant_tensor")

 152 class IdentityIndexedDataset(IndexedDataset):
 153   """IdentityIndexedDataset is a trivial indexed dataset used for testing.
 154   """
 155 
 156   def __init__(self, size):
 157     super(IdentityIndexedDataset, self).__init__()
 158     # TODO(saeta): Verify _size is a scalar!
 159     self._size = ops.convert_to_tensor(size, dtype=dtypes.uint64, name="size")
 160 
 161   @property
 162   def output_types(self):
 163     return dtypes.uint64
 164 
 165   @property
 166   def output_classes(self):
 167     return ops.Tensor
 168 
 169   @property
 170   def output_shapes(self):
 171     return tensor_shape.scalar()
 172 
 173   def _as_variant_tensor(self):
 174     return ged_ops.experimental_identity_indexed_dataset(self._size)
 175 
 176   def _inputs(self):
 177     return []
