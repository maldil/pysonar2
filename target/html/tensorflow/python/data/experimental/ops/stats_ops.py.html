  27 @tf_export("data.experimental.StatsAggregator")
  28 class StatsAggregator(object):
  29   """A stateful resource that aggregates statistics from one or more iterators.
  30 
  31   To record statistics, use one of the custom transformation functions defined
  32   in this module when defining your `tf.data.Dataset`. All statistics will be
  33   aggregated by the `StatsAggregator` that is associated with a particular
  34   iterator (see below). For example, to record the latency of producing each
  35   element by iterating over a dataset:
  36 
  37   ```python
  38   dataset = ...
  39   dataset = dataset.apply(tf.data.experimental.latency_stats("total_bytes"))
  40   ```
  41 
  42   To associate a `StatsAggregator` with a `tf.data.Dataset` object, use
  43   the following pattern:
  44 
  45   ```python
  46   stats_aggregator = stats_ops.StatsAggregator()
  47   dataset = ...
  48 
  49   # Apply `set_stats_aggregator` to associate `dataset` with `stats_aggregator`.
  50   dataset = dataset.apply(
  51       tf.data.experimental.set_stats_aggregator(stats_aggregator))
  52   iterator = dataset.make_one_shot_iterator()
  53   ```
  54 
  55   To get a protocol buffer summary of the currently aggregated statistics,
  56   use the `StatsAggregator.get_summary()` tensor. The easiest way to do this
  57   is to add the returned tensor to the `tf.GraphKeys.SUMMARIES` collection,
  58   so that the summaries will be included with any existing summaries.
  59 
  60   ```python
  61   stats_aggregator = stats_ops.StatsAggregator()
  62   # ...
  63   stats_summary = stats_aggregator.get_summary()
  64   tf.add_to_collection(tf.GraphKeys.SUMMARIES, stats_summary)
  65   ```
  66 
  67   Note: This interface is experimental and expected to change. In particular,
  68   we expect to add other implementations of `StatsAggregator` that provide
  69   different ways of exporting statistics, and add more types of statistics.
  70   """
  71 
  72   def __init__(self):
  73     """Creates a `StatsAggregator`."""
  74     self._resource = gen_dataset_ops.stats_aggregator_handle()
  75 
  76   # TODO(b/116314787): Update this/add support for V2 summary API.
  77   def get_summary(self):
  78     """Returns a string `tf.Tensor` that summarizes the aggregated statistics.
  79 
  80     The returned tensor will contain a serialized `tf.summary.Summary` protocol
  81     buffer, which can be used with the standard TensorBoard logging facilities.
  82 
  83     Returns:
  84       A scalar string `tf.Tensor` that summarizes the aggregated statistics.
  85     """
  86     return gen_dataset_ops.stats_aggregator_summary(self._resource)

  89 class _SetStatsAggregatorDataset(dataset_ops.UnaryDataset):
  90   """A `Dataset` that acts as an identity, and sets given stats_aggregator."""
  91 
  92   def __init__(self, input_dataset, stats_aggregator):
  93     super(_SetStatsAggregatorDataset, self).__init__(input_dataset)
  94     self._input_dataset = input_dataset
  95     self._stats_aggregator = stats_aggregator
  96 
  97   def _as_variant_tensor(self):
  98     return gen_dataset_ops.set_stats_aggregator_dataset(
  99         self._input_dataset._as_variant_tensor(),  # pylint: disable=protected-access
 100         self._stats_aggregator._resource,  # pylint: disable=protected-access
 101         **dataset_ops.flat_structure(self))
 102 
 103   @property
 104   def output_shapes(self):
 105     return self._input_dataset.output_shapes
 106 
 107   @property
 108   def output_types(self):
 109     return self._input_dataset.output_types
 110 
 111   @property
 112   def output_classes(self):
 113     return self._input_dataset.output_classes

 116 @tf_export("data.experimental.set_stats_aggregator")
 117 def set_stats_aggregator(stats_aggregator):
 118   """Set the given `stats_aggregator` for aggregating the input dataset stats.
 119 
 120   Args:
 121     stats_aggregator: A `tf.data.experimental.StatsAggregator` object.
 122 
 123   Returns:
 124     A `Dataset` transformation function, which can be passed to
 125     `tf.data.Dataset.apply`.
 126   """
 127 
 128   def _apply_fn(dataset):
 129     return _SetStatsAggregatorDataset(dataset, stats_aggregator)
 130 
 131   return _apply_fn

 148     `tf.data.Dataset.apply`.

 152     return _StatsDataset(dataset, gen_dataset_ops.bytes_produced_stats_dataset,

 180 class _StatsDataset(dataset_ops.UnaryDataset):
 181   """A `Dataset` that acts as an identity, and also records statistics."""
 182 
 183   def __init__(self, input_dataset, op_function, tag):
 184     super(_StatsDataset, self).__init__(input_dataset)
 185     self._input_dataset = input_dataset
 186     self._op_function = op_function
 187     self._tag = ops.convert_to_tensor(tag, dtype=dtypes.string)
 188 
 189   def _as_variant_tensor(self):
 190     return self._op_function(
 191         self._input_dataset._as_variant_tensor(),  # pylint: disable=protected-access
 192         self._tag,
 193         **dataset_ops.flat_structure(self))
 194 
 195   @property
 196   def output_shapes(self):
 197     return self._input_dataset.output_shapes
 198 
 199   @property
 200   def output_types(self):
 201     return self._input_dataset.output_types
 202 
 203   @property
 204   def output_classes(self):
 205     return self._input_dataset.output_classes
