  48 def _is_valid_int32(str_val):
  49   try:
  50     # Checks equality to prevent int32 overflow
  51     return dtypes.int32.as_numpy_dtype(str_val) == dtypes.int64.as_numpy_dtype(
  52         str_val)
  53   except (ValueError, OverflowError):
  54     return False
  55 
  56 
  57 def _is_valid_int64(str_val):
  58   try:
  59     dtypes.int64.as_numpy_dtype(str_val)
  60     return True
  61   except (ValueError, OverflowError):
  62     return False
  63

  65 def _is_valid_float(str_val, float_dtype):
  66   try:
  67     return float_dtype.as_numpy_dtype(str_val) < np.inf
  68   except ValueError:
  69     return False

  72 def _infer_type(str_val, na_value, prev_type):
  73   """Given a string, infers its tensor type.
  74 
  75   Infers the type of a value by picking the least 'permissive' type possible,
  76   while still allowing the previous type inference for this column to be valid.
  77 
  78   Args:
  79     str_val: String value to infer the type of.
  80     na_value: Additional string to recognize as a NA/NaN CSV value.
  81     prev_type: Type previously inferred based on values of this column that
  82       we've seen up till now.
  83   Returns:
  84     Inferred dtype.
  85   """
  86   if str_val in ("", na_value):
  87     # If the field is null, it gives no extra information about its type
  88     return prev_type
  89 
  90   type_list = [
  91       dtypes.int32, dtypes.int64, dtypes.float32, dtypes.float64, dtypes.string
  92   ]  # list of types to try, ordered from least permissive to most
  93 
  94   type_functions = [
  95       _is_valid_int32,
  96       _is_valid_int64,
  97       lambda str_val: _is_valid_float(str_val, dtypes.float32),
  98       lambda str_val: _is_valid_float(str_val, dtypes.float64),
  99       lambda str_val: True,
 100   ]  # Corresponding list of validation functions
 101 
 102   for i in range(len(type_list)):
 103     validation_fn = type_functions[i]
 104     if validation_fn(str_val) and (prev_type is None or
 105                                    prev_type in type_list[:i + 1]):
 106       return type_list[i]

 109 def _next_csv_row(filenames, num_cols, field_delim, use_quote_delim, header):
 110   """Generator that yields rows of CSV file(s) in order."""
 111   for fn in filenames:
 112     with file_io.FileIO(fn, "r") as f:
 113       rdr = csv.reader(
 114           f,
 115           delimiter=field_delim,
 116           quoting=csv.QUOTE_MINIMAL if use_quote_delim else csv.QUOTE_NONE)
 117       if header:
 118         next(rdr)  # Skip header lines
 119 
 120       for csv_row in rdr:
 121         if len(csv_row) != num_cols:
 122           raise ValueError(
 123               "Problem inferring types: CSV row has different number of fields "
 124               "than expected.")
 125         yield csv_row

 128 def _infer_column_defaults(filenames, num_cols, field_delim, use_quote_delim,
 129                            na_value, header, num_rows_for_inference,
 130                            select_columns):
 131   """Infers column types from the first N valid CSV records of files."""
 132   if select_columns is None:
 133     select_columns = range(num_cols)
 134   inferred_types = [None] * len(select_columns)
 135 
 136   for i, csv_row in enumerate(
 137       _next_csv_row(filenames, num_cols, field_delim, use_quote_delim, header)):
 138     if num_rows_for_inference is not None and i >= num_rows_for_inference:
 139       break
 140 
 141     for j, col_index in enumerate(select_columns):
 142       inferred_types[j] = _infer_type(csv_row[col_index], na_value,
 143                                       inferred_types[j])
 144 
 145   # Replace None's with a default type
 146   inferred_types = [t or dtypes.string for t in inferred_types]
 147   # Default to 0 or '' for null values
 148   return [
 149       constant_op.constant([0 if t is not dtypes.string else ""], dtype=t)
 150       for t in inferred_types
 151   ]

 154 def _infer_column_names(filenames, field_delim, use_quote_delim):
 155   """Infers column names from first rows of files."""
 156   csv_kwargs = {
 157       "delimiter": field_delim,
 158       "quoting": csv.QUOTE_MINIMAL if use_quote_delim else csv.QUOTE_NONE
 159   }
 160   with file_io.FileIO(filenames[0], "r") as f:
 161     try:
 162       column_names = next(csv.reader(f, **csv_kwargs))
 163     except StopIteration:
 164       raise ValueError(("Received StopIteration when reading the header line "
 165                         "of %s.  Empty file?") % filenames[0])
 166 
 167   for name in filenames[1:]:
 168     with file_io.FileIO(name, "r") as f:
 169       try:
 170         if next(csv.reader(f, **csv_kwargs)) != column_names:
 171           raise ValueError(
 172               "Files have different column names in the header row.")
 173       except StopIteration:
 174         raise ValueError(("Received StopIteration when reading the header line "
 175                           "of %s.  Empty file?") % filenames[0])
 176   return column_names

 179 def _get_sorted_col_indices(select_columns, column_names):
 180   """Transforms select_columns argument into sorted column indices."""
 181   names_to_indices = {n: i for i, n in enumerate(column_names)}
 182   num_cols = len(column_names)
 183   for i, v in enumerate(select_columns):
 184     if isinstance(v, int):
 185       if v < 0 or v >= num_cols:
 186         raise ValueError(
 187             "Column index %d specified in select_columns out of valid range." %
 188             v)
 189       continue
 190     if v not in names_to_indices:
 191       raise ValueError(
 192           "Value '%s' specified in select_columns not a valid column index or "
 193           "name." % v)
 194     select_columns[i] = names_to_indices[v]
 195 
 196   # Sort and ensure there are no duplicates
 197   result = sorted(set(select_columns))
 198   if len(result) != len(select_columns):
 199     raise ValueError("select_columns contains duplicate columns")
 200   return result

 203 def _maybe_shuffle_and_repeat(
 204     dataset, num_epochs, shuffle, shuffle_buffer_size, shuffle_seed):
 205   """Optionally shuffle and repeat dataset, as requested."""
 206   if num_epochs != 1 and shuffle:
 207     # Use shuffle_and_repeat for perf
 208     return dataset.apply(
 209         shuffle_ops.shuffle_and_repeat(shuffle_buffer_size, num_epochs,
 210                                        shuffle_seed))
 211   elif shuffle:
 212     return dataset.shuffle(shuffle_buffer_size, shuffle_seed)
 213   elif num_epochs != 1:
 214     return dataset.repeat(num_epochs)
 215   return dataset

 218 def make_tf_record_dataset(file_pattern,
 219                            batch_size,
 220                            parser_fn=None,
 221                            num_epochs=None,
 222                            shuffle=True,
 223                            shuffle_buffer_size=None,
 224                            shuffle_seed=None,
 225                            prefetch_buffer_size=optimization.AUTOTUNE,
 226                            num_parallel_reads=None,
 227                            num_parallel_parser_calls=None,
 228                            drop_final_batch=False):
 229   """Reads and optionally parses TFRecord files into a dataset.
 230 
 231   Provides common functionality such as batching, optional parsing, shuffling,
 232   and performant defaults.
 233 
 234   Args:
 235     file_pattern: List of files or patterns of TFRecord file paths.
 236       See `tf.gfile.Glob` for pattern rules.
 237     batch_size: An int representing the number of records to combine
 238       in a single batch.
 239     parser_fn: (Optional.) A function accepting string input to parse
 240       and process the record contents. This function must map records
 241       to components of a fixed shape, so they may be batched. By
 242       default, uses the record contents unmodified.
 243     num_epochs: (Optional.) An int specifying the number of times this
 244       dataset is repeated.  If None (the default), cycles through the
 245       dataset forever.
 246     shuffle: (Optional.) A bool that indicates whether the input
 247       should be shuffled. Defaults to `True`.
 248     shuffle_buffer_size: (Optional.) Buffer size to use for
 249       shuffling. A large buffer size ensures better shuffling, but
 250       increases memory usage and startup time.
 251     shuffle_seed: (Optional.) Randomization seed to use for shuffling.
 252     prefetch_buffer_size: (Optional.) An int specifying the number of
 253       feature batches to prefetch for performance improvement.
 254       Defaults to auto-tune. Set to 0 to disable prefetching.
 255     num_parallel_reads: (Optional.) Number of threads used to read
 256       records from files. By default or if set to a value >1, the
 257       results will be interleaved.
 258     num_parallel_parser_calls: (Optional.) Number of parallel
 259       records to parse in parallel. Defaults to an automatic selection.
 260     drop_final_batch: (Optional.) Whether the last batch should be
 261       dropped in case its size is smaller than `batch_size`; the
 262       default behavior is not to drop the smaller batch.
 263 
 264   Returns:
 265     A dataset, where each element matches the output of `parser_fn`
 266     except it will have an additional leading `batch-size` dimension,
 267     or a `batch_size`-length 1-D tensor of strings if `parser_fn` is
 268     unspecified.
 269   """
 270   files = dataset_ops.Dataset.list_files(
 271       file_pattern, shuffle=shuffle, seed=shuffle_seed)
 272 
 273   if num_parallel_reads is None:
 274     # Note: We considered auto-tuning this value, but there is a concern
 275     # that this affects the mixing of records from different files, which
 276     # could affect training convergence/accuracy, so we are defaulting to
 277     # a constant for now.
 278     num_parallel_reads = 24
 279   dataset = core_readers.TFRecordDataset(
 280       files, num_parallel_reads=num_parallel_reads)
 281 
 282   if shuffle_buffer_size is None:
 283     # TODO(josh11b): Auto-tune this value when not specified
 284     shuffle_buffer_size = 10000
 285   dataset = _maybe_shuffle_and_repeat(
 286       dataset, num_epochs, shuffle, shuffle_buffer_size, shuffle_seed)
 287 
 288   # NOTE(mrry): We set `drop_final_batch=True` when `num_epochs is None` to
 289   # improve the shape inference, because it makes the batch dimension static.
 290   # It is safe to do this because in that case we are repeating the input
 291   # indefinitely, and all batches will be full-sized.
 292   drop_final_batch = drop_final_batch or num_epochs is None
 293 
 294   if parser_fn is None:
 295     dataset = dataset.batch(batch_size, drop_remainder=drop_final_batch)
 296   else:
 297     # TODO(josh11b): if num_parallel_parser_calls is None, use some function
 298     # of num cores instead of map_and_batch's default behavior of one batch.
 299     dataset = dataset.apply(batching.map_and_batch(
 300         parser_fn, batch_size, num_parallel_calls=num_parallel_parser_calls,
 301         drop_remainder=drop_final_batch))
 302 
 303   if prefetch_buffer_size == 0:
 304     return dataset
 305   else:
 306     return dataset.prefetch(buffer_size=prefetch_buffer_size)

 309 @tf_export("data.experimental.make_csv_dataset")
 310 def make_csv_dataset(
 311     file_pattern,
 312     batch_size,
 313     column_names=None,
 314     column_defaults=None,
 315     label_name=None,
 316     select_columns=None,
 317     field_delim=",",
 318     use_quote_delim=True,
 319     na_value="",
 320     header=True,
 321     num_epochs=None,
 322     shuffle=True,
 323     shuffle_buffer_size=10000,
 324     shuffle_seed=None,
 325     prefetch_buffer_size=optimization.AUTOTUNE,
 326     num_parallel_reads=1,
 327     sloppy=False,
 328     num_rows_for_inference=100,
 329     compression_type=None,
 330 ):
 331   """Reads CSV files into a dataset.
 332 
 333   Reads CSV files into a dataset, where each element is a (features, labels)
 334   tuple that corresponds to a batch of CSV rows. The features dictionary
 335   maps feature column names to `Tensor`s containing the corresponding
 336   feature data, and labels is a `Tensor` containing the batch's label data.
 337 
 338   Args:
 339     file_pattern: List of files or patterns of file paths containing CSV
 340       records. See `tf.gfile.Glob` for pattern rules.
 341     batch_size: An int representing the number of records to combine
 342       in a single batch.
 343     column_names: An optional list of strings that corresponds to the CSV
 344       columns, in order. One per column of the input record. If this is not
 345       provided, infers the column names from the first row of the records.
 346       These names will be the keys of the features dict of each dataset element.
 347     column_defaults: A optional list of default values for the CSV fields. One
 348       item per selected column of the input record. Each item in the list is
 349       either a valid CSV dtype (float32, float64, int32, int64, or string), or a
 350       `Tensor` with one of the aforementioned types. The tensor can either be
 351       a scalar default value (if the column is optional), or an empty tensor (if
 352       the column is required). If a dtype is provided instead of a tensor, the
 353       column is also treated as required. If this list is not provided, tries
 354       to infer types based on reading the first num_rows_for_inference rows of
 355       files specified, and assumes all columns are optional, defaulting to `0`
 356       for numeric values and `""` for string values. If both this and
 357       `select_columns` are specified, these must have the same lengths, and
 358       `column_defaults` is assumed to be sorted in order of increasing column
 359       index.
 360     label_name: A optional string corresponding to the label column. If
 361       provided, the data for this column is returned as a separate `Tensor` from
 362       the features dictionary, so that the dataset complies with the format
 363       expected by a `tf.Estimator.train` or `tf.Estimator.evaluate` input
 364       function.
 365     select_columns: An optional list of integer indices or string column
 366       names, that specifies a subset of columns of CSV data to select. If
 367       column names are provided, these must correspond to names provided in
 368       `column_names` or inferred from the file header lines. When this argument
 369       is specified, only a subset of CSV columns will be parsed and returned,
 370       corresponding to the columns specified. Using this results in faster
 371       parsing and lower memory usage. If both this and `column_defaults` are
 372       specified, these must have the same lengths, and `column_defaults` is
 373       assumed to be sorted in order of increasing column index.
 374     field_delim: An optional `string`. Defaults to `","`. Char delimiter to
 375       separate fields in a record.
 376     use_quote_delim: An optional bool. Defaults to `True`. If false, treats
 377       double quotation marks as regular characters inside of the string fields.
 378     na_value: Additional string to recognize as NA/NaN.
 379     header: A bool that indicates whether the first rows of provided CSV files
 380       correspond to header lines with column names, and should not be included
 381       in the data.
 382     num_epochs: An int specifying the number of times this dataset is repeated.
 383       If None, cycles through the dataset forever.
 384     shuffle: A bool that indicates whether the input should be shuffled.
 385     shuffle_buffer_size: Buffer size to use for shuffling. A large buffer size
 386       ensures better shuffling, but increases memory usage and startup time.
 387     shuffle_seed: Randomization seed to use for shuffling.
 388     prefetch_buffer_size: An int specifying the number of feature
 389       batches to prefetch for performance improvement. Recommended value is the
 390       number of batches consumed per training step. Defaults to auto-tune.
 391 
 392     num_parallel_reads: Number of threads used to read CSV records from files.
 393       If >1, the results will be interleaved.
 394     sloppy: If `True`, reading performance will be improved at
 395       the cost of non-deterministic ordering. If `False`, the order of elements
 396       produced is deterministic prior to shuffling (elements are still
 397       randomized if `shuffle=True`. Note that if the seed is set, then order
 398       of elements after shuffling is deterministic). Defaults to `False`.
 399     num_rows_for_inference: Number of rows of a file to use for type inference
 400       if record_defaults is not provided. If None, reads all the rows of all
 401       the files. Defaults to 100.
 402     compression_type: (Optional.) A `tf.string` scalar evaluating to one of
 403       `""` (no compression), `"ZLIB"`, or `"GZIP"`. Defaults to no compression.
 404 
 405   Returns:
 406     A dataset, where each element is a (features, labels) tuple that corresponds
 407     to a batch of `batch_size` CSV rows. The features dictionary maps feature
 408     column names to `Tensor`s containing the corresponding column data, and
 409     labels is a `Tensor` containing the column data for the label column
 410     specified by `label_name`.
 411 
 412   Raises:
 413     ValueError: If any of the arguments is malformed.
 414   """
 415   # Create dataset of all matching filenames
 416   filenames = _get_file_names(file_pattern, False)
 417   dataset = dataset_ops.Dataset.from_tensor_slices(filenames)
 418   if shuffle:
 419     dataset = dataset.shuffle(len(filenames), shuffle_seed)
 420 
 421   # Clean arguments; figure out column names and defaults
 422 
 423   if column_names is None:
 424     if not header:
 425       raise ValueError("Cannot infer column names without a header line.")
 426     # If column names are not provided, infer from the header lines
 427     column_names = _infer_column_names(filenames, field_delim, use_quote_delim)
 428   if len(column_names) != len(set(column_names)):
 429     raise ValueError("Cannot have duplicate column names.")
 430 
 431   if select_columns is not None:
 432     select_columns = _get_sorted_col_indices(select_columns, column_names)
 433 
 434   if column_defaults is not None:
 435     column_defaults = [
 436         constant_op.constant([], dtype=x) if x in _ACCEPTABLE_CSV_TYPES else x
 437         for x in column_defaults
 438     ]
 439   else:
 440     # If column defaults are not provided, infer from records at graph
 441     # construction time
 442     column_defaults = _infer_column_defaults(
 443         filenames, len(column_names), field_delim, use_quote_delim, na_value,
 444         header, num_rows_for_inference, select_columns)
 445 
 446   if select_columns is not None and len(column_defaults) != len(select_columns):
 447     raise ValueError(
 448         "If specified, column_defaults and select_columns must have same "
 449         "length."
 450     )
 451   if select_columns is not None and len(column_names) > len(select_columns):
 452     # Pick the relevant subset of column names
 453     column_names = [column_names[i] for i in select_columns]
 454 
 455   if label_name is not None and label_name not in column_names:
 456     raise ValueError("`label_name` provided must be one of the columns.")
 457 
 458   def filename_to_dataset(filename):
 459     return CsvDataset(
 460         filename,
 461         record_defaults=column_defaults,
 462         field_delim=field_delim,
 463         use_quote_delim=use_quote_delim,
 464         na_value=na_value,
 465         select_cols=select_columns,
 466         header=header,
 467         compression_type=compression_type,
 468     )
 469 
 470   def map_fn(*columns):
 471     """Organizes columns into a features dictionary.
 472 
 473     Args:
 474       *columns: list of `Tensor`s corresponding to one csv record.
 475     Returns:
 476       An OrderedDict of feature names to values for that particular record. If
 477       label_name is provided, extracts the label feature to be returned as the
 478       second element of the tuple.
 479     """
 480     features = collections.OrderedDict(zip(column_names, columns))
 481     if label_name is not None:
 482       label = features.pop(label_name)
 483       return features, label
 484     return features
 485 
 486   # Read files sequentially (if num_parallel_reads=1) or in parallel
 487   dataset = dataset.apply(
 488       interleave_ops.parallel_interleave(
 489           filename_to_dataset, cycle_length=num_parallel_reads, sloppy=sloppy))
 490 
 491   dataset = _maybe_shuffle_and_repeat(
 492       dataset, num_epochs, shuffle, shuffle_buffer_size, shuffle_seed)
 493 
 494   # Apply batch before map for perf, because map has high overhead relative
 495   # to the size of the computation in each map.
 496   # NOTE(mrry): We set `drop_remainder=True` when `num_epochs is None` to
 497   # improve the shape inference, because it makes the batch dimension static.
 498   # It is safe to do this because in that case we are repeating the input
 499   # indefinitely, and all batches will be full-sized.
 500   dataset = dataset.batch(batch_size=batch_size,
 501                           drop_remainder=num_epochs is None)
 502   dataset = dataset_ops.MapDataset(
 503       dataset, map_fn, use_inter_op_parallelism=False)
 504   dataset = dataset.prefetch(prefetch_buffer_size)
 505 
 506   return dataset

 512 @tf_export("data.experimental.CsvDataset")
 513 class CsvDataset(dataset_ops.DatasetSource):
 514   """A Dataset comprising lines from one or more CSV files."""
 515 
 516   def __init__(self,
 517                filenames,
 518                record_defaults,
 519                compression_type=None,
 520                buffer_size=None,
 521                header=False,
 522                field_delim=",",
 523                use_quote_delim=True,
 524                na_value="",
 525                select_cols=None):
 526     """Creates a `CsvDataset` by reading and decoding CSV files.
 527 
 528     The elements of this dataset correspond to records from the file(s).
 529     RFC 4180 format is expected for CSV files
 530     (https://tools.ietf.org/html/rfc4180)
 531     Note that we allow leading and trailing spaces with int or float field.
 532 
 533 
 534     For example, suppose we have a file 'my_file0.csv' with four CSV columns of
 535     different data types:
 536     ```
 537     abcdefg,4.28E10,5.55E6,12
 538     hijklmn,-5.3E14,,2
 539     ```
 540 
 541     We can construct a CsvDataset from it as follows:
 542     ```python
 543     dataset = tf.data.experimental.CsvDataset(
 544         "my_file*.csv",
 545         [tf.float32,  # Required field, use dtype or empty tensor
 546          tf.constant([0.0], dtype=tf.float32),  # Optional field, default to 0.0
 547          tf.int32,  # Required field, use dtype or empty tensor
 548          ],
 549         select_cols=[1,2,3]  # Only parse last three columns
 550     )
 551     ```
 552 
 553     The expected output of its iterations is:
 554     ```python
 555     next_element = dataset.make_one_shot_iterator().get_next()
 556     with tf.Session() as sess:
 557       while True:
 558         try:
 559           print(sess.run(next_element))
 560         except tf.errors.OutOfRangeError:
 561           break
 562 
 563     >> (4.28e10, 5.55e6, 12)
 564     >> (-5.3e14, 0.0, 2)
 565     ```
 566 
 567     Args:
 568       filenames: A `tf.string` tensor containing one or more filenames.
 569       record_defaults: A list of default values for the CSV fields. Each item in
 570         the list is either a valid CSV `DType` (float32, float64, int32, int64,
 571         string), or a `Tensor` object with one of the above types. One per
 572         column of CSV data, with either a scalar `Tensor` default value for the
 573         column if it is optional, or `DType` or empty `Tensor` if required. If

 660 @tf_export("data.experimental.make_batched_features_dataset")
 661 def make_batched_features_dataset(file_pattern,
 662                                   batch_size,
 663                                   features,
 664                                   reader=core_readers.TFRecordDataset,
 665                                   label_key=None,
 666                                   reader_args=None,
 667                                   num_epochs=None,
 668                                   shuffle=True,
 669                                   shuffle_buffer_size=10000,
 670                                   shuffle_seed=None,
 671                                   prefetch_buffer_size=optimization.AUTOTUNE,
 672                                   reader_num_threads=1,
 673                                   parser_num_threads=2,
 674                                   sloppy_ordering=False,
 675                                   drop_final_batch=False):
 676   """Returns a `Dataset` of feature dictionaries from `Example` protos.
 677 
 678   If label_key argument is provided, returns a `Dataset` of tuple
 679   comprising of feature dictionaries and label.
 680 
 681   Example:
 682 
 683   ```
 684   serialized_examples = [
 685     features {
 686       feature { key: "age" value { int64_list { value: [ 0 ] } } }
 687       feature { key: "gender" value { bytes_list { value: [ "f" ] } } }
 688       feature { key: "kws" value { bytes_list { value: [ "code", "art" ] } } }
 689     },
 690     features {
 691       feature { key: "age" value { int64_list { value: [] } } }
 692       feature { key: "gender" value { bytes_list { value: [ "f" ] } } }
 693       feature { key: "kws" value { bytes_list { value: [ "sports" ] } } }
 694     }
 695   ]
 696   ```
 697 
 698   We can use arguments:
 699 
 700   ```
 701   features: {
 702     "age": FixedLenFeature([], dtype=tf.int64, default_value=-1),
 703     "gender": FixedLenFeature([], dtype=tf.string),
 704     "kws": VarLenFeature(dtype=tf.string),
 705   }
 706   ```
 707 
 708   And the expected output is:
 709 
 710   ```python
 711   {
 712     "age": [[0], [-1]],
 713     "gender": [["f"], ["f"]],
 714     "kws": SparseTensor(
 715       indices=[[0, 0], [0, 1], [1, 0]],
 716       values=["code", "art", "sports"]
 717       dense_shape=[2, 2]),
 718   }
 719   ```
 720 
 721   Args:
 722     file_pattern: List of files or patterns of file paths containing
 723       `Example` records. See `tf.gfile.Glob` for pattern rules.
 724     batch_size: An int representing the number of records to combine
 725       in a single batch.
 726     features: A `dict` mapping feature keys to `FixedLenFeature` or
 727       `VarLenFeature` values. See `tf.parse_example`.
 728     reader: A function or class that can be
 729       called with a `filenames` tensor and (optional) `reader_args` and returns
 730       a `Dataset` of `Example` tensors. Defaults to `tf.data.TFRecordDataset`.
 731     label_key: (Optional) A string corresponding to the key labels are stored in
 732       `tf.Examples`. If provided, it must be one of the `features` key,
 733       otherwise results in `ValueError`.
 734     reader_args: Additional arguments to pass to the reader class.
 735     num_epochs: Integer specifying the number of times to read through the
 736       dataset. If None, cycles through the dataset forever. Defaults to `None`.
 737     shuffle: A boolean, indicates whether the input should be shuffled. Defaults
 738       to `True`.
 739     shuffle_buffer_size: Buffer size of the ShuffleDataset. A large capacity
 740       ensures better shuffling but would increase memory usage and startup time.
 741     shuffle_seed: Randomization seed to use for shuffling.
 742     prefetch_buffer_size: Number of feature batches to prefetch in order to
 743       improve performance. Recommended value is the number of batches consumed
 744       per training step. Defaults to auto-tune.
 745     reader_num_threads: Number of threads used to read `Example` records. If >1,
 746       the results will be interleaved.
 747     parser_num_threads: Number of threads to use for parsing `Example` tensors
 748       into a dictionary of `Feature` tensors.
 749     sloppy_ordering: If `True`, reading performance will be improved at
 750       the cost of non-deterministic ordering. If `False`, the order of elements
 751       produced is deterministic prior to shuffling (elements are still
 752       randomized if `shuffle=True`. Note that if the seed is set, then order
 753       of elements after shuffling is deterministic). Defaults to `False`.
 754     drop_final_batch: If `True`, and the batch size does not evenly divide the
 755       input dataset size, the final smaller batch will be dropped. Defaults to
 756       `False`.
 757 
 758   Returns:
 759     A dataset of `dict` elements, (or a tuple of `dict` elements and label).
 760     Each `dict` maps feature keys to `Tensor` or `SparseTensor` objects.
 761 
 762   Raises:
 763     ValueError: If `label_key` is not one of the `features` keys.
 764   """
 765   # Create dataset of all matching filenames
 766   filenames = _get_file_names(file_pattern, False)
 767   dataset = dataset_ops.Dataset.from_tensor_slices(filenames)
 768   if shuffle:
 769     dataset = dataset.shuffle(len(filenames), shuffle_seed)
 770 
 771   # Read `Example` records from files as tensor objects.
 772   if reader_args is None:
 773     reader_args = []
 774 
 775   # Read files sequentially (if reader_num_threads=1) or in parallel
 776   dataset = dataset.apply(
 777       interleave_ops.parallel_interleave(
 778           lambda filename: reader(filename, *reader_args),
 779           cycle_length=reader_num_threads,
 780           sloppy=sloppy_ordering))
 781 
 782   # Extract values if the `Example` tensors are stored as key-value tuples.
 783   if dataset.output_types == (dtypes.string, dtypes.string):
 784     dataset = dataset_ops.MapDataset(
 785         dataset, lambda _, v: v, use_inter_op_parallelism=False)
 786 
 787   # Apply dataset repeat and shuffle transformations.
 788   dataset = _maybe_shuffle_and_repeat(
 789       dataset, num_epochs, shuffle, shuffle_buffer_size, shuffle_seed)
 790 
 791   # NOTE(mrry): We set `drop_remainder=True` when `num_epochs is None` to
 792   # improve the shape inference, because it makes the batch dimension static.
 793   # It is safe to do this because in that case we are repeating the input
 794   # indefinitely, and all batches will be full-sized.
 795   dataset = dataset.batch(
 796       batch_size, drop_remainder=drop_final_batch or num_epochs is None)
 797 
 798   # Parse `Example` tensors to a dictionary of `Feature` tensors.
 799   dataset = dataset.apply(
 800       parsing_ops.parse_example_dataset(
 801           features, num_parallel_calls=parser_num_threads))
 802 
 803   if label_key:
 804     if label_key not in features:
 805       raise ValueError(
 806           "The `label_key` provided (%r) must be one of the `features` keys." %
 807           label_key)
 808     dataset = dataset.map(lambda x: (x, x.pop(label_key)))
 809 
 810   dataset = dataset.prefetch(prefetch_buffer_size)
 811   return dataset

 814 def _get_file_names(file_pattern, shuffle):
 815   """Parse list of file names from pattern, optionally shuffled.
 816 
 817   Args:
 818     file_pattern: File glob pattern, or list of glob patterns.
 819     shuffle: Whether to shuffle the order of file names.
 820 
 821   Returns:
 822     List of file names matching `file_pattern`.
 823 
 824   Raises:
 825     ValueError: If `file_pattern` is empty, or pattern matches no files.
 826   """
 827   if isinstance(file_pattern, list):
 828     if not file_pattern:
 829       raise ValueError("File pattern is empty.")
 830     file_names = []
 831     for entry in file_pattern:
 832       file_names.extend(gfile.Glob(entry))
 833   else:
 834     file_names = list(gfile.Glob(file_pattern))
 835 
 836   if not file_names:
 837     raise ValueError("No files match %s." % file_pattern)
 838 
 839   # Sort files so it will be deterministic for unit tests.
 840   if not shuffle:
 841     file_names = sorted(file_names)
 842   return file_names
