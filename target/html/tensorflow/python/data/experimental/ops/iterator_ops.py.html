  30 @tf_export("data.experimental.make_saveable_from_iterator")
  31 def make_saveable_from_iterator(iterator):
  32   """Returns a SaveableObject for saving/restore iterator state using Saver.
  33 
  34   Args:
  35     iterator: Iterator.
  36 
  37   For example:
  38 
  39   ```python
  40   with tf.Graph().as_default():
  41     ds = tf.data.Dataset.range(10)
  42     iterator = ds.make_initializable_iterator()
  43     # Build the iterator SaveableObject.
  44     saveable_obj = tf.data.experimental.make_saveable_from_iterator(iterator)
  45     # Add the SaveableObject to the SAVEABLE_OBJECTS collection so
  46     # it can be automatically saved using Saver.
  47     tf.add_to_collection(tf.GraphKeys.SAVEABLE_OBJECTS, saveable_obj)
  48     saver = tf.train.Saver()
  49 
  50     while continue_training:
  51       ... Perform training ...
  52       if should_save_checkpoint:
  53         saver.save()
  54   ```
  55 
  56   Note: When restoring the iterator, the existing iterator state is completely
  57   discarded. This means that any changes you may have made to the Dataset
  58   graph will be discarded as well! This includes the new Dataset graph
  59   that you may have built during validation. So, while running validation,
  60   make sure to run the initializer for the validation input pipeline after
  61   restoring the checkpoint.
  62 
  63   Note: Not all iterators support checkpointing yet. Attempting to save the
  64   state of an unsupported iterator will throw an error.
  65   """
  66   return _Saveable(iterator._iterator_resource)  # pylint: disable=protected-access
  67

  69 class _Saveable(saver_lib.BaseSaverBuilder.SaveableObject):
  70   """SaveableObject for saving/restoring iterator state."""
  71 
  72   def __init__(self, iterator_resource):
  73     serialized_iterator = gen_dataset_ops.serialize_iterator(iterator_resource)
  74     specs = [
  75         saver_lib.BaseSaverBuilder.SaveSpec(serialized_iterator, "",
  76                                             iterator_resource.name + "-state")
  77     ]
  78     super(_Saveable, self).__init__(iterator_resource, specs,
  79                                     iterator_resource.name)
  80 
  81   def restore(self, restored_tensors, unused_restored_shapes):
  82     with ops.colocate_with(self.op):
  83       return gen_dataset_ops.deserialize_iterator(self.op, restored_tensors[0])

  86 @tf_export("data.experimental.CheckpointInputPipelineHook")
  87 class CheckpointInputPipelineHook(session_run_hook.SessionRunHook):
  88   """Checkpoints input pipeline state every N steps or seconds.
  89 
  90   This hook saves the state of the iterators in the `Graph` so that when
  91   training is resumed the input pipeline continues from where it left off.
  92   This could potentially avoid overfitting in certain pipelines where the
  93   number of training steps per eval are small compared to the dataset
  94   size or if the training pipeline is pre-empted.
  95 
  96   Differences from `CheckpointSaverHook`:
  97   1. Saves only the input pipelines in the "iterators" collection and not the
  98      global variables or other saveable objects.
  99   2. Does not write the `GraphDef` and `MetaGraphDef` to the summary.
 100 
 101   Example of checkpointing the training pipeline:
 102 
 103   ```python
 104   est = tf.estimator.Estimator(model_fn)
 105   while True:
 106     est.train(
 107         train_input_fn,
 108         hooks=[tf.data.experimental.CheckpointInputPipelineHook(est)],
 109         steps=train_steps_per_eval)
 110     # Note: We do not pass the hook here.
 111     metrics = est.evaluate(eval_input_fn)
 112     if should_stop_the_training(metrics):
 113       break
 114   ```
 115 
 116   This hook should be used if the input pipeline state needs to be saved
 117   separate from the model checkpoint. Doing so may be useful for a few reasons:
 118   1. The input pipeline checkpoint may be large, if there are large shuffle
 119      or prefetch buffers for instance, and may bloat the checkpoint size.
 120   2. If the input pipeline is shared between training and validation, restoring
 121      the checkpoint during validation may override the validation input
 122      pipeline.
 123 
 124   For saving the input pipeline checkpoint alongside the model weights use
 125   `tf.data.experimental.make_saveable_from_iterator` directly to create a
 126   `SaveableObject` and add to the `SAVEABLE_OBJECTS` collection. Note, however,
 127   that you will need to be careful not to restore the training iterator during
 128   eval. You can do that by not adding the iterator to the SAVEABLE_OBJECTS
 129   collector when building the eval graph.
 130   """
 131 
 132   def __init__(self, estimator):
 133     """Initializes a `CheckpointInputPipelineHook`.
 134 
 135     Args:
 136       estimator: Estimator.
 137 
 138     Raises:
 139       ValueError: One of `save_steps` or `save_secs` should be set.
 140       ValueError: At most one of saver or scaffold should be set.
 141     """
 142     # `checkpoint_basename` is "input.ckpt" for non-distributed pipelines or
 143     # of the form "input_<task_type>_<task_id>.ckpt" for distributed pipelines.
 144     # Note: The default `checkpoint_basename` used by `CheckpointSaverHook` is
 145     # "model.ckpt". We intentionally choose the input pipeline checkpoint prefix
 146     # to be different to avoid conflicts with the model checkpoint.
 147 
 148     # pylint: disable=protected-access
 149     checkpoint_prefix = "input"
 150     if estimator._config.num_worker_replicas > 1:
 151       # Distributed setting.
 152       suffix = "_{}_{}".format(estimator._config.task_type,
 153                                estimator._config.task_id)
 154       checkpoint_prefix += suffix
 155     # pylint: enable=protected-access
 156 
 157     # We use a composition paradigm instead of inheriting from
 158     # `CheckpointSaverHook` because `Estimator` does an `isinstance` check
 159     # to check whether a `CheckpointSaverHook` is already present in the list
 160     # of hooks and if not, adds one. Inheriting from `CheckpointSaverHook`
 161     # would thwart this behavior. This hook checkpoints *only the iterators*
 162     # and not the graph variables.
 163     self._checkpoint_saver_hook = basic_session_run_hooks.CheckpointSaverHook(
 164         estimator.model_dir,
 165         save_secs=estimator._config.save_checkpoints_secs,  # pylint: disable=protected-access
 166         save_steps=estimator._config.save_checkpoints_steps,  # pylint: disable=protected-access
 167         checkpoint_basename=checkpoint_prefix + ".ckpt")
 168 
 169     # Name for the protocol buffer file that will contain the list of most
 170     # recent checkpoints stored as a `CheckpointState` protocol buffer.
 171     # This file, kept in the same directory as the checkpoint files, is
 172     # automatically managed by the `Saver` to keep track of recent checkpoints.
 173     # The default name used by the `Saver` for this file is "checkpoint". Here
 174     # we use the name "checkpoint_<checkpoint_prefix>" so that in case the
 175     # `checkpoint_dir` is the same as the model checkpoint directory, there are
 176     # no conflicts during restore.
 177     self._latest_filename = "checkpoint_" + checkpoint_prefix
 178     self._first_run = True
 179 
 180   def begin(self):
 181     # Build a Saver that saves all iterators in the `GLOBAL_ITERATORS`
 182     # collection if no `Saver` or `Scaffold` is provided.
 183     # pylint: disable=protected-access
 184     if (self._checkpoint_saver_hook._saver is None and
 185         self._checkpoint_saver_hook._scaffold is None):
 186       iterators = ops.get_collection(iterator_ops.GLOBAL_ITERATORS)
 187       saveables = [_Saveable(i) for i in iterators]
 188       self._checkpoint_saver_hook._saver = _CustomSaver(saveables,
 189                                                         self._latest_filename)
 190     # pylint: enable=protected-access
 191     self._checkpoint_saver_hook.begin()
 192 
 193   def _restore_or_save_initial_ckpt(self, session):
 194     # Ideally this should be run in after_create_session but is not for the
 195     # following reason:
 196     # Currently there is no way of enforcing an order of running the
 197     # `SessionRunHooks`. Hence it is possible that the `_DatasetInitializerHook`
 198     # is run *after* this hook. That is troublesome because
 199     # 1. If a checkpoint exists and this hook restores it, the initializer hook
 200     #    will override it.
 201     # 2. If no checkpoint exists, this hook will try to save an initialized
 202     #    iterator which will result in an exception.
 203     #
 204     # As a temporary fix we enter the following implicit contract between this
 205     # hook and the _DatasetInitializerHook.
 206     # 1. The _DatasetInitializerHook initializes the iterator in the call to
 207     #    after_create_session.
 208     # 2. This hook saves the iterator on the first call to `before_run()`, which
 209     #    is guaranteed to happen after `after_create_session()` of all hooks
 210     #    have been run.
 211 
 212     # Check if there is an existing checkpoint. If so, restore from it.
 213     # pylint: disable=protected-access
 214     latest_checkpoint_path = checkpoint_management.latest_checkpoint(
 215         self._checkpoint_saver_hook._checkpoint_dir,
 216         latest_filename=self._latest_filename)
 217     if latest_checkpoint_path:
 218       self._checkpoint_saver_hook._get_saver().restore(session,
 219                                                        latest_checkpoint_path)
 220     else:
 221       # The checkpoint saved here is the state at step "global_step".
 222       # Note: We do not save the GraphDef or MetaGraphDef here.
 223       global_step = session.run(self._checkpoint_saver_hook._global_step_tensor)
 224       self._checkpoint_saver_hook._save(session, global_step)
 225       self._checkpoint_saver_hook._timer.update_last_triggered_step(global_step)
 226     # pylint: enable=protected-access
 227 
 228   def before_run(self, run_context):
 229     if self._first_run:
 230       self._restore_or_save_initial_ckpt(run_context.session)
 231       self._first_run = False
 232     return self._checkpoint_saver_hook.before_run(run_context)
 233 
 234   def after_run(self, run_context, run_values):
 235     self._checkpoint_saver_hook.after_run(run_context, run_values)
 236 
 237   def end(self, session):
 238     self._checkpoint_saver_hook.end(session)

 241 class _CustomSaver(saver_lib.Saver):
 242   """`Saver` with a different default `latest_filename`.
 243 
 244   This is used in the `CheckpointInputPipelineHook` to avoid conflicts with
 245   the model ckpt saved by the `CheckpointSaverHook`.
 246   """
 247 
 248   def __init__(self, var_list, latest_filename):
 249     super(_CustomSaver, self).__init__(var_list)
 250     self._latest_filename = latest_filename
 251 
 252   def save(self,
 253            sess,
 254            save_path,
 255            global_step=None,
 256            latest_filename=None,
 257            meta_graph_suffix="meta",
 258            write_meta_graph=True,
 259            write_state=True,
 260            strip_default_attrs=False):
 261     return super(_CustomSaver, self).save(
 262         sess, save_path, global_step, latest_filename or self._latest_filename,
 263         meta_graph_suffix, write_meta_graph, write_state, strip_default_attrs)
