  39 def function_buffering_resource(string_arg,
  40                                 target_device,
  41                                 f,
  42                                 buffer_size,
  43                                 output_types,
  44                                 container="",
  45                                 shared_name=None,
  46                                 name=None):
  47   """Creates a FunctionBufferingResource.
  48 
  49   A FunctionBufferingResource fills up a buffer by calling a function `f` on
  50   `target_device`. `f` should take in only a single string argument as input.
  51 
  52   Args:
  53     string_arg: The single string argument to the function.
  54     target_device: The device to run `f` on.
  55     f: The function to be executed.
  56     buffer_size: Size of the buffer to be populated.
  57     output_types: The output types generated by the function.
  58     container: (Optional) string. Defaults to "".
  59     shared_name: (Optional) string.
  60     name: (Optional) string to name the op.
  61 
  62   Returns:
  63     Handle to a FunctionBufferingResource.
  64   """
  65   if shared_name is None:
  66     shared_name = ""
  67   return ged_ops.experimental_function_buffering_resource(
  68       string_arg=string_arg,
  69       target_device=target_device,
  70       shared_name=shared_name,
  71       f=f,
  72       buffer_size=buffer_size,
  73       container=container,
  74       name=name,
  75       output_types=output_types)
  76

  78 def function_buffering_resource_get_next(function_buffer_resource,
  79                                          output_types,
  80                                          name=None):
  81   return ged_ops.experimental_function_buffering_resource_get_next(
  82       function_buffer_resource=function_buffer_resource,
  83       output_types=output_types,
  84       name=name)

  88   return ged_ops.experimental_function_buffering_resource_reset(
  89       function_buffer_resource=function_buffer_resource, name=name)

  93 class _PrefetchToDeviceIterator(object):
  94   """A replacement for `tf.data.Iterator` that prefetches to another device.
  95 
  96   Args:
  97     input_dataset: The input dataset
  98     one_shot: If true, we make a one shot iterator that's already initialized.
  99     device: A fully specified device string where we want to prefetch to
 100     buffer_size: Size of the prefetching buffer.
 101     shared_name: (Optional.) If non-empty, the returned iterator will be
 102         shared under the given name across multiple sessions that share the
 103         same devices (e.g. when using a remote server).
 104 
 105   Returns:
 106     An Iterator type object.
 107   """
 108 
 109   def __init__(self,
 110                input_dataset,
 111                one_shot,
 112                device,
 113                buffer_size,
 114                shared_name=None):
 115     self._input_dataset = input_dataset
 116     self._get_next_call_count = 0
 117     self._one_shot = one_shot
 118     if shared_name is None:
 119       shared_name = ""
 120 
 121     if self._one_shot:
 122       self._input_iterator = input_dataset.make_one_shot_iterator()
 123     else:
 124       self._input_iterator = iterator_ops.Iterator.from_structure(
 125           self._input_dataset.output_types, self._input_dataset.output_shapes,
 126           shared_name, self._input_dataset.output_classes)
 127     input_iterator_handle = self._input_iterator.string_handle()
 128 
 129     @function.Defun(dtypes.string)
 130     def _prefetch_fn(handle):
 131       """Prefetches one element from `input_iterator`."""
 132       remote_iterator = iterator_ops.Iterator.from_string_handle(
 133           handle, self._input_iterator.output_types,
 134           self._input_iterator.output_shapes,
 135           self._input_iterator.output_classes)
 136       ret = remote_iterator.get_next()
 137       return nest.flatten(sparse.serialize_sparse_tensors(ret))
 138 
 139     iterator_device = ged_ops.experimental_iterator_get_device(
 140         self._input_iterator._iterator_resource)
 141 
 142     with ops.device(device):
 143       self._buffering_resource = function_buffering_resource(
 144           f=_prefetch_fn,
 145           target_device=iterator_device,
 146           string_arg=input_iterator_handle,
 147           buffer_size=buffer_size,
 148           shared_name=shared_name,
 149           output_types=nest.flatten(
 150               sparse.as_dense_types(self._input_dataset.output_types,
 151                                     self._input_dataset.output_classes)))
 152 
 153     if not self._one_shot:
 154       reset_op = function_buffering_resource_reset(self._buffering_resource)
 155       with ops.control_dependencies([reset_op]):
 156         self._initializer = self._input_iterator.make_initializer(
 157             self._input_dataset)
 158 
 159   def get_next(self, name=None):
 160     """See `tf.data.Iterator.get_next`."""
 161     self._get_next_call_count += 1
 162     if self._get_next_call_count > iterator_ops.GET_NEXT_CALL_WARNING_THRESHOLD:
 163       warnings.warn(iterator_ops.GET_NEXT_CALL_WARNING_MESSAGE)
 164 
 165     flat_ret = ged_ops.experimental_function_buffering_resource_get_next(
 166         self._buffering_resource,
 167         output_types=nest.flatten(
 168             sparse.as_dense_types(self.output_types, self.output_classes)),
 169         name=name)
 170 
 171     ret = sparse.deserialize_sparse_tensors(
 172         nest.pack_sequence_as(self.output_types, flat_ret),
 173         self.output_types, self.output_shapes, self.output_classes)
 174 
 175     for tensor, shape in zip(
 176         nest.flatten(ret), nest.flatten(self.output_shapes)):
 177       if isinstance(tensor, ops.Tensor):
 178         tensor.set_shape(shape)
 179 
 180     return ret
 181 
 182   @property
 183   def initializer(self):
 184     if self._one_shot:
 185       raise NotImplementedError("Can't initialize a one_shot_iterator")
 186     return self._initializer
 187 
 188   @property
 189   def output_classes(self):
 190     return self._input_dataset.output_classes
 191 
 192   @property
 193   def output_shapes(self):
 194     return self._input_dataset.output_shapes
 195 
 196   @property
 197   def output_types(self):
 198     return self._input_dataset.output_types

 201 class _PrefetchToDeviceEagerIterator(iterator_ops.EagerIterator):
 202   """A replacement for `tf.data.Iterator` that prefetches to another device.
 203 
 204   Args:
 205     input_dataset: The input dataset
 206     one_shot: If true, we make a one shot iterator that's already initialized.
 207     device: A fully specified device string where we want to prefetch to
 208     buffer_size: Size of the prefetching buffer.
 209     shared_name: (Optional.) If non-empty, the returned iterator will be
 210         shared under the given name across multiple sessions that share the
 211         same devices (e.g. when using a remote server).
 212 
 213   Returns:
 214     An Iterator type object.
 215   """
 216 
 217   def __init__(self,
 218                input_dataset,
 219                device,
 220                buffer_size):
 221     with ops.device("/device:CPU:0"):
 222       super(_PrefetchToDeviceEagerIterator, self).__init__(input_dataset)
 223       input_iterator_handle = gen_dataset_ops.iterator_to_string_handle(
 224           self._resource)
 225 
 226     self._device = device
 227 
 228     @function.Defun(dtypes.string)
 229     def _prefetch_fn(handle):
 230       """Prefetches one element from `input_iterator`."""
 231       remote_iterator = iterator_ops.Iterator.from_string_handle(
 232           handle, self.output_types, self.output_shapes, self.output_classes)
 233       ret = remote_iterator.get_next()
 234       return nest.flatten(sparse.serialize_sparse_tensors(ret))
 235 
 236     _prefetch_fn.add_to_graph(None)
 237 
 238     with ops.device(device):
 239       self._buffering_resource = function_buffering_resource(
 240           f=_prefetch_fn,
 241           output_types=self._flat_output_types,
 242           target_device=ged_ops.experimental_iterator_get_device(
 243               self._resource),
 244           string_arg=input_iterator_handle,
 245           buffer_size=buffer_size,
 246           shared_name=iterator_ops._generate_shared_name(
 247               "function_buffer_resource"))
 248 
 249   def _next_internal(self):
 250     """Returns a nested structure of `tf.Tensor`s containing the next element.
 251     """
 252     # This runs in sync mode as iterators use an error status to communicate
 253     # that there is no more data to iterate over.
 254     # TODO(b/77291417): Fix
 255     with context.execution_mode(context.SYNC):
 256       with ops.device(self._device):
 257         ret = ged_ops.experimental_function_buffering_resource_get_next(
 258             function_buffer_resource=self._buffering_resource,
 259             output_types=self._flat_output_types)
 260       return sparse.deserialize_sparse_tensors(
 261           nest.pack_sequence_as(self._output_types, ret), self._output_types,
 262           self._output_shapes, self._output_classes)
 263 # pylint: enable=protected-access

 266 class _PrefetchToDeviceDataset(dataset_ops.UnaryDataset):
 267   """A `Dataset` whose iterator prefetches elements to another device."""
 268 
 269   def __init__(self, input_dataset, device, buffer_size):
 270     super(_PrefetchToDeviceDataset, self).__init__(input_dataset)
 271     self._input_dataset = input_dataset
 272     self._device = device
 273     self._buffer_size = buffer_size if buffer_size is not None else 1
 274 
 275   # The static analysis cannot tell that the eager iterator's superclass has
 276   # a `next()` method.
 277   # pylint: disable=non-iterator-returned
 278   def __iter__(self):
 279     """Creates an `Iterator` for enumerating the elements of this dataset.
 280 
 281     The returned iterator implements the Python iterator protocol and therefore
 282     can only be used in eager mode.
 283 
 284     Returns:
 285       An `Iterator` over the elements of this dataset.
 286 
 287     Raises:
 288       RuntimeError: If eager execution is enabled.
 289     """
 290     if context.executing_eagerly():
 291       return _PrefetchToDeviceEagerIterator(self._input_dataset, self._device,
 292                                             self._buffer_size)
 293     else:
 294       raise RuntimeError("dataset.__iter__() is only supported when eager "
 295                          "execution is enabled.")
 296   # pylint: enable=non-iterator-returned
 297 
 298   def make_one_shot_iterator(self):
 299     if context.executing_eagerly():
 300       return _PrefetchToDeviceEagerIterator(self._input_dataset, self._device,
 301                                             self._buffer_size)
 302     else:
 303       return _PrefetchToDeviceIterator(self._input_dataset, one_shot=True,
 304                                        device=self._device,
 305                                        buffer_size=self._buffer_size)
 306 
 307   def make_initializable_iterator(self, shared_name=None):
 308     return _PrefetchToDeviceIterator(
 309         self._input_dataset,
 310         one_shot=False,
 311         device=self._device,
 312         buffer_size=self._buffer_size,
 313         shared_name=shared_name)
 314 
 315   def _as_variant_tensor(self):
 316     # TODO(mrry): Raise this error earlier (e.g. when one of the Dataset
 317     # transformation methods is called.
 318     # TODO(mrry): Investigate support for chaining further transformations after
 319     # the prefetch, including GPU support.
 320     raise NotImplementedError("`prefetch_to_device()` must be the last "
 321                               "transformation in a dataset pipeline.")
 322 
 323   @property
 324   def output_types(self):
 325     return self._input_dataset.output_types
 326 
 327   @property
 328   def output_shapes(self):
 329     return self._input_dataset.output_shapes
 330 
 331   @property
 332   def output_classes(self):
 333     return self._input_dataset.output_classes

 336 @tf_export("data.experimental.prefetch_to_device")
 337 def prefetch_to_device(device, buffer_size=None):
 338   """A transformation that prefetches dataset values to the given `device`.
 339 
 340   NOTE: Although the transformation creates a `tf.data.Dataset`, the
 341   transformation must be the final `Dataset` in the input pipeline.
 342 
 343   Args:
 344     device: A string. The name of a device to which elements will be prefetched.
 345     buffer_size: (Optional.) The number of elements to buffer on `device`.
 346       Defaults to an automatically chosen value.
 347 
 348   Returns:
 349     A `Dataset` transformation function, which can be passed to
 350     `tf.data.Dataset.apply`.
 351   """
 352   def _apply_fn(dataset):
 353     return _PrefetchToDeviceDataset(dataset, device, buffer_size)
 354 
 355   return _apply_fn

 358 @tf_export("data.experimental.copy_to_device")
 359 def copy_to_device(target_device, source_device="/cpu:0"):
 360   """A transformation that copies dataset elements to the given `target_device`.
 361 
 362   Args:
 363     target_device: The name of a device to which elements will be copied.
 364     source_device: The original device on which `input_dataset` will be placed.
 365 
 366   Returns:
 367     A `Dataset` transformation function, which can be passed to
 368     `tf.data.Dataset.apply`.
 369   """
 370 
 371   def _apply_fn(dataset):
 372     return _CopyToDeviceDataset(
 373         dataset, target_device=target_device, source_device=source_device)
 374 
 375   return _apply_fn

 378 # TODO(rohanj): Use the _input_hostmem attr on the RemoteCall ops to indicate
 379 # all inputs to the Op are in host memory, thereby avoiding some unnecessary
 380 # Sends and Recvs.
 381 class _CopyToDeviceDataset(dataset_ops.UnaryDataset):
 382   """A `Dataset` that copies elements to another device."""
 383 
 384   def __init__(self, input_dataset, target_device, source_device="/cpu:0"):
 385     """Constructs a _CopyToDeviceDataset.
 386 
 387     Args:
 388       input_dataset: `Dataset` to be copied
 389       target_device: The name of the device to which elements would be copied.
 390       source_device: Device where input_dataset would be placed.
 391     """
 392     super(_CopyToDeviceDataset, self).__init__(input_dataset)
 393     self._input_dataset = input_dataset
 394     self._target_device = target_device
 395     spec = framework_device.DeviceSpec().from_string(self._target_device)
 396     self._is_gpu_target = (spec.device_type == "GPU")
 397     self._source_device_string = source_device
 398     self._source_device = ops.convert_to_tensor(source_device)
 399 
 400     self._flat_output_shapes = nest.flatten(
 401         sparse.as_dense_shapes(self._input_dataset.output_shapes,
 402                                self._input_dataset.output_classes))
 403     self._flat_output_types = nest.flatten(
 404         sparse.as_dense_types(self._input_dataset.output_types,
 405                               self._input_dataset.output_classes))
 406 
 407     @function.Defun()
 408     def _init_func():
 409       """Creates an iterator for the input dataset.
 410 
 411       Returns:
 412         A `string` tensor that encapsulates the iterator created.
 413       """
 414       # pylint: disable=protected-access
 415       ds_variant = self._input_dataset._as_variant_tensor()
 416       resource = gen_dataset_ops.anonymous_iterator(
 417           output_types=self._flat_output_types,
 418           output_shapes=self._flat_output_shapes)
 419       with ops.control_dependencies(
 420           [gen_dataset_ops.make_iterator(ds_variant, resource)]):
 421         return gen_dataset_ops.iterator_to_string_handle(resource)
 422 
 423     @function.Defun()
 424     def _remote_init_func():
 425       return functional_ops.remote_call(
 426           target=self._source_device,
 427           args=_init_func.captured_inputs,
 428           Tout=[dtypes.string],
 429           f=_init_func)
 430 
 431     self._init_func = _remote_init_func
 432     self._init_captured_args = _remote_init_func.captured_inputs
 433 
 434     @function.Defun(dtypes.string)
 435     def _next_func(string_handle):
 436       """Calls get_next for created iterator.
 437 
 438       Args:
 439         string_handle: An iterator string handle created by _init_func
 440       Returns:
 441         The elements generated from `input_dataset`
 442       """
 443       with ops.device(self._source_device_string):
 444         iterator = iterator_ops.Iterator.from_string_handle(
 445             string_handle, self.output_types, self.output_shapes,
 446             self.output_classes)
 447       ret = iterator.get_next()
 448       return nest.flatten(sparse.serialize_sparse_tensors(ret))
 449 
 450     @function.Defun(dtypes.string)
 451     def _remote_next_func(string_handle):
 452       return functional_ops.remote_call(
 453           target=self._source_device,
 454           args=[string_handle] + _next_func.captured_inputs,
 455           Tout=self._flat_output_types,
 456           f=_next_func)
 457 
 458     self._next_func = _remote_next_func
 459     self._next_captured_args = _remote_next_func.captured_inputs
 460 
 461     @function.Defun(dtypes.string)
 462     def _finalize_func(string_handle):
 463       """Destroys the iterator resource created.
 464 
 465       Args:
 466         string_handle: An iterator string handle created by _init_func
 467       Returns:
 468         Tensor constant 0
 469       """
 470       iterator_resource = gen_dataset_ops.iterator_from_string_handle_v2(
 471           string_handle,
 472           output_types=self._flat_output_types,
 473           output_shapes=self._flat_output_shapes)
 474       with ops.control_dependencies([
 475           resource_variable_ops.destroy_resource_op(
 476               iterator_resource, ignore_lookup_error=True)]):
 477         return array_ops.constant(0, dtypes.int64)
 478 
 479     @function.Defun(dtypes.string)
 480     def _remote_finalize_func(string_handle):
 481       return functional_ops.remote_call(
 482           target=self._source_device,
 483           args=[string_handle] + _finalize_func.captured_inputs,
 484           Tout=[dtypes.int64],
 485           f=_finalize_func)
 486 
 487     self._finalize_func = _remote_finalize_func
 488     self._finalize_captured_args = _remote_finalize_func.captured_inputs
 489 
 490     g = ops.get_default_graph()
 491     _remote_init_func.add_to_graph(g)
 492     _remote_next_func.add_to_graph(g)
 493     _remote_finalize_func.add_to_graph(g)
 494     # pylint: enable=protected-scope
 495 
 496   # The one_shot_iterator implementation needs a 0 arg _make_dataset function
 497   # that thereby captures all the inputs required to create the dataset. Since
 498   # there are strings that are inputs to the GeneratorDataset which can't be
 499   # placed on a GPU, this fails for the GPU case. Therefore, disabling it for
 500   # GPU
 501   def make_one_shot_iterator(self):
 502     if self._is_gpu_target:
 503       raise ValueError("Cannot create a one shot iterator when using "
 504                        "`tf.data.experimental.copy_to_device()` on GPU. Please "
 505                        "use `Dataset.make_initializable_iterator()` instead.")
 506     else:
 507       return super(_CopyToDeviceDataset, self).make_one_shot_iterator()
 508 
 509   def _as_variant_tensor(self):
 510     with ops.device(self._target_device):
 511       return gen_dataset_ops.generator_dataset(
 512           self._init_captured_args,
 513           self._next_captured_args,
 514           self._finalize_captured_args,
 515           init_func=self._init_func,
 516           next_func=self._next_func,
 517           finalize_func=self._finalize_func,
 518           output_types=self._flat_output_types,
 519           output_shapes=self._flat_output_shapes)
 520 
 521   @property
 522   def output_types(self):
 523     return self._input_dataset.output_types
 524 
 525   @property
 526   def output_shapes(self):
 527     return self._input_dataset.output_shapes
 528 
 529   @property
 530   def output_classes(self):
 531     return self._input_dataset.output_classes
