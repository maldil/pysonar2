  31 def assert_next(transformations):
  32   """A transformation that asserts which transformations happen next.
  33 
  34   Args:
  35     transformations: A `tf.string` vector `tf.Tensor` identifying the
  36       transformations that are expected to happen next.
  37 
  38   Returns:
  39     A `Dataset` transformation function, which can be passed to
  40     `tf.data.Dataset.apply`.
  41   """
  42 
  43   def _apply_fn(dataset):
  44     """Function from `Dataset` to `Dataset` that applies the transformation."""
  45     return _AssertNextDataset(dataset, transformations)
  46 
  47   return _apply_fn

  65 def optimize(optimizations=None):
  66   """A transformation that applies optimizations.
  67 
  68   Args:
  69     optimizations: (Optional.) A `tf.string` vector `tf.Tensor` identifying
  70       optimizations to use. If not specified, the default set of optimizations
  71       is applied.
  72 
  73   Returns:
  74     A `Dataset` transformation function, which can be passed to
  75     `tf.data.Dataset.apply`.
  76   """
  77 
  78   def _apply_fn(dataset):
  79     """Function from `Dataset` to `Dataset` that applies the transformation."""
  80     return dataset_ops._OptimizeDataset(dataset, optimizations)  # pylint: disable=protected-access
  81 
  82   return _apply_fn

  85 class _AssertNextDataset(dataset_ops.UnaryDataset):
  86   """A `Dataset` that asserts which transformations happen next."""
  87 
  88   def __init__(self, input_dataset, transformations):
  89     """See `assert_next()` for details."""
  90     super(_AssertNextDataset, self).__init__(input_dataset)
  91     self._input_dataset = input_dataset
  92     if transformations is None:
  93       raise ValueError("At least one transformation should be specified")
  94     self._transformations = ops.convert_to_tensor(
  95         transformations, dtype=dtypes.string, name="transformations")
  96 
  97   def _as_variant_tensor(self):
  98     return gen_experimental_dataset_ops.experimental_assert_next_dataset(
  99         self._input_dataset._as_variant_tensor(),  # pylint: disable=protected-access
 100         self._transformations,
 101         **dataset_ops.flat_structure(self))
 102 
 103   @property
 104   def output_classes(self):
 105     return self._input_dataset.output_classes
 106 
 107   @property
 108   def output_shapes(self):
 109     return self._input_dataset.output_shapes
 110 
 111   @property
 112   def output_types(self):
 113     return self._input_dataset.output_types
