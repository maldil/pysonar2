  31 class _ScanDataset(dataset_ops.UnaryDataset):
  32   """A dataset that scans a function across its input."""
  33 
  34   def __init__(self, input_dataset, initial_state, scan_func):
  35     """See `scan()` for details."""
  36     super(_ScanDataset, self).__init__(input_dataset)
  37     self._input_dataset = input_dataset
  38 
  39     with ops.name_scope("initial_state"):
  40       # Convert any `SparseTensorValue`s to `SparseTensor`s and all other
  41       # values to tensors.
  42       self._initial_state = nest.pack_sequence_as(initial_state, [
  43           sparse_tensor.SparseTensor.from_value(t)
  44           if sparse_tensor.is_sparse(t) else ops.convert_to_tensor(
  45               t, name="component_%d" % i)
  46           for i, t in enumerate(nest.flatten(initial_state))
  47       ])
  48 
  49     # Compute initial values for the state classes, shapes and types based on
  50     # the initial state. The shapes may be refined by running `tf_scan_func` one
  51     # or more times below.
  52     self._state_classes = sparse.get_classes(self._initial_state)
  53     self._state_shapes = nest.pack_sequence_as(
  54         self._initial_state,
  55         [t.get_shape() for t in nest.flatten(self._initial_state)])
  56     self._state_types = nest.pack_sequence_as(
  57         self._initial_state,
  58         [t.dtype for t in nest.flatten(self._initial_state)])
  59 
  60     # Will be populated by calling `tf_scan_func`.
  61     self._output_classes = None
  62     self._output_shapes = None
  63     self._output_types = None
  64 
  65     # Iteratively rerun the scan function until reaching a fixed point on
  66     # `self._state_shapes`.
  67     need_to_rerun = True
  68     while need_to_rerun:
  69 
  70       wrapped_func = dataset_ops.StructuredFunctionWrapper(
  71           scan_func,
  72           "tf.data.experimental.scan()",
  73           input_classes=(self._state_classes, input_dataset.output_classes),
  74           input_shapes=(self._state_shapes, input_dataset.output_shapes),
  75           input_types=(self._state_types, input_dataset.output_types),
  76           add_to_graph=False)
  77       if not (
  78           isinstance(wrapped_func.output_types, collections.Sequence) and
  79           len(wrapped_func.output_types) == 2):
  80         raise TypeError("The scan function must return a pair comprising the "
  81                         "new state and the output value.")
  82 
  83       new_state_classes, self._output_classes = wrapped_func.output_classes
  84 
  85       # Extract and validate class information from the returned values.
  86       for new_state_class, state_class in zip(
  87           nest.flatten(new_state_classes),
  88           nest.flatten(self._state_classes)):
  89         if not issubclass(new_state_class, state_class):
  90           raise TypeError(
  91               "The element classes for the new state must match the initial "
  92               "state. Expected %s; got %s." %
  93               (self._state_classes, new_state_classes))
  94 
  95       # Extract and validate type information from the returned values.
  96       new_state_types, self._output_types = wrapped_func.output_types
  97       for new_state_type, state_type in zip(
  98           nest.flatten(new_state_types), nest.flatten(self._state_types)):
  99         if new_state_type != state_type:
 100           raise TypeError(
 101               "The element types for the new state must match the initial "
 102               "state. Expected %s; got %s." %
 103               (self._state_types, new_state_types))
 104 
 105       # Extract shape information from the returned values.
 106       new_state_shapes, self._output_shapes = wrapped_func.output_shapes
 107 
 108       flat_state_shapes = nest.flatten(self._state_shapes)
 109       flat_new_state_shapes = nest.flatten(new_state_shapes)
 110       weakened_state_shapes = [
 111           original.most_specific_compatible_shape(new)
 112           for original, new in zip(flat_state_shapes, flat_new_state_shapes)
 113       ]
 114 
 115       need_to_rerun = False
 116       for original_shape, weakened_shape in zip(flat_state_shapes,
 117                                                 weakened_state_shapes):
 118         if original_shape.ndims is not None and (
 119             weakened_shape.ndims is None or
 120             original_shape.as_list() != weakened_shape.as_list()):
 121           need_to_rerun = True
 122           break
 123 
 124       if need_to_rerun:
 125         self._state_shapes = nest.pack_sequence_as(self._state_shapes,
 126                                                    weakened_state_shapes)
 127 
 128     self._scan_func = wrapped_func.function
 129     self._scan_func.add_to_graph(ops.get_default_graph())
 130 
 131   def _as_variant_tensor(self):
 132     input_t = self._input_dataset._as_variant_tensor()  # pylint: disable=protected-access
 133     return gen_dataset_ops.scan_dataset(
 134         input_t,
 135         nest.flatten(sparse.serialize_sparse_tensors(self._initial_state)),
 136         self._scan_func.captured_inputs,
 137         f=self._scan_func,
 138         **dataset_ops.flat_structure(self))
 139 
 140   @property
 141   def output_classes(self):
 142     return self._output_classes
 143 
 144   @property
 145   def output_shapes(self):
 146     return self._output_shapes
 147 
 148   @property
 149   def output_types(self):
 150     return self._output_types

 153 @tf_export("data.experimental.scan")
 154 def scan(initial_state, scan_func):
 155   """A transformation that scans a function across an input dataset.
 156 
 157   This transformation is a stateful relative of `tf.data.Dataset.map`.
 158   In addition to mapping `scan_func` across the elements of the input dataset,
 159   `scan()` accumulates one or more state tensors, whose initial values are
 160   `initial_state`.
 161 
 162   Args:
 163     initial_state: A nested structure of tensors, representing the initial state
 164       of the accumulator.
 165     scan_func: A function that maps `(old_state, input_element)` to
 166       `(new_state, output_element). It must take two arguments and return a
 167       pair of nested structures of tensors. The `new_state` must match the
 168       structure of `initial_state`.
 169 
 170   Returns:
 171     A `Dataset` transformation function, which can be passed to
 172     `tf.data.Dataset.apply`.
 173   """
 174   def _apply_fn(dataset):
 175     return _ScanDataset(dataset, initial_state, scan_func)
 176 
 177   return _apply_fn
