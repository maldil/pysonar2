  35 @tf_export("data.experimental.group_by_reducer")
  36 def group_by_reducer(key_func, reducer):
  37   """A transformation that groups elements and performs a reduction.
  38 
  39   This transformation maps element of a dataset to a key using `key_func` and
  40   groups the elements by key. The `reducer` is used to process each group; its
  41   `init_func` is used to initialize state for each group when it is created, the
  42   `reduce_func` is used to update the state every time an element is mapped to
  43   the matching group, and the `finalize_func` is used to map the final state to
  44   an output value.
  45 
  46   Args:
  47     key_func: A function mapping a nested structure of tensors
  48       (having shapes and types defined by `self.output_shapes` and
  49       `self.output_types`) to a scalar `tf.int64` tensor.
  50     reducer: An instance of `Reducer`, which captures the reduction logic using
  51       the `init_func`, `reduce_func`, and `finalize_func` functions.
  52 
  53   Returns:
  54     A `Dataset` transformation function, which can be passed to
  55     `tf.data.Dataset.apply`.
  56   """
  57 
  58   def _apply_fn(dataset):
  59     """Function from `Dataset` to `Dataset` that applies the transformation."""
  60     return _GroupByReducerDataset(dataset, key_func, reducer)
  61 
  62   return _apply_fn

  65 @tf_export("data.experimental.group_by_window")
  66 def group_by_window(key_func,
  67                     reduce_func,
  68                     window_size=None,
  69                     window_size_func=None):
  70   """A transformation that groups windows of elements by key and reduces them.
  71 
  72   This transformation maps each consecutive element in a dataset to a key
  73   using `key_func` and groups the elements by key. It then applies
  74   `reduce_func` to at most `window_size_func(key)` elements matching the same
  75   key. All except the final window for each key will contain
  76   `window_size_func(key)` elements; the final window may be smaller.
  77 
  78   You may provide either a constant `window_size` or a window size determined by
  79   the key through `window_size_func`.
  80 
  81   Args:
  82     key_func: A function mapping a nested structure of tensors
  83       (having shapes and types defined by `self.output_shapes` and
  84       `self.output_types`) to a scalar `tf.int64` tensor.
  85     reduce_func: A function mapping a key and a dataset of up to `window_size`
  86       consecutive elements matching that key to another dataset.
  87     window_size: A `tf.int64` scalar `tf.Tensor`, representing the number of
  88       consecutive elements matching the same key to combine in a single
  89       batch, which will be passed to `reduce_func`. Mutually exclusive with
  90       `window_size_func`.
  91     window_size_func: A function mapping a key to a `tf.int64` scalar
  92       `tf.Tensor`, representing the number of consecutive elements matching
  93       the same key to combine in a single batch, which will be passed to
  94       `reduce_func`. Mutually exclusive with `window_size`.
  95 
  96   Returns:
  97     A `Dataset` transformation function, which can be passed to
  98     `tf.data.Dataset.apply`.
  99 
 100   Raises:
 101     ValueError: if neither or both of {`window_size`, `window_size_func`} are
 102       passed.
 103   """
 104   if (window_size is not None and window_size_func or
 105       not (window_size is not None or window_size_func)):
 106     raise ValueError("Must pass either window_size or window_size_func.")
 107 
 108   if window_size is not None:
 109 
 110     def constant_window_func(unused_key):
 111       return ops.convert_to_tensor(window_size, dtype=dtypes.int64)
 112 
 113     window_size_func = constant_window_func
 114 
 115   assert window_size_func is not None
 116 
 117   def _apply_fn(dataset):
 118     """Function from `Dataset` to `Dataset` that applies the transformation."""
 119     return _GroupByWindowDataset(dataset, key_func, reduce_func,
 120                                  window_size_func)
 121 
 122   return _apply_fn

 125 @tf_export("data.experimental.bucket_by_sequence_length")
 126 def bucket_by_sequence_length(element_length_func,
 127                               bucket_boundaries,
 128                               bucket_batch_sizes,
 129                               padded_shapes=None,
 130                               padding_values=None,
 131                               pad_to_bucket_boundary=False,
 132                               no_padding=False):
 133   """A transformation that buckets elements in a `Dataset` by length.
 134 
 135   Elements of the `Dataset` are grouped together by length and then are padded
 136   and batched.
 137 
 138   This is useful for sequence tasks in which the elements have variable length.
 139   Grouping together elements that have similar lengths reduces the total
 140   fraction of padding in a batch which increases training step efficiency.
 141 
 142   Args:
 143     element_length_func: function from element in `Dataset` to `tf.int32`,
 144       determines the length of the element, which will determine the bucket it
 145       goes into.
 146     bucket_boundaries: `list<int>`, upper length boundaries of the buckets.
 147     bucket_batch_sizes: `list<int>`, batch size per bucket. Length should be
 148       `len(bucket_boundaries) + 1`.
 149     padded_shapes: Nested structure of `tf.TensorShape` to pass to
 150       `tf.data.Dataset.padded_batch`. If not provided, will use
 151       `dataset.output_shapes`, which will result in variable length dimensions
 152       being padded out to the maximum length in each batch.
 153     padding_values: Values to pad with, passed to
 154       `tf.data.Dataset.padded_batch`. Defaults to padding with 0.
 155     pad_to_bucket_boundary: bool, if `False`, will pad dimensions with unknown
 156       size to maximum length in batch. If `True`, will pad dimensions with
 157       unknown size to bucket boundary minus 1 (i.e., the maximum length in each
 158       bucket), and caller must ensure that the source `Dataset` does not contain
 159       any elements with length longer than `max(bucket_boundaries)`.
 160     no_padding: `bool`, indicates whether to pad the batch features (features
 161       need to be either of type `tf.SparseTensor` or of same shape).
 162 
 163   Returns:
 164     A `Dataset` transformation function, which can be passed to
 165     `tf.data.Dataset.apply`.
 166 
 167   Raises:
 168     ValueError: if `len(bucket_batch_sizes) != len(bucket_boundaries) + 1`.
 169   """
 170   with ops.name_scope("bucket_by_seq_length"):
 171     if len(bucket_batch_sizes) != (len(bucket_boundaries) + 1):
 172       raise ValueError(
 173           "len(bucket_batch_sizes) must equal len(bucket_boundaries) + 1")
 174 
 175     batch_sizes = constant_op.constant(bucket_batch_sizes, dtype=dtypes.int64)
 176 
 177     def element_to_bucket_id(*args):
 178       """Return int64 id of the length bucket for this element."""
 179       seq_length = element_length_func(*args)
 180 
 181       boundaries = list(bucket_boundaries)
 182       buckets_min = [np.iinfo(np.int32).min] + boundaries
 183       buckets_max = boundaries + [np.iinfo(np.int32).max]
 184       conditions_c = math_ops.logical_and(
 185           math_ops.less_equal(buckets_min, seq_length),
 186           math_ops.less(seq_length, buckets_max))
 187       bucket_id = math_ops.reduce_min(array_ops.where(conditions_c))
 188 
 189       return bucket_id
 190 
 191     def window_size_fn(bucket_id):
 192       # The window size is set to the batch size for this bucket
 193       window_size = batch_sizes[bucket_id]
 194       return window_size
 195 
 196     def make_padded_shapes(shapes, none_filler=None):
 197       padded = []
 198       for shape in nest.flatten(shapes):
 199         shape = tensor_shape.TensorShape(shape)
 200         shape = [
 201             none_filler if d.value is None else d
 202             for d in shape
 203         ]
 204         padded.append(shape)
 205       return nest.pack_sequence_as(shapes, padded)
 206 
 207     def batching_fn(bucket_id, grouped_dataset):
 208       """Batch elements in dataset."""
 209       batch_size = window_size_fn(bucket_id)
 210       if no_padding:
 211         return grouped_dataset.batch(batch_size)
 212       none_filler = None
 213       if pad_to_bucket_boundary:
 214         err_msg = ("When pad_to_bucket_boundary=True, elements must have "
 215                    "length < max(bucket_boundaries).")
 216         check = check_ops.assert_less(
 217             bucket_id,
 218             constant_op.constant(len(bucket_batch_sizes) - 1,
 219                                  dtype=dtypes.int64),
 220             message=err_msg)
 221         with ops.control_dependencies([check]):
 222           boundaries = constant_op.constant(bucket_boundaries,
 223                                             dtype=dtypes.int64)
 224           bucket_boundary = boundaries[bucket_id]
 225           none_filler = bucket_boundary - 1
 226       shapes = make_padded_shapes(
 227           padded_shapes or grouped_dataset.output_shapes,
 228           none_filler=none_filler)
 229       return grouped_dataset.padded_batch(batch_size, shapes, padding_values)
 230 
 231     def _apply_fn(dataset):
 232       return dataset.apply(
 233           group_by_window(element_to_bucket_id, batching_fn,
 234                           window_size_func=window_size_fn))
 235 
 236     return _apply_fn

 262 class _GroupByReducerDataset(dataset_ops.UnaryDataset):
 263   """A `Dataset` that groups its input and performs a reduction."""
 264 
 265   def __init__(self, input_dataset, key_func, reducer):
 266     """See `group_by_reducer()` for details."""
 267     super(_GroupByReducerDataset, self).__init__(input_dataset)
 268 
 269     self._input_dataset = input_dataset
 270 
 271     self._make_key_func(key_func, input_dataset)
 272     self._make_init_func(reducer.init_func)
 273     self._make_reduce_func(reducer.reduce_func, input_dataset)
 274     self._make_finalize_func(reducer.finalize_func)
 275 
 276   def _make_key_func(self, key_func, input_dataset):
 277     """Make wrapping Defun for key_func."""
 278     wrapped_func = dataset_ops.StructuredFunctionWrapper(
 279         key_func, "tf.data.experimental.group_by_reducer()", input_dataset)
 280     if not (
 281         wrapped_func.output_types == dtypes.int64 and
 282         wrapped_func.output_shapes.is_compatible_with(tensor_shape.scalar())):
 283       raise ValueError(
 284           "`key_func` must return a single tf.int64 tensor. "
 285           "Got type=%s and shape=%s"
 286           % (wrapped_func.output_types, wrapped_func.output_shapes))
 287     self._key_func = wrapped_func.function
 288 
 289   def _make_init_func(self, init_func):
 290     """Make wrapping Defun for init_func."""
 291     wrapped_func = dataset_ops.StructuredFunctionWrapper(
 292         init_func,
 293         "tf.data.experimental.group_by_reducer()",
 294         input_classes=ops.Tensor,
 295         input_shapes=tensor_shape.scalar(),
 296         input_types=dtypes.int64)
 297     self._init_func = wrapped_func.function
 298     self._state_classes = wrapped_func.output_classes
 299     self._state_shapes = wrapped_func.output_shapes
 300     self._state_types = wrapped_func.output_types
 301 
 302   def _make_reduce_func(self, reduce_func, input_dataset):
 303     """Make wrapping Defun for reduce_func."""
 304 
 305     # Iteratively rerun the reduce function until reaching a fixed point on
 306     # `self._state_shapes`.
 307     need_to_rerun = True
 308     while need_to_rerun:
 309 
 310       wrapped_func = dataset_ops.StructuredFunctionWrapper(
 311           reduce_func,
 312           "tf.data.experimental.group_by_reducer()",
 313           input_classes=(self._state_classes, input_dataset.output_classes),
 314           input_shapes=(self._state_shapes, input_dataset.output_shapes),
 315           input_types=(self._state_types, input_dataset.output_types),
 316           add_to_graph=False)
 317 
 318       # Extract and validate class information from the returned values.
 319       for new_state_class, state_class in zip(
 320           nest.flatten(wrapped_func.output_classes),
 321           nest.flatten(self._state_classes)):
 322         if not issubclass(new_state_class, state_class):
 323           raise TypeError(
 324               "The element classes for the new state must match the initial "
 325               "state. Expected %s; got %s." %
 326               (self._state_classes, wrapped_func.output_classes))
 327 
 328       # Extract and validate type information from the returned values.
 329       for new_state_type, state_type in zip(
 330           nest.flatten(wrapped_func.output_types),
 331           nest.flatten(self._state_types)):
 332         if new_state_type != state_type:
 333           raise TypeError(
 334               "The element types for the new state must match the initial "
 335               "state. Expected %s; got %s." %
 336               (self._state_types, wrapped_func.output_types))
 337 
 338       # Extract shape information from the returned values.
 339       flat_state_shapes = nest.flatten(self._state_shapes)
 340       flat_new_state_shapes = nest.flatten(wrapped_func.output_shapes)
 341       weakened_state_shapes = [
 342           original.most_specific_compatible_shape(new)
 343           for original, new in zip(flat_state_shapes, flat_new_state_shapes)
 344       ]
 345 
 346       need_to_rerun = False
 347       for original_shape, weakened_shape in zip(flat_state_shapes,
 348                                                 weakened_state_shapes):
 349         if original_shape.ndims is not None and (
 350             weakened_shape.ndims is None or
 351             original_shape.as_list() != weakened_shape.as_list()):
 352           need_to_rerun = True
 353           break
 354 
 355       if need_to_rerun:
 356         self._state_shapes = nest.pack_sequence_as(self._state_shapes,
 357                                                    weakened_state_shapes)
 358 
 359     self._reduce_func = wrapped_func.function
 360     self._reduce_func.add_to_graph(ops.get_default_graph())
 361 
 362   def _make_finalize_func(self, finalize_func):
 363     """Make wrapping Defun for finalize_func."""
 364     wrapped_func = dataset_ops.StructuredFunctionWrapper(
 365         finalize_func,
 366         "tf.data.experimental.group_by_reducer()",
 367         input_classes=self._state_classes,
 368         input_shapes=self._state_shapes,
 369         input_types=self._state_types)
 370     self._finalize_func = wrapped_func.function
 371     self._output_classes = wrapped_func.output_classes
 372     self._output_shapes = wrapped_func.output_shapes
 373     self._output_types = wrapped_func.output_types
 374 
 375   @property
 376   def output_classes(self):
 377     return self._output_classes
 378 
 379   @property
 380   def output_shapes(self):
 381     return self._output_shapes
 382 
 383   @property
 384   def output_types(self):
 385     return self._output_types
 386 
 387   def _as_variant_tensor(self):
 388     return gen_dataset_ops.group_by_reducer_dataset(
 389         self._input_dataset._as_variant_tensor(),  # pylint: disable=protected-access
 390         self._key_func.captured_inputs,
 391         self._init_func.captured_inputs,
 392         self._reduce_func.captured_inputs,
 393         self._finalize_func.captured_inputs,
 394         key_func=self._key_func,
 395         init_func=self._init_func,
 396         reduce_func=self._reduce_func,
 397         finalize_func=self._finalize_func,
 398         **dataset_ops.flat_structure(self))

 401 class _GroupByWindowDataset(dataset_ops.UnaryDataset):
 402   """A `Dataset` that groups its input and performs a windowed reduction."""
 403 
 404   def __init__(self, input_dataset, key_func, reduce_func, window_size_func):
 405     """See `group_by_window()` for details."""
 406     super(_GroupByWindowDataset, self).__init__(input_dataset)
 407 
 408     self._input_dataset = input_dataset
 409 
 410     self._make_key_func(key_func, input_dataset)
 411     self._make_reduce_func(reduce_func, input_dataset)
 412     self._make_window_size_func(window_size_func)
 413 
 414   def _make_window_size_func(self, window_size_func):
 415     """Make wrapping Defun for window_size_func."""
 416     def window_size_func_wrapper(key):
 417       return ops.convert_to_tensor(window_size_func(key), dtype=dtypes.int64)
 418     wrapped_func = dataset_ops.StructuredFunctionWrapper(
 419         window_size_func_wrapper,
 420         "tf.data.experimental.group_by_window()",
 421         input_classes=ops.Tensor,
 422         input_shapes=tensor_shape.scalar(),
 423         input_types=dtypes.int64)
 424     if not (
 425         wrapped_func.output_types == dtypes.int64 and
 426         wrapped_func.output_shapes.is_compatible_with(tensor_shape.scalar())):
 427       raise ValueError(
 428           "`window_size_func` must return a single tf.int64 scalar tensor.")
 429     self._window_size_func = wrapped_func.function
 430 
 431   def _make_key_func(self, key_func, input_dataset):
 432     """Make wrapping Defun for key_func."""
 433     def key_func_wrapper(*args):
 434       return ops.convert_to_tensor(key_func(*args), dtype=dtypes.int64)
 435     wrapped_func = dataset_ops.StructuredFunctionWrapper(
 436         key_func_wrapper, "tf.data.experimental.group_by_window()",
 437         input_dataset)
 438     if not (
 439         wrapped_func.output_types == dtypes.int64 and
 440         wrapped_func.output_shapes.is_compatible_with(tensor_shape.scalar())):
 441       raise ValueError(
 442           "`key_func` must return a single tf.int64 scalar tensor.")
 443     self._key_func = wrapped_func.function
 444 
 445   def _make_reduce_func(self, reduce_func, input_dataset):
 446     """Make wrapping Defun for reduce_func."""
 447     nested_dataset = dataset_ops._NestedDatasetComponent(input_dataset)  # pylint: disable=protected-access
 448     wrapped_func = dataset_ops.StructuredFunctionWrapper(
 449         reduce_func,
 450         "tf.data.experimental.reduce_by_window()",
 451         input_classes=(ops.Tensor, nested_dataset),
 452         input_shapes=(tensor_shape.scalar(), nested_dataset),
 453         input_types=(dtypes.int64, nested_dataset),
 454         experimental_nested_dataset_support=True)
 455     if not isinstance(
 456         wrapped_func.output_classes, dataset_ops._NestedDatasetComponent):  # pylint: disable=protected-access
 457       raise TypeError("`reduce_func` must return a `Dataset` object.")
 458     self._output_classes = wrapped_func.output_classes.output_classes
 459     self._output_types = wrapped_func.output_types.output_types
 460     self._output_shapes = wrapped_func.output_shapes.output_shapes
 461     self._reduce_func = wrapped_func.function
 462 
 463   @property
 464   def output_classes(self):
 465     return self._output_classes
 466 
 467   @property
 468   def output_shapes(self):
 469     return self._output_shapes
 470 
 471   @property
 472   def output_types(self):
 473     return self._output_types
 474 
 475   def _as_variant_tensor(self):
 476     return gen_dataset_ops.group_by_window_dataset(
 477         self._input_dataset._as_variant_tensor(),  # pylint: disable=protected-access
 478         self._key_func.captured_inputs,
 479         self._reduce_func.captured_inputs,
 480         self._window_size_func.captured_inputs,
 481         key_func=self._key_func,
 482         reduce_func=self._reduce_func,
 483         window_size_func=self._window_size_func,
 484         **dataset_ops.flat_structure(self))

 498     self._init_func = init_func
 499     self._reduce_func = reduce_func

 501

 504     return self._init_func
 505 
 506   @property
 507   def reduce_func(self):
 508     return self._reduce_func
 509

 512     return self._finalize_func
