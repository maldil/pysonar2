  43 def batch_window(dataset):
  44   """Batches a window of tensors.
  45 
  46   Args:
  47     dataset: the input dataset.
  48 
  49   Returns:
  50     A `Tensor` representing the batch of the entire input dataset.
  51   """
  52   if isinstance(dataset.output_classes, tuple):
  53     raise TypeError("Input dataset expected to have a single component")
  54   if dataset.output_classes is ops.Tensor:
  55     return _batch_dense_window(dataset)
  56   elif dataset.output_classes is sparse_tensor.SparseTensor:
  57     return _batch_sparse_window(dataset)
  58   else:
  59     raise TypeError("Unsupported dataset type: %s" % dataset.output_classes)

  62 def _batch_dense_window(dataset):
  63   """Batches a window of dense tensors."""
  64 
  65   def key_fn(_):
  66     return np.int64(0)
  67 
  68   def shape_init_fn(_):
  69     return array_ops.shape(first_element)
  70 
  71   def shape_reduce_fn(state, value):
  72     check_ops.assert_equal(state, array_ops.shape(value))
  73     return state
  74 
  75   def finalize_fn(state):
  76     return state
  77 
  78   if dataset.output_shapes.is_fully_defined():
  79     shape = dataset.output_shapes
  80   else:
  81     first_element = get_single_element.get_single_element(dataset.take(1))
  82     shape_reducer = grouping.Reducer(shape_init_fn, shape_reduce_fn,
  83                                      finalize_fn)
  84     shape = get_single_element.get_single_element(
  85         dataset.apply(grouping.group_by_reducer(key_fn, shape_reducer)))
  86 
  87   def batch_init_fn(_):
  88     batch_shape = array_ops.concat([[0], shape], 0)
  89     return gen_array_ops.empty(batch_shape, dtype=dataset.output_types)
  90 
  91   def batch_reduce_fn(state, value):
  92     return array_ops.concat([state, [value]], 0)
  93 
  94   batch_reducer = grouping.Reducer(batch_init_fn, batch_reduce_fn, finalize_fn)
  95   return get_single_element.get_single_element(
  96       dataset.apply(grouping.group_by_reducer(key_fn, batch_reducer)))

  99 def _batch_sparse_window(dataset):
 100   """Batches a window of sparse tensors."""
 101 
 102   def key_fn(_):
 103     return np.int64(0)
 104 
 105   def shape_init_fn(_):
 106     return first_element.dense_shape
 107 
 108   def shape_reduce_fn(state, value):
 109     check_ops.assert_equal(state, value.dense_shape)
 110     return state
 111 
 112   def finalize_fn(state):
 113     return state
 114 
 115   if dataset.output_shapes.is_fully_defined():
 116     shape = dataset.output_shapes
 117   else:
 118     first_element = get_single_element.get_single_element(dataset.take(1))
 119     shape_reducer = grouping.Reducer(shape_init_fn, shape_reduce_fn,
 120                                      finalize_fn)
 121     shape = get_single_element.get_single_element(
 122         dataset.apply(grouping.group_by_reducer(key_fn, shape_reducer)))
 123 
 124   def batch_init_fn(_):
 125     indices_shape = array_ops.concat([[0], [array_ops.size(shape) + 1]], 0)
 126     return sparse_tensor.SparseTensor(
 127         indices=gen_array_ops.empty(indices_shape, dtype=dtypes.int64),
 128         values=constant_op.constant([], shape=[0], dtype=dataset.output_types),
 129         dense_shape=array_ops.concat(
 130             [np.array([0], dtype=np.int64),
 131              math_ops.cast(shape, dtypes.int64)], 0))
 132 
 133   def batch_reduce_fn(state, value):
 134     return sparse_ops.sparse_concat(0, [state, value])
 135 
 136   def reshape_fn(value):
 137     return sparse_ops.sparse_reshape(
 138         value,
 139         array_ops.concat([np.array([1], dtype=np.int64), value.dense_shape], 0))
 140 
 141   batch_reducer = grouping.Reducer(batch_init_fn, batch_reduce_fn, finalize_fn)
 142   return get_single_element.get_single_element(
 143       dataset.map(reshape_fn).apply(
 144           grouping.group_by_reducer(key_fn, batch_reducer)))

 147 @tf_export("data.experimental.dense_to_sparse_batch")
 148 def dense_to_sparse_batch(batch_size, row_shape):
 149   """A transformation that batches ragged elements into `tf.SparseTensor`s.
 150 
 151   Like `Dataset.padded_batch()`, this transformation combines multiple
 152   consecutive elements of the dataset, which might have different
 153   shapes, into a single element. The resulting element has three
 154   components (`indices`, `values`, and `dense_shape`), which
 155   comprise a `tf.SparseTensor` that represents the same data. The
 156   `row_shape` represents the dense shape of each row in the
 157   resulting `tf.SparseTensor`, to which the effective batch size is
 158   prepended. For example:
 159 
 160   ```python
 161   # NOTE: The following examples use `{ ... }` to represent the
 162   # contents of a dataset.
 163   a = { ['a', 'b', 'c'], ['a', 'b'], ['a', 'b', 'c', 'd'] }
 164 
 165   a.apply(tf.data.experimental.dense_to_sparse_batch(
 166       batch_size=2, row_shape=[6])) ==
 167   {
 168       ([[0, 0], [0, 1], [0, 2], [1, 0], [1, 1]],  # indices
 169        ['a', 'b', 'c', 'a', 'b'],                 # values
 170        [2, 6]),                                   # dense_shape
 171       ([[0, 0], [0, 1], [0, 2], [0, 3]],
 172        ['a', 'b', 'c', 'd'],
 173        [1, 6])
 174   }
 175   ```
 176 
 177   Args:
 178     batch_size: A `tf.int64` scalar `tf.Tensor`, representing the
 179       number of consecutive elements of this dataset to combine in a
 180       single batch.
 181     row_shape: A `tf.TensorShape` or `tf.int64` vector tensor-like
 182       object representing the equivalent dense shape of a row in the
 183       resulting `tf.SparseTensor`. Each element of this dataset must
 184       have the same rank as `row_shape`, and must have size less
 185       than or equal to `row_shape` in each dimension.
 186 
 187   Returns:
 188     A `Dataset` transformation function, which can be passed to
 189     `tf.data.Dataset.apply`.
 190   """
 191 
 192   def _apply_fn(dataset):
 193     return _DenseToSparseBatchDataset(dataset, batch_size, row_shape)
 194 
 195   return _apply_fn

 198 def padded_batch_window(dataset, padded_shape, padding_value=None):
 199   """Batches a window of tensors with padding.
 200 
 201   Args:
 202     dataset: the input dataset.
 203     padded_shape: (Optional.) `tf.TensorShape` or `tf.int64` vector tensor-like
 204       object representing the shape to which the input elements should be padded
 205       prior to batching. Any unknown dimensions (e.g. `tf.Dimension(None)` in a
 206       `tf.TensorShape` or `-1` in a tensor-like object) will be padded to the
 207       maximum size of that dimension in each batch.
 208     padding_value: (Optional.) A scalar-shaped `tf.Tensor`, representing the
 209       padding value to use. Defaults are `0` for numeric types and the empty
 210       string for string types. If `dataset` contains `tf.SparseTensor`, this
 211       value is ignored.
 212 
 213   Returns:
 214     A `Tensor` representing the batch of the entire input dataset.
 215 
 216   Raises:
 217     ValueError: if invalid arguments are provided.
 218   """
 219   if not issubclass(dataset.output_classes,
 220                     (ops.Tensor, sparse_tensor.SparseTensor)):
 221     raise TypeError("Input dataset expected to have a single tensor component")
 222   if issubclass(dataset.output_classes, (ops.Tensor)):
 223     return _padded_batch_dense_window(dataset, padded_shape, padding_value)
 224   elif issubclass(dataset.output_classes, (sparse_tensor.SparseTensor)):
 225     if padding_value is not None:
 226       raise ValueError("Padding value not allowed for sparse tensors")
 227     return _padded_batch_sparse_window(dataset, padded_shape)
 228   else:
 229     raise TypeError("Unsupported dataset type: %s" % dataset.output_classes)

 232 def _padded_batch_dense_window(dataset, padded_shape, padding_value=None):
 233   """Batches a window of dense tensors with padding."""
 234 
 235   padded_shape = math_ops.cast(
 236       convert.partial_shape_to_tensor(padded_shape), dtypes.int32)
 237 
 238   def key_fn(_):
 239     return np.int64(0)
 240 
 241   def max_init_fn(_):
 242     return padded_shape
 243 
 244   def max_reduce_fn(state, value):
 245     """Computes the maximum shape to pad to."""
 246     condition = math_ops.reduce_all(
 247         math_ops.logical_or(
 248             math_ops.less_equal(array_ops.shape(value), padded_shape),
 249             math_ops.equal(padded_shape, -1)))
 250     assert_op = control_flow_ops.Assert(condition, [
 251         "Actual shape greater than padded shape: ",
 252         array_ops.shape(value), padded_shape
 253     ])
 254     with ops.control_dependencies([assert_op]):
 255       return math_ops.maximum(state, array_ops.shape(value))
 256 
 257   def finalize_fn(state):
 258     return state
 259 
 260   # Compute the padded shape.
 261   max_reducer = grouping.Reducer(max_init_fn, max_reduce_fn, finalize_fn)
 262   padded_shape = get_single_element.get_single_element(
 263       dataset.apply(grouping.group_by_reducer(key_fn, max_reducer)))
 264 
 265   if padding_value is None:
 266     if dataset.output_types == dtypes.string:
 267       padding_value = ""
 268     elif dataset.output_types == dtypes.bool:
 269       padding_value = False
 270     elif dataset.output_types == dtypes.variant:
 271       raise TypeError("Unable to create padding for field of type 'variant'")
 272     else:
 273       padding_value = 0
 274 
 275   def batch_init_fn(_):
 276     batch_shape = array_ops.concat(
 277         [np.array([0], dtype=np.int32), padded_shape], 0)
 278     return gen_array_ops.empty(batch_shape, dtype=dataset.output_types)
 279 
 280   def batch_reduce_fn(state, value):
 281     return array_ops.concat([state, [value]], 0)
 282 
 283   def pad_fn(value):
 284     shape = array_ops.shape(value)
 285     left = array_ops.zeros_like(shape)
 286     right = padded_shape - shape
 287     return array_ops.pad(
 288         value, array_ops.stack([left, right], 1), constant_values=padding_value)
 289 
 290   batch_reducer = grouping.Reducer(batch_init_fn, batch_reduce_fn, finalize_fn)
 291   return get_single_element.get_single_element(
 292       dataset.map(pad_fn).apply(
 293           grouping.group_by_reducer(key_fn, batch_reducer)))

 296 def _padded_batch_sparse_window(dataset, padded_shape):
 297   """Batches a window of sparse tensors with padding."""
 298 
 299   def key_fn(_):
 300     return np.int64(0)
 301 
 302   def max_init_fn(_):
 303     return convert.partial_shape_to_tensor(padded_shape)
 304 
 305   def max_reduce_fn(state, value):
 306     """Computes the maximum shape to pad to."""
 307     condition = math_ops.reduce_all(
 308         math_ops.logical_or(
 309             math_ops.less_equal(value.dense_shape, padded_shape),
 310             math_ops.equal(padded_shape, -1)))
 311     assert_op = control_flow_ops.Assert(condition, [
 312         "Actual shape greater than padded shape: ", value.dense_shape,
 313         padded_shape
 314     ])
 315     with ops.control_dependencies([assert_op]):
 316       return math_ops.maximum(state, value.dense_shape)
 317 
 318   def finalize_fn(state):
 319     return state
 320 
 321   # Compute the padded shape.
 322   max_reducer = grouping.Reducer(max_init_fn, max_reduce_fn, finalize_fn)
 323   padded_shape = get_single_element.get_single_element(
 324       dataset.apply(grouping.group_by_reducer(key_fn, max_reducer)))
 325 
 326   def batch_init_fn(_):
 327     indices_shape = array_ops.concat([[0], [array_ops.size(padded_shape) + 1]],
 328                                      0)
 329     return sparse_tensor.SparseTensor(
 330         indices=gen_array_ops.empty(indices_shape, dtype=dtypes.int64),
 331         values=constant_op.constant([], shape=[0], dtype=dataset.output_types),
 332         dense_shape=array_ops.concat(
 333             [np.array([0], dtype=np.int64), padded_shape], 0))
 334 
 335   def batch_reduce_fn(state, value):
 336     padded_value = sparse_tensor.SparseTensor(
 337         indices=value.indices, values=value.values, dense_shape=padded_shape)
 338     reshaped_value = sparse_ops.sparse_reshape(
 339         padded_value,
 340         array_ops.concat(
 341             [np.array([1], dtype=np.int64), padded_value.dense_shape], 0))
 342     return sparse_ops.sparse_concat(0, [state, reshaped_value])
 343 
 344   reducer = grouping.Reducer(batch_init_fn, batch_reduce_fn, finalize_fn)
 345   return get_single_element.get_single_element(
 346       dataset.apply(grouping.group_by_reducer(key_fn, reducer)))

 349 class _UnbatchDataset(dataset_ops.UnaryDataset):
 350   """A dataset that splits the elements of its input into multiple elements."""
 351 
 352   def __init__(self, input_dataset):
 353     """See `unbatch()` for more details."""
 354     super(_UnbatchDataset, self).__init__(input_dataset)
 355     flat_shapes = nest.flatten(input_dataset.output_shapes)
 356     if any(s.ndims == 0 for s in flat_shapes):
 357       raise ValueError("Cannot unbatch an input with scalar components.")
 358     known_batch_dim = tensor_shape.Dimension(None)
 359     for s in flat_shapes:
 360       try:
 361         known_batch_dim = known_batch_dim.merge_with(s[0])
 362       except ValueError:
 363         raise ValueError("Cannot unbatch an input whose components have "
 364                          "different batch sizes.")
 365     self._input_dataset = input_dataset
 366 
 367   def _as_variant_tensor(self):
 368     return gen_dataset_ops.unbatch_dataset(
 369         self._input_dataset._as_variant_tensor(),  # pylint: disable=protected-access
 370         **dataset_ops.flat_structure(self))
 371 
 372   @property
 373   def output_classes(self):
 374     return self._input_dataset.output_classes
 375 
 376   @property
 377   def output_shapes(self):
 378     return nest.map_structure(lambda s: s[1:],
 379                               self._input_dataset.output_shapes)
 380 
 381   @property
 382   def output_types(self):
 383     return self._input_dataset.output_types

 386 @tf_export("data.experimental.unbatch")
 387 def unbatch():
 388   """Splits elements of a dataset into multiple elements on the batch dimension.
 389 
 390   For example, if elements of the dataset are shaped `[B, a0, a1, ...]`,
 391   where `B` may vary for each input element, then for each element in the
 392   dataset, the unbatched dataset will contain `B` consecutive elements
 393   of shape `[a0, a1, ...]`.
 394 
 395   ```python
 396   # NOTE: The following example uses `{ ... }` to represent the contents
 397   # of a dataset.
 398   a = { ['a', 'b', 'c'], ['a', 'b'], ['a', 'b', 'c', 'd'] }
 399 
 400   a.apply(tf.data.experimental.unbatch()) == {
 401       'a', 'b', 'c', 'a', 'b', 'a', 'b', 'c', 'd'}
 402   ```
 403 
 404   Returns:
 405     A `Dataset` transformation function, which can be passed to
 406     `tf.data.Dataset.apply`.
 407   """
 408 
 409   def _apply_fn(dataset):
 410     """Function from `Dataset` to `Dataset` that applies the transformation."""
 411     if not sparse.any_sparse(dataset.output_classes):
 412       return _UnbatchDataset(dataset)
 413 
 414     # NOTE(mrry): We must ensure that any SparseTensors in `dataset`
 415     # are normalized to the rank-1 dense representation, so that the
 416     # sparse-oblivious unbatching logic will slice them
 417     # appropriately. This leads to a somewhat inefficient re-encoding step
 418     # for all SparseTensor components.
 419     # TODO(mrry): Consider optimizing this in future
 420     # if it turns out to be a bottleneck.
 421     def normalize(arg, *rest):
 422       if rest:
 423         return sparse.serialize_many_sparse_tensors((arg,) + rest)
 424       else:
 425         return sparse.serialize_many_sparse_tensors(arg)
 426 
 427     normalized_dataset = dataset.map(normalize)
 428 
 429     # NOTE(mrry): Our `map()` has lost information about the sparseness
 430     # of any SparseTensor components, so re-apply the structure of the
 431     # original dataset.
 432     restructured_dataset = _RestructuredDataset(
 433         normalized_dataset,
 434         dataset.output_types,
 435         dataset.output_shapes,
 436         dataset.output_classes,
 437         allow_unsafe_cast=True)
 438     return _UnbatchDataset(restructured_dataset)
 439 
 440   return _apply_fn

 477 class _RestructuredDataset(dataset_ops.UnaryDataset):
 478   """An internal helper for changing the structure and shape of a dataset."""
 479 
 480   def __init__(self,
 481                dataset,
 482                output_types,
 483                output_shapes=None,
 484                output_classes=None,
 485                allow_unsafe_cast=False):
 486     """Creates a new dataset with the given output types and shapes.
 487 
 488     The given `dataset` must have a structure that is convertible:
 489     * `dataset.output_types` must be the same as `output_types` module nesting.
 490     * Each shape in `dataset.output_shapes` must be compatible with each shape
 491       in `output_shapes` (if given).
 492 
 493     Note: This helper permits "unsafe casts" for shapes, equivalent to using
 494     `tf.Tensor.set_shape()` where domain-specific knowledge is available.
 495 
 496     Args:
 497       dataset: A `Dataset` object.
 498       output_types: A nested structure of `tf.DType` objects.
 499       output_shapes: (Optional.) A nested structure of `tf.TensorShape` objects.
 500         If omitted, the shapes will be inherited from `dataset`.
 501       output_classes: (Optional.) A nested structure of class types.
 502         If omitted, the class types will be inherited from `dataset`.
 503       allow_unsafe_cast: (Optional.) If `True`, the caller may switch the
 504         reported output types and shapes of the restructured dataset, e.g. to
 505         switch a sparse tensor represented as `tf.variant` to its user-visible
 506         type and shape.
 507 
 508     Raises:
 509       ValueError: If either `output_types` or `output_shapes` is not compatible
 510         with the structure of `dataset`.
 511     """
 512     super(_RestructuredDataset, self).__init__(dataset)
 513     self._input_dataset = dataset
 514 
 515     if not allow_unsafe_cast:
 516       # Validate that the types are compatible.
 517       output_types = nest.map_structure(dtypes.as_dtype, output_types)
 518       flat_original_types = nest.flatten(dataset.output_types)
 519       flat_new_types = nest.flatten(output_types)
 520       if flat_original_types != flat_new_types:
 521         raise ValueError(
 522             "Dataset with output types %r cannot be restructured to have "
 523             "output types %r" % (dataset.output_types, output_types))
 524 
 525     self._output_types = output_types
 526 
 527     if output_shapes is None:
 528       # Inherit shapes from the original `dataset`.
 529       self._output_shapes = nest.pack_sequence_as(output_types,
 530                                                   nest.flatten(
 531                                                       dataset.output_shapes))
 532     else:
 533       if not allow_unsafe_cast:
 534         # Validate that the shapes are compatible.
 535         nest.assert_same_structure(output_types, output_shapes)
 536         flat_original_shapes = nest.flatten(dataset.output_shapes)
 537         flat_new_shapes = nest.flatten_up_to(output_types, output_shapes)
 538 
 539         for original_shape, new_shape in zip(flat_original_shapes,
 540                                              flat_new_shapes):
 541           if not original_shape.is_compatible_with(new_shape):
 542             raise ValueError(
 543                 "Dataset with output shapes %r cannot be restructured to have "
 544                 "incompatible output shapes %r" % (dataset.output_shapes,
 545                                                    output_shapes))
 546       self._output_shapes = nest.map_structure_up_to(
 547           output_types, tensor_shape.as_shape, output_shapes)
 548     if output_classes is None:
 549       # Inherit class types from the original `dataset`.
 550       self._output_classes = nest.pack_sequence_as(output_types,
 551                                                    nest.flatten(
 552                                                        dataset.output_classes))
 553     else:
 554       self._output_classes = output_classes
 555 
 556   def _as_variant_tensor(self):
 557     return self._input_dataset._as_variant_tensor()  # pylint: disable=protected-access
 558 
 559   @property
 560   def output_classes(self):
 561     return self._output_classes
 562 
 563   @property
 564   def output_types(self):
 565     return self._output_types
 566 
 567   @property
 568   def output_shapes(self):
 569     return self._output_shapes

 572 class _MapAndBatchDataset(dataset_ops.MapDataset):
 573   """A `Dataset` that maps a function over a batch of elements."""
 574 
 575   def __init__(self, input_dataset, map_func, batch_size, num_parallel_calls,
 576                drop_remainder):
 577     """See `Dataset.map()` for details."""
 578     super(_MapAndBatchDataset, self).__init__(input_dataset, map_func)
 579     self._batch_size_t = ops.convert_to_tensor(
 580         batch_size, dtype=dtypes.int64, name="batch_size")

 583     self._drop_remainder_t = ops.convert_to_tensor(
 584         drop_remainder, dtype=dtypes.bool, name="drop_remainder")

 586     self._batch_size = batch_size
 587     self._drop_remainder = drop_remainder

 615 @tf_export("data.experimental.map_and_batch")
 616 def map_and_batch(map_func,
 617                   batch_size,
 618                   num_parallel_batches=None,
 619                   drop_remainder=False,
 620                   num_parallel_calls=None):
 621   """Fused implementation of `map` and `batch`.
 622 
 623   Maps `map_func` across `batch_size` consecutive elements of this dataset
 624   and then combines them into a batch. Functionally, it is equivalent to `map`
 625   followed by `batch`. However, by fusing the two transformations together, the
 626   implementation can be more efficient. Surfacing this transformation in the API
 627   is temporary. Once automatic input pipeline optimization is implemented,
 628   the fusing of `map` and `batch` will happen automatically and this API will be
 629   deprecated.
 630 
 631   Args:
 632     map_func: A function mapping a nested structure of tensors to another
 633       nested structure of tensors.
 634     batch_size: A `tf.int64` scalar `tf.Tensor`, representing the number of
 635       consecutive elements of this dataset to combine in a single batch.
 636     num_parallel_batches: (Optional.) A `tf.int64` scalar `tf.Tensor`,
 637       representing the number of batches to create in parallel. On one hand,
 638       higher values can help mitigate the effect of stragglers. On the other
 639       hand, higher values can increase contention if CPU is scarce.
 640     drop_remainder: (Optional.) A `tf.bool` scalar `tf.Tensor`, representing
 641       whether the last batch should be dropped in case its size is smaller than
 642       desired; the default behavior is not to drop the smaller batch.
 643     num_parallel_calls: (Optional.) A `tf.int32` scalar `tf.Tensor`,
 644         representing the number of elements to process in parallel. If not
 645         specified, `batch_size * num_parallel_batches` elements will be
 646         processed in parallel.
 647 
 648   Returns:
 649     A `Dataset` transformation function, which can be passed to
 650     `tf.data.Dataset.apply`.
 651 
 652   Raises:
 653     ValueError: If both `num_parallel_batches` and `num_parallel_calls` are
 654       specified.
 655   """
 656 
 657   if num_parallel_batches is None and num_parallel_calls is None:
 658     num_parallel_calls = batch_size
 659   elif num_parallel_batches is not None and num_parallel_calls is None:
 660     num_parallel_calls = batch_size * num_parallel_batches
 661   elif num_parallel_batches is not None and num_parallel_calls is not None:
 662     raise ValueError("The `num_parallel_batches` and `num_parallel_calls` "
 663                      "arguments are mutually exclusive.")
 664 
 665   def _apply_fn(dataset):
 666     return _MapAndBatchDataset(dataset, map_func, batch_size,
 667                                num_parallel_calls, drop_remainder)
 668 
 669   return _apply_fn
