  30 class _ParseExampleDataset(dataset_ops.UnaryDataset):
  31   """A `Dataset` that parses `example` dataset into a `dict` dataset."""
  32 
  33   def __init__(self, input_dataset, features, num_parallel_calls):
  34     super(_ParseExampleDataset, self).__init__(input_dataset)
  35     self._input_dataset = input_dataset
  36     if not all(types == dtypes.string
  37                for types in nest.flatten(input_dataset.output_types)):
  38       raise TypeError("Input dataset should be a dataset of vectors of strings")
  39     self._num_parallel_calls = num_parallel_calls
  40     # pylint: disable=protected-access
  41     self._features = parsing_ops._prepend_none_dimension(features)
  42     # sparse_keys and dense_keys come back sorted here.
  43     (sparse_keys, sparse_types, dense_keys, dense_types, dense_defaults,
  44      dense_shapes) = parsing_ops._features_to_raw_params(
  45          self._features, [
  46              parsing_ops.VarLenFeature, parsing_ops.SparseFeature,
  47              parsing_ops.FixedLenFeature, parsing_ops.FixedLenSequenceFeature
  48          ])
  49     # TODO(b/112859642): Pass sparse_index and sparse_values for SparseFeature.
  50     (_, dense_defaults_vec, sparse_keys, sparse_types, dense_keys, dense_shapes,
  51      dense_shape_as_shape) = parsing_ops._process_raw_parameters(
  52          None, dense_defaults, sparse_keys, sparse_types, dense_keys,
  53          dense_types, dense_shapes)
  54     # pylint: enable=protected-access
  55     self._sparse_keys = sparse_keys
  56     self._sparse_types = sparse_types
  57     self._dense_keys = dense_keys
  58     self._dense_defaults = dense_defaults_vec
  59     self._dense_shapes = dense_shapes
  60     self._dense_types = dense_types
  61     dense_output_shapes = [
  62         self._input_dataset.output_shapes.concatenate(shape)
  63         for shape in dense_shape_as_shape
  64     ]
  65     sparse_output_shapes = [
  66         self._input_dataset.output_shapes.concatenate([None])
  67         for _ in range(len(sparse_keys))
  68     ]
  69 
  70     self._output_shapes = dict(
  71         zip(self._dense_keys + self._sparse_keys,
  72             dense_output_shapes + sparse_output_shapes))
  73     self._output_types = dict(
  74         zip(self._dense_keys + self._sparse_keys,
  75             self._dense_types + self._sparse_types))
  76     self._output_classes = dict(
  77         zip(self._dense_keys + self._sparse_keys,
  78             [ops.Tensor for _ in range(len(self._dense_defaults))] +
  79             [sparse_tensor.SparseTensor for _ in range(len(self._sparse_keys))
  80             ]))
  81 
  82   def _as_variant_tensor(self):
  83     return gen_dataset_ops.parse_example_dataset(
  84         self._input_dataset._as_variant_tensor(),  # pylint: disable=protected-access
  85         self._num_parallel_calls,
  86         self._dense_defaults,
  87         self._sparse_keys,
  88         self._dense_keys,
  89         self._sparse_types,
  90         self._dense_shapes,
  91         **dataset_ops.flat_structure(self))
  92 
  93   @property
  94   def output_shapes(self):
  95     return self._output_shapes
  96 
  97   @property
  98   def output_types(self):
  99     return self._output_types
 100 
 101   @property
 102   def output_classes(self):
 103     return self._output_classes

 107 @tf_export("data.experimental.parse_example_dataset")
 108 def parse_example_dataset(features, num_parallel_calls=1):
 109   """A transformation that parses `Example` protos into a `dict` of tensors.
 110 
 111   Parses a number of serialized `Example` protos given in `serialized`. We refer
 112   to `serialized` as a batch with `batch_size` many entries of individual
 113   `Example` protos.
 114 
 115   This op parses serialized examples into a dictionary mapping keys to `Tensor`
 116   and `SparseTensor` objects. `features` is a dict from keys to `VarLenFeature`,
 117   `SparseFeature`, and `FixedLenFeature` objects. Each `VarLenFeature`
 118   and `SparseFeature` is mapped to a `SparseTensor`, and each
 119   `FixedLenFeature` is mapped to a `Tensor`. See `tf.parse_example` for more
 120   details about feature dictionaries.
 121 
 122   Args:
 123    features: A `dict` mapping feature keys to `FixedLenFeature`,
 124      `VarLenFeature`, and `SparseFeature` values.
 125    num_parallel_calls: (Optional.) A `tf.int32` scalar `tf.Tensor`,
 126       representing the number of parsing processes to call in parallel.
 127 
 128   Returns:
 129     A dataset transformation function, which can be passed to
 130     `tf.data.Dataset.apply`.
 131 
 132   Raises:
 133     ValueError: if features argument is None.
 134   """
 135   if features is None:
 136     raise ValueError("Missing: features was %s." % features)
 137 
 138   def _apply_fn(dataset):
 139     """Function from `Dataset` to `Dataset` that applies the transformation."""
 140     out_dataset = _ParseExampleDataset(dataset, features, num_parallel_calls)
 141     if any([
 142         isinstance(feature, parsing_ops.SparseFeature)
 143         for _, feature in features.items()
 144     ]):
 145       # pylint: disable=protected-access
 146       # pylint: disable=g-long-lambda
 147       out_dataset = out_dataset.map(
 148           lambda x: parsing_ops._construct_sparse_tensors_for_sparse_features(
 149               features, x), num_parallel_calls=num_parallel_calls)
 150     return out_dataset
 151 
 152   return _apply_fn
