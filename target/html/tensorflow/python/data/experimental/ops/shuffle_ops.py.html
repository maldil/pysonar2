  29 class _ShuffleAndRepeatDataset(dataset_ops.UnaryDataset):
  30   """A `Dataset` that fuses `shuffle` and `repeat`."""
  31 
  32   def __init__(self, input_dataset, buffer_size, count=None, seed=None):
  33     super(_ShuffleAndRepeatDataset, self).__init__(input_dataset)
  34     self._input_dataset = input_dataset
  35     self._buffer_size = ops.convert_to_tensor(
  36         buffer_size, dtype=dtypes.int64, name="buffer_size")
  37     if count is None:
  38       self._count = constant_op.constant(-1, dtype=dtypes.int64, name="count")
  39     else:
  40       self._count = ops.convert_to_tensor(
  41           count, dtype=dtypes.int64, name="count")
  42     self._seed, self._seed2 = random_seed.get_seed(seed)
  43 
  44   def _as_variant_tensor(self):
  45     # pylint: disable=protected-access
  46     input_resource = self._input_dataset._as_variant_tensor()
  47     return gen_dataset_ops.shuffle_and_repeat_dataset(
  48         input_resource,
  49         buffer_size=self._buffer_size,
  50         count=self._count,
  51         seed=self._seed,
  52         seed2=self._seed2,
  53         **dataset_ops.flat_structure(self))
  54     # pylint: enable=protected-access
  55 
  56   @property
  57   def output_classes(self):
  58     return self._input_dataset.output_classes
  59 
  60   @property
  61   def output_shapes(self):
  62     return self._input_dataset.output_shapes
  63 
  64   @property
  65   def output_types(self):
  66     return self._input_dataset.output_types

  69 @tf_export("data.experimental.shuffle_and_repeat")
  70 def shuffle_and_repeat(buffer_size, count=None, seed=None):
  71   """Shuffles and repeats a Dataset returning a new permutation for each epoch.
  72 
  73   `dataset.apply(tf.data.experimental.shuffle_and_repeat(buffer_size, count))`
  74 
  75   is equivalent to
  76 
  77   `dataset.shuffle(buffer_size, reshuffle_each_iteration=True).repeat(count)`
  78 
  79   The difference is that the latter dataset is not serializable. So,
  80   if you need to checkpoint an input pipeline with reshuffling you must use
  81   this implementation.
  82 
  83   Args:
  84     buffer_size: A `tf.int64` scalar `tf.Tensor`, representing the
  85       maximum number elements that will be buffered when prefetching.
  86     count: (Optional.) A `tf.int64` scalar `tf.Tensor`, representing the
  87       number of times the dataset should be repeated. The default behavior
  88       (if `count` is `None` or `-1`) is for the dataset be repeated
  89       indefinitely.
  90     seed: (Optional.) A `tf.int64` scalar `tf.Tensor`, representing the
  91       random seed that will be used to create the distribution. See
  92       `tf.set_random_seed` for behavior.
  93 
  94   Returns:
  95     A `Dataset` transformation function, which can be passed to
  96     `tf.data.Dataset.apply`.
  97   """
  98 
  99   def _apply_fn(dataset):  # pylint: disable=missing-docstring
 100     return _ShuffleAndRepeatDataset(dataset, buffer_size, count, seed)
 101 
 102   return _apply_fn
