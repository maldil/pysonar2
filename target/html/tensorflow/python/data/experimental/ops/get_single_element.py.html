  15 """Python wrappers for Datasets and Iterators."""
  16 from __future__ import absolute_import
  17 from __future__ import division
  18 from __future__ import print_function

  20 from tensorflow.python.data.ops import dataset_ops
  21 from tensorflow.python.data.util import nest
  22 from tensorflow.python.data.util import sparse
  23 from tensorflow.python.ops import gen_dataset_ops
  24 from tensorflow.python.util.tf_export import tf_export
  25 
  26 
  27 @tf_export("data.experimental.get_single_element")
  28 def get_single_element(dataset):
  29   """Returns the single element in `dataset` as a nested structure of tensors.
  30 
  31   This function enables you to use a `tf.data.Dataset` in a stateless
  32   "tensor-in tensor-out" expression, without creating a `tf.data.Iterator`.
  33   This can be useful when your preprocessing transformations are expressed
  34   as a `Dataset`, and you want to use the transformation at serving time.
  35   For example:
  36 
  37   ```python
  38   input_batch = tf.placeholder(tf.string, shape=[BATCH_SIZE])
  39 
  40   def preprocessing_fn(input_str):
  41     # ...
  42     return image, label
  43 
  44   dataset = (tf.data.Dataset.from_tensor_slices(input_batch)
  45              .map(preprocessing_fn, num_parallel_calls=BATCH_SIZE)
  46              .batch(BATCH_SIZE))
  47 
  48   image_batch, label_batch = tf.data.experimental.get_single_element(dataset)
  49   ```
  50 
  51   Args:
  52     dataset: A `tf.data.Dataset` object containing a single element.
  53 
  54   Returns:
  55     A nested structure of `tf.Tensor` objects, corresponding to the single
  56     element of `dataset`.
  57 
  58   Raises:
  59     TypeError: if `dataset` is not a `tf.data.Dataset` object.
  60     InvalidArgumentError (at runtime): if `dataset` does not contain exactly
  61       one element.
  62   """
  63   if not isinstance(dataset, dataset_ops.Dataset):
  64     raise TypeError("`dataset` must be a `tf.data.Dataset` object.")
  65 
  66   nested_ret = nest.pack_sequence_as(
  67       dataset.output_types, gen_dataset_ops.dataset_to_single_element(
  68           dataset._as_variant_tensor(),  # pylint: disable=protected-access
  69           **dataset_ops.flat_structure(dataset)))
  70   return sparse.deserialize_sparse_tensors(
  71       nested_ret, dataset.output_types, dataset.output_shapes,
  72       dataset.output_classes)
