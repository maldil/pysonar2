  25 @tf_export("data.experimental.ignore_errors")
  26 def ignore_errors():
  27   """Creates a `Dataset` from another `Dataset` and silently ignores any errors.
  28 
  29   Use this transformation to produce a dataset that contains the same elements
  30   as the input, but silently drops any elements that caused an error. For
  31   example:
  32 
  33   ```python
  34   dataset = tf.data.Dataset.from_tensor_slices([1., 2., 0., 4.])
  35 
  36   # Computing `tf.check_numerics(1. / 0.)` will raise an InvalidArgumentError.
  37   dataset = dataset.map(lambda x: tf.check_numerics(1. / x, "error"))
  38 
  39   # Using `ignore_errors()` will drop the element that causes an error.
  40   dataset =
  41       dataset.apply(tf.data.experimental.ignore_errors())  # ==> {1., 0.5, 0.2}
  42   ```
  43 
  44   Returns:
  45     A `Dataset` transformation function, which can be passed to
  46     `tf.data.Dataset.apply`.
  47   """
  48 
  49   def _apply_fn(dataset):
  50     return _IgnoreErrorsDataset(dataset)
  51 
  52   return _apply_fn
  53 
  54 
  55 class _IgnoreErrorsDataset(dataset_ops.UnaryDataset):
  56   """A `Dataset` that silently ignores errors when computing its input."""
  57 
  58   def __init__(self, input_dataset):
  59     """See `Dataset.ignore_errors()` for details."""
  60     super(_IgnoreErrorsDataset, self).__init__(input_dataset)
  61     self._input_dataset = input_dataset
  62 
  63   def _as_variant_tensor(self):
  64     return gen_experimental_dataset_ops.experimental_ignore_errors_dataset(
  65         self._input_dataset._as_variant_tensor(),  # pylint: disable=protected-access
  66         **dataset_ops.flat_structure(self))
  67 
  68   @property
  69   def output_classes(self):
  70     return self._input_dataset.output_classes
  71 
  72   @property
  73   def output_shapes(self):
  74     return self._input_dataset.output_shapes
  75 
  76   @property
  77   def output_types(self):
  78     return self._input_dataset.output_types
