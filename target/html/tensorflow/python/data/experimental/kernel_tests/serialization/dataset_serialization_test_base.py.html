  40 def remove_variants(get_next_op):
  41   # TODO(b/72408568): Remove this once session.run can get
  42   # variant tensors.
  43   """Remove variants from a nest structure, so sess.run will execute."""
  44 
  45   def _remove_variant(x):
  46     if isinstance(x, ops.Tensor) and x.dtype == dtypes.variant:
  47       return ()
  48     else:
  49       return x
  50 
  51   return nest.map_structure(_remove_variant, get_next_op)

 212   def verify_reset_restored_iterator(self,
 213                                      ds_fn,
 214                                      num_outputs,
 215                                      break_point=None,
 216                                      sparse_tensors=False,
 217                                      verify_exhausted=True):
 218     """Attempts to re-initialize a restored iterator.
 219 
 220     This is useful when restoring a training checkpoint during validation.
 221 
 222     Args:
 223       ds_fn: See `run_core_tests`.
 224       num_outputs: See `run_core_tests`.
 225       break_point: Break point. Optional. Defaults to num_outputs/2.
 226       sparse_tensors: See `run_core_tests`.
 227       verify_exhausted: See `gen_outputs`.
 228 
 229     Raises:
 230       AssertionError if any test fails.
 231     """
 232     break_point = num_outputs // 2 if not break_point else break_point
 233 
 234     # Collect ground truth containing all outputs.
 235     expected = self.gen_outputs(
 236         ds_fn, [],
 237         num_outputs,
 238         sparse_tensors=sparse_tensors,
 239         verify_exhausted=verify_exhausted)
 240 
 241     # Skip some items and save checkpoint.
 242     self.gen_outputs(
 243         ds_fn, [],
 244         break_point,
 245         sparse_tensors=sparse_tensors,
 246         verify_exhausted=False)
 247 
 248     actual = []
 249     # Restore from checkpoint and then run init_op.
 250     with ops.Graph().as_default() as g:
 251       saver = self._import_meta_graph()
 252       init_op, get_next_op = self._get_iterator_ops_from_collection(
 253           ds_fn, sparse_tensors=sparse_tensors)
 254       get_next_op = remove_variants(get_next_op)
 255       with self.session(graph=g) as sess:
 256         self._restore(saver, sess)
 257         self._initialize(init_op, sess)
 258         for _ in range(num_outputs):
 259           actual.append(sess.run(get_next_op))
 260         if verify_exhausted:
 261           with self.assertRaises(errors.OutOfRangeError):
 262             sess.run(get_next_op)
 263     self.match(expected, actual)
 264 
 265   def verify_restore_in_modified_graph(self,
 266                                        ds_fn1,
 267                                        ds_fn2,
 268                                        num_outputs,
 269                                        break_point=None,
 270                                        sparse_tensors=False,
 271                                        verify_exhausted=True):
 272     """Attempts to restore an iterator in a modified graph.
 273 
 274     Builds an input pipeline using ds_fn1, runs it for `break_point` steps
 275     and saves a checkpoint. Then builds a new graph using ds_fn2, restores
 276     the checkpoint from ds_fn1 and verifies that the restore is successful.
 277 
 278     Args:
 279       ds_fn1: See `run_core_tests`.
 280       ds_fn2: See `run_core_tests`.
 281       num_outputs: See `run_core_tests`.
 282       break_point: Break point. Optional. Defaults to num_outputs/2.
 283       sparse_tensors: See `run_core_tests`.
 284       verify_exhausted: See `gen_outputs`.
 285 
 286     Raises:
 287       AssertionError if any test fails.
 288     """
 289     break_point = num_outputs // 2 if not break_point else break_point
 290 
 291     # Skip `break_point` items and store the remaining produced from ds_fn1
 292     # in `expected`.
 293     self.gen_outputs(
 294         ds_fn1, [],
 295         break_point,
 296         sparse_tensors=sparse_tensors,
 297         verify_exhausted=False)
 298     expected = self.gen_outputs(
 299         ds_fn1, [],
 300         num_outputs - break_point,
 301         ckpt_saved=True,
 302         sparse_tensors=sparse_tensors,
 303         verify_exhausted=verify_exhausted)
 304 
 305     # Generate `break_point` items from ds_fn1 and save checkpoint.
 306     self.gen_outputs(
 307         ds_fn1, [],
 308         break_point,
 309         sparse_tensors=sparse_tensors,
 310         verify_exhausted=False)
 311 
 312     actual = []
 313     # Build graph for ds_fn2 but load checkpoint for ds_fn1.
 314     with ops.Graph().as_default() as g:
 315       _, get_next_op, saver = self._build_graph(
 316           ds_fn2, sparse_tensors=sparse_tensors)
 317       get_next_op = remove_variants(get_next_op)
 318       with self.session(graph=g) as sess:
 319         self._restore(saver, sess)
 320         for _ in range(num_outputs - break_point):
 321           actual.append(sess.run(get_next_op))
 322         if verify_exhausted:
 323           with self.assertRaises(errors.OutOfRangeError):
 324             sess.run(get_next_op)
 325 
 326     self.match(expected, actual)

 420   def verify_run_with_breaks(self,

 464   def gen_outputs(self,
 465                   ds_fn,
 466                   break_points,
 467                   num_outputs,
 468                   ckpt_saved=False,
 469                   init_before_restore=False,
 470                   sparse_tensors=False,
 471                   verify_exhausted=True,
 472                   save_checkpoint_at_end=True):
 473     """Generates elements from input dataset while stopping at break points.
 474 
 475     Produces `num_outputs` outputs and saves the state of the iterator in the
 476     Saver checkpoint.
 477 
 478     Args:
 479       ds_fn: 0-argument function that returns the dataset.
 480       break_points: A list of integers. For each `break_point` in
 481         `break_points`, we produce outputs till `break_point` number of items
 482         have been produced and then checkpoint the state. The current graph
 483         and session are destroyed and a new graph and session are used to
 484         produce outputs till next checkpoint or till `num_outputs` elements
 485         have been produced. `break_point` must be <= `num_outputs`.
 486       num_outputs: The total number of outputs to produce from the iterator.
 487       ckpt_saved: Whether a checkpoint already exists. If False, we build the
 488         graph from ds_fn.
 489       init_before_restore: Whether init should be called before saver.restore.
 490         This is just so that we can verify that restoring an already initialized
 491         iterator works.
 492       sparse_tensors:  Whether dataset is built from SparseTensor(s).
 493       verify_exhausted: Whether to verify that the iterator has been exhausted
 494         after producing `num_outputs` elements.
 495       save_checkpoint_at_end: Whether to save a checkpoint after producing all
 496         outputs. If False, checkpoints are saved each break point but not at the
 497         end. Note that checkpoints overwrite each other so there is always only
 498         a single checkpoint available. Defaults to True.
 499 
 500     Returns:
 501       A list of `num_outputs` items.
 502     """
 503     outputs = []
 504 
 505     def get_ops():
 506       if ckpt_saved:
 507         saver = self._import_meta_graph()
 508         init_op, get_next_op = self._get_iterator_ops_from_collection(
 509             ds_fn, sparse_tensors=sparse_tensors)
 510       else:
 511         init_op, get_next_op, saver = self._build_graph(
 512             ds_fn, sparse_tensors=sparse_tensors)
 513       return init_op, get_next_op, saver
 514 
 515     for i in range(len(break_points) + 1):
 516       with ops.Graph().as_default() as g:
 517         init_op, get_next_op, saver = get_ops()
 518         get_next_op = remove_variants(get_next_op)
 519         with self.session(graph=g) as sess:
 520           if ckpt_saved:
 521             if init_before_restore:
 522               self._initialize(init_op, sess)
 523             self._restore(saver, sess)
 524           else:
 525             self._initialize(init_op, sess)
 526           start = break_points[i - 1] if i > 0 else 0
 527           end = break_points[i] if i < len(break_points) else num_outputs
 528           num_iters = end - start
 529           for _ in range(num_iters):
 530             outputs.append(sess.run(get_next_op))
 531           if i == len(break_points) and verify_exhausted:
 532             with self.assertRaises(errors.OutOfRangeError):
 533               sess.run(get_next_op)
 534           if save_checkpoint_at_end or i < len(break_points):
 535             self._save(sess, saver)
 536             ckpt_saved = True
 537 
 538     return outputs

 540   def match(self, expected, actual):
 541     """Matches nested structures.
 542 
 543     Recursively matches shape and values of `expected` and `actual`.
 544     Handles scalars, numpy arrays and other python sequence containers
 545     e.g. list, dict.
 546 
 547     Args:
 548       expected: Nested structure 1.
 549       actual: Nested structure 2.
 550 
 551     Raises:
 552       AssertionError if matching fails.
 553     """
 554     if isinstance(expected, np.ndarray):
 555       expected = expected.tolist()
 556     if isinstance(actual, np.ndarray):
 557       actual = actual.tolist()
 558     self.assertEqual(type(expected), type(actual))
 559 
 560     if nest.is_sequence(expected):
 561       self.assertEqual(len(expected), len(actual))
 562       if isinstance(expected, dict):
 563         for key1, key2 in zip(sorted(expected), sorted(actual)):
 564           self.assertEqual(key1, key2)
 565           self.match(expected[key1], actual[key2])
 566       else:
 567         for item1, item2 in zip(expected, actual):
 568           self.match(item1, item2)
 569     else:
 570       self.assertEqual(expected, actual)
 571 
 572   def does_not_match(self, expected, actual):
 573     with self.assertRaises(AssertionError):
 574       self.match(expected, actual)

 576   def gen_break_points(self, num_outputs, num_samples=10):
 577     """Generates `num_samples` breaks points in [0, num_outputs]."""
 578     return np.linspace(0, num_outputs, num_samples, dtype=int)

 580   def _build_graph(self, ds_fn, sparse_tensors=False):
 581     iterator = ds_fn().make_initializable_iterator()
 582 
 583     saveable = contrib_iterator_ops.make_saveable_from_iterator(iterator)
 584     ops.add_to_collection(ops.GraphKeys.SAVEABLE_OBJECTS, saveable)
 585     init_op = iterator.initializer
 586     if sparse_tensors:
 587       get_next = sparse_tensor.SparseTensor(*iterator.get_next())
 588     else:
 589       get_next = iterator.get_next()
 590     self._add_iterator_ops_to_collection(init_op, get_next, ds_fn,
 591                                          sparse_tensors)
 592     saver = saver_lib.Saver(allow_empty=True)
 593     return init_op, get_next, saver
 594 
 595   def _build_empty_graph(self, ds_fn, sparse_tensors=False):
 596     iterator = iterator_ops.Iterator.from_structure(
 597         self._get_output_types(ds_fn),
 598         output_shapes=self._get_output_shapes(ds_fn),
 599         output_classes=self._get_output_classes(ds_fn))
 600     saveable = contrib_iterator_ops.make_saveable_from_iterator(iterator)
 601     ops.add_to_collection(ops.GraphKeys.SAVEABLE_OBJECTS, saveable)
 602     if sparse_tensors:
 603       get_next = sparse_tensor.SparseTensor(*iterator.get_next())
 604     else:
 605       get_next = iterator.get_next()
 606     saver = saver_lib.Saver(allow_empty=True)
 607     return get_next, saver

 609   def _add_iterator_ops_to_collection(self,
 610                                       init_op,
 611                                       get_next,
 612                                       ds_fn,
 613                                       sparse_tensors=False):
 614     ops.add_to_collection("iterator_ops", init_op)
 615     # `get_next` may be a tuple e.g. in TensorSliceDataset. Since Collections
 616     # do not support tuples we flatten the tensors and restore the shape in
 617     # `_get_iterator_ops_from_collection`.
 618     if sparse_tensors:  # specific for deprecated `from_sparse_tensor_slices`.
 619       ops.add_to_collection("iterator_ops", get_next.indices)
 620       ops.add_to_collection("iterator_ops", get_next.values)
 621       ops.add_to_collection("iterator_ops", get_next.dense_shape)
 622       return
 623 
 624     get_next_list = nest.flatten(get_next)
 625     for i, output_class in enumerate(
 626         nest.flatten(self._get_output_classes(ds_fn))):
 627       if output_class is sparse_tensor.SparseTensor:
 628         ops.add_to_collection("iterator_ops", get_next_list[i].indices)
 629         ops.add_to_collection("iterator_ops", get_next_list[i].values)
 630         ops.add_to_collection("iterator_ops", get_next_list[i].dense_shape)
 631       else:
 632         ops.add_to_collection("iterator_ops", get_next_list[i])
 633 
 634   def _get_iterator_ops_from_collection(self, ds_fn, sparse_tensors=False):
 635     all_ops = ops.get_collection("iterator_ops")
 636     if sparse_tensors:  # specific for deprecated `from_sparse_tensor_slices`.
 637       init_op, indices, values, dense_shape = all_ops
 638       return init_op, sparse_tensor.SparseTensor(indices, values, dense_shape)
 639     get_next_list = []
 640     i = 1
 641     for output_class in nest.flatten(self._get_output_classes(ds_fn)):
 642       if output_class is sparse_tensor.SparseTensor:
 643         indices, values, dense_shape = all_ops[i:i + 3]
 644         i += 3
 645         get_next_list.append(
 646             sparse_tensor.SparseTensor(indices, values, dense_shape))
 647       else:
 648         get_next_list.append(all_ops[i])
 649         i += 1
 650     return all_ops[0], nest.pack_sequence_as(
 651         self._get_output_types(ds_fn), get_next_list)

 687   def _delete_ckpt(self):
 688     # Remove all checkpoint files.
 689     prefix = self._ckpt_path()
 690     pattern = prefix + "*"
 691     files = gfile.Glob(pattern)
 692     map(gfile.Remove, files)
