  15 """Base class for testing the input pipeline statistics gathering ops."""
  16 from __future__ import absolute_import
  17 from __future__ import division
  18 from __future__ import print_function

  20 
  21 from tensorflow.core.framework import summary_pb2
  22 from tensorflow.python.data.kernel_tests import test_base
  23 
  24 
  25 class StatsDatasetTestBase(test_base.DatasetTestBase):
  26   """Base class for testing statistics gathered in `StatsAggregator`."""
  27 
  28   def _assertSummaryContains(self, summary_str, tag):
  29     summary_proto = summary_pb2.Summary()
  30     summary_proto.ParseFromString(summary_str)
  31     for value in summary_proto.value:
  32       if tag == value.tag:
  33         return
  34     self.fail("Expected tag %r not found in summary %r" % (tag, summary_proto))
  35 
  36   def _assertSummaryHasCount(self, summary_str, tag, expected_value):
  37     summary_proto = summary_pb2.Summary()
  38     summary_proto.ParseFromString(summary_str)
  39     for value in summary_proto.value:
  40       if tag == value.tag:
  41         self.assertEqual(expected_value, value.histo.num)
  42         return
  43     self.fail("Expected tag %r not found in summary %r" % (tag, summary_proto))
  44 
  45   def _assertSummaryHasRange(self, summary_str, tag, min_value, max_value):
  46     summary_proto = summary_pb2.Summary()
  47     summary_proto.ParseFromString(summary_str)
  48     for value in summary_proto.value:
  49       if tag == value.tag:
  50         self.assertLessEqual(min_value, value.histo.min)
  51         self.assertGreaterEqual(max_value, value.histo.max)
  52         return
  53     self.fail("Expected tag %r not found in summary %r" % (tag, summary_proto))
  54

  63
