  34 class Structure(object):
  35   """Represents structural information, such as type and shape, about a value.
  36 
  37   A `Structure` generalizes the `tf.Tensor.dtype` and `tf.Tensor.shape`
  38   properties, so that we can define generic containers of objects including:
  39 
  40   * `tf.Tensor`
  41   * `tf.SparseTensor`
  42   * Nested structures of the above.
  43 
  44   TODO(b/110122868): In the future, a single `Structure` will replace the
  45   `tf.data.Dataset.output_types`, `tf.data.Dataset.output_shapes`,
  46   and `tf.data.Dataset.output_classes`, and similar properties and arguments in
  47   the `tf.data.Iterator` and `Optional` classes.
  48   """
  49   __metaclass__ = abc.ABCMeta
  50 
  51   @abc.abstractproperty
  52   def _flat_shapes(self):
  53     """A list of shapes matching the shapes of `self._to_tensor_list()`.
  54 
  55     Returns:
  56       A list of `tf.TensorShape` objects.
  57     """
  58     raise NotImplementedError("Structure._flat_shapes")
  59 
  60   @abc.abstractproperty
  61   def _flat_types(self):
  62     """A list of types matching the types of `self._to_tensor_list()`.
  63 
  64     Returns:
  65       A list of `tf.DType` objects.
  66     """
  67     raise NotImplementedError("Structure._flat_shapes")
  68 
  69   @abc.abstractmethod
  70   def is_compatible_with(self, other):
  71     """Returns `True` if `other` is compatible with this structure.
  72 
  73     A structure `t` is a "subtype" of `s` if:
  74 
  75     * `s` and `t` are instances of the same `Structure` subclass.
  76     * The nested structures (if any) of `s` and `t` are the same, according to
  77       `tf.contrib.framework.nest.assert_same_structure`, and each nested
  78       structure of `t` is a "subtype" of the corresponding nested structure of
  79       `s`.
  80     * Any `tf.DType` components of `t` are the same as the corresponding
  81       components in `s`.
  82     * Any `tf.TensorShape` components of `t` are compatible with the
  83       corresponding components in `s`, according to
  84       `tf.TensorShape.is_compatible_with`.
  85 
  86     Args:
  87       other: A `Structure`.
  88 
  89     Returns:
  90       `True` if `other` is a subtype of this structure, otherwise `False`.
  91     """
  92     raise NotImplementedError("Structure.is_compatible_with()")
  93 
  94   @abc.abstractmethod
  95   def _to_tensor_list(self, value):
  96     """Returns a flat list of `tf.Tensor` representing `value`.
  97 
  98     This method can be used, along with `self._flat_shapes` and
  99     `self._flat_types` to represent structured values in lower level APIs
 100     (such as plain TensorFlow operations) that do not understand structure.
 101 
 102     Requires: `self.is_compatible_with(Structure.from_value(value))`.
 103 
 104     Args:
 105       value: A value with compatible structure.
 106 
 107     Returns:
 108       A flat list of `tf.Tensor` representing `value`.
 109     """
 110     raise NotImplementedError("Structure._to_tensor_list()")
 111 
 112   @abc.abstractmethod
 113   def _from_tensor_list(self, flat_value):
 114     """Builds a flat list of `tf.Tensor` into a value matching this structure.
 115 
 116     Requires: The shapes and types of the tensors in `flat_value` must be
 117     compatible with `self._flat_shapes` and `self._flat_types` respectively.
 118 
 119     Args:
 120       flat_value: A list of `tf.Tensor` with compatible flat structure.
 121 
 122     Returns:
 123       A structured object matching this structure.
 124     """
 125     raise NotImplementedError("Structure._from_tensor_list()")
 126 
 127   @staticmethod
 128   def from_value(value):
 129     """Returns a `Structure` that represents the given `value`.
 130 
 131     Args:
 132       value: A potentially structured value.
 133 
 134     Returns:
 135       A `Structure` that is compatible with `value`.
 136 
 137     Raises:
 138       TypeError: If a structure cannot be built for `value`, because its type
 139         or one of its component types is not supported.
 140     """
 141     # TODO(b/110122868): Add support for custom types and Dataset to this
 142     # method.
 143     if isinstance(
 144         value,
 145         (sparse_tensor_lib.SparseTensor, sparse_tensor_lib.SparseTensorValue)):
 146       return SparseTensorStructure.from_value(value)
 147     elif isinstance(value, (tuple, dict)):
 148       return NestedStructure.from_value(value)
 149     else:
 150       for converter_type, converter_fn in (
 151           _STRUCTURE_CONVERSION_FUNCTION_REGISTRY.items()):
 152         if isinstance(value, converter_type):
 153           return converter_fn(value)
 154       try:
 155         tensor = ops.convert_to_tensor(value)
 156       except (ValueError, TypeError):
 157         raise TypeError("Could not build a structure for %r" % value)
 158       return TensorStructure.from_value(tensor)
 159 
 160   @staticmethod
 161   def _from_legacy_structure(output_types, output_shapes, output_classes):
 162     """Returns a `Structure` that represents the given legacy structure.
 163 
 164     This method provides a way to convert from the existing `Dataset` and
 165     `Iterator` structure-related properties to a `Structure` object.
 166 
 167     TODO(b/110122868): Remove this method once `Structure` is used throughout
 168     `tf.data`.
 169 
 170     Args:
 171       output_types: A nested structure of `tf.DType` objects corresponding to
 172         each component of a structured value.
 173       output_shapes: A nested structure of `tf.TensorShape` objects
 174         corresponding to each component a structured value.
 175       output_classes: A nested structure of Python `type` objects corresponding
 176         to each component of a structured value.
 177 
 178     Returns:
 179       A `Structure`.
 180 
 181     Raises:
 182       TypeError: If a structure cannot be built the arguments, because one of
 183         the component classes in `output_classes` is not supported.
 184     """
 185     flat_types = nest.flatten(output_types)
 186     flat_shapes = nest.flatten(output_shapes)
 187     flat_classes = nest.flatten(output_classes)
 188     flat_ret = []
 189     for flat_type, flat_shape, flat_class in zip(flat_types, flat_shapes,
 190                                                  flat_classes):
 191       if issubclass(flat_class, sparse_tensor_lib.SparseTensor):
 192         flat_ret.append(SparseTensorStructure(flat_type, flat_shape))
 193       elif issubclass(flat_class, ops.Tensor):
 194         flat_ret.append(TensorStructure(flat_type, flat_shape))
 195       else:
 196         # NOTE(mrry): Since legacy structures produced by iterators only
 197         # comprise Tensors, SparseTensors, and nests, we do not need to support
 198         # all structure types here.
 199         raise TypeError(
 200             "Could not build a structure for output class %r" % flat_type)
 201 
 202     ret = nest.pack_sequence_as(output_classes, flat_ret)
 203     if isinstance(ret, Structure):
 204       return ret
 205     else:
 206       return NestedStructure(ret)
 207 
 208   @staticmethod
 209   def _register_custom_converter(type_object, converter_fn):
 210     """Registers `converter_fn` for converting values of the given type.
 211 
 212     Args:
 213       type_object: A Python `type` object representing the type of values
 214         accepted by `converter_fn`.
 215       converter_fn: A function that takes one argument (an instance of the
 216         type represented by `type_object`) and returns a `Structure`.
 217     """
 218     _STRUCTURE_CONVERSION_FUNCTION_REGISTRY[type_object] = converter_fn

 221 # NOTE(mrry): The following classes make extensive use of non-public methods of
 222 # their base class, so we disable the protected-access lint warning once here.
 223 # pylint: disable=protected-access
 224 class NestedStructure(Structure):
 225   """Represents a nested structure in which each leaf is a `Structure`."""
 226 
 227   def __init__(self, nested_structure):
 228     self._nested_structure = nested_structure
 229     self._flat_shapes_list = []
 230     self._flat_types_list = []
 231     for s in nest.flatten(nested_structure):
 232       if not isinstance(s, Structure):
 233         raise TypeError("nested_structure must be a (potentially nested) tuple "
 234                         "or dictionary of Structure objects.")
 235       self._flat_shapes_list.extend(s._flat_shapes)
 236       self._flat_types_list.extend(s._flat_types)
 237 
 238   @property
 239   def _flat_shapes(self):
 240     return self._flat_shapes_list
 241 
 242   @property
 243   def _flat_types(self):
 244     return self._flat_types_list
 245 
 246   def is_compatible_with(self, other):
 247     if not isinstance(other, NestedStructure):
 248       return False
 249     try:
 250       # pylint: disable=protected-access
 251       nest.assert_same_structure(self._nested_structure,
 252                                  other._nested_structure)
 253     except (ValueError, TypeError):
 254       return False
 255 
 256     return all(
 257         substructure.is_compatible_with(other_substructure)
 258         for substructure, other_substructure in zip(
 259             nest.flatten(self._nested_structure),
 260             nest.flatten(other._nested_structure)))
 261 
 262   def _to_tensor_list(self, value):
 263     ret = []
 264 
 265     try:
 266       flat_value = nest.flatten_up_to(self._nested_structure, value)
 267     except (ValueError, TypeError):
 268       raise ValueError("The value %r is not compatible with the nested "
 269                        "structure %r." % (value, self._nested_structure))
 270 
 271     for sub_value, structure in zip(flat_value,
 272                                     nest.flatten(self._nested_structure)):
 273       if not structure.is_compatible_with(Structure.from_value(sub_value)):
 274         raise ValueError("Component value %r is not compatible with the nested "
 275                          "structure %r." % (sub_value, structure))
 276       ret.extend(structure._to_tensor_list(sub_value))
 277     return ret
 278 
 279   def _from_tensor_list(self, flat_value):
 280     if len(flat_value) != len(self._flat_types):
 281       raise ValueError("Expected %d flat values in NestedStructure but got %d."
 282                        % (len(self._flat_types), len(flat_value)))
 283 
 284     flat_ret = []
 285     for sub_value, structure in zip(flat_value,
 286                                     nest.flatten(self._nested_structure)):
 287       flat_ret.append(structure._from_tensor_list([sub_value]))
 288 
 289     return nest.pack_sequence_as(self._nested_structure, flat_ret)
 290 
 291   @staticmethod
 292   def from_value(value):
 293     flat_nested_structure = [
 294         Structure.from_value(sub_value) for sub_value in nest.flatten(value)
 295     ]
 296     return NestedStructure(nest.pack_sequence_as(value, flat_nested_structure))

 299 class TensorStructure(Structure):
 300   """Represents structural information about a `tf.Tensor`."""
 301 
 302   def __init__(self, dtype, shape):
 303     self._dtype = dtypes.as_dtype(dtype)
 304     self._shape = tensor_shape.as_shape(shape)
 305 
 306   @property
 307   def _flat_shapes(self):
 308     return [self._shape]
 309 
 310   @property
 311   def _flat_types(self):
 312     return [self._dtype]
 313 
 314   def is_compatible_with(self, other):
 315     return (isinstance(other, TensorStructure) and
 316             self._dtype.is_compatible_with(other._dtype) and
 317             self._shape.is_compatible_with(other._shape))
 318 
 319   def _to_tensor_list(self, value):
 320     if not self.is_compatible_with(Structure.from_value(value)):
 321       raise ValueError("Value %r is not convertible to a tensor with dtype %s "
 322                        "and shape %s." % (value, self._dtype, self._shape))
 323     return [value]
 324 
 325   def _from_tensor_list(self, flat_value):
 326     if len(flat_value) != 1:
 327       raise ValueError("TensorStructure corresponds to a single tf.Tensor.")
 328     if not self.is_compatible_with(Structure.from_value(flat_value[0])):
 329       raise ValueError("Cannot convert %r to a tensor with dtype %s and shape "
 330                        "%s." % (flat_value[0], self._dtype, self._shape))
 331     return flat_value[0]
 332 
 333   @staticmethod
 334   def from_value(value):
 335     return TensorStructure(value.dtype, value.shape)

 338 class SparseTensorStructure(Structure):
 339   """Represents structural information about a `tf.SparseTensor`."""
 340 
 341   def __init__(self, dtype, dense_shape):
 342     self._dtype = dtypes.as_dtype(dtype)
 343     self._dense_shape = tensor_shape.as_shape(dense_shape)
 344 
 345   @property
 346   def _flat_shapes(self):
 347     return [tensor_shape.vector(3)]
 348 
 349   @property
 350   def _flat_types(self):
 351     return [dtypes.variant]
 352 
 353   def is_compatible_with(self, other):
 354     return (isinstance(other, SparseTensorStructure) and
 355             self._dtype.is_compatible_with(other._dtype) and
 356             self._dense_shape.is_compatible_with(other._dense_shape))
 357 
 358   def _to_tensor_list(self, value):
 359     return [sparse_ops.serialize_sparse(value, out_type=dtypes.variant)]
 360 
 361   def _from_tensor_list(self, flat_value):
 362     if (len(flat_value) != 1 or flat_value[0].dtype != dtypes.variant or
 363         not flat_value[0].shape.is_compatible_with(tensor_shape.vector(3))):
 364       raise ValueError("SparseTensorStructure corresponds to a single "
 365                        "tf.variant vector of length 3.")
 366     return sparse_ops.deserialize_sparse(
 367         flat_value[0], dtype=self._dtype, rank=self._dense_shape.ndims)
 368 
 369   @staticmethod
 370   def from_value(value):
 371     sparse_tensor = sparse_tensor_lib.SparseTensor.from_value(value)
 372     return SparseTensorStructure(
 373         sparse_tensor.dtype,
 374         tensor_util.constant_value_as_shape(sparse_tensor.dense_shape))
