  16 """## Functions for working with arbitrarily nested sequences of elements.
  17 
  18 NOTE(mrry): This fork of the `tensorflow.python.util.nest` module
  19 makes two changes:
  20 
  21 1. It removes support for lists as a level of nesting in nested structures.
  22 2. It adds support for `SparseTensorValue` as an atomic element.
  23 
  24 The motivation for this change is twofold:
  25 
  26 1. It seems more natural for lists to be treated (e.g. in Dataset constructors)
  27    as tensors, rather than lists of (lists of...) tensors.
  28 2. This is needed because `SparseTensorValue` is implemented as a `namedtuple`
  29    that would normally be flattened and we want to be able to create sparse
  30    tensor from `SparseTensorValue's similarly to creating tensors from numpy
  31    arrays.
  32 """

  46 def _sorted(dict_):
  47   """Returns a sorted list of the dict keys, with error if keys not sortable."""
  48   try:
  49     return sorted(_six.iterkeys(dict_))
  50   except TypeError:
  51     raise TypeError("nest only supports dicts with sortable keys.")

  54 def _sequence_like(instance, args):
  55   """Converts the sequence `args` to the same type as `instance`.
  56 
  57   Args:
  58     instance: an instance of `tuple`, `list`, or a `namedtuple` class.
  59     args: elements to be converted to a sequence.
  60 
  61   Returns:
  62     `args` with the type of `instance`.
  63   """
  64   if isinstance(instance, dict):
  65     # Pack dictionaries in a deterministic order by sorting the keys.
  66     # Notice this means that we ignore the original order of `OrderedDict`
  67     # instances. This is intentional, to avoid potential bugs caused by mixing
  68     # ordered and plain dicts (e.g., flattening a dict but using a
  69     # corresponding `OrderedDict` to pack it back).
  70     result = dict(zip(_sorted(instance), args))
  71     return type(instance)((key, result[key]) for key in _six.iterkeys(instance))
  72   elif (isinstance(instance, tuple) and
  73         hasattr(instance, "_fields") and
  74         isinstance(instance._fields, _collections.Sequence) and
  75         all(isinstance(f, _six.string_types) for f in instance._fields)):
  76     # This is a namedtuple
  77     return type(instance)(*args)
  78   else:
  79     # Not a namedtuple
  80     return type(instance)(args)

  83 def _yield_value(iterable):
  84   if isinstance(iterable, dict):
  85     # Iterate through dictionaries in a deterministic order by sorting the
  86     # keys. Notice this means that we ignore the original order of `OrderedDict`
  87     # instances. This is intentional, to avoid potential bugs caused by mixing
  88     # ordered and plain dicts (e.g., flattening a dict but using a
  89     # corresponding `OrderedDict` to pack it back).
  90     for key in _sorted(iterable):
  91       yield iterable[key]
  92   elif isinstance(iterable, _sparse_tensor.SparseTensorValue):
  93     yield iterable
  94   else:
  95     for value in iterable:
  96       yield value

 106 def assert_same_structure(nest1, nest2, check_types=True):
 107   """Asserts that two structures are nested in the same way.
 108 
 109   Args:
 110     nest1: an arbitrarily nested structure.
 111     nest2: an arbitrarily nested structure.
 112     check_types: if `True` (default) types of sequences should be same as
 113       well. For dictionary, "type" of dictionary is considered to include its
 114       keys. In other words, two dictionaries with different keys are considered
 115       to have a different "type". If set to `False`, two iterables are
 116       considered same as long as they yield the elements that have same
 117       structures.
 118 
 119   Raises:
 120     ValueError: If the two structures do not have the same number of elements or
 121       if the two structures are not nested in the same way.
 122     TypeError: If the two structures differ in the type of sequence in any of
 123       their substructures. Only possible if `check_types` is `True`.
 124   """
 125   _pywrap_tensorflow.AssertSameStructureForData(nest1, nest2, check_types)

 128 def _packed_nest_with_indices(structure, flat, index):
 129   """Helper function for pack_nest_as.
 130 
 131   Args:
 132     structure: Substructure (tuple of elements and/or tuples) to mimic
 133     flat: Flattened values to output substructure for.
 134     index: Index at which to start reading from flat.
 135 
 136   Returns:
 137     The tuple (new_index, child), where:
 138       * new_index - the updated index into `flat` having processed `structure`.
 139       * packed - the subset of `flat` corresponding to `structure`,
 140                  having started at `index`, and packed into the same nested
 141                  format.
 142 
 143   Raises:
 144     ValueError: if `structure` contains more elements than `flat`
 145       (assuming indexing starts from `index`).
 146   """
 147   packed = []
 148   for s in _yield_value(structure):
 149     if is_sequence(s):
 150       new_index, child = _packed_nest_with_indices(s, flat, index)
 151       packed.append(_sequence_like(s, child))
 152       index = new_index
 153     else:
 154       packed.append(flat[index])
 155       index += 1
 156   return index, packed

 159 def pack_sequence_as(structure, flat_sequence):
 160   """Returns a given flattened sequence packed into a nest.
 161 
 162   If `structure` is a scalar, `flat_sequence` must be a single-element list;
 163   in this case the return value is `flat_sequence[0]`.
 164 
 165   Args:
 166     structure: tuple or list constructed of scalars and/or other tuples/lists,
 167       or a scalar.  Note: numpy arrays are considered scalars.
 168     flat_sequence: flat sequence to pack.
 169 
 170   Returns:
 171     packed: `flat_sequence` converted to have the same recursive structure as
 172       `structure`.
 173 
 174   Raises:
 175     ValueError: If nest and structure have different element counts.
 176   """
 177   if not (is_sequence(flat_sequence) or isinstance(flat_sequence, list)):
 178     raise TypeError("flat_sequence must be a sequence")
 179 
 180   if not is_sequence(structure):
 181     if len(flat_sequence) != 1:
 182       raise ValueError("Structure is a scalar but len(flat_sequence) == %d > 1"
 183                        % len(flat_sequence))
 184     return flat_sequence[0]
 185 
 186   flat_structure = flatten(structure)
 187   if len(flat_structure) != len(flat_sequence):
 188     raise ValueError(
 189         "Could not pack sequence. Structure had %d elements, but flat_sequence "
 190         "had %d elements.  Structure: %s, flat_sequence: %s."
 191         % (len(flat_structure), len(flat_sequence), structure, flat_sequence))
 192 
 193   _, packed = _packed_nest_with_indices(structure, flat_sequence, 0)
 194   return _sequence_like(structure, packed)

 197 def map_structure(func, *structure, **check_types_dict):
 198   """Applies `func` to each entry in `structure` and returns a new structure.
 199 
 200   Applies `func(x[0], x[1], ...)` where x[i] is an entry in
 201   `structure[i]`.  All structures in `structure` must have the same arity,
 202   and the return value will contain the results in the same structure.
 203 
 204   Args:
 205     func: A callable that accepts as many arguments are there are structures.
 206     *structure: scalar, or tuple or list of constructed scalars and/or other
 207       tuples/lists, or scalars.  Note: numpy arrays are considered scalars.
 208     **check_types_dict: only valid keyword argument is `check_types`. If set to
 209       `True` (default) the types of iterables within the structures have to be
 210       same (e.g. `map_structure(func, [1], (1,))` raises a `TypeError`
 211       exception). To allow this set this argument to `False`.
 212 
 213   Returns:
 214     A new structure with the same arity as `structure`, whose values correspond
 215     to `func(x[0], x[1], ...)` where `x[i]` is a value in the corresponding
 216     location in `structure[i]`. If there are different sequence types and
 217     `check_types` is `False` the sequence types of the first structure will be
 218     used.
 219 
 220   Raises:
 221     TypeError: If `func` is not callable or if the structures do not match
 222       each other by depth tree.
 223     ValueError: If no structure is provided or if the structures do not match
 224       each other by type.
 225     ValueError: If wrong keyword arguments are provided.
 226   """
 227   if not callable(func):
 228     raise TypeError("func must be callable, got: %s" % func)
 229 
 230   if not structure:
 231     raise ValueError("Must provide at least one structure")
 232 
 233   if check_types_dict:
 234     if "check_types" not in check_types_dict or len(check_types_dict) > 1:
 235       raise ValueError("Only valid keyword argument is check_types")
 236     check_types = check_types_dict["check_types"]
 237   else:
 238     check_types = True
 239 
 240   for other in structure[1:]:
 241     assert_same_structure(structure[0], other, check_types=check_types)
 242 
 243   flat_structure = [flatten(s) for s in structure]
 244   entries = zip(*flat_structure)
 245 
 246   return pack_sequence_as(
 247       structure[0], [func(*x) for x in entries])

 250 def _yield_flat_up_to(shallow_tree, input_tree):
 251   """Yields elements `input_tree` partially flattened up to `shallow_tree`."""
 252   if is_sequence(shallow_tree):
 253     for shallow_branch, input_branch in zip(_yield_value(shallow_tree),
 254                                             _yield_value(input_tree)):
 255       for input_leaf in _yield_flat_up_to(shallow_branch, input_branch):
 256         yield input_leaf
 257   else:
 258     yield input_tree

 261 def assert_shallow_structure(shallow_tree, input_tree, check_types=True):
 262   """Asserts that `shallow_tree` is a shallow structure of `input_tree`.
 263 
 264   That is, this function tests if the `input_tree` structure can be created from
 265   the `shallow_tree` structure by replacing its leaf nodes with deeper
 266   tree structures.
 267 
 268   Examples:
 269 
 270   The following code will raise an exception:
 271   ```python
 272     shallow_tree = ["a", "b"]
 273     input_tree = ["c", ["d", "e"], "f"]
 274     assert_shallow_structure(shallow_tree, input_tree)
 275   ```
 276 
 277   The following code will not raise an exception:
 278   ```python
 279     shallow_tree = ["a", "b"]
 280     input_tree = ["c", ["d", "e"]]
 281     assert_shallow_structure(shallow_tree, input_tree)
 282   ```
 283 
 284   Args:
 285     shallow_tree: an arbitrarily nested structure.
 286     input_tree: an arbitrarily nested structure.
 287     check_types: if `True` (default) the sequence types of `shallow_tree` and
 288       `input_tree` have to be the same.
 289 
 290   Raises:
 291     TypeError: If `shallow_tree` is a sequence but `input_tree` is not.
 292     TypeError: If the sequence types of `shallow_tree` are different from
 293       `input_tree`. Only raised if `check_types` is `True`.
 294     ValueError: If the sequence lengths of `shallow_tree` are different from
 295       `input_tree`.
 296   """
 297   if is_sequence(shallow_tree):
 298     if not is_sequence(input_tree):
 299       raise TypeError(
 300           "If shallow structure is a sequence, input must also be a sequence. "
 301           "Input has type: %s." % type(input_tree))
 302 
 303     if check_types and not isinstance(input_tree, type(shallow_tree)):
 304       raise TypeError(
 305           "The two structures don't have the same sequence type. Input "
 306           "structure has type %s, while shallow structure has type %s."
 307           % (type(input_tree), type(shallow_tree)))
 308 
 309     if len(input_tree) != len(shallow_tree):
 310       raise ValueError(
 311           "The two structures don't have the same sequence length. Input "
 312           "structure has length %s, while shallow structure has length %s."
 313           % (len(input_tree), len(shallow_tree)))
 314 
 315     if check_types and isinstance(shallow_tree, dict):
 316       if set(input_tree) != set(shallow_tree):
 317         raise ValueError(
 318             "The two structures don't have the same keys. Input "
 319             "structure has keys %s, while shallow structure has keys %s." %
 320             (list(_six.iterkeys(input_tree)),
 321              list(_six.iterkeys(shallow_tree))))
 322       input_tree = list(sorted(_six.iteritems(input_tree)))
 323       shallow_tree = list(sorted(_six.iteritems(shallow_tree)))
 324 
 325     for shallow_branch, input_branch in zip(shallow_tree, input_tree):
 326       assert_shallow_structure(shallow_branch, input_branch,
 327                                check_types=check_types)

 330 def flatten_up_to(shallow_tree, input_tree):
 331   """Flattens `input_tree` up to `shallow_tree`.
 332 
 333   Any further depth in structure in `input_tree` is retained as elements in the
 334   partially flatten output.
 335 
 336   If `shallow_tree` and `input_tree` are not sequences, this returns a
 337   single-element list: `[input_tree]`.
 338 
 339   Use Case:
 340 
 341   Sometimes we may wish to partially flatten a nested sequence, retaining some
 342   of the nested structure. We achieve this by specifying a shallow structure,
 343   `shallow_tree`, we wish to flatten up to.
 344 
 345   The input, `input_tree`, can be thought of as having the same structure as
 346   `shallow_tree`, but with leaf nodes that are themselves tree structures.
 347 
 348   Examples:
 349 
 350   ```python
 351   input_tree = [[[2, 2], [3, 3]], [[4, 9], [5, 5]]]
 352   shallow_tree = [[True, True], [False, True]]
 353 
 354   flattened_input_tree = flatten_up_to(shallow_tree, input_tree)
 355   flattened_shallow_tree = flatten_up_to(shallow_tree, shallow_tree)
 356 
 357   # Output is:
 358   # [[2, 2], [3, 3], [4, 9], [5, 5]]
 359   # [True, True, False, True]
 360   ```
 361 
 362   ```python
 363   input_tree = [[('a', 1), [('b', 2), [('c', 3), [('d', 4)]]]]]
 364   shallow_tree = [['level_1', ['level_2', ['level_3', ['level_4']]]]]
 365 
 366   input_tree_flattened_as_shallow_tree = flatten_up_to(shallow_tree, input_tree)
 367   input_tree_flattened = flatten(input_tree)
 368 
 369   # Output is:
 370   # [('a', 1), ('b', 2), ('c', 3), ('d', 4)]
 371   # ['a', 1, 'b', 2, 'c', 3, 'd', 4]
 372   ```
 373 
 374   Non-Sequence Edge Cases:
 375 
 376   ```python
 377   flatten_up_to(0, 0)  # Output: [0]
 378   flatten_up_to(0, [0, 1, 2])  # Output: [[0, 1, 2]]
 379   flatten_up_to([0, 1, 2], 0)  # Output: TypeError
 380   flatten_up_to([0, 1, 2], [0, 1, 2])  # Output: [0, 1, 2]
 381   ```
 382 
 383   Args:
 384     shallow_tree: a possibly pruned structure of input_tree.
 385     input_tree: an arbitrarily nested structure or a scalar object.
 386       Note, numpy arrays are considered scalars.
 387 
 388   Returns:
 389     A Python list, the partially flattened version of `input_tree` according to
 390     the structure of `shallow_tree`.
 391 
 392   Raises:
 393     TypeError: If `shallow_tree` is a sequence but `input_tree` is not.
 394     TypeError: If the sequence types of `shallow_tree` are different from
 395       `input_tree`.
 396     ValueError: If the sequence lengths of `shallow_tree` are different from
 397       `input_tree`.
 398   """
 399   assert_shallow_structure(shallow_tree, input_tree)
 400   return list(_yield_flat_up_to(shallow_tree, input_tree))

 403 def map_structure_up_to(shallow_tree, func, *inputs):
 404   """Applies a function or op to a number of partially flattened inputs.
 405 
 406   The `inputs` are flattened up to `shallow_tree` before being mapped.
 407 
 408   Use Case:
 409 
 410   Sometimes we wish to apply a function to a partially flattened
 411   sequence (for example when the function itself takes sequence inputs). We
 412   achieve this by specifying a shallow structure, `shallow_tree` we wish to
 413   flatten up to.
 414 
 415   The `inputs`, can be thought of as having the same structure as
 416   `shallow_tree`, but with leaf nodes that are themselves tree structures.
 417 
 418   This function, therefore, will return something with the same base structure
 419   as `shallow_tree`.
 420 
 421   Examples:
 422 
 423   ```python
 424   ab_tuple = collections.namedtuple("ab_tuple", "a, b")
 425   op_tuple = collections.namedtuple("op_tuple", "add, mul")
 426   inp_val = ab_tuple(a=2, b=3)
 427   inp_ops = ab_tuple(a=op_tuple(add=1, mul=2), b=op_tuple(add=2, mul=3))
 428   out = map_structure_up_to(inp_val, lambda val, ops: (val + ops.add) * ops.mul,
 429                             inp_val, inp_ops)
 430 
 431   # Output is: ab_tuple(a=6, b=15)
 432   ```
 433 
 434   ```python
 435   data_list = [[2, 4, 6, 8], [[1, 3, 5, 7, 9], [3, 5, 7]]]
 436   name_list = ['evens', ['odds', 'primes']]
 437   out = map_structure_up_to(
 438       name_list,
 439       lambda name, sec: "first_{}_{}".format(len(sec), name),
 440       name_list, data_list)
 441 
 442   # Output is: ['first_4_evens', ['first_5_odds', 'first_3_primes']]
 443   ```
 444 
 445   Args:
 446     shallow_tree: a shallow tree, common to all the inputs.
 447     func: callable which will be applied to each input individually.
 448     *inputs: arbitrarily nested combination of objects that are compatible with
 449         shallow_tree. The function `func` is applied to corresponding
 450         partially flattened elements of each input, so the function must support
 451         arity of `len(inputs)`.
 452 
 453   Raises:
 454     TypeError: If `shallow_tree` is a sequence but `input_tree` is not.
 455     TypeError: If the sequence types of `shallow_tree` are different from
 456       `input_tree`.
 457     ValueError: If the sequence lengths of `shallow_tree` are different from
 458       `input_tree`.
 459 
 460   Returns:
 461     result of repeatedly applying `func`, with same structure as
 462     `shallow_tree`.
 463   """
 464   if not inputs:
 465     raise ValueError("Cannot map over no sequences")
 466   for input_tree in inputs:
 467     assert_shallow_structure(shallow_tree, input_tree)
 468 
 469   # Flatten each input separately, apply the function to corresponding elements,
 470   # then repack based on the structure of the first input.
 471   all_flattened_up_to = [flatten_up_to(shallow_tree, input_tree)
 472                          for input_tree in inputs]
 473 
 474   results = [func(*tensors) for tensors in zip(*all_flattened_up_to)]
 475   return pack_sequence_as(structure=shallow_tree, flat_sequence=results)
