  29 class DatasetTestBase(test.TestCase):
  30   """Base class for dataset tests."""
  31 
  32   def assertSparseValuesEqual(self, a, b):
  33     """Asserts that two SparseTensors/SparseTensorValues are equal."""
  34     self.assertAllEqual(a.indices, b.indices)
  35     self.assertAllEqual(a.values, b.values)
  36     self.assertAllEqual(a.dense_shape, b.dense_shape)
  37 
  38   def getNext(self, dataset):
  39     """Returns a callable that returns the next element of the dataset.
  40 
  41     Example use:
  42     ```python
  43     # In both graph and eager modes
  44     dataset = ...
  45     nxt = self.getNext(dataset)
  46     result = self.evaluate(nxt())
  47     ```
  48 
  49     Args:
  50       dataset: A dataset whose next element is returned
  51 
  52     Returns:
  53       A callable that returns the next element of `dataset`
  54     """
  55     it = dataset.make_one_shot_iterator()
  56     if context.executing_eagerly():
  57       return it.get_next
  58     else:
  59       nxt = it.get_next()
  60       return lambda: nxt
  61 
  62   def assertDatasetsEqual(self, dataset1, dataset2):
  63     """Checks that datasets are equal. Supports both graph and eager mode."""
  64     self.assertEqual(dataset1.output_types, dataset2.output_types)
  65     self.assertEqual(dataset1.output_classes, dataset2.output_classes)
  66 
  67     next1 = self.getNext(dataset1)
  68     next2 = self.getNext(dataset2)
  69     while True:
  70       try:
  71         op1 = self.evaluate(next1())
  72       except errors.OutOfRangeError:
  73         with self.assertRaises(errors.OutOfRangeError):
  74           self.evaluate(next2())
  75         break
  76       op2 = self.evaluate(next2())
  77 
  78       op1 = nest.flatten(op1)
  79       op2 = nest.flatten(op2)
  80       assert len(op1) == len(op2)
  81       for i in range(len(op1)):
  82         if isinstance(
  83             op1[i],
  84             (sparse_tensor.SparseTensor, sparse_tensor.SparseTensorValue)):
  85           self.assertSparseValuesEqual(op1[i], op2[i])
  86         else:
  87           self.assertAllEqual(op1[i], op2[i])
  88 
  89   def assertDatasetsRaiseSameError(self,
  90                                    dataset1,
  91                                    dataset2,
  92                                    exception_class,
  93                                    replacements=None):
  94     """Checks that datasets raise the same error on the first get_next call."""
  95     next1 = self.getNext(dataset1)
  96     next2 = self.getNext(dataset2)
  97     try:
  98       self.evaluate(next1())
  99       raise ValueError(
 100           'Expected dataset to raise an error of type %s, but it did not.' %
 101           repr(exception_class))
 102     except exception_class as e:
 103       expected_message = e.message
 104       for old, new, count in replacements:
 105         expected_message = expected_message.replace(old, new, count)
 106       # Check that the first segment of the error messages are the same.
 107       with self.assertRaisesRegexp(exception_class,
 108                                    re.escape(expected_message)):
 109         self.evaluate(next2())
