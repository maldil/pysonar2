  32 @tf_export("compat.forward_compatible")
  33 def forward_compatible(year, month, day):
  34   """Return true if the forward compatibility window has expired.
  35 
  36   See [Version
  37   compatibility](https://tensorflow.org/guide/version_compat#backward_forward).
  38 
  39   Forward-compatibility refers to scenarios where the producer of a TensorFlow
  40   model (a GraphDef or SavedModel) is compiled against a version of the
  41   TensorFlow library newer than what the consumer was compiled against. The
  42   "producer" is typically a Python program that constructs and trains a model
  43   while the "consumer" is typically another program that loads and serves the
  44   model.
  45 
  46   TensorFlow has been supporting a 3 week forward-compatibility window for
  47   programs compiled from source at HEAD.
  48 
  49   For example, consider the case where a new operation `MyNewAwesomeAdd` is
  50   created with the intent of replacing the implementation of an existing Python
  51   wrapper - `tf.add`.  The Python wrapper implementation should change from
  52   something like:
  53 
  54   ```python
  55   def add(inputs, name=None):
  56     return gen_math_ops.add(inputs, name)
  57   ```
  58 
  59   to:
  60 
  61   ```python
  62   from tensorflow.python.compat import compat
  63 
  64   def add(inputs, name=None):
  65     if compat.forward_compatible(year, month, day):
  66       # Can use the awesome new implementation.
  67       return gen_math_ops.my_new_awesome_add(inputs, name)
  68     # To maintain forward compatibiltiy, use the old implementation.
  69     return gen_math_ops.add(inputs, name)
  70   ```
  71 
  72   Where `year`, `month`, and `day` specify the date beyond which binaries
  73   that consume a model are expected to have been updated to include the
  74   new operations. This date is typically at least 3 weeks beyond the date
  75   the code that adds the new operation is committed.
  76 
  77   Args:
  78     year:  A year (e.g., 2018).
  79     month: A month (1 <= month <= 12) in year.
  80     day:   A day (1 <= day <= 31, or 30, or 29, or 28) in month.
  81 
  82   Returns:
  83     True if the caller can expect that serialized TensorFlow graphs produced
  84     can be consumed by programs that are compiled with the TensorFlow library
  85     source code after (year, month, day).
  86   """
  87   return _FORWARD_COMPATIBILITY_HORIZON > datetime.date(year, month, day)

  90 @tf_export("compat.forward_compatibility_horizon")
  91 @tf_contextlib.contextmanager
  92 def forward_compatibility_horizon(year, month, day):
  93   """Context manager for testing forward compatibility of generated graphs.
  94 
  95   See [Version
  96   compatibility](https://tensorflow.org/guide/version_compat#backward_forward).
  97 
  98   To ensure forward compatibility of generated graphs (see `forward_compatible`)
  99   with older binaries, new features can be gated with:
 100 
 101   ```python
 102   if compat.forward_compatible(year=2018, month=08, date=01):
 103     generate_graph_with_new_features()
 104   else:
 105     generate_graph_so_older_binaries_can_consume_it()
 106   ```
 107 
 108   However, when adding new features, one may want to unittest it before
 109   the forward compatibility window expires. This context manager enables
 110   such tests. For example:
 111 
 112   ```python
 113   from tensorflow.python.compat import compat
 114 
 115   def testMyNewFeature(self):
 116     with compat.forward_compatibility_horizon(2018, 08, 02):
 117        # Test that generate_graph_with_new_features() has an effect
 118   ```
 119 
 120   Args :
 121     year:  A year (e.g. 2018).
 122     month: A month (1 <= month <= 12) in year.
 123     day:   A day (1 <= day <= 31, or 30, or 29, or 28) in month.
 124 
 125   Yields:
 126     Nothing.
 127   """
 128   global _FORWARD_COMPATIBILITY_HORIZON
 129   try:
 130     old_compat_date = _FORWARD_COMPATIBILITY_HORIZON
 131     _FORWARD_COMPATIBILITY_HORIZON = datetime.date(year, month, day)
 132     yield
 133   finally:
 134     _FORWARD_COMPATIBILITY_HORIZON = old_compat_date
