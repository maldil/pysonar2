  51   def __init__(self, profile_datum_list, time_unit=cli_shared.TIME_UNIT_US):
  52     """Constructor.
  53 
  54     Args:
  55       profile_datum_list: List of `ProfileDatum` objects.
  56       time_unit: must be in cli_shared.TIME_UNITS.
  57     """
  58     self._profile_datum_list = profile_datum_list
  59     self.formatted_start_time = [
  60         datum.start_time for datum in profile_datum_list]
  61     self.formatted_op_time = [
  62         cli_shared.time_to_readable_str(datum.op_time,
  63                                         force_time_unit=time_unit)
  64         for datum in profile_datum_list]
  65     self.formatted_exec_time = [
  66         cli_shared.time_to_readable_str(
  67             datum.node_exec_stats.all_end_rel_micros,
  68             force_time_unit=time_unit)
  69         for datum in profile_datum_list]
  70 
  71     self._column_names = ["Node",
  72                           "Op Type",
  73                           "Start Time (us)",
  74                           "Op Time (%s)" % time_unit,
  75                           "Exec Time (%s)" % time_unit,
  76                           "Filename:Lineno(function)"]
  77     self._column_sort_ids = [SORT_OPS_BY_OP_NAME, SORT_OPS_BY_OP_TYPE,
  78                              SORT_OPS_BY_START_TIME, SORT_OPS_BY_OP_TIME,
  79                              SORT_OPS_BY_EXEC_TIME, SORT_OPS_BY_LINE]

  81   def value(self,
  82             row,
  83             col,
  84             device_name_filter=None,
  85             node_name_filter=None,
  86             op_type_filter=None):
  87     """Get the content of a cell of the table.
  88 
  89     Args:
  90       row: (int) row index.
  91       col: (int) column index.
  92       device_name_filter: Regular expression to filter by device name.
  93       node_name_filter: Regular expression to filter by node name.
  94       op_type_filter: Regular expression to filter by op type.
  95 
  96     Returns:
  97       A debuggre_cli_common.RichLine object representing the content of the
  98       cell, potentially with a clickable MenuItem.
  99 
 100     Raises:
 101       IndexError: if row index is out of range.
 102     """
 103     menu_item = None
 104     if col == 0:
 105       text = self._profile_datum_list[row].node_exec_stats.node_name
 106     elif col == 1:
 107       text = self._profile_datum_list[row].op_type
 108     elif col == 2:
 109       text = str(self.formatted_start_time[row])
 110     elif col == 3:
 111       text = str(self.formatted_op_time[row])
 112     elif col == 4:
 113       text = str(self.formatted_exec_time[row])
 114     elif col == 5:
 115       command = "ps"
 116       if device_name_filter:
 117         command += " --%s %s" % (_DEVICE_NAME_FILTER_FLAG,
 118                                  device_name_filter)
 119       if node_name_filter:
 120         command += " --%s %s" % (_NODE_NAME_FILTER_FLAG, node_name_filter)
 121       if op_type_filter:
 122         command += " --%s %s" % (_OP_TYPE_FILTER_FLAG, op_type_filter)
 123       command += " %s --init_line %d" % (
 124           self._profile_datum_list[row].file_path,
 125           self._profile_datum_list[row].line_number)
 126       menu_item = debugger_cli_common.MenuItem(None, command)
 127       text = self._profile_datum_list[row].file_line_func
 128     else:
 129       raise IndexError("Invalid column index %d." % col)
 130 
 131     return RL(text, font_attr=menu_item)

 146 def _list_profile_filter(
 147     profile_datum,
 148     node_name_regex,
 149     file_path_regex,
 150     op_type_regex,
 151     op_time_interval,
 152     exec_time_interval,
 153     min_lineno=-1,
 154     max_lineno=-1):
 155   """Filter function for list_profile command.
 156 
 157   Args:
 158     profile_datum: A `ProfileDatum` object.
 159     node_name_regex: Regular expression pattern object to filter by name.
 160     file_path_regex: Regular expression pattern object to filter by file path.
 161     op_type_regex: Regular expression pattern object to filter by op type.
 162     op_time_interval: `Interval` for filtering op time.
 163     exec_time_interval: `Interval` for filtering exec time.
 164     min_lineno: Lower bound for 1-based line number, inclusive.
 165       If <= 0, has no effect.
 166     max_lineno: Upper bound for 1-based line number, exclusive.
 167       If <= 0, has no effect.
 168     # TODO(cais): Maybe filter by function name.
 169 
 170   Returns:
 171     True iff profile_datum should be included.
 172   """
 173   if node_name_regex and not node_name_regex.match(
 174       profile_datum.node_exec_stats.node_name):
 175     return False
 176   if file_path_regex:
 177     if (not profile_datum.file_path or
 178         not file_path_regex.match(profile_datum.file_path)):
 179       return False
 180   if (min_lineno > 0 and profile_datum.line_number and
 181       profile_datum.line_number < min_lineno):
 182     return False
 183   if (max_lineno > 0 and profile_datum.line_number and
 184       profile_datum.line_number >= max_lineno):
 185     return False
 186   if (profile_datum.op_type is not None and op_type_regex and
 187       not op_type_regex.match(profile_datum.op_type)):
 188     return False
 189   if op_time_interval is not None and not op_time_interval.contains(
 190       profile_datum.op_time):
 191     return False
 192   if exec_time_interval and not exec_time_interval.contains(
 193       profile_datum.node_exec_stats.all_end_rel_micros):
 194     return False
 195   return True

 198 def _list_profile_sort_key(profile_datum, sort_by):
 199   """Get a profile_datum property to sort by in list_profile command.
 200 
 201   Args:
 202     profile_datum: A `ProfileDatum` object.
 203     sort_by: (string) indicates a value to sort by.
 204       Must be one of SORT_BY* constants.
 205 
 206   Returns:
 207     profile_datum property to sort by.
 208   """
 209   if sort_by == SORT_OPS_BY_OP_NAME:
 210     return profile_datum.node_exec_stats.node_name
 211   elif sort_by == SORT_OPS_BY_OP_TYPE:
 212     return profile_datum.op_type
 213   elif sort_by == SORT_OPS_BY_LINE:
 214     return profile_datum.file_line_func
 215   elif sort_by == SORT_OPS_BY_OP_TIME:
 216     return profile_datum.op_time
 217   elif sort_by == SORT_OPS_BY_EXEC_TIME:
 218     return profile_datum.node_exec_stats.all_end_rel_micros
 219   else:  # sort by start time
 220     return profile_datum.node_exec_stats.all_start_micros

 382   def list_profile(self, args, screen_info=None):
 383     """Command handler for list_profile.
 384 
 385     List per-operation profile information.
 386 
 387     Args:
 388       args: Command-line arguments, excluding the command prefix, as a list of
 389         str.
 390       screen_info: Optional dict input containing screen information such as
 391         cols.
 392 
 393     Returns:
 394       Output text lines as a RichTextLines object.
 395     """
 396     screen_cols = 80
 397     if screen_info and "cols" in screen_info:
 398       screen_cols = screen_info["cols"]
 399 
 400     parsed = self._arg_parsers["list_profile"].parse_args(args)
 401     op_time_interval = (command_parser.parse_time_interval(parsed.op_time)
 402                         if parsed.op_time else None)
 403     exec_time_interval = (
 404         command_parser.parse_time_interval(parsed.execution_time)
 405         if parsed.execution_time else None)
 406     node_name_regex = (re.compile(parsed.node_name_filter)
 407                        if parsed.node_name_filter else None)
 408     file_path_regex = (re.compile(parsed.file_path_filter)
 409                        if parsed.file_path_filter else None)
 410     op_type_regex = (re.compile(parsed.op_type_filter)
 411                      if parsed.op_type_filter else None)
 412 
 413     output = debugger_cli_common.RichTextLines([""])
 414     device_name_regex = (re.compile(parsed.device_name_filter)
 415                          if parsed.device_name_filter else None)
 416     data_generator = self._get_profile_data_generator()
 417     device_count = len(self._run_metadata.step_stats.dev_stats)
 418     for index in range(device_count):
 419       device_stats = self._run_metadata.step_stats.dev_stats[index]
 420       if not device_name_regex or device_name_regex.match(device_stats.device):
 421         profile_data = [
 422             datum for datum in data_generator(device_stats)
 423             if _list_profile_filter(
 424                 datum, node_name_regex, file_path_regex, op_type_regex,
 425                 op_time_interval, exec_time_interval,
 426                 min_lineno=parsed.min_lineno, max_lineno=parsed.max_lineno)]
 427         profile_data = sorted(
 428             profile_data,
 429             key=lambda datum: _list_profile_sort_key(datum, parsed.sort_by),
 430             reverse=parsed.reverse)
 431         output.extend(
 432             self._get_list_profile_lines(
 433                 device_stats.device, index, device_count,
 434                 profile_data, parsed.sort_by, parsed.reverse, parsed.time_unit,
 435                 device_name_filter=parsed.device_name_filter,
 436                 node_name_filter=parsed.node_name_filter,
 437                 op_type_filter=parsed.op_type_filter,
 438                 screen_cols=screen_cols))
 439     return output

 441   def _get_profile_data_generator(self):
 442     """Get function that generates `ProfileDatum` objects.
 443 
 444     Returns:
 445       A function that generates `ProfileDatum` objects.
 446     """
 447     node_to_file_path = {}
 448     node_to_line_number = {}
 449     node_to_func_name = {}
 450     node_to_op_type = {}
 451     for op in self._graph.get_operations():
 452       for trace_entry in reversed(op.traceback):
 453         file_path = trace_entry[0]
 454         line_num = trace_entry[1]
 455         func_name = trace_entry[2]
 456         if not source_utils.guess_is_tensorflow_py_library(file_path):
 457           break
 458       node_to_file_path[op.name] = file_path
 459       node_to_line_number[op.name] = line_num
 460       node_to_func_name[op.name] = func_name
 461       node_to_op_type[op.name] = op.type
 462 
 463     def profile_data_generator(device_step_stats):
 464       for node_stats in device_step_stats.node_stats:
 465         if node_stats.node_name == "_SOURCE" or node_stats.node_name == "_SINK":
 466           continue
 467         yield profiling.ProfileDatum(
 468             device_step_stats.device,
 469             node_stats,
 470             node_to_file_path.get(node_stats.node_name, ""),
 471             node_to_line_number.get(node_stats.node_name, 0),
 472             node_to_func_name.get(node_stats.node_name, ""),
 473             node_to_op_type.get(node_stats.node_name, ""))
 474     return profile_data_generator

 476   def _get_list_profile_lines(
 477       self, device_name, device_index, device_count,
 478       profile_datum_list, sort_by, sort_reverse, time_unit,
 479       device_name_filter=None, node_name_filter=None, op_type_filter=None,
 480       screen_cols=80):
 481     """Get `RichTextLines` object for list_profile command for a given device.
 482 
 483     Args:
 484       device_name: (string) Device name.
 485       device_index: (int) Device index.
 486       device_count: (int) Number of devices.
 487       profile_datum_list: List of `ProfileDatum` objects.
 488       sort_by: (string) Identifier of column to sort. Sort identifier
 489           must match value of SORT_OPS_BY_OP_NAME, SORT_OPS_BY_OP_TYPE,
 490           SORT_OPS_BY_EXEC_TIME, SORT_OPS_BY_MEMORY or SORT_OPS_BY_LINE.
 491       sort_reverse: (bool) Whether to sort in descending instead of default
 492           (ascending) order.
 493       time_unit: time unit, must be in cli_shared.TIME_UNITS.
 494       device_name_filter: Regular expression to filter by device name.
 495       node_name_filter: Regular expression to filter by node name.
 496       op_type_filter: Regular expression to filter by op type.
 497       screen_cols: (int) Number of columns available on the screen (i.e.,
 498         available screen width).
 499 
 500     Returns:
 501       `RichTextLines` object containing a table that displays profiling
 502       information for each op.
 503     """
 504     profile_data = ProfileDataTableView(profile_datum_list, time_unit=time_unit)
 505 
 506     # Calculate total time early to calculate column widths.
 507     total_op_time = sum(datum.op_time for datum in profile_datum_list)
 508     total_exec_time = sum(datum.node_exec_stats.all_end_rel_micros
 509                           for datum in profile_datum_list)
 510     device_total_row = [
 511         "Device Total", "",
 512         cli_shared.time_to_readable_str(total_op_time,
 513                                         force_time_unit=time_unit),
 514         cli_shared.time_to_readable_str(total_exec_time,
 515                                         force_time_unit=time_unit)]
 516 
 517     # Calculate column widths.
 518     column_widths = [
 519         len(column_name) for column_name in profile_data.column_names()]
 520     for col in range(len(device_total_row)):
 521       column_widths[col] = max(column_widths[col], len(device_total_row[col]))
 522     for col in range(len(column_widths)):
 523       for row in range(profile_data.row_count()):
 524         column_widths[col] = max(
 525             column_widths[col], len(profile_data.value(
 526                 row,
 527                 col,
 528                 device_name_filter=device_name_filter,
 529                 node_name_filter=node_name_filter,
 530                 op_type_filter=op_type_filter)))
 531       column_widths[col] += 2  # add margin between columns
 532 
 533     # Add device name.
 534     output = [RL("-" * screen_cols)]
 535     device_row = "Device %d of %d: %s" % (
 536         device_index + 1, device_count, device_name)
 537     output.append(RL(device_row))
 538     output.append(RL())
 539 
 540     # Add headers.
 541     base_command = "list_profile"
 542     row = RL()
 543     for col in range(profile_data.column_count()):
 544       column_name = profile_data.column_names()[col]
 545       sort_id = profile_data.column_sort_id(col)
 546       command = "%s -s %s" % (base_command, sort_id)
 547       if sort_by == sort_id and not sort_reverse:
 548         command += " -r"
 549       head_menu_item = debugger_cli_common.MenuItem(None, command)
 550       row += RL(column_name, font_attr=[head_menu_item, "bold"])
 551       row += RL(" " * (column_widths[col] - len(column_name)))
 552 
 553     output.append(row)
 554 
 555     # Add data rows.
 556     for row in range(profile_data.row_count()):
 557       new_row = RL()
 558       for col in range(profile_data.column_count()):
 559         new_cell = profile_data.value(
 560             row,
 561             col,
 562             device_name_filter=device_name_filter,
 563             node_name_filter=node_name_filter,
 564             op_type_filter=op_type_filter)
 565         new_row += new_cell
 566         new_row += RL(" " * (column_widths[col] - len(new_cell)))
 567       output.append(new_row)
 568 
 569     # Add stat totals.
 570     row_str = ""
 571     for col in range(len(device_total_row)):
 572       row_str += ("{:<%d}" % column_widths[col]).format(device_total_row[col])
 573     output.append(RL())
 574     output.append(RL(row_str))
 575     return debugger_cli_common.rich_text_lines_from_rich_line_list(output)

 577   def _measure_list_profile_column_widths(self, profile_data):
 578     """Determine the maximum column widths for each data list.
 579 
 580     Args:
 581       profile_data: list of ProfileDatum objects.
 582 
 583     Returns:
 584       List of column widths in the same order as columns in data.
 585     """
 586     num_columns = len(profile_data.column_names())
 587     widths = [len(column_name) for column_name in profile_data.column_names()]
 588     for row in range(profile_data.row_count()):
 589       for col in range(num_columns):
 590         widths[col] = max(
 591             widths[col], len(str(profile_data.row_values(row)[col])) + 2)
 592     return widths
 593 
 594   _LINE_COST_ATTR = cli_shared.COLOR_CYAN
 595   _LINE_NUM_ATTR = cli_shared.COLOR_YELLOW
 596   _NUM_NODES_HEAD = "#nodes"
 597   _NUM_EXECS_SUB_HEAD = "(#execs)"
 598   _LINENO_HEAD = "lineno"
 599   _SOURCE_HEAD = "source"

 601   def print_source(self, args, screen_info=None):
 602     """Print a Python source file with line-level profile information.
 603 
 604     Args:
 605       args: Command-line arguments, excluding the command prefix, as a list of
 606         str.
 607       screen_info: Optional dict input containing screen information such as
 608         cols.
 609 
 610     Returns:
 611       Output text lines as a RichTextLines object.
 612     """
 613     del screen_info
 614 
 615     parsed = self._arg_parsers["print_source"].parse_args(args)
 616 
 617     device_name_regex = (re.compile(parsed.device_name_filter)
 618                          if parsed.device_name_filter else None)
 619 
 620     profile_data = []
 621     data_generator = self._get_profile_data_generator()
 622     device_count = len(self._run_metadata.step_stats.dev_stats)
 623     for index in range(device_count):
 624       device_stats = self._run_metadata.step_stats.dev_stats[index]
 625       if device_name_regex and not device_name_regex.match(device_stats.device):
 626         continue
 627       profile_data.extend([datum for datum in data_generator(device_stats)])
 628 
 629     source_annotation = source_utils.annotate_source_against_profile(
 630         profile_data,
 631         os.path.expanduser(parsed.source_file_path),
 632         node_name_filter=parsed.node_name_filter,
 633         op_type_filter=parsed.op_type_filter)
 634     if not source_annotation:
 635       return debugger_cli_common.RichTextLines(
 636           ["The source file %s does not contain any profile information for "
 637            "the previous Session run under the following "
 638            "filters:" % parsed.source_file_path,
 639            "  --%s: %s" % (_DEVICE_NAME_FILTER_FLAG, parsed.device_name_filter),
 640            "  --%s: %s" % (_NODE_NAME_FILTER_FLAG, parsed.node_name_filter),
 641            "  --%s: %s" % (_OP_TYPE_FILTER_FLAG, parsed.op_type_filter)])
 642 
 643     max_total_cost = 0
 644     for line_index in source_annotation:
 645       total_cost = self._get_total_cost(source_annotation[line_index],
 646                                         parsed.cost_type)
 647       max_total_cost = max(max_total_cost, total_cost)
 648 
 649     source_lines, line_num_width = source_utils.load_source(
 650         parsed.source_file_path)
 651 
 652     cost_bar_max_length = 10
 653     total_cost_head = parsed.cost_type
 654     column_widths = {
 655         "cost_bar": cost_bar_max_length + 3,
 656         "total_cost": len(total_cost_head) + 3,
 657         "num_nodes_execs": len(self._NUM_EXECS_SUB_HEAD) + 1,
 658         "line_number": line_num_width,
 659     }
 660 
 661     head = RL(
 662         " " * column_widths["cost_bar"] +
 663         total_cost_head +
 664         " " * (column_widths["total_cost"] - len(total_cost_head)) +
 665         self._NUM_NODES_HEAD +
 666         " " * (column_widths["num_nodes_execs"] - len(self._NUM_NODES_HEAD)),
 667         font_attr=self._LINE_COST_ATTR)
 668     head += RL(self._LINENO_HEAD, font_attr=self._LINE_NUM_ATTR)
 669     sub_head = RL(
 670         " " * (column_widths["cost_bar"] +
 671                column_widths["total_cost"]) +
 672         self._NUM_EXECS_SUB_HEAD +
 673         " " * (column_widths["num_nodes_execs"] -
 674                len(self._NUM_EXECS_SUB_HEAD)) +
 675         " " * column_widths["line_number"],
 676         font_attr=self._LINE_COST_ATTR)
 677     sub_head += RL(self._SOURCE_HEAD, font_attr="bold")
 678     lines = [head, sub_head]
 679 
 680     output_annotations = {}
 681     for i, line in enumerate(source_lines):
 682       lineno = i + 1
 683       if lineno in source_annotation:
 684         annotation = source_annotation[lineno]
 685         cost_bar = self._render_normalized_cost_bar(
 686             self._get_total_cost(annotation, parsed.cost_type), max_total_cost,
 687             cost_bar_max_length)
 688         annotated_line = cost_bar
 689         annotated_line += " " * (column_widths["cost_bar"] - len(cost_bar))
 690 
 691         total_cost = RL(cli_shared.time_to_readable_str(
 692             self._get_total_cost(annotation, parsed.cost_type),
 693             force_time_unit=parsed.time_unit),
 694                         font_attr=self._LINE_COST_ATTR)
 695         total_cost += " " * (column_widths["total_cost"] - len(total_cost))
 696         annotated_line += total_cost
 697 
 698         file_path_filter = re.escape(parsed.source_file_path) + "$"
 699         command = "lp --file_path_filter %s --min_lineno %d --max_lineno %d" % (
 700             file_path_filter, lineno, lineno + 1)
 701         if parsed.device_name_filter:
 702           command += " --%s %s" % (_DEVICE_NAME_FILTER_FLAG,
 703                                    parsed.device_name_filter)
 704         if parsed.node_name_filter:
 705           command += " --%s %s" % (_NODE_NAME_FILTER_FLAG,
 706                                    parsed.node_name_filter)
 707         if parsed.op_type_filter:
 708           command += " --%s %s" % (_OP_TYPE_FILTER_FLAG,
 709                                    parsed.op_type_filter)
 710         menu_item = debugger_cli_common.MenuItem(None, command)
 711         num_nodes_execs = RL("%d(%d)" % (annotation.node_count,
 712                                          annotation.node_exec_count),
 713                              font_attr=[self._LINE_COST_ATTR, menu_item])
 714         num_nodes_execs += " " * (
 715             column_widths["num_nodes_execs"] - len(num_nodes_execs))
 716         annotated_line += num_nodes_execs
 717       else:
 718         annotated_line = RL(
 719             " " * sum(column_widths[col_name] for col_name in column_widths
 720                       if col_name != "line_number"))
 721 
 722       line_num_column = RL(" L%d" % (lineno), self._LINE_NUM_ATTR)
 723       line_num_column += " " * (
 724           column_widths["line_number"] - len(line_num_column))
 725       annotated_line += line_num_column
 726       annotated_line += line
 727       lines.append(annotated_line)
 728 
 729       if parsed.init_line == lineno:
 730         output_annotations[
 731             debugger_cli_common.INIT_SCROLL_POS_KEY] = len(lines) - 1
 732 
 733     return debugger_cli_common.rich_text_lines_from_rich_line_list(
 734         lines, annotations=output_annotations)
 735 
 736   def _get_total_cost(self, aggregated_profile, cost_type):
 737     if cost_type == "exec_time":
 738       return aggregated_profile.total_exec_time
 739     elif cost_type == "op_time":
 740       return aggregated_profile.total_op_time
 741     else:
 742       raise ValueError("Unsupported cost type: %s" % cost_type)
 743 
 744   def _render_normalized_cost_bar(self, cost, max_cost, length):
 745     """Render a text bar representing a normalized cost.
 746 
 747     Args:
 748       cost: the absolute value of the cost.
 749       max_cost: the maximum cost value to normalize the absolute cost with.
 750       length: (int) length of the cost bar, in number of characters, excluding
 751         the brackets on the two ends.
 752 
 753     Returns:
 754       An instance of debugger_cli_common.RichTextLine.
 755     """
 756     num_ticks = int(np.ceil(float(cost) / max_cost * length))
 757     num_ticks = num_ticks or 1  # Minimum is 1 tick.
 758     output = RL("[", font_attr=self._LINE_COST_ATTR)
 759     output += RL("|" * num_ticks + " " * (length - num_ticks),
 760                  font_attr=["bold", self._LINE_COST_ATTR])
 761     output += RL("]", font_attr=self._LINE_COST_ATTR)
 762     return output
