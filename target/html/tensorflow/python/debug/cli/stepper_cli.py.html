 197   def _initialize_state(self):
 198     """Initialize the state of this stepper CLI."""
 199 
 200     # Get the elements in the sorted transitive closure, as a list of str.
 201     self._sorted_nodes = self._node_stepper.sorted_nodes()
 202     self._closure_elements = self._node_stepper.closure_elements()
 203     self._placeholders = self._node_stepper.placeholders()
 204     self._completed_nodes = set()
 205 
 206     self._calculate_next()
 207 
 208   def _calculate_next(self):
 209     """Calculate the next target for "step" action based on current state."""
 210 
 211     override_names = self._node_stepper.override_names()
 212 
 213     next_i = -1
 214     for i in xrange(len(self._sorted_nodes)):
 215       if (i > next_i and (self._sorted_nodes[i] in self._completed_nodes) or
 216           (self._sorted_nodes[i] in override_names)):
 217         next_i = i
 218 
 219     next_i += 1
 220     self._next = next_i

 222   def list_sorted_nodes(self, args, screen_info=None):
 223     """List the sorted transitive closure of the stepper's fetches."""
 224 
 225     # TODO(cais): Use pattern such as del args, del screen_info python/debug.
 226     _ = args
 227     _ = screen_info
 228 
 229     parsed = self.arg_parsers["list_sorted_nodes"].parse_args(args)
 230 
 231     if parsed.lower_bound != -1 and parsed.upper_bound != -1:
 232       index_range = [
 233           max(0, parsed.lower_bound),
 234           min(len(self._sorted_nodes), parsed.upper_bound)
 235       ]
 236       verbose = False
 237     else:
 238       index_range = [0, len(self._sorted_nodes)]
 239       verbose = True
 240 
 241     handle_node_names = self._node_stepper.handle_node_names()
 242     intermediate_tensor_names = self._node_stepper.intermediate_tensor_names()
 243     override_names = self._node_stepper.override_names()
 244     dirty_variable_names = [
 245         dirty_variable.split(":")[0]
 246         for dirty_variable in self._node_stepper.dirty_variables()
 247     ]
 248 
 249     lines = []
 250     if verbose:
 251       lines.extend(
 252           ["Topologically-sorted transitive input(s) and fetch(es):", ""])
 253 
 254     for i, element_name in enumerate(self._sorted_nodes):
 255       if i < index_range[0] or i >= index_range[1]:
 256         continue
 257 
 258       # TODO(cais): Use fixed-width text to show node index.
 259       if i == self._next:
 260         node_prefix = RL("  ") + RL(self.NEXT_NODE_POINTER_STR, "bold")
 261       else:
 262         node_prefix = RL("     ")
 263 
 264       node_prefix += "(%d / %d)" % (i + 1, len(self._sorted_nodes)) + "  ["
 265       node_prefix += self._get_status_labels(
 266           element_name,
 267           handle_node_names,
 268           intermediate_tensor_names,
 269           override_names,
 270           dirty_variable_names)
 271 
 272       lines.append(node_prefix + "] " + element_name)
 273 
 274     output = debugger_cli_common.rich_text_lines_from_rich_line_list(lines)
 275 
 276     if verbose:
 277       output.extend(self._node_status_label_legend())
 278 
 279     return output

 281   def _get_status_labels(self,
 282                          element_name,
 283                          handle_node_names,
 284                          intermediate_tensor_names,
 285                          override_names,
 286                          dirty_variable_names):
 287     """Get a string of status labels for a graph element.
 288 
 289     A status label indicates that a node has a certain state in this
 290     node-stepper CLI invocation. For example, 1) that the node has been
 291     continued-to and a handle to its output tensor is available to the node
 292     stepper; 2) the node is a Variable and its value has been altered, e.g.,
 293     by continuing to a variable-updating node, since the beginning of this
 294     node-stepper invocation (i.e., "dirty variable").
 295 
 296     Args:
 297       element_name: (str) name of the graph element.
 298       handle_node_names: (list of str) Names of the nodes of which the output
 299         tensors' handles are available.
 300       intermediate_tensor_names: (list of str) Names of the intermediate tensor
 301         dumps generated from the graph element.
 302       override_names: (list of str) Names of the tensors of which the values
 303         are overridden.
 304       dirty_variable_names: (list of str) Names of the dirty variables.
 305 
 306     Returns:
 307       (RichLine) The rich text string of status labels that currently apply to
 308         the graph element.
 309     """
 310 
 311     status = RL()
 312 
 313     node_name = element_name.split(":")[0]
 314     status += (RL(self.STATE_IS_PLACEHOLDER,
 315                   self._STATE_COLORS[self.STATE_IS_PLACEHOLDER])
 316                if node_name in self._placeholders else " ")
 317     status += (RL(self.STATE_UNFEEDABLE,
 318                   self._STATE_COLORS[self.STATE_UNFEEDABLE])
 319                if not self._node_stepper.is_feedable(str(element_name))
 320                else " ")
 321     status += (RL(self.STATE_CONT, self._STATE_COLORS[self.STATE_CONT])
 322                if element_name in handle_node_names else " ")
 323 
 324     intermediate_node_names = [
 325         tensor_name.split(":")[0] for tensor_name in intermediate_tensor_names]
 326     status += (RL(self.STATE_DUMPED_INTERMEDIATE,
 327                   self._STATE_COLORS[self.STATE_DUMPED_INTERMEDIATE])
 328                if element_name in intermediate_node_names else " ")
 329 
 330     slots = self._node_stepper.output_slots_in_closure(element_name)
 331     has_override = any(element_name + ":%d" % slot in override_names
 332                        for slot in slots)
 333     status += (RL(self.STATE_OVERRIDDEN,
 334                   self._STATE_COLORS[self.STATE_OVERRIDDEN])
 335                if has_override else " ")
 336     status += (RL(self.STATE_DIRTY_VARIABLE,
 337                   self._STATE_COLORS[self.STATE_DIRTY_VARIABLE])
 338                if element_name in dirty_variable_names else " ")
 339 
 340     return status

 378   def cont(self, args, screen_info=None):
 379     """Continue-to action on the graph."""
 380 
 381     _ = screen_info
 382 
 383     parsed = self.arg_parsers["cont"].parse_args(args)
 384 
 385     # Determine which node is being continued to, so the _next pointer can be
 386     # set properly.
 387     node_name = parsed.target_name.split(":")[0]
 388     if node_name not in self._sorted_nodes:
 389       return cli_shared.error(self._MESSAGE_TEMPLATES["NOT_IN_CLOSURE"] %
 390                               parsed.target_name)
 391     self._next = self._sorted_nodes.index(node_name)
 392 
 393     cont_result = self._node_stepper.cont(
 394         parsed.target_name,
 395         invalidate_from_updated_variables=(
 396             parsed.invalidate_from_updated_variables),
 397         restore_variable_values=parsed.restore_variable_values)
 398     self._completed_nodes.add(parsed.target_name.split(":")[0])
 399 
 400     screen_output = debugger_cli_common.RichTextLines(
 401         ["Continued to %s:" % parsed.target_name, ""])
 402     screen_output.extend(self._report_last_feed_types())
 403     screen_output.extend(self._report_last_updated())
 404     screen_output.extend(
 405         tensor_format.format_tensor(
 406             cont_result, parsed.target_name, include_metadata=True))
 407 
 408     # Generate windowed view of the sorted transitive closure on which the
 409     # stepping is occurring.
 410     lower_bound = max(0, self._next - 2)
 411     upper_bound = min(len(self._sorted_nodes), self._next + 3)
 412 
 413     final_output = self.list_sorted_nodes(
 414         ["-l", str(lower_bound), "-u", str(upper_bound)])
 415     final_output.extend(debugger_cli_common.RichTextLines([""]))
 416     final_output.extend(screen_output)
 417 
 418     # Re-calculate the target of the next "step" action.
 419     self._calculate_next()
 420 
 421     return final_output

 423   def _report_last_feed_types(self):
 424     """Generate a report of the feed types used in the cont/step call.
 425 
 426     Returns:
 427       (debugger_cli_common.RichTextLines) A RichTextLines representation of the
 428         feeds used in the last cont/step call.
 429     """
 430     feed_types = self._node_stepper.last_feed_types()
 431 
 432     out = ["Stepper used feeds:"]
 433     if feed_types:
 434       for feed_name in feed_types:
 435         feed_info = RL("  %s : " % feed_name)
 436         feed_info += RL(feed_types[feed_name],
 437                         self._FEED_COLORS[feed_types[feed_name]])
 438         out.append(feed_info)
 439     else:
 440       out.append("  (No feeds)")
 441     out.append("")
 442 
 443     return debugger_cli_common.rich_text_lines_from_rich_line_list(out)

 445   def _report_last_updated(self):
 446     """Generate a report of the variables updated in the last cont/step call.
 447 
 448     Returns:
 449       (debugger_cli_common.RichTextLines) A RichTextLines representation of the
 450         variables updated in the last cont/step call.
 451     """
 452 
 453     last_updated = self._node_stepper.last_updated()
 454     if not last_updated:
 455       return debugger_cli_common.RichTextLines([])
 456 
 457     rich_lines = [RL("Updated:", self._UPDATED_ATTRIBUTE)]
 458     sorted_last_updated = sorted(list(last_updated))
 459     for updated in sorted_last_updated:
 460       rich_lines.append("  %s" % updated)
 461     rich_lines.append("")
 462     return debugger_cli_common.rich_text_lines_from_rich_line_list(rich_lines)

 464   def step(self, args, screen_info=None):
 465     """Step once.
 466 
 467     Args:
 468       args: (list of str) command-line arguments for the "step" command.
 469       screen_info: Information about screen.
 470 
 471     Returns:
 472       (RichTextLines) Screen output for the result of the stepping action.
 473     """
 474 
 475     parsed = self.arg_parsers["step"].parse_args(args)
 476 
 477     if parsed.num_times < 0:
 478       return debugger_cli_common.RichTextLines(
 479           "ERROR: Invalid number of times to step: %d" % parsed.num_times)
 480 
 481     for _ in xrange(parsed.num_times):
 482       if self._next >= len(self._sorted_nodes):
 483         return debugger_cli_common.RichTextLines(
 484             "ERROR: Cannot step any further because the end of the sorted "
 485             "transitive closure has been reached.")
 486       else:
 487         screen_output = self.cont([self._sorted_nodes[self._next]], screen_info)
 488 
 489     return screen_output

 491   def print_tensor(self, args, screen_info=None):
 492     """Print the value of a tensor that the stepper has access to."""
 493 
 494     parsed = self.arg_parsers["print_tensor"].parse_args(args)
 495 
 496     if screen_info and "cols" in screen_info:
 497       np_printoptions = {"linewidth": screen_info["cols"]}
 498     else:
 499       np_printoptions = {}
 500 
 501     # Determine if any range-highlighting is required.
 502     highlight_options = cli_shared.parse_ranges_highlight(parsed.ranges)
 503 
 504     tensor_name, tensor_slicing = (
 505         command_parser.parse_tensor_name_with_slicing(parsed.tensor_name))
 506 
 507     tensor_names = self._resolve_tensor_names(tensor_name)
 508     if not tensor_names:
 509       return cli_shared.error(
 510           self._MESSAGE_TEMPLATES["NOT_IN_CLOSURE"] % tensor_name)
 511     elif len(tensor_names) > 1:
 512       return cli_shared.error(
 513           self._MESSAGE_TEMPLATES["MULTIPLE_TENSORS"] % tensor_name)
 514     else:
 515       tensor_name = tensor_names[0]
 516 
 517     try:
 518       tensor_value = self._node_stepper.get_tensor_value(tensor_name)
 519     except ValueError as e:
 520       return debugger_cli_common.RichTextLines([str(e)])
 521 
 522     return cli_shared.format_tensor(
 523         tensor_value,
 524         tensor_name,
 525         np_printoptions,
 526         print_all=parsed.print_all,
 527         tensor_slicing=tensor_slicing,
 528         highlight_options=highlight_options)

 530   def inject_value(self, args, screen_info=None):
 531     """Inject value to a given tensor.
 532 
 533     Args:
 534       args: (list of str) command-line arguments for the "step" command.
 535       screen_info: Information about screen.
 536 
 537     Returns:
 538       (RichTextLines) Screen output for the result of the stepping action.
 539     """
 540 
 541     _ = screen_info  # Currently unused.
 542 
 543     if screen_info and "cols" in screen_info:
 544       np_printoptions = {"linewidth": screen_info["cols"]}
 545     else:
 546       np_printoptions = {}
 547 
 548     parsed = self.arg_parsers["inject_value"].parse_args(args)
 549 
 550     tensor_names = self._resolve_tensor_names(parsed.tensor_name)
 551     if not tensor_names:
 552       return cli_shared.error(
 553           self._MESSAGE_TEMPLATES["NOT_IN_CLOSURE"] % parsed.tensor_name)
 554     elif len(tensor_names) > 1:
 555       return cli_shared.error(
 556           self._MESSAGE_TEMPLATES["MULTIPLE_TENSORS"] % parsed.tensor_name)
 557     else:
 558       tensor_name = tensor_names[0]
 559 
 560     tensor_value = eval(parsed.tensor_value_str)  # pylint: disable=eval-used
 561 
 562     try:
 563       self._node_stepper.override_tensor(tensor_name, tensor_value)
 564       lines = [
 565           "Injected value \"%s\"" % parsed.tensor_value_str,
 566           "  to tensor \"%s\":" % tensor_name, ""
 567       ]
 568 
 569       tensor_lines = tensor_format.format_tensor(
 570           tensor_value,
 571           tensor_name,
 572           include_metadata=True,
 573           np_printoptions=np_printoptions).lines
 574       lines.extend(tensor_lines)
 575 
 576     except ValueError:
 577       lines = [
 578           "ERROR: Failed to inject value to tensor %s" % parsed.tensor_name
 579       ]
 580 
 581     return debugger_cli_common.RichTextLines(lines)

 587   def _resolve_tensor_names(self, element_name):
 588     """Resolve tensor name from graph element name.
 589 
 590     Args:
 591       element_name: (str) Name of the graph element to resolve.
 592 
 593     Returns:
 594       (list) Name of the tensor(s). If element_name is the name of a tensor in
 595       the transitive closure, return [element_name]. If element_name is the
 596       name of a node in the transitive closure, return the list of output
 597       tensors from the node that are in the transitive closure. Otherwise,
 598       return empty list.
 599     """
 600 
 601     if element_name in self._closure_elements and ":" in element_name:
 602       return [element_name]
 603     if (element_name in self._sorted_nodes or
 604         (element_name in self._closure_elements and ":" not in element_name)):
 605       slots = self._node_stepper.output_slots_in_closure(element_name)
 606       return [(element_name + ":%d" % slot) for slot in slots]
 607     else:
 608       return []
