 422   def add_tensor_filter(self, filter_name, filter_callable):
 423     """Add a tensor filter.
 424 
 425     A tensor filter is a named callable of the signature:
 426       filter_callable(dump_datum, tensor),
 427 
 428     wherein dump_datum is an instance of debug_data.DebugTensorDatum carrying
 429     metadata about the dumped tensor, including tensor name, timestamps, etc.
 430     tensor is the value of the dumped tensor as an numpy.ndarray object.
 431     The return value of the function is a bool.
 432     This is the same signature as the input argument to
 433     debug_data.DebugDumpDir.find().
 434 
 435     Args:
 436       filter_name: (str) name of the filter. Cannot be empty.
 437       filter_callable: (callable) a filter function of the signature described
 438         as above.
 439 
 440     Raises:
 441       ValueError: If filter_name is an empty str.
 442       TypeError: If filter_name is not a str.
 443                  Or if filter_callable is not callable.
 444     """
 445 
 446     if not isinstance(filter_name, str):
 447       raise TypeError("Input argument filter_name is expected to be str, "
 448                       "but is not.")
 449 
 450     # Check that filter_name is not an empty str.
 451     if not filter_name:
 452       raise ValueError("Input argument filter_name cannot be empty.")
 453 
 454     # Check that filter_callable is callable.
 455     if not callable(filter_callable):
 456       raise TypeError(
 457           "Input argument filter_callable is expected to be callable, "
 458           "but is not.")
 459 
 460     self._tensor_filters[filter_name] = filter_callable
 461 
 462   def get_tensor_filter(self, filter_name):
 463     """Retrieve filter function by name.
 464 
 465     Args:
 466       filter_name: Name of the filter set during add_tensor_filter() call.
 467 
 468     Returns:
 469       The callable associated with the filter name.
 470 
 471     Raises:
 472       ValueError: If there is no tensor filter of the specified filter name.
 473     """
 474 
 475     if filter_name not in self._tensor_filters:
 476       raise ValueError("There is no tensor filter named \"%s\"" % filter_name)
 477 
 478     return self._tensor_filters[filter_name]

 483   def list_tensors(self, args, screen_info=None):
 484     """Command handler for list_tensors.
 485 
 486     List tensors dumped during debugged Session.run() call.
 487 
 488     Args:
 489       args: Command-line arguments, excluding the command prefix, as a list of
 490         str.
 491       screen_info: Optional dict input containing screen information such as
 492         cols.
 493 
 494     Returns:
 495       Output text lines as a RichTextLines object.
 496 
 497     Raises:
 498       ValueError: If `--filter_exclude_node_names` is used without `-f` or
 499         `--tensor_filter` being used.
 500     """
 501 
 502     # TODO(cais): Add annotations of substrings for dumped tensor names, to
 503     # facilitate on-screen highlighting/selection of node names.
 504     _ = screen_info
 505 
 506     parsed = self._arg_parsers["list_tensors"].parse_args(args)
 507 
 508     output = []
 509 
 510     filter_strs = []
 511     if parsed.op_type_filter:
 512       op_type_regex = re.compile(parsed.op_type_filter)
 513       filter_strs.append("Op type regex filter: \"%s\"" % parsed.op_type_filter)
 514     else:
 515       op_type_regex = None
 516 
 517     if parsed.node_name_filter:
 518       node_name_regex = re.compile(parsed.node_name_filter)
 519       filter_strs.append("Node name regex filter: \"%s\"" %
 520                          parsed.node_name_filter)
 521     else:
 522       node_name_regex = None
 523 
 524     output = debugger_cli_common.RichTextLines(filter_strs)
 525     output.append("")
 526 
 527     if parsed.tensor_filter:
 528       try:
 529         filter_callable = self.get_tensor_filter(parsed.tensor_filter)
 530       except ValueError:
 531         output = cli_shared.error("There is no tensor filter named \"%s\"." %
 532                                   parsed.tensor_filter)
 533         _add_main_menu(output, node_name=None, enable_list_tensors=False)
 534         return output
 535 
 536       data_to_show = self._debug_dump.find(
 537           filter_callable,
 538           exclude_node_names=parsed.filter_exclude_node_names)
 539     else:
 540       if parsed.filter_exclude_node_names:
 541         raise ValueError(
 542             "The flag --filter_exclude_node_names is valid only when "
 543             "the flag -f or --tensor_filter is used.")
 544 
 545       data_to_show = self._debug_dump.dumped_tensor_data
 546 
 547     # TODO(cais): Implement filter by lambda on tensor value.
 548 
 549     max_timestamp_width, max_dump_size_width, max_op_type_width = (
 550         self._measure_tensor_list_column_widths(data_to_show))
 551 
 552     # Sort the data.
 553     data_to_show = self._sort_dump_data_by(
 554         data_to_show, parsed.sort_by, parsed.reverse)
 555 
 556     output.extend(
 557         self._tensor_list_column_heads(parsed, max_timestamp_width,
 558                                        max_dump_size_width, max_op_type_width))
 559 
 560     dump_count = 0
 561     for dump in data_to_show:
 562       if node_name_regex and not node_name_regex.match(dump.node_name):
 563         continue
 564 
 565       if op_type_regex:
 566         op_type = self._debug_dump.node_op_type(dump.node_name)
 567         if not op_type_regex.match(op_type):
 568           continue
 569 
 570       rel_time = (dump.timestamp - self._debug_dump.t0) / 1000.0
 571       dump_size_str = cli_shared.bytes_to_readable_str(dump.dump_size_bytes)
 572       dumped_tensor_name = "%s:%d" % (dump.node_name, dump.output_slot)
 573       op_type = self._debug_dump.node_op_type(dump.node_name)
 574 
 575       line = "[%.3f]" % rel_time
 576       line += " " * (max_timestamp_width - len(line))
 577       line += dump_size_str
 578       line += " " * (max_timestamp_width + max_dump_size_width - len(line))
 579       line += op_type
 580       line += " " * (max_timestamp_width + max_dump_size_width +
 581                      max_op_type_width - len(line))
 582       line += dumped_tensor_name
 583 
 584       output.append(
 585           line,
 586           font_attr_segs=[(
 587               len(line) - len(dumped_tensor_name), len(line),
 588               debugger_cli_common.MenuItem("", "pt %s" % dumped_tensor_name))])
 589       dump_count += 1
 590 
 591     if parsed.tensor_filter:
 592       output.prepend([
 593           "%d dumped tensor(s) passing filter \"%s\":" %
 594           (dump_count, parsed.tensor_filter)
 595       ])
 596     else:
 597       output.prepend(["%d dumped tensor(s):" % dump_count])
 598 
 599     _add_main_menu(output, node_name=None, enable_list_tensors=False)
 600     return output

 602   def _measure_tensor_list_column_widths(self, data):
 603     """Determine the maximum widths of the timestamp and op-type column.
 604 
 605     This method assumes that data is sorted in the default order, i.e.,
 606     by ascending timestamps.
 607 
 608     Args:
 609       data: (list of DebugTensorDaum) the data based on which the maximum
 610         column widths will be determined.
 611 
 612     Returns:
 613       (int) maximum width of the timestamp column. 0 if data is empty.
 614       (int) maximum width of the dump size column. 0 if data is empty.
 615       (int) maximum width of the op type column. 0 if data is empty.
 616     """
 617

1366   def _dfs_from_node(self,
1367                      lines,
1368                      attr_segs,
1369                      node_name,
1370                      tracker,
1371                      max_depth,
1372                      depth,
1373                      unfinished,
1374                      include_control=False,
1375                      show_op_type=False,
1376                      command_template=None):
1377     """Perform depth-first search (DFS) traversal of a node's input tree.
1378 
1379     It recursively tracks the inputs (or output recipients) of the node called
1380     node_name, and append these inputs (or output recipients) to a list of text
1381     lines (lines) with proper indentation that reflects the recursion depth,
1382     together with some formatting attributes (to attr_segs). The formatting
1383     attributes can include command shortcuts, for example.
1384 
1385     Args:
1386       lines: Text lines to append to, as a list of str.
1387       attr_segs: (dict) Attribute segments dictionary to append to.
1388       node_name: Name of the node, as a str. This arg is updated during the
1389         recursion.
1390       tracker: A callable that takes one str as the node name input and
1391         returns a list of str as the inputs/outputs.
1392         This makes it this function general enough to be used with both
1393         node-input and node-output tracking.
1394       max_depth: Maximum recursion depth, as an int.
1395       depth: Current recursion depth. This arg is updated during the
1396         recursion.
1397       unfinished: A stack of unfinished recursion depths, as a list of int.
1398       include_control: Whether control dependencies are to be included as
1399         inputs (and marked as such).
1400       show_op_type: Whether op type of the input nodes are to be displayed
1401         alongside the nodes' names.
1402       command_template: (str) Template for command shortcut of the node names.
1403     """
1404 
1405     # Make a shallow copy of the list because it may be extended later.
1406     all_inputs = self._exclude_blacklisted_ops(
1407         copy.copy(tracker(node_name, is_control=False)))
1408     is_ctrl = [False] * len(all_inputs)
1409     if include_control:
1410       # Sort control inputs or recipients in alphabetical order of the node
1411       # names.
1412       ctrl_inputs = self._exclude_blacklisted_ops(
1413           sorted(tracker(node_name, is_control=True)))
1414       all_inputs.extend(ctrl_inputs)
1415       is_ctrl.extend([True] * len(ctrl_inputs))
1416 
1417     if not all_inputs:
1418       if depth == 1:
1419         lines.append("  [None]")
1420 
1421       return
1422 
1423     unfinished.append(depth)
1424 
1425     # Create depth-dependent hanging indent for the line.
1426     hang = ""
1427     for k in xrange(depth):
1428       if k < depth - 1:
1429         if k + 1 in unfinished:
1430           hang += HANG_UNFINISHED
1431         else:
1432           hang += HANG_FINISHED
1433       else:
1434         hang += HANG_SUFFIX
1435 
1436     if all_inputs and depth > max_depth:
1437       lines.append(hang + ELLIPSIS)
1438       unfinished.pop()
1439       return
1440 
1441     hang += DEPTH_TEMPLATE % depth
1442 
1443     for i in xrange(len(all_inputs)):
1444       inp = all_inputs[i]
1445       op_type = self._debug_dump.node_op_type(debug_graphs.get_node_name(inp))
1446       if op_type in self._GRAPH_STRUCT_OP_TYPE_BLACKLIST:
1447         continue
1448 
1449       if is_ctrl[i]:
1450         ctrl_str = CTRL_LABEL
1451       else:
1452         ctrl_str = ""
1453 
1454       op_type_str = ""
1455       if show_op_type:
1456         op_type_str = OP_TYPE_TEMPLATE % op_type
1457 
1458       if i == len(all_inputs) - 1:
1459         unfinished.pop()
1460 
1461       line = hang + ctrl_str + op_type_str + inp
1462       lines.append(line)
1463       if command_template:
1464         attr_segs[len(lines) - 1] = [(
1465             len(line) - len(inp), len(line),
1466             debugger_cli_common.MenuItem(None, command_template % inp))]
1467 
1468       # Recursive call.
1469       # The input's/output's name can be a tensor name, in the case of node
1470       # with >1 output slots.
1471       inp_node_name, _ = debug_graphs.parse_node_or_tensor_name(inp)
1472       self._dfs_from_node(
1473           lines,
1474           attr_segs,
1475           inp_node_name,
1476           tracker,
1477           max_depth,
1478           depth + 1,
1479           unfinished,
1480           include_control=include_control,
1481           show_op_type=show_op_type,
1482           command_template=command_template)
