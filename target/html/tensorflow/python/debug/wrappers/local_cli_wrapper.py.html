  42 class LocalCLIDebugWrapperSession(framework.BaseDebugWrapperSession):
  43   """Concrete subclass of BaseDebugWrapperSession implementing a local CLI.
  44 
  45   This class has all the methods that a `session.Session` object has, in order
  46   to support debugging with minimal code changes. Invoking its `run()` method
  47   will launch the command-line interface (CLI) of tfdbg.
  48   """
  49 
  50   def __init__(self,
  51                sess,
  52                dump_root=None,
  53                log_usage=True,
  54                ui_type="curses",
  55                thread_name_filter=None):
  56     """Constructor of LocalCLIDebugWrapperSession.
  57 
  58     Args:
  59       sess: The TensorFlow `Session` object being wrapped.
  60       dump_root: (`str`) optional path to the dump root directory. Must be a
  61         directory that does not exist or an empty directory. If the directory
  62         does not exist, it will be created by the debugger core during debug
  63         `run()` calls and removed afterwards. If `None`, the debug dumps will
  64         be at tfdbg_<random_string> under the system temp directory.
  65       log_usage: (`bool`) whether the usage of this class is to be logged.
  66       ui_type: (`str`) requested UI type. Currently supported:
  67         (curses | readline)
  68       thread_name_filter: Regular-expression white list for thread name. See
  69         the doc of `BaseDebugWrapperSession` for details.
  70 
  71     Raises:
  72       ValueError: If dump_root is an existing and non-empty directory or if
  73         dump_root is a file.
  74     """
  75 
  76     if log_usage:
  77       pass  # No logging for open-source.
  78 
  79     framework.BaseDebugWrapperSession.__init__(
  80         self, sess, thread_name_filter=thread_name_filter)
  81 
  82     if not dump_root:
  83       self._dump_root = tempfile.mktemp(prefix=_DUMP_ROOT_PREFIX)
  84     else:
  85       dump_root = os.path.expanduser(dump_root)
  86       if os.path.isfile(dump_root):
  87         raise ValueError("dump_root path points to a file: %s" % dump_root)
  88       elif os.path.isdir(dump_root) and os.listdir(dump_root):
  89         raise ValueError("dump_root path points to a non-empty directory: %s" %
  90                          dump_root)
  91 
  92       self._dump_root = dump_root
  93 
  94     self._initialize_argparsers()
  95 
  96     # Registered tensor filters.
  97     self._tensor_filters = {}
  98     # Register frequently-used filter(s).
  99     self.add_tensor_filter("has_inf_or_nan", debug_data.has_inf_or_nan)
 100 
 101     # Below are the state variables of this wrapper object.
 102     # _active_tensor_filter: what (if any) tensor filter is in effect. If such
 103     #   a filter is in effect, this object will call run() method of the
 104     #   underlying TensorFlow Session object until the filter passes. This is
 105     #   activated by the "-f" flag of the "run" command.
 106     # _run_through_times: keeps track of how many times the wrapper needs to
 107     #   run through without stopping at the run-end CLI. It is activated by the
 108     #   "-t" option of the "run" command.
 109     # _skip_debug: keeps track of whether the current run should be executed
 110     #   without debugging. It is activated by the "-n" option of the "run"
 111     #   command.
 112     #
 113     # _run_start_response: keeps track what OnRunStartResponse the wrapper
 114     #   should return at the next run-start callback. If this information is
 115     #   unavailable (i.e., is None), the run-start CLI will be launched to ask
 116     #   the user. This is the case, e.g., right before the first run starts.
 117     self._active_tensor_filter = None
 118     self._active_filter_exclude_node_names = None
 119     self._active_tensor_filter_run_start_response = None
 120     self._run_through_times = 1
 121     self._skip_debug = False
 122     self._run_start_response = None
 123     self._is_run_start = True
 124 
 125     self._ui_type = ui_type

 127   def _is_disk_usage_reset_each_run(self):
 128     # The dumped tensors are all cleaned up after every Session.run
 129     # in a command-line wrapper.
 130     return True

 156         help="Run until a tensor in the graph passes the specified filter.")

 167         "--node_name_filter",

 232   def on_run_start(self, request):
 233     """Overrides on-run-start callback.
 234 
 235     Invoke the CLI to let user choose what action to take:
 236       `run` / `invoke_stepper`.
 237 
 238     Args:
 239       request: An instance of `OnRunStartRequest`.
 240 
 241     Returns:
 242       An instance of `OnRunStartResponse`.
 243     """
 244     self._is_run_start = True
 245     self._update_run_calls_state(
 246         request.run_call_count, request.fetches, request.feed_dict,
 247         is_callable_runner=request.is_callable_runner)
 248 
 249     if self._active_tensor_filter:
 250       # If we are running until a filter passes, we just need to keep running
 251       # with the previous `OnRunStartResponse`.
 252       return self._active_tensor_filter_run_start_response
 253 
 254     self._exit_if_requested_by_user()
 255 
 256     if self._run_call_count > 1 and not self._skip_debug:
 257       if self._run_through_times > 0:
 258         # Just run through without debugging.
 259         return framework.OnRunStartResponse(
 260             framework.OnRunStartAction.NON_DEBUG_RUN, [])
 261       elif self._run_through_times == 0:
 262         # It is the run at which the run-end CLI will be launched: activate
 263         # debugging.
 264         return (self._run_start_response or
 265                 framework.OnRunStartResponse(
 266                     framework.OnRunStartAction.DEBUG_RUN,
 267                     self._get_run_debug_urls()))
 268 
 269     if self._run_start_response is None:
 270       self._prep_cli_for_run_start()
 271 
 272       self._run_start_response = self._launch_cli()
 273       if self._active_tensor_filter:
 274         self._active_tensor_filter_run_start_response = self._run_start_response
 275       if self._run_through_times > 1:
 276         self._run_through_times -= 1
 277 
 278     self._exit_if_requested_by_user()
 279     return self._run_start_response
 280 
 281   def _exit_if_requested_by_user(self):
 282     if self._run_start_response == debugger_cli_common.EXPLICIT_USER_EXIT:
 283       # Explicit user "exit" command leads to sys.exit(1).
 284       print(
 285           "Note: user exited from debugger CLI: Calling sys.exit(1).",
 286           file=sys.stderr)
 287       sys.exit(1)

 289   def _prep_cli_for_run_start(self):
 290     """Prepare (but not launch) the CLI for run-start."""
 291 
 292     self._run_cli = ui_factory.get_ui(self._ui_type)
 293 
 294     help_intro = debugger_cli_common.RichTextLines([])
 295     if self._run_call_count == 1:
 296       # Show logo at the onset of the first run.
 297       help_intro.extend(cli_shared.get_tfdbg_logo())
 298       help_intro.extend(debugger_cli_common.get_tensorflow_version_lines())
 299     help_intro.extend(debugger_cli_common.RichTextLines("Upcoming run:"))
 300     help_intro.extend(self._run_info)
 301 
 302     self._run_cli.set_help_intro(help_intro)
 303 
 304     # Create initial screen output detailing the run.
 305     self._title = "run-start: " + self._run_description
 306     self._init_command = "run_info"
 307     self._title_color = "blue_on_white"

 309   def on_run_end(self, request):
 310     """Overrides on-run-end callback.
 311 
 312     Actions taken:
 313       1) Load the debug dump.
 314       2) Bring up the Analyzer CLI.
 315 
 316     Args:
 317       request: An instance of OnSessionInitRequest.
 318 
 319     Returns:
 320       An instance of OnSessionInitResponse.
 321     """
 322 
 323     self._is_run_start = False
 324     if request.performed_action == framework.OnRunStartAction.DEBUG_RUN:
 325       partition_graphs = None
 326       if request.run_metadata and request.run_metadata.partition_graphs:
 327         partition_graphs = request.run_metadata.partition_graphs
 328       elif request.client_graph_def:
 329         partition_graphs = [request.client_graph_def]
 330 
 331       if request.tf_error and not os.path.isdir(self._dump_root):
 332         # It is possible that the dump root may not exist due to errors that
 333         # have occurred prior to graph execution (e.g., invalid device
 334         # assignments), in which case we will just raise the exception as the
 335         # unwrapped Session does.
 336         raise request.tf_error
 337 
 338       debug_dump = debug_data.DebugDumpDir(
 339           self._dump_root, partition_graphs=partition_graphs)
 340       debug_dump.set_python_graph(self._sess.graph)
 341 
 342       passed_filter = None
 343       passed_filter_exclude_node_names = None
 344       if self._active_tensor_filter:
 345         if not debug_dump.find(
 346             self._tensor_filters[self._active_tensor_filter], first_n=1,
 347             exclude_node_names=self._active_filter_exclude_node_names):
 348           # No dumped tensor passes the filter in this run. Clean up the dump
 349           # directory and move on.
 350           self._remove_dump_root()
 351           return framework.OnRunEndResponse()
 352         else:
 353           # Some dumped tensor(s) from this run passed the filter.
 354           passed_filter = self._active_tensor_filter
 355           passed_filter_exclude_node_names = (
 356               self._active_filter_exclude_node_names)
 357           self._active_tensor_filter = None
 358           self._active_filter_exclude_node_names = None
 359 
 360       self._prep_debug_cli_for_run_end(
 361           debug_dump, request.tf_error, passed_filter,
 362           passed_filter_exclude_node_names)
 363 
 364       self._run_start_response = self._launch_cli()
 365 
 366       # Clean up the dump generated by this run.
 367       self._remove_dump_root()
 368     elif request.performed_action == framework.OnRunStartAction.PROFILE_RUN:
 369       self._prep_profile_cli_for_run_end(self._sess.graph, request.run_metadata)
 370       self._run_start_response = self._launch_cli()
 371     else:
 372       # No debug information to show following a non-debug run() call.
 373       self._run_start_response = None
 374 
 375     # Return placeholder response that currently holds no additional
 376     # information.
 377     return framework.OnRunEndResponse()
 378 
 379   def _remove_dump_root(self):
 380     if os.path.isdir(self._dump_root):
 381       shutil.rmtree(self._dump_root)

 383   def _prep_debug_cli_for_run_end(self,
 384                                   debug_dump,
 385                                   tf_error,
 386                                   passed_filter,
 387                                   passed_filter_exclude_node_names):
 388     """Prepare (but not launch) CLI for run-end, with debug dump from the run.
 389 
 390     Args:
 391       debug_dump: (debug_data.DebugDumpDir) The debug dump directory from this
 392         run.
 393       tf_error: (None or OpError) OpError that happened during the run() call
 394         (if any).
 395       passed_filter: (None or str) Name of the tensor filter that just passed
 396         and caused the preparation of this run-end CLI (if any).
 397       passed_filter_exclude_node_names: (None or str) Regular expression used
 398         with the tensor filter to exclude ops with names matching the regular
 399         expresssion.
 400     """
 401 
 402     if tf_error:
 403       help_intro = cli_shared.get_error_intro(tf_error)
 404 
 405       self._init_command = "help"
 406       self._title_color = "red_on_white"
 407     else:
 408       help_intro = None
 409       self._init_command = "lt"
 410 
 411       self._title_color = "black_on_white"
 412       if passed_filter is not None:
 413         # Some dumped tensor(s) from this run passed the filter.
 414         self._init_command = "lt -f %s" % passed_filter
 415         if passed_filter_exclude_node_names:
 416           self._init_command += (" --filter_exclude_node_names %s" %
 417                                  passed_filter_exclude_node_names)
 418         self._title_color = "red_on_white"
 419 
 420     self._run_cli = analyzer_cli.create_analyzer_ui(
 421         debug_dump, self._tensor_filters, ui_type=self._ui_type,
 422         on_ui_exit=self._remove_dump_root)
 423 
 424     # Get names of all dumped tensors.
 425     dumped_tensor_names = []
 426     for datum in debug_dump.dumped_tensor_data:
 427       dumped_tensor_names.append("%s:%d" %
 428                                  (datum.node_name, datum.output_slot))
 429 
 430     # Tab completions for command "print_tensors".
 431     self._run_cli.register_tab_comp_context(["print_tensor", "pt"],
 432                                             dumped_tensor_names)
 433 
 434     # Tab completion for commands "node_info", "list_inputs" and
 435     # "list_outputs". The list comprehension is used below because nodes()
 436     # output can be unicodes and they need to be converted to strs.
 437     self._run_cli.register_tab_comp_context(
 438         ["node_info", "ni", "list_inputs", "li", "list_outputs", "lo"],
 439         [str(node_name) for node_name in debug_dump.nodes()])
 440     # TODO(cais): Reduce API surface area for aliases vis-a-vis tab
 441     #    completion contexts and registered command handlers.
 442 
 443     self._title = "run-end: " + self._run_description
 444 
 445     if help_intro:
 446       self._run_cli.set_help_intro(help_intro)
 447

 455   def _launch_cli(self):
 456     """Launch the interactive command-line interface.
 457 
 458     Returns:
 459       The OnRunStartResponse specified by the user using the "run" command.
 460     """
 461 
 462     self._register_this_run_info(self._run_cli)
 463     response = self._run_cli.run_ui(
 464         init_command=self._init_command,
 465         title=self._title,
 466         title_color=self._title_color)
 467 
 468     return response

 470   def _run_info_handler(self, args, screen_info=None):
 471     output = debugger_cli_common.RichTextLines([])
 472 
 473     if self._run_call_count == 1:
 474       output.extend(cli_shared.get_tfdbg_logo())
 475       output.extend(debugger_cli_common.get_tensorflow_version_lines())
 476     output.extend(self._run_info)
 477 
 478     if (not self._is_run_start and
 479         debugger_cli_common.MAIN_MENU_KEY in output.annotations):
 480       menu = output.annotations[debugger_cli_common.MAIN_MENU_KEY]
 481       if "list_tensors" not in menu.captions():
 482         menu.insert(
 483             0, debugger_cli_common.MenuItem("list_tensors", "list_tensors"))
 484 
 485     return output

 487   def _print_feed_handler(self, args, screen_info=None):
 488     np_printoptions = cli_shared.numpy_printoptions_from_screen_info(
 489         screen_info)
 490 
 491     if not self._feed_dict:
 492       return cli_shared.error(
 493           "The feed_dict of the current run is None or empty.")
 494 
 495     parsed = self._argparsers["print_feed"].parse_args(args)
 496     tensor_name, tensor_slicing = (
 497         command_parser.parse_tensor_name_with_slicing(parsed.tensor_name))
 498 
 499     feed_key = None
 500     feed_value = None
 501     for key in self._feed_dict:
 502       key_name = common.get_graph_element_name(key)
 503       if key_name == tensor_name:
 504         feed_key = key_name
 505         feed_value = self._feed_dict[key]
 506         break
 507 
 508     if feed_key is None:
 509       return cli_shared.error(
 510           "The feed_dict of the current run does not contain the key %s" %
 511           tensor_name)
 512     else:
 513       return cli_shared.format_tensor(
 514           feed_value,
 515           feed_key + " (feed)",
 516           np_printoptions,
 517           print_all=parsed.print_all,
 518           tensor_slicing=tensor_slicing,
 519           highlight_options=cli_shared.parse_ranges_highlight(parsed.ranges),
 520           include_numeric_summary=parsed.numeric_summary)

 522   def _run_handler(self, args, screen_info=None):
 523     """Command handler for "run" command during on-run-start."""
 524 
 525     del screen_info  # Currently unused.
 526 
 527     parsed = self._argparsers["run"].parse_args(args)
 528     parsed.node_name_filter = parsed.node_name_filter or None
 529     parsed.op_type_filter = parsed.op_type_filter or None
 530     parsed.tensor_dtype_filter = parsed.tensor_dtype_filter or None
 531 
 532     if parsed.filter_exclude_node_names and not parsed.till_filter_pass:
 533       raise ValueError(
 534           "The --filter_exclude_node_names (or -feon) flag is valid only if "
 535           "the --till_filter_pass (or -f) flag is used.")
 536 
 537     if parsed.profile:
 538       raise debugger_cli_common.CommandLineExit(
 539           exit_token=framework.OnRunStartResponse(
 540               framework.OnRunStartAction.PROFILE_RUN, []))
 541 
 542     self._skip_debug = parsed.no_debug
 543     self._run_through_times = parsed.times
 544 
 545     if parsed.times > 1 or parsed.no_debug:
 546       # If requested -t times > 1, the very next run will be a non-debug run.
 547       action = framework.OnRunStartAction.NON_DEBUG_RUN
 548       debug_urls = []
 549     else:
 550       action = framework.OnRunStartAction.DEBUG_RUN
 551       debug_urls = self._get_run_debug_urls()
 552     run_start_response = framework.OnRunStartResponse(
 553         action,
 554         debug_urls,
 555         node_name_regex_whitelist=parsed.node_name_filter,
 556         op_type_regex_whitelist=parsed.op_type_filter,
 557         tensor_dtype_regex_whitelist=parsed.tensor_dtype_filter)
 558 
 559     if parsed.till_filter_pass:
 560       # For the run-till-filter-pass (run -f) mode, use the DEBUG_RUN
 561       # option to access the intermediate tensors, and set the corresponding
 562       # state flag of the class itself to True.
 563       if parsed.till_filter_pass in self._tensor_filters:
 564         action = framework.OnRunStartAction.DEBUG_RUN
 565         self._active_tensor_filter = parsed.till_filter_pass
 566         self._active_filter_exclude_node_names = (
 567             parsed.filter_exclude_node_names)
 568         self._active_tensor_filter_run_start_response = run_start_response
 569       else:
 570         # Handle invalid filter name.
 571         return debugger_cli_common.RichTextLines(
 572             ["ERROR: tensor filter \"%s\" does not exist." %
 573              parsed.till_filter_pass])
 574 
 575     # Raise CommandLineExit exception to cause the CLI to exit.
 576     raise debugger_cli_common.CommandLineExit(exit_token=run_start_response)

 593         prefix_aliases=["ri"])
 594     curses_cli.register_command_handler(

 667   def invoke_node_stepper(self,
 668                           node_stepper,
 669                           restore_variable_values_on_exit=True):
 670     """Overrides method in base class to implement interactive node stepper.
 671 
 672     Args:
 673       node_stepper: (`stepper.NodeStepper`) The underlying NodeStepper API
 674         object.
 675       restore_variable_values_on_exit: (`bool`) Whether any variables whose
 676         values have been altered during this node-stepper invocation should be
 677         restored to their old values when this invocation ends.
 678 
 679     Returns:
 680       The same return values as the `Session.run()` call on the same fetches as
 681         the NodeStepper.
 682     """
 683 
 684     stepper = stepper_cli.NodeStepperCLI(node_stepper)
 685 
 686     # On exiting the node-stepper CLI, the finalize method of the node_stepper
 687     # object will be called, ensuring that the state of the graph will be the
 688     # same as if the stepping did not happen.
 689     # TODO(cais): Perhaps some users will want the effect of the interactive
 690     # stepping and value injection to persist. When that happens, make the call
 691     # to finalize optional.
 692     stepper_ui = ui_factory.get_ui(
 693         self._ui_type,
 694         on_ui_exit=(node_stepper.restore_variable_values if
 695                     restore_variable_values_on_exit else None))
 696 
 697     stepper_ui.register_command_handler(
 698         "list_sorted_nodes",
 699         stepper.list_sorted_nodes,
 700         stepper.arg_parsers["list_sorted_nodes"].format_help(),
 701         prefix_aliases=["lt", "lsn"])
 702     stepper_ui.register_command_handler(
 703         "cont",
 704         stepper.cont,
 705         stepper.arg_parsers["cont"].format_help(),
 706         prefix_aliases=["ct", "c"])
 707     stepper_ui.register_command_handler(
 708         "step",
 709         stepper.step,
 710         stepper.arg_parsers["step"].format_help(),
 711         prefix_aliases=["st", "s"])
 712     stepper_ui.register_command_handler(
 713         "print_tensor",
 714         stepper.print_tensor,
 715         stepper.arg_parsers["print_tensor"].format_help(),
 716         prefix_aliases=["pt"])
 717     stepper_ui.register_command_handler(
 718         "inject_value",
 719         stepper.inject_value,
 720         stepper.arg_parsers["inject_value"].format_help(),
 721         prefix_aliases=["inject", "override_value", "override"])
 722 
 723     # Register tab completion candidates.
 724     stepper_ui.register_tab_comp_context([
 725         "cont", "ct", "c", "pt", "inject_value", "inject", "override_value",
 726         "override"
 727     ], [str(elem) for elem in node_stepper.sorted_nodes()])
 728     # TODO(cais): Tie up register_tab_comp_context to a single alias to shorten
 729     # calls like this.
 730 
 731     return stepper_ui.run_ui(
 732         init_command="lt",
 733         title="Node Stepper: " + self._run_description,
 734         title_color="blue_on_white")
