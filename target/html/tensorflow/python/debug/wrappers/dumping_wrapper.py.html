  15 """Debugger wrapper session that dumps debug data to file:// URLs."""
  16 from __future__ import absolute_import
  17 from __future__ import division
  18 from __future__ import print_function
  19 
  20 import os
  21 import threading
  22 import time
  23 
  24 # Google-internal import(s).
  25 from tensorflow.core.util import event_pb2
  26 from tensorflow.python.debug.lib import debug_data
  27 from tensorflow.python.debug.wrappers import framework
  28 from tensorflow.python.platform import gfile
  29 
  30 
  31 class DumpingDebugWrapperSession(framework.NonInteractiveDebugWrapperSession):
  32   """Debug Session wrapper that dumps debug data to filesystem."""
  33 
  34   def __init__(self,
  35                sess,
  36                session_root,
  37                watch_fn=None,
  38                thread_name_filter=None,
  39                pass_through_operrors=None,
  40                log_usage=True):
  41     """Constructor of DumpingDebugWrapperSession.
  42 
  43     Args:
  44       sess: The TensorFlow `Session` object being wrapped.
  45       session_root: (`str`) Path to the session root directory. Must be a
  46         directory that does not exist or an empty directory. If the directory
  47         does not exist, it will be created by the debugger core during debug
  48         `tf.Session.run`
  49         calls.
  50         As the `run()` calls occur, subdirectories will be added to
  51         `session_root`. The subdirectories' names has the following pattern:
  52           run_<epoch_time_stamp>_<zero_based_run_counter>
  53         E.g., run_1480734393835964_ad4c953a85444900ae79fc1b652fb324
  54       watch_fn: (`Callable`) A Callable that can be used to define per-run
  55         debug ops and watched tensors. See the doc of
  56         `NonInteractiveDebugWrapperSession.__init__()` for details.
  57       thread_name_filter: Regular-expression white list for threads on which the
  58         wrapper session will be active. See doc of `BaseDebugWrapperSession` for
  59         more details.
  60       pass_through_operrors: If true, all captured OpErrors will be
  61         propagated. By default this captures all OpErrors.
  62       log_usage: (`bool`) whether the usage of this class is to be logged.
  63 
  64     Raises:
  65        ValueError: If `session_root` is an existing and non-empty directory or
  66        if `session_root` is a file.
  67     """
  68 
  69     if log_usage:
  70       pass  # No logging for open-source.
  71 
  72     framework.NonInteractiveDebugWrapperSession.__init__(
  73         self, sess, watch_fn=watch_fn, thread_name_filter=thread_name_filter,
  74         pass_through_operrors=pass_through_operrors)
  75 
  76     session_root = os.path.expanduser(session_root)
  77     if gfile.Exists(session_root):
  78       if not gfile.IsDirectory(session_root):
  79         raise ValueError(
  80             "session_root path points to a file: %s" % session_root)
  81       elif gfile.ListDirectory(session_root):
  82         raise ValueError(
  83             "session_root path points to a non-empty directory: %s" %
  84             session_root)
  85     else:
  86       gfile.MakeDirs(session_root)
  87     self._session_root = session_root
  88 
  89     self._run_counter = 0
  90     self._run_counter_lock = threading.Lock()
  91 
  92   def prepare_run_debug_urls(self, fetches, feed_dict):
  93     """Implementation of abstrat method in superclass.
  94 
  95     See doc of `NonInteractiveDebugWrapperSession.prepare_run_debug_urls()`
  96     for details. This implementation creates a run-specific subdirectory under
  97     self._session_root and stores information regarding run `fetches` and
  98     `feed_dict.keys()` in the subdirectory.
  99 
 100     Args:
 101       fetches: Same as the `fetches` argument to `Session.run()`
 102       feed_dict: Same as the `feed_dict` argument to `Session.run()`
 103 
 104     Returns:
 105       debug_urls: (`str` or `list` of `str`) file:// debug URLs to be used in
 106         this `Session.run()` call.
 107     """
 108 
 109     # Add a UUID to accommodate the possibility of concurrent run() calls.
 110     self._run_counter_lock.acquire()
 111     run_dir = os.path.join(self._session_root, "run_%d_%d" %
 112                            (int(time.time() * 1e6), self._run_counter))
 113     self._run_counter += 1
 114     self._run_counter_lock.release()
 115     gfile.MkDir(run_dir)
 116 
 117     fetches_event = event_pb2.Event()
 118     fetches_event.log_message.message = repr(fetches)
 119     fetches_path = os.path.join(
 120         run_dir,
 121         debug_data.METADATA_FILE_PREFIX + debug_data.FETCHES_INFO_FILE_TAG)
 122     with gfile.Open(os.path.join(fetches_path), "wb") as f:
 123       f.write(fetches_event.SerializeToString())
 124 
 125     feed_keys_event = event_pb2.Event()
 126     feed_keys_event.log_message.message = (repr(feed_dict.keys()) if feed_dict
 127                                            else repr(feed_dict))
 128 
 129     feed_keys_path = os.path.join(
 130         run_dir,
 131         debug_data.METADATA_FILE_PREFIX + debug_data.FEED_KEYS_INFO_FILE_TAG)
 132     with gfile.Open(os.path.join(feed_keys_path), "wb") as f:
 133       f.write(feed_keys_event.SerializeToString())
 134 
 135     return ["file://" + run_dir]
