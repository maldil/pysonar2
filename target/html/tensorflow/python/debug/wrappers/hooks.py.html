  31 class LocalCLIDebugHook(session_run_hook.SessionRunHook):
  32   """Command-line-interface debugger hook.
  33 
  34   Can be used as a hook for `tf.train.MonitoredSession`s and
  35   `tf.estimator.Estimator`s. Provides a substitute for
  36   `tfdbg.LocalCLIDebugWrapperSession` in cases where the session is not directly
  37   available.
  38   """
  39 
  40   def __init__(self, ui_type="curses", dump_root=None, thread_name_filter=None):
  41     """Create a local debugger command-line interface (CLI) hook.
  42 
  43     Args:
  44       ui_type: (`str`) requested user-interface type. Currently supported:
  45         (curses | readline).
  46       dump_root: (`str`) optional path to the dump root directory. Must be a
  47         directory that does not exist or an empty directory. If the directory
  48         does not exist, it will be created by the debugger core during debug
  49         `run()` calls and removed afterwards.
  50       thread_name_filter: Regular-expression white list for threads on which the
  51         wrapper session will be active. See doc of `BaseDebugWrapperSession` for
  52         more details.
  53     """
  54 
  55     self._ui_type = ui_type
  56     self._dump_root = dump_root
  57     self._thread_name_filter = thread_name_filter
  58     self._session_wrapper = None
  59     self._pending_tensor_filters = {}
  60 
  61   def add_tensor_filter(self, filter_name, tensor_filter):
  62     """Add a tensor filter.
  63 
  64     See doc of `LocalCLIDebugWrapperSession.add_tensor_filter()` for details.
  65     Override default behavior to accommodate the possibility of this method
  66     being
  67     called prior to the initialization of the underlying
  68     `LocalCLIDebugWrapperSession` object.
  69 
  70     Args:
  71       filter_name: See doc of `LocalCLIDebugWrapperSession.add_tensor_filter()`
  72         for details.
  73       tensor_filter: See doc of
  74         `LocalCLIDebugWrapperSession.add_tensor_filter()` for details.
  75     """
  76 
  77     if self._session_wrapper:
  78       self._session_wrapper.add_tensor_filter(filter_name, tensor_filter)
  79     else:
  80       self._pending_tensor_filters[filter_name] = tensor_filter
  81 
  82   def begin(self):
  83     pass
  84 
  85   def before_run(self, run_context):
  86     if not self._session_wrapper:
  87       self._session_wrapper = local_cli_wrapper.LocalCLIDebugWrapperSession(
  88           run_context.session,
  89           ui_type=self._ui_type,
  90           dump_root=self._dump_root,
  91           thread_name_filter=self._thread_name_filter)
  92 
  93       # Actually register tensor filters registered prior to the construction
  94       # of the underlying LocalCLIDebugWrapperSession object.
  95       for filter_name in self._pending_tensor_filters:
  96         self._session_wrapper.add_tensor_filter(
  97             filter_name, self._pending_tensor_filters[filter_name])
  98 
  99     # Increment run call counter.
 100     self._session_wrapper.increment_run_call_count()
 101 
 102     # Adapt run_context to an instance of OnRunStartRequest for invoking
 103     # superclass on_run_start().
 104     on_run_start_request = framework.OnRunStartRequest(
 105         run_context.original_args.fetches, run_context.original_args.feed_dict,
 106         None, None, self._session_wrapper.run_call_count)
 107 
 108     on_run_start_response = self._session_wrapper.on_run_start(
 109         on_run_start_request)
 110     self._performed_action = on_run_start_response.action
 111 
 112     run_args = session_run_hook.SessionRunArgs(
 113         None, feed_dict=None, options=config_pb2.RunOptions())
 114     if self._performed_action == framework.OnRunStartAction.DEBUG_RUN:
 115       # pylint: disable=protected-access
 116       self._session_wrapper._decorate_run_options_for_debug(
 117           run_args.options,
 118           on_run_start_response.debug_urls,
 119           debug_ops=on_run_start_response.debug_ops,
 120           node_name_regex_whitelist=(
 121               on_run_start_response.node_name_regex_whitelist),
 122           op_type_regex_whitelist=(
 123               on_run_start_response.op_type_regex_whitelist),
 124           tensor_dtype_regex_whitelist=(
 125               on_run_start_response.tensor_dtype_regex_whitelist),
 126           tolerate_debug_op_creation_failures=(
 127               on_run_start_response.tolerate_debug_op_creation_failures))
 128       # pylint: enable=protected-access
 129     elif self._performed_action == framework.OnRunStartAction.PROFILE_RUN:
 130       # pylint: disable=protected-access
 131       self._session_wrapper._decorate_run_options_for_profile(run_args.options)
 132       # pylint: enable=protected-access
 133     elif self._performed_action == framework.OnRunStartAction.INVOKE_STEPPER:
 134       # The _finalized property must be set to False so that the NodeStepper
 135       # can insert ops for retrieving TensorHandles.
 136       # pylint: disable=protected-access
 137       run_context.session.graph._finalized = False
 138       # pylint: enable=protected-access
 139 
 140       with stepper.NodeStepper(
 141           run_context.session, run_context.original_args.fetches,
 142           run_context.original_args.feed_dict) as node_stepper:
 143         self._session_wrapper.invoke_node_stepper(
 144             node_stepper, restore_variable_values_on_exit=True)
 145 
 146     return run_args
 147 
 148   def after_run(self, run_context, run_values):
 149     # Adapt run_context and run_values to OnRunEndRequest and invoke superclass
 150     # on_run_end()
 151     on_run_end_request = framework.OnRunEndRequest(self._performed_action,
 152                                                    run_values.run_metadata)
 153     self._session_wrapper.on_run_end(on_run_end_request)

 156 class DumpingDebugHook(session_run_hook.SessionRunHook):
 157   """A debugger hook that dumps debug data to filesystem.
 158 
 159   Can be used as a hook for `tf.train.MonitoredSession`s and
 160   `tf.estimator.Estimator`s.
 161   """
 162 
 163   def __init__(self,
 164                session_root,
 165                watch_fn=None,
 166                thread_name_filter=None,
 167                log_usage=True):
 168     """Create a local debugger command-line interface (CLI) hook.
 169 
 170     Args:
 171       session_root: See doc of
 172         `dumping_wrapper.DumpingDebugWrapperSession.__init__`.
 173       watch_fn: See doc of
 174         `dumping_wrapper.DumpingDebugWrapperSession.__init__`.
 175       thread_name_filter: Regular-expression white list for threads on which the
 176         wrapper session will be active. See doc of `BaseDebugWrapperSession` for
 177         more details.
 178       log_usage: (bool) Whether usage is to be logged.
 179     """
 180 
 181     self._session_root = session_root
 182     self._watch_fn = watch_fn
 183     self._thread_name_filter = thread_name_filter
 184     self._log_usage = log_usage
 185     self._session_wrapper = None
 186 
 187   def begin(self):
 188     pass
 189 
 190   def before_run(self, run_context):
 191     reset_disk_byte_usage = False
 192     if not self._session_wrapper:
 193       self._session_wrapper = dumping_wrapper.DumpingDebugWrapperSession(
 194           run_context.session,
 195           self._session_root,
 196           watch_fn=self._watch_fn,
 197           thread_name_filter=self._thread_name_filter,
 198           log_usage=self._log_usage)
 199       reset_disk_byte_usage = True
 200 
 201     self._session_wrapper.increment_run_call_count()
 202 
 203     # pylint: disable=protected-access
 204     debug_urls, watch_options = self._session_wrapper._prepare_run_watch_config(
 205         run_context.original_args.fetches, run_context.original_args.feed_dict)
 206     # pylint: enable=protected-access
 207     run_options = config_pb2.RunOptions()
 208     debug_utils.watch_graph(
 209         run_options,
 210         run_context.session.graph,
 211         debug_urls=debug_urls,
 212         debug_ops=watch_options.debug_ops,
 213         node_name_regex_whitelist=watch_options.node_name_regex_whitelist,
 214         op_type_regex_whitelist=watch_options.op_type_regex_whitelist,
 215         tensor_dtype_regex_whitelist=watch_options.tensor_dtype_regex_whitelist,
 216         tolerate_debug_op_creation_failures=(
 217             watch_options.tolerate_debug_op_creation_failures),
 218         reset_disk_byte_usage=reset_disk_byte_usage)
 219 
 220     run_args = session_run_hook.SessionRunArgs(
 221         None, feed_dict=None, options=run_options)
 222     return run_args
 223 
 224   def after_run(self, run_context, run_values):
 225     pass

 228 class GrpcDebugHook(session_run_hook.SessionRunHook):
 229   """A hook that streams debugger-related events to any grpc_debug_server.
 230 
 231   For example, the debugger data server is a grpc_debug_server. The debugger
 232   data server writes debugger-related events it receives via GRPC to logdir.
 233   This enables debugging features in Tensorboard such as health pills.
 234 
 235   When the arguments of debug_utils.watch_graph changes, strongly consider
 236   changing arguments here too so that features are available to tflearn users.
 237 
 238   Can be used as a hook for `tf.train.MonitoredSession`s and
 239   `tf.estimator.Estimator`s.
 240   """
 241 
 242   def __init__(self,
 243                grpc_debug_server_addresses,
 244                watch_fn=None,
 245                thread_name_filter=None,
 246                log_usage=True):
 247     """Constructs a GrpcDebugHook.
 248 
 249     Args:
 250       grpc_debug_server_addresses: (`list` of `str`) A list of the gRPC debug
 251         server addresses, in the format of <host:port>, with or without the
 252         "grpc://" prefix. For example: ["localhost:7000", "192.168.0.2:8000"]
 253       watch_fn: A function that allows for customizing which ops to watch at
 254         which specific steps. See doc of
 255         `dumping_wrapper.DumpingDebugWrapperSession.__init__` for details.
 256       thread_name_filter: Regular-expression white list for threads on which the
 257         wrapper session will be active. See doc of `BaseDebugWrapperSession` for
 258         more details.
 259       log_usage: (bool) Whether usage is to be logged.
 260     """
 261     self._grpc_debug_wrapper_session = None
 262     self._thread_name_filter = thread_name_filter
 263     self._grpc_debug_server_addresses = (
 264         grpc_debug_server_addresses
 265         if isinstance(grpc_debug_server_addresses, list) else
 266         [grpc_debug_server_addresses])
 267 
 268     self._watch_fn = watch_fn
 269     self._log_usage = log_usage
 270 
 271   def before_run(self, run_context):
 272     """Called right before a session is run.
 273 
 274     Args:
 275       run_context: A session_run_hook.SessionRunContext. Encapsulates
 276         information on the run.
 277 
 278     Returns:
 279       A session_run_hook.SessionRunArgs object.
 280     """
 281 
 282     if not self._grpc_debug_wrapper_session:
 283       self._grpc_debug_wrapper_session = grpc_wrapper.GrpcDebugWrapperSession(
 284           run_context.session,
 285           self._grpc_debug_server_addresses,
 286           watch_fn=self._watch_fn,
 287           thread_name_filter=self._thread_name_filter,
 288           log_usage=self._log_usage)
 289 
 290     fetches = run_context.original_args.fetches
 291     feed_dict = run_context.original_args.feed_dict
 292     watch_options = self._watch_fn(fetches, feed_dict)
 293     run_options = config_pb2.RunOptions()
 294     debug_utils.watch_graph(
 295         run_options,
 296         run_context.session.graph,
 297         debug_urls=self._grpc_debug_wrapper_session.prepare_run_debug_urls(
 298             fetches, feed_dict),
 299         debug_ops=watch_options.debug_ops,
 300         node_name_regex_whitelist=watch_options.node_name_regex_whitelist,
 301         op_type_regex_whitelist=watch_options.op_type_regex_whitelist,
 302         tensor_dtype_regex_whitelist=watch_options.tensor_dtype_regex_whitelist,
 303         tolerate_debug_op_creation_failures=(
 304             watch_options.tolerate_debug_op_creation_failures))
 305 
 306     return session_run_hook.SessionRunArgs(
 307         None, feed_dict=None, options=run_options)

 310 class TensorBoardDebugHook(GrpcDebugHook):
 311   """A tfdbg hook that can be used with TensorBoard Debugger Plugin.
 312 
 313   This hook is the same as `GrpcDebugHook`, except that it uses a predefined
 314     `watch_fn` that
 315     1) uses `DebugIdentity` debug ops with the `gated_grpc` attribute set to
 316         `True`, to allow the interactive enabling and disabling of tensor
 317        breakpoints.
 318     2) watches all tensors in the graph.
 319   This saves the need for the user to define a `watch_fn`.
 320   """
 321 
 322   def __init__(self,
 323                grpc_debug_server_addresses,
 324                thread_name_filter=None,
 325                send_traceback_and_source_code=True,
 326                log_usage=True):
 327     """Constructor of TensorBoardDebugHook.
 328 
 329     Args:
 330       grpc_debug_server_addresses: gRPC address(es) of debug server(s), as a
 331         `str` or a `list` of `str`s. E.g., "localhost:2333",
 332         "grpc://localhost:2333", ["192.168.0.7:2333", "192.168.0.8:2333"].
 333       thread_name_filter: Optional filter for thread names.
 334       send_traceback_and_source_code: Whether traceback of graph elements and
 335         the source code are to be sent to the debug server(s).
 336       log_usage: Whether the usage of this class is to be logged (if
 337         applicable).
 338     """
 339 
 340     def _gated_grpc_watch_fn(fetches, feeds):
 341       del fetches, feeds  # Unused.
 342       return framework.WatchOptions(
 343           debug_ops=["DebugIdentity(gated_grpc=true)"])
 344 
 345     super(TensorBoardDebugHook, self).__init__(
 346         grpc_debug_server_addresses,
 347         watch_fn=_gated_grpc_watch_fn,
 348         thread_name_filter=thread_name_filter,
 349         log_usage=log_usage)
 350 
 351     self._grpc_debug_server_addresses = grpc_debug_server_addresses
 352     self._send_traceback_and_source_code = send_traceback_and_source_code
 353     self._sent_graph_version = -1
 354     grpc_wrapper.register_signal_handler()
 355 
 356   def before_run(self, run_context):
 357     if self._send_traceback_and_source_code:
 358       self._sent_graph_version = grpc_wrapper.publish_traceback(
 359           self._grpc_debug_server_addresses, run_context.session.graph,
 360           run_context.original_args.feed_dict,
 361           run_context.original_args.fetches, self._sent_graph_version)
 362     return super(TensorBoardDebugHook, self).before_run(run_context)
