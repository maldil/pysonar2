  32 # URLs to download data sets from, if necessary.
  33 IRIS_TRAINING_DATA_URL = "https://raw.githubusercontent.com/tensorflow/tensorflow/master/tensorflow/examples/tutorials/monitors/iris_training.csv"
  34 IRIS_TEST_DATA_URL = "https://raw.githubusercontent.com/tensorflow/tensorflow/master/tensorflow/examples/tutorials/monitors/iris_test.csv"

  37 def maybe_download_data(data_dir):
  38   """Download data sets if necessary.
  39 
  40   Args:
  41     data_dir: Path to where data should be downloaded.
  42 
  43   Returns:
  44     Paths to the training and test data files.
  45   """
  46 
  47   if not os.path.isdir(data_dir):
  48     os.makedirs(data_dir)
  49 
  50   training_data_path = os.path.join(data_dir,
  51                                     os.path.basename(IRIS_TRAINING_DATA_URL))
  52   if not os.path.isfile(training_data_path):
  53     train_file = open(training_data_path, "wt")
  54     urllib.request.urlretrieve(IRIS_TRAINING_DATA_URL, train_file.name)
  55     train_file.close()
  56 
  57     print("Training data are downloaded to %s" % train_file.name)
  58 
  59   test_data_path = os.path.join(data_dir, os.path.basename(IRIS_TEST_DATA_URL))
  60   if not os.path.isfile(test_data_path):
  61     test_file = open(test_data_path, "wt")
  62     urllib.request.urlretrieve(IRIS_TEST_DATA_URL, test_file.name)
  63     test_file.close()
  64 
  65     print("Test data are downloaded to %s" % test_file.name)
  66 
  67   return training_data_path, test_data_path

  70 _IRIS_INPUT_DIM = 4
  71 
  72 
  73 def iris_input_fn():
  74   iris = base.load_iris()
  75   features = tf.reshape(tf.constant(iris.data), [-1, _IRIS_INPUT_DIM])
  76   labels = tf.reshape(tf.constant(iris.target), [-1])
  77   return features, labels

  80 def main(_):
  81   # Load datasets.
  82   if FLAGS.fake_data:
  83     def training_input_fn():
  84       return ({"features": tf.random_normal([128, 4])},
  85               tf.random_uniform([128], minval=0, maxval=3, dtype=tf.int32))
  86     def test_input_fn():
  87       return ({"features": tf.random_normal([32, 4])},
  88               tf.random_uniform([32], minval=0, maxval=3, dtype=tf.int32))
  89     feature_columns = [
  90         tf.feature_column.numeric_column("features", shape=(4,))]
  91   else:
  92     training_data_path, test_data_path = maybe_download_data(FLAGS.data_dir)
  93     column_names = [
  94         "sepal_length", "sepal_width", "petal_length", "petal_width", "label"]
  95     batch_size = 32
  96     def training_input_fn():
  97       return tf.data.experimental.make_csv_dataset([training_data_path],
  98                                                    batch_size,
  99                                                    column_names=column_names,
 100                                                    label_name="label")
 101     def test_input_fn():
 102       return tf.data.experimental.make_csv_dataset([test_data_path],
 103                                                    batch_size,
 104                                                    column_names=column_names,
 105                                                    label_name="label")
 106     feature_columns = [tf.feature_column.numeric_column(feature)
 107                        for feature in column_names[:-1]]
 108 
 109   # Build 3 layer DNN with 10, 20, 10 units respectively.
 110   model_dir = FLAGS.model_dir or tempfile.mkdtemp(prefix="debug_tflearn_iris_")
 111 
 112   classifier = tf.estimator.DNNClassifier(
 113       feature_columns=feature_columns,
 114       hidden_units=[10, 20, 10],
 115       n_classes=3,
 116       model_dir=model_dir)
 117 
 118   if FLAGS.debug and FLAGS.tensorboard_debug_address:
 119     raise ValueError(
 120         "The --debug and --tensorboard_debug_address flags are mutually "
 121         "exclusive.")
 122   hooks = []
 123   if FLAGS.debug:
 124     hooks.append(tf_debug.LocalCLIDebugHook(ui_type=FLAGS.ui_type,
 125                                             dump_root=FLAGS.dump_root))
 126   elif FLAGS.tensorboard_debug_address:
 127     hooks.append(tf_debug.TensorBoardDebugHook(FLAGS.tensorboard_debug_address))
 128 
 129   # Train model, using tfdbg hook.
 130   classifier.train(training_input_fn,
 131                    steps=FLAGS.train_steps,
 132                    hooks=hooks)
 133 
 134   # Evaluate accuracy, using tfdbg hook.
 135   accuracy_score = classifier.evaluate(test_input_fn,
 136                                        steps=FLAGS.eval_steps,
 137                                        hooks=hooks)["accuracy"]
 138 
 139   print("After training %d steps, Accuracy = %f" %
 140         (FLAGS.train_steps, accuracy_score))
 141 
 142   # Make predictions, using tfdbg hook.
 143   predict_results = classifier.predict(test_input_fn, hooks=hooks)
 144   print("A prediction result: %s" % next(predict_results))
