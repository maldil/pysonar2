  15 """tfdbg example: debugging tf.keras models training on tf.data.Dataset."""
  16 
  17 from __future__ import absolute_import
  18 from __future__ import division
  19 from __future__ import print_function
  20 
  21 import argparse
  22 import sys
  23 
  24 import numpy as np
  25 import tensorflow as tf
  26 
  27 from tensorflow.python import debug as tf_debug
  28 
  29 
  30 def main(_):
  31   # Create a dummy dataset.
  32   num_examples = 8
  33   steps_per_epoch = 2
  34   input_dims = 3
  35   output_dims = 1
  36   xs = np.zeros([num_examples, input_dims])
  37   ys = np.zeros([num_examples, output_dims])
  38   dataset = tf.data.Dataset.from_tensor_slices(
  39       (xs, ys)).repeat(num_examples).batch(int(num_examples / steps_per_epoch))
  40 
  41   sess = tf.Session()
  42   if FLAGS.debug:
  43     # Use the command-line interface (CLI) of tfdbg.
  44     sess = tf_debug.LocalCLIDebugWrapperSession(sess, ui_type=FLAGS.ui_type)
  45   elif FLAGS.tensorboard_debug_address:
  46     # Use the TensorBoard Debugger Plugin (GUI of tfdbg).
  47     sess = tf_debug.TensorBoardDebugWrapperSession(
  48         sess, FLAGS.tensorboard_debug_address)
  49   tf.keras.backend.set_session(sess)
  50 
  51   # Create a dummy model.
  52   model = tf.keras.Sequential([
  53       tf.keras.layers.Dense(1, input_shape=[input_dims])])
  54   model.compile(loss="mse", optimizer="sgd")
  55 
  56   # Train the model using the dummy dataset created above.
  57   model.fit(dataset, epochs=FLAGS.epochs, steps_per_epoch=steps_per_epoch)
  58 
  59 
  60 if __name__ == "__main__":
  61   parser = argparse.ArgumentParser()
  62   parser.register("type", "bool", lambda v: v.lower() == "true")
  63   parser.add_argument(
  64       "--debug",
  65       type="bool",
  66       nargs="?",
  67       const=True,
  68       default=False,
  69       help="Use debugger to track down bad values during training. "
  70       "Mutually exclusive with the --tensorboard_debug_address flag.")
  71   parser.add_argument(
  72       "--ui_type",
  73       type=str,
  74       default="curses",
  75       help="Command-line user interface type (curses | readline).")
  76   parser.add_argument(
  77       "--tensorboard_debug_address",
  78       type=str,
  79       default=None,
  80       help="Connect to the TensorBoard Debugger Plugin backend specified by "
  81       "the gRPC address (e.g., localhost:1234). Mutually exclusive with the "
  82       "--debug flag.")
  83   parser.add_argument(
  84       "--epochs",
  85       type=int,
  86       default=2,
  87       help="Number of epochs to train the model for.")
  88   FLAGS, unparsed = parser.parse_known_args()
  89   tf.app.run(main=main, argv=[sys.argv[0]] + unparsed)
