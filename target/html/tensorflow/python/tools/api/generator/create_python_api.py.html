  59 def format_import(source_module_name, source_name, dest_name):
  60   """Formats import statement.
  61 
  62   Args:
  63     source_module_name: (string) Source module to import from.
  64     source_name: (string) Source symbol name to import.
  65     dest_name: (string) Destination alias name.
  66 
  67   Returns:
  68     An import statement string.
  69   """
  70   if source_module_name:
  71     if source_name == dest_name:
  72       return 'from %s import %s' % (source_module_name, source_name)
  73     else:
  74       return 'from %s import %s as %s' % (
  75           source_module_name, source_name, dest_name)
  76   else:
  77     if source_name == dest_name:
  78       return 'import %s' % source_name
  79     else:
  80       return 'import %s as %s' % (source_name, dest_name)

  83 class _ModuleInitCodeBuilder(object):
  84   """Builds a map from module name to imports included in that module."""
  85 
  86   def __init__(self, output_package):
  87     self._output_package = output_package
  88     self._module_imports = collections.defaultdict(
  89         lambda: collections.defaultdict(set))
  90     self._dest_import_to_id = collections.defaultdict(int)
  91     # Names that start with underscore in the root module.
  92     self._underscore_names_in_root = []
  93 
  94   def add_import(
  95       self, symbol_id, dest_module_name, source_module_name, source_name,
  96       dest_name):
  97     """Adds this import to module_imports.
  98 
  99     Args:
 100       symbol_id: (number) Unique identifier of the symbol to import.
 101       dest_module_name: (string) Module name to add import to.
 102       source_module_name: (string) Module to import from.
 103       source_name: (string) Name of the symbol to import.
 104       dest_name: (string) Import the symbol using this name.
 105 
 106     Raises:
 107       SymbolExposedTwiceError: Raised when an import with the same
 108         dest_name has already been added to dest_module_name.
 109     """
 110     import_str = format_import(source_module_name, source_name, dest_name)
 111 
 112     # Check if we are trying to expose two different symbols with same name.
 113     full_api_name = dest_name
 114     if dest_module_name:
 115       full_api_name = dest_module_name + '.' + full_api_name
 116     if (full_api_name in self._dest_import_to_id and
 117         symbol_id != self._dest_import_to_id[full_api_name] and
 118         symbol_id != -1):
 119       raise SymbolExposedTwiceError(
 120           'Trying to export multiple symbols with same name: %s.' %
 121           full_api_name)
 122     self._dest_import_to_id[full_api_name] = symbol_id
 123 
 124     if not dest_module_name and dest_name.startswith('_'):
 125       self._underscore_names_in_root.append(dest_name)
 126 
 127     # The same symbol can be available in multiple modules.
 128     # We store all possible ways of importing this symbol and later pick just
 129     # one.
 130     self._module_imports[dest_module_name][full_api_name].add(import_str)
 131 
 132   def _import_submodules(self):
 133     """Add imports for all destination modules in self._module_imports."""
 134     # Import all required modules in their parent modules.
 135     # For e.g. if we import 'foo.bar.Value'. Then, we also
 136     # import 'bar' in 'foo'.
 137     imported_modules = set(self._module_imports.keys())
 138     for module in imported_modules:
 139       if not module:
 140         continue
 141       module_split = module.split('.')
 142       parent_module = ''  # we import submodules in their parent_module
 143 
 144       for submodule_index in range(len(module_split)):
 145         if submodule_index > 0:
 146           submodule = module_split[submodule_index-1]
 147           parent_module += '.' + submodule if parent_module else submodule
 148         import_from = self._output_package
 149         if submodule_index > 0:
 150           import_from += '.' + '.'.join(module_split[:submodule_index])
 151         self.add_import(
 152             -1, parent_module, import_from,
 153             module_split[submodule_index], module_split[submodule_index])
 154 
 155   def build(self):
 156     """Get a map from destination module to __init__.py code for that module.
 157 
 158     Returns:
 159       A dictionary where
 160         key: (string) destination module (for e.g. tf or tf.consts).
 161         value: (string) text that should be in __init__.py files for
 162           corresponding modules.
 163     """
 164     self._import_submodules()
 165     module_text_map = {}
 166     for dest_module, dest_name_to_imports in self._module_imports.items():
 167       # Sort all possible imports for a symbol and pick the first one.
 168       imports_list = [
 169           sorted(imports)[0]
 170           for _, imports in dest_name_to_imports.items()]
 171       module_text_map[dest_module] = '\n'.join(sorted(imports_list))
 172 
 173     # Expose exported symbols with underscores in root module
 174     # since we import from it using * import.
 175     underscore_names_str = ', '.join(
 176         '\'%s\'' % name for name in self._underscore_names_in_root)
 177     # We will always generate a root __init__.py file to let us handle *
 178     # imports consistently. Be sure to have a root __init__.py file listed in
 179     # the script outputs.
 180     module_text_map[''] = module_text_map.get('', '') + '''
 181 _names_with_underscore = [%s]
 182 __all__ = [_s for _s in dir() if not _s.startswith('_')]
 183 __all__.extend([_s for _s in _names_with_underscore])
 184 ''' % underscore_names_str
 185 
 186     return module_text_map

 189 def _get_name_and_module(full_name):
 190   """Split full_name into module and short name.
 191 
 192   Args:
 193     full_name: Full name of symbol that includes module.
 194 
 195   Returns:
 196     Full module name and short symbol name.
 197   """
 198   name_segments = full_name.split('.')
 199   return '.'.join(name_segments[:-1]), name_segments[-1]

 202 def _join_modules(module1, module2):
 203   """Concatenate 2 module components.
 204 
 205   Args:
 206     module1: First module to join.
 207     module2: Second module to join.
 208 
 209   Returns:
 210     Given two modules aaa.bbb and ccc.ddd, returns a joined
 211     module aaa.bbb.ccc.ddd.
 212   """
 213   if not module1:
 214     return module2
 215   if not module2:
 216     return module1
 217   return '%s.%s' % (module1, module2)

 220 def add_imports_for_symbol(
 221     module_code_builder,
 222     symbol,
 223     source_module_name,
 224     source_name,
 225     api_name,
 226     api_version,
 227     output_module_prefix=''):
 228   """Add imports for the given symbol to `module_code_builder`.
 229 
 230   Args:
 231     module_code_builder: `_ModuleInitCodeBuilder` instance.
 232     symbol: A symbol.
 233     source_module_name: Module that we can import the symbol from.
 234     source_name: Name we can import the symbol with.
 235     api_name: API name. Currently, must be either `tensorflow` or `estimator`.
 236     api_version: API version.
 237     output_module_prefix: Prefix to prepend to destination module.
 238   """
 239   if api_version == 1:
 240     names_attr = API_ATTRS_V1[api_name].names
 241     constants_attr = API_ATTRS_V1[api_name].constants
 242   else:
 243     names_attr = API_ATTRS[api_name].names
 244     constants_attr = API_ATTRS[api_name].constants
 245 
 246   # If symbol is _tf_api_constants attribute, then add the constants.
 247   if source_name == constants_attr:
 248     for exports, name in symbol:
 249       for export in exports:
 250         dest_module, dest_name = _get_name_and_module(export)
 251         dest_module = _join_modules(output_module_prefix, dest_module)
 252         module_code_builder.add_import(
 253             -1, dest_module, source_module_name, name, dest_name)
 254 
 255   # If symbol has _tf_api_names attribute, then add import for it.
 256   if (hasattr(symbol, '__dict__') and names_attr in symbol.__dict__):
 257     for export in getattr(symbol, names_attr):  # pylint: disable=protected-access
 258       dest_module, dest_name = _get_name_and_module(export)
 259       dest_module = _join_modules(output_module_prefix, dest_module)
 260       module_code_builder.add_import(
 261           id(symbol), dest_module, source_module_name, source_name, dest_name)

 264 def get_api_init_text(
 265     package, output_package, api_name, api_version, compat_api_versions=None):
 266   """Get a map from destination module to __init__.py code for that module.
 267 
 268   Args:
 269     package: Base python package containing python with target tf_export
 270       decorators.
 271     output_package: Base output python package where generated API will
 272       be added.
 273     api_name: API you want to generate (e.g. `tensorflow` or `estimator`).
 274     api_version: API version you want to generate (1 or 2).
 275     compat_api_versions: Additional API versions to generate under compat/
 276       directory.
 277 
 278   Returns:
 279     A dictionary where
 280       key: (string) destination module (for e.g. tf or tf.consts).
 281       value: (string) text that should be in __init__.py files for
 282         corresponding modules.
 283   """
 284   if compat_api_versions is None:
 285     compat_api_versions = []
 286   module_code_builder = _ModuleInitCodeBuilder(output_package)
 287   # Traverse over everything imported above. Specifically,
 288   # we want to traverse over TensorFlow Python modules.
 289   for module in list(sys.modules.values()):
 290     # Only look at tensorflow modules.
 291     if (not module or not hasattr(module, '__name__') or
 292         module.__name__ is None or package not in module.__name__):
 293       continue
 294     # Do not generate __init__.py files for contrib modules for now.
 295     if '.contrib.' in module.__name__ or module.__name__.endswith('.contrib'):
 296       continue
 297 
 298     for module_contents_name in dir(module):
 299       if (module.__name__ + '.' + module_contents_name
 300           in _SYMBOLS_TO_SKIP_EXPLICITLY):
 301         continue
 302       attr = getattr(module, module_contents_name)
 303       _, attr = tf_decorator.unwrap(attr)
 304 
 305       add_imports_for_symbol(
 306           module_code_builder, attr, module.__name__, module_contents_name,
 307           api_name, api_version)
 308       for compat_api_version in compat_api_versions:
 309         add_imports_for_symbol(
 310             module_code_builder, attr, module.__name__, module_contents_name,
 311             api_name, compat_api_version,
 312             _COMPAT_MODULE_TEMPLATE % compat_api_version)
 313 
 314   return module_code_builder.build()

 317 def get_module(dir_path, relative_to_dir):
 318   """Get module that corresponds to path relative to relative_to_dir.
 319 
 320   Args:
 321     dir_path: Path to directory.
 322     relative_to_dir: Get module relative to this directory.
 323 
 324   Returns:
 325     Name of module that corresponds to the given directory.
 326   """
 327   dir_path = dir_path[len(relative_to_dir):]
 328   # Convert path separators to '/' for easier parsing below.
 329   dir_path = dir_path.replace(os.sep, '/')
 330   return dir_path.replace('/', '.').strip('.')

 333 def get_module_docstring(module_name, package, api_name):
 334   """Get docstring for the given module.
 335 
 336   This method looks for docstring in the following order:
 337   1. Checks if module has a docstring specified in doc_srcs.
 338   2. Checks if module has a docstring source module specified
 339      in doc_srcs. If it does, gets docstring from that module.
 340   3. Checks if module with module_name exists under base package.
 341      If it does, gets docstring from that module.
 342   4. Returns a default docstring.
 343 
 344   Args:
 345     module_name: module name relative to tensorflow
 346       (excluding 'tensorflow.' prefix) to get a docstring for.
 347     package: Base python package containing python with target tf_export
 348       decorators.
 349     api_name: API you want to generate (e.g. `tensorflow` or `estimator`).
 350 
 351   Returns:
 352     One-line docstring to describe the module.
 353   """
 354   # Get the same module doc strings for any version. That is, for module
 355   # 'compat.v1.foo' we can get docstring from module 'foo'.
 356   for version in _API_VERSIONS:
 357     compat_prefix = _COMPAT_MODULE_TEMPLATE % version
 358     if module_name.startswith(compat_prefix):
 359       module_name = module_name[len(compat_prefix):].strip('.')
 360 
 361   # Module under base package to get a docstring from.
 362   docstring_module_name = module_name
 363 
 364   doc_sources = doc_srcs.get_doc_sources(api_name)
 365 
 366   if module_name in doc_sources:
 367     docsrc = doc_sources[module_name]
 368     if docsrc.docstring:
 369       return docsrc.docstring
 370     if docsrc.docstring_module_name:
 371       docstring_module_name = docsrc.docstring_module_name
 372 
 373   docstring_module_name = package + '.' + docstring_module_name
 374   if (docstring_module_name in sys.modules and
 375       sys.modules[docstring_module_name].__doc__):
 376     return sys.modules[docstring_module_name].__doc__
 377 
 378   return 'Public API for tf.%s namespace.' % module_name

 381 def create_api_files(
 382     output_files,
 383     package,
 384     root_init_template,
 385     output_dir,
 386     output_package,
 387     api_name,
 388     api_version,
 389     compat_api_versions):
 390   """Creates __init__.py files for the Python API.
 391 
 392   Args:
 393     output_files: List of __init__.py file paths to create.
 394     package: Base python package containing python with target tf_export
 395       decorators.
 396     root_init_template: Template for top-level __init__.py file.
 397       "# API IMPORTS PLACEHOLDER" comment in the template file will be replaced
 398       with imports.
 399     output_dir: output API root directory.
 400     output_package: Base output package where generated API will be added.
 401     api_name: API you want to generate (e.g. `tensorflow` or `estimator`).
 402     api_version: API version to generate (`v1` or `v2`).
 403     compat_api_versions: Additional API versions to generate in compat/
 404       subdirectory.
 405 
 406   Raises:
 407     ValueError: if output_files list is missing a required file.
 408   """
 409   module_name_to_file_path = {}
 410   for output_file in output_files:
 411     module_name = get_module(os.path.dirname(output_file), output_dir)
 412     module_name_to_file_path[module_name] = os.path.normpath(output_file)
 413 
 414   # Create file for each expected output in genrule.
 415   for module, file_path in module_name_to_file_path.items():
 416     if not os.path.isdir(os.path.dirname(file_path)):
 417       os.makedirs(os.path.dirname(file_path))
 418     open(file_path, 'a').close()
 419 
 420   module_text_map = get_api_init_text(
 421       package, output_package, api_name, api_version, compat_api_versions)
 422 
 423   # Add imports to output files.
 424   missing_output_files = []
 425   # Root modules are "" and "compat.v*".
 426   root_modules = set(_COMPAT_MODULE_TEMPLATE % v for v in compat_api_versions)
 427   root_modules.add('')
 428   for module, text in module_text_map.items():
 429     # Make sure genrule output file list is in sync with API exports.
 430     if module not in module_name_to_file_path:
 431       module_file_path = '"%s/__init__.py"' %  (
 432           module.replace('.', '/'))
 433       missing_output_files.append(module_file_path)
 434       continue
 435 
 436     contents = ''
 437     if module not in root_modules or not root_init_template:
 438       contents = (
 439           _GENERATED_FILE_HEADER %
 440           get_module_docstring(module, package, api_name) +
 441           text + _GENERATED_FILE_FOOTER)
 442     else:
 443       # Read base init file
 444       with open(root_init_template, 'r') as root_init_template_file:
 445         contents = root_init_template_file.read()
 446         contents = contents.replace('# API IMPORTS PLACEHOLDER', text)
 447     with open(module_name_to_file_path[module], 'w') as fp:
 448       fp.write(contents)
 449 
 450   if missing_output_files:
 451     raise ValueError(
 452         'Missing outputs for genrule:\n%s.' %
 453         ',\n'.join(sorted(missing_output_files)))

 456 def main():
 457   parser = argparse.ArgumentParser()
 458   parser.add_argument(
 459       'outputs', metavar='O', type=str, nargs='+',
 460       help='If a single file is passed in, then we we assume it contains a '
 461       'semicolon-separated list of Python files that we expect this script to '
 462       'output. If multiple files are passed in, then we assume output files '
 463       'are listed directly as arguments.')
 464   parser.add_argument(
 465       '--package', default=_DEFAULT_PACKAGE, type=str,
 466       help='Base package that imports modules containing the target tf_export '
 467            'decorators.')
 468   parser.add_argument(
 469       '--root_init_template', default='', type=str,
 470       help='Template for top level __init__.py file. '
 471            '"#API IMPORTS PLACEHOLDER" comment will be replaced with imports.')
 472   parser.add_argument(
 473       '--apidir', type=str, required=True,
 474       help='Directory where generated output files are placed. '
 475            'gendir should be a prefix of apidir. Also, apidir '
 476            'should be a prefix of every directory in outputs.')
 477   parser.add_argument(
 478       '--apiname', required=True, type=str,
 479       choices=API_ATTRS.keys(),
 480       help='The API you want to generate.')
 481   parser.add_argument(
 482       '--apiversion', default=2, type=int,
 483       choices=_API_VERSIONS,
 484       help='The API version you want to generate.')
 485   parser.add_argument(
 486       '--compat_apiversions', default=[], type=int, action='append',
 487       help='Additional versions to generate in compat/ subdirectory. '
 488            'If set to 0, then no additional version would be generated.')
 489   parser.add_argument(
 490       '--output_package', default='tensorflow', type=str,
 491       help='Root output package.')
 492   args = parser.parse_args()
 493 
 494   if len(args.outputs) == 1:
 495     # If we only get a single argument, then it must be a file containing
 496     # list of outputs.
 497     with open(args.outputs[0]) as output_list_file:
 498       outputs = [line.strip() for line in output_list_file.read().split(';')]
 499   else:
 500     outputs = args.outputs
 501 
 502   # Populate `sys.modules` with modules containing tf_export().
 503   importlib.import_module(args.package)
 504   create_api_files(outputs, args.package, args.root_init_template,
 505                    args.apidir, args.output_package, args.apiname,
 506                    args.apiversion, args.compat_apiversions)
 507 
 508 
 509 if __name__ == '__main__':
 510   main()
