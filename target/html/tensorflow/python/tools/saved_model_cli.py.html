 120 def _show_inputs_outputs(saved_model_dir, tag_set, signature_def_key, indent=0):
 121   """Prints input and output TensorInfos.
 122 
 123   Prints the details of input and output TensorInfos for the SignatureDef mapped
 124   by the given signature_def_key.
 125 
 126   Args:
 127     saved_model_dir: Directory containing the SavedModel to inspect.
 128     tag_set: Group of tag(s) of the MetaGraphDef, in string format, separated by
 129         ','. For tag-set contains multiple tags, all tags must be passed in.
 130     signature_def_key: A SignatureDef key string.
 131     indent: How far (in increments of 2 spaces) to indent each line of output.
 132   """
 133   meta_graph_def = saved_model_utils.get_meta_graph_def(saved_model_dir,
 134                                                         tag_set)
 135   inputs_tensor_info = _get_inputs_tensor_info_from_meta_graph_def(
 136       meta_graph_def, signature_def_key)
 137   outputs_tensor_info = _get_outputs_tensor_info_from_meta_graph_def(
 138       meta_graph_def, signature_def_key)
 139 
 140   indent_str = '  ' * indent
 141   def in_print(s):
 142     print(indent_str + s)
 143 
 144   in_print('The given SavedModel SignatureDef contains the following input(s):')
 145   for input_key, input_tensor in sorted(inputs_tensor_info.items()):
 146     in_print('  inputs[\'%s\'] tensor_info:' % input_key)
 147     _print_tensor_info(input_tensor, indent+1)
 148 
 149   in_print('The given SavedModel SignatureDef contains the following '
 150            'output(s):')
 151   for output_key, output_tensor in sorted(outputs_tensor_info.items()):
 152     in_print('  outputs[\'%s\'] tensor_info:' % output_key)
 153     _print_tensor_info(output_tensor, indent+1)
 154 
 155   in_print('Method name is: %s' %
 156            meta_graph_def.signature_def[signature_def_key].method_name)

 159 def _print_tensor_info(tensor_info, indent=0):
 160   """Prints details of the given tensor_info.
 161 
 162   Args:
 163     tensor_info: TensorInfo object to be printed.
 164     indent: How far (in increments of 2 spaces) to indent each line output
 165   """
 166   indent_str = '  ' * indent
 167   def in_print(s):
 168     print(indent_str + s)
 169 
 170   in_print('    dtype: ' +
 171            {value: key
 172             for (key, value) in types_pb2.DataType.items()}[tensor_info.dtype])
 173   # Display shape as tuple.
 174   if tensor_info.tensor_shape.unknown_rank:
 175     shape = 'unknown_rank'
 176   else:
 177     dims = [str(dim.size) for dim in tensor_info.tensor_shape.dim]
 178     shape = ', '.join(dims)
 179     shape = '(' + shape + ')'
 180   in_print('    shape: ' + shape)
 181   in_print('    name: ' + tensor_info.name)

 247 def scan_meta_graph_def(meta_graph_def):
 248   """Scans meta_graph_def and reports if there are ops on blacklist.
 249 
 250   Print ops if they are on black list, or print success if no blacklisted ops
 251   found.
 252 
 253   Args:
 254     meta_graph_def: MetaGraphDef protocol buffer.
 255   """
 256   all_ops_set = set(
 257       meta_graph_lib.ops_used_by_graph_def(meta_graph_def.graph_def))
 258   blacklisted_ops = _OP_BLACKLIST & all_ops_set
 259   if blacklisted_ops:
 260     # TODO(yifeif): print more warnings
 261     print('MetaGraph with tag set %s contains the following blacklisted ops:' %
 262           meta_graph_def.meta_info_def.tags, blacklisted_ops)
 263   else:
 264     print('MetaGraph with tag set %s does not contain blacklisted ops.' %
 265           meta_graph_def.meta_info_def.tags)

 268 def run_saved_model_with_feed_dict(saved_model_dir, tag_set, signature_def_key,
 269                                    input_tensor_key_feed_dict, outdir,
 270                                    overwrite_flag, worker=None, init_tpu=False,
 271                                    tf_debug=False):
 272   """Runs SavedModel and fetch all outputs.
 273 
 274   Runs the input dictionary through the MetaGraphDef within a SavedModel
 275   specified by the given tag_set and SignatureDef. Also save the outputs to file
 276   if outdir is not None.
 277 
 278   Args:
 279     saved_model_dir: Directory containing the SavedModel to execute.
 280     tag_set: Group of tag(s) of the MetaGraphDef with the SignatureDef map, in
 281         string format, separated by ','. For tag-set contains multiple tags, all
 282         tags must be passed in.
 283     signature_def_key: A SignatureDef key string.
 284     input_tensor_key_feed_dict: A dictionary maps input keys to numpy ndarrays.
 285     outdir: A directory to save the outputs to. If the directory doesn't exist,
 286         it will be created.
 287     overwrite_flag: A boolean flag to allow overwrite output file if file with
 288         the same name exists.
 289     worker: If provided, the session will be run on the worker.  Valid worker
 290         specification is a bns or gRPC path.
 291     init_tpu: If true, the TPU system will be initialized after the session
 292         is created.
 293     tf_debug: A boolean flag to use TensorFlow Debugger (TFDBG) to observe the
 294         intermediate Tensor values and runtime GraphDefs while running the
 295         SavedModel.
 296 
 297   Raises:
 298     ValueError: When any of the input tensor keys is not valid.
 299     RuntimeError: An error when output file already exists and overwrite is not
 300     enabled.
 301   """
 302   # Get a list of output tensor names.
 303   meta_graph_def = saved_model_utils.get_meta_graph_def(saved_model_dir,
 304                                                         tag_set)
 305 
 306   # Re-create feed_dict based on input tensor name instead of key as session.run
 307   # uses tensor name.
 308   inputs_tensor_info = _get_inputs_tensor_info_from_meta_graph_def(
 309       meta_graph_def, signature_def_key)
 310 
 311   # Check if input tensor keys are valid.
 312   for input_key_name in input_tensor_key_feed_dict.keys():
 313     if input_key_name not in inputs_tensor_info:
 314       raise ValueError(
 315           '"%s" is not a valid input key. Please choose from %s, or use '
 316           '--show option.' %
 317           (input_key_name, '"' + '", "'.join(inputs_tensor_info.keys()) + '"'))
 318 
 319   inputs_feed_dict = {
 320       inputs_tensor_info[key].name: tensor
 321       for key, tensor in input_tensor_key_feed_dict.items()
 322   }
 323   # Get outputs
 324   outputs_tensor_info = _get_outputs_tensor_info_from_meta_graph_def(
 325       meta_graph_def, signature_def_key)
 326   # Sort to preserve order because we need to go from value to key later.
 327   output_tensor_keys_sorted = sorted(outputs_tensor_info.keys())
 328   output_tensor_names_sorted = [
 329       outputs_tensor_info[tensor_key].name
 330       for tensor_key in output_tensor_keys_sorted
 331   ]
 332 
 333   with session.Session(worker, graph=ops_lib.Graph()) as sess:
 334     if init_tpu:
 335       print('Initializing TPU System ...')
 336       # This is needed for freshly started worker, or if the job
 337       # restarts after a preemption.
 338       sess.run(tf.contrib.tpu.initialize_system())
 339 
 340     loader.load(sess, tag_set.split(','), saved_model_dir)
 341 
 342     if tf_debug:
 343       sess = local_cli_wrapper.LocalCLIDebugWrapperSession(sess)
 344 
 345     outputs = sess.run(output_tensor_names_sorted, feed_dict=inputs_feed_dict)
 346 
 347     for i, output in enumerate(outputs):
 348       output_tensor_key = output_tensor_keys_sorted[i]
 349       print('Result for output key %s:\n%s' % (output_tensor_key, output))
 350 
 351       # Only save if outdir is specified.
 352       if outdir:
 353         # Create directory if outdir does not exist
 354         if not os.path.isdir(outdir):
 355           os.makedirs(outdir)
 356         output_full_path = os.path.join(outdir, output_tensor_key + '.npy')
 357 
 358         # If overwrite not enabled and file already exist, error out
 359         if not overwrite_flag and os.path.exists(output_full_path):
 360           raise RuntimeError(
 361               'Output file %s already exists. Add \"--overwrite\" to overwrite'
 362               ' the existing output files.' % output_full_path)
 363 
 364         np.save(output_full_path, output)
 365         print('Output %s is saved to %s' % (output_tensor_key,
 366                                             output_full_path))

 369 def preprocess_inputs_arg_string(inputs_str):
 370   """Parses input arg into dictionary that maps input to file/variable tuple.
 371 
 372   Parses input string in the format of, for example,
 373   "input1=filename1[variable_name1],input2=filename2" into a
 374   dictionary looks like
 375   {'input_key1': (filename1, variable_name1),
 376    'input_key2': (file2, None)}
 377   , which maps input keys to a tuple of file name and variable name(None if
 378   empty).
 379 
 380   Args:
 381     inputs_str: A string that specified where to load inputs. Inputs are
 382     separated by semicolons.
 383         * For each input key:
 384             '<input_key>=<filename>' or
 385             '<input_key>=<filename>[<variable_name>]'
 386         * The optional 'variable_name' key will be set to None if not specified.
 387 
 388   Returns:
 389     A dictionary that maps input keys to a tuple of file name and variable name.
 390 
 391   Raises:
 392     RuntimeError: An error when the given input string is in a bad format.
 393   """
 394   input_dict = {}
 395   inputs_raw = inputs_str.split(';')
 396   for input_raw in filter(bool, inputs_raw):  # skip empty strings
 397     # Format of input=filename[variable_name]'
 398     match = re.match(r'([^=]+)=([^\[\]]+)\[([^\[\]]+)\]$', input_raw)
 399 
 400     if match:
 401       input_dict[match.group(1)] = match.group(2), match.group(3)
 402     else:
 403       # Format of input=filename'
 404       match = re.match(r'([^=]+)=([^\[\]]+)$', input_raw)
 405       if match:
 406         input_dict[match.group(1)] = match.group(2), None
 407       else:
 408         raise RuntimeError(
 409             '--inputs "%s" format is incorrect. Please follow'
 410             '"<input_key>=<filename>", or'
 411             '"<input_key>=<filename>[<variable_name>]"' % input_raw)
 412 
 413   return input_dict

 416 def preprocess_input_exprs_arg_string(input_exprs_str):
 417   """Parses input arg into dictionary that maps input key to python expression.
 418 
 419   Parses input string in the format of 'input_key=<python expression>' into a
 420   dictionary that maps each input_key to its python expression.
 421 
 422   Args:
 423     input_exprs_str: A string that specifies python expression for input keys.
 424     Each input is separated by semicolon. For each input key:
 425         'input_key=<python expression>'
 426 
 427   Returns:
 428     A dictionary that maps input keys to their values.
 429 
 430   Raises:
 431     RuntimeError: An error when the given input string is in a bad format.
 432   """
 433   input_dict = {}
 434 
 435   for input_raw in filter(bool, input_exprs_str.split(';')):
 436     if '=' not in input_exprs_str:
 437       raise RuntimeError('--input_exprs "%s" format is incorrect. Please follow'
 438                          '"<input_key>=<python expression>"' % input_exprs_str)
 439     input_key, expr = input_raw.split('=', 1)
 440     # ast.literal_eval does not work with numpy expressions
 441     input_dict[input_key] = eval(expr)  # pylint: disable=eval-used
 442   return input_dict

 445 def preprocess_input_examples_arg_string(input_examples_str):
 446   """Parses input into dict that maps input keys to lists of tf.Example.
 447 
 448   Parses input string in the format of 'input_key1=[{feature_name:
 449   feature_list}];input_key2=[{feature_name:feature_list}];' into a dictionary
 450   that maps each input_key to its list of serialized tf.Example.
 451 
 452   Args:
 453     input_examples_str: A string that specifies a list of dictionaries of
 454     feature_names and their feature_lists for each input.
 455     Each input is separated by semicolon. For each input key:
 456       'input=[{feature_name1: feature_list1, feature_name2:feature_list2}]'
 457       items in feature_list can be the type of float, int, long or str.
 458 
 459   Returns:
 460     A dictionary that maps input keys to lists of serialized tf.Example.
 461 
 462   Raises:
 463     ValueError: An error when the given tf.Example is not a list.
 464   """
 465   input_dict = preprocess_input_exprs_arg_string(input_examples_str)
 466   for input_key, example_list in input_dict.items():
 467     if not isinstance(example_list, list):
 468       raise ValueError(
 469           'tf.Example input must be a list of dictionaries, but "%s" is %s' %
 470           (example_list, type(example_list)))
 471     input_dict[input_key] = [
 472         _create_example_string(example) for example in example_list
 473     ]
 474   return input_dict

 477 def _create_example_string(example_dict):
 478   """Create a serialized tf.example from feature dictionary."""
 479   example = example_pb2.Example()
 480   for feature_name, feature_list in example_dict.items():
 481     if not isinstance(feature_list, list):
 482       raise ValueError('feature value must be a list, but %s: "%s" is %s' %
 483                        (feature_name, feature_list, type(feature_list)))
 484     if isinstance(feature_list[0], float):
 485       example.features.feature[feature_name].float_list.value.extend(
 486           feature_list)
 487     elif isinstance(feature_list[0], str):
 488       example.features.feature[feature_name].bytes_list.value.extend(
 489           feature_list)
 490     elif isinstance(feature_list[0], integer_types):
 491       example.features.feature[feature_name].int64_list.value.extend(
 492           feature_list)
 493     else:
 494       raise ValueError(
 495           'Type %s for value %s is not supported for tf.train.Feature.' %
 496           (type(feature_list[0]), feature_list[0]))
 497   return example.SerializeToString()

 500 def load_inputs_from_input_arg_string(inputs_str, input_exprs_str,
 501                                       input_examples_str):
 502   """Parses input arg strings and create inputs feed_dict.
 503 
 504   Parses '--inputs' string for inputs to be loaded from file, and parses
 505   '--input_exprs' string for inputs to be evaluated from python expression.
 506   '--input_examples' string for inputs to be created from tf.example feature
 507   dictionary list.
 508 
 509   Args:
 510     inputs_str: A string that specified where to load inputs. Each input is
 511         separated by semicolon.
 512         * For each input key:
 513             '<input_key>=<filename>' or
 514             '<input_key>=<filename>[<variable_name>]'
 515         * The optional 'variable_name' key will be set to None if not specified.
 516         * File specified by 'filename' will be loaded using numpy.load. Inputs
 517             can be loaded from only .npy, .npz or pickle files.
 518         * The "[variable_name]" key is optional depending on the input file type
 519             as descripted in more details below.
 520         When loading from a npy file, which always contains a numpy ndarray, the
 521         content will be directly assigned to the specified input tensor. If a
 522         variable_name is specified, it will be ignored and a warning will be
 523         issued.
 524         When loading from a npz zip file, user can specify which variable within
 525         the zip file to load for the input tensor inside the square brackets. If
 526         nothing is specified, this function will check that only one file is
 527         included in the zip and load it for the specified input tensor.
 528         When loading from a pickle file, if no variable_name is specified in the
 529         square brackets, whatever that is inside the pickle file will be passed
 530         to the specified input tensor, else SavedModel CLI will assume a
 531         dictionary is stored in the pickle file and the value corresponding to
 532         the variable_name will be used.
 533     input_exprs_str: A string that specifies python expressions for inputs.
 534         * In the format of: '<input_key>=<python expression>'.
 535         * numpy module is available as np.
 536     input_examples_str: A string that specifies tf.Example with dictionary.
 537         * In the format of: '<input_key>=<[{feature:value list}]>'
 538 
 539   Returns:
 540     A dictionary that maps input tensor keys to numpy ndarrays.
 541 
 542   Raises:
 543     RuntimeError: An error when a key is specified, but the input file contains
 544         multiple numpy ndarrays, none of which matches the given key.
 545     RuntimeError: An error when no key is specified, but the input file contains
 546         more than one numpy ndarrays.
 547   """
 548   tensor_key_feed_dict = {}
 549 
 550   inputs = preprocess_inputs_arg_string(inputs_str)
 551   input_exprs = preprocess_input_exprs_arg_string(input_exprs_str)
 552   input_examples = preprocess_input_examples_arg_string(input_examples_str)
 553 
 554   for input_tensor_key, (filename, variable_name) in inputs.items():
 555     data = np.load(file_io.FileIO(filename, mode='rb'))
 556 
 557     # When a variable_name key is specified for the input file
 558     if variable_name:
 559       # if file contains a single ndarray, ignore the input name
 560       if isinstance(data, np.ndarray):
 561         warnings.warn(
 562             'Input file %s contains a single ndarray. Name key \"%s\" ignored.'
 563             % (filename, variable_name))
 564         tensor_key_feed_dict[input_tensor_key] = data
 565       else:
 566         if variable_name in data:
 567           tensor_key_feed_dict[input_tensor_key] = data[variable_name]
 568         else:
 569           raise RuntimeError(
 570               'Input file %s does not contain variable with name \"%s\".' %
 571               (filename, variable_name))
 572     # When no key is specified for the input file.
 573     else:
 574       # Check if npz file only contains a single numpy ndarray.
 575       if isinstance(data, np.lib.npyio.NpzFile):
 576         variable_name_list = data.files
 577         if len(variable_name_list) != 1:
 578           raise RuntimeError(
 579               'Input file %s contains more than one ndarrays. Please specify '
 580               'the name of ndarray to use.' % filename)
 581         tensor_key_feed_dict[input_tensor_key] = data[variable_name_list[0]]
 582       else:
 583         tensor_key_feed_dict[input_tensor_key] = data
 584 
 585   # When input is a python expression:
 586   for input_tensor_key, py_expr_evaluated in input_exprs.items():
 587     if input_tensor_key in tensor_key_feed_dict:
 588       warnings.warn(
 589           'input_key %s has been specified with both --inputs and --input_exprs'
 590           ' options. Value in --input_exprs will be used.' % input_tensor_key)
 591     tensor_key_feed_dict[input_tensor_key] = py_expr_evaluated
 592 
 593   # When input is a tf.Example:
 594   for input_tensor_key, example in input_examples.items():
 595     if input_tensor_key in tensor_key_feed_dict:
 596       warnings.warn(
 597           'input_key %s has been specified in multiple options. Value in '
 598           '--input_examples will be used.' % input_tensor_key)
 599     tensor_key_feed_dict[input_tensor_key] = example
 600   return tensor_key_feed_dict

 603 def show(args):
 604   """Function triggered by show command.
 605 
 606   Args:
 607     args: A namespace parsed from command line.
 608   """
 609   # If all tag is specified, display all information.
 610   if args.all:
 611     _show_all(args.dir)
 612   else:
 613     # If no tag is specified, display all tag_set, if no signaure_def key is
 614     # specified, display all SignatureDef keys, else show input output tensor
 615     # information corresponding to the given SignatureDef key
 616     if args.tag_set is None:
 617       _show_tag_sets(args.dir)
 618     else:
 619       if args.signature_def is None:
 620         _show_signature_def_map_keys(args.dir, args.tag_set)
 621       else:
 622         _show_inputs_outputs(args.dir, args.tag_set, args.signature_def)

 625 def run(args):
 626   """Function triggered by run command.
 627 
 628   Args:
 629     args: A namespace parsed from command line.
 630 
 631   Raises:
 632     AttributeError: An error when neither --inputs nor --input_exprs is passed
 633     to run command.
 634   """
 635   if not args.inputs and not args.input_exprs and not args.input_examples:
 636     raise AttributeError(
 637         'At least one of --inputs, --input_exprs or --input_examples must be '
 638         'required')
 639   tensor_key_feed_dict = load_inputs_from_input_arg_string(
 640       args.inputs, args.input_exprs, args.input_examples)
 641   run_saved_model_with_feed_dict(args.dir, args.tag_set, args.signature_def,
 642                                  tensor_key_feed_dict, args.outdir,
 643                                  args.overwrite, worker=args.worker,
 644                                  init_tpu=args.init_tpu, tf_debug=args.tf_debug)

 647 def scan(args):
 648   """Function triggered by scan command.
 649 
 650   Args:
 651     args: A namespace parsed from command line.
 652   """
 653   if args.tag_set:
 654     scan_meta_graph_def(
 655         saved_model_utils.get_meta_graph_def(args.dir, args.tag_set))
 656   else:
 657     saved_model = reader.read_saved_model(args.dir)
 658     for meta_graph_def in saved_model.meta_graphs:
 659       scan_meta_graph_def(meta_graph_def)

 662 def create_parser():
 663   """Creates a parser that parse the command line arguments.
 664 
 665   Returns:
 666     A namespace parsed from command line arguments.
 667   """
 668   parser = argparse.ArgumentParser(
 669       description='saved_model_cli: Command-line interface for SavedModel')
 670   parser.add_argument('-v', '--version', action='version', version='0.1.0')
 671 
 672   subparsers = parser.add_subparsers(
 673       title='commands', description='valid commands', help='additional help')
 674 
 675   # show command
 676   show_msg = (
 677       'Usage examples:\n'
 678       'To show all tag-sets in a SavedModel:\n'
 679       '$saved_model_cli show --dir /tmp/saved_model\n\n'
 680       'To show all available SignatureDef keys in a '
 681       'MetaGraphDef specified by its tag-set:\n'
 682       '$saved_model_cli show --dir /tmp/saved_model --tag_set serve\n\n'
 683       'For a MetaGraphDef with multiple tags in the tag-set, all tags must be '
 684       'passed in, separated by \';\':\n'
 685       '$saved_model_cli show --dir /tmp/saved_model --tag_set serve,gpu\n\n'
 686       'To show all inputs and outputs TensorInfo for a specific'
 687       ' SignatureDef specified by the SignatureDef key in a'
 688       ' MetaGraph.\n'
 689       '$saved_model_cli show --dir /tmp/saved_model --tag_set serve'
 690       ' --signature_def serving_default\n\n'
 691       'To show all available information in the SavedModel:\n'
 692       '$saved_model_cli show --dir /tmp/saved_model --all')
 693   parser_show = subparsers.add_parser(
 694       'show',
 695       description=show_msg,
 696       formatter_class=argparse.RawTextHelpFormatter)
 697   parser_show.add_argument(
 698       '--dir',
 699       type=str,
 700       required=True,
 701       help='directory containing the SavedModel to inspect')
 702   parser_show.add_argument(
 703       '--all',
 704       action='store_true',
 705       help='if set, will output all information in given SavedModel')
 706   parser_show.add_argument(
 707       '--tag_set',
 708       type=str,
 709       default=None,
 710       help='tag-set of graph in SavedModel to show, separated by \',\'')
 711   parser_show.add_argument(
 712       '--signature_def',
 713       type=str,
 714       default=None,
 715       metavar='SIGNATURE_DEF_KEY',
 716       help='key of SignatureDef to display input(s) and output(s) for')
 717   parser_show.set_defaults(func=show)
 718 
 719   # run command
 720   run_msg = ('Usage example:\n'
 721              'To run input tensors from files through a MetaGraphDef and save'
 722              ' the output tensors to files:\n'
 723              '$saved_model_cli show --dir /tmp/saved_model --tag_set serve \\\n'
 724              '   --signature_def serving_default \\\n'
 725              '   --inputs input1_key=/tmp/124.npz[x],input2_key=/tmp/123.npy '
 726              '\\\n'
 727              '   --input_exprs \'input3_key=np.ones(2)\' \\\n'
 728              '   --input_examples '
 729              '\'input4_key=[{"id":[26],"weights":[0.5, 0.5]}]\' \\\n'
 730              '   --outdir=/out\n\n'
 731              'For more information about input file format, please see:\n'
 732              'https://www.tensorflow.org/guide/saved_model_cli\n')
 733   parser_run = subparsers.add_parser(
 734       'run', description=run_msg, formatter_class=argparse.RawTextHelpFormatter)
 735   parser_run.add_argument(
 736       '--dir',
 737       type=str,
 738       required=True,
 739       help='directory containing the SavedModel to execute')
 740   parser_run.add_argument(
 741       '--tag_set',
 742       type=str,
 743       required=True,
 744       help='tag-set of graph in SavedModel to load, separated by \',\'')
 745   parser_run.add_argument(
 746       '--signature_def',
 747       type=str,
 748       required=True,
 749       metavar='SIGNATURE_DEF_KEY',
 750       help='key of SignatureDef to run')
 751   msg = ('Loading inputs from files, in the format of \'<input_key>=<filename>,'
 752          ' or \'<input_key>=<filename>[<variable_name>]\', separated by \';\'.'
 753          ' The file format can only be from .npy, .npz or pickle.')
 754   parser_run.add_argument('--inputs', type=str, default='', help=msg)
 755   msg = ('Specifying inputs by python expressions, in the format of'
 756          ' "<input_key>=\'<python expression>\'", separated by \';\'. '
 757          'numpy module is available as \'np\'. '
 758          'Will override duplicate input keys from --inputs option.')
 759   parser_run.add_argument('--input_exprs', type=str, default='', help=msg)
 760   msg = (
 761       'Specifying tf.Example inputs as list of dictionaries. For example: '
 762       '<input_key>=[{feature0:value_list,feature1:value_list}]. Use ";" to '
 763       'separate input keys. Will override duplicate input keys from --inputs '
 764       'and --input_exprs option.')
 765   parser_run.add_argument('--input_examples', type=str, default='', help=msg)
 766   parser_run.add_argument(
 767       '--outdir',
 768       type=str,
 769       default=None,
 770       help='if specified, output tensor(s) will be saved to given directory')
 771   parser_run.add_argument(
 772       '--overwrite',
 773       action='store_true',
 774       help='if set, output file will be overwritten if it already exists.')
 775   parser_run.add_argument(
 776       '--tf_debug',
 777       action='store_true',
 778       help='if set, will use TensorFlow Debugger (tfdbg) to watch the '
 779            'intermediate Tensors and runtime GraphDefs while running the '
 780            'SavedModel.')
 781   parser_run.add_argument(
 782       '--worker',
 783       type=str,
 784       default=None,
 785       help='if specified, a Session will be run on the worker. '
 786            'Valid worker specification is a bns or gRPC path.')
 787   parser_run.add_argument(
 788       '--init_tpu',
 789       action='store_true',
 790       default=None,
 791       help='if specified, tpu.initialize_system will be called on the Session. '
 792            'This option should be only used if the worker is a TPU job.')
 793   parser_run.set_defaults(func=run)
 794 
 795   # scan command
 796   scan_msg = ('Usage example:\n'
 797               'To scan for blacklisted ops in SavedModel:\n'
 798               '$saved_model_cli scan --dir /tmp/saved_model\n'
 799               'To scan a specific MetaGraph, pass in --tag_set\n')
 800   parser_scan = subparsers.add_parser(
 801       'scan',
 802       description=scan_msg,
 803       formatter_class=argparse.RawTextHelpFormatter)
 804   parser_scan.add_argument(
 805       '--dir',
 806       type=str,
 807       required=True,
 808       help='directory containing the SavedModel to execute')
 809   parser_scan.add_argument(
 810       '--tag_set',
 811       type=str,
 812       help='tag-set of graph in SavedModel to scan, separated by \',\'')
 813   parser_scan.set_defaults(func=scan)
 814 
 815   return parser
