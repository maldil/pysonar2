  16 r"""Removes parts of a graph that are only needed for training.
  17 
  18 There are several common transformations that can be applied to GraphDefs
  19 created to train a model, that help reduce the amount of computation needed when
  20 the network is used only for inference. These include:
  21 
  22  - Removing training-only operations like checkpoint saving.
  23 
  24  - Stripping out parts of the graph that are never reached.
  25 
  26  - Removing debug operations like CheckNumerics.
  27 
  28  - Folding batch normalization ops into the pre-calculated weights.
  29 
  30  - Fusing common operations into unified versions.
  31 
  32 This script takes a frozen GraphDef file (where the weight variables have been
  33 converted into constants by the freeze_graph script) and outputs a new GraphDef
  34 with the optimizations applied.
  35 
  36 An example of command-line usage is:
  37 
  38 bazel build tensorflow/python/tools:optimize_for_inference && \
  39 bazel-bin/tensorflow/python/tools/optimize_for_inference \
  40 --input_graph=some_graph_def.pb \
  41 --output_graph=/tmp/optimized_graph.pb \
  42 --input_names=Mul \
  43 --output_names=softmax
  44 
  45 """

  70 # Support folding two types of batch norm ops:
  71 # BatchNormWithGlobalNormalization and FusedBatchNorm.  The two types only
  72 # differ in input order and attribute names, so we've collected their
  73 # differences up front.
  74 INPUT_ORDER = {
  75     # Order of inputs for BatchNormWithGlobalNormalization.
  76     "BatchNormWithGlobalNormalization": [
  77         "conv_op", "mean_op", "var_op", "beta_op", "gamma_op"
  78     ],
  79     # Order of inputs for FusedBatchNorm.
  80     "FusedBatchNorm": ["conv_op", "gamma_op", "beta_op", "mean_op", "var_op"]
  81 }
  82 # Name of the attribute epsilon value is stored in.
  83 EPSILON_ATTR = {
  84     "BatchNormWithGlobalNormalization": "variance_epsilon",
  85     "FusedBatchNorm": "epsilon"
  86 }

  89 def optimize_for_inference(input_graph_def, input_node_names, output_node_names,
  90                            placeholder_type_enum, toco_compatible=False):
  91   """Applies a series of inference optimizations on the input graph.
  92 
  93   Args:
  94     input_graph_def: A GraphDef containing a training model.
  95     input_node_names: A list of names of the nodes that are fed inputs during
  96       inference.
  97     output_node_names: A list of names of the nodes that produce the final
  98       results.
  99     placeholder_type_enum: The AttrValue enum for the placeholder data type, or
 100         a list that specifies one value per input node name.
 101     toco_compatible: Boolean, if True, only runs optimizations that result in
 102       TOCO compatible graph operations (default=False).
 103 
 104   Returns:
 105     An optimized version of the input graph.
 106   """
 107   ensure_graph_is_valid(input_graph_def)
 108   optimized_graph_def = input_graph_def
 109   optimized_graph_def = strip_unused_lib.strip_unused(
 110       optimized_graph_def, input_node_names, output_node_names,
 111       placeholder_type_enum)
 112   optimized_graph_def = graph_util.remove_training_nodes(
 113       optimized_graph_def, output_node_names)
 114   optimized_graph_def = fold_batch_norms(optimized_graph_def)
 115   if not toco_compatible:
 116     optimized_graph_def = fuse_resize_and_conv(optimized_graph_def,
 117                                                output_node_names)
 118   ensure_graph_is_valid(optimized_graph_def)
 119   return optimized_graph_def

 122 def ensure_graph_is_valid(graph_def):
 123   """Makes sure that the graph is internally consistent.
 124 
 125   Checks basic properties of the graph def and raises an exception if there are
 126   input references to missing nodes, duplicated names, or other logic errors.
 127 
 128   Args:
 129     graph_def: Definition of a graph to be checked.
 130 
 131   Raises:
 132     ValueError: If the graph is incorrectly constructed.
 133   """
 134   node_map = {}
 135   for node in graph_def.node:
 136     if node.name not in node_map:
 137       node_map[node.name] = node
 138     else:
 139       raise ValueError("Duplicate node names detected for ", node.name)
 140   for node in graph_def.node:
 141     for input_name in node.input:
 142       input_node_name = node_name_from_input(input_name)
 143       if input_node_name not in node_map:
 144         raise ValueError("Input for ", node.name, " not found: ", input_name)
 145 
 146 
 147 def node_name_from_input(node_name):
 148   """Strips off ports and other decorations to get the underlying node name."""
 149   if node_name.startswith("^"):
 150     node_name = node_name[1:]
 151   m = re.search(r"(.*):\d+$", node_name)
 152   if m:
 153     node_name = m.group(1)
 154   return node_name

 157 def node_from_map(node_map, name):
 158   """Pulls a node def from a dictionary for a given name.
 159 
 160   Args:
 161     node_map: Dictionary containing an entry indexed by name for every node.
 162     name: Identifies the node we want to find.
 163 
 164   Returns:
 165     NodeDef of the node with the given name.
 166 
 167   Raises:
 168     ValueError: If the node isn't present in the dictionary.
 169   """
 170   stripped_name = node_name_from_input(name)
 171   if stripped_name not in node_map:
 172     raise ValueError("No node named '%s' found in map." % name)
 173   return node_map[stripped_name]

 176 def values_from_const(node_def):
 177   """Extracts the values from a const NodeDef as a numpy ndarray.
 178 
 179   Args:
 180     node_def: Const NodeDef that has the values we want to access.
 181 
 182   Returns:
 183     Numpy ndarray containing the values.
 184 
 185   Raises:
 186     ValueError: If the node isn't a Const.
 187   """
 188   if node_def.op != "Const":
 189     raise ValueError(
 190         "Node named '%s' should be a Const op for values_from_const." %
 191         node_def.name)
 192   input_tensor = node_def.attr["value"].tensor
 193   tensor_value = tensor_util.MakeNdarray(input_tensor)
 194   return tensor_value

 197 # Whether to scale by gamma after normalization.
 198 def scale_after_normalization(node):
 199   if node.op == "BatchNormWithGlobalNormalization":
 200     return node.attr["scale_after_normalization"].b
 201   return True

 204 def fold_batch_norms(input_graph_def):
 205   """Removes batch normalization ops by folding them into convolutions.
 206 
 207   Batch normalization during training has multiple dynamic parameters that are
 208   updated, but once the graph is finalized these become constants. That means
 209   there's an opportunity to reduce the computations down to a scale and
 210   addition, rather than the more expensive multiple ops, and even bake the
 211   scaling into the convolution weights. This function identifies the typical
 212   pattern of batch normalization subgraphs, and performs the transformation to
 213   fold the computations down into a simpler form. It currently only spots batch
 214   normalization that's performed by the BatchNormWithGlobalNormalization op, and
 215   will need to be extended in the future to handle the newer style.
 216 
 217   Args:
 218     input_graph_def: A GraphDef containing a model.
 219 
 220   Returns:
 221     Modified graph with BN ops removed, and modified weights.
 222 
 223   Raises:
 224     ValueError: If the graph is badly formed with duplicate node names.
 225   """
 226   input_node_map = {}
 227   for node in input_graph_def.node:
 228     if node.name not in input_node_map:
 229       input_node_map[node.name] = node
 230     else:
 231       raise ValueError("Duplicate node names detected for ", node.name)
 232 
 233   nodes_to_skip = {}
 234   new_ops = []
 235   for node in input_graph_def.node:
 236     if node.op not in ("BatchNormWithGlobalNormalization", "FusedBatchNorm"):
 237       continue
 238 
 239     conv_op = node_from_map(input_node_map,
 240                             node.input[INPUT_ORDER[node.op].index("conv_op")])
 241     if conv_op.op != "Conv2D":
 242       tf_logging.warning(
 243           "Didn't find expected Conv2D input to '%s'" % node.name)
 244       continue
 245 
 246     weights_op = node_from_map(input_node_map, conv_op.input[1])
 247     if weights_op.op != "Const":
 248       tf_logging.warning("Didn't find expected conv Constant input to '%s',"
 249                          " found %s instead. Maybe because freeze_graph wasn't"
 250                          " run first?" % (conv_op.name, weights_op))
 251       continue
 252     weights = values_from_const(weights_op)
 253     channel_count = weights.shape[3]
 254 
 255     mean_op = node_from_map(input_node_map,
 256                             node.input[INPUT_ORDER[node.op].index("mean_op")])
 257     if mean_op.op != "Const":
 258       tf_logging.warning("Didn't find expected mean Constant input to '%s',"
 259                          " found %s instead. Maybe because freeze_graph wasn't"
 260                          " run first?" % (node.name, mean_op))
 261       continue
 262     mean_value = values_from_const(mean_op)
 263     if mean_value.shape != (channel_count,):
 264       tf_logging.warning("Incorrect shape for mean, found %s, expected %s,"
 265                          " for node %s" % (str(mean_value.shape), str(
 266                              (channel_count,)), node.name))
 267       continue
 268 
 269     var_op = node_from_map(input_node_map,
 270                            node.input[INPUT_ORDER[node.op].index("var_op")])
 271     if var_op.op != "Const":
 272       tf_logging.warning("Didn't find expected var Constant input to '%s',"
 273                          " found %s instead. Maybe because freeze_graph wasn't"
 274                          " run first?" % (node.name, var_op))
 275       continue
 276     var_value = values_from_const(var_op)
 277     if var_value.shape != (channel_count,):
 278       tf_logging.warning("Incorrect shape for var, found %s, expected %s,"
 279                          " for node %s" % (str(var_value.shape), str(
 280                              (channel_count,)), node.name))
 281       continue
 282 
 283     beta_op = node_from_map(input_node_map,
 284                             node.input[INPUT_ORDER[node.op].index("beta_op")])
 285     if beta_op.op != "Const":
 286       tf_logging.warning("Didn't find expected beta Constant input to '%s',"
 287                          " found %s instead. Maybe because freeze_graph wasn't"
 288                          " run first?" % (node.name, beta_op))
 289       continue
 290     beta_value = values_from_const(beta_op)
 291     if beta_value.shape != (channel_count,):
 292       tf_logging.warning("Incorrect shape for beta, found %s, expected %s,"
 293                          " for node %s" % (str(beta_value.shape), str(
 294                              (channel_count,)), node.name))
 295       continue
 296 
 297     gamma_op = node_from_map(input_node_map,
 298                              node.input[INPUT_ORDER[node.op].index("gamma_op")])
 299     if gamma_op.op != "Const":
 300       tf_logging.warning("Didn't find expected gamma Constant input to '%s',"
 301                          " found %s instead. Maybe because freeze_graph wasn't"
 302                          " run first?" % (node.name, gamma_op))
 303       continue
 304     gamma_value = values_from_const(gamma_op)
 305     if gamma_value.shape != (channel_count,):
 306       tf_logging.warning("Incorrect shape for gamma, found %s, expected %s,"
 307                          " for node %s" % (str(gamma_value.shape), str(
 308                              (channel_count,)), node.name))
 309       continue
 310 
 311     variance_epsilon_value = node.attr[EPSILON_ATTR[node.op]].f
 312     nodes_to_skip[node.name] = True
 313     nodes_to_skip[weights_op.name] = True
 314     nodes_to_skip[mean_op.name] = True
 315     nodes_to_skip[var_op.name] = True
 316     nodes_to_skip[beta_op.name] = True
 317     nodes_to_skip[gamma_op.name] = True
 318     nodes_to_skip[conv_op.name] = True
 319 
 320     if scale_after_normalization(node):
 321       scale_value = (
 322           (1.0 / np.vectorize(math.sqrt)(var_value + variance_epsilon_value)) *
 323           gamma_value)
 324     else:
 325       scale_value = (
 326           1.0 / np.vectorize(math.sqrt)(var_value + variance_epsilon_value))
 327     offset_value = (-mean_value * scale_value) + beta_value
 328     scaled_weights = np.copy(weights)
 329     it = np.nditer(
 330         scaled_weights, flags=["multi_index"], op_flags=["readwrite"])
 331     while not it.finished:
 332       current_scale = scale_value[it.multi_index[3]]
 333       it[0] *= current_scale
 334       it.iternext()
 335     scaled_weights_op = node_def_pb2.NodeDef()
 336     scaled_weights_op.op = "Const"
 337     scaled_weights_op.name = weights_op.name
 338     scaled_weights_op.attr["dtype"].CopyFrom(weights_op.attr["dtype"])
 339     scaled_weights_op.attr["value"].CopyFrom(
 340         attr_value_pb2.AttrValue(tensor=tensor_util.make_tensor_proto(
 341             scaled_weights, weights.dtype.type, weights.shape)))
 342     new_conv_op = node_def_pb2.NodeDef()
 343     new_conv_op.CopyFrom(conv_op)
 344     offset_op = node_def_pb2.NodeDef()
 345     offset_op.op = "Const"
 346     offset_op.name = conv_op.name + "_bn_offset"
 347     offset_op.attr["dtype"].CopyFrom(mean_op.attr["dtype"])
 348     offset_op.attr["value"].CopyFrom(
 349         attr_value_pb2.AttrValue(tensor=tensor_util.make_tensor_proto(
 350             offset_value, mean_value.dtype.type, offset_value.shape)))
 351     bias_add_op = node_def_pb2.NodeDef()
 352     bias_add_op.op = "BiasAdd"
 353     bias_add_op.name = node.name
 354     bias_add_op.attr["T"].CopyFrom(conv_op.attr["T"])
 355     bias_add_op.attr["data_format"].CopyFrom(conv_op.attr["data_format"])
 356     bias_add_op.input.extend([new_conv_op.name, offset_op.name])
 357     new_ops.extend([scaled_weights_op, new_conv_op, offset_op, bias_add_op])
 358 
 359   result_graph_def = graph_pb2.GraphDef()
 360   for node in input_graph_def.node:
 361     if node.name in nodes_to_skip:
 362       continue
 363     new_node = node_def_pb2.NodeDef()
 364     new_node.CopyFrom(node)
 365     result_graph_def.node.extend([new_node])
 366 
 367   result_graph_def.node.extend(new_ops)
 368   return result_graph_def

 371 def fuse_resize_and_conv(input_graph_def, output_node_names):
 372   """Merges preceding resize and mirror pad ops into a specialized convolution.
 373 
 374   There's a common pattern of enlarging the input to a convolution using a
 375   resize operation, and also using MirrorPad to extend the boundaries to that
 376   zero edge pixels don't bleed inwards when convolving. This routine looks for
 377   that pattern of operations, and fuses them together into a Conv2DWithResizeOp.
 378 
 379   Args:
 380     input_graph_def: A GraphDef containing a model.
 381     output_node_names: A list of names of the nodes that produce the final
 382       results.
 383 
 384   Returns:
 385     Modified graph with resize and pad ops merged.
 386 
 387   Raises:
 388     ValueError: If the graph is badly formed with duplicate node names.
 389   """
 390 
 391   input_node_map = {}
 392   for node in input_graph_def.node:
 393     if node.name not in input_node_map:
 394       input_node_map[node.name] = node
 395     else:
 396       raise ValueError("Duplicate node names detected for ", node.name)
 397 
 398   node_reference_count = collections.defaultdict(int)
 399   for node in input_graph_def.node:
 400     for input_name in node.input:
 401       stripped_name = node_name_from_input(input_name)
 402       node_reference_count[stripped_name] += 1
 403   for output_name in output_node_names:
 404     node_reference_count[output_name] += 1
 405 
 406   new_ops = []
 407   for node in input_graph_def.node:
 408 
 409     if node.op != "Conv2D":
 410       continue
 411     conv_op = node
 412 
 413     input_op = node_from_map(input_node_map, conv_op.input[0])
 414     if input_op.op == "MirrorPad":
 415       mirror_pad_op = input_op
 416       resize_op = node_from_map(input_node_map, mirror_pad_op.input[0])
 417       if resize_op.op != "ResizeBilinear":
 418         resize_op = None
 419     else:
 420       mirror_pad_op = None
 421       if input_op.op == "ResizeBilinear":
 422         resize_op = input_op
 423       else:
 424         resize_op = None
 425 
 426     # There are no ops to be fused into the conv, so skip replacing this one.
 427     if not mirror_pad_op and not resize_op:
 428       continue
 429 
 430     # We're replacing this node, so make sure the old one is removed.
 431     node_reference_count[conv_op.name] = 0
 432     if mirror_pad_op:
 433       node_reference_count[mirror_pad_op.name] -= 1
 434     if resize_op:
 435       node_reference_count[resize_op.name] -= 1
 436 
 437     fused_conv_op = node_def_pb2.NodeDef()
 438     if resize_op:
 439       fused_conv_op.op = "FusedResizeAndPadConv2D"
 440     else:
 441       fused_conv_op.op = "FusedPadConv2D"
 442     fused_conv_op.name = conv_op.name
 443     if mirror_pad_op:
 444       mirror_paddings_name = mirror_pad_op.input[1]
 445       mirror_paddings_mode = mirror_pad_op.attr["mode"]
 446     else:
 447       # If there was no MirrorPad op, then create settings that make the padding
 448       # stage of the fused operation a no-op.
 449       paddings_op = node_def_pb2.NodeDef()
 450       paddings_op.op = "Const"
 451       paddings_op.name = conv_op.name + "_dummy_paddings"
 452       paddings_op.attr["dtype"].CopyFrom(
 453           attr_value_pb2.AttrValue(type=dtypes.int32.as_datatype_enum))
 454       paddings_op.attr["value"].CopyFrom(
 455           attr_value_pb2.AttrValue(tensor=tensor_util.make_tensor_proto(
 456               [0, 0, 0, 0, 0, 0, 0, 0], dtypes.int32, [4, 2])))
 457       new_ops.extend([paddings_op])
 458       mirror_paddings_name = paddings_op.name
 459       mirror_paddings_mode = attr_value_pb2.AttrValue(s=b"REFLECT")
 460     if resize_op:
 461       fused_conv_op.input.extend([
 462           resize_op.input[0], resize_op.input[1], mirror_paddings_name,
 463           conv_op.input[1]
 464       ])
 465       fused_conv_op.attr["resize_align_corners"].CopyFrom(
 466           resize_op.attr["align_corners"])
 467     else:
 468       fused_conv_op.input.extend(
 469           [mirror_pad_op.input[0], mirror_paddings_name, conv_op.input[1]])
 470     fused_conv_op.attr["T"].CopyFrom(conv_op.attr["T"])
 471     fused_conv_op.attr["mode"].CopyFrom(mirror_paddings_mode)
 472     fused_conv_op.attr["strides"].CopyFrom(conv_op.attr["strides"])
 473     fused_conv_op.attr["padding"].CopyFrom(conv_op.attr["padding"])
 474     new_ops.extend([fused_conv_op])
 475 
 476   result_graph_def = graph_pb2.GraphDef()
 477   for node in input_graph_def.node:
 478     if node_reference_count[node.name] < 1:
 479       continue
 480     new_node = node_def_pb2.NodeDef()
 481     new_node.CopyFrom(node)
 482     result_graph_def.node.extend([new_node])
 483 
 484   result_graph_def.node.extend(new_ops)
 485   return result_graph_def
