  38 @tf_export("io.TFRecordOptions", "python_io.TFRecordOptions")
  39 @deprecation.deprecated_endpoints("python_io.TFRecordOptions")
  40 class TFRecordOptions(object):
  41   """Options used for manipulating TFRecord files."""
  42   compression_type_map = {
  43       TFRecordCompressionType.ZLIB: "ZLIB",
  44       TFRecordCompressionType.GZIP: "GZIP",
  45       TFRecordCompressionType.NONE: ""
  46   }
  47 
  48   def __init__(self,
  49                compression_type=None,
  50                flush_mode=None,
  51                input_buffer_size=None,
  52                output_buffer_size=None,
  53                window_bits=None,
  54                compression_level=None,
  55                compression_method=None,
  56                mem_level=None,
  57                compression_strategy=None):
  58     # pylint: disable=line-too-long
  59     """Creates a `TFRecordOptions` instance.
  60 
  61     Options only effect TFRecordWriter when compression_type is not `None`.
  62     Documentation, details, and defaults can be found in
  63     [`zlib_compression_options.h`](https://www.tensorflow.org/code/tensorflow/core/lib/io/zlib_compression_options.h)
  64     and in the [zlib manual](http://www.zlib.net/manual.html).
  65     Leaving an option as `None` allows C++ to set a reasonable default.
  66 
  67     Args:
  68       compression_type: `TFRecordCompressionType` or `None`.
  69       flush_mode: flush mode or `None`, Default: Z_NO_FLUSH.
  70       input_buffer_size: int or `None`.
  71       output_buffer_size: int or `None`.
  72       window_bits: int or `None`.
  73       compression_level: 0 to 9, or `None`.
  74       compression_method: compression method or `None`.
  75       mem_level: 1 to 9, or `None`.
  76       compression_strategy: strategy or `None`. Default: Z_DEFAULT_STRATEGY.
  77 
  78     Returns:
  79       A `TFRecordOptions` object.
  80 
  81     Raises:
  82       ValueError: If compression_type is invalid.
  83     """
  84     # pylint: enable=line-too-long
  85     # Check compression_type is valid, but for backwards compatibility don't
  86     # immediately convert to a string.
  87     self.get_compression_type_string(compression_type)
  88     self.compression_type = compression_type
  89     self.flush_mode = flush_mode
  90     self.input_buffer_size = input_buffer_size
  91     self.output_buffer_size = output_buffer_size
  92     self.window_bits = window_bits
  93     self.compression_level = compression_level
  94     self.compression_method = compression_method
  95     self.mem_level = mem_level
  96     self.compression_strategy = compression_strategy
  97 
  98   @classmethod
  99   def get_compression_type_string(cls, options):
 100     """Convert various option types to a unified string.
 101 
 102     Args:
 103       options: `TFRecordOption`, `TFRecordCompressionType`, or string.
 104 
 105     Returns:
 106       Compression type as string (e.g. `'ZLIB'`, `'GZIP'`, or `''`).
 107 
 108     Raises:
 109       ValueError: If compression_type is invalid.
 110     """
 111     if not options:
 112       return ""
 113     elif isinstance(options, TFRecordOptions):
 114       return cls.get_compression_type_string(options.compression_type)
 115     elif isinstance(options, TFRecordCompressionType):
 116       return cls.compression_type_map[options]
 117     elif options in TFRecordOptions.compression_type_map:
 118       return cls.compression_type_map[options]
 119     elif options in TFRecordOptions.compression_type_map.values():
 120       return options
 121     else:
 122       raise ValueError('Not a valid compression_type: "{}"'.format(options))
 123 
 124   def _as_record_writer_options(self):
 125     """Convert to RecordWriterOptions for use with PyRecordWriter."""
 126     options = pywrap_tensorflow.RecordWriterOptions_CreateRecordWriterOptions(
 127         compat.as_bytes(
 128             self.get_compression_type_string(self.compression_type)))
 129 
 130     if self.flush_mode is not None:
 131       options.zlib_options.flush_mode = self.flush_mode
 132     if self.input_buffer_size is not None:
 133       options.zlib_options.input_buffer_size = self.input_buffer_size
 134     if self.output_buffer_size is not None:
 135       options.zlib_options.output_buffer_size = self.output_buffer_size
 136     if self.window_bits is not None:
 137       options.zlib_options.window_bits = self.window_bits
 138     if self.compression_level is not None:
 139       options.zlib_options.compression_level = self.compression_level
 140     if self.compression_method is not None:
 141       options.zlib_options.compression_method = self.compression_method
 142     if self.mem_level is not None:
 143       options.zlib_options.mem_level = self.mem_level
 144     if self.compression_strategy is not None:
 145       options.zlib_options.compression_strategy = self.compression_strategy
 146     return options

 149 @tf_export("io.tf_record_iterator", "python_io.tf_record_iterator")
 150 @deprecation.deprecated_endpoints("python_io.tf_record_iterator")
 151 def tf_record_iterator(path, options=None):
 152   """An iterator that read the records from a TFRecords file.
 153 
 154   Args:
 155     path: The path to the TFRecords file.
 156     options: (optional) A TFRecordOptions object.
 157 
 158   Yields:
 159     Strings.
 160 
 161   Raises:
 162     IOError: If `path` cannot be opened for reading.
 163   """
 164   compression_type = TFRecordOptions.get_compression_type_string(options)
 165   with errors.raise_exception_on_not_ok_status() as status:
 166     reader = pywrap_tensorflow.PyRecordReader_New(
 167         compat.as_bytes(path), 0, compat.as_bytes(compression_type), status)
 168 
 169   if reader is None:
 170     raise IOError("Could not open %s." % path)
 171   try:
 172     while True:
 173       try:
 174         reader.GetNext()
 175       except errors.OutOfRangeError:
 176         break
 177       yield reader.record()
 178   finally:
 179     reader.Close()

 182 @tf_export("io.TFRecordWriter", "python_io.TFRecordWriter")
 183 @deprecation.deprecated_endpoints("python_io.TFRecordWriter")
 184 class TFRecordWriter(object):
 185   """A class to write records to a TFRecords file.
 186 
 187   This class implements `__enter__` and `__exit__`, and can be used
 188   in `with` blocks like a normal file.
 189   """
 190 
 191   # TODO(josh11b): Support appending?
 192   def __init__(self, path, options=None):
 193     """Opens file `path` and creates a `TFRecordWriter` writing to it.
 194 
 195     Args:
 196       path: The path to the TFRecords file.
 197       options: (optional) String specifying compression type,
 198           `TFRecordCompressionType`, or `TFRecordOptions` object.
 199 
 200     Raises:
 201       IOError: If `path` cannot be opened for writing.
 202       ValueError: If valid compression_type can't be determined from `options`.
 203     """
 204     if not isinstance(options, TFRecordOptions):
 205       options = TFRecordOptions(compression_type=options)
 206 
 207     with errors.raise_exception_on_not_ok_status() as status:
 208       # pylint: disable=protected-access
 209       self._writer = pywrap_tensorflow.PyRecordWriter_New(
 210           compat.as_bytes(path), options._as_record_writer_options(), status)
 211       # pylint: enable=protected-access
 212 
 213   def __enter__(self):
 214     """Enter a `with` block."""
 215     return self
 216 
 217   def __exit__(self, unused_type, unused_value, unused_traceback):
 218     """Exit a `with` block, closing the file."""
 219     self.close()
 220 
 221   def write(self, record):
 222     """Write a string record to the file.
 223 
 224     Args:
 225       record: str
 226     """
 227     with errors.raise_exception_on_not_ok_status() as status:
 228       self._writer.WriteRecord(record, status)
 229 
 230   def flush(self):
 231     """Flush the file."""
 232     with errors.raise_exception_on_not_ok_status() as status:
 233       self._writer.Flush(status)
 234 
 235   def close(self):
 236     """Close the file."""
 237     with errors.raise_exception_on_not_ok_status() as status:
 238       self._writer.Close(status)
