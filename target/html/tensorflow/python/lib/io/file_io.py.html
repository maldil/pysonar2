  42 class FileIO(object):
  43   """FileIO class that exposes methods to read / write to / from files.
  44 
  45   The constructor takes the following arguments:
  46   name: name of the file
  47   mode: one of 'r', 'w', 'a', 'r+', 'w+', 'a+'. Append 'b' for bytes mode.
  48 
  49   Can be used as an iterator to iterate over lines in the file.
  50 
  51   The default buffer size used for the BufferedInputStream used for reading
  52   the file line by line is 1024 * 512 bytes.
  53   """
  54 
  55   def __init__(self, name, mode):
  56     self.__name = name
  57     self.__mode = mode
  58     self._read_buf = None
  59     self._writable_file = None
  60     self._binary_mode = "b" in mode
  61     mode = mode.replace("b", "")
  62     if mode not in ("r", "w", "a", "r+", "w+", "a+"):
  63       raise errors.InvalidArgumentError(
  64           None, None, "mode is not 'r' or 'w' or 'a' or 'r+' or 'w+' or 'a+'")
  65     self._read_check_passed = mode in ("r", "r+", "a+", "w+")
  66     self._write_check_passed = mode in ("a", "w", "r+", "a+", "w+")
  67 
  68   @property
  69   def name(self):
  70     """Returns the file name."""
  71     return self.__name
  72 
  73   @property
  74   def mode(self):
  75     """Returns the mode in which the file was opened."""
  76     return self.__mode
  77 
  78   def _preread_check(self):
  79     if not self._read_buf:
  80       if not self._read_check_passed:
  81         raise errors.PermissionDeniedError(None, None,
  82                                            "File isn't open for reading")
  83       with errors.raise_exception_on_not_ok_status() as status:
  84         self._read_buf = pywrap_tensorflow.CreateBufferedInputStream(
  85             compat.as_bytes(self.__name), 1024 * 512, status)
  86 
  87   def _prewrite_check(self):
  88     if not self._writable_file:
  89       if not self._write_check_passed:
  90         raise errors.PermissionDeniedError(None, None,
  91                                            "File isn't open for writing")
  92       with errors.raise_exception_on_not_ok_status() as status:
  93         self._writable_file = pywrap_tensorflow.CreateWritableFile(
  94             compat.as_bytes(self.__name), compat.as_bytes(self.__mode), status)
  95 
  96   def _prepare_value(self, val):
  97     if self._binary_mode:
  98       return compat.as_bytes(val)
  99     else:
 100       return compat.as_str_any(val)
 101 
 102   def size(self):
 103     """Returns the size of the file."""
 104     return stat(self.__name).length
 105 
 106   def write(self, file_content):
 107     """Writes file_content to the file. Appends to the end of the file."""
 108     self._prewrite_check()
 109     with errors.raise_exception_on_not_ok_status() as status:
 110       pywrap_tensorflow.AppendToFile(
 111           compat.as_bytes(file_content), self._writable_file, status)
 112 
 113   def read(self, n=-1):
 114     """Returns the contents of a file as a string.
 115 
 116     Starts reading from current position in file.
 117 
 118     Args:
 119       n: Read 'n' bytes if n != -1. If n = -1, reads to end of file.
 120 
 121     Returns:
 122       'n' bytes of the file (or whole file) in bytes mode or 'n' bytes of the
 123       string if in string (regular) mode.
 124     """
 125     self._preread_check()
 126     with errors.raise_exception_on_not_ok_status() as status:
 127       if n == -1:
 128         length = self.size() - self.tell()
 129       else:
 130         length = n
 131       return self._prepare_value(
 132           pywrap_tensorflow.ReadFromStream(self._read_buf, length, status))
 133 
 134   @deprecation.deprecated_args(
 135       None,
 136       "position is deprecated in favor of the offset argument.",
 137       "position")
 138   def seek(self, offset=None, whence=0, position=None):
 139     # TODO(jhseu): Delete later. Used to omit `position` from docs.
 140     # pylint: disable=g-doc-args
 141     """Seeks to the offset in the file.
 142 
 143     Args:
 144       offset: The byte count relative to the whence argument.
 145       whence: Valid values for whence are:
 146         0: start of the file (default)
 147         1: relative to the current position of the file
 148         2: relative to the end of file. offset is usually negative.
 149     """
 150     # pylint: enable=g-doc-args
 151     self._preread_check()
 152     # We needed to make offset a keyword argument for backwards-compatibility.
 153     # This check exists so that we can convert back to having offset be a
 154     # positional argument.
 155     # TODO(jhseu): Make `offset` a positional argument after `position` is
 156     # deleted.
 157     if offset is None and position is None:
 158       raise TypeError("seek(): offset argument required")
 159     if offset is not None and position is not None:
 160       raise TypeError("seek(): offset and position may not be set "
 161                       "simultaneously.")
 162 
 163     if position is not None:
 164       offset = position
 165 
 166     with errors.raise_exception_on_not_ok_status() as status:
 167       if whence == 0:
 168         pass
 169       elif whence == 1:
 170         offset += self.tell()
 171       elif whence == 2:
 172         offset += self.size()
 173       else:
 174         raise errors.InvalidArgumentError(
 175             None, None,
 176             "Invalid whence argument: {}. Valid values are 0, 1, or 2."
 177             .format(whence))
 178       ret_status = self._read_buf.Seek(offset)
 179       pywrap_tensorflow.Set_TF_Status_from_Status(status, ret_status)
 180 
 181   def readline(self):
 182     r"""Reads the next line from the file. Leaves the '\n' at the end."""
 183     self._preread_check()
 184     return self._prepare_value(self._read_buf.ReadLineAsString())
 185 
 186   def readlines(self):
 187     """Returns all lines from the file in a list."""
 188     self._preread_check()
 189     lines = []
 190     while True:
 191       s = self.readline()
 192       if not s:
 193         break
 194       lines.append(s)
 195     return lines
 196 
 197   def tell(self):
 198     """Returns the current position in the file."""
 199     self._preread_check()
 200     return self._read_buf.Tell()
 201 
 202   def __enter__(self):
 203     """Make usable with "with" statement."""
 204     return self
 205 
 206   def __exit__(self, unused_type, unused_value, unused_traceback):
 207     """Make usable with "with" statement."""
 208     self.close()
 209 
 210   def __iter__(self):
 211     return self
 212 
 213   def next(self):
 214     retval = self.readline()
 215     if not retval:
 216       raise StopIteration()
 217     return retval
 218 
 219   def __next__(self):
 220     return self.next()
 221 
 222   def flush(self):
 223     """Flushes the Writable file.
 224 
 225     This only ensures that the data has made its way out of the process without
 226     any guarantees on whether it's written to disk. This means that the
 227     data would survive an application crash but not necessarily an OS crash.
 228     """
 229     if self._writable_file:
 230       with errors.raise_exception_on_not_ok_status() as status:
 231         ret_status = self._writable_file.Flush()
 232         pywrap_tensorflow.Set_TF_Status_from_Status(status, ret_status)
 233 
 234   def close(self):
 235     """Closes FileIO. Should be called for the WritableFile to be flushed."""
 236     self._read_buf = None
 237     if self._writable_file:
 238       with errors.raise_exception_on_not_ok_status() as status:
 239         ret_status = self._writable_file.Close()
 240         pywrap_tensorflow.Set_TF_Status_from_Status(status, ret_status)
 241     self._writable_file = None

 244 @tf_export("gfile.Exists")
 245 def file_exists(filename):
 246   """Determines whether a path exists or not.
 247 
 248   Args:
 249     filename: string, a path
 250 
 251   Returns:
 252     True if the path exists, whether its a file or a directory.
 253     False if the path does not exist and there are no filesystem errors.
 254 
 255   Raises:
 256     errors.OpError: Propagates any errors reported by the FileSystem API.
 257   """
 258   try:
 259     with errors.raise_exception_on_not_ok_status() as status:
 260       pywrap_tensorflow.FileExists(compat.as_bytes(filename), status)
 261   except errors.NotFoundError:
 262     return False
 263   return True

 281 def read_file_to_string(filename, binary_mode=False):
 282   """Reads the entire contents of a file to a string.
 283 
 284   Args:
 285     filename: string, path to a file
 286     binary_mode: whether to open the file in binary mode or not. This changes
 287         the type of the object returned.
 288 
 289   Returns:
 290     contents of the file as a string or bytes.
 291 
 292   Raises:
 293     errors.OpError: Raises variety of errors that are subtypes e.g.
 294     NotFoundError etc.
 295   """
 296   if binary_mode:
 297     f = FileIO(filename, mode="rb")
 298   else:
 299     f = FileIO(filename, mode="r")
 300   return f.read()

 317 @tf_export("gfile.Glob")
 318 def get_matching_files(filename):
 319   """Returns a list of files that match the given pattern(s).
 320 
 321   Args:
 322     filename: string or iterable of strings. The glob pattern(s).
 323 
 324   Returns:
 325     A list of strings containing filenames that match the given pattern(s).
 326 
 327   Raises:
 328     errors.OpError: If there are filesystem / directory listing errors.
 329   """
 330   with errors.raise_exception_on_not_ok_status() as status:

 333           # Convert the filenames to string from bytes.

 338     else:
 339       return [
 340           # Convert the filenames to string from bytes.
 341           compat.as_str_any(matching_filename)
 342           for single_filename in filename
 343           for matching_filename in pywrap_tensorflow.GetMatchingFiles(
 344               compat.as_bytes(single_filename), status)
 345       ]

 418 def atomic_write_string_to_file(filename, contents, overwrite=True):
 419   """Writes to `filename` atomically.
 420 
 421   This means that when `filename` appears in the filesystem, it will contain
 422   all of `contents`. With write_string_to_file, it is possible for the file
 423   to appear in the filesystem with `contents` only partially written.
 424 
 425   Accomplished by writing to a temp file and then renaming it.
 426 
 427   Args:
 428     filename: string, pathname for a file
 429     contents: string, contents that need to be written to the file
 430     overwrite: boolean, if false it's an error for `filename` to be occupied by
 431         an existing file.
 432   """
 433   temp_pathname = filename + ".tmp" + uuid.uuid4().hex
 434   write_string_to_file(temp_pathname, contents)
 435   try:
 436     rename(temp_pathname, filename, overwrite)
 437   except errors.OpError:
 438     delete_file(temp_pathname)
 439     raise

 470 @tf_export("gfile.ListDirectory")
 471 def list_directory(dirname):
 472   """Returns a list of entries contained within a directory.
 473 
 474   The list is in arbitrary order. It does not contain the special entries "."
 475   and "..".
 476 
 477   Args:
 478     dirname: string, path to a directory
 479 
 480   Returns:
 481     [filename1, filename2, ... filenameN] as strings
 482 
 483   Raises:
 484     errors.NotFoundError if directory doesn't exist
 485   """
 486   if not is_directory(dirname):
 487     raise errors.NotFoundError(None, None, "Could not find directory")
 488   with errors.raise_exception_on_not_ok_status() as status:
 489     # Convert each element to string, since the return values of the
 490     # vector of string should be interpreted as strings, not bytes.
 491     return [
 492         compat.as_str_any(filename)
 493         for filename in pywrap_tensorflow.GetChildren(
 494             compat.as_bytes(dirname), status)
 495     ]

 498 @tf_export("gfile.Walk")
 499 def walk(top, in_order=True):
 500   """Recursive directory tree generator for directories.
 501 
 502   Args:
 503     top: string, a Directory name
 504     in_order: bool, Traverse in order if True, post order if False.
 505 
 506   Errors that happen while listing directories are ignored.
 507 
 508   Yields:
 509     Each yield is a 3-tuple:  the pathname of a directory, followed by lists of
 510     all its subdirectories and leaf files.
 511     (dirname, [subdirname, subdirname, ...], [filename, filename, ...])
 512     as strings
 513   """
 514   top = compat.as_str_any(top)
 515   try:
 516     listing = list_directory(top)
 517   except errors.NotFoundError:
 518     return
 519 
 520   files = []
 521   subdirs = []
 522   for item in listing:
 523     full_path = os.path.join(top, item)
 524     if is_directory(full_path):
 525       subdirs.append(item)
 526     else:
 527       files.append(item)
 528 
 529   here = (top, subdirs, files)
 530 
 531   if in_order:
 532     yield here
 533 
 534   for subdir in subdirs:
 535     for subitem in walk(os.path.join(top, subdir), in_order):
 536       yield subitem
 537 
 538   if not in_order:
 539     yield here

 561 def filecmp(filename_a, filename_b):
 562   """Compare two files, returning True if they are the same, False otherwise.
 563 
 564   We check size first and return False quickly if the files are different sizes.
 565   If they are the same size, we continue to generating a crc for the whole file.
 566 
 567   You might wonder: why not use Python's filecmp.cmp() instead? The answer is
 568   that the builtin library is not robust to the many different filesystems
 569   TensorFlow runs on, and so we here perform a similar comparison with
 570   the more robust FileIO.
 571 
 572   Args:
 573     filename_a: string path to the first file.
 574     filename_b: string path to the second file.
 575 
 576   Returns:
 577     True if the files are the same, False otherwise.
 578   """
 579   size_a = FileIO(filename_a, "rb").size()
 580   size_b = FileIO(filename_b, "rb").size()
 581   if size_a != size_b:
 582     return False
 583 
 584   # Size is the same. Do a full check.
 585   crc_a = file_crc32(filename_a)
 586   crc_b = file_crc32(filename_b)
 587   return crc_a == crc_b

 590 def file_crc32(filename, block_size=_DEFAULT_BLOCK_SIZE):
 591   """Get the crc32 of the passed file.
 592 
 593   The crc32 of a file can be used for error checking; two files with the same
 594   crc32 are considered equivalent. Note that the entire file must be read
 595   to produce the crc32.
 596 
 597   Args:
 598     filename: string, path to a file
 599     block_size: Integer, process the files by reading blocks of `block_size`
 600       bytes. Use -1 to read the file as once.
 601 
 602   Returns:
 603     hexadecimal as string, the crc32 of the passed file.
 604   """
 605   crc = 0
 606   with FileIO(filename, mode="rb") as f:
 607     chunk = f.read(n=block_size)
 608     while chunk:
 609       crc = binascii.crc32(chunk, crc)
 610       chunk = f.read(n=block_size)
 611   return hex(crc & 0xFFFFFFFF)
