  42 # On UNIX-based platforms, pywrap_tensorflow is a SWIG-generated
  43 # python library that dynamically loads _pywrap_tensorflow.so.
  44 _can_set_rtld_local = (hasattr(sys, 'getdlopenflags')
  45                        and hasattr(sys, 'setdlopenflags'))
  46 if _can_set_rtld_local:
  47   _default_dlopen_flags = sys.getdlopenflags()

  52   elif _can_set_rtld_local:
  53     # Ensure RTLD_LOCAL behavior for platforms where it isn't the default
  54     # (macOS). On Linux RTLD_LOCAL is 0, so this does nothing (and would not
  55     # override an RTLD_GLOBAL in _default_dlopen_flags).
  56     sys.setdlopenflags(_default_dlopen_flags | ctypes.RTLD_LOCAL)
