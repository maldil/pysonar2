  15 """A python interface for Grappler clusters."""
  16 
  17 from __future__ import absolute_import
  18 from __future__ import division
  19 from __future__ import print_function

  21 import contextlib
  22 
  23 from tensorflow.core.framework import step_stats_pb2
  24 from tensorflow.core.grappler.costs import op_performance_data_pb2
  25 from tensorflow.core.protobuf import device_properties_pb2
  26 from tensorflow.python import pywrap_tensorflow as tf_cluster
  27 from tensorflow.python.framework import errors
  28 
  29 
  30 class Cluster(object):
  31   """Grappler Clusters."""
  32 
  33   def __init__(self,
  34                allow_soft_placement=True,
  35                disable_detailed_stats=True,
  36                disable_timeline=True,
  37                devices=None):
  38     """Creates a Cluster.
  39 
  40     Args:
  41       allow_soft_placement: If True, TF will automatically fix illegal
  42         placements instead of erroring out if the placement isn't legal.
  43       disable_detailed_stats: If True, detailed statistics will not be
  44         available.
  45       disable_timeline: If True, the timeline information will not be reported.
  46       devices: A list of devices of type device_properties_pb2.NamedDevice.
  47         If None, a device list will be created based on the spec of
  48         the local machine.
  49     """
  50     self._tf_cluster = None
  51     self._generate_timeline = not disable_timeline
  52     with errors.raise_exception_on_not_ok_status() as status:
  53       if devices is None:
  54         self._tf_cluster = tf_cluster.TF_NewCluster(
  55             allow_soft_placement, disable_detailed_stats, status)
  56       else:
  57         devices_serialized = [device.SerializeToString() for device in devices]
  58         self._tf_cluster = tf_cluster.TF_NewVirtualCluster(
  59             devices_serialized, status)
  60 
  61   def Shutdown(self):
  62     if self._tf_cluster is not None:
  63       tf_cluster.TF_ShutdownCluster(self._tf_cluster)
  64       self._tf_cluster = None
  65 
  66   def __del__(self):
  67     self.Shutdown()
  68 
  69   @property
  70   def tf_cluster(self):
  71     return self._tf_cluster
  72 
  73   def ListDevices(self):
  74     """Returns the list of available hardware devices."""
  75     devices = []
  76     if self._tf_cluster is not None:
  77       ret_from_swig = tf_cluster.TF_ListDevices(self._tf_cluster)
  78       devices = []
  79       for raw_dev in ret_from_swig:
  80         devices.append(device_properties_pb2.NamedDevice.FromString(raw_dev))
  81     return devices
  82 
  83   def ListAvailableOps(self):
  84     """Returns a list of all the available operations (sorted alphatically)."""
  85     return tf_cluster.TF_ListAvailableOps()
  86 
  87   def GetSupportedDevices(self, item):
  88     return tf_cluster.TF_GetSupportedDevices(self._tf_cluster, item.tf_item)
  89 
  90   def EstimatePerformance(self, device):
  91     """Estimate the performance of the specified device."""
  92     serialized = device.SerializeToString()
  93     return tf_cluster.TF_EstimatePerformance(serialized)
  94 
  95   def MeasureCosts(self, item):
  96     """Returns the cost of running the specified item.
  97 
  98     Args:
  99       item: The item for which to measure the costs.
 100     Returns: The triplet op_perfs, runtime, step_stats.
 101     """
 102     with errors.raise_exception_on_not_ok_status() as status:
 103       ret_from_swig = tf_cluster.TF_MeasureCosts(
 104           item.tf_item, self._tf_cluster, self._generate_timeline, status)
 105 
 106     if ret_from_swig is None:
 107       return None
 108 
 109     op_perf_bytes_list, run_time, step_stats_bytes = ret_from_swig
 110     op_perfs = []
 111     for op_perf_bytes in op_perf_bytes_list:
 112       op_perfs.append(
 113           op_performance_data_pb2.OpPerformance.FromString(op_perf_bytes))
 114     return (op_perfs, run_time,
 115             step_stats_pb2.StepStats.FromString(step_stats_bytes))
 116 
 117   def DeterminePeakMemoryUsage(self, item):
 118     """Returns a snapshot of the peak memory usage.
 119 
 120     Args:
 121       item: The item for which to measure the costs.
 122     Returns: A hashtable indexed by device name.
 123     """
 124     with errors.raise_exception_on_not_ok_status() as status:
 125       ret_from_swig = tf_cluster.TF_DeterminePeakMemoryUsage(
 126           item.tf_item, self._tf_cluster, status)
 127 
 128     return ret_from_swig
 129

 131 @contextlib.contextmanager
 132 def Provision(allow_soft_placement=True,
 133               disable_detailed_stats=True,
 134               disable_timeline=True,
 135               devices=None):
 136   cluster = Cluster(allow_soft_placement, disable_detailed_stats,
 137                     disable_timeline, devices)
 138   yield cluster
 139   cluster.Shutdown()
