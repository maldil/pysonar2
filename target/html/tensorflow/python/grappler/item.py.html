  15 """A python interface for Grappler items."""
  16 
  17 from __future__ import absolute_import
  18 from __future__ import division
  19 from __future__ import print_function
  20 
  21 from tensorflow.core.grappler.costs import op_performance_data_pb2
  22 from tensorflow.core.protobuf import meta_graph_pb2
  23 from tensorflow.python import pywrap_tensorflow as tf_item
  24 from tensorflow.python.framework import errors
  25 
  26 
  27 class Item(object):
  28   """GrapplerItem."""
  29 
  30   def __init__(self,
  31                metagraph,
  32                ignore_colocation=True,
  33                ignore_user_placement=False):
  34     """Creates an Item.
  35 
  36     Args:
  37       metagraph: a TensorFlow metagraph.
  38       ignore_colocation: if set, the tool will ignore all the colocation
  39         constraints generated by TensorFlow.
  40       ignore_user_placement: if set, all the placement annotations annotated in
  41         the metagraph will be ignored.
  42     Raises:
  43       ValueError: the metagraph is incomplete or invalid.
  44     """
  45     self._metagraph = metagraph
  46     self._item_graph = meta_graph_pb2.MetaGraphDef()
  47     self._item_graph.CopyFrom(metagraph)
  48     self._ignore_colocation = ignore_colocation
  49     self._ignore_user_placement = ignore_user_placement
  50     self._tf_item = None
  51     self._BuildTFItem()
  52 
  53   def IdentifyImportantOps(self, sort_topologically=False):
  54     return tf_item.TF_IdentifyImportantOps(self.tf_item, sort_topologically)
  55 
  56   def GetOpProperties(self):
  57     ret_from_swig = tf_item.TF_GetOpProperties(self.tf_item)
  58     properties = {}
  59     for key, values in ret_from_swig.items():
  60       prop = []
  61       for value in values:
  62         prop.append(
  63             op_performance_data_pb2.OpInfo.TensorProperties.FromString(value))
  64       properties[key] = prop
  65     return properties
  66 
  67   def GetColocationGroups(self):
  68     """Return a list of hard colocation constraints.
  69 
  70     All the nodes in a colocation tuple must be placed on the same device for
  71     the model to work.
  72 
  73     Returns:
  74       A list of colocation tuples.
  75     """
  76     return tf_item.TF_GetColocationGroups(self.tf_item)
  77 
  78   @property
  79   def metagraph(self):
  80     return self._metagraph
  81 
  82   @property
  83   def tf_item(self):
  84     if self._item_graph != self._metagraph:
  85       self._BuildTFItem()
  86       self._item_graph.CopyFrom(self._metagraph)
  87     return self._tf_item
  88 
  89   def _BuildTFItem(self):
  90     with errors.raise_exception_on_not_ok_status() as status:
  91       self._tf_item = tf_item.TF_NewItem(self._metagraph.SerializeToString(),
  92                                          self._ignore_colocation,
  93                                          self._ignore_user_placement, status)
