  65 @tf_export('summary.scalar')
  66 def scalar(name, tensor, collections=None, family=None):
  67   """Outputs a `Summary` protocol buffer containing a single scalar value.
  68 
  69   The generated Summary has a Tensor.proto containing the input Tensor.
  70 
  71   Args:
  72     name: A name for the generated node. Will also serve as the series name in
  73       TensorBoard.
  74     tensor: A real numeric Tensor containing a single value.
  75     collections: Optional list of graph collections keys. The new summary op is
  76       added to these collections. Defaults to `[GraphKeys.SUMMARIES]`.
  77     family: Optional; if provided, used as the prefix of the summary tag name,
  78       which controls the tab name used for display on Tensorboard.
  79 
  80   Returns:
  81     A scalar `Tensor` of type `string`. Which contains a `Summary` protobuf.
  82 
  83   Raises:
  84     ValueError: If tensor has the wrong shape or type.
  85   """
  86   if _summary_op_util.skip_summary():
  87     return _constant_op.constant('')
  88   with _summary_op_util.summary_scope(
  89       name, family, values=[tensor]) as (tag, scope):
  90     val = _gen_logging_ops.scalar_summary(tags=tag, values=tensor, name=scope)
  91     _summary_op_util.collect(val, collections, [_ops.GraphKeys.SUMMARIES])
  92   return val

  95 @tf_export('summary.image')
  96 def image(name, tensor, max_outputs=3, collections=None, family=None):
  97   """Outputs a `Summary` protocol buffer with images.
  98 
  99   The summary has up to `max_outputs` summary values containing images. The
 100   images are built from `tensor` which must be 4-D with shape `[batch_size,
 101   height, width, channels]` and where `channels` can be:
 102 
 103   *  1: `tensor` is interpreted as Grayscale.
 104   *  3: `tensor` is interpreted as RGB.
 105   *  4: `tensor` is interpreted as RGBA.
 106 
 107   The images have the same number of channels as the input tensor. For float
 108   input, the values are normalized one image at a time to fit in the range
 109   `[0, 255]`.  `uint8` values are unchanged.  The op uses two different
 110   normalization algorithms:
 111 
 112   *  If the input values are all positive, they are rescaled so the largest one
 113      is 255.
 114 
 115   *  If any input value is negative, the values are shifted so input value 0.0
 116      is at 127.  They are then rescaled so that either the smallest value is 0,
 117      or the largest one is 255.
 118 
 119   The `tag` in the outputted Summary.Value protobufs is generated based on the
 120   name, with a suffix depending on the max_outputs setting:
 121 
 122   *  If `max_outputs` is 1, the summary value tag is '*name*/image'.
 123   *  If `max_outputs` is greater than 1, the summary value tags are
 124      generated sequentially as '*name*/image/0', '*name*/image/1', etc.
 125 
 126   Args:
 127     name: A name for the generated node. Will also serve as a series name in
 128       TensorBoard.
 129     tensor: A 4-D `uint8` or `float32` `Tensor` of shape `[batch_size, height,
 130       width, channels]` where `channels` is 1, 3, or 4.
 131     max_outputs: Max number of batch elements to generate images for.
 132     collections: Optional list of ops.GraphKeys.  The collections to add the
 133       summary to.  Defaults to [_ops.GraphKeys.SUMMARIES]
 134     family: Optional; if provided, used as the prefix of the summary tag name,
 135       which controls the tab name used for display on Tensorboard.
 136 
 137   Returns:
 138     A scalar `Tensor` of type `string`. The serialized `Summary` protocol
 139     buffer.
 140   """
 141   if _summary_op_util.skip_summary():
 142     return _constant_op.constant('')
 143   with _summary_op_util.summary_scope(
 144       name, family, values=[tensor]) as (tag, scope):
 145     val = _gen_logging_ops.image_summary(
 146         tag=tag, tensor=tensor, max_images=max_outputs, name=scope)
 147     _summary_op_util.collect(val, collections, [_ops.GraphKeys.SUMMARIES])
 148   return val

 151 @tf_export('summary.histogram')
 152 def histogram(name, values, collections=None, family=None):
 153   # pylint: disable=line-too-long
 154   """Outputs a `Summary` protocol buffer with a histogram.
 155 
 156   Adding a histogram summary makes it possible to visualize your data's
 157   distribution in TensorBoard. You can see a detailed explanation of the
 158   TensorBoard histogram dashboard
 159   [here](https://www.tensorflow.org/get_started/tensorboard_histograms).
 160 
 161   The generated
 162   [`Summary`](https://www.tensorflow.org/code/tensorflow/core/framework/summary.proto)
 163   has one summary value containing a histogram for `values`.
 164 
 165   This op reports an `InvalidArgument` error if any value is not finite.
 166 
 167   Args:
 168     name: A name for the generated node. Will also serve as a series name in
 169       TensorBoard.
 170     values: A real numeric `Tensor`. Any shape. Values to use to
 171       build the histogram.
 172     collections: Optional list of graph collections keys. The new summary op is
 173       added to these collections. Defaults to `[GraphKeys.SUMMARIES]`.
 174     family: Optional; if provided, used as the prefix of the summary tag name,
 175       which controls the tab name used for display on Tensorboard.
 176 
 177   Returns:
 178     A scalar `Tensor` of type `string`. The serialized `Summary` protocol
 179     buffer.
 180   """
 181   if _summary_op_util.skip_summary():
 182     return _constant_op.constant('')
 183   with _summary_op_util.summary_scope(
 184       name, family, values=[values],
 185       default_name='HistogramSummary') as (tag, scope):
 186     val = _gen_logging_ops.histogram_summary(
 187         tag=tag, values=values, name=scope)
 188     _summary_op_util.collect(val, collections, [_ops.GraphKeys.SUMMARIES])
 189   return val

 192 @tf_export('summary.audio')
 193 def audio(name, tensor, sample_rate, max_outputs=3, collections=None,
 194           family=None):
 195   # pylint: disable=line-too-long
 196   """Outputs a `Summary` protocol buffer with audio.
 197 
 198   The summary has up to `max_outputs` summary values containing audio. The
 199   audio is built from `tensor` which must be 3-D with shape `[batch_size,
 200   frames, channels]` or 2-D with shape `[batch_size, frames]`. The values are
 201   assumed to be in the range of `[-1.0, 1.0]` with a sample rate of
 202   `sample_rate`.
 203 
 204   The `tag` in the outputted Summary.Value protobufs is generated based on the
 205   name, with a suffix depending on the max_outputs setting:
 206 
 207   *  If `max_outputs` is 1, the summary value tag is '*name*/audio'.
 208   *  If `max_outputs` is greater than 1, the summary value tags are
 209      generated sequentially as '*name*/audio/0', '*name*/audio/1', etc
 210 
 211   Args:
 212     name: A name for the generated node. Will also serve as a series name in
 213       TensorBoard.
 214     tensor: A 3-D `float32` `Tensor` of shape `[batch_size, frames, channels]`
 215       or a 2-D `float32` `Tensor` of shape `[batch_size, frames]`.
 216     sample_rate: A Scalar `float32` `Tensor` indicating the sample rate of the
 217       signal in hertz.
 218     max_outputs: Max number of batch elements to generate audio for.
 219     collections: Optional list of ops.GraphKeys.  The collections to add the
 220       summary to.  Defaults to [_ops.GraphKeys.SUMMARIES]
 221     family: Optional; if provided, used as the prefix of the summary tag name,
 222       which controls the tab name used for display on Tensorboard.
 223 
 224   Returns:
 225     A scalar `Tensor` of type `string`. The serialized `Summary` protocol
 226     buffer.
 227   """
 228   if _summary_op_util.skip_summary():
 229     return _constant_op.constant('')
 230   with _summary_op_util.summary_scope(
 231       name, family=family, values=[tensor]) as (tag, scope):
 232     sample_rate = _ops.convert_to_tensor(
 233         sample_rate, dtype=_dtypes.float32, name='sample_rate')
 234     val = _gen_logging_ops.audio_summary_v2(
 235         tag=tag, tensor=tensor, max_outputs=max_outputs,
 236         sample_rate=sample_rate, name=scope)
 237     _summary_op_util.collect(val, collections, [_ops.GraphKeys.SUMMARIES])
 238   return val

 241 @tf_export('summary.merge')
 242 def merge(inputs, collections=None, name=None):
 243   # pylint: disable=line-too-long
 244   """Merges summaries.
 245 
 246   This op creates a
 247   [`Summary`](https://www.tensorflow.org/code/tensorflow/core/framework/summary.proto)
 248   protocol buffer that contains the union of all the values in the input
 249   summaries.
 250 
 251   When the Op is run, it reports an `InvalidArgument` error if multiple values
 252   in the summaries to merge use the same tag.
 253 
 254   Args:
 255     inputs: A list of `string` `Tensor` objects containing serialized `Summary`
 256       protocol buffers.
 257     collections: Optional list of graph collections keys. The new summary op is
 258       added to these collections. Defaults to `[]`.
 259     name: A name for the operation (optional).
 260 
 261   Returns:
 262     A scalar `Tensor` of type `string`. The serialized `Summary` protocol
 263     buffer resulting from the merging.
 264 
 265   Raises:
 266     RuntimeError: If called with eager mode enabled.
 267 
 268   @compatibility(eager)
 269   Not compatible with eager execution. To write TensorBoard
 270   summaries under eager execution, use `tf.contrib.summary` instead.
 271   @end_compatibility
 272   """
 273   # pylint: enable=line-too-long
 274   if _context.executing_eagerly():
 275     raise RuntimeError(
 276         'Merging tf.summary.* ops is not compatible with eager execution. '
 277         'Use tf.contrib.summary instead.')
 278   if _summary_op_util.skip_summary():
 279     return _constant_op.constant('')
 280   name = _summary_op_util.clean_tag(name)
 281   with _ops.name_scope(name, 'Merge', inputs):
 282     val = _gen_logging_ops.merge_summary(inputs=inputs, name=name)
 283     _summary_op_util.collect(val, collections, [])
 284   return val

 287 @tf_export('summary.merge_all')
 288 def merge_all(key=_ops.GraphKeys.SUMMARIES, scope=None, name=None):
 289   """Merges all summaries collected in the default graph.
 290 
 291   Args:
 292     key: `GraphKey` used to collect the summaries.  Defaults to
 293       `GraphKeys.SUMMARIES`.
 294     scope: Optional scope used to filter the summary ops, using `re.match`
 295 
 296   Returns:
 297     If no summaries were collected, returns None.  Otherwise returns a scalar
 298     `Tensor` of type `string` containing the serialized `Summary` protocol
 299     buffer resulting from the merging.
 300 
 301   Raises:
 302     RuntimeError: If called with eager execution enabled.
 303 
 304   @compatibility(eager)
 305   Not compatible with eager execution. To write TensorBoard
 306   summaries under eager execution, use `tf.contrib.summary` instead.
 307   @end_compatibility
 308   """
 309   if _context.executing_eagerly():
 310     raise RuntimeError(
 311         'Merging tf.summary.* ops is not compatible with eager execution. '
 312         'Use tf.contrib.summary instead.')
 313   summary_ops = _ops.get_collection(key, scope=scope)
 314   if not summary_ops:
 315     return None
 316   else:
 317     return merge(summary_ops, name=name)

 320 @tf_export('summary.get_summary_description')
 321 def get_summary_description(node_def):
 322   """Given a TensorSummary node_def, retrieve its SummaryDescription.
 323 
 324   When a Summary op is instantiated, a SummaryDescription of associated
 325   metadata is stored in its NodeDef. This method retrieves the description.
 326 
 327   Args:
 328     node_def: the node_def_pb2.NodeDef of a TensorSummary op
 329 
 330   Returns:
 331     a summary_pb2.SummaryDescription
 332 
 333   Raises:
 334     ValueError: if the node is not a summary op.
 335 
 336   @compatibility(eager)
 337   Not compatible with eager execution. To write TensorBoard
 338   summaries under eager execution, use `tf.contrib.summary` instead.
 339   @end_compatibility
 340   """
 341 
 342   if node_def.op != 'TensorSummary':
 343     raise ValueError("Can't get_summary_description on %s" % node_def.op)
 344   description_str = _compat.as_str_any(node_def.attr['description'].s)
 345   summary_description = SummaryDescription()
 346   _json_format.Parse(description_str, summary_description)
 347   return summary_description
