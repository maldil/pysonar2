  41 class SummaryToEventTransformer(object):
  42   """Abstractly implements the SummaryWriter API.
  43 
  44   This API basically implements a number of endpoints (add_summary,
  45   add_session_log, etc). The endpoints all generate an event protobuf, which is
  46   passed to the contained event_writer.
  47   """
  48 
  49   def __init__(self, event_writer, graph=None, graph_def=None):
  50     """Creates a `SummaryWriter` and an event file.
  51 
  52     On construction the summary writer creates a new event file in `logdir`.
  53     This event file will contain `Event` protocol buffers constructed when you
  54     call one of the following functions: `add_summary()`, `add_session_log()`,
  55     `add_event()`, or `add_graph()`.
  56 
  57     If you pass a `Graph` to the constructor it is added to
  58     the event file. (This is equivalent to calling `add_graph()` later).
  59 
  60     TensorBoard will pick the graph from the file and display it graphically so
  61     you can interactively explore the graph you built. You will usually pass
  62     the graph from the session in which you launched it:
  63 
  64     ```python
  65     ...create a graph...
  66     # Launch the graph in a session.
  67     sess = tf.Session()
  68     # Create a summary writer, add the 'graph' to the event file.
  69     writer = tf.summary.FileWriter(<some-directory>, sess.graph)
  70     ```
  71 
  72 
  73     Args:
  74       event_writer: An EventWriter. Implements add_event and get_logdir.
  75       graph: A `Graph` object, such as `sess.graph`.
  76       graph_def: DEPRECATED: Use the `graph` argument instead.
  77     """
  78     self.event_writer = event_writer
  79     # For storing used tags for session.run() outputs.
  80     self._session_run_tags = {}
  81     if graph is not None or graph_def is not None:
  82       # Calling it with both graph and graph_def for backward compatibility.
  83       self.add_graph(graph=graph, graph_def=graph_def)
  84       # Also export the meta_graph_def in this case.
  85       # graph may itself be a graph_def due to positional arguments
  86       maybe_graph_as_def = (graph.as_graph_def(add_shapes=True)
  87                             if isinstance(graph, ops.Graph) else graph)
  88       self.add_meta_graph(
  89           meta_graph.create_meta_graph_def(graph_def=graph_def or
  90                                            maybe_graph_as_def))
  91 
  92     # This set contains tags of Summary Values that have been encountered
  93     # already. The motivation here is that the SummaryWriter only keeps the
  94     # metadata property (which is a SummaryMetadata proto) of the first Summary
  95     # Value encountered for each tag. The SummaryWriter strips away the
  96     # SummaryMetadata for all subsequent Summary Values with tags seen
  97     # previously. This saves space.
  98     self._seen_summary_tags = set()
  99 
 100   def add_summary(self, summary, global_step=None):
 101     """Adds a `Summary` protocol buffer to the event file.
 102 
 103     This method wraps the provided summary in an `Event` protocol buffer
 104     and adds it to the event file.
 105 
 106     You can pass the result of evaluating any summary op, using
 107     `tf.Session.run` or
 108     `tf.Tensor.eval`, to this
 109     function. Alternatively, you can pass a `tf.Summary` protocol
 110     buffer that you populate with your own data. The latter is
 111     commonly done to report evaluation results in event files.
 112 
 113     Args:
 114       summary: A `Summary` protocol buffer, optionally serialized as a string.
 115       global_step: Number. Optional global step value to record with the
 116         summary.
 117     """
 118     if isinstance(summary, bytes):
 119       summ = summary_pb2.Summary()
 120       summ.ParseFromString(summary)
 121       summary = summ
 122 
 123     # We strip metadata from values with tags that we have seen before in order
 124     # to save space - we just store the metadata on the first value with a
 125     # specific tag.
 126     for value in summary.value:
 127       if not value.metadata:
 128         continue
 129 
 130       if value.tag in self._seen_summary_tags:
 131         # This tag has been encountered before. Strip the metadata.
 132         value.ClearField("metadata")
 133         continue
 134 
 135       # We encounter a value with a tag we have not encountered previously. And
 136       # it has metadata. Remember to strip metadata from future values with this
 137       # tag string.
 138       self._seen_summary_tags.add(value.tag)
 139 
 140     event = event_pb2.Event(summary=summary)
 141     self._add_event(event, global_step)
 142 
 143   def add_session_log(self, session_log, global_step=None):
 144     """Adds a `SessionLog` protocol buffer to the event file.
 145 
 146     This method wraps the provided session in an `Event` protocol buffer
 147     and adds it to the event file.
 148 
 149     Args:
 150       session_log: A `SessionLog` protocol buffer.
 151       global_step: Number. Optional global step value to record with the
 152         summary.
 153     """
 154     event = event_pb2.Event(session_log=session_log)
 155     self._add_event(event, global_step)
 156 
 157   def _add_graph_def(self, graph_def, global_step=None):
 158     graph_bytes = graph_def.SerializeToString()
 159     event = event_pb2.Event(graph_def=graph_bytes)
 160     self._add_event(event, global_step)
 161 
 162   def add_graph(self, graph, global_step=None, graph_def=None):
 163     """Adds a `Graph` to the event file.
 164 
 165     The graph described by the protocol buffer will be displayed by
 166     TensorBoard. Most users pass a graph in the constructor instead.
 167 
 168     Args:
 169       graph: A `Graph` object, such as `sess.graph`.
 170       global_step: Number. Optional global step counter to record with the
 171         graph.
 172       graph_def: DEPRECATED. Use the `graph` parameter instead.
 173 
 174     Raises:
 175       ValueError: If both graph and graph_def are passed to the method.
 176     """
 177 
 178     if graph is not None and graph_def is not None:
 179       raise ValueError("Please pass only graph, or graph_def (deprecated), "
 180                        "but not both.")
 181 
 182     if isinstance(graph, ops.Graph) or isinstance(graph_def, ops.Graph):
 183       # The user passed a `Graph`.
 184 
 185       # Check if the user passed it via the graph or the graph_def argument and
 186       # correct for that.
 187       if not isinstance(graph, ops.Graph):
 188         logging.warning("When passing a `Graph` object, please use the `graph`"
 189                         " named argument instead of `graph_def`.")
 190         graph = graph_def
 191 
 192       # Serialize the graph with additional info.
 193       true_graph_def = graph.as_graph_def(add_shapes=True)
 194       self._write_plugin_assets(graph)
 195     elif (isinstance(graph, graph_pb2.GraphDef) or
 196           isinstance(graph_def, graph_pb2.GraphDef)):
 197       # The user passed a `GraphDef`.
 198       logging.warning("Passing a `GraphDef` to the SummaryWriter is deprecated."
 199                       " Pass a `Graph` object instead, such as `sess.graph`.")
 200 
 201       # Check if the user passed it via the graph or the graph_def argument and
 202       # correct for that.
 203       if isinstance(graph, graph_pb2.GraphDef):
 204         true_graph_def = graph
 205       else:
 206         true_graph_def = graph_def
 207 
 208     else:
 209       # The user passed neither `Graph`, nor `GraphDef`.
 210       raise TypeError("The passed graph must be an instance of `Graph` "
 211                       "or the deprecated `GraphDef`")
 212     # Finally, add the graph_def to the summary writer.
 213     self._add_graph_def(true_graph_def, global_step)
 214 
 215   def _write_plugin_assets(self, graph):
 216     plugin_assets = plugin_asset.get_all_plugin_assets(graph)
 217     logdir = self.event_writer.get_logdir()
 218     for asset_container in plugin_assets:
 219       plugin_name = asset_container.plugin_name
 220       plugin_dir = os.path.join(logdir, _PLUGINS_DIR, plugin_name)
 221       gfile.MakeDirs(plugin_dir)
 222       assets = asset_container.assets()
 223       for (asset_name, content) in assets.items():
 224         asset_path = os.path.join(plugin_dir, asset_name)
 225         with gfile.Open(asset_path, "w") as f:
 226           f.write(content)
 227 
 228   def add_meta_graph(self, meta_graph_def, global_step=None):
 229     """Adds a `MetaGraphDef` to the event file.
 230 
 231     The `MetaGraphDef` allows running the given graph via
 232     `saver.import_meta_graph()`.
 233 
 234     Args:
 235       meta_graph_def: A `MetaGraphDef` object, often as returned by
 236         `saver.export_meta_graph()`.
 237       global_step: Number. Optional global step counter to record with the
 238         graph.
 239 
 240     Raises:
 241       TypeError: If both `meta_graph_def` is not an instance of `MetaGraphDef`.
 242     """
 243     if not isinstance(meta_graph_def, meta_graph_pb2.MetaGraphDef):
 244       raise TypeError("meta_graph_def must be type MetaGraphDef, saw type: %s" %
 245                       type(meta_graph_def))
 246     meta_graph_bytes = meta_graph_def.SerializeToString()
 247     event = event_pb2.Event(meta_graph_def=meta_graph_bytes)
 248     self._add_event(event, global_step)
 249 
 250   def add_run_metadata(self, run_metadata, tag, global_step=None):
 251     """Adds a metadata information for a single session.run() call.
 252 
 253     Args:
 254       run_metadata: A `RunMetadata` protobuf object.
 255       tag: The tag name for this metadata.
 256       global_step: Number. Optional global step counter to record with the
 257         StepStats.
 258 
 259     Raises:
 260       ValueError: If the provided tag was already used for this type of event.
 261     """
 262     if tag in self._session_run_tags:
 263       raise ValueError("The provided tag was already used for this event type")
 264     self._session_run_tags[tag] = True
 265 
 266     tagged_metadata = event_pb2.TaggedRunMetadata()
 267     tagged_metadata.tag = tag
 268     # Store the `RunMetadata` object as bytes in order to have postponed
 269     # (lazy) deserialization when used later.
 270     tagged_metadata.run_metadata = run_metadata.SerializeToString()
 271     event = event_pb2.Event(tagged_run_metadata=tagged_metadata)
 272     self._add_event(event, global_step)
 273 
 274   def _add_event(self, event, step):
 275     event.wall_time = time.time()
 276     if step is not None:
 277       event.step = int(step)
 278     self.event_writer.add_event(event)

 281 @tf_export("summary.FileWriter")
 282 class FileWriter(SummaryToEventTransformer):
 283   """Writes `Summary` protocol buffers to event files.
 284 
 285   The `FileWriter` class provides a mechanism to create an event file in a
 286   given directory and add summaries and events to it. The class updates the
 287   file contents asynchronously. This allows a training program to call methods
 288   to add data to the file directly from the training loop, without slowing down
 289   training.
 290 
 291   When constructed with a `tf.Session` parameter, a `FileWriter` instead forms
 292   a compatibility layer over new graph-based summaries (`tf.contrib.summary`)
 293   to facilitate the use of new summary writing with pre-existing code that
 294   expects a `FileWriter` instance.
 295   """
 296 
 297   def __init__(self,
 298                logdir,
 299                graph=None,
 300                max_queue=10,
 301                flush_secs=120,
 302                graph_def=None,
 303                filename_suffix=None,
 304                session=None):
 305     """Creates a `FileWriter`, optionally shared within the given session.
 306 
 307     Typically, constructing a file writer creates a new event file in `logdir`.
 308     This event file will contain `Event` protocol buffers constructed when you
 309     call one of the following functions: `add_summary()`, `add_session_log()`,
 310     `add_event()`, or `add_graph()`.
 311 
 312     If you pass a `Graph` to the constructor it is added to
 313     the event file. (This is equivalent to calling `add_graph()` later).
 314 
 315     TensorBoard will pick the graph from the file and display it graphically so
 316     you can interactively explore the graph you built. You will usually pass
 317     the graph from the session in which you launched it:
 318 
 319     ```python
 320     ...create a graph...
 321     # Launch the graph in a session.
 322     sess = tf.Session()
 323     # Create a summary writer, add the 'graph' to the event file.
 324     writer = tf.summary.FileWriter(<some-directory>, sess.graph)
 325     ```
 326 
 327     The `session` argument to the constructor makes the returned `FileWriter` a
 328     compatibility layer over new graph-based summaries (`tf.contrib.summary`).
 329     Crucially, this means the underlying writer resource and events file will
 330     be shared with any other `FileWriter` using the same `session` and `logdir`,
 331     and with any `tf.contrib.summary.SummaryWriter` in this session using the
 332     the same shared resource name (which by default scoped to the logdir). If
 333     no such resource exists, one will be created using the remaining arguments
 334     to this constructor, but if one already exists those arguments are ignored.
 335     In either case, ops will be added to `session.graph` to control the
 336     underlying file writer resource. See `tf.contrib.summary` for more details.
 337 
 338     Args:
 339       logdir: A string. Directory where event file will be written.
 340       graph: A `Graph` object, such as `sess.graph`.
 341       max_queue: Integer. Size of the queue for pending events and summaries.
 342       flush_secs: Number. How often, in seconds, to flush the
 343         pending events and summaries to disk.
 344       graph_def: DEPRECATED: Use the `graph` argument instead.
 345       filename_suffix: A string. Every event file's name is suffixed with
 346         `suffix`.
 347       session: A `tf.Session` object. See details above.
 348 
 349     Raises:
 350       RuntimeError: If called with eager execution enabled.
 351 
 352     @compatibility(eager)
 353     `FileWriter` is not compatible with eager execution. To write TensorBoard
 354     summaries under eager execution, use `tf.contrib.summary` instead.
 355     @end_compatibility
 356     """
 357     if context.executing_eagerly():
 358       raise RuntimeError(
 359           "tf.summary.FileWriter is not compatible with eager execution. "
 360           "Use tf.contrib.summary instead.")
 361     if session is not None:
 362       event_writer = EventFileWriterV2(
 363           session, logdir, max_queue, flush_secs, filename_suffix)
 364     else:
 365       event_writer = EventFileWriter(logdir, max_queue, flush_secs,
 366                                      filename_suffix)
 367     super(FileWriter, self).__init__(event_writer, graph, graph_def)
 368 
 369   def __enter__(self):
 370     """Make usable with "with" statement."""
 371     return self
 372 
 373   def __exit__(self, unused_type, unused_value, unused_traceback):
 374     """Make usable with "with" statement."""
 375     self.close()
 376 
 377   def get_logdir(self):
 378     """Returns the directory where event file will be written."""
 379     return self.event_writer.get_logdir()
 380 
 381   def add_event(self, event):
 382     """Adds an event to the event file.
 383 
 384     Args:
 385       event: An `Event` protocol buffer.
 386     """
 387     self.event_writer.add_event(event)
 388 
 389   def flush(self):
 390     """Flushes the event file to disk.
 391 
 392     Call this method to make sure that all pending events have been written to
 393     disk.
 394     """
 395     self.event_writer.flush()
 396 
 397   def close(self):
 398     """Flushes the event file to disk and close the file.
 399 
 400     Call this method when you do not need the summary writer anymore.
 401     """
 402     self.event_writer.close()
 403 
 404   def reopen(self):
 405     """Reopens the EventFileWriter.
 406 
 407     Can be called after `close()` to add more events in the same directory.
 408     The events will go into a new events file.
 409 
 410     Does nothing if the EventFileWriter was not closed.
 411     """
 412     self.event_writer.reopen()
