  29 class EventFileWriterV2(object):
  30   """Writes `Event` protocol buffers to an event file via the graph.
  31 
  32   The `EventFileWriterV2` class is backed by the summary file writer in the v2
  33   summary API (currently in tf.contrib.summary), so it uses a shared summary
  34   writer resource and graph ops to write events.
  35 
  36   As with the original EventFileWriter, this class will asynchronously write
  37   Event protocol buffers to the backing file. The Event file is encoded using
  38   the tfrecord format, which is similar to RecordIO.
  39   """
  40 
  41   def __init__(self, session, logdir, max_queue=10, flush_secs=120,
  42                filename_suffix=''):
  43     """Creates an `EventFileWriterV2` and an event file to write to.
  44 
  45     On construction, this calls `tf.contrib.summary.create_file_writer` within
  46     the graph from `session.graph` to look up a shared summary writer resource
  47     for `logdir` if one exists, and create one if not. Creating the summary
  48     writer resource in turn creates a new event file in `logdir` to be filled
  49     with `Event` protocol buffers passed to `add_event`. Graph ops to control
  50     this writer resource are added to `session.graph` during this init call;
  51     stateful methods on this class will call `session.run()` on these ops.
  52 
  53     Note that because the underlying resource is shared, it is possible that
  54     other parts of the code using the same session may interact independently
  55     with the resource, e.g. by flushing or even closing it. It is the caller's
  56     responsibility to avoid any undesirable sharing in this regard.
  57 
  58     The remaining arguments to the constructor (`flush_secs`, `max_queue`, and
  59     `filename_suffix`) control the construction of the shared writer resource
  60     if one is created. If an existing resource is reused, these arguments have
  61     no effect.  See `tf.contrib.summary.create_file_writer` for details.
  62 
  63     Args:
  64       session: A `tf.Session`. Session that will hold shared writer resource.
  65         The writer ops will be added to session.graph during this init call.
  66       logdir: A string. Directory where event file will be written.
  67       max_queue: Integer. Size of the queue for pending events and summaries.
  68       flush_secs: Number. How often, in seconds, to flush the
  69         pending events and summaries to disk.
  70       filename_suffix: A string. Every event file's name is suffixed with
  71         `filename_suffix`.
  72     """
  73     self._session = session
  74     self._logdir = logdir
  75     self._closed = False
  76     if not gfile.IsDirectory(self._logdir):
  77       gfile.MakeDirs(self._logdir)
  78 
  79     with self._session.graph.as_default():
  80       with ops.name_scope('filewriter'):
  81         file_writer = summary_ops_v2.create_file_writer(
  82             logdir=self._logdir,
  83             max_queue=max_queue,
  84             flush_millis=flush_secs * 1000,
  85             filename_suffix=filename_suffix)
  86         with summary_ops_v2.always_record_summaries(), file_writer.as_default():
  87           self._event_placeholder = array_ops.placeholder_with_default(
  88               constant_op.constant('unused', dtypes.string),
  89               shape=[])
  90           self._add_event_op = summary_ops_v2.import_event(
  91               self._event_placeholder)
  92         self._init_op = file_writer.init()
  93         self._flush_op = file_writer.flush()
  94         self._close_op = file_writer.close()
  95       self._session.run(self._init_op)
  96 
  97   def get_logdir(self):
  98     """Returns the directory where event file will be written."""
  99     return self._logdir
 100 
 101   def reopen(self):
 102     """Reopens the EventFileWriter.
 103 
 104     Can be called after `close()` to add more events in the same directory.
 105     The events will go into a new events file.
 106 
 107     Does nothing if the EventFileWriter was not closed.
 108     """
 109     if self._closed:
 110       self._closed = False
 111       self._session.run(self._init_op)
 112 
 113   def add_event(self, event):
 114     """Adds an event to the event file.
 115 
 116     Args:
 117       event: An `Event` protocol buffer.
 118     """
 119     if not self._closed:
 120       event_pb = event.SerializeToString()
 121       self._session.run(
 122           self._add_event_op, feed_dict={self._event_placeholder: event_pb})
 123 
 124   def flush(self):
 125     """Flushes the event file to disk.
 126 
 127     Call this method to make sure that all pending events have been written to
 128     disk.
 129     """
 130     self._session.run(self._flush_op)
 131 
 132   def close(self):
 133     """Flushes the event file to disk and close the file.
 134 
 135     Call this method when you do not need the summary writer anymore.
 136     """
 137     if not self._closed:
 138       self.flush()
 139       self._session.run(self._close_op)
 140       self._closed = True
