  17 from __future__ import absolute_import
  18 from __future__ import division
  19 from __future__ import print_function

  21 from tensorflow.core.framework import device_attributes_pb2
  22 from tensorflow.python import pywrap_tensorflow
  23 
  24 
  25 def list_local_devices(session_config=None):
  26   """List the available devices available in the local process.
  27 
  28   Args:
  29     session_config: a session config proto or None to use the default config.
  30 
  31   Returns:
  32     A list of `DeviceAttribute` protocol buffers.
  33   """
  34   def _convert(pb_str):
  35     m = device_attributes_pb2.DeviceAttributes()
  36     m.ParseFromString(pb_str)
  37     return m
  38 
  39   return [
  40       _convert(s)
  41       for s in pywrap_tensorflow.list_devices(session_config=session_config)
  42   ]
