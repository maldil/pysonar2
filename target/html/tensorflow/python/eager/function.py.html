  76 def _create_substitute_placeholder(value, name=None, dtype=None):
  77   """Creates a placeholder for `value` and propagates shape info to it."""
  78   # Note: setting ops.control_dependencies(None) ensures we always put
  79   # capturing placeholders outside of any control flow context.
  80   with ops.control_dependencies(None):
  81     placeholder = graph_placeholder(
  82         dtype=dtype or value.dtype, shape=value.shape, name=name)
  83   _copy_handle_data(value, placeholder)
  84   return placeholder

  87 def _copy_handle_data(source_t, target_t):
  88   """Copies HandleData for variant and resource type tensors if available.
  89 
  90   The CppShapeInferenceResult::HandleData proto contains information about the
  91   shapes and types of the element tensors of resource/variant type tensors.
  92   We need to copy this across function boundaries, i.e., when capturing a
  93   placeholder or when returning a function tensor as output. If we don't do this
  94   the element tensors will have unknown shapes, e.g., if a TensorList variant
  95   tensor is captured as a placeholder, elements popped from that list would have
  96   unknown shape.
  97 
  98   Args:
  99     source_t: The tensor to copy HandleData from.
 100     target_t: The tensor to copy HandleData to.
 101   """
 102   if (target_t.dtype == dtypes_module.resource or
 103       target_t.dtype == dtypes_module.variant):
 104     if isinstance(source_t, ops.EagerTensor):
 105       handle_data = source_t._handle_data  # pylint: disable=protected-access
 106     else:
 107       handle_data = resource_variable_ops.get_resource_handle_data(source_t)
 108     if handle_data is not None and handle_data.is_set:
 109       # pylint: disable=protected-access
 110       pywrap_tensorflow.SetHandleShapeAndType(target_t.graph._c_graph,
 111                                               target_t._as_tf_output(),
 112                                               handle_data.SerializeToString())
 113       # pylint: enable=protected-access
 114       # Ensure that shapes and dtypes are propagated.
 115       shapes, types = zip(*[(pair.shape, pair.dtype)
 116                             for pair in handle_data.shape_and_type])
 117       ranks = [len(s.dim) if not s.unknown_rank else -1 for s in shapes]
 118       shapes = [[d.size for d in s.dim]
 119                 if not s.unknown_rank else None for s in shapes]
 120       pywrap_tensorflow.TF_GraphSetOutputHandleShapesAndTypes_wrapper(
 121           target_t._op._graph._c_graph,  # pylint: disable=protected-access
 122           target_t._as_tf_output(),  # pylint: disable=protected-access
 123           shapes, ranks, types)

 134 def _parse_func_attrs(attributes):
 135   """Convert the keyword arguments into function_def attributes.
 136 
 137   Currently only support primitive types: bool, int, float and string.
 138 
 139   Args:
 140     attributes: the dictionary of attributes.
 141   Returns:
 142     A dict of attributes where the key is the name of attribute and the value
 143       is the AttrValue proto.
 144   Raises:
 145     ValueError: If the kwargs contains unwhitelisted name or unsupported value
 146       types.
 147   """
 148   attrs = {}
 149   for key, value in attributes.items():
 150     if not any([re.match(reg, key)
 151                 for reg in WHITELIST_FUNCTION_ATTRIBUTE_REGEX]):
 152       raise ValueError("Attribute name is not whitelisted. "
 153                        "Whitelisted: prefix %s, got: %s" %
 154                        (WHITELIST_FUNCTION_ATTRIBUTE_REGEX, key))
 155 
 156     if isinstance(value, attr_value_pb2.AttrValue):
 157       attrs[key] = value
 158     # bool type check has to happen before int since bool is a subclass of int.
 159     elif isinstance(value, bool):
 160       attrs[key] = attr_value_pb2.AttrValue(b=value)
 161     elif isinstance(value, int):
 162       attrs[key] = attr_value_pb2.AttrValue(i=value)
 163     elif isinstance(value, float):
 164       attrs[key] = attr_value_pb2.AttrValue(f=value)
 165     elif isinstance(value, (str, bytes)):
 166       attrs[key] = attr_value_pb2.AttrValue(s=compat.as_bytes(value))
 167     else:
 168       raise ValueError("Unsupported attribute type for %s with type %s" %
 169                        (key, type(value)))
 170   return attrs

 173 class FuncGraph(ops.Graph):
 174   """Graph representing a function body.
 175 
 176   Attributes:
 177     name: The name of the function.
 178     inputs: Placeholder tensors representing the inputs to this function. The
 179       tensors are in this FuncGraph. This represents "regular" inputs as well as
 180       captured inputs (i.e. the values of self.captures), with the regular
 181       inputs coming first.
 182     outputs: Tensors that will be returned by this function. The tensors are in
 183       this FuncGraph.
 184     structured_outputs: A possibly-nested python object which will be returned
 185       by this function. The Tensors in this structure are the same as those of
 186       self.outputs. Note that this structure might contain Python `None`s.
 187     variables: Variables that should be watched during function execution.
 188     outer_graph: The graph this function is defined in. May be another FuncGraph
 189       or the global default Graph.
 190     captures: Maps external tensor -> internal tensor (i.e. input placeholder).
 191       The entries are in the order they were captured.
 192     seed: The graph-level random seed.
 193   """
 194 
 195   def __init__(self, name):
 196     """Construct a new FuncGraph.
 197 
 198     The graph will inherit its graph key, collections, seed, device stack, and
 199     distribution strategy stack from the current context or graph.
 200 
 201     Args:
 202       name: the name of the function.
 203     """
 204     super(FuncGraph, self).__init__()
 205 
 206     self.name = name
 207     self.inputs = []
 208     self.outputs = []
 209     self.structured_outputs = None
 210     self._weak_variables = []
 211     self.outer_graph = ops.get_default_graph()
 212     self.captures = collections.OrderedDict()
 213 
 214     self._building_function = True
 215     # Map from resource tensor name to last op (in program order) which uses
 216     # this tensor. Used to enforce that execution order matches program order
 217     # for resource tensors.
 218     self._last_op_using_resource_tensor = {}
 219 
 220     graph = self.outer_graph
 221 
 222     if context.executing_eagerly():
 223       self.seed = context.global_seed()
 224       self._xla_compile = (context.context().device_spec.device_type == "TPU")
 225       self._add_device_to_stack(context.context().device_name)
 226     else:
 227       self.seed = graph.seed
 228       self._xla_compile = getattr(graph, "_xla_compile", False)
 229       self._device_function_stack = graph._device_function_stack.copy()  # pylint: disable=protected-access
 230       self._colocation_stack = graph._colocation_stack.copy()  # pylint: disable=protected-access
 231 
 232     # TODO(b/112165328, b/112906995): summaries depend on inheriting collections
 233     # from the default graph even in eager mode. It'd be nice to not have a
 234     # default graph with eager execution, so hopefully this will go away when we
 235     # remove collections.
 236     # pylint: disable=protected-access
 237     self._collections = graph._collections
 238     # TODO(b/112906995): distribution strategy depends on inheriting this stack
 239     # from the default graph even in eager mode. Maybe it should be part of the
 240     # eager context?
 241     self._distribution_strategy_stack = graph._distribution_strategy_stack
 242     # Inherit the graph key, since this is used for matching variables in
 243     # optimizers.
 244     self._graph_key = graph._graph_key
 245     # pylint: enable=protected-access
 246 
 247   @property
 248   def variables(self):
 249     """A list of variables accessed by this FuncGraph.
 250 
 251     Note that functions keep only weak references to variables. Calling the
 252     function after a variable it accesses has been deleted is an error.
 253 
 254     Yields:
 255       Strong references to variables accessed by this FuncGraph.
 256     """
 257     for weak_v in self._weak_variables:
 258       v = weak_v()
 259       if v is None:
 260         raise AssertionError(
 261             "Called a function referencing variables which have been deleted. "
 262             "This likely means that function-local variables were created and "
 263             "not referenced elsewhere in the program. This is generally a "
 264             "mistake; consider storing variables in an object attribute on "
 265             "first call.")
 266       yield v
 267 
 268   @variables.setter
 269   def variables(self, var_list):
 270     self._weak_variables = [weakref.ref(v) for v in var_list]

 273     # Drop control dependencies to outside of the graph. TODO(b/117109273)
 274     # unclear how to capture an op, not a tensor.
 275     if not control_inputs:
 276       return super(FuncGraph, self).control_dependencies(control_inputs)
 277     return super(FuncGraph, self).control_dependencies(
 278         [c for c in control_inputs

 281   def create_op(
 282       self,
 283       op_type,
 284       inputs,
 285       dtypes,
 286       input_types=None,
 287       name=None,
 288       attrs=None,
 289       op_def=None,
 290       compute_shapes=True,
 291       compute_device=True):
 292     """Like Graph.create_op, except handles external input tensors.
 293 
 294     This overload adds functionality to create_op to "capture" any external
 295     input tensors, i.e. tensors from the eager context or outer function graphs
 296     if this is a nested function. See `capture` for more information.
 297 
 298     Args:
 299       op_type: The `Operation` type to create. This corresponds to the
 300         `OpDef.name` field for the proto that defines the operation.
 301       inputs: A list of `Tensor` objects that will be inputs to the `Operation`.
 302       dtypes: A list of `DType` objects that will be the types of the tensors
 303         that the operation produces.
 304       input_types: (Optional.) A list of `DType`s that will be the types of
 305         the tensors that the operation consumes. By default, uses the base
 306         `DType` of each input in `inputs`. Operations that expect
 307         reference-typed inputs must specify `input_types` explicitly.
 308       name: (Optional.) A string name for the operation. If not specified, a
 309         name is generated based on `op_type`.
 310       attrs: (Optional.) A dictionary where the key is the attribute name (a
 311         string) and the value is the respective `attr` attribute of the
 312         `NodeDef` proto that will represent the operation (an `AttrValue`
 313         proto).
 314       op_def: (Optional.) The `OpDef` proto that describes the `op_type` that
 315         the operation will have.
 316       compute_shapes: (Optional.) Deprecated. Has no effect (shapes are always
 317         computed).
 318       compute_device: (Optional.) If True, device functions will be executed
 319         to compute the device property of the Operation.
 320 
 321     Returns:
 322       An `Operation` object.
 323     """
 324     # This capturing logic interacts poorly with control flow contexts which
 325     # want to replace inputs of ops far too late in the process. This can lead
 326     # the context to get confused and try to create an Enter for an Enter. We
 327     # can detect this here and skip the additional Enter which can confuse loop
 328     # validation logic.
 329     if op_type == "Enter" and inputs[0].op.type == "Enter":
 330       if inputs[0].op.get_attr("frame_name") == attrs["frame_name"].s:
 331         return inputs[0].op
 332     # Calling AddValue on the control flow contexts to force creation of the
 333     # backward accumulators in the original graph before we create placeholders
 334     # to capture the inputs.
 335     ctxt = ops.get_default_graph()._control_flow_context  # pylint: disable=protected-access
 336     for i, inp in enumerate(inputs):
 337       # TPU Estimator defines a control flow context with no AddValue method.
 338       if ctxt is not None and hasattr(ctxt, "AddValue"):
 339         inp = ctxt.AddValue(inp)
 340       inp = self.capture(inp)
 341       inputs[i] = inp
 342     return super(FuncGraph, self).create_op(
 343         op_type, inputs, dtypes, input_types, name, attrs, op_def,
 344         compute_device=compute_device)

 346   def capture(self, tensor, name=None):
 347     """Captures `tensor` if it's external to this graph.
 348 
 349     If `tensor` is from a different graph, returns a placeholder for it.
 350     `tensor` and the placeholder will appear in self.captures, and the
 351     placeholder will appear in self.inputs.  Multiple calls to this method with
 352     the same `tensor` argument will return the same placeholder. If `tensor` is
 353     from this graph, returns `tensor`.
 354 
 355     Args:
 356       tensor: Tensor. May be from this FuncGraph or a different graph.
 357       name: Optional name if a placeholder is created.
 358 
 359     Returns:
 360       Tensor from this FuncGraph.
 361     """
 362     if isinstance(tensor, ops.EagerTensor):
 363       if name is None:
 364         name = str(ops.uid())
 365       return self._capture_helper(tensor, name)
 366     if tensor.graph is not self:
 367       if name is None:
 368         name = tensor.op.name
 369       return self._capture_helper(tensor, name)
 370     return tensor
 371 
 372   def _capture_helper(self, tensor, name):
 373     captured_tensor = self.captures.get(tensor, None)
 374     if captured_tensor is None:
 375       captured_tensor = _create_substitute_placeholder(tensor, name=name,
 376                                                        dtype=tensor.dtype)
 377       self.captures[tensor] = captured_tensor
 378       self.inputs.append(captured_tensor)
 379     tape.record_operation("captured_value", [captured_tensor], [tensor],
 380                           lambda x: [x])
 381     return captured_tensor

 417 class _EagerDefinedFunction(object):
 418   """Callable with the interface of `framework.function._DefinedFunction.`
 419 
 420   `_EagerDefinedFunction` encapsulates a function definition and its properties,
 421   and it provides a method for calling the encapsulated function. Some Ops
 422   take functions as attributes, which have type `func`; an instance of this
 423   class may be provided as the value of these `func` attributes.
 424   """
 425 
 426   def __init__(self, name, graph, inputs, outputs, attrs):
 427     """Initializes an eager defined function.
 428 
 429     Args:
 430       name: str, the name for the created function.
 431       graph: Graph, the graph containing the operations in the function
 432       inputs: the tensors in the graph to be used as inputs to the function
 433       outputs: the tensors in the graph which will be outputs to the function
 434       attrs: dict mapping names of attributes to their AttrValue values
 435     """
 436     operations = [
 437         op for op in graph.get_operations()
 438         if op not in set(arg.op for arg in inputs)
 439     ]
 440     fn = pywrap_tensorflow.TF_GraphToFunction_wrapper(
 441         graph._c_graph,  # pylint: disable=protected-access
 442         compat.as_str(name),
 443         False,
 444         [o._c_op for o in operations],  # pylint: disable=protected-access
 445         [t._as_tf_output() for t in inputs],  # pylint: disable=protected-access
 446         [t._as_tf_output() for t in outputs],  # pylint: disable=protected-access
 447         [],
 448         None,
 449         compat.as_str(""))
 450 
 451     for name, attr_value in attrs.items():
 452       serialized = attr_value.SerializeToString()
 453       # TODO(iga): this creates and deletes a new TF_Status for every attr.
 454       # It might be worth creating a convenient way to re-use status.
 455       pywrap_tensorflow.TF_FunctionSetAttrValueProto(
 456           fn, compat.as_str(name), serialized)
 457 
 458     # TODO(apassos) avoid creating a FunctionDef (specially to grab the
 459     # signature, but also in general it's nice not to depend on it.
 460     with c_api_util.tf_buffer() as buffer_:
 461       pywrap_tensorflow.TF_FunctionToFunctionDef(fn, buffer_)
 462       proto_data = pywrap_tensorflow.TF_GetBuffer(buffer_)
 463     function_def = function_pb2.FunctionDef()
 464     function_def.ParseFromString(compat.as_bytes(proto_data))
 465     if context.executing_eagerly():
 466       _register(fn)
 467     self.definition = function_def
 468     self.name = compat.as_bytes(function_def.signature.name)
 469     self.signature = function_def.signature
 470     self._num_outputs = len(self.signature.output_arg)
 471     self._output_types = [o.type for o in self.signature.output_arg]
 472     self._output_shapes = [o.shape for o in outputs]
 473     self._func_graph_outputs = outputs
 474     self.grad_func_name = None
 475     self.python_grad_func = None
 476     self._c_func = c_api_util.ScopedTFFunction(fn)
 477     self._grad_func = None
 478     self._graph = graph
 479     self._stateful_ops = tuple(op for op in operations if op.op_def.is_stateful)
 480

 494   def call(self, ctx, args):
 495     """Calls this function with `args` as inputs.
 496 
 497     Function execution respects device annotations only if the function won't
 498     be compiled with xla.
 499 
 500     Args:
 501       ctx: a Context object
 502       args: a list of arguments to supply this function with.
 503 
 504     Returns:
 505       The outputs of the function call.
 506     """
 507 
 508     executing_eagerly = ctx.executing_eagerly()
 509 
 510     if self._graph._xla_compile:  # pylint: disable=protected-access
 511       # XLA compilation relies upon a custom kernel creator to run functions.
 512       signature = self.signature
 513       if executing_eagerly:
 514         outputs = execute.execute(
 515             str(signature.name),
 516             num_outputs=self._num_outputs,
 517             inputs=args,
 518             attrs=None,
 519             ctx=ctx)
 520       else:
 521         g = ops.get_default_graph()
 522         self.add_to_graph(g)
 523         op = g.create_op(
 524             signature.name,
 525             [ops.internal_convert_to_tensor(x, ctx=ctx) for x in args],
 526             tuple(dtypes_module.DType(x.type) for x in signature.output_arg),
 527             op_def=signature,
 528             name="FunctionCall",
 529             compute_shapes=False)
 530         outputs = op.outputs
 531         if not outputs:
 532           return op
 533         outputs = [outputs] if isinstance(
 534             outputs, (ops.Tensor, type(None))) else list(outputs)
 535     else:
 536       # TODO(akshayka): Either remove this if the FunctionLibraryRuntime
 537       # creates `PartitionedCallOp` kernels by default, or remove the previous
 538       # branch if a TPU kernel is registered for `PartitionedCall`.
 539       outputs = functional_ops.partitioned_call(
 540           args=args,
 541           f=self,
 542           tout=self._output_types,
 543           executing_eagerly=executing_eagerly)
 544 
 545     if executing_eagerly:
 546       return outputs
 547     else:
 548       for i, shape in enumerate(self._output_shapes):
 549         outputs[i].set_shape(shape)
 550       for i, func_graph_output in enumerate(self._func_graph_outputs):
 551         _copy_handle_data(func_graph_output, outputs[i])
 552       return outputs

 555 def _flatten(sequence):
 556   """A wrapper around `nest.flatten` that also unpacks `IndexedSlices`."""
 557   # TODO(akshayka): Support `SparseTensor` in a similar fashion.
 558   flat_sequence = nest.flatten(sequence)
 559   outputs = []
 560   for item in flat_sequence:
 561     if isinstance(item, ops.IndexedSlices):
 562       if item.dense_shape is not None:
 563         outputs.extend([item.values, item.indices, item.dense_shape])
 564       else:
 565         outputs.extend([item.values, item.indices])
 566     else:
 567       outputs.append(item)
 568   return outputs

 578   def __init__(self, func_graph, attrs=None):
 579     """Initialize a Function.
 580 
 581     Args:
 582       func_graph: An instance of FuncGraph: the function body to wrap.
 583       attrs: (optional) dict mapping names of attributes to their AttrValue
 584         values. Attributes in `attrs` will be included in this function's
 585         definition.
 586 
 587     Raises:
 588       ValueError: If number of input_placeholders is not equal to the number
 589         of function inputs.
 590     """
 591     self._func_graph = func_graph
 592     self._captured_inputs = list(self._func_graph.captures.keys())
 593     self._num_outputs = len(self._func_graph.outputs)
 594     self._output_shapes = tuple(
 595         output.shape for output in self._func_graph.outputs)
 596     self._attrs = _parse_func_attrs(attrs or {})
 597     self._device_functions = tuple(
 598         self._func_graph._device_functions_outer_to_inner)  # pylint: disable=protected-access
 599 
 600     self._inference_function = _EagerDefinedFunction(
 601         _inference_name(self._func_graph.name), self._func_graph,
 602         self._func_graph.inputs, self._func_graph.outputs, self._attrs)
 603     self._backward_graph_function = None
 604 
 605     # Map holding distributed variables, keyed by resource handle tensors.
 606     self._distributed_variables = {}
 607     strategy = distribution_strategy_context.get_distribution_strategy()
 608     for variable in self._func_graph.variables:
 609       # If variable is not distributed, unwrap returns [variable].
 610       component_variables = strategy.unwrap(variable)
 611       # Only update the dictionary when the variable is actually distributed.
 612       if (len(component_variables) > 1 or component_variables[0] != variable):
 613         for component_variable in component_variables:
 614           self._distributed_variables[component_variable.handle] = variable

 616   def __call__(self, *args):
 617     """Executes the wrapped function.
 618 
 619     Args:
 620       *args: a list of Tensors or Variables.
 621 
 622     Returns:
 623       The result of applying the TF function to `args`.
 624 
 625     Raises:
 626       ValueError: If the current device stack does not match the device stack
 627         under which the function was created, or if `args` contains anything
 628         other than Tensors or Variables.
 629     """
 630     ctx = context.context()
 631     device_functions = _get_device_functions(ctx, ops.get_default_graph())
 632     if device_functions != self._device_functions:
 633       raise ValueError(
 634           "The current device stack does not match the device stack under "
 635           "which the TensorFlow function '%s' was created.\n"
 636           "Current device stack: %s\n%s device stack: %s" %
 637           (self._inference_function.name, device_functions,
 638            self._inference_function.name, self._device_functions))
 639 
 640     for v in self._func_graph.variables:
 641       if v.trainable:
 642         tape.variable_accessed(v)
 643 
 644     captures = self._resolve_captured_inputs()
 645     tensor_inputs = []
 646     for i, arg in enumerate(nest.flatten(args)):
 647       if isinstance(arg, resource_variable_ops.ResourceVariable):
 648         if arg.trainable:
 649           tape.variable_accessed(arg)
 650         tensor_inputs.append(arg.handle)
 651       elif isinstance(arg, ops.Tensor):
 652         tensor_inputs.append(arg)
 653       else:
 654         raise ValueError("All inputs to `Function`s must be Tensors; "
 655                          "on invocation of %s, the %d-th input (%s) was not a "
 656                          "Tensor." % (self._func_graph.name, i, str(arg)))
 657     args = tensor_inputs + captures
 658 
 659     if tape.should_record(tensor_inputs) or tape.should_record(captures):
 660       return self._backprop_call(args)
 661 
 662     outputs = self._inference_function.call(ctx, args)
 663     return self._build_call_outputs(outputs)

 722   def _construct_backprop_function(self):
 723     """Constructs the backprop function object for this function."""
 724     backwards_graph = FuncGraph(_backward_name(self._func_graph.name))
 725     forward_function_name = _forward_name(self._func_graph.name)
 726     with backwards_graph.as_default():
 727       gradients_wrt_outputs = [
 728           graph_placeholder(x.dtype, x.shape) for x in self._func_graph.outputs
 729       ]
 730       gradients_wrt_inputs = gradients_impl._GradientsHelper(  # pylint: disable=protected-access
 731           self._func_graph.outputs,
 732           self._func_graph.inputs,
 733           grad_ys=gradients_wrt_outputs,
 734           src_graph=self._func_graph)
 735 
 736     backwards_graph_captures = list(backwards_graph.captures.keys())
 737 
 738     backward_function_attr = _parse_func_attrs(
 739         {FORWARD_FUNCTION_ATTRIBUTE_NAME: forward_function_name})
 740     backward_function_attr.update(self._attrs)
 741 
 742     # The ordering of `backwards_graph.inputs` is important: inputs of
 743     # `self._backward_graph_function` correspond to outputs of
 744     # `self._forward_function`.
 745     backwards_graph.inputs = gradients_wrt_outputs + list(
 746         backwards_graph.captures.values())
 747     # Clear captures, since we pass them in as inputs.
 748     backwards_graph.captures = {}
 749     backwards_graph.outputs.extend(
 750         grad for grad in _flatten(gradients_wrt_inputs) if grad is not None)
 751     backwards_graph.structured_outputs = gradients_wrt_inputs
 752     self._backward_graph_function = Function(
 753         backwards_graph, attrs=backward_function_attr)
 754 
 755     forward_function_attr = _parse_func_attrs({
 756         BACKWARD_FUNCTION_ATTRIBUTE_NAME:
 757             self._backward_graph_function._inference_function.name})  # pylint: disable=protected-access
 758     forward_function_attr.update(self._attrs)
 759 
 760     self._forward_function = _EagerDefinedFunction(
 761         forward_function_name, self._func_graph, self._func_graph.inputs,
 762         self._func_graph.outputs + backwards_graph_captures,
 763         forward_function_attr)

 765   def _backprop_call(self, args):
 766     """Calls the forward function and records the result on a tape.
 767 
 768     (Only records results on a tape if the function has outputs)
 769 
 770     Args:
 771       args: All inputs to the function, including resolved captured inputs
 772 
 773     Returns:
 774       The call output.
 775     """
 776     if self._backward_graph_function is None:
 777       self._construct_backprop_function()
 778 
 779     ctx = context.context()
 780     outputs = self._forward_function.call(ctx, args)
 781     if isinstance(outputs, ops.Operation) or outputs is None:
 782       return outputs
 783 
 784     # `real_outputs` are the actual outputs of the inference graph function;
 785     # `side_outputs` are the intermediate Tensors that were added as outputs to
 786     # the forward graph function so that we can compute its gradient.
 787     real_outputs = outputs[:self._num_outputs]
 788     side_outputs = outputs[self._num_outputs:]
 789 
 790     def backward_function(*args):
 791       return self._backward_graph_function(*(list(args) + side_outputs))  # pylint: disable=not-callable
 792 
 793     tape.record_operation(self._forward_function.signature.name, real_outputs,
 794                           args, backward_function)
 795     return self._build_call_outputs(real_outputs)

 797   def _resolve_captured_inputs(self):
 798     """Resolve captured distributed variables to their current values.
 799 
 800     Some inputs can be distributed variables. Such variables yield a different
 801     component (i.e. actual tf.Variable) variables depending on the context of
 802     execution.
 803 
 804     Returns:
 805       a list of resolved captured input tensors.
 806     """
 807     if self._distributed_variables:
 808       # Loop over each captured input and check if it corresponds to something
 809       # distributed. If so, get its _distributed_container and fetch the
 810       # component appropriate for the current execution context.
 811       resolved_captured_inputs = self._captured_inputs[:]
 812       for i, captured_input in enumerate(self._captured_inputs):
 813         distributed_var = self._distributed_variables.get(captured_input, None)
 814         if distributed_var is not None:
 815           # distributed variables override __getattr__ and substitute the
 816           # right component variable. In here, `distributed_var.handle`
 817           # actually does the equivalent of
 818           # distributed_var.get_current_component_var().handle.
 819           resolved_captured_inputs[i] = distributed_var.handle
 820       return resolved_captured_inputs
 821     return self._captured_inputs

 823   def _build_call_outputs(self, result):
 824     """Maps the fdef output list to actual output structure.
 825 
 826     Args:
 827       result: Output lists defined by FunctionDef.
 828     Returns:
 829       The actual call output.
 830     """
 831     if self._func_graph.structured_outputs is None:
 832       return result
 833 
 834     # Use `nest.flatten` instead of `_flatten` in order to preserve any
 835     # IndexedSlices in `self._func_graph.structured_outputs`.
 836     outputs_list = nest.flatten(self._func_graph.structured_outputs)
 837     j = 0
 838     for i, o in enumerate(outputs_list):
 839       if o is not None:
 840         if isinstance(o, ops.IndexedSlices):
 841           # Repack Tensors for IndexedSlices.
 842           if o.dense_shape is not None:
 843             outputs_list[i] = ops.IndexedSlices(
 844                 values=result[j],
 845                 indices=result[j + 1],
 846                 dense_shape=result[j + 2])
 847             j += 3
 848           else:
 849             outputs_list[i] = ops.IndexedSlices(
 850                 values=result[j], indices=result[j + 1])
 851             j += 2
 852         else:
 853           outputs_list[i] = result[j]
 854           j += 1
 855     ret = nest.pack_sequence_as(self._func_graph.structured_outputs,
 856                                 outputs_list)
 857     return ret

 878 def func_graph_from_py_func(name,
 879                             python_func,
 880                             args,
 881                             kwargs,
 882                             signature=None,
 883                             func_graph=None):
 884   """Returns a `FuncGraph` generated from `python_func`.
 885 
 886   Args:
 887     name: an identifier for the function.
 888     python_func: the Python function to trace.
 889     args: the positional args with which the Python function should be called;
 890       ignored if a signature is provided.
 891     kwargs: the keyword args with which the Python function should be called;
 892       ignored if a signature is provided.
 893     signature: a possibly nested sequence of `TensorSpecs` specifying the shapes
 894       and dtypes of the arguments. When a signature is provided, `args` and
 895       `kwargs` are ignored, and `python_func` is traced with Tensors conforming
 896       to `signature`. If `None`, the shapes and dtypes are inferred from the
 897       inputs.
 898     func_graph: Optional. An instance of FuncGraph. If provided, we will use
 899       this graph else a new one is built and returned.
 900 
 901   Returns:
 902     A FuncGraph.
 903 
 904   Raises:
 905     TypeError: If any of `python_func`'s return values is neither `None` nor a
 906       `Tensor`.
 907   """
 908   if func_graph is None:
 909     func_graph = FuncGraph(name)
 910   assert isinstance(func_graph, FuncGraph)
 911   with func_graph.as_default(), AutomaticControlDependencies() as a:
 912     variable_scope.get_variable_scope().set_use_resource(True)
 913 
 914     if signature is None:
 915       func_args = _get_defun_inputs_from_args(args)
 916       func_kwargs = _get_defun_inputs_from_args(kwargs)
 917     else:
 918       func_args = _get_defun_inputs_from_signature(signature)
 919       func_kwargs = {}
 920 
 921     # Note: `nest.flatten` sorts by keys, as does `_deterministic_dict_values`.
 922     # Variables to help check whether mutation happens in calling the function
 923     # Copy the recursive list, tuple and map structure, but not base objects
 924     func_args_before = nest.pack_sequence_as(func_args, nest.flatten(func_args))
 925     func_kwargs_before = nest.pack_sequence_as(
 926         func_kwargs, nest.flatten(func_kwargs))
 927 
 928     def convert(x):
 929       """Converts an argument to a Tensor."""
 930       if x is None:
 931         return None
 932       try:
 933         x = ops.convert_to_tensor_or_indexed_slices(x)
 934       except (ValueError, TypeError):
 935         raise TypeError(
 936             "To be compatible with tf.contrib.eager.defun, Python functions "
 937             "must return zero or more Tensors; in compilation of %s, found "
 938             "return value of type %s, which is not a Tensor." %
 939             (str(python_func), type(x)))
 940       x = a.mark_as_return(x)
 941       return x
 942 
 943     this_tape = tape.push_new_tape()
 944     try:
 945       func_outputs = python_func(*func_args, **func_kwargs)
 946       # invariant: `func_outputs` contains only Tensors and `None`s.
 947       func_outputs = nest.map_structure(convert, func_outputs)
 948 
 949       def check_mutation(n1, n2):
 950         """Check if two list of arguments are exactly the same."""
 951         errmsg = ("Function to be traced should not modify structure of input "
 952                   "arguments. Check if your function has list and dictionary "
 953                   "operations that alter input arguments, "
 954                   "such as `list.pop`, `list.append`")
 955         try:
 956           nest.assert_same_structure(n1, n2)
 957         except ValueError:
 958           raise ValueError(errmsg)
 959 
 960         for arg1, arg2 in zip(nest.flatten(n1), nest.flatten(n2)):
 961           if arg1 is not arg2:
 962             raise ValueError(errmsg)
 963 
 964       check_mutation(func_args_before, func_args)
 965       check_mutation(func_kwargs_before, func_kwargs)
 966     finally:
 967       tape.pop_tape(this_tape)
 968 
 969     # Variables in `func_args`, `func_kwargs` should be explicit inputs
 970     # to the function, not captured inputs.
 971     tape_variables = this_tape.watched_variables()
 972     arg_variables = set()
 973     inputs = []
 974     for arg in nest.flatten(func_args) + nest.flatten(func_kwargs):
 975       if isinstance(arg, resource_variable_ops.ResourceVariable):
 976         try:
 977           resource_placeholder = func_graph.captures.pop(arg.handle)
 978           arg_variables.add(arg)
 979         except KeyError:
 980           # This case occurs if a Variable among the inputs is not actually
 981           # used by the function; we still add an explicit input for it
 982           # because the user should presumably pass the Variable as an input
 983           # to the corresponding graph function.
 984           resource_placeholder = _create_substitute_placeholder(arg.handle)
 985         inputs.append(resource_placeholder)
 986       elif isinstance(arg, ops.Tensor):
 987         inputs.append(arg)
 988     variables = [v for v in tape_variables if v not in arg_variables]
 989     func_graph.inputs = inputs + list(func_graph.captures.values())
 990 
 991     func_graph.structured_outputs = func_outputs
 992     # Returning a closed-over tensor does not trigger convert_to_tensor.
 993     func_graph.outputs.extend(
 994         func_graph.capture(x)
 995         for x in _flatten(func_graph.structured_outputs)
 996         if x is not None)
 997 
 998     # Some captured variables might be components of DistributedValues.
 999     # Instead of storing non-distributed component variables, we
1000     # store their distributed containers so we can retrieve the correct
1001     # component variables at call-time.
1002     strategy = distribution_strategy_context.get_distribution_strategy()
1003     for i, variable in enumerate(variables):
1004       # If variable is not distributed value_container returns itself.
1005       variables[i] = strategy.value_container(variable)
1006     func_graph.variables = variables
1007 
1008   # Register any other functions defined in the graph.
1009   if context.executing_eagerly():
1010     for f in func_graph._functions.values():  # pylint: disable=protected-access
1011       # TODO(ashankar): What about the gradient registry?
1012       _register(f._c_func.func)  # pylint: disable=protected-access
1013 
1014   return func_graph

1025 class PolymorphicFunction(object):
1026   """Wrapper class for the graph functions defined for a Python function.
1027 
1028   See the documentation for `defun` for more information on the semantics of
1029   defined functions.
1030 
1031   PolymorphicFunction class is thread-compatible meaning that minimal
1032   usage of defuns (defining and calling) is thread-safe, but if users call other
1033   methods or invoke the base `python_function` themselves, external
1034   synchronization is necessary.
1035   """
1036 
1037   def __init__(self,
1038                python_function,
1039                name,
1040                input_signature=None,
1041                attributes=None):
1042     """Initializes a polymorphic function.
1043 
1044     Args:
1045       python_function: the function to be wrapped.
1046       name: the name given to it.
1047       input_signature: a possibly nested sequence of `TensorSpec` objects
1048         specifying the input signature of this function. If `None`, a separate
1049         function is instantiated for each inferred input signature.
1050       attributes: dict, extra keyword arguments that will be added as attribute
1051          of the function.
1052 
1053     Raises:
1054       ValueError: if `input_signature` is not None and the `python_function`'s
1055         argspec has keyword arguments.
1056     """
1057 
1058     if isinstance(python_function, functools.partial):
1059       self._python_function = python_function.func
1060       self._args_to_prepend = python_function.args or tuple()
1061       self._kwargs_to_include = python_function.keywords or {}
1062     else:
1063       self._python_function = python_function
1064       self._args_to_prepend = tuple()
1065       self._kwargs_to_include = {}
1066     self._name = name
1067     self._function_cache = collections.OrderedDict()
1068     self._function_attributes = attributes or {}
1069 
1070     self._lock = threading.Lock()
1071 
1072     fullargspec = tf_inspect.getfullargspec(self._python_function)
1073     if tf_inspect.ismethod(self._python_function):
1074       # Remove `self`: default arguments shouldn't be matched to it.
1075       args = fullargspec.args[1:]
1076     else:
1077       args = fullargspec.args
1078 
1079     # A cache mapping from argument name to index, for canonicalizing
1080     # arguments that are called in a keyword-like fashion.
1081     self._args_to_indices = {arg: i for i, arg in enumerate(args)}
1082     # A cache mapping from arg index to default value, for canonicalization.
1083     offset = len(args) - len(fullargspec.defaults or [])
1084     self._arg_indices_to_default_values = {
1085         offset + index: default
1086         for index, default in enumerate(fullargspec.defaults or [])
1087     }
1088     self._default_values = fullargspec.defaults
1089     self._default_values_start_index = offset
1090     if input_signature is None:
1091       self._input_signature = None
1092     else:
1093       if fullargspec.varkw is not None or fullargspec.kwonlyargs:
1094         raise ValueError("Cannot define a TensorFlow function from a Python "
1095                          "function with keyword arguments when "
1096                          "input_signature is provided.")
1097 
1098       if not isinstance(input_signature, (tuple, list)):
1099         raise TypeError("input_signature must be either a tuple or a "
1100                         "list, received " + str(type(input_signature)))
1101 
1102       self._input_signature = tuple(input_signature)
1103       self._flat_input_signature = tuple(nest.flatten(input_signature))

1128   def __get__(self, instance, owner):
1129     """Makes it possible to defun instance methods."""
1130     del owner
1131     # `instance` here is the instance that this `PolymorphicFunction` was
1132     # accessed through; e.g., for
1133     #
1134     #   class Foo(object):
1135     #
1136     #     @function.defun
1137     #     def bar(self):
1138     #       ...
1139     #
1140     #   foo = Foo()
1141     #   foo.bar()  # `foo.bar` is a `PolymorphicFunction` instance
1142     #
1143     # then `instance` will be `foo` (and `owner` will be `Foo`).
1144     return functools.partial(self.__call__, instance)

1146   def _cache_key(self, args, kwargs):
1147     """Computes the cache key given inputs and execution context."""
1148     if self._input_signature is None:
1149       inputs = (args, kwargs) if kwargs else args
1150       cache_key = pywrap_tensorflow.TFE_Py_EncodeArg(inputs)
1151     else:
1152       del args, kwargs
1153       cache_key = self._flat_input_signature
1154 
1155     with ops.init_scope():
1156       init_graph = ops.get_default_graph()
1157 
1158       # The graph, or whether we're executing eagerly, should be a part of the
1159       # cache key so we don't improperly capture tensors such as variables.
1160       executing_eagerly = context.executing_eagerly()
1161       execution_context = executing_eagerly or init_graph
1162 
1163     default_graph = ops.get_default_graph()

1175   def _canonicalize_function_inputs(self, *args, **kwargs):
1176     """Canonicalizes `args` and `kwargs`.
1177 
1178     Canonicalize the inputs to the Python function using its fullargspec. In
1179     particular, we parse the varags and kwargs that this
1180     `PolymorphicFunction` was called with into a tuple corresponding to the
1181     Python function's positional (named) arguments and a dictionary
1182     corresponding to its kwargs.
1183 
1184     Args:
1185       *args: The varargs this object was called with.
1186       **kwargs: The keyword args this function was called with.
1187 
1188     Returns:
1189       A canonicalized ordering of the inputs.
1190 
1191     Raises:
1192       ValueError: If a keyword in `kwargs` cannot be matched with a positional
1193         argument when an input signature is specified, or when the inputs
1194         do not conform to the input signature.
1195     """
1196     args = self._args_to_prepend + args
1197     kwargs = dict(kwargs, **self._kwargs_to_include)
1198     # Maps from index of arg to its corresponding value, according to `args`
1199     # and `kwargs`; seeded with the default values for the named args that
1200     # aren't in `args`.
1201     if not kwargs:
1202       if self._default_values:
1203         inputs = args + self._default_values[len(args) -
1204                                              self._default_values_start_index:]
1205       else:
1206         inputs = args
1207     else:
1208       arg_indices_to_values = {
1209           index: default for index, default in six.iteritems(
1210               self._arg_indices_to_default_values) if index >= len(args)
1211       }
1212       consumed_args = []
1213       for arg, value in six.iteritems(kwargs):
1214         index = self._args_to_indices.get(arg, None)
1215         if index is not None:
1216           arg_indices_to_values[index] = value
1217           consumed_args.append(arg)
1218         elif self._input_signature is not None:
1219           raise ValueError("Cannot define a TensorFlow function from a Python "
1220                            "function with keyword arguments when "
1221                            "input_signature is provided.")
1222       for arg in consumed_args:
1223         # After this loop, `kwargs` will only contain true keyword arguments, as
1224         # opposed to named arguments called in a keyword-like fashion.
1225         kwargs.pop(arg)
1226       inputs = args + _deterministic_dict_values(arg_indices_to_values)
1227     flat_inputs = nest.flatten(inputs)
1228 
1229     # Check for NumPy arrays in arguments and convert them to Tensors.
1230     need_packing = False
1231     for index, value in enumerate(flat_inputs):
1232       if isinstance(value, np.ndarray):
1233         flat_inputs[index] = constant_op.constant(value)
1234         need_packing = True
1235     if need_packing:
1236       inputs = nest.pack_sequence_as(structure=inputs,
1237                                      flat_sequence=flat_inputs)
1238     if self._input_signature is None:
1239       return inputs, kwargs
1240     else:
1241       assert not kwargs
1242       try:
1243         nest.assert_same_structure(self._input_signature, inputs)
1244       except (ValueError, TypeError):
1245         raise ValueError("Structure of Python function inputs does not match "
1246                          "input_signature.")
1247       if any(not isinstance(arg, ops.Tensor) for arg in flat_inputs):
1248         raise ValueError("When input_signature is provided, all inputs to "
1249                          "the Python function must be Tensors.")
1250       tensor_specs = [
1251           tensor_spec.TensorSpec.from_tensor(tensor) for tensor in flat_inputs
1252       ]
1253       if any(not spec.is_compatible_with(other)
1254              for spec, other in zip(self._flat_input_signature, tensor_specs)):
1255         raise ValueError("Python inputs incompatible with input_signature: "
1256                          "inputs (%s), input_signature (%s)" %
1257                          (str(inputs), str(self._input_signature)))
1258       return inputs, {}

1301   """Register the defun function into the graph.
1302 
1303   This won't actually call the function with the inputs, and only put the
1304   function definition into graph. Register function with different input param
1305   will result into multiple version of functions registered in graph.
1306 
1307   Args:
1308     func: the PolymorphicFunction instance that generated by a @defun
1309     *args: input arguments for the Python function.
1310     **kwargs: input keyword arguments for the Python function.
1311 
1312   Returns:
1313     a `Function` object specialized to inputs and execution context.
1314 
1315   Raises:
1316     ValueError: When the input function is not a defun wrapped python function.
1317   """

1319     raise ValueError("Only defun function is allowed to be registered. "
1320                      "Got type: %s" % type(func))

1322   graph = ops.get_default_graph()
1323 
1324   # There are two situations for the actual call of a defun:
1325   # 1. If none of the input args are resource variables or watch by any tape,
1326   #   it will run the _inference_function of concrete_func for forward pass, and
1327   #   the gradient will be generated by standard mechanism.

1339   backward_function.add_to_graph(graph)

1352 def defun(func=None, input_signature=None):
1353   """Compiles a Python function into a callable TensorFlow graph.
1354 
1355   `defun` (short for "define function") trace-compiles a Python function
1356   composed of TensorFlow operations into a callable that executes a `tf.Graph`
1357   containing those operations. The callable produced by `defun` contains only
1358   the subgraph of TensorFlow operations that were executed when the Python
1359   function was called with a particular input signature, defined as a list
1360   of the shapes and dtypes of the Python function's Tensor-valued arguments and
1361   the values of its non-Tensor Python objects. In particular, `defun` is _not_ a
1362   compiler for arbitrary Python code.
1363 
1364   When eager execution is enabled, the ability to create graphs from Python
1365   functions makes it possible to incrementally trade off debugability and
1366   interactivity for performance.  Functions compiled with `defun` cannot be
1367   inspected with `pdb` and `print` statements; however, executing a graph
1368   generated by `defun` sometimes takes less time and memory than eagerly
1369   executing the corresponding Python function, since specifying computations as
1370   graphs allows for optimizations like automatic buffer reuse and
1371   parallelization among ops. Note that executing a `defun`-compiled function
1372   incurs a small constant overhead, so eagerly executing sufficiently small
1373   Python functions might take less time than executing their corresponding
1374   `defun`-generated graphs.
1375 
1376   For a Python function to be compatible with `defun`, all of its arguments must
1377   be hashable Python objects or lists thereof. The function itself may not
1378   modify the list/map structure of its arguments. Additionally, it must return
1379   zero or more `tf.Tensor` objects. If the Python function returns
1380   a `tf.Variable`, its compiled version will return the value of that variable
1381   as a `tf.Tensor`.
1382 
1383   Executing a graph generated by `defun` respects device annotations (i.e.,
1384   all `with tf.device` directives present in a Python function will also be
1385   present in its corresponding graph), but it is not yet possible to execute the
1386   generated graphs across multiple machines.
1387 
1388   _Example Usage_
1389 
1390   ```python
1391   import tensorflow as tf
1392 
1393   tf.enable_eager_execution()
1394 
1395   # A simple example.
1396   def f(x, y):
1397     return tf.reduce_mean(tf.multiply(x ** 2, 3) + y)
1398 
1399   g = tf.contrib.eager.defun(f)
1400 
1401   x = tf.constant([[2.0, 3.0]])
1402   y = tf.constant([[3.0, -2.0]])
1403 
1404   # `f` and `g` will return the same value, but `g` will be executed as a
1405   # TensorFlow graph.
1406   assert f(x, y).numpy() == g(x, y).numpy()
1407 
1408   # `defun` is capable of compiling Python functions that close over Python
1409   # objects, including Tensors and Variables.
1410   @tf.contrib.eager.defun
1411   def h():
1412     return f(x, y)
1413 
1414   assert (h().numpy() == f(x, y).numpy()).all()
1415 
1416   # `defun` automatically lifts variables out of the graphs it creates,
1417   # allowing you to compile the `call` methods of `tf.keras.layers.Layer` and
1418   # `tf.keras.Model` objects.
1419   class MyModel(tf.keras.Model):
1420 
1421     def __init__(self, keep_probability=0.2):
1422       super(MyModel, self).__init__()
1423       self.dense1 = tf.keras.layers.Dense(4, activation=tf.nn.relu)
1424       self.dense2 = tf.keras.layers.Dense(5, activation=tf.nn.softmax)
1425       self.keep_probability = keep_probability
1426 
1427     @tf.contrib.eager.defun
1428     def call(self, inputs, training=True):
1429       x = self.dense2(self.dense1(inputs))
1430       if training:
1431         return tf.nn.dropout(x, self.keep_probability)
1432       else:
1433         return x
1434 
1435   model = MyModel()
1436   model(x, training=True)  # executes a graph, with dropout
1437   model(x, training=False) # executes a graph, without dropout
1438 
1439   # `defun`-compiled functions are differentiable.
1440   optimizer = tf.train.GradientDescentOptimizer(learning_rate=0.01)
1441   with tf.GradientTape() as tape:
1442     outputs = model(x)
1443   gradient = tape.gradient(outputs, model.trainable_variables)
1444   optimizer.apply_gradients((grad, var) for grad, var in zip(gradient,
1445                             model.trainable_variables))
1446   ```
1447 
1448   When using `defun`, there are subtleties regarding inputs, Python control
1449   flow, and variable creation that one should be aware of. For concreteness, let
1450   `f` be a Python function that returns zero or more `tf.Tensor` objects and
1451   let `F = defun(f)`. `F` builds a graph for each unique input signature it
1452   sees, Python control flow is baked into graphs, and operations related to
1453   variable initialization are automatically lifted out of the graphs that `F`
1454   generates and placed in the eager context if executing eagerly or into an
1455   outer graph otherwise.
1456 
1457   _Input Signatures_
1458   By default, `F = tf.contrib.eager.defun(f)` instantiates a separate graph
1459   for every unique sequence of the shapes and dtypes of Tensor arguments and
1460   the values of Python objects it is invoked with. For example, calling
1461   `F(tf.random_uniform([2])` will execute a different graph than
1462   `F(tf.random_uniform([3])` because the two inputs have different shapes.
1463   The first time that `F(*args, **kwargs)` is called with a particular sequence
1464   of Tensor shapes and dtypes and Python values, it constructs a graph by
1465   tracing the execution of `f(*args, **kwargs)`; this graph is bound to an
1466   input signature inferred from `(*args, **kwargs)` and cached for future reuse.
1467 
1468   NumPy arrays passed as inputs to `F` are converted to `tf.Tensor` objects
1469   before being passed to `f`, and are treated as Tensors for caching. This
1470   allows a function to be called multiple times with NumPy arrays having
1471   different values but the same shape and dtype without re-tracing each time.
1472 
1473   `tf.contrib.eager.defun` caches graphs for your convenience, letting you
1474   define TensorFlow functions without explicitly specifying their signatures.
1475   However, this policy is conservative and potentially expensive; for example,
1476   when different invocations of your function have differently-shaped Tensor
1477   inputs, this policy might generate more graph functions than necessary. To
1478   eliminate such costs, `tf.contrib.eager.defun` allows you to supply an
1479   optional `input_signature` argument specifying the shapes and dtypes of the
1480   inputs. In particular, the shapes may be partially unspecified, with `None`s
1481   in the unknown dimensions.  When an input signature is provided,
1482   `tf.contrib.eager.defun` will only instantiate a single graph for the
1483   decorated Python function. The following is an example:
1484 
1485   ```python
1486   import tensorflow as tf
1487 
1488   # The first `TensorSpec` below describes the shape and dtype of `words`,
1489   # and the second describes the shape and dtype of `another_tensor`. Note that
1490   # the last dimension of the `words` `TensorSpec` is left unspecified.
1491   @tf.contrib.eager.defun(input_signature=[
1492     tf.contrib.eager.TensorSpec(shape=[50, 300, None], dtype=tf.float32),
1493     tf.contrib.eager.TensorSpec(shape=[300, 100], dtype=tf.float32)
1494   ])
1495   def my_sequence_model(words, another_tensor):
1496     ...
1497 
1498   # Note how the third dimension of the first input can vary freely.
1499   words = tf.random_uniform(([50, 300, 10])
1500   second_input = tf.random_uniform([300, 100])
1501   my_sequence_model(words, second_input)
1502 
1503   words = tf.random_uniform(([50, 300, 20])
1504   my_sequence_model(words, second_input)
1505 
1506   # Passing an input with an incompatible shape will raise an error.
1507   words = tf.random_uniform(([50, 100, 20])
1508   my_sequence_model(words, second_input)  # <---- This will raise an error.
1509 
1510   ```
1511 
1512   Python functions that are compiled with an `input_signature` must only accept
1513   Tensors as arguments and must not take unnamed keyword arguments (**kwargs).
1514 
1515   _Tracing_
1516   Be aware that because `F` only logs TensorFlow operations, all the other
1517   Python code that `f` executes will only shape the _construction_ of the graphs
1518   that `F` executes: the Python code won't be executed when the graphs
1519   themselves are executed, though it will be executed every time the Python
1520   function is traced (and a given Python function might be traced multiple
1521   times, once for each input signature it is invoked with). For example, whereas
1522   the Python function
1523 
1524   ```python
1525   import tensorflow as tf
1526   import numpy as np
1527 
1528   tf.enable_eager_execution()
1529 
1530   def add_noise():
1531     return tf.eye(5) + np.random.randn(5, 5)
1532   ```
1533 
1534   will return a different output everytime it is invoked, the compiled function
1535   `compiled = tf.contrib.eager.defun(add_noise)` will return the same value
1536   every time it is called, since a particular random offset generated by NumPy
1537   will be inserted into the graph as a TensorFlow constant. The solution is to
1538   replace the call to `np.random.randn` with `tf.random_normal((5, 5))`.
1539 
1540   _Python Side-Effects_
1541   A corollary of the previous discussion on tracing is the following: If a
1542   Python function `f` has Python side-effects, then executing `f` multiple times
1543   will not necessarily be semantically equivalent to executing `F =
1544   tf.contrib.eager.defun(f)` multiple times; this difference is due to the fact
1545   that `defun` only captures the subgraph of TensorFlow operations that is
1546   constructed when `f` is called in a graph-building context.
1547 
1548   _Python Control Flow_.
1549   The structure of many machine learning computations depend upon whether one is
1550   training or validating, and it is common to nest specialized logic under `if
1551   training:` blocks. By mapping each input signature to a unique graph, `defun`
1552   lets users transparently compile such code, as the following code snippet
1553   demonstrates:
1554 
1555   ```python
1556   import tensorflow as tf
1557 
1558   tf.enable_eager_execution()
1559 
1560   @tf.contrib.eager.defun
1561   def lossy_matmul(W, x, training=True):
1562     outputs = tf.matmul(W, x)
1563     if training:
1564       outputs = tf.nn.dropout(outputs, keep_probability=0.2)
1565     return outputs
1566 
1567   W = tf.random_normal((3, 5))
1568   x = tf.random_normal((5, 1))
1569 
1570   # Executes a graph that applies dropout.
1571   lossy_outputs = lossy_matmul(W, x, training=True)
1572 
1573   # Executes a graph that does not apply dropout.
1574   exact_outputs = lossy_matmul(W, x, training=False)
1575   ```
1576 
1577   On the other hand, because `defun` generates graphs by tracing and not by
1578   source code analysis, it fully unrolls Python `for` and `while` loops,
1579   potentially creating large graphs. If your Python function has native loops
1580   that run for many iterations, consider replacing them with `tf.while_loop`
1581   operations.
1582 
1583   When constructing graphs, `tf.Tensor` objects cannot be used as Python
1584   `bool` objects. This means, for example, that you should replace code in `f`
1585   resembling
1586 
1587   ```python
1588 
1589   if tensor < 10:
1590     true_fn()
1591   else:
1592     false_fn()
1593   ```
1594 
1595   with `tf.cond(tensor < 10, true_fn, false_fn)`.
1596 
1597   _Variables_
1598   TensorFlow operations related to variable creation and initialization are
1599   automatically lifted out of the graphs generated by `defun`. In practice, this
1600   implies that variable creation and initialization only happen the first time
1601   `F` is called, and that variables are reused every time thereafter. Many
1602   TensorFlow APIs, like `tf.keras.layers.Layer` objects, create variables the
1603   first time they are called and reuse them thereafter. Automatic variable
1604   lifting makes it possible to compile these APIs without extra effort, at the
1605   cost of introducing a discrepancy between the semantics of executing Python
1606   functions and their corresponding compiled functions. For example:
1607 
1608   ```python
1609   import tensorflow as tf
1610 
1611   tf.enable_eager_execution()
1612 
1613   def fn():
1614     x = tf.Variable(0.0)
1615     x.assign_add(1.0)
1616     return x.read_value()
1617 
1618   # `fn` is a Python function, so x is created, initialized, and destroyed upon
1619   # every invocation
1620   assert fn().numpy() == fn().numpy() == 1.0
1621 
1622   compiled = tf.contrib.eager.defun(fn)
1623 
1624   # Compiling `fn` with `defun` hoists all variables outside of the generated
1625   # graph, so initialization happens exactly once.
1626   assert compiled().numpy() == 1.0
1627   assert compiled().numpy() == 2.0
1628   ```
1629 
1630   Finally, because each input signature is bound to a unique graph, if your
1631   Python function constructs `tf.Variable` objects, then each graph constructed
1632   for that Python function will reference a unique set of variables. To
1633   circumvent this problem, we recommend against compiling Python functions that
1634   create `tf.Variable` objects. Instead, Python functions should either
1635   lexically close over `tf.Variable` objects or accept them as arguments,
1636   preferably encapsulated in an object-oriented container. If you must create
1637   variables inside your Python function and you want each graph generated for it
1638   to reference the same set of variables, add logic to your Python function that
1639   ensures that variables are only created the first time it is called and are
1640   reused for every subsequent invocation; note that this is precisely what
1641   `tf.keras.layers.Layer` objects do, so we recommend using them to represent
1642   variable-bearing computations whenever possible.
1643 
1644   Args:
1645     func: function to be compiled. If `func` is None, returns a
1646       decorator that can be invoked with a single argument - `func`. The
1647       end result is equivalent to providing all the arguments up front.
1648       In other words, defun(input_signature=...)(func) is equivalent to
1649       defun(func, input_signature=...). The former allows
1650       the following use case:
1651         @tf.contrib.eager.defun(input_signature=...)
1652         def foo(...):
1653           ...
1654 
1655     input_signature: A possibly nested sequence of
1656       `tf.contrib.eager.TensorSpec` objects specifying the shapes and dtypes of
1657       the Tensors that will be supplied to this function. If `None`, a separate
1658       function is instantiated for each inferred input signature.  If a
1659       signature is specified, every input to `func` must be a `Tensor`, and
1660       `func` cannot accept `**kwargs`.
1661 
1662   Returns:
1663      If `func` is not None, returns a callable that will execute the compiled
1664      function (and return zero or more `tf.Tensor` objects).
1665      If `func` is None, returns a decorator that, when invoked with a single
1666      `func` argument, returns a callable equivalent to the case above.
1667 
1668   Raises:
1669     TypeError: If `input_signature` is neither `None` nor a sequence of
1670       `tf.contrib.eager.TensorSpec` objects.
1671   """
1672   return defun_with_attributes(func=func, input_signature=input_signature)

1732   supported (the value of the variables will never change as they will keep
1733   getting reinitialized).
1734 
1735   NOT THREAD SAFE
1736   """
1737 
1738   def __init__(self):
1739     self._returned_tensors = set()

1741   def mark_as_return(self, tensor):
1742     """Acts like identity but marks the `Tensor` as a return value.
1743 
1744     This will possibly return a copy of the `Tensor`. Usage:
1745 
1746     ```
1747       with AutomaticControlDependencies() as a:
1748        ...
1749        t = a.mark_as_return(t)
1750       _ = ...(t...)  # i.e. it's safe to use t here
1751     ```
1752 
1753     Args:
1754       tensor: the `Tensor` to be marked
1755 
1756     Returns:
1757       a copy of the `Tensor`.
1758     """
1759     if isinstance(tensor, ops.IndexedSlices):
1760       values = array_ops.identity(tensor.values)
1761       indices = array_ops.identity(tensor.indices)
1762       self._returned_tensors.add(indices)
1763       self._returned_tensors.add(values)
1764       return ops.IndexedSlices(values, indices, dense_shape=tensor.dense_shape)
1765     # We want to make the return values depend on the stateful operations, but
1766     # we don't want to introduce a cycle, so we make the return value the result
1767     # of a new identity operation that the stateful operations definitely don't
1768     # depend on.
1769     tensor = array_ops.identity(tensor)
1770     self._returned_tensors.add(tensor)
1771     return tensor

1773   def __enter__(self):
1774     if context.executing_eagerly():
1775       return self
1776     # This code assumes no other thread is adding ops to the graph while
1777     # we're adding ops to the graph.
1778     # TODO(apassos): Fix this by locking the graph or using a temporary
1779     # graph (but that would mess up devices and collections at least,
1780     # probably other things as well).
1781     self._graph = ops.get_default_graph()
1782     self._n_operations = len(self._graph.get_operations())
1783     return self

1785   def _process_switch(self, switch_op, ops_which_must_run,
1786                       last_op_using_resource_tensor, merge_for_resource):
1787     """Processes a switch node for a resource input.
1788 
1789     When tensorflow creates a cond, it creates a control flow context for each
1790     branch of the cond. Each external tensor accessed by that branch is routed
1791     through a switch op, which gets created in the graph _after_ the op which
1792     uses that tensor get created.
1793 
1794     If the resource comes from another switch op we process that one first.
1795 
1796     _process_switch creates a corresponding merge node for the switch node. This
1797     merge node is added to the outer control flow context of the switch
1798     node. We also ensure that:
1799 
1800       1. The switch node executes after the previous op which used the resource
1801          tensor
1802 
1803       2. Any op which uses a resource output of the switch node executes before
1804          the merge for the switch node.
1805 
1806       3. The next op which uses the input resource to the switch node (which
1807          might be another switch node for the other branch of the conditional)
1808          will execute after the merge node is done.
1809 
1810       4. The merge node is marked as must_run so it will run even if no
1811          subsequent operation uses the resource.
1812 
1813     Args:
1814       switch_op: the switch op to be processed
1815       ops_which_must_run: the set of ops which must run
1816       last_op_using_resource_tensor: map from resource tensor to last op using
1817         it
1818       merge_for_resource: map from resource tensor to merge which must follow
1819         all usages of it.
1820     """
1821     inp = switch_op.inputs[0]
1822     if inp.dtype == dtypes_module.resource and inp.op.type == "Switch":
1823       self._process_switch(inp.op, ops_which_must_run,
1824                            last_op_using_resource_tensor, merge_for_resource)
1825     if switch_op.outputs[0] in merge_for_resource:
1826       return
1827     new_merge = control_flow_ops.merge(switch_op.outputs,
1828                                        name="artificial_merge")
1829     new_merge[0].op._control_flow_context = (  # pylint: disable=protected-access
1830         switch_op._control_flow_context.outer_context)  # pylint: disable=protected-access
1831     # Ensures the merge always runs
1832     ops_which_must_run.add(new_merge[0].op)
1833     if inp in last_op_using_resource_tensor:
1834       # Ensures the switch executes after the previous op using the resource.
1835       switch_op._add_control_input(last_op_using_resource_tensor[inp])  # pylint: disable=protected-access
1836     # Ensure the next op outside the cond happens after the merge.
1837     last_op_using_resource_tensor[inp] = new_merge[0].op
1838     if inp in merge_for_resource:
1839       merge_for_resource[inp]._add_control_input(new_merge[0].op)  # pylint: disable=protected-access
1840     for o in switch_op.outputs:
1841       # Ensures the merge will execute after all ops inside the cond
1842       merge_for_resource[o] = new_merge[0].op
1843 
1844   def __exit__(self, unused_type, unused_value, unused_traceback):
1845     if context.executing_eagerly():
1846       return
1847 
1848     if self._graph is not ops.get_default_graph():
1849       raise RuntimeError(
1850           "Graph changed while trying to add control dependencies.")
1851 
1852     # map from resource tensor to the last op which used it
1853     last_op_using_resource_tensor = {}
1854     # set of conditional and loop exits
1855     ops_which_must_run = set()
1856     # merge which must depend on ops which use this resource
1857     merge_for_resource = {}
1858 
1859     new_operations = self._graph.get_operations()[self._n_operations:]
1860 
1861     # Ensures that uses of resource tensors get serialized properly and all
1862     # execute. This is done by keeping a map from resource tensor to the last op
1863     # in graph-construction order which used it (last_op_using_resource_tensor).
1864     #
1865     # Conditionals are written in TensorFlow such that every external tensor
1866     # accessed in the conditional goes through a switch op and every return
1867     # tensor (it's guaranteed that there will be at least one) goes through a
1868     # merge op.
1869     #
1870     # To handle conditionals, switches are handled in a special way (see
1871     # comments for _process_switch). Merge nodes created by TF's conditional
1872     # logic (as opposed to by _process_switch) are forced to run and also get a
1873     # control dependency added to them to ensure all stateful ops inside their
1874     # control flow context run.
1875     #
1876     # We also ensure that if an op is using a resource output by a switch node
1877     # (that is, a resource tensor for which there's a value in
1878     # merge_for_resource) this op will run before the merge for that resource.
1879     #
1880     # We try to add control inputs to nodes respecting their control flow
1881     # contexts to avoid dead nodes propagating everywhere and leading to
1882     # "retval[0] doesn't have value" errors. If a node gets a control dependency
1883     # on a dead node (i.e. a note from an untaken control flow branch) that node
1884     # will be marked as dead unless it's a merge node.
1885     #
1886     # TODO(apassos): serialize non-resource-taking stateful ops as well, and
1887     # test that it works. Support while loops. Support init_scope escaping from
1888     # this.
1889     for op in new_operations:
1890       # TODO(apassos) make this code safely support while loops.
1891       if isinstance(op._control_flow_context, control_flow_ops.WhileContext):  # pylint: disable=protected-access
1892         continue
1893       control_inputs = set()
1894       # Ensure stateful ops run
1895       if (op.type not in self._graph._registered_ops  # pylint: disable=protected-access
1896           or self._graph._registered_ops[op.type].is_stateful):  # pylint: disable=protected-access
1897         ops_which_must_run.add(op)
1898       # Ignore switches (they're handled separately)
1899       if op.type == "Switch" and op.inputs[0].dtype == dtypes_module.resource:
1900         continue
1901       # Make merges trigger all other computation which must run
1902       if op.type == "Merge":
1903         for o in ops_which_must_run:
1904           op._add_control_input(o)  # pylint: disable=protected-access
1905           for inp in o.inputs:
1906             if inp in last_op_using_resource_tensor:
1907               last_op_using_resource_tensor[inp] = op
1908         ops_which_must_run = set([op])
1909         continue
1910       for inp in op.inputs:
1911         if inp.dtype == dtypes_module.resource:
1912           # Deal with switches, finally.
1913           if inp.op.type == "Switch":
1914             self._process_switch(inp.op, ops_which_must_run,
1915                                  last_op_using_resource_tensor,
1916                                  merge_for_resource)
1917           # Ensure uses of resources are serialized
1918           if inp in last_op_using_resource_tensor:
1919             if (last_op_using_resource_tensor[inp]._control_flow_context  # pylint: disable=protected-access
1920                 is op._control_flow_context):  # pylint: disable=protected-access
1921               control_inputs.add(last_op_using_resource_tensor[inp])
1922           # Ensure merges happen after the closing of a cond block
1923           if inp in merge_for_resource:
1924             merge_for_resource[inp]._add_control_input(op)  # pylint: disable=protected-access
1925           last_op_using_resource_tensor[inp] = op
1926       control_inputs = [c for c in control_inputs
1927                         if c._control_flow_context is op._control_flow_context]  # pylint: disable=protected-access
1928       op._add_control_inputs(control_inputs)  # pylint: disable=protected-access
1929 
1930     # Ensure all ops which must run do run
1931     for r in self._returned_tensors:
1932       if ops_which_must_run:
1933         r.op._add_control_inputs(  # pylint: disable=protected-access
1934             [o for o in ops_which_must_run
1935              if o._control_flow_context is r.op._control_flow_context])  # pylint: disable=protected-access

1938 def automatic_control_dependencies(f):
1939   """Wraps f to automatically insert control dependencies.
1940 
1941   The inserted dependencies ensure that:
1942     1. All stateful ops in f run when the result of f runs
1943     2. Updates to the same resources happen in order.
1944 
1945   Args:
1946     f: the function to be wrapped.
1947 
1948   Returns:
1949     The wrapped function.
1950   """
1951 
1952   def wrapper(*args, **kwargs):
1953     with AutomaticControlDependencies() as a:
1954       result = f(*args, **kwargs)
1955       result_flat = [a.mark_as_return(t) for t in nest.flatten(result)]
1956       return nest.pack_sequence_as(result, result_flat)
1957 
1958   return tf_decorator.make_decorator(f, wrapper)
