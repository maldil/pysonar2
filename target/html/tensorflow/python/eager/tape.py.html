  15 """Gradient tape utilites."""

  17 from __future__ import absolute_import
  18 from __future__ import division
  19 from __future__ import print_function
  20 
  21 import contextlib
  22 
  23 from tensorflow.python import pywrap_tensorflow
  24 
  25 
  26 class Tape(object):
  27   """Represents a gradient propagation trace."""
  28 
  29   def __init__(self, tape):
  30     self._tape = tape
  31 
  32   def watched_variables(self):
  33     return pywrap_tensorflow.TFE_Py_TapeWatchedVariables(self._tape)
  34 
  35 
  36 def push_new_tape(persistent=False, watch_accessed_variables=True):
  37   """Pushes a new tape onto the tape stack."""
  38   tape = pywrap_tensorflow.TFE_Py_TapeSetNew(persistent,
  39                                              watch_accessed_variables)
  40   return Tape(tape)
  41 
  42 
  43 def push_tape(tape):
  44   """Pushes an existing tape onto the tape stack."""
  45   pywrap_tensorflow.TFE_Py_TapeSetAdd(tape._tape)  # pylint: disable=protected-access
  46 
  47 
  48 def watch(tape, tensor):
  49   """Marks this tensor to be watched by the given tape."""
  50   pywrap_tensorflow.TFE_Py_TapeWatch(tape._tape, tensor)  # pylint: disable=protected-access
  51 
  52 
  53 def watch_variable(tape, variable):
  54   """Marks this variable to be watched by the given tape."""
  55   pywrap_tensorflow.TFE_Py_TapeWatchVariable(tape._tape, variable)  # pylint: disable=protected-access
  56 
  57 
  58 def variable_accessed(variable):
  59   """Notifies all tapes in the stack that a variable has been accessed."""
  60   pywrap_tensorflow.TFE_Py_TapeVariableAccessed(variable)
  61 
  62 
  63 def pop_tape(tape):
  64   """Pops the top tape in the stack, if any."""
  65   pywrap_tensorflow.TFE_Py_TapeSetRemove(tape._tape)  # pylint: disable=protected-access
  66 
  67 
  68 @contextlib.contextmanager
  69 def stop_recording():
  70   try:
  71     pywrap_tensorflow.TFE_Py_TapeSetStopOnThread()
  72     yield
  73   finally:
  74     pywrap_tensorflow.TFE_Py_TapeSetRestartOnThread()
  75 
  76 
  77 def should_record(tensors):
  78   """Returns true if any tape in the stack watches any of these tensors."""
  79   return pywrap_tensorflow.TFE_Py_TapeSetShouldRecord(tensors)
  80 
  81 
  82 def record_operation(op_type, output_tensors, input_tensors, backward_function):
  83   """Records the operation on all tapes in the stack."""
  84   pywrap_tensorflow.TFE_Py_TapeSetRecordOperation(
  85       op_type, output_tensors, input_tensors, backward_function)
  86 
  87 
  88 def delete_trace(tensor_id):
  89   """Deletes traces for this Tensor from all tapes in the stack."""
  90   pywrap_tensorflow.TFE_Py_TapeSetDeleteTrace(tensor_id)
  91 
  92 
  93 def could_possibly_record():
  94   """Returns True if any tape is active."""
  95   return not pywrap_tensorflow.TFE_Py_TapeSetIsEmpty()
