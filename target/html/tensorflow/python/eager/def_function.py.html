  31 class UnliftedInitializerVariable(resource_variable_ops.ResourceVariable):
  32   """Variable which does not lift its initializer out of function context.
  33 
  34   Instances of this variable, when created, build a graph which runs their
  35   initializer inside a tf.cond(is_initialized) block.
  36 
  37   This can only be created inside a defun called from (eventually) eager
  38   mode. That is, non-function-building graphs are not supported.
  39   """
  40 
  41   def __init__(self,  # pylint: disable=super-init-not-called
  42                initial_value=None,
  43                trainable=True,
  44                caching_device=None,
  45                name=None,
  46                dtype=None,
  47                constraint=None,
  48                **unused_kwargs):
  49     """Creates a variable.
  50 
  51     Args:
  52       initial_value: A `Tensor`, or Python object convertible to a `Tensor`,
  53         which is the initial value for the Variable. The initial value must have
  54         a shape specified unless `validate_shape` is set to False. Can also be a
  55         callable with no argument that returns the initial value when called.
  56         (Note that initializer functions from init_ops.py must first be bound
  57          to a shape before being used here.)
  58       trainable: If `True`, GradientTapes automatically watch uses of this
  59         Variable.
  60       caching_device: Optional device string or function describing where the
  61         Variable should be cached for reading.  Defaults to the Variable's
  62         device.  If not `None`, caches on another device.  Typical use is to
  63         cache on the device where the Ops using the Variable reside, to
  64         deduplicate copying through `Switch` and other conditional statements.
  65       name: Optional name for the variable. Defaults to `'Variable'` and gets
  66         uniquified automatically.
  67       dtype: If set, initial_value will be converted to the given type.
  68         If None, either the datatype will be kept (if initial_value is
  69        a Tensor) or float32 will be used (if it is a Python object convertible
  70        to a Tensor).
  71       constraint: An optional projection function to be applied to the variable
  72         after being updated by an `Optimizer` (e.g. used to implement norm
  73         constraints or value constraints for layer weights). The function must
  74         take as input the unprojected Tensor representing the value of the
  75         variable and return the Tensor for the projected value
  76         (which must have the same shape). Constraints are not safe to
  77         use when doing asynchronous distributed training.
  78 
  79     Raises:
  80       ValueError: If the initial value is not specified, or does not have a
  81         shape and `validate_shape` is `True`.
  82       RuntimeError: If called outside of a function definition.
  83     """
  84     if context.executing_eagerly():
  85       raise RuntimeError(
  86           "UnliftedInitializerVariable should not be created "
  87           "outside of functions.")
  88     with ops.init_scope():
  89       if not context.executing_eagerly():
  90         raise RuntimeError(
  91             "UnliftedInitializerVariable does not support legacy graph mode.")
  92     self._in_graph_mode = False
  93     if initial_value is None:
  94       raise ValueError("initial_value must be specified.")
  95     init_from_fn = callable(initial_value)
  96 
  97     if constraint is not None and not callable(constraint):
  98       raise ValueError("The `constraint` argument must be a callable.")
  99 
 100     if isinstance(initial_value, checkpointable.CheckpointInitialValue):
 101       self._maybe_initialize_checkpointable()
 102       self._update_uid = initial_value.checkpoint_position.restore_uid
 103       initial_value = initial_value.wrapped_value
 104 
 105     self._trainable = trainable
 106     self._save_slice_info = None
 107     self._initial_value = None
 108     self._initializer_op = None
 109     self._is_initialized_op = None
 110     self._graph_element = None
 111     self._cached_value = None
 112     # Store the graph key so optimizers know how to only retrieve variables from
 113     # this graph. Guaranteed to be the same as the eager graph_key.
 114     self._graph_key = ops.get_default_graph()._graph_key  # pylint: disable=protected-access
 115     with ops.name_scope(name, "Variable", []
 116                         if init_from_fn else [initial_value]) as name:
 117       # pylint: disable=protected-access
 118       with ops.init_scope():
 119         assert context.executing_eagerly()
 120         shared_name = ops._name_from_scope_name(name)
 121         shared_name = "%s_%d" % (shared_name, ops.uid())
 122       # Use attr_scope and device(None) to simulate the behavior of
 123       # colocate_with when the variable we want to colocate with doesn't
 124       # yet exist.
 125       with ops.name_scope("Initializer"), ops.device(None):
 126         initial_value = ops.convert_to_tensor(
 127             initial_value() if init_from_fn else initial_value,
 128             name="initial_value", dtype=dtype)
 129       with ops.init_scope():
 130         self._handle = resource_variable_ops.eager_safe_variable_handle(
 131             shape=initial_value.get_shape(),
 132             dtype=initial_value.dtype.base_dtype,
 133             shared_name=shared_name,
 134             name=name,
 135             graph_mode=False)
 136       self._shape = initial_value.shape
 137       self._unique_id = shared_name
 138       self._handle_name = shared_name + ":0"
 139       self._dtype = initial_value.dtype.base_dtype
 140       self._constraint = constraint
 141       assert initial_value is not None
 142       def assign_fn():
 143         with ops.name_scope("Assign") as n, ops.colocate_with(self._handle):
 144           resource_variable_ops.assign_variable_op(
 145               self._handle,
 146               initial_value,
 147               name=n)
 148         # Returning values to keep tf.cond happy.
 149         return ops.convert_to_tensor(1)
 150       def not_assign_fn():
 151         return ops.convert_to_tensor(0)
 152       # Note: this cond is always guaranteed to run because we're inside a defun
 153       # which will insert automatic control dependencies.
 154       control_flow_ops.cond(
 155           resource_variable_ops.var_is_initialized_op(self._handle),
 156           not_assign_fn, assign_fn)
 157 
 158     # After the handle has been created, set up a way to clean it up when
 159     # executing eagerly. We'll hold the only reference to the deleter, so that
 160     # when this object is garbage collected the deleter will be too. This
 161     # means ResourceVariables can be part of reference cycles without those
 162     # cycles being uncollectable.
 163     self._handle_deleter = resource_variable_ops.EagerResourceDeleter(
 164         handle=self._handle, handle_device=self._handle.device)
 165     self._cached_shape_as_list = None
 166

 168 def _defun_with_scope(scope, fn):
 169 
 170   def wrapped_fn(*args, **kwds):
 171     with variable_scope.variable_creator_scope(scope):
 172       return fn(*args, **kwds)
 173 
 174   return function.defun(wrapped_fn)

 177 def def_function(fn):
 178   """Defines a function as per the "functions, not sessions" document."""
 179 
 180   # Wrapping the values in lists to bypass python's lack of way to mutate
 181   # symbols from an outer scope.
 182   first_call = [True]
 183   function_to_call = []
 184 
 185   # TODO(apassos) represent this as an object and not as a closure.
 186   def decorated_fn(*args, **kwds):
 187     """Graph function for fn."""
 188     if not first_call[0]:
 189       return function_to_call[0](*args, **kwds)
 190 
 191     first_call[0] = False
 192     created_variables = []
 193 
 194     def variable_creator_scope(unused_next_creator, **kwds):
 195       """Creates UnliftedInitializerVariables and saves references to them."""
 196       v = UnliftedInitializerVariable(**kwds)
 197       created_variables.append(v)
 198       return v
 199 
 200     first_graph_function = _defun_with_scope(variable_creator_scope, fn)
 201 
 202     # Force the definition of the function for these arguments
 203     first_concrete = first_graph_function.get_concrete_function(*args, **kwds)
 204 
 205     def invalid_creator_scope(*unused_args, **unused_kwds):
 206       """Disables variable creation."""
 207       raise ValueError(
 208           "def_function-decorated function tried to create "
 209           "variables on second call.")
 210 
 211     second_graph_function = _defun_with_scope(invalid_creator_scope, fn)
 212 
 213     function_to_call.append(second_graph_function)
 214     if not created_variables:
 215       # Note: this retracing might be unnecessary, but running the function
 216       # forever in the scope which disallows variable creation is safer than not
 217       # doing so.
 218       return second_graph_function(*args, **kwds)
 219 
 220     def fn_with_cond(*inner_args, **inner_kwds):
 221       """Conditionally runs initialization if it's needed."""
 222       condition = True
 223       for variable in created_variables:
 224         condition = condition and resource_variable_ops.var_is_initialized_op(
 225             variable.handle)
 226       # We want to call second_graph_function if possible because it avoids
 227       # recomputing potentially expensive initializers.
 228       return control_flow_ops.cond(
 229           condition,
 230           lambda: second_graph_function(*inner_args, **inner_kwds),
 231           lambda: first_concrete(*inner_args, **inner_kwds))
 232 
 233     return function.defun(fn_with_cond)(*args, **kwds)
 234 
 235   return decorated_fn
