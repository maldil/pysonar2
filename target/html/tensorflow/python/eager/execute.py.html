  33 def quick_execute(op_name, num_outputs, inputs, attrs, ctx, name=None):
  34   """Execute a TensorFlow operation.
  35 
  36   Args:
  37     op_name: Name of the TensorFlow operation (see REGISTER_OP in C++ code) to
  38       execute.
  39     num_outputs: The number of outputs of the operation to fetch.
  40                  (Explicitly provided instead of being inferred for performance
  41                  reasons).
  42     inputs: A list of inputs to the operation. Each entry should be a Tensor, or
  43       a value which can be passed to the Tensor constructor to create one.
  44     attrs: A tuple with alternating string attr names and attr values for this
  45       operation.
  46     ctx: The value of context.context().
  47     name: Customized name for the operation.
  48 
  49   Returns:
  50     List of output Tensor objects. The list is empty if there are no outputs
  51 
  52   Raises:
  53     An exception on error.
  54   """
  55   device_name = ctx.device_name
  56   # pylint: disable=protected-access
  57   try:
  58     tensors = pywrap_tensorflow.TFE_Py_Execute(ctx._handle, device_name,
  59                                                op_name, inputs, attrs,
  60                                                num_outputs)
  61   except core._NotOkStatusException as e:
  62     if name is not None:
  63       message = e.message + " name: " + name
  64     else:
  65       message = e.message
  66     six.raise_from(core._status_to_exception(e.code, message), None)
  67   # pylint: enable=protected-access
  68   return tensors

  71 def execute_with_callbacks(op_name, num_outputs, inputs, attrs, ctx, name=None):
  72   """Monkey-patch to execute to enable execution callbacks."""
  73   tensors = quick_execute(op_name, num_outputs, inputs, attrs, ctx, name)
  74   for callback in ctx.post_execution_callbacks:
  75     callback(op_name, inputs, attrs, tensors, name)
  76 
  77   return tensors

  89 def make_float(v, arg_name):
  90   if not isinstance(v, compat.real_types):
  91     raise TypeError("Expected float for argument '%s' not %s." %
  92                     (arg_name, repr(v)))
  93   return float(v)

  96 def make_int(v, arg_name):
  97   if isinstance(v, six.string_types):
  98     raise TypeError("Expected int for argument '%s' not %s." %
  99                     (arg_name, repr(v)))
 100   try:
 101     return int(v)
 102   except (ValueError, TypeError):
 103     raise TypeError("Expected int for argument '%s' not %s." %
 104                     (arg_name, repr(v)))

 107 def make_str(v, arg_name):
 108   if not isinstance(v, compat.bytes_or_text_types):
 109     raise TypeError("Expected string for argument '%s' not %s." %
 110                     (arg_name, repr(v)))
 111   return compat.as_bytes(v)  # Convert unicode strings to bytes.

 117                     (arg_name, repr(v)))

 121 def make_type(v, arg_name):
 122   try:
 123     v = dtypes.as_dtype(v).base_dtype
 124   except TypeError:
 125     raise TypeError("Expected DataType for argument '%s' not %s." %
 126                     (arg_name, repr(v)))
 127   i = v.as_datatype_enum
 128   return i

 131 def make_shape(v, arg_name):
 132   """Convert v into a list."""
 133   # Args:
 134   #   v: A TensorShapeProto, a list of ints, or a tensor_shape.TensorShape.
 135   #   arg_name: String, for error messages.
 136 
 137   # Returns:
 138   #   None if the rank is unknown, otherwise a list of ints (or Nones in the
 139   #   position where the dimension is unknown).
 140   try:
 141     shape = tensor_shape.as_shape(v)
 142   except TypeError as e:
 143     raise TypeError("Error converting %s to a TensorShape: %s." % (arg_name, e))
 144   except ValueError as e:
 145     raise ValueError("Error converting %s to a TensorShape: %s." % (arg_name,
 146                                                                     e))
 147   if shape.ndims is None:
 148     return None
 149   else:
 150     return shape.as_list()

 153 def make_tensor(v, arg_name):
 154   """Ensure v is a TensorProto."""
 155   if isinstance(v, tensor_pb2.TensorProto):
 156     return v
 157   elif isinstance(v, six.string_types):
 158     pb = tensor_pb2.TensorProto()
 159     text_format.Merge(v, pb)
 160     return pb
 161   raise TypeError(
 162       "Don't know how to convert %s to a TensorProto for argument '%s'." %
 163       (repr(v), arg_name))

 166 def args_to_matching_eager(l, ctx, default_dtype=None):
 167   """Convert sequence `l` to eager same-type Tensors."""
 168   EagerTensor = ops.EagerTensor  # pylint: disable=invalid-name
 169   for x in l:
 170     if not isinstance(x, EagerTensor):
 171       break
 172   else:  # note: intentional for-else
 173     return l[0]._datatype_enum(), l  # pylint: disable=protected-access
 174   # TODO(josh11b): Could we do a better job if we also passed in the
 175   # allowed dtypes when that was known?
 176 
 177   # Is some input already a Tensor with a dtype?
 178   dtype = None
 179   for t in l:
 180     if isinstance(t, EagerTensor):
 181       dtype = t.dtype
 182       break
 183 
 184   internal_convert_to_tensor = ops.internal_convert_to_tensor
 185   if dtype is None:
 186     # Infer a dtype based on the first value, and use that dtype for the
 187     # remaining values.
 188     ret = []
 189     for t in l:
 190       ret.append(internal_convert_to_tensor(
 191           t, dtype, preferred_dtype=default_dtype, ctx=ctx))
 192       if dtype is None:
 193         dtype = ret[-1].dtype
 194   else:
 195     ret = [internal_convert_to_tensor(t, dtype, ctx=ctx) for t in l]
 196 
 197   return dtype.as_datatype_enum, ret

 200 def convert_to_mixed_eager_tensors(values, ctx):
 201   v = [ops.internal_convert_to_tensor(t, ctx=ctx) for t in values]
 202   types = [t._datatype_enum() for t in v]  # pylint: disable=protected-access
 203   return types, v

 206 def args_to_mixed_eager_tensors(lists, ctx):
 207   """Converts a list of same-length lists of values to eager tensors."""
 208   assert len(lists) > 1
 209 
 210   # Generate an error if len(lists[i]) is not the same for all i.
 211   lists_ret = []
 212   for l in lists[1:]:
 213     if len(l) != len(lists[0]):
 214       raise ValueError(
 215           "Expected list arguments to be the same length: %d != %d (%r vs. %r)."
 216           % (len(lists[0]), len(l), lists[0], l))
 217     lists_ret.append([])
 218 
 219   # Convert the first element of each list first, then the second element, etc.
 220   types = []
 221   for i in range(len(lists[0])):
 222     dtype = None
 223     # If any list has a Tensor, use that dtype
 224     for l in lists:
 225       if isinstance(l[i], ops.EagerTensor):
 226         dtype = l[i].dtype
 227         break
 228     if dtype is None:
 229       # Convert the first one and use its dtype.
 230       lists_ret[0].append(ops.internal_convert_to_tensor(lists[0][i], ctx=ctx))
 231       dtype = lists_ret[0][i].dtype
 232       for j in range(1, len(lists)):
 233         lists_ret[j].append(
 234             ops.internal_convert_to_tensor(lists[j][i], dtype=dtype, ctx=ctx))
 235     else:
 236       # Convert everything to the found dtype.
 237       for j in range(len(lists)):
 238         lists_ret[j].append(
 239             ops.internal_convert_to_tensor(lists[j][i], dtype=dtype, ctx=ctx))
 240     types.append(dtype.as_datatype_enum)
 241   return types, lists_ret
