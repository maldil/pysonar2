  59 class _EagerTensorCache(object):
  60   """Simple cache which evicts items based on length in a FIFO manner."""
  61 
  62   def __init__(self, max_items=256, max_tensor_size=10000):
  63     self._data = collections.OrderedDict()
  64     self._max_items = max_items
  65     self._max_tensor_size = max_tensor_size
  66 
  67   def put(self, key, value):
  68     if value._num_elements() > self._max_tensor_size:  # pylint: disable=protected-access
  69       return
  70 
  71     self._data[key] = value
  72 
  73     if len(self._data) > self._max_items:
  74       self._data.popitem(last=False)
  75 
  76   def get(self, key):
  77     return self._data.get(key, None)
  78 
  79   def flush(self):
  80     self._data = {}

 106 # `_ContextSwitchStack` is a `threading.local` to match the semantics of
 107 # ``DefaultGraphStack`, which is also a `threading.local`.
 108 class _ContextSwitchStack(threading.local):
 109   """A thread-local stack of context switches."""
 110 
 111   def __init__(self, eager):
 112     super(_ContextSwitchStack, self).__init__()
 113     self.stack = []
 114     if eager:
 115       # Initialize the stack with a pointer to enter the eager context; this
 116       # ensures that the fact that eager execution was enabled is propagated
 117       # across threads, since (1) `enable_eager_execution` modifies a
 118       # process-level flag (`default_execution_mode`) and (2) `__init__` is
 119       # called each time a threading.local object is used in a separate thread.
 120       self.push(is_building_function=False, enter_context_fn=eager_mode)
 121 
 122   def push(self, is_building_function, enter_context_fn):
 123     """Push metadata about a context switch onto the stack.
 124 
 125     A context switch can take one of two forms: installing a graph as the
 126     default graph, or entering the eager context. For each context switch,
 127     we record whether or not the entered context is building a function.
 128 
 129     Args:
 130       is_building_function: (bool.) Whether the context is building a function.
 131       enter_context_fn: (function.) A callable that executes the context switch.
 132         For example, `graph.as_default` or `eager_mode`.
 133     """
 134 
 135     self.stack.append(
 136         ContextSwitch(is_building_function, enter_context_fn))
 137 
 138   def pop(self):
 139     """Pop the stack."""
 140 
 141     self.stack.pop()

 149   # TODO(agarwal): create and link in some documentation for `execution_mode`.
 150   # pylint: disable=redefined-outer-name
 151   def __init__(self,
 152                config=None,
 153                device_policy=None,
 154                execution_mode=None,
 155                server_def=None):
 156     """Creates a new Context.
 157 
 158     Args:
 159       config: (Optional.) A `ConfigProto` protocol buffer with configuration
 160         options for the Context. Note that a lot of these options may be
 161         currently unimplemented or irrelevant when eager execution is enabled.
 162       device_policy: (Optional.) What policy to use when trying to run an
 163          operation on a device with inputs which are not on that device.
 164          When set to None, an appropriate value will be picked automatically.
 165          The value picked may change between TensorFlow releases.
 166 
 167          Defaults to tf.contrib.eager.DEVICE_PLACEMENT_SILENT_FOR_INT32.
 168          Valid values:
 169          - tfe.DEVICE_PLACEMENT_EXPLICIT: raises an error if the placement is
 170            not correct.
 171          - tfe.DEVICE_PLACEMENT_WARN: copies the tensors which are not on the
 172            right device but raises a warning.
 173          - tfe.DEVICE_PLACEMENT_SILENT: silently copies the tensors. This might
 174            hide performance problems.
 175          - tfe.DEVICE_PLACEMENT_SILENT_FOR_INT32: silently copies int32 tensors,
 176            raising errors on the other ones.
 177       execution_mode: (Optional.) Policy controlling how operations dispatched
 178         are actually executed. When set to None, an appropriate value will be
 179         picked automatically. The value picked may change between TensorFlow
 180         releases.
 181         Valid values:
 182         - tf.contrib.eager.SYNC: executes each operation synchronously.
 183         - tf.contrib.eager.ASYNC: executes each operation asynchronously. These
 184           operations may return "non-ready" handles.
 185       server_def: (Optional.) A tensorflow::ServerDef proto.
 186         Enables execution on remote devices. GrpcServers need to be started by
 187         creating an identical server_def to this, and setting the appropriate
 188         task_indexes, so that the servers can communicate. It will then be
 189         possible to execute operations on remote devices.
 190 
 191     Raises:
 192      ValueError: If execution_mode is not valid.
 193     """
 194     self._eager_context = _EagerContext()
 195     self._context_switches = _ContextSwitchStack(self.executing_eagerly())
 196     self._context_handle = None
 197     self._context_devices = None
 198     self._post_execution_callbacks = []
 199     self._config = config
 200     self._seed = None
 201     self._initialize_lock = threading.Lock()
 202     self._device_policy = device_policy
 203     if execution_mode not in (None, SYNC, ASYNC):
 204       raise ValueError(
 205           "execution_mode should be None/SYNC/ASYNC. Got %s" % execution_mode)
 206     if execution_mode is None:
 207       execution_mode = SYNC
 208     self._execution_mode = execution_mode
 209     self._server_def = server_def

 213   def _set_global_seed(self, seed):
 214     """Set a global eager mode seed for random ops."""
 215     self._seed = seed
 216     self._rng = random.Random(self._seed)
 217     # Also clear the kernel cache, to reset any existing seeds
 218     if self._context_handle is not None:
 219       pywrap_tensorflow.TFE_ContextClearCaches(self._context_handle)

 221   def _internal_operation_seed(self):
 222     """Returns a fake operation seed.
 223 
 224       In eager mode, user shouldn't set or depend on operation seed.
 225       Here, we generate a random seed based on global seed to make
 226       operation's randomness different and depend on the global seed.
 227 
 228     Returns:
 229       A fake operation seed based on global seed.
 230     """
 231     return self._rng.randint(0, _MAXINT32)

 233   def _initialize_devices(self):
 234     """Helper to initialize devices."""
 235     # Store list of devices
 236     self._context_devices = []
 237     device_list = pywrap_tensorflow.TFE_ContextListDevices(
 238         self._context_handle)
 239     try:
 240       self._num_gpus = 0
 241       for i in range(pywrap_tensorflow.TF_DeviceListCount(device_list)):
 242         dev_name = pywrap_tensorflow.TF_DeviceListName(device_list, i)
 243         self._context_devices.append(pydev.canonical_name(dev_name))
 244         dev_type = pywrap_tensorflow.TF_DeviceListType(device_list, i)
 245         if dev_type == "GPU":
 246           self._num_gpus += 1
 247 
 248     finally:
 249       pywrap_tensorflow.TF_DeleteDeviceList(device_list)
 250 
 251   def _initialize_handle_and_devices(self):
 252     """Initialize handle and devices."""
 253     with self._initialize_lock:
 254       if self._context_handle is not None:
 255         return
 256       assert self._context_devices is None
 257       opts = pywrap_tensorflow.TFE_NewContextOptions()
 258       try:
 259         if self._config is not None:
 260           config_str = self._config.SerializeToString()
 261           pywrap_tensorflow.TFE_ContextOptionsSetConfig(opts, config_str)
 262         if self._device_policy is not None:
 263           pywrap_tensorflow.TFE_ContextOptionsSetDevicePlacementPolicy(
 264               opts, self._device_policy)
 265         if self._execution_mode == ASYNC:
 266           pywrap_tensorflow.TFE_ContextOptionsSetAsync(opts, True)
 267         self._context_handle = pywrap_tensorflow.TFE_NewContext(opts)
 268       finally:
 269         pywrap_tensorflow.TFE_DeleteContextOptions(opts)
 270       if self._server_def is not None:
 271         server_def_str = self._server_def.SerializeToString()
 272         pywrap_tensorflow.TFE_ContextSetServerDef(self._context_handle, 600,
 273                                                   server_def_str)
 274 
 275       self._initialize_devices()

 277   def _clear_caches(self):
 278     self.scalar_cache().clear()

 282   def set_server_def(self, server_def, keep_alive_secs=600):
 283     """Allow setting a server_def on the context.
 284 
 285     When a server def is replaced, it effectively clears a bunch of caches
 286     within the context. If you attempt to use a tensor object that was pointing
 287     to a tensor on the remote device, it will raise an error.
 288 
 289     Args:
 290       server_def: A tensorflow::ServerDef proto.
 291         Enables execution on remote devices.
 292       keep_alive_secs: Num. seconds after which the remote end will hang up.
 293         As long as the client is still alive, the server state for the context
 294         will be kept alive. If the client is killed (or there is some failure),
 295         the server will clean up its context keep_alive_secs after the final RPC
 296         it receives.
 297 
 298     Raises:
 299       ValueError: if server_def is None.
 300     """
 301     if not server_def:
 302       raise ValueError("server_def is None.")
 303     if not self._context_handle:
 304       self._server_def = server_def
 305     else:
 306       server_def_str = server_def.SerializeToString()
 307       pywrap_tensorflow.TFE_ContextSetServerDef(self._context_handle,
 308                                                 keep_alive_secs, server_def_str)
 309 
 310       # Clear all the caches in case there are remote tensors in them.
 311       self._clear_caches()
 312 
 313       self._initialize_devices()
 314

 334     if self._context_handle is None:
 335       return "Eager TensorFlow Context. Devices currently uninitialized."

 343   @tf_contextlib.contextmanager
 344   def _mode(self, mode):
 345     """A context manager to allow setting the mode to EAGER/GRAPH."""
 346     ctx = self._eager_context
 347     old_mode = ctx.mode
 348     old_is_eager = ctx.is_eager
 349     ctx.mode = mode
 350     ctx.is_eager = mode == EAGER_MODE
 351     if mode == EAGER_MODE:
 352       # Entering graph mode does not provide us with sufficient information to
 353       # record a context switch; graph-based context switches are only logged
 354       # when a graph is registered as the default graph.
 355       self.context_switches.push(False, eager_mode)
 356     try:
 357       yield
 358     finally:
 359       ctx.is_eager = old_is_eager
 360       ctx.mode = old_mode
 361       if mode == EAGER_MODE:
 362         self.context_switches.pop()

 410   @tf_contextlib.contextmanager
 411   def device(self, name):
 412     """Context-manager to force placement of operations and Tensors on a device.
 413 
 414     Args:
 415       name: Name of the device or None to get default placement.
 416 
 417     Yields:
 418       Nothing.
 419 
 420     Raises:
 421       ValueError: If name is not a string or is an invalid device name.
 422     """
 423     eager_context = self._eager_context
 424     old_device_name = eager_context.device_name
 425     old_device_spec = eager_context.device_spec
 426     cache_key = (old_device_name, name)
 427     try:
 428       new_device_name, new_device_spec = _device_parsing_cache[cache_key]
 429     except TypeError:
 430       # Error while trying to compute the cache key.
 431       raise ValueError("Expecting a string device name. Got %s(%s)" %
 432                        (type(name), name))
 433     except KeyError:
 434       # Handle a cache miss.
 435       if name is not None:
 436         if not isinstance(name, str):
 437           raise ValueError("Expecting a string device name. Got %s(%s)" %
 438                            (type(name), name))
 439         device_spec = pydev.DeviceSpec.from_string(name)
 440         if old_device_name:
 441           new_device_spec = copy.copy(old_device_spec)
 442         else:
 443           new_device_spec = pydev.DeviceSpec.from_string(
 444               "/job:localhost/replica:0/task:0/device:CPU:0")
 445         new_device_spec.merge_from(device_spec)
 446       else:
 447         new_device_spec = pydev.DeviceSpec.from_string("")
 448       new_device_name = new_device_spec.to_string()
 449       _device_parsing_cache[cache_key] = (new_device_name, new_device_spec)
 450 
 451     try:
 452       eager_context.device_name = new_device_name
 453       eager_context.device_spec = new_device_spec
 454       yield
 455     finally:
 456       eager_context.device_name = old_device_name
 457       eager_context.device_spec = old_device_spec

 526   def add_post_execution_callback(self, callback):
 527     """Add a post-execution callback to the context.
 528 
 529     A post-execution callback is invoked immediately after an eager operation or
 530     function has finished execution, providing access to the op's type, name
 531     input and output tensors. Multiple execution callbacks can be added, in
 532     which case the callbacks will be invoked in the order in which they are
 533     added.
 534 
 535     Args:
 536       callback: a callable of the signature
 537       `f(op_type, op_name, attrs, inputs, outputs)`.
 538       `op_type` is the type of the operation that was just executed (e.g.,
 539         `MatMul`).
 540       `op_name` is the name of the operation that has was just executed. This
 541         name is set by the client who created the operation and can be `None` if
 542         it is unset.
 543       `attrs` contains the attributes of the operation as a `tuple` of
 544         alternating attribute names and attribute values.
 545       `inputs` is the `list` of input `Tensor`(s) to the op.
 546       `outputs` is the `list` of output `Tensor`(s) from the op.
 547        Return value(s) from the callback are ignored.
 548     """
 549     # TODO(cais): (b/64674139) Allow access to function-internal operations.
 550     self._post_execution_callbacks.append(callback)

 592 
 593     Returns:
 594       A RunMetadata protocol buffer. Or None if not enabled.
 595     """
 596     if not self._context_handle:
 597       return None
 598     with c_api_util.tf_buffer() as buffer_:
 599       pywrap_tensorflow.TFE_ContextExportRunMetadata(
 600           self._context_handle, buffer_)
 601       proto_data = pywrap_tensorflow.TF_GetBuffer(buffer_)
 602     run_metadata = config_pb2.RunMetadata()
 603     run_metadata.ParseFromString(compat.as_bytes(proto_data))
 604     return run_metadata
