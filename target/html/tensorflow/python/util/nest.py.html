  16 """## Functions for working with arbitrarily nested sequences of elements.
  17 
  18 This module can perform operations on nested structures. A nested structure is a
  19 Python sequence, tuple (including `namedtuple`), or dict that can contain
  20 further sequences, tuples, and dicts.
  21 
  22 attr.s decorated classes (http://www.attrs.org) are also supported, in the
  23 same way as `namedtuple`.
  24 
  25 The utilities here assume (and do not check) that the nested structures form a
  26 'tree', i.e., no references in the structure of the input of these functions
  27 should be recursive.
  28 
  29 Example structures: `((3, 4), 5, (6, 7, (9, 10), 8))`, `(np.array(0),
  30   (np.array([3, 4]), tf.constant([3, 4])))`
  31 """

  44 def _get_attrs_values(obj):
  45   """Returns the list of values from an attrs instance."""
  46   attrs = getattr(obj.__class__, "__attrs_attrs__")
  47   return [getattr(obj, a.name) for a in attrs]

  50 def _sorted(dict_):
  51   """Returns a sorted list of the dict keys, with error if keys not sortable."""
  52   try:
  53     return sorted(_six.iterkeys(dict_))
  54   except TypeError:
  55     raise TypeError("nest only supports dicts with sortable keys.")

  79 def _sequence_like(instance, args):
  80   """Converts the sequence `args` to the same type as `instance`.
  81 
  82   Args:
  83     instance: an instance of `tuple`, `list`, `namedtuple`, `dict`, or
  84         `collections.OrderedDict`.
  85     args: elements to be converted to the `instance` type.
  86 
  87   Returns:
  88     `args` with the type of `instance`.
  89   """
  90   if _is_mapping(instance):
  91     # Pack dictionaries in a deterministic order by sorting the keys.
  92     # Notice this means that we ignore the original order of `OrderedDict`
  93     # instances. This is intentional, to avoid potential bugs caused by mixing
  94     # ordered and plain dicts (e.g., flattening a dict but using a
  95     # corresponding `OrderedDict` to pack it back).
  96     result = dict(zip(_sorted(instance), args))
  97     return type(instance)((key, result[key]) for key in _six.iterkeys(instance))
  98   elif _is_namedtuple(instance) or _is_attrs(instance):
  99     return type(instance)(*args)
 100   else:
 101     # Not a namedtuple
 102     return type(instance)(args)

 105 def _yield_value(iterable):
 106   """Yields the next value from the given iterable."""
 107   if _is_mapping(iterable):
 108     # Iterate through dictionaries in a deterministic order by sorting the
 109     # keys. Notice this means that we ignore the original order of `OrderedDict`
 110     # instances. This is intentional, to avoid potential bugs caused by mixing
 111     # ordered and plain dicts (e.g., flattening a dict but using a
 112     # corresponding `OrderedDict` to pack it back).
 113     for key in _sorted(iterable):
 114       yield iterable[key]
 115   elif _is_attrs(iterable):
 116     for value in _get_attrs_values(iterable):
 117       yield value
 118   else:
 119     for value in iterable:
 120       yield value

 147 def assert_same_structure(nest1, nest2, check_types=True):
 148   """Asserts that two structures are nested in the same way.
 149 
 150   Note that namedtuples with identical name and fields are always considered
 151   to have the same shallow structure (even with `check_types=True`).
 152   For intance, this code will print `True`:
 153 
 154   ```python
 155   def nt(a, b):
 156     return collections.namedtuple('foo', 'a b')(a, b)
 157   print(assert_same_structure(nt(0, 1), nt(2, 3)))
 158   ```
 159 
 160   Args:
 161     nest1: an arbitrarily nested structure.
 162     nest2: an arbitrarily nested structure.
 163     check_types: if `True` (default) types of sequences are checked as well,
 164         including the keys of dictionaries. If set to `False`, for example a
 165         list and a tuple of objects will look the same if they have the same
 166         size. Note that namedtuples with identical name and fields are always
 167         considered to have the same shallow structure. Two types will also be
 168         considered the same if they are both list subtypes (which allows "list"
 169         and "_ListWrapper" from checkpointable dependency tracking to compare
 170         equal).
 171 
 172   Raises:
 173     ValueError: If the two structures do not have the same number of elements or
 174       if the two structures are not nested in the same way.
 175     TypeError: If the two structures differ in the type of sequence in any of
 176       their substructures. Only possible if `check_types` is `True`.
 177   """
 178   try:
 179     _pywrap_tensorflow.AssertSameStructure(nest1, nest2, check_types)
 180   except (ValueError, TypeError) as e:
 181     str1 = str(map_structure(lambda _: _DOT, nest1))
 182     str2 = str(map_structure(lambda _: _DOT, nest2))
 183     raise type(e)("%s\n"
 184                   "Entire first structure:\n%s\n"
 185                   "Entire second structure:\n%s"
 186                   % (str(e), str1, str2))

 189 def flatten_dict_items(dictionary):
 190   """Returns a dictionary with flattened keys and values.
 191 
 192   This function flattens the keys and values of a dictionary, which can be
 193   arbitrarily nested structures, and returns the flattened version of such
 194   structures:
 195 
 196   ```python
 197   example_dictionary = {(4, 5, (6, 8)): ("a", "b", ("c", "d"))}
 198   result = {4: "a", 5: "b", 6: "c", 8: "d"}
 199   flatten_dict_items(example_dictionary) == result
 200   ```
 201 
 202   The input dictionary must satisfy two properties:
 203 
 204   1. Its keys and values should have the same exact nested structure.
 205   2. The set of all flattened keys of the dictionary must not contain repeated
 206      keys.
 207 
 208   Args:
 209     dictionary: the dictionary to zip
 210 
 211   Returns:
 212     The zipped dictionary.
 213 
 214   Raises:
 215     TypeError: If the input is not a dictionary.
 216     ValueError: If any key and value have not the same structure, or if keys are
 217       not unique.
 218   """
 219   if not isinstance(dictionary, (dict, _collections.Mapping)):
 220     raise TypeError("input must be a dictionary")
 221   flat_dictionary = {}
 222   for i, v in _six.iteritems(dictionary):
 223     if not is_sequence(i):
 224       if i in flat_dictionary:
 225         raise ValueError(
 226             "Could not flatten dictionary: key %s is not unique." % i)
 227       flat_dictionary[i] = v
 228     else:
 229       flat_i = flatten(i)
 230       flat_v = flatten(v)
 231       if len(flat_i) != len(flat_v):
 232         raise ValueError(
 233             "Could not flatten dictionary. Key had %d elements, but value had "
 234             "%d elements. Key: %s, value: %s."
 235             % (len(flat_i), len(flat_v), flat_i, flat_v))
 236       for new_i, new_v in zip(flat_i, flat_v):
 237         if new_i in flat_dictionary:
 238           raise ValueError(
 239               "Could not flatten dictionary: key %s is not unique."
 240               % (new_i))
 241         flat_dictionary[new_i] = new_v
 242   return flat_dictionary

 245 def _packed_nest_with_indices(structure, flat, index):
 246   """Helper function for pack_sequence_as.
 247 
 248   Args:
 249     structure: Substructure (list / tuple / dict) to mimic.
 250     flat: Flattened values to output substructure for.
 251     index: Index at which to start reading from flat.
 252 
 253   Returns:
 254     The tuple (new_index, child), where:
 255       * new_index - the updated index into `flat` having processed `structure`.
 256       * packed - the subset of `flat` corresponding to `structure`,
 257                  having started at `index`, and packed into the same nested
 258                  format.
 259 
 260   Raises:
 261     ValueError: if `structure` contains more elements than `flat`
 262       (assuming indexing starts from `index`).
 263   """
 264   packed = []
 265   for s in _yield_value(structure):
 266     if is_sequence(s):
 267       new_index, child = _packed_nest_with_indices(s, flat, index)
 268       packed.append(_sequence_like(s, child))
 269       index = new_index
 270     else:
 271       packed.append(flat[index])
 272       index += 1
 273   return index, packed

 276 def pack_sequence_as(structure, flat_sequence):
 277   """Returns a given flattened sequence packed into a given structure.
 278 
 279   If `structure` is a scalar, `flat_sequence` must be a single-element list;
 280   in this case the return value is `flat_sequence[0]`.
 281 
 282   If `structure` is or contains a dict instance, the keys will be sorted to
 283   pack the flat sequence in deterministic order. This is true also for
 284   `OrderedDict` instances: their sequence order is ignored, the sorting order of
 285   keys is used instead. The same convention is followed in `flatten`.
 286   This correctly repacks dicts and `OrderedDict`s after they have been
 287   flattened, and also allows flattening an `OrderedDict` and then repacking it
 288   back using a corresponding plain dict, or vice-versa.
 289   Dictionaries with non-sortable keys cannot be flattened.
 290 
 291   Args:
 292     structure: Nested structure, whose structure is given by nested lists,
 293         tuples, and dicts. Note: numpy arrays and strings are considered
 294         scalars.
 295     flat_sequence: flat sequence to pack.
 296 
 297   Returns:
 298     packed: `flat_sequence` converted to have the same recursive structure as
 299       `structure`.
 300 
 301   Raises:
 302     ValueError: If `flat_sequence` and `structure` have different
 303       element counts.
 304     TypeError: `structure` is or contains a dict with non-sortable keys.
 305   """
 306   if not is_sequence(flat_sequence):
 307     raise TypeError("flat_sequence must be a sequence")
 308 
 309   if not is_sequence(structure):
 310     if len(flat_sequence) != 1:
 311       raise ValueError("Structure is a scalar but len(flat_sequence) == %d > 1"
 312                        % len(flat_sequence))
 313     return flat_sequence[0]
 314 
 315   try:
 316     final_index, packed = _packed_nest_with_indices(structure, flat_sequence, 0)
 317     if final_index < len(flat_sequence):
 318       raise IndexError
 319   except IndexError:
 320     flat_structure = flatten(structure)
 321     if len(flat_structure) != len(flat_sequence):
 322       raise ValueError(
 323           "Could not pack sequence. Structure had %d elements, but "
 324           "flat_sequence had %d elements.  Structure: %s, flat_sequence: %s." %
 325           (len(flat_structure), len(flat_sequence), structure, flat_sequence))
 326   return _sequence_like(structure, packed)

 329 def map_structure(func, *structure, **check_types_dict):
 330   """Applies `func` to each entry in `structure` and returns a new structure.
 331 
 332   Applies `func(x[0], x[1], ...)` where x[i] is an entry in
 333   `structure[i]`.  All structures in `structure` must have the same arity,
 334   and the return value will contain the results in the same structure.
 335 
 336   Args:
 337     func: A callable that accepts as many arguments as there are structures.
 338     *structure: scalar, or tuple or list of constructed scalars and/or other
 339       tuples/lists, or scalars.  Note: numpy arrays are considered as scalars.
 340     **check_types_dict: only valid keyword argument is `check_types`. If set to
 341       `True` (default) the types of iterables within the structures have to be
 342       same (e.g. `map_structure(func, [1], (1,))` raises a `TypeError`
 343       exception). To allow this set this argument to `False`.
 344       Note that namedtuples with identical name and fields are always
 345       considered to have the same shallow structure.
 346 
 347   Returns:
 348     A new structure with the same arity as `structure`, whose values correspond
 349     to `func(x[0], x[1], ...)` where `x[i]` is a value in the corresponding
 350     location in `structure[i]`. If there are different sequence types and
 351     `check_types` is `False` the sequence types of the first structure will be
 352     used.
 353 
 354   Raises:
 355     TypeError: If `func` is not callable or if the structures do not match
 356       each other by depth tree.
 357     ValueError: If no structure is provided or if the structures do not match
 358       each other by type.
 359     ValueError: If wrong keyword arguments are provided.
 360   """
 361   if not callable(func):
 362     raise TypeError("func must be callable, got: %s" % func)
 363 
 364   if not structure:
 365     raise ValueError("Must provide at least one structure")
 366 
 367   if check_types_dict:
 368     if "check_types" not in check_types_dict or len(check_types_dict) > 1:
 369       raise ValueError("Only valid keyword argument is check_types")
 370     check_types = check_types_dict["check_types"]
 371   else:
 372     check_types = True
 373 
 374   for other in structure[1:]:
 375     assert_same_structure(structure[0], other, check_types=check_types)
 376 
 377   flat_structure = [flatten(s) for s in structure]
 378   entries = zip(*flat_structure)
 379 
 380   return pack_sequence_as(
 381       structure[0], [func(*x) for x in entries])

 384 def map_structure_with_paths(func, *structure, **kwargs):
 385   """Applies `func` to each entry in `structure` and returns a new structure.
 386 
 387   Applies `func(path, x[0], x[1], ..., **kwargs)` where x[i] is an entry in
 388   `structure[i]` and `path` is the common path to x[i] in the structures.  All
 389   structures in `structure` must have the same arity, and the return value will
 390   contain the results in the same structure. Special kwarg `check_types`
 391   determines whether the types of iterables within the structure must be the
 392   same-- see **kwargs definition below.
 393 
 394   Args:
 395     func: A callable with the signature func(path, *values, **kwargs) that is
 396       evaluated on the leaves of the structure.
 397     *structure: A variable number of compatible structures to process.
 398     **kwargs: Optional kwargs to be passed through to func. Special kwarg
 399       `check_types` is not passed to func, but instead determines whether the
 400       types of iterables within the structures have to be same (e.g.,
 401       `map_structure(func, [1], (1,))` raises a `TypeError` exception). By
 402       default, the types must match. To allow iteration over structures of
 403       different types (but common arity), set this kwarg to `False`.
 404 
 405   Returns:
 406     A structure of the same form as the input structures whose leaves are the
 407     result of evaluating func on corresponding leaves of the input structures.
 408 
 409   Raises:
 410     TypeError: If `func` is not callable or if the structures do not match
 411       each other by depth tree.
 412     TypeError: If `check_types` is not `False` and the two structures differ in
 413       the type of sequence in any of their substructures.
 414     ValueError: If no structures are provided.
 415   """
 416   if not callable(func):
 417     raise TypeError("func must be callable, got: %s" % func)
 418   if not structure:
 419     raise ValueError("Must provide at least one structure")
 420 
 421   check_types = kwargs.pop("check_types", True)
 422   for other in structure[1:]:
 423     assert_same_structure(structure[0], other, check_types=check_types)
 424 
 425   # First set paths_and_values to:
 426   # [[(p11, v11), ... (p1n, v1n)], ... [(pm1, vm1), ... (pmn, vmn)]]
 427   paths_and_values = [flatten_with_joined_string_paths(s) for s in structure]
 428 
 429   # Now zip(*paths_and_values) would be:
 430   # [((p11, v11), ... (pm1, vm1)), ... ((p1n, v1n), ... (pmn, vmn))]
 431   # so grouped_by_path is set to:
 432   # [[(p11, ... pm1), (v11, ... vm1)], ... [(p1n, ... pmn), (v1n, ... vmn)]]
 433   # Note that p1i, ... pmi must all be equal since the structures are the same.
 434   grouped_by_path = [zip(*p_v) for p_v in zip(*paths_and_values)]
 435 
 436   return pack_sequence_as(structure[0], [
 437       func(paths[0], *values, **kwargs) for paths, values in grouped_by_path])

 440 def _yield_flat_up_to(shallow_tree, input_tree):
 441   """Yields elements `input_tree` partially flattened up to `shallow_tree`."""
 442   if is_sequence(shallow_tree):
 443     for shallow_branch, input_branch in zip(_yield_value(shallow_tree),
 444                                             _yield_value(input_tree)):
 445       for input_leaf in _yield_flat_up_to(shallow_branch, input_branch):
 446         yield input_leaf
 447   else:
 448     yield input_tree

 451 def assert_shallow_structure(shallow_tree, input_tree, check_types=True):
 452   """Asserts that `shallow_tree` is a shallow structure of `input_tree`.
 453 
 454   That is, this function tests if the `input_tree` structure can be created from
 455   the `shallow_tree` structure by replacing its leaf nodes with deeper
 456   tree structures.
 457 
 458   Examples:
 459 
 460   The following code will raise an exception:
 461   ```python
 462     shallow_tree = ["a", "b"]
 463     input_tree = ["c", ["d", "e"], "f"]
 464     assert_shallow_structure(shallow_tree, input_tree)
 465   ```
 466 
 467   The following code will not raise an exception:
 468   ```python
 469     shallow_tree = ["a", "b"]
 470     input_tree = ["c", ["d", "e"]]
 471     assert_shallow_structure(shallow_tree, input_tree)
 472   ```
 473 
 474   Args:
 475     shallow_tree: an arbitrarily nested structure.
 476     input_tree: an arbitrarily nested structure.
 477     check_types: if `True` (default) the sequence types of `shallow_tree` and
 478       `input_tree` have to be the same. Note that even with check_types==True,
 479       this function will consider two different namedtuple classes with the same
 480       name and _fields attribute to be the same class.
 481 
 482   Raises:
 483     TypeError: If `shallow_tree` is a sequence but `input_tree` is not.
 484     TypeError: If the sequence types of `shallow_tree` are different from
 485       `input_tree`. Only raised if `check_types` is `True`.
 486     ValueError: If the sequence lengths of `shallow_tree` are different from
 487       `input_tree`.
 488   """
 489   if is_sequence(shallow_tree):
 490     if not is_sequence(input_tree):
 491       raise TypeError(
 492           "If shallow structure is a sequence, input must also be a sequence. "
 493           "Input has type: %s." % type(input_tree))
 494 
 495     if check_types and not isinstance(input_tree, type(shallow_tree)):
 496       # Duck-typing means that nest should be fine with two different
 497       # namedtuples with identical name and fields.
 498       shallow_is_namedtuple = _is_namedtuple(shallow_tree, False)
 499       input_is_namedtuple = _is_namedtuple(input_tree, False)
 500       if shallow_is_namedtuple and input_is_namedtuple:
 501         if not _same_namedtuples(shallow_tree, input_tree):
 502           raise TypeError(
 503               "The two namedtuples don't have the same sequence type. Input "
 504               "structure has type %s, while shallow structure has type %s."
 505               % (type(input_tree), type(shallow_tree)))
 506       else:
 507         raise TypeError(
 508             "The two structures don't have the same sequence type. Input "
 509             "structure has type %s, while shallow structure has type %s."
 510             % (type(input_tree), type(shallow_tree)))
 511 
 512     if len(input_tree) != len(shallow_tree):
 513       raise ValueError(
 514           "The two structures don't have the same sequence length. Input "
 515           "structure has length %s, while shallow structure has length %s."
 516           % (len(input_tree), len(shallow_tree)))
 517 
 518     if check_types and isinstance(shallow_tree, (dict, _collections.Mapping)):
 519       if set(input_tree) != set(shallow_tree):
 520         raise ValueError(
 521             "The two structures don't have the same keys. Input "
 522             "structure has keys %s, while shallow structure has keys %s." %
 523             (list(_six.iterkeys(input_tree)),
 524              list(_six.iterkeys(shallow_tree))))
 525 
 526       input_tree = list(sorted(_six.iteritems(input_tree)))
 527       shallow_tree = list(sorted(_six.iteritems(shallow_tree)))
 528 
 529     for shallow_branch, input_branch in zip(shallow_tree, input_tree):
 530       assert_shallow_structure(shallow_branch, input_branch,
 531                                check_types=check_types)

 534 def flatten_up_to(shallow_tree, input_tree):
 535   """Flattens `input_tree` up to `shallow_tree`.
 536 
 537   Any further depth in structure in `input_tree` is retained as elements in the
 538   partially flatten output.
 539 
 540   If `shallow_tree` and `input_tree` are not sequences, this returns a
 541   single-element list: `[input_tree]`.
 542 
 543   Use Case:
 544 
 545   Sometimes we may wish to partially flatten a nested sequence, retaining some
 546   of the nested structure. We achieve this by specifying a shallow structure,
 547   `shallow_tree`, we wish to flatten up to.
 548 
 549   The input, `input_tree`, can be thought of as having the same structure as
 550   `shallow_tree`, but with leaf nodes that are themselves tree structures.
 551 
 552   Examples:
 553 
 554   ```python
 555   input_tree = [[[2, 2], [3, 3]], [[4, 9], [5, 5]]]
 556   shallow_tree = [[True, True], [False, True]]
 557 
 558   flattened_input_tree = flatten_up_to(shallow_tree, input_tree)
 559   flattened_shallow_tree = flatten_up_to(shallow_tree, shallow_tree)
 560 
 561   # Output is:
 562   # [[2, 2], [3, 3], [4, 9], [5, 5]]
 563   # [True, True, False, True]
 564   ```
 565 
 566   ```python
 567   input_tree = [[('a', 1), [('b', 2), [('c', 3), [('d', 4)]]]]]
 568   shallow_tree = [['level_1', ['level_2', ['level_3', ['level_4']]]]]
 569 
 570   input_tree_flattened_as_shallow_tree = flatten_up_to(shallow_tree, input_tree)
 571   input_tree_flattened = flatten(input_tree)
 572 
 573   # Output is:
 574   # [('a', 1), ('b', 2), ('c', 3), ('d', 4)]
 575   # ['a', 1, 'b', 2, 'c', 3, 'd', 4]
 576   ```
 577 
 578   Non-Sequence Edge Cases:
 579 
 580   ```python
 581   flatten_up_to(0, 0)  # Output: [0]
 582   flatten_up_to(0, [0, 1, 2])  # Output: [[0, 1, 2]]
 583   flatten_up_to([0, 1, 2], 0)  # Output: TypeError
 584   flatten_up_to([0, 1, 2], [0, 1, 2])  # Output: [0, 1, 2]
 585   ```
 586 
 587   Args:
 588     shallow_tree: a possibly pruned structure of input_tree.
 589     input_tree: an arbitrarily nested structure or a scalar object.
 590       Note, numpy arrays are considered scalars.
 591 
 592   Returns:
 593     A Python list, the partially flattened version of `input_tree` according to
 594     the structure of `shallow_tree`.
 595 
 596   Raises:
 597     TypeError: If `shallow_tree` is a sequence but `input_tree` is not.
 598     TypeError: If the sequence types of `shallow_tree` are different from
 599       `input_tree`.
 600     ValueError: If the sequence lengths of `shallow_tree` are different from
 601       `input_tree`.
 602   """
 603   assert_shallow_structure(shallow_tree, input_tree)
 604   return list(_yield_flat_up_to(shallow_tree, input_tree))

 607 def map_structure_up_to(shallow_tree, func, *inputs):
 608   """Applies a function or op to a number of partially flattened inputs.
 609 
 610   The `inputs` are flattened up to `shallow_tree` before being mapped.
 611 
 612   Use Case:
 613 
 614   Sometimes we wish to apply a function to a partially flattened
 615   sequence (for example when the function itself takes sequence inputs). We
 616   achieve this by specifying a shallow structure, `shallow_tree` we wish to
 617   flatten up to.
 618 
 619   The `inputs`, can be thought of as having the same structure as
 620   `shallow_tree`, but with leaf nodes that are themselves tree structures.
 621 
 622   This function therefore will return something with the same base structure as
 623   `shallow_tree`.
 624 
 625   Examples:
 626 
 627   ```python
 628   ab_tuple = collections.namedtuple("ab_tuple", "a, b")
 629   op_tuple = collections.namedtuple("op_tuple", "add, mul")
 630   inp_val = ab_tuple(a=2, b=3)
 631   inp_ops = ab_tuple(a=op_tuple(add=1, mul=2), b=op_tuple(add=2, mul=3))
 632   out = map_structure_up_to(inp_val, lambda val, ops: (val + ops.add) * ops.mul,
 633                             inp_val, inp_ops)
 634 
 635   # Output is: ab_tuple(a=6, b=15)
 636   ```
 637 
 638   ```python
 639   data_list = [[2, 4, 6, 8], [[1, 3, 5, 7, 9], [3, 5, 7]]]
 640   name_list = ['evens', ['odds', 'primes']]
 641   out = map_structure_up_to(
 642       name_list,
 643       lambda name, sec: "first_{}_{}".format(len(sec), name),
 644       name_list, data_list)
 645 
 646   # Output is: ['first_4_evens', ['first_5_odds', 'first_3_primes']]
 647   ```
 648 
 649   Args:
 650     shallow_tree: a shallow tree, common to all the inputs.
 651     func: callable which will be applied to each input individually.
 652     *inputs: arbitrarily nested combination of objects that are compatible with
 653         shallow_tree. The function `func` is applied to corresponding
 654         partially flattened elements of each input, so the function must support
 655         arity of `len(inputs)`.
 656 
 657   Raises:
 658     TypeError: If `shallow_tree` is a sequence but `input_tree` is not.
 659     TypeError: If the sequence types of `shallow_tree` are different from
 660       `input_tree`.
 661     ValueError: If the sequence lengths of `shallow_tree` are different from
 662       `input_tree`.
 663 
 664   Returns:
 665     result of repeatedly applying `func`, with same structure as
 666     `shallow_tree`.
 667   """
 668   if not inputs:
 669     raise ValueError("Cannot map over no sequences")
 670   for input_tree in inputs:
 671     assert_shallow_structure(shallow_tree, input_tree)
 672 
 673   # Flatten each input separately, apply the function to corresponding elements,
 674   # then repack based on the structure of the first input.
 675   all_flattened_up_to = [flatten_up_to(shallow_tree, input_tree)
 676                          for input_tree in inputs]
 677   results = [func(*tensors) for tensors in zip(*all_flattened_up_to)]
 678   return pack_sequence_as(structure=shallow_tree, flat_sequence=results)

 681 def get_traverse_shallow_structure(traverse_fn, structure):
 682   """Generates a shallow structure from a `traverse_fn` and `structure`.
 683 
 684   `traverse_fn` must accept any possible subtree of `structure` and return
 685   a depth=1 structure containing `True` or `False` values, describing which
 686   of the top-level subtrees may be traversed.  It may also
 687   return scalar `True` or `False` "traversal is OK / not OK for all subtrees."
 688 
 689   Examples are available in the unit tests (nest_test.py).
 690 
 691   Args:
 692     traverse_fn: Function taking a substructure and returning either a scalar
 693       `bool` (whether to traverse that substructure or not) or a depth=1
 694       shallow structure of the same type, describing which parts of the
 695       substructure to traverse.
 696     structure: The structure to traverse.
 697 
 698   Returns:
 699     A shallow structure containing python bools, which can be passed to
 700     `map_structure_up_to` and `flatten_up_to`.
 701 
 702   Raises:
 703     TypeError: if `traverse_fn` returns a sequence for a non-sequence input,
 704       or a structure with depth higher than 1 for a sequence input,
 705       or if any leaf values in the returned structure or scalar are not type
 706       `bool`.
 707   """
 708   to_traverse = traverse_fn(structure)
 709   if not is_sequence(structure):
 710     if not isinstance(to_traverse, bool):
 711       raise TypeError("traverse_fn returned structure: %s for non-structure: %s"
 712                       % (to_traverse, structure))
 713     return to_traverse
 714   level_traverse = []
 715   if isinstance(to_traverse, bool):
 716     if not to_traverse:
 717       # Do not traverse this substructure at all.  Exit early.
 718       return False
 719     else:
 720       # Traverse the entire substructure.
 721       for branch in _yield_value(structure):
 722         level_traverse.append(
 723             get_traverse_shallow_structure(traverse_fn, branch))
 724   elif not is_sequence(to_traverse):
 725     raise TypeError("traverse_fn returned a non-bool scalar: %s for input: %s"
 726                     % (to_traverse, structure))
 727   else:
 728     # Traverse some subset of this substructure.
 729     assert_shallow_structure(to_traverse, structure)
 730     for t, branch in zip(_yield_value(to_traverse), _yield_value(structure)):
 731       if not isinstance(t, bool):
 732         raise TypeError(
 733             "traverse_fn didn't return a depth=1 structure of bools.  saw: %s "
 734             " for structure: %s" % (to_traverse, structure))
 735       if t:
 736         level_traverse.append(
 737             get_traverse_shallow_structure(traverse_fn, branch))
 738       else:
 739         level_traverse.append(False)
 740   return _sequence_like(structure, level_traverse)

 743 def yield_flat_paths(nest):
 744   """Yields paths for some nested structure.
 745 
 746   Paths are lists of objects which can be str-converted, which may include
 747   integers or other types which are used as indices in a dict.
 748 
 749   The flat list will be in the corresponding order as if you called
 750   `snt.nest.flatten` on the structure. This is handy for naming Tensors such
 751   the TF scope structure matches the tuple structure.
 752 
 753   E.g. if we have a tuple `value = Foo(a=3, b=Bar(c=23, d=42))`
 754 
 755   ```shell
 756   >>> nest.flatten(value)
 757   [3, 23, 42]
 758   >>> list(nest.yield_flat_paths(value))
 759   [('a',), ('b', 'c'), ('b', 'd')]
 760   ```
 761 
 762   ```shell
 763   >>> list(nest.yield_flat_paths({'a': [3]}))
 764   [('a', 0)]
 765   >>> list(nest.yield_flat_paths({'a': 3}))
 766   [('a',)]
 767   ```
 768 
 769   Args:
 770     nest: the value to produce a flattened paths list for.
 771 
 772   Yields:
 773     Tuples containing index or key values which form the path to a specific
 774       leaf value in the nested structure.
 775   """
 776 
 777   # The _maybe_add_final_path_element function is used below in order to avoid
 778   # adding trailing slashes when the sub-element recursed into is a leaf.
 779   if isinstance(nest, (dict, _collections.Mapping)):
 780     for key in _sorted(nest):
 781       value = nest[key]
 782       for sub_path in yield_flat_paths(value):
 783         yield (key,) + sub_path
 784   elif _is_namedtuple(nest):
 785     for key in nest._fields:
 786       value = getattr(nest, key)
 787       for sub_path in yield_flat_paths(value):
 788         yield (key,) + sub_path
 789   elif isinstance(nest, _six.string_types):
 790     yield ()
 791   elif isinstance(nest, _collections.Sequence):
 792     for idx, value in enumerate(nest):
 793       for sub_path in yield_flat_paths(value):
 794         yield (idx,) + sub_path
 795   else:
 796     yield ()
 797

 799 def flatten_with_joined_string_paths(structure, separator="/"):
 800   """Returns a list of (string path, data element) tuples.
 801 
 802   The order of tuples produced matches that of `nest.flatten`. This allows you
 803   to flatten a nested structure while keeping information about where in the
 804   structure each data element was located. See `nest.yield_flat_paths`
 805   for more information.
 806 
 807   Args:
 808     structure: the nested structure to flatten.
 809     separator: string to separate levels of hierarchy in the results, defaults
 810       to '/'.
 811 
 812   Returns:
 813     A list of (string, data element) tuples.
 814   """
 815   flat_paths = yield_flat_paths(structure)
 816   def stringify_and_join(path_elements):
 817     return separator.join(str(path_element) for path_element in path_elements)
 818   flat_string_paths = [stringify_and_join(path) for path in flat_paths]
 819   return list(zip(flat_string_paths, flatten(structure)))
