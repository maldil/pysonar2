  45 def _add_deprecated_function_notice_to_docstring(doc, date, instructions):
  46   """Adds a deprecation notice to a docstring for deprecated functions."""
  47   main_text = ['THIS FUNCTION IS DEPRECATED. It will be removed %s.' %
  48                ('in a future version' if date is None else ('after %s' % date))]
  49   if instructions:
  50     main_text.append('Instructions for updating:')
  51   return decorator_utils.add_notice_to_docstring(
  52       doc, instructions,
  53       'DEPRECATED FUNCTION',
  54       '(deprecated)', main_text)

  69 def _validate_deprecation_args(date, instructions):
  70   if date is not None and not re.match(r'20\d\d-[01]\d-[0123]\d', date):
  71     raise ValueError('Date must be YYYY-MM-DD.')
  72   if not instructions:
  73     raise ValueError('Don\'t deprecate things without conversion instructions!')

  76 def _call_location(outer=False):
  77   """Returns call location given level up from current call."""
  78   frame = tf_inspect.currentframe()
  79   if frame:
  80     # CPython internals are available, use them for performance.
  81     # walk back two frames to get to deprecated function caller.
  82     frame = frame.f_back
  83     if frame.f_back:
  84       frame = frame.f_back
  85     if outer and frame.f_back:
  86       frame = frame.f_back
  87     return '%s:%d' % (frame.f_code.co_filename, frame.f_lineno)
  88   else:
  89     # Slow fallback path
  90     stack = tf_inspect.stack(0)  # 0 avoids generating unused context
  91     entry = stack[3 if outer else 2]
  92     return '%s:%d' % (entry[1], entry[2])

  95 def _wrap_decorator(wrapped_function):
  96   """Indicate that one function wraps another.
  97 
  98   This decorator wraps a function using `tf_decorator.make_decorator`
  99   so that doc generation scripts can pick up original function
 100   signature.
 101   It would be better to use @functools.wrap decorator, but it would
 102   not update function signature to match wrapped function in Python 2.
 103 
 104   Args:
 105     wrapped_function: The function that decorated function wraps.
 106 
 107   Returns:
 108     Function that accepts wrapper function as an argument and returns
 109     `TFDecorator` instance.
 110   """
 111   def wrapper(wrapper_func):
 112     return tf_decorator.make_decorator(wrapped_function, wrapper_func)
 113   return wrapper

 116 def deprecated_alias(deprecated_name, name, func_or_class, warn_once=True):
 117   """Deprecate a symbol in favor of a new name with identical semantics.
 118 
 119   This function is meant to be used when defining a backwards-compatibility
 120   alias for a symbol which has been moved. For example:
 121 
 122   module1.py:
 123   ```python
 124   class NewNameForClass: pass
 125   ```
 126 
 127   module2.py:
 128   ```python
 129   import module1
 130 
 131   DeprecatedNameForClass = deprecated_alias(
 132     deprecated_name='module2.DeprecatedNameForClass',
 133     name='module1.NewNameForClass',
 134     module1.NewNameForClass)
 135   ```
 136 
 137   This function works for classes and functions.
 138 
 139   For classes, it creates a new class which is functionally identical (it
 140   inherits from the original, and overrides its constructor), but which prints
 141   a deprecation warning when an instance is created. It also adds a deprecation
 142   notice to the class' docstring.
 143 
 144   For functions, it returns a function wrapped by `tf_decorator.make_decorator`.
 145   That function prints a warning when used, and has a deprecation notice in its
 146   docstring. This is more or less equivalent (the deprecation warning has
 147   slightly different text) to writing:
 148 
 149   ```python
 150   @deprecated
 151   def deprecated_alias(original_args):
 152     real_function(original_args)
 153   ```
 154 
 155   Args:
 156     deprecated_name: The name of the symbol that is being deprecated, to be used
 157       in the warning message. This should be its fully qualified name to avoid
 158       confusion.
 159     name: The name of the symbol that is to be used instead of the deprecated
 160       name. This should be a fully qualified name to avoid confusion.
 161     func_or_class: The (non-deprecated) class or function for which a deprecated
 162       alias should be created.
 163     warn_once: If True (the default), only print a deprecation warning the first
 164       time this function is used, or the class is instantiated.
 165 
 166   Returns:
 167     A wrapped version of `func_or_class` which prints a deprecation warning on
 168     use and has a modified docstring.
 169   """
 170   if tf_inspect.isclass(func_or_class):
 171 
 172     # Make a new class with __init__ wrapped in a warning.
 173     class _NewClass(func_or_class):  # pylint: disable=missing-docstring
 174       __doc__ = decorator_utils.add_notice_to_docstring(
 175           func_or_class.__doc__, 'Please use %s instead.' % name,
 176           'DEPRECATED CLASS',
 177           '(deprecated)', ['THIS CLASS IS DEPRECATED. '
 178                            'It will be removed in a future version. '])
 179       __name__ = func_or_class.__name__
 180       __module__ = _call_location(outer=True)
 181 
 182       @_wrap_decorator(func_or_class.__init__)
 183       def __init__(self, *args, **kwargs):
 184         if hasattr(_NewClass.__init__, '__func__'):
 185           # Python 2
 186           _NewClass.__init__.__func__.__doc__ = func_or_class.__init__.__doc__
 187         else:
 188           # Python 3
 189           _NewClass.__init__.__doc__ = func_or_class.__init__.__doc__
 190 
 191         if _PRINT_DEPRECATION_WARNINGS:
 192           # We're making the alias as we speak. The original may have other
 193           # aliases, so we cannot use it to check for whether it's already been
 194           # warned about.
 195           if _NewClass.__init__ not in _PRINTED_WARNING:
 196             if warn_once:
 197               _PRINTED_WARNING[_NewClass.__init__] = True
 198             logging.warning(
 199                 'From %s: The name %s is deprecated. Please use %s instead.\n',
 200                 _call_location(), deprecated_name, name)
 201         super(_NewClass, self).__init__(*args, **kwargs)
 202 
 203     return _NewClass
 204   else:
 205     decorator_utils.validate_callable(func_or_class, 'deprecated')
 206 
 207     # Make a wrapper for the original
 208     @functools.wraps(func_or_class)
 209     def new_func(*args, **kwargs):  # pylint: disable=missing-docstring
 210       if _PRINT_DEPRECATION_WARNINGS:
 211         # We're making the alias as we speak. The original may have other
 212         # aliases, so we cannot use it to check for whether it's already been
 213         # warned about.
 214         if new_func not in _PRINTED_WARNING:
 215           if warn_once:
 216             _PRINTED_WARNING[new_func] = True
 217           logging.warning(
 218               'From %s: The name %s is deprecated. Please use %s instead.\n',
 219               _call_location(), deprecated_name, name)
 220       return func_or_class(*args, **kwargs)
 221     return tf_decorator.make_decorator(
 222         func_or_class, new_func, 'deprecated',
 223         _add_deprecated_function_notice_to_docstring(
 224             func_or_class.__doc__, None, 'Please use %s instead.' % name))

 227 def deprecated_endpoints(*args):
 228   """Decorator for marking endpoints deprecated.
 229 
 230   This decorator does not print deprecation messages.
 231   TODO(annarev): eventually start printing deprecation warnings when
 232   @deprecation_endpoints decorator is added.
 233 
 234   Args:
 235     *args: Deprecated endpoint names.
 236 
 237   Returns:
 238     A function that takes symbol as an argument and adds
 239     _tf_deprecated_api_names to that symbol.
 240     _tf_deprecated_api_names would be set to a list of deprecated
 241     endpoint names for the symbol.
 242   """
 243   def deprecated_wrapper(func):
 244     # pylint: disable=protected-access
 245     if '_tf_deprecated_api_names' in func.__dict__:
 246       raise DeprecatedNamesAlreadySet(
 247           'Cannot set deprecated names for %s to %s. '
 248           'Deprecated names are already set to %s.' % (
 249               func.__name__, str(args), str(func._tf_deprecated_api_names)))
 250     func._tf_deprecated_api_names = args
 251     # pylint: disable=protected-access
 252     return func
 253   return deprecated_wrapper

 256 def deprecated(date, instructions, warn_once=True):
 257   """Decorator for marking functions or methods deprecated.
 258 
 259   This decorator logs a deprecation warning whenever the decorated function is
 260   called. It has the following format:
 261 
 262     <function> (from <module>) is deprecated and will be removed after <date>.
 263     Instructions for updating:
 264     <instructions>
 265 
 266   If `date` is None, 'after <date>' is replaced with 'in a future version'.
 267   <function> will include the class name if it is a method.
 268 
 269   It also edits the docstring of the function: ' (deprecated)' is appended
 270   to the first line of the docstring and a deprecation notice is prepended
 271   to the rest of the docstring.
 272 
 273   Args:
 274     date: String or None. The date the function is scheduled to be removed.
 275       Must be ISO 8601 (YYYY-MM-DD), or None.
 276     instructions: String. Instructions on how to update code using the
 277       deprecated function.
 278     warn_once: Boolean. Set to `True` to warn only the first time the decorated
 279       function is called. Otherwise, every call will log a warning.
 280 
 281   Returns:
 282     Decorated function or method.
 283 
 284   Raises:
 285     ValueError: If date is not None or in ISO 8601 format, or instructions are
 286       empty.
 287   """
 288   _validate_deprecation_args(date, instructions)
 289 
 290   def deprecated_wrapper(func):
 291     """Deprecation wrapper."""
 292     decorator_utils.validate_callable(func, 'deprecated')
 293     @functools.wraps(func)
 294     def new_func(*args, **kwargs):  # pylint: disable=missing-docstring
 295       if _PRINT_DEPRECATION_WARNINGS:
 296         if func not in _PRINTED_WARNING:
 297           if warn_once:
 298             _PRINTED_WARNING[func] = True
 299           logging.warning(
 300               'From %s: %s (from %s) is deprecated and will be removed %s.\n'
 301               'Instructions for updating:\n%s',
 302               _call_location(), decorator_utils.get_qualified_name(func),
 303               func.__module__,
 304               'in a future version' if date is None else ('after %s' % date),
 305               instructions)
 306       return func(*args, **kwargs)
 307     return tf_decorator.make_decorator(
 308         func, new_func, 'deprecated',
 309         _add_deprecated_function_notice_to_docstring(func.__doc__, date,
 310                                                      instructions))
 311   return deprecated_wrapper

 318 def deprecated_args(date, instructions, *deprecated_arg_names_or_tuples,
 319                     **kwargs):
 320   """Decorator for marking specific function arguments as deprecated.
 321 
 322   This decorator logs a deprecation warning whenever the decorated function is
 323   called with the deprecated argument. It has the following format:
 324 
 325     Calling <function> (from <module>) with <arg> is deprecated and will be
 326     removed after <date>. Instructions for updating:
 327       <instructions>
 328 
 329   If `date` is None, 'after <date>' is replaced with 'in a future version'.
 330   <function> includes the class name if it is a method.
 331 
 332   It also edits the docstring of the function: ' (deprecated arguments)' is
 333   appended to the first line of the docstring and a deprecation notice is
 334   prepended to the rest of the docstring.
 335 
 336   Args:
 337     date: String or None. The date the function is scheduled to be removed.
 338       Must be ISO 8601 (YYYY-MM-DD), or None.
 339     instructions: String. Instructions on how to update code using the
 340       deprecated function.
 341     *deprecated_arg_names_or_tuples: String or 2-Tuple(String,
 342       [ok_vals]).  The string is the deprecated argument name.
 343       Optionally, an ok-value may be provided.  If the user provided
 344       argument equals this value, the warning is suppressed.
 345     **kwargs: If `warn_once=False` is passed, every call with a deprecated
 346       argument will log a warning. The default behavior is to only warn the
 347       first time the function is called with any given deprecated argument.
 348       All other kwargs raise `ValueError`.
 349 
 350   Returns:
 351     Decorated function or method.
 352 
 353   Raises:
 354     ValueError: If date is not None or in ISO 8601 format, instructions are
 355       empty, the deprecated arguments are not present in the function
 356       signature, the second element of a deprecated_tuple is not a
 357       list, or if a kwarg other than `warn_once` is passed.
 358   """
 359   _validate_deprecation_args(date, instructions)
 360   if not deprecated_arg_names_or_tuples:
 361     raise ValueError('Specify which argument is deprecated.')
 362   if kwargs and list(kwargs.keys()) != ['warn_once']:
 363     kwargs.pop('warn_once', None)
 364     raise ValueError('Illegal argument to deprecated_args: %s' % kwargs)
 365   warn_once = kwargs.get('warn_once', True)
 366 
 367   def _get_arg_names_to_ok_vals():
 368     """Returns a dict mapping arg_name to DeprecatedArgSpec w/o position."""
 369     d = {}
 370     for name_or_tuple in deprecated_arg_names_or_tuples:
 371       if isinstance(name_or_tuple, tuple):
 372         d[name_or_tuple[0]] = DeprecatedArgSpec(-1, True, name_or_tuple[1])
 373       else:
 374         d[name_or_tuple] = DeprecatedArgSpec(-1, False, None)
 375     return d
 376 
 377   def _get_deprecated_positional_arguments(names_to_ok_vals, arg_spec):
 378     """Builds a dictionary from deprecated arguments to their spec.
 379 
 380     Returned dict is keyed by argument name.
 381     Each value is a DeprecatedArgSpec with the following fields:
 382        position: The zero-based argument position of the argument
 383          within the signature.  None if the argument isn't found in
 384          the signature.
 385        ok_values:  Values of this argument for which warning will be
 386          suppressed.
 387 
 388     Args:
 389       names_to_ok_vals: dict from string arg_name to a list of values,
 390         possibly empty, which should not elicit a warning.
 391       arg_spec: Output from tf_inspect.getfullargspec on the called function.
 392 
 393     Returns:
 394       Dictionary from arg_name to DeprecatedArgSpec.
 395     """
 396     arg_name_to_pos = {
 397         name: pos for pos, name in enumerate(arg_spec.args)}
 398     deprecated_positional_args = {}
 399     for arg_name, spec in iter(names_to_ok_vals.items()):
 400       if arg_name in arg_name_to_pos:
 401         pos = arg_name_to_pos[arg_name]
 402         deprecated_positional_args[arg_name] = DeprecatedArgSpec(
 403             pos, spec.has_ok_value, spec.ok_value)
 404     return deprecated_positional_args
 405 
 406   def deprecated_wrapper(func):
 407     """Deprecation decorator."""
 408     decorator_utils.validate_callable(func, 'deprecated_args')
 409     deprecated_arg_names = _get_arg_names_to_ok_vals()
 410 
 411     arg_spec = tf_inspect.getfullargspec(func)
 412     deprecated_positions = _get_deprecated_positional_arguments(
 413         deprecated_arg_names, arg_spec)
 414 
 415     is_varargs_deprecated = arg_spec.varargs in deprecated_arg_names
 416     is_kwargs_deprecated = arg_spec.varkw in deprecated_arg_names
 417 
 418     if (len(deprecated_positions) + is_varargs_deprecated + is_kwargs_deprecated
 419         != len(deprecated_arg_names_or_tuples)):
 420       known_args = arg_spec.args + [arg_spec.varargs, arg_spec.varkw]
 421       missing_args = [arg_name for arg_name in deprecated_arg_names
 422                       if arg_name not in known_args]
 423       raise ValueError('The following deprecated arguments are not present '
 424                        'in the function signature: %s. '
 425                        'Found next arguments: %s.' % (missing_args, known_args))
 426 
 427     def _same_value(a, b):
 428       """A comparison operation that works for multiple object types.
 429 
 430       Returns True for two empty lists, two numeric values with the
 431       same value, etc.
 432 
 433       Returns False for (pd.DataFrame, None), and other pairs which
 434       should not be considered equivalent.
 435 
 436       Args:
 437         a: value one of the comparison.
 438         b: value two of the comparison.
 439 
 440       Returns:
 441         A boolean indicating whether the two inputs are the same value
 442         for the purposes of deprecation.
 443       """
 444       if a is b:
 445         return True
 446       try:
 447         equality = a == b
 448         if isinstance(equality, bool):
 449           return equality
 450       except TypeError:
 451         return False
 452       return False
 453 
 454     @functools.wraps(func)
 455     def new_func(*args, **kwargs):
 456       """Deprecation wrapper."""
 457       # TODO(apassos) figure out a way to have reasonable performance with
 458       # deprecation warnings and eager mode.
 459       if is_in_graph_mode.IS_IN_GRAPH_MODE() and _PRINT_DEPRECATION_WARNINGS:
 460         invalid_args = []
 461         named_args = tf_inspect.getcallargs(func, *args, **kwargs)
 462         for arg_name, spec in iter(deprecated_positions.items()):
 463           if (spec.position < len(args) and
 464               not (spec.has_ok_value and
 465                    _same_value(named_args[arg_name], spec.ok_value))):
 466             invalid_args.append(arg_name)
 467         if is_varargs_deprecated and len(args) > len(arg_spec.args):
 468           invalid_args.append(arg_spec.varargs)
 469         if is_kwargs_deprecated and kwargs:
 470           invalid_args.append(arg_spec.varkw)
 471         for arg_name in deprecated_arg_names:
 472           if (arg_name in kwargs and
 473               not (deprecated_positions[arg_name].has_ok_value and
 474                    _same_value(named_args[arg_name],
 475                                deprecated_positions[arg_name].ok_value))):
 476             invalid_args.append(arg_name)
 477         for arg_name in invalid_args:
 478           if (func, arg_name) not in _PRINTED_WARNING:
 479             if warn_once:
 480               _PRINTED_WARNING[(func, arg_name)] = True
 481             logging.warning(
 482                 'From %s: calling %s (from %s) with %s is deprecated and will '
 483                 'be removed %s.\nInstructions for updating:\n%s',
 484                 _call_location(), decorator_utils.get_qualified_name(func),
 485                 func.__module__, arg_name,
 486                 'in a future version' if date is None else ('after %s' % date),
 487                 instructions)
 488       return func(*args, **kwargs)
 489     return tf_decorator.make_decorator(func, new_func, 'deprecated',
 490                                        _add_deprecated_arg_notice_to_docstring(
 491                                            func.__doc__, date, instructions))
 492   return deprecated_wrapper

 495 def deprecated_arg_values(date, instructions, warn_once=True,
 496                           **deprecated_kwargs):
 497   """Decorator for marking specific function argument values as deprecated.
 498 
 499   This decorator logs a deprecation warning whenever the decorated function is
 500   called with the deprecated argument values. It has the following format:
 501 
 502     Calling <function> (from <module>) with <arg>=<value> is deprecated and
 503     will be removed after <date>. Instructions for updating:
 504       <instructions>
 505 
 506   If `date` is None, 'after <date>' is replaced with 'in a future version'.
 507   <function> will include the class name if it is a method.
 508 
 509   It also edits the docstring of the function: ' (deprecated arguments)' is
 510   appended to the first line of the docstring and a deprecation notice is
 511   prepended to the rest of the docstring.
 512 
 513   Args:
 514     date: String or None. The date the function is scheduled to be removed.
 515       Must be ISO 8601 (YYYY-MM-DD), or None
 516     instructions: String. Instructions on how to update code using the
 517       deprecated function.
 518     warn_once: If `True`, warn only the first time this function is called with
 519       deprecated argument values. Otherwise, every call (with a deprecated
 520       argument value) will log a warning.
 521     **deprecated_kwargs: The deprecated argument values.
 522 
 523   Returns:
 524     Decorated function or method.
 525 
 526   Raises:
 527     ValueError: If date is not None or in ISO 8601 format, or instructions are
 528       empty.
 529   """
 530   _validate_deprecation_args(date, instructions)
 531   if not deprecated_kwargs:
 532     raise ValueError('Specify which argument values are deprecated.')
 533 
 534   def deprecated_wrapper(func):
 535     """Deprecation decorator."""
 536     decorator_utils.validate_callable(func, 'deprecated_arg_values')
 537     @functools.wraps(func)
 538     def new_func(*args, **kwargs):
 539       """Deprecation wrapper."""
 540       if _PRINT_DEPRECATION_WARNINGS:
 541         named_args = tf_inspect.getcallargs(func, *args, **kwargs)
 542         for arg_name, arg_value in deprecated_kwargs.items():
 543           if arg_name in named_args and named_args[arg_name] == arg_value:
 544             if (func, arg_name) not in _PRINTED_WARNING:
 545               if warn_once:
 546                 _PRINTED_WARNING[(func, arg_name)] = True
 547               logging.warning(
 548                   'From %s: calling %s (from %s) with %s=%s is deprecated and '
 549                   'will be removed %s.\nInstructions for updating:\n%s',
 550                   _call_location(), decorator_utils.get_qualified_name(func),
 551                   func.__module__, arg_name, arg_value, 'in a future version'
 552                   if date is None else ('after %s' % date), instructions)
 553       return func(*args, **kwargs)
 554     return tf_decorator.make_decorator(func, new_func, 'deprecated',
 555                                        _add_deprecated_arg_notice_to_docstring(
 556                                            func.__doc__, date, instructions))
 557   return deprecated_wrapper

 560 def deprecated_argument_lookup(new_name, new_value, old_name, old_value):
 561   """Looks up deprecated argument name and ensures both are not used.
 562 
 563   Args:
 564     new_name: new name of argument
 565     new_value: value of new argument (or None if not used)
 566     old_name: old name of argument
 567     old_value: value of old argument (or None if not used)
 568   Returns:
 569     The effective argument that should be used.
 570   Raises:
 571     ValueError: if new_value and old_value are both non-null
 572   """
 573   if old_value is not None:
 574     if new_value is not None:
 575       raise ValueError("Cannot specify both '%s' and '%s'" %
 576                        (old_name, new_name))
 577     return old_value
 578   return new_value

 581 def rewrite_argument_docstring(old_doc, old_argument, new_argument):
 582   return old_doc.replace('`%s`' % old_argument, '`%s`' % new_argument).replace(
 583       '%s:' % old_argument, '%s:' % new_argument)

 586 @tf_contextlib.contextmanager
 587 def silence():
 588   """Temporarily silence deprecation warnings."""
 589   global _PRINT_DEPRECATION_WARNINGS
 590   print_deprecation_warnings = _PRINT_DEPRECATION_WARNINGS
 591   _PRINT_DEPRECATION_WARNINGS = False
 592   yield
 593   _PRINT_DEPRECATION_WARNINGS = print_deprecation_warnings
