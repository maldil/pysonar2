  15 """Utilities for exporting TensorFlow symbols to the API.
  16 
  17 Exporting a function or a class:
  18 
  19 To export a function or a class use tf_export decorator. For e.g.:
  20 ```python
  21 @tf_export('foo', 'bar.foo')
  22 def foo(...):
  23   ...
  24 ```
  25 
  26 If a function is assigned to a variable, you can export it by calling
  27 tf_export explicitly. For e.g.:
  28 ```python
  29 foo = get_foo(...)
  30 tf_export('foo', 'bar.foo')(foo)
  31 ```
  32 
  33 
  34 Exporting a constant
  35 ```python
  36 foo = 1
  37 tf_export("consts.foo").export_constant(__name__, 'foo')
  38 ```
  39 """
  40 from __future__ import absolute_import
  41 from __future__ import division

  44 import collections
  45 import functools
  46 import sys

  48 from tensorflow.python.util import tf_decorator
  49 
  50 ESTIMATOR_API_NAME = 'estimator'
  51 TENSORFLOW_API_NAME = 'tensorflow'
  52 
  53 _Attributes = collections.namedtuple(
  54     'ExportedApiAttributes', ['names', 'constants'])
  55 
  56 # Attribute values must be unique to each API.
  57 API_ATTRS = {
  58     TENSORFLOW_API_NAME: _Attributes(
  59         '_tf_api_names',
  60         '_tf_api_constants'),
  61     ESTIMATOR_API_NAME: _Attributes(
  62         '_estimator_api_names',
  63         '_estimator_api_constants')
  64 }
  65 
  66 API_ATTRS_V1 = {
  67     TENSORFLOW_API_NAME: _Attributes(
  68         '_tf_api_names_v1',
  69         '_tf_api_constants_v1'),
  70     ESTIMATOR_API_NAME: _Attributes(
  71         '_estimator_api_names_v1',
  72         '_estimator_api_constants_v1')
  73 }
  74

  76 class SymbolAlreadyExposedError(Exception):
  77   """Raised when adding API names to symbol that already has API names."""
  78   pass

  81 def get_canonical_name_for_symbol(symbol, api_name=TENSORFLOW_API_NAME):
  82   """Get canonical name for the API symbol.
  83 
  84   Canonical name is the first non-deprecated endpoint name.
  85 
  86   Args:
  87     symbol: API function or class.
  88     api_name: API name (tensorflow or estimator).
  89 
  90   Returns:
  91     Canonical name for the API symbol (for e.g. initializers.zeros) if
  92     canonical name could be determined. Otherwise, returns None.
  93   """
  94   if not hasattr(symbol, '__dict__'):
  95     return None
  96   api_names_attr = API_ATTRS[api_name].names
  97   _, undecorated_symbol = tf_decorator.unwrap(symbol)
  98   if api_names_attr not in undecorated_symbol.__dict__:
  99     return None
 100   api_names = getattr(undecorated_symbol, api_names_attr)
 101   # TODO(annarev): may be add a separate deprecated attribute
 102   # for estimator names.
 103   deprecated_api_names = undecorated_symbol.__dict__.get(
 104       '_tf_deprecated_api_names', [])
 105   return get_canonical_name(api_names, deprecated_api_names)
 106 
 107 
 108 def get_canonical_name(api_names, deprecated_api_names):
 109   """Get first non-deprecated endpoint name.
 110 
 111   Args:
 112     api_names: API names iterable.
 113     deprecated_api_names: Deprecated API names iterable.
 114   Returns:
 115     Canonical name if there is at least one non-deprecated endpoint.
 116     Otherwise returns None.
 117   """
 118   return next(
 119       (name for name in api_names if name not in deprecated_api_names),
 120       None)

 123 class api_export(object):  # pylint: disable=invalid-name
 124   """Provides ways to export symbols to the TensorFlow API."""
 125 
 126   def __init__(self, *args, **kwargs):
 127     """Export under the names *args (first one is considered canonical).
 128 
 129     Args:
 130       *args: API names in dot delimited format.
 131       **kwargs: Optional keyed arguments.
 132         v1: Names for the TensorFlow V1 API. If not set, we will use V2 API
 133           names both for TensorFlow V1 and V2 APIs.
 134         overrides: List of symbols that this is overriding
 135           (those overrided api exports will be removed). Note: passing overrides
 136           has no effect on exporting a constant.
 137         api_name: Name of the API you want to generate (e.g. `tensorflow` or
 138           `estimator`). Default is `tensorflow`.
 139         allow_multiple_exports: Allow symbol to be exported multiple time under
 140           different names.
 141     """
 142     self._names = args
 143     self._names_v1 = kwargs.get('v1', args)
 144     self._api_name = kwargs.get('api_name', TENSORFLOW_API_NAME)
 145     self._overrides = kwargs.get('overrides', [])
 146     self._allow_multiple_exports = kwargs.get('allow_multiple_exports', False)
 147 
 148   def __call__(self, func):
 149     """Calls this decorator.
 150 
 151     Args:
 152       func: decorated symbol (function or class).
 153 
 154     Returns:
 155       The input function with _tf_api_names attribute set.
 156 
 157     Raises:
 158       SymbolAlreadyExposedError: Raised when a symbol already has API names
 159         and kwarg `allow_multiple_exports` not set.
 160     """
 161     api_names_attr = API_ATTRS[self._api_name].names
 162     api_names_attr_v1 = API_ATTRS_V1[self._api_name].names
 163     # Undecorate overridden names
 164     for f in self._overrides:
 165       _, undecorated_f = tf_decorator.unwrap(f)
 166       delattr(undecorated_f, api_names_attr)
 167       delattr(undecorated_f, api_names_attr_v1)
 168 
 169     _, undecorated_func = tf_decorator.unwrap(func)
 170     self.set_attr(undecorated_func, api_names_attr, self._names)
 171     self.set_attr(undecorated_func, api_names_attr_v1, self._names_v1)
 172     return func
 173 
 174   def set_attr(self, func, api_names_attr, names):
 175     # Check for an existing api. We check if attribute name is in
 176     # __dict__ instead of using hasattr to verify that subclasses have
 177     # their own _tf_api_names as opposed to just inheriting it.
 178     if api_names_attr in func.__dict__:
 179       if not self._allow_multiple_exports:
 180         raise SymbolAlreadyExposedError(
 181             'Symbol %s is already exposed as %s.' %
 182             (func.__name__, getattr(func, api_names_attr)))  # pylint: disable=protected-access
 183     setattr(func, api_names_attr, names)
 184 
 185   def export_constant(self, module_name, name):
 186     """Store export information for constants/string literals.
 187 
 188     Export information is stored in the module where constants/string literals
 189     are defined.
 190 
 191     e.g.
 192     ```python
 193     foo = 1
 194     bar = 2
 195     tf_export("consts.foo").export_constant(__name__, 'foo')
 196     tf_export("consts.bar").export_constant(__name__, 'bar')
 197     ```
 198 
 199     Args:
 200       module_name: (string) Name of the module to store constant at.
 201       name: (string) Current constant name.
 202     """
 203     module = sys.modules[module_name]
 204     api_constants_attr = API_ATTRS[self._api_name].constants
 205     api_constants_attr_v1 = API_ATTRS_V1[self._api_name].constants
 206 
 207     if not hasattr(module, api_constants_attr):
 208       setattr(module, api_constants_attr, [])
 209     # pylint: disable=protected-access
 210     getattr(module, api_constants_attr).append(
 211         (self._names, name))
 212 
 213     if not hasattr(module, api_constants_attr_v1):
 214       setattr(module, api_constants_attr_v1, [])
 215     getattr(module, api_constants_attr_v1).append(
 216         (self._names_v1, name))
 217

 219 tf_export = functools.partial(api_export, api_name=TENSORFLOW_API_NAME)
 220 estimator_export = functools.partial(
 221     api_export, api_name=ESTIMATOR_API_NAME, allow_multiple_exports=True)
