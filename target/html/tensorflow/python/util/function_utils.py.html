  29 def _is_bounded_method(fn):
  30   _, fn = tf_decorator.unwrap(fn)
  31   return tf_inspect.ismethod(fn) and (fn.__self__ is not None)
  32 
  33 
  34 def _is_callable_object(obj):
  35   return hasattr(obj, '__call__') and tf_inspect.ismethod(obj.__call__)

  38 def fn_args(fn):
  39   """Get argument names for function-like object.
  40 
  41   Args:
  42     fn: Function, or function-like object (e.g., result of `functools.partial`).
  43 
  44   Returns:
  45     `tuple` of string argument names.
  46 
  47   Raises:
  48     ValueError: if partial function has positionally bound arguments
  49   """
  50   if isinstance(fn, functools.partial):
  51     args = fn_args(fn.func)
  52     args = [a for a in args[len(fn.args):] if a not in (fn.keywords or [])]
  53   else:
  54     if _is_callable_object(fn):
  55       fn = fn.__call__
  56     args = tf_inspect.getfullargspec(fn).args
  57     if _is_bounded_method(fn):
  58       args.remove('self')
  59   return tuple(args)
  60

  62 def has_kwargs(fn):
  63   """Returns whether the passed callable has **kwargs in its signature.
  64 
  65   Args:
  66     fn: Function, or function-like object (e.g., result of `functools.partial`).
  67 
  68   Returns:
  69     `bool`: if `fn` has **kwargs in its signature.
  70 
  71   Raises:
  72      `TypeError`: If fn is not a Function, or function-like object.
  73   """
  74   if isinstance(fn, functools.partial):
  75     fn = fn.func
  76   elif _is_callable_object(fn):
  77     fn = fn.__call__
  78   elif not callable(fn):
  79     raise TypeError(
  80         'fn should be a function-like object, but is of type {}.'.format(
  81             type(fn)))
  82   return tf_inspect.getfullargspec(fn).varkw is not None

  85 def get_func_name(func):
  86   """Returns name of passed callable."""
  87   _, func = tf_decorator.unwrap(func)
  88   if callable(func):
  89     if tf_inspect.isfunction(func):
  90       return func.__name__
  91     elif tf_inspect.ismethod(func):
  92       return '%s.%s' % (six.get_method_self(func).__class__.__name__,
  93                         six.get_method_function(func).__name__)
  94     else:  # Probably a class instance with __call__
  95       return str(type(func))
  96   else:
  97     raise ValueError('Argument must be callable')

 100 def get_func_code(func):
 101   """Returns func_code of passed callable, or None if not available."""
 102   _, func = tf_decorator.unwrap(func)
 103   if callable(func):
 104     if tf_inspect.isfunction(func) or tf_inspect.ismethod(func):
 105       return six.get_function_code(func)
 106     # Since the object is not a function or method, but is a callable, we will
 107     # try to access the __call__method as a function.  This works with callable
 108     # classes but fails with functool.partial objects despite their __call__
 109     # attribute.
 110     try:
 111       return six.get_function_code(func.__call__)
 112     except AttributeError:
 113       return None
 114   else:
 115     raise ValueError('Argument must be callable')
