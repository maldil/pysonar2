  31 class _TFShouldUseHelper(object):
  32   """Object stored in TFShouldUse-wrapped objects.
  33 
  34   When it is deleted it will emit a warning or error if its `sate` method
  35   has not been called by time of deletion.
  36   """
  37 
  38   def __init__(self, type_, repr_, stack_frame, fatal_error_if_unsated):
  39     self._type = type_
  40     self._repr = repr_
  41     self._stack_frame = stack_frame
  42     self._fatal_error_if_unsated = fatal_error_if_unsated
  43     self._sated = False
  44 
  45   def sate(self):
  46     self._sated = True
  47     self._type = None
  48     self._repr = None
  49     self._stack_frame = None
  50     self._logging_module = None
  51 
  52   def __del__(self):
  53     if self._sated:
  54       return
  55     if self._fatal_error_if_unsated:
  56       logger = tf_logging.fatal
  57     else:
  58       logger = tf_logging.error
  59     creation_stack = ''.join(
  60         [line.rstrip() for line in traceback.format_stack(self._stack_frame)])
  61     logger(
  62         '==================================\n'
  63         'Object was never used (type %s):\n%s\nIf you want to mark it as '
  64         'used call its "mark_used()" method.\nIt was originally created '
  65         'here:\n%s\n'
  66         '==================================' %
  67         (self._type, self._repr, creation_stack))
  68

  70 def _new__init__(self, true_value, tf_should_use_helper):
  71   # pylint: disable=protected-access
  72   self._tf_should_use_helper = tf_should_use_helper
  73   self._true_value = true_value
  74 
  75 
  76 def _new__setattr__(self, key, value):
  77   if key in ('_tf_should_use_helper', '_true_value'):
  78     return object.__setattr__(self, key, value)
  79   return setattr(
  80       object.__getattribute__(self, '_true_value'),
  81       key, value)
  82 
  83 
  84 def _new__getattribute__(self, key):
  85   if key not in ('_tf_should_use_helper', '_true_value'):
  86     object.__getattribute__(self, '_tf_should_use_helper').sate()
  87   if key in ('_tf_should_use_helper', 'mark_used', '__setatt__'):
  88     return object.__getattribute__(self, key)
  89   return getattr(object.__getattribute__(self, '_true_value'), key)
  90 
  91 
  92 def _new_mark_used(self, *args, **kwargs):
  93   object.__getattribute__(self, '_tf_should_use_helper').sate()
  94   try:
  95     mu = object.__getattribute__(
  96         object.__getattribute__(self, '_true_value'),
  97         'mark_used')
  98     return mu(*args, **kwargs)
  99   except AttributeError:
 100     pass

 106 def _get_wrapper(x, tf_should_use_helper):
 107   """Create a wrapper for object x, whose class subclasses type(x).
 108 
 109   The wrapper will emit a warning if it is deleted without any of its
 110   properties being accessed or methods being called.
 111 
 112   Args:
 113     x: The instance to wrap.
 114     tf_should_use_helper: The object that tracks usage.
 115 
 116   Returns:
 117     An object wrapping `x`, of type `type(x)`.
 118   """
 119   type_x = type(x)
 120   memoized = _WRAPPERS.get(type_x, None)
 121   if memoized:
 122     return memoized(x, tf_should_use_helper)
 123 
 124   tx = copy.deepcopy(type_x)
 125   copy_tx = type(tx.__name__, tx.__bases__, dict(tx.__dict__))
 126   copy_tx.__init__ = _new__init__
 127   copy_tx.__getattribute__ = _new__getattribute__
 128   copy_tx.mark_used = _new_mark_used
 129   copy_tx.__setattr__ = _new__setattr__
 130   _WRAPPERS[type_x] = copy_tx
 131 
 132   return copy_tx(x, tf_should_use_helper)

 135 def _add_should_use_warning(x, fatal_error=False):
 136   """Wraps object x so that if it is never used, a warning is logged.
 137 
 138   Args:
 139     x: Python object.
 140     fatal_error: Python bool.  If `True`, tf.logging.fatal is raised
 141       if the returned value is never used.
 142 
 143   Returns:
 144     An instance of `TFShouldUseWarningWrapper` which subclasses `type(x)`
 145     and is a very shallow wrapper for `x` which logs access into `x`.
 146   """
 147   if x is None or x == []:  # pylint: disable=g-explicit-bool-comparison
 148     return x
 149 
 150   # Extract the current frame for later use by traceback printing.
 151   try:
 152     raise ValueError()
 153   except ValueError:
 154     stack_frame = sys.exc_info()[2].tb_frame.f_back
 155 
 156   tf_should_use_helper = _TFShouldUseHelper(
 157       type_=type(x),
 158       repr_=repr(x),
 159       stack_frame=stack_frame,
 160       fatal_error_if_unsated=fatal_error)
 161 
 162   return _get_wrapper(x, tf_should_use_helper)

 165 def should_use_result(fn):
 166   """Function wrapper that ensures the function's output is used.
 167 
 168   If the output is not used, a `tf.logging.error` is logged.
 169 
 170   An output is marked as used if any of its attributes are read, modified, or
 171   updated.  Examples when the output is a `Tensor` include:
 172 
 173   - Using it in any capacity (e.g. `y = t + 0`, `sess.run(t)`)
 174   - Accessing a property (e.g. getting `t.name` or `t.op`).
 175 
 176   Note, certain behaviors cannot be tracked - for these the object may not
 177   be marked as used.  Examples include:
 178 
 179   - `t != 0`.  In this case, comparison is done on types / ids.
 180   - `isinstance(t, tf.Tensor)`.  Similar to above.
 181 
 182   Args:
 183     fn: The function to wrap.
 184 
 185   Returns:
 186     The wrapped function.
 187   """
 188   def wrapped(*args, **kwargs):
 189     return _add_should_use_warning(fn(*args, **kwargs))
 190   return tf_decorator.make_decorator(
 191       fn, wrapped, 'should_use_result',
 192       ((fn.__doc__ or '') +
 193        ('\n\n  '
 194         '**NOTE** The output of this function should be used.  If it is not, '
 195         'a warning will be logged.  To mark the output as used, '
 196         'call its .mark_used() method.')))

 199 def must_use_result_or_fatal(fn):
 200   """Function wrapper that ensures the function's output is used.
 201 
 202   If the output is not used, a `tf.logging.fatal` error is raised.
 203 
 204   An output is marked as used if any of its attributes are read, modified, or
 205   updated.  Examples when the output is a `Tensor` include:
 206 
 207   - Using it in any capacity (e.g. `y = t + 0`, `sess.run(t)`)
 208   - Accessing a property (e.g. getting `t.name` or `t.op`).
 209 
 210   Note, certain behaviors cannot be tracked - for these the object may not
 211   be marked as used.  Examples include:
 212 
 213   - `t != 0`.  In this case, comparison is done on types / ids.
 214   - `isinstance(t, tf.Tensor)`.  Similar to above.
 215 
 216   Args:
 217     fn: The function to wrap.
 218 
 219   Returns:
 220     The wrapped function.
 221   """
 222   def wrapped(*args, **kwargs):
 223     return _add_should_use_warning(fn(*args, **kwargs), fatal_error=True)
 224   return tf_decorator.make_decorator(
 225       fn, wrapped, 'must_use_result_or_fatal',
 226       ((fn.__doc__ or '') +
 227        ('\n\n  '
 228         '**NOTE** The output of this function must be used.  If it is not, '
 229         'a fatal error will be raised.  To mark the output as used, '
 230         'call its .mark_used() method.')))
