  16 """Utility functions for comparing proto2 messages in Python.
  17 
  18 ProtoEq() compares two proto2 messages for equality.
  19 
  20 ClearDefaultValuedFields() recursively clears the fields that are set to their
  21 default values. This is useful for comparing protocol buffers where the
  22 semantics of unset fields and default valued fields are the same.
  23 
  24 assertProtoEqual() is useful for unit tests.  It produces much more helpful
  25 output than assertEqual() for proto2 messages, e.g. this:
  26 
  27   outer {
  28     inner {
  29 -     strings: "x"
  30 ?               ^
  31 +     strings: "y"
  32 ?               ^
  33     }
  34   }
  35 
  36 ...compared to the default output from assertEqual() that looks like this:
  37 
  38 AssertionError: <my.Msg object at 0x9fb353c> != <my.Msg object at 0x9fb35cc>
  39 
  40 Call it inside your unit test's googletest.TestCase subclasses like this:
  41 
  42   from tensorflow.python.util.protobuf import compare
  43 
  44   class MyTest(googletest.TestCase):
  45     ...
  46     def testXXX(self):
  47       ...
  48       compare.assertProtoEqual(self, a, b)
  49 
  50 Alternatively:
  51 
  52   from tensorflow.python.util.protobuf import compare
  53 
  54   class MyTest(compare.ProtoAssertions, googletest.TestCase):
  55     ...
  56     def testXXX(self):
  57       ...
  58       self.assertProtoEqual(a, b)
  59 """

  75 def assertProtoEqual(self, a, b, check_initialized=True,  # pylint: disable=invalid-name
  76                      normalize_numbers=False, msg=None):
  77   """Fails with a useful error if a and b aren't equal.
  78 
  79   Comparison of repeated fields matches the semantics of
  80   unittest.TestCase.assertEqual(), ie order and extra duplicates fields matter.
  81 
  82   Args:
  83     self: googletest.TestCase
  84     a: proto2 PB instance, or text string representing one.
  85     b: proto2 PB instance -- message.Message or subclass thereof.
  86     check_initialized: boolean, whether to fail if either a or b isn't
  87       initialized.
  88     normalize_numbers: boolean, whether to normalize types and precision of
  89       numbers before comparison.
  90     msg: if specified, is used as the error message on failure.
  91   """
  92   pool = descriptor_pool.Default()
  93   if isinstance(a, six.string_types):
  94     a = text_format.Merge(a, b.__class__(), descriptor_pool=pool)
  95 
  96   for pb in a, b:
  97     if check_initialized:
  98       errors = pb.FindInitializationErrors()
  99       if errors:
 100         self.fail('Initialization errors: %s\n%s' % (errors, pb))
 101     if normalize_numbers:
 102       NormalizeNumberFields(pb)
 103 
 104   self.assertMultiLineEqual(
 105       text_format.MessageToString(a, descriptor_pool=pool),
 106       text_format.MessageToString(b, descriptor_pool=pool),
 107       msg=msg)

 110 def NormalizeNumberFields(pb):
 111   """Normalizes types and precisions of number fields in a protocol buffer.
 112 
 113   Due to subtleties in the python protocol buffer implementation, it is possible
 114   for values to have different types and precision depending on whether they
 115   were set and retrieved directly or deserialized from a protobuf. This function
 116   normalizes integer values to ints and longs based on width, 32-bit floats to
 117   five digits of precision to account for python always storing them as 64-bit,
 118   and ensures doubles are floating point for when they're set to integers.
 119 
 120   Modifies pb in place. Recurses into nested objects.
 121 
 122   Args:
 123     pb: proto2 message.
 124 
 125   Returns:
 126     the given pb, modified in place.
 127   """
 128   for desc, values in pb.ListFields():
 129     is_repeated = True
 130     if desc.label is not descriptor.FieldDescriptor.LABEL_REPEATED:
 131       is_repeated = False
 132       values = [values]
 133 
 134     normalized_values = None
 135 
 136     # We force 32-bit values to int and 64-bit values to long to make
 137     # alternate implementations where the distinction is more significant
 138     # (e.g. the C++ implementation) simpler.
 139     if desc.type in (descriptor.FieldDescriptor.TYPE_INT64,
 140                      descriptor.FieldDescriptor.TYPE_UINT64,
 141                      descriptor.FieldDescriptor.TYPE_SINT64):
 142       normalized_values = [int(x) for x in values]
 143     elif desc.type in (descriptor.FieldDescriptor.TYPE_INT32,
 144                        descriptor.FieldDescriptor.TYPE_UINT32,
 145                        descriptor.FieldDescriptor.TYPE_SINT32,
 146                        descriptor.FieldDescriptor.TYPE_ENUM):
 147       normalized_values = [int(x) for x in values]
 148     elif desc.type == descriptor.FieldDescriptor.TYPE_FLOAT:
 149       normalized_values = [round(x, 6) for x in values]
 150     elif desc.type == descriptor.FieldDescriptor.TYPE_DOUBLE:
 151       normalized_values = [round(float(x), 7) for x in values]
 152 
 153     if normalized_values is not None:
 154       if is_repeated:
 155         pb.ClearField(desc.name)
 156         getattr(pb, desc.name).extend(normalized_values)
 157       else:
 158         setattr(pb, desc.name, normalized_values[0])
 159 
 160     if (desc.type == descriptor.FieldDescriptor.TYPE_MESSAGE or
 161         desc.type == descriptor.FieldDescriptor.TYPE_GROUP):
 162       if (desc.type == descriptor.FieldDescriptor.TYPE_MESSAGE and
 163           desc.message_type.has_options and
 164           desc.message_type.GetOptions().map_entry):
 165         # This is a map, only recurse if the values have a message type.
 166         if (desc.message_type.fields_by_number[2].type ==
 167             descriptor.FieldDescriptor.TYPE_MESSAGE):
 168           for v in six.itervalues(values):
 169             NormalizeNumberFields(v)
 170       else:
 171         for v in values:
 172           # recursive step
 173           NormalizeNumberFields(v)
 174 
 175   return pb

 178 def _IsMap(value):
 179   return isinstance(value, collections.Mapping)

 182 def _IsRepeatedContainer(value):
 183   if isinstance(value, six.string_types):
 184     return False
 185   try:
 186     iter(value)
 187     return True
 188   except TypeError:
 189     return False

 192 def ProtoEq(a, b):
 193   """Compares two proto2 objects for equality.
 194 
 195   Recurses into nested messages. Uses list (not set) semantics for comparing
 196   repeated fields, ie duplicates and order matter.
 197 
 198   Args:
 199     a: A proto2 message or a primitive.
 200     b: A proto2 message or a primitive.
 201 
 202   Returns:
 203     `True` if the messages are equal.
 204   """
 205   def Format(pb):
 206     """Returns a dictionary or unchanged pb bases on its type.
 207 
 208     Specifically, this function returns a dictionary that maps tag
 209     number (for messages) or element index (for repeated fields) to
 210     value, or just pb unchanged if it's neither.
 211 
 212     Args:
 213       pb: A proto2 message or a primitive.
 214     Returns:
 215       A dict or unchanged pb.
 216     """
 217     if isinstance(pb, message.Message):
 218       return dict((desc.number, value) for desc, value in pb.ListFields())
 219     elif _IsMap(pb):
 220       return dict(pb.items())
 221     elif _IsRepeatedContainer(pb):
 222       return dict(enumerate(list(pb)))
 223     else:
 224       return pb
 225 
 226   a, b = Format(a), Format(b)
 227 
 228   # Base case
 229   if not isinstance(a, dict) or not isinstance(b, dict):
 230     return a == b
 231 
 232   # This list performs double duty: it compares two messages by tag value *or*
 233   # two repeated fields by element, in order. the magic is in the format()
 234   # function, which converts them both to the same easily comparable format.
 235   for tag in sorted(set(a.keys()) | set(b.keys())):
 236     if tag not in a or tag not in b:
 237       return False
 238     else:
 239       # Recursive step
 240       if not ProtoEq(a[tag], b[tag]):
 241         return False
 242 
 243   # Didn't find any values that differed, so they're equal!
 244   return True

 247 class ProtoAssertions(object):
 248   """Mix this into a googletest.TestCase class to get proto2 assertions.
 249 
 250   Usage:
 251 
 252   class SomeTestCase(compare.ProtoAssertions, googletest.TestCase):
 253     ...
 254     def testSomething(self):
 255       ...
 256       self.assertProtoEqual(a, b)
 257 
 258   See module-level definitions for method documentation.
 259   """
 260 
 261   # pylint: disable=invalid-name
 262   def assertProtoEqual(self, *args, **kwargs):
 263     return assertProtoEqual(self, *args, **kwargs)
