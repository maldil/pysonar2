  45 def getargspec(obj):
  46   """TFDecorator-aware replacement for inspect.getargspec.
  47 
  48   Args:
  49     obj: A function, partial function, or callable object, possibly
  50     decorated.
  51 
  52   Returns:
  53     The `ArgSpec` that describes the signature of the outermost decorator that
  54     changes the callable's signature. If the callable is not decorated,
  55     `inspect.getargspec()` will be called directly on the object.
  56 
  57   Raises:
  58     ValueError: When callable's signature can not be expressed with
  59       ArgSpec.
  60     TypeError: For objects of unsupported types.
  61   """
  62   if isinstance(obj, functools.partial):
  63     return _get_argspec_for_partial(obj)
  64 
  65   decorators, target = tf_decorator.unwrap(obj)
  66 
  67   spec = next((d.decorator_argspec
  68                for d in decorators
  69                if d.decorator_argspec is not None), None)
  70   if spec:
  71     return spec
  72 
  73   try:
  74     # Python3 will handle most callables here (not partial).
  75     return _inspect.getargspec(target)
  76   except TypeError:
  77     pass
  78 
  79   if isinstance(target, type):
  80     try:
  81       return _inspect.getargspec(target.__init__)
  82     except TypeError:
  83       pass
  84 
  85     try:
  86       return _inspect.getargspec(target.__new__)
  87     except TypeError:
  88       pass
  89 
  90   # The `type(target)` ensures that if a class is received we don't return
  91   # the signature of it's __call__ method.
  92   return _inspect.getargspec(type(target).__call__)

  95 def _get_argspec_for_partial(obj):
  96   """Implements `getargspec` for `functools.partial` objects.
  97 
  98   Args:
  99     obj: The `functools.partial` obeject
 100   Returns:
 101     An `inspect.ArgSpec`
 102   Raises:
 103     ValueError: When callable's signature can not be expressed with
 104       ArgSpec.
 105   """
 106   # When callable is a functools.partial object, we construct its ArgSpec with
 107   # following strategy:
 108   # - If callable partial contains default value for positional arguments (ie.
 109   # object.args), then final ArgSpec doesn't contain those positional arguments.
 110   # - If callable partial contains default value for keyword arguments (ie.
 111   # object.keywords), then we merge them with wrapped target. Default values
 112   # from callable partial takes precedence over those from wrapped target.
 113   #
 114   # However, there is a case where it is impossible to construct a valid
 115   # ArgSpec. Python requires arguments that have no default values must be
 116   # defined before those with default values. ArgSpec structure is only valid
 117   # when this presumption holds true because default values are expressed as a
 118   # tuple of values without keywords and they are always assumed to belong to
 119   # last K arguments where K is number of default values present.
 120   #
 121   # Since functools.partial can give default value to any argument, this
 122   # presumption may no longer hold in some cases. For example:
 123   #
 124   # def func(m, n):
 125   #   return 2 * m + n
 126   # partialed = functools.partial(func, m=1)
 127   #
 128   # This example will result in m having a default value but n doesn't. This is
 129   # usually not allowed in Python and can not be expressed in ArgSpec correctly.
 130   #
 131   # Thus, we must detect cases like this by finding first argument with default
 132   # value and ensures all following arguments also have default values. When
 133   # this is not true, a ValueError is raised.
 134 
 135   n_prune_args = len(obj.args)
 136   partial_keywords = obj.keywords or {}
 137 
 138   args, varargs, keywords, defaults = getargspec(obj.func)
 139 
 140   # Pruning first n_prune_args arguments.
 141   args = args[n_prune_args:]
 142 
 143   # Partial function may give default value to any argument, therefore length
 144   # of default value list must be len(args) to allow each argument to
 145   # potentially be given a default value.
 146   all_defaults = [None] * len(args)
 147   if defaults:
 148     all_defaults[-len(defaults):] = defaults
 149 
 150   # Fill in default values provided by partial function in all_defaults.
 151   for kw, default in six.iteritems(partial_keywords):
 152     idx = args.index(kw)
 153     all_defaults[idx] = default
 154 
 155   # Find first argument with default value set.
 156   first_default = next((idx for idx, x in enumerate(all_defaults) if x), None)
 157 
 158   # If no default values are found, return ArgSpec with defaults=None.
 159   if first_default is None:
 160     return ArgSpec(args, varargs, keywords, None)
 161 
 162   # Checks if all arguments have default value set after first one.
 163   invalid_default_values = [
 164       args[i] for i, j in enumerate(all_defaults) if not j and i > first_default
 165   ]
 166 
 167   if invalid_default_values:
 168     raise ValueError('Some arguments %s do not have default value, but they '
 169                      'are positioned after those with default values. This can '
 170                      'not be expressed with ArgSpec.' % invalid_default_values)
 171 
 172   return ArgSpec(args, varargs, keywords, tuple(all_defaults[first_default:]))

 179   def _getfullargspec(target):
 180     """A python2 version of getfullargspec.
 181 
 182     Args:
 183       target: the target object to inspect.
 184     Returns:
 185       A FullArgSpec with empty kwonlyargs, kwonlydefaults and annotations.
 186     """
 187     argspecs = getargspec(target)
 188     fullargspecs = FullArgSpec(
 189         args=argspecs.args,
 190         varargs=argspecs.varargs,
 191         varkw=argspecs.keywords,
 192         defaults=argspecs.defaults,
 193         kwonlyargs=[],
 194         kwonlydefaults=None,
 195         annotations={})
 196     return fullargspecs

 199 def getfullargspec(obj):
 200   """TFDecorator-aware replacement for `inspect.getfullargspec`.
 201 
 202   This wrapper emulates `inspect.getfullargspec` in[^)]* Python2.
 203 
 204   Args:
 205     obj: A callable, possibly decorated.
 206 
 207   Returns:
 208     The `FullArgSpec` that describes the signature of
 209     the outermost decorator that changes the callable's signature. If the
 210     callable is not decorated, `inspect.getfullargspec()` will be called
 211     directly on the callable.
 212   """
 213   decorators, target = tf_decorator.unwrap(obj)
 214   return next((d.decorator_argspec
 215                for d in decorators
 216                if d.decorator_argspec is not None), _getfullargspec(target))

 219 def getcallargs(func, *positional, **named):
 220   """TFDecorator-aware replacement for inspect.getcallargs.
 221 
 222   Args:
 223     func: A callable, possibly decorated
 224     *positional: The positional arguments that would be passed to `func`.
 225     **named: The named argument dictionary that would be passed to `func`.
 226 
 227   Returns:
 228     A dictionary mapping `func`'s named arguments to the values they would
 229     receive if `func(*positional, **named)` were called.
 230 
 231   `getcallargs` will use the argspec from the outermost decorator that provides
 232   it. If no attached decorators modify argspec, the final unwrapped target's
 233   argspec will be used.
 234   """
 235   argspec = getfullargspec(func)
 236   call_args = named.copy()
 237   this = getattr(func, 'im_self', None) or getattr(func, '__self__', None)
 238   if ismethod(func) and this:
 239     positional = (this,) + positional
 240   remaining_positionals = [arg for arg in argspec.args if arg not in call_args]
 241   call_args.update(dict(zip(remaining_positionals, positional)))
 242   default_count = 0 if not argspec.defaults else len(argspec.defaults)
 243   if default_count:
 244     for arg, value in zip(argspec.args[-default_count:], argspec.defaults):
 245       if arg not in call_args:
 246         call_args[arg] = value
 247   return call_args

 269 def getfile(object):  # pylint: disable=redefined-builtin
 270   """TFDecorator-aware replacement for inspect.getfile."""
 271   unwrapped_object = tf_decorator.unwrap(object)[1]
 272 
 273   # Work around for the case when object is a stack frame
 274   # and only .pyc files are used. In this case, getfile
 275   # might return incorrect path. So, we get the path from f_globals
 276   # instead.
 277   if (hasattr(unwrapped_object, 'f_globals') and
 278       '__file__' in unwrapped_object.f_globals):
 279     return unwrapped_object.f_globals['__file__']
 280   return _inspect.getfile(unwrapped_object)
