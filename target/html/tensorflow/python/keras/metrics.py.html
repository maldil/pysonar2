  81 def clone_metrics(metrics):
  82   """Clones the given metric list/dict."""

  90 def update_state_wrapper(update_state_fn):
  91   """Decorator to wrap metric `update_state()` with `add_update()`.
  92 
  93   Args:
  94     update_state_fn: function that accumulates metric statistics.
  95 
  96   Returns:
  97     Decorated function that wraps `update_state_fn()` with `add_update()`.
  98   """
  99 
 100   def decorated(metric_obj, *args, **kwargs):
 101     """Decorated function with `add_update()`."""
 102 
 103     update_op = update_state_fn(*args, **kwargs)
 104     if update_op is not None:  # update_op will be None in eager execution.
 105       metric_obj.add_update(update_op, inputs=True)
 106       check_is_tensor_or_operation(
 107           update_op, 'Metric {0}\'s update'.format(metric_obj.name))
 108     return update_op
 109 
 110   return tf_decorator.make_decorator(update_state_fn, decorated)

 113 def result_wrapper(result_fn):
 114   """Decorator to wrap metric `result()` function in `merge_call()`.
 115 
 116   Result computation is an idempotent operation that simply calculates the
 117   metric value using the state variables.
 118 
 119   If metric state variables are distributed across towers/devices and
 120   `result()` is requested from the context of one device - This function wraps
 121   `result()` in a distribution strategy `merge_call()`. With this,
 122   the metric state variables will be aggregated across devices.
 123 
 124   Args:
 125     result_fn: function that computes the metric result.
 126 
 127   Returns:
 128     Decorated function that wraps `result_fn()` in distribution strategy
 129     `merge_call()`.
 130   """
 131 
 132   def decorated(metric_obj, *args):
 133     """Decorated function with merge_call."""
 134     tower_context = distribution_strategy_context.get_tower_context()
 135     if tower_context is None:  # if in cross tower context already
 136       result_t = result_fn(*args)
 137     else:
 138       # TODO(psv): Test distribution of metrics using different distribution
 139       # strategies.
 140 
 141       # Creating a wrapper for merge_fn. merge_call invokes the given merge_fn
 142       # with distribution object as the first parameter. We create a wrapper
 143       # here so that the result function need not have that parameter.
 144       def merge_fn_wrapper(distribution, merge_fn, *args):
 145         # We will get `PerDevice` merge function. Taking the first one as all
 146         # are identical copies of the function that we had passed below.
 147         return distribution.unwrap(merge_fn)[0](*args)
 148 
 149       # Wrapping result in merge_call. merge_call is used when we want to leave
 150       # tower mode and compute a value in cross tower mode.
 151       result_t = tower_context.merge_call(merge_fn_wrapper, result_fn, *args)
 152     check_is_tensor_or_operation(result_t,
 153                                  'Metric {0}\'s result'.format(metric_obj.name))
 154     return result_t
 155 
 156   return tf_decorator.make_decorator(result_fn, decorated)

 159 def weakmethod(method):
 160   """Creates a weak reference to the bound method."""
 161 
 162   cls = method.im_class
 163   func = method.im_func
 164   instance_ref = weakref.ref(method.im_self)
 165 
 166   @functools.wraps(method)
 167   def inner(*args, **kwargs):
 168     return func.__get__(instance_ref(), cls)(*args, **kwargs)
 169 
 170   del method
 171   return inner

 174 def safe_div(numerator, denominator):
 175   """Divides two tensors element-wise, returning 0 if the denominator is <= 0.
 176 
 177   Args:
 178     numerator: A `Tensor`.
 179     denominator: A `Tensor`, with dtype matching `numerator`.
 180 
 181   Returns:
 182     0 if `denominator` <= 0, else `numerator` / `denominator`
 183   """
 184   t = math_ops.truediv(numerator, denominator)
 185   zero = array_ops.zeros_like(t, dtype=denominator.dtype)
 186   condition = math_ops.greater(denominator, zero)
 187   zero = math_ops.cast(zero, t.dtype)
 188   return array_ops.where(condition, t, zero)

 191 def squeeze_or_expand_dimensions(y_pred, y_true, sample_weight):
 192   """Squeeze or expand last dimension if needed.
 193 
 194   1. Squeezes last dim of `y_pred` or `y_true` if their rank differs by 1
 195   (using `confusion_matrix.remove_squeezable_dimensions`).
 196   2. Squeezes or expands last dim of `sample_weight` if its rank differs by 1
 197   from the new rank of `y_pred`.
 198   If `sample_weight` is scalar, it is kept scalar.
 199 
 200   This will use static shape if available. Otherwise, it will add graph
 201   operations, which could result in a performance hit.
 202 
 203   Args:
 204     y_pred: Predicted values, a `Tensor` of arbitrary dimensions.
 205     y_true: Optional label `Tensor` whose dimensions match `y_pred`.
 206     sample_weight: Optional weight scalar or `Tensor` whose dimensions match
 207       `y_pred`.
 208 
 209   Returns:
 210     Tuple of `y_pred`, `y_true` and `sample_weight`. Each of them possibly has
 211     the last dimension squeezed,
 212     `sample_weight` could be extended by one dimension.
 213   """
 214   if y_true is not None:
 215     # squeeze last dim of `y_pred` or `y_true` if their rank differs by 1
 216     y_true, y_pred = confusion_matrix.remove_squeezable_dimensions(
 217         y_true, y_pred)
 218 
 219   if sample_weight is None:
 220     return y_pred, y_true, None
 221 
 222   sample_weight = ops.convert_to_tensor(sample_weight)
 223   weights_shape = sample_weight.get_shape()
 224   weights_rank = weights_shape.ndims
 225   if weights_rank == 0:  # If weights is scalar, do nothing.
 226     return y_pred, y_true, sample_weight
 227 
 228   y_pred_shape = y_pred.get_shape()
 229   y_pred_rank = y_pred_shape.ndims
 230   if (y_pred_rank is not None) and (weights_rank is not None):
 231     # Use static rank.
 232     if weights_rank - y_pred_rank == 1:
 233       sample_weight = array_ops.squeeze(sample_weight, [-1])
 234     elif y_pred_rank - weights_rank == 1:
 235       sample_weight = array_ops.expand_dims(sample_weight, [-1])
 236     return y_pred, y_true, sample_weight
 237 
 238   # Use dynamic rank.
 239   weights_rank_tensor = array_ops.rank(sample_weight)
 240   rank_diff = weights_rank_tensor - array_ops.rank(y_pred)
 241   maybe_squeeze_weights = lambda: array_ops.squeeze(sample_weight, [-1])
 242 
 243   def _maybe_expand_weights():
 244     return control_flow_ops.cond(
 245         math_ops.equal(rank_diff,
 246                        -1), lambda: array_ops.expand_dims(sample_weight, [-1]),
 247         lambda: sample_weight)
 248 
 249   def _maybe_adjust_weights():
 250     return control_flow_ops.cond(
 251         math_ops.equal(rank_diff, 1), maybe_squeeze_weights,
 252         _maybe_expand_weights)
 253 
 254   # squeeze or expand last dim of `sample_weight` if its rank differs by 1
 255   # from the new rank of `y_pred`.
 256   sample_weight = control_flow_ops.cond(
 257       math_ops.equal(weights_rank_tensor, 0), lambda: sample_weight,
 258       _maybe_adjust_weights)
 259   return y_pred, y_true, sample_weight

 262 class Metric(Layer):
 263   """Encapsulates metric logic and state.
 264 
 265   Usage with eager execution:
 266 
 267   ```python
 268   m = SomeMetric(...)
 269   for input in ...:
 270     m.update_state(input)
 271   print('Final result: ', m.result().numpy())
 272   ```
 273 
 274   Usage with graph execution:
 275 
 276   ```python
 277   m = SomeMetric(...)
 278   init_op = tf.variables_initializer(m.variables)  # Initialize variables
 279   with tf.Session() as sess:
 280     sess.run(init_op)
 281     for input in ...:
 282       update_op = m.update_state(input)
 283       sess.run(update_op)
 284     print('Final result: ', sess.run(m.result()))
 285   ```
 286 
 287   Usage with tf.keras API:
 288 
 289   ```python
 290   model = tf.keras.Sequential()
 291   model.add(tf.keras.layers.Dense(64, activation='relu'))
 292   model.add(tf.keras.layers.Dense(64, activation='relu'))
 293   model.add(tf.keras.layers.Dense(10, activation='softmax'))
 294 
 295   model.compile(optimizer=tf.train.RMSPropOptimizer(0.01),
 296                 loss=tf.keras.losses.categorical_crossentropy,
 297                 metrics=[tf.keras.metrics.CategoricalAccuracy()])
 298 
 299   data = np.random.random((1000, 32))
 300   labels = np.random.random((1000, 10))
 301 
 302   dataset = tf.data.Dataset.from_tensor_slices((data, labels))
 303   dataset = dataset.batch(32)
 304   dataset = dataset.repeat()
 305 
 306   model.fit(dataset, epochs=10, steps_per_epoch=30)
 307   ```
 308 
 309   To be implemented by subclasses:
 310   * `__init__()`: All state variables should be created in this method by
 311     calling `self.add_weight()` like: `self.var = self.add_weight(...)`
 312   * `update_state()`: Has all updates to the state variables like:
 313     self.var.assign_add(...).
 314   * `result()`: Computes and returns a value for the metric
 315     from the state variables.
 316 
 317   Example subclass implementation:
 318 
 319   ```
 320   class BinaryTruePositives(Metric):
 321     def __init__(self, name='binary_true_positives', dtype=None):
 322       super(BinaryTruePositives, self).__init__(name=name, dtype=dtype)
 323       self.true_positives = self.add_weight(
 324           'true_positives', initializer=init_ops.zeros_initializer)
 325 
 326     def update_state(self, y_true, y_pred, sample_weight=None):
 327       y_true = math_ops.cast(y_true, dtypes.bool)
 328       y_pred = math_ops.cast(y_pred, dtypes.bool)
 329       y_pred, y_true, sample_weight = squeeze_or_expand_dimensions(
 330           y_pred, y_true, sample_weight)
 331 
 332       values = math_ops.logical_and(
 333           math_ops.equal(y_true, True), math_ops.equal(y_pred, True))
 334       values = math_ops.cast(values, self._dtype)
 335       if sample_weight is not None:
 336         sample_weight = math_ops.cast(sample_weight, self._dtype)
 337         values = math_ops.multiply(values, sample_weight)
 338       state_ops.assign_add(self.true_positives, math_ops.reduce_sum(values))
 339 
 340     def result(self):
 341       return array_ops.identity(self.true_positives)
 342   ```
 343   """
 344   __metaclass__ = ABCMeta
 345 
 346   def __init__(self, name=None, dtype=None):
 347     super(Metric, self).__init__(name=name, dtype=dtype)
 348     self.stateful = True  # All metric layers are stateful.
 349     self.built = True
 350     self._dtype = K.floatx() if dtype is None else dtypes.as_dtype(dtype).name
 351 
 352   def __new__(cls, *args, **kwargs):
 353     obj = super(Metric, cls).__new__(cls)
 354 
 355     if sys.version_info < (3,):
 356       # Wrap methods in `weakmethod` function to remove binding and create a
 357       # weak reference. This is to remove reference cycle that is created here.
 358       # This is not an issue in python versions > 3.
 359       if context.executing_eagerly():
 360         obj.update_state = weakmethod(obj.update_state)
 361       obj.update_state = weakmethod(
 362           types.MethodType(update_state_wrapper(obj.update_state), obj))
 363       result = weakmethod(obj.result)
 364       obj.result = weakmethod(types.MethodType(result_wrapper(result), obj))
 365     else:
 366       obj.update_state = types.MethodType(
 367           update_state_wrapper(obj.update_state), obj)
 368       obj.result = types.MethodType(result_wrapper(obj.result), obj)
 369 
 370     return obj
 371 
 372   def __call__(self, *args, **kwargs):
 373     """Accumulates statistics and then computes metric result value.
 374 
 375     Args:
 376       *args:
 377       **kwargs: A mini-batch of inputs to the Metric,
 378         passed on to `update_state()`.
 379 
 380     Returns:
 381       The metric value tensor.
 382     """
 383     update_op = self.update_state(*args, **kwargs)  # pylint: disable=not-callable
 384     with ops.control_dependencies([update_op]):
 385       return self.result()  # pylint: disable=not-callable
 386 
 387   def reset_states(self):
 388     """Resets all of the metric state variables.
 389 
 390     This function is called between epochs/steps,
 391     when a metric is evaluated during training.
 392     """
 393     for v in self.variables:
 394       K.set_value(v, 0)
 395 
 396   @abstractmethod
 397   def update_state(self, *args, **kwargs):
 398     """Accumulates statistics for the metric.
 399 
 400     Note: This function is executed as a graph function in graph mode.
 401     This means:
 402       a) Operations on the same resource are executed in textual order.
 403          This should make it easier to do things like add the updated
 404          value of a variable to another, for example.
 405       b) You don't need to worry about collecting the update ops to execute.
 406          All update ops added to the graph by this function will be executed.
 407       As a result, code should generally work the same way with graph or
 408       eager execution.
 409     and adds the update op to the metric layer.
 410 
 411     Args:
 412       *args:
 413       **kwargs: A mini-batch of inputs to the Metric.
 414     """
 415     NotImplementedError('Must be implemented in subclasses.')
 416 
 417   @abstractmethod
 418   def result(self):
 419     """Computes and returns the metric value tensor.
 420 
 421     Result computation is an idempotent operation that simply calculates the
 422     metric value using the state variables.
 423     """
 424     NotImplementedError('Must be implemented in subclasses.')
 425 
 426   @classmethod
 427   def from_config(cls, config):
 428     if 'trainable' in config:
 429       config.pop('trainable')
 430     return cls(**config)
 431 
 432   ### For use by subclasses ###
 433   @doc_controls.for_subclass_implementers
 434   def add_weight(self,
 435                  name,
 436                  shape=(),
 437                  aggregation=tf_variables.VariableAggregation.SUM,
 438                  synchronization=tf_variables.VariableSynchronization.ON_READ,
 439                  initializer=None):
 440     """Adds state variable. Only for use by subclasses."""
 441     return super(Metric, self).add_weight(
 442         name=name,
 443         shape=shape,
 444         dtype=self._dtype,
 445         trainable=False,
 446         initializer=initializer,
 447         collections=[],
 448         synchronization=synchronization,
 449         aggregation=aggregation)
 450 
 451   ### End: For use by subclasses ###

 454 class Mean(Metric):
 455   """Computes the (weighted) mean of the given values.
 456 
 457   This metric creates two variables, `total` and `count` that are used to
 458   compute the average of `values`. This average is ultimately returned as `mean`
 459   which is an idempotent operation that simply divides `total` by `count`.
 460 
 461   If `sample_weight` is `None`, weights default to 1.
 462   Use `sample_weight` of 0 to mask values.
 463   """
 464 
 465   def __init__(self, name='mean', dtype=None):
 466     """Creates a `Mean` instance.
 467 
 468     Args:
 469       name: (Optional) string name of the metric instance.
 470       dtype: (Optional) data type of the metric result.
 471     """
 472     super(Mean, self).__init__(name=name, dtype=dtype)
 473     # Create new state variables
 474     self.total = self.add_weight(
 475         'total', initializer=init_ops.zeros_initializer)
 476     self.count = self.add_weight(
 477         'count', initializer=init_ops.zeros_initializer)
 478 
 479   def update_state(self, values, sample_weight=None):
 480     """Accumulates statistics for computing the mean.
 481 
 482     For example, if `values` is [1, 3, 5, 7] then the mean is 4. If
 483     the `sample_weight` is specified as [1, 1, 0, 0] then the mean would be 2.
 484 
 485     Args:
 486       values: Per-example value.
 487       sample_weight: Optional weighting of each example. Defaults to 1.
 488 
 489     Returns:
 490       Update op.
 491     """
 492     values = math_ops.cast(values, self._dtype)
 493     if sample_weight is None:
 494       num_values = math_ops.cast(array_ops.size(values), self._dtype)
 495     else:
 496       sample_weight = math_ops.cast(sample_weight, self._dtype)
 497 
 498       # Update dimensions of weights to match with values if possible.
 499       values, _, sample_weight = squeeze_or_expand_dimensions(
 500           values, None, sample_weight)
 501       try:
 502         # Broadcast weights if possible.
 503         sample_weight = weights_broadcast_ops.broadcast_weights(
 504             sample_weight, values)
 505       except ValueError:
 506         # Reduce values to same ndim as weight array
 507         ndim = K.ndim(values)
 508         weight_ndim = K.ndim(sample_weight)
 509         values = math_ops.reduce_mean(
 510             values, axis=list(range(weight_ndim, ndim)))
 511 
 512       num_values = math_ops.reduce_sum(sample_weight)
 513       values = math_ops.multiply(values, sample_weight)
 514     values = math_ops.reduce_sum(values)
 515 
 516     # Update state variables
 517     update_total_op = state_ops.assign_add(self.total, values)
 518     update_count_op = state_ops.assign_add(self.count, num_values)
 519     return control_flow_ops.group(update_total_op, update_count_op)
 520 
 521   def result(self):
 522     return safe_div(self.total, self.count)

 525 class MeanMetricWrapper(Mean):
 526   """Wraps a stateless metric function with the Mean metric."""
 527 
 528   def __init__(self, fn, name=None, dtype=None, **kwargs):
 529     """Creates a `MeanMetricWrapper` instance.
 530 
 531     Args:
 532       fn: The metric function to wrap, with signature
 533         `fn(y_true, y_pred, **kwargs)`.
 534       name: (Optional) string name of the metric instance.
 535       dtype: (Optional) data type of the metric result.
 536       **kwargs: The keyword arguments that are passed on to `fn`.
 537     """
 538     super(MeanMetricWrapper, self).__init__(name=name, dtype=dtype)
 539     self._fn = fn
 540     self._fn_kwargs = kwargs
 541 
 542   def update_state(self, y_true, y_pred, sample_weight=None):
 543     """Accumulates metric statistics.
 544 
 545     `y_true` and `y_pred` should have the same shape.
 546 
 547     Args:
 548       y_true: The ground truth values.
 549       y_pred: The predicted values.
 550       sample_weight: Optional weighting of each example. Defaults to 1. Can be
 551         a `Tensor` whose rank is either 0, or the same rank as `y_true`,
 552         and must be broadcastable to `y_true`.
 553 
 554     Returns:
 555       Update op.
 556     """
 557     y_true = math_ops.cast(y_true, self._dtype)
 558     y_pred = math_ops.cast(y_pred, self._dtype)
 559     y_pred, y_true, sample_weight = squeeze_or_expand_dimensions(
 560         y_pred, y_true, sample_weight)
 561 
 562     matches = self._fn(y_true, y_pred, **self._fn_kwargs)
 563     return super(MeanMetricWrapper, self).update_state(
 564         matches, sample_weight=sample_weight)
 565 
 566   def get_config(self):
 567     config = self._fn_kwargs
 568     base_config = super(MeanMetricWrapper, self).get_config()
 569     return dict(list(base_config.items()) + list(config.items()))

 637 @tf_export('keras.metrics.binary_accuracy')
 638 def binary_accuracy(y_true, y_pred, threshold=0.5):
 639   threshold = math_ops.cast(threshold, y_pred.dtype)
 640   y_pred = math_ops.cast(y_pred > threshold, y_pred.dtype)
 641   return K.mean(math_ops.equal(y_true, y_pred), axis=-1)

 644 @tf_export('keras.metrics.categorical_accuracy')
 645 def categorical_accuracy(y_true, y_pred):
 646   return math_ops.cast(
 647       math_ops.equal(
 648           math_ops.argmax(y_true, axis=-1), math_ops.argmax(y_pred, axis=-1)),
 649       K.floatx())

 652 @tf_export('keras.metrics.sparse_categorical_accuracy')
 653 def sparse_categorical_accuracy(y_true, y_pred):
 654   y_true = math_ops.reduce_max(y_true, axis=-1)
 655   y_pred = math_ops.argmax(y_pred, axis=-1)
 656 
 657   # If the expected labels are float, we need to cast the int returned by
 658   # argmax to compare.
 659   if K.dtype(y_true) == K.floatx():
 660     y_pred = math_ops.cast(y_pred, K.floatx())
 661 
 662   return math_ops.cast(math_ops.equal(y_true, y_pred), K.floatx())

 668       nn.in_top_k(y_pred, math_ops.argmax(y_true, axis=-1), k), axis=-1)
