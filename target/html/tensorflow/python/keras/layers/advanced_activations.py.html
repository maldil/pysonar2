  32 @tf_export('keras.layers.LeakyReLU')
  33 class LeakyReLU(Layer):
  34   """Leaky version of a Rectified Linear Unit.
  35 
  36   It allows a small gradient when the unit is not active:
  37   `f(x) = alpha * x for x < 0`,
  38   `f(x) = x for x >= 0`.
  39 
  40   Input shape:
  41       Arbitrary. Use the keyword argument `input_shape`
  42       (tuple of integers, does not include the samples axis)
  43       when using this layer as the first layer in a model.
  44 
  45   Output shape:
  46       Same shape as the input.
  47 
  48   Arguments:
  49       alpha: float >= 0. Negative slope coefficient.
  50 
  51   """
  52 
  53   def __init__(self, alpha=0.3, **kwargs):
  54     super(LeakyReLU, self).__init__(**kwargs)
  55     self.supports_masking = True
  56     self.alpha = K.cast_to_floatx(alpha)
  57 
  58   def call(self, inputs):
  59     return K.relu(inputs, alpha=self.alpha)
  60 
  61   def get_config(self):
  62     config = {'alpha': float(self.alpha)}
  63     base_config = super(LeakyReLU, self).get_config()
  64     return dict(list(base_config.items()) + list(config.items()))
  65 
  66   @tf_utils.shape_type_conversion
  67   def compute_output_shape(self, input_shape):
  68     return input_shape

  71 @tf_export('keras.layers.PReLU')
  72 class PReLU(Layer):
  73   """Parametric Rectified Linear Unit.
  74 
  75   It follows:
  76   `f(x) = alpha * x for x < 0`,
  77   `f(x) = x for x >= 0`,
  78   where `alpha` is a learned array with the same shape as x.
  79 
  80   Input shape:
  81       Arbitrary. Use the keyword argument `input_shape`
  82       (tuple of integers, does not include the samples axis)
  83       when using this layer as the first layer in a model.
  84 
  85   Output shape:
  86       Same shape as the input.
  87 
  88   Arguments:
  89       alpha_initializer: initializer function for the weights.
  90       alpha_regularizer: regularizer for the weights.
  91       alpha_constraint: constraint for the weights.
  92       shared_axes: the axes along which to share learnable
  93           parameters for the activation function.
  94           For example, if the incoming feature maps
  95           are from a 2D convolution
  96           with output shape `(batch, height, width, channels)`,
  97           and you wish to share parameters across space
  98           so that each filter only has one set of parameters,
  99           set `shared_axes=[1, 2]`.
 100 
 101   """
 102 
 103   def __init__(self,
 104                alpha_initializer='zeros',
 105                alpha_regularizer=None,
 106                alpha_constraint=None,
 107                shared_axes=None,
 108                **kwargs):
 109     super(PReLU, self).__init__(**kwargs)
 110     self.supports_masking = True
 111     self.alpha_initializer = initializers.get(alpha_initializer)
 112     self.alpha_regularizer = regularizers.get(alpha_regularizer)
 113     self.alpha_constraint = constraints.get(alpha_constraint)
 114     if shared_axes is None:
 115       self.shared_axes = None
 116     elif not isinstance(shared_axes, (list, tuple)):
 117       self.shared_axes = [shared_axes]
 118     else:
 119       self.shared_axes = list(shared_axes)
 120 
 121   @tf_utils.shape_type_conversion
 122   def build(self, input_shape):
 123     param_shape = list(input_shape[1:])
 124     self.param_broadcast = [False] * len(param_shape)
 125     if self.shared_axes is not None:
 126       for i in self.shared_axes:
 127         param_shape[i - 1] = 1
 128         self.param_broadcast[i - 1] = True
 129     self.alpha = self.add_weight(
 130         shape=param_shape,
 131         name='alpha',
 132         initializer=self.alpha_initializer,
 133         regularizer=self.alpha_regularizer,
 134         constraint=self.alpha_constraint)
 135     # Set input spec
 136     axes = {}
 137     if self.shared_axes:
 138       for i in range(1, len(input_shape)):
 139         if i not in self.shared_axes:
 140           axes[i] = input_shape[i]
 141     self.input_spec = InputSpec(ndim=len(input_shape), axes=axes)
 142     self.built = True
 143 
 144   def call(self, inputs, mask=None):
 145     pos = K.relu(inputs)
 146     if K.backend() == 'theano':
 147       neg = (
 148           K.pattern_broadcast(self.alpha, self.param_broadcast) *
 149           (inputs - math_ops.abs(inputs)) * 0.5)
 150     else:
 151       neg = -self.alpha * K.relu(-inputs)
 152     return pos + neg
 153 
 154   def get_config(self):
 155     config = {
 156         'alpha_initializer': initializers.serialize(self.alpha_initializer),
 157         'alpha_regularizer': regularizers.serialize(self.alpha_regularizer),
 158         'alpha_constraint': constraints.serialize(self.alpha_constraint),
 159         'shared_axes': self.shared_axes
 160     }
 161     base_config = super(PReLU, self).get_config()
 162     return dict(list(base_config.items()) + list(config.items()))
 163 
 164   @tf_utils.shape_type_conversion
 165   def compute_output_shape(self, input_shape):
 166     return input_shape

 169 @tf_export('keras.layers.ELU')
 170 class ELU(Layer):
 171   """Exponential Linear Unit.
 172 
 173   It follows:
 174   `f(x) =  alpha * (exp(x) - 1.) for x < 0`,
 175   `f(x) = x for x >= 0`.
 176 
 177   Input shape:
 178       Arbitrary. Use the keyword argument `input_shape`
 179       (tuple of integers, does not include the samples axis)
 180       when using this layer as the first layer in a model.
 181 
 182   Output shape:
 183       Same shape as the input.
 184 
 185   Arguments:
 186       alpha: scale for the negative factor.
 187 
 188   """
 189 
 190   def __init__(self, alpha=1.0, **kwargs):
 191     super(ELU, self).__init__(**kwargs)
 192     self.supports_masking = True
 193     self.alpha = K.cast_to_floatx(alpha)
 194 
 195   def call(self, inputs):
 196     return K.elu(inputs, self.alpha)
 197 
 198   def get_config(self):
 199     config = {'alpha': float(self.alpha)}
 200     base_config = super(ELU, self).get_config()
 201     return dict(list(base_config.items()) + list(config.items()))
 202 
 203   @tf_utils.shape_type_conversion
 204   def compute_output_shape(self, input_shape):
 205     return input_shape

 208 @tf_export('keras.layers.ThresholdedReLU')
 209 class ThresholdedReLU(Layer):
 210   """Thresholded Rectified Linear Unit.
 211 
 212   It follows:
 213   `f(x) = x for x > theta`,
 214   `f(x) = 0 otherwise`.
 215 
 216   Input shape:
 217       Arbitrary. Use the keyword argument `input_shape`
 218       (tuple of integers, does not include the samples axis)
 219       when using this layer as the first layer in a model.
 220 
 221   Output shape:
 222       Same shape as the input.
 223 
 224   Arguments:
 225       theta: float >= 0. Threshold location of activation.
 226 
 227   """
 228 
 229   def __init__(self, theta=1.0, **kwargs):
 230     super(ThresholdedReLU, self).__init__(**kwargs)
 231     self.supports_masking = True
 232     self.theta = K.cast_to_floatx(theta)
 233 
 234   def call(self, inputs, mask=None):
 235     return inputs * math_ops.cast(
 236         math_ops.greater(inputs, self.theta), K.floatx())
 237 
 238   def get_config(self):
 239     config = {'theta': float(self.theta)}
 240     base_config = super(ThresholdedReLU, self).get_config()
 241     return dict(list(base_config.items()) + list(config.items()))
 242 
 243   @tf_utils.shape_type_conversion
 244   def compute_output_shape(self, input_shape):
 245     return input_shape

 282 @tf_export('keras.layers.ReLU')
 283 class ReLU(Layer):
 284   """Rectified Linear Unit activation function.
 285 
 286   With default values, it returns element-wise `max(x, 0)`.
 287 
 288   Otherwise, it follows:
 289   `f(x) = max_value` for `x >= max_value`,
 290   `f(x) = x` for `threshold <= x < max_value`,
 291   `f(x) = negative_slope * (x - threshold)` otherwise.
 292 
 293   Input shape:
 294       Arbitrary. Use the keyword argument `input_shape`
 295       (tuple of integers, does not include the samples axis)
 296       when using this layer as the first layer in a model.
 297 
 298   Output shape:
 299       Same shape as the input.
 300 
 301   Arguments:
 302       max_value: float >= 0. Maximum activation value.
 303       negative_slope: float >= 0. Negative slope coefficient.
 304       threshold: float. Threshold value for thresholded activation.
 305   """
 306 
 307   def __init__(self, max_value=None, negative_slope=0, threshold=0, **kwargs):
 308     super(ReLU, self).__init__(**kwargs)
 309     if max_value is not None and max_value < 0.:
 310       raise ValueError('max_value of Relu layer '
 311                        'cannot be negative value: ' + str(max_value))
 312     if negative_slope < 0.:
 313       raise ValueError('negative_slope of Relu layer '
 314                        'cannot be negative value: ' + str(negative_slope))
 315 
 316     self.support_masking = True
 317     if max_value is not None:
 318       max_value = K.cast_to_floatx(max_value)
 319     self.max_value = max_value
 320     self.negative_slope = K.cast_to_floatx(negative_slope)
 321     self.threshold = K.cast_to_floatx(threshold)
 322 
 323   def call(self, inputs):
 324     # alpha is used for leaky relu slope in activations instead of
 325     # negative_slope.
 326     return K.relu(inputs,
 327                   alpha=self.negative_slope,
 328                   max_value=self.max_value,
 329                   threshold=self.threshold)
 330 
 331   def get_config(self):
 332     config = {
 333         'max_value': self.max_value,
 334         'negative_slope': self.negative_slope,
 335         'threshold': self.threshold
 336     }
 337     base_config = super(ReLU, self).get_config()
 338     return dict(list(base_config.items()) + list(config.items()))
 339 
 340   @tf_utils.shape_type_conversion
 341   def compute_output_shape(self, input_shape):
 342     return input_shape
