  43 @tf_export('keras.layers.BatchNormalization')
  44 class BatchNormalization(Layer):
  45   """Batch normalization layer (Ioffe and Szegedy, 2014).
  46 
  47   Normalize the activations of the previous layer at each batch,
  48   i.e. applies a transformation that maintains the mean activation
  49   close to 0 and the activation standard deviation close to 1.
  50 
  51   Arguments:
  52     axis: Integer, the axis that should be normalized
  53         (typically the features axis).
  54         For instance, after a `Conv2D` layer with
  55         `data_format="channels_first"`,
  56         set `axis=1` in `BatchNormalization`.
  57     momentum: Momentum for the moving average.
  58     epsilon: Small float added to variance to avoid dividing by zero.
  59     center: If True, add offset of `beta` to normalized tensor.
  60         If False, `beta` is ignored.
  61     scale: If True, multiply by `gamma`.
  62         If False, `gamma` is not used.
  63         When the next layer is linear (also e.g. `nn.relu`),
  64         this can be disabled since the scaling
  65         will be done by the next layer.
  66     beta_initializer: Initializer for the beta weight.
  67     gamma_initializer: Initializer for the gamma weight.
  68     moving_mean_initializer: Initializer for the moving mean.
  69     moving_variance_initializer: Initializer for the moving variance.
  70     beta_regularizer: Optional regularizer for the beta weight.
  71     gamma_regularizer: Optional regularizer for the gamma weight.
  72     beta_constraint: Optional constraint for the beta weight.
  73     gamma_constraint: Optional constraint for the gamma weight.
  74     renorm: Whether to use Batch Renormalization
  75       (https://arxiv.org/abs/1702.03275). This adds extra variables during
  76       training. The inference is the same for either value of this parameter.
  77     renorm_clipping: A dictionary that may map keys 'rmax', 'rmin', 'dmax' to
  78       scalar `Tensors` used to clip the renorm correction. The correction
  79       `(r, d)` is used as `corrected_value = normalized_value * r + d`, with
  80       `r` clipped to [rmin, rmax], and `d` to [-dmax, dmax]. Missing rmax, rmin,
  81       dmax are set to inf, 0, inf, respectively.
  82     renorm_momentum: Momentum used to update the moving means and standard
  83       deviations with renorm. Unlike `momentum`, this affects training
  84       and should be neither too small (which would add noise) nor too large
  85       (which would give stale estimates). Note that `momentum` is still applied
  86       to get the means and variances for inference.
  87     fused: if `None` or `True`, use a faster, fused implementation if possible.
  88       If `False`, use the system recommended implementation.
  89     trainable: Boolean, if `True` also add variables to the graph collection
  90       `GraphKeys.TRAINABLE_VARIABLES` (see tf.Variable).
  91     virtual_batch_size: An `int`. By default, `virtual_batch_size` is `None`,
  92       which means batch normalization is performed across the whole batch. When
  93       `virtual_batch_size` is not `None`, instead perform "Ghost Batch
  94       Normalization", which creates virtual sub-batches which are each
  95       normalized separately (with shared gamma, beta, and moving statistics).
  96       Must divide the actual batch size during execution.
  97     adjustment: A function taking the `Tensor` containing the (dynamic) shape of
  98       the input tensor and returning a pair (scale, bias) to apply to the
  99       normalized values (before gamma and beta), only during training. For
 100       example, if axis==-1,
 101         `adjustment = lambda shape: (
 102           tf.random_uniform(shape[-1:], 0.93, 1.07),
 103           tf.random_uniform(shape[-1:], -0.1, 0.1))`
 104       will scale the normalized value by up to 7% up or down, then shift the
 105       result by up to 0.1 (with independent scaling and bias for each feature
 106       but shared across all examples), and finally apply gamma and/or beta. If
 107       `None`, no adjustment is applied. Cannot be specified if
 108       virtual_batch_size is specified.
 109 
 110   Input shape:
 111       Arbitrary. Use the keyword argument `input_shape`
 112       (tuple of integers, does not include the samples axis)
 113       when using this layer as the first layer in a model.
 114 
 115   Output shape:
 116       Same shape as input.
 117 
 118   References:
 119       - [Batch Normalization: Accelerating Deep Network Training by Reducing
 120         Internal Covariate Shift](https://arxiv.org/abs/1502.03167)
 121   """
 122 
 123   def __init__(self,
 124                axis=-1,
 125                momentum=0.99,
 126                epsilon=1e-3,
 127                center=True,
 128                scale=True,
 129                beta_initializer='zeros',
 130                gamma_initializer='ones',
 131                moving_mean_initializer='zeros',
 132                moving_variance_initializer='ones',
 133                beta_regularizer=None,
 134                gamma_regularizer=None,
 135                beta_constraint=None,
 136                gamma_constraint=None,
 137                renorm=False,
 138                renorm_clipping=None,
 139                renorm_momentum=0.99,
 140                fused=None,
 141                trainable=True,
 142                virtual_batch_size=None,
 143                adjustment=None,
 144                name=None,
 145                **kwargs):
 146     super(BatchNormalization, self).__init__(
 147         name=name, trainable=trainable, **kwargs)
 148     if isinstance(axis, list):
 149       self.axis = axis[:]
 150     else:
 151       self.axis = axis
 152     self.momentum = momentum
 153     self.epsilon = epsilon
 154     self.center = center
 155     self.scale = scale
 156     self.beta_initializer = initializers.get(beta_initializer)
 157     self.gamma_initializer = initializers.get(gamma_initializer)
 158     self.moving_mean_initializer = initializers.get(moving_mean_initializer)
 159     self.moving_variance_initializer = initializers.get(
 160         moving_variance_initializer)
 161     self.beta_regularizer = regularizers.get(beta_regularizer)
 162     self.gamma_regularizer = regularizers.get(gamma_regularizer)
 163     self.beta_constraint = constraints.get(beta_constraint)
 164     self.gamma_constraint = constraints.get(gamma_constraint)
 165     self.renorm = renorm
 166     self.virtual_batch_size = virtual_batch_size
 167     self.adjustment = adjustment
 168     if fused is None:
 169       fused = True
 170     self.supports_masking = True
 171 
 172     self.fused = fused
 173     self._bessels_correction_test_only = True
 174 
 175     if renorm:
 176       renorm_clipping = renorm_clipping or {}
 177       keys = ['rmax', 'rmin', 'dmax']
 178       if set(renorm_clipping) - set(keys):
 179         raise ValueError('renorm_clipping %s contains keys not in %s' %
 180                          (renorm_clipping, keys))
 181       self.renorm_clipping = renorm_clipping
 182       self.renorm_momentum = renorm_momentum

 184   def build(self, input_shape):
 185     input_shape = tensor_shape.TensorShape(input_shape)
 186     if not input_shape.ndims:
 187       raise ValueError('Input has undefined rank:', input_shape)
 188     ndims = len(input_shape)
 189 
 190     # Convert axis to list and resolve negatives
 191     if isinstance(self.axis, int):
 192       self.axis = [self.axis]
 193 
 194     if not isinstance(self.axis, list):
 195       raise TypeError('axis must be int or list, type given: %s'
 196                       % type(self.axis))
 197 
 198     for idx, x in enumerate(self.axis):
 199       if x < 0:
 200         self.axis[idx] = ndims + x
 201 
 202     # Validate axes
 203     for x in self.axis:
 204       if x < 0 or x >= ndims:
 205         raise ValueError('Invalid axis: %d' % x)
 206     if len(self.axis) != len(set(self.axis)):
 207       raise ValueError('Duplicate axis: %s' % self.axis)
 208 
 209     if self.virtual_batch_size is not None:
 210       if self.virtual_batch_size <= 0:
 211         raise ValueError('virtual_batch_size must be a positive integer that '
 212                          'divides the true batch size of the input Tensor')
 213       # If using virtual batches, the first dimension must be the batch
 214       # dimension and cannot be the batch norm axis
 215       if 0 in self.axis:
 216         raise ValueError('When using virtual_batch_size, the batch dimension '
 217                          'must be 0 and thus axis cannot include 0')
 218       if self.adjustment is not None:
 219         raise ValueError('When using virtual_batch_size, adjustment cannot '
 220                          'be specified')
 221 
 222     if self.fused:
 223       # Currently fused batch norm doesn't support renorm. It also only supports
 224       # an input tensor of rank 4 and a channel dimension on axis 1 or 3.
 225       # TODO(yaozhang): if input is not 4D, reshape it to 4D and reshape the
 226       # output back to its original shape accordingly.
 227       self.fused = (not self.renorm and
 228                     ndims == 4 and
 229                     self.axis in [[1], [3]] and
 230                     self.virtual_batch_size is None and
 231                     self.adjustment is None)
 232       # TODO(chrisying): fused batch norm is currently not supported for
 233       # multi-axis batch norm and by extension virtual batches. In some cases,
 234       # it might be possible to use fused batch norm but would require reshaping
 235       # the Tensor to 4D with the axis in 1 or 3 (preferred 1) which is
 236       # particularly tricky. A compromise might be to just support the most
 237       # common use case (turning 5D w/ virtual batch to NCHW)
 238 
 239     if self.fused:
 240       if self.axis == [1]:
 241         self._data_format = 'NCHW'
 242       elif self.axis == [3]:
 243         self._data_format = 'NHWC'
 244       else:
 245         raise ValueError('Unsupported axis, fused batch norm only supports '
 246                          'axis == [1] or axis == [3]')
 247 
 248     # Raise parameters of fp16 batch norm to fp32
 249     if self.dtype == dtypes.float16 or self.dtype == dtypes.bfloat16:
 250       param_dtype = dtypes.float32
 251     else:
 252       param_dtype = self.dtype or dtypes.float32
 253 
 254     axis_to_dim = {x: input_shape[x].value for x in self.axis}
 255     for x in axis_to_dim:
 256       if axis_to_dim[x] is None:
 257         raise ValueError('Input has undefined `axis` dimension. Input shape: ',
 258                          input_shape)
 259     self.input_spec = InputSpec(ndim=ndims, axes=axis_to_dim)
 260 
 261     if len(axis_to_dim) == 1 and self.virtual_batch_size is None:
 262       # Single axis batch norm (most common/default use-case)
 263       param_shape = (list(axis_to_dim.values())[0],)
 264     else:
 265       # Parameter shape is the original shape but with 1 in all non-axis dims
 266       param_shape = [axis_to_dim[i] if i in axis_to_dim
 267                      else 1 for i in range(ndims)]
 268       if self.virtual_batch_size is not None:
 269         # When using virtual batches, add an extra dim at index 1
 270         param_shape.insert(1, 1)
 271         for idx, x in enumerate(self.axis):
 272           self.axis[idx] = x + 1      # Account for added dimension
 273 
 274     if self.scale:
 275       self.gamma = self.add_weight(
 276           name='gamma',
 277           shape=param_shape,
 278           dtype=param_dtype,
 279           initializer=self.gamma_initializer,
 280           regularizer=self.gamma_regularizer,
 281           constraint=self.gamma_constraint,
 282           trainable=True)
 283     else:
 284       self.gamma = None
 285       if self.fused:
 286         self._gamma_const = array_ops.constant(
 287             1.0, dtype=param_dtype, shape=param_shape)
 288 
 289     if self.center:
 290       self.beta = self.add_weight(
 291           name='beta',
 292           shape=param_shape,
 293           dtype=param_dtype,
 294           initializer=self.beta_initializer,
 295           regularizer=self.beta_regularizer,
 296           constraint=self.beta_constraint,
 297           trainable=True)
 298     else:
 299       self.beta = None
 300       if self.fused:
 301         self._beta_const = array_ops.constant(
 302             0.0, dtype=param_dtype, shape=param_shape)
 303 
 304     try:
 305       # Disable variable partitioning when creating the moving mean and variance
 306       if hasattr(self, '_scope') and self._scope:
 307         partitioner = self._scope.partitioner
 308         self._scope.set_partitioner(None)
 309       else:
 310         partitioner = None
 311       self.moving_mean = self.add_weight(
 312           name='moving_mean',
 313           shape=param_shape,
 314           dtype=param_dtype,
 315           initializer=self.moving_mean_initializer,
 316           synchronization=tf_variables.VariableSynchronization.ON_READ,
 317           trainable=False,
 318           aggregation=tf_variables.VariableAggregation.MEAN)
 319 
 320       self.moving_variance = self.add_weight(
 321           name='moving_variance',
 322           shape=param_shape,
 323           dtype=param_dtype,
 324           initializer=self.moving_variance_initializer,
 325           synchronization=tf_variables.VariableSynchronization.ON_READ,
 326           trainable=False,
 327           aggregation=tf_variables.VariableAggregation.MEAN)
 328 
 329       if self.renorm:
 330         # Create variables to maintain the moving mean and standard deviation.
 331         # These are used in training and thus are different from the moving
 332         # averages above. The renorm variables are colocated with moving_mean
 333         # and moving_variance.
 334         # NOTE: below, the outer `with device` block causes the current device
 335         # stack to be cleared. The nested ones use a `lambda` to set the desired
 336         # device and ignore any devices that may be set by the custom getter.
 337         def _renorm_variable(name, shape):
 338           var = self.add_weight(
 339               name=name,
 340               shape=shape,
 341               dtype=param_dtype,
 342               initializer=init_ops.zeros_initializer(),
 343               synchronization=tf_variables.VariableSynchronization.ON_READ,
 344               trainable=False,
 345               aggregation=tf_variables.VariableAggregation.MEAN)
 346           return var
 347 
 348         with distribution_strategy_context.get_distribution_strategy(
 349         ).colocate_vars_with(self.moving_mean):
 350           self.renorm_mean = _renorm_variable('renorm_mean', param_shape)
 351           self.renorm_mean_weight = _renorm_variable('renorm_mean_weight', ())
 352         # We initialize renorm_stddev to 0, and maintain the (0-initialized)
 353         # renorm_stddev_weight. This allows us to (1) mix the average
 354         # stddev with the minibatch stddev early in training, and (2) compute
 355         # the unbiased average stddev by dividing renorm_stddev by the weight.
 356         with distribution_strategy_context.get_distribution_strategy(
 357         ).colocate_vars_with(self.moving_variance):
 358           self.renorm_stddev = _renorm_variable('renorm_stddev', param_shape)
 359           self.renorm_stddev_weight = _renorm_variable('renorm_stddev_weight',
 360                                                        ())
 361     finally:
 362       if partitioner:
 363         self._scope.set_partitioner(partitioner)
 364     self.built = True
 365 
 366   def _assign_moving_average(self, variable, value, momentum):
 367     with ops.name_scope(None, 'AssignMovingAvg',
 368                         [variable, value, momentum]) as scope:
 369       with ops.colocate_with(variable):
 370         decay = ops.convert_to_tensor(1.0 - momentum, name='decay')
 371         if decay.dtype != variable.dtype.base_dtype:
 372           decay = math_ops.cast(decay, variable.dtype.base_dtype)
 373         update_delta = (variable - math_ops.cast(value, variable.dtype)) * decay
 374         return state_ops.assign_sub(variable, update_delta, name=scope)
 375 
 376   def _fused_batch_norm(self, inputs, training):
 377     """Returns the output of fused batch norm."""
 378     beta = self.beta if self.center else self._beta_const
 379     gamma = self.gamma if self.scale else self._gamma_const
 380 
 381     def _fused_batch_norm_training():
 382       return nn.fused_batch_norm(
 383           inputs,
 384           gamma,
 385           beta,
 386           epsilon=self.epsilon,
 387           data_format=self._data_format)
 388 
 389     def _fused_batch_norm_inference():
 390       return nn.fused_batch_norm(
 391           inputs,
 392           gamma,
 393           beta,
 394           mean=self.moving_mean,
 395           variance=self.moving_variance,
 396           epsilon=self.epsilon,
 397           is_training=False,
 398           data_format=self._data_format)
 399 
 400     output, mean, variance = tf_utils.smart_cond(
 401         training, _fused_batch_norm_training, _fused_batch_norm_inference)
 402     if not self._bessels_correction_test_only:
 403       # Remove Bessel's correction to be consistent with non-fused batch norm.
 404       # Note that the variance computed by fused batch norm is
 405       # with Bessel's correction.
 406       sample_size = math_ops.cast(
 407           array_ops.size(inputs) / array_ops.size(variance), variance.dtype)
 408       factor = (sample_size - math_ops.cast(1.0, variance.dtype)) / sample_size
 409       variance *= factor
 410 
 411     training_value = tf_utils.constant_value(training)
 412     if training_value is None:
 413       momentum = tf_utils.smart_cond(training,
 414                                      lambda: self.momentum,
 415                                      lambda: 1.0)
 416     else:
 417       momentum = ops.convert_to_tensor(self.momentum)
 418     if training_value or training_value is None:
 419       mean_update = self._assign_moving_average(self.moving_mean, mean,
 420                                                 momentum)
 421       variance_update = self._assign_moving_average(self.moving_variance,
 422                                                     variance, momentum)
 423       self.add_update(mean_update, inputs=True)
 424       self.add_update(variance_update, inputs=True)
 425 
 426     return output
 427 
 428   def _renorm_correction_and_moments(self, mean, variance, training):
 429     """Returns the correction and update values for renorm."""
 430     stddev = math_ops.sqrt(variance + self.epsilon)
 431     # Compute the average mean and standard deviation, as if they were
 432     # initialized with this batch's moments.
 433     mixed_renorm_mean = (self.renorm_mean +
 434                          (1. - self.renorm_mean_weight) * mean)
 435     mixed_renorm_stddev = (self.renorm_stddev +
 436                            (1. - self.renorm_stddev_weight) * stddev)
 437     # Compute the corrections for batch renorm.
 438     r = stddev / mixed_renorm_stddev
 439     d = (mean - mixed_renorm_mean) / mixed_renorm_stddev
 440     # Ensure the corrections use pre-update moving averages.
 441     with ops.control_dependencies([r, d]):
 442       mean = array_ops.identity(mean)
 443       stddev = array_ops.identity(stddev)
 444     rmin, rmax, dmax = [self.renorm_clipping.get(key)
 445                         for key in ['rmin', 'rmax', 'dmax']]
 446     if rmin is not None:
 447       r = math_ops.maximum(r, rmin)
 448     if rmax is not None:
 449       r = math_ops.minimum(r, rmax)
 450     if dmax is not None:
 451       d = math_ops.maximum(d, -dmax)
 452       d = math_ops.minimum(d, dmax)
 453     # When not training, use r=1, d=0.
 454     r = tf_utils.smart_cond(training, lambda: r, lambda: array_ops.ones_like(r))
 455     d = tf_utils.smart_cond(training,
 456                             lambda: d,
 457                             lambda: array_ops.zeros_like(d))
 458 
 459     def _update_renorm_variable(var, weight, value):
 460       """Updates a moving average and weight, returns the unbiased value."""
 461       value = array_ops.identity(value)
 462       def _do_update():
 463         """Updates the var and weight, returns their updated ratio."""
 464         # Update the variables without zero debiasing. The debiasing will be
 465         # accomplished by dividing the exponential moving average by the weight.
 466         # For example, after a single update, the moving average would be
 467         # (1-decay) * value. and the weight will be 1-decay, with their ratio
 468         # giving the value.
 469         # Make sure the weight is not updated until before r and d computation.
 470         with ops.control_dependencies([value]):
 471           weight_value = array_ops.constant(1., dtype=weight.dtype)
 472         new_var = self._assign_moving_average(var, value, self.renorm_momentum)
 473         new_weight = self._assign_moving_average(weight, weight_value,
 474                                                  self.renorm_momentum)
 475         # TODO(yuefengz): the updates to var and weighted can not be batched
 476         # together if we fetch their updated values here. Consider calculating
 477         # new values and delaying the updates.
 478         return new_var / new_weight
 479 
 480       def _fake_update():
 481         return array_ops.identity(var)
 482       return tf_utils.smart_cond(training, _do_update, _fake_update)
 483 
 484     # TODO(yuefengz): colocate the operations
 485     new_mean = _update_renorm_variable(self.renorm_mean,
 486                                        self.renorm_mean_weight, mean)
 487     new_stddev = _update_renorm_variable(self.renorm_stddev,
 488                                          self.renorm_stddev_weight, stddev)
 489     # Make sqrt(moving_variance + epsilon) = new_stddev.
 490     new_variance = math_ops.square(new_stddev) - self.epsilon
 491 
 492     return (r, d, new_mean, new_variance)
 493 
 494   def call(self, inputs, training=None):
 495     original_training_value = training
 496     if training is None:
 497       training = K.learning_phase()
 498 
 499     in_eager_mode = context.executing_eagerly()
 500     if self.virtual_batch_size is not None:
 501       # Virtual batches (aka ghost batches) can be simulated by reshaping the
 502       # Tensor and reusing the existing batch norm implementation
 503       original_shape = [-1] + inputs.shape.as_list()[1:]
 504       expanded_shape = [self.virtual_batch_size, -1] + original_shape[1:]
 505 
 506       # Will cause errors if virtual_batch_size does not divide the batch size
 507       inputs = array_ops.reshape(inputs, expanded_shape)
 508 
 509       def undo_virtual_batching(outputs):
 510         outputs = array_ops.reshape(outputs, original_shape)
 511         return outputs
 512 
 513     if self.fused:
 514       outputs = self._fused_batch_norm(inputs, training=training)
 515       if self.virtual_batch_size is not None:
 516         # Currently never reaches here since fused_batch_norm does not support
 517         # virtual batching
 518         outputs = undo_virtual_batching(outputs)
 519       if not context.executing_eagerly() and original_training_value is None:
 520         outputs._uses_learning_phase = True  # pylint: disable=protected-access
 521       return outputs
 522 
 523     # Compute the axes along which to reduce the mean / variance
 524     input_shape = inputs.get_shape()
 525     ndims = len(input_shape)
 526     reduction_axes = [i for i in range(ndims) if i not in self.axis]
 527     if self.virtual_batch_size is not None:
 528       del reduction_axes[1]     # Do not reduce along virtual batch dim
 529 
 530     # Broadcasting only necessary for single-axis batch norm where the axis is
 531     # not the last dimension
 532     broadcast_shape = [1] * ndims
 533     broadcast_shape[self.axis[0]] = input_shape[self.axis[0]].value
 534     def _broadcast(v):
 535       if (v is not None and
 536           len(v.get_shape()) != ndims and
 537           reduction_axes != list(range(ndims - 1))):
 538         return array_ops.reshape(v, broadcast_shape)
 539       return v
 540 
 541     scale, offset = _broadcast(self.gamma), _broadcast(self.beta)
 542 
 543     def _compose_transforms(scale, offset, then_scale, then_offset):
 544       if then_scale is not None:
 545         scale *= then_scale
 546         offset *= then_scale
 547       if then_offset is not None:
 548         offset += then_offset
 549       return (scale, offset)
 550 
 551     # Determine a boolean value for `training`: could be True, False, or None.
 552     training_value = tf_utils.constant_value(training)
 553     if training_value is not False:
 554       if self.adjustment:
 555         adj_scale, adj_bias = self.adjustment(array_ops.shape(inputs))
 556         # Adjust only during training.
 557         adj_scale = tf_utils.smart_cond(training,
 558                                         lambda: adj_scale,
 559                                         lambda: array_ops.ones_like(adj_scale))
 560         adj_bias = tf_utils.smart_cond(training,
 561                                        lambda: adj_bias,
 562                                        lambda: array_ops.zeros_like(adj_bias))
 563         scale, offset = _compose_transforms(adj_scale, adj_bias, scale, offset)
 564 
 565       # Some of the computations here are not necessary when training==False
 566       # but not a constant. However, this makes the code simpler.
 567       keep_dims = self.virtual_batch_size is not None or len(self.axis) > 1
 568       mean, variance = nn.moments(inputs, reduction_axes, keep_dims=keep_dims)
 569 
 570       moving_mean = self.moving_mean
 571       moving_variance = self.moving_variance
 572 
 573       mean = tf_utils.smart_cond(training,
 574                                  lambda: mean,
 575                                  lambda: moving_mean)
 576       variance = tf_utils.smart_cond(training,
 577                                      lambda: variance,
 578                                      lambda: moving_variance)
 579 
 580       if self.virtual_batch_size is not None:
 581         # This isn't strictly correct since in ghost batch norm, you are
 582         # supposed to sequentially update the moving_mean and moving_variance
 583         # with each sub-batch. However, since the moving statistics are only
 584         # used during evaluation, it is more efficient to just update in one
 585         # step and should not make a significant difference in the result.
 586         new_mean = math_ops.reduce_mean(mean, axis=1, keepdims=True)
 587         new_variance = math_ops.reduce_mean(variance, axis=1, keepdims=True)
 588       else:
 589         new_mean, new_variance = mean, variance
 590 
 591       if self.renorm:
 592         r, d, new_mean, new_variance = self._renorm_correction_and_moments(
 593             new_mean, new_variance, training)
 594         # When training, the normalized values (say, x) will be transformed as
 595         # x * gamma + beta without renorm, and (x * r + d) * gamma + beta
 596         # = x * (r * gamma) + (d * gamma + beta) with renorm.
 597         r = _broadcast(array_ops.stop_gradient(r, name='renorm_r'))
 598         d = _broadcast(array_ops.stop_gradient(d, name='renorm_d'))
 599         scale, offset = _compose_transforms(r, d, scale, offset)
 600 
 601       def _do_update(var, value):
 602         if in_eager_mode and not self.trainable:
 603           return
 604 
 605         return self._assign_moving_average(var, value, self.momentum)
 606 
 607       mean_update = tf_utils.smart_cond(
 608           training,
 609           lambda: _do_update(self.moving_mean, new_mean),
 610           lambda: self.moving_mean)
 611       variance_update = tf_utils.smart_cond(
 612           training,
 613           lambda: _do_update(self.moving_variance, new_variance),
 614           lambda: self.moving_variance)
 615       if not context.executing_eagerly():
 616         self.add_update(mean_update, inputs=True)
 617         self.add_update(variance_update, inputs=True)
 618 
 619     else:
 620       mean, variance = self.moving_mean, self.moving_variance
 621 
 622     mean = math_ops.cast(mean, inputs.dtype)
 623     variance = math_ops.cast(variance, inputs.dtype)
 624     if offset is not None:
 625       offset = math_ops.cast(offset, inputs.dtype)
 626     outputs = nn.batch_normalization(inputs,
 627                                      _broadcast(mean),
 628                                      _broadcast(variance),
 629                                      offset,
 630                                      scale,
 631                                      self.epsilon)
 632     # If some components of the shape got lost due to adjustments, fix that.
 633     outputs.set_shape(input_shape)
 634 
 635     if self.virtual_batch_size is not None:
 636       outputs = undo_virtual_batching(outputs)
 637     if not context.executing_eagerly() and original_training_value is None:
 638       outputs._uses_learning_phase = True  # pylint: disable=protected-access
 639     return outputs
 640 
 641   def compute_output_shape(self, input_shape):
 642     return input_shape

 645     config = {
 646         'axis': self.axis,
 647         'momentum': self.momentum,
 648         'epsilon': self.epsilon,
 649         'center': self.center,
 650         'scale': self.scale,
 651         'beta_initializer': initializers.serialize(self.beta_initializer),
 652         'gamma_initializer': initializers.serialize(self.gamma_initializer),
 653         'moving_mean_initializer':
 654             initializers.serialize(self.moving_mean_initializer),
 655         'moving_variance_initializer':
 656             initializers.serialize(self.moving_variance_initializer),
 657         'beta_regularizer': regularizers.serialize(self.beta_regularizer),
 658         'gamma_regularizer': regularizers.serialize(self.gamma_regularizer),
 659         'beta_constraint': constraints.serialize(self.beta_constraint),
 660         'gamma_constraint': constraints.serialize(self.gamma_constraint)
 661     }
 662     # Only add TensorFlow-specific parameters if they are set, so as to preserve
 663     # model compatibility with external Keras.
 664     if self.renorm:
 665       config['renorm'] = True
 666       config['renorm_clipping'] = self.renorm_clipping
 667       config['renorm_momentum'] = self.renorm_momentum
 668     if self.virtual_batch_size is not None:
 669       config['virtual_batch_size'] = self.virtual_batch_size
 670     # Note: adjustment is not serializable.
 671     if self.adjustment is not None:
 672       logging.warning('The `adjustment` function of this `BatchNormalization` '
 673                       'layer cannot be serialized and has been omitted from '
 674                       'the layer config. It will not be included when '
 675                       're-creating the layer from the saved config.')
 676     base_config = super(BatchNormalization, self).get_config()
 677     return dict(list(base_config.items()) + list(config.items()))
