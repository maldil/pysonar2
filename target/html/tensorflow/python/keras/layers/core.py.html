  51 @tf_export('keras.layers.Masking')
  52 class Masking(Layer):
  53   """Masks a sequence by using a mask value to skip timesteps.
  54 
  55   For each timestep in the input tensor (dimension #1 in the tensor),
  56   if all values in the input tensor at that timestep
  57   are equal to `mask_value`, then the timestep will be masked (skipped)
  58   in all downstream layers (as long as they support masking).
  59 
  60   If any downstream layer does not support masking yet receives such
  61   an input mask, an exception will be raised.
  62 
  63   Example:
  64 
  65   Consider a Numpy data array `x` of shape `(samples, timesteps, features)`,
  66   to be fed to an LSTM layer.
  67   You want to mask timestep #3 and #5 because you lack data for
  68   these timesteps. You can:
  69 
  70       - set `x[:, 3, :] = 0.` and `x[:, 5, :] = 0.`
  71       - insert a `Masking` layer with `mask_value=0.` before the LSTM layer:
  72 
  73   ```python
  74       model = Sequential()
  75       model.add(Masking(mask_value=0., input_shape=(timesteps, features)))
  76       model.add(LSTM(32))
  77   ```
  78   """
  79 
  80   def __init__(self, mask_value=0., **kwargs):
  81     super(Masking, self).__init__(**kwargs)
  82     self.supports_masking = True
  83     self.mask_value = mask_value
  84 
  85   def compute_mask(self, inputs, mask=None):
  86     return K.any(math_ops.not_equal(inputs, self.mask_value), axis=-1)
  87 
  88   def call(self, inputs):
  89     boolean_mask = K.any(
  90         math_ops.not_equal(inputs, self.mask_value), axis=-1, keepdims=True)
  91     return inputs * math_ops.cast(boolean_mask, inputs.dtype)
  92 
  93   def compute_output_shape(self, input_shape):
  94     return input_shape
  95 
  96   def get_config(self):
  97     config = {'mask_value': self.mask_value}
  98     base_config = super(Masking, self).get_config()
  99     return dict(list(base_config.items()) + list(config.items()))

 102 @tf_export('keras.layers.Dropout')
 103 class Dropout(Layer):
 104   """Applies Dropout to the input.
 105 
 106   Dropout consists in randomly setting
 107   a fraction `rate` of input units to 0 at each update during training time,
 108   which helps prevent overfitting.
 109 
 110   Arguments:
 111       rate: float between 0 and 1. Fraction of the input units to drop.
 112       noise_shape: 1D integer tensor representing the shape of the
 113           binary dropout mask that will be multiplied with the input.
 114           For instance, if your inputs have shape
 115           `(batch_size, timesteps, features)` and
 116           you want the dropout mask to be the same for all timesteps,
 117           you can use `noise_shape=(batch_size, 1, features)`.
 118       seed: A Python integer to use as random seed.
 119   """
 120 
 121   def __init__(self, rate, noise_shape=None, seed=None, **kwargs):
 122     super(Dropout, self).__init__(**kwargs)
 123     self.rate = rate
 124     self.noise_shape = noise_shape
 125     self.seed = seed
 126     self.supports_masking = True
 127 
 128   def _get_noise_shape(self, inputs):
 129     # Subclasses of `Dropout` may implement `_get_noise_shape(self, inputs)`,
 130     # which will override `self.noise_shape`, and allows for custom noise
 131     # shapes with dynamically sized inputs.
 132     if self.noise_shape is None:
 133       return self.noise_shape
 134     return nn_ops._get_noise_shape(inputs, self.noise_shape)  # pylint: disable=protected-access
 135 
 136   def call(self, inputs, training=None):
 137     original_training_value = training
 138     if training is None:
 139       training = K.learning_phase()
 140 
 141     def dropped_inputs():
 142       return nn.dropout(inputs, 1  - self.rate,
 143                         noise_shape=self._get_noise_shape(inputs),
 144                         seed=self.seed)
 145     output = tf_utils.smart_cond(training,
 146                                  dropped_inputs,
 147                                  lambda: array_ops.identity(inputs))
 148     # EagerTensor object has no attribute _uses_learning_phase
 149     if not context.executing_eagerly() and original_training_value is None:
 150       output._uses_learning_phase = True  # pylint: disable=protected-access
 151     return output
 152 
 153   def compute_output_shape(self, input_shape):
 154     return input_shape
 155 
 156   def get_config(self):
 157     config = {
 158         'rate': self.rate,
 159         'noise_shape': self.noise_shape,
 160         'seed': self.seed
 161     }
 162     base_config = super(Dropout, self).get_config()
 163     return dict(list(base_config.items()) + list(config.items()))

 166 @tf_export('keras.layers.SpatialDropout1D')
 167 class SpatialDropout1D(Dropout):
 168   """Spatial 1D version of Dropout.
 169 
 170   This version performs the same function as Dropout, however it drops
 171   entire 1D feature maps instead of individual elements. If adjacent frames
 172   within feature maps are strongly correlated (as is normally the case in
 173   early convolution layers) then regular dropout will not regularize the
 174   activations and will otherwise just result in an effective learning rate
 175   decrease. In this case, SpatialDropout1D will help promote independence
 176   between feature maps and should be used instead.
 177 
 178   Arguments:
 179       rate: float between 0 and 1. Fraction of the input units to drop.
 180 
 181   Input shape:
 182       3D tensor with shape:
 183       `(samples, timesteps, channels)`
 184 
 185   Output shape:
 186       Same as input
 187 
 188   References:
 189       - [Efficient Object Localization Using Convolutional
 190         Networks](https://arxiv.org/abs/1411.4280)
 191   """
 192 
 193   def __init__(self, rate, **kwargs):
 194     super(SpatialDropout1D, self).__init__(rate, **kwargs)
 195     self.input_spec = InputSpec(ndim=3)
 196 
 197   def _get_noise_shape(self, inputs):
 198     input_shape = array_ops.shape(inputs)
 199     noise_shape = (input_shape[0], 1, input_shape[2])
 200     return noise_shape

 344 @tf_export('keras.layers.Reshape')
 345 class Reshape(Layer):
 346   """Reshapes an output to a certain shape.
 347 
 348   Arguments:
 349       target_shape: target shape. Tuple of integers,
 350           does not include the samples dimension (batch size).
 351 
 352   Input shape:
 353       Arbitrary, although all dimensions in the input shaped must be fixed.
 354       Use the keyword argument `input_shape`
 355       (tuple of integers, does not include the samples axis)
 356       when using this layer as the first layer in a model.
 357 
 358   Output shape:
 359       `(batch_size,) + target_shape`
 360 
 361   Example:
 362 
 363   ```python
 364       # as first layer in a Sequential model
 365       model = Sequential()
 366       model.add(Reshape((3, 4), input_shape=(12,)))
 367       # now: model.output_shape == (None, 3, 4)
 368       # note: `None` is the batch dimension
 369 
 370       # as intermediate layer in a Sequential model
 371       model.add(Reshape((6, 2)))
 372       # now: model.output_shape == (None, 6, 2)
 373 
 374       # also supports shape inference using `-1` as dimension
 375       model.add(Reshape((-1, 2, 2)))
 376       # now: model.output_shape == (None, 3, 2, 2)
 377   ```
 378   """
 379 
 380   def __init__(self, target_shape, **kwargs):
 381     super(Reshape, self).__init__(**kwargs)
 382     self.target_shape = tuple(target_shape)
 383 
 384   def _fix_unknown_dimension(self, input_shape, output_shape):
 385     """Find and replace a missing dimension in an output shape.
 386 
 387     This is a near direct port of the internal Numpy function
 388     `_fix_unknown_dimension` in `numpy/core/src/multiarray/shape.c`
 389 
 390     Arguments:
 391         input_shape: shape of array being reshaped
 392         output_shape: desired shape of the array with at most
 393             a single -1 which indicates a dimension that should be
 394             derived from the input shape.
 395 
 396     Returns:
 397         The new output shape with a -1 replaced with its computed value.
 398 
 399         Raises a ValueError if the total array size of the output_shape is
 400         different then the input_shape, or more than one unknown dimension
 401         is specified.
 402 
 403     Raises:
 404         ValueError: in case of invalid values
 405             for `input_shape` or `input_shape`.
 406     """
 407     output_shape = list(output_shape)
 408     msg = 'total size of new array must be unchanged'
 409 
 410     known, unknown = 1, None
 411     for index, dim in enumerate(output_shape):
 412       if dim < 0:
 413         if unknown is None:
 414           unknown = index
 415         else:
 416           raise ValueError('Can only specify one unknown dimension.')
 417       else:
 418         known *= dim
 419 
 420     original = np.prod(input_shape, dtype=int)
 421     if unknown is not None:
 422       if known == 0 or original % known != 0:
 423         raise ValueError(msg)
 424       output_shape[unknown] = original // known
 425     elif original != known:
 426       raise ValueError(msg)
 427     return output_shape
 428 
 429   def compute_output_shape(self, input_shape):
 430     input_shape = tensor_shape.TensorShape(input_shape).as_list()
 431     if None in input_shape[1:]:
 432       output_shape = [input_shape[0]]
 433       # input shape (partially) unknown? replace -1's with None's
 434       output_shape += tuple(s if s != -1 else None for s in self.target_shape)
 435     else:
 436       output_shape = [input_shape[0]]
 437       output_shape += self._fix_unknown_dimension(input_shape[1:],
 438                                                   self.target_shape)
 439     return tensor_shape.TensorShape(output_shape)
 440 
 441   def call(self, inputs):
 442     return array_ops.reshape(inputs,
 443                              (array_ops.shape(inputs)[0],) + self.target_shape)
 444 
 445   def get_config(self):
 446     config = {'target_shape': self.target_shape}
 447     base_config = super(Reshape, self).get_config()
 448     return dict(list(base_config.items()) + list(config.items()))

 451 @tf_export('keras.layers.Permute')
 452 class Permute(Layer):
 453   """Permutes the dimensions of the input according to a given pattern.
 454 
 455   Useful for e.g. connecting RNNs and convnets together.
 456 
 457   Example:
 458 
 459   ```python
 460       model = Sequential()
 461       model.add(Permute((2, 1), input_shape=(10, 64)))
 462       # now: model.output_shape == (None, 64, 10)
 463       # note: `None` is the batch dimension
 464   ```
 465 
 466   Arguments:
 467       dims: Tuple of integers. Permutation pattern, does not include the
 468           samples dimension. Indexing starts at 1.
 469           For instance, `(2, 1)` permutes the first and second dimensions
 470           of the input.
 471 
 472   Input shape:
 473       Arbitrary. Use the keyword argument `input_shape`
 474       (tuple of integers, does not include the samples axis)
 475       when using this layer as the first layer in a model.
 476 
 477   Output shape:
 478       Same as the input shape, but with the dimensions re-ordered according
 479       to the specified pattern.
 480   """
 481 
 482   def __init__(self, dims, **kwargs):
 483     super(Permute, self).__init__(**kwargs)
 484     self.dims = tuple(dims)
 485     if sorted(dims) != list(range(1, len(dims) + 1)):
 486       raise ValueError(
 487           'Invalid permutation `dims` for Permute Layer: %s. '
 488           'The set of indices in `dims` must be consecutive and start from 1.' %
 489           (dims,))
 490     self.input_spec = InputSpec(ndim=len(self.dims) + 1)
 491 
 492   def compute_output_shape(self, input_shape):
 493     input_shape = tensor_shape.TensorShape(input_shape).as_list()
 494     output_shape = copy.copy(input_shape)
 495     for i, dim in enumerate(self.dims):
 496       target_dim = input_shape[dim]
 497       output_shape[i + 1] = target_dim
 498     return tensor_shape.TensorShape(output_shape)
 499 
 500   def call(self, inputs):
 501     return array_ops.transpose(inputs, perm=(0,) + self.dims)
 502 
 503   def get_config(self):
 504     config = {'dims': self.dims}
 505     base_config = super(Permute, self).get_config()
 506     return dict(list(base_config.items()) + list(config.items()))

 509 @tf_export('keras.layers.Flatten')
 510 class Flatten(Layer):
 511   """Flattens the input. Does not affect the batch size.
 512 
 513   Arguments:
 514       data_format: A string,
 515           one of `channels_last` (default) or `channels_first`.
 516           The ordering of the dimensions in the inputs.
 517           `channels_last` corresponds to inputs with shape
 518           `(batch, ..., channels)` while `channels_first` corresponds to
 519           inputs with shape `(batch, channels, ...)`.
 520           It defaults to the `image_data_format` value found in your
 521           Keras config file at `~/.keras/keras.json`.
 522           If you never set it, then it will be "channels_last".
 523 
 524   Example:
 525 
 526   ```python
 527       model = Sequential()
 528       model.add(Convolution2D(64, 3, 3,
 529                               border_mode='same',
 530                               input_shape=(3, 32, 32)))
 531       # now: model.output_shape == (None, 64, 32, 32)
 532 
 533       model.add(Flatten())
 534       # now: model.output_shape == (None, 65536)
 535   ```
 536   """
 537 
 538   def __init__(self, data_format=None, **kwargs):
 539     super(Flatten, self).__init__(**kwargs)
 540     self.data_format = conv_utils.normalize_data_format(data_format)
 541     self.input_spec = InputSpec(min_ndim=2)
 542 
 543   def call(self, inputs):
 544     if self.data_format == 'channels_first':
 545       permutation = [0]
 546       permutation.extend([i for i in
 547                           range(2, K.ndim(inputs))])
 548       permutation.append(1)
 549       inputs = array_ops.transpose(inputs, perm=permutation)
 550 
 551     outputs = array_ops.reshape(inputs, (array_ops.shape(inputs)[0], -1))
 552     if not context.executing_eagerly():
 553       outputs.set_shape(self.compute_output_shape(inputs.get_shape()))
 554     return outputs
 555 
 556   def compute_output_shape(self, input_shape):
 557     input_shape = tensor_shape.TensorShape(input_shape).as_list()
 558     output_shape = [input_shape[0]]
 559     if all(input_shape[1:]):
 560       output_shape += [np.prod(input_shape[1:])]
 561     else:
 562       output_shape += [None]
 563     return tensor_shape.TensorShape(output_shape)
 564 
 565   def get_config(self):
 566     config = {'data_format': self.data_format}
 567     base_config = super(Flatten, self).get_config()
 568     return dict(list(base_config.items()) + list(config.items()))

 571 @tf_export('keras.layers.RepeatVector')
 572 class RepeatVector(Layer):
 573   """Repeats the input n times.
 574 
 575   Example:
 576 
 577   ```python
 578       model = Sequential()
 579       model.add(Dense(32, input_dim=32))
 580       # now: model.output_shape == (None, 32)
 581       # note: `None` is the batch dimension
 582 
 583       model.add(RepeatVector(3))
 584       # now: model.output_shape == (None, 3, 32)
 585   ```
 586 
 587   Arguments:
 588       n: integer, repetition factor.
 589 
 590   Input shape:
 591       2D tensor of shape `(num_samples, features)`.
 592 
 593   Output shape:
 594       3D tensor of shape `(num_samples, n, features)`.
 595   """
 596 
 597   def __init__(self, n, **kwargs):
 598     super(RepeatVector, self).__init__(**kwargs)
 599     self.n = n
 600     self.input_spec = InputSpec(ndim=2)
 601 
 602   def compute_output_shape(self, input_shape):
 603     input_shape = tensor_shape.TensorShape(input_shape).as_list()
 604     return tensor_shape.TensorShape([input_shape[0], self.n, input_shape[1]])
 605 
 606   def call(self, inputs):
 607     return K.repeat(inputs, self.n)
 608 
 609   def get_config(self):

 844 @tf_export('keras.layers.Dense')
 845 class Dense(Layer):
 846   """Just your regular densely-connected NN layer.
 847 
 848   `Dense` implements the operation:
 849   `output = activation(dot(input, kernel) + bias)`
 850   where `activation` is the element-wise activation function
 851   passed as the `activation` argument, `kernel` is a weights matrix
 852   created by the layer, and `bias` is a bias vector created by the layer
 853   (only applicable if `use_bias` is `True`).
 854 
 855   Note: if the input to the layer has a rank greater than 2, then
 856   it is flattened prior to the initial dot product with `kernel`.
 857 
 858   Example:
 859 
 860   ```python
 861       # as first layer in a sequential model:
 862       model = Sequential()
 863       model.add(Dense(32, input_shape=(16,)))
 864       # now the model will take as input arrays of shape (*, 16)
 865       # and output arrays of shape (*, 32)
 866 
 867       # after the first layer, you don't need to specify
 868       # the size of the input anymore:
 869       model.add(Dense(32))
 870   ```
 871 
 872   Arguments:
 873       units: Positive integer, dimensionality of the output space.
 874       activation: Activation function to use.
 875           If you don't specify anything, no activation is applied
 876           (ie. "linear" activation: `a(x) = x`).
 877       use_bias: Boolean, whether the layer uses a bias vector.
 878       kernel_initializer: Initializer for the `kernel` weights matrix.
 879       bias_initializer: Initializer for the bias vector.
 880       kernel_regularizer: Regularizer function applied to
 881           the `kernel` weights matrix.
 882       bias_regularizer: Regularizer function applied to the bias vector.
 883       activity_regularizer: Regularizer function applied to
 884           the output of the layer (its "activation")..
 885       kernel_constraint: Constraint function applied to
 886           the `kernel` weights matrix.
 887       bias_constraint: Constraint function applied to the bias vector.
 888 
 889   Input shape:
 890       nD tensor with shape: `(batch_size, ..., input_dim)`.
 891       The most common situation would be
 892       a 2D input with shape `(batch_size, input_dim)`.
 893 
 894   Output shape:
 895       nD tensor with shape: `(batch_size, ..., units)`.
 896       For instance, for a 2D input with shape `(batch_size, input_dim)`,
 897       the output would have shape `(batch_size, units)`.
 898   """
 899

 901                units,

 905                bias_initializer='zeros',

 958   def call(self, inputs):
 959     inputs = ops.convert_to_tensor(inputs, dtype=self.dtype)
 960     rank = common_shapes.rank(inputs)
 961     if rank > 2:
 962       # Broadcasting is required for the inputs.
 963       outputs = standard_ops.tensordot(inputs, self.kernel, [[rank - 1], [0]])
 964       # Reshape the output back to the original ndim of the input.
 965       if not context.executing_eagerly():
 966         shape = inputs.get_shape().as_list()
 967         output_shape = shape[:-1] + [self.units]
 968         outputs.set_shape(output_shape)
 969     else:
 970       outputs = gen_math_ops.mat_mul(inputs, self.kernel)
 971     if self.use_bias:
 972       outputs = nn.bias_add(outputs, self.bias)
 973     if self.activation is not None:
 974       return self.activation(outputs)  # pylint: disable=not-callable
 975     return outputs

 981       raise ValueError(

 996             regularizers.serialize(self.activity_regularizer),
