  34 @tf_export('keras.layers.Embedding')
  35 class Embedding(Layer):
  36   """Turns positive integers (indexes) into dense vectors of fixed size.
  37 
  38   eg. [[4], [20]] -> [[0.25, 0.1], [0.6, -0.2]]
  39 
  40   This layer can only be used as the first layer in a model.
  41 
  42   Example:
  43 
  44   ```python
  45     model = Sequential()
  46     model.add(Embedding(1000, 64, input_length=10))
  47     # the model will take as input an integer matrix of size (batch,
  48     input_length).
  49     # the largest integer (i.e. word index) in the input should be no larger
  50     than 999 (vocabulary size).
  51     # now model.output_shape == (None, 10, 64), where None is the batch
  52     dimension.
  53 
  54     input_array = np.random.randint(1000, size=(32, 10))
  55 
  56     model.compile('rmsprop', 'mse')
  57     output_array = model.predict(input_array)
  58     assert output_array.shape == (32, 10, 64)
  59   ```
  60 
  61   Arguments:
  62     input_dim: int > 0. Size of the vocabulary,
  63         i.e. maximum integer index + 1.
  64     output_dim: int >= 0. Dimension of the dense embedding.
  65     embeddings_initializer: Initializer for the `embeddings` matrix.
  66     embeddings_regularizer: Regularizer function applied to
  67         the `embeddings` matrix.
  68     embeddings_constraint: Constraint function applied to
  69         the `embeddings` matrix.
  70     mask_zero: Whether or not the input value 0 is a special "padding"
  71         value that should be masked out.
  72         This is useful when using recurrent layers
  73         which may take variable length input.
  74         If this is `True` then all subsequent layers
  75         in the model need to support masking or an exception will be raised.
  76         If mask_zero is set to True, as a consequence, index 0 cannot be
  77         used in the vocabulary (input_dim should equal size of
  78         vocabulary + 1).
  79     input_length: Length of input sequences, when it is constant.
  80         This argument is required if you are going to connect
  81         `Flatten` then `Dense` layers upstream
  82         (without it, the shape of the dense outputs cannot be computed).
  83 
  84   Input shape:
  85       2D tensor with shape: `(batch_size, sequence_length)`.
  86 
  87   Output shape:
  88       3D tensor with shape: `(batch_size, sequence_length, output_dim)`.
  89 
  90   """
  91 
  92   def __init__(self,
  93                input_dim,
  94                output_dim,
  95                embeddings_initializer='uniform',
  96                embeddings_regularizer=None,
  97                activity_regularizer=None,
  98                embeddings_constraint=None,
  99                mask_zero=False,
 100                input_length=None,
 101                **kwargs):
 102     if 'input_shape' not in kwargs:
 103       if input_length:
 104         kwargs['input_shape'] = (input_length,)
 105       else:
 106         kwargs['input_shape'] = (None,)
 107     dtype = kwargs.pop('dtype', K.floatx())
 108     super(Embedding, self).__init__(dtype=dtype, **kwargs)
 109 
 110     self.input_dim = input_dim
 111     self.output_dim = output_dim
 112     self.embeddings_initializer = initializers.get(embeddings_initializer)
 113     self.embeddings_regularizer = regularizers.get(embeddings_regularizer)
 114     self.activity_regularizer = regularizers.get(activity_regularizer)
 115     self.embeddings_constraint = constraints.get(embeddings_constraint)
 116     self.mask_zero = mask_zero
 117     self.supports_masking = mask_zero
 118     self.input_length = input_length
 119 
 120   @tf_utils.shape_type_conversion
 121   def build(self, input_shape):
 122     # Note: most sparse optimizers do not have GPU kernels defined. When
 123     # building graphs, the placement algorithm is able to place variables on CPU
 124     # since it knows all kernels using the variable only exist on CPU.
 125     # When eager execution is enabled, the placement decision has to be made
 126     # right now. Checking for the presence of GPUs to avoid complicating the
 127     # TPU codepaths which can handle sparse optimizers.
 128     if context.executing_eagerly() and context.context().num_gpus():
 129       with ops.device('cpu:0'):
 130         self.embeddings = self.add_weight(
 131             shape=(self.input_dim, self.output_dim),
 132             initializer=self.embeddings_initializer,
 133             name='embeddings',
 134             regularizer=self.embeddings_regularizer,
 135             constraint=self.embeddings_constraint)
 136     else:
 137       self.embeddings = self.add_weight(
 138           shape=(self.input_dim, self.output_dim),
 139           initializer=self.embeddings_initializer,
 140           name='embeddings',
 141           regularizer=self.embeddings_regularizer,
 142           constraint=self.embeddings_constraint)
 143     self.built = True
 144 
 145   def compute_mask(self, inputs, mask=None):
 146     if not self.mask_zero:
 147       return None
 148 
 149     return math_ops.not_equal(inputs, 0)
 150 
 151   @tf_utils.shape_type_conversion
 152   def compute_output_shape(self, input_shape):
 153     if self.input_length is None:
 154       return input_shape + (self.output_dim,)
 155     else:
 156       # input_length can be tuple if input is 3D or higher
 157       if isinstance(self.input_length, (list, tuple)):
 158         in_lens = list(self.input_length)
 159       else:
 160         in_lens = [self.input_length]
 161       if len(in_lens) != len(input_shape) - 1:
 162         ValueError('"input_length" is %s, but received input has shape %s' %
 163                    (str(self.input_length), str(input_shape)))
 164       else:
 165         for i, (s1, s2) in enumerate(zip(in_lens, input_shape[1:])):
 166           if s1 is not None and s2 is not None and s1 != s2:
 167             ValueError('"input_length" is %s, but received input has shape %s' %
 168                        (str(self.input_length), str(input_shape)))
 169           elif s1 is None:
 170             in_lens[i] = s2
 171       return (input_shape[0],) + tuple(in_lens) + (self.output_dim,)
 172 
 173   def call(self, inputs):
 174     dtype = K.dtype(inputs)
 175     if dtype != 'int32' and dtype != 'int64':
 176       inputs = math_ops.cast(inputs, 'int32')
 177     out = embedding_ops.embedding_lookup(self.embeddings, inputs)
 178     return out

 180   def get_config(self):
 181     config = {
 182         'input_dim':
 183             self.input_dim,
 184         'output_dim':
 185             self.output_dim,
 186         'embeddings_initializer':
 187             initializers.serialize(self.embeddings_initializer),
 188         'embeddings_regularizer':
 189             regularizers.serialize(self.embeddings_regularizer),
 190         'activity_regularizer':
 191             regularizers.serialize(self.activity_regularizer),
 192         'embeddings_constraint':
 193             constraints.serialize(self.embeddings_constraint),
 194         'mask_zero':
 195             self.mask_zero,
 196         'input_length':
 197             self.input_length
 198     }
 199     base_config = super(Embedding, self).get_config()
 200     return dict(list(base_config.items()) + list(config.items()))
