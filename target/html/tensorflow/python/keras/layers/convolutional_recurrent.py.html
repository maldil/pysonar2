  40 class ConvRNN2D(RNN):
  41   """Base class for convolutional-recurrent layers.
  42 
  43   Arguments:
  44     cell: A RNN cell instance. A RNN cell is a class that has:
  45         - a `call(input_at_t, states_at_t)` method, returning
  46             `(output_at_t, states_at_t_plus_1)`. The call method of the
  47             cell can also take the optional argument `constants`, see
  48             section "Note on passing external constants" below.
  49         - a `state_size` attribute. This can be a single integer
  50             (single state) in which case it is
  51             the number of channels of the recurrent state
  52             (which should be the same as the number of channels of the cell
  53             output). This can also be a list/tuple of integers
  54             (one size per state). In this case, the first entry
  55             (`state_size[0]`) should be the same as
  56             the size of the cell output.
  57     return_sequences: Boolean. Whether to return the last output.
  58         in the output sequence, or the full sequence.
  59     return_state: Boolean. Whether to return the last state
  60         in addition to the output.
  61     go_backwards: Boolean (default False).
  62         If True, process the input sequence backwards and return the
  63         reversed sequence.
  64     stateful: Boolean (default False). If True, the last state
  65         for each sample at index i in a batch will be used as initial
  66         state for the sample of index i in the following batch.
  67     input_shape: Use this argument to specify the shape of the
  68         input when this layer is the first one in a model.
  69 
  70   Input shape:
  71     5D tensor with shape:
  72     `(samples, timesteps, channels, rows, cols)`
  73     if data_format='channels_first' or 5D tensor with shape:
  74     `(samples, timesteps, rows, cols, channels)`
  75     if data_format='channels_last'.
  76 
  77   Output shape:
  78     - if `return_state`: a list of tensors. The first tensor is
  79         the output. The remaining tensors are the last states,
  80         each 5D tensor with shape:
  81         `(samples, timesteps, filters, new_rows, new_cols)`
  82         if data_format='channels_first'
  83         or 5D tensor with shape:
  84         `(samples, timesteps, new_rows, new_cols, filters)`
  85         if data_format='channels_last'.
  86         `rows` and `cols` values might have changed due to padding.
  87     - if `return_sequences`: 5D tensor with shape:
  88         `(samples, timesteps, filters, new_rows, new_cols)`
  89         if data_format='channels_first'
  90         or 5D tensor with shape:
  91         `(samples, timesteps, new_rows, new_cols, filters)`
  92         if data_format='channels_last'.
  93     - else, 4D tensor with shape:
  94         `(samples, filters, new_rows, new_cols)`
  95         if data_format='channels_first'
  96         or 4D tensor with shape:
  97         `(samples, new_rows, new_cols, filters)`
  98         if data_format='channels_last'.
  99 
 100   Masking:
 101     This layer supports masking for input data with a variable number
 102     of timesteps. To introduce masks to your data,
 103     use an Embedding layer with the `mask_zero` parameter
 104     set to `True`.
 105 
 106   Note on using statefulness in RNNs:
 107     You can set RNN layers to be 'stateful', which means that the states
 108     computed for the samples in one batch will be reused as initial states
 109     for the samples in the next batch. This assumes a one-to-one mapping
 110     between samples in different successive batches.
 111     To enable statefulness:
 112         - specify `stateful=True` in the layer constructor.
 113         - specify a fixed batch size for your model, by passing
 114              - if sequential model:
 115                 `batch_input_shape=(...)` to the first layer in your model.
 116              - if functional model with 1 or more Input layers:
 117                 `batch_shape=(...)` to all the first layers in your model.
 118                 This is the expected shape of your inputs
 119                 *including the batch size*.
 120                 It should be a tuple of integers,
 121                 e.g. `(32, 10, 100, 100, 32)`.
 122                 Note that the number of rows and columns should be specified
 123                 too.
 124         - specify `shuffle=False` when calling fit().
 125     To reset the states of your model, call `.reset_states()` on either
 126     a specific layer, or on your entire model.
 127 
 128   Note on specifying the initial state of RNNs:
 129     You can specify the initial state of RNN layers symbolically by
 130     calling them with the keyword argument `initial_state`. The value of
 131     `initial_state` should be a tensor or list of tensors representing
 132     the initial state of the RNN layer.
 133     You can specify the initial state of RNN layers numerically by
 134     calling `reset_states` with the keyword argument `states`. The value of
 135     `states` should be a numpy array or list of numpy arrays representing
 136     the initial state of the RNN layer.
 137 
 138   Note on passing external constants to RNNs:
 139     You can pass "external" constants to the cell using the `constants`
 140     keyword argument of `RNN.__call__` (as well as `RNN.call`) method. This
 141     requires that the `cell.call` method accepts the same keyword argument
 142     `constants`. Such constants can be used to condition the cell
 143     transformation on additional static inputs (not changing over time),
 144     a.k.a. an attention mechanism.
 145   """

 147   def __init__(self,
 148                cell,
 149                return_sequences=False,
 150                return_state=False,
 151                go_backwards=False,
 152                stateful=False,
 153                unroll=False,
 154                **kwargs):
 155     if unroll:
 156       raise TypeError('Unrolling isn\'t possible with '
 157                       'convolutional RNNs.')
 158     if isinstance(cell, (list, tuple)):
 159       # The StackedConvRNN2DCells isn't implemented yet.
 160       raise TypeError('It is not possible at the moment to'
 161                       'stack convolutional cells.')
 162     super(ConvRNN2D, self).__init__(cell,
 163                                     return_sequences,
 164                                     return_state,
 165                                     go_backwards,
 166                                     stateful,
 167                                     unroll,
 168                                     **kwargs)
 169     self.input_spec = [InputSpec(ndim=5)]
 170     self.states = None
 171     self._num_constants = None

 173   @tf_utils.shape_type_conversion
 174   def compute_output_shape(self, input_shape):
 175     if isinstance(input_shape, list):
 176       input_shape = input_shape[0]
 177 
 178     cell = self.cell
 179     if cell.data_format == 'channels_first':
 180       rows = input_shape[3]
 181       cols = input_shape[4]
 182     elif cell.data_format == 'channels_last':
 183       rows = input_shape[2]
 184       cols = input_shape[3]
 185     rows = conv_utils.conv_output_length(rows,
 186                                          cell.kernel_size[0],
 187                                          padding=cell.padding,
 188                                          stride=cell.strides[0],
 189                                          dilation=cell.dilation_rate[0])
 190     cols = conv_utils.conv_output_length(cols,
 191                                          cell.kernel_size[1],
 192                                          padding=cell.padding,
 193                                          stride=cell.strides[1],
 194                                          dilation=cell.dilation_rate[1])
 195 
 196     if cell.data_format == 'channels_first':
 197       output_shape = input_shape[:2] + (cell.filters, rows, cols)
 198     elif cell.data_format == 'channels_last':
 199       output_shape = input_shape[:2] + (rows, cols, cell.filters)
 200 
 201     if not self.return_sequences:
 202       output_shape = output_shape[:1] + output_shape[2:]
 203 
 204     if self.return_state:
 205       output_shape = [output_shape]
 206       if cell.data_format == 'channels_first':
 207         output_shape += [(input_shape[0], cell.filters, rows, cols)
 208                          for _ in range(2)]
 209       elif cell.data_format == 'channels_last':
 210         output_shape += [(input_shape[0], rows, cols, cell.filters)
 211                          for _ in range(2)]
 212     return output_shape

 214   @tf_utils.shape_type_conversion
 215   def build(self, input_shape):
 216     # Note input_shape will be list of shapes of initial states and
 217     # constants if these are passed in __call__.
 218     if self._num_constants is not None:
 219       constants_shape = input_shape[-self._num_constants:]  # pylint: disable=E1130
 220     else:
 221       constants_shape = None
 222 
 223     if isinstance(input_shape, list):
 224       input_shape = input_shape[0]
 225 
 226     batch_size = input_shape[0] if self.stateful else None
 227     self.input_spec[0] = InputSpec(shape=(batch_size, None) + input_shape[2:5])
 228 
 229     # allow cell (if layer) to build before we set or validate state_spec
 230     if isinstance(self.cell, Layer):
 231       step_input_shape = (input_shape[0],) + input_shape[2:]
 232       if constants_shape is not None:
 233         self.cell.build([step_input_shape] + constants_shape)
 234       else:
 235         self.cell.build(step_input_shape)
 236 
 237     # set or validate state_spec
 238     if hasattr(self.cell.state_size, '__len__'):
 239       state_size = list(self.cell.state_size)
 240     else:
 241       state_size = [self.cell.state_size]
 242 
 243     if self.state_spec is not None:
 244       # initial_state was passed in call, check compatibility
 245       if self.cell.data_format == 'channels_first':
 246         ch_dim = 1
 247       elif self.cell.data_format == 'channels_last':
 248         ch_dim = 3
 249       if [spec.shape[ch_dim] for spec in self.state_spec] != state_size:
 250         raise ValueError(
 251             'An initial_state was passed that is not compatible with '
 252             '`cell.state_size`. Received `state_spec`={}; '
 253             'However `cell.state_size` is '
 254             '{}'.format([spec.shape for spec in self.state_spec],
 255                         self.cell.state_size))
 256     else:
 257       if self.cell.data_format == 'channels_first':
 258         self.state_spec = [InputSpec(shape=(None, dim, None, None))
 259                            for dim in state_size]
 260       elif self.cell.data_format == 'channels_last':
 261         self.state_spec = [InputSpec(shape=(None, None, None, dim))
 262                            for dim in state_size]
 263     if self.stateful:
 264       self.reset_states()
 265     self.built = True

 267   def get_initial_state(self, inputs):
 268     # (samples, timesteps, rows, cols, filters)
 269     initial_state = K.zeros_like(inputs)
 270     # (samples, rows, cols, filters)
 271     initial_state = K.sum(initial_state, axis=1)
 272     shape = list(self.cell.kernel_shape)
 273     shape[-1] = self.cell.filters
 274     initial_state = self.cell.input_conv(initial_state,
 275                                          K.zeros(tuple(shape)),
 276                                          padding=self.cell.padding)
 277 
 278     if hasattr(self.cell.state_size, '__len__'):
 279       return [initial_state for _ in self.cell.state_size]
 280     else:
 281       return [initial_state]

 283   def __call__(self, inputs, initial_state=None, constants=None, **kwargs):
 284     inputs, initial_state, constants = _standardize_args(
 285         inputs, initial_state, constants, self._num_constants)
 286 
 287     if initial_state is None and constants is None:
 288       return super(ConvRNN2D, self).__call__(inputs, **kwargs)
 289 
 290     # If any of `initial_state` or `constants` are specified and are Keras
 291     # tensors, then add them to the inputs and temporarily modify the
 292     # input_spec to include them.
 293 
 294     additional_inputs = []
 295     additional_specs = []
 296     if initial_state is not None:
 297       kwargs['initial_state'] = initial_state
 298       additional_inputs += initial_state
 299       self.state_spec = []
 300       for state in initial_state:
 301         shape = K.int_shape(state)
 302         self.state_spec.append(InputSpec(shape=shape))
 303 
 304       additional_specs += self.state_spec
 305     if constants is not None:
 306       kwargs['constants'] = constants
 307       additional_inputs += constants
 308       self.constants_spec = [InputSpec(shape=K.int_shape(constant))
 309                              for constant in constants]
 310       self._num_constants = len(constants)
 311       additional_specs += self.constants_spec
 312     # at this point additional_inputs cannot be empty
 313     for tensor in additional_inputs:
 314       if K.is_keras_tensor(tensor) != K.is_keras_tensor(additional_inputs[0]):
 315         raise ValueError('The initial state or constants of an RNN'
 316                          ' layer cannot be specified with a mix of'
 317                          ' Keras tensors and non-Keras tensors')
 318 
 319     if K.is_keras_tensor(additional_inputs[0]):
 320       # Compute the full input spec, including state and constants
 321       full_input = [inputs] + additional_inputs
 322       full_input_spec = self.input_spec + additional_specs
 323       # Perform the call with temporarily replaced input_spec
 324       original_input_spec = self.input_spec
 325       self.input_spec = full_input_spec
 326       output = super(ConvRNN2D, self).__call__(full_input, **kwargs)
 327       self.input_spec = original_input_spec
 328       return output
 329     else:
 330       return super(ConvRNN2D, self).__call__(inputs, **kwargs)

 332   def call(self,
 333            inputs,
 334            mask=None,
 335            training=None,
 336            initial_state=None,
 337            constants=None):
 338     # note that the .build() method of subclasses MUST define
 339     # self.input_spec and self.state_spec with complete input shapes.
 340     if isinstance(inputs, list):
 341       inputs = inputs[0]
 342     if initial_state is not None:
 343       pass
 344     elif self.stateful:
 345       initial_state = self.states
 346     else:
 347       initial_state = self.get_initial_state(inputs)
 348 
 349     if isinstance(mask, list):
 350       mask = mask[0]
 351 
 352     if len(initial_state) != len(self.states):
 353       raise ValueError('Layer has ' + str(len(self.states)) +
 354                        ' states but was passed ' +
 355                        str(len(initial_state)) +
 356                        ' initial states.')
 357     timesteps = K.int_shape(inputs)[1]
 358 
 359     kwargs = {}
 360     if generic_utils.has_arg(self.cell.call, 'training'):
 361       kwargs['training'] = training
 362 
 363     if constants:
 364       if not generic_utils.has_arg(self.cell.call, 'constants'):
 365         raise ValueError('RNN cell does not support constants')
 366 
 367       def step(inputs, states):
 368         constants = states[-self._num_constants:]
 369         states = states[:-self._num_constants]
 370         return self.cell.call(inputs, states, constants=constants,
 371                               **kwargs)
 372     else:
 373       def step(inputs, states):
 374         return self.cell.call(inputs, states, **kwargs)
 375 
 376     last_output, outputs, states = K.rnn(step,
 377                                          inputs,
 378                                          initial_state,
 379                                          constants=constants,
 380                                          go_backwards=self.go_backwards,
 381                                          mask=mask,
 382                                          input_length=timesteps)
 383     if self.stateful:
 384       updates = []
 385       for i in range(len(states)):
 386         updates.append(K.update(self.states[i], states[i]))
 387       self.add_update(updates, inputs=True)
 388 
 389     if self.return_sequences:
 390       output = outputs
 391     else:
 392       output = last_output
 393 
 394     # Properly set learning phase
 395     if getattr(last_output, '_uses_learning_phase', False):
 396       output._uses_learning_phase = True
 397 
 398     if self.return_state:
 399       if not isinstance(states, (list, tuple)):
 400         states = [states]
 401       else:
 402         states = list(states)
 403       return [output] + states
 404     else:
 405       return output

 407   def reset_states(self, states=None):
 408     if not self.stateful:
 409       raise AttributeError('Layer must be stateful.')
 410     input_shape = self.input_spec[0].shape
 411     state_shape = self.compute_output_shape(input_shape)
 412     if self.return_state:
 413       state_shape = state_shape[0]
 414     if self.return_sequences:
 415       state_shape = state_shape[:1].concatenate(state_shape[2:])
 416     if None in state_shape:
 417       raise ValueError('If a RNN is stateful, it needs to know '
 418                        'its batch size. Specify the batch size '
 419                        'of your input tensors: \n'
 420                        '- If using a Sequential model, '
 421                        'specify the batch size by passing '
 422                        'a `batch_input_shape` '
 423                        'argument to your first layer.\n'
 424                        '- If using the functional API, specify '
 425                        'the time dimension by passing a '
 426                        '`batch_shape` argument to your Input layer.\n'
 427                        'The same thing goes for the number of rows and '
 428                        'columns.')
 429 
 430     # helper function
 431     def get_tuple_shape(nb_channels):
 432       result = list(state_shape)
 433       if self.cell.data_format == 'channels_first':
 434         result[1] = nb_channels
 435       elif self.cell.data_format == 'channels_last':
 436         result[3] = nb_channels
 437       else:
 438         raise KeyError
 439       return tuple(result)
 440 
 441     # initialize state if None
 442     if self.states[0] is None:
 443       if hasattr(self.cell.state_size, '__len__'):
 444         self.states = [K.zeros(get_tuple_shape(dim))
 445                        for dim in self.cell.state_size]
 446       else:
 447         self.states = [K.zeros(get_tuple_shape(self.cell.state_size))]
 448     elif states is None:
 449       if hasattr(self.cell.state_size, '__len__'):
 450         for state, dim in zip(self.states, self.cell.state_size):
 451           K.set_value(state, np.zeros(get_tuple_shape(dim)))
 452       else:
 453         K.set_value(self.states[0],
 454                     np.zeros(get_tuple_shape(self.cell.state_size)))
 455     else:
 456       if not isinstance(states, (list, tuple)):
 457         states = [states]
 458       if len(states) != len(self.states):
 459         raise ValueError('Layer ' + self.name + ' expects ' +
 460                          str(len(self.states)) + ' states, ' +
 461                          'but it received ' + str(len(states)) +
 462                          ' state values. Input received: ' + str(states))
 463       for index, (value, state) in enumerate(zip(states, self.states)):
 464         if hasattr(self.cell.state_size, '__len__'):
 465           dim = self.cell.state_size[index]
 466         else:
 467           dim = self.cell.state_size
 468         if value.shape != get_tuple_shape(dim):
 469           raise ValueError('State ' + str(index) +
 470                            ' is incompatible with layer ' +
 471                            self.name + ': expected shape=' +
 472                            str(get_tuple_shape(dim)) +
 473                            ', found shape=' + str(value.shape))
 474         # TODO(anjalisridhar): consider batch calls to `set_value`.
 475         K.set_value(state, value)

 478 class ConvLSTM2DCell(Layer):
 479   """Cell class for the ConvLSTM2D layer.
 480 
 481   # Arguments
 482       filters: Integer, the dimensionality of the output space
 483           (i.e. the number of output filters in the convolution).
 484       kernel_size: An integer or tuple/list of n integers, specifying the
 485           dimensions of the convolution window.
 486       strides: An integer or tuple/list of n integers,
 487           specifying the strides of the convolution.
 488           Specifying any stride value != 1 is incompatible with specifying
 489           any `dilation_rate` value != 1.
 490       padding: One of `"valid"` or `"same"` (case-insensitive).
 491       data_format: A string,
 492           one of `channels_last` (default) or `channels_first`.
 493           It defaults to the `image_data_format` value found in your
 494           Keras config file at `~/.keras/keras.json`.
 495           If you never set it, then it will be "channels_last".
 496       dilation_rate: An integer or tuple/list of n integers, specifying
 497           the dilation rate to use for dilated convolution.
 498           Currently, specifying any `dilation_rate` value != 1 is
 499           incompatible with specifying any `strides` value != 1.
 500       activation: Activation function to use.
 501           If you don't specify anything, no activation is applied
 502           (ie. "linear" activation: `a(x) = x`).
 503       recurrent_activation: Activation function to use
 504           for the recurrent step.
 505       use_bias: Boolean, whether the layer uses a bias vector.
 506       kernel_initializer: Initializer for the `kernel` weights matrix,
 507           used for the linear transformation of the inputs.
 508       recurrent_initializer: Initializer for the `recurrent_kernel`
 509           weights matrix,
 510           used for the linear transformation of the recurrent state.
 511       bias_initializer: Initializer for the bias vector.
 512       unit_forget_bias: Boolean.
 513           If True, add 1 to the bias of the forget gate at initialization.
 514           Use in combination with `bias_initializer="zeros"`.
 515           This is recommended in [Jozefowicz et al.]
 516           (http://www.jmlr.org/proceedings/papers/v37/jozefowicz15.pdf)
 517       kernel_regularizer: Regularizer function applied to
 518           the `kernel` weights matrix.
 519       recurrent_regularizer: Regularizer function applied to
 520           the `recurrent_kernel` weights matrix.
 521       bias_regularizer: Regularizer function applied to the bias vector.
 522       kernel_constraint: Constraint function applied to
 523           the `kernel` weights matrix.
 524       recurrent_constraint: Constraint function applied to
 525           the `recurrent_kernel` weights matrix.
 526       bias_constraint: Constraint function applied to the bias vector.
 527       dropout: Float between 0 and 1.
 528           Fraction of the units to drop for
 529           the linear transformation of the inputs.
 530       recurrent_dropout: Float between 0 and 1.
 531           Fraction of the units to drop for
 532           the linear transformation of the recurrent state.
 533   """
 534 
 535   def __init__(self,
 536                filters,
 537                kernel_size,
 538                strides=(1, 1),
 539                padding='valid',
 540                data_format=None,
 541                dilation_rate=(1, 1),
 542                activation='tanh',
 543                recurrent_activation='hard_sigmoid',
 544                use_bias=True,
 545                kernel_initializer='glorot_uniform',
 546                recurrent_initializer='orthogonal',
 547                bias_initializer='zeros',
 548                unit_forget_bias=True,
 549                kernel_regularizer=None,
 550                recurrent_regularizer=None,
 551                bias_regularizer=None,
 552                kernel_constraint=None,
 553                recurrent_constraint=None,
 554                bias_constraint=None,
 555                dropout=0.,
 556                recurrent_dropout=0.,
 557                **kwargs):
 558     super(ConvLSTM2DCell, self).__init__(**kwargs)
 559     self.filters = filters
 560     self.kernel_size = conv_utils.normalize_tuple(kernel_size, 2, 'kernel_size')
 561     self.strides = conv_utils.normalize_tuple(strides, 2, 'strides')
 562     self.padding = conv_utils.normalize_padding(padding)
 563     self.data_format = conv_utils.normalize_data_format(data_format)
 564     self.dilation_rate = conv_utils.normalize_tuple(dilation_rate, 2,
 565                                                     'dilation_rate')
 566     self.activation = activations.get(activation)
 567     self.recurrent_activation = activations.get(recurrent_activation)
 568     self.use_bias = use_bias
 569 
 570     self.kernel_initializer = initializers.get(kernel_initializer)
 571     self.recurrent_initializer = initializers.get(recurrent_initializer)
 572     self.bias_initializer = initializers.get(bias_initializer)
 573     self.unit_forget_bias = unit_forget_bias
 574 
 575     self.kernel_regularizer = regularizers.get(kernel_regularizer)
 576     self.recurrent_regularizer = regularizers.get(recurrent_regularizer)
 577     self.bias_regularizer = regularizers.get(bias_regularizer)
 578 
 579     self.kernel_constraint = constraints.get(kernel_constraint)
 580     self.recurrent_constraint = constraints.get(recurrent_constraint)
 581     self.bias_constraint = constraints.get(bias_constraint)
 582 
 583     self.dropout = min(1., max(0., dropout))
 584     self.recurrent_dropout = min(1., max(0., recurrent_dropout))
 585     self.state_size = (self.filters, self.filters)
 586     self._dropout_mask = None
 587     self._recurrent_dropout_mask = None
 588 
 589   def build(self, input_shape):
 590 
 591     if self.data_format == 'channels_first':
 592       channel_axis = 1
 593     else:
 594       channel_axis = -1
 595     if input_shape[channel_axis] is None:
 596       raise ValueError('The channel dimension of the inputs '
 597                        'should be defined. Found `None`.')
 598     input_dim = input_shape[channel_axis]
 599     kernel_shape = self.kernel_size + (input_dim, self.filters * 4)
 600     self.kernel_shape = kernel_shape
 601     recurrent_kernel_shape = self.kernel_size + (self.filters, self.filters * 4)
 602 
 603     self.kernel = self.add_weight(shape=kernel_shape,
 604                                   initializer=self.kernel_initializer,
 605                                   name='kernel',
 606                                   regularizer=self.kernel_regularizer,
 607                                   constraint=self.kernel_constraint)
 608     self.recurrent_kernel = self.add_weight(
 609         shape=recurrent_kernel_shape,
 610         initializer=self.recurrent_initializer,
 611         name='recurrent_kernel',
 612         regularizer=self.recurrent_regularizer,
 613         constraint=self.recurrent_constraint)
 614 
 615     if self.use_bias:
 616       if self.unit_forget_bias:
 617 
 618         def bias_initializer(_, *args, **kwargs):
 619           return K.concatenate([
 620               self.bias_initializer((self.filters,), *args, **kwargs),
 621               initializers.Ones()((self.filters,), *args, **kwargs),
 622               self.bias_initializer((self.filters * 2,), *args, **kwargs),
 623           ])
 624       else:
 625         bias_initializer = self.bias_initializer
 626       self.bias = self.add_weight(
 627           shape=(self.filters * 4,),
 628           name='bias',
 629           initializer=bias_initializer,
 630           regularizer=self.bias_regularizer,
 631           constraint=self.bias_constraint)
 632 
 633     else:
 634       self.bias = None
 635 
 636     self.kernel_i = self.kernel[:, :, :, :self.filters]
 637     self.recurrent_kernel_i = self.recurrent_kernel[:, :, :, :self.filters]
 638     self.kernel_f = self.kernel[:, :, :, self.filters: self.filters * 2]
 639     self.recurrent_kernel_f = self.recurrent_kernel[:, :, :, self.filters:
 640                                                     self.filters * 2]
 641     self.kernel_c = self.kernel[:, :, :, self.filters * 2: self.filters * 3]
 642     self.recurrent_kernel_c = self.recurrent_kernel[:, :, :, self.filters * 2:
 643                                                     self.filters * 3]
 644     self.kernel_o = self.kernel[:, :, :, self.filters * 3:]
 645     self.recurrent_kernel_o = self.recurrent_kernel[:, :, :, self.filters * 3:]
 646 
 647     if self.use_bias:
 648       self.bias_i = self.bias[:self.filters]
 649       self.bias_f = self.bias[self.filters: self.filters * 2]
 650       self.bias_c = self.bias[self.filters * 2: self.filters * 3]
 651       self.bias_o = self.bias[self.filters * 3:]
 652     else:
 653       self.bias_i = None
 654       self.bias_f = None
 655       self.bias_c = None
 656       self.bias_o = None
 657     self.built = True
 658 
 659   def call(self, inputs, states, training=None):
 660     if 0 < self.dropout < 1 and self._dropout_mask is None:
 661       self._dropout_mask = _generate_dropout_mask(
 662           K.ones_like(inputs),
 663           self.dropout,
 664           training=training,
 665           count=4)
 666     if (0 < self.recurrent_dropout < 1 and
 667         self._recurrent_dropout_mask is None):
 668       self._recurrent_dropout_mask = _generate_dropout_mask(
 669           K.ones_like(states[1]),
 670           self.recurrent_dropout,
 671           training=training,
 672           count=4)
 673 
 674     # dropout matrices for input units
 675     dp_mask = self._dropout_mask
 676     # dropout matrices for recurrent units
 677     rec_dp_mask = self._recurrent_dropout_mask
 678 
 679     h_tm1 = states[0]  # previous memory state
 680     c_tm1 = states[1]  # previous carry state
 681 
 682     if 0 < self.dropout < 1.:
 683       inputs_i = inputs * dp_mask[0]
 684       inputs_f = inputs * dp_mask[1]
 685       inputs_c = inputs * dp_mask[2]
 686       inputs_o = inputs * dp_mask[3]
 687     else:
 688       inputs_i = inputs
 689       inputs_f = inputs
 690       inputs_c = inputs
 691       inputs_o = inputs
 692 
 693     if 0 < self.recurrent_dropout < 1.:
 694       h_tm1_i = h_tm1 * rec_dp_mask[0]
 695       h_tm1_f = h_tm1 * rec_dp_mask[1]
 696       h_tm1_c = h_tm1 * rec_dp_mask[2]
 697       h_tm1_o = h_tm1 * rec_dp_mask[3]
 698     else:
 699       h_tm1_i = h_tm1
 700       h_tm1_f = h_tm1
 701       h_tm1_c = h_tm1
 702       h_tm1_o = h_tm1
 703 
 704     x_i = self.input_conv(inputs_i, self.kernel_i, self.bias_i,
 705                           padding=self.padding)
 706     x_f = self.input_conv(inputs_f, self.kernel_f, self.bias_f,
 707                           padding=self.padding)
 708     x_c = self.input_conv(inputs_c, self.kernel_c, self.bias_c,
 709                           padding=self.padding)
 710     x_o = self.input_conv(inputs_o, self.kernel_o, self.bias_o,
 711                           padding=self.padding)
 712     h_i = self.recurrent_conv(h_tm1_i,
 713                               self.recurrent_kernel_i)
 714     h_f = self.recurrent_conv(h_tm1_f,
 715                               self.recurrent_kernel_f)
 716     h_c = self.recurrent_conv(h_tm1_c,
 717                               self.recurrent_kernel_c)
 718     h_o = self.recurrent_conv(h_tm1_o,
 719                               self.recurrent_kernel_o)
 720 
 721     i = self.recurrent_activation(x_i + h_i)
 722     f = self.recurrent_activation(x_f + h_f)
 723     c = f * c_tm1 + i * self.activation(x_c + h_c)
 724     o = self.recurrent_activation(x_o + h_o)
 725     h = o * self.activation(c)
 726 
 727     if 0 < self.dropout + self.recurrent_dropout:
 728       if training is None:
 729         h._uses_learning_phase = True
 730 
 731     return h, [h, c]
 732 
 733   def input_conv(self, x, w, b=None, padding='valid'):
 734     conv_out = K.conv2d(x, w, strides=self.strides,
 735                         padding=padding,
 736                         data_format=self.data_format,
 737                         dilation_rate=self.dilation_rate)
 738     if b is not None:
 739       conv_out = K.bias_add(conv_out, b,
 740                             data_format=self.data_format)
 741     return conv_out
 742 
 743   def recurrent_conv(self, x, w):
 744     conv_out = K.conv2d(x, w, strides=(1, 1),
 745                         padding='same',
 746                         data_format=self.data_format)
 747     return conv_out
