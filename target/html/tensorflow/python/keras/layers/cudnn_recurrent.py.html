  36 class _CuDNNRNN(RNN):
  37   """Private base class for CuDNNGRU and CuDNNLSTM layers.
  38 
  39   Arguments:
  40     return_sequences: Boolean. Whether to return the last output
  41         in the output sequence, or the full sequence.
  42     return_state: Boolean. Whether to return the last state
  43         in addition to the output.
  44     go_backwards: Boolean (default False).
  45         If True, process the input sequence backwards and return the
  46         reversed sequence.
  47     stateful: Boolean (default False). If True, the last state
  48         for each sample at index i in a batch will be used as initial
  49         state for the sample of index i in the following batch.
  50   """
  51 
  52   def __init__(self,
  53                return_sequences=False,
  54                return_state=False,
  55                go_backwards=False,
  56                stateful=False,
  57                **kwargs):
  58     # We invoke the base layer's initializer directly here because we do not
  59     # want to create RNN cell instance.
  60     super(RNN, self).__init__(**kwargs)  # pylint: disable=bad-super-call
  61     self.return_sequences = return_sequences
  62     self.return_state = return_state
  63     self.go_backwards = go_backwards
  64     self.stateful = stateful
  65     self.supports_masking = False
  66     self.input_spec = [InputSpec(ndim=3)]
  67     if hasattr(self.cell.state_size, '__len__'):
  68       state_size = self.cell.state_size
  69     else:
  70       state_size = [self.cell.state_size]
  71     self.state_spec = [InputSpec(shape=(None, dim)) for dim in state_size]
  72     self.constants_spec = None
  73     self._states = None
  74     self._num_constants = None
  75     self._vector_shape = constant_op.constant([-1])
  76 
  77   def _canonical_to_params(self, weights, biases):
  78     weights = [array_ops.reshape(x, self._vector_shape) for x in weights]
  79     biases = [array_ops.reshape(x, self._vector_shape) for x in biases]
  80     return array_ops.concat(weights + biases, axis=0)
  81 
  82   def call(self, inputs, mask=None, training=None, initial_state=None):
  83     if isinstance(mask, list):
  84       mask = mask[0]
  85     if mask is not None:
  86       raise ValueError('Masking is not supported for CuDNN RNNs.')
  87 
  88     # input shape: `(samples, time (padded with zeros), input_dim)`
  89     # note that the .build() method of subclasses MUST define
  90     # self.input_spec and self.state_spec with complete input shapes.
  91     if isinstance(inputs, list):
  92       initial_state = inputs[1:]
  93       inputs = inputs[0]
  94     elif initial_state is not None:
  95       pass
  96     elif self.stateful:
  97       initial_state = self.states
  98     else:
  99       initial_state = self.get_initial_state(inputs)
 100 
 101     if len(initial_state) != len(self.states):
 102       raise ValueError('Layer has ' + str(len(self.states)) +
 103                        ' states but was passed ' + str(len(initial_state)) +
 104                        ' initial states.')
 105 
 106     if self.go_backwards:
 107       # Reverse time axis.
 108       inputs = K.reverse(inputs, 1)
 109     output, states = self._process_batch(inputs, initial_state)
 110 
 111     if self.stateful:
 112       updates = []
 113       for i in range(len(states)):
 114         updates.append(state_ops.assign(self.states[i], states[i]))
 115       self.add_update(updates, inputs)
 116 
 117     if self.return_state:
 118       return [output] + states
 119     else:
 120       return output
 121 
 122   def get_config(self):
 123     config = {
 124         'return_sequences': self.return_sequences,
 125         'return_state': self.return_state,
 126         'go_backwards': self.go_backwards,
 127         'stateful': self.stateful
 128     }
 129     base_config = super(  # pylint: disable=bad-super-call
 130         RNN, self).get_config()
 131     return dict(list(base_config.items()) + list(config.items()))
 132 
 133   @classmethod
 134   def from_config(cls, config):
 135     return cls(**config)
 136 
 137   @property
 138   def trainable_weights(self):
 139     if self.trainable and self.built:
 140       return [self.kernel, self.recurrent_kernel, self.bias]
 141     return []
 142 
 143   @property
 144   def non_trainable_weights(self):
 145     if not self.trainable and self.built:
 146       return [self.kernel, self.recurrent_kernel, self.bias]
 147     return []
 148 
 149   @property
 150   def losses(self):
 151     return super(RNN, self).losses
 152 
 153   def get_losses_for(self, inputs=None):
 154     return super(  # pylint: disable=bad-super-call
 155         RNN, self).get_losses_for(inputs=inputs)

 158 @tf_export('keras.layers.CuDNNGRU')
 159 class CuDNNGRU(_CuDNNRNN):
 160   """Fast GRU implementation backed by cuDNN.
 161 
 162   More information about cuDNN can be found on the [NVIDIA
 163   developer website](https://developer.nvidia.com/cudnn).
 164   Can only be run on GPU.
 165 
 166   Arguments:
 167       units: Positive integer, dimensionality of the output space.
 168       kernel_initializer: Initializer for the `kernel` weights matrix, used for
 169         the linear transformation of the inputs.
 170       recurrent_initializer: Initializer for the `recurrent_kernel` weights
 171         matrix, used for the linear transformation of the recurrent state.
 172       bias_initializer: Initializer for the bias vector.
 173       kernel_regularizer: Regularizer function applied to the `kernel` weights
 174         matrix.
 175       recurrent_regularizer: Regularizer function applied to the
 176         `recurrent_kernel` weights matrix.
 177       bias_regularizer: Regularizer function applied to the bias vector.
 178       activity_regularizer: Regularizer function applied to the output of the
 179         layer (its "activation").
 180       kernel_constraint: Constraint function applied to the `kernel` weights
 181         matrix.
 182       recurrent_constraint: Constraint function applied to the
 183         `recurrent_kernel` weights matrix.
 184       bias_constraint: Constraint function applied to the bias vector.
 185       return_sequences: Boolean. Whether to return the last output in the output
 186         sequence, or the full sequence.
 187       return_state: Boolean. Whether to return the last state in addition to the
 188         output.
 189       go_backwards: Boolean (default False). If True, process the input sequence
 190         backwards and return the reversed sequence.
 191       stateful: Boolean (default False). If True, the last state for each sample
 192         at index i in a batch will be used as initial state for the sample of
 193         index i in the following batch.
 194   """
 195 
 196   def __init__(self,
 197                units,
 198                kernel_initializer='glorot_uniform',
 199                recurrent_initializer='orthogonal',
 200                bias_initializer='zeros',
 201                kernel_regularizer=None,
 202                recurrent_regularizer=None,
 203                bias_regularizer=None,
 204                activity_regularizer=None,
 205                kernel_constraint=None,
 206                recurrent_constraint=None,
 207                bias_constraint=None,
 208                return_sequences=False,
 209                return_state=False,
 210                go_backwards=False,
 211                stateful=False,
 212                **kwargs):
 213     self.units = units
 214     cell_spec = collections.namedtuple('cell', 'state_size')
 215     self._cell = cell_spec(state_size=self.units)
 216     super(CuDNNGRU, self).__init__(
 217         return_sequences=return_sequences,
 218         return_state=return_state,
 219         go_backwards=go_backwards,
 220         stateful=stateful,
 221         **kwargs)
 222 
 223     self.kernel_initializer = initializers.get(kernel_initializer)
 224     self.recurrent_initializer = initializers.get(recurrent_initializer)
 225     self.bias_initializer = initializers.get(bias_initializer)
 226 
 227     self.kernel_regularizer = regularizers.get(kernel_regularizer)
 228     self.recurrent_regularizer = regularizers.get(recurrent_regularizer)
 229     self.bias_regularizer = regularizers.get(bias_regularizer)
 230     self.activity_regularizer = regularizers.get(activity_regularizer)
 231 
 232     self.kernel_constraint = constraints.get(kernel_constraint)
 233     self.recurrent_constraint = constraints.get(recurrent_constraint)
 234     self.bias_constraint = constraints.get(bias_constraint)
 235 
 236   @property
 237   def cell(self):
 238     return self._cell
 239 
 240   def build(self, input_shape):
 241     super(CuDNNGRU, self).build(input_shape)
 242     if isinstance(input_shape, list):
 243       input_shape = input_shape[0]
 244     input_dim = int(input_shape[-1])
 245 
 246     self.kernel = self.add_weight(
 247         shape=(input_dim, self.units * 3),
 248         name='kernel',
 249         initializer=self.kernel_initializer,
 250         regularizer=self.kernel_regularizer,
 251         constraint=self.kernel_constraint)
 252 
 253     self.recurrent_kernel = self.add_weight(
 254         shape=(self.units, self.units * 3),
 255         name='recurrent_kernel',
 256         initializer=self.recurrent_initializer,
 257         regularizer=self.recurrent_regularizer,
 258         constraint=self.recurrent_constraint)
 259 
 260     self.bias = self.add_weight(
 261         shape=(self.units * 6,),
 262         name='bias',
 263         initializer=self.bias_initializer,
 264         regularizer=self.bias_regularizer,
 265         constraint=self.bias_constraint)
 266 
 267     self.built = True
 268 
 269   def _process_batch(self, inputs, initial_state):
 270     inputs = array_ops.transpose(inputs, perm=(1, 0, 2))
 271     input_h = initial_state[0]
 272     input_h = array_ops.expand_dims(input_h, axis=0)
 273 
 274     params = self._canonical_to_params(
 275         weights=[
 276             self.kernel[:, self.units:self.units * 2],
 277             self.kernel[:, :self.units],
 278             self.kernel[:, self.units * 2:],
 279             self.recurrent_kernel[:, self.units:self.units * 2],
 280             self.recurrent_kernel[:, :self.units],
 281             self.recurrent_kernel[:, self.units * 2:],
 282         ],
 283         biases=[
 284             self.bias[self.units:self.units * 2],
 285             self.bias[:self.units],
 286             self.bias[self.units * 2:self.units * 3],
 287             self.bias[self.units * 4:self.units * 5],
 288             self.bias[self.units * 3:self.units * 4],
 289             self.bias[self.units * 5:],
 290         ],
 291     )
 292 
 293     outputs, h, _, _ = gen_cudnn_rnn_ops.cudnn_rnn(
 294         inputs,
 295         input_h=input_h,
 296         input_c=0,
 297         params=params,
 298         is_training=True,
 299         rnn_mode='gru')
 300 
 301     if self.stateful or self.return_state:
 302       h = h[0]
 303     if self.return_sequences:
 304       output = array_ops.transpose(outputs, perm=(1, 0, 2))
 305     else:
 306       output = outputs[-1]
 307     return output, [h]
 308 
 309   def get_config(self):
 310     config = {
 311         'units': self.units,
 312         'kernel_initializer': initializers.serialize(self.kernel_initializer),
 313         'recurrent_initializer':
 314             initializers.serialize(self.recurrent_initializer),
 315         'bias_initializer': initializers.serialize(self.bias_initializer),
 316         'kernel_regularizer': regularizers.serialize(self.kernel_regularizer),
 317         'recurrent_regularizer':
 318             regularizers.serialize(self.recurrent_regularizer),
 319         'bias_regularizer': regularizers.serialize(self.bias_regularizer),
 320         'activity_regularizer':
 321             regularizers.serialize(self.activity_regularizer),
 322         'kernel_constraint': constraints.serialize(self.kernel_constraint),
 323         'recurrent_constraint':
 324             constraints.serialize(self.recurrent_constraint),
 325         'bias_constraint': constraints.serialize(self.bias_constraint)
 326     }
 327     base_config = super(CuDNNGRU, self).get_config()
 328     return dict(list(base_config.items()) + list(config.items()))

 331 @tf_export('keras.layers.CuDNNLSTM')
 332 class CuDNNLSTM(_CuDNNRNN):
 333   """Fast LSTM implementation backed by cuDNN.
 334 
 335   More information about cuDNN can be found on the [NVIDIA
 336   developer website](https://developer.nvidia.com/cudnn).
 337   Can only be run on GPU.
 338 
 339   Arguments:
 340       units: Positive integer, dimensionality of the output space.
 341       kernel_initializer: Initializer for the `kernel` weights matrix, used for
 342         the linear transformation of the inputs.
 343       unit_forget_bias: Boolean. If True, add 1 to the bias of the forget gate
 344         at initialization. Setting it to true will also force
 345         `bias_initializer="zeros"`. This is recommended in [Jozefowicz et
 346         al.](http://www.jmlr.org/proceedings/papers/v37/jozefowicz15.pdf)
 347       recurrent_initializer: Initializer for the `recurrent_kernel` weights
 348         matrix, used for the linear transformation of the recurrent state.
 349       bias_initializer: Initializer for the bias vector.
 350       kernel_regularizer: Regularizer function applied to the `kernel` weights
 351         matrix.
 352       recurrent_regularizer: Regularizer function applied to the
 353         `recurrent_kernel` weights matrix.
 354       bias_regularizer: Regularizer function applied to the bias vector.
 355       activity_regularizer: Regularizer function applied to the output of the
 356         layer (its "activation").
 357       kernel_constraint: Constraint function applied to the `kernel` weights
 358         matrix.
 359       recurrent_constraint: Constraint function applied to the
 360         `recurrent_kernel` weights matrix.
 361       bias_constraint: Constraint function applied to the bias vector.
 362       return_sequences: Boolean. Whether to return the last output. in the
 363         output sequence, or the full sequence.
 364       return_state: Boolean. Whether to return the last state in addition to the
 365         output.
 366       go_backwards: Boolean (default False). If True, process the input sequence
 367         backwards and return the reversed sequence.
 368       stateful: Boolean (default False). If True, the last state for each sample
 369         at index i in a batch will be used as initial state for the sample of
 370         index i in the following batch.
 371   """
 372 
 373   def __init__(self,
 374                units,
 375                kernel_initializer='glorot_uniform',
 376                recurrent_initializer='orthogonal',
 377                bias_initializer='zeros',
 378                unit_forget_bias=True,
 379                kernel_regularizer=None,
 380                recurrent_regularizer=None,
 381                bias_regularizer=None,
 382                activity_regularizer=None,
 383                kernel_constraint=None,
 384                recurrent_constraint=None,
 385                bias_constraint=None,
 386                return_sequences=False,
 387                return_state=False,
 388                go_backwards=False,
 389                stateful=False,
 390                **kwargs):
 391     self.units = units
 392     cell_spec = collections.namedtuple('cell', 'state_size')
 393     self._cell = cell_spec(state_size=(self.units, self.units))
 394     super(CuDNNLSTM, self).__init__(
 395         return_sequences=return_sequences,
 396         return_state=return_state,
 397         go_backwards=go_backwards,
 398         stateful=stateful,
 399         **kwargs)
 400 
 401     self.kernel_initializer = initializers.get(kernel_initializer)
 402     self.recurrent_initializer = initializers.get(recurrent_initializer)
 403     self.bias_initializer = initializers.get(bias_initializer)
 404     self.unit_forget_bias = unit_forget_bias
 405 
 406     self.kernel_regularizer = regularizers.get(kernel_regularizer)
 407     self.recurrent_regularizer = regularizers.get(recurrent_regularizer)
 408     self.bias_regularizer = regularizers.get(bias_regularizer)
 409     self.activity_regularizer = regularizers.get(activity_regularizer)
 410 
 411     self.kernel_constraint = constraints.get(kernel_constraint)
 412     self.recurrent_constraint = constraints.get(recurrent_constraint)
 413     self.bias_constraint = constraints.get(bias_constraint)
 414 
 415   @property
 416   def cell(self):
 417     return self._cell
 418 
 419   def build(self, input_shape):
 420     super(CuDNNLSTM, self).build(input_shape)
 421     if isinstance(input_shape, list):
 422       input_shape = input_shape[0]
 423     input_dim = int(input_shape[-1])
 424 
 425     self.kernel = self.add_weight(
 426         shape=(input_dim, self.units * 4),
 427         name='kernel',
 428         initializer=self.kernel_initializer,
 429         regularizer=self.kernel_regularizer,
 430         constraint=self.kernel_constraint)
 431 
 432     self.recurrent_kernel = self.add_weight(
 433         shape=(self.units, self.units * 4),
 434         name='recurrent_kernel',
 435         initializer=self.recurrent_initializer,
 436         regularizer=self.recurrent_regularizer,
 437         constraint=self.recurrent_constraint)
 438 
 439     if self.unit_forget_bias:
 440 
 441       def bias_initializer(_, *args, **kwargs):
 442         return array_ops.concat([
 443             self.bias_initializer((self.units * 5,), *args, **kwargs),
 444             initializers.Ones()((self.units,), *args, **kwargs),
 445             self.bias_initializer((self.units * 2,), *args, **kwargs),
 446         ], axis=0)
 447     else:
 448       bias_initializer = self.bias_initializer
 449     self.bias = self.add_weight(
 450         shape=(self.units * 8,),
 451         name='bias',
 452         initializer=bias_initializer,
 453         regularizer=self.bias_regularizer,
 454         constraint=self.bias_constraint)
 455 
 456     self.built = True
 457 
 458   def _process_batch(self, inputs, initial_state):
 459     inputs = array_ops.transpose(inputs, perm=(1, 0, 2))
 460     input_h = initial_state[0]
 461     input_c = initial_state[1]
 462     input_h = array_ops.expand_dims(input_h, axis=0)
 463     input_c = array_ops.expand_dims(input_c, axis=0)
 464 
 465     params = self._canonical_to_params(
 466         weights=[
 467             self.kernel[:, :self.units],
 468             self.kernel[:, self.units:self.units * 2],
 469             self.kernel[:, self.units * 2:self.units * 3],
 470             self.kernel[:, self.units * 3:],
 471             self.recurrent_kernel[:, :self.units],
 472             self.recurrent_kernel[:, self.units:self.units * 2],
 473             self.recurrent_kernel[:, self.units * 2:self.units * 3],
 474             self.recurrent_kernel[:, self.units * 3:],
 475         ],
 476         biases=[
 477             self.bias[:self.units],
 478             self.bias[self.units:self.units * 2],
 479             self.bias[self.units * 2:self.units * 3],
 480             self.bias[self.units * 3:self.units * 4],
 481             self.bias[self.units * 4:self.units * 5],
 482             self.bias[self.units * 5:self.units * 6],
 483             self.bias[self.units * 6:self.units * 7],
 484             self.bias[self.units * 7:],
 485         ],
 486     )
 487 
 488     outputs, h, c, _ = gen_cudnn_rnn_ops.cudnn_rnn(
 489         inputs,
 490         input_h=input_h,
 491         input_c=input_c,
 492         params=params,
 493         is_training=True)
 494 
 495     if self.stateful or self.return_state:
 496       h = h[0]
 497       c = c[0]
 498     if self.return_sequences:
 499       output = array_ops.transpose(outputs, perm=(1, 0, 2))
 500     else:
 501       output = outputs[-1]
 502     return output, [h, c]
 503 
 504   def get_config(self):
 505     config = {
 506         'units': self.units,
 507         'kernel_initializer': initializers.serialize(self.kernel_initializer),
 508         'recurrent_initializer':
 509             initializers.serialize(self.recurrent_initializer),
 510         'bias_initializer': initializers.serialize(self.bias_initializer),
 511         'unit_forget_bias': self.unit_forget_bias,
 512         'kernel_regularizer': regularizers.serialize(self.kernel_regularizer),
 513         'recurrent_regularizer':
 514             regularizers.serialize(self.recurrent_regularizer),
 515         'bias_regularizer': regularizers.serialize(self.bias_regularizer),
 516         'activity_regularizer':
 517             regularizers.serialize(self.activity_regularizer),
 518         'kernel_constraint': constraints.serialize(self.kernel_constraint),
 519         'recurrent_constraint':
 520             constraints.serialize(self.recurrent_constraint),
 521         'bias_constraint': constraints.serialize(self.bias_constraint)
 522     }
 523     base_config = super(CuDNNLSTM, self).get_config()
 524     return dict(list(base_config.items()) + list(config.items()))
