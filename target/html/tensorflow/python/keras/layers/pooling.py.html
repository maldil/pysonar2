  31 class Pooling1D(Layer):
  32   """Pooling layer for arbitrary pooling functions, for 1D inputs.
  33 
  34   This class only exists for code reuse. It will never be an exposed API.
  35 
  36   Arguments:
  37     pool_function: The pooling function to apply, e.g. `tf.nn.max_pool`.
  38     pool_size: An integer or tuple/list of a single integer,
  39       representing the size of the pooling window.
  40     strides: An integer or tuple/list of a single integer, specifying the
  41       strides of the pooling operation.
  42     padding: A string. The padding method, either 'valid' or 'same'.
  43       Case-insensitive.
  44     data_format: A string, one of `channels_last` (default) or `channels_first`.
  45       The ordering of the dimensions in the inputs.
  46       `channels_last` corresponds to inputs with shape
  47       `(batch, length, channels)` while `channels_first` corresponds to
  48       inputs with shape `(batch, channels, length)`.
  49     name: A string, the name of the layer.
  50   """
  51 
  52   def __init__(self, pool_function, pool_size, strides,
  53                padding='valid', data_format=None,
  54                name=None, **kwargs):
  55     super(Pooling1D, self).__init__(name=name, **kwargs)
  56     if data_format is None:
  57       data_format = backend.image_data_format()
  58     if strides is None:
  59       strides = pool_size
  60     self.pool_function = pool_function
  61     self.pool_size = conv_utils.normalize_tuple(pool_size, 1, 'pool_size')
  62     self.strides = conv_utils.normalize_tuple(strides, 1, 'strides')
  63     self.padding = conv_utils.normalize_padding(padding)
  64     self.data_format = conv_utils.normalize_data_format(data_format)
  65     self.input_spec = InputSpec(ndim=3)
  66 
  67   def call(self, inputs):
  68     # There is no TF op for 1D pooling, hence we make the inputs 4D.
  69     if self.data_format == 'channels_last':
  70       # input is NWC, make it NHWC
  71       inputs = array_ops.expand_dims(inputs, 1)
  72       # pool on the W dim
  73       pool_shape = (1, 1) + self.pool_size + (1,)
  74       strides = (1, 1) + self.strides + (1,)
  75       data_format = 'NHWC'
  76     else:
  77       # input is NCW, make it NCHW
  78       inputs = array_ops.expand_dims(inputs, 2)
  79       # pool on the W dim
  80       pool_shape = (1, 1, 1) + self.pool_size
  81       strides = (1, 1, 1) + self.strides
  82       data_format = 'NCHW'
  83 
  84     outputs = self.pool_function(
  85         inputs,
  86         ksize=pool_shape,
  87         strides=strides,
  88         padding=self.padding.upper(),
  89         data_format=data_format)
  90 
  91     if self.data_format == 'channels_last':
  92       return array_ops.squeeze(outputs, 1)
  93     else:
  94       return array_ops.squeeze(outputs, 2)
  95 
  96   def compute_output_shape(self, input_shape):
  97     input_shape = tensor_shape.TensorShape(input_shape).as_list()
  98     length = conv_utils.conv_output_length(input_shape[1], self.pool_size[0],
  99                                            self.padding, self.strides[0])
 100     return tensor_shape.TensorShape([input_shape[0], length, input_shape[2]])
 101 
 102   def get_config(self):
 103     config = {
 104         'strides': self.strides,
 105         'pool_size': self.pool_size,
 106         'padding': self.padding
 107     }
 108     base_config = super(Pooling1D, self).get_config()
 109     return dict(list(base_config.items()) + list(config.items()))

 171 class Pooling2D(Layer):
 172   """Pooling layer for arbitrary pooling functions, for 2D inputs (e.g. images).
 173 
 174   This class only exists for code reuse. It will never be an exposed API.
 175 
 176   Arguments:
 177     pool_function: The pooling function to apply, e.g. `tf.nn.max_pool`.
 178     pool_size: An integer or tuple/list of 2 integers: (pool_height, pool_width)
 179       specifying the size of the pooling window.
 180       Can be a single integer to specify the same value for
 181       all spatial dimensions.
 182     strides: An integer or tuple/list of 2 integers,
 183       specifying the strides of the pooling operation.
 184       Can be a single integer to specify the same value for
 185       all spatial dimensions.
 186     padding: A string. The padding method, either 'valid' or 'same'.
 187       Case-insensitive.
 188     data_format: A string, one of `channels_last` (default) or `channels_first`.
 189       The ordering of the dimensions in the inputs.
 190       `channels_last` corresponds to inputs with shape
 191       `(batch, height, width, channels)` while `channels_first` corresponds to
 192       inputs with shape `(batch, channels, height, width)`.
 193     name: A string, the name of the layer.
 194   """
 195 
 196   def __init__(self, pool_function, pool_size, strides,
 197                padding='valid', data_format=None,
 198                name=None, **kwargs):
 199     super(Pooling2D, self).__init__(name=name, **kwargs)
 200     if data_format is None:
 201       data_format = backend.image_data_format()
 202     if strides is None:
 203       strides = pool_size
 204     self.pool_function = pool_function
 205     self.pool_size = conv_utils.normalize_tuple(pool_size, 2, 'pool_size')
 206     self.strides = conv_utils.normalize_tuple(strides, 2, 'strides')
 207     self.padding = conv_utils.normalize_padding(padding)
 208     self.data_format = conv_utils.normalize_data_format(data_format)
 209     self.input_spec = InputSpec(ndim=4)
 210 
 211   def call(self, inputs):
 212     if self.data_format == 'channels_last':
 213       pool_shape = (1,) + self.pool_size + (1,)
 214       strides = (1,) + self.strides + (1,)
 215     else:
 216       pool_shape = (1, 1) + self.pool_size
 217       strides = (1, 1) + self.strides
 218     outputs = self.pool_function(
 219         inputs,
 220         ksize=pool_shape,
 221         strides=strides,
 222         padding=self.padding.upper(),
 223         data_format=conv_utils.convert_data_format(self.data_format, 4))
 224     return outputs
 225 
 226   def compute_output_shape(self, input_shape):
 227     input_shape = tensor_shape.TensorShape(input_shape).as_list()
 228     if self.data_format == 'channels_first':
 229       rows = input_shape[2]
 230       cols = input_shape[3]
 231     else:
 232       rows = input_shape[1]
 233       cols = input_shape[2]
 234     rows = conv_utils.conv_output_length(rows, self.pool_size[0], self.padding,
 235                                          self.strides[0])
 236     cols = conv_utils.conv_output_length(cols, self.pool_size[1], self.padding,
 237                                          self.strides[1])
 238     if self.data_format == 'channels_first':
 239       return tensor_shape.TensorShape(
 240           [input_shape[0], input_shape[1], rows, cols])
 241     else:
 242       return tensor_shape.TensorShape(
 243           [input_shape[0], rows, cols, input_shape[3]])
 244 
 245   def get_config(self):
 246     config = {
 247         'pool_size': self.pool_size,
 248         'padding': self.padding,
 249         'strides': self.strides,
 250         'data_format': self.data_format
 251     }
 252     base_config = super(Pooling2D, self).get_config()
 253     return dict(list(base_config.items()) + list(config.items()))

 364 class Pooling3D(Layer):
 365   """Pooling layer for arbitrary pooling functions, for 3D inputs.
 366 
 367   This class only exists for code reuse. It will never be an exposed API.
 368 
 369   Arguments:
 370     pool_function: The pooling function to apply, e.g. `tf.nn.max_pool`.
 371     pool_size: An integer or tuple/list of 3 integers:
 372       (pool_depth, pool_height, pool_width)
 373       specifying the size of the pooling window.
 374       Can be a single integer to specify the same value for
 375       all spatial dimensions.
 376     strides: An integer or tuple/list of 3 integers,
 377       specifying the strides of the pooling operation.
 378       Can be a single integer to specify the same value for
 379       all spatial dimensions.
 380     padding: A string. The padding method, either 'valid' or 'same'.
 381       Case-insensitive.
 382     data_format: A string, one of `channels_last` (default) or `channels_first`.
 383       The ordering of the dimensions in the inputs.
 384       `channels_last` corresponds to inputs with shape
 385       `(batch, depth, height, width, channels)`
 386       while `channels_first` corresponds to
 387       inputs with shape `(batch, channels, depth, height, width)`.
 388     name: A string, the name of the layer.
 389   """
 390 
 391   def __init__(self, pool_function, pool_size, strides,
 392                padding='valid', data_format='channels_last',
 393                name=None, **kwargs):
 394     super(Pooling3D, self).__init__(name=name, **kwargs)
 395     if data_format is None:
 396       data_format = backend.image_data_format()
 397     if strides is None:
 398       strides = pool_size
 399     self.pool_function = pool_function
 400     self.pool_size = conv_utils.normalize_tuple(pool_size, 3, 'pool_size')
 401     self.strides = conv_utils.normalize_tuple(strides, 3, 'strides')
 402     self.padding = conv_utils.normalize_padding(padding)
 403     self.data_format = conv_utils.normalize_data_format(data_format)
 404     self.input_spec = InputSpec(ndim=5)
 405 
 406   def call(self, inputs):
 407     pool_shape = (1,) + self.pool_size + (1,)
 408     strides = (1,) + self.strides + (1,)
 409 
 410     if self.data_format == 'channels_first':
 411       # TF does not support `channels_first` with 3D pooling operations,
 412       # so we must handle this case manually.
 413       # TODO(fchollet): remove this when TF pooling is feature-complete.
 414       inputs = array_ops.transpose(inputs, (0, 2, 3, 4, 1))
 415 
 416     outputs = self.pool_function(
 417         inputs,
 418         ksize=pool_shape,
 419         strides=strides,
 420         padding=self.padding.upper())
 421 
 422     if self.data_format == 'channels_first':
 423       outputs = array_ops.transpose(outputs, (0, 4, 1, 2, 3))
 424     return outputs
 425 
 426   def compute_output_shape(self, input_shape):
 427     input_shape = tensor_shape.TensorShape(input_shape).as_list()
 428     if self.data_format == 'channels_first':
 429       len_dim1 = input_shape[2]
 430       len_dim2 = input_shape[3]
 431       len_dim3 = input_shape[4]
 432     else:
 433       len_dim1 = input_shape[1]
 434       len_dim2 = input_shape[2]
 435       len_dim3 = input_shape[3]
 436     len_dim1 = conv_utils.conv_output_length(len_dim1, self.pool_size[0],
 437                                              self.padding, self.strides[0])
 438     len_dim2 = conv_utils.conv_output_length(len_dim2, self.pool_size[1],
 439                                              self.padding, self.strides[1])
 440     len_dim3 = conv_utils.conv_output_length(len_dim3, self.pool_size[2],
 441                                              self.padding, self.strides[2])
 442     if self.data_format == 'channels_first':
 443       return tensor_shape.TensorShape(
 444           [input_shape[0], input_shape[1], len_dim1, len_dim2, len_dim3])
 445     else:
 446       return tensor_shape.TensorShape(
 447           [input_shape[0], len_dim1, len_dim2, len_dim3, input_shape[4]])
 448 
 449   def get_config(self):
 450     config = {
 451         'pool_size': self.pool_size,
 452         'padding': self.padding,
 453         'strides': self.strides,
 454         'data_format': self.data_format
 455     }
 456     base_config = super(Pooling3D, self).get_config()
 457     return dict(list(base_config.items()) + list(config.items()))
