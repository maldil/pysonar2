  45 def deserialize(config, custom_objects=None):
  46   """Instantiates a layer from a config dictionary.
  47 
  48   Arguments:
  49       config: dict of the form {'class_name': str, 'config': dict}
  50       custom_objects: dict mapping class names (or function names)
  51           of custom (non-Keras) objects to class/functions
  52 
  53   Returns:
  54       Layer instance (may be Model, Sequential, Layer...)
  55   """
  56   from tensorflow.python.keras import models  # pylint: disable=g-import-not-at-top
  57   globs = globals()  # All layers.
  58   globs['Model'] = models.Model
  59   globs['Sequential'] = models.Sequential
  60   return deserialize_keras_object(
  61       config,
  62       module_objects=globs,
  63       custom_objects=custom_objects,
  64       printable_module_name='layer')
