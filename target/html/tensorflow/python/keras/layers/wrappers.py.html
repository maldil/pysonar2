  35 @tf_export('keras.layers.Wrapper')
  36 class Wrapper(Layer):
  37   """Abstract wrapper base class.
  38 
  39   Wrappers take another layer and augment it in various ways.
  40   Do not use this class as a layer, it is only an abstract base class.
  41   Two usable wrappers are the `TimeDistributed` and `Bidirectional` wrappers.
  42 
  43   Arguments:
  44       layer: The layer to be wrapped.
  45   """
  46 
  47   def __init__(self, layer, **kwargs):
  48     assert isinstance(layer, Layer)
  49     self.layer = layer
  50     # Tracks mapping of Wrapper inputs to inner layer inputs. Useful when
  51     # the inner layer has update ops that depend on its inputs (as opposed
  52     # to the inputs to the Wrapper layer).
  53     self._input_map = {}
  54     super(Wrapper, self).__init__(**kwargs)
  55 
  56   def build(self, input_shape=None):
  57     self.built = True
  58 
  59   @property
  60   def activity_regularizer(self):
  61     if hasattr(self.layer, 'activity_regularizer'):
  62       return self.layer.activity_regularizer
  63     else:
  64       return None
  65 
  66   @property
  67   def trainable(self):
  68     return self.layer.trainable
  69 
  70   @trainable.setter
  71   def trainable(self, value):
  72     self.layer.trainable = value
  73 
  74   @property
  75   def trainable_weights(self):
  76     return self.layer.trainable_weights
  77 
  78   @property
  79   def non_trainable_weights(self):
  80     return self.layer.non_trainable_weights
  81 
  82   @property
  83   def updates(self):
  84     return self.layer.updates + self._updates
  85 
  86   @property
  87   def losses(self):
  88     return self.layer.losses + self._losses
  89 
  90   def get_weights(self):
  91     return self.layer.get_weights()
  92 
  93   def set_weights(self, weights):
  94     self.layer.set_weights(weights)
  95 
  96   def get_config(self):
  97     config = {
  98         'layer': {
  99             'class_name': self.layer.__class__.__name__,
 100             'config': self.layer.get_config()
 101         }
 102     }
 103     base_config = super(Wrapper, self).get_config()
 104     return dict(list(base_config.items()) + list(config.items()))
 105 
 106   @classmethod
 107   def from_config(cls, config, custom_objects=None):
 108     from tensorflow.python.keras.layers import deserialize as deserialize_layer  # pylint: disable=g-import-not-at-top
 109     layer = deserialize_layer(
 110         config.pop('layer'), custom_objects=custom_objects)
 111     return cls(layer, **config)

 114 @tf_export('keras.layers.TimeDistributed')
 115 class TimeDistributed(Wrapper):
 116   """This wrapper allows to apply a layer to every temporal slice of an input.
 117 
 118   The input should be at least 3D, and the dimension of index one
 119   will be considered to be the temporal dimension.
 120 
 121   Consider a batch of 32 samples,
 122   where each sample is a sequence of 10 vectors of 16 dimensions.
 123   The batch input shape of the layer is then `(32, 10, 16)`,
 124   and the `input_shape`, not including the samples dimension, is `(10, 16)`.
 125 
 126   You can then use `TimeDistributed` to apply a `Dense` layer
 127   to each of the 10 timesteps, independently:
 128 
 129   ```python
 130       # as the first layer in a model
 131       model = Sequential()
 132       model.add(TimeDistributed(Dense(8), input_shape=(10, 16)))
 133       # now model.output_shape == (None, 10, 8)
 134   ```
 135 
 136   The output will then have shape `(32, 10, 8)`.
 137 
 138   In subsequent layers, there is no need for the `input_shape`:
 139 
 140   ```python
 141       model.add(TimeDistributed(Dense(32)))
 142       # now model.output_shape == (None, 10, 32)
 143   ```
 144 
 145   The output will then have shape `(32, 10, 32)`.
 146 
 147   `TimeDistributed` can be used with arbitrary layers, not just `Dense`,
 148   for instance with a `Conv2D` layer:
 149 
 150   ```python
 151       model = Sequential()
 152       model.add(TimeDistributed(Conv2D(64, (3, 3)),
 153                                 input_shape=(10, 299, 299, 3)))
 154   ```
 155 
 156   Arguments:
 157       layer: a layer instance.
 158 
 159   Raises:
 160       ValueError: If not initialized with a `Layer` instance.
 161   """
 162 
 163   def __init__(self, layer, **kwargs):
 164     if not isinstance(layer, Layer):
 165       raise ValueError(
 166           'Please initialize `TimeDistributed` layer with a '
 167           '`Layer` instance. You passed: {input}'.format(input=layer))
 168     super(TimeDistributed, self).__init__(layer, **kwargs)
 169     self.supports_masking = True
 170     self._track_checkpointable(layer, name='layer')
 171 
 172   def _get_shape_tuple(self, init_tuple, tensor, start_idx, int_shape=None):
 173     """Finds non-specific dimensions in the static shapes.
 174 
 175     The static shapes are replaced with the corresponding dynamic shapes of the
 176     tensor.
 177 
 178     Arguments:
 179         init_tuple: a tuple, the first part of the output shape
 180         tensor: the tensor from which to get the (static and dynamic) shapes
 181             as the last part of the output shape
 182         start_idx: int, which indicate the first dimension to take from
 183             the static shape of the tensor
 184         int_shape: an alternative static shape to take as the last part
 185             of the output shape
 186     Returns:
 187         The new int_shape with the first part from init_tuple
 188         and the last part from either `int_shape` (if provided)
 189         or `tensor.shape`, where every `None` is replaced by
 190         the corresponding dimension from `tf.shape(tensor)`.
 191     """
 192     # replace all None in int_shape by K.shape
 193     if int_shape is None:
 194       int_shape = K.int_shape(tensor)[start_idx:]
 195     if not any(not s for s in int_shape):
 196       return init_tuple + tuple(int_shape)
 197     shape = K.shape(tensor)
 198     int_shape = list(int_shape)
 199     for i, s in enumerate(int_shape):
 200       if not s:
 201         int_shape[i] = shape[start_idx + i]
 202     return init_tuple + tuple(int_shape)
 203 
 204   def build(self, input_shape):
 205     input_shape = tensor_shape.TensorShape(input_shape).as_list()
 206     assert len(input_shape) >= 3
 207     self.input_spec = InputSpec(shape=input_shape)
 208     child_input_shape = [input_shape[0]] + input_shape[2:]
 209     if not self.layer.built:
 210       # The base layer class calls a conversion function on the input shape to
 211       # convert it to a TensorShape. The conversion function requires a
 212       # tuple which is why we cast the shape.
 213       self.layer.build(tuple(child_input_shape))
 214       self.layer.built = True
 215     super(TimeDistributed, self).build()
 216     self.built = True
 217 
 218   def compute_output_shape(self, input_shape):
 219     input_shape = tensor_shape.TensorShape(input_shape).as_list()
 220     child_input_shape = tensor_shape.TensorShape([input_shape[0]] +
 221                                                  input_shape[2:])
 222     child_output_shape = self.layer.compute_output_shape(
 223         child_input_shape).as_list()
 224     timesteps = input_shape[1]
 225     return tensor_shape.TensorShape([child_output_shape[0], timesteps] +
 226                                     child_output_shape[1:])
 227 
 228   def call(self, inputs, training=None, mask=None):
 229     kwargs = {}
 230     if generic_utils.has_arg(self.layer.call, 'training'):
 231       kwargs['training'] = training
 232     uses_learning_phase = False  # pylint: disable=redefined-outer-name
 233 
 234     input_shape = K.int_shape(inputs)
 235     if input_shape[0]:
 236       # batch size matters, use rnn-based implementation
 237       def step(x, _):
 238         global uses_learning_phase  # pylint: disable=global-variable-undefined
 239         output = self.layer.call(x, **kwargs)
 240         if hasattr(output, '_uses_learning_phase'):
 241           uses_learning_phase = (output._uses_learning_phase or
 242                                  uses_learning_phase)
 243         return output, []
 244 
 245       _, outputs, _ = K.rnn(
 246           step,
 247           inputs,
 248           initial_states=[],
 249           input_length=input_shape[1],
 250           unroll=False)
 251       y = outputs
 252     else:
 253       # No batch size specified, therefore the layer will be able
 254       # to process batches of any size.
 255       # We can go with reshape-based implementation for performance.
 256       input_length = input_shape[1]
 257       if not input_length:
 258         input_length = array_ops.shape(inputs)[1]
 259       inner_input_shape = self._get_shape_tuple((-1,), inputs, 2)
 260       # Shape: (num_samples * timesteps, ...). And track the
 261       # transformation in self._input_map.
 262       input_uid = generic_utils.object_list_uid(inputs)
 263       inputs = array_ops.reshape(inputs, inner_input_shape)
 264       self._input_map[input_uid] = inputs
 265       # (num_samples * timesteps, ...)
 266       if generic_utils.has_arg(self.layer.call, 'mask') and mask is not None:
 267         inner_mask_shape = self._get_shape_tuple((-1,), mask, 2)
 268         kwargs['mask'] = K.reshape(mask, inner_mask_shape)
 269       y = self.layer.call(inputs, **kwargs)
 270       if hasattr(y, '_uses_learning_phase'):
 271         uses_learning_phase = y._uses_learning_phase
 272       # Shape: (num_samples, timesteps, ...)
 273       output_shape = self.compute_output_shape(input_shape).as_list()
 274       output_shape = self._get_shape_tuple(
 275           (-1, input_length), y, 1, output_shape[2:])
 276       y = array_ops.reshape(y, output_shape)
 277 
 278     # Apply activity regularizer if any:
 279     if (hasattr(self.layer, 'activity_regularizer') and
 280         self.layer.activity_regularizer is not None):
 281       regularization_loss = self.layer.activity_regularizer(y)
 282       self.add_loss(regularization_loss, inputs)
 283 
 284     if uses_learning_phase:
 285       y._uses_learning_phase = True
 286     return y
 287 
 288   def compute_mask(self, inputs, mask=None):
 289     """Computes an output mask tensor for Embedding layer.
 290 
 291     This is based on the inputs, mask, and the inner layer.
 292     If batch size is specified:
 293     Simply return the input `mask`. (An rnn-based implementation with
 294     more than one rnn inputs is required but not supported in tf.keras yet.)
 295     Otherwise we call `compute_mask` of the inner layer at each time step.
 296     If the output mask at each time step is not `None`:
 297     (E.g., inner layer is Masking or RNN)
 298     Concatenate all of them and return the concatenation.
 299     If the output mask at each time step is `None` and the input mask is not
 300     `None`:(E.g., inner layer is Dense)
 301     Reduce the input_mask to 2 dimensions and return it.
 302     Otherwise (both the output mask and the input mask are `None`):
 303     (E.g., `mask` is not used at all)
 304     Return `None`.
 305 
 306     Arguments:
 307       inputs: Tensor with shape [batch size, timesteps, ...] indicating the
 308           input to TimeDistributed. If static shape information is available for
 309           "batch size", `mask` is returned unmodified.
 310       mask: Either None (indicating no masking) or a Tensor indicating the
 311           input mask for TimeDistributed. The shape can be static or dynamic.
 312 
 313     Returns:
 314       Either None (no masking), or a [batch size, timesteps, ...] Tensor with
 315       an output mask for the TimeDistributed layer with the shape beyond the
 316       second dimension being the value of the input mask shape(if the computed
 317       output mask is none), an output mask with the shape beyond the first
 318       dimension being the value of the mask shape(if mask is not None) or
 319       output mask with the shape beyond the first dimension being the
 320       value of the computed output shape.
 321 
 322     """
 323     # cases need to call the layer.compute_mask when input_mask is None:
 324     # Masking layer and Embedding layer with mask_zero
 325     input_shape = K.int_shape(inputs)
 326     if input_shape[0]:
 327       # batch size matters, we currently do not handle mask explicitly
 328       return mask
 329     inner_mask = mask
 330     if inner_mask is not None:
 331       inner_mask_shape = self._get_shape_tuple((-1,), mask, 2)
 332       inner_mask = K.reshape(inner_mask, inner_mask_shape)
 333     input_uid = generic_utils.object_list_uid(inputs)
 334     inner_inputs = self._input_map.get(input_uid, inputs)
 335     output_mask = self.layer.compute_mask(inner_inputs, inner_mask)
 336     if output_mask is None:
 337       if mask is None:
 338         return None
 339       # input_mask is not None, and output_mask is None:
 340       # we should return a not-None mask
 341       output_mask = mask
 342       for _ in range(2, len(K.int_shape(mask))):
 343         output_mask = K.any(output_mask, axis=-1)
 344     else:
 345       # output_mask is not None. We need to reshape it
 346       input_length = input_shape[1]
 347       if not input_length:
 348         input_length = K.shape(inputs)[1]
 349       output_mask_int_shape = K.int_shape(output_mask)
 350       if output_mask_int_shape is None:
 351         # if the output_mask does not have a static shape,
 352         # its shape must be the same as mask's
 353         if mask is not None:
 354           output_mask_int_shape = K.int_shape(mask)
 355         else:
 356           output_mask_int_shape = K.compute_output_shape(input_shape)[:-1]
 357       output_mask_shape = self._get_shape_tuple(
 358           (-1, input_length), output_mask, 1, output_mask_int_shape[1:])
 359       output_mask = K.reshape(output_mask, output_mask_shape)
 360     return output_mask

 363 @tf_export('keras.layers.Bidirectional')
 364 class Bidirectional(Wrapper):
 365   """Bidirectional wrapper for RNNs.
 366 
 367   Arguments:
 368       layer: `Recurrent` instance.
 369       merge_mode: Mode by which outputs of the
 370           forward and backward RNNs will be combined.
 371           One of {'sum', 'mul', 'concat', 'ave', None}.
 372           If None, the outputs will not be combined,
 373           they will be returned as a list.
 374 
 375   Raises:
 376       ValueError: If not initialized with a `Layer` instance or
 377           In case of invalid `merge_mode` argument.
 378 
 379   Examples:
 380 
 381   ```python
 382       model = Sequential()
 383       model.add(Bidirectional(LSTM(10, return_sequences=True), input_shape=(5,
 384       10)))
 385       model.add(Bidirectional(LSTM(10)))
 386       model.add(Dense(5))
 387       model.add(Activation('softmax'))
 388       model.compile(loss='categorical_crossentropy', optimizer='rmsprop')
 389   ```
 390   """
 391 
 392   def __init__(self, layer, merge_mode='concat', weights=None, **kwargs):
 393     if not isinstance(layer, Layer):
 394       raise ValueError(
 395           'Please initialize `Bidirectional` layer with a '
 396           '`Layer` instance. You passed: {input}'.format(input=layer))
 397     if merge_mode not in ['sum', 'mul', 'ave', 'concat', None]:
 398       raise ValueError('Invalid merge mode. '
 399                        'Merge mode should be one of '
 400                        '{"sum", "mul", "ave", "concat", None}')
 401     self.forward_layer = copy.copy(layer)
 402     config = layer.get_config()
 403     config['go_backwards'] = not config['go_backwards']
 404     self.backward_layer = layer.__class__.from_config(config)
 405     self.forward_layer._name = 'forward_' + self.forward_layer.name
 406     self.backward_layer._name = 'backward_' + self.backward_layer.name
 407     self.merge_mode = merge_mode
 408     if weights:
 409       nw = len(weights)
 410       self.forward_layer.initial_weights = weights[:nw // 2]
 411       self.backward_layer.initial_weights = weights[nw // 2:]
 412     self.stateful = layer.stateful
 413     self.return_sequences = layer.return_sequences
 414     self.return_state = layer.return_state
 415     self.supports_masking = True
 416     self._trainable = True
 417     self._num_constants = None
 418     super(Bidirectional, self).__init__(layer, **kwargs)
 419     self.input_spec = layer.input_spec
 420     self._track_checkpointable(self.forward_layer, name='forward_layer')
 421     self._track_checkpointable(self.backward_layer, name='backward_layer')
 422 
 423   @property
 424   def trainable(self):
 425     return self._trainable
 426 
 427   @trainable.setter
 428   def trainable(self, value):
 429     self._trainable = value
 430     self.forward_layer.trainable = value
 431     self.backward_layer.trainable = value
 432 
 433   def get_weights(self):
 434     return self.forward_layer.get_weights() + self.backward_layer.get_weights()
 435 
 436   def set_weights(self, weights):
 437     nw = len(weights)
 438     self.forward_layer.set_weights(weights[:nw // 2])
 439     self.backward_layer.set_weights(weights[nw // 2:])
 440 
 441   @tf_utils.shape_type_conversion
 442   def compute_output_shape(self, input_shape):
 443     output_shape = tuple(self.forward_layer.compute_output_shape(
 444         input_shape).as_list())
 445     if self.return_state:
 446       state_shape = output_shape[1:]
 447       output_shape = output_shape[0]
 448 
 449     if self.merge_mode == 'concat':
 450       output_shape = list(output_shape)
 451       output_shape[-1] *= 2
 452       output_shape = tuple(output_shape)
 453     elif self.merge_mode is None:
 454       output_shape = [output_shape, copy.copy(output_shape)]
 455 
 456     if self.return_state:
 457       if self.merge_mode is None:
 458         return output_shape + state_shape + copy.copy(state_shape)
 459       return [output_shape] + state_shape + copy.copy(state_shape)
 460     return output_shape
 461 
 462   def __call__(self, inputs, initial_state=None, constants=None, **kwargs):
 463     """`Bidirectional.__call__` implements the same API as the wrapped `RNN`."""
 464     inputs, initial_state, constants = _standardize_args(
 465         inputs, initial_state, constants, self._num_constants)
 466 
 467     if isinstance(inputs, list):
 468       if len(inputs) > 1:
 469         initial_state = inputs[1:]
 470       inputs = inputs[0]
 471 
 472     if initial_state is None and constants is None:
 473       return super(Bidirectional, self).__call__(inputs, **kwargs)
 474 
 475     # Applies the same workaround as in `RNN.__call__`
 476     additional_inputs = []
 477     additional_specs = []
 478     if initial_state is not None:
 479       # Check if `initial_state` can be splitted into half
 480       num_states = len(initial_state)
 481       if num_states % 2 > 0:
 482         raise ValueError(
 483             'When passing `initial_state` to a Bidirectional RNN, '
 484             'the state should be a list containing the states of '
 485             'the underlying RNNs. '
 486             'Found: ' + str(initial_state))
 487 
 488       kwargs['initial_state'] = initial_state
 489       additional_inputs += initial_state
 490       state_specs = [InputSpec(shape=K.int_shape(state))
 491                      for state in initial_state]
 492       self.forward_layer.state_spec = state_specs[:num_states // 2]
 493       self.backward_layer.state_spec = state_specs[num_states // 2:]
 494       additional_specs += state_specs
 495     if constants is not None:
 496       kwargs['constants'] = constants
 497       additional_inputs += constants
 498       constants_spec = [InputSpec(shape=K.int_shape(constant))
 499                         for constant in constants]
 500       self.forward_layer.constants_spec = constants_spec
 501       self.backward_layer.constants_spec = constants_spec
 502       additional_specs += constants_spec
 503 
 504       self._num_constants = len(constants)
 505       self.forward_layer._num_constants = self._num_constants
 506       self.backward_layer._num_constants = self._num_constants
 507 
 508     is_keras_tensor = K.is_keras_tensor(additional_inputs[0])
 509     for tensor in additional_inputs:
 510       if K.is_keras_tensor(tensor) != is_keras_tensor:
 511         raise ValueError('The initial state of a Bidirectional'
 512                          ' layer cannot be specified with a mix of'
 513                          ' Keras tensors and non-Keras tensors'
 514                          ' (a "Keras tensor" is a tensor that was'
 515                          ' returned by a Keras layer, or by `Input`)')
 516 
 517     if is_keras_tensor:
 518       # Compute the full input spec, including state
 519       full_input = [inputs] + additional_inputs
 520       full_input_spec = self.input_spec + additional_specs
 521 
 522       # Perform the call with temporarily replaced input_spec
 523       original_input_spec = self.input_spec
 524       self.input_spec = full_input_spec
 525       output = super(Bidirectional, self).__call__(full_input, **kwargs)
 526       self.input_spec = original_input_spec
 527       return output
 528     else:
 529       return super(Bidirectional, self).__call__(inputs, **kwargs)
 530 
 531   def call(self,
 532            inputs,
 533            training=None,
 534            mask=None,
 535            initial_state=None,
 536            constants=None):
 537     """`Bidirectional.call` implements the same API as the wrapped `RNN`."""
 538     kwargs = {}
 539     if generic_utils.has_arg(self.layer.call, 'training'):
 540       kwargs['training'] = training
 541     if generic_utils.has_arg(self.layer.call, 'mask'):
 542       kwargs['mask'] = mask
 543     if generic_utils.has_arg(self.layer.call, 'constants'):
 544       kwargs['constants'] = constants
 545 
 546     if initial_state is not None and generic_utils.has_arg(
 547         self.layer.call, 'initial_state'):
 548       forward_inputs = [inputs[0]]
 549       backward_inputs = [inputs[0]]
 550       pivot = len(initial_state) // 2 + 1
 551       # add forward initial state
 552       forward_state = inputs[1:pivot]
 553       forward_inputs += forward_state
 554       if self._num_constants is None:
 555         # add backward initial state
 556         backward_state = inputs[pivot:]
 557         backward_inputs += backward_state
 558       else:
 559         # add backward initial state
 560         backward_state = inputs[pivot:-self._num_constants]
 561         backward_inputs += backward_state
 562         # add constants for forward and backward layers
 563         forward_inputs += inputs[-self._num_constants:]
 564         backward_inputs += inputs[-self._num_constants:]
 565       y = self.forward_layer.call(forward_inputs,
 566                                   initial_state=forward_state, **kwargs)
 567       y_rev = self.backward_layer.call(backward_inputs,
 568                                        initial_state=backward_state, **kwargs)
 569     else:
 570       y = self.forward_layer.call(inputs, **kwargs)
 571       y_rev = self.backward_layer.call(inputs, **kwargs)
 572 
 573     if self.return_state:
 574       states = y[1:] + y_rev[1:]
 575       y = y[0]
 576       y_rev = y_rev[0]
 577 
 578     if self.return_sequences:
 579       y_rev = K.reverse(y_rev, 1)
 580     if self.merge_mode == 'concat':
 581       output = K.concatenate([y, y_rev])
 582     elif self.merge_mode == 'sum':
 583       output = y + y_rev
 584     elif self.merge_mode == 'ave':
 585       output = (y + y_rev) / 2
 586     elif self.merge_mode == 'mul':
 587       output = y * y_rev
 588     elif self.merge_mode is None:
 589       output = [y, y_rev]
 590 
 591     # Properly set learning phase
 592     if (getattr(y, '_uses_learning_phase', False) or
 593         getattr(y_rev, '_uses_learning_phase', False)):
 594       if self.merge_mode is None:
 595         for out in output:
 596           out._uses_learning_phase = True
 597       else:
 598         output._uses_learning_phase = True
 599 
 600     if self.return_state:
 601       if self.merge_mode is None:
 602         return output + states
 603       return [output] + states
 604     return output

 617   def compute_mask(self, inputs, mask):
 618     if isinstance(mask, list):
 619       mask = mask[0]
 620     if self.return_sequences:
 621       if not self.merge_mode:
 622         output_mask = [mask, mask]
 623       else:
 624         output_mask = mask
 625     else:
 626       output_mask = [None, None] if not self.merge_mode else None
 627 
 628     if self.return_state:
 629       states = self.forward_layer.states
 630       state_mask = [None for _ in states]
 631       if isinstance(output_mask, list):
 632         return output_mask + state_mask * 2
 633       return [output_mask] + state_mask * 2
 634     return output_mask
