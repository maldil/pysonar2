  31 @tf_export('keras.layers.GaussianNoise')
  32 class GaussianNoise(Layer):
  33   """Apply additive zero-centered Gaussian noise.
  34 
  35   This is useful to mitigate overfitting
  36   (you could see it as a form of random data augmentation).
  37   Gaussian Noise (GS) is a natural choice as corruption process
  38   for real valued inputs.
  39 
  40   As it is a regularization layer, it is only active at training time.
  41 
  42   Arguments:
  43       stddev: float, standard deviation of the noise distribution.
  44 
  45   Input shape:
  46       Arbitrary. Use the keyword argument `input_shape`
  47       (tuple of integers, does not include the samples axis)
  48       when using this layer as the first layer in a model.
  49 
  50   Output shape:
  51       Same shape as input.
  52   """
  53 
  54   def __init__(self, stddev, **kwargs):
  55     super(GaussianNoise, self).__init__(**kwargs)
  56     self.supports_masking = True
  57     self.stddev = stddev
  58 
  59   def call(self, inputs, training=None):
  60 
  61     def noised():
  62       return inputs + K.random_normal(
  63           shape=array_ops.shape(inputs), mean=0., stddev=self.stddev)
  64 
  65     return K.in_train_phase(noised, inputs, training=training)
  66 
  67   def get_config(self):
  68     config = {'stddev': self.stddev}
  69     base_config = super(GaussianNoise, self).get_config()
  70     return dict(list(base_config.items()) + list(config.items()))
  71 
  72   @tf_utils.shape_type_conversion
  73   def compute_output_shape(self, input_shape):
  74     return input_shape

  77 @tf_export('keras.layers.GaussianDropout')
  78 class GaussianDropout(Layer):
  79   """Apply multiplicative 1-centered Gaussian noise.
  80 
  81   As it is a regularization layer, it is only active at training time.
  82 
  83   Arguments:
  84       rate: float, drop probability (as with `Dropout`).
  85           The multiplicative noise will have
  86           standard deviation `sqrt(rate / (1 - rate))`.
  87 
  88   Input shape:
  89       Arbitrary. Use the keyword argument `input_shape`
  90       (tuple of integers, does not include the samples axis)
  91       when using this layer as the first layer in a model.
  92 
  93   Output shape:
  94       Same shape as input.
  95 
  96   """
  97 
  98   def __init__(self, rate, **kwargs):
  99     super(GaussianDropout, self).__init__(**kwargs)
 100     self.supports_masking = True
 101     self.rate = rate
 102 
 103   def call(self, inputs, training=None):
 104     if 0 < self.rate < 1:
 105 
 106       def noised():
 107         stddev = np.sqrt(self.rate / (1.0 - self.rate))
 108         return inputs * K.random_normal(
 109             shape=array_ops.shape(inputs), mean=1.0, stddev=stddev)
 110 
 111       return K.in_train_phase(noised, inputs, training=training)
 112     return inputs
 113 
 114   def get_config(self):
 115     config = {'rate': self.rate}
 116     base_config = super(GaussianDropout, self).get_config()
 117     return dict(list(base_config.items()) + list(config.items()))
 118 
 119   @tf_utils.shape_type_conversion
 120   def compute_output_shape(self, input_shape):
 121     return input_shape

 124 @tf_export('keras.layers.AlphaDropout')
 125 class AlphaDropout(Layer):
 126   """Applies Alpha Dropout to the input.
 127 
 128   Alpha Dropout is a `Dropout` that keeps mean and variance of inputs
 129   to their original values, in order to ensure the self-normalizing property
 130   even after this dropout.
 131   Alpha Dropout fits well to Scaled Exponential Linear Units
 132   by randomly setting activations to the negative saturation value.
 133 
 134   Arguments:
 135       rate: float, drop probability (as with `Dropout`).
 136           The multiplicative noise will have
 137           standard deviation `sqrt(rate / (1 - rate))`.
 138       seed: A Python integer to use as random seed.
 139 
 140   Input shape:
 141       Arbitrary. Use the keyword argument `input_shape`
 142       (tuple of integers, does not include the samples axis)
 143       when using this layer as the first layer in a model.
 144 
 145   Output shape:
 146       Same shape as input.
 147 
 148   """
 149 
 150   def __init__(self, rate, noise_shape=None, seed=None, **kwargs):
 151     super(AlphaDropout, self).__init__(**kwargs)
 152     self.rate = rate
 153     self.noise_shape = noise_shape
 154     self.seed = seed
 155     self.supports_masking = True
 156 
 157   def _get_noise_shape(self, inputs):
 158     return self.noise_shape if self.noise_shape else array_ops.shape(inputs)
 159 
 160   def call(self, inputs, training=None):
 161     if 0. < self.rate < 1.:
 162       noise_shape = self._get_noise_shape(inputs)
 163 
 164       def dropped_inputs(inputs=inputs, rate=self.rate, seed=self.seed):  # pylint: disable=missing-docstring
 165         alpha = 1.6732632423543772848170429916717
 166         scale = 1.0507009873554804934193349852946
 167         alpha_p = -alpha * scale
 168 
 169         kept_idx = math_ops.greater_equal(
 170             K.random_uniform(noise_shape, seed=seed), rate)
 171         kept_idx = math_ops.cast(kept_idx, K.floatx())
 172 
 173         # Get affine transformation params
 174         a = ((1 - rate) * (1 + rate * alpha_p**2))**-0.5
 175         b = -a * alpha_p * rate
 176 
 177         # Apply mask
 178         x = inputs * kept_idx + alpha_p * (1 - kept_idx)
 179 
 180         # Do affine transformation
 181         return a * x + b
 182 
 183       return K.in_train_phase(dropped_inputs, inputs, training=training)
 184     return inputs
 185 
 186   def get_config(self):
 187     config = {'rate': self.rate}
 188     base_config = super(AlphaDropout, self).get_config()
 189     return dict(list(base_config.items()) + list(config.items()))
 190 
 191   @tf_utils.shape_type_conversion
 192   def compute_output_shape(self, input_shape):
 193     return input_shape
