  31 @tf_export('keras.datasets.imdb.load_data')
  32 def load_data(path='imdb.npz',
  33               num_words=None,
  34               skip_top=0,
  35               maxlen=None,
  36               seed=113,
  37               start_char=1,
  38               oov_char=2,
  39               index_from=3,
  40               **kwargs):
  41   """Loads the IMDB dataset.
  42 
  43   Arguments:
  44       path: where to cache the data (relative to `~/.keras/dataset`).
  45       num_words: max number of words to include. Words are ranked
  46           by how often they occur (in the training set) and only
  47           the most frequent words are kept
  48       skip_top: skip the top N most frequently occurring words
  49           (which may not be informative).
  50       maxlen: sequences longer than this will be filtered out.
  51       seed: random seed for sample shuffling.
  52       start_char: The start of a sequence will be marked with this character.
  53           Set to 1 because 0 is usually the padding character.
  54       oov_char: words that were cut out because of the `num_words`
  55           or `skip_top` limit will be replaced with this character.
  56       index_from: index actual words with this index and higher.
  57       **kwargs: Used for backwards compatibility.
  58 
  59   Returns:
  60       Tuple of Numpy arrays: `(x_train, y_train), (x_test, y_test)`.
  61 
  62   Raises:
  63       ValueError: in case `maxlen` is so low
  64           that no input sequence could be kept.
  65 
  66   Note that the 'out of vocabulary' character is only used for
  67   words that were present in the training set but are not included
  68   because they're not making the `num_words` cut here.
  69   Words that were not seen in the training set but are in the test set
  70   have simply been skipped.
  71   """
  72   # Legacy support
  73   if 'nb_words' in kwargs:
  74     logging.warning('The `nb_words` argument in `load_data` '
  75                     'has been renamed `num_words`.')
  76     num_words = kwargs.pop('nb_words')
  77   if kwargs:
  78     raise TypeError('Unrecognized keyword arguments: ' + str(kwargs))
  79 
  80   origin_folder = 'https://storage.googleapis.com/tensorflow/tf-keras-datasets/'
  81   path = get_file(
  82       path,
  83       origin=origin_folder + 'imdb.npz',
  84       file_hash='599dadb1135973df5b59232a0e9a887c')
  85   with np.load(path) as f:
  86     x_train, labels_train = f['x_train'], f['y_train']
  87     x_test, labels_test = f['x_test'], f['y_test']
  88 
  89   np.random.seed(seed)
  90   indices = np.arange(len(x_train))
  91   np.random.shuffle(indices)
  92   x_train = x_train[indices]
  93   labels_train = labels_train[indices]
  94 
  95   indices = np.arange(len(x_test))
  96   np.random.shuffle(indices)
  97   x_test = x_test[indices]
  98   labels_test = labels_test[indices]
  99 
 100   xs = np.concatenate([x_train, x_test])
 101   labels = np.concatenate([labels_train, labels_test])
 102 
 103   if start_char is not None:
 104     xs = [[start_char] + [w + index_from for w in x] for x in xs]
 105   elif index_from:
 106     xs = [[w + index_from for w in x] for x in xs]
 107 
 108   if maxlen:
 109     xs, labels = _remove_long_seq(maxlen, xs, labels)
 110     if not xs:
 111       raise ValueError('After filtering for sequences shorter than maxlen=' +
 112                        str(maxlen) + ', no sequence was kept. '
 113                        'Increase maxlen.')
 114   if not num_words:
 115     num_words = max([max(x) for x in xs])
 116 
 117   # by convention, use 2 as OOV word
 118   # reserve 'index_from' (=3 by default) characters:
 119   # 0 (padding), 1 (start), 2 (OOV)
 120   if oov_char is not None:
 121     xs = [
 122         [w if (skip_top <= w < num_words) else oov_char for w in x] for x in xs
 123     ]
 124   else:
 125     xs = [[w for w in x if skip_top <= w < num_words] for x in xs]
 126 
 127   idx = len(x_train)
 128   x_train, y_train = np.array(xs[:idx]), np.array(labels[:idx])
 129   x_test, y_test = np.array(xs[idx:]), np.array(labels[idx:])
 130 
 131   return (x_train, y_train), (x_test, y_test)

 134 @tf_export('keras.datasets.imdb.get_word_index')
 135 def get_word_index(path='imdb_word_index.json'):
 136   """Retrieves the dictionary mapping word indices back to words.
 137 
 138   Arguments:
 139       path: where to cache the data (relative to `~/.keras/dataset`).
 140 
 141   Returns:
 142       The word index dictionary.
 143   """
 144   origin_folder = 'https://storage.googleapis.com/tensorflow/tf-keras-datasets/'
 145   path = get_file(
 146       path,
 147       origin=origin_folder + 'imdb_word_index.json',
 148       file_hash='bfafd718b763782e994055a2d397834f')
 149   with open(path) as f:
 150     return json.load(f)
