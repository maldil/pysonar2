  31 @tf_export('keras.datasets.reuters.load_data')
  32 def load_data(path='reuters.npz',
  33               num_words=None,
  34               skip_top=0,
  35               maxlen=None,
  36               test_split=0.2,
  37               seed=113,
  38               start_char=1,
  39               oov_char=2,
  40               index_from=3,
  41               **kwargs):
  42   """Loads the Reuters newswire classification dataset.
  43 
  44   Arguments:
  45       path: where to cache the data (relative to `~/.keras/dataset`).
  46       num_words: max number of words to include. Words are ranked
  47           by how often they occur (in the training set) and only
  48           the most frequent words are kept
  49       skip_top: skip the top N most frequently occurring words
  50           (which may not be informative).
  51       maxlen: truncate sequences after this length.
  52       test_split: Fraction of the dataset to be used as test data.
  53       seed: random seed for sample shuffling.
  54       start_char: The start of a sequence will be marked with this character.
  55           Set to 1 because 0 is usually the padding character.
  56       oov_char: words that were cut out because of the `num_words`
  57           or `skip_top` limit will be replaced with this character.
  58       index_from: index actual words with this index and higher.
  59       **kwargs: Used for backwards compatibility.
  60 
  61   Returns:
  62       Tuple of Numpy arrays: `(x_train, y_train), (x_test, y_test)`.
  63 
  64   Note that the 'out of vocabulary' character is only used for
  65   words that were present in the training set but are not included
  66   because they're not making the `num_words` cut here.
  67   Words that were not seen in the training set but are in the test set
  68   have simply been skipped.
  69   """
  70   # Legacy support
  71   if 'nb_words' in kwargs:
  72     logging.warning('The `nb_words` argument in `load_data` '
  73                     'has been renamed `num_words`.')
  74     num_words = kwargs.pop('nb_words')
  75   if kwargs:
  76     raise TypeError('Unrecognized keyword arguments: ' + str(kwargs))
  77 
  78   origin_folder = 'https://storage.googleapis.com/tensorflow/tf-keras-datasets/'
  79   path = get_file(
  80       path,
  81       origin=origin_folder + 'reuters.npz',
  82       file_hash='87aedbeb0cb229e378797a632c1997b6')
  83   with np.load(path) as f:
  84     xs, labels = f['x'], f['y']
  85 
  86   np.random.seed(seed)
  87   indices = np.arange(len(xs))
  88   np.random.shuffle(indices)
  89   xs = xs[indices]
  90   labels = labels[indices]
  91 
  92   if start_char is not None:
  93     xs = [[start_char] + [w + index_from for w in x] for x in xs]
  94   elif index_from:
  95     xs = [[w + index_from for w in x] for x in xs]
  96 
  97   if maxlen:
  98     xs, labels = _remove_long_seq(maxlen, xs, labels)
  99 
 100   if not num_words:
 101     num_words = max([max(x) for x in xs])
 102 
 103   # by convention, use 2 as OOV word
 104   # reserve 'index_from' (=3 by default) characters:
 105   # 0 (padding), 1 (start), 2 (OOV)
 106   if oov_char is not None:
 107     xs = [[w if skip_top <= w < num_words else oov_char for w in x] for x in xs]
 108   else:
 109     xs = [[w for w in x if skip_top <= w < num_words] for x in xs]
 110 
 111   idx = int(len(xs) * (1 - test_split))
 112   x_train, y_train = np.array(xs[:idx]), np.array(labels[:idx])
 113   x_test, y_test = np.array(xs[idx:]), np.array(labels[idx:])
 114 
 115   return (x_train, y_train), (x_test, y_test)

 118 @tf_export('keras.datasets.reuters.get_word_index')
 119 def get_word_index(path='reuters_word_index.json'):
 120   """Retrieves the dictionary mapping word indices back to words.
 121 
 122   Arguments:
 123       path: where to cache the data (relative to `~/.keras/dataset`).
 124 
 125   Returns:
 126       The word index dictionary.
 127   """
 128   origin_folder = 'https://storage.googleapis.com/tensorflow/tf-keras-datasets/'
 129   path = get_file(
 130       path,
 131       origin=origin_folder + 'reuters_word_index.json',
 132       file_hash='4d44cc38712099c9e383dc6e5f11a921')
 133   with open(path) as f:
 134     return json.load(f)
