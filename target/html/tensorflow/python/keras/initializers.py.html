  49 @tf_export('keras.initializers.TruncatedNormal',
  50            'keras.initializers.truncated_normal')
  51 class TruncatedNormal(TFTruncatedNormal):
  52   """Initializer that generates a truncated normal distribution.
  53 
  54   These values are similar to values from a `random_normal_initializer`
  55   except that values more than two standard deviations from the mean
  56   are discarded and re-drawn. This is the recommended initializer for
  57   neural network weights and filters.
  58 
  59   Args:
  60     mean: a python scalar or a scalar tensor. Mean of the random values to
  61       generate. Defaults to 0.
  62     stddev: a python scalar or a scalar tensor. Standard deviation of the random
  63       values to generate. Defaults to 0.05.
  64     seed: A Python integer. Used to create random seeds. See
  65       `tf.set_random_seed` for behavior.
  66     dtype: The data type. Only floating point types are supported.
  67   """
  68 
  69   def __init__(self, mean=0.0, stddev=0.05, seed=None, dtype=dtypes.float32):
  70     super(TruncatedNormal, self).__init__(
  71         mean=mean, stddev=stddev, seed=seed, dtype=dtype)

 136 @tf_export('keras.initializers.serialize')
 137 def serialize(initializer):
 138   return serialize_keras_object(initializer)
 139 
 140 
 141 @tf_export('keras.initializers.deserialize')
 142 def deserialize(config, custom_objects=None):
 143   return deserialize_keras_object(
 144       config,
 145       module_objects=globals(),
 146       custom_objects=custom_objects,
 147       printable_module_name='initializer')
 148 
 149 
 150 @tf_export('keras.initializers.get')
 151 def get(identifier):
 152   if identifier is None:
 153     return None
 154   if isinstance(identifier, dict):
 155     return deserialize(identifier)
 156   elif isinstance(identifier, six.string_types):
 157     config = {'class_name': str(identifier), 'config': {}}
 158     return deserialize(config)
 159   elif callable(identifier):
 160     return identifier
 161   else:
 162     raise ValueError('Could not interpret initializer identifier: ' +
 163                      str(identifier))
