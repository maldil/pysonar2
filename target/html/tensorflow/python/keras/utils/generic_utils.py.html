  40 @tf_export('keras.utils.CustomObjectScope')
  41 class CustomObjectScope(object):
  42   """Provides a scope that changes to `_GLOBAL_CUSTOM_OBJECTS` cannot escape.
  43 
  44   Code within a `with` statement will be able to access custom objects
  45   by name. Changes to global custom objects persist
  46   within the enclosing `with` statement. At end of the `with` statement,
  47   global custom objects are reverted to state
  48   at beginning of the `with` statement.
  49 
  50   Example:
  51 
  52   Consider a custom object `MyObject` (e.g. a class):
  53 
  54   ```python
  55       with CustomObjectScope({'MyObject':MyObject}):
  56           layer = Dense(..., kernel_regularizer='MyObject')
  57           # save, load, etc. will recognize custom object by name
  58   ```
  59   """
  60 
  61   def __init__(self, *args):
  62     self.custom_objects = args
  63     self.backup = None
  64 
  65   def __enter__(self):
  66     self.backup = _GLOBAL_CUSTOM_OBJECTS.copy()
  67     for objects in self.custom_objects:
  68       _GLOBAL_CUSTOM_OBJECTS.update(objects)
  69     return self
  70 
  71   def __exit__(self, *args, **kwargs):
  72     _GLOBAL_CUSTOM_OBJECTS.clear()
  73     _GLOBAL_CUSTOM_OBJECTS.update(self.backup)

  76 @tf_export('keras.utils.custom_object_scope')
  77 def custom_object_scope(*args):
  78   """Provides a scope that changes to `_GLOBAL_CUSTOM_OBJECTS` cannot escape.
  79 
  80   Convenience wrapper for `CustomObjectScope`.
  81   Code within a `with` statement will be able to access custom objects
  82   by name. Changes to global custom objects persist
  83   within the enclosing `with` statement. At end of the `with` statement,
  84   global custom objects are reverted to state
  85   at beginning of the `with` statement.
  86 
  87   Example:
  88 
  89   Consider a custom object `MyObject`
  90 
  91   ```python
  92       with custom_object_scope({'MyObject':MyObject}):
  93           layer = Dense(..., kernel_regularizer='MyObject')
  94           # save, load, etc. will recognize custom object by name
  95   ```
  96 
  97   Arguments:
  98       *args: Variable length list of dictionaries of name,
  99           class pairs to add to custom objects.
 100 
 101   Returns:
 102       Object of type `CustomObjectScope`.
 103   """
 104   return CustomObjectScope(*args)

 109   """Retrieves a live reference to the global dictionary of custom objects.
 110 
 111   Updating and clearing custom objects using `custom_object_scope`
 112   is preferred, but `get_custom_objects` can
 113   be used to directly access `_GLOBAL_CUSTOM_OBJECTS`.

 115   Example:
 116 
 117   ```python
 118       get_custom_objects().clear()
 119       get_custom_objects()['MyObject'] = MyObject

 122   Returns:
 123       Global dictionary of names to classes (`_GLOBAL_CUSTOM_OBJECTS`).
 124   """
 125   return _GLOBAL_CUSTOM_OBJECTS

 128 @tf_export('keras.utils.serialize_keras_object')
 129 def serialize_keras_object(instance):
 130   _, instance = tf_decorator.unwrap(instance)
 131   if instance is None:
 132     return None
 133   if hasattr(instance, 'get_config'):
 134     return {
 135         'class_name': instance.__class__.__name__,
 136         'config': instance.get_config()
 137     }
 138   if hasattr(instance, '__name__'):
 139     return instance.__name__
 140   else:
 141     raise ValueError('Cannot serialize', instance)

 144 @tf_export('keras.utils.deserialize_keras_object')
 145 def deserialize_keras_object(identifier,
 146                              module_objects=None,
 147                              custom_objects=None,
 148                              printable_module_name='object'):
 149   if isinstance(identifier, dict):
 150     # In this case we are dealing with a Keras config dictionary.
 151     config = identifier
 152     if 'class_name' not in config or 'config' not in config:
 153       raise ValueError('Improper config format: ' + str(config))
 154     class_name = config['class_name']
 155     if custom_objects and class_name in custom_objects:
 156       cls = custom_objects[class_name]
 157     elif class_name in _GLOBAL_CUSTOM_OBJECTS:
 158       cls = _GLOBAL_CUSTOM_OBJECTS[class_name]
 159     else:
 160       module_objects = module_objects or {}
 161       cls = module_objects.get(class_name)
 162       if cls is None:
 163         raise ValueError('Unknown ' + printable_module_name + ': ' + class_name)
 164     if hasattr(cls, 'from_config'):
 165       arg_spec = tf_inspect.getfullargspec(cls.from_config)
 166       custom_objects = custom_objects or {}
 167 
 168       if 'custom_objects' in arg_spec.args:
 169         return cls.from_config(
 170             config['config'],
 171             custom_objects=dict(
 172                 list(_GLOBAL_CUSTOM_OBJECTS.items()) +
 173                 list(custom_objects.items())))
 174       with CustomObjectScope(custom_objects):
 175         return cls.from_config(config['config'])
 176     else:
 177       # Then `cls` may be a function returning a class.
 178       # in this case by convention `config` holds
 179       # the kwargs of the function.
 180       custom_objects = custom_objects or {}
 181       with CustomObjectScope(custom_objects):
 182         return cls(**config['config'])
 183   elif isinstance(identifier, six.string_types):
 184     function_name = identifier
 185     if custom_objects and function_name in custom_objects:
 186       fn = custom_objects.get(function_name)
 187     elif function_name in _GLOBAL_CUSTOM_OBJECTS:
 188       fn = _GLOBAL_CUSTOM_OBJECTS[function_name]
 189     else:
 190       fn = module_objects.get(function_name)
 191       if fn is None:
 192         raise ValueError('Unknown ' + printable_module_name + ':' +
 193                          function_name)
 194     return fn
 195   else:
 196     raise ValueError('Could not interpret serialized ' + printable_module_name +
 197                      ': ' + identifier)

 200 def func_dump(func):
 201   """Serializes a user defined function.
 202 
 203   Arguments:
 204       func: the function to serialize.
 205 
 206   Returns:
 207       A tuple `(code, defaults, closure)`.
 208   """
 209   if os.name == 'nt':
 210     raw_code = marshal.dumps(func.__code__).replace(b'\\', b'/')
 211     code = codecs.encode(raw_code, 'base64').decode('ascii')
 212   else:
 213     raw_code = marshal.dumps(func.__code__)
 214     code = codecs.encode(raw_code, 'base64').decode('ascii')
 215   defaults = func.__defaults__
 216   if func.__closure__:
 217     closure = tuple(c.cell_contents for c in func.__closure__)
 218   else:
 219     closure = None
 220   return code, defaults, closure

 223 def func_load(code, defaults=None, closure=None, globs=None):
 224   """Deserializes a user defined function.
 225 
 226   Arguments:
 227       code: bytecode of the function.
 228       defaults: defaults of the function.
 229       closure: closure of the function.
 230       globs: dictionary of global objects.
 231 
 232   Returns:
 233       A function object.
 234   """
 235   if isinstance(code, (tuple, list)):  # unpack previous dump
 236     code, defaults, closure = code
 237     if isinstance(defaults, list):
 238       defaults = tuple(defaults)
 239 
 240   def ensure_value_to_cell(value):
 241     """Ensures that a value is converted to a python cell object.
 242 
 243     Arguments:
 244         value: Any value that needs to be casted to the cell type
 245 
 246     Returns:
 247         A value wrapped as a cell object (see function "func_load")
 248     """
 249     def dummy_fn():
 250       # pylint: disable=pointless-statement
 251       value  # just access it so it gets captured in .__closure__
 252 
 253     cell_value = dummy_fn.__closure__[0]
 254     if not isinstance(value, type(cell_value)):
 255       return cell_value
 256     else:
 257       return value
 258 
 259   if closure is not None:
 260     closure = tuple(ensure_value_to_cell(_) for _ in closure)
 261   try:
 262     raw_code = codecs.decode(code.encode('ascii'), 'base64')
 263   except (UnicodeEncodeError, binascii.Error):
 264     raw_code = code.encode('raw_unicode_escape')
 265   code = marshal.loads(raw_code)
 266   if globs is None:
 267     globs = globals()
 268   return python_types.FunctionType(
 269       code, globs, name=code.co_name, argdefs=defaults, closure=closure)

 272 def has_arg(fn, name, accept_all=False):
 273   """Checks if a callable accepts a given keyword argument.
 274 
 275   Arguments:
 276       fn: Callable to inspect.
 277       name: Check if `fn` can be called with `name` as a keyword argument.
 278       accept_all: What to return if there is no parameter called `name`
 279                   but the function accepts a `**kwargs` argument.
 280 
 281   Returns:
 282       bool, whether `fn` accepts a `name` keyword argument.
 283   """
 284   arg_spec = tf_inspect.getfullargspec(fn)
 285   if accept_all and arg_spec.varkw is not None:
 286     return True
 287   return name in arg_spec.args

 290 @tf_export('keras.utils.Progbar')
 291 class Progbar(object):
 292   """Displays a progress bar.
 293 
 294   Arguments:
 295       target: Total number of steps expected, None if unknown.
 296       width: Progress bar width on screen.
 297       verbose: Verbosity mode, 0 (silent), 1 (verbose), 2 (semi-verbose)
 298       stateful_metrics: Iterable of string names of metrics that
 299           should *not* be averaged over time. Metrics in this list
 300           will be displayed as-is. All others will be averaged
 301           by the progbar before display.
 302       interval: Minimum visual progress update interval (in seconds).
 303   """
 304 
 305   def __init__(self, target, width=30, verbose=1, interval=0.05,
 306                stateful_metrics=None):
 307     self.target = target
 308     self.width = width
 309     self.verbose = verbose
 310     self.interval = interval
 311     if stateful_metrics:
 312       self.stateful_metrics = set(stateful_metrics)
 313     else:
 314       self.stateful_metrics = set()
 315 
 316     self._dynamic_display = ((hasattr(sys.stdout, 'isatty') and
 317                               sys.stdout.isatty()) or
 318                              'ipykernel' in sys.modules or
 319                              'posix' in sys.modules)
 320     self._total_width = 0
 321     self._seen_so_far = 0
 322     # We use a dict + list to avoid garbage collection
 323     # issues found in OrderedDict
 324     self._values = {}
 325     self._values_order = []
 326     self._start = time.time()
 327     self._last_update = 0
 328 
 329   def update(self, current, values=None):
 330     """Updates the progress bar.
 331 
 332     Arguments:
 333         current: Index of current step.
 334         values: List of tuples:
 335             `(name, value_for_last_step)`.
 336             If `name` is in `stateful_metrics`,
 337             `value_for_last_step` will be displayed as-is.
 338             Else, an average of the metric over time will be displayed.
 339     """
 340     values = values or []
 341     for k, v in values:
 342       if k not in self._values_order:
 343         self._values_order.append(k)
 344       if k not in self.stateful_metrics:
 345         if k not in self._values:
 346           self._values[k] = [v * (current - self._seen_so_far),
 347                              current - self._seen_so_far]
 348         else:
 349           self._values[k][0] += v * (current - self._seen_so_far)
 350           self._values[k][1] += (current - self._seen_so_far)
 351       else:
 352         # Stateful metrics output a numeric value. This representation
 353         # means "take an average from a single value" but keeps the
 354         # numeric formatting.
 355         self._values[k] = [v, 1]
 356     self._seen_so_far = current
 357 
 358     now = time.time()
 359     info = ' - %.0fs' % (now - self._start)
 360     if self.verbose == 1:
 361       if (now - self._last_update < self.interval and
 362           self.target is not None and current < self.target):
 363         return
 364 
 365       prev_total_width = self._total_width
 366       if self._dynamic_display:
 367         sys.stdout.write('\b' * prev_total_width)
 368         sys.stdout.write('\r')
 369       else:
 370         sys.stdout.write('\n')
 371 
 372       if self.target is not None:
 373         numdigits = int(np.floor(np.log10(self.target))) + 1
 374         barstr = '%%%dd/%d [' % (numdigits, self.target)
 375         bar = barstr % current
 376         prog = float(current) / self.target
 377         prog_width = int(self.width * prog)
 378         if prog_width > 0:
 379           bar += ('=' * (prog_width - 1))
 380           if current < self.target:
 381             bar += '>'
 382           else:
 383             bar += '='
 384         bar += ('.' * (self.width - prog_width))
 385         bar += ']'
 386       else:
 387         bar = '%7d/Unknown' % current
 388 
 389       self._total_width = len(bar)
 390       sys.stdout.write(bar)
 391 
 392       if current:
 393         time_per_unit = (now - self._start) / current
 394       else:
 395         time_per_unit = 0
 396       if self.target is not None and current < self.target:
 397         eta = time_per_unit * (self.target - current)
 398         if eta > 3600:
 399           eta_format = '%d:%02d:%02d' % (eta // 3600,
 400                                          (eta % 3600) // 60,
 401                                          eta % 60)
 402         elif eta > 60:
 403           eta_format = '%d:%02d' % (eta // 60, eta % 60)
 404         else:
 405           eta_format = '%ds' % eta
 406 
 407         info = ' - ETA: %s' % eta_format
 408       else:
 409         if time_per_unit >= 1:
 410           info += ' %.0fs/step' % time_per_unit
 411         elif time_per_unit >= 1e-3:
 412           info += ' %.0fms/step' % (time_per_unit * 1e3)
 413         else:
 414           info += ' %.0fus/step' % (time_per_unit * 1e6)
 415 
 416       for k in self._values_order:
 417         info += ' - %s:' % k
 418         if isinstance(self._values[k], list):
 419           avg = np.mean(self._values[k][0] / max(1, self._values[k][1]))
 420           if abs(avg) > 1e-3:
 421             info += ' %.4f' % avg
 422           else:
 423             info += ' %.4e' % avg
 424         else:
 425           info += ' %s' % self._values[k]
 426 
 427       self._total_width += len(info)
 428       if prev_total_width > self._total_width:
 429         info += (' ' * (prev_total_width - self._total_width))
 430 
 431       if self.target is not None and current >= self.target:
 432         info += '\n'
 433 
 434       sys.stdout.write(info)
 435       sys.stdout.flush()
 436 
 437     elif self.verbose == 2:
 438       if self.target is None or current >= self.target:
 439         for k in self._values_order:
 440           info += ' - %s:' % k
 441           avg = np.mean(self._values[k][0] / max(1, self._values[k][1]))
 442           if avg > 1e-3:
 443             info += ' %.4f' % avg
 444           else:
 445             info += ' %.4e' % avg
 446         info += '\n'
 447 
 448         sys.stdout.write(info)
 449         sys.stdout.flush()
 450 
 451     self._last_update = now
 452 
 453   def add(self, n, values=None):
 454     self.update(self._seen_so_far + n, values)

 457 def make_batches(size, batch_size):
 458   """Returns a list of batch indices (tuples of indices).
 459 
 460   Arguments:
 461       size: Integer, total size of the data to slice into batches.
 462       batch_size: Integer, batch size.
 463 
 464   Returns:
 465       A list of tuples of array indices.
 466   """
 467   num_batches = int(np.ceil(size / float(batch_size)))
 468   return [(i * batch_size, min(size, (i + 1) * batch_size))
 469           for i in range(0, num_batches)]

 472 def slice_arrays(arrays, start=None, stop=None):
 473   """Slice an array or list of arrays.
 474 
 475   This takes an array-like, or a list of
 476   array-likes, and outputs:
 477       - arrays[start:stop] if `arrays` is an array-like
 478       - [x[start:stop] for x in arrays] if `arrays` is a list
 479 
 480   Can also work on list/array of indices: `slice_arrays(x, indices)`
 481 
 482   Arguments:
 483       arrays: Single array or list of arrays.
 484       start: can be an integer index (start index)
 485           or a list/array of indices
 486       stop: integer (stop index); should be None if
 487           `start` was a list.
 488 
 489   Returns:
 490       A slice of the array(s).
 491 
 492   Raises:
 493       ValueError: If the value of start is a list and stop is not None.
 494   """
 495   if arrays is None:
 496     return [None]
 497   if isinstance(start, list) and stop is not None:
 498     raise ValueError('The stop argument has to be None if the value of start '
 499                      'is a list.')
 500   elif isinstance(arrays, list):
 501     if hasattr(start, '__len__'):
 502       # hdf5 datasets only support list objects as indices
 503       if hasattr(start, 'shape'):
 504         start = start.tolist()
 505       return [None if x is None else x[start] for x in arrays]
 506     else:
 507       return [None if x is None else x[start:stop] for x in arrays]
 508   else:
 509     if hasattr(start, '__len__'):
 510       if hasattr(start, 'shape'):
 511         start = start.tolist()
 512       return arrays[start]
 513     elif hasattr(start, '__getitem__'):
 514       return arrays[start:stop]
 515     else:
 516       return [None]

 536 def object_list_uid(object_list):
 537   """Creates a single string from object ids."""
 538   object_list = nest.flatten(object_list)
 539   return ', '.join([str(abs(id(x))) for x in object_list])

 542 def to_snake_case(name):
 543   intermediate = re.sub('(.)([A-Z][a-z0-9]+)', r'\1_\2', name)
 544   insecure = re.sub('([a-z])([A-Z])', r'\1_\2', intermediate).lower()
 545   # If the class is private the name starts with "_" which is not secure
 546   # for creating scopes. We prefix the name with "private" in this case.
 547   if insecure[0] != '_':
 548     return insecure
 549   return 'private' + insecure

 552 def is_all_none(iterable_or_element):
 553   if not isinstance(iterable_or_element, (list, tuple)):
 554     iterable = [iterable_or_element]
 555   else:
 556     iterable = iterable_or_element
 557   # We cannot use Python's `any` because the iterable may return Tensors.
 558   for element in iterable:
 559     if element is not None:
 560       return False
 561   return True
