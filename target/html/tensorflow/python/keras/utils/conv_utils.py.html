  28 def convert_data_format(data_format, ndim):
  29   if data_format == 'channels_last':
  30     if ndim == 3:
  31       return 'NWC'
  32     elif ndim == 4:
  33       return 'NHWC'
  34     elif ndim == 5:
  35       return 'NDHWC'
  36     else:
  37       raise ValueError('Input rank not supported:', ndim)
  38   elif data_format == 'channels_first':
  39     if ndim == 3:
  40       return 'NCW'
  41     elif ndim == 4:
  42       return 'NCHW'
  43     elif ndim == 5:
  44       return 'NCDHW'
  45     else:
  46       raise ValueError('Input rank not supported:', ndim)
  47   else:
  48     raise ValueError('Invalid data_format:', data_format)

  51 def normalize_tuple(value, n, name):
  52   """Transforms a single integer or iterable of integers into an integer tuple.
  53 
  54   Arguments:
  55     value: The value to validate and convert. Could an int, or any iterable
  56       of ints.
  57     n: The size of the tuple to be returned.
  58     name: The name of the argument being validated, e.g. "strides" or
  59       "kernel_size". This is only used to format error messages.
  60 
  61   Returns:
  62     A tuple of n integers.
  63 
  64   Raises:
  65     ValueError: If something else than an int/long or iterable thereof was
  66       passed.
  67   """
  68   if isinstance(value, int):
  69     return (value,) * n
  70   else:
  71     try:
  72       value_tuple = tuple(value)
  73     except TypeError:
  74       raise ValueError('The `' + name + '` argument must be a tuple of ' +
  75                        str(n) + ' integers. Received: ' + str(value))
  76     if len(value_tuple) != n:
  77       raise ValueError('The `' + name + '` argument must be a tuple of ' +
  78                        str(n) + ' integers. Received: ' + str(value))
  79     for single_value in value_tuple:
  80       try:
  81         int(single_value)
  82       except (ValueError, TypeError):
  83         raise ValueError('The `' + name + '` argument must be a tuple of ' +
  84                          str(n) + ' integers. Received: ' + str(value) + ' '
  85                          'including element ' + str(single_value) + ' of type' +
  86                          ' ' + str(type(single_value)))
  87     return value_tuple

  90 def conv_output_length(input_length, filter_size, padding, stride, dilation=1):
  91   """Determines output length of a convolution given input length.
  92 
  93   Arguments:
  94       input_length: integer.
  95       filter_size: integer.
  96       padding: one of "same", "valid", "full", "causal"
  97       stride: integer.
  98       dilation: dilation rate, integer.
  99 
 100   Returns:
 101       The output length (integer).
 102   """
 103   if input_length is None:
 104     return None
 105   assert padding in {'same', 'valid', 'full', 'causal'}
 106   dilated_filter_size = filter_size + (filter_size - 1) * (dilation - 1)
 107   if padding in ['same', 'causal']:
 108     output_length = input_length
 109   elif padding == 'valid':
 110     output_length = input_length - dilated_filter_size + 1
 111   elif padding == 'full':
 112     output_length = input_length + dilated_filter_size - 1
 113   return (output_length + stride - 1) // stride

 116 def conv_input_length(output_length, filter_size, padding, stride):
 117   """Determines input length of a convolution given output length.
 118 
 119   Arguments:
 120       output_length: integer.
 121       filter_size: integer.
 122       padding: one of "same", "valid", "full".
 123       stride: integer.
 124 
 125   Returns:
 126       The input length (integer).
 127   """
 128   if output_length is None:
 129     return None
 130   assert padding in {'same', 'valid', 'full'}
 131   if padding == 'same':
 132     pad = filter_size // 2
 133   elif padding == 'valid':
 134     pad = 0
 135   elif padding == 'full':
 136     pad = filter_size - 1
 137   return (output_length - 1) * stride - 2 * pad + filter_size

 140 def deconv_output_length(input_length, filter_size, padding, stride):
 141   """Determines output length of a transposed convolution given input length.
 142 
 143   Arguments:
 144       input_length: integer.
 145       filter_size: integer.
 146       padding: one of "same", "valid", "full".
 147       stride: integer.
 148 
 149   Returns:
 150       The output length (integer).
 151   """
 152   if input_length is None:
 153     return None
 154   input_length *= stride
 155   if padding == 'valid':
 156     input_length += max(filter_size - stride, 0)
 157   elif padding == 'full':
 158     input_length -= (stride + filter_size - 2)
 159   return input_length

 162 def normalize_data_format(value):
 163   if value is None:
 164     value = backend.image_data_format()
 165   data_format = value.lower()
 166   if data_format not in {'channels_first', 'channels_last'}:
 167     raise ValueError('The `data_format` argument must be one of '
 168                      '"channels_first", "channels_last". Received: ' +
 169                      str(value))
 170   return data_format

 173 def normalize_padding(value):
 174   padding = value.lower()
 175   if padding not in {'valid', 'same', 'causal'}:
 176     raise ValueError('The `padding` argument must be one of '
 177                      '"valid", "same" (or "causal", only for `Conv1D). '
 178                      'Received: ' + str(padding))
 179   return padding

 182 def convert_kernel(kernel):
 183   """Converts a Numpy kernel matrix from Theano format to TensorFlow format.
 184 
 185   Also works reciprocally, since the transformation is its own inverse.
 186 
 187   Arguments:
 188       kernel: Numpy array (3D, 4D or 5D).
 189 
 190   Returns:
 191       The converted kernel.
 192 
 193   Raises:
 194       ValueError: in case of invalid kernel shape or invalid data_format.
 195   """
 196   kernel = np.asarray(kernel)
 197   if not 3 <= kernel.ndim <= 5:
 198     raise ValueError('Invalid kernel shape:', kernel.shape)
 199   slices = [slice(None, None, -1) for _ in range(kernel.ndim)]
 200   no_flip = (slice(None, None), slice(None, None))
 201   slices[-2:] = no_flip
 202   return np.copy(kernel[slices])

 205 def conv_kernel_mask(input_shape, kernel_shape, strides, padding):
 206   """Compute a mask representing the connectivity of a convolution operation.
 207 
 208   Assume a convolution with given parameters is applied to an input having N
 209   spatial dimensions with `input_shape = (d_in1, ..., d_inN)` to produce an
 210   output with shape `(d_out1, ..., d_outN)`. This method returns a boolean array
 211   of shape `(d_in1, ..., d_inN, d_out1, ..., d_outN)` with `True` entries
 212   indicating pairs of input and output locations that are connected by a weight.
 213 
 214   Example:
 215     ```python
 216         >>> input_shape = (4,)
 217         >>> kernel_shape = (2,)
 218         >>> strides = (1,)
 219         >>> padding = "valid"
 220         >>> conv_kernel_mask(input_shape, kernel_shape, strides, padding)
 221         array([[ True, False, False],
 222                [ True,  True, False],
 223                [False,  True,  True],
 224                [False, False,  True]], dtype=bool)
 225     ```
 226     where rows and columns correspond to inputs and outputs respectively.
 227 
 228 
 229   Args:
 230     input_shape: tuple of size N: `(d_in1, ..., d_inN)`,
 231                  spatial shape of the input.
 232     kernel_shape: tuple of size N, spatial shape of the convolutional kernel
 233                   / receptive field.
 234     strides: tuple of size N, strides along each spatial dimension.
 235     padding: type of padding, string `"same"` or `"valid"`.
 236 
 237   Returns:
 238     A boolean 2N-D `np.ndarray` of shape
 239     `(d_in1, ..., d_inN, d_out1, ..., d_outN)`, where `(d_out1, ..., d_outN)`
 240     is the spatial shape of the output. `True` entries in the mask represent
 241     pairs of input-output locations that are connected by a weight.
 242 
 243   Raises:
 244     ValueError: if `input_shape`, `kernel_shape` and `strides` don't have the
 245         same number of dimensions.
 246     NotImplementedError: if `padding` is not in {`"same"`, `"valid"`}.
 247   """
 248   if padding not in {'same', 'valid'}:
 249     raise NotImplementedError('Padding type %s not supported. '
 250                               'Only "valid" and "same" '
 251                               'are implemented.' % padding)
 252 
 253   in_dims = len(input_shape)
 254   if isinstance(kernel_shape, int):
 255     kernel_shape = (kernel_shape,) * in_dims
 256   if isinstance(strides, int):
 257     strides = (strides,) * in_dims
 258 
 259   kernel_dims = len(kernel_shape)
 260   stride_dims = len(strides)
 261   if kernel_dims != in_dims or stride_dims != in_dims:
 262     raise ValueError('Number of strides, input and kernel dimensions must all '
 263                      'match. Received: %d, %d, %d.' %
 264                      (stride_dims, in_dims, kernel_dims))
 265 
 266   output_shape = conv_output_shape(input_shape, kernel_shape, strides, padding)
 267 
 268   mask_shape = input_shape + output_shape
 269   mask = np.zeros(mask_shape, np.bool)
 270 
 271   output_axes_ticks = [range(dim) for dim in output_shape]
 272   for output_position in itertools.product(*output_axes_ticks):
 273     input_axes_ticks = conv_connected_inputs(input_shape,
 274                                              kernel_shape,
 275                                              output_position,
 276                                              strides,
 277                                              padding)
 278     for input_position in itertools.product(*input_axes_ticks):
 279       mask[input_position + output_position] = True
 280 
 281   return mask

 284 def conv_connected_inputs(input_shape,
 285                           kernel_shape,
 286                           output_position,
 287                           strides,
 288                           padding):
 289   """Return locations of the input connected to an output position.
 290 
 291   Assume a convolution with given parameters is applied to an input having N
 292   spatial dimensions with `input_shape = (d_in1, ..., d_inN)`. This method
 293   returns N ranges specifying the input region that was convolved with the
 294   kernel to produce the output at position
 295   `output_position = (p_out1, ..., p_outN)`.
 296 
 297   Example:
 298     ```python
 299         >>> input_shape = (4, 4)
 300         >>> kernel_shape = (2, 1)
 301         >>> output_position = (1, 1)
 302         >>> strides = (1, 1)
 303         >>> padding = "valid"
 304         >>> conv_connected_inputs(input_shape, kernel_shape, output_position,
 305         >>>                       strides, padding)
 306         [xrange(1, 3), xrange(1, 2)]
 307     ```
 308   Args:
 309     input_shape: tuple of size N: `(d_in1, ..., d_inN)`,
 310                  spatial shape of the input.
 311     kernel_shape: tuple of size N, spatial shape of the convolutional kernel
 312                   / receptive field.
 313     output_position: tuple of size N: `(p_out1, ..., p_outN)`,
 314                      a single position in the output of the convolution.
 315     strides: tuple of size N, strides along each spatial dimension.
 316     padding: type of padding, string `"same"` or `"valid"`.
 317 
 318   Returns:
 319     N ranges `[[p_in_left1, ..., p_in_right1], ...,
 320               [p_in_leftN, ..., p_in_rightN]]` specifying the region in the
 321     input connected to output_position.
 322   """
 323   ranges = []
 324 
 325   ndims = len(input_shape)
 326   for d in range(ndims):
 327     left_shift = int(kernel_shape[d] / 2)
 328     right_shift = kernel_shape[d] - left_shift
 329 
 330     center = output_position[d] * strides[d]
 331 
 332     if padding == 'valid':
 333       center += left_shift
 334 
 335     start = max(0, center - left_shift)
 336     end = min(input_shape[d], center + right_shift)
 337 
 338     ranges.append(range(start, end))
 339 
 340   return ranges

 343 def conv_output_shape(input_shape, kernel_shape, strides, padding):
 344   """Return the output shape of an N-D convolution.
 345 
 346   Forces dimensions where input is empty (size 0) to remain empty.
 347 
 348   Args:
 349     input_shape: tuple of size N: `(d_in1, ..., d_inN)`,
 350                  spatial shape of the input.
 351     kernel_shape: tuple of size N, spatial shape of the convolutional kernel
 352                   / receptive field.
 353     strides: tuple of size N, strides along each spatial dimension.
 354     padding: type of padding, string `"same"` or `"valid"`.
 355 
 356   Returns:
 357     tuple of size N: `(d_out1, ..., d_outN)`, spatial shape of the output.
 358   """
 359   dims = range(len(kernel_shape))
 360   output_shape = [conv_output_length(input_shape[d],
 361                                      kernel_shape[d],
 362                                      padding,
 363                                      strides[d])
 364                   for d in dims]
 365   output_shape = tuple([0 if input_shape[d] == 0 else output_shape[d]
 366                         for d in dims])
 367   return output_shape
