  55   def urlretrieve(url, filename, reporthook=None, data=None):
  56     """Replacement for `urlretrive` for Python 2.
  57 
  58     Under Python 2, `urlretrieve` relies on `FancyURLopener` from legacy
  59     `urllib` module, known to have issues with proxy management.
  60 
  61     Arguments:
  62         url: url to retrieve.
  63         filename: where to store the retrieved data locally.
  64         reporthook: a hook function that will be called once
  65             on establishment of the network connection and once
  66             after each block read thereafter.
  67             The hook will be passed three arguments;
  68             a count of blocks transferred so far,
  69             a block size in bytes, and the total size of the file.
  70         data: `data` argument passed to `urlopen`.
  71     """
  72 
  73     def chunk_read(response, chunk_size=8192, reporthook=None):
  74       content_type = response.info().get('Content-Length')
  75       total_size = -1
  76       if content_type is not None:
  77         total_size = int(content_type.strip())
  78       count = 0
  79       while True:
  80         chunk = response.read(chunk_size)
  81         count += 1
  82         if reporthook is not None:
  83           reporthook(count, chunk_size, total_size)
  84         if chunk:
  85           yield chunk
  86         else:
  87           break
  88 
  89     response = urlopen(url, data)
  90     with open(filename, 'wb') as fd:
  91       for chunk in chunk_read(response, reporthook=reporthook):
  92         fd.write(chunk)

 102 def _extract_archive(file_path, path='.', archive_format='auto'):
 103   """Extracts an archive if it matches tar, tar.gz, tar.bz, or zip formats.
 104 
 105   Arguments:
 106       file_path: path to the archive file
 107       path: path to extract the archive file
 108       archive_format: Archive format to try for extracting the file.
 109           Options are 'auto', 'tar', 'zip', and None.
 110           'tar' includes tar, tar.gz, and tar.bz files.
 111           The default 'auto' is ['tar', 'zip'].
 112           None or an empty list will return no matches found.
 113 
 114   Returns:
 115       True if a match was found and an archive extraction was completed,
 116       False otherwise.
 117   """
 118   if archive_format is None:
 119     return False
 120   if archive_format is 'auto':
 121     archive_format = ['tar', 'zip']
 122   if isinstance(archive_format, six.string_types):
 123     archive_format = [archive_format]
 124 
 125   for archive_type in archive_format:
 126     if archive_type is 'tar':
 127       open_fn = tarfile.open
 128       is_match_fn = tarfile.is_tarfile
 129     if archive_type is 'zip':
 130       open_fn = zipfile.ZipFile
 131       is_match_fn = zipfile.is_zipfile
 132 
 133     if is_match_fn(file_path):
 134       with open_fn(file_path) as archive:
 135         try:
 136           archive.extractall(path)
 137         except (tarfile.TarError, RuntimeError, KeyboardInterrupt):
 138           if os.path.exists(path):
 139             if os.path.isfile(path):
 140               os.remove(path)
 141             else:
 142               shutil.rmtree(path)
 143           raise
 144       return True
 145   return False

 148 @tf_export('keras.utils.get_file')
 149 def get_file(fname,
 150              origin,
 151              untar=False,
 152              md5_hash=None,
 153              file_hash=None,
 154              cache_subdir='datasets',
 155              hash_algorithm='auto',
 156              extract=False,
 157              archive_format='auto',
 158              cache_dir=None):
 159   """Downloads a file from a URL if it not already in the cache.
 160 
 161   By default the file at the url `origin` is downloaded to the
 162   cache_dir `~/.keras`, placed in the cache_subdir `datasets`,
 163   and given the filename `fname`. The final location of a file
 164   `example.txt` would therefore be `~/.keras/datasets/example.txt`.
 165 
 166   Files in tar, tar.gz, tar.bz, and zip formats can also be extracted.
 167   Passing a hash will verify the file after download. The command line
 168   programs `shasum` and `sha256sum` can compute the hash.
 169 
 170   Arguments:
 171       fname: Name of the file. If an absolute path `/path/to/file.txt` is
 172           specified the file will be saved at that location.
 173       origin: Original URL of the file.
 174       untar: Deprecated in favor of 'extract'.
 175           boolean, whether the file should be decompressed
 176       md5_hash: Deprecated in favor of 'file_hash'.
 177           md5 hash of the file for verification
 178       file_hash: The expected hash string of the file after download.
 179           The sha256 and md5 hash algorithms are both supported.
 180       cache_subdir: Subdirectory under the Keras cache dir where the file is
 181           saved. If an absolute path `/path/to/folder` is
 182           specified the file will be saved at that location.
 183       hash_algorithm: Select the hash algorithm to verify the file.
 184           options are 'md5', 'sha256', and 'auto'.
 185           The default 'auto' detects the hash algorithm in use.
 186       extract: True tries extracting the file as an Archive, like tar or zip.
 187       archive_format: Archive format to try for extracting the file.
 188           Options are 'auto', 'tar', 'zip', and None.
 189           'tar' includes tar, tar.gz, and tar.bz files.
 190           The default 'auto' is ['tar', 'zip'].
 191           None or an empty list will return no matches found.
 192       cache_dir: Location to store cached files, when None it
 193           defaults to the [Keras
 194             Directory](/faq/#where-is-the-keras-configuration-filed-stored).
 195 
 196   Returns:
 197       Path to the downloaded file
 198   """
 199   if cache_dir is None:
 200     cache_dir = os.path.join(os.path.expanduser('~'), '.keras')
 201   if md5_hash is not None and file_hash is None:
 202     file_hash = md5_hash
 203     hash_algorithm = 'md5'
 204   datadir_base = os.path.expanduser(cache_dir)
 205   if not os.access(datadir_base, os.W_OK):
 206     datadir_base = os.path.join('/tmp', '.keras')
 207   datadir = os.path.join(datadir_base, cache_subdir)
 208   if not os.path.exists(datadir):
 209     os.makedirs(datadir)
 210 
 211   if untar:
 212     untar_fpath = os.path.join(datadir, fname)
 213     fpath = untar_fpath + '.tar.gz'
 214   else:
 215     fpath = os.path.join(datadir, fname)
 216 
 217   download = False
 218   if os.path.exists(fpath):
 219     # File found; verify integrity if a hash was provided.
 220     if file_hash is not None:
 221       if not validate_file(fpath, file_hash, algorithm=hash_algorithm):
 222         print('A local file was found, but it seems to be '
 223               'incomplete or outdated because the ' + hash_algorithm +
 224               ' file hash does not match the original value of ' + file_hash +
 225               ' so we will re-download the data.')
 226         download = True
 227   else:
 228     download = True
 229 
 230   if download:
 231     print('Downloading data from', origin)
 232 
 233     class ProgressTracker(object):
 234       # Maintain progbar for the lifetime of download.
 235       # This design was chosen for Python 2.7 compatibility.
 236       progbar = None
 237 
 238     def dl_progress(count, block_size, total_size):
 239       if ProgressTracker.progbar is None:
 240         if total_size is -1:
 241           total_size = None
 242         ProgressTracker.progbar = Progbar(total_size)
 243       else:
 244         ProgressTracker.progbar.update(count * block_size)
 245 
 246     error_msg = 'URL fetch failure on {}: {} -- {}'
 247     try:
 248       try:
 249         urlretrieve(origin, fpath, dl_progress)
 250       except URLError as e:
 251         raise Exception(error_msg.format(origin, e.errno, e.reason))
 252       except HTTPError as e:
 253         raise Exception(error_msg.format(origin, e.code, e.msg))
 254     except (Exception, KeyboardInterrupt) as e:
 255       if os.path.exists(fpath):
 256         os.remove(fpath)
 257       raise
 258     ProgressTracker.progbar = None
 259 
 260   if untar:
 261     if not os.path.exists(untar_fpath):
 262       _extract_archive(fpath, datadir, archive_format='tar')
 263     return untar_fpath
 264 
 265   if extract:
 266     _extract_archive(fpath, datadir, archive_format)
 267 
 268   return fpath

 271 def _hash_file(fpath, algorithm='sha256', chunk_size=65535):
 272   """Calculates a file sha256 or md5 hash.
 273 
 274   Example:
 275 
 276   ```python
 277       >>> from keras.data_utils import _hash_file
 278       >>> _hash_file('/path/to/file.zip')
 279       'e3b0c44298fc1c149afbf4c8996fb92427ae41e4649b934ca495991b7852b855'
 280   ```
 281 
 282   Arguments:
 283       fpath: path to the file being validated
 284       algorithm: hash algorithm, one of 'auto', 'sha256', or 'md5'.
 285           The default 'auto' detects the hash algorithm in use.
 286       chunk_size: Bytes to read at a time, important for large files.
 287 
 288   Returns:
 289       The file hash
 290   """
 291   if (algorithm is 'sha256') or (algorithm is 'auto' and len(hash) is 64):
 292     hasher = hashlib.sha256()
 293   else:
 294     hasher = hashlib.md5()
 295 
 296   with open(fpath, 'rb') as fpath_file:
 297     for chunk in iter(lambda: fpath_file.read(chunk_size), b''):
 298       hasher.update(chunk)
 299 
 300   return hasher.hexdigest()

 303 def validate_file(fpath, file_hash, algorithm='auto', chunk_size=65535):
 304   """Validates a file against a sha256 or md5 hash.
 305 
 306   Arguments:
 307       fpath: path to the file being validated
 308       file_hash:  The expected hash string of the file.
 309           The sha256 and md5 hash algorithms are both supported.
 310       algorithm: Hash algorithm, one of 'auto', 'sha256', or 'md5'.
 311           The default 'auto' detects the hash algorithm in use.
 312       chunk_size: Bytes to read at a time, important for large files.
 313 
 314   Returns:
 315       Whether the file is valid
 316   """
 317   if ((algorithm is 'sha256') or
 318       (algorithm is 'auto' and len(file_hash) is 64)):
 319     hasher = 'sha256'
 320   else:
 321     hasher = 'md5'
 322 
 323   if str(_hash_file(fpath, hasher, chunk_size)) == str(file_hash):
 324     return True
 325   else:
 326     return False

 329 @tf_export('keras.utils.Sequence')
 330 class Sequence(object):
 331   """Base object for fitting to a sequence of data, such as a dataset.
 332 
 333   Every `Sequence` must implement the `__getitem__` and the `__len__` methods.
 334   If you want to modify your dataset between epochs you may implement
 335   `on_epoch_end`.
 336   The method `__getitem__` should return a complete batch.
 337 
 338   Notes:
 339 
 340   `Sequence` are a safer way to do multiprocessing. This structure guarantees
 341   that the network will only train once
 342    on each sample per epoch which is not the case with generators.
 343 
 344   Examples:
 345 
 346   ```python
 347       from skimage.io import imread
 348       from skimage.transform import resize
 349       import numpy as np
 350       import math
 351 
 352       # Here, `x_set` is list of path to the images
 353       # and `y_set` are the associated classes.
 354 
 355       class CIFAR10Sequence(Sequence):
 356 
 357           def __init__(self, x_set, y_set, batch_size):
 358               self.x, self.y = x_set, y_set
 359               self.batch_size = batch_size
 360 
 361           def __len__(self):
 362               return math.ceil(len(self.x) / self.batch_size)
 363 
 364           def __getitem__(self, idx):
 365               batch_x = self.x[idx * self.batch_size:(idx + 1) *
 366               self.batch_size]
 367               batch_y = self.y[idx * self.batch_size:(idx + 1) *
 368               self.batch_size]
 369 
 370               return np.array([
 371                   resize(imread(file_name), (200, 200))
 372                      for file_name in batch_x]), np.array(batch_y)
 373   ```
 374   """
 375 
 376   @abstractmethod
 377   def __getitem__(self, index):
 378     """Gets batch at position `index`.
 379 
 380     Arguments:
 381         index: position of the batch in the Sequence.
 382 
 383     Returns:
 384         A batch
 385     """
 386     raise NotImplementedError
 387 
 388   @abstractmethod
 389   def __len__(self):
 390     """Number of batch in the Sequence.
 391 
 392     Returns:
 393         The number of batches in the Sequence.
 394     """
 395     raise NotImplementedError
 396 
 397   def on_epoch_end(self):
 398     """Method called at the end of every epoch.
 399     """
 400     pass
 401 
 402   def __iter__(self):
 403     """Creates an infinite generator that iterate over the Sequence.
 404 
 405     Yields:
 406       Sequence items.
 407     """
 408     while True:
 409       for item in (self[i] for i in range(len(self))):
 410         yield item

 413 # Global variables to be shared across processes
 414 _SHARED_SEQUENCES = {}
 415 # We use a Value to provide unique id to different processes.
 416 _SEQUENCE_COUNTER = None
 417 
 418 
 419 def init_pool(seqs):
 420   global _SHARED_SEQUENCES
 421   _SHARED_SEQUENCES = seqs

 424 def get_index(uid, i):
 425   """Get the value from the Sequence `uid` at index `i`.
 426 
 427   To allow multiple Sequences to be used at the same time, we use `uid` to
 428   get a specific one. A single Sequence would cause the validation to
 429   overwrite the training Sequence.
 430 
 431   Arguments:
 432       uid: int, Sequence identifier
 433       i: index
 434 
 435   Returns:
 436       The value at index `i`.
 437   """
 438   return _SHARED_SEQUENCES[uid][i]

 441 @tf_export('keras.utils.SequenceEnqueuer')
 442 class SequenceEnqueuer(object):
 443   """Base class to enqueue inputs.
 444 
 445   The task of an Enqueuer is to use parallelism to speed up preprocessing.
 446   This is done with processes or threads.
 447 
 448   Examples:
 449 
 450   ```python
 451       enqueuer = SequenceEnqueuer(...)
 452       enqueuer.start()
 453       datas = enqueuer.get()
 454       for data in datas:
 455           # Use the inputs; training, evaluating, predicting.
 456           # ... stop sometime.
 457       enqueuer.close()
 458   ```
 459 
 460   The `enqueuer.get()` should be an infinite stream of datas.
 461 
 462   """
 463 
 464   @abstractmethod
 465   def is_running(self):
 466     raise NotImplementedError
 467 
 468   @abstractmethod
 469   def start(self, workers=1, max_queue_size=10):
 470     """Starts the handler's workers.
 471 
 472     Arguments:
 473         workers: number of worker threads
 474         max_queue_size: queue size
 475             (when full, threads could block on `put()`).
 476     """
 477     raise NotImplementedError
 478 
 479   @abstractmethod
 480   def stop(self, timeout=None):
 481     """Stop running threads and wait for them to exit, if necessary.
 482 
 483     Should be called by the same thread which called start().
 484 
 485     Arguments:
 486         timeout: maximum time to wait on thread.join()
 487     """
 488     raise NotImplementedError
 489 
 490   @abstractmethod
 491   def get(self):
 492     """Creates a generator to extract data from the queue.
 493 
 494     Skip the data if it is `None`.
 495 
 496     Returns:
 497         Generator yielding tuples `(inputs, targets)`
 498             or `(inputs, targets, sample_weights)`.
 499     """
 500     raise NotImplementedError

 503 @tf_export('keras.utils.OrderedEnqueuer')
 504 class OrderedEnqueuer(SequenceEnqueuer):
 505   """Builds a Enqueuer from a Sequence.
 506 
 507   Used in `fit_generator`, `evaluate_generator`, `predict_generator`.
 508 
 509   Arguments:
 510       sequence: A `keras.utils.data_utils.Sequence` object.
 511       use_multiprocessing: use multiprocessing if True, otherwise threading
 512       shuffle: whether to shuffle the data at the beginning of each epoch
 513   """
 514 
 515   def __init__(self, sequence, use_multiprocessing=False, shuffle=False):
 516     self.sequence = sequence
 517     self.use_multiprocessing = use_multiprocessing
 518 
 519     global _SEQUENCE_COUNTER
 520     if _SEQUENCE_COUNTER is None:
 521       try:
 522         _SEQUENCE_COUNTER = multiprocessing.Value('i', 0)
 523       except OSError:
 524         # In this case the OS does not allow us to use
 525         # multiprocessing. We resort to an int
 526         # for enqueuer indexing.
 527         _SEQUENCE_COUNTER = 0
 528 
 529     if isinstance(_SEQUENCE_COUNTER, int):
 530       self.uid = _SEQUENCE_COUNTER
 531       _SEQUENCE_COUNTER += 1
 532     else:
 533       # Doing Multiprocessing.Value += x is not process-safe.
 534       with _SEQUENCE_COUNTER.get_lock():
 535         self.uid = _SEQUENCE_COUNTER.value
 536         _SEQUENCE_COUNTER.value += 1
 537 
 538     self.shuffle = shuffle
 539     self.workers = 0
 540     self.executor_fn = None
 541     self.queue = None
 542     self.run_thread = None
 543     self.stop_signal = None
 544 
 545   def is_running(self):
 546     return self.stop_signal is not None and not self.stop_signal.is_set()
 547 
 548   def start(self, workers=1, max_queue_size=10):
 549     """Start the handler's workers.
 550 
 551     Arguments:
 552         workers: number of worker threads
 553         max_queue_size: queue size
 554             (when full, workers could block on `put()`)
 555     """
 556     if self.use_multiprocessing:
 557       self.executor_fn = lambda seqs: multiprocessing.Pool(  # pylint: disable=g-long-lambda
 558           workers, initializer=init_pool, initargs=(seqs,))
 559     else:
 560       # We do not need the init since it's threads.
 561       self.executor_fn = lambda _: ThreadPool(workers)
 562     self.workers = workers
 563     self.queue = queue.Queue(max_queue_size)
 564     self.stop_signal = threading.Event()
 565     self.run_thread = threading.Thread(target=self._run)
 566     self.run_thread.daemon = True
 567     self.run_thread.start()
 568 
 569   def _wait_queue(self):
 570     """Wait for the queue to be empty."""
 571     while True:
 572       time.sleep(0.1)
 573       if self.queue.unfinished_tasks == 0 or self.stop_signal.is_set():
 574         return
 575 
 576   def _run(self):
 577     """Submits request to the executor and queue the `Future` objects."""
 578     sequence = list(range(len(self.sequence)))
 579     self._send_sequence()  # Share the initial sequence
 580     while True:
 581       if self.shuffle:
 582         random.shuffle(sequence)
 583 
 584       with closing(self.executor_fn(_SHARED_SEQUENCES)) as executor:
 585         for i in sequence:
 586           if self.stop_signal.is_set():
 587             return
 588           self.queue.put(
 589               executor.apply_async(get_index, (self.uid, i)), block=True)
 590 
 591         # Done with the current epoch, waiting for the final batches
 592         self._wait_queue()
 593 
 594         if self.stop_signal.is_set():
 595           # We're done
 596           return
 597 
 598       # Call the internal on epoch end.
 599       self.sequence.on_epoch_end()
 600       self._send_sequence()  # Update the pool
 601 
 602   def get(self):
 603     """Creates a generator to extract data from the queue.
 604 
 605     Skip the data if it is `None`.
 606 
 607     Yields:
 608         The next element in the queue, i.e. a tuple
 609         `(inputs, targets)` or
 610         `(inputs, targets, sample_weights)`.
 611     """
 612     try:
 613       while self.is_running():
 614         inputs = self.queue.get(block=True).get()
 615         self.queue.task_done()
 616         if inputs is not None:
 617           yield inputs
 618     except Exception as e:  # pylint: disable=broad-except
 619       self.stop()
 620       six.raise_from(StopIteration(e), e)
 621 
 622   def _send_sequence(self):
 623     """Send current Sequence to all workers."""
 624     # For new processes that may spawn
 625     _SHARED_SEQUENCES[self.uid] = self.sequence
 626 
 627   def stop(self, timeout=None):
 628     """Stops running threads and wait for them to exit, if necessary.
 629 
 630     Should be called by the same thread which called `start()`.
 631 
 632     Arguments:
 633         timeout: maximum time to wait on `thread.join()`
 634     """
 635     self.stop_signal.set()
 636     with self.queue.mutex:
 637       self.queue.queue.clear()
 638       self.queue.unfinished_tasks = 0
 639       self.queue.not_full.notify()
 640     self.run_thread.join(timeout)
 641     _SHARED_SEQUENCES[self.uid] = None

 644 @tf_export('keras.utils.GeneratorEnqueuer')
 645 class GeneratorEnqueuer(SequenceEnqueuer):
 646   """Builds a queue out of a data generator.
 647 
 648   The provided generator can be finite in which case the class will throw
 649   a `StopIteration` exception.
 650 
 651   Used in `fit_generator`, `evaluate_generator`, `predict_generator`.
 652 
 653   Arguments:
 654       generator: a generator function which yields data
 655       use_multiprocessing: use multiprocessing if True, otherwise threading
 656       wait_time: time to sleep in-between calls to `put()`
 657       random_seed: Initial seed for workers,
 658           will be incremented by one for each worker.
 659   """
 660 
 661   def __init__(self,
 662                generator,
 663                use_multiprocessing=False,
 664                wait_time=0.05,
 665                seed=None):
 666     self.wait_time = wait_time
 667     self._generator = generator
 668     if os.name is 'nt' and use_multiprocessing is True:
 669       # On Windows, avoid **SYSTEMATIC** error in `multiprocessing`:
 670       # `TypeError: can't pickle generator objects`
 671       # => Suggest multithreading instead of multiprocessing on Windows
 672       raise ValueError('Using a generator with `use_multiprocessing=True`'
 673                        ' is not supported on Windows (no marshalling of'
 674                        ' generators across process boundaries). Instead,'
 675                        ' use single thread/process or multithreading.')
 676     else:
 677       self._use_multiprocessing = use_multiprocessing
 678     self._threads = []
 679     self._stop_event = None
 680     self._manager = None
 681     self.queue = None
 682     self.seed = seed
 683 
 684   def _data_generator_task(self):
 685     if self._use_multiprocessing is False:
 686       while not self._stop_event.is_set():
 687         with self.genlock:
 688           try:
 689             if (self.queue is not None and
 690                 self.queue.qsize() < self.max_queue_size):
 691               # On all OSes, avoid **SYSTEMATIC** error
 692               # in multithreading mode:
 693               # `ValueError: generator already executing`
 694               # => Serialize calls to
 695               # infinite iterator/generator's next() function
 696               generator_output = next(self._generator)
 697               self.queue.put((True, generator_output))
 698             else:
 699               time.sleep(self.wait_time)
 700           except StopIteration:
 701             break
 702           except Exception as e:  # pylint: disable=broad-except
 703             # Can't pickle tracebacks.
 704             # As a compromise, print the traceback and pickle None instead.
 705             if not hasattr(e, '__traceback__'):
 706               setattr(e, '__traceback__', sys.exc_info()[2])
 707             self.queue.put((False, e))
 708             self._stop_event.set()
 709             break
 710     else:
 711       while not self._stop_event.is_set():
 712         try:
 713           if (self.queue is not None and
 714               self.queue.qsize() < self.max_queue_size):
 715             generator_output = next(self._generator)
 716             self.queue.put((True, generator_output))
 717           else:
 718             time.sleep(self.wait_time)
 719         except StopIteration:
 720           break
 721         except Exception as e:  # pylint: disable=broad-except
 722           # Can't pickle tracebacks.
 723           # As a compromise, print the traceback and pickle None instead.
 724           traceback.print_exc()
 725           setattr(e, '__traceback__', None)
 726           self.queue.put((False, e))
 727           self._stop_event.set()
 728           break
 729 
 730   def start(self, workers=1, max_queue_size=10):
 731     """Kicks off threads which add data from the generator into the queue.
 732 
 733     Arguments:
 734         workers: number of worker threads
 735         max_queue_size: queue size
 736             (when full, threads could block on `put()`)
 737     """
 738     try:
 739       self.max_queue_size = max_queue_size
 740       if self._use_multiprocessing:
 741         self._manager = multiprocessing.Manager()
 742         self.queue = self._manager.Queue(maxsize=max_queue_size)
 743         self._stop_event = multiprocessing.Event()
 744       else:
 745         # On all OSes, avoid **SYSTEMATIC** error in multithreading mode:
 746         # `ValueError: generator already executing`
 747         # => Serialize calls to infinite iterator/generator's next() function
 748         self.genlock = threading.Lock()
 749         self.queue = queue.Queue(maxsize=max_queue_size)
 750         self._stop_event = threading.Event()
 751 
 752       for _ in range(workers):
 753         if self._use_multiprocessing:
 754           # Reset random seed else all children processes
 755           # share the same seed
 756           np.random.seed(self.seed)
 757           thread = multiprocessing.Process(target=self._data_generator_task)
 758           thread.daemon = True
 759           if self.seed is not None:
 760             self.seed += 1
 761         else:
 762           thread = threading.Thread(target=self._data_generator_task)
 763         self._threads.append(thread)
 764         thread.start()
 765     except:
 766       self.stop()
 767       raise
 768 
 769   def is_running(self):
 770     return self._stop_event is not None and not self._stop_event.is_set()
 771 
 772   def stop(self, timeout=None):
 773     """Stops running threads and wait for them to exit, if necessary.
 774 
 775     Should be called by the same thread which called `start()`.
 776 
 777     Arguments:
 778         timeout: maximum time to wait on `thread.join()`.
 779     """
 780     if self.is_running():
 781       self._stop_event.set()
 782 
 783     for thread in self._threads:
 784       if self._use_multiprocessing:
 785         if thread.is_alive():
 786           thread.terminate()
 787       else:
 788         # The thread.is_alive() test is subject to a race condition:
 789         # the thread could terminate right after the test and before the
 790         # join, rendering this test meaningless -> Call thread.join()
 791         # always, which is ok no matter what the status of the thread.
 792         thread.join(timeout)
 793 
 794     if self._manager:
 795       self._manager.shutdown()
 796 
 797     self._threads = []
 798     self._stop_event = None
 799     self.queue = None

 801   def get(self):
 802     """Creates a generator to extract data from the queue.
 803 
 804     Skip the data if it is `None`.
 805 
 806     Yields:
 807         The next element in the queue, i.e. a tuple
 808         `(inputs, targets)` or
 809         `(inputs, targets, sample_weights)`.
 810     """
 811     while self.is_running():
 812       if not self.queue.empty():
 813         success, value = self.queue.get()
 814         # Rethrow any exceptions found in the queue
 815         if not success:
 816           six.reraise(value.__class__, value, value.__traceback__)
 817         # Yield regular values
 818         if value is not None:
 819           yield value
 820       else:
 821         all_finished = all([not thread.is_alive() for thread in self._threads])
 822         if all_finished and self.queue.empty():
 823           raise StopIteration()
 824         else:
 825           time.sleep(self.wait_time)
 826 
 827     # Make sure to rethrow the first exception in the queue, if any
 828     while not self.queue.empty():
 829       success, value = self.queue.get()
 830       if not success:
 831         six.reraise(value.__class__, value, value.__traceback__)
