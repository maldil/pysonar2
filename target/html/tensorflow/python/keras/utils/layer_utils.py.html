  29 @tf_export('keras.utils.get_source_inputs')
  30 def get_source_inputs(tensor, layer=None, node_index=None):
  31   """Returns the list of input tensors necessary to compute `tensor`.
  32 
  33   Output will always be a list of tensors
  34   (potentially with 1 element).
  35 
  36   Arguments:
  37       tensor: The tensor to start from.
  38       layer: Origin layer of the tensor. Will be
  39           determined via tensor._keras_history if not provided.
  40       node_index: Origin node index of the tensor.
  41 
  42   Returns:
  43       List of input tensors.
  44   """
  45   if not hasattr(tensor, '_keras_history'):
  46     return tensor
  47 
  48   if layer is None or node_index:
  49     layer, node_index, _ = tensor._keras_history
  50   if not layer._inbound_nodes:
  51     return [tensor]
  52   else:
  53     node = layer._inbound_nodes[node_index]
  54     if not node.inbound_layers:
  55       # Reached an Input layer, stop recursion.
  56       return node.input_tensors
  57     else:
  58       source_tensors = []
  59       for i in range(len(node.inbound_layers)):
  60         x = node.input_tensors[i]
  61         layer = node.inbound_layers[i]
  62         node_index = node.node_indices[i]
  63         previous_sources = get_source_inputs(x, layer, node_index)
  64         # Avoid input redundancy.
  65         for x in previous_sources:
  66           if x not in source_tensors:
  67             source_tensors.append(x)
  68       return source_tensors

  71 def count_params(weights):
  72   """Count the total number of scalars composing the weights.
  73 
  74   Arguments:
  75       weights: An iterable containing the weights on which to compute params
  76 
  77   Returns:
  78       The total number of scalars composing the weights
  79   """
  80   return int(np.sum([np.prod(p.get_shape().as_list()) for p in set(weights)]))

  83 def print_summary(model, line_length=None, positions=None, print_fn=None):
  84   """Prints a summary of a model.
  85 
  86   Arguments:
  87       model: Keras model instance.
  88       line_length: Total length of printed lines
  89           (e.g. set this to adapt the display to different
  90           terminal window sizes).
  91       positions: Relative or absolute positions of log elements in each line.
  92           If not provided, defaults to `[.33, .55, .67, 1.]`.
  93       print_fn: Print function to use.
  94           It will be called on each line of the summary.
  95           You can set it to a custom function
  96           in order to capture the string summary.
  97           It defaults to `print` (prints to stdout).
  98   """
  99   if print_fn is None:
 100     print_fn = print
 101 
 102   if model.__class__.__name__ == 'Sequential':
 103     sequential_like = True
 104   elif not model._is_graph_network:
 105     # We treat subclassed models as a simple sequence of layers, for logging
 106     # purposes.
 107     sequential_like = True
 108   else:
 109     sequential_like = True
 110     nodes_by_depth = model._nodes_by_depth.values()
 111     nodes = []
 112     for v in nodes_by_depth:
 113       if (len(v) > 1) or (len(v) == 1 and len(v[0].inbound_layers) > 1):
 114         # if the model has multiple nodes
 115         # or if the nodes have multiple inbound_layers
 116         # the model is no longer sequential
 117         sequential_like = False
 118         break
 119       nodes += v
 120     if sequential_like:
 121       # search for shared layers
 122       for layer in model.layers:
 123         flag = False
 124         for node in layer._inbound_nodes:
 125           if node in nodes:
 126             if flag:
 127               sequential_like = False
 128               break
 129             else:
 130               flag = True
 131         if not sequential_like:
 132           break
 133 
 134   if sequential_like:
 135     line_length = line_length or 65
 136     positions = positions or [.45, .85, 1.]
 137     if positions[-1] <= 1:
 138       positions = [int(line_length * p) for p in positions]
 139     # header names for the different log elements
 140     to_display = ['Layer (type)', 'Output Shape', 'Param #']
 141   else:
 142     line_length = line_length or 98
 143     positions = positions or [.33, .55, .67, 1.]
 144     if positions[-1] <= 1:
 145       positions = [int(line_length * p) for p in positions]
 146     # header names for the different log elements
 147     to_display = ['Layer (type)', 'Output Shape', 'Param #', 'Connected to']
 148     relevant_nodes = []
 149     for v in model._nodes_by_depth.values():
 150       relevant_nodes += v
 151 
 152   def print_row(fields, positions):
 153     line = ''
 154     for i in range(len(fields)):
 155       if i > 0:
 156         line = line[:-1] + ' '
 157       line += str(fields[i])
 158       line = line[:positions[i]]
 159       line += ' ' * (positions[i] - len(line))
 160     print_fn(line)
 161 
 162   print_fn('_' * line_length)
 163   print_row(to_display, positions)
 164   print_fn('=' * line_length)
 165 
 166   def print_layer_summary(layer):
 167     """Prints a summary for a single layer.
 168 
 169     Arguments:
 170         layer: target layer.
 171     """
 172     try:
 173       output_shape = layer.output_shape
 174     except AttributeError:
 175       output_shape = 'multiple'
 176     except RuntimeError:  # output_shape unknown in Eager mode.
 177       output_shape = '?'
 178     name = layer.name
 179     cls_name = layer.__class__.__name__
 180     fields = [name + ' (' + cls_name + ')', output_shape, layer.count_params()]
 181     print_row(fields, positions)
 182 
 183   def print_layer_summary_with_connections(layer):
 184     """Prints a summary for a single layer (including topological connections).
 185 
 186     Arguments:
 187         layer: target layer.
 188     """
 189     try:
 190       output_shape = layer.output_shape
 191     except AttributeError:
 192       output_shape = 'multiple'
 193     connections = []
 194     for node in layer._inbound_nodes:
 195       if relevant_nodes and node not in relevant_nodes:
 196         # node is not part of the current network
 197         continue
 198       for i in range(len(node.inbound_layers)):
 199         inbound_layer = node.inbound_layers[i].name
 200         inbound_node_index = node.node_indices[i]
 201         inbound_tensor_index = node.tensor_indices[i]
 202         connections.append(inbound_layer + '[' + str(inbound_node_index) +
 203                            '][' + str(inbound_tensor_index) + ']')
 204 
 205     name = layer.name
 206     cls_name = layer.__class__.__name__
 207     if not connections:
 208       first_connection = ''
 209     else:
 210       first_connection = connections[0]
 211     fields = [
 212         name + ' (' + cls_name + ')', output_shape,
 213         layer.count_params(), first_connection
 214     ]
 215     print_row(fields, positions)
 216     if len(connections) > 1:
 217       for i in range(1, len(connections)):
 218         fields = ['', '', '', connections[i]]
 219         print_row(fields, positions)
 220 
 221   layers = model.layers
 222   for i in range(len(layers)):
 223     if sequential_like:
 224       print_layer_summary(layers[i])
 225     else:
 226       print_layer_summary_with_connections(layers[i])
 227     if i == len(layers) - 1:
 228       print_fn('=' * line_length)
 229     else:
 230       print_fn('_' * line_length)
 231 
 232   model._check_trainable_weights_consistency()
 233   if hasattr(model, '_collected_trainable_weights'):
 234     trainable_count = count_params(model._collected_trainable_weights)
 235   else:
 236     trainable_count = count_params(model.trainable_weights)
 237 
 238   non_trainable_count = count_params(model.non_trainable_weights)
 239 
 240   print_fn('Total params: {:,}'.format(trainable_count + non_trainable_count))
 241   print_fn('Trainable params: {:,}'.format(trainable_count))
 242   print_fn('Non-trainable params: {:,}'.format(non_trainable_count))
 243   print_fn('_' * line_length)

 246 def gather_trainable_weights(trainable, sub_layers, extra_variables):
 247   """Lists the trainable weights for an object with sub-layers.
 248 
 249   Args:
 250     trainable: Whether the object collecting the variables is trainable.
 251     sub_layers: A flat list of Layer objects owned by this object, to collect
 252       variables from.
 253     extra_variables: Any extra variables to include. Their `.trainable` property
 254       is used to categorize them.
 255 
 256   Returns:
 257     A list of collected trainable weights/variables.
 258   """
 259   if not trainable:
 260     return []
 261   weights = []
 262   for layer in sub_layers:
 263     weights += layer.trainable_weights
 264   trainable_extra_variables = [
 265       v for v in extra_variables if v.trainable]
 266   return weights + trainable_extra_variables

 269 def gather_non_trainable_weights(trainable, sub_layers, extra_variables):
 270   """Lists the non-trainable weights for an object with sub-layers.
 271 
 272   Args:
 273     trainable: Whether the object collecting the variables is trainable.
 274     sub_layers: A flat list of Layer objects owned by this object, to collect
 275       variables from.
 276     extra_variables: Any extra variables to include. Their `.trainable` property
 277       is used to categorize them.
 278 
 279   Returns:
 280     A list of collected non-trainable weights/variables.
 281   """
 282   trainable_extra_variables = []
 283   non_trainable_extra_variables = []
 284   for v in extra_variables:
 285     if v.trainable:
 286       trainable_extra_variables.append(v)
 287     else:
 288       non_trainable_extra_variables.append(v)
 289   weights = []
 290   for layer in sub_layers:
 291     weights += layer.non_trainable_weights
 292   if not trainable:
 293     trainable_weights = []
 294     for layer in sub_layers:
 295       trainable_weights += layer.trainable_weights
 296     return (trainable_weights + trainable_extra_variables
 297             + weights + non_trainable_extra_variables)
 298   return weights + non_trainable_extra_variables

 301 @tf_export('keras.utils.convert_all_kernels_in_model')
 302 def convert_all_kernels_in_model(model):
 303   """Converts all convolution kernels in a model from Theano to TensorFlow.
 304 
 305   Also works from TensorFlow to Theano.
 306 
 307   Arguments:
 308       model: target model for the conversion.
 309   """
 310   # Note: SeparableConvolution not included
 311   # since only supported by TF.
 312   conv_classes = {
 313       'Conv1D',
 314       'Conv2D',
 315       'Conv3D',
 316       'Conv2DTranspose',
 317   }
 318   to_assign = []
 319   for layer in model.layers:
 320     if layer.__class__.__name__ in conv_classes:
 321       original_kernel = K.get_value(layer.kernel)
 322       converted_kernel = convert_kernel(original_kernel)
 323       to_assign.append((layer.kernel, converted_kernel))
 324   K.batch_set_value(to_assign)

 327 def convert_dense_weights_data_format(dense,
 328                                       previous_feature_map_shape,
 329                                       target_data_format='channels_first'):
 330   """Utility useful when changing a convnet's `data_format`.
 331 
 332   When porting the weights of a convnet from one data format to the other,
 333   if the convnet includes a `Flatten` layer
 334   (applied to the last convolutional feature map)
 335   followed by a `Dense` layer, the weights of that `Dense` layer
 336   should be updated to reflect the new dimension ordering.
 337 
 338   Arguments:
 339       dense: The target `Dense` layer.
 340       previous_feature_map_shape: A shape tuple of 3 integers,
 341           e.g. `(512, 7, 7)`. The shape of the convolutional
 342           feature map right before the `Flatten` layer that
 343           came before the target `Dense` layer.
 344       target_data_format: One of "channels_last", "channels_first".
 345           Set it "channels_last"
 346           if converting a "channels_first" model to "channels_last",
 347           or reciprocally.
 348   """
 349   assert target_data_format in {'channels_last', 'channels_first'}
 350   kernel, bias = dense.get_weights()
 351   for i in range(kernel.shape[1]):
 352     if target_data_format == 'channels_first':
 353       c, h, w = previous_feature_map_shape
 354       original_fm_shape = (h, w, c)
 355       ki = kernel[:, i].reshape(original_fm_shape)
 356       ki = np.transpose(ki, (2, 0, 1))  # last -> first
 357     else:
 358       h, w, c = previous_feature_map_shape
 359       original_fm_shape = (c, h, w)
 360       ki = kernel[:, i].reshape(original_fm_shape)
 361       ki = np.transpose(ki, (1, 2, 0))  # first -> last
 362     kernel[:, i] = np.reshape(ki, (np.prod(previous_feature_map_shape),))
 363   dense.set_weights([kernel, bias])
