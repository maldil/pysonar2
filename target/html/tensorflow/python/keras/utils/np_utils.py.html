  24 @tf_export('keras.utils.to_categorical')
  25 def to_categorical(y, num_classes=None):
  26   """Converts a class vector (integers) to binary class matrix.
  27 
  28   E.g. for use with categorical_crossentropy.
  29 
  30   Arguments:
  31       y: class vector to be converted into a matrix
  32           (integers from 0 to num_classes).
  33       num_classes: total number of classes.
  34 
  35   Returns:
  36       A binary matrix representation of the input. The classes axis is placed
  37       last.
  38   """
  39   y = np.array(y, dtype='int')
  40   input_shape = y.shape
  41   if input_shape and input_shape[-1] == 1 and len(input_shape) > 1:
  42     input_shape = tuple(input_shape[:-1])
  43   y = y.ravel()
  44   if not num_classes:
  45     num_classes = np.max(y) + 1
  46   n = y.shape[0]
  47   categorical = np.zeros((n, num_classes), dtype=np.float32)
  48   categorical[np.arange(n), y] = 1
  49   output_shape = input_shape + (num_classes,)
  50   categorical = np.reshape(categorical, output_shape)
  51   return categorical

  54 @tf_export('keras.utils.normalize')
  55 def normalize(x, axis=-1, order=2):
  56   """Normalizes a Numpy array.
  57 
  58   Arguments:
  59       x: Numpy array to normalize.
  60       axis: axis along which to normalize.
  61       order: Normalization order (e.g. 2 for L2 norm).
  62 
  63   Returns:
  64       A normalized copy of the array.
  65   """
  66   l2 = np.atleast_1d(np.linalg.norm(x, order, axis))
  67   l2[l2 == 0] = 1
  68   return x / np.expand_dims(l2, axis)
