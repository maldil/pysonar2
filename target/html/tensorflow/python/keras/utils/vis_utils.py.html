  41 def _check_pydot():
  42   try:
  43     # Attempt to create an image of a blank graph
  44     # to check the pydot/graphviz installation.
  45     pydot.Dot.create(pydot.Dot())
  46   except Exception:
  47     # pydot raises a generic Exception here,
  48     # so no specific class can be caught.
  49     raise ImportError('Failed to import pydot. You must install pydot'
  50                       ' and graphviz for `pydotprint` to work.')

  53 def model_to_dot(model, show_shapes=False, show_layer_names=True, rankdir='TB'):
  54   """Convert a Keras model to dot format.
  55 
  56   Arguments:
  57       model: A Keras model instance.
  58       show_shapes: whether to display shape information.
  59       show_layer_names: whether to display layer names.
  60       rankdir: `rankdir` argument passed to PyDot,
  61           a string specifying the format of the plot:
  62           'TB' creates a vertical plot;
  63           'LR' creates a horizontal plot.
  64 
  65   Returns:
  66       A `pydot.Dot` instance representing the Keras model.
  67   """
  68   from tensorflow.python.keras.layers.wrappers import Wrapper
  69   from tensorflow.python.keras.models import Sequential
  70 
  71   _check_pydot()
  72   dot = pydot.Dot()
  73   dot.set('rankdir', rankdir)
  74   dot.set('concentrate', True)
  75   dot.set_node_defaults(shape='record')
  76 
  77   if isinstance(model, Sequential):
  78     if not model.built:
  79       model.build()
  80   layers = model.layers
  81 
  82   # Create graph nodes.
  83   for layer in layers:
  84     layer_id = str(id(layer))
  85 
  86     # Append a wrapped layer's label to node's label, if it exists.
  87     layer_name = layer.name
  88     class_name = layer.__class__.__name__
  89     if isinstance(layer, Wrapper):
  90       layer_name = '{}({})'.format(layer_name, layer.layer.name)
  91       child_class_name = layer.layer.__class__.__name__
  92       class_name = '{}({})'.format(class_name, child_class_name)
  93 
  94     # Create node's label.
  95     if show_layer_names:
  96       label = '{}: {}'.format(layer_name, class_name)
  97     else:
  98       label = class_name
  99 
 100     # Rebuild the label as a table including input/output shapes.
 101     if show_shapes:
 102       try:
 103         outputlabels = str(layer.output_shape)
 104       except AttributeError:
 105         outputlabels = 'multiple'
 106       if hasattr(layer, 'input_shape'):
 107         inputlabels = str(layer.input_shape)
 108       elif hasattr(layer, 'input_shapes'):
 109         inputlabels = ', '.join([str(ishape) for ishape in layer.input_shapes])
 110       else:
 111         inputlabels = 'multiple'
 112       label = '%s\n|{input:|output:}|{{%s}|{%s}}' % (label, inputlabels,
 113                                                      outputlabels)
 114     node = pydot.Node(layer_id, label=label)
 115     dot.add_node(node)
 116 
 117   # Connect nodes with edges.
 118   for layer in layers:
 119     layer_id = str(id(layer))
 120     for i, node in enumerate(layer._inbound_nodes):
 121       node_key = layer.name + '_ib-' + str(i)
 122       if node_key in model._network_nodes:  # pylint: disable=protected-access
 123         for inbound_layer in node.inbound_layers:
 124           inbound_layer_id = str(id(inbound_layer))
 125           layer_id = str(id(layer))
 126           dot.add_edge(pydot.Edge(inbound_layer_id, layer_id))
 127   return dot

 130 @tf_export('keras.utils.plot_model')
 131 def plot_model(model,
 132                to_file='model.png',
 133                show_shapes=False,
 134                show_layer_names=True,
 135                rankdir='TB'):
 136   """Converts a Keras model to dot format and save to a file.
 137 
 138   Arguments:
 139       model: A Keras model instance
 140       to_file: File name of the plot image.
 141       show_shapes: whether to display shape information.
 142       show_layer_names: whether to display layer names.
 143       rankdir: `rankdir` argument passed to PyDot,
 144           a string specifying the format of the plot:
 145           'TB' creates a vertical plot;
 146           'LR' creates a horizontal plot.
 147   """
 148   dot = model_to_dot(model, show_shapes, show_layer_names, rankdir)
 149   _, extension = os.path.splitext(to_file)
 150   if not extension:
 151     extension = 'png'
 152   else:
 153     extension = extension[1:]
 154   dot.write(to_file, format=extension)
