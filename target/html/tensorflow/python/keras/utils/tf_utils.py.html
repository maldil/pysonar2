  29 def smart_cond(pred, true_fn=None, false_fn=None, name=None):
  30   """Return either `true_fn()` if predicate `pred` is true else `false_fn()`.
  31 
  32   If `pred` is a bool or has a constant value, we return either `true_fn()`
  33   or `false_fn()`, otherwise we use `tf.cond` to dynamically route to both.
  34 
  35   Arguments:
  36     pred: A scalar determining whether to return the result of `true_fn` or
  37       `false_fn`.
  38     true_fn: The callable to be performed if pred is true.
  39     false_fn: The callable to be performed if pred is false.
  40     name: Optional name prefix when using `tf.cond`.
  41 
  42   Returns:
  43     Tensors returned by the call to either `true_fn` or `false_fn`.
  44 
  45   Raises:
  46     TypeError: If `true_fn` or `false_fn` is not callable.
  47   """
  48   if isinstance(pred, variables.Variable):
  49     return control_flow_ops.cond(
  50         pred, true_fn=true_fn, false_fn=false_fn, name=name)
  51   return smart_module.smart_cond(
  52       pred, true_fn=true_fn, false_fn=false_fn, name=name)

  55 def constant_value(pred):
  56   """Return the bool value for `pred`, or None if `pred` had a dynamic value.
  57 
  58   Arguments:
  59     pred: A scalar, either a Python bool or a TensorFlow boolean variable
  60       or tensor, or the Python integer 1 or 0.
  61 
  62   Returns:
  63     True or False if `pred` has a constant boolean value, None otherwise.
  64 
  65   Raises:
  66     TypeError: If `pred` is not a Variable, Tensor or bool, or Python
  67       integer 1 or 0.
  68   """
  69   # Allow integer booleans.
  70   if isinstance(pred, int):
  71     if pred == 1:
  72       pred = True
  73     elif pred == 0:
  74       pred = False
  75 
  76   if isinstance(pred, variables.Variable):
  77     return None
  78   return smart_module.smart_constant_value(pred)

  81 def is_tensor_or_tensor_list(v):
  82   v = nest.flatten(v)
  83   if v and isinstance(v[0], ops.Tensor):
  84     return True
  85   else:
  86     return False

  89 def get_reachable_from_inputs(inputs, targets=None):
  90   """Returns the set of tensors/ops reachable from `inputs`.
  91 
  92   Stops if all targets have been found (target is optional).
  93 
  94   Only valid in Symbolic mode, not Eager mode.
  95 
  96   Args:
  97     inputs: List of tensors.
  98     targets: List of tensors.
  99 
 100   Returns:
 101     A set of tensors reachable from the inputs (includes the inputs themselves).
 102   """
 103   reachable = set(inputs)
 104   if targets:
 105     targets = set(targets)
 106   queue = inputs[:]
 107 
 108   while queue:
 109     x = queue.pop()
 110     if isinstance(x, ops.Operation):
 111       outputs = x.outputs[:] or []
 112       outputs += x._control_outputs  # pylint: disable=protected-access
 113     elif isinstance(x, variables.Variable):
 114       outputs = [x.op]
 115     elif tensor_util.is_tensor(x):
 116       outputs = x.consumers()
 117     else:
 118       raise TypeError('Expected Operation, Variable, or Tensor, got ' + str(x))
 119 
 120     for y in outputs:
 121       if y not in reachable:
 122         reachable.add(y)
 123         queue.insert(0, y)
 124 
 125     if targets and targets.issubset(reachable):
 126       return reachable
 127   return reachable

 130 def shape_type_conversion(fn):
 131   """Decorator that handles tuple/TensorShape conversion.
 132 
 133   Used in `compute_output_shape` and `build`.
 134 
 135   Arguments:
 136     fn: function to wrap.
 137 
 138   Returns:
 139     Wrapped function.
 140   """
 141 
 142   def wrapper(instance, input_shape):
 143     if input_shape is not None:
 144       if isinstance(input_shape, list):
 145         input_shape = [
 146             tuple(tensor_shape.TensorShape(x).as_list()) for x in input_shape]
 147       else:
 148         input_shape = tuple(tensor_shape.TensorShape(input_shape).as_list())
 149     output_shape = fn(instance, input_shape)
 150     if output_shape is not None:
 151       if isinstance(output_shape, list):
 152         return [tensor_shape.TensorShape(x) for x in output_shape]
 153       return tensor_shape.TensorShape(output_shape)
 154 
 155   return wrapper
