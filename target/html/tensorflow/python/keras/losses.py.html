  32 @tf_export('keras.metrics.mean_squared_error',
  33            'keras.metrics.mse',
  34            'keras.metrics.MSE',
  35            'keras.losses.mean_squared_error',

  38 def mean_squared_error(y_true, y_pred):
  39   return K.mean(math_ops.square(y_pred - y_true), axis=-1)

  52 @tf_export('keras.metrics.mean_absolute_percentage_error',
  53            'keras.metrics.mape',
  54            'keras.metrics.MAPE',
  55            'keras.losses.mean_absolute_percentage_error',
  56            'keras.losses.mape',
  57            'keras.losses.MAPE')
  58 def mean_absolute_percentage_error(y_true, y_pred):
  59   diff = math_ops.abs(
  60       (y_true - y_pred) / K.clip(math_ops.abs(y_true), K.epsilon(), None))
  61   return 100. * K.mean(diff, axis=-1)

  64 @tf_export('keras.metrics.mean_squared_logarithmic_error',
  65            'keras.metrics.msle',
  66            'keras.metrics.MSLE',
  67            'keras.losses.mean_squared_logarithmic_error',
  68            'keras.losses.msle',
  69            'keras.losses.MSLE')
  70 def mean_squared_logarithmic_error(y_true, y_pred):
  71   first_log = math_ops.log(K.clip(y_pred, K.epsilon(), None) + 1.)
  72   second_log = math_ops.log(K.clip(y_true, K.epsilon(), None) + 1.)
  73   return K.mean(math_ops.square(first_log - second_log), axis=-1)

  87 @tf_export('keras.losses.categorical_hinge')
  88 def categorical_hinge(y_true, y_pred):
  89   pos = math_ops.reduce_sum(y_true * y_pred, axis=-1)
  90   neg = math_ops.reduce_max((1. - y_true) * y_pred, axis=-1)
  91   return math_ops.maximum(0., neg - pos + 1.)

  94 @tf_export('keras.losses.logcosh')
  95 def logcosh(y_true, y_pred):
  96   """Logarithm of the hyperbolic cosine of the prediction error.
  97 
  98   `log(cosh(x))` is approximately equal to `(x ** 2) / 2` for small `x` and
  99   to `abs(x) - log(2)` for large `x`. This means that 'logcosh' works mostly
 100   like the mean squared error, but will not be so strongly affected by the
 101   occasional wildly incorrect prediction.
 102 
 103   Arguments:
 104       y_true: tensor of true targets.
 105       y_pred: tensor of predicted targets.
 106 
 107   Returns:
 108       Tensor with one scalar loss entry per sample.
 109   """
 110 
 111   def _logcosh(x):
 112     return x + nn.softplus(-2. * x) - math_ops.log(2.)
 113 
 114   return K.mean(_logcosh(y_pred - y_true), axis=-1)

 135 @tf_export('keras.metrics.kullback_leibler_divergence',
 136            'keras.metrics.kld',
 137            'keras.metrics.KLD',
 138            'keras.losses.kullback_leibler_divergence',
 139            'keras.losses.kld',
 140            'keras.losses.KLD')
 141 def kullback_leibler_divergence(y_true, y_pred):
 142   y_true = K.clip(y_true, K.epsilon(), 1)
 143   y_pred = K.clip(y_pred, K.epsilon(), 1)
 144   return math_ops.reduce_sum(y_true * math_ops.log(y_true / y_pred), axis=-1)

 147 @tf_export('keras.metrics.poisson', 'keras.losses.poisson')
 148 def poisson(y_true, y_pred):
 149   return K.mean(y_pred - y_true * math_ops.log(y_pred + K.epsilon()), axis=-1)

 152 @tf_export('keras.metrics.cosine_proximity',
 153            'keras.metrics.cosine',
 154            'keras.losses.cosine_proximity',
 155            'keras.losses.cosine')
 156 def cosine_proximity(y_true, y_pred):
 157   y_true = nn.l2_normalize(y_true, axis=-1)
 158   y_pred = nn.l2_normalize(y_pred, axis=-1)
 159   return -math_ops.reduce_sum(y_true * y_pred, axis=-1)

 186 @tf_export('keras.losses.get')
 187 def get(identifier):
 188   if identifier is None:
 189     return None
 190   if isinstance(identifier, six.string_types):
 191     identifier = str(identifier)
 192     return deserialize(identifier)
 193   if isinstance(identifier, dict):
 194     return deserialize(identifier)
 195   elif callable(identifier):
 196     return identifier
 197   else:
 198     raise ValueError('Could not interpret '
 199                      'loss function identifier:', identifier)
