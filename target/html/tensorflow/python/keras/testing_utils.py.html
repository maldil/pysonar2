  29 def get_test_data(train_samples,
  30                   test_samples,
  31                   input_shape,
  32                   num_classes,
  33                   random_seed=None):
  34   """Generates test data to train a model on.
  35 
  36   Arguments:
  37     train_samples: Integer, how many training samples to generate.
  38     test_samples: Integer, how many test samples to generate.
  39     input_shape: Tuple of integers, shape of the inputs.
  40     num_classes: Integer, number of classes for the data and targets.
  41     random_seed: Integer, random seed used by numpy to generate data.
  42 
  43   Returns:
  44     A tuple of Numpy arrays: `(x_train, y_train), (x_test, y_test)`.
  45   """
  46   if random_seed is not None:
  47     np.random.seed(random_seed)
  48   num_sample = train_samples + test_samples
  49   templates = 2 * num_classes * np.random.random((num_classes,) + input_shape)
  50   y = np.random.randint(0, num_classes, size=(num_sample,))
  51   x = np.zeros((num_sample,) + input_shape, dtype=np.float32)
  52   for i in range(num_sample):
  53     x[i] = templates[y[i]] + np.random.normal(loc=0, scale=1., size=input_shape)
  54   return ((x[:train_samples], y[:train_samples]),
  55           (x[train_samples:], y[train_samples:]))

  58 def layer_test(layer_cls, kwargs=None, input_shape=None, input_dtype=None,
  59                input_data=None, expected_output=None,
  60                expected_output_dtype=None):
  61   """Test routine for a layer with a single input and single output.
  62 
  63   Arguments:
  64     layer_cls: Layer class object.
  65     kwargs: Optional dictionary of keyword arguments for instantiating the
  66       layer.
  67     input_shape: Input shape tuple.
  68     input_dtype: Data type of the input data.
  69     input_data: Numpy array of input data.
  70     expected_output: Shape tuple for the expected shape of the output.
  71     expected_output_dtype: Data type expected for the output.
  72 
  73   Returns:
  74     The output data (Numpy array) returned by the layer, for additional
  75     checks to be done by the calling code.
  76   """
  77   if input_data is None:
  78     assert input_shape
  79     if not input_dtype:
  80       input_dtype = 'float32'
  81     input_data_shape = list(input_shape)
  82     for i, e in enumerate(input_data_shape):
  83       if e is None:
  84         input_data_shape[i] = np.random.randint(1, 4)
  85     input_data = 10 * np.random.random(input_data_shape)
  86     if input_dtype[:5] == 'float':
  87       input_data -= 0.5
  88     input_data = input_data.astype(input_dtype)
  89   elif input_shape is None:
  90     input_shape = input_data.shape
  91   if input_dtype is None:
  92     input_dtype = input_data.dtype
  93   if expected_output_dtype is None:
  94     expected_output_dtype = input_dtype
  95 
  96   # instantiation
  97   kwargs = kwargs or {}
  98   layer = layer_cls(**kwargs)
  99 
 100   # test get_weights , set_weights at layer level
 101   weights = layer.get_weights()
 102   layer.set_weights(weights)
 103 
 104   # test and instantiation from weights
 105   if 'weights' in tf_inspect.getargspec(layer_cls.__init__):
 106     kwargs['weights'] = weights
 107     layer = layer_cls(**kwargs)
 108 
 109   # test in functional API
 110   x = keras.layers.Input(shape=input_shape[1:], dtype=input_dtype)
 111   y = layer(x)
 112   if keras.backend.dtype(y) != expected_output_dtype:
 113     raise AssertionError('When testing layer %s, for input %s, found output '
 114                          'dtype=%s but expected to find %s.\nFull kwargs: %s' %
 115                          (layer_cls.__name__,
 116                           x,
 117                           keras.backend.dtype(y),
 118                           expected_output_dtype,
 119                           kwargs))
 120   # check shape inference
 121   model = keras.models.Model(x, y)
 122   expected_output_shape = tuple(
 123       layer.compute_output_shape(
 124           tensor_shape.TensorShape(input_shape)).as_list())
 125   actual_output = model.predict(input_data)
 126   actual_output_shape = actual_output.shape
 127   for expected_dim, actual_dim in zip(expected_output_shape,
 128                                       actual_output_shape):
 129     if expected_dim is not None:
 130       if expected_dim != actual_dim:
 131         raise AssertionError(
 132             'When testing layer %s, for input %s, found output_shape='
 133             '%s but expected to find %s.\nFull kwargs: %s' %
 134             (layer_cls.__name__,
 135              x,
 136              actual_output_shape,
 137              expected_output_shape,
 138              kwargs))
 139   if expected_output is not None:
 140     np.testing.assert_allclose(actual_output, expected_output, rtol=1e-3)
 141 
 142   # test serialization, weight setting at model level
 143   model_config = model.get_config()
 144   recovered_model = keras.models.Model.from_config(model_config)
 145   if model.weights:
 146     weights = model.get_weights()
 147     recovered_model.set_weights(weights)
 148     output = recovered_model.predict(input_data)
 149     np.testing.assert_allclose(output, actual_output, rtol=1e-3)
 150 
 151   # test training mode (e.g. useful for dropout tests)
 152   model.compile(RMSPropOptimizer(0.01), 'mse')
 153   model.train_on_batch(input_data, actual_output)
 154 
 155   # test as first layer in Sequential API
 156   layer_config = layer.get_config()
 157   layer_config['batch_input_shape'] = input_shape
 158   layer = layer.__class__.from_config(layer_config)
 159 
 160   model = keras.models.Sequential()
 161   model.add(layer)
 162   actual_output = model.predict(input_data)
 163   actual_output_shape = actual_output.shape
 164   for expected_dim, actual_dim in zip(expected_output_shape,
 165                                       actual_output_shape):
 166     if expected_dim is not None:
 167       if expected_dim != actual_dim:
 168         raise AssertionError(
 169             'When testing layer %s, for input %s, found output_shape='
 170             '%s but expected to find %s.\nFull kwargs: %s' %
 171             (layer_cls.__name__,
 172              x,
 173              actual_output_shape,
 174              expected_output_shape,
 175              kwargs))
 176   if expected_output is not None:
 177     np.testing.assert_allclose(actual_output, expected_output, rtol=1e-3)
 178 
 179   # test serialization, weight setting at model level
 180   model_config = model.get_config()
 181   recovered_model = keras.models.Sequential.from_config(model_config)
 182   if model.weights:
 183     weights = model.get_weights()
 184     recovered_model.set_weights(weights)
 185     output = recovered_model.predict(input_data)
 186     np.testing.assert_allclose(output, actual_output, rtol=1e-3)
 187 
 188   # for further checks in the caller function
 189   return actual_output

 192 def get_small_sequential_mlp(num_hidden, num_classes, input_dim=None):
 193   model = keras.models.Sequential()
 194   if input_dim:
 195     model.add(keras.layers.Dense(num_hidden, activation='relu',
 196                                  input_dim=input_dim))
 197   else:
 198     model.add(keras.layers.Dense(num_hidden, activation='relu'))
 199   activation = 'sigmoid' if num_classes == 1 else 'softmax'
 200   model.add(keras.layers.Dense(num_classes, activation=activation))
 201   return model

 204 def get_small_functional_mlp(num_hidden, num_classes, input_dim):
 205   inputs = keras.Input(shape=(input_dim,))
 206   outputs = keras.layers.Dense(num_hidden, activation='relu')(inputs)
 207   activation = 'sigmoid' if num_classes == 1 else 'softmax'
 208   outputs = keras.layers.Dense(num_classes, activation=activation)(outputs)
 209   return keras.Model(inputs, outputs)
