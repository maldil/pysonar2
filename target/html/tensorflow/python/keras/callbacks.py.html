  58 def configure_callbacks(callbacks,
  59                         model,
  60                         do_validation=False,
  61                         val_inputs=None,
  62                         val_targets=None,
  63                         val_sample_weights=None,
  64                         batch_size=None,
  65                         epochs=None,
  66                         steps_per_epoch=None,
  67                         samples=None,
  68                         validation_steps=None,
  69                         verbose=1,
  70                         count_mode='steps'):
  71   """Configures callbacks for use in various training loops.
  72 
  73   Arguments:
  74       callbacks: List of Callbacks.
  75       model: Model being trained.
  76       do_validation: Whether or not validation loop will be run.
  77       val_inputs: Inputs to Model for validation loop. Can be any
  78         data format Keras accepts.
  79       val_targets: Targets for Model for validation loop. Can be any
  80         data format Keras accepts.
  81       val_sample_weights: Sample weights for Model for validation loop.
  82         Can be any data format Keras accepts.
  83       batch_size: Number of samples per batch.
  84       epochs: Number of epoch to train.
  85       steps_per_epoch: Number of batches to run per training epoch.
  86       samples: Number of training samples.
  87       validation_steps: Number of batches to run per validation epoch.
  88       verbose: int, 0 or 1. Keras logging verbosity to pass to ProgbarLogger.
  89       count_mode: One of 'steps' or 'samples'. Per-batch or per-sample count.
  90 
  91   Returns:
  92       Instance of CallbackList used to control all Callbacks.
  93   """
  94 
  95   # Add additional callbacks
  96   model.history = History()
  97   stateful_metric_names = None
  98   if hasattr(model, 'stateful_metric_names'):
  99     stateful_metric_names = model.stateful_metric_names
 100   callbacks = [BaseLogger(stateful_metrics=stateful_metric_names)
 101               ] + (callbacks or []) + [model.history]
 102   if verbose:
 103     callbacks.append(
 104         ProgbarLogger(count_mode, stateful_metrics=stateful_metric_names))
 105   callback_list = CallbackList(callbacks)
 106 
 107   # Set callback model
 108   callback_model = model._get_callback_model()  # pylint: disable=protected-access
 109   if do_validation and val_inputs and not context.executing_eagerly():
 110     # Need to create the test_function before start of the first epoch
 111     # because TensorBoard callback on_epoch_begin adds summary to the
 112     # list of fetches of the test_function
 113     callback_model._make_test_function()  # pylint: disable=protected-access
 114   callback_list.set_model(callback_model)
 115 
 116   # Set callback parameters
 117   callback_metrics = []
 118   # When we have deferred build scenario with iterator input, we will compile
 119   # when we standardize first batch of data.
 120   if model._is_compiled:  # pylint: disable=protected-access
 121     callback_metrics = copy.copy(model.metrics_names)
 122     if do_validation:
 123       callback_metrics += ['val_' + n for n in model.metrics_names]
 124   if validation_steps is None and isinstance(val_inputs, Sequence):
 125     validation_steps = len(val_inputs)
 126   callback_params = {
 127       'batch_size': batch_size,
 128       'epochs': epochs,
 129       'steps': steps_per_epoch,
 130       'samples': samples,
 131       'verbose': verbose,
 132       'do_validation': do_validation,
 133       'metrics': callback_metrics,
 134       'validation_steps': validation_steps
 135   }
 136   callback_list.set_params(callback_params)
 137 
 138   # Pass validation data to callbacks
 139   if not val_inputs:
 140     val_data = []
 141   elif _is_generator_like(val_inputs):
 142     val_data = val_inputs
 143   else:
 144     val_data = val_inputs + val_targets
 145     if val_sample_weights:
 146       val_data += val_sample_weights
 147     if model.uses_learning_phase and not isinstance(K.learning_phase(), int):
 148       val_data += [0.]
 149   for cbk in callbacks:
 150     cbk.validation_data = val_data
 151 
 152   callback_list.model.stop_training = False
 153   return callback_list
 154

 156 def _is_generator_like(data):
 157   """Checks if data is a generator, Sequence, or Iterator."""
 158   return (hasattr(data, 'next') or hasattr(data, '__next__') or isinstance(
 159       data, (Sequence, iterator_ops.Iterator, iterator_ops.EagerIterator)))

 162 class CallbackList(object):
 163   """Container abstracting a list of callbacks.
 164 
 165   Arguments:
 166       callbacks: List of `Callback` instances.
 167       queue_length: Queue length for keeping
 168           running statistics over callback execution time.
 169   """
 170 
 171   def __init__(self, callbacks=None, queue_length=10):
 172     callbacks = callbacks or []
 173     self.callbacks = [c for c in callbacks]
 174     self.queue_length = queue_length
 175     self.params = {}
 176     self.model = None
 177 
 178   def append(self, callback):
 179     self.callbacks.append(callback)
 180 
 181   def set_params(self, params):
 182     self.params = params
 183     for callback in self.callbacks:
 184       callback.set_params(params)
 185 
 186   def set_model(self, model):
 187     self.model = model
 188     for callback in self.callbacks:
 189       callback.set_model(model)
 190 
 191   def on_epoch_begin(self, epoch, logs=None):
 192     """Called at the start of an epoch.
 193 
 194     Arguments:
 195         epoch: integer, index of epoch.
 196         logs: dictionary of logs.
 197     """
 198     logs = logs or {}
 199     for callback in self.callbacks:
 200       callback.on_epoch_begin(epoch, logs)
 201     self._delta_t_batch = 0.
 202     self._delta_ts_batch_begin = deque([], maxlen=self.queue_length)
 203     self._delta_ts_batch_end = deque([], maxlen=self.queue_length)
 204 
 205   def on_epoch_end(self, epoch, logs=None):
 206     """Called at the end of an epoch.
 207 
 208     Arguments:
 209         epoch: integer, index of epoch.
 210         logs: dictionary of logs.
 211     """
 212     logs = logs or {}
 213     for callback in self.callbacks:
 214       callback.on_epoch_end(epoch, logs)
 215 
 216   def on_batch_begin(self, batch, logs=None):
 217     """Called right before processing a batch.
 218 
 219     Arguments:
 220         batch: integer, index of batch within the current epoch.
 221         logs: dictionary of logs.
 222     """
 223     logs = logs or {}
 224     t_before_callbacks = time.time()
 225     for callback in self.callbacks:
 226       callback.on_batch_begin(batch, logs)
 227     self._delta_ts_batch_begin.append(time.time() - t_before_callbacks)
 228     delta_t_median = np.median(self._delta_ts_batch_begin)
 229     if (self._delta_t_batch > 0. and
 230         delta_t_median > 0.95 * self._delta_t_batch and delta_t_median > 0.1):
 231       logging.warning('Method on_batch_begin() is slow compared '
 232                       'to the batch update (%f). Check your callbacks.',
 233                       delta_t_median)
 234     self._t_enter_batch = time.time()
 235 
 236   def on_batch_end(self, batch, logs=None):
 237     """Called at the end of a batch.
 238 
 239     Arguments:
 240         batch: integer, index of batch within the current epoch.
 241         logs: dictionary of logs.
 242     """
 243     logs = logs or {}
 244     if not hasattr(self, '_t_enter_batch'):
 245       self._t_enter_batch = time.time()
 246     self._delta_t_batch = time.time() - self._t_enter_batch
 247     t_before_callbacks = time.time()
 248     for callback in self.callbacks:
 249       callback.on_batch_end(batch, logs)
 250     self._delta_ts_batch_end.append(time.time() - t_before_callbacks)
 251     delta_t_median = np.median(self._delta_ts_batch_end)
 252     if (self._delta_t_batch > 0. and
 253         (delta_t_median > 0.95 * self._delta_t_batch and delta_t_median > 0.1)):
 254       logging.warning('Method on_batch_end() is slow compared '
 255                       'to the batch update (%f). Check your callbacks.',
 256                       delta_t_median)
 257 
 258   def on_train_begin(self, logs=None):
 259     """Called at the beginning of training.
 260 
 261     Arguments:
 262         logs: dictionary of logs.
 263     """
 264     logs = logs or {}
 265     for callback in self.callbacks:
 266       callback.on_train_begin(logs)
 267 
 268   def on_train_end(self, logs=None):
 269     """Called at the end of training.
 270 
 271     Arguments:
 272         logs: dictionary of logs.
 273     """
 274     logs = logs or {}
 275     for callback in self.callbacks:
 276       callback.on_train_end(logs)
 277 
 278   def __iter__(self):
 279     return iter(self.callbacks)

 282 @tf_export('keras.callbacks.Callback')
 283 class Callback(object):
 284   """Abstract base class used to build new callbacks.
 285 
 286   Attributes:
 287       params: dict. Training parameters
 288           (eg. verbosity, batch size, number of epochs...).
 289       model: instance of `keras.models.Model`.
 290           Reference of the model being trained.
 291 
 292   The `logs` dictionary that callback methods
 293   take as argument will contain keys for quantities relevant to
 294   the current batch or epoch.
 295 
 296   Currently, the `.fit()` method of the `Sequential` model class
 297   will include the following quantities in the `logs` that
 298   it passes to its callbacks:
 299 
 300       on_epoch_end: logs include `acc` and `loss`, and
 301           optionally include `val_loss`
 302           (if validation is enabled in `fit`), and `val_acc`
 303           (if validation and accuracy monitoring are enabled).
 304       on_batch_begin: logs include `size`,
 305           the number of samples in the current batch.
 306       on_batch_end: logs include `loss`, and optionally `acc`
 307           (if accuracy monitoring is enabled).
 308   """
 309 
 310   def __init__(self):
 311     self.validation_data = None
 312     self.model = None
 313 
 314   def set_params(self, params):
 315     self.params = params
 316 
 317   def set_model(self, model):
 318     self.model = model
 319 
 320   def on_epoch_begin(self, epoch, logs=None):
 321     pass
 322 
 323   def on_epoch_end(self, epoch, logs=None):
 324     pass
 325 
 326   def on_batch_begin(self, batch, logs=None):
 327     pass
 328 
 329   def on_batch_end(self, batch, logs=None):
 330     pass
 331 
 332   def on_train_begin(self, logs=None):
 333     pass
 334 
 335   def on_train_end(self, logs=None):
 336     pass

 339 @tf_export('keras.callbacks.BaseLogger')
 340 class BaseLogger(Callback):
 341   """Callback that accumulates epoch averages of metrics.
 342 
 343   This callback is automatically applied to every Keras model.
 344 
 345   Arguments:
 346       stateful_metrics: Iterable of string names of metrics that
 347           should *not* be averaged over an epoch.
 348           Metrics in this list will be logged as-is in `on_epoch_end`.
 349           All others will be averaged in `on_epoch_end`.
 350   """
 351 
 352   def __init__(self, stateful_metrics=None):
 353     super(BaseLogger, self).__init__()
 354     self.stateful_metrics = set(stateful_metrics or [])
 355 
 356   def on_epoch_begin(self, epoch, logs=None):
 357     self.seen = 0
 358     self.totals = {}
 359 
 360   def on_batch_end(self, batch, logs=None):
 361     logs = logs or {}
 362     batch_size = logs.get('size', 0)
 363     # In case of distribution strategy we can potentially run multiple steps
 364     # at the same time, we should account for that in the `seen` calculation.
 365     num_steps = logs.get('num_steps', 1)
 366     self.seen += batch_size * num_steps
 367 
 368     for k, v in logs.items():
 369       if k in self.stateful_metrics:
 370         self.totals[k] = v
 371       else:
 372         if k in self.totals:
 373           self.totals[k] += v * batch_size
 374         else:
 375           self.totals[k] = v * batch_size
 376 
 377   def on_epoch_end(self, epoch, logs=None):
 378     if logs is not None:
 379       for k in self.params['metrics']:
 380         if k in self.totals:
 381           # Make value available to next callbacks.
 382           if k in self.stateful_metrics:
 383             logs[k] = self.totals[k]
 384           else:
 385             logs[k] = self.totals[k] / self.seen

 388 @tf_export('keras.callbacks.TerminateOnNaN')
 389 class TerminateOnNaN(Callback):
 390   """Callback that terminates training when a NaN loss is encountered.
 391   """
 392 
 393   def on_batch_end(self, batch, logs=None):
 394     logs = logs or {}
 395     loss = logs.get('loss')

 399         self.model.stop_training = True

 402 @tf_export('keras.callbacks.ProgbarLogger')
 403 class ProgbarLogger(Callback):
 404   """Callback that prints metrics to stdout.
 405 
 406   Arguments:
 407       count_mode: One of "steps" or "samples".
 408           Whether the progress bar should
 409           count samples seen or steps (batches) seen.
 410       stateful_metrics: Iterable of string names of metrics that
 411           should *not* be averaged over an epoch.
 412           Metrics in this list will be logged as-is.
 413           All others will be averaged over time (e.g. loss, etc).
 414 
 415   Raises:
 416       ValueError: In case of invalid `count_mode`.
 417   """
 418 
 419   def __init__(self, count_mode='samples', stateful_metrics=None):
 420     super(ProgbarLogger, self).__init__()
 421     if count_mode == 'samples':
 422       self.use_steps = False
 423     elif count_mode == 'steps':
 424       self.use_steps = True
 425     else:
 426       raise ValueError('Unknown `count_mode`: ' + str(count_mode))
 427     self.stateful_metrics = set(stateful_metrics or [])
 428 
 429   def on_train_begin(self, logs=None):
 430     self.verbose = self.params['verbose']
 431     self.epochs = self.params['epochs']
 432 
 433   def on_epoch_begin(self, epoch, logs=None):
 434     if self.verbose:
 435       print('Epoch %d/%d' % (epoch + 1, self.epochs))
 436       if self.use_steps:
 437         target = self.params['steps']
 438       else:
 439         target = self.params['samples']
 440       self.target = target
 441       self.progbar = Progbar(
 442           target=self.target,
 443           verbose=self.verbose,
 444           stateful_metrics=self.stateful_metrics)
 445     self.seen = 0
 446 
 447   def on_batch_begin(self, batch, logs=None):
 448     if self.seen < self.target:
 449       self.log_values = []
 450 
 451   def on_batch_end(self, batch, logs=None):
 452     logs = logs or {}
 453     batch_size = logs.get('size', 0)
 454     # In case of distribution strategy we can potentially run multiple steps
 455     # at the same time, we should account for that in the `seen` calculation.
 456     num_steps = logs.get('num_steps', 1)
 457     if self.use_steps:
 458       self.seen += num_steps
 459     else:
 460       self.seen += batch_size * num_steps
 461 
 462     for k in self.params['metrics']:
 463       if k in logs:
 464         self.log_values.append((k, logs[k]))
 465 
 466     # Skip progbar update for the last batch;
 467     # will be handled by on_epoch_end.
 468     if self.verbose and self.seen < self.target:
 469       self.progbar.update(self.seen, self.log_values)
 470 
 471   def on_epoch_end(self, epoch, logs=None):
 472     logs = logs or {}
 473     for k in self.params['metrics']:
 474       if k in logs:
 475         self.log_values.append((k, logs[k]))
 476     if self.verbose:
 477       self.progbar.update(self.seen, self.log_values)

 500 @tf_export('keras.callbacks.ModelCheckpoint')
 501 class ModelCheckpoint(Callback):
 502   """Save the model after every epoch.
 503 
 504   `filepath` can contain named formatting options,
 505   which will be filled the value of `epoch` and
 506   keys in `logs` (passed in `on_epoch_end`).
 507 
 508   For example: if `filepath` is `weights.{epoch:02d}-{val_loss:.2f}.hdf5`,
 509   then the model checkpoints will be saved with the epoch number and
 510   the validation loss in the filename.
 511 
 512   Arguments:
 513       filepath: string, path to save the model file.
 514       monitor: quantity to monitor.
 515       verbose: verbosity mode, 0 or 1.
 516       save_best_only: if `save_best_only=True`,
 517           the latest best model according to
 518           the quantity monitored will not be overwritten.
 519       mode: one of {auto, min, max}.
 520           If `save_best_only=True`, the decision
 521           to overwrite the current save file is made
 522           based on either the maximization or the
 523           minimization of the monitored quantity. For `val_acc`,
 524           this should be `max`, for `val_loss` this should
 525           be `min`, etc. In `auto` mode, the direction is
 526           automatically inferred from the name of the monitored quantity.
 527       save_weights_only: if True, then only the model's weights will be
 528           saved (`model.save_weights(filepath)`), else the full model
 529           is saved (`model.save(filepath)`).
 530       period: Interval (number of epochs) between checkpoints.
 531   """
 532 
 533   def __init__(self,
 534                filepath,
 535                monitor='val_loss',
 536                verbose=0,
 537                save_best_only=False,
 538                save_weights_only=False,
 539                mode='auto',
 540                period=1):
 541     super(ModelCheckpoint, self).__init__()
 542     self.monitor = monitor
 543     self.verbose = verbose
 544     self.filepath = filepath
 545     self.save_best_only = save_best_only
 546     self.save_weights_only = save_weights_only
 547     self.period = period
 548     self.epochs_since_last_save = 0
 549 
 550     if mode not in ['auto', 'min', 'max']:
 551       logging.warning('ModelCheckpoint mode %s is unknown, '
 552                       'fallback to auto mode.', mode)
 553       mode = 'auto'
 554 
 555     if mode == 'min':
 556       self.monitor_op = np.less
 557       self.best = np.Inf
 558     elif mode == 'max':
 559       self.monitor_op = np.greater
 560       self.best = -np.Inf
 561     else:
 562       if 'acc' in self.monitor or self.monitor.startswith('fmeasure'):
 563         self.monitor_op = np.greater
 564         self.best = -np.Inf
 565       else:
 566         self.monitor_op = np.less
 567         self.best = np.Inf
 568 
 569   def on_epoch_end(self, epoch, logs=None):
 570     logs = logs or {}
 571     self.epochs_since_last_save += 1
 572     if self.epochs_since_last_save >= self.period:
 573       self.epochs_since_last_save = 0
 574       filepath = self.filepath.format(epoch=epoch + 1, **logs)
 575       if self.save_best_only:
 576         current = logs.get(self.monitor)
 577         if current is None:
 578           logging.warning('Can save best model only with %s available, '
 579                           'skipping.', self.monitor)
 580         else:
 581           if self.monitor_op(current, self.best):
 582             if self.verbose > 0:
 583               print('\nEpoch %05d: %s improved from %0.5f to %0.5f,'
 584                     ' saving model to %s' % (epoch + 1, self.monitor, self.best,
 585                                              current, filepath))
 586             self.best = current
 587             if self.save_weights_only:
 588               self.model.save_weights(filepath, overwrite=True)
 589             else:
 590               self.model.save(filepath, overwrite=True)
 591           else:
 592             if self.verbose > 0:
 593               print('\nEpoch %05d: %s did not improve from %0.5f' %
 594                     (epoch + 1, self.monitor, self.best))
 595       else:
 596         if self.verbose > 0:
 597           print('\nEpoch %05d: saving model to %s' % (epoch + 1, filepath))
 598         if self.save_weights_only:
 599           self.model.save_weights(filepath, overwrite=True)
 600         else:
 601           self.model.save(filepath, overwrite=True)

 604 @tf_export('keras.callbacks.EarlyStopping')
 605 class EarlyStopping(Callback):
 606   """Stop training when a monitored quantity has stopped improving.
 607 
 608   Arguments:
 609       monitor: quantity to be monitored.
 610       min_delta: minimum change in the monitored quantity
 611           to qualify as an improvement, i.e. an absolute
 612           change of less than min_delta, will count as no
 613           improvement.
 614       patience: number of epochs with no improvement
 615           after which training will be stopped.
 616       verbose: verbosity mode.
 617       mode: one of {auto, min, max}. In `min` mode,
 618           training will stop when the quantity

 632                patience=0,
 633                verbose=0,

 639     self.patience = patience

 653     elif mode == 'max':

 657         self.monitor_op = np.greater
 658       else:

 672     else:
 673       self.best = np.Inf if self.monitor_op == np.less else -np.Inf

 689         self.model.stop_training = True

 891   def _make_histogram_ops(self, model):
 892     """Defines histogram ops when histogram_freq > 0."""
 893     # only make histogram summary op if it hasn't already been made
 894     if self.histogram_freq and self.merged is None:
 895       for layer in self.model.layers:

 899           if self.write_images:

 901             shape = K.int_shape(w_img)

 906               w_img = array_ops.reshape(w_img, [1, shape[0], shape[1], 1])

 908               if K.image_data_format() == 'channels_last':

 917             else:
 918               # not possible to handle 3D convnets etc.
 919               continue

1095     """Checks if summary ops should run next epoch, logs scalar summaries."""
1096 
1097     # don't output batch_size and
1098     # batch number as Tensorboard summaries
1099     logs = {('epoch_' + k): v
1100             for k, v in logs.items()
1101             if k not in ['batch', 'size', 'num_steps']}
1102     self._write_custom_summaries(epoch, logs)
1103 
1104     # pop the histogram summary op after each epoch
1105     if self.histogram_freq:
1106       if self.merged in self.model.test_function.fetches:
1107         self.model.test_function.fetches.remove(self.merged)
1108       if self.merged in self.model.test_function.fetch_callbacks:
1109         self.model.test_function.fetch_callbacks.pop(self.merged)
1110 
1111     if self.embeddings_data is None and self.embeddings_freq:
1112       raise ValueError('To visualize embeddings, embeddings_data must '
1113                        'be provided.')
1114 
1115     if self.embeddings_freq and self.embeddings_data is not None:
1116       if epoch % self.embeddings_freq == 0:
1117         # We need a second forward-pass here because we're passing
1118         # the `embeddings_data` explicitly. This design allows to pass
1119         # arbitrary data as `embeddings_data` and results from the fact
1120         # that we need to know the size of the `tf.Variable`s which
1121         # hold the embeddings in `set_model`. At this point, however,
1122         # the `validation_data` is not yet set.
1123 
1124         embeddings_data = self.embeddings_data
1125         n_samples = embeddings_data[0].shape[0]
1126         i = 0
1127         while i < n_samples:
1128           step = min(self.batch_size, n_samples - i)
1129           batch = slice(i, i + step)
1130 
1131           if isinstance(self.model.input, list):
1132             feed_dict = {
1133                 model_input: embeddings_data[idx][batch]
1134                 for idx, model_input in enumerate(self.model.input)
1135             }
1136           else:
1137             feed_dict = {self.model.input: embeddings_data[0][batch]}
1138 
1139           feed_dict.update({self.batch_id: i, self.step: step})
1140 
1141           if self.model.uses_learning_phase:
1142             feed_dict[K.learning_phase()] = False
1143 
1144           self.sess.run(self.assign_embeddings, feed_dict=feed_dict)
1145           self.saver.save(self.sess,
1146                           os.path.join(self.log_dir, 'keras_embedding.ckpt'),
1147                           epoch)
1148

1161   quantity and if no improvement is seen for a 'patience' number
1162   of epochs, the learning rate is reduced.

1164   Example:
1165 
1166   ```python

1173       monitor: quantity to be monitored.
1174       factor: factor by which the learning rate will
1175           be reduced. new_lr = lr * factor
1176       patience: number of epochs with no improvement
1177           after which learning rate will be reduced.
1178       verbose: int. 0: quiet, 1: update messages.
1179       mode: one of {auto, min, max}. In `min` mode,
1180           lr will be reduced when the quantity
1181           monitored has stopped decreasing; in `max`
1182           mode it will be reduced when the quantity
1183           monitored has stopped increasing; in `auto`
1184           mode, the direction is automatically inferred
1185           from the name of the monitored quantity.
1186       min_delta: threshold for measuring the new optimum,
1187           to only focus on significant changes.
1188       cooldown: number of epochs to wait before resuming
1189           normal operation after lr has been reduced.
1190       min_lr: lower bound on the learning rate.
1191   """
1192 
1193   def __init__(self,
1194                monitor='val_loss',
1195                factor=0.1,
1196                patience=10,
1197                verbose=0,
1198                mode='auto',
1199                min_delta=1e-4,
1200                cooldown=0,
1201                min_lr=0,
1202                **kwargs):
1203     super(ReduceLROnPlateau, self).__init__()
1204 
1205     self.monitor = monitor
1206     if factor >= 1.0:
1207       raise ValueError('ReduceLROnPlateau ' 'does not support a factor >= 1.0.')
1208     if 'epsilon' in kwargs:
1209       min_delta = kwargs.pop('epsilon')
1210       logging.warning('`epsilon` argument is deprecated and '
1211                       'will be removed, use `min_delta` instead.')
1212     self.factor = factor
1213     self.min_lr = min_lr
1214     self.min_delta = min_delta
1215     self.patience = patience
1216     self.verbose = verbose
1217     self.cooldown = cooldown
1218     self.cooldown_counter = 0  # Cooldown counter.
1219     self.wait = 0
1220     self.best = 0
1221     self.mode = mode
1222     self.monitor_op = None
1223     self._reset()
1224 
1225   def _reset(self):
1226     """Resets wait counter and cooldown counter.
1227     """
1228     if self.mode not in ['auto', 'min', 'max']:
1229       logging.warning('Learning Rate Plateau Reducing mode %s is unknown, '
1230                       'fallback to auto mode.', self.mode)
1231       self.mode = 'auto'
1232     if (self.mode == 'min' or
1233         (self.mode == 'auto' and 'acc' not in self.monitor)):
1234       self.monitor_op = lambda a, b: np.less(a, b - self.min_delta)
1235       self.best = np.Inf
1236     else:
1237       self.monitor_op = lambda a, b: np.greater(a, b + self.min_delta)
1238       self.best = -np.Inf
1239     self.cooldown_counter = 0
1240     self.wait = 0
1241 
1242   def on_train_begin(self, logs=None):
1243     self._reset()
1244 
1245   def on_epoch_end(self, epoch, logs=None):
1246     logs = logs or {}
1247     logs['lr'] = K.get_value(self.model.optimizer.lr)
1248     current = logs.get(self.monitor)
1249     if current is None:
1250       logging.warning('Reduce LR on plateau conditioned on metric `%s` '
1251                       'which is not available. Available metrics are: %s',
1252                       self.monitor, ','.join(list(logs.keys())))
1253 
1254     else:
1255       if self.in_cooldown():
1256         self.cooldown_counter -= 1
1257         self.wait = 0
1258 
1259       if self.monitor_op(current, self.best):
1260         self.best = current
1261         self.wait = 0
1262       elif not self.in_cooldown():
1263         self.wait += 1
1264         if self.wait >= self.patience:
1265           old_lr = float(K.get_value(self.model.optimizer.lr))
1266           if old_lr > self.min_lr:
1267             new_lr = old_lr * self.factor
1268             new_lr = max(new_lr, self.min_lr)
1269             K.set_value(self.model.optimizer.lr, new_lr)
1270             if self.verbose > 0:
1271               print('\nEpoch %05d: ReduceLROnPlateau reducing learning '
1272                     'rate to %s.' % (epoch + 1, new_lr))
1273             self.cooldown_counter = self.cooldown
1274             self.wait = 0
1275 
1276   def in_cooldown(self):
1277     return self.cooldown_counter > 0

1281 class CSVLogger(Callback):
1282   """Callback that streams epoch results to a csv file.
1283 
1284   Supports all values that can be represented as a string,
1285   including 1D iterables such as np.ndarray.
1286 
1287   Example:
1288 
1289   ```python
1290   csv_logger = CSVLogger('training.log')
1291   model.fit(X_train, Y_train, callbacks=[csv_logger])
1292   ```
1293 
1294   Arguments:
1295       filename: filename of the csv file, e.g. 'run/log.csv'.
1296       separator: string used to separate elements in the csv file.
1297       append: True: append if file exists (useful for continuing
1298           training). False: overwrite existing file,
1299   """
1300 
1301   def __init__(self, filename, separator=',', append=False):
1302     self.sep = separator
1303     self.filename = filename
1304     self.append = append
1305     self.writer = None
1306     self.keys = None
1307     self.append_header = True
1308     self.file_flags = 'b' if six.PY2 and os.name == 'nt' else ''
1309     super(CSVLogger, self).__init__()
1310 
1311   def on_train_begin(self, logs=None):
1312     if self.append:
1313       if os.path.exists(self.filename):
1314         with open(self.filename, 'r' + self.file_flags) as f:
1315           self.append_header = not bool(len(f.readline()))
1316       self.csv_file = open(self.filename, 'a' + self.file_flags)
1317     else:
1318       self.csv_file = open(self.filename, 'w' + self.file_flags)
1319 
1320   def on_epoch_end(self, epoch, logs=None):
1321     logs = logs or {}
1322 
1323     def handle_value(k):
1324       is_zero_dim_ndarray = isinstance(k, np.ndarray) and k.ndim == 0
1325       if isinstance(k, six.string_types):
1326         return k
1327       elif isinstance(k, Iterable) and not is_zero_dim_ndarray:
1328         return '"[%s]"' % (', '.join(map(str, k)))
1329       else:
1330         return k
1331 
1332     if self.keys is None:
1333       self.keys = sorted(logs.keys())
1334 
1335     if self.model.stop_training:
1336       # We set NA so that csv parsers do not fail for this last epoch.
1337       logs = dict([(k, logs[k]) if k in logs else (k, 'NA') for k in self.keys])
1338 
1339     if not self.writer:
1340 
1341       class CustomDialect(csv.excel):
1342         delimiter = self.sep
1343 
1344       self.writer = csv.DictWriter(
1345           self.csv_file,
1346           fieldnames=['epoch'] + self.keys,
1347           dialect=CustomDialect)
1348       if self.append_header:
1349         self.writer.writeheader()
1350 
1351     row_dict = OrderedDict({'epoch': epoch})
1352     row_dict.update((key, handle_value(logs[key])) for key in self.keys)
1353     self.writer.writerow(row_dict)
1354     self.csv_file.flush()
1355 
1356   def on_train_end(self, logs=None):
1357     self.csv_file.close()
1358     self.writer = None

1361 @tf_export('keras.callbacks.LambdaCallback')
1362 class LambdaCallback(Callback):
1363   r"""Callback for creating simple, custom callbacks on-the-fly.
1364 
1365   This callback is constructed with anonymous functions that will be called
1366   at the appropriate time. Note that the callbacks expects positional
1367   arguments, as:
1368 
1369    - `on_epoch_begin` and `on_epoch_end` expect two positional arguments:
1370       `epoch`, `logs`
1371    - `on_batch_begin` and `on_batch_end` expect two positional arguments:
1372       `batch`, `logs`
1373    - `on_train_begin` and `on_train_end` expect one positional argument:
1374       `logs`
1375 
1376   Arguments:
1377       on_epoch_begin: called at the beginning of every epoch.
1378       on_epoch_end: called at the end of every epoch.
1379       on_batch_begin: called at the beginning of every batch.
1380       on_batch_end: called at the end of every batch.
1381       on_train_begin: called at the beginning of model training.
1382       on_train_end: called at the end of model training.
1383 
1384   Example:
1385 
1386   ```python
1387   # Print the batch number at the beginning of every batch.
1388   batch_print_callback = LambdaCallback(
1389       on_batch_begin=lambda batch,logs: print(batch))
1390 
1391   # Stream the epoch loss to a file in JSON format. The file content
1392   # is not well-formed JSON but rather has a JSON object per line.
1393   import json
1394   json_log = open('loss_log.json', mode='wt', buffering=1)
1395   json_logging_callback = LambdaCallback(
1396       on_epoch_end=lambda epoch, logs: json_log.write(
1397           json.dumps({'epoch': epoch, 'loss': logs['loss']}) + '\n'),
1398       on_train_end=lambda logs: json_log.close()
1399   )
1400 
1401   # Terminate some processes after having finished model training.
1402   processes = ...
1403   cleanup_callback = LambdaCallback(
1404       on_train_end=lambda logs: [
1405           p.terminate() for p in processes if p.is_alive()])
1406 
1407   model.fit(...,
1408             callbacks=[batch_print_callback,
1409                        json_logging_callback,
1410                        cleanup_callback])
1411   ```
1412   """
1413 
1414   def __init__(self,
1415                on_epoch_begin=None,
1416                on_epoch_end=None,
1417                on_batch_begin=None,
1418                on_batch_end=None,
1419                on_train_begin=None,
1420                on_train_end=None,
1421                **kwargs):
1422     super(LambdaCallback, self).__init__()
1423     self.__dict__.update(kwargs)
1424     if on_epoch_begin is not None:
1425       self.on_epoch_begin = on_epoch_begin
1426     else:
1427       self.on_epoch_begin = lambda epoch, logs: None
1428     if on_epoch_end is not None:
1429       self.on_epoch_end = on_epoch_end
1430     else:
1431       self.on_epoch_end = lambda epoch, logs: None
1432     if on_batch_begin is not None:
1433       self.on_batch_begin = on_batch_begin
1434     else:
1435       self.on_batch_begin = lambda batch, logs: None
1436     if on_batch_end is not None:
1437       self.on_batch_end = on_batch_end
1438     else:
1439       self.on_batch_end = lambda batch, logs: None
1440     if on_train_begin is not None:
1441       self.on_train_begin = on_train_begin
1442     else:
1443       self.on_train_begin = lambda logs: None
1444     if on_train_end is not None:
1445       self.on_train_end = on_train_end
1446     else:
1447       self.on_train_end = lambda logs: None
