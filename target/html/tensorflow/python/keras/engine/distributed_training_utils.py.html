  31 def set_weights(distribution_strategy, dist_model, weights):
  32   """Sets the weights of the replicated models.
  33 
  34   The weights of the replicated models are set to the weights of the original
  35   model. The weights of the replicated model are Mirrored variables and hence
  36   we need to use the `update` call within a DistributionStrategy scope.
  37 
  38   Args:
  39     distribution_strategy: DistributionStrategy used to distribute training
  40         and validation.
  41     dist_model: The replicated models on the different devices.
  42     weights: The weights of the original model.
  43   """
  44   assign_ops = []
  45   for layer in dist_model.layers:
  46     num_param = len(layer.weights)
  47     layer_weights = weights[:num_param]
  48     for sw, w in zip(layer.weights, layer_weights):
  49       assign_ops.append(distribution_strategy.unwrap(sw.assign(w)))
  50 
  51     weights = weights[num_param:]
  52   K.get_session().run(assign_ops)

  55 def unwrap_values(distribution_strategy, grouped_inputs, grouped_outputs,
  56                   grouped_updates, grouped_session_args,
  57                   with_loss_tensor=False):
  58   """Unwrap and return the list of values contained in the PerDevice parameters.
  59 
  60   This function calls `flatten_perdevice_values` to parse each of the input
  61   parameters into a list of values on the different devices. If we set
  62   `with_loss_tensor` to be True, we also call `reduce` on the list of losses on
  63   the different devices to give us one loss tensor.
  64 
  65   Args:
  66     distribution_strategy: DistributionStrategy used to distribute training and
  67         validation.
  68     grouped_inputs: PerDevice inputs returned from the train or test function
  69         that we ran on each device.
  70     grouped_outputs: PerDevice outputs returned from the train or test function
  71         that we ran on each device.
  72     grouped_updates: PerDevice updates returned from the train or test function
  73         that we ran on each device.
  74     grouped_session_args: PerDevice session args returned from the train or
  75         test function that we ran on each device.
  76     with_loss_tensor: Boolean that indicates if we need to add the reduced loss
  77         tensor as one of the outputs.
  78 
  79   Returns:
  80     Values of each of the PerDevice parameters.
  81 
  82   """
  83   # Unwrap per device values returned from each model's train function.
  84   # This will be used to construct the main train function.
  85   all_inputs = flatten_perdevice_values(distribution_strategy,
  86                                         grouped_inputs)
  87   if with_loss_tensor:
  88     # reduce loss tensor before adding it to the list of fetches
  89     loss = distribution_strategy.unwrap(
  90         distribution_strategy.reduce(distribute_lib.get_loss_reduction(),
  91                                      grouped_outputs[0],
  92                                      destinations='/device:CPU:0'))[0]
  93 
  94     all_outputs = flatten_perdevice_values(distribution_strategy,
  95                                            grouped_outputs[1:])
  96     all_outputs = [loss] + all_outputs
  97   else:
  98     all_outputs = flatten_perdevice_values(distribution_strategy,
  99                                            grouped_outputs)
 100 
 101   all_updates = flatten_perdevice_values(distribution_strategy,
 102                                          grouped_updates)
 103 
 104   all_session_args = {}
 105   grouped_feed_dict = grouped_session_args.get('feed_dict')
 106   if grouped_feed_dict:
 107     all_session_args['feed_dict'] = flatten_perdevice_values(
 108         distribution_strategy, grouped_feed_dict)
 109 
 110   grouped_fetches = grouped_session_args.get('fetches')
 111   if grouped_fetches:
 112     all_session_args['fetches'] = flatten_perdevice_values(
 113         distribution_strategy, grouped_fetches)
 114 
 115   return all_inputs, all_outputs, all_updates, all_session_args

 118 def flatten_perdevice_values(distribution_strategy, perdevice_values):
 119   """Unwraps and flattens a nest of PerDevice parameters.
 120 
 121   PerDevice values have one value associated with each device. Each entry in
 122   the PerDevice dict has a device `key` and the corresponding value on the
 123   device as the `value`. In this function we take a PerDevice value or a list of
 124   PerDevice values and return all the values in the PerDevice dict.
 125 
 126   Args:
 127     distribution_strategy: DistributionStrategy used to distribute training and
 128         validation.
 129     perdevice_values: List of PerDevice object or a single PerDevice object.
 130 
 131   Returns:
 132     List of values of all the PerDevice objects.
 133 
 134   """
 135   # This function takes a PerDevice object or a list of PerDevice objects and
 136   # returns all the values associated with it.
 137   return [e for flattened in nest.flatten(perdevice_values)
 138           for e in distribution_strategy.unwrap(flattened)]

 141 def validate_callbacks(input_callbacks):
 142   """Validate whether given callbacks are supported by DistributionStrategy.
 143 
 144   Args:
 145     input_callbacks: List of callbacks passed by the user to fit.
 146 
 147   Raises:
 148     ValueError: If `LearningRateScheduler` or `ReduceLROnPlateau` is one of the
 149         callbacks passed.
 150     ValueError: If `histogram_freq` or `write_grads` is one of the parameters
 151         passed as part of the TensorBoard callback.
 152   """
 153   if input_callbacks:
 154     for callback in input_callbacks:
 155       if callback not in [callbacks.TensorBoard, callbacks.ReduceLROnPlateau,
 156                           callbacks.LearningRateScheduler, callbacks.CSVLogger,
 157                           callbacks.EarlyStopping, callbacks.ModelCheckpoint,
 158                           callbacks.TerminateOnNaN, callbacks.ProgbarLogger,
 159                           callbacks.History, callbacks.RemoteMonitor]:
 160         logging.warning('Your input callback is not one of the predefined '
 161                         'Callbacks that supports DistributionStrategy. You '
 162                         'might encounter an error if you access one of the '
 163                         'model\'s attributes as part of the callback since '
 164                         'these attributes are not set. You can access each of '
 165                         'the individual distributed models using the '
 166                         '`_grouped_model` attribute of your original model.')
 167       if isinstance(callback, callbacks.LearningRateScheduler):
 168         raise ValueError('LearningRateScheduler callback is not supported with '
 169                          'DistributionStrategy.')
 170       if isinstance(callback, callbacks.ReduceLROnPlateau):
 171         raise ValueError('ReduceLROnPlateau callback is not supported with '
 172                          'DistributionStrategy.')
 173 
 174       # If users want to use the TensorBoard callback they cannot use certain
 175       # features of the callback that involve accessing model attributes and
 176       # running ops.
 177       if isinstance(callback, callbacks.TensorBoard):
 178         if callback.__getattribute__('histogram_freq'):
 179           raise ValueError('histogram_freq in the TensorBoard callback is not '
 180                            'supported when using DistributionStrategy.')
 181         if callback.__getattribute__('write_grads'):
 182           raise ValueError('write_grads in the TensorBoard callback is not '
 183                            'supported when using DistributionStrategy.')

 186 def validate_distributed_dataset_inputs(distribution_strategy, x, y):
 187   """Validate all the components of a DistributedValue Dataset input.
 188 
 189   Args:
 190     distribution_strategy: The current DistributionStrategy used to call
 191         `fit`/`evaluate`.
 192     x: Input Dataset DistributedValue object. For example, when we use
 193         `MirroredStrategy` this is a PerDevice object with a tensor for each
 194         device set in the dict. x can also be a tuple or dict. The keys of the
 195         dict should match the names of the input layers of the model.
 196     y: Target Dataset DistributedValue object. For example, when we use
 197         `MirroredStrategy` this is a PerDevice object with a tensor for each
 198         device set in the dict. y can also be a tuple or dict. The keys of the
 199         dict should match the names of the output layers of the model.
 200 
 201   Returns:
 202     The unwrapped values list of the x and y DistributedValues inputs.
 203 
 204   Raises:
 205     ValueError: If x and y do not have support for being evaluated as tensors.
 206         or if x and y contain elements that are not tensors or if x and y
 207         contain elements that have a shape or dtype mismatch.
 208   """
 209   # If the input and target used to call the model are not dataset tensors,
 210   # we need to raise an error. When using a DistributionStrategy, the input
 211   # and targets to a model should be from a `tf.data.Dataset`.
 212 
 213   # If each element of x and y are not tensors, we cannot standardize and
 214   # validate the input and targets.
 215   x_values_list = validate_per_device_inputs(distribution_strategy, x)
 216 
 217   if y is not None:
 218     y_values_list = validate_per_device_inputs(distribution_strategy, y)
 219   else:
 220     y_values_list = None
 221 
 222   # Return the unwrapped values to avoid calling `unwrap` a second time.
 223   return x_values_list, y_values_list

 226 def validate_per_device_inputs(distribution_strategy, x):
 227   """Validates PerDevice dataset input list.
 228 
 229   Args:
 230     distribution_strategy: The current DistributionStrategy used to call
 231       `fit`, `evaluate` and `predict`.
 232     x: A list of PerDevice objects that represent the input or
 233       target values.
 234 
 235   Returns:
 236     List containing the first element of each of the PerDevice objects in
 237     the input list.
 238 
 239   Raises:
 240     ValueError: If any of the objects in the `per_device_list` is not a tensor.
 241 
 242   """
 243   # Convert the inputs and targets into a list of PerDevice objects.
 244   per_device_list = nest.flatten(x)
 245   x_values_list = []
 246   for x in per_device_list:
 247     if not tensor_util.is_tensor(x):
 248       raise ValueError('Dataset input to the model should be tensors instead '
 249                        'they are of type {}'.format(type(x)))
 250 
 251     # At this point both x and y contain tensors in the `DistributedValues`
 252     # structure.
 253     x_values = distribution_strategy.unwrap(x)
 254 
 255     # Validate that the shape and dtype of all the elements in x are the same.
 256     validate_all_tensor_shapes(x, x_values)
 257     validate_all_tensor_types(x, x_values)
 258 
 259     x_values_list.append(x_values[0])
 260   return x_values_list
 261

 263 def validate_all_tensor_types(x, x_values):
 264   x_dtype = x_values[0].dtype
 265   for i in range(1, len(x_values)):
 266     if x_dtype != x_values[i].dtype:
 267       raise ValueError('Input tensor dtypes do not match for distributed tensor'
 268                        ' inputs {}'.format(x))
 269 
 270 
 271 def validate_all_tensor_shapes(x, x_values):
 272   # Validate that the shape of all the elements in x have the same shape
 273   x_shape = x_values[0].get_shape().as_list()
 274   for i in range(1, len(x_values)):
 275     if x_shape != x_values[i].get_shape().as_list():
 276       raise ValueError('Input tensor shapes do not match for distributed tensor'
 277                        ' inputs {}'.format(x))

 280 def configure_and_create_session(distribution_strategy):
 281   """Configure session config and create a session with it."""
 282   # TODO(priyag): Throw error if a session already exists.
 283   session_config = K.get_default_session_config()
 284   distribution_strategy.configure(session_config)
 285 
 286   if distribution_strategy.__class__.__name__ == 'TPUStrategy':
 287     # TODO(priyag): Remove this workaround when Distributed Coordinator is
 288     # integrated with keras and we can create a session from there.
 289     master = distribution_strategy._tpu_cluster_resolver.master()  # pylint: disable=protected-access
 290     session = session_module.Session(config=session_config, target=master)
 291   else:
 292     session = session_module.Session(config=session_config)
 293 
 294   K.set_session(session)

 297 def validate_inputs(x, y, distribution_strategy):
 298   """Validate inputs when using DistributionStrategy.
 299 
 300   Args:
 301     x: Model Inputs.
 302     y: Model Targets.
 303     distribution_strategy: The DistributionStrategy with which the model is
 304       compiled.
 305 
 306   Raises:
 307     ValueError: if input is not a Dataset or a numpy array.
 308   """
 309   if isinstance(x, list) or isinstance(y, list):
 310     raise ValueError('DistributionStrategy does not support lists of numpy'
 311                      'arrays. You must pass a Dataset object or a numpy array '
 312                      'as input.')
 313 
 314   if isinstance(x, dict) or isinstance(y, dict):
 315     raise ValueError('DistributionStrategy does not support inputs of type '
 316                      'dict. You must pass a Dataset object or a numpy array as '
 317                      'input.')
 318 
 319   if isinstance(x, iterator_ops.Iterator) or \
 320       isinstance(y, iterator_ops.Iterator):
 321     raise ValueError('DistributionStrategy does not support inputs of type '
 322                      'Iterator. You must pass a Dataset object or a numpy '
 323                      'array as input.')
 324 
 325   if distribution_strategy.__class__.__name__ == 'TPUStrategy':
 326     for i in [x, y]:
 327       if isinstance(i, dataset_ops.Dataset):
 328         shapes = nest.flatten(i.output_shapes)
 329         if any([not s.is_fully_defined() for s in shapes]):
 330           raise ValueError(
 331               'Using TPUs currently requires fully defined shapes. Either use '
 332               'set_shape() on the input tensors or use '
 333               'dataset.batch(..., drop_remainder=True).'
 334               'Found unknown shape {} in input {}.'.format(s, i))
 335

 337 def get_input_batch_params(first_x_value, batch_size, current_strategy):
 338   """Calculate the number of batches and steps/steps_per_epoch.
 339 
 340   Args:
 341     first_x_value: This is the first input numpy array that is passed in as the
 342       model input.
 343     batch_size: The specified batch_size or the default batch_size of 32.
 344     current_strategy: The current DistributionStrategy used to compile the
 345       model.
 346 
 347   Returns:
 348     The steps or steps_per_epoch argument depending on if a user is
 349     calling `fit`, `evaluate` or `predict`.
 350 
 351   Raises:
 352     ValueError: If the number of batches or steps evaluates to 0.
 353 
 354   """
 355   num_batches = first_x_value.shape[0] // batch_size
 356   if not num_batches:
 357     raise ValueError('Please specify a batch_size that is smaller than'
 358                      'the number of input samples %d.' % first_x_value.shape[0])
 359   # TODO(anjalisridhar): TPU currently supports using the num_towers property.
 360   # We might want to look into implementing worker_devices. In multi worker
 361   # strategy, perhaps num_towers works better?
 362   steps = num_batches // current_strategy.num_towers
 363   if not steps:
 364     # TODO(anjalisridhar): Number of towers in the error message may not convey
 365     # what we want to the user. Is there another terminology that we can use
 366     # that is consistent across different strategies.
 367     raise ValueError('The number of batches %d is smaller than the number '
 368                      'of towers %d used for DistributionStrategy. ' %
 369                      num_batches, current_strategy.num_towers)
 370   return steps

 373 def get_batch_dimension(iterator):
 374   shapes = nest.flatten(iterator.output_shapes)
 375   # Take the batch size from the first element, as it should be the same for
 376   # all.
 377   dims = shapes[0].dims
 378   return dims[0] if dims else None
