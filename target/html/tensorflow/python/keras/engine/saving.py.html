  50 @tf_export('keras.models.save_model')
  51 def save_model(model, filepath, overwrite=True, include_optimizer=True):
  52   """Saves a model to a HDF5 file.
  53 
  54   The saved model contains:
  55       - the model's configuration (topology)
  56       - the model's weights
  57       - the model's optimizer's state (if any)
  58 
  59   Thus the saved model can be reinstantiated in
  60   the exact same state, without any of the code
  61   used for model definition or training.
  62 
  63   Arguments:
  64       model: Keras model instance to be saved.
  65       filepath: One of the following:
  66           - String, path where to save the model
  67           - `h5py.File` object where to save the model
  68       overwrite: Whether we should overwrite any existing
  69           model at the target location, or instead
  70           ask the user with a manual prompt.
  71       include_optimizer: If True, save optimizer's state together.
  72 
  73   Raises:
  74       ImportError: if h5py is not available.
  75   """
  76 
  77   if h5py is None:
  78     raise ImportError('`save_model` requires h5py.')
  79 
  80   from tensorflow.python.keras import __version__ as keras_version  # pylint: disable=g-import-not-at-top
  81 
  82   if not isinstance(filepath, h5py.File):
  83     # If file exists and should not be overwritten.
  84     if not overwrite and os.path.isfile(filepath):
  85       proceed = ask_to_proceed_with_overwrite(filepath)
  86       if not proceed:
  87         return
  88 
  89     f = h5py.File(filepath, mode='w')
  90     opened_new_file = True
  91   else:
  92     f = filepath
  93     opened_new_file = False
  94 
  95   try:
  96     f.attrs['keras_version'] = str(keras_version).encode('utf8')
  97     f.attrs['backend'] = K.backend().encode('utf8')
  98     f.attrs['model_config'] = json.dumps(
  99         {
 100             'class_name': model.__class__.__name__,
 101             'config': model.get_config()
 102         },
 103         default=serialization.get_json_type).encode('utf8')
 104 
 105     model_weights_group = f.create_group('model_weights')
 106     model_layers = model.layers
 107     save_weights_to_hdf5_group(model_weights_group, model_layers)
 108 
 109     if include_optimizer and model.optimizer:
 110       if isinstance(model.optimizer, optimizers.TFOptimizer):
 111         logging.warning(
 112             'TensorFlow optimizers do not '
 113             'make it possible to access '
 114             'optimizer attributes or optimizer state '
 115             'after instantiation. '
 116             'As a result, we cannot save the optimizer '
 117             'as part of the model save file.'
 118             'You will have to compile your model again after loading it. '
 119             'Prefer using a Keras optimizer instead '
 120             '(see keras.io/optimizers).')
 121       else:
 122         f.attrs['training_config'] = json.dumps(
 123             {
 124                 'optimizer_config': {
 125                     'class_name': model.optimizer.__class__.__name__,
 126                     'config': model.optimizer.get_config()
 127                 },
 128                 'loss': model.loss,
 129                 'metrics': model.metrics,
 130                 'weighted_metrics': model.weighted_metrics,
 131                 'sample_weight_mode': model.sample_weight_mode,
 132                 'loss_weights': model.loss_weights,
 133             },
 134             default=serialization.get_json_type).encode('utf8')
 135 
 136         # Save optimizer weights.
 137         symbolic_weights = getattr(model.optimizer, 'weights')
 138         if symbolic_weights:
 139           optimizer_weights_group = f.create_group('optimizer_weights')
 140           weight_values = K.batch_get_value(symbolic_weights)
 141           weight_names = []
 142           for w, val in zip(symbolic_weights, weight_values):
 143             name = str(w.name)
 144             weight_names.append(name.encode('utf8'))
 145           optimizer_weights_group.attrs['weight_names'] = weight_names
 146           for name, val in zip(weight_names, weight_values):
 147             param_dset = optimizer_weights_group.create_dataset(
 148                 name, val.shape, dtype=val.dtype)
 149             if not val.shape:
 150               # scalar
 151               param_dset[()] = val
 152             else:
 153               param_dset[:] = val
 154     f.flush()
 155   finally:
 156     if opened_new_file:
 157       f.close()

 160 @tf_export('keras.models.load_model')
 161 def load_model(filepath, custom_objects=None, compile=True):  # pylint: disable=redefined-builtin
 162   """Loads a model saved via `save_model`.
 163 
 164   Arguments:
 165       filepath: One of the following:
 166           - String, path to the saved model
 167           - `h5py.File` object from which to load the model
 168       custom_objects: Optional dictionary mapping names
 169           (strings) to custom classes or functions to be
 170           considered during deserialization.
 171       compile: Boolean, whether to compile the model
 172           after loading.
 173 
 174   Returns:
 175       A Keras model instance. If an optimizer was found
 176       as part of the saved model, the model is already
 177       compiled. Otherwise, the model is uncompiled and
 178       a warning will be displayed. When `compile` is set
 179       to False, the compilation is omitted without any
 180       warning.
 181 
 182   Raises:
 183       ImportError: if h5py is not available.
 184       ValueError: In case of an invalid savefile.
 185   """
 186   if h5py is None:
 187     raise ImportError('`load_model` requires h5py.')
 188 
 189   if not custom_objects:
 190     custom_objects = {}
 191 
 192   def convert_custom_objects(obj):
 193     """Handles custom object lookup.
 194 
 195     Arguments:
 196         obj: object, dict, or list.
 197 
 198     Returns:
 199         The same structure, where occurrences
 200             of a custom object name have been replaced
 201             with the custom object.
 202     """
 203     if isinstance(obj, list):
 204       deserialized = []
 205       for value in obj:
 206         deserialized.append(convert_custom_objects(value))
 207       return deserialized
 208     if isinstance(obj, dict):
 209       deserialized = {}
 210       for key, value in obj.items():
 211         deserialized[key] = convert_custom_objects(value)
 212       return deserialized
 213     if obj in custom_objects:
 214       return custom_objects[obj]
 215     return obj
 216 
 217   opened_new_file = not isinstance(filepath, h5py.File)
 218   if opened_new_file:
 219     f = h5py.File(filepath, mode='r')
 220   else:
 221     f = filepath
 222 
 223   model = None
 224   try:
 225     # instantiate model
 226     model_config = f.attrs.get('model_config')
 227     if model_config is None:
 228       raise ValueError('No model found in config file.')
 229     model_config = json.loads(model_config.decode('utf-8'))
 230     model = model_from_config(model_config, custom_objects=custom_objects)
 231 
 232     # set weights
 233     load_weights_from_hdf5_group(f['model_weights'], model.layers)
 234 
 235     if compile:
 236       # instantiate optimizer
 237       training_config = f.attrs.get('training_config')
 238       if training_config is None:
 239         logging.warning('No training configuration found in save file: '
 240                         'the model was *not* compiled. Compile it manually.')
 241         return model
 242       training_config = json.loads(training_config.decode('utf-8'))
 243       optimizer_config = training_config['optimizer_config']
 244       optimizer = optimizers.deserialize(
 245           optimizer_config, custom_objects=custom_objects)
 246 
 247       # Recover loss functions and metrics.
 248       loss = convert_custom_objects(training_config['loss'])
 249       metrics = convert_custom_objects(training_config['metrics'])
 250       weighted_metrics = convert_custom_objects(
 251           training_config.get('weighted_metrics', None))
 252       sample_weight_mode = training_config['sample_weight_mode']
 253       loss_weights = training_config['loss_weights']
 254 
 255       # Compile model.
 256       model.compile(
 257           optimizer=optimizer,
 258           loss=loss,
 259           metrics=metrics,
 260           weighted_metrics=weighted_metrics,
 261           loss_weights=loss_weights,
 262           sample_weight_mode=sample_weight_mode)
 263 
 264       # Set optimizer weights.
 265       if 'optimizer_weights' in f:
 266         # Build train function (to get weight updates).
 267         model._make_train_function()
 268         optimizer_weights_group = f['optimizer_weights']
 269         optimizer_weight_names = [
 270             n.decode('utf8')
 271             for n in optimizer_weights_group.attrs['weight_names']
 272         ]
 273         optimizer_weight_values = [
 274             optimizer_weights_group[n] for n in optimizer_weight_names
 275         ]
 276         try:
 277           model.optimizer.set_weights(optimizer_weight_values)
 278         except ValueError:
 279           logging.warning('Error in loading the saved optimizer '
 280                           'state. As a result, your model is '
 281                           'starting with a freshly initialized '
 282                           'optimizer.')
 283   finally:
 284     if opened_new_file:
 285       f.close()
 286   return model

 354 def preprocess_weights_for_loading(layer,
 355                                    weights,
 356                                    original_keras_version=None,
 357                                    original_backend=None):
 358   """Preprocess layer weights between different Keras formats.
 359 
 360   Converts layers weights from Keras 1 format to Keras 2 and also weights of
 361   CuDNN layers in Keras 2.
 362 
 363   Arguments:
 364       layer: Layer instance.
 365       weights: List of weights values (Numpy arrays).
 366       original_keras_version: Keras version for the weights, as a string.
 367       original_backend: Keras backend the weights were trained with,
 368           as a string.
 369 
 370   Returns:
 371       A list of weights values (Numpy arrays).
 372   """
 373   def convert_nested_bidirectional(weights):
 374     """Converts layers nested in `Bidirectional` wrapper.
 375 
 376     This function uses `preprocess_weights_for_loading()` for converting
 377     layers.
 378 
 379     Arguments:
 380         weights: List of weights values (Numpy arrays).
 381 
 382     Returns:
 383         A list of weights values (Numpy arrays).
 384     """
 385     num_weights_per_layer = len(weights) // 2
 386     forward_weights = preprocess_weights_for_loading(
 387         layer.forward_layer, weights[:num_weights_per_layer],
 388         original_keras_version, original_backend)
 389     backward_weights = preprocess_weights_for_loading(
 390         layer.backward_layer, weights[num_weights_per_layer:],
 391         original_keras_version, original_backend)
 392     return forward_weights + backward_weights
 393 
 394   def convert_nested_time_distributed(weights):
 395     """Converts layers nested in `TimeDistributed` wrapper.
 396 
 397     This function uses `preprocess_weights_for_loading()` for converting nested
 398     layers.
 399 
 400     Arguments:
 401         weights: List of weights values (Numpy arrays).
 402 
 403     Returns:
 404         A list of weights values (Numpy arrays).
 405     """
 406     return preprocess_weights_for_loading(
 407         layer.layer, weights, original_keras_version, original_backend)
 408 
 409   def convert_nested_model(weights):
 410     """Converts layers nested in `Model` or `Sequential`.
 411 
 412     This function uses `preprocess_weights_for_loading()` for converting nested
 413     layers.
 414 
 415     Arguments:
 416         weights: List of weights values (Numpy arrays).
 417 
 418     Returns:
 419         A list of weights values (Numpy arrays).
 420     """
 421     new_weights = []
 422     # trainable weights
 423     for sublayer in layer.layers:
 424       num_weights = len(sublayer.trainable_weights)
 425       if num_weights > 0:
 426         new_weights.extend(preprocess_weights_for_loading(
 427             layer=sublayer,
 428             weights=weights[:num_weights],
 429             original_keras_version=original_keras_version,
 430             original_backend=original_backend))
 431         weights = weights[num_weights:]
 432 
 433     # non-trainable weights
 434     for sublayer in layer.layers:
 435       num_weights = len([l for l in sublayer.weights
 436                          if l not in sublayer.trainable_weights])
 437       if num_weights > 0:
 438         new_weights.extend(preprocess_weights_for_loading(
 439             layer=sublayer,
 440             weights=weights[:num_weights],
 441             original_keras_version=original_keras_version,
 442             original_backend=original_backend))
 443         weights = weights[num_weights:]
 444     return new_weights
 445 
 446   # Convert layers nested in Bidirectional/Model/Sequential.
 447   # Both transformation should be ran for both Keras 1->2 conversion
 448   # and for conversion of CuDNN layers.
 449   if layer.__class__.__name__ == 'Bidirectional':
 450     weights = convert_nested_bidirectional(weights)
 451   if layer.__class__.__name__ == 'TimeDistributed':
 452     weights = convert_nested_time_distributed(weights)
 453   elif layer.__class__.__name__ in ['Model', 'Sequential']:
 454     weights = convert_nested_model(weights)
 455 
 456   if original_keras_version == '1':
 457     if layer.__class__.__name__ == 'TimeDistributed':
 458       weights = preprocess_weights_for_loading(
 459           layer.layer, weights, original_keras_version, original_backend)
 460 
 461     if layer.__class__.__name__ == 'Conv1D':
 462       shape = weights[0].shape
 463       # Handle Keras 1.1 format
 464       if shape[:2] != (layer.kernel_size[0], 1) or shape[3] != layer.filters:
 465         # Legacy shape:
 466         # (filters, input_dim, filter_length, 1)
 467         assert shape[0] == layer.filters and shape[2:] == (layer.kernel_size[0],
 468                                                            1)
 469         weights[0] = np.transpose(weights[0], (2, 3, 1, 0))
 470       weights[0] = weights[0][:, 0, :, :]
 471 
 472     if layer.__class__.__name__ == 'Conv2D':
 473       if layer.data_format == 'channels_first':
 474         # old: (filters, stack_size, kernel_rows, kernel_cols)
 475         # new: (kernel_rows, kernel_cols, stack_size, filters)
 476         weights[0] = np.transpose(weights[0], (2, 3, 1, 0))
 477 
 478     if layer.__class__.__name__ == 'Conv2DTranspose':
 479       if layer.data_format == 'channels_last':
 480         # old: (kernel_rows, kernel_cols, stack_size, filters)
 481         # new: (kernel_rows, kernel_cols, filters, stack_size)
 482         weights[0] = np.transpose(weights[0], (0, 1, 3, 2))
 483       if layer.data_format == 'channels_first':
 484         # old: (filters, stack_size, kernel_rows, kernel_cols)
 485         # new: (kernel_rows, kernel_cols, filters, stack_size)
 486         weights[0] = np.transpose(weights[0], (2, 3, 0, 1))
 487 
 488     if layer.__class__.__name__ == 'Conv3D':
 489       if layer.data_format == 'channels_first':
 490         # old: (filters, stack_size, ...)
 491         # new: (..., stack_size, filters)
 492         weights[0] = np.transpose(weights[0], (2, 3, 4, 1, 0))
 493 
 494     if layer.__class__.__name__ == 'GRU':
 495       if len(weights) == 9:
 496         kernel = np.concatenate([weights[0], weights[3], weights[6]], axis=-1)
 497         recurrent_kernel = np.concatenate(
 498             [weights[1], weights[4], weights[7]], axis=-1)
 499         bias = np.concatenate([weights[2], weights[5], weights[8]], axis=-1)
 500         weights = [kernel, recurrent_kernel, bias]
 501 
 502     if layer.__class__.__name__ == 'LSTM':
 503       if len(weights) == 12:
 504         # old: i, c, f, o
 505         # new: i, f, c, o
 506         kernel = np.concatenate(
 507             [weights[0], weights[6], weights[3], weights[9]], axis=-1)
 508         recurrent_kernel = np.concatenate(
 509             [weights[1], weights[7], weights[4], weights[10]], axis=-1)
 510         bias = np.concatenate(
 511             [weights[2], weights[8], weights[5], weights[11]], axis=-1)
 512         weights = [kernel, recurrent_kernel, bias]
 513 
 514     if layer.__class__.__name__ == 'ConvLSTM2D':
 515       if len(weights) == 12:
 516         kernel = np.concatenate(
 517             [weights[0], weights[6], weights[3], weights[9]], axis=-1)
 518         recurrent_kernel = np.concatenate(
 519             [weights[1], weights[7], weights[4], weights[10]], axis=-1)
 520         bias = np.concatenate(
 521             [weights[2], weights[8], weights[5], weights[11]], axis=-1)
 522         if layer.data_format == 'channels_first':
 523           # old: (filters, stack_size, kernel_rows, kernel_cols)
 524           # new: (kernel_rows, kernel_cols, stack_size, filters)
 525           kernel = np.transpose(kernel, (2, 3, 1, 0))
 526           recurrent_kernel = np.transpose(recurrent_kernel, (2, 3, 1, 0))
 527         weights = [kernel, recurrent_kernel, bias]
 528 
 529   conv_layers = ['Conv1D', 'Conv2D', 'Conv3D', 'Conv2DTranspose', 'ConvLSTM2D']
 530   if layer.__class__.__name__ in conv_layers:
 531     if original_backend == 'theano':
 532       weights[0] = conv_utils.convert_kernel(weights[0])
 533       if layer.__class__.__name__ == 'ConvLSTM2D':
 534         weights[1] = conv_utils.convert_kernel(weights[1])
 535     if K.int_shape(layer.weights[0]) != weights[0].shape:
 536       weights[0] = np.transpose(weights[0], (3, 2, 0, 1))
 537       if layer.__class__.__name__ == 'ConvLSTM2D':
 538         weights[1] = np.transpose(weights[1], (3, 2, 0, 1))
 539 
 540   # convert CuDNN layers
 541   return _convert_rnn_weights(layer, weights)

 544 def _convert_rnn_weights(layer, weights):
 545   """Converts weights for RNN layers between native and CuDNN format.
 546 
 547   Input kernels for each gate are transposed and converted between Fortran
 548   and C layout, recurrent kernels are transposed. For LSTM biases are summed/
 549   split in half, for GRU biases are reshaped.
 550 
 551   Weights can be converted in both directions between `LSTM` and`CuDNNSLTM`
 552   and between `CuDNNGRU` and `GRU(reset_after=True)`. Default `GRU` is not
 553   compatible with `CuDNNGRU`.
 554 
 555   For missing biases in `LSTM`/`GRU` (`use_bias=False`) no conversion is made.
 556 
 557   Arguments:
 558       layer: Target layer instance.
 559       weights: List of source weights values (input kernels, recurrent
 560           kernels, [biases]) (Numpy arrays).
 561 
 562   Returns:
 563       A list of converted weights values (Numpy arrays).
 564 
 565   Raises:
 566       ValueError: for incompatible GRU layer/weights or incompatible biases
 567   """
 568 
 569   def transform_kernels(kernels, func, n_gates):
 570     """Transforms kernel for each gate separately using given function.
 571 
 572     Arguments:
 573         kernels: Stacked array of kernels for individual gates.
 574         func: Function applied to kernel of each gate.
 575         n_gates: Number of gates (4 for LSTM, 3 for GRU).
 576 
 577     Returns:
 578         Stacked array of transformed kernels.
 579     """
 580     return np.hstack([func(k) for k in np.hsplit(kernels, n_gates)])
 581 
 582   def transpose_input(from_cudnn):
 583     """Makes a function that transforms input kernels from/to CuDNN format.
 584 
 585     It keeps the shape, but changes between the layout (Fortran/C). Eg.:
 586 
 587     ```
 588     Keras                 CuDNN
 589     [[0, 1, 2],  <--->  [[0, 2, 4],
 590      [3, 4, 5]]          [1, 3, 5]]
 591     ```
 592 
 593     It can be passed to `transform_kernels()`.
 594 
 595     Arguments:
 596         from_cudnn: `True` if source weights are in CuDNN format, `False`
 597             if they're in plain Keras format.
 598 
 599     Returns:
 600         Function that converts input kernel to the other format.
 601     """
 602     order = 'F' if from_cudnn else 'C'
 603 
 604     def transform(kernel):
 605       return kernel.T.reshape(kernel.shape, order=order)
 606 
 607     return transform
 608 
 609   target_class = layer.__class__.__name__
 610 
 611   # convert the weights between CuDNNLSTM and LSTM
 612   if target_class in ['LSTM', 'CuDNNLSTM'] and len(weights) == 3:
 613     # determine if we're loading a CuDNNLSTM layer
 614     # from the number of bias weights:
 615     # CuDNNLSTM has (units * 8) weights; while LSTM has (units * 4)
 616     # if there's no bias weight in the file, skip this conversion
 617     units = weights[1].shape[0]
 618     bias_shape = weights[2].shape
 619     n_gates = 4
 620 
 621     if bias_shape == (2 * units * n_gates,):
 622       source = 'CuDNNLSTM'
 623     elif bias_shape == (units * n_gates,):
 624       source = 'LSTM'
 625     else:
 626       raise ValueError('Invalid bias shape: ' + str(bias_shape))

 628     def convert_lstm_weights(weights, from_cudnn=True):
 629       """Converts the weights between CuDNNLSTM and LSTM.
 630 
 631       Arguments:
 632         weights: Original weights.
 633         from_cudnn: Indicates whether original weights are from CuDNN layer.
 634 
 635       Returns:
 636         Updated weights compatible with LSTM.
 637       """
 638 
 639       # Transpose (and reshape) input and recurrent kernels
 640       kernels = transform_kernels(weights[0], transpose_input(from_cudnn),
 641                                   n_gates)
 642       recurrent_kernels = transform_kernels(weights[1], lambda k: k.T, n_gates)
 643       if from_cudnn:
 644         # merge input and recurrent biases into a single set
 645         biases = np.sum(np.split(weights[2], 2, axis=0), axis=0)
 646       else:
 647         # Split single set of biases evenly to two sets. The way of
 648         # splitting doesn't matter as long as the two sets sum is kept.
 649         biases = np.tile(0.5 * weights[2], 2)
 650       return [kernels, recurrent_kernels, biases]
 651 
 652     if source != target_class:
 653       weights = convert_lstm_weights(weights, from_cudnn=source == 'CuDNNLSTM')
 654 
 655   # convert the weights between CuDNNGRU and GRU(reset_after=True)
 656   if target_class in ['GRU', 'CuDNNGRU'] and len(weights) == 3:
 657     # We can determine the source of the weights from the shape of the bias.
 658     # If there is no bias we skip the conversion since
 659     # CuDNNGRU always has biases.
 660 
 661     units = weights[1].shape[0]
 662     bias_shape = weights[2].shape
 663     n_gates = 3
 664 
 665     def convert_gru_weights(weights, from_cudnn=True):
 666       """Converts the weights between CuDNNGRU and GRU.
 667 
 668       Arguments:
 669         weights: Original weights.
 670         from_cudnn: Indicates whether original weights are from CuDNN layer.
 671 
 672       Returns:
 673         Updated weights compatible with GRU.
 674       """
 675 
 676       kernels = transform_kernels(weights[0], transpose_input(from_cudnn),
 677                                   n_gates)
 678       recurrent_kernels = transform_kernels(weights[1], lambda k: k.T, n_gates)
 679       biases = np.array(weights[2]).reshape((2, -1) if from_cudnn else -1)
 680       return [kernels, recurrent_kernels, biases]
 681 
 682     if bias_shape == (2 * units * n_gates,):
 683       source = 'CuDNNGRU'
 684     elif bias_shape == (2, units * n_gates):
 685       source = 'GRU(reset_after=True)'
 686     elif bias_shape == (units * n_gates,):
 687       source = 'GRU(reset_after=False)'
 688     else:
 689       raise ValueError('Invalid bias shape: ' + str(bias_shape))
 690 
 691     if target_class == 'CuDNNGRU':
 692       target = 'CuDNNGRU'
 693     elif layer.reset_after:
 694       target = 'GRU(reset_after=True)'
 695     else:
 696       target = 'GRU(reset_after=False)'
 697 
 698     # only convert between different types
 699     if source != target:
 700       types = (source, target)
 701       if 'GRU(reset_after=False)' in types:
 702         raise ValueError('%s is not compatible with %s' % types)
 703       if source == 'CuDNNGRU':
 704         weights = convert_gru_weights(weights, from_cudnn=True)
 705       elif source == 'GRU(reset_after=True)':
 706         weights = convert_gru_weights(weights, from_cudnn=False)
 707 
 708   return weights
 709

 711 def save_weights_to_hdf5_group(f, layers):
 712   """Saves the weights of a list of layers to a HDF5 group.
 713 
 714   Arguments:
 715       f: HDF5 group.
 716       layers: List of layer instances.

 718   from tensorflow.python.keras import __version__ as keras_version  # pylint: disable=g-import-not-at-top
 719 
 720   save_attributes_to_hdf5_group(
 721       f, 'layer_names', [layer.name.encode('utf8') for layer in layers])

 723   f.attrs['keras_version'] = str(keras_version).encode('utf8')

 739       if not val.shape:
 740         # scalar

 746 def load_weights_from_hdf5_group(f, layers):
 747   """Implements topological (order-based) weight loading.
 748 
 749   Arguments:
 750       f: A pointer to a HDF5 group.
 751       layers: a list of target layers.
 752 
 753   Raises:
 754       ValueError: in case of mismatch between provided layers
 755           and weights file.
 756   """
 757   if 'keras_version' in f.attrs:
 758     original_keras_version = f.attrs['keras_version'].decode('utf8')
 759   else:
 760     original_keras_version = '1'
 761   if 'backend' in f.attrs:
 762     original_backend = f.attrs['backend'].decode('utf8')
 763   else:
 764     original_backend = None
 765 
 766   filtered_layers = []
 767   for layer in layers:
 768     weights = layer.weights
 769     if weights:
 770       filtered_layers.append(layer)
 771 
 772   layer_names = load_attributes_from_hdf5_group(f, 'layer_names')
 773   filtered_layer_names = []
 774   for name in layer_names:
 775     g = f[name]
 776     weight_names = load_attributes_from_hdf5_group(g, 'weight_names')
 777     if weight_names:
 778       filtered_layer_names.append(name)
 779   layer_names = filtered_layer_names
 780   if len(layer_names) != len(filtered_layers):
 781     raise ValueError('You are trying to load a weight file '
 782                      'containing ' + str(len(layer_names)) +
 783                      ' layers into a model with ' + str(len(filtered_layers)) +
 784                      ' layers.')
 785 
 786   # We batch weight value assignments in a single backend call
 787   # which provides a speedup in TensorFlow.
 788   weight_value_tuples = []
 789   for k, name in enumerate(layer_names):
 790     g = f[name]
 791     weight_names = load_attributes_from_hdf5_group(g, 'weight_names')
 792     weight_values = [np.asarray(g[weight_name]) for weight_name in weight_names]
 793     layer = filtered_layers[k]
 794     symbolic_weights = layer.weights
 795     weight_values = preprocess_weights_for_loading(
 796         layer, weight_values, original_keras_version, original_backend)
 797     if len(weight_values) != len(symbolic_weights):
 798       raise ValueError('Layer #' + str(k) + ' (named "' + layer.name +
 799                        '" in the current model) was found to '
 800                        'correspond to layer ' + name + ' in the save file. '
 801                        'However the new layer ' + layer.name + ' expects ' +
 802                        str(len(symbolic_weights)) +
 803                        ' weights, but the saved weights have ' +
 804                        str(len(weight_values)) + ' elements.')
 805     weight_value_tuples += zip(symbolic_weights, weight_values)
 806   K.batch_set_value(weight_value_tuples)
 807

 809 def load_weights_from_hdf5_group_by_name(f, layers):
 810   """Implements name-based weight loading.
 811 
 812   (instead of topological weight loading).
 813 
 814   Layers that have no matching name are skipped.
 815 
 816   Arguments:
 817       f: A pointer to a HDF5 group.
 818       layers: a list of target layers.
 819 
 820   Raises:
 821       ValueError: in case of mismatch between provided layers
 822           and weights file.
 823   """
 824   if 'keras_version' in f.attrs:
 825     original_keras_version = f.attrs['keras_version'].decode('utf8')
 826   else:
 827     original_keras_version = '1'
 828   if 'backend' in f.attrs:
 829     original_backend = f.attrs['backend'].decode('utf8')
 830   else:
 831     original_backend = None
 832 
 833   # New file format.
 834   layer_names = load_attributes_from_hdf5_group(f, 'layer_names')
 835 
 836   # Reverse index of layer name to list of layers with name.
 837   index = {}
 838   for layer in layers:
 839     if layer.name:
 840       index.setdefault(layer.name, []).append(layer)
 841 
 842   # We batch weight value assignments in a single backend call
 843   # which provides a speedup in TensorFlow.
 844   weight_value_tuples = []
 845   for k, name in enumerate(layer_names):
 846     g = f[name]
 847     weight_names = load_attributes_from_hdf5_group(g, 'weight_names')
 848     weight_values = [np.asarray(g[weight_name]) for weight_name in weight_names]
 849 
 850     for layer in index.get(name, []):
 851       symbolic_weights = layer.weights
 852       weight_values = preprocess_weights_for_loading(
 853           layer, weight_values, original_keras_version, original_backend)
 854       if len(weight_values) != len(symbolic_weights):
 855         raise ValueError('Layer #' + str(k) + ' (named "' + layer.name +
 856                          '") expects ' + str(len(symbolic_weights)) +
 857                          ' weight(s), but the saved weights' + ' have ' +
 858                          str(len(weight_values)) + ' element(s).')
 859       # Set values.
 860       for i in range(len(weight_values)):
 861         if K.int_shape(symbolic_weights[i]) != weight_values[i].shape:
 862           raise ValueError('Layer #' + str(k) +' (named "' + layer.name +
 863                            '"), weight ' + str(symbolic_weights[i]) +
 864                            ' has shape {}'.format(K.int_shape(
 865                                symbolic_weights[i])) +
 866                            ', but the saved weight has shape ' +
 867                            str(weight_values[i].shape) + '.')
 868 
 869         else:
 870           weight_value_tuples.append((symbolic_weights[i], weight_values[i]))
 871   K.batch_set_value(weight_value_tuples)

 874 def save_attributes_to_hdf5_group(group, name, data):
 875   """Saves attributes (data) of the specified name into the HDF5 group.
 876 
 877   This method deals with an inherent problem of HDF5 file which is not
 878   able to store data larger than HDF5_OBJECT_HEADER_LIMIT bytes.
 879 
 880   Arguments:
 881       group: A pointer to a HDF5 group.
 882       name: A name of the attributes to save.
 883       data: Attributes data to store.
 884 
 885   Raises:
 886     RuntimeError: If any single attribute is too large to be saved.
 887   """
 888   # Check that no item in `data` is larger than `HDF5_OBJECT_HEADER_LIMIT`
 889   # because in that case even chunking the array would not make the saving
 890   # possible.
 891   bad_attributes = [x for x in data if len(x) > HDF5_OBJECT_HEADER_LIMIT]
 892 
 893   # Expecting this to never be true.
 894   if bad_attributes:
 895     raise RuntimeError('The following attributes cannot be saved to HDF5 '
 896                        'file because they are larger than %d bytes: %s' %
 897                        (HDF5_OBJECT_HEADER_LIMIT,
 898                         ', '.join([x for x in bad_attributes])))
 899 
 900   data_npy = np.asarray(data)
 901 
 902   num_chunks = 1
 903   chunked_data = np.array_split(data_npy, num_chunks)
 904 
 905   # This will never loop forever thanks to the test above.
 906   while any([x.nbytes > HDF5_OBJECT_HEADER_LIMIT for x in chunked_data]):
 907     num_chunks += 1
 908     chunked_data = np.array_split(data_npy, num_chunks)
 909 
 910   if num_chunks > 1:
 911     for chunk_id, chunk_data in enumerate(chunked_data):
 912       group.attrs['%s%d' % (name, chunk_id)] = chunk_data
 913   else:
 914     group.attrs[name] = data
 915

 917 def load_attributes_from_hdf5_group(group, name):
 918   """Loads attributes of the specified name from the HDF5 group.
 919 
 920   This method deals with an inherent problem
 921   of HDF5 file which is not able to store
 922   data larger than HDF5_OBJECT_HEADER_LIMIT bytes.
 923 
 924   Arguments:
 925       group: A pointer to a HDF5 group.
 926       name: A name of the attributes to load.
 927 
 928   Returns:
 929       data: Attributes data.
 930   """
 931   if name in group.attrs:
 932     data = [n.decode('utf8') for n in group.attrs[name]]
 933   else:
 934     data = []
 935     chunk_id = 0
 936     while '%s%d' % (name, chunk_id) in group.attrs:
 937       data.extend(
 938           [n.decode('utf8') for n in group.attrs['%s%d' % (name, chunk_id)]])
 939       chunk_id += 1
 940   return data
