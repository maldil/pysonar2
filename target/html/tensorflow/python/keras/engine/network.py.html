  70   It is the topological form of a "model". A `Model`
  71   is simply a `Network` with added training routines.

  85   # Several Network methods have "no_automatic_dependency_tracking"
  86   # annotations. Since Network does automatic dependency tracking on attribute
  87   # assignment, including for common data structures such as lists, by default
  88   # we'd have quite a few empty dependencies which users don't care about (or
  89   # would need some way to ignore dependencies automatically, which is confusing
  90   # when applied to user code). Some attributes, such as _layers, would cause
  91   # structural issues (_layers being the place where Layers assigned to tracked
  92   # attributes are stored).
  93   #
  94   # Aside from these aesthetic and structural issues, useless dependencies on
  95   # empty lists shouldn't cause issues; adding or removing them will not break
  96   # checkpoints, but may cause "all Python objects matched" assertions to fail
  97   # (in which case less strict assertions may be substituted if necessary).
  98   @checkpointable.no_automatic_dependency_tracking
  99   def _base_init(self, name=None):
 100     # The following are implemented as property functions:
 101     # self.trainable_weights
 102     # self.non_trainable_weights
 103     # self.input_spec
 104     # self.losses
 105     # self.updates
 106 
 107     self._init_set_name(name, zero_based=True)
 108     self._activity_regularizer = None
 109     # This acts just like the `trainable` attribute of any layer instance.
 110     # It does not affect users of the underlying layers, only users of the
 111     # Network instance.
 112     self.trainable = True
 113     self._is_compiled = False
 114     self._expects_training_arg = False
 115     # A list of "extra" variables assigned to attributes of this class, included
 116     # in self.weights and self.variables. Always empty for graph networks (but
 117     # included in base_init to avoid excessive special casing when retrieving
 118     # the value).
 119     self._extra_variables = []
 120     # In many internal cases one needs to compute both the model's output
 121     # and its output mask without relying on `__call__` (which would do both and
 122     # set mask metadata), but for models, computing the mask requires to
 123     # recompute the output.
 124     # Hence the pattern `output = model.call(); mask = model.compute_mask()`
 125     # would be redundant, and internal logic
 126     # (susceptible to use `call` directly) should prefer using the
 127     # internal method `output, mask = _call_and_compute_mask()`.
 128     # This is True for Sequential networks and graph networks.
 129     self._compute_output_and_mask_jointly = False
 130 
 131     self.supports_masking = False
 132     if not hasattr(self, 'optimizer'):
 133       # Don't reset optimizer if already set.
 134       self.optimizer = None
 135 
 136     # Private attributes to implement compatibility with Layer.
 137     self._updates = []  # Used in symbolic mode only.
 138     self._losses = []   # Used in symbolic mode only.
 139     self._scope = None  # Never used.
 140     self._reuse = None  # Never used.
 141     if context.executing_eagerly():
 142       self._graph = None
 143     else:
 144       self._graph = ops.get_default_graph()  # Used in symbolic mode only.
 145       # A Network does not create weights of its own, thus has no dtype.
 146     self._dtype = None
 147 
 148     # All layers in order of horizontal graph traversal.
 149     # Entries are unique. Includes input and output layers.
 150     self._layers = []
 151 
 152     # Used in symbolic mode only, only in conjunction with graph-networks
 153     self._outbound_nodes = []
 154     self._inbound_nodes = []
 155 
 156     self._checkpointable_saver = checkpointable_utils.CheckpointableSaver(
 157         weakref.ref(self))

 159   @checkpointable.no_automatic_dependency_tracking
 160   def _init_graph_network(self, inputs, outputs, name=None):
 161     self._call_convention = base_layer.CallConvention.EXPLICIT_INPUTS_ARGUMENT
 162     # Normalize and set self.inputs, self.outputs.
 163     if isinstance(inputs, (list, tuple)):
 164       self.inputs = list(inputs)  # Tensor or list of tensors.
 165     else:
 166       self.inputs = [inputs]
 167     if isinstance(outputs, (list, tuple)):
 168       self.outputs = list(outputs)
 169     else:
 170       self.outputs = [outputs]
 171 
 172     # User-provided argument validation.
 173     if context.executing_eagerly():
 174       # Check that all inputs/outputs are DeferredTensors.
 175       for tensor in self.inputs:
 176         if not isinstance(tensor, base_layer.DeferredTensor):  # pylint: disable=protected-access
 177           raise TypeError('When eager execution is enabled, '
 178                           'inputs must come from a call to '
 179                           '`tf.keras.Input` (called after '
 180                           'tf.enable_eager_execution()). '
 181                           'Received invalid input: ' + str(tensor))
 182       for tensor in self.outputs:
 183         if not isinstance(tensor, base_layer.DeferredTensor):  # pylint: disable=protected-access
 184           raise TypeError('When eager execution is enabled, '
 185                           'outputs must come from a call to '
 186                           'a layer (called after '
 187                           'tf.enable_eager_execution()). '
 188                           'Received invalid output: ' + str(tensor))
 189     # Check for redundancy in inputs.
 190     if len(set(self.inputs)) != len(self.inputs):
 191       raise ValueError('The list of inputs passed to the model '
 192                        'is redundant. '
 193                        'All inputs should only appear once.'
 194                        ' Found: ' + str(self.inputs))
 195     for x in self.inputs:
 196       # Check that x has appropriate `_keras_history` metadata.
 197       if not hasattr(x, '_keras_history'):
 198         cls_name = self.__class__.__name__
 199         raise ValueError('Input tensors to a ' + cls_name + ' ' +
 200                          'must come from `tf.layers.Input`. '
 201                          'Received: ' + str(x) +
 202                          ' (missing previous layer metadata).')
 203       # Check that x is an input tensor.
 204       # pylint: disable=protected-access
 205       layer, node_index, tensor_index = x._keras_history
 206       if len(layer._inbound_nodes) > 1 or (
 207           layer._inbound_nodes and layer._inbound_nodes[0].inbound_layers):
 208         cls_name = self.__class__.__name__
 209         logging.warning(cls_name + ' inputs must come from '
 210                         '`tf.layers.Input` (thus holding past layer metadata), '
 211                         'they cannot be the output of '
 212                         'a previous non-Input layer. '
 213                         'Here, a tensor specified as '
 214                         'input to "' + self.name + '" was not an Input tensor, '
 215                         'it was generated by layer ' + layer.name + '.\n'
 216                         'Note that input tensors are '
 217                         'instantiated via `tensor = tf.layers.Input(shape)`.\n'
 218                         'The tensor that caused the issue was: ' + str(x.name))
 219     for x in self.outputs:
 220       if not hasattr(x, '_keras_history'):
 221         cls_name = self.__class__.__name__
 222         raise ValueError('Output tensors to a ' + cls_name + ' must be '
 223                          'the output of a TensorFlow `Layer` '
 224                          '(thus holding past layer metadata). Found: ' + str(x))
 225 
 226     self._base_init(name=name)
 227     self._compute_previous_mask = (
 228         'mask' in tf_inspect.getfullargspec(self.call).args or
 229         hasattr(self, 'compute_mask'))
 230     # A Network does not create weights of its own, thus it is already
 231     # built.
 232     self.built = True
 233     self._compute_output_and_mask_jointly = True
 234     self._is_graph_network = True
 235 
 236     self._input_layers = []
 237     self._output_layers = []
 238     self._input_coordinates = []
 239     self._output_coordinates = []
 240 
 241     # This is for performance optimization when calling the Network on new
 242     # inputs. Every time the Network is called on a set on input tensors,
 243     # we compute the output tensors, output masks and output shapes in one pass,
 244     # then cache them here. When any of these outputs is queried later, we
 245     # retrieve it from there instead of recomputing it.
 246     self._output_mask_cache = {}
 247     self._output_tensor_cache = {}
 248     self._output_shape_cache = {}
 249 
 250     # Build self._output_layers:
 251     for x in self.outputs:
 252       layer, node_index, tensor_index = x._keras_history  # pylint: disable=protected-access
 253       self._output_layers.append(layer)
 254       self._output_coordinates.append((layer, node_index, tensor_index))
 255 
 256     # Build self._input_layers:
 257     for x in self.inputs:
 258       layer, node_index, tensor_index = x._keras_history  # pylint: disable=protected-access
 259       # It's supposed to be an input layer, so only one node
 260       # and one tensor output.
 261       assert node_index == 0
 262       assert tensor_index == 0
 263       self._input_layers.append(layer)
 264       self._input_coordinates.append((layer, node_index, tensor_index))
 265 
 266     # Keep track of the network's nodes and layers.
 267     nodes, nodes_by_depth, layers, layers_by_depth = _map_graph_network(
 268         self.inputs, self.outputs)
 269     self._network_nodes = nodes
 270     self._nodes_by_depth = nodes_by_depth
 271     self._layers = layers
 272     self._layers_by_depth = layers_by_depth
 273 
 274     self._track_layers(layers)
 275 
 276     # Create the node linking internal inputs to internal outputs.
 277     base_layer.Node(
 278         outbound_layer=self,
 279         inbound_layers=[],
 280         node_indices=[],
 281         tensor_indices=[],
 282         input_tensors=self.inputs,
 283         output_tensors=self.outputs)
 284 
 285     # Build self.input_names and self.output_names.
 286     self.input_names = []
 287     self.output_names = []
 288     self._feed_input_names = []
 289     self._feed_inputs = []
 290     self._feed_input_shapes = []
 291     for i, layer in enumerate(self._input_layers):
 292       self.input_names.append(layer.name)
 293       if layer.is_placeholder:
 294         self._feed_input_names.append(layer.name)
 295         self._feed_input_shapes.append(backend.int_shape(self.inputs[i]))
 296         # layer.input gives an error in eager mode
 297         if not context.executing_eagerly():
 298           self._feed_inputs.append(layer.input)
 299     for layer in self._output_layers:
 300       self.output_names.append(layer.name)

 316   def _determine_call_convention(self, call_argspec):
 317     """Decides how `self.call()` is invoked. See base_layer.CallConvention."""
 318     if call_argspec.varargs:
 319       may_take_single_argument = False
 320     else:
 321       try:
 322         # Note: tf_inspect doesn't raise a TypeError when regular inspect would,
 323         # so we need to keep in mind that "getcallargs" may have returned
 324         # something even though we under-specified positional arguments.
 325         all_args = tf_inspect.getcallargs(self.call, None)
 326         self_args = set()
 327         for arg_name, obj in all_args.items():
 328           if obj is self:
 329             self_args.add(arg_name)
 330         may_take_single_argument = True
 331       except TypeError:
 332         may_take_single_argument = False
 333     if may_take_single_argument:
 334       # A single positional argument (plus "self") is considered equivalent to
 335       # an "inputs" argument.
 336       all_positional_args = len(call_argspec.args)
 337       if call_argspec.defaults is not None:
 338         all_positional_args -= len(call_argspec.defaults)
 339       non_self_positional_args = all_positional_args
 340       for positional_arg_name in call_argspec.args[:all_positional_args]:
 341         if positional_arg_name in self_args:
 342           non_self_positional_args -= 1
 343       if non_self_positional_args == 1:
 344         if 'inputs' in call_argspec.args[all_positional_args:]:
 345           raise TypeError(
 346               "Model.call() takes a single positional argument (to which "
 347               "inputs are passed by convention) and a separate 'inputs' "
 348               "argument. Unable to determine which arguments are inputs.")
 349         return base_layer.CallConvention.SINGLE_POSITIONAL_ARGUMENT
 350     if 'inputs' in call_argspec.args:
 351       return base_layer.CallConvention.EXPLICIT_INPUTS_ARGUMENT
 352     else:
 353       return base_layer.CallConvention.POSITIONAL_ARGUMENTS_ARE_INPUTS

 355   def _track_layers(self, layers):
 356     """Add Checkpointable dependencies on a list of Layers."""
 357     weight_layer_index = 0
 358     for layer_index, layer in enumerate(layers):
 359       if layer.weights:
 360         # Keep a separate index for layers which have weights. This allows users
 361         # to insert Layers without weights anywhere in the network without
 362         # breaking checkpoints.
 363         self._track_checkpointable(
 364             layer, name='layer_with_weights-%d' % weight_layer_index,
 365             overwrite=True)
 366         weight_layer_index += 1
 367       # Even if it doesn't have weights, we should still track everything in
 368       # case it has/will have Checkpointable dependencies.
 369       self._track_checkpointable(
 370           layer, name='layer-%d' % layer_index, overwrite=True)

 372   def _no_dependency(self, value):
 373     """Override to allow `Layer` to disable dependency tracking.
 374 
 375     `CheckpointableBase` defines this method, whose semantics are "if a subclass
 376     does dependency tracking, this method exempts `value`." Layer uses
 377     `_no_dependency` to exempt some of its attribute assignments (conditional on
 378     attribute assignment causing tracking in the subclass).
 379 
 380     Args:
 381       value: An object which will be assigned to an object attribute, whose
 382         value should not be tracked.
 383 
 384     Returns:
 385       A wrapped object which, when assigned to an attribute, will not be
 386       tracked (`value` will be stored in the attribute).
 387     """
 388     return data_structures.NoDependency(value)

 390   def __setattr__(self, name, value):
 391     if not getattr(self, '_setattr_tracking', True):
 392       super(Network, self).__setattr__(name, value)
 393       return
 394     no_dependency = isinstance(value, data_structures.NoDependency)
 395     value = data_structures.sticky_attribute_assignment(
 396         checkpointable=self, value=value, name=name)
 397     if (isinstance(value, (base_layer.Layer,
 398                            Network,
 399                            data_structures.CheckpointableDataStructure))
 400         or checkpointable_layer_utils.has_weights(value)):
 401       try:
 402         is_graph_network = self._is_graph_network
 403       except AttributeError:
 404         raise RuntimeError('It looks like you are subclassing `Model` and you '
 405                            'forgot to call `super(YourClass, self).__init__()`.'
 406                            ' Always start with this line.')
 407       if not is_graph_network:
 408         # We need to check object identity to avoid de-duplicating empty
 409         # container types which compare equal.
 410         if not any((layer is value for layer in self._layers)):
 411           self._layers.append(value)
 412           if hasattr(value, '_use_resource_variables'):
 413             # In subclassed models, legacy layers (tf.layers) must always use
 414             # resource variables.
 415             value._use_resource_variables = True
 416     if (not no_dependency
 417         and isinstance(value, checkpointable.CheckpointableBase)):
 418       if (  # For subclassed models only, users may add extra weights/variables
 419             # simply by assigning them to attributes.
 420           not self._is_graph_network
 421           and isinstance(value, variables.Variable)):
 422         self._extra_variables.append(value)
 423     super(Network, self).__setattr__(name, value)

 475     """Retrieves the weights of the model.

 485   def set_weights(self, weights):
 486     """Sets the weights of the model.
 487 
 488     Arguments:
 489         weights: A list of Numpy arrays with shapes and types matching
 490             the output of `model.get_weights()`.
 491     """
 492     tuples = []
 493     for layer in self.layers:
 494       num_param = len(layer.weights)
 495       layer_weights = weights[:num_param]
 496       for sw, w in zip(layer.weights, layer_weights):
 497         tuples.append((sw, w))
 498       weights = weights[num_param:]
 499     backend.batch_set_value(tuples)

 501   def compute_mask(self, inputs, mask):
 502     if not self._is_graph_network:
 503       return None
 504 
 505     inputs = generic_utils.to_list(inputs)

 509       masks = generic_utils.to_list(mask)

 511     _, output_masks = self._run_internal_graph(inputs, mask=masks)

 519   def get_layer(self, name=None, index=None):
 520     """Retrieves a layer based on either its name (unique) or index.
 521 
 522     If `name` and `index` are both provided, `index` will take precedence.
 523     Indices are based on order of horizontal graph traversal (bottom-up).
 524 
 525     Arguments:
 526         name: String, name of layer.
 527         index: Integer, index of layer.
 528 
 529     Returns:
 530         A layer instance.
 531 
 532     Raises:
 533         ValueError: In case of invalid layer name or index.
 534     """
 535     # TODO(fchollet): We could build a dictionary based on layer names
 536     # since they are constant, but we have not done that yet.
 537     if index is not None:
 538       if len(self.layers) <= index:
 539         raise ValueError('Was asked to retrieve layer at index ' + str(index) +
 540                          ' but model only has ' + str(len(self.layers)) +
 541                          ' layers.')
 542       else:
 543         return self.layers[index]
 544     else:
 545       if not name:
 546         raise ValueError('Provide either a layer name or layer index.')
 547     for layer in self.layers:
 548       if layer.name == name:
 549         return layer
 550     raise ValueError('No such layer: ' + name)

 574   @property
 575   def updates(self):
 576     """Retrieves the network's updates.
 577 
 578     Will only include updates that are either
 579     unconditional, or conditional on inputs to this model
 580     (e.g. will not include updates that were created by layers of this model
 581     outside of the model).
 582 
 583     When the network has no registered inputs, all updates are returned.
 584 
 585     Effectively, `network.updates` behaves like `layer.updates`.
 586 
 587     Concrete example:
 588 
 589     ```python
 590       bn = keras.layers.BatchNormalization()
 591       x1 = keras.layers.Input(shape=(10,))
 592       _ = bn(x1)  # This creates 2 updates.
 593 
 594       x2 = keras.layers.Input(shape=(10,))
 595       y2 = bn(x2)  # This creates 2 more updates.
 596 
 597       # The BN layer has now 4 updates.
 598       self.assertEqual(len(bn.updates), 4)
 599 
 600       # Let's create a model from x2 to y2.
 601       model = keras.models.Model(x2, y2)
 602 
 603       # The model does not list all updates from its underlying layers,
 604       # but only the updates that are relevant to it. Updates created by layers
 605       # outside of the model are discarded.
 606       self.assertEqual(len(model.updates), 2)
 607 
 608       # If you keep calling the model, you append to its updates, just like
 609       # what happens for a layer.
 610       x3 = keras.layers.Input(shape=(10,))
 611       y3 = model(x3)
 612       self.assertEqual(len(model.updates), 4)
 613 
 614       # But if you call the inner BN layer independently, you don't affect
 615       # the model's updates.
 616       x4 = keras.layers.Input(shape=(10,))
 617       _ = bn(x4)
 618       self.assertEqual(len(model.updates), 4)
 619     ```
 620 
 621     Returns:
 622         A list of update ops.
 623     """
 624     if context.executing_eagerly():
 625       return []
 626 
 627     if not self.trainable and not self.stateful:
 628       return []
 629 
 630     updates = self._unfiltered_updates
 631 
 632     # `updates` might contain irrelevant updates, so it needs to be filtered
 633     # with respect to inputs the model has been called on.
 634     relevant_inputs = []
 635     for i in range(0, len(self._inbound_nodes)):
 636       inputs = self.get_input_at(i)
 637       if isinstance(inputs, list):
 638         relevant_inputs += inputs
 639       else:
 640         relevant_inputs.append(inputs)
 641     if not relevant_inputs:
 642       return updates
 643 
 644     reachable = tf_utils.get_reachable_from_inputs(relevant_inputs, updates)
 645     relevant_conditional_updates = [x for x in updates if x in reachable]
 646     unconditional_updates = [
 647         x for x in updates if x._unconditional_update]  # pylint: disable=protected-access
 648     # A layer could be used multiple times in a nested structure,
 649     # so the updates list must be de-duped.
 650     return list(set(
 651         relevant_conditional_updates + unconditional_updates + self._updates))

 729   @base_layer.default
 730   def build(self, input_shape):
 731     """Builds the model based on input shapes received.
 732 
 733     This is to be used for subclassed models, which do not know at instantiation
 734     time what their inputs look like.
 735 
 736     Args:
 737      input_shape: Single tuple, TensorShape, or list of shapes, where shapes
 738          are tuples, integers, or TensorShapes.
 739 
 740     Raises:
 741       ValueError:
 742         1. In case of invalid user-provided data (not of type tuple,
 743            list, or TensorShape).
 744         2. If the model requires call arguments that are agnostic
 745            to the input shapes (positional or kwarg in call signature).
 746         3. If not all layers were properly built.
 747         4. If float type inputs are not supported within the layers.
 748 
 749       In each of these cases, the user should build their model by calling it
 750       on real tensor data.
 751     """
 752     if self._is_graph_network:
 753       self.built = True
 754       return
 755 
 756     # If subclass network
 757     if input_shape is None:
 758       raise ValueError('Input shape must be defined when calling build on a '
 759                        'model subclass network.')
 760     valid_types = (tuple, list, tensor_shape.TensorShape)
 761     if not isinstance(input_shape, valid_types):
 762       raise ValueError('Specified input shape is not one of the valid types. '
 763                        'Please specify a batch input shape of type tuple or '
 764                        'list of input shapes. User provided '
 765                        'input type: {}'.format(type(input_shape)))
 766 
 767     if input_shape and not self.inputs:
 768       # We create placeholders for the `None`s in the shape and build the model
 769       # in a Graph. Since tf.Variable is compatible with both eager execution
 770       # and graph building, the variables created after building the model in
 771       # a Graph are still valid when executing eagerly.
 772       with context.graph_mode():
 773         graph = eager_function.FuncGraph('graph')
 774         with graph.as_default():
 775           if isinstance(input_shape, list):
 776             x = [base_layer.generate_placeholders_from_shape(shape)
 777                  for shape in input_shape]
 778           else:
 779             x = base_layer.generate_placeholders_from_shape(input_shape)
 780 
 781           kwargs = {}
 782           num_call_args = len(tf_inspect.getfullargspec(self.call).args)
 783           if self._expects_training_arg and num_call_args == 3:
 784             # Has call signature of call(self, input, training)
 785             kwargs['training'] = False
 786           elif num_call_args > 2:
 787             # Has invalid call signature of call(self, input, *args, **kwargs)
 788             raise ValueError('Currently, you cannot build your model if it has '
 789                              'positional or keyword arguments that are not '
 790                              'inputs to the model, but are required for its '
 791                              '`call` method. Instead, in order to instantiate '
 792                              'and build your model, `call` your model on real '
 793                              'tensor data with all expected call arguments.')
 794 
 795           try:
 796             self.call(x, **kwargs)
 797           except (errors.InvalidArgumentError, TypeError):
 798             raise ValueError('You cannot build your model by calling `build` '
 799                              'if your layers do not support float type inputs. '
 800                              'Instead, in order to instantiate and build your '
 801                              'model, `call` your model on real tensor data (of '
 802                              'the correct dtype).')
 803 
 804     if self._layers:
 805       self._track_layers(self._layers)
 806     if self.layers:
 807       for layer in self.layers:
 808         if not layer.built:
 809           raise ValueError('Layer: {} was not built in your model. Calling '
 810                            '`build` manually on a subclassed model is only '
 811                            'allowed for models with a static topology. '
 812                            'In this case, you can build your model by '
 813                            'calling it on real tensor data.'.format(layer))
 814     self.built = True

 816   def call(self, inputs, training=None, mask=None):
 817     """Calls the model on new inputs.
 818 
 819     In this case `call` just reapplies
 820     all ops in the graph to the new inputs
 821     (e.g. build a new computational graph from the provided inputs).
 822 
 823     Arguments:
 824         inputs: A tensor or list of tensors.
 825         training: Boolean or boolean scalar tensor, indicating whether to run
 826           the `Network` in training mode or inference mode.
 827         mask: A mask or list of masks. A mask can be
 828             either a tensor or None (no mask).
 829 
 830     Returns:
 831         A tensor if there is a single output, or
 832         a list of tensors if there are more than one outputs.
 833     """
 834     if not self._is_graph_network:
 835       raise NotImplementedError('When subclassing the `Model` class, you should'
 836                                 ' implement a `call` method.')
 837 
 838     inputs = generic_utils.to_list(inputs)
 839     if mask is None:
 840       masks = [None for _ in range(len(inputs))]
 841     else:
 842       masks = generic_utils.to_list(mask)
 843     outputs, _ = self._run_internal_graph(inputs,
 844                                           training=training,
 845                                           mask=masks)
 846     return outputs

 858   def compute_output_shape(self, input_shape):
 859     if not self._is_graph_network:
 860       if context.executing_eagerly():
 861         return super(Network, self).compute_output_shape(input_shape)
 862       raise NotImplementedError
 863 
 864     if isinstance(input_shape, list):
 865       input_shapes = []
 866       for shape in input_shape:
 867         if shape is not None:
 868           input_shapes.append(tuple(tensor_shape.TensorShape(shape).as_list()))
 869         else:
 870           input_shapes.append(None)
 871     else:
 872       if input_shape is not None:
 873         input_shapes = [tuple(tensor_shape.TensorShape(input_shape).as_list())]
 874       else:
 875         input_shapes = [None]
 876 
 877     if len(input_shapes) != len(self._input_layers):
 878       raise ValueError('Invalid input_shape argument ' + str(input_shape) +
 879                        ': model has ' + str(len(self._input_layers)) +
 880                        ' tensor inputs.')
 881 
 882     cache_key = generic_utils.object_list_uid(input_shapes)
 883     if cache_key in self._output_shape_cache:
 884       # Cache hit.
 885       output_shapes = self._output_shape_cache[cache_key]
 886     else:
 887       layers_to_output_shapes = {}
 888       for i in range(len(input_shapes)):
 889         layer = self._input_layers[i]
 890         input_shape = input_shapes[i]
 891         # It's an input layer: then `compute_output_shape` is identity,
 892         # and there is only one node and one tensor output.
 893         shape_key = layer.name + '_0_0'
 894         layers_to_output_shapes[shape_key] = input_shape
 895 
 896       depth_keys = list(self._nodes_by_depth.keys())
 897       depth_keys.sort(reverse=True)
 898       # Iterate over nodes, by depth level.
 899       if len(depth_keys) > 1:
 900         for depth in depth_keys:
 901           nodes = self._nodes_by_depth[depth]
 902           for node in nodes:
 903             # This is always a single layer, never a list.
 904             layer = node.outbound_layer
 905             if layer in self._input_layers:
 906               # We've already covered the input layers
 907               # a few lines above.
 908               continue
 909             # Potentially redundant list,
 910             # same size as node.input_tensors.
 911             input_shapes = []
 912             for j in range(len(node.inbound_layers)):
 913               inbound_layer = node.inbound_layers[j]
 914               node_index = node.node_indices[j]
 915               tensor_index = node.tensor_indices[j]
 916               shape_key = inbound_layer.name + '_%s_%s' % (node_index,
 917                                                            tensor_index)
 918               input_shape = layers_to_output_shapes[shape_key]
 919               input_shapes.append(input_shape)
 920 
 921             if len(input_shapes) == 1:
 922               output_shape = layer.compute_output_shape(input_shapes[0])
 923             else:
 924               output_shape = layer.compute_output_shape(input_shapes)
 925             if isinstance(output_shape, list):
 926               output_shapes = [
 927                   tuple(tensor_shape.TensorShape(shape).as_list())
 928                   for shape in output_shape
 929               ]
 930             else:
 931               output_shapes = [
 932                   tuple(tensor_shape.TensorShape(output_shape).as_list())
 933               ]
 934 
 935             node_index = layer._inbound_nodes.index(node)  # pylint: disable=protected-access
 936             for j in range(len(output_shapes)):
 937               shape_key = layer.name + '_%s_%s' % (node_index, j)
 938               layers_to_output_shapes[shape_key] = output_shapes[j]
 939 
 940         # Read final output shapes from layers_to_output_shapes.
 941         output_shapes = []
 942         for i in range(len(self._output_layers)):
 943           layer, node_index, tensor_index = self._output_coordinates[i]
 944           shape_key = layer.name + '_%s_%s' % (node_index, tensor_index)
 945           output_shapes.append(layers_to_output_shapes[shape_key])
 946         # Store in cache.
 947         self._output_shape_cache[cache_key] = output_shapes
 948 
 949     if isinstance(output_shapes, list):
 950       if len(output_shapes) == 1:
 951         return tensor_shape.TensorShape(output_shapes[0])
 952       else:
 953         return [tensor_shape.TensorShape(shape) for shape in output_shapes]
 954     else:
 955       return tensor_shape.TensorShape(output_shapes)

 957   def _run_internal_graph(self, inputs, training=None, mask=None):
 958     """Computes output tensors for new inputs.
 959 
 960     # Note:
 961         - Expects `inputs` to be a list (potentially with 1 element).
 962         - Can be run on non-Keras tensors.
 963 
 964     Arguments:
 965         inputs: List of tensors
 966         training: Boolean learning phase.
 967         mask: List of masks (tensors or None).
 968 
 969     Returns:
 970         Two lists: output_tensors, output_masks
 971     """
 972     # Note: masking support is relevant mainly for Keras.
 973     # It cannot be factored out without having the fully reimplement the network
 974     # calling logic on the Keras side. We choose to incorporate it in
 975     # Network because 1) it may be useful to fully support in tf.layers in
 976     # the future and 2) Keras is a major user of Network.  If you don't
 977     # use masking, it does not interfere with regular behavior at all and you
 978     # can ignore it.
 979     if mask is None:
 980       masks = [None for _ in range(len(inputs))]
 981     else:
 982       masks = mask
 983 
 984     # Dictionary mapping reference tensors to tuples
 985     # (computed tensor, compute mask)
 986     # we assume a 1:1 mapping from tensor to mask
 987     tensor_map = {}
 988     for x, y, mask in zip(self.inputs, inputs, masks):
 989       tensor_map[str(id(x))] = (y, mask)
 990 
 991     depth_keys = list(self._nodes_by_depth.keys())
 992     depth_keys.sort(reverse=True)
 993     for depth in depth_keys:
 994       nodes = self._nodes_by_depth[depth]
 995       for node in nodes:
 996         # This is always a single layer, never a list.
 997         layer = node.outbound_layer
 998         reference_input_tensors = node.input_tensors
 999         reference_output_tensors = node.output_tensors
1000 
1001         # If all previous input tensors are available in tensor_map,
1002         # then call node.inbound_layer on them.
1003         computed_data = []  # List of tuples (input, mask).
1004         for x in reference_input_tensors:
1005           if str(id(x)) in tensor_map:
1006             computed_data.append(tensor_map[str(id(x))])
1007 
1008         if len(computed_data) == len(reference_input_tensors):
1009           # Call layer (reapplying ops to new inputs).
1010           with ops.name_scope(layer.name):
1011             if node.arguments:
1012               kwargs = node.arguments
1013             else:
1014               kwargs = {}
1015             # Ensure `training` arg propagation if applicable.
1016             if 'training' in tf_inspect.getfullargspec(layer.call).args:
1017               kwargs.setdefault('training', training)
1018 
1019             if len(computed_data) == 1:
1020               computed_tensor, computed_mask = computed_data[0]
1021               # Ensure mask propagation if applicable.
1022               if 'mask' in tf_inspect.getfullargspec(layer.call).args:
1023                 kwargs.setdefault('mask', computed_mask)
1024 
1025               # Compute outputs and masks.
1026               if (isinstance(layer, Network) and
1027                   layer._compute_output_and_mask_jointly):
1028                 output_tensors, output_masks = layer._call_and_compute_mask(
1029                     computed_tensor, **kwargs)
1030               else:
1031                 output_tensors = layer.call(computed_tensor, **kwargs)
1032                 if hasattr(layer, 'compute_mask'):
1033                   output_masks = layer.compute_mask(computed_tensor,
1034                                                     computed_mask)
1035                 else:
1036                   output_masks = [None for _ in output_tensors]
1037               computed_tensors = [computed_tensor]
1038 
1039             else:
1040               computed_tensors = [x[0] for x in computed_data]
1041               computed_masks = [x[1] for x in computed_data]
1042               # Ensure mask propagation if applicable.
1043               if 'mask' in tf_inspect.getfullargspec(layer.call).args:
1044                 kwargs.setdefault('mask', computed_masks)
1045 
1046               # Compute outputs and masks.
1047               if (isinstance(layer, Network) and
1048                   layer._compute_output_and_mask_jointly):
1049                 output_tensors, output_masks = layer._call_and_compute_mask(
1050                     computed_tensors, **kwargs)
1051               else:
1052                 output_tensors = layer.call(computed_tensors, **kwargs)
1053                 if hasattr(layer, 'compute_mask'):
1054                   output_masks = layer.compute_mask(computed_tensors,
1055                                                     computed_masks)
1056                 else:
1057                   output_masks = [None for _ in output_tensors]
1058 
1059             output_tensors = generic_utils.to_list(output_tensors)
1060             if output_masks is None:
1061               output_masks = [None for _ in output_tensors]
1062             else:
1063               output_masks = generic_utils.to_list(output_masks)
1064 
1065             if not context.executing_eagerly():
1066               # Set mask metadata.
1067               for x, m in zip(output_tensors, output_masks):
1068                 try:
1069                   x._keras_mask = m
1070                 except AttributeError:
1071                   pass
1072 
1073               # Apply activity regularizer if any.
1074               if layer.activity_regularizer is not None:
1075                 regularization_losses = [
1076                     layer.activity_regularizer(x) for x in output_tensors
1077                 ]
1078                 layer.add_loss(regularization_losses, computed_tensors)
1079 
1080           # Update tensor_map.
1081           for x, y, mask in zip(reference_output_tensors, output_tensors,
1082                                 output_masks):
1083             tensor_map[str(id(x))] = (y, mask)
1084 
1085     output_tensors = []
1086     output_masks = []
1087     output_shapes = []
1088     for x in self.outputs:
1089       assert str(id(x)) in tensor_map, 'Could not compute output ' + str(x)
1090       tensor, mask = tensor_map[str(id(x))]
1091       output_shapes.append(backend.int_shape(x))
1092       output_tensors.append(tensor)
1093       output_masks.append(mask)
1094 
1095     if len(output_tensors) == 1:
1096       output_tensors = output_tensors[0]
1097       if output_shapes is not None:
1098         output_shapes = output_shapes[0]
1099       if output_masks is not None:
1100         output_masks = output_masks[0]
1101 
1102     if output_shapes is not None:
1103       input_shapes = [backend.int_shape(x) for x in inputs]
1104       cache_key = generic_utils.object_list_uid(input_shapes)
1105       self._output_shape_cache[cache_key] = output_shapes
1106 
1107     return output_tensors, output_masks

1109   def get_config(self):
1110     if not self._is_graph_network:
1111       raise NotImplementedError
1112 
1113     config = {
1114         'name': self.name,
1115     }
1116     node_conversion_map = {}
1117     for layer in self.layers:
1118       if issubclass(layer.__class__, Network):
1119         # Networks start with a pre-existing node
1120         # linking their input to output.
1121         kept_nodes = 1
1122       else:
1123         kept_nodes = 0
1124       for original_node_index, node in enumerate(layer._inbound_nodes):
1125         node_key = _make_node_key(layer.name, original_node_index)
1126         if node_key in self._network_nodes:
1127           node_conversion_map[node_key] = kept_nodes
1128           kept_nodes += 1

1130     for layer in self.layers:  # From the earliest layers on.
1131       layer_class_name = layer.__class__.__name__
1132       layer_config = layer.get_config()

1134       for original_node_index, node in enumerate(layer._inbound_nodes):
1135         node_key = _make_node_key(layer.name, original_node_index)

1143             except TypeError:

1147                   str(node.arguments) + '. They will not be included '
1148                   'in the serialized model (and thus will be missing '
1149                   'at deserialization time).')
1150               kwargs = {}
1151           else:
1152             kwargs = {}

1154             node_data = []
1155             for i in range(len(node.inbound_layers)):
1156               inbound_layer = node.inbound_layers[i]

1162                   [inbound_layer.name, new_node_index, tensor_index, kwargs])
1163             filtered_inbound_nodes.append(node_data)

1165           'name': layer.name,

1169       })

1171

1173     model_inputs = []
1174     for i in range(len(self._input_layers)):
1175       layer, node_index, tensor_index = self._input_coordinates[i]
1176       node_key = _make_node_key(layer.name, node_index)
1177       if node_key not in self._network_nodes:
1178         continue
1179       new_node_index = node_conversion_map[node_key]
1180       model_inputs.append([layer.name, new_node_index, tensor_index])
1181     config['input_layers'] = model_inputs
1182     model_outputs = []
1183     for i in range(len(self._output_layers)):
1184       layer, node_index, tensor_index = self._output_coordinates[i]
1185       node_key = _make_node_key(layer.name, node_index)
1186       if node_key not in self._network_nodes:
1187         continue
1188       new_node_index = node_conversion_map[node_key]
1189       model_outputs.append([layer.name, new_node_index, tensor_index])
1190     config['output_layers'] = model_outputs

1193   @classmethod
1194   def from_config(cls, config, custom_objects=None):
1195     """Instantiates a Model from its config (output of `get_config()`).
1196 
1197     Arguments:
1198         config: Model config dictionary.
1199         custom_objects: Optional dictionary mapping names
1200             (strings) to custom classes or functions to be
1201             considered during deserialization.
1202 
1203     Returns:
1204         A model instance.
1205 
1206     Raises:
1207         ValueError: In case of improperly formatted config dict.
1208     """
1209     # Layer instances created during
1210     # the graph reconstruction process
1211     created_layers = {}
1212 
1213     # Dictionary mapping layer instances to
1214     # node data that specifies a layer call.
1215     # It acts as a queue that maintains any unprocessed
1216     # layer call until it becomes possible to process it
1217     # (i.e. until the input tensors to the call all exist).
1218     unprocessed_nodes = {}
1219 
1220     def add_unprocessed_node(layer, node_data):
1221       if layer not in unprocessed_nodes:
1222         unprocessed_nodes[layer] = [node_data]
1223       else:
1224         unprocessed_nodes[layer].append(node_data)
1225 
1226     def process_node(layer, node_data):
1227       """Deserialize a node.
1228 
1229       Arguments:
1230           layer: layer instance.
1231           node_data: node config dict.
1232 
1233       Raises:
1234           ValueError: In case of improperly formatted `node_data` dict.
1235       """
1236       input_tensors = []
1237       for input_data in node_data:
1238         inbound_layer_name = input_data[0]
1239         inbound_node_index = input_data[1]
1240         inbound_tensor_index = input_data[2]
1241         if len(input_data) == 3:
1242           kwargs = {}
1243         elif len(input_data) == 4:
1244           kwargs = input_data[3]
1245         else:
1246           raise ValueError('Improperly formatted model config.')
1247         if inbound_layer_name not in created_layers:
1248           add_unprocessed_node(layer, node_data)
1249           return
1250         inbound_layer = created_layers[inbound_layer_name]
1251         if len(inbound_layer._inbound_nodes) <= inbound_node_index:
1252           add_unprocessed_node(layer, node_data)
1253           return
1254         inbound_node = inbound_layer._inbound_nodes[inbound_node_index]
1255         input_tensors.append(inbound_node.output_tensors[inbound_tensor_index])
1256       # Call layer on its inputs, thus creating the node
1257       # and building the layer if needed.
1258       if input_tensors:
1259         if len(input_tensors) == 1:
1260           layer(input_tensors[0], **kwargs)
1261         else:
1262           layer(input_tensors, **kwargs)
1263 
1264     def process_layer(layer_data):
1265       """Deserializes a layer, then call it on appropriate inputs.
1266 
1267       Arguments:
1268           layer_data: layer config dict.
1269 
1270       Raises:
1271           ValueError: In case of improperly formatted `layer_data` dict.
1272       """
1273       layer_name = layer_data['name']
1274 
1275       # Instantiate layer.
1276       from tensorflow.python.keras.layers import deserialize as deserialize_layer  # pylint: disable=g-import-not-at-top
1277 
1278       layer = deserialize_layer(layer_data, custom_objects=custom_objects)
1279       created_layers[layer_name] = layer
1280 
1281       # Gather layer inputs.
1282       inbound_nodes_data = layer_data['inbound_nodes']
1283       for node_data in inbound_nodes_data:
1284         # We don't process nodes (i.e. make layer calls)
1285         # on the fly because the inbound node may not yet exist,
1286         # in case of layer shared at different topological depths
1287         # (e.g. a model such as A(B(A(B(x)))))
1288         add_unprocessed_node(layer, node_data)
1289 
1290     # First, we create all layers and enqueue nodes to be processed
1291     for layer_data in config['layers']:
1292       process_layer(layer_data)
1293     # Then we process nodes in order of layer depth.
1294     # Nodes that cannot yet be processed (if the inbound node
1295     # does not yet exist) are re-enqueued, and the process
1296     # is repeated until all nodes are processed.
1297     while unprocessed_nodes:
1298       for layer_data in config['layers']:
1299         layer = created_layers[layer_data['name']]
1300         if layer in unprocessed_nodes:
1301           for node_data in unprocessed_nodes.pop(layer):
1302             process_node(layer, node_data)
1303 
1304     name = config.get('name')
1305     input_tensors = []
1306     output_tensors = []
1307     for layer_data in config['input_layers']:
1308       layer_name, node_index, tensor_index = layer_data
1309       assert layer_name in created_layers
1310       layer = created_layers[layer_name]
1311       layer_output_tensors = layer._inbound_nodes[node_index].output_tensors
1312       input_tensors.append(layer_output_tensors[tensor_index])
1313     for layer_data in config['output_layers']:
1314       layer_name, node_index, tensor_index = layer_data
1315       assert layer_name in created_layers
1316       layer = created_layers[layer_name]
1317       layer_output_tensors = layer._inbound_nodes[node_index].output_tensors
1318       output_tensors.append(layer_output_tensors[tensor_index])
1319     return cls(inputs=input_tensors, outputs=output_tensors, name=name)

1365   def save_weights(self, filepath, overwrite=True, save_format=None):

1367 
1368     Either saves in HDF5 or in TensorFlow format based on the `save_format`
1369     argument.
1370 
1371     When saving in HDF5 format, the weight file has:
1372       - `layer_names` (attribute), a list of strings
1373           (ordered names of model layers).
1374       - For every layer, a `group` named `layer.name`
1375           - For every such layer group, a group attribute `weight_names`,
1376               a list of strings

1378           - For every weight in the layer, a dataset
1379               storing the weight value, named after the weight tensor.
1380 
1381     When saving in TensorFlow format, all objects referenced by the network are
1382     saved in the same format as `tf.train.Checkpoint`, including any `Layer`
1383     instances or `Optimizer` instances assigned to object attributes. For
1384     networks constructed from inputs and outputs using `tf.keras.Model(inputs,
1385     outputs)`, `Layer` instances used by the network are tracked/saved
1386     automatically. For user-defined classes which inherit from `tf.keras.Model`,
1387     `Layer` instances must be assigned to object attributes, typically in the
1388     constructor. See the documentation of `tf.train.Checkpoint` and
1389     `tf.keras.Model` for details.
1390 
1391     Arguments:
1392         filepath: String, path to the file to save the weights to. When saving
1393             in TensorFlow format, this is the prefix used for checkpoint files
1394             (multiple files are generated). Note that the '.h5' suffix causes
1395             weights to be saved in HDF5 format.
1396         overwrite: Whether to silently overwrite any existing file at the
1397             target location, or provide the user with a manual prompt.
1398         save_format: Either 'tf' or 'h5'. A `filepath` ending in '.h5' or
1399             '.keras' will default to HDF5 if `save_format` is `None`. Otherwise
1400             `None` defaults to 'tf'.
1401 
1402     Raises:
1403         ImportError: If h5py is not available when attempting to save in HDF5

1405         ValueError: For invalid/unknown format arguments.

1407     filepath_is_h5 = _is_hdf5_filepath(filepath)
1408     if save_format is None:
1409       if filepath_is_h5:
1410         save_format = 'h5'
1411       else:
1412         save_format = 'tf'
1413     else:
1414       user_format = save_format.lower().strip()
1415       if user_format in ('tensorflow', 'tf'):
1416         save_format = 'tf'
1417       elif user_format in ('hdf5', 'h5', 'keras'):
1418         save_format = 'h5'
1419       else:
1420         raise ValueError(
1421             'Unknown format "%s". Was expecting one of {"tf", "h5"}.' % (
1422                 save_format,))
1423     if save_format == 'tf' and filepath_is_h5:
1424       raise ValueError(
1425           ('save_weights got save_format="tf"/"tensorflow", but the '
1426            'filepath ("%s") looks like an HDF5 file. Omit the ".h5"/".keras" '
1427            'when saving in TensorFlow format.')
1428           % filepath)
1429 
1430     if save_format == 'h5' and h5py is None:
1431       raise ImportError(
1432           '`save_weights` requires h5py when saving in hdf5.')
1433     if save_format == 'tf':
1434       check_filepath = filepath + '.index'
1435     else:
1436       check_filepath = filepath
1437     # If file exists and should not be overwritten:
1438     if not overwrite and os.path.isfile(check_filepath):
1439       proceed = ask_to_proceed_with_overwrite(check_filepath)
1440       if not proceed:
1441         return

1445     else:
1446       if context.executing_eagerly():

1448       else:

1461       # Record this checkpoint so it's visible from tf.train.latest_checkpoint.

1467   def load_weights(self, filepath, by_name=False):
1468     """Loads all layer weights, either from a TensorFlow or an HDF5 weight file.
1469 
1470     If `by_name` is False weights are loaded based on the network's
1471     topology. This means the architecture should be the same as when the weights
1472     were saved.  Note that layers that don't have weights are not taken into
1473     account in the topological ordering, so adding or removing layers is fine as
1474     long as they don't have weights.
1475 
1476     If `by_name` is True, weights are loaded into layers only if they share the
1477     same name. This is useful for fine-tuning or transfer-learning models where
1478     some of the layers have changed.
1479 
1480     Only topological loading (`by_name=False`) is supported when loading weights
1481     from the TensorFlow format. Note that topological loading differs slightly
1482     between TensorFlow and HDF5 formats for user-defined classes inheriting from
1483     `tf.keras.Model`: HDF5 loads based on a flattened list of weights, while the
1484     TensorFlow format loads based on the object-local names of attributes to
1485     which layers are assigned in the `Model`'s constructor.
1486 
1487     Arguments:
1488         filepath: String, path to the weights file to load. For weight files in
1489             TensorFlow format, this is the file prefix (the same as was passed
1490             to `save_weights`).
1491         by_name: Boolean, whether to load weights by name or by topological
1492             order. Only topological loading is supported for weight files in
1493             TensorFlow format.
1494 
1495     Returns:
1496         When loading a weight file in TensorFlow format, returns the same status
1497         object as `tf.train.Checkpoint.restore`. When graph building, restore
1498         ops are run automatically as soon as the network is built (on first call
1499         for user-defined classes inheriting from `Model`, immediately if it is
1500         already built).
1501 
1502         When loading weights in HDF5 format, returns `None`.
1503 
1504     Raises:
1505         ImportError: If h5py is not available and the weight file is in HDF5
1506             format.
1507     """
1508     if _is_hdf5_filepath(filepath):
1509       save_format = 'h5'
1510     else:
1511       try:
1512         pywrap_tensorflow.NewCheckpointReader(filepath)
1513         save_format = 'tf'
1514       except errors_impl.DataLossError:
1515         # The checkpoint is not readable in TensorFlow format. Try HDF5.
1516         save_format = 'h5'
1517     if save_format == 'tf':
1518       status = self._checkpointable_saver.restore(filepath)
1519       if by_name:
1520         raise NotImplementedError(
1521             'Weights may only be loaded based on topology into Models when '
1522             'loading TensorFlow-formatted weights (got by_name=True to '
1523             'load_weights).')
1524       if not context.executing_eagerly():
1525         session = backend.get_session()
1526         # Restore existing variables (if any) immediately, and set up a
1527         # streaming restore for any variables created in the future.
1528         checkpointable_utils.streaming_restore(status=status, session=session)
1529       return status
1530     if h5py is None:
1531       raise ImportError(
1532           '`load_weights` requires h5py when loading weights from HDF5.')
1533     if self._is_graph_network and not self.built:
1534       raise NotImplementedError(
1535           'Unable to load weights saved in HDF5 format into a subclassed '
1536           'Model which has not created its variables yet. Call the Model '
1537           'first, then load the weights.')
1538     with h5py.File(filepath, 'r') as f:
1539       if 'layer_names' not in f.attrs and 'model_weights' in f:
1540         f = f['model_weights']
1541       if by_name:
1542         saving.load_weights_from_hdf5_group_by_name(f, self.layers)
1543       else:
1544         saving.load_weights_from_hdf5_group(f, self.layers)

1655 def _map_graph_network(inputs, outputs):
1656   """Validates a network's topology and gather its layers and nodes.
1657 
1658   Arguments:
1659     inputs: List of input tensors.
1660     outputs: List of outputs tensors.
1661 
1662   Returns:
1663     A tuple `(nodes, nodes_by_depth, layers, layers_by_depth)`.
1664     - nodes: list of Node instances.
1665     - nodes_by_depth: dict mapping ints (depth) to lists of node instances.
1666     - layers: list of Layer instances.
1667     - layers_by_depth: dict mapping ints (depth) to lists of layer instances.
1668 
1669   Raises:
1670     ValueError: In case the network is not valid (e.g. disconnected graph).
1671   """
1672   # Network_nodes: set of nodes included in the graph of layers
1673   # (not all nodes included in the layers are relevant to the current graph).
1674   network_nodes = set()  # ids of all nodes relevant to the Network
1675   nodes_depths = {}  # dict {node: depth value}
1676   layers_depths = {}  # dict {layer: depth value}
1677   layer_indices = {}  # dict {layer: index in traversal}
1678   nodes_in_decreasing_depth = []
1679 
1680   def build_map(tensor,
1681                 finished_nodes,
1682                 nodes_in_progress,
1683                 layer,
1684                 node_index,
1685                 tensor_index):
1686     """Builds a map of the graph of layers.
1687 
1688     This recursively updates the map `layer_indices`,
1689     the list `nodes_in_decreasing_depth` and the set `network_nodes`.
1690 
1691     Arguments:
1692         tensor: Some tensor in a graph.
1693         finished_nodes: Set of nodes whose subgraphs have been traversed
1694             completely. Useful to prevent duplicated work.
1695         nodes_in_progress: Set of nodes that are currently active on the
1696             recursion stack. Useful to detect cycles.
1697         layer: Layer from which `tensor` comes from. If not provided,
1698             will be obtained from `tensor._keras_history`.
1699         node_index: Node index from which `tensor` comes from.
1700         tensor_index: Tensor_index from which `tensor` comes from.
1701 
1702     Raises:
1703         ValueError: if a cycle is detected.
1704     """
1705     node = layer._inbound_nodes[node_index]  # pylint: disable=protected-access
1706 
1707     # Prevent cycles.
1708     if node in nodes_in_progress:
1709       raise ValueError('The tensor ' + str(tensor) + ' at layer "' +
1710                        layer.name + '" is part of a cycle.')
1711 
1712     # Don't repeat work for shared subgraphs
1713     if node in finished_nodes:
1714       return
1715 
1716     node_key = _make_node_key(layer.name, node_index)
1717     # Update network_nodes.
1718     network_nodes.add(node_key)
1719 
1720     # Store the traversal order for layer sorting.
1721     if layer not in layer_indices:
1722       layer_indices[layer] = len(layer_indices)
1723 
1724     nodes_in_progress.add(node)
1725 
1726     # Propagate to all previous tensors connected to this node.
1727     for i in range(len(node.inbound_layers)):
1728       x = node.input_tensors[i]
1729       layer = node.inbound_layers[i]
1730       node_index = node.node_indices[i]
1731       tensor_index = node.tensor_indices[i]
1732       build_map(x, finished_nodes, nodes_in_progress, layer,
1733                 node_index, tensor_index)
1734 
1735     finished_nodes.add(node)
1736     nodes_in_progress.remove(node)
1737     nodes_in_decreasing_depth.append(node)
1738 
1739   finished_nodes = set()
1740   nodes_in_progress = set()
1741   for x in outputs:
1742     layer, node_index, tensor_index = x._keras_history  # pylint: disable=protected-access
1743     build_map(x, finished_nodes, nodes_in_progress,
1744               layer=layer,
1745               node_index=node_index,
1746               tensor_index=tensor_index)
1747 
1748   for node in reversed(nodes_in_decreasing_depth):
1749     # If the depth is not set, the node has no outbound nodes (depth 0).
1750     depth = nodes_depths.setdefault(node, 0)
1751 
1752     # Update the depth of the corresponding layer
1753     previous_depth = layers_depths.get(node.outbound_layer, 0)
1754     # If we've seen this layer before at a higher depth,
1755     # we should use that depth instead of the node depth.
1756     # This is necessary for shared layers that have inputs at different
1757     # depth levels in the graph.
1758     depth = max(depth, previous_depth)
1759     layers_depths[node.outbound_layer] = depth
1760     nodes_depths[node] = depth
1761 
1762     # Update the depth of inbound nodes.
1763     # The "depth" of a node is the max of the depths
1764     # of all layers it is connected to.
1765     for i in range(len(node.inbound_layers)):
1766       inbound_layer = node.inbound_layers[i]
1767       node_index = node.node_indices[i]
1768       inbound_node = inbound_layer._inbound_nodes[node_index]  # pylint: disable=protected-access
1769       previous_depth = nodes_depths.get(inbound_node, 0)
1770       nodes_depths[inbound_node] = max(depth + 1, previous_depth)
1771 
1772   # Build a dict {depth: list of nodes with this depth}
1773   nodes_by_depth = {}
1774   for node, depth in nodes_depths.items():
1775     if depth not in nodes_by_depth:
1776       nodes_by_depth[depth] = []
1777     nodes_by_depth[depth].append(node)
1778 
1779   # Build a dict {depth: list of layers with this depth}
1780   layers_by_depth = {}
1781   for layer, depth in layers_depths.items():
1782     if depth not in layers_by_depth:
1783       layers_by_depth[depth] = []
1784     layers_by_depth[depth].append(layer)
1785 
1786   # Get sorted list of layer depths.
1787   depth_keys = list(layers_by_depth.keys())
1788   depth_keys.sort(reverse=True)
1789 
1790   # Set self.layers and self._layers_by_depth.
1791   layers = []
1792   for depth in depth_keys:
1793     layers_for_depth = layers_by_depth[depth]
1794     # Network.layers needs to have a deterministic order:
1795     # here we order them by traversal order.
1796     layers_for_depth.sort(key=lambda x: layer_indices[x])
1797     layers.extend(layers_for_depth)
1798 
1799   # Get sorted list of node depths.
1800   depth_keys = list(nodes_by_depth.keys())
1801   depth_keys.sort(reverse=True)
1802 
1803   # Check that all tensors required are computable.
1804   # computable_tensors: all tensors in the graph
1805   # that can be computed from the inputs provided.
1806   computable_tensors = []
1807   for x in inputs:
1808     computable_tensors.append(x)
1809 
1810   layers_with_complete_input = []  # To provide a better error msg.
1811   for depth in depth_keys:
1812     for node in nodes_by_depth[depth]:
1813       layer = node.outbound_layer
1814       if layer:
1815         for x in node.input_tensors:
1816           if x not in computable_tensors:
1817             raise ValueError('Graph disconnected: '
1818                              'cannot obtain value for tensor ' + str(x) +
1819                              ' at layer "' + layer.name + '". '
1820                              'The following previous layers '
1821                              'were accessed without issue: ' +
1822                              str(layers_with_complete_input))
1823         for x in node.output_tensors:
1824           computable_tensors.append(x)
1825         layers_with_complete_input.append(layer.name)
1826 
1827   # Ensure name unicity, which will be crucial for serialization
1828   # (since serialized nodes refer to layers by their name).
1829   all_names = [layer.name for layer in layers]
1830   for name in all_names:
1831     if all_names.count(name) != 1:
1832       raise ValueError('The name "' + name + '" is used ' +
1833                        str(all_names.count(name)) + ' times in the model. '
1834                        'All layer names should be unique.')
1835   return network_nodes, nodes_by_depth, layers, layers_by_depth
