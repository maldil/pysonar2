  39 @tf_export('keras.models.Sequential', 'keras.Sequential')
  40 class Sequential(Model):
  41   """Linear stack of layers.
  42 
  43   Arguments:
  44       layers: list of layers to add to the model.
  45 
  46   Example:
  47 
  48   ```python
  49   # Optionally, the first layer can receive an `input_shape` argument:
  50   model = Sequential()
  51   model.add(Dense(32, input_shape=(500,)))
  52   # Afterwards, we do automatic shape inference:
  53   model.add(Dense(32))
  54 
  55   # This is identical to the following:
  56   model = Sequential()
  57   model.add(Dense(32, input_dim=500))
  58 
  59   # And to the following:
  60   model = Sequential()
  61   model.add(Dense(32, batch_input_shape=(None, 500)))
  62 
  63   # Note that you can also omit the `input_shape` argument:
  64   # In that case the model gets built the first time you call `fit` (or other
  65   # training and evaluation methods).
  66   model = Sequential()
  67   model.add(Dense(32))
  68   model.add(Dense(32))
  69   model.compile(optimizer=optimizer, loss=loss)
  70   # This builds the model for the first time:
  71   model.fit(x, y, batch_size=32, epochs=10)
  72 
  73   # Note that when using this delayed-build pattern (no input shape specified),
  74   # the model doesn't have any weights until the first call
  75   # to a training/evaluation method (since it isn't yet built):
  76   model = Sequential()
  77   model.add(Dense(32))
  78   model.add(Dense(32))
  79   model.weights  # returns []
  80 
  81   # Whereas if you specify the input shape, the model gets built continuously
  82   # as you are adding layers:
  83   model = Sequential()
  84   model.add(Dense(32, input_shape=(500,)))
  85   model.add(Dense(32))
  86   model.weights  # returns list of length 4
  87 
  88   When using the delayed-build pattern (no input shape specified), you can
  89   choose to manually build your model by calling `build(batch_input_shape)`:
  90   model = Sequential()
  91   model.add(Dense(32))
  92   model.add(Dense(32))
  93   model.build((None, 500))
  94   model.weights  # returns list of length 4
  95   ```
  96   """
  97 
  98   @checkpointable.no_automatic_dependency_tracking
  99   def __init__(self, layers=None, name=None):
 100     super(Sequential, self).__init__(name=name)
 101     self.supports_masking = True
 102     self._build_input_shape = None
 103     self._compute_output_and_mask_jointly = True
 104 
 105     # Add to the model any layers passed to the constructor.
 106     if layers:
 107       for layer in layers:
 108         self.add(layer)
 109 
 110   @property
 111   def layers(self):
 112     # Historically, `sequential.layers` only returns layers that were added
 113     # via `add`, and omits the auto-generated `InputLayer` that comes at the
 114     # bottom of the stack.
 115     # `CheckpointableBase` manages the `_layers` attributes and does filtering
 116     # over it.
 117     layers = super(Sequential, self).layers
 118     if layers and isinstance(layers[0], InputLayer):
 119       return layers[1:]
 120     return layers[:]
 121 
 122   @checkpointable.no_automatic_dependency_tracking
 123   def add(self, layer):
 124     """Adds a layer instance on top of the layer stack.
 125 
 126     Arguments:
 127         layer: layer instance.
 128 
 129     Raises:
 130         TypeError: If `layer` is not a layer instance.
 131         ValueError: In case the `layer` argument does not
 132             know its input shape.
 133         ValueError: In case the `layer` argument has
 134             multiple output tensors, or is already connected
 135             somewhere else (forbidden in `Sequential` models).
 136     """
 137     if not isinstance(layer, base_layer.Layer):
 138       raise TypeError('The added layer must be '
 139                       'an instance of class Layer. '
 140                       'Found: ' + str(layer))
 141     self.built = False
 142     set_inputs = False
 143     if not self._layers:
 144       if isinstance(layer, InputLayer):
 145         # Corner case where the user passes an InputLayer layer via `add`.
 146         assert len(layer._inbound_nodes[-1].output_tensors) == 1
 147         set_inputs = True
 148       else:
 149         batch_shape, dtype = get_input_shape_and_dtype(layer)
 150         if batch_shape:
 151           # Instantiate an input layer.
 152           x = Input(
 153               batch_shape=batch_shape,
 154               dtype=dtype,
 155               name=layer.name + '_input')
 156           # This will build the current layer
 157           # and create the node connecting the current layer
 158           # to the input layer we just created.
 159           layer(x)
 160           set_inputs = True
 161 
 162       if set_inputs:
 163         # If an input layer (placeholder) is available.
 164         if len(layer._inbound_nodes[-1].output_tensors) != 1:
 165           raise ValueError('All layers in a Sequential model '
 166                            'should have a single output tensor. '
 167                            'For multi-output layers, '
 168                            'use the functional API.')
 169         self.outputs = [layer._inbound_nodes[-1].output_tensors[0]]
 170         self.inputs = layer_utils.get_source_inputs(self.outputs[0])
 171 
 172     elif self.outputs:
 173       # If the model is being built continuously on top of an input layer:
 174       # refresh its output.
 175       output_tensor = layer(self.outputs[0])
 176       if isinstance(output_tensor, list):
 177         raise TypeError('All layers in a Sequential model '
 178                         'should have a single output tensor. '
 179                         'For multi-output layers, '
 180                         'use the functional API.')
 181       self.outputs = [output_tensor]
 182     if set_inputs or self._is_graph_network:
 183       self._init_graph_network(self.inputs, self.outputs, name=self.name)
 184       self.built = True
 185     else:
 186       self._layers.append(layer)
 187     if self._layers:
 188       self._track_layers(self._layers)
 189 
 190   @checkpointable.no_automatic_dependency_tracking
 191   def pop(self):
 192     """Removes the last layer in the model.
 193 
 194     Raises:
 195         TypeError: if there are no layers in the model.
 196     """
 197     if not self.layers:
 198       raise TypeError('There are no layers in the model.')
 199 
 200     self._layers.pop()
 201     if not self.layers:
 202       self.outputs = None
 203       self.inputs = None
 204       self.built = False
 205     elif self._is_graph_network:
 206       self.layers[-1]._outbound_nodes = []
 207       self.outputs = [self.layers[-1].output]
 208       self._init_graph_network(self.inputs, self.outputs, name=self.name)
 209       self.built = True
 210 
 211   def build(self, input_shape=None):
 212     if self._is_graph_network:
 213       self._init_graph_network(self.inputs, self.outputs, name=self.name)
 214     else:
 215       if input_shape is None:
 216         raise ValueError('You must provide an `input_shape` argument.')
 217       self._build_input_shape = input_shape
 218       shape = input_shape
 219       for layer in self.layers:
 220         if not layer.built:
 221           with ops.name_scope(layer._name_scope()):
 222             layer.build(shape)
 223           layer.built = True
 224         shape = layer.compute_output_shape(shape)
 225     self.built = True
 226 
 227   def call(self, inputs, training=None, mask=None):
 228     if self._is_graph_network:
 229       return super(Sequential, self).call(inputs, training=training, mask=mask)
 230 
 231     outputs, _ = self._call_and_compute_mask(
 232         inputs, training=training, mask=mask)
 233     return outputs
 234 
 235   def _call_and_compute_mask(self, inputs, training=None, mask=None):
 236     if not self.built:
 237       self.build(inputs.shape)
 238 
 239     x = inputs
 240     for layer in self.layers:
 241       kwargs = {}
 242       if 'mask' in tf_inspect.getfullargspec(layer.call).args:
 243         kwargs['mask'] = mask
 244       if 'training' in tf_inspect.getfullargspec(layer.call).args:
 245         kwargs['training'] = training
 246 
 247       if isinstance(layer, Network) and layer._compute_output_and_mask_jointly:
 248         x, mask = layer._call_and_compute_mask(x, **kwargs)
 249       else:
 250         x = layer.call(x, **kwargs)
 251         if layer.supports_masking:
 252           mask = layer.compute_mask(x, mask)
 253         else:
 254           mask = None
 255       if not context.executing_eagerly():
 256         x._keras_mask = mask
 257     return x, mask
 258 
 259   def compute_output_shape(self, input_shape):
 260     shape = input_shape
 261     for layer in self.layers:
 262       shape = layer.compute_output_shape(shape)
 263     return shape
 264 
 265   def compute_mask(self, inputs, mask):
 266     _, mask = self._call_and_compute_mask(inputs, mask=mask)
 267     return mask

 269   def predict_proba(self, x, batch_size=32, verbose=0):
 270     """Generates class probability predictions for the input samples.
 271 
 272     The input samples are processed batch by batch.
 273 
 274     Arguments:
 275         x: input data, as a Numpy array or list of Numpy arrays
 276             (if the model has multiple inputs).
 277         batch_size: integer.
 278         verbose: verbosity mode, 0 or 1.
 279 
 280     Returns:
 281         A Numpy array of probability predictions.
 282     """
 283     preds = self.predict(x, batch_size, verbose)
 284     if preds.min() < 0. or preds.max() > 1.:
 285       logging.warning('Network returning invalid probability values. '
 286                       'The last layer might not normalize predictions '
 287                       'into probabilities '
 288                       '(like softmax or sigmoid would).')
 289     return preds
 290 
 291   def predict_classes(self, x, batch_size=32, verbose=0):
 292     """Generate class predictions for the input samples.
 293 
 294     The input samples are processed batch by batch.
 295 
 296     Arguments:
 297         x: input data, as a Numpy array or list of Numpy arrays
 298             (if the model has multiple inputs).
 299         batch_size: integer.
 300         verbose: verbosity mode, 0 or 1.
 301 
 302     Returns:
 303         A numpy array of class predictions.
 304     """
 305     proba = self.predict(x, batch_size=batch_size, verbose=verbose)
 306     if proba.shape[-1] > 1:
 307       return proba.argmax(axis=-1)
 308     else:
 309       return (proba > 0.5).astype('int32')

 311   def get_config(self):
 312     layer_configs = []
 313     for layer in self.layers:
 314       layer_configs.append({
 315           'class_name': layer.__class__.__name__,
 316           'config': layer.get_config()
 317       })
 318     config = {
 319         'name': self.name,
 320         'layers': copy.deepcopy(layer_configs)
 321     }
 322     if self._build_input_shape:
 323       config['build_input_shape'] = self._build_input_shape
 324     return config
 325 
 326   @classmethod
 327   def from_config(cls, config, custom_objects=None):
 328     if 'name' in config:
 329       name = config['name']
 330       build_input_shape = config.get('build_input_shape')
 331       layer_configs = config['layers']
 332     else:
 333       name = None
 334       build_input_shape = None
 335       layer_configs = config
 336     model = cls(name=name)
 337     for layer_config in layer_configs:
 338       layer = layer_module.deserialize(layer_config,
 339                                        custom_objects=custom_objects)
 340       model.add(layer)
 341     if not model.inputs and build_input_shape:
 342       model.build(build_input_shape)
 343     return model

 346 def get_input_shape_and_dtype(layer):
 347   """Retrieve input shape and input dtype of layer if applicable.
 348 
 349   Args:
 350     layer: Layer (or model) instance.
 351 
 352   Returns:
 353     Tuple (input_shape, input_dtype). Both could be None if the layer
 354       does not have a defined input shape.
 355 
 356   Raises:
 357     ValueError: in case an empty Sequential or Graph Network is passed.
 358   """
 359   if ((isinstance(layer, Model) and layer._is_graph_network)
 360       or isinstance(layer, Sequential)):
 361     # We were passed a model as first layer.
 362     # This requires a specific way to figure out the
 363     # input shape and dtype.
 364     if not layer.layers:
 365       raise ValueError('Cannot add an empty model '
 366                        'to a `Sequential` model.')
 367     # In case of nested models: recover the first layer
 368     # of the deepest model to infer input shape and dtype.
 369     layer = layer.layers[0]
 370     while ((isinstance(layer, Model) and layer._is_graph_network)
 371            or isinstance(layer, Sequential)):
 372       layer = layer.layers[0]
 373 
 374   if hasattr(layer, '_batch_input_shape'):
 375     batch_shape = layer._batch_input_shape
 376     dtype = layer.dtype
 377     return batch_shape, dtype
 378   return None, None
