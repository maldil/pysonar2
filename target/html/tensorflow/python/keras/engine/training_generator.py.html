  33 def fit_generator(model,
  34                   generator,
  35                   steps_per_epoch=None,
  36                   epochs=1,
  37                   verbose=1,
  38                   callbacks=None,
  39                   validation_data=None,
  40                   validation_steps=None,
  41                   class_weight=None,
  42                   max_queue_size=10,
  43                   workers=1,
  44                   use_multiprocessing=False,
  45                   shuffle=True,
  46                   initial_epoch=0):
  47   """See docstring for `Model.fit_generator`."""
  48   wait_time = 0.01  # in seconds
  49   epoch = initial_epoch
  50 
  51   do_validation = bool(validation_data)
  52   if not context.executing_eagerly():
  53     model._make_train_function()
  54     if do_validation:
  55       model._make_test_function()
  56 
  57   is_sequence = isinstance(generator, Sequence)
  58   if not is_sequence and use_multiprocessing and workers > 1:
  59     logging.warning(
  60         UserWarning('Using a generator with `use_multiprocessing=True`'
  61                     ' and multiple workers may duplicate your data.'
  62                     ' Please consider using the`keras.utils.Sequence'
  63                     ' class.'))
  64   if steps_per_epoch is None:
  65     if is_sequence:
  66       steps_per_epoch = len(generator)
  67     else:
  68       raise ValueError('`steps_per_epoch=None` is only valid for a'
  69                        ' generator based on the `keras.utils.Sequence`'
  70                        ' class. Please specify `steps_per_epoch` or use'
  71                        ' the `keras.utils.Sequence` class.')
  72 
  73   # python 2 has 'next', 3 has '__next__'
  74   # avoid any explicit version checks
  75   val_gen = (
  76       hasattr(validation_data, 'next') or
  77       hasattr(validation_data, '__next__') or
  78       isinstance(validation_data, Sequence))
  79   if (val_gen and not isinstance(validation_data, Sequence) and
  80       not validation_steps):
  81     raise ValueError('`validation_steps=None` is only valid for a'
  82                      ' generator based on the `keras.utils.Sequence`'
  83                      ' class. Please specify `validation_steps` or use'
  84                      ' the `keras.utils.Sequence` class.')
  85 
  86   enqueuer = None
  87   val_enqueuer = None
  88 
  89   try:
  90     val_x, val_y, val_sample_weights = validation_data, None, None
  91     if do_validation and not val_gen:
  92       # Prepare data for validation
  93       if len(validation_data) == 2:
  94         val_x, val_y = validation_data  # pylint: disable=unpacking-non-sequence
  95         val_sample_weights = None
  96       elif len(validation_data) == 3:
  97         val_x, val_y, val_sample_weights = validation_data  # pylint: disable=unpacking-non-sequence
  98       else:
  99         raise ValueError(
 100             '`validation_data` should be a tuple '
 101             '`(val_x, val_y, val_sample_weight)` '
 102             'or `(val_x, val_y)`. Found: ' + str(validation_data))
 103       val_x, val_y, val_sample_weights = model._standardize_user_data(
 104           val_x, val_y, val_sample_weights)
 105 
 106     callbacks = cbks.configure_callbacks(
 107         callbacks,
 108         model,
 109         do_validation=do_validation,
 110         val_inputs=val_x,
 111         val_targets=val_y,
 112         val_sample_weights=val_sample_weights,
 113         epochs=epochs,
 114         validation_steps=validation_steps,
 115         steps_per_epoch=steps_per_epoch,
 116         verbose=verbose)
 117 
 118     if workers > 0:
 119       if is_sequence:
 120         enqueuer = OrderedEnqueuer(
 121             generator,
 122             use_multiprocessing=use_multiprocessing,
 123             shuffle=shuffle)
 124       else:
 125         enqueuer = GeneratorEnqueuer(
 126             generator,
 127             use_multiprocessing=use_multiprocessing,
 128             wait_time=wait_time)
 129       enqueuer.start(workers=workers, max_queue_size=max_queue_size)
 130       output_generator = enqueuer.get()
 131     else:
 132       if is_sequence:
 133         output_generator = iter(generator)
 134       else:
 135         output_generator = generator
 136 
 137     callbacks.on_train_begin()
 138     # Construct epoch logs.
 139     epoch_logs = {}
 140     while epoch < epochs:
 141       for m in model.stateful_metric_functions:
 142         m.reset_states()
 143       callbacks.on_epoch_begin(epoch)
 144       steps_done = 0
 145       batch_index = 0
 146       while steps_done < steps_per_epoch:
 147         generator_output = next(output_generator)
 148 
 149         if not hasattr(generator_output, '__len__'):
 150           raise ValueError('Output of generator should be '
 151                            'a tuple `(x, y, sample_weight)` '
 152                            'or `(x, y)`. Found: ' + str(generator_output))
 153 
 154         if len(generator_output) == 2:
 155           x, y = generator_output
 156           sample_weight = None
 157         elif len(generator_output) == 3:
 158           x, y, sample_weight = generator_output
 159         else:
 160           raise ValueError('Output of generator should be '
 161                            'a tuple `(x, y, sample_weight)` '
 162                            'or `(x, y)`. Found: ' + str(generator_output))
 163         # build batch logs
 164         batch_logs = {}
 165         if isinstance(x, list):
 166           batch_size = x[0].shape[0]
 167         elif isinstance(x, dict):
 168           batch_size = list(x.values())[0].shape[0]
 169         else:
 170           batch_size = x.shape[0]
 171         batch_logs['batch'] = batch_index
 172         batch_logs['size'] = batch_size
 173         callbacks.on_batch_begin(batch_index, batch_logs)
 174 
 175         outs = model.train_on_batch(
 176             x, y, sample_weight=sample_weight, class_weight=class_weight)
 177 
 178         if not isinstance(outs, list):
 179           outs = [outs]
 180         for l, o in zip(model.metrics_names, outs):
 181           batch_logs[l] = o
 182 
 183         callbacks.on_batch_end(batch_index, batch_logs)
 184 
 185         batch_index += 1
 186         steps_done += 1
 187 
 188         # Epoch finished.
 189         if steps_done >= steps_per_epoch and do_validation:
 190           if val_gen:
 191             val_outs = evaluate_generator(
 192                 model,
 193                 validation_data,
 194                 validation_steps,
 195                 workers=workers,
 196                 use_multiprocessing=use_multiprocessing,
 197                 max_queue_size=max_queue_size)
 198           else:
 199             # No need for try/except because
 200             # data has already been validated.
 201             val_outs = model.evaluate(
 202                 val_x,
 203                 val_y,
 204                 batch_size=batch_size,
 205                 sample_weight=val_sample_weights,
 206                 verbose=0)
 207           if not isinstance(val_outs, list):
 208             val_outs = [val_outs]
 209           # Same labels assumed.
 210           for l, o in zip(model.metrics_names, val_outs):
 211             epoch_logs['val_' + l] = o
 212 
 213         if callbacks.model.stop_training:
 214           break
 215 
 216       callbacks.on_epoch_end(epoch, epoch_logs)
 217       epoch += 1
 218       if callbacks.model.stop_training:
 219         break
 220 
 221   finally:
 222     try:
 223       if enqueuer is not None:
 224         enqueuer.stop()
 225     finally:
 226       if val_enqueuer is not None:
 227         val_enqueuer.stop()
 228 
 229   callbacks.on_train_end()
 230   return model.history

 233 def evaluate_generator(model,
 234                        generator,
 235                        steps=None,
 236                        max_queue_size=10,
 237                        workers=1,
 238                        use_multiprocessing=False,
 239                        verbose=0):
 240   """See docstring for `Model.evaluate_generator`."""
 241   if not context.executing_eagerly():
 242     model._make_test_function()
 243 
 244   if hasattr(model, 'metrics'):
 245     for m in model.stateful_metric_functions:
 246       m.reset_states()
 247     stateful_metric_indices = [
 248         i for i, name in enumerate(model.metrics_names)
 249         if str(name) in model.stateful_metric_names]
 250   else:
 251     stateful_metric_indices = []
 252 
 253   steps_done = 0
 254   wait_time = 0.01
 255   all_outs = []
 256   batch_sizes = []
 257   is_sequence = isinstance(generator, Sequence)
 258   if not is_sequence and use_multiprocessing and workers > 1:
 259     logging.warning(
 260         UserWarning('Using a generator with `use_multiprocessing=True`'
 261                     ' and multiple workers may duplicate your data.'
 262                     ' Please consider using the`keras.utils.Sequence'
 263                     ' class.'))
 264   if steps is None:
 265     if is_sequence:
 266       steps = len(generator)
 267     else:
 268       raise ValueError('`steps=None` is only valid for a generator'
 269                        ' based on the `keras.utils.Sequence` class.'
 270                        ' Please specify `steps` or use the'
 271                        ' `keras.utils.Sequence` class.')
 272   enqueuer = None
 273 
 274   try:
 275     if workers > 0:
 276       if is_sequence:
 277         enqueuer = OrderedEnqueuer(
 278             generator, use_multiprocessing=use_multiprocessing)
 279       else:
 280         enqueuer = GeneratorEnqueuer(
 281             generator,
 282             use_multiprocessing=use_multiprocessing,
 283             wait_time=wait_time)
 284       enqueuer.start(workers=workers, max_queue_size=max_queue_size)
 285       output_generator = enqueuer.get()
 286     else:
 287       if is_sequence:
 288         output_generator = iter(generator)
 289       else:
 290         output_generator = generator
 291 
 292     if verbose == 1:
 293       progbar = Progbar(target=steps)
 294 
 295     while steps_done < steps:
 296       generator_output = next(output_generator)
 297       if not hasattr(generator_output, '__len__'):
 298         raise ValueError('Output of generator should be a tuple '
 299                          '(x, y, sample_weight) '
 300                          'or (x, y). Found: ' + str(generator_output))
 301       if len(generator_output) == 2:
 302         x, y = generator_output
 303         sample_weight = None
 304       elif len(generator_output) == 3:
 305         x, y, sample_weight = generator_output
 306       else:
 307         raise ValueError('Output of generator should be a tuple '
 308                          '(x, y, sample_weight) '
 309                          'or (x, y). Found: ' + str(generator_output))
 310       outs = model.test_on_batch(x, y, sample_weight=sample_weight)
 311 
 312       if isinstance(x, list):
 313         batch_size = x[0].shape[0]
 314       elif isinstance(x, dict):
 315         batch_size = list(x.values())[0].shape[0]
 316       else:
 317         batch_size = x.shape[0]
 318       if batch_size == 0:
 319         raise ValueError('Received an empty batch. '
 320                          'Batches should at least contain one item.')
 321       all_outs.append(outs)
 322 
 323       steps_done += 1
 324       batch_sizes.append(batch_size)
 325       if verbose == 1:
 326         progbar.update(steps_done)
 327 
 328   finally:
 329     if enqueuer is not None:
 330       enqueuer.stop()
 331 
 332   if not isinstance(outs, list):
 333     return np.average(np.asarray(all_outs), weights=batch_sizes)
 334   else:
 335     averages = []
 336     for i in range(len(outs)):
 337       if i not in stateful_metric_indices:
 338         averages.append(
 339             np.average([out[i] for out in all_outs], weights=batch_sizes))
 340       else:
 341         averages.append(np.float64(all_outs[-1][i]))
 342     return averages

 345 def predict_generator(model,
 346                       generator,
 347                       steps=None,
 348                       max_queue_size=10,
 349                       workers=1,
 350                       use_multiprocessing=False,
 351                       verbose=0):
 352   """See docstring for `Model.predict_generator`."""
 353   if not context.executing_eagerly():
 354     model._make_test_function()
 355 
 356   steps_done = 0
 357   wait_time = 0.01
 358   all_outs = []
 359   is_sequence = isinstance(generator, Sequence)
 360   if not is_sequence and use_multiprocessing and workers > 1:
 361     logging.warning(
 362         UserWarning('Using a generator with `use_multiprocessing=True`'
 363                     ' and multiple workers may duplicate your data.'
 364                     ' Please consider using the`keras.utils.Sequence'
 365                     ' class.'))
 366   if steps is None:
 367     if is_sequence:
 368       steps = len(generator)
 369     else:
 370       raise ValueError('`steps=None` is only valid for a generator'
 371                        ' based on the `keras.utils.Sequence` class.'
 372                        ' Please specify `steps` or use the'
 373                        ' `keras.utils.Sequence` class.')
 374   enqueuer = None
 375 
 376   try:
 377     if workers > 0:
 378       if is_sequence:
 379         enqueuer = OrderedEnqueuer(
 380             generator, use_multiprocessing=use_multiprocessing)
 381       else:
 382         enqueuer = GeneratorEnqueuer(
 383             generator,
 384             use_multiprocessing=use_multiprocessing,
 385             wait_time=wait_time)
 386       enqueuer.start(workers=workers, max_queue_size=max_queue_size)
 387       output_generator = enqueuer.get()
 388     else:
 389       if is_sequence:
 390         output_generator = iter(generator)
 391       else:
 392         output_generator = generator
 393 
 394     if verbose == 1:
 395       progbar = Progbar(target=steps)
 396 
 397     while steps_done < steps:
 398       generator_output = next(output_generator)
 399       if isinstance(generator_output, tuple):
 400         # Compatibility with the generators
 401         # used for training.
 402         if len(generator_output) == 2:
 403           x, _ = generator_output
 404         elif len(generator_output) == 3:
 405           x, _, _ = generator_output
 406         else:
 407           raise ValueError('Output of generator should be '
 408                            'a tuple `(x, y, sample_weight)` '
 409                            'or `(x, y)`. Found: ' + str(generator_output))
 410       else:
 411         # Assumes a generator that only
 412         # yields inputs (not targets and sample weights).
 413         x = generator_output
 414 
 415       outs = model.predict_on_batch(x)
 416       if not isinstance(outs, list):
 417         outs = [outs]
 418 
 419       if not all_outs:
 420         for out in outs:
 421           all_outs.append([])
 422 
 423       for i, out in enumerate(outs):
 424         all_outs[i].append(out)
 425       steps_done += 1
 426       if verbose == 1:
 427         progbar.update(steps_done)
 428 
 429   finally:
 430     if enqueuer is not None:
 431       enqueuer.stop()
 432 
 433   if len(all_outs) == 1:
 434     if steps_done == 1:
 435       return all_outs[0][0]
 436     else:
 437       return np.concatenate(all_outs[0])
 438   if steps_done == 1:
 439     return [out[0] for out in all_outs]
 440   else:
 441     return [np.concatenate(out) for out in all_outs]
