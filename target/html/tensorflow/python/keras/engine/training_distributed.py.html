  40 def fit_loop(
  41     model,
  42     iterator,
  43     epochs=100,
  44     verbose=1,
  45     callbacks=None,
  46     val_iterator=None,
  47     initial_epoch=0,
  48     steps_per_epoch=None,
  49     validation_steps=None):
  50   """Fit loop for training with DistributionStrategy.
  51 
  52   Arguments:
  53       model: Keras Model instance.
  54       iterator: Iterator for input data.
  55       epochs: Number of times to iterate over the data
  56       verbose: Integer, Verbosity mode, 0, 1 or 2
  57       callbacks: List of callbacks to be called during training
  58       val_iterator: Iterator for validation data.
  59       initial_epoch: Epoch at which to start training
  60           (useful for resuming a previous training run)
  61       steps_per_epoch: Total number of steps (batches of samples)
  62           before declaring one epoch finished and starting the
  63           next epoch. Ignored with the default value of `None`.
  64       validation_steps: Number of steps to run validation for
  65           (only if doing validation from data tensors).
  66           Ignored with the default value of `None`.
  67 
  68   Returns:
  69       `History` object.
  70 
  71   Raises:
  72       ValueError: in case of invalid arguments.
  73   """
  74   current_strategy = model._distribution_strategy
  75 
  76   # TODO(priyag, sourabhbajaj): Remove this when the codepaths are merged.
  77   if current_strategy.__class__.__name__ == 'TPUStrategy':
  78     return _experimental_fit_loop(
  79         model, iterator, epochs, verbose, callbacks, initial_epoch,
  80         steps_per_epoch)
  81 
  82   if not model._grouped_model:
  83     clone_model_on_towers(model, current_strategy, make_callback_model=True)
  84 
  85   def _per_device_train_function(model):
  86     model._make_train_function()
  87     return (model.train_function.inputs,
  88             model.train_function.outputs,
  89             model.train_function.updates_op,
  90             model.train_function.session_kwargs)
  91 
  92   inputs, targets = _get_input_from_iterator(iterator, model)
  93   with current_strategy.scope():
  94     # Create train ops on each of the devices when we call
  95     # `_per_device_train_function`.
  96     (grouped_inputs, grouped_outputs, grouped_updates,
  97      grouped_session_args) = current_strategy.call_for_each_tower(
  98          _per_device_train_function, model._grouped_model)
  99     # Unwrap all the per device values returned from `call_for_each_tower`.
 100     # Unwrapping per device values gives you a list of values that can be
 101     # used to construct a new train function that is composed of update ops on
 102     # all the devices over which the model is distributed.
 103     (all_inputs, all_outputs, all_updates,
 104      all_session_args) = distributed_training_utils.unwrap_values(
 105          current_strategy, grouped_inputs, grouped_outputs,
 106          grouped_updates, grouped_session_args, with_loss_tensor=True)
 107 
 108     # Dataset inputs and targets are also per devices values that need to be
 109     # unwrapped.
 110     dataset_inputs = distributed_training_utils.flatten_perdevice_values(
 111         current_strategy, inputs)
 112     dataset_targets = distributed_training_utils.flatten_perdevice_values(
 113         current_strategy, targets)
 114 
 115     # Create a train function that is composed of all the parameters above.
 116     distributed_train_function = K.Function(
 117         all_inputs, all_outputs,
 118         updates=all_updates,
 119         name='distributed_train_function',
 120         **all_session_args)
 121 
 122     # We need to set sample_weights to None since there are sample weight
 123     # placeholders that are created with default values.
 124     sample_weights = [None for _ in range(len(model.outputs) *
 125                                           current_strategy.num_towers)]
 126     if model.uses_learning_phase and not isinstance(K.learning_phase(), int):
 127       ins = dataset_inputs + dataset_targets + sample_weights + [1]
 128     else:
 129       ins = dataset_inputs + dataset_targets
 130 
 131     do_validation = False
 132     if validation_steps:
 133       do_validation = True
 134 
 135     # Copy the weights from the original model to each of the replicated models.
 136     orig_model_weights = model.get_weights()
 137     distributed_model = current_strategy.unwrap(model._grouped_model)[0]
 138     distributed_training_utils.set_weights(
 139         current_strategy, distributed_model, orig_model_weights)
 140 
 141     callbacks = cbks.configure_callbacks(
 142         callbacks,
 143         model,
 144         do_validation=do_validation,
 145         val_inputs=None,
 146         val_targets=None,
 147         epochs=epochs,
 148         steps_per_epoch=steps_per_epoch,
 149         verbose=verbose)
 150     out_labels = model.metrics_names or []
 151     callbacks.on_train_begin()
 152 
 153     assert steps_per_epoch is not None
 154 
 155     for epoch in range(initial_epoch, epochs):
 156       # Reset stateful metrics
 157       for m in model.stateful_metric_functions:
 158         m.reset_states()
 159       callbacks.on_epoch_begin(epoch)
 160       epoch_logs = {}
 161       for step_index in range(steps_per_epoch):
 162         batch_logs = {'batch': step_index, 'size': 1}
 163         callbacks.on_batch_begin(step_index, batch_logs)
 164         try:
 165           outs = distributed_train_function(ins)
 166         except errors.OutOfRangeError:
 167           logging.warning('Your dataset iterator ran out of data; '
 168                           'interrupting training. Make sure that your dataset '
 169                           'can generate at least `steps_per_epoch * epochs` '
 170                           'batches (in this case, %d batches).' %
 171                           steps_per_epoch * epochs)
 172           break
 173 
 174         if not isinstance(outs, list):
 175           outs = [outs]
 176 
 177         outs = _aggregate_metrics_across_towers(current_strategy.num_towers,
 178                                                 out_labels,
 179                                                 model.stateful_metric_names,
 180                                                 outs)
 181         for l, o in zip(out_labels, outs):
 182           batch_logs[l] = o
 183         callbacks.on_batch_end(step_index, batch_logs)
 184         if callbacks.model.stop_training:
 185           break
 186       if do_validation:
 187         val_outs = test_loop(
 188             model,
 189             val_iterator,
 190             steps=validation_steps,
 191             verbose=0)
 192         if not isinstance(val_outs, list):
 193           val_outs = [val_outs]
 194         # Same labels assumed.
 195         for l, o in zip(out_labels, val_outs):
 196           epoch_logs['val_' + l] = o
 197 
 198       callbacks.on_epoch_end(epoch, epoch_logs)
 199       if callbacks.model.stop_training:
 200         break
 201     callbacks.on_train_end()
 202 
 203     # Copy the weights back from the replicated model to the original model.
 204     updated_weights = current_strategy.unwrap(
 205         model._grouped_model)[0].get_weights()
 206     model.set_weights(updated_weights)
 207     return model.history

 210 def _experimental_fit_loop(
 211     model,
 212     iterator,
 213     epochs=100,
 214     verbose=1,
 215     callbacks=None,
 216     initial_epoch=0,
 217     steps_per_epoch=None):
 218   """Fit loop for training with TPU DistributionStrategy.
 219 
 220   Arguments:
 221       model: Keras Model instance.
 222       iterator: Iterator that returns inputs and targets
 223       epochs: Number of times to iterate over the data
 224       verbose: Integer, Verbosity mode, 0, 1 or 2
 225       callbacks: List of callbacks to be called during training
 226       initial_epoch: Epoch at which to start training
 227           (useful for resuming a previous training run)
 228       steps_per_epoch: Total number of steps (batches of samples)
 229           before declaring one epoch finished and starting the
 230           next epoch. Ignored with the default value of `None`.
 231 
 232   Returns:
 233       Returns `None`.
 234 
 235   Raises:
 236       ValueError: in case of invalid arguments.
 237   """
 238   current_strategy = model._distribution_strategy
 239 
 240   K.get_session().run(current_strategy.initialize())
 241 
 242   def _per_device_train_function(model):
 243     model._make_train_function()
 244     return (model.train_function.inputs,
 245             model.train_function.outputs,
 246             model.train_function.updates_op,
 247             model.train_function.session_kwargs)
 248 
 249   # TODO(priyag, sourabhbajaj): This should likely not be hardcoded here.
 250   K.set_learning_phase(1)
 251 
 252   def step_fn(ctx, inputs, targets):
 253     """Clones the model and calls make_train_function."""
 254     # TODO(priyag, sourabhbajaj): The model gets cloned every time
 255     # fit/test/predict is called. We should look into caching this keyed on
 256     # input shapes.
 257     clone_model_on_towers(
 258         model,
 259         current_strategy,
 260         make_callback_model=True,
 261         inputs=inputs,
 262         targets=targets)
 263 
 264     (grouped_inputs, grouped_outputs, grouped_updates,
 265      grouped_session_args) = current_strategy.call_for_each_tower(
 266          _per_device_train_function, model._grouped_model)
 267     (all_inputs, all_outputs, all_updates,
 268      all_session_args) = distributed_training_utils.unwrap_values(
 269          current_strategy, grouped_inputs, grouped_outputs,
 270          grouped_updates, grouped_session_args)
 271     combined_fn = K.Function(
 272         all_inputs, all_outputs,
 273         updates=all_updates,
 274         name='distributed_train_function',
 275         **all_session_args)
 276 
 277     out_labels = model.metrics_names or []
 278     for label, output in zip(out_labels, combined_fn.outputs):
 279       if label == 'loss':
 280         aggregation = distribute_lib.get_loss_reduction()
 281       else:
 282         # We aggregate all other metrics using mean for now. This is temporary
 283         # workaround until new metrics are in place.
 284         aggregation = variable_scope.VariableAggregation.MEAN
 285       ctx.set_last_step_output(label, output, aggregation)
 286 
 287     # TODO(priyag, sourabhbajaj): Ignoring these things from the combined_fn:
 288     # feed_dict, session kwargs, run options, run_metadata for now. These should
 289     # be handled appropriately
 290     return combined_fn.updates_op
 291 
 292   # Add initial dummy values for loss and other metric tensors.
 293   initial_loop_values = {}
 294   initial_loop_values['loss'] = constant_op.constant(1e7)
 295   for name, tensor in zip(model.metrics_names[1:], model.metrics_tensors):
 296     initial_loop_values[name] = array_ops.zeros(tensor.shape, tensor.dtype)
 297 
 298   if steps_per_epoch is None:
 299     raise ValueError('steps_per_epoch should be specified in the fit call.')
 300   steps_per_run_var = K.variable(
 301       value=min(steps_per_epoch, current_strategy.steps_per_run),
 302       dtype='int32',
 303       name='steps_per_run_var')
 304 
 305   with current_strategy.scope():
 306     ctx = current_strategy.run_steps_on_dataset(
 307         step_fn, iterator, iterations=steps_per_run_var,
 308         initial_loop_values=initial_loop_values)
 309 
 310   train_op = ctx.run_op
 311   output_tensors = ctx.last_step_outputs
 312 
 313   # Copy the weights from the original model to each of the replicated models.
 314   orig_model_weights = model.get_weights()
 315   with current_strategy.scope():
 316     distributed_model = current_strategy.unwrap(model._grouped_model)[0]
 317     distributed_training_utils.set_weights(
 318         current_strategy, distributed_model, orig_model_weights)
 319   callbacks = cbks.configure_callbacks(
 320       callbacks,
 321       model,
 322       do_validation=False,
 323       val_inputs=None,
 324       val_targets=None,
 325       epochs=epochs,
 326       steps_per_epoch=steps_per_epoch,
 327       verbose=verbose)
 328   # TODO(priyag, sourabhbajaj): Add callbacks support for per step callback
 329   # TODO(priyag, sourabhbajaj): Add validation.
 330 
 331   # Calculate the steps each time on the device.
 332   steps_to_run = [current_strategy.steps_per_run] * (
 333       steps_per_epoch // current_strategy.steps_per_run)
 334   if steps_per_epoch % current_strategy.steps_per_run:
 335     steps_to_run.append(steps_per_epoch % current_strategy.steps_per_run)
 336 
 337   callbacks.on_train_begin()
 338   for epoch in range(initial_epoch, epochs):
 339     callbacks.on_epoch_begin(epoch)
 340     epoch_logs = {}
 341     step_index = 0
 342     prev_step_count = None
 343     for step_count in steps_to_run:
 344       batch_logs = {'batch': step_index, 'size': 1, 'num_steps': step_count}
 345       callbacks.on_batch_begin(step_index, batch_logs)
 346       if prev_step_count is None or step_count != prev_step_count:
 347         steps_per_run_var.load(step_count, K.get_session())
 348         prev_step_count = step_count
 349       try:
 350         _, outputs = K.get_session().run([train_op, output_tensors])
 351       except errors.OutOfRangeError:
 352         logging.warning('Your dataset iterator ran out of data; '
 353                         'interrupting training. Make sure that your dataset '
 354                         'can generate at least `steps_per_epoch * epochs` '
 355                         'batches (in this case, %d batches).' %
 356                         steps_per_epoch * epochs)
 357         break
 358 
 359       batch_logs.update(outputs)
 360       callbacks.on_batch_end(step_index, batch_logs)
 361       step_index = step_index + step_count
 362       if callbacks.model.stop_training:
 363         break
 364 
 365     callbacks.on_epoch_end(epoch, epoch_logs)
 366     if callbacks.model.stop_training:
 367       break
 368   callbacks.on_train_end()
 369 
 370   # Copy the weights back from the replicated model to the original model.
 371   with current_strategy.scope():
 372     updated_weights = current_strategy.unwrap(
 373         model._grouped_model)[0].get_weights()
 374     model.set_weights(updated_weights)
 375 
 376   K.get_session().run(current_strategy.finalize())
 377   return model.history

 380 def test_loop(model, iterator, verbose=0, steps=None):
 381   """Test loop for evaluating with DistributionStrategy.
 382 
 383   Arguments:
 384       model: Keras Model instance.
 385       iterator: Iterator for input data.
 386       verbose: Integer, Verbosity mode 0 or 1.
 387       steps: Total number of steps (batches of samples)
 388           before declaring predictions finished.
 389           Ignored with the default value of `None`.
 390 
 391   Returns:
 392       Scalar loss (if the model has a single output and no metrics)
 393       or list of scalars (if the model has multiple outputs
 394       and/or metrics). The attribute `model.metrics_names` will give you
 395       the display labels for the outputs.
 396   """
 397   current_strategy = model._distribution_strategy
 398 
 399   # TODO(priyag, sourabhbajaj): Remove this when the codepaths are merged.
 400   if current_strategy.__class__.__name__ == 'TPUStrategy':
 401     return _experimental_test_loop(model, iterator, verbose, steps)
 402 
 403   if not model._grouped_model:
 404     clone_model_on_towers(model, current_strategy)
 405 
 406   def _per_device_test_function(model):
 407     model._make_test_function()
 408     return (model.test_function.inputs,
 409             model.test_function.outputs,
 410             model.test_function.updates_op,
 411             model.test_function.session_kwargs)
 412 
 413   inputs, targets = _get_input_from_iterator(iterator, model)
 414   with current_strategy.scope():
 415     (grouped_inputs, grouped_outputs, grouped_updates,
 416      grouped_session_args) = current_strategy.call_for_each_tower(
 417          _per_device_test_function, model._grouped_model)
 418 
 419     (all_inputs, all_outputs, all_updates,
 420      all_session_args) = distributed_training_utils.unwrap_values(
 421          current_strategy, grouped_inputs, grouped_outputs, grouped_updates,
 422          grouped_session_args, with_loss_tensor=True)
 423 
 424     dataset_inputs = distributed_training_utils.flatten_perdevice_values(
 425         current_strategy, inputs)
 426     dataset_targets = distributed_training_utils.flatten_perdevice_values(
 427         current_strategy, targets)
 428 
 429     distributed_test_function = K.Function(
 430         all_inputs, all_outputs,
 431         updates=all_updates,
 432         name='distributed_test_function',
 433         **all_session_args)
 434 
 435     # We need to set sample_weights to None since there are sample weight
 436     # placeholders that are created with default values.
 437     sample_weights = [None for _ in range(len(model.outputs) *
 438                                           current_strategy.num_towers)]
 439     if model.uses_learning_phase and not isinstance(K.learning_phase(), int):
 440       ins = dataset_inputs + dataset_targets + sample_weights + [0]
 441     else:
 442       ins = dataset_inputs + dataset_targets
 443 
 444     for m in model.stateful_metric_functions:
 445       m.reset_states()
 446     stateful_metric_indices = [
 447         i for i, name in enumerate(model.metrics_names)
 448         if str(name) in model.stateful_metric_names
 449     ]
 450 
 451     outs = []
 452     if verbose == 1:
 453       progbar = Progbar(target=steps)
 454 
 455     # Copy the weights from the original model to each of the replicated models.
 456     orig_model_weights = model.get_weights()
 457     distributed_model = current_strategy.unwrap(model._grouped_model)[0]
 458     distributed_training_utils.set_weights(
 459         current_strategy, distributed_model, orig_model_weights)
 460 
 461     assert steps is not None
 462     for step in range(steps):
 463       batch_outs = distributed_test_function(ins)
 464       batch_outs = _aggregate_metrics_across_towers(
 465           current_strategy.num_towers, model.metrics_names,
 466           model.stateful_metric_names, batch_outs)
 467       if isinstance(batch_outs, list):
 468         if step == 0:
 469           outs = [0.] * len(batch_outs)
 470         for i, batch_out in enumerate(batch_outs):
 471           if i in stateful_metric_indices:
 472             outs[i] = batch_out
 473           else:
 474             outs[i] += batch_out
 475       else:
 476         if step == 0:
 477           outs.append(0.)
 478         outs[0] += batch_outs
 479       if verbose >= 1:
 480         progbar.update(step + 1)
 481     for i in range(len(outs)):
 482       if i not in stateful_metric_indices:
 483         outs[i] /= steps
 484 
 485     if len(outs) == 1:
 486       return outs[0]
 487     return outs

 490 def _experimental_test_loop(model, iterator, verbose=0, steps=None):
 491   """Test loop for evaluating with TPU DistributionStrategy.
 492 
 493   Arguments:
 494       model: Keras Model instance.
 495       iterator: Iterator for input data.
 496       verbose: Integer, Verbosity mode 0 or 1.
 497       steps: Total number of steps (batches of samples)
 498           before declaring predictions finished.
 499           Ignored with the default value of `None`.
 500 
 501   Returns:
 502       Scalar loss (if the model has a single output and no metrics)
 503       or list of scalars (if the model has multiple outputs
 504       and/or metrics). The attribute `model.metrics_names` will give you
 505       the display labels for the outputs.
 506   """
 507   current_strategy = model._distribution_strategy
 508   K.get_session().run(current_strategy.initialize())
 509 
 510   def _per_device_test_function(model):
 511     model._make_test_function()
 512     return (model.test_function.inputs,
 513             model.test_function.outputs,
 514             model.test_function.updates_op,
 515             model.test_function.session_kwargs)
 516 
 517   # TODO(priyag, sourabhbajaj): This should likely not be hardcoded here.
 518   K.set_learning_phase(0)
 519 
 520   def step_fn(ctx, inputs, targets):
 521     """Clones the model and calls make_test_function."""
 522     # TODO(priyag, sourabhbajaj): The model gets cloned every time
 523     # fit/test/predict is called. We should look into caching this keyed on
 524     # input shapes.
 525     clone_model_on_towers(
 526         model,
 527         current_strategy,
 528         make_callback_model=False,
 529         inputs=inputs,
 530         targets=targets)
 531 
 532     (grouped_inputs, grouped_outputs, grouped_updates,
 533      grouped_session_args) = current_strategy.call_for_each_tower(
 534          _per_device_test_function, model._grouped_model)
 535 
 536     (all_inputs, all_outputs, all_updates,
 537      all_session_args) = distributed_training_utils.unwrap_values(
 538          current_strategy, grouped_inputs, grouped_outputs, grouped_updates,
 539          grouped_session_args)
 540 
 541     combined_fn = K.Function(
 542         all_inputs, all_outputs,
 543         updates=all_updates,
 544         name='distributed_test_function',
 545         **all_session_args)
 546 
 547     for label, output in zip(model.metrics_names, combined_fn.outputs):
 548       if label == 'loss':
 549         aggregation = distribute_lib.get_loss_reduction()
 550       else:
 551         # We aggregate all other metrics using mean for now. This is temporary
 552         # workaround until new metrics are in place.
 553         aggregation = variable_scope.VariableAggregation.MEAN
 554       ctx.set_last_step_output(label, output, aggregation)
 555 
 556     return combined_fn.updates_op
 557 
 558   # Add initial dummy values for loss and other metric tensors.
 559   initial_loop_values = {}
 560   initial_loop_values['loss'] = constant_op.constant(1e7)
 561   for name, tensor in zip(model.metrics_names[1:], model.metrics_tensors):
 562     initial_loop_values[name] = array_ops.zeros(tensor.shape, tensor.dtype)
 563 
 564   with current_strategy.scope():
 565     # TODO(priyag): Use steps_per_run when we use new metrics as they will
 566     # allow handling metric computation at each step using variables.
 567     ctx = current_strategy.run_steps_on_dataset(
 568         step_fn, iterator, iterations=1,
 569         initial_loop_values=initial_loop_values)
 570 
 571   test_op = ctx.run_op
 572   output_tensors = ctx.last_step_outputs
 573 
 574   if verbose == 1:
 575     progbar = Progbar(target=steps)
 576 
 577   # Copy the weights from the original model to each of the replicated models.
 578   orig_model_weights = model.get_weights()
 579   with current_strategy.scope():
 580     distributed_model = current_strategy.unwrap(model._grouped_model)[0]
 581     distributed_training_utils.set_weights(
 582         current_strategy, distributed_model, orig_model_weights)
 583 
 584   assert steps is not None
 585   outs = [0.] * len(model.metrics_names)
 586   for step in range(steps):
 587     _, batch_outs = K.get_session().run([test_op, output_tensors])
 588     for i, label in enumerate(model.metrics_names):
 589       outs[i] += batch_outs[label]
 590     if verbose >= 1:
 591       progbar.update(step + 1)
 592   for i in range(len(outs)):
 593     outs[i] /= (steps)
 594 
 595   K.get_session().run(current_strategy.finalize())
 596 
 597   if len(outs) == 1:
 598     return outs[0]
 599   return outs

 602 def predict_loop(model, iterator, verbose=0, steps=None):
 603   """Predict loop for predicting with DistributionStrategy.
 604 
 605   Arguments:
 606       model: Keras Model instance.
 607       iterator: Iterator for input data.
 608       verbose: Integer, Verbosity mode 0 or 1.
 609       steps: Total number of steps (batches of samples)
 610           before declaring `_predict_loop` finished.
 611           Ignored with the default value of `None`.
 612 
 613   Returns:
 614       Array of predictions (if the model has a single output)
 615       or list of arrays of predictions
 616       (if the model has multiple outputs).
 617   """
 618   current_strategy = model._distribution_strategy
 619 
 620   # TODO(priyag, sourabhbajaj): Remove this when the codepaths are merged.
 621   if current_strategy.__class__.__name__ == 'TPUStrategy':
 622     return _experimental_predict_loop(model, iterator, verbose, steps)
 623 
 624   if not model._grouped_model:
 625     clone_model_on_towers(model, current_strategy)
 626 
 627   def _per_device_predict_function(model):
 628     model._make_predict_function()
 629     return (model.predict_function.inputs,
 630             model.predict_function.outputs,
 631             model.predict_function.updates_op,
 632             model.predict_function.session_kwargs)
 633 
 634   inputs, _ = _get_input_from_iterator(iterator, model)
 635   with current_strategy.scope():
 636     (grouped_inputs, grouped_outputs, grouped_updates,
 637      grouped_session_args) = current_strategy.call_for_each_tower(
 638          _per_device_predict_function, model._grouped_model)
 639 
 640     (all_inputs, all_outputs, all_updates,
 641      all_session_args) = distributed_training_utils.unwrap_values(
 642          current_strategy, grouped_inputs, grouped_outputs, grouped_updates,
 643          grouped_session_args)
 644 
 645     dataset_inputs = distributed_training_utils.flatten_perdevice_values(
 646         current_strategy, inputs)
 647 
 648     distributed_predict_function = K.Function(
 649         all_inputs, all_outputs,
 650         updates=all_updates,
 651         name='distributed_predict_function',
 652         **all_session_args)
 653 
 654     if model.uses_learning_phase and not isinstance(K.learning_phase(), int):
 655       ins = dataset_inputs + [0]
 656     else:
 657       ins = dataset_inputs
 658 
 659     if verbose == 1:
 660       progbar = Progbar(target=steps)
 661 
 662     # Copy the weights from the original model to each of the replicated models.
 663     orig_model_weights = model.get_weights()
 664     distributed_model = current_strategy.unwrap(model._grouped_model)[0]
 665     distributed_training_utils.set_weights(
 666         current_strategy, distributed_model, orig_model_weights)
 667 
 668     if steps is not None:
 669       # Since we do not know how many samples we will see, we cannot
 670       # pre-allocate the returned Numpy arrays. Instead, we store one array per
 671       # batch seen and concatenate them upon returning.
 672       unconcatenated_outs = []
 673       for step in range(steps):
 674         batch_outs = distributed_predict_function(ins)
 675         if not isinstance(batch_outs, list):
 676           batch_outs = [batch_outs]
 677         if step == 0:
 678           for _ in batch_outs:
 679             unconcatenated_outs.append([])
 680         # TODO(anjalisridhar): Should combine the outputs from multiple towers
 681         # correctly here.
 682         for i, batch_out in enumerate(batch_outs):
 683           unconcatenated_outs[i].append(batch_out)
 684         if verbose >= 1:
 685           progbar.update(step + 1)
 686       if len(unconcatenated_outs) == 1:
 687         return np.concatenate(unconcatenated_outs[0], axis=0)
 688       return [
 689           np.concatenate(unconcatenated_outs[i], axis=0)
 690           for i in range(len(unconcatenated_outs))
 691       ]

 694 def _experimental_predict_loop(model, iterator, verbose=0, steps=None):
 695   """Predict loop for predicting with TPU DistributionStrategy.
 696 
 697   Arguments:
 698       model: Keras Model instance.
 699       iterator: Iterator for input data.
 700       verbose: Integer, Verbosity mode 0 or 1.
 701       steps: Total number of steps (batches of samples)
 702           before declaring `_predict_loop` finished.
 703           Ignored with the default value of `None`.
 704 
 705   Returns:
 706       Array of predictions (if the model has a single output)
 707       or list of arrays of predictions
 708       (if the model has multiple outputs).
 709   """
 710   current_strategy = model._distribution_strategy
 711   K.get_session().run(current_strategy.initialize())
 712 
 713   # TODO(priyag, sourabhbajaj): This should likely not be hardcoded here.
 714   K.set_learning_phase(0)
 715 
 716   def _per_device_predict_function(model):
 717     model._make_predict_function()
 718     return (model.predict_function.inputs,
 719             model.predict_function.outputs,
 720             model.predict_function.updates_op,
 721             model.predict_function.session_kwargs)
 722 
 723   def step_fn(ctx, inputs, targets):
 724     """Clones the model and calls make_predict_function."""
 725 
 726     # TODO(anjalisridhar): Support predict input correctly as it will not
 727     # contain targets, only inputs.
 728     del targets
 729 
 730     # TODO(priyag, sourabhbajaj): The model gets cloned every time
 731     # fit/test/predict is called. We should look into caching this keyed on
 732     # input shapes.
 733     clone_model_on_towers(
 734         model,
 735         current_strategy,
 736         make_callback_model=False,
 737         inputs=inputs)
 738 
 739     (grouped_inputs, grouped_outputs, grouped_updates,
 740      grouped_session_args) = current_strategy.call_for_each_tower(
 741          _per_device_predict_function, model._grouped_model)
 742 
 743     (all_inputs, all_outputs, all_updates,
 744      all_session_args) = distributed_training_utils.unwrap_values(
 745          current_strategy, grouped_inputs, grouped_outputs, grouped_updates,
 746          grouped_session_args)
 747 
 748     combined_fn = K.Function(
 749         all_inputs, all_outputs,
 750         updates=all_updates,
 751         name='distributed_predict_function',
 752         **all_session_args)
 753 
 754     for label, output in zip(model.output_names, combined_fn.outputs):
 755       ctx.set_last_step_output(label, output)
 756 
 757     return combined_fn.updates_op
 758 
 759   # Add initial dummy values for outputs.
 760   initial_loop_values = {}
 761   batch_dimension = distributed_training_utils.get_batch_dimension(iterator)
 762   for name, tensor in zip(model.output_names, model.outputs):
 763     # TODO(priyag): This is a workaround as we do not know the batch dimension
 764     # of the model's output at this point.
 765     shape = tensor_shape.TensorShape(tensor.shape.dims)
 766     shape.dims = [batch_dimension] + shape.dims[1:]
 767     initial_loop_values[name] = array_ops.zeros(shape, tensor.dtype)
 768 
 769   with current_strategy.scope():
 770     # TODO(priyag, sourabhbajaj): Support steps_per_run if/when we add outfeed.
 771     ctx = current_strategy.run_steps_on_dataset(
 772         step_fn, iterator, iterations=1,
 773         initial_loop_values=initial_loop_values)
 774 
 775   predict_op = ctx.run_op
 776   output_tensors = ctx.last_step_outputs
 777 
 778   if verbose == 1:
 779     progbar = Progbar(target=steps)
 780 
 781   # Copy the weights from the original model to each of the replicated models.
 782   orig_model_weights = model.get_weights()
 783   with current_strategy.scope():
 784     distributed_model = current_strategy.unwrap(model._grouped_model)[0]
 785     distributed_training_utils.set_weights(
 786         current_strategy, distributed_model, orig_model_weights)
 787 
 788   assert steps is not None
 789   # Since we do not know how many samples we will see, we cannot pre-allocate
 790   # the returned Numpy arrays. Instead, we store one array per batch seen
 791   # and concatenate them upon returning.
 792   unconcatenated_outs = [[] for _ in model.outputs]
 793   for step in range(steps):
 794     _, batch_outs = K.get_session().run([predict_op, output_tensors])
 795     # TODO(priyag): maybe need to unwrap the outputs first for MirroredStrategy.
 796     for i, label in enumerate(model.output_names):
 797       unconcatenated_outs[i].extend(batch_outs[label])
 798     if verbose >= 1:
 799       progbar.update(step + 1)
 800 
 801   K.get_session().run(current_strategy.finalize())
 802 
 803   if len(unconcatenated_outs) == 1:
 804     return np.concatenate(unconcatenated_outs[0], axis=0)
 805   return [
 806       np.concatenate(unconcatenated_outs[i], axis=0)
 807       for i in range(len(unconcatenated_outs))
 808   ]

 811 def _clone_and_build_model(model, inputs=None, targets=None):
 812   """Clone and build the given keras_model."""
 813   # We need to set the import here since we run into a circular dependency
 814   # error.
 815   from tensorflow.python.keras import models  # pylint: disable=g-import-not-at-top
 816   cloned_model = models.clone_model(model, input_tensors=inputs)
 817 
 818   # Compile and build model.
 819   if isinstance(model.optimizer, optimizers.TFOptimizer):
 820     optimizer = model.optimizer
 821   else:
 822     optimizer_config = model.optimizer.get_config()
 823     optimizer = model.optimizer.__class__.from_config(optimizer_config)
 824 
 825   # TODO(priyag): Is there a cleaner way to do this? The API doc suggests a
 826   # single tensor should be OK but it throws an error in that case.
 827   if (targets is not None and not isinstance(targets, list) and
 828       not isinstance(targets, dict)):
 829     targets = [targets]
 830   cloned_model.compile(
 831       optimizer,
 832       model.loss,
 833       metrics=metrics_module.clone_metrics(model.metrics),
 834       loss_weights=model.loss_weights,
 835       sample_weight_mode=model.sample_weight_mode,
 836       weighted_metrics=metrics_module.clone_metrics(model.weighted_metrics),
 837       target_tensors=targets)
 838   return cloned_model

 841 def clone_model_on_towers(
 842     model, strategy, make_callback_model=False, inputs=None, targets=None):
 843   """Create a cloned model on each tower."""

 846         _clone_and_build_model, model, inputs, targets)
 847   if make_callback_model:
 848     model._make_callback_model()

 851 def _aggregate_metrics_across_towers(num_devices, out_labels,
 852                                      stateful_metric_names, outs):
 853   """Aggregates stateless metrics values across towers.
 854 
 855   When using `MirroredStrategy`, the number of towers is equal to the
 856   number of devices over which training is distributed. This may not always be
 857   the case.
 858 
 859   Args:
 860     num_devices: Number of devices over which the model is being distributed.
 861     out_labels: The list of metric names passed to `compile`.
 862     stateful_metric_names: List of stateful metric names on the model.
 863     outs: The output from all the towers.
 864 
 865   Returns:
 866     The average value of each metric across the towers.
 867   """
 868   # TODO(anjalisridhar): Temporary workaround for aggregating metrics
 869   # across towers. Replace with the new metrics module eventually.
 870   merged_output = []
 871   # The first output is the total loss.
 872   merged_output.append(outs[0])
 873   current_index = 1
 874   # Each label in `out_labels` corresponds to one set of metrics. The
 875   # number of metric values corresponds to the number of devices. We
 876   # currently take the mean of the values.
 877   for metric_name in out_labels[1:]:
 878     if metric_name in stateful_metric_names:
 879       # For stateful metrics, we get one aggregated result value.
 880       merged_output.append(outs[current_index])
 881       current_index += 1
 882     else:
 883       m = np.mean(outs[current_index:current_index + num_devices])
 884       merged_output.append(m)
 885       current_index += num_devices
 886 
 887   return merged_output

 890 def _get_input_from_iterator(iterator, model):
 891   """Get elements from the iterator and verify the input shape and type."""
 892   next_element = iterator.get_next()
 893 
 894   if isinstance(next_element, tuple):
 895     x, y = next_element
 896   else:
 897     x = next_element
 898     y = None
 899   # Validate that all the elements in x and y are of the same type and shape.
 900   # We can then pass the first element of x and y to `_standardize_weights`
 901   # below and be confident of the output.
 902   x_values, y_values = distributed_training_utils.\
 903     validate_distributed_dataset_inputs(model._distribution_strategy, x, y)
 904   # TODO(sourabhbajaj): Add support for sample weights in distribution
 905   # strategy.
 906   model._standardize_weights(x_values, y_values)
 907   return x, y
