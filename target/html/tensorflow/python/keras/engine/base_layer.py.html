  56 class CallConvention(enum.Enum):
  57   """Calling conventions for passing `Layer` inputs to `Layer.call`."""
  58   # The Layer takes inputs as its first argument, named "inputs" for
  59   # compatibility with the signature of Layer.__call__. This is the mode assumed
  60   # for Layers which are not subclassed Models.
  61   EXPLICIT_INPUTS_ARGUMENT = 1
  62   # The Layer takes a single positional argument, not named "inputs". It's
  63   # treated like an "inputs" argument.
  64   SINGLE_POSITIONAL_ARGUMENT = 2
  65   # The Layer has multiple positional arguments to which its inputs should be
  66   # bound.
  67   POSITIONAL_ARGUMENTS_ARE_INPUTS = 3

  71 class Layer(checkpointable.CheckpointableBase):
  72   """Base layer class.
  73 
  74   This is the class from which all layers inherit.
  75 
  76   A layer is a class implementing common neural networks operations, such
  77   as convolution, batch norm, etc. These operations require managing weights,
  78   losses, updates, and inter-layer connectivity.
  79 
  80   Users will just instantiate a layer and then treat it as a callable.
  81 
  82   We recommend that descendants of `Layer` implement the following methods:
  83 
  84   * `__init__()`: Save configuration in member variables
  85   * `build()`: Called once from `__call__`, when we know the shapes of inputs
  86     and `dtype`. Should have the calls to `add_weight()`, and then
  87     call the super's `build()` (which sets `self.built = True`, which is
  88     nice in case the user wants to call `build()` manually before the
  89     first `__call__`).
  90   * `call()`: Called in `__call__` after making sure `build()` has been called
  91     once. Should actually perform the logic of applying the layer to the
  92     input tensors (which should be passed in as the first argument).
  93 
  94   Arguments:
  95     trainable: Boolean, whether the layer's variables should be trainable.
  96     name: String name of the layer.
  97     dtype: Default dtype of the layer's weights (default of `None` means use the
  98       type of the first input).
  99 
 100   Read-only properties:
 101     name: The name of the layer (string).
 102     dtype: Default dtype of the layer's weights (default of `None` means use the
 103       type of the first input).
 104     trainable_variables: List of trainable variables.
 105     non_trainable_variables: List of non-trainable variables.
 106     variables: List of all variables of this layer, trainable and
 107       non-trainable.
 108     updates: List of update ops of this layer.
 109     losses: List of losses added by this layer.
 110     trainable_weights: List of variables to be included in backprop.
 111     non_trainable_weights: List of variables that should not be
 112       included in backprop.
 113     weights: The concatenation of the lists trainable_weights and
 114       non_trainable_weights (in this order).
 115 
 116   Mutable properties:
 117     trainable: Whether the layer should be trained (boolean).
 118     input_spec: Optional (list of) `InputSpec` object(s) specifying the
 119       constraints on inputs that can be accepted by the layer.
 120   """
 121 
 122   @checkpointable.no_automatic_dependency_tracking
 123   def __init__(self, trainable=True, name=None, dtype=None, **kwargs):
 124     # These properties should be set by the user via keyword arguments.
 125     # note that 'dtype', 'input_shape' and 'batch_input_shape'
 126     # are only applicable to input layers: do not pass these keywords
 127     # to non-input layers.
 128     allowed_kwargs = {
 129         'input_shape',
 130         'batch_input_shape',
 131         'batch_size',
 132         'weights',
 133         'activity_regularizer',
 134     }
 135     # Validate optional keyword arguments.
 136     for kwarg in kwargs:
 137       if kwarg not in allowed_kwargs:
 138         raise TypeError('Keyword argument not understood:', kwarg)
 139 
 140     # Mutable properties
 141     # Indicates whether the layer's weights are updated during training
 142     # and whether the layer's updates are run during training
 143     self.trainable = trainable
 144     # A stateful layer is a layer whose updates are run during inference too,
 145     # for instance stateful RNNs.
 146     self.stateful = False
 147     # Indicates whether `build` needs to be called upon layer call, to create
 148     # the layer's weights.
 149     self.built = False
 150     # Provides information about which inputs are compatible with the layer.
 151     self.input_spec = None
 152 
 153     self._init_set_name(name)
 154 
 155     activity_regularizer = kwargs.pop('activity_regularizer', None)
 156     if activity_regularizer and context.executing_eagerly():
 157       raise ValueError(
 158           ('Activity regularization is not supported when executing eagerly. '
 159            'Got activity_regularizer=%s') % (activity_regularizer,))
 160     self._activity_regularizer = activity_regularizer
 161     self._trainable_weights = []
 162     self._non_trainable_weights = []
 163     self._updates = []
 164     # A list of zero-argument lambdas which return Tensors, used for variable
 165     # regularizers.
 166     self._callable_losses = []
 167     # A list of Tensors containing activity regularizers and losses manually
 168     # added through `add_loss`. Empty when executing eagerly.
 169     self._losses = []
 170     self._in_call = False  # Flag for error checking in add_loss
 171     self._dtype = None if dtype is None else dtypes.as_dtype(dtype).name
 172     self._call_fn_args = function_utils.fn_args(self.call)
 173     self._compute_previous_mask = ('mask' in self._call_fn_args or
 174                                    hasattr(self, 'compute_mask'))
 175     self._call_convention = CallConvention.EXPLICIT_INPUTS_ARGUMENT
 176 
 177     # These lists will be filled via successive calls
 178     # to self._add_inbound_node().
 179     self._inbound_nodes = []
 180     self._outbound_nodes = []
 181 
 182     self.supports_masking = False
 183 
 184     call_argspec = tf_inspect.getfullargspec(self.call)
 185     if 'training' in call_argspec.args:
 186       self._expects_training_arg = True
 187     else:
 188       self._expects_training_arg = False
 189 
 190     # Manage input shape information if passed.
 191     if 'input_shape' in kwargs or 'batch_input_shape' in kwargs:
 192       # In this case we will later create an input layer
 193       # to insert before the current layer
 194       if 'batch_input_shape' in kwargs:
 195         batch_input_shape = tuple(kwargs['batch_input_shape'])
 196       elif 'input_shape' in kwargs:
 197         if 'batch_size' in kwargs:
 198           batch_size = kwargs['batch_size']
 199         else:
 200           batch_size = None
 201         batch_input_shape = (batch_size,) + tuple(kwargs['input_shape'])
 202       self._batch_input_shape = batch_input_shape
 203 
 204     # Manage initial weight values if passed.
 205     if 'weights' in kwargs:
 206       self._initial_weights = kwargs['weights']
 207     else:
 208       self._initial_weights = None

 281   @doc_controls.for_subclass_implementers
 282   def add_update(self, updates, inputs=None):
 283     """Add update op(s), potentially dependent on layer inputs.
 284 
 285     Weight updates (for instance, the updates of the moving mean and variance
 286     in a BatchNormalization layer) may be dependent on the inputs passed
 287     when calling a layer. Hence, when reusing the same layer on
 288     different inputs `a` and `b`, some entries in `layer.updates` may be
 289     dependent on `a` and some on `b`. This method automatically keeps track
 290     of dependencies.
 291 
 292     The `get_updates_for` method allows to retrieve the updates relevant to a
 293     specific set of inputs.
 294 
 295     This call is ignored when eager execution is enabled (in that case, variable
 296     updates are run on the fly and thus do not need to be tracked for later
 297     execution).
 298 
 299     Arguments:
 300       updates: Update op, or list/tuple of update ops.
 301       inputs: If anything other than None is passed, it signals the updates
 302         are conditional on some of the layer's inputs,
 303         and thus they should only be run where these inputs are available.
 304         This is the case for BatchNormalization updates, for instance.
 305         If None, the updates will be taken into account unconditionally,
 306         and you are responsible for making sure that any dependency they might
 307         have is available at runtime.
 308         A step counter might fall into this category.
 309     """
 310     if context.executing_eagerly():
 311       return  # Updates already applied when in eager mode.
 312 
 313     def process_update(x):
 314       if isinstance(x, ops.Operation):
 315         return x
 316       elif hasattr(x, 'op'):
 317         return x.op
 318       else:
 319         return ops.convert_to_tensor(x)
 320 
 321     updates = generic_utils.to_list(updates)
 322     updates = [process_update(x) for x in updates]
 323     self._updates += updates
 324     if inputs is None:
 325       for u in updates:
 326         u._unconditional_update = True  # pylint: disable=protected-access
 327     else:
 328       for u in updates:
 329         u._unconditional_update = False  # pylint: disable=protected-access
 330 
 331   def get_updates_for(self, inputs):
 332     """Retrieves updates relevant to a specific set of inputs.
 333 
 334     Arguments:
 335       inputs: Input tensor or list/tuple of input tensors.
 336 
 337     Returns:
 338       List of update ops of the layer that depend on `inputs`.
 339 
 340     Raises:
 341       RuntimeError: If called in Eager mode.
 342     """
 343     if context.executing_eagerly():
 344       raise RuntimeError('`get_updates_for()` not supported in Eager mode.')
 345 
 346     # Updates disabled if layer is not trainable and not explicitly stateful.
 347     if not self.trainable and not self.stateful:
 348       return []
 349 
 350     if inputs is None:
 351       # Requesting unconditional updates.
 352       return [x for x in self.updates if x._unconditional_update]  # pylint: disable=protected-access
 353 
 354     # Requesting input-conditional updates.
 355     inputs = nest.flatten(inputs)
 356     reachable = tf_utils.get_reachable_from_inputs(inputs, self.updates)
 357     updates = []
 358     for update in self.updates:
 359       if update in reachable:
 360         updates.append(update)
 361     return updates

 363   @property
 364   def losses(self):
 365     """Losses which are associated with this `Layer`.
 366 
 367     Variable regularization tensors are created when this property is accessed,
 368     so it is eager safe: accessing `losses` under a `tf.GradientTape` will
 369     propagate gradients back to the corresponding variables.
 370 
 371     Returns:
 372       A list of tensors.
 373     """
 374     collected_losses = []
 375     collected_losses.extend(self._losses)
 376     for regularizer in self._callable_losses:
 377       loss_tensor = regularizer()
 378       if loss_tensor is not None:
 379         collected_losses.append(loss_tensor)
 380     return collected_losses

 382   @doc_controls.for_subclass_implementers
 383   def add_loss(self, losses, inputs=None):
 384     """Add loss tensor(s), potentially dependent on layer inputs.
 385 
 386     Some losses (for instance, activity regularization losses) may be dependent
 387     on the inputs passed when calling a layer. Hence, when reusing the same
 388     layer on different inputs `a` and `b`, some entries in `layer.losses` may
 389     be dependent on `a` and some on `b`. This method automatically keeps track
 390     of dependencies.
 391 
 392     The `get_losses_for` method allows to retrieve the losses relevant to a
 393     specific set of inputs.
 394 
 395     Note that `add_loss` is not supported when executing eagerly. Instead,
 396     variable regularizers may be added through `add_variable`. Activity
 397     regularization is not supported directly (but such losses may be returned
 398     from `Layer.call()`).
 399 
 400     Arguments:
 401       losses: Loss tensor, or list/tuple of tensors. Rather than tensors, losses
 402         may also be zero-argument callables which create a loss tensor. Only
 403         callable losses are supported when executing eagerly.
 404       inputs: If anything other than None is passed, it signals the losses
 405         are conditional on some of the layer's inputs,
 406         and thus they should only be run where these inputs are available.
 407         This is the case for activity regularization losses, for instance.
 408         If `None` is passed, the losses are assumed
 409         to be unconditional, and will apply across all dataflows of the layer
 410         (e.g. weight regularization losses).
 411 
 412     Raises:
 413       RuntimeError: If called in Eager mode with a `Tensor` rather than a
 414         callable, or if `inputs` is not None.
 415     """
 416     executing_eagerly = context.executing_eagerly()
 417     if executing_eagerly:
 418       if inputs is not None:
 419         raise RuntimeError(
 420             'Activity regularization (via the "inputs" argument to '
 421             'Layer.add_loss) is not supported when executing eagerly. Consider '
 422             'returning activity regularization losses from a Model\'s call() '
 423             'method.')
 424       if getattr(self, '_in_call', False):
 425         # TODO(psv): Support activity regularization and a way to reset losses.
 426         raise RuntimeError(
 427             'Adding losses inside a Layer\'s call() method is not currently '
 428             'supported when executing eagerly. Please file a feature request '
 429             'if you need this limitation lifted.')
 430     losses = generic_utils.to_list(losses)
 431 
 432     def _tag_unconditional(loss):
 433       if callable(loss):
 434         loss = loss()
 435       if loss is None:
 436         return None  # Will be filtered out when computing the .losses property
 437       if not tensor_util.is_tensor(loss):
 438         loss = ops.convert_to_tensor(loss, dtype=backend.floatx())
 439       loss._unconditional_loss = (inputs is None)  # pylint: disable=protected-access
 440       return loss
 441 
 442     for loss in losses:
 443       if callable(loss):
 444         self._callable_losses.append(
 445             functools.partial(_tag_unconditional, loss))
 446       else:
 447         if executing_eagerly:
 448           raise RuntimeError(
 449               'Layer.add_loss only supported for zero-argument lambdas when '
 450               'executing eagerly.')
 451         self._losses.append(_tag_unconditional(loss))
 452 
 453   def get_losses_for(self, inputs):
 454     """Retrieves losses relevant to a specific set of inputs.
 455 
 456     Arguments:
 457       inputs: Input tensor or list/tuple of input tensors.
 458 
 459     Returns:
 460       List of loss tensors of the layer that depend on `inputs`.
 461 
 462     Raises:
 463       RuntimeError: If called in Eager mode.
 464     """
 465     if context.executing_eagerly():
 466       raise RuntimeError('Layer.get_losses_for not supported in Eager mode.')
 467 
 468     if inputs is None:
 469       # Requesting unconditional losses.
 470       return [x for x in self.losses if x._unconditional_loss]  # pylint: disable=protected-access
 471 
 472     # Requesting input-conditional losses.
 473     inputs = nest.flatten(inputs)
 474     # Retrieve the set of tensors in the TF graph that depend on `inputs`.
 475     # The losses we want to return will be part of this set.
 476     # To avoid unnecessary work, we stop the search in case all of
 477     # `self.losses` have been retrieved.
 478     reachable = tf_utils.get_reachable_from_inputs(inputs, self.losses)
 479     losses = []
 480     for loss in self.losses:
 481       if loss in reachable:
 482         losses.append(loss)
 483     return losses

 497   @doc_controls.for_subclass_implementers
 498   def add_weight(self,
 499                  name,
 500                  shape,
 501                  dtype=None,
 502                  initializer=None,
 503                  regularizer=None,
 504                  trainable=None,
 505                  constraint=None,
 506                  partitioner=None,
 507                  use_resource=None,
 508                  synchronization=tf_variables.VariableSynchronization.AUTO,
 509                  aggregation=tf_variables.VariableAggregation.NONE,
 510                  **kwargs):
 511     """Adds a new variable to the layer, or gets an existing one; returns it.
 512 
 513     Arguments:
 514       name: variable name.
 515       shape: variable shape.
 516       dtype: The type of the variable. Defaults to `self.dtype` or `float32`.
 517       initializer: initializer instance (callable).
 518       regularizer: regularizer instance (callable).
 519       trainable: whether the variable should be part of the layer's
 520         "trainable_variables" (e.g. variables, biases)
 521         or "non_trainable_variables" (e.g. BatchNorm mean, stddev).
 522         Note, if the current variable scope is marked as non-trainable
 523         then this parameter is ignored and any added variables are also
 524         marked as non-trainable. `trainable` defaults to `True` unless
 525         `synchronization` is set to `ON_READ`.
 526       constraint: constraint instance (callable).
 527       partitioner: Partitioner to be passed to the `Checkpointable` API.
 528       use_resource: Whether to use `ResourceVariable`.
 529       synchronization: Indicates when a distributed a variable will be
 530         aggregated. Accepted values are constants defined in the class
 531         `tf.VariableSynchronization`. By default the synchronization is set to
 532         `AUTO` and the current `DistributionStrategy` chooses
 533         when to synchronize. If `synchronization` is set to `ON_READ`,
 534         `trainable` must not be set to `True`.
 535       aggregation: Indicates how a distributed variable will be aggregated.
 536         Accepted values are constants defined in the class
 537         `tf.VariableAggregation`.
 538       **kwargs: Additional keyword arguments. Accepted values are `getter` and
 539         `collections`.
 540 
 541     Returns:
 542       The created variable.  Usually either a `Variable` or `ResourceVariable`
 543       instance.  If `partitioner` is not `None`, a `PartitionedVariable`
 544       instance is returned.
 545 
 546     Raises:
 547       RuntimeError: If called with partioned variable regularization and
 548         eager execution is enabled.
 549       ValueError: When giving unsupported dtype and no initializer or when
 550         trainable has been set to True with synchronization set as `ON_READ`.
 551     """
 552     # Validate optional keyword arguments.
 553     for kwarg in kwargs:
 554       if kwarg not in ['getter', 'collections']:
 555         raise TypeError('Unknown keyword argument:', kwarg)
 556     getter = kwargs.pop('getter', None)
 557     collections = kwargs.pop('collections', None)
 558 
 559     if dtype is None:
 560       dtype = self.dtype or backend.floatx()
 561     dtype = dtypes.as_dtype(dtype)
 562     initializer = initializers.get(initializer)
 563     regularizer = regularizers.get(regularizer)
 564     constraint = constraints.get(constraint)
 565 
 566     if synchronization == tf_variables.VariableSynchronization.ON_READ:
 567       if trainable:
 568         raise ValueError(
 569             'Synchronization value can be set to '
 570             'VariableSynchronization.ON_READ only for non-trainable variables. '
 571             'You have specified trainable=True and '
 572             'synchronization=VariableSynchronization.ON_READ.')
 573       else:
 574         # Set trainable to be false when variable is to be synced on read.
 575         trainable = False
 576     elif trainable is None:
 577       trainable = True
 578 
 579     # Initialize variable when no initializer provided
 580     if initializer is None:
 581       # If dtype is DT_FLOAT, provide a uniform unit scaling initializer
 582       if dtype.is_floating:
 583         initializer = initializers.glorot_uniform()
 584       # If dtype is DT_INT/DT_UINT, provide a default value `zero`
 585       # If dtype is DT_BOOL, provide a default value `FALSE`
 586       elif dtype.is_integer or dtype.is_unsigned or dtype.is_bool:
 587         initializer = initializers.zeros()
 588       # NOTES:Do we need to support for handling DT_STRING and DT_COMPLEX here?
 589       else:

 664   def __call__(self, inputs, *args, **kwargs):
 665     """Wraps `call`, applying pre- and post-processing steps.
 666 
 667     Arguments:
 668       inputs: input tensor(s).
 669       *args: additional positional arguments to be passed to `self.call`.
 670       **kwargs: additional keyword arguments to be passed to `self.call`.
 671 
 672     Returns:
 673       Output tensor(s).
 674 
 675     Note:
 676       - The following optional keyword arguments are reserved for specific uses:
 677         * `training`: Boolean scalar tensor of Python boolean indicating
 678           whether the `call` is meant for training or inference.
 679         * `mask`: Boolean input mask.
 680       - If the layer's `call` method takes a `mask` argument (as some Keras
 681         layers do), its default value will be set to the mask generated
 682         for `inputs` by the previous layer (if `input` did come from
 683         a layer that generated a corresponding mask, i.e. if it came from
 684         a Keras layer with masking support.
 685 
 686     Raises:
 687       ValueError: if the layer's `call` method returns None (an invalid value).
 688     """
 689     input_list = nest.flatten(inputs)
 690 
 691     build_graph = not context.executing_eagerly()
 692     # TODO(fchollet, allenl): Make deferred mode work with subclassed Models
 693     # which don't use an "inputs" argument.
 694     in_deferred_mode = isinstance(input_list[0], DeferredTensor)
 695 
 696     # Handle Keras mask propagation from previous layer to current layer.
 697     previous_mask = None
 698     if build_graph and (not hasattr(self, '_compute_previous_mask') or
 699                         self._compute_previous_mask):
 700       previous_mask = collect_previous_mask(inputs)
 701       if not hasattr(self, '_call_fn_args'):
 702         self._call_fn_args = self._no_dependency(
 703             function_utils.fn_args(self.call))
 704       if ('mask' in self._call_fn_args and 'mask' not in kwargs and
 705           not generic_utils.is_all_none(previous_mask)):
 706         # The previous layer generated a mask, and mask was not explicitly pass
 707         # to __call__, hence we set previous_mask as the default value.
 708         kwargs['mask'] = previous_mask
 709 
 710     input_shapes = None
 711 
 712     with ops.name_scope(self._name_scope()):
 713       if not self.built:
 714         if not build_graph:
 715           # Activity regularization is currently unsupported in Eager mode.
 716           if self._activity_regularizer:
 717             raise ValueError(
 718                 'activity_regularizer currently unsupported with '
 719                 'eager execution enabled. Found an activity_regularizer in '
 720                 '%s(%s).' % (self.__class__.__name__, self))
 721         if not build_graph and not in_deferred_mode:
 722           for x in input_list:
 723             if hasattr(x, '_keras_history'):
 724               raise ValueError('_keras_history currently unsupported in '
 725                                'Eager mode. Found _keras_history in %s while '
 726                                'executing __call__ for %s(%s)' %
 727                                (x, self.__class_.__name__, self))
 728 
 729         # Check input assumptions set before layer building, e.g. input rank.
 730         self._assert_input_compatibility(inputs)
 731         if input_list and self._dtype is None:
 732           try:
 733             self._dtype = input_list[0].dtype.base_dtype.name
 734           except AttributeError:
 735             pass
 736 
 737         if all(hasattr(x, 'shape') for x in input_list):
 738           input_shapes = nest.map_structure(lambda x: x.shape, inputs)
 739 
 740         if (not hasattr(self, '_is_graph_network') or
 741             self.__class__.__name__ == 'Sequential' or
 742             not hasattr(self.build, '_is_default')):
 743           # Only if self is a layer, an instance of a sequential model, or
 744           # the user has manually overwritten the build method do we need to
 745           # build it.
 746           self.build(input_shapes)
 747         # We must set self.built since user defined build functions are not
 748         # constrained to set self.built.
 749         self.built = True
 750 
 751       # Check input assumptions set after layer building, e.g. input shape.
 752       if build_graph or in_deferred_mode:
 753         self._assert_input_compatibility(inputs)
 754 
 755       if not in_deferred_mode:
 756         self._in_call = True
 757         outputs = self.call(inputs, *args, **kwargs)
 758         self._in_call = False
 759         if outputs is None:
 760           raise ValueError('A layer\'s `call` method should return a Tensor '
 761                            'or a list of Tensors, not None (layer: ' +
 762                            self.name + ').')
 763       else:
 764         # Deferred mode behavior: use `compute_output_shape` to
 765         # infer the number of outputs of the layer and their shapes.
 766         if input_shapes is None:
 767           input_shapes = nest.map_structure(lambda x: x.shape, inputs)
 768 
 769         output_shapes = self.compute_output_shape(input_shapes)
 770         output_shapes = nest.flatten(output_shapes)
 771         outputs = [
 772             # TODO(fchollet): name the deferred tensors?
 773             DeferredTensor(shape=shape, dtype=self._dtype)
 774             for shape in output_shapes
 775         ]
 776         if len(outputs) == 1:
 777           outputs = outputs[0]
 778 
 779       if build_graph:
 780         self._handle_activity_regularization(inputs, outputs)
 781         self._set_mask_metadata(inputs, outputs, previous_mask)
 782 
 783       if in_deferred_mode or build_graph and have_all_keras_metadata(inputs):
 784         inputs, outputs = self._set_connectivity_metadata_(
 785             inputs, outputs, args, kwargs)
 786       if context.executing_eagerly():
 787         return outputs
 788 
 789       if hasattr(self, '_symbolic_set_inputs') and not self.inputs:
 790         # Subclassed network: explicitly set metadata normally set by a call to
 791         # self._set_inputs(). This is not relevant in eager execution.
 792         self._symbolic_set_inputs(inputs, outputs)
 793 
 794       if in_deferred_mode or build_graph:
 795         self._set_learning_phase_metadata(inputs, outputs)
 796 
 797     # Optionally load weight values that were specified at layer instantiation.
 798     # TODO(fchollet): consider enabling this with eager execution too.
 799     if hasattr(self, '_initial_weights') and self._initial_weights is not None:
 800       self.set_weights(self._initial_weights)
 801       del self._initial_weights
 802     return outputs

 860     call_convention = getattr(self, '_call_convention',
 861                               CallConvention.EXPLICIT_INPUTS_ARGUMENT)

 863       if call_convention == CallConvention.EXPLICIT_INPUTS_ARGUMENT:

 869       elif call_convention == CallConvention.SINGLE_POSITIONAL_ARGUMENT:
 870         raise TypeError(
 871             'This Layer takes a single positional argument to call(), which is '
 872             'by convention the inputs argument, and only this argument may be '
 873             'specified as a positional argument. Pass everything else as a '
 874             'keyword argument (those arguments will not be tracked as inputs '
 875             'to the Layer).')
 876

 878     # we copy them to avoid loss of tensor metadata.
 879     output_ls = nest.flatten(outputs)
 880     output_ls_copy = []

 882       if x in nest.flatten(inputs):

 884           x = array_ops.identity(x)
 885       output_ls_copy.append(x)

 894     # This updates the layer history of the output tensor(s).
 895     kwargs.pop('mask', None)  # `mask` should not be serialized.
 896     self._add_inbound_node(
 897         input_tensors=inputs, output_tensors=outputs, arguments=kwargs)

 900   def _inputs_from_call_args(self, call_args, call_kwargs):
 901     """Get Layer inputs from __call__ *args and **kwargs.
 902 
 903     Args:
 904       call_args: The positional arguments passed to __call__.
 905       call_kwargs: The keyword argument dict passed to __call__.
 906 
 907     Returns:
 908       A tuple of (inputs, non_input_kwargs). These may be the same objects as
 909       were passed in (call_args and call_kwargs).
 910     """
 911     call_convention = getattr(self, '_call_convention',
 912                               CallConvention.EXPLICIT_INPUTS_ARGUMENT)
 913     if (call_convention in (
 914         CallConvention.EXPLICIT_INPUTS_ARGUMENT,
 915         CallConvention.SINGLE_POSITIONAL_ARGUMENT)):
 916       assert len(call_args) == 1  # TypeError raised earlier in __call__.
 917       return call_args[0], call_kwargs
 918     else:
 919       call_arg_spec = tf_inspect.getfullargspec(self.call)
 920       # There is no explicit "inputs" argument expected or provided to
 921       # call(). Arguments which have default values are considered non-inputs,
 922       # and arguments without are considered inputs.
 923       if call_arg_spec.defaults:
 924         if call_arg_spec.varargs is not None:
 925           raise TypeError(
 926               'Layer.call() may not accept both *args and arguments with '
 927               'default values (unable to determine which are inputs to the '
 928               'Layer).')
 929         keyword_arg_names = set(
 930             call_arg_spec.args[-len(call_arg_spec.defaults):])
 931       else:
 932         keyword_arg_names = set()
 933         # Training is never an input argument name, to allow signatures like
 934         # call(x, training).
 935       keyword_arg_names.add('training')
 936       _, unwrapped_call = tf_decorator.unwrap(self.call)
 937       bound_args = inspect.getcallargs(
 938           unwrapped_call, *call_args, **call_kwargs)
 939       if call_arg_spec.varkw is not None:
 940         var_kwargs = bound_args.pop(call_arg_spec.varkw)
 941         bound_args.update(var_kwargs)
 942         keyword_arg_names = keyword_arg_names.union(var_kwargs.keys())
 943       all_args = call_arg_spec.args
 944       if all_args and bound_args[all_args[0]] is self:
 945         # Ignore the 'self' argument of methods
 946         bound_args.pop(call_arg_spec.args[0])
 947         all_args = all_args[1:]
 948       non_input_arg_values = {}
 949       input_arg_values = []
 950       remaining_args_are_keyword = False
 951       for argument_name in all_args:
 952         if argument_name in keyword_arg_names:
 953           remaining_args_are_keyword = True
 954         else:
 955           if remaining_args_are_keyword:
 956             raise TypeError(
 957                 'Found a positional argument to call() after a non-input '
 958                 'argument. All arguments after "training" must be keyword '
 959                 'arguments, and are not tracked as inputs to the Layer.')
 960         if remaining_args_are_keyword:
 961           non_input_arg_values[argument_name] = bound_args[argument_name]
 962         else:
 963           input_arg_values.append(bound_args[argument_name])
 964       if call_arg_spec.varargs is not None:
 965         input_arg_values.extend(bound_args[call_arg_spec.varargs])
 966       return input_arg_values, non_input_arg_values

 968   def compute_output_shape(self, input_shape):
 969     """Computes the output shape of the layer.
 970 
 971     Assumes that the layer will be built
 972     to match that input shape provided.
 973 
 974     Arguments:
 975         input_shape: Shape tuple (tuple of integers)
 976             or list of shape tuples (one per output tensor of the layer).
 977             Shape tuples can include None for free dimensions,
 978             instead of an integer.
 979 
 980     Returns:
 981         An input shape tuple.
 982     """
 983     if context.executing_eagerly():
 984       # In this case we build the model first in order to do shape inference.
 985       # This is acceptable because the framework only calls
 986       # `compute_output_shape` on shape values that the layer would later be
 987       # built for. It would however cause issues in case a user attempts to
 988       # use `compute_output_shape` manually (these users will have to
 989       # implement `compute_output_shape` themselves).
 990       self.build(input_shape)
 991 
 992       with context.graph_mode():

1009                                       ' Please implement the '

1431   def _assert_input_compatibility(self, inputs):
1432     """Checks compatibility between the layer and provided inputs.
1433 
1434     This checks that the tensor(s) `inputs` verify the input assumptions
1435     of the layer (if any). If not, a clear and actional exception gets raised.
1436 
1437     Arguments:
1438         inputs: input tensor or list of input tensors.
1439 
1440     Raises:
1441         ValueError: in case of mismatch between
1442             the provided inputs and the expectations of the layer.
1443     """
1444     if not self.input_spec:
1445       return
1446     if not isinstance(self.input_spec, (list, tuple)):
1447       input_spec = nest.flatten(self.input_spec)
1448     else:
1449       input_spec = self.input_spec
1450     inputs = nest.flatten(inputs)
1451     if len(inputs) != len(input_spec):
1452       raise ValueError('Layer ' + self.name + ' expects ' +
1453                        str(len(input_spec)) + ' inputs, '
1454                        'but it received ' + str(len(inputs)) +
1455                        ' input tensors. Inputs received: ' + str(inputs))
1456     for input_index, (x, spec) in enumerate(zip(inputs, input_spec)):
1457       if spec is None:
1458         continue
1459 
1460       if (spec.ndim is not None or
1461           spec.min_ndim is not None or
1462           spec.max_ndim is not None):
1463         if x.shape.ndims is None:
1464           raise ValueError('Input ' + str(input_index) + ' of layer ' +
1465                            self.name + ' is incompatible with the layer: '
1466                            'its rank is undefined, but the layer requires a '
1467                            'defined rank.')
1468 
1469       # Check ndim.
1470       if spec.ndim is not None:
1471         ndim = x.shape.ndims
1472         if ndim != spec.ndim:
1473           raise ValueError('Input ' + str(input_index) + ' of layer ' +
1474                            self.name + ' is incompatible with the layer: '
1475                            'expected ndim=' + str(spec.ndim) + ', found ndim=' +
1476                            str(ndim) + '. Full shape received: ' +
1477                            str(x.shape.as_list()))
1478       if spec.max_ndim is not None:
1479         ndim = x.shape.ndims
1480         if ndim is not None and ndim > spec.max_ndim:
1481           raise ValueError('Input ' + str(input_index) + ' of layer ' +
1482                            self.name + ' is incompatible with the layer: '
1483                            'expected max_ndim=' + str(spec.max_ndim) +
1484                            ', found ndim=' + str(ndim))
1485       if spec.min_ndim is not None:
1486         ndim = x.shape.ndims
1487         if ndim is not None and ndim < spec.min_ndim:
1488           raise ValueError('Input ' + str(input_index) + ' of layer ' +
1489                            self.name + ' is incompatible with the layer: '
1490                            ': expected min_ndim=' + str(spec.min_ndim) +
1491                            ', found ndim=' + str(ndim) +
1492                            '. Full shape received: ' +
1493                            str(x.shape.as_list()))
1494       # Check dtype.
1495       if spec.dtype is not None:
1496         if x.dtype != spec.dtype:
1497           raise ValueError('Input ' + str(input_index) + ' of layer ' +
1498                            self.name + ' is incompatible with the layer: '
1499                            'expected dtype=' + str(spec.dtype) +
1500                            ', found dtype=' + str(x.dtype))
1501       # Check specific shape axes.
1502       if spec.axes:
1503         shape = x.shape.as_list()
1504         if shape is not None:
1505           for axis, value in spec.axes.items():
1506             if hasattr(value, 'value'):
1507               value = value.value
1508             if value is not None and shape[int(axis)] not in {value, None}:
1509               raise ValueError(
1510                   'Input ' + str(input_index) + ' of layer ' + self.name + ' is'
1511                   ' incompatible with the layer: expected axis ' + str(axis) +
1512                   ' of input shape to have value ' + str(value) +
1513                   ' but received input with shape ' + str(shape))
1514       # Check shape.
1515       if spec.shape is not None:
1516         shape = x.shape.as_list()
1517         if shape is not None:
1518           for spec_dim, dim in zip(spec.shape, shape):
1519             if spec_dim is not None and dim is not None:
1520               if spec_dim != dim:
1521                 raise ValueError('Input ' + str(input_index) +
1522                                  ' is incompatible with layer ' + self.name +
1523                                  ': expected shape=' + str(spec.shape) +
1524                                  ', found shape=' + str(shape))

1657 class Node(object):
1658   """A `Node` describes the connectivity between two layers.
1659 
1660   Each time a layer is connected to some new input,
1661   a node is added to `layer._inbound_nodes`.
1662   Each time the output of a layer is used by another layer,
1663   a node is added to `layer._outbound_nodes`.
1664 
1665   Arguments:
1666       outbound_layer: the layer that takes
1667           `input_tensors` and turns them into `output_tensors`
1668           (the node gets created when the `call`
1669           method of the layer was called).
1670       inbound_layers: a list of layers, the same length as `input_tensors`,
1671           the layers from where `input_tensors` originate.
1672       node_indices: a list of integers, the same length as `inbound_layers`.
1673           `node_indices[i]` is the origin node of `input_tensors[i]`
1674           (necessary since each inbound layer might have several nodes,
1675           e.g. if the layer is being shared with a different data stream).
1676       tensor_indices: a list of integers,
1677           the same length as `inbound_layers`.
1678           `tensor_indices[i]` is the index of `input_tensors[i]` within the
1679           output of the inbound layer
1680           (necessary since each inbound layer might
1681           have multiple tensor outputs, with each one being
1682           independently manipulable).
1683       input_tensors: list of input tensors.
1684       output_tensors: list of output tensors.
1685       arguments: dictionary of keyword arguments that were passed to the
1686           `call` method of the layer at the call that created the node.
1687 
1688   `node_indices` and `tensor_indices` are basically fine-grained coordinates
1689   describing the origin of the `input_tensors`.
1690 
1691   A node from layer A to layer B is added to:
1692     - A._outbound_nodes
1693     - B._inbound_nodes
1694   """
1695 
1696   def __init__(self,
1697                outbound_layer,
1698                inbound_layers,
1699                node_indices,
1700                tensor_indices,
1701                input_tensors,
1702                output_tensors,
1703                arguments=None):
1704     # Layer instance (NOT a list).
1705     if isinstance(outbound_layer, list):
1706       raise ValueError(
1707           '`outbound_layer` should be a layer instance, not a list.')
1708     # this is the layer that takes a list of input tensors
1709     # and turns them into a list of output tensors.
1710     # the current node will be added to
1711     # the inbound_nodes of outbound_layer.
1712     self.outbound_layer = outbound_layer
1713 
1714     # The following 3 properties describe where
1715     # the input tensors come from: which layers,
1716     # and for each layer, which node and which
1717     # tensor output of each node.
1718 
1719     # List of layer instances.
1720     self.inbound_layers = inbound_layers
1721     # List of integers, 1:1 mapping with inbound_layers.
1722     self.node_indices = node_indices
1723     # List of integers, 1:1 mapping with inbound_layers.
1724     self.tensor_indices = tensor_indices
1725 
1726     # Following 2 properties:
1727     # tensor inputs and outputs of outbound_layer.
1728 
1729     # List of tensors. 1:1 mapping with inbound_layers.
1730     self.input_tensors = input_tensors
1731     # List of tensors, created by outbound_layer.call().
1732     self.output_tensors = output_tensors
1733 
1734     # Following 2 properties: input and output shapes.
1735 
1736     # List of shape tuples, shapes of input_tensors.
1737     self.input_shapes = [backend.int_shape(x) for x in input_tensors]
1738     # List of shape tuples, shapes of output_tensors.
1739     self.output_shapes = [backend.int_shape(x) for x in output_tensors]
1740 
1741     # Optional keyword arguments to layer's `call`.
1742     self.arguments = arguments
1743 
1744     # Add nodes to all layers involved.
1745     for layer in inbound_layers:
1746       if layer is not None:
1747         # For compatibility with external Keras, we use the deprecated
1748         # accessor here.
1749         layer.outbound_nodes.append(self)
1750     # For compatibility with external Keras, we use the deprecated
1751     # accessor here.
1752     outbound_layer.inbound_nodes.append(self)

1755     inbound_names = []

1757       if layer:
1758         inbound_names.append(layer.name)
1759       else:
1760         inbound_names.append(None)
1761     return {
1762         'outbound_layer': self.outbound_layer.name,
1763         'inbound_layers': inbound_names,
1764         'node_indices': self.node_indices,
1765         'tensor_indices': self.tensor_indices

1800 def unique_layer_name(name, name_uid_map=None, avoid_names=None, namespace='',
1801                       zero_based=False):
1802   """Makes a layer name (or arbitrary string) unique within a TensorFlow graph.
1803 
1804   Arguments:
1805     name: String name to make unique.
1806     name_uid_map: An optional defaultdict(int) to use when creating unique
1807       names. If None (default), uses a per-Graph dictionary.
1808     avoid_names: An optional set or dict with names which should not be used. If
1809       None (default) does not avoid any names.
1810     namespace: Gets a name which is unique within the (graph, namespace). Layers
1811       which are not Networks use a blank namespace and so get graph-global
1812       names.
1813     zero_based: If True, name sequences start with no suffix (e.g. "dense",
1814       "dense_1"). If False, naming is one-based ("dense_1", "dense_2").
1815 
1816   Returns:
1817     Unique string name.
1818 
1819   Example:
1820 
1821   ```python
1822   _unique_layer_name('dense')  # dense_1
1823   _unique_layer_name('dense')  # dense_2
1824   ```
1825   """
1826   if name_uid_map is None:
1827     name_uid_map = get_default_graph_uid_map()
1828   if avoid_names is None:
1829     avoid_names = set()
1830   proposed_name = None
1831   while proposed_name is None or proposed_name in avoid_names:
1832     name_key = (namespace, name)
1833     if zero_based:
1834       number = name_uid_map[name_key]
1835       if number:
1836         proposed_name = name + '_' + str(number)
1837       else:
1838         proposed_name = name
1839       name_uid_map[name_key] += 1
1840     else:
1841       name_uid_map[name_key] += 1
1842       proposed_name = name + '_' + str(name_uid_map[name_key])
1843   return proposed_name

1886 def make_variable(name,
1887                   shape=None,
1888                   dtype=dtypes.float32,
1889                   initializer=None,
1890                   partition_info=None,
1891                   trainable=None,
1892                   caching_device=None,
1893                   validate_shape=True,
1894                   constraint=None,
1895                   use_resource=None,
1896                   collections=None,
1897                   synchronization=tf_variables.VariableSynchronization.AUTO,
1898                   aggregation=tf_variables.VariableAggregation.NONE,
1899                   partitioner=None):  # pylint: disable=unused-argument
1900   """Temporary util to create a variable (relies on `variable_scope.variable`).
1901 
1902   Some reuse-related technicalities prevent us from using
1903   `variable_scope.get_variable()` directly, so we use a subcomponent
1904   that has fewer constraints (`variable_scope.variable()`).
1905 
1906   In the longer term, it seems like a similar "default variable creator" method
1907   should exist in `CheckpointableBase` instead. When this happens, we can get
1908   rid of this temporary solution.
1909 
1910   TODO(fchollet): remove this method when no longer needed.
1911   TODO(fchollet): handle `partitioner` argument.
1912 
1913   Arguments:
1914     name: Variable name.
1915     shape: Variable shape.
1916     dtype: The type of the variable. Defaults to `self.dtype` or `float32`.
1917     initializer: Initializer instance (callable).
1918     partition_info: Not handled at this time.
1919     trainable: Whether the variable should be part of the layer's
1920       "trainable_variables" (e.g. variables, biases)
1921       or "non_trainable_variables" (e.g. BatchNorm mean, stddev).
1922       Note, if the current variable scope is marked as non-trainable
1923       then this parameter is ignored and any added variables are also
1924       marked as non-trainable. `trainable` defaults to `True` unless
1925       `synchronization` is set to `ON_READ`.
1926     caching_device: Passed to `tf.Variable`.
1927     validate_shape: Passed to `tf.Variable`.
1928     constraint: Constraint instance (callable).
1929     use_resource: Whether to use a `ResourceVariable`.
1930     collections: List of graph collections keys. The new variable is added to
1931       these collections. Defaults to `[GraphKeys.GLOBAL_VARIABLES]`.
1932     synchronization: Indicates when a distributed a variable will be
1933       aggregated. Accepted values are constants defined in the class
1934       `tf.VariableSynchronization`. By default the synchronization is set to
1935       `AUTO` and the current `DistributionStrategy` chooses
1936       when to synchronize. If `synchronization` is set to `ON_READ`,
1937       `trainable` must not be set to `True`.
1938     aggregation: Indicates how a distributed variable will be aggregated.
1939       Accepted values are constants defined in the class
1940       `tf.VariableAggregation`.
1941     partitioner: Not handled at this time.
1942 
1943   Returns:
1944     Variable instance.
1945   """
1946   initializing_from_value = False
1947   if initializer is not None and not callable(initializer):
1948     initializing_from_value = True
1949 
1950   with ops.init_scope():
1951     if initializing_from_value:
1952       init_val = initializer
1953       variable_dtype = None
1954     else:
1955       # Instantiate initializer if provided initializer is a type object.
1956       if isinstance(initializer, type(init_ops.Initializer)):
1957         initializer = initializer(dtype=dtype)
1958       init_val = lambda: initializer(  # pylint: disable=g-long-lambda
1959           shape, dtype=dtype, partition_info=partition_info)
1960       variable_dtype = dtype.base_dtype
1961   if use_resource is None:
1962     use_resource = True
1963 
1964   # TODO(apassos,rohanj) figure out how to remove collections from here so we
1965   # can remove the V1.
1966   v = tf_variables.VariableV1(
1967       initial_value=init_val,
1968       name=name,
1969       trainable=trainable,
1970       caching_device=caching_device,
1971       dtype=variable_dtype,
1972       validate_shape=validate_shape,
1973       constraint=constraint,
1974       use_resource=use_resource,
1975       collections=collections,
1976       synchronization=synchronization,
1977       aggregation=aggregation)
1978   return v
