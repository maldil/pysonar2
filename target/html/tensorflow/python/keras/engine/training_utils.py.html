  44 def _map_nested(data, func):
  45   """Maps each nested element using func."""
  46   if isinstance(data, list):
  47     return [_map_nested(nested_data, func) for nested_data in data]
  48   elif isinstance(data, tuple):
  49     return tuple(_map_nested(nested_data, func) for nested_data in data)
  50   elif isinstance(data, dict):
  51     return {
  52         k: _map_nested(nested_data, func) for k, nested_data in data.items()
  53     }
  54   else:
  55     return func(data)

  58 def _nested_all(data, cond_func):
  59   """Checks if all elements in a nested structure satisfy cond_func."""
  60   if isinstance(data, (tuple, list)):
  61     return all([_nested_all(nested_data, cond_func) for nested_data in data])
  62   elif isinstance(data, dict):
  63     return all(
  64         [_nested_all(nested_data, cond_func) for nested_data in data.values()])
  65   else:
  66     return cond_func(data)
  67

  69 def _nested_any(data, cond_func):
  70   """Checks if any nested_elements in a nested structure satisfy cond_func."""
  71   if isinstance(data, (tuple, list)):
  72     return any([_nested_any(nested_data, cond_func) for nested_data in data])
  73   elif isinstance(data, dict):
  74     return any(
  75         [_nested_any(nested_data, cond_func) for nested_data in data.values()])
  76   else:
  77     return cond_func(data)

  80 def _convert_lists_to_tuples(data):
  81   """Converts all lists to tuples, since Datasets expect tuples."""
  82   if isinstance(data, (tuple, list)):
  83     return tuple(_convert_lists_to_tuples(nested_data) for nested_data in data)
  84   elif isinstance(data, dict):
  85     return {
  86         k: _convert_lists_to_tuples(nested_data)
  87         for k, nested_data in data.items()
  88     }
  89   else:
  90     return data

  93 def _get_batch_axis_size(data):
  94   """Returns batch axis shape for nested data."""
  95   if isinstance(data, (tuple, list)):

  99   else:

 103 def convert_to_iterator(x=None,
 104                         y=None,
 105                         sample_weights=None,
 106                         batch_size=None,
 107                         steps_per_epoch=None,
 108                         epochs=1,
 109                         shuffle=False):
 110   """Converts NumPy arrays or EagerTensors to an EagerIterator.
 111 
 112   Combines all provided data into a single EagerIterator.
 113 
 114   Arguments:
 115       x: NumPy array or EagerTensor,  or list of Numpy arrays or EagerTensors
 116         representing inputs to a model.
 117       y: Optional. NumPy array or EagerTensor, or list of Numpy arrays or
 118         EagerTensors representing targets of a model.
 119       sample_weights: Optional NumPy array or EagerTensor representing sample
 120         weights.
 121       batch_size: Used to batch data and calculate how many steps EagerIterator
 122         should take per epoch.
 123       steps_per_epoch: If provided, how many steps EagerIterator should take per
 124         epoch.
 125       epochs: Epochs to repeat iterator for.
 126       shuffle: Whether to shuffle data after each epoch.
 127 
 128   Raises:
 129       ValueError: if steps_per_epoch cannot be calculated from the data
 130       provided.
 131 
 132   Returns:
 133       (Iterator, steps_per_epoch).
 134 
 135   """
 136   if isinstance(x, iterator_ops.EagerIterator):
 137     return x, steps_per_epoch
 138 
 139   if not _nested_any(sample_weights, lambda x: x is None):
 140     data = (x, y, sample_weights)
 141   elif not _nested_any(y, lambda x: x is None):
 142     data = (x, y)
 143   else:
 144     # always wrap in a tuple, so we know y, sample_weights weren't set
 145     # even when x has multiple elements
 146     data = (x,)
 147 
 148   data = _convert_lists_to_tuples(data)
 149   if steps_per_epoch is None and batch_size is not None:
 150     num_samples = _get_batch_axis_size(data)
 151     steps_per_epoch = int(math.ceil(num_samples / batch_size))
 152 
 153   if steps_per_epoch is None:
 154     raise ValueError('Could not determine steps_per_epoch.'
 155                      'Please provide either batch_size or'
 156                      'steps_per_epoch.')
 157 
 158   # TODO(omalleyt) for NumPy arrays in graph mode
 159   # placeholder ops should be used
 160   # this is only ideal for eager mode
 161   dataset = dataset_ops.Dataset.from_tensor_slices(data)
 162 
 163   if batch_size is not None:
 164     dataset = dataset.batch(batch_size)
 165   if shuffle:
 166     dataset = dataset.shuffle(buffer_size=10000)
 167   dataset = dataset.repeat(epochs)
 168   iterator = dataset.make_one_shot_iterator()
 169 
 170   return iterator, steps_per_epoch

 173 def check_num_samples(ins,
 174                       batch_size=None,
 175                       steps=None,
 176                       steps_name='steps'):
 177   """Determine the number of samples provided for training and evaluation.
 178 
 179   The number of samples is not defined when running with `steps`,
 180   in which case the number of samples is set to `None`.
 181 
 182   Arguments:
 183       ins: List of tensors to be fed to the Keras function.
 184       batch_size: Integer batch size or `None` if not defined.
 185       steps: Total number of steps (batches of samples)
 186           before declaring `_predict_loop` finished.
 187           Ignored with the default value of `None`.
 188       steps_name: The public API's parameter name for `steps`.
 189 
 190   Raises:
 191       ValueError: when `steps` is `None` and the attribute `ins.shape`
 192       does not exist. Also raises ValueError when `steps` is not `None`
 193       and `batch_size` is not `None` because they are mutually
 194       exclusive.
 195 
 196   Returns:
 197       When steps is `None`, returns the number of samples to be
 198       processed based on the size of the first dimension of the
 199       first input numpy array. When steps is not `None` and
 200       `batch_size` is `None`, returns `None`.
 201 
 202   Raises:
 203       ValueError: In case of invalid arguments.
 204   """
 205   if steps is not None and batch_size is not None:
 206     raise ValueError(
 207         'If ' + steps_name + ' is set, the `batch_size` must be None.')
 208   if check_steps_argument(ins, steps, steps_name):
 209     return None
 210   if hasattr(ins[0], 'shape'):
 211     return int(ins[0].shape[0])
 212   return None  # Edge case where ins == [static_learning_phase]

 215 def standardize_single_array(x):
 216   if x is None:
 217     return None
 218   if x.shape is not None and len(x.shape) == 1:
 219     if tensor_util.is_tensor(x):
 220       return array_ops.expand_dims(x, axis=1)
 221     else:
 222       return np.expand_dims(x, 1)
 223   return x

 226 def standardize_input_data(data,
 227                            names,
 228                            shapes=None,
 229                            check_batch_axis=True,
 230                            exception_prefix=''):
 231   """Normalizes inputs and targets provided by users.
 232 
 233   Users may pass data as a list of arrays, dictionary of arrays,
 234   or as a single array. We normalize this to an ordered list of
 235   arrays (same order as `names`), while checking that the provided
 236   arrays have shapes that match the network's expectations.
 237 
 238   Arguments:
 239       data: User-provided input data (polymorphic).
 240       names: List of expected array names.
 241       shapes: Optional list of expected array shapes.
 242       check_batch_axis: Boolean; whether to check that
 243           the batch axis of the arrays matches the expected
 244           value found in `shapes`.
 245       exception_prefix: String prefix used for exception formatting.
 246 
 247   Returns:
 248       List of standardized input arrays (one array per model input).
 249 
 250   Raises:
 251       ValueError: in case of improperly formatted user-provided data.
 252   """
 253   if not names:
 254     if (data is not None and hasattr(data, '__len__') and len(data) and
 255         not isinstance(data, dict)):
 256       raise ValueError('Error when checking model ' + exception_prefix + ': '
 257                        'expected no data, but got:', data)
 258     return []
 259   if data is None:
 260     return [None for _ in range(len(names))]
 261 
 262   if isinstance(data, dict):
 263     try:
 264       data = [
 265           data[x].values
 266           if data[x].__class__.__name__ == 'DataFrame' else data[x]
 267           for x in names
 268       ]
 269     except KeyError as e:
 270       raise ValueError('No data provided for "' + e.args[0] + '". Need data '
 271                        'for each key in: ' + str(names))
 272   elif isinstance(data, (list, tuple)):
 273     if isinstance(data[0], (list, tuple)):
 274       data = [np.asarray(d) for d in data]
 275     elif len(names) == 1 and isinstance(data[0], (float, int)):
 276       data = [np.asarray(data)]
 277     else:
 278       data = [
 279           x.values if x.__class__.__name__ == 'DataFrame' else x for x in data
 280       ]
 281   else:
 282     data = data.values if data.__class__.__name__ == 'DataFrame' else data
 283     data = [data]
 284   data = [standardize_single_array(x) for x in data]
 285 
 286   if len(data) != len(names):
 287     if data and hasattr(data[0], 'shape'):
 288       raise ValueError('Error when checking model ' + exception_prefix +
 289                        ': the list of Numpy arrays that you are passing to '
 290                        'your model is not the size the model expected. '
 291                        'Expected to see ' + str(len(names)) + ' array(s), '
 292                        'but instead got the following list of ' +
 293                        str(len(data)) + ' arrays: ' + str(data)[:200] + '...')
 294     elif len(names) > 1:
 295       raise ValueError(
 296           'Error when checking model ' + exception_prefix +
 297           ': you are passing a list as input to your model, '
 298           'but the model expects a list of ' + str(len(names)) +
 299           ' Numpy arrays instead. The list you passed was: ' + str(data)[:200])
 300     elif len(data) == 1 and not hasattr(data[0], 'shape'):
 301       raise TypeError('Error when checking model ' + exception_prefix +
 302                       ': data should be a Numpy array, or list/dict of '
 303                       'Numpy arrays. Found: ' + str(data)[:200] + '...')
 304     elif len(names) == 1:
 305       data = [np.asarray(data)]
 306 
 307   # Check shapes compatibility.
 308   if shapes:
 309     for i in range(len(names)):
 310       if shapes[i] is not None:
 311         if tensor_util.is_tensor(data[i]):
 312           tensorshape = data[i].get_shape()
 313           if not tensorshape:
 314             continue
 315           data_shape = tuple(tensorshape.as_list())
 316         else:
 317           data_shape = data[i].shape
 318         shape = shapes[i]
 319         if len(data_shape) != len(shape):
 320           raise ValueError('Error when checking ' + exception_prefix +
 321                            ': expected ' + names[i] + ' to have ' +
 322                            str(len(shape)) + ' dimensions, but got array '
 323                            'with shape ' + str(data_shape))
 324         if not check_batch_axis:
 325           data_shape = data_shape[1:]
 326           shape = shape[1:]
 327         for dim, ref_dim in zip(data_shape, shape):
 328           if ref_dim != dim and ref_dim is not None and dim is not None:
 329             raise ValueError(
 330                 'Error when checking ' + exception_prefix + ': expected ' +
 331                 names[i] + ' to have shape ' + str(shape) +
 332                 ' but got array with shape ' + str(data_shape))
 333   return data

 336 def standardize_sample_or_class_weights(x_weight, output_names, weight_type):
 337   """Maps `sample_weight` or `class_weight` to model outputs.
 338 
 339   Arguments:
 340       x_weight: User-provided `sample_weight` or `class_weight` argument.
 341       output_names: List of output names (strings) in the model.
 342       weight_type: A string used purely for exception printing.
 343 
 344   Returns:
 345       A list of `sample_weight` or `class_weight` where there are exactly
 346           one element per model output.
 347 
 348   Raises:
 349       ValueError: In case of invalid user-provided argument.
 350   """
 351   if x_weight is None or (isinstance(x_weight, list) and len(x_weight) == 0):  # pylint: disable=g-explicit-length-test
 352     return [None for _ in output_names]
 353   if len(output_names) == 1:
 354     if isinstance(x_weight, list) and len(x_weight) == 1:
 355       return x_weight
 356     if isinstance(x_weight, dict) and output_names[0] in x_weight:
 357       return [x_weight[output_names[0]]]
 358     else:
 359       return [x_weight]
 360   if isinstance(x_weight, list):
 361     if len(x_weight) != len(output_names):
 362       raise ValueError('Provided `' + weight_type + '` was a list of ' +
 363                        str(len(x_weight)) + ' elements, but the model has ' +
 364                        str(len(output_names)) + ' outputs. '
 365                        'You should provide one `' + weight_type + '`'
 366                        'array per model output.')
 367     return x_weight
 368   if isinstance(x_weight, dict):
 369     x_weights = []
 370     for name in output_names:
 371       x_weights.append(x_weight.get(name))
 372     return x_weights
 373   else:
 374     raise TypeError(
 375         'The model has multiple outputs, so `' + weight_type + '` '
 376         'should be either a list or a dict. '
 377         'Provided `' + weight_type + '` type not understood: ' + str(x_weight))
 378

 382                                              'class_weight')

 390 def check_array_lengths(inputs, targets, weights=None):
 391   """Does user input validation for numpy arrays.
 392 
 393   Arguments:
 394       inputs: list of Numpy arrays of inputs.
 395       targets: list of Numpy arrays of targets.
 396       weights: list of Numpy arrays of sample weights.
 397 
 398   Raises:
 399       ValueError: in case of incorrectly formatted data.
 400   """
 401 
 402   def set_of_lengths(x):
 403     # Returns a set with the variation between
 404     # different shapes, with None => 0
 405     if x is None:
 406       return {}
 407     else:
 408       return set([y.shape[0] for y in x
 409                   if y is not None and not tensor_util.is_tensor(y)])
 410 
 411   set_x = set_of_lengths(inputs)
 412   set_y = set_of_lengths(targets)
 413   set_w = set_of_lengths(weights)
 414   if len(set_x) > 1:
 415     raise ValueError('All input arrays (x) should have '
 416                      'the same number of samples. Got array shapes: ' +
 417                      str([x.shape for x in inputs]))
 418   if len(set_y) > 1:
 419     raise ValueError('All target arrays (y) should have '
 420                      'the same number of samples. Got array shapes: ' +
 421                      str([y.shape for y in targets]))
 422   if set_x and set_y and list(set_x)[0] != list(set_y)[0]:
 423     raise ValueError('Input arrays should have '
 424                      'the same number of samples as target arrays. '
 425                      'Found ' + str(list(set_x)[0]) + ' input samples '
 426                      'and ' + str(list(set_y)[0]) + ' target samples.')
 427   if len(set_w) > 1:
 428     raise ValueError('All sample_weight arrays should have '
 429                      'the same number of samples. Got array shapes: ' +
 430                      str([w.shape for w in weights]))
 431   if set_y and set_w and list(set_y)[0] != list(set_w)[0]:
 432     raise ValueError('Sample_weight arrays should have '
 433                      'the same number of samples as target arrays. Got ' +
 434                      str(list(set_y)[0]) + ' input samples and ' +
 435                      str(list(set_w)[0]) + ' target samples.')

 438 def check_loss_and_target_compatibility(targets, loss_fns, output_shapes):
 439   """Does validation on the compatibility of targets and loss functions.
 440 
 441   This helps prevent users from using loss functions incorrectly. This check
 442   is purely for UX purposes.
 443 
 444   Arguments:
 445       targets: list of Numpy arrays of targets.
 446       loss_fns: list of loss functions.
 447       output_shapes: list of shapes of model outputs.
 448 
 449   Raises:
 450       ValueError: if a loss function or target array
 451           is incompatible with an output.
 452   """
 453   key_losses = {
 454       losses.mean_squared_error, losses.binary_crossentropy,
 455       losses.categorical_crossentropy
 456   }
 457   for y, loss, shape in zip(targets, loss_fns, output_shapes):
 458     if y is None or loss is None or tensor_util.is_tensor(y):
 459       continue
 460     if loss is losses.categorical_crossentropy:
 461       if y.shape[-1] == 1:
 462         raise ValueError('You are passing a target array of shape ' + str(
 463             y.shape) + ' while using as loss `categorical_crossentropy`. '
 464                          '`categorical_crossentropy` expects '
 465                          'targets to be binary matrices (1s and 0s) '
 466                          'of shape (samples, classes). '
 467                          'If your targets are integer classes, '
 468                          'you can convert them to the expected format via:\n'
 469                          '```\n'
 470                          'from keras.utils import to_categorical\n'
 471                          'y_binary = to_categorical(y_int)\n'
 472                          '```\n'
 473                          '\n'
 474                          'Alternatively, you can use the loss function '
 475                          '`sparse_categorical_crossentropy` instead, '
 476                          'which does expect integer targets.')
 477     if loss in key_losses:
 478       for target_dim, out_dim in zip(y.shape[1:], shape[1:]):
 479         if out_dim is not None and target_dim != out_dim:
 480           raise ValueError('A target array with shape ' + str(y.shape) +
 481                            ' was passed for an output of shape ' + str(shape) +
 482                            ' while using as loss `' + loss.__name__ + '`. '
 483                            'This loss expects '
 484                            'targets to have the same shape '
 485                            'as the output.')

 488 def collect_per_output_metric_info(metrics,
 489                                    output_names,
 490                                    output_shapes,
 491                                    loss_fns,
 492                                    sample_weights=None):
 493   """Maps metric names and functions to model outputs.
 494 
 495   Arguments:
 496       metrics: a list or dict of metric functions.
 497       output_names: a list of the names (strings) of model outputs.
 498       output_shapes: a list of the shapes (strings) of model outputs.
 499       loss_fns: a list of the loss functions corresponding to the model outputs.
 500       sample_weights: a list of weights to be applied on the model outputs.
 501 
 502   Returns:
 503       A list (one entry per model output) of dicts.
 504       For instance, if the model has 2 outputs, and for the first output
 505       we want to compute "binary_accuracy" and "binary_crossentropy",
 506       and just "binary_accuracy" for the second output,
 507       the list would look like: `[[('acc', binary_accuracy()),
 508       ('ce', binary_crossentropy())], [('acc', binary_accuracy())]]`
 509 
 510   Raises:
 511       TypeError: if an incorrect type is passed for the `metrics` argument.
 512   """
 513   if not metrics:
 514     return [{} for _ in output_names]
 515   if isinstance(metrics, list):
 516     # we then apply all metrics to all outputs.
 517     nested_metrics = [copy.copy(metrics) for _ in output_names]
 518   elif isinstance(metrics, dict):
 519     nested_metrics = []
 520     for name in output_names:
 521       output_metrics = metrics.get(name, [])
 522       if not isinstance(output_metrics, list):
 523         output_metrics = [output_metrics]
 524       nested_metrics.append(output_metrics)
 525   else:
 526     raise TypeError('Type of `metrics` argument not understood. '
 527                     'Expected a list or dictionary, found: ' + str(metrics))
 528 
 529   per_output_metrics = []
 530   for i, metrics in enumerate(nested_metrics):
 531     metrics_dict = OrderedDict()
 532     for metric in metrics:
 533       weighted = False if (sample_weights is None) else (
 534           sample_weights[i] is not None)
 535       metric_name = get_metric_name(metric, weighted)
 536       metric_fn = get_metric_function(
 537           metric, output_shape=output_shapes[i], loss_fn=loss_fns[i])
 538       metrics_dict[metric_name] = metric_fn
 539     per_output_metrics.append(metrics_dict)
 540 
 541   return per_output_metrics

 544 def batch_shuffle(index_array, batch_size):
 545   """Shuffles an array in a batch-wise fashion.
 546 
 547   Useful for shuffling HDF5 arrays
 548   (where one cannot access arbitrary indices).
 549 
 550   Arguments:
 551       index_array: array of indices to be shuffled.
 552       batch_size: integer.
 553 
 554   Returns:
 555       The `index_array` array, shuffled in a batch-wise fashion.
 556   """
 557   batch_count = int(len(index_array) / batch_size)
 558   # to reshape we need to be cleanly divisible by batch size
 559   # we stash extra items and reappend them after shuffling
 560   last_batch = index_array[batch_count * batch_size:]
 561   index_array = index_array[:batch_count * batch_size]
 562   index_array = index_array.reshape((batch_count, batch_size))
 563   np.random.shuffle(index_array)
 564   index_array = index_array.flatten()
 565   return np.append(index_array, last_batch)

 568 def weighted_masked_objective(fn):
 569   """Adds support for masking and sample-weighting to an objective function.
 570 
 571   It transforms an objective function `fn(y_true, y_pred)`
 572   into a sample-weighted, cost-masked objective function
 573   `fn(y_true, y_pred, weights, mask)`.
 574 
 575   Arguments:
 576       fn: The objective function to wrap,
 577           with signature `fn(y_true, y_pred)`.
 578 
 579   Returns:
 580       A function with signature `fn(y_true, y_pred, weights, mask)`.
 581   """
 582   if fn is None:
 583     return None
 584 
 585   def weighted(y_true, y_pred, weights, mask=None):
 586     """Wrapper function.
 587 
 588     Arguments:
 589         y_true: `y_true` argument of `fn`.
 590         y_pred: `y_pred` argument of `fn`.
 591         weights: Weights tensor.
 592         mask: Mask tensor.
 593 
 594     Returns:
 595         Scalar tensor.
 596     """
 597     # score_array has ndim >= 2
 598     score_array = fn(y_true, y_pred)
 599     if mask is not None:
 600       mask = math_ops.cast(mask, y_pred.dtype)
 601       # Update weights with mask.
 602       if weights is None:
 603         weights = mask
 604       else:
 605         # Update shape of weights if possible before adding mask.
 606         # Update dimensions of weights to match with mask if possible.
 607         mask, _, weights = metrics_module.squeeze_or_expand_dimensions(
 608             mask, None, weights)
 609         try:
 610           # Broadcast weights if possible.
 611           weights = weights_broadcast_ops.broadcast_weights(weights, mask)
 612           weights *= mask
 613         except ValueError:
 614           score_array *= mask
 615           score_array /= K.mean(mask)
 616           # TODO(psv): Handle case when mask and weight shapes are not
 617           # compatible.
 618 
 619     # Apply sample weighting.
 620     if weights is not None:
 621 
 622       # Update dimensions of weights to match with values if possible.
 623       score_array, _, weights = metrics_module.squeeze_or_expand_dimensions(
 624           score_array, None, weights)
 625       try:
 626         # Broadcast weights if possible.
 627         weights = weights_broadcast_ops.broadcast_weights(weights, score_array)
 628       except ValueError:
 629         # Reduce values to same ndim as weight array.
 630         ndim = K.ndim(score_array)
 631         weight_ndim = K.ndim(weights)
 632         score_array = K.mean(score_array, axis=list(range(weight_ndim, ndim)))
 633 
 634       score_array = math_ops.multiply(score_array, weights)
 635       score_array = math_ops.reduce_sum(score_array)
 636       weights = math_ops.reduce_sum(weights)
 637       score_array = metrics_module.safe_div(score_array, weights)
 638     return K.mean(score_array)
 639 
 640   return weighted

 643 def standardize_weights(y,
 644                         sample_weight=None,
 645                         class_weight=None,
 646                         sample_weight_mode=None):
 647   """Performs sample weight validation and standardization.
 648 
 649   Everything gets normalized to a single sample-wise (or timestep-wise)
 650   weight array.
 651 
 652   Arguments:
 653       y: Numpy array of model targets to be weighted.
 654       sample_weight: User-provided `sample_weight` argument.
 655       class_weight: User-provided `class_weight` argument.
 656       sample_weight_mode: One of `None` or `"temporal"`.
 657           `"temporal"` indicated that we expect 2D weight data
 658           that will be applied to the last 2 dimensions of
 659           the targets (i.e. we are weighting timesteps, not samples).
 660 
 661   Returns:
 662       A numpy array of target weights, one entry per sample to weight.
 663 
 664   Raises:
 665       ValueError: In case of invalid user-provided arguments.
 666   """
 667   # Iterator may return sample_weight as 1-tuple
 668   if isinstance(sample_weight, tuple):
 669     sample_weight = sample_weight[0]
 670   if sample_weight_mode is not None:
 671     if sample_weight_mode != 'temporal':
 672       raise ValueError('"sample_weight_mode '
 673                        'should be None or "temporal". '
 674                        'Found: ' + str(sample_weight_mode))
 675     if len(y.shape) < 3:
 676       raise ValueError('Found a sample_weight array for '
 677                        'an input with shape ' + str(y.shape) + '. '
 678                        'Timestep-wise sample weighting (use of '
 679                        'sample_weight_mode="temporal") is restricted to '
 680                        'outputs that are at least 3D, i.e. that have '
 681                        'a time dimension.')
 682     if sample_weight is not None and len(sample_weight.shape) != 2:
 683       raise ValueError('Found a sample_weight array with shape ' +
 684                        str(sample_weight.shape) + '. '
 685                        'In order to use timestep-wise sample weighting, '
 686                        'you should pass a 2D sample_weight array.')
 687   else:
 688     if sample_weight is not None and len(sample_weight.shape) != 1:
 689       raise ValueError('Found a sample_weight array with shape ' +
 690                        str(sample_weight.shape) + '. '
 691                        'In order to use timestep-wise sample weights, '
 692                        'you should specify '
 693                        'sample_weight_mode="temporal" '
 694                        'in compile(). If you just mean to use '
 695                        'sample-wise weights, make sure your '
 696                        'sample_weight array is 1D.')
 697 
 698   if sample_weight is not None:
 699     if len(sample_weight.shape) > len(y.shape):
 700       raise ValueError(
 701           'Found a sample_weight with shape' + str(sample_weight.shape) + '.'
 702           'Expected sample_weight with rank '
 703           'less than or equal to ' + str(len(y.shape)))
 704 
 705     if (not tensor_util.is_tensor(sample_weight) and
 706         y.shape[:sample_weight.ndim] != sample_weight.shape):
 707       raise ValueError(
 708           'Found a sample_weight array with shape ' + str(sample_weight.shape) +
 709           ' for an input with shape ' + str(y.shape) + '. '
 710           'sample_weight cannot be broadcast.')
 711     return sample_weight
 712   elif isinstance(class_weight, dict):
 713     if len(y.shape) > 2:
 714       raise ValueError('`class_weight` not supported for '
 715                        '3+ dimensional targets.')
 716     if y.shape[1] > 1:
 717       y_classes = np.argmax(y, axis=1)
 718     elif y.shape[1] == 1:
 719       y_classes = np.reshape(y, y.shape[0])
 720     else:
 721       y_classes = y
 722 
 723     weights = np.asarray(
 724         [class_weight[cls] for cls in y_classes if cls in class_weight])
 725 
 726     if len(weights) != len(y_classes):
 727       # subtract the sets to pick all missing classes
 728       existing_classes = set(y_classes)
 729       existing_class_weight = set(class_weight.keys())
 730       raise ValueError('`class_weight` must contain all classes in the data.'
 731                        ' The classes %s exist in the data but not in '
 732                        '`class_weight`.' %
 733                        (existing_classes - existing_class_weight))
 734     return weights
 735   else:
 736     return None

 753 def get_metric_name(metric, weighted=False):
 754   """Returns the name corresponding to the given metric input.
 755 
 756   Arguments:
 757     metric: Metric function name or reference.
 758     weighted: Boolean indicating if the given metric is weighted.
 759 
 760   Returns:
 761       The metric name.
 762   """
 763   metric_name_prefix = 'weighted_' if weighted else ''
 764   if metric in ('accuracy', 'acc', 'crossentropy', 'ce'):
 765     if metric in ('accuracy', 'acc'):
 766       suffix = 'acc'
 767     elif metric in ('crossentropy', 'ce'):
 768       suffix = 'ce'
 769   else:
 770     metric_fn = metrics_module.get(metric)
 771     # Get metric name as string
 772     if hasattr(metric_fn, 'name'):
 773       suffix = metric_fn.name
 774     else:
 775       suffix = metric_fn.__name__
 776   metric_name = metric_name_prefix + suffix
 777   return metric_name

 780 def get_metric_function(metric, output_shape=None, loss_fn=None):
 781   """Returns the metric function corresponding to the given metric input.
 782 
 783   Arguments:
 784       metric: Metric function name or reference.
 785       output_shape: The shape of the output that this metric
 786           will be calculated for.
 787       loss_fn: The loss function used.
 788 
 789   Returns:
 790       The metric function.
 791   """
 792   if metric in ['accuracy', 'acc']:
 793     if output_shape[-1] == 1 or loss_fn == losses.binary_crossentropy:
 794       return metrics_module.binary_accuracy  # case: binary accuracy
 795     elif loss_fn == losses.sparse_categorical_crossentropy:
 796       # case: categorical accuracy with sparse targets
 797       return metrics_module.sparse_categorical_accuracy
 798     return metrics_module.categorical_accuracy  # case: categorical accuracy
 799   elif metric in ['crossentropy', 'ce']:
 800     if output_shape[-1] == 1 or loss_fn == losses.binary_crossentropy:
 801       return metrics_module.binary_crossentropy  # case: binary cross-entropy
 802     elif loss_fn == losses.sparse_categorical_crossentropy:
 803       # case: categorical cross-entropy with sparse targets
 804       return metrics_module.sparse_categorical_crossentropy
 805     # case: categorical cross-entropy
 806     return metrics_module.categorical_crossentropy
 807   return metrics_module.get(metric)

 860 def check_steps_argument(input_data, steps, steps_name):
 861   """Validates `steps` argument based on input data's type.
 862 
 863   The cases when `steps` value must be provided are when
 864     1. input data passed is an iterator.
 865     2. model was built on top of symbolic tensors, input data is not
 866        required and is `None`.
 867     3. input data passed is a symbolic tensor.
 868 
 869   Arguments:
 870       input_data: Input data. Can be Numpy array(s) or TensorFlow tensor(s) or
 871         tf.data.Dataset iterator or `None`.
 872       steps: Integer or `None`. Total number of steps (batches of samples) to
 873         execute.
 874       steps_name: The public API's parameter name for `steps`.
 875 
 876   Returns:
 877     boolean, True if `steps` argument is required, else False.
 878 
 879   Raises:
 880       ValueError: if `steps` argument is required for given input data type
 881         but not provided.
 882   """
 883 
 884   is_x_iterator = (
 885       isinstance(input_data, iterator_ops.Iterator) or
 886       isinstance(input_data, iterator_ops.EagerIterator))
 887 
 888   if (input_data is None or is_x_iterator or has_symbolic_tensors(input_data) or
 889       (isinstance(input_data, list) and not input_data)):
 890     if steps is None:
 891       input_type_str = 'iterators' if is_x_iterator else 'data tensors'
 892       raise ValueError('When using {input_type} as input to a model, you should'
 893                        ' specify the `{steps_name}` argument.'.format(
 894                            input_type=input_type_str, steps_name=steps_name))
 895     return True
 896   return False

 925 def get_output_sample_weight_and_mode(skip_target_weighing_indices,
 926                                       sample_weight_mode, output_name,
 927                                       output_index):
 928   """Returns the sample weight and weight mode for a single output."""
 929   if output_index in skip_target_weighing_indices:
 930     return None, None
 931 
 932   if sample_weight_mode == 'temporal':
 933     default_value = [[1.]]
 934     shape = [None, None]
 935     mode = 'temporal'
 936   else:
 937     default_value = [1.]
 938     shape = [None]
 939     mode = None
 940   if context.executing_eagerly():
 941     weight = None
 942   else:
 943     weight = array_ops.placeholder_with_default(
 944         constant_op.constant(default_value, dtype=K.floatx()),
 945         shape=shape,
 946         name=output_name + '_sample_weights')
 947   return weight, mode

 950 def prepare_sample_weights(output_names, sample_weight_mode,
 951                            skip_target_weighing_indices):
 952   """Prepares sample weights for the model.
 953 
 954   Args:
 955     output_names: List of model output names.
 956     sample_weight_mode: sample weight mode user input passed from compile API.
 957     skip_target_weighing_indices: Indices of output for which sample weights
 958       should be skipped.
 959 
 960   Returns:
 961     A pair of list of sample weights and sample weight modes
 962       (one for each output).
 963 
 964   Raises:
 965     ValueError: In case of invalid `sample_weight_mode` input.
 966   """
 967   sample_weights = []
 968   sample_weight_modes = []
 969   if isinstance(sample_weight_mode, dict):
 970     unknown_output = set(sample_weight_mode.keys()) - set(output_names)
 971     if unknown_output:
 972       raise ValueError('Unknown entry in '
 973                        'sample_weight_mode dictionary: "' + unknown_output +
 974                        '". Only expected the following keys: ' +
 975                        str(output_names))
 976     for i, name in enumerate(output_names):
 977       if (i not in skip_target_weighing_indices and
 978           name not in sample_weight_mode):
 979         raise ValueError('Output missing from sample_weight_modes dictionary')
 980       weight, mode = get_output_sample_weight_and_mode(
 981           skip_target_weighing_indices, sample_weight_mode.get(name), name, i)
 982       sample_weights.append(weight)
 983       sample_weight_modes.append(mode)
 984   elif isinstance(sample_weight_mode, list):
 985     if len(sample_weight_mode) != len(output_names):
 986       raise ValueError('When passing a list as sample_weight_mode, '
 987                        'it should have one entry per model output. '
 988                        'The model has ' + str(len(output_names)) +
 989                        ' outputs, but you passed ' +
 990                        str(len(sample_weight_mode)) + 'sample_weight_modes')
 991     for i, name in enumerate(output_names):
 992       weight, mode = get_output_sample_weight_and_mode(
 993           skip_target_weighing_indices, sample_weight_mode[i], name, i)
 994       sample_weights.append(weight)
 995       sample_weight_modes.append(mode)
 996   else:
 997     for i, name in enumerate(output_names):
 998       weight, mode = get_output_sample_weight_and_mode(
 999           skip_target_weighing_indices, sample_weight_mode, name, i)
1000       sample_weights.append(weight)
1001       sample_weight_modes.append(mode)
1002   return sample_weights, sample_weight_modes

1012 class ModelInputs(object):
1013   """Encapsulates model inputs.
1014 
1015   Allows for transforming model inputs while keeping the same structure.
1016   """
1017 
1018   def __init__(self, inputs):
1019     self._inputs = inputs
1020     self._is_dict = isinstance(self._inputs, dict)
1021     self._is_single_input = not isinstance(self._inputs, (list, tuple, dict))
1022     self._flattened_inputs = []
1023     self._input_names = []
1024     if isinstance(self._inputs, dict):
1025       for k in sorted(self._inputs.keys()):
1026         self._flattened_inputs.append(self._inputs[k])
1027         self._input_names.append(k)
1028     else:
1029       self._flattened_inputs = nest.flatten(self._inputs)
1030       self._input_names = [
1031           'input_%d' % (i + 1) for i in range(len(self._flattened_inputs))
1032       ]
1033     assert len(self._input_names) == len(self._flattened_inputs)
1034 
1035   def get_input_names(self):
1036     """Returns keys to name inputs by.
1037 
1038     In case inputs provided were a list, tuple or single entry, we make up a
1039     key 'input_%d'. For dictionary case, we return a sorted list of keys.
1040     """
1041     return self._input_names
1042 
1043   def _get(self, return_single_as_list=False):
1044     """Returns provided inputs, potentially transformed.
1045 
1046     Inputs are returned in the same format they were provided i.e. lists
1047     are returned as lists, single entries as single entries (unless
1048     `return_single_as_list` is true), dictionaries as dictionaries.
1049 
1050     Args:
1051       return_single_as_list: Returns a list of size 1 for single entry case.
1052     """
1053     if self._is_dict:
1054       return dict(zip(self._input_names, self._flattened_inputs))
1055     if self._is_single_input and not return_single_as_list:
1056       return self._flattened_inputs[0]
1057     return self._flattened_inputs
1058 
1059   def get_input_values(self):
1060     """Returns input values passed in."""
1061     if context.executing_eagerly():
1062       for i in range(len(self._flattened_inputs)):
1063         v = self._flattened_inputs[i]
1064         if tensor_util.is_tensor(v):
1065           v = cast_single_tensor(v)
1066         else:
1067           v = ops.convert_to_tensor(v, dtype=K.floatx())
1068         self._flattened_inputs[i] = v
1069     return self._get(return_single_as_list=False)
1070 
1071   def get_symbolic_inputs(self, return_single_as_list=False):
1072     """Returns inputs to be set as self.inputs for a model."""
1073     for i in range(len(self._flattened_inputs)):
1074       k = self._input_names[i]
1075       v = self._flattened_inputs[i]
1076       if context.executing_eagerly():
1077         v = base_layer.DeferredTensor(
1078             shape=(None for _ in v.shape), dtype=v.dtype)
1079       else:
1080         if isinstance(v, list):
1081           v = np.asarray(v)
1082           if v.ndim == 1:
1083             v = np.expand_dims(v, 1)
1084         if isinstance(v, (np.ndarray)):
1085           # We fix the placeholder shape except the batch size.
1086           # This is suboptimal, but it is the best we can do with the info
1087           # we have. The user should call `model._set_inputs(placeholders)`
1088           # to specify custom placeholders if the need arises.
1089           shape = (None,) + v.shape[1:]
1090           v = K.placeholder(shape=shape, name=k)
1091       self._flattened_inputs[i] = v
1092 
1093     return self._get(return_single_as_list)
1094 
1095   def as_dict(self):
1096     """An iterable over a dictionary version of inputs."""
1097     for i in range(len(self._flattened_inputs)):
1098       yield self._input_names[i], self._flattened_inputs[i]
1099 
1100   def as_list(self):
1101     """Returning the inputs as a list."""
1102     return self._flattened_inputs
