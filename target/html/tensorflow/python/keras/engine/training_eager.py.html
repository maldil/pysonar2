  38 def _eager_loss_fn(outputs, targets, loss_fn, output_name):
  39   with backend.name_scope(output_name + '_loss'):
  40     loss = loss_fn(targets, outputs)
  41   return loss

  44 def _eager_metrics_fn(model, outputs, targets, sample_weights=None, masks=None):
  45   """Calculates the metrics for each output of the given model.
  46 
  47   Arguments:
  48       model: The model on which metrics are being calculated.
  49       outputs: The outputs of the given model.
  50       targets: The predictions or targets of the given model.
  51       sample_weights: Optional list of sample weights for each output.
  52       masks: Optional list of masks for each output.
  53 
  54   Returns:
  55       Returns the metric results for each output of the model.
  56   """
  57   outputs = generic_utils.to_list(outputs)
  58   targets = generic_utils.to_list(targets)
  59   # TODO(psv): Consider supporting skip target indices in eager mode?
  60   metric_results = model._handle_metrics(
  61       outputs, targets=targets, sample_weights=sample_weights, masks=masks)
  62   return [backend.mean(t) for t in metric_results]

  65 def _model_loss(model, inputs, targets, sample_weights=None, training=False):
  66   """Calculates the loss for a given model.
  67 
  68   Arguments:
  69       model: The model on which metrics are being calculated.
  70       inputs: Either a dictionary of inputs to the model or a list of input
  71         arrays.
  72       targets: List of target arrays.
  73       sample_weights: Optional list of sample weight arrays.
  74       training: Whether the model should be run in inference or training mode.
  75 
  76   Returns:
  77      Returns the model output, total loss, loss value calculated using the
  78      specified loss function and masks for each output. The total loss includes
  79      regularization losses and applies masking and sample weighting
  80      to the loss value.
  81   """
  82   total_loss = 0
  83   kwargs = {}
  84   if model._expects_training_arg:
  85     kwargs['training'] = training
  86   if len(inputs) == 1 and not isinstance(inputs, dict):
  87     inputs = inputs[0]
  88 
  89   if model._compute_output_and_mask_jointly:
  90     outs, masks = model._call_and_compute_mask(inputs, **kwargs)
  91     masks = generic_utils.to_list(masks)
  92   else:
  93     outs = model.call(inputs, **kwargs)
  94     masks = None
  95 
  96   outs = generic_utils.to_list(outs)
  97   if masks is None:
  98     masks = [None for _ in outs]
  99   targets = generic_utils.to_list(targets)
 100 
 101   loss_metrics = []
 102   with backend.name_scope('loss'):
 103     for i, loss_fn in enumerate(model.loss_functions):
 104       if sample_weights:
 105         weights = sample_weights[i]
 106       else:
 107         weights = None
 108       mask = masks[i]
 109 
 110       weighted_masked_fn = training_utils.weighted_masked_objective(loss_fn)
 111       with backend.name_scope(model.output_names[i] + '_loss'):
 112         output_loss = weighted_masked_fn(
 113             targets[i], outs[i], weights, mask=mask)
 114       # If the number of outputs is 1 then we don't append the loss metric
 115       # associated with each model output. When there are multiple outputs
 116       # associated with a model, each output's loss is calculated and returned
 117       # as part of the loss_metrics.
 118       if len(model.outputs) > 1:
 119         loss_metrics.append(backend.mean(output_loss))
 120 
 121       loss_weight = model.loss_weights_list[i]
 122       if total_loss is None:
 123         total_loss = loss_weight * output_loss
 124       else:
 125         total_loss += loss_weight * output_loss
 126 
 127     total_loss = backend.mean(total_loss)
 128     # Add regularization losses
 129     custom_losses = []
 130     for layer in model.layers:
 131       if layer.losses:
 132         custom_losses += layer.losses
 133 
 134     if custom_losses:
 135       total_loss += sum(custom_losses)
 136 
 137   return outs, total_loss, loss_metrics, masks

 140 def iterator_fit_loop(model,
 141                       inputs,
 142                       class_weight,
 143                       steps_per_epoch,
 144                       epoch_logs,
 145                       val_inputs=None,
 146                       val_targets=None,
 147                       val_sample_weights=None,
 148                       epochs=1,
 149                       verbose=1,
 150                       callbacks=None,
 151                       validation_steps=None,
 152                       do_validation=False,
 153                       batch_size=None):
 154   """Fit function for eager execution when input is given as dataset iterator.
 155 
 156   Updates the given epoch logs.
 157 
 158   Arguments:
 159       model: Instance of the `Model`.
 160       inputs: Input dataset iterator.
 161       class_weight: Optional class-weight array to weight the importance of
 162           samples in `inputs` based on the class they belong to, as conveyed by
 163           the targets from the `inputs` iterator.
 164       steps_per_epoch: Total number of steps (batches of samples)
 165           before declaring one epoch finished and starting the
 166           next epoch.
 167       epoch_logs: Dictionary of logs from every epoch.
 168       val_inputs: Input data for validation.
 169       val_targets: Target data for validation.
 170       val_sample_weights: Sample weight data for validation.
 171       epochs: Number of times to iterate over the data
 172       verbose: Verbosity mode, 0, 1 or 2
 173       callbacks: CallbackList instance. Controls callbacks during training.
 174       validation_steps: Number of steps to run validation for (only if doing
 175         validation from data tensors). Ignored with default value of `None`.
 176       do_validation: Boolean value indicating whether we should do validation.
 177       batch_size: int, val_inputs and val_targets will be evaled batch by
 178         batch with size batch_size if they are array.
 179 
 180   Raises:
 181       ValueError: In case of mismatch between given number of inputs and
 182         expectations of the model.
 183   """
 184   assert isinstance(inputs, iterator_ops.EagerIterator)
 185 
 186   # make sure either x,y or x,y,sample_weights is provided
 187   if (not isinstance(inputs.output_shapes, (list, tuple)) or
 188       len(inputs.output_shapes) not in (2, 3)):
 189     raise ValueError('Please provide either inputs and targets '
 190                      'or inputs, targets, and sample_weights')
 191 
 192   for step_index in range(steps_per_epoch):
 193     batch_logs = {'batch': step_index, 'size': 1}
 194     callbacks.on_batch_begin(step_index, batch_logs)
 195 
 196     # Get data from the iterator.
 197     try:
 198       next_element = inputs.get_next()
 199     except errors.OutOfRangeError:
 200       logging.warning(
 201           'Your dataset iterator ran out of data; interrupting training. Make '
 202           'sure that your dataset can generate at least '
 203           '`steps_per_epoch * epochs` batches (in this case, %d batches). You '
 204           'may need to use the repeat() function when building your '
 205           'dataset.' % steps_per_epoch * epochs)
 206       break
 207 
 208     if len(inputs.output_shapes) == 2:
 209       x, y = next_element
 210       sample_weights = None
 211     else:
 212       x, y, sample_weights = next_element
 213 
 214     # Validate and standardize data.
 215     x, y, sample_weights = model._standardize_user_data(
 216         x, y, sample_weight=sample_weights, class_weight=class_weight)
 217     x = training_utils.cast_if_floating_dtype(x)
 218     y = training_utils.cast_if_floating_dtype(y)
 219     if sample_weights:
 220       sample_weights = [
 221           training_utils.cast_if_floating_dtype(
 222               ops.convert_to_tensor(val, dtype=backend.floatx()))
 223           if val is not None else None for val in sample_weights
 224       ]
 225 
 226     # Set stateful_metrics in callbacks. We do not do this before the
 227     # `steps_per_epoch` loop because model will be compiled only in the first
 228     # iteration of this loop in the deferred build scenario.
 229     if step_index == 0:
 230       for cbk in callbacks:
 231         if (isinstance(cbk, cbks.BaseLogger) or
 232             isinstance(cbk, cbks.ProgbarLogger)):
 233           cbk.stateful_metrics = model.stateful_metric_names
 234 
 235     if step_index == 0 and not callbacks.params['metrics']:
 236       callback_metrics = copy.copy(model.metrics_names)
 237       if do_validation:
 238         callback_metrics += ['val_' + n for n in model.metrics_names]
 239       callbacks.set_params({
 240           'batch_size': batch_size,
 241           'epochs': epochs,
 242           'steps': steps_per_epoch,
 243           'verbose': verbose,
 244           'do_validation': do_validation,
 245           'metrics': callback_metrics or [],
 246           'validation_steps': validation_steps
 247       })
 248 
 249     # Train model.
 250     outs, loss, loss_metrics, masks = _process_single_batch(
 251         model, x, y, sample_weights=sample_weights, training=True)
 252     outs = generic_utils.to_list(outs)
 253 
 254     # Calculate metrics.
 255     for l, o in zip(model.metrics_names, outs):
 256       batch_logs[l] = o
 257     # Required for eager execution
 258     metrics_results = _eager_metrics_fn(
 259         model, outs, y, sample_weights=sample_weights, masks=masks)
 260     batch_logs['loss'] = tensor_util.constant_value(backend.mean(loss))
 261 
 262     for k, v in zip(model.metrics_names,
 263                     [backend.mean(loss)] + loss_metrics + metrics_results):
 264       batch_logs[k] = tensor_util.constant_value(v)
 265     callbacks.on_batch_end(step_index, batch_logs)
 266     if callbacks.model.stop_training:
 267       break
 268 
 269     if step_index == steps_per_epoch - 1:
 270       if do_validation:
 271         val_outs = test_loop(
 272             model,
 273             val_inputs,
 274             val_targets,
 275             sample_weights=val_sample_weights,
 276             steps=validation_steps,
 277             verbose=0,
 278             batch_size=batch_size)
 279         if not isinstance(val_outs, list):
 280           val_outs = [val_outs]
 281         # Same labels assumed.
 282         for l, o in zip(model.metrics_names, val_outs):
 283           epoch_logs['val_' + l] = o

 286 def iterator_test_loop(model, inputs, steps, verbose=0):
 287   """Test function for eager execution when input is given as dataset iterator.
 288 
 289   Arguments:
 290       model: Model instance that is being evaluated in Eager mode.
 291       inputs: Input dataset iterator.
 292       steps: Total number of steps (batches of samples) before declaring
 293       predictions finished.
 294       verbose: Verbosity mode.
 295 
 296   Returns:
 297       Scalar loss (if the model has a single output and no metrics)
 298       or list of scalars (if the model has multiple outputs
 299       and/or metrics). The attribute `model.metrics_names` will give you
 300       the display labels for the scalar outputs.
 301 
 302   Raises:
 303       ValueError: In case of mismatch between given number of inputs and
 304         expectations of the model.
 305   """
 306   assert isinstance(inputs, iterator_ops.EagerIterator)
 307   # make sure either x,y or x,y,sample_weights is provided
 308   if (not isinstance(inputs.output_shapes, (list, tuple)) or
 309       len(inputs.output_shapes) < 2 or len(inputs.output_shapes) > 3):
 310     raise ValueError('Please provide either inputs and targets'
 311                      'or inputs, targets, and sample_weights')
 312   outs = []
 313   num_samples = 0
 314   if verbose == 1:
 315     progbar = generic_utils.Progbar(target=steps)
 316   for step_index in range(steps):
 317     # Get data from the iterator.
 318     try:
 319       next_element = inputs.get_next()
 320     except errors.OutOfRangeError:
 321       logging.warning(
 322           'Your dataset iterator ran out of data interrupting testing. '
 323           'Make sure that your dataset can generate at least `steps` batches '
 324           '(in this case, %d batches). You may need to use the repeat() '
 325           'function when building your dataset.', steps)
 326       break
 327 
 328     if len(inputs.output_shapes) == 2:
 329       x, y = next_element
 330       sample_weights = None
 331     else:
 332       x, y, sample_weights = next_element
 333 
 334     # Validate and standardize data.
 335     x, y, sample_weights = model._standardize_user_data(
 336         x, y, sample_weight=sample_weights)
 337     x = training_utils.cast_if_floating_dtype(x)
 338     y = training_utils.cast_if_floating_dtype(y)
 339     if sample_weights:
 340       sample_weights = [
 341           training_utils.cast_if_floating_dtype(
 342               ops.convert_to_tensor(val, dtype=backend.floatx()))
 343           if val is not None else None for val in sample_weights
 344       ]
 345 
 346     if step_index == 0:
 347       # Get stateful metrics indices. We do not do this before the `steps` loop
 348       # because model will be compiled only in the first iteration of this loop
 349       # in the deferred build scenario.
 350       if hasattr(model, 'metrics'):
 351         for m in model.stateful_metric_functions:
 352           m.reset_states()
 353         stateful_metric_indices = [
 354             i for i, name in enumerate(model.metrics_names)
 355             if str(name) in model.stateful_metric_names
 356         ]
 357       else:
 358         stateful_metric_indices = []
 359 
 360     # Calculate model output, loss values.
 361     loss_outs, loss, loss_metrics, masks = _model_loss(
 362         model, x, y, sample_weights=sample_weights, training=False)
 363     metrics_results = _eager_metrics_fn(
 364         model, loss_outs, y, sample_weights=sample_weights, masks=masks)
 365     batch_outs = []
 366     for _, v in zip(model.metrics_names,
 367                     [backend.mean(loss)] + loss_metrics + metrics_results):
 368       batch_outs.append(tensor_util.constant_value(v))
 369 
 370     # Get current step size.
 371     if isinstance(x, list):
 372       step_size = x[0].get_shape().as_list()[0]
 373     elif isinstance(x, dict):
 374       step_size = list(x.values())[0].get_shape().as_list()[0]
 375     else:
 376       step_size = x.get_shape().as_list()[0]
 377 
 378     # Accumulate results in output array.
 379     if not isinstance(batch_outs, list):
 380       batch_outs = [batch_outs]
 381     if step_index == 0:
 382       for _ in enumerate(batch_outs):
 383         outs.append(0.)
 384     for i, batch_out in enumerate(batch_outs):
 385       if i in stateful_metric_indices:
 386         outs[i] = batch_out
 387       else:
 388         outs[i] += batch_out * step_size
 389 
 390     # Calculate sample size.
 391     num_samples += step_size
 392     if verbose == 1:
 393       progbar.update(step_index + 1)
 394 
 395   for i in range(len(outs)):
 396     if i not in stateful_metric_indices:
 397       outs[i] /= num_samples
 398   if len(outs) == 1:
 399     return outs[0]
 400   return outs

 403 def iterator_predict_loop(model, inputs, steps, verbose=0):
 404   """Predict function for eager execution when input is dataset iterator.
 405 
 406   Arguments:
 407       model: Instance of `Model`.
 408       inputs: Input dataset iterator.
 409       steps: Total number of steps (batches of samples) before declaring
 410           `_predict_loop` finished.
 411       verbose: Verbosity mode.
 412 
 413   Returns:
 414       Array of predictions (if the model has a single output)
 415       or list of arrays of predictions (if the model has multiple outputs).
 416 
 417   Raises:
 418       ValueError: In case of mismatch between given number of inputs and
 419         expectations of the model.
 420   """
 421   assert isinstance(inputs, iterator_ops.EagerIterator)
 422   if not isinstance(inputs.output_shapes,
 423                     (list, tuple)) or len(inputs.output_shapes) > 3:
 424     raise ValueError(
 425         'Please provide data as a list or tuple of 1, 2, or 3 elements '
 426         ' - `(input)`, or `(input, target)`, or `(input, target,'
 427         'sample_weights)`. Received %s. We do not use the `target` or'
 428         '`sample_weights` value here.' % inputs.output_shapes)
 429   outs = []
 430   if verbose == 1:
 431     progbar = generic_utils.Progbar(target=steps)
 432   for step_index in range(steps):
 433     # Get data from the iterator.
 434     try:
 435       next_element = inputs.get_next()
 436     except errors.OutOfRangeError:
 437       logging.warning(
 438           'Your dataset iterator ran out of data; interrupting prediction. '
 439           'Make sure that your dataset can generate at least `steps` batches '
 440           '(in this case, %d batches). You may need to use the repeat() '
 441           'function when building your dataset.', steps)
 442       break
 443 
 444     # expects a tuple, where first element of tuple represents inputs
 445     x = next_element[0]
 446 
 447     # Validate and standardize data.
 448     x, _, _ = model._standardize_user_data(x)
 449     x = training_utils.cast_if_floating_dtype(x)
 450 
 451     if isinstance(x, list) and len(x) == 1:
 452       x = x[0]
 453 
 454     if model._expects_training_arg:
 455       batch_outs = model.call(x, training=False)
 456     else:
 457       batch_outs = model.call(x)
 458     if not isinstance(batch_outs, list):
 459       batch_outs = [batch_outs]
 460 
 461     # We collect the results from every step and then concatenate them once
 462     # in the end. This is an expensive process. We are doing this because we
 463     # do not know the number of samples beforehand.
 464     if step_index == 0:
 465       for _ in batch_outs:
 466         outs.append([])
 467     for i, batch_out in enumerate(batch_outs):
 468       outs[i].append(backend.get_value(batch_out))
 469 
 470     if verbose == 1:
 471       progbar.update(step_index + 1)
 472   for i, out in enumerate(outs):
 473     outs[i] = np.concatenate(tuple(out), axis=0)
 474   if len(outs) == 1:
 475     return outs[0]
 476   return outs

 479 def _process_single_batch(model,
 480                           inputs,
 481                           targets,
 482                           sample_weights=None,
 483                           training=False):
 484   """Calculate the loss and gradient for one input batch.
 485 
 486      The model weights are updated if training is set to True.
 487 
 488   Arguments:
 489       model: Model whose loss has to be calculated.
 490       inputs: List of input arrays.
 491       targets: List of target arrays.
 492       sample_weights: Optional list of sample weight arrays.
 493       training: The boolean represents if the weights of the model are updated.
 494               'fit' methods will set this to True while 'evaluate' methods will
 495               set this to False.
 496 
 497   Returns:
 498       output of the model, total loss, the loss and the mask
 499       associated with each output.
 500 
 501   Raises:
 502       ValueError: If the model has no loss to optimize.
 503   """
 504   with backend.learning_phase_scope(1 if training else 0):
 505     with GradientTape() as tape:
 506       outs, loss, loss_metrics, masks = _model_loss(
 507           model,
 508           inputs,
 509           targets,
 510           sample_weights=sample_weights,
 511           training=training)
 512       if loss is None:
 513         raise ValueError('The model cannot be run '
 514                          'because it has no loss to optimize.')
 515     if training:
 516       if not model._collected_trainable_weights:
 517         logging.warning('The list of trainable weights is empty. Make sure that'
 518                         ' you are not setting model.trainable to False before '
 519                         'compiling the model.')
 520       else:
 521         grads = tape.gradient(loss, model._collected_trainable_weights)
 522         model.optimizer.apply_gradients(zip(grads,
 523                                             model._collected_trainable_weights))
 524     return outs, loss, loss_metrics, masks

 527 def train_on_batch(model, inputs, targets, sample_weights=None):
 528   """Calculates the loss and gradient updates for one input batch.
 529 
 530   Arguments:
 531       model: Model whose loss has to be calculated.
 532       inputs: Input batch data.
 533       targets: Target batch data.
 534       sample_weights: Sample weight batch data.
 535 
 536   Returns:
 537       total loss and the loss associated with each output.
 538   """
 539   if len(inputs) and tensor_util.is_tensor(inputs[0]):
 540     inputs = training_utils.cast_if_floating_dtype(inputs)
 541     targets = training_utils.cast_if_floating_dtype(targets)
 542   else:
 543     inputs = [
 544         ops.convert_to_tensor(val, dtype=backend.floatx()) for val in inputs
 545     ]
 546     targets = [
 547         ops.convert_to_tensor(val, dtype=backend.floatx()) for val in targets
 548     ]
 549   if sample_weights:
 550     sample_weights = [
 551         ops.convert_to_tensor(val, dtype=backend.floatx())
 552         if val is not None else None for val in sample_weights
 553     ]
 554 
 555   outs, loss, loss_metrics, masks = _process_single_batch(
 556       model, inputs, targets, sample_weights=sample_weights, training=True)
 557   if not isinstance(outs, list):
 558     outs = [outs]
 559   metrics_results = _eager_metrics_fn(
 560       model, outs, targets, sample_weights=sample_weights, masks=masks)
 561   loss = generic_utils.to_list(loss)
 562 
 563   return [
 564       tensor_util.constant_value(v)
 565       for v in loss + loss_metrics + metrics_results
 566   ]

 569 def test_on_batch(model, inputs, targets, sample_weights=None):
 570   """Calculates the loss for one input batch.
 571 
 572   Arguments:
 573       model: Model whose loss has to be calculated.
 574       inputs: Input batch data.
 575       targets: Target batch data.
 576       sample_weights: Sample weight batch data.
 577 
 578   Returns:
 579       total loss, loss and metrics associated with each output.
 580   """
 581   if len(inputs) and tensor_util.is_tensor(inputs[0]):
 582     inputs = training_utils.cast_if_floating_dtype(inputs)
 583     targets = training_utils.cast_if_floating_dtype(targets)
 584   else:
 585     inputs = [
 586         ops.convert_to_tensor(val, dtype=backend.floatx()) for val in inputs
 587     ]
 588     targets = [
 589         ops.convert_to_tensor(val, dtype=backend.floatx()) for val in targets
 590     ]
 591   if sample_weights:
 592     sample_weights = [
 593         ops.convert_to_tensor(val, dtype=backend.floatx())
 594         if val is not None else None for val in sample_weights
 595     ]
 596   outs, loss, loss_metrics, masks = _model_loss(
 597       model, inputs, targets, sample_weights=sample_weights, training=False)
 598   if not isinstance(outs, list):
 599     outs = [outs]
 600   metrics_results = _eager_metrics_fn(
 601       model, outs, targets, sample_weights=sample_weights, masks=masks)
 602   loss = generic_utils.to_list(loss)
 603 
 604   return [
 605       tensor_util.constant_value(v)
 606       for v in loss + loss_metrics + metrics_results
 607   ]

 610 def fit_loop(model,
 611              inputs,
 612              targets,
 613              sample_weights=None,
 614              class_weight=None,
 615              val_inputs=None,
 616              val_targets=None,
 617              val_sample_weights=None,
 618              batch_size=None,
 619              epochs=1,
 620              verbose=1,
 621              callbacks=None,
 622              shuffle=True,
 623              initial_epoch=0,
 624              steps_per_epoch=None,
 625              validation_steps=None):
 626   """Fit function for eager execution.
 627 
 628   Arguments:
 629       model: Instance of the model that is being executed in Eager mode.
 630       inputs: List of input arrays.
 631       targets: List of target arrays.
 632       sample_weights: Optional list of sample weight arrays.
 633       class_weight: Optional class-weight array to weight the importance of
 634           samples in `inputs` based on the class they belong to, as conveyed by
 635           `targets`.
 636       val_inputs: Input data for validation.
 637       val_targets: Target data for validation.
 638       val_sample_weights: Sample weight data for validation.
 639       batch_size: Integer batch size or None if unknown.
 640       epochs: Number of times to iterate over the data
 641       verbose: Verbosity mode, 0, 1 or 2
 642       callbacks: List of callbacks to be called during training
 643       shuffle: Whether to shuffle the data at the beginning of each epoch
 644       initial_epoch: Epoch at which to start training
 645           (useful for resuming a previous training run)
 646       steps_per_epoch: Total number of steps (batches of samples)
 647           before declaring one epoch finished and starting the
 648           next epoch. Ignored with the default value of `None`.
 649       validation_steps: Number of steps to run validation for (only if doing
 650         validation from data tensors). Ignored with default value of `None`.
 651 
 652   Returns:
 653       `History` object.
 654 
 655   Raises:
 656     ValueError: In case of invalid argument values.
 657   """
 658   # Convert training inputs to an EagerIterator
 659   inputs, steps_per_epoch = training_utils.convert_to_iterator(
 660       x=inputs,
 661       y=targets,
 662       sample_weights=sample_weights,
 663       batch_size=batch_size,
 664       steps_per_epoch=steps_per_epoch,
 665       epochs=epochs,
 666       shuffle=shuffle)
 667   # Required for eager execution
 668   with backend.learning_phase_scope(1):
 669     do_validation = val_inputs is not None
 670     callbacks = cbks.configure_callbacks(
 671         callbacks,
 672         model,
 673         do_validation=do_validation,
 674         batch_size=batch_size,
 675         epochs=epochs,
 676         steps_per_epoch=steps_per_epoch,
 677         val_inputs=val_inputs,
 678         val_targets=val_targets,
 679         val_sample_weights=val_sample_weights,
 680         validation_steps=validation_steps,
 681         verbose=verbose)
 682 
 683     callbacks.on_train_begin()
 684     for epoch in range(initial_epoch, epochs):
 685       if model._is_compiled:  # Model may not be compiled the first time.
 686         # Reset stateful metrics
 687         for m in model.stateful_metric_functions:
 688           m.reset_states()
 689       callbacks.on_epoch_begin(epoch)
 690       epoch_logs = {}
 691       iterator_fit_loop(
 692           model,
 693           inputs,
 694           class_weight,
 695           steps_per_epoch=steps_per_epoch,
 696           epoch_logs=epoch_logs,
 697           val_inputs=val_inputs,
 698           val_targets=val_targets,
 699           val_sample_weights=val_sample_weights,
 700           epochs=epochs,
 701           verbose=verbose,
 702           callbacks=callbacks,
 703           validation_steps=validation_steps,
 704           do_validation=do_validation,
 705           batch_size=batch_size)
 706       callbacks.on_epoch_end(epoch, epoch_logs)
 707       if callbacks.model.stop_training:
 708         break
 709   callbacks.on_train_end()
 710   return model.history

 713 def test_loop(model, inputs, targets,
 714               sample_weights=None,
 715               batch_size=None,
 716               verbose=0,
 717               steps=None):
 718   """Test function for eager execution.
 719 
 720   Arguments:
 721       model: Model instance that is being evaluated in Eager mode.
 722       inputs: List of input arrays.
 723       targets: List of target arrays.
 724       sample_weights: Optional list of sample weight arrays.
 725       batch_size: integer batch size or `None`.
 726       verbose: verbosity mode.
 727       steps: Total number of steps (batches of samples)
 728           before declaring predictions finished.
 729           Ignored with the default value of `None`.
 730 
 731   Returns:
 732       Scalar loss (if the model has a single output and no metrics)
 733       or list of scalars (if the model has multiple outputs
 734       and/or metrics). The attribute `model.metrics_names` will give you
 735       the display labels for the scalar outputs.
 736   """

 738       x=inputs,
 739       y=targets,
 740       sample_weights=sample_weights,
 741       batch_size=batch_size,
 742       steps_per_epoch=steps)
 743   with backend.learning_phase_scope(0):
 744     return iterator_test_loop(model, inputs, steps, verbose=verbose)
