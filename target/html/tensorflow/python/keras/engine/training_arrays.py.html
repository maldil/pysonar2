  39 def fit_loop(model,
  40              inputs,
  41              targets,
  42              sample_weights=None,
  43              batch_size=None,
  44              epochs=100,
  45              verbose=1,
  46              callbacks=None,
  47              val_inputs=None,
  48              val_targets=None,
  49              val_sample_weights=None,
  50              shuffle=True,
  51              initial_epoch=0,
  52              steps_per_epoch=None,
  53              validation_steps=None):
  54   """Abstract fit function for arrays of data.
  55 
  56   Arguments:
  57       model: Keras Model instance.
  58       inputs: Either a list of arrays or a dictionary.
  59       targets: List of target arrays.
  60       sample_weights: Optional list of sample weight arrays.
  61       batch_size: Integer batch size or None if unknown.
  62       epochs: Number of times to iterate over the data
  63       verbose: Verbosity mode, 0, 1 or 2
  64       callbacks: List of callbacks to be called during training
  65       val_inputs: List of input arrays.
  66       val_targets: List of target arrays.
  67       val_sample_weights: Optional list of sample weight arrays.
  68       shuffle: Whether to shuffle the data at the beginning of each epoch
  69           concatenation of list the display names of the outputs of
  70            `f` and the list of display names of the outputs of `f_val`.
  71       initial_epoch: Epoch at which to start training
  72           (useful for resuming a previous training run)
  73       steps_per_epoch: Total number of steps (batches of samples)
  74           before declaring one epoch finished and starting the
  75           next epoch. Ignored with the default value of `None`.
  76       validation_steps: Number of steps to run validation for
  77           (only if doing validation from data tensors).
  78           Ignored with the default value of `None`.
  79 
  80   Returns:
  81       `History` object.
  82 
  83   Raises:
  84       ValueError: in case of invalid arguments.
  85   """
  86   model._make_train_function()
  87   f = model.train_function
  88 
  89   sample_weights = sample_weights or []
  90   val_sample_weights = val_sample_weights or []
  91   inputs = training_utils.ModelInputs(inputs).as_list()
  92   if model.uses_learning_phase and not isinstance(K.learning_phase(), int):
  93     ins = inputs + targets + sample_weights + [1]
  94   else:
  95     ins = inputs + targets + sample_weights
  96 
  97   do_validation = False
  98   if val_inputs:
  99     do_validation = True
 100     if (steps_per_epoch is None and verbose and inputs and
 101         hasattr(inputs[0], 'shape') and hasattr(val_inputs[0], 'shape')):
 102       print('Train on %d samples, validate on %d samples' %
 103             (inputs[0].shape[0], val_inputs[0].shape[0]))
 104   if validation_steps:
 105     do_validation = True
 106     if steps_per_epoch is None:
 107       raise ValueError('Can only use `validation_steps` '
 108                        'when doing step-wise '
 109                        'training, i.e. `steps_per_epoch` '
 110                        'must be set.')
 111 
 112   num_train_samples = training_utils.check_num_samples(
 113       ins, batch_size, steps_per_epoch, 'steps_per_epoch')
 114   count_mode = 'steps' if steps_per_epoch else 'samples'
 115   callbacks = cbks.configure_callbacks(
 116       callbacks,
 117       model,
 118       do_validation=do_validation,
 119       val_inputs=val_inputs,
 120       val_targets=val_targets,
 121       val_sample_weights=val_sample_weights,
 122       batch_size=batch_size,
 123       epochs=epochs,
 124       steps_per_epoch=steps_per_epoch,
 125       samples=num_train_samples,
 126       validation_steps=validation_steps,
 127       verbose=verbose,
 128       count_mode=count_mode)
 129 
 130   if num_train_samples is not None:
 131     index_array = np.arange(num_train_samples)
 132 
 133   # To prevent a slowdown, we find beforehand the arrays that need conversion.
 134   feed = model._feed_inputs + model._feed_targets + model._feed_sample_weights
 135   indices_for_conversion_to_dense = []
 136   for i in range(len(feed)):
 137     if issparse is not None and issparse(ins[i]) and not K.is_sparse(feed[i]):
 138       indices_for_conversion_to_dense.append(i)
 139 
 140   callbacks.on_train_begin()
 141   for epoch in range(initial_epoch, epochs):
 142     # Reset stateful metrics
 143     for m in model.stateful_metric_functions:
 144       m.reset_states()
 145     # Update callbacks
 146     callbacks.on_epoch_begin(epoch)
 147     epoch_logs = {}
 148     if steps_per_epoch is not None:
 149       # Step-wise fit loop.
 150       for step_index in range(steps_per_epoch):
 151         batch_logs = {'batch': step_index, 'size': 1}
 152         callbacks.on_batch_begin(step_index, batch_logs)
 153         try:
 154           outs = f(ins)
 155         except errors.OutOfRangeError:
 156           logging.warning('Your dataset iterator ran out of data; '
 157                           'interrupting training. Make sure that your dataset '
 158                           'can generate at least `steps_per_epoch * epochs` '
 159                           'batches (in this case, %d batches). You may need to'
 160                           'use the repeat() function when building your '
 161                           'dataset.' %
 162                           steps_per_epoch * epochs)
 163           break
 164 
 165         if not isinstance(outs, list):
 166           outs = [outs]
 167         for l, o in zip(model.metrics_names, outs):
 168           batch_logs[l] = o
 169 
 170         callbacks.on_batch_end(step_index, batch_logs)
 171         if callbacks.model.stop_training:
 172           break
 173 
 174       if do_validation:
 175         val_outs = test_loop(
 176             model,
 177             val_inputs,
 178             val_targets,
 179             sample_weights=val_sample_weights,
 180             steps=validation_steps,
 181             verbose=0)
 182         if not isinstance(val_outs, list):
 183           val_outs = [val_outs]
 184         # Same labels assumed.
 185         for l, o in zip(model.metrics_names, val_outs):
 186           epoch_logs['val_' + l] = o
 187     else:
 188       # Sample-wise fit loop.
 189       if shuffle == 'batch':
 190         index_array = training_utils.batch_shuffle(index_array, batch_size)
 191       elif shuffle:
 192         np.random.shuffle(index_array)
 193 
 194       batches = make_batches(num_train_samples, batch_size)
 195 
 196       for batch_index, (batch_start, batch_end) in enumerate(batches):
 197         batch_ids = index_array[batch_start:batch_end]
 198         try:
 199           if isinstance(ins[-1], int):
 200             # Do not slice the training phase flag.
 201             ins_batch = slice_arrays(ins[:-1], batch_ids) + [ins[-1]]
 202           else:
 203             ins_batch = slice_arrays(ins, batch_ids)
 204         except TypeError:
 205           raise TypeError('TypeError while preparing batch. '
 206                           'If using HDF5 input data, '
 207                           'pass shuffle="batch".')
 208         batch_logs = {}
 209         batch_logs['batch'] = batch_index
 210         batch_logs['size'] = len(batch_ids)
 211         callbacks.on_batch_begin(batch_index, batch_logs)
 212         for i in indices_for_conversion_to_dense:
 213           ins_batch[i] = ins_batch[i].toarray()
 214 
 215         outs = f(ins_batch)
 216         if not isinstance(outs, list):
 217           outs = [outs]
 218         for l, o in zip(model.metrics_names, outs):
 219           batch_logs[l] = o
 220 
 221         callbacks.on_batch_end(batch_index, batch_logs)
 222         if callbacks.model.stop_training:
 223           break
 224 
 225         if batch_index == len(batches) - 1:  # Last batch.
 226           if do_validation:
 227             val_outs = test_loop(
 228                 model,
 229                 val_inputs,
 230                 val_targets,
 231                 sample_weights=val_sample_weights,
 232                 batch_size=batch_size,
 233                 verbose=0)
 234             if not isinstance(val_outs, list):
 235               val_outs = [val_outs]
 236             # Same labels assumed.
 237             for l, o in zip(model.metrics_names, val_outs):
 238               epoch_logs['val_' + l] = o
 239     callbacks.on_epoch_end(epoch, epoch_logs)
 240     if callbacks.model.stop_training:
 241       break
 242   callbacks.on_train_end()
 243   return model.history

 246 def predict_loop(model, inputs, batch_size=32, verbose=0, steps=None):
 247   """Abstract method to loop over some data in batches.
 248 
 249   Arguments:
 250       model: Keras Model instance.
 251       inputs: list of tensors to be fed to `f`.
 252       batch_size: integer batch size.
 253       verbose: verbosity mode.
 254       steps: Total number of steps (batches of samples)
 255           before declaring `_predict_loop` finished.
 256           Ignored with the default value of `None`.
 257 
 258   Returns:
 259       Array of predictions (if the model has a single output)
 260       or list of arrays of predictions
 261       (if the model has multiple outputs).
 262   """
 263   model._make_predict_function()
 264   f = model.predict_function
 265 
 266   inputs = training_utils.ModelInputs(inputs).as_list()
 267   if model.uses_learning_phase and not isinstance(K.learning_phase(), int):
 268     ins = inputs + [0]
 269   else:
 270     ins = inputs
 271 
 272   num_samples = training_utils.check_num_samples(
 273       inputs, batch_size, steps, 'steps')
 274   if verbose == 1:
 275     if steps is not None:
 276       progbar = Progbar(target=steps)
 277     else:
 278       progbar = Progbar(target=num_samples)
 279 
 280   indices_for_conversion_to_dense = []
 281   for i in range(len(model._feed_inputs)):
 282     if (issparse is not None and issparse(inputs[i]) and
 283         not K.is_sparse(model._feed_inputs[i])):
 284       indices_for_conversion_to_dense.append(i)
 285 
 286   if steps is not None:
 287     # Step-based predictions.
 288     # Since we do not know how many samples
 289     # we will see, we cannot pre-allocate
 290     # the returned Numpy arrays.
 291     # Instead, we store one array per batch seen
 292     # and concatenate them upon returning.
 293     unconcatenated_outs = []
 294     for step in range(steps):
 295       batch_outs = f(ins)
 296       if not isinstance(batch_outs, list):
 297         batch_outs = [batch_outs]
 298       if step == 0:
 299         for batch_out in batch_outs:
 300           unconcatenated_outs.append([])
 301       for i, batch_out in enumerate(batch_outs):
 302         unconcatenated_outs[i].append(batch_out)
 303       if verbose == 1:
 304         progbar.update(step + 1)
 305     if len(unconcatenated_outs) == 1:
 306       return np.concatenate(unconcatenated_outs[0], axis=0)
 307     return [
 308         np.concatenate(unconcatenated_outs[i], axis=0)
 309         for i in range(len(unconcatenated_outs))
 310     ]
 311   else:
 312     # Sample-based predictions.
 313     outs = []
 314     batches = make_batches(num_samples, batch_size)
 315     index_array = np.arange(num_samples)
 316     for batch_index, (batch_start, batch_end) in enumerate(batches):
 317       batch_ids = index_array[batch_start:batch_end]
 318       if ins and isinstance(ins[-1], int):
 319         # Do not slice the training phase flag.
 320         ins_batch = slice_arrays(ins[:-1], batch_ids) + [ins[-1]]
 321       else:
 322         ins_batch = slice_arrays(ins, batch_ids)
 323       for i in indices_for_conversion_to_dense:
 324         ins_batch[i] = ins_batch[i].toarray()
 325 
 326       batch_outs = f(ins_batch)
 327       if not isinstance(batch_outs, list):
 328         batch_outs = [batch_outs]
 329       if batch_index == 0:
 330         # Pre-allocate the results arrays.
 331         for batch_out in batch_outs:
 332           shape = (num_samples,) + batch_out.shape[1:]
 333           outs.append(np.zeros(shape, dtype=batch_out.dtype))
 334       for i, batch_out in enumerate(batch_outs):
 335         outs[i][batch_start:batch_end] = batch_out
 336       if verbose == 1:
 337         progbar.update(batch_end)
 338     if len(outs) == 1:
 339       return outs[0]
 340     return outs

 343 def test_loop(model,
 344               inputs,
 345               targets,
 346               sample_weights=None,
 347               batch_size=None,
 348               verbose=0,
 349               steps=None):
 350   """Abstract method to loop over some data in batches.
 351 
 352   Arguments:
 353       model: Keras Model instance.
 354       inputs: List of input arrays.
 355       targets: List of target arrays.
 356       sample_weights: Optional list of sample weight arrays.
 357       batch_size: integer batch size or `None`.
 358       verbose: verbosity mode.
 359       steps: Total number of steps (batches of samples)
 360           before declaring predictions finished.
 361           Ignored with the default value of `None`.
 362 
 363   Returns:
 364       Scalar loss (if the model has a single output and no metrics)
 365       or list of scalars (if the model has multiple outputs
 366       and/or metrics). The attribute `model.metrics_names` will give you
 367       the display labels for the scalar outputs.
 368   """
 369   model._make_test_function()
 370   f = model.test_function
 371 
 372   sample_weights = sample_weights or []
 373   inputs = training_utils.ModelInputs(inputs).as_list()
 374   if model.uses_learning_phase and not isinstance(K.learning_phase(), int):
 375     ins = inputs + targets + sample_weights + [0]
 376   else:
 377     ins = inputs + targets + sample_weights
 378 
 379   if hasattr(model, 'metrics'):
 380     for m in model.stateful_metric_functions:
 381       m.reset_states()
 382     stateful_metric_indices = [
 383         i for i, name in enumerate(model.metrics_names)
 384         if str(name) in model.stateful_metric_names
 385     ]
 386   else:
 387     stateful_metric_indices = []
 388 
 389   num_samples = training_utils.check_num_samples(
 390       ins, batch_size, steps, 'steps')
 391   outs = []
 392   if verbose == 1:
 393     if steps is not None:
 394       progbar = Progbar(target=steps)
 395     else:
 396       progbar = Progbar(target=num_samples)
 397 
 398   # To prevent a slowdown, we find beforehand the arrays that need conversion.
 399   feed = model._feed_inputs + model._feed_targets + model._feed_sample_weights
 400   indices_for_conversion_to_dense = []
 401   for i in range(len(feed)):
 402     if issparse is not None and issparse(ins[i]) and not K.is_sparse(feed[i]):
 403       indices_for_conversion_to_dense.append(i)
 404 
 405   if steps is not None:
 406     for step in range(steps):
 407       batch_outs = f(ins)
 408       if isinstance(batch_outs, list):
 409         if step == 0:
 410           for _ in enumerate(batch_outs):
 411             outs.append(0.)
 412         for i, batch_out in enumerate(batch_outs):
 413           if i in stateful_metric_indices:
 414             outs[i] = batch_out
 415           else:
 416             outs[i] += batch_out
 417       else:
 418         if step == 0:
 419           outs.append(0.)
 420         outs[0] += batch_outs
 421       if verbose == 1:
 422         progbar.update(step + 1)
 423     for i in range(len(outs)):
 424       if i not in stateful_metric_indices:
 425         outs[i] /= steps
 426   else:
 427     batches = make_batches(num_samples, batch_size)
 428     index_array = np.arange(num_samples)
 429     for batch_index, (batch_start, batch_end) in enumerate(batches):
 430       batch_ids = index_array[batch_start:batch_end]
 431       if isinstance(ins[-1], int):
 432         # Do not slice the training phase flag.
 433         ins_batch = slice_arrays(ins[:-1], batch_ids) + [ins[-1]]
 434       else:
 435         ins_batch = slice_arrays(ins, batch_ids)
 436       for i in indices_for_conversion_to_dense:
 437         ins_batch[i] = ins_batch[i].toarray()
 438 
 439       batch_outs = f(ins_batch)
 440 
 441       if isinstance(batch_outs, list):
 442         if batch_index == 0:
 443           outs.extend([0.] * len(batch_outs))
 444         for i, batch_out in enumerate(batch_outs):
 445           if i in stateful_metric_indices:
 446             outs[i] = batch_out
 447           else:
 448             outs[i] += batch_out * len(batch_ids)
 449       else:
 450         if batch_index == 0:
 451           outs.append(0.)
 452         outs[0] += batch_outs * len(batch_ids)
 453       if verbose == 1:
 454         progbar.update(batch_end)
 455     for i in range(len(outs)):
 456       if i not in stateful_metric_indices:
 457         outs[i] /= num_samples
 458   if len(outs) == 1:
 459     return outs[0]
 460   return outs
