  30 @tf_export('keras.layers.InputLayer')
  31 class InputLayer(base_layer.Layer):
  32   """Layer to be used as an entry point into a Network (a graph of layers).
  33 
  34   It can either wrap an existing tensor (pass an `input_tensor` argument)
  35   or create its a placeholder tensor (pass arguments `input_shape`, and
  36   optionally, `dtype`).
  37 
  38   It is generally recommend to use the functional layer API via `Input`,
  39   (which creates an `InputLayer`) without directly using `InputLayer`.
  40 
  41   Arguments:
  42       input_shape: Shape tuple (not including the batch axis), or `TensorShape`
  43         instance (not including the batch axis).
  44       batch_size: Optional input batch size (integer or None).
  45       dtype: Datatype of the input.
  46       input_tensor: Optional tensor to use as layer input
  47           instead of creating a placeholder.
  48       sparse: Boolean, whether the placeholder created
  49           is meant to be sparse.
  50       name: Name of the layer (string).
  51   """
  52 
  53   def __init__(self,
  54                input_shape=None,
  55                batch_size=None,
  56                dtype=None,
  57                input_tensor=None,
  58                sparse=False,
  59                name=None,
  60                **kwargs):
  61     if 'batch_input_shape' in kwargs:
  62       batch_input_shape = kwargs.pop('batch_input_shape')
  63       if input_shape and batch_input_shape:
  64         raise ValueError('Only provide the input_shape OR '
  65                          'batch_input_shape argument to '
  66                          'InputLayer, not both at the same time.')
  67       batch_size = batch_input_shape[0]
  68       input_shape = batch_input_shape[1:]
  69     if kwargs:
  70       raise ValueError('Unrecognized keyword arguments:', kwargs.keys())
  71 
  72     if not name:
  73       prefix = 'input'
  74       name = prefix + '_' + str(K.get_uid(prefix))
  75 
  76     if not dtype:
  77       if input_tensor is None:
  78         dtype = K.floatx()
  79       else:
  80         dtype = K.dtype(input_tensor)
  81     super(InputLayer, self).__init__(dtype=dtype, name=name)
  82     self.built = True
  83     self.sparse = sparse
  84     self.batch_size = batch_size
  85 
  86     if isinstance(input_shape, tensor_shape.TensorShape):
  87       input_shape = tuple(input_shape.as_list())
  88 
  89     if input_tensor is None:
  90       if input_shape is not None:
  91         batch_input_shape = (batch_size,) + tuple(input_shape)
  92       else:
  93         batch_input_shape = None
  94 
  95       if context.executing_eagerly():
  96         # In eager mode, create a temporary placeholder to call the layer on.
  97         input_tensor = base_layer.DeferredTensor(  # pylint: disable=protected-access
  98             shape=batch_input_shape,
  99             dtype=dtype,
 100             name=self.name)
 101       else:
 102         # In graph mode, create a graph placeholder to call the layer on.
 103         if sparse:
 104           input_tensor = array_ops.sparse_placeholder(
 105               shape=batch_input_shape,
 106               dtype=dtype,
 107               name=self.name)
 108         else:
 109           input_tensor = array_ops.placeholder(
 110               shape=batch_input_shape,
 111               dtype=dtype,
 112               name=self.name)
 113 
 114       # For compatibility with Keras API.
 115       self.is_placeholder = True
 116       self._batch_input_shape = batch_input_shape
 117     else:
 118       # For compatibility with Keras API.
 119       self.is_placeholder = False
 120       self._batch_input_shape = tuple(input_tensor.get_shape().as_list())
 121 
 122       if context.executing_eagerly():
 123         raise ValueError('You should not pass an input tensor when executing '
 124                          'in eager mode. For example, instead of creating an '
 125                          'InputLayer, you should instantiate your model and '
 126                          'directly call it on your input.')
 127 
 128     # Create an input node to add to self.outbound_node
 129     # and set output_tensors' _keras_history.
 130     input_tensor._keras_history = (self, 0, 0)  # pylint: disable=protected-access
 131     base_layer.Node(
 132         self,
 133         inbound_layers=[],
 134         node_indices=[],
 135         tensor_indices=[],
 136         input_tensors=[input_tensor],
 137         output_tensors=[input_tensor])
 138 
 139   def get_config(self):
 140     config = {
 141         'batch_input_shape': self._batch_input_shape,
 142         'dtype': self.dtype,
 143         'sparse': self.sparse,
 144         'name': self.name
 145     }
 146     return config

 149 @tf_export('keras.layers.Input', 'keras.Input')
 150 def Input(  # pylint: disable=invalid-name
 151     shape=None,
 152     batch_size=None,
 153     name=None,
 154     dtype=None,
 155     sparse=False,
 156     tensor=None,
 157     **kwargs):
 158   """`Input()` is used to instantiate a Keras tensor.
 159 
 160   A Keras tensor is a tensor object from the underlying backend
 161   (Theano or TensorFlow), which we augment with certain
 162   attributes that allow us to build a Keras model
 163   just by knowing the inputs and outputs of the model.
 164 
 165   For instance, if a, b and c are Keras tensors,
 166   it becomes possible to do:
 167   `model = Model(input=[a, b], output=c)`
 168 
 169   The added Keras attribute is:
 170       `_keras_history`: Last layer applied to the tensor.
 171           the entire layer graph is retrievable from that layer,
 172           recursively.
 173 
 174   Arguments:
 175       shape: A shape tuple (integers), not including the batch size.
 176           For instance, `shape=(32,)` indicates that the expected input
 177           will be batches of 32-dimensional vectors.
 178       batch_size: optional static batch size (integer).
 179       name: An optional name string for the layer.
 180           Should be unique in a model (do not reuse the same name twice).
 181           It will be autogenerated if it isn't provided.
 182       dtype: The data type expected by the input, as a string
 183           (`float32`, `float64`, `int32`...)
 184       sparse: A boolean specifying whether the placeholder
 185           to be created is sparse.
 186       tensor: Optional existing tensor to wrap into the `Input` layer.
 187           If set, the layer will not create a placeholder tensor.
 188       **kwargs: deprecated arguments support.
 189 
 190   Returns:
 191       A tensor.
 192 
 193   Example:
 194 
 195       ```python
 196       # this is a logistic regression in Keras
 197       x = Input(shape=(32,))
 198       y = Dense(16, activation='softmax')(x)
 199       model = Model(x, y)
 200       ```
 201 
 202   Raises:
 203     ValueError: in case of invalid arguments.
 204   """
 205   if 'batch_shape' in kwargs:
 206     batch_shape = kwargs.pop('batch_shape')
 207     if shape and batch_shape:
 208       raise ValueError('Only provide the shape OR '
 209                        'batch_shape argument to '
 210                        'Input, not both at the same time.')
 211     batch_size = batch_shape[0]
 212     shape = batch_shape[1:]
 213   if kwargs:
 214     raise ValueError('Unrecognized keyword arguments:', kwargs.keys())
 215 
 216   if dtype is None:
 217     dtype = K.floatx()
 218   if shape is None and tensor is None:
 219     raise ValueError('Please provide to Input either a `shape`'
 220                      ' or a `tensor` argument. Note that '
 221                      '`shape` does not include the batch '
 222                      'dimension.')
 223   input_layer = InputLayer(
 224       input_shape=shape,
 225       batch_size=batch_size,
 226       name=name,
 227       dtype=dtype,
 228       sparse=sparse,
 229       input_tensor=tensor)
 230   # Return tensor including `_keras_history`.
 231   # Note that in this case train_output and test_output are the same pointer.
 232   outputs = input_layer._inbound_nodes[0].output_tensors
 233   if len(outputs) == 1:
 234     return outputs[0]
 235   else:
 236     return outputs
