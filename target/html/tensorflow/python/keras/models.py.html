  48 def _clone_functional_model(model, input_tensors=None):
  49   """Clone a functional `Model` instance.
  50 
  51   Model cloning is similar to calling a model on new inputs,
  52   except that it creates new layers (and thus new weights) instead
  53   of sharing the weights of the existing layers.
  54 
  55   Arguments:
  56       model: Instance of `Model`.
  57       input_tensors: optional list of input tensors
  58           to build the model upon. If not provided,
  59           placeholders will be created.
  60 
  61   Returns:
  62       An instance of `Model` reproducing the behavior
  63       of the original model, on top of new inputs tensors,
  64       using newly instantiated weights.
  65 
  66   Raises:
  67       ValueError: in case of invalid `model` argument value.
  68   """
  69   if not isinstance(model, Model):
  70     raise ValueError('Expected `model` argument '
  71                      'to be a `Model` instance, got ', model)
  72   if isinstance(model, Sequential):
  73     raise ValueError('Expected `model` argument '
  74                      'to be a functional `Model` instance, '
  75                      'got a `Sequential` instance instead:', model)
  76 
  77   layer_map = {}  # Cache for created layers.
  78   tensor_map = {}  # Map {reference_tensor: corresponding_tensor}
  79   if input_tensors is None:
  80     # Create placeholders to build the model on top of.
  81     input_layers = []
  82     input_tensors = []
  83     for layer in model._input_layers:
  84       input_tensor = Input(
  85           batch_shape=layer._batch_input_shape,
  86           dtype=layer.dtype,
  87           sparse=layer.sparse,
  88           name=layer.name)
  89       input_tensors.append(input_tensor)
  90       # Cache newly created input layer.
  91       newly_created_input_layer = input_tensor._keras_history[0]
  92       layer_map[layer] = newly_created_input_layer
  93     for original_input_layer, cloned_input_layer in zip(model._input_layers,
  94                                                         input_layers):
  95       layer_map[original_input_layer] = cloned_input_layer
  96   else:
  97     # Make sure that all input tensors come from a Keras layer.
  98     # If tensor comes from an input layer: cache the input layer.
  99     input_tensors = generic_utils.to_list(input_tensors)
 100     input_tensors_ = []
 101     for i, x in enumerate(input_tensors):
 102       if not K.is_keras_tensor(x):
 103         name = model._input_layers[i].name
 104         input_tensor = Input(tensor=x, name='input_wrapper_for_' + name)
 105         input_tensors_.append(input_tensor)
 106         # Cache newly created input layer.
 107         original_input_layer = x._keras_history[0]
 108         newly_created_input_layer = input_tensor._keras_history[0]
 109         layer_map[original_input_layer] = newly_created_input_layer
 110       else:
 111         input_tensors_.append(x)
 112     input_tensors = input_tensors_
 113 
 114   for x, y in zip(model.inputs, input_tensors):
 115     tensor_map[x] = y
 116 
 117   # Iterated over every node in the reference model, in depth order.
 118   depth_keys = list(model._nodes_by_depth.keys())
 119   depth_keys.sort(reverse=True)
 120   for depth in depth_keys:
 121     nodes = model._nodes_by_depth[depth]
 122     for node in nodes:
 123       # Recover the corresponding layer.
 124       layer = node.outbound_layer
 125 
 126       # Get or create layer.
 127       if layer not in layer_map:
 128         # Clone layer.
 129         new_layer = layer.__class__.from_config(layer.get_config())
 130         layer_map[layer] = new_layer
 131         layer = new_layer
 132       else:
 133         # Reuse previously cloned layer.
 134         layer = layer_map[layer]
 135         # Don't call InputLayer multiple times.
 136         if isinstance(layer, InputLayer):
 137           continue
 138 
 139       # Gather inputs to call the new layer.
 140       reference_input_tensors = node.input_tensors
 141       reference_output_tensors = node.output_tensors
 142 
 143       # If all previous input tensors are available in tensor_map,
 144       # then call node.inbound_layer on them.
 145       computed_tensors = []
 146       for x in reference_input_tensors:
 147         if x in tensor_map:
 148           computed_tensors.append(tensor_map[x])
 149 
 150       if len(computed_tensors) == len(reference_input_tensors):
 151         # Call layer.
 152         if node.arguments:
 153           kwargs = node.arguments
 154         else:
 155           kwargs = {}
 156         if len(computed_tensors) == 1:
 157           computed_tensor = computed_tensors[0]
 158           output_tensors = generic_utils.to_list(layer(computed_tensor,
 159                                                        **kwargs))
 160           computed_tensors = [computed_tensor]
 161         else:
 162           computed_tensors = computed_tensors
 163           output_tensors = generic_utils.to_list(layer(computed_tensors,
 164                                                        **kwargs))
 165 
 166         for x, y in zip(reference_output_tensors, output_tensors):
 167           tensor_map[x] = y
 168 
 169   # Check that we did compute the model outputs,
 170   # then instantiate a new model from inputs and outputs.
 171   output_tensors = []
 172   for x in model.outputs:
 173     assert x in tensor_map, 'Could not compute output ' + str(x)
 174     output_tensors.append(tensor_map[x])
 175   return Model(input_tensors, output_tensors, name=model.name)

 178 def _clone_sequential_model(model, input_tensors=None):
 179   """Clone a `Sequential` model instance.
 180 
 181   Model cloning is similar to calling a model on new inputs,
 182   except that it creates new layers (and thus new weights) instead
 183   of sharing the weights of the existing layers.
 184 
 185   Arguments:
 186       model: Instance of `Sequential`.
 187       input_tensors: optional list of input tensors
 188           to build the model upon. If not provided,
 189           placeholders will be created.
 190 
 191   Returns:
 192       An instance of `Sequential` reproducing the behavior
 193       of the original model, on top of new inputs tensors,
 194       using newly instantiated weights.
 195 
 196   Raises:
 197       ValueError: in case of invalid `model` argument value.
 198   """
 199   if not isinstance(model, Sequential):
 200     raise ValueError('Expected `model` argument '
 201                      'to be a `Sequential` model instance, '
 202                      'but got:', model)
 203 
 204   def clone(layer):
 205     return layer.__class__.from_config(layer.get_config())
 206 
 207   layers = [clone(layer) for layer in model.layers]
 208   if input_tensors is None:
 209     return Sequential(layers=layers, name=model.name)
 210   else:
 211     if len(generic_utils.to_list(input_tensors)) != 1:
 212       raise ValueError('To clone a `Sequential` model, we expect '
 213                        ' at most one tensor '
 214                        'as part of `input_tensors`.')
 215     x = generic_utils.to_list(input_tensors)[0]
 216     if K.is_keras_tensor(x):
 217       origin_layer = x._keras_history[0]
 218       if isinstance(origin_layer, InputLayer):
 219         return Sequential(layers=[origin_layer] + layers, name=model.name)
 220       else:
 221         raise ValueError('Cannot clone a `Sequential` model on top '
 222                          'of a tensor that comes from a Keras layer '
 223                          'other than an `InputLayer`. '
 224                          'Use the functional API instead.')
 225     input_tensor = Input(tensor=x, name='input_wrapper_for_' + str(x.name))
 226     input_layer = input_tensor._keras_history[0]
 227     return Sequential(layers=[input_layer] + layers, name=model.name)

 230 @tf_export('keras.models.clone_model')
 231 def clone_model(model, input_tensors=None):
 232   """Clone any `Model` instance.
 233 
 234   Model cloning is similar to calling a model on new inputs,
 235   except that it creates new layers (and thus new weights) instead
 236   of sharing the weights of the existing layers.
 237 
 238   Arguments:
 239       model: Instance of `Model`
 240           (could be a functional model or a Sequential model).
 241       input_tensors: optional list of input tensors
 242           to build the model upon. If not provided,
 243           placeholders will be created.
 244 
 245   Returns:
 246       An instance of `Model` reproducing the behavior
 247       of the original model, on top of new inputs tensors,
 248       using newly instantiated weights.
 249 
 250   Raises:
 251       ValueError: in case of invalid `model` argument value.
 252   """
 253   if isinstance(model, Sequential):
 254     return _clone_sequential_model(model, input_tensors=input_tensors)
 255   else:
 256     return _clone_functional_model(model, input_tensors=input_tensors)

 262 def _in_place_subclassed_model_reset(model):
 263   """Substitute for model cloning that works for subclassed models.
 264 
 265   Subclassed models cannot be cloned because their topology is not serializable.
 266   To "instantiate" an identical model in a new TF graph, we reuse the original
 267   model object, but we clear its state.
 268 
 269   After calling this function on a model instance, you can use the model
 270   instance as if it were a model clone (in particular you can use it in a new
 271   graph).
 272 
 273   This method clears the state of the input model. It is thus destructive.
 274   However the original state can be restored fully by calling
 275   `_in_place_subclassed_model_state_restoration`.
 276 
 277   Args:
 278     model: Instance of a Keras model created via subclassing.
 279 
 280   Raises:
 281     ValueError: In case the model uses a subclassed model as inner layer.
 282   """
 283   assert not model._is_graph_network  # Only makes sense for subclassed networks
 284   # Retrieve all layers tracked by the model as well as their attribute names
 285   attributes_cache = {}
 286   for name in dir(model):
 287     try:
 288       value = getattr(model, name)
 289     except (AttributeError, ValueError, TypeError):
 290       continue
 291     if isinstance(value, Layer):
 292       attributes_cache[name] = value
 293       assert value in model._layers
 294     elif isinstance(
 295         value, (list, tuple)) and name not in ('layers', '_layers',
 296                                                'stateful_metric_functions'):
 297       # Handle case: list/tuple of layers (also tracked by the Network API).
 298       if value and all(isinstance(val, Layer) for val in value):
 299         raise ValueError('We do not support the use of list-of-layers '
 300                          'attributes in subclassed models used with '
 301                          '`model_to_estimator` at this time. Found list '
 302                          'model: %s' % name)
 303 
 304   # Replace layers on the model with fresh layers
 305   layers_to_names = {value: key for key, value in attributes_cache.items()}
 306   original_layers = model._layers[:]
 307   model._layers = data_structures.NoDependency([])
 308   for layer in original_layers:  # We preserve layer order.
 309     config = layer.get_config()
 310     # This will not work for nested subclassed models used as layers.
 311     # This would be theoretically possible to support, but would add complexity.
 312     # Only do it if users complain.
 313     if isinstance(layer, Network) and not layer._is_graph_network:
 314       raise ValueError('We do not support the use of nested subclassed models '
 315                        'in `model_to_estimator` at this time. Found nested '
 316                        'model: %s' % layer)
 317     fresh_layer = layer.__class__.from_config(config)
 318     name = layers_to_names[layer]
 319     setattr(model, name, fresh_layer)
 320 
 321   # Cache original model build attributes (in addition to layers)
 322   if (not hasattr(model, '_original_attributes_cache') or
 323       model._original_attributes_cache is None):
 324     if model.built:
 325       attributes_to_cache = [
 326           'inputs',
 327           'outputs',
 328           '_feed_outputs',
 329           '_feed_output_names',
 330           '_feed_output_shapes',
 331           '_feed_loss_fns',
 332           'loss_weights_list',
 333           'targets',
 334           '_feed_targets',
 335           'sample_weight_modes',
 336           'weighted_metrics',
 337           'metrics_names',
 338           'metrics_tensors',
 339           'metrics_updates',
 340           'stateful_metric_names',
 341           'total_loss',
 342           'sample_weights',
 343           '_feed_sample_weights',
 344           'train_function',
 345           'test_function',
 346           'predict_function',
 347           '_collected_trainable_weights',
 348           '_feed_inputs',
 349           '_feed_input_names',
 350           '_feed_input_shapes',
 351           'optimizer',
 352       ]
 353       for name in attributes_to_cache:
 354         attributes_cache[name] = getattr(model, name)
 355   model._original_attributes_cache = data_structures.NoDependency(
 356       attributes_cache)
 357   # Reset built state
 358   model.built = False
 359   model.inputs = None
 360   model.outputs = None

 363 def in_place_subclassed_model_state_restoration(model):
 364   """Restores the original state of a model after it was "reset".
 365 
 366   This undoes this action of `_in_place_subclassed_model_reset`, which is called
 367   in `clone_and_build_model` if `in_place_reset` is set to True.
 368 
 369   Args:
 370     model: Instance of a Keras model created via subclassing, on which
 371       `_in_place_subclassed_model_reset` was previously called.
 372   """
 373   assert not model._is_graph_network
 374   # Restore layers and build attributes
 375   if (hasattr(model, '_original_attributes_cache') and
 376       model._original_attributes_cache is not None):
 377     # Models have sticky attribute assignment, so we want to be careful to add
 378     # back the previous attributes and track Layers by their original names
 379     # without adding dependencies on "utility" attributes which Models exempt
 380     # when they're constructed.
 381     model._layers = data_structures.NoDependency([])
 382     for name, value in model._original_attributes_cache.items():
 383       if not isinstance(value, checkpointable.CheckpointableBase):
 384         # If this value is not already checkpointable, it's probably that way
 385         # for a reason; we don't want to start tracking data structures that the
 386         # original Model didn't.
 387         value = data_structures.NoDependency(value)
 388       setattr(model, name, value)
 389     model._original_attributes_cache = None
 390   else:
 391     # Restore to the state of a never-called model.
 392     model.built = False
 393     model.inputs = None
 394     model.outputs = None

 397 def clone_and_build_model(
 398     model, input_tensors=None, target_tensors=None, custom_objects=None,
 399     compile_clone=True, in_place_reset=False, optimizer_iterations=None):
 400   """Clone a `Model` and build/compile it with the same settings used before.
 401 
 402   This function can be be run in the same graph or in a separate graph from the
 403   model. When using a separate graph, `in_place_reset` must be `False`.
 404 
 405   Args:
 406     model: `tf.keras.Model` object. Can be Functional, Sequential, or
 407       sub-classed.
 408     input_tensors: Optional list of input tensors to build the model upon. If
 409       not provided, placeholders will be created.
 410     target_tensors: Optional list of target tensors for compiling the model. If
 411       not provided, placeholders will be created.
 412     custom_objects: Optional dictionary mapping string names to custom classes
 413       or functions.
 414     compile_clone: Boolean, whether to compile model clone (default `True`).
 415     in_place_reset: Boolean, whether to reset the model in place. Only used if
 416       the model is not a graph network. If the model is a subclassed model, then
 417       this argument must be set to `True` (default `False`). To restore the
 418       original model, use the function
 419       `in_place_subclassed_model_state_restoration(model)`.
 420     optimizer_iterations: An iterations variable that will be incremented by the
 421       optimizer if the clone is compiled. This argument is used when a Keras
 422       model is cloned into an Estimator model function, because Estimators
 423       create their own global step variable.
 424 
 425   Returns:
 426     Clone of the model.
 427 
 428   Raises:
 429     ValueError: if trying to clone a subclassed model, and `in_place_reset` is
 430       set to False.
 431   """
 432   if model._is_graph_network:
 433     if custom_objects:
 434       with CustomObjectScope(custom_objects):
 435         clone = clone_model(model, input_tensors=input_tensors)
 436     else:
 437       clone = clone_model(model, input_tensors=input_tensors)
 438   else:
 439     if not in_place_reset:
 440       raise ValueError(
 441           'Model is not a graph network (usually means that it is a subclassed '
 442           'model). The model cannot be cloned, but there is a workaround where '
 443           'the model is reset in-place. To use this, please set the argument '
 444           '`in_place_reset` to `True`. This will reset the attributes in the '
 445           'original model. To restore the attributes, call '
 446           '`in_place_subclassed_model_state_restoration(model)`.')
 447     clone = model
 448     _in_place_subclassed_model_reset(clone)
 449     if input_tensors is not None:
 450       if isinstance(input_tensors, (list, tuple)) and len(input_tensors) == 1:
 451         input_tensors = input_tensors[0]
 452       clone._set_inputs(input_tensors)
 453 
 454   # Compile/Build model
 455   if not compile_clone:
 456     if isinstance(clone, Sequential):
 457       clone.build()
 458   elif model.optimizer:
 459     if isinstance(model.optimizer, optimizers.TFOptimizer):
 460       optimizer = optimizers.TFOptimizer(
 461           model.optimizer.optimizer, optimizer_iterations)
 462       K.track_tf_optimizer(optimizer)
 463     else:
 464       optimizer_config = model.optimizer.get_config()
 465       optimizer = model.optimizer.__class__.from_config(optimizer_config)
 466       if optimizer_iterations is not None:
 467         optimizer.iterations = optimizer_iterations
 468 
 469     clone.compile(
 470         optimizer,
 471         model.loss,
 472         metrics=metrics_module.clone_metrics(model.metrics),
 473         loss_weights=model.loss_weights,
 474         sample_weight_mode=model.sample_weight_mode,
 475         weighted_metrics=metrics_module.clone_metrics(model.weighted_metrics),
 476         target_tensors=target_tensors)
 477 
 478   return clone
