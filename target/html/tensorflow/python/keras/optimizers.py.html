  38 @tf_export('keras.optimizers.Optimizer')
  39 class Optimizer(object):
  40   """Abstract optimizer base class.
  41 
  42   Note: this is the parent class of all optimizers, not an actual optimizer
  43   that can be used for training models.
  44 
  45   All Keras optimizers support the following keyword arguments:
  46 
  47       clipnorm: float >= 0. Gradients will be clipped
  48           when their L2 norm exceeds this value.
  49       clipvalue: float >= 0. Gradients will be clipped
  50           when their absolute value exceeds this value.
  51   """
  52 
  53   def __init__(self, **kwargs):
  54     allowed_kwargs = {'clipnorm', 'clipvalue'}
  55     for k in kwargs:
  56       if k not in allowed_kwargs:
  57         raise TypeError('Unexpected keyword argument '
  58                         'passed to optimizer: ' + str(k))
  59       # checks that clipnorm >= 0 and clipvalue >= 0
  60       if kwargs[k] < 0:
  61         raise ValueError('Expected {} >= 0, received: {}'.format(k, kwargs[k]))
  62     self.__dict__.update(kwargs)
  63     self.updates = []
  64     self.weights = []
  65 
  66   def get_updates(self, loss, params):
  67     raise NotImplementedError
  68 
  69   def get_gradients(self, loss, params):
  70     """Returns gradients of `loss` with respect to `params`.
  71 
  72     Arguments:
  73         loss: Loss tensor.
  74         params: List of variables.
  75 
  76     Returns:
  77         List of gradient tensors.
  78 
  79     Raises:
  80         ValueError: In case any gradient cannot be computed (e.g. if gradient
  81           function not implemented).
  82     """
  83     grads = K.gradients(loss, params)
  84     if None in grads:
  85       raise ValueError('An operation has `None` for gradient. '
  86                        'Please make sure that all of your ops have a '
  87                        'gradient defined (i.e. are differentiable). '
  88                        'Common ops without gradient: '
  89                        'K.argmax, K.round, K.eval.')
  90     if hasattr(self, 'clipnorm'):
  91       grads = [clip_ops.clip_by_norm(g, self.clipnorm) for g in grads]
  92     if hasattr(self, 'clipvalue'):
  93       grads = [
  94           clip_ops.clip_by_value(g, -self.clipvalue, self.clipvalue)
  95           for g in grads
  96       ]
  97     return grads
  98 
  99   def set_weights(self, weights):
 100     """Sets the weights of the optimizer, from Numpy arrays.
 101 
 102     Should only be called after computing the gradients
 103     (otherwise the optimizer has no weights).
 104 
 105     Arguments:
 106         weights: a list of Numpy arrays. The number
 107             of arrays and their shape must match
 108             number of the dimensions of the weights
 109             of the optimizer (i.e. it should match the
 110             output of `get_weights`).
 111 
 112     Raises:
 113         ValueError: in case of incompatible weight shapes.
 114     """
 115     params = self.weights
 116     if len(params) != len(weights):
 117       raise ValueError(
 118           'Length of the specified weight list (' + str(len(weights)) +
 119           ') does not match the number of weights '
 120           'of the optimizer (' + str(len(params)) + ')')
 121     weight_value_tuples = []
 122     param_values = K.batch_get_value(params)
 123     for pv, p, w in zip(param_values, params, weights):
 124       if pv.shape != w.shape:
 125         raise ValueError(
 126             'Optimizer weight shape ' + str(pv.shape) + ' not compatible with '
 127             'provided weight shape ' + str(w.shape))
 128       weight_value_tuples.append((p, w))
 129     K.batch_set_value(weight_value_tuples)
 130 
 131   def get_weights(self):
 132     """Returns the current value of the weights of the optimizer.
 133 
 134     Returns:
 135         A list of numpy arrays.
 136     """
 137     return K.batch_get_value(self.weights)
 138 
 139   def get_config(self):
 140     config = {}
 141     if hasattr(self, 'clipnorm'):
 142       config['clipnorm'] = self.clipnorm
 143     if hasattr(self, 'clipvalue'):
 144       config['clipvalue'] = self.clipvalue
 145     return config
 146 
 147   @classmethod
 148   def from_config(cls, config):
 149     return cls(**config)

 152 @tf_export('keras.optimizers.SGD')
 153 class SGD(Optimizer):
 154   """Stochastic gradient descent optimizer.
 155 
 156   Includes support for momentum,
 157   learning rate decay, and Nesterov momentum.
 158 
 159   Arguments:
 160       lr: float >= 0. Learning rate.
 161       momentum: float >= 0. Parameter that accelerates SGD
 162           in the relevant direction and dampens oscillations.
 163       decay: float >= 0. Learning rate decay over each update.
 164       nesterov: boolean. Whether to apply Nesterov momentum.
 165   """
 166 
 167   def __init__(self, lr=0.01, momentum=0., decay=0., nesterov=False, **kwargs):
 168     super(SGD, self).__init__(**kwargs)
 169     with K.name_scope(self.__class__.__name__):
 170       self.iterations = K.variable(0, dtype='int64', name='iterations')
 171       self.lr = K.variable(lr, name='lr')
 172       self.momentum = K.variable(momentum, name='momentum')
 173       self.decay = K.variable(decay, name='decay')
 174     self.initial_decay = decay
 175     self.nesterov = nesterov
 176 
 177   def get_updates(self, loss, params):
 178     grads = self.get_gradients(loss, params)
 179     self.updates = [state_ops.assign_add(self.iterations, 1)]
 180 
 181     lr = self.lr
 182     if self.initial_decay > 0:
 183       lr = lr * (  # pylint: disable=g-no-augmented-assignment
 184           1. / (1. + self.decay * math_ops.cast(self.iterations,
 185                                                 K.dtype(self.decay))))
 186     # momentum
 187     shapes = [K.int_shape(p) for p in params]
 188     moments = [K.zeros(shape) for shape in shapes]
 189     self.weights = [self.iterations] + moments
 190     for p, g, m in zip(params, grads, moments):
 191       v = self.momentum * m - lr * g  # velocity
 192       self.updates.append(state_ops.assign(m, v))
 193 
 194       if self.nesterov:
 195         new_p = p + self.momentum * v - lr * g
 196       else:
 197         new_p = p + v
 198 
 199       # Apply constraints.
 200       if getattr(p, 'constraint', None) is not None:
 201         new_p = p.constraint(new_p)
 202 
 203       self.updates.append(state_ops.assign(p, new_p))
 204     return self.updates
 205 
 206   def get_config(self):
 207     config = {
 208         'lr': float(K.get_value(self.lr)),
 209         'momentum': float(K.get_value(self.momentum)),
 210         'decay': float(K.get_value(self.decay)),
 211         'nesterov': self.nesterov
 212     }
 213     base_config = super(SGD, self).get_config()
 214     return dict(list(base_config.items()) + list(config.items()))

 217 @tf_export('keras.optimizers.RMSprop')
 218 class RMSprop(Optimizer):
 219   """RMSProp optimizer.
 220 
 221   It is recommended to leave the parameters of this optimizer
 222   at their default values
 223   (except the learning rate, which can be freely tuned).
 224 
 225   This optimizer is usually a good choice for recurrent
 226   neural networks.
 227 
 228   Arguments:
 229       lr: float >= 0. Learning rate.
 230       rho: float >= 0.
 231       epsilon: float >= 0. Fuzz factor. If `None`, defaults to `K.epsilon()`.
 232       decay: float >= 0. Learning rate decay over each update.
 233 
 234   """
 235 
 236   def __init__(self, lr=0.001, rho=0.9, epsilon=None, decay=0., **kwargs):
 237     super(RMSprop, self).__init__(**kwargs)
 238     with K.name_scope(self.__class__.__name__):
 239       self.lr = K.variable(lr, name='lr')
 240       self.rho = K.variable(rho, name='rho')
 241       self.decay = K.variable(decay, name='decay')
 242       self.iterations = K.variable(0, dtype='int64', name='iterations')
 243     if epsilon is None:
 244       epsilon = K.epsilon()
 245     self.epsilon = epsilon
 246     self.initial_decay = decay
 247 
 248   def get_updates(self, loss, params):
 249     grads = self.get_gradients(loss, params)
 250     accumulators = [K.zeros(K.int_shape(p), dtype=K.dtype(p)) for p in params]
 251     self.weights = accumulators
 252     self.updates = [state_ops.assign_add(self.iterations, 1)]
 253 
 254     lr = self.lr
 255     if self.initial_decay > 0:
 256       lr = lr * (  # pylint: disable=g-no-augmented-assignment
 257           1. / (1. + self.decay * math_ops.cast(self.iterations,
 258                                                 K.dtype(self.decay))))
 259 
 260     for p, g, a in zip(params, grads, accumulators):
 261       # update accumulator
 262       new_a = self.rho * a + (1. - self.rho) * math_ops.square(g)
 263       self.updates.append(state_ops.assign(a, new_a))
 264       new_p = p - lr * g / (K.sqrt(new_a) + self.epsilon)
 265 
 266       # Apply constraints.
 267       if getattr(p, 'constraint', None) is not None:
 268         new_p = p.constraint(new_p)
 269 
 270       self.updates.append(state_ops.assign(p, new_p))
 271     return self.updates
 272 
 273   def get_config(self):
 274     config = {
 275         'lr': float(K.get_value(self.lr)),
 276         'rho': float(K.get_value(self.rho)),
 277         'decay': float(K.get_value(self.decay)),
 278         'epsilon': self.epsilon
 279     }
 280     base_config = super(RMSprop, self).get_config()
 281     return dict(list(base_config.items()) + list(config.items()))

 344 @tf_export('keras.optimizers.Adadelta')
 345 class Adadelta(Optimizer):
 346   """Adadelta optimizer.
 347 
 348   It is recommended to leave the parameters of this optimizer
 349   at their default values.
 350 
 351   Arguments:
 352       lr: float >= 0. Learning rate.
 353           It is recommended to leave it at the default value.
 354       rho: float >= 0.
 355       epsilon: float >= 0. Fuzz factor. If `None`, defaults to `K.epsilon()`.
 356       decay: float >= 0. Learning rate decay over each update.
 357 
 358   """
 359 
 360   def __init__(self, lr=1.0, rho=0.95, epsilon=None, decay=0., **kwargs):
 361     super(Adadelta, self).__init__(**kwargs)
 362     with K.name_scope(self.__class__.__name__):
 363       self.lr = K.variable(lr, name='lr')
 364       self.decay = K.variable(decay, name='decay')
 365       self.iterations = K.variable(0, dtype='int64', name='iterations')
 366     if epsilon is None:
 367       epsilon = K.epsilon()
 368     self.rho = rho
 369     self.epsilon = epsilon
 370     self.initial_decay = decay
 371 
 372   def get_updates(self, loss, params):
 373     grads = self.get_gradients(loss, params)
 374     shapes = [K.int_shape(p) for p in params]
 375     accumulators = [K.zeros(shape) for shape in shapes]
 376     delta_accumulators = [K.zeros(shape) for shape in shapes]
 377     self.weights = accumulators + delta_accumulators
 378     self.updates = [state_ops.assign_add(self.iterations, 1)]
 379 
 380     lr = self.lr
 381     if self.initial_decay > 0:
 382       lr = lr * (  # pylint: disable=g-no-augmented-assignment
 383           1. / (1. + self.decay * math_ops.cast(self.iterations,
 384                                                 K.dtype(self.decay))))
 385 
 386     for p, g, a, d_a in zip(params, grads, accumulators, delta_accumulators):
 387       # update accumulator
 388       new_a = self.rho * a + (1. - self.rho) * math_ops.square(g)
 389       self.updates.append(state_ops.assign(a, new_a))
 390 
 391       # use the new accumulator and the *old* delta_accumulator
 392       update = g * K.sqrt(d_a + self.epsilon) / K.sqrt(new_a + self.epsilon)
 393       new_p = p - lr * update
 394 
 395       # Apply constraints.
 396       if getattr(p, 'constraint', None) is not None:
 397         new_p = p.constraint(new_p)
 398 
 399       self.updates.append(state_ops.assign(p, new_p))
 400 
 401       # update delta_accumulator
 402       new_d_a = self.rho * d_a + (1 - self.rho) * math_ops.square(update)
 403       self.updates.append(state_ops.assign(d_a, new_d_a))
 404     return self.updates
 405 
 406   def get_config(self):
 407     config = {
 408         'lr': float(K.get_value(self.lr)),
 409         'rho': self.rho,
 410         'decay': float(K.get_value(self.decay)),
 411         'epsilon': self.epsilon
 412     }
 413     base_config = super(Adadelta, self).get_config()
 414     return dict(list(base_config.items()) + list(config.items()))

 417 @tf_export('keras.optimizers.Adam')
 418 class Adam(Optimizer):
 419   """Adam optimizer.
 420 
 421   Default parameters follow those provided in the original paper.
 422 
 423   Arguments:
 424       lr: float >= 0. Learning rate.
 425       beta_1: float, 0 < beta < 1. Generally close to 1.
 426       beta_2: float, 0 < beta < 1. Generally close to 1.
 427       epsilon: float >= 0. Fuzz factor. If `None`, defaults to `K.epsilon()`.
 428       decay: float >= 0. Learning rate decay over each update.
 429       amsgrad: boolean. Whether to apply the AMSGrad variant of this
 430           algorithm from the paper "On the Convergence of Adam and
 431           Beyond".
 432 
 433   """
 434 
 435   def __init__(self,
 436                lr=0.001,
 437                beta_1=0.9,
 438                beta_2=0.999,
 439                epsilon=None,
 440                decay=0.,
 441                amsgrad=False,
 442                **kwargs):
 443     super(Adam, self).__init__(**kwargs)
 444     with K.name_scope(self.__class__.__name__):
 445       self.iterations = K.variable(0, dtype='int64', name='iterations')
 446       self.lr = K.variable(lr, name='lr')
 447       self.beta_1 = K.variable(beta_1, name='beta_1')
 448       self.beta_2 = K.variable(beta_2, name='beta_2')
 449       self.decay = K.variable(decay, name='decay')
 450     if epsilon is None:
 451       epsilon = K.epsilon()
 452     self.epsilon = epsilon
 453     self.initial_decay = decay
 454     self.amsgrad = amsgrad
 455 
 456   def get_updates(self, loss, params):
 457     grads = self.get_gradients(loss, params)
 458     self.updates = [state_ops.assign_add(self.iterations, 1)]
 459 
 460     lr = self.lr
 461     if self.initial_decay > 0:
 462       lr = lr * (  # pylint: disable=g-no-augmented-assignment
 463           1. / (1. + self.decay * math_ops.cast(self.iterations,
 464                                                 K.dtype(self.decay))))
 465 
 466     t = math_ops.cast(self.iterations, K.floatx()) + 1
 467     lr_t = lr * (
 468         K.sqrt(1. - math_ops.pow(self.beta_2, t)) /
 469         (1. - math_ops.pow(self.beta_1, t)))
 470 
 471     ms = [K.zeros(K.int_shape(p), dtype=K.dtype(p)) for p in params]
 472     vs = [K.zeros(K.int_shape(p), dtype=K.dtype(p)) for p in params]
 473     if self.amsgrad:
 474       vhats = [K.zeros(K.int_shape(p), dtype=K.dtype(p)) for p in params]
 475     else:
 476       vhats = [K.zeros(1) for _ in params]
 477     self.weights = [self.iterations] + ms + vs + vhats
 478 
 479     for p, g, m, v, vhat in zip(params, grads, ms, vs, vhats):
 480       m_t = (self.beta_1 * m) + (1. - self.beta_1) * g
 481       v_t = (self.beta_2 * v) + (1. - self.beta_2) * math_ops.square(g)
 482       if self.amsgrad:
 483         vhat_t = math_ops.maximum(vhat, v_t)
 484         p_t = p - lr_t * m_t / (K.sqrt(vhat_t) + self.epsilon)
 485         self.updates.append(state_ops.assign(vhat, vhat_t))
 486       else:
 487         p_t = p - lr_t * m_t / (K.sqrt(v_t) + self.epsilon)
 488 
 489       self.updates.append(state_ops.assign(m, m_t))
 490       self.updates.append(state_ops.assign(v, v_t))
 491       new_p = p_t
 492 
 493       # Apply constraints.
 494       if getattr(p, 'constraint', None) is not None:
 495         new_p = p.constraint(new_p)
 496 
 497       self.updates.append(state_ops.assign(p, new_p))
 498     return self.updates
 499 
 500   def get_config(self):
 501     config = {
 502         'lr': float(K.get_value(self.lr)),
 503         'beta_1': float(K.get_value(self.beta_1)),
 504         'beta_2': float(K.get_value(self.beta_2)),
 505         'decay': float(K.get_value(self.decay)),
 506         'epsilon': self.epsilon,
 507         'amsgrad': self.amsgrad
 508     }
 509     base_config = super(Adam, self).get_config()
 510     return dict(list(base_config.items()) + list(config.items()))

 513 @tf_export('keras.optimizers.Adamax')
 514 class Adamax(Optimizer):
 515   """Adamax optimizer from Adam paper's Section 7.
 516 
 517   It is a variant of Adam based on the infinity norm.
 518   Default parameters follow those provided in the paper.
 519 
 520   Arguments:
 521       lr: float >= 0. Learning rate.
 522       beta_1/beta_2: floats, 0 < beta < 1. Generally close to 1.
 523       epsilon: float >= 0. Fuzz factor. If `None`, defaults to `K.epsilon()`.
 524       decay: float >= 0. Learning rate decay over each update.
 525 
 526   """
 527 
 528   def __init__(self,
 529                lr=0.002,
 530                beta_1=0.9,
 531                beta_2=0.999,
 532                epsilon=None,
 533                decay=0.,
 534                **kwargs):
 535     super(Adamax, self).__init__(**kwargs)
 536     with K.name_scope(self.__class__.__name__):
 537       self.iterations = K.variable(0, dtype='int64', name='iterations')
 538       self.lr = K.variable(lr, name='lr')
 539       self.beta_1 = K.variable(beta_1, name='beta_1')
 540       self.beta_2 = K.variable(beta_2, name='beta_2')
 541       self.decay = K.variable(decay, name='decay')
 542     if epsilon is None:
 543       epsilon = K.epsilon()
 544     self.epsilon = epsilon
 545     self.initial_decay = decay
 546 
 547   def get_updates(self, loss, params):
 548     grads = self.get_gradients(loss, params)
 549     self.updates = [state_ops.assign_add(self.iterations, 1)]
 550 
 551     lr = self.lr
 552     if self.initial_decay > 0:
 553       lr = lr * (  # pylint: disable=g-no-augmented-assignment
 554           1. / (1. + self.decay * math_ops.cast(self.iterations,
 555                                                 K.dtype(self.decay))))
 556 
 557     t = math_ops.cast(self.iterations, K.floatx()) + 1
 558     lr_t = lr / (1. - math_ops.pow(self.beta_1, t))
 559 
 560     shapes = [K.int_shape(p) for p in params]
 561     # zero init of 1st moment
 562     ms = [K.zeros(shape) for shape in shapes]
 563     # zero init of exponentially weighted infinity norm
 564     us = [K.zeros(shape) for shape in shapes]
 565     self.weights = [self.iterations] + ms + us
 566 
 567     for p, g, m, u in zip(params, grads, ms, us):
 568 
 569       m_t = (self.beta_1 * m) + (1. - self.beta_1) * g
 570       u_t = math_ops.maximum(self.beta_2 * u, math_ops.abs(g))
 571       p_t = p - lr_t * m_t / (u_t + self.epsilon)
 572 
 573       self.updates.append(state_ops.assign(m, m_t))
 574       self.updates.append(state_ops.assign(u, u_t))
 575       new_p = p_t
 576 
 577       # Apply constraints.
 578       if getattr(p, 'constraint', None) is not None:
 579         new_p = p.constraint(new_p)
 580 
 581       self.updates.append(state_ops.assign(p, new_p))
 582     return self.updates
 583 
 584   def get_config(self):
 585     config = {
 586         'lr': float(K.get_value(self.lr)),
 587         'beta_1': float(K.get_value(self.beta_1)),
 588         'beta_2': float(K.get_value(self.beta_2)),
 589         'decay': float(K.get_value(self.decay)),
 590         'epsilon': self.epsilon
 591     }
 592     base_config = super(Adamax, self).get_config()
 593     return dict(list(base_config.items()) + list(config.items()))

 596 @tf_export('keras.optimizers.Nadam')
 597 class Nadam(Optimizer):
 598   """Nesterov Adam optimizer.
 599 
 600   Much like Adam is essentially RMSprop with momentum,
 601   Nadam is Adam RMSprop with Nesterov momentum.
 602 
 603   Default parameters follow those provided in the paper.
 604   It is recommended to leave the parameters of this optimizer
 605   at their default values.
 606 
 607   Arguments:
 608       lr: float >= 0. Learning rate.
 609       beta_1/beta_2: floats, 0 < beta < 1. Generally close to 1.
 610       epsilon: float >= 0. Fuzz factor. If `None`, defaults to `K.epsilon()`.
 611 
 612   """
 613 
 614   def __init__(self,
 615                lr=0.002,
 616                beta_1=0.9,
 617                beta_2=0.999,
 618                epsilon=None,
 619                schedule_decay=0.004,
 620                **kwargs):
 621     super(Nadam, self).__init__(**kwargs)
 622     with K.name_scope(self.__class__.__name__):
 623       self.iterations = K.variable(0, dtype='int64', name='iterations')
 624       self.m_schedule = K.variable(1., name='m_schedule')
 625       self.lr = K.variable(lr, name='lr')
 626       self.beta_1 = K.variable(beta_1, name='beta_1')
 627       self.beta_2 = K.variable(beta_2, name='beta_2')
 628     if epsilon is None:
 629       epsilon = K.epsilon()
 630     self.epsilon = epsilon
 631     self.schedule_decay = schedule_decay
 632 
 633   def get_updates(self, loss, params):
 634     grads = self.get_gradients(loss, params)
 635     self.updates = [state_ops.assign_add(self.iterations, 1)]
 636 
 637     t = math_ops.cast(self.iterations, K.floatx()) + 1
 638 
 639     # Due to the recommendations in [2], i.e. warming momentum schedule
 640     momentum_cache_t = self.beta_1 * (
 641         1. - 0.5 *
 642         (math_ops.pow(K.cast_to_floatx(0.96), t * self.schedule_decay)))
 643     momentum_cache_t_1 = self.beta_1 * (
 644         1. - 0.5 *
 645         (math_ops.pow(K.cast_to_floatx(0.96), (t + 1) * self.schedule_decay)))
 646     m_schedule_new = self.m_schedule * momentum_cache_t
 647     m_schedule_next = self.m_schedule * momentum_cache_t * momentum_cache_t_1
 648     self.updates.append((self.m_schedule, m_schedule_new))
 649 
 650     shapes = [K.int_shape(p) for p in params]
 651     ms = [K.zeros(shape) for shape in shapes]
 652     vs = [K.zeros(shape) for shape in shapes]
 653 
 654     self.weights = [self.iterations] + ms + vs
 655 
 656     for p, g, m, v in zip(params, grads, ms, vs):
 657       # the following equations given in [1]
 658       g_prime = g / (1. - m_schedule_new)
 659       m_t = self.beta_1 * m + (1. - self.beta_1) * g
 660       m_t_prime = m_t / (1. - m_schedule_next)
 661       v_t = self.beta_2 * v + (1. - self.beta_2) * math_ops.square(g)
 662       v_t_prime = v_t / (1. - math_ops.pow(self.beta_2, t))
 663       m_t_bar = (
 664           1. - momentum_cache_t) * g_prime + momentum_cache_t_1 * m_t_prime
 665 
 666       self.updates.append(state_ops.assign(m, m_t))
 667       self.updates.append(state_ops.assign(v, v_t))
 668 
 669       p_t = p - self.lr * m_t_bar / (K.sqrt(v_t_prime) + self.epsilon)
 670       new_p = p_t
 671 
 672       # Apply constraints.
 673       if getattr(p, 'constraint', None) is not None:
 674         new_p = p.constraint(new_p)
 675 
 676       self.updates.append(state_ops.assign(p, new_p))
 677     return self.updates
 678 
 679   def get_config(self):
 680     config = {
 681         'lr': float(K.get_value(self.lr)),
 682         'beta_1': float(K.get_value(self.beta_1)),
 683         'beta_2': float(K.get_value(self.beta_2)),
 684         'epsilon': self.epsilon,
 685         'schedule_decay': self.schedule_decay
 686     }
 687     base_config = super(Nadam, self).get_config()
 688     return dict(list(base_config.items()) + list(config.items()))

 691 class TFOptimizer(Optimizer, checkpointable.CheckpointableBase):
 692   """Wrapper class for native TensorFlow optimizers.
 693   """
 694 
 695   def __init__(self, optimizer, iterations=None):  # pylint: disable=super-init-not-called
 696     self.optimizer = optimizer
 697     self._track_checkpointable(optimizer, name='optimizer')
 698     if iterations is None:
 699       with K.name_scope(self.__class__.__name__):
 700         self.iterations = K.variable(0, dtype='int64', name='iterations')
 701     else:
 702       self.iterations = iterations
 703     self._track_checkpointable(self.iterations, name='global_step')
 704 
 705   def apply_gradients(self, grads):
 706     self.optimizer.apply_gradients(grads, global_step=self.iterations)
 707 
 708   def get_grads(self, loss, params):
 709     return self.optimizer.compute_gradients(loss, params)
 710 
 711   def get_updates(self, loss, params):
 712     if distribution_strategy_context.has_distribution_strategy():
 713       self.updates = []
 714 
 715       if not params:
 716         # After the model vars have been created, the second call to get_updates
 717         # is called with params as an empty list. This ensures that we call
 718         # compute_gradients with params=None.
 719         grads = self.optimizer.compute_gradients(loss)
 720       else:
 721         grads = self.optimizer.compute_gradients(loss, params)
 722       global_step = training_util.get_global_step()
 723       opt_update = self.optimizer.apply_gradients(grads, global_step)
 724     else:
 725       if not params:
 726         self.updates = [state_ops.assign_add(self.iterations, 1)]
 727         return self.updates
 728 
 729       # Updates list starts out empty because the iterations variable is
 730       # incremented in optimizer.apply_gradients()
 731       self.updates = []
 732       grads = self.optimizer.compute_gradients(loss, params)
 733       opt_update = self.optimizer.apply_gradients(
 734           grads, global_step=self.iterations)
 735 
 736     self.updates.append(opt_update)
 737     return self.updates

 768   """Inverse of the `serialize` function.
 769 
 770   Arguments:
 771       config: Optimizer configuration dictionary.
 772       custom_objects: Optional dictionary mapping
 773           names (strings) to custom objects
 774           (classes and functions)
 775           to be considered during deserialization.
 776 
 777   Returns:
 778       A Keras Optimizer instance.

 781       'sgd': SGD,
 782       'rmsprop': RMSprop,

 786       'adamax': Adamax,
 787       'nadam': Nadam,
 788       'tfoptimizer': TFOptimizer,

 791   if config['class_name'].lower() in all_classes:
 792     config['class_name'] = config['class_name'].lower()
 793   return deserialize_keras_object(
 794       config,
 795       module_objects=all_classes,

 800 @tf_export('keras.optimizers.get')
 801 def get(identifier):
 802   """Retrieves a Keras Optimizer instance.
 803 
 804   Arguments:
 805       identifier: Optimizer identifier, one of
 806           - String: name of an optimizer
 807           - Dictionary: configuration dictionary.
 808           - Keras Optimizer instance (it will be returned unchanged).
 809           - TensorFlow Optimizer instance
 810               (it will be wrapped as a Keras Optimizer).
 811 
 812   Returns:
 813       A Keras Optimizer instance.
 814 
 815   Raises:
 816       ValueError: If `identifier` cannot be interpreted.
 817   """
 818   # Wrap TF optimizer instances
 819   if isinstance(identifier, tf_optimizer_module.Optimizer):
 820     opt = TFOptimizer(identifier)
 821     K.track_tf_optimizer(opt)
 822     return opt
 823   if isinstance(identifier, dict):
 824     return deserialize(identifier)
 825   elif isinstance(identifier, six.string_types):
 826     config = {'class_name': str(identifier), 'config': {}}
 827     return deserialize(config)
 828   if isinstance(identifier, Optimizer):
 829     return identifier
 830   else:
 831     raise ValueError('Could not interpret optimizer identifier:', identifier)
