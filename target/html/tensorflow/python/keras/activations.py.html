  30 @tf_export('keras.activations.softmax')
  31 def softmax(x, axis=-1):
  32   """Softmax activation function.
  33 
  34   Arguments:
  35       x : Input tensor.
  36       axis: Integer, axis along which the softmax normalization is applied.
  37 
  38   Returns:
  39       Tensor, output of softmax transformation.
  40 
  41   Raises:
  42       ValueError: In case `dim(x) == 1`.
  43   """
  44   ndim = K.ndim(x)
  45   if ndim == 2:
  46     return nn.softmax(x)
  47   elif ndim > 2:
  48     e = math_ops.exp(x - math_ops.reduce_max(x, axis=axis, keepdims=True))
  49     s = math_ops.reduce_sum(e, axis=axis, keepdims=True)
  50     return e / s
  51   else:
  52     raise ValueError('Cannot apply softmax to a tensor that is 1D. '
  53                      'Received input: %s' % (x,))

  56 @tf_export('keras.activations.elu')
  57 def elu(x, alpha=1.0):
  58   """Exponential linear unit.
  59 
  60   Arguments:
  61       x: Input tensor.
  62       alpha: A scalar, slope of negative section.
  63 
  64   Returns:
  65       The exponential linear activation: `x` if `x > 0` and
  66         `alpha * (exp(x)-1)` if `x < 0`.
  67 
  68   Reference:
  69       - [Fast and Accurate Deep Network Learning by Exponential
  70         Linear Units (ELUs)](https://arxiv.org/abs/1511.07289)
  71   """
  72   return K.elu(x, alpha)

  75 @tf_export('keras.activations.selu')
  76 def selu(x):
  77   """Scaled Exponential Linear Unit (SELU).
  78 
  79   SELU is equal to: `scale * elu(x, alpha)`, where alpha and scale
  80   are pre-defined constants. The values of `alpha` and `scale` are
  81   chosen so that the mean and variance of the inputs are preserved
  82   between two consecutive layers as long as the weights are initialized
  83   correctly (see `lecun_normal` initialization) and the number of inputs
  84   is "large enough" (see references for more information).
  85 
  86   Arguments:
  87       x: A tensor or variable to compute the activation function for.
  88 
  89   Returns:
  90       The scaled exponential unit activation: `scale * elu(x, alpha)`.
  91 
  92   # Note
  93       - To be used together with the initialization "lecun_normal".
  94       - To be used together with the dropout variant "AlphaDropout".
  95 
  96   References:
  97       - [Self-Normalizing Neural Networks](https://arxiv.org/abs/1706.02515)
  98   """
  99   alpha = 1.6732632423543772848170429916717
 100   scale = 1.0507009873554804934193349852946
 101   return scale * K.elu(x, alpha)

 130 @tf_export('keras.activations.relu')
 131 def relu(x, alpha=0., max_value=None, threshold=0):
 132   """Rectified Linear Unit.
 133 
 134   With default values, it returns element-wise `max(x, 0)`.
 135 
 136   Otherwise, it follows:
 137   `f(x) = max_value` for `x >= max_value`,
 138   `f(x) = x` for `threshold <= x < max_value`,
 139   `f(x) = alpha * (x - threshold)` otherwise.
 140 
 141   Arguments:
 142       x: A tensor or variable.
 143       alpha: A scalar, slope of negative section (default=`0.`).
 144       max_value: float. Saturation threshold.
 145       threshold: float. Threshold value for thresholded activation.
 146 
 147   Returns:
 148       A tensor.
 149   """
 150   return K.relu(x, alpha=alpha, max_value=max_value, threshold=threshold)

 163 @tf_export('keras.activations.hard_sigmoid')
 164 def hard_sigmoid(x):
 165   """Hard sigmoid activation function.
 166 
 167   Faster to compute than sigmoid activation.
 168 
 169   Arguments:
 170       x: Input tensor.
 171 
 172   Returns:
 173       Hard sigmoid activation:
 174       - `0` if `x < -2.5`

 177   """
 178   return K.hard_sigmoid(x)

 200 @tf_export('keras.activations.get')
 201 def get(identifier):
 202   if identifier is None:
 203     return linear
 204   if isinstance(identifier, six.string_types):
 205     identifier = str(identifier)
 206     return deserialize(identifier)
 207   elif callable(identifier):
 208     return identifier
 209   else:
 210     raise ValueError('Could not interpret '
 211                      'activation function identifier:', identifier)
