  28 def convert_data_format(data_format, ndim):
  29   if data_format == 'channels_last':
  30     if ndim == 3:
  31       return 'NWC'
  32     elif ndim == 4:
  33       return 'NHWC'
  34     elif ndim == 5:
  35       return 'NDHWC'
  36     else:
  37       raise ValueError('Input rank not supported:', ndim)
  38   elif data_format == 'channels_first':
  39     if ndim == 3:
  40       return 'NCW'
  41     elif ndim == 4:
  42       return 'NCHW'
  43     elif ndim == 5:
  44       return 'NCDHW'
  45     else:
  46       raise ValueError('Input rank not supported:', ndim)
  47   else:
  48     raise ValueError('Invalid data_format:', data_format)

  51 def normalize_tuple(value, n, name):
  52   """Transforms a single integer or iterable of integers into an integer tuple.
  53 
  54   Arguments:
  55     value: The value to validate and convert. Could an int, or any iterable
  56       of ints.
  57     n: The size of the tuple to be returned.
  58     name: The name of the argument being validated, e.g. "strides" or
  59       "kernel_size". This is only used to format error messages.
  60 
  61   Returns:
  62     A tuple of n integers.
  63 
  64   Raises:
  65     ValueError: If something else than an int/long or iterable thereof was
  66       passed.
  67   """
  68   if isinstance(value, int):
  69     return (value,) * n
  70   else:
  71     try:
  72       value_tuple = tuple(value)
  73     except TypeError:
  74       raise ValueError('The `' + name + '` argument must be a tuple of ' +
  75                        str(n) + ' integers. Received: ' + str(value))
  76     if len(value_tuple) != n:
  77       raise ValueError('The `' + name + '` argument must be a tuple of ' +
  78                        str(n) + ' integers. Received: ' + str(value))
  79     for single_value in value_tuple:
  80       try:
  81         int(single_value)
  82       except (ValueError, TypeError):
  83         raise ValueError('The `' + name + '` argument must be a tuple of ' +
  84                          str(n) + ' integers. Received: ' + str(value) + ' '
  85                          'including element ' + str(single_value) + ' of type' +
  86                          ' ' + str(type(single_value)))
  87     return value_tuple

  90 def normalize_data_format(value):
  91   data_format = value.lower()
  92   if data_format not in {'channels_first', 'channels_last'}:
  93     raise ValueError('The `data_format` argument must be one of '
  94                      '"channels_first", "channels_last". Received: ' +
  95                      str(value))
  96   return data_format

  99 def normalize_padding(value):
 100   padding = value.lower()
 101   if padding not in {'valid', 'same'}:
 102     raise ValueError('The `padding` argument must be one of "valid", "same". '
 103                      'Received: ' + str(padding))
 104   return padding

 107 def conv_output_length(input_length, filter_size, padding, stride, dilation=1):
 108   """Determines output length of a convolution given input length.
 109 
 110   Arguments:
 111       input_length: integer.
 112       filter_size: integer.
 113       padding: one of "same", "valid", "full".
 114       stride: integer.
 115       dilation: dilation rate, integer.
 116 
 117   Returns:
 118       The output length (integer).
 119   """
 120   if input_length is None:
 121     return None
 122   assert padding in {'same', 'valid', 'full'}
 123   dilated_filter_size = filter_size + (filter_size - 1) * (dilation - 1)
 124   if padding == 'same':
 125     output_length = input_length
 126   elif padding == 'valid':
 127     output_length = input_length - dilated_filter_size + 1
 128   elif padding == 'full':
 129     output_length = input_length + dilated_filter_size - 1
 130   return (output_length + stride - 1) // stride

 133 def conv_input_length(output_length, filter_size, padding, stride):
 134   """Determines input length of a convolution given output length.
 135 
 136   Arguments:
 137       output_length: integer.
 138       filter_size: integer.
 139       padding: one of "same", "valid", "full".
 140       stride: integer.
 141 
 142   Returns:
 143       The input length (integer).
 144   """
 145   if output_length is None:
 146     return None
 147   assert padding in {'same', 'valid', 'full'}
 148   if padding == 'same':
 149     pad = filter_size // 2
 150   elif padding == 'valid':
 151     pad = 0
 152   elif padding == 'full':
 153     pad = filter_size - 1
 154   return (output_length - 1) * stride - 2 * pad + filter_size

 157 def deconv_output_length(input_length, filter_size, padding, stride):
 158   """Determines output length of a transposed convolution given input length.
 159 
 160   Arguments:
 161       input_length: integer.
 162       filter_size: integer.
 163       padding: one of "same", "valid", "full".
 164       stride: integer.
 165 
 166   Returns:
 167       The output length (integer).
 168   """
 169   if input_length is None:
 170     return None
 171   input_length *= stride
 172   if padding == 'valid':
 173     input_length += max(filter_size - stride, 0)
 174   elif padding == 'full':
 175     input_length -= (stride + filter_size - 2)
 176   return input_length

 179 def smart_cond(pred, true_fn=None, false_fn=None, name=None):
 180   """Return either `true_fn()` if predicate `pred` is true else `false_fn()`.
 181 
 182   If `pred` is a bool or has a constant value, we return either `true_fn()`
 183   or `false_fn()`, otherwise we use `tf.cond` to dynamically route to both.
 184 
 185   Arguments:
 186     pred: A scalar determining whether to return the result of `true_fn` or
 187       `false_fn`.
 188     true_fn: The callable to be performed if pred is true.
 189     false_fn: The callable to be performed if pred is false.
 190     name: Optional name prefix when using `tf.cond`.
 191 
 192   Returns:
 193     Tensors returned by the call to either `true_fn` or `false_fn`.
 194 
 195   Raises:
 196     TypeError: If `true_fn` or `false_fn` is not callable.
 197   """
 198   if isinstance(pred, variables.Variable):
 199     return control_flow_ops.cond(
 200         pred, true_fn=true_fn, false_fn=false_fn, name=name)
 201   return smart_module.smart_cond(
 202       pred, true_fn=true_fn, false_fn=false_fn, name=name)

 205 def constant_value(pred):
 206   """Return the bool value for `pred`, or None if `pred` had a dynamic value.
 207 
 208     Arguments:
 209       pred: A scalar, either a Python bool or a TensorFlow boolean variable
 210         or tensor, or the Python integer 1 or 0.
 211 
 212     Returns:
 213       True or False if `pred` has a constant boolean value, None otherwise.
 214 
 215     Raises:
 216       TypeError: If `pred` is not a Variable, Tensor or bool, or Python
 217         interger 1 or 0.
 218     """
 219   # Allow integer booleans.
 220   if isinstance(pred, int):
 221     if pred == 1:
 222       pred = True
 223     elif pred == 0:
 224       pred = False
 225 
 226   if isinstance(pred, variables.Variable):
 227     return None
 228   return smart_module.smart_constant_value(pred)

 231 def object_list_uid(object_list):
 232   """Creates a single string from object ids."""
 233   object_list = nest.flatten(object_list)
 234   return ', '.join([str(abs(id(x))) for x in object_list])

 237 def static_shape(x):
 238   """Get the static shape of a Tensor, or None if it is unavailable."""
 239   if x is None:
 240     return None
 241   try:
 242     return tuple(x.get_shape().as_list())
 243   except ValueError:
 244     return None

 247 def get_reachable_from_inputs(inputs, targets=None):
 248   """Returns the set of tensors reachable from `inputs`.
 249 
 250   Stops if all targets have been found (target is optional).
 251 
 252   Only valid in Symbolic mode, not Eager mode.
 253 
 254   Args:
 255     inputs: List of tensors.
 256     targets: List of tensors.
 257 
 258   Returns:
 259     A set of tensors reachable from the inputs (includes the inputs themselves).
 260   """
 261   reachable = set(inputs)
 262   if targets:
 263     targets = set(targets)
 264   queue = inputs[:]
 265 
 266   while queue:
 267     x = queue.pop()
 268     outputs = []
 269     try:
 270       consumers = x.consumers()
 271     except AttributeError:
 272       # Case where x is a variable type
 273       consumers = [x.op]
 274     for z in consumers:
 275       consumer_outputs = z.outputs
 276       if consumer_outputs:  # May be None
 277         outputs += consumer_outputs
 278 
 279     for y in outputs:
 280       if y not in reachable:
 281         reachable.add(y)
 282         queue.insert(0, y)
 283 
 284     if targets and targets.issubset(reachable):
 285       return reachable
 286   return reachable
