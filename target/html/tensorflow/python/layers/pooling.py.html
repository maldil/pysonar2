  27 @tf_export('layers.AveragePooling1D')
  28 class AveragePooling1D(keras_layers.AveragePooling1D, base.Layer):
  29   """Average Pooling layer for 1D inputs.
  30 
  31   Arguments:
  32     pool_size: An integer or tuple/list of a single integer,
  33       representing the size of the pooling window.
  34     strides: An integer or tuple/list of a single integer, specifying the
  35       strides of the pooling operation.
  36     padding: A string. The padding method, either 'valid' or 'same'.
  37       Case-insensitive.
  38     data_format: A string, one of `channels_last` (default) or `channels_first`.
  39       The ordering of the dimensions in the inputs.
  40       `channels_last` corresponds to inputs with shape
  41       `(batch, length, channels)` while `channels_first` corresponds to
  42       inputs with shape `(batch, channels, length)`.
  43     name: A string, the name of the layer.
  44   """
  45 
  46   def __init__(self, pool_size, strides,
  47                padding='valid', data_format='channels_last',
  48                name=None, **kwargs):
  49     if strides is None:
  50       raise ValueError('Argument `strides` must not be None.')
  51     super(AveragePooling1D, self).__init__(
  52         pool_size=pool_size,
  53         strides=strides,
  54         padding=padding,
  55         data_format=data_format,
  56         name=name,
  57         **kwargs)

  78       inputs with shape `(batch, channels, length)`.
  79     name: A string, the name of the layer.
  80 
  81   Returns:
