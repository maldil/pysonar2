  29 @tf_export('layers.BatchNormalization')
  30 class BatchNormalization(keras_layers.BatchNormalization, base.Layer):
  31   """Batch Normalization layer from http://arxiv.org/abs/1502.03167.
  32 
  33   "Batch Normalization: Accelerating Deep Network Training by Reducing
  34   Internal Covariate Shift"
  35 
  36   Sergey Ioffe, Christian Szegedy
  37 
  38   Arguments:
  39     axis: An `int` or list of `int`, the axis or axes that should be
  40         normalized, typically the features axis/axes. For instance, after a
  41         `Conv2D` layer with `data_format="channels_first"`, set `axis=1`. If a
  42         list of axes is provided, each axis in `axis` will be normalized
  43         simultaneously. Default is `-1` which uses the last axis. Note: when
  44         using multi-axis batch norm, the `beta`, `gamma`, `moving_mean`, and
  45         `moving_variance` variables are the same rank as the input Tensor, with
  46         dimension size 1 in all reduced (non-axis) dimensions).
  47     momentum: Momentum for the moving average.
  48     epsilon: Small float added to variance to avoid dividing by zero.
  49     center: If True, add offset of `beta` to normalized tensor. If False, `beta`
  50       is ignored.
  51     scale: If True, multiply by `gamma`. If False, `gamma` is
  52       not used. When the next layer is linear (also e.g. `nn.relu`), this can be
  53       disabled since the scaling can be done by the next layer.
  54     beta_initializer: Initializer for the beta weight.
  55     gamma_initializer: Initializer for the gamma weight.
  56     moving_mean_initializer: Initializer for the moving mean.
  57     moving_variance_initializer: Initializer for the moving variance.
  58     beta_regularizer: Optional regularizer for the beta weight.
  59     gamma_regularizer: Optional regularizer for the gamma weight.
  60     beta_constraint: An optional projection function to be applied to the `beta`
  61         weight after being updated by an `Optimizer` (e.g. used to implement
  62         norm constraints or value constraints for layer weights). The function
  63         must take as input the unprojected variable and must return the
  64         projected variable (which must have the same shape). Constraints are
  65         not safe to use when doing asynchronous distributed training.
  66     gamma_constraint: An optional projection function to be applied to the
  67         `gamma` weight after being updated by an `Optimizer`.
  68     renorm: Whether to use Batch Renormalization
  69       (https://arxiv.org/abs/1702.03275). This adds extra variables during
  70       training. The inference is the same for either value of this parameter.
  71     renorm_clipping: A dictionary that may map keys 'rmax', 'rmin', 'dmax' to
  72       scalar `Tensors` used to clip the renorm correction. The correction
  73       `(r, d)` is used as `corrected_value = normalized_value * r + d`, with
  74       `r` clipped to [rmin, rmax], and `d` to [-dmax, dmax]. Missing rmax, rmin,
  75       dmax are set to inf, 0, inf, respectively.
  76     renorm_momentum: Momentum used to update the moving means and standard
  77       deviations with renorm. Unlike `momentum`, this affects training
  78       and should be neither too small (which would add noise) nor too large
  79       (which would give stale estimates). Note that `momentum` is still applied
  80       to get the means and variances for inference.
  81     fused: if `None` or `True`, use a faster, fused implementation if possible.
  82       If `False`, use the system recommended implementation.
  83     trainable: Boolean, if `True` also add variables to the graph collection
  84       `GraphKeys.TRAINABLE_VARIABLES` (see tf.Variable).
  85     virtual_batch_size: An `int`. By default, `virtual_batch_size` is `None`,
  86       which means batch normalization is performed across the whole batch. When
  87       `virtual_batch_size` is not `None`, instead perform "Ghost Batch
  88       Normalization", which creates virtual sub-batches which are each
  89       normalized separately (with shared gamma, beta, and moving statistics).
  90       Must divide the actual batch size during execution.
  91     adjustment: A function taking the `Tensor` containing the (dynamic) shape of
  92       the input tensor and returning a pair (scale, bias) to apply to the
  93       normalized values (before gamma and beta), only during training. For
  94       example, if axis==-1,
  95         `adjustment = lambda shape: (
  96           tf.random_uniform(shape[-1:], 0.93, 1.07),
  97           tf.random_uniform(shape[-1:], -0.1, 0.1))`
  98       will scale the normalized value by up to 7% up or down, then shift the
  99       result by up to 0.1 (with independent scaling and bias for each feature
 100       but shared across all examples), and finally apply gamma and/or beta. If
 101       `None`, no adjustment is applied. Cannot be specified if
 102       virtual_batch_size is specified.
 103     name: A string, the name of the layer.
 104   """

 157 @tf_export('layers.batch_normalization')
 158 def batch_normalization(inputs,
 159                         axis=-1,
 160                         momentum=0.99,
 161                         epsilon=1e-3,
 162                         center=True,
 163                         scale=True,
 164                         beta_initializer=init_ops.zeros_initializer(),
 165                         gamma_initializer=init_ops.ones_initializer(),
 166                         moving_mean_initializer=init_ops.zeros_initializer(),
 167                         moving_variance_initializer=init_ops.ones_initializer(),
 168                         beta_regularizer=None,
 169                         gamma_regularizer=None,
 170                         beta_constraint=None,
 171                         gamma_constraint=None,
 172                         training=False,
 173                         trainable=True,
 174                         name=None,
 175                         reuse=None,
 176                         renorm=False,
 177                         renorm_clipping=None,
 178                         renorm_momentum=0.99,
 179                         fused=None,
 180                         virtual_batch_size=None,
 181                         adjustment=None):
 182   """Functional interface for the batch normalization layer.
 183 
 184   Reference: http://arxiv.org/abs/1502.03167
 185 
 186   "Batch Normalization: Accelerating Deep Network Training by Reducing
 187   Internal Covariate Shift"
 188 
 189   Sergey Ioffe, Christian Szegedy
 190 
 191   Note: when training, the moving_mean and moving_variance need to be updated.
 192   By default the update ops are placed in `tf.GraphKeys.UPDATE_OPS`, so they
 193   need to be added as a dependency to the `train_op`. Also, be sure to add
 194   any batch_normalization ops before getting the update_ops collection.
 195   Otherwise, update_ops will be empty, and training/inference will not work
 196   properly. For example:
 197 
 198   ```python
 199     x_norm = tf.layers.batch_normalization(x, training=training)
 200 
 201     # ...
 202 
 203     update_ops = tf.get_collection(tf.GraphKeys.UPDATE_OPS)
 204     with tf.control_dependencies(update_ops):
 205       train_op = optimizer.minimize(loss)
 206   ```
 207 
 208   Arguments:
 209     inputs: Tensor input.
 210     axis: An `int`, the axis that should be normalized (typically the features
 211       axis). For instance, after a `Convolution2D` layer with
 212       `data_format="channels_first"`, set `axis=1` in `BatchNormalization`.
 213     momentum: Momentum for the moving average.
 214     epsilon: Small float added to variance to avoid dividing by zero.
 215     center: If True, add offset of `beta` to normalized tensor. If False, `beta`
 216       is ignored.
 217     scale: If True, multiply by `gamma`. If False, `gamma` is
 218       not used. When the next layer is linear (also e.g. `nn.relu`), this can be
 219       disabled since the scaling can be done by the next layer.
 220     beta_initializer: Initializer for the beta weight.
 221     gamma_initializer: Initializer for the gamma weight.
 222     moving_mean_initializer: Initializer for the moving mean.
 223     moving_variance_initializer: Initializer for the moving variance.
 224     beta_regularizer: Optional regularizer for the beta weight.
 225     gamma_regularizer: Optional regularizer for the gamma weight.
 226     beta_constraint: An optional projection function to be applied to the `beta`
 227         weight after being updated by an `Optimizer` (e.g. used to implement
 228         norm constraints or value constraints for layer weights). The function
 229         must take as input the unprojected variable and must return the
 230         projected variable (which must have the same shape). Constraints are
 231         not safe to use when doing asynchronous distributed training.
 232     gamma_constraint: An optional projection function to be applied to the
 233         `gamma` weight after being updated by an `Optimizer`.
 234     training: Either a Python boolean, or a TensorFlow boolean scalar tensor
 235       (e.g. a placeholder). Whether to return the output in training mode
 236       (normalized with statistics of the current batch) or in inference mode
 237       (normalized with moving statistics). **NOTE**: make sure to set this
 238       parameter correctly, or else your training/inference will not work
 239       properly.
 240     trainable: Boolean, if `True` also add variables to the graph collection
 241       `GraphKeys.TRAINABLE_VARIABLES` (see tf.Variable).
 242     name: String, the name of the layer.
 243     reuse: Boolean, whether to reuse the weights of a previous layer
 244       by the same name.
 245     renorm: Whether to use Batch Renormalization
 246       (https://arxiv.org/abs/1702.03275). This adds extra variables during
 247       training. The inference is the same for either value of this parameter.
 248     renorm_clipping: A dictionary that may map keys 'rmax', 'rmin', 'dmax' to
 249       scalar `Tensors` used to clip the renorm correction. The correction
 250       `(r, d)` is used as `corrected_value = normalized_value * r + d`, with
 251       `r` clipped to [rmin, rmax], and `d` to [-dmax, dmax]. Missing rmax, rmin,
 252       dmax are set to inf, 0, inf, respectively.
 253     renorm_momentum: Momentum used to update the moving means and standard
 254       deviations with renorm. Unlike `momentum`, this affects training
 255       and should be neither too small (which would add noise) nor too large
 256       (which would give stale estimates). Note that `momentum` is still applied
 257       to get the means and variances for inference.
 258     fused: if `None` or `True`, use a faster, fused implementation if possible.
 259       If `False`, use the system recommended implementation.
 260     virtual_batch_size: An `int`. By default, `virtual_batch_size` is `None`,
 261       which means batch normalization is performed across the whole batch. When
 262       `virtual_batch_size` is not `None`, instead perform "Ghost Batch
 263       Normalization", which creates virtual sub-batches which are each
 264       normalized separately (with shared gamma, beta, and moving statistics).
 265       Must divide the actual batch size during execution.
 266     adjustment: A function taking the `Tensor` containing the (dynamic) shape of
 267       the input tensor and returning a pair (scale, bias) to apply to the
 268       normalized values (before gamma and beta), only during training. For
 269       example, if axis==-1,
 270         `adjustment = lambda shape: (
 271           tf.random_uniform(shape[-1:], 0.93, 1.07),
 272           tf.random_uniform(shape[-1:], -0.1, 0.1))`
 273       will scale the normalized value by up to 7% up or down, then shift the
 274       result by up to 0.1 (with independent scaling and bias for each feature
 275       but shared across all examples), and finally apply gamma and/or beta. If
 276       `None`, no adjustment is applied. Cannot be specified if
 277       virtual_batch_size is specified.
 278 
 279   Returns:
 280     Output tensor.
 281 
 282   Raises:
 283     ValueError: if eager execution is enabled.
 284   """
 285   layer = BatchNormalization(
 286       axis=axis,
 287       momentum=momentum,
 288       epsilon=epsilon,
 289       center=center,
 290       scale=scale,
 291       beta_initializer=beta_initializer,
 292       gamma_initializer=gamma_initializer,
 293       moving_mean_initializer=moving_mean_initializer,
 294       moving_variance_initializer=moving_variance_initializer,
 295       beta_regularizer=beta_regularizer,
 296       gamma_regularizer=gamma_regularizer,
 297       beta_constraint=beta_constraint,
 298       gamma_constraint=gamma_constraint,
 299       renorm=renorm,
 300       renorm_clipping=renorm_clipping,
 301       renorm_momentum=renorm_momentum,
 302       fused=fused,
 303       trainable=trainable,
 304       virtual_batch_size=virtual_batch_size,
 305       adjustment=adjustment,
 306       name=name,
 307       _reuse=reuse,
 308       _scope=name)
 309   return layer.apply(inputs, training=training)
