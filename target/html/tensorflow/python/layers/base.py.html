  36 @tf_export('layers.Layer')
  37 class Layer(base_layer.Layer):
  38   """Base layer class.
  39 
  40   It is considered legacy, and we recommend the use of `tf.keras.layers.Layer`
  41   instead.
  42 
  43   Arguments:
  44     trainable: Boolean, whether the layer's variables should be trainable.
  45     name: String name of the layer.
  46     dtype: Default dtype of the layer's weights (default of `None` means use the
  47       type of the first input).
  48 
  49   Read-only properties:
  50     name: The name of the layer (string).
  51     dtype: Default dtype of the layer's weights (default of `None` means use the
  52       type of the first input).
  53     trainable_variables: List of trainable variables.
  54     non_trainable_variables: List of non-trainable variables.
  55     variables: List of all variables of this layer, trainable and
  56       non-trainable.
  57     updates: List of update ops of this layer.
  58     losses: List of losses added by this layer.
  59     trainable_weights: List of variables to be included in backprop.
  60     non_trainable_weights: List of variables that should not be
  61       included in backprop.
  62     weights: The concatenation of the lists trainable_weights and
  63       non_trainable_weights (in this order).
  64 
  65   Mutable properties:
  66     trainable: Whether the layer should be trained (boolean).
  67     input_spec: Optional (list of) `InputSpec` object(s) specifying the
  68       constraints on inputs that can be accepted by the layer.
  69   """
  70 
  71   def __init__(self, trainable=True, name=None, dtype=None,
  72                **kwargs):
  73     # For backwards compatibility, legacy layers do not use `ResourceVariable`
  74     # by default.
  75     self._use_resource_variables = False
  76     scope = kwargs.pop('_scope', None)
  77     self._reuse = kwargs.pop('_reuse', None)
  78 
  79     # Avoid an incorrect lint error
  80     self._trainable_weights = []
  81     self.built = False
  82 
  83     super(Layer, self).__init__(trainable=trainable, name=name, dtype=dtype,
  84                                 **kwargs)
  85 
  86     self._graph = None
  87     self._call_has_scope_arg = 'scope' in self._call_fn_args
  88     if scope:
  89       with vs.variable_scope(scope) as captured_scope:
  90         self._scope = captured_scope
  91     else:
  92       self._scope = None
  93     self._current_scope = None
  94 
  95   @property
  96   def graph(self):
  97     if context.executing_eagerly():
  98       raise RuntimeError('Layer.graph not supported when executing eagerly.')
  99     return self._graph

 101   def _init_set_name(self, name):
 102     # Determine layer name (non-unique).
 103     if isinstance(name, vs.VariableScope):
 104       base_name = name.name
 105     else:
 106       base_name = name
 107       self._name = name
 108     if not name:
 109       self._name, base_name = self._make_unique_name()
 110     self._base_name = base_name
 111 
 112   def _make_unique_name(self, name_uid_map=None, avoid_names=None,
 113                         namespace='', zero_based=False):
 114     base_name = base_layer.to_snake_case(self.__class__.__name__)
 115     name = base_layer.unique_layer_name(base_name,
 116                                         name_uid_map=name_uid_map,
 117                                         avoid_names=avoid_names,
 118                                         namespace=namespace,
 119                                         zero_based=zero_based)
 120     return (name, base_name)
 121 
 122   @property
 123   def scope_name(self):
 124     if not self._scope:
 125       raise ValueError('No name available for layer scope because the layer "' +
 126                        self._name + '" has not been used yet. The scope name ' +
 127                        ' is determined the first time the layer instance is ' +
 128                        'called. You must therefore call the layer before ' +
 129                        'querying `scope_name`.')
 130     return self._scope.name

 132   def add_loss(self, losses, inputs=None):
 133     previous_losses_length = len(self._losses)
 134     previous_callable_losses_length = len(self._callable_losses)
 135     super(Layer, self).add_loss(losses, inputs=inputs)
 136     if not context.executing_eagerly():
 137       # TODO(fchollet): deprecate collection below.
 138       new_losses = self._losses[previous_losses_length:]
 139       new_callable_losses = self._callable_losses[
 140           previous_callable_losses_length:]
 141       for regularizer in new_callable_losses:
 142         loss_tensor = regularizer()
 143         if loss_tensor is not None:
 144           new_losses.append(loss_tensor)
 145       _add_elements_to_collection(
 146           new_losses,
 147           ops.GraphKeys.REGULARIZATION_LOSSES)

 149   def _name_scope(self):
 150     """Determines op naming for the Layer."""
 151     return self._current_scope.original_name_scope
 152 
 153   def _set_scope(self, scope=None):
 154     if self._scope is None:
 155       # If constructed with _scope=None, lazy setting of scope.
 156       if self._reuse:
 157         with vs.variable_scope(
 158             scope if scope is not None else self._base_name) as captured_scope:
 159           self._scope = captured_scope
 160       else:
 161         with vs.variable_scope(
 162             scope, default_name=self._base_name) as captured_scope:
 163           self._scope = captured_scope

 165   def add_weight(self,
 166                  name,
 167                  shape,
 168                  dtype=None,
 169                  initializer=None,
 170                  regularizer=None,
 171                  trainable=None,
 172                  constraint=None,
 173                  use_resource=None,
 174                  synchronization=vs.VariableSynchronization.AUTO,
 175                  aggregation=vs.VariableAggregation.NONE,
 176                  partitioner=None):
 177     """Adds a new variable to the layer, or gets an existing one; returns it.
 178 
 179     Arguments:
 180       name: variable name.
 181       shape: variable shape.
 182       dtype: The type of the variable. Defaults to `self.dtype` or `float32`.
 183       initializer: initializer instance (callable).
 184       regularizer: regularizer instance (callable).
 185       trainable: whether the variable should be part of the layer's
 186         "trainable_variables" (e.g. variables, biases)
 187         or "non_trainable_variables" (e.g. BatchNorm mean, stddev).
 188         Note, if the current variable scope is marked as non-trainable
 189         then this parameter is ignored and any added variables are also
 190         marked as non-trainable. `trainable` defaults to `True` unless
 191         `synchronization` is set to `ON_READ`.
 192       constraint: constraint instance (callable).
 193       use_resource: Whether to use `ResourceVariable`.
 194       synchronization: Indicates when a distributed a variable will be
 195         aggregated. Accepted values are constants defined in the class
 196         `tf.VariableSynchronization`. By default the synchronization is set to
 197         `AUTO` and the current `DistributionStrategy` chooses
 198         when to synchronize. If `synchronization` is set to `ON_READ`,
 199         `trainable` must not be set to `True`.
 200       aggregation: Indicates how a distributed variable will be aggregated.
 201         Accepted values are constants defined in the class
 202         `tf.VariableAggregation`.
 203       partitioner: (optional) partitioner instance (callable).  If
 204         provided, when the requested variable is created it will be split
 205         into multiple partitions according to `partitioner`.  In this case,
 206         an instance of `PartitionedVariable` is returned.  Available
 207         partitioners include `tf.fixed_size_partitioner` and
 208         `tf.variable_axis_size_partitioner`.  For more details, see the
 209         documentation of `tf.get_variable` and the  "Variable Partitioners
 210         and Sharding" section of the API guide.
 211 
 212     Returns:
 213       The created variable.  Usually either a `Variable` or `ResourceVariable`
 214       instance.  If `partitioner` is not `None`, a `PartitionedVariable`
 215       instance is returned.
 216 
 217     Raises:
 218       RuntimeError: If called with partioned variable regularization and
 219         eager execution is enabled.
 220       ValueError: When trainable has been set to True with synchronization
 221         set as `ON_READ`.
 222     """
 223     if synchronization == vs.VariableSynchronization.ON_READ:
 224       if trainable:
 225         raise ValueError(
 226             'Synchronization value can be set to '
 227             'VariableSynchronization.ON_READ only for non-trainable variables. '
 228             'You have specified trainable=True and '
 229             'synchronization=VariableSynchronization.ON_READ.')
 230       else:
 231         # Set trainable to be false when variable is to be synced on read.
 232         trainable = False
 233     elif trainable is None:
 234       trainable = True
 235 
 236     def _should_add_regularizer(variable, existing_variable_set):
 237       if isinstance(variable, tf_variables.PartitionedVariable):
 238         for var in variable:
 239           if var in existing_variable_set:
 240             return False
 241         return True
 242       else:
 243         return variable not in existing_variable_set
 244 
 245     init_graph = None
 246     if not context.executing_eagerly():
 247       default_graph = ops.get_default_graph()
 248       if default_graph.building_function:
 249         with ops.init_scope():
 250           # Retrieve the variables from the graph into which variables
 251           # will be lifted; if initialization ops will be lifted into
 252           # the eager context, then there is nothing to retrieve, since variable
 253           # collections are not supported when eager execution is enabled.
 254           if not context.executing_eagerly():
 255             init_graph = ops.get_default_graph()
 256             existing_variables = set(tf_variables.global_variables())
 257       else:
 258         # Initialization ops will not be lifted out of the default graph.
 259         init_graph = default_graph
 260         existing_variables = set(tf_variables.global_variables())
 261 
 262     if dtype is None:
 263       dtype = self.dtype or dtypes.float32
 264 
 265     self._set_scope(None)
 266     reuse = self.built or self._reuse
 267     prev_len_trainable = len(self._trainable_weights)
 268     with vs.variable_scope(
 269         self._scope, reuse=reuse, auxiliary_name_scope=False) as scope:
 270       self._current_scope = scope
 271       with ops.name_scope(self._name_scope()):
 272         use_resource = (use_resource or
 273                         self._use_resource_variables or
 274                         scope.use_resource)
 275         if initializer is None:
 276           initializer = scope.initializer
 277         variable = super(Layer, self).add_weight(
 278             name,
 279             shape,
 280             dtype=dtypes.as_dtype(dtype),
 281             initializer=initializer,
 282             trainable=trainable,
 283             constraint=constraint,
 284             partitioner=partitioner,
 285             use_resource=use_resource,
 286             synchronization=synchronization,
 287             aggregation=aggregation,
 288             getter=vs.get_variable)
 289 
 290         if regularizer:
 291           if context.executing_eagerly() or _should_add_regularizer(
 292               variable, existing_variables):
 293             self._handle_weight_regularization(name, variable, regularizer)
 294 
 295         if init_graph is not None:
 296           # Handle edge case where a custom getter has overridden `trainable`.
 297           # There is one known occurrence of this, in unit test
 298           # testBasicRNNCellNotTrainable in
 299           # contrib.rnn.python.kernel_tests.core_rnn_cell_test
 300           with init_graph.as_default():
 301             trainable_variables = tf_variables.trainable_variables()
 302           if (trainable and self.trainable and
 303               variable not in trainable_variables):
 304             # A custom getter / variable scope overrode the trainable flag.
 305             extra_trainable_vars = self._trainable_weights[prev_len_trainable:]
 306             self._trainable_weights = self._trainable_weights[
 307                 :prev_len_trainable]
 308             self._non_trainable_weights += extra_trainable_vars
 309     return variable

 311   def __call__(self, inputs, *args, **kwargs):
 312     """Wraps `call`, applying pre- and post-processing steps.
 313 
 314     Arguments:
 315       inputs: input tensor(s).
 316       *args: additional positional arguments to be passed to `self.call`.
 317       **kwargs: additional keyword arguments to be passed to `self.call`.
 318         **Note**: kwarg `scope` is reserved for use by the layer.
 319 
 320     Returns:
 321       Output tensor(s).
 322 
 323     Note:
 324       - If the layer's `call` method takes a `scope` keyword argument,
 325         this argument will be automatically set to the current variable scope.
 326       - If the layer's `call` method takes a `mask` argument (as some Keras
 327         layers do), its default value will be set to the mask generated
 328         for `inputs` by the previous layer (if `input` did come from
 329         a layer that generated a corresponding mask, i.e. if it came from
 330         a Keras layer with masking support.
 331 
 332     Raises:
 333       ValueError: if the layer's `call` method returns None (an invalid value).
 334     """
 335     self._set_scope(kwargs.pop('scope', None))
 336 
 337     if not context.executing_eagerly():
 338       try:
 339         # Set layer's "graph" at build time
 340         self._graph = ops._get_graph_from_inputs(nest.flatten(inputs),  # pylint: disable=protected-access
 341                                                  graph=self._graph)
 342       except ValueError as e:
 343         raise ValueError('Input graph and Layer graph are not the same: %s' % e)
 344 
 345     if self.built:
 346       try:
 347         # Some classes which inherit from Layer do not use its constructor, so
 348         # rather than initializing to None we check for an AttributeError.
 349         scope_context_manager = self._always_reuse_variable_scope
 350       except AttributeError:
 351         # From this point we will always set reuse=True, so create a "final"
 352         # variable scope with this setting. We avoid re-creating variable scopes
 353         # after this point as an optimization.
 354         self._always_reuse_variable_scope = vs.variable_scope(
 355             self._scope, reuse=True, auxiliary_name_scope=False)
 356         scope_context_manager = self._always_reuse_variable_scope
 357     else:
 358       scope_context_manager = vs.variable_scope(
 359           self._scope, reuse=self._reuse, auxiliary_name_scope=False)
 360 
 361     with scope_context_manager as scope:
 362       self._current_scope = scope
 363 
 364       try:
 365         call_has_scope_arg = self._call_has_scope_arg
 366       except AttributeError:
 367         self._call_fn_args = function_utils.fn_args(self.call)
 368         self._call_has_scope_arg = 'scope' in self._call_fn_args
 369         call_has_scope_arg = self._call_has_scope_arg
 370       if call_has_scope_arg:
 371         kwargs['scope'] = scope
 372 
 373       # Actually call layer
 374       outputs = super(Layer, self).__call__(inputs, *args, **kwargs)
 375 
 376     if not context.executing_eagerly():
 377       # Update global default collections.
 378       _add_elements_to_collection(self.updates, ops.GraphKeys.UPDATE_OPS)
 379     return outputs
 380 
 381   def __deepcopy__(self, memo):
 382     no_copy = set(['_graph'])
 383     shallow_copy = set(['_scope', '_always_reuse_variable_scope'])
 384     cls = self.__class__
 385     result = cls.__new__(cls)
 386     memo[id(self)] = result
 387     for k, v in self.__dict__.items():
 388       if k in no_copy:
 389         setattr(result, k, v)
 390       elif k in shallow_copy:
 391         setattr(result, k, copy.copy(v))
 392       elif base_layer.is_tensor_or_tensor_list(v):
 393         setattr(result, k, v)
 394       else:
 395         setattr(result, k, copy.deepcopy(v, memo))
 396     return result

 399 def _add_elements_to_collection(elements, collection_list):
 400   if context.executing_eagerly():
 401     raise RuntimeError('Using collections from Layers not supported in Eager '
 402                        'mode. Tried to add %s to %s' % (elements,
 403                                                         collection_list))
 404   elements = nest.flatten(elements)
 405   collection_list = nest.flatten(collection_list)
 406   for name in collection_list:
 407     collection = ops.get_collection_ref(name)
 408     collection_set = set(collection)
 409     for element in elements:
 410       if element not in collection_set:
 411         collection.append(element)
