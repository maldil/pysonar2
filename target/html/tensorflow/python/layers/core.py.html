  32 class Dense(keras_layers.Dense, base.Layer):
  33   """Densely-connected layer class.
  34 
  35   This layer implements the operation:
  36   `outputs = activation(inputs * kernel + bias)`
  37   Where `activation` is the activation function passed as the `activation`
  38   argument (if not `None`), `kernel` is a weights matrix created by the layer,
  39   and `bias` is a bias vector created by the layer
  40   (only if `use_bias` is `True`).
  41 
  42   Arguments:
  43     units: Integer or Long, dimensionality of the output space.
  44     activation: Activation function (callable). Set it to None to maintain a
  45       linear activation.
  46     use_bias: Boolean, whether the layer uses a bias.
  47     kernel_initializer: Initializer function for the weight matrix.
  48       If `None` (default), weights are initialized using the default
  49       initializer used by `tf.get_variable`.
  50     bias_initializer: Initializer function for the bias.
  51     kernel_regularizer: Regularizer function for the weight matrix.
  52     bias_regularizer: Regularizer function for the bias.
  53     activity_regularizer: Regularizer function for the output.
  54     kernel_constraint: An optional projection function to be applied to the
  55         kernel after being updated by an `Optimizer` (e.g. used to implement
  56         norm constraints or value constraints for layer weights). The function
  57         must take as input the unprojected variable and must return the
  58         projected variable (which must have the same shape). Constraints are
  59         not safe to use when doing asynchronous distributed training.
  60     bias_constraint: An optional projection function to be applied to the
  61         bias after being updated by an `Optimizer`.
  62     trainable: Boolean, if `True` also add variables to the graph collection
  63       `GraphKeys.TRAINABLE_VARIABLES` (see `tf.Variable`).
  64     name: String, the name of the layer. Layers with the same name will
  65       share weights, but to avoid mistakes we require reuse=True in such cases.
  66     reuse: Boolean, whether to reuse the weights of a previous layer
  67       by the same name.
  68 
  69   Properties:
  70     units: Python integer, dimensionality of the output space.
  71     activation: Activation function (callable).
  72     use_bias: Boolean, whether the layer uses a bias.
  73     kernel_initializer: Initializer instance (or name) for the kernel matrix.
  74     bias_initializer: Initializer instance (or name) for the bias.
  75     kernel_regularizer: Regularizer instance for the kernel matrix (callable)
  76     bias_regularizer: Regularizer instance for the bias (callable).
  77     activity_regularizer: Regularizer instance for the output (callable)
  78     kernel_constraint: Constraint function for the kernel matrix.
  79     bias_constraint: Constraint function for the bias.
  80     kernel: Weight matrix (TensorFlow variable or tensor).
  81     bias: Bias vector, if applicable (TensorFlow variable or tensor).
  82   """
  83 
  84   def __init__(self, units,
  85                activation=None,
  86                use_bias=True,
  87                kernel_initializer=None,
  88                bias_initializer=init_ops.zeros_initializer(),

  91                activity_regularizer=None,
  92                kernel_constraint=None,

 112 @tf_export('layers.dense')
 113 def dense(
 114     inputs, units,
 115     activation=None,
 116     use_bias=True,
 117     kernel_initializer=None,
 118     bias_initializer=init_ops.zeros_initializer(),
 119     kernel_regularizer=None,
 120     bias_regularizer=None,
 121     activity_regularizer=None,
 122     kernel_constraint=None,
 123     bias_constraint=None,
 124     trainable=True,
 125     name=None,
 126     reuse=None):
 127   """Functional interface for the densely-connected layer.
 128 
 129   This layer implements the operation:
 130   `outputs = activation(inputs * kernel + bias)`
 131   where `activation` is the activation function passed as the `activation`
 132   argument (if not `None`), `kernel` is a weights matrix created by the layer,
 133   and `bias` is a bias vector created by the layer
 134   (only if `use_bias` is `True`).
 135 
 136   Arguments:
 137     inputs: Tensor input.
 138     units: Integer or Long, dimensionality of the output space.
 139     activation: Activation function (callable). Set it to None to maintain a
 140       linear activation.
 141     use_bias: Boolean, whether the layer uses a bias.
 142     kernel_initializer: Initializer function for the weight matrix.
 143       If `None` (default), weights are initialized using the default
 144       initializer used by `tf.get_variable`.
 145     bias_initializer: Initializer function for the bias.
 146     kernel_regularizer: Regularizer function for the weight matrix.
 147     bias_regularizer: Regularizer function for the bias.
 148     activity_regularizer: Regularizer function for the output.
 149     kernel_constraint: An optional projection function to be applied to the
 150         kernel after being updated by an `Optimizer` (e.g. used to implement
 151         norm constraints or value constraints for layer weights). The function
 152         must take as input the unprojected variable and must return the
 153         projected variable (which must have the same shape). Constraints are
 154         not safe to use when doing asynchronous distributed training.
 155     bias_constraint: An optional projection function to be applied to the
 156         bias after being updated by an `Optimizer`.
 157     trainable: Boolean, if `True` also add variables to the graph collection
 158       `GraphKeys.TRAINABLE_VARIABLES` (see `tf.Variable`).
 159     name: String, the name of the layer.
 160     reuse: Boolean, whether to reuse the weights of a previous layer
 161       by the same name.
 162 
 163   Returns:
 164     Output tensor the same shape as `inputs` except the last dimension is of
 165     size `units`.
 166 
 167   Raises:
 168     ValueError: if eager execution is enabled.
 169   """
 170   layer = Dense(units,
 171                 activation=activation,
 172                 use_bias=use_bias,
 173                 kernel_initializer=kernel_initializer,
 174                 bias_initializer=bias_initializer,
 175                 kernel_regularizer=kernel_regularizer,
 176                 bias_regularizer=bias_regularizer,
 177                 activity_regularizer=activity_regularizer,
 178                 kernel_constraint=kernel_constraint,
 179                 bias_constraint=bias_constraint,
 180                 trainable=trainable,
 181                 name=name,
 182                 _scope=name,
 183                 _reuse=reuse)
 184   return layer.apply(inputs)

 187 @tf_export('layers.Dropout')
 188 class Dropout(keras_layers.Dropout, base.Layer):
 189   """Applies Dropout to the input.
 190 
 191   Dropout consists in randomly setting a fraction `rate` of input units to 0
 192   at each update during training time, which helps prevent overfitting.
 193   The units that are kept are scaled by `1 / (1 - rate)`, so that their
 194   sum is unchanged at training time and inference time.
 195 
 196   Arguments:
 197     rate: The dropout rate, between 0 and 1. E.g. `rate=0.1` would drop out
 198       10% of input units.
 199     noise_shape: 1D tensor of type `int32` representing the shape of the
 200       binary dropout mask that will be multiplied with the input.
 201       For instance, if your inputs have shape
 202       `(batch_size, timesteps, features)`, and you want the dropout mask
 203       to be the same for all timesteps, you can use
 204       `noise_shape=[batch_size, 1, features]`.
 205     seed: A Python integer. Used to create random seeds. See
 206       `tf.set_random_seed`.
 207       for behavior.
 208     name: The name of the layer (string).
 209   """
 210 
 211   def __init__(self, rate=0.5,
 212                noise_shape=None,
 213                seed=None,
 214                name=None,
 215                **kwargs):
 216     super(Dropout, self).__init__(rate=rate,
 217                                   noise_shape=noise_shape,
 218                                   seed=seed,
 219                                   name=name,
 220                                   **kwargs)
 221 
 222   def call(self, inputs, training=False):
 223     return super(Dropout, self).call(inputs, training=training)

 226 @tf_export('layers.dropout')
 227 def dropout(inputs,
 228             rate=0.5,
 229             noise_shape=None,
 230             seed=None,
 231             training=False,
 232             name=None):
 233   """Applies Dropout to the input.
 234 
 235   Dropout consists in randomly setting a fraction `rate` of input units to 0
 236   at each update during training time, which helps prevent overfitting.
 237   The units that are kept are scaled by `1 / (1 - rate)`, so that their
 238   sum is unchanged at training time and inference time.
 239 
 240   Arguments:
 241     inputs: Tensor input.
 242     rate: The dropout rate, between 0 and 1. E.g. "rate=0.1" would drop out
 243       10% of input units.
 244     noise_shape: 1D tensor of type `int32` representing the shape of the
 245       binary dropout mask that will be multiplied with the input.
 246       For instance, if your inputs have shape
 247       `(batch_size, timesteps, features)`, and you want the dropout mask
 248       to be the same for all timesteps, you can use
 249       `noise_shape=[batch_size, 1, features]`.
 250     seed: A Python integer. Used to create random seeds. See
 251       `tf.set_random_seed`
 252       for behavior.
 253     training: Either a Python boolean, or a TensorFlow boolean scalar tensor
 254       (e.g. a placeholder). Whether to return the output in training mode
 255       (apply dropout) or in inference mode (return the input untouched).
 256     name: The name of the layer (string).
 257 
 258   Returns:
 259     Output tensor.
 260 
 261   Raises:
 262     ValueError: if eager execution is enabled.
 263   """
 264   layer = Dropout(rate, noise_shape=noise_shape, seed=seed, name=name)
 265   return layer.apply(inputs, training=training)

 287 @tf_export('layers.flatten')
 288 def flatten(inputs, name=None):
 289   """Flattens an input tensor while preserving the batch axis (axis 0).
 290 
 291   Arguments:
 292     inputs: Tensor input.
 293     name: The name of the layer (string).
 294 
 295   Returns:
 296     Reshaped tensor.
 297 
 298   Examples:
 299 
 300   ```
 301     x = tf.placeholder(shape=(None, 4, 4), dtype='float32')
 302     y = flatten(x)
 303     # now `y` has shape `(None, 16)`
 304 
 305     x = tf.placeholder(shape=(None, 3, None), dtype='float32')
 306     y = flatten(x)
 307     # now `y` has shape `(None, None)`
 308   ```
 309   """
 310   layer = Flatten(name=name)
 311   return layer.apply(inputs)
