  50 def _validate_input_fn(input_fn):
  51   """Validates the `input_fn`."""
  52   if not callable(input_fn):
  53     raise TypeError('`input_fn` must be callable, given: {}'.format(input_fn))

  56 def _validate_hooks(hooks):
  57   """Validates the `hooks`."""
  58   hooks = tuple(hooks or [])
  59   for hook in hooks:
  60     if not isinstance(hook, session_run_hook.SessionRunHook):
  61       raise TypeError(
  62           'All hooks must be `SessionRunHook` instances, given: {}'.format(
  63               hook))
  64   return hooks

  67 def _validate_exporters(exporters):
  68   """Validates `exporters` and returns them as a tuple."""
  69   if not exporters:
  70     return ()
  71 
  72   if isinstance(exporters, exporter_lib.Exporter):
  73     exporters = [exporters]
  74 
  75   unique_names = []  # `Exporter`s should have unique names.
  76   try:
  77     for exporter in exporters:
  78       if not isinstance(exporter, exporter_lib.Exporter):
  79         # Error message will be printed out by the outer try/except.
  80         raise TypeError
  81 
  82       if not exporter.name:
  83         full_list_of_names = [e.name for e in exporters]
  84         raise ValueError('An Exporter cannot have a name that is `None` or'
  85                          ' empty. All exporter names:'
  86                          ' {}'.format(full_list_of_names))
  87 
  88       if not isinstance(exporter.name, six.string_types):
  89         raise ValueError('An Exporter must have a string name. Given: '
  90                          '{}'.format(type(exporter.name)))
  91 
  92       if exporter.name in unique_names:
  93         full_list_of_names = [e.name for e in exporters]
  94         raise ValueError(
  95             '`exporters` must have unique names. Such a name cannot be `None`.'
  96             ' All exporter names: {}'.format(full_list_of_names))
  97       unique_names.append(exporter.name)
  98   except TypeError:
  99     # Two possibilities:
 100     # - `exporters` is neither `Exporter` nor iterable.  Python has
 101     #   raised a `TypeError` when iterating over `exporters`.
 102     # - an `exporter` was None or not of type `Exporter`, so we raised a
 103     #   `TypeError`.
 104     raise TypeError('`exporters` must be an Exporter,'
 105                     ' an iterable of Exporter, or `None`,'
 106                     ' found %s.' % exporters)
 107 
 108   return tuple(exporters)

 111 def _is_google_env():
 112   """Detects whether current environment is google."""
 113   tf_config = json.loads(os.environ.get(_TF_CONFIG_ENV) or '{}')
 114   if not tf_config:

 116   return tf_config.get(_ENVIRONMENT_KEY) == _ENVIRONMENT_GOOGLE_VALUE

 122   """Configuration for the "train" part for the `train_and_evaluate` call.

 128   def __new__(cls, input_fn, max_steps=None, hooks=None):
 129     """Creates a validated `TrainSpec` instance.
 130 
 131     Args:
 132       input_fn: A function that provides input data for training as minibatches.
 133         See [Premade Estimators](https://tensorflow.org/guide/premade_estimators#create_input_functions)
 134         for more information. The function should construct and return one of
 135         the following:
 136           * A 'tf.data.Dataset' object: Outputs of `Dataset` object must be a
 137             tuple (features, labels) with same constraints as below.
 138           * A tuple (features, labels): Where features is a `Tensor` or a
 139             dictionary of string feature name to `Tensor` and labels is a
 140             `Tensor` or a dictionary of string label name to `Tensor`.
 141 
 142       max_steps: Int. Positive number of total steps for which to train model.
 143         If `None`, train forever. The training `input_fn` is not expected to
 144         generate `OutOfRangeError` or `StopIteration` exceptions. See the
 145         `train_and_evaluate` stop condition section for details.
 146       hooks: Iterable of `tf.train.SessionRunHook` objects to run

 154       TypeError: If any of the arguments is not of the expected type.
 155     """
 156     # Validate input_fn.
 157     _validate_input_fn(input_fn)
 158 
 159     # Validate max_steps.
 160     if max_steps is not None and max_steps <= 0:
 161       raise ValueError(
 162           'Must specify max_steps > 0, given: {}'.format(max_steps))
 163 
 164     # Validate hooks.
 165     hooks = _validate_hooks(hooks)
 166 
 167     return super(TrainSpec, cls).__new__(
 168         cls, input_fn=input_fn, max_steps=max_steps, hooks=hooks)

 171 @estimator_export('estimator.EvalSpec')
 172 class EvalSpec(
 173     collections.namedtuple('EvalSpec', [
 174         'input_fn', 'steps', 'name', 'hooks', 'exporters', 'start_delay_secs',
 175         'throttle_secs'
 176     ])):
 177   """Configuration for the "eval" part for the `train_and_evaluate` call.
 178 
 179   `EvalSpec` combines details of evaluation of the trained model as well as its
 180   export. Evaluation consists of computing metrics to judge the performance of
 181   the trained model.  Export writes out the trained model on to external
 182   storage.
 183   """
 184 
 185   def __new__(cls,
 186               input_fn,
 187               steps=100,
 188               name=None,
 189               hooks=None,
 190               exporters=None,
 191               start_delay_secs=120,
 192               throttle_secs=600):
 193     """Creates a validated `EvalSpec` instance.
 194 
 195     Args:
 196       input_fn: A function that constructs the input data for evaluation.
 197         See [Premade Estimators](https://tensorflow.org/api_guides/premade_estimators#create_input_functions)
 198         for more information. The function should construct and return one of
 199         the following:
 200           * A 'tf.data.Dataset' object: Outputs of `Dataset` object must be a
 201             tuple (features, labels) with same constraints as below.
 202           * A tuple (features, labels): Where features is a `Tensor` or a
 203             dictionary of string feature name to `Tensor` and labels is a
 204             `Tensor` or a dictionary of string label name to `Tensor`.
 205 
 206       steps: Int. Positive number of steps for which to evaluate model. If
 207         `None`, evaluates until `input_fn` raises an end-of-input exception.
 208         See `Estimator.evaluate` for details.
 209       name: String. Name of the evaluation if user needs to run multiple
 210         evaluations on different data sets. Metrics for different evaluations
 211         are saved in separate folders, and appear separately in tensorboard.
 212       hooks: Iterable of `tf.train.SessionRunHook` objects to run
 213         during evaluation.
 214       exporters: Iterable of `Exporter`s, or a single one, or `None`.
 215         `exporters` will be invoked after each evaluation.
 216       start_delay_secs: Int. Start evaluating after waiting for this many
 217         seconds.
 218       throttle_secs: Int. Do not re-evaluate unless the last evaluation was
 219         started at least this many seconds ago. Of course, evaluation does not
 220         occur if no new checkpoints are available, hence, this is the minimum.
 221 
 222     Returns:
 223       A validated `EvalSpec` object.
 224 
 225     Raises:
 226       ValueError: If any of the input arguments is invalid.
 227       TypeError: If any of the arguments is not of the expected type.
 228     """
 229     # Validate input_fn.
 230     _validate_input_fn(input_fn)
 231 
 232     # Validate steps.
 233     if steps is not None and steps <= 0:
 234       raise ValueError('Must specify steps > 0, given: {}'.format(steps))
 235 
 236     # Validate name.
 237     if name is not None and not isinstance(name, six.string_types):
 238       raise TypeError('`name` must be string, given: {}'.format(name))
 239 
 240     # Validate hooks.
 241     hooks = _validate_hooks(hooks)
 242 
 243     # Validate exporters.
 244     exporters = _validate_exporters(exporters)
 245 
 246     # Validate start_delay_secs.
 247     if start_delay_secs < 0:
 248       raise ValueError('Must specify start_delay_secs >= 0, given: {}'.format(
 249           start_delay_secs))
 250 
 251     # Validate throttle_secs.
 252     if throttle_secs < 0:
 253       raise ValueError(
 254           'Must specify throttle_secs >= 0, given: {}'.format(throttle_secs))
 255 
 256     return super(EvalSpec, cls).__new__(
 257         cls,
 258         input_fn=input_fn,
 259         steps=steps,
 260         name=name,
 261         hooks=hooks,
 262         exporters=exporters,
 263         start_delay_secs=start_delay_secs,
 264         throttle_secs=throttle_secs)

 267 @estimator_export('estimator.train_and_evaluate')
 268 def train_and_evaluate(estimator, train_spec, eval_spec):
 269   """Train and evaluate the `estimator`.
 270 
 271   This utility function trains, evaluates, and (optionally) exports the model by
 272   using the given `estimator`. All training related specification is held in
 273   `train_spec`, including training `input_fn` and training max steps, etc. All
 274   evaluation and export related specification is held in `eval_spec`, including
 275   evaluation `input_fn`, steps, etc.
 276 
 277   This utility function provides consistent behavior for both local
 278   (non-distributed) and distributed configurations. The default distribution
 279   configuration is parameter server-based between-graph replication. For other
 280   types of distribution configurations such as all-reduce training, please use
 281   [DistributionStrategies](https://github.com/tensorflow/tensorflow/tree/master/tensorflow/contrib/distribute).  # pylint: disable=line-too-long
 282 
 283   Overfitting: In order to avoid overfitting, it is recommended to set up the
 284   training `input_fn` to shuffle the training data properly.
 285 
 286   Stop condition: In order to support both distributed and non-distributed
 287   configuration reliably, the only supported stop condition for model
 288   training is `train_spec.max_steps`. If `train_spec.max_steps` is `None`, the
 289   model is trained forever. *Use with care* if model stop condition is
 290   different. For example, assume that the model is expected to be trained with
 291   one epoch of training data, and the training `input_fn` is configured to throw
 292   `OutOfRangeError` after going through one epoch, which stops the
 293   `Estimator.train`. For a three-training-worker distributed configuration, each
 294   training worker is likely to go through the whole epoch independently. So, the
 295   model will be trained with three epochs of training data instead of one epoch.
 296 
 297   Example of local (non-distributed) training:
 298 
 299   ```python
 300   # Set up feature columns.
 301   categorial_feature_a = categorial_column_with_hash_bucket(...)
 302   categorial_feature_a_emb = embedding_column(
 303       categorical_column=categorial_feature_a, ...)
 304   ...  # other feature columns
 305 
 306   estimator = DNNClassifier(
 307       feature_columns=[categorial_feature_a_emb, ...],
 308       hidden_units=[1024, 512, 256])
 309 
 310   # Or set up the model directory
 311   #   estimator = DNNClassifier(
 312   #       config=tf.estimator.RunConfig(
 313   #           model_dir='/my_model', save_summary_steps=100),
 314   #       feature_columns=[categorial_feature_a_emb, ...],
 315   #       hidden_units=[1024, 512, 256])
 316 
 317   # Input pipeline for train and evaluate.
 318   def train_input_fn(): # returns x, y
 319     # please shuffle the data.
 320     pass
 321   def eval_input_fn(): # returns x, y
 322     pass
 323 
 324   train_spec = tf.estimator.TrainSpec(input_fn=train_input_fn, max_steps=1000)
 325   eval_spec = tf.estimator.EvalSpec(input_fn=eval_input_fn)
 326 
 327   tf.estimator.train_and_evaluate(estimator, train_spec, eval_spec)
 328   ```
 329   Note that in current implementation `estimator.evaluate` will be called
 330   multiple times. This means that evaluation graph (including eval_input_fn)
 331   will be re-created for each `evaluate` call. `estimator.train` will be called
 332   only once.
 333 
 334   Example of distributed training:
 335 
 336   Regarding the example of distributed training, the code above can be used
 337   without a change (Please do make sure that the `RunConfig.model_dir` for all
 338   workers is set to the same directory, i.e., a shared file system all workers
 339   can read and write). The only extra work to do is setting the environment
 340   variable `TF_CONFIG` properly for each worker correspondingly.
 341 
 342   Also see
 343   [Distributed TensorFlow](https://www.tensorflow.org/deploy/distributed).
 344 
 345   Setting environment variable depends on the platform. For example, on Linux,
 346   it can be done as follows (`$` is the shell prompt):
 347 
 348   ```
 349   $ TF_CONFIG='<replace_with_real_content>' python train_model.py
 350   ```
 351 
 352   For the content in `TF_CONFIG`, assume that the training cluster spec looks
 353   like:
 354 
 355   ```
 356   cluster = {"chief": ["host0:2222"],
 357              "worker": ["host1:2222", "host2:2222", "host3:2222"],
 358              "ps": ["host4:2222", "host5:2222"]}
 359   ```
 360 
 361   Example of `TF_CONFIG` for chief training worker (must have one and only one):
 362 
 363   ```
 364   # This should be a JSON string, which is set as environment variable. Usually
 365   # the cluster manager handles that.
 366   TF_CONFIG='{
 367       "cluster": {
 368           "chief": ["host0:2222"],
 369           "worker": ["host1:2222", "host2:2222", "host3:2222"],
 370           "ps": ["host4:2222", "host5:2222"]
 371       },
 372       "task": {"type": "chief", "index": 0}
 373   }'
 374   ```
 375   Note that the chief worker also does the model training job, similar to other
 376   non-chief training workers (see next paragraph). In addition to the model
 377   training, it manages some extra work, e.g., checkpoint saving and restoring,
 378   writing summaries, etc.
 379 
 380   Example of `TF_CONFIG` for non-chief training worker (optional, could be
 381   multiple):
 382 
 383   ```
 384   # This should be a JSON string, which is set as environment variable. Usually
 385   # the cluster manager handles that.
 386   TF_CONFIG='{
 387       "cluster": {
 388           "chief": ["host0:2222"],
 389           "worker": ["host1:2222", "host2:2222", "host3:2222"],
 390           "ps": ["host4:2222", "host5:2222"]
 391       },
 392       "task": {"type": "worker", "index": 0}
 393   }'
 394   ```
 395   where the `task.index` should be set as 0, 1, 2, in this example, respectively
 396   for non-chief training workers.
 397 
 398   Example of `TF_CONFIG` for parameter server, aka ps (could be multiple):
 399 
 400   ```
 401   # This should be a JSON string, which is set as environment variable. Usually
 402   # the cluster manager handles that.
 403   TF_CONFIG='{
 404       "cluster": {
 405           "chief": ["host0:2222"],
 406           "worker": ["host1:2222", "host2:2222", "host3:2222"],
 407           "ps": ["host4:2222", "host5:2222"]
 408       },
 409       "task": {"type": "ps", "index": 0}
 410   }'
 411   ```
 412   where the `task.index` should be set as 0 and 1, in this example, respectively
 413   for parameter servers.
 414 
 415   Example of `TF_CONFIG` for evaluator task. Evaluator is a special task that is
 416   not part of the training cluster. There could be only one. It is used for
 417   model evaluation.
 418 
 419   ```
 420   # This should be a JSON string, which is set as environment variable. Usually
 421   # the cluster manager handles that.
 422   TF_CONFIG='{
 423       "cluster": {
 424           "chief": ["host0:2222"],
 425           "worker": ["host1:2222", "host2:2222", "host3:2222"],
 426           "ps": ["host4:2222", "host5:2222"]
 427       },
 428       "task": {"type": "evaluator", "index": 0}
 429   }'
 430   ```
 431 
 432   When `distribute` or `experimental_distribute.train_distribute` and
 433   `experimental_distribute.remote_cluster` is set, this method will start a
 434   client running on the current host which connects to the `remote_cluster` for
 435   training and evaluation.
 436 
 437   Args:
 438     estimator: An `Estimator` instance to train and evaluate.
 439     train_spec: A `TrainSpec` instance to specify the training specification.
 440     eval_spec: A `EvalSpec` instance to specify the evaluation and export
 441       specification.
 442 
 443   Returns:
 444     A tuple of the result of the `evaluate` call to the `Estimator` and the
 445     export results using the specified `ExportStrategy`.
 446     Currently, the return value is undefined for distributed training mode.
 447 
 448   Raises:
 449     ValueError: if environment variable `TF_CONFIG` is incorrectly set.
 450   """
 451   _assert_eval_spec(eval_spec)  # fail fast if eval_spec is invalid.
 452 
 453   executor = _TrainingExecutor(
 454       estimator=estimator, train_spec=train_spec, eval_spec=eval_spec)
 455   config = estimator.config
 456 
 457   # If `distribute_coordinator_mode` is set and running in distributed
 458   # environment, we run `train_and_evaluate` via distribute coordinator.
 459   if distribute_coordinator_training.should_run_distribute_coordinator(config):
 460     logging.info('Running `train_and_evaluate` with Distribute Coordinator.')
 461     distribute_coordinator_training.train_and_evaluate(
 462         estimator, train_spec, eval_spec, _TrainingExecutor)
 463     return
 464 
 465   if (config.task_type == run_config_lib.TaskType.EVALUATOR and
 466       config.task_id > 0):
 467     raise ValueError(
 468         'For distributed training, there can only be one `evaluator` task '
 469         '(with task id 0).  Given task id {}'.format(config.task_id))
 470 
 471   return executor.run()

 474 class _StopAtSecsHook(session_run_hook.SessionRunHook):
 475   """Stops given secs after begin is called."""
 476 
 477   def __init__(self, stop_after_secs):
 478     self._stop_after_secs = stop_after_secs
 479     self._start_time = None
 480 
 481   def begin(self):
 482     self._start_time = time.time()
 483 
 484   def after_run(self, run_context, run_values):
 485     del run_values
 486     if time.time() - self._start_time >= self._stop_after_secs:
 487       run_context.request_stop()

 490 class _NewCheckpointListenerForEvaluate(
 491     basic_session_run_hooks.CheckpointSaverListener):
 492   """A saver listener to run evaluate with every checkpoint."""
 493 
 494   def __init__(self, evaluator, eval_throttle_secs, continuous_eval_listener):
 495     self._evaluator = evaluator
 496     self._eval_throttle_secs = eval_throttle_secs
 497     self._continuous_eval_listener = continuous_eval_listener
 498     self.eval_result, self.export_results = None, None
 499 
 500   def begin(self):
 501     self._timer = basic_session_run_hooks.SecondOrStepTimer(
 502         every_secs=self._eval_throttle_secs)
 503     self._is_first_run = True
 504 
 505   def after_save(self, session, global_step_value):
 506     del session  # unused; required by signature.
 507     # skip first run model is not trained yet.
 508     if self._is_first_run:
 509       self._is_first_run = False
 510       return
 511 
 512     if not self._continuous_eval_listener.before_eval():
 513       logging.info('Exiting training and evaluation loop, as requested by '
 514                    '_ContinuousEvalListener.before_eval.')
 515       return True
 516     if self._timer.should_trigger_for_step(global_step_value):
 517       self._evaluate(global_step_value)  # updates self.eval_result
 518       if not self._continuous_eval_listener.after_eval(self.eval_result):
 519         logging.info('Exiting evaluation, as requested by '
 520                      '_ContinuousEvalListener.after_eval.')
 521         return True
 522     else:
 523       # TODO(ispir): add remaining time in the log.
 524       logging.info('Skip the current checkpoint eval due to throttle secs '
 525                    '({} secs).'.format(self._eval_throttle_secs))
 526 
 527   def end(self, session, global_step_value):
 528     # Evaluate if the last step has not been evaluated, yet.
 529     if global_step_value != self._timer.last_triggered_step():
 530       if self._continuous_eval_listener.before_eval():
 531         self._evaluate(global_step_value)
 532         self._continuous_eval_listener.after_eval(self.eval_result)
 533 
 534   def _evaluate(self, global_step_value):
 535     self._timer.update_last_triggered_step(global_step_value)
 536     self.eval_result, self.export_results = (
 537         self._evaluator.evaluate_and_export())
 538     if self.eval_result.status != _EvalStatus.EVALUATED:
 539       #  This is unexpected; should never happen.
 540       #  Training should always end with a new checkpoint.
 541       raise RuntimeError('There was no new checkpoint after the training. '
 542                          'Eval status: {}'.format(self.eval_result.status))
 543

 545 class _TrainingExecutor(object):
 546   """The executor to run `Estimator` training and evaluation.
 547 
 548   This implementation supports both distributed and non-distributed (aka local)
 549   training and evaluation based on the setting in `tf.estimator.RunConfig`.
 550   """
 551 
 552   def __init__(self,
 553                estimator,
 554                train_spec,
 555                eval_spec,
 556                train_hooks=None,
 557                continuous_eval_listener=None):
 558     if not isinstance(estimator, estimator_lib.Estimator):
 559       raise TypeError(
 560           '`estimator` must have type `tf.estimator.Estimator`. '
 561           'Got: {}'.format(type(estimator)))
 562     self._estimator = estimator
 563 
 564     if not isinstance(train_spec, TrainSpec):
 565       raise TypeError(
 566           '`train_spec` must have type `tf.estimator.TrainSpec`. '
 567           'Got: {}'.format(type(train_spec)))
 568     self._train_spec = train_spec
 569 
 570     if eval_spec and not isinstance(eval_spec, EvalSpec):
 571       raise TypeError('`eval_spec` must be either `None` or have type '
 572                       '`tf.estimator.EvalSpec`. Got: {}'.format(
 573                           type(eval_spec)))
 574     self._eval_spec = eval_spec
 575 
 576     self._train_hooks = _validate_hooks(train_hooks)
 577 
 578     if (continuous_eval_listener and
 579         not isinstance(continuous_eval_listener, _ContinuousEvalListener)):
 580       raise TypeError('`continuous_eval_listener` must have type '
 581                       '`_ContinuousEvalListener`.')
 582     self._continuous_eval_listener = (
 583         continuous_eval_listener or _ContinuousEvalListener())

 589   def run(self):
 590     """Executes the run_foo for task type `foo`.
 591 
 592     `_TrainingExecutor` predefines the procedure for task type 'chief',
 593     'worker', 'ps', and 'evaluator'. For task type `foo`, the corresponding
 594     procedure is `run_foo'. This `run` method invoke the procedure base on the
 595     `RunConfig.task_type`.
 596 
 597     Returns:
 598       A tuple of the result of the `evaluate` call to the `Estimator` and the
 599       export results using the specified `ExportStrategy`.
 600       Currently undefined for distributed training mode.
 601 
 602     Raises:
 603       ValueError: if the estimator.config is mis-configured.
 604     """
 605     config = self._estimator.config
 606 
 607     if (not config.cluster_spec and
 608         config.task_type != run_config_lib.TaskType.EVALUATOR):
 609       logging.info('Running training and evaluation locally (non-distributed).')
 610       return self.run_local()
 611 
 612     # Distributed case.
 613     if not config.task_type:
 614       # TODO(xiejw): Improve the error message about how to set the TF_CONFIG
 615       # correctly.
 616       raise ValueError(
 617           '`estimator.config` must have task_type set. This usually means '
 618           'TF_CONFIG environment is not set correctly.')
 619 
 620     if config.task_type == 'local':
 621       raise ValueError(
 622           '`task.type` in TF_CONFIG cannot be `local`. Leaving `cluster` and '
 623           '`task` properties in TF_CONFIG absent triggers train and evaluate '
 624           '`Estimator` locally (non-distributed).')
 625 
 626     # For task type foo, call executor.run_foo.
 627     available_tasks = [
 628         x for x in dir(self)
 629         if x.startswith('run_') and x != 'run_local' and
 630         callable(getattr(self, x))
 631     ]
 632     task_to_run = 'run_' + config.task_type
 633     if task_to_run not in available_tasks:
 634       raise ValueError(
 635           'Task type {} is not supported. Supported task types are {}'.format(
 636               config.task_type, [x[len('run_'):] for x in available_tasks]))
 637     getattr(self, task_to_run)()
 638

 649   def run_master(self):
 650     """Runs task master."""
 651     _assert_eval_spec(self._eval_spec)
 652 
 653     # Final export signal: For any eval result with global_step >= train
 654     # max_steps, the evaluator will send the final export signal. There is a
 655     # small chance that the Estimator.train stopping logic sees a different
 656     # global_step value (due to global step race condition and the fact the
 657     # saver sees a larger value for checkpoint saving), which does not end
 658     # the training. When the training ends, a new checkpoint is generated, which
 659     # triggers the listener again. So, it could be the case the final export is
 660     # triggered twice.
 661     #
 662     # But here, throttle_secs will skip the next intermediate checkpoint and,
 663     # so, the double final export chance is very small.
 664     evaluator = _TrainingExecutor._Evaluator(self._estimator, self._eval_spec,
 665                                              self._train_spec.max_steps)
 666 
 667     # When the underlying `Estimator` object saves a new checkpoint, we would
 668     # like this callback to be called so that evaluation and export can trigger.
 669     saving_listeners = [
 670         _NewCheckpointListenerForEvaluate(evaluator,
 671                                           self._eval_spec.throttle_secs,
 672                                           _ContinuousEvalListener())
 673     ]
 674     self._start_distributed_training(saving_listeners=saving_listeners)
 675

 687   def run_local(self):
 688     """Runs training and evaluation locally (non-distributed)."""
 689     _assert_eval_spec(self._eval_spec)
 690 
 691     train_hooks = list(self._train_spec.hooks) + list(self._train_hooks)
 692     logging.info('Start train and evaluate loop. The evaluate will happen '
 693                  'after every checkpoint. Checkpoint frequency is determined '
 694                  'based on RunConfig arguments: save_checkpoints_steps {} or '
 695                  'save_checkpoints_secs {}.'.format(
 696                      self._estimator.config.save_checkpoints_steps,
 697                      self._estimator.config.save_checkpoints_secs))
 698 
 699     evaluator = _TrainingExecutor._Evaluator(self._estimator, self._eval_spec,
 700                                              self._train_spec.max_steps)
 701 
 702     listener_for_eval = _NewCheckpointListenerForEvaluate(
 703         evaluator, self._eval_spec.throttle_secs,
 704         self._continuous_eval_listener)
 705     saving_listeners = [listener_for_eval]
 706 
 707     self._estimator.train(
 708         input_fn=self._train_spec.input_fn,
 709         max_steps=self._train_spec.max_steps,
 710         hooks=train_hooks,
 711         saving_listeners=saving_listeners)
 712 
 713     eval_result = listener_for_eval.eval_result or _EvalResult(
 714         status=_EvalStatus.MISSING_CHECKPOINT)
 715     return eval_result.metrics, listener_for_eval.export_results

 717   def _start_std_server(self, config):
 718     """Creates, starts, and returns a server_lib.Server."""
 719     if (not config.cluster_spec or not config.task_type or
 720         config.task_id is None):
 721       raise RuntimeError('Could not start server; be sure to specify '
 722                          'cluster_spec, task_type, and task in '
 723                          'RunConfig or set the TF_CONFIG environment variable.')
 724 
 725     if not config.master:
 726       jobs = config.cluster_spec.jobs
 727       if (len(jobs) == 1 and
 728           len(config.cluster_spec.job_tasks(jobs[0])) == 1 and
 729           config.task_type in _TRAINER_JOBS):
 730         # For distributed training, config.master is empty if and only if it has
 731         # a single node in the cluster spec. In this case, we should not start
 732         # the server.
 733         logging.info('Skip starting Tensorflow server as there is only one '
 734                      'node in the cluster.')
 735         return
 736       else:
 737         raise RuntimeError(
 738             'Could not start server; be sure to specify master in '
 739             'RunConfig or set the TF_CONFIG environment variable.')
 740 
 741     logging.info('Start Tensorflow server.')
 742 
 743     if config.session_config is None:
 744       session_config = config_pb2.ConfigProto(log_device_placement=False)
 745     else:
 746       session_config = config_pb2.ConfigProto(
 747           log_device_placement=False,
 748           gpu_options=config.session_config.gpu_options)
 749 
 750     server = server_lib.Server(
 751         config.cluster_spec,
 752         job_name=config.task_type,
 753         task_index=config.task_id,
 754         config=session_config,
 755         start=False,
 756         protocol=config.protocol)
 757     server.start()
 758     return server

 760   def _start_distributed_training(self, saving_listeners=None):
 761     """Calls `Estimator` train in a distributed setting."""
 762     config = self._estimator.config
 763 
 764     # Start in-process TensorFlow server if needed. It's important to start the
 765     # server before we (optionally) sleep. Otherwise, the servers will wait to
 766     # connect to each other before starting to train.
 767     if not _is_google_env():
 768       self._start_std_server(config)
 769 
 770     # Delay worker to start. For asynchronous training, this usually helps model
 771     # to converge faster.  Chief starts the training immediately, so, worker
 772     # with task id x (0-based) should wait (x+1) * _DELAY_SECS_PER_WORKER.
 773     start_delay_secs = 0
 774     if config.task_type == run_config_lib.TaskType.WORKER:
 775       # TODO(xiejw): Replace the hard code logic (task_id + 1) with unique id in
 776       # training cluster.
 777       start_delay_secs = min(_MAX_DELAY_SECS,
 778                              (config.task_id + 1) * _DELAY_SECS_PER_WORKER)
 779     if start_delay_secs > 0:
 780       logging.info('Waiting %d secs before starting training.',
 781                    start_delay_secs)
 782       time.sleep(start_delay_secs)
 783 
 784     self._estimator.train(
 785         input_fn=self._train_spec.input_fn,
 786         max_steps=self._train_spec.max_steps,
 787         hooks=list(self._train_spec.hooks) + list(self._train_hooks),
 788         saving_listeners=saving_listeners)
 789 
 790   def _start_continuous_evaluation(self):
 791     """Repeatedly calls `Estimator` evaluate and export until training ends."""
 792 
 793     _assert_eval_spec(self._eval_spec)
 794 
 795     start_delay_secs = self._eval_spec.start_delay_secs
 796     if start_delay_secs:
 797       logging.info('Waiting %f secs before starting eval.', start_delay_secs)
 798       time.sleep(start_delay_secs)
 799 
 800     latest_eval_result = None
 801     evaluator = _TrainingExecutor._Evaluator(self._estimator, self._eval_spec,
 802                                              self._train_spec.max_steps)
 803 
 804     should_early_stop = False
 805     while not should_early_stop:
 806       if (latest_eval_result and
 807           latest_eval_result.status == _EvalStatus.EVALUATED):
 808         global_step = latest_eval_result.metrics.get(ops.GraphKeys.GLOBAL_STEP)
 809         if (global_step and self._train_spec.max_steps and
 810             global_step >= self._train_spec.max_steps):
 811           logging.info(
 812               'Exiting evaluation, global_step=%s >= train max_steps=%s',
 813               global_step, self._train_spec.max_steps)
 814           return
 815 
 816       latest_eval_result, should_early_stop = self._execute_evaluator_once(
 817           evaluator, self._continuous_eval_listener,
 818           self._eval_spec.throttle_secs)
 819 
 820   def _execute_evaluator_once(self, evaluator, continuous_eval_listener,
 821                               throttle_secs):
 822     """Executes the `evaluator`."""
 823 
 824     _assert_eval_spec(self._eval_spec)
 825 
 826     start = time.time()
 827 
 828     eval_result = None
 829     should_early_stop = False
 830 
 831     if not continuous_eval_listener.before_eval():
 832       logging.info('Exiting evaluation, as requested by '
 833                    '_ContinuousEvalListener.before_eval.')
 834       should_early_stop = True
 835       return (eval_result, should_early_stop)
 836 
 837     # Final export signal: For any eval result with global_step >= train
 838     # max_steps, the evaluator will send the final export signal. The next
 839     # iteration of while loop will end the continuous eval as the stopping
 840     # condition is satisfied (both checks use the same global_step value,
 841     # i.e., no race condition)
 842     eval_result, _ = evaluator.evaluate_and_export()
 843 
 844     if not self._continuous_eval_listener.after_eval(eval_result):
 845       logging.info('Exiting evaluation, as requested by '
 846                    '_ContinuousEvalListener.after_eval.')
 847       should_early_stop = True
 848       return (eval_result, should_early_stop)
 849 
 850     # Throttle if necessary.
 851     elapsed_time = time.time() - start
 852     difference = throttle_secs - elapsed_time
 853     if difference > 0:
 854       logging.info('Waiting %f secs before starting next eval run.', difference)
 855       time.sleep(difference)
 856     elif (throttle_secs == 0 and
 857           eval_result.status != _EvalStatus.EVALUATED):
 858       # Prints a user-actionable warning to avoid unnecessary load on evaluator.
 859       logging.warning(
 860           'EvalSpec.throttle_secs is set as 0. This might overload the job '
 861           'before finding (next) new checkpoint. Please consider to increase '
 862           'it.')
 863 
 864     return (eval_result, should_early_stop)

 866   class _Evaluator(object):
 867     """A helper class to call `Estimator.evaluate` and export model."""
 868 
 869     def __init__(self, estimator, eval_spec, max_training_steps):
 870       self._estimator = estimator
 871 
 872       _assert_eval_spec(eval_spec)
 873       self._eval_spec = eval_spec
 874 
 875       self._is_final_export_triggered = False
 876       self._previous_ckpt_path = None
 877       self._last_warning_time = 0
 878       self._max_training_steps = max_training_steps
 879 
 880     @property
 881     def is_final_export_triggered(self):
 882       return self._is_final_export_triggered
 883 
 884     def evaluate_and_export(self):
 885       """Evaluate and (maybe) export the current model.
 886 
 887       Returns:
 888         A tuple of `EvalResult` instance and the export results.
 889 
 890       Raises:
 891         RuntimeError: for any unexpected internal error.
 892         TypeError: if evaluation result has wrong type.
 893       """
 894       latest_ckpt_path = self._estimator.latest_checkpoint()
 895       if not latest_ckpt_path:
 896         self._log_err_msg('Estimator is not trained yet. Will start an '
 897                           'evaluation when a checkpoint is ready.')
 898         return _EvalResult(status=_EvalStatus.MISSING_CHECKPOINT), []
 899 
 900       if latest_ckpt_path == self._previous_ckpt_path:
 901         self._log_err_msg(
 902             'No new checkpoint ready for evaluation. Skip the current '
 903             'evaluation pass as evaluation results are expected to be same '
 904             'for the same checkpoint.')
 905         return _EvalResult(status=_EvalStatus.NO_NEW_CHECKPOINT), []
 906 
 907       metrics = self._estimator.evaluate(
 908           input_fn=self._eval_spec.input_fn,
 909           steps=self._eval_spec.steps,
 910           name=self._eval_spec.name,
 911           checkpoint_path=latest_ckpt_path,
 912           hooks=self._eval_spec.hooks)
 913 
 914       # _EvalResult validates the metrics.
 915       eval_result = _EvalResult(
 916           status=_EvalStatus.EVALUATED,
 917           metrics=metrics,
 918           checkpoint_path=latest_ckpt_path)
 919 
 920       is_the_final_export = (
 921           eval_result.metrics[ops.GraphKeys.GLOBAL_STEP] >=
 922           self._max_training_steps if self._max_training_steps else False)
 923       export_results = self._export_eval_result(eval_result,
 924                                                 is_the_final_export)
 925 
 926       if is_the_final_export:
 927         logging.debug('Calling exporter with the `is_the_final_export=True`.')
 928         self._is_final_export_triggered = True
 929 
 930       self._last_warning_time = 0
 931       self._previous_ckpt_path = latest_ckpt_path
 932       return eval_result, export_results
 933 
 934     def _log_err_msg(self, message):
 935       """Prints warning `message` every 10 mins."""
 936       current_time = time.time()
 937       if current_time - self._last_warning_time > 600:
 938         logging.warning(message)
 939         self._last_warning_time = current_time
 940 
 941     def _export_eval_result(self, eval_result, is_the_final_export):
 942       """Export `eval_result` according to exporters in `EvalSpec`."""
 943       export_dir_base = os.path.join(
 944           compat.as_str_any(self._estimator.model_dir),
 945           compat.as_str_any('export'))
 946 
 947       export_results = []
 948       for exporter in self._eval_spec.exporters:
 949         export_results.append(
 950             exporter.export(
 951                 estimator=self._estimator,
 952                 export_path=os.path.join(
 953                     compat.as_str_any(export_dir_base),
 954                     compat.as_str_any(exporter.name)),
 955                 checkpoint_path=eval_result.checkpoint_path,
 956                 eval_result=eval_result.metrics,
 957                 is_the_final_export=is_the_final_export))
 958       return export_results

 961 class _EvalStatus(object):
 962   """The status of an evaluation event.
 963 
 964   For local training and evaluation, the status can only be `EVALUATED` as
 965   `Estimator.train` always generates a new checkpoint.
 966 
 967   For distributed training and evaluation, a separated evaluator keeps looking
 968   for new checkpoint. So, multiple situations might occur:
 969 
 970   - EVALUATED: A new checkpoint is found since last evaluation.
 971       `Estimator.evaluate` will be invoked.
 972   - MISSING_CHECKPOINT: No checkpoint can be found. Typically, this means
 973       the trainer has not yet produced any checkpoint.
 974   - NO_NEW_CHECKPOINT: No new checkpoint can be found since last evaluation.
 975       Typically, this means the trainer has not yet produced any new checkpoint.
 976   """
 977 
 978   EVALUATED = 'evaluated'
 979   MISSING_CHECKPOINT = 'missing checkpoint'
 980   NO_NEW_CHECKPOINT = 'no new checkpoint'
 981 
 982 
 983 class _EvalResult(
 984     collections.namedtuple('EvalResult',
 985                            ['status', 'metrics', 'checkpoint_path'])):
 986   """_EvalResult holds the result of an evaluation event."""
 987 
 988   def __new__(cls, status, metrics=None, checkpoint_path=None):
 989     """Creates a validated `_EvalResult`.
 990 
 991     Args:
 992       status: See `_EvalStatus`.
 993       metrics: The evaluation results returned by `Estimator.evaluate`. Only set
 994           if status is `EVALUATED`.
 995       checkpoint_path: The corresponding checkpoint path for the `metrics`. Only
 996           set if status is `EVALUATED`.
 997     Returns:
 998       A validated `_EvalResult` object.
 999 
1000     Raises:
1001       ValueError: If validation fails.
1002       TypeError: If any of the arguments is not the expected type.
1003     """
1004 
1005     if status != _EvalStatus.EVALUATED:
1006       if metrics:
1007         raise ValueError(
1008             'metrics must be `None` if status is not {}; got status {},'
1009             ' metrics {}'.format(_EvalStatus.EVALUATED, status, metrics))
1010       if checkpoint_path:
1011         raise ValueError(
1012             'checkpoint must be `None` if status is not {}; got status {}, '
1013             'checkpoint_path {}'.format(_EvalStatus.EVALUATED, status,
1014                                         checkpoint_path))
1015       return super(_EvalResult, cls).__new__(cls, status, metrics,
1016                                              checkpoint_path)
1017 
1018     # Now, evaluated case.
1019     assert status == _EvalStatus.EVALUATED
1020 
1021     # Validates metrics.
1022     if not metrics:
1023       raise ValueError(
1024           'Internal error: `Estimator.evaluate` should never return empty '
1025           'metrics.')
1026     if not isinstance(metrics, dict):
1027       raise TypeError(
1028           '`Estimator.evaluate` should return dict. Given {}.'.format(
1029               type(metrics)))
1030     if ops.GraphKeys.GLOBAL_STEP not in metrics:
1031       raise ValueError(
1032           'Internal error: `Estimator.evaluate` result should have '
1033           '`global_step` in result. Given {}'.format(metrics))
1034 
1035     # Validates checkpoint_path.
1036     if not checkpoint_path:
1037       raise ValueError(
1038           'Internal error: `checkpoint_path` should never be empty.')
1039 
1040     return super(_EvalResult, cls).__new__(cls, status, metrics,
1041                                            checkpoint_path)
