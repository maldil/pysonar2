  54 def _cast_tensor_to_floatx(x):
  55   """Cast tensor to keras's floatx dtype if it is not already the same dtype."""
  56   if x.dtype == K.floatx():
  57     return x
  58   else:
  59     return math_ops.cast(x, K.floatx())

  62 def _convert_tensor(x):
  63   """Create or cast tensor if needed."""
  64   if not tensor_util.is_tensor(x):
  65     # x is a numpy array
  66     x = sparse_tensor_lib.convert_to_tensor_or_sparse_tensor(x)
  67   if check_ops.is_numeric_tensor(x):
  68     # is_numeric_tensor returns False if provided with a numpy array
  69     x = _cast_tensor_to_floatx(x)
  70   return x

  73 def _any_weight_initialized(keras_model):
  74   """Check if any weights has been initialized in the Keras model.
  75 
  76   Args:
  77     keras_model: An instance of compiled keras model.
  78 
  79   Returns:
  80     boolean, True if at least one weight has been initialized, else False.
  81     Currently keras initialize all weights at get_session().
  82   """
  83   if keras_model is None:
  84     return False
  85   for layer in keras_model.layers:
  86     for weight in layer.weights:
  87       if hasattr(weight, '_keras_initialized'):
  88         return True
  89   return False

  92 def _convert_estimator_io_to_keras(keras_model, features, labels):
  93   """Converts estimator features and labels to keras input and target tensors.
  94 
  95   Args:
  96     keras_model: a compiled `tf.keras.Model` instance, used to determine the
  97       order of the returned lists.
  98     features: Dict of tensors or `None`.
  99     labels: Dict of tensors, a single tensor, or `None`.
 100 
 101   Returns:
 102     Tuple of (
 103       list of input tensors or `None`,
 104       list of target tensors or `None`)
 105     The order of tensors is determined by the order set in the keras model.
 106   """
 107 
 108   def _to_ordered_tensor_list(obj, key_order, obj_name, order_name):
 109     """Convert obj to an ordered list of tensors.
 110 
 111     Args:
 112       obj: List, dict, or single tensor. May be `None`.
 113       key_order: List of strings with the order to return (used if obj is a
 114         dict).
 115       obj_name: String name of object (e.g. "features" or "labels")
 116       order_name: String name of the key order (e.g. "inputs" or "outputs")
 117 
 118     Returns:
 119       List of tensors, or `None`
 120 
 121     Raises:
 122       KeyError: If obj has invalid keys.
 123     """
 124     if obj is None:
 125       return None
 126     elif isinstance(obj, (list, tuple)):
 127       return [_convert_tensor(x) for x in obj]
 128     elif isinstance(obj, dict):
 129       # Ensure that the obj keys and keys in key_order are exactly the same.
 130       different_keys = set(obj.keys()) ^ set(key_order)
 131 
 132       if different_keys:
 133         raise KeyError(
 134             'The dictionary passed into {obj_name} does not have the expected '
 135             '{order_name} keys defined in the keras model.'
 136             '\n\tExpected keys: {order_keys}'
 137             '\n\t{obj_name} keys: {obj_keys}'
 138             '\n\tDifference: {different_keys}'.format(
 139                 order_name=order_name, order_keys=set(key_order),
 140                 obj_name=obj_name, obj_keys=set(obj.keys()),
 141                 different_keys=different_keys))
 142 
 143       return [_convert_tensor(obj[key]) for key in key_order]
 144     else:  # Assume obj is a tensor.
 145       return [_convert_tensor(obj)]
 146 
 147   input_names = None
 148   output_names = None
 149   if isinstance(features, dict):
 150     input_names = (
 151         keras_model.input_names if keras_model._is_graph_network else
 152         ['input_%d' % i for i in range(1, len(features) + 1)])
 153   if isinstance(labels, dict):
 154     output_names = (
 155         keras_model.output_names if keras_model._is_graph_network else
 156         ['output_%d' % i for i in range(1, len(labels) + 1)])
 157 
 158   input_tensors = _to_ordered_tensor_list(
 159       features, input_names, 'features', 'inputs')
 160   target_tensors = _to_ordered_tensor_list(
 161       labels, output_names, 'labels', 'outputs')
 162 
 163   return input_tensors, target_tensors

 166 def _clone_and_build_model(mode,
 167                            keras_model,
 168                            custom_objects,
 169                            features=None,
 170                            labels=None):
 171   """Clone and build the given keras_model.
 172 
 173   Args:
 174     mode: training mode.
 175     keras_model: an instance of compiled keras model.
 176     custom_objects: Dictionary for custom objects.
 177     features: Dict of tensors.
 178     labels: Dict of tensors, or single tensor instance.
 179 
 180   Returns:
 181     The newly built model.
 182   """
 183   # Set to True during training, False for inference or testing.
 184   K.set_learning_phase(mode == model_fn_lib.ModeKeys.TRAIN)
 185   input_tensors, target_tensors = _convert_estimator_io_to_keras(
 186       keras_model, features, labels)
 187 
 188   compile_clone = (mode != model_fn_lib.ModeKeys.PREDICT)
 189 
 190   global_step = None
 191   if compile_clone:
 192     # Set iterations to the global step created by tf.train.create_global_step()
 193     # which is automatically run in the estimator framework.
 194     global_step = training_util.get_or_create_global_step()
 195     K.track_variable(global_step)
 196 
 197   clone = models.clone_and_build_model(
 198       keras_model, input_tensors, target_tensors, custom_objects,
 199       compile_clone=compile_clone,
 200       in_place_reset=(not keras_model._is_graph_network),
 201       optimizer_iterations=global_step)
 202 
 203   return clone

 206 def _convert_keras_metrics_to_estimator(model):
 207   """Convert metrics from a Keras model to ops used by the Estimator framework.
 208 
 209   Args:
 210     model: A `tf.keras.Model` object.
 211 
 212   Returns:
 213     Dictionary mapping metric names to tuples of (value, update) ops. May return
 214     `None` if the model does not contain any metrics.
 215   """
 216   if not getattr(model, 'metrics', None):
 217     return None
 218 
 219   eval_metric_ops = {}
 220 
 221   def get_metric_name(metric):
 222     if isinstance(metric, metrics.Metric):
 223       return metric.name
 224     if callable(metric):
 225       return metric.__name__
 226     assert isinstance(metric, six.string_types)
 227     return metric
 228 
 229   # When each metric maps to an output
 230   if isinstance(model.metrics, dict):
 231     for i, output_name in enumerate(model.metrics.keys()):
 232       # `metric` is the user given metric value in `compile`. This can be
 233       # metric name (`acc`), metric function (binary_accuracy) or a metric
 234       # object (BinaryAccuracy()).
 235       metric = model.metrics[output_name]
 236       metric_name = get_metric_name(metric)
 237       # When some outputs use the same metric
 238       if list(model.metrics.values()).count(metric_name) > 1:
 239         metric_name += '_' + output_name
 240       if isinstance(metric, metrics.Metric):
 241         eval_metric_ops[metric_name] = metric
 242       else:
 243         eval_metric_ops[metric_name] = metrics_module.mean(
 244             model.metrics_tensors[i - len(model.metrics)])
 245   else:
 246     for i, metric in enumerate(model.metrics):
 247       metric_name = get_metric_name(metric)
 248       if isinstance(metric, metrics.Metric):
 249         eval_metric_ops[metric_name] = metric
 250       else:
 251         eval_metric_ops[metric_name] = metrics_module.mean(
 252             model.metrics_tensors[i])
 253   return eval_metric_ops

 256 def _create_keras_model_fn(keras_model, custom_objects=None):
 257   """Creates model_fn for keras Estimator.
 258 
 259   Args:
 260     keras_model: an instance of compiled keras model.
 261     custom_objects: Dictionary for custom objects.
 262 
 263   Returns:
 264     The model_fn for a keras Estimator.
 265   """
 266 
 267   def model_fn(features, labels, mode):
 268     """model_fn for keras Estimator."""
 269     # Raise an error when users use DistributionStrategy with native Keras
 270     # optimizers. Currently we only support native TensorFlow optimizers.
 271     if distribution_strategy_context.has_distribution_strategy() and \
 272         not isinstance(keras_model.optimizer,
 273                        (tf_optimizer_module.Optimizer, optimizers.TFOptimizer)):
 274       raise ValueError('Only TensorFlow native optimizers are supported with '
 275                        'DistributionStrategy.')
 276 
 277     model = _clone_and_build_model(mode, keras_model, custom_objects, features,
 278                                    labels)
 279     model_output_names = []
 280     # We need to make sure that the output names of the last layer in the model
 281     # is the same for each of the cloned models. This is required for mirrored
 282     # strategy when we call regroup.
 283     if distribution_strategy_context.has_distribution_strategy():
 284       for name in model.output_names:
 285         name = re.compile(r'_\d$').sub('', name)
 286         model_output_names.append(name)
 287     else:
 288       model_output_names = model.output_names
 289 
 290     # Get inputs to EstimatorSpec
 291     predictions = dict(zip(model_output_names, model.outputs))
 292 
 293     loss = None
 294     train_op = None
 295     eval_metric_ops = None
 296 
 297     # Set loss and metric only during train and evaluate.
 298     if mode is not model_fn_lib.ModeKeys.PREDICT:
 299       if mode is model_fn_lib.ModeKeys.TRAIN:
 300         model._make_train_function()  # pylint: disable=protected-access
 301       else:
 302         model._make_test_function()  # pylint: disable=protected-access
 303       loss = model.total_loss
 304 
 305       eval_metric_ops = _convert_keras_metrics_to_estimator(model)
 306 
 307     # Set train_op only during train.
 308     if mode is model_fn_lib.ModeKeys.TRAIN:
 309       train_op = model.train_function.updates_op
 310 
 311     if not model._is_graph_network:
 312       # Reset model state to original state,
 313       # to avoid `model_fn` being destructive for the initial model argument.
 314       models.in_place_subclassed_model_state_restoration(keras_model)
 315     return model_fn_lib.EstimatorSpec(
 316         mode=mode,
 317         predictions=predictions,
 318         loss=loss,
 319         train_op=train_op,
 320         eval_metric_ops=eval_metric_ops,
 321         export_outputs={
 322             _DEFAULT_SERVING_KEY:
 323             export_lib.export_output.PredictOutput(predictions)
 324         })
 325 
 326   return model_fn

 329 def _save_first_checkpoint(keras_model, custom_objects, config):
 330   """Save first checkpoint for the keras Estimator.
 331 
 332   Args:
 333     keras_model: an instance of compiled keras model.
 334     custom_objects: Dictionary for custom objects.
 335     config: Estimator config.
 336 
 337   Returns:
 338     The path where keras model checkpoint is saved.
 339   """
 340   # save checkpoint into subdirectory to allow warm start
 341   keras_model_dir = os.path.join(config.model_dir, 'keras')
 342   # Load weights and save to checkpoint if there is no checkpoint
 343   latest_path = checkpoint_management.latest_checkpoint(keras_model_dir)
 344   if not latest_path:
 345     keras_weights = None
 346     if _any_weight_initialized(keras_model):
 347       keras_weights = keras_model.get_weights()
 348     if not gfile.IsDirectory(keras_model_dir):
 349       gfile.MakeDirs(keras_model_dir)
 350     with ops.Graph().as_default():
 351       random_seed.set_random_seed(config.tf_random_seed)
 352       training_util.create_global_step()
 353       model = _clone_and_build_model(model_fn_lib.ModeKeys.TRAIN, keras_model,
 354                                      custom_objects)
 355       # save to checkpoint
 356       with session.Session(config=config.session_config) as sess:
 357         if keras_weights:
 358           model.set_weights(keras_weights)
 359         # Make update ops and initialize all variables.
 360         if not model.train_function:
 361           # pylint: disable=protected-access
 362           model._make_train_function()
 363           K._initialize_variables(sess)
 364           # pylint: enable=protected-access
 365         saver = saver_lib.Saver()
 366         latest_path = os.path.join(keras_model_dir, 'keras_model.ckpt')
 367         saver.save(sess, latest_path)
 368   return latest_path

 371 def _get_file_from_google_storage(keras_model_path, model_dir):
 372   """Get file from google storage and download to local file.
 373 
 374   Args:
 375     keras_model_path: a google storage path for compiled keras model.
 376     model_dir: the directory from estimator config.
 377 
 378   Returns:
 379     The path where keras model is saved.
 380 
 381   Raises:
 382     ValueError: if storage object name does not end with .h5.
 383   """
 384   try:
 385     from google.cloud import storage  # pylint:disable=g-import-not-at-top
 386   except ImportError:
 387     raise TypeError('Could not save model to Google cloud storage; please '
 388                     'install `google-cloud-storage` via '
 389                     '`pip install google-cloud-storage`.')
 390   storage_client = storage.Client()
 391   path, blob_name = os.path.split(keras_model_path)
 392   _, bucket_name = os.path.split(path)
 393   keras_model_dir = os.path.join(model_dir, 'keras')
 394   if not gfile.Exists(keras_model_dir):
 395     gfile.MakeDirs(keras_model_dir)
 396   file_name = os.path.join(keras_model_dir, 'keras_model.h5')
 397   try:
 398     blob = storage_client.get_bucket(bucket_name).blob(blob_name)
 399     blob.download_to_filename(file_name)
 400   except:
 401     raise ValueError('Failed to download keras model, please check '
 402                      'environment variable GOOGLE_APPLICATION_CREDENTIALS '
 403                      'and model path storage.googleapis.com/{bucket}/{object}.')
 404   logging.info('Saving model to {}'.format(file_name))
 405   del storage_client
 406   return file_name

 409 def model_to_estimator(keras_model=None,
 410                        keras_model_path=None,
 411                        custom_objects=None,
 412                        model_dir=None,
 413                        config=None):
 414   """Constructs an `Estimator` instance from given keras model.
 415 
 416   For usage example, please see:
 417   [Creating estimators from Keras
 418   Models](https://tensorflow.org/guide/estimators#model_to_estimator).
 419 
 420   Args:
 421     keras_model: A compiled Keras model object. This argument is mutually
 422       exclusive with `keras_model_path`.
 423     keras_model_path: Path to a compiled Keras model saved on disk, in HDF5
 424       format, which can be generated with the `save()` method of a Keras model.
 425       This argument is mutually exclusive with `keras_model`.
 426     custom_objects: Dictionary for custom objects.
 427     model_dir: Directory to save `Estimator` model parameters, graph, summary
 428       files for TensorBoard, etc.
 429     config: `RunConfig` to config `Estimator`.
 430 
 431   Returns:
 432     An Estimator from given keras model.
 433 
 434   Raises:
 435     ValueError: if neither keras_model nor keras_model_path was given.
 436     ValueError: if both keras_model and keras_model_path was given.
 437     ValueError: if the keras_model_path is a GCS URI.
 438     ValueError: if keras_model has not been compiled.
 439   """
 440   if not (keras_model or keras_model_path):
 441     raise ValueError(
 442         'Either `keras_model` or `keras_model_path` needs to be provided.')
 443   if keras_model and keras_model_path:
 444     raise ValueError(
 445         'Please specity either `keras_model` or `keras_model_path`, '
 446         'but not both.')
 447 
 448   config = estimator_lib.maybe_overwrite_model_dir_and_session_config(
 449       config, model_dir)
 450   if not keras_model:
 451     if keras_model_path.startswith(
 452         'gs://') or 'storage.googleapis.com' in keras_model_path:
 453       keras_model_path = _get_file_from_google_storage(keras_model_path,
 454                                                        config.model_dir)
 455     logging.info('Loading models from %s', keras_model_path)
 456     keras_model = models.load_model(keras_model_path)
 457   else:
 458     logging.info('Using the Keras model provided.')
 459     keras_model = keras_model
 460 
 461   if not hasattr(keras_model, 'optimizer') or not keras_model.optimizer:
 462     raise ValueError(
 463         'The given keras model has not been compiled yet. '
 464         'Please compile the model with `model.compile()` '
 465         'before calling `model_to_estimator()`.')
 466 
 467   keras_model_fn = _create_keras_model_fn(keras_model, custom_objects)
 468   if _any_weight_initialized(keras_model):
 469     # Warn if config passed to estimator tries to update GPUOptions. If a
 470     # session has already been created, the GPUOptions passed to the first
 471     # session sticks.
 472     if config.session_config.HasField('gpu_options'):
 473       logging.warning(
 474           'The Keras backend session has already been set. '
 475           'The _session_config passed to model_to_estimator will not be used.')
 476   else:
 477     # Pass the config into keras backend's default session.
 478     sess = session.Session(config=config.session_config)
 479     K.set_session(sess)
 480 
 481   warm_start_path = None
 482   if keras_model._is_graph_network:
 483     warm_start_path = _save_first_checkpoint(keras_model, custom_objects,
 484                                              config)
 485   elif keras_model.built:
 486     logging.warning('You are creating an Estimator from a Keras model manually '
 487                     'subclassed from `Model`, that was already called on some '
 488                     'inputs (and thus already had weights). We are currently '
 489                     'unable to preserve the model\'s state (its weights) as '
 490                     'part of the estimator in this case. Be warned that the '
 491                     'estimator has been created using a freshly initialized '
 492                     'version of your model.\n'
 493                     'Note that this doesn\'t affect the state of the model '
 494                     'instance you passed as `keras_model` argument.')
 495 
 496   estimator = estimator_lib.Estimator(keras_model_fn,
 497                                       config=config,
 498                                       warm_start_from=warm_start_path)
 499 
 500   return estimator
