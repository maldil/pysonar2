  29 class _FeedingQueueRunner(qr.QueueRunner):
  30   """A queue runner that allows the feeding of values such as numpy arrays."""
  31 
  32   def __init__(self, queue=None, enqueue_ops=None, close_op=None,
  33                cancel_op=None, feed_fns=None,
  34                queue_closed_exception_types=None):
  35     """Initialize the queue runner.
  36 
  37     For further documentation, see `queue_runner.py`. Note that
  38     `FeedingQueueRunner` does not support construction from protobuffer nor
  39     serialization to protobuffer.
  40 
  41     Args:
  42       queue: A `Queue`.
  43       enqueue_ops: List of enqueue ops to run in threads later.
  44       close_op: Op to close the queue. Pending enqueue ops are preserved.
  45       cancel_op: Op to close the queue and cancel pending enqueue ops.
  46       feed_fns: a list of functions that return a dictionary mapping fed
  47         `Tensor`s to values. Must be the same length as `enqueue_ops`.
  48       queue_closed_exception_types: Optional tuple of Exception types that
  49         indicate that the queue has been closed when raised during an enqueue
  50         operation.  Defaults to
  51         `(tf.errors.OutOfRangeError, tf.errors.CancelledError)`.
  52 
  53     Raises:
  54       ValueError: `feed_fns` is not `None` and has different length than
  55         `enqueue_ops`.
  56     """
  57     if queue_closed_exception_types is None:
  58       queue_closed_exception_types = (
  59           errors.OutOfRangeError, errors.CancelledError)
  60     super(_FeedingQueueRunner, self).__init__(
  61         queue, enqueue_ops, close_op,
  62         cancel_op, queue_closed_exception_types=queue_closed_exception_types)
  63     if feed_fns is None:
  64       self._feed_fns = [None for _ in enqueue_ops]
  65     else:
  66       if len(feed_fns) != len(enqueue_ops):
  67         raise ValueError(
  68             "If feed_fns is not None, it must have the same length as "
  69             "enqueue_ops.")
  70       self._feed_fns = feed_fns
  71 
  72   # pylint: disable=broad-except
  73   def _run(self, sess, enqueue_op, feed_fn, coord=None):
  74     """Execute the enqueue op in a loop, close the queue in case of error.
  75 
  76     Args:
  77       sess: A `Session`.
  78       enqueue_op: The `Operation` to run.
  79       feed_fn: the feed function to pass to `sess.run`.
  80       coord: Optional `Coordinator` object for reporting errors and checking
  81         for stop conditions.
  82 
  83     """
  84     # TODO(jamieas): Reduce code duplication with `QueueRunner`.
  85     if coord:
  86       coord.register_thread(threading.current_thread())
  87     decremented = False
  88     try:
  89       while True:
  90         if coord and coord.should_stop():
  91           break
  92         try:
  93           feed_dict = None if feed_fn is None else feed_fn()
  94           sess.run(enqueue_op, feed_dict=feed_dict)
  95         except (errors.OutOfRangeError, errors.CancelledError):
  96           # This exception indicates that a queue was closed.
  97           with self._lock:
  98             self._runs_per_session[sess] -= 1
  99             decremented = True
 100             if self._runs_per_session[sess] == 0:
 101               try:
 102                 sess.run(self._close_op)
 103               except Exception as e:
 104                 # Intentionally ignore errors from close_op.
 105                 logging.vlog(1, "Ignored exception: %s", str(e))
 106             return
 107     except Exception as e:
 108       # This catches all other exceptions.
 109       if coord:
 110         coord.request_stop(e)
 111       else:
 112         logging.error("Exception in QueueRunner: %s", str(e))
 113         with self._lock:
 114           self._exceptions_raised.append(e)
 115         raise
 116     finally:
 117       # Make sure we account for all terminations: normal or errors.
 118       if not decremented:
 119         with self._lock:
 120           self._runs_per_session[sess] -= 1
 121 
 122   def create_threads(self, sess, coord=None, daemon=False, start=False):
 123     """Create threads to run the enqueue ops for the given session.
 124 
 125     This method requires a session in which the graph was launched.  It creates
 126     a list of threads, optionally starting them.  There is one thread for each
 127     op passed in `enqueue_ops`.
 128 
 129     The `coord` argument is an optional coordinator, that the threads will use
 130     to terminate together and report exceptions.  If a coordinator is given,
 131     this method starts an additional thread to close the queue when the
 132     coordinator requests a stop.
 133 
 134     If previously created threads for the given session are still running, no
 135     new threads will be created.
 136 
 137     Args:
 138       sess: A `Session`.
 139       coord: Optional `Coordinator` object for reporting errors and checking
 140         stop conditions.
 141       daemon: Boolean.  If `True` make the threads daemon threads.
 142       start: Boolean.  If `True` starts the threads.  If `False` the
 143         caller must call the `start()` method of the returned threads.
 144 
 145     Returns:
 146       A list of threads.
 147     """
 148     with self._lock:
 149       try:
 150         if self._runs_per_session[sess] > 0:
 151           # Already started: no new threads to return.
 152           return []
 153       except KeyError:
 154         # We haven't seen this session yet.
 155         pass
 156       self._runs_per_session[sess] = len(self._enqueue_ops)
 157       self._exceptions_raised = []
 158 
 159     ret_threads = [threading.Thread(target=self._run,
 160                                     args=(sess, op, feed_fn, coord))
 161                    for op, feed_fn in zip(self._enqueue_ops, self._feed_fns)]
 162     if coord:
 163       ret_threads.append(threading.Thread(target=self._close_on_stop,
 164                                           args=(sess, self._cancel_op, coord)))
 165     for t in ret_threads:
 166       if daemon:
 167         t.daemon = True
 168       if start:
 169         t.start()
 170     return ret_threads
