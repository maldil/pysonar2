  49 def _fill_array(arr, seq, fillvalue=0):
  50   """Recursively fills padded arr with elements from seq.
  51 
  52   If length of seq is less than arr padded length, fillvalue used.
  53   Args:
  54     arr: Padded tensor of shape [batch_size, ..., max_padded_dim_len].
  55     seq: Non-padded list of data samples of shape
  56       [batch_size, ..., padded_dim(None)]
  57     fillvalue: Default fillvalue to use.
  58   """
  59   if arr.ndim == 1:
  60     try:
  61       len_ = len(seq)
  62     except TypeError:
  63       len_ = 0
  64     arr[:len_] = seq
  65     arr[len_:] = fillvalue
  66   else:
  67     for subarr, subseq in six.moves.zip_longest(arr, seq, fillvalue=()):
  68       _fill_array(subarr, subseq, fillvalue)

  71 def _pad_if_needed(batch_key_item, fillvalue=0):
  72   """ Returns padded batch.
  73 
  74   Args:
  75     batch_key_item: List of data samples of any type with shape
  76       [batch_size, ..., padded_dim(None)].
  77     fillvalue: Default fillvalue to use.
  78 
  79   Returns:
  80     Padded with zeros tensor of same type and shape
  81       [batch_size, ..., max_padded_dim_len].
  82 
  83   Raises:
  84     ValueError if data samples have different shapes (except last padded dim).
  85   """
  86   shapes = [
  87       seq.shape[:-1] if len(seq.shape) > 0 else -1 for seq in batch_key_item
  88   ]
  89   if not all(shapes[0] == x for x in shapes):
  90     raise ValueError("Array shapes must match.")
  91 
  92   last_length = [
  93       seq.shape[-1] if len(seq.shape) > 0 else 0 for seq in batch_key_item
  94   ]
  95   if all([x == last_length[0] for x in last_length]):
  96     return batch_key_item
  97 
  98   batch_size = len(batch_key_item)
  99   max_sequence_length = max(last_length)
 100   result_batch = np.zeros(
 101       shape=[batch_size] + list(shapes[0]) + [max_sequence_length],
 102       dtype=batch_key_item[0].dtype)
 103   _fill_array(result_batch, batch_key_item, fillvalue)
 104   return result_batch

 107 def _get_integer_indices_for_next_batch(batch_indices_start, batch_size,
 108                                         epoch_end, array_length, current_epoch,
 109                                         total_epochs):
 110   """Returns the integer indices for next batch.
 111 
 112   If total epochs is not None and current epoch is the final epoch, the end
 113   index of the next batch should not exceed the `epoch_end` (i.e., the final
 114   batch might not have size `batch_size` to avoid overshooting the last epoch).
 115 
 116   Args:
 117     batch_indices_start: Integer, the index to start next batch.
 118     batch_size: Integer, size of batches to return.
 119     epoch_end: Integer, the end index of the epoch. The epoch could start from a
 120       random position, so `epoch_end` provides the end index for that.
 121     array_length: Integer, the length of the array.
 122     current_epoch: Integer, the epoch number has been emitted.
 123     total_epochs: Integer or `None`, the total number of epochs to emit. If
 124       `None` will run forever.
 125 
 126   Returns:
 127     A tuple of a list with integer indices for next batch and `current_epoch`
 128     value after the next batch.
 129 
 130   Raises:
 131     OutOfRangeError if `current_epoch` is not less than `total_epochs`.
 132 
 133   """
 134   if total_epochs is not None and current_epoch >= total_epochs:
 135     raise errors.OutOfRangeError(None, None,
 136                                  "Already emitted %s epochs." % current_epoch)
 137 
 138   batch_indices_end = batch_indices_start + batch_size
 139   batch_indices = [
 140       j % array_length for j in range(batch_indices_start, batch_indices_end)
 141   ]
 142   epoch_end_indices = [i for i, x in enumerate(batch_indices) if x == epoch_end]
 143   current_epoch += len(epoch_end_indices)
 144 
 145   if total_epochs is None or current_epoch < total_epochs:
 146     return (batch_indices, current_epoch)
 147 
 148   # Now we might have emitted more data for expected epochs. Need to trim.
 149   final_epoch_end_inclusive = epoch_end_indices[
 150       -(current_epoch - total_epochs + 1)]
 151   batch_indices = batch_indices[:final_epoch_end_inclusive + 1]
 152 
 153   return (batch_indices, total_epochs)

 156 class _ArrayFeedFn(object):
 157   """Creates feed dictionaries from numpy arrays."""
 158 
 159   def __init__(self,
 160                placeholders,
 161                array,
 162                batch_size,
 163                random_start=False,
 164                seed=None,
 165                num_epochs=None):
 166     if len(placeholders) != 2:
 167       raise ValueError("_array_feed_fn expects 2 placeholders; got {}.".format(
 168           len(placeholders)))
 169     self._placeholders = placeholders
 170     self._array = array
 171     self._max = len(array)
 172     self._batch_size = batch_size
 173     self._num_epochs = num_epochs
 174     self._epoch = 0
 175     random.seed(seed)
 176     self._trav = random.randrange(self._max) if random_start else 0
 177     self._epoch_end = (self._trav - 1) % self._max
 178 
 179   def __call__(self):
 180     integer_indexes, self._epoch = _get_integer_indices_for_next_batch(
 181         batch_indices_start=self._trav,
 182         batch_size=self._batch_size,
 183         epoch_end=self._epoch_end,
 184         array_length=self._max,
 185         current_epoch=self._epoch,
 186         total_epochs=self._num_epochs)
 187 
 188     self._trav = (integer_indexes[-1] + 1) % self._max
 189     return {
 190         self._placeholders[0]: integer_indexes,
 191         self._placeholders[1]: self._array[integer_indexes]
 192     }

 195 class _OrderedDictNumpyFeedFn(object):
 196   """Creates feed dictionaries from `OrderedDict`s of numpy arrays."""
 197 
 198   def __init__(self,
 199                placeholders,
 200                ordered_dict_of_arrays,
 201                batch_size,
 202                random_start=False,
 203                seed=None,
 204                num_epochs=None):
 205     if len(placeholders) != len(ordered_dict_of_arrays) + 1:
 206       raise ValueError("Expected {} placeholders; got {}.".format(
 207           len(ordered_dict_of_arrays), len(placeholders)))
 208     self._index_placeholder = placeholders[0]
 209     self._col_placeholders = placeholders[1:]
 210     self._ordered_dict_of_arrays = ordered_dict_of_arrays
 211     self._max = len(next(iter(ordered_dict_of_arrays.values())))
 212     for _, v in ordered_dict_of_arrays.items():
 213       if len(v) != self._max:
 214         raise ValueError("Array lengths must match.")
 215     self._batch_size = batch_size
 216     self._num_epochs = num_epochs
 217     self._epoch = 0
 218     random.seed(seed)
 219     self._trav = random.randrange(self._max) if random_start else 0
 220     self._epoch_end = (self._trav - 1) % self._max
 221 
 222   def __call__(self):
 223     integer_indexes, self._epoch = _get_integer_indices_for_next_batch(
 224         batch_indices_start=self._trav,
 225         batch_size=self._batch_size,
 226         epoch_end=self._epoch_end,
 227         array_length=self._max,
 228         current_epoch=self._epoch,
 229         total_epochs=self._num_epochs)
 230 
 231     self._trav = (integer_indexes[-1] + 1) % self._max
 232     feed_dict = {self._index_placeholder: integer_indexes}
 233     cols = [
 234         column[integer_indexes]
 235         for column in self._ordered_dict_of_arrays.values()
 236     ]
 237     feed_dict.update(dict(zip(self._col_placeholders, cols)))
 238     return feed_dict

 241 class _PandasFeedFn(object):
 242   """Creates feed dictionaries from pandas `DataFrames`."""
 243 
 244   def __init__(self,
 245                placeholders,
 246                dataframe,
 247                batch_size,
 248                random_start=False,
 249                seed=None,
 250                num_epochs=None):
 251     if len(placeholders) != len(dataframe.columns) + 1:
 252       raise ValueError("Expected {} placeholders; got {}.".format(
 253           len(dataframe.columns) + 1, len(placeholders)))
 254     self._index_placeholder = placeholders[0]
 255     self._col_placeholders = placeholders[1:]
 256     self._dataframe = dataframe
 257     self._max = len(dataframe)
 258     self._batch_size = batch_size
 259     self._num_epochs = num_epochs
 260     self._epoch = 0
 261     random.seed(seed)
 262     self._trav = random.randrange(self._max) if random_start else 0
 263     self._epoch_end = (self._trav - 1) % self._max
 264 
 265   def __call__(self):
 266     integer_indexes, self._epoch = _get_integer_indices_for_next_batch(
 267         batch_indices_start=self._trav,
 268         batch_size=self._batch_size,
 269         epoch_end=self._epoch_end,
 270         array_length=self._max,
 271         current_epoch=self._epoch,
 272         total_epochs=self._num_epochs)
 273 
 274     self._trav = (integer_indexes[-1] + 1) % self._max
 275     result = self._dataframe.iloc[integer_indexes]
 276     cols = [result[col].values for col in result.columns]
 277     feed_dict = dict(zip(self._col_placeholders, cols))
 278     feed_dict[self._index_placeholder] = result.index.values
 279     return feed_dict

 282 class _GeneratorFeedFn(object):
 283   """Creates feed dictionaries from `Generator` of `dicts` of numpy arrays."""
 284 
 285   def __init__(self,
 286                placeholders,
 287                generator,
 288                batch_size,
 289                random_start=False,
 290                seed=None,
 291                num_epochs=None,
 292                pad_value=None):
 293     first_sample = next(generator())
 294     if len(placeholders) != len(first_sample):
 295       raise ValueError("Expected {} placeholders; got {}.".format(
 296           len(first_sample), len(placeholders)))
 297     self._keys = sorted(list(first_sample.keys()))
 298     self._col_placeholders = placeholders
 299     self._generator_function = generator
 300     self._iterator = generator()
 301     self._batch_size = batch_size
 302     self._num_epochs = num_epochs
 303     self._epoch = 0
 304     self._pad_value = pad_value
 305     random.seed(seed)
 306 
 307   def __call__(self):
 308     if self._num_epochs and self._epoch >= self._num_epochs:
 309       raise errors.OutOfRangeError(None, None,
 310                                    "Already emitted %s epochs." % self._epoch)
 311     list_dict = {}
 312     list_dict_size = 0
 313     while list_dict_size < self._batch_size:
 314       try:
 315         data_row = next(self._iterator)
 316       except StopIteration:
 317         self._epoch += 1
 318         self._iterator = self._generator_function()
 319         data_row = next(self._iterator)
 320       for index, key in enumerate(self._keys):
 321         if key not in data_row.keys():
 322           raise KeyError("key mismatch between dicts emitted by GenFun "
 323                          "Expected {} keys; got {}".format(
 324                              self._keys, data_row.keys()))
 325         list_dict.setdefault(self._col_placeholders[index], list()).append(
 326             data_row[key])
 327         list_dict_size += 1
 328 
 329     if self._pad_value is not None:
 330       feed_dict = {
 331           key: np.asarray(_pad_if_needed(item, self._pad_value))
 332           for key, item in list(list_dict.items())
 333       }
 334     else:
 335       feed_dict = {
 336           key: np.asarray(item)
 337           for key, item in list(list_dict.items())
 338       }
 339     return feed_dict

 342 def _enqueue_data(data,
 343                   capacity,
 344                   shuffle=False,
 345                   min_after_dequeue=None,
 346                   num_threads=1,
 347                   seed=None,
 348                   name="enqueue_input",
 349                   enqueue_size=1,
 350                   num_epochs=None,
 351                   pad_value=None):
 352   """Creates a queue filled from a numpy array or pandas `DataFrame`.
 353 
 354     Returns a queue filled with the rows of the given (`OrderedDict` of) array
 355     or `DataFrame`. In the case of a pandas `DataFrame`, the first enqueued
 356     `Tensor` corresponds to the index of the `DataFrame`. For (`OrderedDict` of)
 357     numpy arrays, the first enqueued `Tensor` contains the row number.
 358 
 359   Args:
 360     data: a numpy `ndarray`, `OrderedDict` of numpy arrays, or a generator
 361        yielding `dict`s of numpy arrays or pandas `DataFrame` that will be read
 362        into the queue.
 363     capacity: the capacity of the queue.
 364     shuffle: whether or not to shuffle the rows of the array.
 365     min_after_dequeue: minimum number of elements that can remain in the queue
 366     after a dequeue operation. Only used when `shuffle` is true. If not set,
 367     defaults to `capacity` / 4.
 368     num_threads: number of threads used for reading and enqueueing.
 369     seed: used to seed shuffling and reader starting points.
 370     name: a scope name identifying the data.
 371     enqueue_size: the number of rows to enqueue per step.
 372     num_epochs: limit enqueuing to a specified number of epochs, if provided.
 373     pad_value: default value for dynamic padding of data samples, if provided.
 374 
 375   Returns:
 376     A queue filled with the rows of the given (`OrderedDict` of) array or
 377       `DataFrame`.
 378 
 379   Raises:
 380     TypeError: `data` is not a Pandas `DataFrame`, an `OrderedDict` of numpy
 381       arrays, a numpy `ndarray`, or a generator producing these.
 382     NotImplementedError: padding and shuffling data at the same time.
 383     NotImplementedError: padding usage with non generator data type.
 384   """
 385   with ops.name_scope(name):
 386     if isinstance(data, np.ndarray):
 387       types = [dtypes.int64, dtypes.as_dtype(data.dtype)]
 388       queue_shapes = [(), data.shape[1:]]
 389       get_feed_fn = _ArrayFeedFn
 390     elif isinstance(data, collections.OrderedDict):
 391       types = [dtypes.int64
 392               ] + [dtypes.as_dtype(col.dtype) for col in data.values()]
 393       queue_shapes = [()] + [col.shape[1:] for col in data.values()]
 394       get_feed_fn = _OrderedDictNumpyFeedFn
 395     elif isinstance(data, tp.FunctionType):
 396       x_first_el = six.next(data())
 397       x_first_keys = sorted(x_first_el.keys())
 398       x_first_values = [x_first_el[key] for key in x_first_keys]
 399       types = [dtypes.as_dtype(col.dtype) for col in x_first_values]
 400       queue_shapes = [col.shape for col in x_first_values]
 401       get_feed_fn = _GeneratorFeedFn
 402     elif HAS_PANDAS and isinstance(data, pd.DataFrame):
 403       types = [
 404           dtypes.as_dtype(dt) for dt in [data.index.dtype] + list(data.dtypes)
 405       ]
 406       queue_shapes = [() for _ in types]
 407       get_feed_fn = _PandasFeedFn
 408     else:
 409       raise TypeError(
 410           "data must be either a numpy array or pandas DataFrame if pandas is "
 411           "installed; got {}".format(type(data).__name__))
 412 
 413     pad_data = pad_value is not None
 414     if pad_data and get_feed_fn is not _GeneratorFeedFn:
 415       raise NotImplementedError(
 416           "padding is only available with generator usage")
 417     if shuffle and pad_data:
 418       raise NotImplementedError(
 419           "padding and shuffling data at the same time is not implemented")
 420 
 421     # TODO(jamieas): TensorBoard warnings for all warnings below once available.
 422 
 423     if num_threads > 1 and num_epochs is not None:
 424       logging.warning(
 425           "enqueue_data was called with num_epochs and num_threads > 1. "
 426           "num_epochs is applied per thread, so this will produce more "
 427           "epochs than you probably intend. "
 428           "If you want to limit epochs, use one thread.")
 429 
 430     if shuffle and num_threads > 1 and num_epochs is not None:
 431       logging.warning(
 432           "enqueue_data was called with shuffle=True, num_threads > 1, and "
 433           "num_epochs. This will create multiple threads, all reading the "
 434           "array/dataframe in order adding to the same shuffling queue; the "
 435           "results will likely not be sufficiently shuffled.")
 436 
 437     if not shuffle and num_threads > 1:
 438       logging.warning(
 439           "enqueue_data was called with shuffle=False and num_threads > 1. "
 440           "This will create multiple threads, all reading the "
 441           "array/dataframe in order. If you want examples read in order, use"
 442           " one thread; if you want multiple threads, enable shuffling.")
 443 
 444     if shuffle:
 445       min_after_dequeue = int(capacity / 4 if min_after_dequeue is None else
 446                               min_after_dequeue)
 447       queue = data_flow_ops.RandomShuffleQueue(
 448           capacity,
 449           min_after_dequeue,
 450           dtypes=types,
 451           shapes=queue_shapes,
 452           seed=seed)
 453     elif pad_data:
 454       min_after_dequeue = 0  # just for the summary text
 455       queue_shapes = list(
 456           map(lambda x: tuple(list(x[:-1]) + [None]) if len(x) > 0 else x,
 457               queue_shapes))
 458       queue = data_flow_ops.PaddingFIFOQueue(
 459           capacity, dtypes=types, shapes=queue_shapes)
 460     else:
 461       min_after_dequeue = 0  # just for the summary text
 462       queue = data_flow_ops.FIFOQueue(
 463           capacity, dtypes=types, shapes=queue_shapes)
 464 
 465     enqueue_ops = []
 466     feed_fns = []
 467 
 468     for i in range(num_threads):
 469       # Note the placeholders have no shapes, so they will accept any
 470       # enqueue_size.  enqueue_many below will break them up.
 471       placeholders = [array_ops.placeholder(t) for t in types]
 472 
 473       enqueue_ops.append(queue.enqueue_many(placeholders))
 474       seed_i = None if seed is None else (i + 1) * seed
 475 
 476       if not pad_data:
 477         feed_fns.append(
 478             get_feed_fn(
 479                 placeholders,
 480                 data,
 481                 enqueue_size,
 482                 random_start=shuffle,
 483                 seed=seed_i,
 484                 num_epochs=num_epochs))
 485       else:
 486         feed_fns.append(
 487             get_feed_fn(
 488                 placeholders,
 489                 data,
 490                 enqueue_size,
 491                 random_start=shuffle,
 492                 seed=seed_i,
 493                 num_epochs=num_epochs,
 494                 pad_value=pad_value))
 495 
 496     runner = fqr._FeedingQueueRunner(  # pylint: disable=protected-access
 497         queue=queue,
 498         enqueue_ops=enqueue_ops,
 499         feed_fns=feed_fns)
 500     queue_runner.add_queue_runner(runner)
 501 
 502     full = (
 503         math_ops.cast(
 504             math_ops.maximum(0,
 505                              queue.size() - min_after_dequeue), dtypes.float32)
 506         * (1. / (capacity - min_after_dequeue)))
 507     # Note that name contains a '/' at the end so we intentionally do not place
 508     # a '/' after %s below.
 509     summary_name = ("queue/%sfraction_over_%d_of_%d_full" %
 510                     (queue.name, min_after_dequeue,
 511                      capacity - min_after_dequeue))
 512     summary.scalar(summary_name, full)
 513     return queue
