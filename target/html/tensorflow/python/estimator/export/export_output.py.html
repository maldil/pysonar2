  34 @estimator_export('estimator.export.ExportOutput')
  35 class ExportOutput(object):
  36   """Represents an output of a model that can be served.
  37 
  38   These typically correspond to model heads.
  39   """
  40 
  41   __metaclass__ = abc.ABCMeta
  42 
  43   _SEPARATOR_CHAR = '/'
  44 
  45   @abc.abstractmethod
  46   def as_signature_def(self, receiver_tensors):
  47     """Generate a SignatureDef proto for inclusion in a MetaGraphDef.
  48 
  49     The SignatureDef will specify outputs as described in this ExportOutput,
  50     and will use the provided receiver_tensors as inputs.
  51 
  52     Args:
  53       receiver_tensors: a `Tensor`, or a dict of string to `Tensor`, specifying
  54         input nodes that will be fed.
  55     """
  56     pass
  57 
  58   def _check_output_key(self, key, error_label):
  59     # For multi-head models, the key can be a tuple.
  60     if isinstance(key, tuple):
  61       key = self._SEPARATOR_CHAR.join(key)
  62 
  63     if not isinstance(key, six.string_types):
  64       raise ValueError(
  65           '{} output key must be a string; got {}.'.format(error_label, key))
  66     return key
  67 
  68   def _wrap_and_check_outputs(
  69       self, outputs, single_output_default_name, error_label=None):
  70     """Wraps raw tensors as dicts and checks type.
  71 
  72     Note that we create a new dict here so that we can overwrite the keys
  73     if necessary.
  74 
  75     Args:
  76       outputs: A `Tensor` or a dict of string to `Tensor`.
  77       single_output_default_name: A string key for use in the output dict
  78         if the provided `outputs` is a raw tensor.
  79       error_label: descriptive string for use in error messages. If none,
  80         single_output_default_name will be used.
  81 
  82     Returns:
  83       A dict of tensors
  84 
  85     Raises:
  86       ValueError: if the outputs dict keys are not strings or tuples of strings
  87         or the values are not Tensors.
  88     """
  89     if not isinstance(outputs, dict):
  90       outputs = {single_output_default_name: outputs}
  91 
  92     output_dict = {}
  93     for key, value in outputs.items():
  94       error_name = error_label or single_output_default_name
  95       key = self._check_output_key(key, error_name)
  96       if not isinstance(value, ops.Tensor):
  97         raise ValueError(
  98             '{} output value must be a Tensor; got {}.'.format(
  99                 error_name, value))
 100 
 101       output_dict[key] = value
 102     return output_dict

 105 @estimator_export('estimator.export.ClassificationOutput')
 106 class ClassificationOutput(ExportOutput):
 107   """Represents the output of a classification head.
 108 
 109   Either classes or scores or both must be set.
 110 
 111   The classes `Tensor` must provide string labels, not integer class IDs.
 112 
 113   If only classes is set, it is interpreted as providing top-k results in
 114   descending order.
 115 
 116   If only scores is set, it is interpreted as providing a score for every class
 117   in order of class ID.
 118 
 119   If both classes and scores are set, they are interpreted as zipped, so each
 120   score corresponds to the class at the same index.  Clients should not depend
 121   on the order of the entries.
 122   """
 123 
 124   def __init__(self, scores=None, classes=None):
 125     """Constructor for `ClassificationOutput`.
 126 
 127     Args:
 128       scores: A float `Tensor` giving scores (sometimes but not always
 129           interpretable as probabilities) for each class.  May be `None`, but
 130           only if `classes` is set.  Interpretation varies-- see class doc.
 131       classes: A string `Tensor` giving predicted class labels.  May be `None`,
 132           but only if `scores` is set.  Interpretation varies-- see class doc.
 133 
 134     Raises:
 135       ValueError: if neither classes nor scores is set, or one of them is not a
 136           `Tensor` with the correct dtype.
 137     """
 138     if (scores is not None
 139         and not (isinstance(scores, ops.Tensor)
 140                  and scores.dtype.is_floating)):
 141       raise ValueError('Classification scores must be a float32 Tensor; '
 142                        'got {}'.format(scores))
 143     if (classes is not None
 144         and not (isinstance(classes, ops.Tensor)
 145                  and dtypes.as_dtype(classes.dtype) == dtypes.string)):
 146       raise ValueError('Classification classes must be a string Tensor; '
 147                        'got {}'.format(classes))
 148     if scores is None and classes is None:
 149       raise ValueError('At least one of scores and classes must be set.')
 150 
 151     self._scores = scores
 152     self._classes = classes
 153 
 154   @property
 155   def scores(self):
 156     return self._scores
 157 
 158   @property
 159   def classes(self):
 160     return self._classes
 161 
 162   def as_signature_def(self, receiver_tensors):
 163     if len(receiver_tensors) != 1:
 164       raise ValueError('Classification input must be a single string Tensor; '
 165                        'got {}'.format(receiver_tensors))
 166     (_, examples), = receiver_tensors.items()
 167     if dtypes.as_dtype(examples.dtype) != dtypes.string:
 168       raise ValueError('Classification input must be a single string Tensor; '
 169                        'got {}'.format(receiver_tensors))
 170     return signature_def_utils.classification_signature_def(
 171         examples, self.classes, self.scores)

 176   """Represents the output of a regression head."""

 178   def __init__(self, value):
 179     """Constructor for `RegressionOutput`.
 180

 198       raise ValueError('Regression input must be a single string Tensor; '

 202       raise ValueError('Regression input must be a single string Tensor; '

 204     return signature_def_utils.regression_signature_def(examples, self.value)

 207 @estimator_export('estimator.export.PredictOutput')
 208 class PredictOutput(ExportOutput):
 209   """Represents the output of a generic prediction head.
 210 
 211   A generic prediction need not be either a classification or a regression.
 212 
 213   Named outputs must be provided as a dict from string to `Tensor`,
 214   """
 215   _SINGLE_OUTPUT_DEFAULT_NAME = 'output'
 216 
 217   def __init__(self, outputs):
 218     """Constructor for PredictOutput.
 219 
 220     Args:
 221       outputs: A `Tensor` or a dict of string to `Tensor` representing the
 222         predictions.
 223 
 224     Raises:
 225       ValueError: if the outputs is not dict, or any of its keys are not
 226           strings, or any of its values are not `Tensor`s.
 227     """
 228 
 229     self._outputs = self._wrap_and_check_outputs(
 230         outputs, self._SINGLE_OUTPUT_DEFAULT_NAME, error_label='Prediction')
 231 
 232   @property
 233   def outputs(self):
 234     return self._outputs
 235 
 236   def as_signature_def(self, receiver_tensors):
 237     return signature_def_utils.predict_signature_def(receiver_tensors,
 238                                                      self.outputs)

 241 class _SupervisedOutput(ExportOutput):
 242   """Represents the output of a supervised training or eval process."""
 243   __metaclass__ = abc.ABCMeta
 244 
 245   LOSS_NAME = 'loss'
 246   PREDICTIONS_NAME = 'predictions'
 247   METRICS_NAME = 'metrics'
 248 
 249   METRIC_VALUE_SUFFIX = 'value'
 250   METRIC_UPDATE_SUFFIX = 'update_op'
 251 
 252   _loss = None
 253   _predictions = None
 254   _metrics = None
 255 
 256   def __init__(self, loss=None, predictions=None, metrics=None):
 257     """Constructor for SupervisedOutput (ie, Train or Eval output).
 258 
 259     Args:
 260       loss: dict of Tensors or single Tensor representing calculated loss.
 261       predictions: dict of Tensors or single Tensor representing model
 262         predictions.
 263       metrics: Dict of metric results keyed by name.
 264         The values of the dict can be one of the following:
 265         (1) instance of `Metric` class.
 266         (2) (metric_value, update_op) tuples, or a single tuple.
 267         metric_value must be a Tensor, and update_op must be a Tensor or Op.
 268 
 269     Raises:
 270       ValueError: if any of the outputs' dict keys are not strings or tuples of
 271         strings or the values are not Tensors (or Operations in the case of
 272         update_op).
 273     """
 274 
 275     if loss is not None:
 276       loss_dict = self._wrap_and_check_outputs(loss, self.LOSS_NAME)
 277       self._loss = self._prefix_output_keys(loss_dict, self.LOSS_NAME)
 278     if predictions is not None:
 279       pred_dict = self._wrap_and_check_outputs(
 280           predictions, self.PREDICTIONS_NAME)
 281       self._predictions = self._prefix_output_keys(
 282           pred_dict, self.PREDICTIONS_NAME)
 283     if metrics is not None:
 284       self._metrics = self._wrap_and_check_metrics(metrics)
 285 
 286   def _prefix_output_keys(self, output_dict, output_name):
 287     """Prepend output_name to the output_dict keys if it doesn't exist.
 288 
 289     This produces predictable prefixes for the pre-determined outputs
 290     of SupervisedOutput.
 291 
 292     Args:
 293       output_dict: dict of string to Tensor, assumed valid.
 294       output_name: prefix string to prepend to existing keys.
 295 
 296     Returns:
 297       dict with updated keys and existing values.
 298     """
 299 
 300     new_outputs = {}
 301     for key, val in output_dict.items():
 302       key = self._prefix_key(key, output_name)
 303       new_outputs[key] = val
 304     return new_outputs
 305 
 306   def _prefix_key(self, key, output_name):
 307     if key.find(output_name) != 0:
 308       key = output_name + self._SEPARATOR_CHAR + key
 309     return key
 310 
 311   def _wrap_and_check_metrics(self, metrics):
 312     """Handle the saving of metrics.
 313 
 314     Metrics is either a tuple of (value, update_op), or a dict of such tuples.
 315     Here, we separate out the tuples and create a dict with names to tensors.
 316 
 317     Args:
 318       metrics: Dict of metric results keyed by name.
 319         The values of the dict can be one of the following:
 320         (1) instance of `Metric` class.
 321         (2) (metric_value, update_op) tuples, or a single tuple.
 322         metric_value must be a Tensor, and update_op must be a Tensor or Op.
 323 
 324     Returns:
 325       dict of output_names to tensors
 326 
 327     Raises:
 328       ValueError: if the dict key is not a string, or the metric values or ops
 329         are not tensors.
 330     """
 331     if not isinstance(metrics, dict):
 332       metrics = {self.METRICS_NAME: metrics}
 333 
 334     outputs = {}
 335     for key, value in metrics.items():
 336       if isinstance(value, metrics_module.Metric):
 337         metric_val = value.result()
 338         assert len(value.updates) == 1  # We expect only one update op.
 339         metric_op = value.updates[0]
 340       else:
 341         metric_val, metric_op = value
 342       key = self._check_output_key(key, self.METRICS_NAME)
 343       key = self._prefix_key(key, self.METRICS_NAME)
 344 
 345       val_name = key + self._SEPARATOR_CHAR + self.METRIC_VALUE_SUFFIX
 346       op_name = key + self._SEPARATOR_CHAR + self.METRIC_UPDATE_SUFFIX
 347       if not isinstance(metric_val, ops.Tensor):
 348         raise ValueError(
 349             '{} output value must be a Tensor; got {}.'.format(
 350                 key, metric_val))
 351       if (not isinstance(metric_op, ops.Tensor) and
 352           not isinstance(metric_op, ops.Operation)):
 353         raise ValueError(
 354             '{} update_op must be a Tensor or Operation; got {}.'.format(
 355                 key, metric_op))
 356 
 357       # We must wrap any ops in a Tensor before export, as the SignatureDef
 358       # proto expects tensors only. See b/109740581
 359       metric_op_tensor = metric_op
 360       if isinstance(metric_op, ops.Operation):
 361         with ops.control_dependencies([metric_op]):
 362           metric_op_tensor = constant_op.constant([], name='metric_op_wrapper')
 363 
 364       outputs[val_name] = metric_val
 365       outputs[op_name] = metric_op_tensor
 366 
 367     return outputs
 368 
 369   @property
 370   def loss(self):
 371     return self._loss
 372 
 373   @property
 374   def predictions(self):
 375     return self._predictions
 376 
 377   @property
 378   def metrics(self):
 379     return self._metrics
 380 
 381   @abc.abstractmethod
 382   def _get_signature_def_fn(self):
 383     """Returns a function that produces a SignatureDef given desired outputs."""
 384     pass
 385 
 386   def as_signature_def(self, receiver_tensors):
 387     signature_def_fn = self._get_signature_def_fn()
 388     return signature_def_fn(
 389         receiver_tensors, self.loss, self.predictions, self.metrics)
