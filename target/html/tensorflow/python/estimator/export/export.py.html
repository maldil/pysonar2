  51 def _wrap_and_check_input_tensors(tensors, field_name):
  52   """Ensure that tensors is a dict of str to Tensor mappings.
  53 
  54   Args:
  55     tensors: dict of str to Tensors, or a single Tensor.
  56     field_name: name of the member field of `ServingInputReceiver`
  57       whose value is being passed to `tensors`.
  58 
  59   Returns:
  60     dict of str to Tensors; this is the original dict if one was passed, or
  61     the original tensor wrapped in a dictionary.
  62 
  63   Raises:
  64     ValueError: if tensors is None, or has non-string keys,
  65       or non-Tensor values
  66   """
  67   if tensors is None:
  68     raise ValueError('{}s must be defined.'.format(field_name))
  69   if not isinstance(tensors, dict):
  70     tensors = {_SINGLE_TENSOR_DEFAULT_NAMES[field_name]: tensors}
  71   for name, tensor in tensors.items():
  72     _check_tensor_key(name, error_label=field_name)
  73     _check_tensor(tensor, name, error_label=field_name)
  74   return tensors
  75

  77 def _check_tensor(tensor, name, error_label='feature'):
  78   """Check that passed `tensor` is a Tensor or SparseTensor."""
  79   if not (isinstance(tensor, ops.Tensor) or
  80           isinstance(tensor, sparse_tensor.SparseTensor)):
  81     fmt_name = ' {}'.format(name) if name else ''
  82     value_error = ValueError('{}{} must be a Tensor or SparseTensor.'.format(
  83         error_label, fmt_name))
  84     # NOTE(ericmc): This if-else block is a specific carve-out for
  85     # LabeledTensor, which has a `.tensor` attribute and which is
  86     # convertible to tf.Tensor via ops.convert_to_tensor.
  87     # Allowing all types convertible to tf.Tensor is considered by soergel@
  88     # to be too permissive.
  89     # TODO(soergel): accept any type convertible to Tensor,
  90     # as in cl/193238295 snapshot #6.
  91     if hasattr(tensor, 'tensor'):
  92       try:
  93         ops.convert_to_tensor(tensor)
  94       except TypeError:
  95         raise value_error
  96     else:
  97       raise value_error

 100 def _check_tensor_key(name, error_label='feature'):
 101   if not isinstance(name, six.string_types):
 102     raise ValueError('{} keys must be strings: {}.'.format(error_label, name))

 106 class ServingInputReceiver(
 107     collections.namedtuple(
 108         'ServingInputReceiver',
 109         ['features', 'receiver_tensors', 'receiver_tensors_alternatives'])):
 110   """A return type for a serving_input_receiver_fn.
 111 
 112   The expected return values are:
 113     features: A `Tensor`, `SparseTensor`, or dict of string to `Tensor` or
 114       `SparseTensor`, specifying the features to be passed to the model. Note:
 115       if `features` passed is not a dict, it will be wrapped in a dict with a
 116       single entry, using 'feature' as the key.  Consequently, the model must
 117       accept a feature dict of the form {'feature': tensor}.  You may use
 118       `TensorServingInputReceiver` if you want the tensor to be passed as is.
 119     receiver_tensors: A `Tensor`, `SparseTensor`, or dict of string to `Tensor`
 120       or `SparseTensor`, specifying input nodes where this receiver expects to
 121       be fed by default.  Typically, this is a single placeholder expecting
 122       serialized `tf.Example` protos.
 123     receiver_tensors_alternatives: a dict of string to additional
 124       groups of receiver tensors, each of which may be a `Tensor`,
 125       `SparseTensor`, or dict of string to `Tensor` or`SparseTensor`.
 126       These named receiver tensor alternatives generate additional serving
 127       signatures, which may be used to feed inputs at different points within
 128       the input receiver subgraph.  A typical usage is to allow feeding raw
 129       feature `Tensor`s *downstream* of the tf.parse_example() op.
 130       Defaults to None.
 131   """
 132 
 133   def __new__(cls,
 134               features,
 135               receiver_tensors,
 136               receiver_tensors_alternatives=None):
 137     features = _wrap_and_check_input_tensors(features, 'feature')
 138 
 139     receiver_tensors = _wrap_and_check_input_tensors(receiver_tensors,
 140                                                      'receiver_tensor')
 141 
 142     if receiver_tensors_alternatives is not None:
 143       if not isinstance(receiver_tensors_alternatives, dict):
 144         raise ValueError(
 145             'receiver_tensors_alternatives must be a dict: {}.'.format(
 146                 receiver_tensors_alternatives))
 147       for alternative_name, receiver_tensors_alt in (
 148           six.iteritems(receiver_tensors_alternatives)):
 149         # Updating dict during iteration is OK in this case.
 150         receiver_tensors_alternatives[alternative_name] = (
 151             _wrap_and_check_input_tensors(
 152                 receiver_tensors_alt, 'receiver_tensors_alternative'))
 153 
 154     return super(ServingInputReceiver, cls).__new__(
 155         cls,
 156         features=features,
 157         receiver_tensors=receiver_tensors,
 158         receiver_tensors_alternatives=receiver_tensors_alternatives)

 161 @estimator_export('estimator.export.TensorServingInputReceiver')
 162 class TensorServingInputReceiver(
 163     collections.namedtuple(
 164         'TensorServingInputReceiver',
 165         ['features', 'receiver_tensors', 'receiver_tensors_alternatives'])):
 166   """A return type for a serving_input_receiver_fn.
 167 
 168   This is for use with models that expect a single `Tensor` or `SparseTensor`
 169   as an input feature, as opposed to a dict of features.
 170 
 171   The normal `ServingInputReceiver` always returns a feature dict, even if it
 172   contains only one entry, and so can be used only with models that accept such
 173   a dict.  For models that accept only a single raw feature, the
 174   `serving_input_receiver_fn` provided to `Estimator.export_savedmodel()` should
 175   return this `TensorServingInputReceiver` instead.  See:
 176   https://github.com/tensorflow/tensorflow/issues/11674
 177 
 178   Note that the receiver_tensors and receiver_tensor_alternatives arguments
 179   will be automatically converted to the dict representation in either case,
 180   because the SavedModel format requires each input `Tensor` to have a name
 181   (provided by the dict key).
 182 
 183   The expected return values are:
 184     features: A single `Tensor` or `SparseTensor`, representing the feature
 185       to be passed to the model.
 186     receiver_tensors: A `Tensor`, `SparseTensor`, or dict of string to `Tensor`
 187       or `SparseTensor`, specifying input nodes where this receiver expects to
 188       be fed by default.  Typically, this is a single placeholder expecting
 189       serialized `tf.Example` protos.
 190     receiver_tensors_alternatives: a dict of string to additional
 191       groups of receiver tensors, each of which may be a `Tensor`,
 192       `SparseTensor`, or dict of string to `Tensor` or`SparseTensor`.
 193       These named receiver tensor alternatives generate additional serving
 194       signatures, which may be used to feed inputs at different points within
 195       the input receiver subgraph.  A typical usage is to allow feeding raw
 196       feature `Tensor`s *downstream* of the tf.parse_example() op.
 197       Defaults to None.
 198   """
 199 
 200   def __new__(cls,
 201               features,
 202               receiver_tensors,
 203               receiver_tensors_alternatives=None):
 204     if features is None:
 205       raise ValueError('features must be defined.')
 206     _check_tensor(features, None)
 207 
 208     receiver = ServingInputReceiver(
 209         features=features,
 210         receiver_tensors=receiver_tensors,
 211         receiver_tensors_alternatives=receiver_tensors_alternatives)
 212 
 213     return super(TensorServingInputReceiver, cls).__new__(
 214         cls,
 215         features=receiver.features[_SINGLE_FEATURE_DEFAULT_NAME],
 216         receiver_tensors=receiver.receiver_tensors,
 217         receiver_tensors_alternatives=receiver.receiver_tensors_alternatives)

 243 class SupervisedInputReceiver(
 244     collections.namedtuple('SupervisedInputReceiver',
 245                            ['features', 'labels', 'receiver_tensors'])):
 246   """A return type for a training_input_receiver_fn or eval_input_receiver_fn.
 247 
 248   This differs from a ServingInputReceiver in that (1) this receiver expects
 249   a set of labels to be passed in with features, and (2) this receiver does
 250   not support receiver_tensors_alternatives, which are primarily used for
 251   serving.
 252 
 253   The expected return values are:
 254     features: A `Tensor`, `SparseTensor`, or dict of string to `Tensor` or
 255       `SparseTensor`, specifying the features to be passed to the model.
 256     labels: A `Tensor`, `SparseTensor`, or dict of string to `Tensor` or
 257       `SparseTensor`, specifying the labels to be passed to the model.
 258     receiver_tensors: A `Tensor`, `SparseTensor`, or dict of string to `Tensor`
 259       or `SparseTensor`, specifying input nodes where this receiver expects to
 260       be fed by default.  Typically, this is a single placeholder expecting
 261       serialized `tf.Example` protos.
 262 
 263   """
 264 
 265   def __new__(cls, features, labels, receiver_tensors):
 266     # Both features and labels can be dicts or raw tensors.
 267     for input_vals, error_label in ((features, 'feature'), (labels, 'label')):
 268       # _wrap_and_check_input_tensors is called here only to validate the
 269       # tensors. The wrapped dict that is returned is deliberately discarded.
 270       _wrap_and_check_input_tensors(input_vals, error_label)
 271 
 272     receiver_tensors = _wrap_and_check_input_tensors(receiver_tensors,
 273                                                      'receiver_tensor')
 274 
 275     return super(SupervisedInputReceiver, cls).__new__(
 276         cls,
 277         features=features,
 278         labels=labels,
 279         receiver_tensors=receiver_tensors)

 282 @estimator_export('estimator.export.build_parsing_serving_input_receiver_fn')
 283 def build_parsing_serving_input_receiver_fn(feature_spec,
 284                                             default_batch_size=None):
 285   """Build a serving_input_receiver_fn expecting fed tf.Examples.
 286 
 287   Creates a serving_input_receiver_fn that expects a serialized tf.Example fed
 288   into a string placeholder.  The function parses the tf.Example according to
 289   the provided feature_spec, and returns all parsed Tensors as features.
 290 
 291   Args:
 292     feature_spec: a dict of string to `VarLenFeature`/`FixedLenFeature`.
 293     default_batch_size: the number of query examples expected per batch.
 294         Leave unset for variable batch size (recommended).
 295 
 296   Returns:
 297     A serving_input_receiver_fn suitable for use in serving.
 298   """
 299 
 300   def serving_input_receiver_fn():
 301     """An input_fn that expects a serialized tf.Example."""
 302     serialized_tf_example = array_ops.placeholder(
 303         dtype=dtypes.string,
 304         shape=[default_batch_size],
 305         name='input_example_tensor')
 306     receiver_tensors = {'examples': serialized_tf_example}
 307     features = parsing_ops.parse_example(serialized_tf_example, feature_spec)
 308     return ServingInputReceiver(features, receiver_tensors)
 309 
 310   return serving_input_receiver_fn

 313 def _placeholder_from_tensor(t, default_batch_size=None):
 314   """Creates a placeholder that matches the dtype and shape of passed tensor.
 315 
 316   Args:
 317     t: Tensor or EagerTensor
 318     default_batch_size: the number of query examples expected per batch.
 319         Leave unset for variable batch size (recommended).
 320 
 321   Returns:
 322     Placeholder that matches the passed tensor.
 323   """
 324   batch_shape = tensor_shape.TensorShape([default_batch_size])
 325   shape = batch_shape.concatenate(t.get_shape()[1:])
 326 
 327   # Reuse the feature tensor's op name (t.op.name) for the placeholder,
 328   # excluding the index from the tensor's name (t.name):
 329   # t.name = "%s:%d" % (t.op.name, t._value_index)
 330   try:
 331     name = t.op.name
 332   except AttributeError:
 333     # In Eager mode, tensors don't have ops or names, and while they do have
 334     # IDs, those are not maintained across runs. The name here is used
 335     # primarily for debugging, and is not critical to the placeholder.
 336     # So, in order to make this Eager-compatible, continue with an empty
 337     # name if none is available.
 338     name = None
 339 
 340   return array_ops.placeholder(dtype=t.dtype, shape=shape, name=name)

 376 def build_raw_supervised_input_receiver_fn(features,
 377                                            labels,
 378                                            default_batch_size=None):
 379   """Build a supervised_input_receiver_fn for raw features and labels.
 380 
 381   This function wraps tensor placeholders in a supervised_receiver_fn
 382   with the expectation that the features and labels appear precisely as
 383   the model_fn expects them. Features and labels can therefore be dicts of
 384   tensors, or raw tensors.
 385 
 386   Args:
 387     features: a dict of string to `Tensor` or `Tensor`.
 388     labels: a dict of string to `Tensor` or `Tensor`.
 389     default_batch_size: the number of query examples expected per batch.
 390         Leave unset for variable batch size (recommended).
 391 
 392   Returns:
 393     A supervised_input_receiver_fn.
 394 
 395   Raises:
 396     ValueError: if features and labels have overlapping keys.
 397   """
 398   # Check for overlapping keys before beginning.
 399   try:
 400     feat_keys = features.keys()
 401   except AttributeError:
 402     feat_keys = [_SINGLE_RECEIVER_DEFAULT_NAME]
 403   try:
 404     label_keys = labels.keys()
 405   except AttributeError:
 406     label_keys = [_SINGLE_LABEL_DEFAULT_NAME]
 407 
 408   overlap_keys = set(feat_keys) & set(label_keys)
 409   if overlap_keys:
 410     raise ValueError('Features and labels must have distinct keys. '
 411                      'Found overlapping keys: {}'.format(overlap_keys))
 412 
 413   def supervised_input_receiver_fn():
 414     """A receiver_fn that expects pass-through features and labels."""
 415     if not isinstance(features, dict):
 416       features_cp = _placeholder_from_tensor(features, default_batch_size)
 417       receiver_features = {_SINGLE_RECEIVER_DEFAULT_NAME: features_cp}
 418     else:
 419       receiver_features = _placeholders_from_receiver_tensors_dict(
 420           features, default_batch_size)
 421       features_cp = receiver_features
 422 
 423     if not isinstance(labels, dict):
 424       labels_cp = _placeholder_from_tensor(labels, default_batch_size)
 425       receiver_labels = {_SINGLE_LABEL_DEFAULT_NAME: labels_cp}
 426     else:
 427       receiver_labels = _placeholders_from_receiver_tensors_dict(
 428           labels, default_batch_size)
 429       labels_cp = receiver_labels
 430 
 431     receiver_tensors = dict(receiver_features)
 432     receiver_tensors.update(receiver_labels)
 433     return SupervisedInputReceiver(features_cp, labels_cp, receiver_tensors)
 434 
 435   return supervised_input_receiver_fn

 438 def build_supervised_input_receiver_fn_from_input_fn(input_fn, **input_fn_args):
 439   """Get a function that returns a SupervisedInputReceiver matching an input_fn.
 440 
 441   Note that this function calls the input_fn in a local graph in order to
 442   extract features and labels. Placeholders are then created from those
 443   features and labels in the default graph.
 444 
 445   Args:
 446     input_fn: An Estimator input_fn, which is a function that returns one of:
 447 
 448       * A 'tf.data.Dataset' object: Outputs of `Dataset` object must be a
 449           tuple (features, labels) with same constraints as below.
 450       * A tuple (features, labels): Where `features` is a `Tensor` or a
 451         dictionary of string feature name to `Tensor` and `labels` is a
 452         `Tensor` or a dictionary of string label name to `Tensor`. Both
 453         `features` and `labels` are consumed by `model_fn`. They should
 454         satisfy the expectation of `model_fn` from inputs.
 455 
 456     **input_fn_args: set of kwargs to be passed to the input_fn. Note that
 457       these will not be checked or validated here, and any errors raised by
 458       the input_fn will be thrown to the top.
 459 
 460   Returns:
 461     A function taking no arguments that, when called, returns a
 462     SupervisedInputReceiver. This function can be passed in as part of the
 463     input_receiver_map when exporting SavedModels from Estimator with multiple
 464     modes.
 465   """
 466   # Wrap the input_fn call in a graph to prevent sullying the default namespace
 467   with ops.Graph().as_default():
 468     result = input_fn(**input_fn_args)
 469     features, labels, _ = util.parse_input_fn_result(result)
 470   # Placeholders are created back in the default graph.
 471   return build_raw_supervised_input_receiver_fn(features, labels)

 477 def build_all_signature_defs(receiver_tensors,
 478                              export_outputs,
 479                              receiver_tensors_alternatives=None,
 480                              serving_only=True):
 481   """Build `SignatureDef`s for all export outputs.
 482 
 483   Args:
 484     receiver_tensors: a `Tensor`, or a dict of string to `Tensor`, specifying
 485       input nodes where this receiver expects to be fed by default.  Typically,
 486       this is a single placeholder expecting serialized `tf.Example` protos.
 487     export_outputs: a dict of ExportOutput instances, each of which has
 488       an as_signature_def instance method that will be called to retrieve
 489       the signature_def for all export output tensors.
 490     receiver_tensors_alternatives: a dict of string to additional
 491       groups of receiver tensors, each of which may be a `Tensor` or a dict of
 492       string to `Tensor`.  These named receiver tensor alternatives generate
 493       additional serving signatures, which may be used to feed inputs at
 494       different points within the input receiver subgraph.  A typical usage is
 495       to allow feeding raw feature `Tensor`s *downstream* of the
 496       tf.parse_example() op.  Defaults to None.
 497     serving_only: boolean; if true, resulting signature defs will only include
 498       valid serving signatures. If false, all requested signatures will be
 499       returned.
 500 
 501   Returns:
 502     signature_def representing all passed args.
 503 
 504   Raises:
 505     ValueError: if export_outputs is not a dict
 506   """
 507   if not isinstance(receiver_tensors, dict):
 508     receiver_tensors = {_SINGLE_RECEIVER_DEFAULT_NAME: receiver_tensors}
 509   if export_outputs is None or not isinstance(export_outputs, dict):
 510     raise ValueError('export_outputs must be a dict and not'
 511                      '{}'.format(type(export_outputs)))
 512 
 513   signature_def_map = {}
 514   excluded_signatures = {}
 515   for output_key, export_output in export_outputs.items():
 516     signature_name = '{}'.format(output_key or 'None')
 517     try:
 518       signature = export_output.as_signature_def(receiver_tensors)
 519       signature_def_map[signature_name] = signature
 520     except ValueError as e:
 521       excluded_signatures[signature_name] = str(e)
 522 
 523   if receiver_tensors_alternatives:
 524     for receiver_name, receiver_tensors_alt in (
 525         six.iteritems(receiver_tensors_alternatives)):
 526       if not isinstance(receiver_tensors_alt, dict):
 527         receiver_tensors_alt = {
 528             _SINGLE_RECEIVER_DEFAULT_NAME: receiver_tensors_alt
 529         }
 530       for output_key, export_output in export_outputs.items():
 531         signature_name = '{}:{}'.format(receiver_name or 'None', output_key or
 532                                         'None')
 533         try:
 534           signature = export_output.as_signature_def(receiver_tensors_alt)
 535           signature_def_map[signature_name] = signature
 536         except ValueError as e:
 537           excluded_signatures[signature_name] = str(e)
 538 
 539   _log_signature_report(signature_def_map, excluded_signatures)
 540 
 541   # The above calls to export_output.as_signature_def should return only
 542   # valid signatures; if there is a validity problem, they raise a ValueError,
 543   # in which case we exclude that signature from signature_def_map above.
 544   # The is_valid_signature check ensures that the signatures produced are
 545   # valid for serving, and acts as an additional sanity check for export
 546   # signatures produced for serving. We skip this check for training and eval
 547   # signatures, which are not intended for serving.
 548   if serving_only:
 549     signature_def_map = {
 550         k: v
 551         for k, v in signature_def_map.items()
 552         if signature_def_utils.is_valid_signature(v)
 553     }
 554   return signature_def_map

 566 def _log_signature_report(signature_def_map, excluded_signatures):
 567   """Log a report of which signatures were produced."""
 568   sig_names_by_method_name = collections.defaultdict(list)
 569 
 570   # We'll collect whatever method_names are present, but also we want to make
 571   # sure to output a line for each of the three standard methods even if they
 572   # have no signatures.
 573   for method_name in _FRIENDLY_METHOD_NAMES:
 574     sig_names_by_method_name[method_name] = []
 575 
 576   for signature_name, sig in signature_def_map.items():
 577     sig_names_by_method_name[sig.method_name].append(signature_name)
 578 
 579   # TODO(b/67733540): consider printing the full signatures, not just names
 580   for method_name, sig_names in sig_names_by_method_name.items():
 581     if method_name in _FRIENDLY_METHOD_NAMES:
 582       method_name = _FRIENDLY_METHOD_NAMES[method_name]
 583     logging.info('Signatures INCLUDED in export for {}: {}'.format(
 584         method_name, sig_names if sig_names else 'None'))
 585 
 586   if excluded_signatures:
 587     logging.info('Signatures EXCLUDED from export because they cannot be '
 588                  'be served via TensorFlow Serving APIs:')
 589     for signature_name, message in excluded_signatures.items():
 590       logging.info('\'{}\' : {}'.format(signature_name, message))
 591 
 592   if not signature_def_map:
 593     logging.warn('Export includes no signatures!')
 594   elif (signature_constants.DEFAULT_SERVING_SIGNATURE_DEF_KEY not in
 595         signature_def_map):
 596     logging.warn('Export includes no default signature!')

 599 def get_timestamped_export_dir(export_dir_base):
 600   """Builds a path to a new subdirectory within the base directory.
 601 
 602   Each export is written into a new subdirectory named using the
 603   current time.  This guarantees monotonically increasing version
 604   numbers even across multiple runs of the pipeline.
 605   The timestamp used is the number of seconds since epoch UTC.
 606 
 607   Args:
 608     export_dir_base: A string containing a directory to write the exported
 609         graph and checkpoints.
 610   Returns:
 611     The full path of the new subdirectory (which is not actually created yet).
 612 
 613   Raises:
 614     RuntimeError: if repeated attempts fail to obtain a unique timestamped
 615       directory name.
 616   """
 617   return util.get_timestamped_dir(export_dir_base)

 620 def get_temp_export_dir(timestamped_export_dir):
 621   """Builds a directory name based on the argument but starting with 'temp-'.
 622 
 623   This relies on the fact that TensorFlow Serving ignores subdirectories of
 624   the base directory that can't be parsed as integers.
 625 
 626   Args:
 627     timestamped_export_dir: the name of the eventual export directory, e.g.
 628       /foo/bar/<timestamp>
 629 
 630   Returns:
 631     A sister directory prefixed with 'temp-', e.g. /foo/bar/temp-<timestamp>.
 632   """
 633   (dirname, basename) = os.path.split(timestamped_export_dir)
 634   temp_export_dir = os.path.join(
 635       compat.as_bytes(dirname), compat.as_bytes('temp-{}'.format(basename)))
 636   return temp_export_dir
