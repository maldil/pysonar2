  51 def _dnn_logit_fn_builder(units,
  52                           hidden_units,
  53                           feature_columns,
  54                           activation_fn,
  55                           dropout,
  56                           input_layer_partitioner,
  57                           batch_norm,
  58                           shared_state_manager=None):
  59   """Function builder for a dnn logit_fn.
  60 
  61   Args:
  62     units: An int indicating the dimension of the logit layer.  In the
  63       MultiHead case, this should be the sum of all component Heads' logit
  64       dimensions.
  65     hidden_units: Iterable of integer number of hidden units per layer.
  66     feature_columns: Iterable of `feature_column._FeatureColumn` model inputs.
  67     activation_fn: Activation function applied to each layer.
  68     dropout: When not `None`, the probability we will drop out a given
  69       coordinate.
  70     input_layer_partitioner: Partitioner for input layer.
  71     batch_norm: Whether to use batch normalization after each hidden layer.
  72     shared_state_manager: A SharedEmbeddingStateManager object to hold the
  73       shared state for SharedEmbeddingColumn's.
  74 
  75   Returns:
  76     A logit_fn (see below).
  77 
  78   Raises:
  79     ValueError: If units is not an int.
  80   """
  81   if not isinstance(units, int):
  82     raise ValueError('units must be an int.  Given type: {}'.format(
  83         type(units)))
  84 
  85   def dnn_logit_fn(features, mode):
  86     """Deep Neural Network logit_fn.
  87 
  88     Args:
  89       features: This is the first item returned from the `input_fn`
  90                 passed to `train`, `evaluate`, and `predict`. This should be a
  91                 single `Tensor` or `dict` of same.
  92       mode: Optional. Specifies if this training, evaluation or prediction. See
  93             `ModeKeys`.
  94 
  95     Returns:
  96       A `Tensor` representing the logits, or a list of `Tensor`'s representing
  97       multiple logits in the MultiHead case.
  98     """
  99     dnn_model = _DNNModel(
 100         units,
 101         hidden_units,
 102         feature_columns,
 103         activation_fn,
 104         dropout,
 105         input_layer_partitioner,
 106         batch_norm,
 107         shared_state_manager,
 108         name='dnn')
 109     return dnn_model(features, mode)
 110 
 111   return dnn_logit_fn

 119 class _DNNModel(training.Model):
 120   """A DNN Model."""
 121 
 122   def __init__(self,
 123                units,
 124                hidden_units,
 125                feature_columns,
 126                activation_fn,
 127                dropout,
 128                input_layer_partitioner,
 129                batch_norm,
 130                shared_state_manager,
 131                name=None,
 132                **kwargs):
 133     super(_DNNModel, self).__init__(name=name, **kwargs)
 134     if feature_column_v2.is_feature_column_v2(feature_columns):
 135       self._input_layer = feature_column_v2.FeatureLayer(
 136           feature_columns=feature_columns,
 137           name='input_layer',
 138           shared_state_manager=shared_state_manager)
 139     else:
 140       self._input_layer = feature_column.InputLayer(
 141           feature_columns=feature_columns,
 142           name='input_layer',
 143           create_scope_now=False)
 144 
 145     self._add_layer(self._input_layer, 'input_layer')
 146 
 147     self._dropout = dropout
 148     self._batch_norm = batch_norm
 149 
 150     self._hidden_layers = []
 151     self._dropout_layers = []
 152     self._batch_norm_layers = []
 153     self._hidden_layer_scope_names = []
 154     for layer_id, num_hidden_units in enumerate(hidden_units):
 155       with variable_scope.variable_scope(
 156           'hiddenlayer_%d' % layer_id) as hidden_layer_scope:
 157         hidden_layer = core_layers.Dense(
 158             units=num_hidden_units,
 159             activation=activation_fn,
 160             kernel_initializer=init_ops.glorot_uniform_initializer(),
 161             name=hidden_layer_scope,
 162             _scope=hidden_layer_scope)
 163         self._add_layer(hidden_layer, hidden_layer_scope.name)
 164         self._hidden_layer_scope_names.append(hidden_layer_scope.name)
 165         self._hidden_layers.append(hidden_layer)
 166         if self._dropout is not None:
 167           dropout_layer = core_layers.Dropout(rate=self._dropout)
 168           self._add_layer(dropout_layer, dropout_layer.name)
 169           self._dropout_layers.append(dropout_layer)
 170         if self._batch_norm:
 171           batch_norm_layer = normalization.BatchNormalization(
 172               # The default momentum 0.99 actually crashes on certain
 173               # problem, so here we use 0.999, which is the default of
 174               # tf.contrib.layers.batch_norm.
 175               momentum=0.999,
 176               trainable=True,
 177               name='batchnorm_%d' % layer_id,
 178               _scope='batchnorm_%d' % layer_id)
 179           self._add_layer(batch_norm_layer, batch_norm_layer.name)
 180           self._batch_norm_layers.append(batch_norm_layer)
 181 
 182     with variable_scope.variable_scope('logits') as logits_scope:
 183       self._logits_layer = core_layers.Dense(
 184           units=units,
 185           activation=None,
 186           kernel_initializer=init_ops.glorot_uniform_initializer(),
 187           name=logits_scope,
 188           _scope=logits_scope)
 189       self._add_layer(self._logits_layer, logits_scope.name)
 190       self._logits_scope_name = logits_scope.name
 191     self._input_layer_partitioner = input_layer_partitioner
 192 
 193   def call(self, features, mode):
 194     is_training = mode == model_fn.ModeKeys.TRAIN
 195     # The Keras training.Model adds a name_scope with the name of the model
 196     # which modifies the constructed graph. Hence we add another name_scope
 197     # here which is the one before the training.Model one was applied.
 198     # TODO(rohanj): Remove this in TF 2.0 (b/116728605)
 199     with ops.name_scope(name=_get_previous_name_scope()):
 200       # TODO(rohanj): Remove dependence on variable scope for partitioning.
 201       with variable_scope.variable_scope(
 202           'input_from_feature_columns',
 203           partitioner=self._input_layer_partitioner):
 204         net = self._input_layer(features)
 205       for i in range(len(self._hidden_layers)):
 206         net = self._hidden_layers[i](net)
 207         if self._dropout is not None and is_training:
 208           net = self._dropout_layers[i](net, training=True)
 209         if self._batch_norm:
 210           net = self._batch_norm_layers[i](net, training=is_training)
 211         _add_hidden_layer_summary(net, self._hidden_layer_scope_names[i])
 212 
 213       logits = self._logits_layer(net)
 214       _add_hidden_layer_summary(logits, self._logits_scope_name)
 215       return logits
 216 
 217   def _add_layer(self, layer, layer_name):
 218     # "Magic" required for keras.Model classes to track all the variables in
 219     # a list of layers.Layer objects.
 220     # TODO(ashankar): Figure out API so user code doesn't have to do this.
 221     setattr(self, layer_name, layer)

 224 def _dnn_model_fn(features,
 225                   labels,
 226                   mode,
 227                   head,
 228                   hidden_units,
 229                   feature_columns,
 230                   optimizer='Adagrad',
 231                   activation_fn=nn.relu,
 232                   dropout=None,
 233                   input_layer_partitioner=None,
 234                   config=None,
 235                   use_tpu=False,
 236                   batch_norm=False,
 237                   shared_state_manager=None):
 238   """Deep Neural Net model_fn.
 239 
 240   Args:
 241     features: dict of `Tensor`.
 242     labels: `Tensor` of shape [batch_size, 1] or [batch_size] labels of
 243       dtype `int32` or `int64` in the range `[0, n_classes)`.
 244     mode: Defines whether this is training, evaluation or prediction.
 245       See `ModeKeys`.
 246     head: A `head_lib._Head` instance.
 247     hidden_units: Iterable of integer number of hidden units per layer.
 248     feature_columns: Iterable of `feature_column._FeatureColumn` model inputs.
 249     optimizer: String, `tf.Optimizer` object, or callable that creates the
 250       optimizer to use for training. If not specified, will use the Adagrad
 251       optimizer with a default learning rate of 0.05.
 252     activation_fn: Activation function applied to each layer.
 253     dropout: When not `None`, the probability we will drop out a given
 254       coordinate.
 255     input_layer_partitioner: Partitioner for input layer. Defaults
 256       to `min_max_variable_partitioner` with `min_slice_size` 64 << 20.
 257     config: `RunConfig` object to configure the runtime settings.
 258     use_tpu: Whether to make a DNN model able to run on TPU. Will make function
 259       return a `_TPUEstimatorSpec` instance and disable variable partitioning.
 260     batch_norm: Whether to use batch normalization after each hidden layer.
 261     shared_state_manager: A SharedEmbeddingStateManager object to hold the
 262       shared state for SharedEmbeddingColumn's.
 263 
 264   Returns:
 265     An `EstimatorSpec` instance.
 266 
 267   Raises:
 268     ValueError: If features has the wrong type.
 269   """
 270   if not isinstance(features, dict):
 271     raise ValueError('features should be a dictionary of `Tensor`s. '
 272                      'Given type: {}'.format(type(features)))
 273 
 274   optimizer = optimizers.get_optimizer_instance(
 275       optimizer, learning_rate=_LEARNING_RATE)
 276   num_ps_replicas = config.num_ps_replicas if config else 0
 277 
 278   partitioner = (None if use_tpu else
 279                  partitioned_variables.min_max_variable_partitioner(
 280                      max_partitions=num_ps_replicas))
 281   with variable_scope.variable_scope(
 282       'dnn',
 283       values=tuple(six.itervalues(features)),
 284       partitioner=partitioner):
 285     input_layer_partitioner = input_layer_partitioner or (
 286         None if use_tpu else
 287         partitioned_variables.min_max_variable_partitioner(
 288             max_partitions=num_ps_replicas,
 289             min_slice_size=64 << 20))
 290 
 291     logit_fn = _dnn_logit_fn_builder(
 292         units=head.logits_dimension,
 293         hidden_units=hidden_units,
 294         feature_columns=feature_columns,
 295         activation_fn=activation_fn,
 296         dropout=dropout,
 297         input_layer_partitioner=input_layer_partitioner,
 298         batch_norm=batch_norm,
 299         shared_state_manager=shared_state_manager)
 300     logits = logit_fn(features=features, mode=mode)
 301 
 302     if use_tpu:
 303       return head._create_tpu_estimator_spec(  # pylint: disable=protected-access
 304           features=features,
 305           mode=mode,
 306           labels=labels,
 307           optimizer=optimizer,
 308           logits=logits)
 309     else:
 310       return head.create_estimator_spec(
 311           features=features,
 312           mode=mode,
 313           labels=labels,
 314           optimizer=optimizer,
 315           logits=logits)

 318 @estimator_export('estimator.DNNClassifier')
 319 class DNNClassifier(estimator.Estimator):
 320   """A classifier for TensorFlow DNN models.
 321 
 322   Example:
 323 
 324   ```python
 325   categorical_feature_a = categorical_column_with_hash_bucket(...)
 326   categorical_feature_b = categorical_column_with_hash_bucket(...)
 327 
 328   categorical_feature_a_emb = embedding_column(
 329       categorical_column=categorical_feature_a, ...)
 330   categorical_feature_b_emb = embedding_column(
 331       categorical_column=categorical_feature_b, ...)
 332 
 333   estimator = DNNClassifier(
 334       feature_columns=[categorical_feature_a_emb, categorical_feature_b_emb],
 335       hidden_units=[1024, 512, 256])
 336 
 337   # Or estimator using the ProximalAdagradOptimizer optimizer with
 338   # regularization.
 339   estimator = DNNClassifier(
 340       feature_columns=[categorical_feature_a_emb, categorical_feature_b_emb],
 341       hidden_units=[1024, 512, 256],
 342       optimizer=tf.train.ProximalAdagradOptimizer(
 343         learning_rate=0.1,
 344         l1_regularization_strength=0.001
 345       ))
 346 
 347   # Or estimator using an optimizer with a learning rate decay.
 348   estimator = DNNClassifier(
 349       feature_columns=[categorical_feature_a_emb, categorical_feature_b_emb],
 350       hidden_units=[1024, 512, 256],
 351       optimizer=lambda: tf.AdamOptimizer(
 352           learning_rate=tf.exponential_decay(
 353               learning_rate=0.1,
 354               global_step=tf.get_global_step(),
 355               decay_steps=10000,
 356               decay_rate=0.96))
 357 
 358   # Or estimator with warm-starting from a previous checkpoint.
 359   estimator = DNNClassifier(
 360       feature_columns=[categorical_feature_a_emb, categorical_feature_b_emb],
 361       hidden_units=[1024, 512, 256],
 362       warm_start_from="/path/to/checkpoint/dir")
 363 
 364   # Input builders
 365   def input_fn_train: # returns x, y
 366     pass
 367   estimator.train(input_fn=input_fn_train, steps=100)
 368 
 369   def input_fn_eval: # returns x, y
 370     pass
 371   metrics = estimator.evaluate(input_fn=input_fn_eval, steps=10)
 372   def input_fn_predict: # returns x, None
 373     pass
 374   predictions = estimator.predict(input_fn=input_fn_predict)
 375   ```
 376 
 377   Input of `train` and `evaluate` should have following features,
 378   otherwise there will be a `KeyError`:
 379 
 380   * if `weight_column` is not `None`, a feature with
 381     `key=weight_column` whose value is a `Tensor`.
 382   * for each `column` in `feature_columns`:
 383     - if `column` is a `_CategoricalColumn`, a feature with `key=column.name`
 384       whose `value` is a `SparseTensor`.
 385     - if `column` is a `_WeightedCategoricalColumn`, two features: the first
 386       with `key` the id column name, the second with `key` the weight column
 387       name. Both features' `value` must be a `SparseTensor`.
 388     - if `column` is a `_DenseColumn`, a feature with `key=column.name`
 389       whose `value` is a `Tensor`.
 390 
 391   Loss is calculated by using softmax cross entropy.
 392 
 393   @compatibility(eager)
 394   Estimators can be used while eager execution is enabled. Note that `input_fn`
 395   and all hooks are executed inside a graph context, so they have to be written
 396   to be compatible with graph mode. Note that `input_fn` code using `tf.data`
 397   generally works in both graph and eager modes.
 398   @end_compatibility
 399   """
 400 
 401   def __init__(
 402       self,
 403       hidden_units,
 404       feature_columns,
 405       model_dir=None,
 406       n_classes=2,
 407       weight_column=None,
 408       label_vocabulary=None,
 409       optimizer='Adagrad',
 410       activation_fn=nn.relu,
 411       dropout=None,
 412       input_layer_partitioner=None,
 413       config=None,
 414       warm_start_from=None,
 415       loss_reduction=losses.Reduction.SUM,
 416       batch_norm=False,
 417   ):
 418     """Initializes a `DNNClassifier` instance.
 419 
 420     Args:
 421       hidden_units: Iterable of number hidden units per layer. All layers are
 422         fully connected. Ex. `[64, 32]` means first layer has 64 nodes and
 423         second one has 32.
 424       feature_columns: An iterable containing all the feature columns used by
 425         the model. All items in the set should be instances of classes derived
 426         from `_FeatureColumn`.
 427       model_dir: Directory to save model parameters, graph and etc. This can
 428         also be used to load checkpoints from the directory into a estimator to
 429         continue training a previously saved model.
 430       n_classes: Number of label classes. Defaults to 2, namely binary
 431         classification. Must be > 1.
 432       weight_column: A string or a `_NumericColumn` created by
 433         `tf.feature_column.numeric_column` defining feature column representing
 434         weights. It is used to down weight or boost examples during training. It
 435         will be multiplied by the loss of the example. If it is a string, it is
 436         used as a key to fetch weight tensor from the `features`. If it is a
 437         `_NumericColumn`, raw tensor is fetched by key `weight_column.key`,
 438         then weight_column.normalizer_fn is applied on it to get weight tensor.
 439       label_vocabulary: A list of strings represents possible label values. If
 440         given, labels must be string type and have any value in
 441         `label_vocabulary`. If it is not given, that means labels are
 442         already encoded as integer or float within [0, 1] for `n_classes=2` and
 443         encoded as integer values in {0, 1,..., n_classes-1} for `n_classes`>2 .
 444         Also there will be errors if vocabulary is not provided and labels are
 445         string.
 446       optimizer: An instance of `tf.Optimizer` used to train the model. Can also
 447         be a string (one of 'Adagrad', 'Adam', 'Ftrl', 'RMSProp', 'SGD'), or
 448         callable. Defaults to Adagrad optimizer.
 449       activation_fn: Activation function applied to each layer. If `None`, will
 450         use `tf.nn.relu`.
 451       dropout: When not `None`, the probability we will drop out a given
 452         coordinate.
 453       input_layer_partitioner: Optional. Partitioner for input layer. Defaults
 454         to `min_max_variable_partitioner` with `min_slice_size` 64 << 20.
 455       config: `RunConfig` object to configure the runtime settings.
 456       warm_start_from: A string filepath to a checkpoint to warm-start from, or
 457         a `WarmStartSettings` object to fully configure warm-starting.  If the
 458         string filepath is provided instead of a `WarmStartSettings`, then all
 459         weights are warm-started, and it is assumed that vocabularies and Tensor
 460         names are unchanged.
 461       loss_reduction: One of `tf.losses.Reduction` except `NONE`. Describes how
 462         to reduce training loss over batch. Defaults to `SUM`.
 463       batch_norm: Whether to use batch normalization after each hidden layer.
 464     """
 465     head = head_lib._binary_logistic_or_multi_class_head(  # pylint: disable=protected-access
 466         n_classes, weight_column, label_vocabulary, loss_reduction)
 467 
 468     shared_state_manager = feature_column_v2.maybe_create_shared_state_manager(
 469         feature_columns)
 470 
 471     def _model_fn(features, labels, mode, config):
 472       """Call the defined shared _dnn_model_fn."""
 473       return _dnn_model_fn(
 474           features=features,
 475           labels=labels,
 476           mode=mode,
 477           head=head,
 478           hidden_units=hidden_units,
 479           feature_columns=tuple(feature_columns or []),
 480           optimizer=optimizer,
 481           activation_fn=activation_fn,
 482           dropout=dropout,
 483           input_layer_partitioner=input_layer_partitioner,
 484           config=config,
 485           batch_norm=batch_norm,
 486           shared_state_manager=shared_state_manager)
 487 
 488     super(DNNClassifier, self).__init__(
 489         model_fn=_model_fn, model_dir=model_dir, config=config,
 490         warm_start_from=warm_start_from)
