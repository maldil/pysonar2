  29 @estimator_export('estimator.classifier_parse_example_spec')
  30 def classifier_parse_example_spec(feature_columns,
  31                                   label_key,
  32                                   label_dtype=dtypes.int64,
  33                                   label_default=None,
  34                                   weight_column=None):
  35   """Generates parsing spec for tf.parse_example to be used with classifiers.
  36 
  37   If users keep data in tf.Example format, they need to call tf.parse_example
  38   with a proper feature spec. There are two main things that this utility helps:
  39 
  40   * Users need to combine parsing spec of features with labels and weights
  41     (if any) since they are all parsed from same tf.Example instance. This
  42     utility combines these specs.
  43   * It is difficult to map expected label by a classifier such as
  44     `DNNClassifier` to corresponding tf.parse_example spec. This utility encodes
  45     it by getting related information from users (key, dtype).
  46 
  47   Example output of parsing spec:
  48 
  49   ```python
  50   # Define features and transformations
  51   feature_b = tf.feature_column.numeric_column(...)
  52   feature_c_bucketized = tf.feature_column.bucketized_column(
  53     tf.feature_column.numeric_column("feature_c"), ...)
  54   feature_a_x_feature_c = tf.feature_column.crossed_column(
  55       columns=["feature_a", feature_c_bucketized], ...)
  56 
  57   feature_columns = [feature_b, feature_c_bucketized, feature_a_x_feature_c]
  58   parsing_spec = tf.estimator.classifier_parse_example_spec(
  59       feature_columns, label_key='my-label', label_dtype=tf.string)
  60 
  61   # For the above example, classifier_parse_example_spec would return the dict:
  62   assert parsing_spec == {
  63     "feature_a": parsing_ops.VarLenFeature(tf.string),
  64     "feature_b": parsing_ops.FixedLenFeature([1], dtype=tf.float32),
  65     "feature_c": parsing_ops.FixedLenFeature([1], dtype=tf.float32)
  66     "my-label" : parsing_ops.FixedLenFeature([1], dtype=tf.string)
  67   }
  68   ```
  69 
  70   Example usage with a classifier:
  71 
  72   ```python
  73   feature_columns = # define features via tf.feature_column
  74   estimator = DNNClassifier(
  75       n_classes=1000,
  76       feature_columns=feature_columns,
  77       weight_column='example-weight',
  78       label_vocabulary=['photos', 'keep', ...],
  79       hidden_units=[256, 64, 16])
  80   # This label configuration tells the classifier the following:
  81   # * weights are retrieved with key 'example-weight'
  82   # * label is string and can be one of the following ['photos', 'keep', ...]
  83   # * integer id for label 'photos' is 0, 'keep' is 1, ...
  84 
  85 
  86   # Input builders
  87   def input_fn_train():  # Returns a tuple of features and labels.
  88     features = tf.contrib.learn.read_keyed_batch_features(
  89         file_pattern=train_files,
  90         batch_size=batch_size,
  91         # creates parsing configuration for tf.parse_example
  92         features=tf.estimator.classifier_parse_example_spec(
  93             feature_columns,
  94             label_key='my-label',
  95             label_dtype=tf.string,
  96             weight_column='example-weight'),
  97         reader=tf.RecordIOReader)
  98      labels = features.pop('my-label')
  99      return features, labels
 100 
 101   estimator.train(input_fn=input_fn_train)
 102   ```
 103 
 104   Args:
 105     feature_columns: An iterable containing all feature columns. All items
 106       should be instances of classes derived from `_FeatureColumn`.
 107     label_key: A string identifying the label. It means tf.Example stores labels
 108       with this key.
 109     label_dtype: A `tf.dtype` identifies the type of labels. By default it is
 110       `tf.int64`. If user defines a `label_vocabulary`, this should be set as
 111       `tf.string`. `tf.float32` labels are only supported for binary
 112       classification.
 113     label_default: used as label if label_key does not exist in given
 114       tf.Example. An example usage: let's say `label_key` is 'clicked' and
 115       tf.Example contains clicked data only for positive examples in following
 116       format `key:clicked, value:1`. This means that if there is no data with
 117       key 'clicked' it should count as negative example by setting
 118       `label_deafault=0`. Type of this value should be compatible with
 119       `label_dtype`.
 120     weight_column: A string or a `_NumericColumn` created by
 121       `tf.feature_column.numeric_column` defining feature column representing
 122       weights. It is used to down weight or boost examples during training. It
 123       will be multiplied by the loss of the example. If it is a string, it is
 124       used as a key to fetch weight tensor from the `features`. If it is a
 125       `_NumericColumn`, raw tensor is fetched by key `weight_column.key`,
 126       then weight_column.normalizer_fn is applied on it to get weight tensor.
 127 
 128   Returns:
 129     A dict mapping each feature key to a `FixedLenFeature` or `VarLenFeature`
 130     value.
 131 
 132   Raises:
 133     ValueError: If label is used in `feature_columns`.
 134     ValueError: If weight_column is used in `feature_columns`.
 135     ValueError: If any of the given `feature_columns` is not a `_FeatureColumn`
 136       instance.
 137     ValueError: If `weight_column` is not a `_NumericColumn` instance.
 138     ValueError: if label_key is None.
 139   """
 140   parsing_spec = fc.make_parse_example_spec(feature_columns)
 141   if label_key in parsing_spec:
 142     raise ValueError('label should not be used as feature. '
 143                      'label_key: {}, features: {}'.format(
 144                          label_key, parsing_spec.keys()))
 145   parsing_spec[label_key] = parsing_ops.FixedLenFeature((1,), label_dtype,
 146                                                         label_default)
 147 
 148   if weight_column is None:
 149     return parsing_spec
 150 
 151   if isinstance(weight_column, six.string_types):
 152     weight_column = fc.numeric_column(weight_column)
 153 
 154   if not isinstance(weight_column, fc._NumericColumn):  # pylint: disable=protected-access
 155     raise ValueError('weight_column should be an instance of '
 156                      'tf.feature_column.numeric_column. '
 157                      'Given type: {} value: {}'.format(
 158                          type(weight_column), weight_column))
 159 
 160   if weight_column.key in parsing_spec:
 161     raise ValueError('weight_column should not be used as feature. '
 162                      'weight_column: {}, features: {}'.format(
 163                          weight_column.key, parsing_spec.keys()))
 164 
 165   parsing_spec.update(weight_column._parse_example_spec)  # pylint: disable=protected-access
 166   return parsing_spec

 169 @estimator_export('estimator.regressor_parse_example_spec')
 170 def regressor_parse_example_spec(feature_columns,
 171                                  label_key,
 172                                  label_dtype=dtypes.float32,
 173                                  label_default=None,
 174                                  label_dimension=1,
 175                                  weight_column=None):
 176   """Generates parsing spec for tf.parse_example to be used with regressors.
 177 
 178   If users keep data in tf.Example format, they need to call tf.parse_example
 179   with a proper feature spec. There are two main things that this utility helps:
 180 
 181   * Users need to combine parsing spec of features with labels and weights
 182     (if any) since they are all parsed from same tf.Example instance. This
 183     utility combines these specs.
 184   * It is difficult to map expected label by a regressor such as `DNNRegressor`
 185     to corresponding tf.parse_example spec. This utility encodes it by getting
 186     related information from users (key, dtype).
 187 
 188   Example output of parsing spec:
 189 
 190   ```python
 191   # Define features and transformations
 192   feature_b = tf.feature_column.numeric_column(...)
 193   feature_c_bucketized = tf.feature_column.bucketized_column(
 194     tf.feature_column.numeric_column("feature_c"), ...)
 195   feature_a_x_feature_c = tf.feature_column.crossed_column(
 196       columns=["feature_a", feature_c_bucketized], ...)
 197 
 198   feature_columns = [feature_b, feature_c_bucketized, feature_a_x_feature_c]
 199   parsing_spec = tf.estimator.regressor_parse_example_spec(
 200       feature_columns, label_key='my-label')
 201 
 202   # For the above example, regressor_parse_example_spec would return the dict:
 203   assert parsing_spec == {
 204     "feature_a": parsing_ops.VarLenFeature(tf.string),
 205     "feature_b": parsing_ops.FixedLenFeature([1], dtype=tf.float32),
 206     "feature_c": parsing_ops.FixedLenFeature([1], dtype=tf.float32)
 207     "my-label" : parsing_ops.FixedLenFeature([1], dtype=tf.float32)
 208   }
 209   ```
 210 
 211   Example usage with a regressor:
 212 
 213   ```python
 214   feature_columns = # define features via tf.feature_column
 215   estimator = DNNRegressor(
 216       hidden_units=[256, 64, 16],
 217       feature_columns=feature_columns,
 218       weight_column='example-weight',
 219       label_dimension=3)
 220   # This label configuration tells the regressor the following:
 221   # * weights are retrieved with key 'example-weight'
 222   # * label is a 3 dimension tensor with float32 dtype.
 223 
 224 
 225   # Input builders
 226   def input_fn_train():  # Returns a tuple of features and labels.
 227     features = tf.contrib.learn.read_keyed_batch_features(
 228         file_pattern=train_files,
 229         batch_size=batch_size,
 230         # creates parsing configuration for tf.parse_example
 231         features=tf.estimator.classifier_parse_example_spec(
 232             feature_columns,
 233             label_key='my-label',
 234             label_dimension=3,
 235             weight_column='example-weight'),
 236         reader=tf.RecordIOReader)
 237      labels = features.pop('my-label')
 238      return features, labels
 239 
 240   estimator.train(input_fn=input_fn_train)
 241   ```
 242 
 243   Args:
 244     feature_columns: An iterable containing all feature columns. All items
 245       should be instances of classes derived from `_FeatureColumn`.
 246     label_key: A string identifying the label. It means tf.Example stores labels
 247       with this key.
 248     label_dtype: A `tf.dtype` identifies the type of labels. By default it is
 249       `tf.float32`.
 250     label_default: used as label if label_key does not exist in given
 251       tf.Example. By default default_value is none, which means
 252       `tf.parse_example` will error out if there is any missing label.
 253     label_dimension: Number of regression targets per example. This is the
 254       size of the last dimension of the labels and logits `Tensor` objects
 255       (typically, these have shape `[batch_size, label_dimension]`).
 256     weight_column: A string or a `_NumericColumn` created by
 257       `tf.feature_column.numeric_column` defining feature column representing
 258       weights. It is used to down weight or boost examples during training. It
 259       will be multiplied by the loss of the example. If it is a string, it is
 260       used as a key to fetch weight tensor from the `features`. If it is a
 261       `_NumericColumn`, raw tensor is fetched by key `weight_column.key`,
 262       then weight_column.normalizer_fn is applied on it to get weight tensor.
 263 
 264   Returns:
 265     A dict mapping each feature key to a `FixedLenFeature` or `VarLenFeature`
 266     value.
 267 
 268   Raises:
 269     ValueError: If label is used in `feature_columns`.
 270     ValueError: If weight_column is used in `feature_columns`.
 271     ValueError: If any of the given `feature_columns` is not a `_FeatureColumn`
 272       instance.
 273     ValueError: If `weight_column` is not a `_NumericColumn` instance.
 274     ValueError: if label_key is None.
 275   """
 276   parsing_spec = fc.make_parse_example_spec(feature_columns)
 277   if label_key in parsing_spec:
 278     raise ValueError('label should not be used as feature. '
 279                      'label_key: {}, features: {}'.format(
 280                          label_key, parsing_spec.keys()))
 281   parsing_spec[label_key] = parsing_ops.FixedLenFeature(
 282       (label_dimension,), label_dtype, label_default)
 283 
 284   if weight_column is None:
 285     return parsing_spec
 286 
 287   if isinstance(weight_column, six.string_types):
 288     weight_column = fc.numeric_column(weight_column)
 289 
 290   if not isinstance(weight_column, fc._NumericColumn):  # pylint: disable=protected-access
 291     raise ValueError('weight_column should be an instance of '
 292                      'tf.feature_column.numeric_column. '
 293                      'Given type: {} value: {}'.format(
 294                          type(weight_column), weight_column))
 295 
 296   if weight_column.key in parsing_spec:
 297     raise ValueError('weight_column should not be used as feature. '
 298                      'weight_column: {}, features: {}'.format(
 299                          weight_column.key, parsing_spec.keys()))
 300 
 301   parsing_spec.update(weight_column._parse_example_spec)  # pylint: disable=protected-access
 302   return parsing_spec
