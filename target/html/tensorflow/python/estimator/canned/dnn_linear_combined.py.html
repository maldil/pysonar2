  49 def _check_no_sync_replicas_optimizer(optimizer):
  50   if isinstance(optimizer, sync_replicas_optimizer.SyncReplicasOptimizer):
  51     raise ValueError(
  52         'SyncReplicasOptimizer does not support multi optimizers case. '
  53         'Therefore, it is not supported in DNNLinearCombined model. '
  54         'If you want to use this optimizer, please use either DNN or Linear '
  55         'model.')

  58 def _linear_learning_rate(num_linear_feature_columns):
  59   """Returns the default learning rate of the linear model.
  60 
  61   The calculation is a historical artifact of this initial implementation, but
  62   has proven a reasonable choice.
  63 
  64   Args:
  65     num_linear_feature_columns: The number of feature columns of the linear
  66       model.
  67 
  68   Returns:
  69     A float.
  70   """
  71   default_learning_rate = 1. / math.sqrt(num_linear_feature_columns)
  72   return min(_LINEAR_LEARNING_RATE, default_learning_rate)

  80 def _dnn_linear_combined_model_fn(features,
  81                                   labels,
  82                                   mode,
  83                                   head,
  84                                   linear_feature_columns=None,
  85                                   linear_optimizer='Ftrl',
  86                                   dnn_feature_columns=None,
  87                                   dnn_optimizer='Adagrad',
  88                                   dnn_hidden_units=None,
  89                                   dnn_activation_fn=nn.relu,
  90                                   dnn_dropout=None,
  91                                   input_layer_partitioner=None,
  92                                   config=None,
  93                                   batch_norm=False,
  94                                   linear_sparse_combiner='sum'):
  95   """Deep Neural Net and Linear combined model_fn.
  96 
  97   Args:
  98     features: dict of `Tensor`.
  99     labels: `Tensor` of shape [batch_size, 1] or [batch_size] labels of dtype
 100       `int32` or `int64` in the range `[0, n_classes)`.
 101     mode: Defines whether this is training, evaluation or prediction.
 102       See `ModeKeys`.
 103     head: A `Head` instance.
 104     linear_feature_columns: An iterable containing all the feature columns used
 105       by the Linear model.
 106     linear_optimizer: string, `Optimizer` object, or callable that defines the
 107       optimizer to use for training the Linear model. Defaults to the Ftrl
 108       optimizer.
 109     dnn_feature_columns: An iterable containing all the feature columns used by
 110       the DNN model.
 111     dnn_optimizer: string, `Optimizer` object, or callable that defines the
 112       optimizer to use for training the DNN model. Defaults to the Adagrad
 113       optimizer.
 114     dnn_hidden_units: List of hidden units per DNN layer.
 115     dnn_activation_fn: Activation function applied to each DNN layer. If `None`,
 116       will use `tf.nn.relu`.
 117     dnn_dropout: When not `None`, the probability we will drop out a given DNN
 118       coordinate.
 119     input_layer_partitioner: Partitioner for input layer.
 120     config: `RunConfig` object to configure the runtime settings.
 121     batch_norm: Whether to use batch normalization after each hidden layer.
 122     linear_sparse_combiner: A string specifying how to reduce the linear model
 123       if a categorical column is multivalent.  One of "mean", "sqrtn", and
 124       "sum".
 125   Returns:
 126     An `EstimatorSpec` instance.
 127 
 128   Raises:
 129     ValueError: If both `linear_feature_columns` and `dnn_features_columns`
 130       are empty at the same time, or `input_layer_partitioner` is missing,
 131       or features has the wrong type.
 132   """
 133   if not isinstance(features, dict):
 134     raise ValueError('features should be a dictionary of `Tensor`s. '
 135                      'Given type: {}'.format(type(features)))
 136   if not linear_feature_columns and not dnn_feature_columns:
 137     raise ValueError(
 138         'Either linear_feature_columns or dnn_feature_columns must be defined.')
 139 
 140   num_ps_replicas = config.num_ps_replicas if config else 0
 141   input_layer_partitioner = input_layer_partitioner or (
 142       partitioned_variables.min_max_variable_partitioner(
 143           max_partitions=num_ps_replicas,
 144           min_slice_size=64 << 20))
 145 
 146   shared_state_manager = feature_column_v2.maybe_create_shared_state_manager(
 147       list(linear_feature_columns) + list(dnn_feature_columns))
 148 
 149   # Build DNN Logits.
 150   dnn_parent_scope = 'dnn'
 151 
 152   if not dnn_feature_columns:
 153     dnn_logits = None
 154   else:
 155     dnn_optimizer = optimizers.get_optimizer_instance(
 156         dnn_optimizer, learning_rate=_DNN_LEARNING_RATE)
 157     _check_no_sync_replicas_optimizer(dnn_optimizer)
 158     if not dnn_hidden_units:
 159       raise ValueError(
 160           'dnn_hidden_units must be defined when dnn_feature_columns is '
 161           'specified.')
 162     dnn_partitioner = (
 163         partitioned_variables.min_max_variable_partitioner(
 164             max_partitions=num_ps_replicas))
 165     with variable_scope.variable_scope(
 166         dnn_parent_scope,
 167         values=tuple(six.itervalues(features)),
 168         partitioner=dnn_partitioner) as scope:
 169       dnn_absolute_scope = scope.name
 170       dnn_logit_fn = dnn._dnn_logit_fn_builder(  # pylint: disable=protected-access
 171           units=head.logits_dimension,
 172           hidden_units=dnn_hidden_units,
 173           feature_columns=dnn_feature_columns,
 174           activation_fn=dnn_activation_fn,
 175           dropout=dnn_dropout,
 176           batch_norm=batch_norm,
 177           input_layer_partitioner=input_layer_partitioner,
 178           shared_state_manager=shared_state_manager)
 179       dnn_logits = dnn_logit_fn(features=features, mode=mode)
 180 
 181   linear_parent_scope = 'linear'
 182 
 183   if not linear_feature_columns:
 184     linear_logits = None
 185   else:
 186     linear_optimizer = optimizers.get_optimizer_instance(
 187         linear_optimizer,
 188         learning_rate=_linear_learning_rate(len(linear_feature_columns)))
 189     _check_no_sync_replicas_optimizer(linear_optimizer)
 190     with variable_scope.variable_scope(
 191         linear_parent_scope,
 192         values=tuple(six.itervalues(features)),
 193         partitioner=input_layer_partitioner) as scope:
 194       linear_absolute_scope = scope.name
 195       logit_fn = linear._linear_logit_fn_builder(  # pylint: disable=protected-access
 196           units=head.logits_dimension,
 197           feature_columns=linear_feature_columns,
 198           sparse_combiner=linear_sparse_combiner)
 199       linear_logits = logit_fn(features=features)
 200       _add_layer_summary(linear_logits, scope.name)
 201 
 202   # Combine logits and build full model.
 203   if dnn_logits is not None and linear_logits is not None:
 204     logits = dnn_logits + linear_logits
 205   elif dnn_logits is not None:
 206     logits = dnn_logits
 207   else:
 208     logits = linear_logits
 209 
 210   def _train_op_fn(loss):
 211     """Returns the op to optimize the loss."""
 212     train_ops = []
 213     global_step = training_util.get_global_step()
 214     if dnn_logits is not None:
 215       train_ops.append(
 216           dnn_optimizer.minimize(
 217               loss,
 218               var_list=ops.get_collection(
 219                   ops.GraphKeys.TRAINABLE_VARIABLES,
 220                   scope=dnn_absolute_scope)))
 221     if linear_logits is not None:
 222       train_ops.append(
 223           linear_optimizer.minimize(
 224               loss,
 225               var_list=ops.get_collection(
 226                   ops.GraphKeys.TRAINABLE_VARIABLES,
 227                   scope=linear_absolute_scope)))
 228 
 229     train_op = control_flow_ops.group(*train_ops)
 230     with ops.control_dependencies([train_op]):
 231       return state_ops.assign_add(global_step, 1).op
 232 
 233   return head.create_estimator_spec(
 234       features=features,
 235       mode=mode,
 236       labels=labels,
 237       train_op_fn=_train_op_fn,
 238       logits=logits)

 241 @estimator_export('estimator.DNNLinearCombinedClassifier')
 242 class DNNLinearCombinedClassifier(estimator.Estimator):
 243   """An estimator for TensorFlow Linear and DNN joined classification models.
 244 
 245   Note: This estimator is also known as wide-n-deep.
 246 
 247   Example:
 248 
 249   ```python
 250   numeric_feature = numeric_column(...)
 251   categorical_column_a = categorical_column_with_hash_bucket(...)
 252   categorical_column_b = categorical_column_with_hash_bucket(...)
 253 
 254   categorical_feature_a_x_categorical_feature_b = crossed_column(...)
 255   categorical_feature_a_emb = embedding_column(
 256       categorical_column=categorical_feature_a, ...)
 257   categorical_feature_b_emb = embedding_column(
 258       categorical_id_column=categorical_feature_b, ...)
 259 
 260   estimator = DNNLinearCombinedClassifier(
 261       # wide settings
 262       linear_feature_columns=[categorical_feature_a_x_categorical_feature_b],
 263       linear_optimizer=tf.train.FtrlOptimizer(...),
 264       # deep settings
 265       dnn_feature_columns=[
 266           categorical_feature_a_emb, categorical_feature_b_emb,
 267           numeric_feature],
 268       dnn_hidden_units=[1000, 500, 100],
 269       dnn_optimizer=tf.train.ProximalAdagradOptimizer(...),
 270       # warm-start settings
 271       warm_start_from="/path/to/checkpoint/dir")
 272 
 273   # To apply L1 and L2 regularization, you can set dnn_optimizer to:
 274   tf.train.ProximalAdagradOptimizer(
 275       learning_rate=0.1,
 276       l1_regularization_strength=0.001,
 277       l2_regularization_strength=0.001)
 278   # To apply learning rate decay, you can set dnn_optimizer to a callable:
 279   lambda: tf.AdamOptimizer(
 280       learning_rate=tf.exponential_decay(
 281           learning_rate=0.1,
 282           global_step=tf.get_global_step(),
 283           decay_steps=10000,
 284           decay_rate=0.96)
 285   # It is the same for linear_optimizer.
 286 
 287   # Input builders
 288   def input_fn_train: # returns x, y
 289     pass
 290   estimator.train(input_fn=input_fn_train, steps=100)
 291 
 292   def input_fn_eval: # returns x, y
 293     pass
 294   metrics = estimator.evaluate(input_fn=input_fn_eval, steps=10)
 295   def input_fn_predict: # returns x, None
 296     pass
 297   predictions = estimator.predict(input_fn=input_fn_predict)
 298   ```
 299 
 300   Input of `train` and `evaluate` should have following features,
 301   otherwise there will be a `KeyError`:
 302 
 303   * for each `column` in `dnn_feature_columns` + `linear_feature_columns`:
 304     - if `column` is a `_CategoricalColumn`, a feature with `key=column.name`
 305       whose `value` is a `SparseTensor`.
 306     - if `column` is a `_WeightedCategoricalColumn`, two features: the first
 307       with `key` the id column name, the second with `key` the weight column
 308       name. Both features' `value` must be a `SparseTensor`.
 309     - if `column` is a `_DenseColumn`, a feature with `key=column.name`
 310       whose `value` is a `Tensor`.
 311 
 312   Loss is calculated by using softmax cross entropy.
 313 
 314   @compatibility(eager)
 315   Estimators can be used while eager execution is enabled. Note that `input_fn`
 316   and all hooks are executed inside a graph context, so they have to be written
 317   to be compatible with graph mode. Note that `input_fn` code using `tf.data`
 318   generally works in both graph and eager modes.
 319   @end_compatibility
 320   """
 321 
 322   def __init__(self,
 323                model_dir=None,
 324                linear_feature_columns=None,
 325                linear_optimizer='Ftrl',
 326                dnn_feature_columns=None,
 327                dnn_optimizer='Adagrad',
 328                dnn_hidden_units=None,
 329                dnn_activation_fn=nn.relu,
 330                dnn_dropout=None,
 331                n_classes=2,
 332                weight_column=None,
 333                label_vocabulary=None,
 334                input_layer_partitioner=None,
 335                config=None,
 336                warm_start_from=None,
 337                loss_reduction=losses.Reduction.SUM,
 338                batch_norm=False,
 339                linear_sparse_combiner='sum'):
 340     """Initializes a DNNLinearCombinedClassifier instance.
 341 
 342     Args:
 343       model_dir: Directory to save model parameters, graph and etc. This can
 344         also be used to load checkpoints from the directory into a estimator
 345         to continue training a previously saved model.
 346       linear_feature_columns: An iterable containing all the feature columns
 347         used by linear part of the model. All items in the set must be
 348         instances of classes derived from `FeatureColumn`.
 349       linear_optimizer: An instance of `tf.Optimizer` used to apply gradients to
 350         the linear part of the model. Can also be a string (one of 'Adagrad',
 351         'Adam', 'Ftrl', 'RMSProp', 'SGD'), or callable. Defaults to FTRL
 352         optimizer.
 353       dnn_feature_columns: An iterable containing all the feature columns used
 354         by deep part of the model. All items in the set must be instances of
 355         classes derived from `FeatureColumn`.
 356       dnn_optimizer: An instance of `tf.Optimizer` used to apply gradients to
 357         the deep part of the model. Can also be a string (one of 'Adagrad',
 358         'Adam', 'Ftrl', 'RMSProp', 'SGD'), or callable. Defaults to Adagrad
 359         optimizer.
 360       dnn_hidden_units: List of hidden units per layer. All layers are fully
 361         connected.
 362       dnn_activation_fn: Activation function applied to each layer. If None,
 363         will use `tf.nn.relu`.
 364       dnn_dropout: When not None, the probability we will drop out
 365         a given coordinate.
 366       n_classes: Number of label classes. Defaults to 2, namely binary
 367         classification. Must be > 1.
 368       weight_column: A string or a `_NumericColumn` created by
 369         `tf.feature_column.numeric_column` defining feature column representing
 370         weights. It is used to down weight or boost examples during training. It
 371         will be multiplied by the loss of the example. If it is a string, it is
 372         used as a key to fetch weight tensor from the `features`. If it is a
 373         `_NumericColumn`, raw tensor is fetched by key `weight_column.key`,
 374         then weight_column.normalizer_fn is applied on it to get weight tensor.
 375       label_vocabulary: A list of strings represents possible label values. If
 376         given, labels must be string type and have any value in
 377         `label_vocabulary`. If it is not given, that means labels are
 378         already encoded as integer or float within [0, 1] for `n_classes=2` and
 379         encoded as integer values in {0, 1,..., n_classes-1} for `n_classes`>2 .
 380         Also there will be errors if vocabulary is not provided and labels are
 381         string.
 382       input_layer_partitioner: Partitioner for input layer. Defaults to
 383         `min_max_variable_partitioner` with `min_slice_size` 64 << 20.
 384       config: RunConfig object to configure the runtime settings.
 385       warm_start_from: A string filepath to a checkpoint to warm-start from, or
 386         a `WarmStartSettings` object to fully configure warm-starting.  If the
 387         string filepath is provided instead of a `WarmStartSettings`, then all
 388         weights are warm-started, and it is assumed that vocabularies and Tensor
 389         names are unchanged.
 390       loss_reduction: One of `tf.losses.Reduction` except `NONE`. Describes how
 391         to reduce training loss over batch. Defaults to `SUM`.
 392       batch_norm: Whether to use batch normalization after each hidden layer.
 393       linear_sparse_combiner: A string specifying how to reduce the linear model
 394         if a categorical column is multivalent.  One of "mean", "sqrtn", and
 395         "sum" -- these are effectively different ways to do example-level
 396         normalization, which can be useful for bag-of-words features.  For more
 397         details, see `tf.feature_column.linear_model`.
 398 
 399     Raises:
 400       ValueError: If both linear_feature_columns and dnn_features_columns are
 401         empty at the same time.
 402     """
 403     linear_feature_columns = linear_feature_columns or []
 404     dnn_feature_columns = dnn_feature_columns or []
 405     self._feature_columns = (
 406         list(linear_feature_columns) + list(dnn_feature_columns))
 407     if not self._feature_columns:
 408       raise ValueError('Either linear_feature_columns or dnn_feature_columns '
 409                        'must be defined.')
 410     if n_classes == 2:
 411       head = head_lib._binary_logistic_head_with_sigmoid_cross_entropy_loss(  # pylint: disable=protected-access
 412           weight_column=weight_column,
 413           label_vocabulary=label_vocabulary,
 414           loss_reduction=loss_reduction)
 415     else:
 416       head = head_lib._multi_class_head_with_softmax_cross_entropy_loss(  # pylint: disable=protected-access
 417           n_classes,
 418           weight_column=weight_column,
 419           label_vocabulary=label_vocabulary,
 420           loss_reduction=loss_reduction)
 421 
 422     def _model_fn(features, labels, mode, config):
 423       """Call the _dnn_linear_combined_model_fn."""
 424       return _dnn_linear_combined_model_fn(
 425           features=features,
 426           labels=labels,
 427           mode=mode,
 428           head=head,
 429           linear_feature_columns=linear_feature_columns,
 430           linear_optimizer=linear_optimizer,
 431           dnn_feature_columns=dnn_feature_columns,
 432           dnn_optimizer=dnn_optimizer,
 433           dnn_hidden_units=dnn_hidden_units,
 434           dnn_activation_fn=dnn_activation_fn,
 435           dnn_dropout=dnn_dropout,
 436           input_layer_partitioner=input_layer_partitioner,
 437           config=config,
 438           batch_norm=batch_norm,
 439           linear_sparse_combiner=linear_sparse_combiner)
 440 
 441     super(DNNLinearCombinedClassifier, self).__init__(
 442         model_fn=_model_fn, model_dir=model_dir, config=config,
 443         warm_start_from=warm_start_from)

 561         the deep part of the model. Can also be a string (one of 'Adagrad',
 562         'Adam', 'Ftrl', 'RMSProp', 'SGD'), or callable. Defaults to Adagrad
 563         optimizer.
 564       dnn_hidden_units: List of hidden units per layer. All layers are fully
 565         connected.
 566       dnn_activation_fn: Activation function applied to each layer. If None,
 567         will use `tf.nn.relu`.
 568       dnn_dropout: When not None, the probability we will drop out
 569         a given coordinate.
 570       label_dimension: Number of regression targets per example. This is the
 571         size of the last dimension of the labels and logits `Tensor` objects
 572         (typically, these have shape `[batch_size, label_dimension]`).
 573       weight_column: A string or a `_NumericColumn` created by
 574         `tf.feature_column.numeric_column` defining feature column representing
 575         weights. It is used to down weight or boost examples during training. It
 576         will be multiplied by the loss of the example. If it is a string, it is
 577         used as a key to fetch weight tensor from the `features`. If it is a
 578         `_NumericColumn`, raw tensor is fetched by key `weight_column.key`,
 579         then weight_column.normalizer_fn is applied on it to get weight tensor.
 580       input_layer_partitioner: Partitioner for input layer. Defaults to
 581         `min_max_variable_partitioner` with `min_slice_size` 64 << 20.
 582       config: RunConfig object to configure the runtime settings.
 583       warm_start_from: A string filepath to a checkpoint to warm-start from, or
 584         a `WarmStartSettings` object to fully configure warm-starting.  If the
 585         string filepath is provided instead of a `WarmStartSettings`, then all
 586         weights are warm-started, and it is assumed that vocabularies and Tensor
 587         names are unchanged.
 588       loss_reduction: One of `tf.losses.Reduction` except `NONE`. Describes how
 589         to reduce training loss over batch. Defaults to `SUM`.
 590       batch_norm: Whether to use batch normalization after each hidden layer.
 591       linear_sparse_combiner: A string specifying how to reduce the linear model
 592         if a categorical column is multivalent.  One of "mean", "sqrtn", and
 593         "sum" -- these are effectively different ways to do example-level
 594         normalization, which can be useful for bag-of-words features.  For more
 595         details, see `tf.feature_column.linear_model`.
 596 
 597     Raises:
 598       ValueError: If both linear_feature_columns and dnn_features_columns are
 599         empty at the same time.
 600     """
