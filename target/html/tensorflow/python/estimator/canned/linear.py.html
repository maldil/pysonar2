  48 def _get_default_optimizer(feature_columns):
  49   learning_rate = min(_LEARNING_RATE, 1.0 / math.sqrt(len(feature_columns)))
  50   return ftrl.FtrlOptimizer(learning_rate=learning_rate)

  53 def _get_expanded_variable_list(var_list):
  54   """Given a list of variables, expands them if they are partitioned.
  55 
  56   Args:
  57     var_list: A list of variables.
  58 
  59   Returns:
  60     A list of variables where each partitioned variable is expanded to its
  61     components.
  62   """
  63   returned_list = []
  64   for variable in var_list:
  65     if (isinstance(variable, variable_ops.Variable) or
  66         resource_variable_ops.is_resource_variable(variable)):
  67       returned_list.append(variable)  # Single variable case.
  68     else:  # Must be a PartitionedVariable, so convert into a list.
  69       returned_list.extend(list(variable))
  70   return returned_list

  74 def _compute_fraction_of_zero(variables):
  75   """Given a linear variables list, compute the fraction of zero weights.
  76 
  77   Args:
  78     variables: A list or list of list of variables
  79 
  80   Returns:
  81     The fraction of zeros (sparsity) in the linear model.
  82   """
  83   all_weight_vars = []
  84   for var_or_var_list in variables:
  85     var_list = nest.flatten(var_or_var_list)
  86     # Skip empty-lists associated with columns that created no Variables.
  87     if var_list:
  88       all_weight_vars += [array_ops.reshape(var, [-1]) for var in var_list]
  89   return nn.zero_fraction(array_ops.concat(all_weight_vars, axis=0))

  92 def _linear_logit_fn_builder(units, feature_columns, sparse_combiner='sum'):
  93   """Function builder for a linear logit_fn.
  94 
  95   Args:
  96     units: An int indicating the dimension of the logit layer.
  97     feature_columns: An iterable containing all the feature columns used by
  98       the model.
  99     sparse_combiner: A string specifying how to reduce if a categorical column
 100       is multivalent.  One of "mean", "sqrtn", and "sum".
 101 
 102   Returns:
 103     A logit_fn (see below).
 104 
 105   """
 106 
 107   def linear_logit_fn(features):
 108     """Linear model logit_fn.
 109 
 110     Args:
 111       features: This is the first item returned from the `input_fn`
 112                 passed to `train`, `evaluate`, and `predict`. This should be a
 113                 single `Tensor` or `dict` of same.
 114 
 115     Returns:
 116       A `Tensor` representing the logits.
 117     """
 118     if feature_column_v2.is_feature_column_v2(feature_columns):
 119       shared_state_manager = feature_column_v2.SharedEmbeddingStateManager()
 120       linear_model = feature_column_v2.LinearModel(
 121           feature_columns=feature_columns,
 122           units=units,
 123           sparse_combiner=sparse_combiner,
 124           shared_state_manager=shared_state_manager)
 125       logits = linear_model(features)
 126       bias = linear_model.bias_variable
 127 
 128       # We'd like to get all the non-bias variables associated with this
 129       # LinearModel. This includes the shared embedding variables as well.
 130       variables = linear_model.variables
 131       variables.remove(bias)
 132       variables.extend(shared_state_manager.variables)
 133 
 134       # Expand (potential) Partitioned variables
 135       bias = _get_expanded_variable_list([bias])
 136       variables = _get_expanded_variable_list(variables)
 137     else:
 138       linear_model = feature_column._LinearModel(  # pylint: disable=protected-access
 139           feature_columns=feature_columns,
 140           units=units,
 141           sparse_combiner=sparse_combiner,
 142           name='linear_model')
 143       logits = linear_model(features)
 144       cols_to_vars = linear_model.cols_to_vars()
 145       bias = cols_to_vars.pop('bias')
 146       variables = cols_to_vars.values()
 147 
 148     if units > 1:
 149       summary.histogram('bias', bias)
 150     else:
 151       # If units == 1, the bias value is a length-1 list of a scalar Tensor,
 152       # so we should provide a scalar summary.
 153       summary.scalar('bias', bias[0][0])
 154     summary.scalar('fraction_of_zero_weights',
 155                    _compute_fraction_of_zero(variables))
 156     return logits
 157 
 158   return linear_logit_fn

 161 def _linear_model_fn(features, labels, mode, head, feature_columns, optimizer,
 162                      partitioner, config, sparse_combiner='sum'):
 163   """A model_fn for linear models that use a gradient-based optimizer.
 164 
 165   Args:
 166     features: dict of `Tensor`.
 167     labels: `Tensor` of shape `[batch_size, logits_dimension]`.
 168     mode: Defines whether this is training, evaluation or prediction.
 169       See `ModeKeys`.
 170     head: A `Head` instance.
 171     feature_columns: An iterable containing all the feature columns used by
 172       the model.
 173     optimizer: string, `Optimizer` object, or callable that defines the
 174       optimizer to use for training. If `None`, will use a FTRL optimizer.
 175     partitioner: Partitioner for variables.
 176     config: `RunConfig` object to configure the runtime settings.
 177     sparse_combiner: A string specifying how to reduce if a categorical column
 178       is multivalent.  One of "mean", "sqrtn", and "sum".
 179 
 180   Returns:
 181     An `EstimatorSpec` instance.
 182 
 183   Raises:
 184     ValueError: mode or params are invalid, or features has the wrong type.
 185   """
 186   if not isinstance(features, dict):
 187     raise ValueError('features should be a dictionary of `Tensor`s. '
 188                      'Given type: {}'.format(type(features)))
 189 
 190   optimizer = optimizers.get_optimizer_instance(
 191       optimizer or _get_default_optimizer(feature_columns),
 192       learning_rate=_LEARNING_RATE)
 193   num_ps_replicas = config.num_ps_replicas if config else 0
 194 
 195   partitioner = partitioner or (
 196       partitioned_variables.min_max_variable_partitioner(
 197           max_partitions=num_ps_replicas,
 198           min_slice_size=64 << 20))
 199 
 200   with variable_scope.variable_scope(
 201       'linear',
 202       values=tuple(six.itervalues(features)),
 203       partitioner=partitioner):
 204 
 205     logit_fn = _linear_logit_fn_builder(
 206         units=head.logits_dimension, feature_columns=feature_columns,
 207         sparse_combiner=sparse_combiner)
 208     logits = logit_fn(features=features)
 209 
 210     return head.create_estimator_spec(
 211         features=features,
 212         mode=mode,
 213         labels=labels,
 214         optimizer=optimizer,
 215         logits=logits)

 218 @estimator_export('estimator.LinearClassifier')
 219 class LinearClassifier(estimator.Estimator):
 220   """Linear classifier model.
 221 
 222   Train a linear model to classify instances into one of multiple possible
 223   classes. When number of possible classes is 2, this is binary classification.
 224 
 225   Example:
 226 
 227   ```python
 228   categorical_column_a = categorical_column_with_hash_bucket(...)
 229   categorical_column_b = categorical_column_with_hash_bucket(...)
 230 
 231   categorical_feature_a_x_categorical_feature_b = crossed_column(...)
 232 
 233   # Estimator using the default optimizer.
 234   estimator = LinearClassifier(
 235       feature_columns=[categorical_column_a,
 236                        categorical_feature_a_x_categorical_feature_b])
 237 
 238   # Or estimator using the FTRL optimizer with regularization.
 239   estimator = LinearClassifier(
 240       feature_columns=[categorical_column_a,
 241                        categorical_feature_a_x_categorical_feature_b],
 242       optimizer=tf.train.FtrlOptimizer(
 243         learning_rate=0.1,
 244         l1_regularization_strength=0.001
 245       ))
 246 
 247   # Or estimator using an optimizer with a learning rate decay.
 248   estimator = LinearClassifier(
 249       feature_columns=[categorical_column_a,
 250                        categorical_feature_a_x_categorical_feature_b],
 251       optimizer=lambda: tf.train.FtrlOptimizer(
 252           learning_rate=tf.exponential_decay(
 253               learning_rate=0.1,
 254               global_step=tf.get_global_step(),
 255               decay_steps=10000,
 256               decay_rate=0.96))
 257 
 258   # Or estimator with warm-starting from a previous checkpoint.
 259   estimator = LinearClassifier(
 260       feature_columns=[categorical_column_a,
 261                        categorical_feature_a_x_categorical_feature_b],
 262       warm_start_from="/path/to/checkpoint/dir")
 263 
 264 
 265   # Input builders
 266   def input_fn_train: # returns x, y (where y represents label's class index).
 267     ...
 268   def input_fn_eval: # returns x, y (where y represents label's class index).
 269     ...
 270   estimator.train(input_fn=input_fn_train)
 271   estimator.evaluate(input_fn=input_fn_eval)
 272   estimator.predict(input_fn=input_fn_predict)
 273   ```
 274 
 275   Input of `train` and `evaluate` should have following features,
 276     otherwise there will be a `KeyError`:
 277 
 278   * if `weight_column` is not `None`, a feature with
 279     `key=weight_column` whose value is a `Tensor`.
 280   * for each `column` in `feature_columns`:
 281     - if `column` is a `SparseColumn`, a feature with `key=column.name`
 282       whose `value` is a `SparseTensor`.
 283     - if `column` is a `WeightedSparseColumn`, two features: the first with
 284       `key` the id column name, the second with `key` the weight column name.
 285       Both features' `value` must be a `SparseTensor`.
 286     - if `column` is a `RealValuedColumn`, a feature with `key=column.name`
 287       whose `value` is a `Tensor`.
 288 
 289   Loss is calculated by using softmax cross entropy.
 290 
 291   @compatibility(eager)
 292   Estimators can be used while eager execution is enabled. Note that `input_fn`
 293   and all hooks are executed inside a graph context, so they have to be written
 294   to be compatible with graph mode. Note that `input_fn` code using `tf.data`
 295   generally works in both graph and eager modes.
 296   @end_compatibility
 297   """
 298 
 299   def __init__(self,
 300                feature_columns,
 301                model_dir=None,
 302                n_classes=2,
 303                weight_column=None,
 304                label_vocabulary=None,
 305                optimizer='Ftrl',
 306                config=None,
 307                partitioner=None,
 308                warm_start_from=None,
 309                loss_reduction=losses.Reduction.SUM,
 310                sparse_combiner='sum'):
 311     """Construct a `LinearClassifier` estimator object.
 312 
 313     Args:
 314       feature_columns: An iterable containing all the feature columns used by
 315         the model. All items in the set should be instances of classes derived
 316         from `FeatureColumn`.
 317       model_dir: Directory to save model parameters, graph and etc. This can
 318         also be used to load checkpoints from the directory into a estimator
 319         to continue training a previously saved model.
 320       n_classes: number of label classes. Default is binary classification.
 321         Note that class labels are integers representing the class index (i.e.
 322         values from 0 to n_classes-1). For arbitrary label values (e.g. string
 323         labels), convert to class indices first.
 324       weight_column: A string or a `_NumericColumn` created by
 325         `tf.feature_column.numeric_column` defining feature column representing
 326         weights. It is used to down weight or boost examples during training. It
 327         will be multiplied by the loss of the example. If it is a string, it is
 328         used as a key to fetch weight tensor from the `features`. If it is a
 329         `_NumericColumn`, raw tensor is fetched by key `weight_column.key`,
 330         then weight_column.normalizer_fn is applied on it to get weight tensor.
 331       label_vocabulary: A list of strings represents possible label values. If
 332         given, labels must be string type and have any value in
 333         `label_vocabulary`. If it is not given, that means labels are
 334         already encoded as integer or float within [0, 1] for `n_classes=2` and
 335         encoded as integer values in {0, 1,..., n_classes-1} for `n_classes`>2 .
 336         Also there will be errors if vocabulary is not provided and labels are
 337         string.
 338       optimizer: An instance of `tf.Optimizer` used to train the model. Can also
 339         be a string (one of 'Adagrad', 'Adam', 'Ftrl', 'RMSProp', 'SGD'), or
 340         callable. Defaults to FTRL optimizer.
 341       config: `RunConfig` object to configure the runtime settings.
 342       partitioner: Optional. Partitioner for input layer.
 343       warm_start_from: A string filepath to a checkpoint to warm-start from, or
 344         a `WarmStartSettings` object to fully configure warm-starting.  If the
 345         string filepath is provided instead of a `WarmStartSettings`, then all
 346         weights and biases are warm-started, and it is assumed that vocabularies
 347         and Tensor names are unchanged.
 348       loss_reduction: One of `tf.losses.Reduction` except `NONE`. Describes how
 349         to reduce training loss over batch. Defaults to `SUM`.
 350       sparse_combiner: A string specifying how to reduce if a categorical column
 351         is multivalent.  One of "mean", "sqrtn", and "sum" -- these are
 352         effectively different ways to do example-level normalization, which can
 353         be useful for bag-of-words features. for more details, see
 354         `tf.feature_column.linear_model`.
 355 
 356     Returns:
 357       A `LinearClassifier` estimator.
 358 
 359     Raises:
 360       ValueError: if n_classes < 2.
 361     """
 362     if n_classes == 2:
 363       head = head_lib._binary_logistic_head_with_sigmoid_cross_entropy_loss(  # pylint: disable=protected-access
 364           weight_column=weight_column,
 365           label_vocabulary=label_vocabulary,
 366           loss_reduction=loss_reduction)
 367     else:
 368       head = head_lib._multi_class_head_with_softmax_cross_entropy_loss(  # pylint: disable=protected-access
 369           n_classes, weight_column=weight_column,
 370           label_vocabulary=label_vocabulary,
 371           loss_reduction=loss_reduction)
 372 
 373     def _model_fn(features, labels, mode, config):
 374       """Call the defined shared _linear_model_fn."""
 375       return _linear_model_fn(
 376           features=features,
 377           labels=labels,
 378           mode=mode,
 379           head=head,
 380           feature_columns=tuple(feature_columns or []),
 381           optimizer=optimizer,
 382           partitioner=partitioner,
 383           config=config,
 384           sparse_combiner=sparse_combiner)
 385 
 386     super(LinearClassifier, self).__init__(
 387         model_fn=_model_fn,
 388         model_dir=model_dir,
 389         config=config,
 390         warm_start_from=warm_start_from)
