  15 """Debug and model explainability logic for boosted trees."""

  17 from __future__ import absolute_import

  21 import numpy as np
  22 
  23 from tensorflow.core.kernels.boosted_trees import boosted_trees_pb2
  24 
  25 # For directional feature contributions.
  26 _DEBUG_PROTO_KEY = '_serialized_debug_outputs_proto'
  27 _BIAS_ID = 0
  28 
  29 
  30 def _parse_debug_proto_string(example_proto_serialized):
  31   example_debug_outputs = boosted_trees_pb2.DebugOutput()
  32   example_debug_outputs.ParseFromString(example_proto_serialized)
  33   feature_ids = example_debug_outputs.feature_ids
  34   logits_path = example_debug_outputs.logits_path
  35   return feature_ids, logits_path
  36 
  37 
  38 def _compute_directional_feature_contributions(example_feature_ids,
  39                                                example_logits_paths, activation,
  40                                                num_bucketized_features):
  41   """Directional feature contributions and bias, per example."""
  42   # Initialize contributions to 0.
  43   dfcs = {k: 0 for k in range(num_bucketized_features)}
  44 
  45   # Traverse tree subtracting child prediction from parent prediction and
  46   # associating change with feature id used to split.
  47   predictions = np.array(activation(example_logits_paths))
  48   delta_pred = predictions[_BIAS_ID + 1:] - predictions[:-1]
  49   # Group by feature id, then sum delta_pred.
  50   contribs = np.bincount(
  51       example_feature_ids,
  52       weights=delta_pred,
  53       minlength=num_bucketized_features)
  54   for f, dfc in zip(range(num_bucketized_features), contribs):
  55     dfcs[f] = dfc
  56   return predictions[_BIAS_ID], dfcs
  57 
  58 
  59 def _identity(logits):
  60   return logits
  61 
  62 
  63 def _sigmoid(logits):
  64   # TODO(crawles): Change to softmax once multiclass support is available.
  65   return 1 / (1 + np.exp(-np.array(logits)))
  66 
  67 
  68 def _parse_explanations_from_prediction(serialized_debug_proto,
  69                                         n_features,
  70                                         classification=False):
  71   """Parse serialized explanability proto, compute dfc, and return bias, dfc."""
  72   feature_ids, logits_path = _parse_debug_proto_string(serialized_debug_proto)
  73   if classification:
  74     activation = _sigmoid
  75   else:
  76     activation = _identity
  77   bias, dfcs = _compute_directional_feature_contributions(
  78       feature_ids, logits_path, activation, n_features)
  79   # TODO(crawles): Prediction path and leaf IDs.
  80   return bias, dfcs
