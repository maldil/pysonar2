  32 _OPTIMIZER_CLS_NAMES = {
  33     'Adagrad': adagrad.AdagradOptimizer,
  34     'Adam': adam.AdamOptimizer,
  35     'Ftrl': ftrl.FtrlOptimizer,
  36     'RMSProp': rmsprop.RMSPropOptimizer,
  37     'SGD': gradient_descent.GradientDescentOptimizer,
  38 }
  39 
  40 
  41 def get_optimizer_instance(opt, learning_rate=None):
  42   """Returns an optimizer instance.
  43 
  44   Supports the following types for the given `opt`:
  45   * An `Optimizer` instance: Returns the given `opt`.
  46   * A string: Creates an `Optimizer` subclass with the given `learning_rate`.
  47     Supported strings:
  48     * 'Adagrad': Returns an `AdagradOptimizer`.
  49     * 'Adam': Returns an `AdamOptimizer`.
  50     * 'Ftrl': Returns an `FtrlOptimizer`.
  51     * 'RMSProp': Returns an `RMSPropOptimizer`.
  52     * 'SGD': Returns a `GradientDescentOptimizer`.
  53 
  54   Args:
  55     opt: An `Optimizer` instance, or string, as discussed above.
  56     learning_rate: A float. Only used if `opt` is a string.
  57 
  58   Returns:
  59     An `Optimizer` instance.
  60 
  61   Raises:
  62     ValueError: If `opt` is an unsupported string.
  63     ValueError: If `opt` is a supported string but `learning_rate` was not
  64       specified.
  65     ValueError: If `opt` is none of the above types.
  66   """
  67   if isinstance(opt, six.string_types):
  68     if opt in six.iterkeys(_OPTIMIZER_CLS_NAMES):
  69       if not learning_rate:
  70         raise ValueError('learning_rate must be specified when opt is string.')
  71       return _OPTIMIZER_CLS_NAMES[opt](learning_rate=learning_rate)
  72     raise ValueError(
  73         'Unsupported optimizer name: {}. Supported names are: {}'.format(
  74             opt, tuple(sorted(six.iterkeys(_OPTIMIZER_CLS_NAMES)))))
  75   if callable(opt):
  76     opt = opt()
  77   if not isinstance(opt, optimizer_lib.Optimizer):
  78     raise ValueError(
  79         'The given object is not an Optimizer instance. Given: {}'.format(opt))
  80   return opt
