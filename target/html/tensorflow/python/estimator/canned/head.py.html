  73 def _create_eval_metrics_tuple(fn, kwargs):
  74   """Creates TPU eval metrics tuple.
  75 
  76   Helper function to make eval_metric tuple (eval_metric_fn, fn_kwargs) used
  77   by `TPUEstimator`. TPUEstimator requires that `eval_metric_fn` take
  78   exclusively Tensor arguments. This helper can help create such a function from
  79   a more generic function that can take both Tensor and non-Tensor arguments.
  80 
  81   Args:
  82     fn: A eval_metric_fn that takes both Tensor and non-Tensor arguments.
  83         This function must return a dict of form
  84         {'metric name': (metric_tensor, eval_op)}
  85     kwargs: Dict of arguments for `fn`.
  86 
  87   Returns:
  88     `eval_metric` tuple that can be passed to a `model_fn._TPUEstimatorSpec`.
  89   """
  90   tensor_kwargs = {}
  91   nontensor_kwargs = {}
  92   for k, v in six.iteritems(kwargs):
  93     if tensor_util.is_tensor(v):
  94       tensor_kwargs[k] = v
  95     else:
  96       nontensor_kwargs[k] = v
  97   def _fn(**tensors):
  98     return fn(**dict(nontensor_kwargs, **tensors))
  99   return (_fn, tensor_kwargs)

 102 class _Head(object):
 103   """Interface for the head/top of a model.
 104 
 105   Given logits (or output of a hidden layer), a Head knows how to compute
 106   predictions, loss, train_op, metrics and export outputs. It is meant to:
 107 
 108   1. Simplify writing model_fn and to make model_fn more configurable
 109   2. Support wide range of machine learning models. Since most heads can work
 110      with logits, they can support DNN, RNN, Wide, Wide&Deep,
 111      Global objectives, Gradient boosted trees and many other types
 112      of machine learning models.
 113 
 114   Common usage:
 115   Here is simplified model_fn to build a DNN regression model.
 116     ```python
 117     def _my_dnn_model_fn(features, labels, mode, params, config=None):
 118       # Optionally your callers can pass head to model_fn as a param.
 119       head = tf.contrib.estimator.regression_head(...)
 120       inputs = tf.feature_column.input_layer(features, ...)
 121       hidden_layer0 = tf.layers.dense(
 122           inputs, units=1000, activation=tf.nn.relu)
 123       hidden_layer1 = tf.layers.dense(
 124           hidden_layer0, units=500, activation=tf.nn.relu)
 125       logits = tf.layers.dense(
 126           hidden_layer1, units=head.logits_dimension, activation=None)
 127 
 128       return head.create_estimator_spec(
 129           features=features,
 130           labels=labels,
 131           mode=mode,
 132           logits=logits,
 133           optimizer=optimizer)
 134     ```
 135 
 136   There are cases where computing and applying gradients can not be meaningfully
 137   captured with optimizer or train_op_fn we support (for example, with sync
 138   optimizer). In such case, you can take the responsibility on your own. Here is
 139   a common use case,
 140     ```python
 141     estimator_spec = head.create_estimator_spec(
 142         features=features,
 143         labels=labels,
 144         mode=mode,
 145         logits=logits,
 146         train_op_fn=lambda _: tf.no_op())
 147     if mode == model_fn.ModeKeys.TRAIN:
 148       optimizer = ...
 149       sync = tf.train.SyncReplicasOptimizer(opt=optimizer, ...)
 150       update_op = sync.minimize(
 151           estimator_spec.loss, global_step=tf.get_global_step())
 152       hooks = [sync.make_session_run_hook(is_chief)]
 153       ... update train_op and hooks in EstimatorSpec and return
 154     ```
 155   """
 156   __metaclass__ = abc.ABCMeta

 278 def _check_dense_labels_match_logits_and_reshape(
 279     labels, logits, expected_labels_dimension):
 280   """Checks that labels shape matches logits and reshapes if needed.
 281 
 282   Consider logits of shape [D0, D1, ... DN, logits_dimension]. Then labels
 283   shape must be [D0, D1, ... DN, expected_labels_dimension].
 284   If expected_labels_dimension=1, labels could be [D0, D1, ... DN] and this
 285   method reshapes them to [D0, D1, ... DN, 1].
 286 
 287   Args:
 288     labels: labels Tensor.
 289     logits: logits Tensor.
 290     expected_labels_dimension: Integer.
 291   Returns:
 292     Validated and reshaped labels Tensor.
 293   Raises:
 294     ValueError: If labels is a SparseTensor.
 295     ValueError: If labels shape is statically defined and fails validation.
 296     OpError: If labels shape is not statically defined and fails validation.
 297   """
 298   if labels is None:
 299     raise ValueError(
 300         'You must provide a labels Tensor. Given: None. '
 301         'Suggested troubleshooting steps: Check that your data contain '
 302         'your label feature. Check that your input_fn properly parses and '
 303         'returns labels.')
 304   with ops.name_scope(None, 'labels', (labels, logits)) as scope:
 305     labels = sparse_tensor.convert_to_tensor_or_sparse_tensor(labels)
 306     if isinstance(labels, sparse_tensor.SparseTensor):
 307       raise ValueError(
 308           'SparseTensor labels are not supported. '
 309           'labels must be a Tensor of shape [D0, D1, ..., DN, %s], '
 310           'e.g. [batch_size, %s]. '
 311           'Suggested Fix (1): Check the label feature in your data. '
 312           'Each example must contain %s value(s). If not, your choice of label '
 313           'was probably incorrect. '
 314           'Suggested Fix (2): In your input_fn, use '
 315           'tf.sparse_tensor_to_dense() to turn labels into a Tensor.'
 316           '' % (expected_labels_dimension, expected_labels_dimension,
 317                 expected_labels_dimension))
 318     if (labels.shape.ndims is not None and logits.shape.ndims is not None and
 319         labels.shape.ndims == logits.shape.ndims - 1):
 320       labels = array_ops.expand_dims(labels, -1)
 321     labels_shape = array_ops.shape(labels)
 322     logits_shape = array_ops.shape(logits)
 323     err_msg = (
 324         'labels shape must be [D0, D1, ... DN, {}]. '
 325         'Suggested Fix: check your n_classes argument to the estimator '
 326         'and/or the shape of your label.'.format(expected_labels_dimension))
 327     assert_rank = check_ops.assert_rank_at_least(labels, 2, message=err_msg)
 328     with ops.control_dependencies([assert_rank]):
 329       static_shape = labels.shape
 330       if static_shape.ndims is not None:
 331         dim1 = static_shape[-1]
 332         if (dim1 is not None) and (dim1 != expected_labels_dimension):
 333           raise ValueError(
 334               'Mismatched label shape. '
 335               'Expected labels dimension=%s.  Received %s. '
 336               'Suggested Fix:'
 337               'If your classifier expects one-hot encoding label,'
 338               'check your n_classes argument to the estimator '
 339               'and/or the shape of your label. '
 340               'Otherwise, check the shape of your label.' %
 341               (expected_labels_dimension, dim1))
 342       expected_labels_shape = array_ops.concat(
 343           [logits_shape[:-1], [expected_labels_dimension]], axis=0)
 344       assert_dimension = check_ops.assert_equal(
 345           expected_labels_shape, labels_shape, message=err_msg,
 346           data=['expected_labels_shape: ', expected_labels_shape,
 347                 'labels_shape: ', labels_shape])
 348       with ops.control_dependencies([assert_dimension]):
 349         return array_ops.identity(labels, name=scope)

 352 def _get_weights_and_check_match_logits(
 353     features, weight_column, logits, allow_per_logit_weights=False):
 354   """Fetches weights from features and checks that the shape matches logits.
 355 
 356   Consider logits of shape [D0, D1, ... DN, logits_dimension]. Weights shape
 357   can be either:
 358   * [D0, D1, ... DN, logits_dimension] if `allow_per_logit_weights=True`.
 359   * [D0, D1, ... DN, 1]
 360   * [D0, D1, ... DN]: In this case, weights is reshaped into
 361     [D0, D1, ... DN, 1] to work with weight broadcasting rules.
 362 
 363   Args:
 364     features: The features dict that contains weights.
 365     weight_column: The weight column. If not given, this method returns 1.
 366     logits: logits Tensor.
 367     allow_per_logit_weights: Boolean. Whether we allow weights along the logits
 368       dimension, namely shape `[D0, D1, ... DN, logits_dimension]`.
 369   Returns:
 370     Validated and reshaped weights Tensor.
 371   Raises:
 372     ValueError: If the weights `Tensor` cannot be cast into float.
 373   """
 374   if allow_per_logit_weights:
 375     err_msg = (
 376         'weights shape must be [D0, D1, ... DN], [D0, D1, ... DN, 1] or '
 377         '[D0, D1, ... DN, logits_dimension]')
 378   else:
 379     err_msg = (
 380         'weights shape must be [D0, D1, ... DN] or [D0, D1, ... DN, 1]')
 381   with ops.name_scope(
 382       None, 'weights',
 383       values=tuple(six.itervalues(features)) + (logits,)) as scope:
 384     # Fetch the weights.
 385     if weight_column is None:
 386       return 1.
 387     if isinstance(weight_column, six.string_types):
 388       weight_column = feature_column_lib.numeric_column(
 389           key=weight_column, shape=(1,))
 390     if not isinstance(weight_column, feature_column_lib._NumericColumn):  # pylint: disable=protected-access
 391       raise TypeError('Weight column must be either a string or _NumericColumn.'
 392                       ' Given type: {}.'.format(type(weight_column)))
 393     weights = weight_column._get_dense_tensor(  # pylint: disable=protected-access
 394         feature_column_lib._LazyBuilder(features))  # pylint: disable=protected-access
 395     if not (weights.dtype.is_floating or weights.dtype.is_integer):
 396       raise ValueError('Weight column should be castable to float. '
 397                        'Given dtype: {}'.format(weights.dtype))
 398     weights = math_ops.to_float(weights, name='weights')
 399 
 400     # Validate the weights shape.
 401     weights_shape = array_ops.shape(weights, name='weights_shape')
 402     logits_shape = array_ops.shape(logits, name='logits_shape')
 403     if (weights.shape.ndims is not None and logits.shape.ndims is not None and
 404         weights.shape.ndims == logits.shape.ndims - 1):
 405       assert_dimension = check_ops.assert_equal(
 406           logits_shape[:-1], weights_shape, message=err_msg,
 407           data=['logits_shape: ', logits_shape,
 408                 'weights_shape: ', weights_shape])
 409       with ops.control_dependencies([assert_dimension]):
 410         return array_ops.expand_dims(weights, -1, name=scope)
 411     supported_weights_shape = array_ops.concat([logits_shape[:-1], [1]], axis=0)
 412     if allow_per_logit_weights:
 413       condition = math_ops.reduce_any(
 414           [math_ops.reduce_all(math_ops.equal(logits_shape, weights_shape)),
 415            math_ops.reduce_all(math_ops.equal(
 416                supported_weights_shape, weights_shape))])
 417       assert_dimension = control_flow_ops.Assert(
 418           condition=condition,
 419           data=[err_msg, 'logits_shape: ', logits_shape,
 420                 'weights_shape: ', weights_shape])
 421     else:
 422       assert_dimension = check_ops.assert_equal(
 423           supported_weights_shape, weights_shape, message=err_msg,
 424           data=['logits_shape: ', logits_shape,
 425                 'weights_shape: ', weights_shape])
 426     with ops.control_dependencies([assert_dimension]):
 427       return array_ops.identity(weights, name=scope)

 430 def _check_logits_final_dim(logits, expected_logits_dimension):
 431   """Checks that logits shape is [D0, D1, ... DN, logits_dimension]."""
 432   with ops.name_scope(None, 'logits', (logits,)) as scope:
 433     logits = math_ops.to_float(logits)
 434     logits_shape = array_ops.shape(logits)
 435     assert_rank = check_ops.assert_rank_at_least(
 436         logits, 2, data=[logits_shape],
 437         message='logits shape must be [D0, D1, ... DN, logits_dimension]')
 438     with ops.control_dependencies([assert_rank]):
 439       static_shape = logits.shape
 440       if static_shape.ndims is not None and static_shape[-1] is not None:
 441         if static_shape[-1] != expected_logits_dimension:
 442           raise ValueError(
 443               'logits shape must be [D0, D1, ... DN, logits_dimension], '
 444               'got %s.' % (static_shape,))
 445         return logits
 446       assert_dimension = check_ops.assert_equal(
 447           expected_logits_dimension, logits_shape[-1], data=[logits_shape],
 448           message='logits shape must be [D0, D1, ... DN, logits_dimension]')
 449       with ops.control_dependencies([assert_dimension]):
 450         return array_ops.identity(logits, name=scope)

 475 def _call_loss_fn(loss_fn, labels, logits, features, expected_loss_dim=1):
 476   """Calls loss_fn and checks the returned shape.
 477 
 478   Args:
 479     loss_fn: The loss function.
 480     labels: Processed labels Tensor.
 481     logits: Logits Tensor of shape [D0, D1, ... DN, logits_dimension].
 482     features: Features dict.
 483     expected_loss_dim: The expected last dimension of loss Tensor.
 484   Returns:
 485     Loss Tensor with shape [D0, D1, ... DN, expected_loss_dim].
 486   """
 487   loss_fn_args = function_utils.fn_args(loss_fn)
 488   kwargs = {}
 489   if 'features' in loss_fn_args:
 490     kwargs['features'] = features
 491   with ops.name_scope(
 492       None, 'call_loss_fn',
 493       values=[labels, logits] + list(six.itervalues(features))):
 494     unweighted_loss = loss_fn(labels=labels, logits=logits, **kwargs)
 495     logits_shape = array_ops.shape(logits, name='logits_shape')
 496     expected_loss_shape = array_ops.concat(
 497         [logits_shape[:-1], [expected_loss_dim]], axis=0,
 498         name='expected_loss_shape')
 499     loss_shape = array_ops.shape(unweighted_loss, name='loss_shape')
 500     check_loss_shape_op = control_flow_ops.Assert(
 501         math_ops.reduce_all(math_ops.equal(loss_shape, expected_loss_shape)),
 502         data=[
 503             'loss_fn must return Tensor of shape '
 504             '[D0, D1, ... DN, {}]. '.format(expected_loss_dim),
 505             'logits_shape: ', logits_shape, 'loss_shape: ', loss_shape],
 506         name='check_loss_shape')
 507     with ops.control_dependencies([check_loss_shape_op]):
 508       return array_ops.identity(unweighted_loss)

 511 def _indicator_labels_mean(labels, weights=None, name=None):
 512   with ops.name_scope(name, 'labels_mean', (labels, weights)) as scope:
 513     labels = math_ops.to_float(labels, name='labels')

 610 
 611   The head expects `logits` with shape `[D0, D1, ... DN, n_classes]`.
 612   In many applications, the shape is `[batch_size, n_classes]`.
 613 
 614   `labels` must be a dense `Tensor` with shape matching `logits`, namely
 615   `[D0, D1, ... DN, 1]`. If `label_vocabulary` given, `labels` must be a string
 616   `Tensor` with values from the vocabulary. If `label_vocabulary` is not given,

 620   `[D0, D1, ... DN]`, or `[D0, D1, ... DN, 1]`.
 621

 623   labels have shape `[batch_size, 1]`, the loss is the weighted sum over
 624   `batch_size`.
 625 
 626   Also supports custom `loss_fn`. `loss_fn` takes `(labels, logits)` or
 627   `(labels, logits, features)` as arguments and returns unreduced loss with

 643       `label_vocabulary` is not provided but labels are strings.

 651     An instance of `_Head` for multi class classification.

 658                                                      (list, tuple)):
 659     raise ValueError(
 660         'label_vocabulary should be a list or a tuple. Given type: {}'.format(
 661             type(label_vocabulary)))

 663       loss_reduction == losses.Reduction.NONE):

 665   if loss_fn:
 666     _validate_loss_fn_args(loss_fn)

 669       weight_column=weight_column,
 670       label_vocabulary=label_vocabulary,

 774   def _create_tpu_estimator_spec(
 775       self, features, mode, logits, labels=None, optimizer=None,
 776       train_op_fn=None, regularization_losses=None):
 777     """Returns a `model_fn._TPUEstimatorSpec`.
 778 
 779     Args:
 780       features: Input `dict` of `Tensor` or `SparseTensor` objects.
 781       mode: Estimator's `ModeKeys`.
 782       logits: logits `Tensor` with shape `[D0, D1, ... DN, logits_dimension]`.
 783         For many applications, the shape is `[batch_size, logits_dimension]`.
 784       labels: Labels integer or string `Tensor` with shape matching `logits`,
 785         namely `[D0, D1, ... DN, 1]` or `[D0, D1, ... DN]`. `labels` is
 786         required argument when `mode` equals `TRAIN` or `EVAL`.
 787       optimizer: `Optimizer` instance to optimize the loss in TRAIN mode.
 788         Namely, sets `train_op = optimizer.minimize(loss, global_step)`, which
 789         updates variables and increments `global_step`.
 790       train_op_fn: Function that takes a scalar loss `Tensor` and returns
 791         `train_op`. Used if `optimizer` is `None`.
 792       regularization_losses: A list of additional scalar losses to be added to
 793         the training loss, such as regularization losses. These losses are
 794         usually expressed as a batch average, so for best results users need to
 795         set `loss_reduction=SUM_OVER_BATCH_SIZE` or
 796         `loss_reduction=SUM_OVER_NONZERO_WEIGHTS` when creating the head to
 797         avoid scaling errors.
 798     Returns:
 799       A `model_fn._TPUEstimatorSpec` instance.
 800     Raises:
 801       ValueError: If both `train_op_fn` and `optimizer` are `None` in TRAIN
 802         mode, or if both are set.
 803     """
 804     with ops.name_scope(self._name, 'head'):
 805       logits = _check_logits_final_dim(logits, self.logits_dimension)
 806 
 807       # Predict.
 808       pred_keys = prediction_keys.PredictionKeys
 809       with ops.name_scope(None, 'predictions', (logits,)):
 810         # class_ids's shape is [D0, D1, ... DN].
 811         class_ids = math_ops.argmax(logits, axis=-1, name=pred_keys.CLASS_IDS)
 812         class_ids = array_ops.expand_dims(class_ids, axis=-1)
 813         if self._label_vocabulary:
 814           table = lookup_ops.index_to_string_table_from_tensor(
 815               vocabulary_list=self._label_vocabulary,
 816               name='class_string_lookup')
 817           classes = table.lookup(class_ids)
 818         else:
 819           classes = string_ops.as_string(class_ids, name='str_classes')
 820 
 821         probabilities = nn.softmax(logits, name=pred_keys.PROBABILITIES)
 822         predictions = {
 823             pred_keys.LOGITS: logits,
 824             pred_keys.PROBABILITIES: probabilities,
 825             # Expand to [batch_size, 1]
 826             pred_keys.CLASS_IDS: class_ids,
 827             pred_keys.CLASSES: classes,
 828         }
 829       if mode == model_fn.ModeKeys.PREDICT:
 830         classifier_output = _classification_output(
 831             scores=probabilities, n_classes=self._n_classes,
 832             label_vocabulary=self._label_vocabulary)
 833         return model_fn._TPUEstimatorSpec(  # pylint: disable=protected-access
 834             mode=model_fn.ModeKeys.PREDICT,
 835             predictions=predictions,
 836             export_outputs={
 837                 _DEFAULT_SERVING_KEY: classifier_output,
 838                 _CLASSIFY_SERVING_KEY: classifier_output,
 839                 _PREDICT_SERVING_KEY: export_output.PredictOutput(predictions)
 840             })
 841 
 842       training_loss, unreduced_loss, weights, label_ids = self.create_loss(
 843           features=features, mode=mode, logits=logits, labels=labels)
 844       if regularization_losses:
 845         regularization_loss = math_ops.add_n(regularization_losses)
 846         regularized_training_loss = math_ops.add_n(
 847             [training_loss, regularization_loss])
 848       else:
 849         regularization_loss = None
 850         regularized_training_loss = training_loss
 851       # Eval.
 852       if mode == model_fn.ModeKeys.EVAL:
 853         return model_fn._TPUEstimatorSpec(  # pylint: disable=protected-access
 854             mode=model_fn.ModeKeys.EVAL,
 855             predictions=predictions,
 856             loss=regularized_training_loss,
 857             eval_metrics=_create_eval_metrics_tuple(self._eval_metric_ops, {
 858                 'labels': label_ids,
 859                 'class_ids': class_ids,
 860                 'weights': weights,
 861                 'unreduced_loss': unreduced_loss,
 862                 'regularization_loss': regularization_loss
 863             }))
 864 
 865       # Train.
 866       if optimizer is not None:
 867         if train_op_fn is not None:
 868           raise ValueError('train_op_fn and optimizer cannot both be set.')
 869         train_op = optimizer.minimize(
 870             regularized_training_loss,
 871             global_step=training_util.get_global_step())
 872       elif train_op_fn is not None:
 873         train_op = train_op_fn(regularized_training_loss)
 874       else:
 875         raise ValueError('train_op_fn and optimizer cannot both be None.')
 876       train_op = _append_update_ops(train_op)
 877       # Only summarize mean_loss for SUM reduction to preserve backwards
 878       # compatibility. Otherwise skip it to avoid unnecessary computation.
 879       if self._loss_reduction == losses.Reduction.SUM:
 880         example_weight_sum = math_ops.reduce_sum(
 881             weights * array_ops.ones_like(unreduced_loss))
 882         mean_loss = training_loss / example_weight_sum
 883       else:
 884         mean_loss = None
 885     with ops.name_scope(''):
 886       keys = metric_keys.MetricKeys
 887       summary.scalar(
 888           _summary_key(self._name, keys.LOSS),
 889           regularized_training_loss)
 890       if mean_loss is not None:
 891         summary.scalar(
 892             _summary_key(self._name, keys.LOSS_MEAN),
 893             mean_loss)
 894       if regularization_loss is not None:
 895         summary.scalar(
 896             _summary_key(self._name, keys.LOSS_REGULARIZATION),
 897             regularization_loss)
 898     return model_fn._TPUEstimatorSpec(  # pylint: disable=protected-access
 899         mode=model_fn.ModeKeys.TRAIN,
 900         predictions=predictions,
 901         loss=regularized_training_loss,
 902         train_op=train_op)

1134   def _create_tpu_estimator_spec(
1135       self, features, mode, logits, labels=None, optimizer=None,
1136       train_op_fn=None, regularization_losses=None):
1137     """Returns an `EstimatorSpec`.
1138 
1139     Args:
1140       features: Input `dict` of `Tensor` or `SparseTensor` objects.
1141       mode: Estimator's `ModeKeys`.
1142       logits: logits `Tensor` with shape `[D0, D1, ... DN, 1]`. For many
1143         applications, the shape is `[batch_size, 1]`.

1149         updates variables and increments `global_step`.
1150       train_op_fn: Function that takes a scalar loss `Tensor` and returns
1151         `train_op`. Used if `optimizer` is `None`.
1152       regularization_losses: A list of additional scalar losses to be added to
1153         the training loss, such as regularization losses. These losses are
1154         usually expressed as a batch average, so for best results users need to
1155         set `loss_reduction=SUM_OVER_BATCH_SIZE` or
1156         `loss_reduction=SUM_OVER_NONZERO_WEIGHTS` when creating the head to
1157         avoid scaling errors.
1158     Returns:
1159       `EstimatorSpec`.
1160     Raises:
1161       ValueError: If both `train_op_fn` and `optimizer` are `None` in TRAIN
1162         mode, or if both are set.
1163     """
1164     # Predict.
1165     with ops.name_scope(self._name, 'head'):
1166       with ops.name_scope(None, 'predictions', (logits,)):
1167         pred_keys = prediction_keys.PredictionKeys
1168         logits = _check_logits_final_dim(logits, self.logits_dimension)
1169         logistic = math_ops.sigmoid(logits, name=pred_keys.LOGISTIC)
1170         two_class_logits = array_ops.concat(
1171             (array_ops.zeros_like(logits), logits),
1172             axis=-1, name='two_class_logits')
1173         probabilities = nn.softmax(
1174             two_class_logits, name=pred_keys.PROBABILITIES)
1175         class_ids = math_ops.argmax(
1176             two_class_logits, axis=-1, name=pred_keys.CLASS_IDS)
1177         class_ids = array_ops.expand_dims(class_ids, axis=-1)
1178         if self._label_vocabulary:
1179           table = lookup_ops.index_to_string_table_from_tensor(
1180               vocabulary_list=self._label_vocabulary,
1181               name='class_string_lookup')
1182           classes = table.lookup(class_ids)
1183         else:
1184           classes = string_ops.as_string(class_ids, name='str_classes')
1185         predictions = {
1186             pred_keys.LOGITS: logits,
1187             pred_keys.LOGISTIC: logistic,
1188             pred_keys.PROBABILITIES: probabilities,
1189             pred_keys.CLASS_IDS: class_ids,
1190             pred_keys.CLASSES: classes,
1191         }
1192       if mode == model_fn.ModeKeys.PREDICT:
1193         classifier_output = _classification_output(
1194             scores=probabilities, n_classes=2,
1195             label_vocabulary=self._label_vocabulary)
1196         return model_fn._TPUEstimatorSpec(  # pylint: disable=protected-access
1197             mode=model_fn.ModeKeys.PREDICT,
1198             predictions=predictions,
1199             export_outputs={
1200                 _DEFAULT_SERVING_KEY: classifier_output,
1201                 _CLASSIFY_SERVING_KEY: classifier_output,
1202                 _REGRESS_SERVING_KEY: export_output.RegressionOutput(
1203                     value=logistic),
1204                 _PREDICT_SERVING_KEY: export_output.PredictOutput(predictions)
1205             })
1206 
1207       (training_loss, unreduced_loss, weights, processed_labels) = (
1208           self.create_loss(
1209               features=features, mode=mode, logits=logits, labels=labels))
1210       if regularization_losses:
1211         regularization_loss = math_ops.add_n(regularization_losses)
1212         regularized_training_loss = math_ops.add_n(
1213             [training_loss, regularization_loss])
1214       else:
1215         regularization_loss = None
1216         regularized_training_loss = training_loss
1217 
1218       # Eval.
1219       if mode == model_fn.ModeKeys.EVAL:
1220         return model_fn._TPUEstimatorSpec(  # pylint: disable=protected-access
1221             mode=model_fn.ModeKeys.EVAL,
1222             predictions=predictions,
1223             loss=regularized_training_loss,
1224             eval_metrics=_create_eval_metrics_tuple(
1225                 self._eval_metric_ops,
1226                 {
1227                     'labels': processed_labels,
1228                     'logits': logits,
1229                     'logistic': logistic,
1230                     'class_ids': class_ids,
1231                     'weights': weights,
1232                     'unreduced_loss': unreduced_loss,
1233                     'regularization_loss': regularization_loss
1234                 }
1235             ))
1236 
1237       # Train.
1238       if optimizer is not None:
1239         if train_op_fn is not None:
1240           raise ValueError('train_op_fn and optimizer cannot both be set.')
1241         train_op = optimizer.minimize(
1242             regularized_training_loss,
1243             global_step=training_util.get_global_step())
1244       elif train_op_fn is not None:
1245         train_op = train_op_fn(regularized_training_loss)
1246       else:
1247         raise ValueError('train_op_fn and optimizer cannot both be None.')
1248       train_op = _append_update_ops(train_op)
1249       # Only summarize mean_loss for SUM reduction to preserve backwards
1250       # compatibility. Otherwise skip it to avoid unnecessary computation.
1251       if self._loss_reduction == losses.Reduction.SUM:
1252         example_weight_sum = math_ops.reduce_sum(
1253             weights * array_ops.ones_like(unreduced_loss))
1254         mean_loss = training_loss / example_weight_sum
1255       else:
1256         mean_loss = None
1257     with ops.name_scope(''):
1258       keys = metric_keys.MetricKeys
1259       summary.scalar(
1260           _summary_key(self._name, keys.LOSS),
1261           regularized_training_loss)
1262       if mean_loss is not None:
1263         summary.scalar(
1264             _summary_key(self._name, keys.LOSS_MEAN), mean_loss)
1265       if regularization_loss is not None:
1266         summary.scalar(
1267             _summary_key(self._name, keys.LOSS_REGULARIZATION),
1268             regularization_loss)
1269     return model_fn._TPUEstimatorSpec(  # pylint: disable=protected-access
1270         mode=model_fn.ModeKeys.TRAIN,
1271         predictions=predictions,
1272         loss=regularized_training_loss,
1273         train_op=train_op)

1276 def _regression_head(
1277     weight_column=None,
1278     label_dimension=1,
1279     loss_reduction=losses.Reduction.SUM,
1280     loss_fn=None,
1281     inverse_link_fn=None,
1282     name=None):
1283   """Creates a `_Head` for regression using the `mean_squared_error` loss.
1284 
1285   The loss is the weighted sum over all input dimensions. Namely, if the input
1286   labels have shape `[batch_size, label_dimension]`, the loss is the weighted
1287   sum over both `batch_size` and `label_dimension`.
1288 
1289   The head expects `logits` with shape `[D0, D1, ... DN, label_dimension]`.
1290   In many applications, the shape is `[batch_size, label_dimension]`.
1291 
1292   The `labels` shape must match `logits`, namely
1293   `[D0, D1, ... DN, label_dimension]`. If `label_dimension=1`, shape
1294   `[D0, D1, ... DN]` is also supported.
1295 
1296   If `weight_column` is specified, weights must be of shape
1297   `[D0, D1, ... DN]`, `[D0, D1, ... DN, 1]` or
1298   `[D0, D1, ... DN, label_dimension]`.
1299 
1300   Supports custom `loss_fn`. `loss_fn` takes `(labels, logits)` or
1301   `(labels, logits, features)` as arguments and returns unreduced loss with
1302   shape `[D0, D1, ... DN, label_dimension]`.
1303 
1304   Also supports custom `inverse_link_fn`, also known as 'mean function'.
1305   `inverse_link_fn` takes `logits` as argument and returns predicted values.
1306   This function is the inverse of the link function defined in
1307   https://en.wikipedia.org/wiki/Generalized_linear_model#Link_function
1308   Namely, for poisson regression, set `inverse_link_fn=tf.exp`.
1309 
1310   Args:
1311     weight_column: A string or a `_NumericColumn` created by
1312       `tf.feature_column.numeric_column` defining feature column representing
1313       weights. It is used to down weight or boost examples during training. It
1314       will be multiplied by the loss of the example.
1315     label_dimension: Number of regression labels per example. This is the size
1316       of the last dimension of the labels `Tensor` (typically, this has shape
1317       `[batch_size, label_dimension]`).
1318     loss_reduction: One of `tf.losses.Reduction` except `NONE`. Describes how to
1319       reduce training loss over batch. Defaults to `SUM`.
1320     loss_fn: Optional loss function. Defaults to `mean_squared_error`.
1321     inverse_link_fn: Optional inverse link function, also known as 'mean
1322       function'. Defaults to identity.
1323     name: name of the head. If provided, summary and metrics keys will be
1324       suffixed by `"/" + name`. Also used as `name_scope` when creating ops.
1325 
1326   Returns:
1327     An instance of `_Head` for linear regression.
1328 
1329   Raises:
1330     ValueError: If `label_dimension` or `loss_reduction` is invalid.
1331   """
1332   if (loss_reduction not in losses.Reduction.all() or
1333       loss_reduction == losses.Reduction.NONE):
1334     raise ValueError('Invalid loss_reduction: {}'.format(loss_reduction))
1335   if loss_fn:
1336     _validate_loss_fn_args(loss_fn)
1337   return _RegressionHeadWithMeanSquaredErrorLoss(
1338       weight_column=weight_column,
1339       label_dimension=label_dimension,
1340       loss_reduction=loss_reduction,
1341       loss_fn=loss_fn,
1342       inverse_link_fn=inverse_link_fn,
1343       name=name)
