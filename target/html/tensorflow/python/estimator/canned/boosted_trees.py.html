  63 def _get_transformed_features(features, sorted_feature_columns):
  64   """Gets the transformed features from features/feature_columns pair.
  65 
  66   Args:
  67     features: a dicionary of name to Tensor.
  68     sorted_feature_columns: a list/set of tf.feature_column, sorted by name.
  69 
  70   Returns:
  71     result_features: a list of the transformed features, sorted by the name.
  72 
  73   Raises:
  74     ValueError: when unsupported features/columns are tried.
  75   """
  76   # pylint:disable=protected-access
  77   transformed_features = feature_column_lib._transform_features(
  78       features, sorted_feature_columns)
  79   result_features = []
  80   for column in sorted_feature_columns:
  81     if isinstance(column, feature_column_lib._BucketizedColumn):
  82       source_name = column.source_column.name
  83       squeezed_tensor = array_ops.squeeze(transformed_features[column], axis=1)
  84       if len(squeezed_tensor.shape) > 1:
  85         raise ValueError('For now, only supports features equivalent to rank 1 '
  86                          'but column `{}` got: {}'.format(
  87                              source_name, features[source_name].shape))
  88       result_features.append(squeezed_tensor)
  89     elif isinstance(column, feature_column_lib._IndicatorColumn):
  90       source_name = column.categorical_column.name
  91       tensor = math_ops.to_int32(transformed_features[column])
  92       if len(tensor.shape) > 2:
  93         raise ValueError('Rank of indicator column must be no more than 2, '
  94                          'but column `{}` got: {}'.format(
  95                              source_name, features[source_name].shape))
  96       unstacked = array_ops.unstack(tensor, axis=1)
  97       result_features.extend(unstacked)
  98     else:
  99       raise ValueError(
 100           'For now, only bucketized_column and indicator_column is supported '
 101           'but got: {}'.format(column))
 102     # pylint:enable=protected-access
 103 
 104   return result_features

 121 def _group_features_by_num_buckets(sorted_feature_columns):
 122   """Groups feature ids by the number of buckets.
 123 
 124   Derives the feature ids based on iterating through ordered feature columns
 125   and groups them by the number of buckets each feature require. Returns a
 126   sorted list of buckets and a list of lists of feature ids for each of those
 127   buckets.
 128 
 129   Args:
 130     sorted_feature_columns: a list/set of tf.feature_column sorted by name.
 131 
 132   Returns:
 133     bucket_size_list: a list of required bucket sizes.
 134     feature_ids_list: a list of lists of feature ids for each bucket size.
 135 
 136   Raises:
 137     ValueError: when unsupported features columns are provided.
 138   """
 139   bucket_size_to_feature_ids_dict = collections.OrderedDict()
 140 
 141   # TODO(nponomareva) for now we preserve the previous functionality and bucket
 142   # all numeric into the same num of buckets. Can be easily changed to using
 143   # each numeric's real buckets num, but we need to test that it does not cause
 144   # a performance hit.
 145 
 146   # We will replace this dummy key with the real max after we calculate it.
 147   bucket_size_to_feature_ids_dict[_DUMMY_NUM_BUCKETS] = []
 148 
 149   max_buckets_for_bucketized = 2
 150   max_buckets_for_indicator = 2
 151 
 152   feature_idx = 0
 153   # pylint:disable=protected-access
 154 
 155   for column in sorted_feature_columns:
 156     if isinstance(column, feature_column_lib._IndicatorColumn):
 157       num_categorical_features = column.categorical_column._num_buckets
 158       if max_buckets_for_indicator not in bucket_size_to_feature_ids_dict:
 159         bucket_size_to_feature_ids_dict[max_buckets_for_indicator] = []
 160 
 161       for _ in range(num_categorical_features):
 162         # We use bucket size of 2 for categorical.
 163         bucket_size_to_feature_ids_dict[max_buckets_for_indicator].append(
 164             feature_idx)
 165         feature_idx += 1
 166     elif isinstance(column, feature_column_lib._BucketizedColumn):
 167       max_buckets_for_bucketized = max(max_buckets_for_bucketized,
 168                                        len(column.boundaries) + 1)
 169       bucket_size_to_feature_ids_dict[_DUMMY_NUM_BUCKETS].append(feature_idx)
 170       feature_idx += 1
 171     elif not isinstance(column, feature_column_lib._IndicatorColumn):  # pylint:disable=protected-access
 172       raise ValueError(
 173           'For now, only bucketized_column and indicator column are supported '
 174           'but got: {}'.format(column))
 175 
 176   # pylint:enable=protected-access
 177   # Replace the dummy key with the real max num of buckets for all bucketized
 178   # columns.
 179   if max_buckets_for_bucketized not in bucket_size_to_feature_ids_dict:
 180     bucket_size_to_feature_ids_dict[max_buckets_for_bucketized] = []
 181   bucket_size_to_feature_ids_dict[max_buckets_for_bucketized].extend(
 182       bucket_size_to_feature_ids_dict[_DUMMY_NUM_BUCKETS])
 183   del bucket_size_to_feature_ids_dict[_DUMMY_NUM_BUCKETS]
 184 
 185   feature_ids_list = list(bucket_size_to_feature_ids_dict.values())
 186   bucket_size_list = list(bucket_size_to_feature_ids_dict.keys())
 187   return bucket_size_list, feature_ids_list

 200 def _generate_feature_name_mapping(sorted_feature_columns):
 201   """Return a list of feature name for feature ids.
 202 
 203   Args:
 204     sorted_feature_columns: a list/set of tf.feature_column sorted by name.
 205 
 206   Returns:
 207     feature_name_mapping: a list of feature names indexed by the feature ids.
 208 
 209   Raises:
 210     ValueError: when unsupported features/columns are tried.
 211   """
 212   names = []
 213   for column in sorted_feature_columns:
 214     if isinstance(column, feature_column_lib._IndicatorColumn):  # pylint:disable=protected-access
 215       categorical_column = column.categorical_column
 216       if isinstance(categorical_column,
 217                     feature_column_lib._VocabularyListCategoricalColumn):  # pylint:disable=protected-access
 218         for value in categorical_column.vocabulary_list:
 219           names.append('{}:{}'.format(column.name, value))
 220       elif isinstance(categorical_column,
 221                       feature_column_lib._BucketizedColumn):  # pylint:disable=protected-access
 222         boundaries = [-np.inf] + list(categorical_column.boundaries) + [np.inf]
 223         for pair in zip(boundaries[:-1], boundaries[1:]):
 224           names.append('{}:{}'.format(column.name, pair))
 225       else:
 226         for num in range(categorical_column._num_buckets):  # pylint:disable=protected-access
 227           names.append('{}:{}'.format(column.name, num))
 228     elif isinstance(column, feature_column_lib._BucketizedColumn):
 229       names.append(column.name)
 230     else:
 231       raise ValueError(
 232           'For now, only bucketized_column and indicator_column is supported '
 233           'but got: {}'.format(column))
 234   return names

 237 def _cache_transformed_features(features, sorted_feature_columns, batch_size):
 238   """Transform features and cache, then returns (cached_features, cache_op)."""
 239   num_features = _calculate_num_features(sorted_feature_columns)
 240   cached_features = [
 241       _local_variable(
 242           array_ops.zeros([batch_size], dtype=dtypes.int32),
 243           name='cached_feature_{}'.format(i)) for i in range(num_features)
 244   ]
 245   are_features_cached = _local_variable(False, name='are_features_cached')
 246 
 247   def cache_features_and_return():
 248     """Caches transformed features.
 249 
 250     The intention is to hide get_transformed_features() from the graph by
 251     caching the result except the first step, since bucketize operation
 252     (inside get_transformed_features) is expensive.
 253 
 254     Returns:
 255       input_feature_list: a list of input features.
 256       cache_flip_op: op to add to graph to make sure cache update is included to
 257           the graph.
 258     """
 259 
 260     transformed_features = _get_transformed_features(features,
 261                                                      sorted_feature_columns)
 262     cached = [
 263         state_ops.assign(cached_features[i], transformed_features[i])
 264         for i in range(num_features)
 265     ]
 266     # TODO(youngheek): Try other combination of dependencies so that the
 267     # function returns a single result, not a tuple.
 268     with ops.control_dependencies(cached):
 269       cache_flip_op = are_features_cached.assign(True)
 270     return cached, cache_flip_op
 271 
 272   input_feature_list, cache_flip_op = control_flow_ops.cond(
 273       are_features_cached, lambda: (cached_features, control_flow_ops.no_op()),
 274       cache_features_and_return)
 275   return input_feature_list, cache_flip_op

 278 class _CacheTrainingStatesUsingHashTable(object):
 279   """Caching logits, etc. using MutableHashTable."""
 280 
 281   def __init__(self, example_ids, logits_dimension):
 282     """Creates a cache with the given configuration.
 283 
 284     It maintains a MutableDenseHashTable for all values.
 285     The API lookup() and insert() would have those specs,
 286       tree_ids: shape=[batch_size], dtype=int32
 287       node_ids: shape=[batch_size], dtype=int32
 288       logits: shape=[batch_size, logits_dimension], dtype=float32
 289     However in the MutableDenseHashTable, ids are bitcasted into float32 and
 290     all values are concatenated as a single tensor (of float32).
 291 
 292     Hence conversion happens internally before inserting to the HashTable and
 293     after lookup from it.
 294 
 295     Args:
 296       example_ids: a Rank 1 tensor to be used as a key of the cache.
 297       logits_dimension: a constant (int) for the dimension of logits.
 298 
 299     Raises:
 300       ValueError: if example_ids is other than int64 or string.
 301     """
 302     if dtypes.as_dtype(dtypes.int64).is_compatible_with(example_ids.dtype):
 303       empty_key = -1 << 62
 304     elif dtypes.as_dtype(dtypes.string).is_compatible_with(example_ids.dtype):
 305       empty_key = ''
 306     else:
 307       raise ValueError(
 308           'Unsupported example_id_feature dtype %s.' % example_ids.dtype)
 309     # Cache holds latest <tree_id, node_id, logits> for each example.
 310     # tree_id and node_id are both int32 but logits is a float32.
 311     # To reduce the overhead, we store all of them together as float32 and
 312     # bitcast the ids to int32.
 313     self._table_ref = lookup_ops.mutable_dense_hash_table_v2(
 314         empty_key=empty_key, value_dtype=dtypes.float32, value_shape=[3])
 315     self._example_ids = ops.convert_to_tensor(example_ids)
 316     if self._example_ids.shape.ndims not in (None, 1):
 317       raise ValueError(
 318           'example_id should have rank 1, but got %s' % self._example_ids)
 319     self._logits_dimension = logits_dimension
 320 
 321   def lookup(self):
 322     """Returns cached_tree_ids, cached_node_ids, cached_logits."""
 323     cached_tree_ids, cached_node_ids, cached_logits = array_ops.split(
 324         lookup_ops.lookup_table_find_v2(
 325             self._table_ref,
 326             self._example_ids,
 327             default_value=[0.0, _DUMMY_NODE_ID, 0.0]),
 328         [1, 1, self._logits_dimension],
 329         axis=1)
 330     cached_tree_ids = array_ops.squeeze(
 331         array_ops.bitcast(cached_tree_ids, dtypes.int32))
 332     cached_node_ids = array_ops.squeeze(
 333         array_ops.bitcast(cached_node_ids, dtypes.int32))
 334     if self._example_ids.shape.ndims is not None:
 335       cached_logits.set_shape(
 336           [self._example_ids.shape[0], self._logits_dimension])
 337     return (cached_tree_ids, cached_node_ids, cached_logits)
 338 
 339   def insert(self, tree_ids, node_ids, logits):
 340     """Inserts values and returns the op."""
 341     insert_op = lookup_ops.lookup_table_insert_v2(
 342         self._table_ref, self._example_ids,
 343         array_ops.concat(
 344             [
 345                 array_ops.expand_dims(
 346                     array_ops.bitcast(tree_ids, dtypes.float32), 1),
 347                 array_ops.expand_dims(
 348                     array_ops.bitcast(node_ids, dtypes.float32), 1),
 349                 logits,
 350             ],
 351             axis=1,
 352             name='value_concat_for_cache_insert'))
 353     return insert_op

 356 class _CacheTrainingStatesUsingVariables(object):
 357   """Caching logits, etc. using Variables."""
 358 
 359   def __init__(self, batch_size, logits_dimension):
 360     """Creates a cache with the given configuration.
 361 
 362     It maintains three variables, tree_ids, node_ids, logits, for caching.
 363       tree_ids: shape=[batch_size], dtype=int32
 364       node_ids: shape=[batch_size], dtype=int32
 365       logits: shape=[batch_size, logits_dimension], dtype=float32
 366 
 367     Note, this can be used only with in-memory data setting.
 368 
 369     Args:
 370       batch_size: `int`, the size of the cache.
 371       logits_dimension: a constant (int) for the dimension of logits.
 372     """
 373     self._logits_dimension = logits_dimension
 374     self._tree_ids = _local_variable(
 375         array_ops.zeros([batch_size], dtype=dtypes.int32),
 376         name='tree_ids_cache')
 377     self._node_ids = _local_variable(
 378         _DUMMY_NODE_ID * array_ops.ones([batch_size], dtype=dtypes.int32),
 379         name='node_ids_cache')
 380     self._logits = _local_variable(
 381         array_ops.zeros([batch_size, logits_dimension], dtype=dtypes.float32),
 382         name='logits_cache')
 383 
 384   def lookup(self):
 385     """Returns cached_tree_ids, cached_node_ids, cached_logits."""
 386     return (self._tree_ids, self._node_ids, self._logits)
 387 
 388   def insert(self, tree_ids, node_ids, logits):
 389     """Inserts values and returns the op."""
 390     return control_flow_ops.group(
 391         [
 392             self._tree_ids.assign(tree_ids),
 393             self._node_ids.assign(node_ids),
 394             self._logits.assign(logits)
 395         ],
 396         name='cache_insert')

 399 class _StopAtAttemptsHook(session_run_hook.SessionRunHook):
 400   """Hook that requests stop at the number of attempts."""
 401 
 402   def __init__(self, num_finalized_trees_tensor, num_attempted_layers_tensor,
 403                max_trees, max_depth):
 404     self._num_finalized_trees_tensor = num_finalized_trees_tensor
 405     self._num_attempted_layers_tensor = num_attempted_layers_tensor
 406     self._max_trees = max_trees
 407     self._max_depth = max_depth
 408 
 409   def before_run(self, run_context):
 410     return session_run_hook.SessionRunArgs(
 411         [self._num_finalized_trees_tensor, self._num_attempted_layers_tensor])
 412 
 413   def after_run(self, run_context, run_values):
 414     # num_* tensors should be retrieved by a separate session than the training
 415     # one, in order to read the values after growing.
 416     # So, if it's approaching to the limit, get the actual value by additional
 417     # session.
 418     num_finalized_trees, num_attempted_layers = run_values.results
 419     if (num_finalized_trees >= self._max_trees - 1 or
 420         num_attempted_layers > 2 * self._max_trees * self._max_depth - 1):
 421       num_finalized_trees, num_attempted_layers = run_context.session.run(
 422           [self._num_finalized_trees_tensor, self._num_attempted_layers_tensor])
 423     if (num_finalized_trees >= self._max_trees or
 424         num_attempted_layers > 2 * self._max_trees * self._max_depth):
 425       run_context.request_stop()

 428 def _get_max_splits(tree_hparams):
 429   """Calculates the max possible number of splits based on tree params."""

 519   def _grow_tree_from_stats_summaries(self, stats_summaries_list,

 521     """Updates ensemble based on the best gains from stats summaries."""

 567 class _InMemoryEnsembleGrower(_EnsembleGrower):
 568   """An in-memory ensemble grower."""

 570   def __init__(self, tree_ensemble, tree_hparams, feature_ids_list):

 576   def center_bias(self, center_bias_var, gradients, hessians):
 577     # For in memory, we already have a full batch of gradients and hessians,
 578     # so just take a mean and proceed with centering.
 579     mean_gradients = array_ops.expand_dims(
 580         math_ops.reduce_mean(gradients, 0), 0)
 581     mean_heassians = array_ops.expand_dims(math_ops.reduce_mean(hessians, 0), 0)

 587     return self._grow_tree_from_stats_summaries(
 588         stats_summaries_list, last_layer_nodes_range)

 591 class _AccumulatorEnsembleGrower(_EnsembleGrower):
 592   """An accumulator based ensemble grower."""
 593 
 594   def __init__(self, tree_ensemble, tree_hparams, stamp_token,
 595                n_batches_per_layer, bucket_size_list, is_chief, center_bias,
 596                feature_ids_list):
 597     super(_AccumulatorEnsembleGrower, self).__init__(
 598         tree_ensemble=tree_ensemble, tree_hparams=tree_hparams,
 599         feature_ids_list=feature_ids_list)
 600     self._stamp_token = stamp_token
 601     self._n_batches_per_layer = n_batches_per_layer
 602     self._bucket_size_list = bucket_size_list
 603     self._is_chief = is_chief
 604     self._growing_accumulators = []
 605     self._chief_init_ops = []
 606     max_splits = _get_max_splits(self._tree_hparams)
 607     for i, feature_ids in enumerate(self._feature_ids_list):
 608       accumulator = data_flow_ops.ConditionalAccumulator(
 609           dtype=dtypes.float32,
 610           # The stats consist of grads and hessians (the last dimension).
 611           shape=[len(feature_ids), max_splits, self._bucket_size_list[i], 2],
 612           shared_name='numeric_stats_summary_accumulator_' + str(i))
 613       self._chief_init_ops.append(
 614           accumulator.set_global_step(self._stamp_token))
 615       self._growing_accumulators.append(accumulator)
 616     self._center_bias = center_bias
 617     if center_bias:
 618       self._bias_accumulator = data_flow_ops.ConditionalAccumulator(
 619           dtype=dtypes.float32,
 620           # The stats consist of grads and hessians means only.
 621           # TODO(nponomareva): this will change for a multiclass
 622           shape=[2, 1],
 623           shared_name='bias_accumulator')
 624       self._chief_init_ops.append(
 625           self._bias_accumulator.set_global_step(self._stamp_token))
 626 
 627   def center_bias(self, center_bias_var, gradients, hessians):
 628     # For not in memory situation, we need to accumulate enough of batches first
 629     # before proceeding with centering bias.
 630 
 631     # Create an accumulator.
 632     if not self._center_bias:
 633       raise RuntimeError('center_bias called but bias centering is disabled.')
 634     bias_dependencies = []
 635     grads_and_hess = array_ops.stack([gradients, hessians], axis=0)
 636     grads_and_hess = math_ops.reduce_mean(grads_and_hess, axis=1)
 637 
 638     apply_grad = self._bias_accumulator.apply_grad(
 639         grads_and_hess, self._stamp_token)
 640     bias_dependencies.append(apply_grad)
 641 
 642     # Center bias if enough batches were processed.
 643     with ops.control_dependencies(bias_dependencies):
 644       if not self._is_chief:
 645         return control_flow_ops.no_op()
 646       def _set_accumulators_stamp():
 647         return control_flow_ops.group(
 648             [acc.set_global_step(self._stamp_token + 1) for acc in
 649              self._growing_accumulators])
 650 
 651       def center_bias_from_accumulator():
 652         accumulated = array_ops.unstack(self._bias_accumulator.take_grad(1),
 653                                         axis=0)
 654         center_bias_op = self._center_bias_fn(
 655             center_bias_var,
 656             array_ops.expand_dims(accumulated[0], 0),
 657             array_ops.expand_dims(accumulated[1], 0))
 658         with ops.control_dependencies([center_bias_op]):
 659           return control_flow_ops.cond(center_bias_var,
 660                                        control_flow_ops.no_op,
 661                                        _set_accumulators_stamp)
 662 
 663       center_bias_op = control_flow_ops.cond(
 664           math_ops.greater_equal(self._bias_accumulator.num_accumulated(),
 665                                  self._n_batches_per_layer),
 666           center_bias_from_accumulator,
 667           control_flow_ops.no_op,
 668           name='wait_until_n_batches_for_bias_accumulated')
 669       return center_bias_op
 670 
 671   def grow_tree(self, stats_summaries_list, last_layer_nodes_range):
 672     dependencies = []
 673     for i in range(len(self._feature_ids_list)):
 674       stats_summaries = stats_summaries_list[i]
 675       apply_grad = self._growing_accumulators[i].apply_grad(
 676           array_ops.stack(stats_summaries, axis=0), self._stamp_token)
 677       dependencies.append(apply_grad)
 678 
 679     # Grow the tree if enough batches is accumulated.
 680     with ops.control_dependencies(dependencies):
 681       if not self._is_chief:
 682         return control_flow_ops.no_op()
 683 
 684       min_accumulated = math_ops.reduce_min(
 685           array_ops.stack([acc.num_accumulated() for acc in
 686                            self._growing_accumulators]))
 687 
 688       def grow_tree_from_accumulated_summaries_fn():
 689         """Updates tree with the best layer from accumulated summaries."""
 690         # Take out the accumulated summaries from the accumulator and grow.
 691         stats_summaries_list = []
 692         stats_summaries_list = [
 693             array_ops.unstack(accumulator.take_grad(1), axis=0)
 694             for accumulator in self._growing_accumulators
 695         ]
 696         grow_op = self._grow_tree_from_stats_summaries(
 697             stats_summaries_list, last_layer_nodes_range
 698         )
 699         return grow_op
 700 
 701       grow_model = control_flow_ops.cond(
 702           math_ops.greater_equal(min_accumulated, self._n_batches_per_layer),
 703           grow_tree_from_accumulated_summaries_fn,
 704           control_flow_ops.no_op,
 705           name='wait_until_n_batches_accumulated')
 706       return grow_model
 707 
 708   def chief_init_op(self):
 709     """Ops that chief needs to run to initialize the state."""
 710     return control_flow_ops.group(self._chief_init_ops)

 713 def _bt_model_fn(
 714     features,
 715     labels,
 716     mode,
 717     head,
 718     feature_columns,
 719     tree_hparams,
 720     n_batches_per_layer,
 721     config,
 722     closed_form_grad_and_hess_fn=None,
 723     example_id_column_name=None,
 724     # TODO(youngheek): replace this later using other options.
 725     train_in_memory=False,
 726     name='boosted_trees'):
 727   """Gradient Boosted Trees model_fn.
 728 
 729   Args:
 730     features: dict of `Tensor`.
 731     labels: `Tensor` of shape [batch_size, 1] or [batch_size] labels of
 732       dtype `int32` or `int64` in the range `[0, n_classes)`.
 733     mode: Defines whether this is training, evaluation or prediction.
 734       See `ModeKeys`.
 735     head: A `head_lib._Head` instance.
 736     feature_columns: Iterable of `feature_column._FeatureColumn` model inputs.
 737     tree_hparams: TODO. collections.namedtuple for hyper parameters.
 738     n_batches_per_layer: A `Tensor` of `int64`. Each layer is built after at
 739       least n_batches_per_layer accumulations.
 740     config: `RunConfig` object to configure the runtime settings.
 741     closed_form_grad_and_hess_fn: a function that accepts logits and labels
 742       and returns gradients and hessians. By default, they are created by
 743       tf.gradients() from the loss.
 744     example_id_column_name: Name of the feature for a unique ID per example.
 745       Currently experimental -- not exposed to public API.
 746     train_in_memory: `bool`, when true, it assumes the dataset is in memory,
 747       i.e., input_fn should return the entire dataset as a single batch, and
 748       also n_batches_per_layer should be set as 1.
 749     name: Name to use for the model.
 750 
 751   Returns:
 752       An `EstimatorSpec` instance.
 753 
 754   Raises:
 755     ValueError: mode or params are invalid, or features has the wrong type.
 756   """
 757   sorted_feature_columns = sorted(feature_columns, key=lambda tc: tc.name)
 758   with ops.name_scope(name) as name:
 759     # Prepare.
 760     global_step = training_util.get_or_create_global_step()
 761     bucket_size_list, feature_ids_list = _group_features_by_num_buckets(
 762         sorted_feature_columns)
 763     # Create Ensemble resources.
 764     tree_ensemble = boosted_trees_ops.TreeEnsemble(name=name)
 765 
 766     # Create logits.
 767     if mode != model_fn_lib.ModeKeys.TRAIN:
 768       input_feature_list = _get_transformed_features(features,
 769                                                      sorted_feature_columns)
 770       logits = boosted_trees_ops.predict(
 771           # For non-TRAIN mode, ensemble doesn't change after initialization,
 772           # so no local copy is needed; using tree_ensemble directly.
 773           tree_ensemble_handle=tree_ensemble.resource_handle,
 774           bucketized_features=input_feature_list,
 775           logits_dimension=head.logits_dimension)
 776       return head.create_estimator_spec(
 777           features=features,
 778           mode=mode,
 779           labels=labels,
 780           train_op_fn=control_flow_ops.no_op,
 781           logits=logits)
 782 
 783     # ============== Training graph ==============
 784     center_bias = tree_hparams.center_bias
 785     is_single_machine = (config.num_worker_replicas <= 1)
 786 
 787     if train_in_memory:
 788       assert n_batches_per_layer == 1, (
 789           'When train_in_memory is enabled, input_fn should return the entire '
 790           'dataset as a single batch, and n_batches_per_layer should be set as '
 791           '1.')
 792       if (not config.is_chief or config.num_worker_replicas > 1 or
 793           config.num_ps_replicas > 0):
 794         raise ValueError('train_in_memory is supported only for '
 795                          'non-distributed training.')
 796     worker_device = control_flow_ops.no_op().device
 797     train_op = []
 798     # Extract input features and set up cache for training.
 799     training_state_cache = None
 800     if train_in_memory:
 801       # cache transformed features as well for in-memory training.
 802       batch_size = array_ops.shape(labels)[0]
 803       input_feature_list, input_cache_op = (
 804           _cache_transformed_features(features, sorted_feature_columns,
 805                                       batch_size))
 806       train_op.append(input_cache_op)
 807       training_state_cache = _CacheTrainingStatesUsingVariables(
 808           batch_size, head.logits_dimension)
 809     else:
 810       input_feature_list = _get_transformed_features(features,
 811                                                      sorted_feature_columns)
 812       if example_id_column_name:
 813         example_ids = features[example_id_column_name]
 814         training_state_cache = _CacheTrainingStatesUsingHashTable(
 815             example_ids, head.logits_dimension)
 816     if training_state_cache:
 817       cached_tree_ids, cached_node_ids, cached_logits = (
 818           training_state_cache.lookup())
 819     else:
 820       # Always start from the beginning when no cache is set up.
 821       batch_size = array_ops.shape(labels)[0]
 822       cached_tree_ids, cached_node_ids, cached_logits = (
 823           array_ops.zeros([batch_size], dtype=dtypes.int32),
 824           _DUMMY_NODE_ID * array_ops.ones([batch_size], dtype=dtypes.int32),
 825           array_ops.zeros(
 826               [batch_size, head.logits_dimension], dtype=dtypes.float32))
 827 
 828     if is_single_machine:
 829       local_tree_ensemble = tree_ensemble
 830       ensemble_reload = control_flow_ops.no_op()
 831     else:
 832       # Have a local copy of ensemble for the distributed setting.
 833       with ops.device(worker_device):
 834         local_tree_ensemble = boosted_trees_ops.TreeEnsemble(
 835             name=name + '_local', is_local=True)
 836       # TODO(soroush): Do partial updates if this becomes a bottleneck.
 837       ensemble_reload = local_tree_ensemble.deserialize(
 838           *tree_ensemble.serialize())
 839     with ops.control_dependencies([ensemble_reload]):
 840       (stamp_token, num_trees, num_finalized_trees, num_attempted_layers,
 841        last_layer_nodes_range) = local_tree_ensemble.get_states()
 842       partial_logits, tree_ids, node_ids = boosted_trees_ops.training_predict(
 843           tree_ensemble_handle=local_tree_ensemble.resource_handle,
 844           cached_tree_ids=cached_tree_ids,
 845           cached_node_ids=cached_node_ids,
 846           bucketized_features=input_feature_list,
 847           logits_dimension=head.logits_dimension)
 848     logits = cached_logits + partial_logits
 849 
 850     if train_in_memory:
 851       grower = _InMemoryEnsembleGrower(tree_ensemble, tree_hparams,
 852                                        feature_ids_list=feature_ids_list)
 853     else:
 854       grower = _AccumulatorEnsembleGrower(tree_ensemble, tree_hparams,
 855                                           stamp_token, n_batches_per_layer,
 856                                           bucket_size_list, config.is_chief,
 857                                           center_bias=center_bias,
 858                                           feature_ids_list=feature_ids_list)
 859 
 860     summary.scalar('ensemble/num_trees', num_trees)
 861     summary.scalar('ensemble/num_finalized_trees', num_finalized_trees)
 862     summary.scalar('ensemble/num_attempted_layers', num_attempted_layers)
 863 
 864     # Variable that determines whether bias centering is needed.
 865     center_bias_var = variable_scope.variable(
 866         initial_value=center_bias, name='center_bias_needed', trainable=False,
 867         use_resource=True)
 868     # Create training graph.
 869     def _train_op_fn(loss):
 870       """Run one training iteration."""
 871       if training_state_cache:
 872         # Cache logits only after center_bias is complete, if it's in progress.
 873         train_op.append(
 874             control_flow_ops.cond(
 875                 center_bias_var, control_flow_ops.no_op,
 876                 lambda: training_state_cache.insert(tree_ids, node_ids, logits))
 877         )
 878 
 879       if closed_form_grad_and_hess_fn:
 880         gradients, hessians = closed_form_grad_and_hess_fn(logits, labels)
 881       else:
 882         gradients = gradients_impl.gradients(loss, logits, name='Gradients')[0]
 883         hessians = gradients_impl.gradients(
 884             gradients, logits, name='Hessians')[0]
 885 
 886       # TODO(youngheek): perhaps storage could be optimized by storing stats
 887       # with the dimension max_splits_per_layer, instead of max_splits (for the
 888       # entire tree).
 889       max_splits = _get_max_splits(tree_hparams)
 890 
 891       stats_summaries_list = []
 892       for i, feature_ids in enumerate(feature_ids_list):
 893         num_buckets = bucket_size_list[i]
 894         summaries = [
 895             array_ops.squeeze(
 896                 boosted_trees_ops.make_stats_summary(
 897                     node_ids=node_ids,
 898                     gradients=gradients,
 899                     hessians=hessians,
 900                     bucketized_features_list=[input_feature_list[f]],
 901                     max_splits=max_splits,
 902                     num_buckets=num_buckets),
 903                 axis=0) for f in feature_ids
 904         ]
 905         stats_summaries_list.append(summaries)
 906       if center_bias:
 907         update_model = control_flow_ops.cond(
 908             center_bias_var,
 909             functools.partial(
 910                 grower.center_bias,
 911                 center_bias_var,
 912                 gradients,
 913                 hessians,
 914             ),
 915             functools.partial(grower.grow_tree, stats_summaries_list,
 916                               last_layer_nodes_range))
 917       else:
 918         update_model = grower.grow_tree(stats_summaries_list,
 919                                         last_layer_nodes_range)
 920       train_op.append(update_model)
 921 
 922       with ops.control_dependencies([update_model]):
 923         increment_global = state_ops.assign_add(global_step, 1).op
 924         train_op.append(increment_global)
 925 
 926       return control_flow_ops.group(train_op, name='train_op')
 927 
 928   estimator_spec = head.create_estimator_spec(
 929       features=features,
 930       mode=mode,
 931       labels=labels,
 932       train_op_fn=_train_op_fn,
 933       logits=logits)
 934 
 935   # Add an early stop hook.
 936   estimator_spec = estimator_spec._replace(
 937       training_hooks=estimator_spec.training_hooks +
 938       (_StopAtAttemptsHook(num_finalized_trees, num_attempted_layers,
 939                            tree_hparams.n_trees, tree_hparams.max_depth),),
 940       training_chief_hooks=[GrowerInitializationHook(grower.chief_init_op())] +
 941       list(estimator_spec.training_chief_hooks))
 942   return estimator_spec

 972 def _create_classification_head_and_closed_form(n_classes, weight_column,
 973                                                 label_vocabulary):

 975   head = _create_classification_head(n_classes, weight_column, label_vocabulary)
 976   if (n_classes == 2 and head.logits_dimension == 1 and
 977       weight_column is None and label_vocabulary is None):

 980       """A closed form gradient and hessian for logistic loss."""
 981       # TODO(youngheek): add weights handling.
 982       predictions = math_ops.reciprocal(math_ops.exp(-logits) + 1.0)

 984           math_ops.cast(array_ops.size(predictions), dtypes.float32))

 986       labels = head_lib._check_dense_labels_match_logits_and_reshape(  # pylint: disable=protected-access
 987           labels, logits, head.logits_dimension)
 988       gradients = (predictions - labels) * normalizer
 989       hessians = predictions * (1.0 - predictions) * normalizer
 990       return gradients, hessians
 991 
 992     closed_form = _grad_and_hess_for_logloss
 993   else:
 994     closed_form = None
 995   return (head, closed_form)

1010 def _compute_feature_importances_per_tree(tree, num_features):
1011   """Computes the importance of each feature in the tree."""
1012   importances = np.zeros(num_features)
1013 
1014   for node in tree.nodes:
1015     node_type = node.WhichOneof('node')
1016     if node_type == 'bucketized_split':
1017       feature_id = node.bucketized_split.feature_id
1018       importances[feature_id] += node.metadata.gain
1019     elif node_type == 'leaf':
1020       assert node.metadata.gain == 0
1021     else:
1022       raise ValueError('Unexpected split type %s', node_type)
1023 
1024   return importances

1027 def _compute_feature_importances(tree_ensemble, num_features, normalize):
1028   """Computes gain-based feature importances.
1029 
1030   The higher the value, the more important the feature.
1031 
1032   Args:
1033     tree_ensemble: a trained tree ensemble, instance of proto
1034       boosted_trees.TreeEnsemble.
1035     num_features: The total number of feature ids.
1036     normalize: If True, normalize the feature importances.
1037 
1038   Returns:
1039     sorted_feature_idx: A list of feature_id which is sorted
1040       by its feature importance.
1041     feature_importances: A list of corresponding feature importances.
1042 
1043   Raises:
1044     AssertionError: When normalize = True, if feature importances
1045       contain negative value, or if normalization is not possible
1046       (e.g. ensemble is empty or trees contain only a root node).
1047   """
1048   tree_importances = [_compute_feature_importances_per_tree(tree, num_features)
1049                       for tree in tree_ensemble.trees]
1050   tree_importances = np.array(tree_importances)
1051   tree_weights = np.array(tree_ensemble.tree_weights).reshape(-1, 1)
1052   feature_importances = np.sum(tree_importances * tree_weights, axis=0)
1053   if normalize:
1054     assert np.all(feature_importances >= 0), ('feature_importances '
1055                                               'must be non-negative.')
1056     normalizer = np.sum(feature_importances)

1186   def experimental_predict_with_explanations(self,
1187                                              input_fn,
1188                                              predict_keys=None,
1189                                              hooks=None,
1190                                              checkpoint_path=None):
1191     """Computes model explainability outputs per example along with predictions.
1192 
1193     Currently supports directional feature contributions (DFCs). For each
1194     instance, DFCs indicate the aggregate contribution of each feature. See
1195     https://arxiv.org/abs/1312.1121 and
1196     http://blog.datadive.net/interpreting-random-forests/ for more details.
1197     Args:
1198       input_fn: A function that provides input data for predicting as
1199         minibatches. See [Premade Estimators](
1200         https://tensorflow.org/guide/premade_estimators#create_input_functions)
1201           for more information. The function should construct and return one of
1202         the following:  * A `tf.data.Dataset` object: Outputs of `Dataset`
1203           object must be a tuple `(features, labels)` with same constraints as
1204         below. * A tuple `(features, labels)`: Where `features` is a `tf.Tensor`
1205           or a dictionary of string feature name to `Tensor` and `labels` is a
1206           `Tensor` or a dictionary of string label name to `Tensor`. Both
1207           `features` and `labels` are consumed by `model_fn`. They should
1208           satisfy the expectation of `model_fn` from inputs.
1209       predict_keys: list of `str`, name of the keys to predict. It is used if
1210         the `tf.estimator.EstimatorSpec.predictions` is a `dict`. If
1211         `predict_keys` is used then rest of the predictions will be filtered
1212         from the dictionary, with the exception of 'bias' and 'dfc', which will
1213         always be in the dictionary. If `None`, returns all keys in prediction
1214         dict, as well as two new keys 'dfc' and 'bias'.
1215       hooks: List of `tf.train.SessionRunHook` subclass instances. Used for
1216         callbacks inside the prediction call.
1217       checkpoint_path: Path of a specific checkpoint to predict. If `None`, the
1218         latest checkpoint in `model_dir` is used.  If there are no checkpoints
1219         in `model_dir`, prediction is run with newly initialized `Variables`
1220         instead of ones restored from checkpoint.
1221 
1222     Yields:
1223       Evaluated values of `predictions` tensors. The `predictions` tensors will
1224       contain at least two keys 'dfc' and 'bias' for model explanations. The
1225       `dfc` value corresponds to the contribution of each feature to the overall
1226       prediction for this instance (positive indicating that the feature makes
1227       it more likely to select class 1 and negative less likely). The 'bias'
1228       value will be the same across all the instances, corresponding to the
1229       probability (classification) or prediction (regression) of the training
1230       data distribution.
1231 
1232     Raises:
1233       ValueError: when wrong arguments are given or unsupported functionalities
1234        are requested.
1235     """
1236     if not self._center_bias:
1237       raise ValueError('center_bias must be enabled during estimator '
1238                        'instantiation when using '
1239                        'experimental_predict_with_explanations.')
1240     # pylint: disable=protected-access
1241     if not self._is_classification:
1242       identity_inverse_link_fn = self._head._inverse_link_fn in (None,
1243                                                                  tf_identity)
1244       # pylint:enable=protected-access
1245       if not identity_inverse_link_fn:
1246         raise ValueError(
1247             'For now only identity inverse_link_fn in regression_head is '
1248             'supported for experimental_predict_with_explanations.')
1249 
1250     # pylint:disable=unused-argument
1251     def new_model_fn(features, labels, mode):
1252       return _bt_explanations_fn(features, self._head,
1253                                  self._sorted_feature_columns)
1254 
1255     # pylint:enable=unused-argument
1256     est = estimator.Estimator(
1257         model_fn=new_model_fn,
1258         model_dir=self.model_dir,
1259         config=self.config,
1260         warm_start_from=self._warm_start_settings)
1261     # Make sure bias and dfc will be in prediction dict.
1262     user_supplied_predict_keys = predict_keys is not None
1263     if user_supplied_predict_keys:
1264       predict_keys = set(predict_keys)
1265       predict_keys.add(boosted_trees_utils._DEBUG_PROTO_KEY)
1266     predictions = est.predict(
1267         input_fn,
1268         predict_keys=predict_keys,
1269         hooks=hooks,
1270         checkpoint_path=checkpoint_path,
1271         yield_single_examples=True)
1272     for pred in predictions:
1273       bias, dfcs = boosted_trees_utils._parse_explanations_from_prediction(
1274           pred[boosted_trees_utils._DEBUG_PROTO_KEY], self._n_features,
1275           self._is_classification)
1276       pred['bias'] = bias
1277       pred['dfc'] = dfcs
1278       # Don't need to expose serialized proto to end user.
1279       del pred[boosted_trees_utils._DEBUG_PROTO_KEY]
1280       yield pred
