  40 class ModeKeys(object):

  42 
  43   The following standard keys are defined:

  46   * `EVAL`: evaluation mode.
  47   * `PREDICT`: inference mode.

  52   PREDICT = 'infer'

  56 AVERAGE_LOSS_METRIC_KEY = 'average_loss'

  66 @estimator_export('estimator.EstimatorSpec')
  67 class EstimatorSpec(
  68     collections.namedtuple('EstimatorSpec', [
  69         'mode', 'predictions', 'loss', 'train_op', 'eval_metric_ops',
  70         'export_outputs', 'training_chief_hooks', 'training_hooks', 'scaffold',
  71         'evaluation_hooks', 'prediction_hooks'
  72     ])):
  73   """Ops and objects returned from a `model_fn` and passed to an `Estimator`.
  74 
  75   `EstimatorSpec` fully defines the model to be run by an `Estimator`.
  76   """
  77 
  78   def __new__(cls,
  79               mode,
  80               predictions=None,
  81               loss=None,
  82               train_op=None,
  83               eval_metric_ops=None,
  84               export_outputs=None,
  85               training_chief_hooks=None,
  86               training_hooks=None,
  87               scaffold=None,
  88               evaluation_hooks=None,
  89               prediction_hooks=None):
  90     """Creates a validated `EstimatorSpec` instance.
  91 
  92     Depending on the value of `mode`, different arguments are required. Namely
  93 
  94     * For `mode == ModeKeys.TRAIN`: required fields are `loss` and `train_op`.
  95     * For `mode == ModeKeys.EVAL`: required field is `loss`.
  96     * For `mode == ModeKeys.PREDICT`: required fields are `predictions`.
  97 
  98     model_fn can populate all arguments independent of mode. In this case, some
  99     arguments will be ignored by an `Estimator`. E.g. `train_op` will be
 100     ignored in eval and infer modes. Example:
 101 
 102     ```python
 103     def my_model_fn(features, labels, mode):
 104       predictions = ...
 105       loss = ...
 106       train_op = ...
 107       return tf.estimator.EstimatorSpec(
 108           mode=mode,
 109           predictions=predictions,
 110           loss=loss,
 111           train_op=train_op)
 112     ```
 113 
 114     Alternatively, model_fn can just populate the arguments appropriate to the
 115     given mode. Example:
 116 
 117     ```python
 118     def my_model_fn(features, labels, mode):
 119       if (mode == tf.estimator.ModeKeys.TRAIN or
 120           mode == tf.estimator.ModeKeys.EVAL):
 121         loss = ...
 122       else:
 123         loss = None
 124       if mode == tf.estimator.ModeKeys.TRAIN:
 125         train_op = ...
 126       else:
 127         train_op = None
 128       if mode == tf.estimator.ModeKeys.PREDICT:
 129         predictions = ...
 130       else:
 131         predictions = None
 132 
 133       return tf.estimator.EstimatorSpec(
 134           mode=mode,
 135           predictions=predictions,
 136           loss=loss,
 137           train_op=train_op)
 138     ```
 139 
 140     Args:
 141       mode: A `ModeKeys`. Specifies if this is training, evaluation or
 142         prediction.
 143       predictions: Predictions `Tensor` or dict of `Tensor`.
 144       loss: Training loss `Tensor`. Must be either scalar, or with shape `[1]`.
 145       train_op: Op for the training step.
 146       eval_metric_ops: Dict of metric results keyed by name.
 147         The values of the dict can be one of the following:
 148         (1) instance of `Metric` class.
 149         (2) Results of calling a metric function, namely a
 150         `(metric_tensor, update_op)` tuple. `metric_tensor` should be
 151         evaluated without any impact on state (typically is a pure computation
 152         results based on variables.). For example, it should not trigger the
 153         `update_op` or requires any input fetching.
 154       export_outputs: Describes the output signatures to be exported to
 155         `SavedModel` and used during serving.
 156         A dict `{name: output}` where:
 157         * name: An arbitrary name for this output.
 158         * output: an `ExportOutput` object such as `ClassificationOutput`,
 159             `RegressionOutput`, or `PredictOutput`.
 160         Single-headed models only need to specify one entry in this dictionary.
 161         Multi-headed models should specify one entry for each head, one of
 162         which must be named using
 163         signature_constants.DEFAULT_SERVING_SIGNATURE_DEF_KEY.
 164         If no entry is provided, a default `PredictOutput` mapping to
 165         `predictions` will be created.
 166       training_chief_hooks: Iterable of `tf.train.SessionRunHook` objects to
 167         run on the chief worker during training.
 168       training_hooks: Iterable of `tf.train.SessionRunHook` objects to run
 169         on all workers during training.
 170       scaffold: A `tf.train.Scaffold` object that can be used to set
 171         initialization, saver, and more to be used in training.
 172       evaluation_hooks: Iterable of `tf.train.SessionRunHook` objects to
 173         run during evaluation.
 174       prediction_hooks: Iterable of `tf.train.SessionRunHook` objects to
 175         run during predictions.
 176 
 177     Returns:
 178       A validated `EstimatorSpec` object.
 179 
 180     Raises:
 181       ValueError: If validation fails.
 182       TypeError: If any of the arguments is not the expected type.
 183     """
 184     # Validate train_op.
 185     if train_op is None:
 186       if mode == ModeKeys.TRAIN:
 187         raise ValueError('Missing train_op.')
 188     else:
 189       _check_is_tensor_or_operation(train_op, 'train_op')
 190 
 191     # Validate loss.
 192     if loss is None:
 193       if mode in (ModeKeys.TRAIN, ModeKeys.EVAL):
 194         raise ValueError('Missing loss.')
 195     else:
 196       loss = _check_is_tensor(loss, 'loss')
 197       loss_shape = loss.get_shape()
 198       if loss_shape.num_elements() not in (None, 1):
 199         raise ValueError('Loss must be scalar, given: {}'.format(loss))
 200       if not loss_shape.is_compatible_with(tensor_shape.scalar()):
 201         loss = array_ops.reshape(loss, [])
 202 
 203     # Validate predictions.
 204     if predictions is None:
 205       if mode == ModeKeys.PREDICT:
 206         raise ValueError('Missing predictions.')
 207       predictions = {}
 208     else:
 209       if isinstance(predictions, dict):
 210         predictions = {
 211             k: _check_is_tensor(v, 'predictions[{}]'.format(k))
 212             for k, v in six.iteritems(predictions)
 213         }
 214       else:
 215         predictions = _check_is_tensor(predictions, 'predictions')
 216 
 217     # Validate eval_metric_ops.
 218     if eval_metric_ops is None:
 219       eval_metric_ops = {}
 220     else:
 221       if not isinstance(eval_metric_ops, dict):
 222         raise TypeError(
 223             'eval_metric_ops must be a dict, given: {}'.format(eval_metric_ops))
 224       for key, value in six.iteritems(eval_metric_ops):
 225         # TODO(psv): When we deprecate the old metrics, throw an error here if
 226         # the value is not an instance of `Metric` class.
 227         if isinstance(value, Metric):
 228           if not value.updates:  # Check if metrics updates are available.
 229             raise ValueError(
 230                 'Please call update_state(...) on the "{metric_name}" metric'
 231                 .format(metric_name=value.name))
 232         else:
 233           if not isinstance(value, tuple) or len(value) != 2:
 234             raise TypeError(
 235                 'Values of eval_metric_ops must be (metric_value, update_op) '
 236                 'tuples, given: {} for key: {}'.format(value, key))
 237           metric_value, metric_update = value
 238           for metric_value_member in nest.flatten(metric_value):
 239             # Allow (possibly nested) tuples for metric values, but require that
 240             # each of them be Tensors or Operations.
 241             _check_is_tensor_or_operation(metric_value_member,
 242                                           'eval_metric_ops[{}]'.format(key))
 243           _check_is_tensor_or_operation(metric_update,
 244                                         'eval_metric_ops[{}]'.format(key))
 245 
 246     # Validate the passed export outputs, or generate defaults.
 247     if mode == ModeKeys.PREDICT:
 248       export_outputs = _get_export_outputs(export_outputs, predictions)
 249 
 250     # Validate that all tensors and ops are from the default graph.
 251     default_graph = ops.get_default_graph()
 252 
 253     # We enumerate possible error causes here to aid in debugging.
 254     error_message_template = (
 255         '{0} with "{1}" must be from the default graph. '
 256         'Possible causes of this error include: \n\n'
 257         '1) {0} was created outside the context of the default graph.'
 258         '\n\n'
 259         '2) The object passed through to EstimatorSpec was not created '
 260         'in the most recent call to "model_fn".')
 261 
 262     if isinstance(predictions, dict):
 263       for key, value in six.iteritems(predictions):
 264         if value.graph is not default_graph:
 265           raise ValueError(error_message_template.format(
 266               'prediction values',
 267               '{0}: {1}'.format(key, value.name)))
 268     elif predictions is not None:
 269       # 'predictions' must be a single Tensor.
 270       if predictions.graph is not default_graph:
 271         raise ValueError(error_message_template.format(
 272             'prediction values', predictions.name))
 273 
 274     if loss is not None and loss.graph is not default_graph:
 275       raise ValueError(error_message_template.format('loss', loss.name))
 276     if train_op is not None and train_op.graph is not default_graph:
 277       raise ValueError(error_message_template.format('train_op', train_op.name))
 278     for key, value in list(six.iteritems(eval_metric_ops)):
 279       if isinstance(value, Metric):
 280         values_to_check = value.updates[:]
 281         values_to_check.append(value.result())
 282       else:
 283         values_to_check = nest.flatten(value)
 284       for val in values_to_check:
 285         if val.graph is not default_graph:
 286           raise ValueError(error_message_template.format(
 287               'eval_metric_ops',
 288               '{0}: {1}'.format(key, val.name)))
 289 
 290     # Validate hooks.
 291     training_chief_hooks = tuple(training_chief_hooks or [])
 292     training_hooks = tuple(training_hooks or [])
 293     evaluation_hooks = tuple(evaluation_hooks or [])
 294     prediction_hooks = tuple(prediction_hooks or [])
 295 
 296     for hook in (training_hooks + training_chief_hooks + evaluation_hooks +
 297                  prediction_hooks):
 298       if not isinstance(hook, session_run_hook.SessionRunHook):
 299         raise TypeError(
 300             'All hooks must be SessionRunHook instances, given: {}'.format(
 301                 hook))
 302 
 303     # Add metric variables to the `LOCAL_VARIABLES` collection. Metric variables
 304     # are by default not added to any collections. We are doing this here, so
 305     # that metric variables get initialized.
 306     local_vars = set(ops.get_collection(ops.GraphKeys.LOCAL_VARIABLES))
 307     vars_to_add = set()
 308     for key, value in six.iteritems(eval_metric_ops):
 309       if isinstance(value, Metric):
 310         vars_to_add.update(value.variables)
 311         # Convert Metric instances to (value_tensor, update_op) tuple.
 312         eval_metric_ops[key] = (value.result(), value.updates[0])
 313     # Remove variables that are in the local variables collection already.
 314     vars_to_add = vars_to_add.difference(local_vars)
 315     for v in vars_to_add:
 316       ops.add_to_collection(ops.GraphKeys.LOCAL_VARIABLES, v)
 317 
 318     scaffold = scaffold or monitored_session.Scaffold()
 319     # Validate scaffold.
 320     if not isinstance(scaffold, monitored_session.Scaffold):
 321       raise TypeError(
 322           'scaffold must be tf.train.Scaffold. Given: {}'.format(scaffold))
 323 
 324     return super(EstimatorSpec, cls).__new__(
 325         cls,
 326         mode=mode,
 327         predictions=predictions,
 328         loss=loss,
 329         train_op=train_op,
 330         eval_metric_ops=eval_metric_ops,
 331         export_outputs=export_outputs,
 332         training_chief_hooks=training_chief_hooks,
 333         training_hooks=training_hooks,
 334         scaffold=scaffold,
 335         evaluation_hooks=evaluation_hooks,
 336         prediction_hooks=prediction_hooks)
 337 
 338   def _replace(self, **kwds):
 339     """Return a new EstimatorSpec replacing specified fields with new values."""
 340     if 'mode' in kwds:
 341       if self.mode != kwds['mode']:
 342         raise ValueError('mode of EstimatorSpec cannot be changed.')
 343     new_fields = map(kwds.pop, self._fields, list(self))
 344     return EstimatorSpec(*new_fields)

 347 def _get_export_outputs(export_outputs, predictions):
 348   """Validate export_outputs or create default export_outputs.
 349 
 350   Args:
 351     export_outputs: Describes the output signatures to be exported to
 352       `SavedModel` and used during serving. Should be a dict or None.
 353     predictions:  Predictions `Tensor` or dict of `Tensor`.
 354 
 355   Returns:
 356     Valid export_outputs dict
 357 
 358   Raises:
 359     TypeError: if export_outputs is not a dict or its values are not
 360       ExportOutput instances.
 361   """
 362   if export_outputs is None:
 363     default_output = export_output_lib.PredictOutput(predictions)
 364     export_outputs = {
 365         signature_constants.DEFAULT_SERVING_SIGNATURE_DEF_KEY: default_output}
 366 
 367   if not isinstance(export_outputs, dict):
 368     raise TypeError('export_outputs must be dict, given: {}'.format(
 369         export_outputs))
 370   for v in six.itervalues(export_outputs):
 371     if not isinstance(v, export_output_lib.ExportOutput):
 372       raise TypeError(
 373           'Values in export_outputs must be ExportOutput objects. '
 374           'Given: {}'.format(export_outputs))
 375 
 376   _maybe_add_default_serving_output(export_outputs)
 377 
 378   return export_outputs

 381 def _maybe_add_default_serving_output(export_outputs):
 382   """Add a default serving output to the export_outputs if not present.
 383 
 384   Args:
 385     export_outputs: Describes the output signatures to be exported to
 386       `SavedModel` and used during serving. Should be a dict.
 387 
 388   Returns:
 389     export_outputs dict with default serving signature added if necessary
 390 
 391   Raises:
 392     ValueError: if multiple export_outputs were provided without a default
 393       serving key.
 394   """
 395   if len(export_outputs) == 1:
 396     (key, value), = export_outputs.items()
 397     if key != signature_constants.DEFAULT_SERVING_SIGNATURE_DEF_KEY:
 398       export_outputs[
 399           signature_constants.DEFAULT_SERVING_SIGNATURE_DEF_KEY] = value
 400   if len(export_outputs) > 1:
 401     if (signature_constants.DEFAULT_SERVING_SIGNATURE_DEF_KEY
 402         not in export_outputs):
 403       raise ValueError(
 404           'Multiple export_outputs were provided, but none of them is '
 405           'specified as the default.  Do this by naming one of them with '
 406           'signature_constants.DEFAULT_SERVING_SIGNATURE_DEF_KEY.')
 407 
 408   return export_outputs

 451   def as_estimator_spec(self):
 452     """Creates an equivalent `EstimatorSpec` used by CPU train/eval."""
 453     if not self.eval_metrics:
 454       eval_metric_ops = None
 455     else:
 456       metric_fn, tensors = self.eval_metrics
 457       eval_metric_ops = metric_fn(**tensors)
 458     return EstimatorSpec(
 459         mode=self.mode,
 460         predictions=self.predictions,
 461         loss=self.loss,
 462         train_op=self.train_op,
 463         eval_metric_ops=eval_metric_ops,
 464         export_outputs=self.export_outputs,
 465         training_hooks=self.training_hooks,
 466         evaluation_hooks=self.evaluation_hooks,
 467         prediction_hooks=self.prediction_hooks)

 470 def _check_is_tensor_or_operation(x, name):
 471   if not (isinstance(x, ops.Operation) or ops.is_dense_tensor_like(x)):
 472     raise TypeError('{} must be Operation or Tensor, given: {}'.format(name, x))
 473 
 474 
 475 def _check_is_tensor(x, tensor_name):
 476   """Returns `x` if it is a `Tensor`, raises TypeError otherwise."""
 477   if not ops.is_dense_tensor_like(x):
 478     raise TypeError('{} must be Tensor, given: {}'.format(tensor_name, x))
 479   return x

 482 def export_outputs_for_mode(
 483     mode, serving_export_outputs=None, predictions=None, loss=None,
 484     metrics=None):
 485   """Util function for constructing a `ExportOutput` dict given a mode.
 486 
 487   The returned dict can be directly passed to `build_all_signature_defs` helper
 488   function as the `export_outputs` argument, used for generating a SignatureDef
 489   map.
 490 
 491   Args:
 492     mode: A `ModeKeys` specifying the mode.
 493     serving_export_outputs: Describes the output signatures to be exported to
 494       `SavedModel` and used during serving. Should be a dict or None.
 495     predictions: A dict of Tensors or single Tensor representing model
 496         predictions. This argument is only used if serving_export_outputs is not
 497         set.
 498     loss: A dict of Tensors or single Tensor representing calculated loss.
 499     metrics: A dict of (metric_value, update_op) tuples, or a single tuple.
 500       metric_value must be a Tensor, and update_op must be a Tensor or Op
 501 
 502   Returns:
 503     Dictionary mapping the a key to an `tf.estimator.export.ExportOutput` object
 504     The key is the expected SignatureDef key for the mode.
 505 
 506   Raises:
 507     ValueError: if an appropriate ExportOutput cannot be found for the mode.
 508   """
 509   # TODO(b/113185250): move all model export helper functions into an util file.
 510   if mode == ModeKeys.PREDICT:
 511     return _get_export_outputs(serving_export_outputs, predictions)
 512   elif mode == ModeKeys.TRAIN:
 513     return {mode: export_output_lib.TrainOutput(
 514         loss=loss, predictions=predictions, metrics=metrics)}
 515   elif mode == ModeKeys.EVAL:
 516     return {mode: export_output_lib.EvalOutput(
 517         loss=loss, predictions=predictions, metrics=metrics)}
 518   else:
 519     raise ValueError(
 520         'Export output type not found for mode: {}'.format(mode))
