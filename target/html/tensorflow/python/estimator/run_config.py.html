  75 def _get_session_master(cluster_spec, task_type, task_id, tf_config):
  76   """Returns the appropriate address for TensorFlow master.
  77 
  78   The order of precedence to deteremine the TF session master is as follows:
  79   1. If `tf_session_master` is set in TF_CONFIG environment variable, takes it.
  80   2. If the cluster has only one node, returns empty string ''.
  81   3. Returns the grpc address according to the task type and id in the cluster.
  82      This is between-graph replication.
  83 
  84   Note: task_type and task_id must be validated. Typically, validated using
  85   `_validate_task_type_and_task_id`.
  86 
  87   Args:
  88     cluster_spec: A `ClusterSpec` instance.
  89     task_type: String. Task type for current node.
  90     task_id: Int. Task id for current node.
  91     tf_config: Dict. Python dict for the TF_CONFIG environment variable.
  92 
  93   Raises:
  94     RuntimeError: If `cluster_spec` is not set.
  95 
  96   """
  97   if _SESSION_MASTER_KEY in tf_config:
  98     return tf_config[_SESSION_MASTER_KEY]
  99 
 100   if not cluster_spec:
 101     raise RuntimeError('Internal error: `_get_session_master` '
 102                        'does not expect empty cluster_spec.')
 103 
 104   jobs = cluster_spec.jobs
 105 
 106   # If there is only one node in the cluster, do things locally by setting
 107   # master to ''.  If a service or user sets TF_CONFIG with a single node, it's
 108   # more performant to use a direct master rather than an RPC service.
 109   if len(jobs) == 1 and len(cluster_spec.job_tasks(jobs[0])) == 1:
 110     return _LOCAL_MASTER
 111 
 112   # Lookup the master in cluster_spec using task_type and task_id,
 113   # if possible.
 114   addresses = cluster_spec.job_tasks(task_type)
 115   return _GRPC_SCHEME + addresses[task_id]

 118 def _get_eval_session_master(task_type, tf_config):
 119   """Returns the appropriate address for TensorFlow evaluation master."""
 120   if task_type == TaskType.EVALUATOR:
 121     return tf_config.get(_EVAL_SESSION_MASTER_KEY, _LOCAL_MASTER)
 122 
 123   if _EVAL_SESSION_MASTER_KEY in tf_config:
 124     raise ValueError('Key ({}) should not be set for task type other than {}. '
 125                      'Task type: {}'.format(_EVAL_SESSION_MASTER_KEY,
 126                                             TaskType.EVALUATOR, task_type))
 127   return _LOCAL_MASTER

 158 def _validate_task_type_and_task_id(cluster_spec, task_env, chief_task_type):
 159   """Validates the task type and index in `task_env` according to cluster."""
 160   if chief_task_type not in cluster_spec.jobs:
 161     raise ValueError(
 162         'If "cluster" is set in TF_CONFIG, it must have one "%s" node.' %
 163         chief_task_type)
 164   if len(cluster_spec.job_tasks(chief_task_type)) > 1:
 165     raise ValueError(
 166         'The "cluster" in TF_CONFIG must have only one "%s" node.' %
 167         chief_task_type)
 168 
 169   task_type = task_env.get(_TASK_TYPE_KEY, None)
 170   task_id = task_env.get(_TASK_ID_KEY, None)
 171 
 172   if not task_type:
 173     raise ValueError(
 174         'If "cluster" is set in TF_CONFIG, task type must be set.')
 175   if task_id is None:
 176     raise ValueError(
 177         'If "cluster" is set in TF_CONFIG, task index must be set.')
 178 
 179   task_id = int(task_id)
 180 
 181   # Check the task id bounds. Upper bound is not necessary as
 182   # - for evaluator, there is no upper bound.
 183   # - for non-evaluator, task id is upper bounded by the number of jobs in
 184   # cluster spec, which will be checked later (when retrieving the `master`)
 185   if task_id < 0:
 186     raise ValueError('Task index must be non-negative number.')
 187 
 188   # Evaluator is not part of the training cluster.
 189   if task_type == TaskType.EVALUATOR:
 190     return task_type, task_id
 191 
 192   if task_type not in cluster_spec.jobs:
 193     raise ValueError(
 194         '%s is not a valid task_type in the cluster_spec:\n'
 195         '%s\n\n'
 196         'Note that these values may be coming from the TF_CONFIG environment '
 197         'variable.' % (task_type, cluster_spec))
 198   addresses = cluster_spec.job_tasks(task_type)
 199   if not 0 <= task_id < len(addresses):
 200     raise ValueError(
 201         '%d is not a valid task_id for task_type %s in the cluster_spec:\n'
 202         '%s\n\n'
 203         'Note that these values may be coming from the TF_CONFIG environment '
 204         'variable.' % (task_id, task_type, cluster_spec))
 205 
 206   return task_type, task_id

 209 def _get_global_id_in_cluster(
 210     cluster_spec, task_type, task_id, chief_task_type):
 211   """Returns the global id in cluster."""
 212   # Note: This is implementation details, which user should not rely on.
 213   # The first id is 0, which is always for the `chief` node. All other nodes,
 214   # except `ps`, are ordered alphabetical based on task type (alphabetically)
 215   # and task id (ascendingly). `ps` are ordered last.
 216 
 217   # Sort task names in cluster
 218   task_type_ordered_list = [chief_task_type]
 219   task_type_ordered_list.extend([
 220       t for t in sorted(cluster_spec.jobs)
 221       if t != chief_task_type and t != TaskType.PS
 222   ])
 223   if TaskType.PS in cluster_spec.jobs:
 224     task_type_ordered_list.append(TaskType.PS)
 225 
 226   next_global_id = 0
 227   for t in task_type_ordered_list:
 228     if t == task_type:
 229       return next_global_id + task_id
 230     next_global_id += len(cluster_spec.job_tasks(t))
 231 
 232   # This should never happen.
 233   raise RuntimeError('Internal Error: `task_type` ({}) is not in '
 234                      'cluster_spec ({}).'.format(task_type, cluster_spec))

 237 def _validate_save_ckpt_with_replaced_keys(new_copy, replaced_keys):
 238   """Validates the save ckpt properties."""
 239   # Ensure one (and only one) of save_steps and save_secs is not None.
 240   # Also, if user sets one save ckpt property, say steps, the other one (secs)
 241   # should be set as None to improve usability.
 242 
 243   save_steps = new_copy.save_checkpoints_steps
 244   save_secs = new_copy.save_checkpoints_secs
 245 
 246   if ('save_checkpoints_steps' in replaced_keys and
 247       'save_checkpoints_secs' in replaced_keys):
 248     # If user sets both properties explicitly, we need to error out if both
 249     # are set or neither of them are set.
 250     if save_steps is not None and save_secs is not None:
 251       raise ValueError(_SAVE_CKPT_ERR)
 252   elif 'save_checkpoints_steps' in replaced_keys and save_steps is not None:
 253     new_copy._save_checkpoints_secs = None  # pylint: disable=protected-access
 254   elif 'save_checkpoints_secs' in replaced_keys and save_secs is not None:
 255     new_copy._save_checkpoints_steps = None  # pylint: disable=protected-access

 258 def _validate_properties(run_config):
 259   """Validates the properties."""
 260   def _validate(property_name, cond, message):
 261     property_value = getattr(run_config, property_name)
 262     if property_value is not None and not cond(property_value):
 263       raise ValueError(message)
 264 
 265   _validate('model_dir', lambda dir: dir,
 266             message='model_dir should be non-empty')
 267 
 268   _validate('save_summary_steps', lambda steps: steps >= 0,
 269             message='save_summary_steps should be >= 0')
 270 
 271   _validate('save_checkpoints_steps', lambda steps: steps >= 0,
 272             message='save_checkpoints_steps should be >= 0')
 273   _validate('save_checkpoints_secs', lambda secs: secs >= 0,
 274             message='save_checkpoints_secs should be >= 0')
 275 
 276   _validate('session_config',
 277             lambda sc: isinstance(sc, config_pb2.ConfigProto),
 278             message='session_config must be instance of ConfigProto')
 279 
 280   _validate('keep_checkpoint_max', lambda keep_max: keep_max >= 0,
 281             message='keep_checkpoint_max should be >= 0')
 282   _validate('keep_checkpoint_every_n_hours', lambda keep_hours: keep_hours > 0,
 283             message='keep_checkpoint_every_n_hours should be > 0')
 284   _validate('log_step_count_steps', lambda num_steps: num_steps > 0,
 285             message='log_step_count_steps should be > 0')
 286 
 287   _validate('tf_random_seed', lambda seed: isinstance(seed, six.integer_types),
 288             message='tf_random_seed must be integer.')
 289 
 290   _validate('device_fn', lambda device_fn: six.callable(device_fn) and
 291             set(function_utils.fn_args(device_fn)) == _VALID_DEVICE_FN_ARGS,
 292             message='device_fn must be callable with exactly'
 293                     ' one argument "op".')
 294 
 295   _validate('protocol',
 296             lambda protocol: protocol in (None, "grpc", "grpc+verbs"),
 297             message='protocol should be grpc or grpc+verbs')

 319 @estimator_export('estimator.RunConfig')
 320 class RunConfig(object):
 321   """This class specifies the configurations for an `Estimator` run."""
 322 
 323   def __init__(self,
 324                model_dir=None,
 325                tf_random_seed=None,
 326                save_summary_steps=100,
 327                save_checkpoints_steps=_USE_DEFAULT,
 328                save_checkpoints_secs=_USE_DEFAULT,
 329                session_config=None,
 330                keep_checkpoint_max=5,
 331                keep_checkpoint_every_n_hours=10000,
 332                log_step_count_steps=100,
 333                train_distribute=None,
 334                device_fn=None,
 335                protocol=None,
 336                eval_distribute=None,
 337                experimental_distribute=None):
 338     """Constructs a RunConfig.
 339 
 340     All distributed training related properties `cluster_spec`, `is_chief`,
 341     `master` , `num_worker_replicas`, `num_ps_replicas`, `task_id`, and
 342     `task_type` are set based on the `TF_CONFIG` environment variable, if the
 343     pertinent information is present. The `TF_CONFIG` environment variable is a
 344     JSON object with attributes: `cluster` and `task`.
 345 
 346     `cluster` is a JSON serialized version of `ClusterSpec`'s Python dict from
 347     `server_lib.py`, mapping task types (usually one of the `TaskType` enums) to
 348     a list of task addresses.
 349 
 350     `task` has two attributes: `type` and `index`, where `type` can be any of
 351     the task types in `cluster`. When `TF_CONFIG` contains said information,
 352     the following properties are set on this class:
 353 
 354     * `cluster_spec` is parsed from `TF_CONFIG['cluster']`. Defaults to {}. If
 355       present, must have one and only one node in the `chief` attribute of
 356       `cluster_spec`.
 357     * `task_type` is set to `TF_CONFIG['task']['type']`. Must set if
 358       `cluster_spec` is present; must be `worker` (the default value) if
 359       `cluster_spec` is not set.
 360     * `task_id` is set to `TF_CONFIG['task']['index']`. Must set if
 361       `cluster_spec` is present; must be 0 (the default value) if
 362       `cluster_spec` is not set.
 363     * `master` is determined by looking up `task_type` and `task_id` in the
 364       `cluster_spec`. Defaults to ''.
 365     * `num_ps_replicas` is set by counting the number of nodes listed
 366       in the `ps` attribute of `cluster_spec`. Defaults to 0.
 367     * `num_worker_replicas` is set by counting the number of nodes listed
 368       in the `worker` and `chief` attributes of `cluster_spec`. Defaults to 1.
 369     * `is_chief` is determined based on `task_type` and `cluster`.
 370 
 371     There is a special node with `task_type` as `evaluator`, which is not part
 372     of the (training) `cluster_spec`. It handles the distributed evaluation job.
 373 
 374     Example of non-chief node:
 375     ```
 376       cluster = {'chief': ['host0:2222'],
 377                  'ps': ['host1:2222', 'host2:2222'],
 378                  'worker': ['host3:2222', 'host4:2222', 'host5:2222']}
 379       os.environ['TF_CONFIG'] = json.dumps(
 380           {'cluster': cluster,
 381            'task': {'type': 'worker', 'index': 1}})
 382       config = RunConfig()
 383       assert config.master == 'host4:2222'
 384       assert config.task_id == 1
 385       assert config.num_ps_replicas == 2
 386       assert config.num_worker_replicas == 4
 387       assert config.cluster_spec == server_lib.ClusterSpec(cluster)
 388       assert config.task_type == 'worker'
 389       assert not config.is_chief
 390     ```
 391 
 392     Example of chief node:
 393     ```
 394       cluster = {'chief': ['host0:2222'],
 395                  'ps': ['host1:2222', 'host2:2222'],
 396                  'worker': ['host3:2222', 'host4:2222', 'host5:2222']}
 397       os.environ['TF_CONFIG'] = json.dumps(
 398           {'cluster': cluster,
 399            'task': {'type': 'chief', 'index': 0}})
 400       config = RunConfig()
 401       assert config.master == 'host0:2222'
 402       assert config.task_id == 0
 403       assert config.num_ps_replicas == 2
 404       assert config.num_worker_replicas == 4
 405       assert config.cluster_spec == server_lib.ClusterSpec(cluster)
 406       assert config.task_type == 'chief'
 407       assert config.is_chief
 408     ```
 409 
 410     Example of evaluator node (evaluator is not part of training cluster):
 411     ```
 412       cluster = {'chief': ['host0:2222'],
 413                  'ps': ['host1:2222', 'host2:2222'],
 414                  'worker': ['host3:2222', 'host4:2222', 'host5:2222']}
 415       os.environ['TF_CONFIG'] = json.dumps(
 416           {'cluster': cluster,
 417            'task': {'type': 'evaluator', 'index': 0}})
 418       config = RunConfig()
 419       assert config.master == ''
 420       assert config.evaluator_master == ''
 421       assert config.task_id == 0
 422       assert config.num_ps_replicas == 0
 423       assert config.num_worker_replicas == 0
 424       assert config.cluster_spec == {}
 425       assert config.task_type == 'evaluator'
 426       assert not config.is_chief
 427     ```
 428 
 429     N.B.: If `save_checkpoints_steps` or `save_checkpoints_secs` is set,
 430     `keep_checkpoint_max` might need to be adjusted accordingly, especially in
 431     distributed training. For example, setting `save_checkpoints_secs` as 60
 432     without adjusting `keep_checkpoint_max` (defaults to 5) leads to situation
 433     that checkpoint would be garbage collected after 5 minutes. In distributed
 434     training, the evaluation job starts asynchronously and might fail to load or
 435     find the checkpoint due to race condition.
 436 
 437     Args:
 438       model_dir: directory where model parameters, graph, etc are saved. If
 439         `PathLike` object, the path will be resolved. If `None`, will use a
 440         default value set by the Estimator.
 441       tf_random_seed: Random seed for TensorFlow initializers.
 442         Setting this value allows consistency between reruns.
 443       save_summary_steps: Save summaries every this many steps.
 444       save_checkpoints_steps: Save checkpoints every this many steps. Can not be
 445           specified with `save_checkpoints_secs`.
 446       save_checkpoints_secs: Save checkpoints every this many seconds. Can not
 447           be specified with `save_checkpoints_steps`. Defaults to 600 seconds if
 448           both `save_checkpoints_steps` and `save_checkpoints_secs` are not set
 449           in constructor.  If both `save_checkpoints_steps` and
 450           `save_checkpoints_secs` are None, then checkpoints are disabled.
 451       session_config: a ConfigProto used to set session parameters, or None.
 452       keep_checkpoint_max: The maximum number of recent checkpoint files to
 453         keep. As new files are created, older files are deleted. If None or 0,
 454         all checkpoint files are kept. Defaults to 5 (that is, the 5 most recent
 455         checkpoint files are kept.)
 456       keep_checkpoint_every_n_hours: Number of hours between each checkpoint
 457         to be saved. The default value of 10,000 hours effectively disables
 458         the feature.
 459       log_step_count_steps: The frequency, in number of global steps, that the
 460         global step/sec and the loss will be logged during training.
 461       train_distribute: An optional instance of
 462         `tf.contrib.distribute.DistributionStrategy`. If specified,
 463         then Estimator will distribute the user's model during training,
 464         according to the policy specified by that strategy. Setting
 465         `experimental_distribute.train_distribute` is preferred.
 466       device_fn: A callable invoked for every `Operation` that takes the
 467         `Operation` and returns the device string. If `None`, defaults to
 468         the device function returned by `tf.train.replica_device_setter`
 469         with round-robin strategy.
 470       protocol: An optional argument which specifies the protocol used when
 471         starting server. None means default to grpc.
 472       eval_distribute: An optional instance of
 473         `tf.contrib.distribute.DistributionStrategy`. If specified,
 474         then Estimator will distribute the user's model during evaluation,
 475         according to the policy specified by that strategy. Setting
 476         `experimental_distribute.eval_distribute` is preferred.
 477       experimental_distribute: an optional
 478         `tf.contrib.distribute.DistributeConfig` object specifying
 479         DistributionStrategy-related configuration. The `train_distribute` and
 480         `eval_distribute` can be passed as parameters to `RunConfig` or set in
 481         `experimental_distribute` but not both.
 482 
 483     Raises:
 484       ValueError: If both `save_checkpoints_steps` and `save_checkpoints_secs`
 485       are set.
 486     """
 487     if (save_checkpoints_steps == _USE_DEFAULT and
 488         save_checkpoints_secs == _USE_DEFAULT):
 489       save_checkpoints_steps = None
 490       save_checkpoints_secs = 600
 491     elif save_checkpoints_secs == _USE_DEFAULT:
 492       save_checkpoints_secs = None
 493     elif save_checkpoints_steps == _USE_DEFAULT:
 494       save_checkpoints_steps = None
 495     elif (save_checkpoints_steps is not None and
 496           save_checkpoints_secs is not None):
 497       raise ValueError(_SAVE_CKPT_ERR)
 498 
 499     tf_config = json.loads(os.environ.get(_TF_CONFIG_ENV, '{}'))
 500     if tf_config:
 501       logging.info('TF_CONFIG environment variable: %s', tf_config)
 502 
 503     model_dir = _get_model_dir(tf_config,
 504                                compat_internal.path_to_str(model_dir))
 505 
 506     RunConfig._replace(
 507         self,
 508         allowed_properties_list=_DEFAULT_REPLACEABLE_LIST,
 509         model_dir=model_dir,
 510         tf_random_seed=tf_random_seed,
 511         save_summary_steps=save_summary_steps,
 512         save_checkpoints_steps=save_checkpoints_steps,
 513         save_checkpoints_secs=save_checkpoints_secs,
 514         session_config=session_config,
 515         keep_checkpoint_max=keep_checkpoint_max,
 516         keep_checkpoint_every_n_hours=keep_checkpoint_every_n_hours,
 517         log_step_count_steps=log_step_count_steps,
 518         train_distribute=train_distribute,
 519         device_fn=device_fn,
 520         protocol=protocol,
 521         eval_distribute=eval_distribute,
 522         experimental_distribute=experimental_distribute)
 523 
 524     # TODO(frankchn,priyag): Eventually use distributed coordinator for TPUs.
 525     if ((train_distribute and
 526          train_distribute.__class__.__name__ != 'TPUStrategy') or
 527         (eval_distribute and
 528          eval_distribute.__class__.__name__ != 'TPUStrategy') or
 529         experimental_distribute):
 530       logging.info('Initializing RunConfig with distribution strategies.')
 531       distribute_coordinator_training.init_run_config(self, tf_config)
 532     else:
 533       self._init_distributed_setting_from_environment_var(tf_config)
 534       self._maybe_overwrite_session_config_for_distributed_training()

 536   def _maybe_overwrite_session_config_for_distributed_training(self):
 537     """Overwrites the session_config for distributed training.
 538 
 539     The default overwrite is optimized for between-graph training. Subclass
 540     should override this method if necessary.
 541     """
 542     # Get session_config only for between-graph distributed mode (cluster_spec
 543     # is present).
 544     if not self._session_config and self._cluster_spec:
 545       RunConfig._replace(
 546           self,
 547           allowed_properties_list=_DEFAULT_REPLACEABLE_LIST,
 548           session_config=self._get_default_session_config_distributed())

 550   def _get_default_session_config_distributed(self):
 551     """Returns None or tf.ConfigProto instance with default device_filters set.
 552 
 553     Device filters are set such that chief/master and worker communicates with
 554     only ps. session_config=None for evaluators or any other TaskType.
 555     """
 556 
 557     rewrite_opts = rewriter_config_pb2.RewriterConfig(
 558         meta_optimizer_iterations=rewriter_config_pb2.RewriterConfig.ONE)
 559     graph_opts = config_pb2.GraphOptions(rewrite_options=rewrite_opts)
 560 
 561     device_filters = None
 562     if self._task_type == TaskType.MASTER:
 563       device_filters = ['/job:ps', '/job:master']
 564     elif self._task_type == TaskType.CHIEF:
 565       device_filters = ['/job:ps', '/job:chief']
 566     elif self._task_type == TaskType.WORKER:
 567       device_filters = ['/job:ps', '/job:worker/task:%d' % self._task_id]
 568     elif self._task_type == TaskType.PS:
 569       device_filters = ['/job:ps', '/job:worker', '/job:master']
 570     else:
 571       # If the task_type is `EVALUATOR` or something other than the ones in
 572       # TaskType then don't set any device filters.
 573       return None
 574 
 575     return config_pb2.ConfigProto(
 576         allow_soft_placement=True,
 577         graph_options=graph_opts,
 578         device_filters=device_filters)

 580   def _init_distributed_setting_from_environment_var(self, tf_config):
 581     """Initialize distributed properties based on `tf_config`."""
 582 
 583     self._service = _validate_service(tf_config.get(_SERVICE_KEY))
 584     self._cluster_spec = server_lib.ClusterSpec(tf_config.get(_CLUSTER_KEY, {}))
 585     task_env = tf_config.get(_TASK_ENV_KEY, {})
 586 
 587     if self._cluster_spec and TaskType.MASTER in self._cluster_spec.jobs:
 588       return self._init_distributed_setting_from_environment_var_with_master(
 589           tf_config)
 590 
 591     if self._cluster_spec:
 592       # Distributed mode.
 593       self._task_type, self._task_id = _validate_task_type_and_task_id(
 594           self._cluster_spec, task_env, TaskType.CHIEF)
 595 
 596       self._evaluation_master = _get_eval_session_master(
 597           self._task_type, tf_config)
 598 
 599       if self._task_type != TaskType.EVALUATOR:
 600         self._master = _get_session_master(self._cluster_spec, self._task_type,
 601                                            self._task_id, tf_config)
 602         self._num_ps_replicas = _count_ps(self._cluster_spec)
 603         self._num_worker_replicas = _count_worker(
 604             self._cluster_spec, chief_task_type=TaskType.CHIEF)
 605         self._global_id_in_cluster = _get_global_id_in_cluster(
 606             self._cluster_spec,
 607             self._task_type,
 608             self._task_id,
 609             chief_task_type=TaskType.CHIEF)
 610       else:
 611         # Evaluator is not part of the training cluster.
 612         self._cluster_spec = server_lib.ClusterSpec({})
 613         self._master = _LOCAL_MASTER
 614         self._num_ps_replicas = 0
 615         self._num_worker_replicas = 0
 616         self._global_id_in_cluster = None  # undefined
 617 
 618       self._is_chief = self._task_type == TaskType.CHIEF
 619     else:
 620       # Local mode.
 621       self._task_type = task_env.get(_TASK_TYPE_KEY, TaskType.WORKER)
 622       self._task_id = int(task_env.get(_TASK_ID_KEY, 0))
 623       self._global_id_in_cluster = 0
 624 
 625       if self._task_type != TaskType.WORKER:
 626         raise ValueError(
 627             'If "cluster" is not set in TF_CONFIG, task type must be WORKER.')
 628       if self._task_id != 0:
 629         raise ValueError(
 630             'If "cluster" is not set in TF_CONFIG, task index must be 0.')
 631 
 632       self._master = tf_config.get(_SESSION_MASTER_KEY, _LOCAL_MASTER)
 633       self._evaluation_master = tf_config.get(_EVAL_SESSION_MASTER_KEY,
 634                                               _LOCAL_MASTER)
 635       self._is_chief = True
 636       self._num_ps_replicas = 0
 637       self._num_worker_replicas = 1
 638 
 639   def _init_distributed_setting_from_environment_var_with_master(self,
 640                                                                  tf_config):
 641     """Initialize distributed properties for legacy cluster with `master`."""
 642     # There is no tech reason, why user cannot have chief and master in the same
 643     # cluster, but it is super confusing (which is really the chief?). So, block
 644     # this case.
 645     if TaskType.CHIEF in self._cluster_spec.jobs:
 646       raise ValueError('If `master` node exists in `cluster`, job '
 647                        '`chief` is not supported.')
 648 
 649     task_env = tf_config.get(_TASK_ENV_KEY, {})
 650 
 651     self._task_type, self._task_id = _validate_task_type_and_task_id(
 652         self._cluster_spec, task_env, TaskType.MASTER)
 653 
 654     if self._task_type == TaskType.EVALUATOR:
 655       raise ValueError('If `master` node exists in `cluster`, task_type '
 656                        '`evaluator` is not supported.')
 657 
 658     self._global_id_in_cluster = _get_global_id_in_cluster(
 659         self._cluster_spec,
 660         self._task_type,
 661         self._task_id,
 662         chief_task_type=TaskType.MASTER)
 663 
 664     self._master = _get_session_master(self._cluster_spec, self._task_type,
 665                                        self._task_id, tf_config)
 666     self._evaluation_master = _get_eval_session_master(self._task_type,
 667                                                        tf_config)
 668     self._num_ps_replicas = _count_ps(self._cluster_spec)
 669     self._num_worker_replicas = _count_worker(
 670         self._cluster_spec, chief_task_type=TaskType.MASTER)
 671 
 672     self._is_chief = self._task_type == TaskType.MASTER

 712   @property
 713   def global_id_in_cluster(self):
 714     """The global id in the training cluster.
 715 
 716     All global ids in the training cluster are assigned from an increasing
 717     sequence of consecutive integers. The first id is 0.
 718 
 719     Note: Task id (the property field `task_id`) is tracking the index of the
 720     node among all nodes with the SAME task type. For example, given the cluster
 721     definition as follows:
 722 
 723     ```
 724       cluster = {'chief': ['host0:2222'],
 725                  'ps': ['host1:2222', 'host2:2222'],
 726                  'worker': ['host3:2222', 'host4:2222', 'host5:2222']}
 727     ```
 728 
 729     Nodes with task type `worker` can have id 0, 1, 2.  Nodes with task type
 730     `ps` can have id, 0, 1. So, `task_id` is not unique, but the pair
 731     (`task_type`, `task_id`) can uniquely determine a node in the cluster.
 732 
 733     Global id, i.e., this field, is tracking the index of the node among ALL
 734     nodes in the cluster. It is uniquely assigned.  For example, for the cluster
 735     spec given above, the global ids are assigned as:
 736     ```
 737       task_type  | task_id  |  global_id
 738       --------------------------------
 739       chief      | 0        |  0
 740       worker     | 0        |  1
 741       worker     | 1        |  2
 742       worker     | 2        |  3
 743       ps         | 0        |  4
 744       ps         | 1        |  5
 745     ```
 746 
 747     Returns:
 748       An integer id.
 749     """
 750     return self._global_id_in_cluster

 814   def replace(self, **kwargs):
 815     """Returns a new instance of `RunConfig` replacing specified properties.
 816 
 817     Only the properties in the following list are allowed to be replaced:
 818 
 819       - `model_dir`,
 820       - `tf_random_seed`,
 821       - `save_summary_steps`,
 822       - `save_checkpoints_steps`,
 823       - `save_checkpoints_secs`,
 824       - `session_config`,
 825       - `keep_checkpoint_max`,
 826       - `keep_checkpoint_every_n_hours`,
 827       - `log_step_count_steps`,
 828       - `train_distribute`,
 829       - `device_fn`,
 830       - `protocol`.
 831       - `eval_distribute`,
 832       - `experimental_distribute`,
 833 
 834     In addition, either `save_checkpoints_steps` or `save_checkpoints_secs`
 835     can be set (should not be both).
 836 
 837     Args:
 838       **kwargs: keyword named properties with new values.
 839 
 840     Raises:
 841       ValueError: If any property name in `kwargs` does not exist or is not
 842         allowed to be replaced, or both `save_checkpoints_steps` and
 843         `save_checkpoints_secs` are set.
 844 
 845     Returns:
 846       a new instance of `RunConfig`.
 847     """
 848     return RunConfig._replace(
 849         copy.deepcopy(self),
 850         allowed_properties_list=_DEFAULT_REPLACEABLE_LIST,
 851         **kwargs)
 852 
 853   @staticmethod
 854   def _replace(config, allowed_properties_list=None, **kwargs):
 855     """See `replace`.
 856 
 857     N.B.: This implementation assumes that for key named "foo", the underlying
 858     property the RunConfig holds is "_foo" (with one leading underscore).
 859 
 860     Args:
 861       config: The RunConfig to replace the values of.
 862       allowed_properties_list: The property name list allowed to be replaced.
 863       **kwargs: keyword named properties with new values.
 864 
 865     Raises:
 866       ValueError: If any property name in `kwargs` does not exist or is not
 867         allowed to be replaced, or both `save_checkpoints_steps` and
 868         `save_checkpoints_secs` are set.
 869 
 870     Returns:
 871       a new instance of `RunConfig`.
 872     """
 873 
 874     allowed_properties_list = allowed_properties_list or []
 875 
 876     for key, new_value in six.iteritems(kwargs):
 877       if key in allowed_properties_list:
 878         setattr(config, '_' + key, new_value)
 879         continue
 880 
 881       raise ValueError(
 882           'Replacing {} is not supported. Allowed properties are {}.'.format(
 883               key, allowed_properties_list))
 884 
 885     _validate_save_ckpt_with_replaced_keys(config, kwargs.keys())
 886     _validate_properties(config)
 887     return config

 890 def _get_model_dir(tf_config, model_dir):
 891   """Returns `model_dir` based user provided `tf_config` or `model_dir`."""
 892   # pylint: disable=g-explicit-bool-comparison
 893 
 894   # Empty string is treated as False in Python condition check, which triggers
 895   # some confusing error messages. For example, 'a or b' returns None if a is ''
 896   # and b is None. `None` is allowed for model_dir but '' is not allowed. Here,
 897   # explicitly check empty string to provide clear error message.
 898   if model_dir == '':
 899     raise ValueError('model_dir should be non-empty.')
 900 
 901   model_dir_in_tf_config = tf_config.get('model_dir')
 902   if model_dir_in_tf_config == '':
 903     raise ValueError('model_dir in TF_CONFIG should be non-empty.')
 904 
 905   if model_dir_in_tf_config:
 906     if model_dir and model_dir_in_tf_config != model_dir:
 907       raise ValueError(
 908           '`model_dir` provided in RunConfig construct, if set, '
 909           'must have the same value as the model_dir in TF_CONFIG. '
 910           'model_dir: {}\nTF_CONFIG["model_dir"]: {}.\n'.format(
 911               model_dir, model_dir_in_tf_config))
 912 
 913     logging.info('Using model_dir in TF_CONFIG: %s', model_dir_in_tf_config)
 914 
 915   return model_dir or model_dir_in_tf_config
