  75 @estimator_export('estimator.Estimator')
  76 class Estimator(object):
  77   """Estimator class to train and evaluate TensorFlow models.
  78 
  79   The `Estimator` object wraps a model which is specified by a `model_fn`,
  80   which, given inputs and a number of other parameters, returns the ops
  81   necessary to perform training, evaluation, or predictions.
  82 
  83   All outputs (checkpoints, event files, etc.) are written to `model_dir`, or a
  84   subdirectory thereof. If `model_dir` is not set, a temporary directory is
  85   used.
  86 
  87   The `config` argument can be passed `tf.estimator.RunConfig` object containing
  88   information about the execution environment. It is passed on to the
  89   `model_fn`, if the `model_fn` has a parameter named "config" (and input
  90   functions in the same manner). If the `config` parameter is not passed, it is
  91   instantiated by the `Estimator`. Not passing config means that defaults useful
  92   for local execution are used. `Estimator` makes config available to the model
  93   (for instance, to allow specialization based on the number of workers
  94   available), and also uses some of its fields to control internals, especially
  95   regarding checkpointing.
  96 
  97   The `params` argument contains hyperparameters. It is passed to the
  98   `model_fn`, if the `model_fn` has a parameter named "params", and to the input
  99   functions in the same manner. `Estimator` only passes params along, it does
 100   not inspect it. The structure of `params` is therefore entirely up to the
 101   developer.
 102 
 103   None of `Estimator`'s methods can be overridden in subclasses (its
 104   constructor enforces this). Subclasses should use `model_fn` to configure
 105   the base class, and may add methods implementing specialized functionality.
 106 
 107   @compatibility(eager)
 108   Calling methods of `Estimator` will work while eager execution is enabled.
 109   However, the `model_fn` and `input_fn` is not executed eagerly, `Estimator`
 110   will switch to graph model before calling all user-provided functions (incl.
 111   hooks), so their code has to be compatible with graph mode execution. Note
 112   that `input_fn` code using `tf.data` generally works in both graph and eager
 113   modes.
 114   @end_compatibility
 115   """
 116 
 117   def __init__(self, model_fn, model_dir=None, config=None, params=None,
 118                warm_start_from=None):
 119     """Constructs an `Estimator` instance.
 120 
 121     See [estimators](https://tensorflow.org/guide/estimators) for more
 122     information.
 123 
 124     To warm-start an `Estimator`:
 125 
 126     ```python
 127     estimator = tf.estimator.DNNClassifier(
 128         feature_columns=[categorical_feature_a_emb, categorical_feature_b_emb],
 129         hidden_units=[1024, 512, 256],
 130         warm_start_from="/path/to/checkpoint/dir")
 131     ```
 132 
 133     For more details on warm-start configuration, see
 134     `tf.estimator.WarmStartSettings`.
 135 
 136     Args:
 137       model_fn: Model function. Follows the signature:
 138 
 139         * Args:
 140 
 141           * `features`: This is the first item returned from the `input_fn`
 142                  passed to `train`, `evaluate`, and `predict`. This should be a
 143                  single `tf.Tensor` or `dict` of same.
 144           * `labels`: This is the second item returned from the `input_fn`
 145                  passed to `train`, `evaluate`, and `predict`. This should be a
 146                  single `tf.Tensor` or `dict` of same (for multi-head models).
 147                  If mode is `tf.estimator.ModeKeys.PREDICT`, `labels=None` will
 148                  be passed. If the `model_fn`'s signature does not accept
 149                  `mode`, the `model_fn` must still be able to handle
 150                  `labels=None`.
 151           * `mode`: Optional. Specifies if this training, evaluation or
 152                  prediction. See `tf.estimator.ModeKeys`.
 153           * `params`: Optional `dict` of hyperparameters.  Will receive what
 154                  is passed to Estimator in `params` parameter. This allows
 155                  to configure Estimators from hyper parameter tuning.
 156           * `config`: Optional `estimator.RunConfig` object. Will receive what
 157                  is passed to Estimator as its `config` parameter, or a default
 158                  value. Allows setting up things in your `model_fn` based on
 159                  configuration such as `num_ps_replicas`, or `model_dir`.
 160 
 161         * Returns:
 162           `tf.estimator.EstimatorSpec`
 163 
 164       model_dir: Directory to save model parameters, graph and etc. This can
 165         also be used to load checkpoints from the directory into an estimator to
 166         continue training a previously saved model. If `PathLike` object, the
 167         path will be resolved. If `None`, the model_dir in `config` will be used
 168         if set. If both are set, they must be same. If both are `None`, a
 169         temporary directory will be used.
 170       config: `estimator.RunConfig` configuration object.
 171       params: `dict` of hyper parameters that will be passed into `model_fn`.
 172               Keys are names of parameters, values are basic python types.
 173       warm_start_from: Optional string filepath to a checkpoint or SavedModel to
 174                        warm-start from, or a `tf.estimator.WarmStartSettings`
 175                        object to fully configure warm-starting.  If the string
 176                        filepath is provided instead of a
 177                        `tf.estimator.WarmStartSettings`, then all variables are
 178                        warm-started, and it is assumed that vocabularies
 179                        and `tf.Tensor` names are unchanged.
 180 
 181     Raises:
 182       ValueError: parameters of `model_fn` don't match `params`.
 183       ValueError: if this is called via a subclass and if that class overrides
 184         a member of `Estimator`.
 185     """
 186     Estimator._assert_members_are_not_overridden(self)
 187 
 188     self._config = maybe_overwrite_model_dir_and_session_config(config,
 189                                                                 model_dir)
 190 
 191     # The distribute field contains an instance of DistributionStrategy.
 192     self._train_distribution = self._config.train_distribute
 193     self._eval_distribution = self._config.eval_distribute
 194     # Model directory.
 195     self._model_dir = self._config.model_dir
 196     self._session_config = self._config.session_config
 197     logging.info('Using config: %s', str(vars(self._config)))
 198 
 199     self._device_fn = (
 200         self._config.device_fn or _get_replica_device_setter(self._config))
 201 
 202     if model_fn is None:
 203       raise ValueError('model_fn must be provided to Estimator.')
 204     _verify_model_fn_args(model_fn, params)
 205     self._model_fn = model_fn
 206     self._params = copy.deepcopy(params or {})
 207 
 208     # pylint: disable=protected-access
 209     self._warm_start_settings = _get_default_warm_start_settings(
 210         warm_start_from)
 211     # pylint: enable=protected-access
 212

 279   def train(self,
 280             input_fn,
 281             hooks=None,
 282             steps=None,
 283             max_steps=None,
 284             saving_listeners=None):
 285     """Trains a model given training data `input_fn`.
 286 
 287     Args:
 288       input_fn: A function that provides input data for training as minibatches.
 289         See [Premade Estimators](
 290         https://tensorflow.org/guide/premade_estimators#create_input_functions)
 291         for more information. The function should construct and return one of
 292         the following:  * A
 293         `tf.data.Dataset` object: Outputs of `Dataset` object must be a tuple
 294         `(features, labels)` with same constraints as below. * A tuple
 295         `(features, labels)`: Where `features` is a `tf.Tensor` or a dictionary
 296         of string feature name to `Tensor` and `labels` is a `Tensor` or a
 297         dictionary of string label name to `Tensor`. Both `features` and
 298         `labels` are consumed by `model_fn`. They should satisfy the expectation
 299         of `model_fn` from inputs.
 300       hooks: List of `tf.train.SessionRunHook` subclass instances. Used for
 301         callbacks inside the training loop.
 302       steps: Number of steps for which to train the model. If `None`, train
 303         forever or train until `input_fn` generates the `tf.errors.OutOfRange`
 304         error or `StopIteration` exception. `steps` works incrementally. If you
 305         call two times `train(steps=10)` then training occurs in total 20 steps.
 306         If `OutOfRange` or `StopIteration` occurs in the middle, training stops
 307         before 20 steps. If you don't want to have incremental behavior please
 308         set `max_steps` instead. If set, `max_steps` must be `None`.
 309       max_steps: Number of total steps for which to train model. If `None`,
 310         train forever or train until `input_fn` generates the
 311         `tf.errors.OutOfRange` error or `StopIteration` exception. If set,
 312         `steps` must be `None`. If `OutOfRange` or `StopIteration` occurs in the
 313         middle, training stops before `max_steps` steps. Two calls to
 314         `train(steps=100)` means 200 training iterations. On the other hand, two
 315         calls to `train(max_steps=100)` means that the second call will not do
 316         any iteration since first call did all 100 steps.
 317       saving_listeners: list of `CheckpointSaverListener` objects. Used for
 318         callbacks that run immediately before or after checkpoint savings.
 319 
 320     Returns:
 321       `self`, for chaining.
 322 
 323     Raises:
 324       ValueError: If both `steps` and `max_steps` are not `None`.
 325       ValueError: If either `steps` or `max_steps <= 0`.
 326     """
 327     if self.config.task_type in (run_config.TaskType.EVALUATOR,
 328                                  run_config.TaskType.PS):
 329       raise ValueError(
 330           'Train has been called wrong configuration. Please use '
 331           'tf.estimator.train_and_evaluate which calls proper API according '
 332           'to given configuration. Current configuration: {}.'.format(
 333               self.config))
 334 
 335     with context.graph_mode():
 336       if (steps is not None) and (max_steps is not None):
 337         raise ValueError('Can not provide both steps and max_steps.')
 338       if steps is not None and steps <= 0:
 339         raise ValueError('Must specify steps > 0, given: {}'.format(steps))
 340       if max_steps is not None and max_steps <= 0:
 341         raise ValueError(
 342             'Must specify max_steps > 0, given: {}'.format(max_steps))
 343 
 344       if max_steps is not None:
 345         start_step = _load_global_step_from_checkpoint_dir(self._model_dir)
 346         if max_steps <= start_step:
 347           logging.info('Skipping training since max_steps has already saved.')
 348           return self
 349 
 350       hooks = _check_hooks_type(hooks)
 351       hooks.extend(self._convert_train_steps_to_hooks(steps, max_steps))
 352 
 353       saving_listeners = _check_listeners_type(saving_listeners)
 354       loss = self._train_model(input_fn, hooks, saving_listeners)
 355       logging.info('Loss for final step: %s.', loss)
 356       return self

 358   def _convert_train_steps_to_hooks(self, steps, max_steps):
 359     """Create hooks to run correct number of steps in training.
 360 
 361     Args:
 362       steps: number of steps to run during training.
 363       max_steps: maximum number of steps to be run during training. It'll be
 364         the maximum number of steps the model will train to after restoring
 365         from checkpoint even across multiple estimator.train calls.
 366 
 367     Returns:
 368       List of hooks to be passed to the estimator.
 369     """
 370     if steps is not None or max_steps is not None:
 371       if self._train_distribution:
 372         steps_per_run = getattr(self._train_distribution, 'steps_per_run', 1)
 373         if steps_per_run > 1:
 374           return [basic_session_run_hooks._MultiStepStopAtStepHook(  # pylint: disable=protected-access
 375               steps, max_steps, steps_per_run)]
 376       return [training.StopAtStepHook(steps, max_steps)]
 377     else:
 378       return []

 395   def evaluate(self, input_fn, steps=None, hooks=None, checkpoint_path=None,
 396                name=None):
 397     """Evaluates the model given evaluation data `input_fn`.
 398 
 399     For each step, calls `input_fn`, which returns one batch of data.
 400     Evaluates until:
 401     - `steps` batches are processed, or
 402     - `input_fn` raises an end-of-input exception (`tf.errors.OutOfRangeError`
 403     or
 404     `StopIteration`).
 405 
 406     Args:
 407       input_fn: A function that constructs the input data for evaluation. See
 408         [Premade Estimators](
 409         https://tensorflow.org/guide/premade#create_input_functions)
 410         for more information. The
 411         function should construct and return one of the following:  * A
 412         `tf.data.Dataset` object: Outputs of `Dataset` object must be a tuple
 413         `(features, labels)` with same constraints as below. * A tuple
 414         `(features, labels)`: Where `features` is a `tf.Tensor` or a dictionary
 415         of string feature name to `Tensor` and `labels` is a `Tensor` or a
 416         dictionary of string label name to `Tensor`. Both `features` and
 417         `labels` are consumed by `model_fn`. They should satisfy the expectation
 418         of `model_fn` from inputs.
 419       steps: Number of steps for which to evaluate model. If `None`, evaluates
 420         until `input_fn` raises an end-of-input exception.
 421       hooks: List of `tf.train.SessionRunHook` subclass instances. Used for
 422         callbacks inside the evaluation call.
 423       checkpoint_path: Path of a specific checkpoint to evaluate. If `None`, the
 424         latest checkpoint in `model_dir` is used.  If there are no checkpoints
 425         in `model_dir`, evaluation is run with newly initialized `Variables`
 426         instead of ones restored from checkpoint.
 427       name: Name of the evaluation if user needs to run multiple evaluations on
 428         different data sets, such as on training data vs test data. Metrics for
 429         different evaluations are saved in separate folders, and appear
 430         separately in tensorboard.
 431 
 432     Returns:
 433       A dict containing the evaluation metrics specified in `model_fn` keyed by
 434       name, as well as an entry `global_step` which contains the value of the
 435       global step for which this evaluation was performed. For canned
 436       estimators, the dict contains the `loss` (mean loss per mini-batch) and
 437       the `average_loss` (mean loss per sample). Canned classifiers also return
 438       the `accuracy`. Canned regressors also return the `label/mean` and the
 439       `prediction/mean`.
 440 
 441     Raises:
 442       ValueError: If `steps <= 0`.
 443       ValueError: If no model has been trained, namely `model_dir`, or the
 444         given `checkpoint_path` is empty.
 445     """
 446     with context.graph_mode():
 447       hooks = _check_hooks_type(hooks)
 448       hooks.extend(self._convert_eval_steps_to_hooks(steps))
 449 
 450       # Check that model has been trained (if nothing has been set explicitly).
 451       if not checkpoint_path:
 452         latest_path = checkpoint_management.latest_checkpoint(self._model_dir)
 453         if not latest_path:
 454           logging.info('Could not find trained model in model_dir: {}, running '
 455                        'initialization to evaluate.'.format(self._model_dir))
 456         checkpoint_path = latest_path
 457 
 458       def _evaluate():
 459         (scaffold, update_op, eval_dict, all_hooks) = (
 460             self._evaluate_build_graph(input_fn, hooks, checkpoint_path))
 461         return self._evaluate_run(
 462             checkpoint_path=checkpoint_path,
 463             scaffold=scaffold,
 464             update_op=update_op,
 465             eval_dict=eval_dict,
 466             all_hooks=all_hooks,
 467             output_dir=self.eval_dir(name))
 468 
 469       with ops.Graph().as_default():
 470         if self._eval_distribution:
 471           # We want to create the iterations variable outside the distribution
 472           # scope as that is just stored on the host and mainly used to drive
 473           # the loop and doesn't need to be a Mirrored/Device variable.
 474           training.get_or_create_steps_per_run_variable()
 475           with self._eval_distribution.scope():
 476             return _evaluate()
 477         else:
 478           return _evaluate()
 479 
 480   def _convert_eval_steps_to_hooks(self, steps):
 481     """Create hooks to run correct number of steps in evaluation.
 482 
 483     Args:
 484       steps: number of steps to run during evaluation.
 485 
 486     Raises:
 487       ValueError: if steps is less than or equal to zero.
 488 
 489     Returns:
 490       List of hooks to be passed to the estimator.
 491     """
 492     if steps is None:
 493       return []
 494 
 495     if steps <= 0:
 496       raise ValueError('Must specify steps > 0, given: {}'.format(steps))
 497 
 498     # The hooks are declared as private in evaluation.py discourage the use
 499     # by other libraries or open source users. This should be the only usage
 500     # of the estimator evaluation hooks.
 501     if self._eval_distribution:
 502       steps_per_run = getattr(self._eval_distribution, 'steps_per_run', 1)
 503       if steps_per_run > 1:
 504         return [evaluation._MultiStepStopAfterNEvalsHook(  # pylint: disable=protected-access
 505             num_evals=steps, steps_per_run=steps_per_run)]
 506     return [evaluation._StopAfterNEvalsHook(num_evals=steps)]  # pylint: disable=protected-access

 508   def predict(self,
 509               input_fn,
 510               predict_keys=None,
 511               hooks=None,
 512               checkpoint_path=None,
 513               yield_single_examples=True):
 514     """Yields predictions for given features.
 515 
 516     Please note that interleaving two predict outputs does not work. See:
 517     [issue/20506](
 518     https://github.com/tensorflow/tensorflow/issues/20506#issuecomment-422208517)
 519 
 520     Args:
 521       input_fn: A function that constructs the features. Prediction continues
 522         until `input_fn` raises an end-of-input exception
 523         (`tf.errors.OutOfRangeError` or `StopIteration`).
 524         See [Premade Estimators](
 525         https://tensorflow.org/guide/premade_estimators#create_input_functions)
 526         for more information. The function should construct and return one of
 527         the following:
 528 
 529           * A `tf.data.Dataset` object: Outputs of `Dataset` object must have
 530             same constraints as below.
 531           * features: A `tf.Tensor` or a dictionary of string feature name to
 532             `Tensor`. features are consumed by `model_fn`. They should satisfy
 533             the expectation of `model_fn` from inputs.
 534           * A tuple, in which case the first item is extracted as features.
 535 
 536       predict_keys: list of `str`, name of the keys to predict. It is used if
 537         the `tf.estimator.EstimatorSpec.predictions` is a `dict`. If
 538         `predict_keys` is used then rest of the predictions will be filtered
 539         from the dictionary. If `None`, returns all.
 540       hooks: List of `tf.train.SessionRunHook` subclass instances. Used for
 541         callbacks inside the prediction call.
 542       checkpoint_path: Path of a specific checkpoint to predict. If `None`, the
 543         latest checkpoint in `model_dir` is used.  If there are no checkpoints
 544         in `model_dir`, prediction is run with newly initialized `Variables`
 545         instead of ones restored from checkpoint.
 546       yield_single_examples: If `False`, yields the whole batch as returned by
 547         the `model_fn` instead of decomposing the batch into individual
 548         elements. This is useful if `model_fn` returns some tensors whose first
 549         dimension is not equal to the batch size.
 550 
 551     Yields:
 552       Evaluated values of `predictions` tensors.
 553 
 554     Raises:
 555       ValueError: Could not find a trained model in `model_dir`.
 556       ValueError: If batch length of predictions is not the same and
 557         `yield_single_examples` is `True`.
 558       ValueError: If there is a conflict between `predict_keys` and
 559         `predictions`. For example if `predict_keys` is not `None` but
 560         `tf.estimator.EstimatorSpec.predictions` is not a `dict`.
 561     """
 562     with context.graph_mode():
 563       hooks = _check_hooks_type(hooks)
 564       # Check that model has been trained.
 565       if not checkpoint_path:
 566         checkpoint_path = checkpoint_management.latest_checkpoint(
 567             self._model_dir)
 568       if not checkpoint_path:
 569         logging.info('Could not find trained model in model_dir: {}, running '
 570                      'initialization to predict.'.format(self._model_dir))
 571       with ops.Graph().as_default() as g:
 572         random_seed.set_random_seed(self._config.tf_random_seed)
 573         self._create_and_assert_global_step(g)
 574         features, input_hooks = self._get_features_from_input_fn(
 575             input_fn, model_fn_lib.ModeKeys.PREDICT)
 576         estimator_spec = self._call_model_fn(
 577             features, None, model_fn_lib.ModeKeys.PREDICT, self.config)
 578 
 579         # Call to warm_start has to be after model_fn is called.
 580         self._maybe_warm_start(checkpoint_path)
 581 
 582         predictions = self._extract_keys(
 583             estimator_spec.predictions, predict_keys)
 584         all_hooks = list(input_hooks)
 585         all_hooks.extend(hooks)
 586         all_hooks.extend(list(estimator_spec.prediction_hooks or []))
 587         with training.MonitoredSession(
 588             session_creator=training.ChiefSessionCreator(
 589                 checkpoint_filename_with_path=checkpoint_path,
 590                 master=self._config.master,
 591                 scaffold=estimator_spec.scaffold,
 592                 config=self._session_config),
 593             hooks=all_hooks) as mon_sess:
 594           while not mon_sess.should_stop():
 595             preds_evaluated = mon_sess.run(predictions)
 596             if not yield_single_examples:
 597               yield preds_evaluated
 598             elif not isinstance(predictions, dict):
 599               for pred in preds_evaluated:
 600                 yield pred
 601             else:
 602               for i in range(self._extract_batch_length(preds_evaluated)):
 603                 yield {
 604                     key: value[i]
 605                     for key, value in six.iteritems(preds_evaluated)
 606                 }

 791   def _export_all_saved_models(
 792       self, export_dir_base, input_receiver_fn_map,
 793       assets_extra=None,
 794       as_text=False,
 795       checkpoint_path=None,
 796       strip_default_attrs=False):
 797     # pylint: disable=line-too-long
 798     """Exports a `SavedModel` containing `tf.MetaGraphDefs` for each requested mode.
 799 
 800     See `tf.contrib.estimator.export_all_saved_models` for the currently
 801     exposed version of this function.
 802 
 803     For each mode passed in via the `input_receiver_fn_map`,
 804     this method builds a new graph by calling the `input_receiver_fn` to obtain
 805     feature and label `Tensor`s. Next, this method calls the `Estimator`'s
 806     `model_fn` in the passed mode to generate the model graph based on
 807     those features and labels, and restores the given checkpoint
 808     (or, lacking that, the most recent checkpoint) into the graph.
 809     Only one of the modes is used for saving variables to the `SavedModel`
 810     (order of preference: `tf.estimator.ModeKeys.TRAIN`,
 811     `tf.estimator.ModeKeys.EVAL`, then
 812     `tf.estimator.ModeKeys.PREDICT`), such that up to three
 813     `tf.MetaGraphDefs` are saved with a single set of variables in a single
 814     `SavedModel` directory.
 815

 817     below
 818     `export_dir_base`, and writes a `SavedModel` into it containing
 819     the `tf.MetaGraphDef` for the given mode and its associated signatures.
 820

 822     for each element of the `export_outputs` dict returned from the `model_fn`,
 823     named using the same keys.  One of these keys is always
 824     `tf.saved_model.signature_constants.DEFAULT_SERVING_SIGNATURE_DEF_KEY`,
 825     indicating which
 826     signature will be served when a serving request does not specify one.
 827     For each signature, the outputs are provided by the corresponding
 828     `tf.estimator.export.ExportOutput`s, and the inputs are always the input
 829     receivers provided by
 830     the `serving_input_receiver_fn`.
 831

 833     collection,
 834     and loss, metrics, and predictions are included in a `SignatureDef` for the
 835     mode in question.
 836 
 837     Extra assets may be written into the `SavedModel` via the `assets_extra`
 838     argument.  This should be a dict, where each key gives a destination path
 839     (including the filename) relative to the assets.extra directory.  The
 840     corresponding value gives the full path of the source file to be copied.
 841     For example, the simple case of copying a single file without renaming it
 842     is specified as `{'my_asset_file.txt': '/path/to/my_asset_file.txt'}`.

 845       export_dir_base: A string containing a directory in which to create
 846         timestamped subdirectories containing exported `SavedModel`s.
 847       input_receiver_fn_map: dict of `tf.estimator.ModeKeys` to
 848         `input_receiver_fn` mappings, where the `input_receiver_fn` is a
 849         function that takes no arguments and returns the appropriate subclass of
 850         `InputReceiver`.
 851       assets_extra: A dict specifying how to populate the assets.extra directory
 852         within the exported `SavedModel`, or `None` if no extra assets are
 853         needed.
 854       as_text: whether to write the `SavedModel` proto in text format.
 855       checkpoint_path: The checkpoint path to export.  If `None` (the default),
 856         the most recent checkpoint found within the model directory is chosen.
 857       strip_default_attrs: Boolean. If `True`, default-valued attributes will be
 858         removed from the `NodeDef`s. For a detailed guide, see [Stripping
 859         Default-Valued
 860         Attributes](https://github.com/tensorflow/tensorflow/blob/master/tensorflow/python/saved_model/README.md#stripping-default-valued-attributes).
 861 
 862     Returns:
 863       A dict of `tf.estimator.ModeKeys` value to string path for each exported
 864       directory.
 865 
 866     Raises:
 867       ValueError: if any `input_receiver_fn` is `None`, no `export_outputs`
 868         are provided, or no checkpoint can be found.
 869     """
 870     # pylint: enable=line-too-long
 871     # TODO(b/65561022): Consider allowing multiple input_receiver_fns per mode.
 872     with context.graph_mode():
 873       if not checkpoint_path:
 874         # Locate the latest checkpoint
 875         checkpoint_path = checkpoint_management.latest_checkpoint(
 876             self._model_dir)
 877       if not checkpoint_path:
 878         raise ValueError("Couldn't find trained model at %s." % self._model_dir)
 879 
 880       export_dir = export_helpers.get_timestamped_export_dir(export_dir_base)
 881       temp_export_dir = export_helpers.get_temp_export_dir(export_dir)
 882 
 883       builder = saved_model_builder.SavedModelBuilder(temp_export_dir)
 884 
 885       save_variables = True
 886       # Note that the order in which we run here matters, as the first
 887       # mode we pass through will be used to save the variables. We run TRAIN
 888       # first, as that is also the mode used for checkpoints, and therefore
 889       # we are not likely to have vars in PREDICT that are not in the checkpoint
 890       # created by TRAIN.
 891       if input_receiver_fn_map.get(model_fn_lib.ModeKeys.TRAIN):
 892         self._add_meta_graph_for_mode(
 893             builder, input_receiver_fn_map, checkpoint_path,
 894             strip_default_attrs, save_variables,
 895             mode=model_fn_lib.ModeKeys.TRAIN)
 896         save_variables = False
 897       if input_receiver_fn_map.get(model_fn_lib.ModeKeys.EVAL):
 898         self._add_meta_graph_for_mode(
 899             builder, input_receiver_fn_map, checkpoint_path,
 900             strip_default_attrs, save_variables,
 901             mode=model_fn_lib.ModeKeys.EVAL)
 902         save_variables = False
 903       if input_receiver_fn_map.get(model_fn_lib.ModeKeys.PREDICT):
 904         self._add_meta_graph_for_mode(
 905             builder, input_receiver_fn_map, checkpoint_path,
 906             strip_default_attrs, save_variables,
 907             mode=model_fn_lib.ModeKeys.PREDICT)
 908         save_variables = False
 909 
 910       if save_variables:
 911         raise ValueError('No valid modes for exporting found. Got {}.'.format(
 912             input_receiver_fn_map.keys()))
 913 
 914       builder.save(as_text)
 915 
 916       # Add the extra assets
 917       if assets_extra:
 918         assets_extra_path = os.path.join(compat.as_bytes(temp_export_dir),
 919                                          compat.as_bytes('assets.extra'))
 920         for dest_relative, source in assets_extra.items():
 921           dest_absolute = os.path.join(compat.as_bytes(assets_extra_path),
 922                                        compat.as_bytes(dest_relative))
 923           dest_path = os.path.dirname(dest_absolute)
 924           gfile.MakeDirs(dest_path)
 925           gfile.Copy(source, dest_absolute)
 926 
 927       gfile.Rename(temp_export_dir, export_dir)
 928       return export_dir

 930   def _add_meta_graph_for_mode(self,
 931                                builder,
 932                                input_receiver_fn_map,
 933                                checkpoint_path,
 934                                strip_default_attrs,
 935                                save_variables=True,
 936                                mode=model_fn_lib.ModeKeys.PREDICT,
 937                                export_tags=None,
 938                                check_variables=True):
 939     # pylint: disable=line-too-long
 940     """Loads variables and adds them along with a `tf.MetaGraphDef` for saving.
 941 
 942     Args:
 943       builder: instance of `tf.saved_modle.builder.SavedModelBuilder` that will
 944         be used for saving.
 945       input_receiver_fn_map: dict of `tf.estimator.ModeKeys` to
 946         `input_receiver_fn` mappings, where the `input_receiver_fn` is a
 947         function that takes no argument and returns the appropriate subclass of
 948         `InputReceiver`.
 949       checkpoint_path: The checkpoint path to export.  If `None` (the default),
 950         the most recent checkpoint found within the model directory is chosen.
 951       strip_default_attrs: Boolean. If `True`, default-valued attributes will be
 952         removed from the `NodeDef`s. For a detailed guide, see [Stripping
 953         Default-Valued
 954         Attributes](https://github.com/tensorflow/tensorflow/blob/master/tensorflow/python/saved_model/README.md#stripping-default-valued-attributes).
 955       save_variables: bool, whether variables should be saved. If `False`, just
 956         the `tf.MetaGraphDef` will be saved. Note that `save_variables` should
 957         only be `True` for the first call to this function, and the
 958         `SavedModelBuilder` will raise an error if that is not the case.
 959       mode: `tf.estimator.ModeKeys` value indicating which mode will be
 960         exported.
 961       export_tags: The set of tags with which to save `tf.MetaGraphDef`. If
 962         `None`, a default set will be selected to matched the passed mode.
 963       check_variables: bool, whether to check the checkpoint has all variables.
 964 
 965     Raises:
 966       ValueError: if `save_variables` is `True` and `check_variable` is `False`.
 967     """
 968     # pylint: enable=line-too-long
 969     if export_tags is None:
 970       export_tags = model_fn_lib.EXPORT_TAG_MAP[mode]
 971     input_receiver_fn = input_receiver_fn_map[mode]
 972 
 973     with ops.Graph().as_default() as g:
 974       self._create_and_assert_global_step(g)
 975       random_seed.set_random_seed(self._config.tf_random_seed)
 976 
 977       input_receiver = input_receiver_fn()
 978 
 979       # Call the model_fn and collect the export_outputs.
 980       estimator_spec = self._call_model_fn(
 981           features=input_receiver.features,
 982           labels=getattr(input_receiver, 'labels', None),
 983           mode=mode,
 984           config=self.config)
 985 
 986       export_outputs = model_fn_lib.export_outputs_for_mode(
 987           mode=estimator_spec.mode,
 988           serving_export_outputs=estimator_spec.export_outputs,
 989           predictions=estimator_spec.predictions,
 990           loss=estimator_spec.loss,
 991           metrics=estimator_spec.eval_metric_ops)
 992 
 993       # Build the SignatureDefs from receivers and all outputs
 994       signature_def_map = export_helpers.build_all_signature_defs(
 995           input_receiver.receiver_tensors,
 996           export_outputs,
 997           getattr(input_receiver, 'receiver_tensors_alternatives', None),
 998           serving_only=(mode == model_fn_lib.ModeKeys.PREDICT))
 999 
1000       with tf_session.Session(config=self._session_config) as session:
1001 
1002         if estimator_spec.scaffold.local_init_op is not None:
1003           local_init_op = estimator_spec.scaffold.local_init_op
1004         else:
1005           local_init_op = monitored_session.Scaffold.default_local_init_op()
1006 
1007         # This saver will be used both for restoring variables now,
1008         # and in saving out the metagraph below. This ensures that any
1009         # Custom Savers stored with the Scaffold are passed through to the
1010         # SavedModel for restore later.
1011         graph_saver = estimator_spec.scaffold.saver or saver.Saver(sharded=True)
1012 
1013         if save_variables and not check_variables:
1014           raise ValueError('If `save_variables` is `True, `check_variables`'
1015                            'must not be `False`.')

1022                    'that were not saved in the checkpoint.\n\n'

1024                    'checkpoint from: `{}`. Full Traceback:\n\n{}').format(

1027 
1028         # We add the train op explicitly for now, so that we don't have to
1029         # change the Builder public interface. Note that this is a no-op
1030         # for prediction, where train_op is None.
1031         builder._add_train_op(estimator_spec.train_op)  # pylint: disable=protected-access
1032 
1033         meta_graph_kwargs = dict(
1034             tags=export_tags,
1035             signature_def_map=signature_def_map,
1036             assets_collection=ops.get_collection(
1037                 ops.GraphKeys.ASSET_FILEPATHS),
1038             strip_default_attrs=strip_default_attrs,
1039             legacy_init_op=local_init_op,
1040             saver=graph_saver)
1041 
1042         if save_variables:
1043           builder.add_meta_graph_and_variables(
1044               session, **meta_graph_kwargs)
1045         else:
1046           builder.add_meta_graph(**meta_graph_kwargs)

1132   def _call_input_fn(self, input_fn, mode):
1133     """Calls the input function.
1134 
1135     Args:
1136       input_fn: The input function.
1137       mode: `tf.estimator.ModeKeys`
1138 
1139     Returns:
1140       The return value of the passed `input_fn`, which should be one of:
1141 
1142         * A 'tf.data.Dataset' object: Outputs of `Dataset` object must be a
1143             tuple `(features, labels)` with same constraints as below.
1144         * A tuple `(features, labels)`: Where `features` is a `Tensor` or a

1147           `features` and `labels` are consumed by `model_fn`. They should

1155     if 'mode' in input_fn_args:
1156       kwargs['mode'] = mode
1157     if 'params' in input_fn_args:
1158       kwargs['params'] = self.params
1159     if 'config' in input_fn_args:
1160       kwargs['config'] = self.config
1161     with ops.device('/cpu:0'):
1162       return input_fn(**kwargs)

1209   def _train_model_default(self, input_fn, hooks, saving_listeners):
1210     """Initiate training with `input_fn`, without `DistributionStrategies`.
1211 
1212     Args:
1213       input_fn: A function that provides input data for training as minibatches.
1214       hooks: List of `tf.train.SessionRunHook` subclass instances. Used for
1215         callbacks inside the training loop.
1216       saving_listeners: list of `tf.train.CheckpointSaverListener` objects. Used
1217         for callbacks that run immediately before or after checkpoint savings.
1218 
1219     Returns:
1220       Loss from training
1221     """
1222     worker_hooks = []
1223     with ops.Graph().as_default() as g, g.device(self._device_fn):
1224       random_seed.set_random_seed(self._config.tf_random_seed)
1225       global_step_tensor = self._create_and_assert_global_step(g)
1226 
1227       # Skip creating a read variable if _create_and_assert_global_step
1228       # returns None (e.g. tf.contrib.estimator.SavedModelEstimator).
1229       if global_step_tensor is not None:
1230         training_util._get_or_create_global_step_read(g)  # pylint: disable=protected-access
1231 
1232       features, labels, input_hooks = (
1233           self._get_features_and_labels_from_input_fn(

1236       estimator_spec = self._call_model_fn(
1237           features, labels, model_fn_lib.ModeKeys.TRAIN, self.config)
1238       global_step_tensor = training_util.get_global_step(g)
1239       return self._train_with_estimator_spec(estimator_spec, worker_hooks,
1240                                              hooks, global_step_tensor,

1243   def _train_model_distributed(self, input_fn, hooks, saving_listeners):
1244     """Initiate training with `input_fn`, using `DistributionStrategies`.
1245 
1246     Args:
1247       input_fn: A function that provides input data for training as minibatches.
1248       hooks: List of `tf.train.SessionRunHook` subclass instances. Used for
1249         callbacks inside the training loop.
1250       saving_listeners: list of `tf.train.CheckpointSaverListener` objects. Used
1251         for callbacks that run immediately before or after checkpoint savings.
1252 
1253     Returns:
1254       Loss from training
1255     """
1256     self._train_distribution.configure(self._session_config)
1257 
1258     # TODO(sourabhbajaj): Remove this hack once we migrate the other strategies
1259     # to use the new API
1260     is_tpu_strategy = (
1261         self._train_distribution.__class__.__name__ == 'TPUStrategy')
1262 
1263     worker_hooks = []
1264     with ops.Graph().as_default() as g:
1265       # We want to create the iterations variable outside the distribution scope
1266       # as that is just stored on the host and mainly used to drive the loop
1267       # and doesn't need to be a Mirrored/Device variable.
1268       if is_tpu_strategy:
1269         steps_per_run_variable = training.get_or_create_steps_per_run_variable()
1270       with self._train_distribution.scope():
1271         random_seed.set_random_seed(self._config.tf_random_seed)
1272         iterator, input_hooks = self._get_iterator_from_input_fn(
1273             input_fn, model_fn_lib.ModeKeys.TRAIN, self._train_distribution)
1274         worker_hooks.extend(input_hooks)
1275         global_step_tensor = self._create_and_assert_global_step(g)
1276         # we want to add to the global collection in the main thread not the
1277         # tower threads.
1278         ops.add_to_collection(
1279             training_util.GLOBAL_STEP_READ_KEY,
1280             self._train_distribution.read_var(global_step_tensor))
1281 
1282         if is_tpu_strategy:
1283           # Create a step_fn from the train_op of grouped_estimator_spec
1284           def step_fn(ctx, features, labels=None):
1285             """A single step that is passed to run_on_dataset."""
1286             estimator_spec = self._train_distribution.call_for_each_tower(
1287                 self._call_model_fn,
1288                 features,
1289                 labels,
1290                 model_fn_lib.ModeKeys.TRAIN,
1291                 self.config)
1292             ctx.set_last_step_output(
1293                 name='loss',
1294                 output=estimator_spec.loss,
1295                 aggregation=distribute_lib.get_loss_reduction())
1296             ctx.set_non_tensor_output(
1297                 name='estimator_spec', output=estimator_spec)
1298             return estimator_spec.train_op
1299 
1300           # Create new train_op post graph rewrites
1301           initial_training_loss = constant_op.constant(1e7)
1302           ctx = self._train_distribution.run_steps_on_dataset(
1303               step_fn, iterator, iterations=steps_per_run_variable,
1304               initial_loop_values={'loss': initial_training_loss})
1305           distributed_train_op = ctx.run_op
1306           loss = ctx.last_step_outputs['loss']
1307           grouped_estimator_spec = ctx.non_tensor_outputs['estimator_spec']
1308         else:
1309           features, labels = estimator_util.parse_iterator_result(
1310               iterator.get_next())
1311           grouped_estimator_spec = self._train_distribution.call_for_each_tower(
1312               self._call_model_fn,
1313               features,
1314               labels,  # although this will be None it seems
1315               model_fn_lib.ModeKeys.TRAIN,
1316               self.config)
1317           loss = self._train_distribution.unwrap(
1318               self._train_distribution.reduce(
1319                   distribute_lib.get_loss_reduction(),
1320                   grouped_estimator_spec.loss,
1321                   destinations='/device:CPU:0'))[0]
1322           distributed_train_op = grouped_estimator_spec.train_op
1323 
1324         scaffold = _combine_distributed_scaffold(
1325             grouped_estimator_spec.scaffold, self._train_distribution)
1326 
1327         # TODO(yuefengz): add a test for unwrapping per_device_hooks.
1328         def get_hooks_from_the_first_device(per_device_hooks):
1329           return [
1330               self._distribution.unwrap(per_device_hook)[0]
1331               for per_device_hook in per_device_hooks
1332           ]
1333 
1334         training_hooks = get_hooks_from_the_first_device(
1335             grouped_estimator_spec.training_hooks)
1336         training_chief_hooks = get_hooks_from_the_first_device(
1337             grouped_estimator_spec.training_chief_hooks)
1338         worker_hooks.append(
1339             estimator_util.StrategyInitFinalizeHook(
1340                 self._train_distribution.initialize,
1341                 self._train_distribution.finalize))
1342 
1343         estimator_spec = model_fn_lib.EstimatorSpec(
1344             mode=grouped_estimator_spec.mode,
1345             loss=loss,
1346             train_op=self._train_distribution.group(distributed_train_op),
1347             training_hooks=training_hooks,
1348             training_chief_hooks=training_chief_hooks,
1349             scaffold=scaffold)
1350         return self._train_with_estimator_spec(estimator_spec, worker_hooks,
1351                                                hooks, global_step_tensor,
1352                                                saving_listeners)

1354   def _train_with_estimator_spec(self, estimator_spec, worker_hooks, hooks,
1355                                  global_step_tensor, saving_listeners):
1356     """Train a model with the given Estimator Spec."""
1357     if self._warm_start_settings:
1358       logging.info('Warm-starting with WarmStartSettings: %s' %
1359                    (self._warm_start_settings,))
1360       warm_starting_util.warm_start(*self._warm_start_settings)
1361     # Check if the user created a loss summary, and add one if they didn't.
1362     # We assume here that the summary is called 'loss'. If it is not, we will
1363     # make another one with the name 'loss' to ensure it shows up in the right
1364     # graph in TensorBoard.
1365     if not any([x.op.name == 'loss'
1366                 for x in ops.get_collection(ops.GraphKeys.SUMMARIES)]):
1367       summary.scalar('loss', estimator_spec.loss)
1368     ops.add_to_collection(ops.GraphKeys.LOSSES, estimator_spec.loss)
1369     worker_hooks.extend(hooks)
1370     worker_hooks.append(
1371         training.NanTensorHook(estimator_spec.loss)
1372     )
1373     if self._config.log_step_count_steps is not None:
1374       worker_hooks.append(
1375           training.LoggingTensorHook(
1376               {
1377                   'loss': estimator_spec.loss,
1378                   'step': global_step_tensor
1379               },
1380               every_n_iter=self._config.log_step_count_steps)
1381       )
1382     worker_hooks.extend(estimator_spec.training_hooks)
1383 
1384     if not (estimator_spec.scaffold.saver or
1385             ops.get_collection(ops.GraphKeys.SAVERS)):
1386       ops.add_to_collection(
1387           ops.GraphKeys.SAVERS,
1388           training.Saver(
1389               sharded=True,
1390               max_to_keep=self._config.keep_checkpoint_max,
1391               keep_checkpoint_every_n_hours=(
1392                   self._config.keep_checkpoint_every_n_hours),
1393               defer_build=True,
1394               save_relative_paths=True))
1395 
1396     chief_hooks = []
1397     all_hooks = worker_hooks + list(estimator_spec.training_chief_hooks)
1398     saver_hooks = [
1399         h for h in all_hooks if isinstance(h, training.CheckpointSaverHook)]
1400     if (self._config.save_checkpoints_secs or
1401         self._config.save_checkpoints_steps):
1402       if not saver_hooks:
1403         chief_hooks = [
1404             training.CheckpointSaverHook(
1405                 self._model_dir,
1406                 save_secs=self._config.save_checkpoints_secs,
1407                 save_steps=self._config.save_checkpoints_steps,
1408                 scaffold=estimator_spec.scaffold)
1409         ]
1410         saver_hooks = [chief_hooks[0]]
1411     if saving_listeners:
1412       if not saver_hooks:
1413         raise ValueError(
1414             'There should be a CheckpointSaverHook to use saving_listeners. '
1415             'Please set one of the RunConfig.save_checkpoints_steps or '
1416             'RunConfig.save_checkpoints_secs.')
1417       else:
1418         # It is expected to have one CheckpointSaverHook. If multiple, we pick
1419         # up the first one to add listener.
1420         saver_hooks[0]._listeners.extend(saving_listeners)  # pylint: disable=protected-access
1421 
1422     # Add summary hooks to worker 0 if we are running with a master, to ensure
1423     # that summaries are written at correct intervals even with long-running
1424     # evaluations.
1425     save_summary_steps = self._config.save_summary_steps
1426     log_step_count_steps = self._config.log_step_count_steps
1427 
1428     # Check existence of appropriate cluster spec fields, as well as master and
1429     # worker nodes. As master also performs evaluation, summary writing must
1430     # occur on a different node. The presence of a worker is also checked to
1431     # prevent reassigning hooks for single-replica jobs with just a master node.
1432     if (self._config.cluster_spec and self._config.cluster_spec.jobs and
1433         (run_config.TaskType.WORKER in self._config.cluster_spec.jobs) and
1434         (run_config.TaskType.MASTER in self._config.cluster_spec.jobs)):
1435       # Update config values to prevent the default hooks from being created on
1436       # the master or other workers.
1437       save_summary_steps = 0
1438       log_step_count_steps = None
1439 
1440       if (self._config.task_type == run_config.TaskType.WORKER and
1441           self._config.task_id == 0):
1442         if (self._config.save_summary_steps and
1443             self._config.save_summary_steps > 0):
1444           worker_hooks.append(
1445               training.SummarySaverHook(
1446                   save_steps=self._config.save_summary_steps,
1447                   output_dir=self._config.model_dir,
1448                   scaffold=estimator_spec.scaffold))
1449 
1450         if (self._config.log_step_count_steps and
1451             self._config.log_step_count_steps > 0):
1452           worker_hooks.append(
1453               training.StepCounterHook(
1454                   every_n_steps=self._config.log_step_count_steps,
1455                   output_dir=self._config.model_dir))
1456 
1457     with training.MonitoredTrainingSession(
1458         master=self._config.master,
1459         is_chief=self._config.is_chief,
1460         checkpoint_dir=self._model_dir,
1461         scaffold=estimator_spec.scaffold,
1462         hooks=worker_hooks,
1463         chief_only_hooks=(
1464             tuple(chief_hooks) + tuple(estimator_spec.training_chief_hooks)),
1465         save_checkpoint_secs=0,  # Saving is handled by a hook.
1466         save_summaries_steps=save_summary_steps,
1467         config=self._session_config,
1468         log_step_count_steps=log_step_count_steps) as mon_sess:
1469       loss = None
1470       while not mon_sess.should_stop():
1471         _, loss = mon_sess.run([estimator_spec.train_op, estimator_spec.loss])
1472     return loss

1474   def _evaluate_build_graph(self, input_fn, hooks=None, checkpoint_path=None):
1475     """Builds the graph and related hooks to run evaluation."""

1477     self._create_and_assert_global_step(ops.get_default_graph())
1478 
1479     if self._eval_distribution:
1480       (scaffold, evaluation_hooks, input_hooks, update_op, eval_dict) = (
1481           self._call_model_fn_eval_distributed(input_fn, self.config))
1482     else:
1483       (scaffold, evaluation_hooks, input_hooks, update_op, eval_dict) = (
1484           self._call_model_fn_eval(input_fn, self.config))
1485 
1486     global_step_tensor = training_util.get_global_step(ops.get_default_graph())
1487     # Call to warm_start has to be after model_fn is called.

1491       raise ValueError(
1492           'Metric with name `global_step` is not allowed, because Estimator '

1501     if scaffold and scaffold.local_init_op:
1502       # Ensure that eval step has been created before updating local init op.
1503       evaluation._get_or_create_eval_step()  # pylint: disable=protected-access
1504 
1505       scaffold = monitored_session.Scaffold(
1506           local_init_op=control_flow_ops.group(
1507               scaffold.local_init_op,
1508               monitored_session.Scaffold.default_local_init_op()),
1509           copy_from_scaffold=scaffold
1510       )

1877 def _write_dict_to_summary(output_dir,
1878                            dictionary,
1879                            current_global_step):
1880   """Writes a `dict` into summary file in given output directory.
1881 
1882   Args:
1883     output_dir: `str`, directory to write the summary file in.
1884     dictionary: the `dict` to be written to summary file.
1885     current_global_step: `int`, the current global step.
1886   """
1887   logging.info('Saving dict for global step %d: %s', current_global_step,
1888                _dict_to_str(dictionary))
1889   summary_writer = writer_cache.FileWriterCache.get(output_dir)
1890   summary_proto = summary_pb2.Summary()
1891   for key in dictionary:
1892     if dictionary[key] is None:
1893       continue
1894     if key == 'global_step':
1895       continue
1896     if (isinstance(dictionary[key], np.float32) or
1897         isinstance(dictionary[key], float)):
1898       summary_proto.value.add(tag=key, simple_value=float(dictionary[key]))
1899     elif (isinstance(dictionary[key], np.int64) or
1900           isinstance(dictionary[key], np.int32) or
1901           isinstance(dictionary[key], int)):
1902       summary_proto.value.add(tag=key, simple_value=int(dictionary[key]))
1903     elif isinstance(dictionary[key], six.binary_type):
1904       try:
1905         summ = summary_pb2.Summary.FromString(dictionary[key])
1906         for i, _ in enumerate(summ.value):
1907           summ.value[i].tag = '%s/%d' % (key, i)
1908         summary_proto.value.extend(summ.value)
1909       except message.DecodeError:
1910         logging.warn('Skipping summary for %s, cannot parse string to Summary.',
1911                      key)
1912         continue
1913     elif isinstance(dictionary[key], np.ndarray):
1914       value = summary_proto.value.add()
1915       value.tag = key
1916       value.node_name = key
1917       tensor_proto = tensor_util.make_tensor_proto(dictionary[key])
1918       value.tensor.CopyFrom(tensor_proto)
1919       # pylint: disable=line-too-long
1920       logging.info(
1921           'Summary for np.ndarray is not visible in Tensorboard by default. '
1922           'Consider using a Tensorboard plugin for visualization (see '
1923           'https://github.com/tensorflow/tensorboard-plugin-example/blob/master/README.md'
1924           ' for more information).')
1925       # pylint: enable=line-too-long
1926     else:
1927       logging.warn(
1928           'Skipping summary for %s, must be a float, np.float32, np.int64, '
1929           'np.int32 or int or np.ndarray or a serialized string of Summary.',
1930           key)
1931   summary_writer.add_summary(summary_proto, current_global_step)
1932   summary_writer.flush()
