  33 # When we create a timestamped directory, there is a small chance that the
  34 # directory already exists because another process is also creating these
  35 # directories. In this case we just wait one second to get a new timestamp and
  36 # try again. If this fails several times in a row, then something is seriously
  37 # wrong.
  38 MAX_DIRECTORY_CREATION_ATTEMPTS = 10
  39 
  40 
  41 def get_timestamped_dir(dir_base):
  42   """Builds a path to a new subdirectory within the base directory.
  43 
  44   The subdirectory will be named using the current time.
  45   This guarantees monotonically increasing directory numbers even across
  46   multiple runs of the pipeline.
  47   The timestamp used is the number of seconds since epoch UTC.
  48 
  49   Args:
  50     dir_base: A string containing a directory to create the subdirectory under.
  51 
  52   Returns:
  53     The full path of the new subdirectory (which is not actually created yet).
  54 
  55   Raises:
  56     RuntimeError: if repeated attempts fail to obtain a unique timestamped
  57       directory name.
  58   """
  59   attempts = 0
  60   while attempts < MAX_DIRECTORY_CREATION_ATTEMPTS:
  61     timestamp = int(time.time())
  62 
  63     result_dir = os.path.join(
  64         compat.as_bytes(dir_base), compat.as_bytes(str(timestamp)))
  65     if not gfile.Exists(result_dir):
  66       # Collisions are still possible (though extremely unlikely): this
  67       # directory is not actually created yet, but it will be almost
  68       # instantly on return from this function.
  69       return result_dir
  70     time.sleep(1)
  71     attempts += 1
  72     logging.warn('Directory {} already exists; retrying (attempt {}/{})'.format(
  73         result_dir, attempts, MAX_DIRECTORY_CREATION_ATTEMPTS))
  74   raise RuntimeError('Failed to obtain a unique export directory name after '
  75                      '{} attempts.'.format(MAX_DIRECTORY_CREATION_ATTEMPTS))

  78 def parse_input_fn_result(result):
  79   """Gets features, labels, and hooks from the result of an Estimator input_fn.
  80 
  81   Args:
  82     result: output of an input_fn to an estimator, which should be one of:
  83 
  84       * A 'tf.data.Dataset' object: Outputs of `Dataset` object must be a
  85           tuple (features, labels) with same constraints as below.
  86       * A tuple (features, labels): Where `features` is a `Tensor` or a
  87         dictionary of string feature name to `Tensor` and `labels` is a
  88         `Tensor` or a dictionary of string label name to `Tensor`. Both
  89         `features` and `labels` are consumed by `model_fn`. They should
  90         satisfy the expectation of `model_fn` from inputs.
  91 
  92   Returns:
  93     Tuple of features, labels, and input_hooks, where features are as described
  94     above, labels are as described above or None, and input_hooks are a list
  95     of SessionRunHooks to be included when running.
  96 
  97   Raises:
  98     ValueError: if the result is a list or tuple of length != 2.
  99   """
 100   input_hooks = []
 101   try:
 102     # We can't just check whether this is a tf.data.Dataset instance here,
 103     # as this is plausibly a PerDeviceDataset. Try treating as a dataset first.
 104     iterator = result.make_initializable_iterator()
 105   except AttributeError:
 106     # Not a dataset or dataset-like-object. Move along.
 107     pass
 108   else:
 109     input_hooks.append(_DatasetInitializerHook(iterator))
 110     result = iterator.get_next()
 111   return parse_iterator_result(result) + (input_hooks,)
 112 
 113 
 114 def parse_iterator_result(result):
 115   """Gets features, labels from result."""
 116   if isinstance(result, (list, tuple)):
 117     if len(result) != 2:
 118       raise ValueError(
 119           'input_fn should return (features, labels) as a len 2 tuple.')
 120     return result[0], result[1]
 121   return result, None

 124 class _DatasetInitializerHook(training.SessionRunHook):
 125   """Creates a SessionRunHook that initializes the passed iterator."""
 126 
 127   def __init__(self, iterator):
 128     self._iterator = iterator
 129 
 130   def begin(self):
 131     self._initializer = self._iterator.initializer
 132 
 133   def after_create_session(self, session, coord):
 134     del coord
 135     session.run(self._initializer)
 136 
 137 
 138 class MultiHostDatasetInitializerHook(training.SessionRunHook):
 139   """Creates a SessionRunHook that initializes all passed iterators."""
 140 
 141   def __init__(self, dataset_initializers):
 142     self._initializers = dataset_initializers
 143 
 144   def after_create_session(self, session, coord):
 145     del coord
 146     start = time.time()
 147     session.run(self._initializers)
 148     logging.info('Initialized dataset iterators in %d seconds',
 149                  time.time() - start)

 151 
 152 class StrategyInitFinalizeHook(training.SessionRunHook):
 153   """Creates a SessionRunHook that initializes and shutsdown devices."""
 154 
 155   def __init__(self, initialization_fn, finalize_fn):
 156     self._initialization_fn = initialization_fn
 157     self._finalize_fn = finalize_fn
 158 
 159   def begin(self):
 160     # We only create the init ops, but don't run it. We rely on SessionManager
 161     # to run it for us.
 162     self._init_ops = self._initialization_fn()
 163     self._finalize_ops = self._finalize_fn()
 164 
 165   def end(self, session):
 166     logging.info('Finalize system.')
 167     session.run(self._finalize_ops)
