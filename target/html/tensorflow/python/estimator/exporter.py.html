  34 @estimator_export('estimator.Exporter')
  35 class Exporter(object):
  36   """A class representing a type of model export."""
  37 
  38   @abc.abstractproperty
  39   def name(self):
  40     """Directory name.
  41 
  42     A directory name under the export base directory where exports of
  43     this type are written.  Should not be `None` nor empty.
  44     """
  45     pass
  46 
  47   @abc.abstractmethod
  48   def export(self, estimator, export_path, checkpoint_path, eval_result,
  49              is_the_final_export):
  50     """Exports the given `Estimator` to a specific format.
  51 
  52     Args:
  53       estimator: the `Estimator` to export.
  54       export_path: A string containing a directory where to write the export.
  55       checkpoint_path: The checkpoint path to export.
  56       eval_result: The output of `Estimator.evaluate` on this checkpoint.
  57       is_the_final_export: This boolean is True when this is an export in the
  58         end of training.  It is False for the intermediate exports during
  59         the training.
  60         When passing `Exporter` to `tf.estimator.train_and_evaluate`
  61         `is_the_final_export` is always False if `TrainSpec.max_steps` is
  62         `None`.
  63 
  64     Returns:
  65       The string path to the exported directory or `None` if export is skipped.
  66     """
  67     pass

  70 class _SavedModelExporter(Exporter):
  71   """This class exports the serving graph and checkpoints.
  72 
  73      This class provides a basic exporting functionality and serves as a
  74      foundation for specialized `Exporter`s.
  75   """
  76 
  77   def __init__(self,
  78                name,
  79                serving_input_receiver_fn,
  80                assets_extra=None,
  81                as_text=False,
  82                strip_default_attrs=True):
  83     """Create an `Exporter` to use with `tf.estimator.EvalSpec`.
  84 
  85     Args:
  86       name: unique name of this `Exporter` that is going to be used in the
  87         export path.
  88       serving_input_receiver_fn: a function that takes no arguments and returns
  89         a `ServingInputReceiver`.
  90       assets_extra: An optional dict specifying how to populate the assets.extra
  91         directory within the exported SavedModel.  Each key should give the
  92         destination path (including the filename) relative to the assets.extra
  93         directory.  The corresponding value gives the full path of the source
  94         file to be copied.  For example, the simple case of copying a single
  95         file without renaming it is specified as
  96         `{'my_asset_file.txt': '/path/to/my_asset_file.txt'}`.
  97       as_text: whether to write the SavedModel proto in text format. Defaults to
  98         `False`.
  99       strip_default_attrs: Boolean. If set, default attrs in the `GraphDef` will
 100         be stripped on write. This is the default behavior and recommended for
 101         better forward compatibility of the resulting `SavedModel`.
 102 
 103     Raises:
 104       ValueError: if any arguments is invalid.
 105     """
 106     self._name = name
 107     self._serving_input_receiver_fn = serving_input_receiver_fn
 108     self._assets_extra = assets_extra
 109     self._as_text = as_text
 110     self._strip_default_attrs = strip_default_attrs
 111 
 112   @property
 113   def name(self):
 114     return self._name
 115 
 116   def export(self, estimator, export_path, checkpoint_path, eval_result,
 117              is_the_final_export):
 118     del is_the_final_export
 119 
 120     export_result = estimator.export_savedmodel(
 121         export_path,
 122         self._serving_input_receiver_fn,
 123         assets_extra=self._assets_extra,
 124         as_text=self._as_text,
 125         checkpoint_path=checkpoint_path,
 126         strip_default_attrs=self._strip_default_attrs)
 127 
 128     return export_result

 131 def _loss_smaller(best_eval_result, current_eval_result):
 132   """Compares two evaluation results and returns true if the 2nd one is smaller.
 133 
 134   Both evaluation results should have the values for MetricKeys.LOSS, which are
 135   used for comparison.
 136 
 137   Args:
 138     best_eval_result: best eval metrics.
 139     current_eval_result: current eval metrics.
 140 
 141   Returns:
 142     True if the loss of current_eval_result is smaller; otherwise, False.
 143 
 144   Raises:
 145     ValueError: If input eval result is None or no loss is available.
 146   """
 147   default_key = metric_keys.MetricKeys.LOSS
 148   if not best_eval_result or default_key not in best_eval_result:
 149     raise ValueError(
 150         'best_eval_result cannot be empty or no loss is found in it.')
 151 
 152   if not current_eval_result or default_key not in current_eval_result:
 153     raise ValueError(
 154         'current_eval_result cannot be empty or no loss is found in it.')
 155 
 156   return best_eval_result[default_key] > current_eval_result[default_key]
 157 
 158 
 159 def _verify_compare_fn_args(compare_fn):
 160   """Verifies compare_fn arguments."""
 161   args = set(util.fn_args(compare_fn))
 162   if 'best_eval_result' not in args:
 163     raise ValueError(
 164         'compare_fn (%s) must include best_eval_result argument.' % compare_fn)
 165   if 'current_eval_result' not in args:
 166     raise ValueError(
 167         'compare_fn (%s) must include current_eval_result argument.' %
 168         compare_fn)
 169   non_valid_args = list(args - set(['best_eval_result', 'current_eval_result']))
 170   if non_valid_args:
 171     raise ValueError('compare_fn (%s) has following not expected args: %s' %
 172                      (compare_fn, non_valid_args))

 175 @estimator_export('estimator.BestExporter')
 176 class BestExporter(Exporter):
 177   """This class exports the serving graph and checkpoints of the best models.
 178 
 179   This class performs a model export everytime when the new model is better
 180   than any exsiting model.
 181   """
 182 
 183   def __init__(self,
 184                name='best_exporter',
 185                serving_input_receiver_fn=None,
 186                event_file_pattern='eval/*.tfevents.*',
 187                compare_fn=_loss_smaller,
 188                assets_extra=None,
 189                as_text=False,
 190                exports_to_keep=5):
 191     """Create an `Exporter` to use with `tf.estimator.EvalSpec`.
 192 
 193     Example of creating a BestExporter for training and evluation:
 194     ```python
 195     def make_train_and_eval_fn():
 196       # Set up feature columns.
 197       categorial_feature_a = (
 198           tf.feature_column.categorical_column_with_hash_bucket(...))
 199       categorial_feature_a_emb = embedding_column(
 200           categorical_column=categorial_feature_a, ...)
 201       ...  # other feature columns
 202 
 203       estimator = tf.estimator.DNNClassifier(
 204           config=tf.estimator.RunConfig(
 205               model_dir='/my_model', save_summary_steps=100),
 206           feature_columns=[categorial_feature_a_emb, ...],
 207           hidden_units=[1024, 512, 256])
 208 
 209       serving_feature_spec = tf.feature_column.make_parse_example_spec(
 210           categorial_feature_a_emb)
 211       serving_input_receiver_fn = (
 212           tf.estimator.export.build_parsing_serving_input_receiver_fn(
 213           serving_feature_spec))
 214 
 215       exporter = tf.estimator.BestExporter(
 216           name="best_exporter",
 217           serving_input_receiver_fn=serving_input_receiver_fn,
 218           exports_to_keep=5)
 219 
 220       train_spec = tf.estimator.TrainSpec(...)
 221 
 222       eval_spec = [tf.estimator.EvalSpec(
 223         input_fn=eval_input_fn,
 224         steps=100,
 225         exporters=exporter,
 226         start_delay_secs=0,
 227         throttle_secs=5)]
 228 
 229       return tf.estimator.DistributedTrainingSpec(estimator, train_spec,
 230                                                   eval_spec)
 231     ```
 232 
 233     Args:
 234       name: unique name of this `Exporter` that is going to be used in the
 235         export path.
 236       serving_input_receiver_fn: a function that takes no arguments and returns
 237         a `ServingInputReceiver`.
 238       event_file_pattern: event file name pattern relative to model_dir. If
 239         None, however, the exporter would not be preemption-safe. To be
 240         preemption-safe, event_file_pattern should be specified.
 241       compare_fn: a function that compares two evaluation results and returns
 242         true if current evaluation result is better. Follows the signature:
 243         * Args:
 244           * `best_eval_result`: This is the evaluation result of the best model.
 245           * `current_eval_result`: This is the evaluation result of current
 246                  candidate model.
 247         * Returns:
 248           True if current evaluation result is better; otherwise, False.
 249       assets_extra: An optional dict specifying how to populate the assets.extra
 250         directory within the exported SavedModel.  Each key should give the
 251         destination path (including the filename) relative to the assets.extra
 252         directory.  The corresponding value gives the full path of the source
 253         file to be copied.  For example, the simple case of copying a single
 254         file without renaming it is specified as `{'my_asset_file.txt':
 255         '/path/to/my_asset_file.txt'}`.
 256       as_text: whether to write the SavedModel proto in text format. Defaults to
 257         `False`.
 258       exports_to_keep: Number of exports to keep.  Older exports will be
 259         garbage-collected.  Defaults to 5.  Set to `None` to disable garbage
 260         collection.
 261 
 262     Raises:
 263       ValueError: if any arguments is invalid.
 264     """
 265     self._compare_fn = compare_fn
 266     if self._compare_fn is None:
 267       raise ValueError('`compare_fn` must not be None.')
 268     _verify_compare_fn_args(self._compare_fn)
 269 
 270     self._saved_model_exporter = _SavedModelExporter(
 271         name, serving_input_receiver_fn, assets_extra, as_text)
 272 
 273     self._event_file_pattern = event_file_pattern
 274     self._model_dir = None
 275     self._best_eval_result = None
 276 
 277     self._exports_to_keep = exports_to_keep
 278     if exports_to_keep is not None and exports_to_keep <= 0:
 279       raise ValueError(
 280           '`exports_to_keep`, if provided, must be positive number')
 281 
 282   @property
 283   def name(self):
 284     return self._saved_model_exporter.name
 285 
 286   def export(self, estimator, export_path, checkpoint_path, eval_result,
 287              is_the_final_export):
 288     export_result = None
 289 
 290     if self._model_dir != estimator.model_dir and self._event_file_pattern:
 291       # Loads best metric from event files.
 292       tf_logging.info('Loading best metric from event files.')
 293 
 294       self._model_dir = estimator.model_dir
 295       full_event_file_pattern = os.path.join(self._model_dir,
 296                                              self._event_file_pattern)
 297       self._best_eval_result = self._get_best_eval_result(
 298           full_event_file_pattern)
 299 
 300     if self._best_eval_result is None or self._compare_fn(
 301         best_eval_result=self._best_eval_result,
 302         current_eval_result=eval_result):
 303       tf_logging.info('Performing best model export.')
 304       self._best_eval_result = eval_result
 305       export_result = self._saved_model_exporter.export(
 306           estimator, export_path, checkpoint_path, eval_result,
 307           is_the_final_export)
 308       self._garbage_collect_exports(export_path)
 309 
 310     return export_result
 311 
 312   def _garbage_collect_exports(self, export_dir_base):
 313     """Deletes older exports, retaining only a given number of the most recent.
 314 
 315     Export subdirectories are assumed to be named with monotonically increasing
 316     integers; the most recent are taken to be those with the largest values.
 317 
 318     Args:
 319       export_dir_base: the base directory under which each export is in a
 320         versioned subdirectory.
 321     """
 322     if self._exports_to_keep is None:
 323       return
 324 
 325     def _export_version_parser(path):
 326       # create a simple parser that pulls the export_version from the directory.
 327       filename = os.path.basename(path.path)
 328       if not (len(filename) == 10 and filename.isdigit()):
 329         return None
 330       return path._replace(export_version=int(filename))
 331 
 332     # pylint: disable=protected-access
 333     keep_filter = gc._largest_export_versions(self._exports_to_keep)
 334     delete_filter = gc._negation(keep_filter)
 335     for p in delete_filter(
 336         gc._get_paths(export_dir_base, parser=_export_version_parser)):
 337       try:
 338         gfile.DeleteRecursively(p.path)
 339       except errors_impl.NotFoundError as e:
 340         tf_logging.warn('Can not delete %s recursively: %s', p.path, e)
 341     # pylint: enable=protected-access
 342 
 343   def _get_best_eval_result(self, event_files):
 344     """Get the best eval result from event files.
 345 
 346     Args:
 347       event_files: Absolute pattern of event files.
 348 
 349     Returns:
 350       The best eval result.
 351     """
 352     if not event_files:
 353       return None
 354 
 355     best_eval_result = None
 356     for event_file in gfile.Glob(os.path.join(event_files)):
 357       for event in summary_iterator.summary_iterator(event_file):
 358         if event.HasField('summary'):
 359           event_eval_result = {}
 360           for value in event.summary.value:
 361             if value.HasField('simple_value'):
 362               event_eval_result[value.tag] = value.simple_value
 363           if event_eval_result:
 364             if best_eval_result is None or self._compare_fn(
 365                 best_eval_result, event_eval_result):
 366               best_eval_result = event_eval_result
 367     return best_eval_result
