 136 class InstallHeaders(Command):
 137   """Override how headers are copied.
 138 
 139   The install_headers that comes with setuptools copies all files to
 140   the same directory. But we need the files to be in a specific directory
 141   hierarchy for -I <include_dir> to work correctly.
 142   """
 143   description = 'install C/C++ header files'
 144 
 145   user_options = [('install-dir=', 'd',
 146                    'directory to install header files to'),
 147                   ('force', 'f',
 148                    'force installation (overwrite existing files)'),
 149                  ]
 150 
 151   boolean_options = ['force']
 152 
 153   def initialize_options(self):
 154     self.install_dir = None
 155     self.force = 0
 156     self.outfiles = []
 157 
 158   def finalize_options(self):
 159     self.set_undefined_options('install',
 160                                ('install_headers', 'install_dir'),
 161                                ('force', 'force'))
 162 
 163   def mkdir_and_copy_file(self, header):
 164     install_dir = os.path.join(self.install_dir, os.path.dirname(header))
 165     # Get rid of some extra intervening directories so we can have fewer
 166     # directories for -I
 167     install_dir = re.sub('/google/protobuf_archive/src', '', install_dir)
 168 
 169     # Copy external code headers into tensorflow/include.
 170     # A symlink would do, but the wheel file that gets created ignores
 171     # symlink within the directory hierarchy.
 172     # NOTE(keveman): Figure out how to customize bdist_wheel package so
 173     # we can do the symlink.
 174     external_header_locations = [
 175         'tensorflow/include/external/eigen_archive/',
 176         'tensorflow/include/external/com_google_absl/',
 177     ]
 178     for location in external_header_locations:
 179       if location in install_dir:
 180         extra_dir = install_dir.replace(location, '')
 181         if not os.path.exists(extra_dir):
 182           self.mkpath(extra_dir)
 183         self.copy_file(header, extra_dir)
 184 
 185     if not os.path.exists(install_dir):
 186       self.mkpath(install_dir)
 187     return self.copy_file(header, install_dir)
 188 
 189   def run(self):
 190     hdrs = self.distribution.headers
 191     if not hdrs:
 192       return
 193 
 194     self.mkpath(self.install_dir)
 195     for header in hdrs:
 196       (out, _) = self.mkdir_and_copy_file(header)
 197       self.outfiles.append(out)
 198 
 199   def get_inputs(self):
 200     return self.distribution.headers or []
 201 
 202   def get_outputs(self):
 203     return self.outfiles

 206 def find_files(pattern, root):
 207   """Return all the files matching pattern below root dir."""
 208   for dirpath, _, files in os.walk(root):
 209     for filename in fnmatch.filter(files, pattern):
 210       yield os.path.join(dirpath, filename)
