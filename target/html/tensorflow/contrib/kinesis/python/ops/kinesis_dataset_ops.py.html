  28 class KinesisDataset(dataset_ops.DatasetSource):
  29   """A Kinesis Dataset that consumes the message.
  30 
  31   Kinesis is a managed service provided by AWS for data streaming.
  32   This dataset reads messages from Kinesis with each message presented
  33   as a `tf.string`.
  34 
  35   For example, we can construct and use the KinesisDataset as follows:
  36   ```python
  37   dataset = tf.contrib.kinesis.KinesisDataset(
  38       "kinesis_stream_name", read_indefinitely=False)
  39   next = dataset.make_one_shot_iterator().get_next()
  40   with tf.Session() as sess:
  41     while True:
  42       try:
  43         print(sess.run(nxt))
  44       except tf.errors.OutOfRangeError:
  45         break
  46   ```
  47 
  48   Since Kinesis is a data streaming service, data may not be available
  49   at the time it is being read. The argument `read_indefinitely` is
  50   used to control the behavior in this situation. If `read_indefinitely`
  51   is `True`, then `KinesisDataset` will keep retrying to retrieve data
  52   from the stream. If `read_indefinitely` is `False`, an `OutOfRangeError`
  53   is returned immediately instead.
  54   """
  55 
  56   def __init__(self,
  57                stream,
  58                shard="",
  59                read_indefinitely=True,
  60                interval=100000):
  61     """Create a KinesisDataset.
  62 
  63     Args:
  64       stream: A `tf.string` tensor containing the name of the stream.
  65       shard: A `tf.string` tensor containing the id of the shard.
  66       read_indefinitely: If `True`, the Kinesis dataset will keep retry
  67         again on `EOF` after the `interval` period. If `False`, then
  68         the dataset will stop on `EOF`. The default value is `True`.
  69       interval: The interval for the Kinesis Client to wait before
  70         it tries to get records again (in millisecond).
  71     """
  72     super(KinesisDataset, self).__init__()
  73     self._stream = ops.convert_to_tensor(
  74         stream, dtype=dtypes.string, name="stream")
  75     self._shard = ops.convert_to_tensor(
  76         shard, dtype=dtypes.string, name="shard")
  77     self._read_indefinitely = ops.convert_to_tensor(
  78         read_indefinitely, dtype=dtypes.bool, name="read_indefinitely")
  79     self._interval = ops.convert_to_tensor(
  80         interval, dtype=dtypes.int64, name="interval")
  81 
  82   def _as_variant_tensor(self):
  83     return gen_dataset_ops.kinesis_dataset(
  84         self._stream, self._shard, self._read_indefinitely, self._interval)
  85 
  86   @property
  87   def output_classes(self):
  88     return ops.Tensor
  89 
  90   @property
  91   def output_shapes(self):
  92     return tensor_shape.scalar()
  93 
  94   @property
  95   def output_types(self):
  96     return dtypes.string
