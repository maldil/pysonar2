  15 """Wrappers for bucketization operations."""
  16 from __future__ import absolute_import
  17 from __future__ import division
  18 from __future__ import print_function
  19 
  20 from tensorflow.python.ops import math_ops
  21 
  22 
  23 def bucketize(input_tensor, boundaries, name=None):
  24   """Bucketizes input_tensor by given boundaries.
  25 
  26   See bucketize_op.cc for more details.
  27 
  28   Args:
  29     input_tensor: A `Tensor` which will be bucketize.
  30     boundaries: A list of floats gives the boundaries. It has to be sorted.
  31     name: A name prefix for the returned tensors (optional).
  32 
  33   Returns:
  34     A `Tensor` with type int32 which indicates the corresponding bucket for
  35       each value in `input_tensor`.
  36 
  37   Raises:
  38     TypeError: If boundaries is not a list.
  39   """
  40   return math_ops._bucketize(  # pylint: disable=protected-access
  41       input_tensor, boundaries=boundaries, name=name)
