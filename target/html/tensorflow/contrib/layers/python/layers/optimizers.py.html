  56 def optimize_loss(loss,
  57                   global_step,
  58                   learning_rate,
  59                   optimizer,
  60                   gradient_noise_scale=None,
  61                   gradient_multipliers=None,
  62                   clip_gradients=None,
  63                   learning_rate_decay_fn=None,
  64                   update_ops=None,
  65                   variables=None,
  66                   name=None,
  67                   summaries=None,
  68                   colocate_gradients_with_ops=False,
  69                   increment_global_step=True):
  70   """Given loss and parameters for optimizer, returns a training op.
  71 
  72   Various ways of passing optimizers include:
  73 
  74   - by string specifying the name of the optimizer. See OPTIMIZER_CLS_NAMES
  75       for full list. E.g. `optimize_loss(..., optimizer='Adam')`.
  76   - by function taking learning rate `Tensor` as argument and returning an
  77       `Optimizer` instance. E.g. `optimize_loss(...,
  78       optimizer=lambda lr: tf.train.MomentumOptimizer(lr, momentum=0.5))`.
  79     Alternatively, if `learning_rate` is `None`, the function takes no
  80     arguments. E.g. `optimize_loss(..., learning_rate=None,
  81       optimizer=lambda: tf.train.MomentumOptimizer(0.5, momentum=0.5))`.
  82   - by a subclass of `Optimizer` having a single-argument constructor
  83       (the argument is the learning rate), such as AdamOptimizer or
  84       AdagradOptimizer. E.g. `optimize_loss(...,
  85       optimizer=tf.train.AdagradOptimizer)`.
  86   - by an instance of a subclass of `Optimizer`.
  87       E.g., `optimize_loss(..., optimizer=tf.train.AdagradOptimizer(0.5))`.
  88 
  89   Args:
  90     loss: Scalar `Tensor`.
  91     global_step: Scalar int `Tensor`, step counter to update on each step
  92                  unless `increment_global_step` is `False`. If not supplied,
  93                  it will be fetched from the default graph (see
  94                  `tf.train.get_global_step` for details). If it has
  95                  not been created, no step will be incremented with each weight
  96                  update. `learning_rate_decay_fn` requires `global_step`.
  97     learning_rate: float or `Tensor`, magnitude of update per each training
  98                    step. Can be `None`.
  99     optimizer: string, class or optimizer instance, used as trainer.
 100                string should be name of optimizer, like 'SGD',
 101                  'Adam', 'Adagrad'. Full list in OPTIMIZER_CLS_NAMES constant.
 102                class should be sub-class of `tf.Optimizer` that implements
 103                  `compute_gradients` and `apply_gradients` functions.
 104                optimizer instance should be instantiation of `tf.Optimizer`
 105                  sub-class and have `compute_gradients` and `apply_gradients`
 106                  functions.
 107     gradient_noise_scale: float or None, adds 0-mean normal noise scaled by this
 108                           value.
 109     gradient_multipliers: dict of variables or variable names to floats.
 110                           If present, gradients for specified
 111                           variables will be multiplied by given constant.
 112     clip_gradients: float, callable or `None`. If float, is provided, a global
 113       clipping is applied to prevent the norm of the gradient to exceed this
 114       value. Alternatively, a callable can be provided e.g.: adaptive_clipping.
 115       This callable takes a `list` of `(gradients, variables)` `tuple`s and
 116       returns the same thing with the gradients modified.
 117     learning_rate_decay_fn: function, takes `learning_rate` and `global_step`
 118                             `Tensor`s, returns `Tensor`.
 119                             Can be used to implement any learning rate decay
 120                             functions.
 121                             For example: `tf.train.exponential_decay`.
 122                             Ignored if `learning_rate` is not supplied.
 123     update_ops: list of update `Operation`s to execute at each step. If `None`,
 124                 uses elements of UPDATE_OPS collection. The order of execution
 125                 between `update_ops` and `loss` is non-deterministic.
 126     variables: list of variables to optimize or
 127                `None` to use all trainable variables.
 128     name: The name for this operation is used to scope operations and summaries.
 129     summaries: List of internal quantities to visualize on tensorboard. If not
 130                set, the loss, the learning rate, and the global norm of the
 131                gradients will be reported. The complete list of possible values
 132                is in OPTIMIZER_SUMMARIES.
 133     colocate_gradients_with_ops: If True, try colocating gradients with the
 134                                  corresponding op.
 135     increment_global_step: Whether to increment `global_step`. If your model
 136       calls `optimize_loss` multiple times per training step (e.g. to optimize
 137       different parts of the model), use this arg to avoid incrementing
 138       `global_step` more times than necessary.
 139 
 140   Returns:
 141     Training op.
 142 
 143   Raises:
 144     ValueError: if:
 145         * `loss` is an invalid type or shape.
 146         * `global_step` is an invalid type or shape.
 147         * `learning_rate` is an invalid type or value.
 148         * `optimizer` has the wrong type.
 149         * `clip_gradients` is neither float nor callable.
 150         * `learning_rate` and `learning_rate_decay_fn` are supplied, but no
 151           `global_step` is available.
 152         * `gradients` is empty.
 153   """
 154   loss = ops.convert_to_tensor(loss)
 155   contrib_framework.assert_scalar(loss)
 156   if global_step is None:
 157     global_step = train.get_global_step()
 158   else:
 159     train.assert_global_step(global_step)
 160   with vs.variable_scope(name, "OptimizeLoss", [loss, global_step]):
 161     # Update ops take UPDATE_OPS collection if not provided.
 162     if update_ops is None:
 163       update_ops = set(ops.get_collection(ops.GraphKeys.UPDATE_OPS))
 164     # Make sure update ops are ran before computing loss.
 165     if update_ops:
 166       loss = control_flow_ops.with_dependencies(list(update_ops), loss)
 167 
 168     # Learning rate variable, with possible decay.
 169     lr = None
 170     if learning_rate is not None:
 171       if (isinstance(learning_rate, ops.Tensor) and
 172           learning_rate.get_shape().ndims == 0):
 173         lr = learning_rate
 174       elif isinstance(learning_rate, float):
 175         if learning_rate < 0.0:
 176           raise ValueError("Invalid learning_rate %s.", learning_rate)
 177         lr = vs.get_variable(
 178             "learning_rate", [],
 179             trainable=False,
 180             initializer=init_ops.constant_initializer(learning_rate))
 181       else:
 182         raise ValueError("Learning rate should be 0d Tensor or float. "
 183                          "Got %s of type %s" % (str(learning_rate),
 184                                                 str(type(learning_rate))))
 185     if summaries is None:
 186       summaries = ["loss", "learning_rate", "global_gradient_norm"]
 187     else:
 188       for summ in summaries:
 189         if summ not in OPTIMIZER_SUMMARIES:
 190           raise ValueError("Summaries should be one of [%s], you provided %s." %
 191                            (", ".join(OPTIMIZER_SUMMARIES), summ))
 192     if learning_rate is not None and learning_rate_decay_fn is not None:
 193       if global_step is None:
 194         raise ValueError("global_step is required for learning_rate_decay_fn.")
 195       lr = learning_rate_decay_fn(lr, global_step)
 196       if "learning_rate" in summaries:
 197         summary.scalar("learning_rate", lr)
 198 
 199     # Create optimizer, given specified parameters.
 200     if isinstance(optimizer, six.string_types):
 201       if lr is None:
 202         raise ValueError("Learning rate is None, but should be specified if "
 203                          "optimizer is string (%s)." % optimizer)
 204       if optimizer not in OPTIMIZER_CLS_NAMES:
 205         raise ValueError(
 206             "Optimizer name should be one of [%s], you provided %s." %
 207             (", ".join(OPTIMIZER_CLS_NAMES), optimizer))
 208       opt = OPTIMIZER_CLS_NAMES[optimizer](learning_rate=lr)
 209     elif (isinstance(optimizer, type) and
 210           issubclass(optimizer, optimizer_.Optimizer)):
 211       if lr is None:
 212         raise ValueError("Learning rate is None, but should be specified if "
 213                          "optimizer is class (%s)." % optimizer)
 214       opt = optimizer(learning_rate=lr)
 215     elif isinstance(optimizer, optimizer_.Optimizer):
 216       opt = optimizer
 217     elif callable(optimizer):
 218       if learning_rate is not None:
 219         opt = optimizer(lr)
 220       else:
 221         opt = optimizer()
 222       if not isinstance(opt, optimizer_.Optimizer):
 223         raise ValueError("Unrecognized optimizer: function should return "
 224                          "subclass of Optimizer. Got %s." % str(opt))
 225     else:
 226       raise ValueError("Unrecognized optimizer: should be string, "
 227                        "subclass of Optimizer, instance of "
 228                        "subclass of Optimizer or function with one argument. "
 229                        "Got %s." % str(optimizer))
 230 
 231     # All trainable variables, if specific variables are not specified.
 232     if variables is None:
 233       variables = vars_.trainable_variables()
 234 
 235     # Compute gradients.
 236     gradients = opt.compute_gradients(
 237         loss,
 238         variables,
 239         colocate_gradients_with_ops=colocate_gradients_with_ops)
 240 
 241     # Optionally add gradient noise.
 242     if gradient_noise_scale is not None:
 243       gradients = _add_scaled_noise_to_gradients(gradients,
 244                                                  gradient_noise_scale)
 245 
 246     # Multiply some gradients.
 247     if gradient_multipliers is not None:
 248       gradients = _multiply_gradients(gradients, gradient_multipliers)
 249       if not gradients:
 250         raise ValueError(
 251             "Empty list of (gradient, var) pairs encountered. This is most "
 252             "likely to be caused by an improper value of gradient_multipliers.")
 253 
 254     if "global_gradient_norm" in summaries or "gradient_norm" in summaries:
 255       summary.scalar("global_norm/gradient_norm",
 256                      clip_ops.global_norm(list(zip(*gradients))[0]))
 257 
 258     # Optionally clip gradients by global norm.
 259     if isinstance(clip_gradients, float):
 260       gradients = _clip_gradients_by_norm(gradients, clip_gradients)
 261     elif callable(clip_gradients):
 262       gradients = clip_gradients(gradients)
 263     elif clip_gradients is not None:
 264       raise ValueError(
 265           "Unknown type %s for clip_gradients" % type(clip_gradients))
 266 
 267     # Add scalar summary for loss.
 268     if "loss" in summaries:
 269       summary.scalar("loss", loss)
 270 
 271     # Add histograms for variables, gradients and gradient norms.
 272     for gradient, variable in gradients:
 273       if isinstance(gradient, ops.IndexedSlices):
 274         grad_values = gradient.values
 275       else:
 276         grad_values = gradient
 277 
 278       if grad_values is not None:
 279         var_name = variable.name.replace(":", "_")
 280         if "gradients" in summaries:
 281           summary.histogram("gradients/%s" % var_name, grad_values)
 282         if "gradient_norm" in summaries:
 283           summary.scalar("gradient_norm/%s" % var_name,
 284                          clip_ops.global_norm([grad_values]))
 285 
 286     if clip_gradients is not None and ("global_gradient_norm" in summaries or
 287                                        "gradient_norm" in summaries):
 288       summary.scalar("global_norm/clipped_gradient_norm",
 289                      clip_ops.global_norm(list(zip(*gradients))[0]))
 290 
 291     # Create gradient updates.
 292     grad_updates = opt.apply_gradients(
 293         gradients,
 294         global_step=global_step if increment_global_step else None,
 295         name="train")
 296 
 297     # Ensure the train_tensor computes grad_updates.
 298     train_tensor = control_flow_ops.with_dependencies([grad_updates], loss)
 299 
 300     return train_tensor

 303 def _clip_gradients_by_norm(grads_and_vars, clip_gradients):
 304   """Clips gradients by global norm."""
 305   gradients, variables = zip(*grads_and_vars)
 306   clipped_gradients, _ = clip_ops.clip_by_global_norm(gradients, clip_gradients)
 307   return list(zip(clipped_gradients, variables))

 310 def _adaptive_max_norm(norm, std_factor, decay, global_step, epsilon, name):
 311   """Find max_norm given norm and previous average."""
 312   with vs.variable_scope(name, "AdaptiveMaxNorm", [norm]):
 313     log_norm = math_ops.log(norm + epsilon)
 314 
 315     def moving_average(name, value, decay):
 316       moving_average_variable = vs.get_variable(
 317           name,
 318           shape=value.get_shape(),
 319           dtype=value.dtype,
 320           initializer=init_ops.zeros_initializer(),
 321           trainable=False)
 322       return moving_averages.assign_moving_average(
 323           moving_average_variable, value, decay, zero_debias=False)
 324 
 325     # quicker adaptation at the beginning
 326     if global_step is not None:
 327       n = math_ops.to_float(global_step)
 328       decay = math_ops.minimum(decay, n / (n + 1.))
 329 
 330     # update averages
 331     mean = moving_average("mean", log_norm, decay)
 332     sq_mean = moving_average("sq_mean", math_ops.square(log_norm), decay)
 333 
 334     variance = sq_mean - math_ops.square(mean)
 335     std = math_ops.sqrt(math_ops.maximum(epsilon, variance))
 336     max_norms = math_ops.exp(mean + std_factor * std)
 337     return max_norms, mean

 340 def adaptive_clipping_fn(std_factor=2.,
 341                          decay=0.95,
 342                          static_max_norm=None,
 343                          global_step=None,
 344                          report_summary=False,
 345                          epsilon=1e-8,
 346                          name=None):
 347   """Adapt the clipping value using statistics on the norms.
 348 
 349   Implement adaptive gradient as presented in section 3.2.1 of
 350   https://arxiv.org/abs/1412.1602.
 351 
 352   Keeps a moving average of the mean and std of the log(norm) of the gradient.
 353   If the norm exceeds `exp(mean + std_factor*std)` then all gradients will be
 354   rescaled such that the global norm becomes `exp(mean)`.
 355 
 356   Args:
 357     std_factor: Python scaler (or tensor).
 358       `max_norm = exp(mean + std_factor*std)`
 359     decay: The smoothing factor of the moving averages.
 360     static_max_norm: If provided, will threshold the norm to this value as an
 361       extra safety.
 362     global_step: Optional global_step. If provided, `decay = decay*n/(n+1)`.
 363       This provides a quicker adaptation of the mean for the first steps.
 364     report_summary: If `True`, will add histogram summaries of the `max_norm`.
 365     epsilon: Small value chosen to avoid zero variance.
 366     name: The name for this operation is used to scope operations and summaries.
 367 
 368   Returns:
 369     A function for applying gradient clipping.
 370   """
 371 
 372   def gradient_clipping(grads_and_vars):
 373     """Internal function for adaptive clipping."""
 374     grads, variables = zip(*grads_and_vars)
 375 
 376     norm = clip_ops.global_norm(grads)
 377 
 378     max_norm, log_mean = _adaptive_max_norm(norm, std_factor, decay,
 379                                             global_step, epsilon, name)
 380 
 381     # reports the max gradient norm for debugging
 382     if report_summary:
 383       summary.scalar("global_norm/adaptive_max_gradient_norm", max_norm)
 384 
 385     # factor will be 1. if norm is smaller than max_norm
 386     factor = array_ops.where(norm < max_norm,
 387                              array_ops.ones_like(norm),
 388                              math_ops.exp(log_mean) / norm)
 389 
 390     if static_max_norm is not None:
 391       factor = math_ops.minimum(static_max_norm / norm, factor)
 392 
 393     # apply factor
 394     clipped_grads = []
 395     for grad in grads:
 396       if grad is None:
 397         clipped_grads.append(None)
 398       elif isinstance(grad, ops.IndexedSlices):
 399         clipped_grads.append(
 400             ops.IndexedSlices(grad.values * factor, grad.indices,
 401                               grad.dense_shape))
 402       else:
 403         clipped_grads.append(grad * factor)
 404 
 405     return list(zip(clipped_grads, variables))
 406 
 407   return gradient_clipping

 410 def _add_scaled_noise_to_gradients(grads_and_vars, gradient_noise_scale):
 411   """Adds scaled noise from a 0-mean normal distribution to gradients."""
 412   gradients, variables = zip(*grads_and_vars)
 413   noisy_gradients = []
 414   for gradient in gradients:
 415     if gradient is None:
 416       noisy_gradients.append(None)
 417       continue
 418     if isinstance(gradient, ops.IndexedSlices):
 419       gradient_shape = gradient.dense_shape
 420     else:
 421       gradient_shape = gradient.get_shape()
 422     noise = random_ops.truncated_normal(gradient_shape) * gradient_noise_scale
 423     noisy_gradients.append(gradient + noise)
 424   return list(zip(noisy_gradients, variables))

 427 def _multiply_gradients(grads_and_vars, gradient_multipliers):
 428   """Multiply specified gradients."""
 429   multiplied_grads_and_vars = []
 430   for grad, var in grads_and_vars:
 431     if (grad is not None and
 432         (var in gradient_multipliers or var.name in gradient_multipliers)):
 433       key = var if var in gradient_multipliers else var.name
 434       multiplier = gradient_multipliers[key]
 435       if isinstance(grad, ops.IndexedSlices):
 436         grad_values = grad.values * multiplier
 437         grad = ops.IndexedSlices(grad_values, grad.indices, grad.dense_shape)
 438       else:
 439         grad *= math_ops.cast(multiplier, grad.dtype)
 440     multiplied_grads_and_vars.append((grad, var))
 441   return multiplied_grads_and_vars
