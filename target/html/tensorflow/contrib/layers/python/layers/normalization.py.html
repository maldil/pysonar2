  41 @add_arg_scope
  42 def instance_norm(inputs,
  43                   center=True,
  44                   scale=True,
  45                   epsilon=1e-6,
  46                   activation_fn=None,
  47                   param_initializers=None,
  48                   reuse=None,
  49                   variables_collections=None,
  50                   outputs_collections=None,
  51                   trainable=True,
  52                   data_format=DATA_FORMAT_NHWC,
  53                   scope=None):
  54   """Functional interface for the instance normalization layer.
  55 
  56   Reference: https://arxiv.org/abs/1607.08022.
  57 
  58     "Instance Normalization: The Missing Ingredient for Fast Stylization"
  59     Dmitry Ulyanov, Andrea Vedaldi, Victor Lempitsky
  60 
  61   Args:
  62     inputs: A tensor with 2 or more dimensions, where the first dimension has
  63       `batch_size`. The normalization is over all but the last dimension if
  64       `data_format` is `NHWC` and the second dimension if `data_format` is
  65       `NCHW`.
  66     center: If True, add offset of `beta` to normalized tensor. If False, `beta`
  67       is ignored.
  68     scale: If True, multiply by `gamma`. If False, `gamma` is
  69       not used. When the next layer is linear (also e.g. `nn.relu`), this can be
  70       disabled since the scaling can be done by the next layer.
  71     epsilon: Small float added to variance to avoid dividing by zero.
  72     activation_fn: Activation function, default set to None to skip it and
  73       maintain a linear activation.
  74     param_initializers: Optional initializers for beta, gamma, moving mean and
  75       moving variance.
  76     reuse: Whether or not the layer and its variables should be reused. To be
  77       able to reuse the layer scope must be given.
  78     variables_collections: Optional collections for the variables.
  79     outputs_collections: Collections to add the outputs.
  80     trainable: If `True` also add variables to the graph collection
  81       `GraphKeys.TRAINABLE_VARIABLES` (see `tf.Variable`).
  82     data_format: A string. `NHWC` (default) and `NCHW` are supported.
  83     scope: Optional scope for `variable_scope`.
  84 
  85   Returns:
  86     A `Tensor` representing the output of the operation.
  87 
  88   Raises:
  89     ValueError: If `data_format` is neither `NHWC` nor `NCHW`.
  90     ValueError: If the rank of `inputs` is undefined.
  91     ValueError: If rank or channels dimension of `inputs` is undefined.
  92   """
  93   inputs = ops.convert_to_tensor(inputs)
  94   inputs_shape = inputs.shape
  95   inputs_rank = inputs.shape.ndims
  96 
  97   if inputs_rank is None:
  98     raise ValueError('Inputs %s has undefined rank.' % inputs.name)
  99   if data_format not in (DATA_FORMAT_NCHW, DATA_FORMAT_NHWC):
 100     raise ValueError('data_format has to be either NCHW or NHWC.')
 101 
 102   with variable_scope.variable_scope(
 103       scope, 'InstanceNorm', [inputs], reuse=reuse) as sc:
 104     if data_format == DATA_FORMAT_NCHW:
 105       reduction_axis = 1
 106       # For NCHW format, rather than relying on implicit broadcasting, we
 107       # explicitly reshape the params to params_shape_broadcast when computing
 108       # the moments and the batch normalization.
 109       params_shape_broadcast = list(
 110           [1, inputs_shape[1].value] + [1 for _ in range(2, inputs_rank)])
 111     else:
 112       reduction_axis = inputs_rank - 1
 113       params_shape_broadcast = None
 114     moments_axes = list(range(inputs_rank))
 115     del moments_axes[reduction_axis]
 116     del moments_axes[0]
 117     params_shape = inputs_shape[reduction_axis:reduction_axis + 1]
 118     if not params_shape.is_fully_defined():
 119       raise ValueError('Inputs %s has undefined channels dimension %s.' % (
 120           inputs.name, params_shape))
 121 
 122     # Allocate parameters for the beta and gamma of the normalization.
 123     beta, gamma = None, None
 124     dtype = inputs.dtype.base_dtype
 125     if param_initializers is None:
 126       param_initializers = {}
 127     if center:
 128       beta_collections = utils.get_variable_collections(
 129           variables_collections, 'beta')
 130       beta_initializer = param_initializers.get(
 131           'beta', init_ops.zeros_initializer())
 132       beta = variables.model_variable('beta',
 133                                       shape=params_shape,
 134                                       dtype=dtype,
 135                                       initializer=beta_initializer,
 136                                       collections=beta_collections,
 137                                       trainable=trainable)
 138       if params_shape_broadcast:
 139         beta = array_ops.reshape(beta, params_shape_broadcast)
 140     if scale:
 141       gamma_collections = utils.get_variable_collections(
 142           variables_collections, 'gamma')
 143       gamma_initializer = param_initializers.get(
 144           'gamma', init_ops.ones_initializer())
 145       gamma = variables.model_variable('gamma',
 146                                        shape=params_shape,
 147                                        dtype=dtype,
 148                                        initializer=gamma_initializer,
 149                                        collections=gamma_collections,
 150                                        trainable=trainable)
 151       if params_shape_broadcast:
 152         gamma = array_ops.reshape(gamma, params_shape_broadcast)
 153 
 154     # Calculate the moments (instance activations).
 155     mean, variance = nn.moments(inputs, moments_axes, keep_dims=True)
 156 
 157     # Compute instance normalization.
 158     outputs = nn.batch_normalization(
 159         inputs, mean, variance, beta, gamma, epsilon, name='instancenorm')
 160     if activation_fn is not None:
 161       outputs = activation_fn(outputs)
 162     return utils.collect_named_outputs(outputs_collections, sc.name, outputs)

 165 @add_arg_scope
 166 def group_norm(inputs,
 167                groups=32,
 168                channels_axis=-1,
 169                reduction_axes=(-3, -2),
 170                center=True,
 171                scale=True,
 172                epsilon=1e-6,
 173                activation_fn=None,
 174                param_initializers=None,
 175                reuse=None,
 176                variables_collections=None,
 177                outputs_collections=None,
 178                trainable=True,
 179                scope=None,
 180                mean_close_to_zero=False):
 181   """Functional interface for the group normalization layer.
 182 
 183   Reference: https://arxiv.org/abs/1803.08494.
 184 
 185     "Group Normalization", Yuxin Wu, Kaiming He
 186 
 187   Args:
 188     inputs: A Tensor with at least 2 dimensions one which is channels. All
 189      shape dimensions must be fully defined.
 190     groups: Integer. Divide the channels into this number of groups over which
 191       normalization statistics are computed. This number must be commensurate
 192       with the number of channels in `inputs`.
 193     channels_axis: An integer. Specifies index of channels axis which will be
 194       broken into `groups`, each of which whose statistics will be computed
 195       across. Must be mutually exclusive with `reduction_axes`. Preferred usage
 196       is to specify negative integers to be agnostic as to whether a batch
 197       dimension is included.
 198     reduction_axes: Tuple of integers. Specifies dimensions over which
 199        statistics will be accumulated. Must be mutually exclusive with
 200        `channels_axis`. Statistics will not be accumulated across axes not
 201        specified in `reduction_axes` nor `channel_axis`. Preferred usage is to
 202        specify negative integers to be agnostic to whether a batch dimension is
 203        included.
 204 
 205       Some sample usage cases:
 206         NHWC format: channels_axis=-1, reduction_axes=[-3, -2]
 207         NCHW format: channels_axis=-3, reduction_axes=[-2, -1]
 208 
 209     center: If True, add offset of `beta` to normalized tensor. If False, `beta`
 210       is ignored.
 211     scale: If True, multiply by `gamma`. If False, `gamma` is
 212       not used. When the next layer is linear (also e.g. `nn.relu`), this can be
 213       disabled since the scaling can be done by the next layer.
 214     epsilon: Small float added to variance to avoid dividing by zero.
 215     activation_fn: Activation function, default set to None to skip it and
 216       maintain a linear activation.
 217     param_initializers: Optional initializers for beta, gamma, moving mean and
 218       moving variance.
 219     reuse: Whether or not the layer and its variables should be reused. To be
 220       able to reuse the layer scope must be given.
 221     variables_collections: Optional collections for the variables.
 222     outputs_collections: Collections to add the outputs.
 223     trainable: If `True` also add variables to the graph collection
 224       `GraphKeys.TRAINABLE_VARIABLES` (see `tf.Variable`).
 225     scope: Optional scope for `variable_scope`.
 226     mean_close_to_zero: The mean of `input` before ReLU will be close to zero
 227       when batch size >= 4k for Resnet-50 on TPU. If `True`, use
 228       `nn.sufficient_statistics` and `nn.normalize_moments` to calculate the
 229       variance. This is the same behavior as `fused` equals `True` in batch
 230       normalization. If `False`, use `nn.moments` to calculate the variance.
 231       When `mean` is close to zero, like 1e-4, use `mean` to calculate the
 232       variance may have poor result due to repeated roundoff error and
 233       denormalization in `mean`.  When `mean` is large, like 1e2,
 234       sum(`input`^2) is so large that only the high-order digits of the elements
 235       are being accumulated. Thus, use sum(`input` - `mean`)^2/n to calculate
 236       the variance has better accuracy compared to (sum(`input`^2)/n - `mean`^2)
 237       when `mean` is large.
 238 
 239 
 240   Returns:
 241     A `Tensor` representing the output of the operation.
 242 
 243   Raises:
 244     ValueError: If the rank of `inputs` is undefined.
 245     ValueError: If rank or channels dimension of `inputs` is undefined.
 246     ValueError: If number of groups is not commensurate with number of channels.
 247     ValueError: If reduction_axes or channels_axis are out of bounds.
 248     ValueError: If reduction_axes are not mutually exclusive with channels_axis.
 249   """
 250   # TODO(shlens): Support partially defined shapes for the inputs.
 251   inputs = ops.convert_to_tensor(inputs)
 252   original_shape = inputs.shape
 253 
 254   if inputs.shape.ndims is None:
 255     raise ValueError('Inputs %s has undefined rank.' % inputs.name)
 256   if channels_axis > (inputs.shape.ndims - 1):
 257     raise ValueError('Axis is out of bounds.')
 258 
 259   # Standardize the channels_axis to be positive and identify # of channels.
 260   if channels_axis < 0:
 261     channels_axis = inputs.shape.ndims + channels_axis
 262   channels = inputs.shape[channels_axis].value
 263 
 264   if channels is None:
 265     raise ValueError('Inputs %s has undefined channel dimension: %d.' % (
 266         inputs.name, channels_axis))
 267 
 268   # Standardize the reduction_axes to be positive.
 269   reduction_axes = list(reduction_axes)
 270   for i in range(len(reduction_axes)):
 271     if reduction_axes[i] < 0:
 272       reduction_axes[i] += inputs.shape.ndims
 273 
 274   for a in reduction_axes:
 275     if a > inputs.shape.ndims:
 276       raise ValueError('Axis is out of bounds.')
 277     if inputs.shape[a].value is None:
 278       raise ValueError('Inputs %s has undefined dimensions %d.' % (
 279           inputs.name, a))
 280     if channels_axis == a:
 281       raise ValueError('reduction_axis must be mutually exclusive '
 282                        'with channels_axis')
 283   if groups > channels:
 284     raise ValueError('Invalid groups %d for %d channels.' % (groups, channels))
 285   if channels % groups != 0:
 286     raise ValueError('%d channels is not commensurate with %d groups.' %
 287                      (channels, groups))
 288 
 289   # Determine axes before channels. Some examples of common image formats:
 290   #  'NCHW': before = [N], after = [HW]
 291   #  'NHWC': before = [NHW], after = []
 292   axes_before_channels = inputs.shape.as_list()[:channels_axis]
 293   axes_after_channels = inputs.shape.as_list()[channels_axis+1:]
 294 
 295   # Manually broadcast the parameters to conform to the number of groups.
 296   params_shape_broadcast = ([1] * len(axes_before_channels) +
 297                             [groups, channels // groups] +
 298                             [1] * len(axes_after_channels))
 299 
 300   # Reshape the input by the group within the channel dimension.
 301   inputs_shape = (axes_before_channels + [groups, channels // groups] +
 302                   axes_after_channels)
 303   inputs = array_ops.reshape(inputs, inputs_shape)
 304 
 305   # Determine the dimensions across which moments are calculated.
 306   moments_axes = [channels_axis + 1]
 307   for a in reduction_axes:
 308     if a > channels_axis:
 309       moments_axes.append(a + 1)
 310     else:
 311       moments_axes.append(a)
 312 
 313   with variable_scope.variable_scope(
 314       scope, 'GroupNorm', [inputs], reuse=reuse) as sc:
 315     # Note that the params_shape is the number of channels always.
 316     params_shape = [channels]
 317 
 318     # Allocate parameters for the beta and gamma of the normalization.
 319     beta, gamma = None, None
 320     dtype = inputs.dtype.base_dtype
 321     if param_initializers is None:
 322       param_initializers = {}
 323     if center:
 324       beta_collections = utils.get_variable_collections(
 325           variables_collections, 'beta')
 326       beta_initializer = param_initializers.get(
 327           'beta', init_ops.zeros_initializer())
 328       beta = variables.model_variable('beta',
 329                                       shape=params_shape,
 330                                       dtype=dtype,
 331                                       initializer=beta_initializer,
 332                                       collections=beta_collections,
 333                                       trainable=trainable)
 334       beta = array_ops.reshape(beta, params_shape_broadcast)
 335 
 336     if scale:
 337       gamma_collections = utils.get_variable_collections(
 338           variables_collections, 'gamma')
 339       gamma_initializer = param_initializers.get(
 340           'gamma', init_ops.ones_initializer())
 341       gamma = variables.model_variable('gamma',
 342                                        shape=params_shape,
 343                                        dtype=dtype,
 344                                        initializer=gamma_initializer,
 345                                        collections=gamma_collections,
 346                                        trainable=trainable)
 347       gamma = array_ops.reshape(gamma, params_shape_broadcast)
 348 
 349     # Calculate the moments.
 350     if mean_close_to_zero:
 351       # One pass algorithm returns better result when mean is close to zero.
 352       counts, means_ss, variance_ss, _ = nn.sufficient_statistics(
 353           inputs, moments_axes, keep_dims=True)
 354       mean, variance = nn.normalize_moments(
 355           counts, means_ss, variance_ss, shift=None)
 356     else:
 357       mean, variance = nn.moments(inputs, moments_axes, keep_dims=True)
 358 
 359     # Compute normalization.
 360     # TODO(shlens): Fix nn.batch_normalization to handle the 5-D Tensor
 361     # appropriately so that this operation may be faster.
 362     gain = math_ops.rsqrt(variance + epsilon)
 363     offset = -mean * gain
 364     if gamma is not None:
 365       gain *= gamma
 366       offset *= gamma
 367     if beta is not None:
 368       offset += beta
 369     outputs = inputs * gain + offset
 370 
 371     # Collapse the groups into the channel dimension.
 372     outputs = array_ops.reshape(outputs, original_shape)
 373 
 374     if activation_fn is not None:
 375       outputs = activation_fn(outputs)
 376     return utils.collect_named_outputs(outputs_collections, sc.name, outputs)
