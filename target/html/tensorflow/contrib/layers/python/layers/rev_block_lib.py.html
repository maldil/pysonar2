  55 LAYER_RE = re.compile(".*revlayer_([0-9]*)/([fg])/.*")
  56 _USE_DEFAULT = "__rev_block_lib_default"
  57 _WRONG_VARS_ERR = """\
  58 The variables used on recompute were different than the variables originally
  59 used. The function wrapped with @recompute_grad likley creates its own variable
  60 scope with a default name and has been called twice in the same enclosing scope.
  61 To fix, ensure each call to the function happens in its own unique variable
  62 scope.
  63 """

  66 def _acc_grads(*lists_of_grads):
  67   """Accumulates lists of gradients."""
  68   acc_grads = []
  69   for grads in zip(*lists_of_grads):
  70     grads = [g for g in grads if g is not None]
  71     if grads:
  72       acc_grads.append(math_ops.add_n(grads))
  73     else:
  74       acc_grads.append(None)
  75   return acc_grads

  78 def _rev_layer_forward(xs, f, g, f_side_input, g_side_input,
  79                        gate_outputs=False):
  80   """Forward for 1 reversible layer."""
  81   x1, x2 = xs
  82   y1 = x1 + (f(x2, f_side_input) if f_side_input else f(x2))
  83   y2 = x2 + (g(y1, g_side_input) if g_side_input else g(y1))
  84   if gate_outputs:
  85     return control_flow_ops.tuple([y1, y2])
  86   else:
  87     return (y1, y2)
  88 
  89 
  90 def _rev_layer_backward(ys, grad_ys, f, g, f_vars, f_side_input, g_vars,
  91                         g_side_input):
  92   """Backprop for 1 layer."""
  93   y1, y2 = ys
  94   grad_y1, grad_y2 = grad_ys
  95 
  96   # Reconstruct intermediates and inputs (x1, x2)
  97   # stop_gradients required on fn inputs to prevent infinite recursion into this
  98   # grad function on the calls to gradients.
  99   y1_stop = array_ops.stop_gradient(y1)
 100   g_side_input = [array_ops.stop_gradient(t) for t in g_side_input]
 101   gy1 = g(y1_stop, g_side_input) if g_side_input else g(y1_stop)
 102 
 103   x2 = y2 - gy1
 104   x2_stop = array_ops.stop_gradient(x2)
 105   f_side_input = [array_ops.stop_gradient(t) for t in f_side_input]
 106   fx2 = f(x2_stop, f_side_input) if f_side_input else f(x2_stop)
 107 
 108   x1 = y1 - fx2
 109 
 110   # Compute gradients wrt to inputs
 111   # dL/dy2 * dG(y1)/y1
 112   grad_gy1_y2 = gradients_impl.gradients(gy1, y1_stop, grad_y2)[0]
 113   grad_x1 = grad_y1 + grad_gy1_y2
 114   grad_x2 = (
 115       gradients_impl.gradients(fx2, x2_stop, grad_y1)[0] + grad_y2 +
 116       gradients_impl.gradients(fx2, x2_stop, grad_gy1_y2)[0])
 117 
 118   # Compute gradients wrt to vars and side inputs in f and g
 119   grads1 = gradients_impl.gradients(gy1, g_vars + g_side_input, grad_y2)
 120   grad_g_vars, grad_g_side = grads1[:len(g_vars)], grads1[len(g_vars):]
 121   grads2 = gradients_impl.gradients(fx2, f_vars + f_side_input, grad_y1)
 122   grad_f_y1, grad_f_side1 = grads2[:len(f_vars)], grads2[len(f_vars):]
 123   grads3 = gradients_impl.gradients(fx2, f_vars + f_side_input, grad_gy1_y2)
 124   grad_f_y2, grad_f_side2 = grads3[:len(f_vars)], grads3[len(f_vars):]
 125   grad_f_vars = _acc_grads(grad_f_y1, grad_f_y2)
 126 
 127   grad_f_side = _acc_grads(grad_f_side1, grad_f_side2)
 128 
 129   # Put returns in a tuple to ensure a constant memory budget (i.e. don't want
 130   # the subsequent layer to start computing and consuming memory based on a
 131   # subset of these values).
 132   outputs = ((x1, x2), (grad_x1, grad_x2), (grad_f_vars, grad_f_side),
 133              (grad_g_vars, grad_g_side))
 134   tupled = control_flow_ops.tuple(nest.flatten(outputs))
 135   return nest.pack_sequence_as(outputs, tupled)

 138 def _rev_block_forward(x1,
 139                        x2,
 140                        f,
 141                        g,
 142                        num_layers=1,
 143                        f_side_input=None,
 144                        g_side_input=None,
 145                        gate_outputs=False):
 146   """Forward for a series of reversible layers."""
 147   out = (x1, x2)
 148   for i in xrange(num_layers):
 149     out = _rev_layer_forward(
 150         out, f[i], g[i], f_side_input, g_side_input, gate_outputs=gate_outputs)
 151 
 152   y1, y2 = out
 153   return y1, y2

 156 def _safe_wraps(fn):
 157   if isinstance(fn, functools.partial):
 158     # functools.partial objects cannot be wrapped as they are missing the
 159     # necessary properties (__name__, __module__, __doc__).
 160     def passthrough(f):
 161       return f
 162     return passthrough
 163   return functools.wraps(fn)

 176 class RevBlock(base.Layer):
 177   """Block of reversible layers. See rev_block."""
 178 
 179   def __init__(self,
 180                f,
 181                g,
 182                num_layers=1,
 183                f_side_input=None,
 184                g_side_input=None,
 185                use_efficient_backprop=True,
 186                name="revblock",
 187                **kwargs):
 188     super(RevBlock, self).__init__(name=name, **kwargs)
 189 
 190     if isinstance(f, list):
 191       assert len(f) == num_layers
 192     else:
 193       f = [f] * num_layers
 194 
 195     if isinstance(g, list):
 196       assert len(g) == num_layers
 197     else:
 198       g = [g] * num_layers
 199 
 200     f = [_scope_wrap(fn, "revlayer_%d/f" % i) for i, fn in enumerate(f)]
 201     g = [_scope_wrap(fn, "revlayer_%d/g" % i) for i, fn in enumerate(g)]
 202 
 203     self.f = f
 204     self.g = g
 205 
 206     self.num_layers = num_layers
 207     self.f_side_input = f_side_input or []
 208     self.g_side_input = g_side_input or []
 209 
 210     self._use_efficient_backprop = use_efficient_backprop
 211 
 212   def call(self, inputs, forward=True):
 213     vs = variable_scope.get_variable_scope()
 214     vars_before = vs.global_variables()
 215 
 216     if forward:
 217       x1, x2 = inputs
 218       out = self._forward(x1, x2)
 219     else:
 220       y1, y2 = inputs
 221       out = self._backward(y1, y2)
 222 
 223     # Add any created variables to the Layer's variable stores
 224     new_vars = vs.global_variables()[len(vars_before):]
 225     train_vars = vs.trainable_variables()
 226     for new_var in new_vars:
 227       if new_var in train_vars:
 228         self._trainable_weights.append(new_var)
 229       else:
 230         self._non_trainable_weights.append(new_var)
 231 
 232     return out
 233 
 234   def forward(self, x1, x2):
 235     return self.apply([x1, x2])
 236 
 237   def backward(self, y1, y2):
 238     return self.apply([y1, y2], forward=False)
 239 
 240   def build(self, _):
 241     logging.warn("RevBlock constructs its variables on first call, not on "
 242                  "build.")
 243     self.built = True

 245   def _make_efficient_grad_fn(self, inputs_, ys_):
 246     def _efficient_grad_fn(*grad_ys, **kwargs):
 247       """Custom gradient fn for a block of reversible residual layers."""
 248       inputs = inputs_
 249       ys = ys_
 250       variables = kwargs["variables"]
 251       side_inputs = inputs[2:]
 252 
 253       f_side_idxs = [None] * len(self.f_side_input)
 254       g_side_idxs = [None] * len(self.g_side_input)
 255       assert len(side_inputs) == len(self.f_side_input) + len(self.g_side_input)
 256 
 257       for i, t in enumerate(side_inputs):
 258         if t in self.f_side_input:
 259           f_side_idxs[self.f_side_input.index(t)] = i
 260         elif t in self.g_side_input:
 261           g_side_idxs[self.g_side_input.index(t)] = i
 262         else:
 263           assert False
 264 
 265       f_vars = [[] for _ in range(self.num_layers)]
 266       g_vars = [[] for _ in range(self.num_layers)]
 267       f_vars_idxs = [[] for _ in range(self.num_layers)]
 268       g_vars_idxs = [[] for _ in range(self.num_layers)]
 269 
 270       for i, ref in enumerate(variables):
 271         # Use the name to identify the layer number and function (f or g)
 272         regex = LAYER_RE.match(ref.name)
 273         layer_no = int(regex.group(1))
 274         fn_name = regex.group(2)
 275         if fn_name == "f":
 276           f_vars[layer_no].append(ref)
 277           f_vars_idxs[layer_no].append(i)
 278         else:
 279           assert fn_name == "g"
 280           g_vars[layer_no].append(ref)
 281           g_vars_idxs[layer_no].append(i)
 282 
 283       f_var_grads = []
 284       g_var_grads = []
 285       f_side_grads = []
 286       g_side_grads = []
 287 
 288       # Reverse variable containers to go backward
 289       f_vars.reverse()
 290       g_vars.reverse()
 291       f = list(self.f)
 292       g = list(self.g)
 293       f.reverse()
 294       g.reverse()
 295 
 296       with variable_scope.variable_scope(self.scope_name, reuse=True):
 297         for i in xrange(self.num_layers):
 298           ys, grad_ys, f_ret, g_ret = _rev_layer_backward(
 299               ys, grad_ys, f[i], g[i], f_vars[i], self.f_side_input, g_vars[i],
 300               self.g_side_input)
 301 
 302           grad_f_vars, grad_f_side = f_ret
 303           grad_g_vars, grad_g_side = g_ret
 304           f_var_grads.append(grad_f_vars)
 305           g_var_grads.append(grad_g_vars)
 306           f_side_grads.append(grad_f_side)
 307           g_side_grads.append(grad_g_side)
 308 
 309       # Accumulate layer gradients for f_side_input and g_side_input
 310       acc_f_side_grads = _acc_grads(*f_side_grads)
 311       acc_g_side_grads = _acc_grads(*g_side_grads)
 312 
 313       # Use the stored idxs to put gradients in the passed-in order.
 314       side_input_grads = [None] * len(side_inputs)
 315       variable_grads = [None] * len(variables)
 316 
 317       # Variable gradients were collected in reverse layer order. Reverse to
 318       # match idxs.
 319       f_var_grads.reverse()
 320       g_var_grads.reverse()
 321       for idxs, grads in list(zip(f_vars_idxs, f_var_grads)) + list(
 322           zip(g_vars_idxs, g_var_grads)):
 323         for i, grad in zip(idxs, grads):
 324           variable_grads[i] = grad
 325 
 326       for i, grad in zip(f_side_idxs, acc_f_side_grads):
 327         side_input_grads[i] = grad
 328       for i, grad in zip(g_side_idxs, acc_g_side_grads):
 329         side_input_grads[i] = grad
 330 
 331       grad_x1, grad_x2 = grad_ys
 332       return [grad_x1, grad_x2] + side_input_grads, variable_grads
 333     return _efficient_grad_fn
 334 
 335   def _forward(self, x1, x2):
 336     """Run forward through the reversible layers."""
 337 
 338     side_inputs = [self.f_side_input, self.g_side_input]
 339     flat_side_inputs = nest.flatten(side_inputs)
 340 
 341     def _forward_wrap(x1_, x2_, *flat_side_inputs):
 342       f_side, g_side = nest.pack_sequence_as(side_inputs, flat_side_inputs)
 343       return _rev_block_forward(
 344           x1_,
 345           x2_,
 346           self.f,
 347           self.g,
 348           num_layers=self.num_layers,
 349           f_side_input=f_side,
 350           g_side_input=g_side,
 351           gate_outputs=self._use_efficient_backprop)
 352 
 353     @custom_gradient.custom_gradient
 354     def _forward_with_custom_grad(*args):
 355       out = _forward_wrap(*args)  # pylint: disable=no-value-for-parameter
 356       grad_fn = self._make_efficient_grad_fn(args, out)
 357       return out, grad_fn
 358 
 359     if self._use_efficient_backprop:
 360       return _forward_with_custom_grad(x1, x2, *flat_side_inputs)
 361     else:
 362       return _forward_wrap(x1, x2, *flat_side_inputs)
 363 
 364   def _backward(self, y1, y2):
 365     """Run backward through the reversible layers."""
 366 
 367     f = list(self.f)
 368     g = list(self.g)
 369     f.reverse()
 370     g.reverse()
 371 
 372     for i in xrange(self.num_layers):
 373       gy1 = g[i](y1, self.g_side_input) if self.g_side_input else g[i](y1)
 374       x2 = y2 - gy1
 375       fx2 = f[i](x2, self.f_side_input) if self.f_side_input else f[i](x2)
 376       x1 = y1 - fx2
 377 
 378       y1, y2 = x1, x2
 379 
 380     return x1, x2

 383 def rev_block(x1,
 384               x2,
 385               f,
 386               g,
 387               num_layers=1,
 388               f_side_input=None,
 389               g_side_input=None,
 390               is_training=True):
 391   """A block of reversible residual layers.
 392 
 393   A reversible residual layer is defined as:
 394 
 395   ```
 396   y1 = x1 + f(x2, f_side_input)
 397   y2 = x2 + g(y1, g_side_input)
 398   ```
 399 
 400   A reversible residual block, defined here, is a series of reversible residual
 401   layers.
 402 
 403   Limitations:
 404   * f and g must not close over any Tensors; all side inputs to f and g should
 405     be passed in with f_side_input and g_side_input which will be forwarded to
 406     f and g.
 407   * f and g must not change the dimensionality of their inputs in order for the
 408     addition in the equations above to work.
 409 
 410   Args:
 411     x1: a float Tensor.
 412     x2: a float Tensor.
 413     f: a function, (Tensor) -> (Tensor) (or list of such of length num_layers).
 414       Should not change the shape of the Tensor. Can make calls to get_variable.
 415       See f_side_input if there are side inputs.
 416     g: a function, (Tensor) -> (Tensor) (or list of such of length num_layers).
 417       Should not change the shape of the Tensor. Can make calls to get_variable.
 418       See g_side_input if there are side inputs.
 419     num_layers: int, number of reversible residual layers. Each layer will
 420       apply f and g according to the equations above, with new variables in each
 421       layer.
 422     f_side_input: list of Tensors, side input to f. If not None, signature of f
 423       should be (Tensor, list<Tensor>) -> (Tensor).
 424     g_side_input: list of Tensors, side input to g. If not None, signature of g
 425       should be (Tensor, list<Tensor>) -> (Tensor).
 426     is_training: bool, whether to actually use the efficient backprop codepath.
 427 
 428   Returns:
 429     y1, y2: tuple of float Tensors.
 430   """
 431   block = RevBlock(
 432       f=f,
 433       g=g,
 434       num_layers=num_layers,
 435       f_side_input=f_side_input,
 436       g_side_input=g_side_input,
 437       use_efficient_backprop=is_training,
 438       _reuse=variable_scope.get_variable_scope().reuse)
 439   return block.forward(x1, x2)

 442 def enable_with_args(dec):
 443   """A decorator for decorators to enable their usage with or without args."""
 444 
 445   @_safe_wraps(dec)
 446   def new_dec(*args, **kwargs):
 447     if len(args) == 1 and not kwargs and callable(args[0]):
 448       # Used as decorator without args
 449       fn = args[0]
 450       return dec(fn)
 451     else:
 452       return lambda fn: dec(fn, *args, **kwargs)
 453 
 454   return new_dec

 457 @enable_with_args
 458 def recompute_grad(fn, use_data_dep=_USE_DEFAULT, tupleize_grads=False):
 459   """Decorator that recomputes the function on the backwards pass.
 460 
 461   To use this function, you must use `ResourceVariable`s (i.e.
 462   `variable_scope(name, use_resource=True), which are the default in Eager mode
 463   and when running on TPU.
 464 
 465   Warning: Because the function will be called again on the backwards pass, the
 466   user should be careful to not use ops in their function that mutate state or
 467   have randomness (for example, batch normalization or dropout). If the function
 468   does have such operations, it is recommended that the function take the
 469   `is_recomputing` keyword argument which will be `False` on the forward pass
 470   and `True` on the backwards pass so that it can disable state changes when
 471   `is_recomputing=True` (for example, not updating the moving averages in batch
 472   normalization).
 473 
 474   Args:
 475     fn: a function that takes Tensors (all as positional arguments) and returns
 476       a tuple of Tensors. Note that `fn` should not close over any other
 477       Tensors or Variables.
 478     use_data_dep: `bool`, if `True` will use a dummy data dependency to force
 479       the recompute to happen. If `False` will use a control dependency. By
 480       default will be `True` if in an XLA context and `False` otherwise. XLA
 481       ignores control dependencies and so this data dependency is necessary.
 482     tupleize_grads: `bool`, if `True` will use control dependencies to ensure
 483       that all gradients are produced before any are consumed by downstream ops.
 484       If `use_data_dep` is also `True`, will use a data dependency instead of
 485       a control dependency.
 486 
 487   Returns:
 488     A wrapped fn that is identical to fn when called, but its activations will
 489     be discarded and recomputed on the backwards pass (i.e. on a call to
 490     tf.gradients).
 491 
 492   Raises:
 493     ValueError: if `fn` closes over any Tensors or Variables.
 494   """
 495   # Check for closed-over Tensors/Variables
 496   if fn.__code__.co_freevars:
 497     closed_over_vars = dict(zip(fn.__code__.co_freevars,
 498                                 [c.cell_contents for c in fn.__closure__]))
 499     for var_name, value in six.iteritems(closed_over_vars):
 500       if isinstance(value, (framework_ops.Tensor, variables_lib.Variable)):
 501         raise ValueError(
 502             "fn decorated with @recompute_grad closes over Tensor %s "
 503             "(local variable name: %s). The decorated fn must not close over "
 504             "Tensors or Variables because gradients will NOT be computed for "
 505             "them through fn. To ensure correct gradients, make the "
 506             "Tensor an input to fn." % (value.name, var_name))
 507 
 508   @_safe_wraps(fn)
 509   def wrapped(*args):
 510     return _recompute_grad(
 511         fn, args, use_data_dep=use_data_dep, tupleize_grads=tupleize_grads)
 512 
 513   return wrapped

 521 def _recomputing_grad_fn(compute_fn,
 522                          original_args,
 523                          original_vars,
 524                          output_grads,
 525                          grad_fn_variables,
 526                          use_data_dep,
 527                          tupleize_grads,
 528                          arg_scope,
 529                          var_scope,
 530                          has_is_recompute_kwarg):
 531   """Grad fn for recompute_grad."""
 532   variables = grad_fn_variables or []
 533 
 534   # Identity ops around the inputs ensures correct gradient graph-walking.
 535   inputs = [array_ops.identity(x) for x in list(original_args)]
 536 
 537   # Recompute outputs
 538   # Use a control dependency to ensure that the recompute is not eliminated by
 539   # CSE and that it happens on the backwards pass.
 540   ctrl_dep_grads = [g for g in output_grads if g is not None]
 541   with framework_ops.control_dependencies(ctrl_dep_grads):
 542     if use_data_dep:
 543       inputs = _force_data_dependency(output_grads, inputs)
 544     # Re-enter scopes
 545     with contrib_framework_ops.arg_scope(arg_scope):
 546       with variable_scope.variable_scope(var_scope, reuse=True):
 547         # Re-call the function and ensure that the touched variables are the
 548         # same as in the first call.
 549         with backprop.GradientTape() as tape:
 550           fn_kwargs = {}
 551           if has_is_recompute_kwarg:
 552             fn_kwargs["is_recomputing"] = True
 553           outputs = compute_fn(*inputs, **fn_kwargs)
 554         recompute_vars = set(tape.watched_variables())
 555         if original_vars != recompute_vars:
 556           raise ValueError(_WRONG_VARS_ERR)
 557 
 558   if not isinstance(outputs, (list, tuple)):
 559     outputs = [outputs]
 560   outputs = list(outputs)
 561 
 562   # Compute gradients
 563   grads = gradients_impl.gradients(outputs, inputs + variables,
 564                                    output_grads)
 565 
 566   if tupleize_grads:
 567     if use_data_dep:
 568       grads = _tuple_with_data_dep(grads)
 569     else:
 570       grads = control_flow_ops.tuple(grads)
 571 
 572   grad_inputs = grads[:len(inputs)]
 573   grad_vars = grads[len(inputs):]
 574   return grad_inputs, grad_vars

 577 def _recompute_grad(fn, args, use_data_dep=_USE_DEFAULT, tupleize_grads=False):
 578   """See recompute_grad."""
 579   has_is_recompute_kwarg = "is_recomputing" in tf_inspect.getargspec(fn).args
 580   for arg in args:
 581     if not isinstance(arg, framework_ops.Tensor):
 582       raise ValueError("All inputs to function must be Tensors")
 583   use_data_dep_ = use_data_dep
 584   if use_data_dep_ == _USE_DEFAULT:
 585     use_data_dep_ = _is_on_tpu()
 586 
 587   # Use custom_gradient and return a grad_fn that recomputes on the backwards
 588   # pass.
 589   @custom_gradient.custom_gradient
 590   def fn_with_recompute(*args):
 591     """Wrapper for fn."""
 592     # Capture the variable and arg scopes so we can re-enter them when
 593     # recomputing.
 594     vs = variable_scope.get_variable_scope()
 595     arg_scope = contrib_framework_ops.current_arg_scope()
 596     # Track all variables touched in the function.
 597     with backprop.GradientTape() as tape:
 598       fn_kwargs = {}
 599       if has_is_recompute_kwarg:
 600         fn_kwargs["is_recomputing"] = False
 601       outputs = fn(*args, **fn_kwargs)
 602     original_vars = set(tape.watched_variables())
 603 
 604     def _grad_fn(output_grads, variables=None):
 605       # Validate that custom_gradient passes the right variables into grad_fn.
 606       if original_vars:
 607         assert variables, ("Fn created variables but the variables were not "
 608                            "passed to the gradient fn.")
 609         if set(variables) != original_vars:
 610           raise ValueError(_WRONG_VARS_ERR)
 611 
 612       return _recomputing_grad_fn(
 613           compute_fn=fn,
 614           original_args=args,
 615           original_vars=original_vars,
 616           output_grads=output_grads,
 617           grad_fn_variables=variables,
 618           use_data_dep=use_data_dep_,
 619           tupleize_grads=tupleize_grads,
 620           arg_scope=arg_scope,
 621           var_scope=vs,
 622           has_is_recompute_kwarg=has_is_recompute_kwarg)
 623 
 624     # custom_gradient inspects the signature of the function to determine
 625     # whether the user expects variables passed in the grad_fn. If the function
 626     # created variables, the grad_fn should accept the "variables" kwarg.
 627     if original_vars:
 628       def grad_fn(*output_grads, **kwargs):
 629         return _grad_fn(output_grads, kwargs["variables"])
 630     else:
 631       def grad_fn(*output_grads):
 632         return _grad_fn(output_grads)
 633 
 634     return outputs, grad_fn
 635 
 636   return fn_with_recompute(*args)

 639 def _underlying_variable_ref(t):
 640   """Find the underlying variable ref.
 641 
 642   Traverses through Identity, ReadVariableOp, and Enter ops.
 643   Stops when op type has Variable or VarHandle in name.
 644 
 645   Args:
 646     t: a Tensor
 647 
 648   Returns:
 649     a Tensor that is a variable ref, or None on error.
 650   """
 651   while t.op.type in ["Identity", "ReadVariableOp", "Enter"]:
 652     t = t.op.inputs[0]
 653 
 654   op_type = t.op.type
 655   if "Variable" in op_type or "VarHandle" in op_type:
 656     return t
 657   else:
 658     return None

 661 def _force_data_dependency(first_compute, then_compute):
 662   """Force all of `then_compute` to depend on all of `first_compute`.
 663 
 664   Uses a dummy data dependency, which is useful when running on TPUs because
 665   XLA ignores control dependencies. Only supports float arguments.
 666 
 667   Args:
 668     first_compute: `list<Tensor>`. These will be made to run before the
 669       `Tensor`s `then_compute`.
 670     then_compute: `list<Tensor>`. These will run after all the `Tensor`s in
 671       `first_compute`.
 672 
 673   Returns:
 674     `list<Tensor>`, same length as `then_compute`.
 675 
 676   Raises:
 677     ValueError: if ranks are unknown or types are not floating.
 678   """
 679 
 680   def _first_element(x):
 681     if x.get_shape().ndims is None:
 682       raise ValueError("Rank of Tensor %s must be known" % x)
 683     ndims = x.get_shape().ndims
 684     begin = framework_ops.convert_to_tensor([0] * ndims, dtype=dtypes.int32)
 685     size = framework_ops.convert_to_tensor([1] * ndims, dtype=dtypes.int32)
 686     return array_ops.reshape(array_ops.slice(x, begin, size), [])
 687 
 688   first_compute_sum = math_ops.add_n(
 689       [_first_element(x) for x in first_compute if x is not None])
 690   dtype = first_compute_sum.dtype
 691   if not dtype.is_floating:
 692     raise ValueError("_force_data_dependency only supports floating dtypes.")
 693   epsilon = np.finfo(dtype.as_numpy_dtype).tiny
 694   zero = array_ops.stop_gradient(epsilon * first_compute_sum)
 695 
 696   return [
 697       array_ops.identity(x) + zero if x is not None else None
 698       for x in then_compute
 699   ]
 700 
 701 
 702 def _tuple_with_data_dep(tensors):
 703   return _force_data_dependency(tensors, tensors)
