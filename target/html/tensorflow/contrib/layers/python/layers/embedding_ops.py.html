  48 def safe_embedding_lookup_sparse(embedding_weights,
  49                                  sparse_ids,
  50                                  sparse_weights=None,
  51                                  combiner=None,
  52                                  default_id=None,
  53                                  name=None,
  54                                  partition_strategy="div",
  55                                  max_norm=None):
  56   """Lookup embedding results, accounting for invalid IDs and empty features.
  57 
  58   The partitioned embedding in `embedding_weights` must all be the same shape
  59   except for the first dimension. The first dimension is allowed to vary as the
  60   vocabulary size is not necessarily a multiple of `P`.  `embedding_weights`
  61   may be a `PartitionedVariable` as returned by using `tf.get_variable()` with a
  62   partitioner.
  63 
  64   Invalid IDs (< 0) are pruned from input IDs and weights, as well as any IDs
  65   with non-positive weight. For an entry with no features, the embedding vector
  66   for `default_id` is returned, or the 0-vector if `default_id` is not supplied.
  67 
  68   The ids and weights may be multi-dimensional. Embeddings are always aggregated
  69   along the last dimension.
  70 
  71   Args:
  72     embedding_weights:  A list of `P` float tensors or values representing
  73         partitioned embedding tensors.  Alternatively, a `PartitionedVariable`,
  74         created by partitioning along dimension 0.  The total unpartitioned
  75         shape should be `[e_0, e_1, ..., e_m]`, where `e_0` represents the
  76         vocab size and `e_1, ..., e_m` are the embedding dimensions.
  77     sparse_ids: `SparseTensor` of shape `[d_0, d_1, ..., d_n]` containing the
  78         ids. `d_0` is typically batch size.
  79     sparse_weights: `SparseTensor` of same shape as `sparse_ids`, containing
  80         float weights corresponding to `sparse_ids`, or `None` if all weights
  81         are be assumed to be 1.0.
  82     combiner: A string specifying how to combine embedding results for each
  83         entry. Currently "mean", "sqrtn" and "sum" are supported, with "mean"
  84         the default.
  85     default_id: The id to use for an entry with no features.
  86     name: A name for this operation (optional).
  87     partition_strategy: A string specifying the partitioning strategy.
  88         Currently `"div"` and `"mod"` are supported. Default is `"div"`.
  89     max_norm: If not None, all embeddings are l2-normalized to max_norm before
  90         combining.
  91 
  92 
  93   Returns:
  94     Dense tensor of shape `[d_0, d_1, ..., d_{n-1}, e_1, ..., e_m]`.
  95 
  96   Raises:
  97     ValueError: if `embedding_weights` is empty.
  98   """
  99   if combiner is None:
 100     logging.warn("The default value of combiner will change from \"mean\" "
 101                  "to \"sqrtn\" after 2016/11/01.")
 102     combiner = "mean"
 103   if embedding_weights is None:
 104     raise ValueError("Missing embedding_weights %s." % embedding_weights)
 105   if isinstance(embedding_weights, variables.PartitionedVariable):
 106     embedding_weights = list(embedding_weights)  # get underlying Variables.
 107   if not isinstance(embedding_weights, list):
 108     embedding_weights = [embedding_weights]
 109   if len(embedding_weights) < 1:
 110     raise ValueError("Missing embedding_weights %s." % embedding_weights)
 111 
 112   dtype = sparse_weights.dtype if sparse_weights is not None else None
 113   if isinstance(embedding_weights, variables.PartitionedVariable):
 114     embedding_weights = list(embedding_weights)
 115   embedding_weights = [
 116       ops.convert_to_tensor(w, dtype=dtype) for w in embedding_weights
 117   ]
 118 
 119   contrib_tensor_util.assert_same_float_dtype(embedding_weights +
 120                                               [sparse_weights])
 121 
 122   with ops.name_scope(name, "embedding_lookup",
 123                       embedding_weights + [sparse_ids,
 124                                            sparse_weights]) as scope:
 125     # Reshape higher-rank sparse ids and weights to linear segment ids.
 126     original_shape = sparse_ids.dense_shape
 127     original_rank_dim = sparse_ids.dense_shape.get_shape()[0]
 128     original_rank = (
 129         array_ops.size(original_shape)
 130         if original_rank_dim.value is None
 131         else original_rank_dim.value)
 132     sparse_ids = sparse_ops.sparse_reshape(sparse_ids, [
 133         math_ops.reduce_prod(
 134             array_ops.slice(original_shape, [0], [original_rank - 1])),
 135         array_ops.gather(original_shape, original_rank - 1)])
 136     if sparse_weights is not None:
 137       sparse_weights = sparse_tensor.SparseTensor(
 138           sparse_ids.indices,
 139           sparse_weights.values, sparse_ids.dense_shape)
 140 
 141     # Prune invalid ids and weights.
 142     sparse_ids, sparse_weights = _prune_invalid_ids(sparse_ids, sparse_weights)
 143     if combiner != "sum":
 144       sparse_ids, sparse_weights = _prune_invalid_weights(
 145           sparse_ids, sparse_weights)
 146 
 147     # Fill in dummy values for empty features, if necessary.
 148     sparse_ids, is_row_empty = sparse_ops.sparse_fill_empty_rows(sparse_ids,
 149                                                                  default_id or
 150                                                                  0)
 151     if sparse_weights is not None:
 152       sparse_weights, _ = sparse_ops.sparse_fill_empty_rows(sparse_weights, 1.0)
 153 
 154     result = embedding_ops.embedding_lookup_sparse(
 155         embedding_weights,
 156         sparse_ids,
 157         sparse_weights,
 158         combiner=combiner,
 159         partition_strategy=partition_strategy,
 160         name=None if default_id is None else scope,
 161         max_norm=max_norm)
 162 
 163     if default_id is None:
 164       # Broadcast is_row_empty to the same shape as embedding_lookup_result,
 165       # for use in Select.
 166       is_row_empty = array_ops.tile(
 167           array_ops.reshape(is_row_empty, [-1, 1]),
 168           array_ops.stack([1, array_ops.shape(result)[1]]))
 169 
 170       result = array_ops.where(is_row_empty,
 171                                array_ops.zeros_like(result),
 172                                result,
 173                                name=scope)
 174 
 175     # Reshape back from linear ids back into higher-dimensional dense result.
 176     final_result = array_ops.reshape(
 177         result,
 178         array_ops.concat([
 179             array_ops.slice(
 180                 math_ops.cast(original_shape, dtypes.int32), [0],
 181                 [original_rank - 1]),
 182             array_ops.slice(array_ops.shape(result), [1], [-1])
 183         ], 0))
 184     final_result.set_shape(tensor_shape.unknown_shape(
 185         (original_rank_dim - 1).value).concatenate(result.get_shape()[1:]))
 186     return final_result
 187

 189 def _prune_invalid_ids(sparse_ids, sparse_weights):
 190   """Prune invalid IDs (< 0) from the input ids and weights."""
 191   is_id_valid = math_ops.greater_equal(sparse_ids.values, 0)
 192   if sparse_weights is not None:
 193     is_id_valid = math_ops.logical_and(
 194         is_id_valid,
 195         array_ops.ones_like(sparse_weights.values, dtype=dtypes.bool))
 196   sparse_ids = sparse_ops.sparse_retain(sparse_ids, is_id_valid)
 197   if sparse_weights is not None:
 198     sparse_weights = sparse_ops.sparse_retain(sparse_weights, is_id_valid)
 199   return sparse_ids, sparse_weights
 200 
 201 
 202 def _prune_invalid_weights(sparse_ids, sparse_weights):
 203   """Prune invalid weights (< 0) from the input ids and weights."""
 204   if sparse_weights is not None:
 205     is_weights_valid = math_ops.greater(sparse_weights.values, 0)
 206     sparse_ids = sparse_ops.sparse_retain(sparse_ids, is_weights_valid)
 207     sparse_weights = sparse_ops.sparse_retain(sparse_weights, is_weights_valid)
 208   return sparse_ids, sparse_weights

 211 def scattered_embedding_lookup(params,
 212                                values,
 213                                dimension,
 214                                name=None,
 215                                hash_key=None):
 216   """Looks up embeddings using parameter hashing for each value in `values`.
 217 
 218   The i-th embedding component of a value v in `values` is found by retrieving
 219   the weight whose index is a fingerprint of the pair (v,i).
 220   The concept is explored as "feature hashing" for model compression in this
 221   paper: http://arxiv.org/pdf/1504.04788.pdf
 222 
 223   Feature hashing has the pleasant effect of allowing us to compute an embedding
 224   without needing a pre-determined vocabulary, relieving some amount of process
 225   complexity. It also allows for us to maintain embeddings for possibly
 226   trillions of features with a fixed amount of memory.
 227 
 228   Note that this is superior to out-of-vocabulary shared "hash buckets" in that
 229   the embedding is extremely likely to be unique for each token as opposed to
 230   being shared across probably-colliding tokens. The price is that we must
 231   compute a hash once for each scalar in the token's embedding as opposed to
 232   once per token.
 233 
 234   If `params` is a list, it represents a partition of the embedding parameters.
 235   Each tensor in the list should have the same length, except for the first ones
 236   which may have an additional element. For instance 10 parameters can be
 237   partitioned in 4 tensors with length `[3, 3, 2, 2]`.
 238 
 239   Args:
 240     params: A `Tensor`, `list` of `Tensors`, or `PartitionedVariable`.
 241       Each tensor must be of rank 1 with fully-defined shape.
 242     values: `Tensor` of values to be embedded with shape `[d0, ..., dn]`.
 243     dimension: Embedding dimension.
 244     name: An optional name for this op.
 245     hash_key: Specify the hash_key that will be used by the `FingerprintCat64`
 246       function to combine the crosses fingerprints on SparseFeatureCrossOp
 247       (optional).
 248 
 249   Returns:
 250     A `Tensor` with shape `[d0, ..., dn, dimension]`.
 251 
 252   Raises:
 253     ValueError: if dimension is not positive or the partition size is invalid.
 254   """
 255   if dimension is None:
 256     raise ValueError("You must specify dimension.")
 257   return _sampled_scattered_embedding_lookup(
 258       params, values, dimension=dimension, sampled_candidates=None,
 259       hash_key=hash_key, name=name)

 262 def _sampled_scattered_embedding_lookup(
 263     params, values, dimension=None, sampled_candidates=None, hash_key=None,
 264     name=None):
 265   """Looks up embeddings using parameter hashing for each value in `values`.
 266 
 267   This method looks up selected embedding dimensions if `sampled_candidates` is
 268   given, otherwise looks up all dimensions.
 269 
 270   The i-th embedding component of a value v in `values` is found by retrieving
 271   the weight whose index is a fingerprint of the pair (v,i).
 272   The concept is explored as "feature hashing" for model compression in this
 273   paper: http://arxiv.org/pdf/1504.04788.pdf
 274 
 275   Feature hashing has the pleasant effect of allowing us to compute an embedding
 276   without needing a pre-determined vocabulary, relieving some amount of process
 277   complexity. It also allows for us to maintain embeddings for possibly
 278   trillions of features with a fixed amount of memory.
 279 
 280   Note that this is superior to out-of-vocabulary shared "hash buckets" in that
 281   the embedding is extremely likely to be unique for each token as opposed to
 282   being shared across probably-colliding tokens. The price is that we must
 283   compute a hash once for each scalar in the token's embedding as opposed to
 284   once per token.
 285 
 286   If `params` is a list, it represents a partition of the embedding parameters.
 287   Each tensor in the list should have the same length, except for the first ones
 288   which may have an additional element. For instance 10 parameters can be
 289   partitioned in 4 tensors with length `[3, 3, 2, 2]`.
 290 
 291   Args:
 292     params: A `Tensor`, `list` of `Tensors`, or `PartitionedVariable`.
 293       Each tensor must be of rank 1 with fully-defined shape.
 294     values: `Tensor` of values to be embedded with shape `[d0, ..., dn]`.
 295     dimension: Embedding dimension. The user must specify either `dimension` or
 296       `sampled_candidates`.
 297     sampled_candidates: An optional `Tensor` of slice indices to keep along the
 298       final dimension with shape `[d0, ..., dn, N]`. If given, `dimension` is
 299       ignored. If `None`, looks up all candidates.
 300     hash_key: Specify the hash_key that will be used by the `FingerprintCat64`
 301       function to combine the crosses fingerprints on SparseFeatureCrossOp
 302       (optional).
 303     name: An optional name for this op.
 304 
 305   Returns:
 306     A `Tensor` with shape `[d0, ..., dn, dimension]`.
 307     If `sampled_candidates` is given, the output shape is `[d0, ..., dn, N]`
 308 
 309   Raises:
 310     ValueError: if dimension is not positive or the partition size is invalid.
 311   """
 312   if isinstance(params, variables.PartitionedVariable):
 313     params = list(params)
 314   if not isinstance(params, list):
 315     params = [params]
 316 
 317   with ops.name_scope(name, "scattered_embedding_lookup",
 318                       params + [dimension, values]):
 319     # Flatten the values
 320     values_shape = array_ops.shape(values)
 321     values = array_ops.reshape(values, [-1, 1])
 322 
 323     if sampled_candidates is None:
 324       if dimension is None:
 325         raise ValueError(
 326             "You must specify either dimension or sampled_candidates.")
 327       if dimension <= 0:
 328         raise ValueError("Dimension must be >0. Given is %d" % dimension)
 329       sampled_candidates = array_ops.tile(array_ops.expand_dims(
 330           math_ops.range(0, dimension), 0), array_ops.shape(values))
 331     else:
 332       dimension = array_ops.shape(sampled_candidates)[
 333           math_ops.subtract(array_ops.rank(sampled_candidates), 1)]
 334       sampled_candidates_shape = array_ops.shape(sampled_candidates)
 335       dimension_tensor = array_ops.reshape(dimension, shape=[1,])
 336       expected_shape = array_ops.concat([values_shape, dimension_tensor], 0)
 337       with ops.control_dependencies([control_flow_ops.Assert(
 338           math_ops.reduce_all(math_ops.equal(sampled_candidates_shape,
 339                                              expected_shape)),
 340           ["The shape of sampled_candidates: ", sampled_candidates_shape,
 341            " does not match the shape of values: ", values_shape])]):
 342         # Flatten sampled_candidates, same way as values are flattened.
 343         sampled_candidates = array_ops.reshape(sampled_candidates,
 344                                                [-1, dimension])
 345 
 346     num_partitions = len(params)
 347     partition_sizes = []
 348     for p in range(num_partitions):
 349       shape = params[p].get_shape()
 350       shape.assert_has_rank(1)
 351       shape.assert_is_fully_defined()
 352       partition_sizes.append(shape[0].value)
 353     num_params = sum(partition_sizes)  # Total number of parameters.
 354 
 355     # Assert the size of each partition.
 356     for p in range(num_partitions):
 357       expected_size = (num_params - p - 1) // num_partitions + 1
 358       if partition_sizes[p] != expected_size:
 359         raise ValueError("Tensor %d in params has size %d, expected %d." %
 360                          (p, partition_sizes[p], expected_size))
 361 
 362     # With two values v1 and v2 and 3 dimensions, we will cross
 363     # [[0, 1, 2], [0, 1, 2]] with [[v1], [v2]].
 364     tensors_to_cross = [sampled_candidates, values]
 365     ids = sparse_feature_cross_op.sparse_feature_cross(
 366         tensors_to_cross, hashed_output=True, num_buckets=num_params,
 367         hash_key=hash_key)
 368     ids = sparse_ops.sparse_tensor_to_dense(ids)
 369 
 370     # No need to validate the indices since we have checked the params
 371     # dimensions and we know the largest id.
 372     result = embedding_ops.embedding_lookup(
 373         params, ids, partition_strategy="div")
 374 
 375     return array_ops.reshape(result,
 376                              array_ops.concat([values_shape, [dimension]], 0))
 377

 379 def scattered_embedding_lookup_sparse(params,
 380                                       sparse_values,
 381                                       dimension,
 382                                       combiner=None,
 383                                       default_value=None,
 384                                       name=None,
 385                                       hash_key=None):
 386   """Looks up embeddings of a sparse feature using parameter hashing.
 387 
 388   See `tf.contrib.layers.scattered_embedding_lookup` for embedding with hashing.
 389 
 390   Args:
 391     params: A `Tensor`, `list` of `Tensors`, or `PartitionedVariable`.
 392       Each tensor must be of rank 1 with fully-defined shape.
 393     sparse_values: A 2-D `SparseTensor` containing the values to be embedded.
 394       Some rows may be empty.
 395     dimension: Embedding dimension
 396     combiner: A string specifying how to combine embedding results for each
 397         entry. Currently "mean", "sqrtn" and "sum" are supported, with "mean"
 398         the default.
 399     default_value: The value to use for an entry with no features.
 400     name: An optional name for this op.
 401     hash_key: Specify the hash_key that will be used by the `FingerprintCat64`
 402       function to combine the crosses fingerprints on SparseFeatureCrossOp
 403       (optional).
 404 
 405   Returns:
 406      Dense tensor with shape [N, dimension] with N the number of rows in
 407        sparse_values.
 408 
 409   Raises:
 410     TypeError: If sparse_values is not a SparseTensor.
 411     ValueError: If combiner is not one of {"mean", "sqrtn", "sum"}.
 412   """
 413   if combiner is None:
 414     logging.warn("The default value of combiner will change from \"mean\" "
 415                  "to \"sqrtn\" after 2016/11/01.")
 416     combiner = "mean"
 417   if isinstance(params, variables.PartitionedVariable):
 418     params = list(params)
 419   if not isinstance(params, list):
 420     params = [params]
 421   if not isinstance(sparse_values, sparse_tensor.SparseTensor):
 422     raise TypeError("sparse_values must be SparseTensor")
 423 
 424   with ops.name_scope(name, "scattered_embedding_lookup_sparse",
 425                       params + [sparse_values]) as scope:
 426     # Fill in the empty rows.
 427     if default_value is None:
 428       # Random default values to reduce the risk of collision.
 429       if sparse_values.dtype == dtypes.string:
 430         default_value = "6ZxWzWOHxZ"
 431       else:
 432         default_value = 1288896567
 433     sparse_values, _ = sparse_ops.sparse_fill_empty_rows(
 434         sparse_values, default_value)
 435 
 436     segment_ids = sparse_values.indices[:, 0]
 437     if segment_ids.dtype != dtypes.int32:
 438       segment_ids = math_ops.cast(segment_ids, dtypes.int32)
 439 
 440     values = sparse_values.values
 441     values, idx = array_ops.unique(values)
 442 
 443     embeddings = scattered_embedding_lookup(
 444         params, values, dimension, hash_key=hash_key)
 445 
 446     if combiner == "sum":
 447       embeddings = math_ops.sparse_segment_sum(embeddings, idx, segment_ids,
 448                                                name=scope)
 449     elif combiner == "mean":
 450       embeddings = math_ops.sparse_segment_mean(embeddings, idx, segment_ids,
 451                                                 name=scope)
 452     elif combiner == "sqrtn":
 453       embeddings = math_ops.sparse_segment_sqrt_n(embeddings, idx, segment_ids,
 454                                                   name=scope)
 455     else:
 456       raise ValueError("Combiner must be one of 'mean', 'sqrtn' or 'sum'.")
 457 
 458     return embeddings
 459

 461 def embedding_lookup_unique(params, ids, partition_strategy="mod", name=None):
 462   """Version of embedding_lookup that avoids duplicate lookups.
 463 
 464   This can save communication in the case of repeated ids.
 465   Same interface as embedding_lookup. Except it supports multi-dimensional `ids`
 466   which allows to not reshape input/output to fit gather.
 467 
 468   Args:
 469     params: A list of tensors with the same shape and type, or a
 470       `PartitionedVariable`. Shape `[index, d1, d2, ...]`.
 471     ids: A one-dimensional `Tensor` with type `int32` or `int64` containing
 472       the ids to be looked up in `params`. Shape `[ids1, ids2, ...]`.
 473     partition_strategy: A string specifying the partitioning strategy, relevant
 474       if `len(params) > 1`. Currently `"div"` and `"mod"` are supported. Default
 475       is `"mod"`.
 476     name: A name for this operation (optional).
 477 
 478   Returns:
 479     A `Tensor` with the same type as the tensors in `params` and dimension of
 480     `[ids1, ids2, d1, d2, ...]`.
 481 
 482   Raises:
 483     ValueError: If `params` is empty.
 484   """
 485   with ops.name_scope(name, "EmbeddingLookupUnique", [params, ids]):
 486     ids = ops.convert_to_tensor(ids)
 487     shape = array_ops.shape(ids)
 488     ids_flat = array_ops.reshape(
 489         ids, math_ops.reduce_prod(shape, keepdims=True))
 490     unique_ids, idx = array_ops.unique(ids_flat)
 491     unique_embeddings = embedding_ops.embedding_lookup(params, unique_ids,
 492                                                        partition_strategy)
 493     embeds_flat = array_ops.gather(unique_embeddings, idx)
 494     embed_shape = array_ops.concat(
 495         [shape, array_ops.shape(unique_embeddings)[1:]], 0)
 496     embeds = array_ops.reshape(embeds_flat, embed_shape)
 497     embeds.set_shape(ids.get_shape().concatenate(
 498         unique_embeddings.get_shape()[1:]))
 499     return embeds

 502 def _sampled_scattered_embedding_lookup_sparse(params,
 503                                                sp_values,
 504                                                dimension=None,
 505                                                sampled_candidates=None,
 506                                                hash_key=None,
 507                                                with_sign_hash=False,
 508                                                name=None):
 509   """Looks up embeddings using parameter hashing for sparse values.
 510 
 511   This method looks up selected embedding dimensions if `sampled_candidates` is
 512   given, otherwise looks up all dimensions.
 513 
 514   The i-th embedding component of a value v in `values` is found by retrieving
 515   the weight whose index is a fingerprint of the pair (v,i).
 516   The concept is explored as "feature hashing" for model compression in this
 517   paper: http://arxiv.org/pdf/1504.04788.pdf
 518 
 519   This is logically equivalent to:
 520   * Transforming `sp_values` (which has shape `[d0, d1]`) into a one-hot
 521     `Tensor` of shape `[d0, N]`.
 522   * Multiplying with a `Tensor` `h` of shape `[N, dimension]`, where
 523     `h(i, j) = params[hash(i, j)]`.
 524 
 525   Args:
 526     params: A float `Tensor` with rank 1 and fully-defined shape.
 527     sp_values: A 2D `SparseTensor` to be embedded with shape `[d0, d1]`.
 528     dimension: An int `Tensor` of the final dimension. The user needs to provide
 529       either `dimension` or `sampled_candidates`.
 530     sampled_candidates: An optional `Tensor` of column indices to keep along
 531       the final dimension with shape `[d0, N]`. If given, `dimension` is
 532       ignored. If `None`, looks up all candidates.
 533     hash_key: Specify the hash_key that will be used by the `FingerprintCat64`
 534       function to combine the crosses fingerprints on SparseFeatureCrossOp
 535       (optional).
 536     with_sign_hash:  A `bool` indicating whether `h(i, j)` should be multiplied
 537       by `+1` or `-1`, where the value selected is determined by hashing
 538       `(i, j)`. This is often necessary to remove bias resulting from hash
 539       collisions.
 540     name: An optional name for this op.
 541 
 542   Returns:
 543     A `Tensor` of shape `[d0, dimension]`.
 544     If `sampled_candidates` is given, the output shape is `[d0, N]`.
 545 
 546   Raises:
 547     TypeError: If sp_values is not `SparseTensor`.
 548     ValueError: If both `dimension` and `sampled_candidates` are `None`.
 549   """
 550   if not isinstance(sp_values, sparse_tensor.SparseTensor):
 551     raise TypeError("sp_values must be SparseTensor")
 552 
 553   with ops.name_scope(
 554       name=name,
 555       default_name="sampled_scattered_embedding_lookup_sparse",
 556       values=[sp_values, params, dimension, sampled_candidates]) as name_scope:
 557     segment_ids = sp_values.indices[:, 0]
 558     if sampled_candidates is not None:
 559       # Tile sampled_candidates so there is one line corresponding to each
 560       # element in sp_values.values
 561       sampled_candidates = array_ops.gather(sampled_candidates, segment_ids)
 562 
 563     embeddings = _sampled_scattered_embedding_lookup(
 564         params, sp_values.values, dimension=dimension,
 565         sampled_candidates=sampled_candidates,
 566         hash_key=hash_key, name="values_lookup")
 567     if with_sign_hash:
 568       signs = _sampled_scattered_embedding_lookup(
 569           array_ops.constant([-1., 1.]), sp_values.values, dimension=dimension,
 570           sampled_candidates=sampled_candidates, hash_key=hash_key,
 571           name="signs_lookup")
 572       embeddings = math_ops.multiply(signs, embeddings, name="signs_hash")
 573 
 574     if segment_ids.dtype != dtypes.int32:
 575       segment_ids = math_ops.cast(segment_ids, dtypes.int32)
 576     num_segments = array_ops.shape(sp_values)[0]
 577 
 578     return math_ops.unsorted_segment_sum(embeddings, segment_ids,
 579                                          num_segments=num_segments,
 580                                          name=name_scope)

 583 def embedding_lookup_sparse_with_distributed_aggregation(
 584     params,
 585     sp_ids,
 586     sp_weights,
 587     partition_strategy="mod",
 588     name=None,
 589     combiner=None,
 590     max_norm=None):
 591   """Computes embeddings for the given ids and weights.
 592 
 593   Embeddings belonging to same param are aggregated on that device first. This
 594   op is intended to decrease data transmission and improve parallelism. See
 595   `tf.nn.embedding_lookup_sparse` for the functionality and example of this op.
 596 
 597   Args:
 598     params: A single tensor representing the complete embedding tensor,
 599       or a list of P tensors all of same shape except for the first dimension,
 600       representing sharded embedding tensors.  Alternatively, a
 601       `PartitionedVariable`, created by partitioning along dimension 0. Each
 602       element must be appropriately sized for the given `partition_strategy`.
 603     sp_ids: N x M SparseTensor of int64 ids (typically from FeatureValueToId),
 604       where N is typically batch size and M is arbitrary.
 605     sp_weights: either a SparseTensor of float / double weights, or None to
 606       indicate all weights should be taken to be 1. If specified, sp_weights
 607       must have exactly the same shape and indices as sp_ids.
 608     partition_strategy: A string specifying the partitioning strategy, relevant
 609       if `len(params) > 1`. Currently `"div"` and `"mod"` are supported. Default
 610       is `"mod"`. See `tf.nn.embedding_lookup` for more details.
 611     name: Optional name for the op.
 612     combiner: A string specifying the reduction op. Currently "mean", "sqrtn"
 613       and "sum" are supported.
 614       "sum" computes the weighted sum of the embedding results for each row.
 615       "mean" is the weighted sum divided by the total weight.
 616       "sqrtn" is the weighted sum divided by the square root of the sum of the
 617       squares of the weights.
 618     max_norm: If not None, each embedding is normalized to have l2 norm equal
 619       to max_norm before combining.
 620 
 621   Returns:
 622     A dense tensor representing the combined embeddings for the
 623     sparse ids. For each row in the dense tensor represented by sp_ids, the op
 624     looks up the embeddings for all ids in that row, multiplies them by the
 625     corresponding weight, and combines these embeddings as specified.
 626 
 627   Raises:
 628     TypeError: If sp_ids is not a SparseTensor, or if sp_weights is neither
 629       None nor SparseTensor.
 630     ValueError: If combiner is not one of {"mean", "sqrtn", "sum"}.
 631   """
 632   if combiner is None:
 633     logging.warn("The default value of combiner will change from \"mean\" "
 634                  "to \"sqrtn\" after 2016/11/01.")
 635     combiner = "mean"
 636   if combiner not in ("mean", "sqrtn", "sum"):
 637     raise ValueError("combiner must be one of 'mean', 'sqrtn' or 'sum'")
 638   if isinstance(params, variables.PartitionedVariable):
 639     params = list(params)  # Iterate to get the underlying Variables.
 640   if not isinstance(params, list):
 641     params = [params]
 642   if not isinstance(sp_ids, sparse_tensor.SparseTensor):
 643     raise TypeError("sp_ids must be SparseTensor")
 644   ignore_weights = sp_weights is None
 645   if not ignore_weights:
 646     if not isinstance(sp_weights, sparse_tensor.SparseTensor):
 647       raise TypeError("sp_weights must be either None or SparseTensor")
 648     sp_ids.values.get_shape().assert_is_compatible_with(
 649         sp_weights.values.get_shape())
 650     sp_ids.indices.get_shape().assert_is_compatible_with(
 651         sp_weights.indices.get_shape())
 652     sp_ids.dense_shape.get_shape().assert_is_compatible_with(
 653         sp_weights.dense_shape.get_shape())
 654     # TODO(yleon): Add enhanced node assertions to verify that sp_ids and
 655     # sp_weights have equal indices and shapes.
 656 
 657   with ops.name_scope(name, "embedding_lookup_sparse",
 658                       params + [sp_ids]) as name:
 659     segment_ids = sp_ids.indices[:, 0]
 660     if segment_ids.dtype != dtypes.int32:
 661       segment_ids = math_ops.cast(segment_ids, dtypes.int32)
 662 
 663     ids = sp_ids.values
 664     if ignore_weights:
 665       ids, idx = array_ops.unique(ids)
 666     else:
 667       idx = None
 668 
 669     weights = None if ignore_weights else sp_weights.values
 670     embeddings = _embedding_lookup_with_distributed_aggregation(
 671         params,
 672         ids,
 673         partition_strategy=partition_strategy,
 674         max_norm=max_norm,
 675         weights=weights,
 676         idx=idx,
 677         segment_ids=segment_ids)
 678     # Set weights to all one if ignore weights.
 679     if ignore_weights:
 680       weights = array_ops.fill([array_ops.shape(segment_ids)[0]], 1)
 681     if weights.dtype != embeddings.dtype:
 682       weights = math_ops.cast(weights, embeddings.dtype)
 683     # Reshape weights.
 684     ones = array_ops.fill(
 685         array_ops.expand_dims(array_ops.rank(embeddings) - 1, 0), 1)
 686     bcast_weights_shape = array_ops.concat([array_ops.shape(weights), ones], 0)
 687     orig_weights_shape = weights.get_shape()
 688     weights = array_ops.reshape(weights, bcast_weights_shape)
 689     if embeddings.get_shape().ndims is not None:
 690       weights.set_shape(
 691           orig_weights_shape.concatenate(
 692               [1 for _ in range(embeddings.get_shape().ndims - 1)]))
 693 
 694     if combiner == "mean":
 695       weight_sum = math_ops.segment_sum(weights, segment_ids)
 696       embeddings = math_ops.div(embeddings, weight_sum)
 697     elif combiner == "sqrtn":
 698       weights_squared = math_ops.pow(weights, 2)
 699       weight_sum = math_ops.segment_sum(weights_squared, segment_ids)
 700       weight_sum_sqrt = math_ops.sqrt(weight_sum)
 701       embeddings = math_ops.div(embeddings, weight_sum_sqrt)
 702     elif combiner != "sum":
 703       assert False, "Unrecognized combiner"
 704     return embeddings

 707 def _do_gather(params, ids, name=None):
 708   """Deals with doing gather differently for resource variables."""
 709   if isinstance(params, resource_variable_ops.ResourceVariable):
 710     return params.sparse_read(ids, name=name)
 711   return array_ops.gather(params, ids, name=name)

 714 def _embedding_lookup_with_distributed_aggregation(params,
 715                                                    ids,
 716                                                    partition_strategy="mod",
 717                                                    name=None,
 718                                                    max_norm=None,
 719                                                    weights=None,
 720                                                    idx=None,
 721                                                    segment_ids=None):
 722   """Lookup helper for embedding_lookup_sparse_with_distributed_aggregation."""
 723   if params is None or params == []:  # pylint: disable=g-explicit-bool-comparison
 724     raise ValueError("Need at least one param")
 725   if isinstance(params, variables.PartitionedVariable):
 726     params = list(params)  # Iterate to get the underlying Variables.
 727   if not isinstance(params, list):
 728     params = [params]
 729 
 730   def maybe_normalize(x):
 731     if max_norm is not None:
 732       if x.get_shape().ndims is not None:
 733         ndims = x.get_shape().ndims
 734       else:
 735         ndims = array_ops.size(array_ops.shape(x))
 736       return clip_ops.clip_by_norm(x, max_norm, axes=list(range(1, ndims)))
 737     return x
 738 
 739   with ops.name_scope(name, "embedding_lookup_with_distributed_aggregation",
 740                       params + [ids]) as name:
 741     np = len(params)  # Number of partitions
 742     # Preserve the resource variable status to avoid accidental dense reads.
 743     if not any(
 744         isinstance(p, resource_variable_ops.ResourceVariable) for p in params):
 745       params = ops.convert_n_to_tensor_or_indexed_slices(params, name="params")
 746     if np == 1:
 747       with ops.colocate_with(params[0]):
 748         ret = maybe_normalize(_do_gather(params[0], ids))
 749         ignore_weights = weights is None
 750         if not ignore_weights:
 751           if weights.dtype != ret.dtype:
 752             weights = math_ops.cast(weights, ret.dtype)
 753           # Reshape to allow broadcast
 754           ones = array_ops.fill(
 755               array_ops.expand_dims(array_ops.rank(ret) - 1, 0), 1)
 756           bcast_weights_shape = array_ops.concat(
 757               [array_ops.shape(weights), ones], 0)
 758           orig_weights_shape = weights.get_shape()
 759           weights = array_ops.reshape(weights, bcast_weights_shape)
 760           # Set weights shape after reshape
 761           if ret.get_shape().ndims is not None:
 762             weights.set_shape(
 763                 orig_weights_shape.concatenate(
 764                     [1 for _ in range(ret.get_shape().ndims - 1)]))
 765           ret *= weights
 766           return math_ops.segment_sum(ret, segment_ids, name=name)
 767         else:
 768           return math_ops.sparse_segment_sum(ret, idx, segment_ids, name=name)
 769     else:
 770       ids = ops.convert_to_tensor(ids, name="ids")
 771       flat_ids = array_ops.reshape(ids, [-1])
 772       original_indices = math_ops.range(array_ops.size(flat_ids))
 773 
 774       # Create p_assignments and set new_ids depending on the strategy.
 775       if partition_strategy == "mod":
 776         p_assignments = flat_ids % np
 777         new_ids = flat_ids // np
 778       elif partition_strategy == "div":
 779         # Compute num_total_ids as the sum of dim-0 of params, then assign to
 780         # partitions based on a constant number of ids per partition. Optimize
 781         # if we already know the full shape statically.
 782         dim_0_size = params[0].get_shape()[0]
 783         for p in xrange(1, np):
 784           dim_0_size += params[p].get_shape()[0]
 785         if dim_0_size.value:
 786           num_total_ids = constant_op.constant(dim_0_size.value, flat_ids.dtype)
 787         else:
 788           dim_0_sizes = []
 789           for p in xrange(np):
 790             if params[p].get_shape()[0].value is not None:
 791               dim_0_sizes.append(params[p].get_shape()[0].value)
 792             else:
 793               with ops.colocate_with(params[p]):
 794                 dim_0_sizes.append(array_ops.shape(params[p])[0])
 795           num_total_ids = math_ops.reduce_sum(
 796               math_ops.cast(array_ops.stack(dim_0_sizes), flat_ids.dtype))
 797         ids_per_partition = num_total_ids // np
 798         extras = num_total_ids % np
 799 
 800         p_assignments = math_ops.maximum(flat_ids // (ids_per_partition + 1), (
 801             flat_ids - extras) // ids_per_partition)
 802 
 803         # Emulate a conditional using a boolean indicator tensor
 804         is_in_first_extras_partitions = math_ops.cast(p_assignments < extras,
 805                                                       flat_ids.dtype)
 806         new_ids = (is_in_first_extras_partitions * (flat_ids %
 807                                                     (ids_per_partition + 1)) +
 808                    (1 - is_in_first_extras_partitions) * (
 809                        (flat_ids - extras) % ids_per_partition))
 810       else:
 811         raise ValueError("Unrecognized partition strategy: " +
 812                          partition_strategy)
 813 
 814       # Cast partition assignments to int32 for use in dynamic_partition.
 815       # There really should not be more than 2^32 partitions.
 816       p_assignments = math_ops.cast(p_assignments, dtypes.int32)
 817       # Partition list of ids based on assignments into np separate lists
 818       gather_ids = data_flow_ops.dynamic_partition(new_ids, p_assignments, np)
 819       # Similarly, partition the original indices.
 820       pindices = data_flow_ops.dynamic_partition(original_indices,
 821                                                  p_assignments, np)
 822       # Do np separate lookups, finding embeddings for plist[p] in params[p]
 823       partitioned_result = []
 824       for p in xrange(np):
 825         with ops.colocate_with(params[p]):
 826           partitioned_result.append(_do_gather(params[p], gather_ids[p]))
 827 
 828       ignore_weights = weights is None
 829       if not ignore_weights:
 830         # Partition weights according to pindices.
 831         partitioned_weight = []
 832         for p in xrange(np):
 833           partitioned_weight.append(array_ops.gather(weights, pindices[p]))
 834       # Reshape each partition result.
 835       element_shape = params[0].get_shape()[1:]
 836       for p in params[1:]:
 837         element_shape = element_shape.merge_with(p.get_shape()[1:])
 838       if element_shape.is_fully_defined():
 839         for p in xrange(np):
 840           with ops.colocate_with(params[p]):
 841             partitioned_result[p] = array_ops.reshape(
 842                 partitioned_result[p],
 843                 array_ops.concat([array_ops.shape(pindices[p]), element_shape],
 844                                  0))
 845       else:
 846         with ops.colocate_with(params[0]):
 847           params_shape = array_ops.shape(params[0])
 848         for p in xrange(np):
 849           with ops.colocate_with(params[p]):
 850             partitioned_result[p] = array_ops.reshape(
 851                 partitioned_result[p],
 852                 array_ops.concat([
 853                     array_ops.shape(pindices[p]), array_ops.slice(
 854                         params_shape, [1], [-1])
 855                 ], 0))
 856       # Normalize each partition result.
 857       for p in xrange(np):
 858         with ops.colocate_with(params[p]):
 859           partitioned_result[p] = maybe_normalize(partitioned_result[p])
 860       if not ignore_weights:
 861         # Multiply each partition result with partition weights.
 862         for p in xrange(np):
 863           with ops.colocate_with(params[p]):
 864             if partitioned_weight[p].dtype != partitioned_result[p].dtype:
 865               partitioned_weight[p] = math_ops.cast(partitioned_weight[p],
 866                                                     partitioned_result[p].dtype)
 867             # Reshape partition weights.
 868             ones = array_ops.fill(
 869                 array_ops.expand_dims(
 870                     array_ops.rank(partitioned_result[p]) - 1, 0), 1)
 871             bcast_weights_shape = array_ops.concat(
 872                 [array_ops.shape(partitioned_weight[p]), ones], 0)
 873             orig_weights_shape = partitioned_weight[p].get_shape()
 874             partitioned_weight[p] = array_ops.reshape(partitioned_weight[p],
 875                                                       bcast_weights_shape)
 876             if partitioned_result[p].get_shape().ndims is not None:
 877               partitioned_weight[p].set_shape(
 878                   orig_weights_shape.concatenate([
 879                       1
 880                       for _ in range(partitioned_result[p].get_shape().ndims -
 881                                      1)
 882                   ]))
 883             partitioned_result[p] *= partitioned_weight[p]
 884       partitioned_segment_ids = []
 885       for p in xrange(np):
 886         if not ignore_weights:
 887           # Partition segment_ids according to pindices.
 888           p_segment_ids = array_ops.gather(segment_ids, pindices[p])
 889           # Number the p_segment_ids to meet segment_sum's requirements. Note
 890           # that unique_p_segment_ids contains unique segment ids of this
 891           # partition and these ids' order is unchanged.
 892           unique_p_segment_ids, unique_p_segment_idx = array_ops.unique(
 893               p_segment_ids)
 894           partitioned_segment_ids.append(unique_p_segment_ids)
 895           # segment_sum this partition's result.
 896           with ops.colocate_with(params[p]):
 897             partitioned_result[p] = math_ops.segment_sum(
 898                 partitioned_result[p], unique_p_segment_idx)
 899         else:
 900           # When ignore weights, we need to get indexs of elements in idx and
 901           # segment_ids.
 902           _, exclude_idx = array_ops.setdiff1d(idx, pindices[p])
 903           all_idx = math_ops.range(array_ops.shape(idx)[0])
 904           _, include_idx = array_ops.setdiff1d(all_idx, exclude_idx)
 905           # Gather segment_ids and idx according to indexs.
 906           p_segment_ids = array_ops.gather(segment_ids, include_idx)
 907           p_idx = array_ops.gather(idx, include_idx)
 908           # Number the p_segment_ids, same as ignore_weights case above.
 909           unique_p_segment_ids, unique_p_segment_idx = array_ops.unique(
 910               p_segment_ids)
 911           _, unique_p_idx_idx = array_ops.unique(p_idx)
 912           partitioned_segment_ids.append(unique_p_segment_ids)
 913           with ops.colocate_with(params[p]):
 914             partitioned_result[p] = math_ops.sparse_segment_sum(
 915                 partitioned_result[p], unique_p_idx_idx, unique_p_segment_idx)
 916       # Concat each partition's segment_ids and result for final segment_sum.
 917       concat_segment_ids = array_ops.concat(partitioned_segment_ids, 0)
 918       concat_partitioned_result = array_ops.concat(partitioned_result, 0)
 919       return math_ops.unsorted_segment_sum(
 920           concat_partitioned_result,
 921           concat_segment_ids,
 922           math_ops.reduce_max(concat_segment_ids) + 1,
 923           name=name)
