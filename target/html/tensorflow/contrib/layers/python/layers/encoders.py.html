  32 def bow_encoder(ids,
  33                 vocab_size,
  34                 embed_dim,
  35                 sparse_lookup=True,
  36                 initializer=None,
  37                 regularizer=None,
  38                 trainable=True,
  39                 scope=None,
  40                 reuse=None):
  41   """Maps a sequence of symbols to a vector per example by averaging embeddings.
  42 
  43   Args:
  44     ids: `[batch_size, doc_length]` `Tensor` or `SparseTensor` of type
  45       `int32` or `int64` with symbol ids.
  46     vocab_size: Integer number of symbols in vocabulary.
  47     embed_dim: Integer number of dimensions for embedding matrix.
  48     sparse_lookup: `bool`, if `True`, converts ids to a `SparseTensor`
  49         and performs a sparse embedding lookup. This is usually faster,
  50         but not desirable if padding tokens should have an embedding. Empty rows
  51         are assigned a special embedding.
  52     initializer: An initializer for the embeddings, if `None` default for
  53         current scope is used.
  54     regularizer: Optional regularizer for the embeddings.
  55     trainable: If `True` also add variables to the graph collection
  56       `GraphKeys.TRAINABLE_VARIABLES` (see tf.Variable).
  57     scope: Optional string specifying the variable scope for the op, required
  58         if `reuse=True`.
  59     reuse: If `True`, variables inside the op will be reused.
  60 
  61   Returns:
  62     Encoding `Tensor` `[batch_size, embed_dim]` produced by
  63     averaging embeddings.
  64 
  65   Raises:
  66     ValueError: If `embed_dim` or `vocab_size` are not specified.
  67   """
  68   if not vocab_size or not embed_dim:
  69     raise ValueError('Must specify vocab size and embedding dimension')
  70   with variable_scope.variable_scope(
  71       scope, 'bow_encoder', [ids], reuse=reuse):
  72     embeddings = variables.model_variable(
  73         'embeddings', shape=[vocab_size, embed_dim],
  74         initializer=initializer, regularizer=regularizer,
  75         trainable=trainable)
  76     if sparse_lookup:
  77       if isinstance(ids, sparse_tensor.SparseTensor):
  78         sparse_ids = ids
  79       else:
  80         sparse_ids = sparse_ops.dense_to_sparse_tensor(ids)
  81       return contrib_embedding_ops.safe_embedding_lookup_sparse(
  82           [embeddings], sparse_ids, combiner='mean', default_id=0)
  83     else:
  84       if isinstance(ids, sparse_tensor.SparseTensor):
  85         raise TypeError('ids are expected to be dense Tensor, got: %s', ids)
  86       return math_ops.reduce_mean(
  87           embedding_ops.embedding_lookup(embeddings, ids),
  88           reduction_indices=1)

  91 def embed_sequence(ids,
  92                    vocab_size=None,
  93                    embed_dim=None,
  94                    unique=False,
  95                    initializer=None,
  96                    regularizer=None,
  97                    trainable=True,
  98                    scope=None,
  99                    reuse=None):
 100   """Maps a sequence of symbols to a sequence of embeddings.
 101 
 102   Typical use case would be reusing embeddings between an encoder and decoder.
 103 
 104   Args:
 105     ids: `[batch_size, doc_length]` `Tensor` of type `int32` or `int64`
 106       with symbol ids.
 107     vocab_size: Integer number of symbols in vocabulary.
 108     embed_dim: Integer number of dimensions for embedding matrix.
 109     unique: If `True`, will first compute the unique set of indices, and then
 110          lookup each embedding once, repeating them in the output as needed.
 111     initializer: An initializer for the embeddings, if `None` default for
 112         current scope is used.
 113     regularizer: Optional regularizer for the embeddings.
 114     trainable: If `True` also add variables to the graph collection
 115       `GraphKeys.TRAINABLE_VARIABLES` (see `tf.Variable`).
 116     scope: Optional string specifying the variable scope for the op, required
 117         if `reuse=True`.
 118     reuse: If `True`, variables inside the op will be reused.
 119 
 120   Returns:
 121     `Tensor` of `[batch_size, doc_length, embed_dim]` with embedded sequences.
 122 
 123   Raises:
 124     ValueError: if `embed_dim` or `vocab_size` are not specified when
 125       `reuse` is `None` or `False`.
 126   """
 127   if not (reuse or (vocab_size and embed_dim)):
 128     raise ValueError('Must specify vocab size and embedding dimension when not '
 129                      'reusing. Got vocab_size=%s and embed_dim=%s' % (
 130                          vocab_size, embed_dim))
 131   with variable_scope.variable_scope(
 132       scope, 'EmbedSequence', [ids], reuse=reuse):
 133     shape = [vocab_size, embed_dim]
 134     if reuse and vocab_size is None or embed_dim is None:
 135       shape = None
 136     embeddings = variables.model_variable(
 137         'embeddings', shape=shape,
 138         initializer=initializer, regularizer=regularizer,
 139         trainable=trainable)
 140     if unique:
 141       return contrib_embedding_ops.embedding_lookup_unique(embeddings, ids)
 142     return embedding_ops.embedding_lookup(embeddings, ids)
