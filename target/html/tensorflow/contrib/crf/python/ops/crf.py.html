  15 """Module for constructing a linear-chain CRF.
  16 
  17 The following snippet is an example of a CRF layer on top of a batched sequence
  18 of unary scores (logits for every word). This example also decodes the most
  19 likely sequence at test time. There are two ways to do decoding. One
  20 is using crf_decode to do decoding in Tensorflow , and the other one is using
  21 viterbi_decode in Numpy.
  22 
  23 log_likelihood, transition_params = tf.contrib.crf.crf_log_likelihood(
  24     unary_scores, gold_tags, sequence_lengths)
  25 
  26 loss = tf.reduce_mean(-log_likelihood)
  27 train_op = tf.train.GradientDescentOptimizer(0.01).minimize(loss)
  28 
  29 # Decoding in Tensorflow.
  30 viterbi_sequence, viterbi_score = tf.contrib.crf.crf_decode(
  31     unary_scores, transition_params, sequence_lengths)
  32 
  33 tf_viterbi_sequence, tf_viterbi_score, _ = session.run(
  34     [viterbi_sequence, viterbi_score, train_op])
  35 
  36 # Decoding in Numpy.
  37 tf_unary_scores, tf_sequence_lengths, tf_transition_params, _ = session.run(
  38     [unary_scores, sequence_lengths, transition_params, train_op])
  39 for tf_unary_scores_, tf_sequence_length_ in zip(tf_unary_scores,
  40                                                  tf_sequence_lengths):
  41 # Remove padding.
  42 tf_unary_scores_ = tf_unary_scores_[:tf_sequence_length_]
  43 
  44 # Compute the highest score and its tag sequence.
  45 tf_viterbi_sequence, tf_viterbi_score = tf.contrib.crf.viterbi_decode(
  46     tf_unary_scores_, tf_transition_params)
  47 """

  73 def crf_sequence_score(inputs, tag_indices, sequence_lengths,
  74                        transition_params):
  75   """Computes the unnormalized score for a tag sequence.
  76 
  77   Args:
  78     inputs: A [batch_size, max_seq_len, num_tags] tensor of unary potentials
  79         to use as input to the CRF layer.
  80     tag_indices: A [batch_size, max_seq_len] matrix of tag indices for which we
  81         compute the unnormalized score.
  82     sequence_lengths: A [batch_size] vector of true sequence lengths.
  83     transition_params: A [num_tags, num_tags] transition matrix.
  84   Returns:
  85     sequence_scores: A [batch_size] vector of unnormalized sequence scores.
  86   """
  87   # If max_seq_len is 1, we skip the score calculation and simply gather the
  88   # unary potentials of the single tag.
  89   def _single_seq_fn():
  90     batch_size = array_ops.shape(inputs, out_type=tag_indices.dtype)[0]
  91     example_inds = array_ops.reshape(
  92         math_ops.range(batch_size, dtype=tag_indices.dtype), [-1, 1])
  93     sequence_scores = array_ops.gather_nd(
  94         array_ops.squeeze(inputs, [1]),
  95         array_ops.concat([example_inds, tag_indices], axis=1))
  96     sequence_scores = array_ops.where(math_ops.less_equal(sequence_lengths, 0),
  97                                       array_ops.zeros_like(sequence_scores),
  98                                       sequence_scores)
  99     return sequence_scores
 100 
 101   def _multi_seq_fn():
 102     # Compute the scores of the given tag sequence.
 103     unary_scores = crf_unary_score(tag_indices, sequence_lengths, inputs)
 104     binary_scores = crf_binary_score(tag_indices, sequence_lengths,
 105                                      transition_params)
 106     sequence_scores = unary_scores + binary_scores
 107     return sequence_scores
 108 
 109   return utils.smart_cond(
 110       pred=math_ops.equal(inputs.shape[1].value or array_ops.shape(inputs)[1],
 111                           1),
 112       true_fn=_single_seq_fn,
 113       false_fn=_multi_seq_fn)

 116 def crf_multitag_sequence_score(inputs, tag_bitmap, sequence_lengths,
 117                                 transition_params):
 118   """Computes the unnormalized score of all tag sequences matching tag_bitmap.
 119 
 120   tag_bitmap enables more than one tag to be considered correct at each time
 121   step. This is useful when an observed output at a given time step is
 122   consistent with more than one tag, and thus the log likelihood of that
 123   observation must take into account all possible consistent tags.
 124 
 125   Using one-hot vectors in tag_bitmap gives results identical to
 126   crf_sequence_score.
 127 
 128   Args:
 129     inputs: A [batch_size, max_seq_len, num_tags] tensor of unary potentials
 130         to use as input to the CRF layer.
 131     tag_bitmap: A [batch_size, max_seq_len, num_tags] boolean tensor
 132         representing all active tags at each index for which to calculate the
 133         unnormalized score.
 134     sequence_lengths: A [batch_size] vector of true sequence lengths.
 135     transition_params: A [num_tags, num_tags] transition matrix.
 136   Returns:
 137     sequence_scores: A [batch_size] vector of unnormalized sequence scores.
 138   """
 139 
 140   # If max_seq_len is 1, we skip the score calculation and simply gather the
 141   # unary potentials of all active tags.
 142   def _single_seq_fn():
 143     filtered_inputs = array_ops.where(
 144         tag_bitmap, inputs,
 145         array_ops.fill(array_ops.shape(inputs), float("-inf")))
 146     return math_ops.reduce_logsumexp(
 147         filtered_inputs, axis=[1, 2], keepdims=False)
 148 
 149   def _multi_seq_fn():
 150     # Compute the logsumexp of all scores of sequences matching the given tags.
 151     filtered_inputs = array_ops.where(
 152         tag_bitmap, inputs,
 153         array_ops.fill(array_ops.shape(inputs), float("-inf")))
 154     return crf_log_norm(
 155         inputs=filtered_inputs,
 156         sequence_lengths=sequence_lengths,
 157         transition_params=transition_params)
 158 
 159   return utils.smart_cond(
 160       pred=math_ops.equal(inputs.shape[1].value or array_ops.shape(inputs)[1],
 161                           1),
 162       true_fn=_single_seq_fn,
 163       false_fn=_multi_seq_fn)

 166 def crf_log_norm(inputs, sequence_lengths, transition_params):
 167   """Computes the normalization for a CRF.
 168 
 169   Args:
 170     inputs: A [batch_size, max_seq_len, num_tags] tensor of unary potentials
 171         to use as input to the CRF layer.
 172     sequence_lengths: A [batch_size] vector of true sequence lengths.
 173     transition_params: A [num_tags, num_tags] transition matrix.
 174   Returns:
 175     log_norm: A [batch_size] vector of normalizers for a CRF.
 176   """
 177   # Split up the first and rest of the inputs in preparation for the forward
 178   # algorithm.
 179   first_input = array_ops.slice(inputs, [0, 0, 0], [-1, 1, -1])
 180   first_input = array_ops.squeeze(first_input, [1])
 181 
 182   # If max_seq_len is 1, we skip the algorithm and simply reduce_logsumexp over
 183   # the "initial state" (the unary potentials).
 184   def _single_seq_fn():
 185     log_norm = math_ops.reduce_logsumexp(first_input, [1])
 186     # Mask `log_norm` of the sequences with length <= zero.
 187     log_norm = array_ops.where(math_ops.less_equal(sequence_lengths, 0),
 188                                array_ops.zeros_like(log_norm),
 189                                log_norm)
 190     return log_norm
 191 
 192   def _multi_seq_fn():
 193     """Forward computation of alpha values."""
 194     rest_of_input = array_ops.slice(inputs, [0, 1, 0], [-1, -1, -1])
 195 
 196     # Compute the alpha values in the forward algorithm in order to get the
 197     # partition function.
 198     forward_cell = CrfForwardRnnCell(transition_params)
 199     # Sequence length is not allowed to be less than zero.
 200     sequence_lengths_less_one = math_ops.maximum(
 201         constant_op.constant(0, dtype=sequence_lengths.dtype),
 202         sequence_lengths - 1)
 203     _, alphas = rnn.dynamic_rnn(
 204         cell=forward_cell,
 205         inputs=rest_of_input,
 206         sequence_length=sequence_lengths_less_one,
 207         initial_state=first_input,
 208         dtype=dtypes.float32)
 209     log_norm = math_ops.reduce_logsumexp(alphas, [1])
 210     # Mask `log_norm` of the sequences with length <= zero.
 211     log_norm = array_ops.where(math_ops.less_equal(sequence_lengths, 0),
 212                                array_ops.zeros_like(log_norm),
 213                                log_norm)
 214     return log_norm
 215 
 216   return utils.smart_cond(
 217       pred=math_ops.equal(inputs.shape[1].value or
 218                           array_ops.shape(inputs)[1], 1),
 219       true_fn=_single_seq_fn,
 220       false_fn=_multi_seq_fn)

 223 def crf_log_likelihood(inputs,
 224                        tag_indices,
 225                        sequence_lengths,
 226                        transition_params=None):
 227   """Computes the log-likelihood of tag sequences in a CRF.
 228 
 229   Args:
 230     inputs: A [batch_size, max_seq_len, num_tags] tensor of unary potentials
 231         to use as input to the CRF layer.
 232     tag_indices: A [batch_size, max_seq_len] matrix of tag indices for which we
 233         compute the log-likelihood.
 234     sequence_lengths: A [batch_size] vector of true sequence lengths.
 235     transition_params: A [num_tags, num_tags] transition matrix, if available.
 236   Returns:
 237     log_likelihood: A [batch_size] `Tensor` containing the log-likelihood of
 238       each example, given the sequence of tag indices.
 239     transition_params: A [num_tags, num_tags] transition matrix. This is either
 240         provided by the caller or created in this function.
 241   """
 242   # Get shape information.
 243   num_tags = inputs.get_shape()[2].value
 244 
 245   # Get the transition matrix if not provided.
 246   if transition_params is None:
 247     transition_params = vs.get_variable("transitions", [num_tags, num_tags])
 248 
 249   sequence_scores = crf_sequence_score(inputs, tag_indices, sequence_lengths,
 250                                        transition_params)
 251   log_norm = crf_log_norm(inputs, sequence_lengths, transition_params)
 252 
 253   # Normalize the scores to get the log-likelihood per example.
 254   log_likelihood = sequence_scores - log_norm
 255   return log_likelihood, transition_params

 258 def crf_unary_score(tag_indices, sequence_lengths, inputs):
 259   """Computes the unary scores of tag sequences.
 260 
 261   Args:
 262     tag_indices: A [batch_size, max_seq_len] matrix of tag indices.
 263     sequence_lengths: A [batch_size] vector of true sequence lengths.
 264     inputs: A [batch_size, max_seq_len, num_tags] tensor of unary potentials.
 265   Returns:
 266     unary_scores: A [batch_size] vector of unary scores.
 267   """
 268   batch_size = array_ops.shape(inputs)[0]
 269   max_seq_len = array_ops.shape(inputs)[1]
 270   num_tags = array_ops.shape(inputs)[2]
 271 
 272   flattened_inputs = array_ops.reshape(inputs, [-1])
 273 
 274   offsets = array_ops.expand_dims(
 275       math_ops.range(batch_size) * max_seq_len * num_tags, 1)
 276   offsets += array_ops.expand_dims(math_ops.range(max_seq_len) * num_tags, 0)
 277   # Use int32 or int64 based on tag_indices' dtype.
 278   if tag_indices.dtype == dtypes.int64:
 279     offsets = math_ops.to_int64(offsets)
 280   flattened_tag_indices = array_ops.reshape(offsets + tag_indices, [-1])
 281 
 282   unary_scores = array_ops.reshape(
 283       array_ops.gather(flattened_inputs, flattened_tag_indices),
 284       [batch_size, max_seq_len])
 285 
 286   masks = array_ops.sequence_mask(sequence_lengths,
 287                                   maxlen=array_ops.shape(tag_indices)[1],
 288                                   dtype=dtypes.float32)
 289 
 290   unary_scores = math_ops.reduce_sum(unary_scores * masks, 1)
 291   return unary_scores

 294 def crf_binary_score(tag_indices, sequence_lengths, transition_params):
 295   """Computes the binary scores of tag sequences.
 296 
 297   Args:
 298     tag_indices: A [batch_size, max_seq_len] matrix of tag indices.
 299     sequence_lengths: A [batch_size] vector of true sequence lengths.
 300     transition_params: A [num_tags, num_tags] matrix of binary potentials.
 301   Returns:
 302     binary_scores: A [batch_size] vector of binary scores.
 303   """
 304   # Get shape information.
 305   num_tags = transition_params.get_shape()[0]
 306   num_transitions = array_ops.shape(tag_indices)[1] - 1
 307 
 308   # Truncate by one on each side of the sequence to get the start and end
 309   # indices of each transition.
 310   start_tag_indices = array_ops.slice(tag_indices, [0, 0],
 311                                       [-1, num_transitions])
 312   end_tag_indices = array_ops.slice(tag_indices, [0, 1], [-1, num_transitions])
 313 
 314   # Encode the indices in a flattened representation.
 315   flattened_transition_indices = start_tag_indices * num_tags + end_tag_indices
 316   flattened_transition_params = array_ops.reshape(transition_params, [-1])
 317 
 318   # Get the binary scores based on the flattened representation.
 319   binary_scores = array_ops.gather(flattened_transition_params,
 320                                    flattened_transition_indices)
 321 
 322   masks = array_ops.sequence_mask(sequence_lengths,
 323                                   maxlen=array_ops.shape(tag_indices)[1],
 324                                   dtype=dtypes.float32)
 325   truncated_masks = array_ops.slice(masks, [0, 1], [-1, -1])
 326   binary_scores = math_ops.reduce_sum(binary_scores * truncated_masks, 1)
 327   return binary_scores

 330 class CrfForwardRnnCell(rnn_cell.RNNCell):
 331   """Computes the alpha values in a linear-chain CRF.
 332 
 333   See http://www.cs.columbia.edu/~mcollins/fb.pdf for reference.
 334   """
 335 
 336   def __init__(self, transition_params):
 337     """Initialize the CrfForwardRnnCell.
 338 
 339     Args:
 340       transition_params: A [num_tags, num_tags] matrix of binary potentials.
 341           This matrix is expanded into a [1, num_tags, num_tags] in preparation
 342           for the broadcast summation occurring within the cell.
 343     """
 344     self._transition_params = array_ops.expand_dims(transition_params, 0)
 345     self._num_tags = transition_params.get_shape()[0].value
 346 
 347   @property
 348   def state_size(self):
 349     return self._num_tags
 350 
 351   @property
 352   def output_size(self):
 353     return self._num_tags
 354 
 355   def __call__(self, inputs, state, scope=None):
 356     """Build the CrfForwardRnnCell.
 357 
 358     Args:
 359       inputs: A [batch_size, num_tags] matrix of unary potentials.
 360       state: A [batch_size, num_tags] matrix containing the previous alpha
 361           values.
 362       scope: Unused variable scope of this cell.
 363 
 364     Returns:
 365       new_alphas, new_alphas: A pair of [batch_size, num_tags] matrices
 366           values containing the new alpha values.
 367     """
 368     state = array_ops.expand_dims(state, 2)
 369 
 370     # This addition op broadcasts self._transitions_params along the zeroth
 371     # dimension and state along the second dimension. This performs the
 372     # multiplication of previous alpha values and the current binary potentials
 373     # in log space.
 374     transition_scores = state + self._transition_params
 375     new_alphas = inputs + math_ops.reduce_logsumexp(transition_scores, [1])
 376 
 377     # Both the state and the output of this RNN cell contain the alphas values.
 378     # The output value is currently unused and simply satisfies the RNN API.
 379     # This could be useful in the future if we need to compute marginal
 380     # probabilities, which would require the accumulated alpha values at every
 381     # time step.
 382     return new_alphas, new_alphas

 385 def viterbi_decode(score, transition_params):
 386   """Decode the highest scoring sequence of tags outside of TensorFlow.
 387 
 388   This should only be used at test time.
 389 
 390   Args:
 391     score: A [seq_len, num_tags] matrix of unary potentials.
 392     transition_params: A [num_tags, num_tags] matrix of binary potentials.
 393 
 394   Returns:
 395     viterbi: A [seq_len] list of integers containing the highest scoring tag
 396         indices.
 397     viterbi_score: A float containing the score for the Viterbi sequence.
 398   """
 399   trellis = np.zeros_like(score)
 400   backpointers = np.zeros_like(score, dtype=np.int32)
 401   trellis[0] = score[0]
 402 
 403   for t in range(1, score.shape[0]):
 404     v = np.expand_dims(trellis[t - 1], 1) + transition_params
 405     trellis[t] = score[t] + np.max(v, 0)
 406     backpointers[t] = np.argmax(v, 0)
 407 
 408   viterbi = [np.argmax(trellis[-1])]
 409   for bp in reversed(backpointers[1:]):
 410     viterbi.append(bp[viterbi[-1]])
 411   viterbi.reverse()
 412 
 413   viterbi_score = np.max(trellis[-1])
 414   return viterbi, viterbi_score

 417 class CrfDecodeForwardRnnCell(rnn_cell.RNNCell):
 418   """Computes the forward decoding in a linear-chain CRF.
 419   """
 420 
 421   def __init__(self, transition_params):
 422     """Initialize the CrfDecodeForwardRnnCell.
 423 
 424     Args:
 425       transition_params: A [num_tags, num_tags] matrix of binary
 426         potentials. This matrix is expanded into a
 427         [1, num_tags, num_tags] in preparation for the broadcast
 428         summation occurring within the cell.
 429     """
 430     self._transition_params = array_ops.expand_dims(transition_params, 0)
 431     self._num_tags = transition_params.get_shape()[0].value
 432 
 433   @property
 434   def state_size(self):
 435     return self._num_tags
 436 
 437   @property
 438   def output_size(self):
 439     return self._num_tags
 440 
 441   def __call__(self, inputs, state, scope=None):
 442     """Build the CrfDecodeForwardRnnCell.
 443 
 444     Args:
 445       inputs: A [batch_size, num_tags] matrix of unary potentials.
 446       state: A [batch_size, num_tags] matrix containing the previous step's
 447             score values.
 448       scope: Unused variable scope of this cell.
 449 
 450     Returns:
 451       backpointers: A [batch_size, num_tags] matrix of backpointers.
 452       new_state: A [batch_size, num_tags] matrix of new score values.
 453     """
 454     # For simplicity, in shape comments, denote:
 455     # 'batch_size' by 'B', 'max_seq_len' by 'T' , 'num_tags' by 'O' (output).
 456     state = array_ops.expand_dims(state, 2)                         # [B, O, 1]
 457 
 458     # This addition op broadcasts self._transitions_params along the zeroth
 459     # dimension and state along the second dimension.
 460     # [B, O, 1] + [1, O, O] -> [B, O, O]
 461     transition_scores = state + self._transition_params             # [B, O, O]
 462     new_state = inputs + math_ops.reduce_max(transition_scores, [1])  # [B, O]
 463     backpointers = math_ops.argmax(transition_scores, 1)
 464     backpointers = math_ops.cast(backpointers, dtype=dtypes.int32)    # [B, O]
 465     return backpointers, new_state

 468 class CrfDecodeBackwardRnnCell(rnn_cell.RNNCell):
 469   """Computes backward decoding in a linear-chain CRF.
 470   """
 471 
 472   def __init__(self, num_tags):
 473     """Initialize the CrfDecodeBackwardRnnCell.
 474 
 475     Args:
 476       num_tags: An integer. The number of tags.
 477     """
 478     self._num_tags = num_tags
 479 
 480   @property
 481   def state_size(self):
 482     return 1
 483 
 484   @property
 485   def output_size(self):
 486     return 1
 487 
 488   def __call__(self, inputs, state, scope=None):
 489     """Build the CrfDecodeBackwardRnnCell.
 490 
 491     Args:
 492       inputs: A [batch_size, num_tags] matrix of
 493             backpointer of next step (in time order).
 494       state: A [batch_size, 1] matrix of tag index of next step.
 495       scope: Unused variable scope of this cell.
 496 
 497     Returns:
 498       new_tags, new_tags: A pair of [batch_size, num_tags]
 499         tensors containing the new tag indices.
 500     """
 501     state = array_ops.squeeze(state, axis=[1])                # [B]
 502     batch_size = array_ops.shape(inputs)[0]
 503     b_indices = math_ops.range(batch_size)                    # [B]
 504     indices = array_ops.stack([b_indices, state], axis=1)     # [B, 2]
 505     new_tags = array_ops.expand_dims(
 506         gen_array_ops.gather_nd(inputs, indices),             # [B]
 507         axis=-1)                                              # [B, 1]
 508 
 509     return new_tags, new_tags

 512 def crf_decode(potentials, transition_params, sequence_length):
 513   """Decode the highest scoring sequence of tags in TensorFlow.
 514 
 515   This is a function for tensor.
 516 
 517   Args:
 518     potentials: A [batch_size, max_seq_len, num_tags] tensor of
 519               unary potentials.
 520     transition_params: A [num_tags, num_tags] matrix of
 521               binary potentials.
 522     sequence_length: A [batch_size] vector of true sequence lengths.
 523 
 524   Returns:
 525     decode_tags: A [batch_size, max_seq_len] matrix, with dtype `tf.int32`.
 526                 Contains the highest scoring tag indices.
 527     best_score: A [batch_size] vector, containing the score of `decode_tags`.
 528   """
 529   # If max_seq_len is 1, we skip the algorithm and simply return the argmax tag
 530   # and the max activation.
 531   def _single_seq_fn():
 532     squeezed_potentials = array_ops.squeeze(potentials, [1])
 533     decode_tags = array_ops.expand_dims(
 534         math_ops.argmax(squeezed_potentials, axis=1), 1)
 535     best_score = math_ops.reduce_max(squeezed_potentials, axis=1)
 536     return math_ops.cast(decode_tags, dtype=dtypes.int32), best_score
 537 
 538   def _multi_seq_fn():
 539     """Decoding of highest scoring sequence."""
 540 
 541     # For simplicity, in shape comments, denote:
 542     # 'batch_size' by 'B', 'max_seq_len' by 'T' , 'num_tags' by 'O' (output).
 543     num_tags = potentials.get_shape()[2].value
 544 
 545     # Computes forward decoding. Get last score and backpointers.
 546     crf_fwd_cell = CrfDecodeForwardRnnCell(transition_params)
 547     initial_state = array_ops.slice(potentials, [0, 0, 0], [-1, 1, -1])
 548     initial_state = array_ops.squeeze(initial_state, axis=[1])  # [B, O]
 549     inputs = array_ops.slice(potentials, [0, 1, 0], [-1, -1, -1])  # [B, T-1, O]
 550     # Sequence length is not allowed to be less than zero.
 551     sequence_length_less_one = math_ops.maximum(
 552         constant_op.constant(0, dtype=sequence_length.dtype),
 553         sequence_length - 1)
 554     backpointers, last_score = rnn.dynamic_rnn(  # [B, T - 1, O], [B, O]
 555         crf_fwd_cell,
 556         inputs=inputs,
