  32 class BlockCacheParams(object):
  33   """BlockCacheParams is a struct used for configuring the GCS Block Cache."""

  35   def __init__(self, block_size=None, max_bytes=None, max_staleness=None):
  36     self._block_size = block_size or 128 * 1024 * 1024
  37     self._max_bytes = max_bytes or 2 * self._block_size
  38     self._max_staleness = max_staleness or 0
  39 
  40   @property
  41   def block_size(self):
  42     return self._block_size
  43 
  44   @property
  45   def max_bytes(self):
  46     return self._max_bytes
  47 
  48   @property
  49   def max_staleness(self):
  50     return self._max_staleness

  53 # @tf_export('contrib.cloud.ConfigureGcsHook')
  54 class ConfigureGcsHook(training.SessionRunHook):
  55   """ConfigureGcsHook configures GCS when used with Estimator/TPUEstimator.
  56 
  57   Warning: GCS `credentials` may be transmitted over the network unencrypted.
  58   Please ensure that the network is trusted before using this function. For
  59   users running code entirely within Google Cloud, your data is protected by
  60   encryption in between data centers. For more information, please take a look
  61   at https://cloud.google.com/security/encryption-in-transit/.
  62 
  63   Example:
  64 
  65   ```
  66   sess = tf.Session()
  67   refresh_token = raw_input("Refresh token: ")
  68   client_secret = raw_input("Client secret: ")
  69   client_id = "<REDACTED>"
  70   creds = {
  71       "client_id": client_id,
  72       "refresh_token": refresh_token,
  73       "client_secret": client_secret,
  74       "type": "authorized_user",
  75   }
  76   tf.contrib.cloud.configure_gcs(sess, credentials=creds)
  77   ```
  78 
  79   """
  80 
  81   def _verify_dictionary(self, creds_dict):
  82     if 'refresh_token' in creds_dict or 'private_key' in creds_dict:
  83       return True
  84     return False
  85 
  86   def __init__(self, credentials=None, block_cache=None):
  87     """Constructs a ConfigureGcsHook.
  88 
  89     Args:
  90       credentials: A json-formatted string.
  91       block_cache: A `BlockCacheParams`
  92 
  93     Raises:
  94       ValueError: If credentials is improperly formatted or block_cache is not a
  95         BlockCacheParams.
  96     """
  97     if credentials is not None:
  98       if isinstance(credentials, str):
  99         try:
 100           data = json.loads(credentials)
 101         except ValueError as e:
 102           raise ValueError('credentials was not a well formed JSON string.', e)
 103         if not self._verify_dictionary(data):
 104           raise ValueError(
 105               'credentials has neither a "refresh_token" nor a "private_key" '
 106               'field.')
 107       elif isinstance(credentials, dict):
 108         if not self._verify_dictionary(credentials):
 109           raise ValueError('credentials has neither a "refresh_token" nor a '
 110                            '"private_key" field.')
 111         credentials = json.dumps(credentials)
 112       else:
 113         raise ValueError('credentials is of an unknown type')
 114 
 115     self._credentials = credentials
 116 
 117     if block_cache and not isinstance(block_cache, BlockCacheParams):
 118       raise ValueError('block_cache must be an instance of BlockCacheParams.')
 119     self._block_cache = block_cache
 120 
 121   def begin(self):
 122     if self._credentials:
 123       self._credentials_placeholder = array_ops.placeholder(dtypes.string)
 124       self._credentials_op = gen_gcs_config_ops.gcs_configure_credentials(
 125           self._credentials_placeholder)
 126     else:
 127       self._credentials_op = None
 128 
 129     if self._block_cache:
 130       self._block_cache_op = gen_gcs_config_ops.gcs_configure_block_cache(
 131           max_cache_size=self._block_cache.max_bytes,
 132           block_size=self._block_cache.block_size,
 133           max_staleness=self._block_cache.max_staleness)
 134     else:
 135       self._block_cache_op = None
 136 
 137   def after_create_session(self, session, coord):
 138     del coord
 139     if self._credentials_op:
 140       session.run(
 141           self._credentials_op,
 142           feed_dict={self._credentials_placeholder: self._credentials})
 143     if self._block_cache_op:
 144       session.run(self._block_cache_op)

 147 def configure_gcs(session, credentials=None, block_cache=None, device=None):
 148   """Configures the GCS file system for a given a session.
 149 
 150   Warning: GCS `credentials` may be transmitted over the network unencrypted.
 151   Please ensure that the network is trusted before using this function. For
 152   users running code entirely within Google Cloud, your data is protected by
 153   encryption in between data centers. For more information, please take a look
 154   at https://cloud.google.com/security/encryption-in-transit/.
 155 
 156   Args:
 157     session: A `tf.Session` session that should be used to configure the GCS
 158       file system.
 159     credentials: [Optional.] A JSON string
 160     block_cache: [Optional.] A BlockCacheParams to configure the block cache .
 161     device: [Optional.] The device to place the configure ops.
 162   """
 163 
 164   def configure(credentials, block_cache):
 165     """Helper function to actually configure GCS."""
 166     if credentials:
 167       if isinstance(credentials, dict):
 168         credentials = json.dumps(credentials)
 169       placeholder = array_ops.placeholder(dtypes.string)
 170       op = gen_gcs_config_ops.gcs_configure_credentials(placeholder)
 171       session.run(op, feed_dict={placeholder: credentials})
 172     if block_cache:
 173       op = gen_gcs_config_ops.gcs_configure_block_cache(
 174           max_cache_size=block_cache.max_bytes,
 175           block_size=block_cache.block_size,
 176           max_staleness=block_cache.max_staleness)
 177       session.run(op)
 178 
 179   if device:
 180     with ops.device(device):
 181       return configure(credentials, block_cache)
 182   return configure(credentials, block_cache)

 185 def configure_colab_session(session):
 186   """ConfigureColabSession configures the GCS file system in Colab.
 187 
 188   Args:
 189     session: A `tf.Session` session.
 190   """
 191   # Read from the application default credentials (adc).
 192   adc_filename = os.environ.get(
 193       'GOOGLE_APPLICATION_CREDENTIALS', '/content/adc.json')
 194   with open(adc_filename) as f:
 195     data = json.load(f)
 196   configure_gcs(session, credentials=data)
