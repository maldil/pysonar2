  39 def weight_mask_variable(var, scope):
  40   """Create a mask for the weights.
  41 
  42   This function adds a variable 'mask' to the graph.
  43 
  44   Args:
  45     var: the weight variable that needs to be masked
  46     scope: The variable scope of the variable var
  47 
  48   Returns:
  49     the mask variable of the same size and shape as var, initialized to all 1s.
  50   """
  51   with variable_scope.variable_scope(scope):
  52     mask = variable_scope.get_variable(
  53         'mask',
  54         var.get_shape(),
  55         initializer=init_ops.ones_initializer(),
  56         trainable=False,
  57         dtype=var.dtype)
  58   return mask

  83 def kronecker_product(mat1, mat2):
  84   """Computes the Kronecker product of two matrices mat1 and mat2.
  85 
  86   Args:
  87     mat1: A matrix of size m x n
  88     mat2: A matrix of size p x q
  89   Returns:
  90     Kronecker product of matrices mat1 and mat2 of size mp x nq
  91   """
  92 
  93   m1, n1 = mat1.get_shape().as_list()
  94   mat1_rsh = array_ops.reshape(mat1, [m1, 1, n1, 1])
  95   m2, n2 = mat2.get_shape().as_list()
  96   mat2_rsh = array_ops.reshape(mat2, [1, m2, 1, n2])
  97   return array_ops.reshape(mat1_rsh * mat2_rsh, [m1 * m2, n1 * n2])

 100 def expand_tensor(tensor, block_dims):
 101   """Expands a 2D tensor by replicating the tensor values.
 102 
 103   This is equivalent to the kronecker product of the tensor and a matrix of
 104   ones of size block_dims.
 105 
 106   Example:
 107 
 108   tensor = [[1,2]
 109             [3,4]]
 110   block_dims = [2,2]
 111 
 112   result = [[1 1 2 2]
 113             [1 1 2 2]
 114             [3 3 4 4]
 115             [3 3 4 4]]
 116 
 117   Args:
 118     tensor: A 2D tensor that needs to be expanded.
 119     block_dims: List of integers specifying the expansion factor.
 120 
 121   Returns:
 122     The expanded tensor
 123 
 124   Raises:
 125     ValueError: if tensor is not rank-2 or block_dims is does not have 2
 126     elements.
 127   """
 128   if tensor.get_shape().ndims != 2:
 129     raise ValueError('Input tensor must be rank 2')
 130 
 131   if len(block_dims) != 2:
 132     raise ValueError('block_dims must have 2 elements')
 133 
 134   block_height, block_width = block_dims
 135 
 136   def _tile_rows(tensor, multiple):
 137     """Create a new tensor by tiling the tensor along rows."""
 138     return array_ops.tile(tensor, [multiple, 1])
 139 
 140   def _generate_indices(num_rows, block_dim):
 141     indices = np.zeros(shape=[num_rows * block_dim, 1], dtype=np.int32)
 142     for k in range(block_dim):
 143       for r in range(num_rows):
 144         indices[k * num_rows + r] = r * block_dim + k
 145     return indices
 146 
 147   def _replicate_rows(tensor, multiple):
 148     tensor_shape = tensor.shape.as_list()
 149     expanded_shape = [tensor_shape[0] * multiple, tensor_shape[1]]
 150     indices = constant_op.constant(_generate_indices(tensor_shape[0], multiple))
 151     return array_ops.scatter_nd(indices, _tile_rows(tensor, multiple),
 152                                 expanded_shape)
 153 
 154   expanded_tensor = tensor
 155 
 156   # Expand rows by factor block_height.
 157   if block_height > 1:
 158     expanded_tensor = _replicate_rows(tensor, block_height)
 159 
 160   # Transpose and expand by factor block_width. Transpose the result.
 161   if block_width > 1:
 162     expanded_tensor = array_ops.transpose(
 163         _replicate_rows(array_ops.transpose(expanded_tensor), block_width))
 164 
 165   return expanded_tensor

 168 def _histogram(values, value_range, nbins=100, dtype=dtypes.int32, name=None):
 169   """Return histogram of values.
 170 
 171   Given the tensor `values`, this operation returns a rank 1 histogram counting
 172   the number of entries in `values` that fell into every bin.  The bins are
 173   equal width and determined by the arguments `value_range` and `nbins`.
 174 
 175   Args:
 176     values:  Numeric `Tensor`.
 177     value_range:  Shape [2] `Tensor` of same `dtype` as `values`.
 178       values <= value_range[0] will be mapped to hist[0],
 179       values >= value_range[1] will be mapped to hist[-1].
 180     nbins:  Scalar `int32 Tensor`.  Number of histogram bins.
 181     dtype:  dtype for returned histogram.
 182     name:  A name for this operation (defaults to 'histogram').
 183 
 184   Returns:
 185     A 1-D `Tensor` holding histogram of values.
 186 
 187   """
 188   with ops.name_scope(name, 'histogram', [values, value_range, nbins]) as scope:
 189     values = ops.convert_to_tensor(values, name='values')
 190     values = array_ops.reshape(values, [-1])
 191     value_range = ops.convert_to_tensor(value_range, name='value_range')
 192     nbins_float = np.float32(nbins)
 193 
 194     # Map tensor values that fall within value_range to [0, 1].
 195     scaled_values = math_ops.truediv(
 196         values - value_range[0],
 197         value_range[1] - value_range[0],
 198         name='scaled_values')
 199 
 200     # map tensor values within the open interval value_range to {0,.., nbins-1},
 201     # values outside the open interval will be zero or less, or nbins or more.
 202     indices = math_ops.floor(nbins_float * scaled_values, name='indices')
 203 
 204     # Clip edge cases (e.g. value = value_range[1]) or "outliers."
 205     indices = math_ops.cast(
 206         clip_ops.clip_by_value(indices, 0, nbins_float - 1), dtypes.int32)
 207 
 208     return math_ops.unsorted_segment_sum(
 209         array_ops.ones_like(indices, dtype=dtype), indices, nbins, name=scope)

 212 def compute_cdf_from_histogram(values, value_range, **kwargs):
 213   """Returns the normalized cumulative distribution of the given values tensor.
 214 
 215   Computes the histogram and uses tf.cumsum to arrive at cdf
 216 
 217   Args:
 218     values:  Numeric `Tensor`.
 219     value_range:  Shape [2] `Tensor` of same `dtype` as `values`.
 220     **kwargs: keyword arguments: nbins, name
 221 
 222   Returns:
 223     A 1-D `Tensor` holding normalized cdf of values.
 224 
 225   """
 226   nbins = kwargs.get('nbins', _NBINS)
 227   name = kwargs.get('name', None)
 228   with ops.name_scope(name, 'cdf', [values, value_range, nbins]):
 229     histogram = _histogram(
 230         values, value_range, dtype=dtypes.float32, nbins=nbins)
 231     cdf = math_ops.cumsum(histogram)
 232     return math_ops.div(cdf, math_ops.reduce_max(cdf))

 235 def compute_cdf(values, value_range, **kwargs):
 236   """Returns the normalized cumulative distribution of the given values tensor.
 237 
 238   Uses tf.while_loop to directly compute the cdf of the values.
 239 
 240   Args:
 241     values:  Numeric `Tensor`.
 242     value_range:  Shape [2] `Tensor` of same `dtype` as `values`
 243     **kwargs: keyword arguments: nbins, name
 244 
 245   Returns:
 246     A 1-D `Tensor` holding normalized cdf of values.
 247 
 248   """
 249   nbins = kwargs.get('nbins', _NBINS)
 250   name = kwargs.get('name', None)
 251   with ops.name_scope(name, 'cdf', [values, value_range, nbins]):
 252     values = ops.convert_to_tensor(values, name='values')
 253     value_range = ops.convert_to_tensor(value_range, name='value_range')
 254     nbins_float = np.float32(nbins)
 255 
 256     # Map tensor values that fall within value_range to [0, 1].
 257     scaled_values = math_ops.truediv(
 258         values - value_range[0],
 259         value_range[1] - value_range[0],
 260         name='scaled_values')
 261 
 262     # map tensor values within the open interval value_range to {0,.., nbins-1},
 263     # values outside the open interval will be zero or less, or nbins or more.
 264     indices = math_ops.floor(nbins_float * scaled_values, name='indices')
 265 
 266     # Clip edge cases (e.g. value = value_range[1]) or "outliers."
 267     indices = math_ops.cast(
 268         clip_ops.clip_by_value(indices, 0, nbins_float - 1), dtypes.int32)
 269 
 270     cdf = array_ops.zeros(nbins)
 271     i = constant_op.constant(0)
 272 
 273     def loop_cond(loop_count, _):
 274       return math_ops.less(loop_count, nbins)
 275 
 276     def loop_body(loop_count, cdf):
 277       temp = math_ops.reduce_sum(
 278           math_ops.cast(
 279               math_ops.less_equal(indices, loop_count), dtypes.float32))
 280       cdf = math_ops.add(
 281           cdf,
 282           array_ops.one_hot(
 283               loop_count, depth=nbins, on_value=temp, off_value=0.0))
 284       return [loop_count + 1, cdf]
 285 
 286     _, cdf = control_flow_ops.while_loop(
 287         loop_cond, loop_body, [i, cdf], maximum_iterations=nbins)
 288 
 289     return math_ops.div(cdf, math_ops.reduce_max(cdf))

 292 def factorized_pool(input_tensor,
 293                     window_shape,
 294                     pooling_type,
 295                     strides,
 296                     padding,
 297                     name=None):
 298   """Performs m x n pooling through a combination of 1xm and 1xn pooling.
 299 
 300   Args:
 301     input_tensor: Input tensor. Must be rank 2
 302     window_shape: Pooling window shape
 303     pooling_type: Either 'MAX' or 'AVG'
 304     strides: The stride of the pooling window
 305     padding: 'SAME' or 'VALID'.
 306     name: Name of the op
 307 
 308   Returns:
 309     A rank 2 tensor containing the pooled output
 310 
 311   Raises:
 312     ValueError: if the input tensor is not rank 2
 313   """
 314   if input_tensor.get_shape().ndims != 2:
 315     raise ValueError('factorized_pool() accepts tensors of rank 2 only')
 316 
 317   [height, width] = input_tensor.get_shape()
 318   with ops.name_scope(name, 'factorized_pool'):
 319     input_tensor_aligned = array_ops.reshape(
 320         input_tensor, [1, 1, height, width],
 321         name=input_tensor.op.name + '_aligned')
 322 
 323     height_pooling = nn_ops.pool(
 324         input_tensor_aligned,
 325         window_shape=[1, window_shape[0]],
 326         pooling_type=pooling_type,
 327         strides=[1, strides[0]],
 328         padding=padding)
 329     swap_height_width = array_ops.transpose(height_pooling, perm=[0, 1, 3, 2])
 330 
 331     width_pooling = nn_ops.pool(
 332         swap_height_width,
 333         window_shape=[1, window_shape[1]],
 334         pooling_type=pooling_type,
 335         strides=[1, strides[1]],
 336         padding=padding)
 337 
 338   return array_ops.squeeze(
 339       array_ops.transpose(width_pooling, perm=[0, 1, 3, 2]))

 342 def determine_partitioned_axis(partitioned_variable):
 343   partitioned_axis = 0
 344   concatenated_variable_shape = partitioned_variable.get_shape()
 345   for partition in partitioned_variable:
 346     partition_shape = partition.get_shape()
 347     maybe_partitioned_axis = np.less(partition_shape,
 348                                      concatenated_variable_shape)
 349     # Sanity check: make sure number of partitioned axis == 1
 350     if np.count_nonzero(maybe_partitioned_axis) != 1:
 351       raise ValueError('Number of partitioned axes %s not equal to 1' %
 352                        np.count_nonzero(maybe_partitioned_axis))
 353     partitioned_axis = np.where(maybe_partitioned_axis)[0][0]
 354   return partitioned_axis

 361 def partitioned_variable_assign(partitioned_var, new_value):
 362   """Assign op for partitioned variables.
 363 
 364   Args:
 365     partitioned_var: A partitioned tensorflow variable
 366     new_value: Value to be assigned to the variable var
 367 
 368   Returns:
 369     A tensorflow op that groups the assign ops for each of the variable slices
 370   """
 371   # Determine which axis was used to partition the variable. Currently
 372   # tensorflow allows partitioning variable only along 1 axis.
 373   axis = 0 if len(partitioned_var) == 1 else determine_partitioned_axis(
 374       partitioned_var)
 375 
 376   partition_sizes = np.array(
 377       [partition.get_shape()[axis] for partition in partitioned_var])
 378   new_partitioned_values = array_ops.split(
 379       new_value,
 380       ops.convert_to_tensor(partition_sizes, dtype=dtypes.int32),
 381       axis=axis)
 382   op_list = []
 383   for partition in partitioned_var:
 384     op_list.append(
 385         variable_assign(partition, new_partitioned_values[len(op_list)]))
 386   return control_flow_ops.group(
 387       *op_list, name=partitioned_var.name + '_group_assign')
