  41 class _MaskedConv(base.Layer):
  42   """Abstract nD convolution layer (private, used as implementation base).
  43 
  44   This layer creates a convolution kernel that is convolved
  45   (actually cross-correlated) with the layer input to produce a tensor of
  46   outputs. The weight tensor of this layer is masked.
  47   If `use_bias` is True (and a `bias_initializer` is provided),
  48   a bias vector is created and added to the outputs. Finally, if
  49   `activation` is not `None`, it is applied to the outputs as well.
  50 
  51   Arguments:
  52     rank: An integer, the rank of the convolution, e.g. "2" for 2D convolution.
  53     filters: Integer, the dimensionality of the output space (i.e. the number
  54       of filters in the convolution).
  55     kernel_size: An integer or tuple/list of n integers, specifying the
  56       length of the convolution window.
  57     strides: An integer or tuple/list of n integers,
  58       specifying the stride length of the convolution.
  59       Specifying any stride value != 1 is incompatible with specifying
  60       any `dilation_rate` value != 1.
  61     padding: One of `"valid"` or `"same"` (case-insensitive).
  62     data_format: A string, one of `channels_last` (default) or `channels_first`.
  63       The ordering of the dimensions in the inputs.
  64       `channels_last` corresponds to inputs with shape
  65       `(batch, ..., channels)` while `channels_first` corresponds to
  66       inputs with shape `(batch, channels, ...)`.
  67     dilation_rate: An integer or tuple/list of n integers, specifying
  68       the dilation rate to use for dilated convolution.
  69       Currently, specifying any `dilation_rate` value != 1 is
  70       incompatible with specifying any `strides` value != 1.
  71     activation: Activation function. Set it to None to maintain a
  72       linear activation.
  73     use_bias: Boolean, whether the layer uses a bias.
  74     kernel_initializer: An initializer for the convolution kernel.
  75     bias_initializer: An initializer for the bias vector. If None, the default
  76       initializer will be used.
  77     kernel_regularizer: Optional regularizer for the convolution kernel.
  78     bias_regularizer: Optional regularizer for the bias vector.
  79     activity_regularizer: Regularizer function for the output.
  80     trainable: Boolean, if `True` also add variables to the graph collection
  81       `GraphKeys.TRAINABLE_VARIABLES` (see `tf.Variable`).
  82     name: A string, the name of the layer.
  83   """
  84 
  85   def __init__(self,
  86                rank,
  87                filters,
  88                kernel_size,
  89                strides=1,
  90                padding='valid',
  91                data_format='channels_last',
  92                dilation_rate=1,
  93                activation=None,
  94                use_bias=True,
  95                kernel_initializer=None,
  96                bias_initializer=init_ops.zeros_initializer(),
  97                kernel_regularizer=None,
  98                bias_regularizer=None,
  99                activity_regularizer=None,
 100                trainable=True,
 101                name=None,
 102                **kwargs):
 103     super(_MaskedConv, self).__init__(
 104         trainable=trainable,
 105         name=name,
 106         activity_regularizer=activity_regularizer,
 107         **kwargs)
 108     self.rank = rank
 109     self.filters = filters
 110     self.kernel_size = utils.normalize_tuple(kernel_size, rank, 'kernel_size')
 111     self.strides = utils.normalize_tuple(strides, rank, 'strides')
 112     self.padding = utils.normalize_padding(padding)
 113     self.data_format = utils.normalize_data_format(data_format)
 114     self.dilation_rate = utils.normalize_tuple(dilation_rate, rank,
 115                                                'dilation_rate')
 116     self.activation = activation
 117     self.use_bias = use_bias
 118     self.kernel_initializer = kernel_initializer
 119     self.bias_initializer = bias_initializer
 120     self.kernel_regularizer = kernel_regularizer
 121     self.bias_regularizer = bias_regularizer
 122     self.input_spec = base.InputSpec(ndim=self.rank + 2)
 123 
 124   def build(self, input_shape):
 125     input_shape = tensor_shape.TensorShape(input_shape)
 126     channel_axis = 1 if self.data_format == 'channels_first' else -1
 127     if input_shape[channel_axis].value is None:
 128       raise ValueError('The channel dimension of the inputs '
 129                        'should be defined. Found `None`.')
 130     input_dim = input_shape[channel_axis].value
 131     kernel_shape = self.kernel_size + (input_dim, self.filters)
 132     self.mask = self.add_variable(
 133         name='mask',
 134         shape=kernel_shape,
 135         initializer=init_ops.ones_initializer(),
 136         trainable=False,
 137         dtype=self.dtype)
 138 
 139     self.kernel = self.add_variable(
 140         name='kernel',
 141         shape=kernel_shape,
 142         initializer=self.kernel_initializer,
 143         regularizer=self.kernel_regularizer,
 144         trainable=True,
 145         dtype=self.dtype)
 146 
 147     self.threshold = self.add_variable(
 148         name='threshold',
 149         shape=[],
 150         initializer=init_ops.zeros_initializer(),
 151         trainable=False,
 152         dtype=self.dtype)
 153 
 154     # Add masked_weights in the weights namescope so as to make it easier
 155     # for the quantization library to add quant ops.
 156     self.masked_kernel = math_ops.multiply(self.mask, self.kernel,
 157                                            MASKED_WEIGHT_NAME)
 158 
 159     ops.add_to_collection(MASK_COLLECTION, self.mask)
 160     ops.add_to_collection(MASKED_WEIGHT_COLLECTION, self.masked_kernel)
 161     ops.add_to_collection(THRESHOLD_COLLECTION, self.threshold)
 162     ops.add_to_collection(WEIGHT_COLLECTION, self.kernel)
 163 
 164     if self.use_bias:
 165       self.bias = self.add_variable(
 166           name='bias',
 167           shape=(self.filters,),
 168           initializer=self.bias_initializer,
 169           regularizer=self.bias_regularizer,
 170           trainable=True,
 171           dtype=self.dtype)
 172     else:
 173       self.bias = None
 174     self.input_spec = base.InputSpec(
 175         ndim=self.rank + 2, axes={channel_axis: input_dim})
 176     self.built = True
 177 
 178   def call(self, inputs):
 179     outputs = nn.convolution(
 180         input=inputs,
 181         filter=self.masked_kernel,
 182         dilation_rate=self.dilation_rate,
 183         strides=self.strides,
 184         padding=self.padding.upper(),
 185         data_format=utils.convert_data_format(self.data_format, self.rank + 2))
 186 
 187     if self.bias is not None:
 188       if self.data_format == 'channels_first':
 189         if self.rank == 1:
 190           # nn.bias_add does not accept a 1D input tensor.
 191           bias = array_ops.reshape(self.bias, (1, self.filters, 1))
 192           outputs += bias
 193         if self.rank == 2:
 194           outputs = nn.bias_add(outputs, self.bias, data_format='NCHW')
 195         if self.rank == 3:
 196           # As of Mar 2017, direct addition is significantly slower than
 197           # bias_add when computing gradients. To use bias_add, we collapse Z
 198           # and Y into a single dimension to obtain a 4D input tensor.
 199           outputs_shape = outputs.shape.as_list()
 200           outputs_4d = array_ops.reshape(outputs, [
 201               outputs_shape[0], outputs_shape[1],
 202               outputs_shape[2] * outputs_shape[3], outputs_shape[4]
 203           ])
 204           outputs_4d = nn.bias_add(outputs_4d, self.bias, data_format='NCHW')
 205           outputs = array_ops.reshape(outputs_4d, outputs_shape)
 206       else:
 207         outputs = nn.bias_add(outputs, self.bias, data_format='NHWC')
 208 
 209     if self.activation is not None:
 210       return self.activation(outputs)
 211     return outputs
 212 
 213   def compute_output_shape(self, input_shape):
 214     input_shape = tensor_shape.TensorShape(input_shape).as_list()
 215     if self.data_format == 'channels_last':
 216       space = input_shape[1:-1]
 217       new_space = []
 218       for i in range(len(space)):
 219         new_dim = utils.conv_output_length(
 220             space[i],
 221             self.kernel_size[i],
 222             padding=self.padding,
 223             stride=self.strides[i],
 224             dilation=self.dilation_rate[i])
 225         new_space.append(new_dim)
 226       return tensor_shape.TensorShape([input_shape[0]] + new_space +
 227                                       [self.filters])
 228     else:
 229       space = input_shape[2:]
 230       new_space = []
 231       for i in range(len(space)):
 232         new_dim = utils.conv_output_length(
 233             space[i],
 234             self.kernel_size[i],
 235             padding=self.padding,
 236             stride=self.strides[i],
 237             dilation=self.dilation_rate[i])
 238         new_space.append(new_dim)
 239       return tensor_shape.TensorShape([input_shape[0], self.filters] +
 240                                       new_space)

 243 class MaskedConv2D(_MaskedConv):
 244   """2D convolution layer (e.g. spatial convolution over images).
 245 
 246   This layer creates a convolution kernel that is convolved
 247   (actually cross-correlated) with the layer input to produce a tensor of
 248   outputs. If `use_bias` is True (and a `bias_initializer` is provided),
 249   a bias vector is created and added to the outputs. Finally, if
 250   `activation` is not `None`, it is applied to the outputs as well.
 251 
 252   Arguments:
 253     filters: Integer, the dimensionality of the output space (i.e. the number
 254       of filters in the convolution).
 255     kernel_size: An integer or tuple/list of 2 integers, specifying the
 256       height and width of the 2D convolution window.
 257       Can be a single integer to specify the same value for
 258       all spatial dimensions.
 259     strides: An integer or tuple/list of 2 integers,
 260       specifying the strides of the convolution along the height and width.
 261       Can be a single integer to specify the same value for
 262       all spatial dimensions.
 263       Specifying any stride value != 1 is incompatible with specifying
 264       any `dilation_rate` value != 1.
 265     padding: One of `"valid"` or `"same"` (case-insensitive).
 266     data_format: A string, one of `channels_last` (default) or `channels_first`.
 267       The ordering of the dimensions in the inputs.
 268       `channels_last` corresponds to inputs with shape
 269       `(batch, height, width, channels)` while `channels_first` corresponds to
 270       inputs with shape `(batch, channels, height, width)`.
 271 
 272     dilation_rate: An integer or tuple/list of 2 integers, specifying
 273       the dilation rate to use for dilated convolution.
 274       Can be a single integer to specify the same value for
 275       all spatial dimensions.
 276       Currently, specifying any `dilation_rate` value != 1 is
 277       incompatible with specifying any stride value != 1.
 278     activation: Activation function. Set it to None to maintain a
 279       linear activation.
 280     use_bias: Boolean, whether the layer uses a bias.
 281     kernel_initializer: An initializer for the convolution kernel.
 282     bias_initializer: An initializer for the bias vector. If None, the default
 283       initializer will be used.
 284     kernel_regularizer: Optional regularizer for the convolution kernel.
 285     bias_regularizer: Optional regularizer for the bias vector.
 286     activity_regularizer: Regularizer function for the output.
 287     trainable: Boolean, if `True` also add variables to the graph collection
 288       `GraphKeys.TRAINABLE_VARIABLES` (see `tf.Variable`).
 289     name: A string, the name of the layer.
 290   """
 291 
 292   def __init__(self,
 293                filters,
 294                kernel_size,
 295                strides=(1, 1),
 296                padding='valid',
 297                data_format='channels_last',
 298                dilation_rate=(1, 1),
 299                activation=None,
 300                use_bias=True,
 301                kernel_initializer=None,
 302                bias_initializer=init_ops.zeros_initializer(),
 303                kernel_regularizer=None,
 304                bias_regularizer=None,
 305                activity_regularizer=None,
 306                trainable=True,
 307                name=None,
 308                **kwargs):
 309     super(MaskedConv2D, self).__init__(
 310         rank=2,
 311         filters=filters,
 312         kernel_size=kernel_size,
 313         strides=strides,
 314         padding=padding,
 315         data_format=data_format,
 316         dilation_rate=dilation_rate,
 317         activation=activation,
 318         use_bias=use_bias,
 319         kernel_initializer=kernel_initializer,
 320         bias_initializer=bias_initializer,
 321         kernel_regularizer=kernel_regularizer,
 322         bias_regularizer=bias_regularizer,
 323         activity_regularizer=activity_regularizer,
 324         trainable=trainable,
 325         name=name,
 326         **kwargs)

 329 class MaskedFullyConnected(base.Layer):
 330   """Fully-connected layer class with masked weights.
 331 
 332   This layer implements the operation:
 333   `outputs = activation(inputs.kernel + bias)`
 334   Where `activation` is the activation function passed as the `activation`
 335   argument (if not `None`), `kernel` is a weights matrix created by the layer,
 336   and `bias` is a bias vector created by the layer
 337   (only if `use_bias` is `True`).
 338 
 339   Note: if the input to the layer has a rank greater than 2, then it is
 340   flattened prior to the initial matrix multiply by `kernel`.
 341 
 342   Arguments:
 343     units: Integer or Long, dimensionality of the output space.
 344     activation: Activation function (callable). Set it to None to maintain a
 345       linear activation.
 346     use_bias: Boolean, whether the layer uses a bias.
 347     kernel_initializer: Initializer function for the weight matrix.
 348     bias_initializer: Initializer function for the bias.
 349     kernel_regularizer: Regularizer function for the weight matrix.
 350     bias_regularizer: Regularizer function for the bias.
 351     activity_regularizer: Regularizer function for the output.
 352     trainable: Boolean, if `True` also add variables to the graph collection
 353       `GraphKeys.TRAINABLE_VARIABLES` (see `tf.Variable`).
 354     name: String, the name of the layer. Layers with the same name will
 355       share weights, but to avoid mistakes we require reuse=True in such cases.
 356     reuse: Boolean, whether to reuse the weights of a previous layer
 357       by the same name.
 358 
 359   Properties:
 360     units: Python integer, dimensionality of the output space.
 361     activation: Activation function (callable).
 362     use_bias: Boolean, whether the layer uses a bias.
 363     kernel_initializer: Initializer instance (or name) for the weight matrix.
 364     bias_initializer: Initializer instance (or name) for the bias.
 365     kernel_regularizer: Regularizer instance for the weight matrix (callable)
 366     bias_regularizer: Regularizer instance for the bias (callable).
 367     activity_regularizer: Regularizer instance for the output (callable)
 368     kernel: Weight matrix (TensorFlow variable or tensor).
 369     bias: Bias vector, if applicable (TensorFlow variable or tensor).
 370   """
 371 
 372   def __init__(self,
 373                units,
 374                activation=None,
 375                use_bias=True,
 376                kernel_initializer=None,
 377                bias_initializer=init_ops.zeros_initializer(),
 378                kernel_regularizer=None,
 379                bias_regularizer=None,
 380                activity_regularizer=None,
 381                trainable=True,
 382                name=None,
 383                **kwargs):
 384     super(MaskedFullyConnected, self).__init__(
 385         trainable=trainable,
 386         name=name,
 387         activity_regularizer=activity_regularizer,
 388         **kwargs)
 389     self.units = units
 390     self.activation = activation
 391     self.use_bias = use_bias
 392     self.kernel_initializer = kernel_initializer
 393     self.bias_initializer = bias_initializer
 394     self.kernel_regularizer = kernel_regularizer
 395     self.bias_regularizer = bias_regularizer
 396     self.input_spec = base.InputSpec(min_ndim=2)
 397 
 398   def build(self, input_shape):
 399     input_shape = tensor_shape.TensorShape(input_shape)
 400     if input_shape[-1].value is None:
 401       raise ValueError('The last dimension of the inputs to `Dense` '
 402                        'should be defined. Found `None`.')
 403     self.input_spec = base.InputSpec(
 404         min_ndim=2, axes={-1: input_shape[-1].value})
 405 
 406     self.kernel = self.add_variable(
 407         'kernel',
 408         shape=[input_shape[-1].value, self.units],
 409         initializer=self.kernel_initializer,
 410         regularizer=self.kernel_regularizer,
 411         dtype=self.dtype,
 412         trainable=True)
 413 
 414     self.mask = self.add_variable(
 415         name='mask',
 416         shape=[input_shape[-1].value, self.units],
 417         initializer=init_ops.ones_initializer(),
 418         trainable=False,
 419         dtype=self.dtype)
 420 
 421     self.threshold = self.add_variable(
 422         name='threshold',
 423         shape=[],
 424         initializer=init_ops.zeros_initializer(),
 425         trainable=False,
 426         dtype=self.dtype)
 427 
 428     # Add masked_weights in the weights namescope so as to make it easier
 429     # for the quantization library to add quant ops.
 430     self.masked_kernel = math_ops.multiply(self.mask, self.kernel,
 431                                            MASKED_WEIGHT_NAME)
 432 
 433     ops.add_to_collection(MASK_COLLECTION, self.mask)
 434     ops.add_to_collection(MASKED_WEIGHT_COLLECTION, self.masked_kernel)
 435     ops.add_to_collection(THRESHOLD_COLLECTION, self.threshold)
 436     ops.add_to_collection(WEIGHT_COLLECTION, self.kernel)
 437 
 438     if self.use_bias:
 439       self.bias = self.add_variable(
 440           'bias',
 441           shape=[
 442               self.units,
 443           ],
 444           initializer=self.bias_initializer,
 445           regularizer=self.bias_regularizer,
 446           dtype=self.dtype,
 447           trainable=True)
 448     else:
 449       self.bias = None
 450     self.built = True
 451 
 452   def call(self, inputs):
 453     inputs = ops.convert_to_tensor(inputs, dtype=self.dtype)
 454     shape = inputs.get_shape().as_list()
 455     output_shape = shape[:-1] + [self.units]
 456     if len(output_shape) > 2:
 457       # Broadcasting is required for the inputs.
 458       outputs = standard_ops.tensordot(inputs, self.masked_kernel,
 459                                        [[len(shape) - 1], [0]])
 460       # Reshape the output back to the original ndim of the input.
 461       outputs.set_shape(output_shape)
 462     else:
 463       outputs = standard_ops.matmul(inputs, self.masked_kernel)
 464     if self.use_bias:
 465       outputs = nn.bias_add(outputs, self.bias)
 466     if self.activation is not None:
 467       return self.activation(outputs)  # pylint: disable=not-callable
 468     return outputs
 469 
 470   def compute_output_shape(self, input_shape):
 471     input_shape = tensor_shape.TensorShape(input_shape)
 472     input_shape = input_shape.with_rank_at_least(2)
 473     if input_shape[-1].value is None:
 474       raise ValueError(
 475           'The innermost dimension of input_shape must be defined, but saw: %s'
 476           % input_shape)
 477     return input_shape[:-1].concatenate(self.units)
