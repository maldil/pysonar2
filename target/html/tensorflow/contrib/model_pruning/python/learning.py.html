  15 """Wrapper around tf-slim's training code contrib/slim/python/slim/learning.py
  16 to support training of pruned models
  17 
  18 *******************************************************************
  19 * A simple working training script with support for model pruning *
  20 *******************************************************************
  21 
  22   # Load data and create the model:
  23   images, labels = LoadData(...)
  24   predictions = MyModel(images)
  25 
  26   # Define the loss:
  27   slim.losses.log_loss(predictions, labels)
  28   total_loss = slim.losses.get_total_loss()
  29 
  30   # Define the optimizer:
  31   optimizer = tf.train.MomentumOptimizer(FLAGS.learning_rate, FLAGS.momentum)
  32 
  33   # Create the train_op
  34   train_op = slim.learning.create_train_op(total_loss, optimizer)
  35 
  36   # Parse pruning hyperparameters
  37   pruning_hparams = pruning.get_pruning_hparams().parse(FLAGS.pruning_hparams)
  38 
  39   # Create a pruning object using the pruning_hparams
  40   p = pruning.Pruning(pruning_hparams)
  41 
  42   # Add mask update ops to the graph
  43   mask_update_op = p.conditional_mask_update_op()
  44 
  45   # Run training.
  46   learning.train(train_op,
  47                  my_log_dir,
  48                  mask_update_op)
  49   see contrib/slim/python/slim/learning.py for additional examples
  50 """
  51 from __future__ import absolute_import
  52 from __future__ import division
  53 from __future__ import print_function
  54 
  55 from tensorflow.contrib import slim as _slim
  56 
  57 _USE_DEFAULT = 0
  58 train_step = _slim.learning.train_step
  59 
  60 
  61 def train(train_op,
  62           logdir,
  63           mask_update_op,
  64           train_step_fn=train_step,
  65           train_step_kwargs=_USE_DEFAULT,
  66           log_every_n_steps=1,
  67           graph=None,
  68           master='',
  69           is_chief=True,
  70           global_step=None,
  71           number_of_steps=None,
  72           init_op=_USE_DEFAULT,
  73           init_feed_dict=None,
  74           local_init_op=_USE_DEFAULT,
  75           init_fn=None,
  76           ready_op=_USE_DEFAULT,
  77           summary_op=_USE_DEFAULT,
  78           save_summaries_secs=600,
  79           summary_writer=_USE_DEFAULT,
  80           startup_delay_steps=0,
  81           saver=None,
  82           save_interval_secs=600,
  83           sync_optimizer=None,
  84           session_config=None,
  85           trace_every_n_steps=None):
  86   """Wrapper around tf-slim's train function.
  87 
  88   Runs a training loop using a TensorFlow supervisor.
  89   When the sync_optimizer is supplied, gradient updates are applied
  90   synchronously. Otherwise, gradient updates are applied asynchronous.
  91 
  92   Args:
  93     train_op: A `Tensor` that, when executed, will apply the gradients and
  94       return the loss value.
  95     logdir: The directory where training logs are written to. If None, model
  96       checkpoints and summaries will not be written.
  97     mask_update_op: Operation that upon execution updates the weight masks and
  98       thresholds.
  99     train_step_fn: The function to call in order to execute a single gradient
 100       step. The function must have take exactly four arguments: the current
 101       session, the `train_op` `Tensor`, a global step `Tensor` and a dictionary.
 102     train_step_kwargs: A dictionary which is passed to the `train_step_fn`. By
 103       default, two `Boolean`, scalar ops called "should_stop" and "should_log"
 104       are provided.
 105     log_every_n_steps: The frequency, in terms of global steps, that the loss
 106       and global step and logged.
 107     graph: The graph to pass to the supervisor. If no graph is supplied the
 108       default graph is used.
 109     master: The address of the tensorflow master.
 110     is_chief: Specifies whether or not the training is being run by the primary
 111       replica during replica training.
 112     global_step: The `Tensor` representing the global step. If left as `None`,
 113       then slim.variables.get_or_create_global_step() is used.
 114     number_of_steps: The max number of gradient steps to take during training,
 115       as measured by 'global_step': training will stop if global_step is
 116       greater than 'number_of_steps'. If the value is left as None, training
 117       proceeds indefinitely.
 118     init_op: The initialization operation. If left to its default value, then
 119       the session is initialized by calling `tf.global_variables_initializer()`.
 120     init_feed_dict: A feed dictionary to use when executing the `init_op`.
 121     local_init_op: The local initialization operation. If left to its default
 122       value, then the session is initialized by calling
 123       `tf.local_variables_initializer()` and `tf.tables_initializer()`.
 124     init_fn: An optional callable to be executed after `init_op` is called. The
 125       callable must accept one argument, the session being initialized.
 126     ready_op: Operation to check if the model is ready to use. If left to its
 127       default value, then the session checks for readiness by calling
 128       `tf.report_uninitialized_variables()`.
 129     summary_op: The summary operation.
 130     save_summaries_secs: How often, in seconds, to save summaries.
 131     summary_writer: `SummaryWriter` to use.  Can be `None`
 132       to indicate that no summaries should be written. If unset, we
 133       create a SummaryWriter.
 134     startup_delay_steps: The number of steps to wait for before beginning. Note
 135       that this must be 0 if a sync_optimizer is supplied.
 136     saver: Saver to save checkpoints. If None, a default one will be created
 137       and used.
 138     save_interval_secs: How often, in seconds, to save the model to `logdir`.
 139     sync_optimizer: an instance of tf.train.SyncReplicasOptimizer, or a list of
 140       them. If the argument is supplied, gradient updates will be synchronous.
 141       If left as `None`, gradient updates will be asynchronous.
 142     session_config: An instance of `tf.ConfigProto` that will be used to
 143       configure the `Session`. If left as `None`, the default will be used.
 144     trace_every_n_steps: produce and save a `Timeline` in Chrome trace format
 145       and add it to the summaries every `trace_every_n_steps`. If None, no trace
 146       information will be produced or saved.
 147 
 148   Returns:
 149     the value of the loss function after training.
 150 
 151   Raises:
 152     ValueError: if `train_op` is empty or if `startup_delay_steps` is
 153       non-zero when `sync_optimizer` is supplied, if `number_of_steps` is
 154       negative, or if `trace_every_n_steps` is not `None` and no `logdir` is
 155       provided.
 156   """
 157 
 158   def train_step_with_pruning_fn(sess, train_op, global_step,
 159                                  train_step_kwargs):
 160     total_loss, should_stop = train_step_fn(sess, train_op, global_step,
 161                                             train_step_kwargs)
 162     sess.run(mask_update_op)
 163     return total_loss, should_stop
 164 
 165   total_loss, _ = _slim.learning.train(
 166       train_op,
 167       logdir,
 168       train_step_fn=train_step_with_pruning_fn,
 169       train_step_kwargs=train_step_kwargs,
 170       log_every_n_steps=log_every_n_steps,
 171       graph=graph,
 172       master=master,
 173       is_chief=is_chief,
 174       global_step=global_step,
 175       number_of_steps=number_of_steps,
 176       init_op=init_op,
 177       init_feed_dict=init_feed_dict,
 178       local_init_op=local_init_op,
 179       init_fn=init_fn,
 180       ready_op=ready_op,
 181       summary_op=summary_op,
 182       save_summaries_secs=save_summaries_secs,
 183       summary_writer=summary_writer,
 184       startup_delay_steps=startup_delay_steps,
 185       saver=saver,
 186       save_interval_secs=save_interval_secs,
 187       sync_optimizer=sync_optimizer,
 188       session_config=session_config,
 189       trace_every_n_steps=trace_every_n_steps)
 190 
 191   return total_loss
