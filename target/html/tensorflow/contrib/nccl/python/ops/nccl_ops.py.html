  35 def all_sum(tensors):
  36   """Returns a list of tensors with the all-reduce sum across `tensors`.
  37 
  38   The computation is done with an all-reduce operation, so if only some of the
  39   returned tensors are evaluated then the computation will hang.
  40 
  41   Args:
  42     tensors: The input tensors across which to sum; must be assigned
  43       to GPU devices.
  44 
  45   Returns:
  46     List of tensors, each with the sum of the input tensors, where tensor i has
  47     the same device as `tensors[i]`.
  48   """
  49   return _apply_all_reduce('sum', tensors)

  52 @ops.RegisterGradient('NcclAllReduce')
  53 def _all_sum_grad(op, grad):
  54   """The gradients for `all_sum`.
  55 
  56   Args:
  57     op: The `all_sum` `Operation` that we are differentiating.
  58     grad: Gradient with respect to the output of the `all_sum` op.
  59 
  60   Returns:
  61     The gradient with respect to the output of `all_sum`.
  62 
  63   Raises:
  64     LookupError: If `reduction` is not `sum`.
  65   """
  66   if op.get_attr('reduction') != b'sum':
  67     raise LookupError('No gradient defined for NcclAllReduce except sum.')
  68 
  69   _check_device(grad, expected=op.device)
  70   num_devices = op.get_attr('num_devices')
  71   shared_name = op.get_attr('shared_name') + b'_grad'
  72 
  73   with ops.device(op.device):
  74     return gen_nccl_ops.nccl_all_reduce(
  75         input=grad,
  76         reduction='sum',
  77         num_devices=num_devices,
  78         shared_name=shared_name)

 132 def reduce_sum(tensors):
 133   """Returns a tensor with the reduce sum across `tensors`.
 134 
 135   The computation is done with a reduce operation, so only one tensor is
 136   returned.
 137 
 138   Args:
 139     tensors: The input tensors across which to sum; must be assigned
 140       to GPU devices.
 141 
 142   Returns:
 143     A tensor containing the sum of the input tensors.
 144 
 145   Raises:
 146     LookupError: If context is not currently using a GPU device.
 147   """
 148   return _apply_reduce('sum', tensors)

 151 @ops.RegisterGradient('NcclReduce')
 152 def _reduce_sum_grad(op, grad):
 153   """The gradients for input `Operation` of `reduce_sum`.
 154 
 155   Args:
 156     op: The `sum send` `Operation` that we are differentiating.
 157     grad: Gradient with respect to the output of the `reduce_sum` op.
 158 
 159   Returns:
 160     The gradient with respect to the input of `reduce_sum` op.
 161 
 162   Raises:
 163     LookupError: If the reduction attribute of op is not `sum`.
 164   """
 165   if op.get_attr('reduction') != b'sum':
 166     raise LookupError('No gradient defined for NcclReduce except sum.')
 167   _check_device(grad, expected=op.device)
 168 
 169   with ops.device(op.device):
 170     result = gen_nccl_ops.nccl_broadcast(input=grad, shape=grad.shape)
 171 
 172   return [result] * len(op.inputs)

 175 def broadcast(tensor):
 176   """Returns a tensor that can be efficiently transferred to other devices.
 177 
 178   Args:
 179     tensor: The tensor to send; must be assigned to a GPU device.
 180 
 181   Returns:
 182     A tensor with the value of `src_tensor`, which can be used as input to
 183     ops on other GPU devices.
 184   """
 185   _validate_and_load_nccl_so()
 186   _check_device(tensor)
 187 
 188   with ops.device(tensor.device):
 189     return gen_nccl_ops.nccl_broadcast(input=tensor, shape=tensor.shape)
 190

 192 @ops.RegisterGradient('NcclBroadcast')
 193 def _broadcast_grad(op, accumulated_grad):
 194   """The gradients for input `Operation` of `broadcast`.
 195 
 196   Args:
 197     op: The `broadcast send` `Operation` that we are differentiating.
 198     accumulated_grad: Accumulated gradients with respect to the output of the
 199       `broadcast` op.
 200 
 201   Returns:
 202     Gradients with respect to the input of `broadcast`.
 203   """
 204   # Grab inputs of accumulated_grad and replace accumulation with reduce_sum.
 205   grads = [t for t in accumulated_grad.op.inputs]
 206   for t in grads:
 207     _check_device(t)
 208 
 209   with ops.device(op.device):
 210     return gen_nccl_ops.nccl_reduce(input=grads, reduction='sum')

 213 def _apply_all_reduce(reduction, tensors):
 214   """Helper function for all_* functions."""
 215   if not tensors:
 216     raise ValueError('Must pass >0 tensors to all reduce operations')
 217   _validate_and_load_nccl_so()
 218 
 219   shared_name = _get_shared_name()
 220   res = []
 221 
 222   for t in tensors:
 223     _check_device(t)
 224     with ops.device(t.device):
 225       res.append(
 226           gen_nccl_ops.nccl_all_reduce(
 227               input=t,
 228               reduction=reduction,
 229               num_devices=len(tensors),
 230               shared_name=shared_name))
 231 
 232   return res

 235 def _apply_reduce(reduction, tensors):
 236   """Helper function for reduce_* functions."""
 237   if not tensors:
 238     raise ValueError('Must pass >0 tensors to reduce operations')
 239   _validate_and_load_nccl_so()
 240 
 241   for t in tensors:
 242     _check_device(t)
 243   result = gen_nccl_ops.nccl_reduce(input=tensors, reduction=reduction)
 244   try:
 245     next(t for t in tensors if t.device == result.device)
 246   except StopIteration:
 247     raise ValueError('One input tensor must be assigned to current device')
 248   return result

 251 def _get_shared_name():
 252   global _shared_name_counter
 253 
 254   with _module_lock:
 255     val = _shared_name_counter
 256     _shared_name_counter += 1
 257   return 'c%s' % val

 260 def _check_device(tensor, expected=None):
 261   if not device.canonical_name(tensor.device):
 262     raise ValueError('Device assignment required for nccl collective ops')
 263   if expected and expected != tensor.device:
 264     raise ValueError('Expected device %s, got %s' % (expected, tensor.device))

 267 def _maybe_load_nccl_ops_so():
 268   """Loads nccl ops so if it hasn't been loaded already."""
 269 
 270   with _module_lock:
 271     global _nccl_ops_so
 272     if not _nccl_ops_so:
 273       _nccl_ops_so = loader.load_op_library(
 274           resource_loader.get_path_to_datafile('_nccl_ops.so'))

 277 def _validate_and_load_nccl_so():
 278   """Validates calling context and loads nccl ops so file.
 279 
 280   Raises:
 281     ValueError: Ops are not supported.
 282     errors_impl.NotFoundError: nccl library is not installed.
 283   """
 284 
 285   if context.executing_eagerly():
 286     raise ValueError('Nccl ops are not supported in eager mode')
 287 
 288   _maybe_load_nccl_ops_so()
