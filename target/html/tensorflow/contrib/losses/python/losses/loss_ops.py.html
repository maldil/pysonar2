  43 def _scale_losses(losses, weights):
  44   """Computes the scaled loss.
  45 
  46   Args:
  47     losses: A `Tensor` of size [batch_size, d1, ... dN].
  48     weights: A `Tensor` of size [1], [batch_size] or [batch_size, d1, ... dN].
  49       The `losses` are reduced (tf.reduce_sum) until its dimension matches
  50       that of `weights` at which point the reduced `losses` are element-wise
  51       multiplied by `weights` and a final reduce_sum is computed on the result.
  52       Conceptually, this operation is equivalent to broadcasting (tiling)
  53       `weights` to be the same size as `losses`, performing an element-wise
  54       multiplication, and summing the result.
  55 
  56   Returns:
  57     A scalar tf.float32 `Tensor` whose value represents the sum of the scaled
  58       `losses`.
  59   """
  60   # First, compute the sum of the losses over all elements:
  61   start_index = max(0, weights.get_shape().ndims)
  62   reduction_indices = list(range(start_index, losses.get_shape().ndims))
  63   reduced_losses = math_ops.reduce_sum(
  64       losses, reduction_indices=reduction_indices)
  65   reduced_losses = math_ops.multiply(reduced_losses, weights)
  66   return math_ops.reduce_sum(reduced_losses)

  69 def _safe_div(numerator, denominator, name="value"):
  70   """Computes a safe divide which returns 0 if the denominator is zero.
  71 
  72   Note that the function contains an additional conditional check that is
  73   necessary for avoiding situations where the loss is zero causing NaNs to
  74   creep into the gradient computation.
  75 
  76   Args:
  77     numerator: An arbitrary `Tensor`.
  78     denominator: A `Tensor` whose shape matches `numerator` and whose values are
  79       assumed to be non-negative.
  80     name: An optional name for the returned op.
  81 
  82   Returns:
  83     The element-wise value of the numerator divided by the denominator.
  84   """
  85   return array_ops.where(
  86       math_ops.greater(denominator, 0),
  87       math_ops.div(numerator,
  88                    array_ops.where(
  89                        math_ops.equal(denominator, 0),
  90                        array_ops.ones_like(denominator), denominator)),
  91       array_ops.zeros_like(numerator),
  92       name=name)

  95 def _safe_mean(losses, num_present):
  96   """Computes a safe mean of the losses.
  97 
  98   Args:
  99     losses: A tensor whose elements contain individual loss measurements.
 100     num_present: The number of measurable losses in the tensor.
 101 
 102   Returns:
 103     A scalar representing the mean of the losses. If `num_present` is zero,
 104       then zero is returned.
 105   """
 106   total_loss = math_ops.reduce_sum(losses)
 107   return _safe_div(total_loss, num_present)

 110 @deprecated("2016-12-30", "Use tf.losses.compute_weighted_loss instead.")
 111 def compute_weighted_loss(losses, weights=1.0, scope=None):
 112   """Computes the weighted loss.
 113 
 114   Args:
 115     losses: A tensor of size [batch_size, d1, ... dN].
 116     weights: A tensor of size [1] or [batch_size, d1, ... dK] where K < N.
 117     scope: the scope for the operations performed in computing the loss.
 118 
 119   Returns:
 120     A scalar `Tensor` that returns the weighted loss.
 121 
 122   Raises:
 123     ValueError: If `weights` is `None` or the shape is not compatible with
 124       `losses`, or if the number of dimensions (rank) of either `losses` or
 125       `weights` is missing.
 126   """
 127   with ops.name_scope(scope, "weighted_loss", [losses, weights]):
 128     losses = ops.convert_to_tensor(losses)
 129     input_dtype = losses.dtype
 130     losses = math_ops.to_float(losses)
 131     weights = math_ops.to_float(ops.convert_to_tensor(weights))
 132 
 133     if losses.get_shape().ndims is None:
 134       raise ValueError("losses.get_shape().ndims cannot be None")
 135     weights_shape = weights.get_shape()
 136     if weights_shape.ndims is None:
 137       raise ValueError("weights.get_shape().ndims cannot be None")
 138 
 139     if weights_shape.ndims > 1 and weights_shape.dims[-1].is_compatible_with(1):
 140       weights = array_ops.squeeze(weights, [-1])
 141 
 142     total_loss = _scale_losses(losses, weights)
 143     num_present = _num_present(losses, weights)
 144     mean_loss = _safe_mean(total_loss, num_present)
 145     # convert the result back to the input type
 146     mean_loss = math_ops.cast(mean_loss, input_dtype)
 147     add_loss(mean_loss)
 148     return mean_loss

 151 def _num_present(losses, weights, per_batch=False):
 152   """Computes the number of elements in the loss function induced by `weights`.
 153 
 154   A given weights tensor induces different numbers of usable elements in the
 155   `losses` tensor. The `weights` tensor is broadcast across `losses` for all
 156   possible dimensions. For example, if `losses` is a tensor of dimension
 157   [4, 5, 6, 3] and `weights` is a tensor of size [4, 5], then `weights` is, in
 158   effect, tiled to match the size of `losses`. Following this effective tile,
 159   the total number of present elements is the number of non-zero weights.
 160 
 161   Args:
 162     losses: A tensor of size [batch_size, d1, ... dN].
 163     weights: A tensor of size [1] or [batch_size, d1, ... dK] where K < N.
 164     per_batch: Whether to return the number of elements per batch or as a sum
 165       total.
 166 
 167   Returns:
 168     The number of present (non-zero) elements in the losses tensor. If
 169       `per_batch` is True, the value is returned as a tensor of size
 170       [batch_size]. Otherwise, a single scalar tensor is returned.
 171   """
 172   # If weights is a scalar, its easy to compute:
 173   if weights.get_shape().ndims == 0:
 174     batch_size = array_ops.reshape(
 175         array_ops.slice(array_ops.shape(losses), [0], [1]), [])
 176     num_per_batch = math_ops.div(
 177         math_ops.to_float(array_ops.size(losses)),
 178         math_ops.to_float(batch_size))
 179     num_per_batch = array_ops.where(
 180         math_ops.equal(weights, 0), 0.0, num_per_batch)
 181     num_per_batch = math_ops.multiply(
 182         array_ops.ones(array_ops.reshape(batch_size, [1])), num_per_batch)
 183     return num_per_batch if per_batch else math_ops.reduce_sum(num_per_batch)
 184 
 185   # First, count the number of nonzero weights:
 186   if weights.get_shape().ndims >= 1:
 187     reduction_indices = list(range(1, weights.get_shape().ndims))
 188     num_nonzero_per_batch = math_ops.reduce_sum(
 189         math_ops.to_float(math_ops.not_equal(weights, 0)),
 190         reduction_indices=reduction_indices)
 191 
 192   # Next, determine the number of elements that weights would broadcast to:
 193   broadcast_dims = array_ops.slice(
 194       array_ops.shape(losses), [weights.get_shape().ndims], [-1])
 195   num_to_broadcast = math_ops.to_float(math_ops.reduce_prod(broadcast_dims))
 196 
 197   num_per_batch = math_ops.multiply(num_nonzero_per_batch, num_to_broadcast)
 198   return num_per_batch if per_batch else math_ops.reduce_sum(num_per_batch)

 264 @deprecated("2016-12-30", "Use tf.losses.absolute_difference instead.")
 265 def absolute_difference(predictions, labels=None, weights=1.0, scope=None):
 266   """Adds an Absolute Difference loss to the training procedure.
 267 
 268   `weights` acts as a coefficient for the loss. If a scalar is provided, then
 269   the loss is simply scaled by the given value. If `weights` is a tensor of size
 270   [batch_size], then the total loss for each sample of the batch is rescaled
 271   by the corresponding element in the `weights` vector. If the shape of
 272   `weights` matches the shape of `predictions`, then the loss of each
 273   measurable element of `predictions` is scaled by the corresponding value of
 274   `weights`.
 275 
 276   Args:
 277     predictions: The predicted outputs.
 278     labels: The ground truth output tensor, same dimensions as 'predictions'.
 279     weights: Coefficients for the loss a scalar, a tensor of shape
 280       [batch_size] or a tensor whose shape matches `predictions`.
 281     scope: The scope for the operations performed in computing the loss.
 282 
 283   Returns:
 284     A scalar `Tensor` representing the loss value.
 285 
 286   Raises:
 287     ValueError: If the shape of `predictions` doesn't match that of `labels` or
 288       if the shape of `weights` is invalid.
 289   """
 290   with ops.name_scope(scope, "absolute_difference",
 291                       [predictions, labels, weights]) as scope:
 292     predictions.get_shape().assert_is_compatible_with(labels.get_shape())
 293     predictions = math_ops.to_float(predictions)
 294     labels = math_ops.to_float(labels)
 295     losses = math_ops.abs(math_ops.subtract(predictions, labels))

 351             "Use tf.losses.softmax_cross_entropy instead. Note that the order "
 352             "of the logits and labels arguments has been changed.")
 353 def softmax_cross_entropy(logits,
 354                           onehot_labels,
 355                           weights=1.0,
 356                           label_smoothing=0,
 357                           scope=None):
 358   """Creates a cross-entropy loss using tf.nn.softmax_cross_entropy_with_logits.
 359 
 360   `weights` acts as a coefficient for the loss. If a scalar is provided,
 361   then the loss is simply scaled by the given value. If `weights` is a
 362   tensor of size [`batch_size`], then the loss weights apply to each
 363   corresponding sample.
 364 
 365   If `label_smoothing` is nonzero, smooth the labels towards 1/num_classes:
 366       new_onehot_labels = onehot_labels * (1 - label_smoothing)
 367                           + label_smoothing / num_classes

 369   Args:
 370     logits: [batch_size, num_classes] logits outputs of the network .
 371     onehot_labels: [batch_size, num_classes] one-hot-encoded labels.
 372     weights: Coefficients for the loss. The tensor must be a scalar or a tensor
 373       of shape [batch_size].
 374     label_smoothing: If greater than 0 then smooth the labels.
 375     scope: the scope for the operations performed in computing the loss.
 376 
 377   Returns:
 378     A scalar `Tensor` representing the mean loss value.
 379 
 380   Raises:
 381     ValueError: If the shape of `logits` doesn't match that of `onehot_labels`
 382       or if the shape of `weights` is invalid or if `weights` is None.
 383   """
 384   with ops.name_scope(scope, "softmax_cross_entropy_loss",
 385                       [logits, onehot_labels, weights]) as scope:
 386     logits.get_shape().assert_is_compatible_with(onehot_labels.get_shape())
 387 
 388     onehot_labels = math_ops.cast(onehot_labels, logits.dtype)
 389 
 390     if label_smoothing > 0:
 391       num_classes = math_ops.cast(
 392           array_ops.shape(onehot_labels)[1], logits.dtype)
 393       smooth_positives = 1.0 - label_smoothing
 394       smooth_negatives = label_smoothing / num_classes
 395       onehot_labels = onehot_labels * smooth_positives + smooth_negatives
 396 
 397     losses = nn.softmax_cross_entropy_with_logits(
 398         labels=onehot_labels, logits=logits, name="xentropy")
 399     return compute_weighted_loss(losses, weights, scope=scope)

 437 @deprecated("2016-12-30",
 438             "Use tf.losses.log_loss instead. Note that the order of the "
 439             "predictions and labels arguments has been changed.")
 440 def log_loss(predictions, labels=None, weights=1.0, epsilon=1e-7, scope=None):
 441   """Adds a Log Loss term to the training procedure.
 442 
 443   `weights` acts as a coefficient for the loss. If a scalar is provided, then
 444   the loss is simply scaled by the given value. If `weights` is a tensor of size
 445   [batch_size], then the total loss for each sample of the batch is rescaled
 446   by the corresponding element in the `weights` vector. If the shape of
 447   `weights` matches the shape of `predictions`, then the loss of each
 448   measurable element of `predictions` is scaled by the corresponding value of
 449   `weights`.
 450 
 451   Args:
 452     predictions: The predicted outputs.
 453     labels: The ground truth output tensor, same dimensions as 'predictions'.
 454     weights: Coefficients for the loss a scalar, a tensor of shape
 455       [batch_size] or a tensor whose shape matches `predictions`.
 456     epsilon: A small increment to add to avoid taking a log of zero.
 457     scope: The scope for the operations performed in computing the loss.
 458 
 459   Returns:
 460     A scalar `Tensor` representing the loss value.
 461 
 462   Raises:
 463     ValueError: If the shape of `predictions` doesn't match that of `labels` or
 464       if the shape of `weights` is invalid.
 465   """
 466   with ops.name_scope(scope, "log_loss",
 467                       [predictions, labels, weights]) as scope:
 468     predictions.get_shape().assert_is_compatible_with(labels.get_shape())
 469     predictions = math_ops.to_float(predictions)
 470     labels = math_ops.to_float(labels)
 471     losses = -math_ops.multiply(
 472         labels, math_ops.log(predictions + epsilon)) - math_ops.multiply(
 473             (1 - labels), math_ops.log(1 - predictions + epsilon))
 474     return compute_weighted_loss(losses, weights, scope=scope)

 477 @deprecated("2016-12-30",
 478             "Use tf.losses.hinge_loss instead. Note that the order of the "
 479             "logits and labels arguments has been changed, and to stay "
 480             "unweighted, reduction=Reduction.NONE")
 481 def hinge_loss(logits, labels=None, scope=None):
 482   """Method that returns the loss tensor for hinge loss.
 483 
 484   Args:
 485     logits: The logits, a float tensor. Note that logits are assumed to be
 486       unbounded and 0-centered. A value > 0 (resp. < 0) is considered a positive
 487       (resp. negative) binary prediction.
 488     labels: The ground truth output tensor. Its shape should match the shape of
 489       logits. The values of the tensor are expected to be 0.0 or 1.0. Internally
 490       the {0,1} labels are converted to {-1,1} when calculating the hinge loss.
 491     scope: The scope for the operations performed in computing the loss.
 492 
 493   Returns:
 494     An unweighted `Tensor` of same shape as `logits` and `labels` representing
 495     the
 496       loss values across the batch.
 497 
 498   Raises:
 499     ValueError: If the shapes of `logits` and `labels` don't match.
 500   """
 501   with ops.name_scope(scope, "hinge_loss", [logits, labels]) as scope:
 502     logits.get_shape().assert_is_compatible_with(labels.get_shape())
 503     # We first need to convert binary labels to -1/1 labels (as floats).
 504     labels = math_ops.to_float(labels)
 505     all_ones = array_ops.ones_like(labels)
 506     labels = math_ops.subtract(2 * labels, all_ones)
 507     return nn_ops.relu(
 508         math_ops.subtract(all_ones, math_ops.multiply(labels, logits)))

 553   """Adds a pairwise-errors-squared loss to the training procedure.
 554 
 555   Unlike `mean_squared_error`, which is a measure of the differences between
 556   corresponding elements of `predictions` and `labels`,
 557   `mean_pairwise_squared_error` is a measure of the differences between pairs of
 558   corresponding elements of `predictions` and `labels`.
 559 
 560   For example, if `labels`=[a, b, c] and `predictions`=[x, y, z], there are
 561   three pairs of differences are summed to compute the loss:
 562     loss = [ ((a-b) - (x-y)).^2 + ((a-c) - (x-z)).^2 + ((b-c) - (y-z)).^2 ] / 3
 563 
 564   Note that since the inputs are of size [batch_size, d0, ... dN], the
 565   corresponding pairs are computed within each batch sample but not across
 566   samples within a batch. For example, if `predictions` represents a batch of
 567   16 grayscale images of dimension [batch_size, 100, 200], then the set of pairs
 568   is drawn from each image, but not across images.
 569 
 570   `weights` acts as a coefficient for the loss. If a scalar is provided, then
 571   the loss is simply scaled by the given value. If `weights` is a tensor of size
 572   [batch_size], then the total loss for each sample of the batch is rescaled
 573   by the corresponding element in the `weights` vector.
 574 
 575   Args:
 576     predictions: The predicted outputs, a tensor of size [batch_size, d0, .. dN]
 577       where N+1 is the total number of dimensions in `predictions`.
 578     labels: The ground truth output tensor, whose shape must match the shape of
 579       the `predictions` tensor.
 580     weights: Coefficients for the loss a scalar, a tensor of shape [batch_size]
 581       or a tensor whose shape matches `predictions`.
 582     scope: The scope for the operations performed in computing the loss.
 583 
 584   Returns:
 585     A scalar `Tensor` representing the loss value.
 586 
 587   Raises:
 588     ValueError: If the shape of `predictions` doesn't match that of `labels` or
 589       if the shape of `weights` is invalid.
 590   """
 591   with ops.name_scope(scope, "mean_pairwise_squared_error",
 592                       [predictions, labels, weights]) as scope:
 593     predictions.get_shape().assert_is_compatible_with(labels.get_shape())
 594     predictions = math_ops.to_float(predictions)
 595     labels = math_ops.to_float(labels)
 596     weights = math_ops.to_float(ops.convert_to_tensor(weights))
 597 
 598     diffs = math_ops.subtract(predictions, labels)
 599 
 600     # Need to verify here since the function doesn't use compute_weighted_loss
 601     if diffs.get_shape().ndims is None:
 602       raise ValueError("diffs.get_shape().ndims cannot be None")
 603     if weights.get_shape().ndims is None:
 604       raise ValueError("weights.get_shape().ndims cannot be None")
 605 
 606     reduction_indices = list(range(1, diffs.get_shape().ndims))
 607 
 608     sum_squares_diff_per_batch = math_ops.reduce_sum(
 609         math_ops.square(diffs), reduction_indices=reduction_indices)
 610     num_present_per_batch = _num_present(diffs, weights, per_batch=True)
 611 
 612     term1 = 2.0 * _safe_div(sum_squares_diff_per_batch, num_present_per_batch)
 613 
 614     sum_diff = math_ops.reduce_sum(diffs, reduction_indices=reduction_indices)
 615     term2 = 2.0 * _safe_div(
 616         math_ops.square(sum_diff), math_ops.square(num_present_per_batch))
 617 
 618     loss = _scale_losses(term1 - term2, weights)
 619 
 620     mean_loss = array_ops.where(
 621         math_ops.reduce_sum(num_present_per_batch) > 0,
 622         loss,
 623         array_ops.zeros_like(loss),
 624         name="value")
 625     add_loss(mean_loss)
 626     return mean_loss
