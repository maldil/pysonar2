  40 def pairwise_distance(feature, squared=False):
  41   """Computes the pairwise distance matrix with numerical stability.
  42 
  43   output[i, j] = || feature[i, :] - feature[j, :] ||_2
  44 
  45   Args:
  46     feature: 2-D Tensor of size [number of data, feature dimension].
  47     squared: Boolean, whether or not to square the pairwise distances.
  48 
  49   Returns:
  50     pairwise_distances: 2-D Tensor of size [number of data, number of data].
  51   """
  52   pairwise_distances_squared = math_ops.add(
  53       math_ops.reduce_sum(math_ops.square(feature), axis=[1], keepdims=True),
  54       math_ops.reduce_sum(
  55           math_ops.square(array_ops.transpose(feature)),
  56           axis=[0],
  57           keepdims=True)) - 2.0 * math_ops.matmul(feature,
  58                                                   array_ops.transpose(feature))
  59 
  60   # Deal with numerical inaccuracies. Set small negatives to zero.
  61   pairwise_distances_squared = math_ops.maximum(pairwise_distances_squared, 0.0)
  62   # Get the mask where the zero distances are at.
  63   error_mask = math_ops.less_equal(pairwise_distances_squared, 0.0)
  64 
  65   # Optionally take the sqrt.
  66   if squared:
  67     pairwise_distances = pairwise_distances_squared
  68   else:
  69     pairwise_distances = math_ops.sqrt(
  70         pairwise_distances_squared + math_ops.to_float(error_mask) * 1e-16)
  71 
  72   # Undo conditionally adding 1e-16.
  73   pairwise_distances = math_ops.multiply(
  74       pairwise_distances, math_ops.to_float(math_ops.logical_not(error_mask)))
  75 
  76   num_data = array_ops.shape(feature)[0]
  77   # Explicitly set diagonals to zero.
  78   mask_offdiagonals = array_ops.ones_like(pairwise_distances) - array_ops.diag(
  79       array_ops.ones([num_data]))
  80   pairwise_distances = math_ops.multiply(pairwise_distances, mask_offdiagonals)
  81   return pairwise_distances

  84 def contrastive_loss(labels, embeddings_anchor, embeddings_positive,
  85                      margin=1.0):
  86   """Computes the contrastive loss.
  87 
  88   This loss encourages the embedding to be close to each other for
  89     the samples of the same label and the embedding to be far apart at least
  90     by the margin constant for the samples of different labels.
  91   See: http://yann.lecun.com/exdb/publis/pdf/hadsell-chopra-lecun-06.pdf
  92 
  93   Args:
  94     labels: 1-D tf.int32 `Tensor` with shape [batch_size] of
  95       binary labels indicating positive vs negative pair.
  96     embeddings_anchor: 2-D float `Tensor` of embedding vectors for the anchor
  97       images. Embeddings should be l2 normalized.
  98     embeddings_positive: 2-D float `Tensor` of embedding vectors for the
  99       positive images. Embeddings should be l2 normalized.
 100     margin: margin term in the loss definition.
 101 
 102   Returns:
 103     contrastive_loss: tf.float32 scalar.
 104   """
 105   # Get per pair distances
 106   distances = math_ops.sqrt(
 107       math_ops.reduce_sum(
 108           math_ops.square(embeddings_anchor - embeddings_positive), 1))
 109 
 110   # Add contrastive loss for the siamese network.
 111   #   label here is {0,1} for neg, pos.
 112   return math_ops.reduce_mean(
 113       math_ops.to_float(labels) * math_ops.square(distances) +
 114       (1. - math_ops.to_float(labels)) *
 115       math_ops.square(math_ops.maximum(margin - distances, 0.)),
 116       name='contrastive_loss')

 119 def masked_maximum(data, mask, dim=1):
 120   """Computes the axis wise maximum over chosen elements.
 121 
 122   Args:
 123     data: 2-D float `Tensor` of size [n, m].
 124     mask: 2-D Boolean `Tensor` of size [n, m].
 125     dim: The dimension over which to compute the maximum.
 126 
 127   Returns:
 128     masked_maximums: N-D `Tensor`.
 129       The maximized dimension is of size 1 after the operation.
 130   """
 131   axis_minimums = math_ops.reduce_min(data, dim, keepdims=True)
 132   masked_maximums = math_ops.reduce_max(
 133       math_ops.multiply(data - axis_minimums, mask), dim,
 134       keepdims=True) + axis_minimums
 135   return masked_maximums

 138 def masked_minimum(data, mask, dim=1):
 139   """Computes the axis wise minimum over chosen elements.
 140 
 141   Args:
 142     data: 2-D float `Tensor` of size [n, m].
 143     mask: 2-D Boolean `Tensor` of size [n, m].
 144     dim: The dimension over which to compute the minimum.
 145 
 146   Returns:
 147     masked_minimums: N-D `Tensor`.
 148       The minimized dimension is of size 1 after the operation.
 149   """
 150   axis_maximums = math_ops.reduce_max(data, dim, keepdims=True)
 151   masked_minimums = math_ops.reduce_min(
 152       math_ops.multiply(data - axis_maximums, mask), dim,
 153       keepdims=True) + axis_maximums
 154   return masked_minimums

 157 def triplet_semihard_loss(labels, embeddings, margin=1.0):
 158   """Computes the triplet loss with semi-hard negative mining.
 159 
 160   The loss encourages the positive distances (between a pair of embeddings with
 161   the same labels) to be smaller than the minimum negative distance among
 162   which are at least greater than the positive distance plus the margin constant
 163   (called semi-hard negative) in the mini-batch. If no such negative exists,
 164   uses the largest negative distance instead.
 165   See: https://arxiv.org/abs/1503.03832.
 166 
 167   Args:
 168     labels: 1-D tf.int32 `Tensor` with shape [batch_size] of
 169       multiclass integer labels.
 170     embeddings: 2-D float `Tensor` of embedding vectors. Embeddings should
 171       be l2 normalized.
 172     margin: Float, margin term in the loss definition.
 173 
 174   Returns:
 175     triplet_loss: tf.float32 scalar.
 176   """
 177   # Reshape [batch_size] label tensor to a [batch_size, 1] label tensor.
 178   lshape = array_ops.shape(labels)
 179   assert lshape.shape == 1
 180   labels = array_ops.reshape(labels, [lshape[0], 1])
 181 
 182   # Build pairwise squared distance matrix.
 183   pdist_matrix = pairwise_distance(embeddings, squared=True)
 184   # Build pairwise binary adjacency matrix.
 185   adjacency = math_ops.equal(labels, array_ops.transpose(labels))
 186   # Invert so we can select negatives only.
 187   adjacency_not = math_ops.logical_not(adjacency)
 188 
 189   batch_size = array_ops.size(labels)
 190 
 191   # Compute the mask.
 192   pdist_matrix_tile = array_ops.tile(pdist_matrix, [batch_size, 1])
 193   mask = math_ops.logical_and(
 194       array_ops.tile(adjacency_not, [batch_size, 1]),
 195       math_ops.greater(
 196           pdist_matrix_tile, array_ops.reshape(
 197               array_ops.transpose(pdist_matrix), [-1, 1])))
 198   mask_final = array_ops.reshape(
 199       math_ops.greater(
 200           math_ops.reduce_sum(
 201               math_ops.cast(mask, dtype=dtypes.float32), 1, keepdims=True),
 202           0.0), [batch_size, batch_size])
 203   mask_final = array_ops.transpose(mask_final)
 204 
 205   adjacency_not = math_ops.cast(adjacency_not, dtype=dtypes.float32)
 206   mask = math_ops.cast(mask, dtype=dtypes.float32)
 207 
 208   # negatives_outside: smallest D_an where D_an > D_ap.
 209   negatives_outside = array_ops.reshape(
 210       masked_minimum(pdist_matrix_tile, mask), [batch_size, batch_size])
 211   negatives_outside = array_ops.transpose(negatives_outside)
 212 
 213   # negatives_inside: largest D_an.
 214   negatives_inside = array_ops.tile(
 215       masked_maximum(pdist_matrix, adjacency_not), [1, batch_size])
 216   semi_hard_negatives = array_ops.where(
 217       mask_final, negatives_outside, negatives_inside)
 218 
 219   loss_mat = math_ops.add(margin, pdist_matrix - semi_hard_negatives)
 220 
 221   mask_positives = math_ops.cast(
 222       adjacency, dtype=dtypes.float32) - array_ops.diag(
 223           array_ops.ones([batch_size]))
 224 
 225   # In lifted-struct, the authors multiply 0.5 for upper triangular
 226   #   in semihard, they take all positive pairs except the diagonal.
 227   num_positives = math_ops.reduce_sum(mask_positives)
 228 
 229   triplet_loss = math_ops.truediv(
 230       math_ops.reduce_sum(
 231           math_ops.maximum(
 232               math_ops.multiply(loss_mat, mask_positives), 0.0)),
 233       num_positives,
 234       name='triplet_semihard_loss')
 235 
 236   return triplet_loss

 240 def npairs_loss(labels, embeddings_anchor, embeddings_positive,
 241                 reg_lambda=0.002, print_losses=False):
 242   """Computes the npairs loss.
 243 
 244   Npairs loss expects paired data where a pair is composed of samples from the
 245   same labels and each pairs in the minibatch have different labels. The loss
 246   has two components. The first component is the L2 regularizer on the
 247   embedding vectors. The second component is the sum of cross entropy loss
 248   which takes each row of the pair-wise similarity matrix as logits and
 249   the remapped one-hot labels as labels.
 250 
 251   See: http://www.nec-labs.com/uploads/images/Department-Images/MediaAnalytics/papers/nips16_npairmetriclearning.pdf
 252 
 253   Args:
 254     labels: 1-D tf.int32 `Tensor` of shape [batch_size/2].
 255     embeddings_anchor: 2-D Tensor of shape [batch_size/2, embedding_dim] for the
 256       embedding vectors for the anchor images. Embeddings should not be
 257       l2 normalized.
 258     embeddings_positive: 2-D Tensor of shape [batch_size/2, embedding_dim] for the
 259       embedding vectors for the positive images. Embeddings should not be
 260       l2 normalized.
 261     reg_lambda: Float. L2 regularization term on the embedding vectors.
 262     print_losses: Boolean. Option to print the xent and l2loss.
 263 
 264   Returns:
 265     npairs_loss: tf.float32 scalar.
 266   """
 267   # pylint: enable=line-too-long
 268   # Add the regularizer on the embedding.
 269   reg_anchor = math_ops.reduce_mean(
 270       math_ops.reduce_sum(math_ops.square(embeddings_anchor), 1))
 271   reg_positive = math_ops.reduce_mean(
 272       math_ops.reduce_sum(math_ops.square(embeddings_positive), 1))
 273   l2loss = math_ops.multiply(
 274       0.25 * reg_lambda, reg_anchor + reg_positive, name='l2loss')
 275 
 276   # Get per pair similarities.
 277   similarity_matrix = math_ops.matmul(
 278       embeddings_anchor, embeddings_positive, transpose_a=False,
 279       transpose_b=True)
 280 
 281   # Reshape [batch_size] label tensor to a [batch_size, 1] label tensor.
 282   lshape = array_ops.shape(labels)
 283   assert lshape.shape == 1
 284   labels = array_ops.reshape(labels, [lshape[0], 1])
 285 
 286   labels_remapped = math_ops.to_float(
 287       math_ops.equal(labels, array_ops.transpose(labels)))
 288   labels_remapped /= math_ops.reduce_sum(labels_remapped, 1, keepdims=True)
 289 
 290   # Add the softmax loss.
 291   xent_loss = nn.softmax_cross_entropy_with_logits(
 292       logits=similarity_matrix, labels=labels_remapped)
 293   xent_loss = math_ops.reduce_mean(xent_loss, name='xentropy')
 294 
 295   if print_losses:
 296     xent_loss = logging_ops.Print(
 297         xent_loss, ['cross entropy:', xent_loss, 'l2loss:', l2loss])
 298 
 299   return l2loss + xent_loss

 302 def _build_multilabel_adjacency(sparse_labels):
 303   """Builds multilabel adjacency matrix.
 304 
 305   As of March 14th, 2017, there's no op for the dot product between
 306   two sparse tensors in TF. However, there is `sparse_minimum` op which is
 307   equivalent to an AND op between two sparse boolean tensors.
 308   This computes the dot product between two sparse boolean inputs.
 309 
 310   Args:
 311     sparse_labels: List of 1-D boolean sparse tensors.
 312 
 313   Returns:
 314     adjacency_matrix: 2-D dense `Tensor`.
 315   """
 316   num_pairs = len(sparse_labels)
 317   adjacency_matrix = array_ops.zeros([num_pairs, num_pairs])
 318   for i in range(num_pairs):
 319     for j in range(num_pairs):
 320       sparse_dot_product = math_ops.to_float(
 321           sparse_ops.sparse_reduce_sum(sparse_ops.sparse_minimum(
 322               sparse_labels[i], sparse_labels[j])))
 323       sparse_dot_product = array_ops.expand_dims(sparse_dot_product, 0)
 324       sparse_dot_product = array_ops.expand_dims(sparse_dot_product, 1)
 325       one_hot_matrix = array_ops.pad(sparse_dot_product,
 326                                      [[i, num_pairs-i-1],
 327                                       [j, num_pairs-j-1]], 'CONSTANT')
 328       adjacency_matrix += one_hot_matrix
 329 
 330   return adjacency_matrix

 333 def npairs_loss_multilabel(sparse_labels, embeddings_anchor,
 334                            embeddings_positive, reg_lambda=0.002,
 335                            print_losses=False):
 336   r"""Computes the npairs loss with multilabel data.
 337 
 338   Npairs loss expects paired data where a pair is composed of samples from the
 339   same labels and each pairs in the minibatch have different labels. The loss
 340   has two components. The first component is the L2 regularizer on the
 341   embedding vectors. The second component is the sum of cross entropy loss
 342   which takes each row of the pair-wise similarity matrix as logits and
 343   the remapped one-hot labels as labels. Here, the similarity is defined by the
 344   dot product between two embedding vectors. S_{i,j} = f(x_i)^T f(x_j)
 345 
 346   To deal with multilabel inputs, we use the count of label intersection
 347   i.e. L_{i,j} = | set_of_labels_for(i) \cap set_of_labels_for(j) |
 348   Then we normalize each rows of the count based label matrix so that each row
 349   sums to one.
 350 
 351   Args:
 352     sparse_labels: List of 1-D Boolean `SparseTensor` of dense_shape
 353                    [batch_size/2, num_classes] labels for the anchor-pos pairs.
 354     embeddings_anchor: 2-D `Tensor` of shape [batch_size/2, embedding_dim] for
 355       the embedding vectors for the anchor images. Embeddings should not be
 356       l2 normalized.
 357     embeddings_positive: 2-D `Tensor` of shape [batch_size/2, embedding_dim] for
 358       the embedding vectors for the positive images. Embeddings should not be
 359       l2 normalized.
 360     reg_lambda: Float. L2 regularization term on the embedding vectors.
 361     print_losses: Boolean. Option to print the xent and l2loss.
 362 
 363   Returns:
 364     npairs_loss: tf.float32 scalar.
 365   Raises:
 366     TypeError: When the specified sparse_labels is not a `SparseTensor`.
 367   """
 368   if False in [isinstance(
 369       l, sparse_tensor.SparseTensor) for l in sparse_labels]:
 370     raise TypeError(
 371         'sparse_labels must be a list of SparseTensors, but got %s' % str(
 372             sparse_labels))
 373 
 374   with ops.name_scope('NpairsLossMultiLabel'):
 375     # Add the regularizer on the embedding.
 376     reg_anchor = math_ops.reduce_mean(
 377         math_ops.reduce_sum(math_ops.square(embeddings_anchor), 1))
 378     reg_positive = math_ops.reduce_mean(
 379         math_ops.reduce_sum(math_ops.square(embeddings_positive), 1))
 380     l2loss = math_ops.multiply(0.25 * reg_lambda,
 381                                reg_anchor + reg_positive, name='l2loss')
 382 
 383     # Get per pair similarities.
 384     similarity_matrix = math_ops.matmul(
 385         embeddings_anchor, embeddings_positive, transpose_a=False,
 386         transpose_b=True)
 387 
 388     # TODO(coreylynch): need to check the sparse values
 389     # TODO(coreylynch): are composed only of 0's and 1's.
 390 
 391     multilabel_adjacency_matrix = _build_multilabel_adjacency(sparse_labels)
 392     labels_remapped = math_ops.to_float(multilabel_adjacency_matrix)
 393     labels_remapped /= math_ops.reduce_sum(labels_remapped, 1, keepdims=True)
 394 
 395     # Add the softmax loss.
 396     xent_loss = nn.softmax_cross_entropy_with_logits(
 397         logits=similarity_matrix, labels=labels_remapped)
 398     xent_loss = math_ops.reduce_mean(xent_loss, name='xentropy')
 399 
 400     if print_losses:
 401       xent_loss = logging_ops.Print(
 402           xent_loss, ['cross entropy:', xent_loss, 'l2loss:', l2loss])
 403 
 404     return l2loss + xent_loss

 407 def lifted_struct_loss(labels, embeddings, margin=1.0):
 408   """Computes the lifted structured loss.
 409 
 410   The loss encourages the positive distances (between a pair of embeddings
 411   with the same labels) to be smaller than any negative distances (between a
 412   pair of embeddings with different labels) in the mini-batch in a way
 413   that is differentiable with respect to the embedding vectors.
 414   See: https://arxiv.org/abs/1511.06452.
 415 
 416   Args:
 417     labels: 1-D tf.int32 `Tensor` with shape [batch_size] of
 418       multiclass integer labels.
 419     embeddings: 2-D float `Tensor` of embedding vectors. Embeddings should not
 420       be l2 normalized.
 421     margin: Float, margin term in the loss definition.
 422 
 423   Returns:
 424     lifted_loss: tf.float32 scalar.
 425   """
 426   # Reshape [batch_size] label tensor to a [batch_size, 1] label tensor.
 427   lshape = array_ops.shape(labels)
 428   assert lshape.shape == 1
 429   labels = array_ops.reshape(labels, [lshape[0], 1])
 430 
 431   # Build pairwise squared distance matrix.
 432   pairwise_distances = pairwise_distance(embeddings)
 433 
 434   # Build pairwise binary adjacency matrix.
 435   adjacency = math_ops.equal(labels, array_ops.transpose(labels))
 436   # Invert so we can select negatives only.
 437   adjacency_not = math_ops.logical_not(adjacency)
 438 
 439   batch_size = array_ops.size(labels)
 440 
 441   diff = margin - pairwise_distances
 442   mask = math_ops.cast(adjacency_not, dtype=dtypes.float32)
 443   # Safe maximum: Temporarily shift negative distances
 444   #   above zero before taking max.
 445   #     this is to take the max only among negatives.
 446   row_minimums = math_ops.reduce_min(diff, 1, keepdims=True)
 447   row_negative_maximums = math_ops.reduce_max(
 448       math_ops.multiply(diff - row_minimums, mask), 1,
 449       keepdims=True) + row_minimums
 450 
 451   # Compute the loss.
 452   # Keep track of matrix of maximums where M_ij = max(m_i, m_j)
 453   #   where m_i is the max of alpha - negative D_i's.
 454   # This matches the Caffe loss layer implementation at:
 455   #   https://github.com/rksltnl/Caffe-Deep-Metric-Learning-CVPR16/blob/0efd7544a9846f58df923c8b992198ba5c355454/src/caffe/layers/lifted_struct_similarity_softmax_layer.cpp  # pylint: disable=line-too-long
 456 
 457   max_elements = math_ops.maximum(
 458       row_negative_maximums, array_ops.transpose(row_negative_maximums))
 459   diff_tiled = array_ops.tile(diff, [batch_size, 1])
 460   mask_tiled = array_ops.tile(mask, [batch_size, 1])
 461   max_elements_vect = array_ops.reshape(
 462       array_ops.transpose(max_elements), [-1, 1])
 463 
 464   loss_exp_left = array_ops.reshape(
 465       math_ops.reduce_sum(
 466           math_ops.multiply(
 467               math_ops.exp(diff_tiled - max_elements_vect), mask_tiled),
 468           1,
 469           keepdims=True), [batch_size, batch_size])
 470 
 471   loss_mat = max_elements + math_ops.log(
 472       loss_exp_left + array_ops.transpose(loss_exp_left))
 473   # Add the positive distance.
 474   loss_mat += pairwise_distances
 475 
 476   mask_positives = math_ops.cast(
 477       adjacency, dtype=dtypes.float32) - array_ops.diag(
 478           array_ops.ones([batch_size]))
 479 
 480   # *0.5 for upper triangular, and another *0.5 for 1/2 factor for loss^2.
 481   num_positives = math_ops.reduce_sum(mask_positives) / 2.0
 482 
 483   lifted_loss = math_ops.truediv(
 484       0.25 * math_ops.reduce_sum(
 485           math_ops.square(
 486               math_ops.maximum(
 487                   math_ops.multiply(loss_mat, mask_positives), 0.0))),
 488       num_positives,
 489       name='liftedstruct_loss')
 490   return lifted_loss

 493 def update_1d_tensor(y, index, value):
 494   """Updates 1d tensor y so that y[index] = value.
 495 
 496   Args:
 497     y: 1-D Tensor.
 498     index: index of y to modify.
 499     value: new value to write at y[index].
 500 
 501   Returns:
 502     y_mod: 1-D Tensor. Tensor y after the update.
 503   """
 504   value = array_ops.squeeze(value)
 505   # modify the 1D tensor x at index with value.
 506   # ex) chosen_ids = update_1D_tensor(chosen_ids, cluster_idx, best_medoid)
 507   y_before = array_ops.slice(y, [0], [index])
 508   y_after = array_ops.slice(y, [index + 1], [-1])
 509   y_mod = array_ops.concat([y_before, [value], y_after], 0)
 510   return y_mod

 513 def get_cluster_assignment(pairwise_distances, centroid_ids):
 514   """Assign data points to the neareset centroids.
 515 
 516   Tensorflow has numerical instability and doesn't always choose
 517     the data point with theoretically zero distance as it's nearest neighbor.
 518     Thus, for each centroid in centroid_ids, explicitly assign
 519     the centroid itself as the nearest centroid.
 520     This is done through the mask tensor and the constraint_vect tensor.
 521 
 522   Args:
 523     pairwise_distances: 2-D Tensor of pairwise distances.
 524     centroid_ids: 1-D Tensor of centroid indices.
 525 
 526   Returns:
 527     y_fixed: 1-D tensor of cluster assignment.
 528   """
 529   predictions = math_ops.argmin(
 530       array_ops.gather(pairwise_distances, centroid_ids), dimension=0)
 531   batch_size = array_ops.shape(pairwise_distances)[0]
 532 
 533   # Deal with numerical instability
 534   mask = math_ops.reduce_any(array_ops.one_hot(
 535       centroid_ids, batch_size, True, False, axis=-1, dtype=dtypes.bool),
 536                              axis=0)
 537   constraint_one_hot = math_ops.multiply(
 538       array_ops.one_hot(centroid_ids,
 539                         batch_size,
 540                         array_ops.constant(1, dtype=dtypes.int64),
 541                         array_ops.constant(0, dtype=dtypes.int64),
 542                         axis=0,
 543                         dtype=dtypes.int64),
 544       math_ops.to_int64(math_ops.range(array_ops.shape(centroid_ids)[0])))
 545   constraint_vect = math_ops.reduce_sum(
 546       array_ops.transpose(constraint_one_hot), axis=0)
 547 
 548   y_fixed = array_ops.where(mask, constraint_vect, predictions)
 549   return y_fixed

 552 def compute_facility_energy(pairwise_distances, centroid_ids):
 553   """Compute the average travel distance to the assigned centroid.
 554 
 555   Args:
 556     pairwise_distances: 2-D Tensor of pairwise distances.
 557     centroid_ids: 1-D Tensor of indices.
 558 
 559   Returns:
 560     facility_energy: dtypes.float32 scalar.
 561   """
 562   return -1.0 * math_ops.reduce_sum(
 563       math_ops.reduce_min(
 564           array_ops.gather(pairwise_distances, centroid_ids), axis=0))

 567 def compute_clustering_score(labels, predictions, margin_type):
 568   """Computes the clustering score via sklearn.metrics functions.
 569 
 570   There are various ways to compute the clustering score. Intuitively,
 571   we want to measure the agreement of two clustering assignments (labels vs
 572   predictions) ignoring the permutations and output a score from zero to one.
 573   (where the values close to one indicate significant agreement).
 574   This code supports following scoring functions:
 575     nmi: normalized mutual information
 576     ami: adjusted mutual information
 577     ari: adjusted random index
 578     vmeasure: v-measure
 579     const: indicator checking whether the two clusterings are the same.
 580   See http://scikit-learn.org/stable/modules/classes.html#clustering-metrics
 581     for the detailed descriptions.
 582   Args:
 583     labels: 1-D Tensor. ground truth cluster assignment.
 584     predictions: 1-D Tensor. predicted cluster assignment.
 585     margin_type: Type of structured margin to use. Default is nmi.
 586   Returns:
 587     clustering_score: dtypes.float32 scalar.
 588       The possible valid values are from zero to one.
 589       Zero means the worst clustering and one means the perfect clustering.
 590   Raises:
 591     ValueError: margin_type is not recognized.
 592   """
 593   margin_type_to_func = {
 594       'nmi': _compute_nmi_score,
 595       'ami': _compute_ami_score,
 596       'ari': _compute_ari_score,
 597       'vmeasure': _compute_vmeasure_score,
 598       'const': _compute_zeroone_score
 599   }
 600 
 601   if margin_type not in margin_type_to_func:
 602     raise ValueError('Unrecognized margin_type: %s' % margin_type)
 603   clustering_score_fn = margin_type_to_func[margin_type]
 604   return array_ops.squeeze(clustering_score_fn(labels, predictions))
 605 
 606 
 607 def _compute_nmi_score(labels, predictions):
 608   return math_ops.to_float(
 609       script_ops.py_func(
 610           metrics.normalized_mutual_info_score, [labels, predictions],
 611           [dtypes.float64],
 612           name='nmi'))
 613 
 614 
 615 def _compute_ami_score(labels, predictions):
 616   ami_score = math_ops.to_float(
 617       script_ops.py_func(
 618           metrics.adjusted_mutual_info_score, [labels, predictions],
 619           [dtypes.float64],
 620           name='ami'))
 621   return math_ops.maximum(0.0, ami_score)
 622 
 623 
 624 def _compute_ari_score(labels, predictions):
 625   ari_score = math_ops.to_float(
 626       script_ops.py_func(
 627           metrics.adjusted_rand_score, [labels, predictions], [dtypes.float64],
 628           name='ari'))
 629   # ari score can go below 0
 630   # http://scikit-learn.org/stable/modules/clustering.html#adjusted-rand-score
 631   return math_ops.maximum(0.0, ari_score)
 632 
 633 
 634 def _compute_vmeasure_score(labels, predictions):
 635   vmeasure_score = math_ops.to_float(
 636       script_ops.py_func(
 637           metrics.v_measure_score, [labels, predictions], [dtypes.float64],
 638           name='vmeasure'))
 639   return math_ops.maximum(0.0, vmeasure_score)
 640

 642 def _compute_zeroone_score(labels, predictions):
 643   zeroone_score = math_ops.to_float(
 644       math_ops.equal(
 645           math_ops.reduce_sum(
 646               math_ops.to_int32(math_ops.equal(labels, predictions))),
 647           array_ops.shape(labels)[0]))
 648   return zeroone_score

 651 def _find_loss_augmented_facility_idx(pairwise_distances, labels, chosen_ids,
 652                                       candidate_ids, margin_multiplier,
 653                                       margin_type):
 654   """Find the next centroid that maximizes the loss augmented inference.
 655 
 656   This function is a subroutine called from compute_augmented_facility_locations
 657 
 658   Args:
 659     pairwise_distances: 2-D Tensor of pairwise distances.
 660     labels: 1-D Tensor of ground truth cluster assignment.
 661     chosen_ids: 1-D Tensor of current centroid indices.
 662     candidate_ids: 1-D Tensor of candidate indices.
 663     margin_multiplier: multiplication constant.
 664     margin_type: Type of structured margin to use. Default is nmi.
 665 
 666   Returns:
 667     integer index.
 668   """
 669   num_candidates = array_ops.shape(candidate_ids)[0]
 670 
 671   pairwise_distances_chosen = array_ops.gather(pairwise_distances, chosen_ids)
 672   pairwise_distances_candidate = array_ops.gather(
 673       pairwise_distances, candidate_ids)
 674   pairwise_distances_chosen_tile = array_ops.tile(
 675       pairwise_distances_chosen, [1, num_candidates])
 676 
 677   candidate_scores = -1.0 * math_ops.reduce_sum(
 678       array_ops.reshape(
 679           math_ops.reduce_min(
 680               array_ops.concat([
 681                   pairwise_distances_chosen_tile,
 682                   array_ops.reshape(pairwise_distances_candidate, [1, -1])
 683               ], 0),
 684               axis=0,
 685               keepdims=True), [num_candidates, -1]),
 686       axis=1)
 687 
 688   nmi_scores = array_ops.zeros([num_candidates])
 689   iteration = array_ops.constant(0)
 690 
 691   def func_cond(iteration, nmi_scores):
 692     del nmi_scores  # Unused in func_cond()
 693     return iteration < num_candidates
 694 
 695   def func_body(iteration, nmi_scores):
 696     predictions = get_cluster_assignment(
 697         pairwise_distances,
 698         array_ops.concat([chosen_ids, [candidate_ids[iteration]]], 0))
 699     nmi_score_i = compute_clustering_score(labels, predictions, margin_type)
 700     pad_before = array_ops.zeros([iteration])
 701     pad_after = array_ops.zeros([num_candidates - 1 - iteration])
 702     # return 1 - NMI score as the structured loss.
 703     #   because NMI is higher the better [0,1].
 704     return iteration + 1, nmi_scores + array_ops.concat(
 705         [pad_before, [1.0 - nmi_score_i], pad_after], 0)
 706 
 707   _, nmi_scores = control_flow_ops.while_loop(
 708       func_cond, func_body, [iteration, nmi_scores])
 709 
 710   candidate_scores = math_ops.add(
 711       candidate_scores, margin_multiplier * nmi_scores)
 712 
 713   argmax_index = math_ops.to_int32(
 714       math_ops.argmax(candidate_scores, axis=0))
 715 
 716   return candidate_ids[argmax_index]
 717

 719 def compute_augmented_facility_locations(pairwise_distances, labels, all_ids,
 720                                          margin_multiplier, margin_type):
 721   """Computes the centroid locations.
 722 
 723   Args:
 724     pairwise_distances: 2-D Tensor of pairwise distances.
 725     labels: 1-D Tensor of ground truth cluster assignment.
 726     all_ids: 1-D Tensor of all data indices.
 727     margin_multiplier: multiplication constant.
 728     margin_type: Type of structured margin to use. Default is nmi.
 729 
 730   Returns:
 731     chosen_ids: 1-D Tensor of chosen centroid indices.
 732   """
 733 
 734   def func_cond_augmented(iteration, chosen_ids):
 735     del chosen_ids  # Unused argument in func_cond_augmented.
 736     return iteration < num_classes
 737 
 738   def func_body_augmented(iteration, chosen_ids):
 739     # find a new facility location to add
 740     #  based on the clustering score and the NMI score
 741     candidate_ids = array_ops.setdiff1d(all_ids, chosen_ids)[0]
 742     new_chosen_idx = _find_loss_augmented_facility_idx(pairwise_distances,
 743                                                        labels, chosen_ids,
 744                                                        candidate_ids,
 745                                                        margin_multiplier,
 746                                                        margin_type)
 747     chosen_ids = array_ops.concat([chosen_ids, [new_chosen_idx]], 0)
 748     return iteration + 1, chosen_ids
 749 
 750   num_classes = array_ops.size(array_ops.unique(labels)[0])
 751   chosen_ids = array_ops.constant(0, dtype=dtypes.int32, shape=[0])
 752 
 753   # num_classes get determined at run time based on the sampled batch.
 754   iteration = array_ops.constant(0)
 755 
 756   _, chosen_ids = control_flow_ops.while_loop(
 757       func_cond_augmented,
 758       func_body_augmented, [iteration, chosen_ids],
 759       shape_invariants=[iteration.get_shape(), tensor_shape.TensorShape(
 760           [None])])
 761   return chosen_ids

 764 def update_medoid_per_cluster(pairwise_distances, pairwise_distances_subset,
 765                               labels, chosen_ids, cluster_member_ids,
 766                               cluster_idx, margin_multiplier, margin_type):
 767   """Updates the cluster medoid per cluster.
 768 
 769   Args:
 770     pairwise_distances: 2-D Tensor of pairwise distances.
 771     pairwise_distances_subset: 2-D Tensor of pairwise distances for one cluster.
 772     labels: 1-D Tensor of ground truth cluster assignment.
 773     chosen_ids: 1-D Tensor of cluster centroid indices.
 774     cluster_member_ids: 1-D Tensor of cluster member indices for one cluster.
 775     cluster_idx: Index of this one cluster.
 776     margin_multiplier: multiplication constant.
 777     margin_type: Type of structured margin to use. Default is nmi.
 778 
 779   Returns:
 780     chosen_ids: Updated 1-D Tensor of cluster centroid indices.
 781   """
 782 
 783   def func_cond(iteration, scores_margin):
 784     del scores_margin  # Unused variable scores_margin.
 785     return iteration < num_candidates
 786 
 787   def func_body(iteration, scores_margin):
 788     # swap the current medoid with the candidate cluster member
 789     candidate_medoid = math_ops.to_int32(cluster_member_ids[iteration])
 790     tmp_chosen_ids = update_1d_tensor(chosen_ids, cluster_idx, candidate_medoid)
 791     predictions = get_cluster_assignment(pairwise_distances, tmp_chosen_ids)
 792     metric_score = compute_clustering_score(labels, predictions, margin_type)
 793     pad_before = array_ops.zeros([iteration])
 794     pad_after = array_ops.zeros([num_candidates - 1 - iteration])
 795     return iteration + 1, scores_margin + array_ops.concat(
 796         [pad_before, [1.0 - metric_score], pad_after], 0)
 797 
 798   # pairwise_distances_subset is of size [p, 1, 1, p],
 799   #   the intermediate dummy dimensions at
 800   #   [1, 2] makes this code work in the edge case where p=1.
 801   #   this happens if the cluster size is one.
 802   scores_fac = -1.0 * math_ops.reduce_sum(
 803       array_ops.squeeze(pairwise_distances_subset, [1, 2]), axis=0)
 804 
 805   iteration = array_ops.constant(0)
 806   num_candidates = array_ops.size(cluster_member_ids)
 807   scores_margin = array_ops.zeros([num_candidates])
 808 
 809   _, scores_margin = control_flow_ops.while_loop(func_cond, func_body,
 810                                                  [iteration, scores_margin])
 811   candidate_scores = math_ops.add(scores_fac, margin_multiplier * scores_margin)
 812 
 813   argmax_index = math_ops.to_int32(
 814       math_ops.argmax(candidate_scores, axis=0))
 815 
 816   best_medoid = math_ops.to_int32(cluster_member_ids[argmax_index])
 817   chosen_ids = update_1d_tensor(chosen_ids, cluster_idx, best_medoid)
 818   return chosen_ids
 819

 821 def update_all_medoids(pairwise_distances, predictions, labels, chosen_ids,
 822                        margin_multiplier, margin_type):
 823   """Updates all cluster medoids a cluster at a time.
 824 
 825   Args:
 826     pairwise_distances: 2-D Tensor of pairwise distances.
 827     predictions: 1-D Tensor of predicted cluster assignment.
 828     labels: 1-D Tensor of ground truth cluster assignment.
 829     chosen_ids: 1-D Tensor of cluster centroid indices.
 830     margin_multiplier: multiplication constant.
 831     margin_type: Type of structured margin to use. Default is nmi.
 832 
 833   Returns:
 834     chosen_ids: Updated 1-D Tensor of cluster centroid indices.
 835   """
 836 
 837   def func_cond_augmented_pam(iteration, chosen_ids):
 838     del chosen_ids  # Unused argument.
 839     return iteration < num_classes
 840 
 841   def func_body_augmented_pam(iteration, chosen_ids):
 842     """Call the update_medoid_per_cluster subroutine."""
 843     mask = math_ops.equal(
 844         math_ops.to_int64(predictions), math_ops.to_int64(iteration))
 845     this_cluster_ids = array_ops.where(mask)
 846 
 847     pairwise_distances_subset = array_ops.transpose(
 848         array_ops.gather(
 849             array_ops.transpose(
 850                 array_ops.gather(pairwise_distances, this_cluster_ids)),
 851             this_cluster_ids))
 852 
 853     chosen_ids = update_medoid_per_cluster(pairwise_distances,
 854                                            pairwise_distances_subset, labels,
 855                                            chosen_ids, this_cluster_ids,
 856                                            iteration, margin_multiplier,
 857                                            margin_type)
 858     return iteration + 1, chosen_ids
 859 
 860   unique_class_ids = array_ops.unique(labels)[0]
 861   num_classes = array_ops.size(unique_class_ids)
 862   iteration = array_ops.constant(0)
 863 
 864   _, chosen_ids = control_flow_ops.while_loop(
 865       func_cond_augmented_pam, func_body_augmented_pam, [iteration, chosen_ids])
 866   return chosen_ids

 869 def compute_augmented_facility_locations_pam(pairwise_distances,
 870                                              labels,
 871                                              margin_multiplier,
 872                                              margin_type,
 873                                              chosen_ids,
 874                                              pam_max_iter=5):
 875   """Refine the cluster centroids with PAM local search.
 876 
 877   For fixed iterations, alternate between updating the cluster assignment
 878     and updating cluster medoids.
 879 
 880   Args:
 881     pairwise_distances: 2-D Tensor of pairwise distances.
 882     labels: 1-D Tensor of ground truth cluster assignment.
 883     margin_multiplier: multiplication constant.
 884     margin_type: Type of structured margin to use. Default is nmi.
 885     chosen_ids: 1-D Tensor of initial estimate of cluster centroids.
 886     pam_max_iter: Number of refinement iterations.
 887 
 888   Returns:
 889     chosen_ids: Updated 1-D Tensor of cluster centroid indices.
 890   """
 891   for _ in range(pam_max_iter):
 892     # update the cluster assignment given the chosen_ids (S_pred)
 893     predictions = get_cluster_assignment(pairwise_distances, chosen_ids)
 894 
 895     # update the medoids per each cluster
 896     chosen_ids = update_all_medoids(pairwise_distances, predictions, labels,
 897                                     chosen_ids, margin_multiplier, margin_type)
 898 
 899   return chosen_ids

 902 def compute_gt_cluster_score(pairwise_distances, labels):
 903   """Compute ground truth facility location score.
 904 
 905   Loop over each unique classes and compute average travel distances.
 906 
 907   Args:
 908     pairwise_distances: 2-D Tensor of pairwise distances.
 909     labels: 1-D Tensor of ground truth cluster assignment.
 910 
 911   Returns:
 912     gt_cluster_score: dtypes.float32 score.
 913   """
 914   unique_class_ids = array_ops.unique(labels)[0]
 915   num_classes = array_ops.size(unique_class_ids)
 916   iteration = array_ops.constant(0)
 917   gt_cluster_score = array_ops.constant(0.0, dtype=dtypes.float32)
 918 
 919   def func_cond(iteration, gt_cluster_score):
 920     del gt_cluster_score  # Unused argument.
 921     return iteration < num_classes
 922 
 923   def func_body(iteration, gt_cluster_score):
 924     """Per each cluster, compute the average travel distance."""
 925     mask = math_ops.equal(labels, unique_class_ids[iteration])
 926     this_cluster_ids = array_ops.where(mask)
 927     pairwise_distances_subset = array_ops.transpose(
 928         array_ops.gather(
 929             array_ops.transpose(
 930                 array_ops.gather(pairwise_distances, this_cluster_ids)),
 931             this_cluster_ids))
 932     this_cluster_score = -1.0 * math_ops.reduce_min(
 933         math_ops.reduce_sum(
 934             pairwise_distances_subset, axis=0))
 935     return iteration + 1, gt_cluster_score + this_cluster_score
 936 
 937   _, gt_cluster_score = control_flow_ops.while_loop(
 938       func_cond, func_body, [iteration, gt_cluster_score])
 939   return gt_cluster_score

 942 def cluster_loss(labels,
 943                  embeddings,
 944                  margin_multiplier,
 945                  enable_pam_finetuning=True,
 946                  margin_type='nmi',
 947                  print_losses=False):
 948   """Computes the clustering loss.
 949 
 950   The following structured margins are supported:
 951     nmi: normalized mutual information
 952     ami: adjusted mutual information
 953     ari: adjusted random index
 954     vmeasure: v-measure
 955     const: indicator checking whether the two clusterings are the same.
 956 
 957   Args:
 958     labels: 2-D Tensor of labels of shape [batch size, 1]
 959     embeddings: 2-D Tensor of embeddings of shape
 960       [batch size, embedding dimension]. Embeddings should be l2 normalized.
 961     margin_multiplier: float32 scalar. multiplier on the structured margin term
 962       See section 3.2 of paper for discussion.
 963     enable_pam_finetuning: Boolean, Whether to run local pam refinement.
 964       See section 3.4 of paper for discussion.
 965     margin_type: Type of structured margin to use. See section 3.2 of
 966       paper for discussion. Can be 'nmi', 'ami', 'ari', 'vmeasure', 'const'.
 967     print_losses: Boolean. Option to print the loss.
 968 
 969   Paper: https://arxiv.org/abs/1612.01213.
 970 
 971   Returns:
 972     clustering_loss: A float32 scalar `Tensor`.
 973   Raises:
 974     ImportError: If sklearn dependency is not installed.
 975   """
 976   if not HAS_SKLEARN:
 977     raise ImportError('Cluster loss depends on sklearn.')
 978   pairwise_distances = pairwise_distance(embeddings)
 979   labels = array_ops.squeeze(labels)
 980   all_ids = math_ops.range(array_ops.shape(embeddings)[0])
 981 
 982   # Compute the loss augmented inference and get the cluster centroids.
 983   chosen_ids = compute_augmented_facility_locations(pairwise_distances, labels,
 984                                                     all_ids, margin_multiplier,
 985                                                     margin_type)
 986   # Given the predicted centroids, compute the clustering score.
 987   score_pred = compute_facility_energy(pairwise_distances, chosen_ids)
 988 
 989   # Branch whether to use PAM finetuning.
 990   if enable_pam_finetuning:
 991     # Initialize with augmented facility solution.
 992     chosen_ids = compute_augmented_facility_locations_pam(pairwise_distances,
 993                                                           labels,
 994                                                           margin_multiplier,
 995                                                           margin_type,
 996                                                           chosen_ids)
 997     score_pred = compute_facility_energy(pairwise_distances, chosen_ids)
 998 
 999   # Given the predicted centroids, compute the cluster assignments.
1000   predictions = get_cluster_assignment(pairwise_distances, chosen_ids)
1001 
1002   # Compute the clustering (i.e. NMI) score between the two assignments.
1003   clustering_score_pred = compute_clustering_score(labels, predictions,
1004                                                    margin_type)
1005 
1006   # Compute the clustering score from labels.
1007   score_gt = compute_gt_cluster_score(pairwise_distances, labels)
1008 
1009   # Compute the hinge loss.
1010   clustering_loss = math_ops.maximum(
1011       score_pred + margin_multiplier * (1.0 - clustering_score_pred) - score_gt,
1012       0.0,
1013       name='clustering_loss')
1014   clustering_loss.set_shape([])
1015 
1016   if print_losses:
1017     clustering_loss = logging_ops.Print(
1018         clustering_loss,
1019         ['clustering_loss: ', clustering_loss, array_ops.shape(
1020             clustering_loss)])
1021 
1022   # Clustering specific summary.
1023   summary.scalar('losses/score_pred', score_pred)
1024   summary.scalar('losses/' + margin_type, clustering_score_pred)
1025   summary.scalar('losses/score_gt', score_gt)
1026 
1027   return clustering_loss
