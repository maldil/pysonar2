  37 def rejection_sample(tensors,
  38                      accept_prob_fn,
  39                      batch_size,
  40                      queue_threads=1,
  41                      enqueue_many=False,
  42                      prebatch_capacity=16,
  43                      prebatch_threads=1,
  44                      runtime_checks=False,
  45                      name=None):
  46   """Stochastically creates batches by rejection sampling.
  47 
  48   Each list of non-batched tensors is evaluated by `accept_prob_fn`, to produce
  49   a scalar tensor between 0 and 1. This tensor corresponds to the probability of
  50   being accepted. When `batch_size` tensor groups have been accepted, the batch
  51   queue will return a mini-batch.
  52 
  53   Args:
  54     tensors: List of tensors for data. All tensors are either one item or a
  55         batch, according to enqueue_many.
  56     accept_prob_fn: A python lambda that takes a non-batch tensor from each
  57         item in `tensors`, and produces a scalar tensor.
  58     batch_size: Size of batch to be returned.
  59     queue_threads: The number of threads for the queue that will hold the final
  60       batch.
  61     enqueue_many: Bool. If true, interpret input tensors as having a batch
  62         dimension.
  63     prebatch_capacity: Capacity for the large queue that is used to convert
  64       batched tensors to single examples.
  65     prebatch_threads: Number of threads for the large queue that is used to
  66       convert batched tensors to single examples.
  67     runtime_checks: Bool. If true, insert runtime checks on the output of
  68         `accept_prob_fn`. Using `True` might have a performance impact.
  69     name: Optional prefix for ops created by this function.
  70   Raises:
  71     ValueError: enqueue_many is True and labels doesn't have a batch
  72         dimension, or if enqueue_many is False and labels isn't a scalar.
  73     ValueError: enqueue_many is True, and batch dimension on data and labels
  74         don't match.
  75     ValueError: if a zero initial probability class has a nonzero target
  76         probability.
  77   Returns:
  78     A list of tensors of the same length as `tensors`, with batch dimension
  79     `batch_size`.
  80 
  81   Example:
  82     # Get tensor for a single data and label example.
  83     data, label = data_provider.Get(['data', 'label'])
  84 
  85     # Get stratified batch according to data tensor.
  86     accept_prob_fn = lambda x: (tf.tanh(x[0]) + 1) / 2
  87     data_batch = tf.contrib.training.rejection_sample(
  88         [data, label], accept_prob_fn, 16)
  89 
  90     # Run batch through network.
  91     ...
  92   """
  93   with variable_scope.variable_scope(name, 'rejection_sample', tensors):
  94     tensor_list = ops.convert_n_to_tensor_or_indexed_slices(tensors)
  95     # Reduce the case of a batched example to that of a batch of a single
  96     # example by taking a batch of size one.
  97     if enqueue_many:
  98       # Validate that batch dimension of the input is consistent.
  99       tensor_list = _verify_data_inputs(tensor_list)
 100 
 101       # Make a single queue to hold input examples. Reshape output so examples
 102       # don't have singleton batch dimension.
 103       batched = input_ops.batch(
 104           tensor_list,
 105           batch_size=1,
 106           num_threads=prebatch_threads,
 107           capacity=prebatch_capacity,
 108           enqueue_many=True)
 109       tensor_list = [array_ops.squeeze(x, [0]) for x in batched]
 110 
 111     # Set up a queue containing batches that have the distribution.
 112     cur_prob = accept_prob_fn(tensor_list)
 113     if runtime_checks:
 114       cur_prob = array_ops.identity(
 115           control_flow_ops.with_dependencies([
 116               check_ops.assert_less_equal(0.0, cur_prob),
 117               check_ops.assert_less_equal(cur_prob, 1.0)
 118           ], cur_prob),
 119           name='prob_with_checks')
 120     minibatch = input_ops.maybe_batch(
 121         tensor_list,
 122         keep_input=random_ops.random_uniform([]) < cur_prob,
 123         batch_size=batch_size,
 124         num_threads=queue_threads)
 125 
 126     # Queues return a single tensor if the list of enqueued tensors is one. Since
 127     # we want the type to always be the same, always return a list.
 128     if isinstance(minibatch, ops.Tensor):
 129       minibatch = [minibatch]
 130 
 131     return minibatch

 134 def stratified_sample(tensors,
 135                       labels,
 136                       target_probs,
 137                       batch_size,
 138                       init_probs=None,
 139                       enqueue_many=False,
 140                       queue_capacity=16,
 141                       threads_per_queue=1,
 142                       name=None):
 143   """Stochastically creates batches based on per-class probabilities.
 144 
 145   This method discards examples. Internally, it creates one queue to amortize
 146   the cost of disk reads, and one queue to hold the properly-proportioned
 147   batch.
 148 
 149   Args:
 150     tensors: List of tensors for data. All tensors are either one item or a
 151         batch, according to enqueue_many.
 152     labels: Tensor for label of data. Label is a single integer or a batch,
 153         depending on `enqueue_many`. It is not a one-hot vector.
 154     target_probs: Target class proportions in batch. An object whose type has a
 155         registered Tensor conversion function.
 156     batch_size: Size of batch to be returned.
 157     init_probs: Class proportions in the data. An object whose type has a
 158         registered Tensor conversion function, or `None` for estimating the
 159         initial distribution.
 160     enqueue_many: Bool. If true, interpret input tensors as having a batch
 161         dimension.
 162     queue_capacity: Capacity of the large queue that holds input examples.
 163     threads_per_queue: Number of threads for the large queue that holds input
 164         examples and for the final queue with the proper class proportions.
 165     name: Optional prefix for ops created by this function.
 166   Raises:
 167     ValueError: If `tensors` isn't iterable.
 168     ValueError: `enqueue_many` is True and labels doesn't have a batch
 169         dimension, or if `enqueue_many` is False and labels isn't a scalar.
 170     ValueError: `enqueue_many` is True, and batch dimension on data and labels
 171         don't match.
 172     ValueError: if probs don't sum to one.
 173     ValueError: if a zero initial probability class has a nonzero target
 174         probability.
 175     TFAssertion: if labels aren't integers in [0, num classes).
 176   Returns:
 177     (data_batch, label_batch), where data_batch is a list of tensors of the same
 178         length as `tensors`
 179 
 180   Example:
 181     # Get tensor for a single data and label example.
 182     data, label = data_provider.Get(['data', 'label'])
 183 
 184     # Get stratified batch according to per-class probabilities.
 185     target_probs = [...distribution you want...]
 186     [data_batch], labels = tf.contrib.training.stratified_sample(
 187         [data], label, target_probs)
 188 
 189     # Run batch through network.
 190     ...
 191   """
 192   with ops.name_scope(name, 'stratified_sample', list(tensors) + [labels]):
 193     tensor_list = ops.convert_n_to_tensor_or_indexed_slices(tensors)
 194     labels = ops.convert_to_tensor(labels)
 195     target_probs = ops.convert_to_tensor(target_probs, dtype=dtypes.float32)
 196     # Reduce the case of a single example to that of a batch of size 1.
 197     if not enqueue_many:
 198       tensor_list = [array_ops.expand_dims(tensor, 0) for tensor in tensor_list]
 199       labels = array_ops.expand_dims(labels, 0)
 200 
 201     # If `init_probs` is `None`, set up online estimation of data distribution.
 202     if init_probs is None:
 203       # We use `target_probs` to get the number of classes, so its shape must be
 204       # fully defined at graph construction time.
 205       target_probs.get_shape().assert_is_fully_defined()
 206       init_probs = _estimate_data_distribution(
 207           labels, target_probs.get_shape().num_elements())
 208     else:
 209       init_probs = ops.convert_to_tensor(init_probs, dtype=dtypes.float32)
 210 
 211     # Validate that input is consistent.
 212     tensor_list, labels, [init_probs, target_probs] = _verify_input(
 213         tensor_list, labels, [init_probs, target_probs])
 214 
 215     # Check that all zero initial probabilities also have zero target
 216     # probabilities.
 217     assert_op = control_flow_ops.Assert(
 218         math_ops.reduce_all(
 219             math_ops.logical_or(
 220                 math_ops.not_equal(init_probs, 0),
 221                 math_ops.equal(target_probs, 0))),
 222         ['All classes with zero initial probability must also have zero target '
 223          'probability: ', init_probs, target_probs
 224         ])
 225     init_probs = control_flow_ops.with_dependencies([assert_op], init_probs)
 226 
 227     # Calculate acceptance sampling probabilities.
 228     accept_probs = _calculate_acceptance_probabilities(init_probs, target_probs)
 229     proportion_rejected = math_ops.reduce_sum((1 - accept_probs) * init_probs)
 230     accept_probs = control_flow_ops.cond(
 231         math_ops.less(proportion_rejected, .5),
 232         lambda: accept_probs,
 233         lambda: logging_ops.Print(  # pylint: disable=g-long-lambda
 234             accept_probs, [accept_probs],
 235             message='Proportion of examples rejected by sampler is high.',
 236             first_n=10))
 237 
 238     # Make a single queue to hold input examples. Reshape output so examples
 239     # don't have singleton batch dimension.
 240     batched = input_ops.batch(
 241         tensor_list + [labels],
 242         batch_size=1,
 243         num_threads=threads_per_queue,
 244         capacity=queue_capacity,
 245         enqueue_many=True)
 246     val_list = [array_ops.squeeze(x, [0]) for x in batched[:-1]]
 247     label = array_ops.squeeze(batched[-1], [0])
 248 
 249     # Set up second queue containing batches that have the desired class
 250     # proportions.
 251     cur_prob = array_ops.gather(accept_probs, label)
 252     batched = input_ops.maybe_batch(
 253         val_list + [label],
 254         keep_input=random_ops.random_uniform([]) < cur_prob,
 255         batch_size=batch_size,
 256         num_threads=threads_per_queue)
 257     return batched[:-1], batched[-1]

 260 def _estimate_data_distribution(labels, num_classes, smoothing_constant=10):
 261   """Estimate data distribution as labels are seen."""
 262   # Variable to track running count of classes. Smooth by a nonzero value to
 263   # avoid division-by-zero. Higher values provide more stability at the cost of
 264   # slower convergence.
 265   if smoothing_constant <= 0:
 266     raise ValueError('smoothing_constant must be nonzero.')
 267   num_examples_per_class_seen = variable_scope.variable(
 268       initial_value=[smoothing_constant] * num_classes,
 269       trainable=False,
 270       name='class_count',
 271       dtype=dtypes.int64)
 272 
 273   # Update the class-count based on what labels are seen in batch.
 274   num_examples_per_class_seen = num_examples_per_class_seen.assign_add(
 275       math_ops.reduce_sum(
 276           array_ops.one_hot(
 277               labels, num_classes, dtype=dtypes.int64), 0))
 278 
 279   # Normalize count into a probability.
 280   # NOTE: Without the `+= 0` line below, the test
 281   # `testMultiThreadedEstimateDataDistribution` fails. The reason is that
 282   # before this line, `num_examples_per_class_seen` is a Tensor that shares a
 283   # buffer with an underlying `ref` object. When the `ref` is changed by another
 284   # thread, `num_examples_per_class_seen` changes as well. Since this can happen
 285   # in the middle of the normalization computation, we get probabilities that
 286   # are very far from summing to one. Adding `+= 0` copies the contents of the
 287   # tensor to a new buffer, which will be consistent from the start to the end
 288   # of the normalization computation.
 289   num_examples_per_class_seen += 0
 290   init_prob_estimate = math_ops.truediv(
 291       num_examples_per_class_seen,
 292       math_ops.reduce_sum(num_examples_per_class_seen))
 293 
 294   # Must return float32 (not float64) to agree with downstream `_verify_input`
 295   # checks.
 296   return math_ops.cast(init_prob_estimate, dtypes.float32)

 299 def _verify_data_inputs(tensor_list):
 300   """Verify that batched data inputs are well-formed."""
 301   for tensor in tensor_list:
 302     # Data tensor should have a batch dimension.
 303     tensor_shape = tensor.get_shape().with_rank_at_least(1)
 304 
 305     # Data batch dimensions must be compatible.
 306     tensor_shape[0].assert_is_compatible_with(tensor_list[0].get_shape()[0])
 307 
 308   return tensor_list

 311 def _verify_input(tensor_list, labels, probs_list):
 312   """Verify that batched inputs are well-formed."""
 313   checked_probs_list = []
 314   for probs in probs_list:
 315     # Since number of classes shouldn't change at runtime, probabilities shape
 316     # should be fully defined.
 317     probs.get_shape().assert_is_fully_defined()
 318 
 319     # Probabilities must be 1D.
 320     probs.get_shape().assert_has_rank(1)
 321 
 322     # Probabilities must be nonnegative and sum to one.
 323     tol = 1e-6
 324     prob_sum = math_ops.reduce_sum(probs)
 325     checked_probs = control_flow_ops.with_dependencies([
 326         check_ops.assert_non_negative(probs),
 327         check_ops.assert_less(prob_sum, 1.0 + tol),
 328         check_ops.assert_less(1.0 - tol, prob_sum)
 329     ], probs)
 330     checked_probs_list.append(checked_probs)
 331 
 332   # All probabilities should be the same length.
 333   prob_length = checked_probs_list[0].get_shape().num_elements()
 334   for checked_prob in checked_probs_list:
 335     if checked_prob.get_shape().num_elements() != prob_length:
 336       raise ValueError('Probability parameters must have the same length.')
 337 
 338   # Labels tensor should only have batch dimension.
 339   labels.get_shape().assert_has_rank(1)
 340 
 341   for tensor in tensor_list:
 342     # Data tensor should have a batch dimension.
 343     tensor_shape = tensor.get_shape().with_rank_at_least(1)
 344 
 345     # Data and label batch dimensions must be compatible.
 346     tensor_shape[0].assert_is_compatible_with(labels.get_shape()[0])
 347 
 348   # Data and labels must have the same, strictly positive batch size. Since we
 349   # can't assume we know the batch size at graph creation, add runtime checks.
 350   labels_batch_size = array_ops.shape(labels)[0]
 351   lbl_assert = check_ops.assert_positive(labels_batch_size)
 352 
 353   # Make each tensor depend on its own checks.
 354   labels = control_flow_ops.with_dependencies([lbl_assert], labels)
 355   tensor_list = [
 356       control_flow_ops.with_dependencies([
 357           lbl_assert,
 358           check_ops.assert_equal(array_ops.shape(x)[0], labels_batch_size)
 359       ], x) for x in tensor_list
 360   ]
 361 
 362   # Label's classes must be integers 0 <= x < num_classes.
 363   labels = control_flow_ops.with_dependencies([
 364       check_ops.assert_integer(labels), check_ops.assert_non_negative(labels),
 365       check_ops.assert_less(labels, math_ops.cast(prob_length, labels.dtype))
 366   ], labels)
 367 
 368   return tensor_list, labels, checked_probs_list

 371 def _calculate_acceptance_probabilities(init_probs, target_probs):
 372   """Calculate the per-class acceptance rates.
 373 
 374   Args:
 375     init_probs: The class probabilities of the data.
 376     target_probs: The desired class proportion in minibatches.
 377   Returns:
 378     A list of the per-class acceptance probabilities.
 379 
 380   This method is based on solving the following analysis:
 381 
 382   Let F be the probability of a rejection (on any example).
 383   Let p_i be the proportion of examples in the data in class i (init_probs)
 384   Let a_i is the rate the rejection sampler should *accept* class i
 385   Let t_i is the target proportion in the minibatches for class i (target_probs)
 386 
 387   ```
 388   F = sum_i(p_i * (1-a_i))
 389     = 1 - sum_i(p_i * a_i)     using sum_i(p_i) = 1
 390   ```
 391 
 392   An example with class `i` will be accepted if `k` rejections occur, then an
 393   example with class `i` is seen by the rejector, and it is accepted. This can
 394   be written as follows:
 395 
 396   ```
 397   t_i = sum_k=0^inf(F^k * p_i * a_i)
 398       = p_i * a_j / (1 - F)    using geometric series identity, since 0 <= F < 1
 399       = p_i * a_i / sum_j(p_j * a_j)        using F from above
 400   ```
 401 
 402   Note that the following constraints hold:
 403   ```
 404   0 <= p_i <= 1, sum_i(p_i) = 1
 405   0 <= a_i <= 1
 406   0 <= t_i <= 1, sum_i(t_i) = 1
 407   ```
 408 
 409 
 410   A solution for a_i in terms of the other variables is the following:
 411     ```a_i = (t_i / p_i) / max_i[t_i / p_i]```
 412   """
 413   # Make list of t_i / p_i.
 414   ratio_l = target_probs / init_probs
 415 
 416   # Replace NaNs with 0s.
 417   ratio_l = array_ops.where(
 418       math_ops.is_nan(ratio_l), array_ops.zeros_like(ratio_l), ratio_l)
 419 
 420   # Calculate list of acceptance probabilities.
 421   max_ratio = math_ops.reduce_max(ratio_l)
 422   return ratio_l / max_ratio
