  28 class RandomStrategy(object):
  29   """Returns a random PS task for op placement.
  30 
  31   This may perform better than the default round-robin placement if you
  32   have a large number of variables. Depending on your architecture and
  33   number of parameter servers, round-robin can lead to situations where
  34   all of one type of variable is placed on a single PS task, which may
  35   lead to contention issues.
  36 
  37   This strategy uses a hash function on the name of each op for deterministic
  38   placement.
  39   """
  40 
  41   def __init__(self, num_ps_tasks, seed=0):
  42     """Creates a new `RandomStrategy`."""
  43     self._num_tasks = num_ps_tasks
  44     self._seed = seed
  45 
  46   def __call__(self, op):
  47     """Chooses a ps task index for the given `Operation`."""
  48     key = "%s_%d" % (op.name, self._seed)
  49     key = key.encode("utf-8")
  50     # Use MD5 instead of Python's built-in hash() to get consistent outputs
  51     # between runs.
  52     n = int(hashlib.md5(key).hexdigest(), 16)
  53     return int(n % self._num_tasks)

  56 class GreedyLoadBalancingStrategy(object):
  57   """Returns the least-loaded ps task for op placement.
  58 
  59   The load is calculated by a user-specified load function passed in at
  60   construction.  There are no units for load, and the load function is
  61   responsible for providing an internally consistent measure.
  62 
  63   Note that this strategy is very sensitive to the exact order in which
  64   ps ops (typically variables) are created, as it greedily places ops
  65   on the least-loaded ps at the point each op is processed.
  66 
  67   One reasonable heuristic is the `byte_size_load_fn`, which
  68   estimates load as the number of bytes that would be used to store and
  69   transmit the entire variable.  More advanced load functions
  70   could consider the difference in access patterns across ops, or trade
  71   off CPU-intensive ops with RAM-intensive ops with network bandwidth.
  72 
  73   This class is intended to be used as a `ps_strategy` in
  74   `tf.train.replica_device_setter`.
  75   """
  76 
  77   def __init__(self, num_tasks, load_fn):
  78     """Create a new `LoadBalancingStrategy`.
  79 
  80     Args:
  81       num_tasks: Number of ps tasks to cycle among.
  82       load_fn: A callable that takes an `Operation` and returns a
  83         numeric load value for that op.
  84     """
  85     self._num_tasks = num_tasks
  86     self._load_fn = load_fn
  87     self._ps_loads = np.zeros(num_tasks)
  88 
  89   def __call__(self, op):
  90     """Choose a ps task index for the given `Operation`.
  91 
  92     Args:
  93       op: A `Operation` to be placed on ps.
  94 
  95     Returns:
  96       The next ps task index to use for the `Operation`. Greedily
  97       places the op on the least-loaded ps task so far, as determined
  98       by the load function.
  99     """
 100     task = np.argmin(self._ps_loads)
 101     self._ps_loads[task] += self._load_fn(op)
 102     return task

 105 def byte_size_load_fn(op):
 106   """Load function that computes the byte size of a single-output `Operation`.
 107 
 108   This is intended to be used with `"Variable"` ops, which have a single
 109   `Tensor` output with the contents of the variable.  However, it can also be
 110   used for calculating the size of any op that has a single output.
 111 
 112   Intended to be used with `GreedyLoadBalancingStrategy`.
 113 
 114   Args:
 115     op: An `Operation` with a single output, typically a "Variable" op.
 116 
 117   Returns:
 118     The number of bytes in the output `Tensor`.
 119 
 120   Raises:
 121     ValueError: if `op` does not have a single output, or if the shape of the
 122       single output is not fully-defined.
 123   """
 124   if len(op.outputs) != 1:
 125     raise ValueError("Op %s must have a single output" % op)
 126   output = op.outputs[0]
 127   elem_size = output.dtype.size
 128   shape = output.get_shape()
 129   if not shape.is_fully_defined():
 130     # Due to legacy behavior, scalar "Variable" ops have output Tensors that
 131     # have unknown shape when the op is created (and hence passed to this
 132     # load function for placement), even though the scalar shape is set
 133     # explicitly immediately afterward.
 134     shape = tensor_shape.TensorShape(op.get_attr("shape"))
 135   shape.assert_is_fully_defined()
 136   return shape.num_elements() * elem_size
