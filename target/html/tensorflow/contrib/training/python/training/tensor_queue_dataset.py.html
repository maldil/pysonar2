  32 class _PrependFromQueueAndPaddedBatchDataset(dataset_ops.UnaryDataset):
  33   """A `Dataset` that prepends a queue to another `Dataset`.
  34 
  35   A vector of handles to the queue is returned as the first component of
  36   the associated iterator.  This vector can be passed to
  37   `enqueue_in_queue_dataset` to add new elements to the queue.
  38   """
  39 
  40   def __init__(self, input_dataset, batch_size, padded_shapes, padding_values):
  41     """Initialize `PrependFromQueueAndPaddedBatchDataset`."""
  42     super(_PrependFromQueueAndPaddedBatchDataset, self).__init__(input_dataset)
  43     if sparse.any_sparse(input_dataset.output_classes):
  44       raise TypeError(
  45           "Batching of padded sparse tensors is not currently supported")
  46     self._input_dataset = input_dataset
  47     self._batch_size = ops.convert_to_tensor(
  48         batch_size, dtype=dtypes.int64, name="batch_size")
  49     if padded_shapes is None:
  50       self._padded_shapes = nest.map_structure(
  51           convert.partial_shape_to_tensor, input_dataset.output_shapes)
  52     else:
  53       self._padded_shapes = nest.map_structure_up_to(
  54           input_dataset.output_shapes, convert.partial_shape_to_tensor,
  55           padded_shapes)
  56     # pylint: disable=protected-access
  57     padding_values = (
  58         padding_values if padding_values is not None else
  59         dataset_ops._default_padding(input_dataset))
  60     self._padding_values = nest.map_structure_up_to(
  61         input_dataset.output_shapes, dataset_ops._padding_value_to_tensor,
  62         padding_values, input_dataset.output_types)
  63     # pylint: enable=protected-access
  64 
  65   def _as_variant_tensor(self):
  66     # pylint: disable=protected-access
  67     return gen_dataset_ops.prepend_from_queue_and_padded_batch_dataset(
  68         self._input_dataset._as_variant_tensor(),
  69         batch_size=self._batch_size,
  70         padded_shapes=[
  71             ops.convert_to_tensor(s, dtype=dtypes.int64)
  72             for s in nest.flatten(self._padded_shapes)
  73         ],
  74         padding_values=nest.flatten(self._padding_values),
  75         output_shapes=nest.flatten(
  76             sparse.as_dense_shapes(self.output_shapes, self.output_classes)))
  77     # pylint: enable=protected-access
  78 
  79   @property
  80   def output_classes(self):
  81     return (ops.Tensor, self._input_dataset.output_classes)
  82 
  83   def _as_batch_shape(self, shape_like):
  84     return tensor_shape.vector(None).concatenate(
  85         tensor_util.constant_value_as_shape(shape_like))
  86 
  87   @property
  88   def output_shapes(self):
  89     # First output is a variant representing the Queue
  90     return (tensor_shape.vector(None),
  91             nest.map_structure(self._as_batch_shape, self._padded_shapes))
  92 
  93   @property
  94   def output_types(self):
  95     # First output is a variant representing the Queue
  96     return (dtypes.variant, self._input_dataset.output_types)

  99 def prepend_from_queue_and_padded_batch_dataset(batch_size,
 100                                                 padding_values=None,
 101                                                 padded_shapes=None):
 102   """A transformation that prepends a queue to a `Dataset` and batches results.
 103 
 104   A vector of handles to the queue is returned as the first component of the
 105   associated iterator.  This vector can be passed to `enqueue_in_queue_dataset`
 106   to add new elements to the queue.
 107 
 108   Below is an example of how this dataset might be used to split incoming
 109   variable-length sequences into "head" and "rest" parts, where "rest" parts
 110   are re-enqueued back into the dataset.  A more realistic example would
 111   perform some calculation on the "head" and modify some components of "rest"
 112   with the result (before re-enqueueing).
 113 
 114   ```python
 115   dataset = tf.data.Dataset.from_tensor_slices([2*x for x in range(10)])
 116   # Make a dataset of variable-length vectors and their lengths.
 117   dataset = dataset.map(lambda count: (count, tf.ones((count,))))
 118   # Emit a queue we can prepend to, and counts/values as padded batch.
 119   dataset = dataset.apply(
 120       tf.contrib.training.prepend_from_queue_and_padded_batch_dataset(
 121         batch_size=10))
 122   dataset = dataset.prefetch(1)
 123 
 124   iterator = dataset.make_one_shot_iterator()
 125   queue, (count, padded_value) = iterator.get_next()
 126 
 127   # Split the padded_value into two pieces: head and rest
 128   rest_indices = tf.squeeze(tf.where(count > 3), axis=1)
 129   bound = tf.minimum(3, tf.reduce_max(count))
 130   value_head = padded_value[:, :bound]
 131   count_rest = tf.gather(count - 3, rest_indices)
 132   value_rest = tf.gather(padded_value[:, bound:], rest_indices)
 133   queue_rest = tf.gather(queue, rest_indices)
 134   enqueue_rest_op = tf.contrib.training.enqueue_in_queue_dataset(
 135     queue_rest, (count_rest, value_rest))
 136   with tf.control_dependencies([enqueue_rest_op]):
 137     calculation = fn(value_head)
 138 
 139   while True:  # Will raise OutOfRange when finished with all pieces.
 140     session.run(calculation)
 141   ```
 142 
 143   Args:
 144     batch_size: `int64` scalar tensor.  The batch size to use when performing
 145       padded batching.
 146     padding_values: (optional) Nested tuple of scalar tensors.  If provided,
 147       the structure and dtypes of padding_values should match that of
 148       incoming dataset's `output_types`.
 149     padded_shapes: (optional) Nested tuple of `int64` vector tensors.
 150       If provided, the structure must match that of the incoming dataset's
 151       `output_types`.  If not provided, the incoming dataset's `output_shapes`
 152       is used.  Any unknown (`None` or `-1`) dimensions in the shapes are
 153       treated as being unique per-batch: for each batch time, an unknown
 154       dimension is replaced with the maximum given value of this dimension
 155       across all tensors for the given component in the batch.
 156 
 157   Returns:
 158     A `Dataset` transformation function, which can be passed to
 159     `tf.data.Dataset.apply`.
 160   """
 161 
 162   def _apply_fn(dataset):
 163     return _PrependFromQueueAndPaddedBatchDataset(
 164         dataset,
 165         batch_size=batch_size,
 166         padding_values=padding_values,
 167         padded_shapes=padded_shapes)
 168 
 169   return _apply_fn

 172 def enqueue_in_queue_dataset(queue, components):
 173   """Enqueue components into queue from `PrependFromQueueAndPaddedBatchDataset`.
 174 
 175   The components' dtypes and shapes must be compatible with the `output_shapes`
 176   attribute of the `dataset` created by
 177   `prepend_from_queue_and_padded_batch_dataset`.  This operation supports both
 178   non-batched and batched modes.
 179 
 180   For more details, see the example in the docstring for
 181   `prepend_from_queue_and_padded_batch_dataset`.
 182 
 183   Args:
 184     queue: `variant` scalar or vector tensor.
 185       The tensor emitted by the first component of the iterator associated with
 186       `prepend_from_queue_and_padded_batch_dataset`.  If this is a scalar,
 187       then the `components` input tensors should not have a prepended batch
 188       dimension.
 189     components: Nested tuple of tensors, each with a leading batch dimension
 190       if `queue` is a vector.  The structure, dtypes, and shapes
 191       (excluding batch dimension) must match the nested tuples
 192       `dataset.output_types[1]` and `dataset.output_shapes[1]` (the non-queue
 193       output types and shapes) of the `dataset` emitted by
 194       the original `prepend_from_queue_and_padded_batch_dataset` call.
 195 
 196   Returns:
 197     An `Operation` that enqueues `components` into the dataset(s) associated
 198     with entries of `queue`.
 199   """
 200   return gen_dataset_ops.enqueue_in_queue_dataset(
 201       queue=queue, components=tf_nest.flatten(components))
