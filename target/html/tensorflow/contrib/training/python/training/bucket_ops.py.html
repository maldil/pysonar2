  56 def _validate_bucket(tensor_list):
  57   tensor_list = ops.convert_n_to_tensor_or_indexed_slices(tensor_list)
  58   if not tensor_list:
  59     raise ValueError("Expected at least one tensor in bucket().")
  60   return tensor_list

  63 def bucket(tensors,
  64            which_bucket,
  65            batch_size,
  66            num_buckets,
  67            num_threads=1,
  68            capacity=32,
  69            bucket_capacities=None,
  70            shapes=None,
  71            dynamic_pad=False,
  72            allow_smaller_final_batch=False,
  73            keep_input=True,
  74            shared_name=None,
  75            name=None):
  76   """Lazy bucketing of input tensors according to `which_bucket`.
  77 
  78   The argument `tensors` can be a list or a dictionary of tensors.
  79   The value returned by the function will be of the same type
  80   as `tensors`.
  81 
  82   The tensors entering this function are put into the bucket given by
  83   `which_bucket`.  Each bucket has its own queue.  When a bucket contains
  84   `batch_size` elements, this minibatch is pushed onto a top queue.  The
  85   tensors returned from this function are a the result of dequeueing the
  86   next minibatch from this top queue.
  87 
  88   This function is implemented using several queues. A `QueueRunner` for the
  89   queues is added to the current `Graph`'s `QUEUE_RUNNER` collection.
  90 
  91   As the returned tensors are the result of a dequeue operation, evaluating
  92   them will throw a `tf.errors.OutOfRangeError` when the input queue is
  93   exhausted.  If these tensors are feeding another input queue, its queue runner
  94   will catch this exception, however, if they are used in your main thread
  95   you are responsible for catching this yourself.
  96 
  97   *N.B.:* If `dynamic_pad` is `False`, you must ensure that either
  98   (i) the `shapes` argument is passed, or (ii) all of the tensors in
  99   `tensors` must have fully-defined shapes. `ValueError` will be
 100   raised if neither of these conditions holds.
 101 
 102   If `dynamic_pad` is `True`, it is sufficient that the *rank* of the
 103   tensors is known, but individual dimensions may have shape `None`.
 104   In this case, for each enqueue the dimensions with value `None`
 105   may have a variable length; upon dequeue, the output tensors will be padded
 106   on the right to the maximum shape of the tensors in the current minibatch.
 107   For numbers, this padding takes value 0.  For strings, this padding is
 108   the empty string.  See `PaddingFIFOQueue` for more info.
 109 
 110   If `allow_smaller_final_batch` is `True`, a smaller batch value than
 111   `batch_size` is returned when the queues are closed and there are not enough
 112   elements to fill the batch, otherwise the pending elements are discarded.
 113   In addition, all output tensors' static shapes, as accessed via the
 114   `get_shape()` method will have a 0th `Dimension` value of `None`, and
 115   operations that depend on fixed batch_size would fail.
 116 
 117   Args:
 118     tensors: The list or dictionary of tensors, representing a single element,
 119       to bucket.  Nested lists are not supported.
 120     which_bucket: An `int32` scalar Tensor taking a value in `[0, num_buckets)`.
 121     batch_size: The new batch size pulled from the queue (all queues will have
 122       the same size).  If a list is passed in then each bucket will have a
 123       different batch_size.
 124       (python int, int32 scalar or iterable of integers of length num_buckets).
 125     num_buckets: A python integer, the number of buckets.
 126     num_threads: An integer.  The number of threads enqueuing `tensors`.
 127     capacity: An integer. The maximum number of minibatches in the top queue,
 128       and also (by default) the maximum number of elements within each bucket.
 129     bucket_capacities: (Optional) None or a list of integers, the capacities of
 130       each bucket. If None, capacity is used (default). If specified, it must
 131       be a list of integers of length num_buckets: the i-th element is used
 132       as capacity for the i-th bucket queue.
 133     shapes: (Optional) The shapes for each example.  Defaults to the
 134       inferred shapes for `tensors`.
 135     dynamic_pad: Boolean.  Allow variable dimensions in input shapes.
 136       The given dimensions are padded upon dequeue so that tensors within a
 137       batch have the same shapes.
 138     allow_smaller_final_batch: (Optional) Boolean. If `True`, allow the final
 139       batches to be smaller if there are insufficient items left in the queues.
 140     keep_input: A `bool` scalar Tensor.  If provided, this tensor controls
 141       whether the input is added to the queue or not.  If it evaluates `True`,
 142       then `tensors` are added to the bucket; otherwise they are dropped.  This
 143       tensor essentially acts as a filtering mechanism.
 144     shared_name: (Optional). If set, the queues will be shared under the given
 145       name across multiple sessions.
 146     name: (Optional) A name for the operations.
 147 
 148   Returns:
 149     A tuple `(bucket, outputs)` where `bucket` is
 150     a `int32` scalar tensor and `outputs` is a list or
 151     dictionary of batched outputs corresponding to elements of `tensors`.
 152     Every step will receive a new bucket of outputs.
 153 
 154   Raises:
 155     ValueError: If the `shapes` are not specified, and cannot be
 156       inferred from the elements of `tensors` or if batch_size is a sequence
 157       but its length != num_buckets. Also if bucket_capacities is not None but
 158       its length != num_buckets.
 159   """
 160   batch_size_per_bucket = False
 161   if isinstance(batch_size, (list, tuple)):
 162     batch_size_per_bucket = True
 163     if len(batch_size) != num_buckets:
 164       raise ValueError(
 165           "If batch_size is a list it must have num_buckets elements")
 166   else:
 167     batch_size = [batch_size] * num_buckets
 168 
 169   if bucket_capacities is None:
 170     bucket_capacities = [capacity] * num_buckets
 171   if len(bucket_capacities) != num_buckets:
 172     raise ValueError(
 173         "The list bucket_capacities (%s) must have exactly num_buckets (%d) "
 174         "elements." % (str(bucket_capacities), num_buckets))
 175 
 176   tensor_list = _as_tensor_list(tensors)
 177   with ops.name_scope(name, "bucket", tensor_list) as name:
 178     tensor_list = _validate_bucket(tensor_list)
 179     keep_input = _validate_keep_input(keep_input, enqueue_many=False)
 180     (tensor_list, sparse_info) = _store_sparse_tensors(
 181         tensor_list, enqueue_many=False, keep_input=keep_input)
 182 
 183     # Round-trip batch_size to a tensor, and possibly back
 184     for i, bucket_batch_size in enumerate(batch_size):
 185       bucket_batch_size = ops.convert_to_tensor(
 186           bucket_batch_size, dtype=dtypes.int32, name="batch_size")
 187       static_batch_size = tensor_util.constant_value(bucket_batch_size)
 188       batch_size[i] = (static_batch_size if static_batch_size is not None else
 189                        bucket_batch_size)
 190 
 191     types = _dtypes([tensor_list])
 192     shapes = _shapes([tensor_list], shapes, enqueue_many=False)
 193 
 194     which_bucket = ops.convert_to_tensor(
 195         which_bucket, dtype=dtypes.int32, name="which_bucket")
 196 
 197     queue_creator = _which_queue(dynamic_pad)
 198     bucket_queues = []
 199     for i in range(num_buckets):
 200       shared_name_i = ("%s_%d" % (shared_name, i) if shared_name is not None
 201                        else None)
 202       bucket_queues.append(
 203           queue_creator(
 204               capacity=bucket_capacities[i],
 205               dtypes=types,
 206               shapes=shapes,
 207               shared_name=shared_name_i,
 208               name="bucket_queue_%d" % i))
 209 
 210     maybe_static_batch_size = (
 211         None if (allow_smaller_final_batch or batch_size_per_bucket)
 212         else static_batch_size)
 213 
 214     bucket_shapes = [
 215         tensor_shape.vector(maybe_static_batch_size).concatenate(s)
 216         for s in bucket_queues[0].shapes
 217     ]
 218     # top_queue is a PaddingFIFOQueue even if the bucket queues are regular FIFO
 219     # queues because if we use allow_smaller_final_batch, shapes will
 220     # contain Nones in their first entry; as a result, a regular
 221     # FIFOQueue would die when being passed shapes that are not fully defined.
 222     top_queue = data_flow_ops.PaddingFIFOQueue(
 223         capacity=capacity,
 224         dtypes=[dtypes.int32] + types,
 225         shapes=[tensor_shape.scalar()] + bucket_shapes,
 226         shared_name=shared_name,
 227         name="top_queue")
 228 
 229     def enqueue_which():
 230       """Return an op that enqueues conditionally in one of the queues."""
 231       def enqueue_single(i):
 232         return bucket_queues[i].enqueue(tensor_list)
 233 
 234       enqueues = [
 235           control_flow_ops.cond(
 236               math_ops.equal(which_bucket, i),
 237               functools.partial(enqueue_single, i), control_flow_ops.no_op)
 238           for i in range(num_buckets)
 239       ]
 240       return control_flow_ops.group(*enqueues, name="group_enqueues")
 241 
 242     maybe_enqueue = utils.smart_cond(
 243         keep_input,
 244         enqueue_which,
 245         control_flow_ops.no_op)
 246 
 247     bucket_enqueue_ops = [maybe_enqueue] * num_threads
 248 
 249     if allow_smaller_final_batch:
 250       which_dequeue = lambda q: q.dequeue_up_to
 251     else:
 252       which_dequeue = lambda q: q.dequeue_many
 253 
 254     def make_list(t):
 255       if isinstance(t, (list, tuple)):
 256         return t
 257       else:
 258         return [t]
 259 
 260     enqueues_to_top = [
 261         top_queue.enqueue(
 262             [constant_op.constant(i)] + make_list(which_dequeue(q)(
 263                 bs, name="read_bucket_%d" % i)),
 264             name="enqueue_from_bucket_%d" % i)
 265         for i, (q, bs) in enumerate(zip(bucket_queues, batch_size))
 266     ]
 267 
 268     queue_runner.add_queue_runner(
 269         queue_runner.QueueRunner(
 270             bucket_queues[0], enqueues_to_top,
 271             close_op=top_queue.close(),
 272             cancel_op=top_queue.close(cancel_pending_enqueues=True),
 273             queue_closed_exception_types=(errors.OutOfRangeError,
 274                                           errors.CancelledError)))
 275     queue_runner.add_queue_runner(
 276         queue_runner.QueueRunner(
 277             top_queue,
 278             bucket_enqueue_ops,
 279             close_op=control_flow_ops.group(
 280                 *[q.close() for q in bucket_queues]),
 281             cancel_op=control_flow_ops.group(
 282                 *[q.close(cancel_pending_enqueues=True)
 283                   for q in bucket_queues]),
 284             queue_closed_exception_types=(errors.OutOfRangeError,
 285                                           errors.CancelledError)))
 286 
 287     for q in bucket_queues:
 288       summary.scalar("bucket/%s/size" % q.name,
 289                      math_ops.cast(top_queue.size(), dtypes.float32))
 290     summary.scalar("bucket/%s/fraction_of_%d_full" % (top_queue.name, capacity),
 291                    math_ops.cast(top_queue.size(), dtypes.float32) *
 292                    (1. / capacity))
 293 
 294     dequeued = top_queue.dequeue(name="dequeue_top")
 295     which_bucket_dequeued = dequeued[0]
 296     dequeued = dequeued[1:]
 297     if len(dequeued) == 1:
 298       dequeued = dequeued[0]
 299     dequeued = _restore_sparse_tensors(dequeued, sparse_info)
 300     return (which_bucket_dequeued, _as_original_type(tensors, dequeued))

 303 def bucket_by_sequence_length(input_length,
 304                               tensors,
 305                               batch_size,
 306                               bucket_boundaries,
 307                               num_threads=1,
 308                               capacity=32,
 309                               bucket_capacities=None,
 310                               shapes=None,
 311                               dynamic_pad=False,
 312                               allow_smaller_final_batch=False,
 313                               keep_input=True,
 314                               shared_name=None,
 315                               name=None):
 316   """Lazy bucketing of inputs according to their length.
 317 
 318   This method calls `tf.contrib.training.bucket` under the hood, after first
 319   subdividing the bucket boundaries into separate buckets and identifying which
 320   bucket the given `input_length` belongs to.  See the documentation for
 321   `which_bucket` for details of the other arguments.
 322 
 323   Args:
 324     input_length: `int32` scalar `Tensor`, the sequence length of tensors.
 325     tensors: The list or dictionary of tensors, representing a single element,
 326       to bucket.  Nested lists are not supported.
 327     batch_size: The new batch size pulled from the queue (all queues will have
 328       the same size).  If a list is passed in then each bucket will have a
 329       different batch_size.
 330       (python int, int32 scalar or iterable of integers of length num_buckets).
 331     bucket_boundaries: int list, increasing non-negative numbers.
 332       The edges of the buckets to use when bucketing tensors.  Two extra buckets
 333       are created, one for `input_length < bucket_boundaries[0]` and
 334       one for `input_length >= bucket_boundaries[-1]`.
 335     num_threads: An integer.  The number of threads enqueuing `tensors`.
 336     capacity: An integer. The maximum number of minibatches in the top queue,
 337       and also the maximum number of elements within each bucket.
 338     bucket_capacities: (Optional) None or a list of integers, the capacities of
 339       each bucket. If None, capacity is used (default). If specified, it must
 340       be a list of integers of length one larger than bucket_boundaries.
 341       Its i-th element is used as capacity for the i-th bucket queue.
 342     shapes: (Optional) The shapes for each example.  Defaults to the
 343       inferred shapes for `tensors`.
 344     dynamic_pad: Boolean.  Allow variable dimensions in input shapes.
 345       The given dimensions are padded upon dequeue so that tensors within a
 346       batch have the same shapes.
 347     allow_smaller_final_batch: (Optional) Boolean. If `True`, allow the final
 348       batches to be smaller if there are insufficient items left in the queues.
 349     keep_input: A `bool` scalar Tensor.  If provided, this tensor controls
 350       whether the input is added to the queue or not.  If it evaluates `True`,
 351       then `tensors` are added to the bucket; otherwise they are dropped.  This
 352       tensor essentially acts as a filtering mechanism.
 353     shared_name: (Optional). If set, the queues will be shared under the given
 354       name across multiple sessions.
 355     name: (Optional) A name for the operations.
 356 
 357   Returns:
 358     A tuple `(sequence_length, outputs)` where `sequence_length` is
 359     a 1-D `Tensor` of size `batch_size` and `outputs` is a list or dictionary
 360     of batched, bucketed, outputs corresponding to elements of `tensors`.
 361 
 362   Raises:
 363     TypeError: if `bucket_boundaries` is not a list of python integers.
 364     ValueError: if `bucket_boundaries` is empty or contains non-increasing
 365       values or if batch_size is a list and it's length doesn't equal the number
 366       of buckets.
 367   """
 368   tensor_list = _as_tensor_list(tensors)
 369   if not isinstance(bucket_boundaries, (list, tuple)):
 370     raise TypeError(
 371         "bucket_boundaries must be a list or tuple, but received: %s" %
 372         bucket_boundaries)
 373   if not bucket_boundaries:
 374     raise ValueError("bucket_boundaries must not be empty")
 375   for (s, e) in zip(bucket_boundaries[:-1], bucket_boundaries[1:]):
 376     if not isinstance(s, int) or not isinstance(e, int):
 377       raise TypeError("bucket boundaries must be integers, but saw: %s and %s" %
 378                       (s, e))
 379     if s >= e:
 380       raise ValueError(
 381           "Buckets must contain sequential increasing lengths, but saw: "
 382           "%d before %d" % (s, e))
 383 
 384   with ops.name_scope(name, "bucket_by_sequence_length",
 385                       [input_length] + tensor_list) as name:
 386     input_length = ops.convert_to_tensor(
 387         input_length, dtype=dtypes.int32, name="input_length")
 388     # Bucketing conditions are:
 389     #   l < b[0]
 390     #   b[0] <= l < b[1]
 391     #   b[1] <= l < b[2]
 392     #   ...
 393     #   b[N-2] <= l < b[N-1]
 394     #   b[N-1] <= l
 395     # Equivalent to:
 396     #   [-inf, b[0], b[1], ..., b[N-1]] <= l < [b[0], b[1], ..., b[N-1], inf]
 397     buckets_min = [np.iinfo(np.int32).min] + list(bucket_boundaries)
 398     buckets_max = list(bucket_boundaries) + [np.iinfo(np.int32).max]
 399     conditions_c = math_ops.logical_and(
 400         math_ops.less_equal(buckets_min, input_length),
 401         math_ops.less(input_length, buckets_max))
 402     which_bucket = math_ops.reduce_min(array_ops.where(conditions_c))
 403     which_bucket = math_ops.to_int32(which_bucket)
 404 
 405     if shapes is not None:
 406       shapes = [tensor_shape.scalar()] + shapes
 407 
 408     _, dequeued = bucket(
 409         tensors=[input_length] + tensor_list,
 410         which_bucket=which_bucket,
 411         batch_size=batch_size,
 412         num_buckets=len(bucket_boundaries) + 1,
 413         num_threads=num_threads,
 414         capacity=capacity,
 415         bucket_capacities=bucket_capacities,
 416         shapes=shapes,
 417         dynamic_pad=dynamic_pad,
 418         allow_smaller_final_batch=allow_smaller_final_batch,
 419         keep_input=keep_input,
 420         shared_name=shared_name)
 421 
 422     return (dequeued[0], _as_original_type(tensors, dequeued[1:]))
