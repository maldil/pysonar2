  15 """Recurrent computation.
  16 
  17 The main interface of this module is Recurrent().
  18 A recurrent computation describes an auto-regressive process, where outputs
  19 of one time step are fed to the output of the next time step.
  20 
  21 This module uses:
  22   theta: the "weights" each RNN uses.
  23   state0: the initial state of each RNN.
  24   cell_fn: A python function describing RNN cell. It must has the following
  25     signature:
  26          cell_fn: (theta, state0, inputs) -> (state1, extras)
  27     state1 is the next RNN state, extras are computed by cell_fn
  28     and the library forwards extras to cell_fn's gradient function.
  29   cell_grad: A python function describing the backprop gradient function
  30     for the RNN cell. It must has the following signature:
  31          cell_grad: (theta, state0, inputs, extras, dstate1) -> (
  32                   dtheta, dstate0, dinputs)
  33     dstate1 is what the backprop algorithm provides representing
  34     gradients of state1 w.r.t. the final loss.
  35 
  36 In this module, we handle structures of tensors for theta, state0, inputs,
  37 and extras. The structure is an arbitrarily nested python structure, such
  38 as a dictionary of named tuples.
  39 
  40 Because the computation is a left-to-right chain, a single in-place accumulator
  41 can be used rather than a stack. Thus a special gradient was written to reduce
  42 unnecessary memory usage.
  43 """

  68 def _Index(struct, index):
  69   """Returns a structure with `x[index]` for each tensor `x` in the structure.
  70 
  71   Args:
  72     struct: A structure of tensors.
  73     index: A scalar integer tensor. Performance is better if `index` is
  74       on the host memory.
  75 
  76   Returns:
  77     A structure of tensors congruent to `struct`.
  78     For each key in `ret`, `rets[key] = struct[key][index]`.
  79   """
  80   index = ops.convert_to_tensor(index)
  81   index.get_shape().assert_has_rank(0)
  82   return nest.map_structure(lambda x: array_ops.gather(x, index), struct)

  85 def _Update(struct_acc, struct_x, t):
  86   """Updates t-th row in accumulators.
  87 
  88   Args:
  89     struct_acc: The accumulators. A structure of tensors.
  90     struct_x: The new values. A structure of tensors congruent to `struct_acc`.
  91     t: A scalar integer. Performance is better if `t` is on the device
  92       memory.
  93 
  94   Returns:
  95     A structure of tensors. Say, ret is a returned dictionary. Then, for
  96     each key, we have:
  97       ret[key] = struct_acc[key];
  98       ret[key][t, :] = struct_x[key]
  99   """
 100   to_skip_update = set()
 101   acc_lst = nest.flatten(struct_acc)
 102   x_lst = nest.flatten(struct_x)
 103   t = math_ops.to_int32([t])  # tf.to_int32 casts on-device tensors.
 104   lst = []
 105   for acc, x in zip(acc_lst, x_lst):
 106     if acc in to_skip_update:
 107       # Until b/62105730 is fixed, we need to avoid inplace update for tensors
 108       # of rank 1.  could reshape to handle it, but we don't really need the
 109       # values applied to these, so just skip their modification.
 110       lst += [acc]
 111     else:
 112       lst += [alias_inplace_update(acc, t, array_ops.expand_dims(x, 0))]
 113   return nest.pack_sequence_as(struct_acc, lst)

 116 def _SeqLenDim(struct):
 117   """Returns the 0-th dim size of tensors in a structure of tensors.
 118 
 119   This is the max sequence length according to the shape of the inputs.
 120 
 121   Args:
 122     struct: A structure of tensors. Every tensor's 0-th dim has the same size.
 123 
 124   Returns:
 125     A scalar tensor which is the size of 0-th dim of every tensors in struct.
 126   """
 127   xs = nest.flatten(struct)
 128   assert xs
 129   dim0 = array_ops.shape(xs[0])[0]
 130   return dim0

 138 def _Pack(elements, struct_template):
 139   """Packs the list of tensors according to the structure.
 140 
 141   In the event that `elements` should be a scalar, `struct_template` must
 142   contain exactly one non-trivial element (for instance, `[[], {'x':elt}]`).
 143 
 144   Args:
 145     elements: Elements to be packed. A list of tensor, or a single tensor.
 146     struct_template: The container structure in which to pack them.
 147   Returns:
 148     A python structure of the same type as `struct_template`, containing
 149     `elements` as its contained elements.
 150   """
 151   if not nest.is_sequence(elements):
 152     return nest.pack_sequence_as(struct_template, [elements])
 153   return nest.pack_sequence_as(struct_template, elements)

 156 def _EmptyAcc(slen, struct_template):
 157   """Creates a set of accumulators for tensors in structure.
 158 
 159   Args:
 160     slen: The sequence length. A scalar tensor.
 161     struct_template: A structure of tensors.
 162 
 163   Returns:
 164     A structure congruent to `struct_template`. Say ret is a returned
 165     dictionary. Then, `ret.key`, a tensor, has the same dtype as
 166     `struct_template.key`. The tensor's shape has 1 more dimension
 167     than the tensor `struct_template.key`. The extra 0-th dimension is of size
 168     `slen`. E.g., if `slen=10` and `struct_template.key`'s shape is `[3, 5]`,
 169     then, `ret.key`'s shape is `[10, 3, 5]`.
 170   """
 171 
 172   def _EmptyAccForTensor(tensor):
 173     return inplace_ops.empty(
 174         array_ops.concat([[slen], array_ops.shape(tensor)], axis=0),
 175         tensor.dtype,
 176         init=True)
 177 
 178   return nest.map_structure(_EmptyAccForTensor, struct_template)

 181 def _EmptyLike(struct):
 182   """Creates a set of empty initialized tensors.
 183 
 184   Args:
 185     struct: A structure of tensors.
 186 
 187   Returns:
 188     A struct of tensors. Each tensor has the same shape and dtype as
 189     its corresponding tensor in `struct`. And each tensor is initialized.
 190   """
 191   return nest.map_structure(
 192       lambda x: inplace_ops.empty_like(x, init=True), struct)

 195 def _Add(struct_x, struct_y):
 196   """Adds tensors in `struct_x` with respective tensors in `struct_y`.
 197 
 198   Args:
 199     struct_x: A struct of tensors.
 200     struct_y: A struct of tensors congruent to `struct_x`.
 201 
 202   Returns:
 203     A struct of tensors. Each element of the returned value
 204   equals `x + y`, with corresponding values in `struct_x` and `struct_y`.
 205   """
 206   list_x = nest.flatten(struct_x)
 207   list_y = nest.flatten(struct_y)
 208   z = []
 209   for x, y in zip(list_x, list_y):
 210     z += [math_ops.add(x, y)]
 211   return nest.pack_sequence_as(struct_x, z)

 219 def _ConvertNoneGradientToZeros(xs, dxs):
 220   """Sanitize dxs so that None becomes zeros appropriately.
 221 
 222   Args:
 223     xs: A list of tensors.
 224     dxs: A list of tensors. dxs[i] corresponds to xs[i]'s gradient.
 225 
 226   Returns:
 227     A structure same as `dxs` with `None` replaced by a zero tensor.
 228   """
 229   list_xs = nest.flatten(xs)
 230   list_dxs = nest.flatten(dxs)
 231 
 232   # If x does not get any backprop-ed gradient, propagate zeros.
 233   rets = []
 234   for (x, dx) in zip(list_xs, list_dxs):
 235     if dx is None:
 236       rets.append(array_ops.zeros_like(x))
 237     else:
 238       rets.append(dx)
 239 
 240   return nest.pack_sequence_as(dxs, rets)

 243 # All structures are flattened for use internally. This is for simplicity
 244 # and also to use the Defun construct.
 245 # In the forward pass (inference), the computation is structured as follows.
 246 # Forward: [gradient = _Recurrent.Grad]
 247 #   Flatten structures, create accumulators.
 248 #   for t = 0..max_input_length:
 249 #     Defun ForwardLoopBody:
 250 #       Defun Fwd: flatten/pack around cell_fn
 251 #       state1 = Fwd(inputs[t], state0)
 252 #       acc_state += [state1]
 253 #   Pack structures.
 254 # During the backward pass (backpropping the gradient from the last time
 255 # step to the first, through the structure), the computation is structured
 256 # as follows.
 257 # Grad:
 258 #   Flatten structures.
 259 #   Defun Backward:
 260 #     Create create accumulated derivatives: d_theta, d_inputs, d_acc_state.
 261 #     Regarding the note at the top of the file, there is only one accumulator
 262 #     for d_theta accumulated over the whole sequence.
 263 #     for t = max_input_length -1..0:
 264 #       Defun BackwardLoopBody:
 265 #         Retrieve acc_state[t] computed in the forward pass.
 266 #         Defun Bak: flatten/back around cell_fn_grad.
 267 #         d_state1 is d_state0 from previous step (ie next time).
 268 #         d_acc_state[dev_t] += d_state1
 269 #         d_theta_t, d_state0, d_inputs_t, = Bak()
 270 #         d_inputs[dev_t] += d_inputs
 271 #         d_theta += d_theta_t
 272 #         d_acc_state[t] += d_state1
 273 #   Pack structures and return.
 274 class _Recurrent(object):
 275   """A helper class to construct a recurrent neural net."""
 276 
 277   def __init__(self,
 278                cell_fn,
 279                cell_grad,
 280                theta,
 281                state0,
 282                inputs,
 283                max_input_length,
 284                extras,
 285                use_tpu,
 286                aligned_end=False):
 287     """RNN helper class.
 288 
 289     Args:
 290       cell_fn: A python function, which computes:
 291          state1, extras = cell_fn(theta, state0, inputs[t, :])
 292       cell_grad: A python function which computes:
 293          dtheta, dstate0, dinputs[t, :] = cell_grad(
 294            theta, state0, inputs[t, :], extras, dstate1)
 295       theta: weights. A structure of tensors.
 296       state0: initial state. A structure of tensors.
 297       inputs: inputs. A structure of tensors.
 298       max_input_length: None, or the maximum effective length of the input over
 299         all batches. A scalar tensor.
 300       extras: A structure of tensors. The 2nd return value of every
 301         invocation of cell_fn is a structure of tensors with matching keys
 302         and shapes of this `extras`.
 303       use_tpu: A boolean indicating whether the computation is mean to
 304         run on a TPU.
 305       aligned_end: A boolean indicating whether the sequence is aligned at
 306         the end.
 307     """
 308     self._theta = theta
 309     self._state = state0
 310     self._inputs = inputs
 311     self._max_input_length = self._MaybeComputeMaxInputLength(
 312         inputs, max_input_length)
 313     self._cell_fn = cell_fn
 314     self._cell_grad = cell_grad
 315     self._extras = extras
 316     self._aligned_end = aligned_end
 317 
 318     # pylint: disable=unbalanced-tuple-unpacking
 319 
 320     # NOTE: TF Function (Fwd, Bak, ForwardLoopBody, BackwardLoopBody,
 321     # Forward and Backward defined below) simply takes a list of
 322     # Tensors and returns a list of Tensors. When we pass in a
 323     # structure (a list of structures of Tensors), we use _Flatten to
 324     # convert the structure into a list of tensor. Conversely, the
 325     # following code often uses _Pack to formulate a structure from a
 326     # list of tensors based on a "template".
 327 
 328     # Wraps cell_fn in a TF Function:
 329     #    state1 = cell_fn(theta, state0, inputs)
 330     fwd_sig = [self._theta, self._state, self._inputs]
 331 
 332     compiled = use_tpu
 333     noinline = not compiled
 334     dev_t_type = dtypes.int32 if use_tpu else dtypes.int64
 335 
 336     @function.Defun(*_Dtypes(fwd_sig))
 337     def Fwd(*args):
 338       (theta, state0, inputs) = _Pack(args, fwd_sig)
 339       state1, extras = self._cell_fn(theta, state0, inputs)
 340       assert not function.get_extra_args(), (
 341           'cell_fn is not pure with extra args: %s.' %
 342           (function.get_extra_args()))
 343       _AssertIsCompatible(state1, self._state)
 344       _AssertIsCompatible(extras, self._extras)
 345       return _Flatten([state1, extras])
 346 
 347     # Wraps cell_fn in a TF Function as a for-loop's body.
 348     #
 349     # The loop state is composed of:
 350     #  t: The loop variable. Timestep id.
 351     #  dev_t: The loop variable mirrored on the device.
 352     #  theta: the recurrent net's weights.
 353     #  state0: the previous recurrent state.
 354     #  inputs: inputs to the recurrent net. inputs[t, :] are for the timestep t.
 355     #  acc_state: Each timestep's computed new state is also stashed into
 356     #    acc_state.
 357     #  acc_extras: Each timestep's computed extras is stashed into acc_extras
 358     fwdloop_sig = [
 359         self._theta, self._state, self._inputs, self._state, self._extras
 360     ]
 361 
 362     @function.Defun(dtypes.int32, dev_t_type, *_Dtypes(fwdloop_sig))
 363     def ForwardLoopBody(*args):
 364       """The body of forward loop."""
 365       t, dev_t = args[0], args[1]
 366       (theta, state0, inputs, acc_state, acc_extras) = _Pack(
 367           args[2:], fwdloop_sig)
 368       inputs_t = _Index(inputs, t)  # external input at time step t.
 369       fwd = Fwd(*_Flatten([theta, state0, inputs_t]))
 370       state1, extras = _Pack(fwd, [self._state, self._extras])
 371       # Saves state1 and extras in their accumulators.
 372       acc_state = _Update(acc_state, state1, dev_t)
 373       acc_extras = _Update(acc_extras, extras, dev_t)
 374 
 375       return [math_ops.add(dev_t, 1)] + _Flatten(
 376           [theta, state1, inputs, acc_state, acc_extras])
 377 
 378     def Grad(op, *args):
 379       """The python grad function for the Forward function."""
 380 
 381       # NOTE: tf.gradient backprops None for int32/int64 while zeros
 382       # for float32/float64. For consistency, we always backprop
 383       # zeros.
 384       args = list(args)
 385       for i, dy in enumerate(args):
 386         if dy is None:
 387           args[i] = array_ops.zeros_like(op.outputs[i])
 388       # TODO(drpng): getting the extra state here?
 389       op_inputs = [x for x in op.inputs]
 390       op_struct = [
 391           self._theta, self._state, self._inputs, self._max_input_length,
 392           self._extras
 393       ]
 394       (theta, state0, inputs, max_input_length, _) = _Pack(op_inputs, op_struct)
 395       # acc_state and acc_extras are computed by the Forward pass and
 396       # needed by the Backward pass.
 397       acc_state, _, acc_extras = _Pack([x for x in op.outputs],
 398                                        [self._state, self._state, self._extras])
 399 
 400       # Forward computes acc_state, the final state and
 401       # acc_extras. tf.gradients gives us their gradients w.r.t. the
 402       # final loss. Because acc_extras are not exposed by Compute(),
 403       # it has no gradients w.r.t. the final loss (i.e., by
 404       # construction, it must be zeros).
 405       d_acc_state, d_state1, _ = _Pack(args,
 406                                        [self._state, self._state, self._extras])
 407       return Backward(*_Flatten([
 408           theta, state0, inputs, max_input_length, acc_state, acc_extras,
 409           d_acc_state, d_state1
 410       ]))
 411 
 412     # Forward calls ForwardLoopBody n times. Each time computes one
 413     # time step of the recurrent net.
 414     forward_sig = [
 415         self._theta, self._state, self._inputs, self._max_input_length,
 416         self._extras
 417     ]
 418 
 419     @function.Defun(
 420         *_Dtypes(forward_sig), python_grad_func=Grad, noinline=noinline)
 421     def Forward(*args):
 422       """Forward pass of the recurrent net."""
 423       theta, state0, inputs, max_input_length, extras = _Pack(args, forward_sig)
 424 
 425       slen_dim = _SeqLenDim(inputs)
 426 
 427       # Creates accumulators for state0 and extras.
 428       acc_state = _EmptyAcc(slen_dim, state0)
 429       acc_extras = _EmptyAcc(slen_dim, extras)
 430 
 431       t = slen_dim - max_input_length if self._aligned_end else 0
 432       dev_t = math_ops.to_int32(t) if use_tpu else math_ops.to_int64(t)
 433       run = functional_ops.For(
 434           start=t,
 435           limit=slen_dim if self._aligned_end else max_input_length,
 436           delta=1,
 437           inputs=[dev_t] + _Flatten(
 438               [theta, state0, inputs, acc_state, acc_extras]),
 439           body=ForwardLoopBody,
 440           rewrite_with_while=compiled)
 441       _, state1, _, acc_state, acc_extras = _Pack(
 442           run[1:],
 443           [self._theta, self._state, self._inputs, self._state, self._extras])
 444 
 445       return _Flatten([acc_state, state1, acc_extras])
 446 
 447     # The per-step backward computes:
 448     #    d_theta, d_state0, d_inputs = cell_grad(
 449     #        theta, state0, inputs, extras, d_state1)
 450     # where d_state1 is the backprop-ed gradient for state1, and
 451     # extras is the computed by the forward step to facilitate the
 452     # backward step.
 453     bak_sig = [
 454         self._theta, self._state, self._inputs, self._extras, self._state
 455     ]
 456 
 457     @function.Defun(*_Dtypes(bak_sig))
 458     def Bak(*args):
 459       """Backward step."""
 460       (theta, state0, inputs, extras, d_state1) = _Pack(args, bak_sig)
 461       (dtheta, dstate0, dinputs) = self._cell_grad(theta, state0, inputs,
 462                                                    extras, d_state1)
 463       assert not function.get_extra_args(), (
 464           'cell_grad is not pure with extra args: %s.' %
 465           (function.get_extra_args()))
 466       _AssertIsCompatible(dtheta, self._theta)
 467       _AssertIsCompatible(dstate0, self._state)
 468       _AssertIsCompatible(dinputs, self._inputs)
 469       return _Flatten(
 470           _ConvertNoneGradientToZeros([theta, state0, inputs],
 471                                       [dtheta, dstate0, dinputs]))
 472 
 473     # Define defuns used by a functional_ops.If in BackwardLoopBody.
 474     state_if_sig = [self._state, self._state]
 475 
 476     @function.Defun(*_Dtypes(state_if_sig))
 477     def ReturnOrigState0(*args):
 478       """Returns original state0 from inputs."""
 479       (_, orig_state0) = _Pack(args, state_if_sig)
 480       return nest.flatten(orig_state0)
 481 
 482     @function.Defun(*_Dtypes(state_if_sig))
 483     def ReturnAccState(*args):
 484       """Returns acc_state[t-1] from inputs."""
 485       (acc_state, _) = _Pack(args, state_if_sig)
 486       return nest.flatten(acc_state)
 487 
 488     # Wraps cell_grad gradient function in a TF Function as a
 489     # for-loop's body for the Backward pass.
 490     #
 491     # The loop state is composed of:
 492     #  t: The loop variable. Timestep id.
 493     #  state0: the initial state for the entire backward loop.
 494     #  dev_t: The loop variable mirrored on the device.
 495     #  theta: the recurrent net's weights.
 496     #  inputs: inputs to the recurrent net. inputs[t, :] are for the timestep t.
 497     #  acc_state: Each timestep's computed new state was stashed into
 498     #    acc_state by the Forward pass.
 499     #  acc_extras: Each timestep's computed extras was stashed into
 500     #    acc_extras by the Forward pass.
 501     #  d_theta: All timestep's gradient for theta is accumulated (added) into
 502     #      d_theta.
 503     #  d_state1: The backprop-ed gradient for the new stated computed by
 504     #      timestep t.
 505     #  d_inputs: d_inputs[t, :] is populated by the backward time step t.
 506     #  d_acc_state: The backprop-ed gradient for acc_state.
 507     bakloop_sig = [
 508         self._theta, self._state, self._inputs, self._state, self._extras,
 509         self._theta, self._state, self._inputs, self._state
 510     ]
 511 
 512     @function.Defun(dtypes.int32, dev_t_type, *_Dtypes(bakloop_sig))
 513     def BackwardLoopBody(*args):
 514       """Backward loop body function."""
 515       t, dev_t = args[0], args[1]
 516       (theta, orig_state0, inputs, acc_state, acc_extras, d_theta, d_state1,
 517        d_inputs, d_acc_state) = _Pack(args[2:], bakloop_sig)
 518 
 519       # The input recurrent state for time step t is previous time step's
 520       # output, or the original state0 when on time step 0.
 521       state_from_acc = _Index(acc_state, math_ops.maximum(0, t - 1))
 522       state0 = functional_ops.If(
 523           math_ops.equal(t, array_ops.constant(0, dtypes.int32)),
 524           _Flatten([state_from_acc, orig_state0]), ReturnOrigState0,
 525           ReturnAccState)
 526       state0 = nest.pack_sequence_as(orig_state0, state0)
 527 
 528       # The external inputs for time step t.
 529       inputs_t = _Index(inputs, t)
 530       # The extras for time step t.
 531       extras_t = _Index(acc_extras, t)
 532 
 533       d_state1 = _Add(_Index(d_acc_state, t), d_state1)
 534       (d_theta_t, d_state0, d_inputs_t) = _Pack(
 535           Bak(*_Flatten([theta, state0, inputs_t, extras_t, d_state1])),
 536           [self._theta, self._state, self._inputs])
 537       d_theta = _Add(d_theta, d_theta_t)
 538       d_inputs = _Update(d_inputs, d_inputs_t, dev_t)
 539       return [math_ops.subtract(dev_t, 1)] + _Flatten([
 540           theta, orig_state0, inputs, acc_state, acc_extras, d_theta, d_state0,
 541           d_inputs, d_acc_state
 542       ])
 543 
 544     # Backward calls BackwardLoopBody n times.  Each time computes the backprop
 545     # for one time step of the recurrent net.
 546     backward_sig = [
 547         self._theta, self._state, self._inputs, self._max_input_length,
 548         self._state, self._extras, self._state, self._state
 549     ]
 550 
 551     @function.Defun(*_Dtypes(backward_sig), noinline=noinline)
 552     def Backward(*args):
 553       """Backward pass for the recurrent net."""
 554       # theta, state0, inputs are Forward's inputs.
 555       # acc_state is the accumulated 1st output of Forward.
 556       # acc_extras is the accumulated 2nd output of Forward.
 557       # d_acc_state is the gradient for acc_state.
 558       # d_state1 is the gradient for the final state computed by Forward.
 559       (theta, state0, inputs, max_input_length, acc_state, acc_extras,
 560        d_acc_state, d_state1) = _Pack(args, backward_sig)
 561 
 562       # Accumulators for gradients.
 563       d_theta = _EmptyLike(theta)
 564       d_inputs = _EmptyLike(inputs)
 565 
 566       slen_dim = _SeqLenDim(inputs)
 567 
 568       # Loop backwards. Note the loop's limit is open-ended, so goes through
 569       # t=0.
 570       t = slen_dim - 1 if self._aligned_end else max_input_length - 1
 571       dev_t = math_ops.to_int32(t) if use_tpu else math_ops.to_int64(t)
 572       limit = slen_dim - max_input_length - 1 if self._aligned_end else -1
 573       run = functional_ops.For(
 574           start=t,
 575           limit=limit,
 576           delta=-1,
 577           inputs=[dev_t] + _Flatten([
 578               theta, state0, inputs, acc_state, acc_extras, d_theta, d_state1,
 579               d_inputs, d_acc_state
 580           ]),
 581           body=BackwardLoopBody,
 582           rewrite_with_while=compiled)
 583 
 584       (theta, state0, inputs, acc_state, acc_extras, d_theta, d_state0,
 585        d_inputs, d_acc_state) = _Pack(run[1:], bakloop_sig)
 586 
 587       d_max_input_length = array_ops.constant(0, dtype=max_input_length.dtype)
 588       return _Flatten(
 589           [d_theta, d_state0, d_inputs, d_max_input_length, acc_extras])
 590 
 591     self._forward = Forward
 592 
 593   def _MaybeComputeMaxInputLength(self, inputs, max_input_length):
 594     if max_input_length is not None:
 595       return max_input_length
 596     return math_ops.reduce_max(array_ops.shape(nest.flatten(inputs)[0])[0])
 597 
 598   def Compute(self):
 599     return _Pack(
 600         self._forward(*_Flatten([
 601             self._theta, self._state, self._inputs, self._max_input_length,
 602             self._extras
 603         ])), [self._state, self._state, self._extras])[:2]

 606 def _GetCellGrad(cell_fn, cell_grad):
 607   """Returns the gradient function for cell_fn.
 608 
 609   Args:
 610     cell_fn: The recurrent neural net's cell function.
 611     cell_grad: If not None, cell_fn's gradient function.
 612 
 613   Returns:
 614     Returns cell_grad if not None. Otherwise, assume cell_fn is a python
 615     function representing the recurrent neural net's cell function, i.e.,
 616       cell_fn: (theta, state0, inputs) -> (state1, extra)
 617     returns its default gradient python function, i.e.,
 618       cell_grad: (theta, state0, inputs, extras, dstate1) -> (
 619                   dtheta, dstate0, dinputs)
 620   """
 621 
 622   if cell_grad:
 623     return cell_grad
 624 
 625   def CellGrad(theta, state0, inputs, extras, dstate1):
 626     """Default gradient function for cell_fn."""
 627     # NOTE: The default grad function recomputes the forward
 628     # function and does not take advantage of 'extras' returned by
 629     # the forward function.
 630     del extras
 631     state1, extras = cell_fn(theta, state0, inputs)
 632     ys = _Flatten([state1])
 633     xs = _Flatten([theta, state0, inputs])
 634     grad_ys = _Flatten([dstate1])
 635     grads = gradients_impl.gradients(ys=ys, xs=xs, grad_ys=grad_ys)
 636     return _ConvertNoneGradientToZeros([theta, state0, inputs],
 637                                        _Pack(grads, [theta, state0, inputs]))
 638 
 639   return CellGrad

 642 def _IsSingleTimeStep(inputs, max_input_length):
 643   """Returns True only if the time dimension of inputs is 1."""
 644   if not isinstance(max_input_length, ops.Tensor):
 645     return max_input_length == 1
 646   for x in nest.flatten(inputs):
 647     if x.shape.dims is None or x.shape[0].value != 1:
 648       return False
 649   return True

 652 def Recurrent(theta,
 653               state0,
 654               inputs,
 655               cell_fn,
 656               cell_grad=None,
 657               extras=None,
 658               max_input_length=None,
 659               use_tpu=False,
 660               aligned_end=False):
 661   """Compute a recurrent neural net.
 662 
 663   Roughly, Recurrent() computes the following:
 664     state = state0
 665     for t in inputs' sequence length:
 666       state = cell_fn(theta, state, inputs[t, :])
 667       accumulate_state[t, :] = state
 668     return accumulate_state, state
 669 
 670   theta, state, inputs are all structures of tensors.
 671 
 672   inputs[t, :] means taking a slice out from every tensor in the inputs.
 673 
 674   accumulate_state[t, :] = state means that we stash every tensor in
 675   'state' into a slice of the corresponding tensor in
 676   accumulate_state.
 677 
 678   cell_fn is a python callable computing (building up a TensorFlow
 679   graph) the recurrent neural network's one forward step. Two calls of
 680   cell_fn must describe two identical computations.
 681 
 682   By construction, Recurrent()'s backward computation does not access
 683   any intermediate values computed by cell_fn during forward
 684   computation. We may extend Recurrent() to support that by taking a
 685   customized backward function of cell_fn.
 686 
 687   Args:
 688     theta: weights. A structure of tensors.
 689     state0: initial state. A structure of tensors.
 690     inputs: inputs. A structure of tensors.
 691     cell_fn: A python function, which computes:
 692       state1, extras = cell_fn(theta, state0, inputs[t, :])
 693     cell_grad: A python function which computes:
 694       dtheta, dstate0, dinputs[t, :] = cell_grad(
 695         theta, state0, inputs[t, :], extras, dstate1)
 696     extras: A structure of tensors. The 2nd return value of every
 697       invocation of cell_fn is a structure of tensors with matching keys
 698       and shapes of  this `extras`.
 699     max_input_length: maximum length of effective input. This is used to
 700       truncate the computation if the inputs have been allocated to a
 701       larger size. A scalar tensor.
 702     use_tpu: whether or not we are on TPU.
 703     aligned_end: A boolean indicating whether the sequence is aligned at
 704       the end.
 705 
 706   Returns:
 707     accumulate_state and the final state.
 708   """
 709   if cell_grad is None and _IsSingleTimeStep(inputs, max_input_length):
 710     # The seqlen length is staticly known as 1. Hence, we just need to
 711     # call cell_fn once without putting it into a loop.
 712     inputs = nest.map_structure(lambda x: array_ops.squeeze(x, axis=0), inputs)
 713     state1, _ = cell_fn(theta, state0, inputs)
 714     acc_state = nest.map_structure(lambda x: array_ops.expand_dims(x, axis=0),
 715                                    state1)
 716     return acc_state, state1
 717 
 718   # If cell_grad is not given, derives the gradient function from
 719   # cell_fn.
 720   cell_grad = _GetCellGrad(cell_fn, cell_grad)
 721 
 722   if extras is None:
 723     # Derives 'extras' so that we can allocate extras' accumulator.
 724     _, extras = cell_fn(theta, state0, _Index(inputs, 0))
 725     extras = nest.map_structure(array_ops.zeros_like, extras)
 726   else:
 727     _, actual = cell_fn(theta, state0, _Index(inputs, 0))
 728     _AssertIsCompatible(extras, actual)
 729 
 730   return _Recurrent(
 731       cell_fn=cell_fn,
 732       cell_grad=cell_grad,
 733       theta=theta,
 734       state0=state0,
 735       inputs=inputs,
 736       max_input_length=max_input_length,
 737       extras=extras,
 738       use_tpu=use_tpu,
 739       aligned_end=aligned_end).Compute()
