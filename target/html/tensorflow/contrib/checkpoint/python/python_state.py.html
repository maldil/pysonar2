  37 class NumpyState(base.CheckpointableBase):
  38   """A checkpointable object whose NumPy array attributes are saved/restored.
  39 
  40   Example usage:
  41 
  42   ```python
  43   arrays = tf.contrib.checkpoint.NumpyState()
  44   checkpoint = tf.train.Checkpoint(numpy_arrays=arrays)
  45   arrays.x = numpy.zeros([3, 4])
  46   save_path = checkpoint.save("/tmp/ckpt")
  47   arrays.x[1, 1] = 4.
  48   checkpoint.restore(save_path)
  49   assert (arrays.x == numpy.zeros([3, 4])).all()
  50 
  51   second_checkpoint = tf.train.Checkpoint(
  52       numpy_arrays=tf.contrib.checkpoint.NumpyState())
  53   # Attributes of NumpyState objects are created automatically by restore()
  54   second_checkpoint.restore(save_path)
  55   assert (second_checkpoint.numpy_arrays.x == numpy.zeros([3, 4])).all()
  56   ```
  57 
  58   Note that `NumpyState` objects re-create the attributes of the previously
  59   saved object on `restore()`. This is in contrast to TensorFlow variables, for
  60   which a `Variable` object must be created and assigned to an attribute.
  61 
  62   This snippet works both when graph building and when executing eagerly. On
  63   save, the NumPy array(s) are fed as strings to be saved in the checkpoint (via
  64   a placeholder when graph building, or as a string constant when executing
  65   eagerly). When restoring they skip the TensorFlow graph entirely, and so no
  66   restore ops need be run. This means that restoration always happens eagerly,
  67   rather than waiting for `checkpoint.restore(...).run_restore_ops()` like
  68   TensorFlow variables when graph building.
  69   """
  70 
  71   def _lookup_dependency(self, name):
  72     """Create placeholder NumPy arrays for to-be-restored attributes.
  73 
  74     Typically `_lookup_dependency` is used to check by name whether a dependency
  75     exists. We cheat slightly by creating a checkpointable object for `name` if
  76     we don't already have one, giving us attribute re-creation behavior when
  77     loading a checkpoint.
  78 
  79     Args:
  80       name: The name of the dependency being checked.
  81     Returns:
  82       An existing dependency if one exists, or a new `_NumpyWrapper` placeholder
  83       dependency (which will generally be restored immediately).
  84     """
  85     value = super(NumpyState, self)._lookup_dependency(name)
  86     if value is None:
  87       value = _NumpyWrapper(numpy.array([]))
  88       new_reference = base.CheckpointableReference(name=name, ref=value)
  89       self._unconditional_checkpoint_dependencies.append(new_reference)
  90       self._unconditional_dependency_names[name] = value
  91       super(NumpyState, self).__setattr__(name, value)
  92     return value
  93 
  94   def __getattribute__(self, name):
  95     """Un-wrap `_NumpyWrapper` objects when accessing attributes."""
  96     value = super(NumpyState, self).__getattribute__(name)
  97     if isinstance(value, _NumpyWrapper):
  98       return value.array
  99     return value
 100 
 101   def __setattr__(self, name, value):
 102     """Automatically wrap NumPy arrays assigned to attributes."""
 103     # TODO(allenl): Consider supporting lists/tuples, either ad-hoc or by making
 104     # ndarrays checkpointable natively and using standard checkpointable list
 105     # tracking.
 106     if isinstance(value, (numpy.ndarray, numpy.generic)):
 107       try:
 108         existing = super(NumpyState, self).__getattribute__(name)
 109         existing.array = value
 110         return
 111       except AttributeError:
 112         value = _NumpyWrapper(value)
 113         self._track_checkpointable(value, name=name, overwrite=True)
 114     elif (name not in ("_setattr_tracking", "_update_uid")
 115           and getattr(self, "_setattr_tracking", True)):
 116       # Mixing restore()-created attributes with user-added checkpointable
 117       # objects is tricky, since we can't use the `_lookup_dependency` trick to
 118       # re-create attributes (we might accidentally steal the restoration for
 119       # another checkpointable object). For now `NumpyState` objects must be
 120       # leaf nodes. Theoretically we could add some extra arguments to
 121       # `_lookup_dependency` to figure out whether we should create a NumPy
 122       # array for the attribute or not.
 123       raise NotImplementedError(
 124           ("Assigned %s to the %s property of %s, which is not a NumPy array. "
 125            "Currently mixing NumPy arrays and other checkpointable objects is "
 126            "not supported. File a feature request if this limitation bothers "
 127            "you.")
 128           % (value, name, self))
 129     super(NumpyState, self).__setattr__(name, value)

 132 @six.add_metaclass(abc.ABCMeta)
 133 class PythonStateWrapper(base.CheckpointableBase):
 134   """Wraps a Python object for storage in an object-based checkpoint."""
 135 
 136   @abc.abstractmethod
 137   def _serialize(self):
 138     """Callback for `PythonStringStateSaveable` to serialize the object."""
 139 
 140   @abc.abstractmethod
 141   def _deserialize(self, string_value):
 142     """Callback for `PythonStringStateSaveable` to deserialize the object."""
 143 
 144   def _gather_saveables_for_checkpoint(self):
 145     """Specify callbacks for saving and restoring `array`."""
 146     return {
 147         "py_state": functools.partial(
 148             base.PythonStringStateSaveable,
 149             state_callback=self._serialize,
 150             restore_callback=self._deserialize)
 151         }
 152 
 153 
 154 class _NumpyWrapper(PythonStateWrapper):
 155   """Wraps a NumPy array for storage in an object-based checkpoint."""
 156 
 157   def __init__(self, array):
 158     """Specify a NumPy array to wrap.
 159 
 160     Args:
 161       array: The NumPy array to save and restore (may be overwritten).
 162     """
 163     self.array = array
 164 
 165   def _serialize(self):
 166     """Callback to serialize the array."""
 167     string_file = BytesIO()
 168     try:
 169       numpy.save(string_file, self.array, allow_pickle=False)
 170       serialized = string_file.getvalue()
 171     finally:
 172       string_file.close()
 173     return serialized
 174 
 175   def _deserialize(self, string_value):
 176     """Callback to deserialize the array."""
 177     string_file = BytesIO(string_value)
 178     try:
 179       self.array = numpy.load(string_file, allow_pickle=False)
 180     finally:
 181       string_file.close()
