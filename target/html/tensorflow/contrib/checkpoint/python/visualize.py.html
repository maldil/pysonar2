  25 def dot_graph_from_checkpoint(save_path):
  26   r"""Visualizes an object-based checkpoint (from `tf.train.Checkpoint`).
  27 
  28   Useful for inspecting checkpoints and debugging loading issues.
  29 
  30   Example usage from Python (requires pydot):
  31   ```python
  32   import tensorflow as tf
  33   import pydot
  34 
  35   dot_string = tf.contrib.checkpoint.dot_graph_from_checkpoint('/path/to/ckpt')
  36   parsed, = pydot.graph_from_dot_data(dot_string)
  37   parsed.write_svg('/tmp/tensorflow/visualized_checkpoint.svg')
  38   ```
  39 
  40   Example command line usage:
  41   ```sh
  42   python -c "import tensorflow as tf;\
  43     print(tf.contrib.checkpoint.dot_graph_from_checkpoint('/path/to/ckpt'))"\
  44     | dot -Tsvg > /tmp/tensorflow/checkpoint_viz.svg
  45   ```
  46 
  47   Args:
  48     save_path: The checkpoint prefix, as returned by `tf.train.Checkpoint.save`
  49       or `tf.train.latest_checkpoint`.
  50   Returns:
  51     A graph in DOT format as a string.
  52   """
  53   reader = pywrap_tensorflow.NewCheckpointReader(save_path)
  54   object_graph = checkpointable_utils.object_metadata(save_path)
  55   shape_map = reader.get_variable_to_shape_map()
  56   dtype_map = reader.get_variable_to_dtype_map()
  57   graph = 'digraph {\n'
  58   def _escape(name):
  59     return name.replace('"', '\\"')
  60   slot_ids = set()
  61   for node in object_graph.nodes:
  62     for slot_reference in node.slot_variables:
  63       slot_ids.add(slot_reference.slot_variable_node_id)
  64   for node_id, node in enumerate(object_graph.nodes):
  65     if (len(node.attributes) == 1
  66         and node.attributes[0].name == checkpointable.VARIABLE_VALUE_KEY):
  67       if node_id in slot_ids:
  68         color = 'orange'
  69         tooltip_prefix = 'Slot variable'
  70       else:
  71         color = 'blue'
  72         tooltip_prefix = 'Variable'
  73       attribute = node.attributes[0]
  74       graph += ('N_%d [shape=point label="" color=%s width=.25'
  75                 ' tooltip="%s %s shape=%s %s"]\n') % (
  76                     node_id,
  77                     color,
  78                     tooltip_prefix,
  79                     _escape(attribute.full_name),
  80                     shape_map[attribute.checkpoint_key],
  81                     dtype_map[attribute.checkpoint_key].name)
  82     elif node.slot_variables:
  83       graph += ('N_%d [shape=point label="" width=.25 color=red,'
  84                 'tooltip="Optimizer"]\n') % node_id
  85     else:
  86       graph += 'N_%d [shape=point label="" width=.25]\n' % node_id
  87     for reference in node.children:
  88       graph += 'N_%d -> N_%d [label="%s"]\n' % (
  89           node_id, reference.node_id, _escape(reference.local_name))
  90     for slot_reference in node.slot_variables:
  91       graph += 'N_%d -> N_%d [label="%s" style=dotted]\n' % (
  92           node_id,
  93           slot_reference.slot_variable_node_id,
  94           _escape(slot_reference.slot_name))
  95       graph += 'N_%d -> N_%d [style=dotted]\n' % (
  96           slot_reference.original_variable_node_id,
  97           slot_reference.slot_variable_node_id)
  98   graph += '}\n'
  99   return graph
