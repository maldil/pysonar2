  50 def replace_t_with_placeholder_handler(info, t):
  51   """Transform a tensor into a placeholder tensor.
  52 
  53   This handler is typically used to transform a subgraph input tensor into a
  54   placeholder.
  55 
  56   Args:
  57     info: Transform._TmpInfo instance.
  58     t: tensor whose input must be transformed into a place holder.
  59   Returns:
  60     The tensor generated by the newly created place holder.
  61   """
  62   with info.graph_.as_default():
  63     t_ = util.make_placeholder_from_tensor(t, scope=info.scope_)
  64   return t_

  67 def keep_t_if_possible_handler(info, t):
  68   """Transform a tensor into itself (identity) if possible.
  69 
  70   This handler transform a tensor into itself if the source and destination
  71   graph are the same. Otherwise it will create a placeholder.
  72   This handler is typically used to transform a hidden input tensors.
  73 
  74   Args:
  75     info: Transform._TmpInfo instance.
  76     t: tensor whose input must be transformed into a place holder.
  77   Returns:
  78     The tensor generated by the newly created place holder.
  79   """
  80   if info.graph is info.graph_:
  81     return t
  82   else:
  83     return replace_t_with_placeholder_handler(info, t)

  86 def assign_renamed_collections_handler(info, elem, elem_):
  87   """Add the transformed elem to the (renamed) collections of elem.
  88 
  89   A collection is renamed only if is not a known key, as described in
  90   `tf.GraphKeys`.
  91 
  92   Args:
  93     info: Transform._TmpInfo instance.
  94     elem: the original element (`tf.Tensor` or `tf.Operation`)
  95     elem_: the transformed element
  96   """
  97   known_collection_names = util.get_predefined_collection_names()
  98   for name, collection in iteritems(info.collections):
  99     if elem not in collection:
 100       continue
 101 
 102     if name in known_collection_names:
 103       transformed_name = name
 104     else:
 105       transformed_name = info.new_name(name)
 106     info.graph_.add_to_collection(transformed_name, elem_)

 109 def transform_op_if_inside_handler(info, op, keep_if_possible=True):
 110   """Transform an optional op only if it is inside the subgraph.
 111 
 112   This handler is typically use to handle original op: it is fine to keep them
 113   if they are inside the subgraph, otherwise they are just ignored.
 114 
 115   Args:
 116     info: Transform._TmpInfo instance.
 117     op: the optional op to transform (or ignore).
 118     keep_if_possible: re-attach to the original op if possible, that is,
 119       if the source graph and the destination graph are the same.
 120   Returns:
 121     The transformed op or None.
 122   """
 123   if op in info.sgv.ops:
 124     return info.transformed_ops[op]
 125   else:
 126     if keep_if_possible and info.graph is info.graph_:
 127       return op
 128     else:
 129       return None

 132 def copy_op_handler(info, op, new_inputs, copy_shape=False, nodedef_fn=None):
 133   """Copy a `tf.Operation`.
 134 
 135   Args:
 136     info: Transform._TmpInfo instance.
 137     op: the `tf.Operation` to be copied.
 138     new_inputs: The new inputs for this op.
 139     copy_shape: also copy the shape of the tensor
 140     nodedef_fn: If provided, a function that will be run on the NodeDef
 141       and should return a mutated NodeDef before a new Operation is created.
 142       This is useful as certain features cannot be set on the Operation and
 143       must be modified in NodeDef.
 144 
 145   Returns:
 146     A `(op, op_outputs)` tuple containing the transformed op and its outputs.
 147   """
 148   # The `new_inputs` was added to this function. For compatibility reason,
 149   # let's raise an error if `new_inputs` is a boolean.
 150   if isinstance(new_inputs, bool):
 151     raise TypeError("the `new_inputs` argument must be an iterable.")
 152 
 153   # pylint: disable=protected-access
 154 
 155   # Clone the node def:
 156   node_def_ = deepcopy(op.node_def)
 157 
 158   # Transform name:
 159   name_ = info.new_name(op.name)
 160   name_ = info.graph_.unique_name(name_)
 161   node_def_.name = name_
 162 
 163   # Mutate NodeDef if requested:
 164   if nodedef_fn is not None:
 165     node_def_ = nodedef_fn(node_def_)
 166 
 167   # Copy the other inputs needed for initialization
 168   output_types_ = op._output_types[:]
 169   input_types_ = op._input_types[:]
 170 
 171   # Make a copy of the op_def too.
 172   # Its unique to every _type_ of Operation.
 173   op_def_ = deepcopy(op.op_def)
 174 
 175   # Initialize a new Operation instance
 176   op_ = tf_ops.Operation(node_def_, info.graph_, new_inputs, output_types_,
 177                          [], input_types_, None, op_def_)
 178 
 179   # copy the shape over
 180   if copy_shape:
 181     for t, t_ in zip(op.outputs, op_.outputs):
 182       t_.set_shape(t.get_shape())
 183 
 184   # Original op cannot be finalised here yet. Because some ops require this
 185   # attribute to exist, we will create a dummy original_op first and then
 186   # later finalise it with the actual original_op when all the ops have
 187   # been copied.
 188   # TODO(fkp): Stop worrying about _original_op and remove this code?
 189   if op._original_op:
 190     op_._original_op = op._original_op
 191 
 192   return op_, op_.outputs

 195 class TransformerInfo(object):
 196   """"Contains information about the result of a transform operation."""
 197 
 198   def __init__(self, info):
 199     """Constructor.
 200 
 201     Args:
 202       info: an instance of Transformer._TmpInfo containing various internal
 203         information about the transform operation.
 204     """
 205     self._graph = info.graph
 206     self._scope = info.scope
 207     self._graph_ = info.graph_
 208     self._scope_ = info.scope_
 209     self._transformed_ops = info.transformed_ops
 210     self._transformed_ts = info.transformed_ts
 211 
 212   def _get_transformed_map(self, top):
 213     """Return the correct container depending on the type of `top`."""
 214     if isinstance(top, tf_ops.Operation):
 215       return self._transformed_ops
 216     elif isinstance(top, tf_ops.Tensor):
 217       return self._transformed_ts
 218     else:
 219       raise TypeError(
 220           "Expected a tf.Tensor or a tf.Operation, got a {}".format(
 221               type(top)))
 222 
 223   def _transformed_elem(self, original_top, missing_fn=None):
 224     """Return the transformed op/tensor corresponding to the original one.
 225 
 226     Args:
 227       original_top: the original tensor/operation.
 228       missing_fn: function handling the case where the counterpart
 229         cannot be found. By default, None is returned.
 230     Returns:
 231       the transformed tensor/operation (or None if no match is found).
 232     """
 233     transformed_map = self._get_transformed_map(original_top)
 234     if isinstance(original_top, string_types):
 235       for original, transformed in iteritems(transformed_map):
 236         if original.name == original_top:
 237           return transformed
 238       return None if missing_fn is None else missing_fn(original_top)
 239     else:
 240       if original_top not in transformed_map:
 241         return None if missing_fn is None else missing_fn(original_top)
 242       return transformed_map[original_top]
 243 
 244   def _original_elem(self, transformed_top, missing_fn=None):
 245     """Return the original op/tensor corresponding to the transformed one.
 246 
 247     Args:
 248       transformed_top: the transformed tensor/operation.
 249       missing_fn: function handling the case where the counterpart
 250         cannot be found. By default, None is returned.
 251     Returns:
 252       the original tensor/operation (or None if no match is found).
 253     """
 254     transformed_map = self._get_transformed_map(transformed_top)
 255     if isinstance(transformed_top, string_types):
 256       finder = lambda transformed: transformed.name == transformed_top
 257     else:
 258       finder = lambda transformed: transformed == transformed_top
 259     for original, transformed in iteritems(transformed_map):
 260       if finder(transformed):
 261         return original
 262     return None if missing_fn is None else missing_fn(transformed_top)
 263 
 264   def transformed(self, original, missing_fn=None):
 265     """Return the transformed op/tensor corresponding to the original one.
 266 
 267     Note that the output of this function mimics the hierarchy
 268     of its input argument `original`.
 269     Given an iterable, it returns a list. Given an operation or a tensor,
 270     it will return an operation or a tensor.
 271 
 272     Args:
 273       original: the original tensor/operation.
 274       missing_fn: function handling the case where the counterpart
 275         cannot be found. By default, None is returned.
 276     Returns:
 277       the transformed tensor/operation (or None if no match is found).
 278     """
 279     transformed_elem = partial(self._transformed_elem, missing_fn=missing_fn)
 280     return util.transform_tree(original, transformed_elem)
 281 
 282   def original(self, transformed, missing_fn=None):
 283     """Return the original op/tensor corresponding to the transformed one.
 284 
 285     Note that the output of this function mimics the hierarchy
 286     of its input argument `transformed`.
 287     Given an iterable, it returns a list. Given an operation or a tensor,
 288     it will return an operation or a tensor.
 289 
 290     Args:
 291       transformed: the transformed tensor/operation.
 292       missing_fn: function handling the case where the counterpart
 293         cannot be found. By default, None is returned.
 294     Returns:
 295       the original tensor/operation (or None if no match is found).
 296     """
 297     original_elem = partial(self._original_elem, missing_fn=missing_fn)
 298     return util.transform_tree(transformed, original_elem)
 299 
 300   def __str__(self):
 301     res = StringIO()
 302     print("Transform result info:", file=res)
 303     if self._graph == self._graph_:
 304       in_place_str = "" if self._scope_ else " IN-PLACE"
 305       print("  Within graph[{}]{}".format(
 306           id(self._graph), in_place_str), file=res)
 307     else:
 308       print("  graph[{}] => graph[{}]".format(
 309           id(self._graph), id(self._graph_)), file=res)
 310     if self._scope:
 311       print("  Relative to source scope: {}".format(self._scope), file=res)
 312     if self._scope_:
 313       print("  Scope destination: {}".format(self._scope_), file=res)
 314     print("Operations mapping:", file=res)
 315     for op, op_ in iteritems(self._transformed_ops):
 316       print("  {} => {}".format(op.name, op_.name), file=res)
 317     return res.getvalue()

 320 class _TmpInfo(object):
 321   """Transformer temporary data.
 322 
 323   An instance of this class holds all the information relevant to a call
 324   to a transformer instance (that is, a call to __call__). An instance
 325   is created for the life-time of the __call__ function and is passed as
 326   argument to the handlers.
 327   """
 328 
 329   def __init__(self, sgv, dst_graph, dst_scope, src_scope):
 330     self.sgv = sgv
 331     self.sgv_inputs_set = frozenset(sgv.inputs)
 332     self.ops = frozenset(sgv.ops)
 333     self.control_outputs = util.ControlOutputs(sgv.graph)
 334     self.graph = sgv.graph
 335     self.scope = src_scope
 336     self.graph_ = dst_graph
 337     self.scope_ = dst_scope
 338     self.transformed_ops = {}
 339     self.transformed_ts = {}
 340     self.collections = dict((key, self.graph.get_collection(key))
 341                             for key in self.graph.get_all_collection_keys())
 342     self.cyclic_ops = []
 343     self.transform_original_op_handler = transform_op_if_inside_handler
 344     # The graph is transformed op by op, in the same order the original ops
 345     # were created. However, this is sometimes not possible due to cycles
 346     # (i.e. while loops). So when the transformer creates a new op whose
 347     # inputs do not exist yet, temporary placeholders are created and stored
 348     # in this `tmp_cyclic_ts` container. During a second pass,
 349     # those temporary tensors are replaced by the proper transformed tensors
 350     # (see the function `_finalize_cycles`).
 351     self.tmp_cyclic_ts = []
 352 
 353   def new_name(self, name):
 354     """Compute a destination name from a source name.
 355 
 356     Args:
 357       name: the name to be "transformed".
 358     Returns:
 359       The transformed name.
 360     Raises:
 361       ValueError: if the source scope is used (that is, not an empty string)
 362         and the source name does not belong to the source scope.
 363     """
 364     scope = self.scope
 365     if not name.startswith(scope):
 366       raise ValueError("{} does not belong to source scope: {}.".format(
 367           name, scope))
 368     rel_name = name[len(scope):]
 369     name_ = self.scope_ + rel_name
 370     return name_

 373 class Transformer(object):
 374   """Transform a subgraph into another one.
 375 
 376   By default, the constructor create a transform which copy a subgraph and
 377   replaces inputs with placeholders. This behavior can be modified by changing
 378   the handlers.
 379   """
 380 
 381   def __init__(self):
 382     """Transformer constructor.
 383 
 384     The following members can be modified:
 385     transform_op_handler: handle the transformation of a `tf.Operation`.
 386       This handler defaults to a simple copy.
 387     assign_collections_handler: handle the assignment of collections.
 388       This handler defaults to assigning new collections created under the
 389       given name-scope.
 390     transform_external_input_handler: handle the transform of the inputs to
 391       the given subgraph. This handler defaults to creating placeholders
 392       instead of the ops just before the input tensors of the subgraph.
 393     transform_external_hidden_input_handler: handle the transform of the
 394       hidden inputs of the subgraph, that is, the inputs which are not listed
 395       in sgv.inputs. This handler defaults to a transform which keep the same
 396       input if the source and destination graphs are the same, otherwise
 397       use placeholders.
 398     transform_original_op_handler: handle the transform of original_op. This
 399       handler defaults to transforming original_op only if they are in the
 400       subgraph, otherwise they are ignored.
 401     """
 402 
 403     # handlers
 404     self.transform_op_handler = copy_op_handler
 405     self.transform_control_input_handler = transform_op_if_inside_handler
 406     self.assign_collections_handler = assign_renamed_collections_handler
 407     self.transform_external_input_handler = replace_t_with_placeholder_handler
 408     self.transform_external_hidden_input_handler = keep_t_if_possible_handler
 409     self.transform_original_op_handler = transform_op_if_inside_handler
 410 
 411   def __call__(self,
 412                sgv,
 413                dst_graph,
 414                dst_scope,
 415                src_scope="",
 416                reuse_dst_scope=False):
 417     """Execute the transformation.
 418 
 419     Args:
 420       sgv: the source subgraph-view.
 421       dst_graph: the destination graph.
 422       dst_scope: the destination scope.
 423       src_scope: the source scope, which specify the path from which the
 424         relative path of the transformed nodes are computed. For instance, if
 425         src_scope is a/ and dst_scoped is b/, then the node a/x/y will have a
 426         relative path of x/y and will be transformed into b/x/y.
 427       reuse_dst_scope: if True the dst_scope is re-used if it already exists.
 428         Otherwise, the scope is given a unique name based on the one given
 429         by appending an underscore followed by a digit (default).
 430     Returns:
 431       A tuple `(sgv, info)` where:
 432         `sgv` is the transformed subgraph view;
 433         `info` is an instance of TransformerInfo containing
 434         information about the transform, including mapping between
 435         original and transformed tensors and operations.
 436     Raises:
 437       ValueError: if the arguments are invalid.
 438     """
 439     sgv = subgraph.make_view(sgv)
 440     if not isinstance(dst_graph, tf_ops.Graph):
 441       raise TypeError("Expected a tf.Graph, got: {}".format(type(dst_graph)))
 442 
 443     src_scope = util.scope_finalize(src_scope)
 444     dst_scope = util.scope_finalize(dst_scope)
 445 
 446     # Potentially create new scope if reuse_dst_scope is False
 447     if dst_scope and not reuse_dst_scope:
 448       dst_scope = util.scope_finalize(dst_graph.unique_name(dst_scope[:-1]))
 449 
 450     # Create temporary info used during this transform call
 451     info = _TmpInfo(sgv, dst_graph, dst_scope, src_scope)
 452 
 453     self._copy_ops(info)
 454     self._finalize_cycles(info)
 455     self._connect_control_inputs(info)
 456 
 457     # Compute information about the transformation
 458     res_info = TransformerInfo(info)
 459     sgv_ = self._transform_sgv(info, sgv)
 460     return sgv_, res_info
 461 
 462   def _copy_ops(self, info):
 463     """Copy ops without connecting them."""
 464     sorted_ops = sorted(info.sgv.ops, key=lambda op: op._id)  # pylint: disable=protected-access
 465     for op in sorted_ops:
 466       new_inputs = [self._transformed_t(info, t, op) for t in op.inputs]
 467       op_, op_outputs_ = self.transform_op_handler(info, op, new_inputs)
 468       if op is op_:
 469         raise ValueError("In-place transformation not allowed.")
 470 
 471       # Process op.
 472       info.transformed_ops[op] = op_
 473       self.assign_collections_handler(info, op, op_)
 474 
 475       # Process output tensors.
 476       for op_output, op_output_ in zip(op.outputs, op_outputs_):
 477         info.transformed_ts[op_output] = op_output_
 478         self.assign_collections_handler(info, op_output, op_output_)
 479 
 480   def _finalize_cycles(self, info):
 481     """Reconnects the cyclic tensors."""
 482     for t, tmp_t_, consumer_op in info.tmp_cyclic_ts:
 483       if t not in info.transformed_ts:
 484         raise ValueError("The tensor {} should be transformed by now.".format(
 485             t.name))
 486       if consumer_op not in info.transformed_ops:
 487         raise ValueError("The op {} should be transformed by now.".format(
 488             consumer_op.name))
 489       t_ = info.transformed_ts[t]
 490       consumer_op_ = info.transformed_ops[consumer_op]
 491       t_index_ = list(consumer_op_.inputs).index(tmp_t_)
 492       consumer_op_._update_input(t_index_, t_)  # pylint: disable=protected-access
 493 
 494   def _connect_control_inputs(self, info):
 495     """Connect the previously copied ops."""
 496     for op in info.sgv.ops:
 497       logging.debug("Connecting control inputs of op: %s", op.name)
 498       op_ = info.transformed_ops[op]
 499 
 500       # Finalize original op.
 501       # TODO(fkp): Stop worrying about _original_op and remove this code?
 502       # pylint: disable=protected-access
 503       if op._original_op:
 504         original_op = self.transform_original_op_handler(info, op._original_op)
 505         if original_op is None:
 506           logging.debug("Could not find original op for: %s", op_.name)
 507         else:
 508           op_._original_op = original_op
 509       # pylint: enable=protected-access
 510 
 511       # Finalize control inputs:
 512       control_inputs_ = [self.transform_control_input_handler(info, ci)
 513                          for ci in op.control_inputs]
 514       control_inputs_ = [ci for ci in control_inputs_ if ci is not None]
 515       reroute.add_control_inputs(op_, control_inputs_)
 516 
 517   def _transform_sgv(self, info, sgv):
 518     """Transform a subgraph view.
 519 
 520     For convenience, a transform operation returns a subgraph view of the
 521     transformed graph.
 522 
 523     Args:
 524       info: Temporary information for this transorfm call.
 525       sgv: the subgraph to be transformed.
 526     Returns:
 527       The transformed subgraph.
 528     """
 529     ops_ = [op_ for _, op_ in iteritems(info.transformed_ops)]
 530     sgv_ = subgraph.SubGraphView(ops_)
 531     sgv_inputs_ = sgv_.inputs
 532     sgv_outputs_ = sgv_.outputs
 533 
 534     # re-order inputs
 535     input_map_ = []
 536     for input_t in sgv.inputs:
 537       if input_t not in info.transformed_ts:
 538         continue
 539       input_t_ = info.transformed_ts[input_t]
 540       if input_t_ not in sgv_inputs_:
 541         continue
 542       input_t_index_ = sgv_.input_index(input_t_)
 543       input_map_.append(input_t_index_)
 544 
 545     # re-order outputs
 546     output_map_ = []
 547     for output_t in sgv.outputs:
 548       if output_t not in info.transformed_ts:
 549         continue
 550       output_t_ = info.transformed_ts[output_t]
 551       if output_t_ not in sgv_outputs_:
 552         continue
 553       output_t_index_ = sgv_.output_index(output_t_)
 554       output_map_.append(output_t_index_)
 555 
 556     return sgv_.remap(input_map_, output_map_)
 557 
 558   def _transformed_t(self, info, t, consumer_op):
 559     """Return tre transformed tensor of `t`."""
 560     if t in info.transformed_ts:
 561       # If op is in the subgraph, just return its transformed counterpart.
 562       return info.transformed_ts[t]
 563 
 564     if t in info.sgv_inputs_set:
 565       # `t` is an input of the subgraph.
 566       return self.transform_external_input_handler(info, t)
 567     elif t.op in info.ops:
 568       # `t` is an internal tensor but is not transformed yet because it
 569       # belongs to a graph cycle.
 570       logging.debug("Cyclic tensor: t.name = %s", t.name)
 571       # Try to find an existing tensor we can use for now,
 572       # otherwise create one. We'll rewire this later.
 573       if consumer_op.type == "Merge":
 574         first_input = consumer_op.inputs[0]
 575         tmp_t_ = self._transformed_t(info, first_input, consumer_op)
 576       elif t.op.type == "Enter":
 577         enter_input = t.op.inputs[0]
 578         tmp_t_ = self._transformed_t(info, enter_input, consumer_op)
 579       else:
 580         with info.graph_.as_default():
 581           tmp_t_ = util.make_placeholder_from_tensor(t, scope=info.scope_,
 582                                                      prefix="geph_tmp")
 583         logging.debug("Created temporary placeholder: %s.", tmp_t_.name)
 584       # Register as temporary and return.
 585       info.tmp_cyclic_ts.append((t, tmp_t_, consumer_op))
 586       return tmp_t_
 587     else:
 588       # `t` is a hidden input of the subgraph.
 589       return self.transform_external_hidden_input_handler(info, t)

 592 def copy(sgv, dst_graph=None, dst_scope="", src_scope="",
 593          reuse_dst_scope=False):
 594   """Copy a subgraph.
 595 
 596   Args:
 597     sgv: the source subgraph-view. This argument is converted to a subgraph
 598       using the same rules than the function subgraph.make_view.
 599     dst_graph: the destination graph.
 600     dst_scope: the destination scope.
 601     src_scope: the source scope.
 602     reuse_dst_scope: if True the dst_scope is re-used if it already exists.
 603       Otherwise, the scope is given a unique name based on the one given
 604       by appending an underscore followed by a digit (default).
 605   Returns:
 606     A tuple `(sgv, info)` where:
 607       `sgv` is the transformed subgraph view;
 608       `info` is an instance of TransformerInfo containing
 609       information about the transform, including mapping between
 610       original and transformed tensors and operations.
 611   Raises:
 612     TypeError: if `dst_graph` is not a `tf.Graph`.
 613     StandardError: if sgv cannot be converted to a SubGraphView using
 614       the same rules than the function subgraph.make_view.
 615   """
 616   sgv = subgraph.make_view(sgv)
 617   if dst_graph is None:
 618     dst_graph = sgv.graph
 619   if not isinstance(dst_graph, tf_ops.Graph):
 620     raise TypeError("Expected a tf.Graph, got: {}".format(type(dst_graph)))
 621 
 622   copier = Transformer()
 623   return copier(
 624       sgv, dst_graph, dst_scope, src_scope, reuse_dst_scope=reuse_dst_scope)

 627 def copy_with_input_replacements(sgv, replacement_ts,
 628                                  dst_graph=None, dst_scope="", src_scope="",
 629                                  reuse_dst_scope=False):
 630   """Copy a subgraph, replacing some of its inputs.
 631 
 632   Note a replacement only happens if the tensor to be replaced
 633   is an input of the given subgraph. The inputs of a subgraph can
 634   be queried using sgv.inputs.
 635 
 636   Args:
 637     sgv: the source subgraph-view. This argument is converted to a subgraph
 638       using the same rules as the function subgraph.make_view.
 639     replacement_ts: dictionary mapping from original tensors to the
 640       replaced one.
 641     dst_graph: the destination graph.
 642     dst_scope: the destination scope.
 643     src_scope: the source scope.
 644     reuse_dst_scope: if True the dst_scope is re-used if it already exists.
 645       Otherwise, the scope is given a unique name based on the one given
 646       by appending an underscore followed by a digit (default).
 647   Returns:
 648     A tuple `(sgv, info)` where:
 649       `sgv` is the transformed subgraph view;
 650       `info` is an instance of TransformerInfo containing
 651       information about the transform, including mapping between
 652       original and transformed tensors and operations.
 653   Raises:
 654     TypeError: if dst_graph is not a tf.Graph.
 655     StandardError: if sgv cannot be converted to a SubGraphView using
 656       the same rules as the function subgraph.make_view.
 657   """
 658   sgv = subgraph.make_view(sgv)
 659   if dst_graph is None:
 660     dst_graph = sgv.graph
 661   if not isinstance(dst_graph, tf_ops.Graph):
 662     raise TypeError("Expected a tf.Graph, got: {}".format(type(dst_graph)))
 663 
 664   copier = Transformer()
 665   # Replace tensor if possible.
 666   def replace_t_with_replacement_handler(info, t):
 667     if t in replacement_ts:
 668       return replacement_ts[t]
 669     else:
 670       return keep_t_if_possible_handler(info, t)
 671   copier.transform_external_input_handler = replace_t_with_replacement_handler
 672   return copier(
 673       sgv, dst_graph, dst_scope, src_scope, reuse_dst_scope=reuse_dst_scope)

 676 def _add_control_flow_ops(ops, control_ios):
 677   """Complete `ops` so that the transformed graph is valid.
 678 
 679   Partially copying a graph can lead to a malformed graph. For instance,
 680   copying half of a while construct is likely to result in an invalid graph.
 681   This function attempts to add missing ops so that the transformation result
 682   in a valid graph.
 683 
 684   Args:
 685     ops: list of ops (modifed in-place).
 686     control_ios: object created by a call to `util.ControlOutputs`.
 687   """
 688   # Find while contexts.
 689   control_flow_contexts = set()
 690   for op in ops:
 691     cfc = op._control_flow_context  # pylint: disable=protected-access
 692     if cfc:
 693       control_flow_contexts.add(cfc)
 694   # Find new ops.
 695   new_ops = []
 696   for cfc in control_flow_contexts:
 697     if cfc.IsWhileContext():
 698       new_ops += select.get_walks_intersection_ops(
 699           [enter_t.op for enter_t in cfc.loop_enters],
 700           [exit_t.op for exit_t in cfc.loop_exits],
 701           control_ios=control_ios)
 702   # Add new ops.
 703   new_ops_set = set(new_ops)
 704   ops_set = frozenset(ops)
 705   for op in new_ops_set:
 706     if op not in ops_set:
 707       ops.append(op)

 710 def graph_replace(target_ts, replacement_ts, dst_scope="",
 711                   src_scope="", reuse_dst_scope=False):
 712   """Create a new graph which compute the targets from the replaced Tensors.
 713 
 714   Args:
 715     target_ts: a single tf.Tensor or an iterable of tf.Tensor.
 716     replacement_ts: dictionary mapping from original tensors to replaced tensors
 717     dst_scope: the destination scope.
 718     src_scope: the source scope.
 719     reuse_dst_scope: if True the dst_scope is re-used if it already exists.
 720       Otherwise, the scope is given a unique name based on the one given
 721       by appending an underscore followed by a digit (default).
 722   Returns:
 723     A single tf.Tensor or a list of target tf.Tensor, depending on
 724     the type of the input argument `target_ts`.
 725     The returned tensors are recomputed using the tensors from replacement_ts.
 726   Raises:
 727     ValueError: if the targets are not connected to replacement_ts.
 728   """
 729   # Identify operations in the graph that will change.
 730   # Start forward walk at Tensors that will be replaced, and
 731   # backward walk at the target output Tensors.
 732   flatten_target_ts = util.flatten_tree(target_ts)
 733   # Construct the forward control dependencies edges so that
 734   # the get_walks_intersection_ops can also traverse the
 735   # control dependencies.
 736   graph = util.get_unique_graph(flatten_target_ts, check_types=(tf_ops.Tensor))
 737   control_ios = util.ControlOutputs(graph)
 738   ops = select.get_walks_intersection_ops(list(iterkeys(replacement_ts)),
 739                                           flatten_target_ts,
 740                                           control_ios=control_ios)
 741   if not ops:
 742     raise ValueError("Targets and replacements are not connected!")
 743 
 744   # Complete ops to avoid malformed control flow.
 745   # TODO(fkp): Consider moving this function deeper (in the transformer?).
 746   _add_control_flow_ops(ops, control_ios)
 747 
 748   # Create a copy of the relevant subgraph
 749   unused_sgv_, info = copy_with_input_replacements(
 750       ops, replacement_ts, None, dst_scope, src_scope, reuse_dst_scope)
 751   # Return the transformed targets but keep the original if the transformed
 752   # counterpart cannot be found
 753   missing_fn = lambda original_t: original_t
 754   return info.transformed(target_ts, missing_fn)
