  36 def _eager_safe_variable_handle(shape, dtype, shared_name, name, graph_mode):
  37   """Creates a variable handle with information to do shape inference."""
  38   container = ops.get_default_graph()._container  # pylint: disable=protected-access
  39   if container is None:
  40     container = ""
  41   handle = resource_variable_ops.var_handle_op(shape=shape, dtype=dtype,
  42                                                shared_name=shared_name,
  43                                                name=name,
  44                                                container=container)
  45   if graph_mode:
  46     return handle
  47 
  48   with context.graph_mode(), ops.Graph().as_default() as graph:
  49     h = resource_variable_ops.var_handle_op(shape=shape, dtype=dtype,
  50                                             shared_name=shared_name,
  51                                             name=name,
  52                                             container=container)
  53 
  54     # Tensor._handle_data contains information for the shape-inference code to
  55     # know the shape and dtype of the variable pointed to by a handle. Since
  56     # shape inference doesn't run in eager mode we copy this data here for when
  57     # the handle is captured by an eager mode function.
  58     # pylint: disable=protected-access
  59     if ops._USE_C_SHAPES:
  60       handle._handle_data = resource_variable_ops.get_resource_handle_data(h)
  61     else:
  62       if h._handle_data is None:
  63         ops.set_shape_and_handle_data_for_outputs(h.op)
  64       handle._handle_data = h._handle_data
  65     # pylint: enable=protected-access
  66   # Clean up op->graph->op reference cycles.
  67   ops.dismantle_graph(graph)
  68   return handle

  71 class SharedVariable(resource_variable_ops.ResourceVariable):
  72   """Experimental Variable designed for parameter server training.
  73 
  74   A SharedVariable has a name and two instances of SharedVariable with the
  75   same name will have the same value, even if they are in different Sessions,
  76   as long as they are placed on the same device.
  77 
  78   The storage associated with SharedVariables is also not deleted when they go
  79   out of scope.
  80   """
  81 
  82   def __init__(self,  # pylint: disable=super-init-not-called
  83                initial_value=None,
  84                trainable=True,
  85                name=None,
  86                dtype=None,
  87                constraint=None,
  88                initialize=True,
  89                **unused_kwargs):
  90     """Creates a variable.
  91 
  92     Args:
  93       initial_value: A `Tensor`, or Python object convertible to a `Tensor`,
  94         which is the initial value for the Variable. The initial value must have
  95         a shape specified unless `validate_shape` is set to False. Can also be a
  96         callable with no argument that returns the initial value when called.
  97         (Note that initializer functions from init_ops.py must first be bound
  98          to a shape before being used here.)
  99       trainable: If `True`, automatically watches this variable on GradientTape
 100         whenever it's used.
 101       name: Optional name for the variable. Defaults to `'Variable'` and gets
 102         uniquified automatically.
 103       dtype: If set, initial_value will be converted to the given type.
 104         If None, either the datatype will be kept (if initial_value is
 105         a Tensor) or float32 will be used (if it is a Python object convertible
 106         to a Tensor).
 107       constraint: An optional projection function to be applied to the variable
 108         after being updated by an `Optimizer` (e.g. used to implement norm
 109         constraints or value constraints for layer weights). The function must
 110         take as input the unprojected Tensor representing the value of the
 111         variable and return the Tensor for the projected value
 112         (which must have the same shape). Constraints are not safe to
 113         use when doing asynchronous distributed training.
 114       initialize: if True, runs initialization in eager execution; leaves the
 115         variable uninitialized otherwise.
 116 
 117     Raises:
 118       ValueError: If the initial value is not specified, or does not have a
 119         shape and `validate_shape` is `True`.
 120     """
 121     if initial_value is None:
 122       raise ValueError("initial_value must be specified.")
 123     init_from_fn = callable(initial_value)
 124 
 125     if isinstance(initial_value, ops.Tensor) and hasattr(
 126         initial_value, "graph") and initial_value.graph.building_function:
 127       raise ValueError("Tensor-typed variable initializers must either be "
 128                        "wrapped in an init_scope or callable "
 129                        "(e.g., `tf.Variable(lambda : "
 130                        "tf.truncated_normal([10, 40]))`) when building "
 131                        "functions. Please file a feature request if this "
 132                        "restriction inconveniences you.")
 133 
 134     if constraint is not None and not callable(constraint):
 135       raise ValueError("The `constraint` argument must be a callable.")
 136 
 137     if isinstance(initial_value, checkpointable.CheckpointInitialValue):
 138       self._maybe_initialize_checkpointable()
 139       self._update_uid = initial_value.checkpoint_position.restore_uid
 140       initial_value = initial_value.wrapped_value
 141 
 142     self._trainable = trainable
 143     self._save_slice_info = None
 144     # Store the graph key so optimizers know how to only retrieve variables from
 145     # this graph.
 146     self._graph_key = ops.get_default_graph()._graph_key  # pylint: disable=protected-access
 147     with ops.init_scope():
 148       self._in_graph_mode = not context.executing_eagerly()
 149       with ops.name_scope(name, "Variable", []
 150                           if init_from_fn else [initial_value]) as name:
 151         # pylint: disable=protected-access
 152         handle_name = ops._name_from_scope_name(name)
 153         shared_name = handle_name
 154         if init_from_fn:
 155           # Use attr_scope and device(None) to simulate the behavior of
 156           # colocate_with when the variable we want to colocate with doesn't
 157           # yet exist.
 158           if self._in_graph_mode:
 159             with ops.name_scope("Initializer"), ops.device(None):
 160               initial_value = ops.convert_to_tensor(
 161                   initial_value(), name="initial_value", dtype=dtype)
 162             self._handle = _eager_safe_variable_handle(
 163                 shape=initial_value.get_shape(),
 164                 dtype=initial_value.dtype.base_dtype,
 165                 shared_name=shared_name,
 166                 name=name,
 167                 graph_mode=self._in_graph_mode)
 168             self._shape = initial_value.get_shape()
 169           else:
 170             initial_value = initial_value()
 171             with ops.name_scope("Initializer"):
 172               initial_value = ops.convert_to_tensor(
 173                   initial_value, name="initial_value", dtype=dtype)
 174             self._handle = _eager_safe_variable_handle(
 175                 shape=initial_value.get_shape(),
 176                 dtype=initial_value.dtype.base_dtype,
 177                 shared_name=shared_name,
 178                 name=name,
 179                 graph_mode=False)
 180             self._shape = initial_value.get_shape()
 181         # pylint: enable=protected-access
 182 
 183         # Or get the initial value from a Tensor or Python object.
 184         else:
 185           with ops.name_scope("Initializer"):
 186             initial_value = ops.convert_to_tensor(
 187                 initial_value, name="initial_value", dtype=dtype)
 188           # pylint: disable=protected-access
 189           if (self._in_graph_mode and initial_value is not None and
 190               initial_value.op._get_control_flow_context() is not None):
 191             raise ValueError(
 192                 "Initializer for variable %s is from inside a control-flow "
 193                 "construct, such as a loop or conditional. When creating a "
 194                 "variable inside a loop or conditional, use a lambda as the "
 195                 "initializer." % name)
 196           # pylint: enable=protected-access
 197           self._handle = _eager_safe_variable_handle(
 198               shape=initial_value.get_shape(),
 199               dtype=initial_value.dtype.base_dtype,
 200               shared_name=shared_name,
 201               name=name,
 202               graph_mode=self._in_graph_mode)
 203           self._shape = initial_value.get_shape()
 204 
 205         self._unique_id = shared_name
 206         self._initial_value = initial_value if self._in_graph_mode else None
 207         self._handle_name = handle_name + ":0"
 208         self._dtype = initial_value.dtype.base_dtype
 209         self._constraint = constraint
 210 
 211         if self._in_graph_mode:
 212           with ops.name_scope("IsInitialized"):
 213             self._is_initialized_op = (
 214                 resource_variable_ops.var_is_initialized_op(self._handle))
 215           if initial_value is not None:
 216             with ops.name_scope("Assign") as n, ops.colocate_with(self._handle):
 217               self._initializer_op = (
 218                   resource_variable_ops.assign_variable_op(
 219                       self._handle,
 220                       self._try_guard_against_uninitialized_dependencies(
 221                           initial_value),
 222                       name=n))
 223           with ops.name_scope("Read"), ops.colocate_with(self._handle):
 224             # Manually assign reads to the handle's device to avoid log
 225             # messages.
 226             with ops.device(self._handle.device):
 227               value = self._read_variable_op()
 228             self._graph_element = value
 229             self._cached_value = None
 230         else:
 231           if initialize:
 232             resource_variable_ops.assign_variable_op(self._handle,
 233                                                      initial_value)
 234           self._is_initialized_op = None
 235           self._initializer_op = None
 236           self._graph_element = None
 237           self._cached_value = None
 238 
 239     self._handle_deleter = None
 240     self._cached_shape_as_list = None

 243 @contextlib.contextmanager
 244 def parameter_server_scope(is_chief, ps_job_name, num_ps_tasks):
 245   """Strategy to use parameter servers in eager.
 246 
 247   Creates SharedVariable objects for variables created in this scope. These
 248   SharedVariable objects will be placed round-robin on the parameter servers
 249   specified by the ps_job_name and num_ps_tasks arguments.
 250 
 251   To use parameter servers you need only to wrap your model initialization in
 252   this scope:
 253 
 254   ```
 255   with tf.contrib.eager.parameter_server_scope(
 256       is_chief, ps_job_name, num_ps_tasks):
 257     my_model = tf.keras.Sequential([...])  # Or
 258     input = tf.keras.Input(...)
 259     ....
 260     my_model = tf.keras.Model(input, output)
 261   my_model.compile(...)
 262   # or other usages of the model.
 263   ```
 264 
 265   Args:
 266     is_chief: Boolean. Whether this worker is responsible for initializing
 267       variables.
 268     ps_job_name: The name of the ps job in this cluster.
 269     num_ps_tasks: The number of ps tasks to use.
 270 
 271   Yields:
 272     a context manager.
 273   """
 274   # Note: capturing in a list to allow assignment.
 275   ps_index = [0]
 276 
 277   def variable_creator_scope(unused_next_creator, **kwargs):
 278     kwargs["initialize"] = is_chief
 279     with ops.device(
 280         "/job:%s/task:%s" % (ps_job_name, ps_index[0] % num_ps_tasks)):
 281       ps_index[0] += 1
 282       v = SharedVariable(**kwargs)
 283       if not is_chief:
 284         while not resource_variable_ops.var_is_initialized_op(v.handle):
 285           time.sleep(10)
 286       return v
 287 
 288   with variable_scope.variable_creator_scope(variable_creator_scope):
 289     yield
