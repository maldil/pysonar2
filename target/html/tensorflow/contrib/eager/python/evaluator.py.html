  33 class Evaluator(object):
  34   """This holds and updates Metrics for the duration of a single eval run.
  35 
  36   Usage:
  37     evaluator = my_model.evaluator() # or MyEvaluator(my_model)
  38     for example_batch in ...:
  39       evaluator(example_batch)
  40     results = evaluator.all_metric_results(optional_summary_logdir)
  41 
  42   Or, if you are getting your examples from a tf.data.Dataset, you can use
  43   the evaluate_on_dataset() method.
  44 
  45   Implementers of Evaluators should
  46   (a) Call `track_metric()` and/or `track_evaluator()` in __init__().
  47   (b) Override the `call()` method. It will be passed the output of the
  48       model's `eval_data()` method, and should call its contained metrics
  49       (treating them as callables) and any child Evaluators (using their
  50       call() method to avoid calling eval_data() again).
  51 
  52   Args:
  53     model: A `Model` object with an `eval_data()` method.
  54   """
  55 
  56   def __init__(self, model):
  57     self._model = model
  58     self._metrics = {}
  59     self._evaluators = {}
  60     if not context.executing_eagerly():
  61       self.call = function.defun(self.call)
  62 
  63   # ---- API for users ----
  64   def __call__(self, *args, **kwargs):
  65     """Update metrics with a minibatch of input examples.
  66 
  67     Args:
  68       *args:
  69       **kwargs: Arguments representing an input mini-batch of examples to
  70         pass to self.model.eval_data().
  71 
  72     Returns:
  73       The op to execute or None if executing eagerly.
  74     """
  75     return self.call(self._model.eval_data(*args, **kwargs))
  76 
  77   def init_variables(self):
  78     """Return an op for initializing all contained uninitialized variables.
  79 
  80     Only for graph execution. Should be called after variables are created
  81     in the first execution of __call__().
  82 
  83     Returns:
  84       An op.
  85 
  86     Raises:
  87       RuntimeError: if eager execution is enabled.
  88 
  89     @compatibility(eager)
  90     Only for graph execution.
  91     @end_compatibility
  92     """
  93     if context.executing_eagerly():
  94       raise RuntimeError("Evaluator.init_variables() not needed when "
  95                          "eager execution is enabled.")
  96     return control_flow_ops.group([m.init_variables() for _, m in self.metrics])
  97 
  98   def all_metric_results(self, summary_logdir=None):
  99     """Computes results for all contained metrics.
 100 
 101     Args:
 102       summary_logdir: An optional string. If specified, metric results
 103         will be written as summaries to this directory.
 104 
 105     Returns:
 106       A `dict` mapping string names to tensors.
 107     """
 108     if summary_logdir is None:
 109       with summary_ops.never_record_summaries():
 110         return self._all_metric_results()
 111     else:
 112       def f():
 113         with summary_ops.create_file_writer(
 114             summary_logdir).as_default(), summary_ops.always_record_summaries():
 115           return self._all_metric_results()
 116 
 117       if context.executing_eagerly():
 118         return f()
 119       else:
 120         return function.defun(f)()
 121 
 122   def _all_metric_results(self):
 123     """Implementation of `all_metric_results` in the summary context."""
 124     results = {}
 125     for name, metric in six.iteritems(self._metrics):
 126       results[name] = metric.result()
 127     for prefix, evaluator in six.iteritems(self._evaluators):
 128       for name, metric in six.iteritems(evaluator._metrics):  # pylint: disable=protected-access
 129         results[prefix + "/" + name] = metric.result()
 130     return results
 131 
 132   def evaluate_on_dataset(self, dataset, *args, **kwargs):
 133     """Convenience method for performing an eval on a Dataset.
 134 
 135     Args:
 136       dataset: Dataset object with the input data to evaluate on.
 137       *args:
 138       **kwargs: Optional additional arguments to __call__(), except
 139         `summary_logdir`: if specified, metrics will be written as summaries
 140         to this directory.
 141 
 142     Returns:
 143       @compatibility(eager)
 144       When eager execution is enabled, this returns the result of performing
 145       an evaluation as a dictionary. With graph execution, this returns a tuple
 146       (init_op, call_op, results_op) which may be executed using this code:
 147       ```python
 148         sess.run(init_op)
 149         try:
 150           while True:
 151             sess.run(call_op)
 152         except tf.errors.OutOfRangeError:
 153           pass
 154         return sess.run(results_op)  # A dictionary
 155 
 156         # equivalently:
 157         return evaluator.run_evaluation(init_op, call_op, results_op, sess=sess)
 158       ```
 159       @end_compatibility
 160     """
 161     summary_logdir = kwargs.pop("summary_logdir", None)
 162     if context.executing_eagerly():
 163       for example in datasets.Iterator(dataset):
 164         self.__call__(example, *args, **kwargs)
 165       return self.all_metric_results(summary_logdir)
 166     # Graph construction
 167     call_op = self.__call__(dataset.make_one_shot_iterator().get_next(), *args,
 168                             **kwargs)
 169     init_op = self.init_variables()
 170     results_op = self.all_metric_results(summary_logdir)
 171     return (init_op, call_op, results_op)
 172 
 173   @staticmethod
 174   def run_evaluation(init_op, call_op, results_op, sess=None):
 175     """Convenience method for running the ops returned by evaluate_on_dataset.
 176 
 177     Args:
 178       init_op: An op that initializes/resets evaluation state.
 179       call_op: An op that updates evaluation state on a mini-batch of examples.
 180         Must generate an tf.errors.OutOfRangeError when done.
 181       results_op: A dictionary of tensors that compute the final evaluation
 182         results from the evaluation state.
 183       sess: The Session to run the evaluation in. Defaults to the default
 184         Session.
 185 
 186     Returns:
 187       A dictionary of values, parallel to results_op.
 188 
 189     Raises:
 190       RuntimeError: if eager execution is enabled.
 191 
 192     @compatibility(eager)
 193     Only for graph execution.
 194     @end_compatibility
 195     """
 196     if context.executing_eagerly():
 197       raise RuntimeError("Evaluator.run_evaluation() not supported when "
 198                          "eager execution is enabled.")
 199     sess = sess or ops.get_default_session()
 200     sess.run(init_op)
 201     try:
 202       while True:
 203         sess.run(call_op)
 204     except errors_impl.OutOfRangeError:
 205       pass
 206     return sess.run(results_op)

 208   # ---- To be implemented by descendants ---
 209   def call(self, eval_data):
 210     """Update metrics using the output of self.model.
 211 
 212     Note: This function is executed as a graph function in graph mode.
 213     This means:
 214     a) Operations on the same resource are executed in textual order.
 215        This should make it easier to do things like add the updated
 216        value of a variable to another, for example.
 217     b) You don't need to worry about collecting the update ops to execute.
 218        All update ops added to the graph by this function will be executed.
 219     As a result, code should generally work the same way with graph or
 220     eager execution.
 221 
 222     Args:
 223       eval_data: The output of self.model.eval_data() on a mini-batch of
 224         examples.
 225     """
 226     raise NotImplementedError("Evaluators must define a call member function.")
 227 
 228   # ---- For use by descendants ---
 229   @property
 230   def model(self):
 231     return self._model

 233   def track_metric(self, metric):
 234     """Add a Metric to be tracked.
 235 
 236     Metrics can only be tracked by one `Evaluator`. Metrics must be
 237     tracked or they will not appear in `all_metric_results()`.
 238 
 239     Args:
 240       metric: A `Metric` object.
 241 
 242     Returns:
 243       The `metric` passed into this function.
 244 
 245     Raises:
 246       RuntimeError: If called before __init__.
 247       TypeError: If `metric` is not of the correct type.
 248       ValueError: If there is a name collision between Metrics or `metric`
 249         has already been added to another `Evaluator`.
 250     """
 251     if not hasattr(self, "_metrics"):
 252       raise RuntimeError(
 253           "Need to call Evaluator.__init__ before adding metrics")
 254     if not isinstance(metric, metrics.Metric):
 255       raise TypeError(
 256           "Evaluator.track_metric() passed type %s, not a tfe.metrics.Metric" %
 257           (type(metric),))
 258     if metric.name in self._metrics:
 259       if metric is self._metrics[metric.name]:
 260         return metric
 261       raise ValueError(
 262           "Attempt to add two Metrics with the name '%s' to the same Evaluator "
 263           "'%s'" % (metric.name, self.name))
 264     # pylint: disable=protected-access
 265     if hasattr(metric, "_added_to_an_evaluator"):
 266       raise ValueError("Metric %s already added to Evaluator %s" %
 267                        (metric.name, metric._added_to_an_evaluator))
 268     metric._added_to_an_evaluator = self.__class__.__name__
 269     # pylint: enable=protected-access
 270     self._metrics[metric.name] = metric
 271     return metric

 273   def track_evaluator(self, prefix, evaluator):
 274     """Add a contained `Evaluator`.
 275 
 276     This is for delegating to another `Evaluator`, e.g. for when you have a
 277     model with multiple heads. Users should manually invoke the child
 278     `Evaluator`'s `call` method from their `call` method.
 279 
 280     Args:
 281       prefix: A string. Metrics from `evaluator` are exported with this
 282         prefix and a '/'.
 283       evaluator: An `Evaluator` object.
 284 
 285     Returns:
 286       The value of `evaluator` passed into this function.
 287 
 288     Raises:
 289       RuntimeError: If called before __init__.
 290       TypeError: If `evaluator` is not of the correct type.
 291       ValueError: If an `Evaluator` has already been added with that `prefix`.
 292     """
 293     if not hasattr(self, "_evaluators"):
 294       raise RuntimeError(
 295           "Need to call Evaluator.__init__ before adding evaluators")
 296     if not isinstance(evaluator, Evaluator):
 297       raise TypeError(
 298           "Evaluator.track_evaluator() passed type %s, not a tfe.Evaluator." %
 299           (type(evaluator),))
 300     if prefix in self._evaluators:
 301       if evaluator is self._evaluators[prefix]:
 302         return evaluator
 303       raise RuntimeError(
 304           "Attempt to add two Evaluators with the same prefix '%s'." % prefix)
 305     self._evaluators[prefix] = evaluator
 306     return evaluator
 307 
 308   @property
 309   def metric_variables(self):
 310     v = []
 311     for metric in six.itervalues(self._metrics):
 312       v += metric.variables
 313     for evaluator in six.itervalues(self._evaluators):
 314       v += evaluator.metric_variables
 315     return v
 316 
 317   @property
 318   def metrics(self):
 319     """Returns a list of (prefix, metric) pairs."""
 320     m = []
 321     for metric in six.itervalues(self._metrics):
 322       m.append(("", metric))
 323     for prefix, evaluator in six.iteritems(self._evaluators):
 324       m += [(prefix + "/" + p, m) for p, m in evaluator.metrics]
 325     return m

 328 class SparseSoftmaxEvaluator(Evaluator):
 329   """Evaluator for a sparse softmax model.
 330 
 331   Computes a standard set of metrics for single-label, multi-class
 332   models.
 333 
 334   Args:
 335     model: A `SparseSoftmaxModel` object or a `Model` whose `eval_data()`
 336       method produces a `dict` containing values for the loss, true
 337       label, predicted class, and optional weights.
 338     loss_key: Optional key for looking up the value of the loss in the
 339       `eval_data()` dict. Defaults to "loss".
 340     label_key: Optional key for looking up the value of the label in the
 341       `eval_data()` dict. Defaults to "label".
 342     predicted_class_key: Optional key for looking up the value of the
 343       predicted class in the `eval_data()` dict. Defaults to "predicted_class".
 344     weights_key: Optional key for looking up the value of the weights
 345       in the `eval_data()` dict. Defaults to "weights". Note that weights
 346       are optional, and default to 1 if not present in `eval_data`.
 347   """
 348 
 349   def __init__(self, model, loss_key="loss", label_key="label",
 350                predicted_class_key="predicted_class", weights_key="weights"):
 351     super(SparseSoftmaxEvaluator, self).__init__(model)
 352     # TODO(josh11b): Expand this to include everything from the standard
 353     # SparseSoftmax Head.
 354     self.avg_loss = self.track_metric(metrics.Mean("Avg Loss"))
 355     self.accuracy = self.track_metric(metrics.Accuracy())
 356     self.loss_key = loss_key
 357     self.label_key = label_key
 358     self.predicted_class_key = predicted_class_key
 359     self.weights_key = weights_key
 360 
 361   def call(self, eval_data):
 362     """Update metrics for `eval_data` dict (described above)."""
 363     weights = eval_data.get(self.weights_key, None)
 364     if weights is None:
 365       self.avg_loss(eval_data[self.loss_key])
 366       self.accuracy(eval_data[self.label_key],
 367                     eval_data[self.predicted_class_key])
 368     else:
 369       self.avg_loss(eval_data[self.loss_key], weights=weights)
 370       self.accuracy(eval_data[self.label_key],
 371                     eval_data[self.predicted_class_key],
 372                     weights=weights)
