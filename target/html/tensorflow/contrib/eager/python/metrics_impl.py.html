  39 class Metric(checkpointable.CheckpointableBase):
  40   """A metric holds state for aggregating statistics over an evaluation run.
  41 
  42   Example use with eager execution:
  43 
  44   ```python
  45   m = SomeMetric(...)
  46   for input in ...:
  47     m(input)
  48   print(m.result())
  49   ```
  50 
  51   Example use with graph execution:
  52 
  53   ```python
  54   m = SomeMetric(...)
  55   inputs = ... # Some tensors to compute the metric on.
  56   m_update = m(inputs)
  57   # Variables defined in first call, so get the initialization op afterwards.
  58   m_init = m.init_variables()  # or tf.global_variables_initializer()
  59   m_result = m.result()
  60   with tf.Session() as sess:
  61     sess.run(m_init)
  62     for input in ...:
  63       sess.run(m_update)
  64     print(sess.run(m_result))
  65   ```
  66   Example use with graph execution with placeholders and feed_dict:
  67   ```python
  68   m = SomeMetric(...)
  69   m_placeholder = tf.placeholder(...)
  70   m_update = m(m_placeholder)
  71   # Variables defined in first call, so get the initialization op afterwards.
  72   m_init = m.init_variables()  # or tf.global_variables_initializer()
  73   m_result = m.result()
  74   with tf.Session() as sess:
  75     sess.run(m_init)
  76     for input in ...:
  77       sess.run(m_update, feed_dict={m_placeholder: input})
  78     print(sess.run(m_result))
  79   ```
  80 
  81   Descendants will implement:
  82   * `build()`: All variables should be created in this method, by calling
  83     `self.add_variable()` as in: `self.var = self.add_variable(...)`
  84     build() will be called in the first invocation of `__call__()`, with
  85     the same arguments passed `call()`.
  86   * `call()`: Has all updates to variables, as in:
  87       self.var.assign_add(...)
  88   * `result()`: Computes and returns a final value for the metric
  89     from the variables in `self`.
  90 
  91   Descendants may override `aggregate()`, but usually won't need to.  It
  92   adds in the state from a list of metrics of the same type as `self`.
  93   (Default is to sum all the variables.) Note that users should not call
  94   `aggregate()`, it is for use by TensorFlow infrastructure.
  95   """
  96 
  97   def __init__(self, name=None, use_global_variables=False):
  98     self._built = False
  99     self._vars = []
 100     self._initial_values = {}
 101     self._updates = []
 102     self._use_global_variables = use_global_variables
 103     name = name or self.__class__.__name__
 104     # Replace things like spaces in name to create a valid scope name.
 105     scope_name = _to_replace.sub("_", name)
 106     # We create the variable scope now to get the unique name that will
 107     # be used as a variable prefix when build() calls add_variable().
 108     with variable_scope.variable_scope(
 109         scope_name, use_resource=True, reuse=False) as scope:
 110       pos = scope.name.rfind(scope_name)
 111       self._name = name + scope.name[pos + len(scope_name):]
 112       self._scope = scope
 113 
 114     # Ensures that if the user calls build directly we still set self._built to
 115     # True to prevent variables from being recreated.
 116     self._build = self.build
 117 
 118     def actual_build(*args, **kwargs):
 119       self._build(*args, **kwargs)
 120       self._built = True
 121     self.build = actual_build
 122     self.build.__doc__ = self._build.__doc__
 123 
 124     # Captures construction scope for proper initialization.
 125     if context.executing_eagerly():
 126       self._construction_scope = context.eager_mode
 127     else:
 128       # We make self.call() into a graph callable here, so that we can
 129       # return a single op that performs all of the variable updates.
 130       self._construction_scope = ops.get_default_graph().as_default
 131       self.call = function.defun(self.call)
 132 
 133   # ---- API for users ----
 134   def __call__(self, *args, **kwargs):
 135     """Returns op to execute to update this metric for these inputs.
 136 
 137     Returns None if eager execution is enabled.
 138     Returns a graph-mode function if graph execution is enabled.
 139 
 140     Args:
 141       *args:
 142       **kwargs: A mini-batch of inputs to the Metric, passed on to `call()`.
 143     """
 144     if not self._built:
 145       with variable_scope.variable_scope(
 146           self._scope), self._construction_scope():
 147         self.build(*args, **kwargs)
 148       self._built = True
 149     return self.call(*args, **kwargs)
 150 
 151   @property
 152   def name(self):
 153     return self._name
 154 
 155   @property
 156   def variables(self):
 157     return self._vars
 158 
 159   def init_variables(self):
 160     """Initializes this Metric's variables.
 161 
 162     Should be called after variables are created in the first execution
 163     of `__call__()`. If using graph execution, the return value should be
 164     `run()` in a session before running the op returned by `__call__()`.
 165     (See example above.)
 166 
 167     Returns:
 168       If using graph execution, this returns an op to perform the
 169       initialization. Under eager execution, the variables are reset to their
 170       initial values as a side effect and this function returns None.
 171     """
 172     if context.executing_eagerly():
 173       for v in self._vars:
 174         v.assign(self._initial_values[v])
 175     else:
 176       return control_flow_ops.group([v.initializer for v in self._vars])
 177 
 178   # ---- To be implemented by descendants ---
 179   def build(self, *args, **kwargs):
 180     """Method to create variables.
 181 
 182     Called by `__call__()` before `call()` for the first time.
 183 
 184     Args:
 185       *args:
 186       **kwargs: The arguments to the first invocation of `__call__()`.
 187        `build()` may use the shape and/or dtype of these arguments
 188        when deciding how to create variables.
 189     """
 190     raise NotImplementedError("Metrics must define a build() member function")
 191 
 192   def call(self, *args, **kwargs):
 193     """Accumulates statistics for the metric. Users should use __call__ instead.
 194 
 195     Note: This function is executed as a graph function in graph mode.
 196     This means:
 197     a) Operations on the same resource are executed in textual order.
 198        This should make it easier to do things like add the updated
 199        value of a variable to another, for example.
 200     b) You don't need to worry about collecting the update ops to execute.
 201        All update ops added to the graph by this function will be executed.
 202     As a result, code should generally work the same way with graph or
 203     eager execution.
 204 
 205     Args:
 206       *args:
 207       **kwargs: A mini-batch of inputs to the Metric, as passed to
 208         `__call__()`.
 209     """
 210     raise NotImplementedError("Metrics must define a call() member function")
 211 
 212   def result(self):  # TODO(josh11b): Add an optional summary_writer parameter.
 213     """Computes and returns a final value for the metric."""
 214     raise NotImplementedError("Metrics must define a result() member function")
 215 
 216   def value(self):
 217     """In graph mode returns the result Tensor while in eager the callable."""
 218     if context.executing_eagerly():
 219       return self.result
 220     else:
 221       return self.result()
 222 
 223   # We can support two different strategies of for doing data-parallel
 224   # distributed metric computations:
 225   # * Put metric variables on the first device and rely on small
 226   #   bandwidth needed to do updates. (Doesn't require any particular
 227   #   code in Metric implementations.)
 228   # * Ask each type of metric to define an aggregation method to run
 229   #   at the end of eval to merge across devices. Note: this is good
 230   #   for the use case where they want to record the metric's state
 231   #   for each example and then later decide which examples they want
 232   #   to aggregate over. (Recommended -- not too much harder and adds
 233   #   flexibility over previous option.)
 234   # I'm going with the second strategy since we can define a default
 235   # implementation of aggregate() that will work for most descendants.
 236   def aggregate(self, metrics):
 237     """Adds in the state from a list of metrics.
 238 
 239     Default implementation sums all the metric variables.
 240 
 241     Args:
 242       metrics: A list of metrics with the same type as `self`.
 243 
 244     Raises:
 245       ValueError: If metrics contains invalid data.
 246     """
 247     for m in metrics:
 248       if type(self) != type(m):  # pylint: disable=unidiomatic-typecheck
 249         raise TypeError("All metrics must be the same type, '%s' != '%s'." %
 250                         (type(self), type(m)))
 251     # pylint: disable=protected-access
 252     for i in range(len(self._vars)):
 253       if any(m._vars[i].name != self._vars[i].name for m in metrics):
 254         raise ValueError("All metrics must have variables in the same order.")
 255       self._vars[i].assign_add(math_ops.add_n([m._vars[i] for m in metrics]))
 256     # pylint: enable=protected-access
 257 
 258   # ---- For use by descendants ---
 259   def add_variable(self, name, shape=None, dtype=None, initializer=None):
 260     """***Only for use by descendants of Metric***."""
 261     if self._built:
 262       raise RuntimeError("Can't call add_variable() except in build().")
 263     if context.executing_eagerly():
 264       collections = None
 265     else:
 266       if self._use_global_variables:
 267         collections = [ops.GraphKeys.GLOBAL_VARIABLES]
 268       else:
 269         collections = [ops.GraphKeys.LOCAL_VARIABLES]
 270       collections += [ops.GraphKeys.METRIC_VARIABLES]
 271     # Variables are Checkpointable dependencies of Metrics regardless of the
 272     # global/local distinction. Users can avoid saving variables by not adding a
 273     # dependency on the Metric.
 274     v = self._add_variable_with_custom_getter(
 275         name=name,
 276         shape=shape,
 277         dtype=dtype,
 278         initializer=initializer,
 279         trainable=False,
 280         collections=collections,
 281         use_resource=True,
 282         getter=variable_scope.get_variable,
 283         # Raise duplicate variable exceptions from get_variable rather than
 284         # Checkpointable.
 285         overwrite=True)
 286     self._vars.append(v)
 287     if context.executing_eagerly():
 288       self._initial_values[v] = v.value()
 289     return v

 292 class Mean(Metric):
 293   """Computes the (weighted) mean of the given values."""
 294 
 295   def __init__(self, name=None, dtype=dtypes.float64,
 296                use_global_variables=False):
 297     super(Mean, self).__init__(name=name,
 298                                use_global_variables=use_global_variables)
 299     self.dtype = dtype
 300 
 301   def build(self, *args, **kwargs):
 302     # build() does not use call's arguments, by using *args, **kwargs
 303     # we make it easier to inherit from Mean().
 304     del args, kwargs
 305     self.numer = self.add_variable(name="numer", shape=(),
 306                                    dtype=self.dtype,
 307                                    initializer=init_ops.zeros_initializer)
 308     self.denom = self.add_variable(name="denom", shape=(),
 309                                    dtype=self.dtype,
 310                                    initializer=init_ops.zeros_initializer)
 311 
 312   def call(self, values, weights=None):
 313     """Accumulate statistics for computing the mean.
 314 
 315     For example, if values is [1, 3, 5, 7] then the mean is 4.
 316     If the weights were specified as [1, 1, 0, 0] then the mean would be 2.
 317 
 318     Args:
 319       values: Tensor with the per-example value.
 320       weights: Optional weighting of each example. Defaults to 1.
 321 
 322     Returns:
 323       The arguments, for easy chaining.
 324     """
 325     if weights is None:
 326       self.denom.assign_add(
 327           math_ops.cast(array_ops.identity(array_ops.size(values)), self.dtype))
 328       values = math_ops.reduce_sum(values)
 329       self.numer.assign_add(math_ops.cast(values, self.dtype))
 330     else:
 331       weights = math_ops.cast(weights, self.dtype)
 332       self.denom.assign_add(math_ops.reduce_sum(weights))
 333       values = math_ops.cast(values, self.dtype) * weights
 334       self.numer.assign_add(math_ops.reduce_sum(values))
 335     if weights is None:
 336       return values
 337     return values, weights
 338 
 339   def result(self, write_summary=True):
 340     """Returns the result of the Metric.
 341 
 342     Args:
 343       write_summary: bool indicating whether to feed the result to the summary
 344         before returning.
 345     Returns:
 346       aggregated metric as float.
 347     Raises:
 348       ValueError: if the optional argument is not bool
 349     """
 350      # Convert the boolean to tensor for tf.cond, if it is not.
 351     if not isinstance(write_summary, ops.Tensor):
 352       write_summary = ops.convert_to_tensor(write_summary)
 353     t = self.numer / self.denom
 354     def write_summary_f():
 355       summary_ops.scalar(name=self.name, tensor=t)
 356       return t
 357     control_flow_ops.cond(write_summary,
 358                           write_summary_f,
 359                           lambda: t)
 360     return t

 363 class Accuracy(Mean):
 364   """Calculates how often `predictions` matches `labels`.
 365   Attributes:
 366     name: name of the accuracy object
 367     dtype: data type of the tensor
 368   """
 369 
 370   def __init__(self, name=None, dtype=dtypes.float64):
 371     """Inits Accuracy class with name and dtype."""
 372     super(Accuracy, self).__init__(name=name, dtype=dtype)
 373 
 374   def call(self, labels, predictions, weights=None):
 375     """Accumulate accuracy statistics.
 376 
 377     For example, if labels is [1, 2, 3, 4] and predictions is [0, 2, 3, 4]
 378     then the accuracy is 3/4 or .75.  If the weights were specified as
 379     [1, 1, 0, 0] then the accuracy would be 1/2 or .5.
 380 
 381     `labels` and `predictions` should have the same shape and type.
 382 
 383     Args:
 384       labels: Tensor with the true labels for each example.  One example
 385         per element of the Tensor.
 386       predictions: Tensor with the predicted label for each example.
 387       weights: Optional weighting of each example. Defaults to 1.
 388 
 389     Returns:
 390       The arguments, for easy chaining.
 391     """
 392     check_ops.assert_equal(
 393         array_ops.shape(labels), array_ops.shape(predictions),
 394         message="Shapes of labels and predictions are unequal")
 395     matches = math_ops.equal(labels, predictions)
 396     matches = math_ops.cast(matches, self.dtype)
 397     super(Accuracy, self).call(matches, weights=weights)
 398     if weights is None:
 399       return labels, predictions
 400     return labels, predictions, weights

 403 class CategoricalAccuracy(Mean):
 404   """Calculates how often `predictions` matches `labels`.
 405 
 406   This class is compatible with `tf.keras.losses.categorical_crossentropy`,
 407   `tf.nn.softmax_cross_entropy_with_logits_v2`,
 408   `tf.losses.softmax_cross_entropy`.
 409 
 410   Attributes:
 411     name: name of the accuracy object.
 412     dtype: data type of tensor.
 413   """
 414 
 415   def __init__(self, name=None, dtype=dtypes.float64):
 416     """Inits CategoricalAccuracy with name and dtype."""
 417     super(CategoricalAccuracy, self).__init__(name=name, dtype=dtype)
 418 
 419   def call(self, labels, predictions, weights=None):
 420     """Accumulate accuracy statistics.
 421 
 422     `labels` and `predictions` should have the same shape.
 423     As argmax is being done here, labels and predictions type
 424     can be different.
 425 
 426     Args:
 427       labels: One-hot Tensor.
 428       predictions: Tensor with the logits or probabilities for each example.
 429       weights: Optional weighting of each example. Defaults to 1.
 430 
 431     Returns:
 432       The arguments, for easy chaining.
 433     """
 434     check_ops.assert_equal(
 435         array_ops.shape(labels), array_ops.shape(predictions),
 436         message="Shapes of labels and predictions are unequal")
 437     labels = math_ops.argmax(labels, axis=-1)
 438     predictions = math_ops.argmax(predictions, axis=-1)
 439     matches = math_ops.equal(labels, predictions)
 440     matches = math_ops.cast(matches, self.dtype)
 441     super(CategoricalAccuracy, self).call(matches, weights=weights)
 442     if weights is None:
 443       return labels, predictions
 444     return labels, predictions, weights

 447 class BinaryAccuracy(Mean):
 448   """Calculates how often `predictions` matches `labels`.
 449 
 450   This class is compatible with `tf.keras.losses.binary_crossentropy`,
 451   `tf.losses.sigmoid_cross_entropy`,
 452   `tf.nn.sigmoid_cross_entropy_with_logits`.
 453   If there is more than one label, this will become multi-label classification.
 454 
 455   Attributes:
 456     name: name of the accuracy object.
 457     threshold: Used for rounding off the predictions.
 458                If the predictions are,
 459                 1. probabilities then set the threshold to 0.5.
 460                 2. logits then set the threshold to 0.
 461               You can set the threshold appropriately,
 462               to trade off with precision and recall.
 463     dtype: data type of tensor.
 464   """
 465 
 466   def __init__(self, threshold, name=None, dtype=dtypes.float64):
 467     """Inits BinaryAccuracy with name, threshold and dtype."""
 468 
 469     super(BinaryAccuracy, self).__init__(name=name, dtype=dtype)
 470     self.threshold = threshold
 471 
 472   def call(self, labels, predictions, weights=None):
 473     """Accumulate accuracy statistics.
 474 
 475     `labels` and `predictions` should have the same shape and type.
 476 
 477     Args:
 478       labels: Binary Tensor(containing 0 or 1).
 479       predictions: Tensor with probabilities or logits.
 480       weights: Optional weighting of each example. Defaults to 1.
 481 
 482     Returns:
 483       The arguments, for easy chaining.
 484     """
 485     check_ops.assert_equal(
 486         array_ops.shape(labels), array_ops.shape(predictions),
 487         message="Shapes of labels and predictions are unequal")
 488     predictions = ops.convert_to_tensor(predictions)
 489     predictions = predictions > self.threshold
 490     matches = math_ops.equal(labels, predictions)
 491     matches = math_ops.cast(matches, self.dtype)
 492     super(BinaryAccuracy, self).call(matches, weights=weights)
 493     if weights is None:
 494       return labels, predictions
 495     return labels, predictions, weights

 498 class SparseAccuracy(Mean):
 499   """Calculates how often `predictions` matches `labels`.
 500 
 501   This class is compatible with
 502   `tf.keras.losses.sparse_categorical_crossentropy`,
 503   `tf.nn.sparse_softmax_cross_entropy_with_logits`,
 504   `tf.losses.sparse_softmax_cross_entropy`.
 505 
 506   Attributes:
 507     name: name of the accuracy object
 508     dtype: data type of tensor.
 509   """
 510 
 511   def __init__(self, name=None, dtype=dtypes.float64):
 512     """Inits SparseAccuracy with name and dtype."""
 513 
 514     super(SparseAccuracy, self).__init__(name=name, dtype=dtype)
 515 
 516   def call(self, labels, predictions, weights=None):
 517     """Accumulate accuracy statistics.
 518 
 519     `labels` and `predictions` should have the same shape except the
 520     predictions must have one additional trailing dimension equal to the
 521     number of classes(you want to predict).
 522 
 523     Type of labels and predictions can be different.
 524 
 525     Args:
 526       labels: Tensor of shape (batch_size, ) containing integers
 527       predictions: Tensor with the logits or probabilities for each example.
 528       weights: Optional weighting of each example. Defaults to 1.
 529 
 530     Returns:
 531       The arguments, for easy chaining.
 532     """
 533     check_ops.assert_equal(
 534         array_ops.shape(labels), array_ops.shape(predictions)[0],
 535         message="First axis of labels and predictions is unequal")
 536     predictions = math_ops.argmax(predictions, axis=-1)
 537     labels = math_ops.cast(labels, dtypes.int64)
 538     matches = math_ops.equal(labels, predictions)
 539     matches = math_ops.cast(matches, self.dtype)
 540     super(SparseAccuracy, self).call(matches, weights=weights)
 541     if weights is None:
 542       return labels, predictions
 543     return labels, predictions, weights
