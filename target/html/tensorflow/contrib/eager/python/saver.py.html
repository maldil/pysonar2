  30 def _init_from_checkpoint(self, *args, **kwargs):
  31   """Overrides default init by loading value from checkpoint."""
  32   # pylint: disable=protected-access
  33   self._old_init(*args, **kwargs)
  34   ckpt_name = self._map_func(self._shared_name)
  35   if ckpt_name not in self._ckpt_var_cache:
  36     raise errors.NotFoundError(None, None,
  37                                "%s not found in checkpoint" % ckpt_name)
  38 
  39   val = self._ckpt_var_cache.get(ckpt_name, None)
  40   if val is not None:
  41     self.assign(val)
  42     # Avoid assigning for the second time.
  43     self._ckpt_var_cache[ckpt_name] = None
  44   # pylint: enable=protected-access

  47 @contextlib.contextmanager
  48 def restore_variables_on_create(save_path, map_func=None):
  49   """ContextManager that restores variables on creation.
  50 
  51     When save_path is None (e.g. No checkpoint), does nothing.
  52     Otherwise, it preloads all values from checkpoint. When the
  53     corresponding variable is first created, it assigns the checkpoint
  54     value to the variable.
  55 
  56     ```python
  57     with restore_variables_on_create(
  58         tf.train.latest_checkpoint(checkpoint_dir)):
  59     ```
  60 
  61   Args:
  62     save_path: The checkpoint file prefix.
  63     map_func: A function that given the variable name as argument
  64         and returns a variable name in checkpoint for restore. If
  65         None, use the variable with the same name in checkpoint to restore.
  66         It's an error that the mapped variable name doesn't exist in
  67         checkpoint.
  68 
  69   Yields:
  70     Nothing.
  71 
  72   Raises:
  73     NotFoundError: If the variable is not found in checkpoint.
  74     ValueError: If not used in eager mode or map_func is not callable.
  75   """
  76   if not context.executing_eagerly():
  77     raise ValueError(
  78         "Currently, restore_variables_on_create can only be used with "
  79         "eager execution enabled.")
  80   if save_path:
  81     if map_func is None:
  82       map_func_wrapper = lambda self, x: x
  83     else:
  84       if not callable(map_func):
  85         raise ValueError("map_func must be callable.")
  86       map_func_wrapper = lambda self, x: map_func(x)
  87 
  88     ckpt_var_cache = dict()
  89     reader = checkpoint_utils.load_checkpoint(save_path)
  90     for k, _ in checkpoint_utils.list_variables(save_path):
  91       ckpt_var_cache[k] = reader.get_tensor(k)
  92 
  93     old_init = getattr(resource_variable_ops.ResourceVariable,
  94                        "_init_from_args", None)
  95     assert old_init, "ResourceVariable misses _init_from_args method."
  96     setattr(resource_variable_ops.ResourceVariable, "_init_from_args",
  97             _init_from_checkpoint)
  98     setattr(resource_variable_ops.ResourceVariable, "_old_init", old_init)
  99     setattr(resource_variable_ops.ResourceVariable, "_map_func",
 100             map_func_wrapper)
 101     setattr(resource_variable_ops.ResourceVariable, "_ckpt_var_cache",
 102             ckpt_var_cache)
 103   try:
 104     yield
 105   except Exception as e:
 106     raise e
 107   finally:
 108     if save_path:
 109       setattr(resource_variable_ops.ResourceVariable, "_init_from_args",
 110               old_init)
 111       setattr(resource_variable_ops.ResourceVariable, "_old_init", None)
 112       setattr(resource_variable_ops.ResourceVariable, "_map_func", None)
 113       setattr(resource_variable_ops.ResourceVariable, "_ckpt_var_cache", None)

 116 class Saver(object):
 117   """A tf.train.Saver adapter for use when eager execution is enabled.
 118   """
 119 
 120   def __init__(self, var_list):
 121     """A  tf.train.Saver adapter for use when eager execution is enabled.
 122 
 123       The API, and on-disk format, mimic tf.train.Saver except that no
 124       Session is needed.
 125 
 126     Args:
 127       var_list: The list of variables that will be saved and restored. Either a
 128         list of `tf.Variable` objects, or a dictionary mapping names to
 129         `tf.Variable` objects.
 130 
 131     Raises:
 132       RuntimeError: if invoked when eager execution has not been enabled.
 133     """
 134     if not context.executing_eagerly():
 135       raise RuntimeError("tfe.Saver can only be used when eager "
 136                          "execution is enabled. Use tf.train.Saver when "
 137                          "building graphs.")
 138     self._saver = _saver.Saver(var_list=var_list)
 139 
 140   def save(self, file_prefix, global_step=None):
 141     """Saves variables.
 142 
 143     Args:
 144       file_prefix: Path prefix of files created for the checkpoint.
 145       global_step: If provided the global step number is appended to file_prefix
 146         to create the checkpoint filename. The optional argument can be a
 147         Tensor, a Variable, or an integer.
 148 
 149     Returns:
 150       A string: prefix of filenames created for the checkpoint. This may be
 151        an extension of file_prefix that is suitable to pass as an argument
 152        to a subsequent call to `restore()`.
 153     """
 154     with ops.device("/device:CPU:0"):
 155       return self._saver.save(
 156           None, file_prefix, write_meta_graph=False, global_step=global_step)
 157 
 158   def restore(self, file_prefix):
 159     """Restores previously saved variables.
 160 
 161     Args:
 162       file_prefix: Path prefix where parameters were previously saved.
 163         Typically obtained from a previous `save()` call, or from
 164         `tf.train.latest_checkpoint`.
 165     """
 166     with ops.device("/device:CPU:0"):
 167       self._saver.restore(None, file_prefix)

 170 def get_optimizer_variables(optimizer):
 171   """Returns a list of variables for the given `tf.train.Optimizer`.
 172 
 173   Equivalent to `optimizer.variables()`.
 174 
 175   Args:
 176     optimizer: An instance of `tf.train.Optimizer` which has created variables
 177       (typically after a call to `Optimizer.minimize`).
 178   Returns:
 179     A list of variables which have been created by the `Optimizer`.
 180   """
 181   return optimizer.variables()
