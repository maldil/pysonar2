  34 class RevBlock(tf.keras.Model):
  35   """Single reversible block containing several `_Residual` blocks.
  36 
  37   Each `_Residual` block in turn contains two _ResidualInner blocks,
  38   corresponding to the `F`/`G` functions in the paper.
  39   """
  40 
  41   def __init__(self,
  42                n_res,
  43                filters,
  44                strides,
  45                input_shape,
  46                batch_norm_first=False,
  47                data_format="channels_first",
  48                bottleneck=False,
  49                fused=True,
  50                dtype=tf.float32):
  51     """Initialization.
  52 
  53     Args:
  54       n_res: number of residual blocks
  55       filters: list/tuple of integers for output filter sizes of each residual
  56       strides: length 2 list/tuple of integers for height and width strides
  57       input_shape: length 3 list/tuple of integers
  58       batch_norm_first: whether to apply activation and batch norm before conv
  59       data_format: tensor data format, "NCHW"/"NHWC"
  60       bottleneck: use bottleneck residual if True
  61       fused: use fused batch normalization if True
  62       dtype: float16, float32, or float64
  63     """
  64     super(RevBlock, self).__init__()
  65     self.blocks = tf.contrib.checkpoint.List()
  66     for i in range(n_res):
  67       curr_batch_norm_first = batch_norm_first and i == 0
  68       curr_strides = strides if i == 0 else (1, 1)
  69       block = _Residual(
  70           filters,
  71           curr_strides,
  72           input_shape,
  73           batch_norm_first=curr_batch_norm_first,
  74           data_format=data_format,
  75           bottleneck=bottleneck,
  76           fused=fused,
  77           dtype=dtype)
  78       self.blocks.append(block)
  79 
  80       if data_format == "channels_first":
  81         input_shape = (filters, input_shape[1] // curr_strides[0],
  82                        input_shape[2] // curr_strides[1])
  83       else:
  84         input_shape = (input_shape[0] // curr_strides[0],
  85                        input_shape[1] // curr_strides[1], filters)
  86 
  87   def call(self, h, training=True):
  88     """Apply reversible block to inputs."""
  89 
  90     for block in self.blocks:
  91       h = block(h, training=training)
  92     return h
  93 
  94   def backward_grads(self, x, y, dy, training=True):
  95     """Apply reversible block backward to outputs."""
  96 
  97     grads_all = []
  98     for i in reversed(range(len(self.blocks))):
  99       block = self.blocks[i]
 100       if i == 0:
 101         # First block usually contains downsampling that can't be reversed
 102         dy, grads = block.backward_grads_with_downsample(
 103             x, y, dy, training=True)
 104       else:
 105         y, dy, grads = block.backward_grads(y, dy, training=training)
 106       grads_all = grads + grads_all
 107 
 108     return dy, grads_all

 111 class _Residual(tf.keras.Model):
 112   """Single residual block contained in a _RevBlock. Each `_Residual` object has
 113   two _ResidualInner objects, corresponding to the `F` and `G` functions in the
 114   paper.
 115   """
 116 
 117   def __init__(self,
 118                filters,
 119                strides,
 120                input_shape,
 121                batch_norm_first=True,
 122                data_format="channels_first",
 123                bottleneck=False,
 124                fused=True,
 125                dtype=tf.float32):
 126     """Initialization.
 127 
 128     Args:
 129       filters: output filter size
 130       strides: length 2 list/tuple of integers for height and width strides
 131       input_shape: length 3 list/tuple of integers
 132       batch_norm_first: whether to apply activation and batch norm before conv
 133       data_format: tensor data format, "NCHW"/"NHWC",
 134       bottleneck: use bottleneck residual if True
 135       fused: use fused batch normalization if True
 136       dtype: float16, float32, or float64
 137     """
 138     super(_Residual, self).__init__()
 139 
 140     self.filters = filters
 141     self.strides = strides
 142     self.axis = 1 if data_format == "channels_first" else 3
 143     if data_format == "channels_first":
 144       f_input_shape = (input_shape[0] // 2,) + input_shape[1:]
 145       g_input_shape = (filters // 2, input_shape[1] // strides[0],
 146                        input_shape[2] // strides[1])
 147     else:
 148       f_input_shape = input_shape[:2] + (input_shape[2] // 2,)
 149       g_input_shape = (input_shape[0] // strides[0],
 150                        input_shape[1] // strides[1], filters // 2)
 151 
 152     factory = _BottleneckResidualInner if bottleneck else _ResidualInner
 153     self.f = factory(
 154         filters=filters // 2,
 155         strides=strides,
 156         input_shape=f_input_shape,
 157         batch_norm_first=batch_norm_first,
 158         data_format=data_format,
 159         fused=fused,
 160         dtype=dtype)
 161     self.g = factory(
 162         filters=filters // 2,
 163         strides=(1, 1),
 164         input_shape=g_input_shape,
 165         batch_norm_first=batch_norm_first,
 166         data_format=data_format,
 167         fused=fused,
 168         dtype=dtype)
 169 
 170   def call(self, x, training=True):
 171     """Apply residual block to inputs."""
 172     x1, x2 = x
 173     f_x2 = self.f(x2, training=training)
 174     x1_down = ops.downsample(
 175         x1, self.filters // 2, self.strides, axis=self.axis)
 176     x2_down = ops.downsample(
 177         x2, self.filters // 2, self.strides, axis=self.axis)
 178     y1 = f_x2 + x1_down
 179     g_y1 = self.g(y1, training=training)
 180     y2 = g_y1 + x2_down
 181 
 182     return y1, y2
 183 
 184   def backward_grads(self, y, dy, training=True):
 185     """Manually compute backward gradients given input and output grads."""
 186     dy1, dy2 = dy
 187     y1, y2 = y
 188 
 189     with tf.GradientTape() as gtape:
 190       gtape.watch(y1)
 191       gy1 = self.g(y1, training=training)
 192     grads_combined = gtape.gradient(
 193         gy1, [y1] + self.g.trainable_variables, output_gradients=dy2)
 194     dg = grads_combined[1:]
 195     dx1 = dy1 + grads_combined[0]
 196     # This doesn't affect eager execution, but improves memory efficiency with
 197     # graphs
 198     with tf.control_dependencies(dg + [dx1]):
 199       x2 = y2 - gy1
 200 
 201     with tf.GradientTape() as ftape:
 202       ftape.watch(x2)
 203       fx2 = self.f(x2, training=training)
 204     grads_combined = ftape.gradient(
 205         fx2, [x2] + self.f.trainable_variables, output_gradients=dx1)
 206     df = grads_combined[1:]
 207     dx2 = dy2 + grads_combined[0]
 208     # Same behavior as above
 209     with tf.control_dependencies(df + [dx2]):
 210       x1 = y1 - fx2
 211 
 212     x = x1, x2
 213     dx = dx1, dx2
 214     grads = df + dg
 215 
 216     return x, dx, grads
 217 
 218   def backward_grads_with_downsample(self, x, y, dy, training=True):
 219     """Manually compute backward gradients given input and output grads."""
 220     # Splitting this from `backward_grads` for better readability
 221     x1, x2 = x
 222     y1, _ = y
 223     dy1, dy2 = dy
 224 
 225     with tf.GradientTape() as gtape:
 226       gtape.watch(y1)
 227       gy1 = self.g(y1, training=training)
 228     grads_combined = gtape.gradient(
 229         gy1, [y1] + self.g.trainable_variables, output_gradients=dy2)
 230     dg = grads_combined[1:]
 231     dz1 = dy1 + grads_combined[0]
 232 
 233     # dx1 need one more step to backprop through downsample
 234     with tf.GradientTape() as x1tape:
 235       x1tape.watch(x1)
 236       z1 = ops.downsample(x1, self.filters // 2, self.strides, axis=self.axis)
 237     dx1 = x1tape.gradient(z1, x1, output_gradients=dz1)
 238 
 239     with tf.GradientTape() as ftape:
 240       ftape.watch(x2)
 241       fx2 = self.f(x2, training=training)
 242     grads_combined = ftape.gradient(
 243         fx2, [x2] + self.f.trainable_variables, output_gradients=dz1)
 244     dx2, df = grads_combined[0], grads_combined[1:]
 245 
 246     # dx2 need one more step to backprop through downsample
 247     with tf.GradientTape() as x2tape:
 248       x2tape.watch(x2)
 249       z2 = ops.downsample(x2, self.filters // 2, self.strides, axis=self.axis)
 250     dx2 += x2tape.gradient(z2, x2, output_gradients=dy2)
 251 
 252     dx = dx1, dx2
 253     grads = df + dg
 254 
 255     return dx, grads

 258 # Ideally, the following should be wrapped in `tf.keras.Sequential`, however
 259 # there are subtle issues with its placeholder insertion policy and batch norm
 260 class _BottleneckResidualInner(tf.keras.Model):
 261   """Single bottleneck residual inner function contained in _Resdual.
 262 
 263   Corresponds to the `F`/`G` functions in the paper.
 264   Suitable for training on ImageNet dataset.
 265   """
 266 
 267   def __init__(self,
 268                filters,
 269                strides,
 270                input_shape,
 271                batch_norm_first=True,
 272                data_format="channels_first",
 273                fused=True,
 274                dtype=tf.float32):
 275     """Initialization.
 276 
 277     Args:
 278       filters: output filter size
 279       strides: length 2 list/tuple of integers for height and width strides
 280       input_shape: length 3 list/tuple of integers
 281       batch_norm_first: whether to apply activation and batch norm before conv
 282       data_format: tensor data format, "NCHW"/"NHWC"
 283       fused: use fused batch normalization if True
 284       dtype: float16, float32, or float64
 285     """
 286     super(_BottleneckResidualInner, self).__init__()
 287     axis = 1 if data_format == "channels_first" else 3
 288     if batch_norm_first:
 289       self.batch_norm_0 = tf.keras.layers.BatchNormalization(
 290           axis=axis, input_shape=input_shape, fused=fused, dtype=dtype)
 291     self.conv2d_1 = tf.keras.layers.Conv2D(
 292         filters=filters // 4,
 293         kernel_size=1,
 294         strides=strides,
 295         input_shape=input_shape,
 296         data_format=data_format,
 297         use_bias=False,
 298         padding="SAME",
 299         dtype=dtype)
 300 
 301     self.batch_norm_1 = tf.keras.layers.BatchNormalization(
 302         axis=axis, fused=fused, dtype=dtype)
 303     self.conv2d_2 = tf.keras.layers.Conv2D(
 304         filters=filters // 4,
 305         kernel_size=3,
 306         strides=(1, 1),
 307         data_format=data_format,
 308         use_bias=False,
 309         padding="SAME",
 310         dtype=dtype)
 311 
 312     self.batch_norm_2 = tf.keras.layers.BatchNormalization(
 313         axis=axis, fused=fused, dtype=dtype)
 314     self.conv2d_3 = tf.keras.layers.Conv2D(
 315         filters=filters,
 316         kernel_size=1,
 317         strides=(1, 1),
 318         data_format=data_format,
 319         use_bias=False,
 320         padding="SAME",
 321         dtype=dtype)
 322 
 323     self.batch_norm_first = batch_norm_first
 324 
 325   def call(self, x, training=True):
 326     net = x
 327     if self.batch_norm_first:
 328       net = self.batch_norm_0(net, training=training)
 329       net = tf.nn.relu(net)
 330     net = self.conv2d_1(net)
 331 
 332     net = self.batch_norm_1(net, training=training)
 333     net = tf.nn.relu(net)
 334     net = self.conv2d_2(net)
 335 
 336     net = self.batch_norm_2(net, training=training)
 337     net = tf.nn.relu(net)
 338     net = self.conv2d_3(net)
 339 
 340     return net

 410 class InitBlock(tf.keras.Model):
 411   """Initial block of RevNet."""
 412 
 413   def __init__(self, config):
 414     """Initialization.
 415 
 416     Args:
 417       config: tf.contrib.training.HParams object; specifies hyperparameters
 418     """
 419     super(InitBlock, self).__init__()
 420     self.config = config
 421     self.axis = 1 if self.config.data_format == "channels_first" else 3
 422     self.conv2d = tf.keras.layers.Conv2D(
 423         filters=self.config.init_filters,
 424         kernel_size=self.config.init_kernel,
 425         strides=(self.config.init_stride, self.config.init_stride),
 426         data_format=self.config.data_format,
 427         use_bias=False,
 428         padding="SAME",
 429         input_shape=self.config.input_shape,
 430         dtype=self.config.dtype)
 431     self.batch_norm = tf.keras.layers.BatchNormalization(
 432         axis=self.axis, fused=self.config.fused, dtype=self.config.dtype)
 433     self.activation = tf.keras.layers.Activation("relu")
 434 
 435     if self.config.init_max_pool:
 436       self.max_pool = tf.keras.layers.MaxPooling2D(
 437           pool_size=(3, 3),
 438           strides=(2, 2),
 439           padding="SAME",
 440           data_format=self.config.data_format,
 441           dtype=self.config.dtype)
 442 
 443   def call(self, x, training=True):
 444     net = x
 445     net = self.conv2d(net)
 446     net = self.batch_norm(net, training=training)
 447     net = self.activation(net)
 448 
 449     if self.config.init_max_pool:
 450       net = self.max_pool(net)
 451 
 452     return tf.split(net, num_or_size_splits=2, axis=self.axis)

 455 class FinalBlock(tf.keras.Model):
 456   """Final block of RevNet."""
 457 
 458   def __init__(self, config):
 459     """Initialization.
 460 
 461     Args:
 462       config: tf.contrib.training.HParams object; specifies hyperparameters
 463 
 464     Raises:
 465       ValueError: Unsupported data format
 466     """
 467     super(FinalBlock, self).__init__()
 468     self.config = config
 469     self.axis = 1 if self.config.data_format == "channels_first" else 3
 470 
 471     f = self.config.filters[-1]  # Number of filters
 472     r = functools.reduce(operator.mul, self.config.strides, 1)  # Reduce ratio
 473     r *= self.config.init_stride
 474     if self.config.init_max_pool:
 475       r *= 2
 476 
 477     if self.config.data_format == "channels_first":
 478       w, h = self.config.input_shape[1], self.config.input_shape[2]
 479       input_shape = (f, w // r, h // r)
 480     elif self.config.data_format == "channels_last":
 481       w, h = self.config.input_shape[0], self.config.input_shape[1]
 482       input_shape = (w // r, h // r, f)
 483     else:
 484       raise ValueError("Data format should be either `channels_first`"
 485                        " or `channels_last`")
 486     self.batch_norm = tf.keras.layers.BatchNormalization(
 487         axis=self.axis,
 488         input_shape=input_shape,
 489         fused=self.config.fused,
 490         dtype=self.config.dtype)
 491     self.activation = tf.keras.layers.Activation("relu")
 492     self.global_avg_pool = tf.keras.layers.GlobalAveragePooling2D(
 493         data_format=self.config.data_format, dtype=self.config.dtype)
 494     self.dense = tf.keras.layers.Dense(
 495         self.config.n_classes, dtype=self.config.dtype)
 496 
 497   def call(self, x, training=True):
 498     net = tf.concat(x, axis=self.axis)
 499     net = self.batch_norm(net, training=training)
 500     net = self.activation(net)
 501     net = self.global_avg_pool(net)
 502     net = self.dense(net)
 503 
 504     return net
