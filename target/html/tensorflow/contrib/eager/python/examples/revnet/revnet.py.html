  15 """Reversible residual network compatible with eager execution.
  16 
  17 Code for main model.
  18 
  19 Reference [The Reversible Residual Network: Backpropagation
  20 Without Storing Activations](https://arxiv.org/pdf/1707.04585.pdf)
  21 """
  22 
  23 from __future__ import absolute_import
  24 from __future__ import division
  25 from __future__ import print_function
  26 
  27 import tensorflow as tf
  28 from tensorflow.contrib.eager.python.examples.revnet import blocks
  29 
  30 
  31 class RevNet(tf.keras.Model):
  32   """RevNet that depends on all the blocks."""
  33 
  34   def __init__(self, config):
  35     """Initialize RevNet with building blocks.
  36 
  37     Args:
  38       config: tf.contrib.training.HParams object; specifies hyperparameters
  39     """
  40     super(RevNet, self).__init__()
  41     self.axis = 1 if config.data_format == "channels_first" else 3
  42     self.config = config
  43 
  44     self._init_block = blocks.InitBlock(config=self.config)
  45     self._final_block = blocks.FinalBlock(config=self.config)
  46     self._block_list = self._construct_intermediate_blocks()
  47     self._moving_average_variables = []
  48 
  49   def _construct_intermediate_blocks(self):
  50     # Precompute input shape after initial block
  51     stride = self.config.init_stride
  52     if self.config.init_max_pool:
  53       stride *= 2
  54     if self.config.data_format == "channels_first":
  55       w, h = self.config.input_shape[1], self.config.input_shape[2]
  56       input_shape = (self.config.init_filters, w // stride, h // stride)
  57     else:
  58       w, h = self.config.input_shape[0], self.config.input_shape[1]
  59       input_shape = (w // stride, h // stride, self.config.init_filters)
  60 
  61     # Aggregate intermediate blocks
  62     block_list = tf.contrib.checkpoint.List()
  63     for i in range(self.config.n_rev_blocks):
  64       # RevBlock configurations
  65       n_res = self.config.n_res[i]
  66       filters = self.config.filters[i]
  67       if filters % 2 != 0:
  68         raise ValueError("Number of output filters must be even to ensure"
  69                          "correct partitioning of channels")
  70       stride = self.config.strides[i]
  71       strides = (self.config.strides[i], self.config.strides[i])
  72 
  73       # Add block
  74       rev_block = blocks.RevBlock(
  75           n_res,
  76           filters,
  77           strides,
  78           input_shape,
  79           batch_norm_first=(i != 0),  # Only skip on first block
  80           data_format=self.config.data_format,
  81           bottleneck=self.config.bottleneck,
  82           fused=self.config.fused,
  83           dtype=self.config.dtype)
  84       block_list.append(rev_block)
  85 
  86       # Precompute input shape for the next block
  87       if self.config.data_format == "channels_first":
  88         w, h = input_shape[1], input_shape[2]
  89         input_shape = (filters, w // stride, h // stride)
  90       else:
  91         w, h = input_shape[0], input_shape[1]
  92         input_shape = (w // stride, h // stride, filters)
  93 
  94     return block_list
  95 
  96   def call(self, inputs, training=True):
  97     """Forward pass."""
  98 
  99     if training:
 100       saved_hidden = [inputs]
 101 
 102     h = self._init_block(inputs, training=training)
 103     if training:
 104       saved_hidden.append(h)
 105 
 106     for block in self._block_list:
 107       h = block(h, training=training)
 108       if training:
 109         saved_hidden.append(h)
 110 
 111     logits = self._final_block(h, training=training)
 112 
 113     return (logits, saved_hidden) if training else (logits, None)
 114 
 115   def compute_loss(self, logits, labels):
 116     """Compute cross entropy loss."""
 117 
 118     if self.config.dtype == tf.float32 or self.config.dtype == tf.float16:
 119       cross_ent = tf.nn.sparse_softmax_cross_entropy_with_logits(
 120           logits=logits, labels=labels)
 121     else:
 122       # `sparse_softmax_cross_entropy_with_logits` does not have a GPU kernel
 123       # for float64, int32 pairs
 124       labels = tf.one_hot(
 125           labels, depth=self.config.n_classes, axis=1, dtype=self.config.dtype)
 126       cross_ent = tf.nn.softmax_cross_entropy_with_logits(
 127           logits=logits, labels=labels)
 128 
 129     return tf.reduce_mean(cross_ent)
 130 
 131   def compute_gradients(self, saved_hidden, labels, training=True, l2_reg=True):
 132     """Manually computes gradients.
 133 
 134     This method silently updates the running averages of batch normalization.
 135 
 136     Args:
 137       saved_hidden: List of hidden states Tensors
 138       labels: One-hot labels for classification
 139       training: Use the mini-batch stats in batch norm if set to True
 140       l2_reg: Apply l2 regularization
 141 
 142     Returns:
 143       A tuple with the first entry being a list of all gradients and the second
 144       being the loss
 145     """
 146 
 147     def _defunable_pop(l):
 148       """Functional style list pop that works with `tfe.defun`."""
 149       t, l = l[-1], l[:-1]
 150       return t, l
 151 
 152     # Backprop through last block
 153     x = saved_hidden[-1]
 154     with tf.GradientTape() as tape:
 155       tape.watch(x)
 156       logits = self._final_block(x, training=training)
 157       loss = self.compute_loss(logits, labels)
 158     grads_combined = tape.gradient(loss,
 159                                    [x] + self._final_block.trainable_variables)
 160     dy, final_grads = grads_combined[0], grads_combined[1:]
 161 
 162     # Backprop through intermediate blocks
 163     intermediate_grads = []
 164     for block in reversed(self._block_list):
 165       y, saved_hidden = _defunable_pop(saved_hidden)
 166       x = saved_hidden[-1]
 167       dy, grads = block.backward_grads(x, y, dy, training=training)
 168       intermediate_grads = grads + intermediate_grads
 169 
 170     # Backprop through first block
 171     _, saved_hidden = _defunable_pop(saved_hidden)
 172     x, saved_hidden = _defunable_pop(saved_hidden)
 173     assert not saved_hidden
 174     with tf.GradientTape() as tape:
 175       y = self._init_block(x, training=training)
 176     init_grads = tape.gradient(
 177         y, self._init_block.trainable_variables, output_gradients=dy)
 178 
 179     # Ordering match up with `model.trainable_variables`
 180     grads_all = init_grads + final_grads + intermediate_grads
 181     if l2_reg:
 182       grads_all = self._apply_weight_decay(grads_all)
 183 
 184     return grads_all, loss
 185 
 186   def _apply_weight_decay(self, grads):
 187     """Update gradients to reflect weight decay."""
 188     return [
 189         g + self.config.weight_decay * v if v.name.endswith("kernel:0") else g
 190         for g, v in zip(grads, self.trainable_variables)
 191     ]

 193   def get_moving_stats(self):
 194     """Get moving averages of batch normalization."""
 195     device = "/gpu:0" if tf.test.is_gpu_available() else "/cpu:0"
 196     with tf.device(device):
 197       return [v.read_value() for v in self.moving_average_variables]
 198 
 199   def restore_moving_stats(self, values):
 200     """Restore moving averages of batch normalization."""
 201     device = "/gpu:0" if tf.test.is_gpu_available() else "/cpu:0"
 202     with tf.device(device):
 203       for var_, val in zip(self.moving_average_variables, values):
 204         var_.assign(val)
 205 
 206   @property
 207   def moving_average_variables(self):
 208     """Get all variables that are batch norm moving averages."""
 209 
 210     def _is_moving_avg(v):
 211       n = v.name
 212       return n.endswith("moving_mean:0") or n.endswith("moving_variance:0")
 213 
 214     if not self._moving_average_variables:
 215       self._moving_average_variables = filter(_is_moving_avg, self.variables)
 216 
 217     return self._moving_average_variables
