  15 """Reversible residual network compatible with eager execution.
  16 
  17 Configuration in format of tf.contrib.training.HParams.
  18 Supports CIFAR-10, CIFAR-100, and ImageNet datasets.
  19 
  20 Reference [The Reversible Residual Network: Backpropagation
  21 Without Storing Activations](https://arxiv.org/pdf/1707.04585.pdf)
  22 
  23 """
  24 
  25 from __future__ import absolute_import
  26 from __future__ import division
  27 from __future__ import print_function
  28 
  29 import tensorflow as tf
  30 
  31 
  32 def get_hparams_cifar_38():
  33   """RevNet-38 configurations for CIFAR-10/CIFAR-100."""
  34 
  35   config = tf.contrib.training.HParams()
  36   config.add_hparam("num_train_images", 50000)
  37   config.add_hparam("num_eval_images", 10000)
  38   config.add_hparam("init_filters", 32)
  39   config.add_hparam("init_kernel", 3)
  40   config.add_hparam("init_stride", 1)
  41   config.add_hparam("n_rev_blocks", 3)
  42   config.add_hparam("n_res", [3, 3, 3])
  43   config.add_hparam("filters", [32, 64, 112])
  44   config.add_hparam("strides", [1, 2, 2])
  45   config.add_hparam("batch_size", 100)
  46   config.add_hparam("bottleneck", False)
  47   config.add_hparam("fused", True)
  48   config.add_hparam("init_max_pool", False)
  49   if tf.test.is_gpu_available():
  50     config.add_hparam("input_shape", (3, 32, 32))
  51     config.add_hparam("data_format", "channels_first")
  52   else:
  53     config.add_hparam("input_shape", (32, 32, 3))
  54     config.add_hparam("data_format", "channels_last")
  55 
  56   # Training details
  57   config.add_hparam("weight_decay", 2e-4)
  58   config.add_hparam("momentum", .9)
  59   config.add_hparam("lr_decay_steps", [40000, 60000])
  60   config.add_hparam("lr_list", [1e-1, 1e-2, 1e-3])
  61   config.add_hparam("max_train_iter", 80000)
  62   config.add_hparam("seed", 1234)
  63   config.add_hparam("shuffle", True)
  64   config.add_hparam("log_every", 500)
  65   config.add_hparam("save_every", 500)
  66   config.add_hparam("dtype", tf.float32)
  67   config.add_hparam("eval_batch_size", 1000)
  68   config.add_hparam("div255", True)
  69   # This is imprecise, when training with validation set,
  70   # we only have 40k images in training data
  71   config.add_hparam("iters_per_epoch",
  72                     config.num_train_images // config.batch_size)
  73   config.add_hparam("epochs", config.max_train_iter // config.iters_per_epoch)
  74 
  75   # Customized TPU hyperparameters due to differing batch size caused by
  76   # TPU architecture specifics
  77   # Suggested batch sizes to reduce overhead from excessive tensor padding
  78   # https://cloud.google.com/tpu/docs/troubleshooting
  79   config.add_hparam("tpu_batch_size", 1024)
  80   config.add_hparam("tpu_eval_batch_size", 1024)
  81   config.add_hparam("tpu_iters_per_epoch",
  82                     config.num_train_images // config.tpu_batch_size)
  83   config.add_hparam("tpu_epochs",
  84                     config.max_train_iter // config.tpu_iters_per_epoch)
  85   config.add_hparam("tpu_eval_steps",
  86                     config.num_eval_images // config.tpu_eval_batch_size)
  87   return config
  88

  90 def get_hparams_cifar_110():
  91   config = get_hparams_cifar_38()
  92   config.filters = [32, 64, 128]
  93   config.n_res = [9, 9, 9]
  94 
  95   return config

  98 def get_hparams_cifar_164():
  99   config = get_hparams_cifar_38()
 100   config.filters = [32, 64, 128]
 101   config.n_res = [9, 9, 9]
 102   config.use_bottleneck = True
 103   # Due to bottleneck residual blocks
 104   filters = [f * 4 for f in config.filters]
 105   config.filters = filters
 106 
 107   return config

 110 def get_hparams_imagenet_56():
 111   """RevNet-56 configurations for ImageNet."""
 112 
 113   config = tf.contrib.training.HParams()
 114   config.add_hparam("n_classes", 1000)
 115   config.add_hparam("dataset", "ImageNet")
 116   config.add_hparam("num_train_images", 1281167)
 117   config.add_hparam("num_eval_images", 50000)
 118   config.add_hparam("init_filters", 128)
 119   config.add_hparam("init_kernel", 7)
 120   config.add_hparam("init_stride", 2)
 121   config.add_hparam("n_rev_blocks", 4)
 122   config.add_hparam("n_res", [2, 2, 2, 2])
 123   config.add_hparam("filters", [128, 256, 512, 832])
 124   config.add_hparam("strides", [1, 2, 2, 2])
 125   config.add_hparam("batch_size", 256)
 126   config.add_hparam("bottleneck", True)
 127   config.add_hparam("fused", True)
 128   config.add_hparam("init_max_pool", True)
 129   if tf.test.is_gpu_available():
 130     config.add_hparam("input_shape", (3, 224, 224))
 131     config.add_hparam("data_format", "channels_first")
 132   else:
 133     config.add_hparam("input_shape", (224, 224, 3))
 134     config.add_hparam("data_format", "channels_last")
 135   # Due to bottleneck residual blocks
 136   filters = [f * 4 for f in config.filters]
 137   config.filters = filters
 138 
 139   # Training details
 140   config.add_hparam("weight_decay", 1e-4)
 141   config.add_hparam("momentum", .9)
 142   config.add_hparam("lr_decay_steps", [160000, 320000, 480000])
 143   config.add_hparam("lr_list", [1e-1, 1e-2, 1e-3, 1e-4])
 144   config.add_hparam("max_train_iter", 600000)
 145   config.add_hparam("seed", 1234)
 146   config.add_hparam("shuffle", True)
 147   config.add_hparam("log_every", 500)
 148   config.add_hparam("save_every", 500)
 149   config.add_hparam("dtype", tf.float32)
 150   config.add_hparam("eval_batch_size", 256)
 151   config.add_hparam("div255", True)
 152   config.add_hparam("iters_per_epoch",
 153                     config.num_train_images // config.batch_size)
 154   config.add_hparam("epochs", config.max_train_iter // config.iters_per_epoch)
 155 
 156   # Customized TPU hyperparameters due to differing batch size caused by
 157   # TPU architecture specifics
 158   # Suggested batch sizes to reduce overhead from excessive tensor padding
 159   # https://cloud.google.com/tpu/docs/troubleshooting
 160   config.add_hparam("tpu_batch_size", 1024)
 161   config.add_hparam("tpu_eval_batch_size", 1024)
 162   config.add_hparam("tpu_iters_per_epoch",
 163                     config.num_train_images // config.tpu_batch_size)
 164   config.add_hparam("tpu_epochs",
 165                     config.max_train_iter // config.tpu_iters_per_epoch)
 166   config.add_hparam("tpu_eval_steps",
 167                     config.num_eval_images // config.tpu_eval_batch_size)
 168   return config
