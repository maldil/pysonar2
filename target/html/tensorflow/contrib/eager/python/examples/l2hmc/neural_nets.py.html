  15 """Neural nets utility for L2HMC compatible with TensorFlow's eager execution.
  16 
  17 Reference [Generalizing Hamiltonian Monte Carlo with Neural
  18 Networks](https://arxiv.org/pdf/1711.09268.pdf)
  19 
  20 Code adapted from the released TensorFlow graph implementation by original
  21 authors https://github.com/brain-research/l2hmc.
  22 """
  23 from __future__ import absolute_import
  24 from __future__ import division
  25 from __future__ import print_function
  26 
  27 import tensorflow as tf
  28 
  29 
  30 class GenericNet(tf.keras.Model):
  31   """Generic neural net with different initialization scale based on input.
  32 
  33   Args:
  34     x_dim: dimensionality of observed data
  35     factor: factor of variance scaling initializer
  36     n_hidden: number of hidden units
  37   """
  38 
  39   def __init__(self, x_dim, factor, n_hidden=10):
  40     super(GenericNet, self).__init__()
  41 
  42     self.v_layer = _custom_dense(n_hidden, 1. / 3.)
  43     self.x_layer = _custom_dense(n_hidden, factor / 3.)
  44     self.t_layer = _custom_dense(n_hidden, 1. / 3.)
  45     self.h_layer = _custom_dense(n_hidden)
  46 
  47     # Scale
  48     self.scale_layer = _custom_dense(x_dim, .001)
  49     self.coeff_scale = tf.Variable(
  50         initial_value=tf.zeros([1, x_dim]), name='coeff_scale', trainable=True)
  51     # Translation
  52     self.translation_layer = _custom_dense(x_dim, factor=.001)
  53     # Transformation
  54     self.transformation_layer = _custom_dense(x_dim, .001)
  55     self.coeff_transformation = tf.Variable(
  56         initial_value=tf.zeros([1, x_dim]),
  57         name='coeff_transformation',
  58         trainable=True)
  59 
  60   def call(self, inputs):
  61     v, x, t = inputs
  62     h = self.v_layer(v) + self.x_layer(x) + self.t_layer(t)
  63     h = tf.nn.relu(h)
  64     h = self.h_layer(h)
  65     h = tf.nn.relu(h)
  66     scale = tf.nn.tanh(self.scale_layer(h)) * tf.exp(self.coeff_scale)
  67     translation = self.translation_layer(h)
  68     transformation = (
  69         tf.nn.tanh(self.transformation_layer(h)) * tf.exp(
  70             self.coeff_transformation))
  71 
  72     return scale, translation, transformation
  73 
  74 
  75 def _custom_dense(units, factor=1.):
  76   """Custom dense layer with specified weight initialization."""
  77 
  78   return tf.keras.layers.Dense(
  79       units=units,
  80       use_bias=True,
  81       kernel_initializer=tf.contrib.layers.variance_scaling_initializer(
  82           factor=factor * 2., mode='FAN_IN', uniform=False),
  83       bias_initializer=tf.constant_initializer(0., dtype=tf.float32))
