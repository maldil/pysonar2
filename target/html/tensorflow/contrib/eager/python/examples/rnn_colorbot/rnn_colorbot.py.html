  15 r"""TensorFlow Eager Execution Example: RNN Colorbot.
  16 
  17 This example builds, trains, and evaluates a multi-layer RNN that can be
  18 run with eager execution enabled. The RNN is trained to map color names to
  19 their RGB values: it takes as input a one-hot encoded character sequence and
  20 outputs a three-tuple (R, G, B) (scaled by 1/255).
  21 
  22 For example, say we'd like the RNN Colorbot to generate the RGB values for the
  23 color white. To represent our query in a form that the Colorbot could
  24 understand, we would create a sequence of five 256-long vectors encoding the
  25 ASCII values of the characters in "white". The first vector in our sequence
  26 would be 0 everywhere except for the ord("w")-th position, where it would be
  27 1, the second vector would be 0 everywhere except for the
  28 ord("h")-th position, where it would be 1, and similarly for the remaining three
  29 vectors. We refer to such indicator vectors as "one-hot encodings" of
  30 characters. After consuming these vectors, a well-trained Colorbot would output
  31 the three tuple (1, 1, 1), since the RGB values for white are (255, 255, 255).
  32 We are of course free to ask the colorbot to generate colors for any string we'd
  33 like, such as "steel gray," "tensorflow orange," or "green apple," though
  34 your mileage may vary as your queries increase in creativity.
  35 
  36 This example shows how to:
  37   1. read, process, (one-hot) encode, and pad text data via the
  38      Datasets API;
  39   2. build a trainable model;
  40   3. implement a multi-layer RNN using Python control flow
  41      constructs (e.g., a for loop);
  42   4. train a model using an iterative gradient-based method; and
  43 
  44 The data used in this example is licensed under the Creative Commons
  45 Attribution-ShareAlike License and is available at
  46   https://en.wikipedia.org/wiki/List_of_colors:_A-F
  47   https://en.wikipedia.org/wiki/List_of_colors:_G-M
  48   https://en.wikipedia.org/wiki/List_of_colors:_N-Z
  49 
  50 This example was adapted from
  51   https://github.com/random-forests/tensorflow-workshop/tree/master/extras/colorbot
  52 """

  79 def parse(line):
  80   """Parse a line from the colors dataset."""
  81 
  82   # Each line of the dataset is comma-separated and formatted as
  83   #    color_name, r, g, b
  84   # so `items` is a list [color_name, r, g, b].
  85   items = tf.string_split([line], ",").values
  86   rgb = tf.string_to_number(items[1:], out_type=tf.float32) / 255.
  87   # Represent the color name as a one-hot encoded character sequence.
  88   color_name = items[0]
  89   chars = tf.one_hot(tf.decode_raw(color_name, tf.uint8), depth=256)
  90   # The sequence length is needed by our RNN.
  91   length = tf.cast(tf.shape(chars)[0], dtype=tf.int64)
  92   return rgb, chars, length

  95 def maybe_download(filename, work_directory, source_url):
  96   """Download the data from source url, unless it's already here.
  97 
  98   Args:
  99     filename: string, name of the file in the directory.
 100     work_directory: string, path to working directory.
 101     source_url: url to download from if file doesn't exist.
 102 
 103   Returns:
 104     Path to resulting file.
 105   """
 106   if not tf.gfile.Exists(work_directory):
 107     tf.gfile.MakeDirs(work_directory)
 108   filepath = os.path.join(work_directory, filename)
 109   if not tf.gfile.Exists(filepath):
 110     temp_file_name, _ = urllib.request.urlretrieve(source_url)
 111     tf.gfile.Copy(temp_file_name, filepath)
 112     with tf.gfile.GFile(filepath) as f:
 113       size = f.size()
 114       print("Successfully downloaded", filename, size, "bytes.")
 115   return filepath

 118 def load_dataset(data_dir, url, batch_size):
 119   """Loads the colors data at path into a PaddedDataset."""
 120 
 121   # Downloads data at url into data_dir/basename(url). The dataset has a header
 122   # row (color_name, r, g, b) followed by comma-separated lines.
 123   path = maybe_download(os.path.basename(url), data_dir, url)
 124 
 125   # This chain of commands loads our data by:
 126   #   1. skipping the header; (.skip(1))
 127   #   2. parsing the subsequent lines; (.map(parse))
 128   #   3. shuffling the data; (.shuffle(...))
 129   #   3. grouping the data into padded batches (.padded_batch(...)).
 130   dataset = tf.data.TextLineDataset(path).skip(1).map(parse).shuffle(
 131       buffer_size=10000).padded_batch(
 132           batch_size, padded_shapes=([None], [None, None], []))
 133   return dataset

 136 # pylint: disable=not-callable
 137 class RNNColorbot(tf.keras.Model):
 138   """Multi-layer (LSTM) RNN that regresses on real-valued vector labels.
 139   """
 140 
 141   def __init__(self, rnn_cell_sizes, label_dimension, keep_prob):
 142     """Constructs an RNNColorbot.
 143 
 144     Args:
 145       rnn_cell_sizes: list of integers denoting the size of each LSTM cell in
 146         the RNN; rnn_cell_sizes[i] is the size of the i-th layer cell
 147       label_dimension: the length of the labels on which to regress
 148       keep_prob: (1 - dropout probability); dropout is applied to the outputs of
 149         each LSTM layer
 150     """
 151     super(RNNColorbot, self).__init__(name="")
 152     self.label_dimension = label_dimension
 153     self.keep_prob = keep_prob
 154 
 155     self.cells = tf.contrib.checkpoint.List(
 156         [tf.nn.rnn_cell.BasicLSTMCell(size) for size in rnn_cell_sizes])
 157     self.relu = layers.Dense(
 158         label_dimension, activation=tf.nn.relu, name="relu")
 159 
 160   def call(self, inputs, training=False):
 161     """Implements the RNN logic and prediction generation.
 162 
 163     Args:
 164       inputs: A tuple (chars, sequence_length), where chars is a batch of
 165         one-hot encoded color names represented as a Tensor with dimensions
 166         [batch_size, time_steps, 256] and sequence_length holds the length
 167         of each character sequence (color name) as a Tensor with dimension
 168         [batch_size].
 169       training: whether the invocation is happening during training
 170 
 171     Returns:
 172       A tensor of dimension [batch_size, label_dimension] that is produced by
 173       passing chars through a multi-layer RNN and applying a ReLU to the final
 174       hidden state.
 175     """
 176     (chars, sequence_length) = inputs
 177     # Transpose the first and second dimensions so that chars is of shape
 178     # [time_steps, batch_size, dimension].
 179     chars = tf.transpose(chars, [1, 0, 2])
 180     # The outer loop cycles through the layers of the RNN; the inner loop
 181     # executes the time steps for a particular layer.
 182     batch_size = int(chars.shape[1])
 183     for l in range(len(self.cells)):
 184       cell = self.cells[l]
 185       outputs = []
 186       state = cell.zero_state(batch_size, tf.float32)
 187       # Unstack the inputs to obtain a list of batches, one for each time step.
 188       chars = tf.unstack(chars, axis=0)
 189       for ch in chars:
 190         output, state = cell(ch, state)
 191         outputs.append(output)
 192       # The outputs of this layer are the inputs of the subsequent layer.
 193       chars = tf.stack(outputs, axis=0)
 194       if training:
 195         chars = tf.nn.dropout(chars, self.keep_prob)
 196     # Extract the correct output (i.e., hidden state) for each example. All the
 197     # character sequences in this batch were padded to the same fixed length so
 198     # that they could be easily fed through the above RNN loop. The
 199     # `sequence_length` vector tells us the true lengths of the character
 200     # sequences, letting us obtain for each sequence the hidden state that was
 201     # generated by its non-padding characters.
 202     batch_range = [i for i in range(batch_size)]
 203     indices = tf.stack([sequence_length - 1, batch_range], axis=1)
 204     hidden_states = tf.gather_nd(chars, indices)
 205     return self.relu(hidden_states)

 208 def loss(labels, predictions):
 209   """Computes mean squared loss."""
 210   return tf.reduce_mean(tf.square(predictions - labels))
 211

 213 def test(model, eval_data):
 214   """Computes the average loss on eval_data, which should be a Dataset."""
 215   avg_loss = tfe.metrics.Mean("loss")
 216   for (labels, chars, sequence_length) in tfe.Iterator(eval_data):
 217     predictions = model((chars, sequence_length), training=False)
 218     avg_loss(loss(labels, predictions))
 219   print("eval/loss: %.6f\n" % avg_loss.result())
 220   with tf.contrib.summary.always_record_summaries():
 221     tf.contrib.summary.scalar("loss", avg_loss.result())
 222 
 223 
 224 def train_one_epoch(model, optimizer, train_data, log_interval=10):
 225   """Trains model on train_data using optimizer."""
 226 
 227   tf.train.get_or_create_global_step()
 228 
 229   def model_loss(labels, chars, sequence_length):
 230     predictions = model((chars, sequence_length), training=True)
 231     loss_value = loss(labels, predictions)
 232     tf.contrib.summary.scalar("loss", loss_value)
 233     return loss_value
 234 
 235   for (batch, (labels, chars, sequence_length)) in enumerate(
 236       tfe.Iterator(train_data)):
 237     with tf.contrib.summary.record_summaries_every_n_global_steps(log_interval):
 238       batch_model_loss = functools.partial(model_loss, labels, chars,
 239                                            sequence_length)
 240       optimizer.minimize(
 241           batch_model_loss, global_step=tf.train.get_global_step())
 242       if log_interval and batch % log_interval == 0:
 243         print("train/batch #%d\tloss: %.6f" % (batch, batch_model_loss()))

 250 def main(_):
 251   data_dir = os.path.join(FLAGS.dir, "data")
 252   train_data = load_dataset(
 253       data_dir=data_dir, url=SOURCE_TRAIN_URL, batch_size=FLAGS.batch_size)
 254   eval_data = load_dataset(
 255       data_dir=data_dir, url=SOURCE_TEST_URL, batch_size=FLAGS.batch_size)
 256 
 257   model = RNNColorbot(
 258       rnn_cell_sizes=FLAGS.rnn_cell_sizes,
 259       label_dimension=3,
 260       keep_prob=FLAGS.keep_probability)
 261   optimizer = tf.train.AdamOptimizer(learning_rate=FLAGS.learning_rate)
 262 
 263   if FLAGS.no_gpu or tfe.num_gpus() <= 0:
 264     print(tfe.num_gpus())
 265     device = "/cpu:0"
 266   else:
 267     device = "/gpu:0"
 268   print("Using device %s." % device)
 269 
 270   log_dir = os.path.join(FLAGS.dir, "summaries")
 271   tf.gfile.MakeDirs(log_dir)
 272   train_summary_writer = tf.contrib.summary.create_file_writer(
 273       os.path.join(log_dir, "train"), flush_millis=10000)
 274   test_summary_writer = tf.contrib.summary.create_file_writer(
 275       os.path.join(log_dir, "eval"), flush_millis=10000, name="eval")
 276 
 277   with tf.device(device):
 278     for epoch in range(FLAGS.num_epochs):
 279       start = time.time()
 280       with train_summary_writer.as_default():
 281         train_one_epoch(model, optimizer, train_data, FLAGS.log_interval)
 282       end = time.time()
 283       print("train/time for epoch #%d: %.2f" % (epoch, end - start))
 284       with test_summary_writer.as_default():
 285         test(model, eval_data)
 286 
 287   print("Colorbot is ready to generate colors!")
 288   while True:
 289     try:
 290       color_name = six.moves.input(
 291           "Give me a color name (or press enter to exit): ")
 292     except EOFError:
 293       return
 294 
 295     if not color_name:
 296       return
 297 
 298     _, chars, length = parse(color_name)
 299     with tf.device(device):
 300       (chars, length) = (tf.identity(chars), tf.identity(length))
 301       chars = tf.expand_dims(chars, 0)
 302       length = tf.expand_dims(length, 0)
 303       preds = tf.unstack(model((chars, length), training=False)[0])
 304 
 305     # Predictions cannot be negative, as they are generated by a ReLU layer;
 306     # they may, however, be greater than 1.
 307     clipped_preds = tuple(min(float(p), 1.0) for p in preds)
 308     rgb = tuple(int(p * 255) for p in clipped_preds)
 309     print("rgb:", rgb)
 310     data = [[clipped_preds]]
 311     if HAS_MATPLOTLIB:
 312       plt.imshow(data)
 313       plt.title(color_name)
 314       plt.show()
