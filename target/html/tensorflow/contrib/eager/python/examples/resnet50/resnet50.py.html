  15 """ResNet50 model definition compatible with TensorFlow's eager execution.
  16 
  17 Reference [Deep Residual Learning for Image
  18 Recognition](https://arxiv.org/abs/1512.03385)
  19 
  20 Adapted from tf.keras.applications.ResNet50. A notable difference is that the
  21 model here outputs logits while the Keras model outputs probability.
  22 """

  24 from __future__ import division
  25 from __future__ import print_function

  27 import functools

  29 import tensorflow as tf

  31 layers = tf.keras.layers

  34 class _IdentityBlock(tf.keras.Model):
  35   """_IdentityBlock is the block that has no conv layer at shortcut.
  36 
  37   Args:
  38     kernel_size: the kernel size of middle conv layer at main path
  39     filters: list of integers, the filters of 3 conv layer at main path
  40     stage: integer, current stage label, used for generating layer names
  41     block: 'a','b'..., current block label, used for generating layer names
  42     data_format: data_format for the input ('channels_first' or
  43       'channels_last').
  44   """
  45 
  46   def __init__(self, kernel_size, filters, stage, block, data_format):
  47     super(_IdentityBlock, self).__init__(name='')
  48     filters1, filters2, filters3 = filters
  49 
  50     conv_name_base = 'res' + str(stage) + block + '_branch'
  51     bn_name_base = 'bn' + str(stage) + block + '_branch'
  52     bn_axis = 1 if data_format == 'channels_first' else 3
  53 
  54     self.conv2a = layers.Conv2D(
  55         filters1, (1, 1), name=conv_name_base + '2a', data_format=data_format)
  56     self.bn2a = layers.BatchNormalization(
  57         axis=bn_axis, name=bn_name_base + '2a')
  58 
  59     self.conv2b = layers.Conv2D(
  60         filters2,
  61         kernel_size,
  62         padding='same',
  63         data_format=data_format,
  64         name=conv_name_base + '2b')
  65     self.bn2b = layers.BatchNormalization(
  66         axis=bn_axis, name=bn_name_base + '2b')
  67 
  68     self.conv2c = layers.Conv2D(
  69         filters3, (1, 1), name=conv_name_base + '2c', data_format=data_format)
  70     self.bn2c = layers.BatchNormalization(
  71         axis=bn_axis, name=bn_name_base + '2c')
  72 
  73   def call(self, input_tensor, training=False):
  74     x = self.conv2a(input_tensor)
  75     x = self.bn2a(x, training=training)
  76     x = tf.nn.relu(x)
  77 
  78     x = self.conv2b(x)
  79     x = self.bn2b(x, training=training)
  80     x = tf.nn.relu(x)
  81 
  82     x = self.conv2c(x)
  83     x = self.bn2c(x, training=training)
  84 
  85     x += input_tensor
  86     return tf.nn.relu(x)

  89 class _ConvBlock(tf.keras.Model):
  90   """_ConvBlock is the block that has a conv layer at shortcut.
  91 
  92   Args:
  93       kernel_size: the kernel size of middle conv layer at main path
  94       filters: list of integers, the filters of 3 conv layer at main path
  95       stage: integer, current stage label, used for generating layer names
  96       block: 'a','b'..., current block label, used for generating layer names
  97       data_format: data_format for the input ('channels_first' or
  98         'channels_last').
  99       strides: strides for the convolution. Note that from stage 3, the first
 100        conv layer at main path is with strides=(2,2), and the shortcut should
 101        have strides=(2,2) as well.
 102   """
 103 
 104   def __init__(self,
 105                kernel_size,
 106                filters,
 107                stage,
 108                block,
 109                data_format,
 110                strides=(2, 2)):
 111     super(_ConvBlock, self).__init__(name='')
 112     filters1, filters2, filters3 = filters
 113 
 114     conv_name_base = 'res' + str(stage) + block + '_branch'
 115     bn_name_base = 'bn' + str(stage) + block + '_branch'
 116     bn_axis = 1 if data_format == 'channels_first' else 3
 117 
 118     self.conv2a = layers.Conv2D(
 119         filters1, (1, 1),
 120         strides=strides,
 121         name=conv_name_base + '2a',
 122         data_format=data_format)
 123     self.bn2a = layers.BatchNormalization(
 124         axis=bn_axis, name=bn_name_base + '2a')
 125 
 126     self.conv2b = layers.Conv2D(
 127         filters2,
 128         kernel_size,
 129         padding='same',
 130         name=conv_name_base + '2b',
 131         data_format=data_format)
 132     self.bn2b = layers.BatchNormalization(
 133         axis=bn_axis, name=bn_name_base + '2b')
 134 
 135     self.conv2c = layers.Conv2D(
 136         filters3, (1, 1), name=conv_name_base + '2c', data_format=data_format)
 137     self.bn2c = layers.BatchNormalization(
 138         axis=bn_axis, name=bn_name_base + '2c')
 139 
 140     self.conv_shortcut = layers.Conv2D(
 141         filters3, (1, 1),
 142         strides=strides,
 143         name=conv_name_base + '1',
 144         data_format=data_format)
 145     self.bn_shortcut = layers.BatchNormalization(
 146         axis=bn_axis, name=bn_name_base + '1')
 147 
 148   def call(self, input_tensor, training=False):
 149     x = self.conv2a(input_tensor)
 150     x = self.bn2a(x, training=training)
 151     x = tf.nn.relu(x)
 152 
 153     x = self.conv2b(x)
 154     x = self.bn2b(x, training=training)
 155     x = tf.nn.relu(x)
 156 
 157     x = self.conv2c(x)
 158     x = self.bn2c(x, training=training)
 159 
 160     shortcut = self.conv_shortcut(input_tensor)
 161     shortcut = self.bn_shortcut(shortcut, training=training)
 162 
 163     x += shortcut
 164     return tf.nn.relu(x)

 168 class ResNet50(tf.keras.Model):
 169   """Instantiates the ResNet50 architecture.
 170 
 171   Args:
 172     data_format: format for the image. Either 'channels_first' or
 173       'channels_last'.  'channels_first' is typically faster on GPUs while
 174       'channels_last' is typically faster on CPUs. See
 175       https://www.tensorflow.org/performance/performance_guide#data_formats
 176     name: Prefix applied to names of variables created in the model.
 177     trainable: Is the model trainable? If true, performs backward
 178         and optimization after call() method.
 179     include_top: whether to include the fully-connected layer at the top of the
 180       network.
 181     pooling: Optional pooling mode for feature extraction when `include_top`
 182       is `False`.
 183       - `None` means that the output of the model will be the 4D tensor
 184           output of the last convolutional layer.
 185       - `avg` means that global average pooling will be applied to the output of
 186           the last convolutional layer, and thus the output of the model will be
 187           a 2D tensor.
 188       - `max` means that global max pooling will be applied.
 189     classes: optional number of classes to classify images into, only to be
 190       specified if `include_top` is True.
 191 
 192   Raises:
 193       ValueError: in case of invalid argument for data_format.
 194   """
 195 
 196   def __init__(self,
 197                data_format,
 198                name='',
 199                trainable=True,
 200                include_top=True,
 201                pooling=None,
 202                classes=1000):
 203     super(ResNet50, self).__init__(name=name)
 204 
 205     valid_channel_values = ('channels_first', 'channels_last')
 206     if data_format not in valid_channel_values:
 207       raise ValueError('Unknown data_format: %s. Valid values: %s' %
 208                        (data_format, valid_channel_values))
 209     self.include_top = include_top
 210 
 211     def conv_block(filters, stage, block, strides=(2, 2)):
 212       return _ConvBlock(
 213           3,
 214           filters,
 215           stage=stage,
 216           block=block,
 217           data_format=data_format,
 218           strides=strides)
 219 
 220     def id_block(filters, stage, block):
 221       return _IdentityBlock(
 222           3, filters, stage=stage, block=block, data_format=data_format)
 223 
 224     self.conv1 = layers.Conv2D(
 225         64, (7, 7),
 226         strides=(2, 2),
 227         data_format=data_format,
 228         padding='same',
 229         name='conv1')
 230     bn_axis = 1 if data_format == 'channels_first' else 3
 231     self.bn_conv1 = layers.BatchNormalization(axis=bn_axis, name='bn_conv1')
 232     self.max_pool = layers.MaxPooling2D(
 233         (3, 3), strides=(2, 2), data_format=data_format)
 234 
 235     self.l2a = conv_block([64, 64, 256], stage=2, block='a', strides=(1, 1))
 236     self.l2b = id_block([64, 64, 256], stage=2, block='b')
 237     self.l2c = id_block([64, 64, 256], stage=2, block='c')
 238 
 239     self.l3a = conv_block([128, 128, 512], stage=3, block='a')
 240     self.l3b = id_block([128, 128, 512], stage=3, block='b')
 241     self.l3c = id_block([128, 128, 512], stage=3, block='c')
 242     self.l3d = id_block([128, 128, 512], stage=3, block='d')
 243 
 244     self.l4a = conv_block([256, 256, 1024], stage=4, block='a')
 245     self.l4b = id_block([256, 256, 1024], stage=4, block='b')
 246     self.l4c = id_block([256, 256, 1024], stage=4, block='c')
 247     self.l4d = id_block([256, 256, 1024], stage=4, block='d')
 248     self.l4e = id_block([256, 256, 1024], stage=4, block='e')
 249     self.l4f = id_block([256, 256, 1024], stage=4, block='f')
 250 
 251     self.l5a = conv_block([512, 512, 2048], stage=5, block='a')
 252     self.l5b = id_block([512, 512, 2048], stage=5, block='b')
 253     self.l5c = id_block([512, 512, 2048], stage=5, block='c')
 254 
 255     self.avg_pool = layers.AveragePooling2D(
 256         (7, 7), strides=(7, 7), data_format=data_format)
 257 
 258     if self.include_top:
 259       self.flatten = layers.Flatten()
 260       self.fc1000 = layers.Dense(classes, name='fc1000')
 261     else:
 262       reduction_indices = [1, 2] if data_format == 'channels_last' else [2, 3]
 263       reduction_indices = tf.constant(reduction_indices)
 264       if pooling == 'avg':
 265         self.global_pooling = functools.partial(
 266             tf.reduce_mean,
 267             reduction_indices=reduction_indices,
 268             keep_dims=False)
 269       elif pooling == 'max':
 270         self.global_pooling = functools.partial(
 271             tf.reduce_max, reduction_indices=reduction_indices, keep_dims=False)
 272       else:
 273         self.global_pooling = None
 274 
 275   def call(self, inputs, training=True):
 276     x = self.conv1(inputs)
 277     x = self.bn_conv1(x, training=training)
 278     x = tf.nn.relu(x)
 279     x = self.max_pool(x)
 280 
 281     x = self.l2a(x, training=training)
 282     x = self.l2b(x, training=training)
 283     x = self.l2c(x, training=training)
 284 
 285     x = self.l3a(x, training=training)
 286     x = self.l3b(x, training=training)
 287     x = self.l3c(x, training=training)
 288     x = self.l3d(x, training=training)
 289 
 290     x = self.l4a(x, training=training)
 291     x = self.l4b(x, training=training)
 292     x = self.l4c(x, training=training)
 293     x = self.l4d(x, training=training)
 294     x = self.l4e(x, training=training)
 295     x = self.l4f(x, training=training)
 296 
 297     x = self.l5a(x, training=training)
 298     x = self.l5b(x, training=training)
 299     x = self.l5c(x, training=training)
 300 
 301     x = self.avg_pool(x)
 302 
 303     if self.include_top:
 304       return self.fc1000(self.flatten(x))
 305     elif self.global_pooling:
 306       return self.global_pooling(x)
 307     else:
 308       return x
