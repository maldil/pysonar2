  37 def get_meta_graph_def(saved_model_dir, tags):
  38   """Gets `MetaGraphDef` from a directory containing a `SavedModel`.
  39 
  40   Returns the `MetaGraphDef` for the given tag-set and SavedModel directory.
  41 
  42   Args:
  43     saved_model_dir: Directory containing the SavedModel.
  44     tags: Comma separated list of tags used to identify the correct
  45       `MetaGraphDef`.
  46 
  47   Raises:
  48     ValueError: An error when the given tags cannot be found.
  49 
  50   Returns:
  51     A `MetaGraphDef` corresponding to the given tags.
  52   """
  53   saved_model = reader.read_saved_model(saved_model_dir)
  54   set_of_tags = set([tag.strip() for tag in tags.split(',')])
  55   for meta_graph_def in saved_model.meta_graphs:
  56     if set(meta_graph_def.meta_info_def.tags) == set_of_tags:
  57       return meta_graph_def
  58   raise ValueError('Could not find MetaGraphDef with tags {}'.format(tags))

  61 def _get_signature_def(signature_def_key, export_dir, tags):
  62   """Construct a `SignatureDef` proto."""
  63   signature_def_key = (
  64       signature_def_key or
  65       signature_constants.DEFAULT_SERVING_SIGNATURE_DEF_KEY)
  66 
  67   metagraph_def = get_meta_graph_def(export_dir, tags)
  68 
  69   try:
  70     signature_def = metagraph_def.signature_def[signature_def_key]
  71   except KeyError as e:
  72     formatted_key = _DEFAULT_INPUT_ALTERNATIVE_FORMAT.format(
  73         signature_def_key)
  74     try:
  75       signature_def = metagraph_def.signature_def[formatted_key]
  76     except KeyError:
  77       raise ValueError(
  78           'Got signature_def_key "{}". Available signatures are {}. '
  79           'Original error:\n{}'.format(
  80               signature_def_key, list(metagraph_def.signature_def), e))
  81     logging.warning('Could not find signature def "%s". '
  82                     'Using "%s" instead', signature_def_key, formatted_key)
  83   return signature_def

  86 def _check_signature_arguments(signature_def_key,
  87                                signature_def,
  88                                input_names,
  89                                output_names):
  90   """Validates signature arguments for `SavedModelPredictor`."""
  91   signature_def_key_specified = signature_def_key is not None
  92   signature_def_specified = signature_def is not None
  93   input_names_specified = input_names is not None
  94   output_names_specified = output_names is not None
  95   if input_names_specified != output_names_specified:
  96     raise ValueError(
  97         'input_names and output_names must both be specified or both be '
  98         'unspecified.'
  99     )
 100 
 101   if (signature_def_key_specified + signature_def_specified +
 102       input_names_specified > 1):
 103     raise ValueError(
 104         'You must specify at most one of signature_def_key OR signature_def OR'
 105         '(input_names AND output_names).'
 106     )

 109 class SavedModelPredictor(predictor.Predictor):
 110   """A `Predictor` constructed from a `SavedModel`."""
 111 
 112   def __init__(self,
 113                export_dir,
 114                signature_def_key=None,
 115                signature_def=None,
 116                input_names=None,
 117                output_names=None,
 118                tags=None,
 119                graph=None,
 120                config=None):
 121     """Initialize a `CoreEstimatorPredictor`.
 122 
 123     Args:
 124       export_dir: a path to a directory containing a `SavedModel`.
 125       signature_def_key: Optional string specifying the signature to use. If
 126         `None`, then `DEFAULT_SERVING_SIGNATURE_DEF_KEY` is used. Only one of
 127         `signature_def_key` and `signature_def` should be specified.
 128       signature_def: A `SignatureDef` proto specifying the inputs and outputs
 129         for prediction. Only one of `signature_def_key` and `signature_def`
 130         should be specified.
 131       input_names: A dictionary mapping strings to `Tensor`s in the `SavedModel`
 132         that represent the input. The keys can be any string of the user's
 133         choosing.
 134       output_names: A dictionary mapping strings to `Tensor`s in the
 135         `SavedModel` that represent the output. The keys can be any string of
 136         the user's choosing.
 137       tags: Optional. Comma separated list of tags that will be used to retrieve
 138         the correct `SignatureDef`. Defaults to `DEFAULT_TAGS`.
 139       graph: Optional. The Tensorflow `graph` in which prediction should be
 140         done.
 141       config: `ConfigProto` proto used to configure the session.
 142     Raises:
 143       ValueError: If more than one of signature_def_key OR signature_def OR
 144         (input_names AND output_names) is specified.
 145     """
 146     _check_signature_arguments(
 147         signature_def_key, signature_def, input_names, output_names)
 148     tags = tags or DEFAULT_TAGS
 149     self._graph = graph or ops.Graph()
 150 
 151     with self._graph.as_default():
 152       self._session = session.Session(config=config)
 153       loader.load(self._session, tags.split(','), export_dir)
 154 
 155     if input_names is None:
 156       if signature_def is None:
 157         signature_def = _get_signature_def(signature_def_key, export_dir, tags)
 158       input_names = {k: v.name for k, v in signature_def.inputs.items()}
 159       output_names = {k: v.name for k, v in signature_def.outputs.items()}
 160 
 161     self._feed_tensors = {k: self._graph.get_tensor_by_name(v)
 162                           for k, v in input_names.items()}
 163     self._fetch_tensors = {k: self._graph.get_tensor_by_name(v)
 164                            for k, v in output_names.items()}
