  29 def from_contrib_estimator(estimator,
  30                            prediction_input_fn,
  31                            input_alternative_key=None,
  32                            output_alternative_key=None,
  33                            graph=None,
  34                            config=None):
  35   """Constructs a `Predictor` from a `tf.contrib.learn.Estimator`.
  36 
  37   Args:
  38     estimator: an instance of `tf.contrib.learn.Estimator`.
  39     prediction_input_fn: a function that takes no arguments and returns an
  40       instance of `InputFnOps`.
  41     input_alternative_key: Optional. Specify the input alternative used for
  42       prediction.
  43     output_alternative_key: Specify the output alternative used for
  44       prediction. Not needed for single-headed models but required for
  45       multi-headed models.
  46     graph: Optional. The Tensorflow `graph` in which prediction should be
  47       done.
  48     config: `ConfigProto` proto used to configure the session.
  49 
  50   Returns:
  51     An initialized `Predictor`.
  52 
  53   Raises:
  54     TypeError: if `estimator` is a core `Estimator` instead of a contrib
  55       `Estimator`.
  56   """
  57   if isinstance(estimator, core_estimator.Estimator):
  58     raise TypeError('Expected estimator to be of type '
  59                     'tf.contrib.learn.Estimator, but got type '
  60                     'tf.python.estimator.Estimator. You likely want to call '
  61                     'from_estimator.')
  62   return contrib_estimator_predictor.ContribEstimatorPredictor(
  63       estimator,
  64       prediction_input_fn,
  65       input_alternative_key=input_alternative_key,
  66       output_alternative_key=output_alternative_key,
  67       graph=graph,
  68       config=config)

  71 def from_estimator(estimator,
  72                    serving_input_receiver_fn,
  73                    output_key=None,
  74                    graph=None,
  75                    config=None):
  76   """Constructs a `Predictor` from a `tf.python.estimator.Estimator`.
  77 
  78   Args:
  79     estimator: an instance of `learn.python.estimator.Estimator`.
  80     serving_input_receiver_fn: a function that takes no arguments and returns
  81       an instance of `ServingInputReceiver` compatible with `estimator`.
  82     output_key: Optional string specifying the export output to use. If
  83       `None`, then `DEFAULT_SERVING_SIGNATURE_DEF_KEY` is used.
  84     graph: Optional. The Tensorflow `graph` in which prediction should be
  85       done.
  86     config: `ConfigProto` proto used to configure the session.
  87 
  88   Returns:
  89     An initialized `Predictor`.
  90 
  91   Raises:
  92     TypeError: if `estimator` is a contrib `Estimator` instead of a core
  93       `Estimator`.
  94   """
  95   if isinstance(estimator, contrib_estimator.Estimator):
  96     raise TypeError('Expected estimator to be of type '
  97                     'tf.python.estimator.Estimator, but got type '
  98                     'tf.contrib.learn.Estimator. You likely want to call '
  99                     'from_contrib_estimator.')
 100   return core_estimator_predictor.CoreEstimatorPredictor(
 101       estimator,
 102       serving_input_receiver_fn,
 103       output_key=output_key,
 104       graph=graph,
 105       config=config)

 108 def from_saved_model(export_dir,
 109                      signature_def_key=None,
 110                      signature_def=None,
 111                      input_names=None,
 112                      output_names=None,
 113                      tags=None,
 114                      graph=None,
 115                      config=None):
 116   """Constructs a `Predictor` from a `SavedModel` on disk.
 117 
 118   Args:
 119     export_dir: a path to a directory containing a `SavedModel`.
 120     signature_def_key: Optional string specifying the signature to use. If
 121       `None`, then `DEFAULT_SERVING_SIGNATURE_DEF_KEY` is used. Only one of
 122     `signature_def_key` and `signature_def`
 123     signature_def: A `SignatureDef` proto specifying the inputs and outputs
 124       for prediction. Only one of `signature_def_key` and `signature_def`
 125       should be specified.
 126       input_names: A dictionary mapping strings to `Tensor`s in the `SavedModel`
 127         that represent the input. The keys can be any string of the user's
 128         choosing.
 129       output_names: A dictionary mapping strings to `Tensor`s in the
 130         `SavedModel` that represent the output. The keys can be any string of
 131         the user's choosing.
 132     tags: Optional. Tags that will be used to retrieve the correct
 133       `SignatureDef`. Defaults to `DEFAULT_TAGS`.
 134     graph: Optional. The Tensorflow `graph` in which prediction should be
 135       done.
 136     config: `ConfigProto` proto used to configure the session.
 137 
 138   Returns:
 139     An initialized `Predictor`.
 140 
 141   Raises:
 142     ValueError: More than one of `signature_def_key` and `signature_def` is
 143       specified.
 144   """
 145   return saved_model_predictor.SavedModelPredictor(
 146       export_dir,
 147       signature_def_key=signature_def_key,
 148       signature_def=signature_def,
 149       input_names=input_names,
 150       output_names=output_names,
 151       tags=tags,
 152       graph=graph,
 153       config=config)
