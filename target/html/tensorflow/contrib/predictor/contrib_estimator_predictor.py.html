  29 class ContribEstimatorPredictor(predictor.Predictor):
  30   """A `Predictor constructed from a `tf.contrib.learn.Estimator`."""
  31 
  32   def __init__(self,
  33                estimator,
  34                prediction_input_fn,
  35                input_alternative_key=None,
  36                output_alternative_key=None,
  37                graph=None,
  38                config=None):
  39     """Initialize a `ContribEstimatorPredictor`.
  40 
  41     Args:
  42       estimator: an instance of `tf.contrib.learn.Estimator`.
  43       prediction_input_fn: a function that takes no arguments and returns an
  44         instance of `InputFnOps`.
  45       input_alternative_key: Optional. Specify the input alternative used for
  46         prediction.
  47       output_alternative_key: Specify the output alternative used for
  48         prediction. Not needed for single-headed models but required for
  49         multi-headed models.
  50       graph: Optional. The Tensorflow `graph` in which prediction should be
  51         done.
  52       config: `ConfigProto` proto used to configure the session.
  53     """
  54     self._graph = graph or ops.Graph()
  55     with self._graph.as_default():
  56       input_fn_ops = prediction_input_fn()
  57       # pylint: disable=protected-access
  58       model_fn_ops = estimator._get_predict_ops(input_fn_ops.features)
  59       # pylint: enable=protected-access
  60       checkpoint_path = checkpoint_management.latest_checkpoint(
  61           estimator.model_dir)
  62       self._session = monitored_session.MonitoredSession(
  63           session_creator=monitored_session.ChiefSessionCreator(
  64               config=config,
  65               checkpoint_filename_with_path=checkpoint_path))
  66 
  67     input_alternative_key = (
  68         input_alternative_key or
  69         saved_model_export_utils.DEFAULT_INPUT_ALTERNATIVE_KEY)
  70     input_alternatives, _ = saved_model_export_utils.get_input_alternatives(
  71         input_fn_ops)
  72     self._feed_tensors = input_alternatives[input_alternative_key]
  73 
  74     (output_alternatives,
  75      output_alternative_key) = saved_model_export_utils.get_output_alternatives(
  76          model_fn_ops, output_alternative_key)
  77     _, fetch_tensors = output_alternatives[output_alternative_key]
  78     self._fetch_tensors = fetch_tensors
