  15 """Abstract base class for all predictors."""
  16 
  17 from __future__ import absolute_import
  18 from __future__ import division
  19 from __future__ import print_function
  20 
  21 import abc
  22 import six
  23 
  24 
  25 @six.add_metaclass(abc.ABCMeta)
  26 class Predictor(object):
  27   """Abstract base class for all predictors."""
  28 
  29   @property
  30   def graph(self):
  31     return self._graph
  32 
  33   @property
  34   def session(self):
  35     return self._session
  36 
  37   @property
  38   def feed_tensors(self):
  39     return self._feed_tensors
  40 
  41   @property
  42   def fetch_tensors(self):
  43     return self._fetch_tensors
  44 
  45   def __repr__(self):
  46     return '{} with feed tensors {} and fetch_tensors {}'.format(
  47         type(self).__name__, self._feed_tensors, self._fetch_tensors)
  48 
  49   def __call__(self, input_dict):
  50     """Returns predictions based on `input_dict`.
  51 
  52     Args:
  53       input_dict: a `dict` mapping strings to numpy arrays. These keys
  54         must match `self._feed_tensors.keys()`.
  55 
  56     Returns:
  57       A `dict` mapping strings to numpy arrays. The keys match
  58       `self.fetch_tensors.keys()`.
  59 
  60     Raises:
  61       ValueError: `input_dict` does not match `feed_tensors`.
  62     """
  63     # TODO(jamieas): make validation optional?
  64     input_keys = set(input_dict.keys())
  65     expected_keys = set(self.feed_tensors.keys())
  66     unexpected_keys = input_keys - expected_keys
  67     if unexpected_keys:
  68       raise ValueError(
  69           'Got unexpected keys in input_dict: {}\nexpected: {}'.format(
  70               unexpected_keys, expected_keys))
  71 
  72     feed_dict = {}
  73     for key in self.feed_tensors.keys():
  74       value = input_dict.get(key)
  75       if value is not None:
  76         feed_dict[self.feed_tensors[key]] = value
  77     return self._session.run(fetches=self.fetch_tensors, feed_dict=feed_dict)
