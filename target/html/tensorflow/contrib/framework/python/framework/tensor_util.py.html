  56 def reduce_sum_n(tensors, name=None):
  57   """Reduce tensors to a scalar sum.
  58 
  59   This reduces each tensor in `tensors` to a scalar via `tf.reduce_sum`, then
  60   adds them via `tf.add_n`.
  61 
  62   Args:
  63     tensors: List of tensors, all of the same numeric type.
  64     name: Tensor name, and scope for all other ops.
  65 
  66   Returns:
  67     Total loss tensor, or None if no losses have been configured.
  68 
  69   Raises:
  70     ValueError: if `losses` is missing or empty.
  71   """
  72   if not tensors:
  73     raise ValueError('No tensors provided.')
  74   with ops.name_scope(name, 'reduce_sum_n', tensors) as name_scope:
  75     tensors = [
  76         math_ops.reduce_sum(t, name='%s/sum' % t.op.name) for t in tensors]
  77     if len(tensors) == 1:
  78       return tensors[0]
  79     return math_ops.add_n(tensors, name=name_scope)

  85 def remove_squeezable_dimensions(predictions, labels, name=None):
  86   """Squeeze last dim if ranks of `predictions` and `labels` differ by 1.
  87 
  88   This will use static shape if available. Otherwise, it will add graph
  89   operations, which could result in a performance hit.
  90 
  91   Args:
  92     predictions: Predicted values, a `Tensor` of arbitrary dimensions.
  93     labels: Label values, a `Tensor` whose dimensions match `predictions`.
  94     name: Name of the op.
  95 
  96   Returns:
  97     Tuple of `predictions` and `labels`, possibly with last dim squeezed.
  98   """
  99   with ops.name_scope(name, 'remove_squeezable_dimensions',
 100                       [predictions, labels]):
 101     predictions = ops.convert_to_tensor(predictions)
 102     labels = ops.convert_to_tensor(labels)
 103     predictions_shape = predictions.get_shape()
 104     predictions_rank = predictions_shape.ndims
 105     labels_shape = labels.get_shape()
 106     labels_rank = labels_shape.ndims
 107     if (labels_rank is not None) and (predictions_rank is not None):
 108       # Use static rank.
 109       rank_diff = predictions_rank - labels_rank
 110       if rank_diff == -1:
 111         labels = array_ops.squeeze(labels, [-1])
 112       elif rank_diff == 1:
 113         predictions = array_ops.squeeze(predictions, [-1])
 114       return predictions, labels
 115 
 116     # Use dynamic rank.
 117     rank_diff = array_ops.rank(predictions) - array_ops.rank(labels)
 118     if (predictions_rank is None) or (
 119         predictions_shape.dims[-1].is_compatible_with(1)):
 120       predictions = control_flow_ops.cond(
 121           math_ops.equal(1, rank_diff),
 122           lambda: array_ops.squeeze(predictions, [-1]),
 123           lambda: predictions)
 124     if (labels_rank is None) or (
 125         labels_shape.dims[-1].is_compatible_with(1)):
 126       labels = control_flow_ops.cond(
 127           math_ops.equal(-1, rank_diff),
 128           lambda: array_ops.squeeze(labels, [-1]),
 129           lambda: labels)
 130     return predictions, labels

 133 def _shape_tensor_compatible(expected_shape, actual_shape):
 134   """Returns whether actual_shape is compatible with expected_shape.
 135 
 136   Note that -1 in `expected_shape` is recognized as unknown dimension.
 137 
 138   Args:
 139     expected_shape: Integer list defining the expected shape, or tensor of same.
 140     actual_shape: Shape of the tensor to test.
 141   Returns:
 142     New tensor.
 143   """
 144   with ops.name_scope('shape_tensor_equal',
 145                       values=[expected_shape, actual_shape]) as scope:
 146     return math_ops.reduce_all(
 147         math_ops.logical_or(
 148             math_ops.equal(expected_shape, -1),
 149             math_ops.equal(expected_shape, actual_shape, 'equal'),
 150             name='exclude_partial_shape'),
 151         name=scope)

 154 def _is_rank(expected_rank, actual_tensor):
 155   """Returns whether actual_tensor's rank is expected_rank.
 156 
 157   Args:
 158     expected_rank: Integer defining the expected rank, or tensor of same.
 159     actual_tensor: Tensor to test.
 160   Returns:
 161     New tensor.
 162   """
 163   with ops.name_scope('is_rank', values=[actual_tensor]) as scope:
 164     expected = ops.convert_to_tensor(expected_rank, name='expected')
 165     actual = array_ops.rank(actual_tensor, name='actual')
 166     return math_ops.equal(expected, actual, name=scope)

 169 def _is_shape(expected_shape, actual_tensor, actual_shape=None):
 170   """Returns whether actual_tensor's shape is expected_shape.
 171 
 172   Note that -1 in `expected_shape` is recognized as unknown dimension.
 173 
 174   Args:
 175     expected_shape: Integer list defining the expected shape, or tensor of same.
 176     actual_tensor: Tensor to test.
 177     actual_shape: Shape of actual_tensor, if we already have it.
 178   Returns:
 179     New tensor.
 180   """
 181   with ops.name_scope('is_shape', values=[actual_tensor]) as scope:
 182     is_rank = _is_rank(array_ops.size(expected_shape), actual_tensor)
 183     if actual_shape is None:
 184       actual_shape = array_ops.shape(actual_tensor, name='actual')
 185     shape_equal = _shape_tensor_compatible(expected_shape, actual_shape)
 186     return math_ops.logical_and(is_rank, shape_equal, name=scope)

 189 def _assert_shape_op(expected_shape, actual_tensor):
 190   """Asserts actual_tensor's shape is expected_shape.
 191 
 192   Note that unknown dimension in `expected_shape` will be ignored.
 193 
 194   Args:
 195     expected_shape: List of integers defining the expected shape, or tensor of
 196         same.
 197     actual_tensor: Tensor to test.
 198   Returns:
 199     New assert tensor.
 200   """
 201   with ops.name_scope('assert_shape', values=[actual_tensor]) as scope:
 202     actual_shape = array_ops.shape(actual_tensor, name='actual')
 203     if (isinstance(expected_shape, tensor_shape.TensorShape)
 204         and not expected_shape.is_fully_defined()):
 205       expected_shape = [d if d else -1 for d in expected_shape.as_list()]
 206     is_shape = _is_shape(expected_shape, actual_tensor, actual_shape)
 207     return control_flow_ops.Assert(
 208         is_shape, [
 209             'Wrong shape for %s [expected] [actual].' % actual_tensor.name,
 210             expected_shape,
 211             actual_shape
 212         ], name=scope)

 215 def with_same_shape(expected_tensor, tensor):
 216   """Assert tensors are the same shape, from the same graph.
 217 
 218   Args:
 219     expected_tensor: Tensor with expected shape.
 220     tensor: Tensor of actual values.
 221   Returns:
 222     The original tensor argument, possibly with assert ops added.
 223   """
 224   with ops.name_scope('%s/' % tensor.op.name, values=[expected_tensor, tensor]):
 225     tensor_shape = expected_tensor.get_shape()
 226     expected_shape = (
 227         tensor_shape.as_list() if tensor_shape.is_fully_defined()
 228         else array_ops.shape(expected_tensor, name='expected_shape'))
 229     return with_shape(expected_shape, tensor)

 232 def with_shape(expected_shape, tensor):
 233   """Asserts tensor has expected shape.
 234 
 235   If tensor shape and expected_shape, are fully defined, assert they match.
 236   Otherwise, add assert op that will validate the shape when tensor is
 237   evaluated, and set shape on tensor.
 238 
 239   Args:
 240     expected_shape: Expected shape to assert, as a 1D array of ints, or tensor
 241         of same.
 242     tensor: Tensor whose shape we're validating.
 243   Returns:
 244     tensor, perhaps with a dependent assert operation.
 245   Raises:
 246     ValueError: if tensor has an invalid shape.
 247   """
 248   if isinstance(tensor, sparse_tensor.SparseTensor):
 249     raise ValueError('SparseTensor not supported.')
 250 
 251   # Shape type must be 1D int32.
 252   if tensor_util.is_tensor(expected_shape):
 253     if expected_shape.dtype.base_dtype != dtypes.int32:
 254       raise ValueError(
 255           'Invalid dtype %s for shape %s expected of tensor %s.' % (
 256               expected_shape.dtype, expected_shape, tensor.name))
 257   if isinstance(expected_shape, (list, tuple)):
 258     if not expected_shape:
 259       expected_shape = np.asarray([], dtype=np.int32)
 260     else:
 261       np_expected_shape = np.asarray(expected_shape)
 262       expected_shape = (
 263           np.asarray(expected_shape, dtype=np.int32)
 264           if np_expected_shape.dtype == np.int64 else np_expected_shape)
 265   if isinstance(expected_shape, np.ndarray):
 266     if expected_shape.ndim > 1:
 267       raise ValueError(
 268           'Invalid rank %s for shape %s expected of tensor %s.' % (
 269               expected_shape.ndim, expected_shape, tensor.name))
 270     if expected_shape.dtype != np.int32:
 271       raise ValueError(
 272           'Invalid dtype %s for shape %s expected of tensor %s.' % (
 273               expected_shape.dtype, expected_shape, tensor.name))
 274 
 275   actual_shape = tensor.get_shape()
 276 
 277   if (not actual_shape.is_fully_defined()
 278       or tensor_util.is_tensor(expected_shape)):
 279     with ops.name_scope('%s/' % tensor.op.name, values=[tensor]):
 280       if (not tensor_util.is_tensor(expected_shape)
 281           and (len(expected_shape) < 1)):
 282         # TODO(irving): Remove scalar special case
 283         return array_ops.reshape(tensor, [])
 284       with ops.control_dependencies([_assert_shape_op(expected_shape, tensor)]):
 285         result = array_ops.identity(tensor)
 286       if not tensor_util.is_tensor(expected_shape):
 287         result.set_shape(expected_shape)
 288       return result
 289 
 290   if (not tensor_util.is_tensor(expected_shape) and
 291       not actual_shape.is_compatible_with(expected_shape)):
 292     if (len(expected_shape) < 1) and actual_shape.is_compatible_with([1]):
 293       # TODO(irving): Remove scalar special case.
 294       with ops.name_scope('%s/' % tensor.op.name, values=[tensor]):
 295         return array_ops.reshape(tensor, [])
 296     raise ValueError('Invalid shape for tensor %s, expected %s, got %s.' % (
 297         tensor.name, expected_shape, actual_shape))
 298 
 299   return tensor

 302 def assert_scalar_int(tensor, name=None):
 303   """Assert `tensor` is 0-D, of type `tf.int32` or `tf.int64`.
 304 
 305   Args:
 306     tensor: `Tensor` to test.
 307     name: Name of the op and of the new `Tensor` if one is created.
 308   Returns:
 309     `tensor`, for chaining.
 310   Raises:
 311     ValueError: if `tensor` is not 0-D, of integer type.
 312   """
 313   with ops.name_scope(name, 'assert_scalar_int', [tensor]) as name_scope:
 314     tensor = ops.convert_to_tensor(tensor)
 315     data_type = tensor.dtype
 316     if not data_type.base_dtype.is_integer:
 317       raise ValueError('Expected integer type for %s, received type: %s.'
 318                        % (tensor.name, data_type))
 319     return check_ops.assert_scalar(tensor, name=name_scope)
