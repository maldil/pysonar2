  27 def _add_experimental_function_notice_to_docstring(doc):
  28   """Adds an experimental notice to a docstring for experimental functions."""
  29   return decorator_utils.add_notice_to_docstring(
  30       doc, '',
  31       'EXPERIMENTAL FUNCTION',
  32       '(experimental)', ['THIS FUNCTION IS EXPERIMENTAL. It may change or '
  33                          'be removed at any time, and without warning.'])

  36 def experimental(func):
  37   """Decorator for marking functions or methods experimental.
  38 
  39   This decorator logs an experimental warning whenever the decorated function is
  40   called. It has the following format:
  41 
  42     <function> (from <module>) is experimental and may change or be removed at
  43     any time, and without warning.
  44 
  45   <function> will include the class name if it is a method.
  46 
  47   It also edits the docstring of the function: ' (experimental)' is appended
  48   to the first line of the docstring and a notice is prepended to the rest of
  49   the docstring.
  50 
  51   Args:
  52     func: A function or method to mark experimental.
  53 
  54   Returns:
  55     Decorated function or method.
  56   """
  57   decorator_utils.validate_callable(func, 'experimental')
  58   @functools.wraps(func)
  59   def new_func(*args, **kwargs):
  60     logging.warning(
  61         '%s (from %s) is experimental and may change or be removed at '
  62         'any time, and without warning.',
  63         decorator_utils.get_qualified_name(func), func.__module__)
  64     return func(*args, **kwargs)
  65   new_func.__doc__ = _add_experimental_function_notice_to_docstring(
  66       func.__doc__)
  67   return new_func
