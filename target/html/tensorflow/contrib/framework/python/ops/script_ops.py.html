  22 from __future__ import absolute_import

  26 from tensorflow.python.framework import tensor_shape
  27 from tensorflow.python.ops.script_ops import py_func as _py_func
  28 from tensorflow.python.util import nest
  29 
  30 __all__ = ['py_func']

  33 def py_func(func,
  34             args=(),
  35             kwargs=None,
  36             output_types=None,
  37             output_shapes=None,
  38             stateful=True,
  39             name=None):
  40   """Wraps a python function and uses it as a TensorFlow op.
  41 
  42   This function is a wrapper around `tf.py_func` and improve it with kwargs
  43   and output_shapes. Further it changed some argument names.
  44 
  45   Given a python function `func`, which takes numpy arrays as its
  46   inputs and returns numpy arrays as its outputs, wrap this function as an
  47   operation in a TensorFlow graph. The following snippet constructs a simple
  48   TensorFlow graph that invokes the `np.sinh()` NumPy function as a operation
  49   in the graph:
  50 
  51   ```python
  52   def my_func(x):
  53     # x will be a numpy array with the contents of the placeholder below
  54     return np.sinh(x)
  55   inp = tf.placeholder(tf.float32)
  56   y = tf.py_func(my_func, [inp], tf.float32)
  57   ```
  58 
  59 
  60   **N.B.** The `tf.py_func()` operation has the following known limitations:
  61 
  62   * The body of the function (i.e. `func`) will not be serialized in a
  63     `GraphDef`. Therefore, you should not use this function if you need to
  64     serialize your model and restore it in a different environment.
  65 
  66   * The operation must run in the same address space as the Python program
  67     that calls `tf.py_func()`. If you are using distributed TensorFlow, you
  68     must run a `tf.train.Server` in the same process as the program that calls
  69     `tf.py_func()` and you must pin the created operation to a device in that
  70     server (e.g. using `with tf.device():`).
  71 
  72   Args:
  73     func: A Python function, which accepts a list of NumPy `ndarray` objects
  74       having element types that match the corresponding `tf.Tensor` objects
  75       in `inp`, and returns a list of `ndarray` objects (or a single `ndarray`)
  76       having element types that match the corresponding values in `Tout`.
  77     args: A list of `Tensor` objects.
  78     kwargs: A dict with `Tensor` objects as values.
  79     output_types: A nested structure of tensorflow data types or a single
  80       tensorflow data type if there is only one, indicating what `func` returns.
  81     output_shapes: Same as output_types, except the types are replaces with
  82       shapes (optional).
  83     stateful: (Boolean.) If True, the function should be considered stateful.
  84       If a function is stateless, when given the same input it will return the
  85       same output and have no observable side effects. Optimizations such as
  86       common subexpression elimination are only performed on stateless
  87       operations.
  88     name: A name for the operation (optional).
  89 
  90   Returns:
  91     Tensorflow op that wraps the input python function.
  92   """
  93 
  94   if kwargs is None:
  95     kwargs = {}
  96 
  97   if not isinstance(args, (list, tuple)):
  98     raise TypeError('args must be list and not {}. args: {}'.format(
  99         type(args), args))
 100 
 101   if not isinstance(kwargs, dict):
 102     raise TypeError('kwargs must be dict and not {}. args: {}'.format(
 103         type(kwargs), kwargs))
 104 
 105   # For dynamic type inference use callable output_types and output_shapes
 106   if callable(output_types):
 107     # If callable assume same signature and call with tensors and get the types
 108     output_types = output_types(*args, **kwargs)
 109   if callable(output_shapes):
 110     # If callable assume same signature and call with tensors and get the shapes
 111     output_shapes = output_shapes(*args, **kwargs)
 112 
 113   flat_output_types = nest.flatten(output_types)
 114   args = (args, kwargs)
 115   flat_args = nest.flatten(args)
 116 
 117   def python_function_wrapper(*py_args):
 118     py_args, py_kwargs = nest.pack_sequence_as(args, py_args)
 119 
 120     ret = func(*py_args, **py_kwargs)
 121     # TODO(alextp): Catch Exceptions and improve msg, because tensorflow
 122     # ist not able to preserve the traceback, i.e. the Exceptions does not
 123     # contain any information where the Exception was raised.
 124     nest.assert_shallow_structure(output_types, ret)
 125     return nest.flatten(ret)
 126 
 127   flat_values = _py_func(
 128       python_function_wrapper,
 129       flat_args,
 130       flat_output_types,
 131       stateful=stateful,
 132       name=name)
 133 
 134   if output_shapes is not None:
 135     # I am not sure if this is nessesary
 136     output_shapes = nest.map_structure_up_to(
 137         output_types, tensor_shape.as_shape, output_shapes)
 138 
 139     flattened_shapes = nest.flatten(output_shapes)
 140     for ret_t, shape in zip(flat_values, flattened_shapes):
 141       ret_t.set_shape(shape)
 142 
 143   return nest.pack_sequence_as(output_types, flat_values)
