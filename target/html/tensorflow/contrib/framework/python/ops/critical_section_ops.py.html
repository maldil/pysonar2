  49 def _identity(x):
  50   """Identity op that recognizes `TensorArray`, `Operation`, and `Tensor`."""
  51   if isinstance(x, tensor_array_ops.TensorArray):
  52     return x.identity()
  53   elif isinstance(x, ops.Operation):
  54     return control_flow_ops.group(x)
  55   elif context.executing_eagerly() and x is None:
  56     return None
  57   else:
  58     return array_ops.identity(x)

  61 def _get_colocation(op):
  62   """Get colocation symbol from op, if any."""
  63   try:
  64     return op.get_attr("_class")
  65   except ValueError:
  66     return None

  69 class CriticalSection(object):
  70   """Critical section.
  71 
  72   A `CriticalSection` object is a resource in the graph which executes subgraphs
  73   in **serial** order.  A common example of a subgraph one may wish to run
  74   exclusively is the one given by the following function:
  75 
  76   ```python
  77   v = resource_variable_ops.ResourceVariable(0.0, name="v")
  78 
  79   def count():
  80     value = v.read_value()
  81     with tf.control_dependencies([value]):
  82       with tf.control_dependencies([v.assign_add(1)]):
  83         return tf.identity(value)
  84   ```
  85 
  86   Here, a snapshot of `v` is captured in `value`; and then `v` is updated.
  87   The snapshot value is returned.
  88 
  89   If multiple workers or threads all execute `count` in parallel, there is no
  90   guarantee that access to the variable `v` is atomic at any point within
  91   any thread's calculation of `count`.  In fact, even implementing an atomic
  92   counter that guarantees that the user will see each value `0, 1, ...,` is
  93   currently impossible.
  94 
  95   The solution is to ensure any access to the underlying resource `v` is
  96   only processed through a critical section:
  97 
  98   ```python
  99   cs = CriticalSection()
 100   f1 = cs.execute(count)
 101   f2 = cs.execute(count)
 102   output = f1 + f2
 103   session.run(output)
 104   ```
 105   The functions `f1` and `f2` will be executed serially, and updates to `v`
 106   will be atomic.
 107 
 108   **NOTES**
 109 
 110   All resource objects, including the critical section and any captured
 111   variables of functions executed on that critical section, will be
 112   colocated to the same device (host and cpu/gpu).
 113 
 114   When using multiple critical sections on the same resources, there is no
 115   guarantee of exclusive access to those resources.  This behavior is disallowed
 116   by default (but see the kwarg `exclusive_resource_access`).
 117 
 118   For example, running the same function in two separate critical sections
 119   will not ensure serial execution:
 120 
 121   ```python
 122   v = tf.get_variable("v", initializer=0.0, use_resource=True)
 123   def accumulate(up):
 124     x = v.read_value()
 125     with tf.control_dependencies([x]):
 126       with tf.control_dependencies([v.assign_add(up)]):
 127         return tf.identity(x)
 128   ex1 = CriticalSection().execute(
 129     accumulate, 1.0, exclusive_resource_access=False)
 130   ex2 = CriticalSection().execute(
 131     accumulate, 1.0, exclusive_resource_access=False)
 132   bad_sum = ex1 + ex2
 133   sess.run(v.initializer)
 134   sess.run(bad_sum)  # May return 0.0
 135   ```
 136   """

 161   def _init_from_args(self, name, shared_name):  # pylint: disable=invalid-name
 162     """Initialize the CriticalSection from constructor arguments."""
 163     with ops.name_scope(name, "CriticalSection", []) as name:
 164       with ops.init_scope():
 165         # pylint: disable=protected-access
 166         container = ops.get_default_graph()._container
 167         # pylint: enable=protected-access
 168         if shared_name is None:
 169           shared_name = name
 170         if container is None:
 171           container = ""
 172         self._handle = gen_resource_variable_ops.mutex_v2(
 173             shared_name=shared_name, container=container, name=name)
 174 
 175     if not context.executing_eagerly():
 176       ops.add_to_collections(CRITICAL_SECTIONS, self)
 177 
 178   @property
 179   def name(self):
 180     return self._handle.op.name

 182   def execute(self, fn, *args, **kwargs):
 183     """Execute function `fn(*args, **kwargs)` inside the CriticalSection.
 184 
 185     Args:
 186       fn: The function to execute.  Must return at least one tensor.
 187       *args: Additional positional arguments to `fn`.
 188       **kwargs: Additional keyword arguments to `fn`.
 189         Several keywords are reserved for `execute`.  These are:
 190 
 191         - name; The name to use when creating the execute operation.
 192         - exclusive_resource_access; Whether the resources required by
 193           `fn` should be exclusive to this `CriticalSection`.  Default: `True`.
 194           You may want to set this to `False` if you will be accessing a
 195           resource in read-only mode in two different CriticalSections.
 196 
 197     Returns:
 198       The tensors returned from `fn(*args, **kwargs)`.
 199 
 200     Raises:
 201       ValueError: If `fn` attempts to lock this `CriticalSection` in any nested
 202         or lazy way that may cause a deadlock.
 203       ValueError: If `exclusive_resource_access` is not provided (is `True`) and
 204         another `CriticalSection` has an execution requesting the same
 205         resources as in `*args`, `**kwargs`, and any additionally captured
 206         inputs in `fn`.  Note, even if `exclusive_resource_access` is `True`,
 207         if another execution in another `CriticalSection` was created without
 208         `exclusive_resource_access=True`, a `ValueError` will be raised.
 209     """
 210     name = kwargs.pop("name", None)
 211     exclusive_resource_access = kwargs.pop("exclusive_resource_access", True)
 212 
 213     with ops.name_scope(name, "critical_section_execute", []):
 214 
 215       # Ensure that mutex locking only happens *after* all args and
 216       # kwargs have been executed.  This avoids certain types of deadlocks.
 217       lock = gen_resource_variable_ops.mutex_lock(self._handle)
 218 
 219       if not context.executing_eagerly():
 220         # NOTE(ebrevdo): This is to ensure we don't pick up spurious
 221         # Operations created by other threads.
 222         with ops.get_default_graph()._lock:  # pylint: disable=protected-access
 223           existing_ops = ops.get_default_graph().get_operations()
 224           with ops.control_dependencies([lock]):
 225             r = fn(*args, **kwargs)
 226           # TODO(ebrevdo): If creating critical sections in a python loop, this
 227           # makes graph creation time quadratic.  Revisit if this
 228           # becomes a problem.
 229           created_ops = (set(ops.get_default_graph().get_operations())
 230                          .difference(existing_ops))
 231       else:
 232         with ops.control_dependencies([lock]):
 233           r = fn(*args, **kwargs)
 234 
 235       if not context.executing_eagerly():
 236         self._add_control_dependencies_to_lock(created_ops, lock.op)
 237 
 238         # captured_resources is a list of resources that are directly
 239         # accessed only by ops created during fn(), not by any
 240         # ancestors of those ops in the graph.
 241         captured_resources = set([
 242             input_ for op in created_ops
 243             for input_ in op.inputs
 244             if input_.dtype == dtypes.resource
 245         ])
 246 
 247         # NOTE(ebrevdo): The only time self._is_self_handle() is True
 248         # in this call is if one of the recently created ops, within
 249         # the execute(), themselves attempt to access the
 250         # CriticalSection.  This will cause a deadlock.
 251         if any(self._is_self_handle(x) for x in captured_resources):
 252           raise ValueError("The function fn attempts to directly access the "
 253                            "CriticalSection in which it would be running.  "
 254                            "This is illegal and would cause deadlocks.")
 255 
 256         self._check_multiple_access_to_resources(
 257             captured_resources, exclusive_resource_access)
 258 
 259       r_flat = [_identity(x) for x in nest.flatten(r)]
 260 
 261       with ops.control_dependencies(r_flat):
 262         # The identity must run on the same machine as self._handle
 263         with ops.colocate_with(self._handle):
 264           # Do not use array_ops.identity as there are special
 265           # optimizations within TensorFlow which seem to elide it
 266           # even when optimizations are disabled(!).
 267           ensure_lock_exists = gen_resource_variable_ops.consume_mutex_lock(
 268               lock)
 269 
 270         # Make sure that if any element of r is accessed, all of
 271         # them are executed together.
 272         r = nest.pack_sequence_as(r, control_flow_ops.tuple(nest.flatten(r)))
 273 
 274       with ops.control_dependencies([ensure_lock_exists]):
 275         outputs = nest.map_structure(_identity, r)
 276 
 277       if not context.executing_eagerly():
 278         signature = _ExecutionSignature(
 279             op=lock.op,
 280             handle=self._handle,
 281             resources=list(captured_resources),
 282             exclusive_resource_access=exclusive_resource_access)
 283         ops.add_to_collections(
 284             CRITICAL_SECTION_EXECUTIONS, signature)
 285 
 286       return outputs
 287 
 288   def _add_control_dependencies_to_lock(self, created_ops, lock_op):
 289     """To avoid deadlocks, all args must be executed before lock_op."""
 290     # Get all arguments (explicit and captured) of all ops created by fn().
 291     all_args = set([input_.op for op in created_ops for input_ in op.inputs])
 292     all_args.update(
 293         input_op for op in created_ops for input_op in op.control_inputs)
 294     # Unfortunately, we can't use sets throughout because TF seems to
 295     # create new Operation objects for the same op sometimes; and we
 296     # can't rely on id(op).
 297 
 298     # pylint: disable=protected-access
 299     all_args_dict = dict((op._id, op) for op in all_args)
 300 
 301     # Remove ops created within fn, or that lock_op already has a
 302     # control dependency on.  Also remove a possible self-loop.
 303     for op in created_ops:
 304       all_args_dict.pop(op._id, None)
 305     for op in lock_op.control_inputs:
 306       all_args_dict.pop(op._id, None)
 307     for input_ in lock_op.inputs:
 308       all_args_dict.pop(input_.op._id, None)
 309     all_args_dict.pop(lock_op._id, None)
 310 
 311     all_args = all_args_dict.values()
 312 
 313     if not all_args:
 314       # No control dependencies to add; return early.
 315       return
 316 
 317     # This group is important: it ensures that any ops in all_args
 318     # outside the control context of the lock_op (and this fn, which
 319     # runs in the same context) are added to this context before
 320     # being added to the control dependencies of lock_op.
 321     all_args = control_flow_ops.group(*all_args)
 322 
 323     lock_op._add_control_input(all_args)
 324     # pylint: enable=protected-access

 326   def _is_self_handle(self, x):
 327     """Check if the tensor `x` is the same Mutex as `self._handle`."""
 328     if isinstance(x, ops.EagerTensor):
 329       return x is self._handle
 330     return (x.op.type == "MutexV2"
 331             # blank shared_name means the op will create a unique one.
 332             and x.op.get_attr("shared_name")
 333             and (x.op.get_attr("shared_name") ==
 334                  self._handle.op.get_attr("shared_name"))
 335             and (x.op.device == self._handle.op.device
 336                  or _get_colocation(x.op) == _get_colocation(self._handle.op)))
 337 
 338   def _check_multiple_access_to_resources(
 339       self, captured_resources, exclusive_resource_access):
 340     """Raise if captured_resources are accessed by another CriticalSection.
 341 
 342     Args:
 343       captured_resources: Set of tensors of type resource.
 344       exclusive_resource_access: Whether this execution requires exclusive
 345         resource access.
 346 
 347     Raises:
 348       ValueError: If any tensors in `captured_resources` are also accessed
 349         by another `CriticalSection`, and at least one of them requires
 350         exclusive resource access.
 351     """
 352     # Collections and op introspection does not work in eager
 353     # mode.  This is generally ok; since eager mode (as of
 354     # writing) executes sequentially anyway.
 355     for sg in ops.get_collection(CRITICAL_SECTION_EXECUTIONS):
 356       if self._is_self_handle(sg.handle):
 357         # Other executions in the same critical section are allowed.
 358         continue
 359       if not (exclusive_resource_access or sg.exclusive_resource_access):
 360         # Neither execution requested exclusive access.
 361         continue
 362       resource_intersection = captured_resources.intersection(sg.resources)
 363       if resource_intersection:
 364         raise ValueError(
 365             "This execution would access resources: %s.  Either this "
 366             "lock (CriticalSection: %s) or lock '%s' "
 367             "(CriticalSection: %s) requested exclusive resource access "
 368             "of this resource.  Did you mean to call execute with keyword "
 369             "argument exclusive_resource_access=False?" %
 370             (list(resource_intersection), self._handle, sg, sg.handle))
