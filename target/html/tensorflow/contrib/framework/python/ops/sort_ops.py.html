  35 def sort(values, axis=-1, direction='ASCENDING', name=None):
  36   """Sorts a tensor.
  37 
  38   Args:
  39     values: 1-D or higher numeric `Tensor`.
  40     axis: The axis along which to sort. The default is -1, which sorts the last
  41         axis.
  42     direction: The direction in which to sort the values (`'ASCENDING'` or
  43         `'DESCENDING'`).
  44     name: Optional name for the operation.
  45 
  46   Returns:
  47     A `Tensor` with the same dtype and shape as `values`, with the elements
  48         sorted along the given `axis`.
  49 
  50   Raises:
  51     ValueError: If axis is not a constant scalar, or the direction is invalid.
  52   """
  53   with framework_ops.name_scope(name, 'sort'):
  54     return _sort_or_argsort(values, axis, direction, return_argsort=False)
  55

  57 def argsort(values, axis=-1, direction='ASCENDING', stable=False, name=None):
  58   """Returns the indices of a tensor that give its sorted order along an axis.
  59 
  60   For a 1D tensor, `tf.gather(values, tf.argsort(values))` is equivalent to
  61   `tf.sort(values)`. For higher dimensions, the output has the same shape as
  62   `values`, but along the given axis, values represent the index of the sorted
  63   element in that slice of the tensor at the given position.
  64 
  65   Args:
  66     values: 1-D or higher numeric `Tensor`.
  67     axis: The axis along which to sort. The default is -1, which sorts the last
  68         axis.
  69     direction: The direction in which to sort the values (`'ASCENDING'` or
  70         `'DESCENDING'`).
  71     stable: If True, equal elements in the original tensor will not be
  72         re-ordered in the returned order. Unstable sort is not yet implemented,
  73         but will eventually be the default for performance reasons. If you
  74         require a stable order, pass `stable=True` for forwards compatibility.
  75     name: Optional name for the operation.
  76 
  77   Returns:
  78     An int32 `Tensor` with the same shape as `values`. The indices that would
  79         sort each slice of the given `values` along the given `axis`.
  80 
  81   Raises:
  82     ValueError: If axis is not a constant scalar, or the direction is invalid.
  83   """
  84   del stable  # Unused.
  85   with framework_ops.name_scope(name, 'argsort'):
  86     return _sort_or_argsort(values, axis, direction, return_argsort=True)

  89 def _sort_or_argsort(values, axis, direction, return_argsort):
  90   """Internal sort/argsort implementation.
  91 
  92   Args:
  93     values: The input values.
  94     axis: The axis along which to sort.
  95     direction: 'ASCENDING' or 'DESCENDING'.
  96     return_argsort: Whether to return the argsort result.
  97 
  98   Returns:
  99     Either the sorted values, or the indices of the sorted values in the
 100         original tensor. See the `sort` and `argsort` docstrings.
 101 
 102   Raises:
 103     ValueError: If axis is not a constant scalar, or the direction is invalid.
 104   """
 105   if direction not in _SORT_IMPL:
 106     raise ValueError('%s should be one of %s' %
 107                      (direction, ', '.join(sorted(_SORT_IMPL.keys()))))
 108   # Axis must be an integer, not a Tensor.
 109   axis = framework_ops.convert_to_tensor(axis, name='axis')
 110   axis_static = tensor_util.constant_value(axis)
 111   if axis.shape.ndims != 0 or axis_static is None:
 112     raise ValueError('axis must be a constant scalar')
 113   axis_static = int(axis_static)  # Avoids NumPy casting error
 114 
 115   values = framework_ops.convert_to_tensor(values, name='values')
 116 
 117   return _SORT_IMPL[direction](values, axis_static, return_argsort)

 120 def _descending_sort(values, axis, return_argsort=False):
 121   """Sorts values in reverse using `top_k`.
 122 
 123   Args:
 124     values: Tensor of numeric values.
 125     axis: Index of the axis which values should be sorted along.
 126     return_argsort: If False, return the sorted values. If True, return the
 127         indices that would sort the values.
 128 
 129   Returns:
 130     The sorted values.
 131   """
 132   k = array_ops.shape(values)[axis]
 133   rank = array_ops.rank(values)
 134   static_rank = values.shape.ndims
 135   # Fast path: sorting the last axis.
 136   if axis == -1 or axis + 1 == values.get_shape().ndims:
 137     top_k_input = values
 138     transposition = None
 139   else:
 140     # Otherwise, transpose the array. Swap axes `axis` and `rank - 1`.
 141     if axis < 0:
 142       # Calculate the actual axis index if counting from the end. Use the static
 143       # rank if available, or else make the axis back into a tensor.
 144       axis += static_rank or rank
 145     if static_rank is not None:
 146       # Prefer to calculate the transposition array in NumPy and make it a
 147       # constant.
 148       transposition = constant_op.constant(
 149           np.r_[
 150               # Axes up to axis are unchanged.
 151               np.arange(axis),
 152               # Swap axis and rank - 1.
 153               [static_rank - 1],
 154               # Axes in [axis + 1, rank - 1) are unchanged.
 155               np.arange(axis + 1, static_rank - 1),
 156               # Swap axis and rank - 1.
 157               [axis]],
 158           name='transposition')
 159     else:
 160       # Generate the transposition array from the tensors.
 161       transposition = array_ops.concat(
 162           [
 163               # Axes up to axis are unchanged.
 164               math_ops.range(axis),
 165               # Swap axis and rank - 1.
 166               [rank - 1],
 167               # Axes in [axis + 1, rank - 1) are unchanged.
 168               math_ops.range(axis + 1, rank - 1),
 169               # Swap axis and rank - 1.
 170               [axis]
 171           ],
 172           axis=0)
 173     top_k_input = array_ops.transpose(values, transposition)
 174 
 175   values, indices = nn_ops.top_k(top_k_input, k)
 176   return_value = indices if return_argsort else values
 177   if transposition is not None:
 178     # transposition contains a single cycle of length 2 (swapping 2 elements),
 179     # so it is an involution (it is its own inverse).
 180     return_value = array_ops.transpose(return_value, transposition)
 181   return return_value

 184 def _ascending_sort(values, axis, return_argsort=False):
 185   # Negate the values to get the ascending order from descending sort.
 186   values_or_indices = _descending_sort(-values, axis, return_argsort)
 187   # If not argsort, negate the values again.
 188   return values_or_indices if return_argsort else -values_or_indices
