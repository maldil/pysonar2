  35 def _get_tensor_repr(t,
  36                      print_tensor_name=True,
  37                      print_tensor_type=True,
  38                      print_shape=True,
  39                      summarize_indicator_vector=True):
  40   """Return a list of Tensors that summarize the given tensor t."""
  41   tensor_list = []
  42   if print_tensor_name and isinstance(t, ops.Tensor):
  43     tensor_list.append(constant_op.constant("Name: " + t.name))
  44 
  45   if print_tensor_type:
  46     if isinstance(t, ops.Tensor):
  47       t_type_str = "Type: Tensor ({})".format(t.dtype.name)
  48     elif isinstance(t, sparse_tensor.SparseTensor):
  49       t_type_str = "Type: SparseTensor ({})".format(t.dtype.name)
  50     elif isinstance(t, tensor_array_ops.TensorArray):
  51       t_type_str = "Type: TensorArray ({})".format(t.dtype.name)
  52     elif isinstance(t, variables.Variable):
  53       t_type_str = "Type: Variable ({})".format(t.dtype.name)
  54     else:
  55       raise ValueError("t must be a Tensor, SparseTensor, TensorArray or "
  56                        "Variable.")
  57 
  58     tensor_list.append(constant_op.constant(t_type_str))
  59 
  60   if print_shape:
  61     if isinstance(t, sparse_tensor.SparseTensor):
  62       tensor_list.append(constant_op.constant("Shape:"))
  63       tensor_list.append(t.dense_shape)
  64     elif isinstance(t, ops.Tensor):
  65       tensor_list.append(constant_op.constant("Shape: " + str(t.get_shape(
  66       ).dims)))
  67     elif isinstance(t, tensor_array_ops.TensorArray):
  68       tensor_list.append(constant_op.constant("Size:"))
  69       tensor_list.append(t.size())
  70 
  71   if summarize_indicator_vector and t.dtype == dtypes.bool:
  72     int_tensor = math_ops.cast(t, dtypes.uint8)
  73     tensor_list.append(constant_op.constant("First True in Boolean tensor at:"))
  74     tensor_list.append(math_ops.argmax(int_tensor, 0))
  75 
  76   if isinstance(t, sparse_tensor.SparseTensor):
  77     tensor_list.append(constant_op.constant("Sparse indices:"))
  78     tensor_list.append(t.indices)
  79     tensor_list.append(constant_op.constant("Sparse values:"))
  80     tensor_list.append(t.values)
  81   elif isinstance(t, ops.Tensor):
  82     tensor_list.append(constant_op.constant("Value:"))
  83     tensor_list.append(t)
  84   elif isinstance(t, tensor_array_ops.TensorArray):
  85     tensor_list.append(constant_op.constant("Value:"))
  86     tensor_list.append(t.stack())
  87 
  88   return tensor_list

  91 def print_op(input_,
  92              data=None,
  93              message=None,
  94              first_n=None,
  95              summarize=20,
  96              print_tensor_name=True,
  97              print_tensor_type=True,
  98              print_shape=True,
  99              summarize_indicator_vector=True,
 100              name=None):
 101   """Creates a print op that will print when a tensor is accessed.
 102 
 103   Wraps the tensor passed in so that whenever that tensor is accessed,
 104   the message `message` is printed, along with the current value of the
 105   tensor `t` and an optional list of other tensors.
 106 
 107   Args:
 108     input_: A Tensor/SparseTensor/TensorArray to print when it is evaluated.
 109     data: A list of other tensors to print.
 110     message: A string message to print as a prefix.
 111     first_n: Only log `first_n` number of times. Negative numbers log always;
 112              this is the default.
 113     summarize: Print this number of elements in the tensor.
 114     print_tensor_name: Print the tensor name.
 115     print_tensor_type: Print the tensor type.
 116     print_shape: Print the tensor's shape.
 117     summarize_indicator_vector: Whether to print the index of the first true
 118       value in an indicator vector (a Boolean tensor).
 119     name: The name to give this op.
 120 
 121   Returns:
 122     A Print op. The Print op returns `input_`.
 123 
 124   Raises:
 125     ValueError: If the tensor `input_` is not a Tensor, SparseTensor or
 126       TensorArray.
 127 
 128   """
 129 
 130   message = message or ""
 131   if input_ is None:
 132     raise ValueError("input_ must be of type "
 133                      "Tensor, SparseTensor or TensorArray")
 134 
 135   tensor_list = _get_tensor_repr(input_, print_tensor_name, print_tensor_type,
 136                                  print_shape, summarize_indicator_vector)
 137 
 138   if data is not None:
 139     for t in data:
 140       tensor_list.extend(_get_tensor_repr(t, print_tensor_name,
 141                                           print_tensor_type, print_shape,
 142                                           summarize_indicator_vector))
 143 
 144   if isinstance(input_, ops.Tensor) or isinstance(input_, variables.Variable):
 145     input_ = logging_ops.Print(input_, tensor_list, message, first_n, summarize,
 146                                name)
 147   elif isinstance(input_, sparse_tensor.SparseTensor):
 148     p = logging_ops.Print(
 149         constant_op.constant([]), tensor_list, message, first_n, summarize,
 150         name)
 151 
 152     with ops.control_dependencies([p]):
 153       input_ = sparse_tensor.SparseTensor(
 154           array_ops.identity(input_.indices),
 155           array_ops.identity(input_.values),
 156           array_ops.identity(input_.dense_shape))
 157   elif isinstance(input_, tensor_array_ops.TensorArray):
 158     p = logging_ops.Print(
 159         constant_op.constant([]), tensor_list, message, first_n, summarize,
 160         name)
 161 
 162     with ops.control_dependencies([p]):
 163       input_ = tensor_array_ops.TensorArray(dtype=input_.dtype,
 164                                             handle=input_.handle,
 165                                             flow=input_.flow)
 166   else:
 167     raise ValueError("input_ must be of type "
 168                      "Tensor, SparseTensor or TensorArray")
 169 
 170   return input_
