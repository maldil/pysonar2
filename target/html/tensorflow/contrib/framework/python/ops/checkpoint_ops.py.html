  32 def load_linear_multiclass_bias_initializer(ckpt_path,
  33                                             bias_tensor_name,
  34                                             new_class_vocab_size,
  35                                             old_class_vocab_file,
  36                                             new_class_vocab_file,
  37                                             num_class_oov_buckets=0,
  38                                             initializer=None,
  39                                             max_rows_in_memory=-1):
  40   """Loads pre-trained multi-class biases for linear models from checkpoint.
  41 
  42   Wrapper around `load_and_remap_matrix_initializer()` specialized for loading
  43   multi-class bias and remapping according to the provided vocab files. See docs
  44   for `load_and_remap_matrix_initializer()` for more details. In this case, the
  45   provided row_vocab is the class vocabulary, and the expected shape is
  46   `[new_class_vocab_size, 1]`.
  47 
  48   Args:
  49     ckpt_path: Path to the TensorFlow checkpoint (version 2, `TensorBundle`)
  50       from which the old matrix `Tensor` will be loaded.
  51     bias_tensor_name: Tensor name to load from in the checkpoints.
  52     new_class_vocab_size: Number of entries in the new class vocab.
  53     old_class_vocab_file: A scalar `Tensor` of type `string` containing the
  54       path to the old class vocabulary file.
  55     new_class_vocab_file: A scalar `Tensor` of type `string` containing the
  56       path to the new class vocabulary file.
  57     num_class_oov_buckets: `int` specifying the number of out-of-vocabulary
  58       buckets to use for the classes. Must be >= 0.
  59     initializer: Initializer function that accepts a 1-D tensor as the arg to
  60       specify the shape of the returned tensor. If `None`, defaults to using
  61       `zeros_initializer()`.
  62     max_rows_in_memory: `int` specifying the maximum number of rows to load from
  63       the checkpoint at once. If less than or equal to 0, the entire matrix will
  64       be loaded into memory. Setting this arg trades increased disk reads for
  65       lower memory usage.
  66 
  67   Returns:
  68     A variable initializer function.
  69   """
  70   # Linear multi-class biases should be zero-initialized.
  71   if initializer is None:
  72     initializer = init_ops.zeros_initializer()
  73 
  74   return load_and_remap_matrix_initializer(
  75       ckpt_path=ckpt_path,
  76       old_tensor_name=bias_tensor_name,
  77       new_row_vocab_size=new_class_vocab_size,
  78       new_col_vocab_size=1,
  79       old_row_vocab_file=old_class_vocab_file,
  80       new_row_vocab_file=new_class_vocab_file,
  81       old_col_vocab_file=None,
  82       new_col_vocab_file=None,
  83       num_row_oov_buckets=num_class_oov_buckets,
  84       num_col_oov_buckets=0,
  85       initializer=initializer,
  86       max_rows_in_memory=max_rows_in_memory)
  87

  89 def load_variable_slot_initializer(ckpt_path,
  90                                    old_tensor_name,
  91                                    primary_partition_info,
  92                                    new_row_vocab_size,
  93                                    new_col_vocab_size,
  94                                    old_row_vocab_file=None,
  95                                    new_row_vocab_file=None,
  96                                    old_col_vocab_file=None,
  97                                    new_col_vocab_file=None,
  98                                    num_row_oov_buckets=0,
  99                                    num_col_oov_buckets=0,
 100                                    initializer=None,
 101                                    max_rows_in_memory=-1):
 102   """Loads pre-trained multi-class slots for linear models from checkpoint.
 103 
 104   Wrapper around `load_and_remap_matrix_initializer()` specialized for loading
 105   multi-class slots (such as optimizer accumulators) and remapping them
 106   according to the provided vocab files. See docs for
 107   `load_and_remap_matrix_initializer()` for more details.  Takes in a
 108   `variable_scope._PartitionInfo` representing the slot's primary `Variable`'s
 109   partitioning.  This is necessary since accumulator `Variable` creation ignores
 110   primary scoping and partitioning information.
 111 
 112   Args:
 113     ckpt_path: Path to the TensorFlow checkpoint (version 2, `TensorBundle`)
 114       from which the old matrix `Tensor` will be loaded.
 115     old_tensor_name: Name of the 2-D `Tensor` to load from checkpoint.
 116     primary_partition_info: A `variable_scope._PartitionInfo` containing this
 117       slot's primary `Variable`'s partitioning information.  This is used to
 118       calculate the offset and override the partition_info passed to the call to
 119       _initialize.
 120     new_row_vocab_size: `int` specifying the number of entries in
 121       `new_row_vocab_file`. If no row remapping is needed (no row vocab
 122       provided), this should be equal to the number of rows to load from the old
 123       matrix (which can theoretically be smaller than the number of rows in the
 124       old matrix).
 125     new_col_vocab_size: `int` specifying the number of entries in
 126       `new_col_vocab_file`. If no column remapping is needed (no column vocab
 127       provided), this should be equal to the number of columns in the old
 128       matrix.
 129     old_row_vocab_file: A scalar `Tensor` of type `string` containing the
 130       path to the old row vocabulary file. Can be None, which represents no
 131       remapping on the row axis.
 132     new_row_vocab_file: A scalar `Tensor` of type `string` containing the path
 133       to the new row vocabulary file. Can be None, which represents no remapping
 134       on the row axis.
 135     old_col_vocab_file: A scalar `Tensor` of type `string` containing the
 136       path to the old column vocabulary file. Can be None, which represents no
 137       remapping on the column axis.
 138     new_col_vocab_file: A scalar `Tensor` of type `string` containing the path
 139       to the new column vocabulary file. Can be None, which represents no
 140       remapping on the column axis.
 141     num_row_oov_buckets: `int` specifying the number of out-of-vocabulary rows
 142       to append. Must be >= 0.
 143     num_col_oov_buckets: `int` specifying the number of out-of-vocabulary
 144       columns to append. Must be >= 0.
 145     initializer: Initializer function to initialize missing values. Accepts a
 146       1-D tensor as the arg to specify the shape of the returned tensor. If
 147       `None`, defaults to using `zeros_initializer()`.
 148     max_rows_in_memory: `int` specifying the maximum number of rows to load from
 149       the checkpoint at once. If less than or equal to 0, the entire matrix will
 150       be loaded into memory. Setting this arg trades increased disk reads for
 151       lower memory usage.
 152 
 153   Returns:
 154     A variable initializer function that should be used to initialize a
 155     (potentially partitioned) `Variable` whose complete shape is
 156     `[new_row_vocab_size + num_row_oov_buckets, new_col_vocab_size +
 157     num_col_oov_buckets]`.
 158 
 159   Raises:
 160     TypeError: If `initializer` is specified but not callable.
 161   """
 162   initializer_fn = load_and_remap_matrix_initializer(
 163       ckpt_path=ckpt_path,
 164       old_tensor_name=old_tensor_name,
 165       new_row_vocab_size=new_row_vocab_size,
 166       new_col_vocab_size=new_col_vocab_size,
 167       old_row_vocab_file=old_row_vocab_file,
 168       new_row_vocab_file=new_row_vocab_file,
 169       old_col_vocab_file=old_col_vocab_file,
 170       new_col_vocab_file=new_col_vocab_file,
 171       num_row_oov_buckets=num_row_oov_buckets,
 172       num_col_oov_buckets=num_col_oov_buckets,
 173       initializer=initializer,
 174       max_rows_in_memory=max_rows_in_memory)
 175 
 176   def _initializer(shape, dtype=dtypes.float32, partition_info=None):
 177     del partition_info  # Unused by this override.
 178     return initializer_fn(shape, dtype, partition_info=primary_partition_info)
 179 
 180   return _initializer
