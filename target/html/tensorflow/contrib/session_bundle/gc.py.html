  16 r"""System for specifying garbage collection (GC) of path based data.
  17 
  18 This framework allows for GC of data specified by path names, for example files
  19 on disk.  gc.Path objects each represent a single item stored at a path and may
  20 be a base directory,
  21   /tmp/exports/0/...
  22   /tmp/exports/1/...
  23   ...
  24 or a fully qualified file,
  25   /tmp/train-1.ckpt
  26   /tmp/train-2.ckpt
  27   ...
  28 
  29 A gc filter function takes and returns a list of gc.Path items.  Filter
  30 functions are responsible for selecting Path items for preservation or deletion.
  31 Note that functions should always return a sorted list.
  32 
  33 For example,
  34   base_dir = "/tmp"
  35   # create the directories
  36   for e in xrange(10):
  37     os.mkdir("%s/%d" % (base_dir, e), 0o755)
  38 
  39   # create a simple parser that pulls the export_version from the directory
  40   def parser(path):
  41     match = re.match("^" + base_dir + "/(\\d+)$", path.path)
  42     if not match:
  43       return None
  44     return path._replace(export_version=int(match.group(1)))
  45 
  46   path_list = gc.get_paths("/tmp", parser)  # contains all ten Paths
  47 
  48   every_fifth = gc.mod_export_version(5)
  49   print(every_fifth(path_list))  # shows ["/tmp/0", "/tmp/5"]
  50 
  51   largest_three = gc.largest_export_versions(3)
  52   print(largest_three(all_paths))  # shows ["/tmp/7", "/tmp/8", "/tmp/9"]
  53 
  54   both = gc.union(every_fifth, largest_three)
  55   print(both(all_paths))  # shows ["/tmp/0", "/tmp/5",
  56                           #        "/tmp/7", "/tmp/8", "/tmp/9"]
  57   # delete everything not in 'both'
  58   to_delete = gc.negation(both)
  59   for p in to_delete(all_paths):
  60     gfile.DeleteRecursively(p.path)  # deletes:  "/tmp/1", "/tmp/2",
  61                                      # "/tmp/3", "/tmp/4", "/tmp/6",
  62 """

  79 @deprecated('2017-06-30',
  80             'No longer supported. Switch to SavedModel immediately.')
  81 def largest_export_versions(n):
  82   """Creates a filter that keeps the largest n export versions.
  83 
  84   Args:
  85     n: number of versions to keep.
  86 
  87   Returns:
  88     A filter function that keeps the n largest paths.
  89   """
  90   def keep(paths):
  91     heap = []
  92     for idx, path in enumerate(paths):
  93       if path.export_version is not None:
  94         heapq.heappush(heap, (path.export_version, idx))
  95     keepers = [paths[i] for _, i in heapq.nlargest(n, heap)]
  96     return sorted(keepers)
  97 
  98   return keep

 101 @deprecated('2017-06-30',
 102             'No longer supported. Switch to SavedModel immediately.')
 103 def one_of_every_n_export_versions(n):
 104   r"""Creates a filter that keeps one of every n export versions.
 105 
 106   Args:
 107     n: interval size.
 108 
 109   Returns:
 110     A filter function that keeps exactly one path from each interval
 111     [0, n], (n, 2n], (2n, 3n], etc...  If more than one path exists in an
 112     interval the largest is kept.
 113   """
 114   def keep(paths):
 115     keeper_map = {}  # map from interval to largest path seen in that interval
 116     for p in paths:
 117       if p.export_version is None:
 118         # Skip missing export_versions.
 119         continue
 120       # Find the interval (with a special case to map export_version = 0 to
 121       # interval 0.
 122       interval = math.floor(
 123           (p.export_version - 1) / n) if p.export_version else 0
 124       existing = keeper_map.get(interval, None)
 125       if (not existing) or (existing.export_version < p.export_version):
 126         keeper_map[interval] = p
 127     return sorted(keeper_map.values())
 128 
 129   return keep

 132 @deprecated('2017-06-30',
 133             'No longer supported. Switch to SavedModel immediately.')
 134 def mod_export_version(n):
 135   """Creates a filter that keeps every export that is a multiple of n.
 136 
 137   Args:
 138     n: step size.
 139 
 140   Returns:
 141     A filter function that keeps paths where export_version % n == 0.
 142   """
 143   def keep(paths):
 144     keepers = []
 145     for p in paths:
 146       if p.export_version % n == 0:
 147         keepers.append(p)
 148     return sorted(keepers)
 149   return keep

 152 @deprecated('2017-06-30',
 153             'No longer supported. Switch to SavedModel immediately.')
 154 def union(lf, rf):
 155   """Creates a filter that keeps the union of two filters.
 156 
 157   Args:
 158     lf: first filter
 159     rf: second filter
 160 
 161   Returns:
 162     A filter function that keeps the n largest paths.
 163   """
 164   def keep(paths):
 165     l = set(lf(paths))
 166     r = set(rf(paths))
 167     return sorted(list(l|r))
 168   return keep

 171 @deprecated('2017-06-30',
 172             'No longer supported. Switch to SavedModel immediately.')
 173 def negation(f):
 174   """Negate a filter.
 175 
 176   Args:
 177     f: filter function to invert
 178 
 179   Returns:
 180     A filter function that returns the negation of f.
 181   """
 182   def keep(paths):
 183     l = set(paths)
 184     r = set(f(paths))
 185     return sorted(list(l-r))
 186   return keep

 189 @deprecated('2017-06-30',
 190             'No longer supported. Switch to SavedModel immediately.')
 191 def get_paths(base_dir, parser):
 192   """Gets a list of Paths in a given directory.
 193 
 194   Args:
 195     base_dir: directory.
 196     parser: a function which gets the raw Path and can augment it with
 197       information such as the export_version, or ignore the path by returning
 198       None.  An example parser may extract the export version from a path
 199       such as "/tmp/exports/100" an another may extract from a full file
 200       name such as "/tmp/checkpoint-99.out".
 201 
 202   Returns:
 203     A list of Paths contained in the base directory with the parsing function
 204     applied.
 205     By default the following fields are populated,
 206       - Path.path
 207     The parsing function is responsible for populating,
 208       - Path.export_version
 209   """
 210   raw_paths = gfile.ListDirectory(base_dir)
 211   paths = []
 212   for r in raw_paths:
 213     p = parser(Path(os.path.join(base_dir, r), None))
 214     if p:
 215       paths.append(p)
 216   return sorted(paths)
