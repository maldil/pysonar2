  36 def _add_input_to_signature_def(tensor_name, map_key, signature_def):
  37   """Add input tensor to signature_def.
  38 
  39   Args:
  40     tensor_name: string name of tensor to add to signature_def inputs
  41     map_key: string key to key into signature_def inputs map
  42     signature_def: object of type meta_graph_pb2.SignatureDef()
  43 
  44   Sideffect:
  45     adds a TensorInfo with tensor_name to signature_def inputs map keyed with
  46     map_key
  47   """
  48   tensor_info = meta_graph_pb2.TensorInfo(name=tensor_name)
  49   signature_def.inputs[map_key].CopyFrom(tensor_info)
  50

  52 def _add_output_to_signature_def(tensor_name, map_key, signature_def):
  53   """Add output tensor to signature_def.
  54 
  55   Args:
  56     tensor_name: string name of tensor to add to signature_def outputs
  57     map_key: string key to key into signature_def outputs map
  58     signature_def: object of type meta_graph_pb2.SignatureDef()
  59 
  60   Sideffect:
  61     adds a TensorInfo with tensor_name to signature_def outputs map keyed with
  62     map_key
  63   """
  64 
  65   tensor_info = meta_graph_pb2.TensorInfo(name=tensor_name)
  66   signature_def.outputs[map_key].CopyFrom(tensor_info)

  69 def _convert_default_signature_to_signature_def(signatures):
  70   """Convert default signature to object of type SignatureDef.
  71 
  72   Args:
  73     signatures: object of type manifest_pb2.Signatures()
  74 
  75   Returns:
  76     object of type SignatureDef which contains a converted version of default
  77     signature from input signatures object
  78 
  79     Returns None if signature is of generic type because it cannot be converted
  80     to SignatureDef.
  81 
  82   """
  83   default_signature = signatures.default_signature
  84   signature_def = meta_graph_pb2.SignatureDef()
  85   if (default_signature.WhichOneof("type") ==
  86       legacy_constants.REGRESSION_SIGNATURE):
  87     regression_signature = default_signature.regression_signature
  88     signature_def.method_name = signature_constants.REGRESS_METHOD_NAME
  89     _add_input_to_signature_def(regression_signature.input.tensor_name,
  90                                 signature_constants.REGRESS_INPUTS,
  91                                 signature_def)
  92     _add_output_to_signature_def(regression_signature.output.tensor_name,
  93                                  signature_constants.REGRESS_OUTPUTS,
  94                                  signature_def)
  95   elif (default_signature.WhichOneof("type") ==
  96         legacy_constants.CLASSIFICATION_SIGNATURE):
  97     classification_signature = default_signature.classification_signature
  98     signature_def.method_name = signature_constants.CLASSIFY_METHOD_NAME
  99     _add_input_to_signature_def(classification_signature.input.tensor_name,
 100                                 signature_constants.CLASSIFY_INPUTS,
 101                                 signature_def)
 102     _add_output_to_signature_def(classification_signature.classes.tensor_name,
 103                                  signature_constants.CLASSIFY_OUTPUT_CLASSES,
 104                                  signature_def)
 105     _add_output_to_signature_def(classification_signature.scores.tensor_name,
 106                                  signature_constants.CLASSIFY_OUTPUT_SCORES,
 107                                  signature_def)
 108   else:
 109     logging.error("Only classification and regression default signatures "
 110                   "are supported for up-conversion. %s is not "
 111                   "supported" % default_signature.WhichOneof("type"))
 112     return None
 113   return signature_def

 116 def _convert_named_signatures_to_signature_def(signatures):
 117   """Convert named signatures to object of type SignatureDef.
 118 
 119   Args:
 120     signatures: object of type manifest_pb2.Signatures()
 121 
 122   Returns:
 123     object of type SignatureDef which contains a converted version of named
 124     signatures from input signatures object
 125 
 126   Raises:
 127     RuntimeError: if input and output named signatures are not of type
 128     GenericSignature
 129   """
 130   signature_def = meta_graph_pb2.SignatureDef()
 131   input_signature = signatures.named_signatures[
 132       signature_constants.PREDICT_INPUTS]
 133   output_signature = signatures.named_signatures[
 134       signature_constants.PREDICT_OUTPUTS]
 135   # TODO(pdudnik): what if there are other signatures? Mimic cr/140900781 once
 136   # it is submitted.
 137   if (input_signature.WhichOneof("type") != legacy_constants.GENERIC_SIGNATURE
 138       or output_signature.WhichOneof("type") !=
 139       legacy_constants.GENERIC_SIGNATURE):
 140     raise RuntimeError("Named input and output signatures can only be "
 141                        "up-converted if they are generic signature. "
 142                        "Input signature type is %s, output signature type is "
 143                        "%s" % (input_signature.WhichOneof("type"),
 144                                output_signature.WhichOneof("type")))
 145 
 146   signature_def.method_name = signature_constants.PREDICT_METHOD_NAME
 147   for key, val in input_signature.generic_signature.map.items():
 148     _add_input_to_signature_def(val.tensor_name, key, signature_def)
 149   for key, val in output_signature.generic_signature.map.items():
 150     _add_output_to_signature_def(val.tensor_name, key, signature_def)
 151   return signature_def

 154 def _convert_signatures_to_signature_defs(metagraph_def):
 155   """Produce default and named upconverted SignatureDef objects from Signatures.
 156 
 157   Args:
 158     metagraph_def: object of type meta_graph_pb2.MetaGraphDef containing legacy
 159     format Session Bundle signatures
 160 
 161   Returns:
 162     default_signature_def: object of type SignatureDef which contains an
 163         upconverted version of default signatures in metagraph_def
 164     named_signature_def: object of type SignatureDef which contains an
 165         upconverted version of named signatures in metagraph_def
 166   """
 167 
 168   collection_def = metagraph_def.collection_def
 169   signatures_proto = manifest_pb2.Signatures()
 170   signatures = collection_def[legacy_constants.SIGNATURES_KEY].any_list.value[0]
 171   signatures.Unpack(signatures_proto)
 172 
 173   default_signature_def = None
 174   named_signature_def = None
 175   if signatures_proto.HasField("default_signature"):
 176     default_signature_def = _convert_default_signature_to_signature_def(
 177         signatures_proto)
 178   if len(signatures_proto.named_signatures) > 1:
 179     named_signature_def = _convert_named_signatures_to_signature_def(
 180         signatures_proto)
 181   return default_signature_def, named_signature_def

 184 def _load_saved_model_from_session_bundle_path(export_dir, target, config):
 185   """Load legacy TF Exporter/SessionBundle checkpoint.
 186 
 187   Args:
 188     export_dir: the directory that contains files exported by exporter.
 189     target: The execution engine to connect to. See target in tf.Session()
 190     config: A ConfigProto proto with configuration options. See config in
 191     tf.Session()
 192 
 193   Returns:
 194     session: a tensorflow session created from the variable files.
 195     metagraph_def: The `MetaGraphDef` protocol buffer loaded in the provided
 196     session. This can be used to further extract signature-defs,
 197     collection-defs, etc.
 198     This model is up-converted to SavedModel format. Specifically, metagraph_def
 199     SignatureDef field is populated with Signatures converted from legacy
 200     signatures contained within CollectionDef
 201 
 202   Raises:
 203     RuntimeError: If metagraph already contains signature_def and cannot be
 204     up-converted.
 205   """
 206 
 207   meta_graph_filename = os.path.join(export_dir,
 208                                      legacy_constants.META_GRAPH_DEF_FILENAME)
 209 
 210   metagraph_def = meta_graph.read_meta_graph_file(meta_graph_filename)
 211   if metagraph_def.signature_def:
 212     raise RuntimeError("Legacy graph contains signature def, unable to "
 213                        "up-convert.")
 214 
 215   # Add SignatureDef to metagraph.
 216   default_signature_def, named_signature_def = (
 217       _convert_signatures_to_signature_defs(metagraph_def))
 218   if default_signature_def:
 219     metagraph_def.signature_def[
 220         signature_constants.DEFAULT_SERVING_SIGNATURE_DEF_KEY].CopyFrom(
 221             default_signature_def)
 222   if named_signature_def:
 223     signature_def_key = signature_constants.DEFAULT_SERVING_SIGNATURE_DEF_KEY
 224     if default_signature_def:
 225       signature_def_key += "_from_named"
 226     metagraph_def.signature_def[signature_def_key].CopyFrom(named_signature_def)
 227 
 228   # We cannot just output session we loaded with older metagraph_def and
 229   # up-converted metagraph definition because Session has an internal object of
 230   # type Graph which is populated from meta_graph_def. If we do not create
 231   # session with our new meta_graph_def, then Graph will be out of sync with
 232   # meta_graph_def.
 233   sess, metagraph_def = session_bundle.load_session_bundle_from_path(
 234       export_dir, target, config, meta_graph_def=metagraph_def)
 235   return sess, metagraph_def

 238 def load_session_bundle_or_saved_model_bundle_from_path(export_dir,
 239                                                         tags=None,
 240                                                         target="",
 241                                                         config=None):
 242   """Load session bundle from the given path.
 243 
 244   The function reads input from the export_dir, constructs the graph data to the
 245   default graph and restores the parameters for the session created.
 246 
 247   Args:
 248     export_dir: the directory that contains files exported by exporter.
 249     tags: Set of string tags to identify the required MetaGraphDef when model is
 250           saved as SavedModel. These should correspond to the tags used when
 251           saving the variables using the SavedModel `save()` API.
 252     target: The execution engine to connect to. See target in tf.Session()
 253     config: A ConfigProto proto with configuration options. See config in
 254             tf.Session()
 255 
 256   Returns:
 257     session: a tensorflow session created from the variable files.
 258     meta_graph: a meta graph proto saved in the exporter directory.
 259 
 260   Raises:
 261     RuntimeError: if the required files are missing or contain unrecognizable
 262     fields, i.e. the exported model is invalid.
 263   """
 264   metagraph_def = None
 265   sess = None
 266   if loader.maybe_saved_model_directory(export_dir):
 267     sess = session.Session(target, graph=None, config=config)
 268     metagraph_def = loader.load(sess, tags, export_dir)
 269   elif session_bundle.maybe_session_bundle_dir(export_dir):
 270     sess, metagraph_def = _load_saved_model_from_session_bundle_path(export_dir,
 271                                                                      target,
 272                                                                      config)
 273   else:
 274     raise RuntimeError("SessionBundle or SavedModelBundle not found at "
 275                        "specified export location: %s" % export_dir)
 276 
 277   return sess, metagraph_def
