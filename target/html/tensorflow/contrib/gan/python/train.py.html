  15 """The TFGAN project provides a lightweight GAN training/testing framework.
  16 
  17 This file contains the core helper functions to create and train a GAN model.
  18 See the README or examples in `tensorflow_models` for details on how to use.
  19 
  20 TFGAN training occurs in four steps:
  21 1) Create a model
  22 2) Add a loss
  23 3) Create train ops
  24 4) Run the train ops
  25 
  26 The functions in this file are organized around these four steps. Each function
  27 corresponds to one of the steps.
  28 """

  73 def gan_model(
  74     # Lambdas defining models.
  75     generator_fn,
  76     discriminator_fn,
  77     # Real data and conditioning.
  78     real_data,
  79     generator_inputs,
  80     # Optional scopes.
  81     generator_scope='Generator',
  82     discriminator_scope='Discriminator',
  83     # Options.
  84     check_shapes=True):
  85   """Returns GAN model outputs and variables.
  86 
  87   Args:
  88     generator_fn: A python lambda that takes `generator_inputs` as inputs and
  89       returns the outputs of the GAN generator.
  90     discriminator_fn: A python lambda that takes `real_data`/`generated data`
  91       and `generator_inputs`. Outputs a Tensor in the range [-inf, inf].
  92     real_data: A Tensor representing the real data.
  93     generator_inputs: A Tensor or list of Tensors to the generator. In the
  94       vanilla GAN case, this might be a single noise Tensor. In the conditional
  95       GAN case, this might be the generator's conditioning.
  96     generator_scope: Optional generator variable scope. Useful if you want to
  97       reuse a subgraph that has already been created.
  98     discriminator_scope: Optional discriminator variable scope. Useful if you
  99       want to reuse a subgraph that has already been created.
 100     check_shapes: If `True`, check that generator produces Tensors that are the
 101       same shape as real data. Otherwise, skip this check.
 102 
 103   Returns:
 104     A GANModel namedtuple.
 105 
 106   Raises:
 107     ValueError: If the generator outputs a Tensor that isn't the same shape as
 108       `real_data`.
 109   """
 110   # Create models
 111   with variable_scope.variable_scope(generator_scope) as gen_scope:
 112     generator_inputs = _convert_tensor_or_l_or_d(generator_inputs)
 113     generated_data = generator_fn(generator_inputs)
 114   with variable_scope.variable_scope(discriminator_scope) as dis_scope:
 115     discriminator_gen_outputs = discriminator_fn(generated_data,
 116                                                  generator_inputs)
 117   with variable_scope.variable_scope(dis_scope, reuse=True):
 118     real_data = ops.convert_to_tensor(real_data)
 119     discriminator_real_outputs = discriminator_fn(real_data, generator_inputs)
 120 
 121   if check_shapes:
 122     if not generated_data.shape.is_compatible_with(real_data.shape):
 123       raise ValueError(
 124           'Generator output shape (%s) must be the same shape as real data '
 125           '(%s).' % (generated_data.shape, real_data.shape))
 126 
 127   # Get model-specific variables.
 128   generator_variables = variables_lib.get_trainable_variables(gen_scope)
 129   discriminator_variables = variables_lib.get_trainable_variables(dis_scope)
 130 
 131   return namedtuples.GANModel(
 132       generator_inputs, generated_data, generator_variables, gen_scope,
 133       generator_fn, real_data, discriminator_real_outputs,
 134       discriminator_gen_outputs, discriminator_variables, dis_scope,
 135       discriminator_fn)

 138 def infogan_model(
 139     # Lambdas defining models.
 140     generator_fn,
 141     discriminator_fn,
 142     # Real data and conditioning.
 143     real_data,
 144     unstructured_generator_inputs,
 145     structured_generator_inputs,
 146     # Optional scopes.
 147     generator_scope='Generator',
 148     discriminator_scope='Discriminator'):
 149   """Returns an InfoGAN model outputs and variables.
 150 
 151   See https://arxiv.org/abs/1606.03657 for more details.
 152 
 153   Args:
 154     generator_fn: A python lambda that takes a list of Tensors as inputs and
 155       returns the outputs of the GAN generator.
 156     discriminator_fn: A python lambda that takes `real_data`/`generated data`
 157       and `generator_inputs`. Outputs a 2-tuple of (logits, distribution_list).
 158       `logits` are in the range [-inf, inf], and `distribution_list` is a list
 159       of Tensorflow distributions representing the predicted noise distribution
 160       of the ith structure noise.
 161     real_data: A Tensor representing the real data.
 162     unstructured_generator_inputs: A list of Tensors to the generator.
 163       These tensors represent the unstructured noise or conditioning.
 164     structured_generator_inputs: A list of Tensors to the generator.
 165       These tensors must have high mutual information with the recognizer.
 166     generator_scope: Optional generator variable scope. Useful if you want to
 167       reuse a subgraph that has already been created.
 168     discriminator_scope: Optional discriminator variable scope. Useful if you
 169       want to reuse a subgraph that has already been created.
 170 
 171   Returns:
 172     An InfoGANModel namedtuple.
 173 
 174   Raises:
 175     ValueError: If the generator outputs a Tensor that isn't the same shape as
 176       `real_data`.
 177     ValueError: If the discriminator output is malformed.
 178   """
 179   # Create models
 180   with variable_scope.variable_scope(generator_scope) as gen_scope:
 181     unstructured_generator_inputs = _convert_tensor_or_l_or_d(
 182         unstructured_generator_inputs)
 183     structured_generator_inputs = _convert_tensor_or_l_or_d(
 184         structured_generator_inputs)
 185     generator_inputs = (
 186         unstructured_generator_inputs + structured_generator_inputs)
 187     generated_data = generator_fn(generator_inputs)
 188   with variable_scope.variable_scope(discriminator_scope) as disc_scope:
 189     dis_gen_outputs, predicted_distributions = discriminator_fn(
 190         generated_data, generator_inputs)
 191   _validate_distributions(predicted_distributions, structured_generator_inputs)
 192   with variable_scope.variable_scope(disc_scope, reuse=True):
 193     real_data = ops.convert_to_tensor(real_data)
 194     dis_real_outputs, _ = discriminator_fn(real_data, generator_inputs)
 195 
 196   if not generated_data.get_shape().is_compatible_with(real_data.get_shape()):
 197     raise ValueError(
 198         'Generator output shape (%s) must be the same shape as real data '
 199         '(%s).' % (generated_data.get_shape(), real_data.get_shape()))
 200 
 201   # Get model-specific variables.
 202   generator_variables = variables_lib.get_trainable_variables(gen_scope)
 203   discriminator_variables = variables_lib.get_trainable_variables(disc_scope)
 204 
 205   return namedtuples.InfoGANModel(
 206       generator_inputs,
 207       generated_data,
 208       generator_variables,
 209       gen_scope,
 210       generator_fn,
 211       real_data,
 212       dis_real_outputs,
 213       dis_gen_outputs,
 214       discriminator_variables,
 215       disc_scope,
 216       lambda x, y: discriminator_fn(x, y)[0],  # conform to non-InfoGAN API
 217       structured_generator_inputs,
 218       predicted_distributions,
 219       discriminator_fn)

 222 def acgan_model(
 223     # Lambdas defining models.
 224     generator_fn,
 225     discriminator_fn,
 226     # Real data and conditioning.
 227     real_data,
 228     generator_inputs,
 229     one_hot_labels,
 230     # Optional scopes.
 231     generator_scope='Generator',
 232     discriminator_scope='Discriminator',
 233     # Options.
 234     check_shapes=True):
 235   """Returns an ACGANModel contains all the pieces needed for ACGAN training.
 236 
 237   The `acgan_model` is the same as the `gan_model` with the only difference
 238   being that the discriminator additionally outputs logits to classify the input
 239   (real or generated).
 240   Therefore, an explicit field holding one_hot_labels is necessary, as well as a
 241   discriminator_fn that outputs a 2-tuple holding the logits for real/fake and
 242   classification.
 243 
 244   See https://arxiv.org/abs/1610.09585 for more details.
 245 
 246   Args:
 247     generator_fn: A python lambda that takes `generator_inputs` as inputs and
 248       returns the outputs of the GAN generator.
 249     discriminator_fn: A python lambda that takes `real_data`/`generated data`
 250       and `generator_inputs`. Outputs a tuple consisting of two Tensors:
 251         (1) real/fake logits in the range [-inf, inf]
 252         (2) classification logits in the range [-inf, inf]
 253     real_data: A Tensor representing the real data.
 254     generator_inputs: A Tensor or list of Tensors to the generator. In the
 255       vanilla GAN case, this might be a single noise Tensor. In the conditional
 256       GAN case, this might be the generator's conditioning.
 257     one_hot_labels: A Tensor holding one-hot-labels for the batch. Needed by
 258       acgan_loss.
 259     generator_scope: Optional generator variable scope. Useful if you want to
 260       reuse a subgraph that has already been created.
 261     discriminator_scope: Optional discriminator variable scope. Useful if you
 262       want to reuse a subgraph that has already been created.
 263     check_shapes: If `True`, check that generator produces Tensors that are the
 264       same shape as real data. Otherwise, skip this check.
 265 
 266   Returns:
 267     A ACGANModel namedtuple.
 268 
 269   Raises:
 270     ValueError: If the generator outputs a Tensor that isn't the same shape as
 271       `real_data`.
 272     TypeError: If the discriminator does not output a tuple consisting of
 273     (discrimination logits, classification logits).
 274   """
 275   # Create models
 276   with variable_scope.variable_scope(generator_scope) as gen_scope:
 277     generator_inputs = _convert_tensor_or_l_or_d(generator_inputs)
 278     generated_data = generator_fn(generator_inputs)
 279   with variable_scope.variable_scope(discriminator_scope) as dis_scope:
 280     with ops.name_scope(dis_scope.name + '/generated/'):
 281       (discriminator_gen_outputs, discriminator_gen_classification_logits
 282       ) = _validate_acgan_discriminator_outputs(
 283           discriminator_fn(generated_data, generator_inputs))
 284   with variable_scope.variable_scope(dis_scope, reuse=True):
 285     with ops.name_scope(dis_scope.name + '/real/'):
 286       real_data = ops.convert_to_tensor(real_data)
 287       (discriminator_real_outputs, discriminator_real_classification_logits
 288       ) = _validate_acgan_discriminator_outputs(
 289           discriminator_fn(real_data, generator_inputs))
 290   if check_shapes:
 291     if not generated_data.shape.is_compatible_with(real_data.shape):
 292       raise ValueError(
 293           'Generator output shape (%s) must be the same shape as real data '
 294           '(%s).' % (generated_data.shape, real_data.shape))
 295 
 296   # Get model-specific variables.
 297   generator_variables = variables_lib.get_trainable_variables(gen_scope)
 298   discriminator_variables = variables_lib.get_trainable_variables(dis_scope)
 299 
 300   return namedtuples.ACGANModel(
 301       generator_inputs, generated_data, generator_variables, gen_scope,
 302       generator_fn, real_data, discriminator_real_outputs,
 303       discriminator_gen_outputs, discriminator_variables, dis_scope,
 304       discriminator_fn, one_hot_labels,
 305       discriminator_real_classification_logits,
 306       discriminator_gen_classification_logits)

 309 def cyclegan_model(
 310     # Lambdas defining models.
 311     generator_fn,
 312     discriminator_fn,
 313     # data X and Y.
 314     data_x,
 315     data_y,
 316     # Optional scopes.
 317     generator_scope='Generator',
 318     discriminator_scope='Discriminator',
 319     model_x2y_scope='ModelX2Y',
 320     model_y2x_scope='ModelY2X',
 321     # Options.
 322     check_shapes=True):
 323   """Returns a CycleGAN model outputs and variables.
 324 
 325   See https://arxiv.org/abs/1703.10593 for more details.
 326 
 327   Args:
 328     generator_fn: A python lambda that takes `data_x` or `data_y` as inputs and
 329       returns the outputs of the GAN generator.
 330     discriminator_fn: A python lambda that takes `real_data`/`generated data`
 331       and `generator_inputs`. Outputs a Tensor in the range [-inf, inf].
 332     data_x: A `Tensor` of dataset X. Must be the same shape as `data_y`.
 333     data_y: A `Tensor` of dataset Y. Must be the same shape as `data_x`.
 334     generator_scope: Optional generator variable scope. Useful if you want to
 335       reuse a subgraph that has already been created. Defaults to 'Generator'.
 336     discriminator_scope: Optional discriminator variable scope. Useful if you
 337       want to reuse a subgraph that has already been created. Defaults to
 338       'Discriminator'.
 339     model_x2y_scope: Optional variable scope for model x2y variables. Defaults
 340       to 'ModelX2Y'.
 341     model_y2x_scope: Optional variable scope for model y2x variables. Defaults
 342       to 'ModelY2X'.
 343     check_shapes: If `True`, check that generator produces Tensors that are the
 344       same shape as `data_x` (`data_y`). Otherwise, skip this check.
 345 
 346   Returns:
 347     A `CycleGANModel` namedtuple.
 348 
 349   Raises:
 350     ValueError: If `check_shapes` is True and `data_x` or the generator output
 351       does not have the same shape as `data_y`.
 352   """
 353 
 354   # Create models.
 355   def _define_partial_model(input_data, output_data):
 356     return gan_model(
 357         generator_fn=generator_fn,
 358         discriminator_fn=discriminator_fn,
 359         real_data=output_data,
 360         generator_inputs=input_data,
 361         generator_scope=generator_scope,
 362         discriminator_scope=discriminator_scope,
 363         check_shapes=check_shapes)
 364 
 365   with variable_scope.variable_scope(model_x2y_scope):
 366     model_x2y = _define_partial_model(data_x, data_y)
 367   with variable_scope.variable_scope(model_y2x_scope):
 368     model_y2x = _define_partial_model(data_y, data_x)
 369 
 370   with variable_scope.variable_scope(model_y2x.generator_scope, reuse=True):
 371     reconstructed_x = model_y2x.generator_fn(model_x2y.generated_data)
 372   with variable_scope.variable_scope(model_x2y.generator_scope, reuse=True):
 373     reconstructed_y = model_x2y.generator_fn(model_y2x.generated_data)
 374 
 375   return namedtuples.CycleGANModel(model_x2y, model_y2x, reconstructed_x,
 376                                    reconstructed_y)

 379 def stargan_model(generator_fn,
 380                   discriminator_fn,
 381                   input_data,
 382                   input_data_domain_label,
 383                   generator_scope='Generator',
 384                   discriminator_scope='Discriminator'):
 385   """Returns a StarGAN model outputs and variables.
 386 
 387   See https://arxiv.org/abs/1711.09020 for more details.
 388 
 389   Args:
 390     generator_fn: A python lambda that takes `inputs` and `targets` as inputs
 391       and returns 'generated_data' as the transformed version of `input` based
 392       on the `target`. `input` has shape (n, h, w, c), `targets` has shape (n,
 393       num_domains), and `generated_data` has the same shape as `input`.
 394     discriminator_fn: A python lambda that takes `inputs` and `num_domains` as
 395       inputs and returns a tuple (`source_prediction`, `domain_prediction`).
 396       `source_prediction` represents the source(real/generated) prediction by
 397       the discriminator, and `domain_prediction` represents the domain
 398       prediction/classification by the discriminator. `source_prediction` has
 399       shape (n) and `domain_prediction` has shape (n, num_domains).
 400     input_data: Tensor or a list of tensor of shape (n, h, w, c) representing
 401       the real input images.
 402     input_data_domain_label: Tensor or a list of tensor of shape (batch_size,
 403       num_domains) representing the domain label associated with the real
 404       images.
 405     generator_scope: Optional generator variable scope. Useful if you want to
 406       reuse a subgraph that has already been created.
 407     discriminator_scope: Optional discriminator variable scope. Useful if you
 408       want to reuse a subgraph that has already been created.
 409 
 410   Returns:
 411     StarGANModel nametuple return the tensor that are needed to compute the
 412     loss.
 413 
 414   Raises:
 415     ValueError: If the shape of `input_data_domain_label` is not rank 2 or fully
 416     defined in every dimensions.
 417   """
 418 
 419   # Convert to tensor.
 420   input_data = _convert_tensor_or_l_or_d(input_data)
 421   input_data_domain_label = _convert_tensor_or_l_or_d(input_data_domain_label)
 422 
 423   # Convert list of tensor to a single tensor if applicable.
 424   if isinstance(input_data, (list, tuple)):
 425     input_data = array_ops.concat(
 426         [ops.convert_to_tensor(x) for x in input_data], 0)
 427   if isinstance(input_data_domain_label, (list, tuple)):
 428     input_data_domain_label = array_ops.concat(
 429         [ops.convert_to_tensor(x) for x in input_data_domain_label], 0)
 430 
 431   # Get batch_size, num_domains from the labels.
 432   input_data_domain_label.shape.assert_has_rank(2)
 433   input_data_domain_label.shape.assert_is_fully_defined()
 434   batch_size, num_domains = input_data_domain_label.shape.as_list()
 435 
 436   # Transform input_data to random target domains.
 437   with variable_scope.variable_scope(generator_scope) as generator_scope:
 438     generated_data_domain_target = _generate_stargan_random_domain_target(
 439         batch_size, num_domains)
 440     generated_data = generator_fn(input_data, generated_data_domain_target)
 441 
 442   # Transform generated_data back to the original input_data domain.
 443   with variable_scope.variable_scope(generator_scope, reuse=True):
 444     reconstructed_data = generator_fn(generated_data, input_data_domain_label)
 445 
 446   # Predict source and domain for the generated_data using the discriminator.
 447   with variable_scope.variable_scope(
 448       discriminator_scope) as discriminator_scope:
 449     disc_gen_data_source_pred, disc_gen_data_domain_pred = discriminator_fn(
 450         generated_data, num_domains)
 451 
 452   # Predict source and domain for the input_data using the discriminator.
 453   with variable_scope.variable_scope(discriminator_scope, reuse=True):
 454     disc_input_data_source_pred, disc_input_data_domain_pred = discriminator_fn(
 455         input_data, num_domains)
 456 
 457   # Collect trainable variables from the neural networks.
 458   generator_variables = variables_lib.get_trainable_variables(generator_scope)
 459   discriminator_variables = variables_lib.get_trainable_variables(
 460       discriminator_scope)
 461 
 462   # Create the StarGANModel namedtuple.
 463   return namedtuples.StarGANModel(
 464       input_data=input_data,
 465       input_data_domain_label=input_data_domain_label,
 466       generated_data=generated_data,
 467       generated_data_domain_target=generated_data_domain_target,
 468       reconstructed_data=reconstructed_data,
 469       discriminator_input_data_source_predication=disc_input_data_source_pred,
 470       discriminator_generated_data_source_predication=disc_gen_data_source_pred,
 471       discriminator_input_data_domain_predication=disc_input_data_domain_pred,
 472       discriminator_generated_data_domain_predication=disc_gen_data_domain_pred,
 473       generator_variables=generator_variables,
 474       generator_scope=generator_scope,
 475       generator_fn=generator_fn,
 476       discriminator_variables=discriminator_variables,
 477       discriminator_scope=discriminator_scope,
 478       discriminator_fn=discriminator_fn)

 481 def _validate_aux_loss_weight(aux_loss_weight, name='aux_loss_weight'):
 482   if isinstance(aux_loss_weight, ops.Tensor):
 483     aux_loss_weight.shape.assert_is_compatible_with([])
 484     with ops.control_dependencies(
 485         [check_ops.assert_greater_equal(aux_loss_weight, 0.0)]):
 486       aux_loss_weight = array_ops.identity(aux_loss_weight)
 487   elif aux_loss_weight is not None and aux_loss_weight < 0:
 488     raise ValueError('`%s` must be greater than 0. Instead, was %s' %
 489                      (name, aux_loss_weight))
 490   return aux_loss_weight
 491

 493 def _use_aux_loss(aux_loss_weight):
 494   if aux_loss_weight is not None:
 495     if not isinstance(aux_loss_weight, ops.Tensor):
 496       return aux_loss_weight > 0
 497     else:
 498       return True
 499   else:
 500     return False

 503 def _tensor_pool_adjusted_model(model, tensor_pool_fn):
 504   """Adjusts model using `tensor_pool_fn`.
 505 
 506   Args:
 507     model: A GANModel tuple.
 508     tensor_pool_fn: A function that takes (generated_data, generator_inputs),
 509       stores them in an internal pool and returns a previously stored
 510       (generated_data, generator_inputs) with some probability. For example
 511       tfgan.features.tensor_pool.
 512 
 513   Returns:
 514     A new GANModel tuple where discriminator outputs are adjusted by taking
 515     pooled generator outputs as inputs. Returns the original model if
 516     `tensor_pool_fn` is None.
 517 
 518   Raises:
 519     ValueError: If tensor pool does not support the `model`.
 520   """
 521   if isinstance(model, namedtuples.GANModel):
 522     pooled_generator_inputs, pooled_generated_data = tensor_pool_fn(
 523         (model.generator_inputs, model.generated_data))
 524     with variable_scope.variable_scope(model.discriminator_scope, reuse=True):
 525       dis_gen_outputs = model.discriminator_fn(pooled_generated_data,
 526                                                pooled_generator_inputs)
 527     return model._replace(
 528         generator_inputs=pooled_generator_inputs,
 529         generated_data=pooled_generated_data,
 530         discriminator_gen_outputs=dis_gen_outputs)
 531   elif isinstance(model, namedtuples.ACGANModel):
 532     pooled_generator_inputs, pooled_generated_data = tensor_pool_fn(
 533         (model.generator_inputs, model.generated_data))
 534     with variable_scope.variable_scope(model.discriminator_scope, reuse=True):
 535       (pooled_discriminator_gen_outputs,
 536        pooled_discriminator_gen_classification_logits) = model.discriminator_fn(
 537            pooled_generated_data, pooled_generator_inputs)
 538     return model._replace(
 539         generator_inputs=pooled_generator_inputs,
 540         generated_data=pooled_generated_data,
 541         discriminator_gen_outputs=pooled_discriminator_gen_outputs,
 542         discriminator_gen_classification_logits=
 543         pooled_discriminator_gen_classification_logits)
 544   elif isinstance(model, namedtuples.InfoGANModel):
 545     pooled_generator_inputs, pooled_generated_data, pooled_structured_input = (
 546         tensor_pool_fn((model.generator_inputs, model.generated_data,
 547                         model.structured_generator_inputs)))
 548     with variable_scope.variable_scope(model.discriminator_scope, reuse=True):
 549       (pooled_discriminator_gen_outputs,
 550        pooled_predicted_distributions) = model.discriminator_and_aux_fn(
 551            pooled_generated_data, pooled_generator_inputs)
 552     return model._replace(
 553         generator_inputs=pooled_generator_inputs,
 554         generated_data=pooled_generated_data,
 555         structured_generator_inputs=pooled_structured_input,
 556         discriminator_gen_outputs=pooled_discriminator_gen_outputs,
 557         predicted_distributions=pooled_predicted_distributions)
 558   else:
 559     raise ValueError('Tensor pool does not support `model`: %s.' % type(model))

 562 def gan_loss(
 563     # GANModel.
 564     model,
 565     # Loss functions.
 566     generator_loss_fn=tfgan_losses.wasserstein_generator_loss,
 567     discriminator_loss_fn=tfgan_losses.wasserstein_discriminator_loss,
 568     # Auxiliary losses.
 569     gradient_penalty_weight=None,
 570     gradient_penalty_epsilon=1e-10,
 571     gradient_penalty_target=1.0,
 572     gradient_penalty_one_sided=False,
 573     mutual_information_penalty_weight=None,
 574     aux_cond_generator_weight=None,
 575     aux_cond_discriminator_weight=None,
 576     tensor_pool_fn=None,
 577     # Options.
 578     add_summaries=True):
 579   """Returns losses necessary to train generator and discriminator.
 580 
 581   Args:
 582     model: A GANModel tuple.
 583     generator_loss_fn: The loss function on the generator. Takes a GANModel
 584       tuple.
 585     discriminator_loss_fn: The loss function on the discriminator. Takes a
 586       GANModel tuple.
 587     gradient_penalty_weight: If not `None`, must be a non-negative Python number
 588       or Tensor indicating how much to weight the gradient penalty. See
 589       https://arxiv.org/pdf/1704.00028.pdf for more details.
 590     gradient_penalty_epsilon: If `gradient_penalty_weight` is not None, the
 591       small positive value used by the gradient penalty function for numerical
 592       stability. Note some applications will need to increase this value to
 593       avoid NaNs.
 594     gradient_penalty_target: If `gradient_penalty_weight` is not None, a Python
 595       number or `Tensor` indicating the target value of gradient norm. See the
 596       CIFAR10 section of https://arxiv.org/abs/1710.10196. Defaults to 1.0.
 597     gradient_penalty_one_sided: If `True`, penalty proposed in
 598       https://arxiv.org/abs/1709.08894 is used. Defaults to `False`.
 599     mutual_information_penalty_weight: If not `None`, must be a non-negative
 600       Python number or Tensor indicating how much to weight the mutual
 601       information penalty. See https://arxiv.org/abs/1606.03657 for more
 602       details.
 603     aux_cond_generator_weight: If not None: add a classification loss as in
 604       https://arxiv.org/abs/1610.09585
 605     aux_cond_discriminator_weight: If not None: add a classification loss as in
 606       https://arxiv.org/abs/1610.09585
 607     tensor_pool_fn: A function that takes (generated_data, generator_inputs),
 608       stores them in an internal pool and returns previous stored
 609       (generated_data, generator_inputs). For example
 610       `tf.gan.features.tensor_pool`. Defaults to None (not using tensor pool).
 611     add_summaries: Whether or not to add summaries for the losses.
 612 
 613   Returns:
 614     A GANLoss 2-tuple of (generator_loss, discriminator_loss). Includes
 615     regularization losses.
 616 
 617   Raises:
 618     ValueError: If any of the auxiliary loss weights is provided and negative.
 619     ValueError: If `mutual_information_penalty_weight` is provided, but the
 620       `model` isn't an `InfoGANModel`.
 621   """
 622   # Validate arguments.
 623   gradient_penalty_weight = _validate_aux_loss_weight(
 624       gradient_penalty_weight, 'gradient_penalty_weight')
 625   mutual_information_penalty_weight = _validate_aux_loss_weight(
 626       mutual_information_penalty_weight, 'infogan_weight')
 627   aux_cond_generator_weight = _validate_aux_loss_weight(
 628       aux_cond_generator_weight, 'aux_cond_generator_weight')
 629   aux_cond_discriminator_weight = _validate_aux_loss_weight(
 630       aux_cond_discriminator_weight, 'aux_cond_discriminator_weight')
 631 
 632   # Verify configuration for mutual information penalty
 633   if (_use_aux_loss(mutual_information_penalty_weight) and
 634       not isinstance(model, namedtuples.InfoGANModel)):
 635     raise ValueError(
 636         'When `mutual_information_penalty_weight` is provided, `model` must be '
 637         'an `InfoGANModel`. Instead, was %s.' % type(model))
 638 
 639   # Verify configuration for mutual auxiliary condition loss (ACGAN).
 640   if ((_use_aux_loss(aux_cond_generator_weight) or
 641        _use_aux_loss(aux_cond_discriminator_weight)) and
 642       not isinstance(model, namedtuples.ACGANModel)):
 643     raise ValueError(
 644         'When `aux_cond_generator_weight` or `aux_cond_discriminator_weight` '
 645         'is provided, `model` must be an `ACGANModel`. Instead, was %s.' %
 646         type(model))
 647 
 648   # Optionally create pooled model.
 649   pooled_model = (
 650       _tensor_pool_adjusted_model(model, tensor_pool_fn)
 651       if tensor_pool_fn else model)
 652 
 653   # Create standard losses.
 654   gen_loss = generator_loss_fn(model, add_summaries=add_summaries)
 655   dis_loss = discriminator_loss_fn(pooled_model, add_summaries=add_summaries)
 656 
 657   # Add optional extra losses.
 658   if _use_aux_loss(gradient_penalty_weight):
 659     gp_loss = tfgan_losses.wasserstein_gradient_penalty(
 660         pooled_model,
 661         epsilon=gradient_penalty_epsilon,
 662         target=gradient_penalty_target,
 663         one_sided=gradient_penalty_one_sided,
 664         add_summaries=add_summaries)
 665     dis_loss += gradient_penalty_weight * gp_loss
 666   if _use_aux_loss(mutual_information_penalty_weight):
 667     gen_info_loss = tfgan_losses.mutual_information_penalty(
 668         model, add_summaries=add_summaries)
 669     dis_info_loss = (
 670         gen_info_loss
 671         if tensor_pool_fn is None else tfgan_losses.mutual_information_penalty(
 672             pooled_model, add_summaries=add_summaries))
 673     gen_loss += mutual_information_penalty_weight * gen_info_loss
 674     dis_loss += mutual_information_penalty_weight * dis_info_loss
 675   if _use_aux_loss(aux_cond_generator_weight):
 676     ac_gen_loss = tfgan_losses.acgan_generator_loss(
 677         model, add_summaries=add_summaries)
 678     gen_loss += aux_cond_generator_weight * ac_gen_loss
 679   if _use_aux_loss(aux_cond_discriminator_weight):
 680     ac_disc_loss = tfgan_losses.acgan_discriminator_loss(
 681         pooled_model, add_summaries=add_summaries)
 682     dis_loss += aux_cond_discriminator_weight * ac_disc_loss
 683   # Gathers auxiliary losses.
 684   if model.generator_scope:
 685     gen_reg_loss = losses.get_regularization_loss(model.generator_scope.name)
 686   else:
 687     gen_reg_loss = 0
 688   if model.discriminator_scope:
 689     dis_reg_loss = losses.get_regularization_loss(
 690         model.discriminator_scope.name)
 691   else:
 692     dis_reg_loss = 0
 693 
 694   return namedtuples.GANLoss(gen_loss + gen_reg_loss, dis_loss + dis_reg_loss)

 697 def cyclegan_loss(
 698     model,
 699     # Loss functions.
 700     generator_loss_fn=tfgan_losses.least_squares_generator_loss,
 701     discriminator_loss_fn=tfgan_losses.least_squares_discriminator_loss,
 702     # Auxiliary losses.
 703     cycle_consistency_loss_fn=tfgan_losses.cycle_consistency_loss,
 704     cycle_consistency_loss_weight=10.0,
 705     # Options
 706     **kwargs):
 707   """Returns the losses for a `CycleGANModel`.
 708 
 709   See https://arxiv.org/abs/1703.10593 for more details.
 710 
 711   Args:
 712     model: A `CycleGANModel` namedtuple.
 713     generator_loss_fn: The loss function on the generator. Takes a `GANModel`
 714       named tuple.
 715     discriminator_loss_fn: The loss function on the discriminator. Takes a
 716       `GANModel` namedtuple.
 717     cycle_consistency_loss_fn: The cycle consistency loss function. Takes a
 718       `CycleGANModel` namedtuple.
 719     cycle_consistency_loss_weight: A non-negative Python number or a scalar
 720       `Tensor` indicating how much to weigh the cycle consistency loss.
 721     **kwargs: Keyword args to pass directly to `gan_loss` to construct the loss
 722       for each partial model of `model`.
 723 
 724   Returns:
 725     A `CycleGANLoss` namedtuple.
 726 
 727   Raises:
 728     ValueError: If `model` is not a `CycleGANModel` namedtuple.
 729   """
 730   # Sanity checks.
 731   if not isinstance(model, namedtuples.CycleGANModel):
 732     raise ValueError(
 733         '`model` must be a `CycleGANModel`. Instead, was %s.' % type(model))
 734 
 735   # Defines cycle consistency loss.
 736   cycle_consistency_loss = cycle_consistency_loss_fn(
 737       model, add_summaries=kwargs.get('add_summaries', True))
 738   cycle_consistency_loss_weight = _validate_aux_loss_weight(
 739       cycle_consistency_loss_weight, 'cycle_consistency_loss_weight')
 740   aux_loss = cycle_consistency_loss_weight * cycle_consistency_loss
 741 
 742   # Defines losses for each partial model.
 743   def _partial_loss(partial_model):
 744     partial_loss = gan_loss(
 745         partial_model,
 746         generator_loss_fn=generator_loss_fn,
 747         discriminator_loss_fn=discriminator_loss_fn,
 748         **kwargs)
 749     return partial_loss._replace(generator_loss=partial_loss.generator_loss +
 750                                  aux_loss)
 751 
 752   with ops.name_scope('cyclegan_loss_x2y'):
 753     loss_x2y = _partial_loss(model.model_x2y)
 754   with ops.name_scope('cyclegan_loss_y2x'):
 755     loss_y2x = _partial_loss(model.model_y2x)
 756 
 757   return namedtuples.CycleGANLoss(loss_x2y, loss_y2x)

 760 def stargan_loss(
 761     model,
 762     generator_loss_fn=tfgan_losses.stargan_generator_loss_wrapper(
 763         tfgan_losses_impl.wasserstein_generator_loss),
 764     discriminator_loss_fn=tfgan_losses.stargan_discriminator_loss_wrapper(
 765         tfgan_losses_impl.wasserstein_discriminator_loss),
 766     gradient_penalty_weight=10.0,
 767     gradient_penalty_epsilon=1e-10,
 768     gradient_penalty_target=1.0,
 769     gradient_penalty_one_sided=False,
 770     reconstruction_loss_fn=losses.absolute_difference,
 771     reconstruction_loss_weight=10.0,
 772     classification_loss_fn=losses.softmax_cross_entropy,
 773     classification_loss_weight=1.0,
 774     classification_one_hot=True,
 775     add_summaries=True):
 776   """StarGAN Loss.
 777 
 778   The four major part can be found here: http://screen/tMRMBAohDYG.
 779 
 780   Args:
 781     model: (StarGAN) Model output of the stargan_model() function call.
 782     generator_loss_fn: The loss function on the generator. Takes a
 783       `StarGANModel` named tuple.
 784     discriminator_loss_fn: The loss function on the discriminator. Takes a
 785       `StarGANModel` namedtuple.
 786     gradient_penalty_weight: (float) Gradient penalty weight. Default to 10 per
 787       the original paper https://arxiv.org/abs/1711.09020. Set to 0 or None to
 788       turn off gradient penalty.
 789     gradient_penalty_epsilon: (float) A small positive number added for
 790       numerical stability when computing the gradient norm.
 791     gradient_penalty_target: (float, or tf.float `Tensor`) The target value of
 792       gradient norm. Defaults to 1.0.
 793     gradient_penalty_one_sided: (bool) If `True`, penalty proposed in
 794       https://arxiv.org/abs/1709.08894 is used. Defaults to `False`.
 795     reconstruction_loss_fn: The reconstruction loss function. Default to L1-norm
 796       and the function must conform to the `tf.losses` API.
 797     reconstruction_loss_weight: Reconstruction loss weight. Default to 10.0.
 798     classification_loss_fn: The loss function on the discriminator's ability to
 799       classify domain of the input. Default to one-hot softmax cross entropy
 800       loss, and the function must conform to the `tf.losses` API.
 801     classification_loss_weight: (float) Classification loss weight. Default to
 802       1.0.
 803     classification_one_hot: (bool) If the label is one hot representation.
 804       Default to True. If False, classification classification_loss_fn need to
 805       be sigmoid cross entropy loss instead.
 806     add_summaries: (bool) Add the loss to the summary
 807 
 808   Returns:
 809     GANLoss namedtuple where we have generator loss and discriminator loss.
 810 
 811   Raises:
 812     ValueError: If input StarGANModel.input_data_domain_label does not have rank
 813     2, or dimension 2 is not defined.
 814   """
 815 
 816   def _classification_loss_helper(true_labels, predict_logits, scope_name):
 817     """Classification Loss Function Helper.
 818 
 819     Args:
 820       true_labels: Tensor of shape [batch_size, num_domains] representing the
 821         label where each row is an one-hot vector.
 822       predict_logits: Tensor of shape [batch_size, num_domains] representing the
 823         predicted label logit, which is UNSCALED output from the NN.
 824       scope_name: (string) Name scope of the loss component.
 825 
 826     Returns:
 827       Single scalar tensor representing the classification loss.
 828     """
 829 
 830     with ops.name_scope(scope_name, values=(true_labels, predict_logits)):
 831 
 832       loss = classification_loss_fn(
 833           onehot_labels=true_labels, logits=predict_logits)
 834 
 835       if not classification_one_hot:
 836         loss = math_ops.reduce_sum(loss, axis=1)
 837       loss = math_ops.reduce_mean(loss)
 838 
 839       if add_summaries:
 840         summary.scalar(scope_name, loss)
 841 
 842       return loss
 843 
 844   # Check input shape.
 845   model.input_data_domain_label.shape.assert_has_rank(2)
 846   model.input_data_domain_label.shape[1:].assert_is_fully_defined()
 847 
 848   # Adversarial Loss.
 849   generator_loss = generator_loss_fn(model, add_summaries=add_summaries)
 850   discriminator_loss = discriminator_loss_fn(model, add_summaries=add_summaries)
 851 
 852   # Gradient Penalty.
 853   if _use_aux_loss(gradient_penalty_weight):
 854     gradient_penalty_fn = tfgan_losses.stargan_gradient_penalty_wrapper(
 855         tfgan_losses_impl.wasserstein_gradient_penalty)
 856     discriminator_loss += gradient_penalty_fn(
 857         model,
 858         epsilon=gradient_penalty_epsilon,
 859         target=gradient_penalty_target,
 860         one_sided=gradient_penalty_one_sided,
 861         add_summaries=add_summaries) * gradient_penalty_weight
 862 
 863   # Reconstruction Loss.
 864   reconstruction_loss = reconstruction_loss_fn(model.input_data,
 865                                                model.reconstructed_data)
 866   generator_loss += reconstruction_loss * reconstruction_loss_weight
 867   if add_summaries:
 868     summary.scalar('reconstruction_loss', reconstruction_loss)
 869 
 870   # Classification Loss.
 871   generator_loss += _classification_loss_helper(
 872       true_labels=model.generated_data_domain_target,
 873       predict_logits=model.discriminator_generated_data_domain_predication,
 874       scope_name='generator_classification_loss') * classification_loss_weight
 875   discriminator_loss += _classification_loss_helper(
 876       true_labels=model.input_data_domain_label,
 877       predict_logits=model.discriminator_input_data_domain_predication,
 878       scope_name='discriminator_classification_loss'
 879   ) * classification_loss_weight
 880 
 881   return namedtuples.GANLoss(generator_loss, discriminator_loss)

 884 def _get_update_ops(kwargs, gen_scope, dis_scope, check_for_unused_ops=True):
 885   """Gets generator and discriminator update ops.
 886 
 887   Args:
 888     kwargs: A dictionary of kwargs to be passed to `create_train_op`.
 889       `update_ops` is removed, if present.
 890     gen_scope: A scope for the generator.
 891     dis_scope: A scope for the discriminator.
 892     check_for_unused_ops: A Python bool. If `True`, throw Exception if there are
 893       unused update ops.
 894 
 895   Returns:
 896     A 2-tuple of (generator update ops, discriminator train ops).
 897 
 898   Raises:
 899     ValueError: If there are update ops outside of the generator or
 900       discriminator scopes.
 901   """
 902   if 'update_ops' in kwargs:
 903     update_ops = set(kwargs['update_ops'])
 904     del kwargs['update_ops']
 905   else:
 906     update_ops = set(ops.get_collection(ops.GraphKeys.UPDATE_OPS))
 907 
 908   all_gen_ops = set(ops.get_collection(ops.GraphKeys.UPDATE_OPS, gen_scope))
 909   all_dis_ops = set(ops.get_collection(ops.GraphKeys.UPDATE_OPS, dis_scope))
 910 
 911   if check_for_unused_ops:
 912     unused_ops = update_ops - all_gen_ops - all_dis_ops
 913     if unused_ops:
 914       raise ValueError('There are unused update ops: %s' % unused_ops)
 915 
 916   gen_update_ops = list(all_gen_ops & update_ops)
 917   dis_update_ops = list(all_dis_ops & update_ops)
 918 
 919   return gen_update_ops, dis_update_ops

 922 def gan_train_ops(
 923     model,
 924     loss,
 925     generator_optimizer,
 926     discriminator_optimizer,
 927     check_for_unused_update_ops=True,
 928     # Optional args to pass directly to the `create_train_op`.
 929     **kwargs):
 930   """Returns GAN train ops.
 931 
 932   The highest-level call in TFGAN. It is composed of functions that can also
 933   be called, should a user require more control over some part of the GAN
 934   training process.
 935 
 936   Args:
 937     model: A GANModel.
 938     loss: A GANLoss.
 939     generator_optimizer: The optimizer for generator updates.
 940     discriminator_optimizer: The optimizer for the discriminator updates.
 941     check_for_unused_update_ops: If `True`, throws an exception if there are
 942       update ops outside of the generator or discriminator scopes.
 943     **kwargs: Keyword args to pass directly to
 944       `training.create_train_op` for both the generator and
 945       discriminator train op.
 946 
 947   Returns:
 948     A GANTrainOps tuple of (generator_train_op, discriminator_train_op) that can
 949     be used to train a generator/discriminator pair.
 950   """
 951   if isinstance(model, namedtuples.CycleGANModel):
 952     # Get and store all arguments other than model and loss from locals.
 953     # Contents of locals should not be modified, may not affect values. So make
 954     # a copy. https://docs.python.org/2/library/functions.html#locals.
 955     saved_params = dict(locals())
 956     saved_params.pop('model', None)
 957     saved_params.pop('loss', None)
 958     kwargs = saved_params.pop('kwargs', {})
 959     saved_params.update(kwargs)
 960     with ops.name_scope('cyclegan_x2y_train'):
 961       train_ops_x2y = gan_train_ops(model.model_x2y, loss.loss_x2y,
 962                                     **saved_params)
 963     with ops.name_scope('cyclegan_y2x_train'):
 964       train_ops_y2x = gan_train_ops(model.model_y2x, loss.loss_y2x,
 965                                     **saved_params)
 966     return namedtuples.GANTrainOps(
 967         (train_ops_x2y.generator_train_op, train_ops_y2x.generator_train_op),
 968         (train_ops_x2y.discriminator_train_op,
 969          train_ops_y2x.discriminator_train_op),
 970         training_util.get_or_create_global_step().assign_add(1))
 971 
 972   # Create global step increment op.
 973   global_step = training_util.get_or_create_global_step()
 974   global_step_inc = global_step.assign_add(1)
 975 
 976   # Get generator and discriminator update ops. We split them so that update
 977   # ops aren't accidentally run multiple times. For now, throw an error if
 978   # there are update ops that aren't associated with either the generator or
 979   # the discriminator. Might modify the `kwargs` dictionary.
 980   gen_update_ops, dis_update_ops = _get_update_ops(
 981       kwargs, model.generator_scope.name, model.discriminator_scope.name,
 982       check_for_unused_update_ops)
 983 
 984   generator_global_step = None
 985   if isinstance(generator_optimizer,
 986                 sync_replicas_optimizer.SyncReplicasOptimizer):
 987     # TODO(joelshor): Figure out a way to get this work without including the
 988     # dummy global step in the checkpoint.
 989     # WARNING: Making this variable a local variable causes sync replicas to
 990     # hang forever.
 991     generator_global_step = variable_scope.get_variable(
 992         'dummy_global_step_generator',
 993         shape=[],
 994         dtype=global_step.dtype.base_dtype,
 995         initializer=init_ops.zeros_initializer(),
 996         trainable=False,
 997         collections=[ops.GraphKeys.GLOBAL_VARIABLES])
 998     gen_update_ops += [generator_global_step.assign(global_step)]
 999   with ops.name_scope('generator_train'):
1000     gen_train_op = training.create_train_op(
1001         total_loss=loss.generator_loss,
1002         optimizer=generator_optimizer,
1003         variables_to_train=model.generator_variables,
1004         global_step=generator_global_step,
1005         update_ops=gen_update_ops,
1006         **kwargs)
1007 
1008   discriminator_global_step = None
1009   if isinstance(discriminator_optimizer,
1010                 sync_replicas_optimizer.SyncReplicasOptimizer):
1011     # See comment above `generator_global_step`.
1012     discriminator_global_step = variable_scope.get_variable(
1013         'dummy_global_step_discriminator',
1014         shape=[],
1015         dtype=global_step.dtype.base_dtype,
1016         initializer=init_ops.zeros_initializer(),
1017         trainable=False,
1018         collections=[ops.GraphKeys.GLOBAL_VARIABLES])
1019     dis_update_ops += [discriminator_global_step.assign(global_step)]
1020   with ops.name_scope('discriminator_train'):
1021     disc_train_op = training.create_train_op(
1022         total_loss=loss.discriminator_loss,
1023         optimizer=discriminator_optimizer,
1024         variables_to_train=model.discriminator_variables,
1025         global_step=discriminator_global_step,
1026         update_ops=dis_update_ops,
1027         **kwargs)
1028 
1029   return namedtuples.GANTrainOps(gen_train_op, disc_train_op, global_step_inc)

1075 def get_joint_train_hooks(train_steps=namedtuples.GANTrainSteps(1, 1)):
1076   """Returns a hooks function for sequential GAN training.
1077 
1078   When using these train hooks, IT IS RECOMMENDED TO USE `use_locking=True` ON
1079   ALL OPTIMIZERS TO AVOID RACE CONDITIONS.
1080 
1081   The order of steps taken is:
1082   1) Combined generator and discriminator steps
1083   2) Generator only steps, if any remain
1084   3) Discriminator only steps, if any remain
1085 
1086   **NOTE**: Unlike `get_sequential_train_hooks`, this method performs updates
1087   for the generator and discriminator simultaneously whenever possible. This
1088   reduces the number of `tf.Session` calls, and can also change the training
1089   semantics.
1090 
1091   To illustrate the difference look at the following example:
1092 
1093   `train_steps=namedtuples.GANTrainSteps(3, 5)` will cause
1094   `get_sequential_train_hooks` to make 8 session calls:
1095     1) 3 generator steps
1096     2) 5 discriminator steps
1097 
1098   In contrast, `get_joint_train_steps` will make 5 session calls:
1099   1) 3 generator + discriminator steps
1100   2) 2 discriminator steps
1101 
1102   Args:
1103     train_steps: A `GANTrainSteps` tuple that determines how many generator
1104       and discriminator training steps to take.
1105 
1106   Returns:
1107     A function that takes a GANTrainOps tuple and returns a list of hooks.
1108   """
1109   g_steps = train_steps.generator_train_steps
1110   d_steps = train_steps.discriminator_train_steps
1111   # Get the number of each type of step that should be run.
1112   num_d_and_g_steps = min(g_steps, d_steps)
1113   num_g_steps = g_steps - num_d_and_g_steps
1114   num_d_steps = d_steps - num_d_and_g_steps
1115 
1116   def get_hooks(train_ops):
1117     g_op = train_ops.generator_train_op
1118     d_op = train_ops.discriminator_train_op
1119 
1120     joint_hook = RunTrainOpsHook([g_op, d_op], num_d_and_g_steps)
1121     g_hook = RunTrainOpsHook(g_op, num_g_steps)
1122     d_hook = RunTrainOpsHook(d_op, num_d_steps)
1123 
1124     return [joint_hook, g_hook, d_hook]
1125 
1126   return get_hooks

1187 def get_sequential_train_steps(train_steps=namedtuples.GANTrainSteps(1, 1)):
1188   """Returns a thin wrapper around slim.learning.train_step, for GANs.
1189 
1190   This function is to provide support for the Supervisor. For new code, please
1191   use `MonitoredSession` and `get_sequential_train_hooks`.
1192 
1193   Args:
1194     train_steps: A `GANTrainSteps` tuple that determines how many generator
1195       and discriminator training steps to take.
1196 
1197   Returns:
1198     A function that can be used for `train_step_fn` for GANs.
1199   """
1200 
1201   def sequential_train_steps(sess, train_ops, global_step, train_step_kwargs):
1202     """A thin wrapper around slim.learning.train_step, for GANs.
1203 
1204     Args:
1205       sess: A Tensorflow session.
1206       train_ops: A GANTrainOps tuple of train ops to run.
1207       global_step: The global step.
1208       train_step_kwargs: Dictionary controlling `train_step` behavior.
1209 
1210     Returns:
1211       A scalar final loss and a bool whether or not the train loop should stop.
1212     """
1213     # Only run `should_stop` at the end, if required. Make a local copy of
1214     # `train_step_kwargs`, if necessary, so as not to modify the caller's
1215     # dictionary.
1216     should_stop_op, train_kwargs = None, train_step_kwargs
1217     if 'should_stop' in train_step_kwargs:
1218       should_stop_op = train_step_kwargs['should_stop']
1219       train_kwargs = train_step_kwargs.copy()
1220       del train_kwargs['should_stop']
1221 
1222     # Run generator training steps.
1223     gen_loss = 0
1224     for _ in range(train_steps.generator_train_steps):
1225       cur_gen_loss, _ = slim_learning.train_step(
1226           sess, train_ops.generator_train_op, global_step, train_kwargs)
1227       gen_loss += cur_gen_loss
1228 
1229     # Run discriminator training steps.
1230     dis_loss = 0
1231     for _ in range(train_steps.discriminator_train_steps):
1232       cur_dis_loss, _ = slim_learning.train_step(
1233           sess, train_ops.discriminator_train_op, global_step, train_kwargs)
1234       dis_loss += cur_dis_loss
1235 
1236     sess.run(train_ops.global_step_inc_op)
1237 
1238     # Run the `should_stop` op after the global step has been incremented, so
1239     # that the `should_stop` aligns with the proper `global_step` count.
1240     if should_stop_op is not None:
1241       should_stop = sess.run(should_stop_op)
1242     else:
1243       should_stop = False
1244 
1245     return gen_loss + dis_loss, should_stop
1246 
1247   return sequential_train_steps

1263 def _validate_distributions(distributions_l, noise_l):
1264   if not isinstance(distributions_l, (tuple, list)):
1265     raise ValueError('`predicted_distributions` must be a list. Instead, found '
1266                      '%s.' % type(distributions_l))
1267   for dist in distributions_l:
1268     if not isinstance(dist, ds.Distribution):
1269       raise ValueError('Every element in `predicted_distributions` must be a '
1270                        '`tf.Distribution`. Instead, found %s.' % type(dist))
1271   if len(distributions_l) != len(noise_l):
1272     raise ValueError('Length of `predicted_distributions` %i must be the same '
1273                      'as the length of structured noise %i.' %
1274                      (len(distributions_l), len(noise_l)))
