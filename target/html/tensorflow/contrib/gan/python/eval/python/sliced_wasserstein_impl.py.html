  15 """Implementation of Sliced Wasserstein Distance.
  16 
  17 Proposed in https://arxiv.org/abs/1710.10196 and the official Theano

  40 _GAUSSIAN_FILTER = np.float32([[1, 4, 6, 4, 1], [4, 16, 24, 16, 4], [
  41     6, 24, 36, 24, 6
  42 ], [4, 16, 24, 16, 4], [1, 4, 6, 4, 1]]).reshape([5, 5, 1, 1]) / 256.0

  45 def _laplacian_pyramid(batch, num_levels):
  46   """Compute a Laplacian pyramid.
  47 
  48   Args:
  49       batch: (tensor) The batch of images (batch, height, width, channels).
  50       num_levels: (int) Desired number of hierarchical levels.
  51   Returns:
  52       List of tensors from the highest to lowest resolution.
  53   """
  54   gaussian_filter = constant_op.constant(_GAUSSIAN_FILTER)
  55 
  56   def spatial_conv(batch, gain):
  57     s = array_ops.shape(batch)
  58     padded = array_ops.pad(batch, [[0, 0], [2, 2], [2, 2], [0, 0]], 'REFLECT')
  59     xt = array_ops.transpose(padded, [0, 3, 1, 2])
  60     xt = array_ops.reshape(xt, [s[0] * s[3], s[1] + 4, s[2] + 4, 1])
  61     conv_out = nn_ops.conv2d(xt, gaussian_filter * gain, [1] * 4, 'VALID')
  62     conv_xt = array_ops.reshape(conv_out, [s[0], s[3], s[1], s[2]])
  63     conv_xt = array_ops.transpose(conv_xt, [0, 2, 3, 1])
  64     return conv_xt
  65 
  66   def pyr_down(batch):  # matches cv2.pyrDown()
  67     return spatial_conv(batch, 1)[:, ::2, ::2]
  68 
  69   def pyr_up(batch):  # matches cv2.pyrUp()
  70     s = array_ops.shape(batch)
  71     zeros = array_ops.zeros([3 * s[0], s[1], s[2], s[3]])
  72     res = array_ops.concat([batch, zeros], 0)
  73     res = array_ops.batch_to_space(res, crops=[[0, 0], [0, 0]], block_size=2)
  74     res = spatial_conv(res, 4)
  75     return res
  76 
  77   pyramid = [math_ops.to_float(batch)]
  78   for _ in range(1, num_levels):
  79     pyramid.append(pyr_down(pyramid[-1]))
  80     pyramid[-2] -= pyr_up(pyramid[-1])
  81   return pyramid

  84 def _batch_to_patches(batch, patches_per_image, patch_size):
  85   """Extract patches from a batch.
  86 
  87   Args:
  88       batch: (tensor) The batch of images (batch, height, width, channels).
  89       patches_per_image: (int) Number of patches to extract per image.
  90       patch_size: (int) Size of the patches (size, size, channels) to extract.
  91   Returns:
  92       Tensor (batch*patches_per_image, patch_size, patch_size, channels) of
  93       patches.
  94   """
  95 
  96   def py_func_random_patches(batch):
  97     """Numpy wrapper."""
  98     batch_size, height, width, channels = batch.shape
  99     patch_count = patches_per_image * batch_size
 100     hs = patch_size // 2
 101     # Randomly pick patches.
 102     patch_id, y, x, chan = np.ogrid[0:patch_count, -hs:hs + 1, -hs:hs + 1, 0:3]
 103     img_id = patch_id // patches_per_image
 104     # pylint: disable=g-no-augmented-assignment
 105     # Need explicit addition for broadcast to work properly.
 106     y = y + np.random.randint(hs, height - hs, size=(patch_count, 1, 1, 1))
 107     x = x + np.random.randint(hs, width - hs, size=(patch_count, 1, 1, 1))
 108     # pylint: enable=g-no-augmented-assignment
 109     idx = ((img_id * height + y) * width + x) * channels + chan
 110     patches = batch.flat[idx]
 111     return patches
 112 
 113   patches = script_ops.py_func(
 114       py_func_random_patches, [batch], batch.dtype, stateful=False)
 115   return patches

 118 def _normalize_patches(patches):
 119   """Normalize patches by their mean and standard deviation.
 120 
 121   Args:
 122       patches: (tensor) The batch of patches (batch, size, size, channels).
 123   Returns:
 124       Tensor (batch, size, size, channels) of the normalized patches.
 125   """
 126   patches = array_ops.concat(patches, 0)
 127   mean, variance = nn.moments(patches, [1, 2, 3], keep_dims=True)
 128   patches = (patches - mean) / math_ops.sqrt(variance)
 129   return array_ops.reshape(patches, [array_ops.shape(patches)[0], -1])

 132 def _sort_rows(matrix, num_rows):
 133   """Sort matrix rows by the last column.
 134 
 135   Args:
 136       matrix: a matrix of values (row,col).
 137       num_rows: (int) number of sorted rows to return from the matrix.
 138   Returns:
 139       Tensor (num_rows, col) of the sorted matrix top K rows.
 140   """
 141   tmatrix = array_ops.transpose(matrix, [1, 0])
 142   sorted_tmatrix = nn_ops.top_k(tmatrix, num_rows)[0]
 143   return array_ops.transpose(sorted_tmatrix, [1, 0])

 146 def _sliced_wasserstein(a, b, random_sampling_count, random_projection_dim):
 147   """Compute the approximate sliced Wasserstein distance.
 148 
 149   Args:
 150       a: (matrix) Distribution "a" of samples (row, col).
 151       b: (matrix) Distribution "b" of samples (row, col).
 152       random_sampling_count: (int) Number of random projections to average.
 153       random_projection_dim: (int) Dimension of the random projection space.
 154   Returns:
 155       Float containing the approximate distance between "a" and "b".
 156   """
 157   s = array_ops.shape(a)
 158   means = []
 159   for _ in range(random_sampling_count):
 160     # Random projection matrix.
 161     proj = random_ops.random_normal(
 162         [array_ops.shape(a)[1], random_projection_dim])
 163     proj *= math_ops.rsqrt(
 164         math_ops.reduce_sum(math_ops.square(proj), 0, keepdims=True))
 165     # Project both distributions and sort them.
 166     proj_a = math_ops.matmul(a, proj)
 167     proj_b = math_ops.matmul(b, proj)
 168     proj_a = _sort_rows(proj_a, s[0])
 169     proj_b = _sort_rows(proj_b, s[0])
 170     # Pairwise Wasserstein distance.
 171     wdist = math_ops.reduce_mean(math_ops.abs(proj_a - proj_b))
 172     means.append(wdist)
 173   return math_ops.reduce_mean(means)

 176 def _sliced_wasserstein_svd(a, b):
 177   """Compute the approximate sliced Wasserstein distance using an SVD.
 178 
 179   This is not part of the paper, it's a variant with possibly more accurate
 180   measure.
 181 
 182   Args:
 183       a: (matrix) Distribution "a" of samples (row, col).
 184       b: (matrix) Distribution "b" of samples (row, col).
 185   Returns:
 186       Float containing the approximate distance between "a" and "b".
 187   """
 188   s = array_ops.shape(a)
 189   # Random projection matrix.
 190   sig, u = linalg_ops.svd(array_ops.concat([a, b], 0))[:2]
 191   proj_a, proj_b = array_ops.split(u * sig, 2, axis=0)
 192   proj_a = _sort_rows(proj_a[:, ::-1], s[0])
 193   proj_b = _sort_rows(proj_b[:, ::-1], s[0])
 194   # Pairwise Wasserstein distance.
 195   wdist = math_ops.reduce_mean(math_ops.abs(proj_a - proj_b))
 196   return wdist

 199 def sliced_wasserstein_distance(real_images,
 200                                 fake_images,
 201                                 resolution_min=16,
 202                                 patches_per_image=64,
 203                                 patch_size=7,
 204                                 random_sampling_count=1,
 205                                 random_projection_dim=7 * 7 * 3,
 206                                 use_svd=False):
 207   """Compute the Wasserstein distance between two distributions of images.
 208 
 209   Note that measure vary with the number of images. Use 8192 images to get
 210   numbers comparable to the ones in the original paper.
 211 
 212   Args:
 213       real_images: (tensor) Real images (batch, height, width, channels).
 214       fake_images: (tensor) Fake images (batch, height, width, channels).
 215       resolution_min: (int) Minimum resolution for the Laplacian pyramid.
 216       patches_per_image: (int) Number of patches to extract per image per
 217         Laplacian level.
 218       patch_size: (int) Width of a square patch.
 219       random_sampling_count: (int) Number of random projections to average.
 220       random_projection_dim: (int) Dimension of the random projection space.
 221       use_svd: experimental method to compute a more accurate distance.
 222   Returns:
 223       List of tuples (distance_real, distance_fake) for each level of the
 224       Laplacian pyramid from the highest resolution to the lowest.
 225         distance_real is the Wasserstein distance between real images
 226         distance_fake is the Wasserstein distance between real and fake images.
 227   Raises:
 228       ValueError: If the inputs shapes are incorrect. Input tensor dimensions
 229       (batch, height, width, channels) are expected to be known at graph
 230       construction time. In addition height and width must be the same and the
 231       number of colors should be exactly 3. Real and fake images must have the
 232       same size.
 233   """
 234   height = real_images.shape[1]
 235   real_images.shape.assert_is_compatible_with([None, None, height, 3])
 236   fake_images.shape.assert_is_compatible_with(real_images.shape)
 237 
 238   # Select resolutions.
 239   resolution_full = int(height)
 240   resolution_min = min(resolution_min, resolution_full)
 241   resolution_max = resolution_full
 242   # Base loss of detail.
 243   resolutions = [
 244       2**i
 245       for i in range(
 246           int(np.log2(resolution_max)),
 247           int(np.log2(resolution_min)) - 1, -1)
 248   ]
 249 
 250   # Gather patches for each level of the Laplacian pyramids.
 251   patches_real, patches_fake, patches_test = (
 252       [[] for _ in resolutions] for _ in range(3))
 253   for lod, level in enumerate(
 254       _laplacian_pyramid(real_images, len(resolutions))):
 255     patches_real[lod].append(
 256         _batch_to_patches(level, patches_per_image, patch_size))
 257     patches_test[lod].append(
 258         _batch_to_patches(level, patches_per_image, patch_size))
 259 
 260   for lod, level in enumerate(
 261       _laplacian_pyramid(fake_images, len(resolutions))):
 262     patches_fake[lod].append(
 263         _batch_to_patches(level, patches_per_image, patch_size))
 264 
 265   for lod in range(len(resolutions)):
 266     for patches in [patches_real, patches_test, patches_fake]:
 267       patches[lod] = _normalize_patches(patches[lod])
 268 
 269   # Evaluate scores.
 270   scores = []
 271   for lod in range(len(resolutions)):
 272     if not use_svd:
 273       scores.append(
 274           (_sliced_wasserstein(patches_real[lod], patches_test[lod],
 275                                random_sampling_count, random_projection_dim),
 276            _sliced_wasserstein(patches_real[lod], patches_fake[lod],
 277                                random_sampling_count, random_projection_dim)))
 278     else:
 279       scores.append(
 280           (_sliced_wasserstein_svd(patches_real[lod], patches_test[lod]),
 281            _sliced_wasserstein_svd(patches_real[lod], patches_fake[lod])))
 282   return scores
