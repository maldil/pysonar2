  78 def _validate_images(images, image_size):
  79   images = ops.convert_to_tensor(images)
  80   images.shape.with_rank(4)
  81   images.shape.assert_is_compatible_with([None, image_size, image_size, None])
  82   return images

  85 def _symmetric_matrix_square_root(mat, eps=1e-10):
  86   """Compute square root of a symmetric matrix.
  87 
  88   Note that this is different from an elementwise square root. We want to
  89   compute M' where M' = sqrt(mat) such that M' * M' = mat.
  90 
  91   Also note that this method **only** works for symmetric matrices.
  92 
  93   Args:
  94     mat: Matrix to take the square root of.
  95     eps: Small epsilon such that any element less than eps will not be square
  96       rooted to guard against numerical instability.
  97 
  98   Returns:
  99     Matrix square root of mat.
 100   """
 101   # Unlike numpy, tensorflow's return order is (s, u, v)
 102   s, u, v = linalg_ops.svd(mat)
 103   # sqrt is unstable around 0, just use 0 in such case
 104   si = array_ops.where(math_ops.less(s, eps), s, math_ops.sqrt(s))
 105   # Note that the v returned by Tensorflow is v = V
 106   # (when referencing the equation A = U S V^T)
 107   # This is unlike Numpy which returns v = V^T
 108   return math_ops.matmul(
 109       math_ops.matmul(u, array_ops.diag(si)), v, transpose_b=True)

 112 def preprocess_image(images,
 113                      height=INCEPTION_DEFAULT_IMAGE_SIZE,
 114                      width=INCEPTION_DEFAULT_IMAGE_SIZE,
 115                      scope=None):
 116   """Prepare a batch of images for evaluation.
 117 
 118   This is the preprocessing portion of the graph from
 119   http://download.tensorflow.org/models/image/imagenet/inception-2015-12-05.tgz.
 120 
 121   Note that it expects Tensors in [0, 255]. This function maps pixel values to
 122   [-1, 1] and resizes to match the InceptionV1 network.
 123 
 124   Args:
 125     images: 3-D or 4-D Tensor of images. Values are in [0, 255].
 126     height: Integer. Height of resized output image.
 127     width: Integer. Width of resized output image.
 128     scope: Optional scope for name_scope.
 129 
 130   Returns:
 131     3-D or 4-D float Tensor of prepared image(s). Values are in [-1, 1].
 132   """
 133   is_single = images.shape.ndims == 3
 134   with ops.name_scope(scope, 'preprocess', [images, height, width]):
 135     if not images.dtype.is_floating:
 136       images = math_ops.to_float(images)
 137     if is_single:
 138       images = array_ops.expand_dims(images, axis=0)
 139     resized = image_ops.resize_bilinear(images, [height, width])
 140     resized = (resized - 128.0) / 128.0
 141     if is_single:
 142       resized = array_ops.squeeze(resized, axis=0)
 143     return resized

 146 def _kl_divergence(p, p_logits, q):
 147   """Computes the Kullback-Liebler divergence between p and q.
 148 
 149   This function uses p's logits in some places to improve numerical stability.
 150 
 151   Specifically:
 152 
 153   KL(p || q) = sum[ p * log(p / q) ]
 154     = sum[ p * ( log(p)                - log(q) ) ]
 155     = sum[ p * ( log_softmax(p_logits) - log(q) ) ]
 156 
 157   Args:
 158     p: A 2-D floating-point Tensor p_ij, where `i` corresponds to the minibatch
 159       example and `j` corresponds to the probability of being in class `j`.
 160     p_logits: A 2-D floating-point Tensor corresponding to logits for `p`.
 161     q: A 1-D floating-point Tensor, where q_j corresponds to the probability
 162       of class `j`.
 163 
 164   Returns:
 165     KL divergence between two distributions. Output dimension is 1D, one entry
 166     per distribution in `p`.
 167 
 168   Raises:
 169     ValueError: If any of the inputs aren't floating-point.
 170     ValueError: If p or p_logits aren't 2D.
 171     ValueError: If q isn't 1D.
 172   """
 173   for tensor in [p, p_logits, q]:
 174     if not tensor.dtype.is_floating:
 175       raise ValueError('Input %s must be floating type.', tensor.name)
 176   p.shape.assert_has_rank(2)
 177   p_logits.shape.assert_has_rank(2)
 178   q.shape.assert_has_rank(1)
 179   return math_ops.reduce_sum(
 180       p * (nn_ops.log_softmax(p_logits) - math_ops.log(q)), axis=1)

 194 def get_graph_def_from_url_tarball(url, filename, tar_filename=None):
 195   """Get a GraphDef proto from a tarball on the web.
 196 
 197   Args:
 198     url: Web address of tarball
 199     filename: Filename of graph definition within tarball
 200     tar_filename: Temporary download filename (None = always download)
 201 
 202   Returns:
 203     A GraphDef loaded from a file in the downloaded tarball.
 204   """
 205   if not (tar_filename and os.path.exists(tar_filename)):
 206 
 207     def _progress(count, block_size, total_size):
 208       sys.stdout.write('\r>> Downloading %s %.1f%%' %
 209                        (url,
 210                         float(count * block_size) / float(total_size) * 100.0))
 211       sys.stdout.flush()
 212 
 213     tar_filename, _ = urllib.request.urlretrieve(url, tar_filename, _progress)
 214   with tarfile.open(tar_filename, 'r:gz') as tar:
 215     proto_str = tar.extractfile(filename).read()
 216   return graph_pb2.GraphDef.FromString(proto_str)

 224 def run_inception(images,
 225                   graph_def=None,
 226                   default_graph_def_fn=_default_graph_def_fn,
 227                   image_size=INCEPTION_DEFAULT_IMAGE_SIZE,
 228                   input_tensor=INCEPTION_INPUT,
 229                   output_tensor=INCEPTION_OUTPUT):
 230   """Run images through a pretrained Inception classifier.
 231 
 232   Args:
 233     images: Input tensors. Must be [batch, height, width, channels]. Input shape
 234       and values must be in [-1, 1], which can be achieved using
 235       `preprocess_image`.
 236     graph_def: A GraphDef proto of a pretrained Inception graph. If `None`,
 237       call `default_graph_def_fn` to get GraphDef.
 238     default_graph_def_fn: A function that returns a GraphDef. Used if
 239       `graph_def` is `None. By default, returns a pretrained InceptionV3 graph.
 240     image_size: Required image width and height. See unit tests for the default
 241       values.
 242     input_tensor: Name of input Tensor.
 243     output_tensor: Name or list of output Tensors. This function will compute
 244       activations at the specified layer. Examples include INCEPTION_V3_OUTPUT
 245       and INCEPTION_V3_FINAL_POOL which would result in this function computing
 246       the final logits or the penultimate pooling layer.
 247 
 248   Returns:
 249     Tensor or Tensors corresponding to computed `output_tensor`.
 250 
 251   Raises:
 252     ValueError: If images are not the correct size.
 253     ValueError: If neither `graph_def` nor `default_graph_def_fn` are provided.
 254   """
 255   images = _validate_images(images, image_size)
 256 
 257   if graph_def is None:
 258     if default_graph_def_fn is None:
 259       raise ValueError('If `graph_def` is `None`, must provide '
 260                        '`default_graph_def_fn`.')
 261     graph_def = default_graph_def_fn()
 262 
 263   activations = run_image_classifier(images, graph_def, input_tensor,
 264                                      output_tensor)
 265   if isinstance(activations, list):
 266     for i, activation in enumerate(activations):
 267       if array_ops.rank(activation) != 2:
 268         activations[i] = layers.flatten(activation)
 269   else:
 270     if array_ops.rank(activations) != 2:
 271       activations = layers.flatten(activations)
 272 
 273   return activations

 276 def run_image_classifier(tensor,
 277                          graph_def,
 278                          input_tensor,
 279                          output_tensor,
 280                          scope='RunClassifier'):
 281   """Runs a network from a frozen graph.
 282 
 283   Args:
 284     tensor: An Input tensor.
 285     graph_def: A GraphDef proto.
 286     input_tensor: Name of input tensor in graph def.
 287     output_tensor: A tensor name or list of tensor names in graph def.
 288     scope: Name scope for classifier.
 289 
 290   Returns:
 291     Classifier output if `output_tensor` is a string, or a list of outputs if
 292     `output_tensor` is a list.
 293 
 294   Raises:
 295     ValueError: If `input_tensor` or `output_tensor` aren't in the graph_def.
 296   """
 297   input_map = {input_tensor: tensor}
 298   is_singleton = isinstance(output_tensor, str)
 299   if is_singleton:
 300     output_tensor = [output_tensor]
 301   classifier_outputs = importer.import_graph_def(
 302       graph_def, input_map, output_tensor, name=scope)
 303   if is_singleton:
 304     classifier_outputs = classifier_outputs[0]
 305 
 306   return classifier_outputs

 309 def classifier_score(images, classifier_fn, num_batches=1):
 310   """Classifier score for evaluating a conditional generative model.
 311 
 312   This is based on the Inception Score, but for an arbitrary classifier.
 313 
 314   This technique is described in detail in https://arxiv.org/abs/1606.03498. In
 315   summary, this function calculates
 316 
 317   exp( E[ KL(p(y|x) || p(y)) ] )
 318 
 319   which captures how different the network's classification prediction is from
 320   the prior distribution over classes.
 321 
 322   NOTE: This function consumes images, computes their logits, and then
 323   computes the classifier score. If you would like to precompute many logits for
 324   large batches, use classifier_score_from_logits(), which this method also
 325   uses.
 326 
 327   Args:
 328     images: Images to calculate the classifier score for.
 329     classifier_fn: A function that takes images and produces logits based on a
 330       classifier.
 331     num_batches: Number of batches to split `generated_images` in to in order to
 332       efficiently run them through the classifier network.
 333 
 334   Returns:
 335     The classifier score. A floating-point scalar of the same type as the output
 336     of `classifier_fn`.
 337   """
 338   generated_images_list = array_ops.split(
 339       images, num_or_size_splits=num_batches)
 340 
 341   # Compute the classifier splits using the memory-efficient `map_fn`.
 342   logits = functional_ops.map_fn(
 343       fn=classifier_fn,
 344       elems=array_ops.stack(generated_images_list),
 345       parallel_iterations=1,
 346       back_prop=False,
 347       swap_memory=True,
 348       name='RunClassifier')
 349   logits = array_ops.concat(array_ops.unstack(logits), 0)
 350 
 351   return classifier_score_from_logits(logits)
 352

 354 def classifier_score_from_logits(logits):
 355   """Classifier score for evaluating a generative model from logits.
 356 
 357   This method computes the classifier score for a set of logits. This can be
 358   used independently of the classifier_score() method, especially in the case
 359   of using large batches during evaluation where we would like precompute all
 360   of the logits before computing the classifier score.
 361 
 362   This technique is described in detail in https://arxiv.org/abs/1606.03498. In
 363   summary, this function calculates:
 364 
 365   exp( E[ KL(p(y|x) || p(y)) ] )
 366 
 367   which captures how different the network's classification prediction is from
 368   the prior distribution over classes.
 369 
 370   Args:
 371     logits: Precomputed 2D tensor of logits that will be used to
 372       compute the classifier score.
 373 
 374   Returns:
 375     The classifier score. A floating-point scalar of the same type as the output
 376     of `logits`.
 377   """
 378   logits.shape.assert_has_rank(2)
 379 
 380   # Use maximum precision for best results.
 381   logits_dtype = logits.dtype
 382   if logits_dtype != dtypes.float64:
 383     logits = math_ops.to_double(logits)
 384 
 385   p = nn_ops.softmax(logits)
 386   q = math_ops.reduce_mean(p, axis=0)
 387   kl = _kl_divergence(p, logits, q)
 388   kl.shape.assert_has_rank(1)
 389   log_score = math_ops.reduce_mean(kl)
 390   final_score = math_ops.exp(log_score)
 391 
 392   if logits_dtype != dtypes.float64:
 393     final_score = math_ops.cast(final_score, logits_dtype)
 394 
 395   return final_score
 396

 404 def trace_sqrt_product(sigma, sigma_v):
 405   """Find the trace of the positive sqrt of product of covariance matrices.
 406 
 407   '_symmetric_matrix_square_root' only works for symmetric matrices, so we
 408   cannot just take _symmetric_matrix_square_root(sigma * sigma_v).
 409   ('sigma' and 'sigma_v' are symmetric, but their product is not necessarily).
 410 
 411   Let sigma = A A so A = sqrt(sigma), and sigma_v = B B.
 412   We want to find trace(sqrt(sigma sigma_v)) = trace(sqrt(A A B B))
 413   Note the following properties:
 414   (i) forall M1, M2: eigenvalues(M1 M2) = eigenvalues(M2 M1)
 415      => eigenvalues(A A B B) = eigenvalues (A B B A)
 416   (ii) if M1 = sqrt(M2), then eigenvalues(M1) = sqrt(eigenvalues(M2))
 417      => eigenvalues(sqrt(sigma sigma_v)) = sqrt(eigenvalues(A B B A))
 418   (iii) forall M: trace(M) = sum(eigenvalues(M))
 419      => trace(sqrt(sigma sigma_v)) = sum(eigenvalues(sqrt(sigma sigma_v)))
 420                                    = sum(sqrt(eigenvalues(A B B A)))
 421                                    = sum(eigenvalues(sqrt(A B B A)))
 422                                    = trace(sqrt(A B B A))
 423                                    = trace(sqrt(A sigma_v A))
 424   A = sqrt(sigma). Both sigma and A sigma_v A are symmetric, so we **can**
 425   use the _symmetric_matrix_square_root function to find the roots of these
 426   matrices.
 427 
 428   Args:
 429     sigma: a square, symmetric, real, positive semi-definite covariance matrix
 430     sigma_v: same as sigma
 431 
 432   Returns:
 433     The trace of the positive square root of sigma*sigma_v
 434   """
 435 
 436   # Note sqrt_sigma is called "A" in the proof above
 437   sqrt_sigma = _symmetric_matrix_square_root(sigma)
 438 
 439   # This is sqrt(A sigma_v A) above
 440   sqrt_a_sigmav_a = math_ops.matmul(sqrt_sigma,
 441                                     math_ops.matmul(sigma_v, sqrt_sigma))
 442 
 443   return math_ops.trace(_symmetric_matrix_square_root(sqrt_a_sigmav_a))

 446 def frechet_classifier_distance(real_images,
 447                                 generated_images,
 448                                 classifier_fn,
 449                                 num_batches=1):
 450   """Classifier distance for evaluating a generative model.
 451 
 452   This is based on the Frechet Inception distance, but for an arbitrary
 453   classifier.
 454 
 455   This technique is described in detail in https://arxiv.org/abs/1706.08500.
 456   Given two Gaussian distribution with means m and m_w and covariance matrices
 457   C and C_w, this function calculates
 458 
 459               |m - m_w|^2 + Tr(C + C_w - 2(C * C_w)^(1/2))
 460 
 461   which captures how different the distributions of real images and generated
 462   images (or more accurately, their visual features) are. Note that unlike the
 463   Inception score, this is a true distance and utilizes information about real
 464   world images.
 465 
 466   Note that when computed using sample means and sample covariance matrices,
 467   Frechet distance is biased. It is more biased for small sample sizes. (e.g.
 468   even if the two distributions are the same, for a small sample size, the
 469   expected Frechet distance is large). It is important to use the same
 470   sample size to compute Frechet classifier distance when comparing two
 471   generative models.
 472 
 473   NOTE: This function consumes images, computes their activations, and then
 474   computes the classifier score. If you would like to precompute many
 475   activations for real and generated images for large batches, please use
 476   frechet_clasifier_distance_from_activations(), which this method also uses.
 477 
 478   Args:
 479     real_images: Real images to use to compute Frechet Inception distance.
 480     generated_images: Generated images to use to compute Frechet Inception
 481       distance.
 482     classifier_fn: A function that takes images and produces activations
 483       based on a classifier.
 484     num_batches: Number of batches to split images in to in order to
 485       efficiently run them through the classifier network.
 486 
 487   Returns:
 488     The Frechet Inception distance. A floating-point scalar of the same type
 489     as the output of `classifier_fn`.
 490   """
 491   real_images_list = array_ops.split(
 492       real_images, num_or_size_splits=num_batches)
 493   generated_images_list = array_ops.split(
 494       generated_images, num_or_size_splits=num_batches)
 495 
 496   real_imgs = array_ops.stack(real_images_list)
 497   generated_imgs = array_ops.stack(generated_images_list)
 498 
 499   # Compute the activations using the memory-efficient `map_fn`.
 500   def compute_activations(elems):
 501     return functional_ops.map_fn(fn=classifier_fn,
 502                                  elems=elems,
 503                                  parallel_iterations=1,
 504                                  back_prop=False,
 505                                  swap_memory=True,
 506                                  name='RunClassifier')
 507 
 508   real_a = compute_activations(real_imgs)
 509   gen_a = compute_activations(generated_imgs)
 510 
 511   # Ensure the activations have the right shapes.
 512   real_a = array_ops.concat(array_ops.unstack(real_a), 0)
 513   gen_a = array_ops.concat(array_ops.unstack(gen_a), 0)
 514 
 515   return frechet_classifier_distance_from_activations(real_a, gen_a)

 518 def mean_only_frechet_classifier_distance_from_activations(
 519     real_activations, generated_activations):
 520   """Classifier distance for evaluating a generative model from activations.
 521 
 522   Given two Gaussian distribution with means m and m_w and covariance matrices
 523   C and C_w, this function calcuates
 524 
 525                                 |m - m_w|^2
 526 
 527   which captures how different the distributions of real images and generated
 528   images (or more accurately, their visual features) are. Note that unlike the
 529   Inception score, this is a true distance and utilizes information about real
 530   world images.
 531 
 532   Note that when computed using sample means and sample covariance matrices,
 533   Frechet distance is biased. It is more biased for small sample sizes. (e.g.
 534   even if the two distributions are the same, for a small sample size, the
 535   expected Frechet distance is large). It is important to use the same
 536   sample size to compute frechet classifier distance when comparing two
 537   generative models.
 538 
 539   In this variant, we only compute the difference between the means of the
 540   fitted Gaussians. The computation leads to O(n) vs. O(n^2) memory usage, yet
 541   still retains much of the same information as FID.
 542 
 543   Args:
 544     real_activations: 2D array of activations of real images of size
 545       [num_images, num_dims] to use to compute Frechet Inception distance.
 546     generated_activations: 2D array of activations of generated images of size
 547       [num_images, num_dims] to use to compute Frechet Inception distance.
 548 
 549   Returns:
 550     The mean-only Frechet Inception distance. A floating-point scalar of the
 551     same type as the output of the activations.
 552   """
 553   real_activations.shape.assert_has_rank(2)
 554   generated_activations.shape.assert_has_rank(2)
 555 
 556   activations_dtype = real_activations.dtype
 557   if activations_dtype != dtypes.float64:
 558     real_activations = math_ops.to_double(real_activations)
 559     generated_activations = math_ops.to_double(generated_activations)
 560 
 561   # Compute means of activations.
 562   m = math_ops.reduce_mean(real_activations, 0)
 563   m_w = math_ops.reduce_mean(generated_activations, 0)
 564 
 565   # Next the distance between means.
 566   mean = math_ops.reduce_sum(
 567       math_ops.squared_difference(m, m_w))  # Equivalent to L2 but more stable.
 568   mofid = mean
 569   if activations_dtype != dtypes.float64:
 570     mofid = math_ops.cast(mofid, activations_dtype)
 571 
 572   return mofid

 575 def diagonal_only_frechet_classifier_distance_from_activations(
 576     real_activations, generated_activations):
 577   """Classifier distance for evaluating a generative model.
 578 
 579   This is based on the Frechet Inception distance, but for an arbitrary
 580   classifier.
 581 
 582   This technique is described in detail in https://arxiv.org/abs/1706.08500.
 583   Given two Gaussian distribution with means m and m_w and covariance matrices
 584   C and C_w, this function calcuates
 585 
 586           |m - m_w|^2 + (sigma + sigma_w - 2(sigma x sigma_w)^(1/2))
 587 
 588   which captures how different the distributions of real images and generated
 589   images (or more accurately, their visual features) are. Note that unlike the
 590   Inception score, this is a true distance and utilizes information about real
 591   world images. In this variant, we compute diagonal-only covariance matrices.
 592   As a result, instead of computing an expensive matrix square root, we can do
 593   something much simpler, and has O(n) vs O(n^2) space complexity.
 594 
 595   Note that when computed using sample means and sample covariance matrices,
 596   Frechet distance is biased. It is more biased for small sample sizes. (e.g.
 597   even if the two distributions are the same, for a small sample size, the
 598   expected Frechet distance is large). It is important to use the same
 599   sample size to compute frechet classifier distance when comparing two
 600   generative models.
 601 
 602   Args:
 603     real_activations: Real images to use to compute Frechet Inception distance.
 604     generated_activations: Generated images to use to compute Frechet Inception
 605       distance.
 606 
 607   Returns:
 608     The diagonal-only Frechet Inception distance. A floating-point scalar of
 609     the same type as the output of the activations.
 610 
 611   Raises:
 612     ValueError: If the shape of the variance and mean vectors are not equal.
 613   """
 614   real_activations.shape.assert_has_rank(2)
 615   generated_activations.shape.assert_has_rank(2)
 616 
 617   activations_dtype = real_activations.dtype
 618   if activations_dtype != dtypes.float64:
 619     real_activations = math_ops.to_double(real_activations)
 620     generated_activations = math_ops.to_double(generated_activations)
 621 
 622   # Compute mean and covariance matrices of activations.
 623   m, var = nn_impl.moments(real_activations, axes=[0])
 624   m_w, var_w = nn_impl.moments(generated_activations, axes=[0])
 625 
 626   actual_shape = var.get_shape()
 627   expected_shape = m.get_shape()
 628 
 629   if actual_shape != expected_shape:
 630     raise ValueError('shape: {} must match expected shape: {}'.format(
 631         actual_shape, expected_shape))
 632 
 633   # Compute the two components of FID.
 634 
 635   # First the covariance component.
 636   # Here, note that trace(A + B) = trace(A) + trace(B)
 637   trace = math_ops.reduce_sum(
 638       (var + var_w) - 2.0 * math_ops.sqrt(math_ops.multiply(var, var_w)))
 639 
 640   # Next the distance between means.
 641   mean = math_ops.reduce_sum(
 642       math_ops.squared_difference(m, m_w))  # Equivalent to L2 but more stable.
 643   dofid = trace + mean
 644   if activations_dtype != dtypes.float64:
 645     dofid = math_ops.cast(dofid, activations_dtype)
 646 
 647   return dofid

 650 def frechet_classifier_distance_from_activations(real_activations,
 651                                                  generated_activations):
 652   """Classifier distance for evaluating a generative model.
 653 
 654   This methods computes the Frechet classifier distance from activations of
 655   real images and generated images. This can be used independently of the
 656   frechet_classifier_distance() method, especially in the case of using large
 657   batches during evaluation where we would like precompute all of the
 658   activations before computing the classifier distance.
 659 
 660   This technique is described in detail in https://arxiv.org/abs/1706.08500.
 661   Given two Gaussian distribution with means m and m_w and covariance matrices
 662   C and C_w, this function calculates
 663 
 664                 |m - m_w|^2 + Tr(C + C_w - 2(C * C_w)^(1/2))
 665 
 666   which captures how different the distributions of real images and generated
 667   images (or more accurately, their visual features) are. Note that unlike the
 668   Inception score, this is a true distance and utilizes information about real
 669   world images.
 670 
 671   Note that when computed using sample means and sample covariance matrices,
 672   Frechet distance is biased. It is more biased for small sample sizes. (e.g.
 673   even if the two distributions are the same, for a small sample size, the
 674   expected Frechet distance is large). It is important to use the same
 675   sample size to compute frechet classifier distance when comparing two
 676   generative models.
 677 
 678   Args:
 679     real_activations: 2D Tensor containing activations of real data. Shape is
 680       [batch_size, activation_size].
 681     generated_activations: 2D Tensor containing activations of generated data.
 682       Shape is [batch_size, activation_size].
 683 
 684   Returns:
 685    The Frechet Inception distance. A floating-point scalar of the same type
 686    as the output of the activations.
 687 
 688   """
 689   real_activations.shape.assert_has_rank(2)
 690   generated_activations.shape.assert_has_rank(2)
 691 
 692   activations_dtype = real_activations.dtype
 693   if activations_dtype != dtypes.float64:
 694     real_activations = math_ops.to_double(real_activations)
 695     generated_activations = math_ops.to_double(generated_activations)
 696 
 697   # Compute mean and covariance matrices of activations.
 698   m = math_ops.reduce_mean(real_activations, 0)
 699   m_w = math_ops.reduce_mean(generated_activations, 0)
 700   num_examples_real = math_ops.to_double(array_ops.shape(real_activations)[0])
 701   num_examples_generated = math_ops.to_double(
 702       array_ops.shape(generated_activations)[0])
 703 
 704   # sigma = (1 / (n - 1)) * (X - mu) (X - mu)^T
 705   real_centered = real_activations - m
 706   sigma = math_ops.matmul(
 707       real_centered, real_centered, transpose_a=True) / (
 708           num_examples_real - 1)
 709 
 710   gen_centered = generated_activations - m_w
 711   sigma_w = math_ops.matmul(
 712       gen_centered, gen_centered, transpose_a=True) / (
 713           num_examples_generated - 1)
 714 
 715   # Find the Tr(sqrt(sigma sigma_w)) component of FID
 716   sqrt_trace_component = trace_sqrt_product(sigma, sigma_w)
 717 
 718   # Compute the two components of FID.
 719 
 720   # First the covariance component.
 721   # Here, note that trace(A + B) = trace(A) + trace(B)
 722   trace = math_ops.trace(sigma + sigma_w) - 2.0 * sqrt_trace_component
 723 
 724   # Next the distance between means.
 725   mean = math_ops.reduce_sum(
 726       math_ops.squared_difference(m, m_w))  # Equivalent to L2 but more stable.
 727   fid = trace + mean
 728   if activations_dtype != dtypes.float64:
 729     fid = math_ops.cast(fid, activations_dtype)
 730 
 731   return fid
