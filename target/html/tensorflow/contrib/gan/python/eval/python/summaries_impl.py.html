  41 def _assert_is_image(data):
  42   data.shape.assert_has_rank(4)
  43   data.shape[1:].assert_is_fully_defined()

  46 def add_gan_model_image_summaries(gan_model, grid_size=4, model_summaries=True):
  47   """Adds image summaries for real and fake images.
  48 
  49   Args:
  50     gan_model: A GANModel tuple.
  51     grid_size: The size of an image grid.
  52     model_summaries: Also add summaries of the model.
  53 
  54   Raises:
  55     ValueError: If real and generated data aren't images.
  56   """
  57   if isinstance(gan_model, namedtuples.CycleGANModel):
  58     raise ValueError(
  59         '`add_gan_model_image_summaries` does not take CycleGANModels. Please '
  60         'use `add_cyclegan_image_summaries` instead.')
  61   _assert_is_image(gan_model.real_data)
  62   _assert_is_image(gan_model.generated_data)
  63 
  64   num_images = grid_size ** 2
  65   real_image_shape = gan_model.real_data.shape.as_list()[1:3]
  66   generated_image_shape = gan_model.generated_data.shape.as_list()[1:3]
  67   real_channels = gan_model.real_data.shape.as_list()[3]
  68   generated_channels = gan_model.generated_data.shape.as_list()[3]
  69 
  70   summary.image(
  71       'real_data',
  72       eval_utils.image_grid(
  73           gan_model.real_data[:num_images],
  74           grid_shape=(grid_size, grid_size),
  75           image_shape=real_image_shape,
  76           num_channels=real_channels),
  77       max_outputs=1)
  78   summary.image(
  79       'generated_data',
  80       eval_utils.image_grid(
  81           gan_model.generated_data[:num_images],
  82           grid_shape=(grid_size, grid_size),
  83           image_shape=generated_image_shape,
  84           num_channels=generated_channels),
  85       max_outputs=1)
  86 
  87   if model_summaries:
  88     add_gan_model_summaries(gan_model)

  91 def add_cyclegan_image_summaries(cyclegan_model):
  92   """Adds image summaries for CycleGAN.
  93 
  94   There are two summaries, one for each generator. The first image is the
  95   generator input, the second is the generator output, and the third is G(F(x)).
  96 
  97   Args:
  98     cyclegan_model: A CycleGANModel tuple.
  99 
 100   Raises:
 101     ValueError: If `cyclegan_model` isn't a CycleGANModel.
 102     ValueError: If generated data, generator inputs, and reconstructions aren't
 103       images.
 104     ValueError: If the generator input, generated data, and reconstructions
 105       aren't all the same size.
 106   """
 107   if not isinstance(cyclegan_model, namedtuples.CycleGANModel):
 108     raise ValueError('`cyclegan_model` was not a CycleGANModel. Instead, was '
 109                      '%s' % type(cyclegan_model))
 110 
 111   _assert_is_image(cyclegan_model.model_x2y.generator_inputs)
 112   _assert_is_image(cyclegan_model.model_x2y.generated_data)
 113   _assert_is_image(cyclegan_model.reconstructed_x)
 114   _assert_is_image(cyclegan_model.model_y2x.generator_inputs)
 115   _assert_is_image(cyclegan_model.model_y2x.generated_data)
 116   _assert_is_image(cyclegan_model.reconstructed_y)
 117 
 118   def _add_comparison_summary(gan_model, reconstructions):
 119     image_list = (array_ops.unstack(gan_model.generator_inputs[:1]) +
 120                   array_ops.unstack(gan_model.generated_data[:1]) +
 121                   array_ops.unstack(reconstructions[:1]))
 122     summary.image(
 123         'image_comparison', eval_utils.image_reshaper(
 124             image_list, num_cols=len(image_list)), max_outputs=1)
 125 
 126   with ops.name_scope('x2y_image_comparison_summaries'):
 127     _add_comparison_summary(
 128         cyclegan_model.model_x2y, cyclegan_model.reconstructed_x)
 129   with ops.name_scope('y2x_image_comparison_summaries'):
 130     _add_comparison_summary(
 131         cyclegan_model.model_y2x, cyclegan_model.reconstructed_y)

 134 def add_image_comparison_summaries(gan_model, num_comparisons=2,
 135                                    display_diffs=False):
 136   """Adds image summaries to compare triplets of images.
 137 
 138   The first image is the generator input, the second is the generator output,
 139   and the third is the real data. This style of comparison is useful for
 140   image translation problems, where the generator input is a corrupted image,
 141   the generator output is the reconstruction, and the real data is the target.
 142 
 143   Args:
 144     gan_model: A GANModel tuple.
 145     num_comparisons: The number of image triplets to display.
 146     display_diffs: Also display the difference between generated and target.
 147 
 148   Raises:
 149     ValueError: If real data, generated data, and generator inputs aren't
 150       images.
 151     ValueError: If the generator input, real, and generated data aren't all the
 152       same size.
 153   """
 154   _assert_is_image(gan_model.generator_inputs)
 155   _assert_is_image(gan_model.generated_data)
 156   _assert_is_image(gan_model.real_data)
 157 
 158   gan_model.generated_data.shape.assert_is_compatible_with(
 159       gan_model.generator_inputs.shape)
 160   gan_model.real_data.shape.assert_is_compatible_with(
 161       gan_model.generated_data.shape)
 162 
 163   image_list = []
 164   image_list.extend(
 165       array_ops.unstack(gan_model.generator_inputs[:num_comparisons]))
 166   image_list.extend(
 167       array_ops.unstack(gan_model.generated_data[:num_comparisons]))
 168   image_list.extend(array_ops.unstack(gan_model.real_data[:num_comparisons]))
 169   if display_diffs:
 170     generated_list = array_ops.unstack(
 171         gan_model.generated_data[:num_comparisons])
 172     real_list = array_ops.unstack(gan_model.real_data[:num_comparisons])
 173     diffs = [
 174         math_ops.abs(math_ops.to_float(generated) - math_ops.to_float(real)) for
 175         generated, real in zip(generated_list, real_list)]
 176     image_list.extend(diffs)
 177 
 178   # Reshape image and display.
 179   summary.image(
 180       'image_comparison',
 181       eval_utils.image_reshaper(image_list, num_cols=num_comparisons),
 182       max_outputs=1)

 185 def add_stargan_image_summaries(stargan_model,
 186                                 num_images=2,
 187                                 display_diffs=False):
 188   """Adds image summaries to see StarGAN image results.
 189 
 190   If display_diffs is True, each image result has `2` rows and `num_domains + 1`
 191   columns.
 192   The first row looks like:
 193     [original_image, transformed_to_domain_0, transformed_to_domain_1, ...]
 194   The second row looks like:
 195     [no_modification_baseline, transformed_to_domain_0-original_image, ...]
 196   If display_diffs is False, only the first row is shown.
 197 
 198   IMPORTANT:
 199     Since the model originally does not transformed the image to every domains,
 200     we will transform them on-the-fly within this function in parallel.
 201 
 202   Args:
 203     stargan_model: A StarGANModel tuple.
 204     num_images: The number of examples/images to be transformed and shown.
 205     display_diffs: Also display the difference between generated and target.
 206 
 207   Raises:
 208     ValueError: If input_data is not images.
 209     ValueError: If input_data_domain_label is not rank 2.
 210     ValueError: If dimension 2 of input_data_domain_label is not fully defined.
 211   """
 212 
 213   _assert_is_image(stargan_model.input_data)
 214   stargan_model.input_data_domain_label.shape.assert_has_rank(2)
 215   stargan_model.input_data_domain_label.shape[1:].assert_is_fully_defined()
 216 
 217   num_domains = stargan_model.input_data_domain_label.get_shape().as_list()[-1]
 218 
 219   def _build_image(image):
 220     """Helper function to create a result for each image on the fly."""
 221 
 222     # Expand the first dimension as batch_size = 1.
 223     images = array_ops.expand_dims(image, axis=0)
 224 
 225     # Tile the image num_domains times, so we can get all transformed together.
 226     images = array_ops.tile(images, [num_domains, 1, 1, 1])
 227 
 228     # Create the targets to 0, 1, 2, ..., num_domains-1.
 229     targets = array_ops.one_hot(list(range(num_domains)), num_domains)
 230 
 231     with variable_scope.variable_scope(
 232         stargan_model.generator_scope, reuse=True):
 233 
 234       # Add the original image.
 235       output_images_list = [image]
 236 
 237       # Generate the image and add to the list.
 238       gen_images = stargan_model.generator_fn(images, targets)
 239       gen_images_list = array_ops.split(gen_images, num_domains)
 240       gen_images_list = [
 241           array_ops.squeeze(img, axis=0) for img in gen_images_list
 242       ]
 243       output_images_list.extend(gen_images_list)
 244 
 245       # Display diffs.
 246       if display_diffs:
 247         diff_images = gen_images - images
 248         diff_images_list = array_ops.split(diff_images, num_domains)
 249         diff_images_list = [
 250             array_ops.squeeze(img, axis=0) for img in diff_images_list
 251         ]
 252         output_images_list.append(array_ops.zeros_like(image))
 253         output_images_list.extend(diff_images_list)
 254 
 255       # Create the final image.
 256       final_image = eval_utils.image_reshaper(
 257           output_images_list, num_cols=num_domains + 1)
 258 
 259     # Reduce the first rank.
 260     return array_ops.squeeze(final_image, axis=0)
 261 
 262   summary.image(
 263       'stargan_image_generation',
 264       functional_ops.map_fn(
 265           _build_image,
 266           stargan_model.input_data[:num_images],
 267           parallel_iterations=num_images,
 268           back_prop=False,
 269           swap_memory=True),
 270       max_outputs=num_images)

 273 def add_gan_model_summaries(gan_model):
 274   """Adds typical GANModel summaries.
 275 
 276   Args:
 277     gan_model: A GANModel tuple.
 278   """
 279   if isinstance(gan_model, namedtuples.CycleGANModel):
 280     with ops.name_scope('cyclegan_x2y_summaries'):
 281       add_gan_model_summaries(gan_model.model_x2y)
 282     with ops.name_scope('cyclegan_y2x_summaries'):
 283       add_gan_model_summaries(gan_model.model_y2x)
 284     return
 285 
 286   with ops.name_scope('generator_variables'):
 287     for var in gan_model.generator_variables:
 288       summary.histogram(var.name, var)
 289   with ops.name_scope('discriminator_variables'):
 290     for var in gan_model.discriminator_variables:
 291       summary.histogram(var.name, var)

 294 def add_regularization_loss_summaries(gan_model):
 295   """Adds summaries for a regularization losses..
 296 
 297   Args:
 298     gan_model: A GANModel tuple.
 299   """
 300   if isinstance(gan_model, namedtuples.CycleGANModel):
 301     with ops.name_scope('cyclegan_x2y_regularization_loss_summaries'):
 302       add_regularization_loss_summaries(gan_model.model_x2y)
 303     with ops.name_scope('cyclegan_y2x_regularization_loss_summaries'):
 304       add_regularization_loss_summaries(gan_model.model_y2x)
 305     return
 306 
 307   if gan_model.generator_scope:
 308     summary.scalar(
 309         'generator_regularization_loss',
 310         loss_util.get_regularization_loss(gan_model.generator_scope.name))
 311   if gan_model.discriminator_scope:
 312     summary.scalar(
 313         'discriminator_regularization_loss',
 314         loss_util.get_regularization_loss(gan_model.discriminator_scope.name))
