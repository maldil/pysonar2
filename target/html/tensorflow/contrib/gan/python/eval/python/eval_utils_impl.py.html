  33 # TODO(joelshor): Make this a special case of `image_reshaper`.
  34 def image_grid(input_tensor, grid_shape, image_shape=(32, 32), num_channels=3):
  35   """Arrange a minibatch of images into a grid to form a single image.
  36 
  37   Args:
  38     input_tensor: Tensor. Minibatch of images to format, either 4D
  39         ([batch size, height, width, num_channels]) or flattened
  40         ([batch size, height * width * num_channels]).
  41     grid_shape: Sequence of int. The shape of the image grid,
  42         formatted as [grid_height, grid_width].
  43     image_shape: Sequence of int. The shape of a single image,
  44         formatted as [image_height, image_width].
  45     num_channels: int. The number of channels in an image.
  46 
  47   Returns:
  48     Tensor representing a single image in which the input images have been
  49     arranged into a grid.
  50 
  51   Raises:
  52     ValueError: The grid shape and minibatch size don't match, or the image
  53         shape and number of channels are incompatible with the input tensor.
  54   """
  55   if grid_shape[0] * grid_shape[1] != int(input_tensor.shape[0]):
  56     raise ValueError("Grid shape %s incompatible with minibatch size %i." %
  57                      (grid_shape, int(input_tensor.shape[0])))
  58   if len(input_tensor.shape) == 2:
  59     num_features = image_shape[0] * image_shape[1] * num_channels
  60     if int(input_tensor.shape[1]) != num_features:
  61       raise ValueError("Image shape and number of channels incompatible with "
  62                        "input tensor.")
  63   elif len(input_tensor.shape) == 4:
  64     if (int(input_tensor.shape[1]) != image_shape[0] or
  65         int(input_tensor.shape[2]) != image_shape[1] or
  66         int(input_tensor.shape[3]) != num_channels):
  67       raise ValueError("Image shape and number of channels incompatible with "
  68                        "input tensor.")
  69   else:
  70     raise ValueError("Unrecognized input tensor format.")
  71   height, width = grid_shape[0] * image_shape[0], grid_shape[1] * image_shape[1]
  72   input_tensor = array_ops.reshape(
  73       input_tensor, tuple(grid_shape) + tuple(image_shape) + (num_channels,))
  74   input_tensor = array_ops.transpose(input_tensor, [0, 1, 3, 2, 4])
  75   input_tensor = array_ops.reshape(
  76       input_tensor, [grid_shape[0], width, image_shape[0], num_channels])
  77   input_tensor = array_ops.transpose(input_tensor, [0, 2, 1, 3])
  78   input_tensor = array_ops.reshape(
  79       input_tensor, [1, height, width, num_channels])
  80   return input_tensor

  83 def _validate_images(images):
  84   for img in images:
  85     img.shape.assert_has_rank(3)
  86     img.shape.assert_is_fully_defined()
  87     if img.shape[-1] not in (1, 3):
  88       raise ValueError("image_reshaper only supports 1 or 3 channel images.")

  92 def image_reshaper(images, num_cols=None):
  93   """A reshaped summary image.
  94 
  95   Returns an image that will contain all elements in the list and will be
  96   laid out in a nearly-square tiling pattern (e.g. 11 images will lead to a
  97   3x4 tiled image).
  98 
  99   Args:
 100     images: Image data to summarize. Can be an RGB or grayscale image, a list of
 101          such images, or a set of RGB images concatenated along the depth
 102          dimension. The shape of each image is assumed to be [batch_size,
 103          height, width, depth].
 104     num_cols: (Optional) If provided, this is the number of columns in the final
 105          output image grid. Otherwise, the number of columns is determined by
 106          the number of images.
 107 
 108   Returns:
 109     A summary image matching the input with automatic tiling if needed.
 110     Output shape is [1, height, width, channels].
 111   """
 112   if isinstance(images, ops.Tensor):
 113     images = array_ops.unstack(images)
 114   _validate_images(images)
 115 
 116   num_images = len(images)
 117   num_columns = (num_cols if num_cols else
 118                  int(math.ceil(math.sqrt(num_images))))
 119   num_rows = int(math.ceil(float(num_images) / num_columns))
 120   rows = [images[x:x+num_columns] for x in range(0, num_images, num_columns)]
 121 
 122   # Add empty image tiles if the last row is incomplete.
 123   num_short = num_rows * num_columns - num_images
 124   assert num_short >= 0 and num_short < num_columns
 125   if num_short > 0:
 126     rows[-1].extend([array_ops.zeros_like(images[-1])] * num_short)
 127 
 128   # Convert each row from a list of tensors to a single tensor.
 129   rows = [array_ops.concat(row, 1) for row in rows]
 130 
 131   # Stack rows vertically.
 132   img = array_ops.concat(rows, 0)
 133 
 134   return array_ops.expand_dims(img, 0)
