  40 def _get_shape(tensor):
  41   tensor_shape = array_ops.shape(tensor)
  42   static_tensor_shape = tensor_util.constant_value(tensor_shape)
  43   return (static_tensor_shape if static_tensor_shape is not None else
  44           tensor_shape)
  45

  47 def condition_tensor(tensor, conditioning):
  48   """Condition the value of a tensor.
  49 
  50   Conditioning scheme based on https://arxiv.org/abs/1609.03499.
  51 
  52   Args:
  53     tensor: A minibatch tensor to be conditioned.
  54     conditioning: A minibatch Tensor of to condition on. Must be 2D, with first
  55       dimension the same as `tensor`.
  56 
  57   Returns:
  58     `tensor` conditioned on `conditioning`.
  59 
  60   Raises:
  61     ValueError: If the non-batch dimensions of `tensor` aren't fully defined.
  62     ValueError: If `conditioning` isn't at least 2D.
  63     ValueError: If the batch dimension for the input Tensors don't match.
  64   """
  65   tensor.shape[1:].assert_is_fully_defined()
  66   num_features = tensor.shape[1:].num_elements()
  67 
  68   mapped_conditioning = layers.linear(
  69       layers.flatten(conditioning), num_features)
  70   if not mapped_conditioning.shape.is_compatible_with(tensor.shape):
  71     mapped_conditioning = array_ops.reshape(
  72         mapped_conditioning, _get_shape(tensor))
  73   return tensor + mapped_conditioning
  74

  76 def _one_hot_to_embedding(one_hot, embedding_size):
  77   """Get a dense embedding vector from a one-hot encoding."""
  78   num_tokens = one_hot.shape[1]
  79   label_id = math_ops.argmax(one_hot, axis=1)
  80   embedding = variable_scope.get_variable(
  81       'embedding', [num_tokens, embedding_size])
  82   return embedding_ops.embedding_lookup(
  83       embedding, label_id, name='token_to_embedding')
  84

  86 def _validate_onehot(one_hot_labels):
  87   one_hot_labels.shape.assert_has_rank(2)
  88   one_hot_labels.shape[1:].assert_is_fully_defined()
  89

  91 def condition_tensor_from_onehot(tensor, one_hot_labels, embedding_size=256):
  92   """Condition a tensor based on a one-hot tensor.
  93 
  94   Conditioning scheme based on https://arxiv.org/abs/1609.03499.
  95 
  96   Args:
  97     tensor: Tensor to be conditioned.
  98     one_hot_labels: A Tensor of one-hot labels. Shape is
  99       [batch_size, num_classes].
 100     embedding_size: The size of the class embedding.
 101 
 102   Returns:
 103     `tensor` conditioned on `one_hot_labels`.
 104 
 105   Raises:
 106     ValueError: `one_hot_labels` isn't 2D, if non-batch dimensions aren't
 107       fully defined, or if batch sizes don't match.
 108   """
 109   _validate_onehot(one_hot_labels)
 110 
 111   conditioning = _one_hot_to_embedding(one_hot_labels, embedding_size)
 112   return condition_tensor(tensor, conditioning)
