  41 def _static_or_dynamic_batch_size(tensor, batch_axis):
  42   """Returns the static or dynamic batch size."""
  43   batch_size = array_ops.shape(tensor)[batch_axis]
  44   static_batch_size = tensor_util.constant_value(batch_size)
  45   return static_batch_size or batch_size

  48 def _statistics(x, axes):
  49   """Calculate the mean and mean square of `x`.
  50 
  51   Modified from the implementation of `tf.nn.moments`.
  52 
  53   Args:
  54     x: A `Tensor`.
  55     axes: Array of ints.  Axes along which to compute mean and
  56       variance.
  57 
  58   Returns:
  59     Two `Tensor` objects: `mean` and `square mean`.
  60   """
  61   # The dynamic range of fp16 is too limited to support the collection of
  62   # sufficient statistics. As a workaround we simply perform the operations
  63   # on 32-bit floats before converting the mean and variance back to fp16
  64   y = math_ops.cast(x, dtypes.float32) if x.dtype == dtypes.float16 else x
  65 
  66   # Compute true mean while keeping the dims for proper broadcasting.
  67   shift = array_ops.stop_gradient(math_ops.reduce_mean(y, axes, keepdims=True))
  68 
  69   shifted_mean = math_ops.reduce_mean(y - shift, axes, keepdims=True)
  70   mean = shifted_mean + shift
  71   mean_squared = math_ops.reduce_mean(math_ops.square(y), axes, keepdims=True)
  72 
  73   mean = array_ops.squeeze(mean, axes)
  74   mean_squared = array_ops.squeeze(mean_squared, axes)
  75   if x.dtype == dtypes.float16:
  76     return (math_ops.cast(mean, dtypes.float16),
  77             math_ops.cast(mean_squared, dtypes.float16))
  78   else:
  79     return (mean, mean_squared)
  80

  82 def _validate_init_input_and_get_axis(reference_batch, axis):
  83   """Validate input and return the used axis value."""
  84   if reference_batch.shape.ndims is None:
  85     raise ValueError('`reference_batch` has unknown dimensions.')
  86 
  87   ndims = reference_batch.shape.ndims
  88   if axis < 0:
  89     used_axis = ndims + axis
  90   else:
  91     used_axis = axis
  92   if used_axis < 0 or used_axis >= ndims:
  93     raise ValueError('Value of `axis` argument ' + str(used_axis) +
  94                      ' is out of range for input with rank ' + str(ndims))
  95   return used_axis

  98 def _validate_call_input(tensor_list, batch_dim):
  99   """Verifies that tensor shapes are compatible, except for `batch_dim`."""
 100   def _get_shape(tensor):
 101     shape = tensor.shape.as_list()
 102     del shape[batch_dim]
 103     return shape
 104   base_shape = tensor_shape.TensorShape(_get_shape(tensor_list[0]))
 105   for tensor in tensor_list:
 106     base_shape.assert_is_compatible_with(_get_shape(tensor))

 109 class VBN(object):
 110   """A class to perform virtual batch normalization.
 111 
 112   This technique was first introduced in `Improved Techniques for Training GANs`
 113   (Salimans et al, https://arxiv.org/abs/1606.03498). Instead of using batch
 114   normalization on a minibatch, it fixes a reference subset of the data to use
 115   for calculating normalization statistics.
 116 
 117   To do this, we calculate the reference batch mean and mean square, and modify
 118   those statistics for each example. We use mean square instead of variance,
 119   since it is linear.
 120 
 121   Note that if `center` or `scale` variables are created, they are shared
 122   between all calls to this object.
 123 
 124   The `__init__` API is intended to mimic `tf.layers.batch_normalization` as
 125   closely as possible.
 126   """
 127 
 128   def __init__(self,
 129                reference_batch,
 130                axis=-1,
 131                epsilon=1e-3,
 132                center=True,
 133                scale=True,
 134                beta_initializer=init_ops.zeros_initializer(),
 135                gamma_initializer=init_ops.ones_initializer(),
 136                beta_regularizer=None,
 137                gamma_regularizer=None,
 138                trainable=True,
 139                name=None,
 140                batch_axis=0):
 141     """Initialize virtual batch normalization object.
 142 
 143     We precompute the 'mean' and 'mean squared' of the reference batch, so that
 144     `__call__` is efficient. This means that the axis must be supplied when the
 145     object is created, not when it is called.
 146 
 147     We precompute 'square mean' instead of 'variance', because the square mean
 148     can be easily adjusted on a per-example basis.
 149 
 150     Args:
 151       reference_batch: A minibatch tensors. This will form the reference data
 152         from which the normalization statistics are calculated. See
 153         https://arxiv.org/abs/1606.03498 for more details.
 154       axis: Integer, the axis that should be normalized (typically the features
 155         axis). For instance, after a `Convolution2D` layer with
 156         `data_format="channels_first"`, set `axis=1` in `BatchNormalization`.
 157       epsilon: Small float added to variance to avoid dividing by zero.
 158       center: If True, add offset of `beta` to normalized tensor. If False,
 159         `beta` is ignored.
 160       scale: If True, multiply by `gamma`. If False, `gamma` is
 161         not used. When the next layer is linear (also e.g. `nn.relu`), this can
 162         be disabled since the scaling can be done by the next layer.
 163       beta_initializer: Initializer for the beta weight.
 164       gamma_initializer: Initializer for the gamma weight.
 165       beta_regularizer: Optional regularizer for the beta weight.
 166       gamma_regularizer: Optional regularizer for the gamma weight.
 167       trainable: Boolean, if `True` also add variables to the graph collection
 168         `GraphKeys.TRAINABLE_VARIABLES` (see tf.Variable).
 169       name: String, the name of the ops.
 170       batch_axis: The axis of the batch dimension. This dimension is treated
 171         differently in `virtual batch normalization` vs `batch normalization`.
 172 
 173     Raises:
 174       ValueError: If `reference_batch` has unknown dimensions at graph
 175         construction.
 176       ValueError: If `batch_axis` is the same as `axis`.
 177     """
 178     axis = _validate_init_input_and_get_axis(reference_batch, axis)
 179     self._epsilon = epsilon
 180     self._beta = 0
 181     self._gamma = 1
 182     self._batch_axis = _validate_init_input_and_get_axis(
 183         reference_batch, batch_axis)
 184 
 185     if axis == self._batch_axis:
 186       raise ValueError('`axis` and `batch_axis` cannot be the same.')
 187 
 188     with variable_scope.variable_scope(name, 'VBN',
 189                                        values=[reference_batch]) as self._vs:
 190       self._reference_batch = reference_batch
 191 
 192       # Calculate important shapes:
 193       #  1) Reduction axes for the reference batch
 194       #  2) Broadcast shape, if necessary
 195       #  3) Reduction axes for the virtual batchnormed batch
 196       #  4) Shape for optional parameters
 197       input_shape = self._reference_batch.shape
 198       ndims = input_shape.ndims
 199       reduction_axes = list(range(ndims))
 200       del reduction_axes[axis]
 201 
 202       self._broadcast_shape = [1] * len(input_shape)
 203       self._broadcast_shape[axis] = input_shape[axis].value
 204 
 205       self._example_reduction_axes = list(range(ndims))
 206       del self._example_reduction_axes[max(axis, self._batch_axis)]
 207       del self._example_reduction_axes[min(axis, self._batch_axis)]
 208 
 209       params_shape = self._reference_batch.shape[axis]
 210 
 211       # Determines whether broadcasting is needed. This is slightly different
 212       # than in the `nn.batch_normalization` case, due to `batch_dim`.
 213       self._needs_broadcasting = (
 214           sorted(self._example_reduction_axes) != list(range(ndims))[:-2])
 215 
 216       # Calculate the sufficient statistics for the reference batch in a way
 217       # that can be easily modified by additional examples.
 218       self._ref_mean, self._ref_mean_squares = _statistics(
 219           self._reference_batch, reduction_axes)
 220       self._ref_variance = (self._ref_mean_squares -
 221                             math_ops.square(self._ref_mean))
 222 
 223       # Virtual batch normalization uses a weighted average between example
 224       # statistics and the reference batch statistics.
 225       ref_batch_size = _static_or_dynamic_batch_size(
 226           self._reference_batch, self._batch_axis)
 227       self._example_weight = 1. / (math_ops.to_float(ref_batch_size) + 1.)
 228       self._ref_weight = 1. - self._example_weight
 229 
 230       # Make the variables, if necessary.
 231       if center:
 232         self._beta = variable_scope.get_variable(
 233             name='beta',
 234             shape=(params_shape,),
 235             initializer=beta_initializer,
 236             regularizer=beta_regularizer,
 237             trainable=trainable)
 238       if scale:
 239         self._gamma = variable_scope.get_variable(
 240             name='gamma',
 241             shape=(params_shape,),
 242             initializer=gamma_initializer,
 243             regularizer=gamma_regularizer,
 244             trainable=trainable)
 245 
 246   def _virtual_statistics(self, inputs, reduction_axes):
 247     """Compute the statistics needed for virtual batch normalization."""
 248     cur_mean, cur_mean_sq = _statistics(inputs, reduction_axes)
 249     vb_mean = (self._example_weight * cur_mean +
 250                self._ref_weight * self._ref_mean)
 251     vb_mean_sq = (self._example_weight * cur_mean_sq +
 252                   self._ref_weight * self._ref_mean_squares)
 253     return (vb_mean, vb_mean_sq)
 254 
 255   def _broadcast(self, v, broadcast_shape=None):
 256     # The exact broadcast shape depends on the current batch, not the reference
 257     # batch, unless we're calculating the batch normalization of the reference
 258     # batch.
 259     b_shape = broadcast_shape or self._broadcast_shape
 260     if self._needs_broadcasting and v is not None:
 261       return array_ops.reshape(v, b_shape)
 262     return v
 263 
 264   def reference_batch_normalization(self):
 265     """Return the reference batch, but batch normalized."""
 266     with ops.name_scope(self._vs.name):
 267       return nn.batch_normalization(self._reference_batch,
 268                                     self._broadcast(self._ref_mean),
 269                                     self._broadcast(self._ref_variance),
 270                                     self._broadcast(self._beta),
 271                                     self._broadcast(self._gamma),
 272                                     self._epsilon)
 273 
 274   def __call__(self, inputs):
 275     """Run virtual batch normalization on inputs.
 276 
 277     Args:
 278       inputs: Tensor input.
 279 
 280     Returns:
 281        A virtual batch normalized version of `inputs`.
 282 
 283     Raises:
 284        ValueError: If `inputs` shape isn't compatible with the reference batch.
 285     """
 286     _validate_call_input([inputs, self._reference_batch], self._batch_axis)
 287 
 288     with ops.name_scope(self._vs.name, values=[inputs, self._reference_batch]):
 289       # Calculate the statistics on the current input on a per-example basis.
 290       vb_mean, vb_mean_sq = self._virtual_statistics(
 291           inputs, self._example_reduction_axes)
 292       vb_variance = vb_mean_sq - math_ops.square(vb_mean)
 293 
 294       # The exact broadcast shape of the input statistic Tensors depends on the
 295       # current batch, not the reference batch. The parameter broadcast shape
 296       # is independent of the shape of the input statistic Tensor dimensions.
 297       b_shape = self._broadcast_shape[:]  # deep copy
 298       b_shape[self._batch_axis] = _static_or_dynamic_batch_size(
 299           inputs, self._batch_axis)
 300       return nn.batch_normalization(
 301           inputs,
 302           self._broadcast(vb_mean, b_shape),
 303           self._broadcast(vb_variance, b_shape),
 304           self._broadcast(self._beta, self._broadcast_shape),
 305           self._broadcast(self._gamma, self._broadcast_shape),
 306           self._epsilon)
