  15 """A tensor pool stores values from an input tensor and returns a stored one.
  16 
  17 We use this to keep a history of values created by a generator, such that
  18 a discriminator can randomly be trained on some older samples, not just the
  19 current one. This can help to not let the discriminator get too far ahead of the
  20 generator and also to keep the system from oscillating, if the discriminator
  21 forgets too fast what past samples from the generator looked like.
  22 
  23 See the following papers for more details.
  24 1) `Learning from simulated and unsupervised images through adversarial
  25     training` (https://arxiv.org/abs/1612.07828).
  26 2) `Unpaired Image-to-Image Translation using Cycle-Consistent Adversarial
  27     Networks` (https://arxiv.org/abs/1703.10593).
  28 """
  29 from __future__ import absolute_import
  30 from __future__ import division
  31 from __future__ import print_function
  32 
  33 from tensorflow.python.framework import dtypes
  34 from tensorflow.python.framework import ops
  35 from tensorflow.python.ops import array_ops
  36 from tensorflow.python.ops import control_flow_ops
  37 from tensorflow.python.ops import data_flow_ops
  38 from tensorflow.python.ops import random_ops
  39 from tensorflow.python.util import nest
  40 
  41 __all__ = [
  42     'tensor_pool',
  43 ]

  46 def _to_list(x):
  47   return [x] if isinstance(x, ops.Tensor) else list(x)
  48 
  49 
  50 def tensor_pool(input_values,
  51                 pool_size=50,
  52                 pooling_probability=0.5,
  53                 name='tensor_pool'):
  54   """Queue storing input values and returning random previously stored ones.
  55 
  56   Every time the returned `output_value` is evaluated, `input_value` is
  57   evaluated and its value either directly returned (with
  58   `1-pooling_probability`) or stored in the pool and a random one of the samples
  59   currently in the pool is popped and returned. As long as the pool in not fully
  60   filled, the input_value is always directly returned, as well as stored in the
  61   pool. Note during inference / testing, it may be appropriate to set
  62   `pool_size` = 0 or `pooling_probability` = 0.
  63 
  64   Args:
  65     input_values: An arbitrarily nested structure of `tf.Tensors`, from which to
  66       read values to be pooled.
  67     pool_size: An integer specifying the maximum size of the pool. Defaults to
  68       50.
  69     pooling_probability: A float `Tensor` specifying the probability of getting
  70       a value from the pool, as opposed to just the current input.
  71     name: A string prefix for the name scope for all tensorflow ops.
  72 
  73   Returns:
  74     A nested structure of `Tensor` objects with the same structure as
  75     `input_values`. With the given probability, the Tensor values are either the
  76     same as in `input_values` or a randomly chosen sample that was previously
  77     inserted in the pool.
  78 
  79   Raises:
  80     ValueError: If `pool_size` is negative.
  81   """
  82   pool_size = int(pool_size)
  83   if pool_size < 0:
  84     raise ValueError('`pool_size` is negative.')
  85   elif pool_size == 0:
  86     return input_values
  87 
  88   original_input_values = input_values
  89   input_values = nest.flatten(input_values)
  90 
  91   with ops.name_scope('{}_pool_queue'.format(name),
  92                       values=input_values + [pooling_probability]):
  93     pool_queue = data_flow_ops.RandomShuffleQueue(
  94         capacity=pool_size,
  95         min_after_dequeue=0,
  96         dtypes=[v.dtype for v in input_values],
  97         shapes=None)
  98 
  99     # In pseudo code this code does the following:
 100     # if not pool_full:
 101     #   enqueue(input_values)
 102     #   return input_values
 103     # else
 104     #   dequeue_values = dequeue_random_sample()
 105     #   enqueue(input_values)
 106     #   if rand() < pooling_probability:
 107     #     return dequeue_values
 108     #   else
 109     #     return input_values
 110 
 111     def _get_input_value_pooled():
 112       enqueue_op = pool_queue.enqueue(input_values)
 113       with ops.control_dependencies([enqueue_op]):
 114         return [array_ops.identity(v) for v in input_values]
 115 
 116     def _get_random_pool_value_and_enqueue_input():
 117       dequeue_values = _to_list(pool_queue.dequeue())
 118       with ops.control_dependencies(dequeue_values):
 119         enqueue_op = pool_queue.enqueue(input_values)
 120         with ops.control_dependencies([enqueue_op]):
 121           prob = random_ops.random_uniform(
 122               (), dtype=dtypes.float32) < pooling_probability
 123           return control_flow_ops.cond(prob, lambda: dequeue_values,
 124                                        lambda: input_values)
 125 
 126     output_values = _to_list(control_flow_ops.cond(
 127         pool_queue.size() < pool_size, _get_input_value_pooled,
 128         _get_random_pool_value_and_enqueue_input))
 129 
 130     # Make sure that the shape of `output_value` is set.
 131     for input_value, output_value in zip(input_values, output_values):
 132       output_value.set_shape(input_value.shape)
 133 
 134   return nest.pack_sequence_as(original_input_values, output_values)
