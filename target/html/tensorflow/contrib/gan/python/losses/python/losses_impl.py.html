  15 """Losses that are useful for training GANs.
  16 
  17 The losses belong to two main groups, but there are others that do not:
  18 1) xxxxx_generator_loss
  19 2) xxxxx_discriminator_loss
  20 
  21 Example:
  22 1) wasserstein_generator_loss
  23 2) wasserstein_discriminator_loss
  24 
  25 Other example:
  26 wasserstein_gradient_penalty
  27 
  28 All losses must be able to accept 1D or 2D Tensors, so as to be compatible with
  29 patchGAN style losses (https://arxiv.org/abs/1611.07004).
  30 
  31 To make these losses usable in the TFGAN framework, please create a tuple
  32 version of the losses with `losses_utils.py`.
  33 """

  74 # Wasserstein losses from `Wasserstein GAN` (https://arxiv.org/abs/1701.07875).
  75 def wasserstein_generator_loss(
  76     discriminator_gen_outputs,
  77     weights=1.0,
  78     scope=None,
  79     loss_collection=ops.GraphKeys.LOSSES,
  80     reduction=losses.Reduction.SUM_BY_NONZERO_WEIGHTS,
  81     add_summaries=False):
  82   """Wasserstein generator loss for GANs.
  83 
  84   See `Wasserstein GAN` (https://arxiv.org/abs/1701.07875) for more details.
  85 
  86   Args:
  87     discriminator_gen_outputs: Discriminator output on generated data. Expected
  88       to be in the range of (-inf, inf).
  89     weights: Optional `Tensor` whose rank is either 0, or the same rank as
  90       `discriminator_gen_outputs`, and must be broadcastable to
  91       `discriminator_gen_outputs` (i.e., all dimensions must be either `1`, or
  92       the same as the corresponding dimension).
  93     scope: The scope for the operations performed in computing the loss.
  94     loss_collection: collection to which this loss will be added.
  95     reduction: A `tf.losses.Reduction` to apply to loss.
  96     add_summaries: Whether or not to add detailed summaries for the loss.
  97 
  98   Returns:
  99     A loss Tensor. The shape depends on `reduction`.
 100   """
 101   with ops.name_scope(scope, 'generator_wasserstein_loss', (
 102       discriminator_gen_outputs, weights)) as scope:
 103     discriminator_gen_outputs = math_ops.to_float(discriminator_gen_outputs)
 104 
 105     loss = - discriminator_gen_outputs
 106     loss = losses.compute_weighted_loss(
 107         loss, weights, scope, loss_collection, reduction)
 108 
 109     if add_summaries:
 110       summary.scalar('generator_wass_loss', loss)
 111 
 112   return loss

 115 def wasserstein_discriminator_loss(
 116     discriminator_real_outputs,
 117     discriminator_gen_outputs,
 118     real_weights=1.0,
 119     generated_weights=1.0,
 120     scope=None,
 121     loss_collection=ops.GraphKeys.LOSSES,
 122     reduction=losses.Reduction.SUM_BY_NONZERO_WEIGHTS,
 123     add_summaries=False):
 124   """Wasserstein discriminator loss for GANs.
 125 
 126   See `Wasserstein GAN` (https://arxiv.org/abs/1701.07875) for more details.
 127 
 128   Args:
 129     discriminator_real_outputs: Discriminator output on real data.
 130     discriminator_gen_outputs: Discriminator output on generated data. Expected
 131       to be in the range of (-inf, inf).
 132     real_weights: Optional `Tensor` whose rank is either 0, or the same rank as
 133       `discriminator_real_outputs`, and must be broadcastable to
 134       `discriminator_real_outputs` (i.e., all dimensions must be either `1`, or
 135       the same as the corresponding dimension).
 136     generated_weights: Same as `real_weights`, but for
 137       `discriminator_gen_outputs`.
 138     scope: The scope for the operations performed in computing the loss.
 139     loss_collection: collection to which this loss will be added.
 140     reduction: A `tf.losses.Reduction` to apply to loss.
 141     add_summaries: Whether or not to add summaries for the loss.
 142 
 143   Returns:
 144     A loss Tensor. The shape depends on `reduction`.
 145   """
 146   with ops.name_scope(scope, 'discriminator_wasserstein_loss', (
 147       discriminator_real_outputs, discriminator_gen_outputs, real_weights,
 148       generated_weights)) as scope:
 149     discriminator_real_outputs = math_ops.to_float(discriminator_real_outputs)
 150     discriminator_gen_outputs = math_ops.to_float(discriminator_gen_outputs)
 151     discriminator_real_outputs.shape.assert_is_compatible_with(
 152         discriminator_gen_outputs.shape)
 153 
 154     loss_on_generated = losses.compute_weighted_loss(
 155         discriminator_gen_outputs, generated_weights, scope,
 156         loss_collection=None, reduction=reduction)
 157     loss_on_real = losses.compute_weighted_loss(
 158         discriminator_real_outputs, real_weights, scope, loss_collection=None,
 159         reduction=reduction)
 160     loss = loss_on_generated - loss_on_real
 161     util.add_loss(loss, loss_collection)
 162 
 163     if add_summaries:
 164       summary.scalar('discriminator_gen_wass_loss', loss_on_generated)
 165       summary.scalar('discriminator_real_wass_loss', loss_on_real)
 166       summary.scalar('discriminator_wass_loss', loss)
 167 
 168   return loss

 171 # ACGAN losses from `Conditional Image Synthesis With Auxiliary Classifier GANs`
 172 # (https://arxiv.org/abs/1610.09585).
 173 def acgan_discriminator_loss(
 174     discriminator_real_classification_logits,
 175     discriminator_gen_classification_logits,
 176     one_hot_labels,
 177     label_smoothing=0.0,
 178     real_weights=1.0,
 179     generated_weights=1.0,
 180     scope=None,
 181     loss_collection=ops.GraphKeys.LOSSES,
 182     reduction=losses.Reduction.SUM_BY_NONZERO_WEIGHTS,
 183     add_summaries=False):
 184   """ACGAN loss for the discriminator.
 185 
 186   The ACGAN loss adds a classification loss to the conditional discriminator.
 187   Therefore, the discriminator must output a tuple consisting of
 188     (1) the real/fake prediction and
 189     (2) the logits for the classification (usually the last conv layer,
 190         flattened).
 191 
 192   For more details:
 193     ACGAN: https://arxiv.org/abs/1610.09585
 194 
 195   Args:
 196     discriminator_real_classification_logits: Classification logits for real
 197       data.
 198     discriminator_gen_classification_logits: Classification logits for generated
 199       data.
 200     one_hot_labels: A Tensor holding one-hot labels for the batch.
 201     label_smoothing: A float in [0, 1]. If greater than 0, smooth the labels for
 202       "discriminator on real data" as suggested in
 203       https://arxiv.org/pdf/1701.00160
 204     real_weights: Optional `Tensor` whose rank is either 0, or the same rank as
 205       `discriminator_real_outputs`, and must be broadcastable to
 206       `discriminator_real_outputs` (i.e., all dimensions must be either `1`, or
 207       the same as the corresponding dimension).
 208     generated_weights: Same as `real_weights`, but for
 209       `discriminator_gen_classification_logits`.
 210     scope: The scope for the operations performed in computing the loss.
 211     loss_collection: collection to which this loss will be added.
 212     reduction: A `tf.losses.Reduction` to apply to loss.
 213     add_summaries: Whether or not to add summaries for the loss.
 214 
 215   Returns:
 216     A loss Tensor. Shape depends on `reduction`.
 217 
 218   Raises:
 219     TypeError: If the discriminator does not output a tuple.
 220   """
 221   with ops.name_scope(
 222       scope, 'acgan_discriminator_loss',
 223       (discriminator_real_classification_logits,
 224        discriminator_gen_classification_logits, one_hot_labels)) as scope:
 225     loss_on_generated = losses.softmax_cross_entropy(
 226         one_hot_labels, discriminator_gen_classification_logits,
 227         weights=generated_weights, scope=scope, loss_collection=None,
 228         reduction=reduction)
 229     loss_on_real = losses.softmax_cross_entropy(
 230         one_hot_labels, discriminator_real_classification_logits,
 231         weights=real_weights, label_smoothing=label_smoothing, scope=scope,
 232         loss_collection=None, reduction=reduction)
 233     loss = loss_on_generated + loss_on_real
 234     util.add_loss(loss, loss_collection)
 235 
 236     if add_summaries:
 237       summary.scalar('discriminator_gen_ac_loss', loss_on_generated)
 238       summary.scalar('discriminator_real_ac_loss', loss_on_real)
 239       summary.scalar('discriminator_ac_loss', loss)
 240 
 241   return loss

 301 def wasserstein_gradient_penalty(
 302     real_data,
 303     generated_data,
 304     generator_inputs,
 305     discriminator_fn,
 306     discriminator_scope,
 307     epsilon=1e-10,
 308     target=1.0,
 309     one_sided=False,
 310     weights=1.0,
 311     scope=None,
 312     loss_collection=ops.GraphKeys.LOSSES,
 313     reduction=losses.Reduction.SUM_BY_NONZERO_WEIGHTS,
 314     add_summaries=False):
 315   """The gradient penalty for the Wasserstein discriminator loss.
 316 
 317   See `Improved Training of Wasserstein GANs`
 318   (https://arxiv.org/abs/1704.00028) for more details.
 319 
 320   Args:
 321     real_data: Real data.
 322     generated_data: Output of the generator.
 323     generator_inputs: Exact argument to pass to the generator, which is used
 324       as optional conditioning to the discriminator.
 325     discriminator_fn: A discriminator function that conforms to TFGAN API.
 326     discriminator_scope: If not `None`, reuse discriminators from this scope.
 327     epsilon: A small positive number added for numerical stability when
 328       computing the gradient norm.
 329     target: Optional Python number or `Tensor` indicating the target value of
 330       gradient norm. Defaults to 1.0.
 331     one_sided: If `True`, penalty proposed in https://arxiv.org/abs/1709.08894
 332       is used. Defaults to `False`.
 333     weights: Optional `Tensor` whose rank is either 0, or the same rank as
 334       `real_data` and `generated_data`, and must be broadcastable to
 335       them (i.e., all dimensions must be either `1`, or the same as the
 336       corresponding dimension).
 337     scope: The scope for the operations performed in computing the loss.
 338     loss_collection: collection to which this loss will be added.
 339     reduction: A `tf.losses.Reduction` to apply to loss.
 340     add_summaries: Whether or not to add summaries for the loss.
 341 
 342   Returns:
 343     A loss Tensor. The shape depends on `reduction`.
 344 
 345   Raises:
 346     ValueError: If the rank of data Tensors is unknown.
 347   """
 348   with ops.name_scope(scope, 'wasserstein_gradient_penalty',
 349                       (real_data, generated_data)) as scope:
 350     real_data = ops.convert_to_tensor(real_data)
 351     generated_data = ops.convert_to_tensor(generated_data)
 352     if real_data.shape.ndims is None:
 353       raise ValueError('`real_data` can\'t have unknown rank.')
 354     if generated_data.shape.ndims is None:
 355       raise ValueError('`generated_data` can\'t have unknown rank.')
 356 
 357     differences = generated_data - real_data
 358     batch_size = differences.shape[0].value or array_ops.shape(differences)[0]
 359     alpha_shape = [batch_size] + [1] * (differences.shape.ndims - 1)
 360     alpha = random_ops.random_uniform(shape=alpha_shape)
 361     interpolates = real_data + (alpha * differences)
 362 
 363     with ops.name_scope(None):  # Clear scope so update ops are added properly.
 364       # Reuse variables if variables already exists.
 365       with variable_scope.variable_scope(discriminator_scope, 'gpenalty_dscope',
 366                                          reuse=variable_scope.AUTO_REUSE):
 367         disc_interpolates = discriminator_fn(interpolates, generator_inputs)
 368 
 369     if isinstance(disc_interpolates, tuple):
 370       # ACGAN case: disc outputs more than one tensor
 371       disc_interpolates = disc_interpolates[0]
 372 
 373     gradients = gradients_impl.gradients(disc_interpolates, interpolates)[0]
 374     gradient_squares = math_ops.reduce_sum(
 375         math_ops.square(gradients), axis=list(range(1, gradients.shape.ndims)))
 376     # Propagate shape information, if possible.
 377     if isinstance(batch_size, int):
 378       gradient_squares.set_shape([
 379           batch_size] + gradient_squares.shape.as_list()[1:])
 380     # For numerical stability, add epsilon to the sum before taking the square
 381     # root. Note tf.norm does not add epsilon.
 382     slopes = math_ops.sqrt(gradient_squares + epsilon)
 383     penalties = slopes / target - 1.0
 384     if one_sided:
 385       penalties = math_ops.maximum(0., penalties)
 386     penalties_squared = math_ops.square(penalties)
 387     penalty = losses.compute_weighted_loss(
 388         penalties_squared, weights, scope=scope,
 389         loss_collection=loss_collection, reduction=reduction)
 390 
 391     if add_summaries:
 392       summary.scalar('gradient_penalty_loss', penalty)
 393 
 394     return penalty

 401 def minimax_discriminator_loss(
 402     discriminator_real_outputs,
 403     discriminator_gen_outputs,
 404     label_smoothing=0.25,
 405     real_weights=1.0,
 406     generated_weights=1.0,
 407     scope=None,
 408     loss_collection=ops.GraphKeys.LOSSES,
 409     reduction=losses.Reduction.SUM_BY_NONZERO_WEIGHTS,
 410     add_summaries=False):
 411   """Original minimax discriminator loss for GANs, with label smoothing.
 412 
 413   Note that the authors don't recommend using this loss. A more practically
 414   useful loss is `modified_discriminator_loss`.
 415 
 416   L = - real_weights * log(sigmoid(D(x)))
 417       - generated_weights * log(1 - sigmoid(D(G(z))))
 418 
 419   See `Generative Adversarial Nets` (https://arxiv.org/abs/1406.2661) for more
 420   details.
 421 
 422   Args:
 423     discriminator_real_outputs: Discriminator output on real data.
 424     discriminator_gen_outputs: Discriminator output on generated data. Expected
 425       to be in the range of (-inf, inf).
 426     label_smoothing: The amount of smoothing for positive labels. This technique
 427       is taken from `Improved Techniques for Training GANs`
 428       (https://arxiv.org/abs/1606.03498). `0.0` means no smoothing.
 429     real_weights: Optional `Tensor` whose rank is either 0, or the same rank as
 430       `real_data`, and must be broadcastable to `real_data` (i.e., all
 431       dimensions must be either `1`, or the same as the corresponding
 432       dimension).
 433     generated_weights: Same as `real_weights`, but for `generated_data`.
 434     scope: The scope for the operations performed in computing the loss.
 435     loss_collection: collection to which this loss will be added.
 436     reduction: A `tf.losses.Reduction` to apply to loss.
 437     add_summaries: Whether or not to add summaries for the loss.
 438 
 439   Returns:
 440     A loss Tensor. The shape depends on `reduction`.
 441   """
 442   with ops.name_scope(scope, 'discriminator_minimax_loss', (
 443       discriminator_real_outputs, discriminator_gen_outputs, real_weights,
 444       generated_weights, label_smoothing)) as scope:
 445 
 446     # -log((1 - label_smoothing) - sigmoid(D(x)))
 447     loss_on_real = losses.sigmoid_cross_entropy(
 448         array_ops.ones_like(discriminator_real_outputs),
 449         discriminator_real_outputs, real_weights, label_smoothing, scope,
 450         loss_collection=None, reduction=reduction)
 451     # -log(- sigmoid(D(G(x))))
 452     loss_on_generated = losses.sigmoid_cross_entropy(
 453         array_ops.zeros_like(discriminator_gen_outputs),
 454         discriminator_gen_outputs, generated_weights, scope=scope,
 455         loss_collection=None, reduction=reduction)
 456 
 457     loss = loss_on_real + loss_on_generated
 458     util.add_loss(loss, loss_collection)
 459 
 460     if add_summaries:
 461       summary.scalar('discriminator_gen_minimax_loss', loss_on_generated)
 462       summary.scalar('discriminator_real_minimax_loss', loss_on_real)
 463       summary.scalar('discriminator_minimax_loss', loss)
 464 
 465   return loss

 468 def minimax_generator_loss(
 469     discriminator_gen_outputs,
 470     label_smoothing=0.0,
 471     weights=1.0,
 472     scope=None,
 473     loss_collection=ops.GraphKeys.LOSSES,
 474     reduction=losses.Reduction.SUM_BY_NONZERO_WEIGHTS,
 475     add_summaries=False):
 476   """Original minimax generator loss for GANs.
 477 
 478   Note that the authors don't recommend using this loss. A more practically
 479   useful loss is `modified_generator_loss`.
 480 
 481   L = log(sigmoid(D(x))) + log(1 - sigmoid(D(G(z))))
 482 
 483   See `Generative Adversarial Nets` (https://arxiv.org/abs/1406.2661) for more
 484   details.
 485 
 486   Args:
 487     discriminator_gen_outputs: Discriminator output on generated data. Expected
 488       to be in the range of (-inf, inf).
 489     label_smoothing: The amount of smoothing for positive labels. This technique
 490       is taken from `Improved Techniques for Training GANs`
 491       (https://arxiv.org/abs/1606.03498). `0.0` means no smoothing.
 492     weights: Optional `Tensor` whose rank is either 0, or the same rank as
 493       `discriminator_gen_outputs`, and must be broadcastable to
 494       `discriminator_gen_outputs` (i.e., all dimensions must be either `1`, or
 495       the same as the corresponding dimension).
 496     scope: The scope for the operations performed in computing the loss.
 497     loss_collection: collection to which this loss will be added.
 498     reduction: A `tf.losses.Reduction` to apply to loss.
 499     add_summaries: Whether or not to add summaries for the loss.
 500 
 501   Returns:
 502     A loss Tensor. The shape depends on `reduction`.
 503   """
 504   with ops.name_scope(scope, 'generator_minimax_loss') as scope:
 505     loss = - minimax_discriminator_loss(
 506         array_ops.ones_like(discriminator_gen_outputs),
 507         discriminator_gen_outputs, label_smoothing, weights, weights, scope,
 508         loss_collection, reduction, add_summaries=False)
 509 
 510   if add_summaries:
 511     summary.scalar('generator_minimax_loss', loss)
 512 
 513   return loss

 571     add_summaries=False):
 572   """Modified generator loss for GANs.
 573 
 574   L = -log(sigmoid(D(G(z))))
 575 
 576   This is the trick used in the original paper to avoid vanishing gradients
 577   early in training. See `Generative Adversarial Nets`
 578   (https://arxiv.org/abs/1406.2661) for more details.
 579 
 580   Args:
 581     discriminator_gen_outputs: Discriminator output on generated data. Expected
 582       to be in the range of (-inf, inf).
 583     label_smoothing: The amount of smoothing for positive labels. This technique
 584       is taken from `Improved Techniques for Training GANs`
 585       (https://arxiv.org/abs/1606.03498). `0.0` means no smoothing.
 586     weights: Optional `Tensor` whose rank is either 0, or the same rank as
 587       `discriminator_gen_outputs`, and must be broadcastable to `labels` (i.e.,
 588       all dimensions must be either `1`, or the same as the corresponding
 589       dimension).
 590     scope: The scope for the operations performed in computing the loss.
 591     loss_collection: collection to which this loss will be added.
 592     reduction: A `tf.losses.Reduction` to apply to loss.
 593     add_summaries: Whether or not to add summaries for the loss.
 594 
 595   Returns:
 596     A loss Tensor. The shape depends on `reduction`.
 597   """
 598   with ops.name_scope(scope, 'generator_modified_loss',
 599                       [discriminator_gen_outputs]) as scope:
 600     loss = losses.sigmoid_cross_entropy(
 601         array_ops.ones_like(discriminator_gen_outputs),
 602         discriminator_gen_outputs, weights, label_smoothing, scope,
 603         loss_collection, reduction)
 604 
 605     if add_summaries:
 606       summary.scalar('generator_modified_loss', loss)
 607 
 608   return loss

 615 def least_squares_generator_loss(
 616     discriminator_gen_outputs,
 617     real_label=1,
 618     weights=1.0,
 619     scope=None,
 620     loss_collection=ops.GraphKeys.LOSSES,
 621     reduction=losses.Reduction.SUM_BY_NONZERO_WEIGHTS,
 622     add_summaries=False):
 623   """Least squares generator loss.
 624 
 625   This loss comes from `Least Squares Generative Adversarial Networks`
 626   (https://arxiv.org/abs/1611.04076).
 627 
 628   L = 1/2 * (D(G(z)) - `real_label`) ** 2
 629 
 630   where D(y) are discriminator logits.
 631 
 632   Args:
 633     discriminator_gen_outputs: Discriminator output on generated data. Expected
 634       to be in the range of (-inf, inf).
 635     real_label: The value that the generator is trying to get the discriminator
 636       to output on generated data.
 637     weights: Optional `Tensor` whose rank is either 0, or the same rank as
 638       `discriminator_gen_outputs`, and must be broadcastable to
 639       `discriminator_gen_outputs` (i.e., all dimensions must be either `1`, or
 640       the same as the corresponding dimension).
 641     scope: The scope for the operations performed in computing the loss.
 642     loss_collection: collection to which this loss will be added.
 643     reduction: A `tf.losses.Reduction` to apply to loss.
 644     add_summaries: Whether or not to add summaries for the loss.
 645 
 646   Returns:
 647     A loss Tensor. The shape depends on `reduction`.
 648   """
 649   with ops.name_scope(scope, 'lsq_generator_loss',
 650                       (discriminator_gen_outputs, real_label)) as scope:
 651     discriminator_gen_outputs = math_ops.to_float(discriminator_gen_outputs)
 652     loss = math_ops.squared_difference(
 653         discriminator_gen_outputs, real_label) / 2.0
 654     loss = losses.compute_weighted_loss(
 655         loss, weights, scope, loss_collection, reduction)
 656 
 657   if add_summaries:
 658     summary.scalar('generator_lsq_loss', loss)
 659 
 660   return loss

 663 def least_squares_discriminator_loss(
 664     discriminator_real_outputs,
 665     discriminator_gen_outputs,
 666     real_label=1,
 667     fake_label=0,
 668     real_weights=1.0,
 669     generated_weights=1.0,
 670     scope=None,
 671     loss_collection=ops.GraphKeys.LOSSES,
 672     reduction=losses.Reduction.SUM_BY_NONZERO_WEIGHTS,
 673     add_summaries=False):
 674   """Least squares discriminator loss.
 675 
 676   This loss comes from `Least Squares Generative Adversarial Networks`
 677   (https://arxiv.org/abs/1611.04076).
 678 
 679   L = 1/2 * (D(x) - `real`) ** 2 +
 680       1/2 * (D(G(z)) - `fake_label`) ** 2
 681 
 682   where D(y) are discriminator logits.
 683 
 684   Args:
 685     discriminator_real_outputs: Discriminator output on real data.
 686     discriminator_gen_outputs: Discriminator output on generated data. Expected
 687       to be in the range of (-inf, inf).
 688     real_label: The value that the discriminator tries to output for real data.
 689     fake_label: The value that the discriminator tries to output for fake data.
 690     real_weights: Optional `Tensor` whose rank is either 0, or the same rank as
 691       `discriminator_real_outputs`, and must be broadcastable to
 692       `discriminator_real_outputs` (i.e., all dimensions must be either `1`, or
 693       the same as the corresponding dimension).
 694     generated_weights: Same as `real_weights`, but for
 695       `discriminator_gen_outputs`.
 696     scope: The scope for the operations performed in computing the loss.
 697     loss_collection: collection to which this loss will be added.
 698     reduction: A `tf.losses.Reduction` to apply to loss.
 699     add_summaries: Whether or not to add summaries for the loss.
 700 
 701   Returns:
 702     A loss Tensor. The shape depends on `reduction`.
 703   """
 704   with ops.name_scope(scope, 'lsq_discriminator_loss',
 705                       (discriminator_gen_outputs, real_label)) as scope:
 706     discriminator_real_outputs = math_ops.to_float(discriminator_real_outputs)
 707     discriminator_gen_outputs = math_ops.to_float(discriminator_gen_outputs)
 708     discriminator_real_outputs.shape.assert_is_compatible_with(
 709         discriminator_gen_outputs.shape)
 710 
 711     real_losses = math_ops.squared_difference(
 712         discriminator_real_outputs, real_label) / 2.0
 713     fake_losses = math_ops.squared_difference(
 714         discriminator_gen_outputs, fake_label) / 2.0
 715 
 716     loss_on_real = losses.compute_weighted_loss(
 717         real_losses, real_weights, scope, loss_collection=None,
 718         reduction=reduction)
 719     loss_on_generated = losses.compute_weighted_loss(
 720         fake_losses, generated_weights, scope, loss_collection=None,
 721         reduction=reduction)
 722 
 723     loss = loss_on_real + loss_on_generated
 724     util.add_loss(loss, loss_collection)
 725 
 726   if add_summaries:
 727     summary.scalar('discriminator_gen_lsq_loss', loss_on_generated)
 728     summary.scalar('discriminator_real_lsq_loss', loss_on_real)
 729     summary.scalar('discriminator_lsq_loss', loss)
 730 
 731   return loss

 760 def mutual_information_penalty(
 761     structured_generator_inputs,
 762     predicted_distributions,
 763     weights=1.0,
 764     scope=None,
 765     loss_collection=ops.GraphKeys.LOSSES,
 766     reduction=losses.Reduction.SUM_BY_NONZERO_WEIGHTS,
 767     add_summaries=False):
 768   """Returns a penalty on the mutual information in an InfoGAN model.
 769 
 770   This loss comes from an InfoGAN paper https://arxiv.org/abs/1606.03657.
 771 
 772   Args:
 773     structured_generator_inputs: A list of Tensors representing the random noise
 774       that must  have high mutual information with the generator output. List
 775       length should match `predicted_distributions`.
 776     predicted_distributions: A list of `tfp.distributions.Distribution`s.
 777       Predicted by the recognizer, and used to evaluate the likelihood of the
 778       structured noise. List length should match `structured_generator_inputs`.
 779     weights: Optional `Tensor` whose rank is either 0, or the same dimensions as
 780       `structured_generator_inputs`.
 781     scope: The scope for the operations performed in computing the loss.
 782     loss_collection: collection to which this loss will be added.
 783     reduction: A `tf.losses.Reduction` to apply to loss.
 784     add_summaries: Whether or not to add summaries for the loss.
 785 
 786   Returns:
 787     A scalar Tensor representing the mutual information loss.
 788   """
 789   _validate_information_penalty_inputs(
 790       structured_generator_inputs, predicted_distributions)
 791 
 792   with ops.name_scope(scope, 'mutual_information_loss') as scope:
 793     # Calculate the negative log-likelihood of the reconstructed noise.
 794     log_probs = [math_ops.reduce_mean(dist.log_prob(noise)) for dist, noise in
 795                  zip(predicted_distributions, structured_generator_inputs)]
 796     loss = -1 * losses.compute_weighted_loss(
 797         log_probs, weights, scope, loss_collection=loss_collection,
 798         reduction=reduction)
 799 
 800     if add_summaries:
 801       summary.scalar('mutual_information_penalty', loss)
 802 
 803   return loss

 806 def _numerically_stable_global_norm(tensor_list):
 807   """Compute the global norm of a list of Tensors, with improved stability.
 808 
 809   The global norm computation sometimes overflows due to the intermediate L2
 810   step. To avoid this, we divide by a cheap-to-compute max over the
 811   matrix elements.
 812 
 813   Args:
 814     tensor_list: A list of tensors, or `None`.
 815 
 816   Returns:
 817     A scalar tensor with the global norm.
 818   """
 819   if np.all([x is None for x in tensor_list]):
 820     return 0.0
 821 
 822   list_max = math_ops.reduce_max([math_ops.reduce_max(math_ops.abs(x)) for x in
 823                                   tensor_list if x is not None])
 824   return list_max * clip_ops.global_norm([x / list_max for x in tensor_list
 825                                           if x is not None])

 845 # TODO(joelshor): Add ability to pass in gradients, to avoid recomputing.
 846 def combine_adversarial_loss(main_loss,
 847                              adversarial_loss,
 848                              weight_factor=None,
 849                              gradient_ratio=None,
 850                              gradient_ratio_epsilon=1e-6,
 851                              variables=None,
 852                              scalar_summaries=True,
 853                              gradient_summaries=True,
 854                              scope=None):
 855   """Utility to combine main and adversarial losses.
 856 
 857   This utility combines the main and adversarial losses in one of two ways.
 858   1) Fixed coefficient on adversarial loss. Use `weight_factor` in this case.
 859   2) Fixed ratio of gradients. Use `gradient_ratio` in this case. This is often
 860     used to make sure both losses affect weights roughly equally, as in
 861     https://arxiv.org/pdf/1705.05823.
 862 
 863   One can optionally also visualize the scalar and gradient behavior of the
 864   losses.
 865 
 866   Args:
 867     main_loss: A floating scalar Tensor indicating the main loss.
 868     adversarial_loss: A floating scalar Tensor indication the adversarial loss.
 869     weight_factor: If not `None`, the coefficient by which to multiply the
 870       adversarial loss. Exactly one of this and `gradient_ratio` must be
 871       non-None.
 872     gradient_ratio: If not `None`, the ratio of the magnitude of the gradients.
 873       Specifically,
 874         gradient_ratio = grad_mag(main_loss) / grad_mag(adversarial_loss)
 875       Exactly one of this and `weight_factor` must be non-None.
 876     gradient_ratio_epsilon: An epsilon to add to the adversarial loss
 877       coefficient denominator, to avoid division-by-zero.
 878     variables: List of variables to calculate gradients with respect to. If not
 879       present, defaults to all trainable variables.
 880     scalar_summaries: Create scalar summaries of losses.
 881     gradient_summaries: Create gradient summaries of losses.
 882     scope: Optional name scope.
 883 
 884   Returns:
 885     A floating scalar Tensor indicating the desired combined loss.
 886 
 887   Raises:
 888     ValueError: Malformed input.
 889   """
 890   _validate_args([main_loss, adversarial_loss], weight_factor, gradient_ratio)
 891   if variables is None:
 892     variables = contrib_variables_lib.get_trainable_variables()
 893 
 894   with ops.name_scope(scope, 'adversarial_loss',
 895                       values=[main_loss, adversarial_loss]):
 896     # Compute gradients if we will need them.
 897     if gradient_summaries or gradient_ratio is not None:
 898       main_loss_grad_mag = _numerically_stable_global_norm(
 899           gradients_impl.gradients(main_loss, variables))
 900       adv_loss_grad_mag = _numerically_stable_global_norm(
 901           gradients_impl.gradients(adversarial_loss, variables))
 902 
 903     # Add summaries, if applicable.
 904     if scalar_summaries:
 905       summary.scalar('main_loss', main_loss)
 906       summary.scalar('adversarial_loss', adversarial_loss)
 907     if gradient_summaries:
 908       summary.scalar('main_loss_gradients', main_loss_grad_mag)
 909       summary.scalar('adversarial_loss_gradients', adv_loss_grad_mag)
 910 
 911     # Combine losses in the appropriate way.
 912     # If `weight_factor` is always `0`, avoid computing the adversarial loss
 913     # tensor entirely.
 914     if _used_weight((weight_factor, gradient_ratio)) == 0:
 915       final_loss = main_loss
 916     elif weight_factor is not None:
 917       final_loss = (main_loss +
 918                     array_ops.stop_gradient(weight_factor) * adversarial_loss)
 919     elif gradient_ratio is not None:
 920       grad_mag_ratio = main_loss_grad_mag / (
 921           adv_loss_grad_mag + gradient_ratio_epsilon)
 922       adv_coeff = grad_mag_ratio / gradient_ratio
 923       summary.scalar('adversarial_coefficient', adv_coeff)
 924       final_loss = (main_loss +
 925                     array_ops.stop_gradient(adv_coeff) * adversarial_loss)
 926 
 927   return final_loss

 930 def cycle_consistency_loss(data_x,
 931                            reconstructed_data_x,
 932                            data_y,
 933                            reconstructed_data_y,
 934                            scope=None,
 935                            add_summaries=False):
 936   """Defines the cycle consistency loss.
 937 
 938   The cyclegan model has two partial models where `model_x2y` generator F maps
 939   data set X to Y, `model_y2x` generator G maps data set Y to X. For a `data_x`
 940   in data set X, we could reconstruct it by
 941   * reconstructed_data_x = G(F(data_x))
 942   Similarly
 943   * reconstructed_data_y = F(G(data_y))
 944 
 945   The cycle consistency loss is about the difference between data and
 946   reconstructed data, namely
 947   * loss_x2x = |data_x - G(F(data_x))| (L1-norm)
 948   * loss_y2y = |data_y - F(G(data_y))| (L1-norm)
 949   * loss = (loss_x2x + loss_y2y) / 2
 950   where `loss` is the final result.
 951 
 952   For the L1-norm, we follow the original implementation:
 953   https://github.com/junyanz/CycleGAN/blob/master/models/cycle_gan_model.lua
 954   we use L1-norm of pixel-wise error normalized by data size such that
 955   `cycle_loss_weight` can be specified independent of image size.
 956 
 957   See https://arxiv.org/abs/1703.10593 for more details.
 958 
 959   Args:
 960     data_x: A `Tensor` of data X.
 961     reconstructed_data_x: A `Tensor` of reconstructed data X.
 962     data_y: A `Tensor` of data Y.
 963     reconstructed_data_y: A `Tensor` of reconstructed data Y.
 964     scope: The scope for the operations performed in computing the loss.
 965       Defaults to None.
 966     add_summaries: Whether or not to add detailed summaries for the loss.
 967       Defaults to False.
 968 
 969   Returns:
 970     A scalar `Tensor` of cycle consistency loss.
 971   """
 972 
 973   with ops.name_scope(
 974       scope,
 975       'cycle_consistency_loss',
 976       values=[data_x, reconstructed_data_x, data_y, reconstructed_data_y]):
 977     loss_x2x = losses.absolute_difference(data_x, reconstructed_data_x)
 978     loss_y2y = losses.absolute_difference(data_y, reconstructed_data_y)
 979     loss = (loss_x2x + loss_y2y) / 2.0
 980     if add_summaries:
 981       summary.scalar('cycle_consistency_loss_x2x', loss_x2x)
 982       summary.scalar('cycle_consistency_loss_y2y', loss_y2y)
 983       summary.scalar('cycle_consistency_loss', loss)
 984 
 985   return loss
