  15 """TFGAN utilities for loss functions that accept GANModel namedtuples.
  16 
  17 The losses and penalties in this file all correspond to losses in
  18 `losses_impl.py`. Losses in that file take individual arguments, whereas in this
  19 file they take a `GANModel` tuple. For example:
  20 
  21 losses_impl.py:
  22   ```python
  23   def wasserstein_discriminator_loss(
  24       discriminator_real_outputs,
  25       discriminator_gen_outputs,
  26       real_weights=1.0,
  27       generated_weights=1.0,
  28       scope=None,
  29       loss_collection=ops.GraphKeys.LOSSES,
  30       reduction=losses.Reduction.SUM_BY_NONZERO_WEIGHTS,
  31       add_summaries=False)
  32   ```
  33 
  34 tuple_losses_impl.py:
  35   ```python
  36   def wasserstein_discriminator_loss(
  37       gan_model,
  38       real_weights=1.0,
  39       generated_weights=1.0,
  40       scope=None,
  41       loss_collection=ops.GraphKeys.LOSSES,
  42       reduction=losses.Reduction.SUM_BY_NONZERO_WEIGHTS,
  43       add_summaries=False)
  44   ```
  45 
  46 
  47 
  48 Example usage:
  49   ```python
  50   # `tfgan.losses.wargs` losses take individual arguments.
  51   w_loss = tfgan.losses.wargs.wasserstein_discriminator_loss(
  52     discriminator_real_outputs,
  53     discriminator_gen_outputs)
  54 
  55   # `tfgan.losses` losses take GANModel namedtuples.
  56   w_loss2 = tfgan.losses.wasserstein_discriminator_loss(gan_model)
  57   ```

  89 def _args_to_gan_model(loss_fn):
  90   """Converts a loss taking individual args to one taking a GANModel namedtuple.
  91 
  92   The new function has the same name as the original one.
  93 
  94   Args:
  95     loss_fn: A python function taking a `GANModel` object and returning a loss
  96       Tensor calculated from that object. The shape of the loss depends on
  97       `reduction`.
  98 
  99   Returns:
 100     A new function that takes a GANModel namedtuples and returns the same loss.
 101   """
 102   # Match arguments in `loss_fn` to elements of `namedtuple`.
 103   # TODO(joelshor): Properly handle `varargs` and `keywords`.
 104   argspec = tf_inspect.getargspec(loss_fn)
 105   defaults = argspec.defaults or []
 106 
 107   required_args = set(argspec.args[:-len(defaults)])
 108   args_with_defaults = argspec.args[-len(defaults):]
 109   default_args_dict = dict(zip(args_with_defaults, defaults))
 110 
 111   def new_loss_fn(gan_model, **kwargs):  # pylint:disable=missing-docstring
 112     def _asdict(namedtuple):
 113       """Returns a namedtuple as a dictionary.
 114 
 115       This is required because `_asdict()` in Python 3.x.x is broken in classes
 116       that inherit from `collections.namedtuple`. See
 117       https://bugs.python.org/issue24931 for more details.
 118 
 119       Args:
 120         namedtuple: An object that inherits from `collections.namedtuple`.
 121 
 122       Returns:
 123         A dictionary version of the tuple.
 124       """
 125       return {k: getattr(namedtuple, k) for k in namedtuple._fields}
 126     gan_model_dict = _asdict(gan_model)
 127 
 128     # Make sure non-tuple required args are supplied.
 129     args_from_tuple = set(argspec.args).intersection(set(gan_model._fields))
 130     required_args_not_from_tuple = required_args - args_from_tuple
 131     for arg in required_args_not_from_tuple:
 132       if arg not in kwargs:
 133         raise ValueError('`%s` must be supplied to %s loss function.' % (
 134             arg, loss_fn.__name__))
 135 
 136     # Make sure tuple args aren't also supplied as keyword args.
 137     ambiguous_args = set(gan_model._fields).intersection(set(kwargs.keys()))
 138     if ambiguous_args:
 139       raise ValueError(
 140           'The following args are present in both the tuple and keyword args '
 141           'for %s: %s' % (loss_fn.__name__, ambiguous_args))
 142 
 143     # Add required args to arg dictionary.
 144     required_args_from_tuple = required_args.intersection(args_from_tuple)
 145     for arg in required_args_from_tuple:
 146       assert arg not in kwargs
 147       kwargs[arg] = gan_model_dict[arg]
 148 
 149     # Add arguments that have defaults.
 150     for arg in default_args_dict:
 151       val_from_tuple = gan_model_dict[arg] if arg in gan_model_dict else None
 152       val_from_kwargs = kwargs[arg] if arg in kwargs else None
 153       assert not (val_from_tuple is not None and val_from_kwargs is not None)
 154       kwargs[arg] = (val_from_tuple if val_from_tuple is not None else
 155                      val_from_kwargs if val_from_kwargs is not None else
 156                      default_args_dict[arg])
 157 
 158     return loss_fn(**kwargs)
 159 
 160   new_docstring = """The gan_model version of %s.""" % loss_fn.__name__
 161   new_loss_fn.__docstring__ = new_docstring
 162   new_loss_fn.__name__ = loss_fn.__name__
 163   new_loss_fn.__module__ = loss_fn.__module__
 164   return new_loss_fn

 210 def combine_adversarial_loss(gan_loss,
 211                              gan_model,
 212                              non_adversarial_loss,
 213                              weight_factor=None,
 214                              gradient_ratio=None,
 215                              gradient_ratio_epsilon=1e-6,
 216                              scalar_summaries=True,
 217                              gradient_summaries=True):
 218   """Combine adversarial loss and main loss.
 219 
 220   Uses `combine_adversarial_loss` to combine the losses, and returns
 221   a modified GANLoss namedtuple.
 222 
 223   Args:
 224     gan_loss: A GANLoss namedtuple. Assume the GANLoss.generator_loss is the
 225       adversarial loss.
 226     gan_model: A GANModel namedtuple. Used to access the generator's variables.
 227     non_adversarial_loss: Same as `main_loss` from
 228       `combine_adversarial_loss`.
 229     weight_factor: Same as `weight_factor` from
 230       `combine_adversarial_loss`.
 231     gradient_ratio: Same as `gradient_ratio` from
 232       `combine_adversarial_loss`.
 233     gradient_ratio_epsilon: Same as `gradient_ratio_epsilon` from
 234       `combine_adversarial_loss`.
 235     scalar_summaries: Same as `scalar_summaries` from
 236       `combine_adversarial_loss`.
 237     gradient_summaries: Same as `gradient_summaries` from
 238       `combine_adversarial_loss`.
 239 
 240   Returns:
 241     A modified GANLoss namedtuple, with `non_adversarial_loss` included
 242     appropriately.
 243   """
 244   combined_loss = losses_impl.combine_adversarial_loss(
 245       non_adversarial_loss,
 246       gan_loss.generator_loss,
 247       weight_factor,
 248       gradient_ratio,
 249       gradient_ratio_epsilon,
 250       gan_model.generator_variables,
 251       scalar_summaries,
 252       gradient_summaries)
 253   return gan_loss._replace(generator_loss=combined_loss)

 256 def cycle_consistency_loss(cyclegan_model, scope=None, add_summaries=False):
 257   """Defines the cycle consistency loss.
 258 
 259   Uses `cycle_consistency_loss` to compute the cycle consistency loss for a
 260   `cyclegan_model`.
 261 
 262   Args:
 263     cyclegan_model: A `CycleGANModel` namedtuple.
 264     scope: The scope for the operations performed in computing the loss.
 265       Defaults to None.
 266     add_summaries: Whether or not to add detailed summaries for the loss.
 267       Defaults to False.
 268 
 269   Returns:
 270     A scalar `Tensor` of cycle consistency loss.
 271 
 272   Raises:
 273     ValueError: If `cyclegan_model` is not a `CycleGANModel` namedtuple.
 274   """
 275   if not isinstance(cyclegan_model, namedtuples.CycleGANModel):
 276     raise ValueError(
 277         '`cyclegan_model` must be a `CycleGANModel`. Instead, was %s.' %
 278         type(cyclegan_model))
 279   return losses_impl.cycle_consistency_loss(
 280       cyclegan_model.model_x2y.generator_inputs, cyclegan_model.reconstructed_x,
 281       cyclegan_model.model_y2x.generator_inputs, cyclegan_model.reconstructed_y,
 282       scope, add_summaries)

 285 def stargan_generator_loss_wrapper(loss_fn):
 286   """Convert a generator loss function to take a StarGANModel.
 287 
 288   The new function has the same name as the original one.
 289 
 290   Args:
 291     loss_fn: A python function taking Discriminator's real/fake prediction for
 292       generated data.
 293 
 294   Returns:
 295     A new function that takes a StarGANModel namedtuple and returns the same
 296     loss.
 297   """
 298 
 299   def new_loss_fn(stargan_model, **kwargs):
 300     return loss_fn(
 301         stargan_model.discriminator_generated_data_source_predication, **kwargs)
 302 
 303   new_docstring = """The stargan_model version of %s.""" % loss_fn.__name__
 304   new_loss_fn.__docstring__ = new_docstring
 305   new_loss_fn.__name__ = loss_fn.__name__
 306   new_loss_fn.__module__ = loss_fn.__module__
 307   return new_loss_fn

 336 def stargan_gradient_penalty_wrapper(loss_fn):
 337   """Convert a gradient penalty function to take a StarGANModel.
 338 
 339   The new function has the same name as the original one.
 340 
 341   Args:
 342     loss_fn: A python function taking real_data, generated_data,
 343       generator_inputs for Discriminator's condition (i.e. number of domains),
 344       discriminator_fn, and discriminator_scope.
 345 
 346   Returns:
 347     A new function that takes a StarGANModel namedtuple and returns the same
 348     loss.
 349   """
 350 
 351   def new_loss_fn(stargan_model, **kwargs):
 352     num_domains = stargan_model.input_data_domain_label.shape.as_list()[-1]
 353     return loss_fn(
 354         real_data=stargan_model.input_data,
 355         generated_data=stargan_model.generated_data,
 356         generator_inputs=num_domains,
 357         discriminator_fn=stargan_model.discriminator_fn,
 358         discriminator_scope=stargan_model.discriminator_scope,
 359         **kwargs)
 360 
 361   new_docstring = """The stargan_model version of %s.""" % loss_fn.__name__
 362   new_loss_fn.__docstring__ = new_docstring
 363   new_loss_fn.__name__ = loss_fn.__name__
 364   new_loss_fn.__module__ = loss_fn.__module__
 365   return new_loss_fn
