  38 def _summary_key(head_name, val):
  39   return '%s/%s' % (val, head_name) if head_name else val

  45 def gan_head(generator_loss_fn, discriminator_loss_fn, generator_optimizer,
  46              discriminator_optimizer, use_loss_summaries=True,
  47              get_hooks_fn=tfgan_train.get_sequential_train_hooks(),
  48              get_eval_metric_ops_fn=None, name=None):
  49   """Creates a `GANHead`.
  50 
  51   Args:
  52     generator_loss_fn: A TFGAN loss function for the generator. Takes a
  53       `GANModel` and returns a scalar.
  54     discriminator_loss_fn: Same as `generator_loss_fn`, but for the
  55       discriminator.
  56     generator_optimizer: The optimizer for generator updates.
  57     discriminator_optimizer: Same as `generator_optimizer`, but for the
  58       discriminator updates.
  59     use_loss_summaries: If `True`, add loss summaries. If `False`, does not.
  60       If `None`, uses defaults.
  61     get_hooks_fn: A function that takes a `GANTrainOps` tuple and returns a
  62       list of hooks.
  63     get_eval_metric_ops_fn: A function that takes a `GANModel`, and returns a
  64       dict of metric results keyed by name. The output of this function is
  65       passed into `tf.estimator.EstimatorSpec` during evaluation.
  66     name: name of the head. If provided, summary and metrics keys will be
  67       suffixed by `"/" + name`.
  68 
  69   Returns:
  70     An instance of `GANHead`.
  71   """
  72   return GANHead(generator_loss_fn=generator_loss_fn,
  73                  discriminator_loss_fn=discriminator_loss_fn,
  74                  generator_optimizer=generator_optimizer,
  75                  discriminator_optimizer=discriminator_optimizer,
  76                  use_loss_summaries=use_loss_summaries,
  77                  get_hooks_fn=get_hooks_fn,
  78                  get_eval_metric_ops_fn=get_eval_metric_ops_fn,
  79                  name=name)

  83   """`Head` for a GAN."""
  84 
  85   @deprecation.deprecated(
  86       None, 'Please use tf.contrib.gan.GANEstimator without explicitly making '
  87       'a GANHead.')
  88   def __init__(self, generator_loss_fn, discriminator_loss_fn,
  89                generator_optimizer, discriminator_optimizer,
  90                use_loss_summaries=True,
  91                get_hooks_fn=None,
  92                get_eval_metric_ops_fn=None,
  93                name=None):
  94     """`Head` for GAN training.
  95 
  96     Args:
  97       generator_loss_fn: A TFGAN loss function for the generator. Takes a
  98         `GANModel` and returns a scalar.
  99       discriminator_loss_fn: Same as `generator_loss_fn`, but for the
 100       discriminator.
 101       generator_optimizer: The optimizer for generator updates.
 102       discriminator_optimizer: Same as `generator_optimizer`, but for the
 103         discriminator updates.
 104       use_loss_summaries: If `True`, add loss summaries. If `False`, does not.
 105         If `None`, uses defaults.
 106       get_hooks_fn: A function that takes a `GANTrainOps` tuple and returns a
 107         list of hooks. Defaults to `train.get_sequential_train_hooks()`
 108       get_eval_metric_ops_fn: A function that takes a `GANModel`, and returns a
 109         dict of metric results keyed by name. The output of this function is
 110         passed into `tf.estimator.EstimatorSpec` during evaluation.
 111       name: name of the head. If provided, summary and metrics keys will be
 112         suffixed by `"/" + name`.
 113     """
 114 
 115     if not callable(generator_loss_fn):
 116       raise TypeError('generator_loss_fn must be callable.')
 117     if not callable(discriminator_loss_fn):
 118       raise TypeError('discriminator_loss_fn must be callable.')
 119     if use_loss_summaries not in [True, False, None]:
 120       raise ValueError('use_loss_summaries must be True, False or None.')
 121     if get_hooks_fn is not None and not callable(get_hooks_fn):
 122       raise TypeError('get_hooks_fn must be callable.')
 123     if name is not None and not isinstance(name, str):
 124       raise TypeError('name must be string.')
 125 
 126     if get_hooks_fn is None:
 127       get_hooks_fn = tfgan_train.get_sequential_train_hooks()
 128 
 129     if use_loss_summaries in [True, False]:
 130       generator_loss_fn = functools.partial(
 131           generator_loss_fn, add_summaries=use_loss_summaries)
 132       discriminator_loss_fn = functools.partial(
 133           discriminator_loss_fn, add_summaries=use_loss_summaries)
 134     self._generator_loss_fn = generator_loss_fn
 135     self._discriminator_loss_fn = discriminator_loss_fn
 136     self._generator_optimizer = generator_optimizer
 137     self._discriminator_optimizer = discriminator_optimizer
 138     self._get_hooks_fn = get_hooks_fn
 139     self._get_eval_metric_ops_fn = get_eval_metric_ops_fn
 140     self._name = name
 141 
 142   @property
 143   def name(self):
 144     return self._name
 145 
 146   @property
 147   def logits_dimension(self):
 148     return None
 149 
 150   def create_loss(self, features, mode, logits, labels):
 151     """Returns a GANLoss tuple from the provided GANModel.
 152 
 153     See `Head` for more details.
 154 
 155     Args:
 156       features: Input `dict` of `Tensor` objects. Unused.
 157       mode: Estimator's `ModeKeys`.
 158       logits: A GANModel tuple.
 159       labels: Must be `None`.
 160 
 161     Returns:
 162       A GANLoss tuple.
 163 
 164     """
 165     _validate_logits_and_labels(logits, labels)
 166     del mode, labels, features  # unused for this head.
 167     gan_model = logits  # rename variable for clarity
 168     return tfgan_tuples.GANLoss(
 169         generator_loss=self._generator_loss_fn(gan_model),
 170         discriminator_loss=self._discriminator_loss_fn(gan_model))
 171 
 172   def create_estimator_spec(
 173       self, features, mode, logits, labels=None,
 174       train_op_fn=tfgan_train.gan_train_ops):
 175     """Returns `EstimatorSpec` that a model_fn can return.
 176 
 177     See `Head` for more details.
 178 
 179     Args:
 180       features: Must be `None`.
 181       mode: Estimator's `ModeKeys`.
 182       logits: A GANModel tuple.
 183       labels: Must be `None`.
 184       train_op_fn: Function that takes a GANModel, GANLoss, generator optimizer,
 185         and discriminator optimizer, and returns a `GANTrainOps` tuple. For
 186         example, this function can come from TFGAN's `train.py` library, or can
 187         be custom.
 188 
 189     Returns:
 190       `EstimatorSpec`.
 191 
 192     Raises:
 193       ValueError: If `features` isn't `None`.
 194       ValueError: If `train_op_fn` isn't provided in train mode.
 195     """
 196     _validate_logits_and_labels(logits, labels)
 197     if features is not None:
 198       raise ValueError('`features` should be `None`. Instead, found: %s' %
 199                        features)
 200     gan_model = logits  # rename variable for clarity
 201     with ops.name_scope('GANHead'):
 202       if mode == model_fn_lib.ModeKeys.PREDICT:
 203         return model_fn_lib.EstimatorSpec(
 204             mode=model_fn_lib.ModeKeys.PREDICT,
 205             predictions=gan_model.generated_data,
 206             export_outputs={
 207                 'predict': export_output.PredictOutput(gan_model.generated_data)
 208             })
 209       elif mode == model_fn_lib.ModeKeys.EVAL:
 210         gan_loss = self.create_loss(
 211             features=None, mode=mode, logits=gan_model, labels=None)
 212         scalar_loss = gan_loss.generator_loss + gan_loss.discriminator_loss
 213         with ops.name_scope(None, 'metrics',
 214                             [gan_loss.generator_loss,
 215                              gan_loss.discriminator_loss]):
 216           eval_metric_ops = {
 217               _summary_key(self._name, 'generator_loss'):
 218                   metrics_lib.mean(gan_loss.generator_loss),
 219               _summary_key(self._name, 'discriminator_loss'):
 220                   metrics_lib.mean(gan_loss.discriminator_loss)
 221           }
 222           if self._get_eval_metric_ops_fn is not None:
 223             custom_eval_metric_ops = self._get_eval_metric_ops_fn(gan_model)
 224             if not isinstance(custom_eval_metric_ops, dict):
 225               raise TypeError('get_eval_metric_ops_fn must return a dict, '
 226                               'received: {}'.format(custom_eval_metric_ops))
 227             eval_metric_ops.update(custom_eval_metric_ops)
 228         return model_fn_lib.EstimatorSpec(
 229             mode=model_fn_lib.ModeKeys.EVAL,
 230             predictions=gan_model.generated_data,
 231             loss=scalar_loss,
 232             eval_metric_ops=eval_metric_ops)
 233       elif mode == model_fn_lib.ModeKeys.TRAIN:
 234         if train_op_fn is None:
 235           raise ValueError('train_op_fn can not be None.')
 236         gan_loss = self.create_loss(None, mode, gan_model, None)
 237         scalar_loss = gan_loss.generator_loss + gan_loss.discriminator_loss
 238         train_ops = train_op_fn(gan_model, gan_loss, self._generator_optimizer,
 239                                 self._discriminator_optimizer)
 240         training_hooks = self._get_hooks_fn(train_ops)
 241         return model_fn_lib.EstimatorSpec(
 242             loss=scalar_loss,
 243             mode=model_fn_lib.ModeKeys.TRAIN,
 244             train_op=train_ops.global_step_inc_op,
 245             training_hooks=training_hooks)
 246       else:
 247         raise ValueError('Mode not recognized: %s' % mode)

 250 def _validate_logits_and_labels(logits, labels):
 251   if labels is not None:
 252     raise ValueError('`GANHead`\'s `create_estimator_spec` input `labels` must '
 253                      'be `None`. Instead, found: %s' % labels)
 254 
 255   if not isinstance(logits, tfgan_tuples.GANModel):
 256     raise ValueError('`GANHead`\'s `create_estimator_spec` input `logits` must '
 257                      'be an instnace of a `GANModel`. Instead, found: %s' %
 258                      logits)
