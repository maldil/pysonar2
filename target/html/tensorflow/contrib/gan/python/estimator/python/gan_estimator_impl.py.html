  56 class GANEstimator(estimator.Estimator):
  57   """An estimator for Generative Adversarial Networks (GANs).
  58 
  59   This Estimator is backed by TFGAN. The network functions follow the TFGAN API
  60   except for one exception: if either `generator_fn` or `discriminator_fn` have
  61   an argument called `mode`, then the tf.Estimator mode is passed in for that
  62   argument. This helps with operations like batch normalization, which have
  63   different train and evaluation behavior.
  64 
  65   Example:
  66 
  67   ```python
  68       import tensorflow as tf
  69       tfgan = tf.contrib.gan
  70 
  71       # See TFGAN's `train.py` for a description of the generator and
  72       # discriminator API.
  73       def generator_fn(generator_inputs):
  74         ...
  75         return generated_data
  76 
  77       def discriminator_fn(data, conditioning):
  78         ...
  79         return logits
  80 
  81       # Create GAN estimator.
  82       gan_estimator = tfgan.estimator.GANEstimator(
  83           model_dir,
  84           generator_fn=generator_fn,
  85           discriminator_fn=discriminator_fn,
  86           generator_loss_fn=tfgan.losses.wasserstein_generator_loss,
  87           discriminator_loss_fn=tfgan.losses.wasserstein_discriminator_loss,
  88           generator_optimizer=tf.train.AdamOptimizer(0.1, 0.5),
  89           discriminator_optimizer=tf.train.AdamOptimizer(0.1, 0.5))
  90 
  91       # Train estimator.
  92       gan_estimator.train(train_input_fn, steps)
  93 
  94       # Evaluate resulting estimator.
  95       gan_estimator.evaluate(eval_input_fn)
  96 
  97       # Generate samples from generator.
  98       predictions = np.array([
  99           x for x in gan_estimator.predict(predict_input_fn)])
 100   ```
 101   """
 102 
 103   def __init__(self,
 104                model_dir=None,
 105                generator_fn=None,
 106                discriminator_fn=None,
 107                generator_loss_fn=None,
 108                discriminator_loss_fn=None,
 109                generator_optimizer=None,
 110                discriminator_optimizer=None,
 111                get_hooks_fn=None,
 112                get_eval_metric_ops_fn=None,
 113                add_summaries=None,
 114                use_loss_summaries=True,
 115                config=None):
 116     """Initializes a GANEstimator instance.
 117 
 118     Args:
 119       model_dir: Directory to save model parameters, graph and etc. This can
 120         also be used to load checkpoints from the directory into a estimator
 121         to continue training a previously saved model.
 122       generator_fn: A python function that takes a Tensor, Tensor list, or
 123         Tensor dictionary as inputs and returns the outputs of the GAN
 124         generator. See `TFGAN` for more details and examples. Additionally, if
 125         it has an argument called `mode`, the Estimator's `mode` will be passed
 126         in (ex TRAIN, EVAL, PREDICT). This is useful for things like batch
 127         normalization.
 128       discriminator_fn: A python function that takes the output of
 129         `generator_fn` or real data in the GAN setup, and `generator_inputs`.
 130         Outputs a Tensor in the range [-inf, inf]. See `TFGAN` for more details
 131         and examples.
 132       generator_loss_fn: The loss function on the generator. Takes a `GANModel`
 133         tuple.
 134       discriminator_loss_fn: The loss function on the discriminator. Takes a
 135         `GANModel` tuple.
 136       generator_optimizer: The optimizer for generator updates, or a function
 137         that takes no arguments and returns an optimizer. This function will
 138         be called when the default graph is the `GANEstimator`'s graph, so
 139         utilities like `tf.contrib.framework.get_or_create_global_step` will
 140         work.
 141       discriminator_optimizer: Same as `generator_optimizer`, but for the
 142         discriminator updates.
 143       get_hooks_fn: A function that takes a `GANTrainOps` tuple and returns a
 144         list of hooks. These hooks are run on the generator and discriminator
 145         train ops, and can be used to implement the GAN training scheme.
 146         Defaults to `train.get_sequential_train_hooks()`.
 147       get_eval_metric_ops_fn: A function that takes a `GANModel`, and returns a
 148         dict of metric results keyed by name. The output of this function is
 149         passed into `tf.estimator.EstimatorSpec` during evaluation.
 150       add_summaries: `None`, a single `SummaryType`, or a list of `SummaryType`.
 151       use_loss_summaries: If `True`, add loss summaries. If `False`, does not.
 152         If `None`, uses defaults.
 153       config: `RunConfig` object to configure the runtime settings.
 154 
 155     Raises:
 156       ValueError: If loss functions aren't callable.
 157       ValueError: If `use_loss_summaries` isn't boolean or `None`.
 158       ValueError: If `get_hooks_fn` isn't callable or `None`.
 159     """
 160     if not callable(generator_loss_fn):
 161       raise ValueError('generator_loss_fn must be callable.')
 162     if not callable(discriminator_loss_fn):
 163       raise ValueError('discriminator_loss_fn must be callable.')
 164     if use_loss_summaries not in [True, False, None]:
 165       raise ValueError('use_loss_summaries must be True, False or None.')
 166     if get_hooks_fn is not None and not callable(get_hooks_fn):
 167       raise TypeError('get_hooks_fn must be callable.')
 168 
 169     def _model_fn(features, labels, mode):
 170       """GANEstimator model function."""
 171       if mode not in [model_fn_lib.ModeKeys.TRAIN, model_fn_lib.ModeKeys.EVAL,
 172                       model_fn_lib.ModeKeys.PREDICT]:
 173         raise ValueError('Mode not recognized: %s' % mode)
 174       real_data = labels  # rename inputs for clarity
 175       generator_inputs = features  # rename inputs for clarity
 176 
 177       # Make GANModel, which encapsulates the GAN model architectures.
 178       gan_model = _get_gan_model(
 179           mode, generator_fn, discriminator_fn, real_data, generator_inputs,
 180           add_summaries)
 181 
 182       # Make the EstimatorSpec, which incorporates the GANModel, losses, eval
 183       # metrics, and optimizers (if required).
 184       return _get_estimator_spec(
 185           mode, gan_model, generator_loss_fn, discriminator_loss_fn,
 186           get_eval_metric_ops_fn, generator_optimizer, discriminator_optimizer,
 187           get_hooks_fn, use_loss_summaries)
 188 
 189     super(GANEstimator, self).__init__(
 190         model_fn=_model_fn, model_dir=model_dir, config=config)
 191 
 192 
 193 def _get_gan_model(
 194     mode, generator_fn, discriminator_fn, real_data, generator_inputs,
 195     add_summaries, generator_scope='Generator'):
 196   """Makes the GANModel tuple, which encapsulates the GAN model architecture."""
 197   if mode == model_fn_lib.ModeKeys.PREDICT:
 198     if real_data is not None:
 199       raise ValueError('`labels` must be `None` when mode is `predict`. '
 200                        'Instead, found %s' % real_data)
 201     gan_model = _make_prediction_gan_model(
 202         generator_inputs, generator_fn, generator_scope)
 203   else:  # model_fn_lib.ModeKeys.TRAIN or model_fn_lib.ModeKeys.EVAL
 204     gan_model = _make_gan_model(
 205         generator_fn, discriminator_fn, real_data, generator_inputs,
 206         generator_scope, add_summaries, mode)
 207 
 208   return gan_model
 209

 211 def _get_estimator_spec(
 212     mode, gan_model, generator_loss_fn, discriminator_loss_fn,
 213     get_eval_metric_ops_fn, generator_optimizer, discriminator_optimizer,
 214     get_hooks_fn=None, use_loss_summaries=True):
 215   """Get the EstimatorSpec for the current mode."""
 216   if mode == model_fn_lib.ModeKeys.PREDICT:
 217     estimator_spec = model_fn_lib.EstimatorSpec(
 218         mode=mode, predictions=gan_model.generated_data)
 219   else:
 220     gan_loss = tfgan_tuples.GANLoss(
 221         generator_loss=generator_loss_fn(
 222             gan_model, add_summaries=use_loss_summaries),
 223         discriminator_loss=discriminator_loss_fn(
 224             gan_model, add_summaries=use_loss_summaries))
 225     if mode == model_fn_lib.ModeKeys.EVAL:
 226       estimator_spec = _get_eval_estimator_spec(
 227           gan_model, gan_loss, get_eval_metric_ops_fn)
 228     else:  # model_fn_lib.ModeKeys.TRAIN:
 229       gopt = (generator_optimizer() if callable(generator_optimizer) else
 230               generator_optimizer)
 231       dopt = (discriminator_optimizer() if callable(discriminator_optimizer)
 232               else discriminator_optimizer)
 233       get_hooks_fn = get_hooks_fn or tfgan_train.get_sequential_train_hooks()
 234       estimator_spec = _get_train_estimator_spec(
 235           gan_model, gan_loss, gopt, dopt, get_hooks_fn)
 236 
 237   return estimator_spec

 240 def _make_gan_model(generator_fn, discriminator_fn, real_data,
 241                     generator_inputs, generator_scope, add_summaries, mode):
 242   """Construct a `GANModel`, and optionally pass in `mode`."""
 243   # If network functions have an argument `mode`, pass mode to it.
 244   if 'mode' in inspect.getargspec(generator_fn).args:
 245     generator_fn = functools.partial(generator_fn, mode=mode)
 246   if 'mode' in inspect.getargspec(discriminator_fn).args:
 247     discriminator_fn = functools.partial(discriminator_fn, mode=mode)
 248   gan_model = tfgan_train.gan_model(
 249       generator_fn,
 250       discriminator_fn,
 251       real_data,
 252       generator_inputs,
 253       generator_scope=generator_scope,
 254       check_shapes=False)
 255   if add_summaries:
 256     if not isinstance(add_summaries, (tuple, list)):
 257       add_summaries = [add_summaries]
 258     with ops.name_scope(None):
 259       for summary_type in add_summaries:
 260         _summary_type_map[summary_type](gan_model)
 261 
 262   return gan_model

 265 def _make_prediction_gan_model(generator_inputs, generator_fn, generator_scope):
 266   """Make a `GANModel` from just the generator."""
 267   # If `generator_fn` has an argument `mode`, pass mode to it.
 268   if 'mode' in inspect.getargspec(generator_fn).args:
 269     generator_fn = functools.partial(generator_fn,
 270                                      mode=model_fn_lib.ModeKeys.PREDICT)
 271   with variable_scope.variable_scope(generator_scope) as gen_scope:
 272     generator_inputs = tfgan_train._convert_tensor_or_l_or_d(generator_inputs)  # pylint:disable=protected-access
 273     generated_data = generator_fn(generator_inputs)
 274   generator_variables = variable_lib.get_trainable_variables(gen_scope)
 275 
 276   return tfgan_tuples.GANModel(
 277       generator_inputs,
 278       generated_data,
 279       generator_variables,
 280       gen_scope,
 281       generator_fn,
 282       real_data=None,
 283       discriminator_real_outputs=None,
 284       discriminator_gen_outputs=None,
 285       discriminator_variables=None,
 286       discriminator_scope=None,
 287       discriminator_fn=None)

 290 def _get_eval_estimator_spec(gan_model, gan_loss, get_eval_metric_ops_fn=None,
 291                              name=None):
 292   """Return an EstimatorSpec for the eval case."""
 293   scalar_loss = gan_loss.generator_loss + gan_loss.discriminator_loss
 294   with ops.name_scope(None, 'metrics',
 295                       [gan_loss.generator_loss,
 296                        gan_loss.discriminator_loss]):
 297     def _summary_key(head_name, val):
 298       return '%s/%s' % (val, head_name) if head_name else val
 299     eval_metric_ops = {
 300         _summary_key(name, 'generator_loss'):
 301             metrics_lib.mean(gan_loss.generator_loss),
 302         _summary_key(name, 'discriminator_loss'):
 303             metrics_lib.mean(gan_loss.discriminator_loss)
 304     }
 305     if get_eval_metric_ops_fn is not None:
 306       custom_eval_metric_ops = get_eval_metric_ops_fn(gan_model)
 307       if not isinstance(custom_eval_metric_ops, dict):
 308         raise TypeError('get_eval_metric_ops_fn must return a dict, '
 309                         'received: {}'.format(custom_eval_metric_ops))
 310       eval_metric_ops.update(custom_eval_metric_ops)
 311   return model_fn_lib.EstimatorSpec(
 312       mode=model_fn_lib.ModeKeys.EVAL,
 313       predictions=gan_model.generated_data,
 314       loss=scalar_loss,
 315       eval_metric_ops=eval_metric_ops)

 318 def _get_train_estimator_spec(
 319     gan_model, gan_loss, generator_optimizer, discriminator_optimizer,
 320     get_hooks_fn, train_op_fn=tfgan_train.gan_train_ops):
 321   """Return an EstimatorSpec for the train case."""
 322   scalar_loss = gan_loss.generator_loss + gan_loss.discriminator_loss
 323   train_ops = train_op_fn(gan_model, gan_loss, generator_optimizer,
 324                           discriminator_optimizer)
 325   training_hooks = get_hooks_fn(train_ops)
 326   return model_fn_lib.EstimatorSpec(
 327       loss=scalar_loss,
 328       mode=model_fn_lib.ModeKeys.TRAIN,
 329       train_op=train_ops.global_step_inc_op,
 330       training_hooks=training_hooks)
