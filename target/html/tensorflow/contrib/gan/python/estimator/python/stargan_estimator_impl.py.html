  51 class StarGANEstimator(estimator.Estimator):
  52   """An estimator for Generative Adversarial Networks (GANs).
  53 
  54   This Estimator is backed by TFGAN. The network functions follow the TFGAN API
  55   except for one exception: if either `generator_fn` or `discriminator_fn` have
  56   an argument called `mode`, then the tf.Estimator mode is passed in for that
  57   argument. This helps with operations like batch normalization, which have
  58   different train and evaluation behavior.
  59 
  60   Example:
  61 
  62   ```python
  63       import tensorflow as tf
  64       tfgan = tf.contrib.gan
  65 
  66       # See TFGAN's `train.py` for a description of the generator and
  67       # discriminator API.
  68       def generator_fn(generator_inputs):
  69         ...
  70         return generated_data
  71 
  72       def discriminator_fn(data, conditioning):
  73         ...
  74         return logits
  75 
  76       # Create GAN estimator.
  77       stargan_estimator = tfgan.estimator.StarGANEstimator(
  78           model_dir,
  79           generator_fn=generator_fn,
  80           discriminator_fn=discriminator_fn,
  81           loss_fn=loss_fn,
  82           generator_optimizer=tf.train.AdamOptimizer(0.1, 0.5),
  83           discriminator_optimizer=tf.train.AdamOptimizer(0.1, 0.5))
  84 
  85       # Train estimator.
  86       stargan_estimator.train(train_input_fn, steps)
  87 
  88       # Evaluate resulting estimator.
  89       stargan_estimator.evaluate(eval_input_fn)
  90 
  91       # Generate samples from generator.
  92       stargan_estimator = np.array([
  93           x for x in stargan_estimator.predict(predict_input_fn)])
  94   ```
  95   """
  96 
  97   def __init__(self,
  98                model_dir=None,
  99                generator_fn=None,
 100                discriminator_fn=None,
 101                loss_fn=None,
 102                generator_optimizer=None,
 103                discriminator_optimizer=None,
 104                get_hooks_fn=None,
 105                get_eval_metric_ops_fn=None,
 106                add_summaries=None,
 107                use_loss_summaries=True,
 108                config=None):
 109     """Initializes a StarGANEstimator instance.
 110 
 111     Args:
 112       model_dir: Directory to save model parameters, graph and etc. This can
 113         also be used to load checkpoints from the directory into a estimator to
 114         continue training a previously saved model.
 115       generator_fn: A python function that takes a Tensor, Tensor list, or
 116         Tensor dictionary as inputs and returns the outputs of the GAN
 117         generator. See `TFGAN` for more details and examples. Additionally, if
 118         it has an argument called `mode`, the Estimator's `mode` will be passed
 119         in (ex TRAIN, EVAL, PREDICT). This is useful for things like batch
 120         normalization.
 121       discriminator_fn: A python function that takes the output of
 122         `generator_fn` or real data in the GAN setup, and `input_data`. Outputs
 123         a Tensor in the range [-inf, inf]. See `TFGAN` for more details and
 124         examples.
 125       loss_fn: The loss function on the generator. Takes a `StarGANModel`
 126         namedtuple and return a `GANLoss` namedtuple.
 127       generator_optimizer: The optimizer for generator updates, or a function
 128         that takes no arguments and returns an optimizer. This function will be
 129         called when the default graph is the `StarGANEstimator`'s graph, so
 130         utilities like `tf.contrib.framework.get_or_create_global_step` will
 131         work.
 132       discriminator_optimizer: Same as `generator_optimizer`, but for the
 133         discriminator updates.
 134       get_hooks_fn: A function that takes a `GANTrainOps` tuple and returns a
 135         list of hooks. These hooks are run on the generator and discriminator
 136         train ops, and can be used to implement the GAN training scheme.
 137         Defaults to `train.get_sequential_train_hooks()`.
 138       get_eval_metric_ops_fn: A function that takes a `GANModel`, and returns a
 139         dict of metric results keyed by name. The output of this function is
 140         passed into `tf.estimator.EstimatorSpec` during evaluation.
 141       add_summaries: `None`, a single `SummaryType`, or a list of `SummaryType`.
 142       use_loss_summaries: If `True`, add loss summaries. If `False`, does not.
 143         If `None`, uses defaults.
 144       config: `RunConfig` object to configure the runtime settings.
 145 
 146     Raises:
 147       ValueError: If loss functions aren't callable.
 148       ValueError: If `use_loss_summaries` isn't boolean or `None`.
 149       ValueError: If `get_hooks_fn` isn't callable or `None`.
 150     """
 151     if not callable(loss_fn):
 152       raise ValueError('loss_fn must be callable.')
 153     if use_loss_summaries not in [True, False, None]:
 154       raise ValueError('use_loss_summaries must be True, False or None.')
 155     if get_hooks_fn is not None and not callable(get_hooks_fn):
 156       raise TypeError('get_hooks_fn must be callable.')
 157 
 158     def _model_fn(features, labels, mode):
 159       """StarGANEstimator model function."""
 160       if mode not in [
 161           model_fn_lib.ModeKeys.TRAIN, model_fn_lib.ModeKeys.EVAL,
 162           model_fn_lib.ModeKeys.PREDICT
 163       ]:
 164         raise ValueError('Mode not recognized: %s' % mode)
 165 
 166       if mode == model_fn_lib.ModeKeys.PREDICT:
 167         input_data = features[0]
 168         input_data_domain_label = features[1]
 169       else:
 170         input_data = features  # rename inputs for clarity
 171         input_data_domain_label = labels  # rename inputs for clarity
 172 
 173       # Make StarGANModel, which encapsulates the GAN model architectures.
 174       gan_model = _get_gan_model(mode, generator_fn, discriminator_fn,
 175                                  input_data, input_data_domain_label,
 176                                  add_summaries)
 177 
 178       # Make the EstimatorSpec, which incorporates the StarGANModel, losses,
 179       # eval, metrics, and optimizers (if required).
 180       return _get_estimator_spec(mode, gan_model, loss_fn,
 181                                  get_eval_metric_ops_fn, generator_optimizer,
 182                                  discriminator_optimizer, get_hooks_fn)
 183 
 184     super(StarGANEstimator, self).__init__(
 185         model_fn=_model_fn, model_dir=model_dir, config=config)
 186

 188 def _get_gan_model(mode,
 189                    generator_fn,
 190                    discriminator_fn,
 191                    input_data,
 192                    input_data_domain_label,
 193                    add_summaries,
 194                    generator_scope='Generator'):
 195   """Makes the StarGANModel tuple."""
 196   if mode == model_fn_lib.ModeKeys.PREDICT:
 197     gan_model = _make_prediction_gan_model(input_data, input_data_domain_label,
 198                                            generator_fn, generator_scope)
 199   else:  # model_fn_lib.ModeKeys.TRAIN or model_fn_lib.ModeKeys.EVAL
 200     gan_model = _make_gan_model(generator_fn, discriminator_fn, input_data,
 201                                 input_data_domain_label, generator_scope,
 202                                 add_summaries, mode)
 203 
 204   return gan_model

 207 def _get_estimator_spec(mode,
 208                         gan_model,
 209                         loss_fn,
 210                         get_eval_metric_ops_fn,
 211                         generator_optimizer,
 212                         discriminator_optimizer,
 213                         get_hooks_fn=None):
 214   """Get the EstimatorSpec for the current mode."""
 215   if mode == model_fn_lib.ModeKeys.PREDICT:
 216     estimator_spec = model_fn_lib.EstimatorSpec(
 217         mode=mode, predictions=gan_model.generated_data)
 218   else:
 219     gan_loss = loss_fn(gan_model)
 220     if mode == model_fn_lib.ModeKeys.EVAL:
 221       estimator_spec = _get_eval_estimator_spec(gan_model, gan_loss,
 222                                                 get_eval_metric_ops_fn)
 223     else:  # model_fn_lib.ModeKeys.TRAIN:
 224       gopt = (
 225           generator_optimizer()
 226           if callable(generator_optimizer) else generator_optimizer)
 227       dopt = (
 228           discriminator_optimizer()
 229           if callable(discriminator_optimizer) else discriminator_optimizer)
 230       get_hooks_fn = get_hooks_fn or tfgan_train.get_sequential_train_hooks()
 231       estimator_spec = _get_train_estimator_spec(gan_model, gan_loss, gopt,
 232                                                  dopt, get_hooks_fn)
 233 
 234   return estimator_spec

 237 def _make_gan_model(generator_fn, discriminator_fn, input_data,
 238                     input_data_domain_label, generator_scope, add_summaries,
 239                     mode):
 240   """Construct a `StarGANModel`, and optionally pass in `mode`."""
 241   # If network functions have an argument `mode`, pass mode to it.
 242   if 'mode' in inspect.getargspec(generator_fn).args:
 243     generator_fn = functools.partial(generator_fn, mode=mode)
 244   if 'mode' in inspect.getargspec(discriminator_fn).args:
 245     discriminator_fn = functools.partial(discriminator_fn, mode=mode)
 246   gan_model = tfgan_train.stargan_model(
 247       generator_fn,
 248       discriminator_fn,
 249       input_data,
 250       input_data_domain_label,
 251       generator_scope=generator_scope)
 252   if add_summaries:
 253     if not isinstance(add_summaries, (tuple, list)):
 254       add_summaries = [add_summaries]
 255     with ops.name_scope(None):
 256       for summary_type in add_summaries:
 257         _summary_type_map[summary_type](gan_model)
 258 
 259   return gan_model

 262 def _make_prediction_gan_model(input_data, input_data_domain_label,
 263                                generator_fn, generator_scope):
 264   """Make a `StarGANModel` from just the generator."""
 265   # If `generator_fn` has an argument `mode`, pass mode to it.
 266   if 'mode' in inspect.getargspec(generator_fn).args:
 267     generator_fn = functools.partial(
 268         generator_fn, mode=model_fn_lib.ModeKeys.PREDICT)
 269   with variable_scope.variable_scope(generator_scope) as gen_scope:
 270     # pylint:disable=protected-access
 271     input_data = tfgan_train._convert_tensor_or_l_or_d(input_data)
 272     input_data_domain_label = tfgan_train._convert_tensor_or_l_or_d(
 273         input_data_domain_label)
 274     # pylint:enable=protected-access
 275     generated_data = generator_fn(input_data, input_data_domain_label)
 276   generator_variables = variable_lib.get_trainable_variables(gen_scope)
 277 
 278   return tfgan_tuples.StarGANModel(
 279       input_data=input_data,
 280       input_data_domain_label=None,
 281       generated_data=generated_data,
 282       generated_data_domain_target=input_data_domain_label,
 283       reconstructed_data=None,
 284       discriminator_input_data_source_predication=None,
 285       discriminator_generated_data_source_predication=None,
 286       discriminator_input_data_domain_predication=None,
 287       discriminator_generated_data_domain_predication=None,
 288       generator_variables=generator_variables,
 289       generator_scope=generator_scope,
 290       generator_fn=generator_fn,
 291       discriminator_variables=None,
 292       discriminator_scope=None,
 293       discriminator_fn=None)

 296 def _get_eval_estimator_spec(gan_model,
 297                              gan_loss,
 298                              get_eval_metric_ops_fn=None,
 299                              name=None):
 300   """Return an EstimatorSpec for the eval case."""
 301   scalar_loss = gan_loss.generator_loss + gan_loss.discriminator_loss
 302   with ops.name_scope(None, 'metrics',
 303                       [gan_loss.generator_loss, gan_loss.discriminator_loss]):
 304 
 305     def _summary_key(head_name, val):
 306       return '%s/%s' % (val, head_name) if head_name else val
 307 
 308     eval_metric_ops = {
 309         _summary_key(name, 'generator_loss'):
 310             metrics_lib.mean(gan_loss.generator_loss),
 311         _summary_key(name, 'discriminator_loss'):
 312             metrics_lib.mean(gan_loss.discriminator_loss)
 313     }
 314     if get_eval_metric_ops_fn is not None:
 315       custom_eval_metric_ops = get_eval_metric_ops_fn(gan_model)
 316       if not isinstance(custom_eval_metric_ops, dict):
 317         raise TypeError('get_eval_metric_ops_fn must return a dict, '
 318                         'received: {}'.format(custom_eval_metric_ops))
 319       eval_metric_ops.update(custom_eval_metric_ops)
 320   return model_fn_lib.EstimatorSpec(
 321       mode=model_fn_lib.ModeKeys.EVAL,
 322       predictions=gan_model.generated_data,
 323       loss=scalar_loss,
 324       eval_metric_ops=eval_metric_ops)

 327 def _get_train_estimator_spec(gan_model,
 328                               gan_loss,
 329                               generator_optimizer,
 330                               discriminator_optimizer,
 331                               get_hooks_fn,
 332                               train_op_fn=tfgan_train.gan_train_ops):
 333   """Return an EstimatorSpec for the train case."""
 334   scalar_loss = gan_loss.generator_loss + gan_loss.discriminator_loss
 335   train_ops = train_op_fn(gan_model, gan_loss, generator_optimizer,
 336                           discriminator_optimizer)
 337   training_hooks = get_hooks_fn(train_ops)
 338   return model_fn_lib.EstimatorSpec(
 339       loss=scalar_loss,
 340       mode=model_fn_lib.ModeKeys.TRAIN,
 341       train_op=train_ops.global_step_inc_op,
 342       training_hooks=training_hooks)

 345 def stargan_prediction_input_fn_wrapper(fn):
 346   """StarGAN Estimator prediction input_fn wrapper.
 347 
 348   Since estimator will disregard the "label" variable pass to the model, we will
 349   use a wrapper to pack the (feature, label) tuple as feature passed to the
 350   model.
 351 
 352   Args:
 353     fn: input_fn for the prediction.
 354 
 355   Returns:
 356     A tuple ((feature, label), None) where the second element is the dummy label
 357     to be disregarded and the first element is the true input to the estimator.
 358   """
 359 
 360   def new_fn():
 361     return fn(), None
 362 
 363   return new_fn
