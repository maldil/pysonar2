  32 def hann_window(window_length, periodic=True, dtype=dtypes.float32, name=None):
  33   """Generate a [Hann window][hann].
  34 
  35   Args:
  36     window_length: A scalar `Tensor` indicating the window length to generate.
  37     periodic: A bool `Tensor` indicating whether to generate a periodic or
  38       symmetric window. Periodic windows are typically used for spectral
  39       analysis while symmetric windows are typically used for digital
  40       filter design.
  41     dtype: The data type to produce. Must be a floating point type.
  42     name: An optional name for the operation.
  43 
  44   Returns:
  45     A `Tensor` of shape `[window_length]` of type `dtype`.
  46 
  47   Raises:
  48     ValueError: If `dtype` is not a floating point type.
  49 
  50   [hann]: https://en.wikipedia.org/wiki/Window_function#Hann_and_Hamming_windows
  51   """
  52   return _raised_cosine_window(name, 'hann_window', window_length, periodic,
  53                                dtype, 0.5, 0.5)

  56 def hamming_window(window_length, periodic=True, dtype=dtypes.float32,
  57                    name=None):
  58   """Generate a [Hamming][hamming] window.
  59 
  60   Args:
  61     window_length: A scalar `Tensor` indicating the window length to generate.
  62     periodic: A bool `Tensor` indicating whether to generate a periodic or
  63       symmetric window. Periodic windows are typically used for spectral
  64       analysis while symmetric windows are typically used for digital
  65       filter design.
  66     dtype: The data type to produce. Must be a floating point type.
  67     name: An optional name for the operation.
  68 
  69   Returns:
  70     A `Tensor` of shape `[window_length]` of type `dtype`.
  71 
  72   Raises:
  73     ValueError: If `dtype` is not a floating point type.
  74 
  75   [hamming]: https://en.wikipedia.org/wiki/Window_function#Hann_and_Hamming_windows
  76   """
  77   return _raised_cosine_window(name, 'hamming_window', window_length, periodic,
  78                                dtype, 0.54, 0.46)

  81 def _raised_cosine_window(name, default_name, window_length, periodic,
  82                           dtype, a, b):
  83   """Helper function for computing a raised cosine window.
  84 
  85   Args:
  86     name: Name to use for the scope.
  87     default_name: Default name to use for the scope.
  88     window_length: A scalar `Tensor` or integer indicating the window length.
  89     periodic: A bool `Tensor` indicating whether to generate a periodic or
  90       symmetric window.
  91     dtype: A floating point `DType`.
  92     a: The alpha parameter to the raised cosine window.
  93     b: The beta parameter to the raised cosine window.
  94 
  95   Returns:
  96     A `Tensor` of shape `[window_length]` of type `dtype`.
  97 
  98   Raises:
  99     ValueError: If `dtype` is not a floating point type or `window_length` is
 100       not scalar or `periodic` is not scalar.
 101   """
 102   if not dtype.is_floating:
 103     raise ValueError('dtype must be a floating point type. Found %s' % dtype)
 104 
 105   with ops.name_scope(name, default_name, [window_length, periodic]):
 106     window_length = ops.convert_to_tensor(window_length, dtype=dtypes.int32,
 107                                           name='window_length')
 108     window_length.shape.assert_has_rank(0)
 109     window_length_const = tensor_util.constant_value(window_length)
 110     if window_length_const == 1:
 111       return array_ops.ones([1], dtype=dtype)
 112     periodic = math_ops.cast(
 113         ops.convert_to_tensor(periodic, dtype=dtypes.bool, name='periodic'),
 114         dtypes.int32)
 115     periodic.shape.assert_has_rank(0)
 116     even = 1 - math_ops.mod(window_length, 2)
 117 
 118     n = math_ops.cast(window_length + periodic * even - 1, dtype=dtype)
 119     count = math_ops.cast(math_ops.range(window_length), dtype)
 120     cos_arg = constant_op.constant(2 * np.pi, dtype=dtype) * count / n
 121 
 122     if window_length_const is not None:
 123       return math_ops.cast(a - b * math_ops.cos(cos_arg), dtype=dtype)
 124     return control_flow_ops.cond(
 125         math_ops.equal(window_length, 1),
 126         lambda: array_ops.ones([1], dtype=dtype),
 127         lambda: math_ops.cast(a - b * math_ops.cos(cos_arg), dtype=dtype))
