  30 def gcd(a, b, name=None):
  31   """Returns the greatest common divisor via Euclid's algorithm.
  32 
  33   Args:
  34     a: The dividend. A scalar integer `Tensor`.
  35     b: The divisor. A scalar integer `Tensor`.
  36     name: An optional name for the operation.
  37 
  38   Returns:
  39     A scalar `Tensor` representing the greatest common divisor between `a` and
  40     `b`.
  41 
  42   Raises:
  43     ValueError: If `a` or `b` are not scalar integers.
  44   """
  45   with ops.name_scope(name, 'gcd', [a, b]):
  46     a = ops.convert_to_tensor(a)
  47     b = ops.convert_to_tensor(b)
  48 
  49     a.shape.assert_has_rank(0)
  50     b.shape.assert_has_rank(0)
  51 
  52     if not a.dtype.is_integer:
  53       raise ValueError('a must be an integer type. Got: %s' % a.dtype)
  54     if not b.dtype.is_integer:
  55       raise ValueError('b must be an integer type. Got: %s' % b.dtype)
  56 
  57     # TPU requires static shape inference. GCD is used for subframe size
  58     # computation, so we should prefer static computation where possible.
  59     const_a = tensor_util.constant_value(a)
  60     const_b = tensor_util.constant_value(b)
  61     if const_a is not None and const_b is not None:
  62       return ops.convert_to_tensor(fractions.gcd(const_a, const_b))
  63 
  64     cond = lambda _, b: math_ops.greater(b, array_ops.zeros_like(b))
  65     body = lambda a, b: [b, math_ops.mod(a, b)]
  66     a, b = control_flow_ops.while_loop(cond, body, [a, b], back_prop=False)
  67     return a
