  27 # mel spectrum constants.
  28 _MEL_BREAK_FREQUENCY_HERTZ = 700.0
  29 _MEL_HIGH_FREQUENCY_Q = 1127.0
  30

  32 def _mel_to_hertz(mel_values, name=None):
  33   """Converts frequencies in `mel_values` from the mel scale to linear scale.
  34 
  35   Args:
  36     mel_values: A `Tensor` of frequencies in the mel scale.
  37     name: An optional name for the operation.
  38 
  39   Returns:
  40     A `Tensor` of the same shape and type as `mel_values` containing linear
  41     scale frequencies in Hertz.
  42   """
  43   with ops.name_scope(name, 'mel_to_hertz', [mel_values]):
  44     mel_values = ops.convert_to_tensor(mel_values)
  45     return _MEL_BREAK_FREQUENCY_HERTZ * (
  46         math_ops.exp(mel_values / _MEL_HIGH_FREQUENCY_Q) - 1.0
  47     )
  48

  50 def _hertz_to_mel(frequencies_hertz, name=None):
  51   """Converts frequencies in `frequencies_hertz` in Hertz to the mel scale.
  52 
  53   Args:
  54     frequencies_hertz: A `Tensor` of frequencies in Hertz.
  55     name: An optional name for the operation.
  56 
  57   Returns:
  58     A `Tensor` of the same shape and type of `frequencies_hertz` containing
  59     frequencies in the mel scale.
  60   """
  61   with ops.name_scope(name, 'hertz_to_mel', [frequencies_hertz]):
  62     frequencies_hertz = ops.convert_to_tensor(frequencies_hertz)
  63     return _MEL_HIGH_FREQUENCY_Q * math_ops.log(
  64         1.0 + (frequencies_hertz / _MEL_BREAK_FREQUENCY_HERTZ))

  67 def _validate_arguments(num_mel_bins, sample_rate,
  68                         lower_edge_hertz, upper_edge_hertz, dtype):
  69   """Checks the inputs to linear_to_mel_weight_matrix."""
  70   if num_mel_bins <= 0:
  71     raise ValueError('num_mel_bins must be positive. Got: %s' % num_mel_bins)
  72   if sample_rate <= 0.0:
  73     raise ValueError('sample_rate must be positive. Got: %s' % sample_rate)
  74   if lower_edge_hertz < 0.0:
  75     raise ValueError('lower_edge_hertz must be non-negative. Got: %s' %
  76                      lower_edge_hertz)
  77   if lower_edge_hertz >= upper_edge_hertz:
  78     raise ValueError('lower_edge_hertz %.1f >= upper_edge_hertz %.1f' %
  79                      (lower_edge_hertz, upper_edge_hertz))
  80   if upper_edge_hertz > sample_rate / 2:
  81     raise ValueError('upper_edge_hertz must not be larger than the Nyquist '
  82                      'frequency (sample_rate / 2). Got: %s for sample_rate: %s'
  83                      % (upper_edge_hertz, sample_rate))
  84   if not dtype.is_floating:
  85     raise ValueError('dtype must be a floating point type. Got: %s' % dtype)
  86

  88 def linear_to_mel_weight_matrix(num_mel_bins=20,
  89                                 num_spectrogram_bins=129,
  90                                 sample_rate=8000,
  91                                 lower_edge_hertz=125.0,
  92                                 upper_edge_hertz=3800.0,
  93                                 dtype=dtypes.float32,
  94                                 name=None):
  95   """Returns a matrix to warp linear scale spectrograms to the [mel scale][mel].
  96 
  97   Returns a weight matrix that can be used to re-weight a `Tensor` containing
  98   `num_spectrogram_bins` linearly sampled frequency information from
  99   `[0, sample_rate / 2]` into `num_mel_bins` frequency information from
 100   `[lower_edge_hertz, upper_edge_hertz]` on the [mel scale][mel].
 101 
 102   For example, the returned matrix `A` can be used to right-multiply a
 103   spectrogram `S` of shape `[frames, num_spectrogram_bins]` of linear
 104   scale spectrum values (e.g. STFT magnitudes) to generate a "mel spectrogram"
 105   `M` of shape `[frames, num_mel_bins]`.
 106 
 107       # `S` has shape [frames, num_spectrogram_bins]
 108       # `M` has shape [frames, num_mel_bins]
 109       M = tf.matmul(S, A)
 110 
 111   The matrix can be used with `tf.tensordot` to convert an arbitrary rank
 112   `Tensor` of linear-scale spectral bins into the mel scale.
 113 
 114       # S has shape [..., num_spectrogram_bins].
 115       # M has shape [..., num_mel_bins].
 116       M = tf.tensordot(S, A, 1)
 117       # tf.tensordot does not support shape inference for this case yet.
 118       M.set_shape(S.shape[:-1].concatenate(A.shape[-1:]))
 119 
 120   Args:
 121     num_mel_bins: Python int. How many bands in the resulting mel spectrum.
 122     num_spectrogram_bins: An integer `Tensor`. How many bins there are in the
 123       source spectrogram data, which is understood to be `fft_size // 2 + 1`,
 124       i.e. the spectrogram only contains the nonredundant FFT bins.
 125     sample_rate: Python float. Samples per second of the input signal used to
 126       create the spectrogram. We need this to figure out the actual frequencies
 127       for each spectrogram bin, which dictates how they are mapped into the mel
 128       scale.
 129     lower_edge_hertz: Python float. Lower bound on the frequencies to be
 130       included in the mel spectrum. This corresponds to the lower edge of the
 131       lowest triangular band.
 132     upper_edge_hertz: Python float. The desired top edge of the highest
 133       frequency band.
 134     dtype: The `DType` of the result matrix. Must be a floating point type.
 135     name: An optional name for the operation.
 136 
 137   Returns:
 138     A `Tensor` of shape `[num_spectrogram_bins, num_mel_bins]`.
 139 
 140   Raises:
 141     ValueError: If num_mel_bins/num_spectrogram_bins/sample_rate are not
 142       positive, lower_edge_hertz is negative, frequency edges are incorrectly
 143       ordered, or upper_edge_hertz is larger than the Nyquist frequency.
 144 
 145   [mel]: https://en.wikipedia.org/wiki/Mel_scale
 146   """
 147   with ops.name_scope(name, 'linear_to_mel_weight_matrix') as name:
 148     # Note: As num_spectrogram_bins is passed to `math_ops.linspace`
 149     # and the validation is already done in linspace (both in shape function
 150     # and in kernel), there is no need to validate num_spectrogram_bins here.
 151     _validate_arguments(num_mel_bins, sample_rate,
 152                         lower_edge_hertz, upper_edge_hertz, dtype)
 153 
 154     # This function can be constant folded by graph optimization since there are
 155     # no Tensor inputs.
 156     sample_rate = ops.convert_to_tensor(
 157         sample_rate, dtype, name='sample_rate')
 158     lower_edge_hertz = ops.convert_to_tensor(
 159         lower_edge_hertz, dtype, name='lower_edge_hertz')
 160     upper_edge_hertz = ops.convert_to_tensor(
 161         upper_edge_hertz, dtype, name='upper_edge_hertz')
 162     zero = ops.convert_to_tensor(0.0, dtype)
 163 
 164     # HTK excludes the spectrogram DC bin.
 165     bands_to_zero = 1
 166     nyquist_hertz = sample_rate / 2.0
 167     linear_frequencies = math_ops.linspace(
 168         zero, nyquist_hertz, num_spectrogram_bins)[bands_to_zero:]
 169     spectrogram_bins_mel = array_ops.expand_dims(
 170         _hertz_to_mel(linear_frequencies), 1)
 171 
 172     # Compute num_mel_bins triples of (lower_edge, center, upper_edge). The
 173     # center of each band is the lower and upper edge of the adjacent bands.
 174     # Accordingly, we divide [lower_edge_hertz, upper_edge_hertz] into
 175     # num_mel_bins + 2 pieces.
 176     band_edges_mel = shape_ops.frame(
 177         math_ops.linspace(_hertz_to_mel(lower_edge_hertz),
 178                           _hertz_to_mel(upper_edge_hertz),
 179                           num_mel_bins + 2), frame_length=3, frame_step=1)
 180 
 181     # Split the triples up and reshape them into [1, num_mel_bins] tensors.
 182     lower_edge_mel, center_mel, upper_edge_mel = tuple(array_ops.reshape(
 183         t, [1, num_mel_bins]) for t in array_ops.split(
 184             band_edges_mel, 3, axis=1))
 185 
 186     # Calculate lower and upper slopes for every spectrogram bin.
 187     # Line segments are linear in the mel domain, not Hertz.
 188     lower_slopes = (spectrogram_bins_mel - lower_edge_mel) / (
 189         center_mel - lower_edge_mel)
 190     upper_slopes = (upper_edge_mel - spectrogram_bins_mel) / (
 191         upper_edge_mel - center_mel)
 192 
 193     # Intersect the line segments with each other and zero.
 194     mel_weights_matrix = math_ops.maximum(
 195         zero, math_ops.minimum(lower_slopes, upper_slopes))
 196 
 197     # Re-add the zeroed lower bins we sliced out above.
 198     return array_ops.pad(
 199         mel_weights_matrix, [[bands_to_zero, 0], [0, 0]], name=name)
