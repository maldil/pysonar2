  36 def stft(signals, frame_length, frame_step, fft_length=None,
  37          window_fn=functools.partial(window_ops.hann_window, periodic=True),
  38          pad_end=False, name=None):
  39   """Computes the [Short-time Fourier Transform][stft] of `signals`.
  40 
  41   Implemented with GPU-compatible ops and supports gradients.
  42 
  43   Args:
  44     signals: A `[..., samples]` `float32` `Tensor` of real-valued signals.
  45     frame_length: An integer scalar `Tensor`. The window length in samples.
  46     frame_step: An integer scalar `Tensor`. The number of samples to step.
  47     fft_length: An integer scalar `Tensor`. The size of the FFT to apply.
  48       If not provided, uses the smallest power of 2 enclosing `frame_length`.
  49     window_fn: A callable that takes a window length and a `dtype` keyword
  50       argument and returns a `[window_length]` `Tensor` of samples in the
  51       provided datatype. If set to `None`, no windowing is used.
  52     pad_end: Whether to pad the end of `signals` with zeros when the provided
  53       frame length and step produces a frame that lies partially past its end.
  54     name: An optional name for the operation.
  55 
  56   Returns:
  57     A `[..., frames, fft_unique_bins]` `Tensor` of `complex64` STFT values where
  58     `fft_unique_bins` is `fft_length // 2 + 1` (the unique components of the
  59     FFT).
  60 
  61   Raises:
  62     ValueError: If `signals` is not at least rank 1, `frame_length` is
  63       not scalar, or `frame_step` is not scalar.
  64 
  65   [stft]: https://en.wikipedia.org/wiki/Short-time_Fourier_transform
  66   """
  67   with ops.name_scope(name, 'stft', [signals, frame_length,
  68                                      frame_step]):
  69     signals = ops.convert_to_tensor(signals, name='signals')
  70     signals.shape.with_rank_at_least(1)
  71     frame_length = ops.convert_to_tensor(frame_length, name='frame_length')
  72     frame_length.shape.assert_has_rank(0)
  73     frame_step = ops.convert_to_tensor(frame_step, name='frame_step')
  74     frame_step.shape.assert_has_rank(0)
  75 
  76     if fft_length is None:
  77       fft_length = _enclosing_power_of_two(frame_length)
  78     else:
  79       fft_length = ops.convert_to_tensor(fft_length, name='fft_length')
  80 
  81     framed_signals = shape_ops.frame(
  82         signals, frame_length, frame_step, pad_end=pad_end)
  83 
  84     # Optionally window the framed signals.
  85     if window_fn is not None:
  86       window = window_fn(frame_length, dtype=framed_signals.dtype)
  87       framed_signals *= window
  88 
  89     # spectral_ops.rfft produces the (fft_length/2 + 1) unique components of the
  90     # FFT of the real windowed signals in framed_signals.
  91     return spectral_ops.rfft(framed_signals, [fft_length])

  94 def inverse_stft_window_fn(frame_step,
  95                            forward_window_fn=functools.partial(
  96                                window_ops.hann_window, periodic=True),
  97                            name=None):
  98   """Generates a window function that can be used in `inverse_stft`.
  99 
 100   Constructs a window that is equal to the forward window with a further
 101   pointwise amplitude correction.  `inverse_stft_window_fn` is equivalent to
 102   `forward_window_fn` in the case where it would produce an exact inverse.
 103 
 104   See examples in `inverse_stft` documentation for usage.
 105 
 106   Args:
 107     frame_step: An integer scalar `Tensor`. The number of samples to step.
 108     forward_window_fn: window_fn used in the forward transform, `stft`.
 109     name: An optional name for the operation.
 110 
 111   Returns:
 112     A callable that takes a window length and a `dtype` keyword argument and
 113       returns a `[window_length]` `Tensor` of samples in the provided datatype.
 114       The returned window is suitable for reconstructing original waveform in
 115       inverse_stft.
 116   """
 117   with ops.name_scope(name, 'inverse_stft_window_fn', [forward_window_fn]):
 118     frame_step = ops.convert_to_tensor(frame_step, name='frame_step')
 119     frame_step.shape.assert_has_rank(0)
 120 
 121   def inverse_stft_window_fn_inner(frame_length, dtype):
 122     """Computes a window that can be used in `inverse_stft`.
 123 
 124     Args:
 125       frame_length: An integer scalar `Tensor`. The window length in samples.
 126       dtype: Data type of waveform passed to `stft`.
 127 
 128     Returns:
 129       A window suitable for reconstructing original waveform in `inverse_stft`.
 130 
 131     Raises:
 132       ValueError: If `frame_length` is not scalar, `forward_window_fn` is not a
 133       callable that takes a window length and a `dtype` keyword argument and
 134       returns a `[window_length]` `Tensor` of samples in the provided datatype
 135       `frame_step` is not scalar, or `frame_step` is not scalar.
 136     """
 137     with ops.name_scope(name, 'inverse_stft_window_fn', [forward_window_fn]):
 138       frame_length = ops.convert_to_tensor(frame_length, name='frame_length')
 139       frame_length.shape.assert_has_rank(0)
 140 
 141       # Use equation 7 from Griffin + Lim.
 142       forward_window = forward_window_fn(frame_length, dtype=dtype)
 143       denom = math_ops.square(forward_window)
 144       overlaps = -(-frame_length // frame_step)  # Ceiling division.
 145       denom = array_ops.pad(denom, [(0, overlaps * frame_step - frame_length)])
 146       denom = array_ops.reshape(denom, [overlaps, frame_step])
 147       denom = math_ops.reduce_sum(denom, 0, keepdims=True)
 148       denom = array_ops.tile(denom, [overlaps, 1])
 149       denom = array_ops.reshape(denom, [overlaps * frame_step])
 150 
 151       return forward_window / denom[:frame_length]
 152   return inverse_stft_window_fn_inner

 155 def inverse_stft(stfts,
 156                  frame_length,
 157                  frame_step,
 158                  fft_length=None,
 159                  window_fn=functools.partial(window_ops.hann_window,
 160                                              periodic=True),
 161                  name=None):
 162   """Computes the inverse [Short-time Fourier Transform][stft] of `stfts`.
 163 
 164   To reconstruct an original waveform, a complimentary window function should
 165   be used in inverse_stft. Such a window function can be constructed with
 166   tf.contrib.signal.inverse_stft_window_fn.
 167 
 168   Example:
 169 
 170   ```python
 171   frame_length = 400
 172   frame_step = 160
 173   waveform = tf.placeholder(dtype=tf.float32, shape=[1000])
 174   stft = tf.contrib.signal.stft(waveform, frame_length, frame_step)
 175   inverse_stft = tf.contrib.signal.inverse_stft(
 176       stft, frame_length, frame_step,
 177       window_fn=tf.contrib.signal.inverse_stft_window_fn(frame_step))
 178   ```
 179 
 180   if a custom window_fn is used in stft, it must be passed to
 181   inverse_stft_window_fn:
 182 
 183   ```python
 184   frame_length = 400
 185   frame_step = 160
 186   window_fn = functools.partial(window_ops.hamming_window, periodic=True),
 187   waveform = tf.placeholder(dtype=tf.float32, shape=[1000])
 188   stft = tf.contrib.signal.stft(
 189       waveform, frame_length, frame_step, window_fn=window_fn)
 190   inverse_stft = tf.contrib.signal.inverse_stft(
 191       stft, frame_length, frame_step,
 192       window_fn=tf.contrib.signal.inverse_stft_window_fn(
 193          frame_step, forward_window_fn=window_fn))
 194   ```
 195 
 196   Implemented with GPU-compatible ops and supports gradients.
 197 
 198   Args:
 199     stfts: A `complex64` `[..., frames, fft_unique_bins]` `Tensor` of STFT bins
 200       representing a batch of `fft_length`-point STFTs where `fft_unique_bins`
 201       is `fft_length // 2 + 1`
 202     frame_length: An integer scalar `Tensor`. The window length in samples.
 203     frame_step: An integer scalar `Tensor`. The number of samples to step.
 204     fft_length: An integer scalar `Tensor`. The size of the FFT that produced
 205       `stfts`. If not provided, uses the smallest power of 2 enclosing
 206       `frame_length`.
 207     window_fn: A callable that takes a window length and a `dtype` keyword
 208       argument and returns a `[window_length]` `Tensor` of samples in the
 209       provided datatype. If set to `None`, no windowing is used.
 210     name: An optional name for the operation.
 211 
 212   Returns:
 213     A `[..., samples]` `Tensor` of `float32` signals representing the inverse
 214     STFT for each input STFT in `stfts`.
 215 
 216   Raises:
 217     ValueError: If `stfts` is not at least rank 2, `frame_length` is not scalar,
 218       `frame_step` is not scalar, or `fft_length` is not scalar.
 219 
 220   [stft]: https://en.wikipedia.org/wiki/Short-time_Fourier_transform
 221   """
 222   with ops.name_scope(name, 'inverse_stft', [stfts]):
 223     stfts = ops.convert_to_tensor(stfts, name='stfts')
 224     stfts.shape.with_rank_at_least(2)
 225     frame_length = ops.convert_to_tensor(frame_length, name='frame_length')
 226     frame_length.shape.assert_has_rank(0)
 227     frame_step = ops.convert_to_tensor(frame_step, name='frame_step')
 228     frame_step.shape.assert_has_rank(0)
 229     if fft_length is None:
 230       fft_length = _enclosing_power_of_two(frame_length)
 231     else:
 232       fft_length = ops.convert_to_tensor(fft_length, name='fft_length')
 233       fft_length.shape.assert_has_rank(0)
 234 
 235     real_frames = spectral_ops.irfft(stfts, [fft_length])
 236 
 237     # frame_length may be larger or smaller than fft_length, so we pad or
 238     # truncate real_frames to frame_length.
 239     frame_length_static = tensor_util.constant_value(frame_length)
 240     # If we don't know the shape of real_frames's inner dimension, pad and
 241     # truncate to frame_length.
 242     if (frame_length_static is None or
 243         real_frames.shape.ndims is None or
 244         real_frames.shape[-1].value is None):
 245       real_frames = real_frames[..., :frame_length]
 246       real_frames_rank = array_ops.rank(real_frames)
 247       real_frames_shape = array_ops.shape(real_frames)
 248       paddings = array_ops.concat(
 249           [array_ops.zeros([real_frames_rank - 1, 2],
 250                            dtype=frame_length.dtype),
 251            [[0, math_ops.maximum(0, frame_length - real_frames_shape[-1])]]], 0)
 252       real_frames = array_ops.pad(real_frames, paddings)
 253     # We know real_frames's last dimension and frame_length statically. If they
 254     # are different, then pad or truncate real_frames to frame_length.
 255     elif real_frames.shape[-1].value > frame_length_static:
 256       real_frames = real_frames[..., :frame_length_static]
 257     elif real_frames.shape[-1].value < frame_length_static:
 258       pad_amount = frame_length_static - real_frames.shape[-1].value
 259       real_frames = array_ops.pad(real_frames,
 260                                   [[0, 0]] * (real_frames.shape.ndims - 1) +
 261                                   [[0, pad_amount]])
 262 
 263     # The above code pads the inner dimension of real_frames to frame_length,
 264     # but it does so in a way that may not be shape-inference friendly.
 265     # Restore shape information if we are able to.
 266     if frame_length_static is not None and real_frames.shape.ndims is not None:
 267       real_frames.set_shape([None] * (real_frames.shape.ndims - 1) +
 268                             [frame_length_static])
 269 
 270     # Optionally window and overlap-add the inner 2 dimensions of real_frames
 271     # into a single [samples] dimension.
 272     if window_fn is not None:
 273       window = window_fn(frame_length, dtype=stfts.dtype.real_dtype)
 274       real_frames *= window
 275     return reconstruction_ops.overlap_and_add(real_frames, frame_step)

 278 def _enclosing_power_of_two(value):
 279   """Return 2**N for integer N such that 2**N >= value."""
 280   value_static = tensor_util.constant_value(value)
 281   if value_static is not None:
 282     return constant_op.constant(
 283         int(2**np.ceil(np.log(value_static) / np.log(2.0))), value.dtype)
 284   return math_ops.cast(
 285       math_ops.pow(2.0, math_ops.ceil(
 286           math_ops.log(math_ops.to_float(value)) / math_ops.log(2.0))),
 287       value.dtype)
