 167 def _IotaNdF32Constant(dim_sizes):
 168 
 169   def MakeList(dims):
 170     if len(dims) == 1:
 171       return [float(1 + f) for f in range(dims[0])]
 172     return [MakeList(dims[1:]) for _ in range(dims[0])]
 173 
 174   return constant_op.constant(MakeList(dim_sizes), dtype=dtypes.float32)

 208   def _SetupValuesForDevice(self, tensor_in_sizes, filter_in_sizes, bias,
 209                             strides, padding, activation_mode, data_format,
 210                             filter_format, dtype):
 211     """Verifies the output values of the convolution function.
 212 
 213     Args:
 214       tensor_in_sizes: Input tensor dimensions in
 215         [batch, input_rows, input_cols, input_depth].
 216       filter_in_sizes: Filter tensor dimensions in
 217         [kernel_rows, kernel_cols, input_depth, output_depth].
 218       bias: 1-D bias tensor of length output_depth.
 219       strides: Stride: [col_stride, row_stride]
 220       padding: Padding type.
 221       activation_mode: Activation mode.
 222       data_format: Format of the data tensors.
 223       filter_format: Filter format to use for the fused convolution.
 224       dtype: Data type for inputs and outputs.
 225     Returns:
 226       Symbolic tensor value and reference value that can be used to
 227       execute the computation and verify the results.
 228     """
 229     input_size = np.prod(tensor_in_sizes)
 230     filter_size = np.prod(filter_in_sizes)
 231     bias_size = filter_in_sizes[-1]  # equals to output depth
 232     # Initializes the input tensor with array containing incrementing
 233     # numbers from 1.
 234     x1 = [f * 1.0 for f in range(1, input_size + 1)]
 235     x2 = [f * 1.0 for f in range(1, filter_size + 1)]
 236     # This is to guarantee that there are always negative values after
 237     # bias add so that we can test whether relu works correctly.
 238     x3 = bias
 239     with self.cached_session(use_gpu=True), self.test_scope():
 240       t1 = constant_op.constant(x1, shape=tensor_in_sizes, dtype=dtype)
 241       t2 = constant_op.constant(x2, shape=filter_in_sizes, dtype=dtype)
 242       fused_t2 = t2
 243       if filter_format == "OIHW":
 244         fused_t2 = _HwioToOihw(t2)
 245       t3 = constant_op.constant(x3, shape=[bias_size], dtype=dtype)
 246       strides = [1] + strides + [1]
 247       if data_format == "NCHW":
 248         t1 = test_util.NHWCToNCHW(t1)
 249         strides = test_util.NHWCToNCHW(strides)
 250       output = fused_conv2d_bias_activation_op.fused_conv2d_bias_activation(
 251           t1,
 252           fused_t2,
 253           t3,
 254           strides=strides,
 255           padding=padding,
 256           data_format=data_format,
 257           filter_format=filter_format,
 258           activation_mode=activation_mode)
 259       ref_conv_output = nn_ops.conv2d(
 260           t1, t2, strides=strides, padding=padding, data_format=data_format)
 261       ref_bias_output = nn_ops.bias_add(
 262           ref_conv_output, t3, data_format=data_format)
 263       ref_output = nn_ops.relu(ref_bias_output)
 264       if data_format == "NCHW":
 265         output = test_util.NCHWToNHWC(output)
 266         ref_output = test_util.NCHWToNHWC(ref_output)
 267 
 268       return output, ref_output

 270   def CompareFwdValues(self, tensor_in_sizes, filter_in_sizes, conv_strides,
 271                        padding):
 272     """Verifies that CPU and GPU produce the same values.
 273 
 274     Args:
 275       tensor_in_sizes: Input tensor dimensions in
 276         [batch, input_rows, input_cols, input_depth].
 277       filter_in_sizes: Filter tensor dimensions in
 278         [kernel_rows, kernel_cols, input_depth, output_depth].
 279       conv_strides: [row_stride, col_stride] for the convolution;
 280       padding: Padding type.
 281     """
 282     x1 = np.random.rand(*tensor_in_sizes).astype(np.float32)
 283     x2 = np.random.rand(*filter_in_sizes).astype(np.float32)
 284     x3 = np.random.rand(*[filter_in_sizes[-1]]).astype(np.float32)
 285 
 286     def _SetupVal(data_format, use_gpu):
 287       with self.cached_session(use_gpu=use_gpu), self.test_scope():
 288         t1 = constant_op.constant(x1, shape=tensor_in_sizes)
 289         t2 = constant_op.constant(x2, shape=filter_in_sizes)
 290         t3 = constant_op.constant(x3, shape=[filter_in_sizes[-1]])
 291         strides = [1] + conv_strides + [1]
 292         if data_format == "NCHW":
 293           t1 = test_util.NHWCToNCHW(t1)
 294           strides = test_util.NHWCToNCHW(strides)
 295         output = fused_conv2d_bias_activation_op.fused_conv2d_bias_activation(
 296             t1,
 297             t2,
 298             t3,
 299             strides=strides,
 300             padding=padding,
 301             data_format=data_format,
 302             activation_mode="Relu")
 303 
 304         if data_format == "NCHW":
 305           output = test_util.NCHWToNHWC(output)
 306         return output
 307 
 308     tensors = []
 309     for (data_format, use_gpu) in _GetTestConfigs():
 310       tensors.append(_SetupVal(data_format, use_gpu))
 311     with self.cached_session() as sess, self.test_scope():
 312       values = sess.run(tensors)
 313       for i in range(1, len(values)):
 314         self.assertAllClose(values[0], values[i], rtol=1e-3, atol=1e-3)
 315 
 316   def _VerifyValues(self, tensor_in_sizes, filter_in_sizes, bias, strides,
 317                     padding):
 318     tensors = []
 319     ref_tensors = []
 320     for (data_format, use_gpu) in _GetTestConfigs():
 321       for dtype in self._DtypesToTest(use_gpu):
 322         for filter_format in self._FilterFormatsToTest(use_gpu):
 323           result, expected = self._SetupValuesForDevice(
 324               tensor_in_sizes, filter_in_sizes, bias, strides, padding, "Relu",
 325               data_format, filter_format, dtype)
 326         tensors.append(result)
 327         ref_tensors.append(expected)
 328       with self.cached_session() as sess, self.test_scope():
 329         values = sess.run(tensors)
 330         ref_values = sess.run(ref_tensors)
 331         for i in range(len(tensors)):
 332           conv = tensors[i]
 333           value = values[i]
 334           ref_value = ref_values[i]
 335           tf_logging.info("expected = %s", ref_value)
 336           tf_logging.info("actual = %s", value)
 337           tol = 1e-5
 338           if value.dtype == np.float16:
 339             tol = 1e-3
 340           self.assertAllClose(
 341               np.ravel(ref_value), np.ravel(value), atol=tol, rtol=tol)
 342           self.assertShapeEqual(value, conv)

 616 def _CalculateConvolvedOutputDim(input_dim, filter_dim, stride, padding_type):
 617   """Calculates the size of an output dimension of a strided convolution.
 618 
 619   Given the sizes of the corresponding dimension of the input and filter shapes,
 620   and the stride and padding_types, calculates the size of the output dimension.
 621   This function can be called separately for each input dimension.
 622 
 623   Args:
 624     input_dim: An `int` specifying the size of the input dimension.
 625     filter_dim: An `int` specifying the size of the filter dimension.
 626     stride: An `int` specifying the step size of the convolution along the
 627       input dimension.
 628     padding_type: either 'VALID' or 'SAME'.
 629 
 630   Returns:
 631     The size of the output dimension.
 632   """
 633   if padding_type == "VALID":
 634     return (input_dim - filter_dim + stride) // stride
 635   else:  # padding_type == 'SAME'
 636     return (input_dim + stride - 1) // stride

 639 def _NchwVectCToNchw(in_tensor):
 640   # [N, C / 4, H, W, 4] => [N, C / 4, 4, H, W] == [N, C, H, W]
 641   t = array_ops.transpose(in_tensor, [0, 1, 4, 2, 3])
 642   n = in_tensor.shape.dims[0].value
 643   c = in_tensor.shape.dims[1].value * in_tensor.shape.dims[4].value
 644   h = in_tensor.shape.dims[2].value
 645   w = in_tensor.shape.dims[3].value
 646   return array_ops.reshape(t, [n, c, h, w])
 647 
 648 
 649 def _OihwVectIToHwio(in_tensor):
 650   # [O, I / 4, H, W, 4] => [O, I / 4, 4, H, W] == [O, I, H, W]
 651   t = array_ops.transpose(in_tensor, [2, 3, 1, 4, 0])
 652   o = in_tensor.shape.dims[0].value
 653   i = in_tensor.shape.dims[1].value * in_tensor.shape.dims[4].value
 654   h = in_tensor.shape.dims[2].value
 655   w = in_tensor.shape.dims[3].value
 656   return array_ops.reshape(t, [h, w, i, o])
 657 
 658 
 659 def _NchwToNchwVectC(in_tensor):
 660   n, c, h, w = in_tensor.shape.as_list()
 661   assert c % 4 == 0
 662   t = array_ops.reshape(in_tensor, [n, c // 4, 4, h, w])
 663   return array_ops.transpose(t, [0, 1, 3, 4, 2])
 664 
 665 
 666 def _HwioToOihw(in_tensor):
 667   return array_ops.transpose(in_tensor, [3, 2, 0, 1])

 670 def _SimulateFusedConv2dBiasActivationInt8(conv_input_scale, conv_input, kernel,
 671                                            padding, strides, side_input_scale,
 672                                            side_input, biases, apply_relu):
 673   """Simulates the int8 fused 2-D convolution op using separate float ops.
 674 
 675     The arguments and return values have the same format, meanings and
 676     restrictions as the actual op.
 677   Args:
 678     conv_input_scale: A scalar 'float'.
 679     conv_input: A `Tensor` of type `qint8` in NCHW_VECT_C layout.
 680     kernel: A `Tensor` of type `qint8` in OIHW_VECT_I layout.
 681     padding: A `string` from: `"SAME", "VALID"`.
 682     strides: A list of `ints`.
 683     side_input_scale: A scalar 'float'.
 684     side_input: A `Tensor` of type `qint8` in NCHW_VECT_C layout.
 685     biases: A `Tensor` of type `float32` in NCHW layout.
 686     apply_relu: A boolean to specify whether to apply "Relu" activation function
 687       that clips outputs to the range [0, 127], or "None" activation that clips
 688       to the range [-128, 127].
 689   Returns:
 690     A `Tensor` of type `qint8` in NCHW_VECT_C layout.
 691   """
 692   conv_result = nn_ops.conv2d(
 693       _NchwVectCToNchw(gen_array_ops.dequantize(conv_input, -128, 127)),
 694       _OihwVectIToHwio(gen_array_ops.dequantize(kernel, -128, 127)),
 695       strides=strides,
 696       padding=padding,
 697       data_format="NCHW") * conv_input_scale
 698 
 699   conv_and_side_inputs = conv_result + side_input_scale * _NchwVectCToNchw(
 700       gen_array_ops.dequantize(side_input, -128, 127))
 701 
 702   output = nn_ops.bias_add(conv_and_side_inputs, biases, data_format="NCHW")
 703   if apply_relu:
 704     output = nn_ops.relu(output)
 705 
 706   result, _, _ = gen_array_ops.quantize_v2(
 707       _NchwToNchwVectC(output), -128, 127, dtypes.qint8)
 708   return result
