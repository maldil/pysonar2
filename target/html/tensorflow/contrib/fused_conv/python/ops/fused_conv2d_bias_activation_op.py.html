  17 from __future__ import absolute_import

  19 from __future__ import print_function

  21 from tensorflow.contrib.fused_conv.ops import gen_fused_conv2d_bias_activation_op
  22 from tensorflow.contrib.util import loader
  23 from tensorflow.python.platform import resource_loader
  24 
  25 _fused_conv2d_bias_activation_op_so = loader.load_op_library(
  26     resource_loader.get_path_to_datafile("_fused_conv2d_bias_activation_op.so"))
  27 
  28 
  29 # pylint: disable=redefined-builtin
  30 def fused_conv2d_bias_activation(conv_input,
  31                                  filter,
  32                                  bias,
  33                                  strides=None,
  34                                  padding=None,
  35                                  conv_input_scale=1.0,
  36                                  side_input_scale=0.0,
  37                                  side_input=None,
  38                                  activation_mode="Relu",
  39                                  data_format=None,
  40                                  filter_format=None,
  41                                  name=None):
  42   """Fused 2D conv, bias and activation with optional side input.
  43 
  44   Computes a fused 2-D convolution scaled by conv_input_scale,
  45   adds an optional side input scaled by side_input_scale, adds biases,
  46   and applies ReLU. As an equation:
  47   output = ReLU(conv_input_scale * Conv(conv_input, filter) +
  48                 side_input_scale * side_input + bias)
  49   Note: In int8 mode, The ReLU will clip the output to the range [0..127].
  50 
  51   Args:
  52     conv_input: A `Tensor` of the format specified by `data_format`.
  53     filter: A `Tensor` whose format depends on `data_format`:
  54         if `data_format` is "NCHW_VECT_C", filter should be "OIHW_VECT_I"
  55         otherwise, it should be "HWIO" format.
  56     bias: A 1-D `Tensor` of type `float32`, and dimensions equal to the
  57         number of output channels.
  58     strides: A list of 4 `ints` specifying convolution strides.
  59         if `data_format` is "NCHW" or "NCHW_VECT_C", the order should be NCHW.
  60         if `data_format` is "NHWC", the order should be NHWC.
  61     padding: A `string` from: `"SAME", "VALID"`.
  62     conv_input_scale: A scalar `float32` that will be multiplied by conv_input.
  63         This is optional and defaults to 1. However it should be set to
  64         specify the quantization scale when `data_format` is "NCHW_VECT_C".
  65     side_input_scale: A scalar `float32` that will be multiplied by side_input.
  66         This is optional and defaults to 0.
  67     side_input: A `Tensor` of the format specified by `data_format`.
  68         This is useful for implementing ResNet blocks.
  69     activation_mode: (optional) currently supports the default "Relu", or
  70         "None" activation function.
  71         Note: in qint8 mode, "None" actually clips to the range [-128, 127],
  72         while "Relu" clips to the range [0, 127].
  73     data_format: Specifies the data format.
  74         Possible values are:
  75         "NHWC" float [batch, height, width, channels]
  76         "NCHW" float [batch, channels, height, width]
  77         "NCHW_VECT_C" qint8 [batch, channels / 4, height, width, channels % 4]
  78         Defaults to `"NHWC"`.
  79         Performance is worst for `"NHWC"` and best for `"NCHW_VECT_C"`.
  80     filter_format: Specifies the filter format.
  81         Possible values are:
  82         "HWIO" float [kernel_height, kernel_width, input_channels,
  83                       output_channels ]
  84         "OIHW" float [output_channels, input_channels, kernel_height,
  85                       kernel_width ]
  86         "OIHW_VECT_I" qint8 [ output_channels, input_channels / 4,
  87                               kernel_height, kernel_width, input_channels % 4 ]
  88         Defaults to `"HWIO"`.
  89     name: A name for the operation (optional).
  90 
  91   Returns:
  92     A `Tensor` of the format specified by `data_format`.
  93   """
  94   if strides is None:
  95     strides = [1, 1, 1, 1]
  96   if side_input is None:
  97     side_input = []
  98   return gen_fused_conv2d_bias_activation_op.fused_conv2d_bias_activation(
  99       conv_input,
 100       filter,
 101       bias,
 102       side_input,
 103       conv_input_scale,
 104       side_input_scale,
 105       padding=padding,
 106       strides=strides,
 107       activation_mode=activation_mode,
 108       data_format=data_format,
 109       filter_format=filter_format,
 110       name=name)
