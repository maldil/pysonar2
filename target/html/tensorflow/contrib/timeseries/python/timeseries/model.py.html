  60   def __init__(self,
  61                num_features,
  62                exogenous_feature_columns=None,
  63                dtype=dtypes.float32):
  64     """Constructor for generative models.
  65 
  66     Args:
  67       num_features: Number of features for the time series
  68       exogenous_feature_columns: A list of `tf.feature_column`s (for example
  69            `tf.feature_column.embedding_column`) corresponding to exogenous
  70            features which provide extra information to the model but are not
  71            part of the series to be predicted. Passed to
  72            `tf.feature_column.input_layer`.
  73       dtype: The floating point datatype to use.
  74     """
  75     if exogenous_feature_columns:
  76       self._exogenous_feature_columns = exogenous_feature_columns
  77     else:
  78       self._exogenous_feature_columns = []
  79     self.num_features = num_features
  80     self.dtype = dtype
  81     self._input_statistics = None
  82     self._graph_initialized = False
  83     self._stats_means = None
  84     self._stats_sigmas = None

  87   def exogenous_feature_columns(self):
  88     """`tf.feature_colum`s for features which are not predicted."""
  89     return self._exogenous_feature_columns

  94   def generate(self, number_of_series, series_length,
  95                model_parameters=None, seed=None):
  96     """Sample synthetic data from model parameters, with optional substitutions.
  97 
  98     Returns `number_of_series` possible sequences of future values, sampled from
  99     the generative model with each conditioned on the previous. Samples are
 100     based on trained parameters, except for those parameters explicitly
 101     overridden in `model_parameters`.
 102 
 103     For distributions over future observations, see predict().
 104 
 105     Args:
 106       number_of_series: Number of time series to create.
 107       series_length: Length of each time series.
 108       model_parameters: A dictionary mapping model parameters to values, which
 109           replace trained parameters when generating data.
 110       seed: If specified, return deterministic time series according to this
 111           value.
 112     Returns:
 113       A dictionary with keys TrainEvalFeatures.TIMES (mapping to an array with
 114       shape [number_of_series, series_length]) and TrainEvalFeatures.VALUES
 115       (mapping to an array with shape [number_of_series, series_length,
 116       num_features]).
 117     """
 118     raise NotImplementedError("This model does not support generation.")

 120   def initialize_graph(self, input_statistics=None):
 121     """Define ops for the model, not depending on any previously defined ops.
 122 
 123     Args:
 124       input_statistics: A math_utils.InputStatistics object containing input
 125           statistics. If None, data-independent defaults are used, which may
 126           result in longer or unstable training.
 127     """
 128     self._graph_initialized = True
 129     self._input_statistics = input_statistics
 130     if self._input_statistics:
 131       self._stats_means, variances = (
 132           self._input_statistics.overall_feature_moments)
 133       self._stats_sigmas = math_ops.sqrt(variances)
 134 
 135   def _scale_data(self, data):
 136     """Scale data according to stats (input scale -> model scale)."""
 137     if self._input_statistics is not None:
 138       return (data - self._stats_means) / self._stats_sigmas
 139     else:
 140       return data
 141 
 142   def _scale_variance(self, variance):
 143     """Scale variances according to stats (input scale -> model scale)."""
 144     if self._input_statistics is not None:
 145       return variance / self._input_statistics.overall_feature_moments.variance
 146     else:
 147       return variance
 148 
 149   def _scale_back_data(self, data):
 150     """Scale back data according to stats (model scale -> input scale)."""
 151     if self._input_statistics is not None:
 152       return (data * self._stats_sigmas) + self._stats_means
 153     else:
 154       return data
 155 
 156   def _scale_back_variance(self, variance):
 157     """Scale back variances according to stats (model scale -> input scale)."""
 158     if self._input_statistics is not None:
 159       return variance * self._input_statistics.overall_feature_moments.variance
 160     else:
 161       return variance
 162 
 163   def _check_graph_initialized(self):
 164     if not self._graph_initialized:
 165       raise ValueError(
 166           "TimeSeriesModels require initialize_graph() to be called before "
 167           "use. This defines variables and ops in the default graph, and "
 168           "allows Tensor-valued input statistics to be specified.")

 170   def define_loss(self, features, mode):
 171     """Default loss definition with state replicated across a batch.
 172 
 173     Time series passed to this model have a batch dimension, and each series in
 174     a batch can be operated on in parallel. This loss definition assumes that
 175     each element of the batch represents an independent sample conditioned on
 176     the same initial state (i.e. it is simply replicated across the batch). A
 177     batch size of one provides sequential operations on a single time series.
 178 
 179     More complex processing may operate instead on get_start_state() and
 180     get_batch_loss() directly.
 181 
 182     Args:
 183       features: A dictionary (such as is produced by a chunker) with at minimum
 184         the following key/value pairs (others corresponding to the
 185         `exogenous_feature_columns` argument to `__init__` may be included
 186         representing exogenous regressors):
 187         TrainEvalFeatures.TIMES: A [batch size x window size] integer Tensor
 188             with times for each observation. If there is no artificial chunking,
 189             the window size is simply the length of the time series.
 190         TrainEvalFeatures.VALUES: A [batch size x window size x num features]
 191             Tensor with values for each observation.
 192       mode: The tf.estimator.ModeKeys mode to use (TRAIN, EVAL). For INFER,
 193         see predict().
 194     Returns:
 195       A ModelOutputs object.
 196     """
 197     self._check_graph_initialized()
 198     start_state = math_utils.replicate_state(
 199         start_state=self.get_start_state(),
 200         batch_size=array_ops.shape(features[TrainEvalFeatures.TIMES])[0])
 201     return self.get_batch_loss(features=features, mode=mode, state=start_state)
 202

 231   @abc.abstractmethod
 232   def predict(self, features):
 233     """Returns predictions of future observations given an initial state.
 234 
 235     Computes distributions for future observations. For sampled draws from the
 236     model where each is conditioned on the previous, see generate().
 237 
 238     Args:
 239       features: A dictionary with at minimum the following key/value pairs
 240         (others corresponding to the `exogenous_feature_columns` argument to
 241         `__init__` may be included representing exogenous regressors):
 242         PredictionFeatures.TIMES: A [batch size x window size] Tensor with
 243           times to make predictions for. Times must be increasing within each
 244           part of the batch, and must be greater than the last time `state` was
 245           updated.
 246         PredictionFeatures.STATE_TUPLE: Model-dependent state, each with size
 247           [batch size x ...]. The number and type will typically be fixed by the
 248           model (for example a mean and variance). Typically these will be the
 249           end state returned by get_batch_loss, predicting beyond that data.
 250     Returns:
 251       A dictionary with model-dependent predictions corresponding to the
 252       requested times. Keys indicate the type of prediction, and values have
 253       shape [batch size x window size x ...]. For example state space models
 254       return a "predicted_mean" and "predicted_covariance".
 255     """
 256     pass

 258   def _get_exogenous_embedding_shape(self):
 259     """Computes the shape of the vector returned by _process_exogenous_features.
 260 
 261     Returns:
 262       The shape as a list. Does not include a batch dimension.
 263     """
 264     if not self._exogenous_feature_columns:
 265       return (0,)
 266     with ops.Graph().as_default():
 267       parsed_features = (
 268           feature_column.make_parse_example_spec(
 269               self._exogenous_feature_columns))
 270       placeholder_features = parsing_ops.parse_example(
 271           serialized=array_ops.placeholder(shape=[None], dtype=dtypes.string),
 272           features=parsed_features)
 273       embedded = feature_column.input_layer(
 274           features=placeholder_features,
 275           feature_columns=self._exogenous_feature_columns)
 276       return embedded.get_shape().as_list()[1:]
 277 
 278   def _process_exogenous_features(self, times, features):
 279     """Create a single vector from exogenous features.
 280 
 281     Args:
 282       times: A [batch size, window size] vector of times for this batch,
 283           primarily used to check the shape information of exogenous features.
 284       features: A dictionary of exogenous features corresponding to the columns
 285           in self._exogenous_feature_columns. Each value should have a shape
 286           prefixed by [batch size, window size].
 287     Returns:
 288       A Tensor with shape [batch size, window size, exogenous dimension], where
 289       the size of the exogenous dimension depends on the exogenous feature
 290       columns passed to the model's constructor.
 291     Raises:
 292       ValueError: If an exogenous feature has an unknown rank.
 293     """
 294     if self._exogenous_feature_columns:
 295       exogenous_features_single_batch_dimension = {}
 296       for name, tensor in features.items():
 297         if tensor.get_shape().ndims is None:
 298           # input_from_feature_columns does not support completely unknown
 299           # feature shapes, so we save on a bit of logic and provide a better
 300           # error message by checking that here.
 301           raise ValueError(
 302               ("Features with unknown rank are not supported. Got shape {} for "
 303                "feature {}.").format(tensor.get_shape(), name))
 304         tensor_shape_dynamic = array_ops.shape(tensor)
 305         tensor = array_ops.reshape(
 306             tensor,
 307             array_ops.concat([[tensor_shape_dynamic[0]
 308                                * tensor_shape_dynamic[1]],
 309                               tensor_shape_dynamic[2:]], axis=0))
 310         # Avoid shape warnings when embedding "scalar" exogenous features (those
 311         # with only batch and window dimensions); input_from_feature_columns
 312         # expects input ranks to match the embedded rank.
 313         if tensor.get_shape().ndims == 1 and tensor.dtype != dtypes.string:
 314           exogenous_features_single_batch_dimension[name] = tensor[:, None]
 315         else:
 316           exogenous_features_single_batch_dimension[name] = tensor
 317       embedded_exogenous_features_single_batch_dimension = (
 318           feature_column.input_layer(
 319               features=exogenous_features_single_batch_dimension,
 320               feature_columns=self._exogenous_feature_columns,
 321               trainable=True))
 322       exogenous_regressors = array_ops.reshape(
 323           embedded_exogenous_features_single_batch_dimension,
 324           array_ops.concat(
 325               [
 326                   array_ops.shape(times), array_ops.shape(
 327                       embedded_exogenous_features_single_batch_dimension)[1:]
 328               ],
 329               axis=0))
 330       exogenous_regressors.set_shape(times.get_shape().concatenate(
 331           embedded_exogenous_features_single_batch_dimension.get_shape()[1:]))
 332       exogenous_regressors = math_ops.cast(
 333           exogenous_regressors, dtype=self.dtype)
 334     else:
 335       # Not having any exogenous features is a special case so that models can
 336       # avoid superfluous updates, which may not be free of side effects due to
 337       # bias terms in transformations.
 338       exogenous_regressors = None
 339     return exogenous_regressors

 342 # TODO(allenl): Add a superclass of SequentialTimeSeriesModel which fuses
 343 # filtering/prediction/exogenous into one step, and move looping constructs to
 344 # that class.
 345 class SequentialTimeSeriesModel(TimeSeriesModel):
 346   """Base class for recurrent generative models.
 347 
 348   Models implementing this interface have three main functions, corresponding to
 349   abstract methods:
 350     _filtering_step: Updates state based on observations and computes a loss.
 351     _prediction_step: Predicts a batch of observations and new model state.
 352     _imputation_step: Updates model state across a gap.
 353     _exogenous_input_step: Updates state to account for exogenous regressors.
 354 
 355   Models may also specify a _window_initializer to prepare for a window of data.
 356 
 357   See StateSpaceModel for a concrete example of a model implementing this
 358   interface.
 359 
 360   """
 361 
 362   def __init__(self,
 363                train_output_names,
 364                predict_output_names,
 365                num_features,
 366                normalize_features=False,
 367                dtype=dtypes.float32,
 368                exogenous_feature_columns=None,
 369                exogenous_update_condition=None,
 370                static_unrolling_window_size_threshold=None):
 371     """Initialize a SequentialTimeSeriesModel.
 372 
 373     Args:
 374       train_output_names: A list of products/predictions returned from
 375           _filtering_step.
 376       predict_output_names: A list of products/predictions returned from
 377           _prediction_step.
 378       num_features: Number of features for the time series
 379       normalize_features: Boolean. If True, `values` are passed normalized to
 380           the model (via self._scale_data). Scaling is done for the whole window
 381           as a batch, which is slightly more efficient than scaling inside the
 382           window loop. The model must then define _scale_back_predictions, which
 383           may use _scale_back_data or _scale_back_variance to return predictions
 384           to the input scale.
 385       dtype: The floating point datatype to use.
 386       exogenous_feature_columns: A list of `tf.feature_column`s objects. See
 387           `TimeSeriesModel`.
 388       exogenous_update_condition: A function taking two Tensor arguments `times`
 389           (shape [batch size]) and `features` (a dictionary mapping exogenous
 390           feature keys to Tensors with shapes [batch size, ...]) and returning a
 391           boolean Tensor with shape [batch size] indicating whether state should
 392           be updated using exogenous features for each part of the batch. Where
 393           it is False, no exogenous update is performed. If None (default),
 394           exogenous updates are always performed. Useful for avoiding "leaky"
 395           frequent exogenous updates when sparse updates are desired. Called
 396           only during graph construction.
 397       static_unrolling_window_size_threshold: Controls whether a `tf.while_loop`
 398           is used when looping over a window of data. If
 399           `static_unrolling_window_size_threshold` is None, a `tf.while_loop` is
 400           always used. Otherwise it must be an integer, and the graph is
 401           replicated for each step taken whenever the window size is less than
 402           or equal to this value (if the window size is available in the static
 403           shape information of the TrainEvalFeatures.TIMES feature). Static
 404           unrolling generally decreases the per-step time for small window/batch
 405           sizes, but increases graph construction time.
 406     """
 407     super(SequentialTimeSeriesModel, self).__init__(
 408         num_features=num_features, dtype=dtype,
 409         exogenous_feature_columns=exogenous_feature_columns)
 410     self._exogenous_update_condition = exogenous_update_condition
 411     self._train_output_names = train_output_names
 412     self._predict_output_names = predict_output_names
 413     self._normalize_features = normalize_features
 414     self._static_unrolling_window_size_threshold = (
 415         static_unrolling_window_size_threshold)
 416 
 417   def _scale_back_predictions(self, predictions):
 418     """Return a window of predictions to input scale.
 419 
 420     Args:
 421       predictions: A dictionary mapping from prediction names to Tensors.
 422     Returns:
 423       A dictionary with values corrected for input normalization (e.g. with
 424       self._scale_back_mean and possibly self._scale_back_variance). May be a

 428         "SequentialTimeSeriesModel normalized input data"

 430         "the predictions back to the input scale.")

 432   @abc.abstractmethod
 433   def _filtering_step(self, current_times, current_values, state, predictions):
 434     """Compute a single-step loss for a batch of data.
 435 
 436     Args:
 437       current_times: A [batch size] Tensor of times for each observation.
 438       current_values: A [batch size] Tensor of values for each observation.
 439       state: Model state, updated to current_times.
 440       predictions: The outputs of _prediction_step

 442       A tuple of (updated state, outputs):

 444         outputs: A dictionary of Tensors with keys corresponding to

 517   def _window_initializer(self, times, state):
 518     """Prepare for training or prediction on a window of data.
 519 
 520     Args:
 521       times: A [batch size x window size] Tensor with times for each
 522           observation.
 523       state: Model-dependent state, each with size [batch size x ...]. The

 531   def get_batch_loss(self, features, mode, state):
 532     """Calls self._filtering_step. See TimeSeriesModel.get_batch_loss."""
 533     per_observation_loss, state, outputs = self.per_step_batch_loss(
 534         features, mode, state)
 535     # per_step_batch_loss returns [batch size, window size, ...] state, whereas
 536     # get_batch_loss is expected to return [batch size, ...] state for the last
 537     # element of a window
 538     state = nest.pack_sequence_as(
 539         state,
 540         [state_element[:, -1] for state_element in nest.flatten(state)])
 541     outputs["observed"] = features[TrainEvalFeatures.VALUES]
 542     return ModelOutputs(
 543         loss=per_observation_loss,
 544         end_state=state,
 545         predictions=outputs,
 546         prediction_times=features[TrainEvalFeatures.TIMES])
 547 
 548   def _apply_exogenous_update(
 549       self, current_times, step_number, state, raw_features,
 550       embedded_exogenous_regressors):
 551     """Performs a conditional state update based on exogenous features."""
 552     if embedded_exogenous_regressors is None:
 553       return state
 554     else:
 555       current_exogenous_regressors = embedded_exogenous_regressors[
 556           :, step_number, :]
 557       exogenous_updated_state = self._exogenous_input_step(
 558           current_times=current_times,
 559           current_exogenous_regressors=current_exogenous_regressors,
 560           state=state)
 561       if self._exogenous_update_condition is not None:
 562         current_raw_exogenous_features = {
 563             key: value[:, step_number] for key, value in raw_features.items()
 564             if key not in [PredictionFeatures.STATE_TUPLE,
 565                            TrainEvalFeatures.TIMES,
 566                            TrainEvalFeatures.VALUES]}
 567         conditionally_updated_state_flat = []
 568         for updated_state_element, original_state_element in zip(
 569             nest.flatten(exogenous_updated_state),
 570             nest.flatten(state)):
 571           conditionally_updated_state_flat.append(
 572               array_ops.where(
 573                   self._exogenous_update_condition(
 574                       times=current_times,
 575                       features=current_raw_exogenous_features),
 576                   updated_state_element,
 577                   original_state_element))
 578         return nest.pack_sequence_as(state, conditionally_updated_state_flat)
 579       else:
 580         return exogenous_updated_state

 582   def per_step_batch_loss(self, features, mode, state):
 583     """Computes predictions, losses, and intermediate model states.
 584 
 585     Args:
 586       features: A dictionary with times, values, and (optionally) exogenous
 587           regressors. See `define_loss`.
 588       mode: The tf.estimator.ModeKeys mode to use (TRAIN, EVAL, INFER).
 589       state: Model-dependent state, each with size [batch size x ...]. The
 590           number and type will typically be fixed by the model (for example a
 591           mean and variance).
 592     Returns:
 593       A tuple of (loss, filtered_states, predictions)
 594         loss: Average loss values across the batch.
 595         filtered_states: For each Tensor in `state` with shape [batch size x
 596             ...], `filtered_states` has a Tensor with shape [batch size x window
 597             size x ...] with filtered state for each part of the batch and
 598             window.
 599         predictions: A dictionary with model-dependent one-step-ahead (or
 600             at-least-one-step-ahead with missing values) predictions, with keys
 601             indicating the type of prediction and values having shape [batch
 602             size x window size x ...]. For example state space models provide
 603             "mean", "covariance", and "log_likelihood".
 604 
 605     """
 606     self._check_graph_initialized()
 607     times = math_ops.cast(features[TrainEvalFeatures.TIMES], dtype=dtypes.int64)
 608     values = math_ops.cast(features[TrainEvalFeatures.VALUES], dtype=self.dtype)
 609     if self._normalize_features:
 610       values = self._scale_data(values)
 611     exogenous_regressors = self._process_exogenous_features(
 612         times=times,
 613         features={key: value for key, value in features.items()
 614                   if key not in [TrainEvalFeatures.TIMES,
 615                                  TrainEvalFeatures.VALUES]})
 616     def _batch_loss_filtering_step(step_number, current_times, state):
 617       """Make a prediction and update it based on data."""
 618       current_values = values[:, step_number, :]
 619       state = self._apply_exogenous_update(
 620           step_number=step_number, current_times=current_times, state=state,
 621           raw_features=features,
 622           embedded_exogenous_regressors=exogenous_regressors)
 623       predicted_state, predictions = self._prediction_step(
 624           current_times=current_times,
 625           state=state)
 626       filtered_state, outputs = self._filtering_step(
 627           current_times=current_times,
 628           current_values=current_values,
 629           state=predicted_state,
 630           predictions=predictions)
 631       return filtered_state, outputs
 632     state, outputs = self._state_update_loop(
 633         times=times, state=state, state_update_fn=_batch_loss_filtering_step,
 634         outputs=["loss"] + self._train_output_names)
 635     outputs["loss"].set_shape(times.get_shape())
 636     loss_sum = math_ops.reduce_sum(outputs["loss"])
 637     per_observation_loss = (loss_sum / math_ops.cast(
 638         math_ops.reduce_prod(array_ops.shape(times)), dtype=self.dtype))
 639     per_observation_loss += self._loss_additions(times, values, mode)
 640     # Since we have window-level additions to the loss, its per-step value is
 641     # misleading, so we avoid returning it.
 642     del outputs["loss"]
 643     if self._normalize_features:
 644       outputs = self._scale_back_predictions(outputs)
 645     return per_observation_loss, state, outputs
 646 
 647   def predict(self, features):
 648     """Calls self._prediction_step in a loop. See TimeSeriesModel.predict."""
 649     predict_times = ops.convert_to_tensor(features[PredictionFeatures.TIMES],
 650                                           dtypes.int64)
 651     start_state = features[PredictionFeatures.STATE_TUPLE]
 652     exogenous_regressors = self._process_exogenous_features(
 653         times=predict_times,
 654         features={
 655             key: value
 656             for key, value in features.items()
 657             if key not in
 658             [PredictionFeatures.TIMES, PredictionFeatures.STATE_TUPLE]
 659         })
 660     def _call_prediction_step(step_number, current_times, state):
 661       state = self._apply_exogenous_update(
 662           step_number=step_number, current_times=current_times, state=state,
 663           raw_features=features,
 664           embedded_exogenous_regressors=exogenous_regressors)
 665       state, outputs = self._prediction_step(
 666           current_times=current_times, state=state)
 667       return state, outputs
 668     _, predictions = self._state_update_loop(
 669         times=predict_times, state=start_state,
 670         state_update_fn=_call_prediction_step,
 671         outputs=self._predict_output_names)
 672     if self._normalize_features:
 673       predictions = self._scale_back_predictions(predictions)
 674     return predictions

 676   class _FakeTensorArray(object):
 677     """An interface for Python lists that is similar to TensorArray.
 678 
 679     Used for easy switching between static and dynamic looping.
 680     """
 681 
 682     def __init__(self):
 683       self.values = []
 684 
 685     def write(self, unused_position, value):
 686       del unused_position
 687       self.values.append(value)
 688       return self
 689 
 690   def _state_update_loop(self, times, state, state_update_fn, outputs):
 691     """Iterates over `times`, calling `state_update_fn` to collect outputs.
 692 
 693     Args:
 694       times: A [batch size x window size] Tensor of integers to iterate over.
 695       state: A list of model-specific state Tensors, each with shape [batch size
 696           x ...].
 697       state_update_fn: A callback taking the following arguments
 698             step_number; A scalar integer Tensor indicating the current position
 699               in the window.
 700             current_times; A [batch size] vector of Integers indicating times
 701               for each part of the batch.
 702             state; Current model state.
 703           It returns a tuple of (updated state, output_values), output_values
 704           being a dictionary of Tensors with keys corresponding to `outputs`.
 705       outputs: A list of strings indicating values which will be saved while
 706           iterating. Must match the keys of the dictionary returned by
 707           state_update_fn.
 708     Returns:
 709       A tuple of (state, output_dict)
 710       state: The final model state.
 711       output_dict: A dictionary of outputs corresponding to those specified in
 712         `outputs` and computed in state_update_fn.
 713     """
 714     times = ops.convert_to_tensor(times, dtype=dtypes.int64)
 715     window_static_shape = times.get_shape()[1].value
 716     if self._static_unrolling_window_size_threshold is None:
 717       static_unroll = False
 718     else:
 719       # The user has specified a threshold for static loop unrolling.
 720       if window_static_shape is None:
 721         # We don't have static shape information for the window size, so dynamic
 722         # looping is our only option.
 723         static_unroll = False
 724       elif window_static_shape <= self._static_unrolling_window_size_threshold:
 725         # The threshold is satisfied; unroll statically
 726         static_unroll = True
 727       else:
 728         # A threshold was set but not satisfied
 729         static_unroll = False
 730 
 731     self._window_initializer(times, state)
 732 
 733     def _run_condition(step_number, *unused):
 734       del unused  # not part of while loop run condition
 735       return math_ops.less(step_number, window_size)
 736 
 737     def _state_update_step(
 738         step_number, state, state_accumulators, output_accumulators,
 739         reuse=False):
 740       """Impute, then take one state_update_fn step, accumulating outputs."""
 741       with variable_scope.variable_scope("state_update_step", reuse=reuse):
 742         current_times = times[:, step_number]
 743         state = self._imputation_step(current_times=current_times, state=state)
 744         output_accumulators_dict = {
 745             accumulator_key: accumulator
 746             for accumulator_key, accumulator
 747             in zip(outputs, output_accumulators)}
 748         step_state, output_values = state_update_fn(
 749             step_number=step_number,
 750             current_times=current_times,
 751             state=state)
 752         assert set(output_values.keys()) == set(outputs)
 753         new_output_accumulators = []
 754         for output_key in outputs:
 755           accumulator = output_accumulators_dict[output_key]
 756           output_value = output_values[output_key]
 757           new_output_accumulators.append(
 758               accumulator.write(step_number, output_value))
 759         flat_step_state = nest.flatten(step_state)
 760         assert len(state_accumulators) == len(flat_step_state)
 761         new_state_accumulators = []
 762         new_state_flat = []
 763         for step_state_value, state_accumulator, original_state in zip(
 764             flat_step_state, state_accumulators, nest.flatten(state)):
 765           # Make sure the static shape information is complete so while_loop
 766           # does not complain about shape information changing.
 767           step_state_value.set_shape(original_state.get_shape())
 768           new_state_flat.append(step_state_value)
 769           new_state_accumulators.append(state_accumulator.write(
 770               step_number, step_state_value))
 771         step_state = nest.pack_sequence_as(state, new_state_flat)
 772         return (step_number + 1, step_state,
 773                 new_state_accumulators, new_output_accumulators)
 774 
 775     window_size = array_ops.shape(times)[1]
 776 
 777     def _window_size_tensor_array(dtype):
 778       if static_unroll:
 779         return self._FakeTensorArray()
 780       else:
 781         return tensor_array_ops.TensorArray(
 782             dtype=dtype, size=window_size, dynamic_size=False)
 783 
 784     initial_loop_arguments = [
 785         array_ops.zeros([], dtypes.int32),
 786         state,
 787         [_window_size_tensor_array(element.dtype)
 788          for element in nest.flatten(state)],
 789         [_window_size_tensor_array(self.dtype) for _ in outputs]]
 790     if static_unroll:
 791       arguments = initial_loop_arguments
 792       for step_number in range(times.get_shape()[1].value):
 793         arguments = _state_update_step(
 794             array_ops.constant(step_number, dtypes.int32), *arguments[1:],
 795             reuse=(step_number > 0))  # Variable sharing between steps
 796     else:
 797       arguments = control_flow_ops.while_loop(
 798           cond=_run_condition,
 799           body=_state_update_step,
 800           loop_vars=initial_loop_arguments)
 801     (_, _, state_loop_result, outputs_loop_result) = arguments
 802 
 803     def _stack_and_transpose(tensor_array):
 804       """Stack and re-order the dimensions of a TensorArray."""
 805       if static_unroll:
 806         return array_ops.stack(tensor_array.values, axis=1)
 807       else:
 808         # TensorArrays from while_loop stack with window size as the first
 809         # dimension, so this function swaps it and the batch dimension to
 810         # maintain the [batch x window size x ...] convention used elsewhere.
 811         stacked = tensor_array.stack()
 812         return array_ops.transpose(
 813             stacked,
 814             perm=array_ops.concat([[1, 0], math_ops.range(
 815                 2, array_ops.rank(stacked))], 0))
 816 
 817     outputs_dict = {output_key: _stack_and_transpose(output)
 818                     for output_key, output
 819                     in zip(outputs, outputs_loop_result)}
 820     full_state = nest.pack_sequence_as(
 821         state,
 822         [_stack_and_transpose(state_element)
 823          for state_element in state_loop_result])
 824     return full_state, outputs_dict
