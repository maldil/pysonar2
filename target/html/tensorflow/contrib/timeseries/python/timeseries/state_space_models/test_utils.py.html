  33 def transition_power_test_template(test_case, model, num_steps):
  34   """Tests the transition_to_powers function of a state space model."""
  35   transition_matrix = ops.convert_to_tensor(
  36       model.get_state_transition(), dtype=model.dtype)
  37   step_number = array_ops.placeholder(shape=[], dtype=dtypes.int64)
  38   state_dimension = transition_matrix.get_shape()[0].value
  39   previous_matrix = array_ops.placeholder(
  40       shape=[state_dimension, state_dimension], dtype=transition_matrix.dtype)
  41   true_single_step_update = math_ops.matmul(previous_matrix,
  42                                             transition_matrix)
  43   model_output_tensor = model.transition_to_powers(powers=array_ops.stack(
  44       [step_number, step_number]))
  45   with test_case.test_session():
  46     starting_matrix = linalg_ops.eye(
  47         state_dimension, batch_shape=array_ops.shape(num_steps)).eval()
  48     evaled_current_matrix = starting_matrix
  49     for iteration_number in range(num_steps):
  50       model_output = model_output_tensor.eval(
  51           feed_dict={step_number: iteration_number})
  52       test_case.assertAllClose(
  53           evaled_current_matrix,
  54           model_output[0],
  55           rtol=1e-8 if evaled_current_matrix.dtype == numpy.float64 else 1e-4)
  56       evaled_current_matrix = true_single_step_update.eval(
  57           feed_dict={previous_matrix: evaled_current_matrix})

  60 def noise_accumulator_test_template(test_case, model, num_steps):
  61   """Tests `model`'s transition_power_noise_accumulator."""
  62   transition_matrix = ops.convert_to_tensor(
  63       model.get_state_transition(), dtype=model.dtype)
  64   noise_transform = ops.convert_to_tensor(
  65       model.get_noise_transform(), dtype=model.dtype)
  66   state_dimension = transition_matrix.get_shape()[0].value
  67   state_noise_dimension = noise_transform.get_shape()[1].value
  68   gen_noise_addition = math_utils.sign_magnitude_positive_definite(
  69       raw=random_ops.random_normal(
  70           shape=[state_noise_dimension, state_noise_dimension],
  71           dtype=model.dtype))
  72   gen_starting_noise = math_utils.sign_magnitude_positive_definite(
  73       random_ops.random_normal(
  74           shape=[state_dimension, state_dimension], dtype=model.dtype))
  75   starting_noise = array_ops.placeholder(
  76       shape=[state_dimension, state_dimension], dtype=model.dtype)
  77   step_number = array_ops.placeholder(shape=[], dtype=dtypes.int64)
  78   starting_transitioned = math_ops.matmul(
  79       math_ops.matmul(transition_matrix, starting_noise),
  80       transition_matrix,
  81       adjoint_b=True)
  82   with test_case.test_session():
  83     evaled_starting_noise = gen_starting_noise.eval()
  84     current_starting_noise_transitioned = evaled_starting_noise
  85     current_noise = evaled_starting_noise
  86     evaled_noise_addition = gen_noise_addition.eval()
  87     evaled_noise_addition_transformed = math_ops.matmul(
  88         math_ops.matmul(noise_transform, evaled_noise_addition),
  89         noise_transform,
  90         adjoint_b=True).eval()
  91     model.state_transition_noise_covariance = evaled_noise_addition
  92     model._window_initializer(  # pylint: disable=protected-access
  93         times=math_ops.range(num_steps + 1)[..., None], state=(None, None, 0))
  94     model_update = model.transition_power_noise_accumulator(
  95         num_steps=step_number)
  96     for iteration_number in range(num_steps):
  97       model_new_noise = model_update.eval(
  98           feed_dict={step_number: iteration_number})
  99       test_case.assertAllClose(
 100           current_noise,
 101           model_new_noise + current_starting_noise_transitioned,
 102           rtol=1e-8 if current_noise.dtype == numpy.float64 else 1e-3)
 103       current_starting_noise_transitioned = starting_transitioned.eval(
 104           feed_dict={starting_noise: current_starting_noise_transitioned})
 105       current_noise = (
 106           starting_transitioned.eval(
 107               feed_dict={starting_noise: current_noise})
 108           + evaled_noise_addition_transformed)
