  35 class FilteringStepPostprocessor(object):
  36   """Base class for processors that are applied after each filter step."""
  37 
  38   __metaclass__ = abc.ABCMeta
  39 
  40   @abc.abstractmethod
  41   def process_filtering_step(self, current_times, current_values,
  42                              predicted_state, filtered_state, outputs):
  43     """Extends/modifies a filtering step, altering state and loss.
  44 
  45     Args:
  46       current_times: A [batch size] integer Tensor of times.
  47       current_values: A [batch size x num features] Tensor of values filtering
  48           is being performed on.
  49       predicted_state: A (possibly nested) list of Tensors indicating model
  50           state which does not take `current_times` and `current_values` into
  51           account.
  52       filtered_state: Same structure as predicted_state, but updated to take
  53           `current_times` and `current_values` into account.
  54       outputs: A dictionary of outputs produced by model filtering
  55           (SequentialTimeSeriesModel._process_filtering_step).
  56     Returns: A tuple of (new_state, updated_outputs);
  57       new_state: Updated state with the same structure as `filtered_state` and
  58           `predicted_state`.
  59       updated_outputs: The `outputs` dictionary, updated with any new outputs
  60           from this filtering postprocessor.
  61     """
  62     pass
  63 
  64   @abc.abstractproperty
  65   def output_names(self):
  66     return []

  69 def cauchy_alternative_to_gaussian(current_times, current_values, outputs):
  70   """A Cauchy anomaly distribution, centered at a Gaussian prediction.
  71 
  72   Performs an entropy-matching approximation of the scale parameters of
  73   independent Cauchy distributions given the covariance matrix of a multivariate
  74   Gaussian in outputs["covariance"], and centers the Cauchy distributions at
  75   outputs["mean"]. This requires that the model that we are creating an
  76   alternative/anomaly distribution for produces a mean and covariance.
  77 
  78   Args:
  79     current_times: A [batch size] Tensor of times, unused.
  80     current_values: A [batch size x num features] Tensor of values to evaluate
  81         the anomaly distribution at.
  82     outputs: A dictionary of Tensors with keys "mean" and "covariance"
  83         describing the Gaussian to construct an anomaly distribution from. The
  84         value corresponding to "mean" has shape [batch size x num features], and
  85         the value corresponding to "covariance" has shape [batch size x num
  86         features x num features].
  87   Returns:
  88     A [batch size] Tensor of log likelihoods; the anomaly log PDF evaluated at
  89     `current_values`.
  90   """
  91   del current_times  # unused
  92   cauchy_scale = math_utils.entropy_matched_cauchy_scale(outputs["covariance"])
  93   individual_log_pdfs = distributions.StudentT(
  94       df=array_ops.ones([], dtype=current_values.dtype),
  95       loc=outputs["mean"],
  96       scale=cauchy_scale).log_prob(current_values)
  97   return math_ops.reduce_sum(individual_log_pdfs, axis=1)

 100 def _interpolate_state_linear(first_state, second_state, first_responsibility):
 101   """Interpolate between two model states linearly."""
 102   interpolated_state_flat = []
 103   for first_state_tensor, second_state_tensor in zip(
 104       nest.flatten(first_state), nest.flatten(second_state)):
 105     assert first_state_tensor.dtype == second_state_tensor.dtype
 106     if first_state_tensor.dtype.is_floating:
 107       # Pad the responsibility shape with ones up to the state's rank so that it
 108       # broadcasts
 109       first_responsibility_padded = array_ops.reshape(
 110           tensor=first_responsibility,
 111           shape=array_ops.concat([
 112               array_ops.shape(first_responsibility), array_ops.ones(
 113                   [array_ops.rank(first_state_tensor) - 1], dtype=dtypes.int32)
 114           ], 0))
 115       interpolated_state = (
 116           first_responsibility_padded * first_state_tensor
 117           + (1. - first_responsibility_padded) * second_state_tensor)
 118       interpolated_state.set_shape(first_state_tensor.get_shape())
 119       interpolated_state_flat.append(interpolated_state)
 120     else:
 121       # Integer dtypes are probably representing times, and don't need
 122       # interpolation. Make sure they're identical to be sure.
 123       with ops.control_dependencies(
 124           [check_ops.assert_equal(first_state_tensor, second_state_tensor)]):
 125         interpolated_state_flat.append(array_ops.identity(first_state_tensor))
 126   return nest.pack_sequence_as(first_state, interpolated_state_flat)

 129 class StateInterpolatingAnomalyDetector(FilteringStepPostprocessor):
 130   """An anomaly detector which guards model state against outliers.
 131 
 132   Smoothly interpolates between a model's predicted and inferred states, based
 133   on the posterior probability of an anomaly, p(anomaly | data). This is useful
 134   if anomalies would otherwise lead to model state which is hard to recover
 135   from (Gaussian state space models suffer from this, for example).
 136 
 137   Relies on (1) an alternative distribution, typically with heavier tails than
 138   the model's normal predictions, and (2) a prior probability of an anomaly. The
 139   prior probability acts as a penalty, discouraging the system from marking too
 140   many points as anomalies. The alternative distribution indicates the
 141   probability of a datapoint given that it is an anomaly, and is a heavy-tailed
 142   distribution (Cauchy) centered around the model's predictions by default.
 143 
 144   Specifically, we have:
 145 
 146     p(anomaly | data) = p(data | anomaly) * anomaly_prior_probability
 147         / (p(data | not anomaly) * (1 - anomaly_prior_probability)
 148            + p(data | anomaly) * anomaly_prior_probability)
 149 
 150   This is simply Bayes' theorem, where p(data | anomaly) is the
 151   alternative/anomaly distribution, p(data | not anomaly) is the model's
 152   predicted distribution, and anomaly_prior_probability is the prior probability
 153   of an anomaly occurring (user-specified, defaulting to 1%).
 154 
 155   Rather than computing p(anomaly | data) directly, we use the odds ratio:
 156 
 157     odds_ratio = p(data | anomaly) * anomaly_prior_probability
 158         / (p(data | not anomaly) * (1 - anomaly_prior_probability))
 159 
 160   This has the same information as p(anomaly | data):
 161 
 162     odds_ratio = p(anomaly | data) / p(not anomaly | data)
 163 
 164   A "responsibility" score is computed for the model based on the log odds
 165   ratio, and state interpolated based on this responsibility:
 166 
 167     model_responsibility = 1 / (1 + exp(-responsibility_scaling
 168                                         * ln(odds_ratio)))
 169     model_state = filtered_model_state * model_responsibility
 170                   + predicted_model_state * (1 - model_responsibility)
 171     loss = model_responsibility
 172              * ln(p(data | not anomaly) * (1 - anomaly_prior_probability))
 173            + (1 - model_responsibility)
 174              * ln(p(data | anomaly) * anomaly_prior_probability)
 175 
 176   """
 177 
 178   output_names = ["anomaly_score"]
 179 
 180   def __init__(self,
 181                anomaly_log_likelihood=cauchy_alternative_to_gaussian,
 182                anomaly_prior_probability=0.01,
 183                responsibility_scaling=1.0):
 184     """Configure the anomaly detector.
 185 
 186     Args:
 187       anomaly_log_likelihood: A function taking `current_times`,
 188           `current_values`, and `outputs` (same as the corresponding arguments
 189           to process_filtering_step) and returning a [batch size] Tensor of log
 190           likelihoods under an anomaly distribution.
 191       anomaly_prior_probability: A scalar value, between 0 and 1, indicating the
 192           prior probability of a particular example being an anomaly.
 193       responsibility_scaling: A positive scalar controlling how fast
 194           interpolation transitions between not-anomaly and anomaly; lower
 195           values (closer to 0) create a smoother/slower transition.
 196     """
 197     self._anomaly_log_likelihood = anomaly_log_likelihood
 198     self._responsibility_scaling = responsibility_scaling
 199     self._anomaly_prior_probability = anomaly_prior_probability
 200 
 201   def process_filtering_step(self, current_times, current_values,
 202                              predicted_state, filtered_state, outputs):
 203     """Fall back on `predicted_state` for anomalies.
 204 
 205     Args:
 206       current_times: A [batch size] integer Tensor of times.
 207       current_values: A [batch size x num features] Tensor of values filtering
 208           is being performed on.
 209       predicted_state: A (possibly nested) list of Tensors indicating model
 210           state which does not take `current_times` and `current_values` into
 211           account.
 212       filtered_state: Same structure as predicted_state, but updated to take
 213           `current_times` and `current_values` into account.
 214       outputs: A dictionary of outputs produced by model filtering. Must
 215           include `log_likelihood`, a [batch size] Tensor indicating the log
 216           likelihood of the observations under the model's predictions.
 217     Returns:
 218       A tuple of (new_state, updated_outputs);
 219         new_state: Updated state with the same structure as `filtered_state` and
 220             `predicted_state`; predicted_state for anomalies and filtered_state
 221             otherwise (per batch element).
 222         updated_outputs: The `outputs` dictionary, updated with a new "loss"
 223             (the interpolated negative log likelihoods under the model and
 224             anomaly distributions) and "anomaly_score" (the log odds ratio of
 225             each part of the batch being an anomaly).
 226     """
 227     anomaly_log_likelihood = self._anomaly_log_likelihood(
 228         current_times=current_times,
 229         current_values=current_values,
 230         outputs=outputs)
 231     anomaly_prior_probability = ops.convert_to_tensor(
 232         self._anomaly_prior_probability, dtype=current_values.dtype)
 233     # p(data | anomaly) * p(anomaly)
 234     data_and_anomaly_log_probability = (
 235         anomaly_log_likelihood + math_ops.log(anomaly_prior_probability))
 236     # p(data | no anomaly) * p(no anomaly)
 237     data_and_no_anomaly_log_probability = (
 238         outputs["log_likelihood"] + math_ops.log(1. - anomaly_prior_probability)
 239     )
 240     # A log odds ratio is slightly nicer here than computing p(anomaly | data),
 241     # since it is centered around zero
 242     anomaly_log_odds_ratio = (
 243         data_and_anomaly_log_probability
 244         - data_and_no_anomaly_log_probability)
 245     model_responsibility = math_ops.sigmoid(-self._responsibility_scaling *
 246                                             anomaly_log_odds_ratio)
 247     # Do a linear interpolation between predicted and inferred model state
 248     # based on the model's "responsibility". If we knew for sure whether
 249     # this was an anomaly or not (binary responsibility), this would be the
 250     # correct thing to do, but given that we don't it's just a
 251     # (differentiable) heuristic.
 252     interpolated_state = _interpolate_state_linear(
 253         first_state=filtered_state,
 254         second_state=predicted_state,
 255         first_responsibility=model_responsibility)
 256     # TODO(allenl): Try different responsibility scalings and interpolation
 257     # methods (e.g. average in probability space rather than log space).
 258     interpolated_log_likelihood = (
 259         model_responsibility * data_and_no_anomaly_log_probability
 260         + (1. - model_responsibility) * data_and_anomaly_log_probability)
 261     outputs["loss"] = -interpolated_log_likelihood
 262     outputs["anomaly_score"] = anomaly_log_odds_ratio
 263     return (interpolated_state, outputs)
