  47     collections.namedtuple(
  48         typename="StateSpaceModelConfiguration",
  49         field_names=[
  50             "num_features", "use_observation_noise", "dtype",
  51             "covariance_prior_fn", "bayesian_prior_weighting",
  52             "filtering_postprocessor", "trainable_start_state",
  53             "exogenous_noise_increases", "exogenous_noise_decreases",
  54             "exogenous_feature_columns", "exogenous_update_condition",
  55             "filtering_maximum_posterior_variance_ratio",
  56             "filtering_minimum_posterior_variance",
  57             "transition_covariance_initial_log_scale_bias",
  58             "static_unrolling_window_size_threshold"])):
  59   """Configuration options for StateSpaceModels."""
  60 
  61   def __new__(
  62       cls,
  63       num_features=1,
  64       use_observation_noise=True,
  65       dtype=dtypes.float32,
  66       covariance_prior_fn=math_utils.log_noninformative_covariance_prior,
  67       bayesian_prior_weighting=True,
  68       filtering_postprocessor=None,
  69       trainable_start_state=False,
  70       exogenous_noise_increases=True,
  71       exogenous_noise_decreases=False,
  72       exogenous_feature_columns=None,
  73       exogenous_update_condition=None,
  74       filtering_maximum_posterior_variance_ratio=1e6,
  75       filtering_minimum_posterior_variance=0.,
  76       transition_covariance_initial_log_scale_bias=-5.,
  77       static_unrolling_window_size_threshold=None):
  78     """Configuration options for StateSpaceModels.
  79 
  80     Args:
  81       num_features: Output dimension for model
  82       use_observation_noise: If true, observations are modeled as noisy
  83         functions of the current state. If false, observations are a
  84         deterministic function of the current state. Only applicable to the
  85         top-level model in an ensemble. Consider also changing the
  86         transition_covariance_initial_log_scale_bias when disabling observation
  87         noise, as its default setting assumes that observation noise is part of
  88         the model.
  89       dtype: The float dtype to use when defining the model.
  90       covariance_prior_fn: A function mapping from a covariance matrix to a
  91           scalar value (e.g. log likelihood) which can be summed across
  92           matrices. Defaults to an independent Jeffreys prior on the diagonal
  93           elements (regularizing as log(1. / variance)). To use a flat prior
  94           (i.e. no regularization), set to `lambda _: 0.`.  Defaults to
  95           relatively uninformative priors on state transition and observation
  96           noise, which have the effect of encouraging low-noise solutions which
  97           provide confident predictions when possible. Without regularization,
  98           transition noise tends to remain high, and multi-step predictions are
  99           under-confident.
 100       bayesian_prior_weighting: If True, weights the prior (covariance_prior_fn)
 101           based on an estimate of the full dataset size. If False, weights it
 102           based on the mini-batch window size, which (while statistically
 103           improper) can lead to more desirable low-noise solutions in cases
 104           where the full dataset is large enough to overwhelm the prior.
 105       filtering_postprocessor: A FilteringStepPostprocessor object to use,
 106           useful for ignoring anomalies in training data.
 107       trainable_start_state: If True, start state may depend on trainable
 108           Variables. If False, it will not.
 109       exogenous_noise_increases: If True, exogenous regressors can add to model
 110           state, increasing uncertainty. If both this parameter and
 111           exogenous_noise_decreases are False, exogenous regressors are ignored.
 112       exogenous_noise_decreases: If True, exogenous regressors can "set" model
 113           state, decreasing uncertainty. If both this parameter and
 114           exogenous_noise_increases are False, exogenous regressors are ignored.
 115       exogenous_feature_columns: A list of `tf.feature_column`s (for example
 116           `tf.feature_column.embedding_column`) corresponding to exogenous
 117           features which provide extra information to the model but are not part
 118           of the series to be predicted. Passed to
 119           `tf.feature_column.input_layer`.
 120       exogenous_update_condition: A function taking two Tensor arguments `times`
 121           (shape [batch size]) and `features` (a dictionary mapping exogenous
 122           feature keys to Tensors with shapes [batch size, ...]) and returning a
 123           boolean Tensor with shape [batch size] indicating whether state should
 124           be updated using exogenous features for each part of the batch. Where
 125           it is False, no exogenous update is performed. If None (default),
 126           exogenous updates are always performed. Useful for avoiding "leaky"
 127           frequent exogenous updates when sparse updates are desired. Called
 128           only during graph construction.
 129       filtering_maximum_posterior_variance_ratio: The maximum allowed ratio of
 130           two diagonal entries in a state covariance matrix just prior to
 131           filtering. Lower values mean that filtering will be more numerically
 132           stable, at the cost of artificially increasing estimated uncertainty
 133           in some cases. This parameter can be important when learning a
 134           transition matrix.
 135       filtering_minimum_posterior_variance: The minimum diagonal value in a
 136           state covariance matrix just prior to filtering, preventing numerical
 137           instability due to deterministic beliefs (sometimes an issue when
 138           learning transition matrices). This value should be set several orders
 139           of magnitude below any expected minimum state uncertainty.
 140       transition_covariance_initial_log_scale_bias: Controls the initial
 141           tradeoff between the transition noise covariance matrix and the
 142           observation noise covariance matrix, on a log scale (the elements of
 143           the transition noise covariance matrix are proportional to `e^{X +
 144           transition_covariance_initial_log_scale_bias}` where `X` is learned
 145           and may depend on input statistics, observation noise covariance is
 146           proportional to `e^{Y -
 147           transition_covariance_initial_log_scale_bias}`). For models *with*
 148           observation noise, -5 is a reasonable value. Models which do not use
 149           observation noise, and are not part of an ensemble which does use
 150           observation noise, should have this set to 0 or more to avoid
 151           numerical issues due to filtering with too little noise.
 152       static_unrolling_window_size_threshold: Only relevant for the top-level
 153           StateSpaceModel in an ensemble; enables switching between static and
 154           dynamic looping (if not None, default, meaning that no static
 155           unrolling is performed) based on the window size (windows with this
 156           size and smaller will have their graphs unrolled statically). See the
 157           SequentialTimeSeriesModel constructor for details.
 158     Returns:
 159       A StateSpaceModelConfiguration object.
 160     """
 161     if exogenous_feature_columns is None:
 162       exogenous_feature_columns = []
 163     return super(StateSpaceModelConfiguration, cls).__new__(
 164         cls, num_features, use_observation_noise, dtype,
 165         covariance_prior_fn, bayesian_prior_weighting,
 166         filtering_postprocessor, trainable_start_state,
 167         exogenous_noise_increases, exogenous_noise_decreases,
 168         exogenous_feature_columns, exogenous_update_condition,
 169         filtering_maximum_posterior_variance_ratio,
 170         filtering_minimum_posterior_variance,
 171         transition_covariance_initial_log_scale_bias,
 172         static_unrolling_window_size_threshold)

 175 class StateSpaceModel(model.SequentialTimeSeriesModel):
 176   """Base class for linear state space models.
 177 
 178   Sub-classes can specify the model to be learned by overriding
 179   get_state_transition, get_noise_transform, and get_observation_model.
 180 
 181   See kalman_filter.py for a detailed description of the class of models covered
 182   by StateSpaceModel.
 183 
 184   Briefly, state space models are defined by a state transition equation:
 185 
 186   state[t] = StateTransition * state[t-1] + NoiseTransform * StateNoise[t]
 187              + ExogenousNoiseIncreasing[t]
 188   StateNoise[t] ~ Gaussian(0, StateNoiseCovariance)
 189   ExogenousNoiseIncreasing[t] ~ Gaussian(ExogenousNoiseIncreasingMean[t],
 190                                          ExogenousNoiseIncreasingCovariance[t])
 191 
 192   And an observation model:
 193 
 194   observation[t] = ObservationModel * state[t] + ObservationNoise[t]
 195   ObservationNoise[t] ~ Gaussian(0, ObservationNoiseCovariance)
 196 
 197   Additionally, exogenous regressors can act as observations, decreasing
 198   uncertainty:
 199 
 200   ExogenousNoiseDecreasingObservation[t] ~ Gaussian(
 201       ExogenousNoiseDecreasingMean[t], ExogenousNoiseDecreasingCovariance[t])
 202 
 203   Attributes:
 204     kalman_filter: If initialize_graph has been called, the initialized
 205         KalmanFilter to use for inference. None otherwise.
 206     prior_state_mean: If initialize_graph has been called, a
 207         Variable-parameterized Tensor with shape [state dimension];
 208         the initial prior mean for one or more time series. None otherwise.
 209     prior_state_var: If initialize_graph has been called, a
 210         Variable-parameterized Tensor with shape [state dimension x state
 211         dimension]; the initial prior covariance. None otherwise.
 212     state_transition_noise_covariance: If initialize_graph has been called, a
 213         Variable-parameterized Tensor with shape [state noise dimension x state
 214         noise dimension] indicating the amount of noise added at each
 215         transition.
 216   """
 217 
 218   def __init__(self, configuration):
 219     """Initialize a state space model.
 220 
 221     Args:
 222       configuration: A StateSpaceModelConfiguration object.
 223     """
 224     self._configuration = configuration
 225     if configuration.filtering_postprocessor is not None:
 226       filtering_postprocessor_names = (
 227           configuration.filtering_postprocessor.output_names)
 228     else:
 229       filtering_postprocessor_names = []
 230     super(StateSpaceModel, self).__init__(
 231         train_output_names=(["mean", "covariance", "log_likelihood"]
 232                             + filtering_postprocessor_names),
 233         predict_output_names=["mean", "covariance"],
 234         num_features=configuration.num_features,
 235         normalize_features=True,
 236         dtype=configuration.dtype,
 237         exogenous_feature_columns=configuration.exogenous_feature_columns,
 238         exogenous_update_condition=configuration.exogenous_update_condition,
 239         static_unrolling_window_size_threshold=
 240         configuration.static_unrolling_window_size_threshold)
 241     self._kalman_filter = None
 242     self.prior_state_mean = None
 243     self.prior_state_var = None
 244     self.state_transition_noise_covariance = None
 245     self._total_observation_count = None
 246     self._observation_noise_covariance = None
 247     # Capture the current variable scope and use it to define all model
 248     # variables. Especially useful for ensembles, where variables may be defined
 249     # for every component model in one function call, which would otherwise
 250     # prevent the user from separating variables from different models into
 251     # different scopes.
 252     self._variable_scope = variable_scope.get_variable_scope()

 254   def transition_power_noise_accumulator(self, num_steps):
 255     r"""Sum a transitioned covariance matrix over a number of steps.
 256 
 257     Computes
 258 
 259       \sum_{i=0}^{num_steps - 1} (
 260         state_transition^i
 261         * state_transition_noise_covariance
 262         * (state_transition^i)^T)
 263 
 264     If special cases are available, overriding this function can lead to more
 265     efficient inferences.
 266 
 267     Args:
 268       num_steps: A [...] shape integer Tensor with numbers of steps to compute
 269         power sums for.
 270     Returns:
 271       The computed power sum, with shape [..., state dimension, state
 272       dimension].
 273     """
 274     # TODO(allenl): This general case should use cumsum if transition_to_powers
 275     # can be computed in constant time (important for correlated ensembles,
 276     # where transition_power_noise_accumulator special cases cannot be
 277     # aggregated from member models).
 278     noise_transform = ops.convert_to_tensor(self.get_noise_transform(),
 279                                             self.dtype)
 280     noise_transformed = math_ops.matmul(
 281         math_ops.matmul(noise_transform,
 282                         self.state_transition_noise_covariance),
 283         noise_transform,
 284         transpose_b=True)
 285     noise_additions = math_utils.power_sums_tensor(
 286         math_ops.reduce_max(num_steps) + 1,
 287         ops.convert_to_tensor(self.get_state_transition(), dtype=self.dtype),
 288         noise_transformed)
 289     return array_ops.gather(noise_additions, indices=num_steps)
 290 
 291   def transition_to_powers(self, powers):
 292     """Raise the transition matrix to a batch of powers.
 293 
 294     Computes state_transition^powers. If special cases are available, overriding
 295     this function can lead to more efficient inferences.
 296 
 297     Args:
 298       powers: A [...] shape integer Tensor with powers to raise the transition
 299         matrix to.
 300     Returns:
 301       The computed matrix powers, with shape [..., state dimension, state
 302       dimension].
 303     """
 304     return math_utils.matrix_to_powers(
 305         ops.convert_to_tensor(self.get_state_transition(), dtype=self.dtype),
 306         powers)
 307 
 308   def _window_initializer(self, times, state):
 309     """Prepare to impute across the gaps in a window."""
 310     _, _, priors_from_time = state
 311     times = ops.convert_to_tensor(times)
 312     priors_from_time = ops.convert_to_tensor(priors_from_time)
 313     intra_batch_gaps = array_ops.reshape(times[:, 1:] - times[:, :-1], [-1])
 314     # Ignore negative starting gaps, since there will be transient start times
 315     # as inputs statistics are computed.
 316     starting_gaps = math_ops.maximum(times[:, 0] - priors_from_time, 0)
 317     # Pre-define transition matrices raised to powers (and their sums) for every
 318     # gap in this window. This avoids duplicate computation (for example many
 319     # steps will use the transition matrix raised to the first power) and
 320     # batches the computation rather than doing it inside the per-step loop.
 321     unique_gaps, _ = array_ops.unique(
 322         array_ops.concat([intra_batch_gaps, starting_gaps], axis=0))
 323     self._window_power_sums = self.transition_power_noise_accumulator(
 324         unique_gaps)
 325     self._window_transition_powers = self.transition_to_powers(unique_gaps)
 326     self._window_gap_sizes = unique_gaps
 327 
 328   def _lookup_window_caches(self, caches, indices):
 329     _, window_power_ids = array_ops.unique(
 330         array_ops.concat(
 331             [
 332                 self._window_gap_sizes, math_ops.cast(
 333                     indices, self._window_gap_sizes.dtype)
 334             ],
 335             axis=0))
 336     all_gathered_indices = []
 337     for cache in caches:
 338       gathered_indices = array_ops.gather(
 339           cache, window_power_ids[-array_ops.shape(indices)[0]:])
 340       gathered_indices.set_shape(indices.get_shape().concatenate(
 341           gathered_indices.get_shape()[-2:]))
 342       all_gathered_indices.append(gathered_indices)
 343     return all_gathered_indices
 344 
 345   def _cached_transition_powers_and_sums(self, num_steps):
 346     return self._lookup_window_caches(
 347         caches=[self._window_transition_powers, self._window_power_sums],
 348         indices=num_steps)

 350   def _imputation_step(self, current_times, state):
 351     """Add state transition noise to catch `state` up to `current_times`.
 352 
 353     State space models are inherently sequential, so we need to "predict
 354     through" any missing time steps to catch up each element of the batch to its
 355     next observation/prediction time.
 356 
 357     Args:
 358       current_times: A [batch size] Tensor of times to impute up to, not
 359           inclusive.
 360       state: A tuple of (mean, covariance, previous_times) having shapes
 361           mean; [batch size x state dimension]
 362           covariance; [batch size x state dimension x state dimension]
 363           previous_times; [batch size]
 364     Returns:
 365       Imputed model state corresponding to the `state` argument.
 366     """
 367     estimated_state, estimated_state_var, previous_times = state
 368     # Ignore negative imputation intervals due to transient start time
 369     # estimates.
 370     catchup_times = math_ops.maximum(current_times - previous_times, 0)
 371     transition_matrices, transition_noise_sums = (  # pylint: disable=unbalanced-tuple-unpacking
 372         self._cached_transition_powers_and_sums(catchup_times))
 373     estimated_state = self._kalman_filter.predict_state_mean(
 374         estimated_state, transition_matrices)
 375     estimated_state_var = self._kalman_filter.predict_state_var(
 376         estimated_state_var, transition_matrices, transition_noise_sums)
 377     return (estimated_state, estimated_state_var,
 378             previous_times + catchup_times)
 379 
 380   def _filtering_step(self, current_times, current_values, state, predictions):
 381     """Compute posteriors and accumulate one-step-ahead predictions.
 382 
 383     Args:
 384       current_times: A [batch size] Tensor for times for each observation.
 385       current_values: A [batch size] Tensor of values for each observation.
 386       state: A tuple of (mean, covariance, previous_times) having shapes
 387           mean; [batch size x state dimension]
 388           covariance; [batch size x state dimension x state dimension]
 389           previous_times; [batch size]
 390       predictions: A dictionary containing mean and covariance Tensors, the
 391           output of _prediction_step.
 392     Returns:
 393       A tuple of (posteriors, outputs):
 394         posteriors: Model state updated to take `current_values` into account.
 395         outputs: The `predictions` dictionary updated to include "loss" and
 396             "log_likelihood" entries (loss simply being negative log
 397             likelihood).
 398     """
 399     estimated_state, estimated_state_covariance, previous_times = state
 400     observation_model = self.get_broadcasted_observation_model(current_times)
 401     imputed_to_current_step_assert = control_flow_ops.Assert(
 402         math_ops.reduce_all(math_ops.equal(current_times, previous_times)),
 403         ["Attempted to perform filtering without imputation/prediction"])
 404     with ops.control_dependencies([imputed_to_current_step_assert]):
 405       estimated_state_covariance = math_utils.clip_covariance(
 406           estimated_state_covariance,
 407           self._configuration.filtering_maximum_posterior_variance_ratio,
 408           self._configuration.filtering_minimum_posterior_variance)
 409       (filtered_state, filtered_state_covariance,
 410        log_prob) = self._kalman_filter.do_filter(
 411            estimated_state=estimated_state,
 412            estimated_state_covariance=estimated_state_covariance,
 413            predicted_observation=predictions["mean"],
 414            predicted_observation_covariance=predictions["covariance"],
 415            observation=current_values,
 416            observation_model=observation_model,
 417            observation_noise=self._observation_noise_covariance)
 418     filtered_state = (filtered_state, filtered_state_covariance, current_times)
 419     log_prob.set_shape(current_times.get_shape())
 420     predictions["loss"] = -log_prob
 421     predictions["log_likelihood"] = log_prob
 422     if self._configuration.filtering_postprocessor is not None:
 423       return self._configuration.filtering_postprocessor.process_filtering_step(
 424           current_times=current_times,
 425           current_values=current_values,
 426           predicted_state=state,
 427           filtered_state=filtered_state,
 428           outputs=predictions)
 429     return (filtered_state, predictions)
 430 
 431   def _scale_back_predictions(self, predictions):
 432     """Return a window of predictions to input scale."""
 433     predictions["mean"] = self._scale_back_data(predictions["mean"])
 434     predictions["covariance"] = self._scale_back_variance(
 435         predictions["covariance"])
 436     return predictions

 438   def _prediction_step(self, current_times, state):
 439     """Make a prediction based on `state`.
 440 
 441     Computes predictions based on the current `state`, checking that it has
 442     already been updated (in `_imputation_step`) to `current_times`.
 443 
 444     Args:
 445       current_times: A [batch size] Tensor for times to make predictions for.
 446       state: A tuple of (mean, covariance, previous_times) having shapes
 447           mean; [batch size x state dimension]
 448           covariance; [batch size x state dimension x state dimension]
 449           previous_times; [batch size]
 450     Returns:
 451       A tuple of (updated state, predictions):
 452         updated state: Model state with added transition noise.
 453         predictions: A dictionary with "mean" and "covariance", having shapes
 454             "mean": [batch size x num features]
 455             "covariance: [batch size x num features x num features]
 456     """
 457     estimated_state, estimated_state_var, previous_times = state
 458     advanced_to_current_assert = control_flow_ops.Assert(
 459         math_ops.reduce_all(math_ops.less_equal(current_times, previous_times)),
 460         ["Attempted to predict without imputation"])
 461     with ops.control_dependencies([advanced_to_current_assert]):
 462       observation_model = self.get_broadcasted_observation_model(current_times)
 463       predicted_obs, predicted_obs_var = (
 464           self._kalman_filter.observed_from_state(
 465               state_mean=estimated_state,
 466               state_var=estimated_state_var,
 467               observation_model=observation_model,
 468               observation_noise=self._observation_noise_covariance))
 469       predicted_obs_var.set_shape(
 470           ops.convert_to_tensor(current_times).get_shape()
 471           .concatenate([self.num_features, self.num_features]))
 472     predicted_obs.set_shape(current_times.get_shape().concatenate(
 473         (self.num_features,)))
 474     predicted_obs_var.set_shape(current_times.get_shape().concatenate(
 475         (self.num_features, self.num_features)))
 476     # Not scaled back to input-scale, since this also feeds into the
 477     # loss. Instead, predictions are scaled back before being returned to the
 478     # user in _scale_back_predictions.
 479     predictions = {
 480         "mean": predicted_obs,
 481         "covariance": predicted_obs_var}
 482     state = (estimated_state, estimated_state_var, current_times)
 483     return (state, predictions)
 484 
 485   def _exogenous_noise_decreasing(self, current_times, exogenous_values, state):
 486     """Update state with exogenous regressors, decreasing uncertainty.
 487 
 488     Constructs a mean and covariance based on transformations of
 489     `exogenous_values`, then performs Bayesian inference on the constructed
 490     observation. This has the effect of lowering uncertainty.
 491 
 492     This update refines or overrides previous inferences, useful for modeling
 493     exogenous inputs which "set" state, e.g. we dumped boiling water on the
 494     thermometer so we're pretty sure it's 100 degrees C.
 495 
 496     Args:
 497       current_times: A [batch size] Tensor of times for the exogenous values
 498           being input.
 499       exogenous_values: A [batch size x exogenous input dimension] Tensor of
 500           exogenous values for each part of the batch.
 501       state: A tuple of (mean, covariance, previous_times) having shapes
 502           mean; [batch size x state dimension]
 503           covariance; [batch size x state dimension x state dimension]
 504           previous_times; [batch size]
 505     Returns:
 506       Updated state taking the exogenous regressors into account (with lower
 507       uncertainty than the input state).
 508 
 509     """
 510     estimated_state, estimated_state_covariance, previous_times = state
 511     state_transition = ops.convert_to_tensor(
 512         self.get_state_transition(), dtype=self.dtype)
 513     state_dimension = state_transition.get_shape()[0].value
 514     # Learning the observation model would be redundant since we transform
 515     # `exogenous_values` to the state space via a linear transformation anyway.
 516     observation_model = linalg_ops.eye(
 517         state_dimension,
 518         batch_shape=array_ops.shape(exogenous_values)[:-1],
 519         dtype=self.dtype)
 520     with variable_scope.variable_scope("exogenous_noise_decreasing_covariance"):
 521       observation_noise = math_utils.transform_to_covariance_matrices(
 522           exogenous_values, state_dimension)
 523     with variable_scope.variable_scope(
 524         "exogenous_noise_decreasing_observation"):
 525       observation = layers.fully_connected(
 526           exogenous_values, state_dimension, activation_fn=None)
 527     # Pretend that we are making an observation with an observation model equal
 528     # to the identity matrix (i.e. a direct observation of the latent state),
 529     # with learned observation noise.
 530     posterior_state, posterior_state_var = (
 531         self._kalman_filter.posterior_from_prior_state(
 532             prior_state=estimated_state,
 533             prior_state_var=estimated_state_covariance,
 534             observation=observation,
 535             observation_model=observation_model,
 536             predicted_observations=(
 537                 estimated_state,
 538                 # The predicted noise covariance is noise due to current state
 539                 # uncertainty plus noise learned based on the exogenous
 540                 # observation (a somewhat trivial call to
 541                 # self._kalman_filter.observed_from_state has been omitted).
 542                 observation_noise + estimated_state_covariance),
 543             observation_noise=observation_noise))
 544     return (posterior_state, posterior_state_var, previous_times)
 545 
 546   def _exogenous_noise_increasing(self, current_times, exogenous_values, state):
 547     """Update state with exogenous regressors, increasing uncertainty.
 548 
 549     Adds to the state mean a linear transformation of `exogenous_values`, and
 550     increases uncertainty by constructing a covariance matrix based on
 551     `exogenous_values` and adding it to the state covariance.
 552 
 553     This update is useful for modeling changes relative to current state,
 554     e.g. the furnace turned on so the temperature will be increasing at an
 555     additional 1 degree per minute with some uncertainty, this uncertainty being
 556     added to our current uncertainty in the per-minute change in temperature.
 557 
 558     Args:
 559       current_times: A [batch size] Tensor of times for the exogenous values
 560           being input.
 561       exogenous_values: A [batch size x exogenous input dimension] Tensor of
 562           exogenous values for each part of the batch.
 563       state: A tuple of (mean, covariance, previous_times) having shapes
 564           mean; [batch size x state dimension]
 565           covariance; [batch size x state dimension x state dimension]
 566           previous_times; [batch size]
 567     Returns:
 568       Updated state taking the exogenous regressors into account (with higher
 569       uncertainty than the input state).
 570 
 571     """
 572     start_mean, start_covariance, previous_times = state
 573     with variable_scope.variable_scope("exogenous_noise_increasing_mean"):
 574       mean_addition = layers.fully_connected(
 575           exogenous_values, start_mean.get_shape()[1].value, activation_fn=None)
 576     state_dimension = start_covariance.get_shape()[1].value
 577     with variable_scope.variable_scope("exogenous_noise_increasing_covariance"):
 578       covariance_addition = (
 579           math_utils.transform_to_covariance_matrices(
 580               exogenous_values, state_dimension))
 581     return (start_mean + mean_addition,
 582             start_covariance + covariance_addition,
 583             previous_times)

 585   def _exogenous_input_step(
 586       self, current_times, current_exogenous_regressors, state):
 587     """Update state with exogenous regressors.
 588 
 589     Allows both increases and decreases in uncertainty.
 590 
 591     Args:
 592       current_times: A [batch size] Tensor of times for the exogenous values
 593           being input.
 594       current_exogenous_regressors: A [batch size x exogenous input dimension]
 595           Tensor of exogenous values for each part of the batch.
 596       state: A tuple of (mean, covariance, previous_times) having shapes
 597           mean; [batch size x state dimension]
 598           covariance; [batch size x state dimension x state dimension]
 599           previous_times; [batch size]
 600     Returns:
 601       Updated state taking the exogenous regressors into account.
 602     """
 603     if self._configuration.exogenous_noise_decreases:
 604       state = self._exogenous_noise_decreasing(
 605           current_times, current_exogenous_regressors, state)
 606     if self._configuration.exogenous_noise_increases:
 607       state = self._exogenous_noise_increasing(
 608           current_times, current_exogenous_regressors, state)
 609     return state

 611   def _loss_additions(self, times, values, mode):
 612     """Add regularization during training."""
 613     if mode == estimator_lib.ModeKeys.TRAIN:
 614       if (self._input_statistics is not None
 615           and self._configuration.bayesian_prior_weighting):
 616         normalization = 1. / math_ops.cast(
 617             self._input_statistics.total_observation_count, self.dtype)
 618       else:
 619         # If there is no total observation count recorded, or if we are not
 620         # doing a Bayesian prior weighting, assumes/pretends that the full
 621         # dataset size is the window size.
 622         normalization = 1. / math_ops.cast(
 623             array_ops.shape(times)[1], self.dtype)
 624       transition_contribution = ops.convert_to_tensor(
 625           self._configuration.covariance_prior_fn(
 626               self.state_transition_noise_covariance),
 627           dtype=self.dtype)
 628       if (self._configuration.use_observation_noise
 629           and self._observation_noise_covariance is not None):
 630         observation_contribution = ops.convert_to_tensor(
 631             self._configuration.covariance_prior_fn(
 632                 self._observation_noise_covariance),
 633             dtype=self.dtype)
 634         regularization_sum = transition_contribution + observation_contribution
 635       else:
 636         regularization_sum = transition_contribution
 637       return -normalization * regularization_sum
 638     else:
 639       return array_ops.zeros([], dtype=self.dtype)

 650   def _define_parameters(self, observation_transition_tradeoff_log=None):
 651     """Define extra model-specific parameters.
 652 
 653     Models should wrap any variables defined here in the model's variable scope.
 654 
 655     Args:
 656       observation_transition_tradeoff_log: An ensemble-global parameter
 657         controlling the tradeoff between observation noise and transition
 658         noise. If its value is not None, component transition noise should scale
 659         with e^-observation_transition_tradeoff_log.
 660     """
 661     with variable_scope.variable_scope(self._variable_scope):
 662       # A scalar which allows the optimizer to quickly shift from observation
 663       # noise to transition noise (this value is subtracted from log transition
 664       # noise and added to log observation noise).
 665       if observation_transition_tradeoff_log is None:
 666         self._observation_transition_tradeoff_log_scale = (
 667             self._variable_observation_transition_tradeoff_log())
 668       else:
 669         self._observation_transition_tradeoff_log_scale = (
 670             observation_transition_tradeoff_log)
 671       self.state_transition_noise_covariance = (
 672           self.get_state_transition_noise_covariance())

 678   def initialize_graph(self, input_statistics=None):
 679     """Define variables and ops relevant to the top-level model in an ensemble.
 680 
 681     For generic model parameters, _define_parameters() is called recursively on
 682     all members of an ensemble.
 683 
 684     Args:
 685       input_statistics: A math_utils.InputStatistics object containing input
 686           statistics. If None, data-independent defaults are used, which may
 687           result in longer or unstable training.
 688     """
 689     self._set_input_statistics(input_statistics=input_statistics)
 690     self._define_parameters()
 691     with variable_scope.variable_scope(self._variable_scope):
 692       self._observation_noise_covariance = ops.convert_to_tensor(
 693           self.get_observation_noise_covariance(), dtype=self.dtype)
 694     self._kalman_filter = kalman_filter.KalmanFilter(dtype=self.dtype)

 696      self.prior_state_var) = self._make_priors()

 698   def _make_priors(self):
 699     """Creates and returns model priors."""
 700     prior_state_covariance = self.get_prior_covariance()
 701     prior_state_mean = self.get_prior_mean()
 702     return (prior_state_mean, prior_state_covariance)

 704   def get_prior_covariance(self):
 705     """Constructs a variable prior covariance with data-based initialization.
 706 
 707     Models should wrap any variables defined here in the model's variable scope.
 708 
 709     Returns:
 710       A two-dimensional [state dimension, state dimension] floating point Tensor
 711       with a (positive definite) prior state covariance matrix.
 712     """
 713     with variable_scope.variable_scope(self._variable_scope):
 714       state_dimension = ops.convert_to_tensor(
 715           self.get_state_transition()).get_shape()[0].value
 716       if self._configuration.trainable_start_state:
 717         base_covariance = math_utils.variable_covariance_matrix(
 718             state_dimension, "prior_state_var",
 719             dtype=self.dtype)
 720       else:
 721         return linalg_ops.eye(state_dimension, dtype=self.dtype)
 722       if self._input_statistics is not None:
 723         # Make sure initial latent value uncertainty is at least on the same
 724         # scale as noise in the data.
 725         covariance_multiplier = math_ops.reduce_max(
 726             self._scale_variance(
 727                 self._input_statistics.series_start_moments.variance))
 728         return base_covariance * gen_math_ops.maximum(
 729             covariance_multiplier, 1.0)
 730       else:
 731         return base_covariance

 736     Models should wrap any variables defined here in the model's variable scope.
 737

 740       indicating the prior mean.

 743       state_transition = ops.convert_to_tensor(

 752   # TODO(allenl): It would be nice if the generation were done with TensorFlow

 755   # special set of initializers?
 756   def random_model_parameters(self, seed=None):
 757     if self.num_features != 1:
 758       raise NotImplementedError("Generation for multivariate state space models"
 759                                 " is not currently implemented.")
 760     if seed:
 761       numpy.random.seed(seed)
 762     state_dimension, noise_dimension = ops.convert_to_tensor(
 763         self.get_noise_transform()).get_shape().as_list()
 764     transition_var = 1.0 / numpy.random.gamma(shape=10., scale=10.,
 765                                               size=[noise_dimension])
 766     initial_state = numpy.random.normal(size=[state_dimension])
 767     params_dict = {}
 768     if self.prior_state_mean is not None:
 769       params_dict[self.prior_state_mean] = initial_state
 770     if self.state_transition_noise_covariance is not None:
 771       params_dict[self.state_transition_noise_covariance] = numpy.diag(
 772           transition_var)
 773     if self.prior_state_var is not None:
 774       params_dict[self.prior_state_var] = numpy.zeros(
 775           [state_dimension, state_dimension])
 776     if self._configuration.use_observation_noise:
 777       observation_var = 1.0 / numpy.random.gamma(shape=4, scale=4)
 778       params_dict[self._observation_noise_covariance] = [[observation_var]]
 779     return params_dict

 781   def generate(self, number_of_series, series_length,
 782                model_parameters=None, seed=None, add_observation_noise=None):
 783     if seed is not None:
 784       numpy.random.seed(seed)
 785     if self.num_features != 1:
 786       raise NotImplementedError("Generation for multivariate state space models"
 787                                 " is not currently implemented.")
 788     if add_observation_noise is None:
 789       add_observation_noise = self._configuration.use_observation_noise
 790     if model_parameters is None:
 791       model_parameters = {}
 792     transitions = ops.convert_to_tensor(
 793         self.get_state_transition(), dtype=self.dtype).eval(
 794             feed_dict=model_parameters)
 795     noise_transform = ops.convert_to_tensor(self.get_noise_transform()).eval(
 796         feed_dict=model_parameters)
 797 
 798     noise_dimension = noise_transform.shape[1]
 799     get_passed_or_trained_value = model_utils.parameter_switch(model_parameters)
 800     transition_var = numpy.diag(get_passed_or_trained_value(
 801         self.state_transition_noise_covariance))
 802     transition_std = numpy.sqrt(transition_var)
 803     if add_observation_noise:
 804       observation_var = get_passed_or_trained_value(
 805           self._observation_noise_covariance)[0][0]
 806       observation_std = numpy.sqrt(observation_var)
 807     initial_state = get_passed_or_trained_value(self.prior_state_mean)
 808     current_state = numpy.tile(numpy.expand_dims(initial_state, 0),
 809                                [number_of_series, 1])
 810     observations = numpy.zeros([number_of_series, series_length])
 811     observation_models = self.get_broadcasted_observation_model(
 812         times=math_ops.range(series_length)).eval(feed_dict=model_parameters)
 813     for timestep, observation_model in enumerate(observation_models):
 814       current_state = numpy.dot(current_state, transitions.T)
 815       current_state += numpy.dot(
 816           numpy.random.normal(
 817               loc=numpy.zeros([number_of_series, noise_dimension]),
 818               scale=numpy.tile(numpy.expand_dims(transition_std, 0),
 819                                [number_of_series, 1])),
 820           noise_transform.T)
 821       observation_mean = numpy.dot(current_state, observation_model[0].T)
 822       if add_observation_noise:
 823         observations[:, timestep] = numpy.random.normal(loc=observation_mean,
 824                                                         scale=observation_std)
 825       else:
 826         observations[:, timestep] = observation_mean
 827     observations = numpy.expand_dims(observations, -1)
 828     times = numpy.tile(
 829         numpy.expand_dims(numpy.arange(observations.shape[1]), 0),
 830         [observations.shape[0], 1])
 831     return {TrainEvalFeatures.TIMES: times,
 832             TrainEvalFeatures.VALUES: observations}

 872   def get_broadcasted_observation_model(self, times):
 873     """Broadcast this model's observation model if necessary.
 874 
 875     The model can define a univariate observation model which will be broadcast
 876     over both self.num_features and the batch dimension of `times`.
 877 
 878     The model can define a multi-variate observation model which does not depend
 879     on `times`, and it will be broadcast over the batch dimension of `times`.
 880 
 881     Finally, the model can define a multi-variate observation model with a batch
 882     dimension, which will not be broadcast.
 883 
 884     Args:
 885       times: A [batch dimension] int32 Tensor with times for each part of the
 886           batch, on which the observation model can depend.
 887     Returns:
 888       A [batch dimension x self.num_features x state dimension] Tensor
 889       specifying the observation model to use for each time in `times` and each
 890       feature.
 891     """
 892     unbroadcasted_model = ops.convert_to_tensor(
 893         self.get_observation_model(times), dtype=self.dtype)
 894     unbroadcasted_shape = (unbroadcasted_model.get_shape()
 895                            .with_rank_at_least(1).with_rank_at_most(3))
 896     if unbroadcasted_shape.ndims is None:
 897       # Pass through fully undefined shapes, but make sure they're rank 3 at
 898       # graph eval time
 899       assert_op = control_flow_ops.Assert(
 900           math_ops.equal(array_ops.rank(unbroadcasted_model), 3),
 901           [array_ops.shape(unbroadcasted_model)])
 902       with ops.control_dependencies([assert_op]):
 903         return array_ops.identity(unbroadcasted_model)
 904     if unbroadcasted_shape.ndims == 1:
 905       # Unbroadcasted shape [state dimension]
 906       broadcasted_model = array_ops.tile(
 907           array_ops.reshape(tensor=unbroadcasted_model, shape=[1, 1, -1]),
 908           [array_ops.shape(times)[0], self.num_features, 1])
 909     elif unbroadcasted_shape.ndims == 2:
 910       # Unbroadcasted shape [num features x state dimension]
 911       broadcasted_model = array_ops.tile(
 912           array_ops.expand_dims(unbroadcasted_model, axis=0),
 913           [array_ops.shape(times)[0], 1, 1])
 914     elif unbroadcasted_shape.ndims == 3:
 915       broadcasted_model = unbroadcasted_model
 916     broadcasted_model.get_shape().assert_has_rank(3)
 917     return broadcasted_model

 919   def get_state_transition_noise_covariance(
 920       self, minimum_initial_variance=1e-5):
 921     state_noise_transform = ops.convert_to_tensor(
 922         self.get_noise_transform(), dtype=self.dtype)
 923     state_noise_dimension = state_noise_transform.get_shape()[1].value
 924     if self._input_statistics is not None:
 925       feature_variance = self._scale_variance(
 926           self._input_statistics.series_start_moments.variance)
 927       initial_transition_noise_scale = math_ops.log(
 928           gen_math_ops.maximum(
 929               math_ops.reduce_mean(feature_variance) / math_ops.cast(
 930                   self._input_statistics.total_observation_count, self.dtype),
 931               minimum_initial_variance))
 932     else:
 933       initial_transition_noise_scale = 0.
 934     # Generally high transition noise is undesirable; we want to set it quite
 935     # low to start so that we don't need too much training to get to good
 936     # solutions (i.e. with confident predictions into the future if possible),
 937     # but not so low that training can't yield a high transition noise if the
 938     # data demands it.
 939     initial_transition_noise_scale -= (
 940         self._observation_transition_tradeoff_log_scale)
 941     return math_utils.variable_covariance_matrix(
 942         state_noise_dimension, "state_transition_noise",
 943         dtype=self.dtype,
 944         initial_overall_scale_log=initial_transition_noise_scale)
 945 
 946   def get_observation_noise_covariance(self, minimum_initial_variance=1e-5):
 947     if self._configuration.use_observation_noise:
 948       if self._input_statistics is not None:
 949         # Get variance across the first few values in each batch for each
 950         # feature, for an initial observation noise (over-)estimate.
 951         feature_variance = self._scale_variance(
 952             self._input_statistics.series_start_moments.variance)
 953       else:
 954         feature_variance = None
 955       if feature_variance is not None:
 956         feature_variance = gen_math_ops.maximum(feature_variance,
 957                                                 minimum_initial_variance)
 958       return math_utils.variable_covariance_matrix(
 959           size=self.num_features,
 960           dtype=self.dtype,
 961           name="observation_noise_covariance",
 962           initial_diagonal_values=feature_variance,
 963           initial_overall_scale_log=(
 964               self._observation_transition_tradeoff_log_scale))
 965     else:
 966       return array_ops.zeros(
 967           shape=[self.num_features, self.num_features],
 968           name="observation_noise_covariance",
 969           dtype=self.dtype)

1094 class StateSpaceIndependentEnsemble(StateSpaceEnsemble):
1095   """Implements ensembles of independent state space models.
1096 
1097   Useful for fitting multiple independent state space models together while
1098   keeping their specifications decoupled. The "ensemble" is simply a state space
1099   model with the observation models of its members concatenated, and the
1100   transition matrices and noise transforms stacked in block-diagonal
1101   matrices. This means that the dimensionality of the ensemble's state is the
1102   sum of those of its components, which can lead to slow and memory-intensive
1103   training and inference as the posterior (shape [state dimension x state
1104   dimension]) gets large.
1105 
1106   Each individual model j's state at time t is defined by:
1107 
1108   state[t, j] = StateTransition[j] * state[t-1, j]
1109       + NoiseTransform[j] * StateNoise[t, j]
1110   StateNoise[t, j] ~ Gaussian(0, StateNoiseCovariance[j])
1111 
1112   and the ensemble observation model is:
1113 
1114   observation[t] = Sum { ObservationModel[j] * state[t, j] }
1115       + ObservationNoise[t]
1116   ObservationNoise[t] ~ Gaussian(0, ObservationNoiseCovariance)
1117   """
1118

1120     return self._compute_blocked(

1122         name="ensemble_power_noise_accumulator")

1146 class StateSpaceCorrelatedFeaturesEnsemble(StateSpaceEnsemble):
1147   """An correlated ensemble where each model represents a feature.
1148 
1149   Unlike `StateSpaceIndependentEnsemble`, a full state transition noise
1150   covariance matrix is learned for this ensemble; the models are not assumed to
1151   be independent. Rather than concatenating observation models (i.e. summing the
1152   contributions of each model to each feature),
1153   StateSpaceCorrelatedFeaturesEnsemble stacks observation models diagonally,
1154   meaning that each model corresponds to one feature of the series.
1155 
1156   Behaves like (and is) a single state space model where:
1157 
1158   StateTransition = Diag(StateTransition[j] for models j)
1159   ObservationModel = Diag(ObservationModel[j] for models j)
1160 
1161   Note that each ObservationModel[j] is a [1 x S_j] matrix (S_j being the state
1162   dimension of model j), i.e. a univariate model. The combined model is
1163   multivariate, the number of features of the series being equal to the number
1164   of component models in the ensemble.
1165   """
1166 
1167   def __init__(self, ensemble_members, configuration):
1168     """Specify the ensemble's configuration and component models.
1169 
1170     Args:
1171       ensemble_members: A list of `StateSpaceModel` objects, with length equal
1172         to `configuration.num_features`. Each of these models, which must be
1173         univariate, corresponds to a single feature of the time series.
1174       configuration: A StateSpaceModelConfiguration object.
1175     Raises:
1176       ValueError: If the length of `ensemble_members` does not equal the number
1177         of features in the series, or any component is not univariate.
1178     """
1179     if len(ensemble_members) != configuration.num_features:
1180       raise ValueError(
1181           "The number of members in a StateSpaceCorrelatedFeaturesEnsemble "
1182           "must equal the number of features in the time series.")
1183     for member in ensemble_members:
1184       if member.num_features != 1:
1185         raise ValueError(
1186             "StateSpaceCorrelatedFeaturesEnsemble components must be "
1187             "univariate.")
1188     super(StateSpaceCorrelatedFeaturesEnsemble, self).__init__(
1189         ensemble_members=ensemble_members, configuration=configuration)

1192     """Use a noise accumulator special case when possible."""
1193     if len(self._ensemble_members) == 1:
