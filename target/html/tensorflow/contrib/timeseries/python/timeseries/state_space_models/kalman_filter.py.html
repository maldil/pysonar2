  35 class KalmanFilter(object):
  36   """Inference on linear state models.
  37 
  38   The model for observations in a given state is:
  39     observation(t) = observation_model * state(t)
  40         + Gaussian(0, observation_noise_covariance)
  41 
  42   State updates take the following form:
  43     state(t) = state_transition * state(t-1)
  44         + state_noise_transform * Gaussian(0, state_transition_noise_covariance)
  45 
  46   This is a real-valued analog to hidden Markov models, with linear transitions
  47   and a Gaussian noise model. Given initial conditions, noise, and state
  48   transition, Kalman filtering recursively estimates states and observations,
  49   along with their associated uncertainty. When fed observations, future state
  50   and uncertainty estimates are conditioned on those observations (in a Bayesian
  51   sense).
  52 
  53   Typically some "given"s mentioned above (noises) will be unknown, and so
  54   optimizing the Kalman filter's probabilistic predictions with respect to these
  55   parameters is a good approach. The state transition and observation models are
  56   usually known a priori as a modeling decision.
  57 
  58   """
  59 
  60   def __init__(self, dtype=dtypes.float32,
  61                simplified_posterior_covariance_computation=False):
  62     """Initialize the Kalman filter.
  63 
  64     Args:
  65       dtype: The data type to use for floating point tensors.
  66       simplified_posterior_covariance_computation: If True, uses an algebraic
  67         simplification of the Kalman filtering posterior covariance update,
  68         which is slightly faster at the cost of numerical stability. The
  69         simplified update is often stable when using double precision on small
  70         models or with fixed transition matrices.
  71     """
  72     self._simplified_posterior_covariance_computation = (
  73         simplified_posterior_covariance_computation)
  74     self.dtype = dtype
  75 
  76   def do_filter(
  77       self, estimated_state, estimated_state_covariance,
  78       predicted_observation, predicted_observation_covariance,
  79       observation, observation_model, observation_noise):
  80     """Convenience function for scoring predictions.
  81 
  82     Scores a prediction against an observation, and computes the updated
  83     posterior over states.
  84 
  85     Shapes given below for arguments are for single-model Kalman filtering
  86     (e.g. KalmanFilter). For ensembles, prior_state and prior_state_var are
  87     same-length tuples of values corresponding to each model.
  88 
  89     Args:
  90       estimated_state: A prior mean over states [batch size x state dimension]
  91       estimated_state_covariance: Covariance of state prior [batch size x D x
  92           D], with D depending on the Kalman filter implementation (typically
  93           the state dimension).
  94       predicted_observation: A prediction for the observed value, such as that
  95           returned by observed_from_state. A [batch size x num features] Tensor.
  96       predicted_observation_covariance: A covariance matrix corresponding to
  97           `predicted_observation`, a [batch size x num features x num features]
  98           Tensor.
  99       observation: The observed value corresponding to the predictions
 100           given [batch size x observation dimension]
 101       observation_model: The [batch size x observation dimension x model state
 102           dimension] Tensor indicating how a particular state is mapped to
 103           (pre-noise) observations for each part of the batch.
 104       observation_noise: A [batch size x observation dimension x observation
 105           dimension] Tensor or [observation dimension x observation dimension]
 106           Tensor with covariance matrices to use for each part of the batch (a
 107           two-dimensional input will be broadcast).
 108     Returns:
 109       posterior_state, posterior_state_var: Posterior mean and
 110           covariance, updated versions of prior_state and
 111           prior_state_var.
 112       log_prediction_prob: Log probability of the observations under
 113           the priors, suitable for optimization (should be maximized).
 114 
 115     """
 116     symmetrized_observation_covariance = 0.5 * (
 117         predicted_observation_covariance + array_ops.matrix_transpose(
 118             predicted_observation_covariance))
 119     instability_message = (
 120         "This may occur due to numerically unstable filtering when there is "
 121         "a large difference in posterior variances, or when inferences are "
 122         "near-deterministic. Considering tuning the "
 123         "'filtering_maximum_posterior_variance_ratio' or "
 124         "'filtering_minimum_posterior_variance' parameters in your "
 125         "StateSpaceModelConfiguration, or tuning the transition matrix.")
 126     symmetrized_observation_covariance = numerics.verify_tensor_all_finite(
 127         symmetrized_observation_covariance,
 128         "Predicted observation covariance was not finite. {}".format(
 129             instability_message))
 130     diag = array_ops.matrix_diag_part(symmetrized_observation_covariance)
 131     min_diag = math_ops.reduce_min(diag)
 132     non_negative_assert = control_flow_ops.Assert(
 133         min_diag >= 0.,
 134         [("The predicted observation covariance "
 135           "has a negative diagonal entry. {}").format(instability_message),
 136          min_diag])
 137     with ops.control_dependencies([non_negative_assert]):
 138       observation_covariance_cholesky = linalg_ops.cholesky(
 139           symmetrized_observation_covariance)
 140     log_prediction_prob = distributions.MultivariateNormalTriL(
 141         predicted_observation, observation_covariance_cholesky).log_prob(
 142             observation)
 143     (posterior_state,
 144      posterior_state_var) = self.posterior_from_prior_state(
 145          prior_state=estimated_state,
 146          prior_state_var=estimated_state_covariance,
 147          observation=observation,
 148          observation_model=observation_model,
 149          predicted_observations=(predicted_observation,
 150                                  predicted_observation_covariance),
 151          observation_noise=observation_noise)
 152     return (posterior_state, posterior_state_var, log_prediction_prob)
 153 
 154   def predict_state_mean(self, prior_state, transition_matrices):
 155     """Compute state transitions.
 156 
 157     Args:
 158       prior_state: Current estimated state mean [batch_size x state_dimension]
 159       transition_matrices: A [batch size, state dimension, state dimension]
 160         batch of matrices (dtype matching the `dtype` argument to the
 161         constructor) with the transition matrix raised to the power of the
 162         number of steps to be taken (not element-wise; use
 163         math_utils.matrix_to_powers if there is no efficient special case) if
 164         more than one step is desired.
 165     Returns:
 166       State mean advanced based on `transition_matrices` (dimensions matching
 167       first argument).
 168     """
 169     advanced_state = array_ops.squeeze(
 170         math_ops.matmul(
 171             transition_matrices,
 172             prior_state[..., None]),
 173         axis=[-1])
 174     return advanced_state
 175 
 176   def predict_state_var(
 177       self, prior_state_var, transition_matrices, transition_noise_sums):
 178     r"""Compute variance for state transitions.
 179 
 180     Computes a noise estimate corresponding to the value returned by
 181     predict_state_mean.
 182 
 183     Args:
 184       prior_state_var: Covariance matrix specifying uncertainty of current state
 185           estimate [batch size x state dimension x state dimension]
 186       transition_matrices: A [batch size, state dimension, state dimension]
 187         batch of matrices (dtype matching the `dtype` argument to the
 188         constructor) with the transition matrix raised to the power of the
 189         number of steps to be taken (not element-wise; use
 190         math_utils.matrix_to_powers if there is no efficient special case).
 191       transition_noise_sums: A [batch size, state dimension, state dimension]
 192         Tensor (dtype matching the `dtype` argument to the constructor) with:
 193 
 194           \sum_{i=0}^{num_steps - 1} (
 195              state_transition_to_powers_fn(i)
 196              * state_transition_noise_covariance
 197              * state_transition_to_powers_fn(i)^T
 198           )
 199 
 200         for the number of steps to be taken in each part of the batch (this
 201         should match `transition_matrices`). Use math_utils.power_sums_tensor
 202         with `tf.gather` if there is no efficient special case.
 203     Returns:
 204       State variance advanced based on `transition_matrices` and
 205       `transition_noise_sums` (dimensions matching first argument).
 206     """
 207     prior_variance_transitioned = math_ops.matmul(
 208         math_ops.matmul(transition_matrices, prior_state_var),
 209         transition_matrices,
 210         adjoint_b=True)
 211     return prior_variance_transitioned + transition_noise_sums

 213   def posterior_from_prior_state(self, prior_state, prior_state_var,
 214                                  observation, observation_model,
 215                                  predicted_observations,
 216                                  observation_noise):
 217     """Compute a posterior over states given an observation.
 218 
 219     Args:
 220       prior_state: Prior state mean [batch size x state dimension]
 221       prior_state_var: Prior state covariance [batch size x state dimension x
 222           state dimension]
 223       observation: The observed value corresponding to the predictions given
 224           [batch size x observation dimension]
 225       observation_model: The [batch size x observation dimension x model state
 226           dimension] Tensor indicating how a particular state is mapped to
 227           (pre-noise) observations for each part of the batch.
 228       predicted_observations: An (observation mean, observation variance) tuple
 229           computed based on the current state, usually the output of
 230           observed_from_state.
 231       observation_noise: A [batch size x observation dimension x observation
 232           dimension] or [observation dimension x observation dimension] Tensor
 233           with covariance matrices to use for each part of the batch (a
 234           two-dimensional input will be broadcast).
 235     Returns:
 236       Posterior mean and covariance (dimensions matching the first two
 237       arguments).
 238 
 239     """
 240     observed_mean, observed_var = predicted_observations
 241     residual = observation - observed_mean
 242     # TODO(allenl): Can more of this be done using matrix_solve_ls?
 243     kalman_solve_rhs = math_ops.matmul(
 244         observation_model, prior_state_var, adjoint_b=True)
 245     # This matrix_solve adjoint doesn't make a difference symbolically (since
 246     # observed_var is a covariance matrix, and should be symmetric), but
 247     # filtering on multivariate series is unstable without it. See
 248     # test_multivariate_symmetric_covariance_float64 in kalman_filter_test.py
 249     # for an example of the instability (fails with adjoint=False).
 250     kalman_gain_transposed = linalg_ops.matrix_solve(
 251         matrix=observed_var, rhs=kalman_solve_rhs, adjoint=True)
 252     posterior_state = prior_state + array_ops.squeeze(
 253         math_ops.matmul(
 254             kalman_gain_transposed,
 255             array_ops.expand_dims(residual, -1),
 256             adjoint_a=True),
 257         axis=[-1])
 258     gain_obs = math_ops.matmul(
 259         kalman_gain_transposed, observation_model, adjoint_a=True)
 260     identity_extradim = linalg_ops.eye(
 261         array_ops.shape(gain_obs)[1], dtype=gain_obs.dtype)[None]
 262     identity_minus_factor = identity_extradim - gain_obs
 263     if self._simplified_posterior_covariance_computation:
 264       # posterior covariance =
 265       #   (I - kalman_gain * observation_model) * prior_state_var
 266       posterior_state_var = math_ops.matmul(identity_minus_factor,
 267                                             prior_state_var)
 268     else:
 269       observation_noise = ops.convert_to_tensor(observation_noise)
 270       # A Joseph form update, which provides better numeric stability than the
 271       # simplified optimal Kalman gain update, at the cost of a few extra
 272       # operations. Joseph form updates are valid for any gain (not just the
 273       # optimal Kalman gain), and so are more forgiving of numerical errors in
 274       # computing the optimal Kalman gain.
 275       #
 276       # posterior covariance =
 277       #   (I - kalman_gain * observation_model) * prior_state_var
 278       #     * (I - kalman_gain * observation_model)^T
 279       #   + kalman_gain * observation_noise * kalman_gain^T
 280       left_multiplied_state_var = math_ops.matmul(identity_minus_factor,
 281                                                   prior_state_var)
 282       multiplied_state_var = math_ops.matmul(
 283           identity_minus_factor, left_multiplied_state_var, adjoint_b=True)
 284       def _batch_observation_noise_update():
 285         return (multiplied_state_var + math_ops.matmul(
 286             math_ops.matmul(
 287                 kalman_gain_transposed, observation_noise, adjoint_a=True),
 288             kalman_gain_transposed))
 289       def _matrix_observation_noise_update():
 290         return (multiplied_state_var + math_ops.matmul(
 291             math_utils.batch_times_matrix(
 292                 kalman_gain_transposed, observation_noise, adj_x=True),
 293             kalman_gain_transposed))
 294       if observation_noise.get_shape().ndims is None:
 295         posterior_state_var = control_flow_ops.cond(
 296             math_ops.equal(array_ops.rank(observation_noise), 2),
 297             _matrix_observation_noise_update, _batch_observation_noise_update)
 298       else:
 299         # If static shape information exists, it gets checked in each cond()
 300         # branch, so we need a special case to avoid graph-build-time
 301         # exceptions.
 302         if observation_noise.get_shape().ndims == 2:
 303           posterior_state_var = _matrix_observation_noise_update()
 304         else:
 305           posterior_state_var = _batch_observation_noise_update()
 306     return posterior_state, posterior_state_var

 308   def observed_from_state(self, state_mean, state_var, observation_model,
 309                           observation_noise):
 310     """Compute an observation distribution given a state distribution.
 311 
 312     Args:
 313       state_mean: State mean vector [batch size x state dimension]
 314       state_var: State covariance [batch size x state dimension x state
 315           dimension]
 316       observation_model: The [batch size x observation dimension x model state
 317           dimension] Tensor indicating how a particular state is mapped to
 318           (pre-noise) observations for each part of the batch.
 319       observation_noise: A [batch size x observation dimension x observation
 320           dimension] Tensor with covariance matrices to use for each part of the
 321           batch. To remove observation noise, pass a Tensor of zeros (or simply
 322           0, which will broadcast).
 323     Returns:
 324       observed_mean: Observation mean vector [batch size x observation
 325           dimension]
 326       observed_var: Observation covariance [batch size x observation dimension x
 327           observation dimension]
 328 
 329     """
 330     observed_mean = array_ops.squeeze(
 331         math_ops.matmul(
 332             array_ops.expand_dims(state_mean, 1),
 333             observation_model,
 334             adjoint_b=True),
 335         axis=[1])
 336     observed_var = math_ops.matmul(
 337         math_ops.matmul(observation_model, state_var),
 338         observation_model,
 339         adjoint_b=True)
 340     observed_var += observation_noise
 341     return observed_mean, observed_var
