  46 def clip_covariance(
  47     covariance_matrix, maximum_variance_ratio, minimum_variance):
  48   """Enforce constraints on a covariance matrix to improve numerical stability.
  49 
  50   Args:
  51     covariance_matrix: A [..., N, N] batch of covariance matrices.
  52     maximum_variance_ratio: The maximum allowed ratio of two diagonal
  53       entries. Any entries lower than the maximum entry divided by this ratio
  54       will be set to that value.
  55     minimum_variance: A floor for diagonal entries in the returned matrix.
  56   Returns:
  57     A new covariance matrix with the requested constraints enforced. If the
  58     input was positive definite, the output will be too.
  59   """
  60   # TODO(allenl): Smarter scaling here so that correlations are preserved when
  61   # fiddling with diagonal elements.
  62   diagonal = array_ops.matrix_diag_part(covariance_matrix)
  63   maximum = math_ops.reduce_max(diagonal, axis=-1, keepdims=True)
  64   new_diagonal = gen_math_ops.maximum(
  65       diagonal, maximum / maximum_variance_ratio)
  66   return array_ops.matrix_set_diag(
  67       covariance_matrix, math_ops.maximum(new_diagonal, minimum_variance))

  70 def block_diagonal(matrices, dtype=dtypes.float32, name="block_diagonal"):
  71   r"""Constructs block-diagonal matrices from a list of batched 2D tensors.
  72 
  73   Args:
  74     matrices: A list of Tensors with shape [..., N_i, M_i] (i.e. a list of
  75       matrices with the same batch dimension).
  76     dtype: Data type to use. The Tensors in `matrices` must match this dtype.
  77     name: A name for the returned op.
  78   Returns:
  79     A matrix with the input matrices stacked along its main diagonal, having
  80     shape [..., \sum_i N_i, \sum_i M_i].
  81   """
  82   matrices = [ops.convert_to_tensor(matrix, dtype=dtype) for matrix in matrices]
  83   blocked_rows = tensor_shape.Dimension(0)
  84   blocked_cols = tensor_shape.Dimension(0)
  85   batch_shape = tensor_shape.TensorShape(None)
  86   for matrix in matrices:
  87     full_matrix_shape = matrix.get_shape().with_rank_at_least(2)
  88     batch_shape = batch_shape.merge_with(full_matrix_shape[:-2])
  89     blocked_rows += full_matrix_shape[-2]
  90     blocked_cols += full_matrix_shape[-1]
  91   ret_columns_list = []
  92   for matrix in matrices:
  93     matrix_shape = array_ops.shape(matrix)
  94     ret_columns_list.append(matrix_shape[-1])
  95   ret_columns = math_ops.add_n(ret_columns_list)
  96   row_blocks = []
  97   current_column = 0
  98   for matrix in matrices:
  99     matrix_shape = array_ops.shape(matrix)
 100     row_before_length = current_column
 101     current_column += matrix_shape[-1]
 102     row_after_length = ret_columns - current_column
 103     row_blocks.append(
 104         array_ops.pad(
 105             tensor=matrix,
 106             paddings=array_ops.concat(
 107                 [
 108                     array_ops.zeros(
 109                         [array_ops.rank(matrix) - 1, 2], dtype=dtypes.int32), [(
 110                             row_before_length, row_after_length)]
 111                 ],
 112                 axis=0)))
 113   blocked = array_ops.concat(row_blocks, -2, name=name)
 114   blocked.set_shape(batch_shape.concatenate((blocked_rows, blocked_cols)))
 115   return blocked

 118 def power_sums_tensor(array_size, power_matrix, multiplier):
 119   r"""Computes \sum_{i=0}^{N-1} A^i B (A^i)^T for N=0..(array_size + 1).
 120 
 121   Args:
 122     array_size: The number of non-trivial sums to pre-compute.
 123     power_matrix: The "A" matrix above.
 124     multiplier: The "B" matrix above
 125   Returns:
 126     A Tensor with S[N] = \sum_{i=0}^{N-1} A^i B (A^i)^T
 127       S[0] is the zero matrix
 128       S[1] is B
 129       S[2] is A B A^T + B
 130       ...and so on
 131   """
 132   array_size = math_ops.cast(array_size, dtypes.int32)
 133   power_matrix = ops.convert_to_tensor(power_matrix)
 134   identity_like_power_matrix = linalg_ops.eye(
 135       array_ops.shape(power_matrix)[0], dtype=power_matrix.dtype)
 136   identity_like_power_matrix.set_shape(
 137       ops.convert_to_tensor(power_matrix).get_shape())
 138   transition_powers = functional_ops.scan(
 139       lambda previous_power, _: math_ops.matmul(previous_power, power_matrix),
 140       math_ops.range(array_size - 1),
 141       initializer=identity_like_power_matrix)
 142   summed = math_ops.cumsum(
 143       array_ops.concat([
 144           array_ops.expand_dims(multiplier, 0), math_ops.matmul(
 145               batch_times_matrix(transition_powers, multiplier),
 146               transition_powers,
 147               adjoint_b=True)
 148       ], 0))
 149   return array_ops.concat(
 150       [array_ops.expand_dims(array_ops.zeros_like(multiplier), 0), summed], 0)

 154   """Raise a single matrix to multiple powers."""
 155   matrix_tiled = array_ops.tile(
 156       array_ops.expand_dims(matrix, 0), [array_ops.size(powers), 1, 1])
 157   return batch_matrix_pow(matrix_tiled, powers)

 160 def batch_matrix_pow(matrices, powers):
 161   """Compute powers of matrices, e.g. A^3 = matmul(matmul(A, A), A).
 162 
 163   Uses exponentiation by squaring, with O(log(p)) matrix multiplications to
 164   compute A^p.
 165 
 166   Args:
 167     matrices: [batch size x N x N]
 168     powers: Which integer power to raise each matrix to [batch size]
 169   Returns:
 170     The matrices raised to their respective powers, same dimensions as the
 171     "matrices" argument.
 172   """
 173 
 174   def terminate_when_all_zero(current_argument, residual_powers, accumulator):
 175     del current_argument, accumulator  # not used for condition
 176     do_exit = math_ops.reduce_any(
 177         math_ops.greater(residual_powers, array_ops.ones_like(residual_powers)))
 178     return do_exit
 179 
 180   def do_iteration(current_argument, residual_powers, accumulator):
 181     """Compute one step of iterative exponentiation by squaring.
 182 
 183     The recursive form is:
 184       power(A, p) = { power(matmul(A, A), p / 2) for even p
 185                     { matmul(A, power(matmul(A, A), (p - 1) / 2)) for odd p
 186       power(A, 0) = I
 187 
 188     The power(A, 0) = I case is handled by starting with accumulator set to the
 189     identity matrix; matrices with zero residual powers are passed through
 190     unchanged.
 191 
 192     Args:
 193       current_argument: On this step, what is the first argument (A^2..^2) to
 194           the (unrolled) recursive function? [batch size x N x N]
 195       residual_powers: On this step, what is the second argument (residual p)?
 196           [batch_size]
 197       accumulator: Accumulates the exterior multiplications from the odd
 198           powers (initially the identity matrix). [batch_size x N x N]
 199     Returns:
 200       Updated versions of each argument for one step of the unrolled
 201       computation. Does not change parts of the batch which have a residual
 202       power of zero.
 203     """
 204     is_even = math_ops.equal(residual_powers % 2,
 205                              array_ops.zeros(
 206                                  array_ops.shape(residual_powers),
 207                                  dtype=dtypes.int32))
 208     new_accumulator = array_ops.where(is_even, accumulator,
 209                                       math_ops.matmul(accumulator,
 210                                                       current_argument))
 211     new_argument = math_ops.matmul(current_argument, current_argument)
 212     do_update = math_ops.greater(residual_powers, 1)
 213     new_residual_powers = residual_powers - residual_powers % 2
 214     new_residual_powers //= 2
 215     # Stop updating if we've reached our base case; some batch elements may
 216     # finish sooner than others
 217     accumulator = array_ops.where(do_update, new_accumulator, accumulator)
 218     current_argument = array_ops.where(do_update, new_argument,
 219                                        current_argument)
 220     residual_powers = array_ops.where(do_update, new_residual_powers,
 221                                       residual_powers)
 222     return (current_argument, residual_powers, accumulator)
 223 
 224   matrices = ops.convert_to_tensor(matrices)
 225   powers = math_ops.cast(powers, dtype=dtypes.int32)
 226   ident = array_ops.expand_dims(
 227       array_ops.diag(
 228           array_ops.ones([array_ops.shape(matrices)[1]], dtype=matrices.dtype)),
 229       0)
 230   ident_tiled = array_ops.tile(ident, [array_ops.shape(matrices)[0], 1, 1])
 231   (final_argument,
 232    final_residual_power, final_accumulator) = control_flow_ops.while_loop(
 233        terminate_when_all_zero, do_iteration, [matrices, powers, ident_tiled])
 234   return array_ops.where(
 235       math_ops.equal(final_residual_power,
 236                      array_ops.zeros_like(
 237                          final_residual_power, dtype=dtypes.int32)),
 238       ident_tiled, math_ops.matmul(final_argument, final_accumulator))

 242 def batch_times_matrix(batch, matrix, adj_x=False, adj_y=False):
 243   """Multiply a batch of matrices by a single matrix.
 244 
 245   Functionally equivalent to:
 246   tf.matmul(batch, array_ops.tile(gen_math_ops.expand_dims(matrix, 0),
 247                                  [array_ops.shape(batch)[0], 1, 1]),
 248                   adjoint_a=adj_x, adjoint_b=adj_y)
 249 
 250   Args:
 251     batch: [batch_size x N x M] after optional transpose
 252     matrix: [M x P] after optional transpose
 253     adj_x: If true, transpose the second two dimensions of "batch" before
 254         multiplying.
 255     adj_y: If true, transpose "matrix" before multiplying.
 256   Returns:
 257     [batch_size x N x P]
 258   """
 259   batch = ops.convert_to_tensor(batch)
 260   matrix = ops.convert_to_tensor(matrix)
 261   assert batch.get_shape().ndims == 3
 262   assert matrix.get_shape().ndims == 2
 263   if adj_x:
 264     batch = array_ops.transpose(batch, [0, 2, 1])
 265   batch_dimension = batch.get_shape()[0].value
 266   first_dimension = batch.get_shape()[1].value
 267   tensor_batch_shape = array_ops.shape(batch)
 268   if batch_dimension is None:
 269     batch_dimension = tensor_batch_shape[0]
 270   if first_dimension is None:
 271     first_dimension = tensor_batch_shape[1]
 272   matrix_first_dimension, matrix_second_dimension = matrix.get_shape().as_list()
 273   batch_reshaped = array_ops.reshape(batch, [-1, tensor_batch_shape[2]])
 274   if adj_y:
 275     if matrix_first_dimension is None:
 276       matrix_first_dimension = array_ops.shape(matrix)[0]
 277     result_shape = [batch_dimension, first_dimension, matrix_first_dimension]
 278   else:
 279     if matrix_second_dimension is None:
 280       matrix_second_dimension = array_ops.shape(matrix)[1]
 281     result_shape = [batch_dimension, first_dimension, matrix_second_dimension]
 282   return array_ops.reshape(
 283       math_ops.matmul(batch_reshaped, matrix, adjoint_b=adj_y), result_shape)

 286 def matrix_times_batch(matrix, batch, adj_x=False, adj_y=False):

 289       batch_times_matrix(
 290           batch=batch, matrix=matrix, adj_x=not adj_y, adj_y=not adj_x),
 291       [0, 2, 1])

 294 def make_toeplitz_matrix(inputs, name=None):
 295   """Make a symmetric Toeplitz matrix from input array of values.
 296 
 297   Args:
 298     inputs: a 3-D tensor of shape [num_blocks, block_size, block_size].
 299     name: the name of the operation.
 300 
 301   Returns:
 302     a symmetric Toeplitz matrix of shape
 303       [num_blocks*block_size, num_blocks*block_size].
 304   """
 305   num_blocks = array_ops.shape(inputs)[0]
 306   block_size = array_ops.shape(inputs)[1]
 307   output_size = block_size * num_blocks
 308   lags = array_ops.reshape(math_ops.range(num_blocks), shape=[1, -1])
 309   indices = math_ops.abs(lags - array_ops.transpose(lags))
 310   output = array_ops.gather(inputs, indices)
 311   output = array_ops.reshape(
 312       array_ops.transpose(output, [0, 2, 1, 3]), [output_size, output_size])
 313   return array_ops.identity(output, name=name)

 317 def sign_magnitude_positive_definite(
 318     raw, off_diagonal_scale=0., overall_scale=0.):
 319   """Constructs a positive definite matrix from an unconstrained input matrix.
 320 
 321   We want to keep the whole matrix on a log scale, but also allow off-diagonal
 322   elements to be negative, so the sign of off-diagonal elements is modeled
 323   separately from their magnitude (using the lower and upper triangles
 324   respectively). Specifically:
 325 
 326   for i < j, we have:
 327     output_cholesky[i, j] = raw[j, i] / (abs(raw[j, i]) + 1) *
 328         exp((off_diagonal_scale + overall_scale + raw[i, j]) / 2)
 329 
 330   output_cholesky[i, i] = exp((raw[i, i] + overall_scale) / 2)
 331 
 332   output = output_cholesky^T * output_cholesky
 333 
 334   where raw, off_diagonal_scale, and overall_scale are
 335   un-constrained real-valued variables. The resulting values are stable
 336   around zero due to the exponential (and the softsign keeps the function
 337   smooth).
 338 
 339   Args:
 340     raw: A [..., M, M] Tensor.
 341     off_diagonal_scale: A scalar or [...] shaped Tensor controlling the relative
 342         scale of off-diagonal values in the output matrix.
 343     overall_scale: A scalar or [...] shaped Tensor controlling the overall scale
 344         of the output matrix.
 345   Returns:
 346     The `output` matrix described above, a [..., M, M] positive definite matrix.
 347 
 348   """
 349   raw = ops.convert_to_tensor(raw)
 350   diagonal = array_ops.matrix_diag_part(raw)
 351   def _right_pad_with_ones(tensor, target_rank):
 352     # Allow broadcasting even if overall_scale and off_diagonal_scale have batch
 353     # dimensions
 354     tensor = ops.convert_to_tensor(tensor, dtype=raw.dtype.base_dtype)
 355     return array_ops.reshape(tensor,
 356                              array_ops.concat(
 357                                  [
 358                                      array_ops.shape(tensor), array_ops.ones(
 359                                          [target_rank - array_ops.rank(tensor)],
 360                                          dtype=target_rank.dtype)
 361                                  ],
 362                                  axis=0))
 363   # We divide the log values by 2 to compensate for the squaring that happens
 364   # when transforming Cholesky factors into positive definite matrices.
 365   sign_magnitude = (gen_math_ops.exp(
 366       (raw + _right_pad_with_ones(off_diagonal_scale, array_ops.rank(raw)) +
 367        _right_pad_with_ones(overall_scale, array_ops.rank(raw))) / 2.) *
 368                     nn.softsign(array_ops.matrix_transpose(raw)))
 369   sign_magnitude.set_shape(raw.get_shape())
 370   cholesky_factor = array_ops.matrix_set_diag(
 371       input=array_ops.matrix_band_part(sign_magnitude, 0, -1),
 372       diagonal=gen_math_ops.exp((diagonal + _right_pad_with_ones(
 373           overall_scale, array_ops.rank(diagonal))) / 2.))
 374   return math_ops.matmul(cholesky_factor, cholesky_factor, transpose_a=True)

 377 def transform_to_covariance_matrices(input_vectors, matrix_size):
 378   """Construct covariance matrices via transformations from input_vectors.
 379 
 380   Args:
 381     input_vectors: A [batch size x input size] batch of vectors to transform.
 382     matrix_size: An integer indicating one dimension of the (square) output
 383         matrix.
 384   Returns:
 385     A [batch size x matrix_size x matrix_size] batch of covariance matrices.
 386   """
 387   combined_values = layers.fully_connected(
 388       input_vectors, matrix_size**2 + 2, activation_fn=None)
 389   return sign_magnitude_positive_definite(
 390       raw=array_ops.reshape(combined_values[..., :-2],
 391                             array_ops.concat([
 392                                 array_ops.shape(combined_values)[:-1],
 393                                 [matrix_size, matrix_size]
 394                             ], 0)),
 395       off_diagonal_scale=combined_values[..., -2],
 396       overall_scale=combined_values[..., -1])

 399 def variable_covariance_matrix(
 400     size, name, dtype, initial_diagonal_values=None,
 401     initial_overall_scale_log=0.):
 402   """Construct a Variable-parameterized positive definite matrix.
 403 
 404   Useful for parameterizing covariance matrices.
 405 
 406   Args:
 407     size: The size of the main diagonal, the returned matrix having shape [size
 408         x size].
 409     name: The name to use when defining variables and ops.
 410     dtype: The floating point data type to use.
 411     initial_diagonal_values: A Tensor with shape [size] with initial values for
 412         the diagonal values of the returned matrix. Must be positive.
 413     initial_overall_scale_log: Initial value of the bias term for every element
 414         of the matrix in log space.
 415   Returns:
 416     A Variable-parameterized covariance matrix with shape [size x size].
 417   """
 418   raw_values = variable_scope.get_variable(
 419       name + "_pre_transform",
 420       dtype=dtype,
 421       shape=[size, size],
 422       initializer=init_ops.zeros_initializer())
 423   if initial_diagonal_values is not None:
 424     raw_values += array_ops.matrix_diag(math_ops.log(initial_diagonal_values))
 425   return array_ops.identity(
 426       sign_magnitude_positive_definite(
 427           raw=raw_values,
 428           off_diagonal_scale=variable_scope.get_variable(
 429               name + "_off_diagonal_scale",
 430               dtype=dtype,
 431               initializer=constant_op.constant(-5., dtype=dtype)),
 432           overall_scale=ops.convert_to_tensor(
 433               initial_overall_scale_log, dtype=dtype) +
 434           variable_scope.get_variable(
 435               name + "_overall_scale",
 436               dtype=dtype,
 437               shape=[],
 438               initializer=init_ops.zeros_initializer())),
 439       name=name)

 450 def log_noninformative_covariance_prior(covariance):
 451   """Compute a relatively uninformative prior for noise parameters.
 452 
 453   Helpful for avoiding noise over-estimation, where noise otherwise decreases
 454   very slowly during optimization.
 455 
 456   See:
 457     Villegas, C. On the A Priori Distribution of the Covariance Matrix.
 458     Ann. Math. Statist. 40 (1969), no. 3, 1098--1099.
 459 
 460   Args:
 461     covariance: A covariance matrix.
 462   Returns:
 463     For a [p x p] matrix:
 464       log(det(covariance)^(-(p + 1) / 2))
 465   """
 466   # Avoid zero/negative determinants due to numerical errors
 467   covariance += array_ops.diag(1e-8 * array_ops.ones(
 468       shape=[array_ops.shape(covariance)[0]], dtype=covariance.dtype))
 469   power = -(math_ops.cast(array_ops.shape(covariance)[0] + 1,
 470                           covariance.dtype) / 2.)
 471   return power * math_ops.log(linalg_ops.matrix_determinant(covariance))

 474 def entropy_matched_cauchy_scale(covariance):
 475   """Approximates a similar Cauchy distribution given a covariance matrix.
 476 
 477   Since Cauchy distributions do not have moments, entropy matching provides one
 478   way to set a Cauchy's scale parameter in a way that provides a similar
 479   distribution. The effect is dividing the standard deviation of an independent
 480   Gaussian by a constant very near 3.
 481 
 482   To set the scale of the Cauchy distribution, we first select the diagonals of
 483   `covariance`. Since this ignores cross terms, it overestimates the entropy of
 484   the Gaussian. For each of these variances, we solve for the Cauchy scale
 485   parameter which gives the same entropy as the Gaussian with that
 486   variance. This means setting the (univariate) Gaussian entropy
 487       0.5 * ln(2 * variance * pi * e)
 488   equal to the Cauchy entropy
 489       ln(4 * pi * scale)
 490   Solving, we get scale = sqrt(variance * (e / (8 pi))).
 491 
 492   Args:
 493     covariance: A [batch size x N x N] batch of covariance matrices to produce
 494         Cauchy scales for.
 495   Returns:
 496     A [batch size x N] set of Cauchy scale parameters for each part of the batch
 497     and each dimension of the input Gaussians.
 498   """
 499   return math_ops.sqrt(math.e / (8. * math.pi) *
 500                        array_ops.matrix_diag_part(covariance))

 503 class TensorValuedMutableDenseHashTable(lookup.MutableDenseHashTable):
 504   """A version of MutableDenseHashTable which stores arbitrary Tensor shapes.
 505 
 506   Since MutableDenseHashTable only allows vectors right now, simply adds reshape
 507   ops on both ends.
 508   """
 509 
 510   def __init__(self, key_dtype, value_dtype, default_value, *args, **kwargs):
 511     self._non_vector_value_shape = array_ops.shape(default_value)
 512     super(TensorValuedMutableDenseHashTable, self).__init__(
 513         key_dtype=key_dtype,
 514         value_dtype=value_dtype,
 515         default_value=array_ops.reshape(default_value, [-1]),
 516         *args,
 517         **kwargs)
 518 
 519   def insert(self, keys, values, name=None):
 520     keys = ops.convert_to_tensor(keys, dtype=self._key_dtype)
 521     keys_flat = array_ops.reshape(keys, [-1])
 522     return super(TensorValuedMutableDenseHashTable, self).insert(
 523         keys=keys_flat,
 524         # Each key has one corresponding value, so the shape of the tensor of
 525         # values for every key is key_shape + value_shape
 526         values=array_ops.reshape(values, [array_ops.shape(keys_flat)[0], -1]),
 527         name=name)
 528 
 529   def lookup(self, keys, name=None):
 530     keys_flat = array_ops.reshape(
 531         ops.convert_to_tensor(keys, dtype=self._key_dtype), [-1])
 532     return array_ops.reshape(
 533         super(TensorValuedMutableDenseHashTable, self).lookup(
 534             keys=keys_flat, name=name),
 535         array_ops.concat([array_ops.shape(keys), self._non_vector_value_shape],
 536                          0))

 539 class TupleOfTensorsLookup(lookup.LookupInterface):
 540   """A LookupInterface with nested tuples of Tensors as values.
 541 
 542   Creates one MutableDenseHashTable per value Tensor, which has some unnecessary
 543   overhead.
 544   """
 545 
 546   def __init__(
 547       self, key_dtype, default_values, empty_key, name, checkpoint=True):
 548     default_values_flat = nest.flatten(default_values)
 549     self._hash_tables = nest.pack_sequence_as(
 550         default_values,
 551         [TensorValuedMutableDenseHashTable(
 552             key_dtype=key_dtype,
 553             value_dtype=default_value.dtype.base_dtype,
 554             default_value=default_value,
 555             empty_key=empty_key,
 556             name=name + "_{}".format(table_number),
 557             checkpoint=checkpoint)
 558          for table_number, default_value
 559          in enumerate(default_values_flat)])
 560     self._name = name
 561 
 562   def lookup(self, keys):
 563     return nest.pack_sequence_as(
 564         self._hash_tables,
 565         [hash_table.lookup(keys)
 566          for hash_table in nest.flatten(self._hash_tables)])
 567 
 568   def insert(self, keys, values):
 569     nest.assert_same_structure(self._hash_tables, values)
 570     # Avoid race conditions by requiring that all inputs are computed before any
 571     # inserts happen (an issue if one key's update relies on another's value).
 572     values_flat = [array_ops.identity(value) for value in nest.flatten(values)]
 573     with ops.control_dependencies(values_flat):
 574       insert_ops = [hash_table.insert(keys, value)
 575                     for hash_table, value
 576                     in zip(nest.flatten(self._hash_tables),
 577                            values_flat)]
 578     return control_flow_ops.group(*insert_ops)

 580   def check_table_dtypes(self, key_dtype, value_dtype):
 581     # dtype checking is done in the objects in self._hash_tables
 582     pass

 585 def replicate_state(start_state, batch_size):
 586   """Create batch versions of state.
 587 
 588   Takes a list of Tensors, adds a batch dimension, and replicates
 589   batch_size times across that batch dimension. Used to replicate the
 590   non-batch state returned by get_start_state in define_loss.
 591 
 592   Args:
 593     start_state: Model-defined state to replicate.
 594     batch_size: Batch dimension for data.
 595   Returns:
 596     Replicated versions of the state.
 597   """
 598   flattened_state = nest.flatten(start_state)
 599   replicated_state = [
 600       array_ops.tile(
 601           array_ops.expand_dims(state_nonbatch, 0),
 602           array_ops.concat([[batch_size], array_ops.ones(
 603               [array_ops.rank(state_nonbatch)], dtype=dtypes.int32)], 0))
 604       for state_nonbatch in flattened_state
 605   ]
 606   return nest.pack_sequence_as(start_state, replicated_state)

 612 # Currently all of these statistics are computed incrementally (i.e. are updated
 613 # every time a new mini-batch of training data is presented) when this object is
 614 # created in InputStatisticsFromMiniBatch.
 615 InputStatistics = collections.namedtuple(
 616     "InputStatistics",
 617     ["series_start_moments",  # The mean and variance of each feature in a chunk
 618                               # (with a size configured in the statistics
 619                               # object) at the start of the series. A tuple of
 620                               # (mean, variance), each with shape [number of
 621                               # features], floating point. One use is in state
 622                               # space models, to keep priors calibrated even as
 623                               # earlier parts of the series are presented. If
 624                               # this object was created by
 625                               # InputStatisticsFromMiniBatch, these moments are
 626                               # computed based on the earliest chunk of data
 627                               # presented so far. However, there is a race
 628                               # condition in the update, so these may reflect
 629                               # statistics later in the series, but should
 630                               # eventually reflect statistics in a chunk at the
 631                               # series start.
 632      "overall_feature_moments",  # The mean and variance of each feature over
 633                                  # the entire series. A tuple of (mean,
 634                                  # variance), each with shape [number of
 635                                  # features]. If this object was created by
 636                                  # InputStatisticsFromMiniBatch, these moments
 637                                  # are estimates based on the data seen so far.
 638      "start_time",  # The first (lowest) time in the series, a scalar
 639                     # integer. If this object was created by
 640                     # InputStatisticsFromMiniBatch, this is the lowest time seen
 641                     # so far rather than the lowest time that will ever be seen
 642                     # (guaranteed to be at least as low as the lowest time
 643                     # presented in the current minibatch).
 644      "total_observation_count",  # Count of data points, a scalar integer. If
 645                                  # this object was created by
 646                                  # InputStatisticsFromMiniBatch, this is an
 647                                  # estimate of the total number of observations
 648                                  # in the whole dataset computed based on the
 649                                  # density of the series and the minimum and
 650                                  # maximum times seen.
 651     ])

 654 # TODO(allenl): It would be nice to do something with full series statistics
 655 # when the user provides that.
 656 class InputStatisticsFromMiniBatch(object):
 657   """Generate statistics from mini-batch input."""
 658 
 659   def __init__(self, num_features, dtype, starting_variance_window_size=16):
 660     """Configure the input statistics object.
 661 
 662     Args:
 663       num_features: Number of features for the time series
 664       dtype: The floating point data type to use.
 665       starting_variance_window_size: The number of datapoints to use when
 666           computing the mean and variance at the start of the series.
 667     """
 668     self._starting_variance_window_size = starting_variance_window_size
 669     self._num_features = num_features
 670     self._dtype = dtype
 671 
 672   def initialize_graph(self, features, update_statistics=True):
 673     """Create any ops needed to provide input statistics.
 674 
 675     Should be called before statistics are requested.
 676 
 677     Args:
 678       features: A dictionary, the output of a `TimeSeriesInputFn` (with keys
 679           TrainEvalFeatures.TIMES and TrainEvalFeatures.VALUES).
 680       update_statistics: Whether `features` should be used to update adaptive
 681           statistics. Typically True for training and false for evaluation.
 682     Returns:
 683       An InputStatistics object composed of Variables, which will be updated
 684       based on mini-batches of data if requested.
 685     """
 686     if (TrainEvalFeatures.TIMES in features
 687         and TrainEvalFeatures.VALUES in features):
 688       times = features[TrainEvalFeatures.TIMES]
 689       values = features[TrainEvalFeatures.VALUES]
 690     else:
 691       # times and values may not be available, for example during prediction. We
 692       # still need to retrieve our variables so that they can be read from, even
 693       # if we're not going to update them.
 694       times = None
 695       values = None
 696     # Create/retrieve variables representing input statistics, initialized
 697     # without data to avoid deadlocking if variables are initialized before
 698     # queue runners are started.
 699     with variable_scope.variable_scope("input_statistics", use_resource=True):
 700       statistics = self._create_variable_statistics_object()
 701     with variable_scope.variable_scope(
 702         "input_statistics_auxiliary", use_resource=True):
 703       # Secondary statistics, necessary for the incremental computation of the
 704       # primary statistics (e.g. counts and sums for computing a mean
 705       # incrementally).
 706       auxiliary_variables = self._AdaptiveInputAuxiliaryStatistics(
 707           num_features=self._num_features, dtype=self._dtype)
 708     if update_statistics and times is not None and values is not None:
 709       # If we have times and values from mini-batch input, create update ops to
 710       # take the new data into account.
 711       assign_op = self._update_statistics_from_mini_batch(
 712           statistics, auxiliary_variables, times, values)
 713       with ops.control_dependencies([assign_op]):
 714         stat_variables = nest.pack_sequence_as(statistics, [
 715             array_ops.identity(tensor) for tensor in nest.flatten(statistics)
 716         ])
 717         # Since start time updates have a race condition, ensure that the
 718         # reported start time is at least as low as the lowest time in this
 719         # mini-batch. The start time should converge on the correct value
 720         # eventually even with the race condition, but for example state space
 721         # models have an assertion which could fail without this
 722         # post-processing.
 723         return stat_variables._replace(start_time=gen_math_ops.minimum(
 724             stat_variables.start_time, math_ops.reduce_min(times)))
 725     else:
 726       return statistics

 728   class _AdaptiveInputAuxiliaryStatistics(collections.namedtuple(
 729       "_AdaptiveInputAuxiliaryStatistics",
 730       ["max_time_seen",  # The maximum time seen (best effort if updated from
 731                          # multiple workers; see notes about race condition
 732                          # below).
 733        "chunk_count",  # The number of chunks seen.
 734        "inter_observation_duration_sum",  # The sum across chunks of their "time
 735                                           # density" (number of times per
 736                                           # example).
 737        "example_count",  # The number of examples seen (each example has a
 738                          # single time associated with it and one or more
 739                          # real-valued features).
 740        "overall_feature_sum",  # The sum of values for each feature. Shape
 741                                # [number of features].
 742        "overall_feature_sum_of_squares",  # The sum of squared values for each
 743                                           # feature. Shape [number of features]
 744       ])):
 745     """Extra statistics used to incrementally update InputStatistics."""
 746 
 747     def __new__(cls, num_features, dtype):
 748       return super(
 749           InputStatisticsFromMiniBatch  # pylint: disable=protected-access
 750           ._AdaptiveInputAuxiliaryStatistics,
 751           cls).__new__(
 752               cls,
 753               max_time_seen=variable_scope.get_variable(
 754                   name="max_time_seen",
 755                   initializer=dtypes.int64.min,
 756                   dtype=dtypes.int64,
 757                   trainable=False),
 758               chunk_count=variable_scope.get_variable(
 759                   name="chunk_count",
 760                   initializer=init_ops.zeros_initializer(),
 761                   shape=[],
 762                   dtype=dtypes.int64,
 763                   trainable=False),
 764               inter_observation_duration_sum=variable_scope.get_variable(
 765                   name="inter_observation_duration_sum",
 766                   initializer=init_ops.zeros_initializer(),
 767                   shape=[],
 768                   dtype=dtype,
 769                   trainable=False),
 770               example_count=variable_scope.get_variable(
 771                   name="example_count",
 772                   shape=[],
 773                   dtype=dtypes.int64,
 774                   trainable=False),
 775               overall_feature_sum=variable_scope.get_variable(
 776                   name="overall_feature_sum",
 777                   shape=[num_features],
 778                   dtype=dtype,
 779                   initializer=init_ops.zeros_initializer(),
 780                   trainable=False),
 781               overall_feature_sum_of_squares=variable_scope.get_variable(
 782                   name="overall_feature_sum_of_squares",
 783                   shape=[num_features],
 784                   dtype=dtype,
 785                   initializer=init_ops.zeros_initializer(),
 786                   trainable=False))

 788   def _update_statistics_from_mini_batch(
 789       self, statistics, auxiliary_variables, times, values):
 790     """Given mini-batch input, update `statistics` and `auxiliary_variables`."""
 791     values = math_ops.cast(values, self._dtype)
 792     # The density (measured in times per observation) that we see in each part
 793     # of the mini-batch.
 794     batch_inter_observation_duration = (math_ops.cast(
 795         math_ops.reduce_max(times, axis=1) - math_ops.reduce_min(times, axis=1),
 796         self._dtype) / math_ops.cast(
 797             array_ops.shape(times)[1] - 1, self._dtype))
 798     # Co-locate updates with their variables to minimize race conditions when
 799     # updating statistics.
 800     with ops.colocate_with(auxiliary_variables.max_time_seen):
 801       # There is a race condition if this value is being updated from multiple
 802       # workers. However, it should eventually reach the correct value if the
 803       # last chunk is presented enough times.
 804       max_time_seen_assign = state_ops.assign(
 805           auxiliary_variables.max_time_seen,
 806           gen_math_ops.maximum(auxiliary_variables.max_time_seen,
 807                                math_ops.reduce_max(times)))
 808     with ops.colocate_with(auxiliary_variables.chunk_count):
 809       chunk_count_assign = state_ops.assign_add(auxiliary_variables.chunk_count,
 810                                                 array_ops.shape(
 811                                                     times,
 812                                                     out_type=dtypes.int64)[0])
 813     with ops.colocate_with(auxiliary_variables.inter_observation_duration_sum):
 814       inter_observation_duration_assign = state_ops.assign_add(
 815           auxiliary_variables.inter_observation_duration_sum,
 816           math_ops.reduce_sum(batch_inter_observation_duration))
 817     with ops.colocate_with(auxiliary_variables.example_count):
 818       example_count_assign = state_ops.assign_add(
 819           auxiliary_variables.example_count,
 820           array_ops.size(times, out_type=dtypes.int64))
 821     # Note: These mean/variance updates assume that all points are equally
 822     # likely, which is not true if _chunks_ are sampled uniformly from the space
 823     # of all possible contiguous chunks, since points at the start and end of
 824     # the series are then members of fewer chunks. For series which are much
 825     # longer than the chunk size (the usual/expected case), this effect becomes
 826     # irrelevant.
 827     with ops.colocate_with(auxiliary_variables.overall_feature_sum):
 828       overall_feature_sum_assign = state_ops.assign_add(
 829           auxiliary_variables.overall_feature_sum,
 830           math_ops.reduce_sum(values, axis=[0, 1]))
 831     with ops.colocate_with(auxiliary_variables.overall_feature_sum_of_squares):
 832       overall_feature_sum_of_squares_assign = state_ops.assign_add(
 833           auxiliary_variables.overall_feature_sum_of_squares,
 834           math_ops.reduce_sum(values**2, axis=[0, 1]))
 835     per_chunk_aux_updates = control_flow_ops.group(
 836         max_time_seen_assign, chunk_count_assign,
 837         inter_observation_duration_assign, example_count_assign,
 838         overall_feature_sum_assign, overall_feature_sum_of_squares_assign)
 839     with ops.control_dependencies([per_chunk_aux_updates]):
 840       example_count_float = math_ops.cast(auxiliary_variables.example_count,
 841                                           self._dtype)
 842       new_feature_mean = (auxiliary_variables.overall_feature_sum /
 843                           example_count_float)
 844       overall_feature_mean_update = state_ops.assign(
 845           statistics.overall_feature_moments.mean, new_feature_mean)
 846       overall_feature_var_update = state_ops.assign(
 847           statistics.overall_feature_moments.variance,
 848           # De-biased n / (n - 1) variance correction
 849           example_count_float / (example_count_float - 1.) *
 850           (auxiliary_variables.overall_feature_sum_of_squares /
 851            example_count_float - new_feature_mean**2))
 852       # TODO(b/35675805): Remove this cast
 853       min_time_batch = math_ops.cast(math_ops.argmin(times[:, 0]), dtypes.int32)
 854       def series_start_updates():
 855         # If this is the lowest-time chunk that we have seen so far, update
 856         # series start moments to reflect that. Note that these statistics are
 857         # "best effort", as there are race conditions in the update (however,
 858         # they should eventually converge if the start of the series is
 859         # presented enough times).
 860         mean, variance = nn.moments(
 861             values[min_time_batch, :self._starting_variance_window_size],
 862             axes=[0])
 863         return control_flow_ops.group(
 864             state_ops.assign(statistics.series_start_moments.mean, mean),
 865             state_ops.assign(statistics.series_start_moments.variance,
 866                              variance))
 867       with ops.colocate_with(statistics.start_time):
 868         series_start_update = control_flow_ops.cond(
 869             # Update moments whenever we even match the lowest time seen so far,
 870             # to ensure that series start statistics are eventually updated to
 871             # their correct values, despite race conditions (i.e. eventually
 872             # statistics.start_time will reflect the global lowest time, and
 873             # given that we will eventually update the series start moments to
 874             # their correct values).
 875             math_ops.less_equal(times[min_time_batch, 0],
 876                                 statistics.start_time),
 877             series_start_updates,
 878             control_flow_ops.no_op)
 879         with ops.control_dependencies([series_start_update]):
 880           # There is a race condition if this update is performed in parallel on
 881           # multiple workers. Since models may be sensitive to being presented
 882           # with times before the putative start time, the value of this
 883           # variable is post-processed above to guarantee that each worker is
 884           # presented with a start time which is at least as low as the lowest
 885           # time in its current mini-batch.
 886           start_time_update = state_ops.assign(statistics.start_time,
 887                                                gen_math_ops.minimum(
 888                                                    statistics.start_time,
 889                                                    math_ops.reduce_min(times)))
 890       inter_observation_duration_estimate = (
 891           auxiliary_variables.inter_observation_duration_sum / math_ops.cast(
 892               auxiliary_variables.chunk_count, self._dtype))
 893       # Estimate the total number of observations as:
 894       #   (end time - start time + 1) * average intra-chunk time density
 895       total_observation_count_update = state_ops.assign(
 896           statistics.total_observation_count,
 897           math_ops.cast(
 898               gen_math_ops.round(
 899                   math_ops.cast(max_time_seen_assign -
 900                                 start_time_update + 1, self._dtype) /
 901                   inter_observation_duration_estimate), dtypes.int64))
 902       per_chunk_stat_updates = control_flow_ops.group(
 903           overall_feature_mean_update, overall_feature_var_update,
 904           series_start_update, start_time_update,
 905           total_observation_count_update)
 906     return per_chunk_stat_updates

 908   def _create_variable_statistics_object(self):
 909     """Creates non-trainable variables representing input statistics."""
 910     series_start_moments = Moments(
 911         mean=variable_scope.get_variable(
 912             name="series_start_mean",
 913             shape=[self._num_features],
 914             dtype=self._dtype,
 915             initializer=init_ops.zeros_initializer(),
 916             trainable=False),
 917         variance=variable_scope.get_variable(
 918             name="series_start_variance",
 919             shape=[self._num_features],
 920             dtype=self._dtype,
 921             initializer=init_ops.ones_initializer(),
 922             trainable=False))
 923     overall_feature_moments = Moments(
 924         mean=variable_scope.get_variable(
 925             name="overall_feature_mean",
 926             shape=[self._num_features],
 927             dtype=self._dtype,
 928             initializer=init_ops.zeros_initializer(),
 929             trainable=False),
 930         variance=variable_scope.get_variable(
 931             name="overall_feature_var",
 932             shape=[self._num_features],
 933             dtype=self._dtype,
 934             initializer=init_ops.ones_initializer(),
 935             trainable=False))
 936     start_time = variable_scope.get_variable(
 937         name="start_time",
 938         dtype=dtypes.int64,
 939         initializer=dtypes.int64.max,
 940         trainable=False)
 941     total_observation_count = variable_scope.get_variable(
 942         name="total_observation_count",
 943         shape=[],
 944         dtype=dtypes.int64,
 945         initializer=init_ops.ones_initializer(),
 946         trainable=False)
 947     return InputStatistics(
 948         series_start_moments=series_start_moments,
 949         overall_feature_moments=overall_feature_moments,
 950         start_time=start_time,
 951         total_observation_count=total_observation_count)
