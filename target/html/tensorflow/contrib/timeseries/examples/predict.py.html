  48 def structural_ensemble_train_and_predict(csv_file_name):
  49   # Cycle between 5 latent values over a period of 100. This leads to a very
  50   # smooth periodic component (and a small model), which is a good fit for our
  51   # example data. Modeling high-frequency periodic variations will require a
  52   # higher cycle_num_latent_values.
  53   structural = tf.contrib.timeseries.StructuralEnsembleRegressor(
  54       periodicities=100, num_features=1, cycle_num_latent_values=5)
  55   return train_and_predict(structural, csv_file_name, training_steps=150)

  58 def ar_train_and_predict(csv_file_name):
  59   # An autoregressive model, with periodicity handled as a time-based
  60   # regression. Note that this requires windows of size 16 (input_window_size +
  61   # output_window_size) for training.
  62   ar = tf.contrib.timeseries.ARRegressor(
  63       periodicities=100, input_window_size=10, output_window_size=6,
  64       num_features=1,
  65       # Use the (default) normal likelihood loss to adaptively fit the
  66       # variance. SQUARED_LOSS overestimates variance when there are trends in
  67       # the series.
  68       loss=tf.contrib.timeseries.ARModel.NORMAL_LIKELIHOOD_LOSS)
  69   return train_and_predict(ar, csv_file_name, training_steps=600)

  72 def train_and_predict(estimator, csv_file_name, training_steps):
  73   """A simple example of training and predicting."""
  74   # Read data in the default "time,value" CSV format with no header
  75   reader = tf.contrib.timeseries.CSVReader(csv_file_name)
  76   # Set up windowing and batching for training
  77   train_input_fn = tf.contrib.timeseries.RandomWindowInputFn(
  78       reader, batch_size=16, window_size=16)
  79   # Fit model parameters to data
  80   estimator.train(input_fn=train_input_fn, steps=training_steps)
  81   # Evaluate on the full dataset sequentially, collecting in-sample predictions
  82   # for a qualitative evaluation. Note that this loads the whole dataset into
  83   # memory. For quantitative evaluation, use RandomWindowChunker.
  84   evaluation_input_fn = tf.contrib.timeseries.WholeDatasetInputFn(reader)
  85   evaluation = estimator.evaluate(input_fn=evaluation_input_fn, steps=1)
  86   # Predict starting after the evaluation
  87   (predictions,) = tuple(estimator.predict(
  88       input_fn=tf.contrib.timeseries.predict_continuation_input_fn(
  89           evaluation, steps=200)))
  90   times = evaluation["times"][0]
  91   observed = evaluation["observed"][0, :, 0]
  92   mean = np.squeeze(np.concatenate(
  93       [evaluation["mean"][0], predictions["mean"]], axis=0))
  94   variance = np.squeeze(np.concatenate(
  95       [evaluation["covariance"][0], predictions["covariance"]], axis=0))
  96   all_times = np.concatenate([times, predictions["times"]], axis=0)
  97   upper_limit = mean + np.sqrt(variance)
  98   lower_limit = mean - np.sqrt(variance)
  99   return times, observed, all_times, mean, upper_limit, lower_limit

 102 def make_plot(name, training_times, observed, all_times, mean,
 103               upper_limit, lower_limit):
 104   """Plot a time series in a new figure."""
 105   pyplot.figure()
 106   pyplot.plot(training_times, observed, "b", label="training series")
 107   pyplot.plot(all_times, mean, "r", label="forecast")
 108   pyplot.plot(all_times, upper_limit, "g", label="forecast upper bound")
 109   pyplot.plot(all_times, lower_limit, "g", label="forecast lower bound")
 110   pyplot.fill_between(all_times, lower_limit, upper_limit, color="grey",
 111                       alpha="0.2")
 112   pyplot.axvline(training_times[-1], color="k", linestyle="--")
 113   pyplot.xlabel("time")
 114   pyplot.ylabel("observations")
 115   pyplot.legend(loc=0)
 116   pyplot.title(name)

 119 def main(unused_argv):
 120   if not HAS_MATPLOTLIB:
 121     raise ImportError(
 122         "Please install matplotlib to generate a plot from this example.")
 123   input_filename = FLAGS.input_filename
 124   if input_filename is None:
 125     input_filename = _DEFAULT_DATA_FILE
 126   make_plot("Structural ensemble",
 127             *structural_ensemble_train_and_predict(input_filename))
 128   make_plot("AR", *ar_train_and_predict(input_filename))
 129   pyplot.show()
