  15 """## Functions for copying elements from one graph to another.
  16 
  17 These functions allow for recursive copying of elements (ops and variables)
  18 from one graph to another. The copied elements are initialized inside a
  19 user-specified scope in the other graph. There are separate functions to
  20 copy ops and variables.
  21 There is also a function to retrieve the copied version of an op from the
  22 first graph inside a scope in the second graph.
  23 
  24 @@copy_op_to_graph
  25 @@copy_variable_to_graph
  26 @@get_copied_op
  27 """

  41 def copy_variable_to_graph(org_instance, to_graph, scope=''):
  42   """Given a `Variable` instance from one `Graph`, initializes and returns
  43   a copy of it from another `Graph`, under the specified scope
  44   (default `""`).
  45 
  46   Args:
  47     org_instance: A `Variable` from some `Graph`.
  48     to_graph: The `Graph` to copy the `Variable` to.
  49     scope: A scope for the new `Variable` (default `""`).
  50 
  51   Returns:
  52     The copied `Variable` from `to_graph`.
  53 
  54   Raises:
  55     TypeError: If `org_instance` is not a `Variable`.
  56   """
  57 
  58   if not isinstance(org_instance, VariableV1):
  59     raise TypeError(str(org_instance) + ' is not a Variable')
  60 
  61   #The name of the new variable
  62   if scope != '':
  63     new_name = (scope + '/' + org_instance.name[:org_instance.name.index(':')])
  64   else:
  65     new_name = org_instance.name[:org_instance.name.index(':')]
  66 
  67   #Get the collections that the new instance needs to be added to.
  68   #The new collections will also be a part of the given scope,
  69   #except the special ones required for variable initialization and
  70   #training.
  71   collections = []
  72   for name, collection in org_instance.graph._collections.items():
  73     if org_instance in collection:
  74       if (name == ops.GraphKeys.GLOBAL_VARIABLES or
  75           name == ops.GraphKeys.TRAINABLE_VARIABLES or scope == ''):
  76         collections.append(name)
  77       else:
  78         collections.append(scope + '/' + name)
  79 
  80   #See if it's trainable.
  81   trainable = (
  82       org_instance in org_instance.graph.get_collection(
  83           ops.GraphKeys.TRAINABLE_VARIABLES))
  84   #Get the initial value
  85   with org_instance.graph.as_default():
  86     temp_session = Session()
  87     init_value = temp_session.run(org_instance.initialized_value())
  88 
  89   #Initialize the new variable
  90   with to_graph.as_default():
  91     new_var = VariableV1(
  92         init_value,
  93         trainable,
  94         name=new_name,
  95         collections=collections,
  96         validate_shape=False)
  97 
  98   return new_var
  99

 101 def copy_op_to_graph(org_instance, to_graph, variables, scope=''):
 102   """Returns a copy of an operation from another Graph under a specified scope.
 103 
 104   Given an `Operation` `org_instance` from one `Graph`,
 105   initializes and returns a copy of it from another `Graph`,
 106   under the specified scope (default `""`).
 107 
 108   The copying is done recursively, so any `Operation` whose output
 109   is required to evaluate the `org_instance`, is also copied (unless
 110   already done).
 111 
 112   Since `Variable` instances are copied separately, those required
 113   to evaluate `org_instance` must be provided as input.
 114 
 115   Args:
 116     org_instance: An `Operation` from some `Graph`. Could be a
 117       `Placeholder` as well.
 118     to_graph: The `Graph` to copy `org_instance` to.
 119     variables: An iterable of `Variable` instances to copy `org_instance` to.
 120     scope: A scope for the new `Variable` (default `""`).
 121 
 122   Returns:
 123     The copied `Operation` from `to_graph`.
 124 
 125   Raises:
 126     TypeError: If `org_instance` is not an `Operation` or `Tensor`.
 127   """
 128 
 129   #The name of the new instance
 130   if scope != '':
 131     new_name = scope + '/' + org_instance.name
 132   else:
 133     new_name = org_instance.name
 134 
 135   #Extract names of variables
 136   copied_variables = dict((x.name, x) for x in variables)
 137 
 138   #If a variable by the new name already exists, return the
 139   #correspondng tensor that will act as an input
 140   if new_name in copied_variables:
 141     return to_graph.get_tensor_by_name(copied_variables[new_name].name)
 142 
 143   #If an instance of the same name exists, return appropriately
 144   try:
 145     already_present = to_graph.as_graph_element(
 146         new_name, allow_tensor=True, allow_operation=True)
 147     return already_present
 148   except:
 149     pass
 150 
 151   #Get the collections that the new instance needs to be added to.
 152   #The new collections will also be a part of the given scope.
 153   collections = []
 154   for name, collection in org_instance.graph._collections.items():
 155     if org_instance in collection:
 156       if scope == '':
 157         collections.append(name)
 158       else:
 159         collections.append(scope + '/' + name)
 160 
 161   #Take action based on the class of the instance
 162 
 163   if isinstance(org_instance, ops.Tensor):
 164 
 165     #If it's a Tensor, it is one of the outputs of the underlying
 166     #op. Therefore, copy the op itself and return the appropriate
 167     #output.
 168     op = org_instance.op
 169     new_op = copy_op_to_graph(op, to_graph, variables, scope)
 170     output_index = op.outputs.index(org_instance)
 171     new_tensor = new_op.outputs[output_index]
 172     #Add to collections if any
 173     for collection in collections:
 174       to_graph.add_to_collection(collection, new_tensor)
 175 
 176     return new_tensor
 177 
 178   elif isinstance(org_instance, ops.Operation):
 179 
 180     op = org_instance
 181 
 182     #If it has an original_op parameter, copy it
 183     if op._original_op is not None:
 184       new_original_op = copy_op_to_graph(op._original_op, to_graph, variables,
 185                                          scope)
 186     else:
 187       new_original_op = None
 188 
 189     #If it has control inputs, call this function recursively on each.
 190     new_control_inputs = [
 191         copy_op_to_graph(x, to_graph, variables, scope)
 192         for x in op.control_inputs
 193     ]
 194 
 195     #If it has inputs, call this function recursively on each.
 196     new_inputs = [
 197         copy_op_to_graph(x, to_graph, variables, scope) for x in op.inputs
 198     ]
 199 
 200     #Make a new node_def based on that of the original.
 201     #An instance of tensorflow.core.framework.node_def_pb2.NodeDef, it
 202     #stores String-based info such as name, device and type of the op.
 203     #Unique to every Operation instance.
 204     new_node_def = deepcopy(op.node_def)
 205     #Change the name
 206     new_node_def.name = new_name
 207 
 208     #Copy the other inputs needed for initialization
 209     output_types = op._output_types[:]
 210     input_types = op._input_types[:]
 211 
 212     #Make a copy of the op_def too.
 213     #Its unique to every _type_ of Operation.
 214     op_def = deepcopy(op.op_def)
 215 
 216     #Initialize a new Operation instance
 217     new_op = ops.Operation(new_node_def, to_graph, new_inputs, output_types,
 218                            new_control_inputs, input_types, new_original_op,
 219                            op_def)
 220     #Use Graph's hidden methods to add the op
 221     to_graph._record_op_seen_by_control_dependencies(new_op)
 222     # pylint: disable=protected-access
 223     for device_function in to_graph._device_functions_outer_to_inner:
 224       new_op._set_device(device_function(new_op))
 225     # pylint: enable=protected-access
 226 
 227     return new_op
 228 
 229   else:
 230     raise TypeError('Could not copy instance: ' + str(org_instance))

 233 def get_copied_op(org_instance, graph, scope=''):
 234   """Given an `Operation` instance from some `Graph`, returns
 235   its namesake from `graph`, under the specified scope
 236   (default `""`).
 237 
 238   If a copy of `org_instance` is present in `graph` under the given
 239   `scope`, it will be returned.
 240 
 241   Args:
 242     org_instance: An `Operation` from some `Graph`.
 243     graph: The `Graph` to be searched for a copr of `org_instance`.
 244     scope: The scope `org_instance` is present in.
 245 
 246   Returns:
 247     The `Operation` copy from `graph`.
 248   """
 249 
 250   #The name of the copied instance
 251   if scope != '':
 252     new_name = scope + '/' + org_instance.name
 253   else:
 254     new_name = org_instance.name
 255 
 256   return graph.as_graph_element(
 257       new_name, allow_tensor=True, allow_operation=True)
