  56           method=self.get_method_name('Increment'),

  65   def testScalarHostPortTryRpc(self):
  66     with self.cached_session() as sess:
  67       request_tensors = (
  68           test_example_pb2.TestCase(values=[1, 2, 3]).SerializeToString())
  69       response_tensors, status_code, status_message = self.try_rpc(
  70           method=self.get_method_name('Increment'),
  71           address=self._address,
  72           request=request_tensors)
  73       self.assertEqual(status_code.shape, ())
  74       self.assertEqual(status_message.shape, ())
  75       self.assertEqual(response_tensors.shape, ())
  76       response_values, status_code_values, status_message_values = (
  77           sess.run((response_tensors, status_code, status_message)))
  78     response_message = test_example_pb2.TestCase()
  79     self.assertTrue(response_message.ParseFromString(response_values))
  80     self.assertAllEqual([2, 3, 4], response_message.values)
  81     # For the base Rpc op, don't expect to get error status back.
  82     self.assertEqual(errors.OK, status_code_values)
  83     self.assertEqual(b'', status_message_values)

  96   def testInvalidMethod(self):
  97     for method in [
  98         '/InvalidService.Increment',
  99         self.get_method_name('InvalidMethodName')
 100     ]:
 101       with self.cached_session() as sess:
 102         with self.assertRaisesOpError(self.invalid_method_string):
 103           sess.run(self.rpc(method=method, address=self._address, request=''))
 104 
 105         _, status_code_value, status_message_value = sess.run(
 106             self.try_rpc(method=method, address=self._address, request=''))
 107         self.assertEqual(errors.UNIMPLEMENTED, status_code_value)
 108         self.assertTrue(
 109             self.invalid_method_string in status_message_value.decode('ascii'))

 130   def testAlwaysFailingMethod(self):
 131     with self.cached_session() as sess:
 132       response_tensors = self.rpc(
 133           method=self.get_method_name('AlwaysFailWithInvalidArgument'),
 134           address=self._address,
 135           request='')

 141           method=self.get_method_name('AlwaysFailWithInvalidArgument'),
 142           address=self._address,
 143           request='')
 144       self.assertEqual(response_tensors.shape, ())
 145       self.assertEqual(status_code.shape, ())
 146       self.assertEqual(status_message.shape, ())

 148                                                           status_message))
 149       self.assertEqual(errors.INVALID_ARGUMENT, status_code_value)
 150       self.assertTrue(I_WARNED_YOU in status_message_value.decode('ascii'))

 152   def testSometimesFailingMethodWithManyRequests(self):
 153     with self.cached_session() as sess:
 154       # Fail hard by default.
 155       response_tensors = self.rpc(
 156           method=self.get_method_name('SometimesFailWithInvalidArgument'),
 157           address=self._address,
 158           request=[''] * 20)
 159       self.assertEqual(response_tensors.shape, (20,))
 160       with self.assertRaisesOpError(I_WARNED_YOU):
 161         sess.run(response_tensors)
 162 
 163       # Don't fail hard, use TryRpc - return the failing status instead.
 164       response_tensors, status_code, status_message = self.try_rpc(
 165           method=self.get_method_name('SometimesFailWithInvalidArgument'),
 166           address=self._address,
 167           request=[''] * 20)
 168       self.assertEqual(response_tensors.shape, (20,))
 169       self.assertEqual(status_code.shape, (20,))
 170       self.assertEqual(status_message.shape, (20,))
 171       status_code_values, status_message_values = sess.run((status_code,
 172                                                             status_message))
 173       self.assertTrue([
 174           x in (errors.OK, errors.INVALID_ARGUMENT) for x in status_code_values
 175       ])
 176       expected_message_values = np.where(
 177           status_code_values == errors.INVALID_ARGUMENT,
 178           I_WARNED_YOU.encode('ascii'), b'')
 179       self.assertAllEqual(expected_message_values, status_message_values)

 199   def testVecHostPortManyParallelRpcs(self):
 200     with self.cached_session() as sess:
 201       request_tensors = [
 202           test_example_pb2.TestCase(
 203               values=[i, i + 1, i + 2]).SerializeToString() for i in range(20)
 204       ]
 205       many_response_tensors = [
 206           self.rpc(
 207               method=self.get_method_name('Increment'),
 208               address=self._address,
 209               request=request_tensors) for _ in range(10)
 210       ]
 211       # Launch parallel 10 calls to the RpcOp, each containing 20 rpc requests.
 212       many_response_values = sess.run(many_response_tensors)
 213     self.assertEqual(10, len(many_response_values))
 214     for response_values in many_response_values:
 215       self.assertEqual(response_values.shape, (20,))
 216       for i in range(20):
 217         response_message = test_example_pb2.TestCase()
 218         self.assertTrue(response_message.ParseFromString(response_values[i]))
 219         self.assertAllEqual([i + 1, i + 2, i + 3], response_message.values)

 221   def testVecHostPortRpcUsingEncodeAndDecodeProto(self):
 222     with self.cached_session() as sess:
 223       request_tensors = encode_proto_op.encode_proto(
 224           message_type='tensorflow.contrib.rpc.TestCase',
 225           field_names=['values'],
 226           sizes=[[3]] * 20,
 227           values=[
 228               [[i, i + 1, i + 2] for i in range(20)],
 229           ])
 230       response_tensor_strings = self.rpc(
 231           method=self.get_method_name('Increment'),
 232           address=self._address,
 233           request=request_tensors)
 234       _, (response_shape,) = decode_proto_op.decode_proto(
 235           bytes=response_tensor_strings,
 236           message_type='tensorflow.contrib.rpc.TestCase',
 237           field_names=['values'],
 238           output_types=[dtypes.int32])
 239       response_shape_values = sess.run(response_shape)
 240     self.assertAllEqual([[i + 1, i + 2, i + 3]
 241                          for i in range(20)], response_shape_values)

 243   def testVecHostPortRpcCancelsUponSessionTimeOutWhenSleepingForever(self):
 244     with self.cached_session() as sess:
 245       request_tensors = [''] * 25  # This will launch 25 RPC requests.
 246       response_tensors = self.rpc(
 247           method=self.get_method_name('SleepForever'),
 248           address=self._address,
 249           request=request_tensors)
 250       for timeout_ms in [1, 500, 1000]:
 251         options = config_pb2.RunOptions(timeout_in_ms=timeout_ms)
 252         with self.assertRaises((errors.UnavailableError,
 253                                 errors.DeadlineExceededError)):
 254           sess.run(response_tensors, options=options)

 282   def testTryRpcWithMultipleAddressesSingleRequest(self):
 283     flatten = lambda x: list(itertools.chain.from_iterable(x))
 284     with self.cached_session() as sess:
 285       addresses = flatten([[
 286           self._address, 'unix:/tmp/this_unix_socket_doesnt_exist_97820348!!@'
 287       ] for _ in range(10)])
 288       request = test_example_pb2.TestCase(values=[0, 1, 2]).SerializeToString()
 289       response_tensors, status_code, _ = self.try_rpc(
 290           method=self.get_method_name('Increment'),
 291           address=addresses,
 292           request=request)
 293       response_tensors_values, status_code_values = sess.run((response_tensors,
 294                                                               status_code))
 295       self.assertAllEqual(
 296           flatten([errors.OK, errors.UNAVAILABLE] for _ in range(10)),
 297           status_code_values)
 298       for i in range(10):
 299         self.assertTrue(response_tensors_values[2 * i])
 300         self.assertFalse(response_tensors_values[2 * i + 1])

 318         self.assertFalse(response_tensors_values[2 * i + 1])

 320   def testTryRpcWithMultipleAddressesAndRequests(self):
 321     flatten = lambda x: list(itertools.chain.from_iterable(x))
 322     with self.cached_session() as sess:
 323       addresses = flatten([[
 324           self._address, 'unix:/tmp/this_unix_socket_doesnt_exist_97820348!!@'
 325       ] for _ in range(10)])
 326       requests = [
 327           test_example_pb2.TestCase(
 328               values=[i, i + 1, i + 2]).SerializeToString() for i in range(20)
 329       ]
 330       response_tensors, status_code, _ = self.try_rpc(
 331           method=self.get_method_name('Increment'),
 332           address=addresses,
 333           request=requests)
 334       response_tensors_values, status_code_values = sess.run((response_tensors,
 335                                                               status_code))
 336       self.assertAllEqual(
 337           flatten([errors.OK, errors.UNAVAILABLE] for _ in range(10)),
 338           status_code_values)
 339       for i in range(20):
 340         if i % 2 == 1:
 341           self.assertFalse(response_tensors_values[i])
 342         else:
 343           response_message = test_example_pb2.TestCase()
 344           self.assertTrue(
 345               response_message.ParseFromString(response_tensors_values[i]))
 346           self.assertAllEqual([i + 1, i + 2, i + 3], response_message.values)
