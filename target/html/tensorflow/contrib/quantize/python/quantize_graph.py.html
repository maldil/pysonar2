  26 def _create_graph(input_graph=None,
  27                   is_training=True,
  28                   weight_bits=8,
  29                   activation_bits=8,
  30                   quant_delay=None,
  31                   freeze_bn_delay=None,
  32                   scope=None):
  33   """Rewrites an input_graph in place for simulated quantization.
  34 
  35   The graph has fake quantization ops inserted to simulate the error
  36   introduced by quantization. Since the graph is transformed in place,
  37   the expected behavior of previously held references to nodes and tensors may
  38   change.
  39 
  40   Args:
  41     input_graph: The tf.Graph to be transformed, if None then defaults to the
  42       default graph.
  43     is_training: Whether quantizing training or eval graph.
  44     weight_bits: Number of bits to use for quantizing weights.
  45     activation_bits: Number of bits to use for quantizing activations.
  46     quant_delay: Number of steps after which weights and activations are
  47       quantized during training.
  48     freeze_bn_delay: Number of steps after which moving mean and variance are
  49       frozen and used instead of batch statistics during training.
  50       freeze_bn_delay should be greater than quant_delay and should correspond
  51       to the number of steps when training has almost converged
  52     scope: The scope to be transformed. If it's not None, only the ops which
  53       are in this scope will be transformed.
  54 
  55   Raises:
  56     ValueError: If elements contains an element that isn't a tf.Tensor or
  57       tf.Operation.
  58   """
  59 
  60   if input_graph is None:
  61     input_graph = ops.get_default_graph()
  62 
  63   # Add check to see if graph has training ops, if so provide error message and
  64   # exit
  65   _check_for_training_ops(input_graph)
  66   with input_graph.as_default():
  67     fold_batch_norms.FoldBatchNorms(
  68         input_graph,
  69         freeze_batch_norm_delay=freeze_bn_delay,
  70         is_training=is_training)
  71     quantize.Quantize(
  72         input_graph,
  73         is_training,
  74         quant_delay=quant_delay,
  75         weight_bits=weight_bits,
  76         activation_bits=activation_bits,
  77         scope=scope)

  80 def create_training_graph(input_graph=None, quant_delay=0):
  81   """Rewrites a training input_graph in place for simulated quantization.
  82 
  83   Variables added by the rewrite get added to the global variables collection.
  84 
  85   This function must be invoked prior to insertion of gradient ops in a graph
  86   as quantization should be modeled in both forward and backward passes.
  87 
  88   The graph has fake quantization ops inserted to simulate the error
  89   introduced by quantization. Since the graph is transformed in place,
  90   the expected behavior of previously held references to nodes and tensors may
  91   change.
  92 
  93   The default value of quant_delay is suitable for finetuning an already trained
  94   floating point model (recommended).
  95   If one wants to train a quantized model from scratch, quant_delay should be
  96   set to the number of steps it take the floating point model to converge.
  97   Quantization will be activated at this point and effectively finetune the
  98   model. If quant_delay is not provided when training from scratch, training can
  99   often fail.
 100 
 101   Args:
 102     input_graph: The tf.Graph to be transformed.
 103     quant_delay: Number of steps after which weights and activations are
 104       quantized during training.
 105 
 106   Raises:
 107     ValueError: If elements contains an element that isn't a tf.Tensor or
 108       tf.Operation.
 109   """
 110   # TODO(raghuramank) Need to have freeze_bn_delay be a function of batch size
 111   # Currently the values below are hardcoded for mobilenetV1 on imagenet
 112   # Please use the experimental API if you need to tune these values.
 113   freeze_bn_delay = None
 114   _create_graph(
 115       input_graph=input_graph,
 116       is_training=True,
 117       quant_delay=quant_delay,
 118       freeze_bn_delay=freeze_bn_delay)

 240 def _check_for_training_ops(g):
 241   """Check if training ops are present in the graph.
 242 
 243   Args:
 244    g: The tf.Graph on which the check for training ops needs to be
 245    performed.
 246 
 247   Raises:
 248     ValueError: If a training op is seen in the graph;
 249   """
 250 
 251   # The list here is obtained
 252   # from https://www.tensorflow.org/api_docs/cc/group/training-ops
 253   training_ops = frozenset([
 254       'ApplyAdagrad', 'ApplyAdagradDA', 'ApplyAdam', 'ApplyAddSign',
 255       'ApplyCenteredRMSProp', 'ApplyFtrl', 'ApplyFtrlV2',
 256       'ApplyGradientDescent', 'ApplyMomentum', 'ApplyPowerSign',
 257       'ApplyProximalAdagrad', 'ApplyProximalGradientDescent', 'ApplyRMSProp',
 258       'ResourceApplyAdadelta', 'ResourceApplyAdagrad', 'ResourceApplyAdagradDA',
 259       'ResourceApplyAdam', 'ResourceApplyAddSign',
 260       'ResourceApplyCenteredRMSProp', 'ResourceApplyFtrl',
 261       'ResourceApplyFtrlV2', 'ResourceApplyGradientDescent',
 262       'ResourceApplyMomentum', 'ResourceApplyPowerSign',
 263       'ResourceApplyProximalAdagrad', 'ResourceApplyProximalGradientDescent',
 264       'ResourceApplyRMSProp', 'ResourceSparseApplyAdadelta',
 265       'ResourceSparseApplyAdagrad', 'ResourceSparseApplyAdagradDA',
 266       'ResourceSparseApplyCenteredRMSProp', 'ResourceSparseApplyFtrl',
 267       'ResourceSparseApplyFtrlV2', 'ResourceSparseApplyMomentum',
 268       'ResourceSparseApplyProximalAdagrad',
 269       'ResourceSparseApplyProximalGradientDescent',
 270       'ResourceSparseApplyRMSProp', 'SparseApplyAdadelta', 'SparseApplyAdagrad',
 271       'SparseApplyAdagradDA', 'SparseApplyCenteredRMSProp', 'SparseApplyFtrl',
 272       'SparseApplyFtrlV2', 'SparseApplyMomentum', 'SparseApplyProximalAdagrad',
 273       'SparseApplyProximalGradientDescent', 'SparseApplyRMSProp'
 274   ])
 275 
 276   op_types = set([op.type for op in g.get_operations()])
 277   train_op_list = op_types.intersection(training_ops)
 278   if train_op_list:
 279     raise ValueError('Training op found in graph, exiting %s' % train_op_list)
