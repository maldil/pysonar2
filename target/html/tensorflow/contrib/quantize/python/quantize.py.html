  32 _QUANTIZABLE_TYPES = {'Conv2D', 'MatMul', 'DepthwiseConv2dNative'}

  35 _ACTIVATION_TYPES = {'Relu', 'Relu6'}

  38 def Quantize(graph,
  39              is_training,
  40              weight_bits=8,
  41              activation_bits=8,
  42              ema_decay=0.999,
  43              quant_delay=None,
  44              vars_collection=ops.GraphKeys.GLOBAL_VARIABLES,
  45              scope=None):
  46   """Updates graph with quantization operations.
  47 
  48   Currently we quantize the following tensors:
  49   * Conv/MatMul: Quantize the weights if it matches.
  50   * Activation: Quantize the output if it matches.
  51   * Bypass/Post-activation Bypass: Quantize both input and output
  52     if it matches.
  53 
  54   Args:
  55     graph: Graph to modify.
  56     is_training: Whether quantizing training graph or eval graph.
  57     weight_bits: Number of bits to use for quantizing weights.
  58     activation_bits: Number of bits to use for quantizing activations.
  59     ema_decay: (Optional) Float, EMA decay parameter.  EMA is used to update
  60       quantization intervals for quantizing activations (see here about EMA:
  61       https://en.wikipedia.org/wiki/Moving_average#Exponential_moving_average).
  62     quant_delay: (Optional, default None) Int, count of global steps for which
  63       to delay quantization.  This helps weights stabilize at the start of
  64       training.
  65     vars_collection: (Optional) Collection where to store the variables for
  66       quantization interval ends.
  67     scope: The scope to be transformed. If it's not None, only the ops which
  68       are in this scope will be transformed.
  69   Raises:
  70     ValueError: When quantization fails.
  71   """
  72   if scope and not scope.endswith('/'):
  73     scope += '/'
  74 
  75   input_to_ops_map = input_to_ops.InputToOps(graph)
  76   for layer_match in _FindLayersToQuantize(graph):
  77     # Quantize the weights.
  78     context = _GetContextFromOp(layer_match.layer_op)
  79 
  80     # If `scope` is given, only quantize it if the consumer of weights
  81     # (the layer op) is in the right scope.
  82     _InsertQuantOp(
  83         context,
  84         'weights_quant',
  85         layer_match.weight_tensor.op, [layer_match.layer_op],
  86         is_training,
  87         moving_avg=False,
  88         ema_decay=ema_decay,
  89         quant_delay=quant_delay,
  90         narrow_range=True,
  91         vars_collection=vars_collection,
  92         bits=weight_bits,
  93         consumer_scope=scope)
  94 
  95     # Quantize the activations.
  96     consumer_ops = input_to_ops_map.ConsumerOperations(
  97         layer_match.activation_op)
  98     add_context = context
  99     if layer_match.bypass_op:
 100       pattern_match_result = re.search(r'^(.*)/([^/]+)', context)
 101       if pattern_match_result is not None:
 102         add_context = pattern_match_result.group(1)
 103       else:
 104         add_context = ''
 105     # If `scope` is given, only quantize it if the producer of weights
 106     # (usually it's the layer op) is in the right scope.
 107     _InsertQuantOp(
 108         add_context,
 109         'act_quant',
 110         layer_match.activation_op,
 111         consumer_ops,
 112         is_training,
 113         moving_avg=True,
 114         ema_decay=ema_decay,
 115         quant_delay=quant_delay,
 116         vars_collection=vars_collection,
 117         bits=activation_bits,
 118         init_min=0.0,
 119         producer_scope=scope)
 120 
 121     # Quantize the inputs and output to the bypass (if it exists). The input to
 122     # the bypass is the bias add, and the output is the activation.
 123     if layer_match.bypass_op is not None:
 124       # If `scope` is given, only quantize it if the both the producer and the
 125       # consumer are in the right scope.
 126       _InsertQuantOp(
 127           context,
 128           'conv_quant',
 129           layer_match.bias_add_op, [layer_match.bypass_op],
 130           is_training,
 131           moving_avg=True,
 132           ema_decay=ema_decay,
 133           quant_delay=quant_delay,
 134           vars_collection=vars_collection,
 135           bits=activation_bits,
 136           producer_scope=scope,
 137           consumer_scope=scope)
 138       # Make sure the op following this isn't an activation. In which case, we
 139       # shouldn't quantize it, since the activation will be Fused into the
 140       # Add at inference time.
 141       consumers = input_to_ops_map.ConsumerOperations(layer_match.bypass_op)
 142       if any([consumer.type in _ACTIVATION_TYPES for consumer in consumers]):
 143         logging.info('Skipping %s, because its followed by an activation.',
 144                      layer_match.bypass_op.name)
 145       else:
 146         _InsertQuantOp(
 147             add_context,
 148             'add_quant',
 149             layer_match.bypass_op,
 150             input_to_ops_map.ConsumerOperations(layer_match.bypass_op),
 151             is_training,
 152             moving_avg=True,
 153             ema_decay=ema_decay,
 154             quant_delay=quant_delay,
 155             vars_collection=vars_collection,
 156             bits=activation_bits,
 157             producer_scope=scope,
 158             consumer_scope=scope)
 159 
 160     # Quantize bypass ops that occur after the activation.
 161     if layer_match.post_activation_bypass_op is not None:
 162       pattern_match_result = re.search(
 163           r'^(.*)/([^/]+)', layer_match.post_activation_bypass_op.name)
 164       if pattern_match_result is not None:
 165         post_activation_bypass_context = pattern_match_result.group(1)
 166       else:
 167         post_activation_bypass_context = ''
 168       # If `scope` is given, only quantize it if the producer is in the right
 169       # scope.
 170       # Make sure the op following this isn't an activation. In which case, we
 171       # shouldn't quantize it, since the activation will be Fused into the
 172       # Add at inference time.
 173       consumers = input_to_ops_map.ConsumerOperations(
 174           layer_match.post_activation_bypass_op)
 175       if any([consumer.type in _ACTIVATION_TYPES for consumer in consumers]):
 176         logging.info('Skipping %s, because its followed by an activation.',
 177                      layer_match.post_activation_bypass_op.name)
 178       else:
 179         _InsertQuantOp(
 180             post_activation_bypass_context,
 181             'post_activation_bypass_quant',
 182             layer_match.post_activation_bypass_op,
 183             consumers,
 184             is_training,
 185             moving_avg=True,
 186             ema_decay=ema_decay,
 187             quant_delay=quant_delay,
 188             vars_collection=vars_collection,
 189             bits=activation_bits,
 190             producer_scope=scope)

 193 def _FindLayersToQuantize(graph):
 194   """Matches layers in graph to quantize.
 195 
 196   The following patterns get matched. Nodes surrounded by [] will be
 197   optionally matched:
 198 
 199           weight|folded_weight
 200                 /
 201          conv|fc
 202             |
 203       [batch_to_space_nd]
 204             |
 205     [post_conv_correction]
 206             |
 207      [biasadd|folded_bias]
 208             |
 209          [bypass]
 210             |
 211         activation
 212             |
 213    [post_activation_bypass]
 214 
 215   Match replacements:
 216     If weight|folded_weight is found, FakeQuant is added afterwards.
 217     If bypass is found, FakeQuant is added before and after.
 218     If activation is found, FakeQuant is added afterwards.
 219     If post_activation_bypass is found, FakeQuant is added afterwards.
 220 
 221   Args:
 222     graph: Graph to perform match on.
 223 
 224   Returns:
 225     list of _LayerMatches.
 226   """
 227   input_pattern = graph_matcher.OpTypePattern('*')
 228   weight_var_pattern = graph_matcher.OpTypePattern('Variable|VariableV2')
 229   weight_partition_identity_pattern = graph_matcher.OpTypePattern(
 230       'Identity', inputs=[weight_var_pattern])
 231   weight_partition_concat_pattern = graph_matcher.OpTypePattern(
 232       'ConcatV2', inputs=[weight_partition_identity_pattern, '*', '*'])
 233   weight_identity_pattern = graph_matcher.OpTypePattern(
 234       'Identity',
 235       inputs=[
 236           graph_matcher.OneofPattern([
 237               weight_partition_identity_pattern,
 238               weight_partition_concat_pattern,
 239               weight_var_pattern,
 240           ])
 241       ])
 242   weight_resource_var_pattern = graph_matcher.OpTypePattern('ReadVariableOp')
 243   folded_weight_pattern = graph_matcher.OpTypePattern('Mul')
 244 
 245   # The weights inputs to the layer operation can either be from the Variable or
 246   # the folded weight (Mul).
 247   layer_pattern = graph_matcher.OpTypePattern(
 248       '|'.join(_QUANTIZABLE_TYPES),
 249       inputs=[
 250           input_pattern,
 251           graph_matcher.OneofPattern([
 252               weight_identity_pattern, weight_resource_var_pattern,
 253               folded_weight_pattern
 254           ])
 255       ],
 256       ordered_inputs=False)
 257 
 258   # For atrous convolutions a BatchToSpaceND will occur after the depthwise
 259   # convolution.
 260   batch_to_space_pattern = graph_matcher.OpTypePattern(
 261       'BatchToSpaceND',
 262       inputs=[
 263           layer_pattern,
 264           graph_matcher.OpTypePattern('*'),
 265           graph_matcher.OpTypePattern('*')
 266       ])
 267 
 268   layer_output_pattern = graph_matcher.OneofPattern(
 269       [batch_to_space_pattern, layer_pattern])
 270 
 271   # For separable convolutions, we are looking for a conv, followed by a conv
 272   # with no activations between the two.
 273   sep_conv_pattern = graph_matcher.OpTypePattern(
 274       '|'.join(_QUANTIZABLE_TYPES),
 275       inputs=[
 276           graph_matcher.OneofPattern([layer_output_pattern]),
 277           graph_matcher.OpTypePattern('*')
 278       ],
 279       ordered_inputs=False)
 280   folded_bias_mul_pattern = graph_matcher.OpTypePattern(
 281       'Mul',
 282       inputs=[graph_matcher.OpTypePattern('*'), layer_output_pattern],
 283       ordered_inputs=False)
 284   post_layer_op_correction_pattern = graph_matcher.OpTypePattern(
 285       'Add',
 286       inputs=[folded_bias_mul_pattern,
 287               graph_matcher.OpTypePattern('*')],
 288       ordered_inputs=False)
 289   folded_bias_add_pattern = graph_matcher.OpTypePattern(
 290       'Add',
 291       inputs=[
 292           post_layer_op_correction_pattern,
 293           graph_matcher.OpTypePattern('*')
 294       ],
 295       ordered_inputs=False)
 296 
 297   # batch_norms with forced updates have an Identity operation at the end.
 298   # TODO(suharshs): Find a way to easily skip extra Identity operations. The
 299   # current issue is that doing so can often match patterns across many layers
 300   # incorrectly.
 301   batch_norm_identity = graph_matcher.OpTypePattern(
 302       'Identity', inputs=[folded_bias_add_pattern])
 303 
 304   bias_add_pattern = graph_matcher.OpTypePattern(
 305       'Add|BiasAdd', inputs=[layer_output_pattern, '*'], ordered_inputs=False)
 306 
 307   # The bias can come from the bias add or the folded bias add.
 308   bypass_pattern = graph_matcher.OpTypePattern(
 309       'Add',
 310       inputs=[
 311           graph_matcher.OneofPattern(
 312               [bias_add_pattern, folded_bias_add_pattern, batch_norm_identity]),
 313           '*'
 314       ],
 315       ordered_inputs=False)
 316 
 317   # The input to the activation can come from bias add, fold bias add, the
 318   # bypasses.
 319   # TODO(suharshs): We should ideally skip Identity operations instead of
 320   # treating them as activations.
 321   activation_pattern = graph_matcher.OpTypePattern(
 322       '|'.join(_ACTIVATION_TYPES) + '|Identity',
 323       inputs=[
 324           graph_matcher.OneofPattern([
 325               bias_add_pattern,
 326               folded_bias_add_pattern,
 327               batch_norm_identity,
 328               bypass_pattern,
 329               layer_pattern,
 330           ])
 331       ])
 332 
 333   post_activation_bypass_pattern = graph_matcher.OpTypePattern(
 334       'Add', inputs=['*', activation_pattern], ordered_inputs=False)
 335 
 336   # The order of the following matching blocks is very important. Since matches
 337   # aren't guaranteed to be disjoint, we structure matches from largest to
 338   # smallest to guarantee that the largest match always wins. Additionally, we
 339   # ensure that we don't match layers multiple times.
 340 
 341   layer_matches = []
 342   # We use matched_layer_set to ensure that layers aren't matched multiple
 343   # times.
 344   matched_layer_set = set()
 345 
 346   # First, we match layers that have a post activation bypass. We do this first
 347   # to ensure we don't match only the first part of this layer, missing the
 348   # post activation bypass node.
 349   post_activation_bypass_layer_matcher = graph_matcher.GraphMatcher(
 350       post_activation_bypass_pattern)
 351   for match_result in post_activation_bypass_layer_matcher.match_graph(graph):
 352     layer_op = match_result.get_op(layer_pattern)
 353     weight_tensor = match_result.get_tensor(weight_identity_pattern)
 354     if weight_tensor is None:
 355       weight_tensor = match_result.get_tensor(weight_resource_var_pattern)
 356     if weight_tensor is None:
 357       weight_tensor = match_result.get_tensor(folded_weight_pattern)
 358     activation_op = match_result.get_op(activation_pattern)
 359     bias_add_op = match_result.get_op(bias_add_pattern)
 360     if bias_add_op is None:
 361       bias_add_op = match_result.get_op(folded_bias_add_pattern)
 362     bypass_op = match_result.get_op(bypass_pattern)
 363     post_activation_bypass_op = match_result.get_op(
 364         post_activation_bypass_pattern)
 365     if layer_op not in matched_layer_set:
 366       matched_layer_set.add(layer_op)
 367       layer_matches.append(
 368           _LayerMatch(layer_op, weight_tensor, activation_op, bypass_op,
 369                       post_activation_bypass_op, bias_add_op))
 370 
 371   # Now, we match the basic layer ending at an activation. We may get duplicate
 372   # matches from above, but we don't add them to layer_matches.
 373   layer_matcher = graph_matcher.GraphMatcher(activation_pattern)
 374   for match_result in layer_matcher.match_graph(graph):
 375     layer_op = match_result.get_op(layer_pattern)
 376     weight_tensor = match_result.get_tensor(weight_identity_pattern)
 377     if weight_tensor is None:
 378       weight_tensor = match_result.get_tensor(weight_resource_var_pattern)
 379     if weight_tensor is None:
 380       weight_tensor = match_result.get_tensor(folded_weight_pattern)
 381     activation_op = match_result.get_op(activation_pattern)
 382     bias_add_op = match_result.get_op(bias_add_pattern)
 383     if bias_add_op is None:
 384       bias_add_op = match_result.get_op(folded_bias_add_pattern)
 385     bypass_op = match_result.get_op(bypass_pattern)
 386     if layer_op not in matched_layer_set:
 387       matched_layer_set.add(layer_op)
 388       layer_matches.append(
 389           _LayerMatch(layer_op, weight_tensor, activation_op, bypass_op, None,
 390                       bias_add_op))
 391 
 392   # Match the final layer, where there may not be an activation and instead
 393   # the output of the final BiasAdd must be quantized. So we treat the BiasAdd
 394   # as the 'activation_op' in the _LayerMatch, to ensure that it's output is
 395   # quantized.
 396   final_layer_matcher = graph_matcher.GraphMatcher(
 397       graph_matcher.OneofPattern([bias_add_pattern, folded_bias_add_pattern]))
 398   for match_result in final_layer_matcher.match_graph(graph):
 399     layer_op = match_result.get_op(layer_pattern)
 400     weight_tensor = match_result.get_tensor(weight_identity_pattern)
 401     if weight_tensor is None:
 402       weight_tensor = match_result.get_tensor(weight_resource_var_pattern)
 403     if weight_tensor is None:
 404       weight_tensor = match_result.get_tensor(folded_weight_pattern)
 405     activation_op = match_result.get_op(bias_add_pattern)
 406     if activation_op is None:
 407       activation_op = match_result.get_op(folded_bias_add_pattern)
 408     if layer_op not in matched_layer_set:
 409       matched_layer_set.add(layer_op)
 410       layer_matches.append(
 411           _LayerMatch(layer_op, weight_tensor, activation_op, None, None, None))
 412 
 413   # Look for separable convolutions here
 414   sep_conv_matcher = graph_matcher.GraphMatcher(sep_conv_pattern)
 415   for match_result in sep_conv_matcher.match_graph(graph):
 416     layer_op = match_result.get_op(layer_pattern)
 417     weight_tensor = match_result.get_tensor(weight_identity_pattern)
 418     activation_op = match_result.get_op(layer_pattern)
 419     if layer_op not in matched_layer_set:
 420       matched_layer_set.add(layer_op)
 421       layer_matches.append(
 422           _LayerMatch(layer_op, weight_tensor, activation_op, None, None, None))
 423 
 424   return layer_matches

 464 def _GetFollowingFakeQuantOp(tensor):
 465   """Returns the following FakeQuant op if it exists else None."""
 466   fake_quant_ops = set([
 467       'FakeQuantWithMinMaxVars', 'FakeQuantWithMinMaxArgs',
 468       'FakeQuantWithMinMaxVarsPerChannel'
 469   ])
 470   pass_through_ops = set(['Reshape', 'Identity'])
 471   consumers = tensor.consumers()
 472   while consumers:
 473     c = consumers.pop()
 474     if c.type in fake_quant_ops:
 475       return c
 476     elif c.type in pass_through_ops:
 477       for output in c.outputs:
 478         consumers.extend(output.consumers())
 479   return None

 482 def _InsertQuantOp(context,
 483                    name,
 484                    producer,
 485                    consumers,
 486                    is_training,
 487                    moving_avg=True,
 488                    init_min=-6.0,
 489                    init_max=6.0,
 490                    bits=8,
 491                    ema_decay=0.999,
 492                    quant_delay=None,
 493                    vars_collection=ops.GraphKeys.GLOBAL_VARIABLES,
 494                    narrow_range=False,
 495                    producer_scope=None,
 496                    consumer_scope=None):
 497   """Inserts a quant op between a producer op and (multiple) consumer ops.
 498 
 499   Args:
 500     context: Context where producer and consumer operations are nested.
 501     name: Name for the new quantization op within the context.
 502     producer: Producer operation of the pairs where quantization will be
 503       inserted.
 504     consumers: Consumer operations of the pairs.
 505     is_training: Whether quantizing training graph or eval graph.
 506     moving_avg: Specifies whether to use exponential moving average or just
 507       the last value seen.
 508     init_min: Starting minimum value for the new quantization op.
 509     init_max: Starting maximum value for the new quantization op.
 510     bits: Number of bits to use for quantization, must be between 2 and 8.
 511     ema_decay: (Optional) Float, EMA decay parameter.  EMA is used to update
 512       quantization intervals for quantizing activations (see here about EMA:
 513       https://en.wikipedia.org/wiki/Moving_average#Exponential_moving_average).
 514     quant_delay: (Optional, default None) Int, count of global steps for which
 515       to delay quantization.  This helps weights stabilize at the start of
 516       training.
 517     vars_collection: (Optional) Collection where to store the variables for
 518       quantization interval ends.
 519     narrow_range: Whether to use the narrow quantization range
 520       [1; 2^bits - 1] or wide range [0; 2^bits - 1].
 521     producer_scope: The restriction of producer scope. If not None, the new op
 522       will be inserted only when the producer is in this scope.
 523     consumer_scope: The restriction of producer scope. If not None, the new op
 524       will be inserted only when all the consumers are in this scope.
 525   Raises:
 526     ValueError: When producer operation is not directly connected to the
 527       consumer operation.
 528   """
 529   if producer_scope and not producer.name.startswith(producer_scope):
 530     logging.info(
 531         '_InsertQuantOp ignores context="%s" name="%s" '
 532         'because producer "%s" is not in scope "%s"',
 533         context, name, producer.name, producer_scope)
 534     return
 535 
 536   if consumer_scope:
 537     consumers_in_scope = []
 538     for consumer in consumers:
 539       if consumer.name.startswith(consumer_scope):
 540         consumers_in_scope.append(consumer)
 541       else:
 542         logging.info(
 543             '_InsertQuantOp context="%s" name="%s" ignores '
 544             'consumer "%s" because it is not in scope "%s"',
 545             context, name, consumer.name, consumer_scope)
 546         return
 547     consumers = consumers_in_scope
 548 
 549   name_prefix = _AddContextToName(context, name)
 550   # This is needed on TPU where name_scope == 'TPUReplicate/loop', and
 551   # name_prefix starts with 'TPUReplicate/loop/'; without dropping it
 552   # variables are created as TPUReplicate/loop/TPUReplicate/loop/..., which
 553   # breaks things later.
 554   name_scope = ops.get_name_scope()
 555   if name_scope:
 556     name_prefix = common.DropStringPrefix(name_prefix, name_scope + '/')
 557 
 558   inputs = producer.outputs[0]
 559   # Prevent ops from being quantized multiple times. Bypass ops can sometimes
 560   # overlap between multiple matches, so we need to ensure that we don't
 561   # add duplicate FakeQuant operations.
 562   fake_quant_op = _GetFollowingFakeQuantOp(inputs)
 563 
 564   # If we find that we are attempting to insert a fake quant op following
 565   # a fake quant, we skip inserting a fake quant op
 566 
 567   if fake_quant_op is None:
 568     if moving_avg:
 569       quant = (
 570           quant_ops.MovingAvgQuantize(
 571               inputs,
 572               init_min=init_min,
 573               init_max=init_max,
 574               ema_decay=ema_decay,
 575               is_training=is_training,
 576               num_bits=bits,
 577               narrow_range=narrow_range,
 578               vars_collection=vars_collection,
 579               name_prefix=name_prefix))
 580     else:
 581       quant = (
 582           quant_ops.LastValueQuantize(
 583               inputs,
 584               init_min=init_min,
 585               init_max=init_max,
 586               is_training=is_training,
 587               num_bits=bits,
 588               narrow_range=narrow_range,
 589               vars_collection=vars_collection,
 590               name_prefix=name_prefix))
 591 
 592     if quant_delay and quant_delay > 0:
 593       activate_quant = math_ops.greater_equal(
 594           common.CreateOrGetQuantizationStep(),
 595           quant_delay,
 596           name=name_prefix + '/activate_quant')
 597       quant = control_flow_ops.cond(
 598           activate_quant,
 599           lambda: quant,
 600           lambda: inputs,
 601           name=name_prefix + '/delayed_quant')
 602   else:
 603     # If a fake quant op is present already, make sure that
 604     # any downstream use of the tensor reroutes to the appropriate quantized
 605     # tensor. If there is no quant_delay, this is simply the output of the
 606     # fake quant op. If there is a quant delay, we reroute to the output
 607     # of the delayed quant operation, which inserts quantization only after
 608     # a specified quant_delay
 609 
 610     quant = fake_quant_op.outputs[0]
 611     if quant_delay and quant_delay > 0:
 612       name_prefix = '/'.join(quant.name.split('/')[:-1])
 613       quant = quant.graph.get_tensor_by_name(name_prefix +
 614                                              '/delayed_quant/Merge:0')
 615     pruned_consumer_set = set()
 616     for consumer in consumers:
 617       fake_quant_dest_op = _GetFollowingFakeQuantOp(consumer.outputs[0])
 618       if (fake_quant_dest_op is None or
 619           fake_quant_dest_op.name != fake_quant_op.name):
 620         pruned_consumer_set.add(consumer)
 621     consumers = pruned_consumer_set
 622 
 623     # If we have
 624     # input->pass_through->fake_quant
 625     # there is nothing to reroute.
 626     #
 627     # If we have
 628     #  input-> pass_through->fake_quant
 629     #                |-> consumer
 630     # Then we reroute such that:
 631     # input-> pass_through->fake_quant
 632     #                            |-> consumer
 633   if consumers:
 634     tensors_modified_count = common.RerouteTensor(
 635         quant, inputs, can_modify=consumers)
 636     # Some operations can have multiple output tensors going to the same
 637     # consumer. Since consumers is a set, we need to ensure that
 638     # tensors_modified_count is greater than or equal to the length of the set
 639     # of consumers.
 640     if tensors_modified_count < len(consumers):
 641       raise ValueError('No inputs quantized for ops: [%s]' % ', '.join(
 642           [consumer.name for consumer in consumers]))

 645 def _GetContextFromOp(op):
 646   """Gets the root context name from the op name."""
 647   context_re = re.search(r'^(.*)/([^/]+)', op.name)
 648   if context_re:
 649     return context_re.group(1)
 650   return ''

 653 def _AddContextToName(context, name):
 654   """Adds the context to the name if it exists."""
 655   if not context:
 656     return name
 657   return context + '/' + name
