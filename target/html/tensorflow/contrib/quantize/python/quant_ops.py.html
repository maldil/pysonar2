  34   """Adds a fake quantize layer with fixed quantization interval.

  46         inputs, min=init_min, max=init_max)

  49 @add_arg_scope
  50 def LastValueQuantize(inputs,
  51                       per_channel=False,
  52                       init_min=-6.0,
  53                       init_max=6.0,
  54                       vars_collection=ops.GraphKeys.MOVING_AVERAGE_VARIABLES,
  55                       name_prefix='LastValueQuant',
  56                       reuse=None,
  57                       is_training=True,
  58                       num_bits=8,
  59                       narrow_range=False):
  60   """Adds a layer that collects quantization ranges as last input ranges.
  61 
  62   LastValueQuantize creates variables called 'min' and 'max', representing the
  63   interval used for quantization and clamping.
  64 
  65   Args:
  66     inputs: a tensor containing values to be quantized.
  67     per_channel: (Optional) a boolean specifying whether to use different
  68       quantization ranges per output channel.
  69     init_min: a float scalar, the initial value for variable min.
  70     init_max: a float scalar, the initial value for variable max.
  71     vars_collection: (Optional) collection where to store variables for
  72       quantization interval ends.
  73     name_prefix: name_prefix for created nodes.
  74     reuse: whether or not the layer and its variables should be reused. To be
  75       able to reuse the layer scope must be given.
  76     is_training: Whether the op is applied to a training or eval graph.
  77     num_bits: Number of bits to use for quantization, must be between 2 and 8.
  78     narrow_range: Whether to use the narrow quantization range
  79       [1; 2^num_bits - 1] or wide range [0; 2^num_bits - 1].
  80   Returns:
  81     a tensor containing quantized values.
  82   """
  83   with variable_scope.variable_scope(
  84       None, default_name=name_prefix, values=[inputs], reuse=reuse) as scope:
  85     scope.set_partitioner(None)
  86     input_shape = inputs.get_shape()
  87     input_dim = len(input_shape)
  88     if per_channel:
  89       # Only support quantizing 1-, 2- and 4-dimensional tensors.
  90       assert input_dim in [1, 2, 4], ('Expected 1D, 2D or 4D input, was: %s in '
  91                                       ' scope: %s' % (input_shape, name_prefix))
  92       min_max_shape = [input_shape[-1]]
  93     else:
  94       min_max_shape = []
  95 
  96     min_var = model_variable(
  97         'min',
  98         shape=min_max_shape,
  99         initializer=init_ops.constant_initializer(init_min),
 100         collections=[vars_collection],
 101         trainable=False)
 102     max_var = model_variable(
 103         'max',
 104         shape=min_max_shape,
 105         initializer=init_ops.constant_initializer(init_max),
 106         collections=[vars_collection],
 107         trainable=False)
 108     if not is_training:
 109       return _FakeQuantWithMinMaxVars(
 110           inputs,
 111           min_var,
 112           max_var,
 113           per_channel=per_channel,
 114           num_bits=num_bits,
 115           narrow_range=narrow_range)
 116 
 117     if per_channel:
 118       if input_dim == 2:
 119         reduce_dims = [0]
 120       elif input_dim == 4:
 121         reduce_dims = [0, 1, 2]
 122 
 123     if per_channel:
 124       if input_dim >= 2:
 125         batch_min = math_ops.reduce_min(
 126             inputs, reduction_indices=reduce_dims, name='BatchMin')
 127       else:
 128         batch_min = inputs
 129     else:
 130       batch_min = math_ops.reduce_min(inputs, name='BatchMin')
 131     # TFLite requires that 0.0 if always in the [min; max] range.
 132     batch_min = math_ops.minimum(batch_min, 0.0)
 133     assign_min = state_ops.assign(min_var, batch_min, name='AssignMinLast')
 134 
 135     if per_channel:
 136       if input_dim >= 2:
 137         batch_max = math_ops.reduce_max(
 138             inputs, reduction_indices=reduce_dims, name='BatchMax')
 139       else:
 140         batch_max = inputs
 141     else:
 142       batch_max = math_ops.reduce_max(inputs, name='BatchMax')
 143     # TFLite requires that 0.0 if always in the [min; max] range.
 144     batch_max = math_ops.maximum(batch_max, 0.0)
 145     assign_max = state_ops.assign(max_var, batch_max, name='AssignMaxLast')
 146 
 147     return _FakeQuantWithMinMaxVars(
 148         inputs,
 149         assign_min,
 150         assign_max,
 151         per_channel=per_channel,
 152         num_bits=num_bits,
 153         narrow_range=narrow_range)

 157 def MovingAvgQuantize(inputs,
 158                       per_channel=False,
 159                       init_min=-6.0,
 160                       init_max=6.0,
 161                       ema_decay=0.999,
 162                       vars_collection=ops.GraphKeys.MOVING_AVERAGE_VARIABLES,
 163                       name_prefix='MovingAvgQuantize',
 164                       reuse=None,
 165                       is_training=True,
 166                       num_bits=8,
 167                       narrow_range=False):
 168   """Adds a layer that collects quantization ranges as EMAs of input ranges.
 169 
 170   MovingAvgQuantize creates variables called 'min' and 'max', representing the
 171   interval used for quantization and clamping.
 172 
 173   Args:
 174     inputs: a tensor containing values to be quantized.
 175     per_channel: (default False) a boolean specifying whether to use different
 176       quantization ranges per output channel.
 177     init_min: a float scalar, the initial value for variable min.
 178     init_max: a float scalar, the initial value for variable max.
 179     ema_decay: EMA decay parameter.
 180     vars_collection: (Optional) collection where to store variables for
 181       quantization interval ends.
 182     name_prefix: name_prefix for created nodes.
 183     reuse: whether or not the layer and its variables should be reused. To be
 184       able to reuse the layer scope must be given.
 185     is_training: Whether the op is applied to a training or eval graph.
 186     num_bits: Number of bits to use for quantization, must be between 2 and 8.
 187     narrow_range: Whether to use the narrow quantization range
 188       [1; 2^num_bits - 1] or wide range [0; 2^num_bits - 1].
 189   Returns:
 190     a tensor containing quantized values.
 191   """
 192   with variable_scope.variable_scope(
 193       None, default_name=name_prefix, values=[inputs], reuse=reuse) as scope:
 194     scope.set_partitioner(None)
 195     input_shape = inputs.get_shape()
 196     input_dim = len(input_shape)
 197     if per_channel:
 198       # Only support quantizing 1-, 2- and 4-dimensional tensors.
 199       assert input_dim in [1, 2, 4], ('Expected 1D, 2D or 4D input, was: %s in '
 200                                       ' scope: %s' % (input_shape, name_prefix))
 201       min_max_shape = [input_shape[-1]]
 202     else:
 203       min_max_shape = []
 204 
 205     min_var = model_variable(
 206         'min',
 207         shape=min_max_shape,
 208         initializer=init_ops.constant_initializer(init_min),
 209         collections=[vars_collection],
 210         trainable=False)
 211     max_var = model_variable(
 212         'max',
 213         shape=min_max_shape,
 214         initializer=init_ops.constant_initializer(init_max),
 215         collections=[vars_collection],
 216         trainable=False)
 217     if not is_training:
 218       return _FakeQuantWithMinMaxVars(
 219           inputs,
 220           min_var,
 221           max_var,
 222           per_channel=per_channel,
 223           num_bits=num_bits,
 224           narrow_range=narrow_range)
 225     if per_channel:
 226       if input_dim == 2:
 227         reduce_dims = [0]
 228       elif input_dim == 4:
 229         reduce_dims = [0, 1, 2]
 230 
 231     if per_channel:
 232       if input_dim >= 2:
 233         batch_min = math_ops.reduce_min(
 234             inputs, reduction_indices=reduce_dims, name='BatchMin')
 235       else:
 236         batch_min = inputs
 237     else:
 238       batch_min = math_ops.reduce_min(inputs, name='BatchMin')
 239     # B-eng requires that 0.0 if always in the [min; max] range.
 240     batch_min = math_ops.minimum(batch_min, 0.0)
 241     assign_min = moving_averages.assign_moving_average(
 242         min_var, batch_min, ema_decay, name='AssignMinEma')
 243 
 244     if per_channel:
 245       if input_dim >= 2:
 246         batch_max = math_ops.reduce_max(
 247             inputs, reduction_indices=reduce_dims, name='BatchMax')
 248       else:
 249         batch_max = inputs
 250     else:
 251       batch_max = math_ops.reduce_max(inputs, name='BatchMax')
 252     # B-eng requires that 0.0 if always in the [min; max] range.
 253     batch_max = math_ops.maximum(batch_max, 0.0)
 254     assign_max = moving_averages.assign_moving_average(
 255         max_var, batch_max, ema_decay, name='AssignMaxEma')
 256 
 257     return _FakeQuantWithMinMaxVars(
 258         inputs,
 259         assign_min,
 260         assign_max,
 261         per_channel=per_channel,
 262         num_bits=num_bits,
 263         narrow_range=narrow_range)

 266 def _FakeQuantWithMinMaxVars(inputs, min_var, max_var, per_channel, num_bits,
 267                              narrow_range):
 268   """Adds a fake quantization operation.
 269 
 270   Depending on value of per_channel, this operation may do global quantization
 271   or per channel quantization.  min_var and max_var should have corresponding
 272   shapes: [1] when per_channel == False and [d] when per_channel == True.
 273 
 274   Args:
 275     inputs: a tensor containing values to be quantized.
 276     min_var: a variable containing quantization range lower end(s).
 277     max_var: a variable containing quantization range upper end(s).
 278     per_channel: a boolean specifying whether to use per-channel quantization.
 279     num_bits: Number of bits to use for quantization, must be between 2 and 8.
 280     narrow_range: Whether to use the narrow quantization range
 281       [1; 2^num_bits - 1] or wide range [0; 2^num_bits - 1].
 282   Returns:
 283     a tensor containing quantized values.
 284   """
 285 
 286   if per_channel:
 287     assert len(min_var.get_shape()) == 1
 288     assert len(max_var.get_shape()) == 1
 289     return array_ops.fake_quant_with_min_max_vars_per_channel(
 290         inputs, min_var, max_var, num_bits=num_bits, narrow_range=narrow_range)
 291   else:
 292     assert min_var.get_shape() == []  # pylint: disable=g-explicit-bool-comparison
 293     assert max_var.get_shape() == []  # pylint: disable=g-explicit-bool-comparison
 294     return array_ops.fake_quant_with_min_max_vars(
 295         inputs, min_var, max_var, num_bits=num_bits, narrow_range=narrow_range)
