  25 class Pattern(object):
  26   """The parent class of all patterns (e.g. OpTypePattern and OneofPattern)."""
  27 
  28   @abc.abstractmethod
  29   def match(self, op, tensor):
  30     """Returns the result of matching op/tensor against this pattern."""
  31     raise NotImplementedError('Method "match" not implemented.')
  32 
  33 
  34 class OpTypePattern(Pattern):
  35   """A tree pattern that matches TF expressions with certain op types."""
  36 
  37   def __init__(self, op_type, name=None, inputs=None, ordered_inputs=True):
  38     """Initializes an OpTypePattern.
  39 
  40     Args:
  41       op_type: string that specifies the allowed types of the root. It can be
  42         (1) an op type, e.g. 'Conv2D',
  43         (2) '*', i.e. wildcard, or
  44         (3) multiple op types separated by '|', e.g., 'Relu|Relu6'.
  45         We could use regex strings, which might be worthwhile when we have many
  46         similar TF op types.
  47       name: Optional string. The name of the pattern that can be looked up in
  48         MatchResult.
  49       inputs: Optional list of `Pattern`s or strings that specify the
  50         patterns for the inputs of a matching op. If None, this pattern accepts
  51         any inputs of a matching op.
  52       ordered_inputs: Defaults to True. If False, will match any op that
  53         matches a permutation of the inputs.
  54 
  55     Raises:
  56       ValueError: if too many inputs are provided when order_inputs is False.
  57     """
  58     self._op_type = op_type
  59     self._name = name
  60     if inputs is None:
  61       inputs = []
  62     if len(inputs) > 8:
  63       raise ValueError(
  64           'Only < 8 inputs are allowed when ordered_inputs is False.')
  65     self._inputs = [
  66         input_pattern
  67         if isinstance(input_pattern, Pattern) else OpTypePattern(input_pattern)
  68         for input_pattern in inputs
  69     ]
  70     self._ordered_inputs = ordered_inputs
  71 
  72   @property
  73   def name(self):
  74     return self._name
  75 
  76   def match(self, op, tensor):
  77     if self._op_type != '*':
  78       if op.type not in self._op_type.split('|'):
  79         return None
  80 
  81     match_result = MatchResult()
  82     match_result.add(self, op, tensor)
  83 
  84     if not self._inputs:
  85       # If pattern.inputs is empty, skips the rest and accepts all the inputs.
  86       return match_result
  87 
  88     if len(op.inputs) != len(self._inputs):
  89       return None
  90 
  91     input_patterns_list = [self._inputs]
  92     # If order doesn't matter for the inputs, then make sure we match at least
  93     # one permutation of the inputs.
  94     if not self._ordered_inputs:
  95       input_patterns_list = list(itertools.permutations(self._inputs))
  96 
  97     for input_patterns in input_patterns_list:
  98       match_failed = False
  99       for input_tensor, input_pattern in zip(op.inputs, input_patterns):
 100         input_match_result = input_pattern.match(input_tensor.op, input_tensor)
 101         if input_match_result is None:
 102           match_failed = True
 103           break
 104         match_result.merge_from(input_match_result)
 105       if not match_failed:
 106         return match_result
 107     return None

 110 class OneofPattern(Pattern):
 111   """Matches one of the given sub-patterns."""
 112 
 113   def __init__(self, sub_patterns):
 114     self._sub_patterns = sub_patterns
 115 
 116   def match(self, op, tensor):
 117     for sub_pattern in self._sub_patterns:
 118       match_result = sub_pattern.match(op, tensor)
 119       if match_result is not None:
 120         return match_result
 121     return None

 124 class MatchResult(object):
 125   r"""Encapsulates the result of a match done by GraphMatcher.
 126 
 127   MatchResult contains a map from Pattern to the matching op and tensor.
 128   When the matching op has multiple output tensors, the matching tensor is the
 129   output tensor used by the matching op of the parent pattern. E.g., when we
 130   match graph
 131 
 132       -         +
 133      / \y0   y1/ \
 134     x    split    z
 135           |
 136           y         (nodes are ops; edges are going up)
 137 
 138   against add_pattern defined as
 139 
 140     y1_pattern = OpTypePattern('*')
 141     z_pattern = OpTypePattern('*')
 142     add_pattern = OpTypePattern('+', inputs=[y1_pattern, z_pattern])
 143 
 144   the matching op of `y1_pattern` is `split`, and the matching tensor of
 145   `y1_pattern`
 146   is `y1` not `y0`.
 147   """
 148 
 149   def __init__(self):
 150     self._pattern_to_op_tensor = {}
 151     self._name_to_pattern = {}
 152 
 153   def add(self, pattern, op, tensor):
 154     self._pattern_to_op_tensor[pattern] = op, tensor
 155     if pattern.name is not None:
 156       if pattern.name in self._name_to_pattern:
 157         raise ValueError(
 158             'Name %s is already bound to another pattern' % pattern.name)
 159       self._name_to_pattern[pattern.name] = pattern
 160 
 161   def _to_pattern(self, pattern_or_name):
 162     if isinstance(pattern_or_name, Pattern):
 163       return pattern_or_name
 164 
 165     if isinstance(pattern_or_name, str):
 166       if pattern_or_name not in self._name_to_pattern:
 167         return None
 168       return self._name_to_pattern[pattern_or_name]
 169 
 170     raise ValueError('pattern_or_name has type %s. Expect Pattern or str.' %
 171                      type(pattern_or_name))
 172 
 173   def _get_op_tensor(self, pattern_or_name):
 174     pattern = self._to_pattern(pattern_or_name)
 175     if pattern is None:
 176       return None
 177 
 178     if pattern not in self._pattern_to_op_tensor:
 179       return None
 180 
 181     return self._pattern_to_op_tensor[pattern]
 182 
 183   def get_op(self, pattern_or_name):
 184     op_tensor = self._get_op_tensor(pattern_or_name)
 185     return op_tensor[0] if op_tensor else None
 186 
 187   def get_tensor(self, pattern_or_name):
 188     op_tensor = self._get_op_tensor(pattern_or_name)
 189     return op_tensor[1] if op_tensor else None
 190 
 191   def merge_from(self, other_match_result):
 192     # pylint: disable=protected-access
 193     self._pattern_to_op_tensor.update(other_match_result._pattern_to_op_tensor)
 194     self._name_to_pattern.update(other_match_result._name_to_pattern)
 195     # pylint: enable=protected-access

 198 class GraphMatcher(object):
 199   """Checks if a particular subgraph matches a given pattern."""
 200 
 201   def __init__(self, pattern):
 202     """Initializes a GraphMatcher.
 203 
 204     Args:
 205       pattern: The `Pattern` against which `GraphMatcher` matches
 206         subgraphs.
 207     """
 208     self._pattern = pattern
 209 
 210   def _match_pattern(self, pattern, op, tensor):
 211     """Returns whether an TF expression rooted at `op` matches `pattern`.
 212 
 213     If there is a match, adds to `self._match_result` the matching op and tensor
 214     with key `pattern`.
 215 
 216     Args:
 217       pattern: An `Pattern`.
 218       op: A `tf.Operation` to match against the pattern.
 219       tensor: the output `tf.Tensor` of `op` that is used by the matching op of
 220         `pattern`'s parent. Can be None if `pattern` is already the root of the
 221         pattern tree.
 222 
 223     Returns:
 224       True if an TF expression rooted at `op` matches `pattern`.
 225     """
 226     match_result = pattern.match(op, tensor)
 227     if match_result is None:
 228       return False
 229     self._match_result.merge_from(match_result)
 230     return True
 231 
 232   def match_op(self, op):
 233     """Matches `op` against `self._pattern`.
 234 
 235     Args:
 236       op: `tf.Operation` to match against the pattern.
 237 
 238     Returns:
 239       Returns a `MatchResult` if `op` matches the pattern; otherwise, returns
 240       None.
 241     """
 242     self._match_result = MatchResult()
 243     if not self._match_pattern(self._pattern, op, tensor=None):
 244       return None
 245     return self._match_result
 246 
 247   def match_ops(self, ops):
 248     """Matches each operation in `ops` against `self._pattern`.
 249 
 250     Args:
 251       ops: collection of `tf.Operation` to match against the pattern.
 252 
 253     Yields:
 254       `MatchResult` for each `tf.Operation` that matches the pattern.
 255     """
 256     for op in ops:
 257       match_result = self.match_op(op)
 258       if match_result:
 259         yield match_result
 260 
 261   def match_graph(self, graph):
 262     """Matches each operation in `graph` against `self._pattern`.
 263 
 264     Args:
 265       graph: `tf.Graph` containing operations to match.
 266 
 267     Yields:
 268       `MatchResult` for each `tf.Operation` in `graph` that matches the pattern.
 269     """
 270     # Python 3.3.2+ implements `yield from`, but for now:
 271     for match_result in self.match_ops(graph.get_operations()):
 272       yield match_result
