  37 def FoldBatchNorms(graph, is_training, freeze_batch_norm_delay=None):
  38   """Finds batch norm layers and folds them into preceding layers.
  39 
  40   Folding only affects the following layers: Conv2D, fully connected, depthwise
  41   convolution.
  42 
  43   Args:
  44     graph: Graph to walk and modify.
  45     is_training: Bool, true if training.
  46     freeze_batch_norm_delay: How many steps to wait before freezing moving mean
  47       and variance and using them for batch normalization. This value is used
  48       only when is_training is True.
  49   Raises:
  50     ValueError: When batch norm folding fails.
  51   """
  52   _FoldFusedBatchNorms(
  53       graph, is_training, freeze_batch_norm_delay=freeze_batch_norm_delay)
  54   _FoldUnfusedBatchNorms(
  55       graph,
  56       is_training=is_training,
  57       freeze_batch_norm_delay=freeze_batch_norm_delay)

  60 def _FoldFusedBatchNorms(graph, is_training, freeze_batch_norm_delay):
  61   """Finds fused batch norm layers and folds them into preceding layers.
  62 
  63   Folding only affects the following layers: Conv2D, fully connected, depthwise
  64   convolution.
  65 
  66   Args:
  67     graph: Graph to walk and modify.
  68     is_training: Bool, true if training.
  69     freeze_batch_norm_delay: How many steps to wait before freezing moving mean
  70       and variance and using them for batch normalization.
  71 
  72   Raises:
  73     ValueError: When batch norm folding fails.
  74   """
  75   for match in _FindFusedBatchNorms(graph):
  76     scope, sep, _ = match.layer_op.name.rpartition('/')
  77     # Make sure new ops are added to `graph` and put on the same device as
  78     # `bn_op`. The '/' (i.e. `sep`) ensures that we reuse the existing scope
  79     # named `scope`. Otherwise, TF creates a unique scope whose name starts with
  80     # `scope`.
  81     with graph.as_default(), graph.name_scope(scope + sep):
  82       with graph.name_scope(scope + sep + 'BatchNorm_Fold' + sep):
  83         # new weights = old weights * gamma / sqrt(variance + epsilon)
  84         # new biases = -mean * gamma / sqrt(variance + epsilon) + beta
  85         multiplier_tensor = match.gamma_tensor * math_ops.rsqrt(
  86             match.variance_tensor + match.bn_op.get_attr('epsilon'))
  87         bias_tensor = math_ops.subtract(
  88             match.beta_tensor,
  89             match.mean_tensor * multiplier_tensor,
  90             name='bias')
  91 
  92         correction_scale, correction_recip, correction_offset = None, None, None
  93         if is_training:
  94           correction_scale, correction_recip, correction_offset = (
  95               _ComputeBatchNormCorrections(
  96                   context='',
  97                   match=match,
  98                   freeze_batch_norm_delay=freeze_batch_norm_delay))
  99         # The shape of depthwise weights is different, so we need to reshape the
 100         # multiplier_tensor to ensure that the scaled_weight_tensor has the
 101         # expected shape.
 102         weights = match.weight_tensor
 103         if match.layer_op.type == 'DepthwiseConv2dNative':
 104           new_shape = [
 105               match.weight_tensor.get_shape().as_list()[2],
 106               match.weight_tensor.get_shape().as_list()[3]
 107           ]
 108           multiplier_tensor = array_ops.reshape(
 109               multiplier_tensor, new_shape, name='scale_reshape')
 110 
 111           if correction_scale is not None:
 112             correction_scale = array_ops.reshape(
 113                 correction_scale, new_shape, name='correction_reshape')
 114 
 115       if correction_scale is not None:
 116         weights = math_ops.multiply(
 117             correction_scale, weights, name='correction_mult')
 118 
 119       scaled_weight_tensor = math_ops.multiply(
 120           weights, multiplier_tensor, name='mul_fold')
 121 
 122       new_layer_tensor = _CloneWithNewOperands(
 123           match.layer_op, match.input_tensor, scaled_weight_tensor,
 124           match.batch_to_space_op)
 125 
 126       if correction_recip is not None:
 127         new_layer_tensor = math_ops.multiply(
 128             correction_recip, new_layer_tensor, name='post_conv_mul')
 129         new_layer_tensor = math_ops.add(new_layer_tensor, (correction_offset),
 130                                         'correction_add')
 131 
 132       bias_add_tensor = math_ops.add(
 133           new_layer_tensor, bias_tensor, name='add_fold')
 134 
 135       nodes_modified_count = common.RerouteTensor(bias_add_tensor,
 136                                                   match.output_tensor)
 137       if nodes_modified_count == 0:
 138         raise ValueError('Folding batch norms failed, %s had no outputs.' %
 139                          match.output_tensor.name)

 142 def _FindFusedBatchNorms(graph):
 143   """Finds all ops and tensors related to found FusedBatchNorms.
 144 
 145   Args:
 146     graph: Graph to inspect.
 147 
 148   Yields:
 149     _FusedBatchNormMatches.
 150   """
 151   input_pattern = graph_matcher.OpTypePattern('*')
 152   # In practice, the weight pattern can match a Variable or a SpaceToBatchND
 153   # operation that follows a variable for atrous convolutions.
 154   weight_pattern = graph_matcher.OpTypePattern('*')
 155   gamma_pattern = graph_matcher.OpTypePattern('*')
 156   beta_pattern = graph_matcher.OpTypePattern('*')
 157   mean_pattern = graph_matcher.OpTypePattern('*')
 158   variance_pattern = graph_matcher.OpTypePattern('*')
 159 
 160   moving_average_pattern = graph_matcher.OpTypePattern('*')
 161   bn_decay_pattern = graph_matcher.OpTypePattern('*')
 162   layer_pattern = graph_matcher.OpTypePattern(
 163       'Conv2D|DepthwiseConv2dNative|MatMul',
 164       inputs=[input_pattern, weight_pattern])
 165   batch_to_space_pattern = graph_matcher.OpTypePattern(
 166       'BatchToSpaceND',
 167       inputs=[
 168           layer_pattern,
 169           graph_matcher.OpTypePattern('*'),
 170           graph_matcher.OpTypePattern('*')
 171       ])
 172   layer_output_pattern = graph_matcher.OneofPattern(
 173       [layer_pattern, batch_to_space_pattern])
 174   # MatMul has a Reshape between it and FusedBatchNorm.
 175   matmul_reshape_pattern = graph_matcher.OpTypePattern(
 176       'Reshape',
 177       inputs=[layer_output_pattern,
 178               graph_matcher.OpTypePattern('*')])
 179 
 180   batch_norm_pattern = graph_matcher.OpTypePattern(
 181       'FusedBatchNorm',
 182       inputs=[
 183           graph_matcher.OneofPattern(
 184               [matmul_reshape_pattern, layer_output_pattern]), gamma_pattern,
 185           beta_pattern, mean_pattern, variance_pattern
 186       ])
 187   matmul_bn_output_reshape_pattern = graph_matcher.OpTypePattern(
 188       'Reshape', inputs=[batch_norm_pattern,
 189                          graph_matcher.OpTypePattern('*')])
 190 
 191   bn_matcher = graph_matcher.GraphMatcher(
 192       graph_matcher.OneofPattern(
 193           [matmul_bn_output_reshape_pattern, batch_norm_pattern]))
 194 
 195   moving_average_sub_pattern = graph_matcher.OpTypePattern(
 196       'Sub', inputs=[moving_average_pattern, batch_norm_pattern])
 197   moving_average_mul_pattern = graph_matcher.OpTypePattern(
 198       'Mul', inputs=[moving_average_sub_pattern, bn_decay_pattern])
 199 
 200   moving_avg_mul_matcher = graph_matcher.GraphMatcher(
 201       moving_average_mul_pattern)
 202 
 203   for match_result in bn_matcher.match_graph(graph):
 204     moving_mean_tensor = None
 205     moving_variance_tensor = None
 206     bn_decay_mean_tensor = None
 207     bn_decay_var_tensor = None
 208     batch_to_space_op = None
 209     layer_op = match_result.get_op(layer_pattern)
 210     layer_tensor = match_result.get_tensor(layer_pattern)
 211     bn_op = match_result.get_op(batch_norm_pattern)
 212     batch_epsilon = bn_op.get_attr('epsilon')
 213 
 214     # In the MatMul case, the output of batch norm is reshaped back into a
 215     # 2D tensor, so the output_tensor is the output of the Reshape op.
 216     output_tensor = bn_op.outputs[0]
 217     if layer_op.type == 'MatMul':
 218       output_reshape_op = match_result.get_op(matmul_bn_output_reshape_pattern)
 219       # If the matcher didn't match matmul_bn_output_reshape, there will be
 220       # another match for this 'MatMul' later, so we can skip this one.
 221       if output_reshape_op is None:
 222         continue
 223       output_tensor = output_reshape_op.outputs[0]
 224 
 225     # Ensure that the output tensor has consumers, otherwise this is a dangling
 226     # node and not a match.
 227     if not output_tensor.consumers():
 228       continue
 229 
 230     batch_to_space_op = match_result.get_op(batch_to_space_pattern)
 231     input_tensor = match_result.get_tensor(input_pattern)
 232     weight_tensor = match_result.get_tensor(weight_pattern)
 233     gamma_tensor = match_result.get_tensor(gamma_pattern)
 234     beta_tensor = match_result.get_tensor(beta_pattern)
 235     # FusedBatchNorm in training is different from that in inference. It takes
 236     # empty 'mean' and empty 'variance', and produces the mean and the variance
 237     # of the batch. Therefore, when is_training is true, mean_tensor and
 238     # variance_tensor point to 1st and 2nd (0-based) output of bn_op,
 239     # respectively; when is_training is false, they point to bn_op's inputs.
 240     is_training = bn_op.get_attr('is_training')
 241     if is_training:
 242       # FusedBatchNormGrad doesn't compute gradients of the batch_mean and
 243       # batch_variance outputs, so we need to substitute our own custom
 244       # gradient.
 245       # TODO(suharshs, raghuramank): Find a way to avoid needing this hack.
 246       # pylint: disable=protected-access
 247       bn_op._set_attr(
 248           '_gradient_op_type',
 249           attr_value_pb2.AttrValue(s=compat.as_bytes('FoldFusedBatchNormGrad')))
 250       # pylint: enable=protected-access
 251       mean_tensor = bn_op.outputs[1]
 252       # The batch variance used during forward and backward prop is biased,
 253       # i.e it is calculated as: V=sum(x(k)-mu)^2/N. For the moving average
 254       # calculation, the variance is corrected by the term N/N-1 (Bessel's
 255       # correction). The variance tensor read from FuseBatchNorm has Bessel's
 256       # correction applied, so we undo it here.
 257       scope, sep, _ = bn_op.name.rpartition('/')
 258       g = ops.get_default_graph()
 259       with g.as_default(), g.name_scope(scope + sep):
 260         n = math_ops.cast(
 261             array_ops.size(layer_tensor) / array_ops.size(mean_tensor),
 262             dtypes.float32)
 263         variance_tensor = math_ops.multiply(
 264             bn_op.outputs[2], (n - 1) / n, name='Undo_Bessel_Correction')
 265       # TODO(suharshs): Find a way to get rid of this inner match.
 266       for mul_match_result in moving_avg_mul_matcher.match_graph(graph):
 267         sub_op = mul_match_result.get_op(moving_average_sub_pattern)
 268         if sub_op.inputs[1].name == bn_op.outputs[1].name:
 269           # During training: Batch Mean is bn_op.outputs[1]
 270           moving_mean_tensor = sub_op.inputs[0]
 271           bn_decay_mean_tensor = mul_match_result.get_tensor(bn_decay_pattern)
 272         if sub_op.inputs[1].name == bn_op.outputs[2].name:
 273           # During training: Batch Var is bn_op.outputs[2]
 274           moving_variance_tensor = sub_op.inputs[0]
 275           bn_decay_var_tensor = mul_match_result.get_tensor(bn_decay_pattern)
 276     else:
 277       mean_tensor = match_result.get_tensor(mean_pattern)
 278       variance_tensor = match_result.get_tensor(variance_pattern)
 279 
 280     yield _BatchNormMatch(
 281         layer_op=layer_op,
 282         bn_op=bn_op,
 283         output_tensor=output_tensor,
 284         input_tensor=input_tensor,
 285         weight_tensor=weight_tensor,
 286         gamma_tensor=gamma_tensor,
 287         beta_tensor=beta_tensor,
 288         mean_tensor=mean_tensor,
 289         variance_tensor=variance_tensor,
 290         moving_mean_tensor=moving_mean_tensor,
 291         moving_variance_tensor=moving_variance_tensor,
 292         bn_decay_mean_tensor=bn_decay_mean_tensor,
 293         bn_decay_var_tensor=bn_decay_var_tensor,
 294         batch_epsilon=batch_epsilon,
 295         batch_to_space_op=batch_to_space_op)

 298 def _ComputeBatchNormCorrections(context, match, freeze_batch_norm_delay):
 299   """Computes batch norm correction params.
 300 
 301      Before batch normalization is frozen:
 302      We use batch statistics for batch norm.
 303        correction_scale = sigma_b/sigma_mv
 304        correction_recip = 1/correction_scale
 305        correction_offset = 0
 306 
 307      After batch normalization is frozen:
 308       correction_scale = sigma_b/sigma_mv
 309       correction_recip = 1
 310       correction_offset =  gamma*(mu_b/sigma_b-mu_mv/sigma_mv).
 311 
 312      Batch norm is frozen if global_step > bn_freeze_delay.
 313      The corrections ensure that:
 314      a) The weights are quantized after scaling by gamma/sigma_mv. This enables
 315      smoother training as the scaling on the weights changes slowly, rather than
 316      jump across mini-batches
 317      b) Changing the values of the corrections allows for one to switch between
 318      using batch statistics to using moving mean and average, without requiring
 319      changes to batch_norm
 320 
 321 
 322   Args:
 323     context: The scope under which we look for batch norm params
 324     match: Object containing required batch norm tensors for correction
 325       computation.
 326     freeze_batch_norm_delay: Delay in steps at which computation switches
 327       from regular batch norm to frozen mean and variance.
 328 
 329 
 330   Returns:
 331     A tuple of correction_scale, correction_recip, correction_offset
 332   """
 333 
 334   g = ops.get_default_graph()
 335   prefix = '' if not context else context
 336   with g.name_scope(prefix + 'batch_norm_correction'):
 337     recip_sigma_mv = math_ops.rsqrt(
 338         match.moving_variance_tensor + match.batch_epsilon)
 339     recip_sigma = math_ops.rsqrt(match.variance_tensor + match.batch_epsilon)
 340     correction_scale = math_ops.divide(
 341         recip_sigma_mv, recip_sigma, name='scale_compute')
 342     correction_scale = array_ops.identity(
 343         correction_scale, name='correction_scale')
 344     correction_recip = math_ops.reciprocal(
 345         correction_scale, name='reciprocal_compute')
 346     correction_offset = math_ops.multiply(
 347         match.gamma_tensor,
 348         match.mean_tensor * recip_sigma -
 349         match.moving_mean_tensor * recip_sigma_mv,
 350         name='offset_compute')
 351 
 352     if freeze_batch_norm_delay is not None:
 353       use_mv_avg = math_ops.greater_equal(
 354           common.CreateOrGetQuantizationStep(),
 355           freeze_batch_norm_delay,
 356           name='use_moving_average')
 357     else:
 358       use_mv_avg = False
 359 
 360     bn_decay_zero = 0.0
 361     bn_decay_mean_consumers = list(match.bn_decay_mean_tensor.consumers())
 362     bn_decay_var_consumers = list(match.bn_decay_mean_tensor.consumers())
 363 
 364     bn_decay_mean_out = utils.smart_cond(
 365         use_mv_avg,
 366         lambda: bn_decay_zero,
 367         lambda: match.bn_decay_mean_tensor,
 368         name='freeze_moving_mean')
 369 
 370     common.RerouteTensor(
 371         bn_decay_mean_out,
 372         match.bn_decay_mean_tensor,
 373         can_modify=bn_decay_mean_consumers)
 374 
 375     bn_decay_var_consumers = list(match.bn_decay_var_tensor.consumers())
 376     bn_decay_var_out = utils.smart_cond(
 377         use_mv_avg,
 378         lambda: bn_decay_zero,
 379         lambda: match.bn_decay_var_tensor,
 380         name='freeze_moving_var')
 381     common.RerouteTensor(
 382         bn_decay_var_out,
 383         match.bn_decay_var_tensor,
 384         can_modify=bn_decay_var_consumers)
 385 
 386     correction_recip = utils.smart_cond(
 387         use_mv_avg,
 388         lambda: array_ops.ones(correction_scale.shape),
 389         lambda: correction_recip,
 390         name='correction_recip')
 391 
 392     correction_offset = utils.smart_cond(
 393         use_mv_avg,
 394         lambda: correction_offset,
 395         lambda: array_ops.zeros(correction_offset.shape),
 396         name='correction_offset')
 397   return correction_scale, correction_recip, correction_offset

 400 def _CloneWithNewOperands(layer_op, input_tensor, weight_tensor,
 401                           batch_to_space_op):
 402   """Clones layer_op with input_tensor and weight_tensor as new inputs."""
 403   new_layer_name = layer_op.name.split('/')[-1] + '_Fold'
 404   if layer_op.type == 'Conv2D':
 405     return nn_ops.conv2d(
 406         input_tensor,
 407         weight_tensor,
 408         strides=layer_op.get_attr('strides'),
 409         padding=layer_op.get_attr('padding'),
 410         use_cudnn_on_gpu=layer_op.get_attr('use_cudnn_on_gpu'),
 411         data_format=layer_op.get_attr('data_format'),
 412         name=new_layer_name)
 413   elif layer_op.type == 'MatMul':
 414     return math_ops.matmul(
 415         input_tensor,
 416         weight_tensor,
 417         transpose_a=layer_op.get_attr('transpose_a'),
 418         transpose_b=layer_op.get_attr('transpose_b'),
 419         name=new_layer_name)
 420   elif layer_op.type == 'DepthwiseConv2dNative':
 421     # We don't copy dilation rate because we reuse the input SpaceToBatch
 422     # and create our own BatchToSpace operation below.
 423     conv = nn.depthwise_conv2d(
 424         input_tensor,
 425         weight_tensor,
 426         strides=layer_op.get_attr('strides'),
 427         padding=layer_op.get_attr('padding'),
 428         name=new_layer_name)
 429     # Copy the batch to space operation if we have a atrous convolution.
 430     if batch_to_space_op:
 431       batch_to_space_op = layer_op.outputs[0].consumers()[0]
 432       # TODO(suharshs): It's hard to make this name match with the unfused name.
 433       # Restructure this code to not rely on scope at all.
 434       new_batch_to_space_name = batch_to_space_op.name.split('/')[-1] + '_Fold'
 435       conv = array_ops.batch_to_space_nd(
 436           conv,
 437           batch_to_space_op.inputs[1],
 438           batch_to_space_op.inputs[2],
 439           name=new_batch_to_space_name)
 440     return conv
 441   else:
 442     raise ValueError('Cannot handle operation of type: %s' % layer_op.type)

 445 @ops.RegisterGradient('FoldFusedBatchNormGrad')
 446 def _FoldFusedBatchNormGrad(op, unused_grad_y, grad_mean, grad_var, unused_1,
 447                             unused_2):
 448   x = op.inputs[0]
 449   n = math_ops.cast(
 450       array_ops.size(x) / array_ops.size(grad_mean), dtypes.float32)
 451   dmean_dx = grad_mean / n
 452   dvar_dx = 2 * grad_var * (x - op.outputs[1]) / (n - 1)
 453   return (dmean_dx + dvar_dx), None, None, None, None

 456 def _FoldUnfusedBatchNorms(graph, is_training, freeze_batch_norm_delay):
 457   """Finds unfused batch norm layers and folds them into preceding layers.
 458 
 459   Folding only affects the following layers: Conv2D, fully connected, depthwise
 460   convolution.
 461 
 462   Args:
 463     graph: Graph to walk and modify.
 464     is_training: Bool, True if training.
 465     freeze_batch_norm_delay: How many steps to wait before freezing moving mean
 466       and variance and using them for batch normalization.
 467 
 468   Raises:
 469     ValueError: When batch norm folding fails.
 470   """
 471   input_to_ops_map = input_to_ops.InputToOps(graph)
 472 
 473   for bn in common.BatchNormGroups(graph):
 474     has_scaling = _HasScaling(graph, input_to_ops_map, bn)
 475 
 476     if not _IsValidUnfusedBatchNorm(graph, bn):
 477       continue
 478 
 479     # The mangling code intimately depends on BatchNorm node's internals.
 480     original_op, folded_op = _CreateFoldedOp(
 481         graph,
 482         bn,
 483         has_scaling=has_scaling,
 484         freeze_batch_norm_delay=freeze_batch_norm_delay,
 485         is_training=is_training)
 486 
 487     activation = common.GetEndpointActivationOp(graph, bn)
 488     if activation:
 489       nodes_modified_count = common.RerouteTensor(
 490           folded_op.outputs[0], original_op.outputs[0], can_modify=[activation])
 491       if nodes_modified_count != 1:
 492         raise ValueError('Unexpected inputs to op: %s' % activation.name)
 493       continue
 494 
 495     # Treat consumer ops in bypass modules differently since they have Add
 496     # operations instead of Relu* above.
 497     # Changes to make sure that the correct scope is selected for the bypass add
 498     # The rule here is that if the scope is of the form: str1/str2 for the
 499     # batch norm,
 500     # the bypass add is at scope str1. If bn is of scope just str1, then the
 501     # bypass add is at scope ''.
 502     # If there is no batch norm, then there is no bypass add.
 503     add_bypass_ctx = ''
 504     if bn:
 505       try:
 506         add_bypass_ctx = re.search(r'^(.*)/([^/]+)', bn).group(1)
 507       except AttributeError:
 508         add_bypass_ctx = ''
 509 
 510     if add_bypass_ctx:
 511       add_bypass_ctx = add_bypass_ctx + '/'
 512 
 513     add_bypass = graph.get_operation_by_name(add_bypass_ctx + 'Add')
 514     nodes_modified_count = common.RerouteTensor(
 515         folded_op.outputs[0], original_op.outputs[0], can_modify=[add_bypass])
 516     if nodes_modified_count != 1:
 517       raise ValueError('Unexpected inputs to op: %s' % add_bypass.name)
 518

 520 def _IsValidUnfusedBatchNorm(graph, context):
 521   """Checks that the output of the unfused batch norm has consumers."""
 522   add_shift = graph.get_operation_by_name(context +
 523                                           'BatchNorm/batchnorm_1/add_1')
 524   # Ensure that the output tensor of batch norm has consumers, otherwise this
 525   # is a dangling node and not a match.
 526   return bool(add_shift.outputs[0].consumers())

 529 def _FindMatchingTensor(graph, match_pattern, scope):
 530   """Finds best match of ops matching match_pattern with scope.
 531 
 532      Example: _FindMatchingTensor(graph,'/BatchNorm/moments/Squeeze',
 533      'MobilenetV1/MobilenetV1/Conv2d_0/') returns:
 534       Tensor('MobilenetV1/Conv2d_0/BatchNorm/moments/Squeeze')
 535 
 536   Args:
 537     graph: Graph to inspect.
 538     match_pattern: Part of the name of the op that we need to match, should
 539     be present in the op's name
 540     scope: The scope of the op. All the elements of the scope need not be
 541     present in the op's name.
 542 
 543   Returns:
 544     Tensor from graph that provides the best match to the match_pattern and
 545     scope
 546   """
 547 
 548   oplist = graph.get_operations()
 549   split_context = set(scope.split('/'))
 550   match_dict = {}
 551   for op in oplist:
 552     if op.name.endswith(match_pattern):
 553       split_name = op.name.split('/')
 554       num_matches = len(set(split_name) & split_context)
 555 
 556       if num_matches > 0 or not scope:
 557         match_dict[op.name] = num_matches
 558   # match_dict contains matching op names from graph with values being
 559   # number of matches to scope. We pick the key with the most matches
 560   if match_dict:
 561     max_key = max(match_dict, key=match_dict.get)
 562     return graph.get_tensor_by_name(max_key + ':0')
 563   else:
 564     return None

 567 def _GetBatchNormParams(graph, context, has_scaling):
 568   """Extracts relevant tensors for folding batch norms.
 569 
 570   Args:
 571     graph: Graph to inspect.
 572     context: The scope under which we look for batch norm params
 573     has_scaling: Bool that specifies if scaling is done as part of batch norm.
 574 
 575   Returns:
 576     _BatchNormMatch containing all required batch norm parameters.
 577   """
 578   gamma_tensor = None
 579   batch_mean_tensor = None
 580   batch_variance_tensor = None
 581   moving_mean_tensor = None
 582   moving_variance_tensor = None
 583   batch_epsilon = None
 584   bn_decay_mean_tensor = None
 585   bn_decay_var_tensor = None
 586 
 587   # TODO(raghuramank) This code relies on string matching and needs to be
 588   # updated if unfused batch norm continues to be widely used
 589   # Matching variable names is brittle and relies on scoping
 590   # conventions. Fused batch norm folding is more robust. Support for unfused
 591   # batch norms will be deprecated as we move forward. Fused batch norms allow
 592   # for faster training and should be used whenever possible.
 593   # context contains part of the names of the tensors we are interested in:
 594   # For MobilenetV1, the context has repetitions:
 595   # MobilenetV1/MobilenetV1/Conv2d_3_depthwise
 596   # when the moving_mean tensor has the name:
 597   # MobilenetV1/Conv2d_3_depthwise/BatchNorm/moving_mean/read
 598   # To pick the correct variable name, it is necessary to ignore the repeating
 599   # header.
 600 
 601   # For MobilenetV2, this problem does not exist:
 602   # The context is: MobilenetV2/expanded_conv_3/depthwise
 603   # and the names of the tensors start with a single MobilenetV2
 604   # The moving mean for example, has the name:
 605   # MobilenetV2/expanded_conv_3/depthwise/BatchNorm/moving_mean/read
 606   # We identify the best match for an op by checking for
 607   # 1. The suffix of the op is exactly matched
 608   # 2. Maximum number of matches with the context.The matching
 609   # score is given by the number of parts of context (split by /) that
 610   # are present in the parts of the tensor name (again split by /).
 611   # For example: scope= MobilenetV2/MobilenetV2/expanded_conv_3 and
 612   # op.name =  MobilenetV2/expanded_conv_3/depthwise/BatchNorm/moving_mean/read
 613   # will have 2 matches,scope with a different conv layer will have one match.
 614 
 615   op_suffix_mean = 'BatchNorm/moments/Squeeze'
 616   op_suffix_variance = 'BatchNorm/moments/Squeeze_1'
 617   op_suffix_epsilon = 'BatchNorm/batchnorm_1/add/y'
 618   op_suffix_bn_decay_mean = 'BatchNorm/AssignMovingAvg/decay'
 619   op_suffix_bn_decay_var = 'BatchNorm/AssignMovingAvg_1/decay'
 620 
 621   if variable_scope.get_variable_scope().use_resource:
 622     op_suffix_gamma = 'BatchNorm/gamma/Read/ReadVariableOp'
 623     op_suffix_moving_variance = (
 624         'BatchNorm/moving_variance/Read/ReadVariableOp')
 625     op_suffix_moving_mean = ('BatchNorm/moving_mean/Read/ReadVariableOp')
 626   else:
 627     op_suffix_gamma = 'BatchNorm/gamma'
 628     op_suffix_moving_variance = 'BatchNorm/moving_variance/read'
 629     op_suffix_moving_mean = 'BatchNorm/moving_mean/read'
 630   # Parse through list of ops to find relevant ops
 631 
 632   batch_mean_tensor = _FindMatchingTensor(graph, op_suffix_mean, context)
 633   batch_variance_tensor = _FindMatchingTensor(graph, op_suffix_variance,
 634                                               context)
 635   moving_mean_tensor = _FindMatchingTensor(graph, op_suffix_moving_mean,
 636                                            context)
 637   moving_variance_tensor = _FindMatchingTensor(graph, op_suffix_moving_variance,
 638                                                context)
 639   batch_epsilon = _FindMatchingTensor(graph, op_suffix_epsilon, context)
 640   bn_decay_mean_tensor = _FindMatchingTensor(graph, op_suffix_bn_decay_mean,
 641                                              context)
 642   bn_decay_var_tensor = _FindMatchingTensor(graph, op_suffix_bn_decay_var,
 643                                             context)
 644   if batch_mean_tensor is None and moving_mean_tensor is None:
 645     ValueError('Error folding unfused batch norms')
 646   if has_scaling:
 647     gamma_tensor = _FindMatchingTensor(graph, op_suffix_gamma, context)
 648 
 649   if not has_scaling:
 650     gamma_tensor = array_ops.ones(moving_mean_tensor.shape)
 651 
 652   return _BatchNormMatch(
 653       layer_op=None,
 654       bn_op=None,
 655       output_tensor=None,
 656       input_tensor=None,
 657       weight_tensor=None,
 658       gamma_tensor=gamma_tensor,
 659       beta_tensor=None,
 660       mean_tensor=batch_mean_tensor,
 661       variance_tensor=batch_variance_tensor,
 662       moving_mean_tensor=moving_mean_tensor,
 663       moving_variance_tensor=moving_variance_tensor,
 664       bn_decay_mean_tensor=bn_decay_mean_tensor,
 665       bn_decay_var_tensor=bn_decay_var_tensor,
 666       batch_epsilon=batch_epsilon,
 667       batch_to_space_op=None)

 670 def _CreateFoldedOp(graph, context, has_scaling, freeze_batch_norm_delay,
 671                     is_training):
 672   """Folds in batch norm layer into preceding convolution or FC layer.
 673 
 674   Creates 3 new nodes, connects their inputs and adds them to the graph:
 675   mul is cloned into mul_fold, Conv2D or MatMul, or DepthwiseConv2d is cloned
 676   into respective *_Fold, add is cloned into add_fold.
 677 
 678   Args:
 679     graph: Graph to modify.
 680     context: String, batch norm context, i.e. node into which BatchNorm is
 681       nested.
 682     has_scaling: Whether the batch norm has scaling enabled.
 683     freeze_batch_norm_delay: How many steps to wait before freezing moving mean
 684       and variance and using them for batch normalization.
 685     is_training: Bool, true if training.
 686 
 687   Raises:
 688     ValueError: When operation type is not supported, or input and output tensor
 689       shapes mismatch for created operations: mul_fold, add_fold.
 690 
 691   Returns:
 692     A pair of Operations, the first is the original consumer node of the batch
 693       norm (../BatchNorm/batchnorm_1/add_1), the second is the consumer node of
 694       the folded graph (add_fold).
 695   """
 696   mul_scale_name = 'mul_1' if has_scaling else 'mul'
 697   mul_scale = graph.get_operation_by_name(context + 'BatchNorm/batchnorm_1/' +
 698                                           mul_scale_name)
 699   op_below = mul_scale.inputs[0].op
 700   # Skip over the BatchToSpace operation in the case of atrous convolutions.
 701   batch_to_space_op = None
 702   if op_below.type == 'BatchToSpaceND':
 703     batch_to_space_op = op_below
 704     op_below = op_below.inputs[0].op
 705   weights = op_below.inputs[1]
 706   match = _GetBatchNormParams(
 707       graph=graph, context=context, has_scaling=has_scaling)
 708   correction_scale, correction_recip, correction_offset = None, None, None
 709   if is_training:
 710     correction_scale, correction_recip, correction_offset = (
 711         _ComputeBatchNormCorrections(
 712             context=context,
 713             match=match,
 714             freeze_batch_norm_delay=freeze_batch_norm_delay))
 715   # Special handling for weights of depthwise convolution.
 716   if op_below.type == 'DepthwiseConv2dNative':
 717     new_shape = [
 718         weights.get_shape().as_list()[2],
 719         weights.get_shape().as_list()[3]
 720     ]
 721     scale_name = 'mul' if has_scaling else 'Rsqrt'
 722     scale = graph.get_operation_by_name(context + 'BatchNorm/batchnorm_1/' +
 723                                         scale_name)
 724     scale = array_ops.reshape(scale.outputs[0], new_shape,
 725                               context + 'scale_reshape')
 726 
 727     if correction_scale is not None:
 728       correction_scale = array_ops.reshape(correction_scale, new_shape,
 729                                            context + 'correction_reshape')
 730       with ops.device(mul_scale.device):
 731         weights = math_ops.multiply(correction_scale, weights,
 732                                     context + 'correction_mult')
 733 
 734     mul_fold = _CloneOp(mul_scale, context + 'mul_fold', [(0, weights),
 735                                                           (1, scale)])
 736   elif op_below.type in ['Conv2D', 'MatMul']:
 737 
 738     if correction_scale is not None:
 739       with ops.device(mul_scale.device):
 740         weights = math_ops.multiply(correction_scale, weights,
 741                                     context + 'correction_mult')
 742     mul_fold = _CloneOp(mul_scale, context + 'mul_fold', [(0, weights)])
 743   else:
 744     raise ValueError('Cannot handle operation of type: %s' % op_below.type)
 745   _AssertShapesMatch('mul_fold', mul_fold.inputs[0], mul_fold.outputs[0])
 746 
 747   conv_or_fc_folded = _CloneOp(op_below, op_below.name + '_Fold',
 748                                [(1, mul_fold.outputs[0])])
 749 
 750   add_shift = graph.get_operation_by_name(context +
 751                                           'BatchNorm/batchnorm_1/add_1')
 752 
 753   corrected_output = conv_or_fc_folded.outputs[0]
 754   # Copy the batch to space operation if we have a atrous convolution.
 755   if batch_to_space_op:
 756     corrected_output = array_ops.batch_to_space_nd(
 757         corrected_output,
 758         batch_to_space_op.inputs[1],
 759         batch_to_space_op.inputs[2],
 760         name=batch_to_space_op.name + '_Fold')
 761   if correction_offset is not None:
 762     with ops.device(conv_or_fc_folded.device):
 763       corrected_output = math_ops.multiply(correction_recip, corrected_output,
 764                                            context + 'post_conv_mul')
 765       corrected_output = math_ops.add(corrected_output, (correction_offset),
 766                                       context + 'correction_add')
 767   add_fold = _CloneOp(add_shift, context + 'add_fold', [(0, corrected_output)])
 768   _AssertShapesMatch('add_fold', add_fold.inputs[0], add_fold.outputs[0])
 769   return add_shift, add_fold

 772 def _CloneOp(op, new_name, new_inputs):
 773   """Clones a given op, replaces its name and some of its inputs.
 774 
 775   Args:
 776     op: Operation to modify.
 777     new_name: String, a new name to set on cloned op.
 778     new_inputs: A list of tuples (idx, tensor), each input with corresponding
 779       index will be replaced by the given Tensor in the cloned op.
 780 
 781   Returns:
 782     Operation, the cloned op.
 783 
 784   Raises:
 785     TypeError: When Operation type is not supported.
 786     ValueError: When input shapes are incompatible.
 787   """
 788   inputs = list(op.inputs)
 789   for new_input in new_inputs:
 790     inputs[new_input[0]] = new_input[1]
 791   return _OP_CLONER.Clone(op, inputs, new_name)
 792 
 793 
 794 class _OpCloner(object):
 795   """Helper class that clones tf.Operations based on their type."""
 796 
 797   def __init__(self):
 798     self.op_type_to_action = {
 799         'Mul': self._CloneMul,
 800         'Add': self._CloneAdd,
 801         'Conv2D': self._CloneConv2d,
 802         'DepthwiseConv2dNative': self._CloneDepthwiseConv2d,
 803         'MatMul': self._CloneMatMul,
 804     }
 805 
 806   def _CloneMul(self, op, inputs, new_name):
 807     del op  # Unused.
 808     return math_ops.multiply(inputs[0], inputs[1], name=new_name).op
 809 
 810   def _CloneAdd(self, op, inputs, new_name):
 811     del op  # Unused.
 812     return math_ops.add(inputs[0], inputs[1], name=new_name).op
 813 
 814   def _CloneConv2d(self, op, inputs, new_name):
 815     input_tensor = inputs[0]
 816     weights = inputs[1]
 817     self._AssertConvShapes(op.name, input_tensor, weights)
 818     return nn_ops.conv2d(
 819         input_tensor,
 820         weights,
 821         strides=op.get_attr('strides'),
 822         padding=op.get_attr('padding'),
 823         use_cudnn_on_gpu=op.get_attr('use_cudnn_on_gpu'),
 824         data_format=op.get_attr('data_format'),
 825         name=new_name).op
 826 
 827   def _CloneDepthwiseConv2d(self, op, inputs, new_name):
 828     input_tensor = inputs[0]
 829     weights = inputs[1]
 830     self._AssertConvShapes(op.name, input_tensor, weights)
 831     return nn.depthwise_conv2d(
 832         input_tensor,
 833         weights,
 834         strides=op.get_attr('strides'),
 835         padding=op.get_attr('padding'),
 836         name=new_name).op
 837 
 838   def _CloneMatMul(self, op, inputs, new_name):
 839     weights = inputs[0]
 840     input_tensor = inputs[1]
 841     self._AssertFCShapes(op.name, weights, input_tensor)
 842     return math_ops.matmul(
 843         weights,
 844         input_tensor,
 845         transpose_a=op.get_attr('transpose_a'),
 846         transpose_b=op.get_attr('transpose_b'),
 847         name=new_name).op
 848 
 849   def Clone(self, op, inputs, new_name):
 850     try:
 851       return self.op_type_to_action[op.type](op, inputs, new_name)
 852     except KeyError:
 853       raise TypeError('Unsupported operation type: %s' % op.type)
 854 
 855   def _AssertConvShapes(self, op_name, input_tensor, weights):
 856     """Makes sure that convolution inputs have compatible shapes.
 857 
 858     Args:
 859       op_name: Operation name, only used in error message.
 860       input_tensor: Input that is convolved.
 861       weights: Weights of the convolution filter.
 862 
 863     Raises:
 864       ValueError: When input shapes are incompatible.
 865     """
 866     input_shape = input_tensor.get_shape()
 867     weights_shape = weights.get_shape()
 868     if (len(input_shape) != 4 or len(weights_shape) != 4 or
 869         input_shape[3] != weights_shape[2]):
 870       raise ValueError('Incompatible shapes for op %s inputs: %s and %s' %
 871                        (op_name, input_shape, weights_shape))
 872 
 873   def _AssertFCShapes(self, op_name, weights, input_tensor):
 874     """Makes sure that FC layer inputs have compatible shapes.
 875 
 876     Args:
 877       op_name: Operation name, only used in error message.
 878       weights: Weights used in FC layer.
 879       input_tensor: Input into FC layer.
 880 
 881     Raises:
 882       ValueError: When input shapes are incompatible.
 883     """
 884     weights_shape = weights.get_shape()
 885     input_shape = input_tensor.get_shape()
 886     if (len(weights_shape) != 2 or len(input_shape) != 2 or
 887         weights_shape[1] != input_shape[0]):
 888       raise ValueError('Incompatible shapes for op %s inputs: %s and %s' %
 889                        (op_name, weights_shape, input_shape))
 890 
 891 _OP_CLONER = _OpCloner()
 892

 894 def _AssertShapesMatch(op_name, in_tensor, out_tensor):
 895   """Makes sure that shapes of input and output tensors are compatible.
 896 
 897   Args:
 898     op_name: String, operation name, only used in error message.
 899     in_tensor: Tensor, input tensor.
 900     out_tensor: Tensor, output tensor.
 901 
 902   Raises:
 903     ValueError: When input and output tensors have different shapes.
 904   """
 905   in_shape = in_tensor.get_shape()
 906   out_shape = out_tensor.get_shape()
 907 
 908   if not in_shape.is_compatible_with(out_shape):
 909     raise ValueError('%s should not change tensor shape: input %s, '
 910                      'output %s' % (op_name, in_shape, out_shape))

 913 def _HasScaling(graph, input_to_ops_map, bn):
 914   r"""Checks if batch norm  has scaling enabled.
 915 
 916   Difference between batch norm with scaling and without is that with scaling:
 917 
 918   Rsqrt -> mul -> mul_1
 919               \-> mul_2
 920 
 921   where
 922     mul multiplies gamma by inverse square root of EMA of batch variance,
 923     mul_1 multiplies output of mul with output from the base operation
 924       (convolution, FC or depthwise convolution),
 925     mul_2 multiplies output of mul with EMA of batch mean,
 926   and without scaling:
 927 
 928   Rsqrt -> mul
 929        \-> mul_1
 930 
 931   where
 932     mul multiplies the inverse square root of EMA of batch variance with output
 933       from the base operation,
 934     mul_1 multiplies inverse square root of EMA of batch variance with EMA
 935       of batch mean.
 936 
 937   Args:
 938     graph: Graph to inspect.
 939     input_to_ops_map: InputToOps object containing mapping from tensor's name
 940       to ops that take it as input.
 941     bn: Batch norm layer prefix string.
 942 
 943   Returns:
 944     A boolean indicating whether this batch norm layer has scaling enabled.
 945   """
 946   rsqrt_op = graph.get_operation_by_name(bn + 'BatchNorm/batchnorm_1/Rsqrt')
 947   rsqrt_consumers = input_to_ops_map.ConsumerOperations(rsqrt_op)
 948 
 949   return sum(1 for op in rsqrt_consumers if op.type == 'Mul') == 1
