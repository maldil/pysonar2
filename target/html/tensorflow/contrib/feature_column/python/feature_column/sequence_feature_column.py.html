  39 def sequence_input_layer(
  40     features,
  41     feature_columns,
  42     weight_collections=None,
  43     trainable=True):
  44   """"Builds input layer for sequence input.
  45 
  46   All `feature_columns` must be sequence dense columns with the same
  47   `sequence_length`. The output of this method can be fed into sequence
  48   networks, such as RNN.
  49 
  50   The output of this method is a 3D `Tensor` of shape `[batch_size, T, D]`.
  51   `T` is the maximum sequence length for this batch, which could differ from
  52   batch to batch.
  53 
  54   If multiple `feature_columns` are given with `Di` `num_elements` each, their
  55   outputs are concatenated. So, the final `Tensor` has shape
  56   `[batch_size, T, D0 + D1 + ... + Dn]`.
  57 
  58   Example:
  59 
  60   ```python
  61   rating = sequence_numeric_column('rating')
  62   watches = sequence_categorical_column_with_identity(
  63       'watches', num_buckets=1000)
  64   watches_embedding = embedding_column(watches, dimension=10)
  65   columns = [rating, watches]
  66 
  67   features = tf.parse_example(..., features=make_parse_example_spec(columns))
  68   input_layer, sequence_length = sequence_input_layer(features, columns)
  69 
  70   rnn_cell = tf.nn.rnn_cell.BasicRNNCell(hidden_size)
  71   outputs, state = tf.nn.dynamic_rnn(
  72       rnn_cell, inputs=input_layer, sequence_length=sequence_length)
  73   ```
  74 
  75   Args:
  76     features: A dict mapping keys to tensors.
  77     feature_columns: An iterable of dense sequence columns. Valid columns are
  78       - `embedding_column` that wraps a `sequence_categorical_column_with_*`
  79       - `sequence_numeric_column`.
  80     weight_collections: A list of collection names to which the Variable will be
  81       added. Note that variables will also be added to collections
  82       `tf.GraphKeys.GLOBAL_VARIABLES` and `ops.GraphKeys.MODEL_VARIABLES`.
  83     trainable: If `True` also add the variable to the graph collection
  84       `GraphKeys.TRAINABLE_VARIABLES`.
  85 
  86   Returns:
  87     An `(input_layer, sequence_length)` tuple where:
  88     - input_layer: A float `Tensor` of shape `[batch_size, T, D]`.
  89         `T` is the maximum sequence length for this batch, which could differ
  90         from batch to batch. `D` is the sum of `num_elements` for all
  91         `feature_columns`.
  92     - sequence_length: An int `Tensor` of shape `[batch_size]`. The sequence
  93         length for each example.
  94 
  95   Raises:
  96     ValueError: If any of the `feature_columns` is the wrong type.
  97   """
  98   feature_columns = fc._normalize_feature_columns(feature_columns)
  99   for c in feature_columns:
 100     if not isinstance(c, fc._SequenceDenseColumn):
 101       raise ValueError(
 102           'All feature_columns must be of type _SequenceDenseColumn. '
 103           'You can wrap a sequence_categorical_column with an embedding_column '
 104           'or indicator_column. '
 105           'Given (type {}): {}'.format(type(c), c))
 106 
 107   with variable_scope.variable_scope(
 108       None, default_name='sequence_input_layer', values=features.values()):
 109     builder = fc._LazyBuilder(features)
 110     output_tensors = []
 111     sequence_lengths = []
 112     ordered_columns = []
 113     for column in sorted(feature_columns, key=lambda x: x.name):
 114       ordered_columns.append(column)
 115       with variable_scope.variable_scope(
 116           None, default_name=column._var_scope_name):
 117         dense_tensor, sequence_length = column._get_sequence_dense_tensor(
 118             builder,
 119             weight_collections=weight_collections,
 120             trainable=trainable)
 121         # Flattens the final dimension to produce a 3D Tensor.
 122         num_elements = column._variable_shape.num_elements()
 123         shape = array_ops.shape(dense_tensor)
 124         output_tensors.append(
 125             array_ops.reshape(
 126                 dense_tensor,
 127                 shape=array_ops.concat([shape[:2], [num_elements]], axis=0)))
 128         sequence_lengths.append(sequence_length)
 129     fc._verify_static_batch_size_equality(output_tensors, ordered_columns)
 130     fc._verify_static_batch_size_equality(sequence_lengths, ordered_columns)
 131     sequence_length = _assert_all_equal_and_return(sequence_lengths)
 132     return array_ops.concat(output_tensors, -1), sequence_length

 135 def sequence_categorical_column_with_identity(
 136     key, num_buckets, default_value=None):
 137   """Returns a feature column that represents sequences of integers.
 138 
 139   Pass this to `embedding_column` or `indicator_column` to convert sequence
 140   categorical data into dense representation for input to sequence NN, such as
 141   RNN.
 142 
 143   Example:
 144 
 145   ```python
 146   watches = sequence_categorical_column_with_identity(
 147       'watches', num_buckets=1000)
 148   watches_embedding = embedding_column(watches, dimension=10)
 149   columns = [watches_embedding]
 150 
 151   features = tf.parse_example(..., features=make_parse_example_spec(columns))
 152   input_layer, sequence_length = sequence_input_layer(features, columns)
 153 
 154   rnn_cell = tf.nn.rnn_cell.BasicRNNCell(hidden_size)
 155   outputs, state = tf.nn.dynamic_rnn(
 156       rnn_cell, inputs=input_layer, sequence_length=sequence_length)
 157   ```
 158 
 159   Args:
 160     key: A unique string identifying the input feature.
 161     num_buckets: Range of inputs. Namely, inputs are expected to be in the
 162       range `[0, num_buckets)`.
 163     default_value: If `None`, this column's graph operations will fail for
 164       out-of-range inputs. Otherwise, this value must be in the range
 165       `[0, num_buckets)`, and will replace out-of-range inputs.
 166 
 167   Returns:
 168     A `_SequenceCategoricalColumn`.
 169 
 170   Raises:
 171     ValueError: if `num_buckets` is less than one.
 172     ValueError: if `default_value` is not in range `[0, num_buckets)`.
 173   """
 174   return fc._SequenceCategoricalColumn(
 175       fc.categorical_column_with_identity(
 176           key=key,
 177           num_buckets=num_buckets,
 178           default_value=default_value))

 181 def sequence_categorical_column_with_hash_bucket(
 182     key, hash_bucket_size, dtype=dtypes.string):
 183   """A sequence of categorical terms where ids are set by hashing.
 184 
 185   Pass this to `embedding_column` or `indicator_column` to convert sequence
 186   categorical data into dense representation for input to sequence NN, such as
 187   RNN.
 188 
 189   Example:
 190 
 191   ```python
 192   tokens = sequence_categorical_column_with_hash_bucket(
 193       'tokens', hash_bucket_size=1000)
 194   tokens_embedding = embedding_column(tokens, dimension=10)
 195   columns = [tokens_embedding]
 196 
 197   features = tf.parse_example(..., features=make_parse_example_spec(columns))
 198   input_layer, sequence_length = sequence_input_layer(features, columns)
 199 
 200   rnn_cell = tf.nn.rnn_cell.BasicRNNCell(hidden_size)
 201   outputs, state = tf.nn.dynamic_rnn(
 202       rnn_cell, inputs=input_layer, sequence_length=sequence_length)
 203   ```
 204 
 205   Args:
 206     key: A unique string identifying the input feature.
 207     hash_bucket_size: An int > 1. The number of buckets.
 208     dtype: The type of features. Only string and integer types are supported.
 209 
 210   Returns:
 211     A `_SequenceCategoricalColumn`.
 212 
 213   Raises:
 214     ValueError: `hash_bucket_size` is not greater than 1.
 215     ValueError: `dtype` is neither string nor integer.
 216   """
 217   return fc._SequenceCategoricalColumn(
 218       fc.categorical_column_with_hash_bucket(
 219           key=key,
 220           hash_bucket_size=hash_bucket_size,
 221           dtype=dtype))

 224 def sequence_categorical_column_with_vocabulary_file(
 225     key, vocabulary_file, vocabulary_size=None, num_oov_buckets=0,
 226     default_value=None, dtype=dtypes.string):
 227   """A sequence of categorical terms where ids use a vocabulary file.
 228 
 229   Pass this to `embedding_column` or `indicator_column` to convert sequence
 230   categorical data into dense representation for input to sequence NN, such as
 231   RNN.
 232 
 233   Example:
 234 
 235   ```python
 236   states = sequence_categorical_column_with_vocabulary_file(
 237       key='states', vocabulary_file='/us/states.txt', vocabulary_size=50,
 238       num_oov_buckets=5)
 239   states_embedding = embedding_column(states, dimension=10)
 240   columns = [states_embedding]
 241 
 242   features = tf.parse_example(..., features=make_parse_example_spec(columns))
 243   input_layer, sequence_length = sequence_input_layer(features, columns)
 244 
 245   rnn_cell = tf.nn.rnn_cell.BasicRNNCell(hidden_size)
 246   outputs, state = tf.nn.dynamic_rnn(
 247       rnn_cell, inputs=input_layer, sequence_length=sequence_length)
 248   ```
 249 
 250   Args:
 251     key: A unique string identifying the input feature.
 252     vocabulary_file: The vocabulary file name.
 253     vocabulary_size: Number of the elements in the vocabulary. This must be no
 254       greater than length of `vocabulary_file`, if less than length, later
 255       values are ignored. If None, it is set to the length of `vocabulary_file`.
 256     num_oov_buckets: Non-negative integer, the number of out-of-vocabulary
 257       buckets. All out-of-vocabulary inputs will be assigned IDs in the range
 258       `[vocabulary_size, vocabulary_size+num_oov_buckets)` based on a hash of
 259       the input value. A positive `num_oov_buckets` can not be specified with
 260       `default_value`.
 261     default_value: The integer ID value to return for out-of-vocabulary feature
 262       values, defaults to `-1`. This can not be specified with a positive
 263       `num_oov_buckets`.
 264     dtype: The type of features. Only string and integer types are supported.
 265 
 266   Returns:
 267     A `_SequenceCategoricalColumn`.
 268 
 269   Raises:
 270     ValueError: `vocabulary_file` is missing or cannot be opened.
 271     ValueError: `vocabulary_size` is missing or < 1.
 272     ValueError: `num_oov_buckets` is a negative integer.
 273     ValueError: `num_oov_buckets` and `default_value` are both specified.
 274     ValueError: `dtype` is neither string nor integer.
 275   """
 276   return fc._SequenceCategoricalColumn(
 277       fc.categorical_column_with_vocabulary_file(
 278           key=key,
 279           vocabulary_file=vocabulary_file,
 280           vocabulary_size=vocabulary_size,
 281           num_oov_buckets=num_oov_buckets,
 282           default_value=default_value,
 283           dtype=dtype))

 286 def sequence_categorical_column_with_vocabulary_list(
 287     key, vocabulary_list, dtype=None, default_value=-1, num_oov_buckets=0):
 288   """A sequence of categorical terms where ids use an in-memory list.
 289 
 290   Pass this to `embedding_column` or `indicator_column` to convert sequence
 291   categorical data into dense representation for input to sequence NN, such as
 292   RNN.
 293 
 294   Example:
 295 
 296   ```python
 297   colors = sequence_categorical_column_with_vocabulary_list(
 298       key='colors', vocabulary_list=('R', 'G', 'B', 'Y'),
 299       num_oov_buckets=2)
 300   colors_embedding = embedding_column(colors, dimension=3)
 301   columns = [colors_embedding]
 302 
 303   features = tf.parse_example(..., features=make_parse_example_spec(columns))
 304   input_layer, sequence_length = sequence_input_layer(features, columns)
 305 
 306   rnn_cell = tf.nn.rnn_cell.BasicRNNCell(hidden_size)
 307   outputs, state = tf.nn.dynamic_rnn(
 308       rnn_cell, inputs=input_layer, sequence_length=sequence_length)
 309   ```
 310 
 311   Args:
 312     key: A unique string identifying the input feature.
 313     vocabulary_list: An ordered iterable defining the vocabulary. Each feature
 314       is mapped to the index of its value (if present) in `vocabulary_list`.
 315       Must be castable to `dtype`.
 316     dtype: The type of features. Only string and integer types are supported.
 317       If `None`, it will be inferred from `vocabulary_list`.
 318     default_value: The integer ID value to return for out-of-vocabulary feature
 319       values, defaults to `-1`. This can not be specified with a positive
 320       `num_oov_buckets`.
 321     num_oov_buckets: Non-negative integer, the number of out-of-vocabulary
 322       buckets. All out-of-vocabulary inputs will be assigned IDs in the range
 323       `[len(vocabulary_list), len(vocabulary_list)+num_oov_buckets)` based on a
 324       hash of the input value. A positive `num_oov_buckets` can not be specified
 325       with `default_value`.
 326 
 327   Returns:
 328     A `_SequenceCategoricalColumn`.
 329 
 330   Raises:
 331     ValueError: if `vocabulary_list` is empty, or contains duplicate keys.
 332     ValueError: `num_oov_buckets` is a negative integer.
 333     ValueError: `num_oov_buckets` and `default_value` are both specified.
 334     ValueError: if `dtype` is not integer or string.
 335   """
 336   return fc._SequenceCategoricalColumn(
 337       fc.categorical_column_with_vocabulary_list(
 338           key=key,
 339           vocabulary_list=vocabulary_list,
 340           dtype=dtype,
 341           default_value=default_value,
 342           num_oov_buckets=num_oov_buckets))

 345 def sequence_numeric_column(
 346     key,
 347     shape=(1,),
 348     default_value=0.,
 349     dtype=dtypes.float32,
 350     normalizer_fn=None):
 351   """Returns a feature column that represents sequences of numeric data.
 352 
 353   Example:
 354 
 355   ```python
 356   temperature = sequence_numeric_column('temperature')
 357   columns = [temperature]
 358 
 359   features = tf.parse_example(..., features=make_parse_example_spec(columns))
 360   input_layer, sequence_length = sequence_input_layer(features, columns)
 361 
 362   rnn_cell = tf.nn.rnn_cell.BasicRNNCell(hidden_size)
 363   outputs, state = tf.nn.dynamic_rnn(
 364       rnn_cell, inputs=input_layer, sequence_length=sequence_length)
 365   ```
 366 
 367   Args:
 368     key: A unique string identifying the input features.
 369     shape: The shape of the input data per sequence id. E.g. if `shape=(2,)`,
 370       each example must contain `2 * sequence_length` values.
 371     default_value: A single value compatible with `dtype` that is used for
 372       padding the sparse data into a dense `Tensor`.
 373     dtype: The type of values.
 374     normalizer_fn: If not `None`, a function that can be used to normalize the
 375       value of the tensor after `default_value` is applied for parsing.
 376       Normalizer function takes the input `Tensor` as its argument, and returns
 377       the output `Tensor`. (e.g. lambda x: (x - 3.0) / 4.2). Please note that
 378       even though the most common use case of this function is normalization, it
 379       can be used for any kind of Tensorflow transformations.
 380 
 381   Returns:
 382     A `_SequenceNumericColumn`.
 383 
 384   Raises:
 385     TypeError: if any dimension in shape is not an int.
 386     ValueError: if any dimension in shape is not a positive integer.
 387     ValueError: if `dtype` is not convertible to `tf.float32`.
 388   """
 389   shape = fc._check_shape(shape=shape, key=key)
 390   if not (dtype.is_integer or dtype.is_floating):
 391     raise ValueError('dtype must be convertible to float. '
 392                      'dtype: {}, key: {}'.format(dtype, key))
 393   if normalizer_fn is not None and not callable(normalizer_fn):
 394     raise TypeError(
 395         'normalizer_fn must be a callable. Given: {}'.format(normalizer_fn))
 396 
 397   return _SequenceNumericColumn(
 398       key,
 399       shape=shape,
 400       default_value=default_value,
 401       dtype=dtype,
 402       normalizer_fn=normalizer_fn)

 405 def _assert_all_equal_and_return(tensors, name=None):
 406   """Asserts that all tensors are equal and returns the first one."""
 407   with ops.name_scope(name, 'assert_all_equal', values=tensors):
 408     if len(tensors) == 1:
 409       return tensors[0]
 410     assert_equal_ops = []
 411     for t in tensors[1:]:
 412       assert_equal_ops.append(check_ops.assert_equal(tensors[0], t))
 413     with ops.control_dependencies(assert_equal_ops):
 414       return array_ops.identity(tensors[0])

 417 class _SequenceNumericColumn(
 418     fc._SequenceDenseColumn,
 419     collections.namedtuple(
 420         '_SequenceNumericColumn',
 421         ['key', 'shape', 'default_value', 'dtype', 'normalizer_fn'])):
 422   """Represents sequences of numeric data."""
 423 
 424   @property
 425   def name(self):
 426     return self.key
 427 
 428   @property
 429   def _parse_example_spec(self):
 430     return {self.key: parsing_ops.VarLenFeature(self.dtype)}
 431 
 432   def _transform_feature(self, inputs):
 433     input_tensor = inputs.get(self.key)
 434     if self.normalizer_fn is not None:
 435       input_tensor = self.normalizer_fn(input_tensor)
 436     return input_tensor
 437 
 438   @property
 439   def _variable_shape(self):
 440     return tensor_shape.TensorShape(self.shape)
 441 
 442   def _get_sequence_dense_tensor(
 443       self, inputs, weight_collections=None, trainable=None):
 444     # Do nothing with weight_collections and trainable since no variables are
 445     # created in this function.
 446     del weight_collections
 447     del trainable
 448     sp_tensor = inputs.get(self)
 449     dense_tensor = sparse_ops.sparse_tensor_to_dense(
 450         sp_tensor, default_value=self.default_value)
 451     # Reshape into [batch_size, T, variable_shape].
 452     dense_shape = array_ops.concat(
 453         [array_ops.shape(dense_tensor)[:1], [-1], self._variable_shape],
 454         axis=0)
 455     dense_tensor = array_ops.reshape(dense_tensor, shape=dense_shape)
 456     sequence_length = fc._sequence_length_from_sparse_tensor(
 457         sp_tensor, num_elements=self._variable_shape.num_elements())
 458     return fc._SequenceDenseColumn.TensorSequenceLengthPair(
 459         dense_tensor=dense_tensor, sequence_length=sequence_length)
