  26 class ClusterResolver(object):
  27   """Abstract class for all implementations of ClusterResolvers.
  28 
  29   This defines the skeleton for all implementations of ClusterResolvers.
  30   ClusterResolvers are a way for TensorFlow to communicate with various cluster
  31   management systems (e.g. GCE, AWS, etc...).
  32 
  33   By letting TensorFlow communicate with these systems, we will be able to
  34   automatically discover and resolve IP addresses for various TensorFlow
  35   workers. This will eventually allow us to automatically recover from
  36   underlying machine failures and scale TensorFlow worker clusters up and down.
  37   """
  38 
  39   @abc.abstractmethod
  40   def cluster_spec(self):
  41     """Retrieve the current state of the cluster and returns a ClusterSpec.
  42 
  43     Returns:
  44       A ClusterSpec representing the state of the cluster at the moment this
  45       function is called.
  46 
  47     Implementors of this function must take care in ensuring that the
  48     ClusterSpec returned is up-to-date at the time of calling this function.
  49     This usually means retrieving the information from the underlying cluster
  50     management system every time this function is invoked and reconstructing
  51     a cluster_spec, rather than attempting to cache anything.
  52     """
  53     raise NotImplementedError(
  54         'cluster_spec is not implemented for {}.'.format(self))
  55 
  56   @abc.abstractmethod
  57   def master(self):
  58     """..."""
  59     raise NotImplementedError('master is not implemented for {}.'.format(self))
  60

  62 class SimpleClusterResolver(ClusterResolver):
  63   """Simple implementation of ClusterResolver that accepts a ClusterSpec."""
  64 
  65   def __init__(self, cluster_spec, master=''):
  66     """Creates a SimpleClusterResolver from a ClusterSpec."""
  67     super(SimpleClusterResolver, self).__init__()
  68 
  69     if not isinstance(cluster_spec, ClusterSpec):
  70       raise TypeError('cluster_spec must be a ClusterSpec.')
  71     self._cluster_spec = cluster_spec
  72 
  73     if not isinstance(master, str):
  74       raise TypeError('master must be a string.')
  75     self._master = master
  76 
  77   def cluster_spec(self):
  78     """Returns the ClusterSpec passed into the constructor."""
  79     return self._cluster_spec
  80 
  81   def master(self):
  82     """Returns the master address to use when creating a session."""
  83     return self._master

  86 class UnionClusterResolver(ClusterResolver):
  87   """Performs a union on underlying ClusterResolvers.
  88 
  89   This class performs a union given two or more existing ClusterResolvers. It
  90   merges the underlying ClusterResolvers, and returns one unified ClusterSpec
  91   when cluster_spec is called. The details of the merge function is
  92   documented in the cluster_spec function.
  93   """
  94 
  95   def __init__(self, *args):
  96     """Initializes a UnionClusterResolver with other ClusterResolvers.
  97 
  98     Args:
  99       *args: `ClusterResolver` objects to be unionized.
 100 
 101     Raises:
 102       TypeError: If any argument is not a subclass of `ClusterResolvers`.
 103       ValueError: If there are no arguments passed.
 104     """
 105     super(UnionClusterResolver, self).__init__()
 106 
 107     if not args:
 108       raise ValueError('At least one ClusterResolver is required.')
 109 
 110     for cluster_resolver in args:
 111       if not isinstance(cluster_resolver, ClusterResolver):
 112         raise TypeError('All arguments must be a sub-class of '
 113                         '`ClusterResolver.`')
 114     self._cluster_resolvers = args
 115 
 116   def cluster_spec(self):
 117     """Returns a union of all the ClusterSpecs from the ClusterResolvers.
 118 
 119     Returns:
 120       A ClusterSpec containing host information merged from all the underlying
 121       ClusterResolvers.
 122 
 123     Raises:
 124       KeyError: If there are conflicting keys detected when merging two or
 125       more dictionaries, this exception is raised.
 126 
 127     Note: If there are multiple ClusterResolvers exposing ClusterSpecs with the
 128     same job name, we will merge the list/dict of workers.
 129 
 130     If *all* underlying ClusterSpecs expose the set of workers as lists, we will
 131     concatenate the lists of workers, starting with the list of workers from
 132     the first ClusterResolver passed into the constructor.
 133 
 134     If *any* of the ClusterSpecs expose the set of workers as a dict, we will
 135     treat all the sets of workers as dicts (even if they are returned as lists)
 136     and will only merge them into a dict if there is no conflicting keys. If
 137     there is a conflicting key, we will raise a `KeyError`.
 138     """
 139 
 140     merged_cluster = {}
 141 
 142     # We figure out whether it is all lists for a particular job, or whether
 143     # there are dicts inside.
 144     for cluster_resolver in self._cluster_resolvers:
 145       cluster_spec = cluster_resolver.cluster_spec()
 146       cluster_dict = cluster_spec.as_dict()
 147 
 148       for job_name, tasks in cluster_dict.items():
 149         if job_name in merged_cluster:
 150           # If we see a dict, then we write a dict out regardless.
 151           if isinstance(tasks, dict):
 152             merged_cluster[job_name] = {}
 153         else:
 154           # We take whichever type is present.
 155           if isinstance(tasks, list):
 156             merged_cluster[job_name] = []
 157           else:
 158             merged_cluster[job_name] = {}
 159 
 160     # We then do the merge as appropriate in merged_cluster[job].
 161     for cluster_resolver in self._cluster_resolvers:
 162       cluster_spec = cluster_resolver.cluster_spec()
 163       cluster_dict = cluster_spec.as_dict()
 164 
 165       for job_name, tasks in cluster_dict.items():
 166         if isinstance(merged_cluster[job_name], list):
 167           # We all have lists, we can just concatenate and be done.
 168           merged_cluster[job_name].extend(tasks)
 169         else:
 170           if isinstance(tasks, list):
 171             # We convert to a dictionary if the type is a list.
 172             task_dict = dict(zip(range(0, len(tasks)), tasks))
 173           else:
 174             # We can simply make a copy (for update) and be done.
 175             task_dict = tasks.copy()
 176 
 177           # We detect if there are duplicates, and raise an error if so.
 178           task_keys = set(task_dict)
 179           merged_keys = set(merged_cluster[job_name].keys())
 180           intersected_keys = task_keys.intersection(merged_keys)
 181           if intersected_keys:
 182             raise KeyError('Duplicate keys detected when merging two '
 183                            'ClusterSpecs: %s' % repr(intersected_keys))
 184 
 185           # We do the merge after all the processing.
 186           merged_cluster[job_name].update(task_dict)
 187 
 188     return ClusterSpec(merged_cluster)
 189 
 190   def master(self):
 191     """master returns the master address from the first cluster resolver."""
 192     return self._cluster_resolvers[0].master()
