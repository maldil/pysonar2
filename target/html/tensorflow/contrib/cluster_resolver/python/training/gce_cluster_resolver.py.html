  33 class GceClusterResolver(ClusterResolver):
  34   """Cluster Resolver for Google Compute Engine.
  35 
  36   This is an implementation of cluster resolvers for the Google Compute Engine
  37   instance group platform. By specifying a project, zone, and instance group,
  38   this will retrieve the IP address of all the instances within the instance
  39   group and return a Cluster Resolver object suitable for use for distributed
  40   TensorFlow.
  41   """
  42 
  43   def __init__(self,
  44                project,
  45                zone,
  46                instance_group,
  47                port,
  48                job_name='worker',
  49                credentials='default',
  50                service=None):
  51     """Creates a new GceClusterResolver object.
  52 
  53     This takes in a few parameters and creates a GceClusterResolver project. It
  54     will then use these parameters to query the GCE API for the IP addresses of
  55     each instance in the instance group.
  56 
  57     Args:
  58       project: Name of the GCE project
  59       zone: Zone of the GCE instance group
  60       instance_group: Name of the GCE instance group
  61       port: Port of the listening TensorFlow server (default: 8470)
  62       job_name: Name of the TensorFlow job this set of instances belongs to
  63       credentials: GCE Credentials. If nothing is specified, this defaults to
  64         GoogleCredentials.get_application_default()
  65       service: The GCE API object returned by the googleapiclient.discovery
  66         function. (Default: discovery.build('compute', 'v1')). If you specify a
  67         custom service object, then the credentials parameter will be ignored.
  68 
  69     Raises:
  70       ImportError: If the googleapiclient is not installed.
  71     """
  72     self._project = project
  73     self._zone = zone
  74     self._instance_group = instance_group
  75     self._job_name = job_name
  76     self._port = port
  77     self._credentials = credentials
  78 
  79     if credentials == 'default':
  80       if _GOOGLE_API_CLIENT_INSTALLED:
  81         self._credentials = GoogleCredentials.get_application_default()
  82 
  83     if service is None:
  84       if not _GOOGLE_API_CLIENT_INSTALLED:
  85         raise ImportError('googleapiclient must be installed before using the '
  86                           'GCE cluster resolver')
  87       self._service = discovery.build(
  88           'compute', 'v1',
  89           credentials=self._credentials)
  90     else:
  91       self._service = service
  92 
  93   def cluster_spec(self):
  94     """Returns a ClusterSpec object based on the latest instance group info.
  95 
  96     This returns a ClusterSpec object for use based on information from the
  97     specified instance group. We will retrieve the information from the GCE APIs
  98     every time this method is called.
  99 
 100     Returns:
 101       A ClusterSpec containing host information retrieved from GCE.
 102     """
 103     request_body = {'instanceState': 'RUNNING'}
 104     request = self._service.instanceGroups().listInstances(
 105         project=self._project,
 106         zone=self._zone,
 107         instanceGroups=self._instance_group,
 108         body=request_body,
 109         orderBy='name')
 110 
 111     worker_list = []
 112 
 113     while request is not None:
 114       response = request.execute()
 115 
 116       items = response['items']
 117       for instance in items:
 118         instance_name = instance['instance'].split('/')[-1]
 119 
 120         instance_request = self._service.instances().get(
 121             project=self._project,
 122             zone=self._zone,
 123             instance=instance_name)
 124 
 125         if instance_request is not None:
 126           instance_details = instance_request.execute()
 127           ip_address = instance_details['networkInterfaces'][0]['networkIP']
 128           instance_url = '%s:%s' % (ip_address, self._port)
 129           worker_list.append(instance_url)
 130 
 131       request = self._service.instanceGroups().listInstances_next(
 132           previous_request=request,
 133           previous_response=response)
 134 
 135     worker_list.sort()
 136     return ClusterSpec({self._job_name: worker_list})
