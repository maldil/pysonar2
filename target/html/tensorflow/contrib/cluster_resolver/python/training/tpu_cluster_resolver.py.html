  44 class TPUClusterResolver(ClusterResolver):
  45   """Cluster Resolver for Google Cloud TPUs.
  46 
  47   This is an implementation of cluster resolvers for the Google Cloud TPU
  48   service. As Cloud TPUs are in alpha, you will need to specify a API definition
  49   file for this to consume, in addition to a list of Cloud TPUs in your Google
  50   Cloud Platform project.
  51   """
  52 
  53   def _requestComputeMetadata(self, path):
  54     req = Request('http://metadata/computeMetadata/v1/%s' % path,
  55                   headers={'Metadata-Flavor': 'Google'})
  56     resp = urlopen(req)
  57     return compat.as_bytes(resp.read())
  58 
  59   def _shouldResolve(self):
  60     if (self._tpu == compat.as_bytes('') or
  61         self._tpu == compat.as_bytes('local') or
  62         self._tpu.startswith(compat.as_bytes('/bns')) or
  63         self._tpu.startswith(compat.as_bytes('grpc://'))):
  64       return False
  65     return True
  66 
  67   @staticmethod
  68   def _inGke():
  69     """When running in GKE, the environment variable will be set."""
  70     return _GKE_ENV_VARIABLE in os.environ
  71 
  72   @staticmethod
  73   def _gkeEndpoints():
  74     return os.environ[_GKE_ENV_VARIABLE]

  76   @staticmethod
  77   def _envVarFallback():
  78     if _DEFAULT_ENV_VARIABLE in os.environ:
  79       return os.environ[_DEFAULT_ENV_VARIABLE]
  80     return None

  86   def __init__(self,
  87                tpu=None,
  88                zone=None,
  89                project=None,
  90                job_name='worker',
  91                coordinator_name=None,
  92                coordinator_address=None,
  93                credentials='default',
  94                service=None,
  95                discovery_url=None):
  96     """Creates a new TPUClusterResolver object.
  97 
  98     The ClusterResolver will then use the parameters to query the Cloud TPU APIs
  99     for the IP addresses and ports of each Cloud TPU listed.
 100 
 101     Args:
 102       tpu: Either a string, or a list of strings corresponding to the TPUs to
 103         use. If the single string is the empty string, the string 'local', or a
 104         string that begins with 'grpc://' or '/bns', then it is assumed to not
 105         correspond with a Cloud TPU and will instead be passed as the session
 106         master and no ClusterSpec propagation will be done.
 107       zone: Zone where the TPUs are located. If omitted or empty, we will assume
 108         that the zone of the TPU is the same as the zone of the GCE VM, which we
 109         will try to discover from the GCE metadata service.
 110       project: Name of the GCP project containing Cloud TPUs. If omitted or
 111         empty, we will try to discover the project name of the GCE VM from the
 112         GCE metadata service.
 113       job_name: Name of the TensorFlow job the TPUs belong to.
 114       coordinator_name: The name to use for the coordinator. Set to None if the
 115         coordinator should not be included in the computed ClusterSpec.
 116       coordinator_address: The address of the coordinator (typically an ip:port
 117         pair). If set to None, a TF server will be started. If coordinator_name
 118         is None, a TF server will not be started even if coordinator_address is
 119         None.
 120       credentials: GCE Credentials. If None, then we use default credentials
 121         from the oauth2client
 122       service: The GCE API object returned by the googleapiclient.discovery
 123         function. If you specify a custom service object, then the credentials
 124         parameter will be ignored.
 125       discovery_url: A URL template that points to the location of
 126         the discovery service. It should have two parameters {api} and
 127         {apiVersion} that when filled in produce an absolute URL to the
 128         discovery document for that service. The environment variable
 129         'TPU_API_DISCOVERY_URL' will override this.
 130 
 131     Raises:
 132       ImportError: If the googleapiclient is not installed.
 133       ValueError: If no TPUs are specified.
 134     """
 135     if isinstance(tpu, list):
 136       if not tpu:
 137         raise ValueError('At least one TPU must be specified.')
 138       if len(tpu) != 1:
 139         raise NotImplementedError(
 140             'Using multiple TPUs in a single session is not yet implemented')
 141       tpu = tpu[0]
 142 
 143     in_gke = self._inGke()
 144     # When using GKE with Cloud TPUs, the env variable will be set.
 145     if tpu is None:
 146       if in_gke:
 147         tpu = self._gkeEndpoints()
 148       else:
 149         tpu = self._envVarFallback()
 150 
 151     if tpu is None:
 152       raise ValueError('Please provide a TPU Name to connect to.')
 153 
 154     self._tpu = compat.as_bytes(tpu)  # self._tpu is always bytes
 155     self._job_name = job_name
 156     self._credentials = credentials
 157 
 158     should_resolve = self._shouldResolve()
 159 
 160     if not project and should_resolve:
 161       project = compat.as_str(
 162           self._requestComputeMetadata('project/project-id'))
 163 
 164     if not zone and should_resolve:
 165       zone_path = compat.as_str(self._requestComputeMetadata('instance/zone'))
 166       zone = zone_path.split('/')[-1]
 167 
 168     self._project = project
 169     self._zone = zone
 170 
 171     if credentials == 'default' and should_resolve:
 172       if _GOOGLE_API_CLIENT_INSTALLED:
 173         self._credentials = GoogleCredentials.get_application_default()
 174 
 175     if service is None and should_resolve:
 176       if not _GOOGLE_API_CLIENT_INSTALLED:
 177         raise ImportError('googleapiclient and oauth2client must be installed '
 178                           'before using the TPU cluster resolver. Execute: '
 179                           '`pip install --upgrade google-api-python-client` '
 180                           'and `pip install --upgrade oauth2client` to '
 181                           'install with pip.')
 182 
 183       final_discovery_url = self._discoveryUrl() or discovery_url
 184       if final_discovery_url:
 185         self._service = discovery.build(
 186             'tpu', 'v1alpha1',
 187             credentials=self._credentials,
 188             discoveryServiceUrl=final_discovery_url)
 189       else:
 190         self._service = discovery.build(
 191             'tpu', 'v1alpha1',
 192             credentials=self._credentials)
 193     else:
 194       self._service = service
 195 
 196     self._coordinator_name = coordinator_name
 197     if coordinator_name and not coordinator_address and (should_resolve or
 198                                                          in_gke):
 199       self._start_local_server()
 200     else:
 201       self._coordinator_address = coordinator_address
 202 
 203   def master(self):
 204     """Get the Master string to be used for the session.
 205 
 206     In the normal case, this returns the grpc path (grpc://1.2.3.4:8470) of
 207     first instance in the ClusterSpec returned by the cluster_spec function.
 208 
 209     If a non-TPU name is used when constructing a TPUClusterResolver, that will
 210     be returned instead (e.g. If the tpus argument's value when constructing
 211     this TPUClusterResolver was 'grpc://10.240.1.2:8470',
 212     'grpc://10.240.1.2:8470' will be returned).
 213 
 214     Returns:
 215       string, the connection string to use when creating a session.
 216 
 217     Raises:
 218       ValueError: If none of the TPUs specified exists.
 219     """
 220     if not self._shouldResolve():
 221       return self._tpu.split(compat.as_bytes(_ENDPOINTS_SEPARATOR))[0]
 222 
 223     job_tasks = self.cluster_spec().job_tasks(self._job_name)
 224     if not job_tasks:
 225       raise ValueError('No TPUs exists with the specified names exist.')
 226 
 227     return 'grpc://' + job_tasks[0]
 228 
 229   def get_master(self):
 230     return self.master()
 231 
 232   def get_job_name(self):
 233     if self._shouldResolve():
 234       return self._job_name

 236   def cluster_spec(self):
 237     """Returns a ClusterSpec object based on the latest TPU information.
 238 
 239     We retrieve the information from the GCE APIs every time this method is
 240     called.
 241 
 242     Returns:
 243       A ClusterSpec containing host information returned from Cloud TPUs.
 244 
 245     Raises:
 246       RuntimeError: If the provided TPU is not healthy.
 247     """
 248     ############################################################################
 249     # There are 5 potential cases this code must handle:
 250     #  1. [Normal case.] We should resolve the TPU name to a set of tasks, and
 251     #      a. Create a ClusterSpec that includes the coordinator job
 252     #      b. Create a ClusterSpec without the coordinator job.
 253     #  2. [GKE / No API Access.] We should not resolve the TPU name to a set of
 254     #     tasks and
 255     #      a. Create a ClusterSpec with the coordinator
 256     #      b. Create a ClusterSpec without the coordinator
 257     #  3. [Other (legacy non-gRPC).] We should return an empty ClusterSpec.
 258     ############################################################################
 259 
 260     if self._shouldResolve():
 261       # Case 1.
 262       full_name = 'projects/%s/locations/%s/nodes/%s' % (
 263           self._project, self._zone, compat.as_text(self._tpu))
 264       request = self._service.projects().locations().nodes().get(name=full_name)
 265       response = request.execute()
 266 
 267       if 'state' in response and response['state'] != 'READY':
 268         raise RuntimeError('TPU "%s" is not yet ready; state: "%s"' %
 269                            (compat.as_text(self._tpu), response['state']))
 270 
 271       if 'health' in response and response['health'] != 'HEALTHY':
 272         raise RuntimeError('TPU "%s" is unhealthy: "%s"' %
 273                            (compat.as_text(self._tpu), response['health']))
 274 
 275       if 'networkEndpoints' in response:
 276         worker_list = [
 277             '%s:%s' % (endpoint['ipAddress'], endpoint['port'])
 278             for endpoint in response['networkEndpoints']
 279         ]
 280       else:
 281         # Fall back to the deprecated response format
 282         instance_url = '%s:%s' % (response['ipAddress'], response['port'])
 283         worker_list = [instance_url]
 284 
 285       cluster_spec = {self._job_name: worker_list}
 286     else:
 287       if not self._tpu.startswith(compat.as_bytes('grpc://')):
 288         # Case 3.
 289         return None
 290       # Case 2.
 291       cluster_spec = {
 292           self._job_name: [
 293               x[len(compat.as_bytes('grpc://')):]
 294               for x in self._tpu.split(compat.as_bytes(_ENDPOINTS_SEPARATOR))
 295           ]
 296       }
 297 
 298     if self._coordinator_address:
 299       # {1, 2}.a
 300       cluster_spec[self._coordinator_name] = [self._coordinator_address]
 301 
 302     return server_lib.ClusterSpec(cluster_spec)
 303 
 304   def _start_local_server(self):
 305     address = self._requestComputeMetadata('instance/network-interfaces/0/ip')
 306     self._server = server_lib.Server(
 307         {
 308             'local': ['0.0.0.0:0']
 309         }, protocol='grpc', config=None, start=True)
 310     # self._server.target is of the form: grpc://ipaddress:port
 311     target = compat.as_bytes(self._server.target)
 312     splits = target.split(compat.as_bytes(':'))
 313     assert len(splits) == 3, self._server.target
 314     assert splits[0] == compat.as_bytes('grpc'), self._server.target
 315     self._coordinator_port = compat.as_text(splits[2])
 316     self._coordinator_address = '%s:%s' % (
 317         address, compat.as_text(self._coordinator_port))
