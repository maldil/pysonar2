  25 # White-listed layer operations, which do not affect the receptive field
  26 # computation.
  27 _UNCHANGED_RF_LAYER_OPS = [
  28     "Add", "BiasAdd", "Cast", "Ceil", "ConcatV2", "Const", "Floor",
  29     "FusedBatchNorm", "Identity", "Log", "Mul", "Pow", "RealDiv", "Relu",
  30     "Relu6", "Round", "Rsqrt", "Softplus", "Sub", "VariableV2", "LRN"
  31 ]
  32 
  33 # Different ways in which padding modes may be spelled.
  34 _VALID_PADDING = ["VALID", b"VALID"]
  35 _SAME_PADDING = ["SAME", b"SAME"]

  38 def _stride_size(node, name_to_node):
  39   """Computes stride size given a TF node.
  40 
  41   Args:
  42     node: Tensorflow node (NodeDef proto).
  43     name_to_node: For MaxPoolV2, mapping from variable name Tensorflow node.
  44 
  45   Returns:
  46     stride_x: Stride size for horizontal direction (integer).
  47     stride_y: Stride size for vertical direction (integer).
  48 
  49   Raises:
  50     ValueError: If stride input cannot be found in `name_to_node`.
  51   """
  52   if node.op == "MaxPoolV2":
  53     strides_input_name = node.input[2]
  54     if not strides_input_name.endswith("/strides"):
  55       raise ValueError("Strides name does not end with '/strides'")
  56     strides_node = name_to_node[strides_input_name]
  57     value = strides_node.attr["value"]
  58     t = make_ndarray(value.tensor)
  59     stride_y = t[1]
  60     stride_x = t[2]
  61   else:
  62     strides_attr = node.attr["strides"]
  63     logging.vlog(4, "strides_attr = %s", strides_attr)
  64     stride_y = strides_attr.list.i[1]
  65     stride_x = strides_attr.list.i[2]
  66   return stride_x, stride_y

  69 def _conv_kernel_size(node, name_to_node):
  70   """Computes kernel size given a TF convolution or pooling node.
  71 
  72   Args:
  73     node: Tensorflow node (NodeDef proto).
  74     name_to_node: Dict keyed by node name, each entry containing the node's
  75       NodeDef.
  76 
  77   Returns:
  78     kernel_size_x: Kernel size for horizontal direction (integer).
  79     kernel_size_y: Kernel size for vertical direction (integer).
  80 
  81   Raises:
  82     ValueError: If the weight layer node is invalid.
  83   """
  84   weights_layer_read_name = node.input[1]
  85   if not weights_layer_read_name.endswith("/read"):
  86     raise ValueError(
  87         "Weight layer's name input to conv layer does not end with '/read'")
  88   weights_layer_param_name = weights_layer_read_name[:-5]
  89   weights_node = name_to_node[weights_layer_param_name]
  90   if weights_node.op != "VariableV2":
  91     raise ValueError("Weight layer is not of type VariableV2")
  92   shape = weights_node.attr["shape"]
  93   logging.vlog(4, "weight shape = %s", shape)
  94   kernel_size_y = shape.shape.dim[0].size
  95   kernel_size_x = shape.shape.dim[1].size
  96   return kernel_size_x, kernel_size_y

  99 def _padding_size_conv_pool(node, kernel_size, stride, input_resolution=None):
 100   """Computes padding size given a TF convolution or pooling node.
 101 
 102   Args:
 103     node: Tensorflow node (NodeDef proto).
 104     kernel_size: Kernel size of node (integer).
 105     stride: Stride size of node (integer).
 106     input_resolution: Input resolution to assume, if not None (integer).
 107 
 108   Returns:
 109     total_padding: Total padding size (integer).
 110     padding: Padding size, applied to the left or top (integer).
 111 
 112   Raises:
 113     ValueError: If padding is invalid.
 114   """
 115   # In this case, we need to carefully consider the different TF padding modes.
 116   # The padding depends on kernel size, and may depend on input size. If it
 117   # depends on input size and input_resolution is None, we raise an exception.
 118   padding_attr = node.attr["padding"]
 119   logging.vlog(4, "padding_attr = %s", padding_attr)
 120   if padding_attr.s in _VALID_PADDING:
 121     total_padding = 0
 122     padding = 0
 123   elif padding_attr.s in _SAME_PADDING:
 124     if input_resolution is None:
 125       # In this case, we do not know the input resolution, so we can only know
 126       # the padding in some special cases.
 127       if kernel_size == 1:
 128         total_padding = 0
 129         padding = 0
 130       elif stride == 1:
 131         total_padding = kernel_size - 1
 132         padding = int(math.floor(float(total_padding) / 2))
 133       elif stride == 2 and kernel_size % 2 == 0:
 134         # In this case, we can be sure of the left/top padding, but not of the
 135         # total padding.
 136         total_padding = None
 137         padding = int(math.floor((float(kernel_size) - 1) / 2))
 138       else:
 139         total_padding = None
 140         padding = None
 141         logging.warning(
 142             "Padding depends on input size, which means that the effective "
 143             "padding may be different depending on the input image "
 144             "dimensionality. In this case, alignment check will be skipped. If"
 145             " you know the input resolution, please set it.")
 146     else:
 147       # First, compute total_padding based on documentation.
 148       if input_resolution % stride == 0:
 149         total_padding = int(max(float(kernel_size - stride), 0.0))
 150       else:
 151         total_padding = int(
 152             max(float(kernel_size - (input_resolution % stride)), 0.0))
 153       # Then, compute left/top padding.
 154       padding = int(math.floor(float(total_padding) / 2))
 155 
 156   else:
 157     raise ValueError("Invalid padding operation %s" % padding_attr.s)
 158   return total_padding, padding

 161 def _pool_kernel_size(node, name_to_node):
 162   """Computes kernel size given a TF pooling node.
 163 
 164   Args:
 165     node: Tensorflow node (NodeDef proto).
 166     name_to_node: For MaxPoolV2, mapping from node name to NodeDef.
 167 
 168   Returns:
 169     kernel_size_x: Kernel size for horizontal direction (integer).
 170     kernel_size_y: Kernel size for vertical direction (integer).
 171 
 172   Raises:
 173     ValueError: If pooling is invalid.
 174   """
 175   if node.op == "MaxPoolV2":
 176     ksize_input_name = node.input[1]
 177     if not ksize_input_name.endswith("/ksize"):
 178       raise ValueError("Kernel size name does not end with '/ksize'")
 179     ksize_node = name_to_node[ksize_input_name]
 180     value = ksize_node.attr["value"]
 181     t = make_ndarray(value.tensor)
 182     kernel_size_y = t[1]
 183     kernel_size_x = t[2]
 184     if t[0] != 1:
 185       raise ValueError("pool ksize for first dim is not 1")
 186     if t[3] != 1:
 187       raise ValueError("pool ksize for last dim is not 1")
 188   else:
 189     ksize = node.attr["ksize"]
 190     kernel_size_y = ksize.list.i[1]
 191     kernel_size_x = ksize.list.i[2]
 192     if ksize.list.i[0] != 1:
 193       raise ValueError("pool ksize for first dim is not 1")
 194     if ksize.list.i[3] != 1:
 195       raise ValueError("pool ksize for last dim is not 1")
 196   return kernel_size_x, kernel_size_y

 199 def _padding_size_pad_layer(node, name_to_node):
 200   """Computes padding size given a TF padding node.
 201 
 202   Args:
 203     node: Tensorflow node (NodeDef proto).
 204     name_to_node: Dict keyed by node name, each entry containing the node's
 205       NodeDef.
 206 
 207   Returns:
 208     total_padding_x: Total padding size for horizontal direction (integer).
 209     padding_x: Padding size for horizontal direction, left side (integer).
 210     total_padding_y: Total padding size for vertical direction (integer).
 211     padding_y: Padding size for vertical direction, top side (integer).
 212 
 213   Raises:
 214     ValueError: If padding layer is invalid.
 215   """
 216   paddings_layer_name = node.input[1]
 217   if not paddings_layer_name.endswith("/paddings"):
 218     raise ValueError("Padding layer name does not end with '/paddings'")
 219   paddings_node = name_to_node[paddings_layer_name]
 220   if paddings_node.op != "Const":
 221     raise ValueError("Padding op is not Const")
 222   value = paddings_node.attr["value"]
 223   t = make_ndarray(value.tensor)
 224   padding_y = t[1][0]
 225   padding_x = t[2][0]
 226   total_padding_y = padding_y + t[1][1]
 227   total_padding_x = padding_x + t[2][1]
 228   if (t[0][0] != 0) or (t[0][1] != 0):
 229     raise ValueError("padding is not zero for first tensor dim")
 230   if (t[3][0] != 0) or (t[3][1] != 0):
 231     raise ValueError("padding is not zero for last tensor dim")
 232   return total_padding_x, padding_x, total_padding_y, padding_y

 235 def get_layer_params(node, name_to_node, input_resolution=None, force=False):
 236   """Gets layer parameters relevant for RF computation.
 237 
 238   Currently, only these nodes are supported:
 239   - Conv2D
 240   - DepthwiseConv2dNative
 241   - Pad
 242   - MaxPool
 243   - AvgPool
 244   - all nodes listed in _UNCHANGED_RF_LAYER_OPS
 245 
 246   Args:
 247     node: Tensorflow node (NodeDef proto).
 248     name_to_node: Dict keyed by node name, each entry containing the node's
 249       NodeDef.
 250     input_resolution: List with 2 dimensions, denoting the height/width of the
 251       input feature map to this layer. If set to None, then the padding may be
 252       undefined (in tensorflow, SAME padding depends on input spatial
 253       resolution).
 254     force: If True, the function does not raise a ValueError if the layer op is
 255       unknown. Instead, in this case it sets each of the returned parameters to
 256       None.
 257 
 258   Returns:
 259     kernel_size_x: Kernel size for horizontal direction (integer).
 260     kernel_size_y: Kernel size for vertical direction (integer).
 261     stride_x: Stride size for horizontal direction (integer).
 262     stride_y: Stride size for vertical direction (integer).
 263     padding_x: Padding size for horizontal direction, left side (integer).
 264     padding_y: Padding size for vertical direction, top side (integer).
 265     total_padding_x: Total padding size for horizontal direction (integer).
 266     total_padding_y: Total padding size for vertical direction (integer).
 267 
 268   Raises:
 269     ValueError: If layer op is unknown and force is False.
 270   """
 271   logging.vlog(3, "node.name = %s", node.name)
 272   logging.vlog(3, "node.op = %s", node.op)
 273   logging.vlog(4, "node = %s", node)
 274   if node.op == "Conv2D" or node.op == "DepthwiseConv2dNative":
 275     stride_x, stride_y = _stride_size(node, name_to_node)
 276     kernel_size_x, kernel_size_y = _conv_kernel_size(node, name_to_node)
 277     # Compute the padding for this node separately for each direction.
 278     total_padding_x, padding_x = _padding_size_conv_pool(
 279         node, kernel_size_x, stride_x, input_resolution[1]
 280         if input_resolution is not None else None)
 281     total_padding_y, padding_y = _padding_size_conv_pool(
 282         node, kernel_size_y, stride_y, input_resolution[0]
 283         if input_resolution is not None else None)
 284   elif node.op == "Pad":
 285     # Kernel and stride are simply 1 in this case.
 286     kernel_size_x = 1
 287     kernel_size_y = 1
 288     stride_x = 1
 289     stride_y = 1
 290     total_padding_x, padding_x, total_padding_y, padding_y = (
 291         _padding_size_pad_layer(node, name_to_node))
 292   elif node.op == "MaxPool" or node.op == "MaxPoolV2" or node.op == "AvgPool":
 293     stride_x, stride_y = _stride_size(node, name_to_node)
 294     kernel_size_x, kernel_size_y = _pool_kernel_size(node, name_to_node)
 295     # Compute the padding for this node separately for each direction.
 296     total_padding_x, padding_x = _padding_size_conv_pool(
 297         node, kernel_size_x, stride_x, input_resolution[1]
 298         if input_resolution is not None else None)
 299     total_padding_y, padding_y = _padding_size_conv_pool(
 300         node, kernel_size_y, stride_y, input_resolution[0]
 301         if input_resolution is not None else None)
 302   elif node.op in _UNCHANGED_RF_LAYER_OPS:
 303     # These nodes do not modify the RF parameters.
 304     kernel_size_x = 1
 305     kernel_size_y = 1
 306     stride_x = 1
 307     stride_y = 1
 308     total_padding_x = 0
 309     padding_x = 0
 310     total_padding_y = 0
 311     padding_y = 0
 312   else:
 313     if force:
 314       kernel_size_x = None
 315       kernel_size_y = None
 316       stride_x = None
 317       stride_y = None
 318       total_padding_x = None
 319       padding_x = None
 320       total_padding_y = None
 321       padding_y = None
 322     else:
 323       raise ValueError("Unknown layer for operation '%s': %s" % (node.name,
 324                                                                  node.op))
 325   return (kernel_size_x, kernel_size_y, stride_x, stride_y, padding_x,
 326           padding_y, total_padding_x, total_padding_y)
