  32 class _VectorStudentT(transformed_distribution.TransformedDistribution):
  33   """A vector version of Student's t-distribution on `R^k`.
  34 
  35   #### Mathematical details
  36 
  37   The probability density function (pdf) is,
  38 
  39   ```none
  40   pdf(x; df, mu, Sigma) = (1 + ||y||**2 / df)**(-0.5 (df + 1)) / Z
  41   where,
  42   y = inv(Sigma) (x - mu)
  43   Z = abs(det(Sigma)) ( sqrt(df pi) Gamma(0.5 df) / Gamma(0.5 (df + 1)) )**k
  44   ```
  45 
  46   where:
  47   * `loc = mu`; a vector in `R^k`,
  48   * `scale = Sigma`; a lower-triangular matrix in `R^{k x k}`,
  49   * `Z` denotes the normalization constant, and,
  50   * `Gamma` is the [gamma function](
  51     https://en.wikipedia.org/wiki/Gamma_function), and,
  52   * `||y||**2` denotes the [squared Euclidean norm](
  53   https://en.wikipedia.org/wiki/Norm_(mathematics)#Euclidean_norm) of `y`.
  54 
  55   The VectorStudentT distribution is a member of the [location-scale family](
  56   https://en.wikipedia.org/wiki/Location-scale_family), i.e., it can be
  57   constructed as,
  58 
  59   ```none
  60   X ~ StudentT(df, loc=0, scale=1)
  61   Y = loc + scale * X
  62   ```
  63 
  64   Notice that the `scale` matrix has semantics closer to std. deviation than
  65   covariance (but it is not std. deviation).
  66 
  67   This distribution is an Affine transformation of iid
  68   [Student's t-distributions](
  69   https://en.wikipedia.org/wiki/Student%27s_t-distribution)
  70   and should not be confused with the [Multivariate Student's t-distribution](
  71   https://en.wikipedia.org/wiki/Multivariate_t-distribution). The
  72   traditional Multivariate Student's t-distribution is type of
  73   [elliptical distribution](
  74   https://en.wikipedia.org/wiki/Elliptical_distribution); it has PDF:
  75 
  76   ```none
  77   pdf(x; df, mu, Sigma) = (1 + ||y||**2 / df)**(-0.5 (df + k)) / Z
  78   where,
  79   y = inv(Sigma) (x - mu)
  80   Z = abs(det(Sigma)) sqrt(df pi)**k Gamma(0.5 df) / Gamma(0.5 (df + k))
  81   ```
  82 
  83   Notice that the Multivariate Student's t-distribution uses `k` where the
  84   Vector Student's t-distribution has a `1`. Conversely the Vector version has a
  85   broader application of the power-`k` in the normalization constant.
  86 
  87   #### Examples
  88 
  89   A single instance of a "Vector Student's t-distribution" is defined by a mean
  90   vector of length `k` and a scale matrix of shape `k x k`.
  91 
  92   Extra leading dimensions, if provided, allow for batches.
  93 
  94   ```python
  95   import tensorflow_probability as tfp
  96   tfd = tfp.distributions
  97 
  98   # Initialize a single 3-variate vector Student's t-distribution.
  99   mu = [1., 2, 3]
 100   chol = [[1., 0, 0.],
 101           [1, 3, 0],
 102           [1, 2, 3]]
 103   vt = tfd.VectorStudentT(df=2, loc=mu, scale_tril=chol)
 104 
 105   # Evaluate this on an observation in R^3, returning a scalar.
 106   vt.prob([-1., 0, 1])
 107 
 108   # Initialize a batch of two 3-variate vector Student's t-distributions.
 109   mu = [[1., 2, 3],
 110         [11, 22, 33]]
 111   chol = ...  # shape 2 x 3 x 3, lower triangular, positive diagonal.
 112   vt = tfd.VectorStudentT(loc=mu, scale_tril=chol)
 113 
 114   # Evaluate this on a two observations, each in R^3, returning a length two
 115   # tensor.
 116   x = [[-1, 0, 1],
 117        [-11, 0, 11]]
 118   vt.prob(x)
 119   ```
 120 
 121   For more examples of how to construct the `scale` matrix, see the
 122   `tf.contrib.distributions.bijectors.Affine` docstring.
 123 
 124   """
 125 
 126   @deprecation.deprecated(
 127       "2018-10-01",
 128       "The TensorFlow Distributions library has moved to "
 129       "TensorFlow Probability "
 130       "(https://github.com/tensorflow/probability). You "
 131       "should update all references to use `tfp.distributions` "
 132       "instead of `tf.contrib.distributions`.",
 133       warn_once=True)
 134   def __init__(self,
 135                df,
 136                loc=None,
 137                scale_identity_multiplier=None,
 138                scale_diag=None,
 139                scale_tril=None,
 140                scale_perturb_factor=None,
 141                scale_perturb_diag=None,
 142                validate_args=False,
 143                allow_nan_stats=True,
 144                name="VectorStudentT"):
 145     """Instantiates the vector Student's t-distributions on `R^k`.
 146 
 147     The `batch_shape` is the broadcast between `df.batch_shape` and
 148     `Affine.batch_shape` where `Affine` is constructed from `loc` and
 149     `scale_*` arguments.
 150 
 151     The `event_shape` is the event shape of `Affine.event_shape`.
 152 
 153     Args:
 154       df: Floating-point `Tensor`. The degrees of freedom of the
 155         distribution(s). `df` must contain only positive values. Must be
 156         scalar if `loc`, `scale_*` imply non-scalar batch_shape or must have the
 157         same `batch_shape` implied by `loc`, `scale_*`.
 158       loc: Floating-point `Tensor`. If this is set to `None`, no `loc` is
 159         applied.
 160       scale_identity_multiplier: floating point rank 0 `Tensor` representing a
 161         scaling done to the identity matrix. When `scale_identity_multiplier =
 162         scale_diag=scale_tril = None` then `scale += IdentityMatrix`. Otherwise
 163         no scaled-identity-matrix is added to `scale`.
 164       scale_diag: Floating-point `Tensor` representing the diagonal matrix.
 165         `scale_diag` has shape [N1, N2, ..., k], which represents a k x k
 166         diagonal matrix. When `None` no diagonal term is added to `scale`.
 167       scale_tril: Floating-point `Tensor` representing the diagonal matrix.
 168         `scale_diag` has shape [N1, N2, ..., k, k], which represents a k x k
 169         lower triangular matrix. When `None` no `scale_tril` term is added to
 170         `scale`. The upper triangular elements above the diagonal are ignored.
 171       scale_perturb_factor: Floating-point `Tensor` representing factor matrix
 172         with last two dimensions of shape `(k, r)`. When `None`, no rank-r
 173         update is added to `scale`.
 174       scale_perturb_diag: Floating-point `Tensor` representing the diagonal
 175         matrix. `scale_perturb_diag` has shape [N1, N2, ..., r], which
 176         represents an r x r Diagonal matrix. When `None` low rank updates will
 177         take the form `scale_perturb_factor * scale_perturb_factor.T`.
 178       validate_args: Python `bool`, default `False`. When `True` distribution
 179         parameters are checked for validity despite possibly degrading runtime
 180         performance. When `False` invalid inputs may silently render incorrect
 181         outputs.
 182       allow_nan_stats: Python `bool`, default `True`. When `True`,
 183         statistics (e.g., mean, mode, variance) use the value "`NaN`" to
 184         indicate the result is undefined. When `False`, an exception is raised
 185         if one or more of the statistic's batch members are undefined.
 186       name: Python `str` name prefixed to Ops created by this class.
 187     """
 188     parameters = dict(locals())
 189     graph_parents = [df, loc, scale_identity_multiplier, scale_diag,
 190                      scale_tril, scale_perturb_factor, scale_perturb_diag]
 191     with ops.name_scope(name) as name:
 192       with ops.name_scope("init", values=graph_parents):
 193         # The shape of the _VectorStudentT distribution is governed by the
 194         # relationship between df.batch_shape and affine.batch_shape. In
 195         # pseudocode the basic procedure is:
 196         #   if df.batch_shape is scalar:
 197         #     if affine.batch_shape is not scalar:
 198         #       # broadcast distribution.sample so
 199         #       # it has affine.batch_shape.
 200         #     self.batch_shape = affine.batch_shape
 201         #   else:
 202         #     if affine.batch_shape is scalar:
 203         #       # let affine broadcasting do its thing.
 204         #     self.batch_shape = df.batch_shape
 205         # All of the above magic is actually handled by TransformedDistribution.
 206         # Here we really only need to collect the affine.batch_shape and decide
 207         # what we're going to pass in to TransformedDistribution's
 208         # (override) batch_shape arg.
 209         affine = bijectors.Affine(
 210             shift=loc,
 211             scale_identity_multiplier=scale_identity_multiplier,
 212             scale_diag=scale_diag,
 213             scale_tril=scale_tril,
 214             scale_perturb_factor=scale_perturb_factor,
 215             scale_perturb_diag=scale_perturb_diag,
 216             validate_args=validate_args)
 217         distribution = student_t.StudentT(
 218             df=df,
 219             loc=array_ops.zeros([], dtype=affine.dtype),
 220             scale=array_ops.ones([], dtype=affine.dtype))
 221         batch_shape, override_event_shape = (
 222             distribution_util.shapes_from_loc_and_scale(
 223                 affine.shift, affine.scale))
 224         override_batch_shape = distribution_util.pick_vector(
 225             distribution.is_scalar_batch(),
 226             batch_shape,
 227             constant_op.constant([], dtype=dtypes.int32))
 228         super(_VectorStudentT, self).__init__(
 229             distribution=distribution,
 230             bijector=affine,
 231             batch_shape=override_batch_shape,
 232             event_shape=override_event_shape,
 233             validate_args=validate_args,
 234             name=name)
 235         self._parameters = parameters
 236 
 237   @property
 238   def df(self):
 239     """Degrees of freedom in these Student's t distribution(s)."""
 240     return self.distribution.df
 241 
 242   @property
 243   def loc(self):
 244     """Locations of these Student's t distribution(s)."""
 245     return self.bijector.shift
 246 
 247   @property
 248   def scale(self):
 249     """Dense (batch) covariance matrix, if available."""
 250     return self.bijector.scale
