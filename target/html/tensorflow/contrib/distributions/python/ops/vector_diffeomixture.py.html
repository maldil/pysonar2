  61 def quadrature_scheme_softmaxnormal_gauss_hermite(
  62     normal_loc, normal_scale, quadrature_size,
  63     validate_args=False, name=None):
  64   """Use Gauss-Hermite quadrature to form quadrature on `K - 1` simplex.
  65 
  66   A `SoftmaxNormal` random variable `Y` may be generated via
  67 
  68   ```
  69   Y = SoftmaxCentered(X),
  70   X = Normal(normal_loc, normal_scale)
  71   ```
  72 
  73   Note: for a given `quadrature_size`, this method is generally less accurate
  74   than `quadrature_scheme_softmaxnormal_quantiles`.
  75 
  76   Args:
  77     normal_loc: `float`-like `Tensor` with shape `[b1, ..., bB, K-1]`, B>=0.
  78       The location parameter of the Normal used to construct the SoftmaxNormal.
  79     normal_scale: `float`-like `Tensor`. Broadcastable with `normal_loc`.
  80       The scale parameter of the Normal used to construct the SoftmaxNormal.
  81     quadrature_size: Python `int` scalar representing the number of quadrature
  82       points.
  83     validate_args: Python `bool`, default `False`. When `True` distribution
  84       parameters are checked for validity despite possibly degrading runtime
  85       performance. When `False` invalid inputs may silently render incorrect
  86       outputs.
  87     name: Python `str` name prefixed to Ops created by this class.
  88 
  89   Returns:
  90     grid: Shape `[b1, ..., bB, K, quadrature_size]` `Tensor` representing the
  91       convex combination of affine parameters for `K` components.
  92       `grid[..., :, n]` is the `n`-th grid point, living in the `K - 1` simplex.
  93     probs:  Shape `[b1, ..., bB, K, quadrature_size]` `Tensor` representing the
  94       associated with each grid point.
  95   """
  96   with ops.name_scope(name, "quadrature_scheme_softmaxnormal_gauss_hermite",
  97                       [normal_loc, normal_scale]):
  98     normal_loc = ops.convert_to_tensor(normal_loc, name="normal_loc")
  99     dt = normal_loc.dtype.base_dtype
 100     normal_scale = ops.convert_to_tensor(
 101         normal_scale, dtype=dt, name="normal_scale")
 102 
 103     normal_scale = maybe_check_quadrature_param(
 104         normal_scale, "normal_scale", validate_args)
 105 
 106     grid, probs = np.polynomial.hermite.hermgauss(deg=quadrature_size)
 107     grid = grid.astype(dt.dtype.as_numpy_dtype)
 108     probs = probs.astype(dt.dtype.as_numpy_dtype)
 109     probs /= np.linalg.norm(probs, ord=1, keepdims=True)
 110     probs = ops.convert_to_tensor(probs, name="probs", dtype=dt)
 111 
 112     grid = softmax(
 113         -distribution_util.pad(
 114             (normal_loc[..., array_ops.newaxis] +
 115              np.sqrt(2.) * normal_scale[..., array_ops.newaxis] * grid),
 116             axis=-2,
 117             front=True),
 118         axis=-2)  # shape: [B, components, deg]
 119 
 120     return grid, probs

 131 def quadrature_scheme_softmaxnormal_quantiles(
 132     normal_loc, normal_scale, quadrature_size,
 133     validate_args=False, name=None):
 134   """Use SoftmaxNormal quantiles to form quadrature on `K - 1` simplex.
 135 
 136   A `SoftmaxNormal` random variable `Y` may be generated via
 137 
 138   ```
 139   Y = SoftmaxCentered(X),
 140   X = Normal(normal_loc, normal_scale)
 141   ```
 142 
 143   Args:
 144     normal_loc: `float`-like `Tensor` with shape `[b1, ..., bB, K-1]`, B>=0.
 145       The location parameter of the Normal used to construct the SoftmaxNormal.
 146     normal_scale: `float`-like `Tensor`. Broadcastable with `normal_loc`.
 147       The scale parameter of the Normal used to construct the SoftmaxNormal.
 148     quadrature_size: Python `int` scalar representing the number of quadrature
 149       points.
 150     validate_args: Python `bool`, default `False`. When `True` distribution
 151       parameters are checked for validity despite possibly degrading runtime
 152       performance. When `False` invalid inputs may silently render incorrect
 153       outputs.
 154     name: Python `str` name prefixed to Ops created by this class.
 155 
 156   Returns:
 157     grid: Shape `[b1, ..., bB, K, quadrature_size]` `Tensor` representing the
 158       convex combination of affine parameters for `K` components.
 159       `grid[..., :, n]` is the `n`-th grid point, living in the `K - 1` simplex.
 160     probs:  Shape `[b1, ..., bB, K, quadrature_size]` `Tensor` representing the
 161       associated with each grid point.
 162   """
 163   with ops.name_scope(name, "softmax_normal_grid_and_probs",
 164                       [normal_loc, normal_scale]):
 165     normal_loc = ops.convert_to_tensor(normal_loc, name="normal_loc")
 166     dt = normal_loc.dtype.base_dtype
 167     normal_scale = ops.convert_to_tensor(
 168         normal_scale, dtype=dt, name="normal_scale")
 169 
 170     normal_scale = maybe_check_quadrature_param(
 171         normal_scale, "normal_scale", validate_args)
 172 
 173     dist = normal_lib.Normal(loc=normal_loc, scale=normal_scale)
 174 
 175     def _get_batch_ndims():
 176       """Helper to get dist.batch_shape.ndims, statically if possible."""
 177       ndims = dist.batch_shape.ndims
 178       if ndims is None:
 179         ndims = array_ops.shape(dist.batch_shape_tensor())[0]
 180       return ndims
 181     batch_ndims = _get_batch_ndims()
 182 
 183     def _get_final_shape(qs):
 184       """Helper to build `TensorShape`."""
 185       bs = dist.batch_shape.with_rank_at_least(1)
 186       num_components = bs[-1].value
 187       if num_components is not None:
 188         num_components += 1
 189       tail = tensor_shape.TensorShape([num_components, qs])
 190       return bs[:-1].concatenate(tail)
 191 
 192     def _compute_quantiles():
 193       """Helper to build quantiles."""
 194       # Omit {0, 1} since they might lead to Inf/NaN.
 195       zero = array_ops.zeros([], dtype=dist.dtype)
 196       edges = math_ops.linspace(zero, 1., quadrature_size + 3)[1:-1]
 197       # Expand edges so its broadcast across batch dims.
 198       edges = array_ops.reshape(edges, shape=array_ops.concat([
 199           [-1], array_ops.ones([batch_ndims], dtype=dtypes.int32)], axis=0))
 200       quantiles = dist.quantile(edges)
 201       quantiles = SoftmaxCentered().forward(quantiles)
 202       # Cyclically permute left by one.
 203       perm = array_ops.concat([
 204           math_ops.range(1, 1 + batch_ndims), [0]], axis=0)
 205       quantiles = array_ops.transpose(quantiles, perm)
 206       quantiles.set_shape(_get_final_shape(quadrature_size + 1))
 207       return quantiles
 208     quantiles = _compute_quantiles()
 209 
 210     # Compute grid as quantile midpoints.
 211     grid = (quantiles[..., :-1] + quantiles[..., 1:]) / 2.
 212     # Set shape hints.
 213     grid.set_shape(_get_final_shape(quadrature_size))
 214 
 215     # By construction probs is constant, i.e., `1 / quadrature_size`. This is
 216     # important, because non-constant probs leads to non-reparameterizable
 217     # samples.
 218     probs = array_ops.fill(
 219         dims=[quadrature_size],
 220         value=1. / math_ops.cast(quadrature_size, dist.dtype))
 221 
 222     return grid, probs

 225 class VectorDiffeomixture(distribution_lib.Distribution):
 226   """VectorDiffeomixture distribution.
 227 
 228   A vector diffeomixture (VDM) is a distribution parameterized by a convex
 229   combination of `K` component `loc` vectors, `loc[k], k = 0,...,K-1`, and `K`
 230   `scale` matrices `scale[k], k = 0,..., K-1`.  It approximates the following
 231   [compound distribution]
 232   (https://en.wikipedia.org/wiki/Compound_probability_distribution)
 233 
 234   ```none
 235   p(x) = int p(x | z) p(z) dz,
 236   where z is in the K-simplex, and
 237   p(x | z) := p(x | loc=sum_k z[k] loc[k], scale=sum_k z[k] scale[k])
 238   ```
 239 
 240   The integral `int p(x | z) p(z) dz` is approximated with a quadrature scheme
 241   adapted to the mixture density `p(z)`.  The `N` quadrature points `z_{N, n}`
 242   and weights `w_{N, n}` (which are non-negative and sum to 1) are chosen
 243   such that
 244 
 245   ```q_N(x) := sum_{n=1}^N w_{n, N} p(x | z_{N, n}) --> p(x)```
 246 
 247   as `N --> infinity`.
 248 
 249   Since `q_N(x)` is in fact a mixture (of `N` points), we may sample from
 250   `q_N` exactly.  It is important to note that the VDM is *defined* as `q_N`
 251   above, and *not* `p(x)`.  Therefore, sampling and pdf may be implemented as
 252   exact (up to floating point error) methods.
 253 
 254   A common choice for the conditional `p(x | z)` is a multivariate Normal.
 255 
 256   The implemented marginal `p(z)` is the `SoftmaxNormal`, which is a
 257   `K-1` dimensional Normal transformed by a `SoftmaxCentered` bijector, making
 258   it a density on the `K`-simplex.  That is,
 259 
 260   ```
 261   Z = SoftmaxCentered(X),
 262   X = Normal(mix_loc / temperature, 1 / temperature)
 263   ```
 264 
 265   The default quadrature scheme chooses `z_{N, n}` as `N` midpoints of
 266   the quantiles of `p(z)` (generalized quantiles if `K > 2`).
 267 
 268   See [Dillon and Langmore (2018)][1] for more details.
 269 
 270   #### About `Vector` distributions in TensorFlow.
 271 
 272   The `VectorDiffeomixture` is a non-standard distribution that has properties
 273   particularly useful in [variational Bayesian
 274   methods](https://en.wikipedia.org/wiki/Variational_Bayesian_methods).
 275 
 276   Conditioned on a draw from the SoftmaxNormal, `X|z` is a vector whose
 277   components are linear combinations of affine transformations, thus is itself
 278   an affine transformation.
 279 
 280   Note: The marginals `X_1|v, ..., X_d|v` are *not* generally identical to some
 281   parameterization of `distribution`.  This is due to the fact that the sum of
 282   draws from `distribution` are not generally itself the same `distribution`.
 283 
 284   #### About `Diffeomixture`s and reparameterization.
 285 
 286   The `VectorDiffeomixture` is designed to be reparameterized, i.e., its
 287   parameters are only used to transform samples from a distribution which has no
 288   trainable parameters. This property is important because backprop stops at
 289   sources of stochasticity. That is, as long as the parameters are used *after*
 290   the underlying source of stochasticity, the computed gradient is accurate.
 291 
 292   Reparametrization means that we can use gradient-descent (via backprop) to
 293   optimize Monte-Carlo objectives. Such objectives are a finite-sample
 294   approximation of an expectation and arise throughout scientific computing.
 295 
 296   WARNING: If you backprop through a VectorDiffeomixture sample and the "base"
 297   distribution is both: not `FULLY_REPARAMETERIZED` and a function of trainable
 298   variables, then the gradient is not guaranteed correct!
 299 
 300   #### Examples
 301 
 302   ```python
 303   import tensorflow_probability as tfp
 304   tfd = tfp.distributions
 305 
 306   # Create two batches of VectorDiffeomixtures, one with mix_loc=[0.],
 307   # another with mix_loc=[1]. In both cases, `K=2` and the affine
 308   # transformations involve:
 309   # k=0: loc=zeros(dims)  scale=LinearOperatorScaledIdentity
 310   # k=1: loc=[2.]*dims    scale=LinOpDiag
 311   dims = 5
 312   vdm = tfd.VectorDiffeomixture(
 313       mix_loc=[[0.], [1]],
 314       temperature=[1.],
 315       distribution=tfd.Normal(loc=0., scale=1.),
 316       loc=[
 317           None,  # Equivalent to `np.zeros(dims, dtype=np.float32)`.
 318           np.float32([2.]*dims),
 319       ],
 320       scale=[
 321           tf.linalg.LinearOperatorScaledIdentity(
 322             num_rows=dims,
 323             multiplier=np.float32(1.1),
 324             is_positive_definite=True),
 325           tf.linalg.LinearOperatorDiag(
 326             diag=np.linspace(2.5, 3.5, dims, dtype=np.float32),
 327             is_positive_definite=True),
 328       ],
 329       validate_args=True)
 330   ```
 331 
 332   #### References
 333 
 334   [1]: Joshua Dillon and Ian Langmore. Quadrature Compound: An approximating
 335        family of distributions. _arXiv preprint arXiv:1801.03080_, 2018.
 336        https://arxiv.org/abs/1801.03080
 337   """
 338 
 339   @deprecation.deprecated(
 340       "2018-10-01",
 341       "The TensorFlow Distributions library has moved to "
 342       "TensorFlow Probability "
 343       "(https://github.com/tensorflow/probability). You "
 344       "should update all references to use `tfp.distributions` "
 345       "instead of `tf.contrib.distributions`.",
 346       warn_once=True)
 347   def __init__(self,
 348                mix_loc,
 349                temperature,
 350                distribution,
 351                loc=None,
 352                scale=None,
 353                quadrature_size=8,
 354                quadrature_fn=quadrature_scheme_softmaxnormal_quantiles,
 355                validate_args=False,
 356                allow_nan_stats=True,
 357                name="VectorDiffeomixture"):
 358     """Constructs the VectorDiffeomixture on `R^d`.
 359 
 360     The vector diffeomixture (VDM) approximates the compound distribution
 361 
 362     ```none
 363     p(x) = int p(x | z) p(z) dz,
 364     where z is in the K-simplex, and
 365     p(x | z) := p(x | loc=sum_k z[k] loc[k], scale=sum_k z[k] scale[k])
 366     ```
 367 
 368     Args:
 369       mix_loc: `float`-like `Tensor` with shape `[b1, ..., bB, K-1]`.
 370         In terms of samples, larger `mix_loc[..., k]` ==>
 371         `Z` is more likely to put more weight on its `kth` component.
 372       temperature: `float`-like `Tensor`. Broadcastable with `mix_loc`.
 373         In terms of samples, smaller `temperature` means one component is more
 374         likely to dominate.  I.e., smaller `temperature` makes the VDM look more
 375         like a standard mixture of `K` components.
 376       distribution: `tf.Distribution`-like instance. Distribution from which `d`
 377         iid samples are used as input to the selected affine transformation.
 378         Must be a scalar-batch, scalar-event distribution.  Typically
 379         `distribution.reparameterization_type = FULLY_REPARAMETERIZED` or it is
 380         a function of non-trainable parameters. WARNING: If you backprop through
 381         a VectorDiffeomixture sample and the `distribution` is not
 382         `FULLY_REPARAMETERIZED` yet is a function of trainable variables, then
 383         the gradient will be incorrect!
 384       loc: Length-`K` list of `float`-type `Tensor`s. The `k`-th element
 385         represents the `shift` used for the `k`-th affine transformation.  If
 386         the `k`-th item is `None`, `loc` is implicitly `0`.  When specified,
 387         must have shape `[B1, ..., Bb, d]` where `b >= 0` and `d` is the event
 388         size.
 389       scale: Length-`K` list of `LinearOperator`s. Each should be
 390         positive-definite and operate on a `d`-dimensional vector space. The
 391         `k`-th element represents the `scale` used for the `k`-th affine
 392         transformation. `LinearOperator`s must have shape `[B1, ..., Bb, d, d]`,
 393         `b >= 0`, i.e., characterizes `b`-batches of `d x d` matrices
 394       quadrature_size: Python `int` scalar representing number of
 395         quadrature points.  Larger `quadrature_size` means `q_N(x)` better
 396         approximates `p(x)`.
 397       quadrature_fn: Python callable taking `normal_loc`, `normal_scale`,
 398         `quadrature_size`, `validate_args` and returning `tuple(grid, probs)`
 399         representing the SoftmaxNormal grid and corresponding normalized weight.
 400         normalized) weight.
 401         Default value: `quadrature_scheme_softmaxnormal_quantiles`.
 402       validate_args: Python `bool`, default `False`. When `True` distribution
 403         parameters are checked for validity despite possibly degrading runtime
 404         performance. When `False` invalid inputs may silently render incorrect
 405         outputs.
 406       allow_nan_stats: Python `bool`, default `True`. When `True`,
 407         statistics (e.g., mean, mode, variance) use the value "`NaN`" to
 408         indicate the result is undefined. When `False`, an exception is raised
 409         if one or more of the statistic's batch members are undefined.
 410       name: Python `str` name prefixed to Ops created by this class.
 411 
 412     Raises:
 413       ValueError: if `not scale or len(scale) < 2`.
 414       ValueError: if `len(loc) != len(scale)`
 415       ValueError: if `quadrature_grid_and_probs is not None` and
 416         `len(quadrature_grid_and_probs[0]) != len(quadrature_grid_and_probs[1])`
 417       ValueError: if `validate_args` and any not scale.is_positive_definite.
 418       TypeError: if any scale.dtype != scale[0].dtype.
 419       TypeError: if any loc.dtype != scale[0].dtype.
 420       NotImplementedError: if `len(scale) != 2`.
 421       ValueError: if `not distribution.is_scalar_batch`.
 422       ValueError: if `not distribution.is_scalar_event`.
 423     """
 424     parameters = dict(locals())
 425     with ops.name_scope(name, values=[mix_loc, temperature]) as name:
 426       if not scale or len(scale) < 2:
 427         raise ValueError("Must specify list (or list-like object) of scale "
 428                          "LinearOperators, one for each component with "
 429                          "num_component >= 2.")
 430 
 431       if loc is None:
 432         loc = [None]*len(scale)
 433 
 434       if len(loc) != len(scale):
 435         raise ValueError("loc/scale must be same-length lists "
 436                          "(or same-length list-like objects).")
 437 
 438       dtype = scale[0].dtype.base_dtype
 439 
 440       loc = [ops.convert_to_tensor(loc_, dtype=dtype, name="loc{}".format(k))
 441              if loc_ is not None else None
 442              for k, loc_ in enumerate(loc)]
 443 
 444       for k, scale_ in enumerate(scale):
 445         if validate_args and not scale_.is_positive_definite:
 446           raise ValueError("scale[{}].is_positive_definite = {} != True".format(
 447               k, scale_.is_positive_definite))
 448         if scale_.dtype.base_dtype != dtype:
 449           raise TypeError(
 450               "dtype mismatch; scale[{}].base_dtype=\"{}\" != \"{}\"".format(
 451                   k, scale_.dtype.base_dtype.name, dtype.name))
 452 
 453       self._endpoint_affine = [
 454           AffineLinearOperator(shift=loc_,
 455                                scale=scale_,
 456                                validate_args=validate_args,
 457                                name="endpoint_affine_{}".format(k))
 458           for k, (loc_, scale_) in enumerate(zip(loc, scale))]
 459 
 460       # TODO(jvdillon): Remove once we support k-mixtures.
 461       # We make this assertion here because otherwise `grid` would need to be a
 462       # vector not a scalar.
 463       if len(scale) != 2:
 464         raise NotImplementedError("Currently only bimixtures are supported; "
 465                                   "len(scale)={} is not 2.".format(len(scale)))
 466 
 467       mix_loc = ops.convert_to_tensor(
 468           mix_loc, dtype=dtype, name="mix_loc")
 469       temperature = ops.convert_to_tensor(
 470           temperature, dtype=dtype, name="temperature")
 471       self._grid, probs = tuple(quadrature_fn(
 472           mix_loc / temperature,
 473           1. / temperature,
 474           quadrature_size,
 475           validate_args))
 476 
 477       # Note: by creating the logits as `log(prob)` we ensure that
 478       # `self.mixture_distribution.logits` is equivalent to
 479       # `math_ops.log(self.mixture_distribution.probs)`.
 480       self._mixture_distribution = categorical_lib.Categorical(
 481           logits=math_ops.log(probs),
 482           validate_args=validate_args,
 483           allow_nan_stats=allow_nan_stats)
 484 
 485       asserts = distribution_util.maybe_check_scalar_distribution(
 486           distribution, dtype, validate_args)
 487       if asserts:
 488         self._grid = control_flow_ops.with_dependencies(
 489             asserts, self._grid)
 490       self._distribution = distribution
 491 
 492       self._interpolated_affine = [
 493           AffineLinearOperator(shift=loc_,
 494                                scale=scale_,
 495                                validate_args=validate_args,
 496                                name="interpolated_affine_{}".format(k))
 497           for k, (loc_, scale_) in enumerate(zip(
 498               interpolate_loc(self._grid, loc),
 499               interpolate_scale(self._grid, scale)))]
 500 
 501       [
 502           self._batch_shape_,
 503           self._batch_shape_tensor_,
 504           self._event_shape_,
 505           self._event_shape_tensor_,
 506       ] = determine_batch_event_shapes(self._grid,
 507                                        self._endpoint_affine)
 508 
 509       super(VectorDiffeomixture, self).__init__(
 510           dtype=dtype,
 511           # We hard-code `FULLY_REPARAMETERIZED` because when
 512           # `validate_args=True` we verify that indeed
 513           # `distribution.reparameterization_type == FULLY_REPARAMETERIZED`. A
 514           # distribution which is a function of only non-trainable parameters
 515           # also implies we can use `FULLY_REPARAMETERIZED`. However, we cannot
 516           # easily test for that possibility thus we use `validate_args=False`
 517           # as a "back-door" to allow users a way to use non
 518           # `FULLY_REPARAMETERIZED` distribution. In such cases IT IS THE USERS
 519           # RESPONSIBILITY to verify that the base distribution is a function of

 568   def _sample_n(self, n, seed=None):
 569     x = self.distribution.sample(
 570         sample_shape=concat_vectors(
 571             [n],
 572             self.batch_shape_tensor(),
 573             self.event_shape_tensor()),
 574         seed=seed)   # shape: [n, B, e]
 575     x = [aff.forward(x) for aff in self.endpoint_affine]
 576 
 577     # Get ids as a [n, batch_size]-shaped matrix, unless batch_shape=[] then get
 578     # ids as a [n]-shaped vector.
 579     batch_size = self.batch_shape.num_elements()
 580     if batch_size is None:
 581       batch_size = array_ops.reduce_prod(self.batch_shape_tensor())
 582     mix_batch_size = self.mixture_distribution.batch_shape.num_elements()
 583     if mix_batch_size is None:
 584       mix_batch_size = math_ops.reduce_prod(
 585           self.mixture_distribution.batch_shape_tensor())
 586     ids = self.mixture_distribution.sample(
 587         sample_shape=concat_vectors(
 588             [n],
 589             distribution_util.pick_vector(
 590                 self.is_scalar_batch(),
 591                 np.int32([]),
 592                 [batch_size // mix_batch_size])),
 593         seed=distribution_util.gen_new_seed(
 594             seed, "vector_diffeomixture"))
 595     # We need to flatten batch dims in case mixture_distribution has its own
 596     # batch dims.
 597     ids = array_ops.reshape(ids, shape=concat_vectors(
 598         [n],
 599         distribution_util.pick_vector(
 600             self.is_scalar_batch(),
 601             np.int32([]),
 602             np.int32([-1]))))
 603 
 604     # Stride `components * quadrature_size` for `batch_size` number of times.
 605     stride = self.grid.shape.with_rank_at_least(
 606         2)[-2:].num_elements()
 607     if stride is None:
 608       stride = array_ops.reduce_prod(
 609           array_ops.shape(self.grid)[-2:])
 610     offset = math_ops.range(start=0,
 611                             limit=batch_size * stride,
 612                             delta=stride,
 613                             dtype=ids.dtype)
 614 
 615     weight = array_ops.gather(
 616         array_ops.reshape(self.grid, shape=[-1]),
 617         ids + offset)
 618     # At this point, weight flattened all batch dims into one.
 619     # We also need to append a singleton to broadcast with event dims.
 620     if self.batch_shape.is_fully_defined():
 621       new_shape = [-1] + self.batch_shape.as_list() + [1]
 622     else:
 623       new_shape = array_ops.concat(
 624           ([-1], self.batch_shape_tensor(), [1]), axis=0)
 625     weight = array_ops.reshape(weight, shape=new_shape)
 626 
 627     if len(x) != 2:
 628       # We actually should have already triggered this exception. However as a
 629       # policy we're putting this exception wherever we exploit the bimixture
 630       # assumption.
 631       raise NotImplementedError("Currently only bimixtures are supported; "
 632                                 "len(scale)={} is not 2.".format(len(x)))
 633 
 634     # Alternatively:
 635     # x = weight * x[0] + (1. - weight) * x[1]
 636     x = weight * (x[0] - x[1]) + x[1]
 637 
 638     return x

 640   def _log_prob(self, x):
 641     # By convention, we always put the grid points right-most.
 642     y = array_ops.stack(
 643         [aff.inverse(x) for aff in self.interpolated_affine],
 644         axis=-1)
 645     log_prob = math_ops.reduce_sum(self.distribution.log_prob(y), axis=-2)
 646     # Because the affine transformation has a constant Jacobian, it is the case
 647     # that `affine.fldj(x) = -affine.ildj(x)`. This is not true in general.
 648     fldj = array_ops.stack([
 649         aff.forward_log_det_jacobian(
 650             x,
 651             event_ndims=array_ops.rank(self.event_shape_tensor())
 652         ) for aff in self.interpolated_affine], axis=-1)
 653     return math_ops.reduce_logsumexp(
 654         self.mixture_distribution.logits - fldj + log_prob, axis=-1)

 656   def _mean(self):
 657     p = self._expand_mix_distribution_probs()
 658     m = self._expand_base_distribution_mean()
 659     mean = None
 660     for k, aff in enumerate(self.interpolated_affine):
 661       # aff.forward is going to do this:
 662       # y = array_ops.squeeze(aff.scale.matmul(m), axis=[-1])
 663       # if aff.shift is not None:
 664       #   y += aff.shift
 665       mean = add(mean, p[..., k] * aff.forward(m))
 666     return mean

 668   def _covariance(self):
 669     # Law of total variance:
 670     #
 671     # Cov[Z] = E[Cov[Z | V]] + Cov[E[Z | V]]
 672     #
 673     # where,
 674     #
 675     # E[Cov[Z | V]] = sum_i mix_prob[i] Scale[i]
 676     # Cov[E[Z | V]] = sum_i mix_prob[i] osquare(loc[i])
 677     #                  - osquare(sum_i mix_prob[i] loc[i])
 678     #
 679     # osquare(x) = x.transpose @ x
 680     return add(
 681         self._mean_of_covariance_given_quadrature_component(diag_only=False),
 682         self._covariance_of_mean_given_quadrature_component(diag_only=False))
 683 
 684   def _variance(self):
 685     # Equivalent to: tf.diag_part(self._covariance()),
 686     return add(
 687         self._mean_of_covariance_given_quadrature_component(diag_only=True),
 688         self._covariance_of_mean_given_quadrature_component(diag_only=True))
 689 
 690   def _mean_of_covariance_given_quadrature_component(self, diag_only):
 691     p = self.mixture_distribution.probs
 692 
 693     # To compute E[Cov(Z|V)], we'll add matrices within three categories:
 694     # scaled-identity, diagonal, and full. Then we'll combine these at the end.
 695     scale_identity_multiplier = None
 696     diag = None
 697     full = None
 698 
 699     for k, aff in enumerate(self.interpolated_affine):
 700       s = aff.scale  # Just in case aff.scale has side-effects, we'll call once.
 701       if (s is None
 702           or isinstance(s, linop_identity_lib.LinearOperatorIdentity)):
 703         scale_identity_multiplier = add(scale_identity_multiplier,
 704                                         p[..., k, array_ops.newaxis])
 705       elif isinstance(s, linop_identity_lib.LinearOperatorScaledIdentity):
 706         scale_identity_multiplier = add(
 707             scale_identity_multiplier,
 708             (p[..., k, array_ops.newaxis] * math_ops.square(s.multiplier)))
 709       elif isinstance(s, linop_diag_lib.LinearOperatorDiag):
 710         diag = add(diag, (p[..., k, array_ops.newaxis] *
 711                           math_ops.square(s.diag_part())))
 712       else:
 713         x = (p[..., k, array_ops.newaxis, array_ops.newaxis] *
 714              s.matmul(s.to_dense(), adjoint_arg=True))
 715         if diag_only:
 716           x = array_ops.matrix_diag_part(x)
 717         full = add(full, x)
 718 
 719     # We must now account for the fact that the base distribution might have a
 720     # non-unity variance. Recall that, since X ~ iid Law(X_0),
 721     #   `Cov(SX+m) = S Cov(X) S.T = S S.T Diag(Var(X_0))`.
 722     # We can scale by `Var(X)` (vs `Cov(X)`) since X corresponds to `d` iid
 723     # samples from a scalar-event distribution.
 724     v = self.distribution.variance()
 725     if scale_identity_multiplier is not None:
 726       scale_identity_multiplier *= v
 727     if diag is not None:
 728       diag *= v[..., array_ops.newaxis]
 729     if full is not None:
 730       full *= v[..., array_ops.newaxis]
 731 
 732     if diag_only:
 733       # Apparently we don't need the full matrix, just the diagonal.
 734       r = add(diag, full)
 735       if r is None and scale_identity_multiplier is not None:
 736         ones = array_ops.ones(self.event_shape_tensor(), dtype=self.dtype)
 737         return scale_identity_multiplier[..., array_ops.newaxis] * ones
 738       return add(r, scale_identity_multiplier)
 739 
 740     # `None` indicates we don't know if the result is positive-definite.
 741     is_positive_definite = (True if all(aff.scale.is_positive_definite
 742                                         for aff in self.endpoint_affine)
 743                             else None)
 744 
 745     to_add = []
 746     if diag is not None:
 747       to_add.append(linop_diag_lib.LinearOperatorDiag(
 748           diag=diag,
 749           is_positive_definite=is_positive_definite))
 750     if full is not None:
 751       to_add.append(linop_full_lib.LinearOperatorFullMatrix(
 752           matrix=full,
 753           is_positive_definite=is_positive_definite))
 754     if scale_identity_multiplier is not None:
 755       to_add.append(linop_identity_lib.LinearOperatorScaledIdentity(
 756           num_rows=self.event_shape_tensor()[0],
 757           multiplier=scale_identity_multiplier,
 758           is_positive_definite=is_positive_definite))
 759 
 760     return (linop_add_lib.add_operators(to_add)[0].to_dense()
 761             if to_add else None)
 762 
 763   def _covariance_of_mean_given_quadrature_component(self, diag_only):
 764     square = math_ops.square if diag_only else vec_osquare
 765 
 766     p = self._expand_mix_distribution_probs()
 767     if not diag_only:
 768       p = p[..., array_ops.newaxis, :]  # Assuming event.ndims=1.
 769     m = self._expand_base_distribution_mean()
 770 
 771     cov_e_z_given_v = None
 772     e_z_given_v = self._mean()
 773     for k, aff in enumerate(self.interpolated_affine):
 774       y = aff.forward(m)
 775       cov_e_z_given_v = add(cov_e_z_given_v,
 776                             p[..., k] * square(y - e_z_given_v))
 777 
 778     return cov_e_z_given_v

 780   def _expand_base_distribution_mean(self):
 781     """Ensures `self.distribution.mean()` has `[batch, event]` shape."""
 782     single_draw_shape = concat_vectors(self.batch_shape_tensor(),
 783                                        self.event_shape_tensor())
 784     m = array_ops.reshape(
 785         self.distribution.mean(),  # A scalar.
 786         shape=array_ops.ones_like(single_draw_shape,
 787                                   dtype=dtypes.int32))
 788     m = array_ops.tile(m, multiples=single_draw_shape)
 789     m.set_shape(self.batch_shape.concatenate(self.event_shape))
 790     return m

 792   def _expand_mix_distribution_probs(self):
 793     p = self.mixture_distribution.probs  # [B, deg]
 794     deg = p.shape.with_rank_at_least(1)[-1].value
 795     if deg is None:
 796       deg = array_ops.shape(p)[-1]
 797     event_ndims = self.event_shape.ndims
 798     if event_ndims is None:
 799       event_ndims = array_ops.shape(self.event_shape_tensor())[0]
 800     expand_shape = array_ops.concat([
 801         self.mixture_distribution.batch_shape_tensor(),
 802         array_ops.ones([event_ndims], dtype=dtypes.int32),
 803         [deg],
 804     ], axis=0)
 805     return array_ops.reshape(p, shape=expand_shape)

 903 def interpolate_loc(grid, loc):
 904   """Helper which interpolates between two locs."""
 905   if len(loc) != 2:
 906     raise NotImplementedError("Currently only bimixtures are supported; "
 907                               "len(scale)={} is not 2.".format(len(loc)))
 908   deg = grid.shape.with_rank_at_least(1)[-1].value
 909   if deg is None:

 911                      "to graph execution.")

 913     if loc is None or loc[0] is None and loc[1] is None:
 914       return [None]*deg

 916     w = grid[..., array_ops.newaxis, :, :]

 918            if x is not None else None for x in loc]

 920       x = w[..., 1, :] * loc[1]                        # shape: [B, e, deg]
 921     elif loc[1] is None:
 922       x = w[..., 0, :] * loc[0]                        # shape: [B, e, deg]
 923     else:
 924       delta = loc[0] - loc[1]
 925       x = w[..., 0, :] * delta + loc[1]                # shape: [B, e, deg]
 926     return [x[..., k] for k in range(deg)]             # list(shape:[B, e])

1050 def softmax(x, axis, name=None):
1051   """Equivalent to tf.nn.softmax but works around b/70297725."""
1052   with ops.name_scope(name, "softmax", [x, axis]):
1053     x = ops.convert_to_tensor(x, name="x")
1054     ndims = (x.shape.ndims if x.shape.ndims is not None
1055              else array_ops.rank(x, name="ndims"))
1056     axis = ops.convert_to_tensor(axis, dtype=dtypes.int32, name="axis")
1057     axis_ = tensor_util.constant_value(axis)
1058     if axis_ is not None:
1059       axis = np.int(ndims + axis_ if axis_ < 0 else axis_)
1060     else:
1061       axis = array_ops.where(axis < 0, ndims + axis, axis)
1062   return nn_ops.softmax(x, axis=axis)
