  44 def mixture_stddev(mixture_weight_vector, mean_vector, stddev_vector):
  45   """Computes the standard deviation of a mixture distribution.
  46 
  47   This function works regardless of the component distribution, so long as
  48   each component's mean and standard deviation can be provided.
  49 
  50   Args:
  51     mixture_weight_vector: A 2D tensor with shape [batch_size, num_components]
  52     mean_vector: A 2D tensor of mixture component means. Has shape
  53       `[batch_size, num_components]`.
  54     stddev_vector: A 2D tensor of mixture component standard deviations. Has
  55       shape `[batch_size, num_components]`.
  56   Returns:
  57     A 1D tensor of shape `[batch_size]` representing the standard deviation of
  58     the mixture distribution with given weights and component means and standard
  59     deviations.
  60   Raises:
  61     ValueError: If the shapes of the input tensors are not as expected.
  62   """
  63   mixture_weight_vector.shape.assert_has_rank(2)
  64   if not mean_vector.shape.is_compatible_with(mixture_weight_vector.shape):
  65     raise ValueError("Expecting means to have same shape as mixture weights.")
  66   if not stddev_vector.shape.is_compatible_with(mixture_weight_vector.shape):
  67     raise ValueError("Expecting stddevs to have same shape as mixture weights.")
  68 
  69   # Reshape the distribution parameters for batched vectorized dot products.
  70   pi_for_dot_prod = array_ops.expand_dims(mixture_weight_vector, axis=1)
  71   mu_for_dot_prod = array_ops.expand_dims(mean_vector, axis=2)
  72   sigma_for_dot_prod = array_ops.expand_dims(stddev_vector, axis=2)
  73 
  74   # weighted average of component means under mixture distribution.
  75   mean_wa = math_ops.matmul(pi_for_dot_prod, mu_for_dot_prod)
  76   mean_wa = array_ops.reshape(mean_wa, (-1,))
  77   # weighted average of component variances under mixture distribution.
  78   var_wa = math_ops.matmul(pi_for_dot_prod,
  79                            math_ops.square(sigma_for_dot_prod))
  80   var_wa = array_ops.reshape(var_wa, (-1,))
  81   # weighted average of component squared means under mixture distribution.
  82   sq_mean_wa = math_ops.matmul(pi_for_dot_prod,
  83                                math_ops.square(mu_for_dot_prod))
  84   sq_mean_wa = array_ops.reshape(sq_mean_wa, (-1,))
  85   mixture_variance = var_wa + sq_mean_wa - math_ops.square(mean_wa)
  86   return math_ops.sqrt(mixture_variance)

  89 def make_tril_scale(
  90     loc=None,
  91     scale_tril=None,
  92     scale_diag=None,
  93     scale_identity_multiplier=None,
  94     shape_hint=None,
  95     validate_args=False,
  96     assert_positive=False,
  97     name=None):
  98   """Creates a LinOp representing a lower triangular matrix.
  99 
 100   Args:
 101     loc: Floating-point `Tensor`. This is used for inferring shape in the case
 102       where only `scale_identity_multiplier` is set.
 103     scale_tril: Floating-point `Tensor` representing the diagonal matrix.
 104       `scale_diag` has shape [N1, N2, ...  k, k], which represents a k x k
 105       lower triangular matrix.
 106       When `None` no `scale_tril` term is added to the LinOp.
 107       The upper triangular elements above the diagonal are ignored.
 108     scale_diag: Floating-point `Tensor` representing the diagonal matrix.
 109       `scale_diag` has shape [N1, N2, ...  k], which represents a k x k
 110       diagonal matrix.
 111       When `None` no diagonal term is added to the LinOp.
 112     scale_identity_multiplier: floating point rank 0 `Tensor` representing a
 113       scaling done to the identity matrix.
 114       When `scale_identity_multiplier = scale_diag = scale_tril = None` then
 115       `scale += IdentityMatrix`. Otherwise no scaled-identity-matrix is added
 116       to `scale`.
 117     shape_hint: scalar integer `Tensor` representing a hint at the dimension of
 118       the identity matrix when only `scale_identity_multiplier` is set.
 119     validate_args: Python `bool` indicating whether arguments should be
 120       checked for correctness.
 121     assert_positive: Python `bool` indicating whether LinOp should be checked
 122       for being positive definite.
 123     name: Python `str` name given to ops managed by this object.
 124 
 125   Returns:
 126     `LinearOperator` representing a lower triangular matrix.
 127 
 128   Raises:
 129     ValueError:  If only `scale_identity_multiplier` is set and `loc` and
 130       `shape_hint` are both None.
 131   """
 132 
 133   def _maybe_attach_assertion(x):
 134     if not validate_args:
 135       return x
 136     if assert_positive:
 137       return control_flow_ops.with_dependencies([
 138           check_ops.assert_positive(
 139               array_ops.matrix_diag_part(x),
 140               message="diagonal part must be positive"),
 141       ], x)
 142     return control_flow_ops.with_dependencies([
 143         check_ops.assert_none_equal(
 144             array_ops.matrix_diag_part(x),
 145             array_ops.zeros([], x.dtype),
 146             message="diagonal part must be non-zero"),
 147     ], x)
 148 
 149   with ops.name_scope(name, "make_tril_scale",
 150                       values=[loc, scale_diag, scale_identity_multiplier]):
 151 
 152     loc = _convert_to_tensor(loc, name="loc")
 153     scale_tril = _convert_to_tensor(scale_tril, name="scale_tril")
 154     scale_diag = _convert_to_tensor(scale_diag, name="scale_diag")
 155     scale_identity_multiplier = _convert_to_tensor(
 156         scale_identity_multiplier,
 157         name="scale_identity_multiplier")
 158 
 159   if scale_tril is not None:
 160     scale_tril = array_ops.matrix_band_part(scale_tril, -1, 0)  # Zero out TriU.
 161     tril_diag = array_ops.matrix_diag_part(scale_tril)
 162     if scale_diag is not None:
 163       tril_diag += scale_diag
 164     if scale_identity_multiplier is not None:
 165       tril_diag += scale_identity_multiplier[..., array_ops.newaxis]
 166 
 167     scale_tril = array_ops.matrix_set_diag(scale_tril, tril_diag)
 168 
 169     return linalg.LinearOperatorLowerTriangular(
 170         tril=_maybe_attach_assertion(scale_tril),
 171         is_non_singular=True,
 172         is_self_adjoint=False,
 173         is_positive_definite=assert_positive)
 174 
 175   return make_diag_scale(
 176       loc=loc,
 177       scale_diag=scale_diag,
 178       scale_identity_multiplier=scale_identity_multiplier,
 179       shape_hint=shape_hint,
 180       validate_args=validate_args,
 181       assert_positive=assert_positive,
 182       name=name)

 185 def make_diag_scale(
 186     loc=None,
 187     scale_diag=None,
 188     scale_identity_multiplier=None,
 189     shape_hint=None,
 190     validate_args=False,
 191     assert_positive=False,
 192     name=None):
 193   """Creates a LinOp representing a diagonal matrix.
 194 
 195   Args:
 196     loc: Floating-point `Tensor`. This is used for inferring shape in the case
 197       where only `scale_identity_multiplier` is set.
 198     scale_diag: Floating-point `Tensor` representing the diagonal matrix.
 199       `scale_diag` has shape [N1, N2, ...  k], which represents a k x k
 200       diagonal matrix.
 201       When `None` no diagonal term is added to the LinOp.
 202     scale_identity_multiplier: floating point rank 0 `Tensor` representing a
 203       scaling done to the identity matrix.
 204       When `scale_identity_multiplier = scale_diag = scale_tril = None` then
 205       `scale += IdentityMatrix`. Otherwise no scaled-identity-matrix is added
 206       to `scale`.
 207     shape_hint: scalar integer `Tensor` representing a hint at the dimension of
 208       the identity matrix when only `scale_identity_multiplier` is set.
 209     validate_args: Python `bool` indicating whether arguments should be
 210       checked for correctness.
 211     assert_positive: Python `bool` indicating whether LinOp should be checked
 212       for being positive definite.
 213     name: Python `str` name given to ops managed by this object.
 214 
 215   Returns:
 216     `LinearOperator` representing a lower triangular matrix.
 217 
 218   Raises:
 219     ValueError:  If only `scale_identity_multiplier` is set and `loc` and
 220       `shape_hint` are both None.
 221   """
 222 
 223   def _maybe_attach_assertion(x):
 224     if not validate_args:
 225       return x
 226     if assert_positive:
 227       return control_flow_ops.with_dependencies([
 228           check_ops.assert_positive(
 229               x, message="diagonal part must be positive"),
 230       ], x)
 231     return control_flow_ops.with_dependencies([
 232         check_ops.assert_none_equal(
 233             x,
 234             array_ops.zeros([], x.dtype),
 235             message="diagonal part must be non-zero")], x)
 236 
 237   with ops.name_scope(name, "make_diag_scale",
 238                       values=[loc, scale_diag, scale_identity_multiplier]):
 239     loc = _convert_to_tensor(loc, name="loc")
 240     scale_diag = _convert_to_tensor(scale_diag, name="scale_diag")
 241     scale_identity_multiplier = _convert_to_tensor(
 242         scale_identity_multiplier,
 243         name="scale_identity_multiplier")
 244 
 245     if scale_diag is not None:
 246       if scale_identity_multiplier is not None:
 247         scale_diag += scale_identity_multiplier[..., array_ops.newaxis]
 248       return linalg.LinearOperatorDiag(
 249           diag=_maybe_attach_assertion(scale_diag),
 250           is_non_singular=True,
 251           is_self_adjoint=True,
 252           is_positive_definite=assert_positive)
 253 
 254     if loc is None and shape_hint is None:
 255       raise ValueError(
 256           "Cannot infer `event_shape` unless `loc` or "
 257           "`shape_hint` is specified.")
 258 
 259     if shape_hint is None:
 260       shape_hint = loc.shape[-1]
 261 
 262     if scale_identity_multiplier is None:
 263       return linalg.LinearOperatorIdentity(
 264           num_rows=shape_hint,
 265           dtype=loc.dtype.base_dtype,
 266           is_self_adjoint=True,
 267           is_positive_definite=True,
 268           assert_proper_shapes=validate_args)
 269 
 270     return linalg.LinearOperatorScaledIdentity(
 271         num_rows=shape_hint,
 272         multiplier=_maybe_attach_assertion(scale_identity_multiplier),
 273         is_non_singular=True,
 274         is_self_adjoint=True,
 275         is_positive_definite=assert_positive,
 276         assert_proper_shapes=validate_args)

 279 def shapes_from_loc_and_scale(loc, scale, name="shapes_from_loc_and_scale"):
 280   """Infer distribution batch and event shapes from a location and scale.
 281 
 282   Location and scale family distributions determine their batch/event shape by
 283   broadcasting the `loc` and `scale` args.  This helper does that broadcast,
 284   statically if possible.
 285 
 286   Batch shape broadcasts as per the normal rules.
 287   We allow the `loc` event shape to broadcast up to that of `scale`.  We do not
 288   allow `scale`'s event shape to change.  Therefore, the last dimension of `loc`
 289   must either be size `1`, or the same as `scale.range_dimension`.
 290 
 291   See `MultivariateNormalLinearOperator` for a usage example.
 292 
 293   Args:
 294     loc:  `N-D` `Tensor` with `N >= 1` (already converted to tensor) or `None`.
 295       If `None`, both batch and event shape are determined by `scale`.
 296     scale:  A `LinearOperator` instance.
 297     name:  A string name to prepend to created ops.
 298 
 299   Returns:
 300     batch_shape:  `TensorShape` (if broadcast is done statically), or `Tensor`.
 301     event_shape:  `TensorShape` (if broadcast is done statically), or `Tensor`.
 302 
 303   Raises:
 304     ValueError:  If the last dimension of `loc` is determined statically to be
 305       different than the range of `scale`.
 306   """
 307   with ops.name_scope(name, values=[loc] + scale.graph_parents):
 308     # Get event shape.
 309     event_size = scale.range_dimension_tensor()
 310     event_size_const = tensor_util.constant_value(event_size)
 311     if event_size_const is not None:
 312       event_shape = event_size_const.reshape([1])
 313     else:
 314       event_shape = event_size[array_ops.newaxis]
 315 
 316     # Static check that event shapes match.
 317     if loc is not None:
 318       loc_event_size = loc.get_shape()[-1].value
 319       if loc_event_size is not None and event_size_const is not None:
 320         if loc_event_size != 1 and loc_event_size != event_size_const:
 321           raise ValueError(
 322               "Event size of 'scale' (%d) could not be broadcast up to that of "
 323               "'loc' (%d)." % (loc_event_size, event_size_const))
 324 
 325     # Get batch shape.
 326     batch_shape = scale.batch_shape_tensor()
 327     if loc is None:
 328       batch_shape_const = tensor_util.constant_value(batch_shape)
 329       batch_shape = (
 330           batch_shape_const if batch_shape_const is not None else batch_shape)
 331     else:
 332       loc_batch_shape = loc.get_shape().with_rank_at_least(1)[:-1]
 333       if (loc.get_shape().ndims is None or
 334           not loc_batch_shape.is_fully_defined()):
 335         loc_batch_shape = array_ops.shape(loc)[:-1]
 336       else:
 337         loc_batch_shape = ops.convert_to_tensor(loc_batch_shape,
 338                                                 name="loc_batch_shape")
 339       # This is defined in the core util module.
 340       # pylint: disable=undefined-variable
 341       batch_shape = prefer_static_broadcast_shape(batch_shape, loc_batch_shape)
 342       # pylint: enable=undefined-variable
 343 
 344   return batch_shape, event_shape

 347 def get_broadcast_shape(*tensors):
 348   """Get broadcast shape as a Python list of integers (preferred) or `Tensor`.
 349 
 350   Args:
 351     *tensors:  One or more `Tensor` objects (already converted!).
 352 
 353   Returns:
 354     broadcast shape:  Python list (if shapes determined statically), otherwise
 355       an `int32` `Tensor`.
 356   """
 357   # Try static.
 358   s_shape = tensors[0].shape
 359   for t in tensors[1:]:
 360     s_shape = array_ops.broadcast_static_shape(s_shape, t.shape)
 361   if s_shape.is_fully_defined():
 362     return s_shape.as_list()
 363 
 364   # Fallback on dynamic.
 365   d_shape = array_ops.shape(tensors[0])
 366   for t in tensors[1:]:
 367     d_shape = array_ops.broadcast_dynamic_shape(d_shape, array_ops.shape(t))
 368   return d_shape

 371 def is_diagonal_scale(scale):
 372   """Returns `True` if `scale` is a `LinearOperator` that is known to be diag.
 373 
 374   Args:
 375     scale:  `LinearOperator` instance.
 376 
 377   Returns:
 378     Python `bool`.
 379 
 380   Raises:
 381     TypeError:  If `scale` is not a `LinearOperator`.
 382   """
 383   if not isinstance(scale, linalg.LinearOperator):
 384     raise TypeError("Expected argument 'scale' to be instance of LinearOperator"
 385                     ". Found: %s" % scale)
 386   return (isinstance(scale, linalg.LinearOperatorIdentity) or
 387           isinstance(scale, linalg.LinearOperatorScaledIdentity) or
 388           isinstance(scale, linalg.LinearOperatorDiag))

 391 def maybe_check_scalar_distribution(
 392     distribution, expected_base_dtype, validate_args):
 393   """Helper which checks validity of a scalar `distribution` init arg.
 394 
 395   Valid here means:
 396 
 397   * `distribution` has scalar batch and event shapes.
 398   * `distribution` is `FULLY_REPARAMETERIZED`
 399   * `distribution` has expected dtype.
 400 
 401   Args:
 402     distribution:  `Distribution`-like object.
 403     expected_base_dtype:  `TensorFlow` `dtype`.
 404     validate_args:  Python `bool`.  Whether to do additional checks:
 405       (i)  check that reparameterization_type is `FULLY_REPARAMETERIZED`.
 406       (ii) add `tf.Assert` ops to the graph to enforce that distribution
 407            is scalar in the event that this cannot be determined statically.
 408 
 409   Returns:
 410     List of `tf.Assert` ops to run to enforce validity checks that could not
 411       be statically determined.  Empty if `not validate_args`.
 412 
 413   Raises:
 414     ValueError:  If validate_args and distribution is not FULLY_REPARAMETERIZED
 415     ValueError:  If distribution is statically determined to not have both
 416       scalar batch and scalar event shapes.
 417   """
 418   if distribution.dtype != expected_base_dtype:
 419     raise TypeError("dtype mismatch; "
 420                     "distribution.dtype=\"{}\" is not \"{}\"".format(
 421                         distribution.dtype.name, expected_base_dtype.name))
 422 
 423   # Although `reparameterization_type` is a static property, we guard it by
 424   # `validate_args`. This allows users to use a `distribution` which is not
 425   # reparameterized itself. However, we tacitly assume that although the
 426   # distribution is not reparameterized, it only depends on non-trainable
 427   # variables.
 428   if validate_args and (distribution.reparameterization_type
 429                         != distribution_lib.FULLY_REPARAMETERIZED):
 430     raise ValueError("Base distribution should be reparameterized or be "
 431                      "a function of non-trainable variables; "
 432                      "distribution.reparameterization_type = \"{}\" "
 433                      "!= \"FULLY_REPARAMETERIZED\".".format(
 434                          distribution.reparameterization_type))
 435   with ops.name_scope(name="check_distribution"):
 436     assertions = []
 437     def check_is_scalar(is_scalar, name):
 438       is_scalar_ = static_value(is_scalar)
 439       if is_scalar_ is not None:
 440         if not is_scalar_:
 441           raise ValueError("distribution must be scalar; "
 442                            "distribution.{}=False is not True".format(name))
 443       elif validate_args:
 444         assertions.append(check_ops.assert_equal(
 445             is_scalar, True,
 446             message=("distribution must be scalar; "
 447                      "distribution.{}=False is not True".format(name))))
 448     check_is_scalar(distribution.is_scalar_event(), "is_scalar_event")
 449     check_is_scalar(distribution.is_scalar_batch(), "is_scalar_batch")
 450     return assertions

 453 def pad_mixture_dimensions(x, mixture_distribution, categorical_distribution,
 454                            event_ndims):
 455   """Pad dimensions of event tensors for mixture distributions.
 456 
 457   See `Mixture._sample_n` and `MixtureSameFamily._sample_n` for usage examples.
 458 
 459   Args:
 460     x: event tensor to pad.
 461     mixture_distribution: Base distribution of the mixture.
 462     categorical_distribution: `Categorical` distribution that mixes the base
 463       distribution.
 464     event_ndims: Integer specifying the number of event dimensions in the event
 465       tensor.
 466 
 467   Returns:
 468     A padded version of `x` that can broadcast with `categorical_distribution`.
 469   """
 470   with ops.name_scope("pad_mix_dims", values=[x]):
 471     def _get_ndims(d):
 472       if d.batch_shape.ndims is not None:
 473         return d.batch_shape.ndims
 474       return array_ops.shape(d.batch_shape_tensor())[0]
 475     dist_batch_ndims = _get_ndims(mixture_distribution)
 476     cat_batch_ndims = _get_ndims(categorical_distribution)
 477     pad_ndims = array_ops.where(
 478         categorical_distribution.is_scalar_batch(),
 479         dist_batch_ndims,
 480         dist_batch_ndims - cat_batch_ndims)
 481     s = array_ops.shape(x)
 482     x = array_ops.reshape(x, shape=array_ops.concat([
 483         s[:-1],
 484         array_ops.ones([pad_ndims], dtype=dtypes.int32),
 485         s[-1:],
 486         array_ops.ones([event_ndims], dtype=dtypes.int32),
 487     ], axis=0))
 488     return x

 496 def move_dimension(x, source_idx, dest_idx):
 497   """Move a single tensor dimension within its shape.
 498 
 499   This is a special case of `tf.transpose()`, which applies
 500   arbitrary permutations to tensor dimensions.
 501 
 502   Args:
 503     x: Tensor of rank `ndims`.
 504     source_idx: Integer index into `x.shape` (negative indexing is
 505       supported).
 506     dest_idx: Integer index into `x.shape` (negative indexing is
 507       supported).
 508 
 509   Returns:
 510     x_perm: Tensor of rank `ndims`, in which the dimension at original
 511      index `source_idx` has been moved to new index `dest_idx`, with
 512      all other dimensions retained in their original order.
 513 
 514   Example:
 515 
 516   ```python
 517   x = tf.placeholder(shape=[200, 30, 4, 1, 6])
 518   x_perm = _move_dimension(x, 1, 1) # no-op
 519   x_perm = _move_dimension(x, 0, 3) # result shape [30, 4, 1, 200, 6]
 520   x_perm = _move_dimension(x, 0, -2) # equivalent to previous
 521   x_perm = _move_dimension(x, 4, 2) # result shape [200, 30, 6, 4, 1]
 522   ```
 523   """
 524   ndims = util.prefer_static_rank(x)
 525   if isinstance(source_idx, int):
 526     dtype = dtypes.int32
 527   else:
 528     dtype = dtypes.as_dtype(source_idx.dtype)
 529 
 530   # Handle negative indexing. Since ndims might be dynamic, this makes
 531   # source_idx and dest_idx also possibly dynamic.
 532   if source_idx < 0:
 533     source_idx = ndims + source_idx
 534   if dest_idx < 0:
 535     dest_idx = ndims + dest_idx
 536 
 537   # Construct the appropriate permutation of dimensions, depending
 538   # whether the source is before or after the destination.
 539   def move_left_permutation():
 540     return util.prefer_static_value(
 541         array_ops.concat([
 542             math_ops.range(0, dest_idx, dtype=dtype),
 543             [source_idx],
 544             math_ops.range(dest_idx, source_idx, dtype=dtype),
 545             math_ops.range(source_idx+1, ndims, dtype=dtype)], axis=0))
 546 
 547   def move_right_permutation():
 548     return util.prefer_static_value(
 549         array_ops.concat([
 550             math_ops.range(0, source_idx, dtype=dtype),
 551             math_ops.range(source_idx+1, dest_idx+1, dtype=dtype),
 552             [source_idx],
 553             math_ops.range(dest_idx+1, ndims, dtype=dtype)], axis=0))
 554 
 555   def x_permuted():
 556     return array_ops.transpose(
 557         x, perm=smart_cond.smart_cond(source_idx < dest_idx,
 558                                       move_right_permutation,
 559                                       move_left_permutation))
 560 
 561   # One final conditional to handle the special case where source
 562   # and destination indices are equal.
 563   return smart_cond.smart_cond(math_ops.equal(source_idx, dest_idx),
 564                                lambda: x,
 565                                x_permuted)
