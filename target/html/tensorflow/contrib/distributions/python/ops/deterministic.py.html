  43 @six.add_metaclass(abc.ABCMeta)
  44 class _BaseDeterministic(distribution.Distribution):
  45   """Base class for Deterministic distributions."""
  46 
  47   @deprecation.deprecated(
  48       "2018-10-01",
  49       "The TensorFlow Distributions library has moved to "
  50       "TensorFlow Probability "
  51       "(https://github.com/tensorflow/probability). You "
  52       "should update all references to use `tfp.distributions` "
  53       "instead of `tf.contrib.distributions`.",
  54       warn_once=True)
  55   def __init__(self,
  56                loc,
  57                atol=None,
  58                rtol=None,
  59                is_vector=False,
  60                validate_args=False,
  61                allow_nan_stats=True,
  62                name="_BaseDeterministic"):
  63     """Initialize a batch of `_BaseDeterministic` distributions.
  64 
  65     The `atol` and `rtol` parameters allow for some slack in `pmf`, `cdf`
  66     computations, e.g. due to floating-point error.
  67 
  68     ```
  69     pmf(x; loc)
  70       = 1, if Abs(x - loc) <= atol + rtol * Abs(loc),
  71       = 0, otherwise.
  72     ```
  73 
  74     Args:
  75       loc: Numeric `Tensor`.  The point (or batch of points) on which this
  76         distribution is supported.
  77       atol:  Non-negative `Tensor` of same `dtype` as `loc` and broadcastable
  78         shape.  The absolute tolerance for comparing closeness to `loc`.
  79         Default is `0`.
  80       rtol:  Non-negative `Tensor` of same `dtype` as `loc` and broadcastable
  81         shape.  The relative tolerance for comparing closeness to `loc`.
  82         Default is `0`.
  83       is_vector:  Python `bool`.  If `True`, this is for `VectorDeterministic`,
  84         else `Deterministic`.
  85       validate_args: Python `bool`, default `False`. When `True` distribution
  86         parameters are checked for validity despite possibly degrading runtime
  87         performance. When `False` invalid inputs may silently render incorrect
  88         outputs.
  89       allow_nan_stats: Python `bool`, default `True`. When `True`, statistics
  90         (e.g., mean, mode, variance) use the value "`NaN`" to indicate the
  91         result is undefined. When `False`, an exception is raised if one or
  92         more of the statistic's batch members are undefined.
  93       name: Python `str` name prefixed to Ops created by this class.
  94 
  95     Raises:
  96       ValueError:  If `loc` is a scalar.
  97     """
  98     parameters = dict(locals())
  99     with ops.name_scope(name, values=[loc, atol, rtol]) as name:
 100       loc = ops.convert_to_tensor(loc, name="loc")
 101       if is_vector and validate_args:
 102         msg = "Argument loc must be at least rank 1."
 103         if loc.get_shape().ndims is not None:
 104           if loc.get_shape().ndims < 1:
 105             raise ValueError(msg)
 106         else:
 107           loc = control_flow_ops.with_dependencies(
 108               [check_ops.assert_rank_at_least(loc, 1, message=msg)], loc)
 109       self._loc = loc
 110 
 111       super(_BaseDeterministic, self).__init__(
 112           dtype=self._loc.dtype,
 113           reparameterization_type=distribution.NOT_REPARAMETERIZED,
 114           validate_args=validate_args,
 115           allow_nan_stats=allow_nan_stats,
 116           parameters=parameters,
 117           graph_parents=[self._loc],
 118           name=name)
 119 
 120       self._atol = self._get_tol(atol)
 121       self._rtol = self._get_tol(rtol)
 122       # Avoid using the large broadcast with self.loc if possible.
 123       if rtol is None:
 124         self._slack = self.atol
 125       else:
 126         self._slack = self.atol + self.rtol * math_ops.abs(self.loc)
 127 
 128   def _get_tol(self, tol):
 129     if tol is None:
 130       return ops.convert_to_tensor(0, dtype=self.loc.dtype)
 131 
 132     tol = ops.convert_to_tensor(tol, dtype=self.loc.dtype)
 133     if self.validate_args:
 134       tol = control_flow_ops.with_dependencies([
 135           check_ops.assert_non_negative(
 136               tol, message="Argument 'tol' must be non-negative")
 137       ], tol)
 138     return tol
 139 
 140   @property
 141   def loc(self):
 142     """Point (or batch of points) at which this distribution is supported."""
 143     return self._loc
 144 
 145   @property
 146   def atol(self):
 147     """Absolute tolerance for comparing points to `self.loc`."""
 148     return self._atol
 149 
 150   @property
 151   def rtol(self):
 152     """Relative tolerance for comparing points to `self.loc`."""
 153     return self._rtol
 154 
 155   def _entropy(self):
 156     return array_ops.zeros(self.batch_shape_tensor(), dtype=self.dtype)
 157 
 158   def _mean(self):
 159     return array_ops.identity(self.loc)
 160 
 161   def _variance(self):
 162     return array_ops.zeros_like(self.loc)
 163 
 164   def _mode(self):
 165     return self.mean()
 166 
 167   def _sample_n(self, n, seed=None):  # pylint: disable=unused-arg
 168     n_static = tensor_util.constant_value(ops.convert_to_tensor(n))
 169     if n_static is not None and self.loc.get_shape().ndims is not None:
 170       ones = [1] * self.loc.get_shape().ndims
 171       multiples = [n_static] + ones
 172     else:
 173       ones = array_ops.ones_like(array_ops.shape(self.loc))
 174       multiples = array_ops.concat(([n], ones), axis=0)
 175 
 176     return array_ops.tile(self.loc[array_ops.newaxis, ...], multiples=multiples)

 179 class Deterministic(_BaseDeterministic):
 180   """Scalar `Deterministic` distribution on the real line.
 181 
 182   The scalar `Deterministic` distribution is parameterized by a [batch] point
 183   `loc` on the real line.  The distribution is supported at this point only,
 184   and corresponds to a random variable that is constant, equal to `loc`.
 185 
 186   See [Degenerate rv](https://en.wikipedia.org/wiki/Degenerate_distribution).
 187 
 188   #### Mathematical Details
 189 
 190   The probability mass function (pmf) and cumulative distribution function (cdf)
 191   are
 192 
 193   ```none
 194   pmf(x; loc) = 1, if x == loc, else 0
 195   cdf(x; loc) = 1, if x >= loc, else 0
 196   ```
 197 
 198   #### Examples
 199 
 200   ```python
 201   import tensorflow_probability as tfp
 202   tfd = tfp.distributions
 203 
 204   # Initialize a single Deterministic supported at zero.
 205   constant = tfd.Deterministic(0.)
 206   constant.prob(0.)
 207   ==> 1.
 208   constant.prob(2.)
 209   ==> 0.
 210 
 211   # Initialize a [2, 2] batch of scalar constants.
 212   loc = [[0., 1.], [2., 3.]]
 213   x = [[0., 1.1], [1.99, 3.]]
 214   constant = tfd.Deterministic(loc)
 215   constant.prob(x)
 216   ==> [[1., 0.], [0., 1.]]
 217   ```
 218 
 219   """
 220 
 221   @deprecation.deprecated(
 222       "2018-10-01",
 223       "The TensorFlow Distributions library has moved to "
 224       "TensorFlow Probability "
 225       "(https://github.com/tensorflow/probability). You "
 226       "should update all references to use `tfp.distributions` "
 227       "instead of `tf.contrib.distributions`.",
 228       warn_once=True)
 229   def __init__(self,
 230                loc,
 231                atol=None,
 232                rtol=None,
 233                validate_args=False,
 234                allow_nan_stats=True,
 235                name="Deterministic"):
 236     """Initialize a scalar `Deterministic` distribution.
 237 
 238     The `atol` and `rtol` parameters allow for some slack in `pmf`, `cdf`
 239     computations, e.g. due to floating-point error.
 240 
 241     ```
 242     pmf(x; loc)
 243       = 1, if Abs(x - loc) <= atol + rtol * Abs(loc),
 244       = 0, otherwise.
 245     ```
 246 
 247     Args:
 248       loc: Numeric `Tensor` of shape `[B1, ..., Bb]`, with `b >= 0`.
 249         The point (or batch of points) on which this distribution is supported.
 250       atol:  Non-negative `Tensor` of same `dtype` as `loc` and broadcastable
 251         shape.  The absolute tolerance for comparing closeness to `loc`.
 252         Default is `0`.
 253       rtol:  Non-negative `Tensor` of same `dtype` as `loc` and broadcastable
 254         shape.  The relative tolerance for comparing closeness to `loc`.
 255         Default is `0`.
 256       validate_args: Python `bool`, default `False`. When `True` distribution
 257         parameters are checked for validity despite possibly degrading runtime
 258         performance. When `False` invalid inputs may silently render incorrect
 259         outputs.
 260       allow_nan_stats: Python `bool`, default `True`. When `True`, statistics
 261         (e.g., mean, mode, variance) use the value "`NaN`" to indicate the
 262         result is undefined. When `False`, an exception is raised if one or
 263         more of the statistic's batch members are undefined.
 264       name: Python `str` name prefixed to Ops created by this class.
 265     """
 266     super(Deterministic, self).__init__(
 267         loc,
 268         atol=atol,
 269         rtol=rtol,
 270         validate_args=validate_args,
 271         allow_nan_stats=allow_nan_stats,
 272         name=name)
 273 
 274   def _batch_shape_tensor(self):
 275     return array_ops.shape(self.loc)
 276 
 277   def _batch_shape(self):
 278     return self.loc.get_shape()
 279 
 280   def _event_shape_tensor(self):
 281     return constant_op.constant([], dtype=dtypes.int32)
 282 
 283   def _event_shape(self):
 284     return tensor_shape.scalar()
 285 
 286   def _prob(self, x):
 287     return math_ops.cast(
 288         math_ops.abs(x - self.loc) <= self._slack, dtype=self.dtype)
 289 
 290   def _cdf(self, x):
 291     return math_ops.cast(x >= self.loc - self._slack, dtype=self.dtype)

 294 class VectorDeterministic(_BaseDeterministic):
 295   """Vector `Deterministic` distribution on `R^k`.
 296 
 297   The `VectorDeterministic` distribution is parameterized by a [batch] point
 298   `loc in R^k`.  The distribution is supported at this point only,
 299   and corresponds to a random variable that is constant, equal to `loc`.
 300 
 301   See [Degenerate rv](https://en.wikipedia.org/wiki/Degenerate_distribution).
 302 
 303   #### Mathematical Details
 304 
 305   The probability mass function (pmf) is
 306 
 307   ```none
 308   pmf(x; loc)
 309     = 1, if All[Abs(x - loc) <= atol + rtol * Abs(loc)],
 310     = 0, otherwise.
 311   ```
 312 
 313   #### Examples
 314 
 315   ```python
 316   import tensorflow_probability as tfp
 317   tfd = tfp.distributions
 318 
 319   # Initialize a single VectorDeterministic supported at [0., 2.] in R^2.
 320   constant = tfd.Deterministic([0., 2.])
 321   constant.prob([0., 2.])
 322   ==> 1.
 323   constant.prob([0., 3.])
 324   ==> 0.
 325 
 326   # Initialize a [3] batch of constants on R^2.
 327   loc = [[0., 1.], [2., 3.], [4., 5.]]
 328   constant = tfd.VectorDeterministic(loc)
 329   constant.prob([[0., 1.], [1.9, 3.], [3.99, 5.]])
 330   ==> [1., 0., 0.]
 331   ```
 332 
 333   """
 334 
 335   @deprecation.deprecated(
 336       "2018-10-01",
 337       "The TensorFlow Distributions library has moved to "
 338       "TensorFlow Probability "
 339       "(https://github.com/tensorflow/probability). You "
 340       "should update all references to use `tfp.distributions` "
 341       "instead of `tf.contrib.distributions`.",
 342       warn_once=True)
 343   def __init__(self,
 344                loc,
 345                atol=None,
 346                rtol=None,
 347                validate_args=False,
 348                allow_nan_stats=True,
 349                name="VectorDeterministic"):
 350     """Initialize a `VectorDeterministic` distribution on `R^k`, for `k >= 0`.
 351 
 352     Note that there is only one point in `R^0`, the "point" `[]`.  So if `k = 0`
 353     then `self.prob([]) == 1`.
 354 
 355     The `atol` and `rtol` parameters allow for some slack in `pmf`
 356     computations, e.g. due to floating-point error.
 357 
 358     ```
 359     pmf(x; loc)
 360       = 1, if All[Abs(x - loc) <= atol + rtol * Abs(loc)],
 361       = 0, otherwise
 362     ```
 363 
 364     Args:
 365       loc: Numeric `Tensor` of shape `[B1, ..., Bb, k]`, with `b >= 0`, `k >= 0`
 366         The point (or batch of points) on which this distribution is supported.
 367       atol:  Non-negative `Tensor` of same `dtype` as `loc` and broadcastable
 368         shape.  The absolute tolerance for comparing closeness to `loc`.
 369         Default is `0`.
 370       rtol:  Non-negative `Tensor` of same `dtype` as `loc` and broadcastable
 371         shape.  The relative tolerance for comparing closeness to `loc`.
 372         Default is `0`.
 373       validate_args: Python `bool`, default `False`. When `True` distribution
 374         parameters are checked for validity despite possibly degrading runtime
 375         performance. When `False` invalid inputs may silently render incorrect
 376         outputs.
 377       allow_nan_stats: Python `bool`, default `True`. When `True`, statistics
 378         (e.g., mean, mode, variance) use the value "`NaN`" to indicate the
 379         result is undefined. When `False`, an exception is raised if one or
 380         more of the statistic's batch members are undefined.
 381       name: Python `str` name prefixed to Ops created by this class.
 382     """
 383     super(VectorDeterministic, self).__init__(
 384         loc,
 385         atol=atol,
 386         rtol=rtol,
 387         is_vector=True,
 388         validate_args=validate_args,
 389         allow_nan_stats=allow_nan_stats,
 390         name=name)
 391 
 392   def _batch_shape_tensor(self):
 393     return array_ops.shape(self.loc)[:-1]
 394 
 395   def _batch_shape(self):
 396     return self.loc.get_shape()[:-1]
 397 
 398   def _event_shape_tensor(self):
 399     return array_ops.shape(self.loc)[-1]
 400 
 401   def _event_shape(self):
 402     return self.loc.get_shape()[-1:]
 403 
 404   def _prob(self, x):
 405     if self.validate_args:
 406       is_vector_check = check_ops.assert_rank_at_least(x, 1)
 407       right_vec_space_check = check_ops.assert_equal(
 408           self.event_shape_tensor(),
 409           array_ops.gather(array_ops.shape(x), array_ops.rank(x) - 1),
 410           message=
 411           "Argument 'x' not defined in the same space R^k as this distribution")
 412       with ops.control_dependencies([is_vector_check]):
 413         with ops.control_dependencies([right_vec_space_check]):
 414           x = array_ops.identity(x)
 415     return math_ops.cast(
 416         math_ops.reduce_all(math_ops.abs(x - self.loc) <= self._slack, axis=-1),
 417         dtype=self.dtype)
