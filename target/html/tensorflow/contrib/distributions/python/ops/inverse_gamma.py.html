  44 class InverseGamma(distribution.Distribution):
  45   """InverseGamma distribution.
  46 
  47   The `InverseGamma` distribution is defined over positive real numbers using
  48   parameters `concentration` (aka "alpha") and `rate` (aka "beta").
  49 
  50   #### Mathematical Details
  51 
  52   The probability density function (pdf) is,
  53 
  54   ```none
  55   pdf(x; alpha, beta, x > 0) = x**(-alpha - 1) exp(-beta / x) / Z
  56   Z = Gamma(alpha) beta**-alpha
  57   ```
  58 
  59   where:
  60 
  61   * `concentration = alpha`,
  62   * `rate = beta`,
  63   * `Z` is the normalizing constant, and,
  64   * `Gamma` is the [gamma function](
  65     https://en.wikipedia.org/wiki/Gamma_function).
  66 
  67   The cumulative density function (cdf) is,
  68 
  69   ```none
  70   cdf(x; alpha, beta, x > 0) = GammaInc(alpha, beta / x) / Gamma(alpha)
  71   ```
  72 
  73   where `GammaInc` is the [upper incomplete Gamma function](
  74   https://en.wikipedia.org/wiki/Incomplete_gamma_function).
  75 
  76   The parameters can be intuited via their relationship to mean and stddev,
  77 
  78   ```none
  79   concentration = alpha = (mean / stddev)**2
  80   rate = beta = mean / stddev**2
  81   ```
  82 
  83   Distribution parameters are automatically broadcast in all functions; see
  84   examples for details.
  85 
  86   WARNING: This distribution may draw 0-valued samples for small concentration
  87   values. See note in `tf.random_gamma` docstring.
  88 
  89   #### Examples
  90 
  91   ```python
  92   import tensorflow_probability as tfp
  93   tfd = tfp.distributions
  94 
  95   dist = tfd.InverseGamma(concentration=3.0, rate=2.0)
  96   dist2 = tfd.InverseGamma(concentration=[3.0, 4.0], rate=[2.0, 3.0])
  97   ```
  98 
  99   """
 100 
 101   @deprecation.deprecated(
 102       "2018-10-01",
 103       "The TensorFlow Distributions library has moved to "
 104       "TensorFlow Probability "
 105       "(https://github.com/tensorflow/probability). You "
 106       "should update all references to use `tfp.distributions` "
 107       "instead of `tf.contrib.distributions`.",
 108       warn_once=True)
 109   def __init__(self,
 110                concentration,
 111                rate,
 112                validate_args=False,
 113                allow_nan_stats=True,
 114                name="InverseGamma"):
 115     """Construct InverseGamma with `concentration` and `rate` parameters.
 116 
 117     The parameters `concentration` and `rate` must be shaped in a way that
 118     supports broadcasting (e.g. `concentration + rate` is a valid operation).
 119 
 120     Args:
 121       concentration: Floating point tensor, the concentration params of the
 122         distribution(s). Must contain only positive values.
 123       rate: Floating point tensor, the inverse scale params of the
 124         distribution(s). Must contain only positive values.
 125       validate_args: Python `bool`, default `False`. When `True` distribution
 126         parameters are checked for validity despite possibly degrading runtime
 127         performance. When `False` invalid inputs may silently render incorrect
 128         outputs.
 129       allow_nan_stats: Python `bool`, default `True`. When `True`, statistics
 130         (e.g., mean, mode, variance) use the value "`NaN`" to indicate the
 131         result is undefined. When `False`, an exception is raised if one or
 132         more of the statistic's batch members are undefined.
 133       name: Python `str` name prefixed to Ops created by this class.
 134 
 135 
 136     Raises:
 137       TypeError: if `concentration` and `rate` are different dtypes.
 138     """
 139     parameters = dict(locals())
 140     with ops.name_scope(name, values=[concentration, rate]) as name:
 141       with ops.control_dependencies([
 142           check_ops.assert_positive(concentration),
 143           check_ops.assert_positive(rate),
 144       ] if validate_args else []):
 145         self._concentration = array_ops.identity(
 146             concentration, name="concentration")
 147         self._rate = array_ops.identity(rate, name="rate")
 148         check_ops.assert_same_float_dtype(
 149             [self._concentration, self._rate])
 150     super(InverseGamma, self).__init__(
 151         dtype=self._concentration.dtype,
 152         validate_args=validate_args,
 153         allow_nan_stats=allow_nan_stats,
 154         reparameterization_type=distribution.NOT_REPARAMETERIZED,
 155         parameters=parameters,
 156         graph_parents=[self._concentration,
 157                        self._rate],
 158         name=name)
 159 
 160   @staticmethod
 161   def _param_shapes(sample_shape):
 162     return dict(
 163         zip(("concentration", "rate"), ([ops.convert_to_tensor(
 164             sample_shape, dtype=dtypes.int32)] * 2)))
 165 
 166   @property
 167   def concentration(self):
 168     """Concentration parameter."""
 169     return self._concentration
 170 
 171   @property
 172   def rate(self):
 173     """Rate parameter."""
 174     return self._rate
 175 
 176   def _batch_shape_tensor(self):
 177     return array_ops.broadcast_dynamic_shape(
 178         array_ops.shape(self.concentration),
 179         array_ops.shape(self.rate))
 180 
 181   def _batch_shape(self):
 182     return array_ops.broadcast_static_shape(
 183         self.concentration.get_shape(),
 184         self.rate.get_shape())
 185 
 186   def _event_shape_tensor(self):
 187     return constant_op.constant([], dtype=dtypes.int32)
 188 
 189   def _event_shape(self):
 190     return tensor_shape.scalar()
 191 
 192   @distribution_util.AppendDocstring(
 193       """Note: See `tf.random_gamma` docstring for sampling details and
 194       caveats.""")
 195   def _sample_n(self, n, seed=None):
 196     return 1. / random_ops.random_gamma(
 197         shape=[n],
 198         alpha=self.concentration,
 199         beta=self.rate,
 200         dtype=self.dtype,
 201         seed=seed)
 202 
 203   def _log_prob(self, x):
 204     return self._log_unnormalized_prob(x) - self._log_normalization()
 205 
 206   def _cdf(self, x):
 207     x = self._maybe_assert_valid_sample(x)
 208     # Note that igammac returns the upper regularized incomplete gamma
 209     # function Q(a, x), which is what we want for the CDF.
 210     return math_ops.igammac(self.concentration, self.rate / x)
 211 
 212   def _log_unnormalized_prob(self, x):
 213     x = self._maybe_assert_valid_sample(x)
 214     return -(1. + self.concentration) * math_ops.log(x) - self.rate / x
 215 
 216   def _log_normalization(self):
 217     return (math_ops.lgamma(self.concentration)
 218             - self.concentration * math_ops.log(self.rate))
 219 
 220   def _entropy(self):
 221     return (self.concentration
 222             + math_ops.log(self.rate)
 223             + math_ops.lgamma(self.concentration)
 224             - ((1. + self.concentration) *
 225                math_ops.digamma(self.concentration)))
 226 
 227   @distribution_util.AppendDocstring(
 228       """The mean of an inverse gamma distribution is
 229       `rate / (concentration - 1)`, when `concentration > 1`, and `NaN`
 230       otherwise. If `self.allow_nan_stats` is `False`, an exception will be
 231       raised rather than returning `NaN`""")
 232   def _mean(self):
 233     mean = self.rate / (self.concentration - 1.)
 234     if self.allow_nan_stats:
 235       nan = array_ops.fill(
 236           self.batch_shape_tensor(),
 237           np.array(np.nan, dtype=self.dtype.as_numpy_dtype()),
 238           name="nan")
 239       return array_ops.where(self.concentration > 1., mean, nan)
 240     else:
 241       return control_flow_ops.with_dependencies([
 242           check_ops.assert_less(
 243               array_ops.ones([], self.dtype), self.concentration,
 244               message="mean undefined when any concentration <= 1"),
 245       ], mean)
 246 
 247   @distribution_util.AppendDocstring(
 248       """Variance for inverse gamma is defined only for `concentration > 2`. If
 249       `self.allow_nan_stats` is `False`, an exception will be raised rather
 250       than returning `NaN`.""")
 251   def _variance(self):
 252     var = (math_ops.square(self.rate)
 253            / math_ops.square(self.concentration - 1.)
 254            / (self.concentration - 2.))
 255     if self.allow_nan_stats:
 256       nan = array_ops.fill(
 257           self.batch_shape_tensor(),
 258           np.array(np.nan, dtype=self.dtype.as_numpy_dtype()),
 259           name="nan")
 260       return array_ops.where(self.concentration > 2., var, nan)
 261     else:
 262       return control_flow_ops.with_dependencies([
 263           check_ops.assert_less(
 264               constant_op.constant(2., dtype=self.dtype),
 265               self.concentration,
 266               message="variance undefined when any concentration <= 2"),
 267       ], var)
 268 
 269   @distribution_util.AppendDocstring(
 270       """The mode of an inverse gamma distribution is `rate / (concentration +
 271       1)`.""")
 272   def _mode(self):
 273     return self.rate / (1. + self.concentration)
 274 
 275   def _maybe_assert_valid_sample(self, x):
 276     check_ops.assert_same_float_dtype(
 277         tensors=[x], dtype=self.dtype)
 278     if not self.validate_args:
 279       return x
 280     return control_flow_ops.with_dependencies([
 281         check_ops.assert_positive(x),
 282     ], x)

 285 class InverseGammaWithSoftplusConcentrationRate(InverseGamma):
 286   """`InverseGamma` with softplus of `concentration` and `rate`."""

 294       "instead of `tf.contrib.distributions`.",
 295       warn_once=True)

 307           rate=nn.softplus(rate, name="softplus_rate"),
 308           validate_args=validate_args,
 309           allow_nan_stats=allow_nan_stats,
 310           name=name)
 311     self._parameters = parameters
