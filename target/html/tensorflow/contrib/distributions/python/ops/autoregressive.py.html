  29 class Autoregressive(distribution_lib.Distribution):
  30   """Autoregressive distributions.
  31 
  32   The Autoregressive distribution enables learning (often) richer multivariate
  33   distributions by repeatedly applying a [diffeomorphic](
  34   https://en.wikipedia.org/wiki/Diffeomorphism) transformation (such as
  35   implemented by `Bijector`s). Regarding terminology,
  36 
  37     "Autoregressive models decompose the joint density as a product of
  38     conditionals, and model each conditional in turn. Normalizing flows
  39     transform a base density (e.g. a standard Gaussian) into the target density
  40     by an invertible transformation with tractable Jacobian." [(Papamakarios et
  41     al., 2016)][1]
  42 
  43   In other words, the "autoregressive property" is equivalent to the
  44   decomposition, `p(x) = prod{ p(x[i] | x[0:i]) : i=0, ..., d }`. The provided
  45   `shift_and_log_scale_fn`, `masked_autoregressive_default_template`, achieves
  46   this property by zeroing out weights in its `masked_dense` layers.
  47 
  48   Practically speaking the autoregressive property means that there exists a
  49   permutation of the event coordinates such that each coordinate is a
  50   diffeomorphic function of only preceding coordinates
  51   [(van den Oord et al., 2016)][2].
  52 
  53   #### Mathematical Details
  54 
  55   The probability function is
  56 
  57   ```none
  58   prob(x; fn, n) = fn(x).prob(x)
  59   ```
  60 
  61   And a sample is generated by
  62 
  63   ```none
  64   x = fn(...fn(fn(x0).sample()).sample()).sample()
  65   ```
  66 
  67   where the ellipses (`...`) represent `n-2` composed calls to `fn`, `fn`
  68   constructs a `tfp.distributions.Distribution`-like instance, and `x0` is a
  69   fixed initializing `Tensor`.
  70 
  71   #### Examples
  72 
  73   ```python
  74   import tensorflow_probability as tfp
  75   tfd = tfp.distributions
  76 
  77   def normal_fn(self, event_size):
  78     n = event_size * (event_size + 1) / 2
  79     p = tf.Variable(tfd.Normal(loc=0., scale=1.).sample(n))
  80     affine = tfd.bijectors.Affine(
  81         scale_tril=tfd.fill_triangular(0.25 * p))
  82     def _fn(samples):
  83       scale = math_ops.exp(affine.forward(samples)).eval()
  84       return independent_lib.Independent(
  85           normal_lib.Normal(loc=0., scale=scale, validate_args=True),
  86           reinterpreted_batch_ndims=1)
  87     return _fn
  88 
  89   batch_and_event_shape = [3, 2, 4]
  90   sample0 = array_ops.zeros(batch_and_event_shape)
  91   ar = autoregressive_lib.Autoregressive(
  92       self._normal_fn(batch_and_event_shape[-1]), sample0)
  93   x = ar.sample([6, 5])
  94   # ==> x.shape = [6, 5, 3, 2, 4]
  95   prob_x = ar.prob(x)
  96   # ==> x.shape = [6, 5, 3, 2]
  97 
  98   ```
  99 
 100   #### References
 101 
 102   [1]: George Papamakarios, Theo Pavlakou, and Iain Murray. Masked
 103        Autoregressive Flow for Density Estimation. In _Neural Information
 104        Processing Systems_, 2017. https://arxiv.org/abs/1705.07057
 105 
 106   [2]: Aaron van den Oord, Nal Kalchbrenner, Oriol Vinyals, Lasse Espeholt,
 107        Alex Graves, and Koray Kavukcuoglu. Conditional Image Generation with
 108        PixelCNN Decoders. In _Neural Information Processing Systems_, 2016.
 109        https://arxiv.org/abs/1606.05328
 110   """
 111 
 112   @deprecation.deprecated(
 113       "2018-10-01",
 114       "The TensorFlow Distributions library has moved to "
 115       "TensorFlow Probability "
 116       "(https://github.com/tensorflow/probability). You "
 117       "should update all references to use `tfp.distributions` "
 118       "instead of `tf.contrib.distributions`.",
 119       warn_once=True)
 120   def __init__(self,
 121                distribution_fn,
 122                sample0=None,
 123                num_steps=None,
 124                validate_args=False,
 125                allow_nan_stats=True,
 126                name="Autoregressive"):
 127     """Construct an `Autoregressive` distribution.
 128 
 129     Args:
 130       distribution_fn: Python `callable` which constructs a
 131         `tfp.distributions.Distribution`-like instance from a `Tensor` (e.g.,
 132         `sample0`). The function must respect the "autoregressive property",
 133         i.e., there exists a permutation of event such that each coordinate is a
 134         diffeomorphic function of on preceding coordinates.
 135       sample0: Initial input to `distribution_fn`; used to
 136         build the distribution in `__init__` which in turn specifies this
 137         distribution's properties, e.g., `event_shape`, `batch_shape`, `dtype`.
 138         If unspecified, then `distribution_fn` should be default constructable.
 139       num_steps: Number of times `distribution_fn` is composed from samples,
 140         e.g., `num_steps=2` implies
 141         `distribution_fn(distribution_fn(sample0).sample(n)).sample()`.
 142       validate_args: Python `bool`.  Whether to validate input with asserts.
 143         If `validate_args` is `False`, and the inputs are invalid,
 144         correct behavior is not guaranteed.
 145       allow_nan_stats: Python `bool`, default `True`. When `True`, statistics
 146         (e.g., mean, mode, variance) use the value "`NaN`" to indicate the
 147         result is undefined. When `False`, an exception is raised if one or
 148         more of the statistic's batch members are undefined.
 149       name: Python `str` name prefixed to Ops created by this class.
 150         Default value: "Autoregressive".
 151 
 152     Raises:
 153       ValueError: if `num_steps` and
 154         `distribution_fn(sample0).event_shape.num_elements()` are both `None`.
 155       ValueError: if `num_steps < 1`.
 156     """
 157     parameters = dict(locals())
 158     with ops.name_scope(name) as name:
 159       self._distribution_fn = distribution_fn
 160       self._sample0 = sample0
 161       self._distribution0 = (distribution_fn() if sample0 is None
 162                              else distribution_fn(sample0))
 163       if num_steps is None:
 164         num_steps = self._distribution0.event_shape.num_elements()
 165         if num_steps is None:
 166           raise ValueError("distribution_fn must generate a distribution "
 167                            "with fully known `event_shape`.")
 168       if num_steps < 1:
 169         raise ValueError("num_steps ({}) must be at least 1.".format(num_steps))
 170       self._num_steps = num_steps
 171     super(Autoregressive, self).__init__(
 172         dtype=self._distribution0.dtype,
 173         reparameterization_type=self._distribution0.reparameterization_type,
 174         validate_args=validate_args,
 175         allow_nan_stats=allow_nan_stats,
 176         parameters=parameters,
 177         graph_parents=self._distribution0._graph_parents,  # pylint: disable=protected-access
 178         name=name)
 179 
 180   @property
 181   def distribution_fn(self):
 182     return self._distribution_fn
 183 
 184   @property
 185   def sample0(self):
 186     return self._sample0
 187 
 188   @property
 189   def num_steps(self):
 190     return self._num_steps
 191 
 192   @property
 193   def distribution0(self):
 194     return self._distribution0
 195 
 196   def _batch_shape(self):
 197     return self.distribution0.batch_shape
 198 
 199   def _batch_shape_tensor(self):
 200     return self.distribution0.batch_shape_tensor()
 201 
 202   def _event_shape(self):
 203     return self.distribution0.event_shape
 204 
 205   def _event_shape_tensor(self):
 206     return self.distribution0.event_shape_tensor()
 207 
 208   def _sample_n(self, n, seed=None):
 209     if seed is None:
 210       seed = distribution_util.gen_new_seed(
 211           seed=np.random.randint(2**32 - 1),
 212           salt="autoregressive")
 213     samples = self.distribution0.sample(n, seed=seed)
 214     for _ in range(self._num_steps):
 215       samples = self.distribution_fn(samples).sample(seed=seed)
 216     return samples
 217 
 218   def _log_prob(self, value):
 219     return self.distribution_fn(value).log_prob(value)
 220 
 221   def _prob(self, value):
 222     return self.distribution_fn(value).prob(value)
