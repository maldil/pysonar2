  35 class _Gumbel(distribution.Distribution):
  36   """The scalar Gumbel distribution with location `loc` and `scale` parameters.
  37 
  38   #### Mathematical details
  39 
  40   The probability density function (pdf) of this distribution is,
  41 
  42   ```none
  43   pdf(x; mu, sigma) = exp(-(x - mu) / sigma - exp(-(x - mu) / sigma))
  44   ```
  45 
  46   where `loc = mu` and `scale = sigma`.
  47 
  48   The cumulative density function of this distribution is,
  49 
  50   ```cdf(x; mu, sigma) = exp(-exp(-(x - mu) / sigma))```
  51 
  52   The Gumbel distribution is a member of the [location-scale family](
  53   https://en.wikipedia.org/wiki/Location-scale_family), i.e., it can be
  54   constructed as,
  55 
  56   ```none
  57   X ~ Gumbel(loc=0, scale=1)
  58   Y = loc + scale * X
  59   ```
  60 
  61   #### Examples
  62 
  63   Examples of initialization of one or a batch of distributions.
  64 
  65   ```python
  66   import tensorflow_probability as tfp
  67   tfd = tfp.distributions
  68 
  69   # Define a single scalar Gumbel distribution.
  70   dist = tfd.Gumbel(loc=0., scale=3.)
  71 
  72   # Evaluate the cdf at 1, returning a scalar.
  73   dist.cdf(1.)
  74 
  75   # Define a batch of two scalar valued Gumbels.
  76   # The first has mean 1 and scale 11, the second 2 and 22.
  77   dist = tfd.Gumbel(loc=[1, 2.], scale=[11, 22.])
  78 
  79   # Evaluate the pdf of the first distribution on 0, and the second on 1.5,
  80   # returning a length two tensor.
  81   dist.prob([0, 1.5])
  82 
  83   # Get 3 samples, returning a 3 x 2 tensor.
  84   dist.sample([3])
  85   ```
  86 
  87   Arguments are broadcast when possible.
  88 
  89   ```python
  90   # Define a batch of two scalar valued Logistics.
  91   # Both have mean 1, but different scales.
  92   dist = tfd.Gumbel(loc=1., scale=[11, 22.])
  93 
  94   # Evaluate the pdf of both distributions on the same point, 3.0,
  95   # returning a length 2 tensor.
  96   dist.prob(3.0)
  97   ```
  98 
  99   """
 100 
 101   @deprecation.deprecated(
 102       "2018-10-01",
 103       "The TensorFlow Distributions library has moved to "
 104       "TensorFlow Probability "
 105       "(https://github.com/tensorflow/probability). You "
 106       "should update all references to use `tfp.distributions` "
 107       "instead of `tf.contrib.distributions`.",
 108       warn_once=True)
 109   def __init__(self,
 110                loc,
 111                scale,
 112                validate_args=False,
 113                allow_nan_stats=True,
 114                name="Gumbel"):
 115     """Construct Gumbel distributions with location and scale `loc` and `scale`.
 116 
 117     The parameters `loc` and `scale` must be shaped in a way that supports
 118     broadcasting (e.g. `loc + scale` is a valid operation).
 119 
 120     Args:
 121       loc: Floating point tensor, the means of the distribution(s).
 122       scale: Floating point tensor, the scales of the distribution(s).
 123         scale must contain only positive values.
 124       validate_args: Python `bool`, default `False`. When `True` distribution
 125         parameters are checked for validity despite possibly degrading runtime
 126         performance. When `False` invalid inputs may silently render incorrect
 127         outputs.
 128       allow_nan_stats: Python `bool`, default `True`. When `True`,
 129         statistics (e.g., mean, mode, variance) use the value "`NaN`" to
 130         indicate the result is undefined. When `False`, an exception is raised
 131         if one or more of the statistic's batch members are undefined.
 132       name: Python `str` name prefixed to Ops created by this class.
 133 
 134     Raises:
 135       TypeError: if loc and scale are different dtypes.
 136     """
 137     parameters = dict(locals())
 138     with ops.name_scope(name, values=[loc, scale]) as name:
 139       with ops.control_dependencies([check_ops.assert_positive(scale)] if
 140                                     validate_args else []):
 141         self._loc = array_ops.identity(loc, name="loc")
 142         self._scale = array_ops.identity(scale, name="scale")
 143         check_ops.assert_same_float_dtype([self._loc, self._scale])
 144     super(_Gumbel, self).__init__(
 145         dtype=self._scale.dtype,
 146         reparameterization_type=distribution.FULLY_REPARAMETERIZED,
 147         validate_args=validate_args,
 148         allow_nan_stats=allow_nan_stats,
 149         parameters=parameters,
 150         graph_parents=[self._loc, self._scale],
 151         name=name)
 152 
 153   @staticmethod
 154   def _param_shapes(sample_shape):
 155     return dict(
 156         zip(("loc", "scale"), ([ops.convert_to_tensor(
 157             sample_shape, dtype=dtypes.int32)] * 2)))
 158 
 159   @property
 160   def loc(self):
 161     """Distribution parameter for the location."""
 162     return self._loc
 163 
 164   @property
 165   def scale(self):
 166     """Distribution parameter for scale."""
 167     return self._scale
 168 
 169   def _batch_shape_tensor(self):
 170     return array_ops.broadcast_dynamic_shape(
 171         array_ops.shape(self.loc), array_ops.shape(self.scale))
 172 
 173   def _batch_shape(self):
 174     return array_ops.broadcast_static_shape(
 175         self.loc.get_shape(), self.scale.get_shape())
 176 
 177   def _event_shape_tensor(self):
 178     return constant_op.constant([], dtype=dtypes.int32)
 179 
 180   def _event_shape(self):
 181     return tensor_shape.scalar()
 182 
 183   def _sample_n(self, n, seed=None):
 184     # Uniform variates must be sampled from the open-interval `(0, 1)` rather
 185     # than `[0, 1)`. To do so, we use `np.finfo(self.dtype.as_numpy_dtype).tiny`
 186     # because it is the smallest, positive, "normal" number. A "normal" number
 187     # is such that the mantissa has an implicit leading 1. Normal, positive
 188     # numbers x, y have the reasonable property that, `x + y >= max(x, y)`. In
 189     # this case, a subnormal number (i.e., np.nextafter) can cause us to sample
 190     # 0.
 191     uniform = random_ops.random_uniform(
 192         shape=array_ops.concat([[n], self.batch_shape_tensor()], 0),
 193         minval=np.finfo(self.dtype.as_numpy_dtype).tiny,
 194         maxval=1.,
 195         dtype=self.dtype,
 196         seed=seed)
 197     sampled = -math_ops.log(-math_ops.log(uniform))
 198     return sampled * self.scale + self.loc
 199 
 200   def _log_prob(self, x):
 201     return self._log_unnormalized_prob(x) - self._log_normalization()
 202 
 203   def _log_cdf(self, x):
 204     return -math_ops.exp(-self._z(x))
 205 
 206   def _cdf(self, x):
 207     return math_ops.exp(-math_ops.exp(-self._z(x)))
 208 
 209   def _log_unnormalized_prob(self, x):
 210     z = self._z(x)
 211     return - z - math_ops.exp(-z)
 212 
 213   def _log_normalization(self):
 214     return math_ops.log(self.scale)
 215 
 216   def _entropy(self):
 217     # Use broadcasting rules to calculate the full broadcast sigma.
 218     scale = self.scale * array_ops.ones_like(self.loc)
 219     return 1 + math_ops.log(scale) + np.euler_gamma
 220 
 221   def _mean(self):
 222     return self.loc + self.scale * np.euler_gamma
 223 
 224   def _stddev(self):
 225     return self.scale * array_ops.ones_like(self.loc) * math.pi / math.sqrt(6)
 226 
 227   def _mode(self):
 228     return self.loc * array_ops.ones_like(self.scale)
 229 
 230   def _z(self, x):
 231     """Standardize input `x` to a unit logistic."""
 232     with ops.name_scope("standardize", values=[x]):
 233       return (x - self.loc) / self.scale
