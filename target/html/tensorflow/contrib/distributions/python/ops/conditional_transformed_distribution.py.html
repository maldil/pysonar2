  52   @distribution_util.AppendDocstring(kwargs_dict=_condition_kwargs_dict)
  53   def _sample_n(self, n, seed=None,
  54                 bijector_kwargs=None,
  55                 distribution_kwargs=None):
  56     sample_shape = _concat_vectors(
  57         distribution_util.pick_vector(self._needs_rotation, self._empty, [n]),
  58         self._override_batch_shape,
  59         self._override_event_shape,
  60         distribution_util.pick_vector(self._needs_rotation, [n], self._empty))
  61     distribution_kwargs = distribution_kwargs or {}
  62     x = self.distribution.sample(sample_shape=sample_shape,
  63                                  seed=seed,
  64                                  **distribution_kwargs)
  65     x = self._maybe_rotate_dims(x)
  66     # We'll apply the bijector in the `_call_sample_n` function.
  67     return x
  68 
  69   def _call_sample_n(self, sample_shape, seed, name,
  70                      bijector_kwargs=None,
  71                      distribution_kwargs=None):
  72     # We override `_call_sample_n` rather than `_sample_n` so we can ensure that
  73     # the result of `self.bijector.forward` is not modified (and thus caching
  74     # works).
  75     with self._name_scope(name, values=[sample_shape]):
  76       sample_shape = ops.convert_to_tensor(
  77           sample_shape, dtype=dtypes.int32, name="sample_shape")
  78       sample_shape, n = self._expand_sample_shape_to_vector(
  79           sample_shape, "sample_shape")
  80 
  81       # First, generate samples. We will possibly generate extra samples in the
  82       # event that we need to reinterpret the samples as part of the
  83       # event_shape.
  84       x = self._sample_n(n, seed, bijector_kwargs, distribution_kwargs)
  85 
  86       # Next, we reshape `x` into its final form. We do this prior to the call
  87       # to the bijector to ensure that the bijector caching works.
  88       batch_event_shape = array_ops.shape(x)[1:]
  89       final_shape = array_ops.concat([sample_shape, batch_event_shape], 0)
  90       x = array_ops.reshape(x, final_shape)
  91 
  92       # Finally, we apply the bijector's forward transformation. For caching to
  93       # work, it is imperative that this is the last modification to the
  94       # returned result.
  95       bijector_kwargs = bijector_kwargs or {}
  96       y = self.bijector.forward(x, **bijector_kwargs)
  97       y = self._set_sample_static_shape(y, sample_shape)
  98 
  99       return y

 101   @distribution_util.AppendDocstring(kwargs_dict=_condition_kwargs_dict)
 102   def _log_prob(self, y, bijector_kwargs=None, distribution_kwargs=None):
 103     # For caching to work, it is imperative that the bijector is the first to
 104     # modify the input.
 105     bijector_kwargs = bijector_kwargs or {}
 106     distribution_kwargs = distribution_kwargs or {}
 107     x = self.bijector.inverse(y, **bijector_kwargs)
 108     event_ndims = self._maybe_get_static_event_ndims()
 109     ildj = self.bijector.inverse_log_det_jacobian(
 110         y, event_ndims=event_ndims, **bijector_kwargs)
 111     if self.bijector._is_injective:  # pylint: disable=protected-access
 112       return self._finish_log_prob_for_one_fiber(y, x, ildj,
 113                                                  distribution_kwargs)
 114 
 115     lp_on_fibers = [
 116         self._finish_log_prob_for_one_fiber(y, x_i, ildj_i, distribution_kwargs)
 117         for x_i, ildj_i in zip(x, ildj)]
 118     return math_ops.reduce_logsumexp(array_ops.stack(lp_on_fibers), axis=0)
 119 
 120   def _finish_log_prob_for_one_fiber(self, y, x, ildj, distribution_kwargs):
 121     """Finish computation of log_prob on one element of the inverse image."""
 122     x = self._maybe_rotate_dims(x, rotate_right=True)
 123     log_prob = self.distribution.log_prob(x, **distribution_kwargs)
 124     if self._is_maybe_event_override:
 125       log_prob = math_ops.reduce_sum(log_prob, self._reduce_event_indices)
 126     return math_ops.cast(ildj, log_prob.dtype) + log_prob

 128   @distribution_util.AppendDocstring(kwargs_dict=_condition_kwargs_dict)
 129   def _prob(self, y, bijector_kwargs=None, distribution_kwargs=None):
 130     bijector_kwargs = bijector_kwargs or {}
 131     distribution_kwargs = distribution_kwargs or {}
 132     x = self.bijector.inverse(y, **bijector_kwargs)
 133     event_ndims = self._maybe_get_static_event_ndims()
 134     ildj = self.bijector.inverse_log_det_jacobian(
 135         y, event_ndims=event_ndims, **bijector_kwargs)
 136     if self.bijector._is_injective:  # pylint: disable=protected-access
 137       return self._finish_prob_for_one_fiber(y, x, ildj, distribution_kwargs)
 138 
 139     prob_on_fibers = [
 140         self._finish_prob_for_one_fiber(y, x_i, ildj_i, distribution_kwargs)
 141         for x_i, ildj_i in zip(x, ildj)]
 142     return sum(prob_on_fibers)
 143 
 144   def _finish_prob_for_one_fiber(self, y, x, ildj, distribution_kwargs):
 145     """Finish computation of prob on one element of the inverse image."""
 146     x = self._maybe_rotate_dims(x, rotate_right=True)
 147     prob = self.distribution.prob(x, **distribution_kwargs)
 148     if self._is_maybe_event_override:
 149       prob = math_ops.reduce_prod(prob, self._reduce_event_indices)
 150     return math_ops.exp(math_ops.cast(ildj, prob.dtype)) * prob

 152   @distribution_util.AppendDocstring(kwargs_dict=_condition_kwargs_dict)
 153   def _log_cdf(self, y, bijector_kwargs=None, distribution_kwargs=None):
 154     if self._is_maybe_event_override:
 155       raise NotImplementedError("log_cdf is not implemented when overriding "
 156                                 "event_shape")
 157     if not self.bijector._is_injective:  # pylint: disable=protected-access
 158       raise NotImplementedError("log_cdf is not implemented when "
 159                                 "bijector is not injective.")
 160     bijector_kwargs = bijector_kwargs or {}
 161     distribution_kwargs = distribution_kwargs or {}
 162     x = self.bijector.inverse(y, **bijector_kwargs)
 163     return self.distribution.log_cdf(x, **distribution_kwargs)

 206   @distribution_util.AppendDocstring(kwargs_dict=_condition_kwargs_dict)
 207   def _quantile(self, value, bijector_kwargs=None, distribution_kwargs=None):
 208     if self._is_maybe_event_override:
 209       raise NotImplementedError("quantile is not implemented when overriding "
 210                                 "event_shape")
 211     if not self.bijector._is_injective:  # pylint: disable=protected-access
 212       raise NotImplementedError("quantile is not implemented when "
 213                                 "bijector is not injective.")
 214     bijector_kwargs = bijector_kwargs or {}
 215     distribution_kwargs = distribution_kwargs or {}
 216     # x_q is the "qth quantile" of X iff q = P[X <= x_q].  Now, since X =
 217     # g^{-1}(Y), q = P[X <= x_q] = P[g^{-1}(Y) <= x_q] = P[Y <= g(x_q)],
 218     # implies the qth quantile of Y is g(x_q).
 219     inv_cdf = self.distribution.quantile(value, **distribution_kwargs)
 220     return self.bijector.forward(inv_cdf, **bijector_kwargs)

 222   def _maybe_get_static_event_ndims(self):
 223     if self.event_shape.ndims is not None:
 224       return self.event_shape.ndims
 225 
 226     event_ndims = array_ops.size(self.event_shape_tensor())
 227     event_ndims_ = distribution_util.maybe_get_static_value(event_ndims)
 228 
 229     if event_ndims_ is not None:
 230       return event_ndims_
 231 
 232     return event_ndims
