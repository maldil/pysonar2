  35 class OneHotCategorical(distribution.Distribution):
  36   """OneHotCategorical distribution.
  37 
  38   The categorical distribution is parameterized by the log-probabilities
  39   of a set of classes. The difference between OneHotCategorical and Categorical
  40   distributions is that OneHotCategorical is a discrete distribution over
  41   one-hot bit vectors whereas Categorical is a discrete distribution over
  42   positive integers. OneHotCategorical is equivalent to Categorical except
  43   Categorical has event_dim=() while OneHotCategorical has event_dim=K, where
  44   K is the number of classes.
  45 
  46   This class provides methods to create indexed batches of OneHotCategorical
  47   distributions. If the provided `logits` or `probs` is rank 2 or higher, for
  48   every fixed set of leading dimensions, the last dimension represents one
  49   single OneHotCategorical distribution. When calling distribution
  50   functions (e.g. `dist.prob(x)`), `logits` and `x` are broadcast to the
  51   same shape (if possible). In all cases, the last dimension of `logits,x`
  52   represents single OneHotCategorical distributions.
  53 
  54   #### Examples
  55 
  56   Creates a 3-class distribution, with the 2nd class, the most likely to be
  57   drawn from.
  58 
  59   ```python
  60   p = [0.1, 0.5, 0.4]
  61   dist = OneHotCategorical(probs=p)
  62   ```
  63 
  64   Creates a 3-class distribution, with the 2nd class the most likely to be
  65   drawn from, using logits.
  66 
  67   ```python
  68   logits = [-2, 2, 0]
  69   dist = OneHotCategorical(logits=logits)
  70   ```
  71 
  72   Creates a 3-class distribution, with the 3rd class is most likely to be drawn.
  73 
  74   ```python
  75   # counts is a scalar.
  76   p = [0.1, 0.4, 0.5]
  77   dist = OneHotCategorical(probs=p)
  78   dist.prob([0,1,0])  # Shape []
  79 
  80   # p will be broadcast to [[0.1, 0.4, 0.5], [0.1, 0.4, 0.5]] to match.
  81   samples = [[0,1,0], [1,0,0]]
  82   dist.prob(samples)  # Shape [2]
  83   ```
  84 
  85   """

  95   def __init__(
  96       self,
  97       logits=None,
  98       probs=None,
  99       dtype=dtypes.int32,
 100       validate_args=False,
 101       allow_nan_stats=True,
 102       name="OneHotCategorical"):
 103     """Initialize OneHotCategorical distributions using class log-probabilities.
 104 
 105     Args:
 106       logits: An N-D `Tensor`, `N >= 1`, representing the log probabilities of a
 107         set of Categorical distributions. The first `N - 1` dimensions index
 108         into a batch of independent distributions and the last dimension
 109         represents a vector of logits for each class. Only one of `logits` or
 110         `probs` should be passed in.
 111       probs: An N-D `Tensor`, `N >= 1`, representing the probabilities of a set
 112         of Categorical distributions. The first `N - 1` dimensions index into a
 113         batch of independent distributions and the last dimension represents a
 114         vector of probabilities for each class. Only one of `logits` or `probs`
 115         should be passed in.
 116       dtype: The type of the event samples (default: int32).
 117       validate_args: Python `bool`, default `False`. When `True` distribution
 118         parameters are checked for validity despite possibly degrading runtime
 119         performance. When `False` invalid inputs may silently render incorrect
 120         outputs.
 121       allow_nan_stats: Python `bool`, default `True`. When `True`, statistics
 122         (e.g., mean, mode, variance) use the value "`NaN`" to indicate the
 123         result is undefined. When `False`, an exception is raised if one or
 124         more of the statistic's batch members are undefined.
 125       name: Python `str` name prefixed to Ops created by this class.
 126     """
 127     parameters = dict(locals())
 128     with ops.name_scope(name, values=[logits, probs]) as name:
 129       self._logits, self._probs = distribution_util.get_logits_and_probs(
 130           name=name, logits=logits, probs=probs, validate_args=validate_args,
 131           multidimensional=True)
 132 
 133       logits_shape_static = self._logits.get_shape().with_rank_at_least(1)
 134       if logits_shape_static.ndims is not None:
 135         self._batch_rank = ops.convert_to_tensor(
 136             logits_shape_static.ndims - 1,
 137             dtype=dtypes.int32,
 138             name="batch_rank")
 139       else:
 140         with ops.name_scope(name="batch_rank"):
 141           self._batch_rank = array_ops.rank(self._logits) - 1
 142 
 143       with ops.name_scope(name="event_size"):
 144         self._event_size = array_ops.shape(self._logits)[-1]
 145 
 146     super(OneHotCategorical, self).__init__(
 147         dtype=dtype,
 148         reparameterization_type=distribution.NOT_REPARAMETERIZED,
 149         validate_args=validate_args,
 150         allow_nan_stats=allow_nan_stats,
 151         parameters=parameters,
 152         graph_parents=[self._logits,
 153                        self._probs],
 154         name=name)
 155 
 156   @property
 157   def event_size(self):
 158     """Scalar `int32` tensor: the number of classes."""
 159     return self._event_size
 160 
 161   @property
 162   def logits(self):
 163     """Vector of coordinatewise logits."""
 164     return self._logits
 165 
 166   @property
 167   def probs(self):
 168     """Vector of coordinatewise probabilities."""
 169     return self._probs
 170 
 171   def _batch_shape_tensor(self):
 172     return array_ops.shape(self.logits)[:-1]
 173 
 174   def _batch_shape(self):
 175     return self.logits.get_shape()[:-1]
 176 
 177   def _event_shape_tensor(self):
 178     return array_ops.shape(self.logits)[-1:]
 179 
 180   def _event_shape(self):
 181     return self.logits.get_shape().with_rank_at_least(1)[-1:]
 182 
 183   def _sample_n(self, n, seed=None):
 184     sample_shape = array_ops.concat([[n], array_ops.shape(self.logits)], 0)
 185     logits = self.logits
 186     if logits.get_shape().ndims == 2:
 187       logits_2d = logits
 188     else:
 189       logits_2d = array_ops.reshape(logits, [-1, self.event_size])
 190     samples = random_ops.multinomial(logits_2d, n, seed=seed)
 191     samples = array_ops.transpose(samples)
 192     samples = array_ops.one_hot(samples, self.event_size, dtype=self.dtype)
 193     ret = array_ops.reshape(samples, sample_shape)
 194     return ret
 195 
 196   def _log_prob(self, x):
 197     x = self._assert_valid_sample(x)
 198     # broadcast logits or x if need be.
 199     logits = self.logits
 200     if (not x.get_shape().is_fully_defined() or
 201         not logits.get_shape().is_fully_defined() or
 202         x.get_shape() != logits.get_shape()):
 203       logits = array_ops.ones_like(x, dtype=logits.dtype) * logits
 204       x = array_ops.ones_like(logits, dtype=x.dtype) * x
 205 
 206     logits_shape = array_ops.shape(math_ops.reduce_sum(logits, -1))
 207     logits_2d = array_ops.reshape(logits, [-1, self.event_size])
 208     x_2d = array_ops.reshape(x, [-1, self.event_size])
 209     ret = -nn_ops.softmax_cross_entropy_with_logits(labels=x_2d,
 210                                                     logits=logits_2d)
 211     # Reshape back to user-supplied batch and sample dims prior to 2D reshape.
 212     ret = array_ops.reshape(ret, logits_shape)
 213     return ret
 214 
 215   def _entropy(self):
 216     return -math_ops.reduce_sum(
 217         nn_ops.log_softmax(self.logits) * self.probs, axis=-1)
 218 
 219   def _mode(self):
 220     ret = math_ops.argmax(self.logits, axis=self._batch_rank)
 221     ret = array_ops.one_hot(ret, self.event_size, dtype=self.dtype)
 222     ret.set_shape(self.logits.get_shape())
 223     return ret
 224 
 225   def _covariance(self):
 226     p = self.probs
 227     ret = -math_ops.matmul(p[..., None], p[..., None, :])
 228     return array_ops.matrix_set_diag(ret, self._variance())
 229 
 230   def _variance(self):
 231     return self.probs * (1. - self.probs)
 232 
 233   def _assert_valid_sample(self, x):
 234     if not self.validate_args:
 235       return x
 236     return control_flow_ops.with_dependencies([
 237         check_ops.assert_non_positive(x),
 238         check_ops.assert_near(
 239             array_ops.zeros([], dtype=self.dtype),
 240             math_ops.reduce_logsumexp(x, axis=[-1])),
 241     ], x)

 253 def _kl_categorical_categorical(a, b, name=None):
 254   """Calculate the batched KL divergence KL(a || b) with a, b OneHotCategorical.
 255 
 256   Args:
 257     a: instance of a OneHotCategorical distribution object.
 258     b: instance of a OneHotCategorical distribution object.
 259     name: (optional) Name to use for created operations.
 260       default is "kl_categorical_categorical".
 261 
 262   Returns:
 263     Batchwise KL(a || b)
 264   """
 265   with ops.name_scope(name, "kl_categorical_categorical", values=[
 266       a.logits, b.logits]):
 267     # sum(p ln(p / q))
 268     return math_ops.reduce_sum(
 269         nn_ops.softmax(a.logits) * (nn_ops.log_softmax(a.logits)
 270                                     - nn_ops.log_softmax(b.logits)),
 271         axis=-1)
