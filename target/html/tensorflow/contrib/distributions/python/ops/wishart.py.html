  47 class _WishartLinearOperator(distribution.Distribution):
  48   """The matrix Wishart distribution on positive definite matrices.
  49 
  50   This distribution is defined by a scalar number of degrees of freedom `df` and
  51   an instance of `LinearOperator`, which provides matrix-free access to a
  52   symmetric positive definite operator, which defines the scale matrix.
  53 
  54   #### Mathematical Details
  55 
  56   The probability density function (pdf) is,
  57 
  58   ```none
  59   pdf(X; df, scale) = det(X)**(0.5 (df-k-1)) exp(-0.5 tr[inv(scale) X]) / Z
  60   Z = 2**(0.5 df k) |det(scale)|**(0.5 df) Gamma_k(0.5 df)
  61   ```
  62 
  63   where:
  64 
  65   * `df >= k` denotes the degrees of freedom,
  66   * `scale` is a symmetric, positive definite, `k x k` matrix,
  67   * `Z` is the normalizing constant, and,
  68   * `Gamma_k` is the [multivariate Gamma function](
  69     https://en.wikipedia.org/wiki/Multivariate_gamma_function).
  70 
  71   #### Examples
  72 
  73   See `WishartFull`, `WishartCholesky` for examples of initializing and using
  74   this class.
  75   """

  85   def __init__(self,
  86                df,
  87                scale_operator,
  88                cholesky_input_output_matrices=False,
  89                validate_args=False,
  90                allow_nan_stats=True,
  91                name=None):
  92     """Construct Wishart distributions.
  93 
  94     Args:
  95       df: `float` or `double` tensor, the degrees of freedom of the
  96         distribution(s). `df` must be greater than or equal to `k`.
  97       scale_operator: `float` or `double` instance of `LinearOperator`.
  98       cholesky_input_output_matrices: Python `bool`. Any function which whose
  99         input or output is a matrix assumes the input is Cholesky and returns a
 100         Cholesky factored matrix. Example `log_prob` input takes a Cholesky and
 101         `sample_n` returns a Cholesky when
 102         `cholesky_input_output_matrices=True`.
 103       validate_args: Python `bool`, default `False`. When `True` distribution
 104         parameters are checked for validity despite possibly degrading runtime
 105         performance. When `False` invalid inputs may silently render incorrect
 106         outputs.
 107       allow_nan_stats: Python `bool`, default `True`. When `True`, statistics
 108         (e.g., mean, mode, variance) use the value "`NaN`" to indicate the
 109         result is undefined. When `False`, an exception is raised if one or
 110         more of the statistic's batch members are undefined.
 111       name: Python `str` name prefixed to Ops created by this class.
 112 
 113     Raises:
 114       TypeError: if scale is not floating-type
 115       TypeError: if scale.dtype != df.dtype
 116       ValueError: if df < k, where scale operator event shape is
 117         `(k, k)`
 118     """
 119     parameters = dict(locals())
 120     self._cholesky_input_output_matrices = cholesky_input_output_matrices
 121     with ops.name_scope(name) as name:
 122       with ops.name_scope("init", values=[df, scale_operator]):
 123         if not scale_operator.dtype.is_floating:
 124           raise TypeError(
 125               "scale_operator.dtype=%s is not a floating-point type" %
 126               scale_operator.dtype)
 127         if not scale_operator.is_square:
 128           print(scale_operator.to_dense().eval())
 129           raise ValueError("scale_operator must be square.")
 130 
 131         self._scale_operator = scale_operator
 132         self._df = ops.convert_to_tensor(
 133             df,
 134             dtype=scale_operator.dtype,
 135             name="df")
 136         contrib_tensor_util.assert_same_float_dtype(
 137             (self._df, self._scale_operator))
 138         if (self._scale_operator.shape.ndims is None or
 139             self._scale_operator.shape[-1].value is None):
 140           self._dimension = math_ops.cast(
 141               self._scale_operator.domain_dimension_tensor(),
 142               dtype=self._scale_operator.dtype, name="dimension")
 143         else:
 144           self._dimension = ops.convert_to_tensor(
 145               self._scale_operator.shape[-1].value,
 146               dtype=self._scale_operator.dtype, name="dimension")
 147         df_val = tensor_util.constant_value(self._df)
 148         dim_val = tensor_util.constant_value(self._dimension)
 149         if df_val is not None and dim_val is not None:
 150           df_val = np.asarray(df_val)
 151           if not df_val.shape:
 152             df_val = [df_val]
 153           if any(df_val < dim_val):
 154             raise ValueError(
 155                 "Degrees of freedom (df = %s) cannot be less than "
 156                 "dimension of scale matrix (scale.dimension = %s)"
 157                 % (df_val, dim_val))
 158         elif validate_args:
 159           assertions = check_ops.assert_less_equal(
 160               self._dimension, self._df,
 161               message=("Degrees of freedom (df = %s) cannot be "
 162                        "less than dimension of scale matrix "
 163                        "(scale.dimension = %s)" %
 164                        (self._dimension, self._df)))
 165           self._df = control_flow_ops.with_dependencies(
 166               [assertions], self._df)
 167     super(_WishartLinearOperator, self).__init__(
 168         dtype=self._scale_operator.dtype,
 169         validate_args=validate_args,
 170         allow_nan_stats=allow_nan_stats,
 171         reparameterization_type=distribution.FULLY_REPARAMETERIZED,
 172         parameters=parameters,
 173         graph_parents=([self._df, self._dimension] +
 174                        self._scale_operator.graph_parents),
 175         name=name)

 182   def _square_scale_operator(self):
 183     return self.scale_operator.matmul(

 187     """Wishart distribution scale matrix."""
 188     if self._cholesky_input_output_matrices:
 189       return self.scale_operator.to_dense()
 190     else:
 191       return self._square_scale_operator()

 222   def _sample_n(self, n, seed):
 223     batch_shape = self.batch_shape_tensor()
 224     event_shape = self.event_shape_tensor()
 225     batch_ndims = array_ops.shape(batch_shape)[0]
 226 
 227     ndims = batch_ndims + 3  # sample_ndims=1, event_ndims=2
 228     shape = array_ops.concat([[n], batch_shape, event_shape], 0)
 229 
 230     # Complexity: O(nbk**2)
 231     x = random_ops.random_normal(shape=shape,
 232                                  mean=0.,
 233                                  stddev=1.,
 234                                  dtype=self.dtype,
 235                                  seed=seed)
 236 
 237     # Complexity: O(nbk)
 238     # This parametrization is equivalent to Chi2, i.e.,
 239     # ChiSquared(k) == Gamma(alpha=k/2, beta=1/2)
 240     expanded_df = self.df * array_ops.ones(
 241         self.scale_operator.batch_shape_tensor(),
 242         dtype=self.df.dtype.base_dtype)
 243     g = random_ops.random_gamma(shape=[n],
 244                                 alpha=self._multi_gamma_sequence(
 245                                     0.5 * expanded_df, self.dimension),
 246                                 beta=0.5,
 247                                 dtype=self.dtype,
 248                                 seed=distribution_util.gen_new_seed(
 249                                     seed, "wishart"))
 250 
 251     # Complexity: O(nbk**2)
 252     x = array_ops.matrix_band_part(x, -1, 0)  # Tri-lower.
 253 
 254     # Complexity: O(nbk)
 255     x = array_ops.matrix_set_diag(x, math_ops.sqrt(g))
 256 
 257     # Make batch-op ready.
 258     # Complexity: O(nbk**2)
 259     perm = array_ops.concat([math_ops.range(1, ndims), [0]], 0)
 260     x = array_ops.transpose(x, perm)
 261     shape = array_ops.concat([batch_shape, [event_shape[0]], [-1]], 0)
 262     x = array_ops.reshape(x, shape)
 263 
 264     # Complexity: O(nbM) where M is the complexity of the operator solving a
 265     # vector system. E.g., for LinearOperatorDiag, each matmul is O(k**2), so
 266     # this complexity is O(nbk**2). For LinearOperatorLowerTriangular,
 267     # each matmul is O(k^3) so this step has complexity O(nbk^3).
 268     x = self.scale_operator.matmul(x)
 269 
 270     # Undo make batch-op ready.
 271     # Complexity: O(nbk**2)
 272     shape = array_ops.concat([batch_shape, event_shape, [n]], 0)
 273     x = array_ops.reshape(x, shape)
 274     perm = array_ops.concat([[ndims - 1], math_ops.range(0, ndims - 1)], 0)
 275     x = array_ops.transpose(x, perm)
 276 
 277     if not self.cholesky_input_output_matrices:
 278       # Complexity: O(nbk^3)
 279       x = math_ops.matmul(x, x, adjoint_b=True)
 280 
 281     return x
 282 
 283   def _log_prob(self, x):
 284     if self.cholesky_input_output_matrices:
 285       x_sqrt = x
 286     else:
 287       # Complexity: O(nbk^3)
 288       x_sqrt = linalg_ops.cholesky(x)
 289 
 290     batch_shape = self.batch_shape_tensor()
 291     event_shape = self.event_shape_tensor()
 292     ndims = array_ops.rank(x_sqrt)
 293     # sample_ndims = ndims - batch_ndims - event_ndims
 294     sample_ndims = ndims - array_ops.shape(batch_shape)[0] - 2
 295     sample_shape = array_ops.strided_slice(
 296         array_ops.shape(x_sqrt), [0], [sample_ndims])
 297 
 298     # We need to be able to pre-multiply each matrix by its corresponding
 299     # batch scale matrix. Since a Distribution Tensor supports multiple
 300     # samples per batch, this means we need to reshape the input matrix `x`
 301     # so that the first b dimensions are batch dimensions and the last two
 302     # are of shape [dimension, dimensions*number_of_samples]. Doing these
 303     # gymnastics allows us to do a batch_solve.
 304     #
 305     # After we're done with sqrt_solve (the batch operation) we need to undo
 306     # this reshaping so what we're left with is a Tensor partitionable by
 307     # sample, batch, event dimensions.
 308 
 309     # Complexity: O(nbk**2) since transpose must access every element.
 310     scale_sqrt_inv_x_sqrt = x_sqrt
 311     perm = array_ops.concat([math_ops.range(sample_ndims, ndims),
 312                              math_ops.range(0, sample_ndims)], 0)
 313     scale_sqrt_inv_x_sqrt = array_ops.transpose(scale_sqrt_inv_x_sqrt, perm)
 314     shape = array_ops.concat(
 315         (batch_shape, (math_ops.cast(
 316             self.dimension, dtype=dtypes.int32), -1)),
 317         0)
 318     scale_sqrt_inv_x_sqrt = array_ops.reshape(scale_sqrt_inv_x_sqrt, shape)
 319 
 320     # Complexity: O(nbM*k) where M is the complexity of the operator solving
 321     # a vector system. E.g., for LinearOperatorDiag, each solve is O(k), so
 322     # this complexity is O(nbk**2). For LinearOperatorLowerTriangular,
 323     # each solve is O(k**2) so this step has complexity O(nbk^3).
 324     scale_sqrt_inv_x_sqrt = self.scale_operator.solve(
 325         scale_sqrt_inv_x_sqrt)
 326 
 327     # Undo make batch-op ready.
 328     # Complexity: O(nbk**2)
 329     shape = array_ops.concat([batch_shape, event_shape, sample_shape], 0)
 330     scale_sqrt_inv_x_sqrt = array_ops.reshape(scale_sqrt_inv_x_sqrt, shape)
 331     perm = array_ops.concat([math_ops.range(ndims - sample_ndims, ndims),
 332                              math_ops.range(0, ndims - sample_ndims)], 0)
 333     scale_sqrt_inv_x_sqrt = array_ops.transpose(scale_sqrt_inv_x_sqrt, perm)
 334 
 335     # Write V = SS', X = LL'. Then:
 336     # tr[inv(V) X] = tr[inv(S)' inv(S) L L']
 337     #              = tr[inv(S) L L' inv(S)']
 338     #              = tr[(inv(S) L) (inv(S) L)']
 339     #              = sum_{ik} (inv(S) L)_{ik}**2
 340     # The second equality follows from the cyclic permutation property.
 341     # Complexity: O(nbk**2)
 342     trace_scale_inv_x = math_ops.reduce_sum(
 343         math_ops.square(scale_sqrt_inv_x_sqrt),
 344         axis=[-2, -1])
 345 
 346     # Complexity: O(nbk)
 347     half_log_det_x = math_ops.reduce_sum(
 348         math_ops.log(array_ops.matrix_diag_part(x_sqrt)),
 349         axis=[-1])
 350 
 351     # Complexity: O(nbk**2)
 352     log_prob = ((self.df - self.dimension - 1.) * half_log_det_x -
 353                 0.5 * trace_scale_inv_x -
 354                 self.log_normalization())
 355 
 356     # Set shape hints.
 357     # Try to merge what we know from the input then what we know from the
 358     # parameters of this distribution.
 359     if x.get_shape().ndims is not None:
 360       log_prob.set_shape(x.get_shape()[:-2])
 361     if (log_prob.get_shape().ndims is not None and
 362         self.batch_shape.ndims is not None and
 363         self.batch_shape.ndims > 0):
 364       log_prob.get_shape()[-self.batch_shape.ndims:].merge_with(
 365           self.batch_shape)
 366 
 367     return log_prob
 368 
 369   def _prob(self, x):
 370     return math_ops.exp(self._log_prob(x))
 371 
 372   def _entropy(self):
 373     half_dp1 = 0.5 * self.dimension + 0.5
 374     half_df = 0.5 * self.df
 375     return (self.dimension * (half_df + half_dp1 * math.log(2.)) +
 376             2 * half_dp1 * self.scale_operator.log_abs_determinant() +
 377             self._multi_lgamma(half_df, self.dimension) +
 378             (half_dp1 - half_df) * self._multi_digamma(half_df, self.dimension))
 379 
 380   def _mean(self):
 381     if self.cholesky_input_output_matrices:
 382       return (math_ops.sqrt(self.df)
 383               * self.scale_operator.to_dense())
 384     return self.df * self._square_scale_operator()
 385 
 386   def _variance(self):
 387     x = math_ops.sqrt(self.df) * self._square_scale_operator()
 388     d = array_ops.expand_dims(array_ops.matrix_diag_part(x), -1)
 389     v = math_ops.square(x) + math_ops.matmul(d, d, adjoint_b=True)
 390     if self.cholesky_input_output_matrices:
 391       return linalg_ops.cholesky(v)
 392     return v
 393 
 394   def _stddev(self):
 395     if self.cholesky_input_output_matrices:
 396       raise ValueError(
 397           "Computing std. dev. when is cholesky_input_output_matrices=True "
 398           "does not make sense.")
 399     return linalg_ops.cholesky(self.variance())
 400 
 401   def _mode(self):
 402     s = self.df - self.dimension - 1.
 403     s = array_ops.where(
 404         math_ops.less(s, 0.),
 405         constant_op.constant(float("NaN"), dtype=self.dtype, name="nan"),
 406         s)
 407     if self.cholesky_input_output_matrices:
 408       return math_ops.sqrt(s) * self.scale_operator.to_dense()
 409     return s * self._square_scale_operator()
 410 
 411   def mean_log_det(self, name="mean_log_det"):
 412     """Computes E[log(det(X))] under this Wishart distribution."""
 413     with self._name_scope(name):
 414       return (self._multi_digamma(0.5 * self.df, self.dimension) +
 415               self.dimension * math.log(2.) +
 416               2 * self.scale_operator.log_abs_determinant())
 417 
 418   def log_normalization(self, name="log_normalization"):
 419     """Computes the log normalizing constant, log(Z)."""
 420     with self._name_scope(name):
 421       return (self.df * self.scale_operator.log_abs_determinant() +
 422               0.5 * self.df * self.dimension * math.log(2.) +
 423               self._multi_lgamma(0.5 * self.df, self.dimension))
 424 
 425   def _multi_gamma_sequence(self, a, p, name="multi_gamma_sequence"):
 426     """Creates sequence used in multivariate (di)gamma; shape = shape(a)+[p]."""
 427     with self._name_scope(name, values=[a, p]):
 428       # Linspace only takes scalars, so we'll add in the offset afterwards.
 429       seq = math_ops.linspace(
 430           constant_op.constant(0., dtype=self.dtype),
 431           0.5 - 0.5 * p,
 432           math_ops.cast(p, dtypes.int32))
 433       return seq + array_ops.expand_dims(a, [-1])
 434 
 435   def _multi_lgamma(self, a, p, name="multi_lgamma"):
 436     """Computes the log multivariate gamma function; log(Gamma_p(a))."""
 437     with self._name_scope(name, values=[a, p]):
 438       seq = self._multi_gamma_sequence(a, p)
 439       return (0.25 * p * (p - 1.) * math.log(math.pi) +
 440               math_ops.reduce_sum(math_ops.lgamma(seq),
 441                                   axis=[-1]))
 442 
 443   def _multi_digamma(self, a, p, name="multi_digamma"):
 444     """Computes the multivariate digamma function; Psi_p(a)."""
 445     with self._name_scope(name, values=[a, p]):
 446       seq = self._multi_gamma_sequence(a, p)
 447       return math_ops.reduce_sum(math_ops.digamma(seq),
 448                                  axis=[-1])

 451 class WishartCholesky(_WishartLinearOperator):
 452   """The matrix Wishart distribution on positive definite matrices.
 453 
 454   This distribution is defined by a scalar degrees of freedom `df` and a
 455   lower, triangular Cholesky factor which characterizes the scale matrix.
 456 
 457   Using WishartCholesky is a constant-time improvement over WishartFull. It
 458   saves an O(nbk^3) operation, i.e., a matrix-product operation for sampling
 459   and a Cholesky factorization in log_prob. For most use-cases it often saves
 460   another O(nbk^3) operation since most uses of Wishart will also use the
 461   Cholesky factorization.
 462 
 463   #### Mathematical Details
 464 
 465   The probability density function (pdf) is,
 466 
 467   ```none
 468   pdf(X; df, scale) = det(X)**(0.5 (df-k-1)) exp(-0.5 tr[inv(scale) X]) / Z
 469   Z = 2**(0.5 df k) |det(scale)|**(0.5 df) Gamma_k(0.5 df)
 470   ```
 471 
 472   where:
 473   * `df >= k` denotes the degrees of freedom,
 474   * `scale` is a symmetric, positive definite, `k x k` matrix,
 475   * `Z` is the normalizing constant, and,
 476   * `Gamma_k` is the [multivariate Gamma function](
 477     https://en.wikipedia.org/wiki/Multivariate_gamma_function).
 478 
 479 
 480   #### Examples
 481 
 482   ```python
 483   import tensorflow_probability as tfp
 484   tfd = tfp.distributions
 485 
 486   # Initialize a single 3x3 Wishart with Cholesky factored scale matrix and 5
 487   # degrees-of-freedom.(*)
 488   df = 5
 489   chol_scale = tf.cholesky(...)  # Shape is [3, 3].
 490   dist = tfd.WishartCholesky(df=df, scale=chol_scale)
 491 
 492   # Evaluate this on an observation in R^3, returning a scalar.
 493   x = ...  # A 3x3 positive definite matrix.
 494   dist.prob(x)  # Shape is [], a scalar.
 495 
 496   # Evaluate this on a two observations, each in R^{3x3}, returning a length two
 497   # Tensor.
 498   x = [x0, x1]  # Shape is [2, 3, 3].
 499   dist.prob(x)  # Shape is [2].
 500 
 501   # Initialize two 3x3 Wisharts with Cholesky factored scale matrices.
 502   df = [5, 4]
 503   chol_scale = tf.cholesky(...)  # Shape is [2, 3, 3].
 504   dist = tfd.WishartCholesky(df=df, scale=chol_scale)
 505 
 506   # Evaluate this on four observations.
 507   x = [[x0, x1], [x2, x3]]  # Shape is [2, 2, 3, 3].
 508   dist.prob(x)  # Shape is [2, 2].
 509 
 510   # (*) - To efficiently create a trainable covariance matrix, see the example
 511   #   in tfp.distributions.matrix_diag_transform.
 512   ```
 513 
 514   """

 524   def __init__(self,

 528                validate_args=False,
 529                allow_nan_stats=True,
 530                name="WishartCholesky"):

 532 
 533     Args:
 534       df: `float` or `double` `Tensor`. Degrees of freedom, must be greater than

 536       scale: `float` or `double` `Tensor`. The Cholesky factorization of

 538       cholesky_input_output_matrices: Python `bool`. Any function which whose
 539         input or output is a matrix assumes the input is Cholesky and returns a
 540         Cholesky factored matrix. Example `log_prob` input takes a Cholesky and
 541         `sample_n` returns a Cholesky when

 582 class WishartFull(_WishartLinearOperator):
 583   """The matrix Wishart distribution on positive definite matrices.
 584 
 585   This distribution is defined by a scalar degrees of freedom `df` and a
 586   symmetric, positive definite scale matrix.
 587 
 588   Evaluation of the pdf, determinant, and sampling are all `O(k^3)` operations
 589   where `(k, k)` is the event space shape.
 590 
 591   #### Mathematical Details
 592 
 593   The probability density function (pdf) is,
 594 
 595   ```none
 596   pdf(X; df, scale) = det(X)**(0.5 (df-k-1)) exp(-0.5 tr[inv(scale) X]) / Z
 597   Z = 2**(0.5 df k) |det(scale)|**(0.5 df) Gamma_k(0.5 df)
 598   ```
 599 
 600   where:
 601   * `df >= k` denotes the degrees of freedom,
 602   * `scale` is a symmetric, positive definite, `k x k` matrix,
 603   * `Z` is the normalizing constant, and,
 604   * `Gamma_k` is the [multivariate Gamma function](
 605     https://en.wikipedia.org/wiki/Multivariate_gamma_function).
 606 
 607   #### Examples
 608 
 609   ```python
 610   import tensorflow_probability as tfp
 611   tfd = tfp.distributions
 612 
 613   # Initialize a single 3x3 Wishart with Full factored scale matrix and 5
 614   # degrees-of-freedom.(*)
 615   df = 5
 616   scale = ...  # Shape is [3, 3]; positive definite.
 617   dist = tfd.WishartFull(df=df, scale=scale)
 618 
 619   # Evaluate this on an observation in R^3, returning a scalar.
 620   x = ...  # A 3x3 positive definite matrix.
 621   dist.prob(x)  # Shape is [], a scalar.
 622 
 623   # Evaluate this on a two observations, each in R^{3x3}, returning a length two
 624   # Tensor.
 625   x = [x0, x1]  # Shape is [2, 3, 3].
 626   dist.prob(x)  # Shape is [2].
 627 
 628   # Initialize two 3x3 Wisharts with Full factored scale matrices.
 629   df = [5, 4]
 630   scale = ...  # Shape is [2, 3, 3].
 631   dist = tfd.WishartFull(df=df, scale=scale)
 632 
 633   # Evaluate this on four observations.
 634   x = [[x0, x1], [x2, x3]]  # Shape is [2, 2, 3, 3]; xi is positive definite.
 635   dist.prob(x)  # Shape is [2, 2].
 636 
 637   # (*) - To efficiently create a trainable covariance matrix, see the example
 638   #   in tfd.matrix_diag_transform.
 639   ```
