  33 _binomial_sample_note = """
  34 For each batch member of counts `value`, `P[value]` is the probability that
  35 after sampling `self.total_count` draws from this Binomial distribution, the
  36 number of successes is `value`. Since different sequences of draws can result in
  37 the same counts, the probability includes a combinatorial coefficient.
  38 
  39 Note: `value` must be a non-negative tensor with dtype `dtype` and whose shape
  40 can be broadcast with `self.probs` and `self.total_count`. `value` is only legal
  41 if it is less than or equal to `self.total_count` and its components are equal
  42 to integer values.
  43 """

  54 def _bdtr(k, n, p):
  55   """The binomial cumulative distribution function.
  56 
  57   Args:
  58     k: floating point `Tensor`.
  59     n: floating point `Tensor`.
  60     p: floating point `Tensor`.
  61 
  62   Returns:
  63     `sum_{j=0}^k p^j (1 - p)^(n - j)`.
  64   """
  65   # Trick for getting safe backprop/gradients into n, k when
  66   #   betainc(a = 0, ..) = nan
  67   # Write:
  68   #   where(unsafe, safe_output, betainc(where(unsafe, safe_input, input)))
  69   ones = array_ops.ones_like(n - k)
  70   k_eq_n = math_ops.equal(k, n)
  71   safe_dn = array_ops.where(k_eq_n, ones, n - k)
  72   dk = math_ops.betainc(a=safe_dn, b=k + 1, x=1 - p)
  73   return array_ops.where(k_eq_n, ones, dk)

  76 class Binomial(distribution.Distribution):
  77   """Binomial distribution.
  78 
  79   This distribution is parameterized by `probs`, a (batch of) probabilities for
  80   drawing a `1` and `total_count`, the number of trials per draw from the
  81   Binomial.
  82 
  83   #### Mathematical Details
  84 
  85   The Binomial is a distribution over the number of `1`'s in `total_count`
  86   independent trials, with each trial having the same probability of `1`, i.e.,
  87   `probs`.
  88 
  89   The probability mass function (pmf) is,
  90 
  91   ```none
  92   pmf(k; n, p) = p**k (1 - p)**(n - k) / Z
  93   Z = k! (n - k)! / n!
  94   ```
  95 
  96   where:
  97   * `total_count = n`,
  98   * `probs = p`,
  99   * `Z` is the normalizing constant, and,
 100   * `n!` is the factorial of `n`.
 101 
 102   #### Examples
 103 
 104   Create a single distribution, corresponding to 5 coin flips.
 105 
 106   ```python
 107   dist = Binomial(total_count=5., probs=.5)
 108   ```
 109 
 110   Create a single distribution (using logits), corresponding to 5 coin flips.
 111 
 112   ```python
 113   dist = Binomial(total_count=5., logits=0.)
 114   ```
 115 
 116   Creates 3 distributions with the third distribution most likely to have
 117   successes.
 118 
 119   ```python
 120   p = [.2, .3, .8]
 121   # n will be broadcast to [4., 4., 4.], to match p.
 122   dist = Binomial(total_count=4., probs=p)
 123   ```
 124 
 125   The distribution functions can be evaluated on counts.
 126 
 127   ```python
 128   # counts same shape as p.
 129   counts = [1., 2, 3]
 130   dist.prob(counts)  # Shape [3]
 131 
 132   # p will be broadcast to [[.2, .3, .8], [.2, .3, .8]] to match counts.
 133   counts = [[1., 2, 1], [2, 2, 4]]
 134   dist.prob(counts)  # Shape [2, 3]
 135 
 136   # p will be broadcast to shape [5, 7, 3] to match counts.
 137   counts = [[...]]  # Shape [5, 7, 3]
 138   dist.prob(counts)  # Shape [5, 7, 3]
 139   ```
 140   """
 141 
 142   @deprecation.deprecated(
 143       "2018-10-01",
 144       "The TensorFlow Distributions library has moved to "
 145       "TensorFlow Probability "
 146       "(https://github.com/tensorflow/probability). You "
 147       "should update all references to use `tfp.distributions` "
 148       "instead of `tf.contrib.distributions`.",
 149       warn_once=True)
 150   def __init__(self,
 151                total_count,
 152                logits=None,
 153                probs=None,
 154                validate_args=False,
 155                allow_nan_stats=True,
 156                name="Binomial"):
 157     """Initialize a batch of Binomial distributions.
 158 
 159     Args:
 160       total_count: Non-negative floating point tensor with shape broadcastable
 161         to `[N1,..., Nm]` with `m >= 0` and the same dtype as `probs` or
 162         `logits`. Defines this as a batch of `N1 x ...  x Nm` different Binomial
 163         distributions. Its components should be equal to integer values.
 164       logits: Floating point tensor representing the log-odds of a
 165         positive event with shape broadcastable to `[N1,..., Nm]` `m >= 0`, and
 166         the same dtype as `total_count`. Each entry represents logits for the
 167         probability of success for independent Binomial distributions. Only one
 168         of `logits` or `probs` should be passed in.
 169       probs: Positive floating point tensor with shape broadcastable to
 170         `[N1,..., Nm]` `m >= 0`, `probs in [0, 1]`. Each entry represents the
 171         probability of success for independent Binomial distributions. Only one
 172         of `logits` or `probs` should be passed in.
 173       validate_args: Python `bool`, default `False`. When `True` distribution
 174         parameters are checked for validity despite possibly degrading runtime
 175         performance. When `False` invalid inputs may silently render incorrect
 176         outputs.
 177       allow_nan_stats: Python `bool`, default `True`. When `True`, statistics
 178         (e.g., mean, mode, variance) use the value "`NaN`" to indicate the
 179         result is undefined. When `False`, an exception is raised if one or
 180         more of the statistic's batch members are undefined.
 181       name: Python `str` name prefixed to Ops created by this class.
 182     """
 183     parameters = dict(locals())
 184     with ops.name_scope(name, values=[total_count, logits, probs]) as name:
 185       self._total_count = self._maybe_assert_valid_total_count(
 186           ops.convert_to_tensor(total_count, name="total_count"),
 187           validate_args)
 188       self._logits, self._probs = distribution_util.get_logits_and_probs(
 189           logits=logits,
 190           probs=probs,
 191           validate_args=validate_args,
 192           name=name)
 193     super(Binomial, self).__init__(
 194         dtype=self._probs.dtype,
 195         reparameterization_type=distribution.NOT_REPARAMETERIZED,
 196         validate_args=validate_args,
 197         allow_nan_stats=allow_nan_stats,
 198         parameters=parameters,
 199         graph_parents=[self._total_count,
 200                        self._logits,
 201                        self._probs],
 202         name=name)
 203 
 204   @property
 205   def total_count(self):
 206     """Number of trials."""
 207     return self._total_count
 208 
 209   @property
 210   def logits(self):
 211     """Log-odds of drawing a `1`."""
 212     return self._logits
 213 
 214   @property
 215   def probs(self):
 216     """Probability of drawing a `1`."""
 217     return self._probs
 218 
 219   def _batch_shape_tensor(self):
 220     return array_ops.broadcast_dynamic_shape(
 221         array_ops.shape(self.total_count),
 222         array_ops.shape(self.probs))
 223 
 224   def _batch_shape(self):
 225     return array_ops.broadcast_static_shape(
 226         self.total_count.get_shape(),
 227         self.probs.get_shape())
 228 
 229   def _event_shape_tensor(self):
 230     return constant_op.constant([], dtype=dtypes.int32)
 231 
 232   def _event_shape(self):
 233     return tensor_shape.scalar()
 234 
 235   @distribution_util.AppendDocstring(_binomial_sample_note)
 236   def _log_prob(self, counts):
 237     return self._log_unnormalized_prob(counts) - self._log_normalization(counts)
 238 
 239   @distribution_util.AppendDocstring(_binomial_sample_note)
 240   def _prob(self, counts):
 241     return math_ops.exp(self._log_prob(counts))
 242 
 243   def _cdf(self, counts):
 244     counts = self._maybe_assert_valid_sample(counts)
 245     probs = self.probs
 246     if not (counts.shape.is_fully_defined()
 247             and self.probs.shape.is_fully_defined()
 248             and counts.shape.is_compatible_with(self.probs.shape)):
 249       # If both shapes are well defined and equal, we skip broadcasting.
 250       probs += array_ops.zeros_like(counts)
 251       counts += array_ops.zeros_like(self.probs)
 252 
 253     return _bdtr(k=counts, n=self.total_count, p=probs)
 254 
 255   def _log_unnormalized_prob(self, counts):
 256     counts = self._maybe_assert_valid_sample(counts)
 257     return (counts * math_ops.log(self.probs) +
 258             (self.total_count - counts) * math_ops.log1p(-self.probs))
 259 
 260   def _log_normalization(self, counts):
 261     counts = self._maybe_assert_valid_sample(counts)
 262     return (math_ops.lgamma(1. + self.total_count - counts)
 263             + math_ops.lgamma(1. + counts)
 264             - math_ops.lgamma(1. + self.total_count))
 265 
 266   def _mean(self):
 267     return self.total_count * self.probs
 268 
 269   def _variance(self):
 270     return self._mean() * (1. - self.probs)
 271 
 272   @distribution_util.AppendDocstring(
 273       """Note that when `(1 + total_count) * probs` is an integer, there are
 274       actually two modes. Namely, `(1 + total_count) * probs` and
 275       `(1 + total_count) * probs - 1` are both modes. Here we return only the
 276       larger of the two modes.""")
 277   def _mode(self):
 278     return math_ops.floor((1. + self.total_count) * self.probs)
 279 
 280   def _maybe_assert_valid_total_count(self, total_count, validate_args):
 281     if not validate_args:
 282       return total_count
 283     return control_flow_ops.with_dependencies([
 284         check_ops.assert_non_negative(
 285             total_count,
 286             message="total_count must be non-negative."),
 287         distribution_util.assert_integer_form(
 288             total_count,
 289             message="total_count cannot contain fractional components."),
 290     ], total_count)
 291 
 292   def _maybe_assert_valid_sample(self, counts):
 293     """Check counts for proper shape, values, then return tensor version."""
 294     if not self.validate_args:
 295       return counts
 296     counts = distribution_util.embed_check_nonnegative_integer_form(counts)
 297     return control_flow_ops.with_dependencies([
 298         check_ops.assert_less_equal(
 299             counts, self.total_count,
 300             message="counts are not less than or equal to n."),
 301     ], counts)
