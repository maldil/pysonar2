  38 class Mixture(distribution.Distribution):
  39   """Mixture distribution.
  40 
  41   The `Mixture` object implements batched mixture distributions.
  42   The mixture model is defined by a `Categorical` distribution (the mixture)
  43   and a python list of `Distribution` objects.
  44 
  45   Methods supported include `log_prob`, `prob`, `mean`, `sample`, and
  46   `entropy_lower_bound`.
  47 
  48 
  49   #### Examples
  50 
  51   ```python
  52   # Create a mixture of two Gaussians:
  53   import tensorflow_probability as tfp
  54   tfd = tfp.distributions
  55 
  56   mix = 0.3
  57   bimix_gauss = tfd.Mixture(
  58     cat=tfd.Categorical(probs=[mix, 1.-mix]),
  59     components=[
  60       tfd.Normal(loc=-1., scale=0.1),
  61       tfd.Normal(loc=+1., scale=0.5),
  62   ])
  63 
  64   # Plot the PDF.
  65   import matplotlib.pyplot as plt
  66   x = tf.linspace(-2., 3., int(1e4)).eval()
  67   plt.plot(x, bimix_gauss.prob(x).eval());
  68   ```
  69 
  70   """
  71 
  72   @deprecation.deprecated(
  73       "2018-10-01",
  74       "The TensorFlow Distributions library has moved to "
  75       "TensorFlow Probability "
  76       "(https://github.com/tensorflow/probability). You "
  77       "should update all references to use `tfp.distributions` "
  78       "instead of `tf.contrib.distributions`.",
  79       warn_once=True)
  80   def __init__(self,
  81                cat,
  82                components,
  83                validate_args=False,
  84                allow_nan_stats=True,
  85                use_static_graph=False,
  86                name="Mixture"):
  87     """Initialize a Mixture distribution.
  88 
  89     A `Mixture` is defined by a `Categorical` (`cat`, representing the
  90     mixture probabilities) and a list of `Distribution` objects
  91     all having matching dtype, batch shape, event shape, and continuity
  92     properties (the components).
  93 
  94     The `num_classes` of `cat` must be possible to infer at graph construction
  95     time and match `len(components)`.
  96 
  97     Args:
  98       cat: A `Categorical` distribution instance, representing the probabilities
  99           of `distributions`.
 100       components: A list or tuple of `Distribution` instances.
 101         Each instance must have the same type, be defined on the same domain,
 102         and have matching `event_shape` and `batch_shape`.
 103       validate_args: Python `bool`, default `False`. If `True`, raise a runtime
 104         error if batch or event ranks are inconsistent between cat and any of
 105         the distributions. This is only checked if the ranks cannot be
 106         determined statically at graph construction time.
 107       allow_nan_stats: Boolean, default `True`. If `False`, raise an
 108        exception if a statistic (e.g. mean/mode/etc...) is undefined for any
 109         batch member. If `True`, batch members with valid parameters leading to
 110         undefined statistics will return NaN for this statistic.
 111       use_static_graph: Calls to `sample` will not rely on dynamic tensor
 112         indexing, allowing for some static graph compilation optimizations, but
 113         at the expense of sampling all underlying distributions in the mixture.
 114         (Possibly useful when running on TPUs).
 115         Default value: `False` (i.e., use dynamic indexing).
 116       name: A name for this distribution (optional).
 117 
 118     Raises:
 119       TypeError: If cat is not a `Categorical`, or `components` is not
 120         a list or tuple, or the elements of `components` are not
 121         instances of `Distribution`, or do not have matching `dtype`.
 122       ValueError: If `components` is an empty list or tuple, or its
 123         elements do not have a statically known event rank.
 124         If `cat.num_classes` cannot be inferred at graph creation time,
 125         or the constant value of `cat.num_classes` is not equal to
 126         `len(components)`, or all `components` and `cat` do not have
 127         matching static batch shapes, or all components do not
 128         have matching static event shapes.
 129     """
 130     parameters = dict(locals())
 131     if not isinstance(cat, categorical.Categorical):
 132       raise TypeError("cat must be a Categorical distribution, but saw: %s" %
 133                       cat)
 134     if not components:
 135       raise ValueError("components must be a non-empty list or tuple")
 136     if not isinstance(components, (list, tuple)):
 137       raise TypeError("components must be a list or tuple, but saw: %s" %
 138                       components)
 139     if not all(isinstance(c, distribution.Distribution) for c in components):
 140       raise TypeError(
 141           "all entries in components must be Distribution instances"
 142           " but saw: %s" % components)
 143 
 144     dtype = components[0].dtype
 145     if not all(d.dtype == dtype for d in components):
 146       raise TypeError("All components must have the same dtype, but saw "
 147                       "dtypes: %s" % [(d.name, d.dtype) for d in components])
 148     static_event_shape = components[0].event_shape
 149     static_batch_shape = cat.batch_shape
 150     for d in components:
 151       static_event_shape = static_event_shape.merge_with(d.event_shape)
 152       static_batch_shape = static_batch_shape.merge_with(d.batch_shape)
 153     if static_event_shape.ndims is None:
 154       raise ValueError(
 155           "Expected to know rank(event_shape) from components, but "
 156           "none of the components provide a static number of ndims")
 157 
 158     # Ensure that all batch and event ndims are consistent.
 159     with ops.name_scope(name, values=[cat.logits]) as name:
 160       num_components = cat.event_size
 161       static_num_components = tensor_util.constant_value(num_components)
 162       if static_num_components is None:
 163         raise ValueError(
 164             "Could not infer number of classes from cat and unable "
 165             "to compare this value to the number of components passed in.")
 166       # Possibly convert from numpy 0-D array.
 167       static_num_components = int(static_num_components)
 168       if static_num_components != len(components):
 169         raise ValueError("cat.num_classes != len(components): %d vs. %d" %
 170                          (static_num_components, len(components)))
 171 
 172       cat_batch_shape = cat.batch_shape_tensor()
 173       cat_batch_rank = array_ops.size(cat_batch_shape)
 174       if validate_args:
 175         batch_shapes = [d.batch_shape_tensor() for d in components]
 176         batch_ranks = [array_ops.size(bs) for bs in batch_shapes]
 177         check_message = ("components[%d] batch shape must match cat "
 178                          "batch shape")
 179         self._assertions = [
 180             check_ops.assert_equal(
 181                 cat_batch_rank, batch_ranks[di], message=check_message % di)
 182             for di in range(len(components))
 183         ]
 184         self._assertions += [
 185             check_ops.assert_equal(
 186                 cat_batch_shape, batch_shapes[di], message=check_message % di)
 187             for di in range(len(components))
 188         ]
 189       else:
 190         self._assertions = []
 191 
 192       self._cat = cat
 193       self._components = list(components)
 194       self._num_components = static_num_components
 195       self._static_event_shape = static_event_shape
 196       self._static_batch_shape = static_batch_shape
 197 
 198       self._use_static_graph = use_static_graph
 199       if use_static_graph and static_num_components is None:
 200         raise ValueError("Number of categories must be known statically when "
 201                          "`static_sample=True`.")
 202     # We let the Mixture distribution access _graph_parents since its arguably
 203     # more like a baseclass.
 204     graph_parents = self._cat._graph_parents  # pylint: disable=protected-access
 205     for c in self._components:
 206       graph_parents += c._graph_parents  # pylint: disable=protected-access
 207 
 208     super(Mixture, self).__init__(
 209         dtype=dtype,
 210         reparameterization_type=distribution.NOT_REPARAMETERIZED,
 211         validate_args=validate_args,
 212         allow_nan_stats=allow_nan_stats,
 213         parameters=parameters,
 214         graph_parents=graph_parents,
 215         name=name)

 241   def _expand_to_event_rank(self, x):
 242     """Expand the rank of x up to static_event_rank times for broadcasting.
 243 
 244     The static event rank was checked to not be None at construction time.
 245 
 246     Args:
 247       x: A tensor to expand.
 248     Returns:
 249       The expanded tensor.
 250     """
 251     expanded_x = x
 252     for _ in range(self.event_shape.ndims):
 253       expanded_x = array_ops.expand_dims(expanded_x, -1)
 254     return expanded_x

 256   def _mean(self):
 257     with ops.control_dependencies(self._assertions):
 258       distribution_means = [d.mean() for d in self.components]
 259       cat_probs = self._cat_probs(log_probs=False)
 260       cat_probs = [self._expand_to_event_rank(c_p) for c_p in cat_probs]
 261       partial_means = [
 262           c_p * m for (c_p, m) in zip(cat_probs, distribution_means)
 263       ]
 264       # These should all be the same shape by virtue of matching
 265       # batch_shape and event_shape.
 266       return math_ops.add_n(partial_means)
 267 
 268   def _stddev(self):
 269     with ops.control_dependencies(self._assertions):
 270       distribution_means = [d.mean() for d in self.components]
 271       distribution_devs = [d.stddev() for d in self.components]
 272       cat_probs = self._cat_probs(log_probs=False)
 273 
 274       stacked_means = array_ops.stack(distribution_means, axis=-1)
 275       stacked_devs = array_ops.stack(distribution_devs, axis=-1)
 276       cat_probs = [self._expand_to_event_rank(c_p) for c_p in cat_probs]
 277       broadcasted_cat_probs = (array_ops.stack(cat_probs, axis=-1) *
 278                                array_ops.ones_like(stacked_means))
 279 
 280       batched_dev = distribution_utils.mixture_stddev(
 281           array_ops.reshape(broadcasted_cat_probs, [-1, len(self.components)]),
 282           array_ops.reshape(stacked_means, [-1, len(self.components)]),
 283           array_ops.reshape(stacked_devs, [-1, len(self.components)]))
 284 
 285       # I.e. re-shape to list(batch_shape) + list(event_shape).
 286       return array_ops.reshape(batched_dev,
 287                                array_ops.shape(broadcasted_cat_probs)[:-1])
 288 
 289   def _log_prob(self, x):
 290     with ops.control_dependencies(self._assertions):
 291       x = ops.convert_to_tensor(x, name="x")
 292       distribution_log_probs = [d.log_prob(x) for d in self.components]
 293       cat_log_probs = self._cat_probs(log_probs=True)
 294       final_log_probs = [
 295           cat_lp + d_lp
 296           for (cat_lp, d_lp) in zip(cat_log_probs, distribution_log_probs)
 297       ]
 298       concat_log_probs = array_ops.stack(final_log_probs, 0)
 299       log_sum_exp = math_ops.reduce_logsumexp(concat_log_probs, [0])
 300       return log_sum_exp
 301 
 302   def _log_cdf(self, x):
 303     with ops.control_dependencies(self._assertions):
 304       x = ops.convert_to_tensor(x, name="x")
 305       distribution_log_cdfs = [d.log_cdf(x) for d in self.components]
 306       cat_log_probs = self._cat_probs(log_probs=True)
 307       final_log_cdfs = [
 308           cat_lp + d_lcdf
 309           for (cat_lp, d_lcdf) in zip(cat_log_probs, distribution_log_cdfs)
 310       ]
 311       concatted_log_cdfs = array_ops.stack(final_log_cdfs, axis=0)
 312       mixture_log_cdf = math_ops.reduce_logsumexp(concatted_log_cdfs, [0])
 313       return mixture_log_cdf

 315   def _sample_n(self, n, seed=None):
 316     if self._use_static_graph:
 317       # This sampling approach is almost the same as the approach used by
 318       # `MixtureSameFamily`. The differences are due to having a list of
 319       # `Distribution` objects rather than a single object, and maintaining
 320       # random seed management that is consistent with the non-static code path.
 321       samples = []
 322       cat_samples = self.cat.sample(n, seed=seed)
 323       for c in range(self.num_components):
 324         seed = distribution_util.gen_new_seed(seed, "mixture")
 325         samples.append(self.components[c].sample(n, seed=seed))
 326       x = array_ops.stack(
 327           samples, -self._static_event_shape.ndims - 1)     # [n, B, k, E]
 328       npdt = x.dtype.as_numpy_dtype
 329       mask = array_ops.one_hot(
 330           indices=cat_samples,                              # [n, B]
 331           depth=self._num_components,                       # == k
 332           on_value=np.ones([], dtype=npdt),
 333           off_value=np.zeros([], dtype=npdt))               # [n, B, k]
 334       mask = distribution_utils.pad_mixture_dimensions(
 335           mask, self, self._cat,
 336           self._static_event_shape.ndims)                   # [n, B, k, [1]*e]
 337       return math_ops.reduce_sum(
 338           x * mask,
 339           axis=-1 - self._static_event_shape.ndims)         # [n, B, E]
 340 
 341     with ops.control_dependencies(self._assertions):
 342       n = ops.convert_to_tensor(n, name="n")
 343       static_n = tensor_util.constant_value(n)
 344       n = int(static_n) if static_n is not None else n
 345       cat_samples = self.cat.sample(n, seed=seed)
 346 
 347       static_samples_shape = cat_samples.get_shape()
 348       if static_samples_shape.is_fully_defined():
 349         samples_shape = static_samples_shape.as_list()
 350         samples_size = static_samples_shape.num_elements()
 351       else:
 352         samples_shape = array_ops.shape(cat_samples)
 353         samples_size = array_ops.size(cat_samples)
 354       static_batch_shape = self.batch_shape
 355       if static_batch_shape.is_fully_defined():
 356         batch_shape = static_batch_shape.as_list()
 357         batch_size = static_batch_shape.num_elements()
 358       else:
 359         batch_shape = self.batch_shape_tensor()
 360         batch_size = math_ops.reduce_prod(batch_shape)
 361       static_event_shape = self.event_shape
 362       if static_event_shape.is_fully_defined():
 363         event_shape = np.array(static_event_shape.as_list(), dtype=np.int32)
 364       else:
 365         event_shape = self.event_shape_tensor()
 366 
 367       # Get indices into the raw cat sampling tensor. We will
 368       # need these to stitch sample values back out after sampling
 369       # within the component partitions.
 370       samples_raw_indices = array_ops.reshape(
 371           math_ops.range(0, samples_size), samples_shape)
 372 
 373       # Partition the raw indices so that we can use
 374       # dynamic_stitch later to reconstruct the samples from the
 375       # known partitions.
 376       partitioned_samples_indices = data_flow_ops.dynamic_partition(
 377           data=samples_raw_indices,
 378           partitions=cat_samples,
 379           num_partitions=self.num_components)
 380 
 381       # Copy the batch indices n times, as we will need to know
 382       # these to pull out the appropriate rows within the
 383       # component partitions.
 384       batch_raw_indices = array_ops.reshape(
 385           array_ops.tile(math_ops.range(0, batch_size), [n]), samples_shape)
 386 
 387       # Explanation of the dynamic partitioning below:
 388       #   batch indices are i.e., [0, 1, 0, 1, 0, 1]
 389       # Suppose partitions are:
 390       #     [1 1 0 0 1 1]
 391       # After partitioning, batch indices are cut as:
 392       #     [batch_indices[x] for x in 2, 3]
 393       #     [batch_indices[x] for x in 0, 1, 4, 5]
 394       # i.e.
 395       #     [1 1] and [0 0 0 0]
 396       # Now we sample n=2 from part 0 and n=4 from part 1.
 397       # For part 0 we want samples from batch entries 1, 1 (samples 0, 1),
 398       # and for part 1 we want samples from batch entries 0, 0, 0, 0
 399       #   (samples 0, 1, 2, 3).
 400       partitioned_batch_indices = data_flow_ops.dynamic_partition(
 401           data=batch_raw_indices,
 402           partitions=cat_samples,
 403           num_partitions=self.num_components)
 404       samples_class = [None for _ in range(self.num_components)]
 405 
 406       for c in range(self.num_components):
 407         n_class = array_ops.size(partitioned_samples_indices[c])
 408         seed = distribution_util.gen_new_seed(seed, "mixture")
 409         samples_class_c = self.components[c].sample(n_class, seed=seed)
 410 
 411         # Pull out the correct batch entries from each index.
 412         # To do this, we may have to flatten the batch shape.
 413 
 414         # For sample s, batch element b of component c, we get the
 415         # partitioned batch indices from
 416         # partitioned_batch_indices[c]; and shift each element by
 417         # the sample index. The final lookup can be thought of as
 418         # a matrix gather along locations (s, b) in
 419         # samples_class_c where the n_class rows correspond to
 420         # samples within this component and the batch_size columns
 421         # correspond to batch elements within the component.
 422         #
 423         # Thus the lookup index is
 424         #   lookup[c, i] = batch_size * s[i] + b[c, i]
 425         # for i = 0 ... n_class[c] - 1.
 426         lookup_partitioned_batch_indices = (
 427             batch_size * math_ops.range(n_class) +
 428             partitioned_batch_indices[c])
 429         samples_class_c = array_ops.reshape(
 430             samples_class_c,
 431             array_ops.concat([[n_class * batch_size], event_shape], 0))
 432         samples_class_c = array_ops.gather(
 433             samples_class_c, lookup_partitioned_batch_indices,
 434             name="samples_class_c_gather")
 435         samples_class[c] = samples_class_c
 436 
 437       # Stitch back together the samples across the components.
 438       lhs_flat_ret = data_flow_ops.dynamic_stitch(
 439           indices=partitioned_samples_indices, data=samples_class)
 440       # Reshape back to proper sample, batch, and event shape.
 441       ret = array_ops.reshape(lhs_flat_ret,
 442                               array_ops.concat([samples_shape,
 443                                                 self.event_shape_tensor()], 0))
 444       ret.set_shape(
 445           tensor_shape.TensorShape(static_samples_shape).concatenate(
 446               self.event_shape))
 447       return ret

 449   def entropy_lower_bound(self, name="entropy_lower_bound"):
 450     r"""A lower bound on the entropy of this mixture model.
 451 
 452     The bound below is not always very tight, and its usefulness depends
 453     on the mixture probabilities and the components in use.
 454 
 455     A lower bound is useful for ELBO when the `Mixture` is the variational
 456     distribution:
 457 
 458     \\(
 459     \log p(x) >= ELBO = \int q(z) \log p(x, z) dz + H[q]
 460     \\)
 461 
 462     where \\( p \\) is the prior distribution, \\( q \\) is the variational,
 463     and \\( H[q] \\) is the entropy of \\( q \\). If there is a lower bound
 464     \\( G[q] \\) such that \\( H[q] \geq G[q] \\) then it can be used in
 465     place of \\( H[q] \\).
 466 
 467     For a mixture of distributions \\( q(Z) = \sum_i c_i q_i(Z) \\) with
 468     \\( \sum_i c_i = 1 \\), by the concavity of \\( f(x) = -x \log x \\), a
 469     simple lower bound is:
 470 
 471     \\(
 472     \begin{align}
 473     H[q] & = - \int q(z) \log q(z) dz \\\
 474        & = - \int (\sum_i c_i q_i(z)) \log(\sum_i c_i q_i(z)) dz \\\
 475        & \geq - \sum_i c_i \int q_i(z) \log q_i(z) dz \\\
 476        & = \sum_i c_i H[q_i]
 477     \end{align}
 478     \\)
 479 
 480     This is the term we calculate below for \\( G[q] \\).
 481 
 482     Args:
 483       name: A name for this operation (optional).
 484 
 485     Returns:
 486       A lower bound on the Mixture's entropy.
 487     """
 488     with self._name_scope(name, values=[self.cat.logits]):
 489       with ops.control_dependencies(self._assertions):
 490         distribution_entropies = [d.entropy() for d in self.components]
 491         cat_probs = self._cat_probs(log_probs=False)
 492         partial_entropies = [
 493             c_p * m for (c_p, m) in zip(cat_probs, distribution_entropies)
 494         ]
 495         # These are all the same shape by virtue of matching batch_shape
 496         return math_ops.add_n(partial_entropies)
 497 
 498   def _cat_probs(self, log_probs):
 499     """Get a list of num_components batchwise probabilities."""
 500     which_softmax = nn_ops.log_softmax if log_probs else nn_ops.softmax
 501     cat_probs = which_softmax(self.cat.logits)
 502     cat_probs = array_ops.unstack(cat_probs, num=self.num_components, axis=-1)
 503     return cat_probs
