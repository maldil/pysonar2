  33 class MultivariateNormalTriL(
  34     mvn_linop.MultivariateNormalLinearOperator):
  35   """The multivariate normal distribution on `R^k`.
  36 
  37   The Multivariate Normal distribution is defined over `R^k` and parameterized
  38   by a (batch of) length-`k` `loc` vector (aka "mu") and a (batch of) `k x k`
  39   `scale` matrix; `covariance = scale @ scale.T` where `@` denotes
  40   matrix-multiplication.
  41 
  42   #### Mathematical Details
  43 
  44   The probability density function (pdf) is,
  45 
  46   ```none
  47   pdf(x; loc, scale) = exp(-0.5 ||y||**2) / Z,
  48   y = inv(scale) @ (x - loc),
  49   Z = (2 pi)**(0.5 k) |det(scale)|,
  50   ```
  51 
  52   where:
  53 
  54   * `loc` is a vector in `R^k`,
  55   * `scale` is a matrix in `R^{k x k}`, `covariance = scale @ scale.T`,
  56   * `Z` denotes the normalization constant, and,
  57   * `||y||**2` denotes the squared Euclidean norm of `y`.
  58 
  59   A (non-batch) `scale` matrix is:
  60 
  61   ```none
  62   scale = scale_tril
  63   ```
  64 
  65   where `scale_tril` is lower-triangular `k x k` matrix with non-zero diagonal,
  66   i.e., `tf.diag_part(scale_tril) != 0`.
  67 
  68   Additional leading dimensions (if any) will index batches.
  69 
  70   The MultivariateNormal distribution is a member of the [location-scale
  71   family](https://en.wikipedia.org/wiki/Location-scale_family), i.e., it can be
  72   constructed as,
  73 
  74   ```none
  75   X ~ MultivariateNormal(loc=0, scale=1)   # Identity scale, zero shift.
  76   Y = scale @ X + loc
  77   ```
  78 
  79   Trainable (batch) lower-triangular matrices can be created with
  80   `tfp.distributions.matrix_diag_transform()` and/or
  81   `tfp.distributions.fill_triangular()`
  82 
  83   #### Examples
  84 
  85   ```python
  86   import tensorflow_probability as tfp
  87   tfd = tfp.distributions
  88 
  89   # Initialize a single 3-variate Gaussian.
  90   mu = [1., 2, 3]
  91   cov = [[ 0.36,  0.12,  0.06],
  92          [ 0.12,  0.29, -0.13],
  93          [ 0.06, -0.13,  0.26]]
  94   scale = tf.cholesky(cov)
  95   # ==> [[ 0.6,  0. ,  0. ],
  96   #      [ 0.2,  0.5,  0. ],
  97   #      [ 0.1, -0.3,  0.4]])
  98   mvn = tfd.MultivariateNormalTriL(
  99       loc=mu,
 100       scale_tril=scale)
 101 
 102   mvn.mean().eval()
 103   # ==> [1., 2, 3]
 104 
 105   # Covariance agrees with cholesky(cov) parameterization.
 106   mvn.covariance().eval()
 107   # ==> [[ 0.36,  0.12,  0.06],
 108   #      [ 0.12,  0.29, -0.13],
 109   #      [ 0.06, -0.13,  0.26]]
 110 
 111   # Compute the pdf of an observation in `R^3` ; return a scalar.
 112   mvn.prob([-1., 0, 1]).eval()  # shape: []
 113 
 114   # Initialize a 2-batch of 3-variate Gaussians.
 115   mu = [[1., 2, 3],
 116         [11, 22, 33]]              # shape: [2, 3]
 117   tril = ...  # shape: [2, 3, 3], lower triangular, non-zero diagonal.
 118   mvn = tfd.MultivariateNormalTriL(
 119       loc=mu,
 120       scale_tril=tril)
 121 
 122   # Compute the pdf of two `R^3` observations; return a length-2 vector.
 123   x = [[-0.9, 0, 0.1],
 124        [-10, 0, 9]]     # shape: [2, 3]
 125   mvn.prob(x).eval()    # shape: [2]
 126 
 127   # Instantiate a "learnable" MVN.
 128   dims = 4
 129   with tf.variable_scope("model"):
 130     mvn = tfd.MultivariateNormalTriL(
 131         loc=tf.get_variable(shape=[dims], dtype=tf.float32, name="mu"),
 132         scale_tril=tfd.fill_triangular(
 133             tf.get_variable(shape=[dims * (dims + 1) / 2],
 134                             dtype=tf.float32, name="chol_Sigma")))
 135   ```
 136 
 137   """
 138 
 139   @deprecation.deprecated(
 140       "2018-10-01",
 141       "The TensorFlow Distributions library has moved to "
 142       "TensorFlow Probability "
 143       "(https://github.com/tensorflow/probability). You "
 144       "should update all references to use `tfp.distributions` "
 145       "instead of `tf.contrib.distributions`.",
 146       warn_once=True)
 147   def __init__(self,
 148                loc=None,
 149                scale_tril=None,
 150                validate_args=False,
 151                allow_nan_stats=True,
 152                name="MultivariateNormalTriL"):
 153     """Construct Multivariate Normal distribution on `R^k`.
 154 
 155     The `batch_shape` is the broadcast shape between `loc` and `scale`
 156     arguments.
 157 
 158     The `event_shape` is given by last dimension of the matrix implied by
 159     `scale`. The last dimension of `loc` (if provided) must broadcast with this.
 160 
 161     Recall that `covariance = scale @ scale.T`. A (non-batch) `scale` matrix is:
 162 
 163     ```none
 164     scale = scale_tril
 165     ```
 166 
 167     where `scale_tril` is lower-triangular `k x k` matrix with non-zero
 168     diagonal, i.e., `tf.diag_part(scale_tril) != 0`.
 169 
 170     Additional leading dimensions (if any) will index batches.
 171 
 172     Args:
 173       loc: Floating-point `Tensor`. If this is set to `None`, `loc` is
 174         implicitly `0`. When specified, may have shape `[B1, ..., Bb, k]` where
 175         `b >= 0` and `k` is the event size.
 176       scale_tril: Floating-point, lower-triangular `Tensor` with non-zero
 177         diagonal elements. `scale_tril` has shape `[B1, ..., Bb, k, k]` where
 178         `b >= 0` and `k` is the event size.
 179       validate_args: Python `bool`, default `False`. When `True` distribution
 180         parameters are checked for validity despite possibly degrading runtime
 181         performance. When `False` invalid inputs may silently render incorrect
 182         outputs.
 183       allow_nan_stats: Python `bool`, default `True`. When `True`,
 184         statistics (e.g., mean, mode, variance) use the value "`NaN`" to
 185         indicate the result is undefined. When `False`, an exception is raised
 186         if one or more of the statistic's batch members are undefined.
 187       name: Python `str` name prefixed to Ops created by this class.
 188 
 189     Raises:
 190       ValueError: if neither `loc` nor `scale_tril` are specified.
 191     """
 192     parameters = dict(locals())
 193     def _convert_to_tensor(x, name):
 194       return None if x is None else ops.convert_to_tensor(x, name=name)
 195     if loc is None and scale_tril is None:
 196       raise ValueError("Must specify one or both of `loc`, `scale_tril`.")
 197     with ops.name_scope(name) as name:
 198       with ops.name_scope("init", values=[loc, scale_tril]):
 199         loc = _convert_to_tensor(loc, name="loc")
 200         scale_tril = _convert_to_tensor(scale_tril, name="scale_tril")
 201         if scale_tril is None:
 202           scale = linalg.LinearOperatorIdentity(
 203               num_rows=distribution_util.dimension_size(loc, -1),
 204               dtype=loc.dtype,
 205               is_self_adjoint=True,
 206               is_positive_definite=True,
 207               assert_proper_shapes=validate_args)
 208         else:
 209           # No need to validate that scale_tril is non-singular.
 210           # LinearOperatorLowerTriangular has an assert_non_singular
 211           # method that is called by the Bijector.
 212           scale = linalg.LinearOperatorLowerTriangular(
 213               scale_tril,
 214               is_non_singular=True,
 215               is_self_adjoint=False,
 216               is_positive_definite=False)
 217     super(MultivariateNormalTriL, self).__init__(
 218         loc=loc,
 219         scale=scale,
 220         validate_args=validate_args,
 221         allow_nan_stats=allow_nan_stats,
 222         name=name)
 223     self._parameters = parameters
