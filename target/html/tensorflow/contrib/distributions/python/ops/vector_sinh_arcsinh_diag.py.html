  35 class VectorSinhArcsinhDiag(transformed_distribution.TransformedDistribution):
  36   """The (diagonal) SinhArcsinh transformation of a distribution on `R^k`.
  37 
  38   This distribution models a random vector `Y = (Y1,...,Yk)`, making use of
  39   a `SinhArcsinh` transformation (which has adjustable tailweight and skew),
  40   a rescaling, and a shift.
  41 
  42   The `SinhArcsinh` transformation of the Normal is described in great depth in
  43   [Sinh-arcsinh distributions](https://www.jstor.org/stable/27798865).
  44   Here we use a slightly different parameterization, in terms of `tailweight`
  45   and `skewness`.  Additionally we allow for distributions other than Normal,
  46   and control over `scale` as well as a "shift" parameter `loc`.
  47 
  48   #### Mathematical Details
  49 
  50   Given iid random vector `Z = (Z1,...,Zk)`, we define the VectorSinhArcsinhDiag
  51   transformation of `Z`, `Y`, parameterized by
  52   `(loc, scale, skewness, tailweight)`, via the relation (with `@` denoting
  53   matrix multiplication):
  54 
  55   ```
  56   Y := loc + scale @ F(Z) * (2 / F_0(2))
  57   F(Z) := Sinh( (Arcsinh(Z) + skewness) * tailweight )
  58   F_0(Z) := Sinh( Arcsinh(Z) * tailweight )
  59   ```
  60 
  61   This distribution is similar to the location-scale transformation
  62   `L(Z) := loc + scale @ Z` in the following ways:
  63 
  64   * If `skewness = 0` and `tailweight = 1` (the defaults), `F(Z) = Z`, and then
  65     `Y = L(Z)` exactly.
  66   * `loc` is used in both to shift the result by a constant factor.
  67   * The multiplication of `scale` by `2 / F_0(2)` ensures that if `skewness = 0`
  68     `P[Y - loc <= 2 * scale] = P[L(Z) - loc <= 2 * scale]`.
  69     Thus it can be said that the weights in the tails of `Y` and `L(Z)` beyond
  70     `loc + 2 * scale` are the same.
  71 
  72   This distribution is different than `loc + scale @ Z` due to the
  73   reshaping done by `F`:
  74 
  75   * Positive (negative) `skewness` leads to positive (negative) skew.
  76     * positive skew means, the mode of `F(Z)` is "tilted" to the right.
  77     * positive skew means positive values of `F(Z)` become more likely, and
  78       negative values become less likely.
  79   * Larger (smaller) `tailweight` leads to fatter (thinner) tails.
  80     * Fatter tails mean larger values of `|F(Z)|` become more likely.
  81     * `tailweight < 1` leads to a distribution that is "flat" around `Y = loc`,
  82       and a very steep drop-off in the tails.
  83     * `tailweight > 1` leads to a distribution more peaked at the mode with
  84       heavier tails.
  85 
  86   To see the argument about the tails, note that for `|Z| >> 1` and
  87   `|Z| >> (|skewness| * tailweight)**tailweight`, we have
  88   `Y approx 0.5 Z**tailweight e**(sign(Z) skewness * tailweight)`.
  89 
  90   To see the argument regarding multiplying `scale` by `2 / F_0(2)`,
  91 
  92   ```
  93   P[(Y - loc) / scale <= 2] = P[F(Z) * (2 / F_0(2)) <= 2]
  94                             = P[F(Z) <= F_0(2)]
  95                             = P[Z <= 2]  (if F = F_0).
  96   ```
  97   """
  98 
  99   @deprecation.deprecated(
 100       "2018-10-01",
 101       "The TensorFlow Distributions library has moved to "
 102       "TensorFlow Probability "
 103       "(https://github.com/tensorflow/probability). You "
 104       "should update all references to use `tfp.distributions` "
 105       "instead of `tf.contrib.distributions`.",
 106       warn_once=True)
 107   def __init__(self,
 108                loc=None,
 109                scale_diag=None,
 110                scale_identity_multiplier=None,
 111                skewness=None,
 112                tailweight=None,
 113                distribution=None,
 114                validate_args=False,
 115                allow_nan_stats=True,
 116                name="MultivariateNormalLinearOperator"):
 117     """Construct VectorSinhArcsinhDiag distribution on `R^k`.
 118 
 119     The arguments `scale_diag` and `scale_identity_multiplier` combine to
 120     define the diagonal `scale` referred to in this class docstring:
 121 
 122     ```none
 123     scale = diag(scale_diag + scale_identity_multiplier * ones(k))
 124     ```
 125 
 126     The `batch_shape` is the broadcast shape between `loc` and `scale`
 127     arguments.
 128 
 129     The `event_shape` is given by last dimension of the matrix implied by
 130     `scale`. The last dimension of `loc` (if provided) must broadcast with this
 131 
 132     Additional leading dimensions (if any) will index batches.
 133 
 134     Args:
 135       loc: Floating-point `Tensor`. If this is set to `None`, `loc` is
 136         implicitly `0`. When specified, may have shape `[B1, ..., Bb, k]` where
 137         `b >= 0` and `k` is the event size.
 138       scale_diag: Non-zero, floating-point `Tensor` representing a diagonal
 139         matrix added to `scale`. May have shape `[B1, ..., Bb, k]`, `b >= 0`,
 140         and characterizes `b`-batches of `k x k` diagonal matrices added to
 141         `scale`. When both `scale_identity_multiplier` and `scale_diag` are
 142         `None` then `scale` is the `Identity`.
 143       scale_identity_multiplier: Non-zero, floating-point `Tensor` representing
 144         a scale-identity-matrix added to `scale`. May have shape
 145         `[B1, ..., Bb]`, `b >= 0`, and characterizes `b`-batches of scale
 146         `k x k` identity matrices added to `scale`. When both
 147         `scale_identity_multiplier` and `scale_diag` are `None` then `scale`
 148         is the `Identity`.
 149       skewness:  Skewness parameter.  floating-point `Tensor` with shape
 150         broadcastable with `event_shape`.
 151       tailweight:  Tailweight parameter.  floating-point `Tensor` with shape
 152         broadcastable with `event_shape`.
 153       distribution: `tf.Distribution`-like instance. Distribution from which `k`
 154         iid samples are used as input to transformation `F`.  Default is
 155         `tfp.distributions.Normal(loc=0., scale=1.)`.
 156         Must be a scalar-batch, scalar-event distribution.  Typically
 157         `distribution.reparameterization_type = FULLY_REPARAMETERIZED` or it is
 158         a function of non-trainable parameters. WARNING: If you backprop through
 159         a VectorSinhArcsinhDiag sample and `distribution` is not
 160         `FULLY_REPARAMETERIZED` yet is a function of trainable variables, then
 161         the gradient will be incorrect!
 162       validate_args: Python `bool`, default `False`. When `True` distribution
 163         parameters are checked for validity despite possibly degrading runtime
 164         performance. When `False` invalid inputs may silently render incorrect
 165         outputs.
 166       allow_nan_stats: Python `bool`, default `True`. When `True`,
 167         statistics (e.g., mean, mode, variance) use the value "`NaN`" to
 168         indicate the result is undefined. When `False`, an exception is raised
 169         if one or more of the statistic's batch members are undefined.
 170       name: Python `str` name prefixed to Ops created by this class.
 171 
 172     Raises:
 173       ValueError: if at most `scale_identity_multiplier` is specified.
 174     """
 175     parameters = dict(locals())
 176 
 177     with ops.name_scope(
 178         name,
 179         values=[
 180             loc, scale_diag, scale_identity_multiplier, skewness, tailweight
 181         ]) as name:
 182       loc = ops.convert_to_tensor(loc, name="loc") if loc is not None else loc
 183       tailweight = 1. if tailweight is None else tailweight
 184       has_default_skewness = skewness is None
 185       skewness = 0. if skewness is None else skewness
 186 
 187       # Recall, with Z a random variable,
 188       #   Y := loc + C * F(Z),
 189       #   F(Z) := Sinh( (Arcsinh(Z) + skewness) * tailweight )
 190       #   F_0(Z) := Sinh( Arcsinh(Z) * tailweight )
 191       #   C := 2 * scale / F_0(2)
 192 
 193       # Construct shapes and 'scale' out of the scale_* and loc kwargs.
 194       # scale_linop is only an intermediary to:
 195       #  1. get shapes from looking at loc and the two scale args.
 196       #  2. combine scale_diag with scale_identity_multiplier, which gives us
 197       #     'scale', which in turn gives us 'C'.
 198       scale_linop = distribution_util.make_diag_scale(
 199           loc=loc,
 200           scale_diag=scale_diag,
 201           scale_identity_multiplier=scale_identity_multiplier,
 202           validate_args=False,
 203           assert_positive=False)
 204       batch_shape, event_shape = distribution_util.shapes_from_loc_and_scale(
 205           loc, scale_linop)
 206       # scale_linop.diag_part() is efficient since it is a diag type linop.
 207       scale_diag_part = scale_linop.diag_part()
 208       dtype = scale_diag_part.dtype
 209 
 210       if distribution is None:
 211         distribution = normal.Normal(
 212             loc=array_ops.zeros([], dtype=dtype),
 213             scale=array_ops.ones([], dtype=dtype),
 214             allow_nan_stats=allow_nan_stats)
 215       else:
 216         asserts = distribution_util.maybe_check_scalar_distribution(
 217             distribution, dtype, validate_args)
 218         if asserts:
 219           scale_diag_part = control_flow_ops.with_dependencies(
 220               asserts, scale_diag_part)
 221 
 222       # Make the SAS bijector, 'F'.
 223       skewness = ops.convert_to_tensor(skewness, dtype=dtype, name="skewness")
 224       tailweight = ops.convert_to_tensor(
 225           tailweight, dtype=dtype, name="tailweight")
 226       f = bijectors.SinhArcsinh(
 227           skewness=skewness, tailweight=tailweight)
 228       if has_default_skewness:
 229         f_noskew = f
 230       else:
 231         f_noskew = bijectors.SinhArcsinh(
 232             skewness=skewness.dtype.as_numpy_dtype(0.),
 233             tailweight=tailweight)
 234 
 235       # Make the Affine bijector, Z --> loc + C * Z.
 236       c = 2 * scale_diag_part / f_noskew.forward(
 237           ops.convert_to_tensor(2, dtype=dtype))
 238       affine = bijectors.Affine(
 239           shift=loc, scale_diag=c, validate_args=validate_args)
 240 
 241       bijector = bijectors.Chain([affine, f])
 242 
 243       super(VectorSinhArcsinhDiag, self).__init__(
 244           distribution=distribution,
 245           bijector=bijector,
 246           batch_shape=batch_shape,
 247           event_shape=event_shape,
 248           validate_args=validate_args,
 249           name=name)
 250     self._parameters = parameters
 251     self._loc = loc
 252     self._scale = scale_linop
 253     self._tailweight = tailweight
 254     self._skewness = skewness
 255 
 256   @property
 257   def loc(self):
 258     """The `loc` in `Y := loc + scale @ F(Z) * (2 / F(2))."""
 259     return self._loc
 260 
 261   @property
 262   def scale(self):
 263     """The `LinearOperator` `scale` in `Y := loc + scale @ F(Z) * (2 / F(2))."""
 264     return self._scale
 265 
 266   @property
 267   def tailweight(self):
 268     """Controls the tail decay.  `tailweight > 1` means faster than Normal."""
 269     return self._tailweight
 270 
 271   @property
 272   def skewness(self):
 273     """Controls the skewness.  `Skewness > 0` means right skew."""
 274     return self._skewness
