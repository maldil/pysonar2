  42 class HalfNormal(distribution.Distribution):
  43   """The Half Normal distribution with scale `scale`.
  44 
  45   #### Mathematical details
  46 
  47   The half normal is a transformation of a centered normal distribution.
  48   If some random variable `X` has normal distribution,
  49   ```none
  50   X ~ Normal(0.0, scale)
  51   Y = |X|
  52   ```
  53   Then `Y` will have half normal distribution. The probability density
  54   function (pdf) is:
  55 
  56   ```none
  57   pdf(x; scale, x > 0) = sqrt(2) / (scale * sqrt(pi)) *
  58     exp(- 1/2 * (x / scale) ** 2)
  59   )
  60   ```
  61   Where `scale = sigma` is the standard deviation of the underlying normal
  62   distribution.
  63 
  64   #### Examples
  65 
  66   Examples of initialization of one or a batch of distributions.
  67 
  68   ```python
  69   import tensorflow_probability as tfp
  70   tfd = tfp.distributions
  71 
  72   # Define a single scalar HalfNormal distribution.
  73   dist = tfd.HalfNormal(scale=3.0)
  74 
  75   # Evaluate the cdf at 1, returning a scalar.
  76   dist.cdf(1.)
  77 
  78   # Define a batch of two scalar valued HalfNormals.
  79   # The first has scale 11.0, the second 22.0
  80   dist = tfd.HalfNormal(scale=[11.0, 22.0])
  81 
  82   # Evaluate the pdf of the first distribution on 1.0, and the second on 1.5,
  83   # returning a length two tensor.
  84   dist.prob([1.0, 1.5])
  85 
  86   # Get 3 samples, returning a 3 x 2 tensor.
  87   dist.sample([3])
  88   ```
  89 
  90   """
  91 
  92   @deprecation.deprecated(
  93       "2018-10-01",
  94       "The TensorFlow Distributions library has moved to "
  95       "TensorFlow Probability "
  96       "(https://github.com/tensorflow/probability). You "
  97       "should update all references to use `tfp.distributions` "
  98       "instead of `tf.contrib.distributions`.",
  99       warn_once=True)
 100   def __init__(self,
 101                scale,
 102                validate_args=False,
 103                allow_nan_stats=True,
 104                name="HalfNormal"):
 105     """Construct HalfNormals with scale `scale`.
 106 
 107     Args:
 108       scale: Floating point tensor; the scales of the distribution(s).
 109         Must contain only positive values.
 110       validate_args: Python `bool`, default `False`. When `True` distribution
 111         parameters are checked for validity despite possibly degrading runtime
 112         performance. When `False` invalid inputs may silently render incorrect
 113         outputs.
 114       allow_nan_stats: Python `bool`, default `True`. When `True`,
 115         statistics (e.g., mean, mode, variance) use the value "`NaN`" to
 116         indicate the result is undefined. When `False`, an exception is raised
 117         if one or more of the statistic's batch members are undefined.
 118       name: Python `str` name prefixed to Ops created by this class.
 119     """
 120     parameters = dict(locals())
 121     with ops.name_scope(name, values=[scale]) as name:
 122       with ops.control_dependencies([check_ops.assert_positive(scale)] if
 123                                     validate_args else []):
 124         self._scale = array_ops.identity(scale, name="scale")
 125     super(HalfNormal, self).__init__(
 126         dtype=self._scale.dtype,
 127         reparameterization_type=distribution.FULLY_REPARAMETERIZED,
 128         validate_args=validate_args,
 129         allow_nan_stats=allow_nan_stats,
 130         parameters=parameters,
 131         graph_parents=[self._scale],
 132         name=name)
 133 
 134   @staticmethod
 135   def _param_shapes(sample_shape):
 136     return {"scale": ops.convert_to_tensor(sample_shape, dtype=dtypes.int32)}
 137 
 138   @property
 139   def scale(self):
 140     """Distribution parameter for the scale."""
 141     return self._scale
 142 
 143   def _batch_shape_tensor(self):
 144     return array_ops.shape(self.scale)
 145 
 146   def _batch_shape(self):
 147     return self.scale.shape
 148 
 149   def _event_shape_tensor(self):
 150     return constant_op.constant([], dtype=dtypes.int32)
 151 
 152   def _event_shape(self):
 153     return tensor_shape.scalar()
 154 
 155   def _sample_n(self, n, seed=None):
 156     shape = array_ops.concat([[n], self.batch_shape_tensor()], 0)
 157     sampled = random_ops.random_normal(
 158         shape=shape, mean=0., stddev=1., dtype=self.dtype, seed=seed)
 159     return math_ops.abs(sampled * self.scale)
 160 
 161   def _prob(self, x):
 162     coeff = np.sqrt(2) / self.scale / np.sqrt(np.pi)
 163     pdf = coeff * math_ops.exp(- 0.5 * (x / self.scale) ** 2)
 164     return pdf * math_ops.cast(x >= 0, self.dtype)
 165 
 166   def _cdf(self, x):
 167     truncated_x = nn.relu(x)
 168     return math_ops.erf(truncated_x / self.scale / np.sqrt(2.0))
 169 
 170   def _entropy(self):
 171     return 0.5 * math_ops.log(np.pi * self.scale ** 2.0 / 2.0) + 0.5
 172 
 173   def _mean(self):
 174     return self.scale * np.sqrt(2.0) / np.sqrt(np.pi)
 175 
 176   def _quantile(self, p):
 177     return np.sqrt(2.0) * self.scale * special_math.erfinv(p)
 178 
 179   def _mode(self):
 180     return array_ops.zeros(self.batch_shape_tensor())
 181 
 182   def _variance(self):
 183     return self.scale ** 2.0 * (1.0 - 2.0 / np.pi)
