  39 class Cauchy(distribution.Distribution):
  40   """The Cauchy distribution with location `loc` and scale `scale`.
  41 
  42   #### Mathematical details
  43 
  44   The probability density function (pdf) is,
  45 
  46   ```none
  47   pdf(x; loc, scale) = 1 / (pi scale (1 + z**2))
  48   z = (x - loc) / scale
  49   ```
  50   where `loc` is the location, and `scale` is the scale.
  51 
  52   The Cauchy distribution is a member of the [location-scale family](
  53   https://en.wikipedia.org/wiki/Location-scale_family), i.e.
  54   `Y ~ Cauchy(loc, scale)` is equivalent to,
  55 
  56   ```none
  57   X ~ Cauchy(loc=0, scale=1)
  58   Y = loc + scale * X
  59   ```
  60 
  61   #### Examples
  62 
  63   Examples of initialization of one or a batch of distributions.
  64 
  65   ```python
  66   import tensorflow_probability as tfp
  67   tfd = tfp.distributions
  68 
  69   # Define a single scalar Cauchy distribution.
  70   dist = tfd.Cauchy(loc=0., scale=3.)
  71 
  72   # Evaluate the cdf at 1, returning a scalar.
  73   dist.cdf(1.)
  74 
  75   # Define a batch of two scalar valued Cauchy distributions.
  76   dist = tfd.Cauchy(loc=[1, 2.], scale=[11, 22.])
  77 
  78   # Evaluate the pdf of the first distribution on 0, and the second on 1.5,
  79   # returning a length two tensor.
  80   dist.prob([0, 1.5])
  81 
  82   # Get 3 samples, returning a 3 x 2 tensor.
  83   dist.sample([3])
  84 
  85   # Arguments are broadcast when possible.
  86   # Define a batch of two scalar valued Cauchy distributions.
  87   # Both have median 1, but different scales.
  88   dist = tfd.Cauchy(loc=1., scale=[11, 22.])
  89 
  90   # Evaluate the pdf of both distributions on the same point, 3.0,
  91   # returning a length 2 tensor.
  92   dist.prob(3.)
  93   ```
  94 
  95   """
  96 
  97   @deprecation.deprecated(
  98       "2018-10-01",
  99       "The TensorFlow Distributions library has moved to "
 100       "TensorFlow Probability "
 101       "(https://github.com/tensorflow/probability). You "
 102       "should update all references to use `tfp.distributions` "
 103       "instead of `tf.contrib.distributions`.",
 104       warn_once=True)
 105   def __init__(self,
 106                loc,
 107                scale,
 108                validate_args=False,
 109                allow_nan_stats=True,
 110                name="Cauchy"):
 111     """Construct Cauchy distributions.
 112 
 113     The parameters `loc` and `scale` must be shaped in a way that supports
 114     broadcasting (e.g. `loc + scale` is a valid operation).
 115 
 116     Args:
 117       loc: Floating point tensor; the modes of the distribution(s).
 118       scale: Floating point tensor; the locations of the distribution(s).
 119         Must contain only positive values.
 120       validate_args: Python `bool`, default `False`. When `True` distribution
 121         parameters are checked for validity despite possibly degrading runtime
 122         performance. When `False` invalid inputs may silently render incorrect
 123         outputs.
 124       allow_nan_stats: Python `bool`, default `True`. When `True`,
 125         statistics (e.g., mean, mode, variance) use the value "`NaN`" to
 126         indicate the result is undefined. When `False`, an exception is raised
 127         if one or more of the statistic's batch members are undefined.
 128       name: Python `str` name prefixed to Ops created by this class.
 129 
 130     Raises:
 131       TypeError: if `loc` and `scale` have different `dtype`.
 132     """
 133     parameters = dict(locals())
 134     with ops.name_scope(name, values=[loc, scale]) as name:
 135       with ops.control_dependencies([check_ops.assert_positive(scale)]
 136                                     if validate_args else []):
 137         self._loc = array_ops.identity(loc, name="loc")
 138         self._scale = array_ops.identity(scale, name="scale")
 139         check_ops.assert_same_float_dtype([self._loc, self._scale])
 140     super(Cauchy, self).__init__(
 141         dtype=self._scale.dtype,
 142         reparameterization_type=distribution.FULLY_REPARAMETERIZED,
 143         validate_args=validate_args,
 144         allow_nan_stats=allow_nan_stats,
 145         parameters=parameters,
 146         graph_parents=[self._loc, self._scale],
 147         name=name)
 148 
 149   @staticmethod
 150   def _param_shapes(sample_shape):
 151     return dict(
 152         zip(("loc", "scale"),
 153             ([ops.convert_to_tensor(sample_shape, dtype=dtypes.int32)] * 2)))
 154 
 155   @property
 156   def loc(self):
 157     """Distribution parameter for the mean."""
 158     return self._loc
 159 
 160   @property
 161   def scale(self):
 162     """Distribution parameter for standard deviation."""
 163     return self._scale
 164 
 165   def _batch_shape_tensor(self):
 166     return array_ops.broadcast_dynamic_shape(
 167         array_ops.shape(self.loc), array_ops.shape(self.scale))
 168 
 169   def _batch_shape(self):
 170     return array_ops.broadcast_static_shape(self.loc.shape, self.scale.shape)
 171 
 172   def _event_shape_tensor(self):
 173     return constant_op.constant([], dtype=dtypes.int32)
 174 
 175   def _event_shape(self):
 176     return tensor_shape.scalar()
 177 
 178   def _sample_n(self, n, seed=None):
 179     shape = array_ops.concat([[n], self.batch_shape_tensor()], 0)
 180     probs = random_ops.random_uniform(
 181         shape=shape, minval=0., maxval=1., dtype=self.dtype, seed=seed)
 182     return self._quantile(probs)
 183 
 184   def _log_prob(self, x):
 185     return self._log_unnormalized_prob(x) - self._log_normalization()
 186 
 187   def _cdf(self, x):
 188     return math_ops.atan(self._z(x)) / np.pi + 0.5
 189 
 190   def _log_cdf(self, x):
 191     return math_ops.log1p(2 / np.pi * math_ops.atan(self._z(x))) - np.log(2)
 192 
 193   def _log_unnormalized_prob(self, x):
 194     return -math_ops.log1p(math_ops.square(self._z(x)))
 195 
 196   def _log_normalization(self):
 197     return np.log(np.pi) + math_ops.log(self.scale)
 198 
 199   def _entropy(self):
 200     h = np.log(4 * np.pi) + math_ops.log(self.scale)
 201     return h * array_ops.ones_like(self.loc)
 202 
 203   def _quantile(self, p):
 204     return self.loc + self.scale * math_ops.tan(np.pi * (p - 0.5))
 205 
 206   def _mode(self):
 207     return self.loc * array_ops.ones_like(self.scale)
 208 
 209   def _z(self, x):
 210     """Standardize input `x`."""
 211     with ops.name_scope("standardize", values=[x]):
 212       return (x - self.loc) / self.scale
 213 
 214   def _inv_z(self, z):
 215     """Reconstruct input `x` from a its normalized version."""
 216     with ops.name_scope("reconstruct", values=[z]):
 217       return z * self.scale + self.loc
 218 
 219   def _mean(self):
 220     if self.allow_nan_stats:
 221       return array_ops.fill(self.batch_shape_tensor(),
 222                             self.dtype.as_numpy_dtype(np.nan))
 223     else:
 224       raise ValueError("`mean` is undefined for Cauchy distribution.")
 225 
 226   def _stddev(self):
 227     if self.allow_nan_stats:
 228       return array_ops.fill(self.batch_shape_tensor(),
 229                             self.dtype.as_numpy_dtype(np.nan))
 230     else:
 231       raise ValueError("`stddev` is undefined for Cauchy distribution.")
