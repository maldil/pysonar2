  55 class VectorLaplaceLinearOperator(
  56     transformed_distribution.TransformedDistribution):
  57   """The vectorization of the Laplace distribution on `R^k`.
  58 
  59   The vector laplace distribution is defined over `R^k`, and parameterized by
  60   a (batch of) length-`k` `loc` vector (the means) and a (batch of) `k x k`
  61   `scale` matrix:  `covariance = 2 * scale @ scale.T`, where `@` denotes
  62   matrix-multiplication.
  63 
  64   #### Mathematical Details
  65 
  66   The probability density function (pdf) is,
  67 
  68   ```none
  69   pdf(x; loc, scale) = exp(-||y||_1) / Z,
  70   y = inv(scale) @ (x - loc),
  71   Z = 2**k |det(scale)|,
  72   ```
  73 
  74   where:
  75 
  76   * `loc` is a vector in `R^k`,
  77   * `scale` is a linear operator in `R^{k x k}`, `cov = scale @ scale.T`,
  78   * `Z` denotes the normalization constant, and,
  79   * `||y||_1` denotes the `l1` norm of `y`, `sum_i |y_i|.
  80 
  81   The VectorLaplace distribution is a member of the [location-scale
  82   family](https://en.wikipedia.org/wiki/Location-scale_family), i.e., it can be
  83   constructed as,
  84 
  85   ```none
  86   X = (X_1, ..., X_k), each X_i ~ Laplace(loc=0, scale=1)
  87   Y = (Y_1, ...,Y_k) = scale @ X + loc
  88   ```
  89 
  90   #### About `VectorLaplace` and `Vector` distributions in TensorFlow.
  91 
  92   The `VectorLaplace` is a non-standard distribution that has useful properties.
  93 
  94   The marginals `Y_1, ..., Y_k` are *not* Laplace random variables, due to
  95   the fact that the sum of Laplace random variables is not Laplace.
  96 
  97   Instead, `Y` is a vector whose components are linear combinations of Laplace
  98   random variables.  Thus, `Y` lives in the vector space generated by `vectors`
  99   of Laplace distributions.  This allows the user to decide the mean and
 100   covariance (by setting `loc` and `scale`), while preserving some properties of
 101   the Laplace distribution.  In particular, the tails of `Y_i` will be (up to
 102   polynomial factors) exponentially decaying.
 103 
 104   To see this last statement, note that the pdf of `Y_i` is the convolution of
 105   the pdf of `k` independent Laplace random variables.  One can then show by
 106   induction that distributions with exponential (up to polynomial factors) tails
 107   are closed under convolution.
 108 
 109 
 110   #### Examples
 111 
 112   ```python
 113   import tensorflow_probability as tfp
 114   tfd = tfp.distributions
 115 
 116   # Initialize a single 3-variate VectorLaplace with some desired covariance.
 117   mu = [1., 2, 3]
 118   cov = [[ 0.36,  0.12,  0.06],
 119          [ 0.12,  0.29, -0.13],
 120          [ 0.06, -0.13,  0.26]]
 121 
 122   scale = tf.cholesky(cov)
 123   # ==> [[ 0.6,  0. ,  0. ],
 124   #      [ 0.2,  0.5,  0. ],
 125   #      [ 0.1, -0.3,  0.4]])
 126 
 127   # Divide scale by sqrt(2) so that the final covariance will be what we want.
 128   vla = tfd.VectorLaplaceLinearOperator(
 129       loc=mu,
 130       scale=tf.linalg.LinearOperatorLowerTriangular(scale / tf.sqrt(2.)))
 131 
 132   # Covariance agrees with cholesky(cov) parameterization.
 133   vla.covariance().eval()
 134   # ==> [[ 0.36,  0.12,  0.06],
 135   #      [ 0.12,  0.29, -0.13],
 136   #      [ 0.06, -0.13,  0.26]]
 137 
 138   # Compute the pdf of an`R^3` observation; return a scalar.
 139   vla.prob([-1., 0, 1]).eval()  # shape: []
 140 
 141   # Initialize a 2-batch of 3-variate Vector Laplace's.
 142   mu = [[1., 2, 3],
 143         [11, 22, 33]]              # shape: [2, 3]
 144   scale_diag = [[1., 2, 3],
 145                 [0.5, 1, 1.5]]     # shape: [2, 3]
 146 
 147   vla = tfd.VectorLaplaceLinearOperator(
 148       loc=mu,
 149       scale=tf.linalg.LinearOperatorDiag(scale_diag))
 150 
 151   # Compute the pdf of two `R^3` observations; return a length-2 vector.
 152   x = [[-0.9, 0, 0.1],
 153        [-10, 0, 9]]     # shape: [2, 3]
 154   vla.prob(x).eval()    # shape: [2]
 155   ```
 156 
 157   """
 158 
 159   @deprecation.deprecated(
 160       "2018-10-01",
 161       "The TensorFlow Distributions library has moved to "
 162       "TensorFlow Probability "
 163       "(https://github.com/tensorflow/probability). You "
 164       "should update all references to use `tfp.distributions` "
 165       "instead of `tf.contrib.distributions`.",
 166       warn_once=True)
 167   def __init__(self,
 168                loc=None,
 169                scale=None,
 170                validate_args=False,
 171                allow_nan_stats=True,
 172                name="VectorLaplaceLinearOperator"):
 173     """Construct Vector Laplace distribution on `R^k`.
 174 
 175     The `batch_shape` is the broadcast shape between `loc` and `scale`
 176     arguments.
 177 
 178     The `event_shape` is given by last dimension of the matrix implied by
 179     `scale`. The last dimension of `loc` (if provided) must broadcast with this.
 180 
 181     Recall that `covariance = 2 * scale @ scale.T`.
 182 
 183     Additional leading dimensions (if any) will index batches.
 184 
 185     Args:
 186       loc: Floating-point `Tensor`. If this is set to `None`, `loc` is
 187         implicitly `0`. When specified, may have shape `[B1, ..., Bb, k]` where
 188         `b >= 0` and `k` is the event size.
 189       scale: Instance of `LinearOperator` with same `dtype` as `loc` and shape
 190         `[B1, ..., Bb, k, k]`.
 191       validate_args: Python `bool`, default `False`. Whether to validate input
 192         with asserts. If `validate_args` is `False`, and the inputs are
 193         invalid, correct behavior is not guaranteed.
 194       allow_nan_stats: Python `bool`, default `True`. If `False`, raise an
 195         exception if a statistic (e.g. mean/mode/etc...) is undefined for any
 196         batch member If `True`, batch members with valid parameters leading to
 197         undefined statistics will return NaN for this statistic.
 198       name: The name to give Ops created by the initializer.
 199 
 200     Raises:
 201       ValueError: if `scale` is unspecified.
 202       TypeError: if not `scale.dtype.is_floating`
 203     """
 204     parameters = dict(locals())
 205     if scale is None:
 206       raise ValueError("Missing required `scale` parameter.")
 207     if not scale.dtype.is_floating:
 208       raise TypeError("`scale` parameter must have floating-point dtype.")
 209 
 210     with ops.name_scope(name, values=[loc] + scale.graph_parents):
 211       # Since expand_dims doesn't preserve constant-ness, we obtain the
 212       # non-dynamic value if possible.
 213       loc = ops.convert_to_tensor(loc, name="loc") if loc is not None else loc
 214       batch_shape, event_shape = distribution_util.shapes_from_loc_and_scale(
 215           loc, scale)
 216 
 217       super(VectorLaplaceLinearOperator, self).__init__(
 218           distribution=laplace.Laplace(
 219               loc=array_ops.zeros([], dtype=scale.dtype),
 220               scale=array_ops.ones([], dtype=scale.dtype)),
 221           bijector=bijectors.AffineLinearOperator(
 222               shift=loc, scale=scale, validate_args=validate_args),
 223           batch_shape=batch_shape,
 224           event_shape=event_shape,
 225           validate_args=validate_args,
 226           name=name)
 227       self._parameters = parameters
 228 
 229   @property
 230   def loc(self):
 231     """The `loc` `Tensor` in `Y = scale @ X + loc`."""
 232     return self.bijector.shift
 233 
 234   @property
 235   def scale(self):
 236     """The `scale` `LinearOperator` in `Y = scale @ X + loc`."""
 237     return self.bijector.scale
 238 
 239   @distribution_util.AppendDocstring(_mvn_sample_note)
 240   def _log_prob(self, x):
 241     return super(VectorLaplaceLinearOperator, self)._log_prob(x)
 242 
 243   @distribution_util.AppendDocstring(_mvn_sample_note)
 244   def _prob(self, x):
 245     return super(VectorLaplaceLinearOperator, self)._prob(x)
 246 
 247   def _mean(self):
 248     shape = self.batch_shape.concatenate(self.event_shape)
 249     has_static_shape = shape.is_fully_defined()
 250     if not has_static_shape:
 251       shape = array_ops.concat([
 252           self.batch_shape_tensor(),
 253           self.event_shape_tensor(),
 254       ], 0)
 255 
 256     if self.loc is None:
 257       return array_ops.zeros(shape, self.dtype)
 258 
 259     if has_static_shape and shape == self.loc.get_shape():
 260       return array_ops.identity(self.loc)
 261 
 262     # Add dummy tensor of zeros to broadcast.  This is only necessary if shape
 263     # != self.loc.shape, but we could not determine if this is the case.
 264     return array_ops.identity(self.loc) + array_ops.zeros(shape, self.dtype)
 265 
 266   def _covariance(self):
 267     # Let
 268     #   W = (w1,...,wk), with wj ~ iid Laplace(0, 1).
 269     # Then this distribution is
 270     #   X = loc + LW,
 271     # and since E[X] = loc,
 272     #   Cov(X) = E[LW W^T L^T] = L E[W W^T] L^T.
 273     # Since E[wi wj] = 0 if i != j, and 2 if i == j, we have
 274     #   Cov(X) = 2 LL^T
 275     if distribution_util.is_diagonal_scale(self.scale):
 276       return 2. * array_ops.matrix_diag(math_ops.square(self.scale.diag_part()))
 277     else:
 278       return 2. * self.scale.matmul(self.scale.to_dense(), adjoint_arg=True)
 279 
 280   def _variance(self):
 281     if distribution_util.is_diagonal_scale(self.scale):
 282       return 2. * math_ops.square(self.scale.diag_part())
 283     elif (isinstance(self.scale, linalg.LinearOperatorLowRankUpdate) and
 284           self.scale.is_self_adjoint):
 285       return array_ops.matrix_diag_part(
 286           2. * self.scale.matmul(self.scale.to_dense()))
 287     else:
 288       return 2. * array_ops.matrix_diag_part(
 289           self.scale.matmul(self.scale.to_dense(), adjoint_arg=True))
 290 
 291   def _stddev(self):
 292     if distribution_util.is_diagonal_scale(self.scale):
 293       return np.sqrt(2) * math_ops.abs(self.scale.diag_part())
 294     elif (isinstance(self.scale, linalg.LinearOperatorLowRankUpdate) and
 295           self.scale.is_self_adjoint):
 296       return np.sqrt(2) * math_ops.sqrt(array_ops.matrix_diag_part(
 297           self.scale.matmul(self.scale.to_dense())))
 298     else:
 299       return np.sqrt(2) * math_ops.sqrt(array_ops.matrix_diag_part(
 300           self.scale.matmul(self.scale.to_dense(), adjoint_arg=True)))
 301 
 302   def _mode(self):
 303     return self._mean()
