  40   def run_test_sample_consistent_log_prob(
  41       self, sess_run_fn, dist,
  42       num_samples=int(1e5), num_threshold=int(1e3), seed=42,
  43       batch_size=None,
  44       rtol=1e-2, atol=0.):
  45     """Tests that sample/log_prob are consistent with each other.
  46 
  47     "Consistency" means that `sample` and `log_prob` correspond to the same
  48     distribution.
  49 
  50     Note: this test only verifies a necessary condition for consistency--it does
  51     does not verify sufficiency hence does not prove `sample`, `log_prob` truly
  52     are consistent.
  53 
  54     Args:
  55       sess_run_fn: Python `callable` taking `list`-like of `Tensor`s and
  56         returning a list of results after running one "step" of TensorFlow
  57         computation, typically set to `sess.run`.
  58       dist: Distribution instance or object which implements `sample`,
  59         `log_prob`, `event_shape_tensor` and `batch_shape_tensor`.
  60       num_samples: Python `int` scalar indicating the number of Monte-Carlo
  61         samples to draw from `dist`.
  62       num_threshold: Python `int` scalar indicating the number of samples a
  63         bucket must contain before being compared to the probability.
  64         Default value: 1e3; must be at least 1.
  65         Warning, set too high will cause test to falsely pass but setting too
  66         low will cause the test to falsely fail.
  67       seed: Python `int` indicating the seed to use when sampling from `dist`.
  68         In general it is not recommended to use `None` during a test as this
  69         increases the likelihood of spurious test failure.
  70       batch_size: Hint for unpacking result of samples. Default: `None` means
  71         batch_size is inferred.
  72       rtol: Python `float`-type indicating the admissible relative error between
  73         analytical and sample statistics.
  74       atol: Python `float`-type indicating the admissible absolute error between
  75         analytical and sample statistics.
  76 
  77     Raises:
  78       ValueError: if `num_threshold < 1`.
  79     """
  80     if num_threshold < 1:
  81       raise ValueError("num_threshold({}) must be at least 1.".format(
  82           num_threshold))
  83     # Histogram only supports vectors so we call it once per batch coordinate.
  84     y = dist.sample(num_samples, seed=seed)
  85     y = array_ops.reshape(y, shape=[num_samples, -1])
  86     if batch_size is None:
  87       batch_size = math_ops.reduce_prod(dist.batch_shape_tensor())
  88     batch_dims = array_ops.shape(dist.batch_shape_tensor())[0]
  89     edges_expanded_shape = 1 + array_ops.pad([-2], paddings=[[0, batch_dims]])
  90     for b, x in enumerate(array_ops.unstack(y, num=batch_size, axis=1)):
  91       counts, edges = self.histogram(x)
  92       edges = array_ops.reshape(edges, edges_expanded_shape)
  93       probs = math_ops.exp(dist.log_prob(edges))
  94       probs = array_ops.reshape(probs, shape=[-1, batch_size])[:, b]
  95 
  96       [counts_, probs_] = sess_run_fn([counts, probs])
  97       valid = counts_ > num_threshold
  98       probs_ = probs_[valid]
  99       counts_ = counts_[valid]
 100       self.assertAllClose(probs_, counts_ / num_samples,
 101                           rtol=rtol, atol=atol)

 103   def run_test_sample_consistent_mean_variance(
 104       self, sess_run_fn, dist,
 105       num_samples=int(1e5), seed=24,
 106       rtol=1e-2, atol=0.):
 107     """Tests that sample/mean/variance are consistent with each other.
 108 
 109     "Consistency" means that `sample`, `mean`, `variance`, etc all correspond
 110     to the same distribution.
 111 
 112     Args:
 113       sess_run_fn: Python `callable` taking `list`-like of `Tensor`s and
 114         returning a list of results after running one "step" of TensorFlow
 115         computation, typically set to `sess.run`.
 116       dist: Distribution instance or object which implements `sample`,
 117         `log_prob`, `event_shape_tensor` and `batch_shape_tensor`.
 118       num_samples: Python `int` scalar indicating the number of Monte-Carlo
 119         samples to draw from `dist`.
 120       seed: Python `int` indicating the seed to use when sampling from `dist`.
 121         In general it is not recommended to use `None` during a test as this
 122         increases the likelihood of spurious test failure.
 123       rtol: Python `float`-type indicating the admissible relative error between
 124         analytical and sample statistics.
 125       atol: Python `float`-type indicating the admissible absolute error between
 126         analytical and sample statistics.
 127     """
 128     x = math_ops.to_float(dist.sample(num_samples, seed=seed))
 129     sample_mean = math_ops.reduce_mean(x, axis=0)
 130     sample_variance = math_ops.reduce_mean(
 131         math_ops.square(x - sample_mean), axis=0)
 132     sample_stddev = math_ops.sqrt(sample_variance)
 133 
 134     [
 135         sample_mean_,
 136         sample_variance_,
 137         sample_stddev_,
 138         mean_,
 139         variance_,
 140         stddev_
 141     ] = sess_run_fn([
 142         sample_mean,
 143         sample_variance,
 144         sample_stddev,
 145         dist.mean(),
 146         dist.variance(),
 147         dist.stddev(),
 148     ])
 149 
 150     self.assertAllClose(mean_, sample_mean_, rtol=rtol, atol=atol)
 151     self.assertAllClose(variance_, sample_variance_, rtol=rtol, atol=atol)
 152     self.assertAllClose(stddev_, sample_stddev_, rtol=rtol, atol=atol)
 153 
 154   def histogram(self, x, value_range=None, nbins=None, name=None):
 155     """Return histogram of values.
 156 
 157     Given the tensor `values`, this operation returns a rank 1 histogram
 158     counting the number of entries in `values` that fell into every bin. The
 159     bins are equal width and determined by the arguments `value_range` and
 160     `nbins`.
 161 
 162     Args:
 163       x: 1D numeric `Tensor` of items to count.
 164       value_range:  Shape [2] `Tensor`. `new_values <= value_range[0]` will be
 165         mapped to `hist[0]`, `values >= value_range[1]` will be mapped to
 166         `hist[-1]`. Must be same dtype as `x`.
 167       nbins:  Scalar `int32 Tensor`.  Number of histogram bins.
 168       name: Python `str` name prefixed to Ops created by this class.
 169 
 170     Returns:
 171       counts: 1D `Tensor` of counts, i.e.,
 172         `counts[i] = sum{ edges[i-1] <= values[j] < edges[i] : j }`.
 173       edges: 1D `Tensor` characterizing intervals used for counting.
 174     """
 175     with ops.name_scope(name, "histogram", [x]):
 176       x = ops.convert_to_tensor(x, name="x")
 177       if value_range is None:
 178         value_range = [math_ops.reduce_min(x), 1 + math_ops.reduce_max(x)]
 179       value_range = ops.convert_to_tensor(value_range, name="value_range")
 180       lo = value_range[0]
 181       hi = value_range[1]
 182       if nbins is None:
 183         nbins = math_ops.to_int32(hi - lo)
 184       delta = (hi - lo) / math_ops.cast(
 185           nbins, dtype=value_range.dtype.base_dtype)
 186       edges = math_ops.range(
 187           start=lo, limit=hi, delta=delta, dtype=x.dtype.base_dtype)
 188       counts = histogram_ops.histogram_fixed_width(
 189           x, value_range=value_range, nbins=nbins)
 190       return counts, edges

 196   def run_test_sample_consistent_log_prob(
 197       self,
 198       sess_run_fn,
 199       dist,
 200       num_samples=int(1e5),
 201       radius=1.,
 202       center=0.,
 203       seed=42,
 204       rtol=1e-2,
 205       atol=0.):
 206     """Tests that sample/log_prob are mutually consistent.
 207 
 208     "Consistency" means that `sample` and `log_prob` correspond to the same
 209     distribution.
 210 
 211     The idea of this test is to compute the Monte-Carlo estimate of the volume
 212     enclosed by a hypersphere, i.e., the volume of an `n`-ball. While we could
 213     choose an arbitrary function to integrate, the hypersphere's volume is nice
 214     because it is intuitive, has an easy analytical expression, and works for
 215     `dimensions > 1`.
 216 
 217     Technical Details:
 218 
 219     Observe that:
 220 
 221     ```none
 222     int_{R**d} dx [x in Ball(radius=r, center=c)]
 223     = E_{p(X)}[ [X in Ball(r, c)] / p(X) ]
 224     = lim_{m->infty} m**-1 sum_j^m [x[j] in Ball(r, c)] / p(x[j]),
 225         where x[j] ~iid p(X)
 226     ```
 227 
 228     Thus, for fixed `m`, the above is approximately true when `sample` and
 229     `log_prob` are mutually consistent.
 230 
 231     Furthermore, the above calculation has the analytical result:
 232     `pi**(d/2) r**d / Gamma(1 + d/2)`.
 233 
 234     Note: this test only verifies a necessary condition for consistency--it does
 235     does not verify sufficiency hence does not prove `sample`, `log_prob` truly
 236     are consistent. For this reason we recommend testing several different
 237     hyperspheres (assuming the hypersphere is supported by the distribution).
 238     Furthermore, we gain additional trust in this test when also tested `sample`
 239     against the first, second moments
 240     (`run_test_sample_consistent_mean_covariance`); it is probably unlikely that
 241     a "best-effort" implementation of `log_prob` would incorrectly pass both
 242     tests and for different hyperspheres.
 243 
 244     For a discussion on the analytical result (second-line) see:
 245       https://en.wikipedia.org/wiki/Volume_of_an_n-ball.
 246 
 247     For a discussion of importance sampling (fourth-line) see:
 248       https://en.wikipedia.org/wiki/Importance_sampling.
 249 
 250     Args:
 251       sess_run_fn: Python `callable` taking `list`-like of `Tensor`s and
 252         returning a list of results after running one "step" of TensorFlow
 253         computation, typically set to `sess.run`.
 254       dist: Distribution instance or object which implements `sample`,
 255         `log_prob`, `event_shape_tensor` and `batch_shape_tensor`. The
 256         distribution must have non-zero probability of sampling every point
 257         enclosed by the hypersphere.
 258       num_samples: Python `int` scalar indicating the number of Monte-Carlo
 259         samples to draw from `dist`.
 260       radius: Python `float`-type indicating the radius of the `n`-ball which
 261         we're computing the volume.
 262       center: Python floating-type vector (or scalar) indicating the center of
 263         the `n`-ball which we're computing the volume. When scalar, the value is
 264         broadcast to all event dims.
 265       seed: Python `int` indicating the seed to use when sampling from `dist`.
 266         In general it is not recommended to use `None` during a test as this
 267         increases the likelihood of spurious test failure.
 268       rtol: Python `float`-type indicating the admissible relative error between
 269         actual- and approximate-volumes.
 270       atol: Python `float`-type indicating the admissible absolute error between
 271         actual- and approximate-volumes. In general this should be zero since
 272         a typical radius implies a non-zero volume.
 273     """
 274 
 275     def actual_hypersphere_volume(dims, radius):
 276       # https://en.wikipedia.org/wiki/Volume_of_an_n-ball
 277       # Using tf.lgamma because we'd have to otherwise use SciPy which is not
 278       # a required dependency of core.
 279       radius = np.asarray(radius)
 280       dims = math_ops.cast(dims, dtype=radius.dtype)
 281       return math_ops.exp(
 282           (dims / 2.) * np.log(np.pi)
 283           - math_ops.lgamma(1. + dims / 2.)
 284           + dims * math_ops.log(radius))
 285 
 286     def is_in_ball(x, radius, center):
 287       return math_ops.cast(linalg_ops.norm(x - center, axis=-1) <= radius,
 288                            dtype=x.dtype)
 289 
 290     def monte_carlo_hypersphere_volume(dist, num_samples, radius, center):
 291       # https://en.wikipedia.org/wiki/Importance_sampling
 292       x = dist.sample(num_samples, seed=seed)
 293       x = array_ops.identity(x)  # Invalidate bijector cacheing.
 294       return math_ops.reduce_mean(
 295           math_ops.exp(-dist.log_prob(x)) * is_in_ball(x, radius, center),
 296           axis=0)
 297 
 298     # Build graph.
 299     with ops.name_scope(
 300         "run_test_sample_consistent_log_prob",
 301         values=[num_samples, radius, center] + dist._graph_parents):  # pylint: disable=protected-access
 302       batch_shape = dist.batch_shape_tensor()
 303       actual_volume = actual_hypersphere_volume(
 304           dims=dist.event_shape_tensor()[0],
 305           radius=radius)
 306       sample_volume = monte_carlo_hypersphere_volume(
 307           dist,
 308           num_samples=num_samples,
 309           radius=radius,
 310           center=center)
 311       init_op = variables_ops.global_variables_initializer()
 312 
 313     # Execute graph.
 314     sess_run_fn(init_op)
 315     [batch_shape_, actual_volume_, sample_volume_] = sess_run_fn([
 316         batch_shape, actual_volume, sample_volume])
 317 
 318     # Check results.
 319     self.assertAllClose(np.tile(actual_volume_, reps=batch_shape_),
 320                         sample_volume_,
 321                         rtol=rtol, atol=atol)
 322 
 323   def run_test_sample_consistent_mean_covariance(
 324       self,
 325       sess_run_fn,
 326       dist,
 327       num_samples=int(1e5),
 328       seed=24,
 329       rtol=1e-2,
 330       atol=0.1,
 331       cov_rtol=None,
 332       cov_atol=None):
 333     """Tests that sample/mean/covariance are consistent with each other.
 334 
 335     "Consistency" means that `sample`, `mean`, `covariance`, etc all correspond
 336     to the same distribution.
 337 
 338     Args:
 339       sess_run_fn: Python `callable` taking `list`-like of `Tensor`s and
 340         returning a list of results after running one "step" of TensorFlow
 341         computation, typically set to `sess.run`.
 342       dist: Distribution instance or object which implements `sample`,
 343         `log_prob`, `event_shape_tensor` and `batch_shape_tensor`.
 344       num_samples: Python `int` scalar indicating the number of Monte-Carlo
 345         samples to draw from `dist`.
 346       seed: Python `int` indicating the seed to use when sampling from `dist`.
 347         In general it is not recommended to use `None` during a test as this
 348         increases the likelihood of spurious test failure.
 349       rtol: Python `float`-type indicating the admissible relative error between
 350         analytical and sample statistics.
 351       atol: Python `float`-type indicating the admissible absolute error between
 352         analytical and sample statistics.
 353       cov_rtol: Python `float`-type indicating the admissible relative error
 354         between analytical and sample covariance. Default: rtol.
 355       cov_atol: Python `float`-type indicating the admissible absolute error
 356         between analytical and sample covariance. Default: atol.
 357     """
 358 
 359     x = dist.sample(num_samples, seed=seed)
 360     sample_mean = math_ops.reduce_mean(x, axis=0)
 361     sample_covariance = math_ops.reduce_mean(
 362         _vec_outer_square(x - sample_mean), axis=0)
 363     sample_variance = array_ops.matrix_diag_part(sample_covariance)
 364     sample_stddev = math_ops.sqrt(sample_variance)
 365 
 366     [
 367         sample_mean_,
 368         sample_covariance_,
 369         sample_variance_,
 370         sample_stddev_,
 371         mean_,
 372         covariance_,
 373         variance_,
 374         stddev_
 375     ] = sess_run_fn([
 376         sample_mean,
 377         sample_covariance,
 378         sample_variance,
 379         sample_stddev,
 380         dist.mean(),
 381         dist.covariance(),
 382         dist.variance(),
 383         dist.stddev(),
 384     ])
 385 
 386     self.assertAllClose(mean_, sample_mean_, rtol=rtol, atol=atol)
 387     self.assertAllClose(covariance_, sample_covariance_,
 388                         rtol=cov_rtol or rtol,
 389                         atol=cov_atol or atol)
 390     self.assertAllClose(variance_, sample_variance_, rtol=rtol, atol=atol)
 391     self.assertAllClose(stddev_, sample_stddev_, rtol=rtol, atol=atol)

 394 def _vec_outer_square(x, name=None):
 395   """Computes the outer-product of a vector, i.e., x.T x."""
 396   with ops.name_scope(name, "vec_osquare", [x]):
 397     return x[..., :, array_ops.newaxis] * x[..., array_ops.newaxis, :]
