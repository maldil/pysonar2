  35 class MultivariateNormalFullCovariance(mvn_tril.MultivariateNormalTriL):
  36   """The multivariate normal distribution on `R^k`.
  37 
  38   The Multivariate Normal distribution is defined over `R^k` and parameterized
  39   by a (batch of) length-`k` `loc` vector (aka "mu") and a (batch of) `k x k`
  40   `covariance_matrix` matrices that are the covariance.
  41   This is different than the other multivariate normals, which are parameterized
  42   by a matrix more akin to the standard deviation.
  43 
  44   #### Mathematical Details
  45 
  46   The probability density function (pdf) is, with `@` as matrix multiplication,
  47 
  48   ```none
  49   pdf(x; loc, covariance_matrix) = exp(-0.5 y) / Z,
  50   y = (x - loc)^T @ inv(covariance_matrix) @ (x - loc)
  51   Z = (2 pi)**(0.5 k) |det(covariance_matrix)|**(0.5).
  52   ```
  53 
  54   where:
  55 
  56   * `loc` is a vector in `R^k`,
  57   * `covariance_matrix` is an `R^{k x k}` symmetric positive definite matrix,
  58   * `Z` denotes the normalization constant.
  59 
  60   Additional leading dimensions (if any) in `loc` and `covariance_matrix` allow
  61   for batch dimensions.
  62 
  63   The MultivariateNormal distribution is a member of the [location-scale
  64   family](https://en.wikipedia.org/wiki/Location-scale_family), i.e., it can be
  65   constructed e.g. as,
  66 
  67   ```none
  68   X ~ MultivariateNormal(loc=0, scale=1)   # Identity scale, zero shift.
  69   scale = Cholesky(covariance_matrix)
  70   Y = scale @ X + loc
  71   ```
  72 
  73   #### Examples
  74 
  75   ```python
  76   import tensorflow_probability as tfp
  77   tfd = tfp.distributions
  78 
  79   # Initialize a single 3-variate Gaussian.
  80   mu = [1., 2, 3]
  81   cov = [[ 0.36,  0.12,  0.06],
  82          [ 0.12,  0.29, -0.13],
  83          [ 0.06, -0.13,  0.26]]
  84   mvn = tfd.MultivariateNormalFullCovariance(
  85       loc=mu,
  86       covariance_matrix=cov)
  87 
  88   mvn.mean().eval()
  89   # ==> [1., 2, 3]
  90 
  91   # Covariance agrees with covariance_matrix.
  92   mvn.covariance().eval()
  93   # ==> [[ 0.36,  0.12,  0.06],
  94   #      [ 0.12,  0.29, -0.13],
  95   #      [ 0.06, -0.13,  0.26]]
  96 
  97   # Compute the pdf of an observation in `R^3` ; return a scalar.
  98   mvn.prob([-1., 0, 1]).eval()  # shape: []
  99 
 100   # Initialize a 2-batch of 3-variate Gaussians.
 101   mu = [[1., 2, 3],
 102         [11, 22, 33]]              # shape: [2, 3]
 103   covariance_matrix = ...  # shape: [2, 3, 3], symmetric, positive definite.
 104   mvn = tfd.MultivariateNormalFullCovariance(
 105       loc=mu,
 106       covariance=covariance_matrix)
 107 
 108   # Compute the pdf of two `R^3` observations; return a length-2 vector.
 109   x = [[-0.9, 0, 0.1],
 110        [-10, 0, 9]]     # shape: [2, 3]
 111   mvn.prob(x).eval()    # shape: [2]
 112 
 113   ```
 114 
 115   """
 116 
 117   @deprecation.deprecated(
 118       "2018-10-01",
 119       "The TensorFlow Distributions library has moved to "
 120       "TensorFlow Probability "
 121       "(https://github.com/tensorflow/probability). You "
 122       "should update all references to use `tfp.distributions` "
 123       "instead of `tf.contrib.distributions`.",
 124       warn_once=True)
 125   def __init__(self,
 126                loc=None,
 127                covariance_matrix=None,
 128                validate_args=False,
 129                allow_nan_stats=True,
 130                name="MultivariateNormalFullCovariance"):
 131     """Construct Multivariate Normal distribution on `R^k`.
 132 
 133     The `batch_shape` is the broadcast shape between `loc` and
 134     `covariance_matrix` arguments.
 135 
 136     The `event_shape` is given by last dimension of the matrix implied by
 137     `covariance_matrix`. The last dimension of `loc` (if provided) must
 138     broadcast with this.
 139 
 140     A non-batch `covariance_matrix` matrix is a `k x k` symmetric positive
 141     definite matrix.  In other words it is (real) symmetric with all eigenvalues
 142     strictly positive.
 143 
 144     Additional leading dimensions (if any) will index batches.
 145 
 146     Args:
 147       loc: Floating-point `Tensor`. If this is set to `None`, `loc` is
 148         implicitly `0`. When specified, may have shape `[B1, ..., Bb, k]` where
 149         `b >= 0` and `k` is the event size.
 150       covariance_matrix: Floating-point, symmetric positive definite `Tensor` of
 151         same `dtype` as `loc`.  The strict upper triangle of `covariance_matrix`
 152         is ignored, so if `covariance_matrix` is not symmetric no error will be
 153         raised (unless `validate_args is True`).  `covariance_matrix` has shape
 154         `[B1, ..., Bb, k, k]` where `b >= 0` and `k` is the event size.
 155       validate_args: Python `bool`, default `False`. When `True` distribution
 156         parameters are checked for validity despite possibly degrading runtime
 157         performance. When `False` invalid inputs may silently render incorrect
 158         outputs.
 159       allow_nan_stats: Python `bool`, default `True`. When `True`,
 160         statistics (e.g., mean, mode, variance) use the value "`NaN`" to
 161         indicate the result is undefined. When `False`, an exception is raised
 162         if one or more of the statistic's batch members are undefined.
 163       name: Python `str` name prefixed to Ops created by this class.
 164 
 165     Raises:
 166       ValueError: if neither `loc` nor `covariance_matrix` are specified.
 167     """
 168     parameters = dict(locals())
 169 
 170     # Convert the covariance_matrix up to a scale_tril and call MVNTriL.
 171     with ops.name_scope(name) as name:
 172       with ops.name_scope("init", values=[loc, covariance_matrix]):
 173         if covariance_matrix is None:
 174           scale_tril = None
 175         else:
 176           covariance_matrix = ops.convert_to_tensor(
 177               covariance_matrix, name="covariance_matrix")
 178           if validate_args:
 179             covariance_matrix = control_flow_ops.with_dependencies([
 180                 check_ops.assert_near(
 181                     covariance_matrix,
 182                     array_ops.matrix_transpose(covariance_matrix),
 183                     message="Matrix was not symmetric")], covariance_matrix)
 184           # No need to validate that covariance_matrix is non-singular.
 185           # LinearOperatorLowerTriangular has an assert_non_singular method that
 186           # is called by the Bijector.
 187           # However, cholesky() ignores the upper triangular part, so we do need
 188           # to separately assert symmetric.
 189           scale_tril = linalg_ops.cholesky(covariance_matrix)
 190         super(MultivariateNormalFullCovariance, self).__init__(
 191             loc=loc,
 192             scale_tril=scale_tril,
 193             validate_args=validate_args,
 194             allow_nan_stats=allow_nan_stats,
 195             name=name)
 196     self._parameters = parameters
