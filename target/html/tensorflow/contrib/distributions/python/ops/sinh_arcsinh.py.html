  35 class SinhArcsinh(transformed_distribution.TransformedDistribution):
  36   """The SinhArcsinh transformation of a distribution on `(-inf, inf)`.
  37 
  38   This distribution models a random variable, making use of
  39   a `SinhArcsinh` transformation (which has adjustable tailweight and skew),
  40   a rescaling, and a shift.
  41 
  42   The `SinhArcsinh` transformation of the Normal is described in great depth in
  43   [Sinh-arcsinh distributions](https://www.jstor.org/stable/27798865).
  44   Here we use a slightly different parameterization, in terms of `tailweight`
  45   and `skewness`.  Additionally we allow for distributions other than Normal,
  46   and control over `scale` as well as a "shift" parameter `loc`.
  47 
  48   #### Mathematical Details
  49 
  50   Given random variable `Z`, we define the SinhArcsinh
  51   transformation of `Z`, `Y`, parameterized by
  52   `(loc, scale, skewness, tailweight)`, via the relation:
  53 
  54   ```
  55   Y := loc + scale * F(Z) * (2 / F_0(2))
  56   F(Z) := Sinh( (Arcsinh(Z) + skewness) * tailweight )
  57   F_0(Z) := Sinh( Arcsinh(Z) * tailweight )
  58   ```
  59 
  60   This distribution is similar to the location-scale transformation
  61   `L(Z) := loc + scale * Z` in the following ways:
  62 
  63   * If `skewness = 0` and `tailweight = 1` (the defaults), `F(Z) = Z`, and then
  64     `Y = L(Z)` exactly.
  65   * `loc` is used in both to shift the result by a constant factor.
  66   * The multiplication of `scale` by `2 / F_0(2)` ensures that if `skewness = 0`
  67     `P[Y - loc <= 2 * scale] = P[L(Z) - loc <= 2 * scale]`.
  68     Thus it can be said that the weights in the tails of `Y` and `L(Z)` beyond
  69     `loc + 2 * scale` are the same.
  70 
  71   This distribution is different than `loc + scale * Z` due to the
  72   reshaping done by `F`:
  73 
  74   * Positive (negative) `skewness` leads to positive (negative) skew.
  75     * positive skew means, the mode of `F(Z)` is "tilted" to the right.
  76     * positive skew means positive values of `F(Z)` become more likely, and
  77       negative values become less likely.
  78   * Larger (smaller) `tailweight` leads to fatter (thinner) tails.
  79     * Fatter tails mean larger values of `|F(Z)|` become more likely.
  80     * `tailweight < 1` leads to a distribution that is "flat" around `Y = loc`,
  81       and a very steep drop-off in the tails.
  82     * `tailweight > 1` leads to a distribution more peaked at the mode with
  83       heavier tails.
  84 
  85   To see the argument about the tails, note that for `|Z| >> 1` and
  86   `|Z| >> (|skewness| * tailweight)**tailweight`, we have
  87   `Y approx 0.5 Z**tailweight e**(sign(Z) skewness * tailweight)`.
  88 
  89   To see the argument regarding multiplying `scale` by `2 / F_0(2)`,
  90 
  91   ```
  92   P[(Y - loc) / scale <= 2] = P[F(Z) * (2 / F_0(2)) <= 2]
  93                             = P[F(Z) <= F_0(2)]
  94                             = P[Z <= 2]  (if F = F_0).
  95   ```
  96   """
  97 
  98   @deprecation.deprecated(
  99       "2018-10-01",
 100       "The TensorFlow Distributions library has moved to "
 101       "TensorFlow Probability "
 102       "(https://github.com/tensorflow/probability). You "
 103       "should update all references to use `tfp.distributions` "
 104       "instead of `tf.contrib.distributions`.",
 105       warn_once=True)
 106   def __init__(self,
 107                loc,
 108                scale,
 109                skewness=None,
 110                tailweight=None,
 111                distribution=None,
 112                validate_args=False,
 113                allow_nan_stats=True,
 114                name="SinhArcsinh"):
 115     """Construct SinhArcsinh distribution on `(-inf, inf)`.
 116 
 117     Arguments `(loc, scale, skewness, tailweight)` must have broadcastable shape
 118     (indexing batch dimensions).  They must all have the same `dtype`.
 119 
 120     Args:
 121       loc: Floating-point `Tensor`.
 122       scale:  `Tensor` of same `dtype` as `loc`.
 123       skewness:  Skewness parameter.  Default is `0.0` (no skew).
 124       tailweight:  Tailweight parameter. Default is `1.0` (unchanged tailweight)
 125       distribution: `tf.Distribution`-like instance. Distribution that is
 126         transformed to produce this distribution.
 127         Default is `tfp.distributions.Normal(0., 1.)`.
 128         Must be a scalar-batch, scalar-event distribution.  Typically
 129         `distribution.reparameterization_type = FULLY_REPARAMETERIZED` or it is
 130         a function of non-trainable parameters. WARNING: If you backprop through
 131         a `SinhArcsinh` sample and `distribution` is not
 132         `FULLY_REPARAMETERIZED` yet is a function of trainable variables, then
 133         the gradient will be incorrect!
 134       validate_args: Python `bool`, default `False`. When `True` distribution
 135         parameters are checked for validity despite possibly degrading runtime
 136         performance. When `False` invalid inputs may silently render incorrect
 137         outputs.
 138       allow_nan_stats: Python `bool`, default `True`. When `True`,
 139         statistics (e.g., mean, mode, variance) use the value "`NaN`" to
 140         indicate the result is undefined. When `False`, an exception is raised
 141         if one or more of the statistic's batch members are undefined.
 142       name: Python `str` name prefixed to Ops created by this class.
 143     """
 144     parameters = dict(locals())
 145 
 146     with ops.name_scope(name,
 147                         values=[loc, scale, skewness, tailweight]) as name:
 148       loc = ops.convert_to_tensor(loc, name="loc")
 149       dtype = loc.dtype
 150       scale = ops.convert_to_tensor(scale, name="scale", dtype=dtype)
 151       tailweight = 1. if tailweight is None else tailweight
 152       has_default_skewness = skewness is None
 153       skewness = 0. if skewness is None else skewness
 154       tailweight = ops.convert_to_tensor(
 155           tailweight, name="tailweight", dtype=dtype)
 156       skewness = ops.convert_to_tensor(skewness, name="skewness", dtype=dtype)
 157 
 158       batch_shape = distribution_util.get_broadcast_shape(
 159           loc, scale, tailweight, skewness)
 160 
 161       # Recall, with Z a random variable,
 162       #   Y := loc + C * F(Z),
 163       #   F(Z) := Sinh( (Arcsinh(Z) + skewness) * tailweight )
 164       #   F_0(Z) := Sinh( Arcsinh(Z) * tailweight )
 165       #   C := 2 * scale / F_0(2)
 166       if distribution is None:
 167         distribution = normal.Normal(
 168             loc=array_ops.zeros([], dtype=dtype),
 169             scale=array_ops.ones([], dtype=dtype),
 170             allow_nan_stats=allow_nan_stats)
 171       else:
 172         asserts = distribution_util.maybe_check_scalar_distribution(
 173             distribution, dtype, validate_args)
 174         if asserts:
 175           loc = control_flow_ops.with_dependencies(asserts, loc)
 176 
 177       # Make the SAS bijector, 'F'.
 178       f = bijectors.SinhArcsinh(
 179           skewness=skewness, tailweight=tailweight)
 180       if has_default_skewness:
 181         f_noskew = f
 182       else:
 183         f_noskew = bijectors.SinhArcsinh(
 184             skewness=skewness.dtype.as_numpy_dtype(0.),
 185             tailweight=tailweight)
 186 
 187       # Make the AffineScalar bijector, Z --> loc + scale * Z (2 / F_0(2))
 188       c = 2 * scale / f_noskew.forward(ops.convert_to_tensor(2, dtype=dtype))
 189       affine = bijectors.AffineScalar(
 190           shift=loc,
 191           scale=c,
 192           validate_args=validate_args)
 193 
 194       bijector = bijectors.Chain([affine, f])
 195 
 196       super(SinhArcsinh, self).__init__(
 197           distribution=distribution,
 198           bijector=bijector,
 199           batch_shape=batch_shape,
 200           validate_args=validate_args,
 201           name=name)
 202     self._parameters = parameters
 203     self._loc = loc
 204     self._scale = scale
 205     self._tailweight = tailweight
 206     self._skewness = skewness
 207 
 208   @property
 209   def loc(self):
 210     """The `loc` in `Y := loc + scale @ F(Z) * (2 / F(2))."""
 211     return self._loc
 212 
 213   @property
 214   def scale(self):
 215     """The `LinearOperator` `scale` in `Y := loc + scale @ F(Z) * (2 / F(2))."""
 216     return self._scale
 217 
 218   @property
 219   def tailweight(self):
 220     """Controls the tail decay.  `tailweight > 1` means faster than Normal."""
 221     return self._tailweight
 222 
 223   @property
 224   def skewness(self):
 225     """Controls the skewness.  `Skewness > 0` means right skew."""
 226     return self._skewness
