  37 class Geometric(distribution.Distribution):
  38   """Geometric distribution.
  39 
  40   The Geometric distribution is parameterized by p, the probability of a
  41   positive event. It represents the probability that in k + 1 Bernoulli trials,
  42   the first k trials failed, before seeing a success.
  43 
  44   The pmf of this distribution is:
  45 
  46   #### Mathematical Details
  47 
  48   ```none
  49   pmf(k; p) = (1 - p)**k * p
  50   ```
  51 
  52   where:
  53 
  54   * `p` is the success probability, `0 < p <= 1`, and,
  55   * `k` is a non-negative integer.
  56 
  57   """
  58 
  59   @deprecation.deprecated(
  60       "2018-10-01",
  61       "The TensorFlow Distributions library has moved to "
  62       "TensorFlow Probability "
  63       "(https://github.com/tensorflow/probability). You "
  64       "should update all references to use `tfp.distributions` "
  65       "instead of `tf.contrib.distributions`.",
  66       warn_once=True)
  67   def __init__(self,
  68                logits=None,
  69                probs=None,
  70                validate_args=False,
  71                allow_nan_stats=True,
  72                name="Geometric"):
  73     """Construct Geometric distributions.
  74 
  75     Args:
  76       logits: Floating-point `Tensor` with shape `[B1, ..., Bb]` where `b >= 0`
  77         indicates the number of batch dimensions. Each entry represents logits
  78         for the probability of success for independent Geometric distributions
  79         and must be in the range `(-inf, inf]`. Only one of `logits` or `probs`
  80         should be specified.
  81       probs: Positive floating-point `Tensor` with shape `[B1, ..., Bb]`
  82         where `b >= 0` indicates the number of batch dimensions. Each entry
  83         represents the probability of success for independent Geometric
  84         distributions and must be in the range `(0, 1]`. Only one of `logits`
  85         or `probs` should be specified.
  86       validate_args: Python `bool`, default `False`. When `True` distribution
  87         parameters are checked for validity despite possibly degrading runtime
  88         performance. When `False` invalid inputs may silently render incorrect
  89         outputs.
  90       allow_nan_stats: Python `bool`, default `True`. When `True`, statistics
  91         (e.g., mean, mode, variance) use the value "`NaN`" to indicate the
  92         result is undefined. When `False`, an exception is raised if one or
  93         more of the statistic's batch members are undefined.
  94       name: Python `str` name prefixed to Ops created by this class.
  95     """
  96 
  97     parameters = dict(locals())
  98     with ops.name_scope(name, values=[logits, probs]) as name:
  99       self._logits, self._probs = distribution_util.get_logits_and_probs(
 100           logits, probs, validate_args=validate_args, name=name)
 101 
 102       with ops.control_dependencies(
 103           [check_ops.assert_positive(self._probs)] if validate_args else []):
 104         self._probs = array_ops.identity(self._probs, name="probs")
 105 
 106     super(Geometric, self).__init__(
 107         dtype=self._probs.dtype,
 108         reparameterization_type=distribution.NOT_REPARAMETERIZED,
 109         validate_args=validate_args,
 110         allow_nan_stats=allow_nan_stats,
 111         parameters=parameters,
 112         graph_parents=[self._probs, self._logits],
 113         name=name)
 114 
 115   @property
 116   def logits(self):
 117     """Log-odds of a `1` outcome (vs `0`)."""
 118     return self._logits
 119 
 120   @property
 121   def probs(self):
 122     """Probability of a `1` outcome (vs `0`)."""
 123     return self._probs
 124 
 125   def _batch_shape_tensor(self):
 126     return array_ops.shape(self._probs)
 127 
 128   def _batch_shape(self):
 129     return self.probs.get_shape()
 130 
 131   def _event_shape_tensor(self):
 132     return array_ops.constant([], dtype=dtypes.int32)
 133 
 134   def _event_shape(self):
 135     return tensor_shape.scalar()
 136 
 137   def _sample_n(self, n, seed=None):
 138     # Uniform variates must be sampled from the open-interval `(0, 1)` rather
 139     # than `[0, 1)`. To do so, we use `np.finfo(self.dtype.as_numpy_dtype).tiny`
 140     # because it is the smallest, positive, "normal" number. A "normal" number
 141     # is such that the mantissa has an implicit leading 1. Normal, positive
 142     # numbers x, y have the reasonable property that, `x + y >= max(x, y)`. In
 143     # this case, a subnormal number (i.e., np.nextafter) can cause us to sample
 144     # 0.
 145     sampled = random_ops.random_uniform(
 146         array_ops.concat([[n], array_ops.shape(self._probs)], 0),
 147         minval=np.finfo(self.dtype.as_numpy_dtype).tiny,
 148         maxval=1.,
 149         seed=seed,
 150         dtype=self.dtype)
 151 
 152     return math_ops.floor(
 153         math_ops.log(sampled) / math_ops.log1p(-self.probs))
 154 
 155   def _cdf(self, x):
 156     if self.validate_args:
 157       x = distribution_util.embed_check_nonnegative_integer_form(x)
 158     else:
 159       # Whether or not x is integer-form, the following is well-defined.
 160       # However, scipy takes the floor, so we do too.
 161       x = math_ops.floor(x)
 162     x *= array_ops.ones_like(self.probs)
 163     return array_ops.where(
 164         x < 0.,
 165         array_ops.zeros_like(x),
 166         -math_ops.expm1((1. + x) * math_ops.log1p(-self.probs)))
 167 
 168   def _log_prob(self, x):
 169     if self.validate_args:
 170       x = distribution_util.embed_check_nonnegative_integer_form(x)
 171     else:
 172       # For consistency with cdf, we take the floor.
 173       x = math_ops.floor(x)
 174     x *= array_ops.ones_like(self.probs)
 175     probs = self.probs * array_ops.ones_like(x)
 176     safe_domain = array_ops.where(
 177         math_ops.equal(x, 0.),
 178         array_ops.zeros_like(probs),
 179         probs)
 180     return x * math_ops.log1p(-safe_domain) + math_ops.log(probs)
 181 
 182   def _entropy(self):
 183     probs = self._probs
 184     if self.validate_args:
 185       probs = control_flow_ops.with_dependencies(
 186           [check_ops.assert_less(
 187               probs,
 188               constant_op.constant(1., probs.dtype),
 189               message="Entropy is undefined when logits = inf or probs = 1.")],
 190           probs)
 191     # Claim: entropy(p) = softplus(s)/p - s
 192     # where s=logits and p=probs.
 193     #
 194     # Proof:
 195     #
 196     # entropy(p)
 197     # := -[(1-p)log(1-p) + plog(p)]/p
 198     # = -[log(1-p) + plog(p/(1-p))]/p
 199     # = -[-softplus(s) + ps]/p
 200     # = softplus(s)/p - s
 201     #
 202     # since,
 203     # log[1-sigmoid(s)]
 204     # = log[1/(1+exp(s)]
 205     # = -log[1+exp(s)]
 206     # = -softplus(s)
 207     #
 208     # using the fact that,
 209     # 1-sigmoid(s) = sigmoid(-s) = 1/(1+exp(s))
 210     return nn.softplus(self.logits) / probs - self.logits
 211 
 212   def _mean(self):
 213     return math_ops.exp(-self.logits)
 214 
 215   def _variance(self):
 216     return self._mean() / self.probs
 217 
 218   def _mode(self):
 219     return array_ops.zeros(self.batch_shape_tensor(), dtype=self.dtype)
