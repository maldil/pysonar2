  25 class ConditionalDistribution(distribution.Distribution):
  26   """Distribution that supports intrinsic parameters (local latents).
  27 
  28   Subclasses of this distribution may have additional keyword arguments passed
  29   to their sample-based methods (i.e. `sample`, `log_prob`, etc.).
  30   """
  31 
  32   @distribution_util.AppendDocstring(kwargs_dict={
  33       "**condition_kwargs":
  34       "Named arguments forwarded to subclass implementation."})
  35   def sample(self, sample_shape=(), seed=None, name="sample",
  36              **condition_kwargs):
  37     return self._call_sample_n(sample_shape, seed, name, **condition_kwargs)
  38 
  39   @distribution_util.AppendDocstring(kwargs_dict={
  40       "**condition_kwargs":
  41       "Named arguments forwarded to subclass implementation."})
  42   def log_prob(self, value, name="log_prob", **condition_kwargs):
  43     return self._call_log_prob(value, name, **condition_kwargs)
