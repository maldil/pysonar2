  32 class VectorLaplaceDiag(
  33     vector_laplace_linop.VectorLaplaceLinearOperator):
  34   """The vectorization of the Laplace distribution on `R^k`.
  35 
  36   The vector laplace distribution is defined over `R^k`, and parameterized by
  37   a (batch of) length-`k` `loc` vector (the means) and a (batch of) `k x k`
  38   `scale` matrix:  `covariance = 2 * scale @ scale.T`, where `@` denotes
  39   matrix-multiplication.
  40 
  41   #### Mathematical Details
  42 
  43   The probability density function (pdf) is,
  44 
  45   ```none
  46   pdf(x; loc, scale) = exp(-||y||_1) / Z,
  47   y = inv(scale) @ (x - loc),
  48   Z = 2**k |det(scale)|,
  49   ```
  50 
  51   where:
  52 
  53   * `loc` is a vector in `R^k`,
  54   * `scale` is a linear operator in `R^{k x k}`, `cov = scale @ scale.T`,
  55   * `Z` denotes the normalization constant, and,
  56   * `||y||_1` denotes the `l1` norm of `y`, `sum_i |y_i|.
  57 
  58   A (non-batch) `scale` matrix is:
  59 
  60   ```none
  61   scale = diag(scale_diag + scale_identity_multiplier * ones(k))
  62   ```
  63 
  64   where:
  65 
  66   * `scale_diag.shape = [k]`, and,
  67   * `scale_identity_multiplier.shape = []`.
  68 
  69   Additional leading dimensions (if any) will index batches.
  70 
  71   If both `scale_diag` and `scale_identity_multiplier` are `None`, then
  72   `scale` is the Identity matrix.
  73 
  74   The VectorLaplace distribution is a member of the [location-scale
  75   family](https://en.wikipedia.org/wiki/Location-scale_family), i.e., it can be
  76   constructed as,
  77 
  78   ```none
  79   X = (X_1, ..., X_k), each X_i ~ Laplace(loc=0, scale=1)
  80   Y = (Y_1, ...,Y_k) = scale @ X + loc
  81   ```
  82 
  83   #### About `VectorLaplace` and `Vector` distributions in TensorFlow.
  84 
  85   The `VectorLaplace` is a non-standard distribution that has useful properties.
  86 
  87   The marginals `Y_1, ..., Y_k` are *not* Laplace random variables, due to
  88   the fact that the sum of Laplace random variables is not Laplace.
  89 
  90   Instead, `Y` is a vector whose components are linear combinations of Laplace
  91   random variables.  Thus, `Y` lives in the vector space generated by `vectors`
  92   of Laplace distributions.  This allows the user to decide the mean and
  93   covariance (by setting `loc` and `scale`), while preserving some properties of
  94   the Laplace distribution.  In particular, the tails of `Y_i` will be (up to
  95   polynomial factors) exponentially decaying.
  96 
  97   To see this last statement, note that the pdf of `Y_i` is the convolution of
  98   the pdf of `k` independent Laplace random variables.  One can then show by
  99   induction that distributions with exponential (up to polynomial factors) tails
 100   are closed under convolution.
 101 
 102   #### Examples
 103 
 104   ```python
 105   import tensorflow_probability as tfp
 106   tfd = tfp.distributions
 107 
 108   # Initialize a single 2-variate VectorLaplace.
 109   vla = tfd.VectorLaplaceDiag(
 110       loc=[1., -1],
 111       scale_diag=[1, 2.])
 112 
 113   vla.mean().eval()
 114   # ==> [1., -1]
 115 
 116   vla.stddev().eval()
 117   # ==> [1., 2] * sqrt(2)
 118 
 119   # Evaluate this on an observation in `R^2`, returning a scalar.
 120   vla.prob([-1., 0]).eval()  # shape: []
 121 
 122   # Initialize a 3-batch, 2-variate scaled-identity VectorLaplace.
 123   vla = tfd.VectorLaplaceDiag(
 124       loc=[1., -1],
 125       scale_identity_multiplier=[1, 2., 3])
 126 
 127   vla.mean().eval()  # shape: [3, 2]
 128   # ==> [[1., -1]
 129   #      [1, -1],
 130   #      [1, -1]]
 131 
 132   vla.stddev().eval()  # shape: [3, 2]
 133   # ==> sqrt(2) * [[1., 1],
 134   #                [2, 2],
 135   #                [3, 3]]
 136 
 137   # Evaluate this on an observation in `R^2`, returning a length-3 vector.
 138   vla.prob([-1., 0]).eval()  # shape: [3]
 139 
 140   # Initialize a 2-batch of 3-variate VectorLaplace's.
 141   vla = tfd.VectorLaplaceDiag(
 142       loc=[[1., 2, 3],
 143            [11, 22, 33]]           # shape: [2, 3]
 144       scale_diag=[[1., 2, 3],
 145                   [0.5, 1, 1.5]])  # shape: [2, 3]
 146 
 147   # Evaluate this on a two observations, each in `R^3`, returning a length-2
 148   # vector.
 149   x = [[-1., 0, 1],
 150        [-11, 0, 11.]]   # shape: [2, 3].
 151   vla.prob(x).eval()    # shape: [2]
 152   ```
 153 
 154   """
 155 
 156   @deprecation.deprecated(
 157       "2018-10-01",
 158       "The TensorFlow Distributions library has moved to "
 159       "TensorFlow Probability "
 160       "(https://github.com/tensorflow/probability). You "
 161       "should update all references to use `tfp.distributions` "
 162       "instead of `tf.contrib.distributions`.",
 163       warn_once=True)
 164   def __init__(self,
 165                loc=None,
 166                scale_diag=None,
 167                scale_identity_multiplier=None,
 168                validate_args=False,
 169                allow_nan_stats=True,
 170                name="VectorLaplaceDiag"):
 171     """Construct Vector Laplace distribution on `R^k`.
 172 
 173     The `batch_shape` is the broadcast shape between `loc` and `scale`
 174     arguments.
 175 
 176     The `event_shape` is given by last dimension of the matrix implied by
 177     `scale`. The last dimension of `loc` (if provided) must broadcast with this.
 178 
 179     Recall that `covariance = 2 * scale @ scale.T`.
 180 
 181     ```none
 182     scale = diag(scale_diag + scale_identity_multiplier * ones(k))
 183     ```
 184 
 185     where:
 186 
 187     * `scale_diag.shape = [k]`, and,
 188     * `scale_identity_multiplier.shape = []`.
 189 
 190     Additional leading dimensions (if any) will index batches.
 191 
 192     If both `scale_diag` and `scale_identity_multiplier` are `None`, then
 193     `scale` is the Identity matrix.
 194 
 195     Args:
 196       loc: Floating-point `Tensor`. If this is set to `None`, `loc` is
 197         implicitly `0`. When specified, may have shape `[B1, ..., Bb, k]` where
 198         `b >= 0` and `k` is the event size.
 199       scale_diag: Non-zero, floating-point `Tensor` representing a diagonal
 200         matrix added to `scale`. May have shape `[B1, ..., Bb, k]`, `b >= 0`,
 201         and characterizes `b`-batches of `k x k` diagonal matrices added to
 202         `scale`. When both `scale_identity_multiplier` and `scale_diag` are
 203         `None` then `scale` is the `Identity`.
 204       scale_identity_multiplier: Non-zero, floating-point `Tensor` representing
 205         a scaled-identity-matrix added to `scale`. May have shape
 206         `[B1, ..., Bb]`, `b >= 0`, and characterizes `b`-batches of scaled
 207         `k x k` identity matrices added to `scale`. When both
 208         `scale_identity_multiplier` and `scale_diag` are `None` then `scale` is
 209         the `Identity`.
 210       validate_args: Python `bool`, default `False`. When `True` distribution
 211         parameters are checked for validity despite possibly degrading runtime
 212         performance. When `False` invalid inputs may silently render incorrect
 213         outputs.
 214       allow_nan_stats: Python `bool`, default `True`. When `True`,
 215         statistics (e.g., mean, mode, variance) use the value "`NaN`" to
 216         indicate the result is undefined. When `False`, an exception is raised
 217         if one or more of the statistic's batch members are undefined.
 218       name: Python `str` name prefixed to Ops created by this class.
 219 
 220     Raises:
 221       ValueError: if at most `scale_identity_multiplier` is specified.
 222     """
 223     parameters = dict(locals())
 224     with ops.name_scope(name):
 225       with ops.name_scope("init", values=[
 226           loc, scale_diag, scale_identity_multiplier]):
 227         # No need to validate_args while making diag_scale.  The returned
 228         # LinearOperatorDiag has an assert_non_singular method that is called by
 229         # the Bijector.
 230         scale = distribution_util.make_diag_scale(
 231             loc=loc,
 232             scale_diag=scale_diag,
 233             scale_identity_multiplier=scale_identity_multiplier,
 234             validate_args=False,
 235             assert_positive=False)
 236     super(VectorLaplaceDiag, self).__init__(
 237         loc=loc,
 238         scale=scale,
 239         validate_args=validate_args,
 240         allow_nan_stats=allow_nan_stats,
 241         name=name)
 242     self._parameters = parameters
