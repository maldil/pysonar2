  52 def _harmonic_number(x):
  53   """Compute the harmonic number from its analytic continuation.
  54 
  55   Derivation from [here](
  56   https://en.wikipedia.org/wiki/Digamma_function#Relation_to_harmonic_numbers)
  57   and [Euler's constant](
  58   https://en.wikipedia.org/wiki/Euler%E2%80%93Mascheroni_constant).
  59 
  60   Args:
  61     x: input float.
  62 
  63   Returns:
  64     z: The analytic continuation of the harmonic number for the input.
  65   """
  66   one = array_ops.ones([], dtype=x.dtype)
  67   return math_ops.digamma(x + one) - math_ops.digamma(one)

  70 class Kumaraswamy(transformed_distribution.TransformedDistribution):
  71   """Kumaraswamy distribution.
  72 
  73   The Kumaraswamy distribution is defined over the `(0, 1)` interval using
  74   parameters
  75   `concentration1` (aka "alpha") and `concentration0` (aka "beta").  It has a
  76   shape similar to the Beta distribution, but is reparameterizeable.
  77 
  78   #### Mathematical Details
  79 
  80   The probability density function (pdf) is,
  81 
  82   ```none
  83   pdf(x; alpha, beta) = alpha * beta * x**(alpha - 1) * (1 - x**alpha)**(beta -
  84   1)
  85   ```
  86 
  87   where:
  88 
  89   * `concentration1 = alpha`,
  90   * `concentration0 = beta`,
  91 
  92   Distribution parameters are automatically broadcast in all functions; see
  93   examples for details.
  94 
  95   #### Examples
  96 
  97   ```python
  98   # Create a batch of three Kumaraswamy distributions.
  99   alpha = [1, 2, 3]
 100   beta = [1, 2, 3]
 101   dist = Kumaraswamy(alpha, beta)
 102 
 103   dist.sample([4, 5])  # Shape [4, 5, 3]
 104 
 105   # `x` has three batch entries, each with two samples.
 106   x = [[.1, .4, .5],
 107        [.2, .3, .5]]
 108   # Calculate the probability of each pair of samples under the corresponding
 109   # distribution in `dist`.
 110   dist.prob(x)         # Shape [2, 3]
 111   ```
 112 
 113   ```python
 114   # Create batch_shape=[2, 3] via parameter broadcast:
 115   alpha = [[1.], [2]]      # Shape [2, 1]
 116   beta = [3., 4, 5]        # Shape [3]
 117   dist = Kumaraswamy(alpha, beta)
 118 
 119   # alpha broadcast as: [[1., 1, 1,],
 120   #                      [2, 2, 2]]
 121   # beta broadcast as:  [[3., 4, 5],
 122   #                      [3, 4, 5]]
 123   # batch_Shape [2, 3]
 124   dist.sample([4, 5])  # Shape [4, 5, 2, 3]
 125 
 126   x = [.2, .3, .5]
 127   # x will be broadcast as [[.2, .3, .5],
 128   #                         [.2, .3, .5]],
 129   # thus matching batch_shape [2, 3].
 130   dist.prob(x)         # Shape [2, 3]
 131   ```
 132 
 133   """
 134 
 135   @deprecation.deprecated(
 136       "2018-10-01",
 137       "The TensorFlow Distributions library has moved to "
 138       "TensorFlow Probability "
 139       "(https://github.com/tensorflow/probability). You "
 140       "should update all references to use `tfp.distributions` "
 141       "instead of `tf.contrib.distributions`.",
 142       warn_once=True)
 143   def __init__(self,
 144                concentration1=None,
 145                concentration0=None,
 146                validate_args=False,
 147                allow_nan_stats=True,
 148                name="Kumaraswamy"):
 149     """Initialize a batch of Kumaraswamy distributions.
 150 
 151     Args:
 152       concentration1: Positive floating-point `Tensor` indicating mean
 153         number of successes; aka "alpha". Implies `self.dtype` and
 154         `self.batch_shape`, i.e.,
 155         `concentration1.shape = [N1, N2, ..., Nm] = self.batch_shape`.
 156       concentration0: Positive floating-point `Tensor` indicating mean
 157         number of failures; aka "beta". Otherwise has same semantics as
 158         `concentration1`.
 159       validate_args: Python `bool`, default `False`. When `True` distribution
 160         parameters are checked for validity despite possibly degrading runtime
 161         performance. When `False` invalid inputs may silently render incorrect
 162         outputs.
 163       allow_nan_stats: Python `bool`, default `True`. When `True`, statistics
 164         (e.g., mean, mode, variance) use the value "`NaN`" to indicate the
 165         result is undefined. When `False`, an exception is raised if one or
 166         more of the statistic's batch members are undefined.
 167       name: Python `str` name prefixed to Ops created by this class.
 168     """
 169     with ops.name_scope(name, values=[concentration1, concentration0]) as name:
 170       concentration1 = ops.convert_to_tensor(
 171           concentration1, name="concentration1")
 172       concentration0 = ops.convert_to_tensor(
 173           concentration0, name="concentration0")
 174     super(Kumaraswamy, self).__init__(
 175         distribution=uniform.Uniform(
 176             low=array_ops.zeros([], dtype=concentration1.dtype),
 177             high=array_ops.ones([], dtype=concentration1.dtype),
 178             allow_nan_stats=allow_nan_stats),
 179         bijector=bijectors.Kumaraswamy(
 180             concentration1=concentration1, concentration0=concentration0,
 181             validate_args=validate_args),
 182         batch_shape=distribution_util.get_broadcast_shape(
 183             concentration1, concentration0),
 184         name=name)
 185     self._reparameterization_type = distribution.FULLY_REPARAMETERIZED
 186 
 187   @property
 188   def concentration1(self):
 189     """Concentration parameter associated with a `1` outcome."""
 190     return self.bijector.concentration1
 191 
 192   @property
 193   def concentration0(self):
 194     """Concentration parameter associated with a `0` outcome."""
 195     return self.bijector.concentration0
 196 
 197   def _entropy(self):
 198     a = self.concentration1
 199     b = self.concentration0
 200     return (1 - 1. / a) + (
 201         1 - 1. / b) * _harmonic_number(b) + math_ops.log(a) + math_ops.log(b)
 202 
 203   def _moment(self, n):
 204     """Compute the n'th (uncentered) moment."""
 205     total_concentration = self.concentration1 + self.concentration0
 206     expanded_concentration1 = array_ops.ones_like(
 207         total_concentration, dtype=self.dtype) * self.concentration1
 208     expanded_concentration0 = array_ops.ones_like(
 209         total_concentration, dtype=self.dtype) * self.concentration0
 210     beta_arg0 = 1 + n / expanded_concentration1
 211     beta_arg = array_ops.stack([beta_arg0, expanded_concentration0], -1)
 212     log_moment = math_ops.log(expanded_concentration0) + special_math_ops.lbeta(
 213         beta_arg)
 214     return math_ops.exp(log_moment)
 215 
 216   def _mean(self):
 217     return self._moment(1)
 218 
 219   def _variance(self):
 220     # TODO(b/72696533): Investigate a more numerically stable version.
 221     return self._moment(2) - math_ops.square(self._moment(1))
 222 
 223   @distribution_util.AppendDocstring(
 224       """Note: The mode is undefined when `concentration1 <= 1` or
 225       `concentration0 <= 1`. If `self.allow_nan_stats` is `True`, `NaN`
 226       is used for undefined modes. If `self.allow_nan_stats` is `False` an
 227       exception is raised when one or more modes are undefined.""")
 228   def _mode(self):
 229     a = self.concentration1
 230     b = self.concentration0
 231     mode = ((a - 1) / (a * b - 1))**(1. / a)
 232     if self.allow_nan_stats:
 233       nan = array_ops.fill(
 234           self.batch_shape_tensor(),
 235           np.array(np.nan, dtype=self.dtype.as_numpy_dtype),
 236           name="nan")
 237       is_defined = (self.concentration1 > 1.) & (self.concentration0 > 1.)
 238       return array_ops.where(is_defined, mode, nan)
 239 
 240     return control_flow_ops.with_dependencies([
 241         check_ops.assert_less(
 242             array_ops.ones([], dtype=self.concentration1.dtype),
 243             self.concentration1,
 244             message="Mode undefined for concentration1 <= 1."),
 245         check_ops.assert_less(
 246             array_ops.ones([], dtype=self.concentration0.dtype),
 247             self.concentration0,
 248             message="Mode undefined for concentration0 <= 1.")
 249     ], mode)
