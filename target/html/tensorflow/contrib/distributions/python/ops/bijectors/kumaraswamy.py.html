  34 class Kumaraswamy(bijector.Bijector):
  35   """Compute `Y = g(X) = (1 - (1 - X)**(1 / b))**(1 / a), X in [0, 1]`.
  36 
  37   This bijector maps inputs from `[0, 1]` to [0, 1]`. The inverse of the
  38   bijector applied to a uniform random variable `X ~ U(0, 1) gives back a
  39   random variable with the [Kumaraswamy distribution](
  40   https://en.wikipedia.org/wiki/Kumaraswamy_distribution):
  41 
  42   ```none
  43   Y ~ Kumaraswamy(a, b)
  44   pdf(y; a, b, 0 <= y <= 1) = a * b * y ** (a - 1) * (1 - y**a) ** (b - 1)
  45   ```
  46   """
  47 
  48   @deprecation.deprecated(
  49       "2018-10-01",
  50       "The TensorFlow Distributions library has moved to "
  51       "TensorFlow Probability "
  52       "(https://github.com/tensorflow/probability). You "
  53       "should update all references to use `tfp.distributions` "
  54       "instead of `tf.contrib.distributions`.",
  55       warn_once=True)
  56   def __init__(self,
  57                concentration1=None,
  58                concentration0=None,
  59                validate_args=False,
  60                name="kumaraswamy"):
  61     """Instantiates the `Kumaraswamy` bijector.
  62 
  63     Args:
  64       concentration1: Python `float` scalar indicating the transform power,
  65         i.e., `Y = g(X) = (1 - (1 - X)**(1 / b))**(1 / a)` where `a` is
  66         `concentration1`.
  67       concentration0: Python `float` scalar indicating the transform power,
  68         i.e., `Y = g(X) = (1 - (1 - X)**(1 / b))**(1 / a)` where `b` is
  69         `concentration0`.
  70       validate_args: Python `bool` indicating whether arguments should be
  71         checked for correctness.
  72       name: Python `str` name given to ops managed by this object.
  73     """
  74     self._graph_parents = []
  75     self._name = name
  76     self._validate_args = validate_args
  77 
  78     with self._name_scope("init", values=[concentration1, concentration0]):
  79       concentration1 = self._maybe_assert_valid_concentration(
  80           ops.convert_to_tensor(concentration1, name="concentration1"),
  81           validate_args=validate_args)
  82       concentration0 = self._maybe_assert_valid_concentration(
  83           ops.convert_to_tensor(concentration0, name="concentration0"),
  84           validate_args=validate_args)
  85 
  86     self._concentration1 = concentration1
  87     self._concentration0 = concentration0
  88     super(Kumaraswamy, self).__init__(
  89         forward_min_event_ndims=0,
  90         validate_args=validate_args,
  91         name=name)
  92 
  93   @property
  94   def concentration1(self):
  95     """The `a` in: `Y = g(X) = (1 - (1 - X)**(1 / b))**(1 / a)`."""
  96     return self._concentration1
  97 
  98   @property
  99   def concentration0(self):
 100     """The `b` in: `Y = g(X) = (1 - (1 - X)**(1 / b))**(1 / a)`."""
 101     return self._concentration0
 102 
 103   def _forward(self, x):
 104     x = self._maybe_assert_valid(x)
 105     return math_ops.exp(
 106         math_ops.log1p(-math_ops.exp(math_ops.log1p(-x) / self.concentration0))
 107         / self.concentration1)
 108 
 109   def _inverse(self, y):
 110     y = self._maybe_assert_valid(y)
 111     return math_ops.exp(math_ops.log1p(
 112         -(1 - y**self.concentration1)**self.concentration0))
 113 
 114   def _inverse_log_det_jacobian(self, y):
 115     y = self._maybe_assert_valid(y)
 116     return (
 117         math_ops.log(self.concentration1) + math_ops.log(self.concentration0) +
 118         (self.concentration1 - 1) * math_ops.log(y) +
 119         (self.concentration0 - 1) * math_ops.log1p(-y**self.concentration1))
 120 
 121   def _maybe_assert_valid_concentration(self, concentration, validate_args):
 122     """Checks the validity of a concentration parameter."""
 123     if not validate_args:
 124       return concentration
 125     return control_flow_ops.with_dependencies([
 126         check_ops.assert_positive(
 127             concentration,
 128             message="Concentration parameter must be positive."),
 129     ], concentration)
 130 
 131   def _maybe_assert_valid(self, x):
 132     if not self.validate_args:
 133       return x
 134     return control_flow_ops.with_dependencies([
 135         check_ops.assert_non_negative(
 136             x,
 137             message="sample must be non-negative"),
 138         check_ops.assert_less_equal(
 139             x, array_ops.ones([], self.concentration0.dtype),
 140             message="sample must be no larger than `1`."),
 141     ], x)
