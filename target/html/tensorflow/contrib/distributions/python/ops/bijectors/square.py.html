  35 class Square(bijector.Bijector):
  36   """Compute `g(X) = X^2`; X is a positive real number.
  37 
  38   g is a bijection between the non-negative real numbers (R_+) and the
  39   non-negative real numbers.
  40 
  41   #### Examples
  42 
  43   ```python
  44   bijector.Square().forward(x=[[1., 0], [2, 1]])
  45   # Result: [[1., 0], [4, 1]], i.e., x^2
  46 
  47   bijector.Square().inverse(y=[[1., 4], [9, 1]])
  48   # Result: [[1., 2], [3, 1]], i.e., sqrt(y).
  49   ```
  50 
  51   """
  52 
  53   @deprecation.deprecated(
  54       "2018-10-01",
  55       "The TensorFlow Distributions library has moved to "
  56       "TensorFlow Probability "
  57       "(https://github.com/tensorflow/probability). You "
  58       "should update all references to use `tfp.distributions` "
  59       "instead of `tf.contrib.distributions`.",
  60       warn_once=True)
  61   def __init__(self, validate_args=False, name="square"):
  62     """Instantiates the `Square` bijector.
  63 
  64     Args:
  65       validate_args: Python `bool` indicating whether arguments should be
  66         checked for correctness.
  67       name: Python `str` name given to ops managed by this object.
  68     """
  69     self._name = name
  70     super(Square, self).__init__(
  71         forward_min_event_ndims=0,
  72         validate_args=validate_args,
  73         name=name)
  74 
  75   def _forward(self, x):
  76     x = self._maybe_assert_valid(x)
  77     return math_ops.square(x)
  78 
  79   def _inverse(self, y):
  80     y = self._maybe_assert_valid(y)
  81     return math_ops.sqrt(y)
  82 
  83   def _forward_log_det_jacobian(self, x):
  84     x = self._maybe_assert_valid(x)
  85     return np.log(2.) + math_ops.log(x)
  86 
  87   def _maybe_assert_valid(self, t):
  88     if not self.validate_args:
  89       return t
  90     is_valid = check_ops.assert_non_negative(
  91         t, message="All elements must be non-negative.")
  92     return control_flow_ops.with_dependencies([is_valid], t)
