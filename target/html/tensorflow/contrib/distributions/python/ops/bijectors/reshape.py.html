  64 class Reshape(bijector.Bijector):
  65   """Reshapes the `event_shape` of a `Tensor`.
  66 
  67   The semantics generally follow that of `tf.reshape()`, with
  68   a few differences:
  69 
  70   * The user must provide both the input and output shape, so that
  71     the transformation can be inverted. If an input shape is not
  72     specified, the default assumes a vector-shaped input, i.e.,
  73     event_shape_in = (-1,).
  74   * The `Reshape` bijector automatically broadcasts over the leftmost
  75     dimensions of its input (`sample_shape` and `batch_shape`); only
  76     the rightmost `event_ndims_in` dimensions are reshaped. The
  77     number of dimensions to reshape is inferred from the provided
  78     `event_shape_in` (`event_ndims_in = len(event_shape_in)`).
  79 
  80   Example usage:
  81   ```python
  82 
  83   import tensorflow_probability as tfp
  84   tfb = tfp.bijectors
  85 
  86   r = tfb.Reshape(event_shape_out=[1, -1])
  87 
  88   r.forward([3., 4.])    # shape [2]
  89   # ==> [[3., 4.]]       # shape [1, 2]
  90 
  91   r.forward([[1., 2.], [3., 4.]])  # shape [2, 2]
  92   # ==> [[[1., 2.]],
  93   #      [[3., 4.]]]   # shape [2, 1, 2]
  94 
  95   r.inverse([[3., 4.]])  # shape [1,2]
  96   # ==> [3., 4.]         # shape [2]
  97 
  98   r.forward_log_det_jacobian(any_value)
  99   # ==> 0.
 100 
 101   r.inverse_log_det_jacobian(any_value)
 102   # ==> 0.
 103   ```
 104 
 105   """
 106 
 107   @deprecation.deprecated(
 108       "2018-10-01",
 109       "The TensorFlow Distributions library has moved to "
 110       "TensorFlow Probability "
 111       "(https://github.com/tensorflow/probability). You "
 112       "should update all references to use `tfp.distributions` "
 113       "instead of `tf.contrib.distributions`.",
 114       warn_once=True)
 115   def __init__(self, event_shape_out, event_shape_in=(-1,),
 116                validate_args=False, name=None):
 117     """Creates a `Reshape` bijector.
 118 
 119     Args:
 120       event_shape_out: An `int`-like vector-shaped `Tensor`
 121         representing the event shape of the transformed output.
 122       event_shape_in: An optional `int`-like vector-shape `Tensor`
 123         representing the event shape of the input. This is required in
 124         order to define inverse operations; the default of (-1,)
 125         assumes a vector-shaped input.
 126       validate_args: Python `bool` indicating whether arguments should
 127         be checked for correctness.
 128       name: Python `str`, name given to ops managed by this object.
 129 
 130     Raises:
 131       TypeError: if either `event_shape_in` or `event_shape_out` has
 132         non-integer `dtype`.
 133       ValueError: if either of `event_shape_in` or `event_shape_out`
 134        has non-vector shape (`rank > 1`), or if their sizes do not
 135        match.
 136     """
 137     with ops.name_scope(name, "reshape",
 138                         values=[event_shape_out, event_shape_in]):
 139 
 140       event_shape_out = ops.convert_to_tensor(event_shape_out,
 141                                               name="event_shape_out",
 142                                               preferred_dtype=dtypes.int32)
 143       event_shape_in = ops.convert_to_tensor(event_shape_in,
 144                                              name="event_shape_in",
 145                                              preferred_dtype=dtypes.int32)
 146 
 147       assertions = []
 148       assertions.extend(self._maybe_check_valid_shape(
 149           event_shape_out, validate_args))
 150       assertions.extend(self._maybe_check_valid_shape(
 151           event_shape_in, validate_args))
 152 
 153       self._assertions = assertions
 154       self._event_shape_in = event_shape_in
 155       self._event_shape_out = event_shape_out
 156 
 157       super(Reshape, self).__init__(
 158           forward_min_event_ndims=0,
 159           is_constant_jacobian=True,
 160           validate_args=validate_args,
 161           name=name or "reshape")
 162 
 163   def _maybe_check_valid_shape(self, shape, validate_args):
 164     """Check that a shape Tensor is int-type and otherwise sane."""
 165     if not shape.dtype.is_integer:
 166       raise TypeError("{} dtype ({}) should be `int`-like.".format(
 167           shape, shape.dtype.name))
 168 
 169     assertions = []
 170 
 171     ndims = array_ops.rank(shape)
 172     ndims_ = tensor_util.constant_value(ndims)
 173     if ndims_ is not None and ndims_ > 1:
 174       raise ValueError("`{}` rank ({}) should be <= 1.".format(
 175           shape, ndims_))
 176     elif validate_args:
 177       assertions.append(check_ops.assert_less_equal(
 178           ndims, 1, message="`{}` rank should be <= 1.".format(shape)))
 179 
 180     shape_ = tensor_util.constant_value_as_shape(shape)
 181     if shape_.is_fully_defined():
 182       es = np.int32(shape_.as_list())
 183       if sum(es == -1) > 1:
 184         raise ValueError(
 185             "`{}` must have at most one `-1` (given {})"
 186             .format(shape, es))
 187       if np.any(es < -1):
 188         raise ValueError(
 189             "`{}` elements must be either positive integers or `-1`"
 190             "(given {})."
 191             .format(shape, es))
 192     elif validate_args:
 193       assertions.extend([
 194           check_ops.assert_less_equal(
 195               math_ops.reduce_sum(
 196                   math_ops.cast(math_ops.equal(shape, -1), dtypes.int32)),
 197               1,
 198               message="`{}` elements must have at most one `-1`."
 199               .format(shape)),
 200           check_ops.assert_greater_equal(
 201               shape, -1,
 202               message="`{}` elements must be either positive integers or `-1`."
 203               .format(shape)),
 204       ])
 205     return assertions
 206 
 207   def _reshape_helper(self, x, event_shape_in, event_shape_out):
 208     """Reshape only the event_shape of an input `Tensor`."""
 209 
 210     event_ndims_in_ = _static_ndims_from_shape(event_shape_in)
 211     event_ndims_in = _ndims_from_shape(event_shape_in)
 212     x_ndims_, x_ndims = x.shape.ndims, array_ops.rank(x)
 213 
 214     assertions = []
 215 
 216     # Ensure x.event_shape is compatible with event_shape_in.
 217     if (event_ndims_in_ is not None
 218         and x_ndims_ is not None
 219         and x.shape.with_rank_at_least(event_ndims_in_)[
 220             x_ndims_-event_ndims_in_:].is_fully_defined()):
 221       x_event_shape_, x_event_shape = [  # pylint: disable=unbalanced-tuple-unpacking
 222           np.int32(x.shape[x_ndims_-event_ndims_in_:])]*2
 223     else:
 224       x_event_shape_, x_event_shape = (
 225           None, array_ops.shape(x)[x_ndims-event_ndims_in:])
 226 
 227     event_shape_in_ = tensor_util.constant_value(event_shape_in)
 228 
 229     if x_event_shape_ is not None and event_shape_in_ is not None:
 230       # Compare the shape dimensions that are fully specified in the
 231       # input (i.e., for which event_shape_in is not -1). If x_event_shape
 232       # matches along all of these dimensions, it is compatible with
 233       # the desired input shape and any further mismatches (i.e.,
 234       # imcompatibility with the desired *output* shape) will be
 235       # caught inside of array_ops.reshape() below.
 236       x_event_shape_specified_ = x_event_shape_[event_shape_in_ >= 0]
 237       event_shape_in_specified_ = event_shape_in_[event_shape_in_ >= 0]
 238       if not np.equal(x_event_shape_specified_,
 239                       event_shape_in_specified_).all():
 240         raise ValueError(
 241             "Input `event_shape` does not match `event_shape_in` ({} vs {}).".
 242             format(x_event_shape_, event_shape_in_))
 243     elif self.validate_args:
 244       # Similarly to the static case, we compare the shape dimensions
 245       # that are fully specified in the input. We extract these
 246       # dimensions using boolean_mask(), which requires that the mask
 247       # have known ndims. We can assume that shape Tensors always have
 248       # ndims==1 (this assumption is verified inside of
 249       # _maybe_check_valid_shape), so the reshape operation is just a
 250       # no-op that formally encodes this fact to make boolean_mask()
 251       # happy.
 252       event_shape_mask = array_ops.reshape(event_shape_in >= 0, [-1])
 253       x_event_shape_specified = array_ops.boolean_mask(x_event_shape,
 254                                                        event_shape_mask)
 255       event_shape_in_specified = array_ops.boolean_mask(event_shape_in,
 256                                                         event_shape_mask)
 257       assertions.append(check_ops.assert_equal(
 258           x_event_shape_specified, event_shape_in_specified,
 259           message="Input `event_shape` does not match `event_shape_in`."))
 260 
 261     if assertions:
 262       x = control_flow_ops.with_dependencies(assertions, x)
 263 
 264     # get the parts of shape(x) that will not change
 265     sample_and_batch_shape = array_ops.shape(x)
 266 
 267     ndims = (x.shape.ndims if x.shape.ndims is not None
 268              else array_ops.rank(x))
 269     sample_and_batch_shape = sample_and_batch_shape[
 270         :(ndims - math_ops.abs(event_ndims_in))]
 271 
 272     if (event_ndims_in_ is not None
 273         and x_ndims_ is not None
 274         and event_ndims_in_ == x_ndims_):
 275       # Hack to allow forward/inverse_event_shape to do shape
 276       # inference by calling this helper method with a dummy Tensor of
 277       # shape event_shape_in. In this special case,
 278       # sample_and_batch_shape will be empty so we can preserve static
 279       # shape information by avoiding the concat operation below
 280       # (which would be a no-op).
 281       new_shape = event_shape_out
 282     else:
 283       new_shape = array_ops.concat(
 284           [sample_and_batch_shape, event_shape_out], axis=0)
 285 
 286     return array_ops.reshape(x, new_shape)
 287 
 288   def _forward(self, x):
 289     with ops.control_dependencies(self._assertions):
 290       return self._reshape_helper(x,
 291                                   self._event_shape_in,
 292                                   self._event_shape_out)
 293 
 294   def _inverse(self, y):
 295     with ops.control_dependencies(self._assertions):
 296       return self._reshape_helper(y,
 297                                   self._event_shape_out,
 298                                   self._event_shape_in)
 299 
 300   def _inverse_log_det_jacobian(self, y):
 301     with ops.control_dependencies(self._assertions):
 302       return constant_op.constant(0., dtype=y.dtype)
 303 
 304   def _forward_log_det_jacobian(self, x):
 305     with ops.control_dependencies(self._assertions):
 306       return constant_op.constant(0., dtype=x.dtype)
 307 
 308   def _forward_event_shape(self, input_shape):
 309     # NOTE: this method and the other *_event_shape* methods
 310     # compute shape by explicit transformation of a dummy
 311     # variable. This approach is not generally recommended because it
 312     # bloats the graph and could in general trigger side effects.
 313     #
 314     # In this particular case of the Reshape bijector, the
 315     # forward and inverse transforms have no side effects, and we
 316     # believe the reduction in code complexity from delegating the
 317     # heavy lifting to tf.reshape() is worth the added graph ops.
 318     # However, you should think hard before implementing this approach
 319     # in other Bijectors; it is strongly preferred to compute
 320     # shapes explicitly whenever it's feasible to do so.
 321     with ops.control_dependencies(self._assertions):
 322       dummy = array_ops.zeros(dtype=dtypes.float32, shape=input_shape)
 323       dummy_reshaped = self.forward(dummy)
 324       return dummy_reshaped.shape
 325 
 326   def _inverse_event_shape(self, output_shape):
 327     with ops.control_dependencies(self._assertions):
 328       dummy = array_ops.zeros(dtype=dtypes.float32, shape=output_shape)
 329       dummy_reshaped = self.inverse(dummy)
 330       return dummy_reshaped.shape
 331 
 332   def _forward_event_shape_tensor(self, input_shape):
 333     with ops.control_dependencies(self._assertions):
 334       dummy = array_ops.zeros(dtype=dtypes.float32, shape=input_shape)
 335       dummy_reshaped = self.forward(dummy)
 336       return array_ops.shape(dummy_reshaped)
 337 
 338   def _inverse_event_shape_tensor(self, output_shape):
 339     with ops.control_dependencies(self._assertions):
 340       dummy = array_ops.zeros(dtype=dtypes.float32, shape=output_shape)
 341       dummy_reshaped = self.inverse(dummy)
 342       return array_ops.shape(dummy_reshaped)
