  44 def _sqrtx2p1(x):
  45   """Implementation of `sqrt(1 + x**2)` which is stable despite large `x`."""
  46   return array_ops.where(
  47       math_ops.abs(x) * np.sqrt(np.finfo(x.dtype.as_numpy_dtype).eps) <= 1.,
  48       math_ops.sqrt(x**2. + 1.),
  49       # For large x, calculating x**2 can overflow. This can be alleviated by
  50       # considering:
  51       # sqrt(1 + x**2)
  52       # = exp(0.5 log(1 + x**2))
  53       # = exp(0.5 log(x**2 * (1 + x**-2)))
  54       # = exp(log(x) + 0.5 * log(1 + x**-2))
  55       # = |x| * exp(0.5 log(1 + x**-2))
  56       # = |x| * sqrt(1 + x**-2)
  57       # We omit the last term in this approximation.
  58       # When |x| > 1 / sqrt(machineepsilon), the second term will be 1,
  59       # due to sqrt(1 + x**-2) = 1. This is also true with the gradient term,
  60       # and higher order gradients, since the first order derivative of
  61       # sqrt(1 + x**-2) is -2 * x**-3 / (1 + x**-2) = -2 / (x**3 + x),
  62       # and all nth-order derivatives will be O(x**-(n + 2)). This makes any
  63       # gradient terms that contain any derivatives of sqrt(1 + x**-2) vanish.
  64       math_ops.abs(x))

  67 class SinhArcsinh(bijector.Bijector):
  68   """Compute `Y = g(X) = Sinh( (Arcsinh(X) + skewness) * tailweight )`.
  69 
  70   For `skewness in (-inf, inf)` and `tailweight in (0, inf)`, this
  71   transformation is a
  72   diffeomorphism of the real line `(-inf, inf)`.  The inverse transform is
  73   `X = g^{-1}(Y) = Sinh( ArcSinh(Y) / tailweight - skewness )`.
  74 
  75   The `SinhArcsinh` transformation of the Normal is described in
  76   [Sinh-arcsinh distributions](https://www.jstor.org/stable/27798865)
  77   This Bijector allows a similar transformation of any distribution supported on
  78   `(-inf, inf)`.
  79 
  80   #### Meaning of the parameters
  81 
  82   * If `skewness = 0` and `tailweight = 1`, this transform is the identity.
  83   * Positive (negative) `skewness` leads to positive (negative) skew.
  84     * positive skew means, for unimodal `X` centered at zero, the mode of `Y` is
  85       "tilted" to the right.
  86     * positive skew means positive values of `Y` become more likely, and
  87       negative values become less likely.
  88   * Larger (smaller) `tailweight` leads to fatter (thinner) tails.
  89     * Fatter tails mean larger values of `|Y|` become more likely.
  90     * If `X` is a unit Normal, `tailweight < 1` leads to a distribution that is
  91       "flat" around `Y = 0`, and a very steep drop-off in the tails.
  92     * If `X` is a unit Normal, `tailweight > 1` leads to a distribution more
  93       peaked at the mode with heavier tails.
  94 
  95   To see the argument about the tails, note that for `|X| >> 1` and
  96   `|X| >> (|skewness| * tailweight)**tailweight`, we have
  97   `Y approx 0.5 X**tailweight e**(sign(X) skewness * tailweight)`.
  98   """
  99 
 100   @deprecation.deprecated(
 101       "2018-10-01",
 102       "The TensorFlow Distributions library has moved to "
 103       "TensorFlow Probability "
 104       "(https://github.com/tensorflow/probability). You "
 105       "should update all references to use `tfp.distributions` "
 106       "instead of `tf.contrib.distributions`.",
 107       warn_once=True)
 108   def __init__(self,
 109                skewness=None,
 110                tailweight=None,
 111                validate_args=False,
 112                name="SinhArcsinh"):
 113     """Instantiates the `SinhArcsinh` bijector.
 114 
 115     Args:
 116       skewness:  Skewness parameter.  Float-type `Tensor`.  Default is `0`
 117         of type `float32`.
 118       tailweight:  Tailweight parameter.  Positive `Tensor` of same `dtype` as
 119         `skewness` and broadcastable `shape`.  Default is `1` of type `float32`.
 120       validate_args: Python `bool` indicating whether arguments should be
 121         checked for correctness.
 122       name: Python `str` name given to ops managed by this object.
 123     """
 124     self._graph_parents = []
 125     self._name = name
 126     self._validate_args = validate_args
 127     with self._name_scope("init", values=[skewness, tailweight]):
 128       tailweight = 1. if tailweight is None else tailweight
 129       skewness = 0. if skewness is None else skewness
 130       self._skewness = ops.convert_to_tensor(
 131           skewness, name="skewness")
 132       self._tailweight = ops.convert_to_tensor(
 133           tailweight, name="tailweight", dtype=self._skewness.dtype)
 134       check_ops.assert_same_float_dtype([self._skewness, self._tailweight])
 135       if validate_args:
 136         self._tailweight = control_flow_ops.with_dependencies([
 137             check_ops.assert_positive(
 138                 self._tailweight,
 139                 message="Argument tailweight was not positive")
 140         ], self._tailweight)
 141     super(SinhArcsinh, self).__init__(
 142         forward_min_event_ndims=0,
 143         validate_args=validate_args,
 144         name=name)
 145 
 146   @property
 147   def skewness(self):
 148     """The `skewness` in: `Y  = Sinh((Arcsinh(X) + skewness) * tailweight)`."""
 149     return self._skewness
 150 
 151   @property
 152   def tailweight(self):
 153     """The `tailweight` in: `Y = Sinh((Arcsinh(X) + skewness) * tailweight)`."""
 154     return self._tailweight
 155 
 156   def _forward(self, x):
 157     return math_ops.sinh((math_ops.asinh(x) + self.skewness) * self.tailweight)
 158 
 159   def _inverse(self, y):
 160     return math_ops.sinh(math_ops.asinh(y) / self.tailweight - self.skewness)
 161 
 162   def _inverse_log_det_jacobian(self, y):
 163     # x = sinh(arcsinh(y) / tailweight - skewness)
 164     # Using sinh' = cosh, arcsinh'(y) = 1 / sqrt(y**2 + 1),
 165     # dx/dy
 166     # = cosh(arcsinh(y) / tailweight - skewness)
 167     #     / (tailweight * sqrt(y**2 + 1))
 168 
 169     # This is computed inside the log to avoid catastrophic cancellations
 170     # from cosh((arcsinh(y) / tailweight) - skewness) and sqrt(x**2 + 1).
 171     return (
 172         math_ops.log(math_ops.cosh(
 173             math_ops.asinh(y) / self.tailweight - self.skewness)
 174                      # TODO(srvasude): Consider using cosh(arcsinh(x)) in cases
 175                      # where (arcsinh(x) / tailweight) - skewness ~= arcsinh(x).
 176                      / _sqrtx2p1(y))
 177         - math_ops.log(self.tailweight))
 178 
 179   def _forward_log_det_jacobian(self, x):
 180     # y = sinh((arcsinh(x) + skewness) * tailweight)
 181     # Using sinh' = cosh, arcsinh'(x) = 1 / sqrt(x**2 + 1),
 182     # dy/dx
 183     # = cosh((arcsinh(x) + skewness) * tailweight) * tailweight / sqrt(x**2 + 1)
 184 
 185     # This is computed inside the log to avoid catastrophic cancellations
 186     # from cosh((arcsinh(x) + skewness) * tailweight) and sqrt(x**2 + 1).
 187     return (
 188         math_ops.log(math_ops.cosh(
 189             (math_ops.asinh(x) + self.skewness) * self.tailweight)
 190                      # TODO(srvasude): Consider using cosh(arcsinh(x)) in cases
 191                      # where (arcsinh(x) + skewness) * tailweight ~= arcsinh(x).
 192                      / _sqrtx2p1(x))
 193         + math_ops.log(self.tailweight))
