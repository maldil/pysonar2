  34 class Gumbel(bijector.Bijector):
  35   """Compute `Y = g(X) = exp(-exp(-(X - loc) / scale))`.
  36 
  37   This bijector maps inputs from `[-inf, inf]` to [0, 1]`. The inverse of the
  38   bijector applied to a uniform random variable `X ~ U(0, 1) gives back a
  39   random variable with the
  40   [Gumbel distribution](https://en.wikipedia.org/wiki/Gumbel_distribution):
  41 
  42   ```none
  43   Y ~ Gumbel(loc, scale)
  44   pdf(y; loc, scale) = exp(
  45     -( (y - loc) / scale + exp(- (y - loc) / scale) ) ) / scale
  46   ```
  47   """
  48 
  49   @deprecation.deprecated(
  50       "2018-10-01",
  51       "The TensorFlow Distributions library has moved to "
  52       "TensorFlow Probability "
  53       "(https://github.com/tensorflow/probability). You "
  54       "should update all references to use `tfp.distributions` "
  55       "instead of `tf.contrib.distributions`.",
  56       warn_once=True)
  57   def __init__(self,
  58                loc=0.,
  59                scale=1.,
  60                validate_args=False,
  61                name="gumbel"):
  62     """Instantiates the `Gumbel` bijector.
  63 
  64     Args:
  65       loc: Float-like `Tensor` that is the same dtype and is
  66         broadcastable with `scale`.
  67         This is `loc` in `Y = g(X) = exp(-exp(-(X - loc) / scale))`.
  68       scale: Positive Float-like `Tensor` that is the same dtype and is
  69         broadcastable with `loc`.
  70         This is `scale` in `Y = g(X) = exp(-exp(-(X - loc) / scale))`.
  71       validate_args: Python `bool` indicating whether arguments should be
  72         checked for correctness.
  73       name: Python `str` name given to ops managed by this object.
  74     """
  75     self._graph_parents = []
  76     self._name = name
  77     self._validate_args = validate_args
  78     with self._name_scope("init", values=[loc, scale]):
  79       self._loc = ops.convert_to_tensor(loc, name="loc")
  80       self._scale = ops.convert_to_tensor(scale, name="scale")
  81       check_ops.assert_same_float_dtype([self._loc, self._scale])
  82       if validate_args:
  83         self._scale = control_flow_ops.with_dependencies([
  84             check_ops.assert_positive(
  85                 self._scale, message="Argument scale was not positive")
  86         ], self._scale)
  87 
  88     super(Gumbel, self).__init__(
  89         validate_args=validate_args,
  90         forward_min_event_ndims=0,
  91         name=name)
  92 
  93   @property
  94   def loc(self):
  95     """The `loc` in `Y = g(X) = exp(-exp(-(X - loc) / scale))`."""
  96     return self._loc
  97 
  98   @property
  99   def scale(self):
 100     """This is `scale` in `Y = g(X) = exp(-exp(-(X - loc) / scale))`."""
 101     return self._scale
 102 
 103   def _forward(self, x):
 104     z = (x - self.loc) / self.scale
 105     return math_ops.exp(-math_ops.exp(-z))
 106 
 107   def _inverse(self, y):
 108     y = self._maybe_assert_valid_y(y)
 109     return self.loc - self.scale * math_ops.log(-math_ops.log(y))
 110 
 111   def _inverse_log_det_jacobian(self, y):
 112     y = self._maybe_assert_valid_y(y)
 113     return math_ops.log(self.scale / (-math_ops.log(y) * y))
 114 
 115   def _forward_log_det_jacobian(self, x):
 116     z = (x - self.loc) / self.scale
 117     return -z - math_ops.exp(-z) - math_ops.log(self.scale)
 118 
 119   def _maybe_assert_valid_y(self, y):
 120     if not self.validate_args:
 121       return y
 122     is_positive = check_ops.assert_non_negative(
 123         y, message="Inverse transformation input must be greater than 0.")
 124     less_than_one = check_ops.assert_less_equal(
 125         y,
 126         constant_op.constant(1., y.dtype),
 127         message="Inverse transformation input must be less than or equal to 1.")
 128     return control_flow_ops.with_dependencies([is_positive, less_than_one], y)
