  36 class Ordered(bijector.Bijector):
  37   """Bijector which maps a tensor x_k that has increasing elements in the last
  38   dimension to an unconstrained tensor y_k.
  39 
  40   Both the domain and the codomain of the mapping is [-inf, inf], however,
  41   the input of the forward mapping must be strictly increasing.
  42   The inverse of the bijector applied to a normal random vector `y ~ N(0, 1)`
  43   gives back a sorted random vector with the same distribution `x ~ N(0, 1)`
  44   where `x = sort(y)`
  45 
  46   On the last dimension of the tensor, Ordered bijector performs:
  47   `y[0] = x[0]`
  48   `y[1:] = math_ops.log(x[1:] - x[:-1])`
  49 
  50   #### Example Use:
  51 
  52   ```python
  53   bijector.Ordered().forward([2, 3, 4])
  54   # Result: [2., 0., 0.]
  55 
  56   bijector.Ordered().inverse([0.06428002, -1.07774478, -0.71530371])
  57   # Result: [0.06428002, 0.40464228, 0.8936858]
  58   ```
  59   """
  60 
  61   @deprecation.deprecated(
  62       "2018-10-01",
  63       "The TensorFlow Distributions library has moved to "
  64       "TensorFlow Probability "
  65       "(https://github.com/tensorflow/probability). You "
  66       "should update all references to use `tfp.distributions` "
  67       "instead of `tf.contrib.distributions`.",
  68       warn_once=True)
  69   def __init__(self, validate_args=False, name="ordered"):
  70     super(Ordered, self).__init__(
  71         forward_min_event_ndims=1,
  72         validate_args=validate_args,
  73         name=name)
  74 
  75   def _forward_event_shape(self, input_shape):
  76     if input_shape.ndims is None or input_shape[-1] is None:
  77       return input_shape
  78     return tensor_shape.TensorShape([input_shape[-1]])
  79 
  80   def _forward_event_shape_tensor(self, input_shape):
  81     return (input_shape[-1])[..., array_ops.newaxis]
  82 
  83   def _inverse_event_shape(self, output_shape):
  84     if output_shape.ndims is None or output_shape[-1] is None:
  85       return output_shape
  86     if output_shape[-1] <= 1:
  87       raise ValueError("output_shape[-1] = %d <= 1" % output_shape[-1])
  88     return tensor_shape.TensorShape([output_shape[-1]])
  89 
  90   def _inverse_event_shape_tensor(self, output_shape):
  91     if self.validate_args:
  92       is_greater_one = check_ops.assert_greater(
  93           output_shape[-1], 1, message="Need last dimension greater than 1.")
  94       output_shape = control_flow_ops.with_dependencies(
  95           [is_greater_one], output_shape)
  96     return (output_shape[-1])[..., array_ops.newaxis]
  97 
  98   def _forward(self, x):
  99     x = self._maybe_assert_valid_x(x)
 100     y0 = x[..., 0, array_ops.newaxis]
 101     yk = math_ops.log(x[..., 1:] - x[..., :-1])
 102     y = array_ops.concat([y0, yk], axis=-1)
 103     return y
 104 
 105   def _inverse(self, y):
 106     x0 = y[..., 0, array_ops.newaxis]
 107     xk = math_ops.exp(y[..., 1:])
 108     x = array_ops.concat([x0, xk], axis=-1)
 109     return math_ops.cumsum(x, axis=-1)
 110 
 111   def _inverse_log_det_jacobian(self, y):
 112     # The Jacobian of the inverse mapping is lower
 113     # triangular, with the diagonal elements being:
 114     # J[i,i] = 1 if i=1, and
 115     #          exp(y_i) if 1<i<=K
 116     # which gives the absolute Jacobian determinant:
 117     # |det(Jac)| = prod_{i=1}^{K} exp(y[i]).
 118     # (1) - Stan Modeling Language User's Guide and Reference Manual
 119     #       Version 2.17.0 session 35.2
 120     return math_ops.reduce_sum(y[..., 1:], axis=-1)
 121 
 122   def _forward_log_det_jacobian(self, x):
 123     x = self._maybe_assert_valid_x(x)
 124     return -math_ops.reduce_sum(
 125         math_ops.log(x[..., 1:] - x[..., :-1]),
 126         axis=-1)
 127 
 128   def _maybe_assert_valid_x(self, x):
 129     if not self.validate_args:
 130       return x
 131     is_valid = check_ops.assert_positive(
 132         x[..., 1:] - x[..., :-1],
 133         message="Forward transformation input must be strictly increasing.")
 134     return control_flow_ops.with_dependencies([is_valid], x)
