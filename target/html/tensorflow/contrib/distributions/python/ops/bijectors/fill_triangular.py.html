  37 class FillTriangular(bijector.Bijector):
  38   """Transforms vectors to triangular.
  39 
  40   Triangular matrix elements are filled in a clockwise spiral.
  41 
  42   Given input with shape `batch_shape + [d]`, produces output with
  43   shape `batch_shape + [n, n]`, where
  44    `n = (-1 + sqrt(1 + 8 * d))/2`.
  45   This follows by solving the quadratic equation
  46    `d = 1 + 2 + ... + n = n * (n + 1)/2`.
  47 
  48   #### Example
  49 
  50   ```python
  51   b = tfb.FillTriangular(upper=False)
  52   b.forward([1, 2, 3, 4, 5, 6])
  53   # ==> [[4, 0, 0],
  54   #      [6, 5, 0],
  55   #      [3, 2, 1]]
  56 
  57   b = tfb.FillTriangular(upper=True)
  58   b.forward([1, 2, 3, 4, 5, 6])
  59   # ==> [[1, 2, 3],
  60   #      [0, 5, 6],
  61   #      [0, 0, 4]]
  62 
  63   ```
  64   """
  65 
  66   @deprecation.deprecated(
  67       "2018-10-01",
  68       "The TensorFlow Distributions library has moved to "
  69       "TensorFlow Probability "
  70       "(https://github.com/tensorflow/probability). You "
  71       "should update all references to use `tfp.distributions` "
  72       "instead of `tf.contrib.distributions`.",
  73       warn_once=True)
  74   def __init__(self,
  75                upper=False,
  76                validate_args=False,
  77                name="fill_triangular"):
  78     """Instantiates the `FillTriangular` bijector.
  79 
  80     Args:
  81       upper: Python `bool` representing whether output matrix should be upper
  82         triangular (`True`) or lower triangular (`False`, default).
  83       validate_args: Python `bool` indicating whether arguments should be
  84         checked for correctness.
  85       name: Python `str` name given to ops managed by this object.
  86     """
  87     self._upper = upper
  88     super(FillTriangular, self).__init__(
  89         forward_min_event_ndims=1,
  90         inverse_min_event_ndims=2,
  91         validate_args=validate_args,
  92         name=name)
  93 
  94   def _forward(self, x):
  95     return dist_util.fill_triangular(x, upper=self._upper)
  96 
  97   def _inverse(self, y):
  98     return dist_util.fill_triangular_inverse(y, upper=self._upper)
  99 
 100   def _forward_log_det_jacobian(self, x):
 101     return array_ops.zeros_like(x[..., 0])
 102 
 103   def _inverse_log_det_jacobian(self, y):
 104     return array_ops.zeros_like(y[..., 0, 0])
 105 
 106   def _forward_event_shape(self, input_shape):
 107     batch_shape, d = input_shape[:-1], input_shape[-1].value
 108     if d is None:
 109       n = None
 110     else:
 111       n = vector_size_to_square_matrix_size(d, self.validate_args)
 112     return batch_shape.concatenate([n, n])
 113 
 114   def _inverse_event_shape(self, output_shape):
 115     batch_shape, n1, n2 = (output_shape[:-2],
 116                            output_shape[-2].value,
 117                            output_shape[-1].value)
 118     if n1 is None or n2 is None:
 119       m = None
 120     elif n1 != n2:
 121       raise ValueError("Matrix must be square. (saw [{}, {}])".format(n1, n2))
 122     else:
 123       m = n1 * (n1 + 1) / 2
 124     return batch_shape.concatenate([m])
 125 
 126   def _forward_event_shape_tensor(self, input_shape_tensor):
 127     batch_shape, d = input_shape_tensor[:-1], input_shape_tensor[-1]
 128     n = vector_size_to_square_matrix_size(d, self.validate_args)
 129     return array_ops.concat([batch_shape, [n, n]], axis=0)
 130 
 131   def _inverse_event_shape_tensor(self, output_shape_tensor):
 132     batch_shape, n = output_shape_tensor[:-2], output_shape_tensor[-1]
 133     if self.validate_args:
 134       is_square_matrix = check_ops.assert_equal(
 135           n, output_shape_tensor[-2], message="Matrix must be square.")
 136       with ops.control_dependencies([is_square_matrix]):
 137         n = array_ops.identity(n)
 138     d = math_ops.cast(n * (n + 1) / 2, output_shape_tensor.dtype)
 139     return array_ops.concat([batch_shape, [d]], axis=0)

 142 @deprecation.deprecated(
 143     "2018-10-01",
 144     "The TensorFlow Distributions library has moved to "
 145     "TensorFlow Probability "
 146     "(https://github.com/tensorflow/probability). You "
 147     "should update all references to use `tfp.distributions` "
 148     "instead of `tf.contrib.distributions`.",
 149     warn_once=True)
 150 def vector_size_to_square_matrix_size(d, validate_args, name=None):
 151   """Convert a vector size to a matrix size."""
 152   if isinstance(d, (float, int, np.generic, np.ndarray)):
 153     n = (-1 + np.sqrt(1 + 8 * d)) / 2.
 154     if float(int(n)) != n:
 155       raise ValueError("Vector length is not a triangular number.")
 156     return int(n)
 157   else:
 158     with ops.name_scope(name, "vector_size_to_square_matrix_size", [d]) as name:
 159       n = (-1. + math_ops.sqrt(1 + 8. * math_ops.to_float(d))) / 2.
 160       if validate_args:
 161         with ops.control_dependencies([check_ops.assert_equal(
 162             math_ops.to_float(math_ops.to_int32(n)), n,
 163             message="Vector length is not a triangular number")]):
 164           n = array_ops.identity(n)
 165       return math_ops.cast(n, d.dtype)
