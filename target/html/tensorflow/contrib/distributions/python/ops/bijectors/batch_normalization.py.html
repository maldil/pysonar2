  46 def _undo_batch_normalization(x,
  47                               mean,
  48                               variance,
  49                               offset,
  50                               scale,
  51                               variance_epsilon,
  52                               name=None):
  53   r"""Inverse of tf.nn.batch_normalization.
  54 
  55   Args:
  56     x: Input `Tensor` of arbitrary dimensionality.
  57     mean: A mean `Tensor`.
  58     variance: A variance `Tensor`.
  59     offset: An offset `Tensor`, often denoted `beta` in equations, or
  60       None. If present, will be added to the normalized tensor.
  61     scale: A scale `Tensor`, often denoted `gamma` in equations, or
  62       `None`. If present, the scale is applied to the normalized tensor.
  63     variance_epsilon: A small `float` added to the minibatch `variance` to
  64       prevent dividing by zero.
  65     name: A name for this operation (optional).
  66 
  67   Returns:
  68     batch_unnormalized: The de-normalized, de-scaled, de-offset `Tensor`.
  69   """
  70   with ops.name_scope(
  71       name, "undo_batchnorm", [x, mean, variance, scale, offset]):
  72     # inv = math_ops.rsqrt(variance + variance_epsilon)
  73     # if scale is not None:
  74     #   inv *= scale
  75     # return x * inv + (
  76     #     offset - mean * inv if offset is not None else -mean * inv)
  77     rescale = math_ops.sqrt(variance + variance_epsilon)
  78     if scale is not None:
  79       rescale /= scale
  80     batch_unnormalized = x * rescale + (
  81         mean - offset * rescale if offset is not None else mean)
  82     return batch_unnormalized

  85 class BatchNormalization(bijector.Bijector):
  86   """Compute `Y = g(X) s.t. X = g^-1(Y) = (Y - mean(Y)) / std(Y)`.
  87 
  88   Applies Batch Normalization [(Ioffe and Szegedy, 2015)][1] to samples from a
  89   data distribution. This can be used to stabilize training of normalizing
  90   flows ([Papamakarios et al., 2016][3]; [Dinh et al., 2017][2])
  91 
  92   When training Deep Neural Networks (DNNs), it is common practice to
  93   normalize or whiten features by shifting them to have zero mean and
  94   scaling them to have unit variance.
  95 
  96   The `inverse()` method of the `BatchNormalization` bijector, which is used in
  97   the log-likelihood computation of data samples, implements the normalization
  98   procedure (shift-and-scale) using the mean and standard deviation of the
  99   current minibatch.
 100 
 101   Conversely, the `forward()` method of the bijector de-normalizes samples (e.g.
 102   `X*std(Y) + mean(Y)` with the running-average mean and standard deviation
 103   computed at training-time. De-normalization is useful for sampling.
 104 
 105   ```python
 106 
 107   dist = tfd.TransformedDistribution(
 108       distribution=tfd.Normal()),
 109       bijector=tfb.BatchNorm())
 110 
 111   y = tfd.MultivariateNormalDiag(loc=1., scale=2.).sample(100)  # ~ N(1, 2)
 112   x = dist.bijector.inverse(y)  # ~ N(0, 1)
 113   y = dist.sample()  # ~ N(1, 2)
 114   ```
 115 
 116   During training time, `BatchNorm.inverse` and `BatchNorm.forward` are not
 117   guaranteed to be inverses of each other because `inverse(y)` uses statistics
 118   of the current minibatch, while `forward(x)` uses running-average statistics
 119   accumulated from training. In other words,
 120   `BatchNorm.inverse(BatchNorm.forward(...))` and
 121   `BatchNorm.forward(BatchNorm.inverse(...))` will be identical when
 122   `training=False` but may be different when `training=True`.
 123 
 124   #### References
 125 
 126   [1]: Sergey Ioffe and Christian Szegedy. Batch Normalization: Accelerating
 127        Deep Network Training by Reducing Internal Covariate Shift. In
 128        _International Conference on Machine Learning_, 2015.
 129        https://arxiv.org/abs/1502.03167
 130 
 131   [2]: Laurent Dinh, Jascha Sohl-Dickstein, and Samy Bengio. Density Estimation
 132        using Real NVP. In _International Conference on Learning
 133        Representations_, 2017. https://arxiv.org/abs/1605.08803
 134 
 135   [3]: George Papamakarios, Theo Pavlakou, and Iain Murray. Masked
 136        Autoregressive Flow for Density Estimation. In _Neural Information
 137        Processing Systems_, 2017. https://arxiv.org/abs/1705.07057
 138   """

 148   def __init__(self,
 149                batchnorm_layer=None,
 150                training=True,
 151                validate_args=False,
 152                name="batch_normalization"):
 153     """Instantiates the `BatchNorm` bijector.
 154 
 155     Args:
 156       batchnorm_layer: `tf.layers.BatchNormalization` layer object. If `None`,
 157         defaults to
 158         `tf.layers.BatchNormalization(gamma_constraint=nn_ops.relu(x) + 1e-6)`.
 159         This ensures positivity of the scale variable.
 160 
 161       training: If True, updates running-average statistics during call to
 162         `inverse()`.
 163       validate_args: Python `bool` indicating whether arguments should be
 164         checked for correctness.
 165       name: Python `str` name given to ops managed by this object.
 166     Raises:
 167       ValueError: If bn_layer is not an instance of
 168         `tf.layers.BatchNormalization`, or if it is specified with `renorm=True`
 169         or a virtual batch size.
 170     """
 171     # Scale must be positive.
 172     g_constraint = lambda x: nn.relu(x) + 1e-6
 173     self.batchnorm = batchnorm_layer or normalization.BatchNormalization(
 174         gamma_constraint=g_constraint)
 175     self._validate_bn_layer(self.batchnorm)
 176     self._training = training
 177     if isinstance(self.batchnorm.axis, int):
 178       forward_min_event_ndims = 1
 179     else:
 180       forward_min_event_ndims = len(self.batchnorm.axis)
 181     super(BatchNormalization, self).__init__(
 182         forward_min_event_ndims=forward_min_event_ndims,
 183         validate_args=validate_args, name=name)
 184 
 185   def _validate_bn_layer(self, layer):
 186     """Check for valid BatchNormalization layer.
 187 
 188     Args:
 189       layer: Instance of `tf.layers.BatchNormalization`.
 190     Raises:
 191       ValueError: If batchnorm_layer argument is not an instance of
 192       `tf.layers.BatchNormalization`, or if `batchnorm_layer.renorm=True` or
 193       if `batchnorm_layer.virtual_batch_size` is specified.
 194     """
 195     if not isinstance(layer, normalization.BatchNormalization):
 196       raise ValueError(
 197           "batchnorm_layer must be an instance of BatchNormalization layer.")
 198     if layer.renorm:
 199       raise ValueError("BatchNorm Bijector does not support renormalization.")
 200     if layer.virtual_batch_size:
 201       raise ValueError(
 202           "BatchNorm Bijector does not support virtual batch sizes.")
 203 
 204   def _get_broadcast_fn(self, x):
 205     # Compute shape to broadcast scale/shift parameters to.
 206     if not x.shape.is_fully_defined():
 207       raise ValueError("Input must have shape known at graph construction.")
 208     input_shape = np.int32(x.shape.as_list())
 209 
 210     ndims = len(input_shape)
 211     reduction_axes = [i for i in range(ndims) if i not in self.batchnorm.axis]
 212     # Broadcasting only necessary for single-axis batch norm where the axis is
 213     # not the last dimension
 214     broadcast_shape = [1] * ndims
 215     broadcast_shape[self.batchnorm.axis[0]] = (
 216         input_shape[self.batchnorm.axis[0]])
 217     def _broadcast(v):
 218       if (v is not None and
 219           len(v.get_shape()) != ndims and
 220           reduction_axes != list(range(ndims - 1))):
 221         return array_ops.reshape(v, broadcast_shape)
 222       return v
 223     return _broadcast
 224 
 225   def _normalize(self, y):
 226     return self.batchnorm.apply(y, training=self._training)
 227 
 228   def _de_normalize(self, x):
 229     # Uses the saved statistics.
 230     if not self.batchnorm.built:
 231       input_shape = x.get_shape()
 232       self.batchnorm.build(input_shape)
 233     broadcast_fn = self._get_broadcast_fn(x)
 234     mean = broadcast_fn(self.batchnorm.moving_mean)
 235     variance = broadcast_fn(self.batchnorm.moving_variance)
 236     beta = broadcast_fn(self.batchnorm.beta) if self.batchnorm.center else None
 237     gamma = broadcast_fn(self.batchnorm.gamma) if self.batchnorm.scale else None
 238     return _undo_batch_normalization(
 239         x, mean, variance, beta, gamma, self.batchnorm.epsilon)
 240 
 241   def _forward(self, x):
 242     return self._de_normalize(x)
 243 
 244   def _inverse(self, y):
 245     return self._normalize(y)
 246 
 247   def _forward_log_det_jacobian(self, x):
 248     # Uses saved statistics to compute volume distortion.
 249     return -self._inverse_log_det_jacobian(x, use_saved_statistics=True)
 250 
 251   def _inverse_log_det_jacobian(self, y, use_saved_statistics=False):
 252     if not y.shape.is_fully_defined():
 253       raise ValueError("Input must have shape known at graph construction.")
 254     input_shape = np.int32(y.shape.as_list())
 255 
 256     if not self.batchnorm.built:
 257       # Create variables.
 258       self.batchnorm.build(input_shape)
 259 
 260     event_dims = self.batchnorm.axis
 261     reduction_axes = [i for i in range(len(input_shape)) if i not in event_dims]
 262 
 263     if use_saved_statistics or not self._training:
 264       log_variance = math_ops.log(
 265           self.batchnorm.moving_variance + self.batchnorm.epsilon)
 266     else:
 267       # At training-time, ildj is computed from the mean and log-variance across
 268       # the current minibatch.
 269       _, v = nn.moments(y, axes=reduction_axes, keep_dims=True)
 270       log_variance = math_ops.log(v + self.batchnorm.epsilon)
 271 
 272     # `gamma` and `log Var(y)` reductions over event_dims.
 273     # Log(total change in area from gamma term).
 274     log_total_gamma = math_ops.reduce_sum(math_ops.log(self.batchnorm.gamma))
 275 
 276     # Log(total change in area from log-variance term).
 277     log_total_variance = math_ops.reduce_sum(log_variance)
 278     # The ildj is scalar, as it does not depend on the values of x and are
 279     # constant across minibatch elements.
 280     return log_total_gamma - 0.5 * log_total_variance
