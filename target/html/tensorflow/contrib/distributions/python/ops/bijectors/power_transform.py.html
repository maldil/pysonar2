  35 class PowerTransform(bijector.Bijector):
  36   """Compute `Y = g(X) = (1 + X * c)**(1 / c), X >= -1 / c`.
  37 
  38   The [power transform](https://en.wikipedia.org/wiki/Power_transform) maps
  39   inputs from `[0, inf]` to `[-1/c, inf]`; this is equivalent to the `inverse`
  40   of this bijector.
  41 
  42   This bijector is equivalent to the `Exp` bijector when `c=0`.
  43   """
  44 
  45   @deprecation.deprecated(
  46       "2018-10-01",
  47       "The TensorFlow Distributions library has moved to "
  48       "TensorFlow Probability "
  49       "(https://github.com/tensorflow/probability). You "
  50       "should update all references to use `tfp.distributions` "
  51       "instead of `tf.contrib.distributions`.",
  52       warn_once=True)
  53   def __init__(self,
  54                power=0.,
  55                validate_args=False,
  56                name="power_transform"):
  57     """Instantiates the `PowerTransform` bijector.
  58 
  59     Args:
  60       power: Python `float` scalar indicating the transform power, i.e.,
  61         `Y = g(X) = (1 + X * c)**(1 / c)` where `c` is the `power`.
  62       validate_args: Python `bool` indicating whether arguments should be
  63         checked for correctness.
  64       name: Python `str` name given to ops managed by this object.
  65 
  66     Raises:
  67       ValueError: if `power < 0` or is not known statically.
  68     """
  69     self._graph_parents = []
  70     self._name = name
  71     self._validate_args = validate_args
  72     with self._name_scope("init", values=[power]):
  73       power = tensor_util.constant_value(
  74           ops.convert_to_tensor(power, name="power"))
  75     if power is None or power < 0:
  76       raise ValueError("`power` must be a non-negative TF constant.")
  77     self._power = power
  78     super(PowerTransform, self).__init__(
  79         forward_min_event_ndims=0,
  80         validate_args=validate_args,
  81         name=name)
  82 
  83   @property
  84   def power(self):
  85     """The `c` in: `Y = g(X) = (1 + X * c)**(1 / c)`."""
  86     return self._power
  87 
  88   def _forward(self, x):
  89     x = self._maybe_assert_valid_x(x)
  90     if self.power == 0.:
  91       return math_ops.exp(x)
  92     # If large x accuracy is an issue, consider using:
  93     # (1. + x * self.power)**(1. / self.power) when x >> 1.
  94     return math_ops.exp(math_ops.log1p(x * self.power) / self.power)
  95 
  96   def _inverse(self, y):
  97     y = self._maybe_assert_valid_y(y)
  98     if self.power == 0.:
  99       return math_ops.log(y)
 100     # If large y accuracy is an issue, consider using:
 101     # (y**self.power - 1.) / self.power when y >> 1.
 102     return math_ops.expm1(math_ops.log(y) * self.power) / self.power
 103 
 104   def _inverse_log_det_jacobian(self, y):
 105     y = self._maybe_assert_valid_y(y)
 106     return (self.power - 1.) * math_ops.log(y)
 107 
 108   def _forward_log_det_jacobian(self, x):
 109     x = self._maybe_assert_valid_x(x)
 110     if self.power == 0.:
 111       return x
 112     return (1. / self.power - 1.) * math_ops.log1p(x * self.power)
 113 
 114   def _maybe_assert_valid_x(self, x):
 115     if not self.validate_args or self.power == 0.:
 116       return x
 117     is_valid = check_ops.assert_non_negative(
 118         1. + self.power * x,
 119         message="Forward transformation input must be at least {}.".format(
 120             -1. / self.power))
 121     return control_flow_ops.with_dependencies([is_valid], x)
 122 
 123   def _maybe_assert_valid_y(self, y):
 124     if not self.validate_args:
 125       return y
 126     is_valid = check_ops.assert_positive(
 127         y, message="Inverse transformation input must be greater than 0.")
 128     return control_flow_ops.with_dependencies([is_valid], y)
