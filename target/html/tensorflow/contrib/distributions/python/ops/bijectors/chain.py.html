  55 def _compute_min_event_ndims(bijector_list, compute_forward=True):
  56   """Computes the min_event_ndims associated with the give list of bijectors.
  57 
  58   Given a list `bijector_list` of bijectors, compute the min_event_ndims that is
  59   associated with the composition of bijectors in that list.
  60 
  61   min_event_ndims is the # of right most dimensions for which the bijector has
  62   done necessary computation on (i.e. the non-broadcastable part of the
  63   computation).
  64 
  65   We can derive the min_event_ndims for a chain of bijectors as follows:
  66 
  67   In the case where there are no rank changing bijectors, this will simply be
  68   `max(b.forward_min_event_ndims for b in bijector_list)`. This is because the
  69   bijector with the most forward_min_event_ndims requires the most dimensions,
  70   and hence the chain also requires operating on those dimensions.
  71 
  72   However in the case of rank changing, more care is needed in determining the
  73   exact amount of dimensions. Padding dimensions causes subsequent bijectors to
  74   operate on the padded dimensions, and Removing dimensions causes bijectors to
  75   operate more left.
  76 
  77   Args:
  78     bijector_list: List of bijectors to be composed by chain.
  79     compute_forward: Boolean. If True, computes the min_event_ndims associated
  80       with a forward call to Chain, and otherwise computes the min_event_ndims
  81       associated with an inverse call to Chain. The latter is the same as the
  82       min_event_ndims associated with a forward call to Invert(Chain(....)).
  83 
  84   Returns:
  85     min_event_ndims
  86   """
  87   min_event_ndims = 0
  88   # This is a mouthful, but what this encapsulates is that if not for rank
  89   # changing bijectors, we'd only need to compute the largest of the min
  90   # required ndims. Hence "max_min". Due to rank changing bijectors, we need to
  91   # account for synthetic rank growth / synthetic rank decrease from a rank
  92   # changing bijector.
  93   rank_changed_adjusted_max_min_event_ndims = 0
  94 
  95   if compute_forward:
  96     bijector_list = reversed(bijector_list)
  97 
  98   for b in bijector_list:
  99     if compute_forward:
 100       current_min_event_ndims = b.forward_min_event_ndims
 101       current_inverse_min_event_ndims = b.inverse_min_event_ndims
 102     else:
 103       current_min_event_ndims = b.inverse_min_event_ndims
 104       current_inverse_min_event_ndims = b.forward_min_event_ndims
 105 
 106     # New dimensions were touched.
 107     if rank_changed_adjusted_max_min_event_ndims < current_min_event_ndims:
 108       min_event_ndims += (
 109           current_min_event_ndims - rank_changed_adjusted_max_min_event_ndims)
 110     rank_changed_adjusted_max_min_event_ndims = max(
 111         current_min_event_ndims, rank_changed_adjusted_max_min_event_ndims)
 112 
 113     # If the number of dimensions has increased via forward, then
 114     # inverse_min_event_ndims > forward_min_event_ndims, and hence the
 115     # dimensions we computed on, have moved left (so we have operated
 116     # on additional dimensions).
 117     # Conversely, if the number of dimensions has decreased via forward,
 118     # then we have inverse_min_event_ndims < forward_min_event_ndims,
 119     # and so we will have operated on fewer right most dimensions.
 120 
 121     number_of_changed_dimensions = (
 122         current_min_event_ndims - current_inverse_min_event_ndims)
 123     rank_changed_adjusted_max_min_event_ndims -= number_of_changed_dimensions
 124   return min_event_ndims

 127 class Chain(bijector.Bijector):
 128   """Bijector which applies a sequence of bijectors.
 129 
 130   Example Use:
 131 
 132   ```python
 133   chain = Chain([Exp(), Softplus()], name="one_plus_exp")
 134   ```
 135 
 136   Results in:
 137 
 138   * Forward:
 139 
 140    ```python
 141    exp = Exp()
 142    softplus = Softplus()
 143    Chain([exp, softplus]).forward(x)
 144    = exp.forward(softplus.forward(x))
 145    = tf.exp(tf.log(1. + tf.exp(x)))
 146    = 1. + tf.exp(x)
 147    ```
 148 
 149   * Inverse:
 150 
 151    ```python
 152    exp = Exp()
 153    softplus = Softplus()
 154    Chain([exp, softplus]).inverse(y)
 155    = softplus.inverse(exp.inverse(y))
 156    = tf.log(tf.exp(tf.log(y)) - 1.)
 157    = tf.log(y - 1.)
 158    ```
 159 
 160   """
 161 
 162   @deprecation.deprecated(
 163       "2018-10-01",
 164       "The TensorFlow Distributions library has moved to "
 165       "TensorFlow Probability "
 166       "(https://github.com/tensorflow/probability). You "
 167       "should update all references to use `tfp.distributions` "
 168       "instead of `tf.contrib.distributions`.",
 169       warn_once=True)
 170   def __init__(self, bijectors=None, validate_args=False, name=None):
 171     """Instantiates `Chain` bijector.
 172 
 173     Args:
 174       bijectors: Python `list` of bijector instances. An empty list makes this
 175         bijector equivalent to the `Identity` bijector.
 176       validate_args: Python `bool` indicating whether arguments should be
 177         checked for correctness.
 178       name: Python `str`, name given to ops managed by this object. Default:
 179         E.g., `Chain([Exp(), Softplus()]).name == "chain_of_exp_of_softplus"`.
 180 
 181     Raises:
 182       ValueError: if bijectors have different dtypes.
 183     """
 184     if bijectors is None:
 185       bijectors = ()
 186     self._bijectors = bijectors
 187 
 188     for a_bijector in bijectors:
 189       if not a_bijector._is_injective:  # pylint: disable=protected-access
 190         raise NotImplementedError(
 191             "Invert is not implemented for non-injective bijector ({})".format(
 192                 a_bijector.name))
 193 
 194     dtype = list(set([b.dtype for b in bijectors]))
 195     if len(dtype) > 2:
 196       raise ValueError("incompatible dtypes: %s" % dtype)
 197     elif len(dtype) == 2:
 198       dtype = dtype[1] if dtype[0] is None else dtype[0]
 199     elif len(dtype) == 1:
 200       dtype = dtype[0]
 201     else:
 202       dtype = None
 203 
 204     inverse_min_event_ndims = _compute_min_event_ndims(
 205         bijectors, compute_forward=False)
 206     forward_min_event_ndims = _compute_min_event_ndims(
 207         bijectors, compute_forward=True)
 208 
 209     super(Chain, self).__init__(
 210         graph_parents=list(itertools.chain.from_iterable(
 211             b.graph_parents for b in bijectors)),
 212         forward_min_event_ndims=forward_min_event_ndims,
 213         inverse_min_event_ndims=inverse_min_event_ndims,
 214         is_constant_jacobian=all(b.is_constant_jacobian for b in bijectors),
 215         validate_args=validate_args,
 216         dtype=dtype,
 217         name=name or ("identity" if not bijectors else
 218                       "_of_".join(["chain"] + [b.name for b in bijectors])))
 219 
 220   @property
 221   def bijectors(self):
 222     return self._bijectors
 223 
 224   def _shape_helper(self, func_name, input_shape, reverse):
 225     new_shape = input_shape
 226     for b in reversed(self.bijectors) if reverse else self.bijectors:
 227       func = getattr(b, func_name, None)
 228       if func is None:
 229         raise ValueError("unable to call %s on bijector %s (%s)" %
 230                          (func_name, b.name, func))
 231       new_shape = func(new_shape)
 232     return new_shape
 233 
 234   def _forward_event_shape(self, input_shape):
 235     return self._shape_helper("forward_event_shape", input_shape,
 236                               reverse=True)
 237 
 238   def _forward_event_shape_tensor(self, input_shape):
 239     return self._shape_helper(
 240         "forward_event_shape_tensor", input_shape, reverse=True)
 241 
 242   def _inverse_event_shape(self, output_shape):
 243     return self._shape_helper("inverse_event_shape", output_shape,
 244                               reverse=False)
 245 
 246   def _inverse_event_shape_tensor(self, output_shape):
 247     return self._shape_helper("inverse_event_shape_tensor", output_shape,
 248                               reverse=False)
 249 
 250   def _inverse(self, y, **kwargs):
 251     for b in self.bijectors:
 252       y = b.inverse(y, **kwargs.get(b.name, {}))
 253     return y

 255   def _inverse_log_det_jacobian(self, y, **kwargs):
 256     y = ops.convert_to_tensor(y, name="y")
 257     ildj = math_ops.cast(0., dtype=y.dtype.base_dtype)
 258 
 259     if not self.bijectors:
 260       return ildj
 261 
 262     event_ndims = self._maybe_get_static_event_ndims(
 263         self.inverse_min_event_ndims)
 264 
 265     if _use_static_shape(y, event_ndims):
 266       event_shape = y.shape[y.shape.ndims - event_ndims:]
 267     else:
 268       event_shape = array_ops.shape(y)[array_ops.rank(y) - event_ndims:]
 269 
 270     for b in self.bijectors:
 271       ildj += b.inverse_log_det_jacobian(
 272           y, event_ndims=event_ndims, **kwargs.get(b.name, {}))
 273 
 274       if _use_static_shape(y, event_ndims):
 275         event_shape = b.inverse_event_shape(event_shape)
 276         event_ndims = self._maybe_get_static_event_ndims(
 277             event_shape.ndims)
 278       else:
 279         event_shape = b.inverse_event_shape_tensor(event_shape)
 280         event_ndims = array_ops.size(event_shape)
 281         event_ndims_ = self._maybe_get_static_event_ndims(event_ndims)
 282         if event_ndims_ is not None:
 283           event_ndims = event_ndims_
 284 
 285       y = b.inverse(y, **kwargs.get(b.name, {}))
 286     return ildj
 287 
 288   def _forward(self, x, **kwargs):
 289     for b in reversed(self.bijectors):
 290       x = b.forward(x, **kwargs.get(b.name, {}))
 291     return x
 292 
 293   def _forward_log_det_jacobian(self, x, **kwargs):
 294     x = ops.convert_to_tensor(x, name="x")
 295 
 296     fldj = math_ops.cast(0., dtype=x.dtype.base_dtype)
 297 
 298     if not self.bijectors:
 299       return fldj
 300 
 301     event_ndims = self._maybe_get_static_event_ndims(
 302         self.forward_min_event_ndims)
 303 
 304     if _use_static_shape(x, event_ndims):
 305       event_shape = x.shape[x.shape.ndims - event_ndims:]
 306     else:
 307       event_shape = array_ops.shape(x)[array_ops.rank(x) - event_ndims:]
 308 
 309     for b in reversed(self.bijectors):
 310       fldj += b.forward_log_det_jacobian(
 311           x, event_ndims=event_ndims, **kwargs.get(b.name, {}))
 312       if _use_static_shape(x, event_ndims):
 313         event_shape = b.forward_event_shape(event_shape)
 314         event_ndims = self._maybe_get_static_event_ndims(event_shape.ndims)
 315       else:
 316         event_shape = b.forward_event_shape_tensor(event_shape)
 317         event_ndims = array_ops.size(event_shape)
 318         event_ndims_ = self._maybe_get_static_event_ndims(event_ndims)
 319         if event_ndims_ is not None:
 320           event_ndims = event_ndims_
 321 
 322       x = b.forward(x, **kwargs.get(b.name, {}))
 323 
 324     return fldj
