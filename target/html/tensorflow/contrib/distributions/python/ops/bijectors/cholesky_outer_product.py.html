  38 class CholeskyOuterProduct(bijector.Bijector):
  39   """Compute `g(X) = X @ X.T`; X is lower-triangular, positive-diagonal matrix.
  40 
  41   Note: the upper-triangular part of X is ignored (whether or not its zero).
  42 
  43   The surjectivity of g as a map from  the set of n x n positive-diagonal
  44   lower-triangular matrices to the set of SPD matrices follows immediately from
  45   executing the Cholesky factorization algorithm on an SPD matrix A to produce a
  46   positive-diagonal lower-triangular matrix L such that `A = L @ L.T`.
  47 
  48   To prove the injectivity of g, suppose that L_1 and L_2 are lower-triangular
  49   with positive diagonals and satisfy `A = L_1 @ L_1.T = L_2 @ L_2.T`. Then
  50     `inv(L_1) @ A @ inv(L_1).T = [inv(L_1) @ L_2] @ [inv(L_1) @ L_2].T = I`.
  51   Setting `L_3 := inv(L_1) @ L_2`, that L_3 is a positive-diagonal
  52   lower-triangular matrix follows from `inv(L_1)` being positive-diagonal
  53   lower-triangular (which follows from the diagonal of a triangular matrix being
  54   its spectrum), and that the product of two positive-diagonal lower-triangular
  55   matrices is another positive-diagonal lower-triangular matrix.
  56 
  57   A simple inductive argument (proceeding one column of L_3 at a time) shows
  58   that, if `I = L_3 @ L_3.T`, with L_3 being lower-triangular with positive-
  59   diagonal, then `L_3 = I`. Thus, `L_1 = L_2`, proving injectivity of g.
  60 
  61   #### Examples
  62 
  63   ```python
  64   bijector.CholeskyOuterProduct().forward(x=[[1., 0], [2, 1]])
  65   # Result: [[1., 2], [2, 5]], i.e., x @ x.T
  66 
  67   bijector.CholeskyOuterProduct().inverse(y=[[1., 2], [2, 5]])
  68   # Result: [[1., 0], [2, 1]], i.e., cholesky(y).
  69   ```
  70 
  71   """
  72

  83 
  84     Args:
  85       validate_args: Python `bool` indicating whether arguments should be
  86         checked for correctness.
  87       name: Python `str` name given to ops managed by this object.
  88     """
  89     self._graph_parents = []
  90     self._name = name
  91     super(CholeskyOuterProduct, self).__init__(
  92         forward_min_event_ndims=2,
  93         validate_args=validate_args,
  94         name=name)

  96   def _forward(self, x):
  97     if self.validate_args:
  98       is_matrix = check_ops.assert_rank_at_least(x, 2)
  99       shape = array_ops.shape(x)
 100       is_square = check_ops.assert_equal(shape[-2], shape[-1])
 101       x = control_flow_ops.with_dependencies([is_matrix, is_square], x)
 102     # For safety, explicitly zero-out the upper triangular part.
 103     x = array_ops.matrix_band_part(x, -1, 0)
 104     return math_ops.matmul(x, x, adjoint_b=True)
 105 
 106   def _inverse(self, y):
 107     return linalg_ops.cholesky(y)
 108 
 109   def _forward_log_det_jacobian(self, x):
 110     # Let Y be a symmetric, positive definite matrix and write:
 111     #   Y = X X.T
 112     # where X is lower-triangular.
 113     #
 114     # Observe that,
 115     #   dY[i,j]/dX[a,b]
 116     #   = d/dX[a,b] { X[i,:] X[j,:] }
 117     #   = sum_{d=1}^p { I[i=a] I[d=b] X[j,d] + I[j=a] I[d=b] X[i,d] }
 118     #
 119     # To compute the Jacobian dX/dY we must represent X,Y as vectors. Since Y is
 120     # symmetric and X is lower-triangular, we need vectors of dimension:
 121     #   d = p (p + 1) / 2
 122     # where X, Y are p x p matrices, p > 0. We use a row-major mapping, i.e.,
 123     #   k = { i (i + 1) / 2 + j   i>=j
 124     #       { undef               i<j
 125     # and assume zero-based indexes. When k is undef, the element is dropped.
 126     # Example:
 127     #           j      k
 128     #        0 1 2 3  /
 129     #    0 [ 0 . . . ]
 130     # i  1 [ 1 2 . . ]
 131     #    2 [ 3 4 5 . ]
 132     #    3 [ 6 7 8 9 ]
 133     # Write vec[.] to indicate transforming a matrix to vector via k(i,j). (With
 134     # slight abuse: k(i,j)=undef means the element is dropped.)
 135     #
 136     # We now show d vec[Y] / d vec[X] is lower triangular. Assuming both are
 137     # defined, observe that k(i,j) < k(a,b) iff (1) i<a or (2) i=a and j<b.
 138     # In both cases dvec[Y]/dvec[X]@[k(i,j),k(a,b)] = 0 since:
 139     # (1) j<=i<a thus i,j!=a.
 140     # (2) i=a>j  thus i,j!=a.
 141     #
 142     # Since the Jacobian is lower-triangular, we need only compute the product
 143     # of diagonal elements:
 144     #   d vec[Y] / d vec[X] @[k(i,j), k(i,j)]
 145     #   = X[j,j] + I[i=j] X[i,j]
 146     #   = 2 X[j,j].
 147     # Since there is a 2 X[j,j] term for every lower-triangular element of X we
 148     # conclude:
 149     #   |Jac(d vec[Y]/d vec[X])| = 2^p prod_{j=0}^{p-1} X[j,j]^{p-j}.
 150     diag = array_ops.matrix_diag_part(x)
 151 
 152     # We now ensure diag is columnar. Eg, if `diag = [1, 2, 3]` then the output
 153     # is `[[1], [2], [3]]` and if `diag = [[1, 2, 3], [4, 5, 6]]` then the
 154     # output is unchanged.
 155     diag = self._make_columnar(diag)
 156 
 157     if self.validate_args:
 158       is_matrix = check_ops.assert_rank_at_least(
 159           x, 2, message="Input must be a (batch of) matrix.")
 160       shape = array_ops.shape(x)
 161       is_square = check_ops.assert_equal(
 162           shape[-2], shape[-1],
 163           message="Input must be a (batch of) square matrix.")
 164       # Assuming lower-triangular means we only need check diag>0.
 165       is_positive_definite = check_ops.assert_positive(
 166           diag, message="Input must be positive definite.")
 167       x = control_flow_ops.with_dependencies(
 168           [is_matrix, is_square, is_positive_definite], x)
 169 
 170     # Create a vector equal to: [p, p-1, ..., 2, 1].
 171     if x.get_shape().ndims is None or x.get_shape()[-1].value is None:
 172       p_int = array_ops.shape(x)[-1]
 173       p_float = math_ops.cast(p_int, dtype=x.dtype)
 174     else:
 175       p_int = x.get_shape()[-1].value
 176       p_float = np.array(p_int, dtype=x.dtype.as_numpy_dtype)
 177     exponents = math_ops.linspace(p_float, 1., p_int)
 178 
 179     sum_weighted_log_diag = array_ops.squeeze(
 180         math_ops.matmul(math_ops.log(diag),
 181                         exponents[..., array_ops.newaxis]),
 182         axis=-1)
 183     fldj = p_float * np.log(2.) + sum_weighted_log_diag
 184 
 185     # We finally need to undo adding an extra column in non-scalar cases
 186     # where there is a single matrix as input.
 187     if x.get_shape().ndims is not None:
 188       if x.get_shape().ndims == 2:
 189         fldj = array_ops.squeeze(fldj, axis=-1)
 190       return fldj
 191 
 192     shape = array_ops.shape(fldj)
 193     maybe_squeeze_shape = array_ops.concat([
 194         shape[:-1],
 195         distribution_util.pick_vector(
 196             math_ops.equal(array_ops.rank(x), 2),
 197             np.array([], dtype=np.int32), shape[-1:])], 0)
 198     return array_ops.reshape(fldj, maybe_squeeze_shape)
 199 
 200   def _make_columnar(self, x):
 201     """Ensures non-scalar input has at least one column.
 202 
 203     Example:
 204       If `x = [1, 2, 3]` then the output is `[[1], [2], [3]]`.
 205 
 206       If `x = [[1, 2, 3], [4, 5, 6]]` then the output is unchanged.
 207 
 208       If `x = 1` then the output is unchanged.
 209 
 210     Args:
 211       x: `Tensor`.
 212 
 213     Returns:
 214       columnar_x: `Tensor` with at least two dimensions.
 215     """
 216     if x.get_shape().ndims is not None:
 217       if x.get_shape().ndims == 1:
 218         x = x[array_ops.newaxis, :]
 219       return x
 220     shape = array_ops.shape(x)
 221     maybe_expanded_shape = array_ops.concat([
 222         shape[:-1],
 223         distribution_util.pick_vector(
 224             math_ops.equal(array_ops.rank(x), 1),
 225             [1], np.array([], dtype=np.int32)),
 226         shape[-1:],
 227     ], 0)
 228     return array_ops.reshape(x, maybe_expanded_shape)
