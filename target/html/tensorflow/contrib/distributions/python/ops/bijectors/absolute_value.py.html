  33 class AbsoluteValue(bijector.Bijector):
  34   """Computes `Y = g(X) = Abs(X)`, element-wise.
  35 
  36   This non-injective bijector allows for transformations of scalar distributions
  37   with the absolute value function, which maps `(-inf, inf)` to `[0, inf)`.
  38 
  39   * For `y in (0, inf)`, `AbsoluteValue.inverse(y)` returns the set inverse
  40     `{x in (-inf, inf) : |x| = y}` as a tuple, `-y, y`.
  41   * `AbsoluteValue.inverse(0)` returns `0, 0`, which is not the set inverse
  42     (the set inverse is the singleton `{0}`), but "works" in conjunction with
  43     `TransformedDistribution` to produce a left semi-continuous pdf.
  44   * For `y < 0`, `AbsoluteValue.inverse(y)` happily returns the
  45     wrong thing, `-y, y`.  This is done for efficiency.  If
  46     `validate_args == True`, `y < 0` will raise an exception.
  47 
  48 
  49   ```python
  50   tfd = tf.contrib.distributions
  51 
  52   abs = tfd.bijectors.AbsoluteValue()
  53 
  54   abs.forward([-1., 0., 1.])
  55   ==> [1., 0.,  1.]
  56 
  57   abs.inverse(1.)
  58   ==> [-1., 1.]
  59 
  60   # The |dX/dY| is constant, == 1.  So Log|dX/dY| == 0.
  61   abs.inverse_log_det_jacobian(1.)
  62   ==> [0., 0.]
  63 
  64   # Special case handling of 0.
  65   abs.inverse(0.)
  66   ==> [0., 0.]
  67 
  68   abs.inverse_log_det_jacobian(0.)
  69   ==> [0., 0.]
  70   ```
  71 
  72   """
  73 
  74   @deprecation.deprecated(
  75       "2018-10-01",
  76       "The TensorFlow Distributions library has moved to "
  77       "TensorFlow Probability "
  78       "(https://github.com/tensorflow/probability). You "
  79       "should update all references to use `tfp.distributions` "
  80       "instead of `tf.contrib.distributions`.",
  81       warn_once=True)
  82   def __init__(self, validate_args=False, name="absolute_value"):
  83     """Instantiates the `AbsoluteValue` bijector.
  84 
  85     Args:
  86       validate_args: Python `bool` indicating whether arguments should be
  87         checked for correctness, in particular whether inputs to `inverse` and
  88         `inverse_log_det_jacobian` are non-negative.
  89       name: Python `str` name given to ops managed by this object.
  90     """
  91     self._graph_parents = []
  92     self._name = name
  93 
  94     with self._name_scope("init"):
  95       super(AbsoluteValue, self).__init__(
  96           forward_min_event_ndims=0,
  97           is_constant_jacobian=True,
  98           validate_args=validate_args,
  99           name=name)
 100 
 101   def _forward(self, x):
 102     return math_ops.abs(x)
 103 
 104   def _inverse(self, y):
 105     if self.validate_args:
 106       y = control_flow_ops.with_dependencies(
 107           [check_ops.assert_non_negative(y, message="Argument y was negative")],
 108           y)
 109     return -y, y
 110 
 111   def _inverse_log_det_jacobian(self, y):
 112     # If event_ndims = 2,
 113     # F^{-1}(y) = (-y, y), so DF^{-1}(y) = (-1, 1),
 114     # so Log|DF^{-1}(y)| = Log[1, 1] = [0, 0].
 115     zeros = constant_op.constant(0., dtype=y.dtype)
 116     if self.validate_args:
 117       zeros = control_flow_ops.with_dependencies(
 118           [check_ops.assert_non_negative(y, message="Argument y was negative")],
 119           zeros)
 120     return zeros, zeros
 121 
 122   @property
 123   def _is_injective(self):
 124     return False
