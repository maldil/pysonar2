  40 class Permute(bijector.Bijector):
  41   """Permutes the rightmost dimension of a `Tensor`.
  42 
  43   ```python
  44   import tensorflow_probability as tfp
  45   tfb = tfp.bijectors
  46 
  47   reverse = tfb.Permute(permutation=[2, 1, 0])
  48 
  49   reverse.forward([-1., 0., 1.])
  50   # ==> [1., 0., -1]
  51 
  52   reverse.inverse([1., 0., -1])
  53   # ==> [-1., 0., 1.]
  54 
  55   reverse.forward_log_det_jacobian(any_value)
  56   # ==> 0.
  57 
  58   reverse.inverse_log_det_jacobian(any_value)
  59   # ==> 0.
  60   ```
  61 
  62   Warning: `tf.estimator` may repeatedly build the graph thus
  63   `Permute(np.random.permutation(event_size)).astype("int32"))` is not a
  64   reliable parameterization (nor would it be even if using `tf.constant`). A
  65   safe alternative is to use `tf.get_variable` to achieve "init once" behavior,
  66   i.e.,
  67 
  68   ```python
  69   def init_once(x, name):
  70     return tf.get_variable(name, initializer=x, trainable=False)
  71 
  72   Permute(permutation=init_once(
  73       np.random.permutation(event_size).astype("int32"),
  74       name="permutation"))
  75   ```
  76 
  77   """
  78 
  79   @deprecation.deprecated(
  80       "2018-10-01",
  81       "The TensorFlow Distributions library has moved to "
  82       "TensorFlow Probability "
  83       "(https://github.com/tensorflow/probability). You "
  84       "should update all references to use `tfp.distributions` "
  85       "instead of `tf.contrib.distributions`.",
  86       warn_once=True)
  87   def __init__(self, permutation, validate_args=False, name=None):
  88     """Creates the `Permute` bijector.
  89 
  90     Args:
  91       permutation: An `int`-like vector-shaped `Tensor` representing the
  92         permutation to apply to the rightmost dimension of the transformed
  93         `Tensor`.
  94       validate_args: Python `bool` indicating whether arguments should be
  95         checked for correctness.
  96       name: Python `str`, name given to ops managed by this object.
  97 
  98     Raises:
  99       TypeError: if `not permutation.dtype.is_integer`.
 100       ValueError: if `permutation` does not contain exactly one of each of
 101         `{0, 1, ..., d}`.
 102     """
 103     with ops.name_scope(name, "permute", values=[permutation]):
 104       permutation = ops.convert_to_tensor(
 105           permutation,
 106           name="permutation")
 107       if not permutation.dtype.is_integer:
 108         raise TypeError("permutation.dtype ({}) should be `int`-like.".format(
 109             permutation.dtype.name))
 110       p = tensor_util.constant_value(permutation)
 111       if p is not None:
 112         if set(p) != set(np.arange(p.size)):
 113           raise ValueError("Permutation over `d` must contain exactly one of "
 114                            "each of `{0, 1, ..., d}`.")
 115       elif validate_args:
 116         p, _ = nn_ops.top_k(-permutation,
 117                             k=array_ops.shape(permutation)[-1],
 118                             sorted=True)
 119         permutation = control_flow_ops.with_dependencies([
 120             check_ops.assert_equal(
 121                 -p, math_ops.range(array_ops.size(p)),
 122                 message=("Permutation over `d` must contain exactly one of "
 123                          "each of `{0, 1, ..., d}`.")),
 124         ], permutation)
 125       self._permutation = permutation
 126       super(Permute, self).__init__(
 127           forward_min_event_ndims=1,
 128           is_constant_jacobian=True,
 129           validate_args=validate_args,
 130           name=name or "permute")
 131 
 132   @property
 133   def permutation(self):
 134     return self._permutation
 135 
 136   def _forward(self, x):
 137     return array_ops.gather(x, self.permutation, axis=-1)
 138 
 139   def _inverse(self, y):
 140     return array_ops.gather(
 141         y,
 142         array_ops.invert_permutation(self.permutation),
 143         axis=-1)
 144 
 145   def _inverse_log_det_jacobian(self, y):
 146     # is_constant_jacobian = True for this bijector, hence the
 147     # `log_det_jacobian` need only be specified for a single input, as this will
 148     # be tiled to match `event_ndims`.
 149     return constant_op.constant(0., dtype=y.dtype.base_dtype)
 150 
 151   def _forward_log_det_jacobian(self, x):
 152     return constant_op.constant(0., dtype=x.dtype.base_dtype)
