  38 class RealNVP(bijector.Bijector):
  39   """RealNVP "affine coupling layer" for vector-valued events.
  40 
  41   Real NVP models a normalizing flow on a `D`-dimensional distribution via a
  42   single `D-d`-dimensional conditional distribution [(Dinh et al., 2017)][1]:
  43 
  44   `y[d:D] = y[d:D] * math_ops.exp(log_scale_fn(y[d:D])) + shift_fn(y[d:D])`
  45   `y[0:d] = x[0:d]`
  46 
  47   The last `D-d` units are scaled and shifted based on the first `d` units only,
  48   while the first `d` units are 'masked' and left unchanged. Real NVP's
  49   `shift_and_log_scale_fn` computes vector-valued quantities. For
  50   scale-and-shift transforms that do not depend on any masked units, i.e.
  51   `d=0`, use the `tfb.Affine` bijector with learned parameters instead.
  52 
  53   Masking is currently only supported for base distributions with
  54   `event_ndims=1`. For more sophisticated masking schemes like checkerboard or
  55   channel-wise masking [(Papamakarios et al., 2016)[4], use the `tfb.Permute`
  56   bijector to re-order desired masked units into the first `d` units. For base
  57   distributions with `event_ndims > 1`, use the `tfb.Reshape` bijector to
  58   flatten the event shape.
  59 
  60   Recall that the MAF bijector [(Papamakarios et al., 2016)][4] implements a
  61   normalizing flow via an autoregressive transformation. MAF and IAF have
  62   opposite computational tradeoffs - MAF can train all units in parallel but
  63   must sample units sequentially, while IAF must train units sequentially but
  64   can sample in parallel. In contrast, Real NVP can compute both forward and
  65   inverse computations in parallel. However, the lack of an autoregressive
  66   transformations makes it less expressive on a per-bijector basis.
  67 
  68   A "valid" `shift_and_log_scale_fn` must compute each `shift` (aka `loc` or
  69   "mu" in [Papamakarios et al. (2016)][4]) and `log(scale)` (aka "alpha" in
  70   [Papamakarios et al. (2016)][4]) such that each are broadcastable with the
  71   arguments to `forward` and `inverse`, i.e., such that the calculations in
  72   `forward`, `inverse` [below] are possible. For convenience,
  73   `real_nvp_default_nvp` is offered as a possible `shift_and_log_scale_fn`
  74   function.
  75 
  76   NICE [(Dinh et al., 2014)][2] is a special case of the Real NVP bijector
  77   which discards the scale transformation, resulting in a constant-time
  78   inverse-log-determinant-Jacobian. To use a NICE bijector instead of Real
  79   NVP, `shift_and_log_scale_fn` should return `(shift, None)`, and
  80   `is_constant_jacobian` should be set to `True` in the `RealNVP` constructor.
  81   Calling `real_nvp_default_template` with `shift_only=True` returns one such
  82   NICE-compatible `shift_and_log_scale_fn`.
  83 
  84   Caching: the scalar input depth `D` of the base distribution is not known at
  85   construction time. The first call to any of `forward(x)`, `inverse(x)`,
  86   `inverse_log_det_jacobian(x)`, or `forward_log_det_jacobian(x)` memoizes
  87   `D`, which is re-used in subsequent calls. This shape must be known prior to
  88   graph execution (which is the case if using tf.layers).
  89 
  90   #### Example Use
  91 
  92   ```python
  93   import tensorflow_probability as tfp
  94   tfd = tfp.distributions
  95   tfb = tfp.bijectors
  96 
  97   # A common choice for a normalizing flow is to use a Gaussian for the base
  98   # distribution. (However, any continuous distribution would work.) E.g.,
  99   nvp = tfd.TransformedDistribution(
 100       distribution=tfd.MultivariateNormalDiag(loc=[0., 0., 0.])),
 101       bijector=tfb.RealNVP(
 102           num_masked=2,
 103           shift_and_log_scale_fn=tfb.real_nvp_default_template(
 104               hidden_layers=[512, 512])))
 105 
 106   x = nvp.sample()
 107   nvp.log_prob(x)
 108   nvp.log_prob(0.)
 109   ```
 110 
 111   For more examples, see [Jang (2018)][3].
 112

 131   @deprecation.deprecated(
 132       "2018-10-01",

 137       "instead of `tf.contrib.distributions`.",
 138       warn_once=True)
 139   def __init__(self,
 140                num_masked,
 141                shift_and_log_scale_fn,
 142                is_constant_jacobian=False,
 143                validate_args=False,
 144                name=None):
 145     """Creates the Real NVP or NICE bijector.
 146 
 147     Args:
 148       num_masked: Python `int` indicating that the first `d` units of the event
 149         should be masked. Must be in the closed interval `[1, D-1]`, where `D`
 150         is the event size of the base distribution.
 151       shift_and_log_scale_fn: Python `callable` which computes `shift` and
 152         `log_scale` from both the forward domain (`x`) and the inverse domain
 153         (`y`). Calculation must respect the "autoregressive property" (see class
 154         docstring). Suggested default
 155         `masked_autoregressive_default_template(hidden_layers=...)`.
 156         Typically the function contains `tf.Variables` and is wrapped using
 157         `tf.make_template`. Returning `None` for either (both) `shift`,
 158         `log_scale` is equivalent to (but more efficient than) returning zero.
 159       is_constant_jacobian: Python `bool`. Default: `False`. When `True` the
 160         implementation assumes `log_scale` does not depend on the forward domain
 161         (`x`) or inverse domain (`y`) values. (No validation is made;
 162         `is_constant_jacobian=False` is always safe but possibly computationally
 163         inefficient.)
 164       validate_args: Python `bool` indicating whether arguments should be
 165         checked for correctness.
 166       name: Python `str`, name given to ops managed by this object.
 167 
 168     Raises:
 169       ValueError: If num_masked < 1.
 170     """
 171     name = name or "real_nvp"
 172     if num_masked <= 0:
 173       raise ValueError("num_masked must be a positive integer.")
 174     self._num_masked = num_masked
 175     # At construction time, we don't know input_depth.
 176     self._input_depth = None
 177     self._shift_and_log_scale_fn = shift_and_log_scale_fn
 178     super(RealNVP, self).__init__(
 179         forward_min_event_ndims=1,
 180         is_constant_jacobian=is_constant_jacobian,
 181         validate_args=validate_args,
 182         name=name)
 183 
 184   def _cache_input_depth(self, x):
 185     if self._input_depth is None:
 186       self._input_depth = x.shape.with_rank_at_least(1)[-1].value
 187       if self._input_depth is None:
 188         raise NotImplementedError(
 189             "Rightmost dimension must be known prior to graph execution.")
 190       if self._num_masked >= self._input_depth:
 191         raise ValueError(
 192             "Number of masked units must be smaller than the event size.")
 193 
 194   def _forward(self, x):
 195     self._cache_input_depth(x)
 196     # Performs scale and shift.
 197     x0, x1 = x[:, :self._num_masked], x[:, self._num_masked:]
 198     shift, log_scale = self._shift_and_log_scale_fn(
 199         x0, self._input_depth - self._num_masked)
 200     y1 = x1
 201     if log_scale is not None:
 202       y1 *= math_ops.exp(log_scale)
 203     if shift is not None:
 204       y1 += shift
 205     y = array_ops.concat([x0, y1], axis=-1)
 206     return y
 207 
 208   def _inverse(self, y):
 209     self._cache_input_depth(y)
 210     # Performs un-shift and un-scale.
 211     y0, y1 = y[:, :self._num_masked], y[:, self._num_masked:]
 212     shift, log_scale = self._shift_and_log_scale_fn(
 213         y0, self._input_depth - self._num_masked)
 214     x1 = y1
 215     if shift is not None:
 216       x1 -= shift
 217     if log_scale is not None:
 218       x1 *= math_ops.exp(-log_scale)
 219     x = array_ops.concat([y0, x1], axis=-1)
 220     return x
 221 
 222   def _inverse_log_det_jacobian(self, y):
 223     self._cache_input_depth(y)
 224     y0 = y[:, :self._num_masked]
 225     _, log_scale = self._shift_and_log_scale_fn(
 226         y0, self._input_depth - self._num_masked)
 227     if log_scale is None:
 228       return constant_op.constant(0., dtype=y.dtype, name="ildj")
 229     return -math_ops.reduce_sum(log_scale, axis=-1)
 230 
 231   def _forward_log_det_jacobian(self, x):
 232     self._cache_input_depth(x)
 233     x0 = x[:, :self._num_masked]
 234     _, log_scale = self._shift_and_log_scale_fn(
 235         x0, self._input_depth - self._num_masked)
 236     if log_scale is None:
 237       return constant_op.constant(0., dtype=x.dtype, name="fldj")
 238     return math_ops.reduce_sum(log_scale, axis=-1)

 241 @deprecation.deprecated(
 242     "2018-10-01",
 243     "The TensorFlow Distributions library has moved to "
 244     "TensorFlow Probability "
 245     "(https://github.com/tensorflow/probability). You "
 246     "should update all references to use `tfp.distributions` "
 247     "instead of `tf.contrib.distributions`.",
 248     warn_once=True)
 249 def real_nvp_default_template(
 250     hidden_layers,
 251     shift_only=False,
 252     activation=nn_ops.relu,
 253     name=None,
 254     *args,
 255     **kwargs):
 256   """Build a scale-and-shift function using a multi-layer neural network.
 257 
 258   This will be wrapped in a make_template to ensure the variables are only
 259   created once. It takes the `d`-dimensional input x[0:d] and returns the `D-d`
 260   dimensional outputs `loc` ("mu") and `log_scale` ("alpha").
 261 
 262   Arguments:
 263     hidden_layers: Python `list`-like of non-negative integer, scalars
 264       indicating the number of units in each hidden layer. Default: `[512, 512].
 265     shift_only: Python `bool` indicating if only the `shift` term shall be
 266       computed (i.e. NICE bijector). Default: `False`.
 267     activation: Activation function (callable). Explicitly setting to `None`
 268       implies a linear activation.
 269     name: A name for ops managed by this function. Default:
 270       "real_nvp_default_template".
 271     *args: `tf.layers.dense` arguments.
 272     **kwargs: `tf.layers.dense` keyword arguments.
 273 
 274   Returns:
 275     shift: `Float`-like `Tensor` of shift terms ("mu" in
 276       [Papamakarios et al.  (2016)][1]).
 277     log_scale: `Float`-like `Tensor` of log(scale) terms ("alpha" in
 278       [Papamakarios et al. (2016)][1]).
 279 
 280   Raises:
 281     NotImplementedError: if rightmost dimension of `inputs` is unknown prior to
 282       graph execution.
 283 
 284   #### References
 285 
 286   [1]: George Papamakarios, Theo Pavlakou, and Iain Murray. Masked
 287        Autoregressive Flow for Density Estimation. In _Neural Information
 288        Processing Systems_, 2017. https://arxiv.org/abs/1705.07057
 289   """
 290 
 291   with ops.name_scope(name, "real_nvp_default_template"):
 292     def _fn(x, output_units):
 293       """Fully connected MLP parameterized via `real_nvp_template`."""
 294       for units in hidden_layers:
 295         x = layers.dense(
 296             inputs=x,
 297             units=units,
 298             activation=activation,
 299             *args,
 300             **kwargs)
 301       x = layers.dense(
 302           inputs=x,
 303           units=(1 if shift_only else 2) * output_units,
 304           activation=None,
 305           *args,
 306           **kwargs)
 307       if shift_only:
 308         return x, None
 309       shift, log_scale = array_ops.split(x, 2, axis=-1)
 310       return shift, log_scale
 311     return template_ops.make_template(
 312         "real_nvp_default_template", _fn)
