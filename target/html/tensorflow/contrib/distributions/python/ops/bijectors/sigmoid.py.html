  32 class Sigmoid(bijector.Bijector):
  33   """Bijector which computes `Y = g(X) = 1 / (1 + exp(-X))`."""
  34 
  35   @deprecation.deprecated(
  36       "2018-10-01",
  37       "The TensorFlow Distributions library has moved to "
  38       "TensorFlow Probability "
  39       "(https://github.com/tensorflow/probability). You "
  40       "should update all references to use `tfp.distributions` "
  41       "instead of `tf.contrib.distributions`.",
  42       warn_once=True)
  43   def __init__(self, validate_args=False, name="sigmoid"):
  44     super(Sigmoid, self).__init__(
  45         forward_min_event_ndims=0,
  46         validate_args=validate_args,
  47         name=name)
  48 
  49   def _forward(self, x):
  50     return math_ops.sigmoid(x)
  51 
  52   def _inverse(self, y):
  53     return math_ops.log(y) - math_ops.log1p(-y)
  54 
  55   def _inverse_log_det_jacobian(self, y):
  56     return -math_ops.log(y) - math_ops.log1p(-y)
  57 
  58   def _forward_log_det_jacobian(self, x):
  59     return -nn_ops.softplus(-x) - nn_ops.softplus(x)
