  30 class Exp(power_transform.PowerTransform):
  31   """Compute `Y = g(X) = exp(X)`.
  32 
  33     Example Use:
  34 
  35     ```python
  36     # Create the Y=g(X)=exp(X) transform which works only on Tensors with 1
  37     # batch ndim 2.
  38     exp = Exp()
  39     x = [[[1., 2],
  40            [3, 4]],
  41           [[5, 6],
  42            [7, 8]]]
  43     exp(x) == exp.forward(x)
  44     log(x) == exp.inverse(x)
  45     ```
  46 
  47     Note: the exp(.) is applied element-wise but the Jacobian is a reduction
  48     over the event space.
  49   """
  50 
  51   @deprecation.deprecated(
  52       "2018-10-01",
  53       "The TensorFlow Distributions library has moved to "
  54       "TensorFlow Probability "
  55       "(https://github.com/tensorflow/probability). You "
  56       "should update all references to use `tfp.distributions` "
  57       "instead of `tf.contrib.distributions`.",
  58       warn_once=True)
  59   def __init__(self,
  60                validate_args=False,
  61                name="exp"):
  62     """Instantiates the `Exp` bijector.
  63 
  64     Args:
  65       validate_args: Python `bool` indicating whether arguments should be
  66         checked for correctness.
  67       name: Python `str` name given to ops managed by this object.
  68     """
  69     # forward_min_event_ndims = 0.
  70     # No forward_min_event_ndims specified as this is done in PowerTransform.
  71     super(Exp, self).__init__(
  72         validate_args=validate_args,
  73         name=name)
