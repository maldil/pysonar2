  33 class Softsign(bijector.Bijector):
  34   """Bijector which computes `Y = g(X) = X / (1 + |X|)`.
  35 
  36   The softsign `Bijector` has the following two useful properties:
  37 
  38   * The domain is all real numbers
  39   * `softsign(x) approx sgn(x)`, for large `|x|`.
  40 
  41   #### Examples
  42 
  43   ```python
  44   # Create the Y = softsign(X) transform.
  45   softsign = Softsign()
  46   x = [[[1., 2],
  47         [3, 4]],
  48        [[5, 6],
  49         [7, 8]]]
  50   x / (1 + abs(x)) == softsign.forward(x)
  51   x / (1 - abs(x)) == softsign.inverse(x)
  52   ```
  53   """
  54 
  55   @deprecation.deprecated(
  56       "2018-10-01",
  57       "The TensorFlow Distributions library has moved to "
  58       "TensorFlow Probability "
  59       "(https://github.com/tensorflow/probability). You "
  60       "should update all references to use `tfp.distributions` "
  61       "instead of `tf.contrib.distributions`.",
  62       warn_once=True)
  63   def __init__(self, validate_args=False, name="softsign"):
  64     super(Softsign, self).__init__(
  65         forward_min_event_ndims=0,
  66         validate_args=validate_args,
  67         name=name)
  68 
  69   def _forward(self, x):
  70     return x / (1. + math_ops.abs(x))
  71 
  72   def _inverse(self, y):
  73     y = self._maybe_assert_valid_y(y)
  74     return y / (1. - math_ops.abs(y))
  75 
  76   def _forward_log_det_jacobian(self, x):
  77     return -2. * math_ops.log1p(math_ops.abs(x))
  78 
  79   def _inverse_log_det_jacobian(self, y):
  80     y = self._maybe_assert_valid_y(y)
  81     return -2. * math_ops.log1p(-math_ops.abs(y))
  82 
  83   def _maybe_assert_valid_y(self, y):
  84     if not self.validate_args:
  85       return y
  86     is_valid = [
  87         check_ops.assert_greater(
  88             y, math_ops.cast(-1., dtype=y.dtype.base_dtype),
  89             message="Inverse transformation input must be greater than -1."),
  90         check_ops.assert_less(
  91             y, math_ops.cast(1., dtype=y.dtype.base_dtype),
  92             message="Inverse transformation input must be less than 1.")
  93     ]
  94 
  95     return control_flow_ops.with_dependencies(is_valid, y)
