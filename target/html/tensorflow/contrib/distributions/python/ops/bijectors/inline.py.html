  30 class Inline(bijector.Bijector):
  31   """Bijector constructed from custom callables.
  32 
  33   Example Use:
  34 
  35   ```python
  36   exp = Inline(
  37     forward_fn=tf.exp,
  38     inverse_fn=tf.log,
  39     inverse_log_det_jacobian_fn=(
  40       lambda y: -tf.reduce_sum(tf.log(y), axis=-1)),
  41     name="exp")
  42   ```
  43 
  44   The above example is equivalent to the `Bijector` `Exp()`.
  45   """
  46 
  47   @deprecation.deprecated(
  48       "2018-10-01",
  49       "The TensorFlow Distributions library has moved to "
  50       "TensorFlow Probability "
  51       "(https://github.com/tensorflow/probability). You "
  52       "should update all references to use `tfp.distributions` "
  53       "instead of `tf.contrib.distributions`.",
  54       warn_once=True)
  55   def __init__(self,
  56                forward_fn=None,
  57                inverse_fn=None,
  58                inverse_log_det_jacobian_fn=None,
  59                forward_log_det_jacobian_fn=None,
  60                forward_event_shape_fn=None,
  61                forward_event_shape_tensor_fn=None,
  62                inverse_event_shape_fn=None,
  63                inverse_event_shape_tensor_fn=None,
  64                is_constant_jacobian=False,
  65                validate_args=False,
  66                forward_min_event_ndims=None,
  67                inverse_min_event_ndims=None,
  68                name="inline"):
  69     """Creates a `Bijector` from callables.
  70 
  71     Args:
  72       forward_fn: Python callable implementing the forward transformation.
  73       inverse_fn: Python callable implementing the inverse transformation.
  74       inverse_log_det_jacobian_fn: Python callable implementing the
  75         log o det o jacobian of the inverse transformation.
  76       forward_log_det_jacobian_fn: Python callable implementing the
  77         log o det o jacobian of the forward transformation.
  78       forward_event_shape_fn: Python callable implementing non-identical
  79         static event shape changes. Default: shape is assumed unchanged.
  80       forward_event_shape_tensor_fn: Python callable implementing non-identical
  81         event shape changes. Default: shape is assumed unchanged.
  82       inverse_event_shape_fn: Python callable implementing non-identical
  83         static event shape changes. Default: shape is assumed unchanged.
  84       inverse_event_shape_tensor_fn: Python callable implementing non-identical
  85         event shape changes. Default: shape is assumed unchanged.
  86       is_constant_jacobian: Python `bool` indicating that the Jacobian is
  87         constant for all input arguments.
  88       validate_args: Python `bool` indicating whether arguments should be
  89         checked for correctness.
  90       forward_min_event_ndims: Python `int` indicating the minimal
  91         dimensionality this bijector acts on.
  92       inverse_min_event_ndims: Python `int` indicating the minimal
  93         dimensionality this bijector acts on.
  94       name: Python `str`, name given to ops managed by this object.
  95     """
  96     super(Inline, self).__init__(
  97         forward_min_event_ndims=forward_min_event_ndims,
  98         inverse_min_event_ndims=inverse_min_event_ndims,
  99         is_constant_jacobian=is_constant_jacobian,
 100         validate_args=validate_args,
 101         name=name)
 102     self._forward_fn = forward_fn
 103     self._inverse_fn = inverse_fn
 104     self._inverse_log_det_jacobian_fn = inverse_log_det_jacobian_fn
 105     self._forward_log_det_jacobian_fn = forward_log_det_jacobian_fn
 106     self._forward_event_shape_fn = forward_event_shape_fn
 107     self._forward_event_shape_tensor_fn = forward_event_shape_tensor_fn
 108     self._inverse_event_shape_fn = inverse_event_shape_fn
 109     self._inverse_event_shape_tensor_fn = inverse_event_shape_tensor_fn
 110 
 111   def _forward_event_shape(self, input_shape):
 112     if self._forward_event_shape_fn is None:
 113       # By default assume shape doesn't change.
 114       return input_shape
 115     return self._forward_event_shape_fn(input_shape)
 116 
 117   def _forward_event_shape_tensor(self, input_shape):
 118     if self._forward_event_shape_tensor_fn is None:
 119       # By default assume shape doesn't change.
 120       return input_shape
 121     return self._forward_event_shape_tensor_fn(input_shape)
 122 
 123   def _inverse_event_shape(self, output_shape):
 124     if self._inverse_event_shape_fn is None:
 125       # By default assume shape doesn't change.
 126       return output_shape
 127     return self._inverse_event_shape_fn(output_shape)
 128 
 129   def _inverse_event_shape_tensor(self, output_shape):
 130     if self._inverse_event_shape_tensor_fn is None:
 131       # By default assume shape doesn't change.
 132       return output_shape
 133     return self._inverse_event_shape_tensor_fn(output_shape)
 134 
 135   def _forward(self, x, **kwargs):
 136     if not callable(self._forward_fn):
 137       raise NotImplementedError(
 138           "forward_fn is not a callable function.")
 139     return self._forward_fn(x, **kwargs)
 140 
 141   def _inverse(self, y, **kwargs):
 142     if not callable(self._inverse_fn):
 143       raise NotImplementedError(
 144           "inverse_fn is not a callable function.")
 145     return self._inverse_fn(y, **kwargs)
 146 
 147   def _inverse_log_det_jacobian(self, y, **kwargs):
 148     if not callable(self._inverse_log_det_jacobian_fn):
 149       raise NotImplementedError(
 150           "inverse_log_det_jacobian_fn is not a callable function.")
 151     return self._inverse_log_det_jacobian_fn(y, **kwargs)
 152 
 153   def _forward_log_det_jacobian(self, x, **kwargs):
 154     if not callable(self._forward_log_det_jacobian_fn):
 155       raise NotImplementedError(
 156           "forward_log_det_jacobian_fn is not a callable function.")
 157     return self._forward_log_det_jacobian_fn(x, **kwargs)
