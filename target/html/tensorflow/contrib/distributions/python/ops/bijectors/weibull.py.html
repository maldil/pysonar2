  35 class Weibull(bijector.Bijector):
  36   """Compute `Y = g(X) = 1 - exp((-X / scale) ** concentration), X >= 0`.
  37 
  38   This bijector maps inputs from `[0, inf]` to [0, 1]`. The inverse of the
  39   bijector applied to a uniform random variable `X ~ U(0, 1) gives back a
  40   random variable with the
  41   [Weibull distribution](https://en.wikipedia.org/wiki/Weibull_distribution):
  42 
  43   ```none
  44   Y ~ Weibull(scale, concentration)
  45   pdf(y; scale, concentration, y >= 0) = (scale / concentration) * (
  46     scale / concentration) ** (concentration - 1) * exp(
  47       -(y / scale) ** concentration)
  48   ```
  49   """
  50 
  51   @deprecation.deprecated(
  52       "2018-10-01",
  53       "The TensorFlow Distributions library has moved to "
  54       "TensorFlow Probability "
  55       "(https://github.com/tensorflow/probability). You "
  56       "should update all references to use `tfp.distributions` "
  57       "instead of `tf.contrib.distributions`.",
  58       warn_once=True)
  59   def __init__(self,
  60                scale=1.,
  61                concentration=1.,
  62                validate_args=False,
  63                name="weibull"):
  64     """Instantiates the `Weibull` bijector.
  65 
  66     Args:
  67       scale: Positive Float-type `Tensor` that is the same dtype and is
  68         broadcastable with `concentration`.
  69         This is `l` in `Y = g(X) = 1 - exp((-x / l) ** k)`.
  70       concentration: Positive Float-type `Tensor` that is the same dtype and is
  71         broadcastable with `scale`.
  72         This is `k` in `Y = g(X) = 1 - exp((-x / l) ** k)`.
  73       validate_args: Python `bool` indicating whether arguments should be
  74         checked for correctness.
  75       name: Python `str` name given to ops managed by this object.
  76     """
  77     self._graph_parents = []
  78     self._name = name
  79     self._validate_args = validate_args
  80     with self._name_scope("init", values=[scale, concentration]):
  81       self._scale = ops.convert_to_tensor(scale, name="scale")
  82       self._concentration = ops.convert_to_tensor(
  83           concentration, name="concentration")
  84       check_ops.assert_same_float_dtype([self._scale, self._concentration])
  85       if validate_args:
  86         self._scale = control_flow_ops.with_dependencies([
  87             check_ops.assert_positive(
  88                 self._scale,
  89                 message="Argument scale was not positive")
  90         ], self._scale)
  91         self._concentration = control_flow_ops.with_dependencies([
  92             check_ops.assert_positive(
  93                 self._concentration,
  94                 message="Argument concentration was not positive")
  95         ], self._concentration)
  96 
  97     super(Weibull, self).__init__(
  98         forward_min_event_ndims=0,
  99         validate_args=validate_args,
 100         name=name)
 101 
 102   @property
 103   def scale(self):
 104     """The `l` in `Y = g(X) = 1 - exp((-x / l) ** k)`."""
 105     return self._scale
 106 
 107   @property
 108   def concentration(self):
 109     """The `k` in `Y = g(X) = 1 - exp((-x / l) ** k)`."""
 110     return self._concentration
 111 
 112   def _forward(self, x):
 113     x = self._maybe_assert_valid_x(x)
 114     return -math_ops.expm1(-((x / self.scale) ** self.concentration))
 115 
 116   def _inverse(self, y):
 117     y = self._maybe_assert_valid_y(y)
 118     return self.scale * (-math_ops.log1p(-y)) ** (1 / self.concentration)
 119 
 120   def _inverse_log_det_jacobian(self, y):
 121     y = self._maybe_assert_valid_y(y)
 122     return (
 123         -math_ops.log1p(-y) +
 124         (1 / self.concentration - 1) * math_ops.log(-math_ops.log1p(-y)) +
 125         math_ops.log(self.scale / self.concentration))
 126 
 127   def _forward_log_det_jacobian(self, x):
 128     x = self._maybe_assert_valid_x(x)
 129     return (
 130         -(x / self.scale) ** self.concentration +
 131         (self.concentration - 1) * math_ops.log(x) +
 132         math_ops.log(self.concentration) +
 133         -self.concentration * math_ops.log(self.scale))
 134 
 135   def _maybe_assert_valid_x(self, x):
 136     if not self.validate_args:
 137       return x
 138     is_valid = check_ops.assert_non_negative(
 139         x,
 140         message="Forward transformation input must be at least 0.")
 141     return control_flow_ops.with_dependencies([is_valid], x)
 142 
 143   def _maybe_assert_valid_y(self, y):
 144     if not self.validate_args:
 145       return y
 146     is_positive = check_ops.assert_non_negative(
 147         y, message="Inverse transformation input must be greater than 0.")
 148     less_than_one = check_ops.assert_less_equal(
 149         y, constant_op.constant(1., y.dtype),
 150         message="Inverse transformation input must be less than or equal to 1.")
 151     return control_flow_ops.with_dependencies([is_positive, less_than_one], y)
