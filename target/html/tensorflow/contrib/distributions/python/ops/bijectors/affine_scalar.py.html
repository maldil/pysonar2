  36 class AffineScalar(bijector.Bijector):
  37   """Compute `Y = g(X; shift, scale) = scale * X + shift`.
  38 
  39   Examples:
  40 
  41   ```python
  42   # Y = X
  43   b = AffineScalar()
  44 
  45   # Y = X + shift
  46   b = AffineScalar(shift=[1., 2, 3])
  47 
  48   # Y = 2 * X + shift
  49   b = AffineScalar(
  50     shift=[1., 2, 3],
  51     scale=2.)
  52   ```
  53 
  54   """
  55 
  56   @deprecation.deprecated(
  57       "2018-10-01",
  58       "The TensorFlow Distributions library has moved to "
  59       "TensorFlow Probability "
  60       "(https://github.com/tensorflow/probability). You "
  61       "should update all references to use `tfp.distributions` "
  62       "instead of `tf.contrib.distributions`.",
  63       warn_once=True)
  64   def __init__(self,
  65                shift=None,
  66                scale=None,
  67                validate_args=False,
  68                name="affine_scalar"):
  69     """Instantiates the `AffineScalar` bijector.
  70 
  71     This `Bijector` is initialized with `shift` `Tensor` and `scale` arguments,
  72     giving the forward operation:
  73 
  74     ```none
  75     Y = g(X) = scale * X + shift
  76     ```
  77 
  78     if `scale` is not specified, then the bijector has the semantics of
  79     `scale = 1.`. Similarly, if `shift` is not specified, then the bijector
  80     has the semantics of `shift = 0.`.
  81 
  82     Args:
  83       shift: Floating-point `Tensor`. If this is set to `None`, no shift is
  84         applied.
  85       scale: Floating-point `Tensor`. If this is set to `None`, no scale is
  86         applied.
  87       validate_args: Python `bool` indicating whether arguments should be
  88         checked for correctness.
  89       name: Python `str` name given to ops managed by this object.
  90     """
  91     self._graph_parents = []
  92     self._name = name
  93     self._validate_args = validate_args
  94 
  95     with self._name_scope("init", values=[scale, shift]):
  96       self._shift = shift
  97       self._scale = scale
  98 
  99       if self._shift is not None:
 100         self._shift = ops.convert_to_tensor(shift, name="shift")
 101 
 102       if self._scale is not None:
 103         self._scale = ops.convert_to_tensor(self._scale, name="scale")
 104         if validate_args:
 105           self._scale = control_flow_ops.with_dependencies(
 106               [check_ops.assert_none_equal(
 107                   self._scale,
 108                   array_ops.zeros([], dtype=self._scale.dtype))],
 109               self._scale)
 110 
 111       super(AffineScalar, self).__init__(
 112           forward_min_event_ndims=0,
 113           is_constant_jacobian=True,
 114           validate_args=validate_args,
 115           name=name)
 116 
 117   @property
 118   def shift(self):
 119     """The `shift` `Tensor` in `Y = scale @ X + shift`."""
 120     return self._shift
 121 
 122   @property
 123   def scale(self):
 124     """The `scale` `LinearOperator` in `Y = scale @ X + shift`."""
 125     return self._scale
 126 
 127   def _forward(self, x):
 128     y = array_ops.identity(x)
 129     if self.scale is not None:
 130       y *= self.scale
 131     if self.shift is not None:
 132       y += self.shift
 133     return y
 134 
 135   def _inverse(self, y):
 136     x = array_ops.identity(y)
 137     if self.shift is not None:
 138       x -= self.shift
 139     if self.scale is not None:
 140       x /= self.scale
 141     return x
 142 
 143   def _forward_log_det_jacobian(self, x):
 144     # is_constant_jacobian = True for this bijector, hence the
 145     # `log_det_jacobian` need only be specified for a single input, as this will
 146     # be tiled to match `event_ndims`.
 147     if self.scale is None:
 148       return constant_op.constant(0., dtype=x.dtype.base_dtype)
 149 
 150     return math_ops.log(math_ops.abs(self.scale))
