  36 class Softplus(bijector.Bijector):
  37   """Bijector which computes `Y = g(X) = Log[1 + exp(X)]`.
  38 
  39   The softplus `Bijector` has the following two useful properties:
  40 
  41   * The domain is the positive real numbers
  42   * `softplus(x) approx x`, for large `x`, so it does not overflow as easily as
  43     the `Exp` `Bijector`.
  44 
  45   The optional nonzero `hinge_softness` parameter changes the transition at
  46   zero.  With `hinge_softness = c`, the bijector is:
  47 
  48     ```f_c(x) := c * g(x / c) = c * Log[1 + exp(x / c)].```
  49 
  50   For large `x >> 1`, `c * Log[1 + exp(x / c)] approx c * Log[exp(x / c)] = x`,
  51   so the behavior for large `x` is the same as the standard softplus.
  52 
  53   As `c > 0` approaches 0 from the right, `f_c(x)` becomes less and less soft,
  54   approaching `max(0, x)`.
  55 
  56   * `c = 1` is the default.
  57   * `c > 0` but small means `f(x) approx ReLu(x) = max(0, x)`.
  58   * `c < 0` flips sign and reflects around the `y-axis`: `f_{-c}(x) = -f_c(-x)`.
  59   * `c = 0` results in a non-bijective transformation and triggers an exception.
  60 
  61     Example Use:
  62 
  63     ```python
  64     # Create the Y=g(X)=softplus(X) transform which works only on Tensors with 1
  65     # batch ndim and 2 event ndims (i.e., vector of matrices).
  66     softplus = Softplus()
  67     x = [[[1., 2],
  68           [3, 4]],
  69          [[5, 6],
  70           [7, 8]]]
  71     log(1 + exp(x)) == softplus.forward(x)
  72     log(exp(x) - 1) == softplus.inverse(x)
  73     ```
  74 
  75     Note: log(.) and exp(.) are applied element-wise but the Jacobian is a
  76     reduction over the event space.
  77   """
  78 
  79   @distribution_util.AppendDocstring(
  80       kwargs_dict={
  81           "hinge_softness": (
  82               "Nonzero floating point `Tensor`.  Controls the softness of what "
  83               "would otherwise be a kink at the origin.  Default is 1.0")})
  84   @deprecation.deprecated(
  85       "2018-10-01",
  86       "The TensorFlow Distributions library has moved to "
  87       "TensorFlow Probability "
  88       "(https://github.com/tensorflow/probability). You "
  89       "should update all references to use `tfp.distributions` "
  90       "instead of `tf.contrib.distributions`.",
  91       warn_once=True)
  92   def __init__(self,
  93                hinge_softness=None,
  94                validate_args=False,
  95                name="softplus"):
  96     with ops.name_scope(name, values=[hinge_softness]):
  97       if hinge_softness is not None:
  98         self._hinge_softness = ops.convert_to_tensor(
  99             hinge_softness, name="hinge_softness")
 100       else:
 101         self._hinge_softness = None
 102       if validate_args:
 103         nonzero_check = check_ops.assert_none_equal(
 104             ops.convert_to_tensor(
 105                 0, dtype=self.hinge_softness.dtype),
 106             self.hinge_softness,
 107             message="hinge_softness must be non-zero")
 108         self._hinge_softness = control_flow_ops.with_dependencies(
 109             [nonzero_check], self.hinge_softness)
 110 
 111     super(Softplus, self).__init__(
 112         forward_min_event_ndims=0,
 113         validate_args=validate_args,
 114         name=name)
 115 
 116   def _forward(self, x):
 117     if self.hinge_softness is None:
 118       return nn_ops.softplus(x)
 119     hinge_softness = math_ops.cast(self.hinge_softness, x.dtype)
 120     return hinge_softness * nn_ops.softplus(x / hinge_softness)
 121 
 122   def _inverse(self, y):
 123     if self.hinge_softness is None:
 124       return distribution_util.softplus_inverse(y)
 125     hinge_softness = math_ops.cast(self.hinge_softness, y.dtype)
 126     return hinge_softness * distribution_util.softplus_inverse(
 127         y / hinge_softness)
 128 
 129   def _inverse_log_det_jacobian(self, y):
 130     # Could also do:
 131     #   ildj = math_ops.reduce_sum(y - distribution_util.softplus_inverse(y),
 132     #                              axis=event_dims)
 133     # but the following is more numerically stable. Ie,
 134     # Y = Log[1 + exp{X}] ==> X = Log[exp{Y} - 1]
 135     # ==> dX/dY = exp{Y} / (exp{Y} - 1)
 136     #           = 1 / (1 - exp{-Y}),
 137     # which is the most stable for large Y > 0. For small Y, we use
 138     # 1 - exp{-Y} approx Y.
 139     if self.hinge_softness is not None:
 140       y /= math_ops.cast(self.hinge_softness, y.dtype)
 141     return -math_ops.log(-math_ops.expm1(-y))
 142 
 143   def _forward_log_det_jacobian(self, x):
 144     if self.hinge_softness is not None:
 145       x /= math_ops.cast(self.hinge_softness, x.dtype)
 146     return -nn_ops.softplus(-x)
 147 
 148   @property
 149   def hinge_softness(self):
 150     return self._hinge_softness
