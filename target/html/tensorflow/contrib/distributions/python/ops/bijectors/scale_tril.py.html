  33 class ScaleTriL(chain.Chain):
  34   """Transforms unconstrained vectors to TriL matrices with positive diagonal.
  35 
  36   This is implemented as a simple `tfb.Chain` of `tfb.FillTriangular`
  37   followed by `tfb.TransformDiagonal`, and provided mostly as a
  38   convenience. The default setup is somewhat opinionated, using a
  39   Softplus transformation followed by a small shift (`1e-5`) which
  40   attempts to avoid numerical issues from zeros on the diagonal.
  41 
  42   #### Examples
  43 
  44   ```python
  45   import tensorflow_probability as tfp
  46   tfd = tfp.distributions
  47   tfb = tfp.bijectors
  48 
  49   b = tfb.ScaleTriL(
  50        diag_bijector=tfb.Exp(),
  51        diag_shift=None)
  52   b.forward(x=[0., 0., 0.])
  53   # Result: [[1., 0.],
  54   #          [0., 1.]]
  55   b.inverse(y=[[1., 0],
  56                [.5, 2]])
  57   # Result: [log(2), .5, log(1)]
  58 
  59   # Define a distribution over PSD matrices of shape `[3, 3]`,
  60   # with `1 + 2 + 3 = 6` degrees of freedom.
  61   dist = tfd.TransformedDistribution(
  62           tfd.Normal(tf.zeros(6), tf.ones(6)),
  63           tfb.Chain([tfb.CholeskyOuterProduct(), tfb.ScaleTriL()]))
  64 
  65   # Using an identity transformation, ScaleTriL is equivalent to
  66   # tfb.FillTriangular.
  67   b = tfb.ScaleTriL(
  68        diag_bijector=tfb.Identity(),
  69        diag_shift=None)
  70 
  71   # For greater control over initialization, one can manually encode
  72   # pre- and post- shifts inside of `diag_bijector`.
  73   b = tfb.ScaleTriL(
  74        diag_bijector=tfb.Chain([
  75          tfb.AffineScalar(shift=1e-3),
  76          tfb.Softplus(),
  77          tfb.AffineScalar(shift=0.5413)]),  # softplus_inverse(1.)
  78                                             #  = log(expm1(1.)) = 0.5413
  79        diag_shift=None)
  80   ```
  81   """
  82 
  83   @deprecation.deprecated(
  84       "2018-10-01",
  85       "The TensorFlow Distributions library has moved to "
  86       "TensorFlow Probability "
  87       "(https://github.com/tensorflow/probability). You "
  88       "should update all references to use `tfp.distributions` "
  89       "instead of `tf.contrib.distributions`.",
  90       warn_once=True)
  91   def __init__(self,
  92                diag_bijector=None,
  93                diag_shift=1e-5,
  94                validate_args=False,
  95                name="scale_tril"):
  96     """Instantiates the `ScaleTriL` bijector.
  97 
  98     Args:
  99       diag_bijector: `Bijector` instance, used to transform the output diagonal
 100         to be positive.
 101         Default value: `None` (i.e., `tfb.Softplus()`).
 102       diag_shift: Float value broadcastable and added to all diagonal entries
 103         after applying the `diag_bijector`. Setting a positive
 104         value forces the output diagonal entries to be positive, but
 105         prevents inverting the transformation for matrices with
 106         diagonal entries less than this value.
 107         Default value: `1e-5` (i.e., no shift is applied).
 108       validate_args: Python `bool` indicating whether arguments should be
 109         checked for correctness.
 110         Default value: `False` (i.e., arguments are not validated).
 111       name: Python `str` name given to ops managed by this object.
 112         Default value: `scale_tril`.
 113     """
 114 
 115     if diag_bijector is None:
 116       diag_bijector = softplus.Softplus(validate_args=validate_args)
 117 
 118     if diag_shift is not None:
 119       diag_bijector = chain.Chain([affine_scalar.AffineScalar(shift=diag_shift),
 120                                    diag_bijector])
 121 
 122     super(ScaleTriL, self).__init__(
 123         [transform_diagonal.TransformDiagonal(diag_bijector=diag_bijector),
 124          fill_triangular.FillTriangular()],
 125         validate_args=validate_args,
 126         name=name)
