  36 class MatrixInverseTriL(bijector.Bijector):
  37   """Computes `g(L) = inv(L)`, where `L` is a lower-triangular matrix.
  38 
  39   `L` must be nonsingular; equivalently, all diagonal entries of `L` must be
  40   nonzero.
  41 
  42   The input must have `rank >= 2`.  The input is treated as a batch of matrices
  43   with batch shape `input.shape[:-2]`, where each matrix has dimensions
  44   `input.shape[-2]` by `input.shape[-1]` (hence `input.shape[-2]` must equal
  45   `input.shape[-1]`).
  46 
  47   #### Examples
  48 
  49   ```python
  50   tfd.bijectors.MatrixInverseTriL().forward(x=[[1., 0], [2, 1]])
  51   # Result: [[1., 0], [-2, 1]], i.e., inv(x)
  52 
  53   tfd.bijectors.MatrixInverseTriL().inverse(y=[[1., 0], [-2, 1]])
  54   # Result: [[1., 0], [2, 1]], i.e., inv(y).
  55   ```
  56 
  57   """
  58 
  59   @deprecation.deprecated(
  60       "2018-10-01",
  61       "The TensorFlow Distributions library has moved to "
  62       "TensorFlow Probability "
  63       "(https://github.com/tensorflow/probability). You "
  64       "should update all references to use `tfp.distributions` "
  65       "instead of `tf.contrib.distributions`.",
  66       warn_once=True)
  67   def __init__(self, validate_args=False, name="matrix_inverse_tril"):
  68     """Instantiates the `MatrixInverseTriL` bijector.
  69 
  70     Args:
  71       validate_args: Python `bool` indicating whether arguments should be
  72         checked for correctness.
  73       name: Python `str` name given to ops managed by this object.
  74     """
  75     self._graph_parents = []
  76     self._name = name
  77     super(MatrixInverseTriL, self).__init__(
  78         forward_min_event_ndims=2,
  79         validate_args=validate_args,
  80         name=name)
  81 
  82   def _forward(self, x):
  83     with ops.control_dependencies(self._assertions(x)):
  84       shape = array_ops.shape(x)
  85       return linalg_ops.matrix_triangular_solve(
  86           x, linalg_ops.eye(shape[-1], batch_shape=shape[:-2]), lower=True)
  87 
  88   def _inverse(self, y):
  89     return self._forward(y)
  90 
  91   def _forward_log_det_jacobian(self, x):
  92     # Calculation of the Jacobian:
  93     #
  94     # Let X = (x_{ij}), 0 <= i,j < n, be a matrix of indeterminates.  Let Z =
  95     # X^{-1} where Z = (z_{ij}).  Then
  96     #
  97     #     dZ/dx_{ij} = (d/dt | t=0) Y(t)^{-1},
  98     #
  99     # where Y(t) = X + t*E_{ij} and E_{ij} is the matrix with a 1 in the (i,j)
 100     # entry and zeros elsewhere.  By the product rule,
 101     #
 102     #     0 = d/dt [Identity matrix]
 103     #       = d/dt [Y Y^{-1}]
 104     #       = Y d/dt[Y^{-1}] + dY/dt Y^{-1}
 105     #
 106     # so
 107     #
 108     #     d/dt[Y^{-1}] = -Y^{-1} dY/dt Y^{-1}
 109     #                  = -Y^{-1} E_{ij} Y^{-1}.
 110     #
 111     # Evaluating at t=0,
 112     #
 113     #     dZ/dx_{ij} = -Z E_{ij} Z.
 114     #
 115     # Taking the (r,s) entry of each side,
 116     #
 117     #     dz_{rs}/dx_{ij} = -z_{ri}z_{sj}.
 118     #
 119     # Now, let J be the Jacobian dZ/dX, arranged as the n^2-by-n^2 matrix whose
 120     # (r*n + s, i*n + j) entry is dz_{rs}/dx_{ij}.  Considering J as an n-by-n
 121     # block matrix with n-by-n blocks, the above expression for dz_{rs}/dx_{ij}
 122     # shows that the block at position (r,i) is -z_{ri}Z.  Hence
 123     #
 124     #          J = -KroneckerProduct(Z, Z),
 125     #     det(J) = (-1)^(n^2) (det Z)^(2n)
 126     #            = (-1)^n (det X)^(-2n).
 127     with ops.control_dependencies(self._assertions(x)):
 128       return (-2. * math_ops.cast(array_ops.shape(x)[-1], x.dtype.base_dtype) *
 129               math_ops.reduce_sum(
 130                   math_ops.log(math_ops.abs(array_ops.matrix_diag_part(x))),
 131                   axis=-1))
 132 
 133   def _assertions(self, x):
 134     if not self.validate_args:
 135       return []
 136     shape = array_ops.shape(x)
 137     is_matrix = check_ops.assert_rank_at_least(
 138         x, 2, message="Input must have rank at least 2.")
 139     is_square = check_ops.assert_equal(
 140         shape[-2], shape[-1], message="Input must be a square matrix.")
 141     above_diagonal = array_ops.matrix_band_part(
 142         array_ops.matrix_set_diag(
 143             x, array_ops.zeros(shape[:-1], dtype=dtypes.float32)),
 144         0, -1)
 145     is_lower_triangular = check_ops.assert_equal(
 146         above_diagonal, array_ops.zeros_like(above_diagonal),
 147         message="Input must be lower triangular.")
 148     # A lower triangular matrix is nonsingular iff all its diagonal entries are
 149     # nonzero.
 150     diag_part = array_ops.matrix_diag_part(x)
 151     is_nonsingular = check_ops.assert_none_equal(
 152         diag_part, array_ops.zeros_like(diag_part),
 153         message="Input must have all diagonal entries nonzero.")
 154     return [is_matrix, is_square, is_lower_triangular, is_nonsingular]
