  37 class SoftmaxCentered(bijector.Bijector):
  38   """Bijector which computes `Y = g(X) = exp([X 0]) / sum(exp([X 0]))`.
  39 
  40   To implement [softmax](https://en.wikipedia.org/wiki/Softmax_function) as a
  41   bijection, the forward transformation appends a value to the input and the
  42   inverse removes this coordinate. The appended coordinate represents a pivot,
  43   e.g., `softmax(x) = exp(x-c) / sum(exp(x-c))` where `c` is the implicit last
  44   coordinate.
  45 
  46   Example Use:
  47 
  48   ```python
  49   bijector.SoftmaxCentered().forward(tf.log([2, 3, 4]))
  50   # Result: [0.2, 0.3, 0.4, 0.1]
  51   # Extra result: 0.1
  52 
  53   bijector.SoftmaxCentered().inverse([0.2, 0.3, 0.4, 0.1])
  54   # Result: tf.log([2, 3, 4])
  55   # Extra coordinate removed.
  56   ```
  57 
  58   At first blush it may seem like the [Invariance of domain](
  59   https://en.wikipedia.org/wiki/Invariance_of_domain) theorem implies this
  60   implementation is not a bijection. However, the appended dimension
  61   makes the (forward) image non-open and the theorem does not directly apply.
  62   """
  63 
  64   @deprecation.deprecated(
  65       "2018-10-01",
  66       "The TensorFlow Distributions library has moved to "
  67       "TensorFlow Probability "
  68       "(https://github.com/tensorflow/probability). You "

  73                validate_args=False,
  74                name="softmax_centered"):

  76     self._name = name

  78         forward_min_event_ndims=1,
  79         validate_args=validate_args,

  82   def _forward_event_shape(self, input_shape):
  83     if input_shape.ndims is None or input_shape[-1] is None:
  84       return input_shape
  85     return tensor_shape.TensorShape([input_shape[-1] + 1])
  86 
  87   def _forward_event_shape_tensor(self, input_shape):
  88     return (input_shape[-1] + 1)[..., array_ops.newaxis]
  89 
  90   def _inverse_event_shape(self, output_shape):
  91     if output_shape.ndims is None or output_shape[-1] is None:
  92       return output_shape
  93     if output_shape[-1] <= 1:
  94       raise ValueError("output_shape[-1] = %d <= 1" % output_shape[-1])
  95     return tensor_shape.TensorShape([output_shape[-1] - 1])
  96 
  97   def _inverse_event_shape_tensor(self, output_shape):
  98     if self.validate_args:
  99       # It is not possible for a negative shape so we need only check <= 1.
 100       is_greater_one = check_ops.assert_greater(
 101           output_shape[-1], 1, message="Need last dimension greater than 1.")
 102       output_shape = control_flow_ops.with_dependencies(
 103           [is_greater_one], output_shape)
 104     return (output_shape[-1] - 1)[..., array_ops.newaxis]
 105 
 106   def _forward(self, x):
 107     # Pad the last dim with a zeros vector. We need this because it lets us
 108     # infer the scale in the inverse function.
 109     y = distribution_util.pad(x, axis=-1, back=True)
 110 
 111     # Set shape hints.
 112     if x.shape.ndims is not None:
 113       shape = x.shape[:-1].concatenate(x.shape[-1] + 1)
 114       y.shape.assert_is_compatible_with(shape)
 115       y.set_shape(shape)
 116 
 117     return nn_ops.softmax(y)
 118 
 119   def _inverse(self, y):
 120     # To derive the inverse mapping note that:
 121     #   y[i] = exp(x[i]) / normalization
 122     # and
 123     #   y[end] = 1 / normalization.
 124     # Thus:
 125     # x[i] = log(exp(x[i])) - log(y[end]) - log(normalization)
 126     #      = log(exp(x[i])/normalization) - log(y[end])
 127     #      = log(y[i]) - log(y[end])
 128 
 129     # Do this first to make sure CSE catches that it'll happen again in
 130     # _inverse_log_det_jacobian.
 131     x = math_ops.log(y)
 132 
 133     log_normalization = (-x[..., -1])[..., array_ops.newaxis]
 134     x = x[..., :-1] + log_normalization
 135 
 136     # Set shape hints.
 137     if y.shape.ndims is not None:
 138       shape = y.shape[:-1].concatenate(y.shape[-1] - 1)
 139       x.shape.assert_is_compatible_with(shape)
 140       x.set_shape(shape)
 141 
 142     return x
 143 
 144   def _inverse_log_det_jacobian(self, y):
 145     # WLOG, consider the vector case:
 146     #   x = log(y[:-1]) - log(y[-1])
 147     # where,
 148     #   y[-1] = 1 - sum(y[:-1]).
 149     # We have:
 150     #   det{ dX/dY } = det{ diag(1 ./ y[:-1]) + 1 / y[-1] }
 151     #                = det{ inv{ diag(y[:-1]) - y[:-1]' y[:-1] } }   (1)
 152     #                = 1 / det{ diag(y[:-1]) - y[:-1]' y[:-1] }
 153     #                = 1 / { (1 + y[:-1]' inv(diag(y[:-1])) y[:-1]) *
 154     #                        det(diag(y[:-1])) }                     (2)
 155     #                = 1 / { y[-1] prod(y[:-1]) }
 156     #                = 1 / prod(y)
 157     # (1) - https://en.wikipedia.org/wiki/Sherman%E2%80%93Morrison_formula
 158     #       or by noting that det{ dX/dY } = 1 / det{ dY/dX } from Bijector
 159     #       docstring "Tip".
 160     # (2) - https://en.wikipedia.org/wiki/Matrix_determinant_lemma
 161     return -math_ops.reduce_sum(math_ops.log(y), axis=-1)
 162 
 163   def _forward_log_det_jacobian(self, x):
 164     # This code is similar to nn_ops.log_softmax but different because we have
 165     # an implicit zero column to handle. I.e., instead of:
 166     #   reduce_sum(logits - reduce_sum(exp(logits), dim))
 167     # we must do:
 168     #   log_normalization = 1 + reduce_sum(exp(logits))
 169     #   -log_normalization + reduce_sum(logits - log_normalization)
 170     log_normalization = nn_ops.softplus(
 171         math_ops.reduce_logsumexp(x, axis=-1, keep_dims=True))
 172     return array_ops.squeeze(
 173         (-log_normalization + math_ops.reduce_sum(
 174             x - log_normalization, axis=-1, keepdims=True)), axis=-1)
