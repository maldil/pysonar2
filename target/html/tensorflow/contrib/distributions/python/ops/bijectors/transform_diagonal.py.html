  30 class TransformDiagonal(bijector.Bijector):
  31   """Applies a Bijector to the diagonal of a matrix.
  32 
  33   #### Example
  34 
  35   ```python
  36   b = tfb.TransformDiagonal(diag_bijector=tfb.Exp())
  37 
  38   b.forward([[1., 0.],
  39              [0., 1.]])
  40   # ==> [[2.718, 0.],
  41          [0., 2.718]]
  42   ```
  43 
  44   """
  45 
  46   @deprecation.deprecated(
  47       "2018-10-01",
  48       "The TensorFlow Distributions library has moved to "
  49       "TensorFlow Probability "
  50       "(https://github.com/tensorflow/probability). You "
  51       "should update all references to use `tfp.distributions` "
  52       "instead of `tf.contrib.distributions`.",
  53       warn_once=True)
  54   def __init__(self,
  55                diag_bijector,
  56                validate_args=False,
  57                name="transform_diagonal"):
  58     """Instantiates the `TransformDiagonal` bijector.
  59 
  60     Args:
  61       diag_bijector: `Bijector` instance used to transform the diagonal.
  62       validate_args: Python `bool` indicating whether arguments should be
  63         checked for correctness.
  64       name: Python `str` name given to ops managed by this object.
  65     """
  66     self._diag_bijector = diag_bijector
  67     super(TransformDiagonal, self).__init__(
  68         forward_min_event_ndims=2,
  69         inverse_min_event_ndims=2,
  70         validate_args=validate_args,
  71         name=name)
  72 
  73   def _forward(self, x):
  74     diag = self._diag_bijector.forward(array_ops.matrix_diag_part(x))
  75     return array_ops.matrix_set_diag(x, diag)
  76 
  77   def _inverse(self, y):
  78     diag = self._diag_bijector.inverse(array_ops.matrix_diag_part(y))
  79     return array_ops.matrix_set_diag(y, diag)
  80 
  81   def _forward_log_det_jacobian(self, x):
  82     # We formulate the Jacobian with respect to the flattened matrices
  83     # `vec(x)` and `vec(y)`. Suppose for notational convenience that
  84     # the first `n` entries of `vec(x)` are the diagonal of `x`, and
  85     # the remaining `n**2-n` entries are the off-diagonals in
  86     # arbitrary order. Then the Jacobian is a block-diagonal matrix,
  87     # with the Jacobian of the diagonal bijector in the first block,
  88     # and the identity Jacobian for the remaining entries (since this
  89     # bijector acts as the identity on non-diagonal entries):
  90     #
  91     # J_vec(x) (vec(y)) =
  92     # -------------------------------
  93     # | J_diag(x) (diag(y))      0  | n entries
  94     # |                             |
  95     # | 0                        I  | n**2-n entries
  96     # -------------------------------
  97     #   n                     n**2-n
  98     #
  99     # Since the log-det of the second (identity) block is zero, the
 100     # overall log-det-jacobian is just the log-det of first block,
 101     # from the diagonal bijector.
 102     #
 103     # Note that for elementwise operations (exp, softplus, etc) the
 104     # first block of the Jacobian will itself be a diagonal matrix,
 105     # but our implementation does not require this to be true.
 106     return self._diag_bijector.forward_log_det_jacobian(
 107         array_ops.matrix_diag_part(x), event_ndims=1)
 108 
 109   def _inverse_log_det_jacobian(self, y):
 110     return self._diag_bijector.inverse_log_det_jacobian(
 111         array_ops.matrix_diag_part(y), event_ndims=1)
