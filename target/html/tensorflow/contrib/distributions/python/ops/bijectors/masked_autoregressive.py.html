  46 class MaskedAutoregressiveFlow(bijector.Bijector):
  47   """Affine MaskedAutoregressiveFlow bijector for vector-valued events.
  48 
  49   The affine autoregressive flow [(Papamakarios et al., 2016)][3] provides a
  50   relatively simple framework for user-specified (deep) architectures to learn
  51   a distribution over vector-valued events. Regarding terminology,
  52 
  53     "Autoregressive models decompose the joint density as a product of
  54     conditionals, and model each conditional in turn. Normalizing flows
  55     transform a base density (e.g. a standard Gaussian) into the target density
  56     by an invertible transformation with tractable Jacobian."
  57     [(Papamakarios et al., 2016)][3]
  58 
  59   In other words, the "autoregressive property" is equivalent to the
  60   decomposition, `p(x) = prod{ p(x[i] | x[0:i]) : i=0, ..., d }`. The provided
  61   `shift_and_log_scale_fn`, `masked_autoregressive_default_template`, achieves
  62   this property by zeroing out weights in its `masked_dense` layers.
  63 
  64   In the `tfp` framework, a "normalizing flow" is implemented as a
  65   `tfp.bijectors.Bijector`. The `forward` "autoregression"
  66   is implemented using a `tf.while_loop` and a deep neural network (DNN) with
  67   masked weights such that the autoregressive property is automatically met in
  68   the `inverse`.
  69 
  70   A `TransformedDistribution` using `MaskedAutoregressiveFlow(...)` uses the
  71   (expensive) forward-mode calculation to draw samples and the (cheap)
  72   reverse-mode calculation to compute log-probabilities. Conversely, a
  73   `TransformedDistribution` using `Invert(MaskedAutoregressiveFlow(...))` uses
  74   the (expensive) forward-mode calculation to compute log-probabilities and the
  75   (cheap) reverse-mode calculation to compute samples.  See "Example Use"
  76   [below] for more details.
  77 
  78   Given a `shift_and_log_scale_fn`, the forward and inverse transformations are
  79   (a sequence of) affine transformations. A "valid" `shift_and_log_scale_fn`
  80   must compute each `shift` (aka `loc` or "mu" in [Germain et al. (2015)][1])
  81   and `log(scale)` (aka "alpha" in [Germain et al. (2015)][1]) such that each
  82   are broadcastable with the arguments to `forward` and `inverse`, i.e., such
  83   that the calculations in `forward`, `inverse` [below] are possible.
  84 
  85   For convenience, `masked_autoregressive_default_template` is offered as a
  86   possible `shift_and_log_scale_fn` function. It implements the MADE
  87   architecture [(Germain et al., 2015)][1]. MADE is a feed-forward network that
  88   computes a `shift` and `log(scale)` using `masked_dense` layers in a deep
  89   neural network. Weights are masked to ensure the autoregressive property. It
  90   is possible that this architecture is suboptimal for your task. To build
  91   alternative networks, either change the arguments to
  92   `masked_autoregressive_default_template`, use the `masked_dense` function to
  93   roll-out your own, or use some other architecture, e.g., using `tf.layers`.
  94 
  95   Warning: no attempt is made to validate that the `shift_and_log_scale_fn`
  96   enforces the "autoregressive property".
  97 
  98   Assuming `shift_and_log_scale_fn` has valid shape and autoregressive
  99   semantics, the forward transformation is
 100 
 101   ```python
 102   def forward(x):
 103     y = zeros_like(x)
 104     event_size = x.shape[-1]
 105     for _ in range(event_size):
 106       shift, log_scale = shift_and_log_scale_fn(y)
 107       y = x * math_ops.exp(log_scale) + shift
 108     return y
 109   ```
 110 
 111   and the inverse transformation is
 112 
 113   ```python
 114   def inverse(y):
 115     shift, log_scale = shift_and_log_scale_fn(y)
 116     return (y - shift) / math_ops.exp(log_scale)
 117   ```
 118 
 119   Notice that the `inverse` does not need a for-loop. This is because in the
 120   forward pass each calculation of `shift` and `log_scale` is based on the `y`
 121   calculated so far (not `x`). In the `inverse`, the `y` is fully known, thus is
 122   equivalent to the scaling used in `forward` after `event_size` passes, i.e.,
 123   the "last" `y` used to compute `shift`, `log_scale`. (Roughly speaking, this
 124   also proves the transform is bijective.)
 125 
 126   #### Examples
 127 
 128   ```python
 129   import tensorflow_probability as tfp
 130   tfd = tfp.distributions
 131   tfb = tfp.bijectors
 132 
 133   dims = 5
 134 
 135   # A common choice for a normalizing flow is to use a Gaussian for the base
 136   # distribution. (However, any continuous distribution would work.) E.g.,
 137   maf = tfd.TransformedDistribution(
 138       distribution=tfd.Normal(loc=0., scale=1.),
 139       bijector=tfb.MaskedAutoregressiveFlow(
 140           shift_and_log_scale_fn=tfb.masked_autoregressive_default_template(
 141               hidden_layers=[512, 512])),
 142       event_shape=[dims])
 143 
 144   x = maf.sample()  # Expensive; uses `tf.while_loop`, no Bijector caching.
 145   maf.log_prob(x)   # Almost free; uses Bijector caching.
 146   maf.log_prob(0.)  # Cheap; no `tf.while_loop` despite no Bijector caching.
 147 
 148   # [Papamakarios et al. (2016)][3] also describe an Inverse Autoregressive
 149   # Flow [(Kingma et al., 2016)][2]:
 150   iaf = tfd.TransformedDistribution(
 151       distribution=tfd.Normal(loc=0., scale=1.),
 152       bijector=tfb.Invert(tfb.MaskedAutoregressiveFlow(
 153           shift_and_log_scale_fn=tfb.masked_autoregressive_default_template(
 154               hidden_layers=[512, 512]))),
 155       event_shape=[dims])
 156 
 157   x = iaf.sample()  # Cheap; no `tf.while_loop` despite no Bijector caching.
 158   iaf.log_prob(x)   # Almost free; uses Bijector caching.
 159   iaf.log_prob(0.)  # Expensive; uses `tf.while_loop`, no Bijector caching.
 160 
 161   # In many (if not most) cases the default `shift_and_log_scale_fn` will be a
 162   # poor choice. Here's an example of using a "shift only" version and with a
 163   # different number/depth of hidden layers.
 164   shift_only = True
 165   maf_no_scale_hidden2 = tfd.TransformedDistribution(
 166       distribution=tfd.Normal(loc=0., scale=1.),
 167       bijector=tfb.MaskedAutoregressiveFlow(
 168           tfb.masked_autoregressive_default_template(
 169               hidden_layers=[32],
 170               shift_only=shift_only),
 171           is_constant_jacobian=shift_only),
 172       event_shape=[dims])
 173   ```

 191   @deprecation.deprecated(
 192       "2018-10-01",
 193       "The TensorFlow Distributions library has moved to "

 199   def __init__(self,
 200                shift_and_log_scale_fn,
 201                is_constant_jacobian=False,
 202                validate_args=False,
 203                unroll_loop=False,
 204                name=None):
 205     """Creates the MaskedAutoregressiveFlow bijector.
 206 
 207     Args:
 208       shift_and_log_scale_fn: Python `callable` which computes `shift` and
 209         `log_scale` from both the forward domain (`x`) and the inverse domain
 210         (`y`). Calculation must respect the "autoregressive property" (see class
 211         docstring). Suggested default
 212         `masked_autoregressive_default_template(hidden_layers=...)`.
 213         Typically the function contains `tf.Variables` and is wrapped using
 214         `tf.make_template`. Returning `None` for either (both) `shift`,
 215         `log_scale` is equivalent to (but more efficient than) returning zero.
 216       is_constant_jacobian: Python `bool`. Default: `False`. When `True` the
 217         implementation assumes `log_scale` does not depend on the forward domain
 218         (`x`) or inverse domain (`y`) values. (No validation is made;
 219         `is_constant_jacobian=False` is always safe but possibly computationally
 220         inefficient.)
 221       validate_args: Python `bool` indicating whether arguments should be
 222         checked for correctness.
 223       unroll_loop: Python `bool` indicating whether the `tf.while_loop` in
 224         `_forward` should be replaced with a static for loop. Requires that
 225         the final dimension of `x` be known at graph construction time. Defaults
 226         to `False`.
 227       name: Python `str`, name given to ops managed by this object.
 228     """
 229     name = name or "masked_autoregressive_flow"
 230     self._shift_and_log_scale_fn = shift_and_log_scale_fn
 231     self._unroll_loop = unroll_loop
 232     super(MaskedAutoregressiveFlow, self).__init__(
 233         forward_min_event_ndims=1,
 234         is_constant_jacobian=is_constant_jacobian,
 235         validate_args=validate_args,
 236         name=name)
 237 
 238   def _forward(self, x):
 239     if self._unroll_loop:
 240       event_size = x.shape.with_rank_at_least(1)[-1].value
 241       if event_size is None:
 242         raise ValueError(
 243             "The final dimension of `x` must be known at graph construction "
 244             "time if `unroll_loop=True`. `x.shape: %r`" % x.shape)
 245       y = array_ops.zeros_like(x, name="y0")
 246 
 247       for _ in range(event_size):
 248         shift, log_scale = self._shift_and_log_scale_fn(y)
 249         # next_y = scale * x + shift
 250         next_y = x
 251         if log_scale is not None:
 252           next_y *= math_ops.exp(log_scale)
 253         if shift is not None:
 254           next_y += shift
 255         y = next_y
 256       return y
 257 
 258     event_size = array_ops.shape(x)[-1]
 259     # If the event size is available at graph construction time, we can inform
 260     # the graph compiler of the maximum number of steps. If not,
 261     # static_event_size will be None, and the maximum_iterations argument will
 262     # have no effect.
 263     static_event_size = x.shape.with_rank_at_least(1)[-1].value
 264     y0 = array_ops.zeros_like(x, name="y0")
 265     # call the template once to ensure creation
 266     _ = self._shift_and_log_scale_fn(y0)
 267     def _loop_body(index, y0):
 268       """While-loop body for autoregression calculation."""
 269       # Set caching device to avoid re-getting the tf.Variable for every while
 270       # loop iteration.
 271       with variable_scope_lib.variable_scope(
 272           variable_scope_lib.get_variable_scope()) as vs:
 273         if vs.caching_device is None:
 274           vs.set_caching_device(lambda op: op.device)
 275         shift, log_scale = self._shift_and_log_scale_fn(y0)
 276       y = x
 277       if log_scale is not None:
 278         y *= math_ops.exp(log_scale)
 279       if shift is not None:
 280         y += shift
 281       return index + 1, y
 282     _, y = control_flow_ops.while_loop(
 283         cond=lambda index, _: index < event_size,
 284         body=_loop_body,
 285         loop_vars=(0, y0),
 286         maximum_iterations=static_event_size)
 287     return y
 288 
 289   def _inverse(self, y):
 290     shift, log_scale = self._shift_and_log_scale_fn(y)
 291     x = y
 292     if shift is not None:
 293       x -= shift
 294     if log_scale is not None:
 295       x *= math_ops.exp(-log_scale)
 296     return x
 297 
 298   def _inverse_log_det_jacobian(self, y):
 299     _, log_scale = self._shift_and_log_scale_fn(y)
 300     if log_scale is None:
 301       return constant_op.constant(0., dtype=y.dtype, name="ildj")
 302     return -math_ops.reduce_sum(log_scale, axis=-1)

 317 def _gen_slices(num_blocks, n_in, n_out, mask_type=MASK_EXCLUSIVE):
 318   """Generate the slices for building an autoregressive mask."""
 319   # TODO(b/67594795): Better support of dynamic shape.
 320   slices = []
 321   col = 0
 322   d_in = n_in // num_blocks
 323   d_out = n_out // num_blocks
 324   row = d_out if mask_type == MASK_EXCLUSIVE else 0
 325   for _ in range(num_blocks):
 326     row_slice = slice(row, None)
 327     col_slice = slice(col, col + d_in)
 328     slices.append([row_slice, col_slice])
 329     col += d_in
 330     row += d_out
 331   return slices

 342 def _gen_mask(num_blocks,
 343               n_in,
 344               n_out,
 345               mask_type=MASK_EXCLUSIVE,
 346               dtype=dtypes.float32):
 347   """Generate the mask for building an autoregressive dense layer."""
 348   # TODO(b/67594795): Better support of dynamic shape.
 349   mask = np.zeros([n_out, n_in], dtype=dtype.as_numpy_dtype())
 350   slices = _gen_slices(num_blocks, n_in, n_out, mask_type=mask_type)
 351   for [row_slice, col_slice] in slices:
 352     mask[row_slice, col_slice] = 1
 353   return mask

 364 def masked_dense(inputs,
 365                  units,
 366                  num_blocks=None,
 367                  exclusive=False,
 368                  kernel_initializer=None,
 369                  reuse=None,
 370                  name=None,
 371                  *args,
 372                  **kwargs):
 373   """A autoregressively masked dense layer. Analogous to `tf.layers.dense`.
 374 
 375   See [Germain et al. (2015)][1] for detailed explanation.
 376 
 377   Arguments:
 378     inputs: Tensor input.
 379     units: Python `int` scalar representing the dimensionality of the output
 380       space.
 381     num_blocks: Python `int` scalar representing the number of blocks for the
 382       MADE masks.
 383     exclusive: Python `bool` scalar representing whether to zero the diagonal of
 384       the mask, used for the first layer of a MADE.
 385     kernel_initializer: Initializer function for the weight matrix.
 386       If `None` (default), weights are initialized using the
 387       `tf.glorot_random_initializer`.
 388     reuse: Python `bool` scalar representing whether to reuse the weights of a
 389       previous layer by the same name.
 390     name: Python `str` used to describe ops managed by this function.
 391     *args: `tf.layers.dense` arguments.
 392     **kwargs: `tf.layers.dense` keyword arguments.
 393 
 394   Returns:
 395     Output tensor.
 396 
 397   Raises:
 398     NotImplementedError: if rightmost dimension of `inputs` is unknown prior to
 399       graph execution.
 400 
 401   #### References
 402 
 403   [1]: Mathieu Germain, Karol Gregor, Iain Murray, and Hugo Larochelle. MADE:
 404        Masked Autoencoder for Distribution Estimation. In _International
 405        Conference on Machine Learning_, 2015. https://arxiv.org/abs/1502.03509
 406   """
 407   # TODO(b/67594795): Better support of dynamic shape.
 408   input_depth = inputs.shape.with_rank_at_least(1)[-1].value
 409   if input_depth is None:
 410     raise NotImplementedError(
 411         "Rightmost dimension must be known prior to graph execution.")
 412 
 413   mask = _gen_mask(num_blocks, input_depth, units,
 414                    MASK_EXCLUSIVE if exclusive else MASK_INCLUSIVE).T
 415 
 416   if kernel_initializer is None:
 417     kernel_initializer = init_ops.glorot_normal_initializer()
 418 
 419   def masked_initializer(shape, dtype=None, partition_info=None):
 420     return mask * kernel_initializer(shape, dtype, partition_info)
 421 
 422   with ops.name_scope(name, "masked_dense", [inputs, units, num_blocks]):
 423     layer = layers.Dense(
 424         units,
 425         kernel_initializer=masked_initializer,
 426         kernel_constraint=lambda x: mask * x,
 427         name=name,
 428         dtype=inputs.dtype.base_dtype,
 429         _scope=name,
 430         _reuse=reuse,
 431         *args,
 432         **kwargs)
 433     return layer.apply(inputs)

 436 @deprecation.deprecated(
 437     "2018-10-01",
 438     "The TensorFlow Distributions library has moved to "
 439     "TensorFlow Probability "

 444 def masked_autoregressive_default_template(
 445     hidden_layers,
 446     shift_only=False,
 447     activation=nn_ops.relu,
 448     log_scale_min_clip=-5.,
 449     log_scale_max_clip=3.,
 450     log_scale_clip_gradient=False,
 451     name=None,
 452     *args,
 453     **kwargs):
 454   """Build the Masked Autoregressive Density Estimator (Germain et al., 2015).
 455 
 456   This will be wrapped in a make_template to ensure the variables are only
 457   created once. It takes the input and returns the `loc` ("mu" in [Germain et
 458   al. (2015)][1]) and `log_scale` ("alpha" in [Germain et al. (2015)][1]) from
 459   the MADE network.
 460 
 461   Warning: This function uses `masked_dense` to create randomly initialized
 462   `tf.Variables`. It is presumed that these will be fit, just as you would any
 463   other neural architecture which uses `tf.layers.dense`.
 464 
 465   #### About Hidden Layers
 466 
 467   Each element of `hidden_layers` should be greater than the `input_depth`
 468   (i.e., `input_depth = tf.shape(input)[-1]` where `input` is the input to the
 469   neural network). This is necessary to ensure the autoregressivity property.
 470 
 471   #### About Clipping
 472 
 473   This function also optionally clips the `log_scale` (but possibly not its
 474   gradient). This is useful because if `log_scale` is too small/large it might
 475   underflow/overflow making it impossible for the `MaskedAutoregressiveFlow`
 476   bijector to implement a bijection. Additionally, the `log_scale_clip_gradient`
 477   `bool` indicates whether the gradient should also be clipped. The default does
 478   not clip the gradient; this is useful because it still provides gradient
 479   information (for fitting) yet solves the numerical stability problem. I.e.,
 480   `log_scale_clip_gradient = False` means
 481   `grad[exp(clip(x))] = grad[x] exp(clip(x))` rather than the usual
 482   `grad[clip(x)] exp(clip(x))`.
 483 
 484   Args:
 485     hidden_layers: Python `list`-like of non-negative integer, scalars
 486       indicating the number of units in each hidden layer. Default: `[512, 512].
 487     shift_only: Python `bool` indicating if only the `shift` term shall be
 488       computed. Default: `False`.
 489     activation: Activation function (callable). Explicitly setting to `None`
 490       implies a linear activation.
 491     log_scale_min_clip: `float`-like scalar `Tensor`, or a `Tensor` with the
 492       same shape as `log_scale`. The minimum value to clip by. Default: -5.
 493     log_scale_max_clip: `float`-like scalar `Tensor`, or a `Tensor` with the
 494       same shape as `log_scale`. The maximum value to clip by. Default: 3.
 495     log_scale_clip_gradient: Python `bool` indicating that the gradient of
 496       `tf.clip_by_value` should be preserved. Default: `False`.
 497     name: A name for ops managed by this function. Default:
 498       "masked_autoregressive_default_template".
 499     *args: `tf.layers.dense` arguments.
 500     **kwargs: `tf.layers.dense` keyword arguments.
 501 
 502   Returns:
 503     shift: `Float`-like `Tensor` of shift terms (the "mu" in
 504       [Germain et al.  (2015)][1]).
 505     log_scale: `Float`-like `Tensor` of log(scale) terms (the "alpha" in
 506       [Germain et al. (2015)][1]).
 507 
 508   Raises:
 509     NotImplementedError: if rightmost dimension of `inputs` is unknown prior to
 510       graph execution.
 511 
 512   #### References
 513 
 514   [1]: Mathieu Germain, Karol Gregor, Iain Murray, and Hugo Larochelle. MADE:
 515        Masked Autoencoder for Distribution Estimation. In _International
 516        Conference on Machine Learning_, 2015. https://arxiv.org/abs/1502.03509
 517   """
 518   name = name or "masked_autoregressive_default_template"
 519   with ops.name_scope(name, values=[log_scale_min_clip, log_scale_max_clip]):
 520     def _fn(x):
 521       """MADE parameterized via `masked_autoregressive_default_template`."""
 522       # TODO(b/67594795): Better support of dynamic shape.
 523       input_depth = x.shape.with_rank_at_least(1)[-1].value
 524       if input_depth is None:
 525         raise NotImplementedError(
 526             "Rightmost dimension must be known prior to graph execution.")
 527       input_shape = (np.int32(x.shape.as_list()) if x.shape.is_fully_defined()
 528                      else array_ops.shape(x))
 529       for i, units in enumerate(hidden_layers):
 530         x = masked_dense(
 531             inputs=x,
 532             units=units,
 533             num_blocks=input_depth,
 534             exclusive=True if i == 0 else False,
 535             activation=activation,
 536             *args,
 537             **kwargs)
 538       x = masked_dense(
 539           inputs=x,
 540           units=(1 if shift_only else 2) * input_depth,
 541           num_blocks=input_depth,
 542           activation=None,
 543           *args,
 544           **kwargs)
 545       if shift_only:
 546         x = array_ops.reshape(x, shape=input_shape)
 547         return x, None
 548       x = array_ops.reshape(
 549           x, shape=array_ops.concat([input_shape, [2]], axis=0))
 550       shift, log_scale = array_ops.unstack(x, num=2, axis=-1)
 551       which_clip = (math_ops.clip_by_value if log_scale_clip_gradient
 552                     else _clip_by_value_preserve_grad)
 553       log_scale = which_clip(log_scale, log_scale_min_clip, log_scale_max_clip)
 554       return shift, log_scale
 555     return template_ops.make_template(name, _fn)
