  29 class Invert(bijector.Bijector):
  30   """Bijector which inverts another Bijector.
  31 
  32   Example Use: [ExpGammaDistribution (see Background & Context)](
  33   https://reference.wolfram.com/language/ref/ExpGammaDistribution.html)
  34   models `Y=log(X)` where `X ~ Gamma`.
  35 
  36   ```python
  37   exp_gamma_distribution = TransformedDistribution(
  38     distribution=Gamma(concentration=1., rate=2.),
  39     bijector=bijector.Invert(bijector.Exp())
  40   ```
  41 
  42   """
  43 
  44   @deprecation.deprecated(
  45       "2018-10-01",
  46       "The TensorFlow Distributions library has moved to "
  47       "TensorFlow Probability "
  48       "(https://github.com/tensorflow/probability). You "
  49       "should update all references to use `tfp.distributions` "
  50       "instead of `tf.contrib.distributions`.",
  51       warn_once=True)
  52   def __init__(self, bijector, validate_args=False, name=None):
  53     """Creates a `Bijector` which swaps the meaning of `inverse` and `forward`.
  54 
  55     Note: An inverted bijector's `inverse_log_det_jacobian` is often more
  56     efficient if the base bijector implements `_forward_log_det_jacobian`. If
  57     `_forward_log_det_jacobian` is not implemented then the following code is
  58     used:
  59 
  60     ```python
  61     y = self.inverse(x, **kwargs)
  62     return -self.inverse_log_det_jacobian(y, **kwargs)
  63     ```
  64 
  65     Args:
  66       bijector: Bijector instance.
  67       validate_args: Python `bool` indicating whether arguments should be
  68         checked for correctness.
  69       name: Python `str`, name given to ops managed by this object.
  70     """
  71 
  72     if not bijector._is_injective:  # pylint: disable=protected-access
  73       raise NotImplementedError(
  74           "Invert is not implemented for non-injective bijectors.")
  75 
  76     self._bijector = bijector
  77     super(Invert, self).__init__(
  78         graph_parents=bijector.graph_parents,
  79         forward_min_event_ndims=bijector.inverse_min_event_ndims,
  80         inverse_min_event_ndims=bijector.forward_min_event_ndims,
  81         is_constant_jacobian=bijector.is_constant_jacobian,
  82         validate_args=validate_args,
  83         dtype=bijector.dtype,
  84         name=name or "_".join(["invert", bijector.name]))
  85 
  86   def _forward_event_shape(self, input_shape):
  87     return self.bijector._inverse_event_shape(input_shape)  # pylint: disable=protected-access
  88 
  89   def _forward_event_shape_tensor(self, input_shape):
  90     return self.bijector._inverse_event_shape_tensor(input_shape)  # pylint: disable=protected-access
  91 
  92   def _inverse_event_shape(self, output_shape):
  93     return self.bijector._forward_event_shape(output_shape)  # pylint: disable=protected-access
  94 
  95   def _inverse_event_shape_tensor(self, output_shape):
  96     return self.bijector._forward_event_shape_tensor(output_shape)  # pylint: disable=protected-access
  97 
  98   @property
  99   def bijector(self):
 100     return self._bijector
 101 
 102   def _forward(self, x, **kwargs):
 103     return self.bijector._inverse(x, **kwargs)  # pylint: disable=protected-access
 104 
 105   def _inverse(self, y, **kwargs):
 106     return self.bijector._forward(y, **kwargs)  # pylint: disable=protected-access
 107 
 108   def _inverse_log_det_jacobian(self, y, **kwargs):
 109     return self.bijector._forward_log_det_jacobian(y, **kwargs)  # pylint: disable=protected-access
 110 
 111   def _forward_log_det_jacobian(self, x, **kwargs):
 112     return self.bijector._inverse_log_det_jacobian(x, **kwargs)  # pylint: disable=protected-access
