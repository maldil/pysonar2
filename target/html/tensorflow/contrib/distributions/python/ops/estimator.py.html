  86 class _DistributionRegressionHead(_RegressionHead):
  87   """Creates a _RegressionHead instance from an arbitrary `Distribution`."""
  88 
  89   @deprecation.deprecated(
  90       "2018-10-01",
  91       "The TensorFlow Distributions library has moved to "
  92       "TensorFlow Probability "
  93       "(https://github.com/tensorflow/probability). You "
  94       "should update all references to use `tfp.distributions` "
  95       "instead of `tf.contrib.distributions`.",
  96       warn_once=True)
  97   def __init__(self,
  98                make_distribution_fn,
  99                label_dimension,
 100                logits_dimension=None,
 101                label_name=None,
 102                weight_column_name=None,
 103                enable_centered_bias=False,
 104                head_name=None):
 105     """`Head` for regression.
 106 
 107     Args:
 108       make_distribution_fn: Python `callable` which returns a `tf.Distribution`
 109         instance created using only logits.
 110       label_dimension: Number of regression labels per example. This is the
 111         size of the last dimension of the labels `Tensor` (typically, this has
 112         shape `[batch_size, label_dimension]`).
 113       logits_dimension: Number of logits per example. This is the size of the
 114         last dimension of the logits `Tensor` (typically, this has shape
 115         `[batch_size, logits_dimension]`).
 116         Default value: `label_dimension`.
 117       label_name: Python `str`, name of the key in label `dict`. Can be `None`
 118         if label is a tensor (single headed models).
 119       weight_column_name: Python `str` defining feature column name representing
 120         weights. It is used to down weight or boost examples during training. It
 121         will be multiplied by the loss of the example.
 122       enable_centered_bias: Python `bool`. If `True`, estimator will learn a
 123         centered bias variable for each class. Rest of the model structure
 124         learns the residual after centered bias.
 125       head_name: Python `str`, name of the head. Predictions, summary and
 126         metrics keys are suffixed by `"/" + head_name` and the default variable
 127         scope is `head_name`.
 128 
 129     Raises:
 130       TypeError: if `make_distribution_fn` is not `callable`.
 131     """
 132     if not callable(make_distribution_fn):
 133       raise TypeError("`make_distribution_fn` must be a callable function.")
 134 
 135     self._distributions = {}
 136     self._make_distribution_fn = make_distribution_fn
 137 
 138     def static_value(x):
 139       """Returns the static value of a `Tensor` or `None`."""
 140       return tensor_util.constant_value(ops.convert_to_tensor(x))
 141 
 142     def concat_vectors(*args):
 143       """Concatenates input vectors, statically if possible."""
 144       args_ = [static_value(x) for x in args]
 145       if any(vec is None for vec in args_):
 146         return array_ops.concat(args, axis=0)
 147       return [val for vec in args_ for val in vec]
 148 
 149     def loss_fn(labels, logits, weights=None):
 150       """Returns the loss of using `logits` to predict `labels`."""
 151       d = self.distribution(logits)
 152       labels_batch_shape = labels.shape.with_rank_at_least(1)[:-1]
 153       labels_batch_shape = (
 154           labels_batch_shape.as_list() if labels_batch_shape.is_fully_defined()
 155           else array_ops.shape(labels)[:-1])
 156       labels = array_ops.reshape(
 157           labels,
 158           shape=concat_vectors(labels_batch_shape, d.event_shape_tensor()))
 159       return _compute_weighted_loss(
 160           loss_unweighted=-d.log_prob(labels),
 161           weight=weights)
 162 
 163     def link_fn(logits):
 164       """Returns the inverse link function at `logits`."""
 165       # Note: What the API calls a "link function" is really the inverse-link
 166       # function, i.e., the "mean".
 167       d = self.distribution(logits)
 168       return d.mean()
 169 
 170     super(_DistributionRegressionHead, self).__init__(
 171         label_dimension=label_dimension,
 172         loss_fn=loss_fn,
 173         link_fn=link_fn,
 174         logits_dimension=logits_dimension,
 175         label_name=label_name,
 176         weight_column_name=weight_column_name,
 177         enable_centered_bias=enable_centered_bias,
 178         head_name=head_name)
 179 
 180   @property
 181   def distributions(self):
 182     """Returns all distributions created by `DistributionRegressionHead`."""
 183     return self._distributions
 184 
 185   def distribution(self, logits, name=None):
 186     """Retrieves a distribution instance, parameterized by `logits`.
 187 
 188     Args:
 189       logits: `float`-like `Tensor` representing the parameters of the
 190         underlying distribution.
 191       name: The Python `str` name to given to this op.
 192         Default value: "distribution".
 193 
 194     Returns:
 195       distribution: `tf.Distribution` instance parameterized by `logits`.
 196     """
 197     with ops.name_scope(name, "distribution", [logits]):
 198       d = self._distributions.get(logits, None)
 199       if d is None:
 200         d = self._make_distribution_fn(logits)
 201         self._distributions[logits] = d
 202       return d
