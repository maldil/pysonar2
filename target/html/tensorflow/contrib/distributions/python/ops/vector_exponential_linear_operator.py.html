  50 class VectorExponentialLinearOperator(
  51     transformed_distribution.TransformedDistribution):
  52   """The vectorization of the Exponential distribution on `R^k`.
  53 
  54   The vector exponential distribution is defined over a subset of `R^k`, and
  55   parameterized by a (batch of) length-`k` `loc` vector and a (batch of) `k x k`
  56   `scale` matrix:  `covariance = scale @ scale.T`, where `@` denotes
  57   matrix-multiplication.
  58 
  59   #### Mathematical Details
  60 
  61   The probability density function (pdf) is
  62 
  63   ```none
  64   pdf(y; loc, scale) = exp(-||x||_1) / Z,  for y in S(loc, scale),
  65   x = inv(scale) @ (y - loc),
  66   Z = |det(scale)|,
  67   ```
  68 
  69   where:
  70 
  71   * `loc` is a vector in `R^k`,
  72   * `scale` is a linear operator in `R^{k x k}`, `cov = scale @ scale.T`,
  73   * `S = {loc + scale @ x : x in R^k, x_1 > 0, ..., x_k > 0}`, is an image of
  74      the positive half-space,
  75   * `||x||_1` denotes the `l1` norm of `x`, `sum_i |x_i|`,
  76   * `Z` denotes the normalization constant.
  77 
  78   The VectorExponential distribution is a member of the [location-scale
  79   family](https://en.wikipedia.org/wiki/Location-scale_family), i.e., it can be
  80   constructed as,
  81 
  82   ```none
  83   X = (X_1, ..., X_k), each X_i ~ Exponential(rate=1)
  84   Y = (Y_1, ...,Y_k) = scale @ X + loc
  85   ```
  86 
  87   #### About `VectorExponential` and `Vector` distributions in TensorFlow.
  88 
  89   The `VectorExponential` is a non-standard distribution that has useful
  90   properties.
  91 
  92   The marginals `Y_1, ..., Y_k` are *not* Exponential random variables, due to
  93   the fact that the sum of Exponential random variables is not Exponential.
  94 
  95   Instead, `Y` is a vector whose components are linear combinations of
  96   Exponential random variables.  Thus, `Y` lives in the vector space generated
  97   by `vectors` of Exponential distributions.  This allows the user to decide the
  98   mean and covariance (by setting `loc` and `scale`), while preserving some
  99   properties of the Exponential distribution.  In particular, the tails of `Y_i`
 100   will be (up to polynomial factors) exponentially decaying.
 101 
 102   To see this last statement, note that the pdf of `Y_i` is the convolution of
 103   the pdf of `k` independent Exponential random variables.  One can then show by
 104   induction that distributions with exponential (up to polynomial factors) tails
 105   are closed under convolution.
 106 
 107 
 108   #### Examples
 109 
 110   ```python
 111   import tensorflow_probability as tfp
 112   tfd = tfp.distributions
 113 
 114   # Initialize a single 2-variate VectorExponential, supported on
 115   # {(x, y) in R^2 : x > 0, y > 0}.
 116   mat = [[1.0, 0.1],
 117          [0.1, 1.0]]
 118 
 119   vex = tfd.VectorExponentialLinearOperator(
 120       scale=tf.linalg.LinearOperatorFullMatrix(mat))
 121 
 122   # Compute the pdf of an`R^2` observation; return a scalar.
 123   vex.prob([1., 2.]).eval()  # shape: []
 124 
 125   # Initialize a 2-batch of 3-variate Vector Exponential's.
 126   mu = [[1., 2, 3],
 127         [1., 0, 0]]              # shape: [2, 3]
 128   scale_diag = [[1., 2, 3],
 129                 [0.5, 1, 1.5]]     # shape: [2, 3]
 130 
 131   vex = tfd.VectorExponentialLinearOperator(
 132       loc=mu,
 133       scale=tf.linalg.LinearOperatorDiag(scale_diag))
 134 
 135   # Compute the pdf of two `R^3` observations; return a length-2 vector.
 136   x = [[1.9, 2.2, 3.1],
 137        [10., 1.0, 9.0]]     # shape: [2, 3]
 138   vex.prob(x).eval()    # shape: [2]
 139   ```

 141   """
 142 
 143   @deprecation.deprecated(
 144       "2018-10-01",
 145       "The TensorFlow Distributions library has moved to "
 146       "TensorFlow Probability "
 147       "(https://github.com/tensorflow/probability). You "
 148       "should update all references to use `tfp.distributions` "
 149       "instead of `tf.contrib.distributions`.",
 150       warn_once=True)
 151   def __init__(self,
 152                loc=None,
 153                scale=None,
 154                validate_args=False,
 155                allow_nan_stats=True,
 156                name="VectorExponentialLinearOperator"):
 157     """Construct Vector Exponential distribution supported on a subset of `R^k`.
 158 
 159     The `batch_shape` is the broadcast shape between `loc` and `scale`
 160     arguments.
 161 
 162     The `event_shape` is given by last dimension of the matrix implied by
 163     `scale`. The last dimension of `loc` (if provided) must broadcast with this.
 164 
 165     Recall that `covariance = scale @ scale.T`.
 166 
 167     Additional leading dimensions (if any) will index batches.
 168 
 169     Args:
 170       loc: Floating-point `Tensor`. If this is set to `None`, `loc` is
 171         implicitly `0`. When specified, may have shape `[B1, ..., Bb, k]` where
 172         `b >= 0` and `k` is the event size.
 173       scale: Instance of `LinearOperator` with same `dtype` as `loc` and shape
 174         `[B1, ..., Bb, k, k]`.
 175       validate_args: Python `bool`, default `False`. Whether to validate input
 176         with asserts. If `validate_args` is `False`, and the inputs are
 177         invalid, correct behavior is not guaranteed.
 178       allow_nan_stats: Python `bool`, default `True`. If `False`, raise an
 179         exception if a statistic (e.g. mean/mode/etc...) is undefined for any
 180         batch member If `True`, batch members with valid parameters leading to
 181         undefined statistics will return NaN for this statistic.
 182       name: The name to give Ops created by the initializer.
 183 
 184     Raises:
 185       ValueError: if `scale` is unspecified.
 186       TypeError: if not `scale.dtype.is_floating`
 187     """
 188     parameters = dict(locals())
 189     if scale is None:
 190       raise ValueError("Missing required `scale` parameter.")
 191     if not scale.dtype.is_floating:
 192       raise TypeError("`scale` parameter must have floating-point dtype.")
 193 
 194     with ops.name_scope(name, values=[loc] + scale.graph_parents) as name:
 195       # Since expand_dims doesn't preserve constant-ness, we obtain the
 196       # non-dynamic value if possible.
 197       loc = ops.convert_to_tensor(loc, name="loc") if loc is not None else loc
 198       batch_shape, event_shape = distribution_util.shapes_from_loc_and_scale(
 199           loc, scale)
 200 
 201       super(VectorExponentialLinearOperator, self).__init__(
 202           distribution=exponential.Exponential(rate=array_ops.ones(
 203               [], dtype=scale.dtype), allow_nan_stats=allow_nan_stats),
 204           bijector=bijectors.AffineLinearOperator(
 205               shift=loc, scale=scale, validate_args=validate_args),
 206           batch_shape=batch_shape,
 207           event_shape=event_shape,
 208           validate_args=validate_args,
 209           name=name)
 210       self._parameters = parameters
 211 
 212   @property
 213   def loc(self):
 214     """The `loc` `Tensor` in `Y = scale @ X + loc`."""
 215     return self.bijector.shift
 216 
 217   @property
 218   def scale(self):
 219     """The `scale` `LinearOperator` in `Y = scale @ X + loc`."""
 220     return self.bijector.scale
 221 
 222   @distribution_util.AppendDocstring(_mvn_sample_note)
 223   def _log_prob(self, x):
 224     return super(VectorExponentialLinearOperator, self)._log_prob(x)
 225 
 226   @distribution_util.AppendDocstring(_mvn_sample_note)
 227   def _prob(self, x):
 228     return super(VectorExponentialLinearOperator, self)._prob(x)
 229 
 230   def _mean(self):
 231     # Let
 232     #   W = (w1,...,wk), with wj ~ iid Exponential(0, 1).
 233     # Then this distribution is
 234     #   X = loc + LW,
 235     # and then E[X] = loc + L1, where 1 is the vector of ones.
 236     scale_x_ones = self.bijector.scale.matvec(
 237         array_ops.ones(self._mode_mean_shape(), self.dtype))
 238 
 239     if self.loc is None:
 240       return scale_x_ones
 241 
 242     return array_ops.identity(self.loc) + scale_x_ones
 243 
 244   def _covariance(self):
 245     # Let
 246     #   W = (w1,...,wk), with wj ~ iid Exponential(0, 1).
 247     # Then this distribution is
 248     #   X = loc + LW,
 249     # and then since Cov(wi, wj) = 1 if i=j, and 0 otherwise,
 250     #   Cov(X) = L Cov(W W^T) L^T = L L^T.
 251     if distribution_util.is_diagonal_scale(self.scale):
 252       return array_ops.matrix_diag(math_ops.square(self.scale.diag_part()))
 253     else:
 254       return self.scale.matmul(self.scale.to_dense(), adjoint_arg=True)
 255 
 256   def _variance(self):
 257     if distribution_util.is_diagonal_scale(self.scale):
 258       return math_ops.square(self.scale.diag_part())
 259     elif (isinstance(self.scale, linalg.LinearOperatorLowRankUpdate) and
 260           self.scale.is_self_adjoint):
 261       return array_ops.matrix_diag_part(
 262           self.scale.matmul(self.scale.to_dense()))
 263     else:
 264       return array_ops.matrix_diag_part(
 265           self.scale.matmul(self.scale.to_dense(), adjoint_arg=True))
 266 
 267   def _stddev(self):
 268     if distribution_util.is_diagonal_scale(self.scale):
 269       return math_ops.abs(self.scale.diag_part())
 270     elif (isinstance(self.scale, linalg.LinearOperatorLowRankUpdate) and
 271           self.scale.is_self_adjoint):
 272       return math_ops.sqrt(
 273           array_ops.matrix_diag_part(self.scale.matmul(self.scale.to_dense())))
 274     else:
 275       return math_ops.sqrt(
 276           array_ops.matrix_diag_part(
 277               self.scale.matmul(self.scale.to_dense(), adjoint_arg=True)))
 278 
 279   def _mode(self):
 280     scale_x_zeros = self.bijector.scale.matvec(
 281         array_ops.zeros(self._mode_mean_shape(), self.dtype))
 282 
 283     if self.loc is None:
 284       return scale_x_zeros
 285 
 286     return array_ops.identity(self.loc) + scale_x_zeros
 287 
 288   def _mode_mean_shape(self):
 289     """Shape for the mode/mean Tensors."""
 290     shape = self.batch_shape.concatenate(self.event_shape)
 291     has_static_shape = shape.is_fully_defined()
 292     if not has_static_shape:
 293       shape = array_ops.concat([
 294           self.batch_shape_tensor(),
 295           self.event_shape_tensor(),
 296       ], 0)
 297     return shape
