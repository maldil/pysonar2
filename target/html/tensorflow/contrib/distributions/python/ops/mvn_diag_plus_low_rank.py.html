  33 class MultivariateNormalDiagPlusLowRank(
  34     mvn_linop.MultivariateNormalLinearOperator):
  35   """The multivariate normal distribution on `R^k`.
  36 
  37   The Multivariate Normal distribution is defined over `R^k` and parameterized
  38   by a (batch of) length-`k` `loc` vector (aka "mu") and a (batch of) `k x k`
  39   `scale` matrix; `covariance = scale @ scale.T` where `@` denotes
  40   matrix-multiplication.
  41 
  42   #### Mathematical Details
  43 
  44   The probability density function (pdf) is,
  45 
  46   ```none
  47   pdf(x; loc, scale) = exp(-0.5 ||y||**2) / Z,
  48   y = inv(scale) @ (x - loc),
  49   Z = (2 pi)**(0.5 k) |det(scale)|,
  50   ```
  51 
  52   where:
  53 
  54   * `loc` is a vector in `R^k`,
  55   * `scale` is a linear operator in `R^{k x k}`, `cov = scale @ scale.T`,
  56   * `Z` denotes the normalization constant, and,
  57   * `||y||**2` denotes the squared Euclidean norm of `y`.
  58 
  59   A (non-batch) `scale` matrix is:
  60 
  61   ```none
  62   scale = diag(scale_diag + scale_identity_multiplier ones(k)) +
  63         scale_perturb_factor @ diag(scale_perturb_diag) @ scale_perturb_factor.T
  64   ```
  65 
  66   where:
  67 
  68   * `scale_diag.shape = [k]`,
  69   * `scale_identity_multiplier.shape = []`,
  70   * `scale_perturb_factor.shape = [k, r]`, typically `k >> r`, and,
  71   * `scale_perturb_diag.shape = [r]`.
  72 
  73   Additional leading dimensions (if any) will index batches.
  74 
  75   If both `scale_diag` and `scale_identity_multiplier` are `None`, then
  76   `scale` is the Identity matrix.
  77 
  78   The MultivariateNormal distribution is a member of the [location-scale
  79   family](https://en.wikipedia.org/wiki/Location-scale_family), i.e., it can be
  80   constructed as,
  81 
  82   ```none
  83   X ~ MultivariateNormal(loc=0, scale=1)   # Identity scale, zero shift.
  84   Y = scale @ X + loc
  85   ```
  86 
  87   #### Examples
  88 
  89   ```python
  90   import tensorflow_probability as tfp
  91   tfd = tfp.distributions
  92 
  93   # Initialize a single 3-variate Gaussian with covariance `cov = S @ S.T`,
  94   # `S = diag(d) + U @ diag(m) @ U.T`. The perturbation, `U @ diag(m) @ U.T`, is
  95   # a rank-2 update.
  96   mu = [-0.5., 0, 0.5]   # shape: [3]
  97   d = [1.5, 0.5, 2]      # shape: [3]
  98   U = [[1., 2],
  99        [-1, 1],
 100        [2, -0.5]]        # shape: [3, 2]
 101   m = [4., 5]            # shape: [2]
 102   mvn = tfd.MultivariateNormalDiagPlusLowRank(
 103       loc=mu
 104       scale_diag=d
 105       scale_perturb_factor=U,
 106       scale_perturb_diag=m)
 107 
 108   # Evaluate this on an observation in `R^3`, returning a scalar.
 109   mvn.prob([-1, 0, 1]).eval()  # shape: []
 110 
 111   # Initialize a 2-batch of 3-variate Gaussians; `S = diag(d) + U @ U.T`.
 112   mu = [[1.,  2,  3],
 113         [11, 22, 33]]      # shape: [b, k] = [2, 3]
 114   U = [[[1., 2],
 115         [3,  4],
 116         [5,  6]],
 117        [[0.5, 0.75],
 118         [1,0, 0.25],
 119         [1.5, 1.25]]]      # shape: [b, k, r] = [2, 3, 2]
 120   m = [[0.1, 0.2],
 121        [0.4, 0.5]]         # shape: [b, r] = [2, 2]
 122 
 123   mvn = tfd.MultivariateNormalDiagPlusLowRank(
 124       loc=mu,
 125       scale_perturb_factor=U,
 126       scale_perturb_diag=m)
 127 
 128   mvn.covariance().eval()   # shape: [2, 3, 3]
 129   # ==> [[[  15.63   31.57    48.51]
 130   #       [  31.57   69.31   105.05]
 131   #       [  48.51  105.05   162.59]]
 132   #
 133   #      [[   2.59    1.41    3.35]
 134   #       [   1.41    2.71    3.34]
 135   #       [   3.35    3.34    8.35]]]
 136 
 137   # Compute the pdf of two `R^3` observations (one from each batch);
 138   # return a length-2 vector.
 139   x = [[-0.9, 0, 0.1],
 140        [-10, 0, 9]]     # shape: [2, 3]
 141   mvn.prob(x).eval()    # shape: [2]
 142   ```

 144   """
 145 
 146   @deprecation.deprecated(
 147       "2018-10-01",
 148       "The TensorFlow Distributions library has moved to "
 149       "TensorFlow Probability "
 150       "(https://github.com/tensorflow/probability). You "
 151       "should update all references to use `tfp.distributions` "
 152       "instead of `tf.contrib.distributions`.",
 153       warn_once=True)
 154   def __init__(self,
 155                loc=None,
 156                scale_diag=None,
 157                scale_identity_multiplier=None,
 158                scale_perturb_factor=None,
 159                scale_perturb_diag=None,
 160                validate_args=False,
 161                allow_nan_stats=True,
 162                name="MultivariateNormalDiagPlusLowRank"):
 163     """Construct Multivariate Normal distribution on `R^k`.
 164 
 165     The `batch_shape` is the broadcast shape between `loc` and `scale`
 166     arguments.
 167 
 168     The `event_shape` is given by last dimension of the matrix implied by
 169     `scale`. The last dimension of `loc` (if provided) must broadcast with this.
 170 
 171     Recall that `covariance = scale @ scale.T`. A (non-batch) `scale` matrix is:
 172 
 173     ```none
 174     scale = diag(scale_diag + scale_identity_multiplier ones(k)) +
 175         scale_perturb_factor @ diag(scale_perturb_diag) @ scale_perturb_factor.T
 176     ```
 177 
 178     where:
 179 
 180     * `scale_diag.shape = [k]`,
 181     * `scale_identity_multiplier.shape = []`,
 182     * `scale_perturb_factor.shape = [k, r]`, typically `k >> r`, and,
 183     * `scale_perturb_diag.shape = [r]`.
 184 
 185     Additional leading dimensions (if any) will index batches.
 186 
 187     If both `scale_diag` and `scale_identity_multiplier` are `None`, then
 188     `scale` is the Identity matrix.
 189 
 190     Args:
 191       loc: Floating-point `Tensor`. If this is set to `None`, `loc` is
 192         implicitly `0`. When specified, may have shape `[B1, ..., Bb, k]` where
 193         `b >= 0` and `k` is the event size.
 194       scale_diag: Non-zero, floating-point `Tensor` representing a diagonal
 195         matrix added to `scale`. May have shape `[B1, ..., Bb, k]`, `b >= 0`,
 196         and characterizes `b`-batches of `k x k` diagonal matrices added to
 197         `scale`. When both `scale_identity_multiplier` and `scale_diag` are
 198         `None` then `scale` is the `Identity`.
 199       scale_identity_multiplier: Non-zero, floating-point `Tensor` representing
 200         a scaled-identity-matrix added to `scale`. May have shape
 201         `[B1, ..., Bb]`, `b >= 0`, and characterizes `b`-batches of scaled
 202         `k x k` identity matrices added to `scale`. When both
 203         `scale_identity_multiplier` and `scale_diag` are `None` then `scale` is
 204         the `Identity`.
 205       scale_perturb_factor: Floating-point `Tensor` representing a rank-`r`
 206         perturbation added to `scale`. May have shape `[B1, ..., Bb, k, r]`,
 207         `b >= 0`, and characterizes `b`-batches of rank-`r` updates to `scale`.
 208         When `None`, no rank-`r` update is added to `scale`.
 209       scale_perturb_diag: Floating-point `Tensor` representing a diagonal matrix
 210         inside the rank-`r` perturbation added to `scale`. May have shape
 211         `[B1, ..., Bb, r]`, `b >= 0`, and characterizes `b`-batches of `r x r`
 212         diagonal matrices inside the perturbation added to `scale`. When
 213         `None`, an identity matrix is used inside the perturbation. Can only be
 214         specified if `scale_perturb_factor` is also specified.
 215       validate_args: Python `bool`, default `False`. When `True` distribution
 216         parameters are checked for validity despite possibly degrading runtime
 217         performance. When `False` invalid inputs may silently render incorrect
 218         outputs.
 219       allow_nan_stats: Python `bool`, default `True`. When `True`,
 220         statistics (e.g., mean, mode, variance) use the value "`NaN`" to
 221         indicate the result is undefined. When `False`, an exception is raised
 222         if one or more of the statistic's batch members are undefined.
 223       name: Python `str` name prefixed to Ops created by this class.
 224 
 225     Raises:
 226       ValueError: if at most `scale_identity_multiplier` is specified.
 227     """
 228     parameters = dict(locals())
 229     def _convert_to_tensor(x, name):
 230       return None if x is None else ops.convert_to_tensor(x, name=name)
 231     with ops.name_scope(name) as name:
 232       with ops.name_scope("init", values=[
 233           loc, scale_diag, scale_identity_multiplier, scale_perturb_factor,
 234           scale_perturb_diag]):
 235         has_low_rank = (scale_perturb_factor is not None or
 236                         scale_perturb_diag is not None)
 237         scale = distribution_util.make_diag_scale(
 238             loc=loc,
 239             scale_diag=scale_diag,
 240             scale_identity_multiplier=scale_identity_multiplier,
 241             validate_args=validate_args,
 242             assert_positive=has_low_rank)
 243         scale_perturb_factor = _convert_to_tensor(
 244             scale_perturb_factor,
 245             name="scale_perturb_factor")
 246         scale_perturb_diag = _convert_to_tensor(
 247             scale_perturb_diag,
 248             name="scale_perturb_diag")
 249         if has_low_rank:
 250           scale = linalg.LinearOperatorLowRankUpdate(
 251               scale,
 252               u=scale_perturb_factor,
 253               diag_update=scale_perturb_diag,
 254               is_diag_update_positive=scale_perturb_diag is None,
 255               is_non_singular=True,  # Implied by is_positive_definite=True.
 256               is_self_adjoint=True,
 257               is_positive_definite=True,
 258               is_square=True)
 259     super(MultivariateNormalDiagPlusLowRank, self).__init__(
 260         loc=loc,
 261         scale=scale,
 262         validate_args=validate_args,
 263         allow_nan_stats=allow_nan_stats,
 264         name=name)
 265     self._parameters = parameters
