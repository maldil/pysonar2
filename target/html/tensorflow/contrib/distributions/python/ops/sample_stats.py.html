  41 # TODO(langmore) Write separate versions of this for real/complex dtype, taking
  42 # advantage of optimized real-fft ops.
  43 def auto_correlation(
  44     x,
  45     axis=-1,
  46     max_lags=None,
  47     center=True,
  48     normalize=True,
  49     name="auto_correlation"):
  50   """Auto correlation along one axis.
  51 
  52   Given a `1-D` wide sense stationary (WSS) sequence `X`, the auto correlation
  53   `RXX` may be defined as  (with `E` expectation and `Conj` complex conjugate)
  54 
  55   ```
  56   RXX[m] := E{ W[m] Conj(W[0]) } = E{ W[0] Conj(W[-m]) },
  57   W[n]   := (X[n] - MU) / S,
  58   MU     := E{ X[0] },
  59   S**2   := E{ (X[0] - MU) Conj(X[0] - MU) }.
  60   ```
  61 
  62   This function takes the viewpoint that `x` is (along one axis) a finite
  63   sub-sequence of a realization of (WSS) `X`, and then uses `x` to produce an
  64   estimate of `RXX[m]` as follows:
  65 
  66   After extending `x` from length `L` to `inf` by zero padding, the auto
  67   correlation estimate `rxx[m]` is computed for `m = 0, 1, ..., max_lags` as
  68 
  69   ```
  70   rxx[m] := (L - m)**-1 sum_n w[n + m] Conj(w[n]),
  71   w[n]   := (x[n] - mu) / s,
  72   mu     := L**-1 sum_n x[n],
  73   s**2   := L**-1 sum_n (x[n] - mu) Conj(x[n] - mu)
  74   ```
  75 
  76   The error in this estimate is proportional to `1 / sqrt(len(x) - m)`, so users
  77   often set `max_lags` small enough so that the entire output is meaningful.
  78 
  79   Note that since `mu` is an imperfect estimate of `E{ X[0] }`, and we divide by
  80   `len(x) - m` rather than `len(x) - m - 1`, our estimate of auto correlation
  81   contains a slight bias, which goes to zero as `len(x) - m --> infinity`.
  82 
  83   Args:
  84     x:  `float32` or `complex64` `Tensor`.
  85     axis:  Python `int`. The axis number along which to compute correlation.
  86       Other dimensions index different batch members.
  87     max_lags:  Positive `int` tensor.  The maximum value of `m` to consider
  88       (in equation above).  If `max_lags >= x.shape[axis]`, we effectively
  89       re-set `max_lags` to `x.shape[axis] - 1`.
  90     center:  Python `bool`.  If `False`, do not subtract the mean estimate `mu`
  91       from `x[n]` when forming `w[n]`.
  92     normalize:  Python `bool`.  If `False`, do not divide by the variance
  93       estimate `s**2` when forming `w[n]`.
  94     name:  `String` name to prepend to created ops.
  95 
  96   Returns:
  97     `rxx`: `Tensor` of same `dtype` as `x`.  `rxx.shape[i] = x.shape[i]` for
  98       `i != axis`, and `rxx.shape[axis] = max_lags + 1`.
  99 
 100   Raises:
 101     TypeError:  If `x` is not a supported type.
 102   """
 103   # Implementation details:
 104   # Extend length N / 2 1-D array x to length N by zero padding onto the end.
 105   # Then, set
 106   #   F[x]_k := sum_n x_n exp{-i 2 pi k n / N }.
 107   # It is not hard to see that
 108   #   F[x]_k Conj(F[x]_k) = F[R]_k, where
 109   #   R_m := sum_n x_n Conj(x_{(n - m) mod N}).
 110   # One can also check that R_m / (N / 2 - m) is an unbiased estimate of RXX[m].
 111 
 112   # Since F[x] is the DFT of x, this leads us to a zero-padding and FFT/IFFT
 113   # based version of estimating RXX.
 114   # Note that this is a special case of the Wiener-Khinchin Theorem.
 115   with ops.name_scope(name, values=[x]):
 116     x = ops.convert_to_tensor(x, name="x")
 117 
 118     # Rotate dimensions of x in order to put axis at the rightmost dim.
 119     # FFT op requires this.
 120     rank = util.prefer_static_rank(x)
 121     if axis < 0:
 122       axis = rank + axis
 123     shift = rank - 1 - axis
 124     # Suppose x.shape[axis] = T, so there are T "time" steps.
 125     #   ==> x_rotated.shape = B + [T],
 126     # where B is x_rotated's batch shape.
 127     x_rotated = util.rotate_transpose(x, shift)
 128 
 129     if center:
 130       x_rotated -= math_ops.reduce_mean(x_rotated, axis=-1, keepdims=True)
 131 
 132     # x_len = N / 2 from above explanation.  The length of x along axis.
 133     # Get a value for x_len that works in all cases.
 134     x_len = util.prefer_static_shape(x_rotated)[-1]
 135 
 136     # TODO(langmore) Investigate whether this zero padding helps or hurts.  At
 137     # the moment is necessary so that all FFT implementations work.
 138     # Zero pad to the next power of 2 greater than 2 * x_len, which equals
 139     # 2**(ceil(Log_2(2 * x_len))).  Note: Log_2(X) = Log_e(X) / Log_e(2).
 140     x_len_float64 = math_ops.cast(x_len, np.float64)
 141     target_length = math_ops.pow(
 142         np.float64(2.),
 143         math_ops.ceil(math_ops.log(x_len_float64 * 2) / np.log(2.)))
 144     pad_length = math_ops.cast(target_length - x_len_float64, np.int32)
 145 
 146     # We should have:
 147     # x_rotated_pad.shape = x_rotated.shape[:-1] + [T + pad_length]
 148     #                     = B + [T + pad_length]
 149     x_rotated_pad = util.pad(x_rotated, axis=-1, back=True, count=pad_length)
 150 
 151     dtype = x.dtype
 152     if not dtype.is_complex:
 153       if not dtype.is_floating:
 154         raise TypeError("Argument x must have either float or complex dtype"
 155                         " found: {}".format(dtype))
 156       x_rotated_pad = math_ops.complex(x_rotated_pad,
 157                                        dtype.real_dtype.as_numpy_dtype(0.))
 158 
 159     # Autocorrelation is IFFT of power-spectral density (up to some scaling).
 160     fft_x_rotated_pad = spectral_ops.fft(x_rotated_pad)
 161     spectral_density = fft_x_rotated_pad * math_ops.conj(fft_x_rotated_pad)
 162     # shifted_product is R[m] from above detailed explanation.
 163     # It is the inner product sum_n X[n] * Conj(X[n - m]).
 164     shifted_product = spectral_ops.ifft(spectral_density)
 165 
 166     # Cast back to real-valued if x was real to begin with.
 167     shifted_product = math_ops.cast(shifted_product, dtype)
 168 
 169     # Figure out if we can deduce the final static shape, and set max_lags.
 170     # Use x_rotated as a reference, because it has the time dimension in the far
 171     # right, and was created before we performed all sorts of crazy shape
 172     # manipulations.
 173     know_static_shape = True
 174     if not x_rotated.shape.is_fully_defined():
 175       know_static_shape = False
 176     if max_lags is None:
 177       max_lags = x_len - 1
 178     else:
 179       max_lags = ops.convert_to_tensor(max_lags, name="max_lags")
 180       max_lags_ = tensor_util.constant_value(max_lags)
 181       if max_lags_ is None or not know_static_shape:
 182         know_static_shape = False
 183         max_lags = math_ops.minimum(x_len - 1, max_lags)
 184       else:
 185         max_lags = min(x_len - 1, max_lags_)
 186 
 187     # Chop off the padding.
 188     # We allow users to provide a huge max_lags, but cut it off here.
 189     # shifted_product_chopped.shape = x_rotated.shape[:-1] + [max_lags]
 190     shifted_product_chopped = shifted_product[..., :max_lags + 1]
 191 
 192     # If possible, set shape.
 193     if know_static_shape:
 194       chopped_shape = x_rotated.shape.as_list()
 195       chopped_shape[-1] = min(x_len, max_lags + 1)
 196       shifted_product_chopped.set_shape(chopped_shape)
 197 
 198     # Recall R[m] is a sum of N / 2 - m nonzero terms x[n] Conj(x[n - m]).  The
 199     # other terms were zeros arising only due to zero padding.
 200     # `denominator = (N / 2 - m)` (defined below) is the proper term to
 201     # divide by to make this an unbiased estimate of the expectation
 202     # E[X[n] Conj(X[n - m])].
 203     x_len = math_ops.cast(x_len, dtype.real_dtype)
 204     max_lags = math_ops.cast(max_lags, dtype.real_dtype)
 205     denominator = x_len - math_ops.range(0., max_lags + 1.)
 206     denominator = math_ops.cast(denominator, dtype)
 207     shifted_product_rotated = shifted_product_chopped / denominator
 208 
 209     if normalize:
 210       shifted_product_rotated /= shifted_product_rotated[..., :1]
 211 
 212     # Transpose dimensions back to those of x.
 213     return util.rotate_transpose(shifted_product_rotated, -shift)

 216 # TODO(langmore) To make equivalent to numpy.percentile:
 217 #  Make work with a sequence of floats or single float for 'q'.
 218 #  Make work with "linear", "midpoint" interpolation. (linear should be default)
 219 def percentile(x,
 220                q,
 221                axis=None,
 222                interpolation=None,
 223                keep_dims=False,
 224                validate_args=False,
 225                name=None):
 226   """Compute the `q`-th percentile of `x`.
 227 
 228   Given a vector `x`, the `q`-th percentile of `x` is the value `q / 100` of the
 229   way from the minimum to the maximum in a sorted copy of `x`.
 230 
 231   The values and distances of the two nearest neighbors as well as the
 232   `interpolation` parameter will determine the percentile if the normalized
 233   ranking does not match the location of `q` exactly.
 234 
 235   This function is the same as the median if `q = 50`, the same as the minimum
 236   if `q = 0` and the same as the maximum if `q = 100`.
 237 
 238 
 239   ```python
 240   # Get 30th percentile with default ('nearest') interpolation.
 241   x = [1., 2., 3., 4.]
 242   percentile(x, q=30.)
 243   ==> 2.0
 244 
 245   # Get 30th percentile with 'lower' interpolation
 246   x = [1., 2., 3., 4.]
 247   percentile(x, q=30., interpolation='lower')
 248   ==> 1.0
 249 
 250   # Get 100th percentile (maximum).  By default, this is computed over every dim
 251   x = [[1., 2.]
 252        [3., 4.]]
 253   percentile(x, q=100.)
 254   ==> 4.0
 255 
 256   # Treat the leading dim as indexing samples, and find the 100th quantile (max)
 257   # over all such samples.
 258   x = [[1., 2.]
 259        [3., 4.]]
 260   percentile(x, q=100., axis=[0])
 261   ==> [3., 4.]
 262   ```
 263 
 264   Compare to `numpy.percentile`.
 265 
 266   Args:
 267     x:  Floating point `N-D` `Tensor` with `N > 0`.  If `axis` is not `None`,
 268       `x` must have statically known number of dimensions.
 269     q:  Scalar `Tensor` in `[0, 100]`. The percentile.
 270     axis:  Optional `0-D` or `1-D` integer `Tensor` with constant values.
 271       The axis that hold independent samples over which to return the desired
 272       percentile.  If `None` (the default), treat every dimension as a sample
 273       dimension, returning a scalar.
 274     interpolation : {"lower", "higher", "nearest"}.  Default: "nearest"
 275       This optional parameter specifies the interpolation method to
 276       use when the desired quantile lies between two data points `i < j`:
 277         * lower: `i`.
 278         * higher: `j`.
 279         * nearest: `i` or `j`, whichever is nearest.
 280     keep_dims:  Python `bool`. If `True`, the last dimension is kept with size 1
 281       If `False`, the last dimension is removed from the output shape.
 282     validate_args:  Whether to add runtime checks of argument validity.
 283       If False, and arguments are incorrect, correct behavior is not guaranteed.
 284     name:  A Python string name to give this `Op`.  Default is "percentile"
 285 
 286   Returns:
 287     A `(N - len(axis))` dimensional `Tensor` of same dtype as `x`, or, if
 288       `axis` is `None`, a scalar.
 289 
 290   Raises:
 291     ValueError:  If argument 'interpolation' is not an allowed type.
 292   """
 293   name = name or "percentile"
 294   allowed_interpolations = {"lower", "higher", "nearest"}
 295 
 296   if interpolation is None:
 297     interpolation = "nearest"
 298   else:
 299     if interpolation not in allowed_interpolations:
 300       raise ValueError("Argument 'interpolation' must be in %s.  Found %s" %
 301                        (allowed_interpolations, interpolation))
 302 
 303   with ops.name_scope(name, [x, q]):
 304     x = ops.convert_to_tensor(x, name="x")
 305     # Double is needed here and below, else we get the wrong index if the array
 306     # is huge along axis.
 307     q = math_ops.to_double(q, name="q")
 308     _get_static_ndims(q, expect_ndims=0)
 309 
 310     if validate_args:
 311       q = control_flow_ops.with_dependencies([
 312           check_ops.assert_rank(q, 0),
 313           check_ops.assert_greater_equal(q, math_ops.to_double(0.)),
 314           check_ops.assert_less_equal(q, math_ops.to_double(100.))
 315       ], q)
 316 
 317     if axis is None:
 318       y = array_ops.reshape(x, [-1])
 319     else:
 320       axis = ops.convert_to_tensor(axis, name="axis")
 321       check_ops.assert_integer(axis)
 322       axis_ndims = _get_static_ndims(
 323           axis, expect_static=True, expect_ndims_no_more_than=1)
 324       axis_const = tensor_util.constant_value(axis)
 325       if axis_const is None:
 326         raise ValueError(
 327             "Expected argument 'axis' to be statically available.  Found: %s" %
 328             axis)
 329       axis = axis_const
 330       if axis_ndims == 0:
 331         axis = [axis]
 332       axis = [int(a) for a in axis]
 333       x_ndims = _get_static_ndims(
 334           x, expect_static=True, expect_ndims_at_least=1)
 335       axis = _make_static_axis_non_negative(axis, x_ndims)
 336       y = _move_dims_to_flat_end(x, axis, x_ndims)
 337 
 338     frac_at_q_or_above = 1. - q / 100.
 339     d = math_ops.to_double(array_ops.shape(y)[-1])
 340 
 341     if interpolation == "lower":
 342       index = math_ops.ceil((d - 1) * frac_at_q_or_above)
 343     elif interpolation == "higher":
 344       index = math_ops.floor((d - 1) * frac_at_q_or_above)
 345     elif interpolation == "nearest":
 346       index = math_ops.round((d - 1) * frac_at_q_or_above)
 347 
 348     # If d is gigantic, then we would have d == d - 1, even in double... So
 349     # let's use max/min to avoid out of bounds errors.
 350     d = array_ops.shape(y)[-1]
 351     # d - 1 will be distinct from d in int32.
 352     index = clip_ops.clip_by_value(math_ops.to_int32(index), 0, d - 1)
 353 
 354     # Sort everything, not just the top 'k' entries, which allows multiple calls
 355     # to sort only once (under the hood) and use CSE.
 356     sorted_y = _sort_tensor(y)
 357 
 358     # result.shape = B
 359     result = sorted_y[..., index]
 360     result.set_shape(y.get_shape()[:-1])
 361 
 362     if keep_dims:
 363       if axis is None:
 364         # ones_vec = [1, 1,..., 1], total length = len(S) + len(B).
 365         ones_vec = array_ops.ones(
 366             shape=[_get_best_effort_ndims(x)], dtype=dtypes.int32)
 367         result *= array_ops.ones(ones_vec, dtype=x.dtype)
 368       else:
 369         result = _insert_back_keep_dims(result, axis)
 370 
 371     return result

 374 def _get_static_ndims(x,
 375                       expect_static=False,
 376                       expect_ndims=None,
 377                       expect_ndims_no_more_than=None,
 378                       expect_ndims_at_least=None):
 379   """Get static number of dimensions and assert that some expectations are met.
 380 
 381   This function returns the number of dimensions "ndims" of x, as a Python int.
 382 
 383   The optional expect arguments are used to check the ndims of x, but this is
 384   only done if the static ndims of x is not None.
 385 
 386   Args:
 387     x:  A Tensor.
 388     expect_static:  Expect `x` to have statically defined `ndims`.
 389     expect_ndims:  Optional Python integer.  If provided, assert that x has
 390       number of dimensions equal to this.
 391     expect_ndims_no_more_than:  Optional Python integer.  If provided, assert
 392       that x has no more than this many dimensions.
 393     expect_ndims_at_least:  Optional Python integer.  If provided, assert that
 394       x has at least this many dimensions.
 395 
 396   Returns:
 397     ndims:  A Python integer.
 398 
 399   Raises:
 400     ValueError:  If any of the expectations above are violated.
 401   """
 402   ndims = x.get_shape().ndims
 403   if ndims is None:
 404     shape_const = tensor_util.constant_value(array_ops.shape(x))
 405     if shape_const is not None:
 406       ndims = shape_const.ndim
 407 
 408   if ndims is None:
 409     if expect_static:
 410       raise ValueError(
 411           "Expected argument 'x' to have statically defined 'ndims'.  Found: " %
 412           x)
 413     return
 414 
 415   if expect_ndims is not None:
 416     ndims_message = ("Expected argument 'x' to have ndims %s.  Found tensor %s"
 417                      % (expect_ndims, x))
 418     if ndims != expect_ndims:
 419       raise ValueError(ndims_message)
 420 
 421   if expect_ndims_at_least is not None:
 422     ndims_at_least_message = (
 423         "Expected argument 'x' to have ndims >= %d.  Found tensor %s" % (
 424             expect_ndims_at_least, x))
 425     if ndims < expect_ndims_at_least:
 426       raise ValueError(ndims_at_least_message)
 427 
 428   if expect_ndims_no_more_than is not None:
 429     ndims_no_more_than_message = (
 430         "Expected argument 'x' to have ndims <= %d.  Found tensor %s" % (
 431             expect_ndims_no_more_than, x))
 432     if ndims > expect_ndims_no_more_than:
 433       raise ValueError(ndims_no_more_than_message)
 434 
 435   return ndims

 438 def _get_best_effort_ndims(x,
 439                            expect_ndims=None,
 440                            expect_ndims_at_least=None,
 441                            expect_ndims_no_more_than=None):
 442   """Get static ndims if possible.  Fallback on `tf.rank(x)`."""
 443   ndims_static = _get_static_ndims(
 444       x,
 445       expect_ndims=expect_ndims,
 446       expect_ndims_at_least=expect_ndims_at_least,
 447       expect_ndims_no_more_than=expect_ndims_no_more_than)
 448   if ndims_static is not None:
 449     return ndims_static
 450   return array_ops.rank(x)

 453 def _insert_back_keep_dims(x, axis):
 454   """Insert the dims in `axis` back as singletons after being removed.
 455 
 456   Args:
 457     x:  `Tensor`.
 458     axis:  Python list of integers.
 459 
 460   Returns:
 461     `Tensor` with same values as `x`, but additional singleton dimensions.
 462   """
 463   for i in sorted(axis):
 464     x = array_ops.expand_dims(x, axis=i)
 465   return x

 468 def _make_static_axis_non_negative(axis, ndims):
 469   """Convert possibly negatively indexed axis to non-negative.
 470 
 471   Args:
 472     axis:  Iterable over Python integers.
 473     ndims:  Number of dimensions into which axis indexes.
 474 
 475   Returns:
 476     A list of non-negative Python integers.
 477 
 478   Raises:
 479     ValueError: If values in `axis` are too big/small to index into `ndims`.
 480   """
 481   non_negative_axis = []
 482   for d in axis:
 483     if d >= 0:
 484       if d >= ndims:
 485         raise ValueError("dim %d not in the interval [0, %d]." % (d, ndims - 1))
 486       non_negative_axis.append(d)
 487     else:
 488       if d < -1 * ndims:
 489         raise ValueError(
 490             "Negatively indexed dim %d not in the interval [-%d, -1]" % (d,
 491                                                                          ndims))
 492       non_negative_axis.append(ndims + d)
 493   return non_negative_axis

 496 def _move_dims_to_flat_end(x, axis, x_ndims):
 497   """Move dims corresponding to `axis` in `x` to the end, then flatten.
 498 
 499   Args:
 500     x: `Tensor` with shape `[B0,B1,...,Bb]`.
 501     axis:  Python list of indices into dimensions of `x`.
 502     x_ndims:  Python integer holding number of dimensions in `x`.
 503 
 504   Returns:
 505     `Tensor` with value from `x` and dims in `axis` moved to end into one single
 506       dimension.
 507   """
 508   # Suppose x.shape = [a, b, c, d]
 509   # Suppose axis = [1, 3]
 510 
 511   # front_dims = [0, 2] in example above.
 512   front_dims = sorted(set(range(x_ndims)).difference(axis))
 513   # x_permed.shape = [a, c, b, d]
 514   x_permed = array_ops.transpose(x, perm=front_dims + list(axis))
 515 
 516   if x.get_shape().is_fully_defined():
 517     x_shape = x.get_shape().as_list()
 518     # front_shape = [a, c], end_shape = [b * d]
 519     front_shape = [x_shape[i] for i in front_dims]
 520     end_shape = [np.prod([x_shape[i] for i in axis])]
 521     full_shape = front_shape + end_shape
 522   else:
 523     front_shape = array_ops.shape(x_permed)[:x_ndims - len(axis)]
 524     end_shape = [-1]
 525     full_shape = array_ops.concat([front_shape, end_shape], axis=0)
 526   return array_ops.reshape(x_permed, shape=full_shape)

 529 def _sort_tensor(tensor):
 530   """Use `top_k` to sort a `Tensor` along the last dimension."""
 531   sorted_, _ = nn_ops.top_k(tensor, k=array_ops.shape(tensor)[-1])
 532   return sorted_
