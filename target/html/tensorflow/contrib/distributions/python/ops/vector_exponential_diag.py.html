  32 class VectorExponentialDiag(
  33     vector_exponential_linop.VectorExponentialLinearOperator):
  34   """The vectorization of the Exponential distribution on `R^k`.
  35 
  36   The vector exponential distribution is defined over a subset of `R^k`, and
  37   parameterized by a (batch of) length-`k` `loc` vector and a (batch of) `k x k`
  38   `scale` matrix:  `covariance = scale @ scale.T`, where `@` denotes
  39   matrix-multiplication.
  40 
  41   #### Mathematical Details
  42 
  43   The probability density function (pdf) is defined over the image of the
  44   `scale` matrix + `loc`, applied to the positive half-space:
  45   `Supp = {loc + scale @ x : x in R^k, x_1 > 0, ..., x_k > 0}`.  On this set,
  46 
  47   ```none
  48   pdf(y; loc, scale) = exp(-||x||_1) / Z,  for y in Supp
  49   x = inv(scale) @ (y - loc),
  50   Z = |det(scale)|,
  51   ```
  52 
  53   where:
  54 
  55   * `loc` is a vector in `R^k`,
  56   * `scale` is a linear operator in `R^{k x k}`, `cov = scale @ scale.T`,
  57   * `Z` denotes the normalization constant, and,
  58   * `||x||_1` denotes the `l1` norm of `x`, `sum_i |x_i|`.
  59 
  60   The VectorExponential distribution is a member of the [location-scale
  61   family](https://en.wikipedia.org/wiki/Location-scale_family), i.e., it can be
  62   constructed as,
  63 
  64   ```none
  65   X = (X_1, ..., X_k), each X_i ~ Exponential(rate=1)
  66   Y = (Y_1, ...,Y_k) = scale @ X + loc
  67   ```
  68 
  69   #### About `VectorExponential` and `Vector` distributions in TensorFlow.
  70 
  71   The `VectorExponential` is a non-standard distribution that has useful
  72   properties.
  73 
  74   The marginals `Y_1, ..., Y_k` are *not* Exponential random variables, due to
  75   the fact that the sum of Exponential random variables is not Exponential.
  76 
  77   Instead, `Y` is a vector whose components are linear combinations of
  78   Exponential random variables.  Thus, `Y` lives in the vector space generated
  79   by `vectors` of Exponential distributions.  This allows the user to decide the
  80   mean and covariance (by setting `loc` and `scale`), while preserving some
  81   properties of the Exponential distribution.  In particular, the tails of `Y_i`
  82   will be (up to polynomial factors) exponentially decaying.
  83 
  84   To see this last statement, note that the pdf of `Y_i` is the convolution of
  85   the pdf of `k` independent Exponential random variables.  One can then show by
  86   induction that distributions with exponential (up to polynomial factors) tails
  87   are closed under convolution.
  88 
  89 
  90   #### Examples
  91 
  92   ```python
  93   import tensorflow_probability as tfp
  94   tfd = tfp.distributions
  95 
  96   # Initialize a single 2-variate VectorExponential, supported on
  97   # {(x, y) in R^2 : x > 0, y > 0}.
  98 
  99   # The first component has pdf exp{-x}, the second 0.5 exp{-x / 2}
 100   vex = tfd.VectorExponentialDiag(scale_diag=[1., 2.])
 101 
 102   # Compute the pdf of an`R^2` observation; return a scalar.
 103   vex.prob([3., 4.]).eval()  # shape: []
 104 
 105   # Initialize a 2-batch of 3-variate Vector Exponential's.
 106   loc = [[1., 2, 3],
 107          [1., 0, 0]]              # shape: [2, 3]
 108   scale_diag = [[1., 2, 3],
 109                 [0.5, 1, 1.5]]     # shape: [2, 3]
 110 
 111   vex = tfd.VectorExponentialDiag(loc, scale_diag)
 112 
 113   # Compute the pdf of two `R^3` observations; return a length-2 vector.
 114   x = [[1.9, 2.2, 3.1],
 115        [10., 1.0, 9.0]]     # shape: [2, 3]
 116   vex.prob(x).eval()    # shape: [2]
 117   ```
 118 
 119   """
 120 
 121   @deprecation.deprecated(
 122       "2018-10-01",
 123       "The TensorFlow Distributions library has moved to "
 124       "TensorFlow Probability "
 125       "(https://github.com/tensorflow/probability). You "
 126       "should update all references to use `tfp.distributions` "
 127       "instead of `tf.contrib.distributions`.",
 128       warn_once=True)
 129   def __init__(self,
 130                loc=None,
 131                scale_diag=None,
 132                scale_identity_multiplier=None,
 133                validate_args=False,
 134                allow_nan_stats=True,
 135                name="VectorExponentialDiag"):
 136     """Construct Vector Exponential distribution supported on a subset of `R^k`.
 137 
 138     The `batch_shape` is the broadcast shape between `loc` and `scale`
 139     arguments.
 140 
 141     The `event_shape` is given by last dimension of the matrix implied by
 142     `scale`. The last dimension of `loc` (if provided) must broadcast with this.
 143 
 144     Recall that `covariance = scale @ scale.T`.
 145 
 146     ```none
 147     scale = diag(scale_diag + scale_identity_multiplier * ones(k))
 148     ```
 149 
 150     where:
 151 
 152     * `scale_diag.shape = [k]`, and,
 153     * `scale_identity_multiplier.shape = []`.
 154 
 155     Additional leading dimensions (if any) will index batches.
 156 
 157     If both `scale_diag` and `scale_identity_multiplier` are `None`, then
 158     `scale` is the Identity matrix.
 159 
 160     Args:
 161       loc: Floating-point `Tensor`. If this is set to `None`, `loc` is
 162         implicitly `0`. When specified, may have shape `[B1, ..., Bb, k]` where
 163         `b >= 0` and `k` is the event size.
 164       scale_diag: Non-zero, floating-point `Tensor` representing a diagonal
 165         matrix added to `scale`. May have shape `[B1, ..., Bb, k]`, `b >= 0`,
 166         and characterizes `b`-batches of `k x k` diagonal matrices added to
 167         `scale`. When both `scale_identity_multiplier` and `scale_diag` are
 168         `None` then `scale` is the `Identity`.
 169       scale_identity_multiplier: Non-zero, floating-point `Tensor` representing
 170         a scaled-identity-matrix added to `scale`. May have shape
 171         `[B1, ..., Bb]`, `b >= 0`, and characterizes `b`-batches of scaled
 172         `k x k` identity matrices added to `scale`. When both
 173         `scale_identity_multiplier` and `scale_diag` are `None` then `scale` is
 174         the `Identity`.
 175       validate_args: Python `bool`, default `False`. When `True` distribution
 176         parameters are checked for validity despite possibly degrading runtime
 177         performance. When `False` invalid inputs may silently render incorrect
 178         outputs.
 179       allow_nan_stats: Python `bool`, default `True`. When `True`,
 180         statistics (e.g., mean, mode, variance) use the value "`NaN`" to
 181         indicate the result is undefined. When `False`, an exception is raised
 182         if one or more of the statistic's batch members are undefined.
 183       name: Python `str` name prefixed to Ops created by this class.
 184 
 185     Raises:
 186       ValueError: if at most `scale_identity_multiplier` is specified.
 187     """
 188     parameters = dict(locals())
 189     with ops.name_scope(name) as name:
 190       with ops.name_scope("init", values=[
 191           loc, scale_diag, scale_identity_multiplier]):
 192         # No need to validate_args while making diag_scale.  The returned
 193         # LinearOperatorDiag has an assert_non_singular method that is called by
 194         # the Bijector.
 195         scale = distribution_util.make_diag_scale(
 196             loc=loc,
 197             scale_diag=scale_diag,
 198             scale_identity_multiplier=scale_identity_multiplier,
 199             validate_args=False,
 200             assert_positive=False)
 201     super(VectorExponentialDiag, self).__init__(
 202         loc=loc,
 203         scale=scale,
 204         validate_args=validate_args,
 205         allow_nan_stats=allow_nan_stats,
 206         name=name)
 207     self._parameters = parameters
