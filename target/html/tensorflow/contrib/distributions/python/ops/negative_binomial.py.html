  33 class NegativeBinomial(distribution.Distribution):
  34   """NegativeBinomial distribution.
  35 
  36   The NegativeBinomial distribution is related to the experiment of performing
  37   Bernoulli trials in sequence. Given a Bernoulli trial with probability `p` of
  38   success, the NegativeBinomial distribution represents the distribution over
  39   the number of successes `s` that occur until we observe `f` failures.
  40 
  41   The probability mass function (pmf) is,
  42 
  43   ```none
  44   pmf(s; f, p) = p**s (1 - p)**f / Z
  45   Z = s! (f - 1)! / (s + f - 1)!
  46   ```
  47 
  48   where:
  49   * `total_count = f`,
  50   * `probs = p`,
  51   * `Z` is the normalizaing constant, and,
  52   * `n!` is the factorial of `n`.
  53   """
  54 
  55   @deprecation.deprecated(
  56       "2018-10-01",
  57       "The TensorFlow Distributions library has moved to "
  58       "TensorFlow Probability "
  59       "(https://github.com/tensorflow/probability). You "
  60       "should update all references to use `tfp.distributions` "
  61       "instead of `tf.contrib.distributions`.",
  62       warn_once=True)
  63   def __init__(self,
  64                total_count,
  65                logits=None,
  66                probs=None,
  67                validate_args=False,
  68                allow_nan_stats=True,
  69                name="NegativeBinomial"):
  70     """Construct NegativeBinomial distributions.
  71 
  72     Args:
  73       total_count: Non-negative floating-point `Tensor` with shape
  74         broadcastable to `[B1,..., Bb]` with `b >= 0` and the same dtype as
  75         `probs` or `logits`. Defines this as a batch of `N1 x ... x Nm`
  76         different Negative Binomial distributions. In practice, this represents
  77         the number of negative Bernoulli trials to stop at (the `total_count`
  78         of failures), but this is still a valid distribution when
  79         `total_count` is a non-integer.
  80       logits: Floating-point `Tensor` with shape broadcastable to
  81         `[B1, ..., Bb]` where `b >= 0` indicates the number of batch dimensions.
  82         Each entry represents logits for the probability of success for
  83         independent Negative Binomial distributions and must be in the open
  84         interval `(-inf, inf)`. Only one of `logits` or `probs` should be
  85         specified.
  86       probs: Positive floating-point `Tensor` with shape broadcastable to
  87         `[B1, ..., Bb]` where `b >= 0` indicates the number of batch dimensions.
  88         Each entry represents the probability of success for independent
  89         Negative Binomial distributions and must be in the open interval
  90         `(0, 1)`. Only one of `logits` or `probs` should be specified.
  91       validate_args: Python `bool`, default `False`. When `True` distribution
  92         parameters are checked for validity despite possibly degrading runtime
  93         performance. When `False` invalid inputs may silently render incorrect
  94         outputs.
  95       allow_nan_stats: Python `bool`, default `True`. When `True`, statistics
  96         (e.g., mean, mode, variance) use the value "`NaN`" to indicate the
  97         result is undefined. When `False`, an exception is raised if one or
  98         more of the statistic's batch members are undefined.
  99       name: Python `str` name prefixed to Ops created by this class.
 100     """
 101 
 102     parameters = dict(locals())
 103     with ops.name_scope(name, values=[total_count, logits, probs]) as name:
 104       self._logits, self._probs = distribution_util.get_logits_and_probs(
 105           logits, probs, validate_args=validate_args, name=name)
 106       with ops.control_dependencies(
 107           [check_ops.assert_positive(total_count)] if validate_args else []):
 108         self._total_count = array_ops.identity(total_count)
 109 
 110     super(NegativeBinomial, self).__init__(
 111         dtype=self._probs.dtype,
 112         reparameterization_type=distribution.NOT_REPARAMETERIZED,
 113         validate_args=validate_args,
 114         allow_nan_stats=allow_nan_stats,
 115         parameters=parameters,
 116         graph_parents=[self._total_count, self._probs, self._logits],
 117         name=name)
 118 
 119   @property
 120   def total_count(self):
 121     """Number of negative trials."""
 122     return self._total_count
 123 
 124   @property
 125   def logits(self):
 126     """Log-odds of a `1` outcome (vs `0`)."""
 127     return self._logits
 128 
 129   @property
 130   def probs(self):
 131     """Probability of a `1` outcome (vs `0`)."""
 132     return self._probs
 133 
 134   def _batch_shape_tensor(self):
 135     return array_ops.broadcast_dynamic_shape(
 136         array_ops.shape(self.total_count),
 137         array_ops.shape(self.probs))
 138 
 139   def _batch_shape(self):
 140     return array_ops.broadcast_static_shape(
 141         self.total_count.get_shape(),
 142         self.probs.get_shape())
 143 
 144   def _event_shape_tensor(self):
 145     return array_ops.constant([], dtype=dtypes.int32)
 146 
 147   def _event_shape(self):
 148     return tensor_shape.scalar()
 149 
 150   def _sample_n(self, n, seed=None):
 151     # Here we use the fact that if:
 152     # lam ~ Gamma(concentration=total_count, rate=(1-probs)/probs)
 153     # then X ~ Poisson(lam) is Negative Binomially distributed.
 154     rate = random_ops.random_gamma(
 155         shape=[n],
 156         alpha=self.total_count,
 157         beta=math_ops.exp(-self.logits),
 158         dtype=self.dtype,
 159         seed=seed)
 160     return random_ops.random_poisson(
 161         rate,
 162         shape=[],
 163         dtype=self.dtype,
 164         seed=distribution_util.gen_new_seed(seed, "negative_binom"))
 165 
 166   def _cdf(self, x):
 167     if self.validate_args:
 168       x = distribution_util.embed_check_nonnegative_integer_form(x)
 169     return math_ops.betainc(self.total_count, 1. + x,
 170                             math_ops.sigmoid(-self.logits))
 171 
 172   def _log_prob(self, x):
 173     return (self._log_unnormalized_prob(x)
 174             - self._log_normalization(x))
 175 
 176   def _log_unnormalized_prob(self, x):
 177     if self.validate_args:
 178       x = distribution_util.embed_check_nonnegative_integer_form(x)
 179     return (self.total_count * math_ops.log_sigmoid(-self.logits)
 180             + x * math_ops.log_sigmoid(self.logits))
 181 
 182   def _log_normalization(self, x):
 183     if self.validate_args:
 184       x = distribution_util.embed_check_nonnegative_integer_form(x)
 185     return (-math_ops.lgamma(self.total_count + x)
 186             + math_ops.lgamma(1. + x)
 187             + math_ops.lgamma(self.total_count))
 188 
 189   def _mean(self):
 190     return self.total_count * math_ops.exp(self.logits)
 191 
 192   def _mode(self):
 193     adjusted_count = array_ops.where(
 194         1. < self.total_count,
 195         self.total_count - 1.,
 196         array_ops.zeros_like(self.total_count))
 197     return math_ops.floor(adjusted_count * math_ops.exp(self.logits))
 198 
 199   def _variance(self):
 200     return self._mean() / math_ops.sigmoid(-self.logits)
