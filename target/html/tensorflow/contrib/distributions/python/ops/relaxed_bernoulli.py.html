  34 class RelaxedBernoulli(transformed_distribution.TransformedDistribution):
  35   """RelaxedBernoulli distribution with temperature and logits parameters.
  36 
  37   The RelaxedBernoulli is a distribution over the unit interval (0,1), which
  38   continuously approximates a Bernoulli. The degree of approximation is
  39   controlled by a temperature: as the temperature goes to 0 the
  40   RelaxedBernoulli becomes discrete with a distribution described by the
  41   `logits` or `probs` parameters, as the temperature goes to infinity the
  42   RelaxedBernoulli becomes the constant distribution that is identically 0.5.
  43 
  44   The RelaxedBernoulli distribution is a reparameterized continuous
  45   distribution that is the binary special case of the RelaxedOneHotCategorical
  46   distribution (Maddison et al., 2016; Jang et al., 2016). For details on the
  47   binary special case see the appendix of Maddison et al. (2016) where it is
  48   referred to as BinConcrete. If you use this distribution, please cite both
  49   papers.
  50 
  51   Some care needs to be taken for loss functions that depend on the
  52   log-probability of RelaxedBernoullis, because computing log-probabilities of
  53   the RelaxedBernoulli can suffer from underflow issues. In many case loss
  54   functions such as these are invariant under invertible transformations of
  55   the random variables. The KL divergence, found in the variational autoencoder
  56   loss, is an example. Because RelaxedBernoullis are sampled by a Logistic
  57   random variable followed by a `tf.sigmoid` op, one solution is to treat
  58   the Logistic as the random variable and `tf.sigmoid` as downstream. The
  59   KL divergences of two Logistics, which are always followed by a `tf.sigmoid`
  60   op, is equivalent to evaluating KL divergences of RelaxedBernoulli samples.
  61   See Maddison et al., 2016 for more details where this distribution is called
  62   the BinConcrete.
  63 
  64   An alternative approach is to evaluate Bernoulli log probability or KL
  65   directly on relaxed samples, as done in Jang et al., 2016. In this case,
  66   guarantees on the loss are usually violated. For instance, using a Bernoulli
  67   KL in a relaxed ELBO is no longer a lower bound on the log marginal
  68   probability of the observation. Thus care and early stopping are important.
  69 
  70   #### Examples
  71 
  72   Creates three continuous distributions, which approximate 3 Bernoullis with
  73   probabilities (0.1, 0.5, 0.4). Samples from these distributions will be in
  74   the unit interval (0,1).
  75 
  76   ```python
  77   temperature = 0.5
  78   p = [0.1, 0.5, 0.4]
  79   dist = RelaxedBernoulli(temperature, probs=p)
  80   ```
  81 
  82   Creates three continuous distributions, which approximate 3 Bernoullis with
  83   logits (-2, 2, 0). Samples from these distributions will be in
  84   the unit interval (0,1).
  85 
  86   ```python
  87   temperature = 0.5
  88   logits = [-2, 2, 0]
  89   dist = RelaxedBernoulli(temperature, logits=logits)
  90   ```
  91 
  92   Creates three continuous distributions, whose sigmoid approximate 3 Bernoullis
  93   with logits (-2, 2, 0).
  94 
  95   ```python
  96   temperature = 0.5
  97   logits = [-2, 2, 0]
  98   dist = Logistic(logits/temperature, 1./temperature)
  99   samples = dist.sample()
 100   sigmoid_samples = tf.sigmoid(samples)
 101   # sigmoid_samples has the same distribution as samples from
 102   # RelaxedBernoulli(temperature, logits=logits)
 103   ```
 104 
 105   Creates three continuous distributions, which approximate 3 Bernoullis with
 106   logits (-2, 2, 0). Samples from these distributions will be in
 107   the unit interval (0,1). Because the temperature is very low, samples from
 108   these distributions are almost discrete, usually taking values very close to 0
 109   or 1.
 110 
 111   ```python
 112   temperature = 1e-5
 113   logits = [-2, 2, 0]
 114   dist = RelaxedBernoulli(temperature, logits=logits)
 115   ```
 116 
 117   Creates three continuous distributions, which approximate 3 Bernoullis with
 118   logits (-2, 2, 0). Samples from these distributions will be in
 119   the unit interval (0,1). Because the temperature is very high, samples from
 120   these distributions are usually close to the (0.5, 0.5, 0.5) vector.
 121 
 122   ```python
 123   temperature = 100
 124   logits = [-2, 2, 0]
 125   dist = RelaxedBernoulli(temperature, logits=logits)
 126   ```
 127 
 128   Chris J. Maddison, Andriy Mnih, and Yee Whye Teh. The Concrete Distribution:
 129   A Continuous Relaxation of Discrete Random Variables. 2016.
 130 
 131   Eric Jang, Shixiang Gu, and Ben Poole. Categorical Reparameterization with
 132   Gumbel-Softmax. 2016.
 133   """
 134 
 135   @deprecation.deprecated(
 136       "2018-10-01",
 137       "The TensorFlow Distributions library has moved to "
 138       "TensorFlow Probability "
 139       "(https://github.com/tensorflow/probability). You "
 140       "should update all references to use `tfp.distributions` "
 141       "instead of `tf.contrib.distributions`.",
 142       warn_once=True)
 143   def __init__(self,
 144                temperature,
 145                logits=None,
 146                probs=None,
 147                validate_args=False,
 148                allow_nan_stats=True,
 149                name="RelaxedBernoulli"):
 150     """Construct RelaxedBernoulli distributions.
 151 
 152     Args:
 153       temperature: An 0-D `Tensor`, representing the temperature
 154         of a set of RelaxedBernoulli distributions. The temperature should be
 155         positive.
 156       logits: An N-D `Tensor` representing the log-odds
 157         of a positive event. Each entry in the `Tensor` parametrizes
 158         an independent RelaxedBernoulli distribution where the probability of an
 159         event is sigmoid(logits). Only one of `logits` or `probs` should be
 160         passed in.
 161       probs: An N-D `Tensor` representing the probability of a positive event.
 162         Each entry in the `Tensor` parameterizes an independent Bernoulli
 163         distribution. Only one of `logits` or `probs` should be passed in.
 164       validate_args: Python `bool`, default `False`. When `True` distribution
 165         parameters are checked for validity despite possibly degrading runtime
 166         performance. When `False` invalid inputs may silently render incorrect
 167         outputs.
 168       allow_nan_stats: Python `bool`, default `True`. When `True`, statistics
 169         (e.g., mean, mode, variance) use the value "`NaN`" to indicate the
 170         result is undefined. When `False`, an exception is raised if one or
 171         more of the statistic's batch members are undefined.
 172       name: Python `str` name prefixed to Ops created by this class.
 173 
 174     Raises:
 175       ValueError: If both `probs` and `logits` are passed, or if neither.
 176     """
 177     parameters = dict(locals())
 178     with ops.name_scope(name, values=[logits, probs, temperature]) as name:
 179       with ops.control_dependencies([check_ops.assert_positive(temperature)]
 180                                     if validate_args else []):
 181         self._temperature = array_ops.identity(temperature, name="temperature")
 182       self._logits, self._probs = distribution_util.get_logits_and_probs(
 183           logits=logits, probs=probs, validate_args=validate_args)
 184       super(RelaxedBernoulli, self).__init__(
 185           distribution=logistic.Logistic(
 186               self._logits / self._temperature,
 187               1. / self._temperature,
 188               validate_args=validate_args,
 189               allow_nan_stats=allow_nan_stats,
 190               name=name + "/Logistic"),
 191           bijector=Sigmoid(validate_args=validate_args),
 192           validate_args=validate_args,
 193           name=name)
 194     self._parameters = parameters
 195 
 196   @staticmethod
 197   def _param_shapes(sample_shape):
 198     return {"logits": ops.convert_to_tensor(sample_shape, dtype=dtypes.int32)}
 199 
 200   @property
 201   def temperature(self):
 202     """Distribution parameter for the location."""
 203     return self._temperature
 204 
 205   @property
 206   def logits(self):
 207     """Log-odds of `1`."""
 208     return self._logits
 209 
 210   @property
 211   def probs(self):
 212     """Probability of `1`."""
 213     return self._probs
