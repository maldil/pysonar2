  37 class Logistic(distribution.Distribution):
  38   """The Logistic distribution with location `loc` and `scale` parameters.
  39 
  40   #### Mathematical details
  41 
  42   The cumulative density function of this distribution is:
  43 
  44   ```none
  45   cdf(x; mu, sigma) = 1 / (1 + exp(-(x - mu) / sigma))
  46   ```
  47 
  48   where `loc = mu` and `scale = sigma`.
  49 
  50   The Logistic distribution is a member of the [location-scale family](
  51   https://en.wikipedia.org/wiki/Location-scale_family), i.e., it can be
  52   constructed as,
  53 
  54   ```none
  55   X ~ Logistic(loc=0, scale=1)
  56   Y = loc + scale * X
  57   ```
  58 
  59   #### Examples
  60 
  61   Examples of initialization of one or a batch of distributions.
  62 
  63   ```python
  64   import tensorflow_probability as tfp
  65   tfd = tfp.distributions
  66 
  67   # Define a single scalar Logistic distribution.
  68   dist = tfd.Logistic(loc=0., scale=3.)
  69 
  70   # Evaluate the cdf at 1, returning a scalar.
  71   dist.cdf(1.)
  72 
  73   # Define a batch of two scalar valued Logistics.
  74   # The first has mean 1 and scale 11, the second 2 and 22.
  75   dist = tfd.Logistic(loc=[1, 2.], scale=[11, 22.])
  76 
  77   # Evaluate the pdf of the first distribution on 0, and the second on 1.5,
  78   # returning a length two tensor.
  79   dist.prob([0, 1.5])
  80 
  81   # Get 3 samples, returning a 3 x 2 tensor.
  82   dist.sample([3])
  83 
  84   # Arguments are broadcast when possible.
  85   # Define a batch of two scalar valued Logistics.
  86   # Both have mean 1, but different scales.
  87   dist = tfd.Logistic(loc=1., scale=[11, 22.])
  88 
  89   # Evaluate the pdf of both distributions on the same point, 3.0,
  90   # returning a length 2 tensor.
  91   dist.prob(3.0)
  92   ```
  93 
  94   """
  95 
  96   @deprecation.deprecated(
  97       "2018-10-01",
  98       "The TensorFlow Distributions library has moved to "
  99       "TensorFlow Probability "
 100       "(https://github.com/tensorflow/probability). You "
 101       "should update all references to use `tfp.distributions` "
 102       "instead of `tf.contrib.distributions`.",
 103       warn_once=True)
 104   def __init__(self,
 105                loc,
 106                scale,
 107                validate_args=False,
 108                allow_nan_stats=True,
 109                name="Logistic"):
 110     """Construct Logistic distributions with mean and scale `loc` and `scale`.
 111 
 112     The parameters `loc` and `scale` must be shaped in a way that supports
 113     broadcasting (e.g. `loc + scale` is a valid operation).
 114 
 115     Args:
 116       loc: Floating point tensor, the means of the distribution(s).
 117       scale: Floating point tensor, the scales of the distribution(s). Must
 118         contain only positive values.
 119       validate_args: Python `bool`, default `False`. When `True` distribution
 120         parameters are checked for validity despite possibly degrading runtime
 121         performance. When `False` invalid inputs may silently render incorrect
 122         outputs.
 123       allow_nan_stats: Python `bool`, default `True`. When `True`, statistics
 124         (e.g., mean, mode, variance) use the value "`NaN`" to indicate the
 125         result is undefined. When `False`, an exception is raised if one or
 126         more of the statistic's batch members are undefined.
 127       name: The name to give Ops created by the initializer.
 128 
 129     Raises:
 130       TypeError: if loc and scale are different dtypes.
 131     """
 132     parameters = dict(locals())
 133     with ops.name_scope(name, values=[loc, scale]) as name:
 134       with ops.control_dependencies([check_ops.assert_positive(scale)] if
 135                                     validate_args else []):
 136         self._loc = array_ops.identity(loc, name="loc")
 137         self._scale = array_ops.identity(scale, name="scale")
 138         check_ops.assert_same_float_dtype([self._loc, self._scale])
 139     super(Logistic, self).__init__(
 140         dtype=self._scale.dtype,
 141         reparameterization_type=distribution.FULLY_REPARAMETERIZED,
 142         validate_args=validate_args,
 143         allow_nan_stats=allow_nan_stats,
 144         parameters=parameters,
 145         graph_parents=[self._loc, self._scale],
 146         name=name)
 147 
 148   @staticmethod
 149   def _param_shapes(sample_shape):
 150     return dict(
 151         zip(("loc", "scale"), ([ops.convert_to_tensor(
 152             sample_shape, dtype=dtypes.int32)] * 2)))
 153 
 154   @property
 155   def loc(self):
 156     """Distribution parameter for the location."""
 157     return self._loc
 158 
 159   @property
 160   def scale(self):
 161     """Distribution parameter for scale."""
 162     return self._scale
 163 
 164   def _batch_shape_tensor(self):
 165     return array_ops.broadcast_dynamic_shape(
 166         array_ops.shape(self.loc), array_ops.shape(self.scale))
 167 
 168   def _batch_shape(self):
 169     return array_ops.broadcast_static_shape(
 170         self.loc.get_shape(), self.scale.get_shape())
 171 
 172   def _event_shape_tensor(self):
 173     return constant_op.constant([], dtype=dtypes.int32)
 174 
 175   def _event_shape(self):
 176     return tensor_shape.scalar()
 177 
 178   def _sample_n(self, n, seed=None):
 179     # Uniform variates must be sampled from the open-interval `(0, 1)` rather
 180     # than `[0, 1)`. To do so, we use `np.finfo(self.dtype.as_numpy_dtype).tiny`
 181     # because it is the smallest, positive, "normal" number. A "normal" number
 182     # is such that the mantissa has an implicit leading 1. Normal, positive
 183     # numbers x, y have the reasonable property that, `x + y >= max(x, y)`. In
 184     # this case, a subnormal number (i.e., np.nextafter) can cause us to sample
 185     # 0.
 186     uniform = random_ops.random_uniform(
 187         shape=array_ops.concat([[n], self.batch_shape_tensor()], 0),
 188         minval=np.finfo(self.dtype.as_numpy_dtype).tiny,
 189         maxval=1.,
 190         dtype=self.dtype,
 191         seed=seed)
 192     sampled = math_ops.log(uniform) - math_ops.log1p(-1. * uniform)
 193     return sampled * self.scale + self.loc
 194 
 195   def _log_prob(self, x):
 196     return self._log_unnormalized_prob(x) - self._log_normalization()
 197 
 198   def _log_cdf(self, x):
 199     return -nn_ops.softplus(-self._z(x))
 200 
 201   def _cdf(self, x):
 202     return math_ops.sigmoid(self._z(x))
 203 
 204   def _log_survival_function(self, x):
 205     return -nn_ops.softplus(self._z(x))
 206 
 207   def _survival_function(self, x):
 208     return math_ops.sigmoid(-self._z(x))
 209 
 210   def _log_unnormalized_prob(self, x):
 211     z = self._z(x)
 212     return - z - 2. * nn_ops.softplus(-z)
 213 
 214   def _log_normalization(self):
 215     return math_ops.log(self.scale)
 216 
 217   def _entropy(self):
 218     # Use broadcasting rules to calculate the full broadcast sigma.
 219     scale = self.scale * array_ops.ones_like(self.loc)
 220     return 2 + math_ops.log(scale)
 221 
 222   def _mean(self):
 223     return self.loc * array_ops.ones_like(self.scale)
 224 
 225   def _stddev(self):
 226     return self.scale * array_ops.ones_like(self.loc) * math.pi / math.sqrt(3)
 227 
 228   def _mode(self):
 229     return self._mean()
 230 
 231   def _z(self, x):
 232     """Standardize input `x` to a unit logistic."""
 233     with ops.name_scope("standardize", values=[x]):
 234       return (x - self.loc) / self.scale
