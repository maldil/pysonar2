  39 class BatchReshape(distribution_lib.Distribution):
  40   """The Batch-Reshaping distribution.
  41 
  42   This "meta-distribution" reshapes the batch dimensions of another
  43   distribution.
  44 
  45   #### Examples
  46 
  47   ```python
  48   import tensorflow_probability as tfp
  49   tfd = tfp.distributions
  50 
  51   dtype = np.float32
  52   dims = 2
  53   new_batch_shape = [1, 2, -1]
  54   old_batch_shape = [6]
  55 
  56   scale = np.ones(old_batch_shape + [dims], dtype)
  57   mvn = tfd.MultivariateNormalDiag(scale_diag=scale)
  58   reshape_mvn = tfd.BatchReshape(
  59       distribution=mvn,
  60       batch_shape=new_batch_shape,
  61       validate_args=True)
  62 
  63   reshape_mvn.batch_shape
  64   # ==> [1, 2, 3]
  65 
  66   x = reshape_mvn.sample(sample_shape=[4, 5])
  67   x.shape
  68   # ==> [4, 5, 1, 2, 3, 2] == sample_shape + new_batch_shape + [dims]
  69 
  70   reshape_mvn.log_prob(x).shape
  71   # ==> [4, 5, 1, 2, 3] == sample_shape + new_batch_shape
  72   ```
  73 
  74   """
  75 
  76   @deprecation.deprecated(
  77       "2018-10-01",
  78       "The TensorFlow Distributions library has moved to "
  79       "TensorFlow Probability "
  80       "(https://github.com/tensorflow/probability). You "
  81       "should update all references to use `tfp.distributions` "
  82       "instead of `tf.contrib.distributions`.",
  83       warn_once=True)
  84   def __init__(self,
  85                distribution,
  86                batch_shape,
  87                validate_args=False,
  88                allow_nan_stats=True,
  89                name=None):
  90     """Construct BatchReshape distribution.
  91 
  92     Args:
  93       distribution: The base distribution instance to reshape. Typically an
  94         instance of `Distribution`.
  95       batch_shape: Positive `int`-like vector-shaped `Tensor` representing
  96         the new shape of the batch dimensions. Up to one dimension may contain
  97         `-1`, meaning the remainder of the batch size.
  98       validate_args: Python `bool`, default `False`. When `True` distribution
  99         parameters are checked for validity despite possibly degrading runtime
 100         performance. When `False` invalid inputs may silently render incorrect
 101         outputs.
 102       allow_nan_stats: Python `bool`, default `True`. When `True`, statistics
 103         (e.g., mean, mode, variance) use the value "`NaN`" to indicate the
 104         result is undefined. When `False`, an exception is raised if one or
 105         more of the statistic's batch members are undefined.
 106       name: The name to give Ops created by the initializer.
 107         Default value: `"BatchReshape" + distribution.name`.
 108 
 109     Raises:
 110       ValueError: if `batch_shape` is not a vector.
 111       ValueError: if `batch_shape` has non-positive elements.
 112       ValueError: if `batch_shape` size is not the same as a
 113         `distribution.batch_shape` size.
 114     """
 115     parameters = dict(locals())
 116     name = name or "BatchReshape" + distribution.name
 117     with ops.name_scope(name, values=[batch_shape]) as name:
 118       # The unexpanded batch shape may contain up to one dimension of -1.
 119       self._batch_shape_unexpanded = ops.convert_to_tensor(
 120           batch_shape, dtype=dtypes.int32, name="batch_shape")
 121       validate_init_args_statically(distribution, self._batch_shape_unexpanded)
 122       batch_shape, batch_shape_static, runtime_assertions = calculate_reshape(
 123           distribution.batch_shape_tensor(), self._batch_shape_unexpanded,
 124           validate_args)
 125       self._distribution = distribution
 126       self._batch_shape_ = batch_shape
 127       self._batch_shape_static = batch_shape_static
 128       self._runtime_assertions = runtime_assertions
 129       super(BatchReshape, self).__init__(
 130           dtype=distribution.dtype,
 131           reparameterization_type=distribution.reparameterization_type,
 132           validate_args=validate_args,
 133           allow_nan_stats=allow_nan_stats,
 134           parameters=parameters,
 135           graph_parents=(
 136               [self._batch_shape_unexpanded] + distribution._graph_parents),  # pylint: disable=protected-access
 137           name=name)
 138 
 139   @property
 140   def distribution(self):
 141     return self._distribution
 142 
 143   def _batch_shape_tensor(self):
 144     with ops.control_dependencies(self._runtime_assertions):
 145       return array_ops.identity(self._batch_shape_)
 146 
 147   def _batch_shape(self):
 148     return self._batch_shape_static
 149 
 150   def _event_shape_tensor(self):
 151     with ops.control_dependencies(self._runtime_assertions):
 152       return array_ops.identity(self.distribution.event_shape_tensor())
 153 
 154   def _event_shape(self):
 155     return self.distribution.event_shape
 156 
 157   def _sample_n(self, n, seed=None):
 158     with ops.control_dependencies(self._runtime_assertions):
 159       x = self.distribution.sample(sample_shape=n, seed=seed)
 160       new_shape = array_ops.concat(
 161           [
 162               [n],
 163               self._batch_shape_unexpanded,
 164               self.event_shape_tensor(),
 165           ],
 166           axis=0)
 167       return array_ops.reshape(x, new_shape)
 168 
 169   def _log_prob(self, x):
 170     return self._call_reshape_input_output(
 171         self.distribution.log_prob, x)
 172 
 173   def _prob(self, x):
 174     return self._call_reshape_input_output(
 175         self.distribution.prob, x)
 176 
 177   def _log_cdf(self, x):
 178     return self._call_reshape_input_output(
 179         self.distribution.log_cdf, x)
 180 
 181   def _cdf(self, x):
 182     return self._call_reshape_input_output(
 183         self.distribution.cdf, x)
 184 
 185   def _log_survival_function(self, x):
 186     return self._call_reshape_input_output(
 187         self.distribution.log_survival_function, x)
 188 
 189   def _survival_function(self, x):
 190     return self._call_reshape_input_output(
 191         self.distribution.survival_function, x)
 192 
 193   def _entropy(self):
 194     return self._call_and_reshape_output(
 195         self.distribution.entropy,
 196         [],
 197         [tensor_shape.scalar()])
 198 
 199   def _mean(self):
 200     return self._call_and_reshape_output(self.distribution.mean)
 201 
 202   def _mode(self):
 203     return self._call_and_reshape_output(self.distribution.mode)
 204 
 205   def _stddev(self):
 206     return self._call_and_reshape_output(self.distribution.stddev)
 207 
 208   def _variance(self):
 209     return self._call_and_reshape_output(self.distribution.variance)
 210 
 211   def _covariance(self):
 212     return self._call_and_reshape_output(
 213         self.distribution.covariance,
 214         [self.event_shape_tensor()]*2,
 215         [self.event_shape]*2)

 217   def _sample_shape(self, x):
 218     """Computes graph and static `sample_shape`."""
 219     x_ndims = (array_ops.rank(x) if x.shape.ndims is None else x.shape.ndims)
 220     event_ndims = (array_ops.size(self.event_shape_tensor())
 221                    if self.event_shape.ndims is None
 222                    else self.event_shape.ndims)
 223     batch_ndims = (
 224         array_ops.size(self._batch_shape_unexpanded)
 225         if self.batch_shape.ndims is None else self.batch_shape.ndims)
 226     sample_ndims = x_ndims - batch_ndims - event_ndims
 227     if isinstance(sample_ndims, int):
 228       static_sample_shape = x.shape[:sample_ndims]
 229     else:
 230       static_sample_shape = tensor_shape.TensorShape(None)
 231     if static_sample_shape.is_fully_defined():
 232       sample_shape = np.int32(static_sample_shape.as_list())
 233     else:
 234       sample_shape = array_ops.shape(x)[:sample_ndims]
 235     return sample_shape, static_sample_shape
 236 
 237   def _call_reshape_input_output(self, fn, x):
 238     """Calls `fn`, appropriately reshaping its input `x` and output."""
 239     with ops.control_dependencies(
 240         self._runtime_assertions + self._validate_sample_arg(x)):
 241       sample_shape, static_sample_shape = self._sample_shape(x)
 242       old_shape = array_ops.concat([
 243           sample_shape,
 244           self.distribution.batch_shape_tensor(),
 245           self.event_shape_tensor(),
 246       ], axis=0)
 247       result = fn(array_ops.reshape(x, old_shape))
 248       new_shape = array_ops.concat(
 249           [
 250               sample_shape,
 251               self._batch_shape_unexpanded,
 252           ], axis=0)
 253       result = array_ops.reshape(result, new_shape)
 254       if (static_sample_shape.ndims is not None and
 255           self.batch_shape.ndims is not None):
 256         new_shape = static_sample_shape.concatenate(self.batch_shape)
 257         result.set_shape(result.shape.merge_with(new_shape))
 258       return result
 259 
 260   def _call_and_reshape_output(
 261       self,
 262       fn,
 263       event_shape_list=None,
 264       static_event_shape_list=None):
 265     """Calls `fn` and appropriately reshapes its output."""
 266     with ops.control_dependencies(self._runtime_assertions):
 267       if event_shape_list is None:
 268         event_shape_list = [self._event_shape_tensor()]
 269       if static_event_shape_list is None:
 270         static_event_shape_list = [self.event_shape]
 271       new_shape = array_ops.concat(
 272           [self._batch_shape_unexpanded] + event_shape_list, axis=0)
 273       result = array_ops.reshape(fn(), new_shape)
 274       if (self.batch_shape.ndims is not None and
 275           self.event_shape.ndims is not None):
 276         event_shape = tensor_shape.TensorShape([])
 277         for rss in static_event_shape_list:
 278           event_shape = event_shape.concatenate(rss)
 279         static_shape = result.shape.merge_with(
 280             self.batch_shape.concatenate(event_shape))
 281         result.set_shape(static_shape)
 282       return result
 283 
 284   def _validate_sample_arg(self, x):
 285     """Helper which validates sample arg, e.g., input to `log_prob`."""
 286     with ops.name_scope(name="validate_sample_arg", values=[x]):
 287       x_ndims = (array_ops.rank(x) if x.shape.ndims is None else x.shape.ndims)
 288       event_ndims = (array_ops.size(self.event_shape_tensor())
 289                      if self.event_shape.ndims is None
 290                      else self.event_shape.ndims)
 291       batch_ndims = (
 292           array_ops.size(self._batch_shape_unexpanded)
 293           if self.batch_shape.ndims is None else self.batch_shape.ndims)
 294       expected_batch_event_ndims = batch_ndims + event_ndims
 295 
 296       if (isinstance(x_ndims, int) and
 297           isinstance(expected_batch_event_ndims, int)):
 298         if x_ndims < expected_batch_event_ndims:
 299           raise NotImplementedError(
 300               "Broadcasting is not supported; too few batch and event dims "
 301               "(expected at least {}, saw {}).".format(
 302                   expected_batch_event_ndims, x_ndims))
 303         ndims_assertion = []
 304       elif self.validate_args:
 305         ndims_assertion = [
 306             check_ops.assert_greater_equal(
 307                 x_ndims,
 308                 expected_batch_event_ndims,
 309                 message=("Broadcasting is not supported; too few "
 310                          "batch and event dims."),
 311                 name="assert_batch_and_event_ndims_large_enough"),
 312         ]
 313 
 314       if (self.batch_shape.is_fully_defined() and
 315           self.event_shape.is_fully_defined()):
 316         expected_batch_event_shape = np.int32(self.batch_shape.concatenate(
 317             self.event_shape).as_list())
 318       else:
 319         expected_batch_event_shape = array_ops.concat([
 320             self.batch_shape_tensor(),
 321             self.event_shape_tensor(),
 322         ], axis=0)
 323 
 324       sample_ndims = x_ndims - expected_batch_event_ndims
 325       if isinstance(sample_ndims, int):
 326         sample_ndims = max(sample_ndims, 0)
 327       if (isinstance(sample_ndims, int) and
 328           x.shape[sample_ndims:].is_fully_defined()):
 329         actual_batch_event_shape = np.int32(x.shape[sample_ndims:].as_list())
 330       else:
 331         sample_ndims = math_ops.maximum(sample_ndims, 0)
 332         actual_batch_event_shape = array_ops.shape(x)[sample_ndims:]
 333 
 334       if (isinstance(expected_batch_event_shape, np.ndarray) and
 335           isinstance(actual_batch_event_shape, np.ndarray)):
 336         if any(expected_batch_event_shape != actual_batch_event_shape):
 337           raise NotImplementedError("Broadcasting is not supported; "
 338                                     "unexpected batch and event shape "
 339                                     "(expected {}, saw {}).".format(
 340                                         expected_batch_event_shape,
 341                                         actual_batch_event_shape))
 342         # We need to set the final runtime-assertions to `ndims_assertion` since
 343         # its possible this assertion was created. We could add a condition to
 344         # only do so if `self.validate_args == True`, however this is redundant
 345         # as `ndims_assertion` already encodes this information.
 346         runtime_assertions = ndims_assertion
 347       elif self.validate_args:
 348         # We need to make the `ndims_assertion` a control dep because otherwise
 349         # TF itself might raise an exception owing to this assertion being
 350         # ill-defined, ie, one cannot even compare different rank Tensors.
 351         with ops.control_dependencies(ndims_assertion):
 352           shape_assertion = check_ops.assert_equal(
 353               expected_batch_event_shape,
 354               actual_batch_event_shape,
 355               message=("Broadcasting is not supported; "
 356                        "unexpected batch and event shape."),
 357               name="assert_batch_and_event_shape_same")
 358         runtime_assertions = [shape_assertion]
 359       else:
 360         runtime_assertions = []
 361 
 362       return runtime_assertions

 373 def calculate_reshape(original_shape, new_shape, validate=False, name=None):
 374   """Calculates the reshaped dimensions (replacing up to one -1 in reshape)."""
 375   batch_shape_static = tensor_util.constant_value_as_shape(new_shape)
 376   if batch_shape_static.is_fully_defined():
 377     return np.int32(batch_shape_static.as_list()), batch_shape_static, []
 378   with ops.name_scope(name, "calculate_reshape", [original_shape, new_shape]):
 379     original_size = math_ops.reduce_prod(original_shape)
 380     implicit_dim = math_ops.equal(new_shape, -1)
 381     size_implicit_dim = (
 382         original_size // math_ops.maximum(1, -math_ops.reduce_prod(new_shape)))
 383     new_ndims = array_ops.shape(new_shape)
 384     expanded_new_shape = array_ops.where(  # Assumes exactly one `-1`.
 385         implicit_dim, array_ops.fill(new_ndims, size_implicit_dim), new_shape)
 386     validations = [] if not validate else [
 387         check_ops.assert_rank(
 388             original_shape, 1, message="Original shape must be a vector."),
 389         check_ops.assert_rank(
 390             new_shape, 1, message="New shape must be a vector."),
 391         check_ops.assert_less_equal(
 392             math_ops.count_nonzero(implicit_dim, dtype=dtypes.int32),
 393             1,
 394             message="At most one dimension can be unknown."),
 395         check_ops.assert_positive(
 396             expanded_new_shape, message="Shape elements must be >=-1."),
 397         check_ops.assert_equal(
 398             math_ops.reduce_prod(expanded_new_shape),
 399             original_size,
 400             message="Shape sizes do not match."),
 401     ]
 402     return expanded_new_shape, batch_shape_static, validations

 406     "2018-10-01",
 407     "The TensorFlow Distributions library has moved to "
 408     "TensorFlow Probability "
 409     "(https://github.com/tensorflow/probability). You "

 412     warn_once=True)
 413 def validate_init_args_statically(distribution, batch_shape):
 414   """Helper to __init__ which makes or raises assertions."""
 415   if batch_shape.shape.ndims is not None:
 416     if batch_shape.shape.ndims != 1:
 417       raise ValueError("`batch_shape` must be a vector "
 418                        "(saw rank: {}).".format(batch_shape.shape.ndims))
 419 
 420   batch_shape_static = tensor_util.constant_value_as_shape(batch_shape)
 421   batch_size_static = batch_shape_static.num_elements()
 422   dist_batch_size_static = distribution.batch_shape.num_elements()
 423 
 424   if batch_size_static is not None and dist_batch_size_static is not None:
 425     if batch_size_static != dist_batch_size_static:
 426       raise ValueError("`batch_shape` size ({}) must match "
 427                        "`distribution.batch_shape` size ({}).".format(
 428                            batch_size_static, dist_batch_size_static))
 429 
 430   if batch_shape_static.dims is not None:
 431     if any(
 432         dim.value is not None and dim.value < 1 for dim in batch_shape_static):
 433       raise ValueError("`batch_shape` elements must be >=-1.")
