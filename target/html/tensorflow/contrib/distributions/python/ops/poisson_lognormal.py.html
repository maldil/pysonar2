  54 def quadrature_scheme_lognormal_gauss_hermite(
  55     loc, scale, quadrature_size,
  56     validate_args=False, name=None):  # pylint: disable=unused-argument
  57   """Use Gauss-Hermite quadrature to form quadrature on positive-reals.
  58 
  59   Note: for a given `quadrature_size`, this method is generally less accurate
  60   than `quadrature_scheme_lognormal_quantiles`.
  61 
  62   Args:
  63     loc: `float`-like (batch of) scalar `Tensor`; the location parameter of
  64       the LogNormal prior.
  65     scale: `float`-like (batch of) scalar `Tensor`; the scale parameter of
  66       the LogNormal prior.
  67     quadrature_size: Python `int` scalar representing the number of quadrature
  68       points.
  69     validate_args: Python `bool`, default `False`. When `True` distribution
  70       parameters are checked for validity despite possibly degrading runtime
  71       performance. When `False` invalid inputs may silently render incorrect
  72       outputs.
  73     name: Python `str` name prefixed to Ops created by this class.
  74 
  75   Returns:
  76     grid: (Batch of) length-`quadrature_size` vectors representing the
  77       `log_rate` parameters of a `Poisson`.
  78     probs: (Batch of) length-`quadrature_size` vectors representing the
  79       weight associate with each `grid` value.
  80   """
  81   with ops.name_scope(name, "vector_diffeomixture_quadrature_gauss_hermite",
  82                       [loc, scale]):
  83     grid, probs = np.polynomial.hermite.hermgauss(deg=quadrature_size)
  84     grid = grid.astype(loc.dtype.as_numpy_dtype)
  85     probs = probs.astype(loc.dtype.as_numpy_dtype)
  86     probs /= np.linalg.norm(probs, ord=1, keepdims=True)
  87     probs = ops.convert_to_tensor(probs, name="probs", dtype=loc.dtype)
  88     # The following maps the broadcast of `loc` and `scale` to each grid
  89     # point, i.e., we are creating several log-rates that correspond to the
  90     # different Gauss-Hermite quadrature points and (possible) batches of
  91     # `loc` and `scale`.
  92     grid = (loc[..., array_ops.newaxis]
  93             + np.sqrt(2.) * scale[..., array_ops.newaxis] * grid)
  94     return grid, probs

 105 def quadrature_scheme_lognormal_quantiles(
 106     loc, scale, quadrature_size,
 107     validate_args=False, name=None):
 108   """Use LogNormal quantiles to form quadrature on positive-reals.
 109 
 110   Args:
 111     loc: `float`-like (batch of) scalar `Tensor`; the location parameter of
 112       the LogNormal prior.
 113     scale: `float`-like (batch of) scalar `Tensor`; the scale parameter of
 114       the LogNormal prior.
 115     quadrature_size: Python `int` scalar representing the number of quadrature
 116       points.
 117     validate_args: Python `bool`, default `False`. When `True` distribution
 118       parameters are checked for validity despite possibly degrading runtime
 119       performance. When `False` invalid inputs may silently render incorrect
 120       outputs.
 121     name: Python `str` name prefixed to Ops created by this class.
 122 
 123   Returns:
 124     grid: (Batch of) length-`quadrature_size` vectors representing the
 125       `log_rate` parameters of a `Poisson`.
 126     probs: (Batch of) length-`quadrature_size` vectors representing the
 127       weight associate with each `grid` value.
 128   """
 129   with ops.name_scope(name, "quadrature_scheme_lognormal_quantiles",
 130                       [loc, scale]):
 131     # Create a LogNormal distribution.
 132     dist = transformed_lib.TransformedDistribution(
 133         distribution=normal_lib.Normal(loc=loc, scale=scale),
 134         bijector=Exp(),
 135         validate_args=validate_args)
 136     batch_ndims = dist.batch_shape.ndims
 137     if batch_ndims is None:
 138       batch_ndims = array_ops.shape(dist.batch_shape_tensor())[0]
 139 
 140     def _compute_quantiles():
 141       """Helper to build quantiles."""
 142       # Omit {0, 1} since they might lead to Inf/NaN.
 143       zero = array_ops.zeros([], dtype=dist.dtype)
 144       edges = math_ops.linspace(zero, 1., quadrature_size + 3)[1:-1]
 145       # Expand edges so its broadcast across batch dims.
 146       edges = array_ops.reshape(edges, shape=array_ops.concat([
 147           [-1], array_ops.ones([batch_ndims], dtype=dtypes.int32)], axis=0))
 148       quantiles = dist.quantile(edges)
 149       # Cyclically permute left by one.
 150       perm = array_ops.concat([
 151           math_ops.range(1, 1 + batch_ndims), [0]], axis=0)
 152       quantiles = array_ops.transpose(quantiles, perm)
 153       return quantiles
 154     quantiles = _compute_quantiles()
 155 
 156     # Compute grid as quantile midpoints.
 157     grid = (quantiles[..., :-1] + quantiles[..., 1:]) / 2.
 158     # Set shape hints.
 159     grid.set_shape(dist.batch_shape.concatenate([quadrature_size]))
 160 
 161     # By construction probs is constant, i.e., `1 / quadrature_size`. This is
 162     # important, because non-constant probs leads to non-reparameterizable
 163     # samples.
 164     probs = array_ops.fill(
 165         dims=[quadrature_size],
 166         value=1. / math_ops.cast(quadrature_size, dist.dtype))
 167 
 168     return grid, probs

 171 class PoissonLogNormalQuadratureCompound(distribution_lib.Distribution):
 172   """`PoissonLogNormalQuadratureCompound` distribution.
 173 
 174   The `PoissonLogNormalQuadratureCompound` is an approximation to a
 175   Poisson-LogNormal [compound distribution](
 176   https://en.wikipedia.org/wiki/Compound_probability_distribution), i.e.,
 177 
 178   ```none
 179   p(k|loc, scale)
 180   = int_{R_+} dl LogNormal(l | loc, scale) Poisson(k | l)
 181   approx= sum{ prob[d] Poisson(k | lambda(grid[d])) : d=0, ..., deg-1 }
 182   ```
 183 
 184   By default, the `grid` is chosen as quantiles of the `LogNormal` distribution
 185   parameterized by `loc`, `scale` and the `prob` vector is
 186   `[1. / quadrature_size]*quadrature_size`.
 187 
 188   In the non-approximation case, a draw from the LogNormal prior represents the
 189   Poisson rate parameter. Unfortunately, the non-approximate distribution lacks
 190   an analytical probability density function (pdf). Therefore the
 191   `PoissonLogNormalQuadratureCompound` class implements an approximation based
 192   on [quadrature](https://en.wikipedia.org/wiki/Numerical_integration).
 193 
 194   Note: although the `PoissonLogNormalQuadratureCompound` is approximately the
 195   Poisson-LogNormal compound distribution, it is itself a valid distribution.
 196   Viz., it possesses a `sample`, `log_prob`, `mean`, `variance`, etc. which are
 197   all mutually consistent.
 198 
 199   #### Mathematical Details
 200 
 201   The `PoissonLogNormalQuadratureCompound` approximates a Poisson-LogNormal
 202   [compound distribution](
 203   https://en.wikipedia.org/wiki/Compound_probability_distribution). Using
 204   variable-substitution and [numerical quadrature](
 205   https://en.wikipedia.org/wiki/Numerical_integration) (default:
 206   based on `LogNormal` quantiles) we can redefine the distribution to be a
 207   parameter-less convex combination of `deg` different Poisson samples.
 208 
 209   That is, defined over positive integers, this distribution is parameterized
 210   by a (batch of) `loc` and `scale` scalars.
 211 
 212   The probability density function (pdf) is,
 213 
 214   ```none
 215   pdf(k | loc, scale, deg)
 216     = sum{ prob[d] Poisson(k | lambda=exp(grid[d]))
 217           : d=0, ..., deg-1 }
 218   ```
 219 
 220   #### Examples
 221 
 222   ```python
 223   import tensorflow_probability as tfp
 224   tfd = tfp.distributions
 225 
 226   # Create two batches of PoissonLogNormalQuadratureCompounds, one with
 227   # prior `loc = 0.` and another with `loc = 1.` In both cases `scale = 1.`
 228   pln = tfd.PoissonLogNormalQuadratureCompound(
 229       loc=[0., -0.5],
 230       scale=1.,
 231       quadrature_size=10,
 232       validate_args=True)
 233   """
 234 
 235   @deprecation.deprecated(
 236       "2018-10-01",
 237       "The TensorFlow Distributions library has moved to "
 238       "TensorFlow Probability "
 239       "(https://github.com/tensorflow/probability). You "

 241       "instead of `tf.contrib.distributions`.",
 242       warn_once=True)
 243   def __init__(self,
 244                loc,
 245                scale,
 246                quadrature_size=8,
 247                quadrature_fn=quadrature_scheme_lognormal_quantiles,
 248                validate_args=False,
 249                allow_nan_stats=True,
 250                name="PoissonLogNormalQuadratureCompound"):
 251     """Constructs the PoissonLogNormalQuadratureCompound`.
 252 
 253     Note: `probs` returned by (optional) `quadrature_fn` are presumed to be
 254     either a length-`quadrature_size` vector or a batch of vectors in 1-to-1
 255     correspondence with the returned `grid`. (I.e., broadcasting is only
 256     partially supported.)
 257 
 258     Args:
 259       loc: `float`-like (batch of) scalar `Tensor`; the location parameter of
 260         the LogNormal prior.
 261       scale: `float`-like (batch of) scalar `Tensor`; the scale parameter of
 262         the LogNormal prior.
 263       quadrature_size: Python `int` scalar representing the number of quadrature
 264         points.
 265       quadrature_fn: Python callable taking `loc`, `scale`,
 266         `quadrature_size`, `validate_args` and returning `tuple(grid, probs)`
 267         representing the LogNormal grid and corresponding normalized weight.
 268         normalized) weight.
 269         Default value: `quadrature_scheme_lognormal_quantiles`.
 270       validate_args: Python `bool`, default `False`. When `True` distribution
 271         parameters are checked for validity despite possibly degrading runtime
 272         performance. When `False` invalid inputs may silently render incorrect
 273         outputs.
 274       allow_nan_stats: Python `bool`, default `True`. When `True`,
 275         statistics (e.g., mean, mode, variance) use the value "`NaN`" to
 276         indicate the result is undefined. When `False`, an exception is raised
 277         if one or more of the statistic's batch members are undefined.
 278       name: Python `str` name prefixed to Ops created by this class.
 279 
 280     Raises:
 281       TypeError: if `quadrature_grid` and `quadrature_probs` have different base
 282         `dtype`.
 283     """
 284     parameters = dict(locals())
 285     with ops.name_scope(name, values=[loc, scale]) as name:
 286       if loc is not None:
 287         loc = ops.convert_to_tensor(loc, name="loc")
 288       if scale is not None:
 289         scale = ops.convert_to_tensor(
 290             scale, dtype=None if loc is None else loc.dtype, name="scale")
 291       self._quadrature_grid, self._quadrature_probs = tuple(quadrature_fn(
 292           loc, scale, quadrature_size, validate_args))
 293 
 294       dt = self._quadrature_grid.dtype
 295       if dt.base_dtype != self._quadrature_probs.dtype.base_dtype:
 296         raise TypeError("Quadrature grid dtype ({}) does not match quadrature "
 297                         "probs dtype ({}).".format(
 298                             dt.name, self._quadrature_probs.dtype.name))
 299 
 300       self._distribution = poisson_lib.Poisson(
 301           log_rate=self._quadrature_grid,
 302           validate_args=validate_args,
 303           allow_nan_stats=allow_nan_stats)
 304 
 305       self._mixture_distribution = categorical_lib.Categorical(
 306           logits=math_ops.log(self._quadrature_probs),
 307           validate_args=validate_args,
 308           allow_nan_stats=allow_nan_stats)
 309 
 310       self._loc = loc
 311       self._scale = scale
 312       self._quadrature_size = quadrature_size
 313 
 314       super(PoissonLogNormalQuadratureCompound, self).__init__(
 315           dtype=dt,
 316           reparameterization_type=distribution_lib.NOT_REPARAMETERIZED,
 317           validate_args=validate_args,
 318           allow_nan_stats=allow_nan_stats,
 319           parameters=parameters,
 320           graph_parents=[loc, scale],
 321           name=name)

 347   def _batch_shape_tensor(self):
 348     return array_ops.broadcast_dynamic_shape(
 349         self.distribution.batch_shape_tensor(),
 350         array_ops.shape(self.mixture_distribution.logits))[:-1]
 351 
 352   def _batch_shape(self):
 353     return array_ops.broadcast_static_shape(
 354         self.distribution.batch_shape,
 355         self.mixture_distribution.logits.shape)[:-1]
 356 
 357   def _event_shape(self):
 358     return tensor_shape.scalar()
 359 
 360   def _sample_n(self, n, seed=None):
 361     # Get ids as a [n, batch_size]-shaped matrix, unless batch_shape=[] then get
 362     # ids as a [n]-shaped vector.
 363     batch_size = self.batch_shape.num_elements()
 364     if batch_size is None:
 365       batch_size = math_ops.reduce_prod(self.batch_shape_tensor())
 366     # We need to "sample extra" from the mixture distribution if it doesn't
 367     # already specify a probs vector for each batch coordinate.
 368     # We only support this kind of reduced broadcasting, i.e., there is exactly
 369     # one probs vector for all batch dims or one for each.
 370     ids = self._mixture_distribution.sample(
 371         sample_shape=concat_vectors(
 372             [n],
 373             distribution_util.pick_vector(
 374                 self.mixture_distribution.is_scalar_batch(),
 375                 [batch_size],
 376                 np.int32([]))),
 377         seed=distribution_util.gen_new_seed(
 378             seed, "poisson_lognormal_quadrature_compound"))
 379     # We need to flatten batch dims in case mixture_distribution has its own
 380     # batch dims.
 381     ids = array_ops.reshape(ids, shape=concat_vectors(
 382         [n],
 383         distribution_util.pick_vector(
 384             self.is_scalar_batch(),
 385             np.int32([]),
 386             np.int32([-1]))))
 387 
 388     # Stride `quadrature_size` for `batch_size` number of times.
 389     offset = math_ops.range(start=0,
 390                             limit=batch_size * self._quadrature_size,
 391                             delta=self._quadrature_size,
 392                             dtype=ids.dtype)
 393     ids += offset
 394     rate = array_ops.gather(
 395         array_ops.reshape(self.distribution.rate, shape=[-1]), ids)
 396     rate = array_ops.reshape(
 397         rate, shape=concat_vectors([n], self.batch_shape_tensor()))
 398     return random_ops.random_poisson(
 399         lam=rate, shape=[], dtype=self.dtype, seed=seed)

 401   def _log_prob(self, x):
 402     return math_ops.reduce_logsumexp(
 403         (self.mixture_distribution.logits
 404          + self.distribution.log_prob(x[..., array_ops.newaxis])),
 405         axis=-1)
 406 
 407   def _mean(self):
 408     return math_ops.exp(
 409         math_ops.reduce_logsumexp(
 410             self.mixture_distribution.logits + self.distribution.log_rate,
 411             axis=-1))
 412 
 413   def _variance(self):
 414     return math_ops.exp(self._log_variance())
 415 
 416   def _stddev(self):
 417     return math_ops.exp(0.5 * self._log_variance())
 418 
 419   def _log_variance(self):
 420     # Following calculation is based on law of total variance:
 421     #
 422     # Var[Z] = E[Var[Z | V]] + Var[E[Z | V]]
 423     #
 424     # where,
 425     #
 426     # Z|v ~ interpolate_affine[v](distribution)
 427     # V ~ mixture_distribution
 428     #
 429     # thus,
 430     #
 431     # E[Var[Z | V]] = sum{ prob[d] Var[d] : d=0, ..., deg-1 }
 432     # Var[E[Z | V]] = sum{ prob[d] (Mean[d] - Mean)**2 : d=0, ..., deg-1 }
 433     v = array_ops.stack([
 434         # log(self.distribution.variance()) = log(Var[d]) = log(rate[d])
 435         self.distribution.log_rate,
 436         # log((Mean[d] - Mean)**2)
 437         2. * math_ops.log(
 438             math_ops.abs(self.distribution.mean()
 439                          - self._mean()[..., array_ops.newaxis])),
 440     ], axis=-1)
 441     return math_ops.reduce_logsumexp(
 442         self.mixture_distribution.logits[..., array_ops.newaxis] + v,
 443         axis=[-2, -1])

 454 def concat_vectors(*args):
 455   """Concatenates input vectors, statically if possible."""
 456   args_ = [distribution_util.static_value(x) for x in args]
 457   if any(vec is None for vec in args_):
 458     return array_ops.concat(args, axis=0)
 459   return [val for vec in args_ for val in vec]
