  37 class Chi2(gamma.Gamma):
  38   """Chi2 distribution.
  39 
  40   The Chi2 distribution is defined over positive real numbers using a degrees of
  41   freedom ("df") parameter.
  42 
  43   #### Mathematical Details
  44 
  45   The probability density function (pdf) is,
  46 
  47   ```none
  48   pdf(x; df, x > 0) = x**(0.5 df - 1) exp(-0.5 x) / Z
  49   Z = 2**(0.5 df) Gamma(0.5 df)
  50   ```
  51 
  52   where:
  53 
  54   * `df` denotes the degrees of freedom,
  55   * `Z` is the normalization constant, and,
  56   * `Gamma` is the [gamma function](
  57     https://en.wikipedia.org/wiki/Gamma_function).
  58 
  59   The Chi2 distribution is a special case of the Gamma distribution, i.e.,
  60 
  61   ```python
  62   Chi2(df) = Gamma(concentration=0.5 * df, rate=0.5)
  63   ```
  64 
  65   """
  66 
  67   @deprecation.deprecated(
  68       "2018-10-01",
  69       "The TensorFlow Distributions library has moved to "
  70       "TensorFlow Probability "
  71       "(https://github.com/tensorflow/probability). You "
  72       "should update all references to use `tfp.distributions` "
  73       "instead of `tf.contrib.distributions`.",
  74       warn_once=True)
  75   def __init__(self,
  76                df,
  77                validate_args=False,
  78                allow_nan_stats=True,
  79                name="Chi2"):
  80     """Construct Chi2 distributions with parameter `df`.
  81 
  82     Args:
  83       df: Floating point tensor, the degrees of freedom of the
  84         distribution(s). `df` must contain only positive values.
  85       validate_args: Python `bool`, default `False`. When `True` distribution
  86         parameters are checked for validity despite possibly degrading runtime
  87         performance. When `False` invalid inputs may silently render incorrect
  88         outputs.
  89       allow_nan_stats: Python `bool`, default `True`. When `True`, statistics
  90         (e.g., mean, mode, variance) use the value "`NaN`" to indicate the
  91         result is undefined. When `False`, an exception is raised if one or
  92         more of the statistic's batch members are undefined.
  93       name: Python `str` name prefixed to Ops created by this class.
  94     """
  95     parameters = dict(locals())
  96     # Even though all stats of chi2 are defined for valid parameters, this is
  97     # not true in the parent class "gamma."  therefore, passing
  98     # allow_nan_stats=True
  99     # through to the parent class results in unnecessary asserts.
 100     with ops.name_scope(name, values=[df]) as name:
 101       with ops.control_dependencies([
 102           check_ops.assert_positive(df),
 103       ] if validate_args else []):
 104         self._df = array_ops.identity(df, name="df")
 105 
 106       super(Chi2, self).__init__(
 107           concentration=0.5 * self._df,
 108           rate=constant_op.constant(0.5, dtype=self._df.dtype),
 109           validate_args=validate_args,
 110           allow_nan_stats=allow_nan_stats,
 111           name=name)
 112     self._parameters = parameters
 113 
 114   @staticmethod
 115   def _param_shapes(sample_shape):
 116     return {"df": ops.convert_to_tensor(sample_shape, dtype=dtypes.int32)}
 117 
 118   @property
 119   def df(self):
 120     return self._df

 123 class Chi2WithAbsDf(Chi2):
 124   """Chi2 with parameter transform `df = floor(abs(df))`."""
 125 
 126   @deprecation.deprecated(
 127       "2018-10-01",
 128       "The TensorFlow Distributions library has moved to "
 129       "TensorFlow Probability "
 130       "(https://github.com/tensorflow/probability). You "
 131       "should update all references to use `tfp.distributions` "
 132       "instead of `tf.contrib.distributions`.",
 133       warn_once=True)
 134   def __init__(self,
 135                df,
 136                validate_args=False,
 137                allow_nan_stats=True,
 138                name="Chi2WithAbsDf"):
 139     parameters = dict(locals())
 140     with ops.name_scope(name, values=[df]) as name:
 141       super(Chi2WithAbsDf, self).__init__(
 142           df=math_ops.floor(
 143               math_ops.abs(df, name="abs_df"),
 144               name="floor_abs_df"),
 145           validate_args=validate_args,
 146           allow_nan_stats=allow_nan_stats,
 147           name=name)
 148     self._parameters = parameters
