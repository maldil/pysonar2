  34 class MixtureSameFamily(distribution.Distribution):
  35   """Mixture (same-family) distribution.
  36 
  37   The `MixtureSameFamily` distribution implements a (batch of) mixture
  38   distribution where all components are from different parameterizations of the
  39   same distribution type. It is parameterized by a `Categorical` "selecting
  40   distribution" (over `k` components) and a components distribution, i.e., a
  41   `Distribution` with a rightmost batch shape (equal to `[k]`) which indexes
  42   each (batch of) component.
  43 
  44   #### Examples
  45 
  46   ```python
  47   import tensorflow_probability as tfp
  48   tfd = tfp.distributions
  49 
  50   ### Create a mixture of two scalar Gaussians:
  51 
  52   gm = tfd.MixtureSameFamily(
  53       mixture_distribution=tfd.Categorical(
  54           probs=[0.3, 0.7]),
  55       components_distribution=tfd.Normal(
  56         loc=[-1., 1],       # One for each component.
  57         scale=[0.1, 0.5]))  # And same here.
  58 
  59   gm.mean()
  60   # ==> 0.4
  61 
  62   gm.variance()
  63   # ==> 1.018
  64 
  65   # Plot PDF.
  66   x = np.linspace(-2., 3., int(1e4), dtype=np.float32)
  67   import matplotlib.pyplot as plt
  68   plt.plot(x, gm.prob(x).eval());
  69 
  70   ### Create a mixture of two Bivariate Gaussians:
  71 
  72   gm = tfd.MixtureSameFamily(
  73       mixture_distribution=tfd.Categorical(
  74           probs=[0.3, 0.7]),
  75       components_distribution=tfd.MultivariateNormalDiag(
  76           loc=[[-1., 1],  # component 1
  77                [1, -1]],  # component 2
  78           scale_identity_multiplier=[.3, .6]))
  79 
  80   gm.mean()
  81   # ==> array([ 0.4, -0.4], dtype=float32)
  82 
  83   gm.covariance()
  84   # ==> array([[ 1.119, -0.84],
  85   #            [-0.84,  1.119]], dtype=float32)
  86 
  87   # Plot PDF contours.
  88   def meshgrid(x, y=x):
  89     [gx, gy] = np.meshgrid(x, y, indexing='ij')
  90     gx, gy = np.float32(gx), np.float32(gy)
  91     grid = np.concatenate([gx.ravel()[None, :], gy.ravel()[None, :]], axis=0)
  92     return grid.T.reshape(x.size, y.size, 2)
  93   grid = meshgrid(np.linspace(-2, 2, 100, dtype=np.float32))
  94   plt.contour(grid[..., 0], grid[..., 1], gm.prob(grid).eval());
  95 
  96   ```
  97 
  98   """
  99 
 100   @deprecation.deprecated(
 101       "2018-10-01",
 102       "The TensorFlow Distributions library has moved to "
 103       "TensorFlow Probability "
 104       "(https://github.com/tensorflow/probability). You "
 105       "should update all references to use `tfp.distributions` "
 106       "instead of `tf.contrib.distributions`.",
 107       warn_once=True)
 108   def __init__(self,
 109                mixture_distribution,
 110                components_distribution,
 111                validate_args=False,
 112                allow_nan_stats=True,
 113                name="MixtureSameFamily"):
 114     """Construct a `MixtureSameFamily` distribution.
 115 
 116     Args:
 117       mixture_distribution: `tfp.distributions.Categorical`-like instance.
 118         Manages the probability of selecting components. The number of
 119         categories must match the rightmost batch dimension of the
 120         `components_distribution`. Must have either scalar `batch_shape` or
 121         `batch_shape` matching `components_distribution.batch_shape[:-1]`.
 122       components_distribution: `tfp.distributions.Distribution`-like instance.
 123         Right-most batch dimension indexes components.
 124       validate_args: Python `bool`, default `False`. When `True` distribution
 125         parameters are checked for validity despite possibly degrading runtime
 126         performance. When `False` invalid inputs may silently render incorrect
 127         outputs.
 128       allow_nan_stats: Python `bool`, default `True`. When `True`, statistics
 129         (e.g., mean, mode, variance) use the value "`NaN`" to indicate the
 130         result is undefined. When `False`, an exception is raised if one or
 131         more of the statistic's batch members are undefined.
 132       name: Python `str` name prefixed to Ops created by this class.
 133 
 134     Raises:
 135       ValueError: `if not mixture_distribution.dtype.is_integer`.
 136       ValueError: if mixture_distribution does not have scalar `event_shape`.
 137       ValueError: if `mixture_distribution.batch_shape` and
 138         `components_distribution.batch_shape[:-1]` are both fully defined and
 139         the former is neither scalar nor equal to the latter.
 140       ValueError: if `mixture_distribution` categories does not equal
 141         `components_distribution` rightmost batch shape.
 142     """
 143     parameters = dict(locals())
 144     with ops.name_scope(name) as name:
 145       self._mixture_distribution = mixture_distribution
 146       self._components_distribution = components_distribution
 147       self._runtime_assertions = []
 148 
 149       s = components_distribution.event_shape_tensor()
 150       self._event_ndims = (s.shape[0].value
 151                            if s.shape.with_rank_at_least(1)[0].value is not None
 152                            else array_ops.shape(s)[0])
 153 
 154       if not mixture_distribution.dtype.is_integer:
 155         raise ValueError(
 156             "`mixture_distribution.dtype` ({}) is not over integers".format(
 157                 mixture_distribution.dtype.name))
 158 
 159       if (mixture_distribution.event_shape.ndims is not None
 160           and mixture_distribution.event_shape.ndims != 0):
 161         raise ValueError("`mixture_distribution` must have scalar `event_dim`s")
 162       elif validate_args:
 163         self._runtime_assertions += [
 164             control_flow_ops.assert_has_rank(
 165                 mixture_distribution.event_shape_tensor(), 0,
 166                 message="`mixture_distribution` must have scalar `event_dim`s"),
 167         ]
 168 
 169       mdbs = mixture_distribution.batch_shape
 170       cdbs = components_distribution.batch_shape.with_rank_at_least(1)[:-1]
 171       if mdbs.is_fully_defined() and cdbs.is_fully_defined():
 172         if mdbs.ndims != 0 and mdbs != cdbs:
 173           raise ValueError(
 174               "`mixture_distribution.batch_shape` (`{}`) is not "
 175               "compatible with `components_distribution.batch_shape` "
 176               "(`{}`)".format(mdbs.as_list(), cdbs.as_list()))
 177       elif validate_args:
 178         mdbs = mixture_distribution.batch_shape_tensor()
 179         cdbs = components_distribution.batch_shape_tensor()[:-1]
 180         self._runtime_assertions += [
 181             control_flow_ops.assert_equal(
 182                 distribution_util.pick_vector(
 183                     mixture_distribution.is_scalar_batch(), cdbs, mdbs),
 184                 cdbs,
 185                 message=(
 186                     "`mixture_distribution.batch_shape` is not "
 187                     "compatible with `components_distribution.batch_shape`"))]
 188 
 189       km = mixture_distribution.logits.shape.with_rank_at_least(1)[-1].value
 190       kc = components_distribution.batch_shape.with_rank_at_least(1)[-1].value
 191       if km is not None and kc is not None and km != kc:
 192         raise ValueError("`mixture_distribution components` ({}) does not "
 193                          "equal `components_distribution.batch_shape[-1]` "
 194                          "({})".format(km, kc))
 195       elif validate_args:
 196         km = array_ops.shape(mixture_distribution.logits)[-1]
 197         kc = components_distribution.batch_shape_tensor()[-1]
 198         self._runtime_assertions += [
 199             control_flow_ops.assert_equal(
 200                 km, kc,
 201                 message=("`mixture_distribution components` does not equal "
 202                          "`components_distribution.batch_shape[-1:]`")),
 203         ]
 204       elif km is None:
 205         km = array_ops.shape(mixture_distribution.logits)[-1]
 206 
 207       self._num_components = km
 208 
 209       super(MixtureSameFamily, self).__init__(
 210           dtype=self._components_distribution.dtype,
 211           reparameterization_type=distribution.NOT_REPARAMETERIZED,
 212           validate_args=validate_args,
 213           allow_nan_stats=allow_nan_stats,
 214           parameters=parameters,
 215           graph_parents=(
 216               self._mixture_distribution._graph_parents  # pylint: disable=protected-access
 217               + self._components_distribution._graph_parents),  # pylint: disable=protected-access
 218           name=name)
 219 
 220   @property
 221   def mixture_distribution(self):
 222     return self._mixture_distribution
 223 
 224   @property
 225   def components_distribution(self):
 226     return self._components_distribution
 227 
 228   def _batch_shape_tensor(self):
 229     with ops.control_dependencies(self._runtime_assertions):
 230       return self.components_distribution.batch_shape_tensor()[:-1]
 231 
 232   def _batch_shape(self):
 233     return self.components_distribution.batch_shape.with_rank_at_least(1)[:-1]
 234 
 235   def _event_shape_tensor(self):
 236     with ops.control_dependencies(self._runtime_assertions):
 237       return self.components_distribution.event_shape_tensor()
 238 
 239   def _event_shape(self):
 240     return self.components_distribution.event_shape
 241 
 242   def _sample_n(self, n, seed):
 243     with ops.control_dependencies(self._runtime_assertions):
 244       x = self.components_distribution.sample(n)             # [n, B, k, E]
 245       # TODO(jvdillon): Consider using tf.gather (by way of index unrolling).
 246       npdt = x.dtype.as_numpy_dtype
 247       mask = array_ops.one_hot(
 248           indices=self.mixture_distribution.sample(n),       # [n, B]
 249           depth=self._num_components,                        # == k
 250           on_value=np.ones([], dtype=npdt),
 251           off_value=np.zeros([], dtype=npdt))                # [n, B, k]
 252       mask = distribution_utils.pad_mixture_dimensions(
 253           mask, self, self.mixture_distribution,
 254           self._event_shape().ndims)                         # [n, B, k, [1]*e]
 255       return math_ops.reduce_sum(
 256           x * mask, axis=-1 - self._event_ndims)             # [n, B, E]
 257 
 258   def _log_prob(self, x):
 259     with ops.control_dependencies(self._runtime_assertions):
 260       x = self._pad_sample_dims(x)
 261       log_prob_x = self.components_distribution.log_prob(x)  # [S, B, k]
 262       log_mix_prob = nn_ops.log_softmax(
 263           self.mixture_distribution.logits, axis=-1)         # [B, k]
 264       return math_ops.reduce_logsumexp(
 265           log_prob_x + log_mix_prob, axis=-1)                # [S, B]
 266 
 267   def _mean(self):
 268     with ops.control_dependencies(self._runtime_assertions):
 269       probs = distribution_utils.pad_mixture_dimensions(
 270           self.mixture_distribution.probs, self, self.mixture_distribution,
 271           self._event_shape().ndims)                         # [B, k, [1]*e]
 272       return math_ops.reduce_sum(
 273           probs * self.components_distribution.mean(),
 274           axis=-1 - self._event_ndims)                       # [B, E]
 275 
 276   def _log_cdf(self, x):
 277     x = self._pad_sample_dims(x)
 278     log_cdf_x = self.components_distribution.log_cdf(x)      # [S, B, k]
 279     log_mix_prob = nn_ops.log_softmax(
 280         self.mixture_distribution.logits, axis=-1)           # [B, k]
 281     return math_ops.reduce_logsumexp(
 282         log_cdf_x + log_mix_prob, axis=-1)                   # [S, B]
 283 
 284   def _variance(self):
 285     with ops.control_dependencies(self._runtime_assertions):
 286       # Law of total variance: Var(Y) = E[Var(Y|X)] + Var(E[Y|X])
 287       probs = distribution_utils.pad_mixture_dimensions(
 288           self.mixture_distribution.probs, self, self.mixture_distribution,
 289           self._event_shape().ndims)                         # [B, k, [1]*e]
 290       mean_cond_var = math_ops.reduce_sum(
 291           probs * self.components_distribution.variance(),
 292           axis=-1 - self._event_ndims)                       # [B, E]
 293       var_cond_mean = math_ops.reduce_sum(
 294           probs * math_ops.squared_difference(
 295               self.components_distribution.mean(),
 296               self._pad_sample_dims(self._mean())),
 297           axis=-1 - self._event_ndims)                       # [B, E]
 298       return mean_cond_var + var_cond_mean                   # [B, E]
 299 
 300   def _covariance(self):
 301     static_event_ndims = self.event_shape.ndims
 302     if static_event_ndims != 1:
 303       # Covariance is defined only for vector distributions.
 304       raise NotImplementedError("covariance is not implemented")
 305 
 306     with ops.control_dependencies(self._runtime_assertions):
 307       # Law of total variance: Var(Y) = E[Var(Y|X)] + Var(E[Y|X])
 308       probs = distribution_utils.pad_mixture_dimensions(
 309           distribution_utils.pad_mixture_dimensions(
 310               self.mixture_distribution.probs, self, self.mixture_distribution,
 311               self._event_shape().ndims),
 312           self, self.mixture_distribution,
 313           self._event_shape().ndims)                         # [B, k, 1, 1]
 314       mean_cond_var = math_ops.reduce_sum(
 315           probs * self.components_distribution.covariance(),
 316           axis=-3)                                           # [B, e, e]
 317       var_cond_mean = math_ops.reduce_sum(
 318           probs * _outer_squared_difference(
 319               self.components_distribution.mean(),
 320               self._pad_sample_dims(self._mean())),
 321           axis=-3)                                           # [B, e, e]
 322       return mean_cond_var + var_cond_mean                   # [B, e, e]
 323 
 324   def _pad_sample_dims(self, x):
 325     with ops.name_scope("pad_sample_dims", values=[x]):
 326       ndims = x.shape.ndims if x.shape.ndims is not None else array_ops.rank(x)
 327       shape = array_ops.shape(x)
 328       d = ndims - self._event_ndims
 329       x = array_ops.reshape(x, shape=array_ops.concat([
 330           shape[:d], [1], shape[d:]], axis=0))
 331       return x

 342 def _outer_squared_difference(x, y):
 343   """Convenience function analogous to tf.squared_difference."""
 344   z = x - y
 345   return z[..., array_ops.newaxis, :] * z[..., array_ops.newaxis]
