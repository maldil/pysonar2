  35 class Independent(distribution_lib.Distribution):
  36   """Independent distribution from batch of distributions.
  37 
  38   This distribution is useful for regarding a collection of independent,
  39   non-identical distributions as a single random variable. For example, the
  40   `Independent` distribution composed of a collection of `Bernoulli`
  41   distributions might define a distribution over an image (where each
  42   `Bernoulli` is a distribution over each pixel).
  43 
  44   More precisely, a collection of `B` (independent) `E`-variate random variables
  45   (rv) `{X_1, ..., X_B}`, can be regarded as a `[B, E]`-variate random variable
  46   `(X_1, ..., X_B)` with probability
  47   `p(x_1, ..., x_B) = p_1(x_1) * ... * p_B(x_B)` where `p_b(X_b)` is the
  48   probability of the `b`-th rv. More generally `B, E` can be arbitrary shapes.
  49 
  50   Similarly, the `Independent` distribution specifies a distribution over `[B,
  51   E]`-shaped events. It operates by reinterpreting the rightmost batch dims as
  52   part of the event dimensions. The `reinterpreted_batch_ndims` parameter
  53   controls the number of batch dims which are absorbed as event dims;
  54   `reinterpreted_batch_ndims < len(batch_shape)`.  For example, the `log_prob`
  55   function entails a `reduce_sum` over the rightmost `reinterpreted_batch_ndims`
  56   after calling the base distribution's `log_prob`.  In other words, since the
  57   batch dimension(s) index independent distributions, the resultant multivariate
  58   will have independent components.
  59 
  60   #### Mathematical Details
  61 
  62   The probability function is,
  63 
  64   ```none
  65   prob(x; reinterpreted_batch_ndims) = tf.reduce_prod(
  66       dist.prob(x),
  67       axis=-1-range(reinterpreted_batch_ndims))
  68   ```
  69 
  70   #### Examples
  71 
  72   ```python
  73   import tensorflow_probability as tfp
  74   tfd = tfp.distributions
  75 
  76   # Make independent distribution from a 2-batch Normal.
  77   ind = tfd.Independent(
  78       distribution=tfd.Normal(loc=[-1., 1], scale=[0.1, 0.5]),
  79       reinterpreted_batch_ndims=1)
  80 
  81   # All batch dims have been "absorbed" into event dims.
  82   ind.batch_shape  # ==> []
  83   ind.event_shape  # ==> [2]
  84 
  85   # Make independent distribution from a 2-batch bivariate Normal.
  86   ind = tfd.Independent(
  87       distribution=tfd.MultivariateNormalDiag(
  88           loc=[[-1., 1], [1, -1]],
  89           scale_identity_multiplier=[1., 0.5]),
  90       reinterpreted_batch_ndims=1)
  91 
  92   # All batch dims have been "absorbed" into event dims.
  93   ind.batch_shape  # ==> []
  94   ind.event_shape  # ==> [2, 2]
  95   ```
  96 
  97   """

  99   @deprecation.deprecated(
 100       "2018-10-01",
 101       "The TensorFlow Distributions library has moved to "
 102       "TensorFlow Probability "
 103       "(https://github.com/tensorflow/probability). You "
 104       "should update all references to use `tfp.distributions` "
 105       "instead of `tf.contrib.distributions`.",
 106       warn_once=True)
 107   def __init__(
 108       self, distribution, reinterpreted_batch_ndims=None,
 109       validate_args=False, name=None):
 110     """Construct a `Independent` distribution.
 111 
 112     Args:
 113       distribution: The base distribution instance to transform. Typically an
 114         instance of `Distribution`.
 115       reinterpreted_batch_ndims: Scalar, integer number of rightmost batch dims
 116         which will be regarded as event dims. When `None` all but the first
 117         batch axis (batch axis 0) will be transferred to event dimensions
 118         (analogous to `tf.layers.flatten`).
 119       validate_args: Python `bool`.  Whether to validate input with asserts.
 120         If `validate_args` is `False`, and the inputs are invalid,
 121         correct behavior is not guaranteed.
 122       name: The name for ops managed by the distribution.
 123         Default value: `Independent + distribution.name`.
 124 
 125     Raises:
 126       ValueError: if `reinterpreted_batch_ndims` exceeds
 127         `distribution.batch_ndims`
 128     """
 129     parameters = dict(locals())
 130     name = name or "Independent" + distribution.name
 131     self._distribution = distribution
 132     with ops.name_scope(name) as name:
 133       if reinterpreted_batch_ndims is None:
 134         reinterpreted_batch_ndims = self._get_default_reinterpreted_batch_ndims(
 135             distribution)
 136       reinterpreted_batch_ndims = ops.convert_to_tensor(
 137           reinterpreted_batch_ndims,
 138           dtype=dtypes.int32,
 139           name="reinterpreted_batch_ndims")
 140       self._reinterpreted_batch_ndims = reinterpreted_batch_ndims
 141       self._static_reinterpreted_batch_ndims = tensor_util.constant_value(
 142           reinterpreted_batch_ndims)
 143       if self._static_reinterpreted_batch_ndims is not None:
 144         self._reinterpreted_batch_ndims = self._static_reinterpreted_batch_ndims
 145       super(Independent, self).__init__(
 146           dtype=self._distribution.dtype,
 147           reparameterization_type=self._distribution.reparameterization_type,
 148           validate_args=validate_args,
 149           allow_nan_stats=self._distribution.allow_nan_stats,
 150           parameters=parameters,
 151           graph_parents=(
 152               [reinterpreted_batch_ndims] +
 153               distribution._graph_parents),  # pylint: disable=protected-access
 154           name=name)
 155       self._runtime_assertions = self._make_runtime_assertions(
 156           distribution, reinterpreted_batch_ndims, validate_args)
 157 
 158   @property
 159   def distribution(self):
 160     return self._distribution
 161 
 162   @property
 163   def reinterpreted_batch_ndims(self):
 164     return self._reinterpreted_batch_ndims
 165 
 166   def _batch_shape_tensor(self):
 167     with ops.control_dependencies(self._runtime_assertions):
 168       batch_shape = self.distribution.batch_shape_tensor()
 169       batch_ndims = (batch_shape.shape[0].value
 170                      if batch_shape.shape.with_rank_at_least(1)[0].value
 171                      else array_ops.shape(batch_shape)[0])
 172       return batch_shape[:batch_ndims - self.reinterpreted_batch_ndims]
 173 
 174   def _batch_shape(self):
 175     batch_shape = self.distribution.batch_shape
 176     if (self._static_reinterpreted_batch_ndims is None
 177         or batch_shape.ndims is None):
 178       return tensor_shape.TensorShape(None)
 179     d = batch_shape.ndims - self._static_reinterpreted_batch_ndims
 180     return batch_shape[:d]
 181 
 182   def _event_shape_tensor(self):
 183     with ops.control_dependencies(self._runtime_assertions):
 184       batch_shape = self.distribution.batch_shape_tensor()
 185       batch_ndims = (batch_shape.shape[0].value
 186                      if batch_shape.shape.with_rank_at_least(1)[0].value
 187                      else array_ops.shape(batch_shape)[0])
 188       return array_ops.concat([
 189           batch_shape[batch_ndims - self.reinterpreted_batch_ndims:],
 190           self.distribution.event_shape_tensor(),
 191       ], axis=0)
 192 
 193   def _event_shape(self):
 194     batch_shape = self.distribution.batch_shape
 195     if (self._static_reinterpreted_batch_ndims is None
 196         or batch_shape.ndims is None):
 197       return tensor_shape.TensorShape(None)
 198     d = batch_shape.ndims - self._static_reinterpreted_batch_ndims
 199     return batch_shape[d:].concatenate(self.distribution.event_shape)
 200 
 201   def _sample_n(self, n, seed):
 202     with ops.control_dependencies(self._runtime_assertions):
 203       return self.distribution.sample(sample_shape=n, seed=seed)
 204 
 205   def _log_prob(self, x):
 206     with ops.control_dependencies(self._runtime_assertions):
 207       return self._reduce_sum(self.distribution.log_prob(x))
 208 
 209   def _entropy(self):
 210     with ops.control_dependencies(self._runtime_assertions):
 211       return self._reduce_sum(self.distribution.entropy())
 212 
 213   def _mean(self):
 214     with ops.control_dependencies(self._runtime_assertions):
 215       return self.distribution.mean()
 216 
 217   def _variance(self):
 218     with ops.control_dependencies(self._runtime_assertions):
 219       return self.distribution.variance()
 220 
 221   def _stddev(self):
 222     with ops.control_dependencies(self._runtime_assertions):
 223       return self.distribution.stddev()
 224 
 225   def _mode(self):
 226     with ops.control_dependencies(self._runtime_assertions):
 227       return self.distribution.mode()
 228 
 229   def _make_runtime_assertions(
 230       self, distribution, reinterpreted_batch_ndims, validate_args):
 231     assertions = []
 232     static_reinterpreted_batch_ndims = tensor_util.constant_value(
 233         reinterpreted_batch_ndims)
 234     batch_ndims = distribution.batch_shape.ndims
 235     if batch_ndims is not None and static_reinterpreted_batch_ndims is not None:
 236       if static_reinterpreted_batch_ndims > batch_ndims:
 237         raise ValueError("reinterpreted_batch_ndims({}) cannot exceed "
 238                          "distribution.batch_ndims({})".format(
 239                              static_reinterpreted_batch_ndims, batch_ndims))
 240     elif validate_args:
 241       batch_shape = distribution.batch_shape_tensor()
 242       batch_ndims = (
 243           batch_shape.shape[0].value
 244           if batch_shape.shape.with_rank_at_least(1)[0].value is not None
 245           else array_ops.shape(batch_shape)[0])
 246       assertions.append(check_ops.assert_less_equal(
 247           reinterpreted_batch_ndims, batch_ndims,
 248           message=("reinterpreted_batch_ndims cannot exceed "
 249                    "distribution.batch_ndims")))
 250     return assertions
 251 
 252   def _reduce_sum(self, stat):
 253     if self._static_reinterpreted_batch_ndims is None:
 254       range_ = math_ops.range(self._reinterpreted_batch_ndims)
 255     else:
 256       range_ = np.arange(self._static_reinterpreted_batch_ndims)
 257     return math_ops.reduce_sum(stat, axis=-1-range_)
 258 
 259   def _get_default_reinterpreted_batch_ndims(self, distribution):
 260     """Computes the default value for reinterpreted_batch_ndim __init__ arg."""
 261     ndims = distribution.batch_shape.ndims
 262     if ndims is None:
 263       which_maximum = math_ops.maximum
 264       ndims = array_ops.shape(distribution.batch_shape_tensor())[0]
 265     else:
 266       which_maximum = np.maximum
 267     return which_maximum(0, ndims - 1)

 270 @kullback_leibler.RegisterKL(Independent, Independent)
 271 @deprecation.deprecated(
 272     "2018-10-01",
 273     "The TensorFlow Distributions library has moved to "
 274     "TensorFlow Probability "
 275     "(https://github.com/tensorflow/probability). You "
 276     "should update all references to use `tfp.distributions` "
 277     "instead of `tf.contrib.distributions`.",
 278     warn_once=True)
 279 def _kl_independent(a, b, name="kl_independent"):
 280   """Batched KL divergence `KL(a || b)` for Independent distributions.
 281 
 282   We can leverage the fact that
 283   ```
 284   KL(Independent(a) || Independent(b)) = sum(KL(a || b))
 285   ```
 286   where the sum is over the `reinterpreted_batch_ndims`.
 287 
 288   Args:
 289     a: Instance of `Independent`.
 290     b: Instance of `Independent`.
 291     name: (optional) name to use for created ops. Default "kl_independent".
 292 
 293   Returns:
 294     Batchwise `KL(a || b)`.
 295 
 296   Raises:
 297     ValueError: If the event space for `a` and `b`, or their underlying
 298       distributions don't match.
 299   """
 300   p = a.distribution
 301   q = b.distribution
 302 
 303   # The KL between any two (non)-batched distributions is a scalar.
 304   # Given that the KL between two factored distributions is the sum, i.e.
 305   # KL(p1(x)p2(y) || q1(x)q2(y)) = KL(p1 || q1) + KL(q1 || q2), we compute
 306   # KL(p || q) and do a `reduce_sum` on the reinterpreted batch dimensions.
 307   if a.event_shape.is_fully_defined() and b.event_shape.is_fully_defined():
 308     if a.event_shape == b.event_shape:
 309       if p.event_shape == q.event_shape:
 310         num_reduce_dims = a.event_shape.ndims - p.event_shape.ndims
 311         reduce_dims = [-i - 1 for i in range(0, num_reduce_dims)]
 312 
 313         return math_ops.reduce_sum(
 314             kullback_leibler.kl_divergence(p, q, name=name), axis=reduce_dims)
 315       else:
 316         raise NotImplementedError("KL between Independents with different "
 317                                   "event shapes not supported.")
 318     else:
 319       raise ValueError("Event shapes do not match.")
 320   else:
 321     with ops.control_dependencies([
 322         check_ops.assert_equal(a.event_shape_tensor(), b.event_shape_tensor()),
 323         check_ops.assert_equal(p.event_shape_tensor(), q.event_shape_tensor())
 324     ]):
 325       num_reduce_dims = (
 326           array_ops.shape(a.event_shape_tensor()[0]) -
 327           array_ops.shape(p.event_shape_tensor()[0]))
 328       reduce_dims = math_ops.range(-num_reduce_dims - 1, -1, 1)
 329       return math_ops.reduce_sum(
 330           kullback_leibler.kl_divergence(p, q, name=name), axis=reduce_dims)
