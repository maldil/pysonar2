  56 class ConverterMode(enum.Enum):
  57   """Enum class defining the converters available to generate TFLite models.
  58 
  59   WARNING: Experimental interface, subject to change.
  60   """
  61   # Convert model using TOCO such that all ops are TensorFlow Lite native ops.
  62   #
  63   # This is the only supported mode for any models that contain operations that
  64   # cannot be resolved in TensorFlow.
  65   DEFAULT = "DEFAULT"
  66 
  67   # Convert model using TOCO such that only unsupported operations are
  68   # represented as TensorFlow ops.
  69   # WARNING: Experimental interface, subject to change.
  70   TOCO_FLEX = "TOCO_FLEX"
  71 
  72   # Convert model using TOCO such that all operations are represented as
  73   # TensorFlow ops.
  74   # WARNING: Experimental interface, subject to change.
  75   TOCO_FLEX_ALL = "TOCO_FLEX_ALL"
  76 
  77   def __str__(self):
  78     return self.value

  81 def toco_convert_protos(model_flags_str, toco_flags_str, input_data_str):
  82   """Convert `input_data_str` according to model and toco parameters.
  83 
  84   Unless you know what you are doing consider using
  85   the more friendly `tf.contrib.lite.toco_convert`.
  86 
  87   Args:
  88     model_flags_str: Serialized proto describing model properties, see
  89       `toco/model_flags.proto`.
  90     toco_flags_str: Serialized proto describing conversion properties, see
  91       `toco/toco_flags.proto`.
  92     input_data_str: Input data in serialized form (e.g. a graphdef is common)
  93   Returns:
  94     Converted model in serialized form (e.g. a TFLITE model is common).
  95   Raises:
  96     RuntimeError: When conversion fails, an exception is raised with the error
  97       message embedded.
  98   """
  99   # TODO(aselle): When toco does not use fatal errors for failure, we can
 100   # switch this on.
 101   if not _toco_from_proto_bin:
 102     return _toco_python.TocoConvert(
 103         model_flags_str, toco_flags_str, input_data_str)
 104 
 105   with _tempfile.NamedTemporaryFile() as fp_toco, \
 106            _tempfile.NamedTemporaryFile() as fp_model, \
 107            _tempfile.NamedTemporaryFile() as fp_input, \
 108            _tempfile.NamedTemporaryFile() as fp_output:
 109     fp_model.write(model_flags_str)
 110     fp_toco.write(toco_flags_str)
 111     fp_input.write(input_data_str)
 112     fp_model.flush()
 113     fp_toco.flush()
 114     fp_input.flush()
 115 
 116     cmd = [
 117         _toco_from_proto_bin, fp_model.name, fp_toco.name, fp_input.name,
 118         fp_output.name
 119     ]
 120     cmdline = " ".join(cmd)
 121     is_windows = _platform.system() == "Windows"
 122     proc = _subprocess.Popen(
 123         cmdline,
 124         shell=True,
 125         stdout=_subprocess.PIPE,
 126         stderr=_subprocess.STDOUT,
 127         close_fds=not is_windows)
 128     stdout, stderr = proc.communicate()
 129     exitcode = proc.returncode
 130     if exitcode == 0:
 131       stuff = fp_output.read()
 132       return stuff
 133     else:
 134       raise RuntimeError("TOCO failed see console for info.\n%s\n%s\n" %
 135                          (stdout, stderr))

 142 def build_toco_convert_protos(input_tensors,
 143                               output_tensors,
 144                               inference_type=lite_constants.FLOAT,
 145                               inference_input_type=None,
 146                               input_format=lite_constants.TENSORFLOW_GRAPHDEF,
 147                               input_shapes=None,
 148                               output_format=lite_constants.TFLITE,
 149                               quantized_input_stats=None,
 150                               default_ranges_stats=None,
 151                               drop_control_dependency=True,
 152                               reorder_across_fake_quant=False,
 153                               allow_custom_ops=False,
 154                               change_concat_input_ranges=False,
 155                               post_training_quantize=False,
 156                               dump_graphviz_dir=None,
 157                               dump_graphviz_video=False,
 158                               converter_mode=ConverterMode.DEFAULT):
 159   """Builds protocol buffers describing a conversion of a model using TOCO.
 160 
 161   Typically this is to convert from TensorFlow GraphDef to TFLite, in which
 162   case the default `input_format` and `output_format` are sufficient.
 163 
 164   Args:
 165     input_tensors: List of input tensors. Type and shape are computed using
 166       `foo.get_shape()` and `foo.dtype`.
 167     output_tensors: List of output tensors (only .name is used from this).
 168     inference_type: Target data type of real-number arrays in the output file.
 169       Must be `{FLOAT, QUANTIZED_UINT8}`.  (default FLOAT)
 170     inference_input_type: Target data type of real-number input arrays. Allows
 171       for a different type for input arrays in the case of quantization.
 172       Must be `{FLOAT, QUANTIZED_UINT8}`. (default `inference_type`)
 173     input_format: Type of data to read Currently must be
 174       `{TENSORFLOW_GRAPHDEF}`. (default TENSORFLOW_GRAPHDEF)
 175     input_shapes: Input array shape. It needs to be a list of the same length
 176       as `input_tensors`, or None. (default None)
 177     output_format: Output file format. Currently must be `{TFLITE,
 178       GRAPHVIZ_DOT}`. (default TFLITE)
 179     quantized_input_stats: List of tuples of floats representing the mean and
 180       standard deviation. Each tuple maps to the corresponding input tensor.
 181       Only need if `inference_input_type` is `QUANTIZED_UINT8`.
 182       real_input_value = (quantized_input_value - mean_value) / std_dev_value.
 183       (default None)
 184     default_ranges_stats: Tuple of integers representing (min, max) range values
 185       for all arrays without a specified range. Intended for experimenting with
 186       quantization via "dummy quantization". (default None)
 187     drop_control_dependency: Boolean indicating whether to drop control
 188       dependencies silently. This is due to TFLite not supporting control
 189       dependencies. (default True)
 190     reorder_across_fake_quant: Boolean indicating whether to reorder FakeQuant
 191       nodes in unexpected locations. Used when the location of the FakeQuant
 192       nodes is preventing graph transformations necessary to convert the graph.
 193       Results in a graph that differs from the quantized training graph,
 194       potentially causing differing arithmetic behavior. (default False)
 195     allow_custom_ops: Boolean indicating whether to allow custom operations.
 196       When false any unknown operation is an error. When true, custom ops are
 197       created for any op that is unknown. The developer will need to provide
 198       these to the TensorFlow Lite runtime with a custom resolver.
 199       (default False)
 200     change_concat_input_ranges: Boolean to change behavior of min/max ranges for
 201       inputs and outputs of the concat operator for quantized models. Changes
 202       the ranges of concat operator overlap when true. (default False)
 203     post_training_quantize: Boolean indicating whether to quantize the weights
 204       of the converted float model. Model size will be reduced and there will be
 205       latency improvements (at the cost of accuracy).
 206       (default False)
 207     dump_graphviz_dir: Full filepath of folder to dump the graphs at various
 208       stages of processing GraphViz .dot files. Preferred over
 209       --output_format=GRAPHVIZ_DOT in order to keep the requirements of the
 210       output file. (default None)
 211     dump_graphviz_video: Boolean indicating whether to dump the graph after
 212       every graph transformation. (default False)
 213     converter_mode: Experimental flag, subject to change. ConverterMode
 214       indicating which converter to use. (default ConverterMode.DEFAULT)
 215 
 216   Returns:
 217     model_flags, toco_flags: two protocol buffers describing the conversion
 218     process.
 219 
 220   Raises:
 221     ValueError: If the input tensor type is unknown
 222     RuntimeError: If TOCO fails to convert (in which case the runtime error's
 223       error text will contain the TOCO error log)
 224   """
 225   toco = _toco_flags_pb2.TocoFlags()
 226   toco.input_format = input_format
 227   toco.output_format = output_format
 228   toco.inference_type = inference_type
 229   if inference_input_type:
 230     toco.inference_input_type = inference_input_type
 231   else:
 232     toco.inference_input_type = toco.inference_type
 233   toco.drop_control_dependency = drop_control_dependency
 234   toco.reorder_across_fake_quant = reorder_across_fake_quant
 235   toco.allow_custom_ops = allow_custom_ops
 236   toco.post_training_quantize = post_training_quantize
 237   if default_ranges_stats:
 238     toco.default_ranges_min = default_ranges_stats[0]
 239     toco.default_ranges_max = default_ranges_stats[1]
 240   if dump_graphviz_dir:
 241     toco.dump_graphviz_dir = dump_graphviz_dir
 242   toco.dump_graphviz_include_video = dump_graphviz_video
 243   if converter_mode == ConverterMode.TOCO_FLEX:
 244     toco.allow_flex_ops = True
 245   elif converter_mode == ConverterMode.TOCO_FLEX_ALL:
 246     toco.allow_flex_ops = True
 247     toco.force_flex_ops = True
 248 
 249   model = _model_flags_pb2.ModelFlags()
 250   model.change_concat_input_ranges = change_concat_input_ranges
 251   for idx, input_tensor in enumerate(input_tensors):
 252     input_array = model.input_arrays.add()
 253     if toco.inference_input_type == lite_constants.QUANTIZED_UINT8:
 254       input_array.mean_value, input_array.std_value = quantized_input_stats[idx]
 255     input_array.name = tensor_name(input_tensor)
 256     if input_shapes is None:
 257       shape = input_tensor.get_shape()
 258     else:
 259       shape = input_shapes[idx]
 260     input_array.shape.dims.extend(map(int, shape))
 261 
 262   for output_tensor in output_tensors:
 263     model.output_arrays.append(tensor_name(output_tensor))
 264   return model, toco

 267 def toco_convert_graph_def(input_data, input_arrays_with_shape, output_arrays,
 268                            *args, **kwargs):
 269   """"Convert a model using TOCO.
 270 
 271   This function is used to convert GraphDefs that cannot be loaded into
 272   TensorFlow to TFLite. Conversion can be customized by providing arguments
 273   that are forwarded to `build_toco_convert_protos` (see documentation for
 274   details).
 275 
 276   Args:
 277     input_data: Input data (i.e. often `sess.graph_def`),
 278     input_arrays_with_shape: Tuple of strings representing input tensor names
 279       and list of integers representing input shapes
 280       (e.g., [("foo" : [1, 16, 16, 3])]). Use only when graph cannot be loaded
 281       into TensorFlow and when `input_tensors` is None. (default None)
 282     output_arrays: List of output tensors to freeze graph with. Use only when
 283       graph cannot be loaded into TensorFlow and when `output_tensors` is None.
 284       (default None)
 285     *args: See `build_toco_convert_protos`,
 286     **kwargs: See `build_toco_convert_protos`.
 287 
 288   Returns:
 289     The converted data. For example if TFLite was the destination, then
 290     this will be a tflite flatbuffer in a bytes array.
 291 
 292   Raises:
 293     Defined in `build_toco_convert_protos`.
 294   """
 295   model_flags, toco_flags = build_toco_convert_protos(
 296       input_tensors=[], output_tensors=[], *args, **kwargs)
 297 
 298   for idx, (name, shape) in enumerate(input_arrays_with_shape):
 299     input_array = model_flags.input_arrays.add()
 300     if kwargs["inference_type"] == lite_constants.QUANTIZED_UINT8:
 301       input_array.mean_value, input_array.std_value = kwargs[
 302           "quantized_input_stats"][idx]
 303     input_array.name = name
 304     input_array.shape.dims.extend(map(int, shape))
 305 
 306   for name in output_arrays:
 307     model_flags.output_arrays.append(name)
 308 
 309   data = toco_convert_protos(model_flags.SerializeToString(),
 310                              toco_flags.SerializeToString(),
 311                              input_data.SerializeToString())
 312   return data

 315 def toco_convert_impl(input_data, input_tensors, output_tensors, *args,
 316                       **kwargs):
 317   """"Convert a model using TOCO.
 318 
 319   Typically this function is used to convert from TensorFlow GraphDef to TFLite.
 320   Conversion can be customized by providing arguments that are forwarded to
 321   `build_toco_convert_protos` (see documentation for details).
 322 
 323   Args:
 324     input_data: Input data (i.e. often `sess.graph_def`),
 325     input_tensors: List of input tensors. Type and shape are computed using
 326       `foo.get_shape()` and `foo.dtype`.
 327     output_tensors: List of output tensors (only .name is used from this).
 328     *args: See `build_toco_convert_protos`,
 329     **kwargs: See `build_toco_convert_protos`.
 330 
 331   Returns:
 332     The converted data. For example if TFLite was the destination, then
 333     this will be a tflite flatbuffer in a bytes array.
 334 
 335   Raises:
 336     Defined in `build_toco_convert_protos`.
 337   """
 338   model_flags, toco_flags = build_toco_convert_protos(
 339       input_tensors, output_tensors, *args, **kwargs)
 340   data = toco_convert_protos(model_flags.SerializeToString(),
 341                              toco_flags.SerializeToString(),
 342                              input_data.SerializeToString())
 343   return data
