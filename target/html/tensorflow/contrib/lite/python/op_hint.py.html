  15 """Define tflite op hints (intrinsic operations).
  16 
  17 This essentially allows defining a TensorFlow API for tflite operations in
  18 Python with hints on how they are represented in TensorFlow Lite. This basically
  19 is a form of tflite intrinsic. It wraps a subpart of a TensorFlow execution
  20 graph and is useful for LSTMs and other complicated TensorFlow constructions
  21 that are difficult to pattern match in TOCO, but are represented by a single
  22 accelerated tflite op.
  23 
  24 Example:
  25   def tflite_cool_activation(input):
  26     # A cool activation function.
  27     custom = tf.contrib.lite.OpHint("cool_activation")
  28     input, = custom.add_inputs(input)
  29     output = tf.sigmoid(input) * input
  30     output, = custom.add_outputs(output)
  31     return output
  32 
  33   image = tf.placeholder(tf.float32, (1, 16, 16, 1))
  34   output = tf.identity(tflite_cool_activation(image))
  35 
  36   session = tf.Session()
  37 
  38   graphdef_to_convert = tf.contrib.lite.convert_op_hints_to_stubs(session)
  39   tflite_graph = tf.contrib.lite.toco_convert(graphdef_to_convert,
  40                                               [image], [output])
  41                                               [image], [output])
  42   with open("/tmp/graph.fb", "wb") as fp:
  43     fp.write(tflite_graph)
  44 
  45 How does it work?:
  46 
  47 OpHint is a helper that you use when defining a vanilla python function.
  48 It allows you to wrap arguments with tf.identities with some custom attributes.
  49 These attributes allow you to find the original block of ops that was created.
  50 For example, if you use cool_activation above you essentially get:
  51 
  52 a_input = tf.identity()
  53 result = tf.multiply(tf.sigmoid(a_input), a_input)
  54 output = tf.identity()
  55 
  56 a_input, output are identities that have parameters representing
  57 what argument they are, what the name of the function they should turn into
  58 in tf lite as well as a guid that uniquely identifies a particular invocation.
  59 
  60 Once you have built your whole tensorflow graph, you can run it and train it
  61 as usual, but after you have done that, you need to convert the graph into
  62 a form that replaces these subgraphs wrapped in identities to stub ops. These
  63 ops don't actually exist in the normal TensorFlow runtime, but will be
  64 understood by toco later.
  65 """

  91 class OpHint(object):
  92   """A class that helps build tflite function invocations.
  93 
  94   It allows you to take a bunch of TensorFlow ops and annotate the construction
  95   such that toco knows how to convert it to tflite. This embeds a pseudo
  96   function in a TensorFlow graph. This allows embedding high-level API usage
  97   information in a lower level TensorFlow implementation so that an alternative
  98   implementation can be substituted later.
  99 
 100   Essentially, any "input" into this pseudo op is fed into an identity, and
 101   attributes are added to that input before being used by the constituent ops
 102   that make up the pseudo op. A similar process is done to any output that
 103   is to be exported from the current op.
 104 
 105   TODO(aselle): When TensorFlow functions functionality works for arbitrary
 106   constructs, this mechanism can be retired and changed to use python defun's.
 107   """
 108 
 109   # Attr constants that are used for representation in the GraphDef. These
 110   # will be used on every Identity op that is involved in a total OpHint.
 111 
 112   # Name of the OpHint function (cosmetic).
 113   FUNCTION_NAME_ATTR = "_tflite_function_name"
 114   # UUID of the function (each OpHint gets a new uuid).
 115   FUNCTION_UUID_ATTR = "_tflite_function_uuid"
 116   # The index index of the input (or nothing if it is an output).
 117   FUNCTION_INPUT_INDEX_ATTR = "_tflite_function_input_index"
 118   # The output index of the output (or nothing if it is an input).
 119   FUNCTION_OUTPUT_INDEX_ATTR = "_tflite_function_output_index"
 120   # An index that orders aggregate arguments. Aggregate arguments are ones
 121   # that are separate but will be fused horizontally. For example a static LSTM
 122   # has a lstm cell for each time step. Each one has a separate opHint, but a
 123   # fused SequentialLSTM will treat this as a single tensor.
 124   FUNCTION_SORT_INDEX_ATTR = "_tflite_function_sort_index"
 125   # The way in which multiple parts of the aggregate argument will be joined
 126   # into a fused operand. Valid options are OpHint.AGGREGATE_FIRST,
 127   # OpHint.AGGREGATE_LAST, OpHint.AGGREGATE_STACK.
 128   FUNCTION_AGGREGATE_ATTR = "_tflite_function_aggregate"
 129   # On fused OpHint stub, the order of inputs that the final LSTM call will
 130   # have. What this means is that the TensorFlow order might be
 131   # "foo", "bar", "stuff" and you might want the TF lite op order to be
 132   # "stuff", "foo", "bar", -1 (where -1 is unused). So you would set this
 133   # attribute to [2, 0, 1, -1].
 134   TFLITE_INPUT_INDICES = "_tflite_input_indices"
 135 
 136   # Types of aggregations
 137   #  stack: stacks all ophints with matching tags. i.e. for a static rnn.
 138   #   specifically, this is good for an input or output to a static rnn cell.
 139   AGGREGATE_STACK = _compat.as_bytes("stack")
 140   # first: only takes the first output (one with lowest sort index)
 141   # of matching tags. This is good for the input state to an RNN.
 142   AGGREGATE_FIRST = _compat.as_bytes("first")
 143   # aggregation last takes only the last tag (one with highest sort index).
 144   # This is good for an output value on the last stack item of a
 145   # static rnn.
 146   AGGREGATE_LAST = _compat.as_bytes("last")

 148   class OpHintArgumentTracker(object):
 149     """Conceptually tracks indices of arguments of "OpHint functions".
 150 
 151     The inputs and arguments of these functions both use an instance
 152     of the class so they can have independent numbering."""
 153 
 154     def __init__(self, function_name, unique_function_id, node_name_prefix,
 155                  attr_name):
 156       """Initialize ophint argument.
 157 
 158       Args:
 159         function_name: Name of the function that this tracks arguments for.
 160         unique_function_id: UUID of function that this tracks arguments for.
 161         node_name_prefix: How identities that are created are named.
 162         attr_name: Name of attribute to use to store the index for this hint.
 163           i.e. FUNCTION_INPUT_INDEX or FUNCTION_OUTPUT_INDEX
 164       """
 165 
 166       # The global index is the argument index of the op. This is in contrast
 167       # to the sort index which is the sequence number of a particular instance
 168       # of a given global index. For example, you may have called add hint
 169       # twice with the tag "foo". Then the global index will be 0 for both
 170       # and the sort index will be 0 for the first added and 1 for the second.
 171       self._function_name = function_name
 172       self._unique_function_id = unique_function_id
 173       self._next_global_index = 0  # The absolute global index
 174       self._used_global_indices = set()
 175       self._tag_to_global_index = {}  # The argument index a given tag maps to
 176       self._tag_to_next_sort_index = {}  # The current index for each tag
 177       self._node_name_prefix = node_name_prefix
 178       self._attr_name = attr_name
 179 
 180     def _get_new_global_index(self, index_override):
 181       """Return the next unused argument index in order or use an override.
 182 
 183       Args:
 184         index_override: An index to use instead of the next available or None
 185           to use the next available.
 186 
 187       Returns:
 188         A valid global_index to use for the next hint argument.
 189 
 190       Raises:
 191         ValueError: If the index_override is already used by another hint.
 192       """
 193       if index_override is None:
 194         global_index = self._next_global_index
 195       else:
 196         if index_override in self._used_global_indices:
 197           raise ValueError("Index %d was already used by another call to add")
 198         global_index = index_override
 199       # Make next_global_index valid
 200       self._used_global_indices.add(global_index)
 201       while self._next_global_index in self._used_global_indices:
 202         self._next_global_index += 1
 203       return global_index
 204 
 205     def add(self, arg, tag=None, name=None, aggregate=None,
 206             index_override=None):
 207       """Return a wrapped tensor of an input tensor as an argument.
 208 
 209       Args:
 210         arg: A TensorFlow tensor that should be considered an argument.
 211         tag: String tag to identify arguments that should be packed.
 212         name: Name of argument. This is included in the Identity hint op names.
 213         aggregate: Strategy to aggregate.
 214         Acceptable values are OpHint.AGGREGATE_FIRST, OpHint.AGGREGATE_LAST,
 215           and OpHint.AGGREGATE_STACK.
 216           Note, aggregate is only valid if tag is specified.
 217         index_override: Specify what input/output index should this be in the
 218           final stub. i.e. add(arg0, index=1); add(arg1, index=0) wil make the
 219           final stub be as stub_func(inputs[arg1, arg0], outputs=[]) rather than
 220           the default call order based ordering.
 221 
 222       Returns:
 223         A tensor representing the wrapped argument.
 224 
 225       Raises:
 226         ValueError: When indices are not consistent.
 227       """
 228 
 229       # Find the appropriate index
 230       if tag is None:
 231         if aggregate is not None:
 232           raise ValueError("You must specify `tag` if using aggregate.")
 233         global_index = self._get_new_global_index(index_override)
 234         sort_index = None
 235       else:
 236         if aggregate is None:
 237           raise ValueError("You must specify `aggregate` if using tag.")
 238         if tag not in self._tag_to_global_index:
 239           self._tag_to_global_index[tag] = (
 240               self._get_new_global_index(index_override))
 241           self._tag_to_next_sort_index[tag] = 0
 242         elif (index_override and
 243               index_override != self._tag_to_global_index[tag]):
 244           raise ValueError(
 245               "Tag %r was called with two indices %r and %r" %
 246               (tag, index_override, self._tag_to_global_index[tag]))
 247         global_index = self._tag_to_global_index[tag]
 248         sort_index = self._tag_to_next_sort_index[tag]
 249         self._tag_to_next_sort_index[tag] += 1
 250 
 251       uuid = self._unique_function_id
 252       name = "%s-%s-%s-%r-%r-%s" % (self._node_name_prefix, self._function_name,
 253                                     uuid, global_index, sort_index, name)
 254       identity_op = _array_ops.identity(arg, name=name)
 255 
 256       # pylint: disable=protected-access
 257       identity_op.op._set_attr(
 258           OpHint.FUNCTION_NAME_ATTR,
 259           _attr_value_pb2.AttrValue(
 260               s=_compat.as_bytes(self._function_name)))
 261       identity_op.op._set_attr(
 262           OpHint.FUNCTION_UUID_ATTR,
 263           _attr_value_pb2.AttrValue(
 264               s=_compat.as_bytes(self._unique_function_id)))
 265       identity_op.op._set_attr(
 266           self._attr_name, _attr_value_pb2.AttrValue(i=global_index))
 267       if sort_index is not None:
 268         identity_op.op._set_attr(
 269             OpHint.FUNCTION_SORT_INDEX_ATTR,
 270             _attr_value_pb2.AttrValue(i=sort_index))
 271       if aggregate is not None:
 272         identity_op.op._set_attr(
 273             OpHint.FUNCTION_AGGREGATE_ATTR,
 274             _attr_value_pb2.AttrValue(s=_compat.as_bytes((aggregate))))
 275       # pylint: enable=protected-access
 276       return identity_op
 277 
 278   def __init__(self, function_name, **kwargs):
 279     """Create a OpHint.
 280 
 281     Args:
 282       function_name: Name of the function (the custom op name in tflite)
 283       **kwargs: Keyword arguments of any constant attributes for the function.
 284     """
 285     self._function_name = function_name
 286     self._unique_function_id = _uuid.uuid1().hex  # TODO(aselle): Unique enough?
 287     self._attrs_to_store_later = kwargs
 288     self._stored_attrs = False
 289     self._inputs = OpHint.OpHintArgumentTracker(
 290         self._function_name, self._unique_function_id, "InputHint",
 291         OpHint.FUNCTION_INPUT_INDEX_ATTR)
 292     self._outputs = OpHint.OpHintArgumentTracker(
 293         self._function_name, self._unique_function_id, "OutputHint",
 294         OpHint.FUNCTION_OUTPUT_INDEX_ATTR)
 295 
 296   def _setattr(self, dest_op, name, value):
 297     tensor_value = _ops.convert_to_tensor(value)
 298     # pylint: disable=protected-access
 299     dest_op.op._set_attr(name, _attr_value_pb2.AttrValue(
 300         tensor=tensor_value.op.node_def.attr["value"].tensor))
 301     # pylint: enable=protected-access

 303   def add_input(self, *args, **kwargs):
 304     """Add a wrapped input argument to the hint.
 305 
 306     Args:
 307       *args: The input tensor.
 308       **kwargs:
 309         "name" label
 310         "tag" a tag to group multiple arguments that will be aggregated. I.e.
 311           a string like 'cool_input'. Basically multiple inputs can be added
 312           to the same hint for parallel operations that will eventually be
 313           combined. An example would be static_rnn which creates multiple copies
 314           of state or inputs.
 315         "aggregate" aggregation strategy that is valid only for tag non None.
 316           Acceptable values are OpHint.AGGREGATE_FIRST, OpHint.AGGREGATE_LAST,
 317           and OpHint.AGGREGATE_STACK.
 318         "index_override" The global index to use. This corresponds to the
 319           argument order in the final stub that will be generated.
 320     Returns:
 321       The wrapped input tensor.
 322     """
 323     return self._inputs.add(*args, **kwargs)
 324 
 325   def add_output(self, *args, **kwargs):
 326     """Add a wrapped output argument to the hint.
 327 
 328     Args:
 329       *args: The output tensor.
 330       **kwargs:
 331         "name" label
 332         "tag" a tag to group multiple arguments that will be aggregated. I.e.
 333           a string like 'cool_input'. Basically multiple inputs can be added
 334           to the same hint for parallel operations that will eventually be
 335           combined. An example would be static_rnn which creates multiple copies
 336           of state or inputs.
 337         "aggregate" aggregation strategy that is valid only for tag non None.
 338           Acceptable values are OpHint.AGGREGATE_FIRST, OpHint.AGGREGATE_LAST,
 339           and OpHint.AGGREGATE_STACK.
 340         "index_override" The global index to use. This corresponds to the
 341           argument order in the final stub that will be generated.
 342     Returns:
 343       The wrapped output tensor.
 344     """
 345     return self._outputs.add(*args, **kwargs)
 346 
 347   def add_inputs(self, *args, **kwargs):
 348     """Add a sequence of inputs to the function invocation.
 349 
 350     Args:
 351       *args: List of inputs to be converted (should be Tf.Tensor).
 352       **kwargs: This allows 'names' which should be a list of names.
 353     Returns:
 354       Wrapped inputs (identity standins that have additional metadata). These
 355       are also are also tf.Tensor's.
 356     """
 357     if "names" in kwargs:
 358       return [
 359           self._inputs.add(arg, name=name)
 360           for arg, name in zip(args, kwargs["names"])
 361       ]
 362     else:
 363       return [self._inputs.add(arg) for arg in args]
 364 
 365   def add_outputs(self, *args, **kwargs):
 366     """Add a sequence of outputs to the function invocation.
 367 
 368     Args:
 369       *args: List of outputs to be converted (should be tf.Tensor).
 370       **kwargs: See
 371     Returns:
 372       Wrapped outputs (identity standins that have additional metadata). These
 373       are also tf.Tensor's.
 374     """
 375     if "names" in kwargs:
 376       return [
 377           self._outputs.add(arg, name=name)
 378           for arg, name in zip(args, kwargs["names"])
 379       ]
 380     else:
 381       return [self._outputs.add(arg) for arg in args]

 384 class _LiteOperand(object):
 385   """Abstract operand for a tflite hint function.
 386 
 387   This is a base class that handles representing arguments to an OpHint.
 388   It also is able to serialize operands to the stubbed graph_def.
 389   Child classes are responsible for being able to
 390   store information about the hint identity operators. They are also responsible
 391   for knowing how to serialize to output graphdefs.
 392 
 393   Typically this will be implemented by holding one or more identity nodes
 394   that were previously discovered as hints.
 395   """
 396 
 397   def aggregate_and_return_name_for_input(self, out_graphdef):
 398     """This adds the node(s) to out_graphdef and returns the input node name.
 399 
 400     Args:
 401       out_graphdef: A graphdef that is ready to have this input added.
 402 
 403     Returns:
 404       The the output that the stub should use as an input for this operand.
 405 
 406     Raises:
 407       RuntimeError: if the method is not implemented.
 408     """
 409     del out_graphdef
 410     raise RuntimeError("Unimplemented abstract method.")
 411 
 412   def aggregate_and_return_name_for_output(self, fused_op_name, output_index,
 413                                            out_graphdef):
 414     """Add node(s) to graph representing output operands and returns type.
 415 
 416     Args:
 417       fused_op_name: name of the fused op stub name.
 418       output_index: Output index that we are currently processing from stub.
 419       out_graphdef: The destination graphdef we are currently building up.
 420 
 421     Returns:
 422       The datatype of this identity.
 423 
 424     Raises:
 425       RuntimeError: if the method is not implemented.
 426     """
 427     del fused_op_name, output_index, out_graphdef
 428     raise RuntimeError("Unimplemented abstract method.")

 457 class _LiteAggregateOperand(_LiteOperand):
 458   """An operand for a tflite hint function that is aggregated from many.
 459 
 460   For example, an LSTM is a grid of operators that are all related. Inputs
 461   going into them may need to be fused, so they should all be tracked as
 462   related arguments.
 463   """
 464 
 465   def __init__(self, aggregation):
 466     _LiteOperand.__init__(self)
 467     self.aggregation = aggregation
 468     self.names = {}
 469     self.nodes = {}
 470     self.flattened = None
 471 
 472   def add(self, sort, node):
 473     self.names[sort] = _tensor_name_base(node.name)
 474     self.nodes[sort] = node
 475 
 476   def flatten_nodes(self):
 477     """Return a list of all the node protos in aggregation sorted order."""
 478     if not self.flattened:
 479       self.flattened = [None] * len(self.nodes)
 480       for idx, node in _six.iteritems(self.nodes):
 481         self.flattened[idx] = node
 482       for n in self.nodes:
 483         if n is None:
 484           raise RuntimeError("Aggregate was missing argument.")
 485       if self.aggregation == OpHint.AGGREGATE_FIRST:
 486         self.flattened = self.flattened[:1]
 487       elif self.aggregation == OpHint.AGGREGATE_LAST:
 488         self.flattened = self.flattened[-1:]
 489       elif self.aggregation == OpHint.AGGREGATE_STACK:
 490         pass
 491       else:
 492         raise ValueError(
 493             "Invalid aggregation type %r specified" % self.aggregation)
 494     return self.flattened
 495 
 496   def flatten(self):
 497     """Return a list of all node names in aggregation sorted sorter."""
 498     return [_tensor_name_base(x.name) for x in self.flatten_nodes()]
 499 
 500   def aggregate_and_return_name_for_input(self, out_graphdef):
 501     """This adds the nodes to out_graphdef and returns an aggregated output.
 502 
 503     In particular, if you have 4 inputs to a hint stub, this will be the
 504     node that you can use as an output. I.e. you have 4 timesteps from a
 505     static rnn, then a fused UnidriecitonalLSTM will expect 1 input with
 506     all 4 time steps. So here we make a pack and return the output name of
 507     that pack.
 508 
 509     Args:
 510       out_graphdef: A graphdef that is ready to have this input added.
 511 
 512     Returns:
 513       The name of a pack that aggregates this node.
 514     """
 515     flattened = self.flatten_nodes()
 516     if len(flattened) == 1:
 517       return _tensor_name_base(flattened[0].name)
 518     else:
 519       new_node = _node_def_pb2.NodeDef()
 520       new_node.op = "Pack"
 521       new_node.name = "OpHintStack-%s" % flattened[0].name
 522       new_node.attr["N"].i = len(flattened)
 523       new_node.attr["T"].type = flattened[0].attr["T"].type
 524       for discrete in flattened:
 525         new_node.input.append(_tensor_name_base(discrete.name))
 526       out_graphdef.node.extend([new_node])
 527       return new_node.name
 528 
 529   def aggregate_and_return_name_for_output(self, fused_op_name, output_index,
 530                                            out_graphdef):
 531     """This adds to `out_graphdef` all the unaggregated outputs.
 532 
 533     I.e. we are outputting from a fused stub, but we need to make it compatible
 534     with the unfused original graph so we insert an unpack. Ideally in a later
 535     stage the unpack -> pack sequences will be removed.
 536 
 537     Args:
 538       fused_op_name: The name of the stub we are in the process of fusing.
 539       output_index: The output output_index this object represents.
 540       out_graphdef: The graphdef we are in the process of buildings
 541 
 542     Returns:
 543       The type of the aggregated output (so we can finish building the stub
 544       op).
 545     """
 546     flattened = self.flatten_nodes()
 547     if len(flattened) == 1:
 548       temp_op = _LiteSingleOperand(flattened[0])
 549       return temp_op.aggregate_and_return_name_for_output(
 550           fused_op_name, output_index, out_graphdef)
 551     else:
 552       stack_node = _node_def_pb2.NodeDef()
 553       stack_node.op = "Unpack"
 554       stack_node.name = "OpHintUnstack-%s" % flattened[0].name
 555       stack_node.attr["num"].i = len(flattened)
 556       output_type = flattened[0].attr["T"].type
 557       stack_node.attr["T"].type = output_type
 558       stack_node.input.append(_tensorflow_output_name(
 559           fused_op_name, output_index))
 560       out_graphdef.node.extend([stack_node])
 561 
 562       for idx, discrete in enumerate(flattened):
 563         output_node = _copy.deepcopy(discrete)
 564         del output_node.input[:]
 565         output_node.input.append(_tensorflow_output_name(stack_node.name, idx))
 566         out_graphdef.node.extend([output_node])
 567 
 568       return output_type
 569 
 570   def __str__(self):
 571     s = "\t\t\tAGGREGATE %s\n" % self.aggregation
 572     for sort, val in self.names.iteritems():
 573       s += "\t\t\t%d: %s\n" % (sort, val)
 574     return s

 577 class _LiteFuncCall(object):
 578   """Represent a TensorFlow Lite custom function.
 579 
 580   This is uses to accumulate found hints in the graphdef into a single
 581   conceptual unit.
 582 
 583   Properties:
 584     self.inputs: inputs to the op (hash from index # to argument)
 585     self.outputs: outputs to the op (hash from index # to argument)
 586     self.function_name: the tflite custom op name to use
 587     self.uuid: a unique call id for this particular call  (i.e.
 588       multiple function calls would have the same function_name but different
 589       uuids.
 590     self.params: A param name to key value for op constant data. I.e. for
 591       axis on a reduction, strides on a convolution, etc.
 592   """
 593 
 594   def __init__(self):
 595     self.inputs = {}
 596     self.outputs = {}
 597     self.function_name = None
 598     self.uuid = None
 599     self.params = {}
 600 
 601   def flattened_inputs_and_outputs(self):
 602     """Return a list of inputs and outputs in a flattened format.
 603 
 604     Returns:
 605       Tuple of (inputs, outputs). where input and output i a list of names.
 606     """
 607     def _flatten(input_or_output_dict):
 608       flattened_items = []
 609       for item in input_or_output_dict.values():
 610         flattened_items.extend(item.flatten())
 611       return flattened_items
 612 
 613     return _flatten(self.inputs), _flatten(self.outputs)
 614 
 615   def __str__(self):
 616     def format_args(items):
 617       s = ""
 618       for idx, item in items.iteritems():
 619         s += ("\t\t%d:\n" % idx) + str(item)
 620       return s
 621 
 622     inputs_str = "\tInputs\n" + format_args(self.inputs)
 623     outputs_str = "\tOutputs\n" + format_args(self.outputs)
 624 
 625     return ("tflite function %s call %s\n\tinputs:\n\t\t%s\n\toutputs:\n\t\t%s"
 626             % (self.function_name, self.uuid, inputs_str, outputs_str))

 629 def _find_all_hints_in_graph_def(graphdef):
 630   """Look at the current default graph and return a list of LiteFuncCall objs.
 631 
 632   Args:
 633     graphdef: A TensorFlow graph_def to look for LiteFuncCalls.
 634   Returns:
 635     a list of `LifeFuncCall` objects in the form
 636 
 637   """
 638   func_calls = _collections.defaultdict(_LiteFuncCall)
 639 
 640   for node in graphdef.node:
 641     attr = node.attr
 642     # This is an op hint if it has a FUNCTION_UUID_ATTR, otherwise skip
 643     uuid = attr[OpHint.FUNCTION_UUID_ATTR].s
 644     if (OpHint.FUNCTION_UUID_ATTR not in attr
 645         or not attr[OpHint.FUNCTION_UUID_ATTR].s):
 646       continue
 647 
 648     # Start building function
 649     call_def = func_calls[uuid]
 650     call_def.uuid = uuid
 651     call_def.function_name = attr[OpHint.FUNCTION_NAME_ATTR].s
 652     # Get sorting and aggregation information
 653 
 654     sort = (attr[OpHint.FUNCTION_SORT_INDEX_ATTR].i
 655             if OpHint.FUNCTION_SORT_INDEX_ATTR in attr else None)
 656     if sort == -1: sort = None
 657     aggregation = None
 658     if OpHint.FUNCTION_AGGREGATE_ATTR in attr:
 659       aggregation = attr[OpHint.FUNCTION_AGGREGATE_ATTR].s
 660 
 661     # Add the input or output
 662     def put_operand(stuff, index, sort, operand, aggregation):
 663       """Add a given index into the function structure."""
 664       if sort is None:
 665         stuff[index] = _LiteSingleOperand(operand)
 666       else:
 667         if index not in stuff:
 668           stuff[index] = _LiteAggregateOperand(aggregation)
 669         stuff[index].add(sort, operand)
 670 
 671     if OpHint.FUNCTION_INPUT_INDEX_ATTR in attr:
 672       put_operand(call_def.inputs, attr[OpHint.FUNCTION_INPUT_INDEX_ATTR].i,
 673                   sort, node, aggregation)
 674     if OpHint.FUNCTION_OUTPUT_INDEX_ATTR in attr:
 675       put_operand(call_def.outputs, attr[OpHint.FUNCTION_OUTPUT_INDEX_ATTR].i,
 676                   sort, node, aggregation)
 677 
 678     # Remember attributes
 679     for a in attr:
 680       if a.startswith("_tflite_attr_"):
 681         call_def.params[a.replace("_tflite_attr_,", "")] = attr[a].tensor
 682 
 683   return func_calls

 686 def _tensor_name_base(full_tensor_name):
 687   """Removes the device assignment code from a tensor.
 688 
 689   e.g. _tensor_name_base("foo:3") => "foo"
 690 
 691   Args:
 692     full_tensor_name: A tensor name that is annotated with a device placement
 693       (this is what tensor flow introspection gives).
 694   Returns:
 695     A name without any device assignment.
 696   """

 699   return full_tensor_name.split(":")[0]

 707 # TODO(aselle): This should be converted to grappler in the future.
 708 def _check_subgraph_closed(n, reachable_by_input, input_nodes_set,
 709                            name_to_input_name):
 710   """Checks to make sure node only connects to predecessor graph through inputs.
 711 
 712   Args:
 713     n: Node to check
 714     reachable_by_input: Nodes that are reachable by all inputs of subgraph
 715     input_nodes_set: The set of nodes that are "inputs".
 716     name_to_input_name: Maps from name to the list of inputs.
 717 
 718   Raises:
 719     TypeError: If the given node uses items past inputs directly.
 720   """
 721   next_to_visit = [n]
 722   visited = set()
 723   while next_to_visit:
 724     current_node = next_to_visit.pop()
 725     visited.add(current_node)
 726     if (current_node in reachable_by_input
 727         and current_node not in input_nodes_set):
 728       raise TypeError(
 729           "Node %s uses input %s not in input_nodes." % (n, current_node))
 730     if current_node not in input_nodes_set:
 731       next_to_visit += [
 732           input_node for input_node in name_to_input_name[current_node]
 733           if input_node not in visited
 734       ]

 737 # TODO(aselle): This should be converted to grappler in the future.
 738 def _convert_single_op_hint_to_stub(call, graph_def):
 739   """Given a graph_def, converts `call` into a stub and returns a new graph_def.
 740 
 741   Args:
 742     call: A single function call to be converted.
 743     graph_def: A graph_def to use as input (that hass call obviously).
 744   Returns:
 745     A new transformed graph-def that has call as a stub (single op).
 746 
 747   Note: after this process, the graph_def can no longer be loaded into
 748       the tensorflow runtime, so all future manipulations are done in graph_def
 749       level.
 750   """
 751   name_to_input_name, name_to_node, name_to_seq_num = _extract_graph_summary(
 752       graph_def)
 753   input_names, output_names = call.flattened_inputs_and_outputs()
 754 
 755   reachable_by_input = _bfs_for_reachable_nodes(input_names, name_to_input_name)
 756   reachable_by_output = _bfs_for_reachable_nodes(output_names,
 757                                                  name_to_input_name)
 758   input_nodes_set = set(input_names)
 759   output_nodes_set = set(output_names)
 760   nodes_after_fuse = []
 761   nodes_deleted_by_fuse = set()
 762   # Classify each node. We want to keep everything reachable by input, but
 763   # we don't know if things that are not reachable by output or input (things
 764   # after fusing).
 765   for node in graph_def.node:
 766     n = _tensor_name_base(node.name)
 767     if n in reachable_by_output:
 768       if n not in reachable_by_input and n not in output_nodes_set:
 769         # n is an internal node. Check to make sure it is really internal.
 770         # TODO(aselle): this could be done more efficiently by flooding
 771         # the graph first.
 772         _check_subgraph_closed(n, reachable_by_input, input_nodes_set,
 773                                name_to_input_name)
 774         nodes_deleted_by_fuse.add(n)
 775     elif n not in reachable_by_input:
 776       # n is a node that after all the fusings, so keep it.
 777       nodes_after_fuse.append(n)
 778     else:
 779       # n is a node that is randomly in the graph but not connected to
 780       # the chain of dependencies.
 781       pass
 782 
 783   # Make a new graphdef with all the pre-input and input nodes
 784   out = _graph_pb2.GraphDef()
 785   reachable_by_input_sorted = sorted(
 786       list(reachable_by_input), key=lambda n: name_to_seq_num[n])
 787   for node in reachable_by_input_sorted:
 788     out.node.extend([_copy.deepcopy(name_to_node[node])])
 789 
 790   # Create any stacks to aggregate arguments into to a single input
 791   # i.e. for static_rnn's.
 792   # TODO(aselle): Check that the inputs are complete i.e. 0 to n-1
 793   sorted_input_indices = list(call.inputs.keys())
 794   sorted_input_indices.sort()
 795   sorted_output_indices = list(call.outputs.keys())
 796   sorted_output_indices.sort()
 797   new_node = _node_def_pb2.NodeDef()
 798   # Delegate to each operand to produce the proper new input for this stub node.
 799   # In particular, an aggregate input will now be a Pack of some previously
 800   # non-fused things.
 801   for input_index in sorted_input_indices:
 802     inputs = call.inputs[input_index]
 803     new_node.input.append(inputs.aggregate_and_return_name_for_input(out))
 804   new_node.attr[OpHint.TFLITE_INPUT_INDICES].list.i.extend(sorted_input_indices)
 805 
 806   # Ceate the function
 807   new_node.op = call.function_name
 808   new_node.name = call.uuid
 809   out.node.extend([new_node])
 810 
 811   # Now call each output argument to give them a chance to make the proper
 812   # output type and add it to our new_node.
 813   output_dtypes = []
 814   for output_index in sorted_output_indices:
 815     output = call.outputs[output_index]
 816     output_dtype = (
 817         output.aggregate_and_return_name_for_output(new_node.name, output_index,
 818                                                     out))
 819     output_dtypes.append(output_dtype)
 820   new_node.attr["_output_types"].list.type[:] = output_dtypes
 821   # TODO(aselle): what is right here?
 822   new_node.attr["_output_quantized"].b = False
 823 
 824   # Add post output nodes that do not depend on the outputs
 825   for n in nodes_after_fuse:
 826     should_keep = True
 827     for input_name in name_to_input_name[n]:
 828       if input_name in nodes_deleted_by_fuse:
 829         should_keep = False
 830     if should_keep:
 831       out.node.extend([_copy.deepcopy(name_to_node[n])])
 832 
 833   # Misc. graph_def data that needs copying.
 834   out.library.CopyFrom(graph_def.library)
 835   out.versions.CopyFrom(graph_def.versions)
 836 
 837   return out

 840 # TODO(aselle): This should be converted to grappler in the future.
 841 def _remove_one_redundant_stack_unstack(in_graph_def):
 842   """Removes a stack->unstack pattern from in_graph_def in a returned graph.
 843 
 844   Args:
 845     in_graph_def: Graph def to use as input.
 846   Returns:
 847     Simplified tuple (graph_def, changed_something) where changed_something
 848     is true if anything was done.
 849   """
 850   name_to_input_name, name_to_node, name_to_seq_num = _extract_graph_summary(
 851       in_graph_def)
 852   del name_to_seq_num
 853 
 854   # TODO(aselle): Make this not hardcoded.
 855   do_generic_pack_unpack = True
 856 
 857   out = _graph_pb2.GraphDef()
 858   out.library.CopyFrom(in_graph_def.library)
 859   out.versions.CopyFrom(in_graph_def.versions)
 860   for n in in_graph_def.node:
 861     node_name = _tensor_name_base(n.name)
 862     if not node_name.startswith("OpHintStack") and not n.op.startswith("Pack"):
 863       continue
 864     next_to_visit = [node_name]
 865     visited = set()
 866 
 867     unpack_nodes = set()
 868     pack_node = node_name
 869 
 870     # Find a pattern of unstack connected to a stack (with identities
 871     # in between.
 872     matches_pattern = True
 873     is_hint_created_stack = False
 874     while next_to_visit:
 875       current_node_name = next_to_visit[0]
 876       visited.add(current_node_name)
 877       del next_to_visit[0]
 878       node = name_to_node[current_node_name]
 879       is_op_hint_stack = node.name.startswith("OpHintStack")
 880       is_op_hint_unstack = node.name.startswith("OpHintUnstack")
 881       if (node.op == "Identity" or is_op_hint_stack
 882           or (do_generic_pack_unpack and node.op == "Pack")):
 883         is_hint_created_stack |= is_op_hint_stack
 884         next_to_visit += [
 885             input_node for input_node in name_to_input_name[current_node_name]
 886             if input_node not in visited
 887         ]
 888       elif (is_op_hint_unstack
 889             or (do_generic_pack_unpack and node.op == "Unpack")):
 890         unpack_nodes.add(node.name)
 891         is_hint_created_stack &= is_op_hint_unstack
 892       else:
 893         matches_pattern = False
 894         break
 895       visited.add(node.name)
 896 
 897     if matches_pattern and len(unpack_nodes) == 1:
 898       pack_node = node_name
 899 
 900       # Check to see if anyone depends on the intermediate identity or the
 901       # Unstacked form
 902       no_external_dependency = True
 903       for other_n in in_graph_def.node:
 904         if other_n.name in visited: continue
 905         for input_tensor in name_to_input_name[other_n.name]:
 906           input_op = _tensor_name_base(input_tensor)
 907           if input_op in visited and input_op != pack_node:
 908             no_external_dependency = False
 909       # Proceed with the substitution if the stack/unstack pair was created
 910       # through hints, or that it was not, but nobody is consuming things
 911       # between the stack and unstack.
 912       if is_hint_created_stack or no_external_dependency:
 913         end = unpack_nodes.pop()
 914         end_input = name_to_node[end].input[0]
 915         # All nodes that depend on the final stack need to be redone to use
 916         for other_n in in_graph_def.node:
 917           node_name = _tensor_name_base(other_n.name)
 918           if node_name not in visited:
 919             new_node = _copy.deepcopy(other_n)
 920             new_node.input[:] = [
 921                 (end_input if stripped == pack_node else
 922                  non_stripped) for stripped, non_stripped in zip(
 923                      name_to_input_name[node_name], new_node.input[:])
 924             ]
 925             out.node.extend([new_node])
 926         return out, True
 927   return in_graph_def, False

 930 def _remove_redundant_stack_unstack(graph_def):
 931   curr = graph_def
 932   del graph_def
 933   changed_stuff = True
 934   while changed_stuff:
 935     curr, changed_stuff = _remove_one_redundant_stack_unstack(curr)
 936   return curr

 939 def _convert_op_hints_to_stubs_helper(
 940     graph_def, write_callback=lambda sess, graph_def: None):
 941   """Converts a graph_def to a new graph_def where all op hints are stubbed.
 942 
 943   Args:
 944     graph_def: A graph def that we should convert.
 945     write_callback: A function pointer that can be used to write intermediate
 946       steps of graph transformation (optional).
 947   Returns:
 948     A new stubbed graph_def.
 949   """
 950 
 951   hints = _find_all_hints_in_graph_def(graph_def)
 952   curr_graph_def = graph_def
 953   del graph_def  # prevent using graph_def again (common source of error)
 954   for hint in _six.itervalues(hints):
 955     curr_graph_def = _convert_single_op_hint_to_stub(
 956         hint, curr_graph_def)
 957     write_callback(curr_graph_def, "initial")
 958   # The stubbing process can create stacks/unstacks in the case of LSTMs
 959   # remove them.
 960   curr_graph_def = _remove_redundant_stack_unstack(curr_graph_def)
 961   return curr_graph_def
