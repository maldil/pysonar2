  32 def _parse_array(values, type_fn=str):
  33   if values:
  34     return [type_fn(val) for val in values.split(",") if val]
  35 
  36 
  37 def _parse_set(values):
  38   if values:
  39     return set(values.split(","))

  42 def _get_toco_converter(flags):
  43   """Makes a TFLiteConverter object based on the flags provided.
  44 
  45   Args:
  46     flags: argparse.Namespace object containing TFLite flags.
  47 
  48   Returns:
  49     TFLiteConverter object.
  50 
  51   Raises:
  52     ValueError: Invalid flags.
  53   """
  54   # Parse input and output arrays.
  55   input_arrays = _parse_array(flags.input_arrays)
  56   input_shapes = None
  57   if flags.input_shapes:
  58     input_shapes_list = [
  59         _parse_array(shape, type_fn=int)
  60         for shape in flags.input_shapes.split(":")
  61     ]
  62     input_shapes = dict(zip(input_arrays, input_shapes_list))
  63   output_arrays = _parse_array(flags.output_arrays)
  64 
  65   converter_kwargs = {
  66       "input_arrays": input_arrays,
  67       "input_shapes": input_shapes,
  68       "output_arrays": output_arrays
  69   }
  70 
  71   # Create TFLiteConverter.
  72   if flags.graph_def_file:
  73     converter_fn = lite.TFLiteConverter.from_frozen_graph
  74     converter_kwargs["graph_def_file"] = flags.graph_def_file
  75   elif flags.saved_model_dir:
  76     converter_fn = lite.TFLiteConverter.from_saved_model
  77     converter_kwargs["saved_model_dir"] = flags.saved_model_dir
  78     converter_kwargs["tag_set"] = _parse_set(flags.saved_model_tag_set)
  79     converter_kwargs["signature_key"] = flags.saved_model_signature_key
  80   elif flags.keras_model_file:
  81     converter_fn = lite.TFLiteConverter.from_keras_model_file
  82     converter_kwargs["model_file"] = flags.keras_model_file
  83   else:
  84     raise ValueError("--graph_def_file, --saved_model_dir, or "
  85                      "--keras_model_file must be specified.")
  86 
  87   return converter_fn(**converter_kwargs)

  90 def _convert_model(flags):
  91   """Calls function to convert the TensorFlow model into a TFLite model.
  92 
  93   Args:
  94     flags: argparse.Namespace object.
  95 
  96   Raises:
  97     ValueError: Invalid flags.
  98   """
  99   # Create converter.
 100   converter = _get_toco_converter(flags)
 101   if flags.inference_type:
 102     converter.inference_type = _types_pb2.IODataType.Value(flags.inference_type)
 103   if flags.inference_input_type:
 104     converter.inference_input_type = _types_pb2.IODataType.Value(
 105         flags.inference_input_type)
 106   if flags.output_format:
 107     converter.output_format = _toco_flags_pb2.FileFormat.Value(
 108         flags.output_format)
 109 
 110   if flags.mean_values and flags.std_dev_values:
 111     input_arrays = converter.get_input_arrays()
 112     std_dev_values = _parse_array(flags.std_dev_values, type_fn=float)
 113 
 114     # In quantized inference, mean_value has to be integer so that the real
 115     # value 0.0 is exactly representable.
 116     if flags.inference_type == lite_constants.QUANTIZED_UINT8:
 117       mean_values = _parse_array(flags.mean_values, type_fn=int)
 118     else:
 119       mean_values = _parse_array(flags.mean_values, type_fn=float)
 120     quant_stats = list(zip(mean_values, std_dev_values))
 121     if ((not flags.input_arrays and len(input_arrays) > 1) or
 122         (len(input_arrays) != len(quant_stats))):
 123       raise ValueError("Mismatching --input_arrays, --std_dev_values, and "
 124                        "--mean_values. The flags must have the same number of "
 125                        "items. The current input arrays are '{0}'. "
 126                        "--input_arrays must be present when specifying "
 127                        "--std_dev_values and --mean_values with multiple input "
 128                        "tensors in order to map between names and "
 129                        "values.".format(",".join(input_arrays)))
 130     converter.quantized_input_stats = dict(zip(input_arrays, quant_stats))
 131   if (flags.default_ranges_min is not None) and (flags.default_ranges_max is
 132                                                  not None):
 133     converter.default_ranges_stats = (flags.default_ranges_min,
 134                                       flags.default_ranges_max)
 135 
 136   if flags.drop_control_dependency:
 137     converter.drop_control_dependency = flags.drop_control_dependency
 138   if flags.reorder_across_fake_quant:
 139     converter.reorder_across_fake_quant = flags.reorder_across_fake_quant
 140   if flags.change_concat_input_ranges:
 141     converter.change_concat_input_ranges = (
 142         flags.change_concat_input_ranges == "TRUE")
 143 
 144   if flags.allow_custom_ops:
 145     converter.allow_custom_ops = flags.allow_custom_ops
 146   if flags.converter_mode:
 147     converter.converter_mode = flags.converter_mode
 148 
 149   if flags.post_training_quantize:
 150     converter.post_training_quantize = flags.post_training_quantize
 151     if flags.inference_type == lite_constants.QUANTIZED_UINT8:
 152       print("--post_training_quantize quantizes a graph of inference_type "
 153             "FLOAT. Overriding inference type QUANTIZED_UINT8 to FLOAT.")
 154       converter.inference_type = lite_constants.FLOAT
 155 
 156   if flags.dump_graphviz_dir:
 157     converter.dump_graphviz_dir = flags.dump_graphviz_dir
 158   if flags.dump_graphviz_video:
 159     converter.dump_graphviz_vode = flags.dump_graphviz_video
 160 
 161   # Convert model.
 162   output_data = converter.convert()
 163   with open(flags.output_file, "wb") as f:
 164     f.write(output_data)

 167 def _check_flags(flags, unparsed):
 168   """Checks the parsed and unparsed flags to ensure they are valid.
 169 
 170   Raises an error if previously support unparsed flags are found. Raises an
 171   error for parsed flags that don't meet the required conditions.
 172 
 173   Args:
 174     flags: argparse.Namespace object containing TFLite flags.
 175     unparsed: List of unparsed flags.
 176 
 177   Raises:
 178     ValueError: Invalid flags.
 179   """
 180 
 181   # Check unparsed flags for common mistakes based on previous TOCO.
 182   def _get_message_unparsed(flag, orig_flag, new_flag):
 183     if flag.startswith(orig_flag):
 184       return "\n  Use {0} instead of {1}".format(new_flag, orig_flag)
 185     return ""
 186 
 187   if unparsed:
 188     output = ""
 189     for flag in unparsed:
 190       output += _get_message_unparsed(flag, "--input_file", "--graph_def_file")
 191       output += _get_message_unparsed(flag, "--savedmodel_directory",
 192                                       "--saved_model_dir")
 193       output += _get_message_unparsed(flag, "--std_value", "--std_dev_values")
 194       output += _get_message_unparsed(flag, "--batch_size", "--input_shapes")
 195       output += _get_message_unparsed(flag, "--dump_graphviz",
 196                                       "--dump_graphviz_dir")
 197     if output:
 198       raise ValueError(output)
 199 
 200   # Check that flags are valid.
 201   if flags.graph_def_file and (not flags.input_arrays or
 202                                not flags.output_arrays):
 203     raise ValueError("--input_arrays and --output_arrays are required with "
 204                      "--graph_def_file")
 205 
 206   if flags.input_shapes:
 207     if not flags.input_arrays:
 208       raise ValueError("--input_shapes must be used with --input_arrays")
 209     if flags.input_shapes.count(":") != flags.input_arrays.count(","):
 210       raise ValueError("--input_shapes and --input_arrays must have the same "
 211                        "number of items")
 212 
 213   if flags.std_dev_values or flags.mean_values:
 214     if bool(flags.std_dev_values) != bool(flags.mean_values):
 215       raise ValueError("--std_dev_values and --mean_values must be used "
 216                        "together")
 217     if flags.std_dev_values.count(",") != flags.mean_values.count(","):
 218       raise ValueError("--std_dev_values, --mean_values must have the same "
 219                        "number of items")
 220 
 221   if (flags.default_ranges_min is None) != (flags.default_ranges_max is None):
 222     raise ValueError("--default_ranges_min and --default_ranges_max must be "
 223                      "used together")
 224 
 225   if flags.dump_graphviz_video and not flags.dump_graphviz_dir:
 226     raise ValueError("--dump_graphviz_video must be used with "
 227                      "--dump_graphviz_dir")
 228

 230 def run_main(_):
 231   """Main in toco_convert.py."""
 232   parser = argparse.ArgumentParser(
 233       description=("Command line tool to run TensorFlow Lite Optimizing "
 234                    "Converter (TOCO)."))
 235 
 236   # Output file flag.
 237   parser.add_argument(
 238       "--output_file",
 239       type=str,
 240       help="Full filepath of the output file.",
 241       required=True)
 242 
 243   # Input file flags.
 244   input_file_group = parser.add_mutually_exclusive_group(required=True)
 245   input_file_group.add_argument(
 246       "--graph_def_file",
 247       type=str,
 248       help="Full filepath of file containing frozen TensorFlow GraphDef.")
 249   input_file_group.add_argument(
 250       "--saved_model_dir",
 251       type=str,
 252       help="Full filepath of directory containing the SavedModel.")
 253   input_file_group.add_argument(
 254       "--keras_model_file",
 255       type=str,
 256       help="Full filepath of HDF5 file containing tf.Keras model.")
 257 
 258   # Model format flags.
 259   parser.add_argument(
 260       "--output_format",
 261       type=str.upper,
 262       choices=["TFLITE", "GRAPHVIZ_DOT"],
 263       help="Output file format.")
 264   parser.add_argument(
 265       "--inference_type",
 266       type=str.upper,
 267       choices=["FLOAT", "QUANTIZED_UINT8"],
 268       help="Target data type of real-number arrays in the output file.")
 269   parser.add_argument(
 270       "--inference_input_type",
 271       type=str.upper,
 272       choices=["FLOAT", "QUANTIZED_UINT8"],
 273       help=("Target data type of real-number input arrays. Allows for a "
 274             "different type for input arrays in the case of quantization."))
 275 
 276   # Input and output arrays flags.
 277   parser.add_argument(
 278       "--input_arrays",
 279       type=str,
 280       help="Names of the input arrays, comma-separated.")
 281   parser.add_argument(
 282       "--input_shapes",
 283       type=str,
 284       help="Shapes corresponding to --input_arrays, colon-separated.")
 285   parser.add_argument(
 286       "--output_arrays",
 287       type=str,
 288       help="Names of the output arrays, comma-separated.")
 289 
 290   # SavedModel related flags.
 291   parser.add_argument(
 292       "--saved_model_tag_set",
 293       type=str,
 294       help=("Comma-separated set of tags identifying the MetaGraphDef within "
 295             "the SavedModel to analyze. All tags must be present. "
 296             "(default \"serve\")"))
 297   parser.add_argument(
 298       "--saved_model_signature_key",
 299       type=str,
 300       help=("Key identifying the SignatureDef containing inputs and outputs. "
 301             "(default DEFAULT_SERVING_SIGNATURE_DEF_KEY)"))
 302 
 303   # Quantization flags.
 304   parser.add_argument(
 305       "--std_dev_values",
 306       type=str,
 307       help=("Standard deviation of training data for each input tensor, "
 308             "comma-separated floats. Used for quantized input tensors. "
 309             "(default None)"))
 310   parser.add_argument(
 311       "--mean_values",
 312       type=str,
 313       help=("Mean of training data for each input tensor, comma-separated "
 314             "floats. Used for quantized input tensors. (default None)"))
 315   parser.add_argument(
 316       "--default_ranges_min",
 317       type=int,
 318       help=("Default value for min bound of min/max range values used for all "
 319             "arrays without a specified range, Intended for experimenting with "
 320             "quantization via \"dummy quantization\". (default None)"))
 321   parser.add_argument(
 322       "--default_ranges_max",
 323       type=int,
 324       help=("Default value for max bound of min/max range values used for all "
 325             "arrays without a specified range, Intended for experimenting with "
 326             "quantization via \"dummy quantization\". (default None)"))
 327   # quantize_weights is DEPRECATED.
 328   parser.add_argument(
 329       "--quantize_weights",
 330       dest="post_training_quantize",
 331       action="store_true",
 332       help=argparse.SUPPRESS)
 333   parser.add_argument(
 334       "--post_training_quantize",
 335       dest="post_training_quantize",
 336       action="store_true",
 337       help=(
 338           "Boolean indicating whether to quantize the weights of the "
 339           "converted float model. Model size will be reduced and there will "
 340           "be latency improvements (at the cost of accuracy). (default False)"))
 341 
 342   # Graph manipulation flags.
 343   parser.add_argument(
 344       "--drop_control_dependency",
 345       action="store_true",
 346       help=("Boolean indicating whether to drop control dependencies silently. "
 347             "This is due to TensorFlow not supporting control dependencies. "
 348             "(default True)"))
 349   parser.add_argument(
 350       "--reorder_across_fake_quant",
 351       action="store_true",
 352       help=("Boolean indicating whether to reorder FakeQuant nodes in "
 353             "unexpected locations. Used when the location of the FakeQuant "
 354             "nodes is preventing graph transformations necessary to convert "
 355             "the graph. Results in a graph that differs from the quantized "
 356             "training graph, potentially causing differing arithmetic "
 357             "behavior. (default False)"))
 358   # Usage for this flag is --change_concat_input_ranges=true or
 359   # --change_concat_input_ranges=false in order to make it clear what the flag
 360   # is set to. This keeps the usage consistent with other usages of the flag
 361   # where the default is different. The default value here is False.
 362   parser.add_argument(
 363       "--change_concat_input_ranges",
 364       type=str.upper,
 365       choices=["TRUE", "FALSE"],
 366       help=("Boolean to change behavior of min/max ranges for inputs and "
 367             "outputs of the concat operator for quantized models. Changes the "
 368             "ranges of concat operator overlap when true. (default False)"))
 369 
 370   # Permitted ops flags.
 371   parser.add_argument(
 372       "--allow_custom_ops",
 373       action="store_true",
 374       help=("Boolean indicating whether to allow custom operations. When false "
 375             "any unknown operation is an error. When true, custom ops are "
 376             "created for any op that is unknown. The developer will need to "
 377             "provide these to the TensorFlow Lite runtime with a custom "
 378             "resolver. (default False)"))
 379   parser.add_argument(
 380       "--converter_mode",
 381       type=lite.ConverterMode,
 382       choices=list(lite.ConverterMode),
 383       help=("Experimental flag, subject to change. ConverterMode indicating "
 384             "which converter to use. (default ConverterMode.DEFAULT)"))
 385 
 386   # Logging flags.
 387   parser.add_argument(
 388       "--dump_graphviz_dir",
 389       type=str,
 390       help=("Full filepath of folder to dump the graphs at various stages of "
 391             "processing GraphViz .dot files. Preferred over --output_format="
 392             "GRAPHVIZ_DOT in order to keep the requirements of the output "
 393             "file."))
 394   parser.add_argument(
 395       "--dump_graphviz_video",
 396       action="store_true",
 397       help=("Boolean indicating whether to dump the graph after every graph "
 398             "transformation"))
 399 
 400   tflite_flags, unparsed = parser.parse_known_args(args=sys.argv[1:])
 401   try:
 402     _check_flags(tflite_flags, unparsed)
 403   except ValueError as e:
 404     parser.print_usage()
 405     file_name = os.path.basename(sys.argv[0])
 406     sys.stderr.write("{0}: error: {1}\n".format(file_name, str(e)))
 407     sys.exit(1)
 408   _convert_model(tflite_flags)
 409 
 410 
 411 def main():
 412   app.run(main=run_main, argv=sys.argv[:1])
