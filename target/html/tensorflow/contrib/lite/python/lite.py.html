  69 class TFLiteConverter(object):
  70   """Convert a TensorFlow model into `output_format` using TOCO.
  71 
  72   This is used to convert from a TensorFlow GraphDef or SavedModel into either a
  73   TFLite FlatBuffer or graph visualization.
  74 
  75   Attributes:
  76 
  77     inference_type: Target data type of real-number arrays in the output file.
  78       Must be `{FLOAT, QUANTIZED_UINT8}`.  (default FLOAT)
  79     inference_input_type: Target data type of real-number input arrays. Allows
  80       for a different type for input arrays in the case of quantization.
  81       Must be `{FLOAT, QUANTIZED_UINT8}`. (default `inference_type`)
  82     output_format: Output file format. Currently must be `{TFLITE,
  83       GRAPHVIZ_DOT}`. (default TFLITE)
  84     quantized_input_stats: Dict of strings representing input tensor names
  85       mapped to tuple of floats representing the mean and standard deviation
  86       of the training data (e.g., {"foo" : (0., 1.)}). Only need if
  87       `inference_input_type` is `QUANTIZED_UINT8`.
  88       real_input_value = (quantized_input_value - mean_value) / std_dev_value.
  89       (default {})
  90     default_ranges_stats: Tuple of integers representing (min, max) range values
  91       for all arrays without a specified range. Intended for experimenting with
  92       quantization via "dummy quantization". (default None)
  93     drop_control_dependency: Boolean indicating whether to drop control
  94       dependencies silently. This is due to TFLite not supporting control
  95       dependencies. (default True)
  96     reorder_across_fake_quant: Boolean indicating whether to reorder FakeQuant
  97       nodes in unexpected locations. Used when the location of the FakeQuant
  98       nodes is preventing graph transformations necessary to convert the graph.
  99       Results in a graph that differs from the quantized training graph,
 100       potentially causing differing arithmetic behavior. (default False)
 101     change_concat_input_ranges: Boolean to change behavior of min/max ranges for
 102       inputs and outputs of the concat operator for quantized models. Changes
 103       the ranges of concat operator overlap when true. (default False)
 104     allow_custom_ops: Boolean indicating whether to allow custom operations.
 105       When false any unknown operation is an error. When true, custom ops are
 106       created for any op that is unknown. The developer will need to provide
 107       these to the TensorFlow Lite runtime with a custom resolver.
 108       (default False)
 109     post_training_quantize: Boolean indicating whether to quantize the weights
 110       of the converted float model. Model size will be reduced and there will be
 111       latency improvements (at the cost of accuracy).
 112       (default False)
 113     dump_graphviz_dir: Full filepath of folder to dump the graphs at various
 114       stages of processing GraphViz .dot files. Preferred over
 115       --output_format=GRAPHVIZ_DOT in order to keep the requirements of the
 116       output file. (default None)
 117     dump_graphviz_video: Boolean indicating whether to dump the graph after
 118       every graph transformation. (default False)
 119     converter_mode: Experimental flag, subject to change. ConverterMode
 120       indicating which converter to use. (default ConverterMode.DEFAULT)
 121 
 122   Example usage:
 123 
 124     ```python
 125     # Converting a GraphDef from session.
 126     converter = lite.TFLiteConverter.from_session(sess, in_tensors, out_tensors)
 127     tflite_model = converter.convert()
 128     open("converted_model.tflite", "wb").write(tflite_model)
 129 
 130     # Converting a GraphDef from file.
 131     converter = lite.TFLiteConverter.from_frozen_graph(
 132       graph_def_file, input_arrays, output_arrays)
 133     tflite_model = converter.convert()
 134     open("converted_model.tflite", "wb").write(tflite_model)
 135 
 136     # Converting a SavedModel.
 137     converter = lite.TFLiteConverter.from_saved_model(saved_model_dir)
 138     tflite_model = converter.convert()
 139 
 140     # Converting a tf.keras model.
 141     converter = lite.TFLiteConverter.from_keras_model_file(keras_model)
 142     tflite_model = converter.convert()
 143     ```
 144   """
 145 
 146   def __init__(self,
 147                graph_def,
 148                input_tensors,
 149                output_tensors,
 150                input_arrays_with_shape=None,
 151                output_arrays=None):
 152     """Constructor for TFLiteConverter.
 153 
 154     Args:
 155       graph_def: Frozen TensorFlow GraphDef.
 156       input_tensors: List of input tensors. Type and shape are computed using
 157         `foo.get_shape()` and `foo.dtype`.
 158       output_tensors: List of output tensors (only .name is used from this).
 159       input_arrays_with_shape: Tuple of strings representing input tensor names
 160         and list of integers representing input shapes
 161         (e.g., [("foo" : [1, 16, 16, 3])]). Use only when graph cannot be loaded
 162           into TensorFlow and when `input_tensors` and `output_tensors` are
 163           None. (default None)
 164       output_arrays: List of output tensors to freeze graph with. Use only when
 165         graph cannot be loaded into TensorFlow and when `input_tensors` and
 166         `output_tensors` are None. (default None)
 167 
 168     Raises:
 169       ValueError: Invalid arguments.
 170     """
 171     self._graph_def = graph_def
 172     self._input_tensors = input_tensors
 173     self._output_tensors = output_tensors
 174     self.inference_type = constants.FLOAT
 175     self.inference_input_type = None
 176     self.output_format = constants.TFLITE
 177     self.quantized_input_stats = {}
 178     self.default_ranges_stats = None
 179     self.drop_control_dependency = True
 180     self.reorder_across_fake_quant = False
 181     self.change_concat_input_ranges = False
 182     self.allow_custom_ops = False
 183     self.post_training_quantize = False
 184     self.dump_graphviz_dir = None
 185     self.dump_graphviz_video = False
 186     self.converter_mode = ConverterMode.DEFAULT
 187 
 188     # Attributes are used by models that cannot be loaded into TensorFlow.
 189     if not self._has_valid_tensors():
 190       if not input_arrays_with_shape or not output_arrays:
 191         raise ValueError(
 192             "If input_tensors and output_tensors are None, both "
 193             "input_arrays_with_shape and output_arrays must be defined.")
 194       self._input_arrays_with_shape = input_arrays_with_shape
 195       self._output_arrays = output_arrays

 199     """Creates a TFLiteConverter class from a TensorFlow Session.
 200 
 201     Args:
 202       sess: TensorFlow Session.
 203       input_tensors: List of input tensors. Type and shape are computed using

 205       output_tensors: List of output tensors (only .name is used from this).

 207     Returns:
 208       TFLiteConverter class.
 209     """
 210     graph_def = _freeze_graph(sess, output_tensors)
 211     return cls(graph_def, input_tensors, output_tensors)

 213   @classmethod
 214   def from_frozen_graph(cls,
 215                         graph_def_file,
 216                         input_arrays,
 217                         output_arrays,
 218                         input_shapes=None):
 219     """Creates a TFLiteConverter class from a file containing a frozen GraphDef.
 220 
 221     Args:
 222       graph_def_file: Full filepath of file containing frozen GraphDef.
 223       input_arrays: List of input tensors to freeze graph with.
 224       output_arrays: List of output tensors to freeze graph with.
 225       input_shapes: Dict of strings representing input tensor names to list of
 226         integers representing input shapes (e.g., {"foo" : [1, 16, 16, 3]}).
 227         Automatically determined when input shapes is None (e.g., {"foo" :
 228           None}). (default None)
 229 
 230     Returns:
 231       TFLiteConverter class.
 232 
 233     Raises:
 234       IOError:
 235         File not found.
 236         Unable to parse input file.
 237       ValueError:
 238         The graph is not frozen.
 239         input_arrays or output_arrays contains an invalid tensor name.
 240         input_shapes is not correctly defined when required
 241     """
 242     with _ops.Graph().as_default():
 243       with _session.Session() as sess:
 244         # Read GraphDef from file.
 245         if not _file_io.file_exists(graph_def_file):
 246           raise IOError("File '{0}' does not exist.".format(graph_def_file))
 247         with _file_io.FileIO(graph_def_file, "rb") as f:
 248           file_content = f.read()
 249 
 250         try:
 251           graph_def = _graph_pb2.GraphDef()
 252           graph_def.ParseFromString(file_content)
 253         except (_text_format.ParseError, DecodeError):
 254           try:
 255             print("Ignore 'tcmalloc: large alloc' warnings.")
 256 
 257             if not isinstance(file_content, str):
 258               if PY3:
 259                 file_content = file_content.decode("utf-8")
 260               else:
 261                 file_content = file_content.encode("utf-8")
 262             graph_def = _graph_pb2.GraphDef()
 263             _text_format.Merge(file_content, graph_def)
 264           except (_text_format.ParseError, DecodeError):
 265             raise IOError(
 266                 "Unable to parse input file '{}'.".format(graph_def_file))
 267 
 268         # Handles models with custom TFLite ops that cannot be resolved in
 269         # TensorFlow.
 270         load_model_in_session = True
 271         try:
 272           _import_graph_def(graph_def, name="")
 273         except _NotFoundError:
 274           load_model_in_session = False
 275 
 276         if load_model_in_session:
 277           # Check if graph is frozen.
 278           if not _is_frozen_graph(sess):
 279             raise ValueError("Please freeze the graph using freeze_graph.py.")
 280 
 281           # Get input and output tensors.
 282           input_tensors = _get_tensors_from_tensor_names(
 283               sess.graph, input_arrays)
 284           output_tensors = _get_tensors_from_tensor_names(
 285               sess.graph, output_arrays)
 286           _set_tensor_shapes(input_tensors, input_shapes)
 287 
 288           return cls(sess.graph_def, input_tensors, output_tensors)
 289         else:
 290           if not input_shapes:
 291             raise ValueError("input_shapes must be defined for this model.")
 292           if set(input_arrays) != set(input_shapes.keys()):
 293             raise ValueError("input_shapes must contain a value for each item "
 294                              "in input_array.")
 295 
 296           input_arrays_with_shape = [
 297               (name, input_shapes[name]) for name in input_arrays
 298           ]
 299           return cls(
 300               graph_def,
 301               input_tensors=None,
 302               output_tensors=None,
 303               input_arrays_with_shape=input_arrays_with_shape,
 304               output_arrays=output_arrays)

 306   @classmethod
 307   def from_saved_model(cls,
 308                        saved_model_dir,
 309                        input_arrays=None,
 310                        input_shapes=None,
 311                        output_arrays=None,
 312                        tag_set=None,
 313                        signature_key=None):
 314     """Creates a TFLiteConverter class from a SavedModel.
 315 
 316     Args:
 317       saved_model_dir: SavedModel directory to convert.
 318       input_arrays: List of input tensors to freeze graph with. Uses input
 319         arrays from SignatureDef when none are provided. (default None)
 320       input_shapes: Dict of strings representing input tensor names to list of
 321         integers representing input shapes (e.g., {"foo" : [1, 16, 16, 3]}).
 322         Automatically determined when input shapes is None (e.g., {"foo" :
 323           None}). (default None)
 324       output_arrays: List of output tensors to freeze graph with. Uses output
 325         arrays from SignatureDef when none are provided. (default None)
 326       tag_set: Set of tags identifying the MetaGraphDef within the SavedModel to
 327         analyze. All tags in the tag set must be present. (default set("serve"))
 328       signature_key: Key identifying SignatureDef containing inputs and outputs.
 329         (default DEFAULT_SERVING_SIGNATURE_DEF_KEY)
 330 
 331     Returns:
 332       TFLiteConverter class.
 333     """
 334     if tag_set is None:
 335       tag_set = set([_tag_constants.SERVING])
 336     if signature_key is None:
 337       signature_key = _signature_constants.DEFAULT_SERVING_SIGNATURE_DEF_KEY
 338 
 339     result = _freeze_saved_model(saved_model_dir, input_arrays, input_shapes,
 340                                  output_arrays, tag_set, signature_key)
 341     return cls(
 342         graph_def=result[0], input_tensors=result[1], output_tensors=result[2])

 344   @classmethod
 345   def from_keras_model_file(cls,
 346                             model_file,
 347                             input_arrays=None,
 348                             input_shapes=None,
 349                             output_arrays=None):
 350     """Creates a TFLiteConverter class from a tf.keras model file.
 351 
 352     Args:
 353       model_file: Full filepath of HDF5 file containing the tf.keras model.
 354       input_arrays: List of input tensors to freeze graph with. Uses input
 355         arrays from SignatureDef when none are provided. (default None)
 356       input_shapes: Dict of strings representing input tensor names to list of
 357         integers representing input shapes (e.g., {"foo" : [1, 16, 16, 3]}).
 358         Automatically determined when input shapes is None (e.g., {"foo" :
 359           None}). (default None)
 360       output_arrays: List of output tensors to freeze graph with. Uses output
 361         arrays from SignatureDef when none are provided. (default None)
 362 
 363     Returns:
 364       TFLiteConverter class.
 365     """
 366     _keras.backend.clear_session()
 367     _keras.backend.set_learning_phase(False)
 368     keras_model = _keras.models.load_model(model_file)
 369     sess = _keras.backend.get_session()
 370 
 371     # Get input and output tensors.
 372     if input_arrays:
 373       input_tensors = _get_tensors_from_tensor_names(sess.graph, input_arrays)
 374     else:
 375       input_tensors = keras_model.inputs
 376 
 377     if output_arrays:
 378       output_tensors = _get_tensors_from_tensor_names(sess.graph, output_arrays)
 379     else:
 380       output_tensors = keras_model.outputs
 381     _set_tensor_shapes(input_tensors, input_shapes)
 382 
 383     graph_def = _freeze_graph(sess, output_tensors)
 384     return cls(graph_def, input_tensors, output_tensors)

 386   def convert(self):
 387     """Converts a TensorFlow GraphDef based on instance variables.
 388 
 389     Returns:
 390       The converted data in serialized format. Either a TFLite Flatbuffer or a
 391       Graphviz graph depending on value in `output_format`.
 392 
 393     Raises:
 394       ValueError:
 395         Input shape is not specified.
 396         None value for dimension in input_tensor.
 397         ConverterMode option is unsupported for the model.
 398     """
 399     # Checks dimensions in input tensor.
 400     if self._has_valid_tensors():
 401       for tensor in self._input_tensors:
 402         if not tensor.get_shape():
 403           raise ValueError("Provide an input shape for input array "
 404                            "'{0}'.".format(_tensor_name(tensor)))
 405         shape = tensor.get_shape().as_list()
 406         if None in shape[1:]:
 407           raise ValueError(
 408               "None is only supported in the 1st dimension. Tensor '{0}' has "
 409               "invalid shape '{1}'.".format(_tensor_name(tensor), shape))
 410         elif shape[0] is None:
 411           self._set_batch_size(batch_size=1)
 412 
 413     # Get quantization stats. Ensures there is one stat per name if the stats
 414     # are specified.
 415     if self.quantized_input_stats:
 416       quantized_stats = []
 417       invalid_stats = []
 418       for name in self.get_input_arrays():
 419         if name in self.quantized_input_stats:
 420           quantized_stats.append(self.quantized_input_stats[name])
 421         else:
 422           invalid_stats.append(name)
 423 
 424       if invalid_stats:
 425         raise ValueError("Quantization input stats are not available for input "
 426                          "tensors '{0}'.".format(",".join(invalid_stats)))
 427     else:
 428       quantized_stats = None
 429 
 430     converter_kwargs = {
 431         "inference_type": self.inference_type,
 432         "inference_input_type": self.inference_input_type,
 433         "input_format": constants.TENSORFLOW_GRAPHDEF,
 434         "output_format": self.output_format,
 435         "quantized_input_stats": quantized_stats,
 436         "default_ranges_stats": self.default_ranges_stats,
 437         "drop_control_dependency": self.drop_control_dependency,
 438         "reorder_across_fake_quant": self.reorder_across_fake_quant,
 439         "change_concat_input_ranges": self.change_concat_input_ranges,
 440         "allow_custom_ops": self.allow_custom_ops,
 441         "post_training_quantize": self.post_training_quantize,
 442         "dump_graphviz_dir": self.dump_graphviz_dir,
 443         "dump_graphviz_video": self.dump_graphviz_video
 444     }
 445 
 446     # Converts model.
 447     if self._has_valid_tensors():
 448       converter_kwargs["converter_mode"] = self.converter_mode
 449       result = _toco_convert_impl(
 450           input_data=self._graph_def,
 451           input_tensors=self._input_tensors,
 452           output_tensors=self._output_tensors,
 453           **converter_kwargs)
 454     else:
 455       # Graphs without valid tensors cannot be loaded into tf.Session since they
 456       # contain TFLite operation(s) that cannot be resolved in TensorFlow.
 457       if self.converter_mode != ConverterMode.DEFAULT:
 458         raise ValueError("This model can only be converted with the default "
 459                          "converter.")
 460       result = _toco_convert_graph_def(
 461           input_data=self._graph_def,
 462           input_arrays_with_shape=self._input_arrays_with_shape,
 463           output_arrays=self._output_arrays,
 464           **converter_kwargs)
 465     return result
 466 
 467   def get_input_arrays(self):
 468     """Returns a list of the names of the input tensors.
 469 
 470     Returns:
 471       List of strings.
 472     """
 473     if self._has_valid_tensors():
 474       return [_tensor_name(tensor) for tensor in self._input_tensors]
 475     else:
 476       return [name for name, _ in self._input_arrays_with_shape]
 477 
 478   def _has_valid_tensors(self):
 479     """Checks if the input and output tensors have been initialized.
 480 
 481     Returns:
 482       Bool.
 483     """
 484     return self._input_tensors and self._output_tensors

 486   def _set_batch_size(self, batch_size):
 487     """Sets the first dimension of the input tensor to `batch_size`.
 488 
 489     Args:
 490       batch_size: Batch size for the model. Replaces the first dimension of an
 491         input size array if undefined. (default 1)
 492 
 493     Raises:
 494       ValueError: input_tensor is not defined.
 495     """
 496     if not self._has_valid_tensors():
 497       raise ValueError("The batch size cannot be set for this model. Please "
 498                        "use input_shapes parameter.")
 499 
 500     for tensor in self._input_tensors:
 501       shape = tensor.get_shape().as_list()
 502       shape[0] = batch_size
 503       tensor.set_shape(shape)

 559 def _is_frozen_graph(sess):
 560   """Determines if the graph is frozen.
 561 
 562   Determines if a graph has previously been frozen by checking for any
 563   operations of type Variable*. If variables are found, the graph is not frozen.
 564 
 565   Args:
 566     sess: TensorFlow Session.
 567 
 568   Returns:
 569     Bool.
 570   """
 571   for op in sess.graph.get_operations():
 572     if op.type.startswith("Variable") or op.type.endswith("VariableOp"):
 573       return False
 574   return True

 577 def _freeze_graph(sess, output_tensors):
 578   """Returns a frozen GraphDef.
 579 
 580   Freezes a graph with Variables in it. Otherwise the existing GraphDef is
 581   returned.
 582 
 583   Args:
 584     sess: TensorFlow Session.
 585     output_tensors: List of output tensors (only .name is used from this).
 586 
 587   Returns:
 588     Frozen GraphDef.
 589   """
 590   if not _is_frozen_graph(sess):
 591     output_arrays = [_tensor_name(tensor) for tensor in output_tensors]
 592     return _tf_graph_util.convert_variables_to_constants(
 593         sess, sess.graph_def, output_arrays)
 594   else:
 595     return sess.graph_def
