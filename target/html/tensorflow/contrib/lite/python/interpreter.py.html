  37 class Interpreter(object):
  38   """Interpreter inferace for TF-Lite Models."""
  39 
  40   def __init__(self, model_path=None, model_content=None):
  41     """Constructor.
  42 
  43     Args:
  44       model_path: Path to TF-Lite Flatbuffer file.
  45       model_content: Content of model.
  46 
  47     Raises:
  48       ValueError: If the interpreter was unable to create.
  49     """
  50     if model_path and not model_content:
  51       self._interpreter = (
  52           _interpreter_wrapper.InterpreterWrapper_CreateWrapperCPPFromFile(
  53               model_path))
  54       if not self._interpreter:
  55         raise ValueError('Failed to open {}'.format(model_path))
  56     elif model_content and not model_path:
  57       # Take a reference, so the pointer remains valid.
  58       # Since python strings are immutable then PyString_XX functions
  59       # will always return the same pointer.
  60       self._model_content = model_content
  61       self._interpreter = (
  62           _interpreter_wrapper.InterpreterWrapper_CreateWrapperCPPFromBuffer(
  63               model_content))
  64     elif not model_path and not model_path:
  65       raise ValueError('`model_path` or `model_content` must be specified.')
  66     else:
  67       raise ValueError('Can\'t both provide `model_path` and `model_content`')
  68 
  69   def allocate_tensors(self):
  70     self._ensure_safe()
  71     return self._interpreter.AllocateTensors()
  72 
  73   def _safe_to_run(self):
  74     """Returns true if there exist no numpy array buffers.
  75 
  76     This means it is safe to run tflite calls that may destroy internally
  77     allocated memory. This works, because in the wrapper.cc we have made
  78     the numpy base be the self._interpreter.
  79     """
  80     # NOTE, our tensor() call in cpp will use _interpreter as a base pointer.
  81     # If this environment is the only _interpreter, then the ref count should be
  82     # 2 (1 in self and 1 in temporary of sys.getrefcount).
  83     return sys.getrefcount(self._interpreter) == 2
  84 
  85   def _ensure_safe(self):
  86     """Makes sure no numpy arrays pointing to internal buffers are active.
  87 
  88     This should be called from any function that will call a function on
  89     _interpreter that may reallocate memory e.g. invoke(), ...
  90 
  91     Raises:
  92       RuntimeError: If there exist numpy objects pointing to internal memory
  93         then we throw.
  94     """
  95     if not self._safe_to_run():
  96       raise RuntimeError("""There is at least 1 reference to internal data
  97       in the interpreter in the form of a numpy array or slice. Be sure to
  98       only hold the function returned from tensor() if you are using raw
  99       data access.""")
 100 
 101   def _get_tensor_details(self, tensor_index):
 102     """Gets tensor details.
 103 
 104     Args:
 105       tensor_index: Tensor index of tensor to query.
 106 
 107     Returns:
 108       a dictionary containing the name, index, shape and type of the tensor.
 109 
 110     Raises:
 111       ValueError: If tensor_index is invalid.
 112     """
 113     tensor_index = int(tensor_index)
 114     tensor_name = self._interpreter.TensorName(tensor_index)
 115     tensor_size = self._interpreter.TensorSize(tensor_index)
 116     tensor_type = self._interpreter.TensorType(tensor_index)
 117     tensor_quantization = self._interpreter.TensorQuantization(tensor_index)
 118 
 119     if not tensor_name or not tensor_type:
 120       raise ValueError('Could not get tensor details')
 121 
 122     details = {
 123         'name': tensor_name,
 124         'index': tensor_index,
 125         'shape': tensor_size,
 126         'dtype': tensor_type,
 127         'quantization': tensor_quantization,
 128     }
 129 
 130     return details
 131 
 132   def get_input_details(self):
 133     """Gets model input details.
 134 
 135     Returns:
 136       A list of input details.
 137     """
 138     return [
 139         self._get_tensor_details(i) for i in self._interpreter.InputIndices()
 140     ]

 142   def set_tensor(self, tensor_index, value):
 143     """Sets the value of the input tensor. Note this copies data in `value`.
 144 
 145     If you want to avoid copying, you can use the `tensor()` function to get a
 146     numpy buffer pointing to the input buffer in the tflite interpreter.
 147 
 148     Args:
 149       tensor_index: Tensor index of tensor to set. This value can be gotten from
 150                     the 'index' field in get_input_details.
 151       value: Value of tensor to set.
 152 
 153     Raises:
 154       ValueError: If the interpreter could not set the tensor.
 155     """
 156     self._interpreter.SetTensor(tensor_index, value)
 157 
 158   def resize_tensor_input(self, input_index, tensor_size):
 159     """Resizes an input tensor.
 160 
 161     Args:
 162       input_index: Tensor index of input to set. This value can be gotten from
 163                    the 'index' field in get_input_details.
 164       tensor_size: The tensor_shape to resize the input to.
 165 
 166     Raises:
 167       ValueError: If the interpreter could not resize the input tensor.
 168     """
 169     self._ensure_safe()
 170     # `ResizeInputTensor` now only accepts int32 numpy array as `tensor_size
 171     # parameter.
 172     tensor_size = np.array(tensor_size, dtype=np.int32)
 173     self._interpreter.ResizeInputTensor(input_index, tensor_size)
 174 
 175   def get_output_details(self):
 176     """Gets model output details.
 177 
 178     Returns:
 179       A list of output details.
 180     """
 181     return [
 182         self._get_tensor_details(i) for i in self._interpreter.OutputIndices()
 183     ]

 185   def get_tensor(self, tensor_index):
 186     """Gets the value of the input tensor (get a copy).
 187 
 188     If you wish to avoid the copy, use `tensor()`.
 189 
 190     Args:
 191       tensor_index: Tensor index of tensor to get. This value can be gotten from
 192                     the 'index' field in get_output_details.
 193 
 194     Returns:
 195       a numpy array.
 196     """
 197     return self._interpreter.GetTensor(tensor_index)
 198 
 199   def tensor(self, tensor_index):
 200     """Returns function that gives a numpy view of the current tensor buffer.
 201 
 202     This allows reading and writing to this tensors w/o copies. This more
 203     closely mirrors the C++ Interpreter class interface's tensor() member, hence
 204     the name. Be careful to not hold these output references through calls
 205     to `allocate_tensors()` and `invoke()`.
 206 
 207     Usage:
 208 
 209     interpreter.allocate_tensors()
 210     input = interpreter.tensor(interpreter.get_input_details()[0]["index"])
 211     output = interpreter.tensor(interpreter.get_output_details()[0]["index"])
 212     for i in range(10):
 213       input().fill(3.)
 214       interpreter.invoke()
 215       print("inference %s" % output())
 216 
 217     Notice how this function avoids making a numpy array directly. This is
 218     because it is important to not hold actual numpy views to the data longer
 219     than necessary. If you do, then the interpreter can no longer be invoked,
 220     because it is possible the interpreter would resize and invalidate the
 221     referenced tensors. The NumPy API doesn't allow any mutability of the
 222     the underlying buffers.
 223 
 224     WRONG:
 225 
 226     input = interpreter.tensor(interpreter.get_input_details()[0]["index"])()
 227     output = interpreter.tensor(interpreter.get_output_details()[0]["index"])()
 228     interpreter.allocate_tensors()  # This will throw RuntimeError
 229     for i in range(10):
 230       input.fill(3.)
 231       interpreter.invoke()  # this will throw RuntimeError since input,output
 232 
 233     Args:
 234       tensor_index: Tensor index of tensor to get. This value can be gotten from
 235                     the 'index' field in get_output_details.
 236 
 237     Returns:
 238       A function that can return a new numpy array pointing to the internal
 239       TFLite tensor state at any point. It is safe to hold the function forever,
 240       but it is not safe to hold the numpy array forever.
 241     """
 242     return lambda: self._interpreter.tensor(self._interpreter, tensor_index)
 243 
 244   def invoke(self):
 245     """Invoke the interpreter.
 246 
 247     Be sure to set the input sizes, allocate tensors and fill values before
 248     calling this.
 249 
 250     Raises:
 251       ValueError: When the underlying interpreter fails raise ValueError.
 252     """
 253     self._ensure_safe()
 254     self._interpreter.Invoke()
 255 
 256   def reset_all_variables(self):
 257     return self._interpreter.ResetVariableTensors()
