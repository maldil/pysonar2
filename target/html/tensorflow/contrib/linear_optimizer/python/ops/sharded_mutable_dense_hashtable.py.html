  33 class ShardedMutableDenseHashTable(lookup.LookupInterface):
  34   """A sharded version of MutableDenseHashTable.
  35 
  36   It is designed to be interface compatible with LookupInterface and
  37   MutableDenseHashTable, with the exception of the export method, which is
  38   replaced by an export_sharded method.
  39 
  40   The _ShardedMutableDenseHashTable keeps `num_shards` MutableDenseHashTable
  41   internally. The shard is computed via the modulo operation on the key.
  42   """
  43 
  44   # TODO(andreasst): consider moving this to lookup module
  45 
  46   def __init__(self,
  47                key_dtype,
  48                value_dtype,
  49                default_value,
  50                empty_key,
  51                num_shards=1,
  52                checkpoint=True,
  53                name='ShardedMutableHashTable'):
  54     with ops.name_scope(name, 'sharded_mutable_hash_table') as scope:
  55       super(ShardedMutableDenseHashTable, self).__init__(key_dtype,
  56                                                          value_dtype, scope)
  57       table_shards = []
  58       for i in range(num_shards):
  59         table_shards.append(
  60             lookup.MutableDenseHashTable(
  61                 key_dtype=key_dtype,
  62                 value_dtype=value_dtype,
  63                 default_value=default_value,
  64                 empty_key=empty_key,
  65                 checkpoint=checkpoint,
  66                 name='%s-%d-of-%d' % (name, i + 1, num_shards)))
  67       self._table_shards = table_shards
  68       # TODO(andreasst): add a value_shape() method to LookupInterface
  69       # pylint: disable=protected-access
  70       self._value_shape = self._table_shards[0]._value_shape
  71       # pylint: enable=protected-access
  72 
  73   @property
  74   def _num_shards(self):
  75     return len(self._table_shards)
  76 
  77   @property
  78   def table_shards(self):
  79     return self._table_shards
  80 
  81   def size(self, name=None):
  82     with ops.name_scope(name, 'sharded_mutable_hash_table_size'):
  83       sizes = [
  84           self._table_shards[i].size() for i in range(self._num_shards)
  85       ]
  86       return math_ops.add_n(sizes)
  87 
  88   def _shard_indices(self, keys):
  89     key_shape = keys.get_shape()
  90     if key_shape.ndims > 1:
  91       # If keys are a matrix (i.e. a single key is a vector), we use the first
  92       # element of each key vector to determine the shard.
  93       keys = array_ops.slice(keys, [0, 0], [key_shape[0].value, 1])
  94       keys = array_ops.reshape(keys, [-1])
  95     indices = math_ops.mod(math_ops.abs(keys), self._num_shards)
  96     return math_ops.cast(indices, dtypes.int32)
  97 
  98   def _check_keys(self, keys):
  99     if not keys.get_shape().is_fully_defined():
 100       raise ValueError('Key shape must be fully defined, got %s.' %
 101                        keys.get_shape())
 102     if keys.get_shape().ndims != 1 and keys.get_shape().ndims != 2:
 103       raise ValueError('Expected a vector or matrix for keys, got %s.' %
 104                        keys.get_shape())
 105 
 106   def lookup(self, keys, name=None):
 107     if keys.dtype.base_dtype != self._key_dtype:
 108       raise TypeError('Signature mismatch. Keys must be dtype %s, got %s.' %
 109                       (self._key_dtype, keys.dtype))
 110     self._check_keys(keys)
 111     num_shards = self._num_shards
 112     if num_shards == 1:
 113       return self._table_shards[0].lookup(keys, name=name)
 114 
 115     shard_indices = self._shard_indices(keys)
 116     # TODO(andreasst): support 'keys' that are not vectors
 117     key_shards = data_flow_ops.dynamic_partition(keys, shard_indices,
 118                                                  num_shards)
 119     value_shards = [
 120         self._table_shards[i].lookup(key_shards[i], name=name)
 121         for i in range(num_shards)
 122     ]
 123 
 124     num_keys = keys.get_shape().dims[0]
 125     original_indices = math_ops.range(num_keys)
 126     partitioned_indices = data_flow_ops.dynamic_partition(original_indices,
 127                                                           shard_indices,
 128                                                           num_shards)
 129     result = data_flow_ops.dynamic_stitch(partitioned_indices, value_shards)
 130     result.set_shape(
 131         tensor_shape.TensorShape([num_keys]).concatenate(self._value_shape))
 132     return result
 133 
 134   def insert(self, keys, values, name=None):
 135     self._check_keys(keys)
 136     num_shards = self._num_shards
 137     if num_shards == 1:
 138       return self._table_shards[0].insert(keys, values, name=name)
 139 
 140     shard_indices = self._shard_indices(keys)
 141     # TODO(andreasst): support 'keys' that are not vectors
 142     key_shards = data_flow_ops.dynamic_partition(keys, shard_indices,
 143                                                  num_shards)
 144     value_shards = data_flow_ops.dynamic_partition(values, shard_indices,
 145                                                    num_shards)
 146     return_values = [
 147         self._table_shards[i].insert(key_shards[i], value_shards[i], name=name)
 148         for i in range(num_shards)
 149     ]
 150 
 151     return control_flow_ops.group(*return_values)
 152 
 153   def export_sharded(self, name=None):
 154     """Returns lists of the keys and values tensors in the sharded table.
 155 
 156     Args:
 157       name: name of the table.
 158 
 159     Returns:
 160       A pair of lists with the first list containing the key tensors and the
 161         second list containing the value tensors from each shard.
 162     """
 163     keys_list = []
 164     values_list = []
 165     for table_shard in self._table_shards:
 166       exported_keys, exported_values = table_shard.export(name=name)
 167       keys_list.append(exported_keys)
 168       values_list.append(exported_values)
 169     return keys_list, values_list
