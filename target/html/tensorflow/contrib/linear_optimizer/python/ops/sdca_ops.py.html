  45 # TODO(sibyl-Aix6ihai): add name_scope to appropriate methods.
  46 class SdcaModel(object):
  47   """Stochastic dual coordinate ascent solver for linear models.
  48 
  49     Loss functions supported:
  50 
  51      * Binary logistic loss
  52      * Squared loss
  53      * Hinge loss
  54      * Smooth hinge loss
  55      * Poisson log loss
  56 
  57     This class defines an optimizer API to train a linear model.
  58 
  59     ### Usage
  60 
  61     ```python
  62     # Create a solver with the desired parameters.
  63     lr = tf.contrib.linear_optimizer.SdcaModel(examples, variables, options)
  64     min_op = lr.minimize()
  65     opt_op = lr.update_weights(min_op)
  66 
  67     predictions = lr.predictions(examples)
  68     # Primal loss + L1 loss + L2 loss.
  69     regularized_loss = lr.regularized_loss(examples)
  70     # Primal loss only
  71     unregularized_loss = lr.unregularized_loss(examples)
  72 
  73     examples: {
  74       sparse_features: list of SparseFeatureColumn.
  75       dense_features: list of dense tensors of type float32.
  76       example_labels: a tensor of type float32 and shape [Num examples]
  77       example_weights: a tensor of type float32 and shape [Num examples]
  78       example_ids: a tensor of type string and shape [Num examples]
  79     }
  80     variables: {
  81       sparse_features_weights: list of tensors of shape [vocab size]
  82       dense_features_weights: list of tensors of shape [dense_feature_dimension]
  83     }
  84     options: {
  85       symmetric_l1_regularization: 0.0
  86       symmetric_l2_regularization: 1.0
  87       loss_type: "logistic_loss"
  88       num_loss_partitions: 1 (Optional, with default value of 1. Number of
  89       partitions of the global loss function, 1 means single machine solver,
  90       and >1 when we have more than one optimizer working concurrently.)
  91       num_table_shards: 1 (Optional, with default value of 1. Number of shards
  92       of the internal state table, typically set to match the number of
  93       parameter servers for large data sets.
  94     }
  95     ```
  96 
  97     In the training program you will just have to run the returned Op from
  98     minimize().
  99 
 100     ```python
 101     # Execute opt_op and train for num_steps.
 102     for _ in range(num_steps):
 103       opt_op.run()
 104 
 105     # You can also check for convergence by calling
 106     lr.approximate_duality_gap()
 107     ```
 108   """

 110   def __init__(self, examples, variables, options):
 111     """Create a new sdca optimizer."""
 112 
 113     if not examples or not variables or not options:
 114       raise ValueError('examples, variables and options must all be specified.')
 115 
 116     supported_losses = ('logistic_loss', 'squared_loss', 'hinge_loss',
 117                         'smooth_hinge_loss', 'poisson_loss')
 118     if options['loss_type'] not in supported_losses:
 119       raise ValueError('Unsupported loss_type: ', options['loss_type'])
 120 
 121     self._assertSpecified([
 122         'example_labels', 'example_weights', 'example_ids', 'sparse_features',
 123         'dense_features'
 124     ], examples)
 125     self._assertList(['sparse_features', 'dense_features'], examples)
 126 
 127     self._assertSpecified(['sparse_features_weights', 'dense_features_weights'],
 128                           variables)
 129     self._assertList(['sparse_features_weights', 'dense_features_weights'],
 130                      variables)
 131 
 132     self._assertSpecified([
 133         'loss_type', 'symmetric_l2_regularization',
 134         'symmetric_l1_regularization'
 135     ], options)
 136 
 137     for name in ['symmetric_l1_regularization', 'symmetric_l2_regularization']:
 138       value = options[name]
 139       if value < 0.0:
 140         raise ValueError('%s should be non-negative. Found (%f)' %
 141                          (name, value))
 142 
 143     self._examples = examples
 144     self._variables = variables
 145     self._options = options
 146     self._create_slots()
 147     self._hashtable = ShardedMutableDenseHashTable(
 148         key_dtype=dtypes.int64,
 149         value_dtype=dtypes.float32,
 150         num_shards=self._num_table_shards(),
 151         default_value=[0.0, 0.0, 0.0, 0.0],
 152         # SdcaFprint never returns 0 or 1 for the low64 bits, so this a safe
 153         # empty_key (that will never collide with actual payloads).
 154         empty_key=[0, 0])
 155 
 156     summary.scalar('approximate_duality_gap', self.approximate_duality_gap())
 157     summary.scalar('examples_seen', self._hashtable.size())
 158 
 159   def _symmetric_l1_regularization(self):
 160     return self._options['symmetric_l1_regularization']
 161 
 162   def _symmetric_l2_regularization(self):
 163     # Algorithmic requirement (for now) is to have minimal l2 of 1.0.
 164     return max(self._options['symmetric_l2_regularization'], 1.0)
 165 
 166   def _num_loss_partitions(self):
 167     # Number of partitions of the global objective.
 168     # TODO(andreasst): set num_loss_partitions automatically based on the number
 169     # of workers
 170     return self._options.get('num_loss_partitions', 1)

 172   def _adaptive(self):
 173     # Perform adaptive sampling.
 174     return self._options.get('adaptive', True)
 175 
 176   def _num_table_shards(self):
 177     # Number of hash table shards.
 178     # Return 1 if not specified or if the value is 'None'
 179     # TODO(andreasst): set num_table_shards automatically based on the number
 180     # of parameter servers
 181     num_shards = self._options.get('num_table_shards')
 182     return 1 if num_shards is None else num_shards

 184   # TODO(sibyl-Aix6ihai): Use optimizer interface to make use of slot creation logic.
 185   def _create_slots(self):
 186     """Make unshrinked internal variables (slots)."""
 187     # Unshrinked variables have the updates before applying L1 regularization.
 188     # Each unshrinked slot variable is either a `Variable` or list of
 189     # `Variable`, depending on the value of its corresponding primary variable.
 190     # We avoid using `PartitionedVariable` for the unshrinked slots since we do
 191     # not need any of the extra information.
 192     self._slots = collections.defaultdict(list)
 193     for name in ['sparse_features_weights', 'dense_features_weights']:
 194       for var in self._variables[name]:
 195         # Our primary variable may be either a PartitionedVariable, or a list
 196         # of Variables (each representing a partition).
 197         if (isinstance(var, var_ops.PartitionedVariable) or
 198             isinstance(var, list)):
 199           var_list = []
 200           # pylint: disable=protected-access
 201           for v in var:
 202             with ops.colocate_with(v):
 203               # TODO(andreasst): remove SDCAOptimizer suffix once bug 30843109
 204               # is fixed.
 205               slot_var = var_ops.Variable(
 206                   initial_value=array_ops.zeros_like(v.initialized_value(),
 207                                                      dtypes.float32),
 208                   name=v.op.name + '_unshrinked/SDCAOptimizer')
 209               var_list.append(slot_var)
 210           self._slots['unshrinked_' + name].append(var_list)
 211           # pylint: enable=protected-access
 212         else:
 213           with ops.device(var.device):
 214             # TODO(andreasst): remove SDCAOptimizer suffix once bug 30843109 is
 215             # fixed.
 216             self._slots['unshrinked_' + name].append(
 217                 var_ops.Variable(
 218                     array_ops.zeros_like(var.initialized_value(),
 219                                          dtypes.float32),
 220                     name=var.op.name + '_unshrinked/SDCAOptimizer'))

 222   def _assertSpecified(self, items, check_in):
 223     for x in items:
 224       if check_in[x] is None:
 225         raise ValueError(check_in[x] + ' must be specified.')

 227   def _assertList(self, items, check_in):
 228     for x in items:
 229       if not isinstance(check_in[x], list):
 230         raise ValueError(x + ' must be a list.')

 232   def _var_to_list(self, var):
 233     """Wraps var in a list if it is not a list or PartitionedVariable."""
 234     if not (isinstance(var, list) or
 235             isinstance(var, var_ops.PartitionedVariable)):
 236       var = [var]
 237     return var

 239   def _l1_loss(self):
 240     """Computes the (un-normalized) l1 loss of the model."""
 241     with name_scope('sdca/l1_loss'):
 242       sums = []
 243       for name in ['sparse_features_weights', 'dense_features_weights']:
 244         for var in self._variables[name]:
 245           for v in self._var_to_list(var):
 246             weights = internal_convert_to_tensor(v)
 247             with ops.device(weights.device):
 248               sums.append(
 249                   math_ops.reduce_sum(
 250                       math_ops.abs(math_ops.cast(weights, dtypes.float64))))
 251       # SDCA L1 regularization cost is: l1 * sum(|weights|)
 252       return self._options['symmetric_l1_regularization'] * math_ops.add_n(sums)
 253 
 254   def _l2_loss(self, l2):
 255     """Computes the (un-normalized) l2 loss of the model."""
 256     with name_scope('sdca/l2_loss'):
 257       sums = []
 258       for name in ['sparse_features_weights', 'dense_features_weights']:
 259         for var in self._variables[name]:
 260           for v in self._var_to_list(var):
 261             weights = internal_convert_to_tensor(v)
 262             with ops.device(weights.device):
 263               sums.append(math_ops.reduce_sum(math_ops.square(math_ops.cast(
 264                   weights, dtypes.float64))))
 265       # SDCA L2 regularization cost is: l2 * sum(weights^2) / 2
 266       return l2 * math_ops.add_n(sums) / 2.0
 267 
 268   def _convert_n_to_tensor(self, input_list, as_ref=False):
 269     """Converts input list to a set of tensors."""
 270     # input_list can be a list of Variables (that are implicitly partitioned),
 271     # in which case the underlying logic in internal_convert_to_tensor will not
 272     # concatenate the partitions together.  This method takes care of the
 273     # concatenating (we only allow partitioning on the first axis).
 274     output_list = []
 275     for x in input_list:
 276       tensor_to_convert = x
 277       if isinstance(x, list) or isinstance(x, var_ops.PartitionedVariable):
 278         # We only allow for partitioning on the first axis.
 279         tensor_to_convert = array_ops.concat(x, axis=0)
 280       output_list.append(internal_convert_to_tensor(
 281           tensor_to_convert, as_ref=as_ref))
 282     return output_list
 283 
 284   def _get_first_dimension_size_statically(self, w, num_partitions):
 285     """Compute the static size of the first dimension for a sharded variable."""
 286     dim_0_size = w[0].get_shape()[0]
 287     for p in range(1, num_partitions):
 288       dim_0_size += w[p].get_shape()[0]
 289     return dim_0_size
 290 
 291   def _linear_predictions(self, examples):
 292     """Returns predictions of the form w*x."""
 293     with name_scope('sdca/prediction'):
 294       sparse_variables = self._convert_n_to_tensor(self._variables[
 295           'sparse_features_weights'])
 296       result_sparse = 0.0
 297       for sfc, sv in zip(examples['sparse_features'], sparse_variables):
 298         # TODO(sibyl-Aix6ihai): following does not take care of missing features.
 299         result_sparse += math_ops.segment_sum(
 300             math_ops.multiply(
 301                 array_ops.gather(sv, sfc.feature_indices), sfc.feature_values),
 302             sfc.example_indices)
 303       dense_features = self._convert_n_to_tensor(examples['dense_features'])
 304       dense_variables = self._convert_n_to_tensor(self._variables[
 305           'dense_features_weights'])
 306 
 307       result_dense = 0.0
 308       for i in range(len(dense_variables)):
 309         result_dense += math_ops.matmul(dense_features[i],
 310                                         array_ops.expand_dims(
 311                                             dense_variables[i], -1))
 312 
 313     # Reshaping to allow shape inference at graph construction time.
 314     return array_ops.reshape(result_dense, [-1]) + result_sparse

 316   def predictions(self, examples):
 317     """Add operations to compute predictions by the model.
 318 
 319     If logistic_loss is being used, predicted probabilities are returned.
 320     If poisson_loss is being used, predictions are exponentiated.
 321     Otherwise, (raw) linear predictions (w*x) are returned.
 322 
 323     Args:
 324       examples: Examples to compute predictions on.
 325 
 326     Returns:
 327       An Operation that computes the predictions for examples.
 328 
 329     Raises:
 330       ValueError: if examples are not well defined.
 331     """
 332     self._assertSpecified(
 333         ['example_weights', 'sparse_features', 'dense_features'], examples)
 334     self._assertList(['sparse_features', 'dense_features'], examples)
 335 
 336     result = self._linear_predictions(examples)
 337     if self._options['loss_type'] == 'logistic_loss':
 338       # Convert logits to probability for logistic loss predictions.
 339       with name_scope('sdca/logistic_prediction'):
 340         result = math_ops.sigmoid(result)
 341     elif self._options['loss_type'] == 'poisson_loss':
 342       # Exponeniate the prediction for poisson loss predictions.
 343       with name_scope('sdca/poisson_prediction'):
 344         result = math_ops.exp(result)
 345     return result

 347   def _get_partitioned_update_ops(self,
 348                                   v_num,
 349                                   num_partitions_by_var,
 350                                   p_assignments_by_var,
 351                                   gather_ids_by_var,
 352                                   weights,
 353                                   full_update,
 354                                   p_assignments,
 355                                   num_partitions):
 356     """Get updates for partitioned variables."""
 357     num_partitions = num_partitions_by_var[v_num]
 358     p_assignments = p_assignments_by_var[v_num]
 359     gather_ids = gather_ids_by_var[v_num]
 360     updates = data_flow_ops.dynamic_partition(
 361         full_update, p_assignments, num_partitions)
 362     update_ops = []
 363     for p in range(num_partitions):
 364       with ops.colocate_with(weights[p]):
 365         result = state_ops.scatter_add(weights[p], gather_ids[p], updates[p])
 366       update_ops.append(result)
 367     return update_ops

 369   def minimize(self, global_step=None, name=None):
 370     """Add operations to train a linear model by minimizing the loss function.
 371 
 372     Args:
 373       global_step: Optional `Variable` to increment by one after the
 374         variables have been updated.
 375       name: Optional name for the returned operation.
 376 
 377     Returns:
 378       An Operation that updates the variables passed in the constructor.
 379     """
 380     # Technically, the op depends on a lot more than the variables,
 381     # but we'll keep the list short.
 382     with name_scope(name, 'sdca/minimize'):
 383       sparse_example_indices = []
 384       sparse_feature_indices = []
 385       sparse_features_values = []
 386       for sf in self._examples['sparse_features']:
 387         sparse_example_indices.append(sf.example_indices)
 388         sparse_feature_indices.append(sf.feature_indices)
 389         # If feature values are missing, sdca assumes a value of 1.0f.
 390         if sf.feature_values is not None:
 391           sparse_features_values.append(sf.feature_values)
 392 
 393       # pylint: disable=protected-access
 394       example_ids_hashed = gen_sdca_ops.sdca_fprint(
 395           internal_convert_to_tensor(self._examples['example_ids']))
 396       # pylint: enable=protected-access
 397       example_state_data = self._hashtable.lookup(example_ids_hashed)
 398       # Solver returns example_state_update, new delta sparse_feature_weights
 399       # and delta dense_feature_weights.
 400 
 401       sparse_weights = []
 402       sparse_indices = []
 403       # If we have partitioned variables, keep a few dictionaries of Tensors
 404       # around that we need for the assign_add after the op call to
 405       # gen_sdca_ops.sdca_optimizer().  These are keyed because we may have a
 406       # mix of partitioned and un-partitioned variables.
 407       num_partitions_by_var = {}
 408       p_assignments_by_var = {}
 409       gather_ids_by_var = {}
 410       for v_num, (w, i) in enumerate(
 411           zip(self._slots['unshrinked_sparse_features_weights'],
 412               sparse_feature_indices)):
 413         # Append the sparse_indices (in full-variable space).
 414         sparse_idx = math_ops.cast(
 415             array_ops.unique(math_ops.cast(i, dtypes.int32))[0],
 416             dtypes.int64)
 417         sparse_indices.append(sparse_idx)
 418         if isinstance(w, list) or isinstance(w, var_ops.PartitionedVariable):
 419           num_partitions = len(w)
 420           flat_ids = array_ops.reshape(sparse_idx, [-1])
 421           # We use div partitioning, which is easiest to support downstream.
 422           # Compute num_total_ids as the sum of dim-0 of w, then assign
 423           # to partitions based on a constant number of ids per partition.
 424           # Optimize if we already know the full shape statically.
 425           dim_0_size = self._get_first_dimension_size_statically(
 426               w, num_partitions)
 427 
 428           if dim_0_size.value:
 429             num_total_ids = constant_op.constant(dim_0_size.value,
 430                                                  flat_ids.dtype)
 431           else:
 432             dim_0_sizes = []
 433             for p in range(num_partitions):
 434               if w[p].get_shape()[0].value is not None:
 435                 dim_0_sizes.append(w[p].get_shape()[0].value)
 436               else:
 437                 with ops.colocate_with(w[p]):
 438                   dim_0_sizes.append(array_ops.shape(w[p])[0])
 439             num_total_ids = math_ops.reduce_sum(
 440                 math_ops.cast(array_ops.stack(dim_0_sizes), flat_ids.dtype))
 441           ids_per_partition = num_total_ids // num_partitions
 442           extras = num_total_ids % num_partitions
 443 
 444           p_assignments = math_ops.maximum(
 445               flat_ids // (ids_per_partition + 1),
 446               (flat_ids - extras) // ids_per_partition)
 447 
 448           # Emulate a conditional using a boolean indicator tensor
 449           new_ids = array_ops.where(p_assignments < extras,
 450                                     flat_ids % (ids_per_partition + 1),
 451                                     (flat_ids - extras) % ids_per_partition)
 452 
 453           # Cast partition assignments to int32 for use in dynamic_partition.
 454           # There really should not be more than 2^32 partitions.
 455           p_assignments = math_ops.cast(p_assignments, dtypes.int32)
 456           # Partition list of ids based on assignments into num_partitions
 457           # separate lists.
 458           gather_ids = data_flow_ops.dynamic_partition(new_ids,
 459                                                        p_assignments,
 460                                                        num_partitions)
 461           # Add these into the dictionaries for use in the later update.
 462           num_partitions_by_var[v_num] = num_partitions
 463           p_assignments_by_var[v_num] = p_assignments
 464           gather_ids_by_var[v_num] = gather_ids
 465 
 466           # Gather the weights from each partition.
 467           partition_gathered_weights = []
 468           for p in range(num_partitions):
 469             with ops.colocate_with(w[p]):
 470               partition_gathered_weights.append(
 471                   array_ops.gather(w[p], gather_ids[p]))
 472 
 473           # Stitch the weights back together in the same order they were before
 474           # we dynamic_partitioned them.
 475           condition_indices = data_flow_ops.dynamic_partition(
 476               math_ops.range(array_ops.shape(new_ids)[0]),
 477               p_assignments, num_partitions)
 478           batch_gathered_weights = data_flow_ops.dynamic_stitch(
 479               condition_indices, partition_gathered_weights)
 480         else:
 481           w_as_tensor = internal_convert_to_tensor(w)
 482           with ops.device(w_as_tensor.device):
 483             batch_gathered_weights = array_ops.gather(
 484                 w_as_tensor, sparse_idx)
 485         sparse_weights.append(batch_gathered_weights)
 486 
 487       # pylint: disable=protected-access
 488       esu, sfw, dfw = gen_sdca_ops.sdca_optimizer(
 489           sparse_example_indices,
 490           sparse_feature_indices,
 491           sparse_features_values,
 492           self._convert_n_to_tensor(self._examples['dense_features']),
 493           internal_convert_to_tensor(self._examples['example_weights']),
 494           internal_convert_to_tensor(self._examples['example_labels']),
 495           sparse_indices,
 496           sparse_weights,
 497           self._convert_n_to_tensor(self._slots[
 498               'unshrinked_dense_features_weights']),
 499           example_state_data,
 500           loss_type=self._options['loss_type'],
 501           l1=self._options['symmetric_l1_regularization'],
 502           l2=self._symmetric_l2_regularization(),
 503           num_loss_partitions=self._num_loss_partitions(),
 504           num_inner_iterations=1,
 505           adaptative=self._adaptive())
 506       # pylint: enable=protected-access
 507 
 508       with ops.control_dependencies([esu]):
 509         update_ops = [self._hashtable.insert(example_ids_hashed, esu)]
 510         # Update the weights before the proximal step.
 511         for v_num, (w, i, u) in enumerate(
 512             zip(self._slots['unshrinked_sparse_features_weights'],
 513                 sparse_indices, sfw)):
 514           if (isinstance(w, var_ops.PartitionedVariable) or
 515               isinstance(w, list)):
 516             update_ops += self._get_partitioned_update_ops(
 517                 v_num, num_partitions_by_var, p_assignments_by_var,
 518                 gather_ids_by_var, w, u, p_assignments, num_partitions)
 519           else:
 520             update_ops.append(state_ops.scatter_add(w, i, u))
 521         for w, u in zip(self._slots['unshrinked_dense_features_weights'], dfw):
 522           if (isinstance(w, var_ops.PartitionedVariable) or
 523               isinstance(w, list)):
 524             split_updates = array_ops.split(
 525                 u, num_or_size_splits=[v.shape.as_list()[0] for v in w])
 526             for v, split_update in zip(w, split_updates):
 527               update_ops.append(state_ops.assign_add(v, split_update))
 528           else:
 529             update_ops.append(state_ops.assign_add(w, u))
 530       if not global_step:
 531         return control_flow_ops.group(*update_ops)
 532       with ops.control_dependencies(update_ops):
 533         return state_ops.assign_add(global_step, 1, name=name).op
 534 
 535   def update_weights(self, train_op):
 536     """Updates the model weights.
 537 
 538     This function must be called on at least one worker after `minimize`.
 539     In distributed training this call can be omitted on non-chief workers to
 540     speed up training.
 541 
 542     Args:
 543       train_op: The operation returned by the `minimize` call.
 544 
 545     Returns:
 546       An Operation that updates the model weights.
 547     """
 548     with ops.control_dependencies([train_op]):
 549       update_ops = []
 550       # Copy over unshrinked weights to user provided variables.
 551       for name in ['sparse_features_weights', 'dense_features_weights']:
 552         for var, slot_var in zip(self._variables[name],
 553                                  self._slots['unshrinked_' + name]):
 554           for v, sv in zip(self._var_to_list(var), self._var_to_list(slot_var)):
 555             update_ops.append(v.assign(sv))
 556 
 557     # Apply proximal step.
 558     with ops.control_dependencies(update_ops):
 559       update_ops = []
 560       for name in ['sparse_features_weights', 'dense_features_weights']:
 561         for var in self._variables[name]:
 562           for v in self._var_to_list(var):
 563             with ops.device(v.device):
 564               # pylint: disable=protected-access
 565               update_ops.append(
 566                   gen_sdca_ops.sdca_shrink_l1(
 567                       self._convert_n_to_tensor([v], as_ref=True),
 568                       l1=self._symmetric_l1_regularization(),
 569                       l2=self._symmetric_l2_regularization()))
 570       return control_flow_ops.group(*update_ops)
 571 
 572   def approximate_duality_gap(self):
 573     """Add operations to compute the approximate duality gap.
 574 
 575     Returns:
 576       An Operation that computes the approximate duality gap over all
 577       examples.
 578     """
 579     with name_scope('sdca/approximate_duality_gap'):
 580       _, values_list = self._hashtable.export_sharded()
 581       shard_sums = []
 582       for values in values_list:
 583         with ops.device(values.device):
 584           # For large tables to_double() below allocates a large temporary
 585           # tensor that is freed once the sum operation completes. To reduce
 586           # peak memory usage in cases where we have multiple large tables on a
 587           # single device, we serialize these operations.
 588           # Note that we need double precision to get accurate results.
 589           with ops.control_dependencies(shard_sums):
 590             shard_sums.append(
 591                 math_ops.reduce_sum(math_ops.to_double(values), 0))
 592       summed_values = math_ops.add_n(shard_sums)
 593 
 594       primal_loss = summed_values[1]
 595       dual_loss = summed_values[2]
 596       example_weights = summed_values[3]
 597       # Note: we return NaN if there are no weights or all weights are 0, e.g.
 598       # if no examples have been processed
 599       return (primal_loss + dual_loss + self._l1_loss() +
 600               (2.0 * self._l2_loss(self._symmetric_l2_regularization()))
 601              ) / example_weights
 602 
 603   def unregularized_loss(self, examples):
 604     """Add operations to compute the loss (without the regularization loss).
 605 
 606     Args:
 607       examples: Examples to compute unregularized loss on.
 608 
 609     Returns:
 610       An Operation that computes mean (unregularized) loss for given set of
 611       examples.
 612 
 613     Raises:
 614       ValueError: if examples are not well defined.
 615     """
 616     self._assertSpecified([
 617         'example_labels', 'example_weights', 'sparse_features', 'dense_features'
 618     ], examples)
 619     self._assertList(['sparse_features', 'dense_features'], examples)
 620     with name_scope('sdca/unregularized_loss'):
 621       predictions = math_ops.cast(
 622           self._linear_predictions(examples), dtypes.float64)
 623       labels = math_ops.cast(
 624           internal_convert_to_tensor(examples['example_labels']),
 625           dtypes.float64)
 626       weights = math_ops.cast(
 627           internal_convert_to_tensor(examples['example_weights']),
 628           dtypes.float64)
 629 
 630       if self._options['loss_type'] == 'logistic_loss':
 631         return math_ops.reduce_sum(math_ops.multiply(
 632             sigmoid_cross_entropy_with_logits(labels=labels,
 633                                               logits=predictions),
 634             weights)) / math_ops.reduce_sum(weights)
 635 
 636       if self._options['loss_type'] == 'poisson_loss':
 637         return math_ops.reduce_sum(math_ops.multiply(
 638             log_poisson_loss(targets=labels, log_input=predictions),
 639             weights)) / math_ops.reduce_sum(weights)
 640 
 641       if self._options['loss_type'] in ['hinge_loss', 'smooth_hinge_loss']:
 642         # hinge_loss = max{0, 1 - y_i w*x} where y_i \in {-1, 1}. So, we need to
 643         # first convert 0/1 labels into -1/1 labels.
 644         all_ones = array_ops.ones_like(predictions)
 645         adjusted_labels = math_ops.subtract(2 * labels, all_ones)
 646         # Tensor that contains (unweighted) error (hinge loss) per
 647         # example.
 648         error = nn_ops.relu(
 649             math_ops.subtract(all_ones,
 650                               math_ops.multiply(adjusted_labels, predictions)))
 651         weighted_error = math_ops.multiply(error, weights)
 652         return math_ops.reduce_sum(weighted_error) / math_ops.reduce_sum(
 653             weights)
 654 
 655       # squared loss
 656       err = math_ops.subtract(labels, predictions)
 657 
 658       weighted_squared_err = math_ops.multiply(math_ops.square(err), weights)
 659       # SDCA squared loss function is sum(err^2) / (2*sum(weights))
 660       return (math_ops.reduce_sum(weighted_squared_err) /
 661               (2.0 * math_ops.reduce_sum(weights)))
 662 
 663   def regularized_loss(self, examples):
 664     """Add operations to compute the loss with regularization loss included.
 665 
 666     Args:
 667       examples: Examples to compute loss on.
 668 
 669     Returns:
 670       An Operation that computes mean (regularized) loss for given set of
 671       examples.
 672     Raises:
 673       ValueError: if examples are not well defined.
 674     """
 675     self._assertSpecified([
 676         'example_labels', 'example_weights', 'sparse_features', 'dense_features'
 677     ], examples)
 678     self._assertList(['sparse_features', 'dense_features'], examples)
 679     with name_scope('sdca/regularized_loss'):
 680       weights = internal_convert_to_tensor(examples['example_weights'])
 681       return ((
 682           self._l1_loss() +
 683           # Note that here we are using the raw regularization
 684           # (as specified by the user) and *not*
 685           # self._symmetric_l2_regularization().
 686           self._l2_loss(self._options['symmetric_l2_regularization'])) /
 687               math_ops.reduce_sum(math_ops.cast(weights, dtypes.float64)) +
 688               self.unregularized_loss(examples))
