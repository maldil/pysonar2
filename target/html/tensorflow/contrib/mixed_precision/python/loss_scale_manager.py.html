  32 @six.add_metaclass(abc.ABCMeta)
  33 class LossScaleManager(object):
  34   """Abstract loss scale manager class.
  35 
  36   Loss scale managers with a different strategy should subclass this class.
  37   Loss scaling is a process that:
  38 
  39   1) Applies a multiplier on the loss before computing gradients, and
  40   2) Applies the reciprocal of the multiplier on the gradients before they are
  41      applied on variables.
  42 
  43   This class is used together with
  44   `tf.contrib.mixed_precision.LossScaleOptimizer` for mixed precision training
  45   (float32 variables and float16 ops) on Nvidia GPUs in order to achieve the
  46   same model quality as single precision training, with the benefits of
  47   potential higher throughput.
  48 
  49   See `tf.contrib.mixed_precision.LossScaleOptimizer` for more details.
  50   """
  51 
  52   @abc.abstractmethod
  53   def get_loss_scale(self):
  54     """Returns the loss scale as a scalar `float32` tensor."""
  55     pass
  56 
  57   @abc.abstractmethod
  58   def update_loss_scale(self, finite_grads):
  59     """Updates loss scale based on if gradients are finite in current step.
  60 
  61     Args:
  62       finite_grads: bool scalar tensor indicating if all gradients are
  63         finite (i.e., not inf or nan).
  64 
  65     Returns:
  66       An op, when executed updates the loss scale. If eager execution is
  67       enabled, does not return anything.
  68     """
  69     del finite_grads
  70     return
  71

  73 class FixedLossScaleManager(LossScaleManager):
  74   """Loss scale manager with a fixed loss scale.
  75 
  76   The loss scale is not updated for the lifetime of the class.
  77   """
  78 
  79   def __init__(self, loss_scale):
  80     """Creates the fixed loss scale manager.
  81 
  82     Args:
  83       loss_scale: A Python float. Its ideal value varies depending on models to
  84         run. Choosing a too small loss_scale might affect model quality; a too
  85         big loss_scale might cause inf or nan. There is no single right
  86         loss_scale to apply. There is no harm choosing a relatively big number
  87         as long as no nan or inf is encountered in training.
  88 
  89     Raises:
  90       ValueError: If loss_scale is less than 1.
  91     """
  92     if loss_scale < 1:
  93       raise ValueError("loss scale must be at least 1.")
  94     self._loss_scale = ops.convert_to_tensor(loss_scale, dtype=dtypes.float32)
  95 
  96   def get_loss_scale(self):
  97     return self._loss_scale
  98 
  99   def update_loss_scale(self, finite_grads):
 100     del finite_grads
 101     return gen_control_flow_ops.no_op()

 104 class ExponentialUpdateLossScaleManager(LossScaleManager):
 105   """Loss scale manager uses an exponential update strategy.
 106 
 107   In general, the strategy increases loss scale by a greater-than-one factor
 108   after encountering a consecutive series of steps with finite gradients;
 109   Similarly, it decreases the loss scale by a factor when the accumulated number
 110   of steps with non-finite (nan or inf) gradients are met. An update is not
 111   applied if its result is less than 1 or overflows the float32 dynamic range.
 112 
 113   The number of finite and non-finite steps are cleared every time the loss
 114   scale is changed. The condition to decrease the loss scale is looser than to
 115   increase it since the former does not require the steps to be consecutive.
 116   """
 117 
 118   def __init__(self,
 119                init_loss_scale,
 120                incr_every_n_steps,
 121                decr_every_n_nan_or_inf=2,
 122                incr_ratio=2,
 123                decr_ratio=0.8):
 124     """Constructor of exponential-update loss scale manager.
 125 
 126     Args:
 127       init_loss_scale: A Python float.  The loss scale to use at the beginning.
 128       incr_every_n_steps: Increases loss scale every n consecutive steps with
 129         finite gradients.
 130       decr_every_n_nan_or_inf: Decreases loss scale every n accumulated steps
 131         with nan or inf gradients.
 132       incr_ratio: The multiplier to use when increasing the loss scale.
 133       decr_ratio: The less-than-one-multiplier to use when decreasing the loss
 134         scale.
 135     """
 136     self._incr_every_n_steps = incr_every_n_steps
 137     self._decr_every_n_nan_or_inf = decr_every_n_nan_or_inf
 138     self._incr_ratio = incr_ratio
 139     self._decr_ratio = decr_ratio
 140     self._loss_scale = variable_scope.variable(
 141         name="loss_scale",
 142         initial_value=ops.convert_to_tensor(init_loss_scale, dtypes.float32),
 143         dtype=dtypes.float32,
 144         trainable=False)
 145     self._num_good_steps = variable_scope.variable(
 146         name="good_steps", initial_value=0, dtype=dtypes.int32, trainable=False)
 147     self._num_bad_steps = variable_scope.variable(
 148         name="bad_steps", initial_value=0, dtype=dtypes.int32, trainable=False)
 149 
 150   def _reset_stats(self):
 151     return control_flow_ops.group(
 152         state_ops.assign(self._num_good_steps, 0),
 153         state_ops.assign(self._num_bad_steps, 0))
 154 
 155   def get_loss_scale(self):
 156     """Returns the loss scale."""
 157     return self._loss_scale
 158 
 159   def update_loss_scale(self, finite_grads):
 160     """Updates loss scale based on if gradients are finite in current step."""
 161 
 162     def update_if_finite_grads():
 163       """Branch function when grads are all finite."""
 164 
 165       def incr_loss_scale():
 166         new_loss_scale = control_flow_ops.cond(
 167             gen_math_ops.is_finite(self._loss_scale * self._incr_ratio),
 168             lambda: self._loss_scale * self._incr_ratio,
 169             lambda: self._loss_scale)
 170         update_op = state_ops.assign(self._loss_scale, new_loss_scale)
 171         # When loss_scale is updated, both good and bad steps are reset.
 172         return control_flow_ops.group(update_op, self._reset_stats())
 173 
 174       return control_flow_ops.cond(
 175           self._num_good_steps + 1 >= self._incr_every_n_steps,
 176           incr_loss_scale,
 177           lambda: state_ops.assign_add(self._num_good_steps, 1).op)
 178 
 179     def update_if_not_finite_grads():
 180       """Branch function when any grad is not finite."""
 181 
 182       def decr_loss_scale():
 183         update_op = state_ops.assign(
 184             self._loss_scale,
 185             gen_math_ops.maximum(1., self._loss_scale * self._decr_ratio))
 186         # When loss_scale is updated, both good and bad steps are reset.
 187         return control_flow_ops.group(update_op, self._reset_stats())
 188 
 189       def just_update_steps():
 190         # When bad_steps is incremented, good_step is reset.
 191         return control_flow_ops.group(
 192             state_ops.assign_add(self._num_bad_steps, 1),
 193             state_ops.assign(self._num_good_steps, 0))
 194 
 195       return control_flow_ops.cond(
 196           self._num_bad_steps + 1 >= self._decr_every_n_nan_or_inf,
 197           decr_loss_scale, just_update_steps)
 198 
 199     return control_flow_ops.cond(finite_grads, update_if_finite_grads,
 200                                  update_if_not_finite_grads)
