  28 class LossScaleOptimizer(optimizer.Optimizer):
  29   # TODO(jamesqin): move mixed precision training explanation to __init__
  30   # docstring.
  31   """An optimizer that applies loss scaling in backprop.
  32 
  33   This class is useful for "mixed precision training" on GPUs (or other
  34   potential accelerators), an approach to improve compute throughput without
  35   compromising model quality.
  36 
  37   The canonical way to perform mixed precision training is the following:
  38   * Model variables are kept in high precision (e.g. float32).
  39   * Computations are done in lower precision (e.g. float16), which enjoys
  40     performance speedup by virtue of hardware support. Variables are casted to
  41     lower precision before they're used.
  42   * Final gradients are casted back to high precision dtype, then used to update
  43     variables.
  44 
  45   The side-effect of performing computation in lower precision, is that it comes
  46   with smaller numerical range. During backproping, small gradients might
  47   underflow in the reduced numerical range, causing a model to converge at
  48   suboptimal level.
  49 
  50   To prevent underflow, this optimizer multiplies the loss by a factor before
  51   backprop starts. Consequently, the gradients are linearly scaled up by the
  52   same factor, thus not falling into the underflow zone. After that, to perserve
  53   the correctness of backprop, the gradients are down-scaled by the same factor,
  54   casted to the (higher) variable precision, then applied on the variables.
  55 
  56   See [Nvidia's manual on mixed precision training](
  57   https://docs.nvidia.com/deeplearning/sdk/mixed-precision-training/index.html)
  58   for more details.
  59 
  60   To use loss scale optimizer, one only needs choose a loss scale strategy and
  61   wrap a regular optimizer. See examples below.
  62 
  63   ```
  64   loss = loss_fn()
  65   opt = tf.AdamOptimizer(learning_rate=...)
  66 
  67   # Choose a loss scale manager which decides how to pick the right loss scale
  68   # throughout the training process.
  69   loss_scale_manager = tf.contrib.mixed_precision.FixedLossScaleManager(5000)
  70 
  71   # Wraps the original optimizer in a LossScaleOptimizer.
  72   loss_scale_optimizer =
  73       tf.contrib.mixed_precision.LossScaleOptimizer(opt, loss_scale_manager)
  74 
  75   # Call minimize() on the loss scale optimizer.
  76   train_op = loss_scale_optimizer.minimize(loss)
  77   ```
  78 
  79   If gradients clipping is applied, one can call
  80   `optimizer.compute_gradients()` and `optimizer.apply_gradients()`
  81   separately.
  82 
  83   Notice the following way of using LossScaleOptimizer is not intended. Always
  84   use `loss_scale_optimizer.compute_gradients()` to compute gradients instead of
  85   `tf.gradients()` if doing mixed precision training.
  86 
  87   ```
  88   # The following is a wrong way to use LossScaleOptimizer along with
  89   # tf.gradients().
  90 
  91   # Always use loss_scale_optimizer.compute_gradients() to compute grads, or
  92   # loss scale is not correctly applied.
  93   grads = tf.gradients(loss, ...)
  94 
  95   # Do some custom grad clipping.
  96   grads = clip_grads(grads, ...)
  97 
  98   loss_scale_optimizer.apply(grads_and_vars)
  99   ```
 100   """
 101 
 102   def __init__(self, opt, loss_scale_manager):
 103     """Construct a loss scaling optimizer.
 104 
 105     Args:
 106       opt: The actual optimizer that will be used to compute and apply the
 107         gradients. Must be an implementation of the `tf.train.Optimizer`
 108         interface.
 109       loss_scale_manager: A LossScaleManager object.
 110     """
 111     self._opt = opt
 112     self._loss_scale_manager = loss_scale_manager
 113 
 114   def compute_gradients(self,
 115                         loss,
 116                         var_list=None,
 117                         gate_gradients=optimizer.Optimizer.GATE_OP,
 118                         aggregation_method=None,
 119                         colocate_gradients_with_ops=False,
 120                         grad_loss=None):
 121     """Compute gradients. See base class `tf.train.Optimizer`."""
 122     loss_scale = self._loss_scale_manager.get_loss_scale()
 123     if context.executing_eagerly():
 124 
 125       def scaled_loss():
 126         loss_val = loss()
 127         return loss_val * math_ops.cast(loss_scale, loss_val.dtype.base_dtype)
 128     else:
 129       if callable(loss):
 130         loss_val = loss()
 131       else:
 132         loss_val = loss
 133       scaled_loss = loss_val * math_ops.cast(loss_scale,
 134                                              loss_val.dtype.base_dtype)
 135     grads_and_vars = self._opt.compute_gradients(
 136         scaled_loss,
 137         var_list=var_list,
 138         gate_gradients=gate_gradients,
 139         aggregation_method=aggregation_method,
 140         colocate_gradients_with_ops=colocate_gradients_with_ops,
 141         grad_loss=grad_loss)
 142     return self._down_scale(grads_and_vars, loss_scale)
 143 
 144   def apply_gradients(self, grads_and_vars, global_step=None, name=None):
 145     """Apply gradients. See base class `tf.train.Optimizer`."""
 146     grads = [g for (g, _) in grads_and_vars]
 147 
 148     is_finite_grad = []
 149     for g in grads:
 150       is_finite_grad.append(math_ops.reduce_all(gen_math_ops.is_finite(g)))
 151     is_overall_finite = math_ops.reduce_all(is_finite_grad)
 152 
 153     # Only update gradients when all grads are finite.
 154     def true_apply_gradients_fn():
 155       return self._opt.apply_gradients(grads_and_vars, global_step, name)
 156 
 157     update_vars = control_flow_ops.cond(
 158         is_overall_finite, true_apply_gradients_fn, gen_control_flow_ops.no_op)
 159     # Potentially adjust gradient scale in case of finite gradients.
 160     return control_flow_ops.group(
 161         update_vars,
 162         self._loss_scale_manager.update_loss_scale(is_overall_finite))
 163 
 164   def _down_scale(self, grads_vars, loss_scale):
 165     # Down scale grads by the loss_scale.
 166     gv = []
 167     inv_loss_scale = gen_math_ops.reciprocal(loss_scale)
 168     for g, v in grads_vars:
 169       if g is not None:
 170         gv.append((g * math_ops.cast(inv_loss_scale, g.dtype.base_dtype), v))
 171       else:
 172         gv.append((g, v))
 173     return gv
