  15 """The Python API for TensorFlow's Cloud Bigtable integration.
  16 
  17 TensorFlow has support for reading from and writing to Cloud Bigtable. To use
  18 TensorFlow + Cloud Bigtable integration, first create a BigtableClient to
  19 configure your connection to Cloud Bigtable, and then create a BigtableTable
  20 object to allow you to create numerous `tf.data.Dataset`s to read data, or
  21 write a `tf.data.Dataset` object to the underlying Cloud Bigtable table.
  22 
  23 For background on Cloud Bigtable, see: https://cloud.google.com/bigtable .
  24 """

  47 class BigtableClient(object):
  48   """BigtableClient is the entrypoint for interacting with Cloud Bigtable in TF.
  49 
  50   BigtableClient encapsulates a connection to Cloud Bigtable, and exposes the
  51   `table` method to open a Bigtable table.
  52   """
  53 
  54   def __init__(self,
  55                project_id,
  56                instance_id,
  57                connection_pool_size=None,
  58                max_receive_message_size=None):
  59     """Creates a BigtableClient that can be used to open connections to tables.
  60 
  61     Args:
  62       project_id: A string representing the GCP project id to connect to.
  63       instance_id: A string representing the Bigtable instance to connect to.
  64       connection_pool_size: (Optional.) A number representing the number of
  65         concurrent connections to the Cloud Bigtable service to make.
  66       max_receive_message_size: (Optional.) The maximum bytes received in a
  67         single gRPC response.
  68 
  69     Raises:
  70       ValueError: if the arguments are invalid (e.g. wrong type, or out of
  71         expected ranges (e.g. negative).)
  72     """
  73     if not isinstance(project_id, str):
  74       raise ValueError("`project_id` must be a string")
  75     self._project_id = project_id
  76 
  77     if not isinstance(instance_id, str):
  78       raise ValueError("`instance_id` must be a string")
  79     self._instance_id = instance_id
  80 
  81     if connection_pool_size is None:
  82       connection_pool_size = -1
  83     elif connection_pool_size < 1:
  84       raise ValueError("`connection_pool_size` must be positive")
  85 
  86     if max_receive_message_size is None:
  87       max_receive_message_size = -1
  88     elif max_receive_message_size < 1:
  89       raise ValueError("`max_receive_message_size` must be positive")
  90 
  91     self._connection_pool_size = connection_pool_size
  92 
  93     self._resource = gen_bigtable_ops.bigtable_client(
  94         project_id, instance_id, connection_pool_size, max_receive_message_size)

  96   def table(self, name, snapshot=None):
  97     """Opens a table and returns a `tf.contrib.bigtable.BigtableTable` object.
  98 
  99     Args:
 100       name: A `tf.string` `tf.Tensor` name of the table to open.

 102         request the creation of a snapshot. (Note: currently unimplemented.)
 103 
 104     Returns:
 105       A `tf.contrib.bigtable.BigtableTable` Python object representing the

 107     """
 108     # TODO(saeta): Implement snapshot functionality.
 109     table = gen_bigtable_ops.bigtable_table(self._resource, name)
 110     return BigtableTable(name, snapshot, table)

 132   def lookup_columns(self, *args, **kwargs):
 133     """Retrieves the values of columns for a dataset of keys.
 134 
 135     Example usage:
 136 
 137     ```python
 138     table = bigtable_client.table("my_table")
 139     key_dataset = table.get_keys_prefix("imagenet")
 140     images = key_dataset.apply(table.lookup_columns(("cf1", "image"),
 141                                                     ("cf2", "label"),
 142                                                     ("cf2", "boundingbox")))
 143     training_data = images.map(parse_and_crop, num_parallel_calls=64).batch(128)
 144     ```
 145 
 146     Alternatively, you can use keyword arguments to specify the columns to
 147     capture. Example (same as above, rewritten):
 148 
 149     ```python
 150     table = bigtable_client.table("my_table")
 151     key_dataset = table.get_keys_prefix("imagenet")
 152     images = key_dataset.apply(table.lookup_columns(
 153         cf1="image", cf2=("label", "boundingbox")))
 154     training_data = images.map(parse_and_crop, num_parallel_calls=64).batch(128)
 155     ```
 156 
 157     Note: certain `kwargs` keys are reserved, and thus, some column families
 158     cannot be identified using the `kwargs` syntax. Instead, please use the
 159     `args` syntax. This list includes:
 160 
 161       - 'name'
 162 
 163     Note: this list can change at any time.
 164 
 165     Args:
 166       *args: A list of tuples containing (column family, column name) pairs.
 167       **kwargs: Column families (keys) and column qualifiers (values).
 168 
 169     Returns:
 170       A function that can be passed to `tf.data.Dataset.apply` to retrieve the
 171       values of columns for the rows.
 172     """
 173     table = self  # Capture self
 174     normalized = args
 175     if normalized is None:
 176       normalized = []
 177     if isinstance(normalized, tuple):
 178       normalized = list(normalized)
 179     for key, value in iteritems(kwargs):
 180       if key == "name":
 181         continue
 182       if isinstance(value, str):
 183         normalized.append((key, value))
 184         continue
 185       for col in value:
 186         normalized.append((key, col))
 187 
 188     def _apply_fn(dataset):
 189       # TODO(saeta): Verify dataset's types are correct!
 190       return _BigtableLookupDataset(dataset, table, normalized)
 191 
 192     return _apply_fn

 239   def scan_prefix(self, prefix, probability=None, columns=None, **kwargs):
 240     """Retrieves row (including values) from the Bigtable service.
 241 
 242     Rows with row-key prefixed by `prefix` will be retrieved.
 243 
 244     Specifying the columns to retrieve for each row is done by either using
 245     kwargs or in the columns parameter. To retrieve values of the columns "c1",
 246     and "c2" from the column family "cfa", and the value of the column "c3"
 247     from column family "cfb", the following datasets (`ds1`, and `ds2`) are
 248     equivalent:
 249 
 250     ```
 251     table = # ...
 252     ds1 = table.scan_prefix("row_prefix", columns=[("cfa", "c1"),
 253                                                    ("cfa", "c2"),
 254                                                    ("cfb", "c3")])
 255     ds2 = table.scan_prefix("row_prefix", cfa=["c1", "c2"], cfb="c3")
 256     ```
 257 
 258     Note: only the latest value of a cell will be retrieved.
 259 
 260     Args:
 261       prefix: The prefix all row keys must match to be retrieved for prefix-
 262         based scans.
 263       probability: (Optional.) A float between 0 (exclusive) and 1 (inclusive).
 264         A non-1 value indicates to probabilistically sample rows with the
 265         provided probability.
 266       columns: The columns to read. Note: most commonly, they are expressed as
 267         kwargs. Use the columns value if you are using column families that are
 268         reserved. The value of columns and kwargs are merged. Columns is a list
 269         of tuples of strings ("column_family", "column_qualifier").
 270       **kwargs: The column families and columns to read. Keys are treated as
 271         column_families, and values can be either lists of strings, or strings
 272         that are treated as the column qualifier (column name).
 273 
 274     Returns:
 275       A `tf.data.Dataset` returning the row keys and the cell contents.
 276 
 277     Raises:
 278       ValueError: If the configured probability is unexpected.
 279     """
 280     probability = _normalize_probability(probability)
 281     normalized = _normalize_columns(columns, kwargs)
 282     return _BigtableScanDataset(self, prefix, "", "", normalized, probability)

 449   def write(self, dataset, column_families, columns, timestamp=None):
 450     """Writes a dataset to the table.
 451 
 452     Args:
 453       dataset: A `tf.data.Dataset` to be written to this table. It must produce
 454         a list of number-of-columns+1 elements, all of which must be strings.
 455         The first value will be used as the row key, and subsequent values will
 456         be used as cell values for the corresponding columns from the
 457         corresponding column_families and columns entries.
 458       column_families: A `tf.Tensor` of `tf.string`s corresponding to the
 459         column names to store the dataset's elements into.
 460       columns: A `tf.Tensor` of `tf.string`s corresponding to the column names
 461         to store the dataset's elements into.
 462       timestamp: (Optional.) An int64 timestamp to write all the values at.
 463         Leave as None to use server-provided timestamps.
 464 
 465     Returns:
 466       A `tf.Operation` that can be run to perform the write.
 467 
 468     Raises:
 469       ValueError: If there are unexpected or incompatible types, or if the
 470         number of columns and column_families does not match the output of
 471         `dataset`.
 472     """
 473     if timestamp is None:
 474       timestamp = -1  # Bigtable server provided timestamp.
 475     for tensor_type in nest.flatten(dataset.output_types):
 476       if tensor_type != dtypes.string:
 477         raise ValueError("Not all elements of the dataset were `tf.string`")
 478     for shape in nest.flatten(dataset.output_shapes):
 479       if not shape.is_compatible_with(tensor_shape.scalar()):
 480         raise ValueError("Not all elements of the dataset were scalars")
 481     if len(column_families) != len(columns):
 482       raise ValueError("len(column_families) != len(columns)")
 483     if len(nest.flatten(dataset.output_types)) != len(columns) + 1:
 484       raise ValueError("A column name must be specified for every component of "
 485                        "the dataset elements. (e.g.: len(columns) != "
 486                        "len(dataset.output_types))")
 487     return gen_bigtable_ops.dataset_to_bigtable(
 488         self._resource,
 489         dataset._as_variant_tensor(),  # pylint: disable=protected-access
 490         column_families,
 491         columns,
 492         timestamp)

 494   def _make_parallel_scan_dataset(self, ds, num_parallel_scans,
 495                                   normalized_probability, normalized_columns):
 496     """Builds a parallel dataset from a given range.
 497 
 498     Args:
 499       ds: A `_BigtableSampleKeyPairsDataset` returning ranges of keys to use.
 500       num_parallel_scans: The number of concurrent parallel scans to use.
 501       normalized_probability: A number between 0 and 1 for the keep probability.
 502       normalized_columns: The column families and column qualifiers to retrieve.
 503 
 504     Returns:
 505       A `tf.data.Dataset` representing the result of the parallel scan.
 506     """
 507     if num_parallel_scans is None:
 508       num_parallel_scans = 50
 509 
 510     ds = ds.shuffle(buffer_size=10000)  # TODO(saeta): Make configurable.
 511 
 512     def _interleave_fn(start, end):
 513       return _BigtableScanDataset(
 514           self,
 515           prefix="",
 516           start=start,
 517           end=end,
 518           normalized=normalized_columns,
 519           probability=normalized_probability)
 520 
 521     # Note prefetch_input_elements must be set in order to avoid rpc timeouts.
 522     ds = ds.apply(
 523         interleave_ops.parallel_interleave(
 524             _interleave_fn,
 525             cycle_length=num_parallel_scans,
 526             sloppy=True,
 527             prefetch_input_elements=1))
 528     return ds

 531 def _normalize_probability(probability):
 532   if probability is None:
 533     probability = 1.0
 534   if isinstance(probability, float) and (probability <= 0.0 or
 535                                          probability > 1.0):
 536     raise ValueError("probability must be in the range (0, 1].")
 537   return probability

 540 def _normalize_columns(columns, provided_kwargs):
 541   """Converts arguments (columns, and kwargs dict) to C++ representation.
 542 
 543   Args:
 544     columns: a datastructure containing the column families and qualifier to
 545       retrieve. Valid types include (1) None, (2) list of tuples, (3) a tuple of
 546       strings.
 547     provided_kwargs: a dictionary containing the column families and qualifiers
 548       to retrieve
 549 
 550   Returns:
 551     A list of pairs of column family+qualifier to retrieve.
 552 
 553   Raises:
 554     ValueError: If there are no cells to retrieve or the columns are in an
 555       incorrect format.
 556   """
 557   normalized = columns
 558   if normalized is None:
 559     normalized = []
 560   if isinstance(normalized, tuple):
 561     if len(normalized) == 2:
 562       normalized = [normalized]
 563     else:
 564       raise ValueError("columns was a tuple of inappropriate length")
 565   for key, value in iteritems(provided_kwargs):
 566     if key == "name":
 567       continue
 568     if isinstance(value, string_types):
 569       normalized.append((key, value))
 570       continue
 571     for col in value:
 572       normalized.append((key, col))
 573   if not normalized:
 574     raise ValueError("At least one column + column family must be specified.")
 575   return normalized
