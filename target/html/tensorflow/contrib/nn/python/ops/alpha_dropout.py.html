  30 def alpha_dropout(x, keep_prob, noise_shape=None, seed=None, name=None): # pylint: disable=invalid-name
  31   """Computes alpha dropout.
  32 
  33   Alpha Dropout is a dropout that maintains the self-normalizing property. For
  34   an input with zero mean and unit standard deviation, the output of
  35   Alpha Dropout maintains the original mean and standard deviation of the input.
  36 
  37   See [Self-Normalizing Neural Networks](https://arxiv.org/abs/1706.02515)
  38 
  39   Args:
  40     x: A tensor.
  41     keep_prob: A scalar `Tensor` with the same type as x. The probability
  42       that each element is kept.
  43     noise_shape: A 1-D `Tensor` of type `int32`, representing the
  44       shape for randomly generated keep/drop flags.
  45     seed: A Python integer. Used to create random seeds. See
  46       `tf.set_random_seed` for behavior.
  47     name: A name for this operation (optional).
  48 
  49   Returns:
  50     A Tensor of the same shape of `x`.
  51 
  52   Raises:
  53     ValueError: If `keep_prob` is not in `(0, 1]`.
  54 
  55   """
  56   with ops.name_scope(name, "alpha_dropout", [x]) as name:
  57     x = ops.convert_to_tensor(x, name="x")
  58     if isinstance(keep_prob, numbers.Real) and not 0 < keep_prob <= 1.:
  59       raise ValueError("keep_prob must be a scalar tensor or a float in the "
  60                        "range (0, 1], got %g" % keep_prob)
  61     keep_prob = ops.convert_to_tensor(keep_prob,
  62                                       dtype=x.dtype,
  63                                       name="keep_prob")
  64     keep_prob.get_shape().assert_is_compatible_with(tensor_shape.scalar())
  65 
  66     # Do nothing if we know keep_prob == 1
  67     if tensor_util.constant_value(keep_prob) == 1:
  68       return x
  69 
  70     alpha = -1.7580993408473766
  71 
  72     noise_shape = noise_shape if noise_shape is not None else array_ops.shape(x)
  73     random_tensor = random_ops.random_uniform(noise_shape,
  74                                               seed=seed,
  75                                               dtype=x.dtype)
  76     kept_idx = gen_math_ops.greater_equal(random_tensor, 1 - keep_prob)
  77     kept_idx = math_ops.cast(kept_idx, x.dtype)
  78     # Mask
  79     x = x * kept_idx + alpha * (1 - kept_idx)
  80 
  81     # Affine transformation parameters
  82     a = (keep_prob + keep_prob * (1 - keep_prob) * alpha ** 2) ** -0.5
  83     b = -a * alpha * (1 - keep_prob)
  84 
  85     # Affine transformation
  86     return a * x + b
