  28 def sparsemax_loss(logits, sparsemax, labels, name=None):
  29   """Computes sparsemax loss function [1].
  30 
  31   [1]: https://arxiv.org/abs/1602.02068
  32 
  33   Args:
  34     logits: A `Tensor`. Must be one of the following types: `half`, `float32`,
  35       `float64`.
  36     sparsemax: A `Tensor`. Must have the same type as `logits`.
  37     labels: A `Tensor`. Must have the same type as `logits`.
  38     name: A name for the operation (optional).
  39 
  40   Returns:
  41     A `Tensor`. Has the same type as `logits`.
  42   """
  43 
  44   with ops.name_scope(name, "sparsemax_loss",
  45                       [logits, sparsemax, labels]) as name:
  46     logits = ops.convert_to_tensor(logits, name="logits")
  47     sparsemax = ops.convert_to_tensor(sparsemax, name="sparsemax")
  48     labels = ops.convert_to_tensor(labels, name="labels")
  49 
  50     shifted_logits = logits - \
  51         math_ops.reduce_mean(logits, axis=1)[:, array_ops.newaxis]
  52 
  53     # sum over support
  54     support = math_ops.cast(sparsemax > 0, sparsemax.dtype)
  55     sum_s = support * sparsemax * (shifted_logits - 0.5 * sparsemax)
  56 
  57     # - z_k + ||q||^2
  58     q_part = labels * (0.5 * labels - shifted_logits)
  59 
  60     return math_ops.reduce_sum(sum_s + q_part, axis=1)
