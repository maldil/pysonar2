  25 class Step(object):
  26   """Interface for performing each step of a training algorithm."""
  27 
  28   def __init__(self, distribution):
  29     self._distribution = distribution
  30 
  31   @property
  32   def distribution(self):
  33     return self._distribution
  34 
  35   def __call__(self):
  36     """Perform one step of this training algorithm."""
  37     raise NotImplementedError("must be implemented in descendants")
  38 
  39   # TODO(priyag): Add an method to access initialization and finalize ops.
  40

  42 class StandardInputStep(Step):
  43   """Step with a standard implementation of input handling.
  44 
  45   Args:
  46     dataset_fn: a function that returns a tf.data Dataset that produces the
  47       input for the model.
  48   """
  49 
  50   def __init__(self, dataset_fn, distribution):
  51     super(StandardInputStep, self).__init__(distribution)
  52     self._distributed_input = distribution.distribute_dataset(dataset_fn)
  53     self._iterator = self._distributed_input.make_one_shot_iterator()
  54 
  55 
  56 class StandardSingleLossStep(StandardInputStep):
  57   """A step function that implements a training step for a feed forward network.
  58 
  59   An instance of this class is intended to be used as a callable:
  60 
  61   ```python
  62   ...
  63   step = step_fn.StandardSingleLossStep(
  64       dataset, loss_fn, optimizer, distribution)
  65 
  66   # Run a single training step on a given DistributionStrategy:
  67   step(distribution)
  68   ...
  69   ```
  70 
  71   Args:
  72     dataset_fn: a function that returns a tf.data Dataset that produces the
  73       input for the model.
  74     loss_fn: a function that takes a context and inputs as arguments. It returns
  75       the loss for those inputs. `context` is an instance of
  76       `values.MultiStepContext` that will be passed when `loss_fn` is run.
  77       `context` can be used to specify the outputs to be returned from
  78       `loss_fn`, among other things.
  79     optimizer: an optimizer that implements an update rule.
  80     distribution: a `DistributionStrategy` object.
  81   """
  82 
  83   def __init__(self, dataset_fn, loss_fn, optimizer, distribution,
  84                iterations_per_step=1):
  85     super(StandardSingleLossStep, self).__init__(dataset_fn, distribution)
  86     self._loss_fn = loss_fn
  87     self._optimizer = optimizer
  88     self._is_run_concurrently = False
  89     self._iterations_per_step = iterations_per_step
  90 
  91   def __call__(self):
  92     with self._distribution.scope():
  93       def step_fn(ctx, *inputs):
  94         """Function to run one iteration with one input."""
  95         gradients_fn = backprop.implicit_grad(self._loss_fn)
  96         gradients_fn = optimizer_lib.get_filtered_grad_fn(gradients_fn)
  97 
  98         grads_and_vars = self.distribution.call_for_each_tower(
  99             gradients_fn,
 100             ctx, *inputs,
 101             run_concurrently=self._is_run_concurrently)
 102         # If threads use layers, then we need to run the first step
 103         # sequentially, so that layers.build() is not executed in parallel.
 104         # Otherwise, multiple sets of mirrored variables are going to be
 105         # created.
 106         self._is_run_concurrently = True
 107         return self._optimizer._distributed_apply(  # pylint: disable=protected-access
 108             self.distribution, grads_and_vars)
 109 
 110       # TODO(priyag): Return the outputs, context, etc as well.
 111       ctx = self.distribution.run_steps_on_dataset(
 112           step_fn, self._iterator, self._iterations_per_step)
 113       return ctx.run_op
