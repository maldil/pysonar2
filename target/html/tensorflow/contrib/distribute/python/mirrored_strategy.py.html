  76 def _call_for_each_tower(distribution, fn, *args, **kwargs):
  77   """Run `fn` in separate threads, once per tower/worker device.
  78 
  79   Args:
  80     distribution: the DistributionStrategy object.
  81     fn: function to run (will be run once per device, each in its own thread).
  82     *args: positional arguments for `fn`
  83     **kwargs: keyword arguments for `fn`.
  84         `"run_concurrently"`: Boolean indicating whether executions of `fn`
  85            can be run concurrently (under eager execution only), defaults to
  86            `True`.
  87 
  88   Returns:
  89     Merged return value of `fn` across all towers.
  90 
  91   Raises:
  92     RuntimeError: If fn() calls get_tower_context().merge_call() a different
  93         number of times from the available devices.
  94   """
  95   run_concurrently = kwargs.pop("run_concurrently", True)
  96   if not context.executing_eagerly():
  97     # Lots of TF library code isn't thread-safe in graph mode, and
  98     # there is little to be gained by turning on multithreading when
  99     # constructing a graph.
 100     run_concurrently = False
 101     # Needed for per-thread device, etc. contexts in graph mode.
 102     ops.get_default_graph().switch_to_thread_local()
 103   elif run_concurrently is None:
 104     run_concurrently = True
 105 
 106   coord = coordinator.Coordinator(clean_stop_exception_types=(_RequestedStop,))
 107 
 108   shared_variable_store = {}
 109 
 110   # TODO(isaprykin): Create these threads once instead of during every run()
 111   # call.
 112   threads = []
 113   for index, d in enumerate(distribution.worker_devices):
 114     variable_creator_fn = shared_variable_creator.make_fn(
 115         shared_variable_store, index)
 116     t = MirroredStrategy._MirroredTowerThread(  # pylint: disable=protected-access
 117         distribution, coord, d, variable_creator_fn, fn,
 118         *values.select_device(d, args), **values.select_device(d, kwargs))
 119     threads.append(t)
 120 
 121   for t in threads:
 122     t.start()
 123 
 124   # When `fn` starts `should_run` event is set on _MirroredTowerThread
 125   # (`MTT`) threads. The execution waits until
 126   # `MTT.has_paused` is set, which indicates that either `fn` is
 127   # complete or a `get_tower_context().merge_call()` is called.  If `fn` is
 128   # complete, then `MTT.done` is set to True.  Otherwise, arguments
 129   # of `get_tower_context().merge_call` from all paused threads are grouped
 130   # and the `merge_fn` is performed.  Results of the
 131   # `get_tower_context().merge_call` are then set to `MTT.merge_result`.
 132   # Each such `get_tower_context().merge_call` call returns the
 133   # `MTT.merge_result` for that thread when `MTT.should_run` event
 134   # is reset again. Execution of `fn` resumes.
 135 
 136   try:
 137     with coord.stop_on_exception():
 138       all_done = False
 139       while not all_done and not coord.should_stop():
 140         done = []
 141         if run_concurrently:
 142           for t in threads:
 143             t.should_run.set()
 144           for t in threads:
 145             t.has_paused.wait()
 146             t.has_paused.clear()
 147             if coord.should_stop():
 148               return None
 149             done.append(t.done)
 150         else:
 151           for t in threads:
 152             t.should_run.set()
 153             t.has_paused.wait()
 154             t.has_paused.clear()
 155             if coord.should_stop():
 156               return None
 157             done.append(t.done)
 158         if coord.should_stop():
 159           return None
 160         all_done = all(done)
 161         if not all_done:
 162           if any(done):
 163             raise RuntimeError("Some towers made a different number of "
 164                                "tower_context().merge_call() calls.")
 165           # get_tower_context().merge_call() case
 166           merge_args = values.regroup({t.device: t.merge_args for t in threads})
 167           merge_kwargs = values.regroup(
 168               {t.device: t.merge_kwargs for t in threads})
 169           # We capture the name_scope of the MTT when we call merge_fn
 170           # to ensure that if we have opened a name scope in the MTT,
 171           # it will be respected when executing the merge function. We only
 172           # capture the name_scope from the first MTT and assume it is
 173           # the same for all other MTTs.
 174           mtt_captured_name_scope = threads[0].captured_name_scope
 175           with ops.name_scope(mtt_captured_name_scope):
 176             merge_result = threads[0].merge_fn(distribution, *merge_args,
 177                                                **merge_kwargs)
 178           for t in threads:
 179             t.merge_result = values.select_device(t.device, merge_result)
 180   finally:
 181     for t in threads:
 182       t.should_run.set()
 183     coord.join(threads)
 184 
 185   return values.regroup({t.device: t.main_result for t in threads})

 188 def _reduce_non_distributed_value(distribution, aggregation, value,
 189                                   destinations):
 190   """Reduce a non-DistributedValue `value` to `destinations`."""
 191   if isinstance(value, values.DistributedValues):
 192     raise ValueError("You are passing a `DistributedValue` to "
 193                      "`_reduce_non_distributed_value`, which is not allowed.")
 194 
 195   # If the same value is present on all towers then the PerDevice value will
 196   # be a single value. We also handle the case when `value` is a single value
 197   # and equal to 0.
 198   if value == 0:
 199     return 0
 200   # If the aggregation type is MEAN or ONLY_FIRST_TOWER, then this
 201   # essentially means that the same value should be on all destinations.
 202   if aggregation in (
 203       variable_scope.VariableAggregation.MEAN,
 204       variable_scope.VariableAggregation.ONLY_FIRST_TOWER):
 205     return value
 206 
 207   cross_tower_ops_lib.validate_destinations(destinations)
 208   # We do not support an aggregation type of SUM if the value is the same across
 209   # all towers. We call this as part of assign functions for MirroredVariables
 210   # and summing up identical values across towers is not clearly defined.
 211   if (len(distribution.worker_devices) != 1 or
 212       not cross_tower_ops_lib.check_destinations(destinations)):
 213     raise ValueError("A non-DistributedValues value %s cannot be reduced with "
 214                      "the given aggregation %s." % (value, aggregation))
 215   # TODO(anjalisridhar): Moves these methods to a device utility file?
 216   devices = cross_tower_ops_lib.get_devices_from(destinations)
 217   if len(devices) == 1:
 218     with ops.device(devices[0]):
 219       return array_ops.identity(value)
 220   else:
 221     value_updates = {}
 222     for d in devices:
 223       with ops.device(d):
 224         value_updates[d] = array_ops.identity(value)
 225     return values.Mirrored(value_updates)

 228 def _create_mirrored_variable(devices, real_mirrored_creator, *args, **kwargs):  # pylint: disable=g-missing-docstring
 229   # Figure out what collections this variable should be added to.
 230   # We'll add the MirroredVariable to those collections instead.
 231   collections = kwargs.pop("collections", None)
 232   if collections is None:
 233     collections = [ops.GraphKeys.GLOBAL_VARIABLES]
 234   kwargs["collections"] = []
 235 
 236   # Get synchronization value
 237   synchronization = kwargs.get("synchronization",
 238                                variable_scope.VariableSynchronization.ON_WRITE)
 239   if synchronization == variable_scope.VariableSynchronization.NONE:
 240     raise ValueError("`NONE` variable synchronization mode is not "
 241                      "supported with `Mirrored` distribution strategy. Please"
 242                      " change the `synchronization` for variable: " +
 243                      kwargs["name"])
 244   elif synchronization == variable_scope.VariableSynchronization.ON_READ:
 245     # Variables that are to be synced on read are tower local.
 246     is_tower_local = True
 247     kwargs["trainable"] = False
 248   elif (synchronization == variable_scope.VariableSynchronization.ON_WRITE or
 249         synchronization == variable_scope.VariableSynchronization.AUTO):
 250     # `AUTO` synchronization for `MirroredStrategy` is `ON_WRITE`.
 251     is_tower_local = False
 252   else:
 253     raise ValueError("Invalid variable synchronization mode: " +
 254                      synchronization + " for variable: " + kwargs["name"])
 255 
 256   # Get aggregation value
 257   aggregation = kwargs.pop("aggregation",
 258                            variable_scope.VariableAggregation.NONE)
 259   if aggregation not in (
 260       variable_scope.VariableAggregation.NONE,
 261       variable_scope.VariableAggregation.SUM,
 262       variable_scope.VariableAggregation.MEAN,
 263       variable_scope.VariableAggregation.ONLY_FIRST_TOWER
 264   ):
 265     raise ValueError("Invalid variable aggregation mode: " + aggregation +
 266                      " for variable: " + kwargs["name"])
 267 
 268   # Ignore user-specified caching device, not needed for mirrored variables.
 269   kwargs.pop("caching_device", None)
 270 
 271   # TODO(josh11b,apassos): It would be better if variable initialization
 272   # was never recorded on the tape instead of having to do this manually
 273   # here.
 274   with tape.stop_recording():
 275     index = real_mirrored_creator(devices, *args, **kwargs)
 276 
 277     if is_tower_local:
 278       result = values.TowerLocalVariable(index, index[devices[0]], aggregation)
 279     else:
 280       result = values.MirroredVariable(index, index[devices[0]], aggregation)
 281 
 282   # Add the wrapped variable to the requested collections.
 283   # The handling of eager mode and the global step matches
 284   # ResourceVariable._init_from_args().
 285   if not context.executing_eagerly():
 286     g = ops.get_default_graph()
 287     # If "trainable" is True, next_creator() will add the member variables
 288     # to the TRAINABLE_VARIABLES collection, so we manually remove
 289     # them and replace with the MirroredVariable. We can't set
 290     # "trainable" to False for next_creator() since that causes functions
 291     # like implicit_gradients to skip those variables.
 292     if kwargs.get("trainable", True):
 293       collections.append(ops.GraphKeys.TRAINABLE_VARIABLES)
 294       l = g.get_collection_ref(ops.GraphKeys.TRAINABLE_VARIABLES)
 295       for v in index.values():
 296         l.remove(v)
 297     g.add_to_collections(collections, result)
 298   elif ops.GraphKeys.GLOBAL_STEP in collections:
 299     ops.add_to_collections(ops.GraphKeys.GLOBAL_STEP, result)
 300 
 301   return result

 304 class MirroredStrategy(distribute_lib.DistributionStrategy):
 305   """Mirrors vars to distribute across multiple devices and machines.
 306 
 307   This strategy uses one tower per device and sync replication for its multi-GPU
 308   version.
 309 
 310   When `cluster_spec` is given by the `configure` method., it turns into the
 311   mulit-worker version that works on multiple workers with in-graph replication.
 312   Note: `configure` will be called by higher-level APIs if running in
 313   distributed environment.
 314 
 315   There are several important concepts for distributed TensorFlow, e.g.
 316   `client`, `job`, 'task', `cluster`, `in-graph replication` and
 317   'synchronous training' and they have already been defined in the
 318   [TensorFlow's documentation](https://www.tensorflow.org/deploy/distributed).
 319   The distribution strategy inherits these concepts as well and in addition to
 320   that we also clarify several more concepts:
 321     * **In-graph replication**: the `client` creates a single `tf.Graph` that
 322     specifies tasks for devices on all workers. The `client` then creates a
 323     client session which will talk to the `master` service of a `worker`. Then
 324     the `master` will partition the graph and distribute the work to all
 325     participating workers.
 326     * **Worker**: A `worker` is a TensorFlow `task` that usually maps to one
 327     physical machine. We will have multiple `worker`s with different `task`
 328     index. They all do similar things except for one worker checkpointing model
 329     variables, writing summaries, etc. in addition to its ordinary work.
 330 
 331   The multi-worker version of this class maps one tower to one device on a
 332   worker. It mirrors all model variables on all towers. For example, if you have
 333   two `worker`s and each `worker` has 4 GPUs, it will create 8 copies of the
 334   model variables on these 8 GPUs. Then like in MirroredStrategy, each tower
 335   performs their computation with their own copy of variables unless in
 336   cross-tower model where variable or tensor reduction happens.
 337 
 338   Args:
 339     devices: a list of device strings.
 340     num_gpus: number of GPUs. For local training, either specify `devices` or
 341       `num_gpus`. In distributed training, this must be specified as number of
 342       GPUs on each worker.
 343     num_gpus_per_worker: number of GPUs per worker. This is the same as
 344       `num_gpus` and only one of `num_gpus` and `num_gpus_per_worker` can be
 345       specified.
 346     cross_tower_ops: optional, a descedant of `CrossTowerOps`. If this is not
 347       set, the `configure` method will try to find the best one.
 348     prefetch_on_device: optional boolean to specify whether to prefetch input
 349       data to devices.
 350     auto_shard_dataset: whether to auto-shard the dataset when there are
 351       multiple workers.
 352   """
 353 
 354   def __init__(self,
 355                devices=None,
 356                num_gpus=None,
 357                num_gpus_per_worker=None,
 358                cross_tower_ops=None,
 359                prefetch_on_device=None,
 360                auto_shard_dataset=False):
 361     super(MirroredStrategy, self).__init__()
 362 
 363     self._cross_tower_ops = cross_tower_ops
 364     self._prefetch_on_device = prefetch_on_device
 365     self._auto_shard_dataset = auto_shard_dataset
 366     # Rememeber num GPUs which might be needed by `configure` method.
 367     if num_gpus is not None and num_gpus_per_worker is not None:
 368       raise ValueError(
 369           "You cannot specify both `num_gpus` and `num_gpus_per_worker`.")
 370     if num_gpus is not None:
 371       self._num_gpus = num_gpus
 372     else:
 373       self._num_gpus = num_gpus_per_worker
 374 
 375     self._initialize_local(self._num_gpus, devices)
 376 
 377   def _initialize_local(self, num_gpus, devices):
 378     """Initializes the object for local training."""
 379     self._cluster_spec = None
 380     # Convert `num_gpus` into `devices`, shouldn't specify both.
 381     if devices is None:
 382       if num_gpus is None:
 383         num_gpus = context.num_gpus()
 384       if num_gpus == 0:
 385         devices = ["/device:CPU:0"]
 386       else:
 387         devices = ["/device:GPU:%d" % d for d in range(num_gpus)]
 388     elif num_gpus is not None:
 389       raise ValueError("Must only specify one of `devices` and `num_gpus`.")
 390     self._num_gpus = num_gpus
 391     # TODO(yuefengz): consider setting the default device.
 392 
 393     assert devices, "Must specify at least one device."
 394     assert len(set(devices)) == len(devices), (
 395         "No duplicates allowed in `devices` argument.")
 396     # TODO(josh11b): Require at least 2 devices?
 397     self._devices = [device_util.resolve(d) for d in devices]
 398     self._canonical_device_set = set(self._devices)
 399     self._device_index = values.PerDevice({d: i for i, d in enumerate(devices)})
 400 
 401   def _initialize_multi_worker(self, num_gpus, cluster_spec):
 402     """Initializes the object for multi-worker training."""
 403     cluster_spec = multi_worker_util.normalize_cluster_spec(cluster_spec)
 404     self._cluster_spec = cluster_spec
 405 
 406     self._workers = []
 407     for job in ["chief", "worker"]:
 408       for task in range(len(cluster_spec.as_dict().get(job, []))):
 409         self._workers.append("/job:%s/task:%d" % (job, task))
 410 
 411     if num_gpus is None:
 412       raise ValueError("`num_gpus` is required if `cluster_spec` is given.")
 413     if num_gpus > 0:
 414       self._worker_device_map = {
 415           worker: [
 416               device_util.canonicalize(worker + "/device:GPU:%d" % gpu)
 417               for gpu in range(num_gpus)
 418           ] for worker in self._workers
 419       }
 420     else:
 421       self._worker_device_map = {
 422           worker: [device_util.canonicalize(worker, "/device:CPU:0")]
 423           for worker in self._workers
 424       }
 425 
 426     devices = nest.flatten(self._worker_device_map)
 427 
 428     # Setting `_default_device` will add a device scope in the
 429     # distribution.scope. We set the default device to the first worker. When
 430     # users specify device under distribution.scope by
 431     #   with tf.device("/cpu:0"):
 432     #     ...
 433     # their ops will end up on the cpu device of its first worker, e.g.
 434     # "/job:worker/task:0/device:CPU:0". Note this is not used in tower mode.
 435     self._default_device = self._workers[0]
 436 
 437     assert devices, "Must specify at least one device."
 438     assert len(set(devices)) == len(devices), (
 439         "No duplicates allowed in `devices` argument.")
 440     # TODO(josh11b): Require at least 2 devices?
 441     self._devices = [device_util.resolve(d) for d in devices]
 442     self._canonical_device_set = set(self._devices)
 443     self._device_index = values.PerDevice(
 444         {d: i for i, d in enumerate(devices)})
 445 
 446   def _create_variable(self, next_creator, *args, **kwargs):
 447     """Create a mirrored variable. See `DistributionStrategy.scope`."""
 448     colocate_with = kwargs.pop("colocate_with", None)
 449     devices = self._get_devices_from(colocate_with)
 450 
 451     def _real_mirrored_creator(devices, *args, **kwargs):  # pylint: disable=g-missing-docstring
 452       index = {}
 453       for i, d in enumerate(devices):
 454         with ops.device(d):
 455           if i > 0:
 456             # Give replicas meaningful distinct names:
 457             var0name = index[devices[0]].name.split(":")[0]
 458             # We append a / to variable names created on towers with id > 0 to
 459             # ensure that we ignore the name scope and instead use the given
 460             # name as the absolute name of the variable.
 461             kwargs["name"] = "%s/replica_%d/" % (var0name, i)
 462             # Initialize replicas with the same value:
 463             if context.executing_eagerly():
 464               kwargs["initial_value"] = array_ops.identity(
 465                   index[devices[0]].value())
 466             else:
 467               def initial_value_fn(device=d):
 468                 with ops.device(device):
 469                   return array_ops.identity(index[devices[0]].initial_value)
 470               kwargs["initial_value"] = initial_value_fn
 471           with context.context().device_policy(context.DEVICE_PLACEMENT_SILENT):
 472             v = next_creator(*args, **kwargs)
 473           assert not isinstance(v, values.DistributedVariable)
 474           index[d] = v
 475       return index
 476 
 477     return _create_mirrored_variable(devices, _real_mirrored_creator, *args,
 478                                      **kwargs)
 479 
 480   def distribute_dataset(self, dataset_fn):
 481     if self._cluster_spec:
 482       return values.MultiWorkerDataset(
 483           partial(self._call_dataset_fn, dataset_fn), self._worker_device_map,
 484           self._prefetch_on_device, self._auto_shard_dataset)
 485     else:
 486       return values.PerDeviceDataset(
 487           self._call_dataset_fn(dataset_fn), self._devices,
 488           self._prefetch_on_device)
 489 
 490   # TODO(priyag): Deal with OutOfRange errors once b/111349762 is fixed.
 491   def _run_steps_on_dataset(self, fn, iterator, iterations,
 492                             initial_loop_values=None):
 493     if initial_loop_values is None:
 494       initial_loop_values = {}
 495     initial_loop_values = nest.flatten(initial_loop_values)
 496 
 497     ctx = values.MultiStepContext()
 498     def body(i, *args):
 499       """A wrapper around `fn` to create the while loop body."""
 500       del args
 501       fn_inputs = iterator.get_next()
 502       if not isinstance(fn_inputs, tuple):
 503         fn_inputs = (fn_inputs,)
 504       fn_result = fn(ctx, *fn_inputs)
 505       for (name, output) in ctx.last_step_outputs.items():
 506         # Convert all outputs to tensors, potentially from `DistributedValues`.
 507         ctx.last_step_outputs[name] = self.unwrap(output)
 508       flat_last_step_outputs = nest.flatten(ctx.last_step_outputs)
 509       with ops.control_dependencies([fn_result]):
 510         return [i + 1] + flat_last_step_outputs
 511 
 512     # We capture the control_flow_context at this point, before we run `fn`
 513     # inside a while_loop. This is useful in cases where we might need to exit
 514     # these contexts and get back to the outer context to do some things, for
 515     # e.g. create an op which should be evaluated only once at the end of the
 516     # loop on the host. One such usage is in creating metrics' value op.
 517     self._outer_control_flow_context = (
 518         ops.get_default_graph()._get_control_flow_context())  # pylint: disable=protected-access
 519 
 520     cond = lambda i, *args: i < iterations
 521     i = constant_op.constant(0)
 522     loop_result = control_flow_ops.while_loop(
 523         cond, body, [i] + initial_loop_values, name="",
 524         parallel_iterations=1, back_prop=False, swap_memory=False,
 525         return_same_structure=True)
 526     del self._outer_control_flow_context
 527 
 528     ctx.run_op = control_flow_ops.group(loop_result)
 529 
 530     # Convert the last_step_outputs from a list to the original dict structure
 531     # of last_step_outputs.
 532     last_step_tensor_outputs = loop_result[1:]
 533     last_step_tensor_outputs_dict = nest.pack_sequence_as(
 534         ctx.last_step_outputs, last_step_tensor_outputs)
 535 
 536     for (name, aggregation) in ctx._last_step_outputs_aggregations.items():  # pylint: disable=protected-access
 537       output = last_step_tensor_outputs_dict[name]
 538       # For outputs that have already been aggregated, wrap them in a Mirrored
 539       # container, else in a PerDevice container.
 540       if aggregation is variables_lib.VariableAggregation.NONE:
 541         last_step_tensor_outputs_dict[name] = values.regroup(
 542             {d: t for d, t in zip(self._devices, output)}, values.PerDevice)
 543       else:
 544         assert len(output) == 1
 545         last_step_tensor_outputs_dict[name] = output[0]
 546 
 547     ctx._set_last_step_outputs(last_step_tensor_outputs_dict)  # pylint: disable=protected-access
 548     return ctx

 558   def map(self, map_over, fn, *args, **kwargs):
 559     # TODO(josh11b): In eager mode, use one thread per device.
 560     index = {}
 561     for i, m in enumerate(map_over):
 562       d = self._devices[i % len(self._devices)]
 563       with ops.device(d):
 564         l = index.get(d, [])
 565         l.append(fn(m,
 566                     *values.select_device_mirrored(d, args),
 567                     **values.select_device_mirrored(d, kwargs)))
 568         index[d] = l
 569     # TODO(josh11b): Need a values.regroup equivalent that handles MapOutput
 570     # in addition to PerDevice data.
 571     return values.PerDevice({k: values.MapOutput(v) for k, v in index.items()})

 573   def configure(self,
 574                 session_config=None,
 575                 cluster_spec=None,
 576                 task_type=None,
 577                 task_id=None):

 580     if session_config:

 588         # It currently cannot detect the toplogy of remote workers. So we
 589         # hard-code the multi-worker all-reduce algorithm for now.

 591           # The default is "nccl".
 592           self._cross_tower_ops = cross_tower_ops_lib.AllReduceCrossTowerOps()
 593         else:
 594           # The default is hierarchical reduce and broadcast.

 596               self._workers, self._num_gpus)

 607   def _reduce(self, aggregation, value, destinations):
 608     assert not isinstance(value, values.Mirrored)
 609     if not isinstance(value, values.DistributedValues):
 610       # This function handles reducing values that are not PerDevice or Mirrored
 611       # values. For example, the same value could be present on all towers in
 612       # which case `value` would be a single value or value could be 0.
 613       return _reduce_non_distributed_value(self, aggregation, value,
 614                                            destinations)
 615     if aggregation == variable_scope.VariableAggregation.ONLY_FIRST_TOWER:
 616       value = value.get(self._devices[0])
 617       if isinstance(value, (int, float)):
 618         return value
 619       return self.broadcast(value, destinations)
 620     return self._get_cross_tower_ops().reduce(
 621         aggregation, value, destinations=destinations)

 623   def _batch_reduce(self, aggregation, value_destination_pairs):

 625       return [self.broadcast(v.get(self._devices[0]), d)
 626               for v, d in value_destination_pairs]
 627     return self._get_cross_tower_ops().batch_reduce(aggregation,
 628                                                     value_destination_pairs)

 632     assert isinstance(var, values.DistributedVariable)
 633     should_group = options.pop("grouped")
 634     assert not options  # Validate that we are processing all of the options.
 635     updates = {}
 636     for d, v in var._index.items():  # pylint: disable=protected-access
 637       name = "update_%d" % self._device_index.get(d)
 638       with ops.device(d), distribute_lib.UpdateContext(d), ops.name_scope(name):
 639         # If args and kwargs are not mirrored, the value is returned as is.

 641                         *values.select_device_mirrored(d, args),
 642                         **values.select_device_mirrored(d, kwargs))

 722   class _MirroredTowerThread(threading.Thread):
 723     """A thread that runs() a function on a device."""
 724 
 725     def __init__(self, dist, coord, device, variable_creator_fn, fn, *args,
 726                  **kwargs):
 727       super(MirroredStrategy._MirroredTowerThread, self).__init__()  # pylint: disable=protected-access
 728       self.coord = coord
 729       self.distribution = dist
 730       self.device = device
 731       self.tower_id = dist.worker_devices.index(device)
 732       self.variable_creator_fn = variable_creator_fn
 733       # State needed to run and return the results of `fn`.
 734       self.main_fn = fn
 735       self.main_args = args
 736       self.main_kwargs = kwargs
 737       self.main_result = None
 738       self.done = False
 739       # State needed to run the next merge_call() (if any) requested via
 740       # TowerContext.
 741       self.merge_fn = None
 742       self.merge_args = None
 743       self.merge_kwargs = None
 744       self.merge_result = None
 745       self.captured_name_scope = None
 746       # We use a thread.Event for the main thread to signal when this
 747       # thread should start running (`should_run`), and another for
 748       # this thread to transfer control back to the main thread
 749       # (`has_paused`, either when it gets to a
 750       # `get_tower_context().merge_call` or when `fn` returns). In
 751       # either case the event starts cleared, is signaled by calling
 752       # set(). The receiving thread waits for the signal by calling
 753       # wait() and then immediately clearing the event using clear().
 754       self.should_run = threading.Event()
 755       self.has_paused = threading.Event()
 756       # These fields have to do with inheriting various contexts from the
 757       # parent thread:
 758       # pylint: disable=protected-access
 759       self.context_mode = context.context()._eager_context.mode
 760       if not context.context()._context_handle:
 761         context.context()._initialize_handle_and_devices()
 762       self.context_device_policy = (
 763           pywrap_tensorflow.TFE_ContextGetDevicePlacementPolicy(
 764               context.context()._context_handle))
 765       self.graph = ops.get_default_graph()
 766       self._variable_creator_stack = self.graph._variable_creator_stack[:]
 767       self._captured_var_scope = variable_scope.get_variable_scope()
 768       # Adding a "/" at end lets us re-enter this scope later.
 769       self._name_scope = self.graph.get_name_scope()
 770       if self._name_scope:
 771         self._name_scope += "/"
 772       if self.tower_id > 0:
 773         if not self._name_scope:
 774           self._name_scope = ""
 775         self._name_scope += "tower_%d/" % self.tower_id
 776 
 777     def run(self):
 778       # pylint: disable=protected-access
 779       self.graph._variable_creator_stack = self._variable_creator_stack
 780       self.should_run.wait()
 781       self.should_run.clear()
 782       try:
 783         if self.coord.should_stop():
 784           return
 785         with self.coord.stop_on_exception(), \
 786             context.context()._mode(self.context_mode), \
 787             context.context().device_policy(self.context_device_policy), \
 788             _enter_graph(self.graph), \
 789             MirroredTowerContext(self.distribution, self.tower_id), \
 790             ops.device(self.device), \
 791             ops.name_scope(self._name_scope), \
 792             variable_scope.variable_scope(
 793                 self._captured_var_scope, reuse=self.tower_id > 0), \
 794             variable_scope.variable_creator_scope(self.variable_creator_fn):
 795           self.main_result = self.main_fn(*self.main_args, **self.main_kwargs)
 796           self.done = True
 797       finally:
 798         self.has_paused.set()

 801 class MirroredTowerContext(distribute_lib.TowerContext):
 802   """TowerContext used in MirroredStrategy.call_for_each_tower().
 803 
 804   Opened in `_MirroredTowerThread`, to allow the user to invoke
 805   `MirroredStrategy`'s specific implementation of `merge_call()`,
 806   which works by delegating the function and its arguments to
 807   the main thread (the one that invoked
 808   `MirroredStrategy.call_for_each_tower()`).
 809   """
 810 
 811   def _merge_call(self, fn, *args, **kwargs):
 812     """Delegate to the main thread to actually perform merge_call()."""
 813     t = threading.current_thread()  # a _MirroredTowerThread
 814     t.merge_fn = fn
 815     t.merge_args = args
 816     t.merge_kwargs = kwargs
 817     t.captured_name_scope = t.graph.get_name_scope()
 818     # Adding a "/" at end lets us re-enter this scope later.
 819     if t.captured_name_scope:
 820       t.captured_name_scope += "/"
 821     t.has_paused.set()
 822     t.should_run.wait()
 823     t.should_run.clear()
 824     if t.coord.should_stop():
 825       raise _RequestedStop()
 826     return t.merge_result
 827 
 828   @property
 829   def device(self):
 830     distribute_lib.require_tower_context(self)
 831     return self._distribution_strategy.worker_devices[self._tower_id]
