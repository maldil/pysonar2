  44 class ParameterServerStrategy(distribute_lib.DistributionStrategy):
  45   """A parameter server DistributionStrategy.
  46 
  47   This strategy class works for both local training and between-graph replicated
  48   training for multiple workers. If `cluster_spec` is specified, either passed
  49   in to __init__() method or parsed from the
  50   ["TF_CONFIG" environment
  51   variable](https://www.tensorflow.org/api_docs/python/tf/estimator/RunConfig),
  52   variables and updates to those variables are assigned to parameter servers and
  53   other operations are assigned to workers. If `cluster_spec` is not set, it
  54   becomes local training where variables are assigned to local CPU or the only
  55   GPU. When each worker has more than one GPU, operations will be replicated on
  56   these GPUs. In both cases, operations are replicated but variables are not and
  57   these workers share a common view for which paramater server a variable is
  58   assigned to.
  59 
  60   This class assumes between-graph replication will be used and works on a graph
  61   for a particular worker. Note that each graph and worker is independent.
  62   This means that while each worker will synchronously compute a single gradient
  63   update across all GPUs, updates between workers proceed asynchronously.
  64   Operations that occur only on the first tower (such as incrementing the global
  65   step), will occur on the first tower *of every worker*.
  66 
  67   It is expected to call `call_for_each_tower(fn, *args, **kwargs)` for any
  68   operations which potentially can be replicated across towers (i.e. multiple
  69   GPUs) even if there is only CPU or one GPU. When defining the `fn`, extra
  70   caution needs to be taken:
  71 
  72   1) Always use `tf.get_variable` instead of `tf.Variable` which is not able
  73   to refer to the same variable on different towers.
  74 
  75   2) It is generally not recommended to open a device scope under the strategy's
  76   scope. A device scope (i.e. calling `tf.device`) will be merged with or
  77   override the device for operations but will not change the device for
  78   variables.
  79 
  80   3) It is also not recommended to open a colocation scope (i.e. calling
  81   `tf.colocate_with`) under the strategy's scope. For colocating variables,
  82   use `distribution.colocate_vars_with` instead. Colocation of ops will possibly
  83   create conflicts of device assignment.
  84   """
  85 
  86   def __init__(self, num_gpus_per_worker=0):
  87     """Initializes this strategy.
  88 
  89     Args:
  90       num_gpus_per_worker: number of local GPUs or GPUs per worker, the default
  91         is 0 meaning CPU only.
  92 
  93     Raises:
  94       ValueError: if `cluster_spec` is given but `task_type` or `task_id` is
  95         not.
  96     """
  97     super(ParameterServerStrategy, self).__init__()
  98     self._num_gpus_per_worker = num_gpus_per_worker
  99     self._initialize_local(num_gpus_per_worker)
 100 
 101     # We typically don't need to do all-reduce in this strategy.
 102     self._cross_tower_ops = (
 103         cross_tower_ops_lib.ReductionToOneDeviceCrossTowerOps(
 104             reduce_to_device=_LOCAL_CPU))
 105 
 106   def _initialize_multi_worker(self, num_gpus_per_worker, cluster_spec,
 107                                task_type, task_id):
 108     """Initialize devices for multiple workers.
 109 
 110     It creates variable devices and compute devices. Variables and operations
 111     will be assigned to them respectively. We have one compute device per tower.
 112     The variable device is a device function or device string. The default
 113     variable device assigns variables to parameter servers in a round-robin
 114     fashion.
 115 
 116     Args:
 117       num_gpus_per_worker: number of local GPUs or GPUs per worker.
 118       cluster_spec: a dict, ClusterDef or ClusterSpec object specifying the
 119         cluster configurations.
 120       task_type: the current task type.
 121       task_id: the current task id.
 122 
 123     Raises:
 124       ValueError: if the cluster_spec doesn't have ps jobs.
 125     """
 126     assert cluster_spec
 127     if not task_type or task_id is None:
 128       raise ValueError("When `cluster_spec` is given, you must also specify "
 129                        "`task_type` and `task_id`")
 130     cluster_spec = multi_worker_util.normalize_cluster_spec(cluster_spec)
 131 
 132     self._worker_device = "/job:%s/task:%d" % (self._task_type, self._task_id)
 133 
 134     # Define compute devices which is a list of device strings and one for each
 135     # tower. When there are GPUs, replicate operations on these GPUs. Otherwise,
 136     # place operations on CPU.
 137     if num_gpus_per_worker > 0:
 138       self._compute_devices = [
 139           "%s/device:GPU:%d" % (self._worker_device, i)
 140           for i in range(num_gpus_per_worker)
 141       ]
 142     else:
 143       self._compute_devices = [self._worker_device]
 144 
 145     self._compute_devices = list(
 146         map(device_util.resolve, self._compute_devices))
 147     self._canonical_compute_device_set = set(self._compute_devices)
 148 
 149     # In distributed mode, place variables on ps jobs in a round-robin fashion.
 150     # Note that devices returned from `replica_device_setter` are not
 151     # canonical and therefore we don't canonicalize all variable devices to
 152     # make them consistent.
 153     # TODO(yuefengz): support passing a strategy object to control variable
 154     # assignment.
 155     # TODO(yuefengz): merge the logic of replica_device_setter into this
 156     # class.
 157     num_ps_replicas = len(cluster_spec.as_dict().get("ps", []))
 158     if num_ps_replicas == 0:
 159       raise ValueError("The cluster spec needs to have `ps` jobs.")
 160     self._variable_device = device_setter.replica_device_setter(
 161         ps_tasks=num_ps_replicas,
 162         worker_device=self._worker_device,
 163         merge_devices=True,
 164         cluster=cluster_spec)
 165 
 166     # The `_parameter_devices` is needed for the `parameter_devices` property
 167     # and is a list of all variable devices. Here parameter devices are all
 168     # tasks of the "ps" job.
 169     self._parameter_devices = map("/job:ps/task:{}".format,
 170                                   range(num_ps_replicas))
 171 
 172     # Add a default device so that ops without specified devices will not end up
 173     # on other workers.
 174     self._default_device = self._worker_device
 175 
 176     self._is_chief = multi_worker_util.is_chief(cluster_spec, task_type,
 177                                                 task_id)
 178     self._cluster_spec = cluster_spec
 179     self._task_type = task_type
 180     self._task_id = task_id
 181 
 182     logging.info(
 183         "Multi-worker ParameterServerStrategy with "
 184         "cluster_spec = %r, task_type = %r, task_id = %r, "
 185         "num_ps_replicas = %r, is_chief = %r, compute_devices = %r, "
 186         "variable_device = %r", cluster_spec.as_dict(), task_type, task_id,
 187         num_ps_replicas, self._is_chief, self._compute_devices,
 188         self._variable_device)
 189 
 190   def _initialize_local(self, num_gpus_per_worker):
 191     """Initialize internal devices for local training."""
 192     # Define compute devices which is a list of device strings and one for each
 193     # tower. When there are GPUs, replicate operations on these GPUs. Otherwise,
 194     # place operations on CPU.
 195     if num_gpus_per_worker > 0:
 196       self._compute_devices = list(
 197           map("/device:GPU:{}".format, range(num_gpus_per_worker)))
 198     else:
 199       self._compute_devices = [_LOCAL_CPU]
 200 
 201     self._compute_devices = list(
 202         map(device_util.resolve, self._compute_devices))
 203     self._canonical_compute_device_set = set(self._compute_devices)
 204 
 205     # If there is only one GPU, put everything on that GPU. Otherwise, place
 206     # variables on CPU.
 207     if num_gpus_per_worker == 1:
 208       assert len(list(self._compute_devices)) == 1
 209       self._variable_device = _LOCAL_GPU_0
 210       self._parameter_devices = [_LOCAL_GPU_0]
 211     else:
 212       self._variable_device = _LOCAL_CPU
 213       self._parameter_devices = [_LOCAL_CPU]
 214 
 215     self._is_chief = True
 216     self._cluster_spec = None
 217     self._task_type = None
 218     self._task_id = None
 219 
 220     logging.info(
 221         "ParameterServerStrategy with compute_devices = %r, "
 222         "variable_device = %r", self._compute_devices, self._variable_device)
 223 
 224   def distribute_dataset(self, dataset_fn):
 225     """Distributes the dataset to each local GPU."""
 226     return values.PerDeviceDataset(
 227         self._call_dataset_fn(dataset_fn), self._compute_devices, True)

 229   def _broadcast(self, tensor, destinations):
 230     if not cross_tower_ops_lib.check_destinations(destinations):
 231       destinations = self._compute_devices
 232     return self._cross_tower_ops.broadcast(tensor, destinations)

 234   # TODO(yuefengz): not all ops in device_setter.STANDARD_PS_OPS will go through
 235   # this creator, such as "MutableHashTable".
 236   def _create_variable(self, next_creator, *args, **kwargs):
 237     if self.num_towers > 1:
 238       aggregation = kwargs.pop("aggregation", vs.VariableAggregation.NONE)
 239       if aggregation not in (
 240           vs.VariableAggregation.NONE,
 241           vs.VariableAggregation.SUM,
 242           vs.VariableAggregation.MEAN,
 243           vs.VariableAggregation.ONLY_FIRST_TOWER
 244       ):
 245         raise ValueError("Invalid variable aggregation mode: " + aggregation +
 246                          " for variable: " + kwargs["name"])
 247 
 248       def var_creator(*args, **kwargs):
 249         # Record what collections this variable should be added to.
 250         collections = kwargs.pop("collections", None)
 251         if collections is None:
 252           collections = [ops.GraphKeys.GLOBAL_VARIABLES]
 253         kwargs["collections"] = []
 254 
 255         # Create and wrap the variable.
 256         v = next_creator(*args, **kwargs)
 257         wrapped = values.AggregatingVariable(v, aggregation)
 258 
 259         # Add the wrapped variable to the requested collections.
 260         # The handling of eager mode and the global step matches
 261         # ResourceVariable._init_from_args().
 262         if not context.executing_eagerly():
 263           g = ops.get_default_graph()
 264           # If "trainable" is True, next_creator() will add the contained
 265           # variable to the TRAINABLE_VARIABLES collection, so we manually
 266           # remove it and replace with the wrapper. We can't set "trainable"
 267           # to False for next_creator() since that causes functions like
 268           # implicit_gradients to skip those variables.
 269           if kwargs.get("trainable", True):
 270             collections.append(ops.GraphKeys.TRAINABLE_VARIABLES)
 271             l = g.get_collection_ref(ops.GraphKeys.TRAINABLE_VARIABLES)
 272             l.remove(v)
 273           g.add_to_collections(collections, wrapped)
 274         elif ops.GraphKeys.GLOBAL_STEP in collections:
 275           ops.add_to_collections(ops.GraphKeys.GLOBAL_STEP, wrapped)
 276 
 277         return wrapped
 278     else:
 279       var_creator = next_creator
 280 
 281     if "colocate_with" in kwargs:
 282       with ops.device(None):
 283         with ops.colocate_with(kwargs["colocate_with"]):
 284           return var_creator(*args, **kwargs)
 285 
 286     with ops.colocate_with(None, ignore_existing=True):
 287       with ops.device(self._variable_device):
 288         return var_creator(*args, **kwargs)
 289 
 290   def _call_for_each_tower(self, fn, *args, **kwargs):
 291     # pylint: disable=protected-access
 292     return mirrored_strategy._call_for_each_tower(self, fn, *args, **kwargs)
 293 
 294   def _verify_destinations_not_different_worker(self, destinations):
 295     if destinations is None:
 296       return
 297     for d in cross_tower_ops_lib.get_devices_from(destinations):
 298       d_spec = tf_device.DeviceSpec.from_string(d)
 299       if d_spec.job == self._task_type and d_spec.task != self._task_id:
 300         raise ValueError(
 301             "Cannot reduce to another worker: %r, current worker is %r" %
 302             (d, self._worker_device))
 303 
 304   def _reduce(self, aggregation, value, destinations):
 305     self._verify_destinations_not_different_worker(destinations)
 306     if not isinstance(value, values.DistributedValues):
 307       # pylint: disable=protected-access
 308       return mirrored_strategy._reduce_non_distributed_value(
 309           self, aggregation, value, destinations)
 310     if aggregation == vs.VariableAggregation.ONLY_FIRST_TOWER:
 311       return self.broadcast(value.get(self._compute_devices[0]), destinations)
 312     return self._cross_tower_ops.reduce(
 313         aggregation, value, destinations=destinations)
 314 
 315   def _batch_reduce(self, aggregation, value_destination_pairs):
 316     if aggregation == vs.VariableAggregation.ONLY_FIRST_TOWER:
 317       return [self.broadcast(v.get(self._compute_devices[0]), d)
 318               for v, d in value_destination_pairs]
 319     for _, destinations in value_destination_pairs:
 320       self._verify_destinations_not_different_worker(destinations)
 321     return self._cross_tower_ops.batch_reduce(aggregation,
 322                                               value_destination_pairs)

 324   def _select_single_value(self, structured):
 325     """Select any single values in `structured`."""
 326 
 327     def _select_fn(x):  # pylint: disable=g-missing-docstring
 328       if isinstance(x, values.Mirrored):
 329         if len(x.devices) == 1:
 330           return list(x._index.values())[0]  # pylint: disable=protected-access
 331         else:
 332           raise ValueError(
 333               "You cannot update variable with a Mirrored object with multiple "
 334               "components %r when using ParameterServerStrategy. You must "
 335               "specify a single value or a Mirrored with a single value." % x)
 336       elif isinstance(x, values.PerDevice):
 337         raise ValueError(
 338             "You cannot update variable with a PerDevice object %r when using "
 339             "ParameterServerStrategy. You must specify a single value or a "
 340             "Mirrored with a single value" % x)
 341       else:
 342         return x
 343 
 344     return nest.map_structure(_select_fn, structured)
 345 
 346   def _update(self, var, options, fn, *args, **kwargs):
 347     if isinstance(var, values.AggregatingVariable):
 348       var = var.get()
 349     if not isinstance(var, resource_variable_ops.ResourceVariable):
 350       raise ValueError(
 351           "You can not update `var` %r. It must be a Variable." % var)
 352     should_group = options.pop("grouped")
 353     assert not options  # Validate that we are processing all of the options.
 354     with ops.colocate_with(var), distribute_lib.UpdateContext(var.device):
 355       result = fn(var, *self._select_single_value(args),
 356                   **self._select_single_value(kwargs))
 357       if should_group:
 358         return result
 359       else:
 360         return nest.map_structure(self._unwrap, result)
 361 
 362   # TODO(yuefengz): does it need to call _select_single_value?
 363   def _update_non_slot(self, colocate_with, options, fn, *args, **kwargs):
 364     should_group = options.pop("grouped")
 365     assert not options  # Validate that we are processing all of the options.
 366     with ops.device(
 367         colocate_with.device), distribute_lib.UpdateContext(colocate_with):
 368       result = fn(*args, **kwargs)
 369       if should_group:
 370         return result
 371       else:
 372         return nest.map_structure(self._unwrap, result)

 374   def _unwrap(self, val):
 375     if isinstance(val, values.DistributedValues):
 376       # Return in a deterministic order.
 377       if set(val.devices) == self._canonical_compute_device_set:
 378         return [val.get(device=d) for d in self._compute_devices]
 379       return [val.get(device=d) for d in sorted(val.devices)]
 380     return [val]

 385   def read_var(self, var):
 386     # No need to distinguish between normal variables and tower-local variables.
 387     return array_ops.identity(var)

 389   def configure(self,
 390                 session_config=None,
 391                 cluster_spec=None,
 392                 task_type=None,
 393                 task_id=None):
 394     """Configures the strategy class.
 395 
 396     The strategy object will be re-initialized if `cluster_spec` is given but
 397     was not passed in the constructor.
 398 
 399     Args:
 400       session_config: not used currently.
 401       cluster_spec: a dict, ClusterDef or ClusterSpec object specifying the
 402         cluster configurations.
 403       task_type: the current task type.
 404       task_id: the current task id.
 405 
 406     Raises:
 407       ValueError: if `cluster_spec` is given but `task_type` or `task_id` is
 408         not.
 409     """
 410     if not self._cluster_spec and cluster_spec:
 411       # If a `cluster_spec` is already passed in, do nothing here.
 412       # TODO(yuefengz): check `cluster_spec` is the same if this object has
 413       # already been initialized with a `cluster_spec`.
 414       if task_type is None or task_id is None:
 415         raise ValueError("When `cluster_spec` is given, must also specify "
 416                          "`task_type` and `task_id`.")
 417       self._cluster_spec = multi_worker_util.normalize_cluster_spec(
 418           cluster_spec)
 419       self._task_type = task_type
 420       self._task_id = task_id
 421       self._initialize_multi_worker(self._num_gpus_per_worker,
 422                                     self._cluster_spec, task_type, task_id)
 423 
 424     if not session_config or not self._cluster_spec:
 425       return
 426 
 427     session_config.isolate_session_state = False
 428 
 429     assert self._cluster_spec
 430     assert self._task_type
 431     assert self._task_id is not None
 432 
 433     # The device filters prevent communication between workers.
 434     if self._task_type not in ["chief", "worker"]:
 435       return
 436     del session_config.device_filters[:]
 437     session_config.device_filters.extend(
 438         ["/job:%s/task:%d" % (self._task_type, self._task_id), "/job:ps"])
