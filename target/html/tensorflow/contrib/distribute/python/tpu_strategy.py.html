  48 def get_tpu_system_metadata(tpu_cluster_resolver):
  49   """Retrieves TPU system metadata given a TPUClusterResolver."""
  50   master = tpu_cluster_resolver.master()
  51 
  52   # pylint: disable=protected-access
  53   cluster_spec = tpu_cluster_resolver.cluster_spec()
  54   cluster_def = cluster_spec.as_cluster_def() if cluster_spec else None
  55   tpu_system_metadata = (
  56       tpu_system_metadata_lib._query_tpu_system_metadata(
  57           master,
  58           cluster_def=cluster_def,
  59           query_topology=False))
  60 
  61   return tpu_system_metadata

  65 def _create_tpu_mirrored_variable(devices, real_mirrored_creator, *args,
  66                                   **kwargs):  # pylint: disable=g-missing-docstring
  67   # Figure out what collections this variable should be added to.
  68   # We'll add the TPUMirroredVariable to those collections instead.
  69   collections = kwargs.pop("collections", None)
  70   if collections is None:
  71     collections = [ops.GraphKeys.GLOBAL_VARIABLES]
  72   kwargs["collections"] = []
  73 
  74   # TODO(jhseu): Should we have different behavior for different
  75   # synchronization settings?
  76 
  77   # Get aggregation value
  78   # TODO(jhseu): Support aggregation in a tower context.
  79   aggregation = kwargs.pop("aggregation", vs.VariableAggregation.NONE)
  80   if aggregation not in [
  81       vs.VariableAggregation.NONE,
  82       vs.VariableAggregation.SUM,
  83       vs.VariableAggregation.MEAN,
  84       vs.VariableAggregation.ONLY_FIRST_TOWER,
  85   ]:
  86     raise ValueError("Invalid variable aggregation mode: {} for variable: {}"
  87                      .format(aggregation, kwargs["name"]))
  88 
  89   # Ignore user-specified caching device, not needed for mirrored variables.
  90   kwargs.pop("caching_device", None)
  91 
  92   # TODO(josh11b,apassos): It would be better if variable initialization
  93   # was never recorded on the tape instead of having to do this manually
  94   # here.
  95   with tape.stop_recording():
  96     index = real_mirrored_creator(devices, *args, **kwargs)
  97     result = values.TPUMirroredVariable(index, index[devices[0]], aggregation)
  98 
  99   if not context.executing_eagerly():
 100     g = ops.get_default_graph()
 101     # If "trainable" is True, next_creator() will add the member variables
 102     # to the TRAINABLE_VARIABLES collection, so we manually remove
 103     # them and replace with the MirroredVariable. We can't set
 104     # "trainable" to False for next_creator() since that causes functions
 105     # like implicit_gradients to skip those variables.
 106     if kwargs.get("trainable", True):
 107       collections.append(ops.GraphKeys.TRAINABLE_VARIABLES)
 108       l = g.get_collection_ref(ops.GraphKeys.TRAINABLE_VARIABLES)
 109       for v in index.values():
 110         l.remove(v)
 111     g.add_to_collections(collections, result)
 112   return result

 119   def __init__(self, tpu_cluster_resolver, steps_per_run, num_cores=None):
 120     """Initializes the TPUStrategy object.
 121 
 122     Args:
 123       tpu_cluster_resolver: A tf.contrib.cluster_resolver.TPUClusterResolver,
 124           which provides information about the TPU cluster.
 125       steps_per_run: Number of steps to run on device before returning to the
 126           host. Note that this can have side-effects on performance, hooks,
 127           metrics, summaries etc.
 128           This parameter is only used when Distribution Strategy is used with
 129           estimator or keras.
 130       num_cores: Number of cores to use on the TPU. If None specified, then
 131           auto-detect the cores and topology of the TPU system.
 132     """
 133     # TODO(sourabhbajaj): OneDeviceStrategy should be initialized with the
 134     # master node fetched from the cluster resolver.
 135     super(TPUStrategy, self).__init__('/device:CPU:0')
 136 
 137     self._tpu_cluster_resolver = tpu_cluster_resolver
 138     self._tpu_metadata = get_tpu_system_metadata(self._tpu_cluster_resolver)
 139     # TODO(sourabhbajaj): Change this from num_cores to metadata_override
 140     self._num_cores_override = num_cores
 141 
 142     # TODO(jhseu): Switch to DeviceAssignment to support pods and model
 143     # parallelism.
 144     device_map = {d.name: i for i, d in enumerate(self._tpu_metadata.devices)
 145                   if "device:TPU:" in d.name}
 146     self._device_index = values.PerDevice(device_map)
 147     self._tpu_devices = sorted(device_map.keys())
 148     # Only create variables for the number of towers we're running.
 149     self._tpu_devices = self._tpu_devices[:self.num_towers]

 151     # TODO(sourabhbajaj): Remove this once performance of running one step
 152     # at a time is comparable to multiple steps.

 155   def _get_enqueue_op_per_host(self, host_id, iterator, input_shapes,
 156                                iterations):
 157     """Create an enqueue op for a single host identified using host_id.
 158 
 159     The while_loop op returned will run `iterations` times and in each run
 160     enqueue batches for each shard.
 161 
 162     Args:
 163       host_id: integer, id of the host to run the enqueue ops on.
 164       iterator: `tf.data` iterator to read the input data.
 165       input_shapes: shape of inputs to be enqueue on the queue. This is same as
 166         the value of `nest.flatten(iterator.output_shapes)`.
 167       iterations: integer, number of iterations to be run; determines the
 168         number of batches to be enqueued.
 169 
 170     Returns:
 171       while_loop_op running `iterations` times; in each run we enqueue a batch
 172       on the infeed queue from the host with id `host_id` for each device shard.
 173     """
 174     host = self.get_host_cpu_device(host_id)
 175 
 176     def _infeed_enqueue_ops_fn():
 177       """Enqueue ops for one iteration."""
 178       control_deps = []
 179       sharded_inputs = []
 180       enqueue_ops = []
 181 
 182       with ops.device(host):
 183         for _ in range(self.num_towers_per_host):
 184           # Use control dependencies to ensure a deterministic ordering.
 185           with ops.control_dependencies(control_deps):
 186             inputs = nest.flatten(iterator.get_next())
 187             control_deps.extend(inputs)
 188             sharded_inputs.append(inputs)
 189 
 190       for core_id, shard_input in enumerate(sharded_inputs):
 191         enqueue_ops.append(
 192             tpu_ops.infeed_enqueue_tuple(
 193                 inputs=shard_input,
 194                 shapes=input_shapes,
 195                 device_ordinal=core_id))
 196       return enqueue_ops
 197 
 198     def enqueue_ops_loop_body(i):
 199       """Callable for the loop body of the while_loop instantiated below."""
 200       with ops.control_dependencies(_infeed_enqueue_ops_fn()):
 201         return i + 1
 202 
 203     with ops.device(host):
 204       enqueue_op_per_host = control_flow_ops.while_loop(
 205           lambda i: i < iterations,
 206           enqueue_ops_loop_body,
 207           [constant_op.constant(0)],
 208           parallel_iterations=1)
 209 
 210     return enqueue_op_per_host

 212   def distribute_dataset(self, dataset_fn):
 213     # TODO(priyag): Perhaps distribute across cores here.
 214     return self._call_dataset_fn(dataset_fn)

 216   # TODO(priyag): Deal with OutOfRange errors once b/111349762 is fixed.
 217   # TODO(sourabhbajaj): Remove the initial_loop_values parameter when we have
 218   # a mechanism to infer the outputs of `fn`. Pending b/110550782.
 219   def _run_steps_on_dataset(self, fn, iterator, iterations,
 220                             initial_loop_values=None):
 221 
 222     shapes = nest.flatten(iterator.output_shapes)
 223     if any([not s.is_fully_defined() for s in shapes]):
 224       raise ValueError(
 225           'TPU currently requires fully defined shapes. Either use '
 226           'set_shape() on the input tensors or use '
 227           'dataset.batch(..., drop_remainder=True).')
 228     types = nest.flatten(iterator.output_types)
 229 
 230     enqueue_ops = [
 231         self._get_enqueue_op_per_host(host_id, iterator, shapes, iterations)
 232         for host_id in range(self.num_hosts)]
 233 
 234     def dequeue_fn():
 235       dequeued = tpu_ops.infeed_dequeue_tuple(dtypes=types, shapes=shapes)
 236       return nest.pack_sequence_as(iterator.output_shapes, dequeued)
 237 
 238     # Wrap `fn` for repeat.
 239     if initial_loop_values is None:
 240       initial_loop_values = {}
 241     initial_loop_values = nest.flatten(initial_loop_values)
 242     ctx = values.MultiStepContext()
 243     def run_fn(*args, **kwargs):
 244       """Single step on the TPU device."""
 245       del args, kwargs
 246       fn_inputs = dequeue_fn()
 247       if not isinstance(fn_inputs, tuple):
 248         fn_inputs = (fn_inputs,)
 249       fn_result = fn(ctx, *fn_inputs)
 250       flat_last_step_outputs = nest.flatten(ctx.last_step_outputs)
 251       if flat_last_step_outputs:
 252         with ops.control_dependencies([fn_result]):
 253           return [array_ops.identity(f) for f in flat_last_step_outputs]
 254       else:
 255         return fn_result
 256 
 257     # TODO(sourabhbajaj): The input to while loop should be based on the output
 258     # type of the step_fn
 259     def iterate_on_tpu():
 260       return training_loop.repeat(iterations, run_fn, initial_loop_values)
 261 
 262     # We capture the control_flow_context at this point, before we run `fn`
 263     # inside a while_loop and TPU replicate context. This is useful in cases
 264     # where we might need to exit these contexts and get back to the outer
 265     # context to do some things, for e.g. create an op which should be
 266     # evaluated only once at the end of the loop on the host. One such usage
 267     # is in creating metrics' value op.
 268     self._outer_control_flow_context = (
 269         ops.get_default_graph()._get_control_flow_context())  # pylint: disable=protected-access
 270 
 271     replicate_inputs = [[]] * self.num_towers
 272     replicate_outputs = tpu.replicate(iterate_on_tpu, replicate_inputs)
 273     del self._outer_control_flow_context
 274     ctx.run_op = control_flow_ops.group(replicate_outputs, enqueue_ops)
 275 
 276     # Filter out any ops from the outputs, typically this would be the case
 277     # when there were no tensor outputs.
 278     last_step_tensor_outputs = [x for x in replicate_outputs
 279                                 if not isinstance(x, ops.Operation)]
 280 
 281     # Outputs are currently of the structure (grouped by device)
 282     # [[output0_device0, output1_device0, output2_device0],
 283     #  [output0_device1, output1_device1, output2_device1]]
 284     # Convert this to the following structure instead: (grouped by output)
 285     # [[output0_device0, output0_device1],
 286     #  [output1_device0, output1_device1],
 287     #  [output2_device0, output2_device1]]
 288     last_step_tensor_outputs = [list(x) for x in zip(*last_step_tensor_outputs)]
 289 
 290     # Convert replicate_outputs to the original dict structure of
 291     # last_step_outputs.
 292     last_step_tensor_outputs_dict = nest.pack_sequence_as(
 293         ctx.last_step_outputs, last_step_tensor_outputs)
 294 
 295     for (name, aggregation) in ctx._last_step_outputs_aggregations.items():  # pylint: disable=protected-access
 296       output = last_step_tensor_outputs_dict[name]
 297       # For outputs that have already been aggregated, take the first value
 298       # from the list as each value should be the same. Else return the full
 299       # list of values.
 300       # TODO(josh11b): If aggregation is NONE, we should return a PerDevice value.
 301       if aggregation is not variables_lib.VariableAggregation.NONE:
 302         # TODO(priyag): Should this return the element or a list with 1 element
 303         last_step_tensor_outputs_dict[name] = output[0]
 304     ctx._set_last_step_outputs(last_step_tensor_outputs_dict)  # pylint: disable=protected-access
 305 
 306     return ctx

 308   def _call_for_each_tower(self, fn, *args, **kwargs):
 309     # TODO(jhseu): Consider making it so call_for_each_tower implies that we're
 310     # in a tpu.rewrite(), and update TPUMirroredVariable accordingly.
 311     kwargs.pop('run_concurrently', None)
 312     with one_device_strategy._OneDeviceTowerContext(self):  # pylint: disable=protected-access
 313       return fn(*args, **kwargs)

 315   def initialize(self):
 316     if context.executing_eagerly():
 317       # TODO(priyag): Add appopriate call here when eager is supported for TPUs.
 318       raise NotImplementedError('Eager mode not supported in TPUStrategy.')
 319     else:
 320       # TODO(jhseu): We need this hack because DistributionStrategies must be
 321       # pickleable for copy.deepcopy(). Remove when initialize_system goes away.
 322       graph = ops.get_default_graph()
 323       tpu_init = graph.get_collection(_TPU_INITIALIZE_SYSTEM_COLLECTION)
 324       if tpu_init:
 325         return tpu_init
 326       graph.add_to_collection(_TPU_INITIALIZE_SYSTEM_COLLECTION,
 327                               tpu.initialize_system())
 328       return graph.get_collection(_TPU_INITIALIZE_SYSTEM_COLLECTION)
 329 
 330   def finalize(self):
 331     if context.executing_eagerly():
 332       # TODO(priyag): Add appopriate call here when eager is supported for TPUs.
 333       raise NotImplementedError('Eager mode not supported in TPUStrategy.')
 334     else:
 335       return [tpu.shutdown_system()]

 341   def _create_variable(self, next_creator, *args, **kwargs):
 342     """Create a TPUMirroredVariable. See `DistributionStrategy.scope`."""
 343     colocate_with = kwargs.pop("colocate_with", None)
 344     devices = self._get_devices_from(colocate_with)
 345 
 346     def _real_mirrored_creator(devices, *args, **kwargs):  # pylint: disable=g-missing-docstring
 347       index = {}
 348       for i, d in enumerate(devices):
 349         with ops.device(d):
 350           if i > 0:
 351             # Give replicas meaningful distinct names:
 352             var0name = index[devices[0]].name.split(":")[0]
 353             # We append a / to variable names created on towers with id > 0 to
 354             # ensure that we ignore the name scope and instead use the given
 355             # name as the absolute name of the variable.
 356             kwargs["name"] = "%s/replica_%d/" % (var0name, i)
 357             # Initialize replicas with the same value:
 358             if context.executing_eagerly():
 359               kwargs["initial_value"] = array_ops.identity(
 360                   index[devices[0]].value())
 361             else:
 362               def initial_value_fn(device=d):
 363                 with ops.device(device):
 364                   return array_ops.identity(index[devices[0]].initial_value)
 365               kwargs["initial_value"] = initial_value_fn
 366           with context.context().device_policy(context.DEVICE_PLACEMENT_SILENT):
 367             v = next_creator(*args, **kwargs)
 368           assert not isinstance(v, values.TPUMirroredVariable)
 369           index[d] = v
 370       return index
 371 
 372     return _create_tpu_mirrored_variable(devices, _real_mirrored_creator, *args,
 373                                          **kwargs)

 375   def _reduce(self, aggregation, value, destinations):
 376     if values._enclosing_tpu_context() is not None:  # pylint: disable=protected-access
 377       if aggregation == vs.VariableAggregation.MEAN:
 378         # TODO(jhseu):  Revisit once we support model-parallelism.
 379         value *= (1. / self.num_towers)
 380       elif aggregation != vs.VariableAggregation.SUM:
 381         raise NotImplementedError(
 382             "Currently only support sum & mean in TPUStrategy.")
 383       return tpu_ops.cross_replica_sum(value)
 384 
 385     # Validate that the destination is same as the host device
 386     # Note we don't do this when in replicate context as the reduction is
 387     # performed on the TPU device itself.
 388     devices = cross_tower_ops_lib.get_devices_from(destinations)
 389     if len(devices) == 1:
 390       assert device_util.canonicalize(devices[0]) == device_util.canonicalize(
 391           self.get_host_cpu_device(0))
 392     else:
 393       raise ValueError('Multiple devices are not supported for TPUStrategy')
 394 
 395     if aggregation == vs.VariableAggregation.ONLY_FIRST_TOWER:
 396       return value[0]
 397     output = math_ops.add_n(value)
 398     if aggregation == vs.VariableAggregation.MEAN:
 399       return output * (1. / len(value))
 400     return output

 402   def _update(self, var, options, fn, *args, **kwargs):
 403     assert isinstance(var, values.TPUMirroredVariable)
 404     should_group = options.pop("grouped")
 405     assert not options  # Validate that we are processing all of the options.
 406 
 407     if values._enclosing_tpu_context() is not None:  # pylint: disable=protected-access
 408       if should_group:
 409         return fn(var, *args, **kwargs)
 410       else:
 411         return [fn(var, *args, **kwargs)]
 412 
 413     # Otherwise, we revert to MirroredStrategy behavior and update each variable
 414     # directly.
 415     updates = {}
 416     for d, v in var._index.items():  # pylint: disable=protected-access
 417       name = "update_%d" % self._device_index.get(d)
 418       with ops.device(d), distribute_lib.UpdateContext(d), ops.name_scope(name):
 419         # If args and kwargs are not mirrored, the value is returned as is.
 420         updates[d] = fn(v,
 421                         *values.select_device_mirrored(d, args),
 422                         **values.select_device_mirrored(d, kwargs))
 423     return values.update_regroup(self, updates, should_group)

 431   def _unwrap(self, val):
 432     if isinstance(val, values.DistributedValues):
 433       # Return in a deterministic order.
 434       return [val.get(device=d) for d in sorted(val.devices)]
 435     elif isinstance(val, list):
 436       # TODO(josh11b): We need to remove this case; per device values should
 437       # be represented using a PerDevice wrapper instead of a list with
 438       # one entry per device.
 439       return val
 440     return [val]

 479   def get_host_cpu_device(self, host_id):
 480     if self._tpu_cluster_resolver.get_master() in ('', 'local'):
 481       return '/replica:0/task:0/device:CPU:0'
 482     job_name = self._tpu_cluster_resolver.get_job_name() or 'tpu_worker'
 483     return '/job:%s/task:%d/device:CPU:0' % (job_name, host_id)

 485   def configure(self,
 486                 session_config=None,
 487                 cluster_spec=None,
 488                 task_type=None,
 489                 task_id=None):
 490     del cluster_spec, task_type, task_id
 491     if session_config:
 492       session_config.isolate_session_state = True
 493       cluster_spec = self._tpu_cluster_resolver.cluster_spec()
 494       if cluster_spec:
 495         session_config.cluster_def.CopyFrom(cluster_spec.as_cluster_def())
