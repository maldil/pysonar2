  36 class _PrefetchToDeviceIterator(object):
  37   """A replacement for `tf.data.Iterator` that prefetches to another device.
  38 
  39   Args:
  40     input_dataset: The input dataset.
  41     one_shot: If true, we make a one shot iterator that's already initialized.
  42     devices: Devices on which to prefetch.
  43     buffer_size: Size of the prefetching buffer.
  44     shared_name: (Optional.) If non-empty, the returned iterator will be shared
  45       under the given name across multiple sessions that share the same devices
  46       (e.g. when using a remote server). Only used if one_shot is False.
  47 
  48   Returns:
  49     An Iterator type object.
  50   """
  51 
  52   def __init__(self,
  53                input_dataset,
  54                one_shot,
  55                devices,
  56                buffer_size,
  57                shared_name=None):
  58     self._input_dataset = input_dataset
  59     self._get_next_call_count = 0
  60     self._one_shot = one_shot
  61     if shared_name is None:
  62       shared_name = ""
  63     self._devices = devices
  64 
  65     if self._one_shot:
  66       self._input_iterator = input_dataset.make_one_shot_iterator()
  67     else:
  68       self._input_iterator = iterator_ops.Iterator.from_structure(
  69           self._input_dataset.output_types, self._input_dataset.output_shapes,
  70           shared_name, self._input_dataset.output_classes)
  71     input_iterator_handle = self._input_iterator.string_handle()
  72 
  73     @function.Defun(dtypes.string)
  74     def _prefetch_fn(handle):
  75       """Prefetches one element from `input_iterator`."""
  76       remote_iterator = iterator_ops.Iterator.from_string_handle(
  77           handle, self._input_iterator.output_types,
  78           self._input_iterator.output_shapes,
  79           self._input_iterator.output_classes)
  80       ret = remote_iterator.get_next()
  81       return nest.flatten(sparse.serialize_sparse_tensors(ret))
  82 
  83     target_device = ged_ops.experimental_iterator_get_device(
  84         self._input_iterator._iterator_resource)
  85     self._buffering_resources = []
  86     for device in nest.flatten(self._devices):
  87       with ops.device(device):
  88         buffer_resource_handle = prefetching_ops.function_buffering_resource(
  89             f=_prefetch_fn,
  90             output_types=data_nest.flatten(
  91                 sparse.as_dense_types(self._input_dataset.output_types,
  92                                       self._input_dataset.output_classes)),
  93             target_device=target_device,
  94             string_arg=input_iterator_handle,
  95             buffer_size=buffer_size,
  96             shared_name=shared_name)
  97         self._buffering_resources.append(buffer_resource_handle)
  98 
  99     if not self._one_shot:
 100       reset_ops = []
 101       for buffer_resource in self._buffering_resources:
 102         reset_ops.append(
 103             ged_ops.experimental_function_buffering_resource_reset(
 104                 buffer_resource))
 105       with ops.control_dependencies(reset_ops):
 106         self._initializer = self._input_iterator.make_initializer(
 107             self._input_dataset)
 108 
 109   def get_next(self, name=None):
 110     """See `tf.data.Iterator.get_next`."""
 111     self._get_next_call_count += 1
 112     if self._get_next_call_count > iterator_ops.GET_NEXT_CALL_WARNING_THRESHOLD:
 113       warnings.warn(iterator_ops.GET_NEXT_CALL_WARNING_MESSAGE)
 114 
 115     flat_result = []
 116     # TODO(priyag): This will fail if the input size (typically number of
 117     # batches) is not divisible by number of devices.
 118     # How do we handle that more gracefully / let the user know?
 119     for buffer_resource in self._buffering_resources:
 120       flat_ret = ged_ops.experimental_function_buffering_resource_get_next(
 121           buffer_resource,
 122           output_types=data_nest.flatten(
 123               sparse.as_dense_types(self.output_types, self.output_classes)),
 124           name=name)
 125 
 126       ret = sparse.deserialize_sparse_tensors(
 127           data_nest.pack_sequence_as(self.output_types, flat_ret),
 128           self.output_types, self.output_shapes, self.output_classes)
 129 
 130       for tensor, shape in zip(
 131           data_nest.flatten(ret), data_nest.flatten(self.output_shapes)):
 132         if isinstance(tensor, ops.Tensor):
 133           tensor.set_shape(shape)
 134       flat_result.append(ret)
 135 
 136     return nest.pack_sequence_as(self._devices, flat_result)
 137 
 138   @property
 139   def initializer(self):
 140     if self._one_shot:
 141       raise NotImplementedError("Can't initialize a one_shot_iterator")
 142     return self._initializer
 143 
 144   @property
 145   def output_classes(self):
 146     return self._input_dataset.output_classes
 147 
 148   @property
 149   def output_shapes(self):
 150     return self._input_dataset.output_shapes
 151 
 152   @property
 153   def output_types(self):
 154     return self._input_dataset.output_types

 160 class _PrefetchToDeviceDataset(dataset_ops.UnaryDataset):
 161   """A `Dataset` whose iterator prefetches elements to other device(s)."""
 162 
 163   def __init__(self, input_dataset, devices, buffer_size):
 164     super(_PrefetchToDeviceDataset, self).__init__(input_dataset)
 165     self._input_dataset = input_dataset
 166     self._devices = devices
 167     self._buffer_size = buffer_size if buffer_size is not None else 1
 168 
 169   def make_one_shot_iterator(self):
 170     return _PrefetchToDeviceIterator(
 171         self._input_dataset,
 172         one_shot=True,
 173         devices=self._devices,
 174         buffer_size=self._buffer_size)
 175 
 176   def make_initializable_iterator(self, shared_name=None):
 177     if context.executing_eagerly():
 178       raise RuntimeError(
 179           "make_initializable_iterator is not supported when eager "
 180           "execution is enabled.")
 181 
 182     return _PrefetchToDeviceIterator(
 183         self._input_dataset,
 184         one_shot=False,
 185         devices=self._devices,
 186         buffer_size=self._buffer_size,
 187         shared_name=shared_name)
 188 
 189   def _as_variant_tensor(self):
 190     # TODO(mrry): Raise this error earlier (e.g. when one of the Dataset
 191     # transformation methods is called.
 192     # TODO(mrry): Investigate support for chaining further transformations after
 193     # the prefetch, including GPU support.
 194     raise NotImplementedError("`prefetch_to_devices()` must be the last "
 195                               "transformation in a dataset pipeline.")
 196 
 197   # TODO(priyag): Fix the output types, shapes and classes to match the result
 198   # of get_next (which has the additional nesting layer of devices now).
 199   @property
 200   def output_types(self):
 201     return self._input_dataset.output_types
 202 
 203   @property
 204   def output_shapes(self):
 205     return self._input_dataset.output_shapes
 206 
 207   @property
 208   def output_classes(self):
 209     return self._input_dataset.output_classes

 212 def prefetch_to_devices(devices, buffer_size=None):
 213   """A transformation that prefetches dataset values to the given `devices`.
 214 
 215   NOTE: Although the transformation creates a `tf.data.Dataset`, the
 216   transformation must be the final `Dataset` in the input pipeline.
 217 
 218   Args:
 219     devices: A nested structure of devices on which to prefetch the data. It can
 220       be a single device name, or a tuple or list of device names.
 221     buffer_size: (Optional.) The number of elements to buffer on each device.
 222       Defaults to an automatically chosen value.
 223 
 224   Returns:
 225     A `Dataset` transformation function, which can be passed to
 226     `tf.data.Dataset.apply`.
 227   """
 228 
 229   def _apply_fn(dataset):
 230     return _PrefetchToDeviceDataset(dataset, devices, buffer_size)
 231 
 232   return _apply_fn
