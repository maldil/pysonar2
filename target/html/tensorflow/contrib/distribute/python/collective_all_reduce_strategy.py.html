  35 class CollectiveAllReduceStrategy(mirrored_strategy.MirroredStrategy):
  36   """Distribution strategy that uses collective ops for all-reduce.
  37 
  38   It is similar to the MirroredStrategy but it uses collective ops for
  39   reduction.
  40 
  41   When `cluster_spec` is given by the `configure` method, it turns into the
  42   mulit-worker version that works on multiple workers with between-graph
  43   replication.
  44 
  45   Note: `configure` will be called by higher-level APIs if running in
  46   distributed environment.
  47   """
  48 
  49   def __init__(self, num_gpus_per_worker=0):
  50     """Initializes the object.
  51 
  52     Args:
  53       num_gpus_per_worker: number of local GPUs or GPUs per worker, the default
  54         is 0 meaning CPU only.
  55     """
  56     self._num_gpus_per_worker = num_gpus_per_worker
  57     self._initialize_local_worker(num_gpus_per_worker)
  58 
  59   def _initialize_local_worker(self, num_gpus_per_worker):
  60     """Initializes the object for local training."""
  61     self._is_chief = True
  62     self._num_workers = 1
  63 
  64     if num_gpus_per_worker:
  65       local_devices = [
  66           "/device:GPU:%d" % i for i in range(num_gpus_per_worker)
  67       ]
  68     else:
  69       local_devices = ["/device:CPU:0"]
  70 
  71     self._collective_keys = cross_tower_utils.CollectiveKeys()
  72     super(CollectiveAllReduceStrategy, self).__init__(
  73         devices=local_devices,
  74         cross_tower_ops=cross_tower_ops_lib.CollectiveAllReduce(
  75             num_workers=1,
  76             num_gpus_per_worker=num_gpus_per_worker,
  77             collective_keys=self._collective_keys))
  78 
  79     self._cluster_spec = None
  80     self._task_type = None
  81     self._task_id = None
  82 
  83     logging.info("CollectiveAllReduceStrategy with local_devices = %r",
  84                  local_devices)
  85 
  86   def _initialize_multi_worker(self, num_gpus_per_worker, cluster_spec,
  87                                task_type, task_id):
  88     """Initializes the object for multi-worker training."""
  89     if task_type is None or task_id is None:
  90       raise ValueError("When `cluster_spec` is given, you must also specify "
  91                        "`task_type` and `task_id`")
  92     if task_type not in ["chief", "worker"]:
  93       raise ValueError(
  94           "Unrecognized task_type: %r, valid task types are: \"chief\", "
  95           "\"worker\"." % task_type)
  96     cluster_spec = multi_worker_util.normalize_cluster_spec(cluster_spec)
  97     self._num_workers = len(cluster_spec.as_dict().get("worker", [])) + len(
  98         cluster_spec.as_dict().get("chief", []))
  99     if not self._num_workers:
 100       raise ValueError("No `worker` or `chief` tasks can be found in "
 101                        "`cluster_spec`.")
 102 
 103     self._is_chief = multi_worker_util.is_chief(cluster_spec, task_type,
 104                                                 task_id)
 105 
 106     worker_device = "/job:%s/task:%d" % (task_type, task_id)
 107     if num_gpus_per_worker:
 108       local_devices = [
 109           "%s/device:GPU:%d" % (worker_device, i)
 110           for i in range(num_gpus_per_worker)
 111       ]
 112     else:
 113       local_devices = [worker_device]
 114 
 115     self._collective_keys = cross_tower_utils.CollectiveKeys()
 116     super(CollectiveAllReduceStrategy, self).__init__(
 117         devices=local_devices,
 118         cross_tower_ops=cross_tower_ops_lib.CollectiveAllReduce(
 119             num_workers=self._num_workers,
 120             num_gpus_per_worker=num_gpus_per_worker,
 121             collective_keys=self._collective_keys))
 122 
 123     # Add a default device so that ops without specified devices will not end up
 124     # on other workers.
 125     self._default_device = "/job:%s/task:%d" % (task_type, task_id)
 126 
 127     self._cluster_spec = multi_worker_util.normalize_cluster_spec(cluster_spec)
 128     self._task_type = task_type
 129     self._task_id = task_id
 130 
 131     logging.info(
 132         "Multi-worker CollectiveAllReduceStrategy with "
 133         "cluster_spec = %r, task_type = %r, task_id = %r, "
 134         "num_workers = %r, local_devices = %r", cluster_spec.as_dict(),
 135         task_type, task_id, self._num_workers, local_devices)
 136 
 137   def _create_variable(self, next_creator, *args, **kwargs):
 138     colocate_with = kwargs.pop("colocate_with", None)
 139     devices = self._get_devices_from(colocate_with)
 140     group_size = len(devices) * self._num_workers
 141     group_key = self._collective_keys.get_group_key(self._devices)
 142 
 143     def _real_mirrored_creator(devices, *args, **kwargs):
 144       """Creates one MirroredVariable on the current worker."""
 145       index = {}
 146       unique_var_name = ops.get_default_graph().unique_name(
 147           kwargs["name"], mark_as_used=False).rstrip("/")
 148       collective_instance_key = self._collective_keys.get_instance_key(
 149           key_id=unique_var_name)
 150       if "initial_value" not in kwargs:
 151         raise ValueError("Initial value must be specified.")
 152       initial_value = kwargs["initial_value"]
 153       if callable(initial_value):
 154         initial_value_fn = initial_value
 155       else:
 156         initial_value_fn = lambda: initial_value
 157 
 158       for i, d in enumerate(devices):
 159         with ops.device(d):
 160           if i > 0:
 161             # Give replicas meaningful distinct names:
 162             var0name = index[devices[0]].name.split(":")[0]
 163             # We append a / to variable names created on towers with id > 0 to
 164             # ensure that we ignore the name scope and instead use the given
 165             # name as the absolute name of the variable.
 166             kwargs["name"] = "%s/replica_%d/" % (var0name, i)
 167 
 168           # The initial value fn makes sure variables all initialized to
 169           # same values. The first device of the chief worker will send their
 170           # variable values to other devices and other workers.
 171           def _overridden_initial_value_fn(device=d, index=i):  # pylint: disable=g-missing-docstring
 172             with ops.device(device):
 173               initial_value = initial_value_fn()
 174               assert not callable(initial_value)
 175               initial_value = ops.convert_to_tensor(initial_value)
 176 
 177               if self._is_chief and index == 0:
 178                 bcast_send = collective_ops.broadcast_send(
 179                     initial_value, initial_value.shape, initial_value.dtype,
 180                     group_size, group_key, collective_instance_key)
 181                 with ops.control_dependencies([bcast_send]):
 182                   return array_ops.identity(initial_value)
 183               else:
 184                 return collective_ops.broadcast_recv(
 185                     initial_value.shape, initial_value.dtype, group_size,
 186                     group_key, collective_instance_key)
 187 
 188           kwargs["initial_value"] = _overridden_initial_value_fn
 189 
 190           with context.context().device_policy(context.DEVICE_PLACEMENT_SILENT):
 191             v = next_creator(*args, **kwargs)
 192 
 193           if i == 0:
 194             actual_var_name = v.name.split(":")[0]
 195             assert unique_var_name == actual_var_name, "%r vs %r" % (
 196                 unique_var_name, actual_var_name)
 197           assert not isinstance(v, values.DistributedVariable)
 198           index[d] = v
 199       return index
 200 
 201     # pylint: disable=protected-access
 202     return mirrored_strategy._create_mirrored_variable(
 203         devices, _real_mirrored_creator, *args, **kwargs)

 205   def distribute_dataset(self, dataset_fn):
 206     """Distributes the dataset to each local GPU."""
 207     # TODO(yuefengz): shard the dataset.
 208     return values.PerDeviceDataset(
 209         self._call_dataset_fn(dataset_fn), self._devices, True)

 211   def configure(self,
 212                 session_config=None,
 213                 cluster_spec=None,
 214                 task_type=None,
 215                 task_id=None):
 216     """Configures the object.
 217 
 218     Args:
 219       session_config: a `tf.ConfigProto`
 220       cluster_spec: a dict, ClusterDef or ClusterSpec object specifying the
 221         cluster configurations.
 222       task_type: the current task type, such as "worker".
 223       task_id: the current task id.
 224 
 225     Raises:
 226       ValueError: if `task_type` is not in the `cluster_spec`.
 227     """
 228     if not self._cluster_spec and cluster_spec:
 229       # If a `cluster_spec` is already passed in, do nothing here.
 230       # TODO(yuefengz): check `cluster_spec` is the same if this object has
 231       # already been initialized with a `cluster_spec`.
 232       self._initialize_multi_worker(self._num_gpus_per_worker, cluster_spec,
 233                                     task_type, task_id)
 234 
 235     if not session_config or not self._cluster_spec:
 236       return
 237 
 238     assert self._task_type
 239     assert self._task_id is not None
 240 
 241     # Collective group leader is needed for collective ops to coordinate
 242     # workers.
 243     if "chief" in self._cluster_spec.jobs:
 244       session_config.experimental.collective_group_leader = (
 245           "/job:chief/replica:0/task:0")
 246     else:
 247       if "worker" not in self._cluster_spec.jobs:
 248         raise ValueError(
 249             "You must have `chief` or `worker` jobs in the `cluster_spec`.")
 250       session_config.experimental.collective_group_leader = (
 251           "/job:worker/replica:0/task:0")
 252 
 253     # The device filters prevent communication between workers.
 254     del session_config.device_filters[:]
 255     session_config.device_filters.append(
 256         "/job:%s/task:%d" % (self._task_type, self._task_id))
 257 
 258     # The scoped_allocator_optimization is to optimize graphs for collective
 259     # ops.
 260     rewrite_options = session_config.graph_options.rewrite_options
 261     rewrite_options.scoped_allocator_optimization = (
 262         rewriter_config_pb2.RewriterConfig.ON)
 263     del rewrite_options.scoped_allocator_opts.enable_op[:]
 264     rewrite_options.scoped_allocator_opts.enable_op.append("CollectiveReduce")
