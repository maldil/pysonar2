  37 class OneDeviceStrategy(distribute_lib.DistributionStrategy):
  38   """A distribution strategy for running on a single device."""
  39   # TODO(josh11b): Do we wrap values in types to generate errors if you are
  40   # doing something that won't work with other DistributionStrategy
  41   # implementations?
  42 
  43   def __init__(self, device, prefetch_on_device=None):
  44     super(OneDeviceStrategy, self).__init__()
  45     self._device = device
  46     self._prefetch_on_device = prefetch_on_device
  47     self._default_device = device
  48 
  49   def _create_variable(self, next_creator, *args, **kwargs):
  50     colocate_with = kwargs.pop("colocate_with", None)
  51     if colocate_with is None:
  52       with ops.device(self._device):
  53         return next_creator(*args, **kwargs)
  54     if isinstance(colocate_with, six.string_types):
  55       with ops.device(colocate_with):
  56         return next_creator(*args, **kwargs)
  57     if (isinstance(colocate_with, list) and len(colocate_with) == 1 and
  58         isinstance(colocate_with[0], six.string_types)):
  59       with ops.device(colocate_with[0]):
  60         return next_creator(*args, **kwargs)
  61     with ops.colocate_with(colocate_with):
  62       return next_creator(*args, **kwargs)
  63 
  64   def distribute_dataset(self, dataset_fn):
  65     return values.PerDeviceDataset(
  66         self._call_dataset_fn(dataset_fn), [self._device],
  67         self._prefetch_on_device)
  68 
  69   def _broadcast(self, tensor, destinations):
  70     del destinations
  71     return tensor
  72 
  73   # TODO(priyag): Deal with OutOfRange errors  once b/111349762 is fixed.
  74   def _run_steps_on_dataset(self, fn, iterator, iterations,
  75                             initial_loop_values=None):
  76     if initial_loop_values is None:
  77       initial_loop_values = {}
  78     initial_loop_values = nest.flatten(initial_loop_values)
  79 
  80     ctx = values.MultiStepContext()
  81     def body(i, *args):
  82       """A wrapper around `fn` to create the while loop body."""
  83       del args
  84       fn_inputs = iterator.get_next()
  85       if not isinstance(fn_inputs, tuple):
  86         fn_inputs = (fn_inputs,)
  87       fn_result = fn(ctx, *fn_inputs)
  88       flat_last_step_outputs = nest.flatten(ctx.last_step_outputs)
  89       with ops.control_dependencies([fn_result]):
  90         return [i + 1] + flat_last_step_outputs
  91 
  92     # We capture the control_flow_context at this point, before we run `fn`
  93     # inside a while_loop. This is useful in cases where we might need to exit
  94     # these contexts and get back to the outer context to do some things, for
  95     # e.g. create an op which should be evaluated only once at the end of the
  96     # loop on the host. One such usage is in creating metrics' value op.
  97     self._outer_control_flow_context = (
  98         ops.get_default_graph()._get_control_flow_context())  # pylint: disable=protected-access
  99 
 100     # TODO(priyag): Use max_iterations instead of an explicit counter.
 101     cond = lambda i, *args: i < iterations
 102     i = constant_op.constant(0)
 103     loop_result = control_flow_ops.while_loop(
 104         cond, body, [i] + initial_loop_values, name="",
 105         parallel_iterations=1, back_prop=False, swap_memory=False,
 106         return_same_structure=True)
 107     del self._outer_control_flow_context
 108 
 109     ctx.run_op = control_flow_ops.group(loop_result)
 110 
 111     # Convert the last_step_outputs from a list to the original dict structure
 112     # of last_step_outputs.
 113     last_step_tensor_outputs = loop_result[1:]
 114     last_step_tensor_outputs_dict = nest.pack_sequence_as(
 115         ctx.last_step_outputs, last_step_tensor_outputs)
 116 
 117     ctx._set_last_step_outputs(last_step_tensor_outputs_dict)  # pylint: disable=protected-access
 118     return ctx
 119 
 120   def _call_for_each_tower(self, fn, *args, **kwargs):
 121     # We don't run `fn` in multiple threads in OneDeviceStrategy.
 122     kwargs.pop("run_concurrently", None)
 123     with ops.device(self._device), _OneDeviceTowerContext(self):
 124       return fn(*args, **kwargs)
 125 
 126   def map(self, map_over, fn, *args, **kwargs):
 127     with ops.device(self._device):
 128       return values.MapOutput([fn(m, *args, **kwargs) for m in map_over])
 129 
 130   def _reduce(self, aggregation, value, destinations):
 131     del destinations
 132     if not isinstance(value, values.MapOutput):
 133       return value
 134     l = value.get()
 135     assert l
 136     with ops.device(self._device):
 137       if aggregation == vs.VariableAggregation.SUM:
 138         return math_ops.add_n(l)
 139       elif aggregation == vs.VariableAggregation.MEAN:
 140         return math_ops.add_n(l) / len(l)
 141       else:
 142         assert False
 143 
 144   def _update(self, var, options, fn, *args, **kwargs):
 145     # The implementations of _update() and _update_non_slot() are identical
 146     # except _update() passes `var` as the first argument to `fn()`.
 147     return self._update_non_slot(var, options, fn, var, *args, **kwargs)
 148 
 149   def _update_non_slot(self, colocate_with, options, fn, *args, **kwargs):
 150     del colocate_with
 151     should_group = options.pop("grouped")
 152     assert not options  # Validate that we are processing all of the options.
 153     with ops.device(self._device), distribute_lib.UpdateContext(self._device):
 154       result = fn(*args, **kwargs)
 155       if should_group:
 156         return result
 157       else:
 158         return nest.map_structure(self._unwrap, result)
 159 
 160   def read_var(self, tower_local_var):
 161     """Read the aggregate value of a tower-local variable."""
 162     return array_ops.identity(tower_local_var)
 163 
 164   def _unwrap(self, value):
 165     return [value]
 166 
 167   def value_container(self, value):
 168     return value
 169 
 170   @property
 171   def is_single_tower(self):
 172     return True
 173 
 174   @property
 175   def num_towers(self):
 176     return 1
 177 
 178   @property
 179   def worker_devices(self):
 180     return [self._device]
 181 
 182   @property
 183   def parameter_devices(self):
 184     return [self._device]
 185 
 186   def non_slot_devices(self, var_list):
 187     del var_list
 188     return [self._device]
 189 
 190   def _worker_device_index(self):
 191     return 0

 194 class _OneDeviceTowerContext(distribute_lib.TowerContext):
 195 
 196   def __init__(self, distribution_strategy):
 197     distribute_lib.TowerContext.__init__(
 198         self, distribution_strategy, tower_id=0)
 199 
 200   @property
 201   def device(self):
 202     return self._distribution_strategy.worker_devices[0]
