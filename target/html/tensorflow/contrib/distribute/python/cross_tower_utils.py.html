  36 def aggregate_gradients_using_nccl(tower_grads):
  37   """Aggregate gradients using nccl allreduce."""
  38   agg_all_g_and_v = []
  39   for single_g_and_v in zip(*tower_grads):
  40     single_grads = [g for g, _ in single_g_and_v]
  41     agg_grads = nccl.all_sum(single_grads)
  42     agg_all_g_and_v.append(
  43         [(g, v) for g, (_, v) in zip(agg_grads, single_g_and_v)])
  44 
  45   agg_all_g_and_v = list(zip(*agg_all_g_and_v))
  46 
  47   return agg_all_g_and_v

  50 def aggregate_gradients_using_hierarchical_copy(avail_devices, tower_grads):
  51   """Aggregate gradients using hierarchical copies.
  52 
  53   Args:
  54     avail_devices: available GPU devices.
  55     tower_grads: List of lists of (gradient, variable) tuples. The outer list
  56       is over towers. The inner list is over individual gradients.
  57 
  58   Returns:
  59     The list of (aggregated_gradient, variable), where the gradient has been
  60       summed across all towers and the variable is chosen from the first tower.
  61   """
  62   # This only works for DGX-1 type of machine topology
  63   # Device peer to peer matrix
  64   # DMA: 0 1 2 3 4 5 6 7
  65   # 0:   Y Y Y Y Y N N N
  66   # 1:   Y Y Y Y N Y N N
  67   # 2:   Y Y Y Y N N Y N
  68   # 3:   Y Y Y Y N N N Y
  69   # 4:   Y N N N Y Y Y Y
  70   # 5:   N Y N N Y Y Y Y
  71   # 6:   N N Y N Y Y Y Y
  72   # 7:   N N N Y Y Y Y Y
  73   agg_grads = []
  74   num_devices = len(avail_devices)
  75   # In the special case of DGX-1 machine topology, the two groups have equal
  76   # size.
  77   group_size = num_devices // 2
  78   for i, single_grads in enumerate(zip(*tower_grads)):
  79     group_0_main_device = i % num_devices
  80     group_1_main_device = (group_0_main_device + group_size) % num_devices
  81     if group_0_main_device < group_size:
  82       group_0_begin = 0
  83       group_1_begin = group_size
  84     else:
  85       group_0_begin = group_size
  86       group_1_begin = 0
  87 
  88     # Aggregate the first group.
  89     group_0_device_grads = single_grads[group_0_begin:
  90                                         group_0_begin + group_size]
  91     with ops.device(avail_devices[group_0_main_device]):
  92       group_0_agg_grads, _ = aggregate_single_gradient_using_copy(
  93           group_0_device_grads, False, False)
  94 
  95     # Aggregate the second group.
  96     group_1_device_grads = single_grads[group_1_begin:
  97                                         group_1_begin + group_size]
  98     with ops.device(avail_devices[group_1_main_device]):
  99       group_1_agg_grads, _ = aggregate_single_gradient_using_copy(
 100           group_1_device_grads, False, False)
 101 
 102     # Aggregate between the groups.
 103     with ops.device(avail_devices[group_0_main_device]):
 104       (agg_total_grads, _), _ = aggregate_single_gradient_using_copy(
 105           [group_0_agg_grads, group_1_agg_grads], False, False)
 106 
 107     # Broadcast the result back into the root of each group.
 108     with ops.device(avail_devices[group_0_main_device]):
 109       group_0_agg_grads_bcast = array_ops.identity(agg_total_grads)
 110     with ops.device(avail_devices[group_1_main_device]):
 111       group_1_agg_grads_bcast = array_ops.identity(agg_total_grads)
 112 
 113     agg_grads_bcast = []
 114     for j in range(len(single_grads)):
 115       with ops.device(avail_devices[j]):
 116         # Broadcast the result back to each member in the group from the root.
 117         if (group_0_main_device < group_size) == (j < group_size):
 118           src_device_grad = group_0_agg_grads_bcast
 119         else:
 120           src_device_grad = group_1_agg_grads_bcast
 121         agg_grads_bcast.append(array_ops.identity(src_device_grad))
 122 
 123     agg_grads.append(
 124         [(g, v) for g, (_, v) in zip(agg_grads_bcast, single_grads)])
 125 
 126   agg_grads = list(zip(*agg_grads))
 127 
 128   return agg_grads

 131 def aggregate_single_gradient_using_copy(grad_and_vars, use_mean,
 132                                          check_inf_nan):
 133   """Calculate the average gradient for a shared variable across all towers.
 134 
 135   Note that this function provides a synchronization point across all towers.
 136 
 137   Args:
 138     grad_and_vars: A list or tuple of (gradient, variable) tuples. Each
 139       (gradient, variable) pair within the outer list represents the gradient
 140       of the variable calculated for a single tower, and the number of pairs
 141       equals the number of towers.
 142     use_mean: if True, mean is taken, else sum of gradients is taken.
 143     check_inf_nan: check grads for nans and infs.
 144 
 145   Returns:
 146     The tuple ([(average_gradient, variable),], has_nan_or_inf) where the
 147       gradient has been averaged across all towers. The variable is chosen from
 148       the first tower. The has_nan_or_inf indicates the grads has nan or inf.
 149   """
 150   grads = [g for g, _ in grad_and_vars]
 151   grad = math_ops.add_n(grads)
 152 
 153   if use_mean and len(grads) > 1:
 154     grad = array_ops.multiply(grad, 1.0 / len(grads))
 155 
 156   v = grad_and_vars[0][1]
 157   if check_inf_nan:
 158     has_nan_or_inf = array_ops.logical_not(
 159         array_ops.reduce_all(array_ops.is_finite(grads)))
 160     return (grad, v), has_nan_or_inf
 161   else:
 162     return (grad, v), None

 165 def group_device_names(devices, group_size):
 166   """Group device names into groups of group_size.
 167 
 168   Args:
 169     devices: a list of canonical device strings.
 170     group_size: integer which is equal to or greater than 1.
 171 
 172   Returns:
 173     list of lists of devices, where each inner list is group_size long,
 174       and each device appears at least once in an inner list.  If
 175       len(devices) % group_size == 0 then each device will appear exactly once.
 176 
 177   Raises:
 178     ValueError: if group_size > len(devices)
 179   """
 180   num_devices = len(devices)
 181   if group_size > num_devices:
 182     raise ValueError(
 183         'only %d devices, but group_size=%d' % (num_devices, group_size))
 184   num_groups = (
 185       num_devices // group_size + (1 if (num_devices % group_size != 0) else 0))
 186   groups = [[] for i in range(num_groups)]
 187   for i in range(num_groups * group_size):
 188     groups[i % num_groups].append(devices[i % num_devices])
 189   return groups

 192 def split_grads_by_size(threshold_size, device_grads):
 193   """Break gradients into two sets according to tensor size.
 194 
 195   Args:
 196     threshold_size: int size cutoff for small vs large tensor.
 197     device_grads: List of lists of (gradient, variable) tuples.  The outer
 198         list is over devices. The inner list is over individual gradients.
 199 
 200   Returns:
 201     small_grads: Subset of device_grads where shape is <= threshold_size
 202        elements.
 203     large_grads: Subset of device_grads where shape is > threshold_size
 204        elements.
 205   """
 206   small_grads = []
 207   large_grads = []
 208   for dl in device_grads:
 209     small_dl = []
 210     large_dl = []
 211     for (g, v) in dl:
 212       tensor_size = g.get_shape().num_elements()
 213       if tensor_size <= threshold_size:
 214         small_dl.append([g, v])
 215       else:
 216         large_dl.append([g, v])
 217     if small_dl:
 218       small_grads.append(small_dl)
 219     if large_dl:
 220       large_grads.append(large_dl)
 221   return small_grads, large_grads

 233 class CollectiveKeys(object):
 234   """Class that manages collective keys.
 235 
 236   We need to manage three different keys for collective:
 237 
 238   *Group key*: an integer key to identify the set of cooperative devices.
 239   Collective ops work under the same set of devices must using the same group
 240   key.
 241 
 242   *Instance key*: an integer key to identify the set of same counterpart of
 243   tensors on different devices in a device group that need to be all-reduced.
 244 
 245   "Graph key": an integer key that is unique key graph. This is used to support
 246   multiple graphs per client session. It must be non-zero and set in the
 247   `config` argument of each call to `session.run`.
 248   """
 249 
 250   def __init__(self,
 251                group_key_start=1,
 252                instance_key_start=100,
 253                instance_key_with_id_start=10000):
 254     """Initializes the object.
 255 
 256     Args:
 257       group_key_start: the starting integer of group key.
 258       instance_key_start: the starting integer of instance key.
 259       instance_key_with_id_start: the starting integer of instance key that is
 260         recorded with an id.
 261     """
 262     self._group_key = group_key_start
 263     self._group_key_table = dict()
 264 
 265     # For instance keys with ids
 266     self._instance_key_id_to_key_table = dict()
 267     self._instance_key_with_id_counter = instance_key_with_id_start
 268 
 269     # For instance keys without ids
 270     self._instance_key_start = instance_key_start
 271 
 272   def _get_thread_local_object(self):
 273     # We make instance key without key ids thread local so that it will work
 274     # with MirroredStrategy and distribute coordinator.
 275     if not hasattr(_thread_local, 'instance_key'):
 276       _thread_local.instance_key = self._instance_key_start
 277     return _thread_local
 278 
 279   def get_group_key(self, devices):
 280     """Returns a group key for the set of devices.
 281 
 282     Args:
 283       devices: list of strings naming devices in a collective group.
 284 
 285     Returns:
 286       int key uniquely identifying the set of device names.
 287     """
 288     parsed = [pydev.DeviceSpec.from_string(d) for d in devices]
 289     # In the between-graph replicated training, different workers need to get
 290     # the same device key. So we remove the task_type and task_id from the
 291     # devices.
 292     # TODO(yuefengz): in the in-graph replicated training, we need to include
 293     # task_type and task_id.
 294     names = sorted(['%s:%d' % (d.device_type, d.device_index) for d in parsed])
 295     key_id = ','.join(names)
 296     with _lock:
 297       if key_id not in self._group_key_table:
 298         new_key = self._group_key
 299         self._group_key += 1
 300         self._group_key_table[key_id] = new_key
 301     return self._group_key_table[key_id]
 302 
 303   def get_instance_key(self, key_id=None):
 304     """Returns a new instance key for use in defining a collective op.
 305 
 306     Args:
 307       key_id: optional string. If set, key will be recorded and the same key
 308         will be returned when the same key_id is provided. If not, an increasing
 309         instance key will be returned.
 310     """
 311     if key_id:
 312       with _lock:
 313         if key_id not in self._instance_key_id_to_key_table:
 314           self._instance_key_with_id_counter += 1
 315           self._instance_key_id_to_key_table[key_id] = (
 316               self._instance_key_with_id_counter)
 317       return self._instance_key_id_to_key_table[key_id]
 318     else:
 319       v = self._get_thread_local_object().instance_key
 320       self._get_thread_local_object().instance_key += 1
 321       return v

 324 def build_collective_reduce(input_tensors,
 325                             num_workers,
 326                             collective_keys,
 327                             reduction_op='Add',
 328                             unary_op='Id'):
 329   """Build a subgraph that does one full all-reduce, using the collective Op.
 330 
 331   Args:
 332     input_tensors: tensors within a single worker graph that are to be reduced
 333       together; must be one per device.
 334     num_workers: total number of workers with identical independent graphs that
 335       will be doing this same reduction.  The reduction will actually include
 336       the corresponding tensors at all these workers.
 337     collective_keys: a CollectiveKeys object.
 338     reduction_op: string naming the reduction op.
 339     unary_op: string naming the unary final op.
 340 
 341   Returns:
 342     An array of final tensors, one per device, computed by the full reduction.
 343 
 344   Raises:
 345     ValueError: There must be at least two tensors over all the workers.
 346   """
 347   group_size = len(input_tensors) * num_workers
 348   if group_size < 2:
 349     raise ValueError('num_workers * len(input_tensors) must be 2 or greater')
 350   devices = [t.device for t in input_tensors]
 351   num_devices = len(devices)
 352   group_key = collective_keys.get_group_key(devices)
 353   instance_key = collective_keys.get_instance_key()
 354   out_tensors = []
 355   subdiv_offsets = [0]  # TODO(tucker): maybe support non-default subdiv spec
 356   for d in range(num_devices):
 357     with ops.device(devices[d]):
 358       reduce_op = collective_ops.all_reduce(
 359           input_tensors[d], group_size, group_key, instance_key, reduction_op,
 360           unary_op, subdiv_offsets)
 361       out_tensors.append(reduce_op)
 362   return out_tensors

 365 def sum_grad_and_var_all_reduce(grad_and_vars,
 366                                 num_workers,
 367                                 alg,
 368                                 gpu_indices,
 369                                 aux_devices=None,
 370                                 num_shards=1):
 371   """Apply all-reduce algorithm over specified gradient tensors."""
 372   with ops.name_scope('allreduce'):
 373     # Note that each grad_and_vars looks like the following:
 374     #   ((grad0_gpu0, var0_gpu0), ... , (grad0_gpuN, var0_gpuN))
 375     scaled_grads = [g for g, _ in grad_and_vars]
 376     if alg == 'nccl':
 377       summed_grads = nccl.all_sum(scaled_grads)
 378     elif alg == 'xring':
 379       summed_grads = all_reduce.build_ring_all_reduce(
 380           scaled_grads, num_workers, num_shards, gpu_indices, math_ops.add)
 381     elif alg == 'nccl/xring':
 382       summed_grads = all_reduce.build_nccl_then_ring(scaled_grads, num_shards,
 383                                                      math_ops.add)
 384     elif alg == 'nccl/rechd':
 385       summed_grads = all_reduce.build_nccl_then_recursive_hd(
 386           scaled_grads, math_ops.add)
 387     elif alg == 'nccl/pscpu':
 388       summed_grads = all_reduce.build_nccl_then_shuffle(
 389           scaled_grads, aux_devices, math_ops.add, math_ops.add_n)
 390     elif alg == 'pscpu/pscpu':
 391       second_gather_devices = aux_devices[:num_shards]
 392       summed_grads = all_reduce.build_shuffle_then_shuffle(
 393           scaled_grads, aux_devices, second_gather_devices, math_ops.add_n)
 394     elif alg in ['pscpu', 'psgpu']:
 395       summed_grads = all_reduce.build_shuffle_all_reduce(
 396           scaled_grads, aux_devices, math_ops.add_n)
 397     else:
 398       raise ValueError('unsupported all_reduce alg: ', alg)
 399 
 400   result = []
 401   for (_, v), g in zip(grad_and_vars, summed_grads):
 402     result.append([g, v])
 403   return result

 406 def sum_gradients_all_reduce(dev_prefixes, tower_grads, num_workers, alg,
 407                              num_shards, gpu_indices):
 408   """Apply all-reduce algorithm over specified gradient tensors.
 409 
 410   Args:
 411     dev_prefixes: list of prefix strings to use to generate PS device names.
 412     tower_grads: the gradients to reduce.
 413     num_workers: number of worker processes across entire job.
 414     alg: the all-reduce algorithm to apply.
 415     num_shards: alg-specific sharding factor.
 416     gpu_indices: indices of local GPUs in order usable for ring-reduce.
 417 
 418   Returns:
 419     list of reduced tensors
 420   """
 421   alg_contains_shuffle = any([n in alg for n in ['pscpu', 'psgpu']])
 422   is_hierarchical = '/' in alg
 423   if 'pscpu' in alg:
 424     aux_devices = [prefix + '/cpu:0' for prefix in dev_prefixes]
 425   elif 'psgpu' in alg:
 426     aux_devices = [
 427         prefix + '/gpu:%d' % i
 428         for i in range(len(gpu_indices))
 429         for prefix in dev_prefixes
 430     ]
 431   else:
 432     aux_devices = ['/job:localhost/cpu:0']
 433   # Auxiliary devices for hierarchical all-reduces.
 434   aux_device_groups = group_device_names(
 435       aux_devices, num_shards if alg_contains_shuffle else 1)
 436   group_index = 0
 437   reduced_gv_list = []
 438   for grad_and_vars in zip(*tower_grads):
 439     reduced_gv_list.append(
 440         sum_grad_and_var_all_reduce(
 441             grad_and_vars, num_workers, alg, gpu_indices, aux_devices
 442             if is_hierarchical else aux_device_groups[group_index], num_shards))
 443     group_index = (group_index + 1) % len(aux_device_groups)
 444   new_tower_grads = [list(x) for x in zip(*reduced_gv_list)]
 445   return new_tower_grads

 448 def extract_ranges(index_list, range_size_limit=32):
 449   """Extract consecutive ranges and singles from index_list.
 450 
 451   Args:
 452     index_list: List of monotone increasing non-negative integers.
 453     range_size_limit: Largest size range to return.  If a larger
 454       consecutive range exists, it will be returned as multiple
 455       ranges.
 456 
 457   Returns:
 458     (ranges, singles) where ranges is a list of [first, last] pairs of
 459       consecutive elements in index_list, and singles is all of the
 460       other elements, in original order.
 461   """
 462   if not index_list:
 463     return [], []
 464   first = index_list[0]
 465   last = first
 466   ranges = []
 467   singles = []
 468   for i in index_list[1:]:
 469     if i == last + 1 and (last - first) <= range_size_limit:
 470       last = i
 471     else:
 472       if last > first:
 473         ranges.append([first, last])
 474       else:
 475         singles.append(first)
 476       first = i
 477       last = i
 478   if last > first:
 479     ranges.append([first, last])
 480   else:
 481     singles.append(first)
 482   return ranges, singles

 488 def pack_range(key, packing, grad_vars, rng):
 489   """Form the concatenation of a specified range of gradient tensors.
 490 
 491   Args:
 492     key: Value under which to store meta-data in packing that will be used
 493       later to restore the grad_var list structure.
 494     packing: Dict holding data describing packed ranges of small tensors.
 495     grad_vars: List of (grad, var) pairs for one tower.
 496     rng: A pair of integers giving the first, last indices of a consecutive
 497       range of tensors to be packed.
 498 
 499   Returns:
 500     A tensor that is the concatenation of all the specified small tensors.
 501   """
 502   to_pack = grad_vars[rng[0]:rng[1] + 1]
 503   members = []
 504   variables = []
 505   restore_shapes = []
 506   with ops.name_scope('pack'):
 507     for g, v in to_pack:
 508       variables.append(v)
 509       restore_shapes.append(g.shape)
 510       with ops.device(g.device):
 511         members.append(array_ops.reshape(g, [-1]))
 512     packing[key] = GradPackTuple(
 513         indices=range(rng[0], rng[1] + 1),
 514         vars=variables,
 515         shapes=restore_shapes)
 516     with ops.device(members[0].device):
 517       return array_ops.concat(members, 0)

 520 def unpack_grad_tuple(gv, gpt):
 521   """Unpack a previously packed collection of gradient tensors.
 522 
 523   Args:
 524     gv: A (grad, var) pair to be unpacked.
 525     gpt: A GradPackTuple describing the packing operation that produced gv.
 526 
 527   Returns:
 528     A list of (grad, var) pairs corresponding to the values that were
 529      originally packed into gv, maybe following subsequent operations like
 530      reduction.
 531   """
 532   elt_widths = [x.num_elements() for x in gpt.shapes]
 533   with ops.device(gv[0][0].device):
 534     with ops.name_scope('unpack'):
 535       splits = array_ops.split(gv[0], elt_widths)
 536       unpacked_gv = []
 537       for idx, s in enumerate(splits):
 538         unpacked_gv.append((array_ops.reshape(s, gpt.shapes[idx]),
 539                             gpt.vars[idx]))
 540   return unpacked_gv

 543 def pack_small_tensors(tower_grads, max_bytes=0, max_group=0):
 544   """Concatenate small gradient tensors together for reduction.
 545 
 546   Args:
 547     tower_grads: List of lists of (gradient, variable) tuples.
 548     max_bytes: Int giving max number of bytes in a tensor that
 549       may be considered small.
 550     max_group: Int giving max number of small tensors that may be
 551       concatenated into one new tensor.
 552 
 553   Returns:
 554     new_tower_grads, packing where new_tower_grads is identical to
 555       tower_grads except that all feasible small_tensors have been removed
 556       from their places and concatenated into larger tensors that are
 557       now in the front of the list for each tower, and packing contains
 558       the data necessary to restore the tower_grads structure.
 559 
 560   Look through the first tower for gradients of the same type (float),
 561   and small size, that are all sequential.  For each such group,
 562   replace by a new tensor that is a flattened concatenation.  Note
 563   that the corresponding variable will be absent, which doesn't matter
 564   because it isn't used during all-reduce.
 565 
 566   Requires:
 567     Every gv_list in towers must have isomorphic structure including identical
 568       tensor sizes and types.
 569   """
 570   small_indices = []
 571   large_indices = []
 572   for idx, (g, _) in enumerate(tower_grads[0]):
 573     if g.dtype == dtypes.float32 and (4 * g.shape.num_elements()) <= max_bytes:
 574       small_indices.append(idx)
 575     else:
 576       large_indices.append(idx)
 577   small_ranges, small_singles = extract_ranges(
 578       small_indices, range_size_limit=max_group)
 579   large_indices = sorted(large_indices + small_singles)
 580   num_gv = len(tower_grads[0])
 581   packing = {}
 582   if small_ranges:
 583     new_tower_grads = []
 584     for dev_idx, gv_list in enumerate(tower_grads):
 585       assert len(gv_list) == num_gv
 586       new_gv_list = []
 587       for r in small_ranges:
 588         key = '%d:%d' % (dev_idx, len(new_gv_list))
 589         new_gv_list.append((pack_range(key, packing, gv_list, r),
 590                             'packing_var_placeholder'))
 591       for i in large_indices:
 592         new_gv_list.append(gv_list[i])
 593       new_tower_grads.append(new_gv_list)
 594     return new_tower_grads, packing
 595   else:
 596     return tower_grads, None

 599 def unpack_small_tensors(tower_grads, packing):
 600   """Undo the structure alterations to tower_grads done by pack_small_tensors.
 601 
 602   Args:
 603     tower_grads: List of List of (grad, var) tuples.
 604     packing: A dict generated by pack_small_tensors describing the changes
 605       it made to tower_grads.
 606 
 607   Returns:
 608     new_tower_grads: identical to tower_grads except that concatenations
 609       of small tensors have been split apart and returned to their original
 610       positions, paired with their original variables.
 611   """
 612   if not packing:
 613     return tower_grads
 614   new_tower_grads = []
 615   num_devices = len(tower_grads)
 616   num_packed = len(packing.keys()) // num_devices
 617   for dev_idx, gv_list in enumerate(tower_grads):
 618     gv_list = list(gv_list)
 619     new_gv_list = gv_list[num_packed:]
 620     for i in range(num_packed):
 621       k = '%d:%d' % (dev_idx, i)
 622       gpt = packing[k]
 623       gv = unpack_grad_tuple(gv_list[i], gpt)
 624       for gi, idx in enumerate(gpt.indices):
 625         assert idx == gpt.indices[gi]
 626         new_gv_list.insert(idx, gv[gi])
 627     new_tower_grads.append(new_gv_list)
 628   return new_tower_grads

 648 def copy_tensor_or_indexed_slices_to_device(value, device):
 649   with ops.device(device):
 650     if isinstance(value, ops.IndexedSlices):
 651       copied_values = array_ops.identity(value.values)
 652       copied_indices = array_ops.identity(value.indices)
 653       copied_shape = array_ops.identity(value.dense_shape)
 654       result = ops.IndexedSlices(copied_values, copied_indices, copied_shape)
 655     else:
 656       result = array_ops.identity(value)

 660 def contains_indexed_slices(value):
 661   """Check whether the value is `IndexedSlices` or contains `IndexedSlices`."""
 662   if isinstance(value, ops.IndexedSlices):
 663     return True
 664   elif isinstance(value, (list, tuple)) and value:
 665     return any(contains_indexed_slices(v) for v in value)
 666   elif isinstance(value, value_lib.DistributedValues):
 667     return contains_indexed_slices(list(value._index.values()))  # pylint: disable=protected-access
 668   elif isinstance(value, value_lib.MapOutput):
 669     return contains_indexed_slices(value.get())
 670   else:
 671     return False
