  21 from tensorflow.python.data.ops import readers
  22 from tensorflow.python.data.util import nest

  28 # TODO(priyag): Any other reader datasets to consider here?
  29 _READER_DATASET_OPS = [
  30     "TextLineDataset",
  31     "TFRecordDataset",
  32     "FixedLengthRecordDataset"
  33 ]
  34 
  35 
  36 # pylint: disable=protected-access
  37 def auto_shard_dataset(dataset, num_shards, index):
  38   """Shard the input pipeline by sharding the underlying list of files.
  39 
  40   Args:
  41     dataset: A `tf.data.Dataset` instance, typically the result of a bunch of
  42       dataset transformations.
  43     num_shards: A `tf.int64` scalar `tf.Tensor`, representing the number of
  44         shards operating in parallel. Same usage as in `Dataset.shard`.
  45     index: A `tf.int64` scalar `tf.Tensor`, representing the worker index.
  46       Same usage as in `Dataset.shard`.
  47 
  48   Returns:
  49     A modified `Dataset` obtained by updating the pipeline sharded by the
  50     files. The input dataset will be returned if we cannot automatically
  51     determine a good way to shard the input dataset.
  52   """
  53 
  54   # TODO(priyag): Clone datasets instead of updating in place, similar to the
  55   # clone method for TFRecordDataset.
  56   def _auto_shard_impl(dataset, found_reader_op):
  57     """Recursive implementation of auto sharding."""
  58 
  59     if not found_reader_op:
  60       # TODO(priyag): Make this check more robust by enforcing some common
  61       # property on reader datasets.
  62       if (isinstance(dataset, readers.TextLineDataset) or
  63           isinstance(dataset, readers.FixedLengthRecordDataset)):
  64         filenames_tensor = dataset._filenames
  65         num_files = array_ops.size(filenames_tensor)
  66         sharded_filenames_tensor = array_ops.gather(
  67             filenames_tensor, math_ops.range(index, num_files, num_shards))
  68         dataset._filenames = sharded_filenames_tensor
  69         return dataset
  70       elif isinstance(dataset, readers.TFRecordDataset):
  71         # `TFRecordDataset` needs to be handled separately than other readers
  72         # because it converts filenames to a dataset first. Also, we clone it
  73         # instead of updating in place because it has special logic in the
  74         # constructor. Eventually we will change all cases to clone datasets
  75         # instead of updating in-place.
  76         return dataset._clone(
  77             filenames=dataset._filenames.shard(num_shards, index))
  78       elif hasattr(dataset, "_map_func"):
  79         # TODO(priyag): Make this check more robust by enforcing some common
  80         # property on all map/flatmap/interleave datasets.
  81         map_func_def = dataset._map_func.definition
  82         for node in map_func_def.node_def:
  83           if node.op in _READER_DATASET_OPS:
  84             found_reader_op = True
  85             break
  86           elif node.op == "FlatMapDataset":
  87             # TODO(priyag): Should this check for other map datasets? Should it
  88             # be recursive? It is too specific to implementation of
  89             # TFRecordDataset right now.
  90             nested_func_name = node.attr["f"].func.name
  91             nested_func = ops.get_default_graph()._functions[nested_func_name]
  92             for nested_node in nested_func.definition.node_def:
  93               if nested_node.op in _READER_DATASET_OPS:
  94                 found_reader_op = True
  95                 break
  96             if found_reader_op:
  97               break
  98         if found_reader_op:
  99           dataset._input_dataset = _auto_shard_impl(
 100               dataset._input_dataset, found_reader_op)
 101           return dataset
 102 
 103     # TODO(priyag): Make _input_dataset(s) a common property of all datasets to
 104     # make this check more robust.
 105     if hasattr(dataset, "_input_dataset"):
 106       dataset._input_dataset = _auto_shard_impl(
 107           dataset._input_dataset, found_reader_op)
 108       if hasattr(dataset, "_dataset_to_concatenate"):
 109         # Special case for `ConcatentateDataset`. We want to shard all input
 110         # datasets.
 111         dataset._dataset_to_concatenate = _auto_shard_impl(
 112             dataset._dataset_to_concatenate, found_reader_op)
 113       return dataset
 114 
 115     if hasattr(dataset, "_datasets"):
 116       # Special case for `ZipDataset`.
 117       dataset._datasets = nest.pack_sequence_as(dataset._datasets, [
 118           _auto_shard_impl(ds, found_reader_op)
 119           for ds in nest.flatten(dataset._datasets)
 120       ])
 121       return dataset
 122 
 123     if not found_reader_op:
 124       tf_logging.warn(
 125           "Could not find a standard reader in the input pipeline"
 126           "(one of TextLineDataset, TFRecordDataset, FixedLengthRecordDataset)."
 127           "So auto-sharding is not done. Please verify correctness of "
 128           "auto-sharding for your input.")
 129       # TODO(yuefengz): maybe still shard it?
 130       return dataset
 131 
 132     # TODO(priyag): What do we want to do if the number of filenames is
 133     # uneven in the number of shards? By default, this will just return as
 134     # many items it can before throwing OutOfRangeError.
 135     # TODO(priyag): This will shard the filenames before any shuffling of the
 136     # filename dataset. It might be desirable to shard after shuffling
 137     # filenames? If so, how do we achieve that?
 138     return dataset.shard(num_shards, index)
 139 
 140   return _auto_shard_impl(dataset=dataset, found_reader_op=False)
