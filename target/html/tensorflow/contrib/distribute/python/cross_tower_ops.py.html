  37 def check_destinations(destinations):
  38   """Checks whether `destinations` is not empty.
  39 
  40   Args:
  41     destinations: a DistributedValues, Variable, string or a list of strings.
  42 
  43   Returns:
  44     Boolean which is True if `destinations` is not empty.
  45   """
  46   # Calling bool() on a ResourceVariable is not allowed.
  47   if isinstance(destinations, resource_variable_ops.ResourceVariable):
  48     return bool(destinations.device)
  49   return bool(destinations)
  50

  52 def validate_destinations(destinations):
  53   if not isinstance(
  54       destinations,
  55       (value_lib.DistributedValues, resource_variable_ops.ResourceVariable,
  56        value_lib.AggregatingVariable, six.string_types, list)):
  57     raise ValueError("destinations must be one of a `DistributedValues` object,"
  58                      " a tf.Variable object, a device string, a list of device "
  59                      "strings")
  60 
  61   if not check_destinations(destinations):
  62     raise ValueError("destinations can not be empty")

  65 def _make_tensor_into_per_device(input_tensor):
  66   """Converts a single tensor into a PerDevice object."""
  67   if isinstance(input_tensor, (tuple, list)):
  68     raise ValueError("Cannot convert `input_tensor` to a `PerDevice` object, "
  69                      "got %r but expected a object that is not a tuple or list."
  70                      % (input_tensor,))
  71   if isinstance(input_tensor, value_lib.PerDevice):
  72     return input_tensor
  73 
  74   try:
  75     device = input_tensor.device
  76   except AttributeError:
  77     raise ValueError("Cannot convert `input_tensor` to a `PerDevice` object "
  78                      "because it doesn't have device set.")
  79 
  80   return value_lib.PerDevice({device: input_tensor})

  83 def _normalize_value_destination_pairs(value_destination_pairs):
  84   """Converts each tensor into a PerDevice object in the input list."""
  85   result = []
  86   if not isinstance(value_destination_pairs, (list, tuple)):
  87     raise ValueError("`value_destination_pairs` should be a list or tuple")
  88   for pair in value_destination_pairs:
  89     if not isinstance(pair, tuple):
  90       raise ValueError(
  91           "Each element of `value_destination_pairs` should be a tuple.")
  92     if len(pair) != 2:
  93       raise ValueError("Each element of `value_destination_pairs` should be a "
  94                        "tuple of size 2.")
  95 
  96     per_device = _make_tensor_into_per_device(pair[0])
  97     result.append((per_device, pair[1]))
  98   return result

 101 def _validate_value_destination_pairs(value_destination_pairs):
 102   # TODO(yuefengz): raise exceptions instead of returning False.
 103   # pylint: disable=g-missing-docstring
 104   if not value_destination_pairs: return False
 105   if not isinstance(value_destination_pairs, (list, tuple)): return False
 106   if not all([isinstance(pair, tuple) for pair in value_destination_pairs]):
 107     return False
 108   if not all([isinstance(v[0], value_lib.PerDevice)
 109               for v in value_destination_pairs]):
 110     return False
 111   return True

 114 # TODO(yuefengz): consider calling this function in the caller of CrossTowerOps.
 115 def get_devices_from(destinations):
 116   if isinstance(destinations, value_lib.DistributedValues):
 117     return list(destinations.devices)
 118   elif isinstance(destinations, (resource_variable_ops.ResourceVariable,
 119                                  value_lib.AggregatingVariable)):
 120     return [destinations.device]
 121   elif isinstance(destinations, six.string_types):
 122     return [device_util.resolve(destinations)]
 123   elif isinstance(destinations, (list, tuple)):
 124     return [device_util.resolve(destination) for destination in destinations]
 125   else:
 126     return [destinations.device]

 129 def _devices_match(left, right):
 130   return set(get_devices_from(left)) == set(get_devices_from(right))

 133 def _all_devices_match(value_destination_pairs):
 134   if not all([_devices_match(v, d) for v, d in value_destination_pairs]):
 135     return False
 136   if not all([_devices_match(v, value_destination_pairs[0][0])
 137               for v, _ in value_destination_pairs[1:]]):
 138     return False
 139   return True

 142 def _simple_broadcast(value, destinations):
 143   index = {}
 144   devices = get_devices_from(destinations)
 145   for d in devices:
 146     index[d] = cross_tower_utils.copy_tensor_or_indexed_slices_to_device(
 147         value, d)
 148   return value_lib.Mirrored(index)

 151 def _simple_reduce(per_device_value, reduce_to_device, accumulation_fn,
 152                    aggregation):
 153   # pylint: disable=g-missing-docstring
 154   all_values = []
 155   count = 0
 156   for v in per_device_value._index.values():  # pylint: disable=protected-access
 157     if isinstance(v, value_lib.MapOutput):
 158       v_list = v.get()
 159       if not v_list:
 160         continue
 161       count += len(v_list)
 162       # Sum within each device before aggregating across devices.
 163       # TODO(yuefengz): Check whether it helps to use accumulation_fn here.
 164       v = cross_tower_utils.aggregate_tensors_or_indexed_slices(
 165           v_list, math_ops.add_n)
 166     else:
 167       count += 1
 168     all_values.append(v)
 169   if not all_values:
 170     raise ValueError("`per_device_value` must be non-empty")
 171 
 172   with ops.device(reduce_to_device):
 173     with context.context().device_policy(context.DEVICE_PLACEMENT_SILENT):
 174       reduced = cross_tower_utils.aggregate_tensors_or_indexed_slices(
 175           all_values, accumulation_fn)
 176       if aggregation == vs.VariableAggregation.MEAN:
 177         reduced = cross_tower_utils.divide_by_n_tensors_or_indexed_slices(
 178             reduced, count)
 179       elif aggregation != vs.VariableAggregation.SUM:
 180         raise ValueError("`aggregation` must be VariableAggregation.SUM "
 181                          "or VariableAggregation.MEAN.")
 182   return reduced

 185 class CrossTowerOps(object):
 186   """Base class for cross-tower reduction and broadcasting algorithms."""
 187 
 188   def __init__(self):
 189     pass
 190 
 191   def reduce(self, aggregation, per_device_value, destinations):
 192     """Reduce `per_device_value` to `destinations`.
 193 
 194     It runs the reduction operation defined by `aggregation` and put the
 195     result on `destinations`.
 196 
 197     Args:
 198       aggregation: Indicates how a variable will be aggregated. Accepted values
 199         are `tf.VariableAggregation.SUM`, `tf.VariableAggregation.MEAN`.
 200       per_device_value: a PerDevice object or a tensor with device set.
 201       destinations: the reduction destinations.
 202 
 203     Returns:
 204       a Mirrored object.
 205 
 206     Raises:
 207       ValueError: if per_device_value is not a PerDevice object.
 208     """
 209     if not isinstance(per_device_value, value_lib.PerDevice):
 210       per_device_value = _make_tensor_into_per_device(per_device_value)
 211 
 212     validate_destinations(destinations)
 213     return self._reduce(aggregation, per_device_value, destinations)
 214 
 215   def batch_reduce(self, aggregation, value_destination_pairs):
 216     """Reduce PerDevice objects in a batch.
 217 
 218     Reduce each first element in `value_destination_pairs` to each second
 219     element which indicates the destinations.
 220 
 221     Args:
 222       aggregation: Indicates how a variable will be aggregated. Accepted values
 223         are `tf.VariableAggregation.SUM`, `tf.VariableAggregation.MEAN`.
 224       value_destination_pairs: a list or a tuple of tuples of PerDevice objects
 225         (or tensors with device set if there is one tower) and destinations.
 226 
 227     Returns:
 228       a list of Mirrored objects.
 229 
 230     Raises:
 231       ValueError: if `value_destination_pairs` is not a list or a tuple of
 232         tuples of PerDevice objects and destinations
 233     """
 234     if not _validate_value_destination_pairs(value_destination_pairs):
 235       # If the first element of each pair is a tensor, we try to turn it into a
 236       # PerDevice object.
 237       value_destination_pairs = _normalize_value_destination_pairs(
 238           value_destination_pairs)
 239 
 240     for _, d in value_destination_pairs:
 241       validate_destinations(d)
 242 
 243     return self._batch_reduce(aggregation, value_destination_pairs)
 244 
 245   def broadcast(self, tensor, destinations):
 246     """Broadcast the `tensor` to destinations.
 247 
 248     Args:
 249       tensor: the tensor to broadcast.
 250       destinations: the broadcast destinations.
 251 
 252     Returns:
 253       a Mirrored object.
 254     """
 255     validate_destinations(destinations)
 256     return self._broadcast(tensor, destinations)
 257 
 258   def _reduce(self, aggregation, per_device_value, destinations):
 259     raise NotImplementedError(
 260         "_reduce method must be implemented in descendants.")
 261 
 262   def _batch_reduce(self, aggregation, value_destination_pairs):
 263     raise NotImplementedError(
 264         "_batch_reduce method must be implemented in descendants.")
 265 
 266   def _broadcast(self, tensor, destinations):
 267     return _simple_broadcast(tensor, destinations)

 270 class ReductionToOneDeviceCrossTowerOps(CrossTowerOps):
 271   """Always do reduction to one device first and then do broadcasting.
 272 
 273     Batch reduction is done by reduction on each element one by one.
 274   """
 275 
 276   def __init__(self, reduce_to_device=None, accumulation_fn=math_ops.add_n):
 277     """Constructor.
 278 
 279     Args:
 280       reduce_to_device: the intermediate device to reduce to. If None, reduce
 281         to the first device in `destinations` of the reduce() method.
 282       accumulation_fn: a function that does accumulation.
 283     """
 284     self.reduce_to_device = reduce_to_device
 285     self.accumulation_fn = accumulation_fn
 286     super(ReductionToOneDeviceCrossTowerOps, self).__init__()
 287 
 288   def _reduce(self, aggregation, per_device_value, destinations):
 289     if check_destinations(destinations):
 290       devices = get_devices_from(destinations)
 291     else:
 292       devices = get_devices_from(per_device_value)
 293     reduce_to_device = self.reduce_to_device or devices[0]
 294     reduced = _simple_reduce(per_device_value, reduce_to_device,
 295                              self.accumulation_fn, aggregation)
 296     return self.broadcast(reduced, devices)
 297 
 298   def _batch_reduce(self, aggregation, value_destination_pairs):
 299     return [
 300         self._reduce(aggregation, t, destinations=v)
 301         for t, v in value_destination_pairs
 302     ]

 305 def _group_value_by_device(per_device_values):
 306   """Group values into sublists by their devices.
 307 
 308   This grouping is needed to call the all-reduce library because it expects a
 309   list of the following form:
 310     [[(grad0_gpu0, v0_gpu0), (grad1_gpu0, v1_gpu0), (grad2_gpu0, v2_gpu0) ...],
 311      [(grad0_gpu1, v0_gpu1), (grad1_gpu1, v1_gpu1), (grad2_gpu1, v2_gpu1) ...],
 312      [(grad0_gpu2, v0_gpu2), (grad1_gpu0, v1_gpu2), (grad2_gpu0, v2_gpu2) ...],
 313      ...
 314     ]
 315 
 316   Args:
 317     per_device_values: a list of PerDevice obejcts.
 318 
 319   Returns:
 320     a list of lists, each sublist has components for its corresponding device of
 321       PerDevice objects, paired with a None.
 322   """
 323   destinations = per_device_values[0].devices
 324   grouped = [[] for _ in range(len(destinations))]
 325   for per_device_value in per_device_values:
 326     # pylint: disable=protected-access
 327     for i, v in enumerate(per_device_value._index.values()):
 328       assert per_device_value.devices == destinations
 329       grouped[i].append((v, None))
 330   return grouped

 333 def _ungroup_and_make_mirrored(grouped_reduced,
 334                                destinations,
 335                                aggregation,
 336                                num_between_graph_workers=1):
 337   """Ungroup results from all-reduce and make Mirrored objects.
 338 
 339   Each all-reduce result will be divided by the number of destinations before
 340   Mirrored objects are created if aggregation is "mean".
 341 
 342   Args:
 343     grouped_reduced: a list of lists, each sublist has components for each
 344       device, paired with a None. It is the result from
 345       cross_tower_utils.aggregate_gradients_using*.
 346     destinations: a list of device strings for returned Mirrored objects.
 347     aggregation: Indicates how a variable will be aggregated. Accepted values
 348       are `tf.VariableAggregation.SUM`, `tf.VariableAggregation.MEAN`.
 349     num_between_graph_workers: number of workers in the between-graph
 350       replication.
 351 
 352   Returns:
 353     a list of Mirrored objects.
 354   """
 355   index = [{} for _ in range(len(grouped_reduced[0]))]
 356   for d, per_device_reduced in enumerate(grouped_reduced):
 357     for i, (v, _) in enumerate(per_device_reduced):
 358       if aggregation == vs.VariableAggregation.MEAN:
 359         index[i][destinations[d]] = v / (
 360             len(destinations) * num_between_graph_workers)
 361       else:
 362         index[i][destinations[d]] = v
 363   return [value_lib.Mirrored(v) for v in index]

 366 class ConcatAndSplitPacker(object):
 367   """Concatenate and split tensors for reduction."""
 368 
 369   def __init__(self, num_packs=1):
 370     """Initialize the ConcatAndSplitPacker object.
 371 
 372     Args:
 373       num_packs: specifies the number of split packs that will be
 374         formed.
 375 
 376     Raises:
 377       ValueError: if num_packs is not greater than 0.
 378     """
 379     if num_packs <= 0:
 380       raise ValueError("num_packs must be greater than zero.")
 381     self.num_packs = num_packs
 382 
 383   def pack(self, grouped_grads_and_vars):
 384     """Pack tensors."""
 385     self.grouped_grads_and_vars = grouped_grads_and_vars
 386     self.all_tower_shapes = []
 387     self.all_tower_sizes = []
 388 
 389     device_grad_packs = []
 390     for tower_grads_and_vars in grouped_grads_and_vars:
 391       with ops.colocate_with(tower_grads_and_vars[0][0]):
 392         # Flatten all the grads.
 393         flat_grads = [
 394             array_ops.reshape(g, [-1]) for g, _ in tower_grads_and_vars
 395         ]
 396         # Remember the original shape of all the grads.
 397         tower_shapes = [array_ops.shape(g) for g, _ in tower_grads_and_vars]
 398         # Remember the original sizes of all the grads.
 399         tower_sizes = [array_ops.size(g) for g, _ in tower_grads_and_vars]
 400         # Concat all the flat grads into a big flat tensor.
 401         concat_grads = array_ops.concat(flat_grads, 0)
 402 
 403         # Split the big tensor into num_splits packs. In cases where the
 404         # total size is not divisible num_splits, the last pack gets
 405         # more elements.
 406         # TODO(zhengxq): it is also possible to optimize away all the concat
 407         # as well.
 408         num_splits = self.num_packs
 409 
 410         # The array_ops.size function will sometimes remove static shapes. So if
 411         # all gradient shapes are defined, we use another method to get the
 412         # total size.
 413         # TODO(yuefengz): move this logic to array_ops.size.
 414         if all([g.shape.is_fully_defined() for g, _ in tower_grads_and_vars]):
 415           total_grad_size = sum(
 416               [g.shape.num_elements() for g, _ in tower_grads_and_vars])
 417         else:
 418           total_grad_size = array_ops.size(concat_grads)
 419 
 420         split_size = total_grad_size // num_splits
 421         split_size_last = total_grad_size - split_size * (num_splits - 1)
 422         split_sizes = [split_size] * (num_splits - 1) + [split_size_last]
 423         grad_packs = array_ops.split(concat_grads, split_sizes)
 424 
 425         # Ready to aggregate the repacked gradients, with fake variables.
 426         # TODO(zhengxq): It is hacky to have to use fake variables.
 427         # We should remove the need for variables in
 428         # aggregate_gradients_using*.
 429         device_grad_packs.append(zip(grad_packs, [None] * num_splits))
 430         self.all_tower_shapes.append(tower_shapes)
 431         self.all_tower_sizes.append(tower_sizes)
 432 
 433     return device_grad_packs
 434 
 435   def unpack(self, summed_device_grad_packs):
 436     """Reverse the pack."""
 437     aggregated_device_grads = []
 438     for (summed_tower_grad_packs,
 439          tower_grads_and_vars, tower_shapes, tower_sizes) in zip(
 440              summed_device_grad_packs, self.grouped_grads_and_vars,
 441              self.all_tower_shapes, self.all_tower_sizes):
 442       # pylint: enable=line-too-long
 443       # Reverse the packing operations in the previous steps. Form the
 444       # summed gradients back into their original shapes.
 445       with ops.colocate_with(summed_tower_grad_packs[0][0]):
 446         # Form a list of the summed grad packs.
 447         device_grad_packs = [g for g, _ in summed_tower_grad_packs]
 448 
 449         # Concat them back into a big flat tensor.
 450         device_grads_concat = array_ops.concat(device_grad_packs, 0)
 451 
 452         # Split the tensors back into their original sizes.
 453         grads_with_sizes = array_ops.split(device_grads_concat, tower_sizes)
 454 
 455         # Reshape the tensors back into their original shapes.
 456         grads_with_shapes = [
 457             array_ops.reshape(grad, shape)
 458             for shape, grad in zip(tower_shapes, grads_with_sizes)
 459         ]
 460 
 461         # Form the list with the original list of variables.
 462         summed_tower_grads = [
 463             (g, v) for g, (_, v) in zip(grads_with_shapes, tower_grads_and_vars)
 464         ]
 465         aggregated_device_grads.append(summed_tower_grads)
 466     return aggregated_device_grads

 469 class AggregateSmallTensorPacker(object):
 470   """Concatenate small gradient tensors together for reduction."""
 471 
 472   def __init__(self,
 473                agg_small_grads_max_bytes=1048576,
 474                agg_small_grads_max_group=16):
 475     """Initialize the AggregateSmallTensorPacker object.
 476 
 477     Args:
 478       agg_small_grads_max_bytes: largest tensor eligible for aggregation,
 479         in number of bytes.
 480       agg_small_grads_max_group: largest permitted aggregation of small
 481         tensors.
 482 
 483     Raises:
 484       ValueError: if `agg_small_grads_max_bytes` or `agg_small_grads_max_group`
 485         is not greater than 0.
 486     """
 487     if agg_small_grads_max_bytes <= 0 or agg_small_grads_max_group <= 0:
 488       raise ValueError("agg_small_grads_max_bytes and agg_small_grads_max_group"
 489                        " should both be greater than zero.")
 490     self.agg_small_grads_max_bytes = agg_small_grads_max_bytes
 491     self.agg_small_grads_max_group = agg_small_grads_max_group
 492 
 493   def pack(self, grouped_grads_and_vars):
 494     """Aggregate small tensors."""
 495     if (self.agg_small_grads_max_bytes > 0 and
 496         self.agg_small_grads_max_group > 0):
 497       tower_grads, self.packing = cross_tower_utils.pack_small_tensors(
 498           grouped_grads_and_vars,
 499           max_bytes=self.agg_small_grads_max_bytes,
 500           max_group=self.agg_small_grads_max_group)
 501     return tower_grads
 502 
 503   def unpack(self, summed_device_grad_packs):
 504     """Reverse the aggregation process."""
 505     return cross_tower_utils.unpack_small_tensors(summed_device_grad_packs,
 506                                                   self.packing)
 507

 509 def _pack_tensors(device_grads,
 510                   num_packs=0,
 511                   agg_small_grads_max_bytes=0,
 512                   agg_small_grads_max_group=0):
 513   """Pack tensors if specified."""
 514   if num_packs > 0:
 515     tensor_packer = ConcatAndSplitPacker(num_packs)
 516     device_grad_packs = tensor_packer.pack(device_grads)
 517   elif agg_small_grads_max_bytes > 0 and agg_small_grads_max_group > 0:
 518     tensor_packer = AggregateSmallTensorPacker(agg_small_grads_max_bytes,
 519                                                agg_small_grads_max_group)
 520     device_grad_packs = tensor_packer.pack(device_grads)
 521   else:
 522     tensor_packer = None
 523     device_grad_packs = device_grads
 524   return device_grad_packs, tensor_packer

 527 def _unpack_tensors(reduced, tensor_packer=None):
 528   """Unpack tensors if they are packed before all-reduce."""
 529   if tensor_packer:
 530     return tensor_packer.unpack(reduced)
 531   return reduced

 534 class AllReduceCrossTowerOps(CrossTowerOps):
 535   """Reduction using all reduce."""
 536 
 537   def __init__(self,
 538                all_reduce_alg="nccl",
 539                num_packs=1,
 540                agg_small_grads_max_bytes=0,
 541                agg_small_grads_max_group=10):
 542     """All-reduce implementation of CrossTowerOps.
 543 
 544     Before performing all-reduce, tensors will be repacked or aggregated for
 545     more efficient cross-device transportation:
 546       1) If `num_packs` is non-zero, pack values into
 547         `num_packs` splits.
 548       2) Otherwise, if `agg_small_grads_max_bytes` > 0 and
 549         `agg_small_grads_max_group` > 0, aggregate values smaller than
 550         `agg_small_grads_max_bytes` into groups with at most
 551         `agg_small_grads_max_group` values.
 552       3) Otherwise, no repacking or grouping will happen.
 553 
 554     Args:
 555       all_reduce_alg: the all-reduce algorithm to use, currently only "nccl" or
 556         "hierarchical_copy" are supported.
 557       num_packs: see above.
 558       agg_small_grads_max_bytes: see above.
 559       agg_small_grads_max_group: see above.
 560         tensors.
 561     """
 562     self._all_reduce_alg = all_reduce_alg
 563     self._num_packs = num_packs
 564     self._agg_small_grads_max_bytes = agg_small_grads_max_bytes
 565     self._agg_small_grads_max_group = agg_small_grads_max_group
 566     super(AllReduceCrossTowerOps, self).__init__()
 567 
 568   def _reduce(self, aggregation, per_device_value, destinations):
 569     contains_indexed_slices = cross_tower_utils.contains_indexed_slices(
 570         per_device_value)
 571     if (_devices_match(per_device_value, destinations)
 572         and not context.executing_eagerly()
 573         and not contains_indexed_slices):
 574       return self._batch_all_reduce(aggregation, [per_device_value])[0]
 575     else:
 576       if contains_indexed_slices:
 577         logging.log_first_n(
 578             logging.WARN,
 579             "Efficient allreduce is not supported for IndexedSlices.", 10)
 580 
 581       if check_destinations(destinations):
 582         devices = get_devices_from(destinations)
 583       else:
 584         devices = get_devices_from(per_device_value)
 585       reduce_to_device = devices[0]
 586       reduced = _simple_reduce(per_device_value, reduce_to_device,
 587                                math_ops.add_n, aggregation)
 588       return self.broadcast(reduced, devices)
 589 
 590   def _batch_reduce(self, aggregation, value_destination_pairs):
 591     all_devices_match = _all_devices_match(value_destination_pairs)
 592     contains_indexed_slices = cross_tower_utils.contains_indexed_slices(
 593         value_destination_pairs)
 594     if (all_devices_match and not context.executing_eagerly()
 595         and not contains_indexed_slices):
 596       return self._batch_all_reduce(aggregation,
 597                                     [v[0] for v in value_destination_pairs])
 598     else:
 599       if not all_devices_match:
 600         logging.log_first_n(logging.WARN,
 601                             "Efficient batch_reduce is not supported if "
 602                             "destinations are different.",
 603                             10)
 604 
 605       return [
 606           self._reduce(aggregation, t, destinations=v)
 607           for t, v in value_destination_pairs
 608       ]

 610   def _batch_all_reduce(self, aggregation, per_device_values):
 611     """All reduce algorithm in a batch."""
 612     logging.log_first_n(
 613         logging.INFO, "batch_all_reduce invoked for batches size = %d with "
 614         "algorithm = %s, num_packs = %d, agg_small_grads_max_bytes = %d and "
 615         "agg_small_grads_max_group = %d" %
 616         (len(per_device_values), self._all_reduce_alg, self._num_packs,
 617          self._agg_small_grads_max_bytes, self._agg_small_grads_max_group), 10)
 618     destinations = per_device_values[0].devices
 619     grouped = _group_value_by_device(per_device_values)
 620 
 621     device_grad_packs, tensor_packer = _pack_tensors(
 622         grouped, self._num_packs, self._agg_small_grads_max_bytes,
 623         self._agg_small_grads_max_group)
 624 
 625     # The actual aggregation of the repacked gradients. Note that they are
 626     # sharded among different aggregation trees. So it is important to strike
 627     # the balance on num_splits.
 628     if self._all_reduce_alg == "nccl":
 629       # TODO(yuefengz): merge this into the all-reduce library.
 630       reduced = cross_tower_utils.aggregate_gradients_using_nccl(
 631           device_grad_packs)
 632     else:
 633       # TODO(yuefengz): check that gpu ids in `destinations` are in ascending
 634       # order.
 635       reduced = (
 636           cross_tower_utils.aggregate_gradients_using_hierarchical_copy(
 637               destinations, device_grad_packs))
 638 
 639     reduced = _unpack_tensors(reduced, tensor_packer)
 640     return _ungroup_and_make_mirrored(reduced, per_device_values[0].devices,
 641                                       aggregation)

 648 class MultiWorkerAllReduce(AllReduceCrossTowerOps):
 649   """All-reduce algorithms for distributed TensorFlow."""
 650 
 651   def __init__(self,
 652                worker_devices,
 653                num_gpus_per_worker,
 654                all_reduce_spec=("pscpu/pscpu", 2, -1),
 655                num_packs=0,
 656                agg_small_grads_max_bytes=0,
 657                agg_small_grads_max_group=10):
 658     """Initialize the all-reduce algorithm.
 659 
 660     Args:
 661       worker_devices: a list of device strings for workers participating in
 662         all-reduce.
 663       num_gpus_per_worker: number of GPU devices per worker.
 664       all_reduce_spec: a tuple or a named tuple or a list of tuples specifying
 665         the all-reduce algorithm.
 666         1. The first element of a tuple is the name of the all-reduce algorithm.
 667         Valid algorithm names are: "nccl", "nccl/xring", "nccl/rechd",
 668         "nccl/pscpu", "xring", "pscpu", "psgpu", "pscpu/pscpu". Algorithms with
 669         a "/" are hierarchical, so two all-reduces are executed, the first one
 670         aggregates tensors within a worker and the second aggregates across
 671         workers.
 672         2. The second element of a tuple is the number of shards when doing
 673         all-reduce. Let's say its values is M, each tensor after packing will be
 674         split into M shards and then M parallel all-reduces would be performed
 675         before finally they are concatenated backed into a complete tensor.
 676         3. The third element is the maximum size of tensors that will be
 677         applicable for the algorithm specified by the first element. For
 678         example, if all_reduce_spec=[("nccl", 2, 1024), ("pscpu/pscpu", 2, -1)],
 679         tensors with size not larger than 1024 bytes will be applied a 2-shard
 680         "nccl" all-reduce and other tensors will be applied a 2-shard
 681         "pscpu/pscpu" algorithm. The third elements should be in increasing
 682         order across tuples and end with -1 which indicates infinity.
 683       num_packs: see AllReduceCrossTowerOps.
 684       agg_small_grads_max_bytes: see AllReduceCrossTowerOps.
 685       agg_small_grads_max_group: see AllReduceCrossTowerOps.
 686     """
 687     self._worker_devices = worker_devices
 688     self._num_gpus_per_worker = num_gpus_per_worker
 689     super(MultiWorkerAllReduce, self).__init__(
 690         num_packs=num_packs,
 691         agg_small_grads_max_bytes=agg_small_grads_max_bytes,
 692         agg_small_grads_max_group=agg_small_grads_max_group)
 693 
 694     def validate_and_complete_spec(spec):
 695       """Validate and complete the all-reduce spec."""
 696       # TODO(yuefengz): support namedtuple.
 697       if not isinstance(spec, tuple):
 698         raise ValueError(
 699             "A tuple is expected for all-reduce spec: %r" % all_reduce_spec)
 700       if not spec or len(spec) > 3:
 701         raise ValueError(
 702             "Too many elements in the all-reduce spec tuple: %r" % spec)
 703       if len(spec) == 1:
 704         return AllReduceSpecTuple(spec[0], 1, -1)
 705       elif len(spec) == 2:
 706         return AllReduceSpecTuple(spec[0], spec[1], -1)
 707       else:
 708         return AllReduceSpecTuple(*spec)
 709 
 710     self._all_reduce_spec = []
 711     if isinstance(all_reduce_spec, six.string_types):
 712       self._all_reduce_spec.append(AllReduceSpecTuple(all_reduce_spec, 1, -1))
 713     elif isinstance(all_reduce_spec, tuple):
 714       self._all_reduce_spec.append(validate_and_complete_spec(all_reduce_spec))
 715     elif isinstance(all_reduce_spec, list):
 716       self._all_reduce_spec = [
 717           validate_and_complete_spec(spec) for spec in all_reduce_spec
 718       ]
 719 
 720   def _batch_all_reduce(self, aggregation, per_device_values):
 721     """All reduce algorithm in a batch."""
 722     logging.log_first_n(
 723         logging.INFO,
 724         "distributed batch_all_reduce invoked for batches size = %d with "
 725         "allreduce_spec = %r, num_packs = %d, agg_small_grads_max_bytes = %d "
 726         "and agg_small_grads_max_group = %d" %
 727         (len(per_device_values), self._all_reduce_spec, self._num_packs,
 728          self._agg_small_grads_max_bytes, self._agg_small_grads_max_group), 10)
 729 
 730     destinations = sorted(per_device_values[0].devices)
 731     device_grads = _group_value_by_device(per_device_values)
 732 
 733     # The all reduce library requires fully defined shapes.
 734     # TODO(yuefengz): when tensor sharding is not needed, static shapes are not
 735     # required as well.
 736     for device_grad in device_grads:
 737       for grad, _ in device_grad:
 738         if not grad.shape.is_fully_defined():
 739           raise ValueError("Shape is unknown for node %r" % grad)
 740 
 741     remaining_grads = device_grads
 742     aggregated_grads = []
 743     for spec_tuple in self._all_reduce_spec:
 744       if spec_tuple.limit < 0:
 745         this_grads = remaining_grads
 746         remaining_grads = []
 747       else:
 748         (this_grads, remaining_grads) = cross_tower_utils.split_grads_by_size(
 749             spec_tuple.limit, remaining_grads)
 750       if this_grads:
 751         device_grad_packs, tensor_packer = _pack_tensors(
 752             this_grads, self._num_packs, self._agg_small_grads_max_bytes,
 753             self._agg_small_grads_max_group)
 754         range_agg_grads = cross_tower_utils.sum_gradients_all_reduce(
 755             self._worker_devices, device_grad_packs, len(self._worker_devices),
 756             spec_tuple.alg, spec_tuple.shards, range(self._num_gpus_per_worker))
 757         range_agg_grads = _unpack_tensors(range_agg_grads, tensor_packer)
 758 
 759         if not aggregated_grads:
 760           aggregated_grads = range_agg_grads
 761         else:
 762           assert len(aggregated_grads) == len(range_agg_grads)
 763           for i in range(len(aggregated_grads)):
 764             aggregated_grads[i] += range_agg_grads[i]
 765     assert not remaining_grads
 766 
 767     return _ungroup_and_make_mirrored(aggregated_grads, destinations,
 768                                       aggregation)

 771 # TODO(yuefengz): support in-graph collective all-reduce.
 772 class CollectiveAllReduce(CrossTowerOps):
 773   """All-reduce cross tower ops using collective ops.
 774 
 775   In the between-graph replicated training, it will still do all-reduces across
 776   all workers and then put results on the right destinations.
 777   """
 778 
 779   def __init__(self,
 780                num_workers=1,
 781                num_gpus_per_worker=0,
 782                all_reduce_merge_scope=32,
 783                collective_keys=None):
 784     """Initializes the object.
 785 
 786     Args:
 787       num_workers: number of workers in the between-graph replicated training.
 788       num_gpus_per_worker: number of GPUs per worker.
 789       all_reduce_merge_scope: size of groups into which to partition consecutive
 790         gradients grouped under a common 'allreduce' name scope. This is useful
 791         for some optimization of collective ops.
 792       collective_keys: an optional CollectiveKey object.
 793     """
 794     self._num_workers = num_workers
 795     self._num_gpus_per_worker = num_gpus_per_worker
 796     self._all_reduce_merge_scope = all_reduce_merge_scope
 797     self._collective_keys = collective_keys or cross_tower_utils.CollectiveKeys(
 798     )
 799     super(CollectiveAllReduce, self).__init__()
 800 
 801   # TODO(yuefengz, tucker): is indexed slices supported by collective ops?
 802   def _reduce(self, aggregation, per_device_value, destinations):
 803     if cross_tower_utils.contains_indexed_slices(per_device_value):
 804       raise ValueError(
 805           "`IndexSlices` is not supported for Collective All-Reduce.")
 806     if context.executing_eagerly():
 807       raise ValueError(
 808           "Eager execution is not supported for Collective All-Reduce")
 809 
 810     all_reduced = self._batch_all_reduce(aggregation, [per_device_value])[0]
 811     if _devices_match(per_device_value, destinations):
 812       return all_reduced
 813     else:
 814       index = {}
 815       for d in get_devices_from(destinations):
 816         # pylint: disable=protected-access
 817         if d in all_reduced._index:
 818           index[d] = all_reduced._index[d]
 819         else:
 820           with ops.control_dependencies(list(
 821               all_reduced._index.values())), ops.device(d):
 822             index[d] = array_ops.identity(list(all_reduced._index.values())[0])
 823 
 824       return value_lib.Mirrored(index)

 826   def _batch_reduce(self, aggregation, value_destination_pairs):
 827     if cross_tower_utils.contains_indexed_slices(value_destination_pairs):
 828       raise ValueError(
 829           "`IndexSlices` is not supported for Collective All-Reduce.")
 830     if context.executing_eagerly():
 831       raise ValueError(
 832           "Eager execution is not supported for Collective All-Reduce")
 833 
 834     all_devices_match = _all_devices_match(value_destination_pairs)
 835     if all_devices_match:
 836       return self._batch_all_reduce(aggregation,
 837                                     [v[0] for v in value_destination_pairs])
 838     else:
 839       if not all_devices_match:
 840         logging.log_first_n(
 841             logging.WARN, "Efficient batch_reduce is not supported if "
 842             "destinations are different.", 10)
 843 
 844       return [
 845           self._reduce(aggregation, t, destinations=v)
 846           for t, v in value_destination_pairs
 847       ]

 849   def _batch_all_reduce(self, aggregation, per_device_values):
 850     """All-reduce across all workers in a batch."""
 851     if context.executing_eagerly():
 852       raise ValueError(
 853           "Eager execution with collective ops is not supported yet.")
 854 
 855     logging.log_first_n(
 856         logging.INFO, "Collective All-reduce invoked with batches size = %d, "
 857         "num_workers = %d" % (len(per_device_values), self._num_workers), 10)
 858 
 859     grouped_by_tower = _group_value_by_device(per_device_values)
 860 
 861     grouped_by_var = list(zip(*grouped_by_tower))
 862     # grouped_by_var is grouped by variables and takes the following format:
 863     # [((grad0_gpu0, v0_gpu0), (grad0_gpu1, v0_gpu1), (grad0_gpu2, v0_gpu2) ..),
 864     #  ((grad1_gpu0, v1_gpu0), (grad1_gpu1, v1_gpu1), (grad1_gpu0, v1_gpu2) ..),
 865     #  ((grad2_gpu0, v2_gpu0), (grad2_gpu1, v2_gpu1), (grad2_gpu0, v2_gpu2) ..),
 866     #  ...
 867     # ]
 868     chunked_gv = [
 869         grouped_by_var[x:x + self._all_reduce_merge_scope]
 870         for x in range(0, len(grouped_by_var), self._all_reduce_merge_scope)
 871     ]
 872 
 873     reduced_gv_list = []
 874     for chunk in chunked_gv:
 875       with ops.name_scope("allreduce"):
 876         for grad_and_vars in chunk:
 877           scaled_grads = [g for g, _ in grad_and_vars]
 878           collective_reduced = cross_tower_utils.build_collective_reduce(
 879               scaled_grads, self._num_workers, self._collective_keys, "Add",
 880               "Id")
 881           result = []
 882           for (_, v), g in zip(grad_and_vars, collective_reduced):
 883             result.append([g, v])
 884           reduced_gv_list.append(result)
 885 
 886     new_tower_grads = [list(x) for x in zip(*reduced_gv_list)]
 887     return _ungroup_and_make_mirrored(
 888         new_tower_grads,
 889         per_device_values[0].devices,
 890         aggregation,
 891         num_between_graph_workers=self._num_workers)

 898 def _has_dgx1_like_links(gpu_links):
 899   if not gpu_links:
 900     return False
 901   # TODO(yuefengz): figure out the right topology for hierarchial copy if
 902   # number of gpus are less than 8.
 903   if len(gpu_links) < 8:
 904     return False
 905   for i, (gpu_link, dgx1_link) in enumerate(zip(gpu_links, _dgx1_links)):
 906     if (set(gpu_link) != set(dgx1_link) and
 907         set(gpu_link) != set(dgx1_link + [i])):
 908       return False
 909   return True

 923 def choose_the_best(devices, session_config=None):
 924   """Find the best subclass of CrossTowerOps given a tensorflow session.
 925 
 926   Args:
 927     devices: a list of devices passed for distribute strategy.
 928     session_config: a tensorflow session config or None. If None, it will make
 929       deciesion based on all local devices.
 930 
 931   Returns:
 932     a subclass of CrossTowerOps.
 933   """
 934   requested_devices = set([device_util.canonicalize(d) for d in devices])
 935   machine_devices = device_lib.list_local_devices(session_config=session_config)
 936   using_devices = []
 937   for d in machine_devices:
 938     if device_util.canonicalize(d.name) in requested_devices:
 939       using_devices.append(d)
 940     else:
 941       logging.info(
 942           "Device is available but not used by distribute strategy: %s", d.name)
 943 
 944   if len(using_devices) != len(requested_devices):
 945     logging.warning("Not all devices in distribute strategy are visible by "
 946                     "TensorFlow sessions.")
 947     return ReductionToOneDeviceCrossTowerOps()
 948 
 949   if any([d.device_type.lower() != "gpu" for d in using_devices]):
 950     logging.warning("Not all devices in DistributionStrategy are visible to "
 951                     "TensorFlow session.")
 952     return ReductionToOneDeviceCrossTowerOps()
 953 
 954   device_links = [[] for _ in range(len(using_devices))]
 955   for i, device in enumerate(using_devices):
 956     for link in device.locality.links.link:
 957       device_links[i].append(link.device_id)
 958 
 959   return _choose_all_reduce_algorithm(device_links)
