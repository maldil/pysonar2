  52 class _CudnnRNN(base_layer.Layer):
  53   # pylint:disable=line-too-long
  54   """Abstract class for RNN layers with Cudnn implementation.
  55 
  56   Cudnn RNNs have two major differences from other platform-independent RNNs tf
  57   provides:
  58   * Cudnn LSTM and GRU are mathematically different from their tf counterparts.
  59     (e.g. `tf.contrib.rnn.LSTMBlockCell` and `tf.nn.rnn_cell.GRUCell`.
  60   * Cudnn-trained checkpoints are not directly compatible with tf RNNs:
  61     * They use a single opaque parameter buffer for the entire (possibly)
  62       multi-layer multi-directional RNN; Whereas tf RNN weights are per-cell and
  63       layer.
  64     * The size and layout of the parameter buffers may change between
  65       CUDA/CuDNN/GPU generations. Because of that, the opaque parameter variable
  66       does not have a static shape and is not partitionable. Instead of using
  67       partitioning to alleviate the PS's traffic load, try building a
  68       multi-tower model and do gradient aggregation locally within the host
  69       before updating the PS. See https://www.tensorflow.org/performance/performance_models#parameter_server_variables
  70       for a detailed performance guide.
  71 
  72   Consequently, if one plans to use Cudnn trained models on both GPU and CPU
  73   for inference and training, one needs to:
  74   * Create a CudnnOpaqueParamsSaveable subclass object to save RNN params in
  75     canonical format. (This is done for you automatically during layer building
  76     process.)
  77   * When not using a Cudnn RNN class, use CudnnCompatibleRNN classes to load the
  78     checkpoints. These classes are platform-independent and perform the same
  79     computation as Cudnn for training and inference.
  80   Similarly, CudnnCompatibleRNN-trained checkpoints can be loaded by CudnnRNN
  81   classes seamlessly.
  82 
  83   Below is a typical workflow(using LSTM as an example):
  84   for detailed performance guide.
  85 
  86   # Use Cudnn-trained checkpoints with CudnnCompatibleRNNs
  87   ```python
  88   with tf.Graph().as_default():
  89     lstm = CudnnLSTM(num_layers, num_units, direction, ...)
  90 
  91     outputs, output_states = lstm(inputs, initial_states, training=True)
  92 
  93     # If user plans to delay calling the cell with inputs, one can do
  94     # lstm.build(input_shape)
  95 
  96     saver = Saver()
  97 
  98     # training subgraph
  99     ...
 100 
 101     # Once in a while save the model.
 102     saver.save(save_path)
 103 
 104   # Inference subgraph for unidirectional RNN on, e.g., CPU or mobile.
 105   with tf.Graph().as_default():
 106     single_cell = lambda: tf.contrib.cudnn_rnn.CudnnCompatibleLSTM(num_units)
 107 
 108     # NOTE: Even if there's only one layer, the cell needs to be wrapped in
 109     # MultiRNNCell.
 110     cell = tf.nn.rnn_cell.MultiRNNCell(
 111       [single_cell() for _ in range(num_layers)])
 112 
 113     # Leave the scope arg unset.
 114     outputs, final_state = tf.nn.dynamic_rnn(cell, inputs, initial_state, ...)
 115 
 116     saver = Saver()
 117 
 118     # Create session
 119     sess = ...
 120 
 121     # Restores
 122     saver.restore(sess, save_path)
 123 
 124   # Inference subgraph for bidirectional RNN
 125   with tf.Graph().as_default():
 126     single_cell = lambda: tf.contrib.cudnn_rnn.CudnnCompatibleLSTM(num_units)
 127     cells_fw = [single_cell() for _ in range(num_layers)]
 128     cells_bw = [single_cell() for _ in range(num_layers)]
 129 
 130     # Leave the scope arg unset.
 131     (outputs, output_state_fw,
 132      output_state_bw) = tf.contrib.rnn.stack_bidirectional_dynamic_rnn(
 133          cells_fw, cells_bw, inputs, ...)
 134     saver = Saver()
 135 
 136     # Create session
 137     sess = ...
 138 
 139     # Restores
 140     saver.restore(sess, save_path)
 141   ```
 142   """

 156   def __init__(self,
 157                num_layers,
 158                num_units,
 159                input_mode=CUDNN_INPUT_LINEAR_MODE,
 160                direction=CUDNN_RNN_UNIDIRECTION,
 161                dropout=0.,
 162                seed=None,
 163                dtype=dtypes.float32,
 164                kernel_initializer=None,
 165                bias_initializer=None,
 166                name=None):
 167     """Creates a CudnnRNN model from model spec.
 168 
 169     Args:
 170       num_layers: the number of layers for the RNN model.
 171       num_units: the number of units within the RNN model.
 172       input_mode: indicate whether there is a linear projection between the
 173           input and the actual computation before the first layer. It can be
 174           'linear_input', 'skip_input' or 'auto_select'.
 175           'linear_input' (default) always applies a linear projection of input
 176           onto RNN hidden state. (standard RNN behavior).
 177           'skip_input' is only allowed when input_size == num_units;
 178           'auto_select' implies 'skip_input' when input_size == num_units;
 179           otherwise, it implies 'linear_input'.
 180       direction: the direction model that the model operates. Can be either
 181           'unidirectional' or 'bidirectional'
 182       dropout: dropout rate, a number between [0, 1]. Dropout is applied between
 183           each layer (no dropout is applied for a model with a single layer).
 184           When set to 0, dropout is disabled.
 185       seed: the op seed used for initializing dropout. See `tf.set_random_seed`
 186           for behavior.
 187       dtype: tf.float16, tf.float32 or tf.float64
 188       kernel_initializer: starting value to initialize the weight.
 189       bias_initializer: starting value to initialize the bias
 190         (default is all zeros).
 191       name: VariableScope for the created subgraph; defaults to class name.
 192         This only serves the default scope if later no scope is specified when
 193         invoking __call__().
 194 
 195     Raises:
 196       ValueError: if direction is invalid. Or dtype is not supported.
 197     """
 198     super(_CudnnRNN, self).__init__(dtype=dtype, name=name)
 199     cudnn_rnn_ops.check_direction(direction)
 200     cudnn_rnn_ops.check_input_mode(input_mode)
 201 
 202     if dtype not in [dtypes.float16, dtypes.float32, dtypes.float64]:
 203       raise ValueError(
 204           "Only support float16, float32, float64, provided %s" % dtype)
 205     # Layer self.dtype is type name, the original DType object is kept here.
 206     self._plain_dtype = dtype
 207     self._num_layers = num_layers
 208     self._num_units = num_units
 209     self._input_mode = input_mode
 210     self._direction = direction
 211     self._dropout = dropout
 212     self._seed = seed
 213     self._kernel_initializer = kernel_initializer
 214     self._bias_initializer = bias_initializer
 215     # Init input_size to None, which will be set after build().
 216     self._input_size = None
 217     self._saveable = None
 218

 291   def _update_trainable_weights(self, getter, *args, **kwargs):
 292     """Custom getter for layer variables."""
 293     # Add variables to layer's `(non_)trainable_weights` list(s).
 294     variable = getter(*args, **kwargs)
 295     trainable = kwargs.get("trainable", True)
 296     if trainable and variable not in self._trainable_weights:
 297       self._trainable_weights.append(variable)
 298     elif not trainable and variable not in self._non_trainable_weights:
 299       self._non_trainable_weights.append(variable)
 300     return variable

 302   def build(self, input_shape):
 303     """Create variables of the Cudnn RNN.
 304 
 305     It can be called manually before `__call__()` or automatically through
 306     `__call__()`. In the former case, subsequent `__call__()`s will skip
 307     creating variables.
 308     Args:
 309       input_shape: network input tensor shape, a python list or a TensorShape
 310         object with 3 dimensions.
 311     Raises:
 312       ValueError: if input_shape has wrong dimension or unknown 3rd dimension.
 313     """
 314     if self.built:
 315       return
 316 
 317     input_shape = tensor_shape.TensorShape(input_shape)
 318     if input_shape.ndims != 3:
 319       raise ValueError("Expecting input_shape with 3 dims, got %d" %
 320                        input_shape.ndims)
 321     if input_shape[-1].value is None:
 322       raise ValueError("The last dimension of the inputs to `CudnnRNN` "
 323                        "should be defined. Found `None`.")
 324     self._input_size = input_shape[-1].value
 325     self.input_spec = base_layer.InputSpec(ndim=3, axes={-1: self._input_size})
 326 
 327     self._set_scope(None)
 328 
 329     # Not using base class `add_variable()` since the it calls
 330     # `tf.get_variable()` with a callable initializer whereas here with a
 331     # tensor. The difference is mandated to support forward-compatibility with
 332     # Cudnn.
 333     with vs.variable_scope(
 334         self._scope,
 335         reuse=self.built,
 336         custom_getter=self._update_trainable_weights):
 337       if self._kernel_initializer is None:
 338         self._kernel_initializer = init_ops.glorot_uniform_initializer(
 339             seed=self._seed, dtype=self._plain_dtype)
 340       if self._bias_initializer is None:
 341         self._bias_initializer = init_ops.constant_initializer(
 342             0.0, dtype=self._plain_dtype)
 343 
 344       weights = [
 345           self._kernel_initializer(sp, dtype=self._plain_dtype)
 346           for sp in self.canonical_weight_shapes
 347       ]
 348       biases = [
 349           self._bias_initializer(sp, dtype=self._plain_dtype)
 350           for sp in self.canonical_bias_shapes
 351       ]
 352       opaque_params_t = self._canonical_to_opaque(weights, biases)
 353 
 354       if vs.get_variable_scope().partitioner is not None:
 355         logging.warn(
 356             "Partitioner is not supported for Cudnn RNN layer variables, using "
 357             "it will create forward-compatibility issues with future "
 358             "CUDA/CuDNN generations.")
 359       # Initialize opaque params with a tensor.
 360       self.kernel = vs.get_variable(
 361           "opaque_kernel", dtype=self._plain_dtype,
 362           initializer=opaque_params_t, validate_shape=False)
 363     # Create saveable in the outer scope of the cudnn subgraph, such that
 364     # alternative subgraph with platform-independent rnn cells can load the
 365     # checkpoints directly.
 366     if not (self.built or vs.get_variable_scope().reuse is True):
 367       self._create_saveable()
 368     self.built = True

 375   def call(self, inputs, initial_state=None, training=True):
 376     """Runs the forward step for the RNN model.
 377 
 378     Args:
 379       inputs: `3-D` tensor with shape `[time_len, batch_size, input_size]`.
 380       initial_state: a tuple of tensor(s) of shape
 381         `[num_layers * num_dirs, batch_size, num_units]`. If not provided, use
 382         zero initial states. The tuple size is 2 for LSTM and 1 for other RNNs.
 383       training: whether this operation will be used in training or inference.
 384     Returns:
 385       output: a tensor of shape `[time_len, batch_size, num_dirs * num_units]`.
 386         It is a `concat([fwd_output, bak_output], axis=2)`.
 387       output_states: a tuple of tensor(s) of the same shape and structure as
 388         `initial_state`.
 389     Raises:
 390       ValueError: initial_state is not a tuple.
 391     """
 392     if initial_state is not None and not isinstance(initial_state, tuple):
 393       raise ValueError("Invalid initial_state type: %s, expecting tuple.",
 394                        type(initial_state))
 395     dtype = self.dtype
 396     inputs = ops.convert_to_tensor(inputs, dtype=dtype)
 397 
 398     batch_size = array_ops.shape(inputs)[1]
 399     if initial_state is None:
 400       initial_state = self._zero_state(batch_size)
 401     if self._rnn_mode == CUDNN_LSTM:
 402       h, c = initial_state  # pylint:disable=unbalanced-tuple-unpacking,unpacking-non-sequence
 403     else:
 404       h, = initial_state  # pylint:disable=unbalanced-tuple-unpacking,unpacking-non-sequence
 405     h = ops.convert_to_tensor(h, dtype=dtype)
 406     if self._rnn_mode == CUDNN_LSTM:
 407       c = ops.convert_to_tensor(c, dtype=dtype)
 408     else:
 409       # For model that doesn't take input_c, replace with a dummy tensor.
 410       c = array_ops.constant([], dtype=dtype)
 411     outputs, (output_h, output_c) = self._forward(inputs, h, c, self.kernel,
 412                                                   training)
 413     if self._rnn_mode == CUDNN_LSTM:
 414       return outputs, (output_h, output_c)
 415     else:
 416       return outputs, (output_h,)

 421   def _zero_state(self, batch_size):
 422     res = []
 423     for sp in self.state_shape(batch_size):
 424       res.append(array_ops.zeros(sp, dtype=self.dtype))
 425     return tuple(res)

 427   def _canonical_weight_shape(self, layer):
 428     """Shapes of Cudnn canonical weight tensors for given layer."""
 429     if layer < 0 or layer >= self._num_layers:
 430       raise ValueError("\'layer\' is not valid, got %s, expecting [%d, %d]" %
 431                        (layer, 0, self._num_layers-1))
 432     if not self._input_size:
 433       raise RuntimeError(
 434           "%s._canonical_weight_shape invoked before input shape is known" %
 435           type(self).__name__)
 436 
 437     input_size = self._input_size
 438     num_units = self._num_units
 439     num_gates = self._num_params_per_layer // 2
 440     is_bidi = self._direction == CUDNN_RNN_BIDIRECTION
 441 
 442     if layer == 0:
 443       wts_applied_on_inputs = [(num_units, input_size)] * num_gates
 444     else:
 445       if is_bidi:
 446         wts_applied_on_inputs = [(num_units, 2 * num_units)] * num_gates
 447       else:
 448         wts_applied_on_inputs = [(num_units, num_units)] * num_gates
 449     wts_applied_on_hidden_states = [(num_units, num_units)] * num_gates
 450     tf_wts = wts_applied_on_inputs + wts_applied_on_hidden_states
 451     return tf_wts if not is_bidi else tf_wts * 2
 452 
 453   def _canonical_bias_shape(self, unused_layer):
 454     """Shapes of Cudnn canonical bias tensors for given layer."""
 455     num_dirs = 1 if self._direction == CUDNN_RNN_UNIDIRECTION else 2
 456     return [[self._num_units]] * num_dirs * self._num_params_per_layer
 457 
 458   def _canonical_to_opaque(self, cu_weights, cu_biases):
 459     if not self._input_size:
 460       raise RuntimeError(
 461           "%s._canonical_to_opaque invoked before input shape is known" %
 462           type(self).__name__)
 463     with ops.device("/gpu:0"):
 464       return cudnn_rnn_ops.cudnn_rnn_canonical_to_opaque_params(
 465           rnn_mode=self._rnn_mode,
 466           num_layers=self._num_layers,
 467           num_units=self._num_units,
 468           input_size=self._input_size,
 469           weights=cu_weights,
 470           biases=cu_biases,
 471           input_mode=self._input_mode,
 472           seed=self._seed,
 473           dropout=self._dropout,
 474           direction=self._direction)
 475 
 476   def _forward(self, inputs, h, c, opaque_params, training):
 477     output, output_h, output_c = cudnn_rnn_ops._cudnn_rnn(  # pylint:disable=protected-access
 478         inputs,
 479         h,
 480         c,
 481         opaque_params,
 482         training,
 483         self._rnn_mode,
 484         input_mode=self._input_mode,
 485         direction=self._direction,
 486         dropout=self._dropout,
 487         seed=self._seed)
 488     return output, (output_h, output_c)

 490   def _create_saveable(self):
 491     """Create custom saveable for the Cudnn layer.
 492 
 493     Called during layer building process to make sharing checkpoints between
 494     Cudnn and Cudnn-compatible RNNs easy.
 495     Returns:
 496       a `CudnnOpaqueParamsSaveable` object.
 497     Raises:
 498       RuntimeError: if any custom saveable is already created for this layer.
 499     """
 500     if self._saveable is not None:
 501       raise RuntimeError("Cudnn saveable already created.")
 502     self._saveable = self._saveable_cls(  # pylint:disable=not-callable
 503         opaque_params=self.trainable_variables[0],
 504         num_layers=self.num_layers,
 505         num_units=self.num_units,
 506         input_size=self.input_size,
 507         input_mode=self.input_mode,
 508         direction=self.direction,
 509         scope=vs.get_variable_scope(),
 510         name="%s_saveable" % self.trainable_variables[0].name.split(":")[0])
 511     self._saveable._add_checkpointable_dependencies(  # pylint: disable=protected-access
 512         checkpointable=self, dtype=self._plain_dtype)
 513     ops.add_to_collection(ops.GraphKeys.SAVEABLE_OBJECTS, self._saveable)

 516 class CudnnLSTM(_CudnnRNN):

 518   _rnn_mode = CUDNN_LSTM

 522   def state_shape(self, batch_size):
 523     """Shape of Cudnn LSTM states.
 524 
 525     Shape is a 2-element tuple. Each is
 526     [num_layers * num_dirs, batch_size, num_units]
 527     Args:
 528       batch_size: an int
 529     Returns:
