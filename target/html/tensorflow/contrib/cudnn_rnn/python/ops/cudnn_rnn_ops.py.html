  76 class CudnnCompatibleGRUCell(rnn_cell_impl.GRUCell):
  77   """Cudnn Compatible GRUCell.
  78 
  79   A GRU impl akin to `tf.nn.rnn_cell.GRUCell` to use along with
  80   `tf.contrib.cudnn_rnn.CudnnGRU`. The latter's params can be used by
  81   it seamlessly.
  82 
  83   It differs from platform-independent GRUs in how the new memory gate is
  84   calculated. Nvidia picks this variant based on GRU author's[1] suggestion and
  85   the fact it has no accuracy impact[2].
  86   [1] https://arxiv.org/abs/1406.1078
  87   [2] http://svail.github.io/diff_graphs/
  88 
  89   Cudnn compatible GRU (from Cudnn library user guide):
  90   ```python
  91   # reset gate
  92   $$r_t = \sigma(x_t * W_r + h_t-1 * R_h + b_{Wr} + b_{Rr})$$
  93   # update gate
  94   $$u_t = \sigma(x_t * W_u + h_t-1 * R_u + b_{Wu} + b_{Ru})$$
  95   # new memory gate
  96   $$h'_t = tanh(x_t * W_h + r_t .* (h_t-1 * R_h + b_{Rh}) + b_{Wh})$$
  97   $$h_t = (1 - u_t) .* h'_t + u_t .* h_t-1$$
  98   ```
  99 
 100   Other GRU (see `tf.nn.rnn_cell.GRUCell` and `tf.contrib.rnn.GRUBlockCell`):
 101   ```python
 102   # new memory gate
 103   \\(h'_t = tanh(x_t * W_h + (r_t .* h_t-1) * R_h + b_{Wh})\\)
 104   ```
 105   which is not equivalent to Cudnn GRU: in addition to the extra bias term b_Rh,
 106   ```python
 107   \\(r .* (h * R) != (r .* h) * R\\)
 108   ```
 109   """
 110 
 111   def __init__(self, num_units, reuse=None, kernel_initializer=None):
 112     super(CudnnCompatibleGRUCell, self).__init__(
 113         num_units,
 114         activation=None,
 115         reuse=reuse,
 116         kernel_initializer=kernel_initializer)
 117 
 118   def build(self, inputs_shape):
 119     if inputs_shape[1].value is None:
 120       raise ValueError("Expected inputs.shape[-1] to be known, saw shape: %s"
 121                        % inputs_shape)
 122 
 123     input_depth = inputs_shape[1].value
 124     self._gate_kernel = self.add_variable(
 125         "gates/%s" % _WEIGHTS_VARIABLE_NAME,
 126         shape=[input_depth + self._num_units, 2 * self._num_units],
 127         initializer=self._kernel_initializer)
 128     self._gate_bias = self.add_variable(
 129         "gates/%s" % _BIAS_VARIABLE_NAME,
 130         shape=[2 * self._num_units],
 131         initializer=(
 132             self._bias_initializer
 133             if self._bias_initializer is not None
 134             else init_ops.constant_initializer(1.0, dtype=self.dtype)))
 135 
 136     self._candidate_input_kernel = self.add_variable(
 137         "candidate/input_projection/%s" % _WEIGHTS_VARIABLE_NAME,
 138         shape=[input_depth, self._num_units],
 139         initializer=self._kernel_initializer)
 140     self._candidate_hidden_kernel = self.add_variable(
 141         "candidate/hidden_projection/%s" % _WEIGHTS_VARIABLE_NAME,
 142         shape=[self._num_units, self._num_units],
 143         initializer=self._kernel_initializer)
 144 
 145     self._candidate_input_bias = self.add_variable(
 146         "candidate/input_projection/%s" % _BIAS_VARIABLE_NAME,
 147         shape=[self._num_units],
 148         initializer=(
 149             self._bias_initializer
 150             if self._bias_initializer is not None
 151             else init_ops.zeros_initializer(dtype=self.dtype)))
 152     self._candidate_hidden_bias = self.add_variable(
 153         "candidate/hidden_projection/%s" % _BIAS_VARIABLE_NAME,
 154         shape=[self._num_units],
 155         initializer=(
 156             self._bias_initializer
 157             if self._bias_initializer is not None
 158             else init_ops.zeros_initializer(dtype=self.dtype)))
 159 
 160   def call(self, inputs, state):
 161     """Gated recurrent unit (GRU) with nunits cells."""
 162     gate_inputs = math_ops.matmul(
 163         array_ops.concat([inputs, state], 1), self._gate_kernel)
 164     gate_inputs = nn_ops.bias_add(gate_inputs, self._gate_bias)
 165 
 166     value = math_ops.sigmoid(gate_inputs)
 167     r, u = array_ops.split(value=value, num_or_size_splits=2, axis=1)
 168 
 169     candidate = nn_ops.bias_add(
 170         math_ops.matmul(inputs, self._candidate_input_kernel),
 171         self._candidate_input_bias)
 172     candidate += r * nn_ops.bias_add(
 173         math_ops.matmul(state, self._candidate_hidden_kernel),
 174         self._candidate_hidden_bias)
 175     candidate = self._activation(candidate)
 176     new_h = (1-u) * candidate + u * state
 177     return new_h, new_h

 188 class CudnnOpaqueParamsSaveable(saver.BaseSaverBuilder.SaveableObject):
 189   """Abstract SaveableObject implementation handling Cudnn opaque params."""
 190 
 191   def __init__(self,
 192                opaque_params,
 193                num_layers,
 194                num_units,
 195                input_size,
 196                input_mode=CUDNN_INPUT_LINEAR_MODE,
 197                direction=CUDNN_RNN_UNIDIRECTION,
 198                scope=None,
 199                name="cudnn_rnn_saveable"):
 200     """Creates a CudnnOpaqueParamsSaveable object.
 201 
 202        CudnnOpaqueParamsSaveable is saveable/restorable in a checkpoint file
 203        and is used to save/restore the weights and biases parameters in a
 204        canonical format which is directly consumable by platform-independent tf
 205        RNN cells. Parameters are saved as tensors layer by layer with weight
 206        tensors followed by bias tensors, and forward direction followed by
 207        backward direction (if applicable). When restoring, a user could name
 208        param_variables as desired, and restore weight and bias tensors to these
 209        variables.
 210 
 211        For CudnnRNNRelu or CudnnRNNTanh, there are 2 tensors per weight and per
 212        bias for each layer: tensor 0 is applied to the input from the previous
 213        layer and tensor 1 to the recurrent input.
 214 
 215        For CudnnLSTM, there are 8 tensors per weight and per bias for each
 216        layer: tensor 0-3 are applied to the input from the previous layer and
 217        tensor 4-7 to the recurrent input. Tensor 0 and 4 are for the input gate;
 218        tensor 1 and 5 the forget gate; tensor 2 and 6 the new memory gate;
 219        tensor 3 and 7 the output gate.
 220 
 221        For CudnnGRU, there are 6 tensors per weight and per bias for each layer:
 222        tensor 0-2 are applied to the input from the previous layer and
 223        tensor 3-5 to the recurrent input. Tensor 0 and 3 are for the reset gate;
 224        tensor 1 and 4 the update gate; tensor 2 and 5 the new memory gate.
 225 
 226     Args:
 227       opaque_params: a variable, Cudnn RNN opaque params.
 228       num_layers: the number of layers for the RNN model.
 229       num_units: the number of units within the RNN model.
 230       input_size: the size of the input, it could be different from the
 231           num_units.
 232       input_mode: indicate whether there is a linear projection between the
 233           input and the actual computation before the first layer. It could be
 234           'linear_input', 'skip_input' or 'auto_select'.
 235           'linear_input' (default) always applies a linear projection of input
 236           onto RNN hidden state. (standard RNN behavior).
 237           'skip_input' is only allowed when input_size == num_units;
 238           'auto_select' implies 'skip_input' when input_size == num_units;
 239           otherwise, it implies 'linear_input'.
 240       direction: the direction model that the model operates. Could be either
 241           'unidirectional' or 'bidirectional'
 242       scope: string of VariableScope, the scope of equivalent subgraph
 243           consisting only platform-independent tf RNN cells.
 244       name: the name of the CudnnOpaqueParamsSaveable object.
 245     """
 246     # Define in subclasses.
 247     self._num_layers = num_layers
 248     self._input_size = input_size
 249     self._num_units = num_units
 250     self._input_mode = input_mode
 251     self._direction = direction
 252     if scope is not None:
 253       scope_name = scope.name if isinstance(scope, vs.VariableScope) else scope
 254       self._scope = scope_name or None
 255     else:
 256       self._scope = None
 257 
 258     self._variables = opaque_params
 259     self._num_dirs = 1 if self._direction == CUDNN_RNN_UNIDIRECTION else 2
 260     self._num_params = (
 261         self._num_params_per_layer * self._num_layers * self._num_dirs)
 262 
 263     weights, biases = self._OpaqueParamsToCanonical()
 264     (weights, weight_names), (biases, bias_names) = self._TransformCanonical(
 265         weights, biases)
 266     # We currently don't use slice_spec. It might be useful in a distributed
 267     # setting where each parameter server node stores a slice of variable,
 268     # instead of having the master pull all slices and then save them.
 269     slice_spec = ""
 270     params = weights + biases
 271     self._weight_names = weight_names
 272     self._bias_names = bias_names
 273     self._param_names = weight_names + bias_names
 274     prefixed_param_names = weight_names + bias_names
 275     if self._scope:
 276       prefixed_param_names = [
 277           "%s/%s" % (self._scope, pn) for pn in prefixed_param_names]
 278     specs = [
 279         saver.BaseSaverBuilder.SaveSpec(param, slice_spec, param_name)
 280         for param, param_name in zip(params, prefixed_param_names)
 281     ]
 282     super(CudnnOpaqueParamsSaveable, self).__init__(
 283         array_ops.identity(self._variables), specs, name)
 284 
 285   def restore(self, restored_tensors, restored_shapes):
 286     weights, biases = self._ReverseTransformCanonical(restored_tensors)
 287     weights = [array_ops.reshape(w, [-1]) for w in weights]
 288     opaque_params = self._CanonicalToOpaqueParams(weights, biases)
 289 
 290     return state_ops.assign(
 291         self._variables, opaque_params, validate_shape=False)
 292 
 293   def _checkpointable_save(self, save_buffer):
 294     weights, biases = self._OpaqueParamsToCanonical()
 295     with ops.device("gpu:0"):
 296       (weights, _), (biases, _) = self._TransformCanonical(
 297           weights, biases)
 298     for name, tensor in zip(self._param_names, weights + biases):
 299       save_buffer[name] = array_ops.identity(tensor)
 300 
 301   def _checkpointable_restore(self, restore_buffer):
 302     tensors = [array_ops.identity(restore_buffer[name])
 303                for name in self._param_names]
 304     return self.restore(
 305         restored_tensors=tensors,
 306         restored_shapes=None  # Unused
 307     )

 311

 326     self._checkpointable_track_params(checkpointable, split_dependencies)

 381   def _TransformCanonical(self, cu_weights, cu_biases):
 382     r"""Transform from Cudnn canonical to tf canonical.
 383 
 384     The elements of argument lists are laid out in the following format:
 385         ------------------------------------------------------------
 386         | weights                    | biases                      |
 387         ------------------------------------------------------------
 388         \                             \
 389          \                             \
 390           -------------------------------
 391           | layer1     |layer2     |... |
 392           -------------------------------
 393           \             \
 394            ---------------
 395            |fwd   |bak   |
 396            ---------------
 397     Args:
 398       cu_weights: a list of tensors of Cudnn canonical weights.
 399       cu_biases: a list of tensors of Cudnn canonical biases.
 400     Returns:
 401       2 tuples, one for weights and the other for bias.
 402       Each tuple has two lists: the 1st for transformed tf canonical tensors
 403       and the 2nd for the names of the tensors under which they are saved.
 404     """
 405     tf_weights, tf_biases = [], []
 406     tf_weights_names, tf_bias_names = [], []
 407 
 408     layer_weights_num = self._num_params_per_layer * self._num_dirs
 409     layer_biases_num = layer_weights_num
 410 
 411     for i in range(self._num_layers):
 412       layer_weights = cu_weights[i * layer_weights_num:
 413                                  (i + 1) * layer_weights_num]
 414       layer_biases = cu_biases[i * layer_biases_num:(i + 1) * layer_biases_num]
 415       if self._direction == CUDNN_RNN_UNIDIRECTION:
 416         prefix = self._TFCanonicalNamePrefix(i)
 417         self._TransformSingleLayerCanonical(layer_weights, layer_biases, prefix,
 418                                             tf_weights, tf_weights_names,
 419                                             tf_biases, tf_bias_names)
 420       else:
 421         fw_prefix = self._TFCanonicalNamePrefix(i, is_fwd=True)
 422         bw_prefix = self._TFCanonicalNamePrefix(i, is_fwd=False)
 423 
 424         fw_weights = layer_weights[:len(layer_weights) // 2]
 425         bw_weights = layer_weights[len(layer_weights) // 2:]
 426         fw_biases = layer_biases[:len(layer_biases) // 2]
 427         bw_biases = layer_biases[len(layer_biases) // 2:]
 428 
 429         self._TransformSingleLayerCanonical(fw_weights, fw_biases, fw_prefix,
 430                                             tf_weights, tf_weights_names,
 431                                             tf_biases, tf_bias_names)
 432 
 433         self._TransformSingleLayerCanonical(bw_weights, bw_biases, bw_prefix,
 434                                             tf_weights, tf_weights_names,
 435                                             tf_biases, tf_bias_names)
 436     return (tf_weights, tf_weights_names), (tf_biases, tf_bias_names)
 437 
 438   def _TransformSingleLayerCanonical(self, cu_weights, cu_biases, prefix,
 439                                      tf_weights, tf_weights_names, tf_biases,
 440                                      tf_bias_names):
 441     r"""Transform single layer Cudnn canonicals to tf canonicals.
 442 
 443     The elements of cu_weights, cu_biases are laid out in the following format:
 444     -------------------------------------------------------------------------
 445     | gate0 param on inputs | gate0 param on hidden state | gate1 ..........|
 446     -------------------------------------------------------------------------
 447     Args:
 448       cu_weights: a list of tensors, single layer weights.
 449       cu_biases: a list of tensors, single layer biases.
 450       prefix: the shared prefix of all tensor names.
 451       tf_weights: a list where transformed weights are stored.
 452       tf_weights_names: a list where names of transformed weights are stored.
 453       tf_biases: a list where transformed biases are stored.
 454       tf_bias_names: a list where names of transformed biases are stored.
 455     """

 458   def _ReverseTransformCanonical(self, tf_canonicals):
 459     r"""Transform from tf canonical to Cudnn canonical.
 460 
 461     This is the reverse routine of _TransformCanonical().
 462     Args:
 463       tf_canonicals: a list of tensors of tf canonical params. The elements are
 464         laid out in the following format:
 465         ------------------------------------------------------------
 466         | weights                    | biases                      |
 467         ------------------------------------------------------------
 468         \                             \
 469          \                             \
 470           -------------------------------
 471           | layer1     |layer2     |... |
 472           -------------------------------
 473           \             \
 474            ---------------
 475            |fwd   |bak   |
 476            ---------------
 477     Returns:
 478       2 lists: the recovered cudnn canonical weights and biases.
 479     """
 480     weights = tf_canonicals[:len(tf_canonicals) // 2]
 481     biases = tf_canonicals[len(tf_canonicals) // 2:]
 482 
 483     cu_weights, cu_biases = [], []
 484     layer_weights_num = len(weights) // self._num_layers
 485     layer_biases_num = len(biases) // self._num_layers
 486     for i in range(self._num_layers):
 487       layer_weights = weights[i * layer_weights_num:(i + 1) * layer_weights_num]
 488       layer_biases = biases[i * layer_biases_num:(i + 1) * layer_biases_num]
 489       if self._direction == CUDNN_RNN_UNIDIRECTION:
 490         cu_weights.extend(self._tf_to_cudnn_weights(i, *layer_weights))
 491         cu_biases.extend(self._tf_to_cudnn_biases(*layer_biases))
 492       else:
 493         fw_weights, bw_weights = layer_weights[:len(
 494             layer_weights) // 2], layer_weights[len(layer_weights) // 2:]
 495         fw_biases, bw_biases = layer_biases[:len(
 496             layer_biases) // 2], layer_biases[len(layer_biases) // 2:]
 497         cu_weights.extend(self._tf_to_cudnn_weights(i, *fw_weights))
 498         cu_biases.extend(self._tf_to_cudnn_biases(*fw_biases))
 499 
 500         cu_weights.extend(self._tf_to_cudnn_weights(i, *bw_weights))
 501         cu_biases.extend(self._tf_to_cudnn_biases(*bw_biases))
 502     return cu_weights, cu_biases

 537   def _cudnn_to_tf_weights(self, *cu_weights):
 538     r"""Stitching cudnn canonical weights to generate tf canonical weights."""
 539     w_i, w_f, w_c, w_o, r_i, r_f, r_c, r_o = cu_weights
 540 
 541     # pylint: disable=invalid-name
 542     W_i = array_ops.concat([w_i, r_i], axis=1)
 543     W_f = array_ops.concat([w_f, r_f], axis=1)
 544     W_c = array_ops.concat([w_c, r_c], axis=1)
 545     W_o = array_ops.concat([w_o, r_o], axis=1)
 546     # pylint: enable=invalid-name
 547     # Cudnn LSTM weights are in ifco order, other tf LSTMs are in icfo order.
 548     reordered = self._cudnn_to_tf_gate_params(* [W_i, W_f, W_c, W_o])
 549     return (array_ops.transpose(array_ops.concat(reordered, axis=0)),)

 551   def _tf_to_cudnn_weights(self, layer, *tf_weights):
 552     r"""Reverse the operations in StitchWeights()."""
 553     input_size = self._input_size
 554     num_units = self._num_units
 555     if layer == 0:
 556       input_weight_width = input_size
 557     else:
 558       input_weight_width = num_units
 559       if self._direction == CUDNN_RNN_BIDIRECTION:
 560         input_weight_width *= 2
 561 
 562     (tf_weight,) = tf_weights
 563     w = array_ops.transpose(tf_weight)
 564     # pylint: disable=invalid-name
 565     W_i, W_f, W_c, W_o = self._tf_to_cudnn_gate_params(*array_ops.split(
 566         w, 4, axis=0))
 567 
 568     w_i, r_i = array_ops.split(W_i, [input_weight_width, num_units], axis=1)
 569     w_c, r_c = array_ops.split(W_c, [input_weight_width, num_units], axis=1)
 570     w_f, r_f = array_ops.split(W_f, [input_weight_width, num_units], axis=1)
 571     w_o, r_o = array_ops.split(W_o, [input_weight_width, num_units], axis=1)
 572     return w_i, w_f, w_c, w_o, r_i, r_f, r_c, r_o
 573     # pylint: enable=invalid-name

 575   def _cudnn_to_tf_biases(self, *cu_biases):
 576     r"""Stitching cudnn canonical biases to generate tf canonical biases."""
 577     b_wi, b_wf, b_wc, b_wo, b_ri, b_rf, b_rc, b_ro = cu_biases
 578     # Save only the sum instead of individual biases. When recovering, return
 579     # two biases each with half the value. Since RNN does not regularize by
 580     # weight decay, it has no side effect in training or inference.
 581     # pylint: disable=invalid-name
 582     B_i = b_wi + b_ri
 583     B_f = b_wf + b_rf
 584     B_c = b_wc + b_rc
 585     B_o = b_wo + b_ro
 586     # pylint: enable=invalid-name
 587     reordered = self._cudnn_to_tf_gate_params(* [B_i, B_f, B_c, B_o])
 588     return (array_ops.concat(reordered, axis=0),)

 590   def _tf_to_cudnn_biases(self, *tf_biases):
 591     r"""Reverse the operations in StitchBiases()."""
 592     (tf_bias,) = tf_biases
 593     # pylint: disable=invalid-name
 594     B_i, B_f, B_c, B_o = self._tf_to_cudnn_gate_params(*array_ops.split(
 595         tf_bias, 4, axis=0))
 596     # pylint: enable=invalid-name
 597     # pylint: disable=unbalanced-tuple-unpacking
 598     b_wi, b_ri = (B_i * 0.5,) * 2
 599     b_wf, b_rf = (B_f * 0.5,) * 2
 600     b_wc, b_rc = (B_c * 0.5,) * 2
 601     b_wo, b_ro = (B_o * 0.5,) * 2
 602     # pylint: enable=unbalanced-tuple-unpacking
 603     # Return ifco order for Cudnn LSTM.
 604     return b_wi, b_wf, b_wc, b_wo, b_ri, b_rf, b_rc, b_ro

 619     """Track parameters for compatibility with CudnnCompatibleLSTMCell."""

 621     weights = []
