  28 def _cross_squared_distance_matrix(x, y):
  29   """Pairwise squared distance between two (batch) matrices' rows (2nd dim).
  30 
  31   Computes the pairwise distances between rows of x and rows of y
  32   Args:
  33     x: [batch_size, n, d] float `Tensor`
  34     y: [batch_size, m, d] float `Tensor`
  35 
  36   Returns:
  37     squared_dists: [batch_size, n, m] float `Tensor`, where
  38     squared_dists[b,i,j] = ||x[b,i,:] - y[b,j,:]||^2
  39   """
  40   x_norm_squared = math_ops.reduce_sum(math_ops.square(x), 2)
  41   y_norm_squared = math_ops.reduce_sum(math_ops.square(y), 2)
  42 
  43   # Expand so that we can broadcast.
  44   x_norm_squared_tile = array_ops.expand_dims(x_norm_squared, 2)
  45   y_norm_squared_tile = array_ops.expand_dims(y_norm_squared, 1)
  46 
  47   x_y_transpose = math_ops.matmul(x, y, adjoint_b=True)
  48 
  49   # squared_dists[b,i,j] = ||x_bi - y_bj||^2 = x_bi'x_bi- 2x_bi'x_bj + x_bj'x_bj
  50   squared_dists = x_norm_squared_tile - 2 * x_y_transpose + y_norm_squared_tile
  51 
  52   return squared_dists

  55 def _pairwise_squared_distance_matrix(x):
  56   """Pairwise squared distance among a (batch) matrix's rows (2nd dim).
  57 
  58   This saves a bit of computation vs. using _cross_squared_distance_matrix(x,x)
  59 
  60   Args:
  61     x: `[batch_size, n, d]` float `Tensor`
  62 
  63   Returns:
  64     squared_dists: `[batch_size, n, n]` float `Tensor`, where
  65     squared_dists[b,i,j] = ||x[b,i,:] - x[b,j,:]||^2
  66   """
  67 
  68   x_x_transpose = math_ops.matmul(x, x, adjoint_b=True)
  69   x_norm_squared = array_ops.matrix_diag_part(x_x_transpose)
  70   x_norm_squared_tile = array_ops.expand_dims(x_norm_squared, 2)
  71 
  72   # squared_dists[b,i,j] = ||x_bi - x_bj||^2 = x_bi'x_bi- 2x_bi'x_bj + x_bj'x_bj
  73   squared_dists = x_norm_squared_tile - 2 * x_x_transpose + array_ops.transpose(
  74       x_norm_squared_tile, [0, 2, 1])
  75 
  76   return squared_dists

  79 def _solve_interpolation(train_points, train_values, order,
  80                          regularization_weight):
  81   """Solve for interpolation coefficients.
  82 
  83   Computes the coefficients of the polyharmonic interpolant for the 'training'
  84   data defined by (train_points, train_values) using the kernel phi.
  85 
  86   Args:
  87     train_points: `[b, n, d]` interpolation centers
  88     train_values: `[b, n, k]` function values
  89     order: order of the interpolation
  90     regularization_weight: weight to place on smoothness regularization term
  91 
  92   Returns:
  93     w: `[b, n, k]` weights on each interpolation center
  94     v: `[b, d, k]` weights on each input dimension
  95   Raises:
  96     ValueError: if d or k is not fully specified.
  97   """
  98 
  99   # These dimensions are set dynamically at runtime.
 100   b, n, _ = array_ops.unstack(array_ops.shape(train_points), num=3)
 101 
 102   d = train_points.shape[-1]
 103   if d.value is None:
 104     raise ValueError('The dimensionality of the input points (d) must be '
 105                      'statically-inferrable.')
 106 
 107   k = train_values.shape[-1]
 108   if k.value is None:
 109     raise ValueError('The dimensionality of the output values (k) must be '
 110                      'statically-inferrable.')
 111 
 112   # First, rename variables so that the notation (c, f, w, v, A, B, etc.)
 113   # follows https://en.wikipedia.org/wiki/Polyharmonic_spline.
 114   # To account for python style guidelines we use
 115   # matrix_a for A and matrix_b for B.
 116 
 117   c = train_points
 118   f = train_values
 119 
 120   # Next, construct the linear system.
 121   with ops.name_scope('construct_linear_system'):
 122 
 123     matrix_a = _phi(_pairwise_squared_distance_matrix(c), order)  # [b, n, n]
 124     if regularization_weight > 0:
 125       batch_identity_matrix = array_ops.expand_dims(
 126           linalg_ops.eye(n, dtype=c.dtype), 0)
 127       matrix_a += regularization_weight * batch_identity_matrix
 128 
 129     # Append ones to the feature values for the bias term in the linear model.
 130     ones = array_ops.ones_like(c[..., :1], dtype=c.dtype)
 131     matrix_b = array_ops.concat([c, ones], 2)  # [b, n, d + 1]
 132 
 133     # [b, n + d + 1, n]
 134     left_block = array_ops.concat(
 135         [matrix_a, array_ops.transpose(matrix_b, [0, 2, 1])], 1)
 136 
 137     num_b_cols = matrix_b.get_shape()[2]  # d + 1
 138     lhs_zeros = array_ops.zeros([b, num_b_cols, num_b_cols], train_points.dtype)
 139     right_block = array_ops.concat([matrix_b, lhs_zeros],
 140                                    1)  # [b, n + d + 1, d + 1]
 141     lhs = array_ops.concat([left_block, right_block],
 142                            2)  # [b, n + d + 1, n + d + 1]
 143 
 144     rhs_zeros = array_ops.zeros([b, d + 1, k], train_points.dtype)
 145     rhs = array_ops.concat([f, rhs_zeros], 1)  # [b, n + d + 1, k]
 146 
 147   # Then, solve the linear system and unpack the results.
 148   with ops.name_scope('solve_linear_system'):
 149     w_v = linalg_ops.matrix_solve(lhs, rhs)
 150     w = w_v[:, :n, :]
 151     v = w_v[:, n:, :]
 152 
 153   return w, v

 156 def _apply_interpolation(query_points, train_points, w, v, order):
 157   """Apply polyharmonic interpolation model to data.
 158 
 159   Given coefficients w and v for the interpolation model, we evaluate
 160   interpolated function values at query_points.
 161 
 162   Args:
 163     query_points: `[b, m, d]` x values to evaluate the interpolation at
 164     train_points: `[b, n, d]` x values that act as the interpolation centers
 165                     ( the c variables in the wikipedia article)
 166     w: `[b, n, k]` weights on each interpolation center
 167     v: `[b, d, k]` weights on each input dimension
 168     order: order of the interpolation
 169 
 170   Returns:
 171     Polyharmonic interpolation evaluated at points defined in query_points.
 172   """
 173 
 174   # First, compute the contribution from the rbf term.
 175   pairwise_dists = _cross_squared_distance_matrix(query_points, train_points)
 176   phi_pairwise_dists = _phi(pairwise_dists, order)
 177 
 178   rbf_term = math_ops.matmul(phi_pairwise_dists, w)
 179 
 180   # Then, compute the contribution from the linear term.
 181   # Pad query_points with ones, for the bias term in the linear model.
 182   query_points_pad = array_ops.concat([
 183       query_points,
 184       array_ops.ones_like(query_points[..., :1], train_points.dtype)
 185   ], 2)
 186   linear_term = math_ops.matmul(query_points_pad, v)
 187 
 188   return rbf_term + linear_term

 191 def _phi(r, order):
 192   """Coordinate-wise nonlinearity used to define the order of the interpolation.
 193 
 194   See https://en.wikipedia.org/wiki/Polyharmonic_spline for the definition.
 195 
 196   Args:
 197     r: input op
 198     order: interpolation order
 199 
 200   Returns:
 201     phi_k evaluated coordinate-wise on r, for k = r
 202   """
 203 
 204   # using EPSILON prevents log(0), sqrt0), etc.
 205   # sqrt(0) is well-defined, but its gradient is not
 206   with ops.name_scope('phi'):
 207     if order == 1:
 208       r = math_ops.maximum(r, EPSILON)
 209       r = math_ops.sqrt(r)
 210       return r
 211     elif order == 2:
 212       return 0.5 * r * math_ops.log(math_ops.maximum(r, EPSILON))
 213     elif order == 4:
 214       return 0.5 * math_ops.square(r) * math_ops.log(
 215           math_ops.maximum(r, EPSILON))
 216     elif order % 2 == 0:
 217       r = math_ops.maximum(r, EPSILON)
 218       return 0.5 * math_ops.pow(r, 0.5 * order) * math_ops.log(r)
 219     else:
 220       r = math_ops.maximum(r, EPSILON)
 221       return math_ops.pow(r, 0.5 * order)

 224 def interpolate_spline(train_points,
 225                        train_values,
 226                        query_points,
 227                        order,
 228                        regularization_weight=0.0,
 229                        name='interpolate_spline'):
 230   r"""Interpolate signal using polyharmonic interpolation.
 231 
 232   The interpolant has the form
 233   $$f(x) = \sum_{i = 1}^n w_i \phi(||x - c_i||) + v^T x + b.$$
 234 
 235   This is a sum of two terms: (1) a weighted sum of radial basis function (RBF)
 236   terms, with the centers \\(c_1, ... c_n\\), and (2) a linear term with a bias.
 237   The \\(c_i\\) vectors are 'training' points. In the code, b is absorbed into v
 238   by appending 1 as a final dimension to x. The coefficients w and v are
 239   estimated such that the interpolant exactly fits the value of the function at
 240   the \\(c_i\\) points, the vector w is orthogonal to each \\(c_i\\), and the
 241   vector w sums to 0. With these constraints, the coefficients can be obtained
 242   by solving a linear system.
 243 
 244   \\(\phi\\) is an RBF, parametrized by an interpolation
 245   order. Using order=2 produces the well-known thin-plate spline.
 246 
 247   We also provide the option to perform regularized interpolation. Here, the
 248   interpolant is selected to trade off between the squared loss on the training
 249   data and a certain measure of its curvature
 250   ([details](https://en.wikipedia.org/wiki/Polyharmonic_spline)).
 251   Using a regularization weight greater than zero has the effect that the
 252   interpolant will no longer exactly fit the training data. However, it may be
 253   less vulnerable to overfitting, particularly for high-order interpolation.
 254 
 255   Note the interpolation procedure is differentiable with respect to all inputs
 256   besides the order parameter.
 257 
 258   We support dynamically-shaped inputs, where batch_size, n, and m are None
 259   at graph construction time. However, d and k must be known.
 260 
 261   Args:
 262     train_points: `[batch_size, n, d]` float `Tensor` of n d-dimensional
 263       locations. These do not need to be regularly-spaced.
 264     train_values: `[batch_size, n, k]` float `Tensor` of n c-dimensional values
 265       evaluated at train_points.
 266     query_points: `[batch_size, m, d]` `Tensor` of m d-dimensional locations
 267       where we will output the interpolant's values.
 268     order: order of the interpolation. Common values are 1 for
 269       \\(\phi(r) = r\\), 2 for \\(\phi(r) = r^2 * log(r)\\) (thin-plate spline),
 270        or 3 for \\(\phi(r) = r^3\\).
 271     regularization_weight: weight placed on the regularization term.
 272       This will depend substantially on the problem, and it should always be
 273       tuned. For many problems, it is reasonable to use no regularization.
 274       If using a non-zero value, we recommend a small value like 0.001.
 275     name: name prefix for ops created by this function
 276 
 277   Returns:
 278     `[b, m, k]` float `Tensor` of query values. We use train_points and
 279     train_values to perform polyharmonic interpolation. The query values are
 280     the values of the interpolant evaluated at the locations specified in
 281     query_points.
 282   """
 283   with ops.name_scope(name):
 284     train_points = ops.convert_to_tensor(train_points)
 285     train_values = ops.convert_to_tensor(train_values)
 286     query_points = ops.convert_to_tensor(query_points)
 287 
 288     # First, fit the spline to the observed data.
 289     with ops.name_scope('solve'):
 290       w, v = _solve_interpolation(train_points, train_values, order,
 291                                   regularization_weight)
 292 
 293     # Then, evaluate the spline at the query locations.
 294     with ops.name_scope('predict'):
 295       query_values = _apply_interpolation(query_points, train_points, w, v,
 296                                           order)
 297 
 298   return query_values
