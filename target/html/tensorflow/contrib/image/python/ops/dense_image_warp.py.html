  30 def _interpolate_bilinear(grid,
  31                           query_points,
  32                           name='interpolate_bilinear',
  33                           indexing='ij'):
  34   """Similar to Matlab's interp2 function.
  35 
  36   Finds values for query points on a grid using bilinear interpolation.
  37 
  38   Args:
  39     grid: a 4-D float `Tensor` of shape `[batch, height, width, channels]`.
  40     query_points: a 3-D float `Tensor` of N points with shape `[batch, N, 2]`.
  41     name: a name for the operation (optional).
  42     indexing: whether the query points are specified as row and column (ij),
  43       or Cartesian coordinates (xy).
  44 
  45   Returns:
  46     values: a 3-D `Tensor` with shape `[batch, N, channels]`
  47 
  48   Raises:
  49     ValueError: if the indexing mode is invalid, or if the shape of the inputs
  50       invalid.
  51   """
  52   if indexing != 'ij' and indexing != 'xy':
  53     raise ValueError('Indexing mode must be \'ij\' or \'xy\'')
  54 
  55   with ops.name_scope(name):
  56     grid = ops.convert_to_tensor(grid)
  57     query_points = ops.convert_to_tensor(query_points)
  58     shape = grid.get_shape().as_list()
  59     if len(shape) != 4:
  60       msg = 'Grid must be 4 dimensional. Received size: '
  61       raise ValueError(msg + str(grid.get_shape()))
  62 
  63     batch_size, height, width, channels = shape
  64     query_type = query_points.dtype
  65     grid_type = grid.dtype
  66 
  67     if (len(query_points.get_shape()) != 3 or
  68         query_points.get_shape()[2].value != 2):
  69       msg = ('Query points must be 3 dimensional and size 2 in dim 2. Received '
  70              'size: ')
  71       raise ValueError(msg + str(query_points.get_shape()))
  72 
  73     _, num_queries, _ = query_points.get_shape().as_list()
  74 
  75     if height < 2 or width < 2:
  76       msg = 'Grid must be at least batch_size x 2 x 2 in size. Received size: '
  77       raise ValueError(msg + str(grid.get_shape()))
  78 
  79     alphas = []
  80     floors = []
  81     ceils = []
  82 
  83     index_order = [0, 1] if indexing == 'ij' else [1, 0]
  84     unstacked_query_points = array_ops.unstack(query_points, axis=2)
  85 
  86     for dim in index_order:
  87       with ops.name_scope('dim-' + str(dim)):
  88         queries = unstacked_query_points[dim]
  89 
  90         size_in_indexing_dimension = shape[dim + 1]
  91 
  92         # max_floor is size_in_indexing_dimension - 2 so that max_floor + 1
  93         # is still a valid index into the grid.
  94         max_floor = math_ops.cast(size_in_indexing_dimension - 2, query_type)
  95         min_floor = constant_op.constant(0.0, dtype=query_type)
  96         floor = math_ops.minimum(
  97             math_ops.maximum(min_floor, math_ops.floor(queries)), max_floor)
  98         int_floor = math_ops.cast(floor, dtypes.int32)
  99         floors.append(int_floor)
 100         ceil = int_floor + 1
 101         ceils.append(ceil)
 102 
 103         # alpha has the same type as the grid, as we will directly use alpha
 104         # when taking linear combinations of pixel values from the image.
 105         alpha = math_ops.cast(queries - floor, grid_type)
 106         min_alpha = constant_op.constant(0.0, dtype=grid_type)
 107         max_alpha = constant_op.constant(1.0, dtype=grid_type)
 108         alpha = math_ops.minimum(math_ops.maximum(min_alpha, alpha), max_alpha)
 109 
 110         # Expand alpha to [b, n, 1] so we can use broadcasting
 111         # (since the alpha values don't depend on the channel).
 112         alpha = array_ops.expand_dims(alpha, 2)
 113         alphas.append(alpha)
 114 
 115     if batch_size * height * width > np.iinfo(np.int32).max / 8:
 116       error_msg = """The image size or batch size is sufficiently large
 117                      that the linearized addresses used by array_ops.gather
 118                      may exceed the int32 limit."""
 119       raise ValueError(error_msg)
 120 
 121     flattened_grid = array_ops.reshape(grid,
 122                                        [batch_size * height * width, channels])
 123     batch_offsets = array_ops.reshape(
 124         math_ops.range(batch_size) * height * width, [batch_size, 1])
 125 
 126     # This wraps array_ops.gather. We reshape the image data such that the
 127     # batch, y, and x coordinates are pulled into the first dimension.
 128     # Then we gather. Finally, we reshape the output back. It's possible this
 129     # code would be made simpler by using array_ops.gather_nd.
 130     def gather(y_coords, x_coords, name):
 131       with ops.name_scope('gather-' + name):
 132         linear_coordinates = batch_offsets + y_coords * width + x_coords
 133         gathered_values = array_ops.gather(flattened_grid, linear_coordinates)
 134         return array_ops.reshape(gathered_values,
 135                                  [batch_size, num_queries, channels])
 136 
 137     # grab the pixel values in the 4 corners around each query point
 138     top_left = gather(floors[0], floors[1], 'top_left')
 139     top_right = gather(floors[0], ceils[1], 'top_right')
 140     bottom_left = gather(ceils[0], floors[1], 'bottom_left')
 141     bottom_right = gather(ceils[0], ceils[1], 'bottom_right')
 142 
 143     # now, do the actual interpolation
 144     with ops.name_scope('interpolate'):
 145       interp_top = alphas[1] * (top_right - top_left) + top_left
 146       interp_bottom = alphas[1] * (bottom_right - bottom_left) + bottom_left
 147       interp = alphas[0] * (interp_bottom - interp_top) + interp_top
 148 
 149     return interp

 152 def dense_image_warp(image, flow, name='dense_image_warp'):
 153   """Image warping using per-pixel flow vectors.
 154 
 155   Apply a non-linear warp to the image, where the warp is specified by a dense
 156   flow field of offset vectors that define the correspondences of pixel values
 157   in the output image back to locations in the  source image. Specifically, the
 158   pixel value at output[b, j, i, c] is
 159   images[b, j - flow[b, j, i, 0], i - flow[b, j, i, 1], c].
 160 
 161   The locations specified by this formula do not necessarily map to an int
 162   index. Therefore, the pixel value is obtained by bilinear
 163   interpolation of the 4 nearest pixels around
 164   (b, j - flow[b, j, i, 0], i - flow[b, j, i, 1]). For locations outside
 165   of the image, we use the nearest pixel values at the image boundary.
 166 
 167 
 168   Args:
 169     image: 4-D float `Tensor` with shape `[batch, height, width, channels]`.
 170     flow: A 4-D float `Tensor` with shape `[batch, height, width, 2]`.
 171     name: A name for the operation (optional).
 172 
 173     Note that image and flow can be of type tf.half, tf.float32, or tf.float64,
 174     and do not necessarily have to be the same type.
 175 
 176   Returns:
 177     A 4-D float `Tensor` with shape`[batch, height, width, channels]`
 178       and same type as input image.
 179 
 180   Raises:
 181     ValueError: if height < 2 or width < 2 or the inputs have the wrong number
 182                 of dimensions.
 183   """
 184   with ops.name_scope(name):
 185     batch_size, height, width, channels = image.get_shape().as_list()
 186     # The flow is defined on the image grid. Turn the flow into a list of query
 187     # points in the grid space.
 188     grid_x, grid_y = array_ops.meshgrid(
 189         math_ops.range(width), math_ops.range(height))
 190     stacked_grid = math_ops.cast(
 191         array_ops.stack([grid_y, grid_x], axis=2), flow.dtype)
 192     batched_grid = array_ops.expand_dims(stacked_grid, axis=0)
 193     query_points_on_grid = batched_grid - flow
 194     query_points_flattened = array_ops.reshape(query_points_on_grid,
 195                                                [batch_size, height * width, 2])
 196     # Compute values at the query points, then reshape the result back to the
 197     # image grid.
 198     interpolated = _interpolate_bilinear(image, query_points_flattened)
 199     interpolated = array_ops.reshape(interpolated,
 200                                      [batch_size, height, width, channels])
 201     return interpolated
