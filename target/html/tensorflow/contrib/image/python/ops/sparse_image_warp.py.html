  30 def _get_grid_locations(image_height, image_width):
  31   """Wrapper for np.meshgrid."""
  32 
  33   y_range = np.linspace(0, image_height - 1, image_height)
  34   x_range = np.linspace(0, image_width - 1, image_width)
  35   y_grid, x_grid = np.meshgrid(y_range, x_range, indexing='ij')
  36   return np.stack((y_grid, x_grid), -1)

  39 def _expand_to_minibatch(np_array, batch_size):
  40   """Tile arbitrarily-sized np_array to include new batch dimension."""
  41   tiles = [batch_size] + [1] * np_array.ndim
  42   return np.tile(np.expand_dims(np_array, 0), tiles)

  45 def _get_boundary_locations(image_height, image_width, num_points_per_edge):
  46   """Compute evenly-spaced indices along edge of image."""
  47   y_range = np.linspace(0, image_height - 1, num_points_per_edge + 2)
  48   x_range = np.linspace(0, image_width - 1, num_points_per_edge + 2)
  49   ys, xs = np.meshgrid(y_range, x_range, indexing='ij')
  50   is_boundary = np.logical_or(
  51       np.logical_or(xs == 0, xs == image_width - 1),
  52       np.logical_or(ys == 0, ys == image_height - 1))
  53   return np.stack([ys[is_boundary], xs[is_boundary]], axis=-1)

  56 def _add_zero_flow_controls_at_boundary(control_point_locations,
  57                                         control_point_flows, image_height,
  58                                         image_width, boundary_points_per_edge):
  59   """Add control points for zero-flow boundary conditions.
  60 
  61    Augment the set of control points with extra points on the
  62    boundary of the image that have zero flow.
  63 
  64   Args:
  65     control_point_locations: input control points
  66     control_point_flows: their flows
  67     image_height: image height
  68     image_width: image width
  69     boundary_points_per_edge: number of points to add in the middle of each
  70                            edge (not including the corners).
  71                            The total number of points added is
  72                            4 + 4*(boundary_points_per_edge).
  73 
  74   Returns:
  75     merged_control_point_locations: augmented set of control point locations
  76     merged_control_point_flows: augmented set of control point flows
  77   """
  78 
  79   batch_size = control_point_locations.get_shape()[0].value
  80 
  81   boundary_point_locations = _get_boundary_locations(image_height, image_width,
  82                                                      boundary_points_per_edge)
  83 
  84   boundary_point_flows = np.zeros([boundary_point_locations.shape[0], 2])
  85 
  86   type_to_use = control_point_locations.dtype
  87   boundary_point_locations = constant_op.constant(
  88       _expand_to_minibatch(boundary_point_locations, batch_size),
  89       dtype=type_to_use)
  90 
  91   boundary_point_flows = constant_op.constant(
  92       _expand_to_minibatch(boundary_point_flows, batch_size), dtype=type_to_use)
  93 
  94   merged_control_point_locations = array_ops.concat(
  95       [control_point_locations, boundary_point_locations], 1)
  96 
  97   merged_control_point_flows = array_ops.concat(
  98       [control_point_flows, boundary_point_flows], 1)
  99 
 100   return merged_control_point_locations, merged_control_point_flows

 103 def sparse_image_warp(image,
 104                       source_control_point_locations,
 105                       dest_control_point_locations,
 106                       interpolation_order=2,
 107                       regularization_weight=0.0,
 108                       num_boundary_points=0,
 109                       name='sparse_image_warp'):
 110   """Image warping using correspondences between sparse control points.
 111 
 112   Apply a non-linear warp to the image, where the warp is specified by
 113   the source and destination locations of a (potentially small) number of
 114   control points. First, we use a polyharmonic spline
 115   (`tf.contrib.image.interpolate_spline`) to interpolate the displacements
 116   between the corresponding control points to a dense flow field.
 117   Then, we warp the image using this dense flow field
 118   (`tf.contrib.image.dense_image_warp`).
 119 
 120   Let t index our control points. For regularization_weight=0, we have:
 121   warped_image[b, dest_control_point_locations[b, t, 0],
 122                   dest_control_point_locations[b, t, 1], :] =
 123   image[b, source_control_point_locations[b, t, 0],
 124            source_control_point_locations[b, t, 1], :].
 125 
 126   For regularization_weight > 0, this condition is met approximately, since
 127   regularized interpolation trades off smoothness of the interpolant vs.
 128   reconstruction of the interpolant at the control points.
 129   See `tf.contrib.image.interpolate_spline` for further documentation of the
 130   interpolation_order and regularization_weight arguments.
 131 
 132 
 133   Args:
 134     image: `[batch, height, width, channels]` float `Tensor`
 135     source_control_point_locations: `[batch, num_control_points, 2]` float
 136       `Tensor`
 137     dest_control_point_locations: `[batch, num_control_points, 2]` float
 138       `Tensor`
 139     interpolation_order: polynomial order used by the spline interpolation
 140     regularization_weight: weight on smoothness regularizer in interpolation
 141     num_boundary_points: How many zero-flow boundary points to include at
 142       each image edge.Usage:
 143         num_boundary_points=0: don't add zero-flow points
 144         num_boundary_points=1: 4 corners of the image
 145         num_boundary_points=2: 4 corners and one in the middle of each edge
 146           (8 points total)
 147         num_boundary_points=n: 4 corners and n-1 along each edge
 148     name: A name for the operation (optional).
 149 
 150     Note that image and offsets can be of type tf.half, tf.float32, or
 151     tf.float64, and do not necessarily have to be the same type.
 152 
 153   Returns:
 154     warped_image: `[batch, height, width, channels]` float `Tensor` with same
 155       type as input image.
 156     flow_field: `[batch, height, width, 2]` float `Tensor` containing the dense
 157       flow field produced by the interpolation.
 158   """
 159 
 160   image = ops.convert_to_tensor(image)
 161   source_control_point_locations = ops.convert_to_tensor(
 162       source_control_point_locations)
 163   dest_control_point_locations = ops.convert_to_tensor(
 164       dest_control_point_locations)
 165 
 166   control_point_flows = (
 167       dest_control_point_locations - source_control_point_locations)
 168 
 169   clamp_boundaries = num_boundary_points > 0
 170   boundary_points_per_edge = num_boundary_points - 1
 171 
 172   with ops.name_scope(name):
 173 
 174     batch_size, image_height, image_width, _ = image.get_shape().as_list()
 175 
 176     # This generates the dense locations where the interpolant
 177     # will be evaluated.
 178     grid_locations = _get_grid_locations(image_height, image_width)
 179 
 180     flattened_grid_locations = np.reshape(grid_locations,
 181                                           [image_height * image_width, 2])
 182 
 183     flattened_grid_locations = constant_op.constant(
 184         _expand_to_minibatch(flattened_grid_locations, batch_size), image.dtype)
 185 
 186     if clamp_boundaries:
 187       (dest_control_point_locations,
 188        control_point_flows) = _add_zero_flow_controls_at_boundary(
 189            dest_control_point_locations, control_point_flows, image_height,
 190            image_width, boundary_points_per_edge)
 191 
 192     flattened_flows = interpolate_spline.interpolate_spline(
 193         dest_control_point_locations, control_point_flows,
 194         flattened_grid_locations, interpolation_order, regularization_weight)
 195 
 196     dense_flows = array_ops.reshape(flattened_flows,
 197                                     [batch_size, image_height, image_width, 2])
 198 
 199     warped_image = dense_image_warp.dense_image_warp(image, dense_flows)
 200 
 201     return warped_image, dense_flows
