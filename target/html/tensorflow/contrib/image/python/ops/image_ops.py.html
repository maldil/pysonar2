  48 def rotate(images, angles, interpolation="NEAREST", name=None):
  49   """Rotate image(s) counterclockwise by the passed angle(s) in radians.
  50 
  51   Args:
  52     images: A tensor of shape (num_images, num_rows, num_columns, num_channels)
  53        (NHWC), (num_rows, num_columns, num_channels) (HWC), or
  54        (num_rows, num_columns) (HW). The rank must be statically known (the
  55        shape is not `TensorShape(None)`.
  56     angles: A scalar angle to rotate all images by, or (if images has rank 4)
  57        a vector of length num_images, with an angle for each image in the batch.
  58     interpolation: Interpolation mode. Supported values: "NEAREST", "BILINEAR".
  59     name: The name of the op.
  60 
  61   Returns:
  62     Image(s) with the same type and shape as `images`, rotated by the given
  63     angle(s). Empty space due to the rotation will be filled with zeros.
  64 
  65   Raises:
  66     TypeError: If `image` is an invalid type.
  67   """
  68   with ops.name_scope(name, "rotate"):
  69     image_or_images = ops.convert_to_tensor(images)
  70     if image_or_images.dtype.base_dtype not in _IMAGE_DTYPES:
  71       raise TypeError("Invalid dtype %s." % image_or_images.dtype)
  72     elif image_or_images.get_shape().ndims is None:
  73       raise TypeError("image_or_images rank must be statically known")
  74     elif len(image_or_images.get_shape()) == 2:
  75       images = image_or_images[None, :, :, None]
  76     elif len(image_or_images.get_shape()) == 3:
  77       images = image_or_images[None, :, :, :]
  78     elif len(image_or_images.get_shape()) == 4:
  79       images = image_or_images
  80     else:
  81       raise TypeError("Images should have rank between 2 and 4.")
  82 
  83     image_height = math_ops.cast(array_ops.shape(images)[1],
  84                                  dtypes.float32)[None]
  85     image_width = math_ops.cast(array_ops.shape(images)[2],
  86                                 dtypes.float32)[None]
  87     output = transform(
  88         images,
  89         angles_to_projective_transforms(angles, image_height, image_width),
  90         interpolation=interpolation)
  91     if image_or_images.get_shape().ndims is None:
  92       raise TypeError("image_or_images rank must be statically known")
  93     elif len(image_or_images.get_shape()) == 2:
  94       return output[0, :, :, 0]
  95     elif len(image_or_images.get_shape()) == 3:
  96       return output[0, :, :, :]
  97     else:
  98       return output

 101 def translate(images, translations, interpolation="NEAREST", name=None):
 102   """Translate image(s) by the passed vectors(s).
 103 
 104   Args:
 105     images: A tensor of shape (num_images, num_rows, num_columns, num_channels)
 106         (NHWC), (num_rows, num_columns, num_channels) (HWC), or
 107         (num_rows, num_columns) (HW). The rank must be statically known (the
 108         shape is not `TensorShape(None)`.
 109     translations: A vector representing [dx, dy] or (if images has rank 4)
 110         a matrix of length num_images, with a [dx, dy] vector for each image in
 111         the batch.
 112     interpolation: Interpolation mode. Supported values: "NEAREST", "BILINEAR".
 113     name: The name of the op.
 114 
 115   Returns:
 116     Image(s) with the same type and shape as `images`, translated by the given
 117         vector(s). Empty space due to the translation will be filled with zeros.
 118 
 119   Raises:
 120     TypeError: If `image` is an invalid type.
 121   """
 122   with ops.name_scope(name, "translate"):
 123     return transform(
 124         images,
 125         translations_to_projective_transforms(translations),
 126         interpolation=interpolation)

 129 def angles_to_projective_transforms(angles,
 130                                     image_height,
 131                                     image_width,
 132                                     name=None):
 133   """Returns projective transform(s) for the given angle(s).
 134 
 135   Args:
 136     angles: A scalar angle to rotate all images by, or (for batches of images)
 137         a vector with an angle to rotate each image in the batch. The rank must
 138         be statically known (the shape is not `TensorShape(None)`.
 139     image_height: Height of the image(s) to be transformed.
 140     image_width: Width of the image(s) to be transformed.
 141 
 142   Returns:
 143     A tensor of shape (num_images, 8). Projective transforms which can be given
 144       to `tf.contrib.image.transform`.
 145   """
 146   with ops.name_scope(name, "angles_to_projective_transforms"):
 147     angle_or_angles = ops.convert_to_tensor(
 148         angles, name="angles", dtype=dtypes.float32)
 149     if len(angle_or_angles.get_shape()) == 0:  # pylint: disable=g-explicit-length-test
 150       angles = angle_or_angles[None]
 151     elif len(angle_or_angles.get_shape()) == 1:
 152       angles = angle_or_angles
 153     else:
 154       raise TypeError("Angles should have rank 0 or 1.")
 155     x_offset = ((image_width - 1) - (math_ops.cos(angles) *
 156                                      (image_width - 1) - math_ops.sin(angles) *
 157                                      (image_height - 1))) / 2.0
 158     y_offset = ((image_height - 1) - (math_ops.sin(angles) *
 159                                       (image_width - 1) + math_ops.cos(angles) *
 160                                       (image_height - 1))) / 2.0
 161     num_angles = array_ops.shape(angles)[0]
 162     return array_ops.concat(
 163         values=[
 164             math_ops.cos(angles)[:, None],
 165             -math_ops.sin(angles)[:, None],
 166             x_offset[:, None],
 167             math_ops.sin(angles)[:, None],
 168             math_ops.cos(angles)[:, None],
 169             y_offset[:, None],
 170             array_ops.zeros((num_angles, 2), dtypes.float32),
 171         ],
 172         axis=1)

 175 def translations_to_projective_transforms(translations, name=None):
 176   """Returns projective transform(s) for the given translation(s).
 177 
 178   Args:
 179       translations: A 2-element list representing [dx, dy] or a matrix of
 180           2-element lists representing [dx, dy] to translate for each image
 181           (for a batch of images). The rank must be statically known (the shape
 182           is not `TensorShape(None)`.
 183       name: The name of the op.
 184 
 185   Returns:
 186       A tensor of shape (num_images, 8) projective transforms which can be given
 187           to `tf.contrib.image.transform`.
 188   """
 189   with ops.name_scope(name, "translations_to_projective_transforms"):
 190     translation_or_translations = ops.convert_to_tensor(
 191         translations, name="translations", dtype=dtypes.float32)
 192     if translation_or_translations.get_shape().ndims is None:
 193       raise TypeError(
 194           "translation_or_translations rank must be statically known")
 195     elif len(translation_or_translations.get_shape()) == 1:
 196       translations = translation_or_translations[None]
 197     elif len(translation_or_translations.get_shape()) == 2:
 198       translations = translation_or_translations
 199     else:
 200       raise TypeError("Translations should have rank 1 or 2.")
 201     num_translations = array_ops.shape(translations)[0]
 202     # The translation matrix looks like:
 203     #     [[1 0 -dx]
 204     #      [0 1 -dy]
 205     #      [0 0 1]]
 206     # where the last entry is implicit.
 207     # Translation matrices are always float32.
 208     return array_ops.concat(
 209         values=[
 210             array_ops.ones((num_translations, 1), dtypes.float32),
 211             array_ops.zeros((num_translations, 1), dtypes.float32),
 212             -translations[:, 0, None],
 213             array_ops.zeros((num_translations, 1), dtypes.float32),
 214             array_ops.ones((num_translations, 1), dtypes.float32),
 215             -translations[:, 1, None],
 216             array_ops.zeros((num_translations, 2), dtypes.float32),
 217         ],
 218         axis=1)

 221 def transform(images,
 222               transforms,
 223               interpolation="NEAREST",
 224               output_shape=None,
 225               name=None):
 226   """Applies the given transform(s) to the image(s).
 227 
 228   Args:
 229     images: A tensor of shape (num_images, num_rows, num_columns, num_channels)
 230        (NHWC), (num_rows, num_columns, num_channels) (HWC), or
 231        (num_rows, num_columns) (HW). The rank must be statically known (the
 232        shape is not `TensorShape(None)`.
 233     transforms: Projective transform matrix/matrices. A vector of length 8 or
 234        tensor of size N x 8. If one row of transforms is
 235        [a0, a1, a2, b0, b1, b2, c0, c1], then it maps the *output* point
 236        `(x, y)` to a transformed *input* point
 237        `(x', y') = ((a0 x + a1 y + a2) / k, (b0 x + b1 y + b2) / k)`,
 238        where `k = c0 x + c1 y + 1`. The transforms are *inverted* compared to
 239        the transform mapping input points to output points. Note that gradients
 240        are not backpropagated into transformation parameters.
 241     interpolation: Interpolation mode. Supported values: "NEAREST", "BILINEAR".
 242     output_shape: Output dimesion after the transform, [height, width].
 243        If None, output is the same size as input image.
 244 
 245     name: The name of the op.
 246 
 247   Returns:
 248     Image(s) with the same type and shape as `images`, with the given
 249     transform(s) applied. Transformed coordinates outside of the input image
 250     will be filled with zeros.
 251 
 252   Raises:
 253     TypeError: If `image` is an invalid type.
 254     ValueError: If output shape is not 1-D int32 Tensor.
 255   """
 256   with ops.name_scope(name, "transform"):
 257     image_or_images = ops.convert_to_tensor(images, name="images")
 258     transform_or_transforms = ops.convert_to_tensor(
 259         transforms, name="transforms", dtype=dtypes.float32)
 260     if image_or_images.dtype.base_dtype not in _IMAGE_DTYPES:
 261       raise TypeError("Invalid dtype %s." % image_or_images.dtype)
 262     elif image_or_images.get_shape().ndims is None:
 263       raise TypeError("image_or_images rank must be statically known")
 264     elif len(image_or_images.get_shape()) == 2:
 265       images = image_or_images[None, :, :, None]
 266     elif len(image_or_images.get_shape()) == 3:
 267       images = image_or_images[None, :, :, :]
 268     elif len(image_or_images.get_shape()) == 4:
 269       images = image_or_images
 270     else:
 271       raise TypeError("Images should have rank between 2 and 4.")
 272 
 273     if output_shape is None:
 274       output_shape = tensor_util.constant_value(
 275           array_ops.shape(images)[1:3]) or array_ops.shape(images)[1:3]
 276 
 277     output_shape = ops.convert_to_tensor(
 278         output_shape, dtypes.int32, name="output_shape")
 279 
 280     if not output_shape.get_shape().is_compatible_with([2]):
 281       raise ValueError("output_shape must be a 1-D Tensor of 2 elements: "
 282                        "new_height, new_width")
 283 
 284     if len(transform_or_transforms.get_shape()) == 1:
 285       transforms = transform_or_transforms[None]
 286     elif transform_or_transforms.get_shape().ndims is None:
 287       raise TypeError(
 288           "transform_or_transforms rank must be statically known")
 289     elif len(transform_or_transforms.get_shape()) == 2:
 290       transforms = transform_or_transforms
 291     else:
 292       raise TypeError("Transforms should have rank 1 or 2.")
 293 
 294     output = gen_image_ops.image_projective_transform_v2(
 295         images,
 296         output_shape=output_shape,
 297         transforms=transforms,
 298         interpolation=interpolation.upper())
 299     if len(image_or_images.get_shape()) == 2:
 300       return output[0, :, :, 0]
 301     elif len(image_or_images.get_shape()) == 3:
 302       return output[0, :, :, :]
 303     else:
 304       return output

 307 def compose_transforms(*transforms):
 308   """Composes the transforms tensors.
 309 
 310   Args:
 311     *transforms: List of image projective transforms to be composed. Each
 312         transform is length 8 (single transform) or shape (N, 8) (batched
 313         transforms). The shapes of all inputs must be equal, and at least one
 314         input must be given.
 315 
 316   Returns:
 317     A composed transform tensor. When passed to `tf.contrib.image.transform`,
 318         equivalent to applying each of the given transforms to the image in
 319         order.
 320   """
 321   assert transforms, "transforms cannot be empty"
 322   with ops.name_scope("compose_transforms"):
 323     composed = flat_transforms_to_matrices(transforms[0])
 324     for tr in transforms[1:]:
 325       # Multiply batches of matrices.
 326       composed = math_ops.matmul(composed, flat_transforms_to_matrices(tr))
 327     return matrices_to_flat_transforms(composed)

 330 def flat_transforms_to_matrices(transforms):
 331   """Converts `tf.contrib.image` projective transforms to affine matrices.
 332 
 333   Note that the output matrices map output coordinates to input coordinates. For
 334   the forward transformation matrix, call `tf.linalg.inv` on the result.
 335 
 336   Args:
 337     transforms: Vector of length 8, or batches of transforms with shape
 338       `(N, 8)`.
 339 
 340   Returns:
 341     3D tensor of matrices with shape `(N, 3, 3)`. The output matrices map the
 342       *output coordinates* (in homogeneous coordinates) of each transform to the
 343       corresponding *input coordinates*.
 344 
 345   Raises:
 346     ValueError: If `transforms` have an invalid shape.
 347   """
 348   with ops.name_scope("flat_transforms_to_matrices"):
 349     transforms = ops.convert_to_tensor(transforms, name="transforms")
 350     if transforms.shape.ndims not in (1, 2):
 351       raise ValueError("Transforms should be 1D or 2D, got: %s" % transforms)
 352     # Make the transform(s) 2D in case the input is a single transform.
 353     transforms = array_ops.reshape(transforms, constant_op.constant([-1, 8]))
 354     num_transforms = array_ops.shape(transforms)[0]
 355     # Add a column of ones for the implicit last entry in the matrix.
 356     return array_ops.reshape(
 357         array_ops.concat(
 358             [transforms, array_ops.ones([num_transforms, 1])], axis=1),
 359         constant_op.constant([-1, 3, 3]))

 362 def matrices_to_flat_transforms(transform_matrices):
 363   """Converts affine matrices to `tf.contrib.image` projective transforms.
 364 
 365   Note that we expect matrices that map output coordinates to input coordinates.
 366   To convert forward transformation matrices, call `tf.linalg.inv` on the
 367   matrices and use the result here.
 368 
 369   Args:
 370     transform_matrices: One or more affine transformation matrices, for the
 371       reverse transformation in homogeneous coordinates. Shape `(3, 3)` or
 372       `(N, 3, 3)`.
 373 
 374   Returns:
 375     2D tensor of flat transforms with shape `(N, 8)`, which may be passed into
 376       `tf.contrib.image.transform`.
 377 
 378   Raises:
 379     ValueError: If `transform_matrices` have an invalid shape.
 380   """
 381   with ops.name_scope("matrices_to_flat_transforms"):
 382     transform_matrices = ops.convert_to_tensor(
 383         transform_matrices, name="transform_matrices")
 384     if transform_matrices.shape.ndims not in (2, 3):
 385       raise ValueError(
 386           "Matrices should be 2D or 3D, got: %s" % transform_matrices)
 387     # Flatten each matrix.
 388     transforms = array_ops.reshape(transform_matrices,
 389                                    constant_op.constant([-1, 9]))
 390     # Divide each matrix by the last entry (normally 1).
 391     transforms /= transforms[:, 8:9]
 392     return transforms[:, :8]

 395 @ops.RegisterGradient("ImageProjectiveTransformV2")
 396 def _image_projective_transform_grad(op, grad):
 397   """Computes the gradient for ImageProjectiveTransform."""
 398   images = op.inputs[0]
 399   transforms = op.inputs[1]
 400   interpolation = op.get_attr("interpolation")
 401 
 402   image_or_images = ops.convert_to_tensor(images, name="images")
 403   transform_or_transforms = ops.convert_to_tensor(
 404       transforms, name="transforms", dtype=dtypes.float32)
 405 
 406   if image_or_images.dtype.base_dtype not in _IMAGE_DTYPES:
 407     raise TypeError("Invalid dtype %s." % image_or_images.dtype)
 408   if len(transform_or_transforms.get_shape()) == 1:
 409     transforms = transform_or_transforms[None]
 410   elif len(transform_or_transforms.get_shape()) == 2:
 411     transforms = transform_or_transforms
 412   else:
 413     raise TypeError("Transforms should have rank 1 or 2.")
 414 
 415   # Invert transformations
 416   transforms = flat_transforms_to_matrices(transforms=transforms)
 417   inverse = linalg_ops.matrix_inverse(transforms)
 418   transforms = matrices_to_flat_transforms(inverse)
 419   output = gen_image_ops.image_projective_transform_v2(
 420       images=grad,
 421       transforms=transforms,
 422       output_shape=array_ops.shape(image_or_images)[1:3],
 423       interpolation=interpolation)
 424   return [output, None, None]

 427 def bipartite_match(distance_mat,
 428                     num_valid_rows,
 429                     top_k=-1,
 430                     name="bipartite_match"):
 431   """Find bipartite matching based on a given distance matrix.
 432 
 433   A greedy bi-partite matching algorithm is used to obtain the matching with
 434   the (greedy) minimum distance.
 435 
 436   Args:
 437     distance_mat: A 2-D float tensor of shape `[num_rows, num_columns]`. It is a
 438       pair-wise distance matrix between the entities represented by each row and
 439       each column. It is an asymmetric matrix. The smaller the distance is, the
 440       more similar the pairs are. The bipartite matching is to minimize the
 441       distances.
 442     num_valid_rows: A scalar or a 1-D tensor with one element describing the
 443       number of valid rows of distance_mat to consider for the bipartite
 444       matching. If set to be negative, then all rows from `distance_mat` are
 445       used.
 446     top_k: A scalar that specifies the number of top-k matches to retrieve.
 447       If set to be negative, then is set according to the maximum number of
 448       matches from `distance_mat`.
 449     name: The name of the op.
 450 
 451   Returns:
 452     row_to_col_match_indices: A vector of length num_rows, which is the number
 453       of rows of the input `distance_matrix`. If `row_to_col_match_indices[i]`
 454       is not -1, row i is matched to column `row_to_col_match_indices[i]`.
 455     col_to_row_match_indices: A vector of length num_columns, which is the
 456       number of columns of the input distance matrix.
 457       If `col_to_row_match_indices[j]` is not -1, column j is matched to row
 458       `col_to_row_match_indices[j]`.
 459   """
 460   result = gen_image_ops.bipartite_match(
 461       distance_mat, num_valid_rows, top_k, name=name)
 462   return result

 465 def connected_components(images):
 466   """Labels the connected components in a batch of images.
 467 
 468   A component is a set of pixels in a single input image, which are all adjacent
 469   and all have the same non-zero value. The components using a squared
 470   connectivity of one (all True entries are joined with their neighbors above,
 471   below, left, and right). Components across all images have consecutive ids 1
 472   through n. Components are labeled according to the first pixel of the
 473   component appearing in row-major order (lexicographic order by
 474   image_index_in_batch, row, col). Zero entries all have an output id of 0.
 475 
 476   This op is equivalent with `scipy.ndimage.measurements.label` on a 2D array
 477   with the default structuring element (which is the connectivity used here).
 478 
 479   Args:
 480     images: A 2D (H, W) or 3D (N, H, W) Tensor of boolean image(s).
 481 
 482   Returns:
 483     Components with the same shape as `images`. False entries in `images` have
 484     value 0, and all True entries map to a component id > 0.
 485 
 486   Raises:
 487     TypeError: if `images` is not 2D or 3D.
 488   """
 489   with ops.name_scope("connected_components"):
 490     image_or_images = ops.convert_to_tensor(images, name="images")
 491     if len(image_or_images.get_shape()) == 2:
 492       images = image_or_images[None, :, :]
 493     elif len(image_or_images.get_shape()) == 3:
 494       images = image_or_images
 495     else:
 496       raise TypeError(
 497           "images should have rank 2 (HW) or 3 (NHW). Static shape is %s" %
 498           image_or_images.get_shape())
 499     components = gen_image_ops.image_connected_components(images)
 500 
 501     # TODO(ringwalt): Component id renaming should be done in the op, to avoid
 502     # constructing multiple additional large tensors.
 503     components_flat = array_ops.reshape(components, [-1])
 504     unique_ids, id_index = array_ops.unique(components_flat)
 505     id_is_zero = array_ops.where(math_ops.equal(unique_ids, 0))[:, 0]
 506     # Map each nonzero id to consecutive values.
 507     nonzero_consecutive_ids = math_ops.range(
 508         array_ops.shape(unique_ids)[0] - array_ops.shape(id_is_zero)[0]) + 1
 509 
 510     def no_zero():
 511       # No need to insert a zero into the ids.
 512       return nonzero_consecutive_ids
 513 
 514     def has_zero():
 515       # Insert a zero in the consecutive ids where zero appears in unique_ids.
 516       # id_is_zero has length 1.
 517       zero_id_ind = math_ops.to_int32(id_is_zero[0])
 518       ids_before = nonzero_consecutive_ids[:zero_id_ind]
 519       ids_after = nonzero_consecutive_ids[zero_id_ind:]
 520       return array_ops.concat([ids_before, [0], ids_after], axis=0)
 521 
 522     new_ids = control_flow_ops.cond(
 523         math_ops.equal(array_ops.shape(id_is_zero)[0], 0), no_zero, has_zero)
 524     components = array_ops.reshape(
 525         array_ops.gather(new_ids, id_index), array_ops.shape(components))
 526     if len(image_or_images.get_shape()) == 2:
 527       return components[0, :, :]
 528     else:
 529       return components
