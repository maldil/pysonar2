  79 def _tile_batch(t, multiplier):
  80   """Core single-tensor implementation of tile_batch."""
  81   t = ops.convert_to_tensor(t, name="t")
  82   shape_t = array_ops.shape(t)
  83   if t.shape.ndims is None or t.shape.ndims < 1:
  84     raise ValueError("t must have statically known rank")
  85   tiling = [1] * (t.shape.ndims + 1)
  86   tiling[1] = multiplier
  87   tiled_static_batch_size = (
  88       t.shape[0].value * multiplier if t.shape[0].value is not None else None)
  89   tiled = array_ops.tile(array_ops.expand_dims(t, 1), tiling)
  90   tiled = array_ops.reshape(tiled,
  91                             array_ops.concat(
  92                                 ([shape_t[0] * multiplier], shape_t[1:]), 0))
  93   tiled.set_shape(
  94       tensor_shape.TensorShape([tiled_static_batch_size]).concatenate(
  95           t.shape[1:]))
  96   return tiled

  99 def tile_batch(t, multiplier, name=None):
 100   """Tile the batch dimension of a (possibly nested structure of) tensor(s) t.
 101 
 102   For each tensor t in a (possibly nested structure) of tensors,
 103   this function takes a tensor t shaped `[batch_size, s0, s1, ...]` composed of
 104   minibatch entries `t[0], ..., t[batch_size - 1]` and tiles it to have a shape
 105   `[batch_size * multiplier, s0, s1, ...]` composed of minibatch entries
 106   `t[0], t[0], ..., t[1], t[1], ...` where each minibatch entry is repeated
 107   `multiplier` times.
 108 
 109   Args:
 110     t: `Tensor` shaped `[batch_size, ...]`.
 111     multiplier: Python int.
 112     name: Name scope for any created operations.
 113 
 114   Returns:
 115     A (possibly nested structure of) `Tensor` shaped
 116     `[batch_size * multiplier, ...]`.
 117 
 118   Raises:
 119     ValueError: if tensor(s) `t` do not have a statically known rank or
 120     the rank is < 1.
 121   """
 122   flat_t = nest.flatten(t)
 123   with ops.name_scope(name, "tile_batch", flat_t + [multiplier]):
 124     return nest.map_structure(lambda t_: _tile_batch(t_, multiplier), t)

 127 def gather_tree_from_array(t, parent_ids, sequence_length):
 128   """Calculates the full beams for `TensorArray`s.
 129 
 130   Args:
 131     t: A stacked `TensorArray` of size `max_time` that contains `Tensor`s of
 132       shape `[batch_size, beam_width, s]` or `[batch_size * beam_width, s]`
 133       where `s` is the depth shape.
 134     parent_ids: The parent ids of shape `[max_time, batch_size, beam_width]`.
 135     sequence_length: The sequence length of shape `[batch_size, beam_width]`.
 136 
 137   Returns:
 138     A `Tensor` which is a stacked `TensorArray` of the same size and type as
 139     `t` and where beams are sorted in each `Tensor` according to `parent_ids`.
 140   """
 141   max_time = parent_ids.shape[0].value or array_ops.shape(parent_ids)[0]
 142   batch_size = parent_ids.shape[1].value or array_ops.shape(parent_ids)[1]
 143   beam_width = parent_ids.shape[2].value or array_ops.shape(parent_ids)[2]
 144 
 145   # Generate beam ids that will be reordered by gather_tree.
 146   beam_ids = array_ops.expand_dims(
 147       array_ops.expand_dims(math_ops.range(beam_width), 0), 0)
 148   beam_ids = array_ops.tile(beam_ids, [max_time, batch_size, 1])
 149 
 150   max_sequence_lengths = math_ops.to_int32(
 151       math_ops.reduce_max(sequence_length, axis=1))
 152   sorted_beam_ids = beam_search_ops.gather_tree(
 153       step_ids=beam_ids,
 154       parent_ids=parent_ids,
 155       max_sequence_lengths=max_sequence_lengths,
 156       end_token=beam_width + 1)
 157 
 158   # For out of range steps, simply copy the same beam.
 159   in_bound_steps = array_ops.transpose(
 160       array_ops.sequence_mask(sequence_length, maxlen=max_time),
 161       perm=[2, 0, 1])
 162   sorted_beam_ids = array_ops.where(
 163       in_bound_steps, x=sorted_beam_ids, y=beam_ids)
 164 
 165   # Generate indices for gather_nd.
 166   time_ind = array_ops.tile(array_ops.reshape(
 167       math_ops.range(max_time), [-1, 1, 1]), [1, batch_size, beam_width])
 168   batch_ind = array_ops.tile(array_ops.reshape(
 169       math_ops.range(batch_size), [-1, 1, 1]), [1, max_time, beam_width])
 170   batch_ind = array_ops.transpose(batch_ind, perm=[1, 0, 2])
 171   indices = array_ops.stack([time_ind, batch_ind, sorted_beam_ids], -1)
 172 
 173   # Gather from a tensor with collapsed additional dimensions.
 174   gather_from = t
 175   final_shape = array_ops.shape(gather_from)
 176   gather_from = array_ops.reshape(
 177       gather_from, [max_time, batch_size, beam_width, -1])
 178   ordered = array_ops.gather_nd(gather_from, indices)
 179   ordered = array_ops.reshape(ordered, final_shape)
 180 
 181   return ordered

 189 def _check_static_batch_beam_maybe(shape, batch_size, beam_width):
 190   """Raises an exception if dimensions are known statically and can not be
 191   reshaped to [batch_size, beam_size, -1].
 192   """
 193   reshaped_shape = tensor_shape.TensorShape([batch_size, beam_width, None])
 194   if (batch_size is not None and shape[0].value is not None
 195       and (shape[0] != batch_size * beam_width
 196            or (shape.ndims >= 2 and shape[1].value is not None
 197                and (shape[0] != batch_size or shape[1] != beam_width)))):
 198     tf_logging.warn("TensorArray reordering expects elements to be "
 199                     "reshapable to %s which is incompatible with the "
 200                     "current shape %s. Consider setting "
 201                     "reorder_tensor_arrays to False to disable TensorArray "
 202                     "reordering during the beam search."
 203                     % (reshaped_shape, shape))
 204     return False
 205   return True
 206 
 207 def _check_batch_beam(t, batch_size, beam_width):
 208   """Returns an Assert operation checking that the elements of the stacked
 209   TensorArray can be reshaped to [batch_size, beam_size, -1]. At this point,
 210   the TensorArray elements have a known rank of at least 1.
 211   """
 212   error_message = ("TensorArray reordering expects elements to be "
 213                    "reshapable to [batch_size, beam_size, -1] which is "
 214                    "incompatible with the dynamic shape of %s elements. "
 215                    "Consider setting reorder_tensor_arrays to False to disable "
 216                    "TensorArray reordering during the beam search."
 217                    % (t.name))
 218   rank = t.shape.ndims
 219   shape = array_ops.shape(t)
 220   if rank == 2:
 221     condition = math_ops.equal(shape[1], batch_size * beam_width)
 222   else:
 223     condition = math_ops.logical_or(
 224         math_ops.equal(shape[1], batch_size * beam_width),
 225         math_ops.logical_and(
 226             math_ops.equal(shape[1], batch_size),
 227             math_ops.equal(shape[2], beam_width)))
 228   return control_flow_ops.Assert(condition, [error_message])

 232 class BeamSearchDecoder(decoder.Decoder):
 233   """BeamSearch sampling decoder.
 234 
 235     **NOTE** If you are using the `BeamSearchDecoder` with a cell wrapped in
 236     `AttentionWrapper`, then you must ensure that:
 237 
 238     - The encoder output has been tiled to `beam_width` via
 239       `tf.contrib.seq2seq.tile_batch` (NOT `tf.tile`).
 240     - The `batch_size` argument passed to the `zero_state` method of this
 241       wrapper is equal to `true_batch_size * beam_width`.
 242     - The initial state created with `zero_state` above contains a
 243       `cell_state` value containing properly tiled final state from the
 244       encoder.
 245 
 246     An example:
 247 
 248     ```
 249     tiled_encoder_outputs = tf.contrib.seq2seq.tile_batch(
 250         encoder_outputs, multiplier=beam_width)
 251     tiled_encoder_final_state = tf.contrib.seq2seq.tile_batch(
 252         encoder_final_state, multiplier=beam_width)
 253     tiled_sequence_length = tf.contrib.seq2seq.tile_batch(
 254         sequence_length, multiplier=beam_width)
 255     attention_mechanism = MyFavoriteAttentionMechanism(
 256         num_units=attention_depth,
 257         memory=tiled_inputs,
 258         memory_sequence_length=tiled_sequence_length)
 259     attention_cell = AttentionWrapper(cell, attention_mechanism, ...)
 260     decoder_initial_state = attention_cell.zero_state(
 261         dtype, batch_size=true_batch_size * beam_width)
 262     decoder_initial_state = decoder_initial_state.clone(
 263         cell_state=tiled_encoder_final_state)
 264     ```
 265 
 266     Meanwhile, with `AttentionWrapper`, coverage penalty is suggested to use
 267     when computing scores(https://arxiv.org/pdf/1609.08144.pdf). It encourages
 268     the translation to cover all inputs.
 269   """
 270 
 271   def __init__(self,
 272                cell,
 273                embedding,
 274                start_tokens,
 275                end_token,
 276                initial_state,
 277                beam_width,
 278                output_layer=None,
 279                length_penalty_weight=0.0,
 280                coverage_penalty_weight=0.0,
 281                reorder_tensor_arrays=True):
 282     """Initialize the BeamSearchDecoder.
 283 
 284     Args:
 285       cell: An `RNNCell` instance.
 286       embedding: A callable that takes a vector tensor of `ids` (argmax ids),
 287         or the `params` argument for `embedding_lookup`.
 288       start_tokens: `int32` vector shaped `[batch_size]`, the start tokens.
 289       end_token: `int32` scalar, the token that marks end of decoding.
 290       initial_state: A (possibly nested tuple of...) tensors and TensorArrays.
 291       beam_width:  Python integer, the number of beams.
 292       output_layer: (Optional) An instance of `tf.layers.Layer`, i.e.,
 293         `tf.layers.Dense`.  Optional layer to apply to the RNN output prior
 294         to storing the result or sampling.
 295       length_penalty_weight: Float weight to penalize length. Disabled with 0.0.
 296       coverage_penalty_weight: Float weight to penalize the coverage of source
 297         sentence. Disabled with 0.0.
 298       reorder_tensor_arrays: If `True`, `TensorArray`s' elements within the cell
 299         state will be reordered according to the beam search path. If the
 300         `TensorArray` can be reordered, the stacked form will be returned.
 301         Otherwise, the `TensorArray` will be returned as is. Set this flag to
 302         `False` if the cell state contains `TensorArray`s that are not amenable
 303         to reordering.
 304 
 305     Raises:
 306       TypeError: if `cell` is not an instance of `RNNCell`,
 307         or `output_layer` is not an instance of `tf.layers.Layer`.
 308       ValueError: If `start_tokens` is not a vector or
 309         `end_token` is not a scalar.
 310     """
 311     rnn_cell_impl.assert_like_rnncell("cell", cell)  # pylint: disable=protected-access
 312     if (output_layer is not None and
 313         not isinstance(output_layer, layers_base.Layer)):
 314       raise TypeError(
 315           "output_layer must be a Layer, received: %s" % type(output_layer))
 316     self._cell = cell
 317     self._output_layer = output_layer
 318     self._reorder_tensor_arrays = reorder_tensor_arrays
 319 
 320     if callable(embedding):
 321       self._embedding_fn = embedding
 322     else:
 323       self._embedding_fn = (
 324           lambda ids: embedding_ops.embedding_lookup(embedding, ids))
 325 
 326     self._start_tokens = ops.convert_to_tensor(
 327         start_tokens, dtype=dtypes.int32, name="start_tokens")
 328     if self._start_tokens.get_shape().ndims != 1:
 329       raise ValueError("start_tokens must be a vector")
 330     self._end_token = ops.convert_to_tensor(
 331         end_token, dtype=dtypes.int32, name="end_token")
 332     if self._end_token.get_shape().ndims != 0:
 333       raise ValueError("end_token must be a scalar")
 334 
 335     self._batch_size = array_ops.size(start_tokens)
 336     self._beam_width = beam_width
 337     self._length_penalty_weight = length_penalty_weight
 338     self._coverage_penalty_weight = coverage_penalty_weight
 339     self._initial_cell_state = nest.map_structure(
 340         self._maybe_split_batch_beams, initial_state, self._cell.state_size)
 341     self._start_tokens = array_ops.tile(
 342         array_ops.expand_dims(self._start_tokens, 1), [1, self._beam_width])
 343     self._start_inputs = self._embedding_fn(self._start_tokens)
 344 
 345     self._finished = array_ops.one_hot(
 346         array_ops.zeros([self._batch_size], dtype=dtypes.int32),
 347         depth=self._beam_width,
 348         on_value=False,
 349         off_value=True,
 350         dtype=dtypes.bool)
 351 
 352   @property
 353   def batch_size(self):
 354     return self._batch_size
 355 
 356   def _rnn_output_size(self):
 357     size = self._cell.output_size
 358     if self._output_layer is None:
 359       return size
 360     else:
 361       # To use layer's compute_output_shape, we need to convert the
 362       # RNNCell's output_size entries into shapes with an unknown
 363       # batch size.  We then pass this through the layer's
 364       # compute_output_shape and read off all but the first (batch)
 365       # dimensions to get the output size of the rnn with the layer
 366       # applied to the top.
 367       output_shape_with_unknown_batch = nest.map_structure(
 368           lambda s: tensor_shape.TensorShape([None]).concatenate(s), size)
 369       layer_output_shape = self._output_layer.compute_output_shape(
 370           output_shape_with_unknown_batch)
 371       return nest.map_structure(lambda s: s[1:], layer_output_shape)
 372 
 373   @property
 374   def tracks_own_finished(self):
 375     """The BeamSearchDecoder shuffles its beams and their finished state.
 376 
 377     For this reason, it conflicts with the `dynamic_decode` function's
 378     tracking of finished states.  Setting this property to true avoids
 379     early stopping of decoding due to mismanagement of the finished state
 380     in `dynamic_decode`.
 381 
 382     Returns:
 383       `True`.
 384     """
 385     return True

 396   def output_dtype(self):
 397     # Assume the dtype of the cell is the output_size structure
 398     # containing the input_state's first component's dtype.
 399     # Return that structure and int32 (the id)
 400     dtype = nest.flatten(self._initial_cell_state)[0].dtype

 406   def initialize(self, name=None):
 407     """Initialize the decoder.
 408 
 409     Args:
 410       name: Name scope for any created operations.
 411 
 412     Returns:
 413       `(finished, start_inputs, initial_state)`.
 414     """
 415     finished, start_inputs = self._finished, self._start_inputs
 416 
 417     dtype = nest.flatten(self._initial_cell_state)[0].dtype
 418     log_probs = array_ops.one_hot(  # shape(batch_sz, beam_sz)
 419         array_ops.zeros([self._batch_size], dtype=dtypes.int32),
 420         depth=self._beam_width,
 421         on_value=ops.convert_to_tensor(0.0, dtype=dtype),
 422         off_value=ops.convert_to_tensor(-np.Inf, dtype=dtype),
 423         dtype=dtype)
 424     init_attention_probs = get_attention_probs(
 425         self._initial_cell_state, self._coverage_penalty_weight)
 426     if init_attention_probs is None:
 427       init_attention_probs = ()
 428 
 429     initial_state = BeamSearchDecoderState(
 430         cell_state=self._initial_cell_state,
 431         log_probs=log_probs,
 432         finished=finished,
 433         lengths=array_ops.zeros(
 434             [self._batch_size, self._beam_width], dtype=dtypes.int64),
 435         accumulated_attention_probs=init_attention_probs)
 436 
 437     return (finished, start_inputs, initial_state)
 438 
 439   def finalize(self, outputs, final_state, sequence_lengths):
 440     """Finalize and return the predicted_ids.
 441 
 442     Args:
 443       outputs: An instance of BeamSearchDecoderOutput.
 444       final_state: An instance of BeamSearchDecoderState. Passed through to the
 445         output.
 446       sequence_lengths: An `int64` tensor shaped `[batch_size, beam_width]`.
 447         The sequence lengths determined for each beam during decode.
 448         **NOTE** These are ignored; the updated sequence lengths are stored in
 449         `final_state.lengths`.
 450 
 451     Returns:
 452       outputs: An instance of `FinalBeamSearchDecoderOutput` where the
 453         predicted_ids are the result of calling _gather_tree.
 454       final_state: The same input instance of `BeamSearchDecoderState`.
 455     """
 456     del sequence_lengths
 457     # Get max_sequence_length across all beams for each batch.
 458     max_sequence_lengths = math_ops.to_int32(
 459         math_ops.reduce_max(final_state.lengths, axis=1))
 460     predicted_ids = beam_search_ops.gather_tree(
 461         outputs.predicted_ids,
 462         outputs.parent_ids,
 463         max_sequence_lengths=max_sequence_lengths,
 464         end_token=self._end_token)
 465     if self._reorder_tensor_arrays:
 466       final_state = final_state._replace(cell_state=nest.map_structure(
 467           lambda t: self._maybe_sort_array_beams(
 468               t, outputs.parent_ids, final_state.lengths),
 469           final_state.cell_state))
 470     outputs = FinalBeamSearchDecoderOutput(
 471         beam_search_decoder_output=outputs, predicted_ids=predicted_ids)
 472     return outputs, final_state

 474   def _merge_batch_beams(self, t, s=None):
 475     """Merges the tensor from a batch of beams into a batch by beams.
 476 
 477     More exactly, t is a tensor of dimension [batch_size, beam_width, s]. We
 478     reshape this into [batch_size*beam_width, s]
 479 
 480     Args:
 481       t: Tensor of dimension [batch_size, beam_width, s]
 482       s: (Possibly known) depth shape.
 483 
 484     Returns:
 485       A reshaped version of t with dimension [batch_size * beam_width, s].
 486     """
 487     if isinstance(s, ops.Tensor):
 488       s = tensor_shape.as_shape(tensor_util.constant_value(s))
 489     else:
 490       s = tensor_shape.TensorShape(s)
 491     t_shape = array_ops.shape(t)
 492     static_batch_size = tensor_util.constant_value(self._batch_size)
 493     batch_size_beam_width = (
 494         None
 495         if static_batch_size is None else static_batch_size * self._beam_width)
 496     reshaped_t = array_ops.reshape(
 497         t,
 498         array_ops.concat(([self._batch_size * self._beam_width], t_shape[2:]),
 499                          0))
 500     reshaped_t.set_shape(
 501         (tensor_shape.TensorShape([batch_size_beam_width]).concatenate(s)))
 502     return reshaped_t
 503 
 504   def _split_batch_beams(self, t, s=None):
 505     """Splits the tensor from a batch by beams into a batch of beams.
 506 
 507     More exactly, t is a tensor of dimension [batch_size*beam_width, s]. We
 508     reshape this into [batch_size, beam_width, s]
 509 
 510     Args:
 511       t: Tensor of dimension [batch_size*beam_width, s].
 512       s: (Possibly known) depth shape.
 513 
 514     Returns:
 515       A reshaped version of t with dimension [batch_size, beam_width, s].
 516 
 517     Raises:
 518       ValueError: If, after reshaping, the new tensor is not shaped
 519         `[batch_size, beam_width, s]` (assuming batch_size and beam_width
 520         are known statically).
 521     """
 522     if isinstance(s, ops.Tensor):
 523       s = tensor_shape.TensorShape(tensor_util.constant_value(s))
 524     else:
 525       s = tensor_shape.TensorShape(s)
 526     t_shape = array_ops.shape(t)
 527     reshaped_t = array_ops.reshape(
 528         t,
 529         array_ops.concat(([self._batch_size, self._beam_width], t_shape[1:]),
 530                          0))
 531     static_batch_size = tensor_util.constant_value(self._batch_size)
 532     expected_reshaped_shape = tensor_shape.TensorShape(
 533         [static_batch_size, self._beam_width]).concatenate(s)
 534     if not reshaped_t.shape.is_compatible_with(expected_reshaped_shape):
 535       raise ValueError("Unexpected behavior when reshaping between beam width "
 536                        "and batch size.  The reshaped tensor has shape: %s.  "
 537                        "We expected it to have shape "
 538                        "(batch_size, beam_width, depth) == %s.  Perhaps you "
 539                        "forgot to create a zero_state with "
 540                        "batch_size=encoder_batch_size * beam_width?" %
 541                        (reshaped_t.shape, expected_reshaped_shape))
 542     reshaped_t.set_shape(expected_reshaped_shape)
 543     return reshaped_t

 545   def _maybe_split_batch_beams(self, t, s):
 546     """Maybe splits the tensor from a batch by beams into a batch of beams.
 547 
 548     We do this so that we can use nest and not run into problems with shapes.
 549 
 550     Args:
 551       t: `Tensor`, either scalar or shaped `[batch_size * beam_width] + s`.
 552       s: `Tensor`, Python int, or `TensorShape`.
 553 
 554     Returns:
 555       If `t` is a matrix or higher order tensor, then the return value is
 556       `t` reshaped to `[batch_size, beam_width] + s`.  Otherwise `t` is
 557       returned unchanged.
 558 
 559     Raises:
 560       ValueError: If the rank of `t` is not statically known.
 561     """
 562     if isinstance(t, tensor_array_ops.TensorArray):
 563       return t
 564     _check_maybe(t)
 565     if t.shape.ndims >= 1:
 566       return self._split_batch_beams(t, s)
 567     else:
 568       return t
 569 
 570   def _maybe_merge_batch_beams(self, t, s):
 571     """Splits the tensor from a batch by beams into a batch of beams.
 572 
 573     More exactly, `t` is a tensor of dimension `[batch_size * beam_width] + s`,
 574     then we reshape it to `[batch_size, beam_width] + s`.
 575 
 576     Args:
 577       t: `Tensor` of dimension `[batch_size * beam_width] + s`.
 578       s: `Tensor`, Python int, or `TensorShape`.
 579 
 580     Returns:
 581       A reshaped version of t with shape `[batch_size, beam_width] + s`.
 582 
 583     Raises:
 584       ValueError:  If the rank of `t` is not statically known.
 585     """
 586     if isinstance(t, tensor_array_ops.TensorArray):
 587       return t
 588     _check_maybe(t)
 589     if t.shape.ndims >= 2:
 590       return self._merge_batch_beams(t, s)
 591     else:
 592       return t

 594   def _maybe_sort_array_beams(self, t, parent_ids, sequence_length):
 595     """Maybe sorts beams within a `TensorArray`.
 596 
 597     Args:
 598       t: A `TensorArray` of size `max_time` that contains `Tensor`s of shape
 599         `[batch_size, beam_width, s]` or `[batch_size * beam_width, s]` where
 600         `s` is the depth shape.
 601       parent_ids: The parent ids of shape `[max_time, batch_size, beam_width]`.
 602       sequence_length: The sequence length of shape `[batch_size, beam_width]`.
 603 
 604     Returns:
 605       A `TensorArray` where beams are sorted in each `Tensor` or `t` itself if
 606       it is not a `TensorArray` or does not meet shape requirements.
 607     """
 608     if not isinstance(t, tensor_array_ops.TensorArray):
 609       return t
 610     # pylint: disable=protected-access
 611     if (not t._infer_shape or not t._element_shape
 612         or t._element_shape[0].ndims is None
 613         or t._element_shape[0].ndims < 1):
 614       shape = (
 615           t._element_shape[0] if t._infer_shape and t._element_shape
 616           else tensor_shape.TensorShape(None))
 617       tf_logging.warn("The TensorArray %s in the cell state is not amenable to "
 618                       "sorting based on the beam search result. For a "
 619                       "TensorArray to be sorted, its elements shape must be "
 620                       "defined and have at least a rank of 1, but saw shape: %s"
 621                       % (t.handle.name, shape))
 622       return t
 623     shape = t._element_shape[0]
 624     # pylint: enable=protected-access
 625     if not _check_static_batch_beam_maybe(
 626         shape, tensor_util.constant_value(self._batch_size), self._beam_width):
 627       return t
 628     t = t.stack()
 629     with ops.control_dependencies(
 630         [_check_batch_beam(t, self._batch_size, self._beam_width)]):
 631       return gather_tree_from_array(t, parent_ids, sequence_length)

 633   def step(self, time, inputs, state, name=None):
 634     """Perform a decoding step.
 635 
 636     Args:
 637       time: scalar `int32` tensor.
 638       inputs: A (structure of) input tensors.
 639       state: A (structure of) state tensors and TensorArrays.
 640       name: Name scope for any created operations.
 641 
 642     Returns:
 643       `(outputs, next_state, next_inputs, finished)`.
 644     """
 645     batch_size = self._batch_size
 646     beam_width = self._beam_width
 647     end_token = self._end_token
 648     length_penalty_weight = self._length_penalty_weight
 649     coverage_penalty_weight = self._coverage_penalty_weight
 650 
 651     with ops.name_scope(name, "BeamSearchDecoderStep", (time, inputs, state)):
 652       cell_state = state.cell_state
 653       inputs = nest.map_structure(
 654           lambda inp: self._merge_batch_beams(inp, s=inp.shape[2:]), inputs)
 655       cell_state = nest.map_structure(self._maybe_merge_batch_beams, cell_state,
 656                                       self._cell.state_size)
 657       cell_outputs, next_cell_state = self._cell(inputs, cell_state)
 658       cell_outputs = nest.map_structure(
 659           lambda out: self._split_batch_beams(out, out.shape[1:]), cell_outputs)
 660       next_cell_state = nest.map_structure(
 661           self._maybe_split_batch_beams, next_cell_state, self._cell.state_size)
 662 
 663       if self._output_layer is not None:
 664         cell_outputs = self._output_layer(cell_outputs)
 665 
 666       beam_search_output, beam_search_state = _beam_search_step(
 667           time=time,
 668           logits=cell_outputs,
 669           next_cell_state=next_cell_state,
 670           beam_state=state,
 671           batch_size=batch_size,
 672           beam_width=beam_width,
 673           end_token=end_token,
 674           length_penalty_weight=length_penalty_weight,
 675           coverage_penalty_weight=coverage_penalty_weight)
 676 
 677       finished = beam_search_state.finished
 678       sample_ids = beam_search_output.predicted_ids
 679       next_inputs = control_flow_ops.cond(
 680           math_ops.reduce_all(finished), lambda: self._start_inputs,
 681           lambda: self._embedding_fn(sample_ids))
 682 
 683     return (beam_search_output, beam_search_state, next_inputs, finished)

 686 def _beam_search_step(time, logits, next_cell_state, beam_state, batch_size,
 687                       beam_width, end_token, length_penalty_weight,
 688                       coverage_penalty_weight):
 689   """Performs a single step of Beam Search Decoding.
 690 
 691   Args:
 692     time: Beam search time step, should start at 0. At time 0 we assume
 693       that all beams are equal and consider only the first beam for
 694       continuations.
 695     logits: Logits at the current time step. A tensor of shape
 696       `[batch_size, beam_width, vocab_size]`
 697     next_cell_state: The next state from the cell, e.g. an instance of
 698       AttentionWrapperState if the cell is attentional.
 699     beam_state: Current state of the beam search.
 700       An instance of `BeamSearchDecoderState`.
 701     batch_size: The batch size for this input.
 702     beam_width: Python int.  The size of the beams.
 703     end_token: The int32 end token.
 704     length_penalty_weight: Float weight to penalize length. Disabled with 0.0.
 705     coverage_penalty_weight: Float weight to penalize the coverage of source
 706       sentence. Disabled with 0.0.
 707 
 708   Returns:
 709     A new beam state.
 710   """
 711   static_batch_size = tensor_util.constant_value(batch_size)
 712 
 713   # Calculate the current lengths of the predictions
 714   prediction_lengths = beam_state.lengths
 715   previously_finished = beam_state.finished
 716   not_finished = math_ops.logical_not(previously_finished)
 717 
 718   # Calculate the total log probs for the new hypotheses
 719   # Final Shape: [batch_size, beam_width, vocab_size]
 720   step_log_probs = nn_ops.log_softmax(logits)
 721   step_log_probs = _mask_probs(step_log_probs, end_token, previously_finished)
 722   total_probs = array_ops.expand_dims(beam_state.log_probs, 2) + step_log_probs
 723 
 724   # Calculate the continuation lengths by adding to all continuing beams.
 725   vocab_size = logits.shape[-1].value or array_ops.shape(logits)[-1]
 726   lengths_to_add = array_ops.one_hot(
 727       indices=array_ops.fill([batch_size, beam_width], end_token),
 728       depth=vocab_size,
 729       on_value=np.int64(0),
 730       off_value=np.int64(1),
 731       dtype=dtypes.int64)
 732   add_mask = math_ops.to_int64(not_finished)
 733   lengths_to_add *= array_ops.expand_dims(add_mask, 2)
 734   new_prediction_lengths = (
 735       lengths_to_add + array_ops.expand_dims(prediction_lengths, 2))
 736 
 737   # Calculate the accumulated attention probabilities if coverage penalty is
 738   # enabled.
 739   accumulated_attention_probs = None
 740   attention_probs = get_attention_probs(
 741       next_cell_state, coverage_penalty_weight)
 742   if attention_probs is not None:
 743     attention_probs *= array_ops.expand_dims(math_ops.to_float(not_finished), 2)
 744     accumulated_attention_probs = (
 745         beam_state.accumulated_attention_probs + attention_probs)
 746 
 747   # Calculate the scores for each beam
 748   scores = _get_scores(
 749       log_probs=total_probs,
 750       sequence_lengths=new_prediction_lengths,
 751       length_penalty_weight=length_penalty_weight,
 752       coverage_penalty_weight=coverage_penalty_weight,
 753       finished=previously_finished,
 754       accumulated_attention_probs=accumulated_attention_probs)
 755 
 756   time = ops.convert_to_tensor(time, name="time")
 757   # During the first time step we only consider the initial beam
 758   scores_flat = array_ops.reshape(scores, [batch_size, -1])
 759 
 760   # Pick the next beams according to the specified successors function
 761   next_beam_size = ops.convert_to_tensor(
 762       beam_width, dtype=dtypes.int32, name="beam_width")
 763   next_beam_scores, word_indices = nn_ops.top_k(scores_flat, k=next_beam_size)
 764 
 765   next_beam_scores.set_shape([static_batch_size, beam_width])
 766   word_indices.set_shape([static_batch_size, beam_width])
 767 
 768   # Pick out the probs, beam_ids, and states according to the chosen predictions
 769   next_beam_probs = _tensor_gather_helper(
 770       gather_indices=word_indices,
 771       gather_from=total_probs,
 772       batch_size=batch_size,
 773       range_size=beam_width * vocab_size,
 774       gather_shape=[-1],
 775       name="next_beam_probs")
 776   # Note: just doing the following
 777   #   math_ops.to_int32(word_indices % vocab_size,
 778   #       name="next_beam_word_ids")
 779   # would be a lot cleaner but for reasons unclear, that hides the results of
 780   # the op which prevents capturing it with tfdbg debug ops.
 781   raw_next_word_ids = math_ops.mod(
 782       word_indices, vocab_size, name="next_beam_word_ids")
 783   next_word_ids = math_ops.to_int32(raw_next_word_ids)
 784   next_beam_ids = math_ops.to_int32(
 785       word_indices / vocab_size, name="next_beam_parent_ids")
 786 
 787   # Append new ids to current predictions
 788   previously_finished = _tensor_gather_helper(
 789       gather_indices=next_beam_ids,
 790       gather_from=previously_finished,
 791       batch_size=batch_size,
 792       range_size=beam_width,
 793       gather_shape=[-1])
 794   next_finished = math_ops.logical_or(
 795       previously_finished,
 796       math_ops.equal(next_word_ids, end_token),
 797       name="next_beam_finished")
 798 
 799   # Calculate the length of the next predictions.
 800   # 1. Finished beams remain unchanged.
 801   # 2. Beams that are now finished (EOS predicted) have their length
 802   #    increased by 1.
 803   # 3. Beams that are not yet finished have their length increased by 1.
 804   lengths_to_add = math_ops.to_int64(math_ops.logical_not(previously_finished))
 805   next_prediction_len = _tensor_gather_helper(
 806       gather_indices=next_beam_ids,
 807       gather_from=beam_state.lengths,
 808       batch_size=batch_size,
 809       range_size=beam_width,
 810       gather_shape=[-1])
 811   next_prediction_len += lengths_to_add
 812   next_accumulated_attention_probs = ()
 813   if accumulated_attention_probs is not None:
 814     next_accumulated_attention_probs = _tensor_gather_helper(
 815         gather_indices=next_beam_ids,
 816         gather_from=accumulated_attention_probs,
 817         batch_size=batch_size,
 818         range_size=beam_width,
 819         gather_shape=[batch_size * beam_width, -1],
 820         name="next_accumulated_attention_probs")
 821 
 822   # Pick out the cell_states according to the next_beam_ids. We use a
 823   # different gather_shape here because the cell_state tensors, i.e.
 824   # the tensors that would be gathered from, all have dimension
 825   # greater than two and we need to preserve those dimensions.
 826   # pylint: disable=g-long-lambda
 827   next_cell_state = nest.map_structure(
 828       lambda gather_from: _maybe_tensor_gather_helper(
 829           gather_indices=next_beam_ids,
 830           gather_from=gather_from,
 831           batch_size=batch_size,
 832           range_size=beam_width,
 833           gather_shape=[batch_size * beam_width, -1]),
 834       next_cell_state)
 835   # pylint: enable=g-long-lambda
 836 
 837   next_state = BeamSearchDecoderState(
 838       cell_state=next_cell_state,
 839       log_probs=next_beam_probs,
 840       lengths=next_prediction_len,
 841       finished=next_finished,
 842       accumulated_attention_probs=next_accumulated_attention_probs)
 843 
 844   output = BeamSearchDecoderOutput(
 845       scores=next_beam_scores,
 846       predicted_ids=next_word_ids,
 847       parent_ids=next_beam_ids)
 848 
 849   return output, next_state

 852 def get_attention_probs(next_cell_state, coverage_penalty_weight):
 853   """Get attention probabilities from the cell state.
 854 
 855   Args:
 856     next_cell_state: The next state from the cell, e.g. an instance of
 857       AttentionWrapperState if the cell is attentional.
 858     coverage_penalty_weight: Float weight to penalize the coverage of source
 859       sentence. Disabled with 0.0.
 860 
 861   Returns:
 862     The attention probabilities with shape `[batch_size, beam_width, max_time]`
 863     if coverage penalty is enabled. Otherwise, returns None.
 864 
 865   Raises:
 866     ValueError: If no cell is attentional but coverage penalty is enabled.
 867   """
 868   if coverage_penalty_weight == 0.0:
 869     return None
 870 
 871   # Attention probabilities of each attention layer. Each with shape
 872   # `[batch_size, beam_width, max_time]`.
 873   probs_per_attn_layer = []
 874   if isinstance(next_cell_state, attention_wrapper.AttentionWrapperState):
 875     probs_per_attn_layer = [attention_probs_from_attn_state(next_cell_state)]
 876   elif isinstance(next_cell_state, tuple):
 877     for state in next_cell_state:
 878       if isinstance(state, attention_wrapper.AttentionWrapperState):
 879         probs_per_attn_layer.append(attention_probs_from_attn_state(state))
 880 
 881   if not probs_per_attn_layer:
 882     raise ValueError(
 883         "coverage_penalty_weight must be 0.0 if no cell is attentional.")
 884 
 885   if len(probs_per_attn_layer) == 1:
 886     attention_probs = probs_per_attn_layer[0]
 887   else:
 888     # Calculate the average attention probabilities from all attention layers.
 889     attention_probs = [
 890         array_ops.expand_dims(prob, -1) for prob in probs_per_attn_layer]
 891     attention_probs = array_ops.concat(attention_probs, -1)
 892     attention_probs = math_ops.reduce_mean(attention_probs, -1)
 893 
 894   return attention_probs

 897 def _get_scores(log_probs, sequence_lengths, length_penalty_weight,
 898                 coverage_penalty_weight, finished, accumulated_attention_probs):
 899   """Calculates scores for beam search hypotheses.
 900 
 901   Args:
 902     log_probs: The log probabilities with shape
 903       `[batch_size, beam_width, vocab_size]`.
 904     sequence_lengths: The array of sequence lengths.
 905     length_penalty_weight: Float weight to penalize length. Disabled with 0.0.
 906     coverage_penalty_weight: Float weight to penalize the coverage of source
 907       sentence. Disabled with 0.0.
 908     finished: A boolean tensor of shape `[batch_size, beam_width]` that
 909       specifies which elements in the beam are finished already.
 910     accumulated_attention_probs: Accumulated attention probabilities up to the
 911       current time step, with shape `[batch_size, beam_width, max_time]` if
 912       coverage_penalty_weight is not 0.0.
 913 
 914   Returns:
 915     The scores normalized by the length_penalty and coverage_penalty.
 916 
 917   Raises:
 918     ValueError: accumulated_attention_probs is None when coverage penalty is
 919       enabled.
 920   """
 921   length_penalty_ = _length_penalty(
 922       sequence_lengths=sequence_lengths, penalty_factor=length_penalty_weight)
 923   scores = log_probs / length_penalty_
 924 
 925   coverage_penalty_weight = ops.convert_to_tensor(
 926       coverage_penalty_weight, name="coverage_penalty_weight")
 927   if coverage_penalty_weight.shape.ndims != 0:
 928     raise ValueError("coverage_penalty_weight should be a scalar, "
 929                      "but saw shape: %s" % coverage_penalty_weight.shape)
 930 
 931   if tensor_util.constant_value(coverage_penalty_weight) == 0.0:
 932     return scores
 933 
 934   if accumulated_attention_probs is None:
 935     raise ValueError(
 936         "accumulated_attention_probs can be None only if coverage penalty is "
 937         "disabled.")
 938 
 939   # Add source sequence length mask before computing coverage penalty.
 940   accumulated_attention_probs = array_ops.where(
 941       math_ops.equal(accumulated_attention_probs, 0.0),
 942       array_ops.ones_like(accumulated_attention_probs),
 943       accumulated_attention_probs)
 944 
 945   # coverage penalty =
 946   #     sum over `max_time` {log(min(accumulated_attention_probs, 1.0))}
 947   coverage_penalty = math_ops.reduce_sum(
 948       math_ops.log(math_ops.minimum(accumulated_attention_probs, 1.0)), 2)
 949   # Apply coverage penalty to finished predictions.
 950   coverage_penalty *= math_ops.to_float(finished)
 951   weighted_coverage_penalty = coverage_penalty * coverage_penalty_weight
 952   # Reshape from [batch_size, beam_width] to [batch_size, beam_width, 1]
 953   weighted_coverage_penalty = array_ops.expand_dims(
 954       weighted_coverage_penalty, 2)
 955   return scores + weighted_coverage_penalty

 958 def attention_probs_from_attn_state(attention_state):
 959   """Calculates the average attention probabilities.
 960 
 961   Args:
 962     attention_state: An instance of `AttentionWrapperState`.
 963 
 964   Returns:
 965     The attention probabilities in the given AttentionWrapperState.
 966     If there're multiple attention mechanisms, return the average value from
 967     all attention mechanisms.
 968   """
 969   # Attention probabilities over time steps, with shape
 970   # `[batch_size, beam_width, max_time]`.
 971   attention_probs = attention_state.alignments
 972   if isinstance(attention_probs, tuple):
 973     attention_probs = [
 974         array_ops.expand_dims(prob, -1) for prob in attention_probs]
 975     attention_probs = array_ops.concat(attention_probs, -1)
 976     attention_probs = math_ops.reduce_mean(attention_probs, -1)
 977   return attention_probs

 980 def _length_penalty(sequence_lengths, penalty_factor):
 981   """Calculates the length penalty. See https://arxiv.org/abs/1609.08144.
 982 
 983   Returns the length penalty tensor:
 984   ```
 985   [(5+sequence_lengths)/6]**penalty_factor
 986   ```
 987   where all operations are performed element-wise.
 988 
 989   Args:
 990     sequence_lengths: `Tensor`, the sequence lengths of each hypotheses.
 991     penalty_factor: A scalar that weights the length penalty.
 992 
 993   Returns:
 994     If the penalty is `0`, returns the scalar `1.0`.  Otherwise returns
 995     the length penalty factor, a tensor with the same shape as
 996     `sequence_lengths`.
 997   """
 998   penalty_factor = ops.convert_to_tensor(penalty_factor, name="penalty_factor")
 999   penalty_factor.set_shape(())  # penalty should be a scalar.
1000   static_penalty = tensor_util.constant_value(penalty_factor)
1001   if static_penalty is not None and static_penalty == 0:
1002     return 1.0
1003   return math_ops.div((5. + math_ops.to_float(sequence_lengths))
1004                       **penalty_factor, (5. + 1.)**penalty_factor)

1007 def _mask_probs(probs, eos_token, finished):
1008   """Masks log probabilities.
1009 
1010   The result is that finished beams allocate all probability mass to eos and
1011   unfinished beams remain unchanged.
1012 
1013   Args:
1014     probs: Log probabilities of shape `[batch_size, beam_width, vocab_size]`
1015     eos_token: An int32 id corresponding to the EOS token to allocate
1016       probability to.
1017     finished: A boolean tensor of shape `[batch_size, beam_width]` that
1018       specifies which elements in the beam are finished already.
1019 
1020   Returns:
1021     A tensor of shape `[batch_size, beam_width, vocab_size]`, where unfinished
1022     beams stay unchanged and finished beams are replaced with a tensor with all
1023     probability on the EOS token.
1024   """
1025   vocab_size = array_ops.shape(probs)[2]
1026   # All finished examples are replaced with a vector that has all
1027   # probability on EOS
1028   finished_row = array_ops.one_hot(
1029       eos_token,
1030       vocab_size,
1031       dtype=probs.dtype,
1032       on_value=ops.convert_to_tensor(0., dtype=probs.dtype),
1033       off_value=probs.dtype.min)
1034   finished_probs = array_ops.tile(
1035       array_ops.reshape(finished_row, [1, 1, -1]),
1036       array_ops.concat([array_ops.shape(finished), [1]], 0))
1037   finished_mask = array_ops.tile(
1038       array_ops.expand_dims(finished, 2), [1, 1, vocab_size])
1039 
1040   return array_ops.where(finished_mask, finished_probs, probs)

1043 def _maybe_tensor_gather_helper(gather_indices, gather_from, batch_size,
1044                                 range_size, gather_shape):
1045   """Maybe applies _tensor_gather_helper.
1046 
1047   This applies _tensor_gather_helper when the gather_from dims is at least as
1048   big as the length of gather_shape. This is used in conjunction with nest so
1049   that we don't apply _tensor_gather_helper to inapplicable values like scalars.
1050 
1051   Args:
1052     gather_indices: The tensor indices that we use to gather.
1053     gather_from: The tensor that we are gathering from.
1054     batch_size: The batch size.
1055     range_size: The number of values in each range. Likely equal to beam_width.
1056     gather_shape: What we should reshape gather_from to in order to preserve the
1057       correct values. An example is when gather_from is the attention from an
1058       AttentionWrapperState with shape [batch_size, beam_width, attention_size].
1059       There, we want to preserve the attention_size elements, so gather_shape is
1060       [batch_size * beam_width, -1]. Then, upon reshape, we still have the
1061       attention_size as desired.
1062 
1063   Returns:
1064     output: Gathered tensor of shape tf.shape(gather_from)[:1+len(gather_shape)]
1065       or the original tensor if its dimensions are too small.
1066   """
1067   if isinstance(gather_from, tensor_array_ops.TensorArray):
1068     return gather_from
1069   _check_maybe(gather_from)
1070   if gather_from.shape.ndims >= len(gather_shape):
1071     return _tensor_gather_helper(
1072         gather_indices=gather_indices,
1073         gather_from=gather_from,
1074         batch_size=batch_size,
1075         range_size=range_size,
1076         gather_shape=gather_shape)
1077   else:
1078     return gather_from

1081 def _tensor_gather_helper(gather_indices,
1082                           gather_from,
1083                           batch_size,
1084                           range_size,
1085                           gather_shape,
1086                           name=None):
1087   """Helper for gathering the right indices from the tensor.
1088 
1089   This works by reshaping gather_from to gather_shape (e.g. [-1]) and then
1090   gathering from that according to the gather_indices, which are offset by
1091   the right amounts in order to preserve the batch order.
1092 
1093   Args:
1094     gather_indices: The tensor indices that we use to gather.
1095     gather_from: The tensor that we are gathering from.
1096     batch_size: The input batch size.
1097     range_size: The number of values in each range. Likely equal to beam_width.
1098     gather_shape: What we should reshape gather_from to in order to preserve the
1099       correct values. An example is when gather_from is the attention from an
1100       AttentionWrapperState with shape [batch_size, beam_width, attention_size].
1101       There, we want to preserve the attention_size elements, so gather_shape is
1102       [batch_size * beam_width, -1]. Then, upon reshape, we still have the
1103       attention_size as desired.
1104     name: The tensor name for set of operations. By default this is
1105       'tensor_gather_helper'. The final output is named 'output'.
1106 
1107   Returns:
1108     output: Gathered tensor of shape tf.shape(gather_from)[:1+len(gather_shape)]
1109   """
1110   with ops.name_scope(name, "tensor_gather_helper"):
1111     range_ = array_ops.expand_dims(math_ops.range(batch_size) * range_size, 1)
1112     gather_indices = array_ops.reshape(gather_indices + range_, [-1])
1113     output = array_ops.gather(
1114         array_ops.reshape(gather_from, gather_shape), gather_indices)
1115     final_shape = array_ops.shape(gather_from)[:1 + len(gather_shape)]
1116     static_batch_size = tensor_util.constant_value(batch_size)
1117     final_static_shape = (
1118         tensor_shape.TensorShape([static_batch_size]).concatenate(
1119             gather_from.shape[1:1 + len(gather_shape)]))
1120     output = array_ops.reshape(output, final_shape, name="output")
1121     output.set_shape(final_static_shape)
1122     return output
