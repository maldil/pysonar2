  48 @six.add_metaclass(abc.ABCMeta)
  49 class Decoder(object):
  50   """An RNN Decoder abstract interface object.
  51 
  52   Concepts used by this interface:
  53   - `inputs`: (structure of) tensors and TensorArrays that is passed as input to
  54     the RNNCell composing the decoder, at each time step.
  55   - `state`: (structure of) tensors and TensorArrays that is passed to the
  56     RNNCell instance as the state.
  57   - `finished`: boolean tensor telling whether each sequence in the batch is
  58     finished.
  59   - `outputs`: Instance of BasicDecoderOutput. Result of the decoding, at each
  60     time step.
  61   """
  62 
  63   @property
  64   def batch_size(self):
  65     """The batch size of input values."""
  66     raise NotImplementedError
  67 
  68   @property
  69   def output_size(self):
  70     """A (possibly nested tuple of...) integer[s] or `TensorShape` object[s]."""
  71     raise NotImplementedError
  72 
  73   @property
  74   def output_dtype(self):
  75     """A (possibly nested tuple of...) dtype[s]."""
  76     raise NotImplementedError
  77 
  78   @abc.abstractmethod
  79   def initialize(self, name=None):
  80     """Called before any decoding iterations.
  81 
  82     This methods must compute initial input values and initial state.
  83 
  84     Args:
  85       name: Name scope for any created operations.
  86 
  87     Returns:
  88       `(finished, initial_inputs, initial_state)`: initial values of
  89       'finished' flags, inputs and state.
  90     """
  91     raise NotImplementedError
  92 
  93   @abc.abstractmethod
  94   def step(self, time, inputs, state, name=None):
  95     """Called per step of decoding (but only once for dynamic decoding).
  96 
  97     Args:
  98       time: Scalar `int32` tensor. Current step number.
  99       inputs: RNNCell input (possibly nested tuple of) tensor[s] for this time
 100         step.
 101       state: RNNCell state (possibly nested tuple of) tensor[s] from previous
 102         time step.
 103       name: Name scope for any created operations.
 104 
 105     Returns:
 106       `(outputs, next_state, next_inputs, finished)`: `outputs` is an object
 107       containing the decoder output, `next_state` is a (structure of) state
 108       tensors and TensorArrays, `next_inputs` is the tensor that should be used
 109       as input for the next step, `finished` is a boolean tensor telling whether
 110       the sequence is complete, for each sequence in the batch.
 111     """
 112     raise NotImplementedError
 113 
 114   def finalize(self, outputs, final_state, sequence_lengths):
 115     raise NotImplementedError
 116 
 117   @property
 118   def tracks_own_finished(self):
 119     """Describes whether the Decoder keeps track of finished states.
 120 
 121     Most decoders will emit a true/false `finished` value independently
 122     at each time step.  In this case, the `dynamic_decode` function keeps track
 123     of which batch entries are already finished, and performs a logical OR to
 124     insert new batches to the finished set.
 125 
 126     Some decoders, however, shuffle batches / beams between time steps and
 127     `dynamic_decode` will mix up the finished state across these entries because
 128     it does not track the reshuffle across time steps.  In this case, it is
 129     up to the decoder to declare that it will keep track of its own finished
 130     state by setting this property to `True`.
 131 
 132     Returns:
 133       Python bool.
 134     """
 135     return False

 138 def _create_zero_outputs(size, dtype, batch_size):
 139   """Create a zero outputs Tensor structure."""
 140   def _create(s, d):
 141     return _zero_state_tensors(s, batch_size, d)
 142 
 143   return nest.map_structure(_create, size, dtype)

 146 def dynamic_decode(decoder,
 147                    output_time_major=False,
 148                    impute_finished=False,
 149                    maximum_iterations=None,
 150                    parallel_iterations=32,
 151                    swap_memory=False,
 152                    scope=None):
 153   """Perform dynamic decoding with `decoder`.
 154 
 155   Calls initialize() once and step() repeatedly on the Decoder object.
 156 
 157   Args:
 158     decoder: A `Decoder` instance.
 159     output_time_major: Python boolean.  Default: `False` (batch major).  If
 160       `True`, outputs are returned as time major tensors (this mode is faster).
 161       Otherwise, outputs are returned as batch major tensors (this adds extra
 162       time to the computation).
 163     impute_finished: Python boolean.  If `True`, then states for batch
 164       entries which are marked as finished get copied through and the
 165       corresponding outputs get zeroed out.  This causes some slowdown at
 166       each time step, but ensures that the final state and outputs have
 167       the correct values and that backprop ignores time steps that were
 168       marked as finished.
 169     maximum_iterations: `int32` scalar, maximum allowed number of decoding
 170        steps.  Default is `None` (decode until the decoder is fully done).
 171     parallel_iterations: Argument passed to `tf.while_loop`.
 172     swap_memory: Argument passed to `tf.while_loop`.
 173     scope: Optional variable scope to use.
 174 
 175   Returns:
 176     `(final_outputs, final_state, final_sequence_lengths)`.
 177 
 178   Raises:
 179     TypeError: if `decoder` is not an instance of `Decoder`.
 180     ValueError: if `maximum_iterations` is provided but is not a scalar.
 181   """
 182   if not isinstance(decoder, Decoder):
 183     raise TypeError("Expected decoder to be type Decoder, but saw: %s" %
 184                     type(decoder))
 185 
 186   with variable_scope.variable_scope(scope, "decoder") as varscope:
 187     # Determine context types.
 188     ctxt = ops.get_default_graph()._get_control_flow_context()  # pylint: disable=protected-access
 189     is_xla = control_flow_util.GetContainingXLAContext(ctxt) is not None
 190     in_while_loop = (
 191         control_flow_util.GetContainingWhileContext(ctxt) is not None)
 192     # Properly cache variable values inside the while_loop.
 193     # Don't set a caching device when running in a loop, since it is possible
 194     # that train steps could be wrapped in a tf.while_loop. In that scenario
 195     # caching prevents forward computations in loop iterations from re-reading
 196     # the updated weights.
 197     if not context.executing_eagerly() and not in_while_loop:
 198       if varscope.caching_device is None:
 199         varscope.set_caching_device(lambda op: op.device)
 200 
 201     if maximum_iterations is not None:
 202       maximum_iterations = ops.convert_to_tensor(
 203           maximum_iterations, dtype=dtypes.int32, name="maximum_iterations")
 204       if maximum_iterations.get_shape().ndims != 0:
 205         raise ValueError("maximum_iterations must be a scalar")
 206 
 207     initial_finished, initial_inputs, initial_state = decoder.initialize()
 208 
 209     zero_outputs = _create_zero_outputs(decoder.output_size,
 210                                         decoder.output_dtype,
 211                                         decoder.batch_size)
 212 
 213     if is_xla and maximum_iterations is None:
 214       raise ValueError("maximum_iterations is required for XLA compilation.")
 215     if maximum_iterations is not None:
 216       initial_finished = math_ops.logical_or(
 217           initial_finished, 0 >= maximum_iterations)
 218     initial_sequence_lengths = array_ops.zeros_like(
 219         initial_finished, dtype=dtypes.int32)
 220     initial_time = constant_op.constant(0, dtype=dtypes.int32)
 221 
 222     def _shape(batch_size, from_shape):
 223       if (not isinstance(from_shape, tensor_shape.TensorShape) or
 224           from_shape.ndims == 0):
 225         return tensor_shape.TensorShape(None)
 226       else:
 227         batch_size = tensor_util.constant_value(
 228             ops.convert_to_tensor(
 229                 batch_size, name="batch_size"))
 230         return tensor_shape.TensorShape([batch_size]).concatenate(from_shape)
 231 
 232     dynamic_size = maximum_iterations is None or not is_xla
 233 
 234     def _create_ta(s, d):
 235       return tensor_array_ops.TensorArray(
 236           dtype=d,
 237           size=0 if dynamic_size else maximum_iterations,
 238           dynamic_size=dynamic_size,
 239           element_shape=_shape(decoder.batch_size, s))
 240 
 241     initial_outputs_ta = nest.map_structure(_create_ta, decoder.output_size,
 242                                             decoder.output_dtype)
 243 
 244     def condition(unused_time, unused_outputs_ta, unused_state, unused_inputs,
 245                   finished, unused_sequence_lengths):
 246       return math_ops.logical_not(math_ops.reduce_all(finished))
 247 
 248     def body(time, outputs_ta, state, inputs, finished, sequence_lengths):
 249       """Internal while_loop body.
 250 
 251       Args:
 252         time: scalar int32 tensor.
 253         outputs_ta: structure of TensorArray.
 254         state: (structure of) state tensors and TensorArrays.
 255         inputs: (structure of) input tensors.
 256         finished: bool tensor (keeping track of what's finished).
 257         sequence_lengths: int32 tensor (keeping track of time of finish).
 258 
 259       Returns:
 260         `(time + 1, outputs_ta, next_state, next_inputs, next_finished,
 261           next_sequence_lengths)`.
 262         ```
 263       """
 264       (next_outputs, decoder_state, next_inputs,
 265        decoder_finished) = decoder.step(time, inputs, state)
 266       if decoder.tracks_own_finished:
 267         next_finished = decoder_finished
 268       else:
 269         next_finished = math_ops.logical_or(decoder_finished, finished)
 270       next_sequence_lengths = array_ops.where(
 271           math_ops.logical_not(finished),
 272           array_ops.fill(array_ops.shape(sequence_lengths), time + 1),
 273           sequence_lengths)
 274 
 275       nest.assert_same_structure(state, decoder_state)
 276       nest.assert_same_structure(outputs_ta, next_outputs)
 277       nest.assert_same_structure(inputs, next_inputs)
 278 
 279       # Zero out output values past finish
 280       if impute_finished:
 281         emit = nest.map_structure(
 282             lambda out, zero: array_ops.where(finished, zero, out),
 283             next_outputs,
 284             zero_outputs)
 285       else:
 286         emit = next_outputs
 287 
 288       # Copy through states past finish
 289       def _maybe_copy_state(new, cur):
 290         # TensorArrays and scalar states get passed through.
 291         if isinstance(cur, tensor_array_ops.TensorArray):
 292           pass_through = True
 293         else:
 294           new.set_shape(cur.shape)
 295           pass_through = (new.shape.ndims == 0)
 296         return new if pass_through else array_ops.where(finished, cur, new)
 297 
 298       if impute_finished:
 299         next_state = nest.map_structure(
 300             _maybe_copy_state, decoder_state, state)
 301       else:
 302         next_state = decoder_state
 303 
 304       outputs_ta = nest.map_structure(lambda ta, out: ta.write(time, out),
 305                                       outputs_ta, emit)
 306       return (time + 1, outputs_ta, next_state, next_inputs, next_finished,
 307               next_sequence_lengths)
 308 
 309     res = control_flow_ops.while_loop(
 310         condition,
 311         body,
 312         loop_vars=(
 313             initial_time,
 314             initial_outputs_ta,
 315             initial_state,
 316             initial_inputs,
 317             initial_finished,
 318             initial_sequence_lengths,
 319         ),
 320         parallel_iterations=parallel_iterations,
 321         maximum_iterations=maximum_iterations,
 322         swap_memory=swap_memory)
 323 
 324     final_outputs_ta = res[1]
 325     final_state = res[2]
 326     final_sequence_lengths = res[5]
 327 
 328     final_outputs = nest.map_structure(lambda ta: ta.stack(), final_outputs_ta)
 329 
 330     try:
 331       final_outputs, final_state = decoder.finalize(
 332           final_outputs, final_state, final_sequence_lengths)
 333     except NotImplementedError:
 334       pass
 335 
 336     if not output_time_major:
 337       final_outputs = nest.map_structure(_transpose_batch_time, final_outputs)
 338 
 339   return final_outputs, final_state, final_sequence_lengths
