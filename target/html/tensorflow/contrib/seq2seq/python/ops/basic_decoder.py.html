  39 class BasicDecoderOutput(
  40     collections.namedtuple("BasicDecoderOutput", ("rnn_output", "sample_id"))):
  41   pass

  43 
  44 class BasicDecoder(decoder.Decoder):
  45   """Basic sampling decoder."""
  46 
  47   def __init__(self, cell, helper, initial_state, output_layer=None):
  48     """Initialize BasicDecoder.
  49 
  50     Args:
  51       cell: An `RNNCell` instance.
  52       helper: A `Helper` instance.
  53       initial_state: A (possibly nested tuple of...) tensors and TensorArrays.
  54         The initial state of the RNNCell.
  55       output_layer: (Optional) An instance of `tf.layers.Layer`, i.e.,
  56         `tf.layers.Dense`. Optional layer to apply to the RNN output prior
  57         to storing the result or sampling.
  58 
  59     Raises:
  60       TypeError: if `cell`, `helper` or `output_layer` have an incorrect type.
  61     """
  62     rnn_cell_impl.assert_like_rnncell("cell", cell)
  63     if not isinstance(helper, helper_py.Helper):
  64       raise TypeError("helper must be a Helper, received: %s" % type(helper))
  65     if (output_layer is not None
  66         and not isinstance(output_layer, layers_base.Layer)):
  67       raise TypeError(
  68           "output_layer must be a Layer, received: %s" % type(output_layer))
  69     self._cell = cell
  70     self._helper = helper
  71     self._initial_state = initial_state
  72     self._output_layer = output_layer
  73 
  74   @property
  75   def batch_size(self):
  76     return self._helper.batch_size
  77 
  78   def _rnn_output_size(self):
  79     size = self._cell.output_size
  80     if self._output_layer is None:
  81       return size
  82     else:
  83       # To use layer's compute_output_shape, we need to convert the
  84       # RNNCell's output_size entries into shapes with an unknown
  85       # batch size.  We then pass this through the layer's
  86       # compute_output_shape and read off all but the first (batch)
  87       # dimensions to get the output size of the rnn with the layer
  88       # applied to the top.
  89       output_shape_with_unknown_batch = nest.map_structure(
  90           lambda s: tensor_shape.TensorShape([None]).concatenate(s),
  91           size)
  92       layer_output_shape = self._output_layer.compute_output_shape(
  93           output_shape_with_unknown_batch)
  94       return nest.map_structure(lambda s: s[1:], layer_output_shape)
  95 
  96   @property
  97   def output_size(self):
  98     # Return the cell output and the id
  99     return BasicDecoderOutput(
 100         rnn_output=self._rnn_output_size(),
 101         sample_id=self._helper.sample_ids_shape)
 102 
 103   @property
 104   def output_dtype(self):
 105     # Assume the dtype of the cell is the output_size structure
 106     # containing the input_state's first component's dtype.
 107     # Return that structure and the sample_ids_dtype from the helper.
 108     dtype = nest.flatten(self._initial_state)[0].dtype
 109     return BasicDecoderOutput(
 110         nest.map_structure(lambda _: dtype, self._rnn_output_size()),
 111         self._helper.sample_ids_dtype)
 112 
 113   def initialize(self, name=None):
 114     """Initialize the decoder.
 115 
 116     Args:
 117       name: Name scope for any created operations.
 118 
 119     Returns:
 120       `(finished, first_inputs, initial_state)`.
 121     """
 122     return self._helper.initialize() + (self._initial_state,)
 123 
 124   def step(self, time, inputs, state, name=None):
 125     """Perform a decoding step.
 126 
 127     Args:
 128       time: scalar `int32` tensor.
 129       inputs: A (structure of) input tensors.
 130       state: A (structure of) state tensors and TensorArrays.
 131       name: Name scope for any created operations.
 132 
 133     Returns:
 134       `(outputs, next_state, next_inputs, finished)`.
 135     """
 136     with ops.name_scope(name, "BasicDecoderStep", (time, inputs, state)):
 137       cell_outputs, cell_state = self._cell(inputs, state)
 138       if self._output_layer is not None:
 139         cell_outputs = self._output_layer(cell_outputs)
 140       sample_ids = self._helper.sample(
 141           time=time, outputs=cell_outputs, state=cell_state)
 142       (finished, next_inputs, next_state) = self._helper.next_inputs(
 143           time=time,
 144           outputs=cell_outputs,
 145           state=cell_state,
 146           sample_ids=sample_ids)
 147     outputs = BasicDecoderOutput(cell_outputs, sample_ids)
 148     return (outputs, next_state, next_inputs, finished)
