  15 """Defines `AdditiveExternalRegretOptimizer`.
  16 
  17 This optimizer minimizes a `ConstrainedMinimizationProblem` by introducing
  18 Lagrange multipliers, and using `tf.train.Optimizer`s to jointly optimize over
  19 the model parameters and Lagrange multipliers.
  20 
  21 For the purposes of constrained optimization, at least in theory,
  22 external-regret minimization suffices if the `ConstrainedMinimizationProblem`
  23 we're optimizing doesn't have any `proxy_constraints`, while swap-regret
  24 minimization should be used if `proxy_constraints` are present.
  25 
  26 For more specifics, please refer to:
  27 
  28 > Cotter, Jiang and Sridharan. "Two-Player Games for Efficient Non-Convex
  29 > Constrained Optimization".
  30 > [https://arxiv.org/abs/1804.06500](https://arxiv.org/abs/1804.06500)
  31 
  32 The formulation used by the AdditiveExternalRegretOptimizer--which is simply the
  33 usual Lagrangian formulation--can be found in Definition 1, and is discussed in
  34 Section 3. This optimizer is most similar to Algorithm 3 in Appendix C.3, with
  35 the two differences being that it uses proxy constraints (if they're provided)
  36 in the update of the model parameters, and uses `tf.train.Optimizer`s, instead
  37 of SGD, for the "inner" updates.
  38 """

  58 def _project_multipliers_wrt_euclidean_norm(multipliers, radius):
  59   """Projects its argument onto the feasible region.
  60 
  61   The feasible region is the set of all vectors with nonnegative elements that
  62   sum to at most `radius`.
  63 
  64   Args:
  65     multipliers: 1d tensor, the Lagrange multipliers to project.
  66     radius: float, the radius of the feasible region.
  67 
  68   Returns:
  69     The 1d tensor that results from projecting `multipliers` onto the feasible
  70       region w.r.t. the Euclidean norm.
  71 
  72   Raises:
  73     ValueError: if the `multipliers` tensor is not floating-point, does not have
  74       a fully-known shape, or is not one-dimensional.
  75   """
  76   if not multipliers.dtype.is_floating:
  77     raise ValueError("multipliers must have a floating-point dtype")
  78   multipliers_shape = multipliers.get_shape()
  79   if multipliers_shape.ndims is None:
  80     raise ValueError("multipliers must have known shape")
  81   if multipliers_shape.ndims != 1:
  82     raise ValueError(
  83         "multipliers must be one dimensional (instead is %d-dimensional)" %
  84         multipliers_shape.ndims)
  85   dimension = multipliers_shape[0].value
  86   if dimension is None:
  87     raise ValueError("multipliers must have fully-known shape")
  88 
  89   def while_loop_condition(iteration, multipliers, inactive, old_inactive):
  90     """Returns false if the while loop should terminate."""
  91     del multipliers  # Needed by the body, but not the condition.
  92     not_done = (iteration < dimension)
  93     not_converged = standard_ops.reduce_any(
  94         standard_ops.not_equal(inactive, old_inactive))
  95     return standard_ops.logical_and(not_done, not_converged)
  96 
  97   def while_loop_body(iteration, multipliers, inactive, old_inactive):
  98     """Performs one iteration of the projection."""
  99     del old_inactive  # Needed by the condition, but not the body.
 100     iteration += 1
 101     scale = standard_ops.minimum(
 102         0.0,
 103         (radius - standard_ops.reduce_sum(multipliers)) / standard_ops.maximum(
 104             1.0, standard_ops.reduce_sum(inactive)))
 105     multipliers = multipliers + (scale * inactive)
 106     new_inactive = standard_ops.cast(multipliers > 0, multipliers.dtype)
 107     multipliers = multipliers * new_inactive
 108     return (iteration, multipliers, new_inactive, inactive)
 109 
 110   iteration = standard_ops.constant(0)
 111   inactive = standard_ops.ones_like(multipliers, dtype=multipliers.dtype)
 112 
 113   # We actually want a do-while loop, so we explicitly call while_loop_body()
 114   # once before tf.while_loop().
 115   iteration, multipliers, inactive, old_inactive = while_loop_body(
 116       iteration, multipliers, inactive, inactive)
 117   iteration, multipliers, inactive, old_inactive = control_flow_ops.while_loop(
 118       while_loop_condition,
 119       while_loop_body,
 120       loop_vars=(iteration, multipliers, inactive, old_inactive),
 121       name="euclidean_projection")
 122 
 123   return multipliers

 127 class _ExternalRegretOptimizer(constrained_optimizer.ConstrainedOptimizer):
 128   """Base class representing an `_ExternalRegretOptimizer`.
 129 
 130   This class contains most of the logic for performing constrained
 131   optimization, minimizing external regret for the constraints player. What it
 132   *doesn't* do is keep track of the internal state (the Lagrange multipliers).
 133   Instead, the state is accessed via the _initial_state(),
 134   _lagrange_multipliers(), _constraint_grad_and_var() and _projection_op()
 135   methods.
 136 
 137   The reason for this is that we want to make it easy to implement different
 138   representations of the internal state.
 139 
 140   For more specifics, please refer to:
 141 
 142   > Cotter, Jiang and Sridharan. "Two-Player Games for Efficient Non-Convex
 143   > Constrained Optimization".
 144   > [https://arxiv.org/abs/1804.06500](https://arxiv.org/abs/1804.06500)
 145 
 146   The formulation used by `_ExternalRegretOptimizer`s--which is simply the usual
 147   Lagrangian formulation--can be found in Definition 1, and is discussed in
 148   Section 3. Such optimizers are most similar to Algorithm 3 in Appendix C.3.
 149   """
 150 
 151   def __init__(self, optimizer, constraint_optimizer=None):
 152     """Constructs a new `_ExternalRegretOptimizer`.
 153 
 154     The difference between `optimizer` and `constraint_optimizer` (if the latter
 155     is provided) is that the former is used for learning the model parameters,
 156     while the latter us used for the Lagrange multipliers. If no
 157     `constraint_optimizer` is provided, then `optimizer` is used for both.
 158 
 159     Args:
 160       optimizer: tf.train.Optimizer, used to optimize the objective and
 161         proxy_constraints portion of the ConstrainedMinimizationProblem. If
 162         constraint_optimizer is not provided, this will also be used to optimize
 163         the Lagrange multipliers.
 164       constraint_optimizer: optional tf.train.Optimizer, used to optimize the
 165         Lagrange multipliers.
 166 
 167     Returns:
 168       A new `_ExternalRegretOptimizer`.
 169     """
 170     super(_ExternalRegretOptimizer, self).__init__(optimizer=optimizer)
 171     self._constraint_optimizer = constraint_optimizer
 172 
 173   @property
 174   def constraint_optimizer(self):
 175     """Returns the `tf.train.Optimizer` used for the Lagrange multipliers."""
 176     return self._constraint_optimizer
 177

 194   def _minimize_constrained(self,
 195                             minimization_problem,
 196                             global_step=None,
 197                             var_list=None,
 198                             gate_gradients=train_optimizer.Optimizer.GATE_OP,
 199                             aggregation_method=None,
 200                             colocate_gradients_with_ops=False,
 201                             name=None,
 202                             grad_loss=None):
 203     """Returns an `Operation` for minimizing the constrained problem.
 204 
 205     The `optimizer` constructor parameter will be used to update the model
 206     parameters, while the Lagrange multipliers will be updated using
 207     `constrained_optimizer` (if provided) or `optimizer` (if not).
 208 
 209     Args:
 210       minimization_problem: ConstrainedMinimizationProblem, the problem to
 211         optimize.
 212       global_step: as in `tf.train.Optimizer`'s `minimize` method.
 213       var_list: as in `tf.train.Optimizer`'s `minimize` method.
 214       gate_gradients: as in `tf.train.Optimizer`'s `minimize` method.
 215       aggregation_method: as in `tf.train.Optimizer`'s `minimize` method.
 216       colocate_gradients_with_ops: as in `tf.train.Optimizer`'s `minimize`
 217         method.
 218       name: as in `tf.train.Optimizer`'s `minimize` method.
 219       grad_loss: as in `tf.train.Optimizer`'s `minimize` method.
 220 
 221     Raises:
 222       ValueError: If the minimization_problem tensors have different dtypes.
 223 
 224     Returns:
 225       `Operation`, the train_op.
 226     """
 227     objective = minimization_problem.objective
 228 
 229     constraints = minimization_problem.constraints
 230     proxy_constraints = minimization_problem.proxy_constraints
 231     if proxy_constraints is None:
 232       proxy_constraints = constraints
 233 
 234     # Make sure that the objective, constraints and proxy constraints all have
 235     # the same dtype.
 236     if (objective.dtype.base_dtype != constraints.dtype.base_dtype or
 237         objective.dtype.base_dtype != proxy_constraints.dtype.base_dtype):
 238       raise ValueError("objective, constraints and proxy_constraints must "
 239                        "have the same dtype")
 240 
 241     # Flatten both constraints tensors to 1d.
 242     num_constraints = minimization_problem.num_constraints
 243     constraints = standard_ops.reshape(constraints, shape=(num_constraints,))
 244     proxy_constraints = standard_ops.reshape(
 245         proxy_constraints, shape=(num_constraints,))
 246 
 247     # We use a lambda to initialize the state so that, if this function call is
 248     # inside the scope of a tf.control_dependencies() block, the dependencies
 249     # will not be applied to the initializer.
 250     state = standard_ops.Variable(
 251         lambda: self._initial_state(num_constraints),
 252         trainable=False,
 253         name="external_regret_optimizer_state")
 254 
 255     multipliers = self._lagrange_multipliers(state)
 256     loss = (
 257         objective + standard_ops.tensordot(
 258             standard_ops.cast(multipliers, proxy_constraints.dtype),
 259             proxy_constraints, 1))
 260     multipliers_gradient = standard_ops.cast(constraints, multipliers.dtype)
 261 
 262     update_ops = []
 263     if self.constraint_optimizer is None:
 264       # If we don't have a separate constraint_optimizer, then we use
 265       # self._optimizer for both the update of the model parameters, and that of
 266       # the internal state.
 267       grads_and_vars = self.optimizer.compute_gradients(
 268           loss,
 269           var_list=var_list,
 270           gate_gradients=gate_gradients,
 271           aggregation_method=aggregation_method,
 272           colocate_gradients_with_ops=colocate_gradients_with_ops,
 273           grad_loss=grad_loss)
 274       grads_and_vars.append(
 275           self._constraint_grad_and_var(state, multipliers_gradient))
 276       update_ops.append(
 277           self.optimizer.apply_gradients(grads_and_vars, name="update"))
 278     else:
 279       # If we have a separate constraint_optimizer, then we use self._optimizer
 280       # for the update of the model parameters, and self._constraint_optimizer
 281       # for that of the internal state.
 282       grads_and_vars = self.optimizer.compute_gradients(
 283           loss,
 284           var_list=var_list,
 285           gate_gradients=gate_gradients,
 286           aggregation_method=aggregation_method,
 287           colocate_gradients_with_ops=colocate_gradients_with_ops,
 288           grad_loss=grad_loss)
 289       multiplier_grads_and_vars = [
 290           self._constraint_grad_and_var(state, multipliers_gradient)
 291       ]
 292 
 293       gradients = [
 294           gradient for gradient, _ in grads_and_vars + multiplier_grads_and_vars
 295           if gradient is not None
 296       ]
 297       with ops.control_dependencies(gradients):
 298         update_ops.append(
 299             self.optimizer.apply_gradients(grads_and_vars, name="update"))
 300         update_ops.append(
 301             self.constraint_optimizer.apply_gradients(
 302                 multiplier_grads_and_vars, name="optimizer_state_update"))
 303 
 304     with ops.control_dependencies(update_ops):
 305       if global_step is None:
 306         # If we don't have a global step, just project, and we're done.
 307         return self._projection_op(state, name=name)
 308       else:
 309         # If we have a global step, then we need to increment it in addition to
 310         # projecting.
 311         projection_op = self._projection_op(state, name="project")
 312         with ops.colocate_with(global_step):
 313           global_step_op = state_ops.assign_add(
 314               global_step, 1, name="global_step_increment")
 315         return control_flow_ops.group(projection_op, global_step_op, name=name)

 318 class AdditiveExternalRegretOptimizer(_ExternalRegretOptimizer):
 319   """A `ConstrainedOptimizer` based on external-regret minimization.
 320 
 321   This `ConstrainedOptimizer` uses the given `tf.train.Optimizer`s to jointly
 322   minimize over the model parameters, and maximize over Lagrange multipliers,
 323   with the latter maximization using additive updates and an algorithm that
 324   minimizes external regret.
 325 
 326   For more specifics, please refer to:
 327 
 328   > Cotter, Jiang and Sridharan. "Two-Player Games for Efficient Non-Convex
 329   > Constrained Optimization".
 330   > [https://arxiv.org/abs/1804.06500](https://arxiv.org/abs/1804.06500)
 331 
 332   The formulation used by this optimizer--which is simply the usual Lagrangian
 333   formulation--can be found in Definition 1, and is discussed in Section 3. It
 334   is most similar to Algorithm 3 in Appendix C.3, with the two differences being
 335   that it uses proxy constraints (if they're provided) in the update of the
 336   model parameters, and uses `tf.train.Optimizer`s, instead of SGD, for the
 337   "inner" updates.
 338   """
 339 
 340   def __init__(self,
 341                optimizer,
 342                constraint_optimizer=None,
 343                maximum_multiplier_radius=None):
 344     """Constructs a new `AdditiveExternalRegretOptimizer`.
 345 
 346     Args:
 347       optimizer: tf.train.Optimizer, used to optimize the objective and
 348         proxy_constraints portion of ConstrainedMinimizationProblem. If
 349         constraint_optimizer is not provided, this will also be used to optimize
 350         the Lagrange multipliers.
 351       constraint_optimizer: optional tf.train.Optimizer, used to optimize the
 352         Lagrange multipliers.
 353       maximum_multiplier_radius: float, an optional upper bound to impose on the
 354         sum of the Lagrange multipliers.
 355 
 356     Returns:
 357       A new `AdditiveExternalRegretOptimizer`.
 358 
 359     Raises:
 360       ValueError: If the maximum_multiplier_radius parameter is nonpositive.
 361     """
 362     super(AdditiveExternalRegretOptimizer, self).__init__(
 363         optimizer=optimizer, constraint_optimizer=constraint_optimizer)
 364 
 365     if maximum_multiplier_radius and (maximum_multiplier_radius <= 0.0):
 366       raise ValueError("maximum_multiplier_radius must be strictly positive")
 367 
 368     self._maximum_multiplier_radius = maximum_multiplier_radius
 369 
 370   def _initial_state(self, num_constraints):
 371     # For an AdditiveExternalRegretOptimizer, the internal state is simply a
 372     # tensor of Lagrange multipliers with shape (m,), where m is the number of
 373     # constraints.
 374     #
 375     # FUTURE WORK: make the dtype a parameter.
 376     return standard_ops.zeros((num_constraints,), dtype=dtypes.float32)
 377 
 378   def _lagrange_multipliers(self, state):
 379     return state
 380 
 381   def _constraint_grad_and_var(self, state, gradient):
 382     # TODO(acotter): tf.colocate_with(), if colocate_gradients_with_ops is True?
 383     return (-gradient, state)
 384 
 385   def _projection_op(self, state, name=None):
 386     with ops.colocate_with(state):
 387       if self._maximum_multiplier_radius:
 388         projected_multipliers = _project_multipliers_wrt_euclidean_norm(
 389             state, self._maximum_multiplier_radius)
 390       else:
 391         projected_multipliers = standard_ops.maximum(state, 0.0)
 392       return state_ops.assign(state, projected_multipliers, name=name)
