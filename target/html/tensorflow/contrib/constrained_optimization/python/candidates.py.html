  15 """Code for optimizing over a set of candidate solutions.
  16 
  17 The functions in this file deal with the constrained problem:
  18 
  19 > minimize f(w)
  20 > s.t. g_i(w) <= 0 for all i in {0,1,...,m-1}
  21 
  22 Here, f(w) is the "objective function", and g_i(w) is the ith (of m) "constraint
  23 function". Given the values of the objective and constraint functions for a set
  24 of n "candidate solutions" {w_0,w_1,...,w_{n-1}} (for a total of n objective
  25 function values, and n*m constraint function values), the
  26 `find_best_candidate_distribution` function finds the best DISTRIBUTION over
  27 these candidates, while `find_best_candidate_index' heuristically finds the
  28 single best candidate.
  29 
  30 Both of these functions have dependencies on `scipy`, so if you want to call
  31 them, then you must make sure that `scipy` is available. The imports are
  32 performed inside the functions themselves, so if they're not actually called,
  33 then `scipy` is not needed.
  34 
  35 For more specifics, please refer to:
  36 
  37 > Cotter, Jiang and Sridharan. "Two-Player Games for Efficient Non-Convex
  38 > Constrained Optimization".
  39 > [https://arxiv.org/abs/1804.06500](https://arxiv.org/abs/1804.06500)
  40 
  41 The `find_best_candidate_distribution` function implements the approach
  42 described in Lemma 3, while `find_best_candidate_index` implements the heuristic
  43 used for hyperparameter search in the experiments of Section 5.2.
  44 """

  54 def _find_best_candidate_distribution_helper(objective_vector,
  55                                              constraints_matrix,
  56                                              maximum_violation=0.0):
  57   """Finds a distribution minimizing an objective subject to constraints.
  58 
  59   This function deals with the constrained problem:
  60 
  61   > minimize f(w)
  62   > s.t. g_i(w) <= 0 for all i in {0,1,...,m-1}
  63 
  64   Here, f(w) is the "objective function", and g_i(w) is the ith (of m)
  65   "constraint function". Given a set of n "candidate solutions"
  66   {w_0,w_1,...,w_{n-1}}, this function finds a distribution over these n
  67   candidates that, in expectation, minimizes the objective while violating
  68   the constraints by no more than `maximum_violation`. If no such distribution
  69   exists, it returns an error (using Go-style error reporting).
  70 
  71   The `objective_vector` parameter should be a numpy array with shape (n,), for
  72   which objective_vector[i] = f(w_i). Likewise, `constraints_matrix` should be a
  73   numpy array with shape (m,n), for which constraints_matrix[i,j] = g_i(w_j).
  74 
  75   This function will return a distribution for which at most m+1 probabilities,
  76   and often fewer, are nonzero.
  77 
  78   Args:
  79     objective_vector: numpy array of shape (n,), where n is the number of
  80       "candidate solutions". Contains the objective function values.
  81     constraints_matrix: numpy array of shape (m,n), where m is the number of
  82       constraints and n is the number of "candidate solutions". Contains the
  83       constraint violation magnitudes.
  84     maximum_violation: nonnegative float, the maximum amount by which any
  85       constraint may be violated, in expectation.
  86 
  87   Returns:
  88     A pair (`result`, `message`), exactly one of which is None. If `message` is
  89       None, then the `result` contains the optimal distribution as a numpy array
  90       of shape (n,). If `result` is None, then `message` contains an error
  91       message.
  92 
  93   Raises:
  94     ValueError: If `objective_vector` and `constraints_matrix` have inconsistent
  95       shapes, or if `maximum_violation` is negative.
  96     ImportError: If we're unable to import `scipy.optimize`.
  97   """
  98   if maximum_violation < 0.0:
  99     raise ValueError("maximum_violation must be nonnegative")
 100 
 101   mm, nn = np.shape(constraints_matrix)
 102   if (nn,) != np.shape(objective_vector):
 103     raise ValueError(
 104         "objective_vector must have shape (n,), and constraints_matrix (m, n),"
 105         " where n is the number of candidates, and m is the number of "
 106         "constraints")
 107 
 108   # We import scipy inline, instead of at the top of the file, so that a scipy
 109   # dependency is only introduced if either find_best_candidate_distribution()
 110   # or find_best_candidate_index() are actually called.
 111   import scipy.optimize  # pylint: disable=g-import-not-at-top
 112 
 113   # Feasibility (within maximum_violation) constraints.
 114   a_ub = constraints_matrix
 115   b_ub = np.full((mm, 1), maximum_violation)
 116   # Sum-to-one constraint.
 117   a_eq = np.ones((1, nn))
 118   b_eq = np.ones((1, 1))
 119   # Nonnegativity constraints.
 120   bounds = (0, None)
 121 
 122   result = scipy.optimize.linprog(
 123       objective_vector,
 124       A_ub=a_ub,
 125       b_ub=b_ub,
 126       A_eq=a_eq,
 127       b_eq=b_eq,
 128       bounds=bounds)
 129   # Go-style error reporting. We don't raise on error, since
 130   # find_best_candidate_distribution() needs to handle the failure case, and we
 131   # shouldn't use exceptions as flow-control.
 132   if not result.success:
 133     return (None, result.message)
 134   else:
 135     return (result.x, None)

 138 def find_best_candidate_distribution(objective_vector,
 139                                      constraints_matrix,
 140                                      epsilon=0.0):
 141   """Finds a distribution minimizing an objective subject to constraints.
 142 
 143   This function deals with the constrained problem:
 144 
 145   > minimize f(w)
 146   > s.t. g_i(w) <= 0 for all i in {0,1,...,m-1}
 147 
 148   Here, f(w) is the "objective function", and g_i(w) is the ith (of m)
 149   "constraint function". Given a set of n "candidate solutions"
 150   {w_0,w_1,...,w_{n-1}}, this function finds a distribution over these n
 151   candidates that, in expectation, minimizes the objective while violating
 152   the constraints by the smallest possible amount (with the amount being found
 153   via bisection search).
 154 
 155   The `objective_vector` parameter should be a numpy array with shape (n,), for
 156   which objective_vector[i] = f(w_i). Likewise, `constraints_matrix` should be a
 157   numpy array with shape (m,n), for which constraints_matrix[i,j] = g_i(w_j).
 158 
 159   This function will return a distribution for which at most m+1 probabilities,
 160   and often fewer, are nonzero.
 161 
 162   For more specifics, please refer to:
 163 
 164   > Cotter, Jiang and Sridharan. "Two-Player Games for Efficient Non-Convex
 165   > Constrained Optimization".
 166   > [https://arxiv.org/abs/1804.06500](https://arxiv.org/abs/1804.06500)
 167 
 168   This function implements the approach described in Lemma 3.
 169 
 170   Args:
 171     objective_vector: numpy array of shape (n,), where n is the number of
 172       "candidate solutions". Contains the objective function values.
 173     constraints_matrix: numpy array of shape (m,n), where m is the number of
 174       constraints and n is the number of "candidate solutions". Contains the
 175       constraint violation magnitudes.
 176     epsilon: nonnegative float, the threshold at which to terminate the binary
 177       search while searching for the minimal expected constraint violation
 178       magnitude.
 179 
 180   Returns:
 181     The optimal distribution, as a numpy array of shape (n,).
 182 
 183   Raises:
 184     ValueError: If `objective_vector` and `constraints_matrix` have inconsistent
 185       shapes, or if `epsilon` is negative.
 186     ImportError: If we're unable to import `scipy.optimize`.
 187   """
 188   if epsilon < 0.0:
 189     raise ValueError("epsilon must be nonnegative")
 190 
 191   # If there is a feasible solution (i.e. with maximum_violation=0), then that's
 192   # what we'll return.
 193   pp, _ = _find_best_candidate_distribution_helper(objective_vector,
 194                                                    constraints_matrix)
 195   if pp is not None:
 196     return pp
 197 
 198   # The bound is the minimum over all candidates, of the maximum per-candidate
 199   # constraint violation.
 200   lower = 0.0
 201   upper = np.min(np.amax(constraints_matrix, axis=0))
 202   best_pp, _ = _find_best_candidate_distribution_helper(
 203       objective_vector, constraints_matrix, maximum_violation=upper)
 204   assert best_pp is not None
 205 
 206   # Throughout this loop, a maximum_violation of "lower" is not achievable,
 207   # but a maximum_violation of "upper" is achievable.
 208   while True:
 209     middle = 0.5 * (lower + upper)
 210     if (middle - lower <= epsilon) or (upper - middle <= epsilon):
 211       break
 212     else:
 213       pp, _ = _find_best_candidate_distribution_helper(
 214           objective_vector, constraints_matrix, maximum_violation=middle)
 215       if pp is None:
 216         lower = middle
 217       else:
 218         best_pp = pp
 219         upper = middle
 220 
 221   return best_pp

 224 def find_best_candidate_index(objective_vector,
 225                               constraints_matrix,
 226                               rank_objectives=False):
 227   """Heuristically finds the best candidate solution to a constrained problem.
 228 
 229   This function deals with the constrained problem:
 230 
 231   > minimize f(w)
 232   > s.t. g_i(w) <= 0 for all i in {0,1,...,m-1}
 233 
 234   Here, f(w) is the "objective function", and g_i(w) is the ith (of m)
 235   "constraint function". Given a set of n "candidate solutions"
 236   {w_0,w_1,...,w_{n-1}}, this function finds the "best" solution according
 237   to the following heuristic:
 238 
 239     1. Across all models, the ith constraint violations (i.e. max{0, g_i(0)})
 240        are ranked, as are the objectives (if rank_objectives=True).
 241     2. Each model is then associated its MAXIMUM rank across all m constraints
 242        (and the objective, if rank_objectives=True).
 243     3. The model with the minimal maximum rank is then identified. Ties are
 244        broken using the objective function value.
 245     4. The index of this "best" model is returned.
 246 
 247   The `objective_vector` parameter should be a numpy array with shape (n,), for
 248   which objective_vector[i] = f(w_i). Likewise, `constraints_matrix` should be a
 249   numpy array with shape (m,n), for which constraints_matrix[i,j] = g_i(w_j).
 250 
 251   For more specifics, please refer to:
 252 
 253   > Cotter, Jiang and Sridharan. "Two-Player Games for Efficient Non-Convex
 254   > Constrained Optimization".
 255   > [https://arxiv.org/abs/1804.06500](https://arxiv.org/abs/1804.06500)
 256 
 257   This function implements the heuristic used for hyperparameter search in the
 258   experiments of Section 5.2.
 259 
 260   Args:
 261     objective_vector: numpy array of shape (n,), where n is the number of
 262       "candidate solutions". Contains the objective function values.
 263     constraints_matrix: numpy array of shape (m,n), where m is the number of
 264       constraints and n is the number of "candidate solutions". Contains the
 265       constraint violation magnitudes.
 266     rank_objectives: bool, whether the objective function values should be
 267       included in the initial ranking step. If True, both the objective and
 268       constraints will be ranked. If False, only the constraints will be ranked.
 269       In either case, the objective function values will be used for
 270       tiebreaking.
 271 
 272   Returns:
 273     The index (in {0,1,...,n-1}) of the "best" model according to the above
 274       heuristic.
 275 
 276   Raises:
 277     ValueError: If `objective_vector` and `constraints_matrix` have inconsistent
 278       shapes.
 279     ImportError: If we're unable to import `scipy.stats`.
 280   """
 281   mm, nn = np.shape(constraints_matrix)
 282   if (nn,) != np.shape(objective_vector):
 283     raise ValueError(
 284         "objective_vector must have shape (n,), and constraints_matrix (m, n),"
 285         " where n is the number of candidates, and m is the number of "
 286         "constraints")
 287 
 288   # We import scipy inline, instead of at the top of the file, so that a scipy
 289   # dependency is only introduced if either find_best_candidate_distribution()
 290   # or find_best_candidate_index() are actually called.
 291   import scipy.stats  # pylint: disable=g-import-not-at-top
 292 
 293   if rank_objectives:
 294     maximum_ranks = scipy.stats.rankdata(objective_vector, method="min")
 295   else:
 296     maximum_ranks = np.zeros(nn, dtype=np.int64)
 297   for ii in xrange(mm):
 298     # Take the maximum of the constraint functions with zero, since we want to
 299     # rank the magnitude of constraint *violations*. If the constraint is
 300     # satisfied, then we don't care how much it's satisfied by (as a result, we
 301     # we expect all models satisfying a constraint to be tied at rank 1).
 302     ranks = scipy.stats.rankdata(
 303         np.maximum(0.0, constraints_matrix[ii, :]), method="min")
 304     maximum_ranks = np.maximum(maximum_ranks, ranks)
 305 
 306   best_index = None
 307   best_rank = float("Inf")
 308   best_objective = float("Inf")
 309   for ii in xrange(nn):
 310     if maximum_ranks[ii] < best_rank:
 311       best_index = ii
 312       best_rank = maximum_ranks[ii]
 313       best_objective = objective_vector[ii]
 314     elif (maximum_ranks[ii] == best_rank) and (objective_vector[ii] <=
 315                                                best_objective):
 316       best_index = ii
 317       best_objective = objective_vector[ii]
 318 
 319   return best_index
