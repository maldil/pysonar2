  27 class ConstantMinimizationProblem(
  28     constrained_minimization_problem.ConstrainedMinimizationProblem):
  29   """A `ConstrainedMinimizationProblem` with constant constraint violations.
  30 
  31   This minimization problem is intended for use in performing simple tests of
  32   the Lagrange multiplier (or equivalent) update in the optimizers. There is a
  33   one-element "dummy" model parameter, but it should be ignored.
  34   """
  35 
  36   def __init__(self, constraints):
  37     """Constructs a new `ConstantMinimizationProblem'.
  38 
  39     Args:
  40       constraints: 1d numpy array, the constant constraint violations.
  41 
  42     Returns:
  43       A new `ConstantMinimizationProblem'.
  44     """
  45     # We make an fake 1-parameter linear objective so that we don't get a "no
  46     # variables to optimize" error.
  47     self._objective = standard_ops.Variable(0.0, dtype=dtypes.float32)
  48     self._constraints = standard_ops.constant(constraints, dtype=dtypes.float32)
  49 
  50   @property
  51   def objective(self):
  52     """Returns the objective function."""
  53     return self._objective
  54 
  55   @property
  56   def constraints(self):
  57     """Returns the constant constraint violations."""
  58     return self._constraints
