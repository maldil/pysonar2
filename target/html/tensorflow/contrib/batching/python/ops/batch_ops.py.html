  30 @ops.RegisterGradient("Batch")
  31 def _BatchGrad(op, *out_grads):  # pylint: disable=invalid-name
  32   """Gradient for batch op."""
  33   gradients = []
  34   for i in range(len(op.inputs)):
  35     gradients.append(
  36         gen_batch_ops.unbatch(
  37             out_grads[i],
  38             op.outputs[-2],
  39             op.outputs[-1],
  40             timeout_micros=op.get_attr("grad_timeout_micros"),
  41             shared_name="batch_gradient_{}_{}".format(op.name, i)))
  42   return gradients
  43

  45 @ops.RegisterGradient("Unbatch")
  46 def _UnbatchGrad(op, grad):   # pylint: disable=invalid-name
  47   return [
  48       gen_batch_ops.unbatch_grad(
  49           op.inputs[0],
  50           op.inputs[1],
  51           grad,
  52           op.inputs[2],
  53           shared_name="unbatch_gradient_{}".format(op.name)), None, None
  54   ]

  57 def batch_function(num_batch_threads,
  58                    max_batch_size,
  59                    batch_timeout_micros,
  60                    allowed_batch_sizes=None,
  61                    max_enqueued_batches=10):
  62   """Batches the computation done by the decorated function.
  63 
  64   So, for example, in the following code
  65 
  66   ```python
  67   @batch_function(1, 2, 3)
  68   def layer(a):
  69     return tf.matmul(a, a)
  70 
  71   b = layer(w)
  72   ```
  73 
  74   if more than one session.run call is simultaneously trying to compute `b`
  75   the values of `w` will be gathered, non-deterministically concatenated
  76   along the first axis, and only one thread will run the computation. See the
  77   documentation of the `Batch` op for more details.
  78 
  79   Assumes that all arguments of the decorated function are Tensors which will
  80   be batched along their first dimension.
  81 
  82   SparseTensor is not supported. The return value of the decorated function
  83   must be a Tensor or a list/tuple of Tensors.
  84 
  85   Args:
  86     num_batch_threads: Number of scheduling threads for processing batches
  87      of work. Determines the number of batches processed in parallel.
  88     max_batch_size: Batch sizes will never be bigger than this.
  89     batch_timeout_micros: Maximum number of microseconds to wait before
  90      outputting an incomplete batch.
  91     allowed_batch_sizes: Optional list of allowed batch sizes. If left empty,
  92      does nothing. Otherwise, supplies a list of batch sizes, causing the op
  93      to pad batches up to one of those sizes. The entries must increase
  94      monotonically, and the final entry must equal max_batch_size.
  95     max_enqueued_batches: The maximum depth of the batch queue. Defaults to 10.
  96 
  97   Returns:
  98     The decorated function will return the unbatched computation output Tensors.
  99   """
 100 
 101   def decorator(fn):  # pylint: disable=missing-docstring
 102 
 103     def decorated(*args):  # pylint: disable=missing-docstring
 104       types = [arg.dtype for arg in args]
 105 
 106       @function.Defun(*types)
 107       def computation(*computation_args):
 108         return fn(*computation_args)
 109 
 110       with ops.name_scope("batch") as name:
 111         for a in args:
 112           if not isinstance(a, ops.Tensor):
 113             raise ValueError("All arguments to functions decorated with "
 114                              "`batch_function`  are supposed to be Tensors; "
 115                              "found %s" % repr(a))
 116         return gen_batch_ops.batch_function(
 117             num_batch_threads=num_batch_threads,
 118             max_batch_size=max_batch_size,
 119             batch_timeout_micros=batch_timeout_micros,
 120             allowed_batch_sizes=allowed_batch_sizes,
 121             max_enqueued_batches=max_enqueued_batches,
 122             shared_name=name,
 123             f=computation,
 124             in_tensors=list(args),
 125             captured_tensors=computation.captured_inputs,
 126             Tout=[o.type for o in computation.definition.signature.output_arg])
 127 
 128     return decorated
 129 
 130   return decorator

 133 def batch_function_v1(num_batch_threads,
 134                       max_batch_size,
 135                       batch_timeout_micros,
 136                       allowed_batch_sizes=None,
 137                       grad_timeout_micros=60 * 1000 * 1000,
 138                       unbatch_timeout_micros=60 * 1000 * 1000,
 139                       max_enqueued_batches=10):
 140   """Batches the computation done by the decorated function.
 141 
 142   This is the older version of batch_function(). Please use the former instead
 143   of this.
 144 
 145   Args:
 146     num_batch_threads: Number of scheduling threads for processing batches
 147      of work. Determines the number of batches processed in parallel.
 148     max_batch_size: Batch sizes will never be bigger than this.
 149     batch_timeout_micros: Maximum number of microseconds to wait before
 150      outputting an incomplete batch.
 151     allowed_batch_sizes: Optional list of allowed batch sizes. If left empty,
 152      does nothing. Otherwise, supplies a list of batch sizes, causing the op
 153      to pad batches up to one of those sizes. The entries must increase
 154      monotonically, and the final entry must equal max_batch_size.
 155     grad_timeout_micros: The timeout to use for the gradient. See the
 156      documentation of the unbatch op for more details. Defaults to 60s.
 157     unbatch_timeout_micros: The timeout to use for unbatching. See the
 158      documentation of the unbatch op for more details. Defaults to 60s.
 159     max_enqueued_batches: The maximum depth of the batch queue. Defaults to 10.
 160 
 161   Returns:
 162     The decorated function will return the unbatched computation output Tensors.
 163   """
 164   def decorator(f):  # pylint: disable=missing-docstring
 165     def decorated(*args):
 166       with ops.name_scope("batch") as name:
 167         for a in args:
 168           if not isinstance(a, ops.Tensor):
 169             raise ValueError("All arguments to functions decorated with "
 170                              "`batch_function`  are supposed to be Tensors; "
 171                              "found %s" % repr(a))
 172         batched_tensors, batch_index, id_t = gen_batch_ops.batch(
 173             args,
 174             num_batch_threads=num_batch_threads,
 175             max_batch_size=max_batch_size,
 176             batch_timeout_micros=batch_timeout_micros,
 177             max_enqueued_batches=max_enqueued_batches,
 178             allowed_batch_sizes=allowed_batch_sizes,
 179             grad_timeout_micros=grad_timeout_micros,
 180             shared_name=name)
 181         outputs = f(*batched_tensors)
 182         if isinstance(outputs, ops.Tensor):
 183           outputs_list = [outputs]
 184         else:
 185           outputs_list = outputs
 186         with ops.name_scope("unbatch") as unbatch_name:
 187           unbatched = [
 188               gen_batch_ops.unbatch(t, batch_index, id_t,
 189                                     timeout_micros=unbatch_timeout_micros,
 190                                     shared_name=unbatch_name + "/" + t.name)
 191               for t in outputs_list]
 192         if isinstance(outputs, ops.Tensor):
 193           return unbatched[0]
 194         return unbatched
 195     return decorated
 196   return decorator
