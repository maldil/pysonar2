  35 def delayed_plus1(x):
  36   """Sleeps for 100ms then returns x+1."""
  37   time.sleep(0.1)
  38   return x + 1

  44   def testBasicBatch(self):
  45     """Tests that a single batched tensor executes together and only once."""
  46     with self.cached_session() as sess:
  47       inp = array_ops.placeholder(dtype=dtypes.int32, shape=[1])
  48       batched, index, _ = batch_ops.batch(
  49           [inp], num_batch_threads=1, max_batch_size=2,
  50           batch_timeout_micros=36000000, grad_timeout_micros=0,
  51           batching_queue="")
  52       thread_results = []
  53 
  54       def worker():
  55         thread_results.extend(
  56             sess.run([batched, index], feed_dict={inp: [1]}))
  57 
  58       worker_thread = threading.Thread(target=worker)
  59       worker_thread.start()
  60       main_results = sess.run([batched, index], feed_dict={inp: [2]})
  61       worker_thread.join()
  62 
  63       # At this point either the thread or the main did the batch and the other
  64       # should have empty results.
  65       if list(thread_results[0][0]):
  66         batch_t = thread_results[0][0]
  67         index_t = thread_results[1]
  68         empty_b = main_results[0][0]
  69         empty_m = main_results[1]
  70       else:
  71         batch_t = main_results[0][0]
  72         index_t = main_results[1]
  73         empty_b = thread_results[0][0]
  74         empty_m = thread_results[1]
  75 
  76       # Check that both the inputs made it out exactly once.
  77       self.assertAllEqual(sorted(batch_t), (1, 2))
  78       # Check that we get 2 rows in the index tensor.
  79       self.assertEqual(len(index_t), 2)
  80       # Check that the other ones are empty.
  81       self.assertEqual(len(empty_b), 0)
  82       self.assertEqual(len(empty_m), 0)

  84   def testBatchWithPadding(self):
  85     """Test that batching with padding up to an allowed batch size works."""
  86     with self.cached_session() as sess:
  87       inp = array_ops.placeholder(dtype=dtypes.int32, shape=[2])
  88       batched, index, _ = batch_ops.batch(
  89           [inp], num_batch_threads=1, max_batch_size=10,
  90           batch_timeout_micros=100000,  # 100ms
  91           allowed_batch_sizes=[5, 10],
  92           grad_timeout_micros=0, batching_queue="")
  93       thread_results = []
  94 
  95       def worker():
  96         thread_results.extend(
  97             sess.run([batched, index], feed_dict={inp: [1, 3]}))
  98 
  99       worker_thread = threading.Thread(target=worker)
 100       worker_thread.start()
 101       main_results = sess.run([batched, index], feed_dict={inp: [2, 4]})
 102       worker_thread.join()
 103 
 104       # At this point either the thread or the main did the batch and the other
 105       # should have empty results.
 106       if list(thread_results[0][0]):
 107         batch_t = thread_results[0][0]
 108       else:
 109         batch_t = main_results[0][0]
 110 
 111       # Check that the batch tensor incorporates the padding.
 112       self.assertEqual(len(batch_t), 5)

 114   def testMultipleBatch(self):
 115     """Tests that multiple batched tensors execute together."""
 116     with self.cached_session() as sess:
 117       inp0 = array_ops.placeholder(dtype=dtypes.int32, shape=[1])
 118       inp1 = array_ops.placeholder(dtype=dtypes.int32, shape=[1])
 119       batched, _, _ = batch_ops.batch(
 120           [inp0, inp1],
 121           num_batch_threads=1,
 122           max_batch_size=2,
 123           batch_timeout_micros=36000000,
 124           grad_timeout_micros=0,
 125           batching_queue="")
 126       thread_results = []
 127 
 128       def worker():
 129         thread_results.extend(
 130             sess.run([batched], feed_dict={inp0: [1],
 131                                            inp1: [2]}))
 132 
 133       worker_thread = threading.Thread(target=worker)
 134       worker_thread.start()
 135       main_results = sess.run([batched], feed_dict={inp0: [2], inp1: [3]})
 136       worker_thread.join()
 137 
 138       # At this point either the thread or the main did the batch and the other
 139       # should have empty results.
 140       if list(thread_results[0][0]):
 141         batch_t = thread_results[0]
 142         empty_t = main_results[0]
 143       else:
 144         batch_t = main_results[0]
 145         empty_t = thread_results[0]
 146 
 147       # Assert that the tensors were batched together.
 148       self.assertAllEqual(sorted(batch_t[0]), [1, 2])
 149       self.assertAllEqual(sorted(batch_t[1]), [2, 3])
 150       self.assertAllEqual(empty_t[0], [])
 151       self.assertAllEqual(empty_t[1], [])

 153   def testIllegalBatchDifferentDim0Sizes(self):
 154     """Tests illegally feeding tensors with different dim0 sizes."""
 155     with self.cached_session() as sess:
 156       inp0 = array_ops.placeholder(dtype=dtypes.int32, shape=[1])
 157       inp1 = array_ops.placeholder(dtype=dtypes.int32, shape=[2])
 158       batched, index, _ = batch_ops.batch(
 159           [inp0, inp1], num_batch_threads=1, max_batch_size=2,
 160           batch_timeout_micros=0, grad_timeout_micros=0, batching_queue="")
 161       with self.assertRaises(Exception) as raised:
 162         _ = sess.run([batched, index], feed_dict={inp0: [0], inp1: [1, 2]})
 163       self.assertGreater(
 164           raised.exception.message.find("must have equal 0th-dimension size"),
 165           0)

 167   def testBasicUnbatch(self):
 168     """Tests that batch and unbatch work together."""
 169     with self.cached_session() as sess:
 170       inp = array_ops.placeholder(dtype=dtypes.int32, shape=[1])
 171       batched, index, id_t = batch_ops.batch(
 172           [inp], num_batch_threads=1, max_batch_size=10,
 173           batch_timeout_micros=100000,  # 100ms
 174           allowed_batch_sizes=[3, 10],
 175           grad_timeout_micros=0, batching_queue="")
 176       computation = batched[0] + 1
 177       result = batch_ops.unbatch(computation, index, id_t,
 178                                  timeout_micros=1000000, shared_name="unbatch")
 179       thread_results = []
 180 
 181       def worker():
 182         thread_results.extend(sess.run([result], feed_dict={inp: [1]}))
 183 
 184       worker_thread = threading.Thread(target=worker)
 185       worker_thread.start()
 186       main_results = sess.run([result], feed_dict={inp: [2]})
 187       worker_thread.join()
 188       self.assertEqual(thread_results[0], [2])
 189       self.assertEqual(main_results[0], [3])

 370   def testUnbatchTimeout(self):
 371     """Tests that the unbatch timeout works."""
 372     with self.cached_session() as sess:
 373       inp = array_ops.placeholder(dtype=dtypes.int32, shape=[1])
 374       batched, index, id_t = batch_ops.batch(
 375           [inp], num_batch_threads=1, max_batch_size=2,
 376           batch_timeout_micros=36000000, grad_timeout_micros=0,
 377           batching_queue="")
 378       computation = batched[0] + 1
 379       timeout_micros = 10
 380       result = batch_ops.unbatch(computation, index, id_t, timeout_micros,
 381                                  shared_name="shared_unbatch")
 382       # Set up a parallel pipeline that delays the computation, but uses the
 383       # same unbatch resource object as the non-delayed pipeline.
 384       computation_delayed = script_ops.py_func(delayed_plus1,
 385                                                [batched[0]],
 386                                                dtypes.int32)
 387       result_delayed = batch_ops.unbatch(computation_delayed,
 388                                          index,
 389                                          id_t,
 390                                          timeout_micros,
 391                                          shared_name="shared_unbatch")
 392 
 393       thread_results = []
 394       def worker():
 395         # A first call using the non-delayed pipeline. The batcher will send an
 396         # empty tensor along the non-delayed pipeline.
 397         thread_results.extend(sess.run([result], feed_dict={inp: [1]}))
 398       worker_thread = threading.Thread(target=worker)
 399       worker_thread.start()
 400       time.sleep(0.1)  # Ensure the thread's call starts first.
 401       # A second call using the delayed pipeline.  The batcher will send the
 402       # batched tensor along the delayed pipeline, thus delaying the arrival of
 403       # the batched tensor at the unbatch op, relative to the empty tensor.
 404       #
 405       # TODO(olston, apassos): Avoid relying on the order in which the batch op
 406       # emits the empty tensor versus the batched one.
 407       _ = sess.run([result_delayed], feed_dict={inp: [2]})
 408       worker_thread.join()
 409       # The thread's call should hit the timeout, and thus get 0 results.
 410       self.assertEqual(len(thread_results), 0)

 412   def testUnbatchGrad(self):
 413     """Tests that batch and unbatch are differentiable."""
 414     with self.cached_session() as sess:
 415       inp = array_ops.placeholder(dtype=dtypes.float32, shape=[1])
 416       batched, index, id_t = batch_ops.batch(
 417           [inp], num_batch_threads=1, max_batch_size=2,
 418           batch_timeout_micros=36000000, grad_timeout_micros=1000000,
 419           batching_queue="")
 420       computation = batched[0] * batched[0]
 421       result = batch_ops.unbatch(computation, index, id_t,
 422                                  timeout_micros=1000000, shared_name="unbatch")
 423       grad = gradients_impl.gradients(result, inp)
 424       thread_results = []
 425 
 426       def worker():
 427         thread_results.extend(sess.run([grad], feed_dict={inp: [1]}))
 428 
 429       worker_thread = threading.Thread(target=worker)
 430       worker_thread.start()
 431       main_results = sess.run([grad], feed_dict={inp: [2]})
 432       worker_thread.join()
 433       self.assertEqual(thread_results[0], [2])
 434       self.assertEqual(main_results[0], [4])
