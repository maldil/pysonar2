  76 def index_table_from_tensor(mapping,
  77                             num_oov_buckets=0,
  78                             default_value=-1,
  79                             hasher_spec=FastHashSpec,
  80                             dtype=dtypes.string,
  81                             name=None):
  82   """Returns a lookup table that converts a string tensor into int64 IDs.
  83 
  84   This operation constructs a lookup table to convert tensor of strings into
  85   int64 IDs. The mapping can be initialized from a string `mapping` 1-D tensor
  86   where each element is a key and corresponding index within the tensor is the
  87   value.
  88 
  89   Any lookup of an out-of-vocabulary token will return a bucket ID based on its
  90   hash if `num_oov_buckets` is greater than zero. Otherwise it is assigned the
  91   `default_value`.
  92   The bucket ID range is `[mapping size, mapping size + num_oov_buckets - 1]`.
  93 
  94   The underlying table must be initialized by calling
  95   `tf.tables_initializer.run()` or `table.init.run()` once.
  96 
  97   Elements in `mapping` cannot have duplicates, otherwise when executing the
  98   table initializer op, it will throw a `FailedPreconditionError`.
  99 
 100   Sample Usages:
 101 
 102   ```python
 103   mapping_strings = tf.constant(["emerson", "lake", "palmer"])
 104   table = tf.contrib.lookup.index_table_from_tensor(
 105       mapping=mapping_strings, num_oov_buckets=1, default_value=-1)
 106   features = tf.constant(["emerson", "lake", "and", "palmer"])
 107   ids = table.lookup(features)
 108   ...
 109   tf.tables_initializer().run()
 110 
 111   ids.eval()  ==> [0, 1, 3, 2]
 112   ```
 113 
 114   Args:
 115     mapping: A 1-D `Tensor` that specifies the mapping of keys to indices. The
 116       type of this object must be castable to `dtype`.
 117     num_oov_buckets: The number of out-of-vocabulary buckets.
 118     default_value: The value to use for out-of-vocabulary feature values.
 119       Defaults to -1.
 120     hasher_spec: A `HasherSpec` to specify the hash function to use for
 121       assignment of out-of-vocabulary buckets.
 122     dtype: The type of values passed to `lookup`. Only string and integers are
 123       supported.
 124     name: A name for this op (optional).
 125 
 126   Returns:
 127     The lookup table to map an input `Tensor` to index `int64` `Tensor`.
 128 
 129   Raises:
 130     ValueError: If `mapping` is invalid.
 131     ValueError: If `num_oov_buckets` is negative.
 132   """
 133   if mapping is None:
 134     raise ValueError("mapping must be specified.")
 135   return lookup_ops.index_table_from_tensor(
 136       vocabulary_list=mapping,
 137       num_oov_buckets=num_oov_buckets,
 138       default_value=default_value,
 139       hasher_spec=hasher_spec,
 140       dtype=dtype,
 141       name=name)

 148 def string_to_index(tensor, mapping, default_value=-1, name=None):
 149   """Maps `tensor` of strings into `int64` indices based on `mapping`.
 150 
 151   This operation converts `tensor` of strings into `int64` indices.
 152   The mapping is initialized from a string `mapping` tensor where each element
 153   is a key and corresponding index within the tensor is the value.
 154 
 155   Any entry in the input which does not have a corresponding entry in 'mapping'
 156   (an out-of-vocabulary entry) is assigned the `default_value`
 157 
 158   Elements in `mapping` cannot be duplicated, otherwise the initialization
 159   will throw a FailedPreconditionError.
 160 
 161   The underlying table must be initialized by calling
 162   `tf.tables_initializer.run()` once.
 163 
 164   For example:
 165 
 166   ```python
 167   mapping_strings = tf.constant(["emerson", "lake", "palmer"])
 168   feats = tf.constant(["emerson", "lake", "and", "palmer"])
 169   ids = tf.contrib.lookup.string_to_index(
 170       feats, mapping=mapping_strings, default_value=-1)
 171   ...
 172   tf.tables_initializer().run()
 173 
 174   ids.eval()  ==> [0, 1, -1, 2]
 175   ```
 176 
 177   Args:
 178     tensor: A 1-D input `Tensor` with the strings to map to indices.
 179     mapping: A 1-D string `Tensor` that specifies the mapping of strings to
 180       indices.
 181     default_value: The `int64` value to use for out-of-vocabulary strings.
 182       Defaults to -1.
 183     name: A name for this op (optional).
 184 
 185   Returns:
 186     The mapped indices. It has the same shape and tensor type (dense or sparse)
 187     as `tensor`.
 188   """
 189   table = index_table_from_tensor(
 190       mapping=mapping, default_value=default_value, name=name)
 191   return table.lookup(tensor)

 194 def index_to_string_table_from_tensor(mapping, default_value="UNK", name=None):
 195   """Returns a lookup table that maps a `Tensor` of indices into strings.
 196 
 197   This operation constructs a lookup table to map int64 indices into string
 198   values. The mapping is initialized from a string `mapping` 1-D `Tensor` where
 199   each element is a value and the corresponding index within the tensor is the
 200   key.
 201 
 202   Any input which does not have a corresponding index in 'mapping'
 203   (an out-of-vocabulary entry) is assigned the `default_value`
 204 
 205   The underlying table must be initialized by calling
 206   `tf.tables_initializer.run()` or `table.init.run()` once.
 207 
 208   Elements in `mapping` cannot have duplicates, otherwise when executing the
 209   table initializer op, it will throw a `FailedPreconditionError`.
 210 
 211   Sample Usages:
 212 
 213   ```python
 214   mapping_string = tf.constant(["emerson", "lake", "palmer"])
 215   indices = tf.constant([1, 5], tf.int64)
 216   table = tf.contrib.lookup.index_to_string_table_from_tensor(
 217       mapping_string, default_value="UNKNOWN")
 218   values = table.lookup(indices)
 219   ...
 220   tf.tables_initializer().run()
 221 
 222   values.eval() ==> ["lake", "UNKNOWN"]
 223   ```
 224 
 225   Args:
 226     mapping: A 1-D string `Tensor` that specifies the strings to map from
 227       indices.
 228     default_value: The value to use for out-of-vocabulary indices.
 229     name: A name for this op (optional).
 230 
 231   Returns:
 232     The lookup table to map a string values associated to a given index `int64`
 233     `Tensors`.
 234 
 235   Raises:
 236     ValueError: when `mapping` is not set.
 237   """
 238 
 239   if mapping is None:
 240     raise ValueError("mapping must be specified.")
 241 
 242   return lookup_ops.index_to_string_table_from_tensor(
 243       vocabulary_list=mapping, default_value=default_value, name=name)

 292 class MutableHashTable(LookupInterface, checkpointable.CheckpointableBase):
 293   """A generic mutable hash table implementation.
 294 
 295   Data can be inserted by calling the insert method. It does not support
 296   initialization via the init method.
 297 
 298   Example usage:
 299 
 300   ```python
 301   table = tf.contrib.lookup.MutableHashTable(key_dtype=tf.string,
 302                                              value_dtype=tf.int64,
 303                                              default_value=-1)
 304   sess.run(table.insert(keys, values))
 305   out = table.lookup(query_keys)
 306   print(out.eval())
 307   ```
 308   """
 309 
 310   def __init__(self,
 311                key_dtype,
 312                value_dtype,
 313                default_value,
 314                shared_name=None,
 315                name="MutableHashTable",
 316                checkpoint=True):
 317     """Creates an empty `MutableHashTable` object.
 318 
 319     Creates a table, the type of its keys and values are specified by key_dtype
 320     and value_dtype, respectively.
 321 
 322     Args:
 323       key_dtype: the type of the key tensors.
 324       value_dtype: the type of the value tensors.
 325       default_value: The value to use if a key is missing in the table.
 326       shared_name: If non-empty, this table will be shared under
 327         the given name across multiple sessions.
 328       name: A name for the operation (optional).
 329       checkpoint: if True, the contents of the table are saved to and restored
 330         from checkpoints. If `shared_name` is empty for a checkpointed table, it
 331         is shared using the table node name.
 332 
 333     Returns:
 334       A `MutableHashTable` object.
 335 
 336     Raises:
 337       ValueError: If checkpoint is True and no name was specified.
 338     """
 339     self._default_value = ops.convert_to_tensor(default_value,
 340                                                 dtype=value_dtype)
 341     self._value_shape = self._default_value.get_shape()
 342 
 343     executing_eagerly = context.executing_eagerly()
 344     if executing_eagerly and shared_name is None:
 345       # TODO(allenl): This will leak memory due to kernel caching by the
 346       # shared_name attribute value (but is better than the alternative of
 347       # sharing everything by default when executing eagerly; hopefully creating
 348       # tables in a loop is uncommon).
 349       shared_name = "table_%d" % (ops.uid(),)
 350     # The table must be shared if checkpointing is requested for multi-worker
 351     # training to work correctly. Use the node name if no shared_name has been
 352     # explicitly specified.
 353     use_node_name_sharing = checkpoint and shared_name is None
 354     if self._default_value.get_shape().ndims == 0:
 355       self._table_ref = gen_lookup_ops.mutable_hash_table_v2(
 356           shared_name=shared_name,
 357           use_node_name_sharing=use_node_name_sharing,
 358           key_dtype=key_dtype,
 359           value_dtype=value_dtype,
 360           name=name)
 361     else:
 362       self._table_ref = gen_lookup_ops.mutable_hash_table_of_tensors_v2(
 363           shared_name=shared_name,
 364           use_node_name_sharing=use_node_name_sharing,
 365           key_dtype=key_dtype,
 366           value_dtype=value_dtype,
 367           value_shape=self._default_value.get_shape(),
 368           name=name)
 369     if executing_eagerly:
 370       op_name = None
 371     else:
 372       op_name = self._table_ref.op.name.split("/")[-1]
 373     super(MutableHashTable, self).__init__(key_dtype, value_dtype,
 374                                            op_name)
 375 
 376     if checkpoint:
 377       saveable = MutableHashTable._Saveable(self, name)
 378       ops.add_to_collection(ops.GraphKeys.SAVEABLE_OBJECTS, saveable)
 379 
 380   def size(self, name=None):
 381     """Compute the number of elements in this table.
 382 
 383     Args:
 384       name: A name for the operation (optional).
 385 
 386     Returns:
 387       A scalar tensor containing the number of elements in this table.
 388     """
 389     with ops.name_scope(name, "%s_Size" % self._name,
 390                         [self._table_ref]) as name:
 391       with ops.colocate_with(self._table_ref):
 392         return gen_lookup_ops.lookup_table_size_v2(self._table_ref, name=name)

 394   def lookup(self, keys, name=None):
 395     """Looks up `keys` in a table, outputs the corresponding values.
 396 
 397     The `default_value` is used for keys not present in the table.
 398 
 399     Args:
 400       keys: Keys to look up. Can be a tensor of any shape. Must match the
 401         table's key_dtype.
 402       name: A name for the operation (optional).
 403 
 404     Returns:
 405       A tensor containing the values in the same shape as `keys` using the
 406         table's value type.
 407 
 408     Raises:
 409       TypeError: when `keys` do not match the table data types.
 410     """
 411     with ops.name_scope(name, "%s_lookup_table_find" % self._name,
 412                         (self._table_ref, keys, self._default_value)) as name:
 413       keys = ops.convert_to_tensor(keys, dtype=self._key_dtype, name="keys")
 414       with ops.colocate_with(self._table_ref):
 415         values = gen_lookup_ops.lookup_table_find_v2(
 416             self._table_ref, keys, self._default_value, name=name)
 417     return values
 418 
 419   def insert(self, keys, values, name=None):
 420     """Associates `keys` with `values`.
 421 
 422     Args:
 423       keys: Keys to insert. Can be a tensor of any shape. Must match the
 424         table's key type.
 425       values: Values to be associated with keys. Must be a tensor of the same
 426         shape as `keys` and match the table's value type.
 427       name: A name for the operation (optional).
 428 
 429     Returns:
 430       The created Operation.
 431 
 432     Raises:
 433       TypeError: when `keys` or `values` doesn't match the table data
 434         types.
 435     """
 436     with ops.name_scope(name, "%s_lookup_table_insert" % self._name,
 437                         [self._table_ref, keys, values]) as name:
 438       keys = ops.convert_to_tensor(keys, self._key_dtype, name="keys")
 439       values = ops.convert_to_tensor(values, self._value_dtype, name="values")
 440       with ops.colocate_with(self._table_ref):
 441         # pylint: disable=protected-access
 442         op = gen_lookup_ops.lookup_table_insert_v2(
 443             self._table_ref, keys, values, name=name)
 444     return op
 445 
 446   def export(self, name=None):
 447     """Returns tensors of all keys and values in the table.
 448 
 449     Args:
 450       name: A name for the operation (optional).
 451 
 452     Returns:
 453       A pair of tensors with the first tensor containing all keys and the
 454         second tensors containing all values in the table.
 455     """
 456     with ops.name_scope(name, "%s_lookup_table_export_values" % self._name,
 457                         [self._table_ref]) as name:
 458       with ops.colocate_with(self._table_ref):
 459         exported_keys, exported_values = gen_lookup_ops.lookup_table_export_v2(
 460             self._table_ref, self._key_dtype, self._value_dtype, name=name)
 461     return exported_keys, exported_values

 463   def _gather_saveables_for_checkpoint(self):
 464     """For object-based checkpointing."""
 465     return {"table": functools.partial(MutableHashTable._Saveable, table=self)}

 467   class _Saveable(BaseSaverBuilder.SaveableObject):
 468     """SaveableObject implementation for MutableHashTable."""
 469 
 470     def __init__(self, table, name):
 471       tensors = table.export()
 472       specs = [
 473           BaseSaverBuilder.SaveSpec(tensors[0], "", name + "-keys"),
 474           BaseSaverBuilder.SaveSpec(tensors[1], "", name + "-values")
 475       ]
 476       # pylint: disable=protected-access
 477       super(MutableHashTable._Saveable, self).__init__(table, specs, name)
 478 
 479     def restore(self, restored_tensors, restored_shapes):
 480       del restored_shapes  # unused
 481       # pylint: disable=protected-access
 482       with ops.colocate_with(self.op._table_ref):
 483         return gen_lookup_ops.lookup_table_import_v2(
 484             self.op._table_ref, restored_tensors[0], restored_tensors[1])

 487 class MutableDenseHashTable(LookupInterface, checkpointable.CheckpointableBase):
 488   """A generic mutable hash table implementation using tensors as backing store.
 489 
 490   Data can be inserted by calling the insert method. It does not support
 491   initialization via the init method.
 492 
 493   It uses "open addressing" with quadratic reprobing to resolve collisions.
 494   Compared to `MutableHashTable` the insert and lookup operations in a
 495   `MutableDenseHashTable` are typically faster, but memory usage can be higher.
 496   However, `MutableDenseHashTable` does not require additional memory for
 497   temporary tensors created during checkpointing and restore operations.
 498 
 499   Example usage:
 500 
 501   ```python
 502   table = tf.contrib.lookup.MutableDenseHashTable(key_dtype=tf.int64,
 503                                                   value_dtype=tf.int64,
 504                                                   default_value=-1,
 505                                                   empty_key=0)
 506   sess.run(table.insert(keys, values))
 507   out = table.lookup(query_keys)
 508   print(out.eval())
 509   ```
 510   """
 511

 568         use_node_name_sharing=use_node_name_sharing,
 569         value_dtype=value_dtype,

 574       op_name = None
 575     else:
 576       op_name = self._table_ref.op.name.split("/")[-1]

 578         key_dtype, value_dtype, op_name)

 581       saveable = MutableDenseHashTable._Saveable(self, name)
 582       ops.add_to_collection(ops.GraphKeys.SAVEABLE_OBJECTS, saveable)

 598   def lookup(self, keys, name=None):
 599     """Looks up `keys` in a table, outputs the corresponding values.
 600 
 601     The `default_value` is used for keys not present in the table.
 602 
 603     Args:
 604       keys: Keys to look up. Can be a tensor of any shape. Must match the
 605         table's key_dtype.
 606       name: A name for the operation (optional).
 607 
 608     Returns:
 609       A tensor containing the values in the same shape as `keys` using the
 610         table's value type.
 611 
 612     Raises:
 613       TypeError: when `keys` do not match the table data types.
 614     """
 615     with ops.name_scope(name, "%s_lookup_table_find" % self._name,
 616                         [self._table_ref, keys]) as name:
 617       keys = ops.convert_to_tensor(keys, dtype=self._key_dtype, name="keys")
 618       with ops.colocate_with(self._table_ref):
 619         values = gen_lookup_ops.lookup_table_find_v2(
 620             self._table_ref, keys, self._default_value, name=name)
 621 
 622     return values

 624   def insert(self, keys, values, name=None):
 625     """Associates `keys` with `values`.
 626 
 627     Args:
 628       keys: Keys to insert. Can be a tensor of any shape. Must match the
 629         table's key type.
 630       values: Values to be associated with keys. Must be a tensor of the same
 631         shape as `keys` and match the table's value type.
 632       name: A name for the operation (optional).
 633 
 634     Returns:
 635       The created Operation.
 636 
 637     Raises:
 638       TypeError: when `keys` or `values` doesn't match the table data
 639         types.
 640     """
 641     with ops.name_scope(name, "%s_lookup_table_insert" % self._name,
 642                         [self._table_ref, keys, values]) as name:
 643       keys = ops.convert_to_tensor(keys, dtype=self._key_dtype, name="keys")
 644       values = ops.convert_to_tensor(
 645           values, dtype=self._value_dtype, name="values")
 646       with ops.colocate_with(self._table_ref):
 647         op = gen_lookup_ops.lookup_table_insert_v2(
 648             self._table_ref, keys, values, name=name)
 649       return op

 651   def export(self, name=None):
 652     """Returns tensors of all keys and values in the table.
 653 
 654     Args:
 655       name: A name for the operation (optional).
 656 
 657     Returns:
 658       A pair of tensors with the first tensor containing all keys and the
 659         second tensors containing all values in the table.
 660     """
 661     with ops.name_scope(name, "%s_lookup_table_export_values" % self._name,
 662                         [self._table_ref]) as name:
 663       with ops.colocate_with(self._table_ref):
 664         exported_keys, exported_values = gen_lookup_ops.lookup_table_export_v2(
 665             self._table_ref, self._key_dtype, self._value_dtype, name=name)
 666 
 667     return exported_keys, exported_values
