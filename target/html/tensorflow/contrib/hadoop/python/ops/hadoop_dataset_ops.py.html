  29 class SequenceFileDataset(dataset_ops.DatasetSource):
  30   """A Sequence File Dataset that reads the sequence file."""
  31 
  32   def __init__(self, filenames):
  33     """Create a `SequenceFileDataset`.
  34 
  35     `SequenceFileDataset` allows a user to read data from a hadoop sequence
  36     file. A sequence file consists of (key value) pairs sequentially. At
  37     the moment, `org.apache.hadoop.io.Text` is the only serialization type
  38     being supported, and there is no compression support.
  39 
  40     For example:
  41 
  42     ```python
  43     dataset = tf.contrib.hadoop.SequenceFileDataset("/foo/bar.seq")
  44     iterator = dataset.make_one_shot_iterator()
  45     next_element = iterator.get_next()
  46     # Prints the (key, value) pairs inside a hadoop sequence file.
  47     while True:
  48       try:
  49         print(sess.run(next_element))
  50       except tf.errors.OutOfRangeError:
  51         break
  52     ```
  53 
  54     Args:
  55       filenames: A `tf.string` tensor containing one or more filenames.
  56     """
  57     super(SequenceFileDataset, self).__init__()
  58     self._filenames = ops.convert_to_tensor(
  59         filenames, dtype=dtypes.string, name="filenames")
  60 
  61   def _as_variant_tensor(self):
  62     return gen_dataset_ops.sequence_file_dataset(
  63         self._filenames, nest.flatten(self.output_types))
  64 
  65   @property
  66   def output_classes(self):
  67     return ops.Tensor, ops.Tensor
  68 
  69   @property
  70   def output_shapes(self):
  71     return (tensor_shape.TensorShape([]), tensor_shape.TensorShape([]))
  72 
  73   @property
  74   def output_types(self):
  75     return dtypes.string, dtypes.string
