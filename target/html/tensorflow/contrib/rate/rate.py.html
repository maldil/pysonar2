  33 _to_replace = re.compile("[^A-Za-z0-9.]")
  34 
  35 
  36 class Rate(object):
  37   """Computes the rate of change since the last rate call."""
  38 
  39   def __init__(self, name=None):
  40     self._built = False
  41     self._vars = []
  42     self._initial_values = {}
  43     name = name or self.__class__.__name__
  44     # Replace things like spaces in name to create a valid scope name.
  45     scope_name = _to_replace.sub("_", name)
  46     # We create the variable scope now to get the unique name that will
  47     # be used as a variable prefix when build() calls _add_variable().
  48     with variable_scope.variable_scope(
  49         scope_name, use_resource=True, reuse=False) as scope:
  50       pos = scope.name.rfind(scope_name)
  51       self._name = name + scope.name[pos + len(scope_name):]
  52       self._scope = scope
  53 
  54     # Ensures that if the user calls build directly we still set self._built to
  55     # True to prevent variables from being recreated.
  56     self._build = self.build
  57     if context.executing_eagerly():
  58       self._construction_scope = context.eager_mode
  59     else:
  60       # We make self.call() into a graph callable here, so that we can
  61       # return a single op that performs all of the variable updates.
  62       self._construction_scope = ops.get_default_graph().as_default
  63       self.call = function.defun(self.call)
  64 
  65   def build(self, values, denominator):
  66     """Method to create variables.
  67 
  68     Called by `__call__()` before `call()` for the first time.
  69 
  70     Args:
  71       values: The numerator for rate.
  72       denominator: Value to which the rate is taken with respect.
  73     """
  74     self.numer = self._add_variable(
  75         name="numer", shape=values.get_shape(), dtype=dtypes.float64)
  76     self.denom = self._add_variable(
  77         name="denom", shape=denominator.get_shape(), dtype=dtypes.float64)
  78     self.prev_values = self._add_variable(
  79         name="prev_values", shape=values.get_shape(), dtype=dtypes.float64)
  80     self.prev_denominator = self._add_variable(
  81         name="prev_denominator",
  82         shape=denominator.get_shape(),
  83         dtype=dtypes.float64)
  84     self._built = True
  85 
  86   def __call__(self, *args, **kwargs):
  87     """Returns op to execute to update.
  88 
  89     Returns None if eager execution is enabled.
  90     Returns a graph-mode function if graph execution is enabled.
  91 
  92     Args:
  93       *args:
  94       **kwargs: A mini-batch of inputs to Rate, passed on to `call()`.
  95     """
  96     if not self._built:
  97       with variable_scope.variable_scope(
  98           self._scope), self._construction_scope():
  99         self.build(*args, **kwargs)
 100       self._built = True
 101     return self.call(*args, **kwargs)
 102 
 103   @property
 104   def name(self):
 105     return self._name
 106 
 107   @property
 108   def variables(self):
 109     return self._vars
 110 
 111   def _safe_div(self, numerator, denominator, name):
 112     t = math_ops.truediv(numerator, denominator)
 113     zero = array_ops.zeros_like(t, dtype=denominator.dtype)
 114     condition = math_ops.greater(denominator, zero)
 115     zero = math_ops.cast(zero, t.dtype)
 116     return array_ops.where(condition, t, zero, name=name)
 117 
 118   def _add_variable(self, name, shape=None, dtype=None):
 119     """Private method for adding variables to the graph."""
 120     if self._built:
 121       raise RuntimeError("Can't call add_variable() except in build().")
 122     v = resource_variable_ops.ResourceVariable(
 123         lambda: array_ops.zeros(shape, dtype),
 124         trainable=False,
 125         validate_shape=True,
 126         name=name,
 127         collections=[ops.GraphKeys.LOCAL_VARIABLES])
 128     return v
 129 
 130   def call(self, values, denominator):
 131     """Computes the rate since the last call.
 132 
 133     Args:
 134       values: Tensor with the per-example value.
 135       denominator: Measure to take the rate with respect to.
 136 
 137     Returns:
 138       The rate or 0 if denominator is unchanged since last call.
 139     """
 140     if denominator.dtype != dtypes.float64:
 141       denominator = math_ops.cast(denominator, dtypes.float64)
 142     if values.dtype != dtypes.float64:
 143       values = math_ops.cast(values, dtypes.float64)
 144 
 145     state_ops.assign(self.numer, math_ops.subtract(values, self.prev_values))
 146     state_ops.assign(self.denom,
 147                      math_ops.subtract(denominator, self.prev_denominator))
 148     state_ops.assign(self.prev_values, values)
 149     state_ops.assign(self.prev_denominator, denominator)
 150 
 151     return self._safe_div(self.numer, self.denom, name="safe_rate")
