  33 def _bilinearly_interpolate(data, x, y):
  34   """Performs bilinenar interpolation of grid data at user defined coordinates.
  35 
  36   This interpolation function:
  37     a) implicitly pads the input data with 0s.
  38     b) returns 0 when sampling outside the (padded) image.
  39   The effect is that the sampled signal smoothly goes to 0 outside the original
  40   input domain, rather than producing a jump discontinuity at the image
  41   boundaries.
  42 
  43   Args:
  44     data: numpy array of shape `[data_height, data_width]` containing data
  45       samples assumed to be defined at the corresponding pixel coordinates.
  46     x: numpy array of shape `[warp_height, warp_width]` containing x coordinates
  47       at which interpolation will be performed.
  48     y: numpy array of shape `[warp_height, warp_width]` containing y coordinates
  49       at which interpolation will be performed.
  50 
  51   Returns:
  52     Numpy array of shape `[warp_height, warp_width]` containing interpolated
  53       values.
  54   """
  55   shape = x.shape
  56   x = np.asarray(x) + 1
  57   y = np.asarray(y) + 1
  58   data = np.lib.pad(data, 1, "constant", constant_values=0)
  59 
  60   x_0 = np.floor(x).astype(int)
  61   x_1 = x_0 + 1
  62   y_0 = np.floor(y).astype(int)
  63   y_1 = y_0 + 1
  64 
  65   x_0 = np.clip(x_0, 0, data.shape[1] - 1)
  66   x_1 = np.clip(x_1, 0, data.shape[1] - 1)
  67   y_0 = np.clip(y_0, 0, data.shape[0] - 1)
  68   y_1 = np.clip(y_1, 0, data.shape[0] - 1)
  69 
  70   i_a = data[y_0, x_0]
  71   i_b = data[y_1, x_0]
  72   i_c = data[y_0, x_1]
  73   i_d = data[y_1, x_1]
  74 
  75   w_a = (x_1 - x) * (y_1 - y)
  76   w_b = (x_1 - x) * (y - y_0)
  77   w_c = (x - x_0) * (y_1 - y)
  78   w_d = (x - x_0) * (y - y_0)
  79 
  80   samples = (w_a * i_a + w_b * i_b + w_c * i_c + w_d * i_d)
  81   samples.reshape(shape)
  82 
  83   return samples
  84

  86 def _make_warp(batch_size, warp_height, warp_width, dtype):
  87   """Creates batch of warping coordinates."""
  88   x, y = np.meshgrid(np.linspace(0, warp_width - 1, warp_width),
  89                      np.linspace(0, warp_height - 1, warp_height))
  90   warp = np.concatenate((x.reshape([warp_height, warp_width, 1]),
  91                          y.reshape([warp_height, warp_width, 1])), 2)
  92   warp = np.tile(warp.reshape([1, warp_height, warp_width, 2]),
  93                  [batch_size, 1, 1, 1])
  94   warp += np.random.randn(*warp.shape)
  95   return warp.astype(dtype)

 127   def _test_op_forward_pass(self, on_gpu, dtype, tol):
 128     np.random.seed(0)
 129     data_width = 7
 130     data_height = 9
 131     data_channels = 5
 132     warp_width = 4
 133     warp_height = 8
 134     batch_size = 10
 135 
 136     warp = _make_warp(batch_size, warp_height, warp_width, dtype.as_numpy_dtype)
 137     data_shape = (batch_size, data_height, data_width, data_channels)
 138     data = np.random.rand(*data_shape).astype(dtype.as_numpy_dtype)
 139 
 140     with self.test_session(use_gpu=on_gpu, force_gpu=False) as sess:
 141       data_ph = array_ops.placeholder(dtype, shape=(None,) + data.shape[1:])
 142       warp_ph = array_ops.placeholder(dtype, shape=(None,) + warp.shape[1:])
 143       outputs = resampler.resampler(data=data_ph, warp=warp_ph)
 144       self.assertEqual(outputs.get_shape().as_list(),
 145                        [None, warp_height, warp_width, data_channels])
 146       out = sess.run(outputs, feed_dict={data_ph: data, warp_ph: warp})
 147 
 148     # Generate reference output via bilinear interpolation in numpy
 149     reference_output = np.zeros_like(out)
 150     for batch in xrange(batch_size):
 151       for c in xrange(data_channels):
 152         reference_output[batch, :, :, c] = _bilinearly_interpolate(
 153             data[batch, :, :, c],
 154             warp[batch, :, :, 0],
 155             warp[batch, :, :, 1])
 156 
 157     self.assertAllClose(out, reference_output, rtol=tol, atol=tol)

 159   def _test_op_backward_pass(self, on_gpu, dtype, tol):
 160     np.random.seed(13)
 161     data_width = 5
 162     data_height = 4
 163     data_channels = 3
 164     warp_width = 2
 165     warp_height = 6
 166     batch_size = 3
 167 
 168     warp = _make_warp(batch_size, warp_height, warp_width, dtype.as_numpy_dtype)
 169     data_shape = (batch_size, data_height, data_width, data_channels)
 170     data = np.random.rand(*data_shape).astype(dtype.as_numpy_dtype)
 171 
 172     with self.test_session(use_gpu=on_gpu, force_gpu=False):
 173       data_tensor = constant_op.constant(data)
 174       warp_tensor = constant_op.constant(warp)
 175       output_tensor = resampler.resampler(data=data_tensor, warp=warp_tensor)
 176 
 177       grads = test.compute_gradient([data_tensor, warp_tensor], [
 178           data_tensor.get_shape().as_list(),
 179           warp_tensor.get_shape().as_list()
 180       ], output_tensor, output_tensor.get_shape().as_list(), [data, warp])
 181 
 182       if not on_gpu:
 183         # On CPU we perform numerical differentiation at the best available
 184         # precision, and compare against that. This is necessary for test to
 185         # pass for float16.
 186         data_tensor_64 = constant_op.constant(data, dtype=dtypes.float64)
 187         warp_tensor_64 = constant_op.constant(warp, dtype=dtypes.float64)
 188         output_tensor_64 = resampler.resampler(data=data_tensor_64,
 189                                                warp=warp_tensor_64)
 190         grads_64 = test.compute_gradient([data_tensor_64, warp_tensor_64], [
 191             data_tensor.get_shape().as_list(),
 192             warp_tensor.get_shape().as_list()
 193         ], output_tensor_64, output_tensor.get_shape().as_list(), [data, warp])
 194 
 195         for g, g_64 in zip(grads, grads_64):
 196           self.assertLess(np.fabs(g[0] - g_64[1]).max(), tol)
 197 
 198       else:
 199         for g in grads:
 200           self.assertLess(np.fabs(g[0] - g[1]).max(), tol)

 202   def test_op_errors(self):
 203     data_width = 7
 204     data_height = 9
 205     data_depth = 3
 206     data_channels = 5
 207     warp_width = 4
 208     warp_height = 8
 209     batch_size = 10
 210 
 211     # Input data shape is not defined over a 2D grid, i.e. its shape is not like
 212     # (batch_size, data_height, data_width, data_channels).
 213     with self.cached_session() as sess:
 214       data_shape = (batch_size, data_height, data_width, data_depth,
 215                     data_channels)
 216       data = np.zeros(data_shape)
 217       warp_shape = (batch_size, warp_height, warp_width, 2)
 218       warp = np.zeros(warp_shape)
 219       outputs = resampler.resampler(constant_op.constant(data),
 220                                     constant_op.constant(warp))
 221 
 222       with self.assertRaisesRegexp(errors_impl.UnimplementedError,
 223                                    "Only bilinear interpolation is currently "
 224                                    "supported."):
 225         sess.run(outputs)
 226 
 227     # Warp tensor must be at least a matrix, with shape [batch_size, 2].
 228     with self.cached_session() as sess:
 229       data_shape = (batch_size, data_height, data_width, data_channels)
 230       data = np.zeros(data_shape)
 231       warp_shape = (batch_size,)
 232       warp = np.zeros(warp_shape)
 233       outputs = resampler.resampler(constant_op.constant(data),
 234                                     constant_op.constant(warp))
 235 
 236       with self.assertRaisesRegexp(errors_impl.InvalidArgumentError,
 237                                    "warp should be at least a matrix"):
 238         sess.run(outputs)
 239 
 240     # The batch size of the data and warp tensors must be the same.
 241     with self.cached_session() as sess:
 242       data_shape = (batch_size, data_height, data_width, data_channels)
 243       data = np.zeros(data_shape)
 244       warp_shape = (batch_size+1, warp_height, warp_width, 2)
 245       warp = np.zeros(warp_shape)
 246       outputs = resampler.resampler(constant_op.constant(data),
 247                                     constant_op.constant(warp))
 248 
 249       with self.assertRaisesRegexp(errors_impl.InvalidArgumentError,
 250                                    "Batch size of data and warp tensor"):
 251         sess.run(outputs)
 252 
 253     # The warp tensor must contain 2D coordinates, i.e. its shape last dimension
 254     # must be 2.
 255     with self.cached_session() as sess:
 256       data_shape = (batch_size, data_height, data_width, data_channels)
 257       data = np.zeros(data_shape)
 258       warp_shape = (batch_size, warp_height, warp_width, 3)
 259       warp = np.zeros(warp_shape)
 260       outputs = resampler.resampler(constant_op.constant(data),
 261                                     constant_op.constant(warp))
 262 
 263       with self.assertRaisesRegexp(errors_impl.UnimplementedError,
 264                                    "Only bilinear interpolation is supported, "
 265                                    "warping"):
 266         sess.run(outputs)
