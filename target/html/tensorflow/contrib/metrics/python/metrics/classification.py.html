  32 def accuracy(predictions, labels, weights=None, name=None):
  33   """Computes the percentage of times that predictions matches labels.
  34 
  35   Args:
  36     predictions: the predicted values, a `Tensor` whose dtype and shape
  37                  matches 'labels'.
  38     labels: the ground truth values, a `Tensor` of any shape and
  39             bool, integer, or string dtype.
  40     weights: None or `Tensor` of float values to reweight the accuracy.
  41     name: A name for the operation (optional).
  42 
  43   Returns:
  44     Accuracy `Tensor`.
  45 
  46   Raises:
  47     ValueError: if dtypes don't match or
  48                 if dtype is not bool, integer, or string.
  49   """
  50   if not (labels.dtype.is_integer or
  51           labels.dtype in (dtypes.bool, dtypes.string)):
  52     raise ValueError(
  53         'Labels should have bool, integer, or string dtype, not %r' %
  54         labels.dtype)
  55   if not labels.dtype.is_compatible_with(predictions.dtype):
  56     raise ValueError('Dtypes of predictions and labels should match. '
  57                      'Given: predictions (%r) and labels (%r)' %
  58                      (predictions.dtype, labels.dtype))
  59   with ops.name_scope(name, 'accuracy', values=[predictions, labels]):
  60     is_correct = math_ops.cast(
  61         math_ops.equal(predictions, labels), dtypes.float32)
  62     if weights is not None:
  63       is_correct = math_ops.multiply(is_correct, weights)
  64       num_values = math_ops.multiply(weights, array_ops.ones_like(is_correct))
  65       return math_ops.div(math_ops.reduce_sum(is_correct),
  66                           math_ops.reduce_sum(num_values))
  67     return math_ops.reduce_mean(is_correct)

  70 def f1_score(labels, predictions, weights=None, num_thresholds=200,
  71              metrics_collections=None, updates_collections=None, name=None):
  72   """Computes the approximately best F1-score across different thresholds.
  73 
  74   The f1_score function applies a range of thresholds to the predictions to
  75   convert them from [0, 1] to bool. Precision and recall are computed by
  76   comparing them to the labels. The F1-Score is then defined as
  77   2 * precision * recall / (precision + recall). The best one across the
  78   thresholds is returned.
  79 
  80   Disclaimer: In practice it may be desirable to choose the best threshold on
  81   the validation set and evaluate the F1 score with this threshold on a
  82   separate test set. Or it may be desirable to use a fixed threshold (e.g. 0.5).
  83 
  84   This function internally creates four local variables, `true_positives`,
  85   `true_negatives`, `false_positives` and `false_negatives` that are used to
  86   compute the pairs of recall and precision values for a linearly spaced set of
  87   thresholds from which the best f1-score is derived.
  88 
  89   This value is ultimately returned as `f1-score`, an idempotent operation that
  90   computes the F1-score (computed using the aforementioned variables). The
  91   `num_thresholds` variable controls the degree of discretization with larger
  92   numbers of thresholds more closely approximating the true best F1-score.
  93 
  94   For estimation of the metric over a stream of data, the function creates an
  95   `update_op` operation that updates these variables and returns the F1-score.
  96 
  97   Example usage with a custom estimator:
  98   def model_fn(features, labels, mode):
  99     predictions = make_predictions(features)
 100     loss = make_loss(predictions, labels)
 101     train_op = tf.contrib.training.create_train_op(
 102           total_loss=loss,
 103           optimizer='Adam')
 104     eval_metric_ops = {'f1': f1_score(labels, predictions)}
 105     return tf.estimator.EstimatorSpec(
 106         mode=mode,
 107         predictions=predictions,
 108         loss=loss,
 109         train_op=train_op,
 110         eval_metric_ops=eval_metric_ops,
 111         export_outputs=export_outputs)
 112   estimator = tf.estimator.Estimator(model_fn=model_fn)
 113 
 114   If `weights` is `None`, weights default to 1. Use weights of 0 to mask values.
 115 
 116   Args:
 117     labels: A `Tensor` whose shape matches `predictions`. Will be cast to
 118       `bool`.
 119     predictions: A floating point `Tensor` of arbitrary shape and whose values
 120       are in the range `[0, 1]`.
 121     weights: Optional `Tensor` whose rank is either 0, or the same rank as
 122       `labels`, and must be broadcastable to `labels` (i.e., all dimensions must
 123       be either `1`, or the same as the corresponding `labels` dimension).
 124     num_thresholds: The number of thresholds to use when discretizing the roc
 125       curve.
 126     metrics_collections: An optional list of collections that `f1_score` should
 127       be added to.
 128     updates_collections: An optional list of collections that `update_op` should
 129       be added to.
 130     name: An optional variable_scope name.
 131 
 132   Returns:
 133     f1_score: A scalar `Tensor` representing the current best f1-score across
 134       different thresholds.
 135     update_op: An operation that increments the `true_positives`,
 136       `true_negatives`, `false_positives` and `false_negatives` variables
 137       appropriately and whose value matches the `f1_score`.
 138 
 139   Raises:
 140     ValueError: If `predictions` and `labels` have mismatched shapes, or if
 141       `weights` is not `None` and its shape doesn't match `predictions`, or if
 142       either `metrics_collections` or `updates_collections` are not a list or
 143       tuple.
 144   """
 145   with variable_scope.variable_scope(
 146       name, 'f1', (labels, predictions, weights)):
 147     predictions, labels, weights = metrics_impl._remove_squeezable_dimensions(  # pylint: disable=protected-access
 148         predictions=predictions, labels=labels, weights=weights)
 149     # To account for floating point imprecisions / avoid division by zero.
 150     epsilon = 1e-7
 151     thresholds = [(i + 1) * 1.0 / (num_thresholds - 1)
 152                   for i in range(num_thresholds - 2)]
 153     thresholds = [0.0 - epsilon] + thresholds + [1.0 + epsilon]
 154 
 155     # Confusion matrix.
 156     values, update_ops = metrics_impl._confusion_matrix_at_thresholds(  # pylint: disable=protected-access
 157         labels, predictions, thresholds, weights, includes=('tp', 'fp', 'fn'))
 158 
 159     # Compute precision and recall at various thresholds.
 160     def compute_best_f1_score(tp, fp, fn, name):
 161       precision_at_t = math_ops.div(tp, epsilon + tp + fp,
 162                                     name='precision_' + name)
 163       recall_at_t = math_ops.div(tp, epsilon + tp + fn, name='recall_' + name)
 164       # Compute F1 score.
 165       f1_at_thresholds = (
 166           2.0 * precision_at_t * recall_at_t /
 167           (precision_at_t + recall_at_t + epsilon))
 168       return math_ops.reduce_max(f1_at_thresholds)
 169 
 170     def f1_across_towers(_, values):
 171       best_f1 = compute_best_f1_score(tp=values['tp'], fp=values['fp'],
 172                                       fn=values['fn'], name='value')
 173       if metrics_collections:
 174         ops.add_to_collections(metrics_collections, best_f1)
 175       return best_f1
 176 
 177     best_f1 = distribution_strategy_context.get_tower_context().merge_call(
 178         f1_across_towers, values)
 179 
 180     update_op = compute_best_f1_score(tp=update_ops['tp'], fp=update_ops['fp'],
 181                                       fn=update_ops['fn'], name='update')
 182     if updates_collections:
 183       ops.add_to_collections(updates_collections, update_op)
 184 
 185     return best_f1, update_op
