  39 def auc_using_histogram(boolean_labels,
  40                         scores,
  41                         score_range,
  42                         nbins=100,
  43                         collections=None,
  44                         check_shape=True,
  45                         name=None):
  46   """AUC computed by maintaining histograms.
  47 
  48   Rather than computing AUC directly, this Op maintains Variables containing
  49   histograms of the scores associated with `True` and `False` labels.  By
  50   comparing these the AUC is generated, with some discretization error.
  51   See: "Efficient AUC Learning Curve Calculation" by Bouckaert.
  52 
  53   This AUC Op updates in `O(batch_size + nbins)` time and works well even with
  54   large class imbalance.  The accuracy is limited by discretization error due
  55   to finite number of bins.  If scores are concentrated in a fewer bins,
  56   accuracy is lower.  If this is a concern, we recommend trying different
  57   numbers of bins and comparing results.
  58 
  59   Args:
  60     boolean_labels:  1-D boolean `Tensor`.  Entry is `True` if the corresponding
  61       record is in class.
  62     scores:  1-D numeric `Tensor`, same shape as boolean_labels.
  63     score_range:  `Tensor` of shape `[2]`, same dtype as `scores`.  The min/max
  64       values of score that we expect.  Scores outside range will be clipped.
  65     nbins:  Integer number of bins to use.  Accuracy strictly increases as the
  66       number of bins increases.
  67     collections: List of graph collections keys. Internal histogram Variables
  68       are added to these collections. Defaults to `[GraphKeys.LOCAL_VARIABLES]`.
  69     check_shape:  Boolean.  If `True`, do a runtime shape check on the scores
  70       and labels.
  71     name:  A name for this Op.  Defaults to "auc_using_histogram".
  72 
  73   Returns:
  74     auc:  `float32` scalar `Tensor`.  Fetching this converts internal histograms
  75       to auc value.
  76     update_op:  `Op`, when run, updates internal histograms.
  77   """
  78   if collections is None:
  79     collections = [ops.GraphKeys.LOCAL_VARIABLES]
  80   with variable_scope.variable_scope(
  81       name, 'auc_using_histogram', [boolean_labels, scores, score_range]):
  82     scores, boolean_labels = tensor_util.remove_squeezable_dimensions(
  83         scores, boolean_labels)
  84     score_range = ops.convert_to_tensor(score_range, name='score_range')
  85     boolean_labels, scores = _check_labels_and_scores(
  86         boolean_labels, scores, check_shape)
  87     hist_true, hist_false = _make_auc_histograms(boolean_labels, scores,
  88                                                  score_range, nbins)
  89     hist_true_acc, hist_false_acc, update_op = _auc_hist_accumulate(hist_true,
  90                                                                     hist_false,
  91                                                                     nbins,
  92                                                                     collections)
  93     auc = _auc_convert_hist_to_auc(hist_true_acc, hist_false_acc, nbins)
  94     return auc, update_op
  95

  97 def _check_labels_and_scores(boolean_labels, scores, check_shape):
  98   """Check the rank of labels/scores, return tensor versions."""
  99   with ops.name_scope('_check_labels_and_scores',
 100                       values=[boolean_labels, scores]):
 101     boolean_labels = ops.convert_to_tensor(boolean_labels,
 102                                            name='boolean_labels')
 103     scores = ops.convert_to_tensor(scores, name='scores')
 104 
 105     if boolean_labels.dtype != dtypes.bool:
 106       raise ValueError(
 107           'Argument boolean_labels should have dtype bool.  Found: %s',
 108           boolean_labels.dtype)
 109 
 110     if check_shape:
 111       labels_rank_1 = control_flow_ops.Assert(
 112           math_ops.equal(1, array_ops.rank(boolean_labels)),
 113           ['Argument boolean_labels should have rank 1.  Found: ',
 114            boolean_labels.name, array_ops.shape(boolean_labels)])
 115 
 116       scores_rank_1 = control_flow_ops.Assert(
 117           math_ops.equal(1, array_ops.rank(scores)),
 118           ['Argument scores should have rank 1.  Found: ', scores.name,
 119            array_ops.shape(scores)])
 120 
 121       with ops.control_dependencies([labels_rank_1, scores_rank_1]):
 122         return boolean_labels, scores
 123     else:
 124       return boolean_labels, scores

 127 def _make_auc_histograms(boolean_labels, scores, score_range, nbins):
 128   """Create histogram tensors from one batch of labels/scores."""
 129 
 130   with variable_scope.variable_scope(
 131       None, 'make_auc_histograms', [boolean_labels, scores, nbins]):
 132     # Histogram of scores for records in this batch with True label.
 133     hist_true = histogram_ops.histogram_fixed_width(
 134         array_ops.boolean_mask(scores, boolean_labels),
 135         score_range,
 136         nbins=nbins,
 137         dtype=dtypes.int64,
 138         name='hist_true')
 139     # Histogram of scores for records in this batch with False label.
 140     hist_false = histogram_ops.histogram_fixed_width(
 141         array_ops.boolean_mask(scores, math_ops.logical_not(boolean_labels)),
 142         score_range,
 143         nbins=nbins,
 144         dtype=dtypes.int64,
 145         name='hist_false')
 146     return hist_true, hist_false

 149 def _auc_hist_accumulate(hist_true, hist_false, nbins, collections):
 150   """Accumulate histograms in new variables."""
 151   with variable_scope.variable_scope(
 152       None, 'hist_accumulate', [hist_true, hist_false]):
 153     # Holds running total histogram of scores for records labeled True.
 154     hist_true_acc = variable_scope.get_variable(
 155         'hist_true_acc',
 156         shape=[nbins],
 157         dtype=hist_true.dtype,
 158         initializer=init_ops.zeros_initializer(),
 159         collections=collections,
 160         trainable=False)
 161     # Holds running total histogram of scores for records labeled False.
 162     hist_false_acc = variable_scope.get_variable(
 163         'hist_false_acc',
 164         shape=[nbins],
 165         dtype=hist_true.dtype,
 166         initializer=init_ops.zeros_initializer(),
 167         collections=collections,
 168         trainable=False)
 169 
 170     update_op = control_flow_ops.group(
 171         hist_true_acc.assign_add(hist_true),
 172         hist_false_acc.assign_add(hist_false),
 173         name='update_op')
 174 
 175     return hist_true_acc, hist_false_acc, update_op

 178 def _auc_convert_hist_to_auc(hist_true_acc, hist_false_acc, nbins):
 179   """Convert histograms to auc.
 180 
 181   Args:
 182     hist_true_acc:  `Tensor` holding accumulated histogram of scores for records
 183       that were `True`.
 184     hist_false_acc:  `Tensor` holding accumulated histogram of scores for
 185       records that were `False`.
 186     nbins:  Integer number of bins in the histograms.
 187 
 188   Returns:
 189     Scalar `Tensor` estimating AUC.
 190   """
 191   # Note that this follows the "Approximating AUC" section in:
 192   # Efficient AUC learning curve calculation, R. R. Bouckaert,
 193   # AI'06 Proceedings of the 19th Australian joint conference on Artificial
 194   # Intelligence: advances in Artificial Intelligence
 195   # Pages 181-191.
 196   # Note that the above paper has an error, and we need to re-order our bins to
 197   # go from high to low score.
 198 
 199   # Normalize histogram so we get fraction in each bin.
 200   normed_hist_true = math_ops.truediv(hist_true_acc,
 201                                       math_ops.reduce_sum(hist_true_acc))
 202   normed_hist_false = math_ops.truediv(hist_false_acc,
 203                                        math_ops.reduce_sum(hist_false_acc))
 204 
 205   # These become delta x, delta y from the paper.
 206   delta_y_t = array_ops.reverse_v2(normed_hist_true, [0], name='delta_y_t')
 207   delta_x_t = array_ops.reverse_v2(normed_hist_false, [0], name='delta_x_t')
 208 
 209   # strict_1d_cumsum requires float32 args.
 210   delta_y_t = math_ops.cast(delta_y_t, dtypes.float32)
 211   delta_x_t = math_ops.cast(delta_x_t, dtypes.float32)
 212 
 213   # Trapezoidal integration, \int_0^1 0.5 * (y_t + y_{t-1}) dx_t
 214   y_t = _strict_1d_cumsum(delta_y_t, nbins)
 215   first_trap = delta_x_t[0] * y_t[0] / 2.0
 216   other_traps = delta_x_t[1:] * (y_t[1:] + y_t[:nbins - 1]) / 2.0
 217   return math_ops.add(first_trap, math_ops.reduce_sum(other_traps), name='auc')

 220 # TODO(langmore) Remove once a faster cumsum (accumulate_sum) Op is available.
 221 # Also see if cast to float32 above can be removed with new cumsum.
 222 # See:  https://github.com/tensorflow/tensorflow/issues/813
 223 def _strict_1d_cumsum(tensor, len_tensor):
 224   """Cumsum of a 1D tensor with defined shape by padding and convolving."""
 225   # Assumes tensor shape is fully defined.
 226   with ops.name_scope('strict_1d_cumsum', values=[tensor]):
 227     if len_tensor == 0:
 228       return constant_op.constant([])
 229     len_pad = len_tensor - 1
 230     x = array_ops.pad(tensor, [[len_pad, 0]])
 231     h = array_ops.ones_like(x)
 232     return _strict_conv1d(x, h)[:len_tensor]
 233

 235 # TODO(langmore) Remove once a faster cumsum (accumulate_sum) Op is available.
 236 # See:  https://github.com/tensorflow/tensorflow/issues/813
 237 def _strict_conv1d(x, h):
 238   """Return x * h for rank 1 tensors x and h."""
 239   with ops.name_scope('strict_conv1d', values=[x, h]):
 240     x = array_ops.reshape(x, (1, -1, 1, 1))
 241     h = array_ops.reshape(h, (-1, 1, 1, 1))
 242     result = nn_ops.conv2d(x, h, [1, 1, 1, 1], 'SAME')
 243     return array_ops.reshape(result, [-1])
