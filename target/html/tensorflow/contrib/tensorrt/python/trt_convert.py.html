  66 def tensorrt_rewriter_config(max_batch_size=1,
  67                              max_workspace_size_bytes=2 << 20,
  68                              precision_mode=TrtPrecisionMode.FP32,
  69                              minimum_segment_size=3,
  70                              is_dynamic_op=False,
  71                              maximum_cached_engines=1,
  72                              cached_engine_batch_sizes=None):
  73   """Returns a RewriterConfig proto for TRT transformation.
  74 
  75   Args:
  76     max_batch_size: max size for the input batch
  77     max_workspace_size_bytes: the maximum GPU temporary memory which the TRT
  78       engine can use at execution time. This corresponds to the 'workspaceSize'
  79       parameter of nvinfer1::IBuilder::setMaxWorkspaceSize().
  80     precision_mode: one of TrtPrecisionMode.supported_precision_modes().
  81     minimum_segment_size: the minimum number of nodes required for a subgraph to
  82       be replaced by TRTEngineOp.
  83     is_dynamic_op: whether to generate dynamic TRT ops which will build the TRT
  84       network and engine at run time.
  85     maximum_cached_engines: max number of cached TRT engines in dynamic TRT ops.
  86       If the number of cached engines is already at max but none of them can
  87       serve the input, the TRTEngineOp will fall back to run the TF function
  88       based on which the TRTEngineOp is created.
  89     cached_engine_batch_sizes: a list of batch sizes used to create cached
  90       engines, only used when is_dynamic_op is True. The length of the list
  91       should be smaller than maximum_cached_engines, and the dynamic TRT op will
  92       use this list to determine the batch sizes of the cached engines, instead
  93       of making the decision on the fly. This is useful when we know the most
  94       common batch size(s) the application is going to generate.
  95 
  96   Returns:
  97     A RewriterConfig proto which sets a TensorRTOptimizer to run Grappler.
  98 
  99   Raises:
 100     TypeError: if the provided precision mode is invalid.
 101     ValueError: if len(cached_engine_batch_sizes) exceed maximum_cached_engines.
 102   """
 103   if precision_mode.upper() not in TrtPrecisionMode.supported_precision_modes():
 104     raise ValueError(("precision mode '{}' is not supported."
 105                       "It should be one of {}").format(
 106                           precision_mode,
 107                           TrtPrecisionMode.supported_precision_modes))
 108 
 109   rewriter_cfg = rewriter_config_pb2.RewriterConfig()
 110   rewriter_cfg.optimizers.extend(["constfold", "layout"])
 111   optimizer = rewriter_cfg.custom_optimizers.add()
 112   optimizer.name = "TensorRTOptimizer"
 113   optimizer.parameter_map["minimum_segment_size"].i = minimum_segment_size
 114   optimizer.parameter_map["max_batch_size"].i = max_batch_size
 115   optimizer.parameter_map["is_dynamic_op"].b = is_dynamic_op
 116   optimizer.parameter_map[
 117       "max_workspace_size_bytes"].i = max_workspace_size_bytes
 118   optimizer.parameter_map["precision_mode"].s = _to_bytes(precision_mode)
 119   optimizer.parameter_map["maximum_cached_engines"].i = maximum_cached_engines
 120   if cached_engine_batch_sizes:
 121     if not isinstance(cached_engine_batch_sizes, list):
 122       raise TypeError("cached_engine_batch_sizes should be a list.")
 123     if len(cached_engine_batch_sizes) > maximum_cached_engines:
 124       raise ValueError("cached_engine_batch_sizes should not contain more than "
 125                        "maximum_cached_engines items.")
 126     optimizer.parameter_map["cached_engine_batches"].list.i.extend(
 127         cached_engine_batch_sizes)
 128   return rewriter_cfg

 131 def create_inference_graph(input_graph_def,
 132                            outputs,
 133                            max_batch_size=1,
 134                            max_workspace_size_bytes=2 << 20,
 135                            precision_mode=TrtPrecisionMode.FP32,
 136                            minimum_segment_size=3,
 137                            is_dynamic_op=False,
 138                            maximum_cached_engines=1,
 139                            cached_engine_batch_sizes=None,
 140                            input_saved_model_dir=None,
 141                            input_saved_model_tags=None,
 142                            output_saved_model_dir=None,
 143                            session_config=None):
 144   """Python wrapper for the TRT transformation.
 145 
 146   Args:
 147     input_graph_def: a GraphDef object containing a model to be transformed. If
 148       set to None, the graph will be read from the SavedModel loaded from
 149       input_saved_model_dir.
 150     outputs: list of tensors or node names for the model outputs. Only used when
 151       input_graph_def is not None.
 152     max_batch_size: max size for the input batch.
 153     max_workspace_size_bytes: the maximum GPU temporary memory which the TRT
 154       engine can use at execution time. This corresponds to the 'workspaceSize'
 155       parameter of nvinfer1::IBuilder::setMaxWorkspaceSize().
 156     precision_mode: one of TrtPrecisionMode.supported_precision_modes().
 157     minimum_segment_size: the minimum number of nodes required for a subgraph to
 158       be replaced by TRTEngineOp.
 159     is_dynamic_op: whether to generate dynamic TRT ops which will build the TRT
 160       network and engine at run time.
 161     maximum_cached_engines: max number of cached TRT engines in dynamic TRT ops.
 162       If the number of cached engines is already at max but none of them can
 163       serve the input, the TRTEngineOp will fall back to run the TF function
 164       based on which the TRTEngineOp is created.
 165     cached_engine_batch_sizes: a list of batch sizes used to create cached
 166       engines, only used when is_dynamic_op is True. The length of the list
 167       should be smaller than maximum_cached_engines, and the dynamic TRT op will
 168       use this list to determine the batch sizes of the cached engines, instead
 169       of making the decision on the fly. This is useful when we know the most
 170       common batch size(s) the application is going to generate.
 171     input_saved_model_dir: the directory to load the SavedModel which contains
 172       the input graph to transforms. Used only when input_graph_def is None.
 173     input_saved_model_tags: list of tags to load the SavedModel.
 174     output_saved_model_dir: if not None, construct a SavedModel using the
 175       returned GraphDef and save it to the specified directory. This option only
 176       works when the input graph is loaded from a SavedModel, i.e. when
 177       input_saved_model_dir is specified and input_graph_def is None.
 178     session_config: the ConfigProto used to create a Session. If not specified,
 179       a default ConfigProto will be used.
 180 
 181   Returns:
 182     A GraphDef transformed from input_graph_def (or the SavedModel graph def
 183     loaded from input_saved_model_dir, if input_graph_def is not present), where
 184     all TRT compatible subgraphs are replaced with TRTEngineOps, and a TF
 185     function is added for each of the subgraphs.
 186 
 187     If is_dynamic_op is True, each TRTEngineOp will contain a serialized
 188     subgraph GraphDef, which will be converted to a TRT engine at execution time
 189     and the TRT engine will be cached for future usage. A new TRT engine will be
 190     created each time when none of the cached engines match the input shapes. If
 191     it fails to execute the TRT engine or the number of cached engines reaches
 192     maximum_cached_engines, the op will fall back to call the corresponding TF
 193     function.
 194 
 195     If is_dynamic_op is False, each TRTEngineOp will contain a serialized TRT
 196     engine created from the corresponding subgraph. No more engines will be
 197     created on the fly, and the op will fall back to call the corresponding TF
 198     function when it fails to execute the engine.
 199 
 200   Raises:
 201     ValueError: if the combination of the parameters is invalid.
 202     RuntimeError: if the TensorRT library version is incompatible.
 203   """
 204   compiled_version = get_linked_tensorrt_version()
 205   loaded_version = get_loaded_tensorrt_version()
 206   version_mismatch = False
 207   if loaded_version[0] < compiled_version[0]:
 208     tf_logging.error(
 209         "TensorRT version mismatch. Tensorflow was compiled against " +
 210         "TensorRT %s but library loaded from environment is TensorRT %s" %
 211         (".".join([str(x) for x in compiled_version]),
 212          ".".join([str(x) for x in loaded_version])) +
 213         ". Please make sure that correct version of TensorRT " +
 214         "is available in the system and added to ldconfig or LD_LIBRARY_PATH")
 215     raise RuntimeError("Incompatible TensorRT library version")
 216   for i in zip(loaded_version, compiled_version):
 217     if i[0] != i[1]:
 218       tf_logging.warn("TensorRT mismatch. Compiled against version " +
 219                       "%s, but loaded %s. Things may not work" %
 220                       (".".join([str(x) for x in compiled_version]),
 221                        ".".join([str(x) for x in loaded_version])))
 222       version_mismatch = True
 223       break
 224   if not version_mismatch:
 225     tf_logging.info("Running against TensorRT version %s" % ".".join(
 226         [str(x) for x in loaded_version]))
 227 
 228   if session_config is None:
 229     session_config = config_pb2.ConfigProto()
 230 
 231   if input_saved_model_tags is None:
 232     input_saved_model_tags = [tag_constants.SERVING]
 233   saved_model_loader = None
 234   grappler_meta_graph_def = None
 235 
 236   if input_graph_def is None:
 237     # Read from SavedModel and freeze the graph if necessary.
 238     if input_saved_model_dir is None:
 239       raise ValueError("input_graph_def and input_saved_model_dir cannot be "
 240                        "both None")
 241     with ops.Graph().as_default():
 242       with session.Session(config=session_config) as sess:
 243         saved_model_loader = loader_impl.SavedModelLoader(input_saved_model_dir)
 244         input_meta_graph_def = saved_model_loader.load(sess,
 245                                                        input_saved_model_tags)
 246         output_node_names = set()
 247 
 248         def _gather_names(tensor_info):
 249           """Get the node names from a TensorInfo."""
 250           return set(
 251               [tensor_info[key].name.split(":")[0] for key in tensor_info])
 252 
 253         # Get input and outputs from all SignatureDef.
 254         for key in input_meta_graph_def.signature_def:
 255           signature_def = input_meta_graph_def.signature_def[key]
 256           output_node_names.update(_gather_names(signature_def.inputs))
 257           output_node_names.update(_gather_names(signature_def.outputs))
 258 
 259         # Freeze the variables in the SavedModel graph and copy the frozen
 260         # graph over.
 261         frozen_graph_def = graph_util.convert_variables_to_constants(
 262             sess, sess.graph.as_graph_def(add_shapes=True),
 263             list(output_node_names))
 264         grappler_meta_graph_def = meta_graph_pb2.MetaGraphDef()
 265         grappler_meta_graph_def.graph_def.CopyFrom(frozen_graph_def)
 266 
 267         # Copy the collections that are not variables.
 268         for key in input_meta_graph_def.collection_def:
 269           # TODO(laigd): currently we use the collection key to filter out
 270           # collections that depend on variable ops, but this may miss some
 271           # other user-defined collections. A better way would be to use
 272           # CollectionDef::NodeList for the filtering.
 273           if key not in [
 274               "variables", "local_variables", "model_variables",
 275               "trainable_variables", "train_op", "table_initializer"
 276           ]:
 277             grappler_meta_graph_def.collection_def[key].CopyFrom(
 278                 input_meta_graph_def.collection_def[key])
 279 
 280         # Copy other information.
 281         grappler_meta_graph_def.meta_info_def.CopyFrom(
 282             input_meta_graph_def.meta_info_def)
 283         for key in input_meta_graph_def.signature_def:
 284           grappler_meta_graph_def.signature_def[key].CopyFrom(
 285               input_meta_graph_def.signature_def[key])
 286         # TODO(laigd): maybe add back AssetFileDef.
 287   else:
 288     if output_saved_model_dir is not None:
 289       raise ValueError("output_saved_model_dir cannot be set when "
 290                        "input_graph_def is set")
 291     # Create MetaGraphDef from input graph.
 292     graph = ops.Graph()
 293     with graph.as_default():
 294       importer.import_graph_def(input_graph_def, name="")
 295     grappler_meta_graph_def = saver.export_meta_graph(
 296         graph_def=graph.as_graph_def(add_shapes=True), graph=graph)
 297     if outputs:
 298       output_collection = meta_graph_pb2.CollectionDef()
 299       output_list = output_collection.node_list.value
 300       for i in outputs:
 301         if isinstance(i, ops.Tensor):
 302           output_list.append(_to_bytes(i.name))
 303         else:
 304           output_list.append(_to_bytes(i))
 305       # TODO(laigd): use another key as the outputs are really not train_op.
 306       grappler_meta_graph_def.collection_def["train_op"].CopyFrom(
 307           output_collection)
 308 
 309   # Create RewriterConfig.
 310   rewriter_cfg = tensorrt_rewriter_config(
 311       max_batch_size, max_workspace_size_bytes, precision_mode,
 312       minimum_segment_size, is_dynamic_op, maximum_cached_engines,
 313       cached_engine_batch_sizes)
 314 
 315   # Run Grappler.
 316   transformed_graph_def = tf_optimizer.OptimizeGraph(
 317       rewriter_cfg, grappler_meta_graph_def, graph_id=b"tf_graph")
 318 
 319   # Optionally write the transformed graphdef as SavedModel.
 320   if output_saved_model_dir is not None:
 321     saved_model_builder = builder.SavedModelBuilder(output_saved_model_dir)
 322     with ops.Graph().as_default():
 323       importer.import_graph_def(transformed_graph_def, name="")
 324       with session.Session(config=session_config) as sess:
 325         saved_model_builder.add_meta_graph_and_variables(
 326             sess,
 327             input_saved_model_tags,
 328             signature_def_map=grappler_meta_graph_def.signature_def)
 329     # Ignore other meta graphs from the input SavedModel.
 330     saved_model_builder.save()
 331 
 332   return transformed_graph_def

 335 def calib_graph_to_infer_graph(calibration_graph_def, is_dynamic_op=False):
 336   """Convert an existing calibration graph to inference graph.
 337 
 338   Args:
 339     calibration_graph_def: the calibration GraphDef object with calibration data
 340     is_dynamic_op: whether to create dynamic static engines from calibration
 341 
 342   Returns:
 343     New GraphDef with TRTEngineOps placed in graph replacing calibration nodes.
 344   Raises:
 345     RuntimeError: if the returned status message is malformed.
 346   """
 347 
 348   is_calib_graph = False
 349   for n in calibration_graph_def.node:
 350     if n.op == "TRTEngineOp":
 351       is_calib_graph = is_calib_graph or not n.attr["calibration_data"].s
 352   if not is_calib_graph:
 353     tf_logging.error(
 354         "Not a calib graph. Doesn't seem to contain any calibration nodes.")
 355     return None
 356   graph_str = calibration_graph_def.SerializeToString()
 357   out = calib_convert(graph_str, is_dynamic_op)
 358   status = _to_string(out[0])
 359   output_graph_def_string = out[1]
 360   del graph_str  # Save some memory
 361   if len(status) < 2:
 362     raise _impl.UnknownError(None, None, status)
 363   if status[:2] != "OK":
 364     msg = status.split(";")
 365     if len(msg) == 1:
 366       raise RuntimeError("Status message is malformed {}".format(status))
 367     # pylint: disable=protected-access
 368     raise _impl._make_specific_exception(None, None, ";".join(msg[1:]),
 369                                          int(msg[0]))
 370     # pylint: enable=protected-access
 371   output_graph_def = graph_pb2.GraphDef()
 372   output_graph_def.ParseFromString(output_graph_def_string)
 373   del output_graph_def_string  # Save some memory
 374   return output_graph_def
