  79   # str is bytes in py2, but unicode in py3.
  80   def _ToUnicode(self, s):
  81     if six.PY2:
  82       if isinstance(s, unicode):
  83         return s
  84       return s.decode("utf-8")
  85     else:
  86       if isinstance(s, str):
  87         return s
  88       return s.decode("utf-8")
  89 
  90   def _ToBytes(self, s):
  91     if six.PY2:
  92       if isinstance(s, unicode):
  93         return s.encode("utf-8")
  94       return s
  95     else:
  96       if isinstance(s, str):
  97         return s.encode("utf-8")
  98       return s
  99 
 100   def _ToString(self, s):
 101     if six.PY2:
 102       if isinstance(s, unicode):
 103         return s.encode("utf-8")
 104       return s
 105     else:
 106       if isinstance(s, str):
 107         return s
 108       return s.decode("utf-8")

 147   def VerifyRunForEngine(self, engine_name, graph_state, expect_run=True):
 148     """Verify the state of a particular engine after sess.run()."""
 149     if graph_state == GraphState.ORIGINAL:
 150       self._ExpectCalibration(engine_name, "")
 151       self._ExpectNativeSegment(engine_name, "")
 152       self._ExpectTrtEngine(engine_name, "")
 153     elif graph_state == GraphState.CALIBRATE:
 154       self._ExpectCalibration(engine_name, "done")
 155       self._ExpectNativeSegment(engine_name, "done")
 156       self._ExpectTrtEngine(engine_name, "")
 157     elif graph_state == GraphState.INFERENCE:
 158       self._ExpectCalibration(engine_name, "")
 159       if expect_run:
 160         self._ExpectNativeSegment(engine_name, "")
 161         self._ExpectTrtEngine(engine_name, "done")
 162       else:
 163         self._ExpectNativeSegment(engine_name, "done")
 164         self._ExpectTrtEngine(engine_name, "")
 165 
 166   def VerifyRun(self, run_params, graph_state):
 167     """Verify the state of all engines after sess.run()."""
 168     for engine_name in self.ExpectedEnginesToBuild(run_params):
 169       expect_run = (engine_name in self.ExpectedEnginesToRun(run_params))
 170       self.VerifyRunForEngine(engine_name, graph_state, expect_run)

 193   def _PrepareRun(self, graph_state):
 194     """Set up necessary testing environment before calling sess.run()."""
 195     # Clear test values added by TRTEngineOp.
 196     trt_convert.clear_test_values("my_trt_op_.*:ExecuteTrtEngine")
 197     trt_convert.clear_test_values("my_trt_op_.*:ExecuteCalibration")
 198     trt_convert.clear_test_values("my_trt_op_.*:ExecuteNativeSegment")

 200   def _GetConfigProto(self, run_params, graph_state):
 201     """Get config proto based on specific settings."""
 202     if graph_state != GraphState.ORIGINAL and run_params.use_optimizer:
 203       trt_params = self.GetConversionParams(run_params)
 204       rewriter_cfg = trt_convert.tensorrt_rewriter_config(
 205           trt_params.max_batch_size, trt_params.max_workspace_size_bytes,
 206           trt_params.precision_mode, trt_params.minimum_segment_size,
 207           trt_params.is_dynamic_op, trt_params.maximum_cached_engines,
 208           trt_params.cached_engine_batch_sizes)
 209 
 210       graph_options = config_pb2.GraphOptions(rewrite_options=rewriter_cfg)
 211     else:
 212       graph_options = config_pb2.GraphOptions()
 213 
 214     gpu_options = config_pb2.GPUOptions()
 215     gpu_options.allow_growth = True
 216     if trt_convert.get_linked_tensorrt_version()[0] == 3:
 217       gpu_options.per_process_gpu_memory_fraction = 0.50
 218 
 219     config = config_pb2.ConfigProto(
 220         gpu_options=gpu_options, graph_options=graph_options)
 221     return config

 223   def _ExpectTestValue(self, engine_name, method, expected_value):
 224     label = "%s:%s" % (engine_name, method)
 225     actual_value = trt_convert.get_test_value(label)
 226     self.assertEqual(
 227         expected_value,
 228         actual_value,
 229         msg="Unexpected test value with label %s. Actual: %s; expected: %s" %
 230         (label, actual_value, expected_value))

 241   def _RunGraph(self,
 242                 run_params,
 243                 gdef,
 244                 input_data,
 245                 config,
 246                 graph_state,
 247                 num_runs=2):
 248     """Run given graphdef multiple times."""
 249     params = self._GetParamsCached()
 250     assert len(params.input_names) == len(input_data)
 251     g = ops.Graph()
 252     with g.as_default():
 253       io_ops = importer.import_graph_def(
 254           graph_def=gdef,
 255           return_elements=params.input_names + params.output_names,
 256           name="")
 257       inputs = [op.outputs[0] for op in io_ops[:len(params.input_names)]]
 258       assert len(inputs) == len(input_data)
 259       outputs = [op.outputs[0] for op in io_ops[len(params.input_names):]]
 260     with self.test_session(
 261         graph=g, config=config, use_gpu=True, force_gpu=True) as sess:
 262       val = None
 263       # Defaults to 2 runs to verify result across multiple runs is same.
 264       for _ in range(num_runs):
 265         self._PrepareRun(graph_state)
 266         new_val = sess.run(
 267             outputs, {inputs[i]: input_data[i] for i in range(len(inputs))})
 268         output_len = len(params.expected_output_dims)
 269         self.assertEqual(output_len, len(new_val))
 270         for i in range(output_len):
 271           self.assertEqual(params.expected_output_dims[i], new_val[i].shape)
 272         if val is not None:
 273           self.assertAllClose(val, new_val, atol=1.e-06, rtol=1.e-06)
 274         val = new_val
 275         self.VerifyRun(run_params, graph_state)
 276     return val

 285   def _GetTrtGraphDef(self, run_params, gdef):
 286     """Return trt converted graphdef."""
 287     params = self._GetParamsCached()
 288     trt_params = self.GetConversionParams(run_params)
 289     logging.info(trt_params)
 290     return trt_convert.create_inference_graph(
 291         input_graph_def=gdef,
 292         outputs=params.input_names + params.output_names,
 293         max_batch_size=trt_params.max_batch_size,
 294         max_workspace_size_bytes=trt_params.max_workspace_size_bytes,
 295         precision_mode=trt_params.precision_mode,
 296         minimum_segment_size=trt_params.minimum_segment_size,
 297         is_dynamic_op=trt_params.is_dynamic_op,
 298         maximum_cached_engines=trt_params.maximum_cached_engines,
 299         cached_engine_batch_sizes=trt_params.cached_engine_batch_sizes)

 301   def _WriteGraph(self, run_params, gdef, graph_state):
 302     if graph_state == GraphState.ORIGINAL:
 303       label = "Original"
 304     elif graph_state == GraphState.CALIBRATE:
 305       label = "CalibEngine"
 306     elif graph_state == GraphState.INFERENCE:
 307       label = "InferEngine"
 308     graph_name = (
 309         self.__class__.__name__ + "_" + run_params.test_name + "_" + label +
 310         ".pbtxt")
 311     temp_dir = os.getenv("TRT_TEST_TMPDIR", self.get_temp_dir())
 312     if temp_dir:
 313       logging.info("Writing graph to %s/%s", temp_dir, graph_name)
 314       graph_io.write_graph(gdef, temp_dir, graph_name)

 316   def _VerifyConnections(self, expected_engines, converted_gdef):
 317     params = self._GetParamsCached()
 318     old_to_new_node_map = {
 319         self._ToString(node.name): self._ToString(node.name)
 320         for node in params.gdef.node
 321     }
 322     for engine_name, node_names in expected_engines.items():
 323       for node_name in node_names:
 324         old_to_new_node_map[node_name] = engine_name
 325     name_to_node_map = {
 326         self._ToString(node.name): node for node in params.gdef.node
 327     }
 328 
 329     def _InputName(inp):
 330       inp = self._ToString(inp)
 331       prefix = ""
 332       if inp[0] == "^":
 333         prefix = "^"
 334         inp = inp[1:]
 335       parts = inp.split(":")
 336       if len(parts) > 1 and parts[-1].isdigit():
 337         inp = inp[:-len(parts[-1]) - 1]
 338       return (prefix, inp)
 339 
 340     expected_input_map = {}
 341     for node in params.gdef.node:
 342       name_str = self._ToString(node.name)
 343       target_node_name = old_to_new_node_map[name_str]
 344       is_engine_op = (target_node_name != name_str)
 345       if target_node_name not in expected_input_map:
 346         expected_input_map[target_node_name] = set()
 347       input_set = expected_input_map[target_node_name]
 348       for inp in node.input:
 349         (prefix, inp_name) = _InputName(inp)
 350         # Add the input only if it's outside the segment (note that it could be
 351         # in a different engine).
 352         if (not is_engine_op or
 353             old_to_new_node_map[inp_name] != target_node_name):
 354           if is_engine_op and name_to_node_map[inp_name].op == "Const":
 355             # Const data input nodes to the segment has been copied to the
 356             # segment graphdef and the engine, and the dependency has been
 357             # converted to control dependendy.
 358             input_set.add("^" + old_to_new_node_map[inp_name])
 359           else:
 360             input_set.add(prefix + old_to_new_node_map[inp_name])
 361 
 362     actual_input_map = {}
 363     for node in converted_gdef.node:
 364       name_str = self._ToString(node.name)
 365       actual_input_map[name_str] = set()
 366       input_set = actual_input_map[name_str]
 367       for inp in node.input:
 368         (prefix, node_name) = _InputName(inp)
 369         input_set.add(prefix + node_name)
 370 
 371     self.assertEqual(
 372         expected_input_map,
 373         actual_input_map,
 374         msg="expected:\n%s\nvs actual:\n%s" % (sorted(
 375             expected_input_map.items()), sorted(actual_input_map.items())))

 377   def _VerifyGraphDef(self, run_params, gdef, graph_state):
 378     self._WriteGraph(run_params, gdef, graph_state)
 379 
 380     expected_engines = self.ExpectedEnginesToBuild(run_params)
 381     num_engines = 0
 382     for node in gdef.node:
 383       if node.op == "TRTEngineOp":
 384         logging.info("Found TRTEngineOp: " + node.name)
 385     for node in gdef.node:
 386       if node.op == "TRTEngineOp":
 387         num_engines += 1
 388         self.assertTrue(node.name in expected_engines, node.name)
 389         self.assertTrue(len(node.attr["serialized_segment"].s), node.name)
 390         self.assertTrue(len(node.attr["segment_funcdef_name"].s), node.name)
 391         self.assertEqual(
 392             self._ToBytes(run_params.precision_mode),
 393             node.attr["precision_mode"].s, node.name)
 394 
 395         is_dynamic_engine = not node.attr["static_engine"].b
 396         self.assertEqual(run_params.dynamic_engine, is_dynamic_engine,
 397                          node.name)
 398 
 399         has_calibration_data = len(node.attr["calibration_data"].s)
 400         if (IsQuantizationMode(run_params.precision_mode) and
 401             graph_state == GraphState.INFERENCE):
 402           self.assertTrue(has_calibration_data, node.name)
 403         else:
 404           self.assertFalse(has_calibration_data, node.name)
 405     if graph_state == GraphState.ORIGINAL:
 406       self.assertEqual(0, num_engines)
 407     else:
 408       self.assertEqual(num_engines, len(expected_engines))
 409       if isinstance(expected_engines, dict):
 410         self._VerifyConnections(expected_engines, gdef)
 411       # TODO(aaroey): consider verifying the corresponding TF function.

 413   def RunTest(self, run_params):
 414     if not self.ShouldRunTest(run_params):
 415       return
 416     assert run_params.precision_mode in PRECISION_MODES
 417     np.random.seed(12345)
 418 
 419     params = self._GetParamsCached()
 420     input_gdef = params.gdef
 421     input_dtypes = {}
 422     for node in input_gdef.node:
 423       if self._ToString(node.name) in params.input_names:
 424         assert self._ToString(node.op) == "Placeholder"
 425         input_dtypes[self._ToString(node.name)] = (
 426             dtypes.as_dtype(node.attr["dtype"].type).as_numpy_dtype())
 427     assert len(params.input_names) == len(input_dtypes)
 428 
 429     input_data = []
 430     for i in range(len(params.input_names)):
 431       dtype = input_dtypes[params.input_names[i]]
 432       # Multiply the input by some constant to avoid all zeros input for integer
 433       # types.
 434       scale = 10.0 if np.issubdtype(dtype, np.integer) else 1.0
 435       dims = params.input_dims[i]
 436       input_data.append((scale * np.random.random_sample(dims)).astype(dtype))
 437     self._VerifyGraphDef(run_params, input_gdef, GraphState.ORIGINAL)
 438 
 439     # Get reference result without running trt.
 440     config_no_trt = self._GetConfigProto(run_params, GraphState.ORIGINAL)
 441     logging.info("Running original graph w/o trt, config:\n%s",
 442                  str(config_no_trt))
 443     ref_result = self._RunGraph(run_params, input_gdef, input_data,
 444                                 config_no_trt, GraphState.ORIGINAL)
 445 
 446     # Run calibration if necessary.
 447     if IsQuantizationMode(run_params.precision_mode):
 448 
 449       calib_config = self._GetConfigProto(run_params, GraphState.CALIBRATE)
 450       logging.info("Running calibration graph, config:\n%s", str(calib_config))
 451       if run_params.use_optimizer:
 452         result = self._RunCalibration(run_params, input_gdef, input_data,
 453                                       calib_config)
 454       else:
 455         calib_gdef = self._GetTrtGraphDef(run_params, input_gdef)
 456         self._VerifyGraphDef(run_params, calib_gdef, GraphState.CALIBRATE)
 457         result = self._RunCalibration(run_params, calib_gdef, input_data,
 458                                       calib_config)
 459       infer_gdef = trt_convert.calib_graph_to_infer_graph(
 460           calib_gdef, run_params.dynamic_engine)
 461       self._VerifyGraphDef(run_params, infer_gdef, GraphState.INFERENCE)
 462 
 463       self.assertAllClose(
 464           ref_result,
 465           result,
 466           atol=self.ExpectedAbsoluteTolerance(run_params),
 467           rtol=self.ExpectedRelativeTolerance(run_params))
 468     else:
 469       infer_gdef = input_gdef
 470 
 471     # Run inference.
 472     infer_config = self._GetConfigProto(run_params, GraphState.INFERENCE)
 473     logging.info("Running final inference graph, config:\n%s",
 474                  str(infer_config))
 475     if not run_params.use_optimizer:
 476       infer_gdef = self._GetTrtGraphDef(run_params, infer_gdef)
 477       self._VerifyGraphDef(run_params, infer_gdef, GraphState.INFERENCE)
 478 
 479     result = self._RunGraph(run_params, infer_gdef, input_data, infer_config,
 480                             GraphState.INFERENCE)
 481     self.assertAllClose(
 482         ref_result,
 483         result,
 484         atol=self.ExpectedAbsoluteTolerance(run_params),
 485         rtol=self.ExpectedRelativeTolerance(run_params))

 487   def testIdempotence(self):
 488     # Test that applying tensorrt optimizer or offline conversion tools multiple
 489     # times to the same graph will result in same graph.
 490     #
 491     # TODO(aaroey): currently the conversion is not deterministic, this is
 492     # mainly because during tensorflow::ConvertGraphDefToGraph(), the graph uses
 493     # EdgeSet which use a map keyed by Edge*, so the order of input/output edges
 494     # of a node is nondeterministic, thus the order for segmenter to contract
 495     # edges is nondeterministic. Need to evaluate whether we should fix this.
 496     pass

 499 def _AddTests(test_class):
 500   """Adds test methods to TfTrtIntegrationTestBase."""
 501 
 502   def _GetTest(run_params):
 503     """Gets a single test method based on the parameters."""
 504 
 505     def _Test(self):
 506       logging.info(
 507           "Running test %s with parameters: use_optimizer=%s, "
 508           "precision_mode=%s, dynamic_engine=%s",
 509           "testTfTrt_" + run_params.test_name, run_params.use_optimizer,
 510           run_params.precision_mode, run_params.dynamic_engine)
 511       self.RunTest(run_params)
 512 
 513     return _Test
 514 
 515   use_optimizer_options = [False, True]
 516   dynamic_engine_options = [False, True]
 517   for (use_optimizer, precision_mode, dynamic_engine) in itertools.product(
 518       use_optimizer_options, PRECISION_MODES, dynamic_engine_options):
 519     if IsQuantizationMode(precision_mode):
 520       if use_optimizer:
 521         # TODO(aaroey): if use_optimizer is True we need to get the inference
 522         # graphdef using custom python wrapper class, which is not currently
 523         # supported yet.
 524         continue
 525       if not dynamic_engine:
 526         # TODO(aaroey): construction of static calibration engine is not
 527         # supported yet.
 528         continue
 529 
 530     conversion = "OptimizerConversion" if use_optimizer else "ToolConversion"
 531     engine_type = ("DynamicEngine" if dynamic_engine else "StaticEngine")
 532     test_name = "%s_%s_%s" % (conversion, precision_mode, engine_type)
 533     run_params = RunParams(
 534         use_optimizer=use_optimizer,
 535         precision_mode=precision_mode,
 536         dynamic_engine=dynamic_engine,
 537         test_name=test_name)
 538     setattr(test_class, "testTfTrt_" + test_name, _GetTest(run_params))
