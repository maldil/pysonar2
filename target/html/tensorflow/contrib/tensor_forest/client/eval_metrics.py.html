  36 def _top_k_generator(k):
  37   def _top_k(probabilities, targets):
  38     targets = math_ops.to_int32(targets)
  39     if targets.get_shape().ndims > 1:
  40       targets = array_ops.squeeze(targets, axis=[1])
  41     return metrics.mean(nn.in_top_k(probabilities, targets, k))
  42   return _top_k

  45 def _accuracy(predictions, targets, weights=None):
  46   return metrics.accuracy(
  47       labels=targets, predictions=predictions, weights=weights)

  50 def _r2(probabilities, targets, weights=None):
  51   targets = math_ops.to_float(targets)
  52   y_mean = math_ops.reduce_mean(targets, 0)
  53   squares_total = math_ops.reduce_sum(math_ops.square(targets - y_mean), 0)
  54   squares_residuals = math_ops.reduce_sum(
  55       math_ops.square(targets - probabilities), 0)
  56   score = 1 - math_ops.reduce_sum(squares_residuals / squares_total)
  57   return metrics.mean(score, weights=weights)

  60 def _squeeze_and_onehot(targets, depth):
  61   targets = array_ops.squeeze(targets, axis=[1])
  62   return array_ops.one_hot(math_ops.to_int32(targets), depth)

  65 def _sigmoid_entropy(probabilities, targets, weights=None):
  66   return metrics.mean(
  67       losses.sigmoid_cross_entropy(probabilities,
  68                                    _squeeze_and_onehot(
  69                                        targets,
  70                                        array_ops.shape(probabilities)[1])),
  71       weights=weights)
  72 
  73 
  74 def _softmax_entropy(probabilities, targets, weights=None):
  75   return metrics.mean(
  76       losses.sparse_softmax_cross_entropy(probabilities,
  77                                           math_ops.to_int32(targets)),
  78       weights=weights)

  81 def _predictions(predictions, unused_targets, **unused_kwargs):

  85 def _class_log_loss(probabilities, targets, weights=None):
  86   return metrics.mean(
  87       losses.log_loss(probabilities,
  88                       _squeeze_and_onehot(targets,
  89                                           array_ops.shape(probabilities)[1])),
  90       weights=weights)

  98 def _precision_at_thresholds(predictions, targets, weights=None):
  99   return metrics.precision_at_thresholds(

 103       weights=weights)
