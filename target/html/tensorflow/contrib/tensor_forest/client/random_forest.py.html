  58 class TensorForestRunOpAtEndHook(session_run_hook.SessionRunHook):
  59 
  60   def __init__(self, op_dict):
  61     """Ops is a dict of {name: op} to run before the session is destroyed."""
  62     self._ops = op_dict
  63 
  64   def end(self, session):
  65     for name in sorted(self._ops.keys()):
  66       logging.info('{0}: {1}'.format(name, session.run(self._ops[name])))

  69 class TensorForestLossHook(session_run_hook.SessionRunHook):
  70   """Monitor to request stop when loss stops decreasing."""
  71 
  72   def __init__(self,
  73                early_stopping_rounds,
  74                early_stopping_loss_threshold=None,
  75                loss_op=None):
  76     self.early_stopping_rounds = early_stopping_rounds
  77     self.early_stopping_loss_threshold = early_stopping_loss_threshold
  78     self.loss_op = loss_op
  79     self.min_loss = None
  80     self.last_step = -1
  81     # self.steps records the number of steps for which the loss has been
  82     # non-decreasing
  83     self.steps = 0
  84 
  85   def before_run(self, run_context):
  86     loss = (self.loss_op if self.loss_op is not None else
  87             run_context.session.graph.get_operation_by_name(
  88                 LOSS_NAME).outputs[0])
  89     return session_run_hook.SessionRunArgs(
  90         {'global_step': training_util.get_global_step(),
  91          'current_loss': loss})
  92 
  93   def after_run(self, run_context, run_values):
  94     current_loss = run_values.results['current_loss']
  95     current_step = run_values.results['global_step']
  96     self.steps += 1
  97     # Guard against the global step going backwards, which might happen
  98     # if we recover from something.
  99     if self.last_step == -1 or self.last_step > current_step:
 100       logging.info('TensorForestLossHook resetting last_step.')
 101       self.last_step = current_step
 102       self.steps = 0
 103       self.min_loss = None
 104       return
 105 
 106     self.last_step = current_step
 107     if (self.min_loss is None or current_loss <
 108         (self.min_loss - self.min_loss * self.early_stopping_loss_threshold)):
 109       self.min_loss = current_loss
 110       self.steps = 0
 111     if self.steps > self.early_stopping_rounds:
 112       logging.info('TensorForestLossHook requesting stop.')
 113       run_context.request_stop()

 116 def _get_default_head(params, weights_name, output_type, name=None):
 117   """Creates a default head based on a type of a problem."""
 118   if output_type == ModelBuilderOutputType.MODEL_FN_OPS:
 119     if params.regression:
 120       return head_lib.regression_head(
 121           weight_column_name=weights_name,
 122           label_dimension=params.num_outputs,
 123           enable_centered_bias=False,
 124           head_name=name)
 125     else:
 126       return head_lib.multi_class_head(
 127           params.num_classes,
 128           weight_column_name=weights_name,
 129           enable_centered_bias=False,
 130           head_name=name)
 131   else:
 132     if params.regression:
 133       return core_head_lib.regression_head(
 134           weight_column=weights_name,
 135           label_dimension=params.num_outputs,
 136           name=name,
 137           loss_reduction=losses.Reduction.SUM_OVER_BATCH_SIZE)
 138     else:
 139       if params.num_classes == 2:
 140         return core_head_lib.binary_classification_head(
 141             weight_column=weights_name,
 142             name=name,
 143             loss_reduction=losses.Reduction.SUM_OVER_BATCH_SIZE)
 144       else:
 145         return core_head_lib.multi_class_head(
 146             n_classes=params.num_classes,
 147             weight_column=weights_name,
 148             name=name,
 149             loss_reduction=losses.Reduction.SUM_OVER_BATCH_SIZE)

 151 def get_model_fn(params,
 152                  graph_builder_class,
 153                  device_assigner,
 154                  feature_columns=None,
 155                  weights_name=None,
 156                  model_head=None,
 157                  keys_name=None,
 158                  early_stopping_rounds=100,
 159                  early_stopping_loss_threshold=0.001,
 160                  num_trainers=1,
 161                  trainer_id=0,
 162                  report_feature_importances=False,
 163                  local_eval=False,
 164                  head_scope=None,
 165                  include_all_in_serving=False,
 166                  output_type=ModelBuilderOutputType.MODEL_FN_OPS):
 167   """Return a model function given a way to construct a graph builder."""
 168   if model_head is None:
 169     model_head = _get_default_head(params, weights_name, output_type)
 170 
 171   def _model_fn(features, labels, mode):
 172     """Function that returns predictions, training loss, and training op."""
 173 
 174     if (isinstance(features, ops.Tensor) or
 175         isinstance(features, sparse_tensor.SparseTensor)):
 176       features = {'features': features}
 177     if feature_columns:
 178       features = features.copy()
 179 
 180       if output_type == ModelBuilderOutputType.MODEL_FN_OPS:
 181         features.update(layers.transform_features(features, feature_columns))
 182       else:
 183         for fc in feature_columns:
 184           tensor = fc_core._transform_features(features, [fc])[fc]  # pylint: disable=protected-access
 185           features[fc.name] = tensor
 186 
 187     weights = None
 188     if weights_name and weights_name in features:
 189       weights = features.pop(weights_name)
 190 
 191     keys = None
 192     if keys_name and keys_name in features:
 193       keys = features.pop(keys_name)
 194 
 195     # If we're doing eval, optionally ignore device_assigner.
 196     # Also ignore device assigner if we're exporting (mode == INFER)
 197     dev_assn = device_assigner
 198     if (mode == model_fn_lib.ModeKeys.INFER or
 199         (local_eval and mode == model_fn_lib.ModeKeys.EVAL)):
 200       dev_assn = None
 201 
 202     graph_builder = graph_builder_class(params,
 203                                         device_assigner=dev_assn)
 204 
 205     logits, tree_paths, regression_variance = graph_builder.inference_graph(
 206         features)
 207 
 208     summary.scalar('average_tree_size', graph_builder.average_size())
 209     # For binary classification problems, convert probabilities to logits.
 210     # Includes hack to get around the fact that a probability might be 0 or 1.
 211     if not params.regression and params.num_classes == 2:
 212       class_1_probs = array_ops.slice(logits, [0, 1], [-1, 1])
 213       logits = math_ops.log(
 214           math_ops.maximum(class_1_probs / math_ops.maximum(
 215               1.0 - class_1_probs, EPSILON), EPSILON))
 216 
 217     # labels might be None if we're doing prediction (which brings up the
 218     # question of why we force everything to adhere to a single model_fn).
 219     training_graph = None
 220     training_hooks = []
 221     if labels is not None and mode == model_fn_lib.ModeKeys.TRAIN:
 222       with ops.control_dependencies([logits.op]):
 223         training_graph = control_flow_ops.group(
 224             graph_builder.training_graph(
 225                 features, labels, input_weights=weights,
 226                 num_trainers=num_trainers,
 227                 trainer_id=trainer_id),
 228             state_ops.assign_add(training_util.get_global_step(), 1))
 229 
 230     # Put weights back in
 231     if weights is not None:
 232       features[weights_name] = weights
 233 
 234     # TensorForest's training graph isn't calculated directly from the loss
 235     # like many other models.
 236     def _train_fn(unused_loss):
 237       return training_graph
 238 
 239 
 240     # Ops are run in lexigraphical order of their keys. Run the resource
 241     # clean-up op last.
 242     all_handles = graph_builder.get_all_resource_handles()
 243     ops_at_end = {
 244         '9: clean up resources':
 245             control_flow_ops.group(*[
 246                 resource_variable_ops.destroy_resource_op(handle)
 247                 for handle in all_handles
 248             ])
 249     }
 250 
 251     if report_feature_importances:
 252       ops_at_end['1: feature_importances'] = (
 253           graph_builder.feature_importances())
 254 
 255     training_hooks = [TensorForestRunOpAtEndHook(ops_at_end)]
 256 
 257     if output_type == ModelBuilderOutputType.MODEL_FN_OPS:
 258       model_ops = model_head.create_model_fn_ops(
 259           features=features,
 260           labels=labels,
 261           mode=mode,
 262           train_op_fn=_train_fn,
 263           logits=logits,
 264           scope=head_scope)
 265 
 266       if early_stopping_rounds:
 267         training_hooks.append(
 268             TensorForestLossHook(
 269                 early_stopping_rounds,
 270                 early_stopping_loss_threshold=early_stopping_loss_threshold,
 271                 loss_op=model_ops.loss))
 272 
 273       model_ops.training_hooks.extend(training_hooks)
 274 
 275       if keys is not None:
 276         model_ops.predictions[keys_name] = keys
 277 
 278       if params.inference_tree_paths:
 279         model_ops.predictions[TREE_PATHS_PREDICTION_KEY] = tree_paths
 280 
 281       model_ops.predictions[VARIANCE_PREDICTION_KEY] = regression_variance
 282 
 283       if include_all_in_serving:
 284         # In order to serve the variance we need to add the prediction dict
 285         # to output_alternatives dict.
 286         if not model_ops.output_alternatives:
 287           model_ops.output_alternatives = {}
 288         model_ops.output_alternatives[ALL_SERVING_KEY] = (
 289             constants.ProblemType.UNSPECIFIED, model_ops.predictions)
 290 
 291       return model_ops
 292 
 293     else:
 294       # Estimator spec
 295       estimator_spec = model_head.create_estimator_spec(
 296           features=features,
 297           mode=mode,
 298           labels=labels,
 299           train_op_fn=_train_fn,
 300           logits=logits)
 301 
 302       if early_stopping_rounds:
 303         training_hooks.append(
 304             TensorForestLossHook(
 305                 early_stopping_rounds,
 306                 early_stopping_loss_threshold=early_stopping_loss_threshold,
 307                 loss_op=estimator_spec.loss))
 308 
 309       estimator_spec = estimator_spec._replace(
 310           training_hooks=training_hooks + list(estimator_spec.training_hooks))
 311       if keys is not None:
 312         estimator_spec.predictions[keys_name] = keys
 313       if params.inference_tree_paths:
 314         estimator_spec.predictions[TREE_PATHS_PREDICTION_KEY] = tree_paths
 315       estimator_spec.predictions[VARIANCE_PREDICTION_KEY] = regression_variance
 316 
 317       if include_all_in_serving:
 318         outputs = estimator_spec.export_outputs
 319         if not outputs:
 320           outputs = {}
 321         outputs = {ALL_SERVING_KEY: PredictOutput(estimator_spec.predictions)}
 322         print(estimator_spec.export_outputs)
 323         # In order to serve the variance we need to add the prediction dict
 324         # to output_alternatives dict.
 325         estimator_spec = estimator_spec._replace(export_outputs=outputs)
 326 
 327       return estimator_spec
 328 
 329   return _model_fn

 332 class TensorForestEstimator(estimator.Estimator):
 333   """An estimator that can train and evaluate a random forest.
 334 
 335   Example:
 336 
 337   ```python
 338   params = tf.contrib.tensor_forest.python.tensor_forest.ForestHParams(
 339       num_classes=2, num_features=40, num_trees=10, max_nodes=1000)
 340 
 341   # Estimator using the default graph builder.
 342   estimator = TensorForestEstimator(params, model_dir=model_dir)
 343 
 344   # Or estimator using TrainingLossForest as the graph builder.
 345   estimator = TensorForestEstimator(
 346       params, graph_builder_class=tensor_forest.TrainingLossForest,
 347       model_dir=model_dir)
 348 
 349   # Input builders
 350   def input_fn_train: # returns x, y
 351     ...
 352   def input_fn_eval: # returns x, y
 353     ...
 354   estimator.fit(input_fn=input_fn_train)
 355   estimator.evaluate(input_fn=input_fn_eval)
 356 
 357   # Predict returns an iterable of dicts.
 358   results = list(estimator.predict(x=x))
 359   prob0 = results[0][eval_metrics.INFERENCE_PROB_NAME]
 360   prediction0 = results[0][eval_metrics.INFERENCE_PRED_NAME]
 361   ```
 362   """
 363 
 364   def __init__(self,
 365                params,
 366                device_assigner=None,
 367                model_dir=None,
 368                feature_columns=None,
 369                graph_builder_class=tensor_forest.RandomForestGraphs,
 370                config=None,
 371                weight_column=None,
 372                keys_column=None,
 373                feature_engineering_fn=None,
 374                early_stopping_rounds=100,
 375                early_stopping_loss_threshold=0.001,
 376                num_trainers=1,
 377                trainer_id=0,
 378                report_feature_importances=False,
 379                local_eval=False,
 380                version=None,
 381                head=None,
 382                include_all_in_serving=False):
 383     """Initializes a TensorForestEstimator instance.
 384 
 385     Args:
 386       params: ForestHParams object that holds random forest hyperparameters.
 387         These parameters will be passed into `model_fn`.
 388       device_assigner: An `object` instance that controls how trees get
 389         assigned to devices. If `None`, will use
 390         `tensor_forest.RandomForestDeviceAssigner`.
 391       model_dir: Directory to save model parameters, graph, etc. To continue
 392         training a previously saved model, load checkpoints saved to this
 393         directory into an estimator.
 394       feature_columns: An iterable containing all the feature columns used by
 395         the model. All items in the set should be instances of classes derived
 396         from `_FeatureColumn`.
 397       graph_builder_class: An `object` instance that defines how TF graphs for
 398         random forest training and inference are built. By default will use
 399         `tensor_forest.RandomForestGraphs`. Can be overridden by version
 400         kwarg.
 401       config: `RunConfig` object to configure the runtime settings.
 402       weight_column: A string defining feature column name representing
 403         weights. Will be multiplied by the loss of the example. Used to
 404         downweight or boost examples during training.
 405       keys_column: A string naming one of the features to strip out and
 406         pass through into the inference/eval results dict.  Useful for
 407         associating specific examples with their prediction.
 408       feature_engineering_fn: Feature engineering function. Takes features and
 409         labels which are the output of `input_fn` and returns features and
 410         labels which will be fed into the model.
 411       early_stopping_rounds: Allows training to terminate early if the forest is
 412         no longer growing. 100 by default.  Set to a Falsy value to disable
 413         the default training hook.
 414       early_stopping_loss_threshold: Percentage (as fraction) that loss must
 415         improve by within early_stopping_rounds steps, otherwise training will
 416         terminate.
 417       num_trainers: Number of training jobs, which will partition trees
 418         among them.
 419       trainer_id: Which trainer this instance is.
 420       report_feature_importances: If True, print out feature importances
 421         during evaluation.
 422       local_eval: If True, don't use a device assigner for eval. This is to
 423         support some common setups where eval is done on a single machine, even
 424         though training might be distributed.
 425       version: Unused.
 426       head: A heads_lib.Head object that calculates losses and such. If None,
 427         one will be automatically created based on params.
 428       include_all_in_serving: if True, allow preparation of the complete
 429         prediction dict including the variance to be exported for serving with
 430         the Servo lib; and it also requires calling export_savedmodel with
 431         default_output_alternative_key=ALL_SERVING_KEY, i.e.
 432         estimator.export_savedmodel(export_dir_base=your_export_dir,
 433           serving_input_fn=your_export_input_fn,
 434           default_output_alternative_key=ALL_SERVING_KEY)
 435         if False, resort to default behavior, i.e. export scores and
 436           probabilities but no variances. In this case
 437           default_output_alternative_key should be None while calling
 438           export_savedmodel().
 439         Note, that due to backward compatibility we cannot always set
 440         include_all_in_serving to True because in this case calling
 441         export_saved_model() without
 442         default_output_alternative_key=ALL_SERVING_KEY (legacy behavior) the
 443         saved_model_export_utils.get_output_alternatives() would raise
 444         ValueError.
 445 
 446     Returns:
 447       A `TensorForestEstimator` instance.
 448     """
 449     # Override default number of trainers if config is provided.
 450     if num_trainers == 1 and config is not None:
 451       num_trainers = max(1, config.num_worker_replicas)
 452 
 453     super(TensorForestEstimator, self).__init__(
 454         model_fn=get_model_fn(
 455             params.fill(),
 456             graph_builder_class,
 457             device_assigner,
 458             feature_columns=feature_columns,
 459             model_head=head,
 460             weights_name=weight_column,
 461             keys_name=keys_column,
 462             early_stopping_rounds=early_stopping_rounds,
 463             early_stopping_loss_threshold=early_stopping_loss_threshold,
 464             num_trainers=num_trainers,
 465             trainer_id=trainer_id,
 466             report_feature_importances=report_feature_importances,
 467             local_eval=local_eval,
 468             include_all_in_serving=include_all_in_serving,
 469         ),
 470         model_dir=model_dir,
 471         config=config,
 472         feature_engineering_fn=feature_engineering_fn)
 473

 475 def get_combined_model_fn(model_fns):
 476   """Get a combined model function given a list of other model fns.
 477 
 478   The model function returned will call the individual model functions and
 479   combine them appropriately.  For:
 480 
 481   training ops: tf.group them.
 482   loss: average them.
 483   predictions: concat probabilities such that predictions[*][0-C1] are the
 484     probabilities for output 1 (where C1 is the number of classes in output 1),
 485     predictions[*][C1-(C1+C2)] are the probabilities for output 2 (where C2
 486     is the number of classes in output 2), etc.  Also stack predictions such
 487     that predictions[i][j] is the class prediction for example i and output j.
 488 
 489   This assumes that labels are 2-dimensional, with labels[i][j] being the
 490   label for example i and output j, where forest j is trained using only
 491   output j.
 492 
 493   Args:
 494     model_fns: A list of model functions obtained from get_model_fn.
 495 
 496   Returns:
 497     A ModelFnOps instance.
 498   """
 499   def _model_fn(features, labels, mode):
 500     """Function that returns predictions, training loss, and training op."""
 501     model_fn_ops = []
 502     for i in range(len(model_fns)):
 503       with variable_scope.variable_scope('label_{0}'.format(i)):
 504         sliced_labels = array_ops.slice(labels, [0, i], [-1, 1])
 505         model_fn_ops.append(
 506             model_fns[i](features, sliced_labels, mode))
 507     training_hooks = []
 508     for mops in model_fn_ops:
 509       training_hooks += mops.training_hooks
 510     predictions = {}
 511     if (mode == model_fn_lib.ModeKeys.EVAL or
 512         mode == model_fn_lib.ModeKeys.INFER):
 513       # Flatten the probabilities into one dimension.
 514       predictions[eval_metrics.INFERENCE_PROB_NAME] = array_ops.concat(
 515           [mops.predictions[eval_metrics.INFERENCE_PROB_NAME]
 516            for mops in model_fn_ops], axis=1)
 517       predictions[eval_metrics.INFERENCE_PRED_NAME] = array_ops.stack(
 518           [mops.predictions[eval_metrics.INFERENCE_PRED_NAME]
 519            for mops in model_fn_ops], axis=1)
 520     loss = None
 521     if (mode == model_fn_lib.ModeKeys.EVAL or
 522         mode == model_fn_lib.ModeKeys.TRAIN):
 523       loss = math_ops.reduce_sum(
 524           array_ops.stack(
 525               [mops.loss for mops in model_fn_ops])) / len(model_fn_ops)
 526 
 527     train_op = None
 528     if mode == model_fn_lib.ModeKeys.TRAIN:
 529       train_op = control_flow_ops.group(
 530           *[mops.train_op for mops in model_fn_ops])
 531     return model_fn_lib.ModelFnOps(
 532         mode=mode,
 533         predictions=predictions,
 534         loss=loss,
 535         train_op=train_op,
 536         training_hooks=training_hooks,
 537         scaffold=None,
 538         output_alternatives=None)
 539 
 540   return _model_fn

 543 class MultiForestMultiHeadEstimator(estimator.Estimator):
 544   """An estimator that can train a forest for a multi-headed problems.
 545 
 546   This class essentially trains separate forests (each with their own
 547   ForestHParams) for each output.
 548 
 549   For multi-headed regression, a single-headed TensorForestEstimator can
 550   be used to train a single model that predicts all outputs.  This class can
 551   be used to train separate forests for each output.
 552   """
 553 
 554   def __init__(self,
 555                params_list,
 556                device_assigner=None,
 557                model_dir=None,
 558                feature_columns=None,
 559                graph_builder_class=tensor_forest.RandomForestGraphs,
 560                config=None,
 561                weight_column=None,
 562                keys_column=None,
 563                feature_engineering_fn=None,
 564                early_stopping_rounds=100,
 565                num_trainers=1,
 566                trainer_id=0,
 567                report_feature_importances=False,
 568                local_eval=False):
 569     """See TensorForestEstimator.__init__."""
 570     model_fns = []
 571     # Override default number of trainers if config is provided.
 572     if num_trainers == 1 and config is not None:
 573       num_trainers = max(1, config.num_worker_replicas)
 574 
 575     for i in range(len(params_list)):
 576       params = params_list[i].fill()
 577       model_fns.append(
 578           get_model_fn(
 579               params,
 580               graph_builder_class,
 581               device_assigner,
 582               model_head=_get_default_head(
 583                   params,
 584                   weight_column,
 585                   name='head{0}'.format(i),
 586                   output_type=ModelBuilderOutputType.MODEL_FN_OPS),
 587               weights_name=weight_column,
 588               keys_name=keys_column,
 589               early_stopping_rounds=early_stopping_rounds,
 590               num_trainers=num_trainers,
 591               trainer_id=trainer_id,
 592               report_feature_importances=report_feature_importances,
 593               local_eval=local_eval,

 604   """A CORE estimator that can train and evaluate a random forest.
 605 
 606   Example:
 607 
 608   ```python

 610       num_classes=2, num_features=40, num_trees=10, max_nodes=1000)
 611

 613   estimator = CoreTensorForestEstimator(params, model_dir=model_dir)
 614

 617       params, graph_builder_class=tensor_forest.TrainingLossForest,
 618       model_dir=model_dir)
 619 
 620   # Input builders
 621   def input_fn_train: # returns x, y
 622     ...
 623   def input_fn_eval: # returns x, y
 624     ...

 626   estimator.evaluate(input_fn=input_fn_eval)
 627 
 628   # Predict returns an iterable of dicts.
 629   results = list(estimator.predict(x=x))
 630   prob0 = results[0][eval_metrics.INFERENCE_PROB_NAME]
 631   prediction0 = results[0][eval_metrics.INFERENCE_PRED_NAME]
 632   ```
 633   """

 700         prediction dict including the variance to be exported for serving with

 702         default_output_alternative_key=ALL_SERVING_KEY, i.e.
 703         estimator.export_savedmodel(export_dir_base=your_export_dir,
 704           serving_input_fn=your_export_input_fn,
 705           default_output_alternative_key=ALL_SERVING_KEY)

 707           probabilities but no variances. In this case

 709           export_savedmodel().
