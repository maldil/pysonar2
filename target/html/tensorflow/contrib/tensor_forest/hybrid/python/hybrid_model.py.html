  15 """Defines the model abstraction for hybrid models."""
  16 from __future__ import absolute_import
  17 from __future__ import division
  18 from __future__ import print_function
  19 
  20 import collections
  21 
  22 from tensorflow.contrib import layers
  23 from tensorflow.contrib.framework.python.ops import variables as framework_variables
  24 
  25 from tensorflow.python.ops import array_ops
  26 from tensorflow.python.ops import math_ops
  27 from tensorflow.python.ops import nn_ops
  28 from tensorflow.python.ops import variables
  29 
  30 from tensorflow.python.training import adagrad
  31 
  32 
  33 class HybridModel(object):
  34   """Defines a hybrid model.
  35 
  36   Models chain together the results of inference layers and provide training
  37   capabilities.
  38   """
  39 
  40   # pylint: disable=unused-argument
  41   def __init__(self,
  42                params,
  43                device_assigner=None,
  44                optimizer_class=adagrad.AdagradOptimizer,
  45                **kwargs):
  46 
  47     self.device_assigner = (
  48         device_assigner or framework_variables.VariableDeviceChooser())
  49 
  50     self.params = params
  51 
  52     self.optimizer = optimizer_class(self.params.learning_rate)
  53 
  54     self.is_regression = params.regression
  55 
  56     self.regularizer = None
  57     if params.regularization == "l1":
  58       self.regularizer = layers.l1_regularizer(
  59           self.params.regularization_strength)
  60     elif params.regularization == "l2":
  61       self.regularizer = layers.l2_regularizer(
  62           self.params.regularization_strength)
  63 
  64   def _do_layer_inference(self, layer, data):
  65 
  66     # If this is a collection of layers, return the mean of their inference
  67     # results.
  68     if isinstance(layer, collections.Iterable):
  69       return math_ops.reduce_mean(
  70           array_ops.stack([l.inference_graph(data) for l in layer]), 0)
  71     # If this is a single layer, return its inference result.
  72     else:
  73       return layer.inference_graph(data)
  74 
  75   def _base_inference(self, data, data_spec=None):
  76     """Returns an op that performs inference without a softmax."""
  77     inference_result = self._do_layer_inference(self.layers[0], data)
  78 
  79     for layer in self.layers[1:]:
  80       inference_result = self._do_layer_inference(layer, inference_result)
  81 
  82     output_size = 1 if self.is_regression else self.params.num_classes
  83     output = layers.fully_connected(
  84         inference_result, output_size, activation_fn=array_ops.identity)
  85 
  86     return output
  87 
  88   def inference_graph(self, data, data_spec=None):
  89     """Returns the op that performs inference on a batch of data."""
  90 
  91     return nn_ops.softmax(self._base_inference(data, data_spec=data_spec))
  92 
  93   def training_inference_graph(self, data, data_spec=None):
  94     """Returns an inference-without-softmax op for training purposes."""
  95 
  96     return self._base_inference(data, data_spec=data_spec)
  97 
  98   def predict_proba(self, data, data_spec=None):
  99     inference_result = self.inference_graph(data, data_spec=data_spec)
 100 
 101     probabilities = nn_ops.softmax(inference_result, name="probabilities")
 102 
 103     return probabilities
 104 
 105   def training_graph(self, data, labels, data_spec=None, epoch=None):
 106     """Returns the op that trains the hybrid model."""
 107     return self.optimizer.minimize(self.training_loss(data, labels))
 108 
 109   def loss(self, data, labels):
 110     """The loss to minimize while training."""
 111 
 112     if self.is_regression:
 113       diff = self.training_inference_graph(data) - math_ops.to_float(labels)
 114       mean_squared_error = math_ops.reduce_mean(diff * diff)
 115       root_mean_squared_error = math_ops.sqrt(mean_squared_error, name="loss")
 116       loss = root_mean_squared_error
 117     else:
 118       loss = math_ops.reduce_mean(
 119           nn_ops.sparse_softmax_cross_entropy_with_logits(
 120               labels=array_ops.squeeze(math_ops.to_int32(labels)),
 121               logits=self.training_inference_graph(data)),
 122           name="loss")
 123     if self.regularizer:
 124       loss += layers.apply_regularization(self.regularizer,
 125                                           variables.trainable_variables())
 126     return loss
 127 
 128   def training_loss(self, data, labels):
 129     return self.loss(data, labels)
 130 
 131   def validation_loss(self, data, labels):
 132     return self.loss(data, labels)
