  46 class FertileStatsVariableSavable(saver.BaseSaverBuilder.SaveableObject):
  47   """SaveableObject implementation for FertileStatsVariable."""
  48 
  49   def __init__(self, params, stats_handle, create_op, name):
  50     """Creates a FertileStatsVariableSavable object.
  51 
  52     Args:
  53       params: A TensorForestParams object.
  54       stats_handle: handle to the tree variable.
  55       create_op: the op to initialize the variable.
  56       name: the name to save the tree variable under.
  57     """
  58     self.params = params
  59     tensor = gen_stats_ops.fertile_stats_serialize(
  60         stats_handle, params=params.serialized_params_proto)
  61     # slice_spec is useful for saving a slice from a variable.
  62     # It's not meaningful the tree variable. So we just pass an empty value.
  63     slice_spec = ""
  64     specs = [saver.BaseSaverBuilder.SaveSpec(tensor, slice_spec, name),]
  65     super(FertileStatsVariableSavable,
  66           self).__init__(stats_handle, specs, name)
  67     self._stats_handle = stats_handle
  68     self._create_op = create_op
  69 
  70   def restore(self, restored_tensors, unused_restored_shapes):
  71     """Restores the associated tree from 'restored_tensors'.
  72 
  73     Args:
  74       restored_tensors: the tensors that were loaded from a checkpoint.
  75       unused_restored_shapes: the shapes this object should conform to after
  76         restore. Not meaningful for trees.
  77 
  78     Returns:
  79       The operation that restores the state of the tree variable.
  80     """
  81     with ops.control_dependencies([self._create_op]):
  82       return gen_stats_ops.fertile_stats_deserialize(
  83           self._stats_handle, restored_tensors[0],
  84           params=self.params.serialized_params_proto)

  87 def fertile_stats_variable(params, stats_config, name,
  88                            container=None):
  89   r"""Creates a stats object and returns a handle to it.
  90 
  91   Args:
  92     params: A TensorForestParams object.
  93     stats_config: A `Tensor` of type `string`. Serialized proto of the stats.
  94     name: A name for the variable.
  95     container: An optional `string`. Defaults to `""`.
  96 
  97   Returns:
  98     A `Tensor` of type mutable `string`. The handle to the stats.
  99   """
 100   with ops.name_scope(name, "FertileStatsVariable") as name:
 101     resource_handle = gen_stats_ops.fertile_stats_resource_handle_op(
 102         container, shared_name=name, name=name)
 103 
 104     create_op = gen_stats_ops.create_fertile_stats_variable(
 105         resource_handle, stats_config,
 106         params=params.serialized_params_proto)
 107     is_initialized_op = gen_stats_ops.fertile_stats_is_initialized_op(
 108         resource_handle)
 109     # Adds the variable to the savable list.
 110     saveable = FertileStatsVariableSavable(params, resource_handle, create_op,
 111                                            resource_handle.name)
 112     ops.add_to_collection(ops.GraphKeys.SAVEABLE_OBJECTS, saveable)
 113     resources.register_resource(resource_handle, create_op, is_initialized_op)
 114     return resource_handle
