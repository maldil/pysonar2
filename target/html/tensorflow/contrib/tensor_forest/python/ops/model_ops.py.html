  49 class TreeVariableSavable(saver.BaseSaverBuilder.SaveableObject):
  50   """SaveableObject implementation for TreeVariable."""
  51 
  52   def __init__(self, params, tree_handle, stats_handle, create_op, name):
  53     """Creates a TreeVariableSavable object.
  54 
  55     Args:
  56       params: A TensorForestParams object.
  57       tree_handle: handle to the tree variable.
  58       stats_handle: handle to the stats variable.
  59       create_op: the op to initialize the variable.
  60       name: the name to save the tree variable under.
  61     """
  62     self.params = params
  63     tensor = gen_model_ops.tree_serialize(tree_handle)
  64     # slice_spec is useful for saving a slice from a variable.
  65     # It's not meaningful the tree variable. So we just pass an empty value.
  66     slice_spec = ""
  67     specs = [saver.BaseSaverBuilder.SaveSpec(tensor, slice_spec, name),]
  68     super(TreeVariableSavable,
  69           self).__init__(tree_handle, specs, name)
  70     self._tree_handle = tree_handle
  71     self._create_op = create_op
  72 
  73   def restore(self, restored_tensors, unused_restored_shapes):
  74     """Restores the associated tree from 'restored_tensors'.
  75 
  76     Args:
  77       restored_tensors: the tensors that were loaded from a checkpoint.
  78       unused_restored_shapes: the shapes this object should conform to after
  79         restore. Not meaningful for trees.
  80 
  81     Returns:
  82       The operation that restores the state of the tree variable.
  83     """
  84     with ops.control_dependencies([self._create_op]):
  85       return gen_model_ops.tree_deserialize(
  86           self._tree_handle,
  87           restored_tensors[0],
  88           params=self.params.serialized_params_proto)

  91 def tree_variable(params, tree_config, stats_handle, name, container=None):
  92   r"""Creates a tree model and returns a handle to it.
  93 
  94   Args:
  95     params: A TensorForestParams object.
  96     tree_config: A `Tensor` of type `string`. Serialized proto of the tree.
  97     stats_handle: Resource handle to the stats object.
  98     name: A name for the variable.
  99     container: An optional `string`. Defaults to `""`.
 100 
 101   Returns:
 102     A `Tensor` of type mutable `string`. The handle to the tree.
 103   """
 104   with ops.name_scope(name, "TreeVariable") as name:
 105     resource_handle = gen_model_ops.decision_tree_resource_handle_op(
 106         container, shared_name=name, name=name)
 107 
 108     create_op = gen_model_ops.create_tree_variable(
 109         resource_handle,
 110         tree_config,
 111         params=params.serialized_params_proto)
 112     is_initialized_op = gen_model_ops.tree_is_initialized_op(resource_handle)
 113     # Adds the variable to the savable list.
 114     saveable = TreeVariableSavable(params, resource_handle, stats_handle,
 115                                    create_op,
 116                                    resource_handle.name)
 117     ops.add_to_collection(ops.GraphKeys.SAVEABLE_OBJECTS, saveable)
 118     resources.register_resource(resource_handle, create_op, is_initialized_op)
 119     return resource_handle
