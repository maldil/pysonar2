  76 def parse_number_or_string_to_proto(proto, param):
  77   if isinstance(param, numbers.Number):
  78     proto.constant_value = param
  79   else:  # assume it's a string
  80     if param.isdigit():
  81       proto.constant_value = int(param)
  82     else:
  83       text_format.Merge(param, proto)

  86 def build_params_proto(params):
  87   """Build a TensorForestParams proto out of the V4ForestHParams object."""
  88   proto = _params_proto.TensorForestParams()
  89   proto.num_trees = params.num_trees
  90   proto.max_nodes = params.max_nodes
  91   proto.is_regression = params.regression
  92   proto.num_outputs = params.num_classes
  93   proto.num_features = params.num_features
  94 
  95   proto.leaf_type = params.leaf_model_type
  96   proto.stats_type = params.stats_model_type
  97   proto.collection_type = _params_proto.COLLECTION_BASIC
  98   proto.pruning_type.type = params.pruning_type
  99   proto.finish_type.type = params.finish_type
 100 
 101   proto.inequality_test_type = params.split_type
 102 
 103   proto.drop_final_class = False
 104   proto.collate_examples = params.collate_examples
 105   proto.checkpoint_stats = params.checkpoint_stats
 106   proto.use_running_stats_method = params.use_running_stats_method
 107   proto.initialize_average_splits = params.initialize_average_splits
 108   proto.inference_tree_paths = params.inference_tree_paths
 109 
 110   parse_number_or_string_to_proto(proto.pruning_type.prune_every_samples,
 111                                   params.prune_every_samples)
 112   parse_number_or_string_to_proto(proto.finish_type.check_every_steps,
 113                                   params.early_finish_check_every_samples)
 114   parse_number_or_string_to_proto(proto.split_after_samples,
 115                                   params.split_after_samples)
 116   parse_number_or_string_to_proto(proto.num_splits_to_consider,
 117                                   params.num_splits_to_consider)
 118 
 119   proto.dominate_fraction.constant_value = params.dominate_fraction
 120 
 121   if params.param_file:
 122     with open(params.param_file) as f:
 123       text_format.Merge(f.read(), proto)
 124 
 125   return proto

 128 # A convenience class for holding random forest hyperparameters.
 129 #
 130 # To just get some good default parameters, use:
 131 #   hparams = ForestHParams(num_classes=2, num_features=40).fill()
 132 #
 133 # Note that num_classes can not be inferred and so must always be specified.
 134 # Also, either num_splits_to_consider or num_features should be set.
 135 #
 136 # To override specific values, pass them to the constructor:
 137 #   hparams = ForestHParams(num_classes=5, num_trees=10, num_features=5).fill()
 138 #
 139 # TODO(thomaswc): Inherit from tf.HParams when that is publicly available.
 140 class ForestHParams(object):
 141   """A base class for holding hyperparameters and calculating good defaults."""
 142 
 143   def __init__(
 144       self,
 145       num_trees=100,
 146       max_nodes=10000,
 147       bagging_fraction=1.0,
 148       num_splits_to_consider=0,
 149       feature_bagging_fraction=1.0,
 150       max_fertile_nodes=0,  # deprecated, unused.
 151       split_after_samples=250,
 152       valid_leaf_threshold=1,
 153       dominate_method='bootstrap',
 154       dominate_fraction=0.99,
 155       model_name='all_dense',
 156       split_finish_name='basic',
 157       split_pruning_name='none',
 158       prune_every_samples=0,
 159       early_finish_check_every_samples=0,
 160       collate_examples=False,
 161       checkpoint_stats=False,
 162       use_running_stats_method=False,
 163       initialize_average_splits=False,
 164       inference_tree_paths=False,
 165       param_file=None,
 166       split_name='less_or_equal',
 167       **kwargs):
 168     self.num_trees = num_trees
 169     self.max_nodes = max_nodes
 170     self.bagging_fraction = bagging_fraction
 171     self.feature_bagging_fraction = feature_bagging_fraction
 172     self.num_splits_to_consider = num_splits_to_consider
 173     self.max_fertile_nodes = max_fertile_nodes
 174     self.split_after_samples = split_after_samples
 175     self.valid_leaf_threshold = valid_leaf_threshold
 176     self.dominate_method = dominate_method
 177     self.dominate_fraction = dominate_fraction
 178     self.model_name = model_name
 179     self.split_finish_name = split_finish_name
 180     self.split_pruning_name = split_pruning_name
 181     self.collate_examples = collate_examples
 182     self.checkpoint_stats = checkpoint_stats
 183     self.use_running_stats_method = use_running_stats_method
 184     self.initialize_average_splits = initialize_average_splits
 185     self.inference_tree_paths = inference_tree_paths
 186     self.param_file = param_file
 187     self.split_name = split_name
 188     self.early_finish_check_every_samples = early_finish_check_every_samples
 189     self.prune_every_samples = prune_every_samples
 190 
 191     for name, value in kwargs.items():
 192       setattr(self, name, value)

 197   def fill(self):
 198     """Intelligently sets any non-specific parameters."""
 199     # Fail fast if num_classes or num_features isn't set.
 200     _ = getattr(self, 'num_classes')
 201     _ = getattr(self, 'num_features')
 202 
 203     self.bagged_num_features = int(self.feature_bagging_fraction *
 204                                    self.num_features)
 205 
 206     self.bagged_features = None
 207     if self.feature_bagging_fraction < 1.0:
 208       self.bagged_features = [random.sample(
 209           range(self.num_features),
 210           self.bagged_num_features) for _ in range(self.num_trees)]
 211 
 212     self.regression = getattr(self, 'regression', False)
 213 
 214     # Num_outputs is the actual number of outputs (a single prediction for
 215     # classification, a N-dimensional point for regression).
 216     self.num_outputs = self.num_classes if self.regression else 1
 217 
 218     # Add an extra column to classes for storing counts, which is needed for
 219     # regression and avoids having to recompute sums for classification.
 220     self.num_output_columns = self.num_classes + 1
 221 
 222     # Our experiments have found that num_splits_to_consider = num_features
 223     # gives good accuracy.
 224     self.num_splits_to_consider = self.num_splits_to_consider or min(
 225         max(10, math.floor(math.sqrt(self.num_features))), 1000)
 226 
 227     # If base_random_seed is 0, the current time will be used to seed the
 228     # random number generators for each tree.  If non-zero, the i-th tree
 229     # will be seeded with base_random_seed + i.
 230     self.base_random_seed = getattr(self, 'base_random_seed', 0)
 231 
 232     # How to store leaf models.
 233     self.leaf_model_type = (
 234         REGRESSION_MODEL_TYPE[0] if self.regression else
 235         CLASSIFICATION_LEAF_MODEL_TYPES[self.model_name][0])
 236 
 237     # How to store stats objects.
 238     self.stats_model_type = (
 239         REGRESSION_MODEL_TYPE[1] if self.regression else
 240         CLASSIFICATION_LEAF_MODEL_TYPES[self.model_name][1])
 241 
 242     self.finish_type = (
 243         _params_proto.SPLIT_FINISH_BASIC if self.regression else
 244         FINISH_TYPES[self.split_finish_name])
 245 
 246     self.pruning_type = PRUNING_TYPES[self.split_pruning_name]
 247 
 248     if self.pruning_type == _params_proto.SPLIT_PRUNE_NONE:
 249       self.prune_every_samples = 0
 250     else:
 251       if (not self.prune_every_samples and
 252           not (isinstance(numbers.Number) or
 253                self.split_after_samples.isdigit())):
 254         logging.error(
 255             'Must specify prune_every_samples if using a depth-dependent '
 256             'split_after_samples')
 257       # Pruning half-way through split_after_samples seems like a decent
 258       # default, making it easy to select the number being pruned with
 259       # pruning_type while not paying the cost of pruning too often.  Note that
 260       # this only holds if not using a depth-dependent split_after_samples.
 261       self.prune_every_samples = (self.prune_every_samples or
 262                                   int(self.split_after_samples) / 2)
 263 
 264     if self.finish_type == _params_proto.SPLIT_FINISH_BASIC:
 265       self.early_finish_check_every_samples = 0
 266     else:
 267       if (not self.early_finish_check_every_samples and
 268           not (isinstance(numbers.Number) or
 269                self.split_after_samples.isdigit())):
 270         logging.error(
 271             'Must specify prune_every_samples if using a depth-dependent '
 272             'split_after_samples')
 273       # Checking for early finish every quarter through split_after_samples
 274       # seems like a decent default. We don't want to incur the checking cost
 275       # too often, but (at least for hoeffding) it's lower than the cost of
 276       # pruning so we can do it a little more frequently.
 277       self.early_finish_check_every_samples = (
 278           self.early_finish_check_every_samples or
 279           int(self.split_after_samples) / 4)
 280 
 281     self.split_type = SPLIT_TYPES[self.split_name]
 282 
 283     return self

 286 def get_epoch_variable():
 287   """Returns the epoch variable, or [0] if not defined."""
 288   # Grab epoch variable defined in
 289   # //third_party/tensorflow/python/training/input.py::limit_epochs
 290   for v in tf_variables.local_variables():
 291     if 'limit_epochs/epoch' in v.op.name:
 292       return array_ops.reshape(v, [1])
 293   # TODO(thomaswc): Access epoch from the data feeder.
 294   return [0]

 297 # A simple container to hold the training variables for a single tree.
 298 class TreeVariables(object):
 299   """Stores tf.Variables for training a single random tree.
 300 
 301   Uses tf.get_variable to get tree-specific names so that this can be used
 302   with a tf.learn-style implementation (one that trains a model, saves it,
 303   then relies on restoring that model to evaluate).
 304   """
 305 
 306   def __init__(self, params, tree_num, training, tree_config='', tree_stat=''):
 307     if (not hasattr(params, 'params_proto') or
 308         not isinstance(params.params_proto,
 309                        _params_proto.TensorForestParams)):
 310       params.params_proto = build_params_proto(params)
 311 
 312     params.serialized_params_proto = params.params_proto.SerializeToString()
 313     self.stats = None
 314     if training:
 315       # TODO(gilberth): Manually shard this to be able to fit it on
 316       # multiple machines.
 317       self.stats = stats_ops.fertile_stats_variable(
 318           params, tree_stat, self.get_tree_name('stats', tree_num))
 319     self.tree = model_ops.tree_variable(
 320         params, tree_config, self.stats, self.get_tree_name('tree', tree_num))
 321 
 322   def get_tree_name(self, name, num):
 323     return '{0}-{1}'.format(name, num)

 326 class ForestVariables(object):
 327   """A container for a forests training data, consisting of multiple trees.
 328 
 329   Instantiates a TreeVariables object for each tree. We override the
 330   __getitem__ and __setitem__ function so that usage looks like this:
 331 
 332     forest_variables = ForestVariables(params)
 333 
 334     ... forest_variables.tree ...
 335   """
 336 
 337   def __init__(self, params, device_assigner, training=True,
 338                tree_variables_class=TreeVariables,
 339                tree_configs=None, tree_stats=None):
 340     self.variables = []
 341     # Set up some scalar variables to run through the device assigner, then
 342     # we can use those to colocate everything related to a tree.
 343     self.device_dummies = []
 344     with ops.device(device_assigner):
 345       for i in range(params.num_trees):
 346         self.device_dummies.append(variable_scope.get_variable(
 347             name='device_dummy_%d' % i, shape=0))
 348 
 349     for i in range(params.num_trees):
 350       with ops.device(self.device_dummies[i].device):
 351         kwargs = {}
 352         if tree_configs is not None:
 353           kwargs.update(dict(tree_config=tree_configs[i]))
 354         if tree_stats is not None:
 355           kwargs.update(dict(tree_stat=tree_stats[i]))
 356         self.variables.append(tree_variables_class(
 357             params, i, training, **kwargs))
 358 
 359   def __setitem__(self, t, val):
 360     self.variables[t] = val
 361 
 362   def __getitem__(self, t):
 363     return self.variables[t]

 366 class RandomForestGraphs(object):
 367   """Builds TF graphs for random forest training and inference."""
 368 
 369   def __init__(self,
 370                params,
 371                tree_configs=None,
 372                tree_stats=None,
 373                device_assigner=None,
 374                variables=None,
 375                tree_variables_class=TreeVariables,
 376                tree_graphs=None,
 377                training=True):
 378     self.params = params
 379     self.device_assigner = (
 380         device_assigner or framework_variables.VariableDeviceChooser())
 381     logging.info('Constructing forest with params = ')
 382     logging.info(self.params.__dict__)
 383     self.variables = variables or ForestVariables(
 384         self.params, device_assigner=self.device_assigner, training=training,
 385         tree_variables_class=tree_variables_class,
 386         tree_configs=tree_configs, tree_stats=tree_stats)
 387     tree_graph_class = tree_graphs or RandomTreeGraphs
 388     self.trees = [
 389         tree_graph_class(self.variables[i], self.params, i)
 390         for i in range(self.params.num_trees)
 391     ]
 392 
 393   def _bag_features(self, tree_num, input_data):
 394     split_data = array_ops.split(
 395         value=input_data, num_or_size_splits=self.params.num_features, axis=1)
 396     return array_ops.concat(
 397         [split_data[ind] for ind in self.params.bagged_features[tree_num]], 1)
 398 
 399   def get_all_resource_handles(self):
 400     return ([self.variables[i].tree for i in range(len(self.trees))] +
 401             [self.variables[i].stats for i in range(len(self.trees))])
 402 
 403   def training_graph(self,
 404                      input_data,
 405                      input_labels,
 406                      num_trainers=1,
 407                      trainer_id=0,
 408                      **tree_kwargs):
 409     """Constructs a TF graph for training a random forest.
 410 
 411     Args:
 412       input_data: A tensor or dict of string->Tensor for input data.
 413       input_labels: A tensor or placeholder for labels associated with
 414         input_data.
 415       num_trainers: Number of parallel trainers to split trees among.
 416       trainer_id: Which trainer this instance is.
 417       **tree_kwargs: Keyword arguments passed to each tree's training_graph.
 418 
 419     Returns:
 420       The last op in the random forest training graph.
 421 
 422     Raises:
 423       NotImplementedError: If trying to use bagging with sparse features.
 424     """
 425     processed_dense_features, processed_sparse_features, data_spec = (
 426         data_ops.ParseDataTensorOrDict(input_data))
 427 
 428     if input_labels is not None:
 429       labels = data_ops.ParseLabelTensorOrDict(input_labels)
 430 
 431     data_spec = data_spec or self.get_default_data_spec(input_data)
 432 
 433     tree_graphs = []
 434     trees_per_trainer = self.params.num_trees / num_trainers
 435     tree_start = int(trainer_id * trees_per_trainer)
 436     tree_end = int((trainer_id + 1) * trees_per_trainer)
 437     for i in range(tree_start, tree_end):
 438       with ops.device(self.variables.device_dummies[i].device):
 439         seed = self.params.base_random_seed
 440         if seed != 0:
 441           seed += i
 442         # If using bagging, randomly select some of the input.
 443         tree_data = processed_dense_features
 444         tree_labels = labels
 445         if self.params.bagging_fraction < 1.0:
 446           # TODO(gilberth): Support bagging for sparse features.
 447           if processed_sparse_features is not None:
 448             raise NotImplementedError(
 449                 'Bagging not supported with sparse features.')
 450           # TODO(thomaswc): This does sampling without replacement.  Consider
 451           # also allowing sampling with replacement as an option.
 452           batch_size = array_ops.strided_slice(
 453               array_ops.shape(processed_dense_features), [0], [1])
 454           r = random_ops.random_uniform(batch_size, seed=seed)
 455           mask = math_ops.less(
 456               r, array_ops.ones_like(r) * self.params.bagging_fraction)
 457           gather_indices = array_ops.squeeze(
 458               array_ops.where(mask), axis=[1])
 459           # TODO(thomaswc): Calculate out-of-bag data and labels, and store
 460           # them for use in calculating statistics later.
 461           tree_data = array_ops.gather(processed_dense_features, gather_indices)
 462           tree_labels = array_ops.gather(labels, gather_indices)
 463         if self.params.bagged_features:
 464           if processed_sparse_features is not None:
 465             raise NotImplementedError(
 466                 'Feature bagging not supported with sparse features.')
 467           tree_data = self._bag_features(i, tree_data)
 468 
 469         tree_graphs.append(self.trees[i].training_graph(
 470             tree_data,
 471             tree_labels,
 472             seed,
 473             data_spec=data_spec,
 474             sparse_features=processed_sparse_features,
 475             **tree_kwargs))
 476 
 477     return control_flow_ops.group(*tree_graphs, name='train')
 478 
 479   def inference_graph(self, input_data, **inference_args):
 480     """Constructs a TF graph for evaluating a random forest.
 481 
 482     Args:
 483       input_data: A tensor or dict of string->Tensor for the input data.
 484                   This input_data must generate the same spec as the
 485                   input_data used in training_graph:  the dict must have
 486                   the same keys, for example, and all tensors must have
 487                   the same size in their first dimension.
 488       **inference_args: Keyword arguments to pass through to each tree.
 489 
 490     Returns:
 491       A tuple of (probabilities, tree_paths, variance).
 492 
 493     Raises:
 494       NotImplementedError: If trying to use feature bagging with sparse
 495         features.
 496     """
 497     processed_dense_features, processed_sparse_features, data_spec = (
 498         data_ops.ParseDataTensorOrDict(input_data))
 499 
 500     probabilities = []
 501     paths = []
 502     for i in range(self.params.num_trees):
 503       with ops.device(self.variables.device_dummies[i].device):
 504         tree_data = processed_dense_features
 505         if self.params.bagged_features:
 506           if processed_sparse_features is not None:
 507             raise NotImplementedError(
 508                 'Feature bagging not supported with sparse features.')
 509           tree_data = self._bag_features(i, tree_data)
 510         probs, path = self.trees[i].inference_graph(
 511             tree_data,
 512             data_spec,
 513             sparse_features=processed_sparse_features,
 514             **inference_args)
 515         probabilities.append(probs)
 516         paths.append(path)
 517     with ops.device(self.variables.device_dummies[0].device):
 518       # shape of all_predict should be [batch_size, num_trees, num_outputs]
 519       all_predict = array_ops.stack(probabilities, axis=1)
 520       average_values = math_ops.div(
 521           math_ops.reduce_sum(all_predict, 1),
 522           self.params.num_trees,
 523           name='probabilities')
 524       tree_paths = array_ops.stack(paths, axis=1)
 525 
 526       expected_squares = math_ops.div(
 527           math_ops.reduce_sum(all_predict * all_predict, 1),
 528           self.params.num_trees)
 529       regression_variance = math_ops.maximum(
 530           0., expected_squares - average_values * average_values)
 531       return average_values, tree_paths, regression_variance
 532 
 533   def average_size(self):
 534     """Constructs a TF graph for evaluating the average size of a forest.
 535 
 536     Returns:
 537       The average number of nodes over the trees.
 538     """
 539     sizes = []
 540     for i in range(self.params.num_trees):
 541       with ops.device(self.variables.device_dummies[i].device):
 542         sizes.append(self.trees[i].size())
 543     return math_ops.reduce_mean(math_ops.to_float(array_ops.stack(sizes)))
 544 
 545   # pylint: disable=unused-argument
 546   def training_loss(self, features, labels, name='training_loss'):
 547     return math_ops.negative(self.average_size(), name=name)
 548 
 549   # pylint: disable=unused-argument
 550   def validation_loss(self, features, labels):
 551     return math_ops.negative(self.average_size())
 552 
 553   def average_impurity(self):
 554     """Constructs a TF graph for evaluating the leaf impurity of a forest.
 555 
 556     Returns:
 557       The last op in the graph.
 558     """
 559     impurities = []
 560     for i in range(self.params.num_trees):
 561       with ops.device(self.variables.device_dummies[i].device):
 562         impurities.append(self.trees[i].average_impurity())
 563     return math_ops.reduce_mean(array_ops.stack(impurities))
 564 
 565   def feature_importances(self):
 566     tree_counts = [self.trees[i].feature_usage_counts()
 567                    for i in range(self.params.num_trees)]
 568     total_counts = math_ops.reduce_sum(array_ops.stack(tree_counts, 0), 0)
 569     return total_counts / math_ops.reduce_sum(total_counts)

 580   def training_graph(self,
 581                      input_data,
 582                      input_labels,
 583                      random_seed,
 584                      data_spec,
 585                      sparse_features=None,
 586                      input_weights=None):
 587 
 588     """Constructs a TF graph for training a random tree.
 589 
 590     Args:
 591       input_data: A tensor or placeholder for input data.
 592       input_labels: A tensor or placeholder for labels associated with
 593         input_data.
 594       random_seed: The random number generator seed to use for this tree.  0
 595         means use the current time as the seed.
 596       data_spec: A data_ops.TensorForestDataSpec object specifying the
 597         original feature/columns of the data.
 598       sparse_features: A tf.SparseTensor for sparse input data.
 599       input_weights: A float tensor or placeholder holding per-input weights,
 600         or None if all inputs are to be weighted equally.
 601 
 602     Returns:
 603       The last op in the random tree training graph.
 604     """
 605     # TODO(gilberth): Use this.
 606     unused_epoch = math_ops.to_int32(get_epoch_variable())
 607 
 608     if input_weights is None:
 609       input_weights = []
 610 
 611     sparse_indices = []
 612     sparse_values = []
 613     sparse_shape = []
 614     if sparse_features is not None:
 615       sparse_indices = sparse_features.indices
 616       sparse_values = sparse_features.values
 617       sparse_shape = sparse_features.dense_shape
 618 
 619     if input_data is None:
 620       input_data = []
 621 
 622     leaf_ids = model_ops.traverse_tree_v4(
 623         self.variables.tree,
 624         input_data,
 625         sparse_indices,
 626         sparse_values,
 627         sparse_shape,
 628         input_spec=data_spec.SerializeToString(),
 629         params=self.params.serialized_params_proto)
 630 
 631     update_model = model_ops.update_model_v4(
 632         self.variables.tree,
 633         leaf_ids,
 634         input_labels,
 635         input_weights,
 636         params=self.params.serialized_params_proto)
 637 
 638     finished_nodes = stats_ops.process_input_v4(
 639         self.variables.tree,
 640         self.variables.stats,
 641         input_data,
 642         sparse_indices,
 643         sparse_values,
 644         sparse_shape,
 645         input_labels,
 646         input_weights,
 647         leaf_ids,
 648         input_spec=data_spec.SerializeToString(),
 649         random_seed=random_seed,
 650         params=self.params.serialized_params_proto)
 651 
 652     with ops.control_dependencies([update_model]):
 653       return stats_ops.grow_tree_v4(
 654           self.variables.tree,
 655           self.variables.stats,
 656           finished_nodes,
 657           params=self.params.serialized_params_proto)
 658

 660     """Constructs a TF graph for evaluating a random tree.

 662     Args:

 665         input columns.
 666       sparse_features: A tf.SparseTensor for sparse input data.
 667 
 668     Returns:
 669       A tuple of (probabilities, tree_paths).
 670     """
 671     sparse_indices = []
 672     sparse_values = []
 673     sparse_shape = []
 674     if sparse_features is not None:
 675       sparse_indices = sparse_features.indices
 676       sparse_values = sparse_features.values
 677       sparse_shape = sparse_features.dense_shape
 678     if input_data is None:
 679       input_data = []

 681     return model_ops.tree_predictions_v4(
 682         self.variables.tree,
 683         input_data,
 684         sparse_indices,
 685         sparse_values,
 686         sparse_shape,
 687         input_spec=data_spec.SerializeToString(),
 688         params=self.params.serialized_params_proto)
