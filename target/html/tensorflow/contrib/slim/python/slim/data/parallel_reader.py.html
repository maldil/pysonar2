  32 class ParallelReader(io_ops.ReaderBase):
  33   """Reader class that uses multiple readers in parallel to improve speed.
  34 
  35   See ReaderBase for supported methods.
  36   """
  37 
  38   def __init__(self,
  39                reader_class,
  40                common_queue,
  41                num_readers=4,
  42                reader_kwargs=None):
  43     """ParallelReader creates num_readers instances of the reader_class.
  44 
  45     Each instance is created by calling the `reader_class` function passing
  46     the arguments specified in `reader_kwargs` as in:
  47       reader_class(**read_kwargs)
  48 
  49     When you read from a ParallelReader, with its `read()` method,
  50     you just dequeue examples from the `common_queue`.
  51 
  52     The readers will read different files in parallel, asynchronously enqueueing
  53     their output into `common_queue`. The `common_queue.dtypes` must be
  54     [tf.string, tf.string]
  55 
  56     Because each reader can read from a different file, the examples in the
  57     `common_queue` could be from different files. Due to the asynchronous
  58     reading there is no guarantee that all the readers will read the same
  59     number of examples.
  60 
  61     If the `common_queue` is a shuffling queue, then the examples are shuffled.
  62 
  63     Usage:
  64       common_queue = tf.RandomShuffleQueue(
  65           capacity=256,
  66           min_after_dequeue=128,
  67           dtypes=[tf.string, tf.string])
  68       p_reader = ParallelReader(tf.TFRecordReader, common_queue)
  69 
  70       common_queue = tf.FIFOQueue(
  71           capacity=256,
  72           dtypes=[tf.string, tf.string])
  73       p_reader = ParallelReader(readers, common_queue, num_readers=2)
  74 
  75 
  76     Args:
  77       reader_class: one of the io_ops.ReaderBase subclasses ex: TFRecordReader
  78       common_queue: a Queue to hold (key, value pairs) with `dtypes` equal to
  79         [tf.string, tf.string]. Must be one of the data_flow_ops.Queues
  80         instances, ex. `tf.FIFOQueue()`, `tf.RandomShuffleQueue()`, ...
  81       num_readers: a integer, number of instances of reader_class to create.
  82       reader_kwargs: an optional dict of kwargs to create the readers.
  83 
  84     Raises:
  85       TypeError: if `common_queue.dtypes` is not [tf.string, tf.string].
  86     """
  87     if len(common_queue.dtypes) != 2:
  88       raise TypeError('common_queue.dtypes must be [tf.string, tf.string]')
  89     for dtype in common_queue.dtypes:
  90       if not dtype.is_compatible_with(tf_dtypes.string):
  91         raise TypeError('common_queue.dtypes must be [tf.string, tf.string]')
  92 
  93     reader_kwargs = reader_kwargs or {}
  94     self._readers = [reader_class(**reader_kwargs) for _ in range(num_readers)]
  95     self._common_queue = common_queue
  96 
  97   @property
  98   def num_readers(self):
  99     return len(self._readers)
 100 
 101   @property
 102   def common_queue(self):
 103     return self._common_queue
 104 
 105   def read(self, queue, name=None):
 106     """Returns the next record (key, value pair) produced by the reader.
 107 
 108     The multiple reader instances are all configured to `read()` from the
 109     filenames listed in `queue` and enqueue their output into the `common_queue`
 110     passed to the constructor, and this method returns the next record dequeued
 111     from that `common_queue`.
 112 
 113 
 114     Readers dequeue a work unit from `queue` if necessary (e.g. when a
 115     reader needs to start reading from a new file since it has finished with
 116     the previous file).
 117 
 118     A queue runner for enqueuing in the `common_queue` is automatically added
 119     to the TF QueueRunners collection.
 120 
 121     Args:
 122       queue: A Queue or a mutable string Tensor representing a handle
 123         to a Queue, with string work items.
 124       name: A name for the operation (optional).
 125 
 126     Returns:
 127       The next record (i.e. (key, value pair)) from the common_queue.
 128     """
 129 
 130     self._configure_readers_by(queue)
 131     return self._common_queue.dequeue(name=name)
 132 
 133   def read_up_to(self, queue, num_records, name=None):
 134     """Returns up to num_records (key, value pairs) produced by a reader.
 135 
 136     Will dequeue a work unit from queue if necessary (e.g., when the
 137     Reader needs to start reading from a new file since it has
 138     finished with the previous file).
 139     It may return less than num_records even before the last batch.
 140 
 141     **Note** This operation is not supported by all types of `common_queue`s.
 142     If a `common_queue` does not support `dequeue_up_to()`, then a
 143     `tf.errors.UnimplementedError` is raised.
 144 
 145     Args:
 146       queue: A Queue or a mutable string Tensor representing a handle
 147         to a Queue, with string work items.
 148       num_records: Number of records to read.
 149       name: A name for the operation (optional).
 150 
 151     Returns:
 152       A tuple of Tensors (keys, values) from common_queue.
 153       keys: A 1-D string Tensor.
 154       values: A 1-D string Tensor.
 155     """
 156     self._configure_readers_by(queue)
 157     return self._common_queue.dequeue_up_to(num_records, name)
 158 
 159   def _configure_readers_by(self, queue):
 160     enqueue_ops = []
 161     for reader in self._readers:
 162       enqueue_ops.append(self._common_queue.enqueue(reader.read(queue)))
 163 
 164     queue_runner.add_queue_runner(
 165         queue_runner.QueueRunner(self._common_queue, enqueue_ops))
 166 
 167   def num_records_produced(self, name=None):
 168     """Returns the number of records this reader has produced.
 169 
 170     Args:
 171       name: A name for the operation (optional).
 172 
 173     Returns:
 174       An int64 Tensor.
 175 
 176     """
 177     num_records = [r.num_records_produced() for r in self._readers]
 178     return math_ops.add_n(num_records, name=name)
 179 
 180   def num_work_units_completed(self, name=None):
 181     """Returns the number of work units this reader has finished processing.
 182 
 183     Args:
 184       name: A name for the operation (optional).
 185 
 186     Returns:
 187       An int64 Tensor.
 188     """
 189     num_work_units = [r.num_work_units_completed() for r in self._readers]
 190     return math_ops.add_n(num_work_units, name=name)

 193 def parallel_read(data_sources,
 194                   reader_class,
 195                   num_epochs=None,
 196                   num_readers=4,
 197                   reader_kwargs=None,
 198                   shuffle=True,
 199                   dtypes=None,
 200                   capacity=256,
 201                   min_after_dequeue=128,
 202                   seed=None,
 203                   scope=None):
 204   """Reads multiple records in parallel from data_sources using n readers.
 205 
 206   It uses a ParallelReader to read from multiple files in parallel using
 207   multiple readers created using `reader_class` with `reader_kwargs'.
 208 
 209   If shuffle is True the common_queue would be a RandomShuffleQueue otherwise
 210   it would be a FIFOQueue.
 211 
 212   Usage:
 213       data_sources = ['path_to/train*']
 214       key, value = parallel_read(data_sources, tf.CSVReader, num_readers=4)
 215 
 216   Args:
 217     data_sources: a list/tuple of files or the location of the data, i.e.
 218       /path/to/train@128, /path/to/train* or /tmp/.../train*
 219     reader_class: one of the io_ops.ReaderBase subclasses ex: TFRecordReader
 220     num_epochs: The number of times each data source is read. If left as None,
 221         the data will be cycled through indefinitely.
 222     num_readers: a integer, number of Readers to create.
 223     reader_kwargs: an optional dict, of kwargs for the reader.
 224     shuffle: boolean, whether should shuffle the files and the records by using
 225       RandomShuffleQueue as common_queue.
 226     dtypes:  A list of types.  The length of dtypes must equal the number
 227         of elements in each record. If it is None it will default to
 228         [tf.string, tf.string] for (key, value).
 229     capacity: integer, capacity of the common_queue.
 230     min_after_dequeue: integer, minimum number of records in the common_queue
 231       after dequeue. Needed for a good shuffle.
 232     seed: A seed for RandomShuffleQueue.
 233     scope: Optional name scope for the ops.
 234 
 235   Returns:
 236     key, value: a tuple of keys and values from the data_source.
 237   """
 238   data_files = get_data_files(data_sources)
 239   with ops.name_scope(scope, 'parallel_read'):
 240     filename_queue = tf_input.string_input_producer(
 241         data_files, num_epochs=num_epochs, shuffle=shuffle, seed=seed,
 242         name='filenames')
 243     dtypes = dtypes or [tf_dtypes.string, tf_dtypes.string]
 244     if shuffle:
 245       common_queue = data_flow_ops.RandomShuffleQueue(
 246           capacity=capacity,
 247           min_after_dequeue=min_after_dequeue,
 248           dtypes=dtypes,
 249           seed=seed,
 250           name='common_queue')
 251     else:
 252       common_queue = data_flow_ops.FIFOQueue(
 253           capacity=capacity, dtypes=dtypes, name='common_queue')
 254 
 255     summary.scalar('fraction_of_%d_full' % capacity,
 256                    math_ops.to_float(common_queue.size()) * (1. / capacity))
 257 
 258     return ParallelReader(
 259         reader_class,
 260         common_queue,
 261         num_readers=num_readers,
 262         reader_kwargs=reader_kwargs).read(filename_queue)

 265 def single_pass_read(data_sources, reader_class, reader_kwargs=None,
 266                      scope=None):
 267   """Reads sequentially the data_sources using the reader, doing a single pass.
 268 
 269   Args:
 270     data_sources: a list/tuple of files or the location of the data, i.e.
 271       /path/to/train@128, /path/to/train* or /tmp/.../train*
 272     reader_class: one of the io_ops.ReaderBase subclasses ex: TFRecordReader.
 273     reader_kwargs: an optional dict, of kwargs for the reader.
 274     scope: Optional name scope for the ops.
 275 
 276   Returns:
 277     key, value: a tuple of keys and values from the data_source.
 278   """
 279   data_files = get_data_files(data_sources)
 280   with ops.name_scope(scope, 'single_pass_read'):
 281     filename_queue = tf_input.string_input_producer(
 282         data_files, num_epochs=1, shuffle=False, capacity=1, name='filenames')
 283     reader_kwargs = reader_kwargs or {}
 284     return reader_class(**reader_kwargs).read(filename_queue)

 287 def get_data_files(data_sources):
 288   """Get data_files from data_sources.
 289 
 290   Args:
 291     data_sources: a list/tuple of files or the location of the data, i.e.
 292       /path/to/train@128, /path/to/train* or /tmp/.../train*
 293 
 294   Returns:
 295     a list of data_files.
 296 
 297   Raises:
 298     ValueError: if not data files are not found
 299 
 300   """
 301   if isinstance(data_sources, (list, tuple)):
 302     data_files = []
 303     for source in data_sources:
 304       data_files += get_data_files(source)
 305   else:
 306     if '*' in data_sources or '?' in data_sources or '[' in data_sources:
 307       data_files = gfile.Glob(data_sources)
 308     else:
 309       data_files = [data_sources]
 310   if not data_files:
 311     raise ValueError('No data files found in %s' % (data_sources,))
 312   return data_files
