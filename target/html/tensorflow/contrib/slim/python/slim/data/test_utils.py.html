  33 def _encoded_int64_feature(ndarray):
  34   return feature_pb2.Feature(int64_list=feature_pb2.Int64List(
  35       value=ndarray.flatten().tolist()))
  36

  38 def _encoded_bytes_feature(tf_encoded):
  39   encoded = tf_encoded.eval()
  40 
  41   def string_to_bytes(value):
  42     return feature_pb2.BytesList(value=[value])
  43 
  44   return feature_pb2.Feature(bytes_list=string_to_bytes(encoded))
  45 
  46 
  47 def _string_feature(value):
  48   value = value.encode('utf-8')
  49   return feature_pb2.Feature(bytes_list=feature_pb2.BytesList(value=[value]))

  52 def _encoder(image, image_format):
  53   assert image_format in ['jpeg', 'png']
  54   if image_format == 'jpeg':
  55     tf_image = constant_op.constant(image, dtype=dtypes.uint8)
  56     return image_ops.encode_jpeg(tf_image)
  57   if image_format == 'png':
  58     tf_image = constant_op.constant(image, dtype=dtypes.uint8)
  59     return image_ops.encode_png(tf_image)

  62 def generate_image(image_shape, image_format='jpeg', label=0):
  63   """Generates an image and an example containing the encoded image.
  64 
  65   GenerateImage must be called within an active session.
  66 
  67   Args:
  68     image_shape: the shape of the image to generate.
  69     image_format: the encoding format of the image.
  70     label: the int64 labels for the image.
  71 
  72   Returns:
  73     image: the generated image.
  74     example: a TF-example with a feature key 'image/encoded' set to the
  75       serialized image and a feature key 'image/format' set to the image
  76       encoding format ['jpeg', 'png'].
  77   """
  78   image = np.random.random_integers(0, 255, size=image_shape)
  79   tf_encoded = _encoder(image, image_format)
  80   example = example_pb2.Example(features=feature_pb2.Features(feature={
  81       'image/encoded': _encoded_bytes_feature(tf_encoded),
  82       'image/format': _string_feature(image_format),
  83       'image/class/label': _encoded_int64_feature(np.array(label)),
  84   }))
  85 
  86   return image, example.SerializeToString()

  89 def create_tfrecord_files(output_dir, num_files=3, num_records_per_file=10):
  90   """Creates TFRecords files.
  91 
  92   The method must be called within an active session.
  93 
  94   Args:
  95     output_dir: The directory where the files are stored.
  96     num_files: The number of files to create.
  97     num_records_per_file: The number of records per file.
  98 
  99   Returns:
 100     A list of the paths to the TFRecord files.
 101   """
 102   tfrecord_paths = []
 103   for i in range(num_files):
 104     path = os.path.join(output_dir,
 105                         'flowers.tfrecord-%d-of-%s' % (i, num_files))
 106     tfrecord_paths.append(path)
 107 
 108     writer = tf_record.TFRecordWriter(path)
 109     for _ in range(num_records_per_file):
 110       _, example = generate_image(image_shape=(10, 10, 3))
 111       writer.write(example)
 112     writer.close()
 113 
 114   return tfrecord_paths
