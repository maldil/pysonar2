  15 """Contains the TFExampleDecoder its associated helper classes.
  16 
  17 The TFExampleDecode is a DataDecoder used to decode TensorFlow Example protos.
  18 In order to do so each requested item must be paired with one or more Example
  19 features that are parsed to produce the Tensor-based manifestation of the item.

  40 class ItemHandler(object):
  41   """Specifies the item-to-Features mapping for tf.parse_example.
  42 
  43   An ItemHandler both specifies a list of Features used for parsing an Example
  44   proto as well as a function that post-processes the results of Example
  45   parsing.
  46   """
  47 
  48   __metaclass__ = abc.ABCMeta
  49 
  50   def __init__(self, keys):
  51     """Constructs the handler with the name of the tf.Feature keys to use.
  52 
  53     See third_party/tensorflow/core/example/feature.proto
  54 
  55     Args:
  56       keys: the name of the TensorFlow Example Feature.
  57     """
  58     if not isinstance(keys, (tuple, list)):
  59       keys = [keys]
  60     self._keys = keys
  61 
  62   @property
  63   def keys(self):
  64     return self._keys
  65 
  66   @abc.abstractmethod
  67   def tensors_to_item(self, keys_to_tensors):
  68     """Maps the given dictionary of tensors to the requested item.
  69 
  70     Args:
  71       keys_to_tensors: a mapping of TF-Example keys to parsed tensors.
  72 
  73     Returns:
  74       the final tensor representing the item being handled.
  75     """
  76     pass

  79 class ItemHandlerCallback(ItemHandler):
  80   """An ItemHandler that converts the parsed tensors via a given function.
  81 
  82   Unlike other ItemHandlers, the ItemHandlerCallback resolves its item via
  83   a callback function rather than using prespecified behavior.
  84   """
  85 
  86   def __init__(self, keys, func):
  87     """Initializes the ItemHandler.
  88 
  89     Args:
  90       keys: a list of TF-Example keys.
  91       func: a function that takes as an argument a dictionary from `keys` to
  92         parsed Tensors.
  93     """
  94     super(ItemHandlerCallback, self).__init__(keys)
  95     self._func = func
  96 
  97   def tensors_to_item(self, keys_to_tensors):
  98     return self._func(keys_to_tensors)

 101 class BoundingBox(ItemHandler):
 102   """An ItemHandler that concatenates a set of parsed Tensors to Bounding Boxes.
 103   """
 104 
 105   def __init__(self, keys=None, prefix=''):
 106     """Initialize the bounding box handler.
 107 
 108     Args:
 109       keys: A list of four key names representing the ymin, xmin, ymax, mmax
 110       prefix: An optional prefix for each of the bounding box keys.
 111         If provided, `prefix` is appended to each key in `keys`.
 112 
 113     Raises:
 114       ValueError: if keys is not `None` and also not a list of exactly 4 keys
 115     """
 116     if keys is None:
 117       keys = ['ymin', 'xmin', 'ymax', 'xmax']
 118     elif len(keys) != 4:
 119       raise ValueError('BoundingBox expects 4 keys but got {}'.format(
 120           len(keys)))
 121     self._prefix = prefix
 122     self._keys = keys
 123     self._full_keys = [prefix + k for k in keys]
 124     super(BoundingBox, self).__init__(self._full_keys)
 125 
 126   def tensors_to_item(self, keys_to_tensors):
 127     """Maps the given dictionary of tensors to a concatenated list of bboxes.
 128 
 129     Args:
 130       keys_to_tensors: a mapping of TF-Example keys to parsed tensors.
 131 
 132     Returns:
 133       [num_boxes, 4] tensor of bounding box coordinates,
 134         i.e. 1 bounding box per row, in order [y_min, x_min, y_max, x_max].
 135     """
 136     sides = []
 137     for key in self._full_keys:
 138       side = keys_to_tensors[key]
 139       if isinstance(side, sparse_tensor.SparseTensor):
 140         side = side.values
 141       side = array_ops.expand_dims(side, 0)
 142       sides.append(side)
 143 
 144     bounding_box = array_ops.concat(sides, 0)
 145     return array_ops.transpose(bounding_box)

 148 class Tensor(ItemHandler):
 149   """An ItemHandler that returns a parsed Tensor."""
 150 
 151   def __init__(self, tensor_key, shape_keys=None, shape=None, default_value=0):
 152     """Initializes the Tensor handler.
 153 
 154     Tensors are, by default, returned without any reshaping. However, there are
 155     two mechanisms which allow reshaping to occur at load time. If `shape_keys`
 156     is provided, both the `Tensor` corresponding to `tensor_key` and
 157     `shape_keys` is loaded and the former `Tensor` is reshaped with the values
 158     of the latter. Alternatively, if a fixed `shape` is provided, the `Tensor`
 159     corresponding to `tensor_key` is loaded and reshape appropriately.
 160     If neither `shape_keys` nor `shape` are provided, the `Tensor` will be
 161     returned without any reshaping.
 162 
 163     Args:
 164       tensor_key: the name of the `TFExample` feature to read the tensor from.
 165       shape_keys: Optional name or list of names of the TF-Example feature in
 166         which the tensor shape is stored. If a list, then each corresponds to
 167         one dimension of the shape.
 168       shape: Optional output shape of the `Tensor`. If provided, the `Tensor` is
 169         reshaped accordingly.
 170       default_value: The value used when the `tensor_key` is not found in a
 171         particular `TFExample`.
 172 
 173     Raises:
 174       ValueError: if both `shape_keys` and `shape` are specified.
 175     """
 176     if shape_keys and shape is not None:
 177       raise ValueError('Cannot specify both shape_keys and shape parameters.')
 178     if shape_keys and not isinstance(shape_keys, list):
 179       shape_keys = [shape_keys]
 180     self._tensor_key = tensor_key
 181     self._shape_keys = shape_keys
 182     self._shape = shape
 183     self._default_value = default_value
 184     keys = [tensor_key]
 185     if shape_keys:
 186       keys.extend(shape_keys)
 187     super(Tensor, self).__init__(keys)
 188 
 189   def tensors_to_item(self, keys_to_tensors):
 190     tensor = keys_to_tensors[self._tensor_key]
 191     shape = self._shape
 192     if self._shape_keys:
 193       shape_dims = []
 194       for k in self._shape_keys:
 195         shape_dim = keys_to_tensors[k]
 196         if isinstance(shape_dim, sparse_tensor.SparseTensor):
 197           shape_dim = sparse_ops.sparse_tensor_to_dense(shape_dim)
 198         shape_dims.append(shape_dim)
 199       shape = array_ops.reshape(array_ops.stack(shape_dims), [-1])
 200     if isinstance(tensor, sparse_tensor.SparseTensor):
 201       if shape is not None:
 202         tensor = sparse_ops.sparse_reshape(tensor, shape)
 203       tensor = sparse_ops.sparse_tensor_to_dense(tensor, self._default_value)
 204     else:
 205       if shape is not None:
 206         tensor = array_ops.reshape(tensor, shape)
 207     return tensor

 210 class LookupTensor(Tensor):
 211   """An ItemHandler that returns a parsed Tensor, the result of a lookup."""
 212 
 213   def __init__(self,
 214                tensor_key,
 215                table,
 216                shape_keys=None,
 217                shape=None,
 218                default_value=''):
 219     """Initializes the LookupTensor handler.
 220 
 221     See Tensor.  Simply calls a vocabulary (most often, a label mapping) lookup.
 222 
 223     Args:
 224       tensor_key: the name of the `TFExample` feature to read the tensor from.
 225       table: A tf.lookup table.
 226       shape_keys: Optional name or list of names of the TF-Example feature in
 227         which the tensor shape is stored. If a list, then each corresponds to
 228         one dimension of the shape.
 229       shape: Optional output shape of the `Tensor`. If provided, the `Tensor` is
 230         reshaped accordingly.
 231       default_value: The value used when the `tensor_key` is not found in a
 232         particular `TFExample`.
 233 
 234     Raises:
 235       ValueError: if both `shape_keys` and `shape` are specified.
 236     """
 237     self._table = table
 238     super(LookupTensor, self).__init__(tensor_key, shape_keys, shape,
 239                                        default_value)
 240 
 241   def tensors_to_item(self, keys_to_tensors):
 242     unmapped_tensor = super(LookupTensor, self).tensors_to_item(keys_to_tensors)
 243     return self._table.lookup(unmapped_tensor)

 246 class BackupHandler(ItemHandler):
 247   """An ItemHandler that tries two ItemHandlers in order."""
 248 
 249   def __init__(self, handler, backup):
 250     """Initializes the BackupHandler handler.
 251 
 252     If the first Handler's tensors_to_item returns a Tensor with no elements,
 253     the second Handler is used.
 254 
 255     Args:
 256       handler: The primary ItemHandler.
 257       backup: The backup ItemHandler.
 258 
 259     Raises:
 260       ValueError: if either is not an ItemHandler.
 261     """
 262     if not isinstance(handler, ItemHandler):
 263       raise ValueError('Primary handler is of type %s instead of ItemHandler'
 264                        % type(handler))
 265     if not isinstance(backup, ItemHandler):
 266       raise ValueError('Backup handler is of type %s instead of ItemHandler'
 267                        % type(backup))
 268     self._handler = handler
 269     self._backup = backup
 270     super(BackupHandler, self).__init__(handler.keys + backup.keys)
 271 
 272   def tensors_to_item(self, keys_to_tensors):
 273     item = self._handler.tensors_to_item(keys_to_tensors)
 274     return control_flow_ops.cond(
 275         pred=math_ops.equal(math_ops.reduce_prod(array_ops.shape(item)), 0),
 276         true_fn=lambda: self._backup.tensors_to_item(keys_to_tensors),
 277         false_fn=lambda: item)

 280 class SparseTensor(ItemHandler):
 281   """An ItemHandler for SparseTensors."""
 282 
 283   def __init__(self,
 284                indices_key=None,
 285                values_key=None,
 286                shape_key=None,
 287                shape=None,
 288                densify=False,
 289                default_value=0):
 290     """Initializes the Tensor handler.
 291 
 292     Args:
 293       indices_key: the name of the TF-Example feature that contains the ids.
 294         Defaults to 'indices'.
 295       values_key: the name of the TF-Example feature that contains the values.
 296         Defaults to 'values'.
 297       shape_key: the name of the TF-Example feature that contains the shape.
 298         If provided it would be used.
 299       shape: the output shape of the SparseTensor. If `shape_key` is not
 300         provided this `shape` would be used.
 301       densify: whether to convert the SparseTensor into a dense Tensor.
 302       default_value: Scalar value to set when making dense for indices not
 303         specified in the `SparseTensor`.
 304     """
 305     indices_key = indices_key or 'indices'
 306     values_key = values_key or 'values'
 307     self._indices_key = indices_key
 308     self._values_key = values_key
 309     self._shape_key = shape_key
 310     self._shape = shape
 311     self._densify = densify
 312     self._default_value = default_value
 313     keys = [indices_key, values_key]
 314     if shape_key:
 315       keys.append(shape_key)
 316     super(SparseTensor, self).__init__(keys)
 317 
 318   def tensors_to_item(self, keys_to_tensors):
 319     indices = keys_to_tensors[self._indices_key]
 320     values = keys_to_tensors[self._values_key]
 321     if self._shape_key:
 322       shape = keys_to_tensors[self._shape_key]
 323       if isinstance(shape, sparse_tensor.SparseTensor):
 324         shape = sparse_ops.sparse_tensor_to_dense(shape)
 325     elif self._shape:
 326       shape = self._shape
 327     else:
 328       shape = indices.dense_shape
 329     indices_shape = array_ops.shape(indices.indices)
 330     rank = indices_shape[1]
 331     ids = math_ops.to_int64(indices.values)
 332     indices_columns_to_preserve = array_ops.slice(
 333         indices.indices, [0, 0], array_ops.stack([-1, rank - 1]))
 334     new_indices = array_ops.concat(
 335         [indices_columns_to_preserve, array_ops.reshape(ids, [-1, 1])], 1)
 336 
 337     tensor = sparse_tensor.SparseTensor(new_indices, values.values, shape)
 338     if self._densify:
 339       tensor = sparse_ops.sparse_tensor_to_dense(tensor, self._default_value)
 340     return tensor

 343 class Image(ItemHandler):
 344   """An ItemHandler that decodes a parsed Tensor as an image."""
 345 
 346   def __init__(self,
 347                image_key=None,
 348                format_key=None,
 349                shape=None,
 350                channels=3,
 351                dtype=dtypes.uint8,
 352                repeated=False,
 353                dct_method=''):
 354     """Initializes the image.
 355 
 356     Args:
 357       image_key: the name of the TF-Example feature in which the encoded image
 358         is stored.
 359       format_key: the name of the TF-Example feature in which the image format
 360         is stored.
 361       shape: the output shape of the image as 1-D `Tensor`
 362         [height, width, channels]. If provided, the image is reshaped
 363         accordingly. If left as None, no reshaping is done. A shape should
 364         be supplied only if all the stored images have the same shape.
 365       channels: the number of channels in the image.
 366       dtype: images will be decoded at this bit depth. Different formats
 367         support different bit depths.
 368           See tf.image.decode_image,
 369               tf.decode_raw,
 370       repeated: if False, decodes a single image. If True, decodes a
 371         variable number of image strings from a 1D tensor of strings.
 372       dct_method: An optional string. Defaults to empty string. It only takes
 373         effect when image format is jpeg, used to specify a hint about the
 374         algorithm used for jpeg decompression. Currently valid values
 375         are ['INTEGER_FAST', 'INTEGER_ACCURATE']. The hint may be ignored, for
 376         example, the jpeg library does not have that specific option.
 377     """
 378     if not image_key:
 379       image_key = 'image/encoded'
 380     if not format_key:
 381       format_key = 'image/format'
 382 
 383     super(Image, self).__init__([image_key, format_key])
 384     self._image_key = image_key
 385     self._format_key = format_key
 386     self._shape = shape
 387     self._channels = channels
 388     self._dtype = dtype
 389     self._repeated = repeated
 390     self._dct_method = dct_method
 391 
 392   def tensors_to_item(self, keys_to_tensors):
 393     """See base class."""
 394     image_buffer = keys_to_tensors[self._image_key]
 395     image_format = keys_to_tensors[self._format_key]
 396 
 397     if self._repeated:
 398       return functional_ops.map_fn(lambda x: self._decode(x, image_format),
 399                                    image_buffer, dtype=self._dtype)
 400     else:
 401       return self._decode(image_buffer, image_format)
 402 
 403   def _decode(self, image_buffer, image_format):
 404     """Decodes the image buffer.
 405 
 406     Args:
 407       image_buffer: The tensor representing the encoded image tensor.
 408       image_format: The image format for the image in `image_buffer`. If image
 409         format is `raw`, all images are expected to be in this format, otherwise
 410         this op can decode a mix of `jpg` and `png` formats.
 411 
 412     Returns:
 413       A tensor that represents decoded image of self._shape, or
 414       (?, ?, self._channels) if self._shape is not specified.
 415     """
 416 
 417     def decode_image():
 418       """Decodes a image based on the headers."""
 419       return math_ops.cast(
 420           image_ops.decode_image(image_buffer, channels=self._channels),
 421           self._dtype)
 422 
 423     def decode_jpeg():
 424       """Decodes a jpeg image with specified '_dct_method'."""
 425       return math_ops.cast(
 426           image_ops.decode_jpeg(
 427               image_buffer,
 428               channels=self._channels,
 429               dct_method=self._dct_method), self._dtype)
 430 
 431     def check_jpeg():
 432       """Checks if an image is jpeg."""
 433       # For jpeg, we directly use image_ops.decode_jpeg rather than decode_image
 434       # in order to feed the jpeg specify parameter 'dct_method'.
 435       return control_flow_ops.cond(
 436           image_ops.is_jpeg(image_buffer),
 437           decode_jpeg,
 438           decode_image,
 439           name='cond_jpeg')
 440 
 441     def decode_raw():
 442       """Decodes a raw image."""
 443       return parsing_ops.decode_raw(image_buffer, out_type=self._dtype)
 444 
 445     pred_fn_pairs = {
 446         math_ops.logical_or(
 447             math_ops.equal(image_format, 'raw'),
 448             math_ops.equal(image_format, 'RAW')): decode_raw,
 449     }
 450     image = control_flow_ops.case(
 451         pred_fn_pairs, default=check_jpeg, exclusive=True)
 452 
 453     image.set_shape([None, None, self._channels])
 454     if self._shape is not None:
 455       image = array_ops.reshape(image, self._shape)
 456 
 457     return image

 460 class TFExampleDecoder(data_decoder.DataDecoder):
 461   """A decoder for TensorFlow Examples.
 462 
 463   Decoding Example proto buffers is comprised of two stages: (1) Example parsing
 464   and (2) tensor manipulation.
 465 
 466   In the first stage, the tf.parse_example function is called with a list of
 467   FixedLenFeatures and SparseLenFeatures. These instances tell TF how to parse
 468   the example. The output of this stage is a set of tensors.
 469 
 470   In the second stage, the resulting tensors are manipulated to provide the
 471   requested 'item' tensors.
 472 
 473   To perform this decoding operation, an ExampleDecoder is given a list of
 474   ItemHandlers. Each ItemHandler indicates the set of features for stage 1 and
 475   contains the instructions for post_processing its tensors for stage 2.
 476   """
 477 
 478   def __init__(self, keys_to_features, items_to_handlers):
 479     """Constructs the decoder.
 480 
 481     Args:
 482       keys_to_features: a dictionary from TF-Example keys to either
 483         tf.VarLenFeature or tf.FixedLenFeature instances. See tensorflow's
 484         parsing_ops.py.
 485       items_to_handlers: a dictionary from items (strings) to ItemHandler
 486         instances. Note that the ItemHandler's are provided the keys that they
 487         use to return the final item Tensors.
 488     """
 489     self._keys_to_features = keys_to_features
 490     self._items_to_handlers = items_to_handlers
 491 
 492   def list_items(self):
 493     """See base class."""
 494     return list(self._items_to_handlers.keys())
 495 
 496   def decode(self, serialized_example, items=None):
 497     """Decodes the given serialized TF-example.
 498 
 499     Args:
 500       serialized_example: a serialized TF-example tensor.
 501       items: the list of items to decode. These must be a subset of the item
 502         keys in self._items_to_handlers. If `items` is left as None, then all
 503         of the items in self._items_to_handlers are decoded.
 504 
 505     Returns:
 506       the decoded items, a list of tensor.
 507     """
 508     example = parsing_ops.parse_single_example(serialized_example,
 509                                                self._keys_to_features)
 510 
 511     # Reshape non-sparse elements just once, adding the reshape ops in
 512     # deterministic order.
 513     for k in sorted(self._keys_to_features):
 514       v = self._keys_to_features[k]
 515       if isinstance(v, parsing_ops.FixedLenFeature):
 516         example[k] = array_ops.reshape(example[k], v.shape)
 517 
 518     if not items:
 519       items = self._items_to_handlers.keys()
 520 
 521     outputs = []
 522     for item in items:
 523       handler = self._items_to_handlers[item]
 524       keys_to_tensors = {key: example[key] for key in handler.keys}
 525       outputs.append(handler.tensors_to_item(keys_to_tensors))
 526     return outputs
