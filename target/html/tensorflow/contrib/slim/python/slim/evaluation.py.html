  15 """Contains functions for evaluation and summarization of metrics.
  16 
  17 The evaluation.py module contains helper functions for evaluating TensorFlow
  18 modules using a variety of metrics and summarizing the results.
  19 
  20 **********************
  21 * Evaluating Metrics *
  22 **********************
  23 
  24 In the simplest use case, we use a model to create the predictions, then specify
  25 the metrics and choose one model checkpoint, finally call the`evaluation_once`
  26 method:
  27 
  28   # Create model and obtain the predictions:
  29   images, labels = LoadData(...)
  30   predictions = MyModel(images)
  31 
  32   # Choose the metrics to compute:
  33   names_to_values, names_to_updates = slim.metrics.aggregate_metric_map({
  34       "accuracy": slim.metrics.accuracy(predictions, labels),
  35       "mse": slim.metrics.mean_squared_error(predictions, labels),
  36   })
  37 
  38   checkpoint_path = '/tmp/my_model_dir/my_checkpoint'
  39   log_dir = '/tmp/my_model_eval/'
  40 
  41   initial_op = tf.group(
  42       tf.global_variables_initializer(),
  43       tf.local_variables_initializer())
  44 
  45   metric_values = slim.evaluate_once(
  46       master='',
  47       checkpoint_path=checkpoint_path,
  48       log_dir=log_dir,
  49       num_evals=1,
  50       initial_op=initial_op,
  51       eval_op=names_to_updates.values(),
  52       final_op=name_to_values.values())
  53 
  54   for metric, value in zip(names_to_values.keys(), metric_values):
  55     logging.info('Metric %s has value: %f', metric, value)
  56 
  57 ************************************************
  58 * Evaluating a Checkpointed Model with Metrics *
  59 ************************************************
  60 
  61 Often, one wants to evaluate a model checkpoint saved on disk. This can be
  62 performed once or repeatedly on a set schedule.
  63 
  64 To evaluate a particular model, users define zero or more metrics and zero or
  65 more summaries and call the evaluation_loop method:
  66 
  67   # Create model and obtain the predictions:
  68   images, labels = LoadData(...)
  69   predictions = MyModel(images)
  70 
  71   # Choose the metrics to compute:
  72   names_to_values, names_to_updates = slim.metrics.aggregate_metric_map({
  73       "accuracy": slim.metrics.accuracy(predictions, labels),
  74       "mse": slim.metrics.mean_squared_error(predictions, labels),
  75   })
  76 
  77   # Define the summaries to write:
  78   for metric_name, metric_value in metrics_to_values.iteritems():
  79     tf.summary.scalar(metric_name, metric_value)
  80 
  81   checkpoint_dir = '/tmp/my_model_dir/'
  82   log_dir = '/tmp/my_model_eval/'
  83 
  84   # We'll evaluate 1000 batches:
  85   num_evals = 1000
  86 
  87   # Evaluate every 10 minutes:
  88   slim.evaluation_loop(
  89       '',
  90       checkpoint_dir,
  91       logdir,
  92       num_evals=num_evals,
  93       eval_op=names_to_updates.values(),
  94       summary_op=tf.contrib.deprecated.merge_summary(summary_ops),
  95       eval_interval_secs=600)
  96 
  97 **************************************************
  98 * Evaluating a Checkpointed Model with Summaries *
  99 **************************************************
 100 
 101 At times, an evaluation can be performed without metrics at all but rather
 102 with only summaries. The user need only leave out the 'eval_op' argument:
 103 
 104   # Create model and obtain the predictions:
 105   images, labels = LoadData(...)
 106   predictions = MyModel(images)
 107 
 108   # Define the summaries to write:
 109   tf.summary.scalar(...)
 110   tf.summary.histogram(...)
 111 
 112   checkpoint_dir = '/tmp/my_model_dir/'
 113   log_dir = '/tmp/my_model_eval/'
 114 
 115   # Evaluate once every 10 minutes.
 116   slim.evaluation_loop(
 117       master='',
 118       checkpoint_dir,
 119       logdir,
 120       num_evals=1,
 121       summary_op=tf.contrib.deprecated.merge_summary(summary_ops),
 122       eval_interval_secs=600)
 123

 148 def evaluate_once(master,
 149                   checkpoint_path,
 150                   logdir,
 151                   num_evals=1,
 152                   initial_op=None,
 153                   initial_op_feed_dict=None,
 154                   eval_op=None,
 155                   eval_op_feed_dict=None,
 156                   final_op=None,
 157                   final_op_feed_dict=None,
 158                   summary_op=_USE_DEFAULT,
 159                   summary_op_feed_dict=None,
 160                   variables_to_restore=None,
 161                   session_config=None,
 162                   hooks=None):
 163   """Evaluates the model at the given checkpoint path.
 164 
 165   Args:
 166     master: The BNS address of the TensorFlow master.
 167     checkpoint_path: The path to a checkpoint to use for evaluation.
 168     logdir: The directory where the TensorFlow summaries are written to.
 169     num_evals: The number of times to run `eval_op`.
 170     initial_op: An operation run at the beginning of evaluation.
 171     initial_op_feed_dict: A feed dictionary to use when executing `initial_op`.
 172     eval_op: A operation run `num_evals` times.
 173     eval_op_feed_dict: The feed dictionary to use when executing the `eval_op`.
 174     final_op: An operation to execute after all of the `eval_op` executions. The
 175       value of `final_op` is returned.
 176     final_op_feed_dict: A feed dictionary to use when executing `final_op`.
 177     summary_op: The summary_op to evaluate after running TF-Slims metric ops. By
 178       default the summary_op is set to tf.summary.merge_all().
 179     summary_op_feed_dict: An optional feed dictionary to use when running the
 180       `summary_op`.
 181     variables_to_restore: A list of TensorFlow variables to restore during
 182       evaluation. If the argument is left as `None` then
 183       slim.variables.GetVariablesToRestore() is used.
 184     session_config: An instance of `tf.ConfigProto` that will be used to
 185       configure the `Session`. If left as `None`, the default will be used.
 186     hooks: A list of additional `SessionRunHook` objects to pass during the
 187       evaluation.
 188 
 189   Returns:
 190     The value of `final_op` or `None` if `final_op` is `None`.
 191   """
 192   if summary_op == _USE_DEFAULT:
 193     summary_op = summary.merge_all()
 194 
 195   all_hooks = [evaluation.StopAfterNEvalsHook(num_evals),]
 196 
 197   if summary_op is not None:
 198     all_hooks.append(evaluation.SummaryAtEndHook(
 199         log_dir=logdir, summary_op=summary_op, feed_dict=summary_op_feed_dict))
 200   if hooks is not None:
 201     all_hooks.extend(hooks)
 202 
 203   saver = None
 204   if variables_to_restore is not None:
 205     saver = tf_saver.Saver(variables_to_restore)
 206 
 207   return evaluation.evaluate_once(
 208       checkpoint_path,
 209       master=master,
 210       scaffold=monitored_session.Scaffold(
 211           init_op=initial_op, init_feed_dict=initial_op_feed_dict, saver=saver),
 212       eval_ops=eval_op,
 213       feed_dict=eval_op_feed_dict,
 214       final_ops=final_op,
 215       final_ops_feed_dict=final_op_feed_dict,
 216       hooks=all_hooks,
 217       config=session_config)

 220 def evaluation_loop(master,
 221                     checkpoint_dir,
 222                     logdir,
 223                     num_evals=1,
 224                     initial_op=None,
 225                     initial_op_feed_dict=None,
 226                     init_fn=None,
 227                     eval_op=None,
 228                     eval_op_feed_dict=None,
 229                     final_op=None,
 230                     final_op_feed_dict=None,
 231                     summary_op=_USE_DEFAULT,
 232                     summary_op_feed_dict=None,
 233                     variables_to_restore=None,
 234                     eval_interval_secs=60,
 235                     max_number_of_evaluations=None,
 236                     session_config=None,
 237                     timeout=None,
 238                     timeout_fn=None,
 239                     hooks=None):
 240   """Runs TF-Slim's Evaluation Loop.
 241 
 242   Args:
 243     master: The BNS address of the TensorFlow master.
 244     checkpoint_dir: The directory where checkpoints are stored.
 245     logdir: The directory where the TensorFlow summaries are written to.
 246     num_evals: The number of times to run `eval_op`.
 247     initial_op: An operation run at the beginning of evaluation.
 248     initial_op_feed_dict: A feed dictionary to use when executing `initial_op`.
 249     init_fn: An optional callable to be executed after `init_op` is called. The
 250       callable must accept one argument, the session being initialized.
 251     eval_op: A operation run `num_evals` times.
 252     eval_op_feed_dict: The feed dictionary to use when executing the `eval_op`.
 253     final_op: An operation to execute after all of the `eval_op` executions. The
 254       value of `final_op` is returned.
 255     final_op_feed_dict: A feed dictionary to use when executing `final_op`.
 256     summary_op: The summary_op to evaluate after running TF-Slims metric ops. By
 257       default the summary_op is set to tf.summary.merge_all().
 258     summary_op_feed_dict: An optional feed dictionary to use when running the
 259       `summary_op`.
 260     variables_to_restore: A list of TensorFlow variables to restore during
 261       evaluation. If the argument is left as `None` then
 262       slim.variables.GetVariablesToRestore() is used.
 263     eval_interval_secs: The minimum number of seconds between evaluations.
 264     max_number_of_evaluations: the max number of iterations of the evaluation.
 265       If the value is left as 'None', the evaluation continues indefinitely.
 266     session_config: An instance of `tf.ConfigProto` that will be used to
 267       configure the `Session`. If left as `None`, the default will be used.
 268     timeout: The maximum amount of time to wait between checkpoints. If left as
 269       `None`, then the process will wait indefinitely.
 270     timeout_fn: Optional function to call after a timeout.  If the function
 271       returns True, then it means that no new checkpoints will be generated and
 272       the iterator will exit.  The function is called with no arguments.
 273     hooks: A list of additional `SessionRunHook` objects to pass during
 274       repeated evaluations.
 275 
 276   Returns:
 277     The value of `final_op` or `None` if `final_op` is `None`.
 278   """
 279   if summary_op == _USE_DEFAULT:
 280     summary_op = summary.merge_all()
 281 
 282   all_hooks = [evaluation.StopAfterNEvalsHook(num_evals),]
 283 
 284   if summary_op is not None:
 285     all_hooks.append(evaluation.SummaryAtEndHook(
 286         log_dir=logdir, summary_op=summary_op, feed_dict=summary_op_feed_dict))
 287 
 288   if hooks is not None:
 289     # Add custom hooks if provided.
 290     all_hooks.extend(hooks)
 291 
 292   saver = None
 293   if variables_to_restore is not None:
 294     saver = tf_saver.Saver(variables_to_restore)
 295 
 296   return evaluation.evaluate_repeatedly(
 297       checkpoint_dir,
 298       master=master,
 299       scaffold=monitored_session.Scaffold(
 300           init_op=initial_op, init_feed_dict=initial_op_feed_dict,
 301           init_fn=init_fn, saver=saver),
 302       eval_ops=eval_op,
 303       feed_dict=eval_op_feed_dict,
 304       final_ops=final_op,
 305       final_ops_feed_dict=final_op_feed_dict,
 306       eval_interval_secs=eval_interval_secs,
 307       hooks=all_hooks,
 308       config=session_config,
 309       max_number_of_evaluations=max_number_of_evaluations,
 310       timeout=timeout,
 311       timeout_fn=timeout_fn)
