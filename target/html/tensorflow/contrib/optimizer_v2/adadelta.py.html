  16 """Adadelta for TensorFlow."""
  17 from __future__ import absolute_import
  18 from __future__ import division
  19 from __future__ import print_function
  20 
  21 from tensorflow.contrib.optimizer_v2 import optimizer_v2
  22 from tensorflow.python.training import training_ops
  23 
  24 
  25 class AdadeltaOptimizer(optimizer_v2.OptimizerV2):
  26   """Optimizer that implements the Adadelta algorithm.
  27 
  28   See [M. D. Zeiler](http://arxiv.org/abs/1212.5701)
  29   ([pdf](http://arxiv.org/pdf/1212.5701v1.pdf))
  30   """
  31 
  32   def __init__(self, learning_rate=0.001, rho=0.95, epsilon=1e-8,
  33                use_locking=False, name="Adadelta"):
  34     """Construct a new Adadelta optimizer.
  35 
  36     Some of the args below are hyperparameters, where a hyperparameter is
  37     defined as a scalar Tensor, a regular Python value or a callable (which
  38     will be evaluated when `apply_gradients` is called) returning a scalar
  39     Tensor or a Python value.
  40 
  41     Args:
  42       learning_rate: A float hyperparameter. The learning rate.
  43         To match the exact form in the original paper use 1.0.
  44       rho: A float hyperparameter. The decay rate.
  45       epsilon: A float hyperparameter. A constant epsilon used to better
  46         condition the grad update.
  47       use_locking: If `True` use locks for update operations.
  48       name: Optional name prefix for the operations created when applying
  49         gradients.  Defaults to "Adadelta".
  50     """
  51     super(AdadeltaOptimizer, self).__init__(use_locking, name)
  52     self._set_hyper("learning_rate", learning_rate)
  53     self._set_hyper("rho", rho)
  54     self._set_hyper("epsilon", epsilon)
  55 
  56   def _create_vars(self, var_list, state):
  57     for v in var_list:
  58       state.zeros_slot(v, "accum")
  59       state.zeros_slot(v, "accum_update")
  60 
  61   def _apply_dense(self, grad, var, state):
  62     accum = state.get_slot(var, "accum")
  63     accum_update = state.get_slot(var, "accum_update")
  64     return training_ops.apply_adadelta(
  65         var,
  66         accum,
  67         accum_update,
  68         state.get_hyper("learning_rate", var.dtype.base_dtype),
  69         state.get_hyper("rho", var.dtype.base_dtype),
  70         state.get_hyper("epsilon", var.dtype.base_dtype),
  71         grad,
  72         use_locking=self._use_locking)
  73 
  74   def _resource_apply_dense(self, grad, var, state):
  75     accum = state.get_slot(var, "accum")
  76     accum_update = state.get_slot(var, "accum_update")
  77     return training_ops.resource_apply_adadelta(
  78         var.handle,
  79         accum.handle,
  80         accum_update.handle,
  81         state.get_hyper("learning_rate", var.dtype.base_dtype),
  82         state.get_hyper("rho", var.dtype.base_dtype),
  83         state.get_hyper("epsilon", var.dtype.base_dtype),
  84         grad,
  85         use_locking=self._use_locking)
  86 
  87   def _apply_sparse(self, grad, var, state):
  88     accum = state.get_slot(var, "accum")
  89     accum_update = state.get_slot(var, "accum_update")
  90     return training_ops.sparse_apply_adadelta(
  91         var,
  92         accum,
  93         accum_update,
  94         state.get_hyper("learning_rate", var.dtype.base_dtype),
  95         state.get_hyper("rho", var.dtype.base_dtype),
  96         state.get_hyper("epsilon", var.dtype.base_dtype),
  97         grad.values,
  98         grad.indices,
  99         use_locking=self._use_locking)
 100 
 101   def _resource_apply_sparse(self, grad, var, indices, state):
 102     accum = state.get_slot(var, "accum")
 103     accum_update = state.get_slot(var, "accum_update")
 104     return training_ops.resource_sparse_apply_adadelta(
 105         var.handle,
 106         accum.handle,
 107         accum_update.handle,
 108         state.get_hyper("learning_rate", var.dtype.base_dtype),
 109         state.get_hyper("rho", var.dtype.base_dtype),
 110         state.get_hyper("epsilon", var.dtype.base_dtype),
 111         grad,
 112         indices,
 113         use_locking=self._use_locking)
