  16 """Momentum for TensorFlow."""
  17 from __future__ import absolute_import
  18 from __future__ import division
  19 from __future__ import print_function
  20 
  21 from tensorflow.contrib.optimizer_v2 import optimizer_v2
  22 from tensorflow.python.training import training_ops
  23 
  24 
  25 class MomentumOptimizer(optimizer_v2.OptimizerV2):
  26   """Optimizer that implements the Momentum algorithm.
  27 
  28   Computes (if `use_nesterov = False`):
  29 
  30   ```
  31   accumulation = momentum * accumulation + gradient
  32   variable -= learning_rate * accumulation
  33   ```
  34 
  35   Note that in the dense version of this algorithm, `accumulation` is updated
  36   and applied regardless of a gradient's value, whereas the sparse version (when
  37   the gradient is an `IndexedSlices`, typically because of `tf.gather` or an
  38   embedding) only updates variable slices and corresponding `accumulation` terms
  39   when that part of the variable was used in the forward pass.
  40   """
  41 
  42   def __init__(self, learning_rate, momentum,
  43                use_locking=False, name="Momentum", use_nesterov=False):
  44     """Construct a new Momentum optimizer.
  45 
  46     Some of the args below are hyperparameters, where a hyperparameter is
  47     defined as a scalar Tensor, a regular Python value or a callable (which
  48     will be evaluated when `apply_gradients` is called) returning a scalar
  49     Tensor or a Python value.
  50 
  51     Args:
  52       learning_rate: A float hyperparameter. The learning rate.
  53       momentum: A float hyperparameter. The momentum.
  54       use_locking: If `True` use locks for update operations.
  55       name: Optional name prefix for the operations created when applying
  56         gradients.  Defaults to "Momentum".
  57       use_nesterov: If `True` use Nesterov Momentum.
  58         See [Sutskever et al., 2013](
  59         http://jmlr.org/proceedings/papers/v28/sutskever13.pdf).
  60         This implementation always computes gradients at the value of the
  61         variable(s) passed to the optimizer. Using Nesterov Momentum makes the
  62         variable(s) track the values called `theta_t + mu*v_t` in the paper.
  63 
  64     @compatibility(eager)
  65     When eager execution is enabled, learning_rate and momentum can each be a
  66     callable that takes no arguments and returns the actual value to use. This
  67     can be useful for changing these values across different invocations of
  68     optimizer functions.
  69     @end_compatibility
  70     """
  71     super(MomentumOptimizer, self).__init__(use_locking, name)
  72     self._set_hyper("learning_rate", learning_rate)
  73     self._set_hyper("momentum", momentum)
  74     self._use_nesterov = use_nesterov
  75 
  76   def _create_vars(self, var_list, state):
  77     for v in var_list:
  78       state.zeros_slot(v, "momentum")
  79 
  80   def _apply_dense(self, grad, var, state):
  81     mom = state.get_slot(var, "momentum")
  82     return training_ops.apply_momentum(
  83         var,
  84         mom,
  85         state.get_hyper("learning_rate", var.dtype.base_dtype),
  86         grad,
  87         state.get_hyper("momentum", var.dtype.base_dtype),
  88         use_locking=self._use_locking,
  89         use_nesterov=self._use_nesterov).op
  90 
  91   def _resource_apply_dense(self, grad, var, state):
  92     mom = state.get_slot(var, "momentum")
  93     return training_ops.resource_apply_momentum(
  94         var.handle,
  95         mom.handle,
  96         state.get_hyper("learning_rate", var.dtype.base_dtype),
  97         grad,
  98         state.get_hyper("momentum", var.dtype.base_dtype),
  99         use_locking=self._use_locking,
 100         use_nesterov=self._use_nesterov)
 101 
 102   def _apply_sparse(self, grad, var, state):
 103     mom = state.get_slot(var, "momentum")
 104     return training_ops.sparse_apply_momentum(
 105         var,
 106         mom,
 107         state.get_hyper("learning_rate", var.dtype.base_dtype),
 108         grad.values,
 109         grad.indices,
 110         state.get_hyper("momentum", var.dtype.base_dtype),
 111         use_locking=self._use_locking,
 112         use_nesterov=self._use_nesterov).op
 113 
 114   def _resource_apply_sparse(self, grad, var, indices, state):
 115     mom = state.get_slot(var, "momentum")
 116     return training_ops.resource_sparse_apply_momentum(
 117         var.handle,
 118         mom.handle,
 119         state.get_hyper("learning_rate", var.dtype.base_dtype),
 120         grad,
 121         indices,
 122         state.get_hyper("momentum", var.dtype.base_dtype),
 123         use_locking=self._use_locking,
 124         use_nesterov=self._use_nesterov)
