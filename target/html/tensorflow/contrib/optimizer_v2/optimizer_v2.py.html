  43 class _OptimizableVariable(object):
  44   """Interface for abstracting over variables in the optimizers."""
  45 
  46   @abc.abstractmethod
  47   def target(self):
  48     """Returns the optimization target for this variable."""
  49     raise NotImplementedError("Calling an abstract method.")
  50 
  51   @abc.abstractmethod
  52   def update_op(self, optimizer, g, *args):
  53     """Returns the update ops for updating the variable."""
  54     raise NotImplementedError("Calling an abstract method.")
  55 
  56 
  57 class _RefVariableProcessor(_OptimizableVariable):
  58   """Processor for Variable."""
  59 
  60   def __init__(self, v):
  61     self._v = v
  62 
  63   def target(self):
  64     return self._v._ref()  # pylint: disable=protected-access
  65 
  66   def update_op(self, optimizer, g, *args):
  67     if isinstance(g, ops.Tensor):
  68       update_op = optimizer._apply_dense(g, self._v, *args)  # pylint: disable=protected-access
  69       if self._v.constraint is not None:
  70         with ops.control_dependencies([update_op]):
  71           return self._v.assign(self._v.constraint(self._v))
  72       else:
  73         return update_op
  74     else:
  75       assert isinstance(g, ops.IndexedSlices), ("Gradient ", g, " is neither a "
  76                                                 "tensor nor IndexedSlices.")
  77       if self._v.constraint is not None:
  78         raise RuntimeError(
  79             "Cannot use a constraint function on a sparse variable.")
  80       # pylint: disable=protected-access
  81       return optimizer._apply_sparse_duplicate_indices(g, self._v, *args)

  84 class _DenseReadResourceVariableProcessor(_OptimizableVariable):
  85   """Processor for dense ResourceVariables."""
  86 
  87   def __init__(self, v):
  88     self._v = v
  89 
  90   def target(self):
  91     return self._v
  92 
  93   def update_op(self, optimizer, g, *args):
  94     # pylint: disable=protected-access
  95     update_op = optimizer._resource_apply_dense(g, self._v.op.inputs[0], *args)
  96     if self._v.constraint is not None:
  97       with ops.control_dependencies([update_op]):
  98         return self._v.assign(self._v.constraint(self._v))
  99     else:
 100       return update_op

 103 class _DenseResourceVariableProcessor(_OptimizableVariable):
 104   """Processor for dense ResourceVariables."""
 105 
 106   def __init__(self, v):
 107     self._v = v
 108 
 109   def target(self):
 110     return self._v
 111 
 112   def update_op(self, optimizer, g, *args):
 113     # pylint: disable=protected-access
 114     if isinstance(g, ops.IndexedSlices):
 115       if self._v.constraint is not None:
 116         raise RuntimeError(
 117             "Cannot use a constraint function on a sparse variable.")
 118       return optimizer._resource_apply_sparse_duplicate_indices(
 119           g.values, self._v, g.indices, *args)
 120     update_op = optimizer._resource_apply_dense(g, self._v, *args)
 121     if self._v.constraint is not None:
 122       with ops.control_dependencies([update_op]):
 123         return self._v.assign(self._v.constraint(self._v))
 124     else:
 125       return update_op

 128 class _TensorProcessor(_OptimizableVariable):
 129   """Processor for ordinary Tensors.
 130 
 131   Even though a Tensor can't really be updated, sometimes it is useful to
 132   compute the gradients with respect to a Tensor using the optimizer. Updating
 133   the Tensor is, of course, unsupported.
 134   """
 135 
 136   def __init__(self, v):
 137     self._v = v
 138 
 139   def target(self):
 140     return self._v
 141 
 142   def update_op(self, optimizer, g, *args):
 143     raise NotImplementedError("Trying to update a Tensor ", self._v)
 144

 146 def _get_processor(v):
 147   """The processor of v."""
 148   if context.executing_eagerly():
 149     if isinstance(v, ops.Tensor):
 150       return _TensorProcessor(v)
 151     else:
 152       return _DenseResourceVariableProcessor(v)
 153   if v.op.type == "VarHandleOp":
 154     return _DenseResourceVariableProcessor(v)
 155   if isinstance(v, variables.Variable):
 156     return _RefVariableProcessor(v)
 157   if isinstance(v, ops.Tensor):
 158     return _TensorProcessor(v)
 159   raise NotImplementedError("Trying to optimize unsupported type ", v)

 162 def _var_key_v2(var):
 163   """Key for representing a primary variable, for looking up slots."""
 164   # pylint: disable=protected-access
 165   if hasattr(var, "_distributed_container"):
 166     distributed_container = var._distributed_container()
 167     assert distributed_container is not None
 168     if context.executing_eagerly():
 169       return distributed_container._unique_id
 170     return distributed_container._shared_name
 171   if context.executing_eagerly():
 172     return var._unique_id
 173   return var.op.name

 176 def _resolve(value, name):
 177   if callable(value):
 178     value = value()
 179   return ops.convert_to_tensor(value, name=name)
 180

 182 def _is_dynamic(value):
 183   """Returns true if __init__ arg `value` should be re-evaluated each step."""
 184   if callable(value): return True
 185   # Don't need to do anything special in graph mode, since dynamic values
 186   # will propagate correctly automatically.
 187   # TODO(josh11b): Add per-device caching across steps using variables for
 188   # truly static values once we add distributed support.
 189   if context.executing_eagerly() and isinstance(
 190       value, resource_variable_ops.ResourceVariable):
 191     return True
 192   return False

 195 class _OptimizerV2State(object):
 196   """Holds per-graph and per-step optimizer state.
 197 
 198   Use _init_with_static_hyper() to create the state for a graph, and then
 199   _copy_with_dynamic_hyper() to convert that to state for a particular step.
 200   The difference between the two is that the former only has hyper
 201   parameter values that are static and the latter also has values that
 202   can change every step (according to _is_dynamic()).
 203   """
 204 
 205   def __init__(self, op_name):
 206     self._op_name = op_name
 207 
 208   def _init_with_static_hyper(self, hyper):
 209     """Initialize a fresh state object from hyper dict."""
 210     # self._hyper contains a dict from name to a dict with the Tensor values.
 211     # This dict starts with a single item with key "None" with the hyper
 212     # parameter value converted to a Tensor. Other items have dtype keys
 213     # with that Tensor cast to that dtype.
 214     with ops.init_scope():
 215       self._hyper = {name: {None: ops.convert_to_tensor(value, name=name)}
 216                      for name, (dynamic, value) in sorted(hyper.items())
 217                      if not dynamic}
 218     self._slots = {}
 219     self._non_slot_dict = {}
 220     # Extra state to help Optimizers implement Checkpointable. Holds information
 221     # about variables which will be restored as soon as they're created.
 222     self._deferred_dependencies = {}  # Non-slot variables
 223     self._deferred_slot_restorations = {}  # Slot variables
 224 
 225   def _copy_with_dynamic_hyper(self, hyper, distribution, non_slot_devices):
 226     """Create a new state object for a particular step."""
 227     ret = _OptimizerV2State(self._op_name)
 228     # pylint: disable=protected-access
 229     ret._slots = self._slots
 230     ret._non_slot_dict = self._non_slot_dict
 231     ret._deferred_dependencies = self._deferred_dependencies
 232     ret._deferred_slot_restorations = self._deferred_slot_restorations
 233     ret._hyper = {name: {None: _resolve(value, name)}
 234                   for name, (dynamic, value) in sorted(hyper.items())
 235                   if dynamic}
 236     ret._hyper.update(self._hyper)
 237     ret._non_slot_devices = non_slot_devices
 238     ret._distribution = distribution
 239     return ret
 240 
 241   def _variables(self):
 242     """Returns a list of all variables held by self."""
 243     optimizer_variables = list(self._non_slot_dict.values())
 244     for variable_dict in self._slots.values():
 245       for slot_for_variable in variable_dict.values():
 246         optimizer_variables.append(slot_for_variable)
 247     # Sort variables by name so that the return is deterministic.
 248     return sorted(optimizer_variables, key=lambda v: v.name)
 249 
 250   def _slot_dict(self, slot_name):
 251     """Returns a dict for caching slots created under the given name.
 252 
 253     Args:
 254       slot_name: Name for the slot.
 255 
 256     Returns:
 257       A dict that maps primary `Variable` objects to the slot created
 258       for that variable, under the given slot name.
 259     """
 260     named_slots = self._slots.get(slot_name, None)
 261     if named_slots is None:
 262       named_slots = {}
 263       self._slots[slot_name] = named_slots
 264     return named_slots
 265 
 266   def create_slot(self, var, val, slot_name, optional_op_name=None):
 267     """Find or create a slot for a variable.
 268 
 269     Args:
 270       var: A `Variable` object.
 271       val: A `Tensor`.  The initial value of the slot.
 272       slot_name: Name for the slot.
 273       optional_op_name: Name to use when scoping the Variable that
 274         needs to be created for the slot.
 275 
 276     Returns:
 277       A `Variable` object.
 278     """
 279     named_slots = self._slot_dict(slot_name)
 280     var_key = _var_key_v2(var)
 281     if var_key not in named_slots:
 282       new_slot_variable = slot_creator.create_slot(
 283           var, val, optional_op_name or self._op_name)
 284       self._restore_slot_variable(
 285           slot_name=slot_name, variable=var,
 286           slot_variable=new_slot_variable)
 287       named_slots[var_key] = new_slot_variable
 288     return named_slots[var_key]
 289 
 290   def create_slot_with_initializer(self, var, initializer, shape, dtype,
 291                                    slot_name, optional_op_name=None):
 292     """Find or create a slot for a variable, using an Initializer.
 293 
 294     Args:
 295       var: A `Variable` object.
 296       initializer: An `Initializer`.  The initial value of the slot.
 297       shape: Shape of the initial value of the slot.
 298       dtype: Type of the value of the slot.
 299       slot_name: Name for the slot.
 300       optional_op_name: Name to use when scoping the Variable that
 301         needs to be created for the slot.
 302 
 303     Returns:
 304       A `Variable` object.
 305     """
 306     named_slots = self._slot_dict(slot_name)
 307     var_key = _var_key_v2(var)
 308     if var_key not in named_slots:
 309       new_slot_variable = slot_creator.create_slot_with_initializer(
 310           var, initializer, shape, dtype, optional_op_name or self._op_name)
 311       self._restore_slot_variable(
 312           slot_name=slot_name, variable=var,
 313           slot_variable=new_slot_variable)
 314       named_slots[var_key] = new_slot_variable
 315     return named_slots[var_key]
 316

 320     Args:
 321       var: A `Variable` object.

 340   def _create_or_restore_slot_variable(
 341       self, slot_variable_position, slot_name, variable,
 342       optional_op_name=None):
 343     """Restore a slot variable's value, possibly creating it.
 344 
 345     Called when a variable which has an associated slot variable is created or
 346     restored. When executing eagerly, we create the slot variable with a
 347     restoring initializer.
 348 
 349     No new variables are created when graph building. Instead,
 350     _restore_slot_variable catches these after normal creation and adds restore
 351     ops to the graph. This method is nonetheless important when graph building
 352     for the case when a slot variable has already been created but `variable`
 353     has just been added to a dependency graph (causing us to realize that the
 354     slot variable needs to be restored).
 355 
 356     Args:
 357       slot_variable_position: A `checkpointable._CheckpointPosition` object
 358         indicating the slot variable `Checkpointable` object to be restored.
 359       slot_name: The name of this `Optimizer`'s slot to restore into.
 360       variable: The variable object this slot is being created for.
 361       optional_op_name: Name to use when scoping the Variable that
 362         needs to be created for the slot.
 363     """
 364     slot_variable = self.get_slot(var=variable, name=slot_name)
 365     if (slot_variable is None and context.executing_eagerly() and
 366         slot_variable_position.is_simple_variable()
 367         # Defer slot variable creation if there is an active variable creator
 368         # scope. Generally we'd like to eagerly create/restore slot variables
 369         # when possible, but this may mean that scopes intended to catch
 370         # `variable` also catch its eagerly created slot variable
 371         # unintentionally (specifically make_template would add a dependency on
 372         # a slot variable if not for this case). Deferring is mostly harmless
 373         # (aside from double initialization), and makes variable creator scopes
 374         # behave the same way they do when graph building.
 375         and not ops.get_default_graph()._variable_creator_stack):  # pylint: disable=protected-access
 376       initializer = checkpointable.CheckpointInitialValue(
 377           checkpoint_position=slot_variable_position)
 378       slot_variable = self.create_slot(
 379           var=variable,
 380           val=initializer,
 381           slot_name=slot_name,
 382           optional_op_name=optional_op_name)
 383       # Optimizers do not have unconditional dependencies on their slot
 384       # variables (nor do any other objects). They are only saved if the
 385       # variables they were created for are also saved.
 386     if slot_variable is not None:
 387       # If we've either made this slot variable, or if we've pulled out an
 388       # existing slot variable, we should restore it.
 389       slot_variable_position.restore(slot_variable)
 390     else:
 391       # We didn't make the slot variable. Defer restoring until it gets created
 392       # normally. We keep a list rather than the one with the highest restore
 393       # UID in case slot variables have their own dependencies, in which case
 394       # those could differ between restores.
 395       variable_key = _var_key_v2(variable)
 396       self._deferred_slot_restorations.setdefault(
 397           slot_name, {}).setdefault(variable_key, []).append(
 398               slot_variable_position)
 399 
 400   def get_slot(self, var, name):
 401     """Return a slot named `name` created for `var` by the Optimizer.
 402 
 403     Some `Optimizer` subclasses use additional variables.  For example
 404     `Momentum` and `Adagrad` use variables to accumulate updates.  This method
 405     gives access to these `Variable` objects if for some reason you need them.
 406 
 407     Use `get_slot_names()` to get the list of slot names created by the
 408     `Optimizer`.
 409 
 410     Args:
 411       var: A variable passed to `minimize()` or `apply_gradients()`.
 412       name: A string.
 413 
 414     Returns:
 415       The `Variable` for the slot if it was created, `None` otherwise.
 416     """
 417     named_slots = self._slots.get(name, None)
 418     if not named_slots:
 419       return None
 420     return named_slots.get(_var_key_v2(var), None)
 421 
 422   def get_slot_names(self):
 423     """Return a list of the names of slots created by the `Optimizer`.
 424 
 425     See `get_slot()`.
 426 
 427     Returns:
 428       A list of strings.
 429     """
 430     return sorted(self._slots.keys())
 431 
 432   def create_non_slot(self, initial_value, name, colocate_with=None):
 433     """Add an extra variable, not associated with a slot."""
 434     v = self._non_slot_dict.get(name, None)
 435     if v is None:
 436       if colocate_with is None: colocate_with = self._non_slot_devices
 437       with self._distribution.colocate_vars_with(colocate_with):
 438         # TODO(josh11b): Use get_variable() except for the legacy Adam use case.
 439         v = variable_scope.variable(initial_value, name=name, trainable=False)
 440       self._non_slot_dict[name] = v
 441       deferred_dependencies_list = self._deferred_dependencies.pop(name, ())
 442       for checkpoint_position in sorted(
 443           deferred_dependencies_list,
 444           key=lambda restore: restore.checkpoint.restore_uid,
 445           reverse=True):
 446         checkpoint_position.restore(v)
 447     return v
 448 
 449   def _restore_slot_variable(self, slot_name, variable, slot_variable):
 450     """Restore a newly created slot variable's value."""
 451     variable_key = _var_key_v2(variable)
 452     deferred_restorations = self._deferred_slot_restorations.get(
 453         slot_name, {}).pop(variable_key, [])
 454     # Iterate over restores, highest restore UID first to minimize the number
 455     # of assignments.
 456     deferred_restorations.sort(key=lambda position: position.restore_uid,
 457                                reverse=True)
 458     for checkpoint_position in deferred_restorations:
 459       checkpoint_position.restore(slot_variable)
 460 
 461   def get_non_slot(self, name):
 462     """Returns the non-slot variable identified by `name`."""
 463     return self._non_slot_dict.get(name, None)
 464 
 465   def get_hyper(self, name, dtype=None):
 466     """Returns the `name` hyper parameter, optionally cast to `dtype`."""
 467     dtype_dict = self._hyper[name]
 468     # Do we have the value cast to dtype already cached? This should always
 469     # succeed when dtype is None.
 470     if dtype in dtype_dict:
 471       return dtype_dict[dtype]
 472     # Not cached, cast to dtype and save the result in the cache.
 473     result = math_ops.cast(dtype_dict[None], dtype)
 474     dtype_dict[dtype] = result
 475     return result

 478 class OptimizerV2(optimizer_v1.Optimizer):
 479   """Updated base class for optimizers.
 480 
 481   This class defines the API to add Ops to train a model.  You never use this
 482   class directly, but instead instantiate one of its subclasses such as
 483   `GradientDescentOptimizer`, `AdagradOptimizer`, or `MomentumOptimizer`.
 484 
 485   ### Usage
 486 
 487   ```python
 488   # Create an optimizer with the desired parameters.
 489   opt = GradientDescentOptimizer(learning_rate=0.1)
 490   # Add Ops to the graph to minimize a cost by updating a list of variables.
 491   # "cost" is a Tensor, and the list of variables contains tf.Variable
 492   # objects.
 493   opt_op = opt.minimize(cost, var_list=<list of variables>)
 494   ```
 495 
 496   In the training program you will just have to run the returned Op.
 497 
 498   ```python
 499   # Execute opt_op to do one step of training:
 500   opt_op.run()
 501   ```
 502 
 503   ### Processing gradients before applying them.
 504 
 505   Calling `minimize()` takes care of both computing the gradients and
 506   applying them to the variables.  If you want to process the gradients
 507   before applying them you can instead use the optimizer in three steps:
 508 
 509   1.  Compute the gradients with `compute_gradients()`.
 510   2.  Process the gradients as you wish.
 511   3.  Apply the processed gradients with `apply_gradients()`.
 512 
 513   Example:
 514 
 515   ```python
 516   # Create an optimizer.
 517   opt = GradientDescentOptimizer(learning_rate=0.1)
 518 
 519   # Compute the gradients for a list of variables.
 520   grads_and_vars = opt.compute_gradients(loss, <list of variables>)
 521 
 522   # grads_and_vars is a list of tuples (gradient, variable).  Do whatever you
 523   # need to the 'gradient' part, for example cap them, etc.
 524   capped_grads_and_vars = [(MyCapper(gv[0]), gv[1]) for gv in grads_and_vars]
 525 
 526   # Ask the optimizer to apply the capped gradients.
 527   opt.apply_gradients(capped_grads_and_vars)
 528   ```
 529 
 530   ### Gating Gradients
 531 
 532   Both `minimize()` and `compute_gradients()` accept a `gate_gradients`
 533   argument that controls the degree of parallelism during the application of
 534   the gradients.
 535 
 536   The possible values are: `GATE_NONE`, `GATE_OP`, and `GATE_GRAPH`.
 537 
 538   <b>`GATE_NONE`</b>: Compute and apply gradients in parallel.  This provides
 539   the maximum parallelism in execution, at the cost of some non-reproducibility
 540   in the results.  For example the two gradients of `matmul` depend on the input
 541   values: With `GATE_NONE` one of the gradients could be applied to one of the
 542   inputs _before_ the other gradient is computed resulting in non-reproducible
 543   results.
 544 
 545   <b>`GATE_OP`</b>: For each Op, make sure all gradients are computed before
 546   they are used.  This prevents race conditions for Ops that generate gradients
 547   for multiple inputs where the gradients depend on the inputs.
 548 
 549   <b>`GATE_GRAPH`</b>: Make sure all gradients for all variables are computed
 550   before any one of them is used.  This provides the least parallelism but can
 551   be useful if you want to process all gradients before applying any of them.
 552 
 553   ### Slots
 554 
 555   Some optimizer subclasses, such as `MomentumOptimizer` and `AdagradOptimizer`
 556   allocate and manage additional variables associated with the variables to
 557   train.  These are called <i>Slots</i>.  Slots have names and you can ask the
 558   optimizer for the names of the slots that it uses.  Once you have a slot name
 559   you can ask the optimizer for the variable it created to hold the slot value.
 560 
 561   This can be useful if you want to log debug a training algorithm, report stats
 562   about the slots, etc.
 563 
 564   ### Non-slot variables
 565 
 566   Some optimizer subclasses, such as `AdamOptimizer` have variables that
 567   are not associated with the variables to train, just the step itself.
 568 
 569   ### Hyper parameters
 570 
 571   These are arguments passed to the optimizer subclass constructor
 572   (the `__init__` method), and then passed to `self._set_hyper()`.
 573   They can be either regular Python values (like 1.0), tensors, or
 574   callables. If they are callable, the callable will be called during
 575   `apply_gradients()` to get the value for the hyper parameter.
 576 
 577   ### State
 578 
 579   Internal methods are passed a `state` argument with the correct
 580   values to use for the slot and non-slot variables, and the hyper
 581   parameters.
 582   """
 583 
 584   # Values for gate_gradients.
 585   GATE_NONE = 0
 586   GATE_OP = 1
 587   GATE_GRAPH = 2

 589   def __init__(self, use_locking, name):
 590     """Create a new Optimizer.
 591 
 592     This must be called by the constructors of subclasses.
 593     Note that Optimizer instances should not bind to a single graph,
 594     and so shouldn't keep Tensors as member variables. Generally
 595     you should be able to use the _set_hyper()/state.get_hyper()
 596     facility instead.
 597 
 598     Args:
 599       use_locking: Bool. If True apply use locks to prevent concurrent updates
 600         to variables.
 601       name: A non-empty string.  The name to use for accumulators created
 602         for the optimizer.
 603 
 604     Raises:
 605       ValueError: If name is malformed.
 606       RuntimeError: If _create_slots has been overridden instead of
 607           _create_vars.
 608     """
 609     # Note: We intentionally don't call parent __init__.
 610 
 611     # Optimizer._create_slots was replaced by _create_vars in OptimizerV2.
 612     if (self.__class__._create_slots.__code__ is not  # pylint: disable=protected-access
 613         OptimizerV2._create_slots.__code__):
 614       raise RuntimeError("Override _create_vars instead of _create_slots when "
 615                          "descending from OptimizerV2 (class %s)" %
 616                          self.__class__.__name__)
 617     if not name:
 618       raise ValueError("Must specify the optimizer name")
 619 
 620     self._use_locking = use_locking
 621     self._name = name
 622     # Map from graph_key to state for that graph. We use the graph_key
 623     # since it works in both eager and graph mode, and gives the outer
 624     # graph inside functions.
 625     tower_context = distribution_strategy_context.get_tower_context()
 626     if tower_context is None:
 627       # In a cross-tower context for a DistributionStrategy, which means
 628       # only one Optimizer will be created, not one per tower.
 629       self._per_graph_state = {}
 630     else:
 631       # We use get_tower_context().merge_call() to get a single dict
 632       # shared across all model replicas when running with a
 633       # DistributionStrategy.
 634       self._per_graph_state = tower_context.merge_call(lambda _: {})
 635 
 636     # Hyper parameters, and whether they should be re-evaluated every step.
 637     self._hyper = {}
 638 
 639   def _set_hyper(self, name, value):
 640     self._hyper[name] = (_is_dynamic(value), value)

 642   def minimize(self, loss, global_step=None, var_list=None,
 643                gate_gradients=GATE_OP, aggregation_method=None,
 644                colocate_gradients_with_ops=False, name=None,
 645                grad_loss=None, stop_gradients=None,
 646                scale_loss_by_num_towers=None):
 647     """Add operations to minimize `loss` by updating `var_list`.
 648 
 649     This method simply combines calls `compute_gradients()` and
 650     `apply_gradients()`. If you want to process the gradient before applying
 651     them call `compute_gradients()` and `apply_gradients()` explicitly instead
 652     of using this function.
 653 
 654     Args:
 655       loss: A `Tensor` containing the value to minimize.
 656       global_step: Optional `Variable` to increment by one after the
 657         variables have been updated.
 658       var_list: Optional list or tuple of `Variable` objects to update to
 659         minimize `loss`.  Defaults to the list of variables collected in
 660         the graph under the key `GraphKeys.TRAINABLE_VARIABLES`.
 661       gate_gradients: How to gate the computation of gradients.  Can be
 662         `GATE_NONE`, `GATE_OP`, or  `GATE_GRAPH`.
 663       aggregation_method: Specifies the method used to combine gradient terms.
 664         Valid values are defined in the class `AggregationMethod`.
 665       colocate_gradients_with_ops: If True, try colocating gradients with
 666         the corresponding op.
 667       name: Optional name for the returned operation.
 668       grad_loss: Optional. A `Tensor` holding the gradient computed for `loss`.
 669       stop_gradients: Optional. A Tensor or list of tensors not to differentiate
 670         through.
 671       scale_loss_by_num_towers: Optional boolean. If true, scale the loss
 672         down by the number of towers. By default, auto-detects whether this
 673         is needed.
 674 
 675     Returns:
 676       An Operation that updates the variables in `var_list`.  If `global_step`
 677       was not `None`, that operation also increments `global_step`.
 678 
 679     Raises:
 680       ValueError: If some of the variables are not `Variable` objects.
 681 
 682     @compatibility(eager)
 683     When eager execution is enabled, `loss` should be a Python function that
 684     takes elements of `var_list` as arguments and computes the value to be
 685     minimized. If `var_list` is None, `loss` should take no arguments.
 686     Minimization (and gradient computation) is done with respect to the
 687     elements of `var_list` if not None, else with respect to any trainable
 688     variables created during the execution of the `loss` function.
 689     `gate_gradients`, `aggregation_method`, `colocate_gradients_with_ops` and
 690     `grad_loss` are ignored when eager execution is enabled.
 691     @end_compatibility
 692     """
 693     grads_and_vars = self.compute_gradients(
 694         loss, var_list=var_list, gate_gradients=gate_gradients,
 695         aggregation_method=aggregation_method,
 696         colocate_gradients_with_ops=colocate_gradients_with_ops,
 697         grad_loss=grad_loss, stop_gradients=stop_gradients,
 698         scale_loss_by_num_towers=scale_loss_by_num_towers)
 699 
 700     vars_with_grad = [v for g, v in grads_and_vars if g is not None]
 701     if not vars_with_grad:
 702       raise ValueError(
 703           "No gradients provided for any variable, check your graph for ops"
 704           " that do not support gradients, between variables %s and loss %s." %
 705           ([str(v) for _, v in grads_and_vars], loss))
 706 
 707     return self.apply_gradients(grads_and_vars, global_step=global_step,
 708                                 name=name)
 709 
 710   def compute_gradients(self, loss, var_list=None,
 711                         gate_gradients=GATE_OP,
 712                         aggregation_method=None,
 713                         colocate_gradients_with_ops=False,
 714                         grad_loss=None, stop_gradients=None,
 715                         scale_loss_by_num_towers=None):
 716     """Compute gradients of `loss` for the variables in `var_list`.
 717 
 718     This is the first part of `minimize()`.  It returns a list
 719     of (gradient, variable) pairs where "gradient" is the gradient
 720     for "variable".  Note that "gradient" can be a `Tensor`, an
 721     `IndexedSlices`, or `None` if there is no gradient for the
 722     given variable.
 723 
 724     Args:
 725       loss: A Tensor containing the value to minimize or a callable taking
 726         no arguments which returns the value to minimize. When eager execution
 727         is enabled it must be a callable.
 728       var_list: Optional list or tuple of `tf.Variable` to update to minimize
 729         `loss`.  Defaults to the list of variables collected in the graph
 730         under the key `GraphKeys.TRAINABLE_VARIABLES`.
 731       gate_gradients: How to gate the computation of gradients.  Can be
 732         `GATE_NONE`, `GATE_OP`, or `GATE_GRAPH`.
 733       aggregation_method: Specifies the method used to combine gradient terms.
 734         Valid values are defined in the class `AggregationMethod`.
 735       colocate_gradients_with_ops: If True, try colocating gradients with
 736         the corresponding op.
 737       grad_loss: Optional. A `Tensor` holding the gradient computed for `loss`.
 738       stop_gradients: Optional. A Tensor or list of tensors not to differentiate
 739         through.
 740       scale_loss_by_num_towers: Optional boolean. If true, scale the loss
 741         down by the number of towers. By default, auto-detects whether this
 742         is needed.
 743 
 744     Returns:
 745       A list of (gradient, variable) pairs. Variable is always present, but
 746       gradient can be `None`.
 747 
 748     Raises:
 749       TypeError: If `var_list` contains anything else than `Variable` objects.
 750       ValueError: If some arguments are invalid.
 751       RuntimeError: If called with eager execution enabled and `loss` is
 752         not callable.
 753 
 754     @compatibility(eager)
 755     When eager execution is enabled, `gate_gradients`, `aggregation_method`,
 756     and `colocate_gradients_with_ops` are ignored.
 757     @end_compatibility
 758     """
 759     # TODO(josh11b): Test that we handle weight decay in a reasonable way.
 760     if callable(loss):
 761       with backprop.GradientTape() as tape:
 762         if var_list is not None:
 763           tape.watch(var_list)
 764         loss_value = loss()
 765 
 766         # Scale loss for number of towers (callable-loss case). In this case,
 767         # we have to be careful to call distribute_lib.get_loss_reduction()
 768         # *after* loss() is evaluated, so we know what loss reduction it uses.
 769         if scale_loss_by_num_towers is None:
 770           scale_loss_by_num_towers = (
 771               distribute_lib.get_loss_reduction() ==
 772               variable_scope.VariableAggregation.MEAN)
 773         if scale_loss_by_num_towers:
 774           num_towers = distribution_strategy_context.get_distribution_strategy(
 775           ).num_towers
 776           if num_towers > 1:
 777             loss_value *= 1. / num_towers
 778 
 779       if var_list is None:
 780         var_list = tape.watched_variables()
 781       grads = tape.gradient(loss_value, var_list, grad_loss)
 782       return list(zip(grads, var_list))
 783     if context.executing_eagerly():
 784       raise RuntimeError(
 785           "`loss` passed to Optimizer.compute_gradients should "
 786           "be a function when eager execution is enabled.")
 787 
 788     # Scale loss for number of towers (non-callable-loss case).
 789     if scale_loss_by_num_towers is None:
 790       scale_loss_by_num_towers = (
 791           distribute_lib.get_loss_reduction() ==
 792           variable_scope.VariableAggregation.MEAN)
 793     if scale_loss_by_num_towers:
 794       num_towers = distribution_strategy_context.get_distribution_strategy(
 795       ).num_towers
 796       if num_towers > 1:
 797         loss *= 1. / num_towers
 798 
 799     if gate_gradients not in [optimizer_v1.Optimizer.GATE_NONE,
 800                               optimizer_v1.Optimizer.GATE_OP,
 801                               optimizer_v1.Optimizer.GATE_GRAPH]:
 802       raise ValueError("gate_gradients must be one of: Optimizer.GATE_NONE, "
 803                        "Optimizer.GATE_OP, Optimizer.GATE_GRAPH.  Not %s" %
 804                        gate_gradients)
 805     self._assert_valid_dtypes([loss])
 806     if grad_loss is not None:
 807       self._assert_valid_dtypes([grad_loss])
 808     if var_list is None:
 809       var_list = (
 810           variables.trainable_variables() +
 811           ops.get_collection(ops.GraphKeys.TRAINABLE_RESOURCE_VARIABLES))
 812     else:
 813       var_list = nest.flatten(var_list)
 814     # pylint: disable=protected-access
 815     var_list += ops.get_collection(ops.GraphKeys._STREAMING_MODEL_PORTS)
 816     # pylint: enable=protected-access
 817     processors = [_get_processor(v) for v in var_list]
 818     if not var_list:
 819       raise ValueError("No variables to optimize.")
 820     var_refs = [p.target() for p in processors]
 821     grads = gradients.gradients(
 822         loss, var_refs, grad_ys=grad_loss,
 823         gate_gradients=(gate_gradients == optimizer_v1.Optimizer.GATE_OP),
 824         aggregation_method=aggregation_method,
 825         colocate_gradients_with_ops=colocate_gradients_with_ops,
 826         stop_gradients=stop_gradients)
 827     if gate_gradients == optimizer_v1.Optimizer.GATE_GRAPH:
 828       grads = control_flow_ops.tuple(grads)
 829     grads_and_vars = list(zip(grads, var_list))
 830     self._assert_valid_dtypes(
 831         [v for g, v in grads_and_vars
 832          if g is not None and v.dtype != dtypes.resource])
 833     return grads_and_vars
 834 
 835   def apply_gradients(self, grads_and_vars, global_step=None, name=None):
 836     """Apply gradients to variables.
 837 
 838     This is the second part of `minimize()`. It returns an `Operation` that
 839     applies gradients.
 840 
 841     Args:
 842       grads_and_vars: List of (gradient, variable) pairs as returned by
 843         `compute_gradients()`.
 844       global_step: Optional `Variable` to increment by one after the
 845         variables have been updated.
 846       name: Optional name for the returned operation.  Default to the
 847         name passed to the `Optimizer` constructor.
 848 
 849     Returns:
 850       An `Operation` that applies the specified gradients. If `global_step`
 851       was not None, that operation also increments `global_step`.
 852 
 853     Raises:
 854       TypeError: If `grads_and_vars` is malformed.
 855       ValueError: If none of the variables have gradients.
 856     """
 857     # This is a default implementation of apply_gradients() that can be shared
 858     # by most optimizers.  It relies on the subclass implementing the following
 859     # methods: _create_vars(), _prepare(), _apply_dense(), and _apply_sparse().
 860 
 861     # Filter out variables with gradients of `None`.
 862     grads_and_vars = tuple(grads_and_vars)  # Make sure repeat iteration works.
 863     if not grads_and_vars:
 864       raise ValueError("No variables provided.")
 865     filtered = tuple((g, v) for (g, v) in grads_and_vars if g is not None)
 866     if not filtered:
 867       raise ValueError("No gradients provided for any variable: %s." %
 868                        ([str(v) for _, v in grads_and_vars],))
 869     return distribution_strategy_context.get_tower_context().merge_call(
 870         self._distributed_apply, filtered, global_step=global_step, name=name)

 872   def _get_or_create_state(self, var_list=None):
 873     """Either looks up or creates `_OptimizerV2State`.
 874 
 875     If any variables are available, they should be passed via the `var_list`
 876     argument, and these will be used to determine the graph to create/retrieve
 877     state for. Otherwise the returned state is for the current default graph.
 878 
 879     Args:
 880       var_list: A list of variables to extract a graph from.
 881 
 882     Returns:
 883       An `_OptimizerV2State` object.
 884     """
 885     # Determine the graph_key from the current graph.
 886     eager_execution = context.executing_eagerly()
 887     if eager_execution or var_list is None:
 888       graph = ops.get_default_graph()
 889     else:
 890       graph = ops._get_graph_from_inputs(var_list)  # pylint: disable=protected-access
 891     assert graph is not None
 892     graph_key = graph._graph_key  # pylint: disable=protected-access
 893 
 894     # Get the per graph state by looking up the graph_key.
 895     if graph_key in self._per_graph_state:
 896       per_graph_state = self._per_graph_state[graph_key]
 897     else:
 898       per_graph_state = _OptimizerV2State(self._name)
 899       per_graph_state._init_with_static_hyper(self._hyper)  # pylint: disable=protected-access
 900       self._per_graph_state[graph_key] = per_graph_state
 901     return per_graph_state

 903   def _distributed_apply(self, distribution, grads_and_vars, global_step, name):
 904     """`apply_gradients` for use with a `DistributionStrategy`."""
 905     reduced_grads = distribution.batch_reduce(
 906         variable_scope.VariableAggregation.SUM, grads_and_vars)
 907     var_list = [v for _, v in grads_and_vars]
 908     grads_and_vars = zip(reduced_grads, var_list)
 909 
 910     unwrapped_var_list = [x for v in var_list for x in distribution.unwrap(v)]
 911     eager_execution = context.executing_eagerly()
 912     if eager_execution:
 913       # Give a clear error in this case instead of "name not supported
 914       # for Eager Tensors" when we compute non_slot_devices.
 915       for v in unwrapped_var_list:
 916         if isinstance(v, ops.Tensor):
 917           raise NotImplementedError("Trying to update a Tensor ", v)
 918 
 919     with ops.name_scope(name, self._name) as name:
 920       per_graph_state = self._get_or_create_state(var_list=unwrapped_var_list)
 921       # Include the current value of any dynamic hyper parameters in `state`.
 922       non_slot_devices = distribution.non_slot_devices(var_list)
 923       state = per_graph_state._copy_with_dynamic_hyper(  # pylint: disable=protected-access
 924           self._hyper, distribution, non_slot_devices)
 925 
 926     # Create any slot and non-slot variables we need in `state`.
 927     with ops.init_scope():
 928       self._create_vars(var_list, state)
 929 
 930     with ops.name_scope(name):  # Re-enter name_scope created above
 931       # Give the child class a chance to do something before we start
 932       # applying gradients.
 933       self._prepare(state)
 934 
 935       def update(v, g):
 936         """Update variable `v` using gradient `g`."""
 937         assert v is not None
 938 
 939         # Convert the grad to Tensor or IndexedSlices if necessary, and
 940         # look up a processor for each variable's type.
 941         try:
 942           g = ops.convert_to_tensor_or_indexed_slices(g)
 943         except TypeError:
 944           raise TypeError(
 945               "Gradient must be convertible to a Tensor"
 946               " or IndexedSlices, or None: %s" % g)
 947         if not isinstance(g, (ops.Tensor, ops.IndexedSlices)):
 948           raise TypeError(
 949               "Gradient must be a Tensor, IndexedSlices, or None: %s" % g)
 950         processor = _get_processor(v)
 951 
 952         # We colocate all ops created in _apply_dense or _apply_sparse
 953         # on the same device as the variable.
 954         # TODO(apassos): figure out how to get the variable name here.
 955         scope_name = "" if eager_execution else v.op.name
 956         # device_policy is set because non-mirrored tensors will be read in
 957         # `update_op`.
 958         # TODO(josh11b): Make different state objects for each device to
 959         # avoid needing to set the device_policy.
 960         with ops.name_scope("update_" + scope_name), \
 961             context.context().device_policy(context.DEVICE_PLACEMENT_SILENT):
 962           return processor.update_op(self, g, state)
 963 
 964       # Use the processors to update the variables.
 965       update_ops = []
 966       for grad, var in grads_and_vars:
 967         update_ops.extend(distribution.update(var, update, grad, grouped=False))
 968 
 969       # Give the child class a chance to do something after applying
 970       # gradients
 971       def finish():
 972         # TODO(josh11b): Make different state objects for each device to
 973         # avoid needing to set the device_policy.
 974         with context.context().device_policy(context.DEVICE_PLACEMENT_SILENT):
 975           return self._finish(state)
 976 
 977       update_ops = control_flow_ops.group(update_ops)
 978       with ops.control_dependencies([update_ops]):
 979         finish_updates = distribution.update_non_slot(
 980             non_slot_devices, finish, grouped=False)
 981       # We said grouped=False, which means finish_updates is always a list.
 982       # It will be [None] when finish() returns None.
 983       if finish_updates == [None]:
 984         finish_updates = [update_ops]
 985 
 986       # Update `global_step` (if any).
 987       if global_step is None:
 988         apply_updates = distribution.group(finish_updates, name=name)
 989       else:
 990         with ops.control_dependencies(finish_updates):
 991 
 992           def update_global_step(global_step, name):
 993             return global_step.assign_add(1, read_value=False, name=name)
 994 
 995           apply_updates = distribution.update(
 996               global_step, update_global_step, name)
 997 
 998       # Add the training op to the TRAIN_OP graph collection in graph mode.
 999       if not eager_execution:
1000         if isinstance(apply_updates, ops.Tensor):
1001           apply_updates = apply_updates.op
1002         train_op = ops.get_collection_ref(ops.GraphKeys.TRAIN_OP)
1003         if apply_updates not in train_op:
1004           train_op.append(apply_updates)
1005 
1006       return apply_updates

1114   def _resource_apply_sparse_duplicate_indices(
1115       self, grad, handle, indices, state):
1116     """Add ops to apply sparse gradients to `handle`, with repeated indices.

1118     Optimizers which override this method must deal with repeated indices. See
1119     the docstring of `_apply_sparse_duplicate_indices` for details. By default
1120     the correct behavior, to sum non-unique indices and their associated
1121     gradients, is enforced by first pre-processing `grad` and `indices` and
1122     passing them on to `_resource_apply_sparse`. Optimizers which deal correctly
1123     with duplicate indices may instead override this method to avoid the
1124     overhead of summing.
1125 
1126     Args:
1127       grad: a `Tensor` representing the gradient for the affected indices.
1128       handle: a `Tensor` of dtype `resource` which points to the variable
1129        to be updated.
1130       indices: a `Tensor` of integral type representing the indices for
1131        which the gradient is nonzero. Indices may be repeated.
1132       state: An object with `get_slot(var, name)`, `get_non_slot(self, name)`,
1133         and `get_hyper(name)` methods.
1134 
1135     Returns:
1136       An `Operation` which updates the value of the variable.
1137     """
1138     # pylint: disable=protected-access
1139     summed_grad, unique_indices = optimizer_v1._deduplicate_indexed_slices(
1140         values=grad, indices=indices)
1141     # pylint: enable=protected-access
1142     return self._resource_apply_sparse(
1143         summed_grad, handle, unique_indices, state)

1167   def _apply_sparse_duplicate_indices(self, grad, var, state):
1168     """Add ops to apply sparse gradients to `var`, with repeated sparse indices.
1169 
1170     Optimizers which override this method must deal with IndexedSlices objects
1171     such as the following:
1172 
1173       IndexedSlicesValue(values=[1, 1], indices=[0, 0], dense_shape=[1])
1174 
1175     The correct interpretation is:
1176 
1177       IndexedSlicesValue(values=[2], indices=[0], dense_shape=[1])
1178 
1179     Many optimizers deal incorrectly with repeated indices when updating based
1180     on sparse gradients (e.g. summing squares rather than squaring the sum, or
1181     applying momentum terms multiple times). Adding first is always the correct
1182     behavior, so this is enforced here by reconstructing the IndexedSlices to
1183     have only unique indices, then calling _apply_sparse.
1184 
1185     Optimizers which deal correctly with repeated indices may instead override
1186     this method to avoid the overhead of summing indices.
1187 
1188     Args:
1189       grad: `IndexedSlices`.
1190       var: A `Variable` object.
1191       state: An object with `get_slot(var, name)`, `get_non_slot(self, name)`,
1192         and `get_hyper(name)` methods.
1193 
1194     Returns:
1195       An `Operation`.
1196     """
1197     # pylint: disable=protected-access
1198     summed_values, unique_indices = optimizer_v1._deduplicate_indexed_slices(
1199         values=grad.values, indices=grad.indices)
1200     # pylint: enable=protected-access
1201     gradient_no_duplicate_indices = ops.IndexedSlices(
1202         indices=unique_indices,
1203         values=summed_values,
1204         dense_shape=grad.dense_shape)
1205     return self._apply_sparse(gradient_no_duplicate_indices, var, state)
