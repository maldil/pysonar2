  15 """RMSprop optimizer for Tensorflow.
  16 
  17 rmsprop algorithm [tieleman2012rmsprop]
  18 
  19 A detailed description of rmsprop.
  20 
  21 - maintain a moving (discounted) average of the square of gradients
  22 - divide gradient by the root of this average
  23 
  24 mean_square = decay * mean_square{t-1} + (1-decay) * gradient ** 2
  25 mom = momentum * mom{t-1} + learning_rate * g_t / sqrt(mean_square)
  26 delta = - mom
  27 
  28 This implementation of RMSProp uses plain momentum, not Nesterov momentum.
  29 
  30 The centered version additionally maintains a moving (discounted) average of the
  31 gradients, and uses that average to estimate the variance:
  32 
  33 mean_grad = decay * mean_square{t-1} + (1-decay) * gradient
  34 mean_square = decay * mean_square{t-1} + (1-decay) * gradient ** 2
  35 mom = momentum * mom{t-1} + learning_rate * g_t /
  36     sqrt(mean_square - mean_grad**2)
  37 delta = - mom
  38 """

  50 class RMSPropOptimizer(optimizer_v2.OptimizerV2):
  51   """Optimizer that implements the RMSProp algorithm.
  52 
  53   See the
  54   [paper](http://www.cs.toronto.edu/~tijmen/csc321/slides/lecture_slides_lec6.pdf).
  55   """
  56 
  57   def __init__(self,
  58                learning_rate,
  59                decay=0.9,
  60                momentum=0.0,
  61                epsilon=1e-10,
  62                use_locking=False,
  63                centered=False,
  64                name="RMSProp"):
  65     """Construct a new RMSProp optimizer.
  66 
  67     Note that in the dense implementation of this algorithm, variables and their
  68     corresponding accumulators (momentum, gradient moving average, square
  69     gradient moving average) will be updated even if the gradient is zero
  70     (i.e. accumulators will decay, momentum will be applied). The sparse
  71     implementation (used when the gradient is an `IndexedSlices` object,
  72     typically because of `tf.gather` or an embedding lookup in the forward pass)
  73     will not update variable slices or their accumulators unless those slices
  74     were used in the forward pass (nor is there an "eventual" correction to
  75     account for these omitted updates). This leads to more efficient updates for
  76     large embedding lookup tables (where most of the slices are not accessed in
  77     a particular graph execution), but differs from the published algorithm.
  78 
  79     Some of the args below are hyperparameters, where a hyperparameter is
  80     defined as a scalar Tensor, a regular Python value or a callable (which
  81     will be evaluated when `apply_gradients` is called) returning a scalar
  82     Tensor or a Python value.
  83 
  84     Args:
  85       learning_rate: A float hyperparameter. The learning rate.
  86       decay: A float hyperparameter. Discounting factor for the history/coming
  87         gradient.
  88       momentum: A float hyperparameter.
  89       epsilon: A float hyperparameter. Small value to initialize the average
  90         square gradient variable and avoid zero denominator.
  91       use_locking: If True use locks for update operation.
  92       centered: If True, gradients are normalized by the estimated variance of
  93         the gradient; if False, by the uncentered second moment. Setting this to
  94         True may help with training, but is slightly more expensive in terms of
  95         computation and memory. Defaults to False.
  96       name: Optional name prefix for the operations created when applying
  97         gradients. Defaults to "RMSProp".
  98     """
  99     super(RMSPropOptimizer, self).__init__(use_locking, name)
 100     self._set_hyper("learning_rate", learning_rate)
 101     self._set_hyper("decay", decay)
 102     self._set_hyper("momentum", momentum)
 103     self._set_hyper("epsilon", epsilon)
 104 
 105     self._centered = centered
 106 
 107   def _create_vars(self, var_list, state):
 108     for v in var_list:
 109       init_rms = state.get_hyper(
 110           "epsilon", v.dtype.base_dtype) * array_ops.ones_like(v)
 111       state.create_slot_with_initializer(v, init_rms, v.get_shape(),
 112                                          v.dtype.base_dtype, "rms")
 113       if self._centered:
 114         state.zeros_slot(v, "mg")
 115       state.zeros_slot(v, "momentum")
 116 
 117   def _apply_dense(self, grad, var, state):
 118     rms = state.get_slot(var, "rms")
 119     mom = state.get_slot(var, "momentum")
 120     if self._centered:
 121       mg = state.get_slot(var, "mg")
 122       return training_ops.apply_centered_rms_prop(
 123           var,
 124           mg,
 125           rms,
 126           mom,
 127           state.get_hyper("learning_rate", var.dtype.base_dtype),
 128           state.get_hyper("decay", var.dtype.base_dtype),
 129           state.get_hyper("momentum", var.dtype.base_dtype),
 130           # epsilon is now the rms initial value and is not added to the
 131           # denominator anymore, hence calling the kernel op with epsilon=0.
 132           0,
 133           grad,
 134           use_locking=self._use_locking).op
 135     else:
 136       return training_ops.apply_rms_prop(
 137           var,
 138           rms,
 139           mom,
 140           state.get_hyper("learning_rate", var.dtype.base_dtype),
 141           state.get_hyper("decay", var.dtype.base_dtype),
 142           state.get_hyper("momentum", var.dtype.base_dtype),
 143           0,
 144           grad,
 145           use_locking=self._use_locking).op
 146 
 147   def _resource_apply_dense(self, grad, var, state):
 148     rms = state.get_slot(var, "rms")
 149     mom = state.get_slot(var, "momentum")
 150     if self._centered:
 151       mg = state.get_slot(var, "mg")
 152       return training_ops.resource_apply_centered_rms_prop(
 153           var.handle,
 154           mg.handle,
 155           rms.handle,
 156           mom.handle,
 157           state.get_hyper("learning_rate", var.dtype.base_dtype),
 158           state.get_hyper("decay", var.dtype.base_dtype),
 159           state.get_hyper("momentum", var.dtype.base_dtype),
 160           0,
 161           grad,
 162           use_locking=self._use_locking)
 163     else:
 164       return training_ops.resource_apply_rms_prop(
 165           var.handle,
 166           rms.handle,
 167           mom.handle,
 168           state.get_hyper("learning_rate", var.dtype.base_dtype),
 169           state.get_hyper("decay", var.dtype.base_dtype),
 170           state.get_hyper("momentum", var.dtype.base_dtype),
 171           0,
 172           grad,
 173           use_locking=self._use_locking)
 174
