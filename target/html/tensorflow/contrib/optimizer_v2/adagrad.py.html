  16 """Adagrad optimizer for TensorFlow."""
  17 from __future__ import absolute_import
  18 from __future__ import division
  19 from __future__ import print_function
  20 
  21 from tensorflow.contrib.optimizer_v2 import optimizer_v2
  22 from tensorflow.python.ops import array_ops
  23 from tensorflow.python.ops import gen_array_ops
  24 from tensorflow.python.ops import init_ops
  25 from tensorflow.python.ops import math_ops
  26 from tensorflow.python.training import training_ops
  27 
  28 
  29 class AdagradOptimizer(optimizer_v2.OptimizerV2):
  30   """Optimizer that implements the Adagrad algorithm.
  31 
  32   See this [paper](http://www.jmlr.org/papers/volume12/duchi11a/duchi11a.pdf)
  33   or this
  34   [intro](https://ppasupat.github.io/a9online/uploads/proximal_notes.pdf).
  35   """
  36 
  37   def __init__(self, learning_rate, initial_accumulator_value=0.1,
  38                use_locking=False, name="Adagrad"):
  39     """Construct a new Adagrad optimizer.
  40 
  41     The learning_rate arg below is a hyperparameter, where a hyperparameter is
  42     defined as a scalar Tensor, a regular Python value or a callable (which
  43     will be evaluated when `apply_gradients` is called) returning a scalar
  44     Tensor or a Python value.
  45 
  46     Args:
  47       learning_rate: A float hyperparameter. The learning rate.
  48       initial_accumulator_value: A floating point value.
  49         Starting value for the accumulators, must be positive.
  50       use_locking: If `True` use locks for update operations.
  51       name: Optional name prefix for the operations created when applying
  52         gradients.  Defaults to "Adagrad".
  53 
  54     Raises:
  55       ValueError: If the `initial_accumulator_value` is invalid.
  56     """
  57     if initial_accumulator_value <= 0.0:
  58       raise ValueError("initial_accumulator_value must be positive: %s" %
  59                        initial_accumulator_value)
  60     super(AdagradOptimizer, self).__init__(use_locking, name)
  61     self._set_hyper("learning_rate", learning_rate)
  62 
  63     self._initial_accumulator_value = initial_accumulator_value
  64 
  65   def _create_vars(self, var_list, state):
  66     for v in var_list:
  67       dtype = v.dtype.base_dtype
  68       if v.get_shape().is_fully_defined():
  69         init = init_ops.constant_initializer(self._initial_accumulator_value,
  70                                              dtype=dtype)
  71       else:
  72         def init(v=v, dtype=dtype):
  73           # Use a Tensor instead of initializer if variable does not have
  74           # static shape.
  75           init_constant = gen_array_ops.fill(array_ops.shape(v),
  76                                              self._initial_accumulator_value)
  77           return math_ops.cast(init_constant, dtype)
  78       state.create_slot_with_initializer(v, init, v.get_shape(), dtype,
  79                                          "accumulator")
  80 
  81   def _apply_dense(self, grad, var, state):
  82     acc = state.get_slot(var, "accumulator")
  83     return training_ops.apply_adagrad(
  84         var,
  85         acc,
  86         state.get_hyper("learning_rate", var.dtype.base_dtype),
  87         grad,
  88         use_locking=self._use_locking)
  89 
  90   def _resource_apply_dense(self, grad, var, state):
  91     acc = state.get_slot(var, "accumulator")
  92     return training_ops.resource_apply_adagrad(
  93         var.handle,
  94         acc.handle,
  95         state.get_hyper("learning_rate", var.dtype.base_dtype),
  96         grad,
  97         use_locking=self._use_locking)
  98 
  99   def _apply_sparse(self, grad, var, state):
 100     acc = state.get_slot(var, "accumulator")
 101     return training_ops.sparse_apply_adagrad(
 102         var,
 103         acc,
 104         state.get_hyper("learning_rate", var.dtype.base_dtype),
 105         grad.values,
 106         grad.indices,
 107         use_locking=self._use_locking)
 108 
 109   def _resource_apply_sparse(self, grad, var, indices, state):
 110     acc = state.get_slot(var, "accumulator")
 111     return training_ops.resource_sparse_apply_adagrad(
 112         var.handle,
 113         acc.handle,
 114         state.get_hyper("learning_rate", var.dtype.base_dtype),
 115         grad,
 116         indices,
 117         use_locking=self._use_locking)
