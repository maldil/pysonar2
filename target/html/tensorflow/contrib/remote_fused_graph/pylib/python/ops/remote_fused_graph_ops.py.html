  30 # RemoteFusedGraphExecute is not differenciable op.
  31 ops.NotDifferentiable("RemoteFusedGraphExecute")

  33 
  34 def remote_fused_graph_execute(inputs,
  35                                output_types,
  36                                graph_def,
  37                                graph_input_node_names,
  38                                graph_output_node_names,
  39                                executor_name,
  40                                serialized_executor_parameters,
  41                                default_graph_input_tensor_type_shapes=None,
  42                                default_graph_output_tensor_type_shapes=None):
  43   """A wrapper for remote_fused_graph_execute."""
  44   info_proto = info_pb2.RemoteFusedGraphExecuteInfo()
  45   info_proto.remote_graph.CopyFrom(graph_def)
  46   info_proto.graph_input_node_name.extend(graph_input_node_names)
  47   info_proto.graph_output_node_name.extend(graph_output_node_names)
  48   info_proto.executor_name = executor_name
  49   info_proto.serialized_executor_parameters = serialized_executor_parameters
  50   if default_graph_input_tensor_type_shapes:
  51     for type_shape in default_graph_input_tensor_type_shapes:
  52       type_shape_proto = info_proto.default_graph_input_tensor_shape.add()
  53       type_shape_proto.dtype = int(dtypes.as_dtype(type_shape[0]))
  54       for dim in type_shape[1]:
  55         type_shape_proto.shape.dim.add().size = dim
  56   if default_graph_output_tensor_type_shapes:
  57     for type_shape in default_graph_output_tensor_type_shapes:
  58       type_shape_proto = info_proto.default_graph_output_tensor_shape.add()
  59       type_shape_proto.dtype = int(dtypes.as_dtype(type_shape[0]))
  60       for dim in type_shape[1]:
  61         type_shape_proto.shape.dim.add().size = dim
  62 
  63   serialized_info = info_proto.SerializeToString()
  64 
  65   return gen_remote_fused_graph_ops.remote_fused_graph_execute(
  66       inputs, output_types, serialized_info)
