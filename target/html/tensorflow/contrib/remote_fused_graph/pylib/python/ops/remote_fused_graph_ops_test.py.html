  42     node.name = "b"
  43     node.op = "op1"
  44     inputs = [ops.convert_n_to_tensor([1], dtypes.int64)]
  45     output_types = [np.int64, np.int64]
  46     graph_input_node_names = ["a"]
  47     graph_output_node_names = ["a", "b"]
  48     executor_name = ""
  49     serialized_executor_parameters = b""
  50     default_graph_input_tensor_type_shapes = [[dtypes.int64, [1]]]
  51     default_graph_output_tensor_type_shapes = [[dtypes.int64, [1]],
  52                                                [dtypes.int64, [1]]]
  53 
  54     output_nodes = remote_fused_graph_ops.remote_fused_graph_execute(
  55         inputs, output_types, graph, graph_input_node_names,
  56         graph_output_node_names, executor_name, serialized_executor_parameters,
  57         default_graph_input_tensor_type_shapes,
  58         default_graph_output_tensor_type_shapes)
  59     self.assertEqual(2, len(output_nodes))
  60     for output_node in output_nodes:
  61       with self.test_session(use_gpu=False):
  62         output_node.eval()
