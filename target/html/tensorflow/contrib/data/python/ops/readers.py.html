  32 @deprecation.deprecated(None,
  33                         "Use `tf.data.experimental.make_csv_dataset(...)`.")
  34 def make_csv_dataset(
  35     file_pattern,
  36     batch_size,
  37     column_names=None,
  38     column_defaults=None,
  39     label_name=None,
  40     select_columns=None,
  41     field_delim=",",
  42     use_quote_delim=True,
  43     na_value="",
  44     header=True,
  45     num_epochs=None,
  46     shuffle=True,
  47     shuffle_buffer_size=10000,
  48     shuffle_seed=None,
  49     prefetch_buffer_size=optimization.AUTOTUNE,
  50     num_parallel_reads=1,
  51     sloppy=False,
  52     num_rows_for_inference=100,
  53     compression_type=None,
  54 ):
  55   """Reads CSV files into a dataset.
  56 
  57   Reads CSV files into a dataset, where each element is a (features, labels)
  58   tuple that corresponds to a batch of CSV rows. The features dictionary
  59   maps feature column names to `Tensor`s containing the corresponding
  60   feature data, and labels is a `Tensor` containing the batch's label data.
  61 
  62   Args:
  63     file_pattern: List of files or patterns of file paths containing CSV
  64       records. See `tf.gfile.Glob` for pattern rules.
  65     batch_size: An int representing the number of records to combine
  66       in a single batch.
  67     column_names: An optional list of strings that corresponds to the CSV
  68       columns, in order. One per column of the input record. If this is not
  69       provided, infers the column names from the first row of the records.
  70       These names will be the keys of the features dict of each dataset element.
  71     column_defaults: A optional list of default values for the CSV fields. One
  72       item per selected column of the input record. Each item in the list is
  73       either a valid CSV dtype (float32, float64, int32, int64, or string), or a
  74       `Tensor` with one of the aforementioned types. The tensor can either be
  75       a scalar default value (if the column is optional), or an empty tensor (if
  76       the column is required). If a dtype is provided instead of a tensor, the
  77       column is also treated as required. If this list is not provided, tries
  78       to infer types based on reading the first num_rows_for_inference rows of
  79       files specified, and assumes all columns are optional, defaulting to `0`
  80       for numeric values and `""` for string values. If both this and
  81       `select_columns` are specified, these must have the same lengths, and
  82       `column_defaults` is assumed to be sorted in order of increasing column
  83       index.
  84     label_name: A optional string corresponding to the label column. If
  85       provided, the data for this column is returned as a separate `Tensor` from
  86       the features dictionary, so that the dataset complies with the format
  87       expected by a `tf.Estimator.train` or `tf.Estimator.evaluate` input
  88       function.
  89     select_columns: An optional list of integer indices or string column
  90       names, that specifies a subset of columns of CSV data to select. If
  91       column names are provided, these must correspond to names provided in
  92       `column_names` or inferred from the file header lines. When this argument
  93       is specified, only a subset of CSV columns will be parsed and returned,
  94       corresponding to the columns specified. Using this results in faster
  95       parsing and lower memory usage. If both this and `column_defaults` are
  96       specified, these must have the same lengths, and `column_defaults` is
  97       assumed to be sorted in order of increasing column index.
  98     field_delim: An optional `string`. Defaults to `","`. Char delimiter to
  99       separate fields in a record.
 100     use_quote_delim: An optional bool. Defaults to `True`. If false, treats
 101       double quotation marks as regular characters inside of the string fields.
 102     na_value: Additional string to recognize as NA/NaN.
 103     header: A bool that indicates whether the first rows of provided CSV files
 104       correspond to header lines with column names, and should not be included
 105       in the data.
 106     num_epochs: An int specifying the number of times this dataset is repeated.
 107       If None, cycles through the dataset forever.
 108     shuffle: A bool that indicates whether the input should be shuffled.
 109     shuffle_buffer_size: Buffer size to use for shuffling. A large buffer size
 110       ensures better shuffling, but increases memory usage and startup time.
 111     shuffle_seed: Randomization seed to use for shuffling.
 112     prefetch_buffer_size: An int specifying the number of feature
 113       batches to prefetch for performance improvement. Recommended value is the
 114       number of batches consumed per training step. Defaults to auto-tune.
 115     num_parallel_reads: Number of threads used to read CSV records from files.
 116       If >1, the results will be interleaved.
 117     sloppy: If `True`, reading performance will be improved at
 118       the cost of non-deterministic ordering. If `False`, the order of elements
 119       produced is deterministic prior to shuffling (elements are still
 120       randomized if `shuffle=True`. Note that if the seed is set, then order
 121       of elements after shuffling is deterministic). Defaults to `False`.
 122     num_rows_for_inference: Number of rows of a file to use for type inference
 123       if record_defaults is not provided. If None, reads all the rows of all
 124       the files. Defaults to 100.
 125     compression_type: (Optional.) A `tf.string` scalar evaluating to one of
 126       `""` (no compression), `"ZLIB"`, or `"GZIP"`. Defaults to no compression.
 127 
 128   Returns:
 129     A dataset, where each element is a (features, labels) tuple that corresponds
 130     to a batch of `batch_size` CSV rows. The features dictionary maps feature
 131     column names to `Tensor`s containing the corresponding column data, and
 132     labels is a `Tensor` containing the column data for the label column
 133     specified by `label_name`.
 134 
 135   Raises:
 136     ValueError: If any of the arguments is malformed.
 137   """
 138   return readers.make_csv_dataset(
 139       file_pattern, batch_size, column_names, column_defaults, label_name,
 140       select_columns, field_delim, use_quote_delim, na_value, header,
 141       num_epochs, shuffle, shuffle_buffer_size, shuffle_seed,
 142       prefetch_buffer_size, num_parallel_reads, sloppy, num_rows_for_inference,
 143       compression_type)

 165 @deprecation.deprecated(
 166     None, "Use `tf.data.experimental.make_batched_features_dataset(...)`.")
 167 def make_batched_features_dataset(file_pattern,
 168                                   batch_size,
 169                                   features,
 170                                   reader=core_readers.TFRecordDataset,
 171                                   label_key=None,
 172                                   reader_args=None,
 173                                   num_epochs=None,
 174                                   shuffle=True,
 175                                   shuffle_buffer_size=10000,
 176                                   shuffle_seed=None,
 177                                   prefetch_buffer_size=optimization.AUTOTUNE,
 178                                   reader_num_threads=1,
 179                                   parser_num_threads=2,
 180                                   sloppy_ordering=False,
 181                                   drop_final_batch=False):
 182   """Returns a `Dataset` of feature dictionaries from `Example` protos.
 183 
 184   If label_key argument is provided, returns a `Dataset` of tuple
 185   comprising of feature dictionaries and label.
 186 
 187   Example:
 188 
 189   ```
 190   serialized_examples = [
 191     features {
 192       feature { key: "age" value { int64_list { value: [ 0 ] } } }
 193       feature { key: "gender" value { bytes_list { value: [ "f" ] } } }
 194       feature { key: "kws" value { bytes_list { value: [ "code", "art" ] } } }
 195     },
 196     features {
 197       feature { key: "age" value { int64_list { value: [] } } }
 198       feature { key: "gender" value { bytes_list { value: [ "f" ] } } }
 199       feature { key: "kws" value { bytes_list { value: [ "sports" ] } } }
 200     }
 201   ]
 202   ```
 203 
 204   We can use arguments:
 205 
 206   ```
 207   features: {
 208     "age": FixedLenFeature([], dtype=tf.int64, default_value=-1),
 209     "gender": FixedLenFeature([], dtype=tf.string),
 210     "kws": VarLenFeature(dtype=tf.string),
 211   }
 212   ```
 213 
 214   And the expected output is:
 215 
 216   ```python
 217   {
 218     "age": [[0], [-1]],
 219     "gender": [["f"], ["f"]],
 220     "kws": SparseTensor(
 221       indices=[[0, 0], [0, 1], [1, 0]],
 222       values=["code", "art", "sports"]
 223       dense_shape=[2, 2]),
 224   }
 225   ```
 226 
 227   Args:
 228     file_pattern: List of files or patterns of file paths containing
 229       `Example` records. See `tf.gfile.Glob` for pattern rules.
 230     batch_size: An int representing the number of records to combine
 231       in a single batch.
 232     features: A `dict` mapping feature keys to `FixedLenFeature` or
 233       `VarLenFeature` values. See `tf.parse_example`.
 234     reader: A function or class that can be
 235       called with a `filenames` tensor and (optional) `reader_args` and returns
 236       a `Dataset` of `Example` tensors. Defaults to `tf.data.TFRecordDataset`.
 237     label_key: (Optional) A string corresponding to the key labels are stored in
 238       `tf.Examples`. If provided, it must be one of the `features` key,
 239       otherwise results in `ValueError`.
 240     reader_args: Additional arguments to pass to the reader class.
 241     num_epochs: Integer specifying the number of times to read through the
 242       dataset. If None, cycles through the dataset forever. Defaults to `None`.
 243     shuffle: A boolean, indicates whether the input should be shuffled. Defaults
 244       to `True`.
 245     shuffle_buffer_size: Buffer size of the ShuffleDataset. A large capacity
 246       ensures better shuffling but would increase memory usage and startup time.
 247     shuffle_seed: Randomization seed to use for shuffling.
 248     prefetch_buffer_size: Number of feature batches to prefetch in order to
 249       improve performance. Recommended value is the number of batches consumed
 250       per training step. Defaults to auto-tune.
 251     reader_num_threads: Number of threads used to read `Example` records. If >1,
 252       the results will be interleaved.
 253     parser_num_threads: Number of threads to use for parsing `Example` tensors
 254       into a dictionary of `Feature` tensors.
 255     sloppy_ordering: If `True`, reading performance will be improved at
 256       the cost of non-deterministic ordering. If `False`, the order of elements
 257       produced is deterministic prior to shuffling (elements are still
 258       randomized if `shuffle=True`. Note that if the seed is set, then order
 259       of elements after shuffling is deterministic). Defaults to `False`.
 260     drop_final_batch: If `True`, and the batch size does not evenly divide the
 261       input dataset size, the final smaller batch will be dropped. Defaults to
 262       `False`.
 263 
 264   Returns:
 265     A dataset of `dict` elements, (or a tuple of `dict` elements and label).
 266     Each `dict` maps feature keys to `Tensor` or `SparseTensor` objects.
 267 
 268   Raises:
 269     ValueError: If `label_key` is not one of the `features` keys.
 270   """
 271   return readers.make_batched_features_dataset(
 272       file_pattern, batch_size, features, reader, label_key, reader_args,
 273       num_epochs, shuffle, shuffle_buffer_size, shuffle_seed,
 274       prefetch_buffer_size, reader_num_threads, parser_num_threads,
 275       sloppy_ordering, drop_final_batch)

 278 @deprecation.deprecated(
 279     None, "Use `tf.data.experimental.make_batched_features_dataset(...)`")
 280 def read_batch_features(file_pattern,
 281                         batch_size,
 282                         features,
 283                         reader=core_readers.TFRecordDataset,
 284                         reader_args=None,
 285                         randomize_input=True,
 286                         num_epochs=None,
 287                         capacity=10000):
 288   """Reads batches of Examples.
 289 
 290   Example:
 291 
 292   ```
 293   serialized_examples = [
 294     features {
 295       feature { key: "age" value { int64_list { value: [ 0 ] } } }
 296       feature { key: "gender" value { bytes_list { value: [ "f" ] } } }
 297       feature { key: "kws" value { bytes_list { value: [ "code", "art" ] } } }
 298     },
 299     features {
 300       feature { key: "age" value { int64_list { value: [] } } }
 301       feature { key: "gender" value { bytes_list { value: [ "f" ] } } }
 302       feature { key: "kws" value { bytes_list { value: [ "sports" ] } } }
 303     }
 304   ]
 305   ```
 306 
 307   We can use arguments:
 308 
 309   ```
 310   features: {
 311     "age": FixedLenFeature([], dtype=tf.int64, default_value=-1),
 312     "gender": FixedLenFeature([], dtype=tf.string),
 313     "kws": VarLenFeature(dtype=tf.string),
 314   }
 315   ```
 316 
 317   And the expected output is:
 318 
 319   ```python
 320   {
 321     "age": [[0], [-1]],
 322     "gender": [["f"], ["f"]],
 323     "kws": SparseTensor(
 324       indices=[[0, 0], [0, 1], [1, 0]],
 325       values=["code", "art", "sports"]
 326       dense_shape=[2, 2]),
 327   }
 328   ```
 329 
 330   Args:
 331     file_pattern: List of files or patterns of file paths containing
 332       `Example` records. See `tf.gfile.Glob` for pattern rules.
 333     batch_size: An int representing the number of records to combine
 334       in a single batch.
 335     features: A `dict` mapping feature keys to `FixedLenFeature` or
 336       `VarLenFeature` values. See `tf.parse_example`.
 337     reader: A function or class that can be
 338       called with a `filenames` tensor and (optional) `reader_args` and returns
 339       a `Dataset` of `Example` tensors. Defaults to `tf.data.TFRecordDataset`.
 340     reader_args: Additional arguments to pass to the reader class.
 341     randomize_input: Whether the input should be randomized.
 342     num_epochs: Integer specifying the number of times to read through the
 343       dataset. If None, cycles through the dataset forever.
 344     capacity: Buffer size of the ShuffleDataset. A large capacity ensures better
 345       shuffling but would increase memory usage and startup time.
 346   Returns:
 347     A dict from keys in features to `Tensor` or `SparseTensor` objects.
 348   """
 349   dataset = readers.make_batched_features_dataset(
 350       file_pattern,
 351       batch_size,
 352       features,
 353       reader=reader,
 354       reader_args=reader_args,
 355       shuffle=randomize_input,
 356       num_epochs=num_epochs,
 357       shuffle_buffer_size=capacity)
 358   iterator = dataset.make_one_shot_iterator()
 359   outputs = iterator.get_next()
 360   return outputs

 372 class LMDBDataset(dataset_ops.DatasetSource):
 373   """A LMDB Dataset that reads the lmdb file."""
 374 
 375   def __init__(self, filenames):
 376     """Create a `LMDBDataset`.
 377 
 378     `LMDBDataset` allows a user to read data from a mdb file as
 379     (key value) pairs sequentially.
 380     For example:
 381     ```python
 382     dataset = tf.contrib.lmdb.LMDBDataset("/foo/bar.mdb")
 383     iterator = dataset.make_one_shot_iterator()
 384     next_element = iterator.get_next()
 385     # Prints the (key, value) pairs inside a lmdb file.
 386     while True:
 387       try:
 388         print(sess.run(next_element))
 389       except tf.errors.OutOfRangeError:
 390         break
 391     ```
 392     Args:
 393       filenames: A `tf.string` tensor containing one or more filenames.
 394     """
 395     super(LMDBDataset, self).__init__()
 396     self._filenames = ops.convert_to_tensor(
 397         filenames, dtype=dtypes.string, name="filenames")
 398 
 399   def _as_variant_tensor(self):
 400     return gen_experimental_dataset_ops.experimental_lmdb_dataset(
 401         self._filenames,
 402         output_types=nest.flatten(self.output_types),
 403         output_shapes=nest.flatten(self.output_shapes))
 404 
 405   @property
 406   def output_classes(self):
 407     return ops.Tensor, ops.Tensor
 408 
 409   @property
 410   def output_shapes(self):
 411     return (tensor_shape.TensorShape([]), tensor_shape.TensorShape([]))
 412 
 413   @property
 414   def output_types(self):
 415     return dtypes.string, dtypes.string
