  26 @deprecation.deprecated(
  27     None, "Use `tf.data.experimental.dense_to_sparse_batch(...)`.")
  28 def dense_to_sparse_batch(batch_size, row_shape):
  29   """A transformation that batches ragged elements into `tf.SparseTensor`s.
  30 
  31   Like `Dataset.padded_batch()`, this transformation combines multiple
  32   consecutive elements of the dataset, which might have different
  33   shapes, into a single element. The resulting element has three
  34   components (`indices`, `values`, and `dense_shape`), which
  35   comprise a `tf.SparseTensor` that represents the same data. The
  36   `row_shape` represents the dense shape of each row in the
  37   resulting `tf.SparseTensor`, to which the effective batch size is
  38   prepended. For example:
  39 
  40   ```python
  41   # NOTE: The following examples use `{ ... }` to represent the
  42   # contents of a dataset.
  43   a = { ['a', 'b', 'c'], ['a', 'b'], ['a', 'b', 'c', 'd'] }
  44 
  45   a.apply(tf.contrib.data.dense_to_sparse_batch(batch_size=2, row_shape=[6])) ==
  46   {
  47       ([[0, 0], [0, 1], [0, 2], [1, 0], [1, 1]],  # indices
  48        ['a', 'b', 'c', 'a', 'b'],                 # values
  49        [2, 6]),                                   # dense_shape
  50       ([[0, 0], [0, 1], [0, 2], [0, 3]],
  51        ['a', 'b', 'c', 'd'],
  52        [1, 6])
  53   }
  54   ```
  55 
  56   Args:
  57     batch_size: A `tf.int64` scalar `tf.Tensor`, representing the
  58       number of consecutive elements of this dataset to combine in a
  59       single batch.
  60     row_shape: A `tf.TensorShape` or `tf.int64` vector tensor-like
  61       object representing the equivalent dense shape of a row in the
  62       resulting `tf.SparseTensor`. Each element of this dataset must
  63       have the same rank as `row_shape`, and must have size less
  64       than or equal to `row_shape` in each dimension.
  65 
  66   Returns:
  67     A `Dataset` transformation function, which can be passed to
  68     `tf.data.Dataset.apply`.
  69   """
  70   return batching.dense_to_sparse_batch(batch_size, row_shape)

  73 @deprecation.deprecated(None, "Use `tf.data.experimental.unbatch()`.")
  74 def unbatch():
  75   """Splits elements of a dataset into multiple elements on the batch dimension.
  76 
  77   For example, if elements of the dataset are shaped `[B, a0, a1, ...]`,
  78   where `B` may vary for each input element, then for each element in the
  79   dataset, the unbatched dataset will contain `B` consecutive elements
  80   of shape `[a0, a1, ...]`.
  81 
  82   ```python
  83   # NOTE: The following example uses `{ ... }` to represent the contents
  84   # of a dataset.
  85   a = { ['a', 'b', 'c'], ['a', 'b'], ['a', 'b', 'c', 'd'] }
  86 
  87   a.apply(tf.contrib.data.unbatch()) == {
  88       'a', 'b', 'c', 'a', 'b', 'a', 'b', 'c', 'd'}
  89   ```
  90 
  91   Returns:
  92     A `Dataset` transformation function, which can be passed to
  93     `tf.data.Dataset.apply`.
  94   """
  95   return batching.unbatch()

  98 @deprecation.deprecated(
  99     None, "Use `tf.data.Dataset.batch(..., drop_remainder=True)`.")
 100 def batch_and_drop_remainder(batch_size):
 101   """A batching transformation that omits the final small batch (if present).
 102 
 103   Like `tf.data.Dataset.batch`, this transformation combines
 104   consecutive elements of this dataset into batches. However, if the batch
 105   size does not evenly divide the input dataset size, this transformation will
 106   drop the final smaller element.
 107 
 108   The following example illustrates the difference between this
 109   transformation and `Dataset.batch()`:
 110 
 111   ```python
 112   dataset = tf.data.Dataset.range(200)
 113   batched = dataset.apply(tf.contrib.data.batch_and_drop_remainder(128))
 114   print(batched.output_shapes)  # ==> "(128,)" (the batch dimension is known)
 115   ```
 116 
 117   By contrast, `dataset.batch(128)` would yield a two-element dataset with
 118   shapes `(128,)` and `(72,)`, so the batch dimension would not be statically
 119   known.
 120 
 121   Args:
 122     batch_size: A `tf.int64` scalar `tf.Tensor`, representing the number of
 123         consecutive elements of this dataset to combine in a single batch.
 124 
 125   Returns:
 126     A `Dataset` transformation function, which can be passed to
 127     `tf.data.Dataset.apply`
 128   """
 129 
 130   def _apply_fn(dataset):
 131     """Function from `Dataset` to `Dataset` that applies the transformation."""
 132     return dataset.batch(batch_size, drop_remainder=True)
 133 
 134   return _apply_fn

 137 @deprecation.deprecated(
 138     None, "Use `tf.data.Dataset.padded_batch(..., drop_remainder=True)`.")
 139 def padded_batch_and_drop_remainder(batch_size,
 140                                     padded_shapes,
 141                                     padding_values=None):
 142   """A batching and padding transformation that omits the final small batch.
 143 
 144   Like `tf.data.Dataset.padded_batch`, this transformation combines
 145   consecutive elements of this dataset into batches. However, if the batch
 146   size does not evenly divide the input dataset size, this transformation will
 147   drop the final smaller element.
 148 
 149   See `tf.contrib.data.batch_and_drop_remainder` for more details.
 150 
 151   Args:
 152     batch_size: A `tf.int64` scalar `tf.Tensor`, representing the number of
 153       consecutive elements of this dataset to combine in a single batch.
 154     padded_shapes: A nested structure of `tf.TensorShape` or
 155       `tf.int64` vector tensor-like objects. See
 156       `tf.data.Dataset.padded_batch` for details.
 157     padding_values: (Optional.) A nested structure of scalar-shaped
 158       `tf.Tensor`. See `tf.data.Dataset.padded_batch` for details.
 159 
 160   Returns:
 161     A `Dataset` transformation function, which can be passed to
 162     `tf.data.Dataset.apply`
 163   """
 164 
 165   def _apply_fn(dataset):
 166     """Function from `Dataset` to `Dataset` that applies the transformation."""
 167     return dataset.padded_batch(
 168         batch_size, padded_shapes=padded_shapes, padding_values=padding_values,
 169         drop_remainder=True)
 170 
 171   return _apply_fn

 174 # TODO(b/116817045): Move this to `tf.data.experimental` when the `with_shape()`
 175 # function is available in the core.
 176 def assert_element_shape(expected_shapes):
 177   """Assert the shape of this `Dataset`.
 178 
 179   ```python
 180   shapes = [tf.TensorShape([16, 256]), tf.TensorShape([None, 2])]
 181   result = dataset.apply(tf.contrib.data.assert_element_shape(shapes))
 182   print(result.output_shapes)  # ==> "((16, 256), (<unknown>, 2))"
 183   ```
 184 
 185   If dataset shapes and expected_shape, are fully defined, assert they match.
 186   Otherwise, add assert op that will validate the shapes when tensors are
 187   evaluated, and set shapes on tensors, respectively.
 188 
 189   Note that unknown dimension in `expected_shapes` will be ignored.
 190 
 191   Args:
 192     expected_shapes: A nested structure of `tf.TensorShape` objects.
 193 
 194   Returns:
 195     A `Dataset` transformation function, which can be passed to
 196     `tf.data.Dataset.apply`
 197   """
 198 
 199   def _merge_output_shapes(original_shapes, expected_shapes):
 200     flat_original_shapes = nest.flatten(original_shapes)
 201     flat_new_shapes = nest.flatten_up_to(original_shapes, expected_shapes)
 202     flat_merged_output_shapes = [
 203         original_shape.merge_with(new_shape)
 204         for original_shape, new_shape in zip(flat_original_shapes,
 205                                              flat_new_shapes)]
 206     return nest.pack_sequence_as(original_shapes, flat_merged_output_shapes)
 207 
 208   def _check_shape(*elements):
 209     flatten_tensors = nest.flatten(elements)
 210     flatten_shapes = nest.flatten(expected_shapes)
 211     checked_tensors = [
 212         with_shape(shape, tensor) if shape else tensor  # Ignore unknown shape
 213         for shape, tensor in zip(flatten_shapes, flatten_tensors)
 214     ]
 215     return nest.pack_sequence_as(elements, checked_tensors)
 216 
 217   def _apply_fn(dataset):
 218     output_shapes = _merge_output_shapes(dataset.output_shapes,
 219                                          expected_shapes)
 220     # pylint: disable=protected-access
 221     return batching._RestructuredDataset(
 222         dataset.map(_check_shape),
 223         dataset.output_types,
 224         output_shapes=output_shapes,
 225         output_classes=dataset.output_classes)
 226 
 227   return _apply_fn

 230 @deprecation.deprecated(None, "Use `tf.data.experimental.map_and_batch(...)`.")
 231 def map_and_batch(map_func,
 232                   batch_size,
 233                   num_parallel_batches=None,
 234                   drop_remainder=False,
 235                   num_parallel_calls=None):
 236   """Fused implementation of `map` and `batch`.
 237 
 238   Maps `map_func` across `batch_size` consecutive elements of this dataset
 239   and then combines them into a batch. Functionally, it is equivalent to `map`
 240   followed by `batch`. However, by fusing the two transformations together, the
 241   implementation can be more efficient. Surfacing this transformation in the API
 242   is temporary. Once automatic input pipeline optimization is implemented,
 243   the fusing of `map` and `batch` will happen automatically and this API will be
 244   deprecated.
 245 
 246   Args:
 247     map_func: A function mapping a nested structure of tensors to another
 248       nested structure of tensors.
 249     batch_size: A `tf.int64` scalar `tf.Tensor`, representing the number of
 250       consecutive elements of this dataset to combine in a single batch.
 251     num_parallel_batches: (Optional.) A `tf.int64` scalar `tf.Tensor`,
 252       representing the number of batches to create in parallel. On one hand,
 253       higher values can help mitigate the effect of stragglers. On the other
 254       hand, higher values can increase contention if CPU is scarce.
 255     drop_remainder: (Optional.) A `tf.bool` scalar `tf.Tensor`, representing
 256       whether the last batch should be dropped in case its size is smaller than
 257       desired; the default behavior is not to drop the smaller batch.
 258     num_parallel_calls: (Optional.) A `tf.int32` scalar `tf.Tensor`,
 259         representing the number of elements to process in parallel. If not
 260         specified, `batch_size * num_parallel_batches` elements will be
 261         processed in parallel.
 262 
 263   Returns:
 264     A `Dataset` transformation function, which can be passed to
 265     `tf.data.Dataset.apply`.
 266 
 267   Raises:
 268     ValueError: If both `num_parallel_batches` and `num_parallel_calls` are
 269       specified.
 270   """
 271   return batching.map_and_batch(map_func, batch_size, num_parallel_batches,
 272                                 drop_remainder, num_parallel_calls)
