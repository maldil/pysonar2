  24 @deprecation.deprecated(None,
  25                         "Use `tf.data.experimental.shuffle_and_repeat(...)`.")
  26 def shuffle_and_repeat(buffer_size, count=None, seed=None):
  27   """Shuffles and repeats a Dataset returning a new permutation for each epoch.
  28 
  29   `dataset.apply(tf.contrib.data.shuffle_and_repeat(buffer_size, count))`
  30 
  31   is equivalent to
  32 
  33   `dataset.shuffle(buffer_size, reshuffle_each_iteration=True).repeat(count)`
  34 
  35   The difference is that the latter dataset is not serializable. So,
  36   if you need to checkpoint an input pipeline with reshuffling you must use
  37   this implementation.
  38 
  39   Args:
  40     buffer_size: A `tf.int64` scalar `tf.Tensor`, representing the
  41       maximum number elements that will be buffered when prefetching.
  42     count: (Optional.) A `tf.int64` scalar `tf.Tensor`, representing the
  43       number of times the dataset should be repeated. The default behavior
  44       (if `count` is `None` or `-1`) is for the dataset be repeated
  45       indefinitely.
  46     seed: (Optional.) A `tf.int64` scalar `tf.Tensor`, representing the
  47       random seed that will be used to create the distribution. See
  48       `tf.set_random_seed` for behavior.
  49 
  50   Returns:
  51     A `Dataset` transformation function, which can be passed to
  52     `tf.data.Dataset.apply`.
  53   """
  54   return shuffle_ops.shuffle_and_repeat(buffer_size, count, seed)
