  29 class _SlideDataset(dataset_ops.UnaryDataset):
  30   """A `Dataset` that passes a sliding window over its input."""
  31 
  32   def __init__(self, input_dataset, window_size, window_shift, window_stride):
  33     """See `sliding_window_batch` for details."""
  34     super(_SlideDataset, self).__init__(input_dataset)
  35     self._input_dataset = input_dataset
  36     self._window_size = ops.convert_to_tensor(
  37         window_size, dtype=dtypes.int64, name="window_stride")
  38     self._window_stride = ops.convert_to_tensor(
  39         window_stride, dtype=dtypes.int64, name="window_stride")
  40     self._window_shift = ops.convert_to_tensor(
  41         window_shift, dtype=dtypes.int64, name="window_shift")
  42 
  43   def _as_variant_tensor(self):
  44     return gen_dataset_ops.slide_dataset(
  45         self._input_dataset._as_variant_tensor(),  # pylint: disable=protected-access
  46         window_size=self._window_size,
  47         window_shift=self._window_shift,
  48         window_stride=self._window_stride,
  49         **dataset_ops.flat_structure(self))
  50 
  51   @property
  52   def output_classes(self):
  53     return self._input_dataset.output_classes
  54 
  55   @property
  56   def output_shapes(self):
  57     input_shapes = self._input_dataset.output_shapes
  58     return nest.pack_sequence_as(input_shapes, [
  59         tensor_shape.vector(None).concatenate(s)
  60         for s in nest.flatten(self._input_dataset.output_shapes)
  61     ])
  62 
  63   @property
  64   def output_types(self):
  65     return self._input_dataset.output_types

  74 def sliding_window_batch(window_size,
  75                          stride=None,
  76                          window_shift=None,
  77                          window_stride=1):
  78   """A sliding window over a dataset.
  79 
  80   This transformation passes a sliding window over this dataset. The window size
  81   is `window_size`, the stride of the input elements is `window_stride`, and the
  82   shift between consecutive windows is `window_shift`. If the remaining elements
  83   cannot fill up the sliding window, this transformation will drop the final
  84   smaller element. For example:
  85 
  86   ```python
  87   # NOTE: The following examples use `{ ... }` to represent the
  88   # contents of a dataset.
  89   a = { [1], [2], [3], [4], [5], [6] }
  90 
  91   a.apply(sliding_window_batch(window_size=3)) ==
  92   { [[1], [2], [3]], [[2], [3], [4]], [[3], [4], [5]], [[4], [5], [6]] }
  93 
  94   a.apply(sliding_window_batch(window_size=3, window_shift=2)) ==
  95   { [[1], [2], [3]], [[3], [4], [5]] }
  96 
  97   a.apply(sliding_window_batch(window_size=3, window_stride=2)) ==
  98   { [[1], [3], [5]], [[2], [4], [6]] }
  99   ```
 100 
 101   Args:
 102     window_size: A `tf.int64` scalar `tf.Tensor`, representing the number of
 103       elements in the sliding window. It must be positive.
 104     stride: (Optional.) A `tf.int64` scalar `tf.Tensor`, representing the
 105       forward shift of the sliding window in each iteration. The default is `1`.
 106       It must be positive. Deprecated alias for `window_shift`.
 107     window_shift: (Optional.) A `tf.int64` scalar `tf.Tensor`, representing the
 108       forward shift of the sliding window in each iteration. The default is `1`.
 109       It must be positive.
 110     window_stride: (Optional.) A `tf.int64` scalar `tf.Tensor`, representing the
 111       stride of the input elements in the sliding window. The default is `1`.
 112       It must be positive.
 113 
 114   Returns:
 115     A `Dataset` transformation function, which can be passed to
 116     `tf.data.Dataset.apply`.
 117 
 118   Raises:
 119     ValueError: if invalid arguments are provided.
 120   """
 121   if stride is None and window_shift is None:
 122     window_shift = 1
 123   elif stride is not None and window_shift is None:
 124     window_shift = stride
 125   elif stride is not None and window_shift is not None:
 126     raise ValueError("Cannot specify both `stride` and `window_shift`")
 127 
 128   def _apply_fn(dataset):
 129     return _SlideDataset(dataset, window_size, window_shift, window_stride)
 130 
 131   return _apply_fn
