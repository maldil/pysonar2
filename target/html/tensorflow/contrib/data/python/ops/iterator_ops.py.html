  24 @deprecation.deprecated(
  25     None, "Use `tf.data.experimental.make_saveable_from_iterator(...)`.")
  26 def make_saveable_from_iterator(iterator):
  27   """Returns a SaveableObject for saving/restore iterator state using Saver.
  28 
  29   Args:
  30     iterator: Iterator.
  31 
  32   For example:
  33 
  34   ```python
  35   with tf.Graph().as_default():
  36     ds = tf.data.Dataset.range(10)
  37     iterator = ds.make_initializable_iterator()
  38     # Build the iterator SaveableObject.
  39     saveable_obj = tf.contrib.data.make_saveable_from_iterator(iterator)
  40     # Add the SaveableObject to the SAVEABLE_OBJECTS collection so
  41     # it can be automatically saved using Saver.
  42     tf.add_to_collection(tf.GraphKeys.SAVEABLE_OBJECTS, saveable_obj)
  43     saver = tf.train.Saver()
  44 
  45     while continue_training:
  46       ... Perform training ...
  47       if should_save_checkpoint:
  48         saver.save()
  49   ```
  50 
  51   Note: When restoring the iterator, the existing iterator state is completely
  52   discarded. This means that any changes you may have made to the Dataset
  53   graph will be discarded as well! This includes the new Dataset graph
  54   that you may have built during validation. So, while running validation,
  55   make sure to run the initializer for the validation input pipeline after
  56   restoring the checkpoint.
  57 
  58   Note: Not all iterators support checkpointing yet. Attempting to save the
  59   state of an unsupported iterator will throw an error.
  60   """
  61   return iterator_ops.make_saveable_from_iterator(iterator)

  64 class CheckpointInputPipelineHook(iterator_ops.CheckpointInputPipelineHook):
  65   """Checkpoints input pipeline state every N steps or seconds.
  66 
  67   This hook saves the state of the iterators in the `Graph` so that when
  68   training is resumed the input pipeline continues from where it left off.
  69   This could potentially avoid overfitting in certain pipelines where the
  70   number of training steps per eval are small compared to the dataset
  71   size or if the training pipeline is pre-empted.
  72 
  73   Differences from `CheckpointSaverHook`:
  74   1. Saves only the input pipelines in the "iterators" collection and not the
  75      global variables or other saveable objects.
  76   2. Does not write the `GraphDef` and `MetaGraphDef` to the summary.
  77 
  78   Example of checkpointing the training pipeline:
  79 
  80   ```python
  81   est = tf.estimator.Estimator(model_fn)
  82   while True:
  83     est.train(
  84         train_input_fn,
  85         hooks=[tf.contrib.data.CheckpointInputPipelineHook(est)],
  86         steps=train_steps_per_eval)
  87     # Note: We do not pass the hook here.
  88     metrics = est.evaluate(eval_input_fn)
  89     if should_stop_the_training(metrics):
  90       break
  91   ```
  92 
  93   This hook should be used if the input pipeline state needs to be saved
  94   separate from the model checkpoint. Doing so may be useful for a few reasons:
  95   1. The input pipeline checkpoint may be large, if there are large shuffle
  96      or prefetch buffers for instance, and may bloat the checkpoint size.
  97   2. If the input pipeline is shared between training and validation, restoring
  98      the checkpoint during validation may override the validation input
  99      pipeline.
 100 
 101   For saving the input pipeline checkpoint alongside the model weights use
 102   `tf.contrib.data.make_saveable_from_iterator` directly to create a
 103   `SaveableObject` and add to the `SAVEABLE_OBJECTS` collection. Note, however,
 104   that you will need to be careful not to restore the training iterator during
 105   eval. You can do that by not adding the iterator to the SAVEABLE_OBJECTS
 106   collector when building the eval graph.
 107   """
 108 
 109   @deprecation.deprecated(
 110       None, "Use `tf.data.experimental.CheckpointInputPipelineHook(...)`.")
 111   def __init__(self, estimator):
 112     super(CheckpointInputPipelineHook, self).__init__(estimator)
