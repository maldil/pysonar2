  24 @deprecation.deprecated(
  25     None, "Use `tf.data.experimental.parse_example_dataset(...)`.")
  26 def parse_example_dataset(features, num_parallel_calls=1):
  27   """A transformation that parses `Example` protos into a `dict` of tensors.
  28 
  29   Parses a number of serialized `Example` protos given in `serialized`. We refer
  30   to `serialized` as a batch with `batch_size` many entries of individual
  31   `Example` protos.
  32 
  33   This op parses serialized examples into a dictionary mapping keys to `Tensor`
  34   and `SparseTensor` objects. `features` is a dict from keys to `VarLenFeature`,
  35   `SparseFeature`, and `FixedLenFeature` objects. Each `VarLenFeature`
  36   and `SparseFeature` is mapped to a `SparseTensor`, and each
  37   `FixedLenFeature` is mapped to a `Tensor`. See `tf.parse_example` for more
  38   details about feature dictionaries.
  39 
  40   Args:
  41    features: A `dict` mapping feature keys to `FixedLenFeature`,
  42      `VarLenFeature`, and `SparseFeature` values.
  43    num_parallel_calls: (Optional.) A `tf.int32` scalar `tf.Tensor`,
  44       representing the number of parsing processes to call in parallel.
  45 
  46   Returns:
  47     A dataset transformation function, which can be passed to
  48     `tf.data.Dataset.apply`.
  49 
  50   Raises:
  51     ValueError: if features argument is None.
  52   """
  53   return parsing_ops.parse_example_dataset(features, num_parallel_calls)
