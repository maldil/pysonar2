  25 @deprecation.deprecated(None,
  26                         "Use `tf.data.experimental.enumerate_dataset(...)`.")
  27 def enumerate_dataset(start=0):
  28   """A transformation that enumerate the elements of a dataset.
  29 
  30   It is Similar to python's `enumerate`.
  31   For example:
  32 
  33   ```python
  34   # NOTE: The following examples use `{ ... }` to represent the
  35   # contents of a dataset.
  36   a = { 1, 2, 3 }
  37   b = { (7, 8), (9, 10) }
  38 
  39   # The nested structure of the `datasets` argument determines the
  40   # structure of elements in the resulting dataset.
  41   a.apply(tf.contrib.data.enumerate(start=5)) == { (5, 1), (6, 2), (7, 3) }
  42   b.apply(tf.contrib.data.enumerate()) == { (0, (7, 8)), (1, (9, 10)) }
  43   ```
  44 
  45   Args:
  46     start: A `tf.int64` scalar `tf.Tensor`, representing the start
  47       value for enumeration.
  48 
  49   Returns:
  50     A `Dataset` transformation function, which can be passed to
  51     `tf.data.Dataset.apply`.
  52   """
  53   return enumerate_ops.enumerate_dataset(start)
