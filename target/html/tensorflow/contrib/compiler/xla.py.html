  63 def compile(computation, inputs=None):  # pylint: disable=redefined-builtin
  64   """Builds an operator that compiles and runs `computation` with XLA.
  65 
  66   Args:
  67     computation: A Python function that builds a computation to apply to the
  68       input. If the function takes n inputs, 'inputs' should be a list of n
  69       tensors.
  70 
  71       `computation` may return a list of operations and tensors.  Tensors must
  72       come before operations in the returned list.  The return value of
  73       `compile` is a list of tensors corresponding to the tensors from the
  74       output of `computation`.
  75 
  76       All `Operation`s returned from `computation` will be executed when
  77       evaluating any of the returned output tensors.
  78     inputs: A list of input tensors or `None` (equivalent to an empty list).
  79 
  80   Returns:
  81     A list of output tensors.
  82   """
  83   # pylint: disable=protected-access
  84   return _compile_internal(computation, inputs)

  87 class XLACompileContext(control_flow_ops.XLAControlFlowContext):
  88   """A `ControlFlowContext` for nodes inside an XLA computation cluster.
  89 
  90   THIS IS ONLY FOR TENSORFLOW INTERNAL IMPLEMENTATION, DO NO USE DIRECTLY.
  91 
  92   The primary role of `XLACompileContext` is to mark operators inside a
  93   xla.compile() computation with attribute "_xla_compile_id=XYZ", where XYZ is
  94   a unique name.
  95 
  96   `ControlFlowContext` is used to perform the annotation since it integrates
  97   with Tensorflow constructs like ResourceVariables. For example, if a
  98   `ResourceVariable` is constructed inside a xla.compile() block, the
  99   `ResourceVariable` implementation can use
 100   `with ops.control_dependencies(None)` to build the variable's definition
 101   outside the compiled computation.
 102   """
 103 
 104   def __init__(self, name, pivot):
 105     """Builds a new XLACompileContext.
 106 
 107     Args:
 108       name: a unique name for the context, used to populate the
 109         `_xla_compile_id` attribute.
 110       pivot: a pivot node. Nodes in the XLACompileContext that do not have any
 111         inputs will have a control dependency on the pivot node. This ensures
 112         that nodes are correctly included in any enclosing control flow
 113         contexts.
 114     """
 115     super(XLACompileContext, self).__init__()
 116     self._name = name
 117     self._name_as_bytes = compat.as_bytes(name)
 118     self._unsupported_ops = []
 119     self._pivot = pivot
 120 
 121   def report_unsupported_operations(self):
 122     if self._unsupported_ops:
 123       op_str = '\n'.join([
 124           '  %s (%s)' % (op.type, op.name)
 125           for op in self._unsupported_ops[:_MAX_WARNING_LINES]
 126       ])
 127       logging.warning('%d unsupported operations found: \n%s',
 128                       len(self._unsupported_ops), op_str)
 129       if len(self._unsupported_ops) > _MAX_WARNING_LINES:
 130         logging.warning('... and %d more',
 131                         len(self._unsupported_ops) - _MAX_WARNING_LINES)
 132 
 133   def AddOp(self, op):
 134     """Create op in XLACompileContext and notifies outer context recursively."""
 135     # pylint: disable=protected-access
 136     if op.type in _BLACKLISTED_OPS:
 137       logging.error(
 138           'Operation of type %s (%s) is not supported in XLA. Execution will '
 139           'fail if this op is used in the graph. ', op.type, op.name)
 140 
 141     # TODO(ycao): Automatically disable summaries instead of reporting them.
 142     if op.type in _UNSUPPORTED_OPS:
 143       self._unsupported_ops.append(op)
 144 
 145     if any(x.dtype._is_ref_dtype for x in op.inputs):
 146       raise NotImplementedError(
 147           'Non-resource Variables are not supported inside XLA computations '
 148           '(operator name: %s)' % op.name)
 149 
 150     if _XLA_COMPILE_ATTR in op.node_def.attr:
 151       raise ValueError('XLA compiled computations cannot be nested, (operator '
 152                        'name: %s)' % op.name)
 153 
 154     op._set_attr(
 155         _XLA_COMPILE_ATTR, attr_value_pb2.AttrValue(s=self._name_as_bytes))
 156 
 157     op.graph.prevent_feeding(op)
 158     op.graph.prevent_fetching(op)
 159 
 160     # Remove any control edges from outer control flow contexts. These may cause
 161     # mismatched frame errors. An example is when one of op's inputs is
 162     # generated in a different While control flow context.
 163     (internal_control_inputs,
 164      external_control_inputs) = self._RemoveExternalControlEdges(op)
 165 
 166     if not op.inputs:
 167       # Add a control edge from the control pivot to this op.
 168       if not internal_control_inputs:
 169         # pylint: disable=protected-access
 170         op._add_control_input(self._pivot)
 171         # pylint: enable=protected-access
 172     else:
 173       for index in xrange(len(op.inputs)):
 174         x = op.inputs[index]
 175         real_x = self.AddValue(x)
 176         if real_x != x:
 177           op._update_input(index, real_x)  # pylint: disable=protected-access
 178 
 179     if external_control_inputs:
 180       # Use an identity to pull control inputs as data inputs. Note that we
 181       # ignore ops which don't have outputs. TODO(phawkins): fix that.
 182       external_control_inputs = [
 183           array_ops.identity(x.outputs[0]).op
 184           for x in external_control_inputs
 185           if x.outputs
 186       ]
 187       # pylint: disable=protected-access
 188       op._add_control_inputs(external_control_inputs)
 189       # pylint: enable=protected-access
 190 
 191     # Mark op's outputs as seen by this context and any outer contexts.
 192     output_names = [x.name for x in op.outputs]
 193     context = self
 194     while context is not None:
 195       # pylint: disable=protected-access
 196       context._values.update(output_names)
 197       context = context._outer_context
 198       # pylint: enable=protected-access
 199 
 200     if self._outer_context:
 201       self._outer_context.AddInnerOp(op)
 202 
 203   def AddValue(self, val):
 204     """Add `val` to the current context and its outer context recursively."""
 205     if val.name in self._values:
 206       # Use the real value if it comes from outer context.
 207       result = self._external_values.get(val.name)
 208       return val if result is None else result
 209 
 210     result = val
 211     self._values.add(val.name)
 212     if self._outer_context:
 213       result = self._outer_context.AddValue(val)
 214       self._values.add(result.name)
 215 
 216     self._external_values[val.name] = result
 217 
 218     return result
 219 
 220   def AddInnerOp(self, op):
 221     self.AddOp(op)
 222     if self._outer_context:
 223       self._outer_context.AddInnerOp(op)
 224 
 225   @property
 226   def grad_state(self):
 227     # Define the gradient loop state associated with the XLACompileContext to
 228     # be None as the XLACompileContext does not get nested nor does the
 229     # grad_state outside the XLACompileContext affect the graph inside so the
 230     # grad_state should be as if this is the top-level gradient state.
 231     return None
 232 
 233   @property
 234   def back_prop(self):
 235     """Forwards to the enclosing while context, if any."""
 236     if self.GetWhileContext():
 237       return self.GetWhileContext().back_prop
 238     return False

 241 def _compile_internal(computation, inputs=None):
 242   """Builds graph operators that compiles and symbolically executes computation.
 243 
 244   Args:
 245     computation: A Python function that builds the computation to compile and
 246       execute.
 247     inputs: A list of input tensors or `None` (equivalent to `[]`). Its order
 248       should match ordering of computation arguments.
 249   Returns:
 250     A list of output tensors from computation.
 251   Raises:
 252     ValueError: If any element in computation outputs is neither an operations
 253       or a value that can be converted to tensor.
 254     TypeError: If `inputs` is not a list or tuple.
 255   """
 256   if inputs is None:
 257     inputs = []
 258 
 259   if not isinstance(inputs, collections.Sequence):
 260     raise TypeError('inputs must be a list')
 261 
 262   # Converts inputs to Tensors.
 263   inputs = [ops.convert_to_tensor(x) for x in inputs]
 264   input_arity = len(inputs)
 265 
 266   arg_error = tpu_function.check_function_argument_count(
 267       computation, input_arity, infeed_queue=None)
 268   if arg_error is not None:
 269     raise TypeError(
 270         'Supplied computation cannot be called with the specified inputs. You '
 271         'specified %d inputs: %s, but the computation needs %s' %
 272         (input_arity, str([i.name for i in inputs[0]]), arg_error))
 273 
 274   cluster_name = ops.get_default_graph().unique_name('cluster')
 275   pivot = control_flow_ops.no_op(name=cluster_name + '/pivot')
 276   context = XLACompileContext(name=cluster_name, pivot=pivot)
 277   try:
 278     context.Enter()
 279 
 280     # Add identity ops so even unused inputs are 'consumed' by the
 281     # computation.
 282     computation_inputs = [
 283         array_ops.identity(x, name='input_{}'.format(i))
 284         for i, x in enumerate(inputs)
 285     ]
 286 
 287     # Only resource variables work inside an XLA computation, so turn on
 288     # resource variables for the computation.
 289     vscope = variable_scope.get_variable_scope()
 290     saved_use_resource = vscope.use_resource
 291     vscope.set_use_resource(True)
 292 
 293     with _disable_summary_context():
 294       outputs = computation(*computation_inputs)
 295 
 296     # Restore variable scope after computation.
 297     vscope.set_use_resource(saved_use_resource)
 298 
 299     # If the computation returns `None`, make it an empty tuple.
 300     if outputs is None:
 301       outputs = tuple()
 302     # If the computation only returned one value, make it a tuple.
 303     if not isinstance(outputs, collections.Sequence):
 304       outputs = (outputs,)
 305 
 306     # Append `no_op` here so that return value of this function always contains
 307     # at least one op that can trigger XlaLaunch node.
 308     outputs += (control_flow_ops.no_op(),)
 309     try:
 310       outputs = [
 311           o if isinstance(o, ops.Operation) else ops.convert_to_tensor(o)
 312           for o in outputs
 313       ]
 314     except Exception as e:
 315       raise ValueError(
 316           'XLA computation function return values must all either be Operations'
 317           ' or convertible to Tensors. Got error: "%s"' % str(e))
 318 
 319     # Separates the returned Operations and Tensors.
 320     output_operations = [o for o in outputs if isinstance(o, ops.Operation)]
 321     output_tensors = [o for o in outputs if not isinstance(o, ops.Operation)]
 322 
 323     if outputs != output_tensors + output_operations:
 324       raise ValueError(
 325           'XLA computation function must return zero or more Tensor values '
 326           'followed by zero or more Operations.')
 327     output_arity = len(output_tensors)
 328 
 329     new_output_tensors = []
 330     for t in output_tensors:
 331       with ops.device(t.device if t.device else ''):
 332         new_output_tensors.append(array_ops.identity(t))
 333 
 334     output_tensors = new_output_tensors
 335     context.ExitResult(output_tensors)
 336   finally:
 337     context.report_unsupported_operations()
 338     context.Exit()
 339 
 340   outputs = [
 341       xla_ops.xla_cluster_output(output_tensors[i], name='output{}'.format(i))
 342       for i in xrange(output_arity)
 343   ]
 344 
 345   with ops.control_dependencies(output_operations):
 346     if output_arity == 0:
 347       # When XLA computation returns only operations and no tensors, a NoOp
 348       # dependent on the operations in outputs is returned. Otherwise final
 349       # outputs would be empty and there is no way to trigger returned
 350       # operations.
 351       return control_flow_ops.no_op(name='output_0')
 352     else:
 353       # Wraps the outputs in identity operators that carries control
 354       # dependencies.
 355       return [
 356           array_ops.identity(outputs[i], name='output_%d' % i)
 357           for i in xrange(output_arity)
 358       ]

 361 @contextlib.contextmanager
 362 def _disable_summary_context():
 363   """Enters a context where all summary ops are skipped.
 364 
 365   Summaries are not yet supported in xla.compile(). So we provide this context
 366   manager that can skip creating summary ops. This is a temporary workaround due
 367   to XLA not supporting summary ops.
 368 
 369   Yields:
 370     None.
 371   """
 372   original_skip_summary_func = summary_op_util.skip_summary
 373   summary_op_util.skip_summary = lambda: True
 374 
 375   try:
 376     yield
 377   finally:
 378     summary_op_util.skip_summary = original_skip_summary_func

 381 class _CapturedObject(object):
 382   """A placeholder to capture an object."""
 383 
 384   def __init__(self):
 385     self._object = None
 386 
 387   def capture(self, o):
 388     if self._object:
 389       raise RuntimeError(
 390           'InternalError: _CapturedObject can capture only once. Please file '
 391           'bug.')
 392 
 393     self._object = o
 394 
 395   def get(self):
 396     return self._object

 399 def _get_scaffold(captured_scaffold_fn):
 400   """Retrieves the Scaffold from `captured_scaffold_fn`."""
 401   scaffold_fn = captured_scaffold_fn.get()
 402 
 403   if not scaffold_fn:
 404     return None
 405 
 406   scaffold = scaffold_fn()
 407   if scaffold is None:
 408     raise ValueError(
 409         'TPUEstimatorSpec.scaffold_fn returns None, which is not allowed')
 410 
 411   return scaffold

 414 class _ModelFnWrapper(object):
 415   """_ModelFnWrapper supports executing model_fn with XLA."""
 416 
 417   def __init__(self, function):
 418     self._model_fn = function
 419 
 420   def __call__(self, features, labels, mode, params):
 421 
 422     # TPUEstimator compiles model_fn when use_tpu=True. To avoid double
 423     # compilation, we use this params['use_tpu'] as a hint. When it is set to
 424     # True, model_fn is called without compilation.
 425     # Note that this condition isn't accurate for the case of exporting a model.
 426     # In that case we should ideally not compile so that user can see detailed
 427     # graph. However, we don't have enough information to tell whether model_fn
 428     # is being called for export mode or not.
 429     # TODO(ycao): Make this condition more accurate when implementing PREDICT
 430     # mode.
 431     if params.get('use_tpu'):
 432       return self._call_model_fn(features, labels, mode, params)
 433 
 434     if mode == model_fn_lib.ModeKeys.TRAIN:
 435       train_step, captured_scaffold_fn = self._make_train_step(
 436           features, labels, params)
 437       (loss,) = compile(train_step)
 438       return model_fn_lib.EstimatorSpec(
 439           mode=mode,
 440           loss=loss,
 441           train_op=array_ops.identity(loss),
 442           scaffold=_get_scaffold(captured_scaffold_fn))
 443     elif mode == model_fn_lib.ModeKeys.EVAL:
 444       eval_step, captured_eval_metric_fn, captured_scaffold_fn = (
 445           self._make_eval_step(features, labels, params))
 446       outputs = compile(eval_step)
 447       loss = outputs[0]
 448 
 449       # Calculate eval_metric_ops if eval_metric_fn is set and captured.
 450       eval_metric_fn = captured_eval_metric_fn.get()
 451       if eval_metric_fn:
 452         eval_metric_fn_tensors = outputs[1:]
 453         eval_metric_ops = eval_metric_fn(*eval_metric_fn_tensors)
 454       else:
 455         eval_metric_ops = None
 456 
 457       return model_fn_lib.EstimatorSpec(
 458           mode=mode,
 459           loss=loss,
 460           eval_metric_ops=eval_metric_ops,
 461           scaffold=_get_scaffold(captured_scaffold_fn))
 462     else:
 463       raise NotImplementedError('%s is not implemented, only TRAIN and EVAL are'
 464                                 ' supported' % mode)
 465 
 466   def _make_train_step(self, features, labels, params):
 467     """Creates a single step of training for xla.compile()."""
 468     captured_scaffold_fn = _CapturedObject()
 469 
 470     def train_step():
 471       """A single step of training."""
 472       estimator_spec = self._call_model_fn(features, labels,
 473                                            model_fn_lib.ModeKeys.TRAIN, params)
 474 
 475       try:
 476         captured_scaffold_fn.capture(estimator_spec.scaffold_fn)
 477       except AttributeError:
 478         captured_scaffold_fn.capture(None)
 479 
 480       # train_step will be run by xla.compile(). xla.compile() only supports
 481       # tensor output while train_op can be either an operation or a tensor.
 482       # Even though xla.compile() automatically adds operation-typed train_op as
 483       # control dependency of other tensor outputs, it doesn't do so for
 484       # tensor-typed train_op. Thus, we need to set it explicitly here.
 485       with ops.control_dependencies([estimator_spec.train_op]):
 486         return array_ops.identity(estimator_spec.loss)
 487 
 488     return train_step, captured_scaffold_fn
 489 
 490   def _make_eval_step(self, features, labels, params):
 491     """Creates a single step of evaluation for xla.compile()."""
 492     captured_eval_metric_fn = _CapturedObject()
 493     captured_scaffold_fn = _CapturedObject()
 494 
 495     def eval_step():
 496       """A single step of evaluation."""
 497       estimator_spec = self._call_model_fn(features, labels,
 498                                            model_fn_lib.ModeKeys.EVAL, params)
 499 
 500       try:
 501         captured_scaffold_fn.capture(estimator_spec.scaffold_fn)
 502       except AttributeError:
 503         captured_scaffold_fn.capture(None)
 504 
 505       eval_metric_fn = None
 506       eval_metric_fn_tensors = []
 507       try:
 508         if estimator_spec.eval_metrics:
 509           (eval_metric_fn, eval_metric_fn_tensors) = estimator_spec.eval_metrics
 510       except AttributeError:
 511         pass
 512 
 513       # If a dictionary is provided, we need to convert it into a list sorted
 514       # according to order of eval_metric_fn positional arguments.
 515       if isinstance(eval_metric_fn_tensors, dict):
 516         eval_metric_fn_args = function_utils.fn_args(eval_metric_fn)
 517         eval_metric_fn_tensors = [
 518             eval_metric_fn_tensors[i] for i in eval_metric_fn_args
 519         ]
 520 
 521       captured_eval_metric_fn.capture(eval_metric_fn)
 522 
 523       return tuple([estimator_spec.loss] + eval_metric_fn_tensors)
 524 
 525     return eval_step, captured_eval_metric_fn, captured_scaffold_fn
 526 
 527   def _call_model_fn(self, features, labels, mode, params):
 528     """Calls the model_fn with required parameters."""
 529     model_fn_args = function_utils.fn_args(self._model_fn)
 530     kwargs = {}
 531 
 532     if 'labels' in model_fn_args:
 533       kwargs['labels'] = labels
 534     elif labels is not None:
 535       raise ValueError(
 536           'model_fn does not take labels, but input_fn returns labels.')
 537     if 'mode' in model_fn_args:
 538       kwargs['mode'] = mode
 539 
 540     if 'params' in model_fn_args:
 541       kwargs['params'] = params
 542 
 543     return self._verify_estimator_spec(
 544         self._model_fn(features=features, **kwargs))
 545 
 546   def _verify_estimator_spec(self, estimator_spec):
 547     """Verifies estimator spec contains correct data."""
 548     # TODO(ycao): Implement estimator spec verification for other modes.
 549 
 550     try:
 551       if estimator_spec.scaffold:
 552         logging.warning('EstimatorSpec.scaffold is ignored with XLA compilation'
 553                         '. Please use TPUEstimatorSpec.scaffold_fn instead.')
 554     except AttributeError:
 555       pass
 556 
 557     try:
 558       if estimator_spec.eval_metric_ops:
 559         raise ValueError('EstimatorSpec.eval_metric_ops is not supported with '
 560                          'XLA compilation. Please use '
 561                          'TPUEstimatorSpec.eval_metrics instead.')
 562     except AttributeError:
 563       pass
 564 
 565     if estimator_spec.mode == model_fn_lib.ModeKeys.EVAL:
 566       # If estimator_spec is of type TPUEstimatorSpec and contains eval_metrics,
 567       # check that eval_metrics contains eval_metric_fn and
 568       # eval_metric_fn_tensors with matching arguments.
 569       try:
 570         eval_metrics = estimator_spec.eval_metrics
 571       except AttributeError:
 572         eval_metrics = None
 573 
 574       if eval_metrics:
 575         (eval_metric_fn, eval_metric_fn_tensors) = eval_metrics
 576         eval_metric_fn_args = function_utils.fn_args(eval_metric_fn)
 577 
 578         if isinstance(eval_metric_fn_tensors, dict):
 579           missing_tensors = [
 580               i for i in eval_metric_fn_args if i not in eval_metric_fn_tensors
 581           ]
 582           additional_tensors = [
 583               i for i in eval_metric_fn_tensors if i not in eval_metric_fn_args
 584           ]
 585 
 586           if missing_tensors:
 587             raise ValueError('Arguments %s are needed by metric_fn (first '
 588                              'element of TPUEstimatorSpec.eval_metrics) but '
 589                              'they are not provided by evaluation tensors '
 590                              '(second element of TPUEstimatorSpec.eval_metrics)'
 591                              '.' % missing_tensors)
 592 
 593           if additional_tensors:
 594             raise ValueError('Arguments %s are provided by evaluation tensors '
 595                              '(second element of TPUEstimatorSpec.eval_metrics)'
 596                              ' but they are not needed by metric_fn (first '
 597                              'element of TPUEstimatorSpec.eval_metrics).' %
 598                              additional_tensors)
 599 
 600     return estimator_spec

 603 def estimator_model_fn(target_model_fn=None):
 604   """estimator_model_fn decorates a model_fn to be compiled for execution.
 605 
 606   Currently only it only works with `TPUEstimator`. If you need to use it with
 607   base `Estimator`, please add `tf.enable_resource_variables()` at beginning of
 608   your program.
 609 
 610   Example 1, decorating model_fn:
 611   ```
 612   @xla.estimator_model_fn()
 613   def model_fn(features, labels, mode, params):
 614     ...
 615     return EstimatorSpec(...)
 616 
 617 
 618   est = Estimator(model_fn=model_fn, ...)
 619   est.train(...)
 620 
 621   ```
 622 
 623   Example 2, decorator as function:
 624   ```
 625   def model_fn(features, labels, mode, params):
 626     ...
 627     return EstimatorSpec(...)
 628 
 629   est = Estimator(model_fn=xla.estimator_model_fn(model_fn), ...)
 630   est.train(...)
 631   ```
 632 
 633   Args:
