  26 class BaseSplitHandler(object):
  27   """Abstract Base class defining split handlers interface."""
  28 
  29   __metaclass__ = abc.ABCMeta
  30 
  31   def __init__(self,
  32                l1_regularization,
  33                l2_regularization,
  34                tree_complexity_regularization,
  35                min_node_weight,
  36                feature_column_group_id,
  37                gradient_shape,
  38                hessian_shape,
  39                multiclass_strategy,
  40                loss_uses_sum_reduction=False,
  41                name=None):
  42     """Constructor for BaseSplitHandler.
  43 
  44     Args:
  45       l1_regularization: L1 regularization applied for this split handler.
  46       l2_regularization: L2 regularization applied for this split handler.
  47       tree_complexity_regularization: Tree complexity regularization applied
  48           for this split handler.
  49       min_node_weight: Minimum sum of weights of examples in each partition to
  50           be considered for splitting.
  51       feature_column_group_id: Feature column group index.
  52       gradient_shape: A TensorShape, containing shape of gradients.
  53       hessian_shape: A TensorShape, containing shape of hessians.
  54       multiclass_strategy: Strategy describing how to treat multiclass problems.
  55       loss_uses_sum_reduction: A scalar boolean tensor that specifies whether
  56           SUM or MEAN reduction was used for the loss.
  57       name: An optional handler name.
  58     """
  59     self._l1_regularization = l1_regularization
  60     self._l2_regularization = l2_regularization
  61     self._tree_complexity_regularization = tree_complexity_regularization
  62     self._min_node_weight = min_node_weight
  63     self._feature_column_group_id = feature_column_group_id
  64     self._name = name or ""
  65     self._multiclass_strategy = multiclass_strategy
  66     self._hessian_shape = hessian_shape
  67     self._gradient_shape = gradient_shape
  68     self._loss_uses_sum_reduction = loss_uses_sum_reduction

  70   def scheduled_reads(self):
  71     """Returns the list of `ScheduledOp`s required for update_stats."""
  72     return []

  74   @abc.abstractmethod
  75   def update_stats(self, stamp_token, example_partition_ids, gradients,
  76                    hessians, empty_gradients, empty_hessians, weights,
  77                    is_active, scheduled_reads):
  78     """Updates the state for this split handler.
  79 
  80     Args:
  81       stamp_token: An int32 scalar tensor containing the current stamp token.
  82       example_partition_ids: A dense tensor, containing an int32 for each
  83         example which is the partition id that the example ends up in.
  84       gradients: A dense tensor of gradients.
  85       hessians: A dense tensor of hessians.
  86       empty_gradients: A dense empty tensor of the same shape (for dimensions >
  87         0) as gradients.
  88       empty_hessians: A dense empty tensor of the same shape (for dimensions >
  89         0) as hessians.
  90       weights: A dense float32 tensor with a weight for each example.
  91       is_active: A boolean tensor that says if this handler is active or not.
  92           One value for the current layer and one value for the next layer.
  93       scheduled_reads: List of results from the scheduled reads.
  94 
  95     Returns:
  96       A tuple of the op that updates the stats for this handler and a list of
  97       `ScheduledOp`s.
  98     """
  99 
 100   def update_stats_sync(self, stamp_token, example_partition_ids, gradients,
 101                         hessians, empty_gradients, empty_hessians, weights,
 102                         is_active):
 103     """Updates the state for this split handler running the scheduled I/O.
 104 
 105     Args:
 106       stamp_token: An int32 scalar tensor containing the current stamp token.
 107       example_partition_ids: A dense tensor, containing an int32 for each
 108         example which is the partition id that the example ends up in.
 109       gradients: A dense tensor of gradients.
 110       hessians: A dense tensor of hessians.
 111       empty_gradients: A dense empty tensor of the same shape (for dimensions >
 112         0) as gradients.
 113       empty_hessians: A dense empty tensor of the same shape (for dimensions >
 114         0) as hessians.
 115       weights: A dense float32 tensor with a weight for each example.
 116       is_active: A boolean tensor that says if this handler is active or not.
 117           One value for the current layer and one value for the next layer.
 118 
 119     Returns:
 120       Op that updates the stats for this handler.
 121     """
 122     handler_reads = {self: self.scheduled_reads()}
 123     handler_results = batch_ops_utils.run_handler_scheduled_ops(
 124         handler_reads, stamp_token, None)
 125     update_1, scheduled_updates = self.update_stats(
 126         stamp_token, example_partition_ids, gradients, hessians,
 127         empty_gradients, empty_hessians, weights, is_active,
 128         handler_results[self])
 129     update_2 = batch_ops_utils.run_handler_scheduled_ops({
 130         self: scheduled_updates
 131     }, stamp_token, None)
 132     return control_flow_ops.group(update_1, *update_2[self])
 133 
 134   @abc.abstractmethod
 135   def reset(self, stamp_token, next_stamp_token):
 136     """Resets the state maintained by the handler."""

 138   @abc.abstractmethod
 139   def make_splits(self, stamp_token, next_stamp_token, class_id):
 140     """Create the best split using the accumulated stats and flush the state.
 141 
 142     This should only be called by the master.
 143 
 144     Args:
 145       stamp_token: An int32 scalar tensor containing the current stamp token.
 146       next_stamp_token: An int32 scalar tensor containing the stamp token for
 147         the next iteration.
 148       class_id: what class id the handler gathers stats for (for tree per class
 149         strategy). When class_id=-1, the strategy is not tree per class.
 150     Returns:
 151       A tuple (are_splits_ready, partition_id, gain, split_info) where
 152       are_splits_ready is a scalar boolean tensor, partition_id is a rank 1,
 153       int32 tensor, gain is a rank 1 float32 tensor and split_info is a rank 1
 154       string tensor containing serialized SplitInfo protos.
 155     """
