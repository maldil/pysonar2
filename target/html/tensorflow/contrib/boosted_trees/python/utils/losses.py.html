  29 def per_example_squared_hinge_loss(labels, weights, predictions):
  30   loss = losses.hinge_loss(labels=labels, logits=predictions, weights=weights)
  31   return math_ops.square(loss), control_flow_ops.no_op()

  34 def per_example_logistic_loss(labels, weights, predictions):
  35   """Logistic loss given labels, example weights and predictions.
  36 
  37   Args:
  38     labels: Rank 2 (N, 1) tensor of per-example labels.
  39     weights: Rank 2 (N, 1) tensor of per-example weights.
  40     predictions: Rank 2 (N, 1) tensor of per-example predictions.
  41 
  42   Returns:
  43     loss: A Rank 2 (N, 1) tensor of per-example logistic loss.
  44     update_op: An update operation to update the loss's internal state.
  45   """
  46   labels = math_ops.to_float(labels)
  47   unweighted_loss = nn.sigmoid_cross_entropy_with_logits(
  48       labels=labels, logits=predictions)
  49   return unweighted_loss * weights, control_flow_ops.no_op()

  52 # This is classical form of Maximum entropy loss, that is twice differentiable
  53 # (sparse_softmax_cross_entropy which is what we go for is not twice
  54 # differentiable).
  55 def per_example_maxent_loss(labels, weights, logits, num_classes, eps=1e-15):
  56   """Maximum entropy loss for multiclass problems.
  57 
  58   Maximum entropy is a generalization of logistic loss for the case when more
  59   than 2 classes are present.
  60 
  61   Args:
  62     labels: Rank 2 (N, 1) or Rank 1 (N) tensor of per-example labels.
  63     weights: Rank 2 (N, 1) tensor of per-example weights.
  64     logits: Rank 2 (N, K) tensor of per-example predictions, K - num of
  65     classes.
  66     num_classes: number of classes in classification task. Used to expand label
  67     indices into one-hot encodings.
  68     eps: tolerance, used as a minimum possible value.
  69 
  70   Returns:
  71     loss: A Rank 2 (N, 1) tensor of per-example maxent loss
  72     update_op: An update operation to update the loss's internal state.
  73   """
  74   labels = math_ops.to_int64(labels)
  75   # If labels are of rank 1, make them rank 2.
  76   labels_shape = labels.get_shape()
  77   if len(labels_shape) != 2:
  78     labels = array_ops.expand_dims(labels, 1)
  79   # Labels are indices of classes, convert them to one hot encodings.
  80   target_one_hot = array_ops.one_hot(indices=labels, depth=num_classes)
  81   labels = math_ops.reduce_sum(
  82       input_tensor=target_one_hot, reduction_indices=[1])
  83   labels = math_ops.to_float(labels)
  84 
  85   # Calculate softmax probabilities for each class.
  86   unnormalized_probs = math_ops.exp(logits)
  87   normalizers = math_ops.reduce_sum(unnormalized_probs, 1, keepdims=True)
  88   softmax_predictions = math_ops.divide(unnormalized_probs,
  89                                         math_ops.add(normalizers, eps))
  90 
  91   # Pull out the probabilities for real label.
  92   probs_for_real_class = math_ops.reduce_sum(labels * softmax_predictions, 1)
  93 
  94   # Add handling for values near 0 and 1.
  95   zeros = array_ops.zeros_like(probs_for_real_class, dtype=logits.dtype) + eps
  96   one_minus_eps = array_ops.ones_like(
  97       probs_for_real_class, dtype=logits.dtype) - eps
  98 
  99   # Take maximum(eps, pred)
 100   cond = (probs_for_real_class >= eps)
 101   probs_for_real_class = array_ops.where(cond, probs_for_real_class, zeros)
 102 
 103   # Take minimum(1-eps, pred)
 104   cond = (probs_for_real_class <= 1 - eps)
 105   probs_for_real_class = array_ops.where(cond, probs_for_real_class,
 106                                          one_minus_eps)
 107 
 108   unweighted_loss = array_ops.expand_dims(-math_ops.log(probs_for_real_class),
 109                                           1)
 110   if weights is None:
 111     return unweighted_loss, control_flow_ops.no_op()
 112   else:
 113     return unweighted_loss * weights, control_flow_ops.no_op()

 116 def per_example_squared_loss(labels, weights, predictions):
 117   """Squared loss given labels, example weights and predictions.
 118 
 119   Args:
 120     labels: Rank 2 (N, D) tensor of per-example labels.
 121     weights: Rank 2 (N, 1) tensor of per-example weights.
 122     predictions: Rank 2 (N, D) tensor of per-example predictions.
 123 
 124   Returns:
 125     loss: A Rank 2 (N, 1) tensor of per-example squared loss.
 126     update_op: An update operation to update the loss's internal state.
 127   """
 128   unweighted_loss = math_ops.reduce_sum(
 129       math_ops.square(predictions - labels), 1, keepdims=True)
 130 
 131   return unweighted_loss * weights, control_flow_ops.no_op()

 134 def per_example_exp_loss(labels, weights, predictions, name=None, eps=0.1):
 135   """Trimmed exponential loss given labels, example weights and predictions.
 136 
 137   Note that this is only for binary classification.
 138   If logistic loss tries to make sure that the classifier is certain of its
 139   predictions, exp loss says: "as long as it got it correct, even barely, i
 140   don't care". Can be used on noisy data, or when you don't care about getting
 141   the actual probabilities from the model, just the correct label.
 142 
 143   The loss returns is exp(-targets*modified_predictions), where
 144   modified_predictions are 1 if sigmoid is >= 0.5+eps (eg we predict positive
 145   class), -1 if sigmoid < 0.5-eps (e.g. we predict negative class) and ax+b in
 146   the interval 0.5-eps, 0.5+eps, where a = 1/eps, b=1/(2eps).
 147 
 148   Args:
 149     labels: Rank 2 (N, D) tensor of per-example labels.
 150     weights: Rank 2 (N, 1) tensor of per-example weights.
 151     predictions: Rank 2 (N, D) tensor of per-example predictions.
 152     name: A name for the operation (optional).
 153     eps: For the range (0.5-eps, 0.5+eps) we set the predictions to be ax+b.
 154 
 155   Returns:
 156     loss: A Rank 2 (N, 1) tensor of per-example exp loss
 157     update_op: An update operation to update the loss's internal state.
 158   """
 159 
 160   def exp_with_logits(name, eps, labels=None, logits=None):
 161     """Computes exponential loss given `logits`.
 162 
 163     The loss returns is exp(-targets*modified_predictions), where
 164     modified_predictions are 1 if sigmoid is >= 0.5+eps (eg we predict positive
 165     class), -1 if sigmoid < 0.5-eps (e.g. we predict negative class) and ax+b in
 166     the interval 0.5-eps, 0.5+eps, where a = 1/eps, b=1/(2eps).
 167 
 168     Args:
 169       name: A name for the operation (optional).
 170       eps: For the range (0.5-eps, 0.5+eps) we set the predictions to be ax+b.
 171       labels: A `Tensor` of the same type and shape as `logits`.
 172       logits: A `Tensor` of type `float32` or `float64`.
 173 
 174     Returns:
 175       A `Tensor` of the same shape as `logits` with the componentwise
 176       exponential losses.
 177 
 178     Raises:
 179       ValueError: If `logits` and `labels` do not have the same shape.
 180     """
 181     with ops.name_scope(name, "exp_loss", [logits, labels]) as name:
 182       logits = ops.convert_to_tensor(logits, name="logits")
 183       labels = ops.convert_to_tensor(labels, name="labels")
 184       try:
 185         labels.get_shape().merge_with(logits.get_shape())
 186       except ValueError:
 187         raise ValueError("logits and labels must have the same shape (%s vs %s)"
 188                          % (logits.get_shape(), labels.get_shape()))
 189 
 190     # Default threshold to switch between classes
 191     zeros = array_ops.zeros_like(logits, dtype=logits.dtype)
 192     ones = array_ops.ones_like(logits, dtype=logits.dtype)
 193     neg_ones = -array_ops.ones_like(logits, dtype=logits.dtype)
 194 
 195     # Convert labels to 1 and -1
 196     cond_labels = (labels > zeros)
 197     labels_converted = array_ops.where(cond_labels, ones, neg_ones)
 198 
 199     # Convert predictions to 1 and -1
 200     # The loss we build is min(1, max(-1,ax+b))
 201     # where a=1/eps, b=-1/2eps.
 202 
 203     a = 1.0 / eps
 204     b = -1.0 / 2 / eps
 205     probs = math_ops.sigmoid(logits)
 206     y = a * probs + b
 207     # Build max(-1, ax+b)
 208     cond = (y < -1)
 209     max_res = array_ops.where(cond, neg_ones, y)
 210     # Build min part
 211     cond = (max_res > 1)
 212     min_res = array_ops.where(cond, ones, max_res)
 213     preds_converted = min_res
 214     return math_ops.exp(-preds_converted * labels_converted)
 215 
 216   labels = math_ops.to_float(labels)
 217   unweighted_loss = exp_with_logits(
 218       name=name, eps=eps, labels=labels, logits=predictions)
 219   return unweighted_loss * weights, control_flow_ops.no_op()

 222 def per_example_full_exp_loss(labels, weights, predictions, name=None):
 223   """Full exponential loss given labels, example weights and predictions.
 224 
 225   Note that this is only for binary classification.
 226   The loss returns is exp(-targets*logits), where targets are converted to -1
 227   and 1.
 228 
 229   Args:
 230     labels: Rank 2 (N, D) tensor of per-example labels.
 231     weights: Rank 2 (N, 1) tensor of per-example weights.
 232     predictions: Rank 2 (N, D) tensor of per-example predictions.
 233     name: A name for the operation (optional).
 234 
 235   Returns:
 236     loss: A Rank 2 (N, 1) tensor of per-example exp loss
 237     update_op: An update operation to update the loss's internal state.
 238   """
 239 
 240   def full_exp_with_logits(name, labels=None, logits=None):
 241     """Computes exponential loss given `logits`.
 242 
 243     Args:
 244       name: A name for the operation (optional).
 245       labels: A `Tensor` of the same type and shape as `logits`.
 246       logits: A `Tensor` of type `float32` or `float64`.
 247 
 248     Returns:
 249       A `Tensor` of the same shape as `logits` with the componentwise
 250       exponential losses.
 251 
 252     Raises:
 253       ValueError: If `logits` and `labels` do not have the same shape.
 254     """
 255     with ops.name_scope(name, "exp_loss", [logits, labels]) as name:
 256       logits = ops.convert_to_tensor(logits, name="logits")
 257       labels = ops.convert_to_tensor(labels, name="labels")
 258       try:
 259         labels.get_shape().merge_with(logits.get_shape())
 260       except ValueError:
 261         raise ValueError("logits and labels must have the same shape (%s vs %s)"
 262                          % (logits.get_shape(), labels.get_shape()))
 263 
 264     # Default threshold of 0 to switch between classes
 265     zeros = array_ops.zeros_like(logits, dtype=logits.dtype)
 266     ones = array_ops.ones_like(logits, dtype=logits.dtype)
 267     neg_ones = -array_ops.ones_like(logits, dtype=logits.dtype)
 268 
 269     # Convert labels to 1 and -1
 270     cond_labels = (labels > zeros)
 271     labels_converted = array_ops.where(cond_labels, ones, neg_ones)
 272 
 273     return math_ops.exp(-1.0 * logits * labels_converted)
 274 
 275   labels = math_ops.to_float(labels)
 276   unweighted_loss = full_exp_with_logits(
 277       name=name, labels=labels, logits=predictions)
 278   return unweighted_loss * weights, control_flow_ops.no_op()
