  74 def _get_column_by_index(tensor, indices):
  75   """Returns columns from a 2-D tensor by index."""
  76   shape = array_ops.shape(tensor)
  77   p_flat = array_ops.reshape(tensor, [-1])
  78   i_flat = array_ops.reshape(
  79       array_ops.reshape(math_ops.range(0, shape[0]) * shape[1], [-1, 1]) +
  80       indices, [-1])
  81   return array_ops.reshape(array_ops.gather(p_flat, i_flat), [shape[0], -1])

  84 def _make_predictions_dict(stamp,
  85                            logits,
  86                            partition_ids,
  87                            ensemble_stats,
  88                            used_handlers,
  89                            leaf_index=None):
  90   """Returns predictions for the given logits and n_classes.
  91 
  92   Args:
  93     stamp: The ensemble stamp.
  94     logits: A rank 2 `Tensor` with shape [batch_size, n_classes - 1]. that
  95       contains predictions when no dropout was applied.
  96     partition_ids: A rank 1 `Tensor` with shape [batch_size].
  97     ensemble_stats: A TreeEnsembleStatsOp result tuple.
  98     used_handlers: A TreeEnsembleUsedHandlerOp result tuple of an int and a
  99       boolean mask.
 100     leaf_index: A rank 2 `Tensor` with shape [batch_size, number of trees]. that
 101       contains leaf id for each example prediction.
 102 
 103   Returns:
 104     A dict of predictions.
 105   """
 106   result = {}
 107   result[ENSEMBLE_STAMP] = stamp
 108   result[PREDICTIONS] = logits
 109   result[PARTITION_IDS] = partition_ids
 110   result[NUM_LAYERS_ATTEMPTED] = ensemble_stats.attempted_layers
 111   result[NUM_TREES_ATTEMPTED] = ensemble_stats.attempted_trees
 112   result[NUM_USED_HANDLERS] = used_handlers.num_used_handlers
 113   result[USED_HANDLERS_MASK] = used_handlers.used_handlers_mask
 114   if leaf_index is not None:
 115     result[LEAF_INDEX] = leaf_index
 116   return result

 119 class _OpRoundRobinStrategy(object):
 120   """Returns the next ps task index for placement via per-Op round-robin order.
 121 
 122   This strategy works slightly better for the GBDT graph because of using
 123   custom resources which vary significantly in compute cost.
 124   """
 125 
 126   def __init__(self, ps_ops, num_tasks):
 127     """Create a new `_RoundRobinStrategy`.
 128 
 129     Args:
 130       ps_ops: List of Op types to place on PS.
 131       num_tasks: Number of ps tasks to cycle among.
 132     """
 133     next_task = 0
 134     self._next_task_per_op = {}
 135     for op in ps_ops:
 136       self._next_task_per_op[op] = next_task
 137       next_task = (next_task + 1) % num_tasks if num_tasks else 0
 138     self._num_tasks = num_tasks
 139 
 140   def __call__(self, op):
 141     """Choose a ps task index for the given `Operation`.
 142 
 143     Args:
 144       op: An `Operation` to be placed on ps.
 145 
 146     Returns:
 147       The next ps task index to use for the `Operation`. Returns the next
 148       index, in the range `[offset, offset + num_tasks)`.
 149 
 150     Raises:
 151       ValueError: If attempting to place non-PS Op.
 152     """
 153     if op.type not in self._next_task_per_op:
 154       raise ValueError("Unknown op type '%s' for placement:" % op.type)
 155     task = self._next_task_per_op[op.type]
 156     self._next_task_per_op[op.type] = ((task + 1) % self._num_tasks
 157                                        if self._num_tasks else 0)
 158     return task

 161 def extract_features(features, feature_columns, use_core_columns):
 162   """Extracts columns from a dictionary of features.
 163 
 164   Args:
 165     features: `dict` of `Tensor` objects.
 166     feature_columns: A list of feature_columns.
 167 
 168   Returns:
 169     Seven values:
 170       - A list of all feature column names.
 171       - A list of dense floats.
 172       - A list of sparse float feature indices.
 173       - A list of sparse float feature values.
 174       - A list of sparse float feature shapes.
 175       - A list of sparse int feature indices.
 176       - A list of sparse int feature values.
 177       - A list of sparse int feature shapes.
 178   Raises:
 179     ValueError: if features is not valid.
 180   """
 181   if not features:
 182     raise ValueError("Features dictionary must be specified.")
 183 
 184   # Make a shallow copy of features to ensure downstream usage
 185   # is unaffected by modifications in the model function.
 186   features = copy.copy(features)
 187   if feature_columns:
 188     scope = "gbdt"
 189     with variable_scope.variable_scope(scope):
 190       feature_columns = list(feature_columns)
 191       transformed_features = collections.OrderedDict()
 192       for fc in feature_columns:
 193         # pylint: disable=protected-access
 194         if use_core_columns:
 195           # pylint: disable=protected-access
 196           tensor = fc_core._transform_features(features, [fc])[fc]
 197           transformed_features[fc.name] = tensor
 198         elif isinstance(fc, feature_column_lib._EmbeddingColumn):
 199           # pylint: enable=protected-access
 200           transformed_features[fc.name] = fc_core.input_layer(
 201               features, [fc], weight_collections=[scope])
 202         else:
 203           result = feature_column_ops.transform_features(features, [fc])
 204           if len(result) > 1:
 205             raise ValueError("Unexpected number of output features")
 206           transformed_features[fc.name] = result[list(result.keys())[0]]
 207     features = transformed_features
 208 
 209   dense_float_names = []
 210   dense_floats = []
 211   sparse_float_names = []
 212   sparse_float_indices = []
 213   sparse_float_values = []
 214   sparse_float_shapes = []
 215   sparse_int_names = []
 216   sparse_int_indices = []
 217   sparse_int_values = []
 218   sparse_int_shapes = []
 219   for key in sorted(features.keys()):
 220     tensor = features[key]
 221     # TODO(nponomareva): consider iterating over feature columns instead.
 222     if isinstance(tensor, tuple):
 223       # Weighted categorical feature.
 224       categorical_tensor = tensor[0]
 225       weight_tensor = tensor[1]
 226 
 227       shape = categorical_tensor.dense_shape
 228       indices = array_ops.concat([
 229           array_ops.slice(categorical_tensor.indices, [0, 0], [-1, 1]),
 230           array_ops.expand_dims(
 231               math_ops.to_int64(categorical_tensor.values), -1)
 232       ], 1)
 233       tensor = sparse_tensor.SparseTensor(
 234           indices=indices, values=weight_tensor.values, dense_shape=shape)
 235 
 236     if isinstance(tensor, sparse_tensor.SparseTensor):
 237       if tensor.values.dtype == dtypes.float32:
 238         sparse_float_names.append(key)
 239         sparse_float_indices.append(tensor.indices)
 240         sparse_float_values.append(tensor.values)
 241         sparse_float_shapes.append(tensor.dense_shape)
 242       elif tensor.values.dtype == dtypes.int64:
 243         sparse_int_names.append(key)
 244         sparse_int_indices.append(tensor.indices)
 245         sparse_int_values.append(tensor.values)
 246         sparse_int_shapes.append(tensor.dense_shape)
 247       else:
 248         raise ValueError("Unsupported sparse feature %s with dtype %s." %
 249                          (tensor.indices.name, tensor.dtype))
 250     else:
 251       if tensor.dtype == dtypes.float32:
 252         if len(tensor.shape) > 1 and tensor.shape[1] > 1:
 253           unstacked = array_ops.unstack(tensor, axis=1)
 254           for i in range(len(unstacked)):
 255             dense_float_names.append(_FEATURE_NAME_TEMPLATE % (key, i))
 256             dense_floats.append(array_ops.reshape(unstacked[i], [-1, 1]))
 257         else:
 258           dense_float_names.append(key)
 259           dense_floats.append(tensor)
 260       else:
 261         raise ValueError("Unsupported dense feature %s with dtype %s." %
 262                          (tensor.name, tensor.dtype))
 263   # Feature columns are logically organized into incrementing slots starting
 264   # from dense floats, then sparse floats then sparse ints.
 265   fc_names = (dense_float_names + sparse_float_names + sparse_int_names)
 266   return (fc_names, dense_floats, sparse_float_indices, sparse_float_values,
 267           sparse_float_shapes, sparse_int_indices, sparse_int_values,
 268           sparse_int_shapes)

 271 def _dropout_params(mode, ensemble_stats):
 272   """Returns parameters relevant for dropout.
 273 
 274   Args:
 275     mode: Train/Eval/Infer
 276     ensemble_stats: A TreeEnsembleStatsOp result tuple.
 277 
 278   Returns:
 279     Whether to apply dropout and a dropout seed.
 280   """
 281   if mode == learn.ModeKeys.TRAIN:
 282     # Do dropout only during training.
 283     apply_dropout = True
 284     seed = ensemble_stats.attempted_trees
 285   else:
 286     seed = -1
 287     apply_dropout = False
 288   return apply_dropout, seed

 291 class GradientBoostedDecisionTreeModel(object):
 292   """A GBDT model function."""
 293 
 294   def __init__(self,
 295                is_chief,
 296                num_ps_replicas,
 297                ensemble_handle,
 298                center_bias,
 299                examples_per_layer,
 300                learner_config,
 301                features,
 302                logits_dimension,
 303                loss_reduction=losses.Reduction.SUM_OVER_NONZERO_WEIGHTS,
 304                feature_columns=None,
 305                use_core_columns=False,
 306                output_leaf_index=False,
 307                output_leaf_index_modes=None,
 308                num_quantiles=100):
 309     """Construct a new GradientBoostedDecisionTreeModel function.
 310 
 311     Args:
 312       is_chief: Whether to build the chief graph.
 313       num_ps_replicas: Number of parameter server replicas, can be 0.
 314       ensemble_handle: A handle to the ensemble variable.
 315       center_bias: Whether to center the bias before growing trees.
 316       examples_per_layer: Number of examples to accumulate before growing a tree
 317         layer. It can also be a function that computes the number of examples
 318         based on the depth of the layer that's being built.
 319       learner_config: A learner config.
 320       features: `dict` of `Tensor` objects.
 321       logits_dimension: An int, the dimension of logits.
 322       loss_reduction: Either `SUM_OVER_NONZERO_WEIGHTS` (mean) or `SUM`.
 323       feature_columns: A list of feature columns.
 324       use_core_columns: A boolean specifying whether core feature columns are
 325         used.
 326       output_leaf_index: A boolean variable indicating whether to output leaf
 327         index into predictions dictionary.
 328       output_leaf_index_modes: A list of modes from (TRAIN, EVAL, INFER) which
 329         dictates when leaf indices will be outputted. By default, leaf indices
 330         are only outputted in INFER mode.
 331       num_quantiles: Number of quantiles to build for numeric feature values.
 332 
 333     Raises:
 334       ValueError: if inputs are not valid.
 335     """
 336     if ensemble_handle is None:
 337       raise ValueError("ensemble_handle must be specified.")
 338 
 339     if learner_config is None:
 340       raise ValueError("learner_config must be specified.")
 341 
 342     if learner_config.num_classes < 2:
 343       raise ValueError("Number of classes must be >=2")
 344 
 345     self._logits_dimension = logits_dimension
 346     self._is_chief = is_chief
 347     self._num_ps_replicas = num_ps_replicas
 348     self._ensemble_handle = ensemble_handle
 349     self._center_bias = center_bias
 350     self._examples_per_layer = examples_per_layer
 351 
 352     # Check loss reduction value.
 353     if (loss_reduction != losses.Reduction.SUM and
 354         loss_reduction != losses.Reduction.SUM_OVER_NONZERO_WEIGHTS):
 355       raise ValueError(
 356           "Invalid loss reduction is provided: %s." % loss_reduction)
 357     self._loss_reduction = loss_reduction
 358 
 359     # Fill in the defaults.
 360     if (learner_config.multi_class_strategy ==
 361         learner_pb2.LearnerConfig.MULTI_CLASS_STRATEGY_UNSPECIFIED):
 362       if logits_dimension == 1:
 363         learner_config.multi_class_strategy = (
 364             learner_pb2.LearnerConfig.TREE_PER_CLASS)
 365       else:
 366         learner_config.multi_class_strategy = (
 367             learner_pb2.LearnerConfig.DIAGONAL_HESSIAN)
 368 
 369     if logits_dimension == 1 or learner_config.multi_class_strategy == (
 370         learner_pb2.LearnerConfig.TREE_PER_CLASS):
 371       self._gradient_shape = tensor_shape.scalar()
 372       self._hessian_shape = tensor_shape.scalar()
 373     else:
 374       if center_bias:
 375         raise ValueError("Center bias should be False for multiclass.")
 376 
 377       self._gradient_shape = tensor_shape.TensorShape([logits_dimension])
 378       if (learner_config.multi_class_strategy ==
 379           learner_pb2.LearnerConfig.FULL_HESSIAN):
 380         self._hessian_shape = tensor_shape.TensorShape(
 381             ([logits_dimension, logits_dimension]))
 382       else:
 383         # Diagonal hessian strategy.
 384         self._hessian_shape = tensor_shape.TensorShape(([logits_dimension]))
 385     if (learner_config.growing_mode ==
 386         learner_pb2.LearnerConfig.GROWING_MODE_UNSPECIFIED):
 387       learner_config.growing_mode = learner_pb2.LearnerConfig.LAYER_BY_LAYER
 388 
 389     if (learner_config.pruning_mode ==
 390         learner_pb2.LearnerConfig.PRUNING_MODE_UNSPECIFIED):
 391       learner_config.pruning_mode = learner_pb2.LearnerConfig.POST_PRUNE
 392 
 393     if learner_config.constraints.max_tree_depth == 0:
 394       # Use 6 as the default maximum depth.
 395       learner_config.constraints.max_tree_depth = 6
 396 
 397     tuner = learner_config.learning_rate_tuner.WhichOneof("tuner")
 398     if not tuner:
 399       learner_config.learning_rate_tuner.fixed.learning_rate = 0.1
 400 
 401     self._learner_config = learner_config
 402     self._feature_columns = feature_columns
 403     self._learner_config_serialized = learner_config.SerializeToString()
 404     self._num_quantiles = num_quantiles
 405     self._max_tree_depth = variables.VariableV1(
 406         initial_value=self._learner_config.constraints.max_tree_depth)
 407     self._attempted_trees = variables.VariableV1(
 408         initial_value=array_ops.zeros([], dtypes.int64),
 409         trainable=False,
 410         name="attempted_trees")
 411     self._finalized_trees = variables.VariableV1(
 412         initial_value=array_ops.zeros([], dtypes.int64),
 413         trainable=False,
 414         name="finalized_trees")
 415     if not features:
 416       raise ValueError("Features dictionary must be specified.")
 417     (fc_names, dense_floats, sparse_float_indices, sparse_float_values,
 418      sparse_float_shapes, sparse_int_indices,
 419      sparse_int_values, sparse_int_shapes) = extract_features(
 420          features, self._feature_columns, use_core_columns)
 421     logging.info("Active Feature Columns: " + str(fc_names))
 422     logging.info("Learner config: " + str(learner_config))
 423     self._fc_names = fc_names
 424     self._dense_floats = dense_floats
 425     self._sparse_float_indices = sparse_float_indices
 426     self._sparse_float_values = sparse_float_values
 427     self._sparse_float_shapes = sparse_float_shapes
 428     self._sparse_int_indices = sparse_int_indices
 429     self._sparse_int_values = sparse_int_values
 430     self._sparse_int_shapes = sparse_int_shapes
 431     self._reduce_dim = (
 432         self._learner_config.multi_class_strategy ==
 433         learner_pb2.LearnerConfig.TREE_PER_CLASS and
 434         learner_config.num_classes == 2)
 435 
 436     if output_leaf_index_modes is None:
 437       output_leaf_index_modes = [learn.ModeKeys.INFER]
 438     elif not all(
 439         mode in (learn.ModeKeys.TRAIN, learn.ModeKeys.EVAL,
 440                  learn.ModeKeys.INFER) for mode in output_leaf_index_modes):
 441       raise ValueError("output_leaf_index_modes should only contain ModeKeys.")
 442 
 443     self._output_leaf_index = output_leaf_index
 444     self._output_leaf_index_modes = output_leaf_index_modes
 445 
 446   def _predict_and_return_dict(self, ensemble_handle, ensemble_stamp, mode):
 447     """Runs prediction and returns a dictionary of the prediction results.
 448 
 449     Args:
 450       ensemble_handle: ensemble resource handle.
 451       ensemble_stamp: stamp of ensemble resource.
 452       mode: learn.ModeKeys.TRAIN or EVAL or INFER.
 453 
 454     Returns:
 455       a dictionary of prediction results -
 456         ENSEMBLE_STAMP, PREDICTION, PARTITION_IDS,
 457         NUM_LAYER_ATTEMPTED, NUM_TREES_ATTEMPTED.
 458     """
 459     ensemble_stats = training_ops.tree_ensemble_stats(ensemble_handle,
 460                                                       ensemble_stamp)
 461     num_handlers = (
 462         len(self._dense_floats) + len(self._sparse_float_shapes) + len(
 463             self._sparse_int_shapes))
 464     # Used during feature selection.
 465     used_handlers = model_ops.tree_ensemble_used_handlers(
 466         ensemble_handle, ensemble_stamp, num_all_handlers=num_handlers)
 467 
 468     # We don't need dropout info - we can always restore it based on the
 469     # seed.
 470     apply_dropout, seed = _dropout_params(mode, ensemble_stats)
 471     # Make sure ensemble stats run. This will check that the ensemble has
 472     # the right stamp.
 473     with ops.control_dependencies(ensemble_stats):
 474       leaf_index = None
 475       if self._output_leaf_index and mode in self._output_leaf_index_modes:
 476         predictions, _, leaf_index = (
 477             prediction_ops).gradient_trees_prediction_verbose(
 478                 ensemble_handle,
 479                 seed,
 480                 self._dense_floats,
 481                 self._sparse_float_indices,
 482                 self._sparse_float_values,
 483                 self._sparse_float_shapes,
 484                 self._sparse_int_indices,
 485                 self._sparse_int_values,
 486                 self._sparse_int_shapes,
 487                 learner_config=self._learner_config_serialized,
 488                 apply_dropout=apply_dropout,
 489                 apply_averaging=mode != learn.ModeKeys.TRAIN,
 490                 use_locking=True,
 491                 center_bias=self._center_bias,
 492                 reduce_dim=self._reduce_dim)
 493       else:
 494         leaf_index = None
 495         predictions, _ = prediction_ops.gradient_trees_prediction(
 496             ensemble_handle,
 497             seed,
 498             self._dense_floats,
 499             self._sparse_float_indices,
 500             self._sparse_float_values,
 501             self._sparse_float_shapes,
 502             self._sparse_int_indices,
 503             self._sparse_int_values,
 504             self._sparse_int_shapes,
 505             learner_config=self._learner_config_serialized,
 506             apply_dropout=apply_dropout,
 507             apply_averaging=mode != learn.ModeKeys.TRAIN,
 508             use_locking=True,
 509             center_bias=self._center_bias,
 510             reduce_dim=self._reduce_dim)
 511       partition_ids = prediction_ops.gradient_trees_partition_examples(
 512           ensemble_handle,
 513           self._dense_floats,
 514           self._sparse_float_indices,
 515           self._sparse_float_values,
 516           self._sparse_float_shapes,
 517           self._sparse_int_indices,
 518           self._sparse_int_values,
 519           self._sparse_int_shapes,
 520           use_locking=True)
 521 
 522     return _make_predictions_dict(ensemble_stamp, predictions, partition_ids,
 523                                   ensemble_stats, used_handlers, leaf_index)
 524 
 525   def predict(self, mode):
 526     """Returns predictions given the features and mode.
 527 
 528     Args:
 529       mode: Mode the graph is running in (train|predict|eval).
 530 
 531     Returns:
 532       A dict of predictions tensors.
 533 
 534     Raises:
 535       ValueError: if features is not valid.
 536     """
 537 
 538     # Use the current ensemble to predict on the current batch of input.
 539     # For faster prediction we check if the inputs are on the same device
 540     # as the model. If not, we create a copy of the model on the worker.
 541     input_deps = (
 542         self._dense_floats + self._sparse_float_indices +
 543         self._sparse_int_indices)
 544     if not input_deps:
 545       raise ValueError("No input tensors for prediction.")
 546 
 547     # Get most current model stamp.
 548     ensemble_stamp = model_ops.tree_ensemble_stamp_token(self._ensemble_handle)
 549 
 550     # Determine if ensemble is colocated with the inputs.
 551     if self._ensemble_handle.device != input_deps[0].device:
 552       # Create a local ensemble and get its local stamp.
 553       with ops.name_scope("local_ensemble", "TreeEnsembleVariable") as name:
 554         local_ensemble_handle = (
 555             gen_model_ops.decision_tree_ensemble_resource_handle_op(name=name))
 556         create_op = gen_model_ops.create_tree_ensemble_variable(
 557             local_ensemble_handle, stamp_token=-1, tree_ensemble_config="")
 558         with ops.control_dependencies([create_op]):
 559           local_stamp = model_ops.tree_ensemble_stamp_token(
 560               local_ensemble_handle)
 561 
 562       # Determine whether the local ensemble is stale and update it if needed.
 563       def _refresh_local_ensemble_fn():
 564         # Serialize the model from parameter server after reading the inputs.
 565         with ops.control_dependencies([input_deps[0]]):
 566           (ensemble_stamp, serialized_model) = (
 567               model_ops.tree_ensemble_serialize(self._ensemble_handle))
 568 
 569         # Update local ensemble with the serialized model from parameter server.
 570         with ops.control_dependencies([create_op]):
 571           return model_ops.tree_ensemble_deserialize(
 572               local_ensemble_handle,
 573               stamp_token=ensemble_stamp,
 574               tree_ensemble_config=serialized_model), ensemble_stamp
 575 
 576       refresh_local_ensemble, ensemble_stamp = control_flow_ops.cond(
 577           math_ops.not_equal(ensemble_stamp,
 578                              local_stamp), _refresh_local_ensemble_fn,
 579           lambda: (control_flow_ops.no_op(), ensemble_stamp))
 580 
 581       # Once updated, use the local model for prediction.
 582       with ops.control_dependencies([refresh_local_ensemble]):
 583         return self._predict_and_return_dict(local_ensemble_handle,
 584                                              ensemble_stamp, mode)
 585     else:
 586       # Use ensemble_handle directly, if colocated.
 587       with ops.device(self._ensemble_handle.device):
 588         return self._predict_and_return_dict(self._ensemble_handle,
 589                                              ensemble_stamp, mode)
 590 
 591   def _get_class_id(self, predictions_dict):
 592     # Handle different multiclass strategies.
 593     if (self._learner_config.multi_class_strategy ==
 594         learner_pb2.LearnerConfig.TREE_PER_CLASS and
 595         self._logits_dimension != 1):
 596       # Choose the class for which the tree is built (one vs rest).
 597       return math_ops.to_int32(
 598           predictions_dict[NUM_TREES_ATTEMPTED] % self._logits_dimension)
 599     return constant_op.constant(-1, dtype=dtypes.int32)

 601   def update_stats(self, loss, predictions_dict):
 602     """Update the accumulators with stats from this batch.
 603 
 604     Args:
 605       loss: A scalar tensor representing average loss of examples.
 606       predictions_dict: Dictionary of Rank 2 `Tensor` representing information
 607           about predictions per example.
 608 
 609     Returns:
 610       Three values:
 611         - An op that adds a new tree to the ensemble, and
 612         - An op that increments the stamp but removes all the trees and resets
 613             the handlers. This can be used to reset the state of the ensemble.
 614         - A dict containing the training state.
 615 
 616     Raises:
 617       ValueError: if inputs are not valid.
 618     """
 619     # Get the worker device from input dependencies.
 620     input_deps = (
 621         self._dense_floats + self._sparse_float_indices +
 622         self._sparse_int_indices)
 623     worker_device = input_deps[0].device
 624 
 625     # Get tensors relevant for training and form the loss.
 626     predictions = predictions_dict[PREDICTIONS]
 627     partition_ids = predictions_dict[PARTITION_IDS]
 628     ensemble_stamp = predictions_dict[ENSEMBLE_STAMP]
 629     gradients = gradients_impl.gradients(
 630         loss,
 631         predictions,
 632         name="Gradients",
 633         colocate_gradients_with_ops=False,
 634         gate_gradients=0,
 635         aggregation_method=None)[0]
 636     strategy = self._learner_config.multi_class_strategy
 637 
 638     class_id = self._get_class_id(predictions_dict)
 639     # Handle different multiclass strategies.
 640     if strategy == learner_pb2.LearnerConfig.TREE_PER_CLASS:
 641       # We build one vs rest trees.
 642       if self._logits_dimension == 1:
 643         # We have only 1 score, gradients is of shape [batch, 1].
 644         hessians = gradients_impl.gradients(
 645             gradients,
 646             predictions,
 647             name="Hessian",
 648             colocate_gradients_with_ops=False,
 649             gate_gradients=0,
 650             aggregation_method=None)[0]
 651 
 652         squeezed_gradients = array_ops.squeeze(gradients, axis=[1])
 653         squeezed_hessians = array_ops.squeeze(hessians, axis=[1])
 654       else:
 655         hessian_list = self._diagonal_hessian(gradients, predictions)
 656         # Assemble hessian list into a tensor.
 657         hessians = array_ops.stack(hessian_list, axis=1)
 658         # Use class id tensor to get the column with that index from gradients
 659         # and hessians.
 660         squeezed_gradients = array_ops.squeeze(
 661             _get_column_by_index(gradients, class_id))
 662         squeezed_hessians = array_ops.squeeze(
 663             _get_column_by_index(hessians, class_id))
 664     else:
 665       # Other multiclass strategies.
 666       if strategy == learner_pb2.LearnerConfig.FULL_HESSIAN:
 667         hessian_list = self._full_hessian(gradients, predictions)
 668       else:
 669         # Diagonal hessian strategy.
 670         hessian_list = self._diagonal_hessian(gradients, predictions)
 671 
 672       squeezed_gradients = gradients
 673       hessians = array_ops.stack(hessian_list, axis=1)
 674       squeezed_hessians = hessians
 675 
 676     # Get the weights for each example for quantiles calculation,
 677     weights = self._get_weights(self._hessian_shape, squeezed_hessians)
 678 
 679     # Create all handlers ensuring resources are evenly allocated across PS.
 680     fc_name_idx = 0
 681     handlers = []
 682     init_stamp_token = constant_op.constant(0, dtype=dtypes.int64)
 683     l1_regularization = constant_op.constant(
 684         self._learner_config.regularization.l1, dtypes.float32)
 685     l2_regularization = constant_op.constant(
 686         self._learner_config.regularization.l2, dtypes.float32)
 687     tree_complexity_regularization = constant_op.constant(
 688         self._learner_config.regularization.tree_complexity, dtypes.float32)
 689     min_node_weight = constant_op.constant(
 690         self._learner_config.constraints.min_node_weight, dtypes.float32)
 691     loss_uses_sum_reduction = self._loss_reduction == losses.Reduction.SUM
 692     loss_uses_sum_reduction = constant_op.constant(loss_uses_sum_reduction)
 693     weak_learner_type = constant_op.constant(
 694         self._learner_config.weak_learner_type)
 695     num_quantiles = self._num_quantiles
 696     epsilon = 1.0 / num_quantiles
 697     strategy_tensor = constant_op.constant(strategy)
 698     with ops.device(self._get_replica_device_setter(worker_device)):
 699       # Create handlers for dense float columns
 700       for dense_float_column_idx in range(len(self._dense_floats)):
 701         fc_name = self._fc_names[fc_name_idx]
 702         handlers.append(
 703             ordinal_split_handler.DenseSplitHandler(
 704                 l1_regularization=l1_regularization,
 705                 l2_regularization=l2_regularization,
 706                 tree_complexity_regularization=tree_complexity_regularization,
 707                 min_node_weight=min_node_weight,
 708                 feature_column_group_id=constant_op.constant(
 709                     dense_float_column_idx),
 710                 epsilon=epsilon,
 711                 num_quantiles=num_quantiles,
 712                 dense_float_column=self._dense_floats[dense_float_column_idx],
 713                 name=fc_name,
 714                 gradient_shape=self._gradient_shape,
 715                 hessian_shape=self._hessian_shape,
 716                 multiclass_strategy=strategy_tensor,
 717                 init_stamp_token=init_stamp_token,
 718                 loss_uses_sum_reduction=loss_uses_sum_reduction,
 719                 weak_learner_type=weak_learner_type,
 720             ))
 721         fc_name_idx += 1
 722 
 723       # Create handlers for sparse float columns.
 724       for sparse_float_column_idx in range(len(self._sparse_float_indices)):
 725         fc_name = self._fc_names[fc_name_idx]
 726         handlers.append(
 727             ordinal_split_handler.SparseSplitHandler(
 728                 l1_regularization=l1_regularization,
 729                 l2_regularization=l2_regularization,
 730                 tree_complexity_regularization=tree_complexity_regularization,
 731                 min_node_weight=min_node_weight,
 732                 feature_column_group_id=constant_op.constant(
 733                     sparse_float_column_idx),
 734                 epsilon=epsilon,
 735                 num_quantiles=num_quantiles,
 736                 sparse_float_column=sparse_tensor.SparseTensor(
 737                     self._sparse_float_indices[sparse_float_column_idx],
 738                     self._sparse_float_values[sparse_float_column_idx],
 739                     self._sparse_float_shapes[sparse_float_column_idx]),
 740                 name=fc_name,
 741                 gradient_shape=self._gradient_shape,
 742                 hessian_shape=self._hessian_shape,
 743                 multiclass_strategy=strategy_tensor,
 744                 init_stamp_token=init_stamp_token,
 745                 loss_uses_sum_reduction=loss_uses_sum_reduction))
 746         fc_name_idx += 1
 747 
 748       # Create handlers for sparse int columns.
 749       for sparse_int_column_idx in range(len(self._sparse_int_indices)):
 750         fc_name = self._fc_names[fc_name_idx]
 751         handlers.append(
 752             categorical_split_handler.EqualitySplitHandler(
 753                 l1_regularization=l1_regularization,
 754                 l2_regularization=l2_regularization,
 755                 tree_complexity_regularization=tree_complexity_regularization,
 756                 min_node_weight=min_node_weight,
 757                 feature_column_group_id=constant_op.constant(
 758                     sparse_int_column_idx),
 759                 sparse_int_column=sparse_tensor.SparseTensor(
 760                     self._sparse_int_indices[sparse_int_column_idx],
 761                     self._sparse_int_values[sparse_int_column_idx],
 762                     self._sparse_int_shapes[sparse_int_column_idx]),
 763                 name=fc_name,
 764                 gradient_shape=self._gradient_shape,
 765                 hessian_shape=self._hessian_shape,
 766                 multiclass_strategy=strategy_tensor,
 767                 init_stamp_token=init_stamp_token,
 768                 loss_uses_sum_reduction=loss_uses_sum_reduction,
 769                 weak_learner_type=weak_learner_type))
 770         fc_name_idx += 1
 771 
 772       # Create ensemble stats variables.
 773       num_layer_examples = variables.VariableV1(
 774           initial_value=array_ops.zeros([], dtypes.int64),
 775           name="num_layer_examples",
 776           trainable=False)
 777       num_layer_steps = variables.VariableV1(
 778           initial_value=array_ops.zeros([], dtypes.int64),
 779           name="num_layer_steps",
 780           trainable=False)
 781       num_layers = variables.VariableV1(
 782           initial_value=array_ops.zeros([], dtypes.int64),
 783           name="num_layers",
 784           trainable=False)
 785       active_tree = variables.VariableV1(
 786           initial_value=array_ops.zeros([], dtypes.int64),
 787           name="active_tree",
 788           trainable=False)
 789       active_layer = variables.VariableV1(
 790           initial_value=array_ops.zeros([], dtypes.int64),
 791           name="active_layer",
 792           trainable=False)
 793       # Variable that becomes false once bias centering is done.
 794       continue_centering = variables.VariableV1(
 795           initial_value=self._center_bias,
 796           name="continue_centering",
 797           trainable=False)
 798       # Create bias stats accumulator.
 799       bias_stats_accumulator = stats_accumulator_ops.StatsAccumulator(
 800           stamp_token=0,
 801           gradient_shape=self._gradient_shape,
 802           hessian_shape=self._hessian_shape,
 803           name="BiasAccumulator")
 804       # Create steps accumulator.
 805       steps_accumulator = stats_accumulator_ops.StatsAccumulator(
 806           stamp_token=0,
 807           gradient_shape=tensor_shape.scalar(),
 808           hessian_shape=tensor_shape.scalar(),
 809           name="StepsAccumulator")
 810     # Create ensemble stats summaries.
 811     summary.scalar("layer_stats/num_examples", num_layer_examples)
 812     summary.scalar("layer_stats/num_steps", num_layer_steps)
 813     summary.scalar("ensemble_stats/active_tree", active_tree)
 814     summary.scalar("ensemble_stats/active_layer", active_layer)
 815 
 816     # Update bias stats.
 817     stats_update_ops = []
 818 
 819     stats_update_ops.append(
 820         control_flow_ops.cond(
 821             continue_centering,
 822             self._make_update_bias_stats_fn(
 823                 ensemble_stamp, predictions, gradients,
 824                 bias_stats_accumulator), control_flow_ops.no_op))
 825 
 826     # Update handler stats.
 827     handler_reads = collections.OrderedDict()
 828     for handler in handlers:
 829       handler_reads[handler] = handler.scheduled_reads()
 830 
 831     handler_results = batch_ops_utils.run_handler_scheduled_ops(
 832         handler_reads, ensemble_stamp, worker_device)
 833     per_handler_updates = collections.OrderedDict()
 834     # Two values per handler. First one is if the handler is active for the
 835     # current layer. The second one is if the handler is going to be active
 836     # for the next layer.
 837     subsampling_type = self._learner_config.WhichOneof("feature_fraction")
 838     if subsampling_type == "feature_fraction_per_level":
 839       seed = predictions_dict[NUM_LAYERS_ATTEMPTED]
 840       active_handlers_current_layer = stateless.stateless_random_uniform(
 841           shape=[len(handlers)], seed=[seed, 1])
 842       active_handlers_next_layer = stateless.stateless_random_uniform(
 843           shape=[len(handlers)], seed=[seed + 1, 1])
 844       active_handlers = array_ops.stack(
 845           [active_handlers_current_layer, active_handlers_next_layer], axis=1)
 846       active_handlers = (
 847           active_handlers < self._learner_config.feature_fraction_per_level)
 848     elif subsampling_type == "feature_fraction_per_tree":
 849       seed = predictions_dict[NUM_TREES_ATTEMPTED]
 850       active_handlers_current_layer = stateless.stateless_random_uniform(
 851           shape=[len(handlers)], seed=[seed, 2])
 852       active_handlers_current_layer = (
 853           active_handlers_current_layer <
 854           self._learner_config.feature_fraction_per_tree)
 855       active_handlers = array_ops.stack(
 856           [
 857               active_handlers_current_layer,
 858               array_ops.ones([len(handlers)], dtype=dtypes.bool)
 859           ],
 860           axis=1)
 861     else:
 862       active_handlers = array_ops.ones([len(handlers), 2], dtype=dtypes.bool)
 863 
 864     if self._learner_config.constraints.max_number_of_unique_feature_columns:
 865       target = (
 866           self._learner_config.constraints.max_number_of_unique_feature_columns)
 867 
 868       def _feature_selection_active_handlers():
 869         # The active list for current and the next iteration.
 870         used_handlers = array_ops.reshape(predictions_dict[USED_HANDLERS_MASK],
 871                                           [-1, 1])
 872         used_handlers = array_ops.concat([used_handlers, used_handlers], axis=1)
 873         return math_ops.logical_and(used_handlers, active_handlers)
 874 
 875       active_handlers = (
 876           control_flow_ops.cond(predictions_dict[NUM_USED_HANDLERS] >= target,
 877                                 _feature_selection_active_handlers,
 878                                 lambda: active_handlers))
 879 
 880     # Prepare empty gradients and hessians when handlers are not ready.
 881     empty_hess_shape = [1] + self._hessian_shape.as_list()
 882     empty_grad_shape = [1] + self._gradient_shape.as_list()
 883 
 884     empty_gradients = constant_op.constant(
 885         [], dtype=dtypes.float32, shape=empty_grad_shape)
 886     empty_hessians = constant_op.constant(
 887         [], dtype=dtypes.float32, shape=empty_hess_shape)
 888 
 889     active_handlers = array_ops.unstack(active_handlers, axis=0)
 890     for handler_idx in range(len(handlers)):
 891       handler = handlers[handler_idx]
 892       is_active = active_handlers[handler_idx]
 893       updates, scheduled_updates = handler.update_stats(
 894           ensemble_stamp, partition_ids, squeezed_gradients, squeezed_hessians,
 895           empty_gradients, empty_hessians, weights, is_active,
 896           handler_results[handler])
 897       stats_update_ops.append(updates)
 898       per_handler_updates[handler] = scheduled_updates
 899 
 900     update_results = batch_ops_utils.run_handler_scheduled_ops(
 901         per_handler_updates, ensemble_stamp, worker_device)
 902     for update in update_results.values():
 903       stats_update_ops += update
 904 
 905     training_state = GBDTTrainingState(
 906         num_layer_examples=num_layer_examples,
 907         num_layer_steps=num_layer_steps,
 908         num_layers=num_layers,
 909         active_tree=active_tree,
 910         active_layer=active_layer,
 911         continue_centering=continue_centering,
 912         bias_stats_accumulator=bias_stats_accumulator,
 913         steps_accumulator=steps_accumulator,
 914         handlers=handlers)
 915 
 916     reset_op = control_flow_ops.no_op()
 917     if self._is_chief:
 918       # Advance the ensemble stamp to throw away staggered workers.
 919       stamp_token, _ = model_ops.tree_ensemble_serialize(self._ensemble_handle)
 920       next_stamp_token = stamp_token + 1
 921 
 922       reset_ops = []
 923       for handler in handlers:
 924         reset_ops.append(handler.reset(stamp_token, next_stamp_token))
 925       if self._center_bias:
 926         reset_ops.append(
 927             bias_stats_accumulator.flush(stamp_token, next_stamp_token))
 928       reset_ops.append(steps_accumulator.flush(stamp_token, next_stamp_token))
 929       reset_ops.append(self._finalized_trees.assign(0).op)
 930       reset_ops.append(self._attempted_trees.assign(0).op)
 931       reset_ops.append(
 932           model_ops.tree_ensemble_deserialize(
 933               self._ensemble_handle,
 934               stamp_token=next_stamp_token,
 935               tree_ensemble_config="",
 936               name="reset_gbdt"))
 937 
 938       reset_op = control_flow_ops.group([reset_ops])
 939 
 940     return stats_update_ops, reset_op, training_state

 942   def increment_step_counter_and_maybe_update_ensemble(self, predictions_dict,
 943                                                        training_state):
 944     """Increments number of visited examples and grows the ensemble.
 945 
 946     If the number of visited examples reaches the target examples_per_layer,
 947     ensemble is updated.
 948 
 949     Args:
 950       predictions_dict: Dictionary of Rank 2 `Tensor` representing information
 951           about predictions per example.
 952       training_state: `dict` returned by update_stats.
 953 
 954     Returns:
 955       An op that updates the counters and potientially grows the ensemble.
 956     """
 957     batch_size = math_ops.cast(
 958         array_ops.shape(predictions_dict[PREDICTIONS])[0], dtypes.float32)
 959     ensemble_stamp = predictions_dict[ENSEMBLE_STAMP]
 960     # Accumulate a step after updating stats.
 961 
 962     steps_accumulator = training_state.steps_accumulator
 963     num_layer_examples = training_state.num_layer_examples
 964     num_layer_steps = training_state.num_layer_steps
 965     active_layer = training_state.active_layer
 966     add_step_op = steps_accumulator.add(
 967         ensemble_stamp, [0], [[0, 0]], [batch_size], [1.0])
 968 
 969     # After adding the step, decide if further processing is needed.
 970     ensemble_update_ops = [add_step_op]
 971     class_id = self._get_class_id(predictions_dict)
 972 
 973     with ops.control_dependencies([add_step_op]):
 974       if self._is_chief:
 975         dropout_seed = predictions_dict[NUM_TREES_ATTEMPTED]
 976 
 977         # Get accumulated steps and examples for the current layer.
 978         _, _, _, _, acc_examples, acc_steps = (
 979             steps_accumulator.serialize())
 980         acc_examples = math_ops.cast(acc_examples[0], dtypes.int64)
 981         acc_steps = math_ops.cast(acc_steps[0], dtypes.int64)
 982         ensemble_update_ops.append(
 983             num_layer_examples.assign(acc_examples))
 984         ensemble_update_ops.append(num_layer_steps.assign(acc_steps))
 985         # Determine whether we need to update tree ensemble.
 986         examples_per_layer = self._examples_per_layer
 987         if callable(examples_per_layer):
 988           examples_per_layer = examples_per_layer(active_layer)
 989         ensemble_update_ops.append(
 990             control_flow_ops.cond(
 991                 acc_examples >= examples_per_layer,
 992                 self.make_update_ensemble_fn(ensemble_stamp, training_state,
 993                                              dropout_seed, class_id),
 994                 control_flow_ops.no_op))
 995 
 996     # Note, the loss is calculated from the prediction considering dropouts, so
 997     # that the value might look staggering over steps when the dropout ratio is
 998     # high. eval_loss might be referred instead in the aspect of convergence.
 999     return control_flow_ops.group(*ensemble_update_ops)

1001   def make_update_ensemble_fn(self, ensemble_stamp, training_state,
1002                               dropout_seed, class_id):
1003     """A method to create the function which updates the tree ensemble."""
1004     # Determine learning rate.
1005     learning_rate_tuner = self._learner_config.learning_rate_tuner.WhichOneof(
1006         "tuner")
1007     if learning_rate_tuner == "fixed" or learning_rate_tuner == "dropout":
1008       tuner = getattr(self._learner_config.learning_rate_tuner,
1009                       learning_rate_tuner)
1010       learning_rate = tuner.learning_rate
1011     else:
1012       # TODO(nponomareva, soroush) do the line search.
1013       raise ValueError("Line search learning rate is not yet supported.")
1014 
1015     def _update_ensemble():
1016       """A method to update the tree ensemble."""
1017       # Get next stamp token.
1018       next_ensemble_stamp = ensemble_stamp + 1
1019       # Finalize bias stats.
1020       _, _, _, bias_grads, bias_hess = (
1021           training_state.bias_stats_accumulator.flush(ensemble_stamp,
1022                                                       next_ensemble_stamp))
1023 
1024       # Finalize handler splits.
1025       are_splits_ready_list = []
1026       partition_ids_list = []
1027       gains_list = []
1028       split_info_list = []
1029 
1030       for handler in training_state.handlers:
1031         (are_splits_ready,
1032          partition_ids, gains, split_info) = handler.make_splits(
1033              ensemble_stamp, next_ensemble_stamp, class_id)
1034         are_splits_ready_list.append(are_splits_ready)
1035         partition_ids_list.append(partition_ids)
1036         gains_list.append(gains)
1037         split_info_list.append(split_info)
1038       # Stack all the inputs to one tensor per type.
1039       # This is a workaround for the slowness of graph building in tf.cond.
1040       # See (b/36554864).
1041       split_sizes = array_ops.reshape(
1042           array_ops.shape_n(partition_ids_list), [len(partition_ids_list)])
1043       partition_ids = array_ops.concat(partition_ids_list, axis=0)
1044       gains = array_ops.concat(gains_list, axis=0)
1045       split_infos = array_ops.concat(split_info_list, axis=0)
1046 
1047       # Determine if all splits are ready.
1048       are_all_splits_ready = math_ops.reduce_all(
1049           array_ops.stack(
1050               are_splits_ready_list, axis=0, name="stack_handler_readiness"))
1051 
1052       # Define bias centering update operation.
1053       def _center_bias_fn():
1054         # Center tree ensemble bias.
1055         delta_updates = array_ops.where(bias_hess > 0, -bias_grads / bias_hess,
1056                                         array_ops.zeros_like(bias_grads))
1057         center_bias = training_ops.center_tree_ensemble_bias(
1058             tree_ensemble_handle=self._ensemble_handle,
1059             stamp_token=ensemble_stamp,
1060             next_stamp_token=next_ensemble_stamp,
1061             delta_updates=delta_updates,
1062             learner_config=self._learner_config_serialized)
1063         return training_state.continue_centering.assign(center_bias)
1064 
1065       # Define ensemble growing operations.
1066       def _grow_ensemble_ready_fn():
1067         # Grow the ensemble given the current candidates.
1068         sizes = array_ops.unstack(split_sizes)
1069         partition_ids_list = list(array_ops.split(partition_ids, sizes, axis=0))
1070         # When using the oblivious decision tree as weak learner, it produces
1071         # one gain and one split per handler and not number of partitions.
1072         if self._learner_config.weak_learner_type == (
1073             learner_pb2.LearnerConfig.OBLIVIOUS_DECISION_TREE):
1074           sizes = len(training_state.handlers)
1075 
1076         gains_list = list(array_ops.split(gains, sizes, axis=0))
1077         split_info_list = list(array_ops.split(split_infos, sizes, axis=0))
1078         return training_ops.grow_tree_ensemble(
1079             tree_ensemble_handle=self._ensemble_handle,
1080             stamp_token=ensemble_stamp,
1081             next_stamp_token=next_ensemble_stamp,
1082             learning_rate=learning_rate,
1083             partition_ids=partition_ids_list,
1084             gains=gains_list,
1085             splits=split_info_list,
1086             learner_config=self._learner_config_serialized,
1087             dropout_seed=dropout_seed,
1088             center_bias=self._center_bias,
1089             max_tree_depth=self._max_tree_depth,
1090             weak_learner_type=self._learner_config.weak_learner_type)
1091 
1092       def _grow_ensemble_not_ready_fn():
1093         # Don't grow the ensemble, just update the stamp.
1094         return training_ops.grow_tree_ensemble(
1095             tree_ensemble_handle=self._ensemble_handle,
1096             stamp_token=ensemble_stamp,
1097             next_stamp_token=next_ensemble_stamp,
1098             learning_rate=0,
1099             partition_ids=[],
1100             gains=[],
1101             splits=[],
1102             learner_config=self._learner_config_serialized,
1103             dropout_seed=dropout_seed,
1104             center_bias=self._center_bias,
1105             max_tree_depth=self._max_tree_depth,
1106             weak_learner_type=self._learner_config.weak_learner_type)
1107 
1108       def _grow_ensemble_fn():
1109         # Conditionally grow an ensemble depending on whether the splits
1110         # from all the handlers are ready.
1111         return control_flow_ops.cond(are_all_splits_ready,
1112                                      _grow_ensemble_ready_fn,
1113                                      _grow_ensemble_not_ready_fn)
1114 
1115       # Update ensemble.
1116       update_ops = [are_all_splits_ready]
1117       if self._center_bias:
1118         update_model = control_flow_ops.cond(training_state.continue_centering,
1119                                              _center_bias_fn, _grow_ensemble_fn)
1120       else:
1121         update_model = _grow_ensemble_fn()
1122       update_ops.append(update_model)
1123 
1124       # Update ensemble stats.
1125       with ops.control_dependencies([update_model]):
1126         stats = training_ops.tree_ensemble_stats(
1127             self._ensemble_handle, stamp_token=next_ensemble_stamp)
1128         update_ops.append(self._finalized_trees.assign(stats.num_trees))
1129         update_ops.append(self._attempted_trees.assign(stats.attempted_trees))
1130         update_ops.append(training_state.num_layers.assign(stats.num_layers))
1131         update_ops.append(training_state.active_tree.assign(stats.active_tree))
1132         update_ops.append(
1133             training_state.active_layer.assign(stats.active_layer))
1134 
1135       # Flush step stats.
1136       update_ops.extend(
1137           training_state.steps_accumulator.flush(ensemble_stamp,
1138                                                  next_ensemble_stamp))
1139       return control_flow_ops.group(*update_ops, name="update_ensemble")
1140 
1141     return _update_ensemble

1171   def _get_weights(self, hessian_shape, hessians):
1172     """Derives weights to be used based on hessians and multiclass strategy."""
1173     if hessian_shape == tensor_shape.scalar():
1174       # This is tree per class.
1175       weights = hessians
1176     elif len(hessian_shape.dims) == 1:
1177       # This is diagonal hessian.
1178       weights = math_ops.reduce_sum(hessians, axis=1)
1179     else:
1180       # This is full hessian.
1181       weights = math_ops.trace(hessians)
1182     return weights

1184   def _full_hessian(self, grads, predictions):
1185     """Prepares hessians for full-hessian multiclass strategy."""
1186     # Because of
1187     # https://github.com/tensorflow/tensorflow/issues/675, we can't just
1188     # compute the full hessian with a single call to gradients, but instead
1189     # must compute it row-by-row.
1190     gradients_list = array_ops.unstack(
1191         grads, num=self._logits_dimension, axis=1)
1192     hessian_rows = []
1193 
1194     for row in range(self._logits_dimension):
1195       # If current row is i, K is number of classes,each row returns a tensor of
1196       # size batch_size x K representing for each example dx_i dx_1, dx_i dx_2
1197       # etc dx_i dx_K
1198       hessian_row = gradients_impl.gradients(
1199           gradients_list[row],
1200           predictions,
1201           name="Hessian_%d" % row,
1202           colocate_gradients_with_ops=False,
1203           gate_gradients=0,
1204           aggregation_method=None)
1205 
1206       # hessian_row is of dimension 1, batch_size, K, => trim first dimension
1207       # to get batch_size x K
1208       hessian_row = array_ops.squeeze(array_ops.unstack(hessian_row), [0])
1209       hessian_rows.append(hessian_row)
1210     return hessian_rows

1212   def _diagonal_hessian(self, grads, predictions):
1213     """Prepares hessians for diagonal-hessian multiclass mode."""
1214     diag_hessian_list = []
1215 
1216     gradients_list = array_ops.unstack(
1217         grads, num=self._logits_dimension, axis=1)
1218 
1219     for row, row_grads in enumerate(gradients_list):
1220       # If current row is i, K is number of classes,each row returns a tensor of
1221       # size batch_size x K representing for each example dx_i dx_1, dx_1 dx_2
1222       # etc dx_i dx_K
1223       hessian_row = gradients_impl.gradients(
1224           row_grads,
1225           predictions,
1226           name="Hessian_%d" % row,
1227           colocate_gradients_with_ops=False,
1228           gate_gradients=0,
1229           aggregation_method=None)
1230 
1231       # hessian_row is of dimension 1, batch_size, K, => trim first dimension
1232       # to get batch_size x K
1233       hessian_row = array_ops.squeeze(array_ops.unstack(hessian_row), [0])
1234 
1235       # Get dx_i^2 for the whole batch.
1236       elem = array_ops.transpose(hessian_row)[row]
1237       diag_hessian_list.append(elem)
1238 
1239     return diag_hessian_list

1259   def _make_update_bias_stats_fn(self, ensemble_stamp, predictions, gradients,
1260                                  bias_stats_accumulator):
1261     """A method to create the function which updates the bias stats."""
1262 
1263     def _update_bias_stats():
1264       """A method to update the bias stats."""
1265       # Get reduced gradients and hessians.
1266       grads_sum = math_ops.reduce_sum(gradients, 0)
1267       hess = gradients_impl.gradients(
1268           grads_sum,
1269           predictions,
1270           name="Hessians",
1271           colocate_gradients_with_ops=False,
1272           gate_gradients=0,
1273           aggregation_method=None)[0]
1274       hess_sum = math_ops.reduce_sum(hess, 0)
1275 
1276       # Accumulate gradients and hessians.
1277       partition_ids = math_ops.range(self._logits_dimension)
1278       feature_ids = array_ops.zeros(
1279           [self._logits_dimension, 2], dtype=dtypes.int64)
1280 
1281       add_stats_op = bias_stats_accumulator.add(
1282           ensemble_stamp, partition_ids, feature_ids, grads_sum, hess_sum)
1283       return control_flow_ops.group(*[add_stats_op], name="update_bias_stats")
1284 
1285     return _update_bias_stats
