  40 class TreeEnsembleVariableSavable(saver.BaseSaverBuilder.SaveableObject):
  41   """SaveableObject implementation for TreeEnsembleVariable."""
  42 
  43   def __init__(self, tree_ensemble_handle, create_op, name):
  44     """Creates a TreeEnsembleVariableSavable object.
  45 
  46     Args:
  47       tree_ensemble_handle: handle to the tree ensemble variable.
  48       create_op: the op to initialize the variable.
  49       name: the name to save the tree ensemble variable under.
  50     """
  51     stamp_token, ensemble_config = tree_ensemble_serialize(tree_ensemble_handle)
  52     # slice_spec is useful for saving a slice from a variable.
  53     # It's not meaningful the tree ensemble variable. So we just pass an empty
  54     # value.
  55     slice_spec = ""
  56     specs = [
  57         saver.BaseSaverBuilder.SaveSpec(stamp_token, slice_spec,
  58                                         name + "_stamp"),
  59         saver.BaseSaverBuilder.SaveSpec(ensemble_config, slice_spec,
  60                                         name + "_config"),
  61     ]
  62     super(TreeEnsembleVariableSavable,
  63           self).__init__(tree_ensemble_handle, specs, name)
  64     self._tree_ensemble_handle = tree_ensemble_handle
  65     self._create_op = create_op
  66 
  67   def restore(self, restored_tensors, unused_restored_shapes):
  68     """Restores the associated tree ensemble from 'restored_tensors'.
  69 
  70     Args:
  71       restored_tensors: the tensors that were loaded from a checkpoint.
  72       unused_restored_shapes: the shapes this object should conform to after
  73         restore. Not meaningful for trees.
  74 
  75     Returns:
  76       The operation that restores the state of the tree ensemble variable.
  77     """
  78     with ops.control_dependencies([self._create_op]):
  79       return tree_ensemble_deserialize(
  80           self._tree_ensemble_handle,
  81           stamp_token=restored_tensors[0],
  82           tree_ensemble_config=restored_tensors[1])

  85 def tree_ensemble_variable(stamp_token,
  86                            tree_ensemble_config,
  87                            name,
  88                            container=None):
  89   r"""Creates a tree ensemble model and returns a handle to it.
  90 
  91   Args:
  92     stamp_token: The initial stamp token value for the ensemble resource.
  93     tree_ensemble_config: A `Tensor` of type `string`.
  94       Serialized proto of the tree ensemble.
  95     name: A name for the ensemble variable.
  96     container: An optional `string`. Defaults to `""`.
  97 
  98   Returns:
  99     A `Tensor` of type mutable `string`. The handle to the tree ensemble.
 100   """
 101   with ops.name_scope(name, "TreeEnsembleVariable") as name:
 102     resource_handle = gen_model_ops.decision_tree_ensemble_resource_handle_op(
 103         container, shared_name=name, name=name)
 104     create_op = gen_model_ops.create_tree_ensemble_variable(
 105         resource_handle, stamp_token, tree_ensemble_config)
 106     is_initialized_op = gen_model_ops.tree_ensemble_is_initialized_op(
 107         resource_handle)
 108     # Adds the variable to the savable list.
 109     saveable = TreeEnsembleVariableSavable(resource_handle, create_op,
 110                                            resource_handle.name)
 111     ops.add_to_collection(ops.GraphKeys.SAVEABLE_OBJECTS, saveable)
 112     resources.register_resource(resource_handle, create_op, is_initialized_op)
 113     return resource_handle
