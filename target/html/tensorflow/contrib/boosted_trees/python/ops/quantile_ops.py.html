  37 # Pattern to remove all non alpha numeric from a string.
  38 _PATTERN = re.compile(r"[\W_]+")

  41 class QuantileAccumulator(saver.BaseSaverBuilder.SaveableObject):
  42   """A resource that allows distributed quantile computation."""
  43 
  44   def __init__(self,
  45                init_stamp_token,
  46                epsilon,
  47                num_quantiles,
  48                max_elements=None,
  49                name=None,
  50                container=None,
  51                generate_quantiles=False):
  52     """Creates a QuantileAccumulator object.
  53 
  54     Args:
  55       init_stamp_token: The initial value for the stamp token.
  56       epsilon: Error bound on the quantile computation.
  57       num_quantiles: Number of quantiles to produce from the final summary.
  58       max_elements: Maximum number of elements added to the accumulator.
  59       name: the name to save the accumulator under.
  60       container: An optional `string`. Defaults to `""`
  61       generate_quantiles: Generate quantiles instead of approximate boundaries.
  62         If true, exactly `num_quantiles` will be produced in the final summary.
  63     """
  64     self._epsilon = epsilon
  65     self._generate_quantiles = generate_quantiles
  66 
  67     name = _PATTERN.sub("", name)
  68     with ops.name_scope(name, "QuantileAccumulator") as name:
  69       self._quantile_accumulator_handle = (
  70           gen_quantile_ops.quantile_stream_resource_handle_op(
  71               container=container, shared_name=name, name=name))
  72       self._create_op = gen_quantile_ops.create_quantile_accumulator(
  73           self._quantile_accumulator_handle,
  74           init_stamp_token,
  75           epsilon=epsilon,
  76           max_elements=max_elements,
  77           num_quantiles=num_quantiles,
  78           generate_quantiles=generate_quantiles)
  79       is_initialized_op = gen_quantile_ops.quantile_accumulator_is_initialized(
  80           self._quantile_accumulator_handle)
  81     resources.register_resource(self._quantile_accumulator_handle,
  82                                 self._create_op, is_initialized_op)
  83     self._make_savable(name)
  84 
  85   def _make_savable(self, name):
  86     stamp_token, state, are_buckets_ready, buckets = (
  87         gen_quantile_ops.quantile_accumulator_serialize(
  88             self._quantile_accumulator_handle))
  89     # slice_spec is useful for saving a slice from a variable.
  90     # It's not meaningful in quantile accumulator.
  91     slice_spec = ""
  92     def make_save_spec(tensor, suffix):
  93       return saver.BaseSaverBuilder.SaveSpec(tensor, slice_spec, name + suffix)
  94 
  95     specs = [make_save_spec(stamp_token, "_stamp")]
  96     specs += [make_save_spec(state, "_state")]
  97     specs += [make_save_spec(are_buckets_ready, "_are_buckets_ready")]
  98     specs += [make_save_spec(buckets, "buckets")]
  99     super(QuantileAccumulator,
 100           self).__init__(self._quantile_accumulator_handle, specs, name)
 101     ops.add_to_collection(ops.GraphKeys.SAVEABLE_OBJECTS, self)
 102 
 103   def restore(self, restored_tensors, unused_restored_shapes):
 104     """Restores the associated quantile accumulator from 'restored_tensors'.
 105 
 106     Args:
 107       restored_tensors: the tensors that were loaded from a checkpoint.
 108       unused_restored_shapes: the shapes this object should conform to after
 109         restore.
 110 
 111     Returns:
 112       The operation that restores the state of the quantile accumulator.
 113     """
 114     # Read the restored tensors with the same order that were added to saving
 115     # spec.
 116     stamp_token = restored_tensors[:1]
 117     state = restored_tensors[1:2]
 118     are_buckets_ready = restored_tensors[2:3]
 119     buckets = restored_tensors[3]
 120     with ops.control_dependencies([self._create_op]):
 121       return gen_quantile_ops.quantile_accumulator_deserialize(
 122           self._quantile_accumulator_handle,
 123           stamp_token=stamp_token,
 124           stream_state=state,
 125           are_buckets_ready=are_buckets_ready,
 126           buckets=buckets)
 127 
 128   def get_buckets(self, stamp_token):
 129     """Returns quantile buckets created during previous flush."""
 130     are_buckets_ready, buckets = (
 131         gen_quantile_ops.quantile_accumulator_get_buckets(
 132             quantile_accumulator_handles=[self._quantile_accumulator_handle],
 133             stamp_token=stamp_token))
 134     return are_buckets_ready[0], buckets[0]
 135 
 136   def schedule_get_buckets(self):
 137     """Returns a scheduled read of buckets created during previous flush."""
 138     return batch_ops_utils.ScheduledStampedResourceOp(
 139         resource_handle=self._quantile_accumulator_handle,
 140         op=gen_quantile_ops.quantile_accumulator_get_buckets)
 141 
 142   def _make_summary(self, column, example_weights):
 143     if isinstance(column, sparse_tensor.SparseTensor):
 144       return gen_quantile_ops.make_quantile_summaries(
 145           dense_float_features=[],
 146           sparse_float_feature_indices=[column.indices],
 147           sparse_float_feature_values=[column.values],
 148           sparse_float_feature_shapes=[column.dense_shape],
 149           example_weights=example_weights,
 150           epsilon=self._epsilon / 2).sparse_summaries[0]
 151     else:
 152       return gen_quantile_ops.make_quantile_summaries(
 153           dense_float_features=[column],
 154           sparse_float_feature_indices=[],
 155           sparse_float_feature_values=[],
 156           sparse_float_feature_shapes=[],
 157           example_weights=example_weights,
 158           epsilon=self._epsilon / 2).dense_summaries[0]
 159 
 160   def add_summary(self, stamp_token, column, example_weights):
 161     """Adds quantile summary to its stream in resource."""
 162     summary = self._make_summary(column, example_weights)
 163     return gen_quantile_ops.quantile_accumulator_add_summaries(
 164         quantile_accumulator_handles=[self._quantile_accumulator_handle],
 165         stamp_token=stamp_token,
 166         summaries=[summary])
 167 
 168   def add_prebuilt_summary(self, stamp_token, summary):
 169     """Adds quantile summary to its stream in resource."""
 170     return gen_quantile_ops.quantile_accumulator_add_summaries(
 171         quantile_accumulator_handles=[self._quantile_accumulator_handle],
 172         stamp_token=stamp_token,
 173         summaries=[summary])
 174 
 175   def schedule_add_summary(self, stamp_token, column, example_weights):
 176     """Schedules to add a quantile summary to its stream in resource."""
 177     summary = self._make_summary(column, example_weights)
 178     return batch_ops_utils.ScheduledStampedResourceOp(
 179         op=gen_quantile_ops.quantile_accumulator_add_summaries,
 180         resource_handle=self._quantile_accumulator_handle,
 181         summaries=summary)
 182 
 183   def flush(self, stamp_token, next_stamp_token):
 184     """Finalizes quantile summary stream and resets it for next iteration.
 185 
 186     Args:
 187       stamp_token: Expected current token.
 188       next_stamp_token: Next value for the token.
 189     Returns:
 190       The flush operation.
 191     """
 192     return gen_quantile_ops.quantile_accumulator_flush(
 193         quantile_accumulator_handle=self._quantile_accumulator_handle,
 194         stamp_token=stamp_token,
 195         next_stamp_token=next_stamp_token)
 196 
 197   def flush_summary(self, stamp_token, next_stamp_token):
 198     """Finalizes quantile summary stream and resets it for next iteration."""
 199     result = gen_quantile_ops.quantile_accumulator_flush_summary(
 200         quantile_accumulator_handle=self._quantile_accumulator_handle,
 201         stamp_token=stamp_token,
 202         next_stamp_token=next_stamp_token)
 203     return result
 204 
 205   def resource(self):
 206     return self._quantile_accumulator_handle
