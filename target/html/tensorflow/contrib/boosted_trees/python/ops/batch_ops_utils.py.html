  30 class ScheduledOp(object):
  31   """Represents a scheduled remote operation."""
  32 
  33   __metaclass__ = abc.ABCMeta
  34 
  35   @abc.abstractmethod
  36   def batching_key(self):
  37     """Returns the key for batching operations."""
  38 
  39   @abc.abstractmethod
  40   def batch_runner_fn(self):
  41     """Returns the function that executes the operation on the batch."""
  42 
  43 
  44 class ScheduledStampedResourceOp(ScheduledOp):
  45   """Wrapper class for batched operations on stamped resources."""
  46 
  47   def __init__(self, resource_handle, op, **kwargs):
  48     self.resource_handle = resource_handle
  49     self.op = op
  50     self.args = kwargs
  51 
  52   def batching_key(self):
  53     # We want to group the same operations on the same device and run them in
  54     # one batch. So we use (device, operation) as the key.
  55     return self.resource_handle.device, self.op
  56 
  57   def batch_runner_fn(self):
  58     return _scheduled_stamp_resource_op_runner
  59

  61 def _move_tensors(tensors, device):
  62   """Moves a list of tensors to a device by concatenating/splitting them."""
  63   # Reset the device setting to avoid weird interactions with device merging
  64   # logic.
  65   zero = constant_op.constant(0, dtype=dtypes.int32)
  66   with ops.device(None):
  67     if all(tensor.shape == tensor_shape.scalar() for tensor in tensors):
  68       with ops.device(tensors[0].device):
  69         values = array_ops.stack(tensors)
  70       with ops.device(device):
  71         return array_ops.unstack(values)
  72     else:
  73       with ops.device(tensors[0].device):
  74         sizes = array_ops.stack(array_ops.shape_n(tensors))[:, 0]
  75         values = array_ops.concat(tensors, axis=zero)
  76       with ops.device(device):
  77         sizes = array_ops.unstack(sizes)
  78         return list(array_ops.split(values, sizes, axis=zero))

  81 def _scheduled_stamp_resource_op_runner(batch, stamp):
  82   """Runs a batch operation on a stamped resource."""
  83   if not batch:
  84     return
  85   arg_keys = set(batch[0].args.keys())
  86   grouped_args = collections.OrderedDict()
  87   resource_handles = []
  88   # Check that the set of arguments is the same across all the scheduled ops.
  89   for op in batch:
  90     if set(op.args.keys()) != arg_keys:
  91       raise ValueError("Mismatching arguments: %s, %s.", op.args, arg_keys)
  92     for key in arg_keys:
  93       grouped_args.setdefault(key, []).append(op.args[key])
  94     resource_handles.append(op.resource_handle)
  95   # Move all the inputs to the op device in one RPC.
  96   grouped_args = collections.OrderedDict(
  97       (k, _move_tensors(v, resource_handles[0].device))
  98       for k, v in sorted(grouped_args.items()))
  99   with ops.device(resource_handles[0].device):
 100     return batch[0].op(resource_handles, stamp, **grouped_args)

 103 def run_handler_scheduled_ops(per_handler_ops, stamp, worker_device):
 104   """Given a dictionary of ops for each handler, runs them in batch."""
 105   batched_ops = collections.OrderedDict()
 106   # Group the ops by their batching_key. Ops that share the same batching key
 107   # can be executed together.
 108   for handler in per_handler_ops.keys():
 109     for op in per_handler_ops[handler]:
 110       key = (op.batching_key(), op.batch_runner_fn())
 111       batched_ops.setdefault(key, []).append(op)
 112   op_results = {}
 113   for batch in batched_ops.values():
 114     # Run each of the batched ops using its runner.
 115     results = batch[0].batch_runner_fn()(batch, stamp)
 116     # If the result is a tuple, move each entry in the tuple in one RPC.
 117     if isinstance(results, tuple):
 118       results = tuple(
 119           _move_tensors(result, worker_device) for result in results)
 120       # Once all the results are on the worker, create individual tuple for
 121       # each scheduled op request.
 122       for i in range(len(batch)):
 123         op_results[batch[i]] = tuple(result[i] for result in results)
 124     # If the result is a tuple, it didn't have any outputs, so use the
 125     # `ops.Operation` as the result for all the scheduled ops.
 126     elif isinstance(results, ops.Operation):
 127       for i in range(len(batch)):
 128         op_results[batch[i]] = results
 129     else:
 130       raise ValueError("Unknown type of result %s.", results)
 131   handler_results = collections.defaultdict(list)
 132   # Dispatch the results of the ScheduledOps to the handlers that requested
 133   # them.
 134   for handler in per_handler_ops.keys():
 135     for op in per_handler_ops[handler]:
 136       handler_results[handler].append(op_results[op])
 137   return handler_results
