  30 # Pattern to remove all non alpha numeric from a string.
  31 _PATTERN = re.compile(r"[\W_]+")
  32 
  33 
  34 class StatsAccumulator(saver.BaseSaverBuilder.SaveableObject):
  35   """A resource that allows to accumulate gradients and hessians.
  36 
  37   For consistency guarantees, we use read and write stamp tokens.
  38   The stamp token on the resource is updated with StatsAccumulator.flush.
  39   Calls to StatsAccumulator.add that don't provide the current stamp token are
  40   ignored.
  41   """
  42 
  43   def __init__(self,
  44                stamp_token,
  45                gradient_shape,
  46                hessian_shape,
  47                name=None,
  48                container=None):
  49     """Creates a stats accumulator and returns a handle to it.
  50 
  51     Args:
  52       stamp_token: An int64, initial value to use for the stamp token.
  53       gradient_shape: A TensorShape, containing shape of gradients.
  54       hessian_shape: A TensorShape, containing shape of hessians.
  55       name: A name for the stats accumulator variable.
  56       container: An optional `string`. Defaults to `""`.
  57 
  58     Returns:
  59       A `Tensor` of type mutable `string`. The handle to the stats accumulator.
  60     """
  61     if name is not None:
  62       name = _PATTERN.sub("", name)
  63     with ops.name_scope(name, "StatsAccumulator") as name:
  64       # Both values are scalars.
  65       if (gradient_shape == tensor_shape.scalar() and
  66           hessian_shape == tensor_shape.scalar()):
  67         self._is_scalar = True
  68         self._resource_handle = (gen_stats_accumulator_ops.
  69                                  stats_accumulator_scalar_resource_handle_op(
  70                                      container, name, name=name))
  71 
  72         create_op = gen_stats_accumulator_ops.create_stats_accumulator_scalar(
  73             self._resource_handle, stamp_token)
  74         is_initialized_op = (
  75             gen_stats_accumulator_ops.stats_accumulator_scalar_is_initialized(
  76                 self._resource_handle))
  77       else:
  78         self._is_scalar = False
  79         self._resource_handle = (gen_stats_accumulator_ops.
  80                                  stats_accumulator_tensor_resource_handle_op(
  81                                      container, name, name=name))
  82         create_op = gen_stats_accumulator_ops.create_stats_accumulator_tensor(
  83             self._resource_handle, stamp_token, gradient_shape.as_list(),
  84             hessian_shape.as_list())
  85         is_initialized_op = (
  86             gen_stats_accumulator_ops.stats_accumulator_tensor_is_initialized(
  87                 self._resource_handle))
  88 
  89     self._create_op = create_op
  90     slice_spec = ""
  91     saver_name = self._resource_handle.name
  92     (stamp_token, num_updates, partition_ids, feature_ids, gradients,
  93      hessians) = self.serialize()
  94     specs = [
  95         saver.BaseSaverBuilder.SaveSpec(stamp_token, slice_spec,
  96                                         saver_name + "_stamp"),
  97         saver.BaseSaverBuilder.SaveSpec(num_updates, slice_spec,
  98                                         saver_name + "_num_updates"),
  99         saver.BaseSaverBuilder.SaveSpec(partition_ids, slice_spec,
 100                                         saver_name + "_partition_ids"),
 101         saver.BaseSaverBuilder.SaveSpec(feature_ids, slice_spec,
 102                                         saver_name + "_feature_ids"),
 103         saver.BaseSaverBuilder.SaveSpec(gradients, slice_spec,
 104                                         saver_name + "_gradients"),
 105         saver.BaseSaverBuilder.SaveSpec(hessians, slice_spec,
 106                                         saver_name + "hessians"),
 107     ]
 108 
 109     super(StatsAccumulator, self).__init__(self._resource_handle, specs, name)
 110     resources.register_resource(self._resource_handle, create_op,
 111                                 is_initialized_op)
 112     ops.add_to_collection(ops.GraphKeys.SAVEABLE_OBJECTS, self)
 113 
 114   def add(self, stamp_token, partition_ids, feature_ids, gradients, hessians):
 115     """Updates the stats accumulator."""
 116     partition_ids, feature_ids, gradients, hessians = (self._make_summary(
 117         partition_ids, feature_ids, gradients, hessians))
 118     if self._is_scalar:
 119       return gen_stats_accumulator_ops.stats_accumulator_scalar_add(
 120           [self._resource_handle], stamp_token, [partition_ids], [feature_ids],
 121           [gradients], [hessians])
 122     else:
 123       return gen_stats_accumulator_ops.stats_accumulator_tensor_add(
 124           [self._resource_handle], stamp_token, [partition_ids], [feature_ids],
 125           [gradients], [hessians])
 126 
 127   def schedule_add(self, partition_ids, feature_ids, gradients, hessians):
 128     """Schedules an update to the stats accumulator."""
 129     partition_ids, feature_ids, gradients, hessians = (self._make_summary(
 130         partition_ids, feature_ids, gradients, hessians))
 131     if self._is_scalar:
 132       return batch_ops_utils.ScheduledStampedResourceOp(
 133           op=gen_stats_accumulator_ops.stats_accumulator_scalar_add,
 134           resource_handle=self._resource_handle,
 135           partition_ids=partition_ids,
 136           feature_ids=feature_ids,
 137           gradients=gradients,
 138           hessians=hessians)
 139     else:
 140       return batch_ops_utils.ScheduledStampedResourceOp(
 141           op=gen_stats_accumulator_ops.stats_accumulator_tensor_add,
 142           resource_handle=self._resource_handle,
 143           partition_ids=partition_ids,
 144           feature_ids=feature_ids,
 145           gradients=gradients,
 146           hessians=hessians)
 147 
 148   def _make_summary(self, partition_ids, feature_ids, gradients, hessians):
 149     if self._is_scalar:
 150       return gen_stats_accumulator_ops.stats_accumulator_scalar_make_summary(
 151           partition_ids, feature_ids, gradients, hessians)
 152     else:
 153       return gen_stats_accumulator_ops.stats_accumulator_tensor_make_summary(
 154           partition_ids, feature_ids, gradients, hessians)
 155 
 156   def deserialize(self, stamp_token, num_updates, partition_ids, feature_ids,
 157                   gradients, hessians):
 158     """Resets the stats accumulator with the serialized state."""
 159     if self._is_scalar:
 160       return gen_stats_accumulator_ops.stats_accumulator_scalar_deserialize(
 161           self._resource_handle, stamp_token, num_updates, partition_ids,
 162           feature_ids, gradients, hessians)
 163     else:
 164       return gen_stats_accumulator_ops.stats_accumulator_tensor_deserialize(
 165           self._resource_handle, stamp_token, num_updates, partition_ids,
 166           feature_ids, gradients, hessians)
 167 
 168   def flush(self, stamp_token, next_stamp_token):
 169     """Flushes the stats accumulator."""
 170     if self._is_scalar:
 171       return gen_stats_accumulator_ops.stats_accumulator_scalar_flush(
 172           self._resource_handle, stamp_token, next_stamp_token)
 173     else:
 174       return gen_stats_accumulator_ops.stats_accumulator_tensor_flush(
 175           self._resource_handle, stamp_token, next_stamp_token)
 176 
 177   def serialize(self):
 178     """Serializes the stats accumulator state."""
 179     if self._is_scalar:
 180       return gen_stats_accumulator_ops.stats_accumulator_scalar_serialize(
 181           self._resource_handle)
 182     else:
 183       return gen_stats_accumulator_ops.stats_accumulator_tensor_serialize(
 184           self._resource_handle)
 185 
 186   def restore(self, restored_tensors, unused_restored_shapes):
 187     """Restores the associated tree ensemble from 'restored_tensors'.
 188 
 189     Args:
 190       restored_tensors: the tensors that were loaded from a checkpoint.
 191       unused_restored_shapes: the shapes this object should conform to after
 192         restore. Not meaningful for trees.
 193 
 194     Returns:
 195       The operation that restores the state of the tree ensemble variable.
 196     """
 197     with ops.control_dependencies([self._create_op]):
 198       return self.deserialize(
 199           stamp_token=restored_tensors[0],
 200           num_updates=restored_tensors[1],
 201           partition_ids=restored_tensors[2],
 202           feature_ids=restored_tensors[3],
 203           gradients=restored_tensors[4],
 204           hessians=restored_tensors[5])
 205 
 206   def resource(self):
 207     return self._resource_handle
