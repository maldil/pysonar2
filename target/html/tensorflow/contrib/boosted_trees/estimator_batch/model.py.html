  37 def model_builder(features,
  38                   labels,
  39                   mode,
  40                   params,
  41                   config,
  42                   output_type=ModelBuilderOutputType.MODEL_FN_OPS):
  43   """Multi-machine batch gradient descent tree model.
  44 
  45   Args:
  46     features: `Tensor` or `dict` of `Tensor` objects.
  47     labels: Labels used to train on.
  48     mode: Mode we are in. (TRAIN/EVAL/INFER)
  49     params: A dict of hyperparameters.
  50       The following hyperparameters are expected:
  51       * head: A `Head` instance.
  52       * learner_config: A config for the learner.
  53       * feature_columns: An iterable containing all the feature columns used by
  54           the model.
  55       * examples_per_layer: Number of examples to accumulate before growing a
  56           layer. It can also be a function that computes the number of examples
  57           based on the depth of the layer that's being built.
  58       * weight_column_name: The name of weight column.
  59       * center_bias: Whether a separate tree should be created for first fitting
  60           the bias.
  61       * override_global_step_value: If after the training is done, global step
  62         value must be reset to this value. This is particularly useful for hyper
  63         parameter tuning, which can't recognize early stopping due to the number
  64         of trees. If None, no override of global step will happen.
  65     config: `RunConfig` of the estimator.
  66     output_type: Whether to return ModelFnOps (old interface) or EstimatorSpec
  67       (new interface).
  68 
  69   Returns:
  70     A `ModelFnOps` object.
  71   Raises:
  72     ValueError: if inputs are not valid.
  73   """
  74   head = params["head"]
  75   learner_config = params["learner_config"]
  76   examples_per_layer = params["examples_per_layer"]
  77   feature_columns = params["feature_columns"]
  78   weight_column_name = params["weight_column_name"]
  79   num_trees = params["num_trees"]
  80   use_core_libs = params["use_core_libs"]
  81   logits_modifier_function = params["logits_modifier_function"]
  82   output_leaf_index = params["output_leaf_index"]
  83   override_global_step_value = params.get("override_global_step_value", None)
  84   num_quantiles = params["num_quantiles"]
  85 
  86   if features is None:
  87     raise ValueError("At least one feature must be specified.")
  88 
  89   if config is None:
  90     raise ValueError("Missing estimator RunConfig.")
  91 
  92   center_bias = params["center_bias"]
  93 
  94   if isinstance(features, ops.Tensor):
  95     features = {features.name: features}
  96 
  97   # Make a shallow copy of features to ensure downstream usage
  98   # is unaffected by modifications in the model function.
  99   training_features = copy.copy(features)
 100   training_features.pop(weight_column_name, None)
 101   global_step = training_util.get_global_step()
 102   with ops.device(global_step.device):
 103     ensemble_handle = model_ops.tree_ensemble_variable(
 104         stamp_token=0,
 105         tree_ensemble_config="",  # Initialize an empty ensemble.
 106         name="ensemble_model")
 107 
 108   # Create GBDT model.
 109   gbdt_model = gbdt_batch.GradientBoostedDecisionTreeModel(
 110       is_chief=config.is_chief,
 111       num_ps_replicas=config.num_ps_replicas,
 112       ensemble_handle=ensemble_handle,
 113       center_bias=center_bias,
 114       examples_per_layer=examples_per_layer,
 115       learner_config=learner_config,
 116       feature_columns=feature_columns,
 117       logits_dimension=head.logits_dimension,
 118       features=training_features,
 119       use_core_columns=use_core_libs,
 120       output_leaf_index=output_leaf_index,
 121       num_quantiles=num_quantiles)
 122   with ops.name_scope("gbdt", "gbdt_optimizer"):
 123     predictions_dict = gbdt_model.predict(mode)
 124     logits = predictions_dict["predictions"]
 125     if logits_modifier_function:
 126       logits = logits_modifier_function(logits, features, mode)
 127 
 128     def _train_op_fn(loss):
 129       """Returns the op to optimize the loss."""
 130       update_op = gbdt_model.train(loss, predictions_dict, labels)
 131       with ops.control_dependencies(
 132           [update_op]), (ops.colocate_with(global_step)):
 133         update_op = state_ops.assign_add(global_step, 1).op
 134         return update_op
 135 
 136   create_estimator_spec_op = getattr(head, "create_estimator_spec", None)
 137 
 138   training_hooks = []
 139   if num_trees:
 140     if center_bias:
 141       num_trees += 1
 142 
 143     finalized_trees, attempted_trees = gbdt_model.get_number_of_trees_tensor()
 144     training_hooks.append(
 145         trainer_hooks.StopAfterNTrees(num_trees, attempted_trees,
 146                                       finalized_trees,
 147                                       override_global_step_value))
 148 
 149   if output_type == ModelBuilderOutputType.MODEL_FN_OPS:
 150     if use_core_libs and callable(create_estimator_spec_op):
 151       model_fn_ops = head.create_estimator_spec(
 152           features=features,
 153           mode=mode,
 154           labels=labels,
 155           train_op_fn=_train_op_fn,
 156           logits=logits)
 157       model_fn_ops = estimator_utils.estimator_spec_to_model_fn_ops(
 158           model_fn_ops)
 159     else:
 160       model_fn_ops = head.create_model_fn_ops(
 161           features=features,
 162           mode=mode,
 163           labels=labels,
 164           train_op_fn=_train_op_fn,
 165           logits=logits)
 166 
 167     if output_leaf_index and gbdt_batch.LEAF_INDEX in predictions_dict:
 168       model_fn_ops.predictions[gbdt_batch.LEAF_INDEX] = predictions_dict[
 169           gbdt_batch.LEAF_INDEX]
 170 
 171     model_fn_ops.training_hooks.extend(training_hooks)
 172     return model_fn_ops
 173   elif output_type == ModelBuilderOutputType.ESTIMATOR_SPEC:
 174     assert callable(create_estimator_spec_op)
 175     estimator_spec = head.create_estimator_spec(
 176         features=features,
 177         mode=mode,
 178         labels=labels,
 179         train_op_fn=_train_op_fn,
 180         logits=logits)
 181 
 182     estimator_spec = estimator_spec._replace(
 183         training_hooks=training_hooks + list(estimator_spec.training_hooks))
 184     return estimator_spec
 185 
 186   return model_fn_ops

 189 def ranking_model_builder(features,
 190                           labels,
 191                           mode,
 192                           params,
 193                           config,
 194                           output_type=ModelBuilderOutputType.MODEL_FN_OPS):
 195   """Multi-machine batch gradient descent tree model for ranking.
 196 
 197   Args:
 198     features: `Tensor` or `dict` of `Tensor` objects.
 199     labels: Labels used to train on.
 200     mode: Mode we are in. (TRAIN/EVAL/INFER)
 201     params: A dict of hyperparameters.
 202       The following hyperparameters are expected:
 203       * head: A `Head` instance.
 204       * learner_config: A config for the learner.
 205       * feature_columns: An iterable containing all the feature columns used by
 206           the model.
 207       * examples_per_layer: Number of examples to accumulate before growing a
 208           layer. It can also be a function that computes the number of examples
 209           based on the depth of the layer that's being built.
 210       * weight_column_name: The name of weight column.
 211       * center_bias: Whether a separate tree should be created for first fitting
 212           the bias.
 213       * ranking_model_pair_keys (Optional): Keys to distinguish between features
 214         for left and right part of the training pairs for ranking. For example,
 215         for an Example with features "a.f1" and "b.f1", the keys would be
 216         ("a", "b").
 217       * override_global_step_value: If after the training is done, global step
 218         value must be reset to this value. This is particularly useful for hyper
 219         parameter tuning, which can't recognize early stopping due to the number
 220         of trees. If None, no override of global step will happen.
 221     config: `RunConfig` of the estimator.
 222     output_type: Whether to return ModelFnOps (old interface) or EstimatorSpec
 223       (new interface).
 224 
 225 
 226   Returns:
 227     A `ModelFnOps` object.
 228   Raises:
 229     ValueError: if inputs are not valid.
 230   """
 231   head = params["head"]
 232   learner_config = params["learner_config"]
 233   examples_per_layer = params["examples_per_layer"]
 234   feature_columns = params["feature_columns"]
 235   weight_column_name = params["weight_column_name"]
 236   num_trees = params["num_trees"]
 237   use_core_libs = params["use_core_libs"]
 238   logits_modifier_function = params["logits_modifier_function"]
 239   output_leaf_index = params["output_leaf_index"]
 240   ranking_model_pair_keys = params["ranking_model_pair_keys"]
 241   override_global_step_value = params.get("override_global_step_value", None)
 242   num_quantiles = params["num_quantiles"]
 243 
 244   if features is None:
 245     raise ValueError("At least one feature must be specified.")
 246 
 247   if config is None:
 248     raise ValueError("Missing estimator RunConfig.")
 249 
 250   center_bias = params["center_bias"]
 251 
 252   if isinstance(features, ops.Tensor):
 253     features = {features.name: features}
 254 
 255   # Make a shallow copy of features to ensure downstream usage
 256   # is unaffected by modifications in the model function.
 257   training_features = copy.copy(features)
 258   training_features.pop(weight_column_name, None)
 259   global_step = training_util.get_global_step()
 260   with ops.device(global_step.device):
 261     ensemble_handle = model_ops.tree_ensemble_variable(
 262         stamp_token=0,
 263         tree_ensemble_config="",  # Initialize an empty ensemble.
 264         name="ensemble_model")
 265 
 266   # Extract the features.
 267   if mode == learn.ModeKeys.TRAIN or mode == learn.ModeKeys.EVAL:
 268     # For ranking pairwise training, we extract two sets of features.
 269     if len(ranking_model_pair_keys) != 2:
 270       raise ValueError("You must provide keys for ranking.")
 271     left_pair_key = ranking_model_pair_keys[0]
 272     right_pair_key = ranking_model_pair_keys[1]
 273     if left_pair_key is None or right_pair_key is None:
 274       raise ValueError("Both pair keys should be provided for ranking.")
 275 
 276     features_1 = {}
 277     features_2 = {}
 278     for name in training_features:
 279       feature = training_features[name]
 280       new_name = name[2:]
 281       if name.startswith(left_pair_key + "."):
 282         features_1[new_name] = feature
 283       else:
 284         assert name.startswith(right_pair_key + ".")
 285         features_2[new_name] = feature
 286 
 287     main_features = features_1
 288     supplementary_features = features_2
 289   else:
 290     # For non-ranking or inference ranking, we have only 1 set of features.
 291     main_features = training_features
 292 
 293   # Create GBDT model.
 294   gbdt_model_main = gbdt_batch.GradientBoostedDecisionTreeModel(
 295       is_chief=config.is_chief,
 296       num_ps_replicas=config.num_ps_replicas,
 297       ensemble_handle=ensemble_handle,
 298       center_bias=center_bias,
 299       examples_per_layer=examples_per_layer,
 300       learner_config=learner_config,
 301       feature_columns=feature_columns,
 302       logits_dimension=head.logits_dimension,
 303       features=main_features,
 304       use_core_columns=use_core_libs,
 305       output_leaf_index=output_leaf_index,
 306       num_quantiles=num_quantiles)
 307 
 308   with ops.name_scope("gbdt", "gbdt_optimizer"):
 309     # Logits for inference.
 310     if mode == learn.ModeKeys.INFER:
 311       predictions_dict = gbdt_model_main.predict(mode)
 312       logits = predictions_dict[gbdt_batch.PREDICTIONS]
 313       if logits_modifier_function:
 314         logits = logits_modifier_function(logits, features, mode)
 315     else:
 316       gbdt_model_supplementary = gbdt_batch.GradientBoostedDecisionTreeModel(
 317           is_chief=config.is_chief,
 318           num_ps_replicas=config.num_ps_replicas,
 319           ensemble_handle=ensemble_handle,
 320           center_bias=center_bias,
 321           examples_per_layer=examples_per_layer,
 322           learner_config=learner_config,
 323           feature_columns=feature_columns,
 324           logits_dimension=head.logits_dimension,
 325           features=supplementary_features,
 326           use_core_columns=use_core_libs,
 327           output_leaf_index=output_leaf_index)
 328 
 329       # Logits for train and eval.
 330       if not supplementary_features:
 331         raise ValueError("Features for ranking must be specified.")
 332 
 333       predictions_dict_1 = gbdt_model_main.predict(mode)
 334       predictions_1 = predictions_dict_1[gbdt_batch.PREDICTIONS]
 335 
 336       predictions_dict_2 = gbdt_model_supplementary.predict(mode)
 337       predictions_2 = predictions_dict_2[gbdt_batch.PREDICTIONS]
 338 
 339       logits = predictions_1 - predictions_2
 340       if logits_modifier_function:
 341         logits = logits_modifier_function(logits, features, mode)
 342 
 343       predictions_dict = predictions_dict_1
 344       predictions_dict[gbdt_batch.PREDICTIONS] = logits
 345 
 346     def _train_op_fn(loss):
 347       """Returns the op to optimize the loss."""
 348       update_op = gbdt_model_main.train(loss, predictions_dict, labels)
 349       with ops.control_dependencies(
 350           [update_op]), (ops.colocate_with(global_step)):
 351         update_op = state_ops.assign_add(global_step, 1).op
 352         return update_op
 353 
 354   create_estimator_spec_op = getattr(head, "create_estimator_spec", None)
 355 
 356   training_hooks = []
 357   if num_trees:
 358     if center_bias:
 359       num_trees += 1
 360 
 361     finalized_trees, attempted_trees = (
 362         gbdt_model_main.get_number_of_trees_tensor())
 363     training_hooks.append(
 364         trainer_hooks.StopAfterNTrees(num_trees, attempted_trees,
 365                                       finalized_trees,
 366                                       override_global_step_value))
 367 
 368   if output_type == ModelBuilderOutputType.MODEL_FN_OPS:
 369     if use_core_libs and callable(create_estimator_spec_op):
 370       model_fn_ops = head.create_estimator_spec(
 371           features=features,
 372           mode=mode,
 373           labels=labels,
 374           train_op_fn=_train_op_fn,
 375           logits=logits)
 376       model_fn_ops = estimator_utils.estimator_spec_to_model_fn_ops(
 377           model_fn_ops)
 378     else:
 379       model_fn_ops = head.create_model_fn_ops(
 380           features=features,
 381           mode=mode,
 382           labels=labels,
 383           train_op_fn=_train_op_fn,
 384           logits=logits)
 385 
 386     if output_leaf_index and gbdt_batch.LEAF_INDEX in predictions_dict:
 387       model_fn_ops.predictions[gbdt_batch.LEAF_INDEX] = predictions_dict[
 388           gbdt_batch.LEAF_INDEX]
 389 
 390     model_fn_ops.training_hooks.extend(training_hooks)
 391     return model_fn_ops
 392 
 393   elif output_type == ModelBuilderOutputType.ESTIMATOR_SPEC:
 394     assert callable(create_estimator_spec_op)
 395     estimator_spec = head.create_estimator_spec(
 396         features=features,
 397         mode=mode,
 398         labels=labels,
 399         train_op_fn=_train_op_fn,
 400         logits=logits)
 401 
 402     estimator_spec = estimator_spec._replace(
 403         training_hooks=training_hooks + list(estimator_spec.training_hooks))
 404     return estimator_spec
 405 
 406   return model_fn_ops
