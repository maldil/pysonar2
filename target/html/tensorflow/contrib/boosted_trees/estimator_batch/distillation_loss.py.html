  15 """Utill functions for distillation loss.
  16 
  17 The distillation loss_fn will be called with the following:
  18 
  19 Args:
  20   dnn_logits: Tensor of logits from the dnn, treated as the "target". This will
  21     be the output of a call to tf.stop_gradient().
  22   tree_logits: Tensor of logits from the tree, treated as the "predictions".
  23   example_weights: Tensor of example weights, or a single scalar.
  24 
  25 Returns:
  26   A scalar indicating the reduced loss for that batch of examples.
  27 
  28 Note: we calls the loss_fn defined in contrib head, which is computing two
  29 losses, first one for training and second one for reporting. We only take the
  30 first one here.
  31 """
  32 from __future__ import absolute_import
  33 from __future__ import division
  34 from __future__ import print_function
  35 
  36 from tensorflow.contrib.learn.python.learn.estimators import head as head_lib
  37 from tensorflow.python.ops import math_ops
  38 from tensorflow.python.ops import nn
  39 
  40 
  41 def _logits_to_label_for_tree(logits, n_classes):
  42   if n_classes == 2:
  43     return math_ops.sigmoid(logits)
  44   else:
  45     return nn.softmax(logits)
  46 
  47 
  48 def create_dnn_to_tree_squared_loss_fn(n_classes):
  49   """Returns a squared loss function for dnn to tree distillation."""
  50 
  51   def _dnn_to_tree_squared_loss(dnn_logits, tree_logits, example_weights):
  52     return head_lib._mean_squared_loss(  # pylint: disable=protected-access
  53         labels=_logits_to_label_for_tree(dnn_logits, n_classes),
  54         logits=_logits_to_label_for_tree(tree_logits, n_classes),
  55         weights=example_weights)[0]
  56 
  57   return _dnn_to_tree_squared_loss
  58 
  59 
  60 def create_dnn_to_tree_cross_entropy_loss_fn(n_classes):
  61   """Returns a cross entropy loss function for dnn to tree distillation."""
  62 
  63   def _dnn_to_tree_cross_entropy_loss(dnn_logits, tree_logits, example_weights):
  64     if n_classes == 2:
  65       return head_lib._log_loss_with_two_classes(  # pylint: disable=protected-access
  66           labels=_logits_to_label_for_tree(dnn_logits, n_classes),
  67           logits=tree_logits,
  68           weights=example_weights)[0]
  69     else:
  70       return head_lib._softmax_cross_entropy_loss(  # pylint: disable=protected-access
  71           labels=_logits_to_label_for_tree(dnn_logits, n_classes),
  72           logits=tree_logits,
  73           weights=example_weights)[0]
  74 
  75   return _dnn_to_tree_cross_entropy_loss
