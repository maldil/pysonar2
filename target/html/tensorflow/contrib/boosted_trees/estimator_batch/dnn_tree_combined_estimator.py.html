  16 
  17 The combined model trains a DNN first, then trains boosted trees to boost the

  62 def _add_hidden_layer_summary(value, tag):
  63   summary.scalar("%s_fraction_of_zero_values" % tag, nn.zero_fraction(value))
  64   summary.histogram("%s_activation" % tag, value)

  67 def _dnn_tree_combined_model_fn(
  68     features,
  69     labels,
  70     mode,
  71     head,
  72     dnn_hidden_units,
  73     dnn_feature_columns,
  74     tree_learner_config,
  75     num_trees,
  76     tree_examples_per_layer,
  77     config=None,
  78     dnn_optimizer="Adagrad",
  79     dnn_activation_fn=nn.relu,
  80     dnn_dropout=None,
  81     dnn_input_layer_partitioner=None,
  82     dnn_input_layer_to_tree=True,
  83     dnn_steps_to_train=10000,
  84     predict_with_tree_only=False,
  85     tree_feature_columns=None,
  86     tree_center_bias=False,
  87     dnn_to_tree_distillation_param=None,
  88     use_core_versions=False,
  89     output_type=model.ModelBuilderOutputType.MODEL_FN_OPS,
  90     override_global_step_value=None):
  91   """DNN and GBDT combined model_fn.
  92 
  93   Args:
  94     features: `dict` of `Tensor` objects.
  95     labels: Labels used to train on.
  96     mode: Mode we are in. (TRAIN/EVAL/INFER)
  97     head: A `Head` instance.
  98     dnn_hidden_units: List of hidden units per layer.
  99     dnn_feature_columns: An iterable containing all the feature columns
 100       used by the model's DNN.
 101     tree_learner_config: A config for the tree learner.
 102     num_trees: Number of trees to grow model to after training DNN.
 103     tree_examples_per_layer: Number of examples to accumulate before
 104       growing the tree a layer. This value has a big impact on model
 105       quality and should be set equal to the number of examples in
 106       training dataset if possible. It can also be a function that computes
 107       the number of examples based on the depth of the layer that's
 108       being built.
 109     config: `RunConfig` of the estimator.
 110     dnn_optimizer: string, `Optimizer` object, or callable that defines the
 111       optimizer to use for training the DNN. If `None`, will use the Adagrad
 112       optimizer with default learning rate of 0.001.
 113     dnn_activation_fn: Activation function applied to each layer of the DNN.
 114       If `None`, will use `tf.nn.relu`.
 115     dnn_dropout: When not `None`, the probability to drop out a given
 116       unit in the DNN.
 117     dnn_input_layer_partitioner: Partitioner for input layer of the DNN.
 118       Defaults to `min_max_variable_partitioner` with `min_slice_size` 64 << 20.
 119     dnn_input_layer_to_tree: Whether to provide the DNN's input layer
 120     as a feature to the tree.
 121     dnn_steps_to_train: Number of steps to train dnn for before switching
 122       to gbdt.
 123     predict_with_tree_only: Whether to use only the tree model output as the
 124       final prediction.
 125     tree_feature_columns: An iterable containing all the feature columns
 126       used by the model's boosted trees. If dnn_input_layer_to_tree is
 127       set to True, these features are in addition to dnn_feature_columns.
 128     tree_center_bias: Whether a separate tree should be created for
 129       first fitting the bias.
 130     dnn_to_tree_distillation_param: A Tuple of (float, loss_fn), where the
 131       float defines the weight of the distillation loss, and the loss_fn, for
 132       computing distillation loss, takes dnn_logits, tree_logits and weight
 133       tensor. If the entire tuple is None, no distillation will be applied. If
 134       only the loss_fn is None, we will take the sigmoid/softmax cross entropy
 135       loss be default. When distillation is applied, `predict_with_tree_only`
 136       will be set to True.
 137     use_core_versions: Whether feature columns and loss are from the core (as
 138       opposed to contrib) version of tensorflow.
 139     output_type: Whether to return ModelFnOps (old interface) or EstimatorSpec
 140       (new interface).
 141     override_global_step_value: If after the training is done, global step
 142       value must be reset to this value. This is particularly useful for hyper
 143       parameter tuning, which can't recognize early stopping due to the number
 144       of trees. If None, no override of global step will happen.
 145 
 146   Returns:
 147     A `ModelFnOps` object.
 148   Raises:
 149     ValueError: if inputs are not valid.
 150   """
 151   if not isinstance(features, dict):
 152     raise ValueError("features should be a dictionary of `Tensor`s. "
 153                      "Given type: {}".format(type(features)))
 154 
 155   if not dnn_feature_columns:
 156     raise ValueError("dnn_feature_columns must be specified")
 157 
 158   if dnn_to_tree_distillation_param:
 159     if not predict_with_tree_only:
 160       logging.warning("update predict_with_tree_only to True since distillation"
 161                       "is specified.")
 162       predict_with_tree_only = True
 163 
 164   # Build DNN Logits.
 165   dnn_parent_scope = "dnn"
 166   dnn_partitioner = dnn_input_layer_partitioner or (
 167       partitioned_variables.min_max_variable_partitioner(
 168           max_partitions=config.num_ps_replicas, min_slice_size=64 << 20))
 169 
 170   if (output_type == model.ModelBuilderOutputType.ESTIMATOR_SPEC and
 171       not use_core_versions):
 172     raise ValueError("You must use core versions with Estimator Spec")
 173 
 174   with variable_scope.variable_scope(
 175       dnn_parent_scope,
 176       values=tuple(six.itervalues(features)),
 177       partitioner=dnn_partitioner):
 178 
 179     with variable_scope.variable_scope(
 180         "input_from_feature_columns",
 181         values=tuple(six.itervalues(features)),
 182         partitioner=dnn_partitioner) as input_layer_scope:
 183       if use_core_versions:
 184         input_layer = feature_column_lib.input_layer(
 185             features=features,
 186             feature_columns=dnn_feature_columns,
 187             weight_collections=[dnn_parent_scope])
 188       else:
 189         input_layer = layers.input_from_feature_columns(
 190             columns_to_tensors=features,
 191             feature_columns=dnn_feature_columns,
 192             weight_collections=[dnn_parent_scope],
 193             scope=input_layer_scope)
 194     previous_layer = input_layer
 195     for layer_id, num_hidden_units in enumerate(dnn_hidden_units):
 196       with variable_scope.variable_scope(
 197           "hiddenlayer_%d" % layer_id,
 198           values=(previous_layer,)) as hidden_layer_scope:
 199         net = layers.fully_connected(
 200             previous_layer,
 201             num_hidden_units,
 202             activation_fn=dnn_activation_fn,
 203             variables_collections=[dnn_parent_scope],
 204             scope=hidden_layer_scope)
 205         if dnn_dropout is not None and mode == model_fn.ModeKeys.TRAIN:
 206           net = layers.dropout(net, keep_prob=(1.0 - dnn_dropout))
 207       _add_hidden_layer_summary(net, hidden_layer_scope.name)
 208       previous_layer = net
 209     with variable_scope.variable_scope(
 210         "logits", values=(previous_layer,)) as logits_scope:
 211       dnn_logits = layers.fully_connected(
 212           previous_layer,
 213           head.logits_dimension,
 214           activation_fn=None,
 215           variables_collections=[dnn_parent_scope],
 216           scope=logits_scope)
 217     _add_hidden_layer_summary(dnn_logits, logits_scope.name)
 218 
 219     def _dnn_train_op_fn(loss):
 220       """Returns the op to optimize the loss."""
 221       return optimizers.optimize_loss(
 222           loss=loss,
 223           global_step=training_util.get_global_step(),
 224           learning_rate=_DNN_LEARNING_RATE,
 225           optimizer=_get_optimizer(dnn_optimizer),
 226           name=dnn_parent_scope,
 227           variables=ops.get_collection(
 228               ops.GraphKeys.TRAINABLE_VARIABLES, scope=dnn_parent_scope),
 229           # Empty summaries to prevent optimizers from logging training_loss.
 230           summaries=[])
 231 
 232   # Build Tree Logits.
 233   global_step = training_util.get_global_step()
 234   with ops.device(global_step.device):
 235     ensemble_handle = model_ops.tree_ensemble_variable(
 236         stamp_token=0,
 237         tree_ensemble_config="",  # Initialize an empty ensemble.
 238         name="ensemble_model")
 239 
 240   tree_features = features.copy()
 241   if dnn_input_layer_to_tree:
 242     tree_features["dnn_input_layer"] = input_layer
 243     tree_feature_columns.append(layers.real_valued_column("dnn_input_layer"))
 244   gbdt_model = gbdt_batch.GradientBoostedDecisionTreeModel(
 245       is_chief=config.is_chief,
 246       num_ps_replicas=config.num_ps_replicas,
 247       ensemble_handle=ensemble_handle,
 248       center_bias=tree_center_bias,
 249       examples_per_layer=tree_examples_per_layer,
 250       learner_config=tree_learner_config,
 251       feature_columns=tree_feature_columns,
 252       logits_dimension=head.logits_dimension,
 253       features=tree_features,
 254       use_core_columns=use_core_versions)
 255 
 256   with ops.name_scope("gbdt"):
 257     predictions_dict = gbdt_model.predict(mode)
 258     tree_logits = predictions_dict["predictions"]
 259 
 260     def _tree_train_op_fn(loss):
 261       """Returns the op to optimize the loss."""
 262       if dnn_to_tree_distillation_param:
 263         loss_weight, loss_fn = dnn_to_tree_distillation_param
 264         weight_tensor = head_lib._weight_tensor(  # pylint: disable=protected-access
 265             features, head.weight_column_name)
 266         dnn_logits_fixed = array_ops.stop_gradient(dnn_logits)
 267 
 268         if loss_fn is None:
 269           # we create the loss_fn similar to the head loss_fn for
 270           # multi_class_head used previously as the default one.
 271           n_classes = 2 if head.logits_dimension == 1 else head.logits_dimension
 272           loss_fn = distillation_loss.create_dnn_to_tree_cross_entropy_loss_fn(
 273               n_classes)
 274 
 275         dnn_to_tree_distillation_loss = loss_weight * loss_fn(
 276             dnn_logits_fixed, tree_logits, weight_tensor)
 277         summary.scalar("dnn_to_tree_distillation_loss",
 278                        dnn_to_tree_distillation_loss)
 279         loss += dnn_to_tree_distillation_loss
 280 
 281       update_op = gbdt_model.train(loss, predictions_dict, labels)
 282       with ops.control_dependencies(
 283           [update_op]), (ops.colocate_with(global_step)):
 284         update_op = state_ops.assign_add(global_step, 1).op
 285         return update_op
 286 
 287   if predict_with_tree_only:
 288     if mode == model_fn.ModeKeys.TRAIN or mode == model_fn.ModeKeys.INFER:
 289       tree_train_logits = tree_logits
 290     else:
 291       tree_train_logits = control_flow_ops.cond(
 292           global_step > dnn_steps_to_train,
 293           lambda: tree_logits,
 294           lambda: dnn_logits)
 295   else:
 296     tree_train_logits = dnn_logits + tree_logits
 297 
 298   def _no_train_op_fn(loss):
 299     """Returns a no-op."""
 300     del loss
 301     return control_flow_ops.no_op()
 302 
 303   if tree_center_bias:
 304     num_trees += 1
 305   finalized_trees, attempted_trees = gbdt_model.get_number_of_trees_tensor()
 306 
 307   if output_type == model.ModelBuilderOutputType.MODEL_FN_OPS:
 308     if use_core_versions:
 309       model_fn_ops = head.create_estimator_spec(
 310           features=features,
 311           mode=mode,
 312           labels=labels,
 313           train_op_fn=_no_train_op_fn,
 314           logits=tree_train_logits)
 315       dnn_train_op = head.create_estimator_spec(
 316           features=features,
 317           mode=mode,
 318           labels=labels,
 319           train_op_fn=_dnn_train_op_fn,
 320           logits=dnn_logits)
 321       dnn_train_op = estimator_utils.estimator_spec_to_model_fn_ops(
 322           dnn_train_op).train_op
 323 
 324       tree_train_op = head.create_estimator_spec(
 325           features=tree_features,
 326           mode=mode,
 327           labels=labels,
 328           train_op_fn=_tree_train_op_fn,
 329           logits=tree_train_logits)
 330       tree_train_op = estimator_utils.estimator_spec_to_model_fn_ops(
 331           tree_train_op).train_op
 332 
 333       model_fn_ops = estimator_utils.estimator_spec_to_model_fn_ops(
 334           model_fn_ops)
 335     else:
 336       model_fn_ops = head.create_model_fn_ops(
 337           features=features,
 338           mode=mode,
 339           labels=labels,
 340           train_op_fn=_no_train_op_fn,
 341           logits=tree_train_logits)
 342       dnn_train_op = head.create_model_fn_ops(
 343           features=features,
 344           mode=mode,
 345           labels=labels,
 346           train_op_fn=_dnn_train_op_fn,
 347           logits=dnn_logits).train_op
 348       tree_train_op = head.create_model_fn_ops(
 349           features=tree_features,
 350           mode=mode,
 351           labels=labels,
 352           train_op_fn=_tree_train_op_fn,
 353           logits=tree_train_logits).train_op
 354 
 355     # Add the hooks
 356     model_fn_ops.training_hooks.extend([
 357         trainer_hooks.SwitchTrainOp(dnn_train_op, dnn_steps_to_train,
 358                                     tree_train_op),
 359         trainer_hooks.StopAfterNTrees(num_trees, attempted_trees,
 360                                       finalized_trees,
 361                                       override_global_step_value)
 362     ])
 363     return model_fn_ops
 364 
 365   elif output_type == model.ModelBuilderOutputType.ESTIMATOR_SPEC:
 366     fusion_spec = head.create_estimator_spec(
 367         features=features,
 368         mode=mode,
 369         labels=labels,
 370         train_op_fn=_no_train_op_fn,
 371         logits=tree_train_logits)
 372     dnn_spec = head.create_estimator_spec(
 373         features=features,
 374         mode=mode,
 375         labels=labels,
 376         train_op_fn=_dnn_train_op_fn,
 377         logits=dnn_logits)
 378     tree_spec = head.create_estimator_spec(
 379         features=tree_features,
 380         mode=mode,
 381         labels=labels,
 382         train_op_fn=_tree_train_op_fn,
 383         logits=tree_train_logits)
 384 
 385     training_hooks = [
 386         trainer_hooks.SwitchTrainOp(dnn_spec.train_op, dnn_steps_to_train,
 387                                     tree_spec.train_op),
 388         trainer_hooks.StopAfterNTrees(num_trees, attempted_trees,
 389                                       finalized_trees,
 390                                       override_global_step_value)
 391     ]
 392     fusion_spec = fusion_spec._replace(training_hooks=training_hooks +
 393                                        list(fusion_spec.training_hooks))
 394     return fusion_spec

 467         used by the model's boosted trees. If dnn_input_layer_to_tree is
 468         set to True, these features are in addition to dnn_feature_columns.
 469       tree_center_bias: Whether a separate tree should be created for
 470         first fitting the bias.
 471       dnn_to_tree_distillation_param: A Tuple of (float, loss_fn), where the
 472         float defines the weight of the distillation loss, and the loss_fn, for
 473         computing distillation loss, takes dnn_logits, tree_logits and weight

 476         loss be default. When distillation is applied, `predict_with_tree_only`
 477         will be set to True.
 478       use_core_versions: Whether feature columns and loss are from the core (as
 479         opposed to contrib) version of tensorflow.
 480       override_global_step_value: If after the training is done, global step
 481         value must be reset to this value. This is particularly useful for hyper
 482         parameter tuning, which can't recognize early stopping due to the number
 483         of trees. If None, no override of global step will happen.
 484     """
 485     head = head_lib.multi_class_head(
 486         n_classes=n_classes,
 487         label_name=label_name,
 488         label_keys=label_keys,
 489         weight_column_name=weight_column_name,
 490         enable_centered_bias=False)
 491 
 492     def _model_fn(features, labels, mode, config):
 493       return _dnn_tree_combined_model_fn(
 494           features=features,
 495           labels=labels,
 496           mode=mode,
 497           head=head,
 498           dnn_hidden_units=dnn_hidden_units,
 499           dnn_feature_columns=dnn_feature_columns,
 500           tree_learner_config=tree_learner_config,
 501           num_trees=num_trees,
 502           tree_examples_per_layer=tree_examples_per_layer,
 503           config=config,
 504           dnn_optimizer=dnn_optimizer,
 505           dnn_activation_fn=dnn_activation_fn,
 506           dnn_dropout=dnn_dropout,
 507           dnn_input_layer_partitioner=dnn_input_layer_partitioner,
 508           dnn_input_layer_to_tree=dnn_input_layer_to_tree,
 509           dnn_steps_to_train=dnn_steps_to_train,
 510           predict_with_tree_only=predict_with_tree_only,
 511           tree_feature_columns=tree_feature_columns,
 512           tree_center_bias=tree_center_bias,
 513           dnn_to_tree_distillation_param=dnn_to_tree_distillation_param,
 514           use_core_versions=use_core_versions,
 515           override_global_step_value=override_global_step_value)
 516 
 517     super(DNNBoostedTreeCombinedClassifier, self).__init__(
 518         model_fn=_model_fn,
 519         model_dir=model_dir,
 520         config=config,
 521         feature_engineering_fn=feature_engineering_fn)

 524 class DNNBoostedTreeCombinedRegressor(estimator.Estimator):
 525   """A regressor that uses a combined DNN/GBDT model."""
 526 
 527   def __init__(self,
 528                dnn_hidden_units,
 529                dnn_feature_columns,
 530                tree_learner_config,
 531                num_trees,
 532                tree_examples_per_layer,
 533                weight_column_name=None,
 534                model_dir=None,
 535                config=None,
 536                label_name=None,
 537                label_dimension=1,
 538                feature_engineering_fn=None,
 539                dnn_optimizer="Adagrad",
 540                dnn_activation_fn=nn.relu,
 541                dnn_dropout=None,
 542                dnn_input_layer_partitioner=None,
 543                dnn_input_layer_to_tree=True,
 544                dnn_steps_to_train=10000,
 545                predict_with_tree_only=False,
 546                tree_feature_columns=None,
 547                tree_center_bias=False,
 548                dnn_to_tree_distillation_param=None,
 549                use_core_versions=False,
 550                override_global_step_value=None):
 551     """Initializes a DNNBoostedTreeCombinedRegressor instance.
 552 
 553     Args:
 554       dnn_hidden_units: List of hidden units per layer for DNN.
