  40 def make_custom_export_strategy(name,
  41                                 convert_fn,
  42                                 feature_columns,
  43                                 export_input_fn,
  44                                 use_core_columns=False):
  45   """Makes custom exporter of GTFlow tree format.
  46 
  47   Args:
  48     name: A string, for the name of the export strategy.
  49     convert_fn: A function that converts the tree proto to desired format and
  50       saves it to the desired location. Can be None to skip conversion.
  51     feature_columns: A list of feature columns.
  52     export_input_fn: A function that takes no arguments and returns an
  53       `InputFnOps`.
  54     use_core_columns: A boolean, whether core feature columns were used.
  55 
  56   Returns:
  57     An `ExportStrategy`.
  58   """
  59   base_strategy = saved_model_export_utils.make_export_strategy(
  60       serving_input_fn=export_input_fn, strip_default_attrs=True)
  61   input_fn = export_input_fn()
  62   (sorted_feature_names, dense_floats, sparse_float_indices, _, _,
  63    sparse_int_indices, _, _) = gbdt_batch.extract_features(
  64        input_fn.features, feature_columns, use_core_columns)
  65 
  66   def export_fn(estimator, export_dir, checkpoint_path=None, eval_result=None):
  67     """A wrapper to export to SavedModel, and convert it to other formats."""
  68     result_dir = base_strategy.export(estimator, export_dir,
  69                                       checkpoint_path,
  70                                       eval_result)
  71     with ops.Graph().as_default() as graph:
  72       with tf_session.Session(graph=graph) as sess:
  73         saved_model_loader.load(
  74             sess, [tag_constants.SERVING], result_dir)
  75         # Note: This is GTFlow internal API and might change.
  76         ensemble_model = graph.get_operation_by_name(
  77             "ensemble_model/TreeEnsembleSerialize")
  78         _, dfec_str = sess.run(ensemble_model.outputs)
  79         dtec = tree_config_pb2.DecisionTreeEnsembleConfig()
  80         dtec.ParseFromString(dfec_str)
  81         # Export the result in the same folder as the saved model.
  82         if convert_fn:
  83           convert_fn(dtec, sorted_feature_names,
  84                      len(dense_floats),
  85                      len(sparse_float_indices),
  86                      len(sparse_int_indices), result_dir, eval_result)
  87         feature_importances = _get_feature_importances(
  88             dtec, sorted_feature_names,
  89             len(dense_floats),
  90             len(sparse_float_indices), len(sparse_int_indices))
  91         sorted_by_importance = sorted(
  92             feature_importances.items(), key=lambda x: -x[1])
  93         assets_dir = os.path.join(
  94             compat.as_bytes(result_dir), compat.as_bytes("assets.extra"))
  95         gfile.MakeDirs(assets_dir)
  96         with gfile.GFile(os.path.join(
  97             compat.as_bytes(assets_dir),
  98             compat.as_bytes("feature_importances")), "w") as f:
  99           f.write("\n".join("%s, %f" % (k, v) for k, v in sorted_by_importance))
 100     return result_dir
 101 
 102   return export_strategy.ExportStrategy(
 103       name, export_fn, strip_default_attrs=True)

 106 def convert_to_universal_format(dtec, sorted_feature_names,
 107                                 num_dense, num_sparse_float,
 108                                 num_sparse_int,
 109                                 feature_name_to_proto=None):
 110   """Convert GTFlow trees to universal format."""
 111   del num_sparse_int  # unused.
 112   model_and_features = generic_tree_model_pb2.ModelAndFeatures()
 113   # TODO(jonasz): Feature descriptions should contain information about how each
 114   # feature is processed before it's fed to the model (e.g. bucketing
 115   # information). As of now, this serves as a list of features the model uses.
 116   for feature_name in sorted_feature_names:
 117     if not feature_name_to_proto:
 118       model_and_features.features[feature_name].SetInParent()
 119     else:
 120       model_and_features.features[feature_name].CopyFrom(
 121           feature_name_to_proto[feature_name])
 122   model = model_and_features.model
 123   model.ensemble.summation_combination_technique.SetInParent()
 124   for tree_idx in range(len(dtec.trees)):
 125     gtflow_tree = dtec.trees[tree_idx]
 126     tree_weight = dtec.tree_weights[tree_idx]
 127     member = model.ensemble.members.add()
 128     member.submodel_id.value = tree_idx
 129     tree = member.submodel.decision_tree
 130     for node_idx in range(len(gtflow_tree.nodes)):
 131       gtflow_node = gtflow_tree.nodes[node_idx]
 132       node = tree.nodes.add()
 133       node_type = gtflow_node.WhichOneof("node")
 134       node.node_id.value = node_idx
 135       if node_type == "leaf":
 136         leaf = gtflow_node.leaf
 137         if leaf.HasField("vector"):
 138           for weight in leaf.vector.value:
 139             new_value = node.leaf.vector.value.add()
 140             new_value.float_value = weight * tree_weight
 141         else:
 142           for index, weight in zip(
 143               leaf.sparse_vector.index, leaf.sparse_vector.value):
 144             new_value = node.leaf.sparse_vector.sparse_value[index]
 145             new_value.float_value = weight * tree_weight
 146       else:
 147         node = node.binary_node
 148         # Binary nodes here.
 149         if node_type == "dense_float_binary_split":
 150           split = gtflow_node.dense_float_binary_split
 151           feature_id = split.feature_column
 152           inequality_test = node.inequality_left_child_test
 153           inequality_test.feature_id.id.value = sorted_feature_names[feature_id]
 154           inequality_test.type = (
 155               generic_tree_model_pb2.InequalityTest.LESS_OR_EQUAL)
 156           inequality_test.threshold.float_value = split.threshold
 157         elif node_type == "sparse_float_binary_split_default_left":
 158           split = gtflow_node.sparse_float_binary_split_default_left.split
 159           node.default_direction = (generic_tree_model_pb2.BinaryNode.LEFT)
 160           feature_id = split.feature_column + num_dense
 161           inequality_test = node.inequality_left_child_test
 162           inequality_test.feature_id.id.value = (
 163               _SPARSE_FLOAT_FEATURE_NAME_TEMPLATE %
 164               (sorted_feature_names[feature_id], split.dimension_id))
 165           model_and_features.features.pop(sorted_feature_names[feature_id])
 166           (model_and_features.features[inequality_test.feature_id.id.value]
 167            .SetInParent())
 168           inequality_test.type = (
 169               generic_tree_model_pb2.InequalityTest.LESS_OR_EQUAL)
 170           inequality_test.threshold.float_value = split.threshold
 171         elif node_type == "sparse_float_binary_split_default_right":
 172           split = gtflow_node.sparse_float_binary_split_default_right.split
 173           node.default_direction = (
 174               generic_tree_model_pb2.BinaryNode.RIGHT)
 175           # TODO(nponomareva): adjust this id assignement when we allow multi-
 176           # column sparse tensors.
 177           feature_id = split.feature_column + num_dense
 178           inequality_test = node.inequality_left_child_test
 179           inequality_test.feature_id.id.value = (
 180               _SPARSE_FLOAT_FEATURE_NAME_TEMPLATE %
 181               (sorted_feature_names[feature_id], split.dimension_id))
 182           model_and_features.features.pop(sorted_feature_names[feature_id])
 183           (model_and_features.features[inequality_test.feature_id.id.value]
 184            .SetInParent())
 185           inequality_test.type = (
 186               generic_tree_model_pb2.InequalityTest.LESS_OR_EQUAL)
 187           inequality_test.threshold.float_value = split.threshold
 188         elif node_type == "categorical_id_binary_split":
 189           split = gtflow_node.categorical_id_binary_split
 190           node.default_direction = generic_tree_model_pb2.BinaryNode.RIGHT
 191           feature_id = split.feature_column + num_dense + num_sparse_float
 192           categorical_test = (
 193               generic_tree_model_extensions_pb2.MatchingValuesTest())
 194           categorical_test.feature_id.id.value = sorted_feature_names[
 195               feature_id]
 196           matching_id = categorical_test.value.add()
 197           matching_id.int64_value = split.feature_id
 198           node.custom_left_child_test.Pack(categorical_test)
 199         else:
 200           raise ValueError("Unexpected node type %s" % node_type)
 201         node.left_child_id.value = split.left_id
 202         node.right_child_id.value = split.right_id
 203   return model_and_features

 206 def _get_feature_importances(dtec, feature_names, num_dense_floats,
 207                              num_sparse_float, num_sparse_int):
 208   """Export the feature importance per feature column."""
 209   del num_sparse_int    # Unused.
 210   sums = collections.defaultdict(lambda: 0)
 211   for tree_idx in range(len(dtec.trees)):
 212     tree = dtec.trees[tree_idx]
 213     for tree_node in tree.nodes:
 214       node_type = tree_node.WhichOneof("node")
 215       if node_type == "dense_float_binary_split":
 216         split = tree_node.dense_float_binary_split
 217         split_column = feature_names[split.feature_column]
 218       elif node_type == "sparse_float_binary_split_default_left":
 219         split = tree_node.sparse_float_binary_split_default_left.split
 220         split_column = _SPARSE_FLOAT_FEATURE_NAME_TEMPLATE % (
 221             feature_names[split.feature_column + num_dense_floats],
 222             split.dimension_id)
 223       elif node_type == "sparse_float_binary_split_default_right":
 224         split = tree_node.sparse_float_binary_split_default_right.split
 225         split_column = _SPARSE_FLOAT_FEATURE_NAME_TEMPLATE % (
 226             feature_names[split.feature_column + num_dense_floats],
 227             split.dimension_id)
 228       elif node_type == "categorical_id_binary_split":
 229         split = tree_node.categorical_id_binary_split
 230         split_column = feature_names[split.feature_column + num_dense_floats +
 231                                      num_sparse_float]
 232       elif node_type == "categorical_id_set_membership_binary_split":
 233         split = tree_node.categorical_id_set_membership_binary_split
 234         split_column = feature_names[split.feature_column + num_dense_floats +
 235                                      num_sparse_float]
 236       elif node_type == "leaf":
 237         assert tree_node.node_metadata.gain == 0
 238         continue
 239       else:
 240         raise ValueError("Unexpected split type %s" % node_type)
 241       # Apply shrinkage factor. It is important since it is not always uniform
 242       # across different trees.
 243       sums[split_column] += (
 244           tree_node.node_metadata.gain * dtec.tree_weights[tree_idx])
 245   return dict(sums)
