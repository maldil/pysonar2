  15 """Implementation of `head.Head` with custom loss and link function."""
  16 
  17 from __future__ import absolute_import
  18 from __future__ import division
  19 from __future__ import print_function
  20 
  21 from tensorflow.contrib.learn.python.learn.estimators import head as head_lib
  22 from tensorflow.python.framework import dtypes
  23 from tensorflow.python.ops import array_ops
  24 from tensorflow.python.ops import math_ops
  25 
  26 
  27 class CustomLossHead(head_lib._RegressionHead):  # pylint: disable=protected-access
  28   """A Head object with custom loss function and link function."""
  29 
  30   def __init__(self,
  31                loss_fn,
  32                link_fn,
  33                logit_dimension,
  34                head_name=None,
  35                weight_column_name=None,
  36                metrics_fn=None):
  37     """`Head` for specifying arbitrary loss function.
  38 
  39     Args:
  40       loss_fn: Loss function.
  41       link_fn: Function that converts logits to prediction.
  42       logit_dimension: Number of dimensions for the logits.
  43       head_name: name of the head. Predictions, summary, metrics keys are
  44         suffixed by `"/" + head_name` and the default variable scope is
  45         `head_name`.
  46       weight_column_name: A string defining feature column name representing
  47         weights. It is used to down weight or boost examples during training. It
  48         will be multiplied by the loss of the example.
  49       metrics_fn: a function that takes predictions dict, labels and weights and
  50         returns a dictionary of metrics to be calculated.
  51     """
  52 
  53     def loss_wrapper(labels, logits, weight_tensor):
  54       if weight_tensor is None:
  55         weight_tensor = array_ops.ones(
  56             shape=[array_ops.shape(labels)[0], 1], dtype=dtypes.float32)
  57       weighted_loss, _ = loss_fn(labels, weight_tensor, logits)
  58       average_loss = math_ops.reduce_mean(weighted_loss)
  59       return average_loss, average_loss / math_ops.reduce_mean(weight_tensor)
  60 
  61     super(CustomLossHead, self).__init__(
  62         loss_fn=loss_wrapper,
  63         link_fn=link_fn,
  64         head_name=head_name,
  65         weight_column_name=weight_column_name,
  66         enable_centered_bias=False,
  67         label_dimension=logit_dimension)
  68 
  69     self._metrics_fn = metrics_fn
  70 
  71   def _metrics(self, eval_loss, predictions, labels, weights):
  72     if self._metrics_fn is not None:
  73       return self._metrics_fn(predictions, labels, weights)
