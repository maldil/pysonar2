  21 from tensorflow.contrib.learn.python.learn.estimators import constants

  23 from tensorflow.contrib.learn.python.learn.estimators import model_fn as contrib_model_fn_lib
  24 from tensorflow.contrib.learn.python.learn.estimators import prediction_key

  28 _CORE_MODE_TO_CONTRIB_MODE_ = {
  29     model_fn_lib.ModeKeys.TRAIN: contrib_model_fn_lib.ModeKeys.TRAIN,
  30     model_fn_lib.ModeKeys.EVAL: contrib_model_fn_lib.ModeKeys.EVAL,
  31     model_fn_lib.ModeKeys.PREDICT: contrib_model_fn_lib.ModeKeys.INFER
  32 }
  33 
  34 
  35 def _core_mode_to_contrib_mode(mode):
  36   return _CORE_MODE_TO_CONTRIB_MODE_[mode]

  38 
  39 def _export_outputs_to_output_alternatives(export_outputs):
  40   """Converts EstimatorSpec.export_outputs to output_alternatives.
  41 
  42   Args:
  43     export_outputs: export_outputs created by create_estimator_spec.
  44   Returns:
  45     converted output_alternatives.
  46   """
  47   output = dict()
  48   if export_outputs is not None:
  49     for key, value in export_outputs.items():
  50       if isinstance(value, export_output.ClassificationOutput):
  51         exported_predictions = {
  52             prediction_key.PredictionKey.SCORES: value.scores,
  53             prediction_key.PredictionKey.CLASSES: value.classes
  54         }
  55         output[key] = (constants.ProblemType.CLASSIFICATION,
  56                        exported_predictions)
  57     return output
  58   return None
  59 
  60 
  61 def estimator_spec_to_model_fn_ops(estimator_spec, export_alternatives=False):
  62   if export_alternatives:
  63     alternatives = _export_outputs_to_output_alternatives(
  64         estimator_spec.export_outputs)
  65   else:
  66     alternatives = []
  67 
  68   return model_fn.ModelFnOps(
  69       mode=_core_mode_to_contrib_mode(estimator_spec.mode),
  70       predictions=estimator_spec.predictions,
  71       loss=estimator_spec.loss,
  72       train_op=estimator_spec.train_op,
  73       eval_metric_ops=estimator_spec.eval_metric_ops,
  74       output_alternatives=alternatives)
