  34 class FeatureImportanceSummarySaver(session_run_hook.SessionRunHook):
  35   """Hook to save feature importance summaries."""
  36 
  37   def __init__(self, model_dir, every_n_steps=1):
  38     """Create a FeatureImportanceSummarySaver Hook.
  39 
  40     This hook creates scalar summaries representing feature importance
  41     for each feature column during training.
  42 
  43     Args:
  44       model_dir: model base output directory.
  45       every_n_steps: frequency, in number of steps, for logging summaries.
  46 
  47     Raises:
  48       ValueError: If one of the arguments is invalid.
  49     """
  50     if model_dir is None:
  51       raise ValueError("model dir must be specified.")
  52     self._model_dir = model_dir
  53     self._every_n_steps = every_n_steps
  54     self._last_triggered_step = None
  55 
  56   def begin(self):
  57     self._global_step_tensor = training_util.get_global_step()
  58     if self._global_step_tensor is None:
  59       raise RuntimeError(
  60           "Global step should be created to use FeatureImportanceSummarySaver.")
  61     graph = ops.get_default_graph()
  62     self._feature_names_tensor = graph.get_tensor_by_name(
  63         "gbdt/feature_names:0")
  64     self._feature_usage_counts_tensor = graph.get_tensor_by_name(
  65         "gbdt/feature_usage_counts:0")
  66     self._feature_gains_tensor = graph.get_tensor_by_name(
  67         "gbdt/feature_gains:0")
  68 
  69   def before_run(self, run_context):
  70     del run_context  # Unused by feature importance summary saver hook.
  71     requests = {
  72         "global_step": self._global_step_tensor,
  73         "feature_names": self._feature_names_tensor,
  74         "feature_usage_counts": self._feature_usage_counts_tensor,
  75         "feature_gains": self._feature_gains_tensor
  76     }
  77     return SessionRunArgs(requests)
  78 
  79   def after_run(self, run_context, run_values):
  80     del run_context  # Unused by feature importance summary saver hook.
  81 
  82     # Read result tensors.
  83     global_step = run_values.results["global_step"]
  84     feature_names = run_values.results["feature_names"]
  85     feature_usage_counts = run_values.results["feature_usage_counts"]
  86     feature_gains = run_values.results["feature_gains"]
  87 
  88     # Ensure summaries are logged at desired frequency
  89     if (self._last_triggered_step is not None and
  90         global_step < self._last_triggered_step + self._every_n_steps):
  91       return
  92 
  93     # Validate tensors.
  94     if (len(feature_names) != len(feature_usage_counts) or
  95         len(feature_names) != len(feature_gains)):
  96       raise RuntimeError(
  97           "Feature names and importance measures have inconsistent lengths.")
  98 
  99     # Compute total usage.
 100     total_usage_count = 0.0
 101     for usage_count in feature_usage_counts:
 102       total_usage_count += usage_count
 103     usage_count_norm = 1.0 / total_usage_count if total_usage_count else 1.0
 104 
 105     # Compute total gain.
 106     total_gain = 0.0
 107     for gain in feature_gains:
 108       total_gain += gain
 109     gain_norm = 1.0 / total_gain if total_gain else 1.0
 110 
 111     # Output summary for each feature.
 112     self._last_triggered_step = global_step
 113     for (name, usage_count, gain) in zip(feature_names, feature_usage_counts,
 114                                          feature_gains):
 115       output_dir = os.path.join(self._model_dir, name.decode("utf-8"))
 116       summary_writer = SummaryWriterCache.get(output_dir)
 117       usage_count_summary = Summary(value=[
 118           Summary.Value(
 119               tag="feature_importance/usage_counts", simple_value=usage_count)
 120       ])
 121       usage_fraction_summary = Summary(value=[
 122           Summary.Value(
 123               tag="feature_importance/usage_fraction",
 124               simple_value=usage_count * usage_count_norm)
 125       ])
 126       summary_writer.add_summary(usage_count_summary, global_step)
 127       summary_writer.add_summary(usage_fraction_summary, global_step)
 128       gains_summary = Summary(value=[
 129           Summary.Value(tag="feature_importance/gains", simple_value=gain)
 130       ])
 131       gains_fraction_summary = Summary(value=[
 132           Summary.Value(
 133               tag="feature_importance/gains_fraction",
 134               simple_value=gain * gain_norm)
 135       ])
 136       summary_writer.add_summary(gains_summary, global_step)
 137       summary_writer.add_summary(gains_fraction_summary, global_step)

 151 class StopAfterNTrees(session_run_hook.SessionRunHook):
 152   """Stop training after building N full trees."""
 153 
 154   def __init__(self, n, num_attempted_trees_tensor, num_finalized_trees_tensor,
 155                override_global_step_value=None):
 156     self._num_trees = n
 157     # num_attempted_trees_tensor and num_finalized_trees_tensor are both
 158     # tensors.
 159     self._num_attempted_trees_tensor = num_attempted_trees_tensor
 160     self._num_finalized_trees_tensor = num_finalized_trees_tensor
 161     self._override_global_step_value = override_global_step_value
 162 
 163   def begin(self):
 164     self._global_step_tensor = training_util.get_global_step()
 165     if self._global_step_tensor is None:
 166       raise RuntimeError("Global step should be created.")
 167 
 168     if self._override_global_step_value is not None:
 169       self._override_global_step_op = state_ops.assign(
 170           self._global_step_tensor, self._override_global_step_value)
 171 
 172   def before_run(self, run_context):
 173     del run_context  # unused by StopTrainingAfterNTrees.
 174     return session_run_hook.SessionRunArgs({
 175         "num_attempted_trees": self._num_attempted_trees_tensor,
 176         "num_finalized_trees": self._num_finalized_trees_tensor,
 177     })
 178 
 179   def after_run(self, run_context, run_values):
 180     num_attempted_trees = run_values.results["num_attempted_trees"]
 181     num_finalized_trees = run_values.results["num_finalized_trees"]
 182     assert num_attempted_trees is not None
 183     assert num_finalized_trees is not None
 184     # Stop when the required number of finalized trees is reached, or when we
 185     # try enough times to build a tree but keep failing.
 186     if (num_finalized_trees >= self._num_trees or
 187         num_attempted_trees > 2 * self._num_trees):
 188       logging.info("Requesting stop since we have reached %d trees.",
 189                    num_finalized_trees)
 190       if self._override_global_step_value is not None:
 191         logging.info("Overriding global steps value.")
 192         run_context.session.run(self._override_global_step_op)
 193       run_context.request_stop()

 196 class SwitchTrainOp(session_run_hook.SessionRunHook):
 197   """Hook that switches the train op after specified number of steps.
 198 
 199   Hook that replaces the train op depending on the number of steps of training
 200   that have taken place. The first_train_op is used till train_steps steps
 201   are reached. Thereafter the second_train_op is used.
 202   """
 203 
 204   def __init__(self, first_train_op, train_steps, second_train_op):
 205     """Initializes a `SwitchTrainOp`."""
 206     self._first_train_op = first_train_op
 207     self._second_train_op = second_train_op
 208     self._train_steps = train_steps
 209 
 210   def _get_train_op_for_global_step(self, current_step):
 211     """Gets train_op for current global step."""
 212     if current_step < self._train_steps:
 213       return self._first_train_op
 214     return self._second_train_op
 215 
 216   def begin(self):
 217     self._global_step_tensor = training_util.get_global_step()
 218     self._current_train_op = control_flow_ops.no_op()
 219     if self._global_step_tensor is None:
 220       raise RuntimeError(
 221           "Global step should be created to use SwitchTrainOp.")
 222 
 223   def before_run(self, run_context):  # pylint: disable=unused-argument
 224     return session_run_hook.SessionRunArgs(
 225         {"global_step": self._global_step_tensor,
 226          "train_op": self._current_train_op})
 227 
 228   def after_run(self, run_context, run_values):
 229     self._current_train_op = self._get_train_op_for_global_step(
 230         run_values.results["global_step"])
