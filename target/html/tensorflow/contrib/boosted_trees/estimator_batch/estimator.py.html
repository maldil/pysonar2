  37 class GradientBoostedDecisionTreeClassifier(estimator.Estimator):
  38   """An estimator using gradient boosted decision trees."""
  39 
  40   def __init__(self,
  41                learner_config,
  42                examples_per_layer,
  43                n_classes=2,
  44                num_trees=None,
  45                feature_columns=None,
  46                weight_column_name=None,
  47                model_dir=None,
  48                config=None,
  49                label_keys=None,
  50                feature_engineering_fn=None,
  51                logits_modifier_function=None,
  52                center_bias=True,
  53                use_core_libs=False,
  54                output_leaf_index=False,
  55                override_global_step_value=None,
  56                num_quantiles=100):
  57     """Initializes a GradientBoostedDecisionTreeClassifier estimator instance.
  58 
  59     Args:
  60       learner_config: A config for the learner.
  61       examples_per_layer: Number of examples to accumulate before growing a
  62         layer. It can also be a function that computes the number of examples
  63         based on the depth of the layer that's being built.
  64       n_classes: Number of classes in the classification.
  65       num_trees: An int, number of trees to build.
  66       feature_columns: A list of feature columns.
  67       weight_column_name: Name of the column for weights, or None if not
  68         weighted.
  69       model_dir: Directory for model exports, etc.
  70       config: `RunConfig` object to configure the runtime settings.
  71       label_keys: Optional list of strings with size `[n_classes]` defining the
  72         label vocabulary. Only supported for `n_classes` > 2.
  73       feature_engineering_fn: Feature engineering function. Takes features and
  74         labels which are the output of `input_fn` and returns features and
  75         labels which will be fed into the model.
  76       logits_modifier_function: A modifier function for the logits.
  77       center_bias: Whether a separate tree should be created for first fitting
  78         the bias.
  79       use_core_libs: Whether feature columns and loss are from the core (as
  80         opposed to contrib) version of tensorflow.
  81       output_leaf_index: whether to output leaf indices along with predictions
  82         during inference. The leaf node indexes are available in predictions
  83         dict by the key 'leaf_index'. It is a Tensor of rank 2 and its shape is
  84         [batch_size, num_trees].
  85         For example,
  86         result_iter = classifier.predict(...)
  87         for result_dict in result_iter:
  88           # access leaf index list by result_dict["leaf_index"]
  89           # which contains one leaf index per tree
  90       override_global_step_value: If after the training is done, global step
  91         value must be reset to this value. This should be used to reset global
  92         step to a number > number of steps used to train the current ensemble.
  93         For example, the usual way is to train a number of trees and set a very
  94         large number of training steps. When the training is done (number of
  95         trees were trained), this parameter can be used to set the global step
  96         to a large value, making it look like that number of training steps ran.
  97         If None, no override of global step will happen.
  98       num_quantiles: Number of quantiles to build for numeric feature values.
  99 
 100     Raises:
 101       ValueError: If learner_config is not valid.
 102     """
 103     if n_classes > 2:
 104       # For multi-class classification, use our loss implementation that
 105       # supports second order derivative.
 106       def loss_fn(labels, logits, weights=None):
 107         result = losses.per_example_maxent_loss(
 108             labels=labels,
 109             logits=logits,
 110             weights=weights,
 111             num_classes=n_classes)
 112         return math_ops.reduce_mean(result[0])
 113     else:
 114       loss_fn = None
 115     head = head_lib.multi_class_head(
 116         n_classes=n_classes,
 117         weight_column_name=weight_column_name,
 118         enable_centered_bias=False,
 119         loss_fn=loss_fn,
 120         label_keys=label_keys)
 121     if learner_config.num_classes == 0:
 122       learner_config.num_classes = n_classes
 123     elif learner_config.num_classes != n_classes:
 124       raise ValueError("n_classes (%d) doesn't match learner_config (%d)." %
 125                        (learner_config.num_classes, n_classes))
 126     super(GradientBoostedDecisionTreeClassifier, self).__init__(
 127         model_fn=model.model_builder,
 128         params={
 129             'head': head,
 130             'feature_columns': feature_columns,
 131             'learner_config': learner_config,
 132             'num_trees': num_trees,
 133             'weight_column_name': weight_column_name,
 134             'examples_per_layer': examples_per_layer,
 135             'center_bias': center_bias,
 136             'logits_modifier_function': logits_modifier_function,
 137             'use_core_libs': use_core_libs,
 138             'output_leaf_index': output_leaf_index,
 139             'override_global_step_value': override_global_step_value,
 140             'num_quantiles': num_quantiles,
 141         },
 142         model_dir=model_dir,
 143         config=config,
 144         feature_engineering_fn=feature_engineering_fn)

 147 class GradientBoostedDecisionTreeRegressor(estimator.Estimator):
 148   """An estimator using gradient boosted decision trees."""
 149 
 150   def __init__(self,
 151                learner_config,
 152                examples_per_layer,
 153                label_dimension=1,
 154                num_trees=None,
 155                feature_columns=None,
 156                label_name=None,
 157                weight_column_name=None,
 158                model_dir=None,
 159                config=None,
 160                feature_engineering_fn=None,
 161                logits_modifier_function=None,
 162                center_bias=True,
 163                use_core_libs=False,
 164                output_leaf_index=False,
 165                override_global_step_value=None,
 166                num_quantiles=100):
 167     """Initializes a GradientBoostedDecisionTreeRegressor estimator instance.
 168 
 169     Args:
 170       learner_config: A config for the learner.
 171       examples_per_layer: Number of examples to accumulate before growing a
 172         layer. It can also be a function that computes the number of examples
 173         based on the depth of the layer that's being built.
 174       label_dimension: Number of regression labels per example. This is the size
 175         of the last dimension of the labels `Tensor` (typically, this has shape
 176         `[batch_size, label_dimension]`).
 177       num_trees: An int, number of trees to build.
 178       feature_columns: A list of feature columns.
 179       label_name: String, name of the key in label dict. Can be null if label
 180           is a tensor (single headed models).
 181       weight_column_name: Name of the column for weights, or None if not
 182         weighted.
 183       model_dir: Directory for model exports, etc.
 184       config: `RunConfig` object to configure the runtime settings.
 185       feature_engineering_fn: Feature engineering function. Takes features and
 186         labels which are the output of `input_fn` and returns features and
 187         labels which will be fed into the model.
 188       logits_modifier_function: A modifier function for the logits.
 189       center_bias: Whether a separate tree should be created for first fitting
 190         the bias.
 191       use_core_libs: Whether feature columns and loss are from the core (as
 192         opposed to contrib) version of tensorflow.
 193       output_leaf_index: whether to output leaf indices along with predictions
 194         during inference. The leaf node indexes are available in predictions
 195         dict by the key 'leaf_index'. For example,
 196         result_dict = classifier.predict(...)
 197         for example_prediction_result in result_dict:
 198           # access leaf index list by example_prediction_result["leaf_index"]
 199           # which contains one leaf index per tree
 200       override_global_step_value: If after the training is done, global step
 201         value must be reset to this value. This should be used to reset global
 202         step to a number > number of steps used to train the current ensemble.
 203         For example, the usual way is to train a number of trees and set a very
 204         large number of training steps. When the training is done (number of
 205         trees were trained), this parameter can be used to set the global step
 206         to a large value, making it look like that number of training steps ran.
 207         If None, no override of global step will happen.
 208       num_quantiles: Number of quantiles to build for numeric feature values.
 209     """
 210     head = head_lib.regression_head(
 211         label_name=label_name,
 212         label_dimension=label_dimension,
 213         weight_column_name=weight_column_name,
 214         enable_centered_bias=False)
 215     if label_dimension == 1:
 216       learner_config.num_classes = 2
 217     else:
 218       learner_config.num_classes = label_dimension
 219     super(GradientBoostedDecisionTreeRegressor, self).__init__(
 220         model_fn=model.model_builder,
 221         params={
 222             'head': head,
 223             'feature_columns': feature_columns,
 224             'learner_config': learner_config,
 225             'num_trees': num_trees,
 226             'weight_column_name': weight_column_name,
 227             'examples_per_layer': examples_per_layer,
 228             'logits_modifier_function': logits_modifier_function,
 229             'center_bias': center_bias,
 230             'use_core_libs': use_core_libs,
 231             'output_leaf_index': False,
 232             'override_global_step_value': override_global_step_value,
 233             'num_quantiles': num_quantiles,
 234         },
 235         model_dir=model_dir,
 236         config=config,
 237         feature_engineering_fn=feature_engineering_fn)

 417 # ================== New Estimator interface===================================
 418 # The estimators below use new core Estimator interface and must be used with
 419 # new feature columns and heads.
 420 
 421 # For multiclass classification, use the following head since it uses loss
 422 # that is twice differentiable.
 423 def core_multiclass_head(n_classes):
 424   """Core head for multiclass problems."""
 425 
 426   def loss_fn(labels, logits):
 427     result = losses.per_example_maxent_loss(
 428         labels=labels, logits=logits, weights=None, num_classes=n_classes)
 429     return result[0]
 430 
 431   # pylint:disable=protected-access
 432   head_fn = core_head_lib._multi_class_head_with_softmax_cross_entropy_loss(
 433       n_classes=n_classes,
 434       loss_fn=loss_fn,
 435       loss_reduction=core_losses.Reduction.SUM_OVER_NONZERO_WEIGHTS)
 436   # pylint:enable=protected-access
 437 
 438   return head_fn
