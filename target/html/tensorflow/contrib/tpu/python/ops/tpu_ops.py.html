  41   def _create_default_group_assignment():
  42     num_shards = tpu_function.get_tpu_context().number_of_shards
  43     if num_shards is None:
  44       logging.warning(
  45           "cross_replica_sum should be used within a tpu_shard_context, but "
  46           "got unset number_of_shards. Assuming 1.")
  47       num_shards = 1
  48     group_assignment = [list(range(num_shards))]
  49     return group_assignment

  51   def all_to_all(x,
  52                  concat_dimension,
  53                  split_dimension,
  54                  split_count,
  55                  group_assignment=None,
  56                  name=None):
  57     """Exchange data across TPU replicas.
  58 
  59     Args:
  60       x: The local tensor.
  61       concat_dimension: The dimension number to concatenate.
  62       split_dimension: The dimension number to split.
  63       split_count: The number of splits, this number must equal to the sub-group
  64         size(group_assignment.get_shape()[1])
  65       group_assignment: Optional 2d int32 lists with shape [num_groups,
  66         num_replicas_per_group]. `group_assignment[i]` represents the replica
  67         ids in the ith subgroup.
  68       name: Optional op name.
  69 
  70     Returns:
  71       A `Tensor` which is concatenated by data from different replicas.
  72     """
  73     if group_assignment is None:
  74       group_assignment = _create_default_group_assignment()
  75     return gen_tpu_ops.all_to_all(
  76         x,
  77         group_assignment,
  78         concat_dimension=concat_dimension,
  79         split_dimension=split_dimension,
  80         split_count=split_count,
  81         name=name)
  82 
  83   @ops.RegisterGradient("AllToAll")
  84   def _all_to_all_grad(op, grad):
  85     # The gradient of a all-to-all is also a all-to-all but the
  86     # split_dimension and concat_dimension is swapped.
  87     # The graident with respect to group_assignment is None.
  88     return [
  89         gen_tpu_ops.all_to_all(
  90             grad,
  91             op.inputs[1],
  92             concat_dimension=op.get_attr("split_dimension"),
  93             split_dimension=op.get_attr("concat_dimension"),
  94             split_count=op.get_attr("split_count")), None
  95     ]

  97   def cross_replica_sum(x, group_assignment=None, name=None):
  98     """Sum the input tensor across replicas according to group_assignment.
  99 
 100     Args:
 101       x: The local tensor to the sum.
 102       group_assignment: Optional 2d int32 lists with shape [num_groups,
 103         num_replicas_per_group]. `group_assignment[i]` represents the replica
 104         ids in the ith subgroup.
 105       name: Optional op name.
 106 
 107     Returns:
 108       A `Tensor` which is summed across replicas.
 109     """
 110     if group_assignment is None:
 111       group_assignment = _create_default_group_assignment()
 112 
 113     return gen_tpu_ops.cross_replica_sum(x, group_assignment, name=name)

 115   def collective_permute(x, source_target_pairs, name=None):
 116     """Permute the input tensor across replicas given source_target_pairs.
 117 
 118     For each source_target_pair <a, b>, we send replica a's input to replica b.
 119     Each replica id must only appear once in the source column. Also it must
 120     only appear once in the target column.
 121     For the replica id not in the target column, this op returns a zero tensor
 122     with the same shape and dtype of the input x.
 123 
 124     For example, suppose there are 4 TPU instances: `[A, B, C, D]`. Passing
 125     source_target_pairs=`[[0,1],[1,2],[2,3]]` gets the outputs:
 126     `[0, A, B, C]`.
 127 
 128     Args:
 129       x: The local tensor to be permuted.
 130       source_target_pairs: 2d int lists with shape [num_pairs, 2].
 131         source_target_pairs[i][0] represents the source replica id and
 132         source_target_pairs[i][1] represents the target replica id.
 133       name: Optional op name.
 134 
 135     Returns:
 136       A `Tensor` which is permuted.
 137     """
 138     return gen_tpu_ops.collective_permute(x, source_target_pairs, name=name)

 140   @ops.RegisterGradient("CrossReplicaSum")

 142     # The gradient of a cross replica sum is also a cross-replica sum.
 143     # The graident with respect to group_assignment is None.
 144     return [gen_tpu_ops.cross_replica_sum(grad, op.inputs[1]), None]

 155   def infeed_dequeue(dtype, shape, name=None):
 156     """A placeholder op for a value that will be fed into the computation.
 157 
 158     Args:
 159       dtype: A `tf.DType`. The type of elements in the tensor.
 160       shape: A `tf.TensorShape` or list of `ints`. The shape of the tensor.
 161       name: A name for the operation (optional).
 162 
 163     Returns:
 164       A `Tensor` of type `dtype`.
 165       A tensor that will be provided using the infeed mechanism.
 166 
 167     Raises:
 168       TypeError: If 'dtype` is not a supported infeed type.
 169     """
 170     if dtype not in _SUPPORTED_INFEED_DTYPES:
 171       raise TypeError(
 172           "{} is not a supported TPU infeed type. Supported types are: "
 173           "{}".format(dtype, list(_SUPPORTED_INFEED_DTYPES)))
 174 
 175     return gen_tpu_ops.infeed_dequeue(dtype, shape, name=name)

 177   # pylint: disable=redefined-outer-name
 178   def infeed_dequeue_tuple(dtypes, shapes, name=None):
 179     """A placeholder op for values fed into the TPU simultaneously as a tuple.
 180 
 181     Args:
 182       dtypes: A list of `tf.DType`s that has length `>= 1`.
 183         The element types of each element in `outputs`.
 184       shapes: A list of shapes (each a `tf.TensorShape` or list of `ints`).
 185         The shapes of each tensor in `outputs`.
 186       name: A name for the operation (optional).
 187 
 188     Returns:
 189       A list of `Tensor` objects of type `dtypes`.
 190       A list of tensors that will be provided using the infeed mechanism.
 191 
 192     Raises:
 193       TypeError: If a type in 'dtypes` is not a supported infeed type.
 194     """
 195     for dtype in dtypes:
 196       if dtype not in _SUPPORTED_INFEED_DTYPES:
 197         raise TypeError(
 198             "{} is not a supported TPU infeed type. Supported types are: "
 199             "{}".format(dtype, list(_SUPPORTED_INFEED_DTYPES)))
 200     return gen_tpu_ops.infeed_dequeue_tuple(dtypes, shapes, name=name)

 267   def enqueue_tpu_embedding_sparse_batch(sample_indices,
 268                                          embedding_indices,
 269                                          aggregation_weights,
 270                                          device_ordinal,
 271                                          combiners=None,
 272                                          mode_override=None,
 273                                          name=None):
 274     """A placeholder op for enqueueing embedding IDs to the TPU.
 275 
 276     Args:
 277       sample_indices: A list of rank 1 Tensors specifying the training example
 278         and feature to which the corresponding embedding_indices and
 279         aggregation_weights values belong. sample_indices[i] must equal b * nf +
 280         f, where nf is the number of features from the corresponding table, f is
 281         in [0, nf), and b is in [0, batch size).
 282       embedding_indices: A list of rank 1 Tensors, indices into the embedding
 283         tables.
 284       aggregation_weights: A list of rank 1 Tensors containing per sample --
 285         i.e. per (training example, feature) -- aggregation weights.
 286       device_ordinal: The TPU device to use. Should be >= 0 and less than the
 287         number of TPU cores in the task on which the node is placed.
 288       combiners: A list of string scalars, one for each embedding table that
 289         specify how to normalize the embedding activations after weighted
 290         summation. Supported combiners are 'mean', 'sum', or 'sqrtn'. It is
 291         invalid to have the sum of the weights be 0 for 'mean' or the sum of the
 292         squared weights be 0 for 'sqrtn'. If combiners isn't passed, the default
 293         is to use 'sum' for all tables (optional).
 294       mode_override: A string input that overrides the mode specified in the
 295         TPUEmbeddingConfiguration. Supported values are {'unspecified',
 296         'inference', 'training', 'backward_pass_only'}. When set to
 297         'unspecified', the mode set in TPUEmbeddingConfiguration is used,
 298         otherwise mode_override is used (optional).
 299       name: A name for the operation (optional).
 300 
 301     Returns:
 302       An EnqueueTPUEmbeddingSparseBatch operation.
 303     """
 304     if mode_override is None:
 305       mode_override = "unspecified"
 306     return gen_tpu_ops._enqueue_tpu_embedding_sparse_batch(
 307         sample_indices=sample_indices,
 308         embedding_indices=embedding_indices,
 309         aggregation_weights=aggregation_weights,
 310         device_ordinal=device_ordinal,
 311         combiners=combiners,
 312         mode_override=mode_override,
 313         name=name)
