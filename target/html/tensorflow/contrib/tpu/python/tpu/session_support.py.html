  44 def _make_heartbeat_op(session, device, request_ph):
  45   """Return a heartbeat op or None if heartbeats are not supported by device."""
  46   try:
  47     # Test if we can connect in a isolated graph + session
  48     with ops.Graph().as_default():
  49       with session_lib.Session(target=session.sess_str) as temp_session:
  50         with ops.device(device):
  51           heartbeat_op = tpu_ops.worker_heartbeat('')
  52           options = config_pb2.RunOptions(timeout_in_ms=5000)
  53           temp_session.run(heartbeat_op, options=options)
  54   except errors.InvalidArgumentError as _:
  55     logging.warning('Error running heartbeat on %s', device)
  56     return None
  57   except errors.DeadlineExceededError as _:
  58     logging.warning('Timeout connecting to %s when testing heartbeat', device)
  59     return None
  60 
  61   # If we successfully connected and pinged the worker, go ahead and construct
  62   # the operation.
  63   with ops.device(device):
  64     return tpu_ops.worker_heartbeat(request_ph)

  67 class WorkerHeartbeatManager(object):
  68   """Manages the status/heartbeat monitor for a set of workers."""
  69 
  70   def __init__(self, session, devices, heartbeat_ops, request_placeholder):
  71     """Construct a new WorkerHeartbeatManager.
  72 
  73     (Prefer using `WorkerHeartbeatManager.from_devices` when possible.)
  74 
  75     Args:
  76       session: `tf.Session`, session to use for heartbeat operations.
  77       devices: `list[string]` Set of devices to connect to.
  78       heartbeat_ops: `list[tf.Operation]` Heartbeat operations.
  79       request_placeholder: `tf.Placeholder[String]` Placeholder used to specify
  80         the WorkerHeartbeatRequest protocol buffer.
  81     """
  82     self._session = session
  83     self._devices = devices
  84     self._ops = heartbeat_ops
  85     self._request_placeholder = request_placeholder
  86 
  87   @staticmethod
  88   def from_devices(session, devices):
  89     """Construct a heartbeat manager for the given devices."""
  90     if not devices:
  91       logging.error('Trying to create heartbeat manager with no devices?')
  92 
  93     logging.info('Creating heartbeat manager for %s', devices)
  94     request_placeholder = array_ops.placeholder(
  95         name='worker_heartbeat_request', dtype=dtypes.string)
  96 
  97     heartbeat_ops = []
  98     kept_devices = []
  99     for device in devices:
 100       heartbeat_op = _make_heartbeat_op(session, device, request_placeholder)
 101       if heartbeat_op is not None:
 102         kept_devices.append(device)
 103         heartbeat_ops.append(heartbeat_op)
 104       else:
 105         logging.warning('Heartbeat support not available for %s', device)
 106 
 107     return WorkerHeartbeatManager(session, kept_devices, heartbeat_ops,
 108                                   request_placeholder)
 109 
 110   def num_workers(self):
 111     return len(self._devices)
 112 
 113   def configure(self, message):
 114     """Configure heartbeat manager for all devices.
 115 
 116     Args:
 117       message: `event_pb2.WorkerHeartbeatRequest`
 118     Returns: `None`
 119     """
 120     logging.info('Configuring worker heartbeat: %s',
 121                  text_format.MessageToString(message))
 122     self._session.run(self._ops,
 123                       {self._request_placeholder: message.SerializeToString()})
 124 
 125   def ping(self, request=None, timeout_in_ms=5000):
 126     """Ping all workers, returning the parsed status results."""
 127     if request is None:
 128       request = event_pb2.WorkerHeartbeatRequest()
 129 
 130     options = config_pb2.RunOptions(timeout_in_ms=timeout_in_ms)
 131     results = self._session.run(
 132         self._ops,
 133         feed_dict={self._request_placeholder: request.SerializeToString()},
 134         options=options)
 135     parsed_results = [
 136         event_pb2.WorkerHeartbeatResponse.FromString(res_pb)
 137         for res_pb in results
 138     ]
 139     logging.debug('Ping results: %s', parsed_results)
 140     return parsed_results
 141 
 142   def lame_workers(self):
 143     """Ping all workers, returning manager containing lame workers (or None)."""
 144     ping_results = self.ping()
 145     lame_workers = []
 146 
 147     for ping_response, device, op in zip(ping_results, self._devices,
 148                                          self._ops):
 149       if ping_response.health_status != event_pb2.OK:
 150         lame_workers.append((device, op))
 151 
 152     if not lame_workers:
 153       return None
 154 
 155     bad_devices, bad_ops = zip(*lame_workers)
 156     return WorkerHeartbeatManager(self._session, bad_devices, bad_ops,
 157                                   self._request_placeholder)
 158 
 159   def __repr__(self):
 160     return 'HeartbeatManager(%s)' % ','.join(self._devices)
 161 
 162   def shutdown(self, timeout_ms=10000):
 163     """Shutdown all workers after `shutdown_timeout_secs`."""
 164     logging.info('Shutting down %s.', self)
 165     req = event_pb2.WorkerHeartbeatRequest(
 166         watchdog_config=event_pb2.WatchdogConfig(timeout_ms=timeout_ms))
 167     self.configure(req)
 168 
 169     # Wait for workers to shutdown.  This isn't strictly required
 170     # but it avoids triggering multiple checkpoints with the same lame worker.
 171     logging.info('Waiting %dms for worker shutdown.', timeout_ms)
 172     time.sleep(timeout_ms / 1000)
 173

 175 def all_worker_devices(session):
 176   """Return a list of devices for each worker in the system."""
 177   devices = session.list_devices()
 178   return [device.name for device in devices if ':CPU:' in device.name]

 181 class WatchdogManager(threading.Thread):
 182   """Configures worker watchdog timer and handles periodic pings.
 183 
 184   Usage:
 185     # Ping workers every minute, shutting down workers if they haven't received
 186     # a ping after 1 hour.
 187     watchdog_manager = WatchdogManager(
 188       ping_interval=60, shutdown_timeout=3600
 189     )
 190 
 191     # Use as a context manager, resetting watchdog on context exit:
 192     with watchdog_manager:
 193       session.run(...)
 194 
 195     # Or setup globally; watchdog will remain active until program exit.
 196     watchdog_manager.configure_and_run()
 197   """
 198 
 199   def __init__(self,
 200                session,
 201                devices=None,
 202                ping_interval=60,
 203                shutdown_timeout=3600):
 204     """Initialize a watchdog manager.
 205 
 206     Args:
 207       session: Session connected to worker devices.  A cloned session and graph
 208         will be created for managing worker pings.
 209       devices: Set of devices to monitor.  If none, all workers will be
 210         monitored.
 211       ping_interval: Time, in seconds, between watchdog pings.
 212       shutdown_timeout: Time, in seconds, before watchdog timeout.
 213     """
 214     threading.Thread.__init__(self)
 215     self.ping_interval = ping_interval
 216     self.shutdown_timeout = shutdown_timeout
 217     self.daemon = True
 218     self._running = False
 219     self._graph = ops.Graph()
 220     self._session = session_lib.Session(
 221         target=session.sess_str,
 222         graph=self._graph,
 223     )
 224 
 225     with self._graph.as_default():
 226       if devices is None:
 227         devices = all_worker_devices(self._session)
 228       self._worker_manager = WorkerHeartbeatManager.from_devices(
 229           self._session, devices)
 230 
 231   def configure_and_run(self):
 232     logging.info('Enabling worker watchdog.')
 233     self._running = True
 234     self._worker_manager.configure(
 235         event_pb2.WorkerHeartbeatRequest(
 236             watchdog_config=event_pb2.WatchdogConfig(
 237                 timeout_ms=self.shutdown_timeout * 1000,)))
 238 
 239     self.start()
 240 
 241   def __enter__(self):
 242     self.configure_and_run()
 243 
 244   def __exit__(self, exc_type, exc_val, exc_tb):
 245     logging.info('Disabling worker watchdog.')
 246     self._worker_manager.configure(
 247         event_pb2.WorkerHeartbeatRequest(
 248             watchdog_config=event_pb2.WatchdogConfig(timeout_ms=-1,)))
 249     self._running = False
 250     self.join()
 251 
 252   def run(self):
 253     # Don't fetch logs or adjust timing: just ping the watchdog.
 254     while self._running:
 255       self._worker_manager.ping(request=None)
 256       time.sleep(self.ping_interval)

 259 class GracefulShutdownHook(session_run_hook.SessionRunHook):
 260   """Session hook that watches for shutdown events.
 261 
 262   If a shutdown is indicated, `saver.save(checkpoint_prefix)` is executed, and a
 263   SystemShutdown exception is raised to terminate the main session.  If `saver`
 264   is None the `SAVERS` collection will be read to find a saver.
 265 
 266   `on_shutdown_hooks` is an optional list of functions that should be called
 267   after checkpointing.  The function is called with (`run_context`,
 268   `all_workers`, `lame_workers`).
 269 
 270   If `heartbeat_group` is not specified, it will default to all CPU workers
 271   in the system.
 272   """
 273 
 274   def __init__(self, checkpoint_prefix, saver=None, on_shutdown_hooks=None):
 275     self._saver = saver
 276     self._checkpoint_prefix = checkpoint_prefix
 277     self._on_shutdown_hooks = on_shutdown_hooks if on_shutdown_hooks else []
 278 
 279     # Worker heartbeats are managed independently of the main training graph.
 280     self._graph = ops.Graph()
 281     self._workers = None
 282     self._session = None
 283     self._heartbeat_supported = False
 284 
 285   def after_create_session(self, training_session, coord):  # pylint: disable=unused-argument
 286     # N.B. We have to pull the global step here to avoid it being unavailable
 287     # at checkpoint time; the graph has been frozen at that point.
 288     if training_util.get_global_step() is None and self.saver() is not None:
 289       raise ValueError(
 290           'Saver defined but no global step.  Run `get_or_create_global_step()`'
 291           ' in your model definition to allow checkpointing.')
 292 
 293     with self._graph.as_default():
 294       logging.info('Installing graceful shutdown hook.')
 295       self._session = session_lib.Session(
 296           target=training_session.sess_str, graph=self._graph)
 297       self._workers = WorkerHeartbeatManager.from_devices(
 298           self._session, all_worker_devices(self._session))
 299       self._heartbeat_supported = self._workers.num_workers() > 0
 300       if self._heartbeat_supported:
 301         self._workers.configure(
 302             event_pb2.WorkerHeartbeatRequest(
 303                 shutdown_mode=event_pb2.WAIT_FOR_COORDINATOR))
 304       else:
 305         logging.warn(
 306             'No workers support hearbeats. Failure handling will be disabled.')
 307 
 308   def saver(self):
 309     if self._saver:
 310       return self._saver
 311 
 312     savers = ops.get_collection(ops.GraphKeys.SAVERS)
 313     if not savers:
 314       return None
 315 
 316     if not isinstance(savers, list):
 317       return savers
 318 
 319     if len(savers) > 1:
 320       logging.error(
 321           'Multiple savers in the SAVERS collection.  On-demand checkpointing '
 322           'will be disabled. Pass an explicit `saver` to the constructor to '
 323           'override this behavior.')
 324       return None
 325 
 326     return savers[0]
 327 
 328   def after_run(self, run_context, run_values):
 329     del run_values
 330 
 331     if not self._heartbeat_supported:
 332       return
 333 
 334     lame_workers = self._workers.lame_workers()
 335     if lame_workers:
 336       logging.info('ShutdownHook: lame workers found: %s', lame_workers)
 337 
 338       if self.saver():
 339         logging.info('ShutdownHook: saving checkpoint to %s',
 340                      self._checkpoint_prefix)
 341         self.saver().save(
 342             run_context.session,
 343             self._checkpoint_prefix,
 344             global_step=training_util.get_global_step(),
 345             write_state=True,
 346         )
 347       else:
 348         logging.info('ShutdownHook: no Saver defined.')
 349 
 350       for fn in self._on_shutdown_hooks:
 351         fn(run_context, self._workers, lame_workers)

 354 class RestartComputation(object):
 355   """Restart the entire computation.
 356 
 357   This hook shuts down all workers and returns control to the top-level by
 358   throwing a CoordinatorShutdownException.
 359   """
 360 
 361   def __init__(self, timeout_ms=10000):
 362     self.timeout_ms = timeout_ms
 363 
 364   def __call__(self, run_context, all_workers, lame_workers):
 365     del run_context, lame_workers
 366     all_workers.shutdown(timeout_ms=self.timeout_ms)
 367 
 368     logging.info('Terminating coordinator.')
 369     raise CoordinatorShutdownException()

 372 class ShutdownLameWorkers(object):
 373   """Shutdown lamed workers.
 374 
 375   Processing will continue normally (typically by waiting for the down
 376   workers to be restarted).
 377   """
 378 
 379   def __init__(self, timeout_ms=10000):
 380     self.timeout_in_ms = timeout_ms
 381 
 382   def __call__(self, run_context, all_workers, lame_workers):
 383     lame_workers.shutdown(timeout_ms=self.timeout_in_ms)
