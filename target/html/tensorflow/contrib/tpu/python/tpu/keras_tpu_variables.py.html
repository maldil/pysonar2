  40 @contextlib.contextmanager
  41 def _handle_graph(handle):
  42   with handle.graph.as_default():

  46 def _enclosing_tpu_context():
  47   # pylint: disable=protected-access
  48   context = ops.get_default_graph()._get_control_flow_context()
  49   # pylint: enable=protected-access
  50   while context is not None and not isinstance(
  51       context, control_flow_ops.XLAControlFlowContext):
  52     context = context.outer_context
  53   return context

  56 class ReplicatedVariable(object):
  57   """A replicated variable for use on TPUs.
  58 
  59   When accessed inside a tpu.replicate() context, this variable acts as if it
  60   is a single variable whose handle is a replicated input to the computation.
  61 
  62   Outside a tpu.replicate() context currently this object has pretty murky
  63   semantics, especially with respect to things such as
  64   * initialization
  65   * colocation.
  66   """
  67 
  68   def __init__(self, name, variables):
  69     self._name = name
  70     self._primary_var = variables[0]
  71     self._vars = variables
  72     self._cached_value = None
  73     self._dtype = variables[0].dtype
  74 
  75   @property
  76   def handle(self):
  77     tpu_context = _enclosing_tpu_context()
  78     if tpu_context is None:
  79       return self._primary_var.handle
  80 
  81     return tpu_context.get_replicated_var_handle(self._name, self._vars)
  82 
  83   @contextlib.contextmanager
  84   def _assign_dependencies(self):
  85     """Makes assignments depend on the cached value, if any.
  86 
  87     This prevents undefined behavior with reads not ordered wrt writes.
  88 
  89     Yields:
  90       None.
  91     """
  92     if self._cached_value is not None:
  93       with ops.control_dependencies([self._cached_value]):
  94         yield
  95     else:
  96       yield
  97 
  98   @property
  99   def initializer(self):
 100     return control_flow_ops.group([v.initializer for v in self._vars])
 101

 103   def graph(self):
 104     return self._primary_var.graph

 144   def _read_variable_op(self):
 145     if _enclosing_tpu_context() is None:
 146       return self._primary_var.read_value()
 147     v = gen_resource_variable_ops.read_variable_op(self.handle, self._dtype)
 148     return v
 149 
 150   def read_value(self):
 151     return self._read_variable_op()
 152 
 153   def is_initialized(self, name=None):
 154     return self._vars[0].is_initialized(name=name)
 155 
 156   def __getitem__(self, *args):
 157     return self.read_value().__getitem__(*args)
 158 
 159   def assign(self, value, use_locking=None, name=None, read_value=False):
 160     """Assign `value` to all replicas.
 161 
 162     Outside of the tpu.rewrite context, assign explicitly to all replicas.
 163     Inside of the tpu.rewrite context, assigns to the local replica.
 164 
 165     Arguments:
 166       value: Tensor to assign
 167       use_locking: ignored
 168       name: ignored
 169       read_value: return the value from the assignment
 170     Returns:
 171       Assignment operation, or new value of the variable if `read_value` is True
 172     """
 173     del use_locking
 174     if _enclosing_tpu_context() is None:
 175       assign_ops = []
 176       with self._assign_dependencies():
 177         for var in self._vars:
 178           assign_ops.append(var.assign(value, use_locking=None, name=name))
 179 
 180         if read_value:
 181           with ops.control_dependencies(assign_ops):
 182             return self.read_value()
 183         else:
 184           return control_flow_ops.group(assign_ops)
 185 
 186     with _handle_graph(self.handle), self._assign_dependencies():
 187       value_tensor = ops.convert_to_tensor(value, dtype=self.dtype)
 188       assign_op = gen_resource_variable_ops.assign_variable_op(
 189           self.handle, value_tensor, name=name)
 190     if read_value:
 191       return self._read_variable_op()
 192     return assign_op
 193 
 194   def assign_add(self, delta, use_locking=None, name=None, read_value=True):
 195     del use_locking
 196     with _handle_graph(self.handle), self._assign_dependencies():
 197       assign_add_op = gen_resource_variable_ops.assign_add_variable_op(
 198           self.handle,
 199           ops.convert_to_tensor(delta, dtype=self.dtype),
 200           name=name)
 201     if read_value:
 202       return self._read_variable_op()
 203     return assign_add_op
 204 
 205   def assign_sub(self, delta, use_locking=None, name=None, read_value=True):
 206     del use_locking
 207     with _handle_graph(self.handle), self._assign_dependencies():
 208       assign_sub_op = gen_resource_variable_ops.assign_sub_variable_op(
 209           self.handle,
 210           ops.convert_to_tensor(delta, dtype=self.dtype),
 211           name=name)
 212     if read_value:
 213       return self._read_variable_op()
 214     return assign_sub_op
 215 
 216   def get(self):
 217     return self._primary_var

 219   def _should_act_as_resource_variable(self):
 220     """Pass resource_variable_ops.is_resource_variable check."""
 221     pass

 223   def _dense_var_to_tensor(self, dtype=None, name=None, as_ref=False):
 224     """Converts a variable to a tensor."""
 225     # pylint: disable=protected-access
 226     if _enclosing_tpu_context() is None:
 227       return self._primary_var._dense_var_to_tensor(dtype, name, as_ref)
 228     # pylint: enable=protected-access
 229     if dtype is not None and dtype != self.dtype:
 230       return NotImplemented
 231     if as_ref:
 232       return self.handle
 233     else:
 234       return self.read_value()

 244   # pylint: disable=protected-access

 255 def replicated_scope(num_replicas):
 256   """Variable scope for constructing replicated variables."""
 257 
 258   def _replicated_variable_getter(getter, name, *args, **kwargs):
 259     """Getter that constructs replicated variables."""
 260     collections = kwargs.pop("collections", None)
 261     if collections is None:
 262       collections = [ops.GraphKeys.GLOBAL_VARIABLES]
 263     kwargs["collections"] = []
 264 
 265     variables = []
 266     index = {}
 267     for i in range(num_replicas):
 268       replica_name = "{}/{}".format(name, i)
 269       with ops.device("device:TPU:{}".format(i)):
 270         v = getter(*args, name=replica_name, **kwargs)
 271         variables.append(v)
 272       index[i] = v
 273     result = ReplicatedVariable(name, variables)
 274 
 275     g = ops.get_default_graph()
 276     # If "trainable" is True, next_creator() will add the member variables
 277     # to the TRAINABLE_VARIABLES collection, so we manually remove
 278     # them and replace with the MirroredVariable. We can't set
 279     # "trainable" to False for next_creator() since that causes functions
 280     # like implicit_gradients to skip those variables.
 281     if kwargs.get("trainable", True):
 282       collections.append(ops.GraphKeys.TRAINABLE_VARIABLES)
 283       l = g.get_collection_ref(ops.GraphKeys.TRAINABLE_VARIABLES)
 284       for v in index.values():
 285         if v in l:
 286           l.remove(v)
 287     g.add_to_collections(collections, result)
 288 
 289     return result
 290 
 291   return variable_scope.variable_scope(
 292       "", custom_getter=_replicated_variable_getter)

 295 @contextlib.contextmanager
 296 def replicated_variable_for_optimizer(num_replicas):
 297   """Context manager for optimizer weights. Overrides K.variable."""
 298   if num_replicas == 1:
 299     yield
 300     return
 301 
 302   try:
 303     old_v = backend.variable
 304 
 305     def opt_variable(value, dtype=None, name=None, constraint=None):
 306       """Instantiates a variable and returns it."""
 307       if dtype is None:
 308         dtype = backend.floatx()
 309 
 310       variables = []
 311       for i in range(num_replicas):
 312         # Keras holds the variables in optimizer class instance , so the name
 313         # does not matter here. ResourceVariable constructor will find a unique
 314         # name (including name=None) for each replica.
 315         with ops.device("device:TPU:{}".format(i)):
 316           v = resource_variable_ops.ResourceVariable(
 317               value,
 318               dtype=dtypes_module.as_dtype(dtype),
 319               name=name,
 320               constraint=constraint)
 321           variables.append(v)
 322       name = "replicate_{}_{}".format("variable" if name is None else name,
 323                                       ops.uid())
 324       v = ReplicatedVariable(name, variables)
 325 
 326       # pylint: disable=protected-access
 327 
 328       if isinstance(value, np.ndarray):
 329         v._keras_shape = value.shape
 330       elif hasattr(value, "shape"):
 331         v._keras_shape = backend.int_shape(value)
 332       v._uses_learning_phase = False
 333       backend.track_variable(v)
 334       return v
 335 
 336     backend.variable = opt_variable
 337     yield
 338 
 339   finally:
 340     backend.variable = old_v
