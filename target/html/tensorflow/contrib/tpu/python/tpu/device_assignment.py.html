  28 def _tpu_device_name(job, task, device):
  29   """Returns the device name for the TPU `device` on `task` of `job`."""
  30   if job is None:
  31     return "/task:%d/device:TPU:%d" % (task, device)
  32   else:
  33     return "/job:%s/task:%d/device:TPU:%d" % (job, task, device)
  34

  36 def _tpu_host_device_name(job, task):
  37   """Returns the device name for the CPU device on `task` of `job`."""
  38   if job is None:
  39     return "/task:%d/device:CPU:0" % task
  40   else:
  41     return "/job:%s/task:%d/device:CPU:0" % (job, task)

  44 class DeviceAssignment(object):
  45   """Mapping from logical cores in a computation to the physical TPU topology.
  46 
  47   Prefer to use the `device_assignment()` helper to construct a
  48   `DeviceAssignment`; it is easier if less flexible than constructing a
  49   `DeviceAssignment` directly.
  50   """
  51 
  52   def __init__(self, topology, core_assignment):
  53     """Constructs a `DeviceAssignment` object.
  54 
  55     Args:
  56       topology: A `Topology` object that describes the physical TPU topology.
  57       core_assignment: A logical to physical core mapping, represented as a
  58         rank 3 numpy array. See the description of the `core_assignment`
  59         property for more details.
  60 
  61     Raises:
  62       ValueError: If `topology` is not `Topology` object.
  63       ValueError: If `core_assignment` is not a rank 3 numpy array.
  64     """
  65     if not isinstance(topology, Topology):
  66       raise ValueError("topology must be a Topology object, got {}".format(
  67           type(topology)))
  68     core_assignment = np.asarray(core_assignment, dtype=np.int32)
  69 
  70     self._topology = topology
  71     self._topology_tasks, self._topology_devices = (
  72         self._invert_topology(topology))
  73 
  74     topology_rank = self._topology_tasks.ndim
  75     if core_assignment.ndim != 3:
  76       raise ValueError("core_assignment must be a rank 3 numpy array, "
  77                        "got shape {}".format(core_assignment.shape))
  78 
  79     self._num_replicas = core_assignment.shape[0]
  80     self._num_cores_per_replica = core_assignment.shape[1]
  81 
  82     if core_assignment.shape[-1] != topology_rank:
  83       raise ValueError(
  84           "minor dimension of core_assignment must have size equal to topology "
  85           "rank ({}), got shape {}".format(topology_rank,
  86                                            core_assignment.shape))
  87 
  88     self._core_assignment = core_assignment
  89     self._task_and_cores_to_replicas = self._compute_task_and_cores_to_replicas(
  90         self._core_assignment, self._topology_tasks)
  91 
  92   def _invert_topology(self, topology):
  93     """Inverts a [task,device,axis] topology to [x,y,z] -> task/device maps."""
  94     mesh_shape = topology.mesh_shape
  95     tasks = np.full(list(mesh_shape), -1, dtype=np.int32)
  96     devices = np.full(list(mesh_shape), -1, dtype=np.int32)
  97     for task in xrange(topology.device_coordinates.shape[0]):
  98       for device in xrange(topology.device_coordinates.shape[1]):
  99         x, y, z = topology.device_coordinates[task, device, :]
 100         tasks[x, y, z] = task
 101         devices[x, y, z] = device
 102     return tasks, devices
 103 
 104   def _compute_task_and_cores_to_replicas(self, core_assignment,
 105                                           topology_tasks):
 106     """Computes a nested dict which maps task and logical core to replicas."""
 107     task_and_cores_to_replicas = {}
 108     for replica in xrange(core_assignment.shape[0]):
 109       for logical_core in xrange(core_assignment.shape[1]):
 110         x, y, z = core_assignment[replica, logical_core, :]
 111         task_id = topology_tasks[x, y, z]
 112         if task_id not in task_and_cores_to_replicas:
 113           task_and_cores_to_replicas[task_id] = {}
 114         if logical_core not in task_and_cores_to_replicas[task_id]:
 115           task_and_cores_to_replicas[task_id][logical_core] = set()
 116 
 117         task_and_cores_to_replicas[task_id][logical_core].add(replica)
 118 
 119     task_to_sorted_replica_id = {}
 120 
 121     for task, core_to_replicas in task_and_cores_to_replicas.items():
 122       core_to_sorted_replicas = {}
 123       for core, replicas in core_to_replicas.items():
 124         core_to_sorted_replicas[core] = sorted(replicas)
 125 
 126       task_to_sorted_replica_id[task] = core_to_sorted_replicas
 127     return task_to_sorted_replica_id
 128 
 129   @property
 130   def topology(self):
 131     """A `Topology` that describes the TPU topology."""
 132     return self._topology
 133 
 134   @property
 135   def num_cores_per_replica(self):
 136     """The number of cores per replica."""
 137     return self._num_cores_per_replica
 138 
 139   @property
 140   def num_replicas(self):
 141     """The number of replicas of the computation."""
 142     return self._num_replicas
 143 
 144   @property
 145   def core_assignment(self):
 146     """The logical to physical core mapping.
 147 
 148     Returns:
 149       An integer numpy array of rank 3, with shape
 150       `[num_replicas, num_cores_per_replica, topology_rank]`. Maps
 151       (replica, logical core) pairs to physical topology coordinates.
 152     """
 153     return self._core_assignment
 154 
 155   def _coordinates(self, replica, logical_core):
 156     """Returns the physical topology coordinates of a logical core."""
 157     return tuple(self.core_assignment[replica, logical_core, :])
 158 
 159   def lookup_replicas(self, task_id, logical_core):
 160     """Lookup replica ids by task number and logical core.
 161 
 162     Args:
 163       task_id: TensorFlow task number.
 164       logical_core: An integer, identifying a logical core.
 165     Returns:
 166       A sorted list of the replicas that are attached to that task and
 167       logical_core.
 168     Raises:
 169       ValueError: If no replica exists in the task which contains the logical
 170       core.
 171     """
 172     try:
 173       return self._task_and_cores_to_replicas[task_id][logical_core]
 174     except KeyError:
 175       raise ValueError(
 176           "Can not find any replica in task: {} contains logical_core: {} ".
 177           format(task_id, logical_core))
 178 
 179   def tpu_ordinal(self, replica=0, logical_core=0):
 180     """Returns the ordinal of the TPU device assigned to a logical core."""
 181     coordinates = self._coordinates(replica, logical_core)
 182     return self._topology_devices[coordinates]
 183 
 184   def host_device(self, replica=0, logical_core=0, job=None):
 185     """Returns the CPU device attached to a logical core."""
 186     coordinates = self._coordinates(replica, logical_core)
 187     return _tpu_host_device_name(job, self._topology_tasks[coordinates])
 188 
 189   def tpu_device(self, replica=0, logical_core=0, job=None):
 190     """Returns the name of the TPU device assigned to a logical core."""
 191     coordinates = self._coordinates(replica, logical_core)
 192     return _tpu_device_name(job, self._topology_tasks[coordinates],
 193                             self._topology_devices[coordinates])

 196 def device_assignment(topology,
 197                       computation_shape=None,
 198                       computation_stride=None,
 199                       num_replicas=1):
 200   """Computes a device_assignment of a computation across a TPU topology.
 201 
 202   Attempts to choose a compact grid of cores for locality.
 203 
 204   Returns a `DeviceAssignment` that describes the cores in the topology assigned
 205   to each core of each replica.
 206 
 207   `computation_shape` and `computation_stride` values should be powers of 2 for
 208   optimal packing.
 209 
 210   Args:
 211     topology: A `Topology` object that describes the TPU cluster topology.
 212       To obtain a TPU topology, evaluate the `Tensor` returned by
 213       `initialize_system` using `Session.run`. Either a serialized
 214       `TopologyProto` or a `Topology` object may be passed. Note: you must
 215       evaluate the `Tensor` first; you cannot pass an unevaluated `Tensor` here.
 216     computation_shape: A rank 1 int32 numpy array with size equal to the
 217       topology rank, describing the shape of the computation's block of cores.
 218       If None, the `computation_shape` is `[1] * topology_rank`.
 219     computation_stride: A rank 1 int32 numpy array of size `topology_rank`,
 220       describing the inter-core spacing of the `computation_shape` cores in the
 221       TPU topology. If None, the `computation_stride` is `[1] * topology_rank`.
 222     num_replicas: The number of computation replicas to run. The replicas will
 223       be packed into the free spaces of the topology.
 224 
 225   Returns:
 226     A DeviceAssignment object, which describes the mapping between the logical
 227     cores in each computation replica and the physical cores in the TPU
 228     topology.
 229 
 230   Raises:
 231     ValueError: If `topology` is not a valid `Topology` object.
 232     ValueError: If `computation_shape` or `computation_stride` are not 1D int32
 233       numpy arrays with shape [3] where all values are positive.
 234     ValueError: If computation's replicas cannot fit into the TPU topology.
 235   """
 236   # Deserialize the Topology proto, if it is a string.
 237   if isinstance(topology, bytes):
 238     topology = Topology(serialized=topology)
 239 
 240   if not isinstance(topology, Topology):
 241     raise ValueError("`topology` is not a Topology object; got {}".format(
 242         type(topology)))
 243 
 244   topology_rank = len(topology.mesh_shape)
 245   mesh_shape = topology.mesh_shape
 246   if computation_shape is None:
 247     computation_shape = np.array([1] * topology_rank, dtype=np.int32)
 248   else:
 249     computation_shape = np.asarray(computation_shape, dtype=np.int32)
 250 
 251   if computation_stride is None:
 252     computation_stride = np.array([1] * topology_rank, dtype=np.int32)
 253   else:
 254     computation_stride = np.asarray(computation_stride, dtype=np.int32)
 255 
 256   if computation_shape.shape != (topology_rank,):
 257     raise ValueError("computation_shape must have shape [{}]; got {}".format(
 258         topology_rank, computation_shape.shape))
 259   if computation_stride.shape != (topology_rank,):
 260     raise ValueError("computation_stride must have shape [{}]; got {}".format(
 261         topology_rank, computation_stride.shape))
 262 
 263   if any(computation_shape < 1):
 264     raise ValueError(
 265         "computation_shape must be positive; got computation_shape={}".format(
 266             computation_shape))
 267   if any(computation_stride < 1):
 268     raise ValueError(
 269         "computation_stride must be positive; got computation_stride={}".format(
 270             computation_stride))
 271 
 272   # Computes the physical size of one computation instance.
 273   computation_footprint = computation_shape * computation_stride
 274   if any(computation_footprint > mesh_shape):
 275     raise ValueError(
 276         "computation footprint {} does not fit in TPU topology shape {}".format(
 277             computation_footprint, mesh_shape))
 278 
 279   # Computes how many copies of the computation footprint fit in the mesh.
 280   block_counts = mesh_shape // computation_footprint
 281 
 282   replica_counts = block_counts * computation_stride
 283   max_replicas = np.prod(replica_counts)
 284   if num_replicas > max_replicas:
 285     raise ValueError(
 286         "requested {} replicas but only {} replicas with shape {} and "
 287         "computation_stride {} fit in a TPU mesh of shape {}".format(
 288             num_replicas, max_replicas, computation_shape, computation_stride,
 289             mesh_shape))
 290 
 291   def ceil_of_ratio(n, m):
 292     return (n + m - 1) // m
 293 
 294   replica_shape = [0] * topology_rank
 295   if num_replicas > 0:
 296     remaining_replicas = num_replicas
 297     remaining_dims = topology_rank
 298 
 299     # Choose dimensions as close to an equal cube as possible, in order of
 300     # increasing dimension size. By visiting dimensions in increasing size, we
 301     # assign the most constrained dimension first, so we won't make infeasible
 302     # choices.
 303     #
 304     # As a secondary sort order, visit the dimensions in reverse order. This
 305     # means we try to use both cores on the same chip in preference to two cores
 306     # on different chips.
 307     for x, ni in sorted(((x, -i) for (i, x) in enumerate(replica_counts))):
 308       i = -ni
 309       target_size = int(math.ceil(remaining_replicas**(1.0 / remaining_dims)))
 310       replica_shape[i] = min(target_size, x)
 311       remaining_replicas = ceil_of_ratio(remaining_replicas, replica_shape[i])
 312       remaining_dims -= 1
 313 
 314     assert remaining_replicas == 1 and remaining_dims == 0
 315 
 316   # Assigns an offset to each replica such that no two replicas overlap.
 317   replica_offsets = np.full([num_replicas, topology_rank], -1, dtype=np.int32)
 318   for replica in xrange(num_replicas):
 319     # Chooses a replica number in each axis.
 320     t = replica
 321     pos = []
 322     for dim in replica_shape[::-1]:
 323       pos.append(t % dim)
 324       t //= dim
 325     replica_pos = np.array(pos[::-1], dtype=np.int32)
 326 
 327     # Determines where that replica starts in each axis.
 328     outer = replica_pos // computation_stride
 329     inner = replica_pos % computation_stride
 330     replica_offsets[replica, :] = outer * computation_footprint + inner
 331 
 332   # Computes a complete logical core -> physical core mapping for each replica.
 333   indices = [
 334       np.arange(0, computation_shape[i] * computation_stride[i],
 335                 computation_stride[i]) for i in xrange(topology_rank)
 336   ]
 337   indices = np.concatenate(
 338       [i[..., np.newaxis] for i in np.meshgrid(*indices, indexing="ij")],
 339       axis=-1)
 340   indices = indices.reshape((-1, topology_rank))
 341   assignment = indices + replica_offsets[:, np.newaxis, :]
 342   return DeviceAssignment(topology, core_assignment=assignment)
