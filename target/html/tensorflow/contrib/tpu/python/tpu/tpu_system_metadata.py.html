  31 _PINGING_MASTER_TIMEOUT_IN_MS = 60 * 1000  # 1 min
  32 _RETRY_TIMES = 120
  33 _INITIAL_TPU_SYSTEM_TIMEOUT_IN_MS = 300 * 1000  # 5 mins
  34 
  35 _TPU_DEVICE_REG = re.compile(r'.*task:(\d+)/.*device:TPU:(\d+)$')
  36 
  37 # _TPUSystemMetadata is used by TPUEstimator to hold TPU configuration,
  38 # including num_cores and num_hosts.
  39 _TPUSystemMetadata = collections.namedtuple('_TPUSystemMetadata', [
  40     'num_cores',
  41     'num_hosts',
  42     'num_of_cores_per_host',
  43     'topology',
  44     'devices',
  45 ])
  46

  48 def _query_tpu_system_metadata(master_address, cluster_def=None,
  49                                query_topology=False):
  50   """Automatically detects the TPU system metadata in the system."""
  51   tpu_core_count = 0
  52   devices = []
  53   device_dict = collections.defaultdict(list)
  54 
  55   retry_count = 1
  56   while True:
  57     logging.info('Querying Tensorflow master (%s) for TPU system metadata.',
  58                  master_address)
  59     try:
  60       with ops.Graph().as_default():
  61         with session_lib.Session(
  62             master_address,
  63             config=get_session_config_with_timeout(
  64                 _PINGING_MASTER_TIMEOUT_IN_MS,
  65                 cluster_def)) as sess:
  66           devices = sess.list_devices()
  67           for device in devices:
  68             match = _TPU_DEVICE_REG.match(device.name)
  69             if match:
  70               host_id = match.group(1)
  71               core_id = match.group(2)
  72               device_dict[host_id].append(core_id)
  73               tpu_core_count += 1
  74           break
  75     except errors.DeadlineExceededError:
  76       msg = ('Failed to connect to the Tensorflow master. The TPU worker may '
  77              'not be ready (still scheduling) or the Tensorflow master address '
  78              'is incorrect: got (%s).' %
  79              (master_address))
  80 
  81       # TODO(xiejw): For local or grpc master we might not need retry logic
  82       # here.
  83       if retry_count <= _RETRY_TIMES:
  84         logging.warning('%s', msg)
  85         logging.warning('Retrying (%d/%d).', retry_count, _RETRY_TIMES)
  86         retry_count += 1
  87       else:
  88         raise ValueError(msg)
  89 
  90   num_of_cores_per_host = 0
  91   if tpu_core_count:
  92     num_cores_per_host_set = set(
  93         [len(core_ids) for core_ids in device_dict.values()])
  94     if len(num_cores_per_host_set) != 1:
  95       raise RuntimeError(
  96           'TPU cores on each host is not same. This should not happen!. '
  97           'devices: {}'.format(devices))
  98     num_of_cores_per_host = num_cores_per_host_set.pop()
  99 
 100   topology = None
 101   if query_topology:
 102     if not tpu_core_count:
 103       raise RuntimeError(
 104           'Cannot find any TPU cores in the system (master address {}). '
 105           'This usually means the master address is incorrect or the '
 106           'TPU worker has some problems. Available devices: {}'.format(
 107               master_address, devices))
 108 
 109     topology = _obtain_topology(master_address, cluster_def)
 110 
 111   metadata = _TPUSystemMetadata(
 112       num_cores=tpu_core_count,
 113       num_hosts=len(device_dict),
 114       num_of_cores_per_host=num_of_cores_per_host,
 115       topology=topology,
 116       devices=devices)
 117 
 118   if tpu_core_count:
 119     logging.info('Found TPU system:')
 120     logging.info('*** Num TPU Cores: %d', metadata.num_cores)
 121     logging.info('*** Num TPU Workers: %d', metadata.num_hosts)
 122     logging.info('*** Num TPU Cores Per Worker: %d',
 123                  metadata.num_of_cores_per_host)
 124     for device in metadata.devices:
 125       logging.info('*** Available Device: %s', device)
 126   else:
 127     logging.info('Failed to find TPU: %s', metadata)
 128   return metadata

 131 def _obtain_topology(master_address, cluster_def):
 132   """Obtains TPU fabric topology."""
 133   try:
 134     logging.info('Initializing TPU system (master: %s) to fetch topology '
 135                  'for model parallelism. This might take a while.',
 136                  master_address)
 137     with ops.Graph().as_default():
 138       session_config = get_session_config_with_timeout(
 139           _INITIAL_TPU_SYSTEM_TIMEOUT_IN_MS, cluster_def)
 140       with session_lib.Session(
 141           master_address, config=session_config) as sess:
 142         topology = sess.run(tpu.initialize_system())
 143         return topology
 144   except errors.DeadlineExceededError:
 145     raise ValueError(
 146         'Fail to initialize TPU system with master (%s). '
 147         'Please double check the TPU system is functional.' % (
 148             master_address))

 151 def get_session_config_with_timeout(timeout_in_secs, cluster_def):
 152   """Returns a session given a timeout and a cluster configuration."""
 153   config = config_pb2.ConfigProto(
 154       operation_timeout_in_ms=timeout_in_secs, cluster_def=cluster_def)
 155   return config
