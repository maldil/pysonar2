  15 """Hook for asynchronous checkpointing.
  16 
  17 This hook dispatches checkpoint writing operations in a separate thread to
  18 allow execution to continue on the main thread.
  19 """
  20 
  21 from __future__ import absolute_import
  22 from __future__ import division
  23 from __future__ import print_function
  24 
  25 import os
  26 import threading
  27 import time
  28 
  29 from tensorflow.core.util.event_pb2 import SessionLog
  30 from tensorflow.python.framework import meta_graph
  31 from tensorflow.python.framework import ops
  32 from tensorflow.python.platform import tf_logging as logging
  33 from tensorflow.python.training import basic_session_run_hooks
  34 from tensorflow.python.training import training_util
  35 from tensorflow.python.training.session_run_hook import SessionRunArgs
  36 from tensorflow.python.training.summary_io import SummaryWriterCache

  39 class AsyncCheckpointSaverHook(basic_session_run_hooks.CheckpointSaverHook):
  40   """Saves checkpoints every N steps or seconds."""
  41 
  42   def __init__(self,
  43                checkpoint_dir,
  44                save_secs=None,
  45                save_steps=None,
  46                saver=None,
  47                checkpoint_basename="model.ckpt",
  48                scaffold=None,
  49                listeners=None):
  50     """Initializes a `CheckpointSaverHook`.
  51 
  52     Args:
  53       checkpoint_dir: `str`, base directory for the checkpoint files.
  54       save_secs: `int`, save every N secs.
  55       save_steps: `int`, save every N steps.
  56       saver: `Saver` object, used for saving.
  57       checkpoint_basename: `str`, base name for the checkpoint files.
  58       scaffold: `Scaffold`, use to get saver object.
  59       listeners: List of `CheckpointSaverListener` subclass instances. Used for
  60         callbacks that run immediately before or after this hook saves the
  61         checkpoint.
  62 
  63     Raises:
  64       ValueError: One of `save_steps` or `save_secs` should be set.
  65       ValueError: At most one of `saver` or `scaffold` should be set.
  66     """
  67     logging.info("Create AsyncCheckpointSaverHook.")
  68     if saver is not None and scaffold is not None:
  69       raise ValueError("You cannot provide both saver and scaffold.")
  70     self._saver = saver
  71     self._save_thread = None
  72     self._write_graph_thread = None
  73     self._checkpoint_dir = checkpoint_dir
  74     self._save_path = os.path.join(checkpoint_dir, checkpoint_basename)
  75     self._scaffold = scaffold
  76     self._timer = basic_session_run_hooks.SecondOrStepTimer(
  77         every_secs=save_secs, every_steps=save_steps)
  78     self._listeners = listeners or []
  79     self._steps_per_run = 1
  80     self._summary_writer = None
  81     self._global_step_tensor = None
  82 
  83   def _set_steps_per_run(self, steps_per_run):
  84     self._steps_per_run = steps_per_run
  85 
  86   def begin(self):
  87     self._summary_writer = SummaryWriterCache.get(self._checkpoint_dir)
  88     self._global_step_tensor = training_util._get_or_create_global_step_read()  # pylint: disable=protected-access
  89     if self._global_step_tensor is None:
  90       raise RuntimeError(
  91           "Global step should be created to use CheckpointSaverHook.")
  92     for l in self._listeners:
  93       l.begin()
  94 
  95   def after_create_session(self, session, coord):
  96     global_step = session.run(self._global_step_tensor)
  97 
  98     # We do write graph and saver_def at the first call of before_run.
  99     # We cannot do this in begin, since we let other hooks to change graph and
 100     # add variables in begin. Graph is finalized after all begin calls.
 101     def _write_graph_fn(self):
 102       training_util.write_graph(
 103           ops.get_default_graph().as_graph_def(add_shapes=True),
 104           self._checkpoint_dir, "graph.pbtxt")
 105     self._write_graph_thread = threading.Thread(target=_write_graph_fn,
 106                                                 args=[self])
 107     self._write_graph_thread.start()
 108 
 109     saver_def = self._get_saver().saver_def if self._get_saver() else None
 110     graph = ops.get_default_graph()
 111     meta_graph_def = meta_graph.create_meta_graph_def(
 112         graph_def=graph.as_graph_def(add_shapes=True), saver_def=saver_def)
 113     self._summary_writer.add_graph(graph)
 114     self._summary_writer.add_meta_graph(meta_graph_def)
 115     # The checkpoint saved here is the state at step "global_step".
 116     self._save(session, global_step)
 117     self._timer.update_last_triggered_step(global_step)
 118 
 119   def before_run(self, run_context):  # pylint: disable=unused-argument
 120     return SessionRunArgs(self._global_step_tensor)
 121 
 122   def after_run(self, run_context, run_values):
 123     global_step = run_context.session.run(self._global_step_tensor)
 124     if self._timer.should_trigger_for_step(global_step):
 125       self._timer.update_last_triggered_step(global_step)
 126       logging.info("Triggering checkpoint. %s", global_step)
 127       if self._save(run_context.session, global_step):
 128         run_context.request_stop()
 129 
 130   def end(self, session):
 131     if self._save_thread:
 132       logging.info("Waiting for any pending checkpoints to finish.")
 133       self._save_thread.join()
 134     if self._write_graph_thread:
 135       logging.info("Waiting for any pending write_graph to finish.")
 136       self._write_graph_thread.join()
 137 
 138     last_step = session.run(self._global_step_tensor)
 139 
 140     # Save the last checkpoint synchronously if needed.
 141     if last_step != self._timer.last_triggered_step():
 142       self._save(session, last_step, asynchronous=False)
 143 
 144     for l in self._listeners:
 145       l.end(session, last_step)
 146 
 147   def _save(self, session, step, asynchronous=True):
 148     """Saves the latest checkpoint, returns should_stop."""
 149 
 150     # Skip saving on step 0
 151     if step == 0:
 152       return
 153 
 154     def _save_fn():
 155       """Run the saver process."""
 156       logging.info("Saving checkpoints for %d into %s.", step, self._save_path)
 157 
 158       start_time = time.time()
 159       for l in self._listeners:
 160         l.before_save(session, step)
 161 
 162       self._get_saver().save(session, self._save_path, global_step=step)
 163       self._summary_writer.add_session_log(
 164           SessionLog(
 165               status=SessionLog.CHECKPOINT, checkpoint_path=self._save_path),
 166           step)
 167       end_time = time.time()
 168       logging.info("Checkpoint actual writing time: (%.3f sec)",
 169                    end_time - start_time)
 170       logging.info("Checkpoint finished for %d into %s.", step, self._save_path)
 171 
 172     for l in self._listeners:
 173       l.before_save(session, step)
 174 
 175     if not asynchronous:
 176       _save_fn()
 177       return
 178 
 179     if self._save_thread is not None:
 180       self._save_thread.join(timeout=0.1)
 181       if self._save_thread.is_alive():
 182         logging.info("Saver thread still in progress, skipping checkpoint.")
 183         return
 184 
 185     self._save_thread = threading.Thread(target=_save_fn)
 186     self._save_thread.start()
 187 
 188   def _get_saver(self):
 189     if self._saver is not None:
 190       return self._saver
 191     elif self._scaffold is not None:
 192       return self._scaffold.saver
 193 
 194     # Get saver from the SAVERS collection if present.
 195     collection_key = ops.GraphKeys.SAVERS
 196     savers = ops.get_collection(collection_key)
 197     if not savers:
 198       raise RuntimeError(
 199           "No items in collection {}. Please add a saver to the collection "
 200           "or provide a saver or scaffold.".format(collection_key))
 201     elif len(savers) > 1:
 202       raise RuntimeError(
 203           "More than one item in collection {}. "
 204           "Please indicate which one to use by passing it to the constructor."
 205           .format(collection_key))
 206 
 207     self._saver = savers[0]
 208     return savers[0]
