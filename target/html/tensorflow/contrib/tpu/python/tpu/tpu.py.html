  75 def initialize_system(embedding_config=None, job=None):
  76   """Initializes a distributed TPU system for use with TensorFlow.
  77 
  78   Args:
  79     embedding_config: If not None, a `TPUEmbeddingConfiguration` proto
  80       describing the desired configuration of the hardware embedding lookup
  81       tables. If embedding_config is None, no hardware embeddings can be used.
  82     job: The job (the XXX in TensorFlow device specification /job:XXX) that
  83       contains the TPU devices that will be initialized. If job=None it is
  84       assumed there is only one job in the TensorFlow flock, and an error will
  85       be returned if this assumption does not hold.
  86   Returns:
  87     A serialized `TopologyProto` that describes the TPU system. Note:
  88       the topology must be evaluated using `Session.run` before it can be used.
  89   """
  90   config_string = ("" if embedding_config is None else
  91                    embedding_config.SerializeToString())
  92   with ops.device(_tpu_system_device_name(job)):
  93     return tpu_ops.configure_distributed_tpu(embedding_config=config_string)

  97   """Shuts down a running a distributed TPU system."""

  99     shutdown_distributed_tpu = tpu_ops.shutdown_distributed_tpu()
 100   return shutdown_distributed_tpu

 115 class TPUReplicateContext(control_flow_ops.XLAControlFlowContext):
 116   """A `ControlFlowContext` for nodes inside a TPU computation.
 117 
 118   The primary role of `TPUReplicateContext` is to mark operators inside a
 119   tpu.replicate() computation with the attribute "_tpu_replicate=XYZ", where XYZ
 120   is a unique name.
 121 
 122   We use a `ControlFlowContext` to perform the annotation since it integrates
 123   with Tensorflow constructs like ResourceVariables. For example, if a
 124   `ResourceVariable` is constructed inside a tpu.replicate() block, the
 125   `ResourceVariable` implementation can use
 126   `with ops.control_dependencies(None)` to build the variable's definition
 127   outside the replicated computation.
 128   """
 129 
 130   def __init__(self, name, num_replicas, pivot):
 131     """Builds a new TPUReplicateContext.
 132 
 133     Args:
 134       name: a unique name for the context, used to populate the `_tpu_replicate`
 135         attribute.
 136       num_replicas: an integer that gives the number of replicas for the
 137         computation.
 138       pivot: a pivot node. Nodes in the TPUReplicateContext that do not have any
 139         inputs will have a control dependency on the pivot node. This ensures
 140         that nodes are correctly included in any enclosing control flow
 141         contexts.
 142     """
 143     super(TPUReplicateContext, self).__init__()
 144     self._num_replicas = num_replicas
 145     self._outer_device_function_stack = None
 146     self._oc_dev_fn_stack = None
 147     self._outside_compilation_cluster = None
 148     self._outside_compilation_counter = 0
 149     self._in_gradient_colocation = None
 150     self._gradient_colocation_stack = []
 151     self._host_compute_core = []
 152     self._name = name
 153     self._name_as_bytes = compat.as_bytes(name)
 154     self._unsupported_ops = []
 155     self._pivot = pivot
 156     self._replicated_vars = {}
 157 
 158   def get_replicated_var_handle(self, name, vars_):
 159     """Returns a variable handle for replicated TPU variable 'var'.
 160 
 161     This is a method used by an experimental replicated variable implementation
 162     and is not intended as a public API.
 163 
 164     Args:
 165       name: The common name of the variable.
 166       vars_: The replicated TPU variables.
 167 
 168     Returns:
 169       The handle of the TPU replicated input node.
 170     """
 171     handle = self._replicated_vars.get(name)
 172     if handle is not None:
 173       return handle
 174 
 175     # Builds a TPUReplicatedInput node for the variable, if one does not already
 176     # exist. The TPUReplicatedInput node must belong to the enclosing
 177     # control-flow scope of the TPUReplicateContext.
 178     # TODO(phawkins): consider changing the contract of the TPU encapsulation
 179     # so the TPUReplicatedInput nodes go inside the TPUReplicateContext scope
 180     # instead.
 181 
 182     # pylint: disable=protected-access
 183     graph = ops.get_default_graph()
 184     saved_context = graph._get_control_flow_context()
 185     graph._set_control_flow_context(self.outer_context)
 186     handle = tpu_ops.tpu_replicated_input(
 187         [v.handle for v in vars_], name=name + "/handle")
 188     graph._set_control_flow_context(saved_context)
 189     # pylint: enable=protected-access
 190     self._replicated_vars[name] = handle
 191     return handle

 193   def report_unsupported_operations(self):
 194     if self._unsupported_ops:
 195       op_str = "\n".join(["  %s (%s)" % (op.type, op.name)
 196                           for op in self._unsupported_ops[:_MAX_WARNING_LINES]])
 197       logging.warning("%d unsupported operations found: \n%s",
 198                       len(self._unsupported_ops), op_str)
 199       if len(self._unsupported_ops) > _MAX_WARNING_LINES:
 200         logging.warning("... and %d more" %
 201                         (len(self._unsupported_ops) - _MAX_WARNING_LINES))

 203   def EnterGradientColocation(self, op, gradient_uid):
 204     if op is not None:
 205       self._gradient_colocation_stack.append(op)
 206       if not self._outside_compilation_cluster:
 207         try:
 208           outside_attr = op.get_attr(_OUTSIDE_COMPILATION_ATTR)
 209           if self._in_gradient_colocation:
 210             raise NotImplementedError(
 211                 "Cannot nest gradient colocation operations outside compilation"
 212             )
 213           if gradient_uid == "__unsupported__":
 214             raise NotImplementedError(
 215                 "No gradient_uid calling gradient within outside_compilation")
 216           # When we take the gradient of an op X in an outside_compilation
 217           # cluster C in a forward computation we would like to put the ops
 218           # corresponding to the gradient of X into a new outside_compilation
 219           # cluster C'. However, if we take the gradient of X twice, the second
 220           # one should get yet another new outside_compilation cluster C''.
 221           #
 222           # The mechanism we adopt is to use a 'root_cluster' which is the
 223           # cluster that X was in before we took gradients, and a 'gradient_uid'
 224           # which is different for every invocation of gradients, and put the
 225           # gradient of X in cluster 'root_cluster.gradient_uid'.
 226           #
 227           # When taking a gradient of a gradient, some ops will be colocated
 228           # with Op in the forward pass (e.g., cluster root_cluster) and some in
 229           # the backward pass (e.g., cluster root_cluster.initial_gradient_uid).
 230           # We need all of the grad-of-grad ops to be in the same cluster to
 231           # avoid cyclic dependencies between clusters. We adopt a heuristic
 232           # that puts any op clustered with root_cluster.<xxx> in
 233           # root_cluster.gradient_uid, even if xxx was initial_gradient_uid.
 234           self._in_gradient_colocation = op
 235           parts = outside_attr.split(".")
 236           cluster = parts[0] + "." + gradient_uid
 237           self._EnterOutsideCompilationScope(cluster=cluster)
 238         except ValueError:
 239           # The attr was not present: do nothing.
 240           pass
 241 
 242   def ExitGradientColocation(self, op, gradient_uid):
 243     if op is not None:
 244       if not self._gradient_colocation_stack:
 245         raise errors.InternalError(
 246             op.node_def, op,
 247             "Badly nested gradient colocation: empty stack when popping Op " +
 248             op.name)
 249       last_op = self._gradient_colocation_stack.pop()
 250       if op is last_op:
 251         if op is self._in_gradient_colocation:
 252           self._in_gradient_colocation = None
 253           self._ExitOutsideCompilationScope()
 254       else:
 255         raise errors.InternalError(
 256             op.node_def, op, "Badly nested gradient colocation, expected " +
 257             last_op + ", got " + op.name)

 259   def _EnterOutsideCompilationScope(self, cluster=None):
 260 
 261     class FakeOp(object):
 262       """A helper class to determine the current device.
 263 
 264       Supports only the type and device set/get methods needed to run the
 265       graph's _apply_device_function method.
 266       """
 267 
 268       def __init__(self):
 269         self._device = ""
 270 
 271       @property
 272       def type(self):
 273         return "FakeOp"
 274 
 275       @property
 276       def device(self):
 277         return self._device
 278 
 279       def _set_device(self, device):
 280         if isinstance(device, pydev.DeviceSpec):
 281           self._device = device.to_string()
 282         else:
 283           self._device = device
 284 
 285     if self._outside_compilation_cluster:
 286       raise NotImplementedError("Cannot nest outside_compilation clusters")
 287     if cluster:
 288       self._outside_compilation_cluster = cluster
 289     else:
 290       self._outside_compilation_cluster = str(self._outside_compilation_counter)
 291       self._outside_compilation_counter += 1
 292     graph = ops.get_default_graph()
 293     fake_op = FakeOp()
 294     graph._apply_device_functions(fake_op)  # pylint: disable=protected-access
 295     device = pydev.DeviceSpec.from_string(fake_op.device)
 296     if (device.device_type == "TPU_REPLICATED_CORE" and
 297         device.device_index is not None):
 298       self._host_compute_core.append(self._outside_compilation_cluster + ":" +
 299                                      str(device.device_index))
 300     self._oc_dev_fn_stack = graph._device_function_stack  # pylint: disable=protected-access
 301     graph._device_function_stack = self._outer_device_function_stack  # pylint: disable=protected-access
 302 
 303   def _ExitOutsideCompilationScope(self):
 304     if not self._outside_compilation_cluster:
 305       raise NotImplementedError(
 306           "Attempted to exit outside_compilation scope when not in scope")
 307     self._outside_compilation_cluster = None
 308     graph = ops.get_default_graph()
 309     graph._device_function_stack = self._oc_dev_fn_stack  # pylint: disable=protected-access
 310 
 311   def Enter(self):
 312     if not self._outer_device_function_stack:
 313       # Capture the device function stack at the time of first entry
 314       # since that is the stack that will be used outside_compilation.
 315       graph = ops.get_default_graph()
 316       # pylint: disable=protected-access
 317       self._outer_device_function_stack = graph._device_function_stack.copy()
 318       # pylint: enable=protected-access
 319     super(TPUReplicateContext, self).Enter()

 321   def HostComputeCore(self):
 322     return self._host_compute_core

 324   def AddOp(self, op):
 325     # pylint: disable=protected-access
 326     if op.type in _BLACKLISTED_OPS:
 327       logging.error("Operation of type %s (%s) is not supported on the TPU. "
 328                     "Execution will fail if this op is used in the graph. " %
 329                     (op.type, op.name))
 330 
 331     if op.type in _UNSUPPORTED_OPS:
 332       self._unsupported_ops.append(op)
 333 
 334     if any(x.dtype._is_ref_dtype for x in op.inputs):
 335       raise NotImplementedError(
 336           "Non-resource Variables are not supported inside TPU computations "
 337           "(operator name: %s)" % op.name)
 338     if _TPU_REPLICATE_ATTR in op.node_def.attr:
 339       raise ValueError("TPU computations cannot be nested")
 340     op._set_attr(_TPU_REPLICATE_ATTR,
 341                  attr_value_pb2.AttrValue(s=self._name_as_bytes))
 342     if self._outside_compilation_cluster:
 343       op._set_attr(
 344           _OUTSIDE_COMPILATION_ATTR,
 345           attr_value_pb2.AttrValue(
 346               s=compat.as_bytes(self._outside_compilation_cluster)))
 347     if self._num_replicas > 1 or not self._outside_compilation_cluster:
 348       # Prevent feeding or fetching anything that is being compiled,
 349       # and any replicated outside_compilation Op.
 350       op.graph.prevent_feeding(op)
 351       op.graph.prevent_fetching(op)
 352 
 353     # Remove any control edges from outer control flow contexts. These may cause
 354     # mismatched frame errors.
 355     (internal_control_inputs,
 356      external_control_inputs) = self._RemoveExternalControlEdges(op)
 357 
 358     if not op.inputs:
 359       # Add a control edge from the control pivot to this op.
 360       if not internal_control_inputs:
 361         # pylint: disable=protected-access
 362         op._add_control_input(self.GetControlPivot())
 363         # pylint: enable=protected-access
 364     else:
 365       for index in xrange(len(op.inputs)):
 366         x = op.inputs[index]
 367         real_x = self.AddValue(x)
 368         if real_x != x:
 369           op._update_input(index, real_x)  # pylint: disable=protected-access
 370 
 371     if external_control_inputs:
 372       # Use an identity to pull control inputs as data inputs. Note that we
 373       # ignore ops which don't have outputs. TODO(phawkins): fix that.
 374       external_control_inputs = [
 375           array_ops.identity(x.outputs[0]).op
 376           for x in external_control_inputs
 377           if x.outputs
 378       ]
 379       # pylint: disable=protected-access
 380       op._add_control_inputs(external_control_inputs)
 381       # pylint: enable=protected-access
 382 
 383     # Mark op's outputs as seen by this context and any outer contexts.
 384     output_names = [x.name for x in op.outputs]
 385     context = self
 386     while context is not None:
 387       # pylint: disable=protected-access
 388       context._values.update(output_names)
 389       context = context._outer_context
 390       # pylint: enable=protected-access
 391 
 392     if self._outer_context:
 393       self._outer_context.AddInnerOp(op)
 394 
 395   def AddValue(self, val):
 396     """Add `val` to the current context and its outer context recursively."""
 397     if val.name in self._values:
 398       # Use the real value if it comes from outer context.
 399       result = self._external_values.get(val.name)
 400       return val if result is None else result
 401 
 402     result = val
 403     self._values.add(val.name)
 404     if self._outer_context:
 405       result = self._outer_context.AddValue(val)
 406       self._values.add(result.name)
 407 
 408     self._external_values[val.name] = result
 409 
 410     return result
 411 
 412   def AddInnerOp(self, op):
 413     self.AddOp(op)
 414     if self._outer_context:
 415       self._outer_context.AddInnerOp(op)
 416

 419     # Define the gradient loop state associated with the TPUReplicateContext to
 420     # be None as the TPUReplicateContext does not get nested nor does the
 421     # grad_state outside the TPUReplicateContext affect the graph inside so the
 422     # grad_state should be as if this is the top-level gradient state.

 436 def outside_compilation(computation, *args, **kwargs):
 437   """Builds part of a computation outside any current TPU replicate scope.
 438 
 439   Args:
 440     computation: A Python function that builds the computation to
 441       place on the host.
 442     *args: the positional arguments for the computation.
 443     **kwargs: the keyword arguments for the computation.
 444 
 445   Returns:
 446     The Tensors returned by computation.
 447   """
 448   args = [] if args is None else args
 449   graph = ops.get_default_graph()
 450 
 451   # If we are in a TPUReplicateContext, signal that we are now
 452   # outside_compilation
 453   initial_context = graph._get_control_flow_context()  # pylint: disable=protected-access
 454   context = initial_context
 455   while context:
 456     if isinstance(context, TPUReplicateContext):
 457       context._EnterOutsideCompilationScope()  # pylint: disable=protected-access
 458     context = context.outer_context
 459 
 460   retval = computation(*args, **kwargs)
 461 
 462   # If we are in a TPUReplicateContext, signal that we are no longer
 463   # outside_compilation
 464   final_context = graph._get_control_flow_context()  # pylint: disable=protected-access
 465   if initial_context is not final_context:
 466     raise NotImplementedError(
 467         "Control-flow context cannot be different at start and end of an "
 468         "outside_compilation scope")
 469   context = initial_context
 470   while context:
 471     if isinstance(context, TPUReplicateContext):
 472       context._ExitOutsideCompilationScope()  # pylint: disable=protected-access
 473     context = context.outer_context
 474 
 475   return retval

 478 def replicate(computation,
 479               inputs=None,
 480               infeed_queue=None,
 481               device_assignment=None,
 482               name=None):
 483   """Builds a graph operator that runs a replicated TPU computation.
 484 
 485   Args:

 487     inputs: A list of lists of input tensors or `None` (equivalent to

 494       the TPU topology. Uses a default device assignment if `None`. The
 495       `DeviceAssignment` may be omitted if each replica of the computation uses
 496       only one core, and there is either only one replica, or the number of
 497       replicas is equal to the number of cores in the TPU system.
 498     name: (Deprecated) Does nothing.

 500     A list of lists of output tensors, indexed by `[replica_num][output_num]`.
 501   Raises:
 502     ValueError: If all replicas do not have equal numbers of input tensors.
 503     ValueError: If the number of inputs per replica does not match
 504       the number of formal parameters to `computation`.
 505   """
 506   return split_compile_and_replicate(computation, inputs, infeed_queue,
 507                                      device_assignment, name)[1]

 510 def split_compile_and_replicate(computation,
 511                                 inputs=None,
 512                                 infeed_queue=None,
 513                                 device_assignment=None,
 514                                 name=None,
 515                                 use_tpu=True):
 516   """Builds graph operators that runs compilation and replicated computation.
 517 
 518   This is a lower level interface than replicate that returns a separate compile
 519   and execute output tensor. In the generated graph the compile op feeds into
 520   the execute op and no additional compilation is incurred when running the
 521   compile op before the execute op. The compile op returns additional
 522   information about the compilation but does not return the compiled program.
 523 
 524   Args:
 525     computation: A Python function that builds the computation to replicate.
 526     inputs: A list of lists of input tensors or `None` (equivalent to
 527       `[[]]`), indexed by `[replica_num][input_num]`. All replicas must
 528       have the same number of inputs.
 529     infeed_queue: If not `None`, the `InfeedQueue` from which to append a tuple
 530       of arguments as inputs to computation.
 531     device_assignment: If not `None`, a `DeviceAssignment` describing the
 532       mapping between logical cores in the computation with physical cores in
 533       the TPU topology. Uses a default device assignment if `None`. The
 534       `DeviceAssignment` may be omitted if each replica of the computation uses
 535       only one core, and there is either only one replica, or the number of
 536       replicas is equal to the number of cores in the TPU system.
 537     name: (Deprecated) Does nothing.
 538     use_tpu: When false, the input `computation` is executed on the XLA CPU/GPU
 539       backends. Currently, only supports a default placement (computation is
 540       placed on GPU if one is available, and on CPU if not).
 541   Returns:
 542     A list of lists with the first list corresponding to the compile op and the
 543     second a list of output tensors, indexed by `[replica_num][output_num]`.
 544   Raises:
 545     ValueError: If all replicas do not have equal numbers of input tensors.
 546     ValueError: If the number of inputs per replica does not match
 547       the number of formal parameters to `computation`.
 548   """
 549   del name
 550   inputs = [[]] if inputs is None else inputs
 551 
 552   metadata_kwargs = {}
 553   if device_assignment is not None:
 554     # Turn the Numpy array into a flattened list so we can pass it as an
 555     # operator attribute.
 556     metadata_kwargs = {
 557         "topology":
 558             device_assignment.topology.serialized(),
 559         "device_assignment":
 560             device_assignment.core_assignment.flatten().tolist()
 561     }
 562     # TODO(phawkins): remove this case after the forward compatibility window
 563     # expires on 2018-10-5.
 564     if api_compat.forward_compatible(2018, 10, 5):
 565       metadata_kwargs["num_cores_per_replica"] = (
 566           device_assignment.num_cores_per_replica)
 567     else:
 568       metadata_kwargs["computation_shape"] = [
 569           device_assignment.num_cores_per_replica
 570       ]
 571 
 572   if ((not isinstance(inputs, list)) or
 573       any(not isinstance(inp, (list, tuple)) for inp in inputs)):
 574     raise TypeError("tpu.replicate() inputs must be a list of lists/tuples")
 575 
 576   num_replicas = len(inputs)
 577 
 578   # No replicas? Nothing to do.
 579   if num_replicas == 0:
 580     return []
 581 
 582   # Converts inputs to Tensors.
 583   inputs = [[ops.convert_to_tensor(x) for x in inp] for inp in inputs]
 584 
 585   # Verifies that all replicas have matching numbers and types of inputs
 586   input_types = [x.dtype for x in inputs[0]]
 587   input_arity = len(input_types)
 588   for i in range(num_replicas):
 589     if len(inputs[i]) != input_arity:
 590       raise ValueError("Replicas must have the same number of inputs. "
 591                        "Replica 0 had {} inputs, replica {} had {} "
 592                        "inputs.".format(input_arity, i, len(inputs[i])))
 593 
 594     types = [x.dtype for x in inputs[i]]
 595     if types != input_types:
 596       raise ValueError(
 597           "Replicas must have matching input types. Replica 0 had "
 598           "input types {}, replica {} had input types {}".format(
 599               input_types, i, types))
 600 
 601   arg_error = tpu_function.check_function_argument_count(
 602       computation, input_arity, infeed_queue)
 603   if arg_error is not None:
 604     if infeed_queue is None:
 605       raise TypeError(
 606           "Supplied computation cannot be called with the specified inputs. "
 607           "You specified %d inputs: %s, but the computation needs %s" % (
 608               input_arity, str([i.name for i in inputs[0]]), arg_error))
 609     else:
 610       raise TypeError(
 611           "Supplied computation cannot be called with the specified inputs. "
 612           "You specified %d inputs: %s and %d additional inputs from infeed,"
 613           " but the computation needs %s" % (input_arity, str(
 614               [i.name
 615                for i in inputs[0]]), infeed_queue.number_of_tuple_elements,
 616                                              arg_error))
 617 
 618   graph = ops.get_default_graph()
 619 
 620   # Fan-in: Builds a TPUReplicatedInput node for each input.
 621   computation_inputs = []
 622   for i in range(0, input_arity):
 623     replicas = [inputs[replica][i] for replica in xrange(num_replicas)]
 624     computation_inputs.append(
 625         tpu_ops.tpu_replicated_input(replicas, name="input{}".format(i)))
 626 
 627   cluster_name = graph.unique_name("cluster")
 628   pivot = control_flow_ops.no_op(name=cluster_name + "/pivot")
 629   context = TPUReplicateContext(
 630       name=cluster_name, num_replicas=num_replicas, pivot=pivot)
 631   try:
 632     context.Enter()
 633 
 634     metadata = tpu_ops.tpu_replicate_metadata(
 635         num_replicas=num_replicas, use_tpu=use_tpu, **metadata_kwargs)
 636 
 637     with tpu_function.tpu_shard_context(
 638         num_replicas), ops.control_dependencies([metadata]):
 639 
 640       # Add identity ops so even unused inputs are "consumed" by the
 641       # computation. This is to avoid orphaned TPUReplicatedInput nodes.
 642       # TODO(phawkins): consider instead pruning unused TPUReplicatedInput
 643       # and eliding trivial TPUReplicatedInput/TPUReplicatedOutput pairs.
 644       computation_inputs = [
 645           array_ops.identity(x, name="replicated_input_{}".format(i))
 646           for i, x in enumerate(computation_inputs)
 647       ]
 648 
 649       # If there is an infeed queue, adds the dequeued values to the
 650       # computation's inputs.
 651       if infeed_queue is not None:
 652         infeed_queue.set_number_of_shards(num_replicas)
 653         for t in infeed_queue.generate_dequeue_op():
 654           computation_inputs.append(t)
 655 
 656       # Only resource variables work inside a TPU computation, so turn on
 657       # resource variables for the computation.
 658       # TODO(phawkins): consider removing this code. It will
 659       # be less confusing to clients if they knowingly choose to use resource
 660       # variables.
 661       # Partitioned variables is not supported (b/112311320).
 662       vscope = variable_scope.get_variable_scope()
 663       saved_use_resource = vscope.use_resource
 664       saved_custom_getter = vscope.custom_getter
 665 
 666       def custom_getter(getter, name, *args, **kwargs):
 667         """Variables on TPU have a few restrictions."""
 668         partitioner = kwargs["partitioner"]
 669         if partitioner is not None:
 670           kwargs["partitioner"] = None
 671           logging.warning(
 672               "Partitioned variables are not supported on TPU. Got "
 673               "`partitioner` that is {} for variable {}. "
 674               "Setting `partitioner` to `None`."
 675               .format(partitioner, name))
 676         if saved_custom_getter is None:
 677           return getter(name, *args, **kwargs)
 678         else:
 679           return saved_custom_getter(getter, name, *args, **kwargs)
 680 
 681       vscope.set_use_resource(True)
 682       vscope.set_custom_getter(custom_getter)
 683 
 684       outputs = computation(*computation_inputs)
 685 
 686       vscope.set_use_resource(saved_use_resource)
 687       vscope.set_custom_getter(saved_custom_getter)
 688 
 689     # If the computation returns `None`, make it an empty tuple.
 690     if outputs is None:
 691       outputs = tuple()
 692     # If the computation only returned one value, makes it a tuple.
 693     if not isinstance(outputs, (list, tuple)):
 694       outputs = (outputs,)
 695 
 696     # Append `no_op` here so that fetching any return value of this function
 697     # will trigger TPUExecute node.
 698     outputs += (control_flow_ops.no_op(),)
 699     try:
 700       with ops.device(core(0)):
 701         outputs = [
 702             o if isinstance(o, ops.Operation) else ops.convert_to_tensor(o)
 703             for o in outputs
 704         ]
 705     except Exception as e:
 706       raise ValueError(
 707           "TPU function return values must all either be Operations or "
 708           "convertible to Tensors. Got '%s'" % str(e))
 709 
 710     # Separates the returned Operations and Tensors.
 711     output_operations = [o for o in outputs if isinstance(o, ops.Operation)]
 712     output_tensors = [o for o in outputs if not isinstance(o, ops.Operation)]
 713 
 714     if outputs != output_tensors + output_operations:
 715       raise ValueError(
 716           "TPU functions must return zero-or more Tensor values followed by "
 717           "zero or more Operations.")
 718     output_arity = len(output_tensors)
 719 
 720     # Wraps outputs in Identity ops. Otherwise a replicated input copied
 721     # straight to an output would bypass the replicate(). This would be bad
 722     # because the TPUReplicatedInput/TPUReplicatedOutput operator would not
 723     # be rewritten away, leading to a runtime error.
 724     # TODO(phawkins): extend the rewrite to elide these nodes instead.
 725     new_output_tensors = []
 726     for t in output_tensors:
 727       with ops.device(t.device if t.device else core(0)):
 728         new_output_tensors.append(array_ops.identity(t))
 729     output_tensors = new_output_tensors
 730     context.ExitResult(output_tensors)
 731   finally:
 732     context.report_unsupported_operations()
 733     context.Exit()
 734     host_compute_core = context.HostComputeCore()
 735 
 736   if host_compute_core:
 737     attr_value = attr_value_pb2.AttrValue()
 738     attr_value.list.s.extend([compat.as_bytes(x) for x in host_compute_core])
 739     metadata._set_attr("host_compute_core", attr_value)  # pylint: disable=protected-access
 740 
 741   # Fan-out: Builds a TPUReplicatedOutput node for each output.
 742   outputs = [tpu_ops.tpu_replicated_output(output_tensors[i], num_replicas,
 743                                            name="output{}".format(i))
 744              for i in xrange(output_arity)]
 745 
 746   with ops.control_dependencies([metadata]):
 747     if use_tpu:
 748       compile_status = tpu_ops.tpu_compilation_result()
 749       op = compile_status.op
 750       attr_value = attr_value_pb2.AttrValue(s=compat.as_bytes(cluster_name))
 751       op._set_attr(_TPU_COMPILATION_STATUS_ATTR, attr_value)  # pylint: disable=protected-access
 752     else:
 753       compile_status = control_flow_ops.no_op(name="compilation_status")
 754 
 755   with ops.control_dependencies(output_operations):
 756     if output_arity == 0:
 757       # Returns a list of NoOps dependent on the replication Op, indexed by
 758       # [replica_num].
 759       return [
 760           compile_status, [
 761               control_flow_ops.no_op(name="shard_%d" % i)
 762               for i in range(num_replicas)
 763           ]
 764       ]
 765     else:
 766       # Wraps the outputs in identity operators so the names of any possible
 767       # `fetch` nodes are preserved by the replication rewrite.
 768       return [
 769           compile_status, [[
 770               array_ops.identity(
 771                   outputs[out][replica],
 772                   name="output_%d_shard_%d" % (out, replica))
 773               for out in xrange(output_arity)
 774           ]
 775                            for replica in xrange(num_replicas)]
 776       ]

 779 def shard(computation,
 780           inputs=None,
 781           num_shards=1,
 782           input_shard_axes=None,
 783           outputs_from_all_shards=True,
 784           output_shard_axes=None,
 785           infeed_queue=None,
 786           device_assignment=None,
 787           name=None):
 788   """Shards `computation` for parallel execution.
 789 
 790   `inputs` must be a list of Tensors or None (equivalent to an empty list), each
 791   of which has a corresponding split axis (from `input_shard_axes`). Each input
 792   is split into `num_shards` pieces along the corresponding axis, and
 793   computation is applied to each shard in parallel.
 794 
 795   Tensors are broadcast to all shards if they are lexically captured by
 796   `computation`. e.g.,
 797 
 798   x = tf.constant(7)
 799   def computation():
 800     return x + 3
 801   ... = shard(computation, ...)
 802 
 803   TODO(phawkins): consider adding support for broadcasting Tensors passed
 804   as inputs.
 805 
 806   If `outputs_from_all_shards` is true, the outputs from all shards of
 807   `computation` are concatenated back together along their `output_shards_axes`.
 808   Otherwise, each output is taken from an arbitrary shard.
 809 
 810   Inputs and outputs of the computation must be at least rank-1 Tensors.
 811 
 812   Args:
 813     computation: A Python function that builds a computation to apply to each
 814       shard of the input.
 815     inputs: A list of input tensors or None (equivalent to an empty list). Each
 816       input tensor has a corresponding shard axes, given by `input_shard_axes`,
 817       which must have size divisible by `num_shards`.
 818     num_shards: The number of shards.
 819     input_shard_axes: A list of dimensions along which to shard `inputs`, or
 820       `None`. `None` means "shard all inputs along dimension 0". If not `None`,
 821       there must be one dimension per input.
 822     outputs_from_all_shards: Boolean or list of boolean. For each output, if
 823       `True`, outputs from all shards are concatenated along the corresponding
 824       `output_shard_axes` entry. Otherwise, each output is taken
 825       from an arbitrary shard. If the argument is a boolean, the argument's
 826       value is used for each output.
 827     output_shard_axes: A list of dimensions along which to concatenate the
 828       outputs of `computation`, or `None`. `None` means "concatenate all outputs
 829       along dimension 0". If not `None`, there must be one dimension per output.
 830       Ignored if `outputs_from_all_shards` is False.
 831     infeed_queue: If not `None`, the `InfeedQueue` to use to augment the inputs
 832       of `computation`.
 833     device_assignment: If not `None`, a `DeviceAssignment` describing the
 834       mapping between logical cores in the computation with physical cores in
 835       the TPU topology. Uses a default device assignment if `None`. The
 836       `DeviceAssignment` may be omitted if each shard of the computation uses
 837       only one core, and there is either only one shard, or the number of shards
 838       is equal to the number of cores in the TPU system.
 839     name: (Deprecated) Does nothing.
 840   Returns:
 841     A list of output tensors.
 842   Raises:
 843     ValueError: If num_shards <= 0
 844     ValueError: If len(input_shard_axes) != len(inputs)
 845     ValueError: If len(output_shard_axes) != len(outputs from `computation`)
 846   """
 847 
 848   if num_shards <= 0:
 849     raise ValueError("num_shards must be a positive integer.")
 850 
 851   inputs = [] if inputs is None else inputs
 852   if not isinstance(inputs, list):
 853     raise TypeError("tpu.shard()'s inputs must be a list of Tensors or None.")
 854 
 855   # Converts inputs to Tensors.
 856   inputs = [ops.convert_to_tensor(x) for x in inputs]
 857 
 858   if input_shard_axes is None:
 859     input_shard_axes = [0] * len(inputs)
 860   if len(inputs) != len(input_shard_axes):
 861     raise ValueError("Length of input_shard_axes must be equal to the number "
 862                      "of inputs.")
 863 
 864   if inputs:
 865     # Splits the `inputs` along the corresponding `input_shard_axes`, giving
 866     # lists with layout [input][shard]
 867     split_inputs = [
 868         array_ops.split(x, num_shards, axis=axis)
 869         for (axis, x) in zip(input_shard_axes, inputs)]
 870 
 871     # Transposes the input lists to have layout [shard][input]
 872     transposed_inputs = [list(i) for i in zip(*split_inputs)]
 873   else:
 874     transposed_inputs = [[]] * num_shards
 875 
 876   outputs = replicate(
 877       computation,
 878       transposed_inputs,
 879       infeed_queue=infeed_queue,
 880       device_assignment=device_assignment,
 881       name=name)
 882 
 883   # There must be at least one shard since num_shards > 0.
 884   # TODO(b/36647078) remove disable when pylint bug is fixed.
 885   # pylint: disable=indexing-exception
 886   if isinstance(outputs[0], ops.Operation):
 887     # pylint: enable=indexing-exception
 888     # There were no outputs from the computation and replicate returned a list
 889     # of NoOps with control dependencies on the computation. Return the first
 890     # one so it can be used as a control dependency or fetch node.
 891     # TODO(b/36647078) remove disable when pylint bug is fixed.
 892     # pylint: disable=indexing-exception
 893     return [outputs[0]]
 894     # pylint: enable=indexing-exception
 895 
 896   # TODO(b/36647078) remove disable when pylint bug is fixed.
 897   # pylint: disable=indexing-exception
 898   num_outputs = len(outputs[0])
 899   # pylint: enable=indexing-exception
 900 
 901   if output_shard_axes is None:
 902     output_shard_axes = [0] * num_outputs
 903   if num_outputs != len(output_shard_axes):
 904     raise ValueError("Length of output_shard_axes must be equal to the number "
 905                      "of outputs.")
 906 
 907   if isinstance(outputs_from_all_shards, bool):
 908     outputs_from_all_shards = [outputs_from_all_shards] * num_outputs
 909 
 910   if num_outputs != len(outputs_from_all_shards):
 911     raise ValueError("Length of outputs_from_all_shards must be equal to the "
 912                      "number of outputs.")
 913 
 914   results = []
 915   for (axis, all_shards, x) in zip(output_shard_axes, outputs_from_all_shards,
 916                                    zip(*outputs)):
 917     if all_shards:
 918       # Concatenate all of the outputs together (use stack for scalars).
 919       shape = x[0].shape
 920       is_scalar = shape is not None and (shape.ndims == 0)
 921       results.append((array_ops.stack(list(x)) if is_scalar
 922                       else array_ops.concat(list(x), axis=axis)))
 923     else:
 924       # TODO(phawkins): use a smarter policy, e.g., round-robin across shards.
 925       results.append(x[0])
 926 
 927   return results

 930 def batch_parallel(computation,
 931                    inputs=None,
 932                    num_shards=1,
 933                    infeed_queue=None,
 934                    device_assignment=None,
 935                    name=None):
 936   """Shards `computation` along the batch dimension for parallel execution.
 937 
 938   Convenience wrapper around shard().
 939 
 940   `inputs` must be a list of Tensors or None (equivalent to an empty list).
 941   Each input is split into `num_shards` pieces along the 0-th dimension, and
 942   computation is applied to each shard in parallel.
 943 
 944   Tensors are broadcast to all shards if they are lexically captured by
 945   `computation`. e.g.,
 946 
 947   x = tf.constant(7)
 948   def computation():
 949     return x + 3
 950   ... = shard(computation, ...)
 951 
 952   The outputs from all shards are concatenated back together along their 0-th
 953   dimension.
 954 
 955   Inputs and outputs of the computation must be at least rank-1 Tensors.
 956 
 957   Args:
 958     computation: A Python function that builds a computation to apply to each
 959       shard of the input.
 960     inputs: A list of input tensors or None (equivalent to an empty list). The
 961       0-th dimension of each Tensor must have size divisible by `num_shards`.
 962     num_shards: The number of shards.
 963     infeed_queue: If not `None`, the `InfeedQueue` from which to append a tuple
 964       of arguments as inputs to `computation`.
 965     device_assignment: If not `None`, a `DeviceAssignment` describing the
 966       mapping between logical cores in the computation with physical cores in
 967       the TPU topology. Uses a default device assignment if `None`. The
 968       `DeviceAssignment` may be omitted if each shard of the computation uses
 969       only one core, and there is either only one shard, or the number of shards
 970       is equal to the number of cores in the TPU system.
 971     name: (Deprecated) Does nothing.
 972   Returns:
 973     A list of output tensors.
 974   Raises:
 975     ValueError: If `num_shards <= 0`
 976   """
 977   return shard(
 978       computation,
 979       inputs,
 980       num_shards=num_shards,
 981       infeed_queue=infeed_queue,
 982       device_assignment=device_assignment,
 983       name=name)

 986 def rewrite(computation,
 987             inputs=None,
 988             infeed_queue=None,
 989             device_assignment=None,
 990             name=None):
 991   """Rewrites `computation` for execution on a TPU system.
 992 
 993   Args:
 994     computation: A Python function that builds a computation to apply to the
 995       input. If the function takes n inputs, 'inputs' should be a list of n
 996       tensors.
 997 
 998       `computation` may return a list of operations and tensors. Tensors must
 999       come before operations in the returned list.  The return value of
1000       `rewrite` is a list of tensors corresponding to the tensors from the
1001       output of `computation`.
1002 
1003       All `Operation`s returned from `computation` will be executed when
1004       evaluating any of the returned output tensors.
1005     inputs: A list of input tensors or `None` (equivalent to an empty list).
1006     infeed_queue: If not `None`, the `InfeedQueue` from which to append a tuple
1007       of arguments as inputs to `computation`.
1008     device_assignment: if not `None`, a `DeviceAssignment` describing the
1009       mapping between logical cores in the computation with physical cores in
1010       the TPU topology. May be omitted for a single-core computation, in which
1011       case the core attached to task 0, TPU device 0 is used.
1012     name: (Deprecated) Does nothing.
1013   Returns:
1014     A list of output tensors.
1015   """
1016   if inputs is not None and not isinstance(inputs, (list, tuple)):
1017     raise TypeError("tpu.rewrite() inputs must be a list or tuple")
1018 
1019   # TODO(b/36647078) remove disable when pylint bug is fixed.
1020   # pylint: disable=indexing-exception
1021   return replicate(
1022       computation,
1023       None if inputs is None else [inputs],
1024       infeed_queue=infeed_queue,
1025       device_assignment=device_assignment,
1026       name=name)[0]
1027   # pylint: enable=indexing-exception
1028 
1029   # Operations that indicate some error in the user's inference graph.
1030 _BLACKLISTED_INFERENCE_OPS = set([
1031     "ReadVariableOp",
1032     "AssignVariableOp",
1033     "AssignAddVariableOp",
1034     "AssignSubVariableOp",
1035     "VarHandleOp",
1036     "Variable",
1037     "VariableV2",
1038 ])

1041 def under_tpu_inference_context():
1042   """Check if it is currently under `tpu.rewrite_for_inference()`."""
1043   graph = ops.get_default_graph()
1044 
1045   context = graph._get_control_flow_context()  # pylint: disable=protected-access
1046   while context:
1047     if isinstance(context, _TPUInferenceContext):
1048       return True
1049     context = context.outer_context
1050 
1051   return False

1054 class _TPUInferenceContext(control_flow_ops.XLAControlFlowContext):
1055   """A `ControlFlowContext` for nodes inside a TPU inference computation.
1056 
1057   The primary role of `TPUReplicateContext` is to sanity check operators inside
1058   a tpu.rewrite_for_inference() computation.
1059   """
1060 
1061   def __init__(self, name):
1062     super(_TPUInferenceContext, self).__init__()
1063     self._name = name
1064 
1065   def AddOp(self, op):
1066     self._AddOpInternal(op)
1067 
1068   def _AddOpInternal(self, op):
1069     # pylint: disable=protected-access
1070     if op.type in _BLACKLISTED_INFERENCE_OPS:
1071       raise NotImplementedError(
1072           "Operation of type %s (%s) is not supported on the TPU for inference."
1073           " Execution will fail if this op is used in the graph. Make sure your"
1074           " variables are using variable_scope." % (op.type, op.name))
1075     if self._outer_context:
1076       self._outer_context.AddInnerOp(op)
1077 
1078   def AddValue(self, val):
1079     result = val
1080     if self._outer_context:
1081       result = self._outer_context.AddValue(val)
1082     return result
1083 
1084   def AddInnerOp(self, op):
1085     self._AddOpInternal(op)
1086 
1087   @property
1088   def grad_state(self):
1089     return None

1121 @experimental
1122 def rewrite_for_inference(computation,
1123                           inputs=None,
1124                           infeed_queue=None,
1125                           device_assignment=None,
1126                           name=None):
1127   """Rewrites `computation` for inference on a TPU system.
1128 
1129      Other than 'rewriting' the computation to run on a TPU, if using variables
1130      in your computation, it moves the ReadVariableOps outside the TPU
1131      computation, and adds GuaranteeConst ops just after the ReadVariableOps.
1132      This mechanism works only if you are using tf.get_variable() to create and
1133      access variables in your tpu computation. You can validate whether this
1134      worked, by calling validate_inference_rewrite_for_variables() method
1135      immediately after this method to check whether GuaranteeConstOps where
1136      added to the graph.
1137 
1138   Args:
1139     computation: A Python function that builds a computation to apply to the
1140       input. If the function takes n inputs, 'inputs' should be a list of n
1141       tensors. If the function returns m outputs, rewrite will return a list of
1142       m tensors.
1143     inputs: A list of input tensors or `None` (equivalent to an empty list).
1144     infeed_queue: If not `None`, the `InfeedQueue` from which to append a tuple
1145       of arguments as inputs to `computation`.
1146     device_assignment: if not `None`, a `DeviceAssignment` describing the
1147       mapping between logical cores in the computation with physical cores in
1148       the TPU topology. May be omitted for a single-core computation, in which
1149       case the core attached to task 0, TPU device 0 is used.
1150     name: The name of the operator.
1151   Returns:
1152     A list of output tensors.
1153   """
1154 
1155   def guarantee_const_getter(getter, name, *args, **kwargs):
1156     with ops.control_dependencies(None):
1157       return array_ops.guarantee_const(
1158           getter(name, *args, **kwargs), name=name + "/GuaranteeConst")
1159 
1160   def wrapped_computation(*args, **kwargs):
