  15 """Defines the `Topology` class, that describes a TPU fabric topology."""
  16 
  17 from __future__ import absolute_import
  18 from __future__ import division
  19 from __future__ import print_function
  20 
  21 import numpy as np
  22 
  23 from tensorflow.contrib.tpu.proto import topology_pb2
  24 
  25 
  26 class Topology(object):
  27   """Describes a set of TPU devices.
  28 
  29   Represents both the shape of the physical mesh, and the mapping between
  30   TensorFlow TPU devices to physical mesh coordinates.
  31   """
  32 
  33   def __init__(self, serialized=None, mesh_shape=None, device_coordinates=None):
  34     """Builds a Topology object.
  35 
  36     If `serialized` is not `None`, the topology is parsed from `serialized` and
  37     the other arguments are ignored. Otherwise, the topology is computed from
  38     `mesh_shape` and `device_coordinates`.
  39 
  40     Args:
  41       serialized: A serialized `TopologyProto`, or `None`. If not `None`, the
  42         serialized proto is parsed to discover the topology.
  43       mesh_shape: A sequence of 3 positive integers, or `None`. If not `None`,
  44         the shape of the TPU topology, in number of cores. Ignored if
  45         `serialized` is not `None`.
  46       device_coordinates: A rank 3 numpy array that describes the mapping from
  47         TensorFlow TPU devices to TPU fabric coordinates, or `None`. Ignored
  48         if `serialized is not `None`.
  49 
  50     Raises:
  51       ValueError: If `serialized` does not describe a well-formed topology.
  52       ValueError: If `serialized` is `None` and `mesh_shape` is not a sequence
  53         of 3 positive integers.
  54       ValueError: If `serialized` is `None` and `device_coordinates` is not a
  55         rank 3 numpy int32 array that describes a valid coordinate mapping.
  56     """
  57 
  58     self._serialized = serialized
  59 
  60     if serialized:
  61       self._parse_topology(serialized)
  62     else:
  63       self._mesh_shape = np.asarray(mesh_shape, dtype=np.int32)
  64       self._device_coordinates = np.asarray(device_coordinates, np.int32)
  65       if len(self._mesh_shape) != 3 or any(self._mesh_shape < 1):
  66         raise ValueError("`mesh_shape` must be a sequence of 3 positive "
  67                          "entries; got {}".format(self._mesh_shape))
  68 
  69       if (len(self._device_coordinates.shape) != 3 or
  70           self._device_coordinates.shape[2] != len(self._mesh_shape)):
  71         raise ValueError("`device_coordinates` must be a rank 3 int32 array "
  72                          "with minor dimension equal to the mesh shape rank")
  73 
  74   def _parse_topology(self, serialized):
  75     """Parses a serialized `TopologyProto` into `self`."""
  76     proto = topology_pb2.TopologyProto()
  77     proto.ParseFromString(serialized)
  78 
  79     self._mesh_shape = np.array(proto.mesh_shape, dtype=np.int32)
  80     if len(self._mesh_shape) != 3 or any(self._mesh_shape < 1):
  81       raise ValueError("`mesh_shape` must be a vector of size 3 with positive "
  82                        "entries; got {}".format(self._mesh_shape))
  83 
  84     if proto.num_tasks < 0:
  85       raise ValueError("`num_tasks` must be >= 0; got {}".format(
  86           proto.num_tasks))
  87     if proto.num_tpu_devices_per_task < 0:
  88       raise ValueError("`num_tpu_devices_per_task` must be >= 0; got {}".format(
  89           proto.num_tpu_devices_per_task))
  90 
  91     expected_coordinates_size = (
  92         proto.num_tasks * proto.num_tpu_devices_per_task * len(
  93             proto.mesh_shape))
  94     if len(proto.device_coordinates) != expected_coordinates_size:
  95       raise ValueError("`device_coordinates` must have shape num_tasks ({}) * "
  96                        "num_tpu_devices_per_task ({}) * len(mesh_shape) ({}); "
  97                        "got shape {}".format(proto.num_tasks,
  98                                              proto.num_tpu_devices_per_task,
  99                                              proto.mesh_shape,
 100                                              len(proto.device_coordinates)))
 101 
 102     coords = np.array(proto.device_coordinates, dtype=np.int32)
 103     if any(coords < 0):
 104       raise ValueError("`device_coordinates` must be >= 0")
 105     coords = coords.reshape((proto.num_tasks, proto.num_tpu_devices_per_task,
 106                              len(proto.mesh_shape)))
 107     self._device_coordinates = coords
 108 
 109   @property
 110   def mesh_shape(self):
 111     """A rank 1 int32 array describing the shape of the TPU topology."""
 112     return self._mesh_shape
 113 
 114   @property
 115   def mesh_rank(self):
 116     """Returns the number of dimensions in the mesh."""
 117     return len(self._mesh_shape)
 118 
 119   @property
 120   def device_coordinates(self):
 121     """Describes the mapping from TPU devices to topology coordinates.
 122 
 123     Returns:
 124       A rank 3 int32 array with shape `[tasks, devices, axis]`.
 125       `tasks` is the number of tasks in the TPU cluster, `devices` is the number
 126       of TPU devices per task, and `axis` is the number of axes in the TPU
 127       cluster topology. Each entry gives the `axis`-th coordinate in the
 128       topology of a task/device pair. TPU topologies are 3-dimensional, with
 129       dimensions `(x, y, core number)`.
 130     """
 131     return self._device_coordinates
 132 
 133   @property
 134   def num_tasks(self):
 135     """Returns the number of TensorFlow tasks in the TPU slice."""
 136     return self._device_coordinates.shape[0]
 137 
 138   @property
 139   def num_tpus_per_task(self):
 140     """Returns the number of TPU devices per task in the TPU slice."""
 141     return self._device_coordinates.shape[1]
 142 
 143   def serialized(self):
 144     """Returns the serialized form of the topology."""
 145     if self._serialized is None:
 146       proto = topology_pb2.TopologyProto()
 147       proto.mesh_shape[:] = list(self._mesh_shape)
 148       proto.num_tasks = self._device_coordinates.shape[0]
 149       proto.num_tpu_devices_per_task = self._device_coordinates.shape[1]
 150       proto.device_coordinates.extend(list(self._device_coordinates.flatten()))
 151       self._serialized = proto.SerializeToString()
 152 
 153     return self._serialized
