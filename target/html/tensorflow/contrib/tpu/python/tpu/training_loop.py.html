  29 def while_loop(condition, body, inputs=None, infeed_queue=None, name=None):
  30   """Builds a training loop for TPUs.
  31 
  32   The set of loop-carried tensors corresponds to `inputs`.  Both
  33   `condition` and `body` take the current value of the loop-carried
  34   tensors. 'body' additionally takes a tuple of infeed from
  35   infeed_queue if infeed_queue is not None. `condition` must return a
  36   single boolean value that determines whether iteration
  37   continues. `body` must return an updated list of values for the
  38   loop-carried tensors.
  39 
  40   Args:
  41     condition: a Python function that builds the loop condition.
  42     body: a Python function that builds the loop body.
  43     inputs: a list of initial values passed into the training loop, or
  44       None (equivalent to an empty list).
  45     infeed_queue: if not None, the infeed queue from which to append a tuple
  46       of arguments as inputs to condition.
  47     name: (Deprecated) Does nothing.
  48 
  49   Returns:
  50     The final values of the loop-carried tensors.
  51 
  52   Raises:
  53     TypeError: if body or condition has the wrong signature.
  54   """
  55   del name
  56   # Converts inputs to Tensors.
  57   inputs = [] if inputs is None else [ops.convert_to_tensor(x) for
  58                                       x in inputs]
  59   input_types = [x.dtype for x in inputs]
  60   input_arity = len(inputs)
  61 
  62   body_arg_error = tpu_function.check_function_argument_count(
  63       body, input_arity, infeed_queue)
  64   if body_arg_error is not None:
  65     if infeed_queue is None:
  66       raise TypeError(
  67           "Supplied loop body function cannot be called with the specified "
  68           "inputs. You specified %d inputs: %s, but the loop body needs %s" % (
  69               input_arity, str([i.name for i in inputs]), body_arg_error))
  70     else:
  71       raise TypeError(
  72           "Supplied loop body function cannot be called with the specified "
  73           "inputs. You specified %d inputs: %s and %d additional inputs from "
  74           "infeed, but the computation needs %s" % (input_arity, str(
  75               [i.name for i in inputs]), infeed_queue.number_of_tuple_elements,
  76                                                     body_arg_error))
  77   condition_arg_error = tpu_function.check_function_argument_count(
  78       condition, input_arity, None)
  79   if condition_arg_error is not None:
  80     if infeed_queue is None:
  81       raise TypeError(
  82           "Supplied loop condition function cannot be called with the "
  83           "specified inputs. You specified %d inputs: %s, but the loop "
  84           "condition needs %s" % (input_arity, str([i.name for i in inputs]),
  85                                   condition_arg_error))
  86     else:
  87       raise TypeError(
  88           "Supplied loop condition function cannot be called with the "
  89           "specified inputs. You specified %d inputs: %s, but the loop "
  90           "condition needs %s. Note that infeed is not passed to the loop "
  91           "condition." % (input_arity, str([i.name for i in inputs]),
  92                           condition_arg_error))
  93 
  94   def condition_wrapper(*inputs):
  95     # Discards the dummy output added for arity-0 loops.
  96     if input_arity == 0:
  97       inputs = []
  98     return condition(*inputs)
  99 
 100   def body_wrapper(*inputs):
 101     """Wrapper around `body` that handles infeed queues and control deps."""
 102     inputs = list(inputs)
 103 
 104     # Discards the dummy output added for arity-0 loops.
 105     if input_arity == 0:
 106       inputs = []
 107 
 108     # Runs `body` with the dequeue_ops appended.
 109     if infeed_queue:
 110       number_of_shards = tpu_function.get_tpu_context().number_of_shards
 111       if number_of_shards is None:
 112         raise ValueError("Can't build training loop with infeed when there is "
 113                          "no tpu_shard_context. Are you building a loop or "
 114                          "graph directly rather than from inside tpu.rewrite, "
 115                          "tpu.batch_parallel, tpu.shard, or tpu.replicate?")
 116       infeed_queue.set_number_of_shards(number_of_shards)
 117       dequeue_ops = [d for d in infeed_queue.generate_dequeue_op()]
 118     else:
 119       dequeue_ops = []
 120     outputs = body(*(inputs + dequeue_ops))
 121 
 122     # If the computation only returned one value, make it a tuple.
 123     if not isinstance(outputs, (list, tuple)):
 124       outputs = (outputs,)
 125 
 126     outputs = [
 127         o if isinstance(o, ops.Operation) else ops.convert_to_tensor(o)
 128         for o in outputs
 129     ]
 130 
 131     # Separates the returned Operations and Tensors.
 132     output_operations = [o for o in outputs if isinstance(o, ops.Operation)]
 133     output_tensors = [o for o in outputs
 134                       if not isinstance(o, ops.Operation)]
 135 
 136     if outputs != output_tensors + output_operations:
 137       raise ValueError(
 138           "TPU training loop body must return zero or more Tensor values "
 139           "followed by zero or more Operations.")
 140 
 141     output_types = [op.dtype for op in output_tensors]
 142     if input_types != output_types:
 143       raise TypeError(
 144           "Mismatch between input types and output types for training loop "
 145           "body: {} vs {}".format(input_types, output_types))
 146 
 147     # Add the dequeue operations to output_operations to ensure they are run
 148     # by the loop, even if the programmer's loop body does not use them.
 149     output_operations += dequeue_ops
 150 
 151     # Add a dummy output, if needed.
 152     if not output_tensors:
 153       output_tensors = array_ops.constant(0)
 154 
 155     if output_operations:
 156       # TODO(phawkins): in principle this is too restrictive since it serializes
 157       # the training loop steps. In practice it does not matter since this loop
 158       # will be compiled by XLA.
 159       return control_flow_ops.tuple(output_tensors,
 160                                     control_inputs=output_operations)
 161     else:
 162       return output_tensors
 163 
 164   # If the body has arity 0, add a dummy loop-carried value to which we can add
 165   # control dependencies from any side-effecting operations.
 166   if input_arity == 0:
 167     inputs = [array_ops.constant(0)]
 168   return control_flow_ops.while_loop(condition_wrapper, body_wrapper, inputs,
 169                                      name="")

 172 def repeat(n, body, inputs=None, infeed_queue=None, name=None):
 173   """Builds a training loop that executes a fixed number of iterations.
 174 
 175   The set of loop-carried tensors correspond to `inputs`.
 176   `body` must be a function that takes and returns the values of the
 177   loop-carried tensors.
 178 
 179   Args:
 180     n: the number of loop iterations
 181     body: a Python function that builds the loop body.
 182     inputs: a list of initial values passed into the training loop or
 183       None (equivalent to an empty list).
 184     infeed_queue: if not None, the infeed queue from which to append a tuple
 185       of arguments as inputs to condition.
 186     name: (Deprecated) Does nothing.
 187   Returns:
 188     The final values of the loop-carried tensors.
 189   Raises:
 190     ValueError: if there is a type error.
 191   """
 192   def _convert_to_list(xs):
 193     if not isinstance(xs, (list, tuple)):
 194       return [xs]
 195     else:
 196       return list(xs)
 197 
 198   def cond(i, *args):
 199     del args
 200     return i < n
 201 
 202   def body_wrapper(i, *args):
 203     return [i + 1] + _convert_to_list(body(*args))
 204 
 205   inputs = [0] if inputs is None else [0] + _convert_to_list(inputs)
 206   outputs = while_loop(
 207       cond, body_wrapper, inputs=inputs, infeed_queue=infeed_queue, name=name)
 208   outputs = _convert_to_list(outputs)
 209   if len(outputs) == 1:
 210     # Returns the Op rather than an empty list.
 211     return outputs[0].op
 212   else:
 213     return outputs[1:]
