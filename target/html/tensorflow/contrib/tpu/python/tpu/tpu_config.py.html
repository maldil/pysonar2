  38 class InputPipelineConfig(object):
  39   r"""Please see the definition of these values in TPUConfig."""
  40   PER_SHARD_V1 = 1
  41   PER_HOST_V1 = 2
  42   PER_HOST_V2 = 3
  43   BROADCAST = 4
  44 
  45 
  46 class TPUConfig(
  47     collections.namedtuple('TPUConfig', [
  48         'iterations_per_loop',
  49         'num_shards',
  50         'num_cores_per_replica',
  51         'per_host_input_for_training',
  52         'tpu_job_name',
  53         'initial_infeed_sleep_secs',
  54         'input_partition_dims',
  55     ])):
  56   r"""TPU related configuration required by `TPUEstimator`.
  57 
  58   Args:
  59     iterations_per_loop: This is the number of train steps running in TPU
  60       system before returning to CPU host for each `Session.run`. This means
  61       global step is increased `iterations_per_loop` times in one `Session.run`.
  62       It is recommended to be set as number of global steps for next checkpoint.
  63     num_shards: (Deprecated, ignored by TPUEstimator).
  64       The number of model replicas in the system. For non-model-parallelism
  65       case, this number equals the total number of TPU cores. For
  66       model-parallelism, the total number of TPU cores equals
  67       num_cores_per_replica * num_shards.
  68     num_cores_per_replica: Defaults to `None`, which disables model parallelism.
  69       An integer which describes the number of TPU cores per model replica. This
  70       is required by model-parallelism which enables partitioning
  71       the model to multiple cores. Currently num_cores_per_replica must be
  72       1, 2, 4, or 8.
  73     per_host_input_for_training: If `True`, `PER_HOST_V1`, or `PER_HOST_V2`,
  74       `input_fn` is invoked once on each host. With the per-core input pipeline
  75       configuration, it is invoked once for each core.
  76       With a global batch size `train_batch_size` in `TPUEstimator` constructor,
  77       the batch size for each shard is `train_batch_size` // #hosts in the
  78       `True` or `PER_HOST_V1` mode. In `PER_HOST_V2` mode, it is
  79       `train_batch_size` // #cores. In `BROADCAST` mode, `input_fn` is only
  80       invoked once on host 0 and the tensors are broadcasted to all other
  81       replicas. The batch size equals to train_batch_size`. With the per-core
  82       input pipeline configuration, the shard batch size is also
  83       `train_batch_size` // #cores.
  84       Note: per_host_input_for_training==PER_SHARD_V1 only supports mode.TRAIN.
  85     tpu_job_name: The name of the TPU job. Typically, this name is auto-inferred
  86       within TPUEstimator, however when using ClusterSpec propagation in more
  87       esoteric cluster configurations, you may need to specify the job name as a
  88       string.
  89     initial_infeed_sleep_secs: The number of seconds the infeed thread should
  90       wait before enqueueing the first batch. This helps avoid timeouts for
  91       models that require a long compilation time.
  92     input_partition_dims: A nested list to describe the partition dims
  93       for all the tensors from input_fn(). The structure of
  94       input_partition_dims must match the structure of `features` and
  95       `labels` from input_fn(). The total number of partitions must match
  96       `num_cores_per_replica`. For example, if input_fn() returns two tensors:
  97       images with shape [N, H, W, C] and labels [N].
  98       input_partition_dims = [[1, 2, 2, 1], None] will split the images to 4
  99       pieces and feed into 4 TPU cores. labels tensor are directly broadcasted
 100       to all the TPU cores since the partition dims is `None`.
 101       Current limitations: This feature is only supported with the PER_HOST_V2
 102       input mode.
 103 
 104     Raises:
 105       ValueError: If `num_cores_per_replica` is not 1, 2, 4, 8 or 16.
 106   """
 107 
 108   def __new__(cls,
 109               iterations_per_loop=2,
 110               num_shards=None,
 111               num_cores_per_replica=None,
 112               per_host_input_for_training=True,
 113               tpu_job_name=None,
 114               initial_infeed_sleep_secs=None,
 115               input_partition_dims=None):
 116 
 117     # Check iterations_per_loop.
 118     util_lib.check_positive_integer(iterations_per_loop,
 119                                     'TPUConfig iterations_per_loop')
 120 
 121     # Check num_shards.
 122     if num_shards is not None:
 123       util_lib.check_positive_integer(num_shards, 'TPUConfig num_shards')
 124 
 125     if input_partition_dims is not None:
 126       if len(input_partition_dims) != 1 and len(input_partition_dims) != 2:
 127         raise ValueError(
 128             'input_partition_dims must be a list/tuple with one or two'
 129             ' elements.')
 130 
 131       if per_host_input_for_training is not InputPipelineConfig.PER_HOST_V2:
 132         raise ValueError(
 133             'input_partition_dims is only supported in PER_HOST_V2 mode.')
 134 
 135       if num_cores_per_replica is None:
 136         raise ValueError(
 137             'input_partition_dims requires setting num_cores_per_replica.')
 138 
 139     # Check num_cores_per_replica
 140     if num_cores_per_replica is not None:
 141       if num_cores_per_replica not in [1, 2, 4, 8, 16]:
 142         raise ValueError(
 143             'num_cores_per_replica must be 1, 2, 4, 8, or 16; got {}'.format(
 144                 str(num_cores_per_replica)))
 145 
 146     # per_host_input_for_training may be True, False, or integer in [1..3].
 147     # Map legacy values (True, False) to numeric values.
 148     if per_host_input_for_training is False:
 149       per_host_input_for_training = InputPipelineConfig.PER_SHARD_V1
 150     elif per_host_input_for_training is True:
 151       per_host_input_for_training = InputPipelineConfig.PER_HOST_V1
 152 
 153     # Check initial_infeed_sleep_secs.
 154     if initial_infeed_sleep_secs:
 155       util_lib.check_positive_integer(initial_infeed_sleep_secs,
 156                                       'TPUConfig initial_infeed_sleep_secs')
 157 
 158     tpu_job_name = tpu_job_name or _get_tpu_job_name_from_tf_config()
 159 
 160     return super(TPUConfig, cls).__new__(
 161         cls,
 162         iterations_per_loop=iterations_per_loop,
 163         num_shards=num_shards,
 164         num_cores_per_replica=num_cores_per_replica,
 165         per_host_input_for_training=per_host_input_for_training,
 166         tpu_job_name=tpu_job_name,
 167         initial_infeed_sleep_secs=initial_infeed_sleep_secs,
 168         input_partition_dims=input_partition_dims)

 171 class RunConfig(run_config_lib.RunConfig):
 172   """RunConfig with TPU support."""
 173 
 174   def __init__(self,
 175                tpu_config=None,
 176                evaluation_master=None,
 177                master=None,
 178                cluster=None,
 179                **kwargs):
 180     """Constructs a RunConfig.
 181 
 182     Args:
 183       tpu_config: the TPUConfig that specifies TPU-specific configuration.
 184       evaluation_master: a string. The address of the master to use for eval.
 185         Defaults to master if not set.
 186       master: a string. The address of the master to use for training.
 187       cluster: a ClusterResolver
 188       **kwargs: keyword config parameters.
 189 
 190     Raises:
 191       ValueError: if cluster is not None and the provided session_config has a
 192         cluster_def already.
 193     """
 194     super(RunConfig, self).__init__(**kwargs)
 195     self._tpu_config = tpu_config or TPUConfig()
 196     self._cluster = cluster
 197 
 198     # If user sets master and/or evaluation_master explicitly, including empty
 199     # string '', take it. Otherwise, take the values set by parent class.
 200     if master is not None:
 201       if cluster is not None:
 202         raise ValueError('Both master and cluster are set.')
 203       self._master = master
 204     else:
 205       if cluster:
 206         self._master = cluster.master()
 207 
 208     if evaluation_master is not None:
 209       self._evaluation_master = evaluation_master
 210     elif (not self._evaluation_master and
 211           self.task_type != run_config_lib.TaskType.EVALUATOR):
 212       # If the task type is EVALUATOR, it means some cluster manager sets the
 213       # TF_CONFIG. In that case, we respect the configuration in TF_CONFIG.
 214       #
 215       # Otherwise, it means user executes the code without external cluster
 216       # manager. For that, we optimize the user experience by setting
 217       # evaluation_master to master, unless user overwrites it.
 218       self._evaluation_master = self._master
 219 
 220     # Set the ClusterSpec to use
 221     if cluster:
 222       self._cluster_spec = cluster.cluster_spec()
 223 
 224       # Merge the cluster_def into the ConfigProto.
 225       if self._session_config is None:  # pylint: disable=access-member-before-definition
 226         self._session_config = config_pb2.ConfigProto(allow_soft_placement=True)
 227       if self._session_config.HasField('cluster_def'):
 228         raise ValueError(
 229             'You cannot provide a ClusterResolver and '
 230             'session_config.cluster_def.')
 231       if self._cluster_spec:
 232         self._session_config.cluster_def.CopyFrom(
 233             self._cluster_spec.as_cluster_def())
 234 
 235   def _maybe_overwrite_session_config_for_distributed_training(self):
 236     # Overrides the parent class session_config overwrite for between-graph. TPU
 237     # runs with in-graph, which should not have device filter. Doing nothing
 238     # ("pass") basically disables it.
 239     pass
 240 
 241   @property
 242   def evaluation_master(self):
 243     return self._evaluation_master
 244 
 245   @property
 246   def master(self):
 247     return self._master
 248 
 249   @property
 250   def tpu_config(self):
 251     return self._tpu_config
 252 
 253   @property
 254   def cluster(self):
 255     return self._cluster
 256 
 257   def replace(self, **kwargs):
 258     if 'tpu_config' not in kwargs:
 259       return super(RunConfig, self).replace(**kwargs)
 260 
 261     tpu_config = kwargs.pop('tpu_config')
 262     new_instance = super(RunConfig, self).replace(**kwargs)
 263     new_instance._tpu_config = tpu_config  # pylint: disable=protected-access
 264     return new_instance

 267 def _get_tpu_job_name_from_tf_config():
 268   """Extracts the TPU job name from TF_CONFIG env variable."""
 269   # TODO(xiejw): Extends this to support both TF_CONFIG env variable and cluster
 270   # spec propagation.
 271   tf_config = json.loads(os.environ.get(_TF_CONFIG_ENV, '{}'))
 272   tpu_job_name = tf_config.get(_SERVICE_KEY, {}).get(_TPU_WORKER_JOB_NAME)
 273   if tpu_job_name:
 274     logging.info('Load TPU job name from TF_CONFIG: %s', tpu_job_name)
 275   return tpu_job_name
