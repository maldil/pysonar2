  41 class InfeedQueue(object):
  42   """A helper object to build a device infeed queue.
  43 
  44   The InfeedQueue builds the host-side and device-side Ops to enqueue and
  45   dequeue elements, respectively, and ensures that their types and
  46   shapes match.
  47   """
  48 
  49   def __init__(self,
  50                number_of_tuple_elements=None,
  51                tuple_types=None,
  52                tuple_shapes=None,
  53                shard_dimensions=None,
  54                name=None):
  55     """Creates a new InfeedQueue with the given configuration.
  56 
  57     The configuration need not be fully specified at creation since it
  58     can be modified subsequently by methods that set the values
  59     explicitly or infer them from the shapes of inputs.
  60 
  61     Args:
  62       number_of_tuple_elements: the number of Tensors fed atomically through the
  63         queue, must be present unless it can be inferred from other arguments.
  64       tuple_types: if not None, a list of types of the elements of the queue.
  65       tuple_shapes: if not None, a list of shapes of the elements of the queue.
  66       shard_dimensions: if not None, a list of dimensions on which the
  67         elements of the queue should be sharded during automatic
  68         parallelization.
  69       name: the name of the queue.
  70 
  71     Raises:
  72       ValueError: if number_of_tuple_elements <= 0; or
  73         number_of_tuple_arguments, tuple_types, tuple_shapes, and
  74         shard_dimensions are all None; or the length of tuple_types,
  75         tuple_shapes, or shard_dimensions is not equal to
  76         number_of_tuple_elements; or any element of shard_dimensions
  77         can't be converted to a Dimension.
  78       TypeError: if any element of tuple_types or tuple_shapes can't
  79         be converted to a dtype or TensorShape, respectively.
  80     """
  81     self._frozen = False
  82     self._generated_enqueue_ops = False
  83     self._generated_dequeue_op = False
  84     self._name = "InfeedQueue" if name is None else name
  85     if number_of_tuple_elements is None:
  86       if tuple_types is not None:
  87         number_of_tuple_elements = len(tuple_types)
  88       elif tuple_shapes is not None:
  89         number_of_tuple_elements = len(tuple_shapes)
  90       elif shard_dimensions is not None:
  91         number_of_tuple_elements = len(shard_dimensions)
  92       else:
  93         raise ValueError(
  94             "number of tuple elements cannot be inferred from InfeedQueue "
  95             "constructor"
  96         )
  97     if number_of_tuple_elements <= 0:
  98       raise ValueError("number_of_tuple_elements %d must be > 0" %
  99                        number_of_tuple_elements)
 100     # Make an empty sharding policy for each tuple element.
 101     self._sharding_policies = [
 102         tpu_sharding.ShardingPolicy()
 103         for _ in xrange(number_of_tuple_elements)
 104     ]
 105     if tuple_types is not None:
 106       self.set_tuple_types(tuple_types)
 107     else:
 108       self._tuple_types = None
 109     if tuple_shapes is not None:
 110       self.set_tuple_shapes(tuple_shapes)
 111     else:
 112       self._tuple_shapes = None
 113     if shard_dimensions is not None:
 114       self.set_shard_dimensions(shard_dimensions)
 115     self._validate()
 116 
 117   def _validate(self):
 118     """Checks that the configuration is self-consistent.
 119 
 120     Raises:
 121       ValueError: if the shapes and sharding policies don't match.
 122     """
 123     if self.tuple_shapes is not None:
 124       for (policy, shape) in zip(self._sharding_policies, self._tuple_shapes):
 125         # Raise an error if the policy is incompatible with the shape.
 126         _ = policy.get_sharded_shape(shape)
 127 
 128   @property
 129   def number_of_tuple_elements(self):
 130     """Returns the number of InfeedQueue tuple elements."""
 131     return len(self._sharding_policies)
 132 
 133   @property
 134   def tuple_types(self):
 135     """Returns the types of the InfeedQueue tuple elements."""
 136     return self._tuple_types
 137 
 138   def set_tuple_types(self, tuple_types):
 139     """Sets the type of each element of the queue.
 140 
 141     tuple_types must be a list of length
 142     self.number_of_tuple_elements, and each element must be
 143     convertible to a dtype.
 144 
 145     Args:
 146       tuple_types: the types of each queue element.
 147 
 148     Raises:
 149       ValueError: if tuple_types is not of length
 150         self.number_of_tuple_elements.
 151       TypeError: if an element of tuple_types cannot be converted to a
 152         dtype.
 153     """
 154     if len(tuple_types) != self.number_of_tuple_elements:
 155       raise ValueError("tuple_types is %s, but must be a list of length %d" %
 156                        (str(tuple_types), self.number_of_tuple_elements))
 157     if self._frozen:
 158       for (frozen, updated) in zip(self._tuple_types, tuple_types):
 159         if frozen != updated:
 160           raise ValueError(
 161               "Trying to update InfeedQueue with frozen configuration with an "
 162               "incompatible type. Frozen types are %s, updated types are %s" % (
 163                   str(self._tuple_types), str(tuple_types)))
 164     else:
 165       try:
 166         self._tuple_types = [dtypes.as_dtype(t) for t in tuple_types]
 167       except (TypeError) as e:
 168         raise TypeError(
 169             "tuple_types is %s, but must be a list of elements each "
 170             "convertible to dtype: got error %s" % (str(tuple_types), str(e)))
 171 
 172   @property
 173   def tuple_shapes(self):
 174     """Returns the shapes of the InfeedQueue tuple elements."""
 175     return self._tuple_shapes
 176 
 177   def set_tuple_shapes(self, tuple_shapes):
 178     """Sets the shape of each element of the queue.
 179 
 180     tuple_shapes must be a list of length
 181     self.number_of_tuple_elements, and each element must be
 182     convertible to a TensorShape.
 183 
 184     Args:
 185       tuple_shapes: the shapes of each queue element.
 186 
 187     Raises:
 188       ValueError: if tuple_shapes is not of length
 189         self.number_of_tuple_elements.
 190       TypeError: if an element of tuple_shapes cannot be converted to
 191         a TensorShape.
 192     """
 193     if len(tuple_shapes) != self.number_of_tuple_elements:
 194       raise ValueError("tuple_shapes is %s, but must be a list of length %d" %
 195                        (str(tuple_shapes), self.number_of_tuple_elements))
 196     try:
 197       tuple_shapes = [tensor_shape.as_shape(shape) for shape in tuple_shapes]
 198     except (ValueError, TypeError) as e:
 199       raise TypeError(
 200           "tuple_shapes is %s, but must be a list of elements each "
 201           "convertible to TensorShape: got error %s" % (str(tuple_shapes),
 202                                                         str(e)))
 203     if self._frozen:
 204       for (frozen, updated) in zip(self._tuple_shapes, tuple_shapes):
 205         if frozen != updated:
 206           raise ValueError(
 207               "Trying to update InfeedQueue with frozen configuration with an "
 208               "incompatible shape. Frozen shapes are %s, updated shapes are %s"
 209               % (str(self._tuple_shapes), str(tuple_shapes)))
 210     else:
 211       self._tuple_shapes = tuple_shapes
 212     self._validate()

 214   @property
 215   def sharding_policies(self):
 216     """Returns the sharding policies of the InfeedQueue tuple elements."""
 217     return self._sharding_policies
 218 
 219   @property
 220   def shard_dimensions(self):
 221     """Gets the shard dimension of each tuple element.
 222 
 223     Returns:
 224       A list of length number_of_tuple_elements, where each list entry
 225       is the shard dimension of that tuple element or None if the
 226       shard dimension has not been set.
 227     """
 228     # The number of shards is always the same for all the policies.
 229     return [policy.shard_dimension for policy in self._sharding_policies]
 230 
 231   def set_shard_dimensions(self, shard_dimensions):
 232     """Sets the shard_dimension of each element of the queue.
 233 
 234     shard_dimensions must be a list of length
 235     self.number_of_tuple_elements, and each element must be
 236     convertible to a Dimension compatible with self.tuple_shapes.
 237 
 238     Args:
 239       shard_dimensions: the dimensions of each queue element.
 240 
 241     Raises:
 242       ValueError: if shard_dimensions is not of length
 243         self.number_of_tuple_elements; or an element of
 244         shard_dimensions cannot be converted to a Dimension; or an
 245         element of shard_dimensions is a Dimension that is out of
 246         range for the corresponding tuple element shape.
 247     """
 248     if len(shard_dimensions) != self.number_of_tuple_elements:
 249       raise ValueError("shard_dimensions is %s, but must be a list of length %d"
 250                        % (str(shard_dimensions),
 251                           self.number_of_tuple_elements))
 252     for (policy, dimension) in zip(self._sharding_policies, shard_dimensions):
 253       policy.set_shard_dimension(dimension)
 254     self._validate()
 255 
 256   @property
 257   def number_of_shards(self):
 258     """Gets the number of shards to use for the InfeedQueue.
 259 
 260     Returns:
 261       Number of shards or None if the number of shards has not been set.
 262     """
 263     # The number of shards is always the same for all the policies.
 264     return self._sharding_policies[0].number_of_shards
 265 
 266   def set_number_of_shards(self, number_of_shards):
 267     """Sets the number of shards to use for the InfeedQueue.
 268 
 269     Args:
 270       number_of_shards: number of ways to shard the InfeedQueue.
 271 
 272     Raises:
 273       ValueError: if number_of_shards is not > 0; or the policies have
 274         been frozen and number_of_shards was already set to something
 275         else.
 276     """
 277     for policy in self._sharding_policies:
 278       policy.set_number_of_shards(number_of_shards)
 279     self._validate()

 281   def set_configuration_from_input_tensors(self, input_tensors):
 282     """Sets the shapes and types of the queue tuple elements.
 283 
 284     input_tensors is a list of Tensors whose types and shapes are used
 285     to set the queue configuration.
 286 
 287     Args:
 288       input_tensors: list of Tensors of the same types and shapes as
 289         the desired queue Tuple.
 290 
 291     Raises:
 292       ValueError: if input_tensors is not a list of length
 293         self.number_of_tuple_elements
 294     """
 295     if len(input_tensors) != self.number_of_tuple_elements:
 296       raise ValueError(
 297           "input_tensors is %s, but should be a list of %d Tensors", (
 298               str(input_tensors), self.number_of_tuple_elements))
 299     self.set_tuple_shapes([t.shape for t in input_tensors])
 300     self.set_tuple_types([t.dtype for t in input_tensors])
 301 
 302   def set_configuration_from_sharded_input_tensors(self, input_tensors):
 303     """Sets the shapes and types of the queue tuple elements.
 304 
 305     input_tensors is a list of lists of Tensors whose types and shapes are used
 306     to set the queue configuration. The length of the outer list is the number
 307     of shards required, and each inner list is the tuple of Tensors to use to
 308     determine the types and shapes of the corresponding shard. This method
 309     depends on the shard dimension, and calling it freezes the shard policy.
 310 
 311     Args:
 312       input_tensors: list of lists of Tensors. The outer list length corresponds
 313         to the desired number of shards, and each inner list is the size
 314         and shape of the desired configuration of the corresponding shard.
 315 
 316     Raises:
 317       ValueError: if any inner list is not a list of length
 318         self.number_of_tuple_elements; or the inner lists do not combine to
 319         form a consistent unsharded shape.
 320       TypeError: if the types of the Tensors in the inner lists do not match.
 321     """
 322     if not self._frozen:
 323       # Unset the tuple shapes in case the configuration becomes
 324       # transiently inconsistent.
 325       self._tuple_shapes = None
 326     number_of_shards = len(input_tensors)
 327     self.set_number_of_shards(number_of_shards)
 328     for t in input_tensors:
 329       if len(t) != self.number_of_tuple_elements:
 330         raise ValueError(
 331             "input_tensors is %s but must be a list of lists, where each inner"
 332             " list has length number_of_tuple_elements=%d" % (
 333                 str(input_tensors), self.number_of_tuple_elements))
 334     # Transpose the inputs to make a list of shard shapes for each tuple
 335     # element.
 336     sharded_shapes = [[t[i].shape for t in input_tensors]
 337                       for i in xrange(self.number_of_tuple_elements)]
 338     # For each tuple, get the unsharded shape using that tuple's policy.
 339     unsharded_shapes = [
 340         policy.get_unsharded_shape(s)
 341         for (policy, s) in zip(self._sharding_policies, sharded_shapes)
 342     ]
 343     self.set_tuple_shapes(unsharded_shapes)
 344     for i in xrange(1, self.number_of_shards):
 345       for (t1, t2) in zip(input_tensors[0], input_tensors[i]):
 346         if t1.dtype != t2.dtype:
 347           raise TypeError(
 348               "types of the tuple elements of input_tensors %s are not "
 349               "consistent" % str(input_tensors))
 350     self.set_tuple_types([t.dtype for t in input_tensors[0]])
 351 
 352   def freeze(self):
 353     """Freezes the InfeedQueue so it can no longer be modified.
 354 
 355     The configuration is implicitly frozen before any host-side or
 356     device-side Ops are generated. The configuration cannot be frozen
 357     until the types and shapes of the tuple elements have been set.
 358 
 359     Raises:
 360       ValueError: if the types or shapes of the tuple elements have not been
 361       set.
 362     """
 363     self._frozen = True
 364     if self._tuple_types is None:
 365       raise ValueError(
 366           "Can't freeze an InfeedQueue without setting all tuple types.")
 367     if self._tuple_shapes is None:
 368       raise ValueError(
 369           "Can't freeze an InfeedQueue without setting all tuple shapes.")
 370     for shape in self._tuple_shapes:
 371       if shape.dims is None:
 372         raise ValueError(
 373             "Can't freeze an InfeedQueue without setting all tuple shapes.")
 374     for policy in self._sharding_policies:
 375       policy.freeze()
 376     self._validate()
 377 
 378   def generate_dequeue_op(self, tpu_device=0):
 379     """Generates the device-side Op to dequeue a tuple from the queue.
 380 
 381     Implicitly freezes the queue configuration if it is not already
 382     frozen, which will raise errors if the shapes and types have not
 383     been fully specified.
 384 
 385     Args:
 386       tpu_device: The TPU device ordinal where the infeed instruction should be
 387         placed. If None, no explicit placement will be performed, and it is up
 388         to the user to call this API from within a proper TPU device scope.
 389         The XLA code will fail if the TPU dequeue instruction is not bound to
 390         any device.
 391 
 392     Returns:
 393       A list of Outputs corresponding to a shard of infeed dequeued
 394       into XLA, suitable for use within a replicated block.
 395 
 396     Raises:
 397       ValueError: if the types or shapes of the tuple elements have not been
 398       set; or if a dequeue op has already been generated.
 399     """
 400     self.freeze()
 401     if self._generated_dequeue_op:
 402       raise ValueError("Can't generate two dequeue Ops from the same queue")
 403     self._generated_dequeue_op = True
 404     full_name = "%s/dequeue" % self._name
 405     sharded_shapes = [
 406         policy.get_sharded_shape(shape)
 407         for (shape, policy) in zip(self._tuple_shapes, self._sharding_policies)
 408     ]
 409     if tpu_device is not None:
 410       with ops.device(tpu.core(tpu_device)):
 411         return tpu_ops.infeed_dequeue_tuple(
 412             dtypes=self._tuple_types, shapes=sharded_shapes, name=full_name)
 413     else:
 414       return tpu_ops.infeed_dequeue_tuple(
 415           dtypes=self._tuple_types, shapes=sharded_shapes, name=full_name)
 416 
 417   def _generate_enqueue_op(self,
 418                            inputs,
 419                            name_prefix,
 420                            index,
 421                            device=None,
 422                            tpu_ordinal=-1):
 423     """Generate a host-side Op to enqueue a tuple to the queue.
 424 
 425     If device is None the inputs are all required to have the same
 426     device specification, and the enqueue Op is colocated with
 427     inputs[0]. Otherwise the enqueue Op is placed on 'device'.
 428 
 429     Args:
 430       inputs: a list of Tensors with the types and shapes of the tuple elements.
 431       name_prefix: the base name for the Op.
 432       index: the shard index, used to uniquify the Op name.
 433       device: device to place the Op on, or None if it should be
 434         colocated with the inputs.
 435       tpu_ordinal: ordinal of the TPU device on the host to use for
 436       infeed if device is a CPU device. Should be set to -1 if device
 437       is a TPU device.
 438 
 439     Returns:
 440       An Op corresponding to a shard of infeed enqueued at the host,
 441       suitable for use within a replicated block.
 442 
 443     Raises:
 444       ValueError: if device is None and inputs do not all have the
 445         same device specification.
 446     """
 447     full_name = "%s/%d" % (name_prefix, index)
 448     shapes = [t.shape for t in inputs]
 449     if device is None:
 450       devices = [t.device for t in inputs]
 451       for i in xrange(1, self.number_of_tuple_elements):
 452         if devices[0] != devices[i]:
 453           raise ValueError(
 454               "input devices for shard %d are %s, but should all be the same",
 455               index, str(devices))
 456       with ops.colocate_with(inputs[0]):
 457         return tpu_ops.infeed_enqueue_tuple(
 458             inputs=inputs,
 459             shapes=shapes,
 460             name=full_name,
 461             device_ordinal=tpu_ordinal)
 462     else:
 463       with ops.device(device):
 464         return tpu_ops.infeed_enqueue_tuple(
 465             inputs=inputs,
 466             shapes=shapes,
 467             name=full_name,
 468             device_ordinal=tpu_ordinal)

 470   def generate_enqueue_ops(self,
 471                            sharded_inputs,
 472                            tpu_ordinal_function=None,
 473                            placement_function=None):
 474     """Generates the host-side Ops to enqueue the shards of a tuple.
 475 
 476     sharded_inputs is a list, one for each shard, of lists of
 477     Tensors. sharded_inputs[0] is the tuple of Tensors to use to feed
 478     shard 0 if the queue. Returns the host-side Ops that must be run to
 479     enqueue the sharded tuple. The Op for shard i is colocated with the inputs
 480     for shard i.
 481 
 482     Implicitly freezes the queue configuration if it is not already
 483     frozen. If the configuration has already been frozen, and is not
 484     compatible with the types and shapes of sharded_inputs, an error
 485     will be raised.
 486 
 487     Args:
 488       sharded_inputs: a list of lists of Tensors. The length of the outer list
 489         determines the number of shards. Each inner list indicates the types
 490         and shapes of the tuples in the corresponding shard.
 491       tpu_ordinal_function: if not None, a function that takes the
 492         shard index as input and returns the ordinal of the TPU device
 493         the shard's infeed should be placed on. tpu_ordinal_function must be
 494         set if the inputs are placed on CPU devices.
 495       placement_function: if not None, a function that takes the shard index as
 496         input and returns the host device where the enqueue op should be placed
 497         on.
 498 
 499     Returns:
 500       A list of host-side Ops, one for each shard, that when executed together
 501       will enqueue a full-size element of infeed.
 502 
 503     Raises:
 504       ValueError: if the queue configuration has previously been frozen and the
 505         shapes of the elements of sharded_inputs are not compatible with the
 506         frozen configuration; or if the shapes of the elements of sharded_inputs
 507         don't form a consistent unsharded tuple; or if the elements of a tuple
 508         have different device constraints.
 509       TypeError: if the queue configuration has previously been frozen and the
 510         types of the elements of sharded_inputs are not compatible with the
 511         frozen configuration; or if the types of the elements of sharded_inputs
 512         don't form a consistent unsharded tuple.
 513     """
 514     self.set_configuration_from_sharded_input_tensors(sharded_inputs)
 515     self.freeze()
 516     if self._generated_enqueue_ops:
 517       raise ValueError("Can't generate two enqueue Ops from the same queue")
 518     self._generated_enqueue_ops = True
 519     if tpu_ordinal_function is None:
 520       tpu_ordinal_function = lambda index: -1
 521     name_prefix = "%s/enqueue" % self._name
 522     return [
 523         self._generate_enqueue_op(
 524             shard,
 525             name_prefix,
 526             index,
 527             tpu_ordinal=tpu_ordinal_function(index),
 528             device=placement_function(index) if placement_function else None)
 529         for (shard, index) in zip(sharded_inputs, xrange(self.number_of_shards))
 530     ]
 531 
 532   # TODO(misard) Generalize this to the case of systems that don't
 533   # have 8 devices per host, and figure out what to do with

 535   def _default_placement_function(self, index):

 538   def _default_ordinal_function(self, index):

 541   # TODO(b/36470756) remove this from tutorials once we have a better story
 542   # for automatic placement of input pipelines.
 543   def split_inputs_and_generate_enqueue_ops(self,
 544                                             inputs,
 545                                             device_assignment=None,
 546                                             placement_function=None,
 547                                             tpu_ordinal_function=None):
 548     """POORLY-PERFORMING ON MULTI-HOST SYSTEMS.
 549 
 550     Generates the host-side Ops to enqueue a tuple.
 551 
 552     This method performs poorly because it takes an entire input on a single
 553     host, splits it, and distributes it to all of the cores. It is present only
 554     to simplify tutorial examples.
 555 
 556     inputs is a list of Tensors to use to feed the queue. Each input is split
 557     into self.number_of_shards shards. Returns an Op for each shard to enqueue
 558     the shard. The Op for shard i is placed on device placement_function(i).
 559 
 560     Implicitly freezes the queue configuration if it is not already
 561     frozen. If the configuration has already been frozen, and is not
 562     compatible with the types and shapes of inputs, an error
 563     will be raised.
 564 
 565     Args:
 566       inputs: a list of Tensors which indicates the types and shapes of the
 567         queue tuple.
 568      device_assignment: if not `None`, a TPU `DeviceAssignment`. If
 569         device_assignment is not `None`, but `placement_function` and
 570         `ordinal_function` are None, then `device_assignment` will be used to
 571         place infeeds on the first k TPU shards, where k is the number of shards
 572         in the queue. If all three are `None`, then default placement and
 573         ordinal functions are used.
 574       placement_function: if not None, a function that takes the shard
 575         index as input and returns a device string indicating which
 576         device the shard's infeed should be placed on. If placement_function
 577         and tpu_ordinal_function are None, inputs are sharded round-robin
 578         across the devices in the system.
 579       tpu_ordinal_function: if not None, a function that takes the
 580         shard index as input and returns the ordinal of the TPU device
 581         the shard's infeed should be placed on. If placement_function
 582         and tpu_ordinal_function are None, inputs are sharded round-robin
 583         across the devices in the system.
 584 
 585     Returns:
 586       A list of host-side Ops, one for each shard, that when executed together
 587       will enqueue a full-size element of infeed.
 588 
 589     Raises:
 590       ValueError: if the queue configuration has previously been frozen and the
 591         shapes of the elements of inputs are not compatible with the frozen
 592         configuration.
 593       TypeError: if the queue configuration has previously been frozen and the
 594         types of the elements of inputs are not compatible with the frozen
 595         configuration.
 596     """
 597     if device_assignment is None:
 598       if placement_function is None:
 599         placement_function = self._default_placement_function
 600       if tpu_ordinal_function is None:
 601         tpu_ordinal_function = self._default_ordinal_function
 602     else:
 603 
 604       def _placement_function_from_map(index):
 605         return device_assignment.host_device(replica=index)
 606 
 607       def _ordinal_function_from_map(index):
 608         return device_assignment.tpu_ordinal(replica=index)
 609 
 610       if placement_function is None:
 611         placement_function = _placement_function_from_map
 612       if tpu_ordinal_function is None:
 613         tpu_ordinal_function = _ordinal_function_from_map
 614     self.set_configuration_from_input_tensors(inputs)
 615     self.freeze()
 616     if self._generated_enqueue_ops:
 617       raise ValueError("Can't generate two enqueue Ops from the same queue")
 618     self._generated_enqueue_ops = True
 619     split_name_prefix = "%s/split" % self._name
 620     if self.number_of_shards == 1:
 621       transposed_sharded_inputs = [[inp] for inp in inputs]
 622     else:
 623 
 624       def split_fn(inp, num_shards, axis, name):
 625         with ops.colocate_with(inp):
 626           return array_ops.split(inp, num_shards, axis=axis, name=name)
 627 
 628       transposed_sharded_inputs = [
 629           split_fn(
 630               inp,
 631               self.number_of_shards,
 632               axis=policy.shard_dimension,
 633               name="%s/%d" % (split_name_prefix, index))
 634           for (inp, policy, index) in zip(inputs, self._sharding_policies,
 635                                           xrange(self.number_of_tuple_elements))
 636       ]
 637     sharded_inputs = [[shard[i] for shard in transposed_sharded_inputs]
 638                       for i in xrange(self.number_of_shards)]
 639     name_prefix = "%s/enqueue" % self._name
 640     return [
 641         self._generate_enqueue_op(
 642             shard,
 643             name_prefix,
 644             index,
 645             device=placement_function(index),
 646             tpu_ordinal=tpu_ordinal_function(index))
 647         for (shard, index) in zip(sharded_inputs, xrange(self.number_of_shards))
 648     ]

 651 class _PartitionedInfeedQueue(InfeedQueue):
 652   """A helper object to build a device infeed queue with input partition.
 653 
 654   Args:
 655     number_of_tuple_elements: the number of Tensors fed atomically through the
 656       queue, must be present unless it can be inferred from other arguments.
 657     device_assignment: A TPU `DeviceAssignment` which is used to place all the
 658       partitions to different TPU infeed queues.
 659     host_id: The id of the host machine.
 660     input_partition_dims: A nested list/tuple of integers. Each inner
 661       list/tuple describes how to partition the corresponding input tensor.
 662     tuple_types: If not None, a list of types of the elements of the queue.
 663     tuple_shapes: If not None, a list of shapes of the elements of the queue.
 664     name: The name of the queue.
 665   """
 666 
 667   def __init__(self,
 668                number_of_tuple_elements,
 669                device_assignment,
 670                host_id,
 671                input_partition_dims=None,
 672                tuple_types=None,
 673                tuple_shapes=None,
 674                name=None):
 675     super(_PartitionedInfeedQueue, self).__init__(
 676         number_of_tuple_elements=number_of_tuple_elements,
 677         tuple_types=tuple_types,
 678         tuple_shapes=None,
 679         shard_dimensions=None,
 680         name="PartitionedInfeedQueue" if name is None else name)
 681     self._input_partition_dims = input_partition_dims
 682     self._host_id = host_id
 683     self._device_assignment = device_assignment
 684 
 685   def generate_dequeue_op(self, tpu_device=0):
 686     """Generate TPU dequeue ops.
 687 
 688     Args:
 689       tpu_device: The TPU device ordinal where the infeed instruction should be
 690         placed.
 691 
 692     Returns:
 693       A list of Outputs corresponding to a partition of infeed dequeued
 694       into XLA, suitable for use within a replicated block.
 695 
 696     Raises:
 697       ValueError: if the types or shapes of the tuple elements have not been
 698       set; or if a dequeue op has already been generated.
 699     """
 700     self.freeze()
 701     if self._generated_dequeue_op:
 702       raise ValueError("Can't generate two dequeue Ops from the same queue")
 703     self._generated_dequeue_op = True
 704     full_name = "%s/dequeue" % self._name
 705     sharded_shapes = [
 706         policy.get_sharded_shape(shape)
 707         for (shape, policy) in zip(self._tuple_shapes, self._sharding_policies)
 708     ]
 709     with ops.device(tpu.core(tpu_device)):
 710       values = tpu_ops.infeed_dequeue_tuple(
 711           dtypes=self._tuple_types, shapes=sharded_shapes, name=full_name)
 712     return self._tag_sharding_attribute_for_dequeued_tensors(
 713         values, self._input_partition_dims)
 714 
 715   def generate_enqueue_ops(self, per_host_sharded_inputs):
 716     """Generates the host-side Ops to enqueue the partitioned inputs.
 717 
 718     per_host_sharded_inputs is a list, one for each replica, of lists of
 719     Tensors. sharded_inputs[i] is the tuple of Tensors to use to feed
 720     replica i.
 721     sharded_inputs[i][j] is partitioned by self._input_partition_dims[j].
 722 
 723     For example, if sharded_inputs[i][j] is a 2-D Tensor:
 724     [[A, B, C, D],
 725      [E ,F, G, H]]
 726     self._input_partition_dims[j] is [2, 4].
 727 
 728     sharded_inputs[i][j] will be partitioned and flattened into:
 729     [A, B, C, D, E, F, G, H] and fed into the logical core ids:
 730     [0, 1, 2, 3, 4, 5, 6, 7] respectively.
 731 
 732     Args:
 733       per_host_sharded_inputs: a list of lists of Tensors. The length of the
 734         outer list determines the number of shards. Each inner list indicates
 735         the types and shapes of the tuples in the corresponding shard.
 736 
 737     Returns:
 738       A list of host-side Ops, one for each shard, that when executed together
 739       will enqueue a full-size element of infeed.
 740 
 741     Raises:
 742       ValueError: if the queue configuration has previously been frozen and the
 743         shapes of the elements of sharded_inputs are not compatible with the
 744         frozen configuration; or if the shapes of the elements of sharded_inputs
 745         don't form a consistent unsharded tuple; or if the elements of a tuple
 746         have different device constraints; or if the partition dims are invalid.
 747       TypeError: if the queue configuration has previously been frozen and the
 748         types of the elements of sharded_inputs are not compatible with the
 749         frozen configuration; or if the types of the elements of sharded_inputs
 750         don't form a consistent unsharded tuple.
 751     """
 752     self.set_configuration_from_sharded_input_tensors(per_host_sharded_inputs)
 753     number_of_replicas_per_host = len(per_host_sharded_inputs)
 754     number_of_tuple_elements = len(per_host_sharded_inputs[0])
 755 
 756     assert len(self._input_partition_dims) == number_of_tuple_elements
 757     per_host_enqueue_ops = []
 758 
 759     for replica_index in range(number_of_replicas_per_host):
 760       flattened_inputs = per_host_sharded_inputs[replica_index]
 761       inputs_part_dims_flat = nest.flatten_up_to(flattened_inputs,
 762                                                  self._input_partition_dims)
 763       inputs_parted_iters = [
 764           iter(self._partition_or_replicate_on_host(x, dims)) for x, dims in
 765           zip(per_host_sharded_inputs[replica_index], inputs_part_dims_flat)
 766       ]
 767 
 768       for logical_core in xrange(self._device_assignment.num_cores_per_replica):
 769         # Places different partitions to different logic cores.
 770         replica_id = self._device_assignment.lookup_replicas(
 771             self._host_id, logical_core)[replica_index]
 772         ordinal = self._device_assignment.tpu_ordinal(
 773             replica=replica_id, logical_core=logical_core)
 774         infeed_inputs = []
 775         for it in inputs_parted_iters:
 776           input_for_device = next(it, None)
 777           if input_for_device is not None:
 778             infeed_inputs.append(input_for_device)
 779 
 780         if infeed_inputs:
 781           per_host_enqueue_ops.append(
 782               tpu_ops.infeed_enqueue_tuple(
 783                   inputs=infeed_inputs,
 784                   shapes=[x.shape for x in infeed_inputs],
 785                   name="enqueue/replica_{0}/input_{1}".format(
 786                       replica_index, logical_core),
 787                   device_ordinal=ordinal))
 788     return per_host_enqueue_ops

 790   def _check_input_partition_dims(self, tensor, dims):
 791     """Checks that input partition dims are valid for the `Tensor`.
 792 
 793     Args:
 794       tensor: Input tensor for partitioning.
 795       dims: A list of integer describes how to partition the input tensor.
 796 
 797     Raises:
 798       ValueError: If the tensor can't be partitioned by dims or the
 799         num_cores_per_replica doesn't match the number of
 800         partitions(dims.prod()).
 801     """
 802     if dims is None:
 803       return
 804 
 805     dims = np.array(dims)
 806 
 807     if (dims < 1).any():
 808       raise ValueError("All input partition dims must be >= 1.")
 809 
 810     # No partitioning, so don't perform further checks.
 811     if dims.prod() == 1:
 812       return
 813 
 814     if dims.prod() != self._device_assignment.num_cores_per_replica:
 815       raise ValueError(
 816           "The product of each input parition dim should equal to "
 817           "num_cores_per_replica. (dim = {}, num_cores_per_replica "
 818           "= {})".format(dims, self._device_assignment.num_cores_per_replica))
 819     if dims.shape[0] != tensor.shape.ndims:
 820       raise ValueError(
 821           "Input partition dims must have the same number of dimensions "
 822           "as the `Tensor` to be partitioned. (tensor shape = {}, input "
 823           "partition dims = {}).".format(tensor.shape.as_list(), dims))
 824 
 825     tensor.shape.assert_is_fully_defined()
 826     if (np.array(tensor.shape.as_list()) % dims != 0).any():
 827       raise ValueError(
 828           "All input partition dims must divide exactly into the `Tensor` "
 829           "shape (tensor shape = {}, input partition dims = {}).".format(
 830               tensor.shape.as_list(), dims))
 831 
 832   def _partition_or_replicate_on_host(self, tensor, dims):
 833     """Partitions or replicates the input tensor.
 834 
 835       The ops inside this function are placed on the host side.
 836 
 837     Args:
 838       tensor: The input tensor which will be partioned or replicated.
 839       dims: A list of integer describes how to partition the input tensor.
 840     Returns:
 841       An iterator of `Tensor`s or a list of partioned tensors.
 842     """
 843     self._check_input_partition_dims(tensor, dims)
 844     if dims is None:
 845       return itertools.repeat(tensor)
 846     else:
 847       output = [tensor]
 848       for axis, dim in enumerate(dims):
 849         if dim > 1:
 850           output = [array_ops.split(x, dim, axis=axis) for x in output]
 851           output = nest.flatten(output)
 852       return output

 854   def _tag_sharding_attribute_for_dequeued_tensor(self, tensor, dims):
 855     """Tags appropriate XLA sharding attribute to the dequeued tensor.
 856 
 857     Args:
 858       tensor: The dequeued tensor on TPU.
 859       dims: A list of integer describes how the tensor is partitioned.
 860 
 861     Returns:
 862       The same tensor with the xla_sharding attribute.
 863     """
 864     if dims is None:
 865       return xla_sharding.replicate(tensor)
 866     elif np.prod(dims) == 1:
 867       return xla_sharding.assign_device(tensor, 0)
 868     else:
 869       tile_shape = np.array(tensor.shape.as_list()) // dims
 870       tile_assignment = np.arange(np.prod(dims)).reshape(dims)
 871       return xla_sharding.tile(
 872           tensor=tensor,
 873           tile_shape=xla_shape.CreateShapeFromDtypeAndTuple(
 874               dtype=np.dtype(tensor.dtype.as_numpy_dtype),
 875               shape_tuple=tile_shape),
 876           tile_assignment=tile_assignment)
 877 
 878   def _tag_sharding_attribute_for_dequeued_tensors(self, dequeues, dims):
 879     """Tags appropriate XLA sharding attribute to the dequeued tensors.
 880 
 881     Args:
 882       dequeues: A list of dequeued tensors on TPU.
 883       dims: A list of integer describes how the tensor is partitioned.
 884 
 885     Returns:
 886       The same dequeues with appropriate xla_sharding attribute.
 887     """
 888     nest.assert_shallow_structure(dequeues, dims)
 889     return nest.map_structure_up_to(
 890         dequeues, self._tag_sharding_attribute_for_dequeued_tensor, dequeues,
 891         dims)
