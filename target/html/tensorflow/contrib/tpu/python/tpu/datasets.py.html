  32 def _TextLineDataset(filename):
  33   buffer_size = 8 * 1024 * 1024  # 8 MiB per file
  34   dataset = readers.TextLineDataset(filename, buffer_size=buffer_size)
  35   return dataset
  36 
  37 
  38 def _TFRecordDataset(filename):
  39   buffer_size = 8 * 1024 * 1024  # 8 MiB per file
  40   dataset = readers.TFRecordDataset(filename, buffer_size=buffer_size)
  41   return dataset

  51 def StreamingFilesDataset(files,
  52                           filetype=None,
  53                           file_reader_job=None,
  54                           worker_job=None,
  55                           num_epochs=None,
  56                           filename_shuffle_buffer_size=None,
  57                           num_parallel_reads=None,
  58                           batch_transfer_size=None,
  59                           sloppy=None):
  60   """StreamingFilesDataset constructs a dataset to stream from workers (GCE VM).
  61 
  62   Because Cloud TPUs are allocated over the network, a Cloud TPU cannot read
  63   files local to your GCE VM. In order to train using files stored on your local
  64   VM (e.g. on local SSD for extreme performance), use the StreamingFilesDataset
  65   helper to generate a dataset to feed your Cloud TPU with files from your GCE
  66   VM.
  67 
  68   The resulting dataset may return an OutOfRangeError if there are no files
  69   found as a result of the fileglob expansion.
  70 
  71   Note: StreamingFilesDataset assumes that the session is using a
  72   TPUClusterResolver and has therefore a worker and a coordinator job. File
  73   loading will be done on the coordinator job.
  74 
  75   Args:
  76     files: A string glob to match files, or a `tf.data.Dataset` generating file
  77       names.
  78     filetype: A string (one of 'tfrecord', or 'textline') or a single-argument
  79       TensorFlow function that when given a filename returns a dataset.
  80     file_reader_job: An optional string that corresponds to the job that should
  81       perform the file reads.
  82     worker_job: An optional string that corresponds to the job that should
  83       process the tensors (i.e. your GPU or TPU worker).
  84     num_epochs: The number of epochs through the training set that should be
  85       generated. By default, it will repeat infinitely.
  86     filename_shuffle_buffer_size: An optional integer whose value controls the
  87       shuffling of the file names. If you would like to read from the files in
  88       the same order, set to 0 or False.
  89     num_parallel_reads: An optional integer controlling the number of files to
  90       read from concurrently. (Set to 1 for no parallelism.)
  91     batch_transfer_size: An optional integer controlling the batching used to
  92       amortize the remote function invocation overhead. Set to a very large
  93       number to increase throughput. Set to a very small number to reduce memory
  94       consumption. Set to False to skip batching.
  95     sloppy: (Optional.) If `False`, read input data while maintaining a
  96       deterministic order. (This may have significant performance impacts.)
  97       sloppy defaults to: True.
  98   Returns:
  99     A `tf.data.Dataset` with an infinite stream of elements generated by a
 100     parallel interleaving of the set of files matched (or generated) by `files`
 101     with a type is the output of the dataset specified by `filetype`.
 102 
 103   Raises:
 104     ValueError: if any argument is not of the expected type.
 105   """
 106   if filetype is None:
 107     filetype = 'tfrecord'
 108 
 109   if isinstance(filetype, str):
 110     if filetype not in _FILETYPE_MAP:
 111       raise ValueError('Unexpected filetype: %s' % filetype)
 112     reader_fn = _FILETYPE_MAP[filetype]
 113   elif callable(filetype):
 114     reader_fn = filetype
 115   else:
 116     raise ValueError('filetype should be a string or a callable')
 117 
 118   file_reader_job = file_reader_job or 'coordinator'
 119 
 120   worker_job = worker_job or 'worker'
 121 
 122   if filename_shuffle_buffer_size is None:
 123     filename_shuffle_buffer_size = 4096
 124 
 125   num_parallel_reads = num_parallel_reads or 8
 126 
 127   if batch_transfer_size is None:
 128     batch_transfer_size = 256
 129 
 130   if sloppy is None:
 131     sloppy = True
 132 
 133   with ops.device('/job:%s' % file_reader_job):
 134     if isinstance(files, str):
 135       source_dataset = dataset_ops.Dataset.list_files(files)
 136     elif isinstance(files, dataset_ops.Dataset):
 137       source_dataset = files
 138     else:
 139       raise ValueError('files was not a string or a dataset: %s' % files)
 140 
 141     if filename_shuffle_buffer_size:
 142       source_dataset = source_dataset.shuffle(
 143           buffer_size=filename_shuffle_buffer_size)
 144 
 145     # NOTE: We perform the `repeat` on the source dataset, because the output
 146     # dataset does not currently have enough information to recreate an iterator
 147     # over the source dataset when it reaches the end.
 148     source_dataset = source_dataset.repeat(num_epochs)
 149 
 150     source_dataset = source_dataset.apply(
 151         interleave_ops.parallel_interleave(
 152             reader_fn, cycle_length=num_parallel_reads, sloppy=sloppy))
 153 
 154     if batch_transfer_size:
 155       source_dataset = source_dataset.batch(batch_transfer_size)
 156 
 157     source_dataset = source_dataset.prefetch(1)
 158 
 159     source_iterator = source_dataset.make_one_shot_iterator()
 160     source_handle = source_iterator.string_handle()
 161 
 162   @function.Defun(dtypes.string)
 163   def LoadingFunc(h):
 164     remote_iterator = iterator_ops.Iterator.from_string_handle(
 165         h, source_dataset.output_types, source_dataset.output_shapes)
 166     return remote_iterator.get_next()
 167 
 168   def MapFn(unused_input):
 169     if isinstance(source_dataset.output_types, dtypes.DType):
 170       output_types = [source_dataset.output_types]
 171     elif isinstance(source_dataset.output_types, (list, tuple)):
 172       output_types = source_dataset.output_types
 173     else:
 174       raise ValueError('source dataset has invalid output types')
 175     remote_calls = functional_ops.remote_call(
 176         args=[source_handle],
 177         Tout=output_types,
 178         f=LoadingFunc,
 179         target='/job:%s/replica:0/task:0/cpu:0' % file_reader_job)
 180     if len(remote_calls) == 1:
 181       return remote_calls[0]
 182     else:
 183       return remote_calls
 184 
 185   with ops.device('/job:%s' % worker_job):
 186     output_dataset = dataset_ops.Dataset.range(2).repeat().map(
 187         MapFn, num_parallel_calls=4 if sloppy else None)
 188     output_dataset = output_dataset.prefetch(1)
 189 
 190     if batch_transfer_size:
 191       # Undo the batching used during the transfer.
 192       output_dataset = output_dataset.apply(batching.unbatch()).prefetch(1)
 193 
 194   return output_dataset
