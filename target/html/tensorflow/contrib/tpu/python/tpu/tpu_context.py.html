  43 class TPUContext(object):
  44   """The context of current input_fn invocation."""
  45 
  46   def __init__(self,
  47                internal_ctx,
  48                input_device=None,
  49                invocation_index=None,
  50                call_from_input_fn=True):
  51     self._internal_ctx = internal_ctx
  52     self._input_device = input_device
  53     self._invocation_index = invocation_index
  54     self._call_from_input_fn = call_from_input_fn
  55 
  56   def current_input_fn_deployment(self):
  57     """The configuration of the current input_fn invocation.
  58 
  59     The configuration depends on `TPUConfig.per_host_input_for_training`. See
  60     `TPUConfig` for details.
  61 
  62     Only set in params dict of input_fn
  63 
  64     Returns:
  65       A tuple of
  66         1. Device spec string: String, is the current CPU host where the
  67            input_fn is invoked.
  68         2. Current invocation index: Int, 0-based index of the input_fn
  69            invocation. See next item for details.
  70         3. Total invocation count: Int, the total number of times to invoke the
  71            input_fn on all CPU hosts. Each invocation will be passed with a new
  72            `TPUContext` instance with current invocation index set properly.
  73         4. Total number of replicas consumed by current_invocation: Int, the
  74            number of replicas fed by the data returned by current input_fn. For
  75            example, for per_core input pipeline deployment
  76            and non-model-parallelism, total invocation count is equal to
  77            the number of cores in the system and num replicas consumed by
  78            current invocation is 1. For per-host v2 input pipeline deployment,
  79            total invocation count is equal to the number of hosts in the system
  80            and num replicas consumed by current invocation is equal to number of
  81            cores per host.
  82 
  83     Raises:
  84       RuntimeError: If this method must not be called from input_fn.
  85     """
  86     if not self._call_from_input_fn:
  87       raise RuntimeError('This TPUContext instance must not be called from'
  88                          ' model_fn.')
  89 
  90     if self._internal_ctx.is_input_sharded_per_core():
  91       total_invocation_count = (self._internal_ctx.num_hosts
  92                                 * self._internal_ctx.num_of_replicas_per_host)
  93       replicas_consumed = 1
  94     elif self._internal_ctx.is_input_broadcast_with_iterators():
  95       total_invocation_count = 1
  96       replicas_consumed = self._internal_ctx.num_replicas
  97     else:
  98       total_invocation_count = self._internal_ctx.num_hosts
  99       replicas_consumed = self._internal_ctx.num_of_replicas_per_host
 100     return (self._input_device, self._invocation_index,
 101             total_invocation_count, replicas_consumed)
 102

 142     """Returns the tuple of (CPU device and device ordinal) for replica.

 144     This should be used for full replicate for non-model-parallelism.

 172 class _InternalTPUContext(object):
 173   """A context holds immutable states of TPU computation.
 174 
 175   This immutable object holds TPUEstimator config, train/eval batch size, and
 176   `TPUEstimator.use_tpu`, which is expected to be passed around. It also
 177   provides utility functions, based on the current state, to determine other
 178   information commonly required by TPU computation, such as TPU device names,
 179   TPU hosts, shard batch size, etc.
 180 
 181   if eval_on_tpu is False, then execution of eval on TPU is disabled.
 182   if eval_on_tpu is True, but use_tpu is False, a warning is issued,
 183   and TPU execution is disabled for all modes.
 184 
 185   N.B. As `mode` is not immutable state in Estimator, but essential to
 186   distinguish between TPU training and evaluation, a common usage for
 187   _InternalTPUContext with `mode` is as follows:
 188   ```
 189   with _ctx.with_mode(mode) as ctx:
 190     if ctx.is_running_on_cpu():
 191        ...
 192   ```
 193   """
 194 
 195   def __init__(self, config, train_batch_size, eval_batch_size,
 196                predict_batch_size, use_tpu, eval_on_tpu=True):
 197     self._config = config
 198     self._train_batch_size = train_batch_size
 199     self._eval_batch_size = eval_batch_size
 200     self._predict_batch_size = predict_batch_size
 201     self._use_tpu = use_tpu
 202     logging.info('_TPUContext: eval_on_tpu %s', eval_on_tpu)
 203     if not use_tpu and eval_on_tpu:
 204       logging.warning('eval_on_tpu ignored because use_tpu is False.')
 205 
 206     self._eval_on_tpu = eval_on_tpu
 207     self._model_parallelism_enabled = (
 208         use_tpu and config.tpu_config.num_cores_per_replica)
 209     self._mode = None
 210     num_cores_per_replica = config.tpu_config.num_cores_per_replica
 211     if num_cores_per_replica:
 212       self._computation_shape = _NUM_CORES_TO_COMPUTATION_SHAPE[
 213           num_cores_per_replica]
 214     else:
 215       self._computation_shape = None
 216     self._lazy_tpu_system_metadata_dict = {}  # key by master address
 217     self._lazy_device_assignment_dict = {}  # key by master address
 218     self._lazy_validation_dict = {}  # key by ModeKeys
 219 
 220   def _assert_mode(self):
 221     if self._mode is None:
 222       raise RuntimeError(
 223           '`mode` needs to be set via contextmanager `with_mode`.')
 224     return self._mode
 225 
 226   @contextmanager
 227   def with_mode(self, mode):
 228     # NOTE(xiejw): Shallow copy is enough. It will share he lazy dictionaries,
 229     # such as _lazy_tpu_system_metadata_dict between new copy and the original
 230     # one. Note that all lazy states stored in properties _lazy_foo are sort of
 231     # immutable as they should be same for the process lifetime.
 232     new_ctx = copy.copy(self)
 233     new_ctx._mode = mode  # pylint: disable=protected-access
 234     yield new_ctx
 235 
 236   @property
 237   def mode(self):
 238     return self._assert_mode()
 239 
 240   def _get_master_address(self):
 241     mode = self._assert_mode()
 242     config = self._config
 243     master = (
 244         config.master
 245         if mode != model_fn_lib.ModeKeys.EVAL else config.evaluation_master)
 246     return master
 247 
 248   def _get_tpu_system_metadata(self):
 249     """Gets the (maybe cached) TPU system metadata."""
 250     master = self._get_master_address()
 251     tpu_system_metadata = self._lazy_tpu_system_metadata_dict.get(master)
 252     if tpu_system_metadata is not None:
 253       return tpu_system_metadata
 254 
 255     cluster_def = None
 256     if (self._config.session_config and
 257         self._config.session_config.cluster_def.job):
 258       cluster_def = self._config.session_config.cluster_def
 259 
 260     # pylint: disable=protected-access
 261     tpu_system_metadata = (
 262         tpu_system_metadata_lib._query_tpu_system_metadata(
 263             master,
 264             cluster_def=cluster_def,
 265             query_topology=self.model_parallelism_enabled))
 266 
 267     self._lazy_tpu_system_metadata_dict[master] = tpu_system_metadata
 268     return tpu_system_metadata
 269 
 270   def _get_device_assignment(self):
 271     """Gets the (maybe cached) TPU device assignment."""
 272     master = self._get_master_address()
 273     device_assignment = self._lazy_device_assignment_dict.get(master)
 274     if device_assignment is not None:
 275       return device_assignment
 276 
 277     tpu_system_metadata = self._get_tpu_system_metadata()
 278 
 279     device_assignment = tpu_device_assignment.device_assignment(
 280         tpu_system_metadata.topology,
 281         computation_shape=self._computation_shape,
 282         num_replicas=self.num_replicas)
 283 
 284     logging.info('num_cores_per_replica: %s',
 285                  str(self._config.tpu_config.num_cores_per_replica))
 286     logging.info('computation_shape: %s', str(self._computation_shape))
 287     logging.info('num_replicas: %d', self.num_replicas)
 288     logging.info('device_assignment.topology.device_coordinates: %s',
 289                  str(device_assignment.topology.device_coordinates))
 290     logging.info('device_assignment.core_assignment: %s',
 291                  str(device_assignment.core_assignment))
 292 
 293     self._lazy_device_assignment_dict[master] = device_assignment
 294     return device_assignment

 319   @property
 320   def num_of_replicas_per_host(self):
 321     """Return the number of replicas per host."""
 322     if self.model_parallelism_enabled:
 323       return self.num_replicas // self.num_hosts
 324     else:
 325       return self.num_of_cores_per_host

 327   @property
 328   def num_replicas(self):
 329     num_cores_in_system = self.num_cores
 330 
 331     if self.model_parallelism_enabled:
 332       num_cores_per_replica = self._config.tpu_config.num_cores_per_replica
 333       if num_cores_per_replica > num_cores_in_system:
 334         raise ValueError(
 335             'The num of cores required by the model parallelism, specified by '
 336             'TPUConfig.num_cores_per_replica, is larger than the total num of '
 337             'TPU cores in the system. num_cores_per_replica: {}, num cores '
 338             'in the system: {}'.format(num_cores_per_replica,
 339                                        num_cores_in_system))
 340 
 341       if num_cores_in_system % num_cores_per_replica != 0:
 342         raise RuntimeError(
 343             'The num of cores in the system ({}) is not divisible by the num '
 344             'of cores ({}) required by the model parallelism, specified by '
 345             'TPUConfig.num_cores_per_replica. This should never happen!'.format(
 346                 num_cores_in_system, num_cores_per_replica))
 347 
 348       return num_cores_in_system // num_cores_per_replica
 349     else:
 350       return num_cores_in_system

 361   def is_input_sharded_per_core(self):
 362     """Return true if input_fn is invoked per-core (other than per-host)."""
 363     mode = self._assert_mode()
 364     return (mode == model_fn_lib.ModeKeys.TRAIN and
 365             (self._config.tpu_config.per_host_input_for_training is
 366              tpu_config.InputPipelineConfig.PER_SHARD_V1))
 367 
 368   def is_input_per_host_with_iterators(self):
 369     """Return true if input_fn should be run in the per-host v2 config."""
 370     return (self._config.tpu_config.per_host_input_for_training is
 371             tpu_config.InputPipelineConfig.PER_HOST_V2)
 372 
 373   def is_input_broadcast_with_iterators(self):
 374     """Return true if input_fn should be run in the full_replicae config."""
 375     return (self._config.tpu_config.per_host_input_for_training is
 376             tpu_config.InputPipelineConfig.BROADCAST)

 378   def is_running_on_cpu(self, is_export_mode=False):
 379     """Determines whether the input_fn and model_fn should be invoked on CPU.
 380 
 381     This API also validates user provided configuration, such as batch size,
 382     according the lazy initialized TPU system metadata.
 383 
 384     Args:
 385       is_export_mode: Indicates whether the current mode is for exporting the
 386         model, when mode == PREDICT. Only with this bool, we could
 387         tell whether user is calling the Estimator.predict or
 388         Estimator.export_savedmodel, which are running on TPU and CPU
 389         respectively. Parent class Estimator does not distinguish these two.
 390 
 391     Returns:
 392       bool, whether current input_fn or model_fn should be running on CPU.
 393 
 394     Raises:
 395       ValueError: any configuration is invalid.
 396     """
 397 
 398     is_running_on_cpu = self._is_running_on_cpu(is_export_mode)
 399     if not is_running_on_cpu:
 400       self._validate_tpu_configuration()
 401     return is_running_on_cpu
 402 
 403   def _is_running_on_cpu(self, is_export_mode):
 404     """Determines whether the input_fn and model_fn should be invoked on CPU."""
 405     mode = self._assert_mode()
 406 
 407     if not self._use_tpu:
 408       return True
 409 
 410     if mode == model_fn_lib.ModeKeys.EVAL and not self._eval_on_tpu:
 411       logging.info('_is_running_on_cpu: eval_on_tpu disabled')
 412       return True
 413 
 414     if is_export_mode:
 415       return True
 416 
 417     return False

 419   @property
 420   def global_batch_size(self):
 421     mode = self._assert_mode()
 422     if mode == model_fn_lib.ModeKeys.TRAIN:
 423       return self._train_batch_size
 424     elif mode == model_fn_lib.ModeKeys.EVAL:

 427       return self._predict_batch_size

 429       return None

 431   @property
 432   def batch_size_for_input_fn(self):
 433     """Returns the shard batch size for `input_fn`."""
 434     global_batch_size = self.global_batch_size
 435 
 436     if (self.is_running_on_cpu() or self.is_input_broadcast_with_iterators()):
 437       return global_batch_size
 438 
 439     # On TPU
 440     if self.is_input_sharded_per_core() or (
 441         self.is_input_per_host_with_iterators()):
 442       return global_batch_size // self.num_replicas
 443     else:
 444       return global_batch_size // self.num_hosts
 445 
 446   @property
 447   def batch_size_for_model_fn(self):
 448     """Returns the shard batch size for `model_fn`."""
 449     global_batch_size = self.global_batch_size
 450 
 451     if (self.is_running_on_cpu() or self.is_input_broadcast_with_iterators()):
 452       return global_batch_size
 453 
 454     # On TPU. always sharded per shard.
 455     return global_batch_size // self.num_replicas

 457   @property
 458   def master_job(self):
 459     """Returns the job name to use to place TPU computations on.
 460 
 461     Returns:
 462       A string containing the job name, or None if no job should be specified.
 463 
 464     Raises:
 465       ValueError: If the user needs to specify a tpu_job_name, because we are
 466         unable to infer the job name automatically, or if the user-specified job
 467         names are inappropriate.
 468     """
 469     run_config = self._config
 470     # If the user specifies the tpu_job_name, use that.
 471     if run_config.tpu_config.tpu_job_name:
 472       return run_config.tpu_config.tpu_job_name
 473 
 474     # The tpu job is determined by the run_config. Right now, this method is
 475     # required as tpu_config is not part of the RunConfig.
 476     mode = self._assert_mode()
 477     master = (
 478         run_config.evaluation_master
 479         if mode == model_fn_lib.ModeKeys.EVAL else run_config.master)
 480     if master in _LOCAL_MASTERS:
 481       return None
 482 
 483     if (not run_config.session_config or
 484         not run_config.session_config.cluster_def.job):
 485       return _DEFAULT_JOB_NAME
 486     cluster_def = run_config.session_config.cluster_def
 487     job_names = set([job.name for job in cluster_def.job])
 488     if _DEFAULT_JOB_NAME in job_names:
 489       # b/37868888 tracks allowing ClusterSpec propagation to reuse job names.
 490       raise ValueError('Currently, tpu_worker is not an allowed job name.')
 491     if len(job_names) == 1:
 492       return cluster_def.job[0].name
 493     if len(job_names) == 2:
 494       if _DEFAULT_COORDINATOR_JOB_NAME in job_names:
 495         job_names.remove(_DEFAULT_COORDINATOR_JOB_NAME)
 496         return job_names.pop()
 497       # TODO(b/67716447): Include more sophisticated heuristics.
 498     raise ValueError(
 499         'Could not infer TPU job name. Please specify a tpu_job_name as part '
 500         'of your TPUConfig.')
 501 
 502   @property
 503   def tpu_host_placement_function(self):
 504     """Returns the TPU host place function."""
 505 
 506     master = self.master_job
 507 
 508     def _placement_function(_sentinal=None, replica_id=None, host_id=None):  # pylint: disable=invalid-name
 509       """Return the host device given replica_id or host_id."""
 510       assert _sentinal is None
 511       if replica_id is not None and host_id is not None:
 512         raise RuntimeError(
 513             'replica_id and host_id can have only one non-None value.')
 514 
 515       if master is None:
 516         return '/replica:0/task:0/device:CPU:0'
 517       else:
 518         if replica_id is not None:
 519           if self.model_parallelism_enabled:
 520             return self.device_assignment.host_device(
 521                 replica=replica_id, job=master)
 522           else:
 523             host_id = replica_id / self.num_of_cores_per_host
 524 
 525         return '/job:%s/task:%d/device:CPU:0' % (master, host_id)
 526 
 527     return _placement_function
 528 
 529   @property
 530   def tpu_device_placement_function(self):
 531     """Returns a TPU device placement Fn."""
 532     master = self.master_job
 533     job_device = '' if master is None else ('/job:%s' % master)
 534 
 535     def _placement_function(i):
 536       if self.model_parallelism_enabled:
 537         return self.device_assignment.tpu_device(replica=i, job=master)
 538       else:
 539         num_of_cores_per_host = self.num_of_cores_per_host
 540         host_id = i / num_of_cores_per_host
 541         ordinal_id = i % num_of_cores_per_host
 542         return '%s/task:%d/device:TPU:%d' % (job_device, host_id, ordinal_id)
 543 
 544     return _placement_function
 545 
 546   def tpu_ordinal_function(self, host_id):
 547     """Returns the TPU ordinal fn."""
 548 
 549     def _tpu_ordinal_function(shard_index_in_host):
 550       """Return the TPU ordinal associated with a shard.
 551 
 552       Required because the enqueue ops are placed on CPU.
 553 
 554       Args:
 555         shard_index_in_host: the shard index
 556 
 557       Returns:
 558         The ordinal of the TPU device the shard's infeed should be placed on.
 559       """
 560       if self.model_parallelism_enabled:
 561         # We put both enqueue/dequeue ops at tpu.core(0) in each replica.
 562         replica = self.device_assignment.lookup_replicas(host_id,
 563                                                          0)[shard_index_in_host]
 564         return self.device_assignment.tpu_ordinal(replica=replica)
 565       else:
 566         return shard_index_in_host % self.num_of_cores_per_host
 567 
 568     return _tpu_ordinal_function

 574       return
 575 
 576     # All following information is obtained from TPU system metadata.
 577     num_cores = self.num_cores
 578     num_replicas = self.num_replicas
 579     num_hosts = self.num_hosts
 580 
 581     if not num_cores:
 582       tpu_system_metadata = self._get_tpu_system_metadata()
 583       raise RuntimeError(
 584           'Cannot find any TPU cores in the system. Please double check '
 585           'Tensorflow master address and TPU worker(s). Available devices '
 586           'are {}.'.format(tpu_system_metadata.devices))
 587 
 588     if self._config.tpu_config.num_shards:
 589       user_provided_num_replicas = self._config.tpu_config.num_shards
 590       if user_provided_num_replicas != num_replicas:
 591         message = (
 592             'TPUConfig.num_shards is not set correctly. According to TPU '
 593             'system metadata for Tensorflow master ({}): num_replicas should '
 594             'be ({}), got ({}). For non-model-parallelism, num_replicas should '
 595             'be the total num of TPU cores in the system. For '
 596             'model-parallelism, the total number of TPU cores should be '
 597             'num_cores_per_replica * num_replicas. Please set it '
 598             'accordingly or leave it as `None`'.format(
 599                 self._get_master_address(), num_replicas,
 600                 user_provided_num_replicas))
 601 
 602         raise ValueError(message)
 603 
 604     if self._config.tpu_config.num_cores_per_replica:
 605       num_cores_per_replica = self._config.tpu_config.num_cores_per_replica
 606       num_cores_per_host = self._get_tpu_system_metadata().num_of_cores_per_host
 607       if num_cores_per_replica > num_cores_per_host:
 608         raise ValueError(
 609             'The num of cores required by the model parallelism, specified by '
 610             'TPUConfig.num_cores_per_replica, is larger than the '
 611             'num_cores_per_host. num_cores_per_replica: {}, '
 612             'num_cores_per_host: {}'.format(num_cores_per_replica,
 613                                             num_cores_per_host))
 614 
 615     if mode == model_fn_lib.ModeKeys.TRAIN:
 616       if (self._train_batch_size % num_replicas != 0 and
 617           not self.is_input_broadcast_with_iterators()):
 618         raise ValueError(
 619             'train batch size {} must be divisible by number of replicas {}'
 620             .format(self._train_batch_size, num_replicas))
 621 
 622     elif mode == model_fn_lib.ModeKeys.EVAL:
 623       if self._eval_batch_size is None:
 624         raise ValueError(
 625             'eval_batch_size in TPUEstimator constructor cannot be `None`'
 626             'if .evaluate is running on TPU.')
 627       if (self._eval_batch_size % num_replicas != 0 and
 628           not self.is_input_broadcast_with_iterators()):
 629         raise ValueError(
 630             'eval batch size {} must be divisible by number of replicas {}'
 631             .format(self._eval_batch_size, num_replicas))
 632       if num_hosts > 1 and not self.is_input_broadcast_with_iterators():
 633         raise ValueError(
 634             'TPUEstimator.evaluate should be running on single TPU'
 635             ' instead of a Pod.')
 636     else:
 637       assert mode == model_fn_lib.ModeKeys.PREDICT
 638       if self._predict_batch_size is None:
 639         raise ValueError(
 640             'predict_batch_size in TPUEstimator constructor should not be '
 641             '`None` if .predict is running on TPU.')
 642       if (self._predict_batch_size % num_replicas != 0 and
 643           not self.is_input_broadcast_with_iterators()):
 644         raise ValueError(
 645             'predict batch size {} must be divisible by number of replicas {}'
 646             .format(self._predict_batch_size, num_replicas))
 647       if num_hosts > 1 and not self.is_input_broadcast_with_iterators():
 648         raise ValueError(
 649             'TPUEstimator.predict should be running on single TPU worker. '
 650             'got {}.'.format(num_hosts))
 651 
 652     # Record the state "validated" into lazy dictionary.
 653     self._lazy_validation_dict[mode] = True
 654 
 655   def device_for_replica(self, replica_id):
 656     """Returns the tuple of (CPU device and device ordinal) for replica.
 657 
 658     This should be used for full replicate for non-model-parallelism.
 659 
 660     Args:
 661        replica_id: Int, the replica index.
 662 
 663     Returns:
 664        A tuple of device spec for CPU device and int device ordinal.
 665     """
 666     master = self.master_job
 667 
 668     if self.model_parallelism_enabled:
 669       return (self.device_assignment.host_device(
 670           replica=replica_id, job=master),
 671               self.device_assignment.tpu_ordinal(replica=replica_id))
 672 
 673     job_device = '' if master is None else ('/job:%s' % master)
 674 
 675     num_of_replicas_per_host = self.num_of_replicas_per_host
 676     host_id = replica_id / num_of_replicas_per_host
 677     ordinal_id = replica_id % num_of_replicas_per_host
 678 
 679     host_device = '%s/task:%d/device:CPU:0' % (job_device, host_id)
 680     return (host_device, ordinal_id)

 683 class _OneCoreTPUContext(_InternalTPUContext):
 684   """Special _InternalTPUContext for one core usage."""
 685 
 686   def __init__(self, config, train_batch_size, eval_batch_size,
 687                predict_batch_size, use_tpu):
 688 
 689     super(_OneCoreTPUContext, self).__init__(
 690         config, train_batch_size, eval_batch_size,
 691         predict_batch_size, use_tpu)
 692 
 693   def _get_tpu_system_metadata(self):
 694     """Gets the (maybe cached) TPU system metadata."""
 695     master = self._get_master_address()
 696     tpu_system_metadata = self._lazy_tpu_system_metadata_dict.get(master)
 697     if tpu_system_metadata is not None:
 698       return tpu_system_metadata
 699 
 700     tpu_system_metadata = (
 701         tpu_system_metadata_lib._TPUSystemMetadata(  # pylint: disable=protected-access
 702             num_cores=1,
 703             num_hosts=1,
 704             num_of_cores_per_host=1,
 705             topology=None,
 706             devices=[]))
 707 
 708     self._lazy_tpu_system_metadata_dict[master] = tpu_system_metadata
 709     return tpu_system_metadata

 712 def _get_tpu_context(config, train_batch_size, eval_batch_size,
 713                      predict_batch_size, use_tpu, eval_on_tpu):
 714   """Returns an instance of `_InternalTPUContext`."""
 715 
 716   if (config.tpu_config.num_shards == 1 and
 717       config.tpu_config.num_cores_per_replica is None):
 718     logging.warning(
 719         'Setting TPUConfig.num_shards==1 is an unsupported behavior. '
 720         'Please fix as soon as possible (leaving num_shards as None.)')
 721     return _OneCoreTPUContext(config, train_batch_size, eval_batch_size,
 722                               predict_batch_size, use_tpu)
 723 
 724   return _InternalTPUContext(config, train_batch_size, eval_batch_size,
 725                              predict_batch_size, use_tpu, eval_on_tpu)
