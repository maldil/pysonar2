  27 class TpuContext(object):
  28   """A context object holding state about the TPU computation being built."""
  29 
  30   def __init__(self):
  31     """Creates a new TpuContext."""
  32     self._number_of_shards = None
  33 
  34   @property
  35   def number_of_shards(self):
  36     return self._number_of_shards
  37 
  38   def set_number_of_shards(self, number_of_shards):
  39     self._number_of_shards = number_of_shards
  40 
  41 
  42 # The Tpu context holds the number of shards when a sharded computation is
  43 # being built, or None if no computation is being built.
  44 _current_tpu_context = TpuContext()
  45 
  46 
  47 @contextlib.contextmanager
  48 def tpu_shard_context(number_of_shards):
  49   if _current_tpu_context.number_of_shards is not None:
  50     raise NotImplementedError("tpu_shard_context cannot be nested.")
  51   try:
  52     _current_tpu_context.set_number_of_shards(number_of_shards)
  53     yield
  54   finally:
  55     _current_tpu_context.set_number_of_shards(None)
  56 
  57 
  58 def get_tpu_context():
  59   return _current_tpu_context

  62 def check_function_argument_count(func, input_arity, infeed_queue):
  63   """Validate the number of input arguments to a tpu function.
  64 
  65   Args:
  66     func: the Python function that will be called to generate the body of an XLA
  67       computation graph.
  68     input_arity: the number of explicit arguments supplied by the caller.
  69     infeed_queue: if not None, the infeed queue that will supply
  70       additional arguments to the function.
  71 
  72   Returns:
  73     None if function can be called with the supplied number of
  74       arguments, or an error string if it cannot.
  75   """
  76   def format_error(complaint, quantity):
  77     return "%s %d argument%s" % (complaint, quantity, ""
  78                                  if quantity == 1 else "s")
  79 
  80   number_of_arguments_needed = input_arity
  81   if infeed_queue is not None:
  82     number_of_arguments_needed += infeed_queue.number_of_tuple_elements
  83   arg_spec = tf_inspect.getargspec(func)
  84   number_of_args = len(arg_spec.args)
  85   if arg_spec.defaults is None:
  86     number_of_defaults = 0
  87   else:
  88     number_of_defaults = len(arg_spec.defaults)
  89   min_required_arguments = number_of_args - number_of_defaults
  90   if number_of_arguments_needed < min_required_arguments:
  91     # The required number of arguments is not enough to call the function.
  92     if number_of_defaults == 0 and arg_spec.varargs is None:
  93       return format_error("exactly", number_of_args)
  94     else:
  95       return format_error("at least", min_required_arguments)
  96   if arg_spec.varargs is None and number_of_arguments_needed > number_of_args:
  97     # The required number of arguments is too many to call the function.
  98     if number_of_defaults == 0:
  99       return format_error("exactly", number_of_args)
 100     else:
 101       return format_error("at most", number_of_args)
 102   # Since there are varargs, func can accept any number of arguments
 103   # greater than the minimum.
 104   return None
