  33 class Readable(object):
  34   """Readable abstract class that exposes methods to do reading-related
  35 
  36      operations.
  37   """
  38 
  39   @abc.abstractmethod
  40   def __init__(self):
  41     pass
  42 
  43   def read_byte(self):
  44     """Reads and returnes byte."""
  45     return self._read("b", 1)
  46 
  47   def read_short(self):
  48     """Reads and returns short (2 bytes, little-endian)."""
  49     return self._read("h", 2)
  50 
  51   def read_int(self):
  52     """Reads and returns int (4 bytes, little-endian)."""
  53     return self._read("i", 4)
  54 
  55   def read_long(self):
  56     """Reads and returns long (8 bytes, little-endian)."""
  57     return self._read("q", 8)
  58 
  59   def skip(self, length):
  60     """Skips the specified number of bytes."""
  61     self.read_data(length)
  62 
  63   @abc.abstractmethod
  64   def read_data(self, length):
  65     """Reads the specified number of bytes and returns them as a buffer."""
  66     return None
  67 
  68   def _read(self, data_type, length):
  69     """Reads, unpacks and returns specified type (little-endian)."""
  70     data_buffer = self.read_data(length)
  71     return struct.unpack("<" + data_type, data_buffer)[0]

  74 class DataBuffer(Readable):
  75   """DataBuffer class that exposes methods to read data from a byte buffer."""
  76 
  77   def __init__(self, data_buffer):
  78     """Constructs a new instance based on the specified byte  buffer.
  79 
  80     Args:
  81       data_buffer: Buffer to be read.
  82     """
  83     Readable.__init__(self)
  84     self.buffer = data_buffer
  85     self.ptr = 0
  86 
  87   def read_data(self, length):
  88     """Reads the specified number of bytes and returns them as a buffer."""
  89     data_buffer = self.buffer[self.ptr:][:length]
  90     self.ptr += length
  91     return data_buffer

  94 class TcpClient(Readable):
  95   """TcpClient class that exposes methods to read data from a socket."""
  96 
  97   def __init__(self, host, port, certfile=None, keyfile=None, password=None):
  98     """Constructs a new instance based on the specified host and port.
  99 
 100     Args:
 101       host: Host to be connected.
 102       port: Port to be connected.
 103       certfile: File in PEM format containing the certificate as well as any
 104         number of CA certificates needed to establish the certificate's
 105         authenticity.
 106       keyfile: File containing the private key (otherwise the private key will
 107         be taken from certfile as well).
 108       password: Password to be used if the private key is encrypted and a
 109         password is necessary.
 110 
 111     Raises:
 112       ValueError: If the wrong combination of arguments is provided.
 113     """
 114     Readable.__init__(self)
 115     self.sock = socket.socket(socket.AF_INET, socket.SOCK_STREAM)
 116 
 117     if certfile is not None:
 118       context = ssl.SSLContext(ssl.PROTOCOL_SSLv23)
 119       context.load_cert_chain(certfile, keyfile, password)
 120       self.sock = context.wrap_socket(self.sock)
 121     else:
 122       if keyfile is not None:
 123         raise ValueError("SSL is disabled, keyfile must not be specified "
 124                          "(to enable SSL specify certfile)")
 125       if password is not None:
 126         raise ValueError("SSL is disabled, password must not be specified "
 127                          "(to enable SSL specify certfile)")
 128 
 129     self.host = host
 130     self.port = port
 131 
 132   def __enter__(self):
 133     """Connects to host and port specified in the constructor."""
 134     self.sock.connect((self.host, self.port))
 135     return self
 136 
 137   def __exit__(self, t, v, traceback):
 138     """Disconnects the socket."""
 139     self.sock.close()
 140 
 141   def write_byte(self, v):
 142     """Writes the specified byte."""
 143     self._write(v, "b")
 144 
 145   def write_short(self, v):
 146     """Writes the specified short (2 bytes, little-endian)."""
 147     self._write(v, "h")
 148 
 149   def write_int(self, v):
 150     """Writes the specified short (4 bytes, little-endian)."""
 151     self._write(v, "i")
 152 
 153   def write_long(self, v):
 154     """Writes the specified int (8 bytes, little-endian)."""
 155     self._write(v, "q")
 156 
 157   def write_string(self, v):
 158     """Writes the specified string."""
 159     self.sock.sendall(v.encode("UTF-8"))
 160 
 161   def read_data(self, length):
 162     """Reads the specified number of bytes and returns them as a buffer."""
 163     data_buffer = None
 164     rem = length
 165     while rem > 0:
 166       buf = self.sock.recv(rem)
 167       rem = rem - len(buf)
 168       if data_buffer is None:
 169         data_buffer = buf
 170       else:
 171         data_buffer += buf
 172     return data_buffer
 173 
 174   def _write(self, value, data_type):
 175     """Packs and writes data using the specified type (little-endian)."""
 176     data_buffer = struct.pack("<" + data_type, value)
 177     self.sock.sendall(data_buffer)

 226 class TypeTreeNode(object):
 227   """TypeTreeNode class exposes methods to format object tree structure
 228 
 229      data.
 230   """
 231 
 232   def __init__(self, name, type_id, fields=None, permutation=None):
 233     """Constructs a new instance of TypeTreeNode.
 234 
 235     Args:
 236       name: Name of the object tree node.
 237       type_id: Type id of the object tree node.
 238       fields: List of fields (children of the object tree node).
 239       permutation: Permutation that should be applied to order object children.
 240     """
 241     self.name = name
 242     self.type_id = type_id
 243     self.fields = fields
 244     self.permutation = permutation
 245 
 246   def to_output_classes(self):
 247     """Formats the tree object as required by `Dataset.output_classes`."""
 248     if self.fields is None:
 249       return ops.Tensor
 250     output_classes = {}
 251     for field in self.fields:
 252       output_classes[field.name] = field.to_output_classes()
 253     return output_classes
 254 
 255   def to_output_shapes(self):
 256     """Formats the tree object as required by `Dataset.output_shapes`."""
 257     if self.fields is None:
 258       if self.type_id in types:
 259         object_type = types[self.type_id]
 260         is_array = object_type[1]
 261         if is_array:
 262           return tensor_shape.TensorShape([None])
 263         return tensor_shape.TensorShape([])
 264       raise ValueError("Unsupported type [type_id=%d]" % self.type_id)
 265     output_shapes = {}
 266     for field in self.fields:
 267       output_shapes[field.name] = field.to_output_shapes()
 268     return output_shapes
 269 
 270   def to_output_types(self):
 271     """Formats the tree object as required by `Dataset.output_types`."""
 272     if self.fields is None:
 273       if self.type_id in types:
 274         object_type = types[self.type_id]
 275         return object_type[0]
 276       raise ValueError("Unsupported type [type_id=%d]" % self.type_id)
 277     else:
 278       output_types = {}
 279       for field in self.fields:
 280         output_types[field.name] = field.to_output_types()
 281       return output_types
 282 
 283   def to_flat(self):
 284     """Returns a list of node types."""
 285     return self.to_flat_rec([])
 286 
 287   def to_permutation(self):
 288     """Returns a permutation that should be applied to order object leaves."""
 289     correct_order_dict = {}
 290     self.traversal_rec(correct_order_dict, 0)
 291     object_order = []
 292     self.traversal_permutation_rec(object_order)
 293     return [correct_order_dict[o] for o in object_order]
 294 
 295   def to_flat_rec(self, flat):
 296     """Formats a list of leaf node types in pre-order."""
 297     if self.fields is None:
 298       flat.append(self.type_id)
 299     else:
 300       for field in self.fields:
 301         field.to_flat_rec(flat)
 302     return flat
 303 
 304   def traversal_permutation_rec(self, permutation):
 305     """Collects nodes in accordance with permutation."""
 306     if self.fields is None:
 307       permutation.append(self)
 308     else:
 309       for idx in self.permutation:
 310         field = self.fields[idx]
 311         field.traversal_permutation_rec(permutation)
 312 
 313   def traversal_rec(self, d, i):
 314     """Collects nodes in pre-order traversal."""
 315     if self.fields is None:
 316       d[self] = i
 317       i += 1
 318     else:
 319       for field in self.fields:
 320         i = field.traversal_rec(d, i)
 321     return i

 324 class IgniteClient(TcpClient):
 325   """IgniteClient enables working with Apache Ignite using a thin client.
 326 
 327   This client works with assumption that all object in the cache
 328   have the same structure (homogeneous objects) and the cache contains at
 329   least one object.
 330   """
 331 
 332   def __init__(self,
 333                host,
 334                port,
 335                username=None,
 336                password=None,
 337                certfile=None,
 338                keyfile=None,
 339                cert_password=None):
 340     """Constructs a new instance of IgniteClient.
 341 
 342     Args:
 343       host: Apache Ignite Thin client host to be connected.
 344       port: Apache Ignite Thin client port to be connected.
 345       username: Apache Ignite Thin Client authentication username.
 346       password: Apache Ignite Thin Client authentication password.
 347       certfile: File in PEM format containing the certificate as well as any
 348         number of CA certificates needed to establish the certificate's
 349         authenticity.
 350       keyfile: File containing the private key (otherwise the private key will
 351         be taken from certfile as well).
 352       cert_password: Password to be used if the private key is encrypted and a
 353         password is necessary.
 354     """
 355     TcpClient.__init__(self, host, port, certfile, keyfile, cert_password)
 356     self.username = username
 357     self.password = password
 358 
 359   def handshake(self):
 360     """Makes a handshake  after connect and before any other calls."""
 361     msg_len = 8
 362 
 363     if self.username is None:
 364       msg_len += 1
 365     else:
 366       msg_len += 5 + len(self.username)
 367 
 368     if self.password is None:
 369       msg_len += 1
 370     else:
 371       msg_len += 5 + len(self.password)
 372 
 373     self.write_int(msg_len)  # Message length
 374     self.write_byte(1)  # Handshake operation
 375     self.write_short(1)  # Version (1.1.0)
 376     self.write_short(1)
 377     self.write_short(0)
 378     self.write_byte(2)  # Thin client
 379 
 380     if self.username is None:  # Username
 381       self.write_byte(101)
 382     else:
 383       self.write_byte(9)
 384       self.write_int(len(self.username))
 385       self.write_string(self.username)
 386 
 387     if self.password is None:  # Password
 388       self.write_byte(101)
 389     else:
 390       self.write_byte(9)
 391       self.write_int(len(self.password))
 392       self.write_string(self.password)
 393 
 394     self.read_int()  # Result length
 395     res = self.read_byte()
 396 
 397     if res != 1:
 398       serv_ver_major = self.read_short()
 399       serv_ver_minor = self.read_short()
 400       serv_ver_patch = self.read_short()
 401       err_msg = self._parse_string()
 402       if err_msg is None:
 403         raise RuntimeError(
 404             "Handshake Error [result=%d, version=%d.%d.%d]" %
 405             (res, serv_ver_major, serv_ver_minor, serv_ver_patch))
 406       else:
 407         raise RuntimeError(
 408             "Handshake Error [result=%d, version=%d.%d.%d, message='%s']" %
 409             (res, serv_ver_major, serv_ver_minor, serv_ver_patch, err_msg))
 410 
 411   def get_cache_type(self, cache_name):
 412     """Collects type information about objects stored in the specified cache."""
 413     cache_name_hash = self._java_hash_code(cache_name)
 414     self.write_int(25)  # Message length
 415     self.write_short(2000)  # Operation code
 416     self.write_long(0)  # Request ID
 417     self.write_int(cache_name_hash)  # Cache name
 418     self.write_byte(0)  # Flags
 419     self.write_byte(101)  # Filter (NULL)
 420     self.write_int(1)  # Cursor page size
 421     self.write_int(-1)  # Partition to query
 422     self.write_byte(0)  # Local flag
 423 
 424     result_length = self.read_int()
 425     self.read_long()  # Request id
 426     status = self.read_int()
 427 
 428     if status != 0:
 429       err_msg = self._parse_string()
 430       if err_msg is None:
 431         raise RuntimeError("Scan Query Error [status=%s]" % status)
 432       else:
 433         raise RuntimeError(
 434             "Scan Query Error [status=%s, message='%s']" % (status, err_msg))
 435 
 436     self.read_long()  # Cursor id
 437     row_count = self.read_int()
 438 
 439     if row_count == 0:
 440       raise RuntimeError("Scan Query returned empty result, so it's "
 441                          "impossible to derive the cache type")
 442 
 443     payload = DataBuffer(self.read_data(result_length - 25))
 444 
 445     self.read_byte()  # Next page
 446 
 447     res = TypeTreeNode("root", 0, [
 448         self._collect_types("key", payload),
 449         self._collect_types("val", payload)
 450     ], [0, 1])
 451 
 452     return res
 453 
 454   def _java_hash_code(self, s):
 455     """Computes hash code of the specified string using Java code."""
 456     h = 0
 457     for c in s:
 458       h = (31 * h + ord(c)) & 0xFFFFFFFF
 459     return ((h + 0x80000000) & 0xFFFFFFFF) - 0x80000000
 460 
 461   def _collect_types(self, field_name, data):
 462     """Extracts type information from the specified object."""
 463     type_id = data.read_byte()
 464 
 465     # Byte scalar.
 466     if type_id == 1:
 467       data.skip(1)
 468       return TypeTreeNode(field_name, type_id)
 469 
 470     # Short scalar.
 471     if type_id == 2:
 472       data.skip(2)
 473       return TypeTreeNode(field_name, type_id)
 474 
 475     # Integer scalar.
 476     if type_id == 3:
 477       data.skip(4)
 478       return TypeTreeNode(field_name, type_id)
 479 
 480     # Long scalar.
 481     if type_id == 4:
 482       data.skip(8)
 483       return TypeTreeNode(field_name, type_id)
 484 
 485     # Float scalar.
 486     if type_id == 5:
 487       data.skip(4)
 488       return TypeTreeNode(field_name, type_id)
 489 
 490     # Double scalar.
 491     if type_id == 6:
 492       data.skip(8)
 493       return TypeTreeNode(field_name, type_id)
 494 
 495     # Char scalar.
 496     if type_id == 7:
 497       data.skip(2)
 498       return TypeTreeNode(field_name, type_id)
 499 
 500     # Bool scalar.
 501     if type_id == 8:
 502       data.skip(1)
 503       return TypeTreeNode(field_name, type_id)
 504 
 505     # String scalar.
 506     if type_id == 9:
 507       length = data.read_int()
 508       data.skip(length)
 509       return TypeTreeNode(field_name, type_id)
 510 
 511     # UUID scalar.
 512     if type_id == 10:
 513       data.skip(16)
 514       return TypeTreeNode(field_name, type_id)
 515 
 516     # Date scalar.
 517     if type_id == 11:
 518       data.skip(8)
 519       return TypeTreeNode(field_name, type_id)
 520 
 521     # Byte array.
 522     if type_id == 12:
 523       length = data.read_int()
 524       data.skip(length)
 525       return TypeTreeNode(field_name, type_id)
 526 
 527     # Short array.
 528     if type_id == 13:
 529       length = data.read_int()
 530       data.skip(length * 2)
 531       return TypeTreeNode(field_name, type_id)
 532 
 533     # Integer array.
 534     if type_id == 14:
 535       length = data.read_int()
 536       data.skip(length * 4)
 537       return TypeTreeNode(field_name, type_id)
 538 
 539     # Long array.
 540     if type_id == 15:
 541       length = data.read_int()
 542       data.skip(length * 8)
 543       return TypeTreeNode(field_name, type_id)
 544 
 545     # Float array.
 546     if type_id == 16:
 547       length = data.read_int()
 548       data.skip(length * 4)
 549       return TypeTreeNode(field_name, type_id)
 550 
 551     # Double array.
 552     if type_id == 17:
 553       length = data.read_int()
 554       data.skip(length * 8)
 555       return TypeTreeNode(field_name, type_id)
 556 
 557     # Char array.
 558     if type_id == 18:
 559       length = data.read_int()
 560       data.skip(length * 2)
 561       return TypeTreeNode(field_name, type_id)
 562 
 563     # Bool array.
 564     if type_id == 19:
 565       length = data.read_int()
 566       data.skip(length)
 567       return TypeTreeNode(field_name, type_id)
 568 
 569     # String array.
 570     if type_id == 20:
 571       length = data.read_int()
 572       for _ in range(length):
 573         header = data.read_byte()
 574         if header == 9:
 575           str_length = data.read_int()
 576           data.skip(str_length)
 577         elif header == 101:
 578           pass
 579         else:
 580           raise RuntimeError(
 581               "Unknown binary type when expected string [type_id=%d]" % header)
 582       return TypeTreeNode(field_name, type_id)
 583 
 584     # UUID array.
 585     if type_id == 21:
 586       length = data.read_int()
 587       data.skip(length * 16)  # TODO(dmitrievanthony): support NULL values.
 588       return TypeTreeNode(field_name, type_id)
 589 
 590     # Date array.
 591     if type_id == 22:
 592       length = data.read_int()
 593       data.skip(length * 8)
 594       return TypeTreeNode(field_name, type_id)
 595 
 596     # Wrapped Binary Object.
 597     if type_id == 27:
 598       length = data.read_int()
 599       inner_data = data.read_data(length)
 600       data.read_int()  # Offset
 601       return self._collect_types(field_name, DataBuffer(inner_data))
 602 
 603     # Complex Object.
 604     if type_id == 103:
 605       data.read_byte()  # Object version
 606       data.read_short()  # Object flags
 607       obj_type_id = data.read_int()
 608       data.read_int()  # Object hash code
 609       obj_length = data.read_int()
 610       data.read_int()  # Object schema id
 611       obj_schema_offset = data.read_int()
 612 
 613       obj_type = self._get_type(obj_type_id)
 614       children = []
 615 
 616       for obj_field in obj_type.fields:
 617         child = self._collect_types(obj_field.field_name, data)
 618         children.append(child)
 619 
 620       children_sorted = sorted(children, key=lambda child: child.name)
 621       permutation = [children_sorted.index(child) for child in children]
 622       children = children_sorted
 623 
 624       data.skip(obj_length - obj_schema_offset)
 625 
 626       return TypeTreeNode(field_name, type_id, children, permutation)
 627 
 628     raise RuntimeError("Unknown binary type [type_id=%d]" % type_id)
 629 
 630   def _get_type(self, type_id):
 631     """Queries Apache Ignite information about type by type id."""
 632     self.write_int(14)  # Message length
 633     self.write_short(3002)  # Operation code
 634     self.write_long(0)  # Request ID
 635     self.write_int(type_id)  # Type ID
 636 
 637     self.read_int()  # Result length
 638     self.read_long()  # Request id
 639     status = self.read_int()
 640 
 641     if status != 0:
 642       err_msg = self._parse_string()
 643       if err_msg is None:
 644         raise RuntimeError("Get Binary Type Error [status=%d, message='%s']" %
 645                            (status, err_msg))
 646       else:
 647         raise RuntimeError("Get Binary Type Error [status=%d]" % status)
 648 
 649     binary_type_exists = self.read_byte()
 650 
 651     if binary_type_exists == 0:
 652       raise RuntimeError("Binary type not found [type_id=%d] " % type_id)
 653 
 654     binary_type_id = self.read_int()
 655     binary_type_name = self._parse_string()
 656     self._parse_string()  # Affinity field name
 657 
 658     fields = []
 659     for _ in range(self.read_int()):
 660       field_name = self._parse_string()
 661       field_type_id = self.read_int()
 662       field_id = self.read_int()
 663 
 664       field = BinaryField(field_name, field_type_id, field_id)
 665       fields.append(field)
 666 
 667     is_enum = self.read_byte()
 668     if is_enum == 1:
 669       raise RuntimeError("Enum fields are not supported yet")
 670 
 671     schema_cnt = self.read_int()
 672     for _ in range(schema_cnt):
 673       self.read_int()  # Schema id
 674       field_cnt = self.read_int()
 675       self.skip(field_cnt * 4)
 676 
 677     return BinaryType(binary_type_id, binary_type_name, fields)
 678 
 679   def _parse_string(self):
 680     """Parses string."""
 681     header = self.read_byte()
 682     if header == 9:
 683       length = self.read_int()
 684       return self.read_data(length).decode("utf-8")
 685     if header == 101:
 686       return None
 687     raise RuntimeError(
 688         "Unknown binary type when expected string [type_id=%d]" % header)

 691 class IgniteDataset(dataset_ops.Dataset):
 692   """Apache Ignite is a memory-centric distributed database, caching, and
 693 
 694      processing platform for transactional, analytical, and streaming workloads,
 695      delivering in-memory speeds at petabyte scale. This contrib package
 696      contains an integration between Apache Ignite and TensorFlow. The
 697      integration is based on tf.data from TensorFlow side and Binary Client
 698      Protocol from Apache Ignite side. It allows to use Apache Ignite as a
 699      datasource for neural network training, inference and all other
 700      computations supported by TensorFlow. Ignite Dataset is based on Apache
 701      Ignite Binary Client Protocol.
 702   """
 703 
 704   def __init__(self,
 705                cache_name,
 706                host="localhost",
 707                port=10800,
 708                local=False,
 709                part=-1,
 710                page_size=100,
 711                username=None,
 712                password=None,
 713                certfile=None,
 714                keyfile=None,
 715                cert_password=None):
 716     """Create a IgniteDataset.
 717 
 718     Args:
 719       cache_name: Cache name to be used as datasource.
 720       host: Apache Ignite Thin Client host to be connected.
 721       port: Apache Ignite Thin Client port to be connected.
 722       local: Local flag that defines to query only local data.
 723       part: Number of partitions to be queried.
 724       page_size: Apache Ignite Thin Client page size.
 725       username: Apache Ignite Thin Client authentication username.
 726       password: Apache Ignite Thin Client authentication password.
 727       certfile: File in PEM format containing the certificate as well as any
 728         number of CA certificates needed to establish the certificate's
 729         authenticity.
 730       keyfile: File containing the private key (otherwise the private key will
 731         be taken from certfile as well).
 732       cert_password: Password to be used if the private key is encrypted and a
 733         password is necessary.
 734     """
 735     super(IgniteDataset, self).__init__()
 736 
 737     with IgniteClient(host, port, username, password, certfile, keyfile,
 738                       cert_password) as client:
 739       client.handshake()
 740       self.cache_type = client.get_cache_type(cache_name)
 741 
 742     self.cache_name = ops.convert_to_tensor(
 743         cache_name, dtype=dtypes.string, name="cache_name")
 744     self.host = ops.convert_to_tensor(host, dtype=dtypes.string, name="host")
 745     self.port = ops.convert_to_tensor(port, dtype=dtypes.int32, name="port")
 746     self.local = ops.convert_to_tensor(local, dtype=dtypes.bool, name="local")
 747     self.part = ops.convert_to_tensor(part, dtype=dtypes.int32, name="part")
 748     self.page_size = ops.convert_to_tensor(
 749         page_size, dtype=dtypes.int32, name="page_size")
 750     self.schema = ops.convert_to_tensor(
 751         self.cache_type.to_flat(), dtype=dtypes.int32, name="schema")
 752     self.permutation = ops.convert_to_tensor(
 753         self.cache_type.to_permutation(),
 754         dtype=dtypes.int32,
 755         name="permutation")
 756 
 757   def _as_variant_tensor(self):
 758     return gen_dataset_ops.ignite_dataset(self.cache_name, self.host, self.port,
 759                                           self.local, self.part, self.page_size,
 760                                           self.schema, self.permutation)
