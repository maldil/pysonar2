  39 class ElasticAverageCustomGetter(object):
  40   """Custom_getter class is used to do:
  41   1. Change trainable variables to local collection and place them at worker
  42     device
  43   2. Generate global variables(global center variables)
  44   3. Generate local variables(local center variables) which record the global
  45     variables and place them at worker device
  46     Notice that the class should be used with tf.replica_device_setter,
  47     so that the global center variables and global step variable can be placed
  48     at ps device. Besides, use 'tf.get_variable' instead of 'tf.Variable' to
  49     use this custom getter.
  50 
  51   For example,
  52   ea_custom_getter = ElasticAverageCustomGetter(worker_device)
  53   with tf.device(
  54     tf.train.replica_device_setter(
  55       worker_device=worker_device,
  56       ps_device="/job:ps",
  57       cluster=cluster)),
  58     tf.variable_scope('',custom_getter=ea_custom_getter):
  59     ...
  60     create your model here
  61     ...
  62     with tf.device(worker_device):
  63       opt = tf.train.MomentumOptimizer(...)
  64       optimizer = ElasticAverageOptimizer(
  65             opt,
  66             num_worker=2,
  67             moving_rate=0.01, # or use default value
  68             communication_period=20,
  69             ea_custom_getter=ea_custom_getter)
  70       ...
  71       train_op = optimizer.apply_gradients(
  72         grads_vars,
  73         global_step=global_step)
  74     ...
  75     hooks = [optimizer.make_session_run_hook(is_chief, task_index)]
  76     ...
  77     with tf.train.MonitoredTrainingSession(master=server.target,
  78                                            is_chief=is_chief,
  79                                            checkpoint_dir=("...),
  80                                            save_checkpoint_secs=600,
  81                                            hooks=hooks) as mon_sess:
  82   """
  83 
  84   def __init__(self, worker_device):
  85     """Create a new `ElasticAverageCustomGetter`.
  86 
  87     Args:
  88       worker_device: String.  Name of the `worker` job.
  89     """
  90     self._worker_device = worker_device
  91     self._local_map = {}
  92     self._global_map = {}
  93 
  94   def __call__(self, getter, name, trainable, collections, *args, **kwargs):
  95     if trainable:
  96       with ops.device(self._worker_device):
  97         local_var = getter(
  98             name,
  99             trainable=True,
 100             collections=[ops.GraphKeys.LOCAL_VARIABLES],
 101             *args,
 102             **kwargs)
 103       if kwargs['reuse'] == True:
 104         return local_var
 105       global_center_variable = getter(
 106           name='%s/%s' % (GLOBAL_VARIABLE_NAME, name),
 107           trainable=False,
 108           collections=[ops.GraphKeys.GLOBAL_VARIABLES],
 109           *args,
 110           **kwargs)
 111 
 112       with ops.device(self._worker_device):
 113         local_center_variable = getter(
 114             name='%s/%s' % (LOCAL_VARIABLE_NAME, name),
 115             trainable=False,
 116             collections=[ops.GraphKeys.LOCAL_VARIABLES],
 117             *args,
 118             **kwargs)
 119       if kwargs['partitioner'] is None:
 120         self._local_map[local_var] = local_center_variable
 121         self._global_map[local_var] = global_center_variable
 122       else:
 123         v_list = list(local_var)
 124         for i in range(len(v_list)):
 125           self._local_map[v_list[i]] \
 126               = list(local_center_variable)[i]
 127           self._global_map[v_list[i]] \
 128               = list(global_center_variable)[i]
 129       return local_var
 130     else:
 131       kwargs['trainable'] = trainable
 132       kwargs['collections'] = collections
 133       if ops.GraphKeys.LOCAL_VARIABLES in collections:
 134         with ops.device(self._worker_device):
 135           return getter(name, *args, **kwargs)
 136       else:
 137         return getter(name, *args, **kwargs)

 141 class ElasticAverageOptimizer(optimizer.Optimizer):
 142   """Wrapper optimizer that implements the Elastic Average SGD algorithm.
 143   This is an async optimizer. During the training, Each worker will update
 144   the local variables and maintains its own local_step, which starts from 0
 145   and is incremented by 1 after each update of local variables. Whenever
 146   the communication period divides the local step, the worker requests
 147   the current global center variables and then computed the elastic difference
 148   between global center variables and local variables. The elastic difference
 149   then be used to update both local variables and global variables.
 150   """
 151 
 152   # Default value as paper described
 153   BETA = 0.9
 154 
 155   def __init__(self,
 156                opt,
 157                num_worker,
 158                ea_custom_getter,
 159                communication_period=10,
 160                moving_rate=None,
 161                rho=None,
 162                use_locking=True,
 163                synchronous=False,
 164                name='ElasticAverageOptimizer'):
 165     """Construct a new gradient descent optimizer.
 166 
 167     Args:
 168       opt: The actual optimizer that will be used to update local variables.
 169         Must be one of the Optimizer classes.
 170       num_worker: The number of workers
 171       ea_custom_getter: The ElasticAverageCustomGetter
 172       communication_period: An int point value to controls the frequency
 173         of the communication between every worker and the ps.
 174       moving_rate: A floating point value to control the elastic difference.
 175       rho: the amount of exploration we allow in the model. The default
 176         value is moving_rate/learning_rate
 177         rho=0.0 is suggested in async mode.
 178       use_locking: If True use locks for update operations.
 179       synchronous: Add_sync_queues_and_barrier or not.
 180               True: all workers will wait for each other before start training
 181               False: worker can start training when its initilization is done,
 182                      no need to wait for everyone is ready.
 183                      in case one worker is restarted, it can join and continue
 184                      training without being blocked.
 185       name: Optional name prefix for the operations created when applying
 186         gradients. Defaults to "ElasticAverageOptimizer".
 187     """
 188     super(ElasticAverageOptimizer, self).__init__(use_locking, name)
 189     self._opt = opt
 190     self._num_worker = num_worker
 191     self._period = communication_period
 192     self._local_map = ea_custom_getter._local_map
 193     self._global_map = ea_custom_getter._global_map
 194     self._synchronous = synchronous
 195 
 196     if moving_rate is None:
 197       self._moving_rate = self.BETA / communication_period / num_worker
 198     else:
 199       self._moving_rate = moving_rate
 200     if rho is None:
 201       self._rho = self._moving_rate / self._opt._learning_rate
 202     else:
 203       self._rho = rho
 204 
 205     self._local_step = variable_scope.get_variable(
 206         initializer=0,
 207         trainable=False,
 208         collections=[ops.GraphKeys.LOCAL_VARIABLES],
 209         name='local_step')
 210     self._opt._prepare()
 211 
 212   def compute_gradients(self,
 213                         loss,
 214                         var_list=None,
 215                         gate_gradients=optimizer.Optimizer.GATE_OP,
 216                         aggregation_method=None,
 217                         colocate_gradients_with_ops=False,
 218                         grad_loss=None):
 219     """Compute gradients of `loss` for the variables in `var_list`.
 220 
 221     Add rho*elastic_difference to loss to control the exploration
 222     This is the first part of `minimize()`.  It returns a list
 223     of (gradient, variable) pairs where "gradient" is the gradient
 224     for "variable".  Note that "gradient" can be a `Tensor`, an
 225     `IndexedSlices`, or `None` if there is no gradient for the
 226     given variable.
 227 
 228     Args:
 229       loss: A Tensor containing the value to minimize.
 230       var_list: Optional list or tuple of `tf.Variable` to update to minimize
 231         `loss`.  Defaults to the list of variables collected in the graph
 232         under the key `GraphKey.TRAINABLE_VARIABLES`.
 233       gate_gradients: How to gate the computation of gradients.  Can be
 234         `GATE_NONE`, `GATE_OP`, or `GATE_GRAPH`.
 235       aggregation_method: Specifies the method used to combine gradient terms.
 236         Valid values are defined in the class `AggregationMethod`.
 237       colocate_gradients_with_ops: If True, try colocating gradients with
 238         the corresponding op.
 239       grad_loss: Optional. A `Tensor` holding the gradient computed for `loss`.
 240 
 241     Returns:
 242       A list of (gradient, variable) pairs. Variable is always present, but
 243       gradient can be `None`.
 244 
 245     Raises:
 246       TypeError: If `var_list` contains anything else than `Variable` objects.
 247       ValueError: If some arguments are invalid.
 248     """
 249     if not var_list:
 250       var_list = variables.trainable_variables()
 251 
 252     elastic_difference = [
 253         math_ops.subtract(v, lv)
 254         for v, lv in zip(variables.trainable_variables(),
 255                          [self._local_map[var] for var in var_list])
 256     ]
 257 
 258     distance_loss = self._rho * math_ops.add_n(
 259         [gen_nn_ops.l2_loss(ed) for ed in elastic_difference])
 260 
 261     total_loss = loss + distance_loss
 262     return self._opt.compute_gradients(total_loss, var_list, gate_gradients,
 263                                        aggregation_method,
 264                                        colocate_gradients_with_ops, grad_loss)
 265 
 266   def apply_gradients(self, grads_and_vars, global_step=None, name=None):
 267     """Apply gradients to global variables.
 268 
 269     This is the second part of `minimize()`. It returns an `Operation` that
 270     applies gradients.
 271 
 272     Args:
 273       grads_and_vars: List of (gradient, variable) pairs as returned by
 274         `compute_gradients()`.
 275       global_step: Optional `Variable` to increment by one after the
 276         variables have been updated.
 277       name: Optional name for the returned operation.  Default to the
 278         name passed to the `Optimizer` constructor.
 279 
 280     Returns:
 281       An `Operation` that applies the specified gradients. If `global_step`
 282       was not None, that operation also increments `global_step`.
 283 
 284     Raises:
 285       TypeError: If `grads_and_vars` is malformed.
 286       ValueError: If none of the variables have gradients.
 287     """
 288     global_old = set(n.op.name for n in variables.global_variables())
 289     apply_updates = self._opt.apply_gradients(grads_and_vars)
 290     global_new = set(n.op.name for n in variables.global_variables())
 291     with ops.control_dependencies([apply_updates]):
 292       local_update = state_ops.assign_add(
 293           self._local_step, 1, name='local_step_update').op
 294 
 295     # this is for place the variables created by optimizer to local collection
 296     # e.g., AdamOptimizer will create beta as global variables
 297     def _adjust_optimizer_variable_collection(opt_vars):
 298       g = ops.get_default_graph()
 299       idx = 0
 300       for _ in range(len(g._collections[ops.GraphKeys.GLOBAL_VARIABLES])):
 301         var = g.get_collection_ref(ops.GraphKeys.GLOBAL_VARIABLES)[idx]
 302         name = var.op.name
 303         if name in opt_vars:
 304           ops.add_to_collection(ops.GraphKeys.LOCAL_VARIABLES, var)
 305           del g.get_collection_ref(ops.GraphKeys.GLOBAL_VARIABLES)[idx]
 306         else:
 307           idx += 1
 308 
 309     _adjust_optimizer_variable_collection(global_new - global_old)
 310 
 311     # update global variables.
 312     def _Update_global_variables():
 313       local_vars = [v for g, v in grads_and_vars if g is not None]
 314       global_center_vars = [self._global_map[var] for var in local_vars]
 315       local_center_vars = [self._local_map[var] for var in local_vars]
 316       local_center_vars_update = []
 317       for lvar, var in zip(local_center_vars, global_center_vars):
 318         local_center_vars_update.append(lvar.assign(var))
 319       update_ops = []
 320       differences = []
 321       with ops.control_dependencies(local_center_vars_update):
 322         for v, lv in zip(local_vars, local_center_vars):
 323           with ops.device(v.device):
 324             differences.append(math_ops.subtract(v, lv))
 325         for lvar, diff in zip(local_vars, differences):
 326           with ops.device(lvar.device):
 327             update_ops.append(
 328                 state_ops.assign_sub(lvar,
 329                                      math_ops.multiply(self._moving_rate,
 330                                                        diff)))
 331         for var, diff in zip(global_center_vars, differences):
 332           with ops.device(var.device):
 333             update_ops.append(
 334                 state_ops.assign_add(var,
 335                                      math_ops.multiply(self._moving_rate,
 336                                                        diff)))
 337         if global_step:
 338           with ops.colocate_with(global_step):
 339             update_ops.append(state_ops.assign_add(global_step, 1))
 340       variable_update = control_flow_ops.group(*(update_ops))
 341       return variable_update
 342 
 343     with ops.control_dependencies([local_update]):
 344       condition = math_ops.equal(
 345           math_ops.mod(self._local_step, self._period), 0)
 346       conditional_update = control_flow_ops.cond(
 347           condition, _Update_global_variables, control_flow_ops.no_op)
 348     return conditional_update
 349 
 350   def get_init_op(self, task_index):
 351     """Returns the op to let all the local variables and local center
 352     variables equal to the global center variables before the training begins"""
 353 
 354     def _Add_sync_queues_and_barrier(enqueue_after_list):
 355       """Adds ops to enqueue on all worker queues"""
 356       sync_queues = [
 357           data_flow_ops.FIFOQueue(
 358               self._num_worker, [dtypes.bool],
 359               shapes=[[]],
 360               shared_name='%s%s' % ('variable_init_sync_queue', i))
 361           for i in range(self._num_worker)
 362       ]
 363       queue_ops = []
 364       # For each other worker, add an entry in a queue
 365       token = constant_op.constant(False)
 366       with ops.control_dependencies(enqueue_after_list):
 367         for i, q in enumerate(sync_queues):
 368           if i == task_index:
 369             queue_ops.append(control_flow_ops.no_op())
 370           else:
 371             queue_ops.append(q.enqueue(token))
 372       queue_ops.append(
 373           sync_queues[task_index].dequeue_many(len(sync_queues) - 1))
 374       return control_flow_ops.group(*queue_ops)
 375 
 376     init_ops = []
 377     local_vars = variables.trainable_variables()
 378     global_center_vars = [self._global_map[var] for var in local_vars]
 379     local_center_vars = [self._local_map[var] for var in local_vars]
 380     if not (local_vars and global_center_vars and local_center_vars):
 381       raise ValueError('The lists of local_variables, global_center_variables, '
 382                        'local_center_variables should not be empty  ')
 383     for lvar, gc_var, lc_var in zip(local_vars, global_center_vars,
 384                                     local_center_vars):
 385       init_ops.append(state_ops.assign(lvar, gc_var))
 386       init_ops.append(state_ops.assign(lc_var, gc_var))
 387 
 388     init_op = control_flow_ops.group(*(init_ops))
 389     if self._synchronous == False:
 390       return init_op
 391 
 392     sync_queue_op = _Add_sync_queues_and_barrier([init_op])
 393     return sync_queue_op

 395   def make_session_run_hook(self, is_chief, task_index):
 396     """Creates a hook to handle ElasticAverageOptimizerHook ops such as initialization."""
 397     return _ElasticAverageOptimizerHook(self, is_chief, task_index)

 399   def swapping_saver(self, var_list=None, name='swapping_saver', **kwargs):
 400     """Create a saver copy global_center_variable to trainable variables
 401     Please call this function after all your variables created with
 402     ElasticAverageCustomGetter. For evaluations or inference, use this saver
 403     during training.  It will save the global_center_variable of the trained
 404     parameters under the original parameter names.
 405     Args:
 406       var_list: List of variables to save, as per `Saver()`.
 407                 If set to None, save all the trainable_variables that have
 408                 been created before this call.
 409       name: The name of the saver.
 410       **kwargs: Keyword arguments of `Saver()`.
 411     Returns:
 412       A `tf.train.Saver` object.
 413     Raises:
 414       RuntimeError: global_center_variable is empty, please make sure
 415                     this is called after model created and
 416                     ElasticAverageCustomGetter is used when declaring you model
 417     """
 418     if not self._global_map:
 419       raise RuntimeError('global_center_variable is empty, please make sure '
 420                          'this is called after model created and '
 421                          'ElasticAverageCustomGetter is used when declaring '
 422                          'you model')
 423 
 424     if var_list is None:
 425       var_list = variables.trainable_variables()
 426     if not isinstance(var_list, dict):
 427       var_list = saver.BaseSaverBuilder.OpListToDict(var_list)
 428 
 429     swapped_var_list = {}
 430     for key, var in var_list.items():
 431       tensor = var
 432 
 433       if not isinstance(var, list):
 434         for tvar in variables.trainable_variables():
 435           if tvar.op.name == var.op.name:
 436             tensor = self._global_map.get(tvar, var)
 437             break
 438       else: #partitioned variable
 439         tensor = [self._global_map.get(lvar, lvar) for lvar in var]
 440 
 441       swapped_var_list[key] = tensor
 442 
 443     return saver.Saver(swapped_var_list, name=name, **kwargs)

 445 class _ElasticAverageOptimizerHook(session_run_hook.SessionRunHook):
 446 
 447   def __init__(self, ea_optimizer, is_chief, task_index):
 448     """Creates hook to handle ElasticAverageOptimizer initialization ops.
 449 
 450     Args:
 451       ea_optimizer: `ElasticAverageOptimizer` which this hook will initialize.
 452       is_chief: `Bool`, whether is this a chief replica or not.
 453     """
 454     self._ea_optimizer = ea_optimizer
 455     self._is_chief = is_chief
 456     self._task_index = task_index
 457 
 458   def begin(self):
 459     self._local_init_op = variables.local_variables_initializer()
 460     self._global_init_op = None
 461     if self._is_chief:
 462       self._global_init_op = variables.global_variables_initializer()
 463     self._variable_init_op = self._ea_optimizer.get_init_op(self._task_index)
 464 
 465   def after_create_session(self, session, coord):
 466     """Run initialization ops"""
 467     session.run(self._variable_init_op)
