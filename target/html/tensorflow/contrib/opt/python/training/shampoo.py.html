  37 def GetParam(var, timestep):
  38   if callable(var):
  39     return var(timestep)
  40   else:
  41     return var

  44 class ShampooOptimizer(optimizer.Optimizer):
  45   """The Shampoo Optimizer
  46 
  47   Variant of Adagrad using one preconditioner matrix per variable dimension.
  48   For details, see https://arxiv.org/abs/1802.09568
  49 
  50   gbar is time-weighted accumulated gradient:
  51   gbar[t] = gbar_decay[t] * gbar[t-1] + gbar_weight[t] * g[t]
  52 
  53   mat_gbar is time-weighted accumulated gradient square:
  54   mat_gbar_j[t] = mat_gbar_decay[t] * mat_gbar_j[t-1]
  55                   + mat_gbar_weight[t] * gg_j[t]
  56   where if g[t] = g_abcd then gg_a[t] = g_abcd g_a'bcd (Einstein notation)
  57 
  58   Update rule:
  59   w[t+1] = w[t] - learning_rate[t] * Prod_j mat_gbar_j[t]^(-alpha/n) gbar[t]
  60      Again, mat_gbar_j[t]^(-alpha) gbar[t] is a tensor contraction along the
  61      j'th dimension of gbar[t] with the first dimension of
  62      mat_gbar_j[t]^(-alpha/n), where alpha is a hyperparameter,
  63      and n = rank of the variable.
  64      Prod_j represents doing this contraction for all j in 0..n-1.
  65 
  66   Typically learning_rate is constant, but could be time dependent by passing
  67   a lambda function that depends on step.
  68   """
  69 
  70   def __init__(self,
  71                global_step=0,
  72                max_matrix_size=768,
  73                gbar_decay=0.0,
  74                gbar_weight=1.0,
  75                mat_gbar_decay=1.0,
  76                mat_gbar_weight=1.0,
  77                learning_rate=1.0,
  78                svd_interval=1,
  79                precond_update_interval=1,
  80                epsilon=1e-4,
  81                alpha=0.5,
  82                use_iterative_root=False,
  83                use_locking=False,
  84                name="Shampoo"):
  85     """Default values of the various hyper-parameters.
  86 
  87     gbar_decay, gbar_weight etc. can be a float or a time varying parameter.
  88     For time-varying parameters use e.g. "lambda T: T / (T + 1.0)"
  89     where the expression in the lambda is a tensorflow expression
  90 
  91     Args:
  92       global_step: tensorflow variable indicating the step.
  93       max_matrix_size: We do not perform SVD for matrices larger than this.
  94       gbar_decay:
  95       gbar_weight:  Used to update gbar:
  96             gbar[t] = gbar_decay[t] * gbar[t-1] + gbar_weight[t] * g[t]
  97       mat_gbar_decay:
  98       mat_gbar_weight:  Used to update mat_gbar:
  99            mat_gbar_j[t] = mat_gbar_decay[t] * mat_gbar_j[t-1]
 100                            + mat_gbar_weight[t] * gg_j[t]
 101       learning_rate: Similar to SGD
 102       svd_interval: We should do SVD after this many steps. Default = 1, i.e.
 103                     every step. Usually 20 leads to no loss of accuracy, and
 104                     50 or 100 is also OK. May also want more often early,
 105                     and less often later - set in caller as for example:
 106                     "svd_interval = lambda(T): tf.cond(
 107                         T < 2000, lambda: 20.0, lambda: 1000.0)"
 108       precond_update_interval: We should update the preconditioners after
 109                                this many steps. Default = 1. Usually less than
 110                                svd_interval.
 111       epsilon:  epsilon * I_n is added to each mat_gbar_j for stability
 112       alpha:  total power of the preconditioners.
 113       use_iterative_root: should the optimizer use SVD (faster) or the
 114                           iterative root method (for TPU) for finding the
 115                           roots of PSD matrices.
 116       use_locking:
 117       name: name of optimizer.
 118     """
 119 
 120     super(ShampooOptimizer, self).__init__(use_locking, name)
 121 
 122     self._global_step = math_ops.to_float(global_step)
 123     self._max_matrix_size = max_matrix_size
 124     self._gbar_decay = gbar_decay
 125     self._gbar_weight = gbar_weight
 126     self._mat_gbar_decay = mat_gbar_decay
 127     self._mat_gbar_weight = mat_gbar_weight
 128     self._learning_rate = learning_rate
 129     self._svd_interval = svd_interval
 130     self._precond_update_interval = precond_update_interval
 131     self._epsilon = epsilon
 132     self._alpha = alpha
 133     self._use_iterative_root = use_iterative_root
 134     self._name = name
 135 
 136   def _create_slots(self, var_list):
 137     for v in var_list:
 138       with ops.colocate_with(v):
 139         _ = self._zeros_slot(v, "gbar", self._name)
 140         shape = np.array(v.get_shape())
 141         for i, d in enumerate(shape):
 142           d_tensor = ops.convert_to_tensor(d)
 143           if d <= self._max_matrix_size:
 144             mat_g_init = array_ops.zeros_like(linalg_ops.eye(d_tensor))
 145             if self._svd_interval > 1:
 146               _ = self._get_or_make_slot(v, linalg_ops.eye(d_tensor),
 147                                          "H_" + str(i), self._name)
 148           else:
 149             mat_g_init = array_ops.zeros([d_tensor])
 150 
 151           _ = self._get_or_make_slot(v, mat_g_init, "Gbar_" + str(i),
 152                                      self._name)
 153

 160   def _resource_apply_sparse(self, grad_values, var, grad_indices):
 161     return self._apply_sparse_shared(grad_values, grad_indices, var)
 162 
 163   def _apply_sparse(self, grad, var):
 164     return self._apply_sparse_shared(grad.values, grad.indices, var)
 165 
 166   def _apply_sparse_shared(self, grad_values, grad_indices, var):
 167     if var.get_shape()[0] <= self._max_matrix_size or self._gbar_decay != 0.0:
 168       # The dimension is small enough, we can make the variable dense and
 169       # do a dense update
 170       dense_grad = array_ops.scatter_nd(
 171           array_ops.expand_dims(grad_indices, axis=1), grad_values,
 172           array_ops.shape(var, out_type=grad_indices.dtype))
 173       return self._apply_gradient(dense_grad, var)
 174     return self._apply_gradient(grad_values, var, grad_indices)

 176   def _weighted_average(self, var, weight, weight_t, rest):
 177     """Computes exponential weighted average: var = weight_t * var + rest.
 178 
 179     Important to ensure that var does not occur in rest, otherwise
 180     we can get race conditions in a distributed setting.
 181 
 182     Args:
 183       var: variable to be updated
 184       weight: parameter to be checked. If it is a constant, we can optimize.
 185       weight_t: current value of parameter, used for weighting
 186       rest: the remaining tensor to be added
 187 
 188     Returns:
 189       updated variable.
 190     """
 191     if weight == 0.0:
 192       return rest       # no need to update var, we will never use it.
 193     if weight == 1.0:   # common case
 194       return state_ops.assign_add(var, rest)
 195     # The op below can cause race conditions in a distributed setting,
 196     # since computing weight_t * var + rest can take some time, during
 197     # which var may be set by another worker. To prevent this, it should
 198     # be implemented as a C++ op.
 199     return var.assign_add((weight_t - 1) * var + rest)
 200 
 201   def _update_mat_g(self, mat_g, grad, axes, mat_gbar_decay,
 202                     mat_gbar_weight, i):
 203     """Updates the cumulative outer products of the gradients.
 204 
 205     Args:
 206       mat_g: the matrix to be updated
 207       grad: the gradient of the variable
 208       axes: a list of k-1 integers 0 to k-1, except i
 209       mat_gbar_decay: constant for weighted average:
 210           mat_g = mat_g * decay + grad * weight
 211       mat_gbar_weight: constant for weighted average
 212       i: index of dimension to be updated.
 213 
 214     Returns:
 215       updated mat_g = mat_g * mat_gbar_decay + grad_outer * mat_gbar_weight
 216 
 217     In Einstein notation if i = 0: grad_outer_aa'= g_abcd g_a'bcd
 218     thus grad_outer is a matrix d_i x d_i, where d_i is the size of the
 219     i'th dimension of g.
 220     Alternate view: If mat_i(grad) is the flattening of grad to a
 221     d_i x (d_1d_2...d_{i-1}d_{i+1}...d_k) matrix, then
 222          grad_outer = mat_i(grad) mat_i(grad).transpose
 223     """
 224     grad_outer = math_ops.tensordot(grad, grad, axes=(axes, axes),
 225                                     name="grad_outer_" + str(i))
 226     return self._weighted_average(mat_g, self._mat_gbar_decay, mat_gbar_decay,
 227                                   mat_gbar_weight * grad_outer)
 228 
 229   def _compute_power_svd(self, var, mat_g, mat_g_size, alpha, mat_h_slot_name):
 230     """Computes mat_h = mat_g^alpha using svd. mat_g is a symmetric PSD matrix.
 231 
 232     Args:
 233       var: the variable we are updating.
 234       mat_g: the symmetric PSD matrix whose power it to be computed
 235       mat_g_size: size of mat_g
 236       alpha: a real number
 237       mat_h_slot_name: name of slot to store the power, if needed.
 238 
 239     Returns:
 240       mat_h = mat_g^alpha
 241 
 242     Stores mat_h in the appropriate slot, if it exists.
 243     Note that mat_g is PSD. So we could use linalg_ops.self_adjoint_eig.
 244     """
 245     if mat_g_size == 1:
 246       mat_h = math_ops.pow(mat_g + self._epsilon, alpha)
 247     else:
 248       damping = self._epsilon * linalg_ops.eye(math_ops.to_int32(mat_g_size))
 249       diag_d, mat_u, mat_v = linalg_ops.svd(mat_g + damping, full_matrices=True)
 250       mat_h = math_ops.matmul(
 251           mat_v * math_ops.pow(math_ops.maximum(diag_d, self._epsilon), alpha),
 252           array_ops.transpose(mat_u))
 253     if mat_h_slot_name is not None:
 254       return state_ops.assign(self.get_slot(var, mat_h_slot_name), mat_h)
 255     return mat_h
 256 
 257   def _compute_power_iter(self, var, mat_g, mat_g_size, alpha, mat_h_slot_name,
 258                           iter_count=100, epsilon=1e-6):
 259     """Computes mat_g^alpha, where alpha = -1/p, p a positive integer."""
 260 
 261     mat_g_sqrt = matrix_functions.matrix_square_root(mat_g, mat_g_size,
 262                                                      iter_count, self._epsilon)
 263     mat_h = matrix_functions.matrix_inverse_pth_root(
 264         mat_g_sqrt,
 265         mat_g_size,
 266         2 * alpha,
 267         iter_count,
 268         epsilon,
 269         ridge_epsilon=0.0)
 270 
 271     if mat_h_slot_name is not None:
 272       return state_ops.assign(self.get_slot(var, mat_h_slot_name), mat_h)
 273     return mat_h
 274 
 275   def _compute_power(self, var, mat_g, mat_g_size, alpha, mat_h_slot_name=None):
 276     """Just a switch between the iterative power vs svd."""
 277     with ops.name_scope("matrix_iterative_power"):
 278       if self._use_iterative_root:
 279         return self._compute_power_iter(var, mat_g, mat_g_size, alpha,
 280                                         mat_h_slot_name)
 281       else:
 282         return self._compute_power_svd(var, mat_g, mat_g_size, alpha,
 283                                        mat_h_slot_name)

 285   def _apply_gradient(self, grad, var, indices=None):
 286     """The main function to update a variable.
 287 
 288     Args:
 289       grad: A Tensor containing gradient to apply.
 290       var: A Tensor containing the variable to update.
 291       indices: An array of integers, for sparse update.
 292 
 293     Returns:
 294       Updated variable var = var - learning_rate * preconditioner * grad
 295 
 296     If the gradient is dense, var and grad have the same shape.
 297     If the update is sparse, then the first dimension of the gradient and var
 298     may differ, others are all the same. In this case the indices array
 299     provides the set of indices of the variable which are to be updated with
 300     each row of the gradient.
 301     """
 302     global_step = self._global_step + 1
 303 
 304     # Update accumulated weighted average of gradients
 305     gbar = self.get_slot(var, "gbar")
 306     gbar_decay_t = GetParam(self._gbar_decay, global_step)
 307     gbar_weight_t = GetParam(self._gbar_weight, global_step)
 308     if indices is not None:
 309       # Note - the sparse update is not easily implemented, since the
 310       # algorithm needs all indices of gbar to be updated
 311       # if mat_gbar_decay != 1 or mat_gbar_decay != 0.
 312       # One way to make mat_gbar_decay = 1 is by rescaling.
 313       # If we want the update:
 314       #         G_{t+1} = a_{t+1} G_t + b_{t+1} w_t
 315       # define:
 316       #         r_{t+1} = a_{t+1} * r_t
 317       #         h_t = G_t / r_t
 318       # Then:
 319       #         h_{t+1} = h_t + (b_{t+1} / r_{t+1}) * w_t
 320       # So we get the mat_gbar_decay = 1 as desired.
 321       # We can implement this in a future version as needed.
 322       # However we still need gbar_decay = 0, otherwise all indices
 323       # of the variable will need to be updated.
 324       if self._gbar_decay != 0.0:
 325         tf_logging.warning("Not applying momentum for variable: %s" % var.name)
 326       gbar_updated = grad
 327     else:
 328       gbar_updated = self._weighted_average(gbar, self._gbar_decay,
 329                                             gbar_decay_t,
 330                                             gbar_weight_t * grad)
 331 
 332     # Update the preconditioners and compute the preconditioned gradient
 333     shape = var.get_shape()
 334     mat_g_list = []
 335     for i in range(len(shape)):
 336       mat_g_list.append(self.get_slot(var, "Gbar_" + str(i)))
 337     mat_gbar_decay_t = GetParam(self._mat_gbar_decay, global_step)
 338     mat_gbar_weight_t = GetParam(self._mat_gbar_weight, global_step)
 339 
 340     preconditioned_grad = gbar_updated
 341     v_rank = len(mat_g_list)
 342     neg_alpha = - GetParam(self._alpha, global_step) / v_rank
 343     svd_interval = GetParam(self._svd_interval, global_step)
 344     precond_update_interval = GetParam(self._precond_update_interval,
 345                                        global_step)
 346     for i, mat_g in enumerate(mat_g_list):
 347       # axes is the list of indices to reduce - everything but the current i.
 348       axes = list(range(i)) + list(range(i+1, v_rank))
 349       if shape[i] <= self._max_matrix_size:
 350         # If the tensor size is sufficiently small perform full Shampoo update
 351         # Note if precond_update_interval > 1 and mat_gbar_decay_t != 1, this
 352         # is not strictly correct. However we will use it for now, and
 353         # fix if needed. (G_1 = aG + bg ==> G_n = a^n G + (1+a+..+a^{n-1})bg)
 354 
 355         # pylint: disable=g-long-lambda,cell-var-from-loop
 356         mat_g_updated = control_flow_ops.cond(
 357             math_ops.mod(global_step, precond_update_interval) < 1,
 358             lambda: self._update_mat_g(
 359                 mat_g, grad, axes, mat_gbar_decay_t,
 360                 mat_gbar_weight_t * precond_update_interval, i),
 361             lambda: mat_g)
 362 
 363         mat_g_updated = mat_g_updated / float(shape[i].value)
 364 
 365         if self._svd_interval == 1:
 366           mat_h = self._compute_power(var, mat_g_updated, shape[i], neg_alpha)
 367         else:
 368           mat_h = control_flow_ops.cond(
 369               math_ops.mod(global_step, svd_interval) < 1,
 370               lambda: self._compute_power(var, mat_g_updated, shape[i],
 371                                           neg_alpha, "H_" + str(i)),
 372               lambda: self.get_slot(var, "H_" + str(i)))
 373 
 374         # mat_h is a square matrix of size d_i x d_i
 375         # preconditioned_grad is a d_i x ... x d_n x d_0 x ... d_{i-1} tensor
 376         # After contraction with a d_i x d_i tensor
 377         # it becomes a d_{i+1} x ... x d_n x d_0 x ... d_i tensor
 378         # (the first dimension is contracted out, and the second dimension of
 379         # mat_h is appended).  After going through all the indices, it becomes
 380         # a d_0 x ... x d_n tensor again.
 381         preconditioned_grad = math_ops.tensordot(preconditioned_grad, mat_h,
 382                                                  axes=([0], [0]),
 383                                                  name="precond_" + str(i))
 384       else:
 385         # Tensor size is too large -- perform diagonal Shampoo update
 386         # Only normalize non-vector cases.
 387         if axes:
 388           normalizer = 1.0 if indices is not None else float(shape[i].value)
 389           grad_outer = math_ops.reduce_sum(grad * grad, axis=axes) / normalizer
 390         else:
 391           grad_outer = grad * grad
 392 
 393         if i == 0 and indices is not None:
 394           assert self._mat_gbar_decay == 1.0
 395           mat_g_updated = state_ops.scatter_add(mat_g, indices,
 396                                                 mat_gbar_weight_t * grad_outer)
 397           mat_h = math_ops.pow(
 398               array_ops.gather(mat_g_updated, indices) + self._epsilon,
 399               neg_alpha)
 400         else:
 401           mat_g_updated = self._weighted_average(mat_g,
 402                                                  self._mat_gbar_decay,
 403                                                  mat_gbar_decay_t,
 404                                                  mat_gbar_weight_t * grad_outer)
 405           mat_h = math_ops.pow(mat_g_updated + self._epsilon, neg_alpha)
 406 
 407         # Need to do the transpose to ensure that the tensor becomes
 408         # a d_{i+1} x ... x d_n x d_0 x ... d_i tensor as described above.
 409         preconditioned_grad = array_ops.transpose(
 410             preconditioned_grad, perm=list(range(1, v_rank)) + [0]) * mat_h
 411 
 412     # Update the variable based on the Shampoo update
 413     learning_rate_t = GetParam(self._learning_rate, global_step)
 414     if indices is not None:
 415       var_updated = state_ops.scatter_add(
 416           var, indices, -learning_rate_t * preconditioned_grad)
 417     else:
 418       var_updated = state_ops.assign_sub(var,
 419                                          learning_rate_t * preconditioned_grad)
 420     return var_updated
