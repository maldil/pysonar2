  25 def matrix_square_root(mat_a, mat_a_size, iter_count=100, ridge_epsilon=1e-4):
  26   """Iterative method to get matrix square root.
  27 
  28   Stable iterations for the matrix square root, Nicholas J. Higham
  29 
  30   Page 231, Eq 2.6b
  31   http://citeseerx.ist.psu.edu/viewdoc/download?doi=10.1.1.6.8799&rep=rep1&type=pdf
  32 
  33   Args:
  34     mat_a: the symmetric PSD matrix whose matrix square root be computed
  35     mat_a_size: size of mat_a.
  36     iter_count: Maximum number of iterations.
  37     ridge_epsilon: Ridge epsilon added to make the matrix positive definite.
  38 
  39   Returns:
  40     mat_a^0.5
  41   """
  42 
  43   def _iter_condition(i, unused_mat_y, unused_old_mat_y, unused_mat_z,
  44                       unused_old_mat_z, err, old_err):
  45     # This method require that we check for divergence every step.
  46     return math_ops.logical_and(i < iter_count, err < old_err)
  47 
  48   def _iter_body(i, mat_y, unused_old_mat_y, mat_z, unused_old_mat_z, err,
  49                  unused_old_err):
  50     current_iterate = 0.5 * (3.0 * identity - math_ops.matmul(mat_z, mat_y))
  51     current_mat_y = math_ops.matmul(mat_y, current_iterate)
  52     current_mat_z = math_ops.matmul(current_iterate, mat_z)
  53     # Compute the error in approximation.
  54     mat_sqrt_a = current_mat_y * math_ops.sqrt(norm)
  55     mat_a_approx = math_ops.matmul(mat_sqrt_a, mat_sqrt_a)
  56     residual = mat_a - mat_a_approx
  57     current_err = math_ops.sqrt(math_ops.reduce_sum(residual * residual)) / norm
  58     return i + 1, current_mat_y, mat_y, current_mat_z, mat_z, current_err, err
  59 
  60   identity = linalg_ops.eye(math_ops.to_int32(mat_a_size))
  61   mat_a = mat_a + ridge_epsilon * identity
  62   norm = math_ops.sqrt(math_ops.reduce_sum(mat_a * mat_a))
  63   mat_init_y = mat_a / norm
  64   mat_init_z = identity
  65   init_err = norm
  66 
  67   _, _, prev_mat_y, _, _, _, _ = control_flow_ops.while_loop(
  68       _iter_condition, _iter_body, [
  69           0, mat_init_y, mat_init_y, mat_init_z, mat_init_z, init_err,
  70           init_err + 1.0
  71       ])
  72   return prev_mat_y * math_ops.sqrt(norm)

  75 def matrix_inverse_pth_root(mat_g,
  76                             mat_g_size,
  77                             alpha,
  78                             iter_count=100,
  79                             epsilon=1e-6,
  80                             ridge_epsilon=1e-6):
  81   """Computes mat_g^alpha, where alpha = -1/p, p a positive integer.
  82 
  83   We use an iterative Schur-Newton method from equation 3.2 on page 9 of:
  84 
  85   A Schur-Newton Method for the Matrix p-th Root and its Inverse
  86   by Chun-Hua Guo and Nicholas J. Higham
  87   SIAM Journal on Matrix Analysis and Applications,
  88   2006, Vol. 28, No. 3 : pp. 788-804
  89   https://pdfs.semanticscholar.org/0abe/7f77433cf5908bfe2b79aa91af881da83858.pdf
  90 
  91   Args:
  92     mat_g: the symmetric PSD matrix whose power it to be computed
  93     mat_g_size: size of mat_g.
  94     alpha: exponent, must be -1/p for p a positive integer.
  95     iter_count: Maximum number of iterations.
  96     epsilon: accuracy indicator, useful for early termination.
  97     ridge_epsilon: Ridge epsilon added to make the matrix positive definite.
  98 
  99   Returns:
 100     mat_g^alpha
 101   """
 102 
 103   identity = linalg_ops.eye(math_ops.to_int32(mat_g_size))
 104 
 105   def mat_power(mat_m, p):
 106     """Computes mat_m^p, for p a positive integer.
 107 
 108     Power p is known at graph compile time, so no need for loop and cond.
 109     Args:
 110       mat_m: a square matrix
 111       p: a positive integer
 112 
 113     Returns:
 114       mat_m^p
 115     """
 116     assert p == int(p) and p > 0
 117     power = None
 118     while p > 0:
 119       if p % 2 == 1:
 120         power = math_ops.matmul(mat_m, power) if power is not None else mat_m
 121       p //= 2
 122       mat_m = math_ops.matmul(mat_m, mat_m)
 123     return power
 124 
 125   def _iter_condition(i, mat_m, _):
 126     return math_ops.logical_and(
 127         i < iter_count,
 128         math_ops.reduce_max(math_ops.abs(mat_m - identity)) > epsilon)
 129 
 130   def _iter_body(i, mat_m, mat_x):
 131     mat_m_i = (1 - alpha) * identity + alpha * mat_m
 132     return (i + 1, math_ops.matmul(mat_power(mat_m_i, -1.0 / alpha), mat_m),
 133             math_ops.matmul(mat_x, mat_m_i))
 134 
 135   if mat_g_size == 1:
 136     mat_h = math_ops.pow(mat_g + ridge_epsilon, alpha)
 137   else:
 138     damped_mat_g = mat_g + ridge_epsilon * identity
 139     z = (1 - 1 / alpha) / (2 * linalg_ops.norm(damped_mat_g))
 140     # The best value for z is
 141     # (1 - 1/alpha) * (c_max^{-alpha} - c_min^{-alpha}) /
 142     #                 (c_max^{1-alpha} - c_min^{1-alpha})
 143     # where c_max and c_min are the largest and smallest singular values of
 144     # damped_mat_g.
 145     # The above estimate assumes that c_max > c_min * 2^p. (p = -1/alpha)
 146     # Can replace above line by the one below, but it is less accurate,
 147     # hence needs more iterations to converge.
 148     # z = (1 - 1/alpha) / math_ops.trace(damped_mat_g)
 149     # If we want the method to always converge, use z = 1 / norm(damped_mat_g)
 150     # or z = 1 / math_ops.trace(damped_mat_g), but these can result in many
 151     # extra iterations.
 152     _, _, mat_h = control_flow_ops.while_loop(
 153         _iter_condition, _iter_body,
 154         [0, damped_mat_g * z, identity * math_ops.pow(z, -alpha)])
 155   return mat_h
