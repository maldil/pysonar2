  28 class NadamOptimizer(adam.AdamOptimizer):
  29   """Optimizer that implements the Nadam algorithm.
  30 
  31   See [Dozat, T., 2015](http://cs229.stanford.edu/proj2015/054_report.pdf).
  32   """
  33 
  34   def _apply_dense(self, grad, var):
  35     m = self.get_slot(var, "m")
  36     v = self.get_slot(var, "v")
  37     beta1_power, beta2_power = self._get_beta_accumulators()
  38     return training_ops.apply_adam(
  39         var,
  40         m,
  41         v,
  42         math_ops.cast(beta1_power, var.dtype.base_dtype),
  43         math_ops.cast(beta2_power, var.dtype.base_dtype),
  44         math_ops.cast(self._lr_t, var.dtype.base_dtype),
  45         math_ops.cast(self._beta1_t, var.dtype.base_dtype),
  46         math_ops.cast(self._beta2_t, var.dtype.base_dtype),
  47         math_ops.cast(self._epsilon_t, var.dtype.base_dtype),
  48         grad,
  49         use_locking=self._use_locking,
  50         use_nesterov=True).op
  51 
  52   def _resource_apply_dense(self, grad, var):
  53     m = self.get_slot(var, "m")
  54     v = self.get_slot(var, "v")
  55     beta1_power, beta2_power = self._get_beta_accumulators()
  56     return training_ops.resource_apply_adam(
  57         var.handle,
  58         m.handle,
  59         v.handle,
  60         math_ops.cast(beta1_power, grad.dtype.base_dtype),
  61         math_ops.cast(beta2_power, grad.dtype.base_dtype),
  62         math_ops.cast(self._lr_t, grad.dtype.base_dtype),
  63         math_ops.cast(self._beta1_t, grad.dtype.base_dtype),
  64         math_ops.cast(self._beta2_t, grad.dtype.base_dtype),
  65         math_ops.cast(self._epsilon_t, grad.dtype.base_dtype),
  66         grad,
  67         use_locking=self._use_locking,
  68         use_nesterov=True)
  69 
  70   def _apply_sparse_shared(self, grad, var, indices, scatter_add):
  71     beta1_power, beta2_power = self._get_beta_accumulators()
  72     beta1_power = math_ops.cast(beta1_power, var.dtype.base_dtype)
  73     beta2_power = math_ops.cast(beta2_power, var.dtype.base_dtype)
  74     lr_t = math_ops.cast(self._lr_t, var.dtype.base_dtype)
  75     beta1_t = math_ops.cast(self._beta1_t, var.dtype.base_dtype)
  76     beta2_t = math_ops.cast(self._beta2_t, var.dtype.base_dtype)
  77     epsilon_t = math_ops.cast(self._epsilon_t, var.dtype.base_dtype)
  78     lr = (lr_t * math_ops.sqrt(1 - beta2_power) / (1 - beta1_power))
  79     # m_t = beta1 * m + (1 - beta1) * g_t
  80     m = self.get_slot(var, "m")
  81     m_scaled_g_values = grad * (1 - beta1_t)
  82     m_t = state_ops.assign(m, m * beta1_t, use_locking=self._use_locking)
  83     with ops.control_dependencies([m_t]):
  84       m_t = scatter_add(m, indices, m_scaled_g_values)
  85       # m_bar = (1 - beta1) * g_t + beta1 * m_t
  86       m_bar = m_scaled_g_values + beta1_t * m_t
  87     # v_t = beta2 * v + (1 - beta2) * (g_t * g_t)
  88     v = self.get_slot(var, "v")
  89     v_scaled_g_values = (grad * grad) * (1 - beta2_t)
  90     v_t = state_ops.assign(v, v * beta2_t, use_locking=self._use_locking)
  91     with ops.control_dependencies([v_t]):
  92       v_t = scatter_add(v, indices, v_scaled_g_values)
  93     v_sqrt = math_ops.sqrt(v_t)
  94     var_update = state_ops.assign_sub(
  95         var, lr * m_bar / (v_sqrt + epsilon_t), use_locking=self._use_locking)
  96     return control_flow_ops.group(*[var_update, m_bar, v_t])
