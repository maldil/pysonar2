  15 """Layer-wise Adaptive Rate Scaling optimizer for large-batch training."""
  16 
  17 from __future__ import absolute_import
  18 from __future__ import division
  19 from __future__ import print_function
  20 
  21 from tensorflow.python.ops import array_ops
  22 from tensorflow.python.ops import linalg_ops
  23 from tensorflow.python.ops import math_ops
  24 from tensorflow.python.training import optimizer
  25 from tensorflow.python.training import training_ops
  26 
  27 
  28 class LARSOptimizer(optimizer.Optimizer):
  29   """Layer-wise Adaptive Rate Scaling for large batch training.
  30 
  31   Introduced by "Large Batch Training of Convolutional Networks" by Y. You,
  32   I. Gitman, and B. Ginsburg. (https://arxiv.org/abs/1708.03888)
  33 
  34   Implements the LARS learning rate scheme presented in the paper above. This
  35   optimizer is useful when scaling the batch size to up to 32K without
  36   significant performance degradation. It is recommended to use the optimizer
  37   in conjunction with:
  38       - Gradual learning rate warm-up
  39       - Linear learning rate scaling
  40       - Poly rule learning rate decay
  41 
  42   Note, LARS scaling is currently only enabled for dense tensors. Sparse tensors
  43   use the default momentum optimizer.
  44   """
  45 
  46   def __init__(
  47       self,
  48       learning_rate,
  49       momentum=0.9,
  50       weight_decay=0.0001,
  51       # The LARS coefficient is a hyperparameter
  52       eeta=0.001,
  53       epsilon=0.0,
  54       name="LARSOptimizer",
  55       # Enable skipping variables from LARS scaling.
  56       # TODO(sameerkm): Enable a direct mechanism to pass a
  57       # subset of variables to the optimizer.
  58       skip_list=None,
  59       use_nesterov=False):
  60     """Construct a new LARS Optimizer.
  61 
  62     Args:
  63       learning_rate: A `Tensor` or floating point value. The base learning rate.
  64       momentum: A floating point value. Momentum hyperparameter.
  65       weight_decay: A floating point value. Weight decay hyperparameter.
  66       eeta: LARS coefficient as used in the paper. Dfault set to LARS
  67         coefficient from the paper. (eeta / weight_decay) determines the highest
  68         scaling factor in LARS.
  69       epsilon: Optional epsilon parameter to be set in models that have very
  70         small gradients. Default set to 0.0.
  71       name: Optional name prefix for variables and ops created by LARSOptimizer.
  72       skip_list: List of strings to enable skipping variables from LARS scaling.
  73         If any of the strings in skip_list is a subset of var.name, variable
  74         'var' is skipped from LARS scaling. For a typical classification model
  75         with batch normalization, the skip_list is ['batch_normalization',
  76         'bias']
  77       use_nesterov: when set to True, nesterov momentum will be enabled
  78 
  79     Raises:
  80       ValueError: If a hyperparameter is set to a non-sensical value.
  81     """
  82     if momentum < 0.0:
  83       raise ValueError("momentum should be positive: %s" % momentum)
  84     if weight_decay < 0.0:
  85       raise ValueError("weight_decay should be positive: %s" % weight_decay)
  86     super(LARSOptimizer, self).__init__(use_locking=False, name=name)
  87 
  88     self._learning_rate = learning_rate
  89     self._momentum = momentum
  90     self._weight_decay = weight_decay
  91     self._eeta = eeta
  92     self._epsilon = epsilon
  93     self._name = name
  94     self._skip_list = skip_list
  95     self._use_nesterov = use_nesterov
  96 
  97   def _create_slots(self, var_list):
  98     for v in var_list:
  99       self._zeros_slot(v, "momentum", self._name)
 100 
 101   def compute_lr(self, grad, var):
 102     scaled_lr = self._learning_rate
 103     if self._skip_list is None or not any(v in var.name
 104                                           for v in self._skip_list):
 105       w_norm = linalg_ops.norm(var, ord=2)
 106       g_norm = linalg_ops.norm(grad, ord=2)
 107       trust_ratio = array_ops.where(
 108           math_ops.greater(w_norm, 0),
 109           array_ops.where(
 110               math_ops.greater(g_norm, 0),
 111               (self._eeta * w_norm /
 112                (g_norm + self._weight_decay * w_norm + self._epsilon)), 1.0),
 113           1.0)
 114       scaled_lr = self._learning_rate * trust_ratio
 115     return scaled_lr
 116 
 117   def _apply_dense(self, grad, var):
 118     scaled_lr = self.compute_lr(grad, var)
 119     mom = self.get_slot(var, "momentum")
 120     return training_ops.apply_momentum(
 121         var,
 122         mom,
 123         scaled_lr,
 124         grad,
 125         self._momentum,
 126         use_locking=False,
 127         use_nesterov=self._use_nesterov)
 128 
 129   def _resource_apply_dense(self, grad, var):
 130     scaled_lr = self.compute_lr(grad, var)
 131     mom = self.get_slot(var, "momentum")
 132     return training_ops.resource_apply_momentum(
 133         var.handle,
 134         mom.handle,
 135         scaled_lr,
 136         grad,
 137         self._momentum,
 138         use_locking=False,
 139         use_nesterov=self._use_nesterov)
 140 
 141   # Fallback to momentum optimizer for sparse tensors
 142   def _apply_sparse(self, grad, var):
 143     mom = self.get_slot(var, "momentum")
 144     return training_ops.sparse_apply_momentum(
 145         var,
 146         mom,
 147         math_ops.cast(self._learning_rate_tensor, var.dtype.base_dtype),
 148         grad.values,
 149         grad.indices,
 150         math_ops.cast(self._momentum_tensor, var.dtype.base_dtype),
 151         use_locking=self._use_locking,
 152         use_nesterov=self._use_nesterov).op
 153 
 154   def _resource_apply_sparse(self, grad, var, indices):
 155     mom = self.get_slot(var, "momentum")
 156     return training_ops.resource_sparse_apply_momentum(
 157         var.handle,
 158         mom.handle,
 159         math_ops.cast(self._learning_rate_tensor, grad.dtype),
 160         grad,
 161         indices,
 162         math_ops.cast(self._momentum_tensor, grad.dtype),
 163         use_locking=self._use_locking,
 164         use_nesterov=self._use_nesterov)
