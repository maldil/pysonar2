  34 class VariableClippingOptimizer(optimizer.Optimizer):
  35   """Wrapper optimizer that clips the norm of specified variables after update.
  36 
  37   This optimizer delegates all aspects of gradient calculation and application
  38   to an underlying optimizer.  After applying gradients, this optimizer then
  39   clips the variable to have a maximum L2 norm along specified dimensions.
  40   NB: this is quite different from clipping the norm of the gradients.
  41 
  42   Multiple instances of `VariableClippingOptimizer` may be chained to specify
  43   different max norms for different subsets of variables.
  44 
  45   This is more efficient at serving-time than using normalization during
  46   embedding lookup, at the expense of more expensive training and fewer
  47   guarantees about the norms.
  48

  53   def __init__(self,
  54                opt,
  55                vars_to_clip_dims,
  56                max_norm,
  57                use_locking=False,
  58                colocate_clip_ops_with_vars=False,
  59                name="VariableClipping"):
  60     """Construct a new clip-norm optimizer.
  61 
  62     Args:
  63       opt: The actual optimizer that will be used to compute and apply the
  64         gradients. Must be one of the Optimizer classes.
  65       vars_to_clip_dims: A dict with keys as Variables and values as lists
  66         of dimensions along which to compute the L2-norm.  See
  67         `tf.clip_by_norm` for more details.
  68       max_norm: The L2-norm to clip to, for all variables specified.
  69       use_locking: If `True` use locks for clip update operations.
  70       colocate_clip_ops_with_vars: If `True`, try colocating the clip norm
  71         ops with the corresponding variable.
  72       name: Optional name prefix for the operations created when applying
  73         gradients.  Defaults to "VariableClipping".
  74     """
  75     super(VariableClippingOptimizer, self).__init__(use_locking, name)
  76     self._opt = opt
  77     # Defensive copy of input dict
  78     self._vars_to_clip_dims = {
  79         var: clip_dims[:] for var, clip_dims in vars_to_clip_dims.items()}
  80     self._max_norm = max_norm
  81     self._colocate_clip_ops_with_vars = colocate_clip_ops_with_vars
  82 
  83   def compute_gradients(self, *args, **kwargs):
  84     return self._opt.compute_gradients(*args, **kwargs)
  85 
  86   def get_slot(self, *args, **kwargs):
  87     return self._opt.get_slot(*args, **kwargs)
  88 
  89   def get_slot_names(self, *args, **kwargs):
  90     return self._opt.get_slot_names(*args, **kwargs)
  91 
  92   def apply_gradients(self, grads_and_vars, global_step=None, name=None):
  93     with ops.name_scope(name, self._name) as name:
  94       update_op = self._opt.apply_gradients(
  95           grads_and_vars, global_step=global_step)
  96       clip_update_ops = []
  97       with ops.control_dependencies([update_op]):
  98         for grad, var in grads_and_vars:
  99           if grad is None or var not in self._vars_to_clip_dims:
 100             continue
 101           with ops.name_scope("clip_" + var.op.name):
 102             if isinstance(grad, ops.Tensor):
 103               clip_update_ops.append(self._clip_dense(var))
 104             else:
 105               clip_update_ops.append(self._clip_sparse(grad, var))
 106 
 107       # In case no var was clipped, still need to run the update_op.
 108       return control_flow_ops.group(*([update_op] + clip_update_ops), name=name)
 109 
 110   def _clip_dense(self, var):
 111     with self._maybe_colocate_with(var):
 112       updated_var_value = var.read_value()
 113       normalized_var = clip_ops.clip_by_norm(
 114           updated_var_value, self._max_norm, self._vars_to_clip_dims[var])
 115       delta = updated_var_value - normalized_var
 116     with ops.colocate_with(var):
 117       return var.assign_sub(delta, use_locking=self._use_locking)
 118 
 119   def _clip_sparse(self, grad, var):
 120     assert isinstance(grad, ops.IndexedSlices)
 121     clip_dims = self._vars_to_clip_dims[var]
 122     if 0 in clip_dims:
 123       logging.warning("Clipping norm across dims %s for %s is inefficient "
 124                       "when including sparse dimension 0.", clip_dims,
 125                       var.op.name)
 126       return self._clip_dense(var)
 127 
 128     with ops.colocate_with(var):
 129       var_subset = array_ops.gather(var, grad.indices)
 130     with self._maybe_colocate_with(var):
 131       normalized_var_subset = clip_ops.clip_by_norm(
 132           var_subset, self._max_norm, clip_dims)
 133       delta = ops.IndexedSlices(
 134           var_subset - normalized_var_subset, grad.indices, grad.dense_shape)
 135     with ops.colocate_with(var):
 136       return var.scatter_sub(delta, use_locking=self._use_locking)
 137 
 138   @contextlib.contextmanager
 139   def _maybe_colocate_with(self, var):
 140     """Context to colocate with `var` if `colocate_clip_ops_with_vars`."""
 141     if self._colocate_clip_ops_with_vars:
 142       with ops.colocate_with(var):
 143         yield
 144     else:
 145       yield
