  33 class AGNCustomGetter(object):
  34   """Custom_getter class is used to do:
  35 
  36   1. Change trainable variables to local collection and place them at worker
  37     device
  38   2. Generate global variables(global center variables)
  39   3. Generate grad variables(gradients) which record the gradients sum
  40     and place them at worker device
  41     Notice that the class should be used with tf.replica_device_setter,
  42     so that the global center variables and global step variable can be placed
  43     at ps device.
  44   """
  45 
  46   def __init__(self, worker_device):
  47     """
  48       Args:
  49         worker_device: put the grad_variables on worker device
  50     """
  51     self._worker_device = worker_device
  52     self._global_map = {}
  53     self._grad_map = {}
  54 
  55   def __call__(self, getter, name, trainable, collections, *args, **kwargs):
  56     if trainable:
  57       with ops.device(self._worker_device):
  58         local_var = getter(
  59             name,
  60             trainable=True,
  61             collections=[ops.GraphKeys.LOCAL_VARIABLES],
  62             *args,
  63             **kwargs)
  64       if kwargs['reuse'] == True:
  65         return local_var
  66       global_center_variable = getter(
  67           name='%s/%s' % (GLOBAL_VARIABLE_NAME, name),
  68           trainable=False,
  69           collections=[ops.GraphKeys.GLOBAL_VARIABLES],
  70           *args,
  71           **kwargs)
  72 
  73       with ops.device(self._worker_device):
  74         grad_variable = getter(
  75             name='%s/%s' % (GRAD_VARIABLE_NAME, name),
  76             trainable=False,
  77             collections=[ops.GraphKeys.LOCAL_VARIABLES],
  78             *args,
  79             **kwargs)
  80       if kwargs['partitioner'] is None:
  81         self._grad_map[local_var] = grad_variable
  82         self._global_map[local_var] = global_center_variable
  83       else:
  84         v_list = list(local_var)
  85         for i in range(len(v_list)):
  86           self._grad_map[v_list[i]] = list(grad_variable)[i]
  87           self._global_map[v_list[i]] = list(global_center_variable)[i]
  88       return local_var
  89     else:
  90       return getter(
  91           name, trainable=trainable, collections=collections, *args, **kwargs)

  94 class AGNOptimizer(optimizer.Optimizer):
  95   """Wrapper that implements the Accumulated GradientNormalization algorithm.
  96 
  97   Reference:
  98     Accumulated Gradient Normalization: Joeri Hermans ACML2017
  99     https://arxiv.org/abs/1710.02368
 100   """
 101 
 102   def __init__(self,
 103                optimizer,
 104                num_worker,
 105                custom_getter,
 106                communication_period=10,
 107                use_locking=True,
 108                name='AGNOptimizer'):
 109     """Construct a new AGN optimizer.
 110 
 111     Args:
 112       optimizer: input optimizer, can be sgd/momentum/adam etc.
 113       num_worker: The number of workers
 114       custom_getter: The AGNCustomGetter
 115       communication_period: An int point value to controls the frequency of the
 116         communication between every worker and the ps.
 117       use_locking: If True use locks for update operations.
 118       name: Optional name prefix for the operations created when applying
 119         gradients. Defaults to "AGNOptimizer".
 120     """
 121     super(AGNOptimizer, self).__init__(use_locking, name)
 122     self._opt = optimizer
 123     self._num_worker = num_worker
 124     self._period = communication_period
 125     self._global_map = custom_getter._global_map
 126     self._grad_map = custom_getter._grad_map
 127     self._local_step = variable_scope.get_variable(
 128         initializer=0,
 129         trainable=False,
 130         collections=[ops.GraphKeys.LOCAL_VARIABLES],
 131         name='local_step')
 132     self._opt._prepare()
 133 
 134   def apply_gradients(self, grads_and_vars, global_step=None, name=None):
 135     """Apply gradients to global variables.
 136 
 137     This is the second part of `minimize()`. It returns an `Operation` that
 138     applies gradients.
 139 
 140     Args:
 141       grads_and_vars: List of (gradient, variable) pairs as returned by
 142         `compute_gradients()`.
 143       global_step: Optional `Variable` to increment by one after the variables
 144         have been updated.
 145       name: Optional name for the returned operation.  Default to the name
 146         passed to the `Optimizer` constructor.
 147 
 148     Returns:
 149       An `Operation` that applies the specified gradients. If `global_step`
 150       was not None, that operation also increments `global_step`.
 151     """
 152     local_vars = [v for g, v in grads_and_vars if g is not None]
 153     grads = [g for g, v in grads_and_vars if g is not None]
 154 
 155     def _variable_creator(next_creator, collections, **kwargs):
 156       if not collections:
 157         collections = [ops.GraphKeys.LOCAL_VARIABLES]
 158       elif ops.GraphKeys.GLOBAL_VARIABLES in collections:
 159         collections = list(collections)
 160         collections.append(ops.GraphKeys.LOCAL_VARIABLES)
 161         collections.remove(ops.GraphKeys.GLOBAL_VARIABLES)
 162       return next_creator(collections=collections, **kwargs)
 163 
 164     # theta = theta - lr * grad
 165     with variable_scope.variable_creator_scope(_variable_creator):
 166       local_update_op = self._opt.apply_gradients(grads_and_vars)
 167 
 168     # a = a + grad
 169     update_ops = []
 170     update_ops.append(local_update_op)
 171     grad_vars = [self._grad_map[var] for var in local_vars]
 172     for g, grad_var in zip(grads, grad_vars):
 173       update_ops.append(state_ops.assign_add(grad_var, g))
 174 
 175     global_center_vars = [self._global_map[var] for var in local_vars]
 176 
 177     # update global variables.
 178     def _Update_global_variables():
 179       global_norm = []
 180       # a = a / t
 181       for g in grad_vars:
 182         global_norm.append(state_ops.assign(g, g / self._period))
 183       # apply
 184       with ops.control_dependencies(global_norm):
 185         apply_global_op = self._opt.apply_gradients(
 186             zip(grad_vars, global_center_vars))
 187 
 188       # pull
 189       with ops.control_dependencies([apply_global_op]):
 190         update_ops = []
 191         if global_step:
 192           with ops.colocate_with(global_step):
 193             update_ops.append(state_ops.assign_add(global_step, 1))
 194 
 195         for lvar in local_vars:
 196           g_val = self._global_map[lvar].read_value()
 197           update_ops.append(state_ops.assign(lvar, g_val))
 198         for grad_var in grad_vars:
 199           update_ops.append(
 200               state_ops.assign(grad_var, array_ops.zeros_like(grad_var)))
 201         variable_update = control_flow_ops.group(*(update_ops))
 202       return variable_update
 203 
 204     local_update = state_ops.assign_add(
 205         self._local_step, 1, name='local_step_update').op
 206 
 207     with ops.control_dependencies([local_update]):
 208       condition = math_ops.equal(
 209           math_ops.mod(self._local_step, self._period), 0)
 210     with ops.control_dependencies(update_ops):
 211       conditional_update = control_flow_ops.cond(
 212           condition, _Update_global_variables, control_flow_ops.no_op)
 213     return conditional_update
 214 
 215   def get_init_op(self, task_index):
 216     """Returns the op to let all the local variables and local center
 217 
 218     variables equal to the global center variables before the training begins
 219     """
 220     init_ops = []
 221     local_vars = variables.trainable_variables()
 222     global_center_vars = [self._global_map[var] for var in local_vars]
 223     grad_vars = [self._grad_map[var] for var in local_vars]
 224     if not (local_vars and global_center_vars and grad_vars):
 225       raise ValueError('The lists of local_variables, global_center_variables,'
 226                        'grad_center_variables should not be empty')
 227     for lvar, gc_var in zip(local_vars, global_center_vars):
 228       init_ops.append(state_ops.assign(lvar, gc_var))
 229     for g in grad_vars:
 230       init_ops.append(state_ops.assign(g, array_ops.zeros_like(g)))
 231     init_op = control_flow_ops.group(*(init_ops))
 232     return init_op

 234   def make_session_run_hook(self, is_chief, task_index):
 235     """Creates a hook to handle AGNOptimizerHook ops such as initialization."""
 236     return _AGNOptimizerHook(self, is_chief, task_index)

 239 class _AGNOptimizerHook(session_run_hook.SessionRunHook):
 240 
 241   def __init__(self, agn_optimizer, is_chief, task_index):
 242     """Creates hook to handle AGNOptimizer initialization ops.
 243 
 244     Args:
 245       agn_optimizer: `AGNOptimizer` which this hook will initialize.
 246       is_chief: `Bool`, whether is this a chief replica or not.
 247       task_index: int, task_index of worker
 248     """
 249     self._agn_optimizer = agn_optimizer
 250     self._is_chief = is_chief
 251     self._task_index = task_index
 252 
 253   def begin(self):
 254     self._local_init_op = variables.local_variables_initializer()
 255     self._global_init_op = None
 256     if self._is_chief:
 257       self._global_init_op = variables.global_variables_initializer()
 258     self._variable_init_op = self._agn_optimizer.get_init_op(self._task_index)
 259 
 260   def after_create_session(self, session, coord):
 261     """Run initialization ops"""
 262     session.run(self._variable_init_op)
