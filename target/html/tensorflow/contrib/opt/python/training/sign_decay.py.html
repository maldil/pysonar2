  31 def get_linear_decay_fn(decay_steps):
  32   """Returns a function that computes a linear decay.
  33 
  34   This decay computes linear annealing:
  35     max(0, (decay_steps - global_step) / decay_steps)
  36 
  37   Example usage:
  38   ```
  39   decay_steps = 1000
  40   linear_decay_fn = get_linear_decay_fn(decay_steps)
  41   decayed = linear_decay_fn(global_step)
  42   x *= decayed
  43   ```
  44   Args:
  45     decay_steps: number of steps to decay over.
  46   Returns:
  47     linear_decay_fn: a function that computes the linear decay.
  48   """
  49   # pylint:disable=missing-docstring
  50   def linear_decay_fn(global_step):
  51     if global_step is None:
  52       raise ValueError("global_step is required for linear_decay.")
  53     global_step = math_ops.minimum(global_step, decay_steps)
  54     remaining_steps = math_ops.to_int32(decay_steps) - math_ops.to_int32(
  55         global_step)
  56     decayed = math_ops.to_float(remaining_steps) / math_ops.to_float(
  57         decay_steps)
  58     return math_ops.maximum(0.0, decayed)
  59   # pylint:enable=missing-docstring
  60   return linear_decay_fn

  63 def get_cosine_decay_fn(decay_steps, num_periods=0.5, zero_after=None):
  64   """Returns a function that computes a cosine decay.
  65 
  66   This decay computes cosine annealing:
  67     0.5 * (1.0 + cos(2.0 * pi * num_periods * global_step / decay_steps))
  68 
  69   This decay can be used to decay the sign quantity in the AddSign and PowerSign
  70   optimizers discovered in
  71   [Bello et al., ICML 2017] Neural Optimizer Search with RL.
  72 
  73   Example usage:
  74   ```
  75   decay_steps = 1000
  76   num_periods = 2
  77   cosine_decay_fn = get_cosine_decay_fn(decay_steps, num_periods=num_periods)
  78   decayed = cosine_decay_fn(global_step)
  79   x *= decayed
  80   ```
  81   Args:
  82     decay_steps: number of steps to decay over.
  83     num_periods: number of periods for cosine signal. 0.5 by default,
  84       which maps the last decay step to 0.
  85     zero_after: if not None, number after which the decay function
  86       will just return 0.
  87   Returns:
  88     cosine_decay_fn: a function that computes the cosine decay.
  89   """
  90   # pylint:disable=missing-docstring
  91   def cosine_decay_fn(global_step):
  92     if global_step is None:
  93       raise ValueError("global_step is required for cosine_decay.")
  94     global_step = math_ops.minimum(global_step, decay_steps)
  95     completed_fraction = math_ops.to_float(global_step) / math_ops.to_float(
  96         decay_steps)
  97     fraction = 2.0 * num_periods * completed_fraction
  98     decayed = 0.5 * (
  99         1.0 + math_ops.cos(constant_op.constant(math.pi) * fraction))
 100     if zero_after is not None:
 101       decayed = array_ops.where(
 102           math_ops.greater_equal(fraction, 2 * zero_after), 0.0, decayed)
 103     return decayed
 104   # pylint:enable=missing-docstring
 105   return cosine_decay_fn

 108 def get_restart_decay_fn(decay_steps, num_periods=1, zero_after=None):
 109   """Returns a function that computes a restart decay.
 110 
 111   This decay computes
 112     0.5 * (1.0 + cos(pi * (num_periods * global_step) % num_training_steps))
 113 
 114   This is a simplified version of the restart decay introduced in
 115   "SGDR: Stochastic Gradient Descent with Warm Restarts"
 116   by Ilya Loshchilov & Frank Hutter, Proceedings of
 117   ICLR'2017, available at https://arxiv.org/pdf/1608.03983.pdf
 118 
 119   This decay can be used to decay the sign quantity in the AddSign and PowerSign
 120   optimizers discovered in
 121   [Bello et al., ICML 2017] Neural Optimizer Search with RL.
 122 
 123   Example usage:
 124   ```
 125   decay_steps = 1000
 126   num_periods = 2.0
 127   restart_decay_fn = get_restart_decay_fn(decay_steps,
 128                                           num_periods=num_periods)
 129   decayed = restart_decay_fn(global_step)
 130   x *= decayed
 131   ```
 132   Args:
 133     decay_steps: number of steps to decay over.
 134     num_periods: number of periods for cosine signal. 1 by default,
 135       which maps the last decay step to 0.
 136     zero_after: if not None, number after which the decay function
 137       will return 0.
 138   Returns:
 139     restart_decay_fn: a function that computes the restart decay.
 140   """
 141   # pylint:disable=missing-docstring
 142   def restart_decay_fn(global_step):
 143     if global_step is None:
 144       raise ValueError("global_step is required for cosine_decay.")
 145     global_step = math_ops.minimum(global_step, decay_steps)
 146     num = math_ops.mod(num_periods * math_ops.to_float(global_step),
 147                        decay_steps)
 148     fraction = num / math_ops.to_float(decay_steps)
 149     decayed = 0.5 * (
 150         1.0 + math_ops.cos(constant_op.constant(math.pi) * fraction))
 151     if zero_after is not None:
 152       tmp = math_ops.to_float(
 153           num_periods * global_step) / math_ops.to_float(decay_steps)
 154       decayed = array_ops.where(
 155           math_ops.greater_equal(tmp, zero_after), 0.0, decayed)
 156     return decayed
 157   # pylint:enable=missing-docstring
 158   return restart_decay_fn
