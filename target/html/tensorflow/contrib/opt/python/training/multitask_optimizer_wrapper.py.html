  35 def _is_all_zeros(grad):
  36   all_zeros = math_ops.equal(math_ops.count_nonzero(grad), 0)
  37   return all_zeros
  38 
  39 
  40 def _get_wrapper(fn, opt):
  41 
  42   def wrapper(self, grad, *args, **kwargs):  # pylint: disable=unused-argument
  43     all_zeros = _is_all_zeros(grad)
  44     def call_fn():
  45       with ops.control_dependencies([fn(grad, *args, **kwargs)]):
  46         return control_flow_ops.no_op()
  47     return control_flow_ops.cond(all_zeros, control_flow_ops.no_op, call_fn)
  48 
  49   wrapper = types.MethodType(wrapper, opt)
  50   return wrapper

  53 class MultitaskOptimizerWrapper(object):
  54   """Optimizer wrapper making all-zero gradients harmless.
  55 
  56   This might be useful when a multi-task loss is used,
  57   and some components of the loss might be
  58   not present (e.g. masked out) in some training batches.
  59   Technically their gradient would be zero,
  60   which would normally affect the optimizer state
  61   (e.g. push running average to zero).
  62   However this is not the desired behaviour,
  63   since the missing loss component
  64   should be treated as unknown rather than zero.
  65 
  66   This wrapper filters out all-zero gradient tensors,
  67   therefore preserving the optimizer state.
  68 
  69   If gradient clipping by global norm is used,
  70   the provided function clip_gradients_by_global_norm
  71   should be used (and specified explicitly by the user).
  72   Otherwise the global norm would be underestimated
  73   because of all-zero tensors that should be ignored.
  74 
  75   The gradient calculation and application
  76   are delegated to an underlying optimizer.
  77   The gradient application is altered only for all-zero tensors.
  78 
  79   Example:
  80   ```python
  81   momentum_optimizer = tf.train.MomentumOptimizer(
  82     learning_rate, momentum=0.9)
  83   multitask_momentum_optimizer = tf.contrib.opt.MultitaskOptimizerWrapper(
  84     momentum_optimizer)
  85   gradvars = multitask_momentum_optimizer.compute_gradients(
  86     loss)
  87   gradvars_clipped, _ = tf.contrib.opt.clip_gradients_by_global_norm(
  88     gradvars, 15.0)
  89   train_op = multitask_momentum_optimizer.apply_gradients(
  90     gradvars_clipped, global_step=batch)
  91   ```
  92   """
  93 
  94   def __init__(self, opt):
  95     """Constructor.
  96 
  97     Args:
  98       opt: an instance of a class that implements tf.train.Optimizer.
  99     """
 100     if not isinstance(opt, optimizer.Optimizer):
 101       raise TypeError(
 102           'Supplied optimizer must be an instance of tf.train.Optimizer')
 103     self._opt = opt
 104     overridden_methods = ('_apply_dense', '_resource_apply_dense',
 105                           '_apply_sparse', '_resource_apply_sparse')
 106     for name in overridden_methods:
 107       fn = getattr(self._opt, name)
 108       wrapper = _get_wrapper(fn, self._opt)
 109       setattr(self._opt, name, wrapper)
 110 
 111   def __getattr__(self, name):
 112     return getattr(self._opt, name)

 115 def clip_gradients_by_global_norm(gradients_variables, clip_norm=20.):
 116   """Clips gradients of a multitask loss by their global norm.
 117 
 118   Ignores all-zero tensors when computing the global norm.
 119 
 120   Args:
 121     gradients_variables: a list of pairs (gradient, variable).
 122     clip_norm: a float Tensor, the global norm to clip on. Default is 20.0.
 123 
 124   Returns:
 125     list: A list of pairs of the same type as gradients_variables,.
 126     fixed_global_norm: A 0-D (scalar) Tensor representing the global norm.
 127   """
 128   gradients, variables = six.moves.zip(*gradients_variables)
 129 
 130   def _replace_nonexisting_grad(grad):
 131     if grad is None:
 132       return grad
 133     all_zeros = _is_all_zeros(grad)
 134     return control_flow_ops.cond(
 135         all_zeros,
 136         lambda: array_ops.zeros([], dtype=dtypes.as_dtype(grad.dtype)),
 137         lambda: grad)
 138 
 139   nonzero_gradients = [_replace_nonexisting_grad(g) for g in gradients]
 140   fixed_global_norm = clip_ops.global_norm(nonzero_gradients)
 141   gradients, _ = clip_ops.clip_by_global_norm(
 142       gradients, clip_norm, use_norm=fixed_global_norm)
 143   return list(six.moves.zip(gradients, variables)), fixed_global_norm
