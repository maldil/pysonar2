  16 from __future__ import absolute_import
  17 from __future__ import division
  18 from __future__ import print_function

  20 from tensorflow.python.ops import math_ops
  21 from tensorflow.python.training import adagrad
  22 from tensorflow.python.training import training_ops
  23 from tensorflow.python.util import tf_contextlib
  24 
  25 
  26 class RegAdagradOptimizer(adagrad.AdagradOptimizer):
  27   """RegAdagrad: Adagrad with updates that optionally skip updating the slots.
  28 
  29   This is meant to address the problem of additional regularization terms in the
  30   loss function affecting learning rate decay and causing hyper-param
  31   entanglement. Example usage:
  32 
  33     loss = tf.nn.cross_entropy(x, labels)
  34     reg_loss = reg_strength * tf.reduce_sum(x * x)
  35     opt = tf.contrib.opt.RegAdagradOptimizer(learning_rate)
  36     loss_update = opt.minimize(loss)
  37     with opt.avoid_updating_slots():
  38       reg_update = opt.minimize(reg_loss)
  39     total_update = tf.group([loss_update, reg_update])
  40 
  41     # ...
  42 
  43     sess.run(total_update, ...)
  44   """
  45 
  46   def __init__(self,
  47                learning_rate,
  48                initial_accumulator_value=0.1,
  49                use_locking=False,
  50                name="RegAdagrad"):
  51     super(RegAdagradOptimizer, self).__init__(
  52         learning_rate,
  53         initial_accumulator_value=initial_accumulator_value,
  54         use_locking=use_locking,
  55         name=name)
  56     self._should_update_slots = True
  57 
  58   @tf_contextlib.contextmanager
  59   def avoid_updating_slots(self):
  60     old = self._should_update_slots
  61     self._should_update_slots = False
  62     try:
  63       yield
  64     finally:
  65       self._should_update_slots = old
  66 
  67   def _apply_dense(self, grad, var):
  68     acc = self.get_slot(var, "accumulator")
  69     return training_ops.apply_adagrad(
  70         var,
  71         acc,
  72         math_ops.cast(self._learning_rate_tensor, var.dtype.base_dtype),
  73         grad,
  74         use_locking=self._use_locking,
  75         update_slots=self._should_update_slots)
  76 
  77   def _resource_apply_dense(self, grad, var, update_slots=True):
  78     acc = self.get_slot(var, "accumulator")
  79     return training_ops.resource_apply_adagrad(
  80         var.handle,
  81         acc.handle,
  82         math_ops.cast(self._learning_rate_tensor, grad.dtype.base_dtype),
  83         grad,
  84         use_locking=self._use_locking,
  85         update_slots=self._should_update_slots)
  86 
  87   def _apply_sparse(self, grad, var, update_slots=True):
  88     acc = self.get_slot(var, "accumulator")
  89     return training_ops.sparse_apply_adagrad(
  90         var,
  91         acc,
  92         math_ops.cast(self._learning_rate_tensor, var.dtype.base_dtype),
  93         grad.values,
  94         grad.indices,
  95         use_locking=self._use_locking,
  96         update_slots=self._should_update_slots)
  97 
  98   def _resource_apply_sparse(self, grad, var, indices, update_slots=True):
  99     acc = self.get_slot(var, "accumulator")
 100     return training_ops.resource_sparse_apply_adagrad(
 101         var.handle,
 102         acc.handle,
 103         math_ops.cast(self._learning_rate_tensor, grad.dtype),
 104         grad,
 105         indices,
 106         use_locking=self._use_locking,
 107         update_slots=self._should_update_slots)
