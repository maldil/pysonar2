  36 class ModelAverageCustomGetter(object):
  37   """Custom_getter class is used to do.
  38 
  39   1. Change trainable variables to local collection and place them at worker
  40     device
  41   2. Generate global variables
  42     Notice that the class should be used with tf.replica_device_setter,
  43     so that the global center variables and global step variable can be placed
  44     at ps device. Besides, use 'tf.get_variable' instead of 'tf.Variable' to
  45     use this custom getter.
  46 
  47   For example,
  48   ma_custom_getter = ModelAverageCustomGetter(worker_device)
  49   with tf.device(
  50     tf.train.replica_device_setter(
  51       worker_device=worker_device,
  52       ps_device="/job:ps/cpu:0",
  53       cluster=cluster)),
  54     tf.variable_scope('',custom_getter=ma_custom_getter):
  55     hid_w = tf.get_variable(
  56       initializer=tf.truncated_normal(
  57           [IMAGE_PIXELS * IMAGE_PIXELS, FLAGS.hidden_units],
  58           stddev=1.0 / IMAGE_PIXELS),
  59       name="hid_w")
  60     hid_b = tf.get_variable(initializer=tf.zeros([FLAGS.hidden_units]),
  61                             name="hid_b")
  62   """
  63 
  64   def __init__(self, worker_device):
  65     """Create a new `ModelAverageCustomGetter`.
  66 
  67     Args:
  68       worker_device: String.  Name of the `worker` job.
  69     """
  70     self._worker_device = worker_device
  71     self._local_2_global = {}
  72 
  73   def __call__(self, getter, name, trainable, collections, *args, **kwargs):
  74     if trainable:
  75       with ops.device(self._worker_device):
  76         local_var = getter(
  77             name,
  78             trainable=True,
  79             collections=[ops.GraphKeys.LOCAL_VARIABLES],
  80             *args,
  81             **kwargs)
  82 
  83       global_variable = variable_scope.variable(
  84           name="%s/%s" % (GLOBAL_VARIABLE_NAME, name),
  85           initial_value=local_var.initialized_value(),
  86           trainable=False,
  87           collections=[ops.GraphKeys.GLOBAL_VARIABLES])
  88 
  89       self._local_2_global[local_var] = global_variable
  90       return local_var
  91     else:
  92       kwargs['trainable'] = trainable
  93       kwargs['collections'] = collections
  94       if ops.GraphKeys.LOCAL_VARIABLES in collections:
  95         with ops.device(self._worker_device):
  96           return getter(name, *args, **kwargs)
  97       else:
  98         return getter(name, *args, **kwargs)

 101 class ModelAverageOptimizer(optimizer.Optimizer):
 102   """Wrapper optimizer that implements the Model Average algorithm.
 103 
 104   This is a sync optimizer. During the training, each worker will update
 105   the local variables and maintains its own local_step, which starts from 0
 106   and is incremented by 1 after each update of local variables. Whenever the
 107   interval_steps divides the local step, the local variables from all the
 108   workers will be averaged and assigned to global center variables. Then the
 109   local variables will be assigned by global center variables.
 110   """
 111 
 112   def __init__(self,
 113                opt,
 114                num_worker,
 115                is_chief,
 116                ma_custom_getter,
 117                interval_steps=100,
 118                use_locking=True,
 119                name="ModelAverageOptimizer"):
 120     """Construct a new model average optimizer.
 121 
 122     Args:
 123       opt: The actual optimizer that will be used to update local variables
 124       num_worker: The number of workers
 125       is_chief: whether chief worker
 126       ma_custom_getter: ModelAverageCustomGetter
 127       interval_steps: An int point value to controls the frequency of the
 128         average of local variables
 129       use_locking: If True use locks for update operations
 130       name: string. Optional name of the returned operation
 131     """
 132     super(ModelAverageOptimizer, self).__init__(use_locking, name)
 133     self._opt = opt
 134     self._num_worker = num_worker
 135     self._is_chief = is_chief
 136     self._local_2_global = ma_custom_getter._local_2_global  # pylint:disable=protected-access
 137     self._interval_steps = interval_steps
 138     self._accumulator_list = []
 139     self._chief_init_op = None
 140 
 141     self._local_step = variable_scope.get_variable(
 142         initializer=0,
 143         trainable=False,
 144         collections=[ops.GraphKeys.LOCAL_VARIABLES],
 145         name="local_step")
 146 
 147     self._opt._prepare()  # pylint:disable=protected-access
 148 
 149   def compute_gradients(self, *args, **kwargs):
 150     """Compute gradients of "loss" for the variables in "var_list".
 151 
 152     This simply wraps the compute_gradients() from the real optimizer.
 153 
 154     Args:
 155       *args: Arguments for compute_gradients().
 156       **kwargs: Keyword arguments for compute_gradients().
 157 
 158     Returns:
 159       A list of (gradient, variable) pairs.
 160     """
 161     return self._opt.compute_gradients(*args, **kwargs)
 162 
 163   def _local_vars_update(self, var_list):
 164     """Get the update ops for the local variables in "var_list".
 165 
 166     Args:
 167       var_list: Optional list or tuple of 'tf.Variable' to update
 168 
 169     Returns:
 170       An update op
 171 
 172     Raises:
 173       ValueError: if var_list is empty.
 174     """
 175     if not var_list:
 176       raise ValueError("The list of local_variables should not be empty")
 177     update_ops = []
 178     global_center_vars = [self._local_2_global[var] for var in var_list]
 179     for lvar, gvar in zip(var_list, global_center_vars):
 180       with ops.device(lvar.device):
 181         update_ops.append(state_ops.assign(lvar, gvar.read_value()))
 182     return control_flow_ops.group(*(update_ops))
 183 
 184   def apply_gradients(self, grads_and_vars, global_step=None, name=None):
 185     """Apply gradients to variables.
 186 
 187     This contains most of the synchronization implementation and also wraps the
 188     apply_gradients() from the real optimizer. The chief work updates global
 189     variables.
 190 
 191     Args:
 192       grads_and_vars: List of (gradient, variable) pairs as returned by
 193         compute_gradients().
 194       global_step: Optional Variable to increment by one after the
 195         variables have been updated.
 196       name: Optional name for the returned operation.  Default to the
 197         name passed to the Optimizer constructor.
 198 
 199     Returns:
 200       A conditional 'Operation' that update both local and global variables or
 201       just local variables
 202 
 203     Raises:
 204       ValueError: If the grads_and_vars is empty.
 205       ValueError: If global step is not provided, the staleness cannot be
 206         checked.
 207     """
 208 
 209     # update local variables
 210     if not grads_and_vars:
 211       raise ValueError("Must supply at least one variable")
 212     if global_step is None:
 213       raise ValueError("Global step is required")
 214 
 215     apply_updates = self._opt.apply_gradients(grads_and_vars)
 216     with ops.control_dependencies([apply_updates]):
 217       local_update = state_ops.assign_add(
 218           self._local_step, 1, name="local_step_update").op
 219 
 220     # update global variables.
 221     def _update_global_variables():  # pylint: disable=missing-docstring
 222       local_vars = [v for g, v in grads_and_vars if g is not None]
 223       global_vars = [self._local_2_global[v] for v in local_vars]
 224       # sync queue
 225       with ops.colocate_with(global_step):
 226         sync_queue = data_flow_ops.FIFOQueue(
 227             -1, [dtypes.bool], shapes=[[]], shared_name="sync_queue")
 228       train_ops = []
 229       aggregated_vars = []
 230       with ops.name_scope(None, self._name + "/global"):
 231         for var, gvar in zip(local_vars, global_vars):
 232           # pylint: disable=protected-access
 233           with ops.device(gvar.device):
 234             if isinstance(var._ref(), ops.Tensor):
 235               var_accum = data_flow_ops.ConditionalAccumulator(
 236                   var.dtype,
 237                   shape=var.get_shape(),
 238                   shared_name=gvar.name + "/var_accum")
 239               train_ops.append(
 240                   var_accum.apply_grad(var._ref(), local_step=global_step))
 241               aggregated_vars.append(var_accum.take_grad(self._num_worker))
 242             else:
 243               raise ValueError("Unknown local variable type!")
 244             self._accumulator_list.append((var_accum, gvar.device))
 245       # chief worker updates global vars and enqueues tokens to the sync queue
 246       if self._is_chief:
 247         update_ops = []
 248         with ops.control_dependencies(train_ops):
 249           for avg_var, gvar in zip(aggregated_vars, global_vars):
 250             with ops.device(gvar.device):
 251               update_ops.append(state_ops.assign(gvar, avg_var))
 252           with ops.device(global_step.device):
 253             update_ops.append(state_ops.assign_add(global_step, 1))
 254         with ops.control_dependencies(update_ops), ops.device(
 255             global_step.device):
 256           tokens = array_ops.fill([self._num_worker - 1],
 257                                   constant_op.constant(False))
 258           sync_op = sync_queue.enqueue_many(tokens)
 259       else:
 260         with ops.control_dependencies(train_ops), ops.device(
 261             global_step.device):
 262           sync_op = sync_queue.dequeue()
 263 
 264       with ops.control_dependencies([sync_op]):
 265         local_update_op = self._local_vars_update(local_vars)
 266       return local_update_op
 267 
 268     with ops.control_dependencies([local_update]):
 269       condition = math_ops.equal(
 270           math_ops.mod(self._local_step, self._interval_steps), 0)
 271       conditional_update = control_flow_ops.cond(
 272           condition, _update_global_variables, control_flow_ops.no_op)
 273 
 274     chief_init_ops = []
 275     for accum, dev in self._accumulator_list:
 276       with ops.device(dev):
 277         chief_init_ops.append(
 278             accum.set_global_step(global_step, name="SetGlobalStep"))
 279     self._chief_init_op = control_flow_ops.group(*(chief_init_ops))
 280 
 281     return conditional_update
 282 
 283   def get_init_op(self):
 284     """Returns the op.
 285 
 286     This method lets all the local variables equal to the global
 287     variables before the training begins.
 288     """
 289     return self._local_vars_update(variables.trainable_variables())
 290 
 291   def make_session_run_hook(self):
 292     """Creates a hook to handle ModelAverage ops such as initialization."""
 293     return _ModelAverageOptimizerHook(self, self._is_chief)

 296 class _ModelAverageOptimizerHook(session_run_hook.SessionRunHook):  # pylint: disable=missing-docstring
 297 
 298   def __init__(self, ma_optimizer, is_chief):
 299     """Creates hook to handle ModelAverageOptimizer initialization ops.
 300 
 301     Args:
 302       ma_optimizer: `ModelAverageOptimizer` which this hook will initialize.
 303       is_chief: `Bool`, whether is this a chief replica or not.
 304     """
 305     self._ma_optimizer = ma_optimizer
 306     self._is_chief = is_chief
 307 
 308   def begin(self):
 309     self._local_init_op = variables.local_variables_initializer()
 310     self._global_init_op = None
 311     if self._is_chief:
 312       self._global_init_op = variables.global_variables_initializer()
 313       self._chief_init_op = self._ma_optimizer._chief_init_op  # pylint: disable=protected-access
 314     self._variable_init_op = self._ma_optimizer.get_init_op()
