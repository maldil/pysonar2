  35 class DropStaleGradientOptimizer(optimizer.Optimizer):
  36   """Wrapper optimizer that checks and drops stale gradient.
  37 
  38   This optimizer records the global step for each worker before computing
  39   gradients and compares it with the global step at the time of applying the
  40   gradients. If the difference is larger than a threshold, it will drop all
  41   the computed gradients.
  42   """
  43 
  44   def __init__(self,
  45                opt,
  46                staleness,
  47                use_locking=False,
  48                name="DropStaleGradient"):
  49     """Constructs a new DropStaleGradientOptimizer.
  50 
  51     Args:
  52       opt: The actual optimizer that will be used to compute and apply the
  53            gradients. Must be one of the Optimizer classes.
  54       staleness: The maximum staleness allowed for the optimizer.
  55       use_locking: If `True` use locks for clip update operations.
  56       name: Optional name prefix for the operations created when applying
  57             gradients. Defaults to "DropStaleGradient".
  58     """
  59     super(DropStaleGradientOptimizer, self).__init__(use_locking, name)
  60     self._opt = opt
  61     self._staleness = staleness
  62 
  63   def compute_gradients(self, loss, *args, **kwargs):
  64     # Record current global step for worker.
  65     with ops.colocate_with(loss):
  66       self._local_step = training_util.get_global_step() + 0
  67 
  68     with ops.control_dependencies([self._local_step]):
  69       loss = gen_array_ops.identity(loss)
  70       return self._opt.compute_gradients(loss, *args, **kwargs)
  71 
  72   def get_slot(self, *args, **kwargs):
  73     return self._opt.get_slot(*args, **kwargs)
  74 
  75   def get_slot_names(self, *args, **kwargs):
  76     return self._opt.get_slot_names(*args, **kwargs)
  77 
  78   def apply_gradients(self, grads_and_vars, global_step=None, name=None):
  79     gradients = []
  80     # Number of stale gradients.
  81     with ops.colocate_with(global_step):
  82       stale_counter = variable_scope.get_variable(
  83           "stale_counter", [],
  84           initializer=init_ops.zeros_initializer(),
  85           trainable=False)
  86 
  87     def _AcceptGradientOp():
  88       with ops.control_dependencies(
  89           [self._opt.apply_gradients(
  90               grads_and_vars, global_step=global_step, name=name)]):
  91         return gen_array_ops.identity(0.0)
  92 
  93     def _DropGradientOp():
  94       return gen_array_ops.identity(1.0)
  95 
  96     for grad_and_var in grads_and_vars:
  97       grad = grad_and_var[0]
  98       if isinstance(grad, ops.Tensor):
  99         gradients.append(grad)
 100       elif grad is not None:
 101         gradients.append(grad.op)
 102 
 103     with ops.control_dependencies(gradients), ops.colocate_with(global_step):
 104       staleness = gen_array_ops.reshape(
 105           global_step - self._local_step, shape=())
 106 
 107     conditional_update = stale_counter.assign_add(control_flow_ops.cond(
 108         gen_math_ops.less_equal(staleness, self._staleness),
 109         _AcceptGradientOp, _DropGradientOp))
 110 
 111     summary.scalar(
 112         "Gradient staleness percentage",
 113         stale_counter / (math_ops.cast(global_step + 1, dtypes.float32)))
 114     return conditional_update
