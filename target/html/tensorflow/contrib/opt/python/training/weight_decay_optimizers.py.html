  33 class DecoupledWeightDecayExtension(object):
  34   """This class allows to extend optimizers with decoupled weight decay.
  35 
  36   It implements the decoupled weight decay described by Loshchilov & Hutter
  37   (https://arxiv.org/pdf/1711.05101.pdf), in which the weight decay is
  38   decoupled from the optimization steps w.r.t. to the loss function.
  39   For SGD variants, this simplifies hyperparameter search since it decouples
  40   the settings of weight decay and learning rate.
  41   For adaptive gradient algorithms, it regularizes variables with large
  42   gradients more than L2 regularization would, which was shown to yield better
  43   training loss and generalization error in the paper above.
  44 
  45   This class alone is not an optimizer but rather extends existing
  46   optimizers with decoupled weight decay. We explicitly define the two examples
  47   used in the above paper (SGDW and AdamW), but in general this can extend
  48   any OptimizerX by using
  49   `extend_with_weight_decay(OptimizerX, weight_decay=weight_decay)`.
  50   In order for it to work, it must be the first class the Optimizer with
  51   weight decay inherits from, e.g.
  52 
  53   ```python
  54   class AdamWOptimizer(DecoupledWeightDecayExtension, adam.AdamOptimizer):
  55     def __init__(self, weight_decay, *args, **kwargs):
  56       super(AdamWOptimizer, self).__init__(weight_decay, *args, **kwargs).
  57   ```
  58 
  59   Note that this extension decays weights BEFORE applying the update based
  60   on the gradient, i.e. this extension only has the desired behaviour for
  61   optimizers which do not depend on the value of'var' in the update step!
  62   """
  63 
  64   def __init__(self, weight_decay, **kwargs):
  65     """Construct the extension class that adds weight decay to an optimizer.
  66 
  67     Args:
  68       weight_decay: A `Tensor` or a floating point value, the factor by which
  69         a variable is decayed in the update step.
  70       **kwargs: Optional list or tuple or set of `Variable` objects to
  71         decay.
  72     """
  73     self._decay_var_list = None  # is set in minimize or apply_gradients
  74     self._weight_decay = weight_decay
  75     # The tensors are initialized in call to _prepare
  76     self._weight_decay_tensor = None
  77     super(DecoupledWeightDecayExtension, self).__init__(**kwargs)
  78 
  79   def minimize(self, loss, global_step=None, var_list=None,
  80                gate_gradients=optimizer.Optimizer.GATE_OP,
  81                aggregation_method=None, colocate_gradients_with_ops=False,
  82                name=None, grad_loss=None, decay_var_list=None):
  83     """Add operations to minimize `loss` by updating `var_list` with decay.
  84 
  85     This function is the same as Optimizer.minimize except that it allows to
  86     specify the variables that should be decayed using decay_var_list.
  87     If decay_var_list is None, all variables in var_list are decayed.
  88 
  89     For more information see the documentation of Optimizer.minimize.
  90 
  91     Args:
  92       loss: A `Tensor` containing the value to minimize.
  93       global_step: Optional `Variable` to increment by one after the
  94         variables have been updated.
  95       var_list: Optional list or tuple of `Variable` objects to update to
  96         minimize `loss`.  Defaults to the list of variables collected in
  97         the graph under the key `GraphKeys.TRAINABLE_VARIABLES`.
  98       gate_gradients: How to gate the computation of gradients.  Can be
  99         `GATE_NONE`, `GATE_OP`, or  `GATE_GRAPH`.
 100       aggregation_method: Specifies the method used to combine gradient terms.
 101         Valid values are defined in the class `AggregationMethod`.
 102       colocate_gradients_with_ops: If True, try colocating gradients with
 103         the corresponding op.
 104       name: Optional name for the returned operation.
 105       grad_loss: Optional. A `Tensor` holding the gradient computed for `loss`.
 106       decay_var_list: Optional list of decay variables.
 107 
 108     Returns:
 109       An Operation that updates the variables in `var_list`.  If `global_step`
 110       was not `None`, that operation also increments `global_step`.
 111 
 112     """
 113     self._decay_var_list = set(decay_var_list) if decay_var_list else False
 114     return super(DecoupledWeightDecayExtension, self).minimize(
 115         loss, global_step=global_step, var_list=var_list,
 116         gate_gradients=gate_gradients, aggregation_method=aggregation_method,
 117         colocate_gradients_with_ops=colocate_gradients_with_ops, name=name,
 118         grad_loss=grad_loss)
 119 
 120   def apply_gradients(self, grads_and_vars, global_step=None, name=None,
 121                       decay_var_list=None):
 122     """Apply gradients to variables and decay the variables.
 123 
 124     This function is the same as Optimizer.apply_gradients except that it
 125     allows to specify the variables that should be decayed using
 126     decay_var_list. If decay_var_list is None, all variables in var_list
 127     are decayed.
 128 
 129     For more information see the documentation of Optimizer.apply_gradients.
 130 
 131     Args:
 132       grads_and_vars: List of (gradient, variable) pairs as returned by
 133         `compute_gradients()`.
 134       global_step: Optional `Variable` to increment by one after the
 135         variables have been updated.
 136       name: Optional name for the returned operation.  Default to the
 137         name passed to the `Optimizer` constructor.
 138       decay_var_list: Optional list of decay variables.
 139 
 140     Returns:
 141       An `Operation` that applies the specified gradients. If `global_step`
 142       was not None, that operation also increments `global_step`.
 143     """
 144     self._decay_var_list = set(decay_var_list) if decay_var_list else False
 145     return super(DecoupledWeightDecayExtension, self).apply_gradients(
 146         grads_and_vars, global_step=global_step, name=name)
 147 
 148   def _prepare(self):
 149     weight_decay = self._weight_decay
 150     if callable(weight_decay):
 151       weight_decay = weight_decay()
 152     self._weight_decay_tensor = ops.convert_to_tensor(
 153         weight_decay, name="weight_decay")
 154     # Call the optimizers _prepare function.
 155     super(DecoupledWeightDecayExtension, self)._prepare()
 156 
 157   def _decay_weights_op(self, var):
 158     if not self._decay_var_list or var in self._decay_var_list:
 159       return var.assign_sub(self._weight_decay * var, self._use_locking)
 160     return control_flow_ops.no_op()
 161 
 162   def _decay_weights_sparse_op(self, var, indices, scatter_add):
 163     if not self._decay_var_list or var in self._decay_var_list:
 164       update = -self._weight_decay * array_ops.gather(var, indices)
 165       return scatter_add(var, indices, update, self._use_locking)
 166     return control_flow_ops.no_op()
 167 
 168   # Here, we overwrite the apply functions that the base optimizer calls.
 169   # super().apply_x resolves to the apply_x function of the BaseOptimizer.
 170   def _apply_dense(self, grad, var):
 171     with ops.control_dependencies([self._decay_weights_op(var)]):
 172       return super(DecoupledWeightDecayExtension, self)._apply_dense(grad, var)
 173 
 174   def _resource_apply_dense(self, grad, var):
 175     with ops.control_dependencies([self._decay_weights_op(var)]):
 176       return super(DecoupledWeightDecayExtension, self)._resource_apply_dense(
 177           grad, var)
 178 
 179   def _apply_sparse(self, grad, var):
 180     scatter_add = state_ops.scatter_add
 181     decay_op = self._decay_weights_sparse_op(var, grad.indices, scatter_add)
 182     with ops.control_dependencies([decay_op]):
 183       return super(DecoupledWeightDecayExtension, self)._apply_sparse(
 184           grad, var)
 185 
 186   def _resource_scatter_add(self, x, i, v, _=None):
 187     # last argument allows for one overflow argument, to have the same function
 188     # signature as state_ops.scatter_add
 189     with ops.control_dependencies(
 190         [resource_variable_ops.resource_scatter_add(x.handle, i, v)]):
 191       return x.value()
 192 
 193   def _resource_apply_sparse(self, grad, var, indices):
 194     scatter_add = self._resource_scatter_add
 195     decay_op = self._decay_weights_sparse_op(var, indices, scatter_add)
 196     with ops.control_dependencies([decay_op]):
 197       return super(DecoupledWeightDecayExtension, self)._resource_apply_sparse(
 198           grad, var, indices)

 201 def extend_with_decoupled_weight_decay(base_optimizer):
 202   """Factory function returning an optimizer class with decoupled weight decay.
 203 
 204   Returns an optimizer class. An instance of the returned class computes the
 205   update step of `base_optimizer` and additionally decays the weights.
 206   E.g., the class returned by
 207   `extend_with_decoupled_weight_decay(tf.train.AdamOptimizer)` is equivalent to
 208   `tf.contrib.opt.AdamWOptimizer`.
 209 
 210   The API of the new optimizer class slightly differs from the API of the
 211   base optimizer:
 212   - The first argument to the constructor is the weight decay rate.
 213   - `minimize` and `apply_gradients` accept the optional keyword argument
 214     `decay_var_list`, which specifies the variables that should be decayed.
 215     If `None`, all variables that are optimized are decayed.
 216 
 217   Usage example:
 218   ```python
 219   # MyAdamW is a new class
 220   MyAdamW = extend_with_decoupled_weight_decay(tf.train.AdamOptimizer)
 221   # Create a MyAdamW object
 222   optimizer = MyAdamW(weight_decay=0.001, learning_rate=0.001)
 223   sess.run(optimizer.minimize(loss, decay_variables=[var1, var2]))
 224 
 225   Note that this extension decays weights BEFORE applying the update based
 226   on the gradient, i.e. this extension only has the desired behaviour for
 227   optimizers which do not depend on the value of'var' in the update step!
 228   ```
 229 
 230   Args:
 231     base_optimizer: An optimizer class that inherits from tf.train.Optimizer.
 232 
 233   Returns:
 234     A new optimizer class that inherits from DecoupledWeightDecayExtension
 235     and base_optimizer.
 236   """
 237 
 238   class OptimizerWithDecoupledWeightDecay(DecoupledWeightDecayExtension,
 239                                           base_optimizer):
 240     """Base_optimizer with decoupled weight decay.
 241 
 242     This class computes the update step of `base_optimizer` and
 243     additionally decays the variable with the weight decay being decoupled from
 244     the optimization steps w.r.t. to the loss function, as described by
 245     Loshchilov & Hutter (https://arxiv.org/pdf/1711.05101.pdf).
 246     For SGD variants, this simplifies hyperparameter search since
 247     it decouples the settings of weight decay and learning rate.
 248     For adaptive gradient algorithms, it regularizes variables with large
 249     gradients more than L2 regularization would, which was shown to yield
 250     better training loss and generalization error in the paper above.
 251     """
 252 
 253     def __init__(self, weight_decay, *args, **kwargs):
 254       # super delegation is necessary here
 255       # pylint: disable=useless-super-delegation
 256       super(OptimizerWithDecoupledWeightDecay, self).__init__(
 257           weight_decay, *args, **kwargs)
 258       # pylint: enable=useless-super-delegation
 259 
 260   return OptimizerWithDecoupledWeightDecay

 263 @tf_export("contrib.opt.MomentumWOptimizer")
 264 class MomentumWOptimizer(DecoupledWeightDecayExtension,
 265                          momentum_opt.MomentumOptimizer):
 266   """Optimizer that implements the Momentum algorithm with weight_decay.
 267 
 268   This is an implementation of the SGDW optimizer described in "Fixing
 269   Weight Decay Regularization in Adam" by Loshchilov & Hutter
 270   (https://arxiv.org/abs/1711.05101)
 271   ([pdf])(https://arxiv.org/pdf/1711.05101.pdf).
 272   It computes the update step of `train.MomentumOptimizer` and additionally
 273   decays the variable. Note that this is different from adding
 274   L2 regularization on the variables to the loss. Decoupling the weight decay
 275   from other hyperparameters (in particular the learning rate) simplifies
 276   hyperparameter search.
 277 
 278   For further information see the documentation of the Momentum Optimizer.
 279 
 280   Note that this optimizer can also be instantiated as
 281   ```python
 282   extend_with_weight_decay(tf.train.MomentumOptimizer,
 283                            weight_decay=weight_decay)
 284   ```
 285   """
 286 
 287   def __init__(self, weight_decay, learning_rate, momentum,
 288                use_locking=False, name="MomentumW", use_nesterov=False):
 289     """Construct a new MomentumW optimizer.
 290 
 291     For further information see the documentation of the Momentum Optimizer.
 292 
 293     Args:
 294       weight_decay:  A `Tensor` or a floating point value.  The weight decay.
 295       learning_rate: A `Tensor` or a floating point value.  The learning rate.
 296       momentum: A `Tensor` or a floating point value.  The momentum.
 297       use_locking: If `True` use locks for update operations.
 298       name: Optional name prefix for the operations created when applying
 299         gradients.  Defaults to "Momentum".
 300       use_nesterov: If `True` use Nesterov Momentum.
 301         See [Sutskever et al., 2013](
 302         http://jmlr.org/proceedings/papers/v28/sutskever13.pdf).
 303         This implementation always computes gradients at the value of the
 304         variable(s) passed to the optimizer. Using Nesterov Momentum makes the
 305         variable(s) track the values called `theta_t + mu*v_t` in the paper.
 306 
 307     @compatibility(eager)
 308     When eager execution is enabled, learning_rate, weight_decay and momentum
 309     can each be a callable that takes no arguments and returns the actual value
 310     to use. This can be useful for changing these values across different
 311     invocations of optimizer functions.
 312     @end_compatibility
 313     """
 314     super(MomentumWOptimizer, self).__init__(
 315         weight_decay, learning_rate=learning_rate, momentum=momentum,
 316         use_locking=use_locking, name=name, use_nesterov=use_nesterov)

 319 @tf_export("contrib.opt.AdamWOptimizer")
 320 class AdamWOptimizer(DecoupledWeightDecayExtension, adam.AdamOptimizer):
 321   """Optimizer that implements the Adam algorithm with weight decay.
 322 
 323   This is an implementation of the AdamW optimizer described in "Fixing
 324   Weight Decay Regularization in Adam" by Loshchilov & Hutter
 325   (https://arxiv.org/abs/1711.05101)
 326   ([pdf])(https://arxiv.org/pdf/1711.05101.pdf).
 327 
 328   It computes the update step of `train.AdamOptimizer` and additionally decays
 329   the variable. Note that this is different from adding L2 regularization on
 330   the variables to the loss: it regularizes variables with large
 331   gradients more than L2 regularization would, which was shown to yield better
 332   training loss and generalization error in the paper above.
 333 
 334   For further information see the documentation of the Adam Optimizer.
 335 
 336   Note that this optimizer can also be instantiated as
 337   ```python
 338   extend_with_weight_decay(tf.train.AdamOptimizer, weight_decay=weight_decay)
 339   ```
 340   """
 341 
 342   def __init__(self, weight_decay, learning_rate=0.001, beta1=0.9, beta2=0.999,
 343                epsilon=1e-8, use_locking=False, name="AdamW"):
 344     """Construct a new AdamW optimizer.
 345 
 346     For further information see the documentation of the Adam Optimizer.
 347 
 348     Args:
 349       weight_decay:  A `Tensor` or a floating point value.  The weight decay.
 350       learning_rate: A Tensor or a floating point value.  The learning rate.
 351       beta1: A float value or a constant float tensor.
 352         The exponential decay rate for the 1st moment estimates.
 353       beta2: A float value or a constant float tensor.
 354         The exponential decay rate for the 2nd moment estimates.
 355       epsilon: A small constant for numerical stability. This epsilon is
 356         "epsilon hat" in the Kingma and Ba paper (in the formula just before
 357         Section 2.1), not the epsilon in Algorithm 1 of the paper.
 358       use_locking: If True use locks for update operations.
 359       name: Optional name for the operations created when applying gradients.
 360         Defaults to "Adam".
 361     """
 362     super(AdamWOptimizer, self).__init__(
 363         weight_decay, learning_rate=learning_rate, beta1=beta1, beta2=beta2,
 364         epsilon=epsilon, use_locking=use_locking, name=name)

 367 @tf_export("contrib.opt.ShampooWOptimizer")
 368 class ShampooWOptimizer(DecoupledWeightDecayExtension,
 369                         shampoo.ShampooOptimizer):
 370   """Optimizer that implements the Shampoo algorithm with weight decay.
 371 
 372   For further information see the documentation of the Shampoo Optimizer.
 373   """
 374 
 375   def __init__(self,
 376                weight_decay,
 377                global_step,
 378                max_matrix_size=768,
 379                gbar_decay=0.0,
 380                gbar_weight=1.0,
 381                mat_gbar_decay=1.0,
 382                mat_gbar_weight=1.0,
 383                learning_rate=1.0,
 384                svd_interval=1,
 385                precond_update_interval=1,
 386                epsilon=1e-4,
 387                alpha=0.5,
 388                use_iterative_root=False,
 389                use_locking=False,
 390                name="ShampooW"):
 391     """Construct a new ShampooW optimizer.
 392 
 393     For further information see the documentation of the Shampoo Optimizer.
 394 
 395     Args:
 396       weight_decay:  A `Tensor` or a floating point value.  The weight decay.
 397       global_step: tensorflow variable indicating the step.
 398       max_matrix_size: We do not perform SVD for matrices larger than this.
 399       gbar_decay:
 400       gbar_weight:  Used to update gbar: gbar[t] = gbar_decay[t] * gbar[t-1] +
 401         gbar_weight[t] * g[t]
 402       mat_gbar_decay:
 403       mat_gbar_weight:  Used to update mat_gbar: mat_gbar_j[t] =
 404         mat_gbar_decay[t] * mat_gbar_j[t-1] + mat_gbar_weight[t] * gg_j[t]
 405       learning_rate: Similar to SGD
 406       svd_interval: We should do SVD after this many steps. Default = 1, i.e.
 407         every step. Usually 20 leads to no loss of accuracy, and 50 or 100 is
 408         also OK. May also want more often early,
 409                     and less often later - set in caller as for example:
 410                     "svd_interval = lambda(T): tf.cond(
 411                         T < 2000, lambda: 20.0, lambda: 1000.0)"
 412       precond_update_interval: We should update the preconditioners after this
 413         many steps. Default = 1. Usually less than svd_interval.
 414       epsilon:  epsilon * I_n is added to each mat_gbar_j for stability
 415       alpha:  total power of the preconditioners.
 416       use_iterative_root: should the optimizer use SVD (faster) or the iterative
 417         root method (for TPU) for finding the roots of PSD matrices.
 418       use_locking: If `True` use locks for update operations.
 419       name: name of optimizer.
 420     """
 421     super(ShampooWOptimizer, self).__init__(
 422         weight_decay,
 423         global_step=global_step,
 424         max_matrix_size=max_matrix_size,
 425         gbar_decay=gbar_decay,
 426         gbar_weight=gbar_weight,
 427         mat_gbar_decay=mat_gbar_weight,
 428         learning_rate=learning_rate,
 429         svd_interval=svd_interval,
 430         precond_update_interval=precond_update_interval,
 431         epsilon=epsilon,
 432         alpha=alpha,
 433         use_iterative_root=use_iterative_root,
 434         use_locking=use_locking,
 435         name=name)
