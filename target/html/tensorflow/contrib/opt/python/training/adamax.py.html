  32 class AdaMaxOptimizer(adam.AdamOptimizer):
  33   """Optimizer that implements the AdaMax algorithm.
  34 
  35   Adamax is sometimes superior to adam, specially in models with embeddings,
  36   see [Kingma et al., 2014](http://arxiv.org/abs/1412.6980)
  37   ([pdf](http://arxiv.org/pdf/1412.6980.pdf)).
  38   """
  39 
  40   def __init__(self, learning_rate=0.001, beta1=0.9, beta2=0.999, epsilon=1e-8,
  41                use_locking=False, name="AdaMax"):
  42     """Construct a new AdaMax optimizer.
  43 
  44     Initialization:
  45 
  46     ```
  47     m_0 <- 0 (Initialize initial 1st moment vector)
  48     v_0 <- 0 (Initialize the exponentially weighted infinity norm)
  49     t <- 0 (Initialize timestep)
  50     ```
  51 
  52     The update rule for `variable` with gradient `g` uses an optimization
  53     described at the end of section 7.1 of the paper:
  54 
  55     ```
  56     t <- t + 1
  57 
  58     m_t <- beta1 * m_{t-1} + (1 - beta1) * g
  59     v_t <- max(beta2 * v_{t-1}, abs(g))
  60     variable <- variable - learning_rate / (1 - beta1^t) * m_t / (v_t + epsilon)
  61     ```
  62 
  63     Similar to AdamOptimizer, the epsilon is added for numerical stability
  64     (especially to get rid of division by zero when v_t = 0).
  65 
  66     Contrast to AdamOptimizer, the sparse implementation of this algorithm
  67     (used when the gradient is an IndexedSlices object, typically because of
  68     `tf.gather` or an embedding lookup in the forward pass) only updates
  69     variable slices and corresponding `m_t`, `v_t` terms when that part of
  70     the variable was used in the forward pass. This means that the sparse
  71     behavior is contrast to the dense behavior (similar to some momentum
  72     implementations which ignore momentum unless a variable slice was actually
  73     used).
  74 
  75     Args:
  76       learning_rate: A Tensor or a floating point value.  The learning rate.
  77       beta1: A float value or a constant float tensor.
  78         The exponential decay rate for the 1st moment estimates.
  79       beta2: A float value or a constant float tensor.
  80         The exponential decay rate for the exponentially weighted infinity norm.
  81       epsilon: A small constant for numerical stability.
  82       use_locking: If True use locks for update operations.
  83       name: Optional name for the operations created when applying gradients.
  84         Defaults to "AdaMax".
  85     """
  86     super(AdaMaxOptimizer, self).__init__(learning_rate, beta1, beta2,
  87                                           epsilon, use_locking, name)
  88 
  89   def _get_beta_accumulators(self):
  90     if context.executing_eagerly():
  91       graph = None
  92     else:
  93       graph = ops.get_default_graph()
  94     return self._get_non_slot_variable("beta1_power", graph=graph)
  95 
  96   def _create_slots(self, var_list):
  97     # Create the beta1 accumulators on the same device as the first
  98     # variable. Sort the var_list to make sure this device is consistent across
  99     # workers (these need to go on the same PS, otherwise some updates are
 100     # silently ignored).
 101     first_var = min(var_list, key=lambda x: x.name)
 102     self._create_non_slot_variable(initial_value=self._beta1,
 103                                    name="beta1_power",
 104                                    colocate_with=first_var)
 105 
 106     # Create slots for the first and second moments.
 107     for v in var_list:
 108       self._zeros_slot(v, "m", self._name)
 109       self._zeros_slot(v, "v", self._name)
 110 
 111   def _apply_dense(self, grad, var):
 112     m = self.get_slot(var, "m")
 113     v = self.get_slot(var, "v")
 114     beta1_power = self._get_beta_accumulators()
 115     return training_ops.apply_ada_max(
 116         var, m, v,
 117         math_ops.cast(beta1_power, var.dtype.base_dtype),
 118         math_ops.cast(self._lr_t, var.dtype.base_dtype),
 119         math_ops.cast(self._beta1_t, var.dtype.base_dtype),
 120         math_ops.cast(self._beta2_t, var.dtype.base_dtype),
 121         math_ops.cast(self._epsilon_t, var.dtype.base_dtype),
 122         grad, use_locking=self._use_locking).op
 123 
 124   def _resource_apply_dense(self, grad, var):
 125     m = self.get_slot(var, "m")
 126     v = self.get_slot(var, "v")
 127     beta1_power = self._get_beta_accumulators()
 128     return training_ops.resource_apply_ada_max(
 129         var.handle, m.handle, v.handle,
 130         math_ops.cast(beta1_power, grad.dtype.base_dtype),
 131         math_ops.cast(self._lr_t, grad.dtype.base_dtype),
 132         math_ops.cast(self._beta1_t, grad.dtype.base_dtype),
 133         math_ops.cast(self._beta2_t, grad.dtype.base_dtype),
 134         math_ops.cast(self._epsilon_t, grad.dtype.base_dtype),
 135         grad, use_locking=self._use_locking)
 136 
 137   def _apply_sparse_shared(self, grad, var, indices,
 138                            scatter_add, scatter_update):
 139     beta1_power = self._get_beta_accumulators()
 140     beta1_power = math_ops.cast(beta1_power, var.dtype.base_dtype)
 141     lr_t = math_ops.cast(self._lr_t, var.dtype.base_dtype)
 142     beta1_t = math_ops.cast(self._beta1_t, var.dtype.base_dtype)
 143     beta2_t = math_ops.cast(self._beta2_t, var.dtype.base_dtype)
 144     epsilon_t = math_ops.cast(self._epsilon_t, var.dtype.base_dtype)
 145     # m_t = beta1 * m + (1 - beta1) * g_t
 146     m = self.get_slot(var, "m")
 147     m_slice = array_ops.gather(m, indices)
 148     m_t_slice = m_slice * beta1_t + grad * (1 - beta1_t)
 149     with ops.control_dependencies([m_t_slice]):
 150       m_t = scatter_update(m, indices, m_t_slice)
 151     # u_t = max(beta2 * u, abs(g_t))
 152     v = self.get_slot(var, "v")
 153     v_slice = array_ops.gather(v, indices)
 154     v_t_slice = math_ops.maximum(v_slice * beta2_t, math_ops.abs(grad))
 155     with ops.control_dependencies([v_t_slice]):
 156       v_t = scatter_update(v, indices, v_t_slice)
 157     # theta_t = theta - lr / (1 - beta1^t) * m_t / u_t
 158     var_slice = -lr_t / (1 - beta1_power) * (m_t_slice /
 159                                              (v_t_slice + epsilon_t))
 160     with ops.control_dependencies([var_slice]):
 161       var_update = scatter_add(var, indices, var_slice)
 162     return control_flow_ops.group(*[var_update, m_t, v_t])
 163 
 164   def _apply_sparse(self, grad, var):
 165     return self._apply_sparse_shared(
 166         grad.values, var, grad.indices,
 167         lambda x, i, v: state_ops.scatter_add(  # pylint: disable=g-long-lambda
 168             x, i, v, use_locking=self._use_locking),
 169         lambda x, i, v: state_ops.scatter_update(  # pylint: disable=g-long-lambda
 170             x, i, v, use_locking=self._use_locking))
 171 
 172   def _resource_scatter_update(self, x, i, v):
 173     with ops.control_dependencies(
 174         [resource_variable_ops.resource_scatter_update(
 175             x.handle, i, v)]):
 176       return x.value()
 177 
 178   def _resource_apply_sparse(self, grad, var, indices):
 179     return self._apply_sparse_shared(
 180         grad, var, indices,
 181         self._resource_scatter_add, self._resource_scatter_update)
 182 
 183   def _finish(self, update_ops, name_scope):
 184     # Update the power accumulators.
 185     with ops.control_dependencies(update_ops):
 186       beta1_power = self._get_beta_accumulators()
 187       with ops.colocate_with(beta1_power):
 188         update_beta1 = beta1_power.assign(
 189             beta1_power * self._beta1_t, use_locking=self._use_locking)
 190     return control_flow_ops.group(*update_ops + [update_beta1],
 191                                   name=name_scope)
