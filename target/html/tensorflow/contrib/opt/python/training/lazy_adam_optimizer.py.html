  16 """Variant of the Adam optimizer that handles sparse updates more efficiently.
  17 
  18 Compared with the original Adam optimizer, the one in this file can provide a
  19 large improvement in model training throughput for some applications. However,
  20 it provides slightly different semantics than the original Adam algorithm, and
  21 may lead to different empirical results.
  22 """
  23 
  24 from __future__ import absolute_import
  25 from __future__ import division
  26 from __future__ import print_function
  27 
  28 from tensorflow.python.ops import array_ops
  29 from tensorflow.python.ops import control_flow_ops
  30 from tensorflow.python.ops import math_ops
  31 from tensorflow.python.ops import resource_variable_ops
  32 from tensorflow.python.ops import state_ops
  33 from tensorflow.python.training import adam
  34 
  35 
  36 class LazyAdamOptimizer(adam.AdamOptimizer):
  37   """Variant of the Adam optimizer that handles sparse updates more efficiently.
  38 
  39   The original Adam algorithm maintains two moving-average accumulators for
  40   each trainable variable; the accumulators are updated at every step.
  41   This class provides lazier handling of gradient updates for sparse variables.
  42   It only updates moving-average accumulators for sparse variable indices that
  43   appear in the current batch, rather than updating the accumulators for all
  44   indices. Compared with the original Adam optimizer, it can provide large
  45   improvements in model training throughput for some applications. However, it
  46   provides slightly different semantics than the original Adam algorithm, and
  47   may lead to different empirical results.
  48   """
  49 
  50   def _apply_sparse(self, grad, var):
  51     beta1_power, beta2_power = self._get_beta_accumulators()
  52     beta1_power = math_ops.cast(beta1_power, var.dtype.base_dtype)
  53     beta2_power = math_ops.cast(beta2_power, var.dtype.base_dtype)
  54     lr_t = math_ops.cast(self._lr_t, var.dtype.base_dtype)
  55     beta1_t = math_ops.cast(self._beta1_t, var.dtype.base_dtype)
  56     beta2_t = math_ops.cast(self._beta2_t, var.dtype.base_dtype)
  57     epsilon_t = math_ops.cast(self._epsilon_t, var.dtype.base_dtype)
  58     lr = (lr_t * math_ops.sqrt(1 - beta2_power) / (1 - beta1_power))
  59 
  60     # \\(m := beta1 * m + (1 - beta1) * g_t\\)
  61     m = self.get_slot(var, "m")
  62     m_t = state_ops.scatter_update(m, grad.indices,
  63                                    beta1_t * array_ops.gather(m, grad.indices) +
  64                                    (1 - beta1_t) * grad.values,
  65                                    use_locking=self._use_locking)
  66 
  67     # \\(v := beta2 * v + (1 - beta2) * (g_t * g_t)\\)
  68     v = self.get_slot(var, "v")
  69     v_t = state_ops.scatter_update(v, grad.indices,
  70                                    beta2_t * array_ops.gather(v, grad.indices) +
  71                                    (1 - beta2_t) * math_ops.square(grad.values),
  72                                    use_locking=self._use_locking)
  73 
  74     # \\(variable -= learning_rate * m_t / (epsilon_t + sqrt(v_t))\\)
  75     m_t_slice = array_ops.gather(m_t, grad.indices)
  76     v_t_slice = array_ops.gather(v_t, grad.indices)
  77     denominator_slice = math_ops.sqrt(v_t_slice) + epsilon_t
  78     var_update = state_ops.scatter_sub(var, grad.indices,
  79                                        lr * m_t_slice / denominator_slice,
  80                                        use_locking=self._use_locking)
  81     return control_flow_ops.group(var_update, m_t, v_t)
  82 
  83   def _resource_apply_sparse(self, grad, var, indices):
  84     beta1_power, beta2_power = self._get_beta_accumulators()
  85     beta1_power = math_ops.cast(beta1_power, var.dtype.base_dtype)
  86     beta2_power = math_ops.cast(beta2_power, var.dtype.base_dtype)
  87     lr_t = math_ops.cast(self._lr_t, var.dtype.base_dtype)
  88     beta1_t = math_ops.cast(self._beta1_t, var.dtype.base_dtype)
  89     beta2_t = math_ops.cast(self._beta2_t, var.dtype.base_dtype)
  90     epsilon_t = math_ops.cast(self._epsilon_t, var.dtype.base_dtype)
  91     lr = (lr_t * math_ops.sqrt(1 - beta2_power) / (1 - beta1_power))
  92 
  93     # \\(m := beta1 * m + (1 - beta1) * g_t\\)
  94     m = self.get_slot(var, "m")
  95     m_t_slice = beta1_t * array_ops.gather(m, indices) + (1 - beta1_t) * grad
  96     m_update_op = resource_variable_ops.resource_scatter_update(m.handle,
  97                                                                 indices,
  98                                                                 m_t_slice)
  99 
 100     # \\(v := beta2 * v + (1 - beta2) * (g_t * g_t)\\)
 101     v = self.get_slot(var, "v")
 102     v_t_slice = (beta2_t * array_ops.gather(v, indices) +
 103                  (1 - beta2_t) * math_ops.square(grad))
 104     v_update_op = resource_variable_ops.resource_scatter_update(v.handle,
 105                                                                 indices,
 106                                                                 v_t_slice)
 107 
 108     # \\(variable -= learning_rate * m_t / (epsilon_t + sqrt(v_t))\\)
 109     var_slice = lr * m_t_slice / (math_ops.sqrt(v_t_slice) + epsilon_t)
 110     var_update_op = resource_variable_ops.resource_scatter_sub(var.handle,
 111                                                                indices,
 112                                                                var_slice)
 113 
 114     return control_flow_ops.group(var_update_op, m_update_op, v_update_op)
