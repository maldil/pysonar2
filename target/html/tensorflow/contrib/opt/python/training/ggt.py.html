  31 class GGTOptimizer(optimizer_v2.OptimizerV2):
  32   """Optimizer that implements the GGT algorithm.
  33 
  34   GGT has an advantage over sgd and adam on large models with poor conditioning,
  35   for example language models and CNNs,
  36   see [[ABCHSZZ 2018]](https://arxiv.org/pdf/1806.02958.pdf).
  37   """
  38 
  39   def __init__(self,
  40                learning_rate=0.001,
  41                beta1=0.9,
  42                use_locking=False,
  43                name="GGT",
  44                window=10,
  45                eps=1e-4,
  46                svd_eps=1e-6,
  47                sigma_eps=1e-2):
  48     """Construct a new GGT optimizer.
  49 
  50     Initialization:
  51 
  52     ```
  53     t <- 0 (Initialize timestep)
  54     grad_buffer <- 0 (Initialize buffer for keeping past gradients)
  55     flat_grad <- 0 (Initialize flattened gradient that contains gradients of all
  56                     variables)
  57     m_0 <- 0 (Initialize 1st moment vector)
  58     ```
  59 
  60     Suppose all variables and their gradients are concatenated into vectors
  61     `flat_vars` and `flat_grad`. The update rule for `flat_vars`
  62     uses an optimization described at the beginning of section 2 of the paper:
  63 
  64     ```
  65     t <- t + 1
  66 
  67     m_t <- beta1 * m_{t-1} + (1 - beta1) * flat_grad
  68     grad_buffer[(t-1) % window, :] <- m_t
  69 
  70     M <- grad_buffer^T / sqrt(min(t, window))
  71     U, sigma, _ <- SVD(M^TM + I * svd_eps)
  72 
  73     sigma_sqrt_inv <- (sqrt(sigma) + sigma_eps)^(-3)
  74     sigma_sqrt_min <- min(sqrt(sigma))
  75 
  76     if sigma_sqrt_min > eps:
  77       new_step <- M U diag(sigma_sqrt_inv) U^T M^T m_t +
  78                   (m_t - M U diag(1/sigma) U^T M^T m_t) / sigma_sqrt_min
  79     else:
  80       new_step <- M U diag(sigma_sqrt_inv) U^T M^T m_t
  81 
  82     flat_vars <- flat_vars - learning_rate * new_step
  83     ```
  84 
  85     GGT provides the power of full-matrix adaptive regularization at a cost not
  86     much larger than SGD. As a result it is suited for large models where the
  87     gradient covariance matrix has a poor condition number that slows down first
  88     order methods.
  89     GGT uses the preconditioner from full-matrix AdaGrad, with gradient history
  90     attenuated exponentially as in Adam, and truncated to a window parameter.
  91     It has provable guarantees even for non-convex optimization that is never
  92     significantly worse than SGD and in some cases better.
  93 
  94     Args:
  95       learning_rate: A float hyperparameter. The learning rate.
  96       beta1: A float hyperparameter. The exponential decay rate for the 1st
  97         moment estimates.
  98       use_locking: If True use locks for update operations.
  99       name: Optional name for the operations created when applying gradients.
 100         Defaults to "GGT".
 101       window: An integer hyperparameter. The number of first moments to keep in
 102         computing the adaptive preconditioner.
 103       eps: A float hyperparameter. Used to truncate small eigenvalues of the
 104         gradient covariance matrix.
 105       svd_eps: A float hyperparameter. Used to stabilize SVD.
 106       sigma_eps: A float hyperparameter. Used to regularize matrix inversion.
 107     """
 108     super(GGTOptimizer, self).__init__(use_locking, name)
 109     self._set_hyper("lr", learning_rate)
 110     self._set_hyper("beta1", beta1)
 111     self._set_hyper("window", window)
 112     self._set_hyper("eps", eps)
 113     self._set_hyper("svd_eps", svd_eps)
 114     self._set_hyper("sigma_eps", sigma_eps)
 115 
 116     self.index_dict = {}
 117     self.shape_dict = {}
 118 
 119   def _create_vars(self, var_list, state):
 120     # Construct ordered dictionary for variable dimensions, sorted by name.
 121     shape_dict = {}
 122     for v in var_list:
 123       shape_dict[v.name] = np.prod(v.get_shape()).value
 124     self.shape_dict = collections.OrderedDict(
 125         sorted(shape_dict.items(), key=lambda t: t[0]))
 126 
 127     # Assign each variable its location in flat_grad. The locations are based on
 128     # the order of sorted names.
 129     idx = 0
 130     for v_name, v_dim in self.shape_dict.items():
 131       self.index_dict[v_name] = idx
 132       idx += v_dim
 133 
 134     state.create_non_slot(
 135         initial_value=math_ops.cast(0., dtype=var_list[0].dtype.base_dtype),
 136         name="global_step")
 137 
 138     # Buffer for keeping past gradients.
 139     window = state.get_hyper("window")
 140     grad_buffer_init = array_ops.zeros(
 141         [window, idx], dtype=var_list[0].dtype.base_dtype)
 142     state.create_non_slot(initial_value=grad_buffer_init, name="grad_buffer")
 143 
 144     state.create_non_slot(
 145         initial_value=array_ops.zeros(
 146             (idx,), dtype=var_list[0].dtype.base_dtype),
 147         name="moment1")
 148 
 149     # Flattened gradient that contains gradients for all variables in the model.
 150     state.create_non_slot(
 151         initial_value=array_ops.zeros(
 152             (idx,), dtype=var_list[0].dtype.base_dtype),
 153         name="flat_grad")
 154 
 155   def _get_global_step(self, state=None):
 156     if state is None:
 157       state = self._get_per_graph_state()
 158     return state.get_non_slot("global_step")
 159 
 160   def _get_moment1(self, state=None):
 161     if state is None:
 162       state = self._get_per_graph_state()
 163     return state.get_non_slot("moment1")
 164 
 165   def _get_grad_buffer(self, state=None):
 166     if state is None:
 167       state = self._get_per_graph_state()
 168     return state.get_non_slot("grad_buffer")
 169 
 170   def _get_flat_grad(self, state=None):
 171     if state is None:
 172       state = self._get_per_graph_state()
 173     return state.get_non_slot("flat_grad")
 174 
 175   def _apply_sparse(self, grad, var):
 176     raise NotImplementedError("Sparse gradient updates are not supported.")
 177 
 178   def _prepare(self, state):
 179     self._variables = []
 180 
 181   def _apply_dense(self, grad, var, state):
 182     self._variables.append(var)
 183     dim = self.shape_dict[var.name]
 184     start_index = self.index_dict[var.name]
 185     end_index = start_index + dim
 186 
 187     # Update flat_gradient at the index associated with the variable.
 188     flat_grad = self._get_flat_grad(state)
 189     new_flat_grad = array_ops.reshape(grad, [-1])
 190     flat_grad_updated = state_ops.scatter_update(
 191         flat_grad, math_ops.range(start_index, end_index), new_flat_grad)
 192 
 193     return flat_grad_updated
 194 
 195   def _resource_apply_dense(self, grad, var, state):
 196     self._variables.append(var)
 197     dim = self.shape_dict[var.name]
 198     start_index = self.index_dict[var.name]
 199     end_index = start_index + dim
 200 
 201     # Update flat_gradient at the index associated with the variable.
 202     flat_grad = self._get_flat_grad(state)
 203     new_flat_grad = array_ops.reshape(grad, [-1])
 204     flat_grad_updated = state_ops.scatter_update(
 205         flat_grad, math_ops.range(start_index, end_index), new_flat_grad)
 206 
 207     return flat_grad_updated
 208 
 209   def _finish(self, state):
 210     var_dtype = self._variables[0].dtype.base_dtype
 211     # Update global step.
 212     global_step = self._get_global_step(state)
 213     update_global_step = state_ops.assign_add(global_step, 1.)
 214 
 215     # Update the first moment estimate.
 216     beta1 = state.get_hyper("beta1", dtype=var_dtype)
 217     moment1 = self._get_moment1(state)
 218     flat_grad = self._get_flat_grad(state)
 219     # moment1_t := beta1 * moment1_{t-1} + (1 - beta1) * flat_grad_t
 220     update_moment1 = moment1.assign(beta1 * moment1 + (1. - beta1) * flat_grad)
 221 
 222     # Update the gradient buffer.
 223     window = state.get_hyper("window")
 224     grad_buffer = self._get_grad_buffer(state)
 225     next_grad_index = math_ops.floormod(
 226         math_ops.to_int32(update_global_step - 1.), window)
 227     # grad_buffer[(t-1) % window] := moment1_t
 228     update_grad_buffer = state_ops.scatter_update(grad_buffer, next_grad_index,
 229                                                   update_moment1)
 230 
 231     # Compute the update step.
 232     eps = state.get_hyper("eps", dtype=var_dtype)
 233     svd_eps = state.get_hyper("svd_eps", dtype=var_dtype)
 234     sigma_eps = state.get_hyper("sigma_eps", dtype=var_dtype)
 235     lr = state.get_hyper("lr", dtype=var_dtype)
 236     denom = math_ops.sqrt(
 237         math_ops.minimum(
 238             ops.convert_to_tensor(update_global_step),
 239             ops.convert_to_tensor(math_ops.cast(window, dtype=var_dtype))))
 240     moment1_2d = array_ops.expand_dims(update_moment1, -1)
 241 
 242     # m = grad_buffer^T / sqrt(min(t, window))
 243     # m has shape [model dimension, window], where model dimension is the sum
 244     # of the dimensions of the flattened variables.
 245     m = array_ops.transpose(math_ops.divide(update_grad_buffer, denom))
 246 
 247     # sigma, u, _ = SVD(m^Tm + I * svd_eps)
 248     mm = math_ops.matmul(m, m, transpose_a=True)
 249     damping = math_ops.cast(linalg_ops.eye(window), dtype=var_dtype) * svd_eps
 250     sigma, u, _ = linalg_ops.svd(mm + damping)
 251     sigma_sqrt = math_ops.sqrt(sigma)
 252     sigma_sqrt_min = math_ops.reduce_min(sigma_sqrt)
 253 
 254     # sigma_sqrt_inv = 1 / (\sqrt{sigma} + sigma_eps) ^ 3
 255     # We add sigma_eps to alleviate numerical instability.
 256     # Note that (m^Tm)^(-3/2) = u diag(sigma_sqrt_inv) u^T.
 257     sigma_sqrt_inv = math_ops.divide(
 258         math_ops.cast(1.0, dtype=var_dtype),
 259         math_ops.pow(sigma_sqrt + sigma_eps, 3))
 260 
 261     # In full matrix AdaGrad, the update step computes (mm^T)^(-1/2)g, where the
 262     # inversion of a model dimension by model dimension matrix is needed. To
 263     # speed up this computation we calculate the following instead:
 264     # m(m^Tm)^(-3/2)m^T moment1 = m u diag(sigma_sqrt_inv) u^T m^T moment1.
 265     new_step = array_ops.expand_dims(
 266         array_ops.zeros(flat_grad.get_shape(), dtype=var_dtype), -1)
 267     head = math_ops.matmul(
 268         m,
 269         math_ops.matmul(
 270             u,
 271             math_ops.matmul(
 272                 array_ops.diag(sigma_sqrt_inv),
 273                 math_ops.matmul(
 274                     u,
 275                     math_ops.matmul(m, moment1_2d, transpose_a=True),
 276                     transpose_a=True))))
 277 
 278     # When inverting (mm^t)^(1/2), we also add epsilon * I regularization for
 279     # degenerate cases. We expand ((mm^t)^(1/2) + epsilon * I)^(-1) using
 280     # Woodbury's identity.
 281     # For full derivation please see paper at
 282     # https://arxiv.org/pdf/1806.02958.pdf
 283     tail = moment1_2d - math_ops.matmul(
 284         m,
 285         math_ops.matmul(
 286             u,
 287             math_ops.matmul(
 288                 array_ops.diag(
 289                     math_ops.divide(math_ops.cast(1.0, dtype=var_dtype),
 290                                     sigma)),
 291                 math_ops.matmul(
 292                     u,
 293                     math_ops.matmul(m, moment1_2d, transpose_a=True),
 294                     transpose_a=True))))
 295     scaled_tail = math_ops.divide(tail, sigma_sqrt_min)
 296 
 297     update_new_step = control_flow_ops.cond(
 298         sigma_sqrt_min > eps, lambda: math_ops.add(head, scaled_tail),
 299         lambda: math_ops.add(new_step, head))
 300 
 301     # Update each variable.
 302     update_step = []
 303     for var in self._variables:
 304       dim = self.shape_dict[var.name]
 305       start_index = self.index_dict[var.name]
 306       end_index = start_index + dim
 307       var_update_correct_shape = array_ops.reshape(
 308           update_new_step[start_index:end_index], var.get_shape())
 309       var_updated = state_ops.assign_sub(var, lr * var_update_correct_shape)
 310       update_step.append(var_updated)
 311 
 312     return control_flow_ops.group(update_step)
