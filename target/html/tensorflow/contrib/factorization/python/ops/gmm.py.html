  39 def _streaming_sum(scalar_tensor):
  40   """Create a sum metric and update op."""
  41   sum_metric = framework.local_variable(constant_op.constant(0.0))
  42   sum_update = sum_metric.assign_add(scalar_tensor)
  43   return sum_metric, sum_update

  46 class _InitializeClustersHook(session_run_hook.SessionRunHook):
  47   """Initializes clusters or waits for cluster initialization."""
  48 
  49   def __init__(self, init_op, is_initialized_op, is_chief):
  50     self._init_op = init_op
  51     self._is_chief = is_chief
  52     self._is_initialized_op = is_initialized_op
  53 
  54   def after_create_session(self, session, _):
  55     assert self._init_op.graph == ops.get_default_graph()
  56     assert self._is_initialized_op.graph == self._init_op.graph
  57     while True:
  58       try:
  59         if session.run(self._is_initialized_op):
  60           break
  61         elif self._is_chief:
  62           session.run(self._init_op)
  63         else:
  64           time.sleep(1)
  65       except RuntimeError as e:
  66         logging.info(e)

  69 class GMM(estimator.Estimator):
  70   """An estimator for GMM clustering."""
  71   SCORES = 'scores'
  72   LOG_LIKELIHOOD = 'loss'
  73   ASSIGNMENTS = 'assignments'
  74 
  75   def __init__(self,
  76                num_clusters,
  77                model_dir=None,
  78                random_seed=0,
  79                params='wmc',
  80                initial_clusters='random',
  81                covariance_type='full',
  82                config=None):
  83     """Creates a model for running GMM training and inference.
  84 
  85     Args:
  86       num_clusters: number of clusters to train.
  87       model_dir: the directory to save the model results and log files.
  88       random_seed: Python integer. Seed for PRNG used to initialize centers.
  89       params: Controls which parameters are updated in the training process.
  90         Can contain any combination of "w" for weights, "m" for means,
  91         and "c" for covars.
  92       initial_clusters: specifies how to initialize the clusters for training.
  93         See gmm_ops.gmm for the possible values.
  94       covariance_type: one of "full", "diag".
  95       config: See Estimator
  96     """
  97     self._num_clusters = num_clusters
  98     self._params = params
  99     self._training_initial_clusters = initial_clusters
 100     self._covariance_type = covariance_type
 101     self._training_graph = None
 102     self._random_seed = random_seed
 103     super(GMM, self).__init__(
 104         model_fn=self._model_builder(), model_dir=model_dir, config=config)
 105 
 106   def predict_assignments(self, input_fn=None, batch_size=None, outputs=None):
 107     """See BaseEstimator.predict."""
 108     results = self.predict(input_fn=input_fn,
 109                            batch_size=batch_size,
 110                            outputs=outputs)
 111     for result in results:
 112       yield result[GMM.ASSIGNMENTS]
 113 
 114   def score(self, input_fn=None, batch_size=None, steps=None):
 115     """Predict total log-likelihood.
 116 
 117     Args:
 118       input_fn: see predict.
 119       batch_size: see predict.
 120       steps: see predict.
 121 
 122     Returns:
 123       Total log-likelihood.
 124     """
 125     results = self.evaluate(input_fn=input_fn, batch_size=batch_size,
 126                             steps=steps)
 127     return np.log(np.sum(np.exp(results[GMM.SCORES])))
 128 
 129   def weights(self):
 130     """Returns the cluster weights."""
 131     return checkpoint_utils.load_variable(
 132         self.model_dir, gmm_ops.GmmAlgorithm.CLUSTERS_WEIGHT)
 133 
 134   def clusters(self):
 135     """Returns cluster centers."""
 136     clusters = checkpoint_utils.load_variable(
 137         self.model_dir, gmm_ops.GmmAlgorithm.CLUSTERS_VARIABLE)
 138     return np.squeeze(clusters, 1)
 139 
 140   def covariances(self):
 141     """Returns the covariances."""
 142     return checkpoint_utils.load_variable(
 143         self.model_dir, gmm_ops.GmmAlgorithm.CLUSTERS_COVS_VARIABLE)
 144 
 145   def _parse_tensor_or_dict(self, features):
 146     if isinstance(features, dict):
 147       return array_ops.concat([features[k] for k in sorted(features.keys())],
 148                               1)
 149     return features
 150 
 151   def _model_builder(self):
 152     """Creates a model function."""
 153 
 154     def _model_fn(features, labels, mode, config):
 155       """Model function."""
 156       assert labels is None, labels
 157       (loss,
 158        scores,
 159        model_predictions,
 160        training_op,
 161        init_op,
 162        is_initialized) = gmm_ops.gmm(self._parse_tensor_or_dict(features),
 163                                      self._training_initial_clusters,
 164                                      self._num_clusters, self._random_seed,
 165                                      self._covariance_type,
 166                                      self._params)
 167       incr_step = state_ops.assign_add(training_util.get_global_step(), 1)
 168       training_op = with_dependencies([training_op, incr_step], loss)
 169       training_hooks = [_InitializeClustersHook(
 170           init_op, is_initialized, config.is_chief)]
 171       predictions = {
 172           GMM.ASSIGNMENTS: model_predictions[0][0],
 173       }
 174       eval_metric_ops = {
 175           GMM.SCORES: scores,
 176           GMM.LOG_LIKELIHOOD: _streaming_sum(loss),
 177       }
 178       return model_fn_lib.ModelFnOps(mode=mode, predictions=predictions,
 179                                      eval_metric_ops=eval_metric_ops,
 180                                      loss=loss, train_op=training_op,
 181                                      training_hooks=training_hooks)
 182 
 183     return _model_fn
