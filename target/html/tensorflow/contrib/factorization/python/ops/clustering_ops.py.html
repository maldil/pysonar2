  44 # Euclidean distance between vectors U and V is defined as \\(||U - V||_F\\)
  45 # which is the square root of the sum of the absolute squares of the elements
  46 # difference.
  47 SQUARED_EUCLIDEAN_DISTANCE = 'squared_euclidean'
  48 # Cosine distance between vectors U and V is defined as
  49 # \\(1 - (U \dot V) / (||U||_F ||V||_F)\\)
  50 COSINE_DISTANCE = 'cosine'

  53 KMEANS_PLUS_PLUS_INIT = 'kmeans_plus_plus'
  54 KMC2_INIT = 'kmc2'

  60 class KMeans(object):
  61   """Creates the graph for k-means clustering."""
  62 
  63   def __init__(self,
  64                inputs,
  65                num_clusters,
  66                initial_clusters=RANDOM_INIT,
  67                distance_metric=SQUARED_EUCLIDEAN_DISTANCE,
  68                use_mini_batch=False,
  69                mini_batch_steps_per_iteration=1,
  70                random_seed=0,
  71                kmeans_plus_plus_num_retries=2,
  72                kmc2_chain_length=200):
  73     """Creates an object for generating KMeans clustering graph.
  74 
  75     This class implements the following variants of K-means algorithm:
  76 
  77     If use_mini_batch is False, it runs standard full batch K-means. Each step
  78     runs a single iteration of K-Means. This step can be run sharded across
  79     multiple workers by passing a list of sharded inputs to this class. Note
  80     however that a single step needs to process the full input at once.
  81 
  82     If use_mini_batch is True, it runs a generalization of the mini-batch
  83     K-means algorithm. It runs multiple iterations, where each iteration is
  84     composed of mini_batch_steps_per_iteration steps. Two copies of cluster
  85     centers are maintained: one that is updated at the end of each iteration,
  86     and one that is updated every step. The first copy is used to compute
  87     cluster allocations for each step, and for inference, while the second copy
  88     is the one updated each step using the mini-batch update rule. After each
  89     iteration is complete, this second copy is copied back the first copy.
  90 
  91     Note that for use_mini_batch=True, when mini_batch_steps_per_iteration=1,
  92     the algorithm reduces to the standard mini-batch algorithm. Also by setting
  93     mini_batch_steps_per_iteration = num_inputs / batch_size, the algorithm
  94     becomes an asynchronous version of the full-batch algorithm. Note however
  95     that there is no guarantee by this implementation that each input is seen
  96     exactly once per iteration. Also, different updates are applied
  97     asynchronously without locking. So this asynchronous version may not behave
  98     exactly like a full-batch version.
  99 
 100     Args:
 101       inputs: An input tensor or list of input tensors. It is assumed that the
 102         data points have been previously randomly permuted.
 103       num_clusters: An integer tensor specifying the number of clusters. This
 104         argument is ignored if initial_clusters is a tensor or numpy array.
 105       initial_clusters: Specifies the clusters used during initialization. One
 106         of the following:
 107         - a tensor or numpy array with the initial cluster centers.
 108         - a function f(inputs, k) that returns up to k centers from `inputs`.
 109         - "random": Choose centers randomly from `inputs`.
 110         - "kmeans_plus_plus": Use kmeans++ to choose centers from `inputs`.
 111         - "kmc2": Use the fast k-MC2 algorithm to choose centers from `inputs`.
 112         In the last three cases, one batch of `inputs` may not yield
 113         `num_clusters` centers, in which case initialization will require
 114         multiple batches until enough centers are chosen. In the case of
 115         "random" or "kmeans_plus_plus", if the input size is <= `num_clusters`
 116         then the entire batch is chosen to be cluster centers.
 117       distance_metric: Distance metric used for clustering. Supported options:
 118         "squared_euclidean", "cosine".
 119       use_mini_batch: If true, use the mini-batch k-means algorithm. Else assume
 120         full batch.
 121       mini_batch_steps_per_iteration: Number of steps after which the updated
 122         cluster centers are synced back to a master copy.
 123       random_seed: Seed for PRNG used to initialize seeds.
 124       kmeans_plus_plus_num_retries: For each point that is sampled during
 125         kmeans++ initialization, this parameter specifies the number of
 126         additional points to draw from the current distribution before selecting
 127         the best. If a negative value is specified, a heuristic is used to
 128         sample O(log(num_to_sample)) additional points.
 129       kmc2_chain_length: Determines how many candidate points are used by the
 130         k-MC2 algorithm to produce one new cluster centers. If a (mini-)batch
 131         contains less points, one new cluster center is generated from the
 132         (mini-)batch.
 133 
 134     Raises:
 135       ValueError: An invalid argument was passed to initial_clusters or
 136         distance_metric.
 137     """
 138     if isinstance(initial_clusters, str) and initial_clusters not in [
 139         RANDOM_INIT, KMEANS_PLUS_PLUS_INIT, KMC2_INIT
 140     ]:
 141       raise ValueError(
 142           "Unsupported initialization algorithm '%s'" % initial_clusters)
 143     if distance_metric not in [SQUARED_EUCLIDEAN_DISTANCE, COSINE_DISTANCE]:
 144       raise ValueError("Unsupported distance metric '%s'" % distance_metric)
 145     self._inputs = inputs if isinstance(inputs, list) else [inputs]
 146     self._num_clusters = num_clusters
 147     self._initial_clusters = initial_clusters
 148     self._distance_metric = distance_metric
 149     self._use_mini_batch = use_mini_batch
 150     self._mini_batch_steps_per_iteration = int(mini_batch_steps_per_iteration)
 151     self._random_seed = random_seed
 152     self._kmeans_plus_plus_num_retries = kmeans_plus_plus_num_retries
 153     self._kmc2_chain_length = kmc2_chain_length
 154 
 155   @classmethod
 156   def _distance_graph(cls, inputs, clusters, distance_metric):
 157     """Computes distance between each input and each cluster center.
 158 
 159     Args:
 160       inputs: list of input Tensors.
 161       clusters: cluster Tensor.
 162       distance_metric: distance metric used for clustering
 163 
 164     Returns:
 165       list of Tensors, where each element corresponds to each element in inputs.
 166       The value is the distance of each row to all the cluster centers.
 167       Currently only Euclidean distance and cosine distance are supported.
 168     """
 169     assert isinstance(inputs, list)
 170     if distance_metric == SQUARED_EUCLIDEAN_DISTANCE:
 171       return cls._compute_euclidean_distance(inputs, clusters)
 172     elif distance_metric == COSINE_DISTANCE:
 173       return cls._compute_cosine_distance(
 174           inputs, clusters, inputs_normalized=True)
 175     else:
 176       assert False, str(distance_metric)
 177 
 178   @classmethod
 179   def _compute_euclidean_distance(cls, inputs, clusters):
 180     """Computes Euclidean distance between each input and each cluster center.
 181 
 182     Args:
 183       inputs: list of input Tensors.
 184       clusters: cluster Tensor.
 185 
 186     Returns:
 187       list of Tensors, where each element corresponds to each element in inputs.
 188       The value is the distance of each row to all the cluster centers.
 189     """
 190     output = []
 191     for inp in inputs:
 192       with ops.colocate_with(inp, ignore_existing=True):
 193         # Computes Euclidean distance. Note the first and third terms are
 194         # broadcast additions.
 195         squared_distance = (
 196             math_ops.reduce_sum(math_ops.square(inp), 1, keepdims=True) -
 197             2 * math_ops.matmul(inp, clusters, transpose_b=True) +
 198             array_ops.transpose(
 199                 math_ops.reduce_sum(
 200                     math_ops.square(clusters), 1, keepdims=True)))
 201         output.append(squared_distance)
 202 
 203     return output
 204 
 205   @classmethod
 206   def _compute_cosine_distance(cls, inputs, clusters, inputs_normalized=True):
 207     """Computes cosine distance between each input and each cluster center.
 208 
 209     Args:
 210       inputs: list of input Tensor.
 211       clusters: cluster Tensor
 212       inputs_normalized: if True, it assumes that inp and clusters are
 213       normalized and computes the dot product which is equivalent to the cosine
 214       distance. Else it L2 normalizes the inputs first.
 215 
 216     Returns:
 217       list of Tensors, where each element corresponds to each element in inp.
 218       The value is the distance of each row to all the cluster centers.
 219     """
 220     output = []
 221     if not inputs_normalized:
 222       with ops.colocate_with(clusters, ignore_existing=True):
 223         clusters = nn_impl.l2_normalize(clusters, dim=1)
 224     for inp in inputs:
 225       with ops.colocate_with(inp, ignore_existing=True):
 226         if not inputs_normalized:
 227           inp = nn_impl.l2_normalize(inp, dim=1)
 228         output.append(1 - math_ops.matmul(inp, clusters, transpose_b=True))
 229     return output

 231   def _infer_graph(self, inputs, clusters):
 232     """Maps input to closest cluster and the score.
 233 
 234     Args:
 235       inputs: list of input Tensors.
 236       clusters: Tensor of cluster centers.
 237 
 238     Returns:
 239       List of tuple, where each value in tuple corresponds to a value in inp.
 240       The tuple has following three elements:
 241       all_scores: distance of each input to each cluster center.
 242       score: distance of each input to closest cluster center.
 243       cluster_idx: index of cluster center closest to the corresponding input.
 244     """
 245     assert isinstance(inputs, list)
 246     # Pairwise distances are used only by transform(). In all other cases, this
 247     # sub-graph is not evaluated.
 248     scores = self._distance_graph(inputs, clusters, self._distance_metric)
 249     output = []
 250     if (self._distance_metric == COSINE_DISTANCE and
 251         not self._clusters_l2_normalized()):
 252       # The cosine distance between normalized vectors x and y is the same as
 253       # 2 * squared_euclidean_distance. We are using this fact and reusing the
 254       # nearest_neighbors op.
 255       # TODO(ands): Support COSINE distance in nearest_neighbors and remove
 256       # this.
 257       with ops.colocate_with(clusters, ignore_existing=True):
 258         clusters = nn_impl.l2_normalize(clusters, dim=1)
 259     for inp, score in zip(inputs, scores):
 260       with ops.colocate_with(inp, ignore_existing=True):
 261         (indices, distances) = gen_clustering_ops.nearest_neighbors(
 262             inp, clusters, 1)
 263         if self._distance_metric == COSINE_DISTANCE:
 264           distances *= 0.5
 265         output.append((score, array_ops.squeeze(distances, [-1]),
 266                        array_ops.squeeze(indices, [-1])))
 267     return zip(*output)
 268 
 269   def _clusters_l2_normalized(self):
 270     """Returns True if clusters centers are kept normalized."""
 271     return (self._distance_metric == COSINE_DISTANCE and
 272             (not self._use_mini_batch or
 273              self._mini_batch_steps_per_iteration > 1))

 275   def _create_variables(self, num_clusters):
 276     """Creates variables.
 277 
 278     Args:
 279       num_clusters: an integer Tensor providing the number of clusters.
 280 
 281     Returns:
 282       Tuple with following elements:
 283       - cluster_centers: a Tensor for storing cluster centers
 284       - cluster_centers_initialized: bool Variable indicating whether clusters
 285             are initialized.
 286       - cluster_counts: a Tensor for storing counts of points assigned to this
 287             cluster. This is used by mini-batch training.
 288       - cluster_centers_updated: Tensor representing copy of cluster centers
 289             that are updated every step.
 290       - update_in_steps: numbers of steps left before we sync
 291             cluster_centers_updated back to cluster_centers.
 292     """
 293     init_value = array_ops.constant([], dtype=dtypes.float32)
 294     cluster_centers = variable_scope.variable(
 295         init_value, name=CLUSTERS_VAR_NAME, validate_shape=False)
 296     cluster_centers_initialized = variable_scope.variable(
 297         False, dtype=dtypes.bool, name='initialized')
 298 
 299     if self._use_mini_batch and self._mini_batch_steps_per_iteration > 1:
 300       # Copy of cluster centers actively updated each step according to
 301       # mini-batch update rule.
 302       cluster_centers_updated = variable_scope.variable(
 303           init_value, name='clusters_updated', validate_shape=False)
 304       # How many steps till we copy the updated clusters to cluster_centers.
 305       update_in_steps = variable_scope.variable(
 306           self._mini_batch_steps_per_iteration,
 307           dtype=dtypes.int64,
 308           name='update_in_steps')
 309       # Count of points assigned to cluster_centers_updated.
 310       cluster_counts = variable_scope.variable(
 311           array_ops.zeros([num_clusters], dtype=dtypes.int64))
 312     else:
 313       cluster_centers_updated = cluster_centers
 314       update_in_steps = None
 315       cluster_counts = (
 316           variable_scope.variable(
 317               array_ops.ones([num_clusters], dtype=dtypes.int64))
 318           if self._use_mini_batch else None)
 319     return (cluster_centers, cluster_centers_initialized, cluster_counts,
 320             cluster_centers_updated, update_in_steps)
 321 
 322   @classmethod
 323   def _l2_normalize_data(cls, inputs):
 324     """Normalized the input data."""
 325     output = []
 326     for inp in inputs:
 327       with ops.colocate_with(inp, ignore_existing=True):
 328         output.append(nn_impl.l2_normalize(inp, dim=1))
 329     return output

 331   def training_graph(self):
 332     """Generate a training graph for kmeans algorithm.
 333 
 334     This returns, among other things, an op that chooses initial centers
 335     (init_op), a boolean variable that is set to True when the initial centers
 336     are chosen (cluster_centers_initialized), and an op to perform either an
 337     entire Lloyd iteration or a mini-batch of a Lloyd iteration (training_op).
 338     The caller should use these components as follows. A single worker should
 339     execute init_op multiple times until cluster_centers_initialized becomes
 340     True. Then multiple workers may execute training_op any number of times.
 341 
 342     Returns:
 343       A tuple consisting of:
 344       all_scores: A matrix (or list of matrices) of dimensions (num_input,
 345         num_clusters) where the value is the distance of an input vector and a
 346         cluster center.
 347       cluster_idx: A vector (or list of vectors). Each element in the vector
 348         corresponds to an input row in 'inp' and specifies the cluster id
 349         corresponding to the input.
 350       scores: Similar to cluster_idx but specifies the distance to the
 351         assigned cluster instead.
 352       cluster_centers_initialized: scalar indicating whether clusters have been
 353         initialized.
 354       init_op: an op to initialize the clusters.
 355       training_op: an op that runs an iteration of training.
 356     """
 357     # Implementation of kmeans.
 358     if (isinstance(self._initial_clusters, str) or
 359         callable(self._initial_clusters)):
 360       initial_clusters = self._initial_clusters
 361       num_clusters = ops.convert_to_tensor(self._num_clusters)
 362     else:
 363       initial_clusters = ops.convert_to_tensor(self._initial_clusters)
 364       num_clusters = array_ops.shape(initial_clusters)[0]
 365 
 366     inputs = self._inputs
 367     (cluster_centers_var, cluster_centers_initialized, total_counts,
 368      cluster_centers_updated,
 369      update_in_steps) = self._create_variables(num_clusters)
 370     init_op = _InitializeClustersOpFactory(
 371         self._inputs, num_clusters, initial_clusters, self._distance_metric,
 372         self._random_seed, self._kmeans_plus_plus_num_retries,
 373         self._kmc2_chain_length, cluster_centers_var, cluster_centers_updated,
 374         cluster_centers_initialized).op()
 375     cluster_centers = cluster_centers_var
 376 
 377     if self._distance_metric == COSINE_DISTANCE:
 378       inputs = self._l2_normalize_data(inputs)
 379       if not self._clusters_l2_normalized():
 380         cluster_centers = nn_impl.l2_normalize(cluster_centers, dim=1)
 381 
 382     all_scores, scores, cluster_idx = self._infer_graph(inputs, cluster_centers)
 383     if self._use_mini_batch:
 384       sync_updates_op = self._mini_batch_sync_updates_op(
 385           update_in_steps, cluster_centers_var, cluster_centers_updated,
 386           total_counts)
 387       assert sync_updates_op is not None
 388       with ops.control_dependencies([sync_updates_op]):
 389         training_op = self._mini_batch_training_op(
 390             inputs, cluster_idx, cluster_centers_updated, total_counts)
 391     else:
 392       assert cluster_centers == cluster_centers_var
 393       training_op = self._full_batch_training_op(
 394           inputs, num_clusters, cluster_idx, cluster_centers_var)
 395 
 396     return (all_scores, cluster_idx, scores, cluster_centers_initialized,
 397             init_op, training_op)
 398 
 399   def _mini_batch_sync_updates_op(self, update_in_steps, cluster_centers_var,
 400                                   cluster_centers_updated, total_counts):
 401     if self._use_mini_batch and self._mini_batch_steps_per_iteration > 1:
 402       assert update_in_steps is not None
 403       with ops.colocate_with(update_in_steps, ignore_existing=True):
 404 
 405         def _f():
 406           # Note that there is a race condition here, so we do a best effort
 407           # updates here. We reset update_in_steps first so that other workers
 408           # don't duplicate the updates. Also we update cluster_center_vars
 409           # before resetting total_counts to avoid large updates to
 410           # cluster_centers_updated based on partially updated
 411           # cluster_center_vars.
 412           with ops.control_dependencies([
 413               state_ops.assign(update_in_steps,
 414                                self._mini_batch_steps_per_iteration - 1)
 415           ]):
 416             with ops.colocate_with(
 417                 cluster_centers_updated, ignore_existing=True):
 418               if self._distance_metric == COSINE_DISTANCE:
 419                 cluster_centers = nn_impl.l2_normalize(
 420                     cluster_centers_updated, dim=1)
 421               else:
 422                 cluster_centers = cluster_centers_updated
 423             with ops.colocate_with(cluster_centers_var, ignore_existing=True):
 424               with ops.control_dependencies(
 425                   [state_ops.assign(cluster_centers_var, cluster_centers)]):
 426                 with ops.colocate_with(None, ignore_existing=True):
 427                   with ops.control_dependencies([
 428                       state_ops.assign(total_counts,
 429                                        array_ops.zeros_like(total_counts))
 430                   ]):
 431                     return array_ops.identity(update_in_steps)
 432 
 433         return control_flow_ops.cond(
 434             update_in_steps <= 0, _f,
 435             lambda: state_ops.assign_sub(update_in_steps, 1))
 436     else:
 437       return control_flow_ops.no_op()

 439   def _mini_batch_training_op(self, inputs, cluster_idx_list, cluster_centers,
 440                               total_counts):
 441     """Creates an op for training for mini batch case.
 442 
 443     Args:
 444       inputs: list of input Tensors.
 445       cluster_idx_list: A vector (or list of vectors). Each element in the
 446         vector corresponds to an input row in 'inp' and specifies the cluster id
 447         corresponding to the input.
 448       cluster_centers: Tensor Ref of cluster centers.
 449       total_counts: Tensor Ref of cluster counts.
 450 
 451     Returns:
 452       An op for doing an update of mini-batch k-means.
 453     """
 454     update_ops = []
 455     for inp, cluster_idx in zip(inputs, cluster_idx_list):
 456       with ops.colocate_with(inp, ignore_existing=True):
 457         assert total_counts is not None
 458         cluster_idx = array_ops.reshape(cluster_idx, [-1])
 459         # Dedupe the unique ids of cluster_centers being updated so that updates
 460         # can be locally aggregated.
 461         unique_ids, unique_idx = array_ops.unique(cluster_idx)
 462         num_unique_cluster_idx = array_ops.size(unique_ids)
 463         # Fetch the old values of counts and cluster_centers.
 464         with ops.colocate_with(total_counts, ignore_existing=True):
 465           old_counts = array_ops.gather(total_counts, unique_ids)
 466         # TODO(agarwal): This colocation seems to run into problems. Fix it.
 467         with ops.colocate_with(cluster_centers, ignore_existing=True):
 468           old_cluster_centers = array_ops.gather(cluster_centers, unique_ids)
 469         # Locally aggregate the increment to counts.
 470         count_updates = math_ops.unsorted_segment_sum(
 471             array_ops.ones_like(unique_idx, dtype=total_counts.dtype),
 472             unique_idx, num_unique_cluster_idx)
 473         # Locally compute the sum of inputs mapped to each id.
 474         # For a cluster with old cluster value x, old count n, and with data
 475         # d_1,...d_k newly assigned to it, we recompute the new value as
 476         # \\(x += (sum_i(d_i) - k * x) / (n + k)\\).
 477         # Compute \\(sum_i(d_i)\\), see comment above.
 478         cluster_center_updates = math_ops.unsorted_segment_sum(
 479             inp, unique_idx, num_unique_cluster_idx)
 480         # Shape to enable broadcasting count_updates and learning_rate to inp.
 481         # It extends the shape with 1's to match the rank of inp.
 482         broadcast_shape = array_ops.concat([
 483             array_ops.reshape(num_unique_cluster_idx, [1]),
 484             array_ops.ones(
 485                 array_ops.reshape(array_ops.rank(inp) - 1, [1]),
 486                 dtype=dtypes.int32)
 487         ], 0)
 488         # Subtract k * x, see comment above.
 489         cluster_center_updates -= math_ops.cast(
 490             array_ops.reshape(count_updates, broadcast_shape),
 491             inp.dtype) * old_cluster_centers
 492         learning_rate = math_ops.reciprocal(
 493             math_ops.cast(old_counts + count_updates, inp.dtype))
 494         learning_rate = array_ops.reshape(learning_rate, broadcast_shape)
 495         # scale by 1 / (n + k), see comment above.
 496         cluster_center_updates *= learning_rate
 497         # Apply the updates.
 498       update_counts = state_ops.scatter_add(total_counts, unique_ids,
 499                                             count_updates)
 500       update_cluster_centers = state_ops.scatter_add(
 501           cluster_centers, unique_ids, cluster_center_updates)
 502       update_ops.extend([update_counts, update_cluster_centers])
 503     return control_flow_ops.group(*update_ops)

 505   def _full_batch_training_op(self, inputs, num_clusters, cluster_idx_list,
 506                               cluster_centers):
 507     """Creates an op for training for full batch case.
 508 
 509     Args:
 510       inputs: list of input Tensors.
 511       num_clusters: an integer Tensor providing the number of clusters.
 512       cluster_idx_list: A vector (or list of vectors). Each element in the
 513         vector corresponds to an input row in 'inp' and specifies the cluster id
 514         corresponding to the input.
 515       cluster_centers: Tensor Ref of cluster centers.
 516 
 517     Returns:
 518       An op for doing an update of mini-batch k-means.
 519     """
 520     cluster_sums = []
 521     cluster_counts = []
 522     epsilon = constant_op.constant(1e-6, dtype=inputs[0].dtype)
 523     for inp, cluster_idx in zip(inputs, cluster_idx_list):
 524       with ops.colocate_with(inp, ignore_existing=True):
 525         cluster_sums.append(
 526             math_ops.unsorted_segment_sum(inp, cluster_idx, num_clusters))
 527         cluster_counts.append(
 528             math_ops.unsorted_segment_sum(
 529                 array_ops.reshape(
 530                     array_ops.ones(
 531                         array_ops.reshape(array_ops.shape(inp)[0], [-1])),
 532                     [-1, 1]), cluster_idx, num_clusters))
 533     with ops.colocate_with(cluster_centers, ignore_existing=True):
 534       new_clusters_centers = math_ops.add_n(cluster_sums) / (
 535           math_ops.cast(math_ops.add_n(cluster_counts), cluster_sums[0].dtype) +
 536           epsilon)
 537       if self._clusters_l2_normalized():
 538         new_clusters_centers = nn_impl.l2_normalize(new_clusters_centers, dim=1)
 539     return state_ops.assign(cluster_centers, new_clusters_centers)
 540

 542 class _InitializeClustersOpFactory(object):
 543   """Internal class to create the op to initialize the clusters.
 544 
 545     The op performs this algorithm (see constructor args):
 546 
 547     num_remaining = num_clusters - length(cluster_centers)
 548     if num_remaining == 0:
 549       assert that cluster_centers_initialized is true
 550     else:
 551       assert that num_remaining > 0
 552       new_centers = choose up to num_remaining initial centers
 553       l2-normalize new_centers if using cosine distance
 554       all_centers = concat(cluster_centers, new_centers)
 555       cluster_centers := all_centers
 556       if there is a cluster_centers_updated variable:
 557         cluster_centers_updated := cluster_centers
 558       num_now_remaining = num_clusters - length(cluster_centers)
 559       if num_now_remaining == 0:
 560         cluster_centers_initialized := true
 561   """
 562 
 563   # TODO(ccolby): Refactor this class so that kmc2 isn't so much a special case.
 564 
 565   def __init__(self, inputs, num_clusters, initial_clusters, distance_metric,
 566                random_seed, kmeans_plus_plus_num_retries, kmc2_chain_length,
 567                cluster_centers, cluster_centers_updated,
 568                cluster_centers_initialized):
 569     """Creates an op factory.
 570 
 571     Args:
 572       inputs: See KMeans constructor.
 573       num_clusters: An integer Tensor providing the number of clusters.
 574       initial_clusters: See KMeans constructor.
 575       distance_metric: See KMeans constructor.
 576       random_seed: See KMeans constructor.
 577       kmeans_plus_plus_num_retries: See KMeans constructor.
 578       kmc2_chain_length: See KMeans constructor.
 579       cluster_centers: The TF variable holding the initial centers. It may
 580           already contain some centers when the op is executed.
 581       cluster_centers_updated: A second TF variable to hold a copy of the
 582           initial centers, used for full-batch mode. In mini-batch mode,
 583           cluster_centers_updated is the same variable as cluster_centers.
 584       cluster_centers_initialized: A boolean TF variable that will be set
 585           to true when all the initial centers have been chosen.
 586     """
 587     # All of these instance variables are constants.
 588     self._inputs = inputs
 589     self._num_clusters = num_clusters
 590     self._initial_clusters = initial_clusters
 591     self._distance_metric = distance_metric
 592     self._random_seed = random_seed
 593     self._kmeans_plus_plus_num_retries = kmeans_plus_plus_num_retries
 594     self._kmc2_chain_length = kmc2_chain_length
 595     self._cluster_centers = cluster_centers
 596     self._cluster_centers_updated = cluster_centers_updated
 597     self._cluster_centers_initialized = cluster_centers_initialized
 598 
 599     self._num_selected = array_ops.shape(self._cluster_centers)[0]
 600     self._num_remaining = self._num_clusters - self._num_selected
 601     self._num_data = math_ops.add_n(
 602         [array_ops.shape(i)[0] for i in self._inputs])
 603 
 604   def _random(self):
 605     indices = random_ops.random_uniform(
 606         array_ops.reshape(self._num_remaining, [-1]),
 607         minval=0,
 608         maxval=math_ops.cast(self._num_data, dtypes.int64),
 609         seed=self._random_seed,
 610         dtype=dtypes.int64)
 611     return embedding_lookup(self._inputs, indices, partition_strategy='div')
 612 
 613   def _kmeans_plus_plus(self):
 614     # Points from only the first shard are used for initializing centers.
 615     # TODO(ands): Use all points.
 616     inp = self._inputs[0]
 617     if self._distance_metric == COSINE_DISTANCE:
 618       inp = nn_impl.l2_normalize(inp, dim=1)
 619     return gen_clustering_ops.kmeans_plus_plus_initialization(
 620         inp,
 621         math_ops.to_int64(self._num_remaining), self._random_seed,
 622         self._kmeans_plus_plus_num_retries)
 623 
 624   def _kmc2_multiple_centers(self):
 625     """Adds new initial cluster centers using the k-MC2 algorithm.
 626 
 627     In each call to the op, the provided batch is split into subsets based on
 628     the specified `kmc2_chain_length`. On each subset, a single Markov chain of
 629     the k-MC2 algorithm is used to add *one* new center cluster center. If there
 630     are less than `kmc2_chain_length` points in the subset, a single center is
 631     added using one Markov chain on the full input. It is assumed that the
 632     provided batch has previously been randomly permuted. Otherwise, k-MC2 may
 633     return suboptimal centers.
 634 
 635     Returns:
 636       An op that adds new cluster centers.
 637     """
 638     # The op only operates on the first shard of data.
 639     first_shard = self._inputs[0]
 640     # Number of points in the input that can be used.
 641     batch_size = array_ops.shape(first_shard)[0]
 642     # Maximum number of subsets such that the size of each subset is at least
 643     # `kmc2_chain_length`. Final subsets may be larger.
 644     max_to_sample = math_ops.cast(
 645         batch_size / self._kmc2_chain_length, dtype=dtypes.int32)
 646     # We sample at least one new center and at most all remaining centers.
 647     num_to_sample = math_ops.maximum(
 648         math_ops.minimum(self._num_remaining, max_to_sample), 1)
 649 
 650     def _cond(i, _):
 651       """Stopping condition for the while loop."""
 652       return math_ops.less(i, num_to_sample)
 653 
 654     def _body(i, _):
 655       """Body that adds a single new center based on a subset."""
 656 
 657       def _sample_random():
 658         """Returns a random point as a cluster center."""
 659         # By assumption the batch is reshuffled and _sample_random is always
 660         # called for i=0. Hence, we simply return the first point.
 661         new_center = array_ops.reshape(first_shard[0], [1, -1])
 662         if self._distance_metric == COSINE_DISTANCE:
 663           new_center = nn_impl.l2_normalize(new_center, dim=1)
 664         return new_center
 665 
 666       def _sample_kmc2_chain():
 667         """Returns previous centers as well as a new center sampled using k-MC2.
 668         """
 669         # Extract the subset from the underlying batch.
 670         start = i * self._kmc2_chain_length
 671         end = start + self._kmc2_chain_length
 672         subset = first_shard[start:end]
 673         # Compute the distances from points in the subset to previous centers.
 674         _, distances = gen_clustering_ops.nearest_neighbors(
 675             subset, self._cluster_centers, 1)
 676         # Sample index of new center using k-MC2 Markov chain.
 677         new_center_index = gen_clustering_ops.kmc2_chain_initialization(
 678             array_ops.squeeze(distances), self._random_seed)
 679         # Extract actual new center.
 680         newly_sampled_center = array_ops.reshape(subset[new_center_index],
 681                                                  [1, -1])
 682         # Return concatenation with previously sampled centers.
 683         if self._distance_metric == COSINE_DISTANCE:
 684           newly_sampled_center = nn_impl.l2_normalize(
 685               newly_sampled_center, dim=1)
 686         return array_ops.concat([self._cluster_centers, newly_sampled_center],
 687                                 0)
 688 
 689       # Obtain a random point if there are no previously sampled centers.
 690       # Otherwise, construct a k-MC2 Markov chain.
 691       new_centers = control_flow_ops.cond(
 692           math_ops.equal(self._num_selected, 0), _sample_random,
 693           _sample_kmc2_chain)
 694       # Assign new cluster centers to underlying variable.
 695       assigned_centers = state_ops.assign(
 696           self._cluster_centers, new_centers, validate_shape=False)
 697       if self._cluster_centers_updated is not self._cluster_centers:
 698         assigned_centers = state_ops.assign(
 699             self._cluster_centers_updated,
 700             assigned_centers,
 701             validate_shape=False)
 702       return i + 1, self._num_clusters - array_ops.shape(assigned_centers)[0]
 703 
 704     # Add num_to_sample new data points.
 705     _, num_remaining = control_flow_ops.while_loop(_cond, _body, [0, 0])
 706     return num_remaining
 707 
 708   def _greedy_batch_sampler(self, sampler):
 709     # If the input dataset size is smaller than the number of centers
 710     # remaining, choose the entire input dataset as centers. This can happen
 711     # with mini-batch. Otherwise, sample the batch according to the provided
 712     # sampler.
 713     return control_flow_ops.cond(self._num_data <= self._num_remaining,
 714                                  lambda: array_ops.concat(self._inputs, 0),
 715                                  sampler)
 716 
 717   def _single_batch_sampler(self, sampler):
 718     # Enforce that there are at least as many data points as centers
 719     # remaining. This gives the provided sampler the chance to select all
 720     # remaining centers from a single batch.
 721     with ops.control_dependencies(
 722         [check_ops.assert_greater_equal(self._num_data, self._num_remaining)]):
 723       return sampler()
 724 
 725   def _choose_initial_centers(self):
 726     if isinstance(self._initial_clusters, str):
 727       if self._initial_clusters == RANDOM_INIT:
 728         return self._greedy_batch_sampler(self._random)
 729       else:  # self._initial_clusters == KMEANS_PLUS_PLUS_INIT
 730         return self._single_batch_sampler(self._kmeans_plus_plus)
 731     elif callable(self._initial_clusters):
 732       return self._initial_clusters(self._inputs, self._num_remaining)
 733     else:
 734       with ops.control_dependencies([
 735           check_ops.assert_equal(self._num_remaining,
 736                                  array_ops.shape(self._initial_clusters)[0])
 737       ]):
 738         return self._initial_clusters
 739 
 740   def _add_new_centers(self):
 741     """Adds some centers and returns the number of centers remaining."""
 742     new_centers = self._choose_initial_centers()
 743     if self._distance_metric == COSINE_DISTANCE:
 744       new_centers = nn_impl.l2_normalize(new_centers, dim=1)
 745     # If cluster_centers is empty, it doesn't have the right shape for concat.
 746     all_centers = control_flow_ops.cond(
 747         math_ops.equal(self._num_selected, 0), lambda: new_centers,
 748         lambda: array_ops.concat([self._cluster_centers, new_centers], 0))
 749     # TODO(ccolby): De-dupe all_centers?
 750     a = state_ops.assign(
 751         self._cluster_centers, all_centers, validate_shape=False)
 752     if self._cluster_centers_updated is not self._cluster_centers:
 753       a = state_ops.assign(
 754           self._cluster_centers_updated, a, validate_shape=False)
 755     return self._num_clusters - array_ops.shape(a)[0]
 756 
 757   def _initialize(self):
 758     with ops.control_dependencies([
 759         check_ops.assert_positive(self._num_remaining),
 760     ]):
 761       if self._initial_clusters == KMC2_INIT:
 762         num_now_remaining = self._kmc2_multiple_centers()
 763       else:
 764         num_now_remaining = self._add_new_centers()
 765       return control_flow_ops.cond(
 766           math_ops.equal(num_now_remaining, 0),
 767           lambda: state_ops.assign(self._cluster_centers_initialized, True),
 768           control_flow_ops.no_op)
 769 
 770   def op(self):
 771     """Returns the cluster initializer op."""
 772     return control_flow_ops.cond(
 773         math_ops.equal(self._num_remaining, 0),
 774         lambda: check_ops.assert_equal(self._cluster_centers_initialized, True),
 775         self._initialize)
