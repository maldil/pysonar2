  43 class _LossRelativeChangeHook(session_run_hook.SessionRunHook):
  44   """Stops when the change in loss goes below a tolerance."""
  45 
  46   def __init__(self, loss_tensor, tolerance):
  47     """Creates a _LossRelativeChangeHook.
  48 
  49     Args:
  50       loss_tensor: A scalar tensor of the loss value.
  51       tolerance: A relative tolerance of loss change between iterations.
  52     """
  53     self._loss_tensor = loss_tensor
  54     self._tolerance = tolerance
  55     self._prev_loss = None
  56 
  57   def before_run(self, run_context):
  58     del run_context  # unused
  59     return session_run_hook.SessionRunArgs(self._loss_tensor)
  60 
  61   def after_run(self, run_context, run_values):
  62     loss = run_values.results
  63     assert loss is not None
  64     if self._prev_loss:
  65       relative_change = (
  66           abs(loss - self._prev_loss) / (1 + abs(self._prev_loss)))
  67       if relative_change < self._tolerance:
  68         run_context.request_stop()
  69     self._prev_loss = loss

  72 class _InitializeClustersHook(session_run_hook.SessionRunHook):
  73   """Initializes the cluster centers.
  74 
  75   The chief repeatedly invokes an initialization op until all cluster centers
  76   are initialized. The workers wait for the initialization phase to complete.
  77   """
  78 
  79   def __init__(self, init_op, is_initialized_var, is_chief):
  80     """Creates an _InitializeClustersHook.
  81 
  82     Args:
  83       init_op: An op that, when run, will choose some initial cluster centers.
  84           This op may need to be run multiple times to choose all the centers.
  85       is_initialized_var: A boolean variable reporting whether all initial
  86           centers have been chosen.
  87       is_chief: A boolean specifying whether this task is the chief.
  88     """
  89     self._init_op = init_op
  90     self._is_initialized_var = is_initialized_var
  91     self._is_chief = is_chief
  92 
  93   def after_create_session(self, session, coord):
  94     del coord  # unused
  95     assert self._init_op.graph is ops.get_default_graph()
  96     assert self._is_initialized_var.graph is self._init_op.graph
  97     while True:
  98       try:
  99         if session.run(self._is_initialized_var):
 100           break
 101         elif self._is_chief:
 102           session.run(self._init_op)
 103         else:
 104           time.sleep(1)
 105       except RuntimeError as e:
 106         logging.info(e)

 109 def _parse_features_if_necessary(features, feature_columns):
 110   """Helper function to convert the input points into a usable format.
 111 
 112   Args:
 113     features: The input features.
 114     feature_columns: An optionable iterable containing all the feature columns
 115       used by the model. All items in the set should be feature column instances
 116       that can be passed to `tf.feature_column.input_layer`. If this is None,
 117       all features will be used.
 118 
 119   Returns:
 120     If `features` is a dict of `k` features (optionally filtered by
 121     `feature_columns`), each of which is a vector of `n` scalars, the return
 122     value is a Tensor of shape `(n, k)` representing `n` input points, where the
 123     items in the `k` dimension are sorted lexicographically by `features` key.
 124     If `features` is not a dict, it is returned unmodified.
 125   """
 126   if not isinstance(features, dict):
 127     return features
 128 
 129   if feature_columns:
 130     return fc.input_layer(features, feature_columns)
 131 
 132   keys = sorted(features.keys())
 133   with ops.colocate_with(features[keys[0]]):
 134     return array_ops.concat([features[k] for k in keys], axis=1)

 137 class _ModelFn(object):
 138   """Model function for the estimator."""
 139 
 140   def __init__(self, num_clusters, initial_clusters, distance_metric,
 141                random_seed, use_mini_batch, mini_batch_steps_per_iteration,
 142                kmeans_plus_plus_num_retries, relative_tolerance,
 143                feature_columns):
 144     self._num_clusters = num_clusters
 145     self._initial_clusters = initial_clusters
 146     self._distance_metric = distance_metric
 147     self._random_seed = random_seed
 148     self._use_mini_batch = use_mini_batch
 149     self._mini_batch_steps_per_iteration = mini_batch_steps_per_iteration
 150     self._kmeans_plus_plus_num_retries = kmeans_plus_plus_num_retries
 151     self._relative_tolerance = relative_tolerance
 152     self._feature_columns = feature_columns
 153 
 154   def model_fn(self, features, mode, config):
 155     """Model function for the estimator.
 156 
 157     Note that this does not take a `labels` arg. This works, but `input_fn` must
 158     return either `features` or, equivalently, `(features, None)`.
 159 
 160     Args:
 161       features: The input points. See `tf.estimator.Estimator`.
 162       mode: See `tf.estimator.Estimator`.
 163       config: See `tf.estimator.Estimator`.
 164 
 165     Returns:
 166       A `tf.estimator.EstimatorSpec` (see `tf.estimator.Estimator`) specifying
 167       this behavior:
 168         * `train_op`: Execute one mini-batch or full-batch run of Lloyd's
 169              algorithm.
 170         * `loss`: The sum of the squared distances from each input point to its
 171              closest center.
 172         * `eval_metric_ops`: Maps `SCORE` to `loss`.
 173         * `predictions`: Maps `ALL_DISTANCES` to the distance from each input
 174              point to each cluster center; maps `CLUSTER_INDEX` to the index of
 175              the closest cluster center for each input point.
 176     """
 177     # input_points is a single Tensor. Therefore, the sharding functionality
 178     # in clustering_ops is unused, and some of the values below are lists of a
 179     # single item.
 180     input_points = _parse_features_if_necessary(features, self._feature_columns)
 181 
 182     # Let N = the number of input_points.
 183     # all_distances: A list of one matrix of shape (N, num_clusters). Each value
 184     #   is the distance from an input point to a cluster center.
 185     # model_predictions: A list of one vector of shape (N). Each value is the
 186     #   cluster id of an input point.
 187     # losses: Similar to cluster_idx but provides the distance to the cluster
 188     #   center.
 189     # is_initialized: scalar indicating whether the initial cluster centers
 190     #   have been chosen; see init_op.
 191     # init_op: an op to choose the initial cluster centers. A single worker
 192     #   repeatedly executes init_op until is_initialized becomes True.
 193     # training_op: an op that runs an iteration of training, either an entire
 194     #   Lloyd iteration or a mini-batch of a Lloyd iteration. Multiple workers
 195     #   may execute this op, but only after is_initialized becomes True.
 196     (all_distances, model_predictions, losses, is_initialized, init_op,
 197      training_op) = clustering_ops.KMeans(
 198          inputs=input_points,
 199          num_clusters=self._num_clusters,
 200          initial_clusters=self._initial_clusters,
 201          distance_metric=self._distance_metric,
 202          use_mini_batch=self._use_mini_batch,
 203          mini_batch_steps_per_iteration=self._mini_batch_steps_per_iteration,
 204          random_seed=self._random_seed,
 205          kmeans_plus_plus_num_retries=self._kmeans_plus_plus_num_retries
 206      ).training_graph()
 207 
 208     loss = math_ops.reduce_sum(losses)
 209     summary.scalar('loss/raw', loss)
 210 
 211     incr_step = state_ops.assign_add(training_util.get_global_step(), 1)
 212     training_op = control_flow_ops.with_dependencies([training_op, incr_step],
 213                                                      loss)
 214 
 215     training_hooks = [
 216         _InitializeClustersHook(init_op, is_initialized, config.is_chief)
 217     ]
 218     if self._relative_tolerance is not None:
 219       training_hooks.append(
 220           _LossRelativeChangeHook(loss, self._relative_tolerance))
 221 
 222     export_outputs = {
 223         KMeansClustering.ALL_DISTANCES:
 224             export_output.PredictOutput(all_distances[0]),
 225         KMeansClustering.CLUSTER_INDEX:
 226             export_output.PredictOutput(model_predictions[0]),
 227         signature_constants.DEFAULT_SERVING_SIGNATURE_DEF_KEY:
 228             export_output.PredictOutput(model_predictions[0])
 229     }
 230 
 231     return model_fn_lib.EstimatorSpec(
 232         mode=mode,
 233         predictions={
 234             KMeansClustering.ALL_DISTANCES: all_distances[0],
 235             KMeansClustering.CLUSTER_INDEX: model_predictions[0],
 236         },
 237         loss=loss,
 238         train_op=training_op,
 239         eval_metric_ops={KMeansClustering.SCORE: metrics.mean(loss)},
 240         training_hooks=training_hooks,
 241         export_outputs=export_outputs)

 245 class KMeansClustering(estimator.Estimator):
 246   """An Estimator for K-Means clustering.
 247 
 248   Example:
 249   ```
 250   import numpy as np
 251   import tensorflow as tf
 252 
 253   num_points = 100
 254   dimensions = 2
 255   points = np.random.uniform(0, 1000, [num_points, dimensions])
 256 
 257   def input_fn():
 258     return tf.train.limit_epochs(
 259         tf.convert_to_tensor(points, dtype=tf.float32), num_epochs=1)
 260 
 261   num_clusters = 5
 262   kmeans = tf.contrib.factorization.KMeansClustering(
 263       num_clusters=num_clusters, use_mini_batch=False)
 264 
 265   # train
 266   num_iterations = 10
 267   previous_centers = None
 268   for _ in xrange(num_iterations):
 269     kmeans.train(input_fn)
 270     cluster_centers = kmeans.cluster_centers()
 271     if previous_centers is not None:
 272       print 'delta:', cluster_centers - previous_centers
 273     previous_centers = cluster_centers
 274     print 'score:', kmeans.score(input_fn)
 275   print 'cluster centers:', cluster_centers
 276 
 277   # map the input points to their clusters
 278   cluster_indices = list(kmeans.predict_cluster_index(input_fn))
 279   for i, point in enumerate(points):
 280     cluster_index = cluster_indices[i]
 281     center = cluster_centers[cluster_index]
 282     print 'point:', point, 'is in cluster', cluster_index, 'centered at', center
 283   ```
 284 
 285   The `SavedModel` saved by the `export_savedmodel` method does not include the
 286   cluster centers. However, the cluster centers may be retrieved by the
 287   latest checkpoint saved during training. Specifically,
 288   ```
 289   kmeans.cluster_centers()
 290   ```
 291   is equivalent to
 292   ```
 293   tf.train.load_variable(
 294       kmeans.model_dir, KMeansClustering.CLUSTER_CENTERS_VAR_NAME)
 295   ```
 296   """
 297 
 298   # Valid values for the distance_metric constructor argument.
 299   SQUARED_EUCLIDEAN_DISTANCE = clustering_ops.SQUARED_EUCLIDEAN_DISTANCE
 300   COSINE_DISTANCE = clustering_ops.COSINE_DISTANCE
 301 
 302   # Values for initial_clusters constructor argument.
 303   RANDOM_INIT = clustering_ops.RANDOM_INIT
 304   KMEANS_PLUS_PLUS_INIT = clustering_ops.KMEANS_PLUS_PLUS_INIT
 305 
 306   # Metric returned by evaluate(): The sum of the squared distances from each
 307   # input point to its closest center.
 308   SCORE = 'score'
 309 
 310   # Keys returned by predict().
 311   # ALL_DISTANCES: The distance from each input  point to each cluster center.
 312   # CLUSTER_INDEX: The index of the closest cluster center for each input point.
 313   CLUSTER_INDEX = 'cluster_index'
 314   ALL_DISTANCES = 'all_distances'
 315 
 316   # Variable name used by cluster_centers().
 317   CLUSTER_CENTERS_VAR_NAME = clustering_ops.CLUSTERS_VAR_NAME
 318 
 319   def __init__(self,
 320                num_clusters,
 321                model_dir=None,
 322                initial_clusters=RANDOM_INIT,
 323                distance_metric=SQUARED_EUCLIDEAN_DISTANCE,
 324                random_seed=0,
 325                use_mini_batch=True,
 326                mini_batch_steps_per_iteration=1,
 327                kmeans_plus_plus_num_retries=2,
 328                relative_tolerance=None,
 329                config=None,
 330                feature_columns=None):
 331     """Creates an Estimator for running KMeans training and inference.
 332 
 333     This Estimator implements the following variants of the K-means algorithm:
 334 
 335     If `use_mini_batch` is False, it runs standard full batch K-means. Each
 336     training step runs a single iteration of K-Means and must process the full
 337     input at once. To run in this mode, the `input_fn` passed to `train` must
 338     return the entire input dataset.
 339 
 340     If `use_mini_batch` is True, it runs a generalization of the mini-batch
 341     K-means algorithm. It runs multiple iterations, where each iteration is
 342     composed of `mini_batch_steps_per_iteration` steps. Each training step
 343     accumulates the contribution from one mini-batch into temporary storage.
 344     Every `mini_batch_steps_per_iteration` steps, the cluster centers are
 345     updated and the temporary storage cleared for the next iteration. Note
 346     that:
 347       * If `mini_batch_steps_per_iteration=1`, the algorithm reduces to the
 348         standard K-means mini-batch algorithm.
 349       * If `mini_batch_steps_per_iteration = num_inputs / batch_size`, the
 350         algorithm becomes an asynchronous version of the full-batch algorithm.
 351         However, there is no guarantee by this implementation that each input
 352         is seen exactly once per iteration. Also, different updates are applied
 353         asynchronously without locking. So this asynchronous version may not
 354         behave exactly like a full-batch version.
 355 
 356     Args:
 357       num_clusters: An integer tensor specifying the number of clusters. This
 358         argument is ignored if `initial_clusters` is a tensor or numpy array.
 359       model_dir: The directory to save the model results and log files.
 360       initial_clusters: Specifies how the initial cluster centers are chosen.
 361         One of the following:
 362         * a tensor or numpy array with the initial cluster centers.
 363         * a callable `f(inputs, k)` that selects and returns up to `k` centers
 364               from an input batch. `f` is free to return any number of centers
 365               from `0` to `k`. It will be invoked on successive input batches
 366               as necessary until all `num_clusters` centers are chosen.
 367         * `KMeansClustering.RANDOM_INIT`: Choose centers randomly from an input
 368               batch. If the batch size is less than `num_clusters` then the
 369               entire batch is chosen to be initial cluster centers and the
 370               remaining centers are chosen from successive input batches.
 371         * `KMeansClustering.KMEANS_PLUS_PLUS_INIT`: Use kmeans++ to choose
 372               centers from the first input batch. If the batch size is less
 373               than `num_clusters`, a TensorFlow runtime error occurs.
 374       distance_metric: The distance metric used for clustering. One of:
 375         * `KMeansClustering.SQUARED_EUCLIDEAN_DISTANCE`: Euclidean distance
 376              between vectors `u` and `v` is defined as \\(||u - v||_2\\)
 377              which is the square root of the sum of the absolute squares of
 378              the elements' difference.
 379         * `KMeansClustering.COSINE_DISTANCE`: Cosine distance between vectors
 380              `u` and `v` is defined as \\(1 - (u . v) / (||u||_2 ||v||_2)\\).
 381       random_seed: Python integer. Seed for PRNG used to initialize centers.
 382       use_mini_batch: A boolean specifying whether to use the mini-batch k-means
 383         algorithm. See explanation above.
 384       mini_batch_steps_per_iteration: The number of steps after which the
 385         updated cluster centers are synced back to a master copy. Used only if
 386         `use_mini_batch=True`. See explanation above.
 387       kmeans_plus_plus_num_retries: For each point that is sampled during
 388         kmeans++ initialization, this parameter specifies the number of
 389         additional points to draw from the current distribution before selecting
 390         the best. If a negative value is specified, a heuristic is used to
 391         sample `O(log(num_to_sample))` additional points. Used only if
 392         `initial_clusters=KMeansClustering.KMEANS_PLUS_PLUS_INIT`.
 393       relative_tolerance: A relative tolerance of change in the loss between
 394         iterations. Stops learning if the loss changes less than this amount.
 395         This may not work correctly if `use_mini_batch=True`.
 396       config: See `tf.estimator.Estimator`.
 397       feature_columns: An optionable iterable containing all the feature columns
 398         used by the model. All items in the set should be feature column
 399         instances that can be passed to `tf.feature_column.input_layer`. If this
 400         is None, all features will be used.
 401 
 402     Raises:
 403       ValueError: An invalid argument was passed to `initial_clusters` or
 404         `distance_metric`.
 405     """
 406     if isinstance(initial_clusters, str) and initial_clusters not in [
 407         KMeansClustering.RANDOM_INIT, KMeansClustering.KMEANS_PLUS_PLUS_INIT
 408     ]:
 409       raise ValueError(
 410           "Unsupported initialization algorithm '%s'" % initial_clusters)
 411     if distance_metric not in [
 412         KMeansClustering.SQUARED_EUCLIDEAN_DISTANCE,
 413         KMeansClustering.COSINE_DISTANCE
 414     ]:
 415       raise ValueError("Unsupported distance metric '%s'" % distance_metric)
 416     super(KMeansClustering, self).__init__(
 417         model_fn=_ModelFn(
 418             num_clusters, initial_clusters, distance_metric, random_seed,
 419             use_mini_batch, mini_batch_steps_per_iteration,
 420             kmeans_plus_plus_num_retries, relative_tolerance,
 421             feature_columns).model_fn,
 422         model_dir=model_dir,
 423         config=config)
 424 
 425   def _predict_one_key(self, input_fn, predict_key):
 426     for result in self.predict(input_fn=input_fn, predict_keys=[predict_key]):
 427       yield result[predict_key]
 428 
 429   def predict_cluster_index(self, input_fn):
 430     """Finds the index of the closest cluster center to each input point.
 431 
 432     Args:
 433       input_fn: Input points. See `tf.estimator.Estimator.predict`.
 434 
 435     Yields:
 436       The index of the closest cluster center for each input point.
 437     """
 438     for index in self._predict_one_key(input_fn,
 439                                        KMeansClustering.CLUSTER_INDEX):
 440       yield index
 441 
 442   def score(self, input_fn):
 443     """Returns the sum of squared distances to nearest clusters.
 444 
 445     Note that this function is different from the corresponding one in sklearn
 446     which returns the negative sum.
 447 
 448     Args:
 449       input_fn: Input points. See `tf.estimator.Estimator.evaluate`. Only one
 450           batch is retrieved.
 451 
 452     Returns:
 453       The sum of the squared distance from each point in the first batch of
 454       inputs to its nearest cluster center.
 455     """
 456     return self.evaluate(input_fn=input_fn, steps=1)[KMeansClustering.SCORE]
 457 
 458   def transform(self, input_fn):
 459     """Transforms each input point to its distances to all cluster centers.
 460 
 461     Note that if `distance_metric=KMeansClustering.SQUARED_EUCLIDEAN_DISTANCE`,
 462     this
 463     function returns the squared Euclidean distance while the corresponding
 464     sklearn function returns the Euclidean distance.
 465 
 466     Args:
 467       input_fn: Input points. See `tf.estimator.Estimator.predict`.
 468 
 469     Yields:
 470       The distances from each input point to each cluster center.
 471     """
 472     for distances in self._predict_one_key(input_fn,
 473                                            KMeansClustering.ALL_DISTANCES):
 474       yield distances
 475 
 476   def cluster_centers(self):
 477     """Returns the cluster centers."""
 478     return self.get_variable_value(KMeansClustering.CLUSTER_CENTERS_VAR_NAME)
