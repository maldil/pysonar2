  22 from tensorflow.python.data.ops import dataset_ops

  28 class KafkaDataset(dataset_ops.DatasetSource):
  29   """A Kafka Dataset that consumes the message.
  30   """
  31 
  32   def __init__(self,
  33                topics,
  34                servers="localhost",
  35                group="",
  36                eof=False,
  37                timeout=1000):
  38     """Create a KafkaReader.
  39 
  40     Args:
  41       topics: A `tf.string` tensor containing one or more subscriptions,
  42               in the format of [topic:partition:offset:length],
  43               by default length is -1 for unlimited.
  44       servers: A list of bootstrap servers.
  45       group: The consumer group id.
  46       eof: If True, the kafka reader will stop on EOF.
  47       timeout: The timeout value for the Kafka Consumer to wait
  48                (in millisecond).
  49     """
  50     super(KafkaDataset, self).__init__()
  51     self._topics = ops.convert_to_tensor(
  52         topics, dtype=dtypes.string, name="topics")
  53     self._servers = ops.convert_to_tensor(
  54         servers, dtype=dtypes.string, name="servers")
  55     self._group = ops.convert_to_tensor(
  56         group, dtype=dtypes.string, name="group")
  57     self._eof = ops.convert_to_tensor(eof, dtype=dtypes.bool, name="eof")
  58     self._timeout = ops.convert_to_tensor(
  59         timeout, dtype=dtypes.int64, name="timeout")
  60 
  61   def _as_variant_tensor(self):
  62     return gen_dataset_ops.kafka_dataset(self._topics, self._servers,
  63                                          self._group, self._eof, self._timeout)
  64 
  65   @property
  66   def output_classes(self):
  67     return ops.Tensor
  68 
  69   @property
  70   def output_shapes(self):
  71     return tensor_shape.scalar()
  72 
  73   @property
  74   def output_types(self):
  75     return dtypes.string
