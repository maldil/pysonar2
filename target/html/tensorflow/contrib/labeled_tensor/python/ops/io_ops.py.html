  31 class FixedLenFeature(object):
  32   """Configuration for parsing a fixed-length input feature.
  33 
  34   Fields:
  35     axes: A list of Axis objects or tuples (axis_name, axis_value),
  36       where `axis_name` is a string and `axis_value` is None (unknown size), an
  37       integer or a list of tick labels.
  38     dtype: Data type of input.
  39     default_value: Value to be used if an example is missing this feature. It
  40         must be compatible with `dtype`.
  41   """
  42 
  43   def __init__(self, axes, dtype, default_value=None):
  44     self._axes = [core.as_axis(a) for a in axes]
  45     self._dtype = dtype
  46     self._default_value = default_value
  47 
  48   @property
  49   def axes(self):
  50     return self._axes
  51 
  52   @property
  53   def dtype(self):
  54     return self._dtype
  55 
  56   @property
  57   def default_value(self):
  58     return self._default_value

  61 @tc.returns(tc.Dict(string_types, parsing_ops.FixedLenFeature))
  62 @tc.accepts(tc.Mapping(string_types, FixedLenFeature))
  63 def _labeled_to_unlabeled_features(features):
  64   """Convert a dict of lt.FixedLenFeature into a dict of tf.FixedLenFeature."""
  65   unlabeled_features = {}
  66   for name, labeled_feature in features.items():
  67     shape = [ax.size for ax in labeled_feature.axes]
  68     if any(size is None for size in shape):
  69       # This should be caught on the TensorFlow side, but it isn't yet:
  70       # https://github.com/tensorflow/tensorflow/issues/2874
  71       raise ValueError('axes with unknown size are not supported')
  72     dtype = labeled_feature.dtype
  73     default_value = labeled_feature.default_value
  74     unlabeled_features[name] = parsing_ops.FixedLenFeature(
  75         shape, dtype, default_value)
  76   return unlabeled_features

  79 @tc.returns(tc.Dict(string_types, core.LabeledTensor))
  80 @tc.accepts(core.LabeledTensorLike, tc.Mapping(string_types, FixedLenFeature),
  81             tc.Optional(string_types), object)
  82 def parse_example(serialized, features, name=None, example_names=None):
  83   """Parse `Example` protos into a `dict` of labeled tensors.
  84 
  85   See tf.parse_example.
  86 
  87   Args:
  88     serialized: A 1-D LabeledTensor of strings, a batch of binary serialized
  89       `Example` protos.
  90     features: A `dict` mapping feature keys to `labeled_tensor.FixedLenFeature`
  91       values.
  92     name: A name for this operation (optional).
  93     example_names: A vector (1-D Tensor) of strings (optional), the names of
  94       the serialized protos in the batch.
  95 
  96   Returns:
  97     A `dict` mapping feature keys to `LabeledTensor` values. The single axis
  98     from `serialized` will be prepended to the axes provided by each feature.
  99 
 100   Raises:
 101     ValueError: if any feature is invalid.
 102   """
 103   serialized = core.convert_to_labeled_tensor(serialized)
 104   unlabeled_features = _labeled_to_unlabeled_features(features)
 105 
 106   unlabeled_parsed = parsing_ops.parse_example(
 107       serialized.tensor, unlabeled_features, name, example_names)
 108 
 109   parsed = {}
 110   for name, parsed_feature in unlabeled_parsed.items():
 111     axes = list(serialized.axes.values()) + features[name].axes
 112     parsed[name] = core.LabeledTensor(parsed_feature, axes)
 113 
 114   return parsed

 120 def parse_single_example(serialized, features, name=None, example_names=None):
 121   """Parses a single `Example` proto.
 122 
 123   See tf.parse_single_example.
 124 
 125   Args:
 126     serialized: A scalar string Tensor or LabeledTensor, a single serialized
 127       Example.
 128     features: A `dict` mapping feature keys to `labeled_tensor.FixedLenFeature`
 129       values.
 130     name: A name for this operation (optional).
 131     example_names: (Optional) A scalar string Tensor, the associated name.
 132 
 133   Returns:
 134     A `dict` mapping feature keys to `LabeledTensor` values.
 135 
 136   Raises:
 137     ValueError: if any feature is invalid.
 138   """
 139   serialized = core.convert_to_labeled_tensor(serialized)
 140   unlabeled_features = _labeled_to_unlabeled_features(features)
 141 
 142   unlabeled_parsed = parsing_ops.parse_single_example(
 143       serialized.tensor, unlabeled_features, name, example_names)
 144 
 145   parsed = {}
 146   for name, parsed_feature in unlabeled_parsed.items():
 147     parsed[name] = core.LabeledTensor(parsed_feature, features[name].axes)
 148 
 149   return parsed

 152 @tc.returns(core.LabeledTensor)
 153 @tc.accepts(dtypes.DType, tc.Collection(tc.Union(string_types, core.AxisLike)),
 154             tc.Optional(string_types))
 155 def placeholder(dtype, axes, name=None):
 156   """Create a placeholder for a labeled tensor.
 157 
 158   For example:
 159 
 160     lt.placeholder(tf.float32, ['batch', ('channel', ['r', 'g', 'b'])])
 161 
 162   See tf.placeholder for more details.
 163 
 164   Args:
 165     dtype: The type of elements in the tensor to be fed.
 166     axes: sequence of strings (denoting axes of unknown size) and/or objects
 167       convertable to lt.Axis to label the result.
 168     name: Optional op name.
 169 
 170   Returns:
 171     Placeholder labeled tensor.
 172   """
 173   with ops.name_scope(name, 'lt_placeholder', []) as scope:
 174     axes = core.Axes([(axis, None) if isinstance(axis, string_types) else axis
 175                       for axis in axes])
 176     shape = [axis.size for axis in axes.values()]
 177     tensor = array_ops.placeholder(dtype, shape, name=scope)
 178     return core.LabeledTensor(tensor, axes)
