  24 from tensorflow.contrib.labeled_tensor.python.ops import core
  25 from tensorflow.contrib.labeled_tensor.python.ops import ops
  26 from tensorflow.python.framework import ops as tf_ops

  29 class ReshapeCoder(object):
  30   """Utility class for mapping to and from another shape.
  31 
  32   For example, say you have a function `crop_center` which expects a
  33   LabeledTensor with axes named ['batch', 'row', 'column', 'depth'], and
  34   you have a LabeledTensor `masked_image_lt` with axes ['batch', 'row',
  35   'column', 'channel', 'mask'].
  36 
  37   To call `crop_center` with `masked_image_lt` you'd normally have to write:
  38 
  39   >>> reshape_lt = lt.reshape(masked_image_lt, ['channel', 'mask'], ['depth'])
  40   >>> crop_lt = crop_center(reshape_lt)
  41   >>> result_lt = lt.reshape(crop_lt, ['depth'],
  42   ...   [masked_image_lt.axes['channel'], masked_image_lt.axes['mask']])
  43 
  44   ReshapeCoder takes care of this renaming logic for you, allowing you to
  45   instead write:
  46 
  47   >>> rc = ReshapeCoder(['channel', 'mask'], ['depth'])
  48   >>> result_lt = rc.decode(crop_center(rc.encode(masked_image_lt)))
  49 
  50   Here, `decode` restores the original axes 'channel' and 'mask', so
  51   `crop_center` must not have modified the size of the 'depth' axis.
  52   """
  53 
  54   @tc.accepts(object, tc.Collection(str),
  55               tc.Collection(tc.Union(str, core.AxisLike)), tc.Optional(str))
  56   def __init__(self, existing_axis_names, new_axes, name=None):
  57     self._name = name
  58     self._existing_axis_names = existing_axis_names
  59     self._new_axes = new_axes
  60 
  61     self._existing_axes = None
  62 
  63   @tc.returns(core.LabeledTensor)
  64   @tc.accepts(object, core.LabeledTensorLike)
  65   def encode(self, labeled_tensor):
  66     """Reshape the input to the target shape.
  67 
  68     If called several times, the axes named in existing_axis_names must be
  69     identical.
  70 
  71     Args:
  72       labeled_tensor: The input tensor.
  73 
  74     Returns:
  75       The input reshaped to the target shape.
  76 
  77     Raises:
  78       ValueError: If the axes in existing_axis_names don't match the axes of
  79         a tensor in a previous invocation of this method.
  80     """
  81     with tf_ops.name_scope(self._name, 'lt_reshape_encode',
  82                            [labeled_tensor]) as scope:
  83       labeled_tensor = core.convert_to_labeled_tensor(labeled_tensor)
  84 
  85       reshape_lt = ops.reshape(labeled_tensor,
  86                                self._existing_axis_names,
  87                                self._new_axes,
  88                                name=scope)
  89 
  90       axes = [labeled_tensor.axes[n] for n in self._existing_axis_names]
  91       if self._existing_axes is not None and self._existing_axes != axes:
  92         raise ValueError(
  93             'input axes %r do not match axes from previous method call %r' %
  94             (axes, self._existing_axes))
  95       else:
  96         self._existing_axes = axes
  97 
  98       return reshape_lt
  99 
 100   @tc.returns(core.LabeledTensor)
 101   @tc.accepts(object, core.LabeledTensorLike)
 102   def decode(self, labeled_tensor):
 103     """Reshape the input to the original shape.
 104 
 105     This is the inverse of encode.
 106     Encode must have been called at least once prior to this method being
 107     called.
 108 
 109     Args:
 110       labeled_tensor: The input tensor.
 111 
 112     Returns:
 113       The input reshaped to the original shape.
 114 
 115     Raises:
 116       ValueError: If this method was called before encode was called.
 117     """
 118     if self._existing_axes is None:
 119       raise ValueError('decode called before encode')
 120 
 121     with tf_ops.name_scope(self._name, 'lt_reshape_decode',
 122                            [labeled_tensor]) as scope:
 123       labeled_tensor = core.convert_to_labeled_tensor(labeled_tensor)
 124 
 125       new_axis_names = [axis if isinstance(axis, string_types) else
 126                         core.as_axis(axis).name for axis in self._new_axes]
 127 
 128       return ops.reshape(labeled_tensor,
 129                          new_axis_names,
 130                          self._existing_axes,
 131                          name=scope)
