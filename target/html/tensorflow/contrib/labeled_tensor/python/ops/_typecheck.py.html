  35 class Type(object):
  36   """Base class for type checker types.
  37 
  38   The custom types defined in this module are based on types in the standard
  39   library's typing module (in Python 3.5):
  40   https://docs.python.org/3/library/typing.html
  41 
  42   The only difference should be that we use actual instances of Type classes to
  43   represent custom types rather than the metaclass magic typing uses to create
  44   new class objects. In practice, all this should mean is that we use
  45   `List(int)` rather than `List[int]`.
  46 
  47   Custom types should implement __instancecheck__ and inherit from Type. Every
  48   argument in the constructor must be a type or Type instance, and these
  49   arguments must be stored as a tuple on the `_types` attribute.
  50   """
  51 
  52   def __init__(self, *types):
  53     self._types = types
  54 
  55   def __repr__(self):
  56     args_repr = ", ".join(repr(t) for t in self._types)
  57     return  "typecheck.%s(%s)" % (type(self).__name__, args_repr)
  58 
  59 
  60 class _SingleArgumentType(Type):
  61   """Use this subclass for parametric types that accept only one argument."""
  62 
  63   def __init__(self, tpe):
  64     super(_SingleArgumentType, self).__init__(tpe)
  65 
  66   @property
  67   def _type(self):
  68     tpe, = self._types  # pylint: disable=unbalanced-tuple-unpacking
  69     return tpe
  70 
  71 
  72 class _TwoArgumentType(Type):
  73   """Use this subclass for parametric types that accept two arguments."""
  74 
  75   def __init__(self, first_type, second_type):
  76     super(_TwoArgumentType, self).__init__(first_type, second_type)
  77

  79 class Union(Type):
  80   """A sum type.
  81 
  82   A correct type is any of the types provided.
  83   """
  84 
  85   def __instancecheck__(self, instance):
  86     return isinstance(instance, self._types)
  87 
  88 
  89 class Optional(_SingleArgumentType):
  90   """An optional type.
  91 
  92   A correct type is either the provided type or NoneType.
  93   """
  94 
  95   def __instancecheck__(self, instance):
  96     # types.NoneType does not exist in Python 3
  97     return isinstance(instance, (self._type, type(None)))

 100 class List(_SingleArgumentType):
 101   """A typed list.
 102 
 103   A correct type is a list where each element has the single provided type.
 104   """
 105 
 106   def __instancecheck__(self, instance):
 107     return (isinstance(instance, list)
 108             and all(isinstance(x, self._type) for x in instance))
 109

 111 class Sequence(_SingleArgumentType):
 112   """A typed sequence.
 113 
 114   A correct type is a sequence where each element has the single provided type.
 115   """
 116 
 117   def __instancecheck__(self, instance):
 118     return (isinstance(instance, collections.Sequence)
 119             and all(isinstance(x, self._type) for x in instance))
 120 
 121 
 122 class Collection(_SingleArgumentType):
 123   """A sized, iterable container.
 124 
 125   A correct type is an iterable and container with known size where each element
 126   has the single provided type.
 127 
 128   We use this in preference to Iterable because we check each instance of the
 129   iterable at runtime, and hence need to avoid iterables that could be
 130   exhausted.
 131   """
 132 
 133   def __instancecheck__(self, instance):
 134     return (isinstance(instance, collections.Iterable)
 135             and isinstance(instance, collections.Sized)
 136             and isinstance(instance, collections.Container)
 137             and all(isinstance(x, self._type) for x in instance))
 138 
 139 
 140 class Tuple(Type):
 141   """A typed tuple.
 142 
 143   A correct type is a tuple with the correct length where each element has
 144   the correct type.
 145   """
 146 
 147   def __instancecheck__(self, instance):
 148     return (isinstance(instance, tuple)
 149             and len(instance) == len(self._types)
 150             and all(isinstance(x, t) for x, t in zip(instance, self._types)))
 151 
 152 
 153 class Mapping(_TwoArgumentType):
 154   """A typed mapping.
 155 
 156   A correct type has the correct parametric types for keys and values.
 157   """
 158 
 159   def __instancecheck__(self, instance):
 160     key_type, value_type = self._types  # pylint: disable=unbalanced-tuple-unpacking
 161     return (isinstance(instance, collections.Mapping)
 162             and all(isinstance(k, key_type) for k in instance.keys())
 163             and all(isinstance(k, value_type) for k in instance.values()))
 164 
 165 
 166 class Dict(Mapping):
 167   """A typed dict.
 168 
 169   A correct type has the correct parametric types for keys and values.
 170   """
 171 
 172   def __instancecheck__(self, instance):
 173     return (isinstance(instance, dict)
 174             and super(Dict, self).__instancecheck__(instance))

 177 def _replace_forward_references(t, context):
 178   """Replace forward references in the given type."""
 179   if isinstance(t, str):
 180     return context[t]
 181   elif isinstance(t, Type):
 182     return type(t)(*[_replace_forward_references(t, context) for t in t._types])  # pylint: disable=protected-access
 183   else:
 184     return t

 187 def register_type_abbreviation(name, alias):
 188   """Register an abbreviation for a type in typecheck tracebacks.
 189 
 190   This makes otherwise very long typecheck errors much more readable.
 191 
 192   Example:
 193     typecheck.register_type_abbreviation(tf.Dimension, 'tf.Dimension')
 194 
 195   Args:
 196     name: type or class to abbreviate.
 197     alias: string alias to substitute.
 198   """
 199   _TYPE_ABBREVIATIONS[name] = alias
 200 
 201 
 202 def _type_repr(t):
 203   """A more succinct repr for typecheck tracebacks."""
 204   string = repr(t)
 205   for type_, alias in _TYPE_ABBREVIATIONS.items():
 206     string = string.replace(repr(type_), alias)
 207   string = re.sub(r"<(class|type) '([\w.]+)'>", r"\2", string)
 208   string = re.sub(r"typecheck\.(\w+)", r"\1", string)
 209   return string

 216 def accepts(*types):
 217   """A decorator which checks the input types of a function.
 218 
 219   Based on:
 220   http://stackoverflow.com/questions/15299878/how-to-use-python-decorators-to-check-function-arguments
 221   The above draws from:
 222   https://www.python.org/dev/peps/pep-0318/
 223 
 224   Args:
 225     *types: A list of Python types.
 226 
 227   Returns:
 228     A function to use as a decorator.
 229   """
 230 
 231   def check_accepts(f):
 232     """Check the types."""
 233     spec = tf_inspect.getargspec(f)
 234 
 235     num_function_arguments = len(spec.args)
 236     if len(types) != num_function_arguments:
 237       raise Error(
 238           "Function %r has %d arguments but only %d types were provided in the "
 239           "annotation." % (f, num_function_arguments, len(types)))
 240 
 241     if spec.defaults:
 242       num_defaults = len(spec.defaults)
 243       for (name, a, t) in zip(spec.args[-num_defaults:],
 244                               spec.defaults,
 245                               types[-num_defaults:]):
 246         allowed_type = _replace_forward_references(t, f.__globals__)
 247         if not isinstance(a, allowed_type):
 248           raise Error("default argument value %r of type %r is not an instance "
 249                       "of the allowed type %s for the %s argument to %r"
 250                       % (a, type(a), _type_repr(allowed_type), name, f))
 251 
 252     @functools.wraps(f)
 253     def new_f(*args, **kwds):
 254       """A helper function."""
 255       for (a, t) in zip(args, types):
 256         allowed_type = _replace_forward_references(t, f.__globals__)
 257         if not isinstance(a, allowed_type):
 258           raise Error("%r of type %r is not an instance of the allowed type %s "
 259                       "for %r" % (a, type(a), _type_repr(allowed_type), f))
 260       return f(*args, **kwds)
 261 
 262     return new_f
 263 
 264   return check_accepts

 267 def returns(*types):
 268   """A decorator which checks the return types of a function.
 269 
 270   Based on:
 271   http://stackoverflow.com/questions/15299878/how-to-use-python-decorators-to-check-function-arguments
 272   The above draws from:
 273   https://www.python.org/dev/peps/pep-0318/
 274 
 275   Args:
 276     *types: A list of Python types.
 277       A list of one element corresponds to a single return value.
 278       A list of several elements corresponds to several return values.
 279       Note that a function with no explicit return value has an implicit
 280       NoneType return and should be annotated correspondingly.
 281 
 282   Returns:
 283     A function to use as a decorator.
 284   """
 285 
 286   def check_returns(f):
 287     """Check the types."""
 288     if not types:
 289       raise TypeError("A return type annotation must contain at least one type")
 290 
 291     @functools.wraps(f)
 292     def new_f(*args, **kwds):
 293       """A helper function."""
 294       return_value = f(*args, **kwds)
 295 
 296       if len(types) == 1:
 297         # The function has a single return value.
 298         allowed_type = _replace_forward_references(types[0], f.__globals__)
 299         if not isinstance(return_value, allowed_type):
 300           raise Error("%r of type %r is not an instance of the allowed type %s "
 301                       "for %r"
 302                       % (return_value, type(return_value),
 303                          _type_repr(allowed_type), f))
 304 
 305       else:
 306         if len(return_value) != len(types):
 307           raise Error(
 308               "Function %r has %d return values but only %d types were "
 309               "provided in the annotation." %
 310               (f, len(return_value), len(types)))
 311 
 312         for (r, t) in zip(return_value, types):
 313           allowed_type = _replace_forward_references(t, f.__globals__)
 314           if not isinstance(r, allowed_type):
 315             raise Error("%r of type %r is not an instance of allowed type %s "
 316                         "for %r" % (r, type(r), _type_repr(allowed_type), f))
 317 
 318       return return_value
 319 
 320     return new_f
 321 
 322   return check_returns
