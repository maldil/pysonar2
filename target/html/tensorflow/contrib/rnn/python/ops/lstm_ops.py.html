  41 def _lstm_block_cell(x,
  42                      cs_prev,
  43                      h_prev,
  44                      w,
  45                      b,
  46                      wci=None,
  47                      wcf=None,
  48                      wco=None,
  49                      forget_bias=None,
  50                      cell_clip=None,
  51                      use_peephole=None,
  52                      name=None):
  53   r"""Computes the LSTM cell forward propagation for 1 time step.
  54 
  55   This implementation uses 1 weight matrix and 1 bias vector, and there's an
  56   optional peephole connection.
  57 
  58   This kernel op implements the following mathematical equations:
  59 
  60   ```python
  61   xh = [x, h_prev]
  62   [i, ci, f, o] = xh * w + b
  63   f = f + forget_bias
  64 
  65   if not use_peephole:
  66     wci = wcf = wco = 0
  67 
  68   i = sigmoid(cs_prev * wci + i)
  69   f = sigmoid(cs_prev * wcf + f)
  70   ci = tanh(ci)
  71 
  72   cs = ci .* i + cs_prev .* f
  73   cs = clip(cs, cell_clip)
  74 
  75   o = sigmoid(cs * wco + o)
  76   co = tanh(cs)
  77   h = co .* o
  78   ```
  79 
  80   Args:
  81     x: A `Tensor`. Must be one of the following types: `float32`.
  82       The input to the LSTM cell, shape (batch_size, num_inputs).
  83     cs_prev: A `Tensor`. Must have the same type as `x`.
  84       Value of the cell state at previous time step.
  85     h_prev: A `Tensor`. Must have the same type as `x`.
  86       Output of the previous cell at previous time step.
  87     w: A `Tensor`. Must have the same type as `x`. The weight matrix.
  88     b: A `Tensor`. Must have the same type as `x`. The bias vector.
  89     wci: A `Tensor`. Must have the same type as `x`.
  90       The weight matrix for input gate peephole connection.
  91     wcf: A `Tensor`. Must have the same type as `x`.
  92       The weight matrix for forget gate peephole connection.
  93     wco: A `Tensor`. Must have the same type as `x`.
  94       The weight matrix for output gate peephole connection.
  95     forget_bias: An optional `float`. Defaults to `1`. The forget gate bias.
  96     cell_clip: An optional `float`. Defaults to `-1` (no clipping).
  97       Value to clip the 'cs' value to. Disable by setting to negative value.
  98     use_peephole: An optional `bool`. Defaults to `False`.
  99       Whether to use peephole weights.
 100     name: A name for the operation (optional).
 101 
 102   Returns:
 103     A tuple of `Tensor` objects (i, cs, f, o, ci, co, h).
 104     i: A `Tensor`. Has the same type as `x`. The input gate.
 105     cs: A `Tensor`. Has the same type as `x`. The cell state before the tanh.
 106     f: A `Tensor`. Has the same type as `x`. The forget gate.
 107     o: A `Tensor`. Has the same type as `x`. The output gate.
 108     ci: A `Tensor`. Has the same type as `x`. The cell input.
 109     co: A `Tensor`. Has the same type as `x`. The cell after the tanh.
 110     h: A `Tensor`. Has the same type as `x`. The output h vector.
 111 
 112   Raises:
 113     ValueError: If cell_size is None.
 114   """
 115   if wci is None:
 116     cell_size = cs_prev.get_shape().with_rank(2)[1].value
 117     if cell_size is None:
 118       raise ValueError("cell_size from `cs_prev` should not be None.")
 119     wci = array_ops.constant(0, dtype=dtypes.float32, shape=[cell_size])
 120     wcf = wci
 121     wco = wci
 122 
 123   # pylint: disable=protected-access
 124   return gen_lstm_ops.lstm_block_cell(
 125       x=x,
 126       cs_prev=cs_prev,
 127       h_prev=h_prev,
 128       w=w,
 129       wci=wci,
 130       wcf=wcf,
 131       wco=wco,
 132       b=b,
 133       forget_bias=forget_bias,
 134       cell_clip=cell_clip if cell_clip is not None else -1,
 135       use_peephole=use_peephole,
 136       name=name)
 137   # pylint: enable=protected-access

 140 def _block_lstm(seq_len_max,
 141                 x,
 142                 w,
 143                 b,
 144                 cs_prev=None,
 145                 h_prev=None,
 146                 wci=None,
 147                 wcf=None,
 148                 wco=None,
 149                 forget_bias=None,
 150                 cell_clip=None,
 151                 use_peephole=None,
 152                 name=None):
 153   r"""TODO(williamchan): add doc.
 154 
 155   Args:
 156     seq_len_max: A `Tensor` of type `int64`.
 157     x: A list of at least 1 `Tensor` objects of the same type in: `float32`.
 158     w: A `Tensor`. Must have the same type as `x`.
 159     b: A `Tensor`. Must have the same type as `x`.
 160     cs_prev: A `Tensor`. Must have the same type as `x`.
 161     h_prev: A `Tensor`. Must have the same type as `x`.
 162     wci: A `Tensor`. Must have the same type as `x`.
 163     wcf: A `Tensor`. Must have the same type as `x`.
 164     wco: A `Tensor`. Must have the same type as `x`.
 165     forget_bias: An optional `float`. Defaults to `1`.
 166     cell_clip: An optional `float`. Defaults to `-1` (no clipping).
 167     use_peephole: An optional `bool`. Defaults to `False`.
 168     name: A name for the operation (optional).
 169 
 170   Returns:
 171     A tuple of `Tensor` objects (i, cs, f, o, ci, co, h).
 172     i: A list with the same number of `Tensor` objects as `x` of `Tensor`
 173     objects of the same type as x.
 174     cs: A list with the same number of `Tensor` objects as `x` of `Tensor`
 175     objects of the same type as x.
 176     f: A list with the same number of `Tensor` objects as `x` of `Tensor`
 177     objects of the same type as x.
 178     o: A list with the same number of `Tensor` objects as `x` of `Tensor`
 179     objects of the same type as x.
 180     ci: A list with the same number of `Tensor` objects as `x` of `Tensor`
 181     objects of the same type as x.
 182     co: A list with the same number of `Tensor` objects as `x` of `Tensor`
 183     objects of the same type as x.
 184     h: A list with the same number of `Tensor` objects as `x` of `Tensor`
 185     objects of the same type as x.
 186 
 187   Raises:
 188     ValueError: If `b` does not have a valid shape.
 189   """
 190   batch_size = x[0].get_shape().with_rank(2)[0].value
 191   cell_size4 = b.get_shape().with_rank(1)[0].value
 192   if cell_size4 is None:
 193     raise ValueError("`b` shape must not be None.")
 194   cell_size = cell_size4 / 4
 195   zero_state = None
 196   if cs_prev is None or h_prev is None:
 197     zero_state = array_ops.constant(
 198         0, dtype=dtypes.float32, shape=[batch_size, cell_size])
 199   if cs_prev is None:
 200     cs_prev = zero_state
 201   if h_prev is None:
 202     h_prev = zero_state
 203   if wci is None:
 204     wci = array_ops.constant(0, dtype=dtypes.float32, shape=[cell_size])
 205     wcf = wci
 206     wco = wci
 207 
 208   # pylint: disable=protected-access
 209   i, cs, f, o, ci, co, h = gen_lstm_ops.block_lstm(
 210       seq_len_max=seq_len_max,
 211       x=array_ops.stack(x),
 212       cs_prev=cs_prev,
 213       h_prev=h_prev,
 214       w=w,
 215       wci=wci,
 216       wcf=wcf,
 217       wco=wco,
 218       b=b,
 219       forget_bias=forget_bias,
 220       cell_clip=cell_clip if cell_clip is not None else -1,
 221       name=name,
 222       use_peephole=use_peephole)
 223 
 224   return array_ops.unstack(i), array_ops.unstack(cs), array_ops.unstack(
 225       f), array_ops.unstack(o), array_ops.unstack(ci), array_ops.unstack(
 226           co), array_ops.unstack(h)
 227   # pylint: enable=protected-access
 228   # pylint: enable=invalid-name

 234 @ops.RegisterGradient("LSTMBlockCell")
 235 def _LSTMBlockCellGrad(op, *grad):
 236   """Gradient for LSTMBlockCell."""
 237   (x, cs_prev, h_prev, w, wci, wcf, wco, b) = op.inputs
 238   (i, cs, f, o, ci, co, _) = op.outputs
 239   (_, cs_grad, _, _, _, _, h_grad) = grad
 240 
 241   batch_size = x.get_shape().with_rank(2)[0].value
 242   if batch_size is None:
 243     batch_size = -1
 244   input_size = x.get_shape().with_rank(2)[1].value
 245   if input_size is None:
 246     raise ValueError("input_size from `x` should not be None.")
 247   cell_size = cs_prev.get_shape().with_rank(2)[1].value
 248   if cell_size is None:
 249     raise ValueError("cell_size from `cs_prev` should not be None.")
 250 
 251   (cs_prev_grad, dicfo, wci_grad, wcf_grad,
 252    wco_grad) = gen_lstm_ops.lstm_block_cell_grad(
 253        x,
 254        cs_prev,
 255        h_prev,
 256        w,
 257        wci,
 258        wcf,
 259        wco,
 260        b,
 261        i,
 262        cs,
 263        f,
 264        o,
 265        ci,
 266        co,
 267        cs_grad,
 268        h_grad,
 269        use_peephole=op.get_attr("use_peephole"))
 270 
 271   # Backprop from dicfo to xh.
 272   xh_grad = math_ops.matmul(dicfo, w, transpose_b=True)
 273 
 274   x_grad = array_ops.slice(xh_grad, (0, 0), (batch_size, input_size))
 275   x_grad.get_shape().merge_with(x.get_shape())
 276 
 277   h_prev_grad = array_ops.slice(xh_grad, (0, input_size),
 278                                 (batch_size, cell_size))
 279   h_prev_grad.get_shape().merge_with(h_prev.get_shape())
 280 
 281   # Backprop from dicfo to w.
 282   xh = array_ops.concat([x, h_prev], 1)
 283   w_grad = math_ops.matmul(xh, dicfo, transpose_a=True)
 284   w_grad.get_shape().merge_with(w.get_shape())
 285 
 286   # Backprop from dicfo to b.
 287   b_grad = nn_ops.bias_add_grad(dicfo)
 288   b_grad.get_shape().merge_with(b.get_shape())
 289 
 290   return (x_grad, cs_prev_grad, h_prev_grad, w_grad, wci_grad, wcf_grad,
 291           wco_grad, b_grad)

 331 class LSTMBlockCell(LayerRNNCell):
 332   """Basic LSTM recurrent network cell.
 333 
 334   The implementation is based on: http://arxiv.org/abs/1409.2329.
 335 
 336   We add `forget_bias` (default: 1) to the biases of the forget gate in order to
 337   reduce the scale of forgetting in the beginning of the training.
 338 
 339   Unlike `rnn_cell_impl.LSTMCell`, this is a monolithic op and should be much
 340   faster.  The weight and bias matrices should be compatible as long as the
 341   variable scope matches.
 342   """
 343 
 344   def __init__(self,
 345                num_units,
 346                forget_bias=1.0,
 347                cell_clip=None,
 348                use_peephole=False,
 349                reuse=None,
 350                name="lstm_cell"):
 351     """Initialize the basic LSTM cell.
 352 
 353     Args:
 354       num_units: int, The number of units in the LSTM cell.
 355       forget_bias: float, The bias added to forget gates (see above).
 356       cell_clip: An optional `float`. Defaults to `-1` (no clipping).
 357       use_peephole: Whether to use peephole connections or not.
 358       reuse: (optional) boolean describing whether to reuse variables in an
 359         existing scope.  If not `True`, and the existing scope already has the
 360         given variables, an error is raised.
 361       name: String, the name of the layer. Layers with the same name will
 362         share weights, but to avoid mistakes we require reuse=True in such
 363         cases.  By default this is "lstm_cell", for variable-name compatibility
 364         with `tf.nn.rnn_cell.LSTMCell`.
 365 
 366       When restoring from CudnnLSTM-trained checkpoints, must use
 367       CudnnCompatibleLSTMBlockCell instead.
 368     """
 369     super(LSTMBlockCell, self).__init__(_reuse=reuse, name=name)
 370     self._num_units = num_units
 371     self._forget_bias = forget_bias
 372     self._use_peephole = use_peephole
 373     self._cell_clip = cell_clip if cell_clip is not None else -1
 374     self._names = {
 375         "W": "kernel",
 376         "b": "bias",
 377         "wci": "w_i_diag",
 378         "wcf": "w_f_diag",
 379         "wco": "w_o_diag",
 380         "scope": "lstm_cell"
 381     }
 382     # Inputs must be 2-dimensional.
 383     self.input_spec = base_layer.InputSpec(ndim=2)
 384 
 385   @property
 386   def state_size(self):
 387     return rnn_cell_impl.LSTMStateTuple(self._num_units, self._num_units)
 388 
 389   @property
 390   def output_size(self):
 391     return self._num_units
 392 
 393   def build(self, inputs_shape):
 394     if not inputs_shape[1].value:
 395       raise ValueError(
 396           "Expecting inputs_shape[1] to be set: %s" % str(inputs_shape))
 397     input_size = inputs_shape[1].value
 398     self._kernel = self.add_variable(
 399         self._names["W"], [input_size + self._num_units, self._num_units * 4])
 400     self._bias = self.add_variable(
 401         self._names["b"], [self._num_units * 4],
 402         initializer=init_ops.constant_initializer(0.0))
 403     if self._use_peephole:
 404       self._w_i_diag = self.add_variable(self._names["wci"], [self._num_units])
 405       self._w_f_diag = self.add_variable(self._names["wcf"], [self._num_units])
 406       self._w_o_diag = self.add_variable(self._names["wco"], [self._num_units])
 407 
 408     self.built = True
 409 
 410   def call(self, inputs, state):
 411     """Long short-term memory cell (LSTM)."""
 412     if len(state) != 2:
 413       raise ValueError("Expecting state to be a tuple with length 2.")
 414 
 415     if self._use_peephole:
 416       wci = self._w_i_diag
 417       wcf = self._w_f_diag
 418       wco = self._w_o_diag
 419     else:
 420       wci = wcf = wco = array_ops.zeros([self._num_units])
 421 
 422     (cs_prev, h_prev) = state
 423     (_, cs, _, _, _, _, h) = _lstm_block_cell(
 424         inputs,
 425         cs_prev,
 426         h_prev,
 427         self._kernel,
 428         self._bias,
 429         wci=wci,
 430         wcf=wcf,
 431         wco=wco,
 432         forget_bias=self._forget_bias,
 433         cell_clip=self._cell_clip,
 434         use_peephole=self._use_peephole)
 435 
 436     new_state = rnn_cell_impl.LSTMStateTuple(cs, h)
 437     return h, new_state

 440 class LSTMBlockWrapper(base_layer.Layer):
 441   """This is a helper class that provides housekeeping for LSTM cells.
 442 
 443   This may be useful for alternative LSTM and similar type of cells.
 444   The subclasses must implement `_call_cell` method and `num_units` property.
 445   """
 446 
 447   @abc.abstractproperty
 448   def num_units(self):
 449     """Number of units in this cell (output dimension)."""
 450     pass
 451 
 452   @abc.abstractmethod
 453   def _call_cell(self, inputs, initial_cell_state, initial_output, dtype,
 454                  sequence_length):
 455     """Run this LSTM on inputs, starting from the given state.
 456 
 457     This method must be implemented by subclasses and does the actual work
 458     of calling the cell.
 459 
 460     Args:
 461       inputs: `3-D` tensor with shape `[time_len, batch_size, input_size]`
 462       initial_cell_state: initial value for cell state, shape `[batch_size,
 463         self._num_units]`
 464       initial_output: initial value of cell output, shape `[batch_size,
 465         self._num_units]`
 466       dtype: The data type for the initial state and expected output.
 467       sequence_length: Specifies the length of each sequence in inputs. An int32
 468         or int64 vector (tensor) size [batch_size], values in [0, time_len) or
 469           None.
 470 
 471     Returns:
 472       A pair containing:
 473 
 474       - State: A `3-D` tensor of shape `[time_len, batch_size, output_size]`
 475       - Output: A `3-D` tensor of shape `[time_len, batch_size, output_size]`
 476     """
 477     pass
 478 
 479   def call(self, inputs, initial_state=None, dtype=None, sequence_length=None):
 480     """Run this LSTM on inputs, starting from the given state.
 481 
 482     Args:
 483       inputs: `3-D` tensor with shape `[time_len, batch_size, input_size]`.
 484       initial_state: a tuple `(initial_cell_state, initial_output)` with tensors
 485         of shape `[batch_size, self._num_units]`. If this is not provided, the
 486         cell is expected to create a zero initial state of type `dtype`.
 487       dtype: The data type for the initial state and expected output. Required
 488         if `initial_state` is not provided or RNN state has a heterogeneous
 489         dtype.
 490       sequence_length: Specifies the length of each sequence in inputs. An
 491         `int32` or `int64` vector (tensor) size `[batch_size]`, values in `[0,
 492         time_len).`
 493         Defaults to `time_len` for each element.
 494 
 495     Returns:
 496       A pair containing:
 497 
 498       - Output: A `3-D` tensor of shape `[time_len, batch_size, output_size]`
 499         or a list of time_len tensors of shape `[batch_size, output_size]`,
 500         to match the type of the `inputs`.
 501       - Final state: a tuple `(cell_state, output)` matching `initial_state`.
 502 
 503     Raises:
 504       ValueError: in case of shape mismatches
 505     """
 506     is_list = isinstance(inputs, list)
 507     if is_list:
 508       inputs = array_ops.stack(inputs)
 509     inputs_shape = inputs.get_shape().with_rank(3)
 510     if not inputs_shape[2]:
 511       raise ValueError("Expecting inputs_shape[2] to be set: %s" % inputs_shape)
 512     batch_size = inputs_shape[1].value
 513     if batch_size is None:
 514       batch_size = array_ops.shape(inputs)[1]
 515     time_len = inputs_shape[0].value
 516     if time_len is None:
 517       time_len = array_ops.shape(inputs)[0]
 518 
 519     # Provide default values for initial_state and dtype
 520     if initial_state is None:
 521       if dtype is None:
 522         raise ValueError("Either initial_state or dtype needs to be specified")
 523       z = array_ops.zeros(
 524           array_ops.stack([batch_size, self.num_units]), dtype=dtype)
 525       initial_state = z, z
 526     else:
 527       if len(initial_state) != 2:
 528         raise ValueError(
 529             "Expecting initial_state to be a tuple with length 2 or None")
 530       if dtype is None:
 531         dtype = initial_state[0].dtype
 532 
 533     # create the actual cell
 534     if sequence_length is not None:
 535       sequence_length = ops.convert_to_tensor(sequence_length)
 536     initial_cell_state, initial_output = initial_state  # pylint: disable=unpacking-non-sequence
 537     cell_states, outputs = self._call_cell(
 538         inputs, initial_cell_state, initial_output, dtype, sequence_length)
 539 
 540     if sequence_length is not None:
 541       # Mask out the part beyond sequence_length
 542       mask = array_ops.transpose(
 543           array_ops.sequence_mask(sequence_length, time_len, dtype=dtype),
 544           [1, 0])
 545       mask = array_ops.tile(
 546           array_ops.expand_dims(mask, [-1]), [1, 1, self.num_units])
 547       outputs *= mask
 548       # Prepend initial states to cell_states and outputs for indexing to work
 549       # correctly,since we want to access the last valid state at
 550       # sequence_length - 1, which can even be -1, corresponding to the
 551       # initial state.
 552       mod_cell_states = array_ops.concat(
 553           [array_ops.expand_dims(initial_cell_state, [0]), cell_states], 0)
 554       mod_outputs = array_ops.concat(
 555           [array_ops.expand_dims(initial_output, [0]), outputs], 0)
 556       final_cell_state = self._gather_states(mod_cell_states, sequence_length,
 557                                              batch_size)
 558       final_output = self._gather_states(mod_outputs, sequence_length,
 559                                          batch_size)
 560     else:
 561       # No sequence_lengths used: final state is the last state
 562       final_cell_state = cell_states[-1]
 563       final_output = outputs[-1]
 564 
 565     if is_list:
 566       # Input was a list, so return a list
 567       outputs = array_ops.unstack(outputs)
 568 
 569     final_state = rnn_cell_impl.LSTMStateTuple(final_cell_state, final_output)
 570     return outputs, final_state
 571 
 572   def _gather_states(self, data, indices, batch_size):
 573     """Produce `out`, s.t. out(i, j) = data(indices(i), i, j)."""
 574     return array_ops.gather_nd(
 575         data, array_ops.stack([indices, math_ops.range(batch_size)], axis=1))

 578 class LSTMBlockFusedCell(LSTMBlockWrapper):
 579   """FusedRNNCell implementation of LSTM.
 580 
 581   This is an extremely efficient LSTM implementation, that uses a single TF op
 582   for the entire LSTM. It should be both faster and more memory-efficient than
 583   LSTMBlockCell defined above.
 584 
 585   The implementation is based on: http://arxiv.org/abs/1409.2329.
 586 
 587   We add forget_bias (default: 1) to the biases of the forget gate in order to
 588   reduce the scale of forgetting in the beginning of the training.
 589 
 590   The variable naming is consistent with `rnn_cell_impl.LSTMCell`.
 591   """
 592 
 593   def __init__(self,
 594                num_units,
 595                forget_bias=1.0,
 596                cell_clip=None,
 597                use_peephole=False,
 598                reuse=None,
 599                dtype=None,
 600                name="lstm_fused_cell"):
 601     """Initialize the LSTM cell.
 602 
 603     Args:
 604       num_units: int, The number of units in the LSTM cell.
 605       forget_bias: float, The bias added to forget gates (see above).
 606       cell_clip: clip the cell to this value. Default is no cell clipping.
 607       use_peephole: Whether to use peephole connections or not.
 608       reuse: (optional) boolean describing whether to reuse variables in an
 609         existing scope.  If not `True`, and the existing scope already has the
 610         given variables, an error is raised.
 611       dtype: the dtype of variables of this layer.
 612       name: String, the name of the layer. Layers with the same name will
 613         share weights, but to avoid mistakes we require reuse=True in such
 614         cases.  By default this is "lstm_cell", for variable-name compatibility
 615         with `tf.nn.rnn_cell.LSTMCell`.
 616     """
 617     super(LSTMBlockFusedCell, self).__init__(
 618         _reuse=reuse, name=name, dtype=dtype)
 619     self._num_units = num_units
 620     self._forget_bias = forget_bias
 621     self._cell_clip = cell_clip if cell_clip is not None else -1
 622     self._use_peephole = use_peephole
 623 
 624     # Inputs must be 3-dimensional.
 625     self.input_spec = base_layer.InputSpec(ndim=3)
 626 
 627   @property
 628   def num_units(self):
 629     """Number of units in this cell (output dimension)."""
 630     return self._num_units
 631 
 632   def build(self, input_shape):
 633     input_size = input_shape[2].value
 634     self._kernel = self.add_variable(
 635         "kernel", [input_size + self._num_units, self._num_units * 4])
 636     self._bias = self.add_variable(
 637         "bias", [self._num_units * 4],
 638         initializer=init_ops.constant_initializer(0.0))
 639     if self._use_peephole:
 640       self._w_i_diag = self.add_variable("w_i_diag", [self._num_units])
 641       self._w_f_diag = self.add_variable("w_f_diag", [self._num_units])
 642       self._w_o_diag = self.add_variable("w_o_diag", [self._num_units])
 643 
 644     self.built = True
 645 
 646   def _call_cell(self,
 647                  inputs,
 648                  initial_cell_state=None,
 649                  initial_output=None,
 650                  dtype=None,
 651                  sequence_length=None):
 652     """Run this LSTM on inputs, starting from the given state.
 653 
 654     Args:
 655       inputs: `3-D` tensor with shape `[time_len, batch_size, input_size]`
 656       initial_cell_state: initial value for cell state, shape `[batch_size,
 657         self._num_units]`
 658       initial_output: initial value of cell output, shape `[batch_size,
 659         self._num_units]`
 660       dtype: The data type for the initial state and expected output.
 661       sequence_length: Specifies the length of each sequence in inputs. An
 662         `int32` or `int64` vector (tensor) size `[batch_size]`, values in `[0,
 663         time_len)` or None.
 664 
 665     Returns:
 666       A pair containing:
 667 
 668       - Cell state (cs): A `3-D` tensor of shape `[time_len, batch_size,
 669                          output_size]`
 670       - Output (h): A `3-D` tensor of shape `[time_len, batch_size,
 671                     output_size]`
 672     """
 673 
 674     inputs_shape = inputs.get_shape().with_rank(3)
 675     time_len = inputs_shape[0].value
 676     if time_len is None:
 677       time_len = array_ops.shape(inputs)[0]
 678 
 679     if self._use_peephole:
 680       wci = self._w_i_diag
 681       wco = self._w_o_diag
 682       wcf = self._w_f_diag
 683     else:
 684       wci = wcf = wco = array_ops.zeros([self._num_units], dtype=dtype)
 685 
 686     if sequence_length is None:
 687       max_seq_len = math_ops.to_int64(time_len)
 688     else:
 689       max_seq_len = math_ops.to_int64(math_ops.reduce_max(sequence_length))
 690 
 691     _, cs, _, _, _, _, h = gen_lstm_ops.block_lstm(
 692         seq_len_max=max_seq_len,
 693         x=inputs,
 694         cs_prev=initial_cell_state,
 695         h_prev=initial_output,
 696         w=self._kernel,
 697         wci=wci,
 698         wcf=wcf,
 699         wco=wco,
 700         b=self._bias,
 701         forget_bias=self._forget_bias,
 702         cell_clip=self._cell_clip,
 703         use_peephole=self._use_peephole)
 704     return cs, h
