  38 @ops.RegisterGradient("GRUBlockCell")
  39 def _GRUBlockCellGrad(op, *grad):
  40   r"""Gradient for GRUBlockCell.
  41 
  42   Args:
  43     op: Op for which the gradient is defined.
  44     *grad: Gradients of the optimization function wrt output
  45       for the Op.
  46 
  47   Returns:
  48     d_x: Gradients wrt to x
  49     d_h: Gradients wrt to h
  50     d_w_ru: Gradients wrt to w_ru
  51     d_w_c: Gradients wrt to w_c
  52     d_b_ru: Gradients wrt to b_ru
  53     d_b_c: Gradients wrt to b_c
  54 
  55   Mathematics behind the Gradients below:
  56   ```
  57   d_c_bar = d_h \circ (1-u) \circ (1-c \circ c)
  58   d_u_bar = d_h \circ (h-c) \circ u \circ (1-u)
  59 
  60   d_r_bar_u_bar = [d_r_bar d_u_bar]
  61 
  62   [d_x_component_1 d_h_prev_component_1] = d_r_bar_u_bar * w_ru^T
  63 
  64   [d_x_component_2 d_h_prevr] = d_c_bar * w_c^T
  65 
  66   d_x = d_x_component_1 + d_x_component_2
  67 
  68   d_h_prev = d_h_prev_component_1 + d_h_prevr \circ r + u
  69   ```
  70   Below calculation is performed in the python wrapper for the Gradients
  71   (not in the gradient kernel.)
  72   ```
  73   d_w_ru = x_h_prevr^T * d_c_bar
  74 
  75   d_w_c = x_h_prev^T * d_r_bar_u_bar
  76 
  77   d_b_ru = sum of d_r_bar_u_bar along axis = 0
  78 
  79   d_b_c = sum of d_c_bar along axis = 0
  80   ```
  81   """
  82   x, h_prev, w_ru, w_c, b_ru, b_c = op.inputs
  83   r, u, c, _ = op.outputs
  84   _, _, _, d_h = grad
  85 
  86   d_x, d_h_prev, d_c_bar, d_r_bar_u_bar = gen_gru_ops.gru_block_cell_grad(
  87       x, h_prev, w_ru, w_c, b_ru, b_c, r, u, c, d_h)
  88 
  89   x_h_prev = array_ops.concat([x, h_prev], 1)
  90   d_w_ru = math_ops.matmul(x_h_prev, d_r_bar_u_bar, transpose_a=True)
  91   d_b_ru = nn_ops.bias_add_grad(d_r_bar_u_bar)
  92 
  93   x_h_prevr = array_ops.concat([x, h_prev * r], 1)
  94   d_w_c = math_ops.matmul(x_h_prevr, d_c_bar, transpose_a=True)
  95   d_b_c = nn_ops.bias_add_grad(d_c_bar)
  96 
  97   return d_x, d_h_prev, d_w_ru, d_w_c, d_b_ru, d_b_c

 100 class GRUBlockCell(LayerRNNCell):
 101   r"""Block GRU cell implementation.
 102 
 103   Deprecated: use GRUBlockCellV2 instead.
 104 
 105   The implementation is based on:  http://arxiv.org/abs/1406.1078
 106   Computes the GRU cell forward propagation for 1 time step.
 107 
 108   This kernel op implements the following mathematical equations:
 109 
 110   Biases are initialized with:
 111 
 112   * `b_ru` - constant_initializer(1.0)
 113   * `b_c` - constant_initializer(0.0)
 114 
 115   ```
 116   x_h_prev = [x, h_prev]
 117 
 118   [r_bar u_bar] = x_h_prev * w_ru + b_ru
 119 
 120   r = sigmoid(r_bar)
 121   u = sigmoid(u_bar)
 122 
 123   h_prevr = h_prev \circ r
 124 
 125   x_h_prevr = [x h_prevr]
 126 
 127   c_bar = x_h_prevr * w_c + b_c
 128   c = tanh(c_bar)
 129 
 130   h = (1-u) \circ c + u \circ h_prev
 131   ```
 132 
 133   """
 134 
 135   @deprecated_args(None, "cell_size is deprecated, use num_units instead",
 136                    "cell_size")
 137   def __init__(self,
 138                num_units=None,
 139                cell_size=None,
 140                reuse=None,
 141                name="gru_cell"):
 142     """Initialize the Block GRU cell.
 143 
 144     Args:
 145       num_units: int, The number of units in the GRU cell.
 146       cell_size: int, The old (deprecated) name for `num_units`.
 147       reuse: (optional) boolean describing whether to reuse variables in an
 148         existing scope.  If not `True`, and the existing scope already has the
 149         given variables, an error is raised.
 150       name: String, the name of the layer. Layers with the same name will
 151         share weights, but to avoid mistakes we require reuse=True in such
 152         cases.  By default this is "lstm_cell", for variable-name compatibility
 153         with `tf.nn.rnn_cell.GRUCell`.
 154 
 155     Raises:
 156       ValueError: if both cell_size and num_units are not None;
 157         or both are None.
 158     """
 159     super(GRUBlockCell, self).__init__(_reuse=reuse, name=name)
 160     if (cell_size is None) == (num_units is None):
 161       raise ValueError(
 162           "Exactly one of num_units or cell_size must be provided.")
 163     if num_units is None:
 164       num_units = cell_size
 165     self._cell_size = num_units
 166     # Inputs must be 2-dimensional.
 167     self.input_spec = base_layer.InputSpec(ndim=2)
 168 
 169   @property
 170   def state_size(self):
 171     return self._cell_size
 172 
 173   @property
 174   def output_size(self):
 175     return self._cell_size
 176 
 177   def build(self, input_shape):
 178     # Check if the input size exist.
 179     input_size = input_shape[1].value
 180     if input_size is None:
 181       raise ValueError("Expecting input_size to be set.")
 182 
 183     self._gate_kernel = self.add_variable(
 184         "w_ru", [input_size + self._cell_size, self._cell_size * 2])
 185     self._gate_bias = self.add_variable(
 186         "b_ru", [self._cell_size * 2],
 187         initializer=init_ops.constant_initializer(1.0))
 188     self._candidate_kernel = self.add_variable(
 189         "w_c", [input_size + self._cell_size, self._cell_size])
 190     self._candidate_bias = self.add_variable(
 191         "b_c", [self._cell_size],
 192         initializer=init_ops.constant_initializer(0.0))
 193 
 194     self.built = True
 195 
 196   def call(self, inputs, h_prev):
 197     """GRU cell."""
 198     # Check cell_size == state_size from h_prev.
 199     cell_size = h_prev.get_shape().with_rank(2)[1]
 200     if cell_size != self._cell_size:
 201       raise ValueError("Shape of h_prev[1] incorrect: cell_size %i vs %s" %
 202                        (self._cell_size, cell_size))
 203 
 204     _gru_block_cell = gen_gru_ops.gru_block_cell  # pylint: disable=invalid-name
 205     _, _, _, new_h = _gru_block_cell(
 206         x=inputs,
 207         h_prev=h_prev,
 208         w_ru=self._gate_kernel,
 209         w_c=self._candidate_kernel,
 210         b_ru=self._gate_bias,
 211         b_c=self._candidate_bias)
 212 
 213     return new_h, new_h

 216 class GRUBlockCellV2(GRUBlockCell):
 217   """Temporary GRUBlockCell impl with a different variable naming scheme.
 218 
 219   Only differs from GRUBlockCell by variable names.
 220   """
 221 
 222   def build(self, input_shape):
 223     """GRU cell."""
 224     input_size = input_shape[1].value
 225     if input_size is None:
 226       raise ValueError("Expecting input_size to be set.")
 227 
 228     self._gate_kernel = self.add_variable(
 229         "gates/kernel", [input_size + self._cell_size, self._cell_size * 2])
 230     self._gate_bias = self.add_variable(
 231         "gates/bias", [self._cell_size * 2],
 232         initializer=init_ops.constant_initializer(1.0))
 233     self._candidate_kernel = self.add_variable(
 234         "candidate/kernel", [input_size + self._cell_size, self._cell_size])
 235     self._candidate_bias = self.add_variable(
 236         "candidate/bias", [self._cell_size],
 237         initializer=init_ops.constant_initializer(0.0))
 238 
 239     self.built = True
