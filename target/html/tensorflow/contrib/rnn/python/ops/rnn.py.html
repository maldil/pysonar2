  25 def stack_bidirectional_rnn(cells_fw,
  26                             cells_bw,
  27                             inputs,
  28                             initial_states_fw=None,
  29                             initial_states_bw=None,
  30                             dtype=None,
  31                             sequence_length=None,
  32                             scope=None):
  33   """Creates a bidirectional recurrent neural network.
  34 
  35   Stacks several bidirectional rnn layers. The combined forward and backward
  36   layer outputs are used as input of the next layer. tf.bidirectional_rnn
  37   does not allow to share forward and backward information between layers.
  38   The input_size of the first forward and backward cells must match.
  39   The initial state for both directions is zero and no intermediate states
  40   are returned.
  41 
  42   As described in https://arxiv.org/abs/1303.5778
  43 
  44   Args:
  45     cells_fw: List of instances of RNNCell, one per layer,
  46       to be used for forward direction.
  47     cells_bw: List of instances of RNNCell, one per layer,
  48       to be used for backward direction.
  49     inputs: A length T list of inputs, each a tensor of shape
  50       [batch_size, input_size], or a nested tuple of such elements.
  51     initial_states_fw: (optional) A list of the initial states (one per layer)
  52       for the forward RNN.
  53       Each tensor must has an appropriate type and shape
  54       `[batch_size, cell_fw.state_size]`.
  55     initial_states_bw: (optional) Same as for `initial_states_fw`, but using
  56       the corresponding properties of `cells_bw`.
  57     dtype: (optional) The data type for the initial state.  Required if
  58       either of the initial states are not provided.
  59     sequence_length: (optional) An int32/int64 vector, size `[batch_size]`,
  60       containing the actual lengths for each of the sequences.
  61     scope: VariableScope for the created subgraph; defaults to None.
  62 
  63   Returns:
  64     A tuple (outputs, output_state_fw, output_state_bw) where:
  65       outputs is a length `T` list of outputs (one for each input), which
  66         are depth-concatenated forward and backward outputs.
  67       output_states_fw is the final states, one tensor per layer,
  68         of the forward rnn.
  69       output_states_bw is the final states, one tensor per layer,
  70         of the backward rnn.
  71 
  72   Raises:
  73     TypeError: If `cell_fw` or `cell_bw` is not an instance of `RNNCell`.
  74     ValueError: If inputs is None, not a list or an empty list.
  75   """
  76   if not cells_fw:
  77     raise ValueError("Must specify at least one fw cell for BidirectionalRNN.")
  78   if not cells_bw:
  79     raise ValueError("Must specify at least one bw cell for BidirectionalRNN.")
  80   if not isinstance(cells_fw, list):
  81     raise ValueError("cells_fw must be a list of RNNCells (one per layer).")
  82   if not isinstance(cells_bw, list):
  83     raise ValueError("cells_bw must be a list of RNNCells (one per layer).")
  84   if len(cells_fw) != len(cells_bw):
  85     raise ValueError("Forward and Backward cells must have the same depth.")
  86   if (initial_states_fw is not None and
  87       (not isinstance(initial_states_fw, list) or
  88        len(initial_states_fw) != len(cells_fw))):
  89     raise ValueError(
  90         "initial_states_fw must be a list of state tensors (one per layer).")
  91   if (initial_states_bw is not None and
  92       (not isinstance(initial_states_bw, list) or
  93        len(initial_states_bw) != len(cells_bw))):
  94     raise ValueError(
  95         "initial_states_bw must be a list of state tensors (one per layer).")
  96   states_fw = []
  97   states_bw = []
  98   prev_layer = inputs
  99 
 100   with vs.variable_scope(scope or "stack_bidirectional_rnn"):
 101     for i, (cell_fw, cell_bw) in enumerate(zip(cells_fw, cells_bw)):
 102       initial_state_fw = None
 103       initial_state_bw = None
 104       if initial_states_fw:
 105         initial_state_fw = initial_states_fw[i]
 106       if initial_states_bw:
 107         initial_state_bw = initial_states_bw[i]
 108 
 109       with vs.variable_scope("cell_%d" % i) as cell_scope:
 110         prev_layer, state_fw, state_bw = rnn.static_bidirectional_rnn(
 111             cell_fw,
 112             cell_bw,
 113             prev_layer,
 114             initial_state_fw=initial_state_fw,
 115             initial_state_bw=initial_state_bw,
 116             sequence_length=sequence_length,
 117             dtype=dtype,
 118             scope=cell_scope)
 119       states_fw.append(state_fw)
 120       states_bw.append(state_bw)
 121 
 122   return prev_layer, tuple(states_fw), tuple(states_bw)

 125 def stack_bidirectional_dynamic_rnn(cells_fw,
 126                                     cells_bw,
 127                                     inputs,
 128                                     initial_states_fw=None,
 129                                     initial_states_bw=None,
 130                                     dtype=None,
 131                                     sequence_length=None,
 132                                     parallel_iterations=None,
 133                                     time_major=False,
 134                                     scope=None):
 135   """Creates a dynamic bidirectional recurrent neural network.
 136 
 137   Stacks several bidirectional rnn layers. The combined forward and backward
 138   layer outputs are used as input of the next layer. tf.bidirectional_rnn
 139   does not allow to share forward and backward information between layers.
 140   The input_size of the first forward and backward cells must match.
 141   The initial state for both directions is zero and no intermediate states
 142   are returned.
 143 
 144   Args:
 145     cells_fw: List of instances of RNNCell, one per layer,
 146       to be used for forward direction.
 147     cells_bw: List of instances of RNNCell, one per layer,
 148       to be used for backward direction.
 149     inputs: The RNN inputs. this must be a tensor of shape:
 150       `[batch_size, max_time, ...]`, or a nested tuple of such elements.
 151     initial_states_fw: (optional) A list of the initial states (one per layer)
 152       for the forward RNN.
 153       Each tensor must has an appropriate type and shape
 154       `[batch_size, cell_fw.state_size]`.
 155     initial_states_bw: (optional) Same as for `initial_states_fw`, but using
 156       the corresponding properties of `cells_bw`.
 157     dtype: (optional) The data type for the initial state.  Required if
 158       either of the initial states are not provided.
 159     sequence_length: (optional) An int32/int64 vector, size `[batch_size]`,
 160       containing the actual lengths for each of the sequences.
 161     parallel_iterations: (Default: 32).  The number of iterations to run in
 162       parallel.  Those operations which do not have any temporal dependency
 163       and can be run in parallel, will be.  This parameter trades off
 164       time for space.  Values >> 1 use more memory but take less time,
 165       while smaller values use less memory but computations take longer.
 166     time_major: The shape format of the inputs and outputs Tensors. If true,
 167       these Tensors must be shaped [max_time, batch_size, depth]. If false,
 168       these Tensors must be shaped [batch_size, max_time, depth]. Using
 169       time_major = True is a bit more efficient because it avoids transposes at
 170       the beginning and end of the RNN calculation. However, most TensorFlow
 171       data is batch-major, so by default this function accepts input and emits
 172       output in batch-major form.
 173     scope: VariableScope for the created subgraph; defaults to None.
 174 
 175   Returns:
 176     A tuple (outputs, output_state_fw, output_state_bw) where:
 177       outputs: Output `Tensor` shaped:
 178         `[batch_size, max_time, layers_output]`. Where layers_output
 179         are depth-concatenated forward and backward outputs.
 180       output_states_fw is the final states, one tensor per layer,
 181         of the forward rnn.
 182       output_states_bw is the final states, one tensor per layer,
 183         of the backward rnn.
 184 
 185   Raises:
 186     TypeError: If `cell_fw` or `cell_bw` is not an instance of `RNNCell`.
 187     ValueError: If inputs is `None`.
 188   """
 189   if not cells_fw:
 190     raise ValueError("Must specify at least one fw cell for BidirectionalRNN.")
 191   if not cells_bw:
 192     raise ValueError("Must specify at least one bw cell for BidirectionalRNN.")
 193   if not isinstance(cells_fw, list):
 194     raise ValueError("cells_fw must be a list of RNNCells (one per layer).")
 195   if not isinstance(cells_bw, list):
 196     raise ValueError("cells_bw must be a list of RNNCells (one per layer).")
 197   if len(cells_fw) != len(cells_bw):
 198     raise ValueError("Forward and Backward cells must have the same depth.")
 199   if (initial_states_fw is not None and
 200       (not isinstance(initial_states_fw, list) or
 201        len(initial_states_fw) != len(cells_fw))):
 202     raise ValueError(
 203         "initial_states_fw must be a list of state tensors (one per layer).")
 204   if (initial_states_bw is not None and
 205       (not isinstance(initial_states_bw, list) or
 206        len(initial_states_bw) != len(cells_bw))):
 207     raise ValueError(
 208         "initial_states_bw must be a list of state tensors (one per layer).")
 209 
 210   states_fw = []
 211   states_bw = []
 212   prev_layer = inputs
 213 
 214   with vs.variable_scope(scope or "stack_bidirectional_rnn"):
 215     for i, (cell_fw, cell_bw) in enumerate(zip(cells_fw, cells_bw)):
 216       initial_state_fw = None
 217       initial_state_bw = None
 218       if initial_states_fw:
 219         initial_state_fw = initial_states_fw[i]
 220       if initial_states_bw:
 221         initial_state_bw = initial_states_bw[i]
 222 
 223       with vs.variable_scope("cell_%d" % i):
 224         outputs, (state_fw, state_bw) = rnn.bidirectional_dynamic_rnn(
 225             cell_fw,
 226             cell_bw,
 227             prev_layer,
 228             initial_state_fw=initial_state_fw,
 229             initial_state_bw=initial_state_bw,
 230             sequence_length=sequence_length,
 231             parallel_iterations=parallel_iterations,
 232             dtype=dtype,
 233             time_major=time_major)
 234         # Concat the outputs to create the new input.
 235         prev_layer = array_ops.concat(outputs, 2)
 236       states_fw.append(state_fw)
 237       states_bw.append(state_bw)
 238 
 239   return prev_layer, tuple(states_fw), tuple(states_bw)
