  33 def conjugate_gradient(operator,
  34                        rhs,
  35                        preconditioner=None,
  36                        x=None,
  37                        tol=1e-4,
  38                        max_iter=20,
  39                        name="conjugate_gradient"):
  40   r"""Conjugate gradient solver.
  41 
  42   Solves a linear system of equations `A*x = rhs` for selfadjoint, positive
  43   definite matrix `A` and right-hand side vector `rhs`, using an iterative,
  44   matrix-free algorithm where the action of the matrix A is represented by
  45   `operator`. The iteration terminates when either the number of iterations
  46   exceeds `max_iter` or when the residual norm has been reduced to `tol`
  47   times its initial value, i.e. \\(||rhs - A x_k|| <= tol ||rhs||\\).
  48 
  49   Args:
  50     operator: An object representing a linear operator with attributes:
  51       - shape: Either a list of integers or a 1-D `Tensor` of type `int32` of
  52         length 2. `shape[0]` is the dimension on the domain of the operator,
  53         `shape[1]` is the dimension of the co-domain of the operator. On other
  54         words, if operator represents an N x N matrix A, `shape` must contain
  55         `[N, N]`.
  56       - dtype: The datatype of input to and output from `apply`.
  57       - apply: Callable object taking a vector `x` as input and returning a
  58         vector with the result of applying the operator to `x`, i.e. if
  59        `operator` represents matrix `A`, `apply` should return `A * x`.
  60     rhs: A rank-1 `Tensor` of shape `[N]` containing the right-hand size vector.
  61     preconditioner: An object representing a linear operator, see `operator`
  62       for detail. The preconditioner should approximate the inverse of `A`.
  63       An efficient preconditioner could dramatically improve the rate of
  64       convergence. If `preconditioner` represents matrix `M`(`M` approximates
  65       `A^{-1}`), the algorithm uses `preconditioner.apply(x)` to estimate
  66       `A^{-1}x`. For this to be useful, the cost of applying `M` should be
  67       much lower than computing `A^{-1}` directly.
  68     x: A rank-1 `Tensor` of shape `[N]` containing the initial guess for the
  69       solution.
  70     tol: A float scalar convergence tolerance.
  71     max_iter: An integer giving the maximum number of iterations.
  72     name: A name scope for the operation.
  73 
  74   Returns:
  75     output: A namedtuple representing the final state with fields:
  76       - i: A scalar `int32` `Tensor`. Number of iterations executed.
  77       - x: A rank-1 `Tensor` of shape `[N]` containing the computed solution.
  78       - r: A rank-1 `Tensor` of shape `[M]` containing the residual vector.
  79       - p: A rank-1 `Tensor` of shape `[N]`. `A`-conjugate basis vector.
  80       - gamma: \\(r \dot M \dot r\\), equivalent to  \\(||r||_2^2\\) when
  81         `preconditioner=None`.
  82   """
  83   # ephemeral class holding CG state.
  84   cg_state = collections.namedtuple("CGState", ["i", "x", "r", "p", "gamma"])
  85 
  86   def stopping_criterion(i, state):
  87     return math_ops.logical_and(i < max_iter, linalg_ops.norm(state.r) > tol)
  88 
  89   def cg_step(i, state):  # pylint: disable=missing-docstring
  90     z = operator.apply(state.p)
  91     alpha = state.gamma / util.dot(state.p, z)
  92     x = state.x + alpha * state.p
  93     r = state.r - alpha * z
  94     if preconditioner is None:
  95       gamma = util.dot(r, r)
  96       beta = gamma / state.gamma
  97       p = r + beta * state.p
  98     else:
  99       q = preconditioner.apply(r)
 100       gamma = util.dot(r, q)
 101       beta = gamma / state.gamma
 102       p = q + beta * state.p
 103     return i + 1, cg_state(i + 1, x, r, p, gamma)
 104 
 105   with ops.name_scope(name):
 106     n = operator.shape[1:]
 107     rhs = array_ops.expand_dims(rhs, -1)
 108     if x is None:
 109       x = array_ops.expand_dims(
 110           array_ops.zeros(n, dtype=rhs.dtype.base_dtype), -1)
 111       r0 = rhs
 112     else:
 113       x = array_ops.expand_dims(x, -1)
 114       r0 = rhs - operator.apply(x)
 115     if preconditioner is None:
 116       p0 = r0
 117     else:
 118       p0 = preconditioner.apply(r0)
 119     gamma0 = util.dot(r0, p0)
 120     tol *= linalg_ops.norm(r0)
 121     i = constant_op.constant(0, dtype=dtypes.int32)
 122     state = cg_state(i=i, x=x, r=r0, p=p0, gamma=gamma0)
 123     _, state = control_flow_ops.while_loop(stopping_criterion, cg_step,
 124                                            [i, state])
 125     return cg_state(
 126         state.i,
 127         x=array_ops.squeeze(state.x),
 128         r=array_ops.squeeze(state.r),
 129         p=array_ops.squeeze(state.p),
 130         gamma=state.gamma)
