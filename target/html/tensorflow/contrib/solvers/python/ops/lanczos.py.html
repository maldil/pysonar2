  36 def lanczos_bidiag(operator,
  37                    k,
  38                    orthogonalize=True,
  39                    starting_vector=None,
  40                    name="lanczos_bidiag"):
  41   """Computes a Lanczos bidiagonalization for a linear operator.
  42 
  43   Computes matrices `U` of shape `[m, k+1]`, `V` of shape `[n, k]` and lower
  44   bidiagonal matrix `B` of shape `[k+1, k]`, that satisfy the equations
  45   `A * V = U * B` and `A' * U[:, :-1] = V * B[:-1, :]'`.
  46 
  47   The columns of `U` are orthonormal and form a basis for the Krylov subspace
  48   `K(A*A', U[:,0])`.
  49 
  50   The columns of `V` are orthonormal and form a basis for the Krylov subspace
  51   `K(A'*A, A' U[:,0])`.
  52 
  53   Args:
  54     operator: An object representing a linear operator with attributes:
  55       - shape: Either a list of integers or a 1-D `Tensor` of type `int32` of
  56         length 2. `shape[0]` is the dimension on the domain of the operator,
  57         `shape[1]` is the dimension of the co-domain of the operator. On other
  58         words, if operator represents an M x N matrix A, `shape` must contain
  59         `[M, N]`.
  60       - dtype: The datatype of input to and output from `apply` and
  61         `apply_adjoint`.
  62       - apply: Callable object taking a vector `x` as input and returning a
  63         vector with the result of applying the operator to `x`, i.e. if
  64        `operator` represents matrix `A`, `apply` should return `A * x`.
  65       - apply_adjoint: Callable object taking a vector `x` as input and
  66         returning a vector with the result of applying the adjoint operator
  67         to `x`, i.e. if `operator` represents matrix `A`, `apply_adjoint` should
  68         return `conj(transpose(A)) * x`.
  69     k: An integer or a scalar Tensor of type `int32`. Determines the maximum
  70       number of steps to run. If an invariant subspace is found, the algorithm
  71       may terminate before `k` steps have been run.
  72     orthogonalize: If `True`, perform full orthogonalization. If `False` no
  73       orthogonalization is performed.
  74     starting_vector: If not null, must be a `Tensor` of shape `[n]`.
  75     name: A name scope for the operation.
  76 
  77   Returns:
  78     output: A namedtuple representing a Lanczos bidiagonalization of
  79       `operator` with attributes:
  80       u: A rank-2 `Tensor` of type `operator.dtype` and shape
  81         `[operator.shape[0], k_actual+1]`, where `k_actual` is the number of
  82         steps run.
  83       v: A rank-2 `Tensor` of type `operator.dtype` and shape
  84         `[operator.shape[1], k_actual]`, where `k_actual` is the number of steps
  85         run.
  86       alpha: A rank-1 `Tensor` of type `operator.dtype` and shape `[k]`.
  87       beta: A rank-1 `Tensor` of type `operator.dtype` and shape `[k]`.
  88   """
  89 
  90   def tarray(size, dtype, name):
  91     return tensor_array_ops.TensorArray(
  92         dtype=dtype, size=size, tensor_array_name=name, clear_after_read=False)
  93 
  94   # Reads a row-vector at location i in tarray and returns it as a
  95   # column-vector.
  96   def read_colvec(tarray, i):
  97     return array_ops.expand_dims(tarray.read(i), -1)
  98 
  99   # Writes an column-vector as a row-vecor at location i in tarray.
 100   def write_colvec(tarray, colvec, i):
 101     return tarray.write(i, array_ops.squeeze(colvec))
 102 
 103   # Ephemeral class holding Lanczos bidiagonalization state:
 104   #   u = left Lanczos vectors
 105   #   v = right Lanczos vectors
 106   #   alpha = diagonal of B_k.
 107   #   beta = subdiagonal of B_k.
 108   # Notice that we store the left and right Lanczos vectors as the _rows_
 109   # of u and v. This is done because tensors are stored row-major and
 110   # TensorArray only supports packing along dimension 0.
 111   lanzcos_bidiag_state = collections.namedtuple("LanczosBidiagState",
 112                                                 ["u", "v", "alpha", "beta"])
 113 
 114   def update_state(old, i, u, v, alpha, beta):
 115     return lanzcos_bidiag_state(
 116         write_colvec(old.u, u, i + 1),
 117         write_colvec(old.v, v, i),
 118         old.alpha.write(i, alpha), old.beta.write(i, beta))
 119 
 120   def gram_schmidt_step(j, basis, v):
 121     """Makes v orthogonal to the j'th vector in basis."""
 122     v_shape = v.get_shape()
 123     basis_vec = read_colvec(basis, j)
 124     v -= math_ops.matmul(basis_vec, v, adjoint_a=True) * basis_vec
 125     v.set_shape(v_shape)
 126     return j + 1, basis, v
 127 
 128   def orthogonalize_once(i, basis, v):
 129     j = constant_op.constant(0, dtype=dtypes.int32)
 130     _, _, v = control_flow_ops.while_loop(lambda j, basis, v: j < i,
 131                                           gram_schmidt_step, [j, basis, v])
 132     return util.l2normalize(v)
 133 
 134   # Iterated modified Gram-Schmidt orthogonalization adapted from PROPACK.
 135   # TODO(rmlarsen): This is possibly the slowest implementation of
 136   # iterated Gram-Schmidt orthogonalization since the abacus. Move to C++.
 137   def orthogonalize_(i, basis, v):
 138     v_norm = util.l2norm(v)
 139     v_new, v_new_norm = orthogonalize_once(i, basis, v)
 140     # If the norm decreases more than 1/sqrt(2), run a second
 141     # round of MGS. See proof in:
 142     #   B. N. Parlett, ``The Symmetric Eigenvalue Problem'',
 143     #   Prentice-Hall, Englewood Cliffs, NJ, 1980. pp. 105-109
 144     return control_flow_ops.cond(v_new_norm < 0.7071 * v_norm,
 145                                  lambda: orthogonalize_once(i, basis, v),
 146                                  lambda: (v_new, v_new_norm))
 147 
 148   def stopping_criterion(i, _):
 149     # TODO(rmlarsen): Stop if an invariant subspace is detected.
 150     return i < k
 151 
 152   def lanczos_bidiag_step(i, ls):
 153     """Extends the Lanczos bidiagonalization ls by one step."""
 154     u = read_colvec(ls.u, i)
 155     r = operator.apply_adjoint(u)
 156     # The shape inference doesn't work across cond, save and reapply the shape.
 157     r_shape = r.get_shape()
 158     r = control_flow_ops.cond(
 159         i > 0, lambda: r - ls.beta.read(i - 1) * read_colvec(ls.v, i - 1),
 160         lambda: r)
 161     r.set_shape(r_shape)
 162     if orthogonalize:
 163       v, alpha = orthogonalize_(i - 1, ls.v, r)
 164     else:
 165       v, alpha = util.l2normalize(r)
 166     p = operator.apply(v) - alpha * u
 167     if orthogonalize:
 168       u, beta = orthogonalize_(i, ls.u, p)
 169     else:
 170       u, beta = util.l2normalize(p)
 171 
 172     return i + 1, update_state(ls, i, u, v, alpha, beta)
 173 
 174   with ops.name_scope(name):
 175     dtype = operator.dtype
 176     if starting_vector is None:
 177       starting_vector = random_ops.random_uniform(
 178           operator.shape[:1], -1, 1, dtype=dtype)
 179     u0, _ = util.l2normalize(starting_vector)
 180     ls = lanzcos_bidiag_state(
 181         u=write_colvec(tarray(k + 1, dtype, "u"), u0, 0),
 182         v=tarray(k, dtype, "v"),
 183         alpha=tarray(k, dtype, "alpha"),
 184         beta=tarray(k, dtype, "beta"))
 185     i = constant_op.constant(0, dtype=dtypes.int32)
 186     _, ls = control_flow_ops.while_loop(stopping_criterion, lanczos_bidiag_step,
 187                                         [i, ls])
 188     return lanzcos_bidiag_state(
 189         array_ops.matrix_transpose(ls.u.stack()),
 190         array_ops.matrix_transpose(ls.v.stack()),
 191         ls.alpha.stack(), ls.beta.stack())

 200 def bidiag_matmul(matrix, alpha, beta, adjoint_b=False, name="bidiag_matmul"):
 201   """Multiplies a matrix by a bidiagonal matrix.
 202 
 203   alpha and beta are length k vectors representing the diagonal and first lower
 204   subdiagonal of (K+1) x K matrix B.
 205   If adjoint_b is False, computes A * B as follows:
 206 
 207     A * B =  A[:, :-1] * diag(alpha) + A[:, 1:] * diag(beta)
 208 
 209   If adjoint_b is True, computes A * B[:-1, :]' as follows
 210 
 211     A * B[:-1, :]' =
 212       A * diag(alpha) + [zeros(m,1), A[:, :-1] * diag(beta[:-1])]
 213 
 214   Args:
 215     matrix: A rank-2 `Tensor` representing matrix A.
 216     alpha: A rank-1 `Tensor` representing the diagonal of B.
 217     beta: A rank-1 `Tensor` representing the lower subdiagonal diagonal of B.
 218     adjoint_b: `bool` determining what to compute.
 219     name: A name scope for the operation.
 220 
 221   Returns:
 222     If `adjoint_b` is False the `A * B` is returned.
 223     If `adjoint_b` is True the `A * B'` is returned.
 224   """
 225   with ops.name_scope(name):
 226     alpha = array_ops.expand_dims(alpha, 0)
 227     if adjoint_b is False:
 228       beta = array_ops.expand_dims(beta, 0)
 229       return matrix[:, :-1] * alpha + matrix[:, 1:] * beta
 230     else:
 231       beta = array_ops.expand_dims(beta[:-1], 0)
 232       shape = array_ops.shape(matrix)
 233       zero_column = array_ops.expand_dims(
 234           array_ops.zeros(
 235               shape[:1], dtype=matrix.dtype), 1)
 236       return matrix * alpha + array_ops.concat(
 237           [zero_column, matrix[:, :-1] * beta], 1)
