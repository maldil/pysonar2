  16 """Categorical vocabulary classes to map categories to indexes (deprecated).
  17 
  18 This module and all its submodules are deprecated. See
  19 [contrib/learn/README.md](https://www.tensorflow.org/code/tensorflow/contrib/learn/README.md)
  20 for migration instructions.
  21 
  22 Can be used for categorical variables, sparse variables and words.
  23 """
  24 
  25 from __future__ import absolute_import
  26 from __future__ import division
  27 from __future__ import print_function
  28 
  29 import collections
  30 import six
  31 
  32 from tensorflow.python.util.deprecation import deprecated
  33 
  34 
  35 class CategoricalVocabulary(object):
  36   """Categorical variables vocabulary class.
  37 
  38   THIS CLASS IS DEPRECATED. See
  39   [contrib/learn/README.md](https://www.tensorflow.org/code/tensorflow/contrib/learn/README.md)
  40   for general migration instructions.
  41 
  42   Accumulates and provides mapping from classes to indexes.
  43   Can be easily used for words.
  44   """
  45 
  46   @deprecated(None, 'Please use tensorflow/transform or tf.data.')
  47   def __init__(self, unknown_token="<UNK>", support_reverse=True):
  48     self._unknown_token = unknown_token
  49     self._mapping = {unknown_token: 0}
  50     self._support_reverse = support_reverse
  51     if support_reverse:
  52       self._reverse_mapping = [unknown_token]
  53     self._freq = collections.defaultdict(int)
  54     self._freeze = False
  55 
  56   def __len__(self):
  57     """Returns total count of mappings. Including unknown token."""
  58     return len(self._mapping)
  59 
  60   def freeze(self, freeze=True):
  61     """Freezes the vocabulary, after which new words return unknown token id.
  62 
  63     Args:
  64       freeze: True to freeze, False to unfreeze.
  65     """
  66     self._freeze = freeze
  67 
  68   def get(self, category):
  69     """Returns word's id in the vocabulary.
  70 
  71     If category is new, creates a new id for it.
  72 
  73     Args:
  74       category: string or integer to lookup in vocabulary.
  75 
  76     Returns:
  77       interger, id in the vocabulary.
  78     """
  79     if category not in self._mapping:
  80       if self._freeze:
  81         return 0
  82       self._mapping[category] = len(self._mapping)
  83       if self._support_reverse:
  84         self._reverse_mapping.append(category)
  85     return self._mapping[category]
  86 
  87   def add(self, category, count=1):
  88     """Adds count of the category to the frequency table.
  89 
  90     Args:
  91       category: string or integer, category to add frequency to.
  92       count: optional integer, how many to add.
  93     """
  94     category_id = self.get(category)
  95     if category_id <= 0:
  96       return
  97     self._freq[category] += count
  98 
  99   def trim(self, min_frequency, max_frequency=-1):
 100     """Trims vocabulary for minimum frequency.
 101 
 102     Remaps ids from 1..n in sort frequency order.
 103     where n - number of elements left.
 104 
 105     Args:
 106       min_frequency: minimum frequency to keep.
 107       max_frequency: optional, maximum frequency to keep.
 108         Useful to remove very frequent categories (like stop words).
 109     """
 110     # Sort by alphabet then reversed frequency.
 111     self._freq = sorted(
 112         sorted(
 113             six.iteritems(self._freq),
 114             key=lambda x: (isinstance(x[0], str), x[0])),
 115         key=lambda x: x[1],
 116         reverse=True)
 117     self._mapping = {self._unknown_token: 0}
 118     if self._support_reverse:
 119       self._reverse_mapping = [self._unknown_token]
 120     idx = 1
 121     for category, count in self._freq:
 122       if max_frequency > 0 and count >= max_frequency:
 123         continue
 124       if count <= min_frequency:
 125         break
 126       self._mapping[category] = idx
 127       idx += 1
 128       if self._support_reverse:
 129         self._reverse_mapping.append(category)
 130     self._freq = dict(self._freq[:idx - 1])
 131 
 132   def reverse(self, class_id):
 133     """Given class id reverse to original class name.
 134 
 135     Args:
 136       class_id: Id of the class.
 137 
 138     Returns:
 139       Class name.
 140 
 141     Raises:
 142       ValueError: if this vocabulary wasn't initialized with support_reverse.
 143     """
 144     if not self._support_reverse:
 145       raise ValueError("This vocabulary wasn't initialized with "
 146                        "support_reverse to support reverse() function.")
 147     return self._reverse_mapping[class_id]
