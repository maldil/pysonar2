  16

  19 for migration instructions.
  20 """

  22 from __future__ import absolute_import
  23 from __future__ import division
  24 from __future__ import print_function
  25 
  26 from collections import Container
  27 from types import FunctionType
  28 from types import GeneratorType
  29 
  30 from tensorflow.python.estimator.inputs.queues.feeding_functions import _enqueue_data as enqueue_data
  31 from tensorflow.python.util.deprecation import deprecated
  32 
  33 
  34 @deprecated(None, 'Please use tf.data.')
  35 def generator_input_fn(x,
  36                        target_key=None,
  37                        batch_size=128,
  38                        num_epochs=1,
  39                        shuffle=True,
  40                        queue_capacity=1000,
  41                        num_threads=1,
  42                        pad_value=None):
  43   """Returns input function that returns dicts of numpy arrays
  44      yielded from a generator.
  45 
  46   It is assumed that every dict of numpy arrays yielded from the dictionary
  47   represents a single sample. The generator should consume a single epoch of the
  48   data.
  49 
  50   This returns a function outputting `features` and `target` based on the dict
  51   of numpy arrays. The dict `features` has the same keys as an element yielded
  52   from x.
  53 
  54   Example:
  55     ```python
  56     def generator():
  57       for index in range(10):
  58         yield {'height': np.random.randint(32,36),
  59               'age': np.random.randint(18, 80),
  60               'label': np.ones(1)}
  61 
  62     with tf.Session() as session:
  63       input_fn = generator_io.generator_input_fn(
  64           generator, target_key="label", batch_size=2, shuffle=False,
  65           num_epochs=1)
  66     ```
  67 
  68   Args:
  69     x: Generator Function, returns a `Generator` that will yield the data
  70       in `dict` of numpy arrays
  71     target_key: String or Container of Strings, the key or Container of keys of
  72       the numpy arrays in x dictionaries to use as target.
  73     batch_size: Integer, size of batches to return.
  74     num_epochs: Integer, number of epochs to iterate over data. If `None` will
  75       run forever.
  76     shuffle: Boolean, if True shuffles the queue. Avoid shuffle at prediction
  77       time.
  78     queue_capacity: Integer, size of queue to accumulate.
  79     num_threads: Integer, number of threads used for reading and enqueueing.
  80     pad_value: default value for dynamic padding of data samples, if provided.
  81 
  82   Returns:
  83     Function, that returns a feature `dict` with `Tensors` and an optional
  84      label `dict` with `Tensors`, or if target_key is `str` label is a `Tensor`
  85 
  86   Raises:
  87     TypeError: `x` is not `FunctionType`.
  88     TypeError: `x()` is not `GeneratorType`.
  89     TypeError: `next(x())` is not `dict`.
  90     TypeError: `target_key` is not `str` or `target_key` is not `Container`
  91        of `str`.
  92     KeyError:  `target_key` not a key or `target_key[index]` not in next(`x()`).
  93     KeyError: `key` mismatch between dicts emitted from `x()`
  94   """
  95   if not isinstance(x, FunctionType):
  96     raise TypeError(
  97         'x must be generator function; got {}'.format(type(x).__name__))
  98   generator = x()
  99   if not isinstance(generator, GeneratorType):
 100     raise TypeError(
 101         'x() must be generator; got {}'.format(type(generator).__name__))
 102   data = next(generator)
 103   if not isinstance(data, dict):
 104     raise TypeError('x() must yield dict; got {}'.format(type(data).__name__))
 105   input_keys = sorted(next(x()).keys())
 106   if target_key is not None:
 107     if isinstance(target_key, str):
 108       target_key = [target_key]
 109     elif isinstance(target_key, Container):
 110       for item in target_key:
 111         if not isinstance(item, str):
 112           raise TypeError('target_key must be str or Container of str; got {}'.
 113                           format(type(item).__name__))
 114         if item not in input_keys:
 115           raise KeyError(
 116               'target_key not in yielded dict. Expected {} keys; got {}'.format(
 117                   input_keys, item))
 118     else:
 119       raise TypeError('target_key must be str or Container of str; got {}'.
 120                       format(type(target_key).__name__))
 121 
 122   def _generator_input_fn():
 123     """generator input function."""
 124     queue = enqueue_data(
 125         x,
 126         queue_capacity,
 127         shuffle=shuffle,
 128         num_threads=num_threads,
 129         enqueue_size=batch_size,
 130         num_epochs=num_epochs,
 131         pad_value=pad_value)
 132 
 133     features = (queue.dequeue_many(batch_size)
 134                 if num_epochs is None else queue.dequeue_up_to(batch_size))
 135     if not isinstance(features, list):
 136       features = [features]
 137     features = dict(zip(input_keys, features))
 138     if target_key is not None:
 139       if len(target_key) > 1:
 140         target = {key: features.pop(key) for key in target_key}
 141       else:
 142         target = features.pop(target_key[0])
 143       return features, target
 144     return features
 145 
 146   return _generator_input_fn
