  16 r"""System for specifying garbage collection (GC) of path based data (deprecated).
  17 
  18 This module and all its submodules are deprecated. See
  19 [contrib/learn/README.md](https://www.tensorflow.org/code/tensorflow/contrib/learn/README.md)
  20 for migration instructions.
  21 
  22 This framework allows for GC of data specified by path names, for example files
  23 on disk.  gc.Path objects each represent a single item stored at a path and may
  24 be a base directory,
  25   /tmp/exports/0/...
  26   /tmp/exports/1/...
  27   ...
  28 or a fully qualified file,
  29   /tmp/train-1.ckpt
  30   /tmp/train-2.ckpt
  31   ...
  32 
  33 A gc filter function takes and returns a list of gc.Path items.  Filter
  34 functions are responsible for selecting Path items for preservation or deletion.
  35 Note that functions should always return a sorted list.
  36 
  37 For example,
  38   base_dir = "/tmp"
  39   # Create the directories.
  40   for e in xrange(10):
  41     os.mkdir("%s/%d" % (base_dir, e), 0o755)
  42 
  43   # Create a simple parser that pulls the export_version from the directory.
  44   path_regex = "^" + re.escape(base_dir) + "/(\\d+)$"
  45   def parser(path):
  46     match = re.match(path_regex, path.path)
  47     if not match:
  48       return None
  49     return path._replace(export_version=int(match.group(1)))
  50 
  51   path_list = gc.get_paths("/tmp", parser)  # contains all ten Paths
  52 
  53   every_fifth = gc.mod_export_version(5)
  54   print(every_fifth(path_list))  # shows ["/tmp/0", "/tmp/5"]
  55 
  56   largest_three = gc.largest_export_versions(3)
  57   print(largest_three(all_paths))  # shows ["/tmp/7", "/tmp/8", "/tmp/9"]
  58 
  59   both = gc.union(every_fifth, largest_three)
  60   print(both(all_paths))  # shows ["/tmp/0", "/tmp/5",
  61                           #        "/tmp/7", "/tmp/8", "/tmp/9"]
  62   # Delete everything not in 'both'.
  63   to_delete = gc.negation(both)
  64   for p in to_delete(all_paths):
  65     gfile.DeleteRecursively(p.path)  # deletes:  "/tmp/1", "/tmp/2",
  66                                      # "/tmp/3", "/tmp/4", "/tmp/6",
  67 """

  85 @deprecated(None, 'Please implement your own file management or use Saver.')
  86 def largest_export_versions(n):
  87   """Creates a filter that keeps the largest n export versions.
  88 
  89   Args:
  90     n: number of versions to keep.
  91 
  92   Returns:
  93     A filter function that keeps the n largest paths.
  94   """
  95   def keep(paths):
  96     heap = []
  97     for idx, path in enumerate(paths):
  98       if path.export_version is not None:
  99         heapq.heappush(heap, (path.export_version, idx))
 100     keepers = [paths[i] for _, i in heapq.nlargest(n, heap)]
 101     return sorted(keepers)
 102 
 103   return keep

 106 @deprecated(None, 'Please implement your own file management or use Saver.')
 107 def one_of_every_n_export_versions(n):
 108   """Creates a filter that keeps one of every n export versions.
 109 
 110   Args:
 111     n: interval size.
 112 
 113   Returns:
 114     A filter function that keeps exactly one path from each interval
 115     [0, n], (n, 2n], (2n, 3n], etc...  If more than one path exists in an
 116     interval the largest is kept.
 117   """
 118   def keep(paths):
 119     """A filter function that keeps exactly one out of every n paths."""
 120 
 121     keeper_map = {}  # map from interval to largest path seen in that interval
 122     for p in paths:
 123       if p.export_version is None:
 124         # Skip missing export_versions.
 125         continue
 126       # Find the interval (with a special case to map export_version = 0 to
 127       # interval 0.
 128       interval = math.floor(
 129           (p.export_version - 1) / n) if p.export_version else 0
 130       existing = keeper_map.get(interval, None)
 131       if (not existing) or (existing.export_version < p.export_version):
 132         keeper_map[interval] = p
 133     return sorted(keeper_map.values())
 134 
 135   return keep

 138 @deprecated(None, 'Please implement your own file management or use Saver.')
 139 def mod_export_version(n):
 140   """Creates a filter that keeps every export that is a multiple of n.
 141 
 142   Args:
 143     n: step size.
 144 
 145   Returns:
 146     A filter function that keeps paths where export_version % n == 0.
 147   """
 148   def keep(paths):
 149     keepers = []
 150     for p in paths:
 151       if p.export_version % n == 0:
 152         keepers.append(p)
 153     return sorted(keepers)
 154   return keep

 157 @deprecated(None, 'Please implement your own file management or use Saver.')
 158 def union(lf, rf):
 159   """Creates a filter that keeps the union of two filters.
 160 
 161   Args:
 162     lf: first filter
 163     rf: second filter
 164 
 165   Returns:
 166     A filter function that keeps the n largest paths.
 167   """
 168   def keep(paths):
 169     l = set(lf(paths))
 170     r = set(rf(paths))
 171     return sorted(list(l|r))
 172   return keep

 175 @deprecated(None, 'Please implement your own file management or use Saver.')
 176 def negation(f):
 177   """Negate a filter.
 178 
 179   Args:
 180     f: filter function to invert
 181 
 182   Returns:
 183     A filter function that returns the negation of f.
 184   """
 185   def keep(paths):
 186     l = set(paths)
 187     r = set(f(paths))
 188     return sorted(list(l-r))
 189   return keep

 192 @deprecated(None, 'Please implement your own file name management.')
 193 def get_paths(base_dir, parser):
 194   """Gets a list of Paths in a given directory.
 195 
 196   Args:
 197     base_dir: directory.
 198     parser: a function which gets the raw Path and can augment it with
 199       information such as the export_version, or ignore the path by returning
 200       None.  An example parser may extract the export version from a path
 201       such as "/tmp/exports/100" an another may extract from a full file
 202       name such as "/tmp/checkpoint-99.out".
 203 
 204   Returns:
 205     A list of Paths contained in the base directory with the parsing function
 206     applied.
 207     By default the following fields are populated,
 208       - Path.path
 209     The parsing function is responsible for populating,
 210       - Path.export_version
 211   """
 212   raw_paths = gfile.ListDirectory(base_dir)
 213   paths = []
 214   for r in raw_paths:
 215     p = parser(Path(os.path.join(compat.as_str_any(base_dir),
 216                                  compat.as_str_any(r)),
 217                     None))
 218     if p:
 219       paths.append(p)
 220   return sorted(paths)
