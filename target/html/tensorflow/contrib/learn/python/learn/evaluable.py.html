  29 class Evaluable(object):
  30   """Interface for objects that are evaluatable by, e.g., `Experiment`.
  31 
  32   THIS CLASS IS DEPRECATED. See
  33   [contrib/learn/README.md](https://www.tensorflow.org/code/tensorflow/contrib/learn/README.md)
  34   for general migration instructions.
  35   """
  36   __metaclass__ = abc.ABCMeta
  37 
  38   @abc.abstractproperty
  39   def model_dir(self):
  40     """Returns a path in which the eval process will look for checkpoints."""
  41     raise NotImplementedError
  42 
  43   @abc.abstractmethod
  44   def evaluate(self,
  45                x=None,
  46                y=None,
  47                input_fn=None,
  48                feed_fn=None,
  49                batch_size=None,
  50                steps=None,
  51                metrics=None,
  52                name=None,
  53                checkpoint_path=None,
  54                hooks=None):
  55     """Evaluates given model with provided evaluation data.
  56 
  57     Stop conditions - we evaluate on the given input data until one of the
  58     following:
  59     - If `steps` is provided, and `steps` batches of size `batch_size` are
  60     processed.
  61     - If `input_fn` is provided, and it raises an end-of-input
  62     exception (`OutOfRangeError` or `StopIteration`).
  63     - If `x` is provided, and all items in `x` have been processed.
  64 
  65     The return value is a dict containing the metrics specified in `metrics`, as
  66     well as an entry `global_step` which contains the value of the global step
  67     for which this evaluation was performed.
  68 
  69     Args:
  70       x: Matrix of shape [n_samples, n_features...] or dictionary of many
  71         matrices
  72         containing the input samples for fitting the model. Can be iterator that
  73           returns
  74         arrays of features or dictionary of array of features. If set,
  75           `input_fn` must
  76         be `None`.
  77       y: Vector or matrix [n_samples] or [n_samples, n_outputs] containing the
  78         label values (class labels in classification, real numbers in
  79         regression) or dictionary of multiple vectors/matrices. Can be iterator
  80         that returns array of targets or dictionary of array of targets. If set,
  81         `input_fn` must be `None`. Note: For classification, label values must
  82         be integers representing the class index (i.e. values from 0 to
  83         n_classes-1).
  84       input_fn: Input function returning a tuple of:
  85         features - Dictionary of string feature name to `Tensor` or `Tensor`.
  86         labels - `Tensor` or dictionary of `Tensor` with labels.
  87         If input_fn is set, `x`, `y`, and `batch_size` must be `None`. If
  88         `steps` is not provided, this should raise `OutOfRangeError` or
  89         `StopIteration` after the desired amount of data (e.g., one epoch) has
  90         been provided. See "Stop conditions" above for specifics.
  91       feed_fn: Function creating a feed dict every time it is called. Called
  92         once per iteration. Must be `None` if `input_fn` is provided.
  93       batch_size: minibatch size to use on the input, defaults to first
  94         dimension of `x`, if specified. Must be `None` if `input_fn` is
  95         provided.
  96       steps: Number of steps for which to evaluate model. If `None`, evaluate
  97         until `x` is consumed or `input_fn` raises an end-of-input exception.
  98         See "Stop conditions" above for specifics.
  99       metrics: Dict of metrics to run. If None, the default metric functions
 100         are used; if {}, no metrics are used. Otherwise, `metrics` should map
 101         friendly names for the metric to a `MetricSpec` object defining which
 102         model outputs to evaluate against which labels with which metric
 103         function.
 104         Metric ops should support streaming, e.g., returning `update_op` and
 105         `value` tensors. For example, see the options defined in
 106         `../../../metrics/python/ops/metrics_ops.py`.
 107       name: Name of the evaluation if user needs to run multiple evaluations on
 108         different data sets, such as on training data vs test data.
 109       checkpoint_path: Path of a specific checkpoint to evaluate. If `None`, the
 110         latest checkpoint in `model_dir` is used.
 111       hooks: List of `SessionRunHook` subclass instances. Used for callbacks
 112         inside the evaluation call.
 113 
 114     Returns:
 115       Returns `dict` with evaluation results.
 116     """
 117     raise NotImplementedError
