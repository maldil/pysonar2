  34 def _execute_schedule(experiment, schedule):
  35   """Execute the method named `schedule` of `experiment`."""
  36   if not hasattr(experiment, schedule):
  37     logging.error('Schedule references non-existent task %s', schedule)
  38     valid_tasks = [x for x in dir(experiment)
  39                    if not x.startswith('_')
  40                    and callable(getattr(experiment, x))]
  41     logging.error('Allowed values for this experiment are: %s', valid_tasks)
  42     raise ValueError('Schedule references non-existent task %s' % schedule)
  43 
  44   task = getattr(experiment, schedule)
  45   if not callable(task):
  46     logging.error('Schedule references non-callable member %s', schedule)
  47     valid_tasks = [x for x in dir(experiment)
  48                    if not x.startswith('_')
  49                    and callable(getattr(experiment, x))]
  50     logging.error('Allowed values for this experiment are: %s', valid_tasks)
  51     raise TypeError('Schedule references non-callable member %s' % schedule)
  52   return task()

  55 def _wrapped_experiment_fn_with_uid_check(experiment_fn, require_hparams=False):
  56   """Wraps the `RunConfig` uid check with `experiment_fn`.
  57 
  58   For `experiment_fn` which takes `run_config`, it is expected that the
  59   `run_config` is passed to the Estimator correctly. Toward that, the wrapped
  60   `experiment_fn` compares the `uid` of the `RunConfig` instance.
  61 
  62   Args:
  63     experiment_fn: The original `experiment_fn` which takes `run_config` and
  64       `hparams`.
  65     require_hparams: If True, the `hparams` passed to `experiment_fn` cannot be
  66       `None`.
  67 
  68   Returns:
  69     A experiment_fn with same signature.
  70   """
  71   def wrapped_experiment_fn(run_config, hparams):
  72     """Calls experiment_fn and checks the uid of `RunConfig`."""
  73     if not isinstance(run_config, run_config_lib.RunConfig):
  74       raise ValueError(
  75           '`run_config` must be `tf.contrib.learn.RunConfig` instance')
  76     if not run_config.model_dir:
  77       raise ValueError(
  78           'Must specify a model directory `model_dir` in `run_config`.')
  79     if hparams is not None and not isinstance(hparams, hparam_lib.HParams):
  80       raise ValueError('`hparams` must be `HParams` instance')
  81     if require_hparams and hparams is None:
  82       raise ValueError('`hparams` cannot be `None`.')
  83 
  84     expected_uid = run_config.uid()
  85     experiment = experiment_fn(run_config, hparams)
  86 
  87     if not isinstance(experiment, Experiment):
  88       raise TypeError('Experiment builder did not return an Experiment '
  89                       'instance, got %s instead.' % type(experiment))
  90 
  91     config_from_estimator = experiment.estimator.config
  92     if not hasattr(config_from_estimator, 'uid'):
  93       raise RuntimeError(
  94           'Pass `run_config` argument of the `experiment_fn` to the Estimator '
  95           'in Experiment. It is likely a different `RunConfig` is passed to '
  96           '`Estimator` or the `config` constructor argument in `Estimator` '
  97           'is not set.')
  98 
  99     if config_from_estimator.uid() != expected_uid:
 100       raise RuntimeError(
 101           '`RunConfig` instance is expected to be used by the `Estimator` '
 102           'inside the `Experiment`. expected {}, but got {}'.format(
 103               expected_uid, experiment.estimator.config.uid()))
 104     return experiment
 105   return wrapped_experiment_fn

 108 @deprecated(None, 'Use tf.estimator.train_and_evaluate.')
 109 def run(experiment_fn, output_dir=None, schedule=None, run_config=None,
 110         hparams=None):
 111   """Make and run an experiment.
 112 
 113   It creates an Experiment by calling `experiment_fn`. Then it calls the
 114   function named as `schedule` of the Experiment.
 115 
 116   If schedule is not provided, then the default schedule for the current task
 117   type is used. The defaults are as follows:
 118 
 119    * 'ps' maps to 'serve'
 120    * 'worker' maps to 'train'
 121    * 'master' maps to 'local_run'
 122 
 123   If the experiment's config does not include a task type, then an exception
 124   is raised.
 125 
 126   Example with `run_config` (Recommended):
 127   ```
 128     def _create_my_experiment(run_config, hparams):
 129 
 130         # You can change a subset of the run_config properties as
 131         #   run_config = run_config.replace(save_checkpoints_steps=500)
 132 
 133         return tf.contrib.learn.Experiment(
 134           estimator=my_estimator(config=run_config, hparams=hparams),
 135           train_input_fn=my_train_input,
 136           eval_input_fn=my_eval_input)
 137 
 138     learn_runner.run(
 139       experiment_fn=_create_my_experiment,
 140       run_config=run_config_lib.RunConfig(model_dir="some/output/dir"),
 141       schedule="train_and_evaluate",
 142       hparams=_create_default_hparams())
 143   ```
 144   or simply as
 145   ```
 146     learn_runner.run(
 147       experiment_fn=_create_my_experiment,
 148       run_config=run_config_lib.RunConfig(model_dir="some/output/dir"))
 149   ```
 150   if `hparams` is not used by the `Estimator`. On a single machine, `schedule`
 151   defaults to `train_and_evaluate`.
 152 
 153   Example with `output_dir` (deprecated):
 154   ```
 155     def _create_my_experiment(output_dir):
 156         return tf.contrib.learn.Experiment(
 157           estimator=my_estimator(model_dir=output_dir),
 158           train_input_fn=my_train_input,
 159           eval_input_fn=my_eval_input)
 160 
 161     learn_runner.run(
 162       experiment_fn=_create_my_experiment,
 163       output_dir="some/output/dir",
 164       schedule="train")
 165   ```
 166   Args:
 167     experiment_fn: A function that creates an `Experiment`. It could be one of
 168       the two following signatures:
 169       1) [Deprecated] It accepts an argument `output_dir` which should be used
 170       to create the `Estimator` (passed as `model_dir` to its constructor). It
 171       must return an `Experiment`. For this case, `run_config` and `hparams`
 172       must be None.
 173       2) It accepts two arguments `run_config` and `hparams`, which should be
 174       used to create the `Estimator` (`run_config` passed as `config` to its
 175       constructor; `hparams` used as the hyper-parameters of the model).
 176       It must return an `Experiment`. For this case, `output_dir` must be None.
 177     output_dir: Base output directory [Deprecated].
 178     schedule: The name of the method in the `Experiment` to run.
 179     run_config: `RunConfig` instance. The `run_config.model_dir` must be
 180       non-empty. If `run_config` is set, `output_dir` must be None.
 181     hparams: `HParams` instance. The default hyper-parameters, which will be
 182       passed to the `experiment_fn` if `run_config` is not None.
 183 
 184   Returns:
 185     The return value of function `schedule`.
 186 
 187   Raises:
 188     ValueError: If both `output_dir` and `run_config` are empty or set,
 189       `schedule` is None but no task type is set in the built experiment's
 190       config, the task type has no default, `run_config.model_dir` is empty or
 191       `schedule` doesn't reference a member of `Experiment`.
 192     TypeError: `schedule` references non-callable member.
 193   """
 194 
 195   if output_dir is not None and run_config is not None:
 196     raise ValueError('Cannot provide both `output_dir` and `run_config`')
 197 
 198   if output_dir is None and run_config is None:
 199     raise ValueError('Must set value for `output_dir` or `run_config`')
 200 
 201   if not callable(experiment_fn):
 202     raise TypeError('Experiment builder "%s" is not callable.' %
 203                     experiment_fn)
 204 
 205   experiment = None
 206   if run_config is not None:
 207     wrapped_experiment_fn = _wrapped_experiment_fn_with_uid_check(experiment_fn)
 208     experiment = wrapped_experiment_fn(run_config=run_config, hparams=hparams)
 209   else:
 210     if not output_dir:
 211       raise ValueError('Must specify an output directory')
 212     if hparams is not None:
 213       raise ValueError(
 214           'Must set `hparams` as None for `experiment_fn` with `output_dir`.')
 215     # Call the builder
 216     experiment = experiment_fn(output_dir=output_dir)
 217     if not isinstance(experiment, Experiment):
 218       raise TypeError('Experiment builder did not return an Experiment '
 219                       'instance, got %s instead.' % type(experiment))
 220 
 221   # Get the schedule
 222   run_config = run_config or experiment.estimator.config
 223   schedule = schedule or _get_default_schedule(run_config)
 224 
 225   return _execute_schedule(experiment, schedule)

 228 @deprecated(None, 'Use tf.estimator.train_and_evaluate.')
 229 def tune(experiment_fn, tuner):
 230   """Tune an experiment with hyper-parameters.
 231 
 232   It iterates trials by running the Experiment for each trial with the
 233   corresponding hyper-parameters. For each trial, it retrieves the
 234   hyper-parameters from `tuner`, creates an Experiment by calling experiment_fn,
 235   and then reports the measure back to `tuner`.
 236 
 237   Example:
 238   ```
 239     def _create_my_experiment(run_config, hparams):
 240       hidden_units = [hparams.unit_per_layer] * hparams.num_hidden_layers
 241 
 242       return tf.contrib.learn.Experiment(
 243           estimator=DNNClassifier(config=run_config, hidden_units=hidden_units),
 244           train_input_fn=my_train_input,
 245           eval_input_fn=my_eval_input)
 246 
 247     tuner = create_tuner(study_configuration, objective_key)
 248 
 249     learn_runner.tune(experiment_fn=_create_my_experiment, tuner)
 250   ```
 251   Args:
 252     experiment_fn: A function that creates an `Experiment`. It should accept an
 253       argument `run_config` which should be used to create the `Estimator` (
 254       passed as `config` to its constructor), and an argument `hparams`, which
 255       should be used for hyper-parameters tuning. It must return an
 256       `Experiment`.
 257     tuner: A `Tuner` instance.
 258   """
 259   while tuner.next_trial():
 260     tuner.run_experiment(
 261         _wrapped_experiment_fn_with_uid_check(
 262             experiment_fn, require_hparams=True))

 265 def _is_distributed(config):
 266   """Returns true if this is a distributed job."""
 267   if not config.cluster_spec:
 268     return False
 269 
 270   # This is considered a distributed job if there is more than one task
 271   # in the cluster spec.
 272   task_count = 0
 273   for job in config.cluster_spec.jobs:
 274     for _ in config.cluster_spec.job_tasks(job):
 275       task_count += 1
 276 
 277   return task_count > 1

 280 def _get_default_schedule(config):
 281   """Returns the default schedule for the provided RunConfig."""
 282   if not config or not _is_distributed(config):
 283     return 'train_and_evaluate'
 284 
 285   if not config.task_type:
 286     raise ValueError('Must specify a schedule')
 287 
 288   if config.task_type == run_config_lib.TaskType.MASTER:
 289     # TODO(rhaertel): handle the case where there is more than one master
 290     # or explicitly disallow such a case.
 291     return 'train_and_evaluate'
 292   elif config.task_type == run_config_lib.TaskType.PS:
 293     return 'run_std_server'
 294   elif config.task_type == run_config_lib.TaskType.WORKER:
 295     return 'train'
 296 
 297   raise ValueError('No default schedule for task type: %s' % (config.task_type))
