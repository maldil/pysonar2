  37 def _get_model_fn_with_logistic_metrics(model_fn):
  38   """Returns a model_fn with additional logistic metrics.
  39 
  40   Args:
  41     model_fn: Model function with the signature:
  42       `(features, labels, mode) -> (predictions, loss, train_op)`.
  43       Expects the returned predictions to be probabilities in [0.0, 1.0].
  44 
  45   Returns:
  46     model_fn that can be used with Estimator.
  47   """
  48 
  49   def _model_fn(features, labels, mode, params):
  50     """Model function that appends logistic evaluation metrics."""
  51     thresholds = params.get('thresholds') or [.5]
  52 
  53     predictions, loss, train_op = model_fn(features, labels, mode)
  54     if mode == model_fn_lib.ModeKeys.EVAL:
  55       eval_metric_ops = _make_logistic_eval_metric_ops(
  56           labels=labels,
  57           predictions=predictions,
  58           thresholds=thresholds)
  59     else:
  60       eval_metric_ops = None
  61     return model_fn_lib.ModelFnOps(
  62         mode=mode,
  63         predictions=predictions,
  64         loss=loss,
  65         train_op=train_op,
  66         eval_metric_ops=eval_metric_ops,
  67         output_alternatives={
  68             'head': (constants.ProblemType.LOGISTIC_REGRESSION, {
  69                 'predictions': predictions
  70             })
  71         })
  72 
  73   return _model_fn

  77 def LogisticRegressor(  # pylint: disable=invalid-name
  78     model_fn, thresholds=None, model_dir=None, config=None,
  79     feature_engineering_fn=None):
  80   """Builds a logistic regression Estimator for binary classification.
  81 
  82   THIS CLASS IS DEPRECATED. See
  83   [contrib/learn/README.md](https://www.tensorflow.org/code/tensorflow/contrib/learn/README.md)
  84   for general migration instructions.
  85 
  86   This method provides a basic Estimator with some additional metrics for custom
  87   binary classification models, including AUC, precision/recall and accuracy.
  88 
  89   Example:
  90 
  91   ```python
  92     # See tf.contrib.learn.Estimator(...) for details on model_fn structure
  93     def my_model_fn(...):
  94       pass
  95 
  96     estimator = LogisticRegressor(model_fn=my_model_fn)
  97 
  98     # Input builders
  99     def input_fn_train:
 100       pass
 101 
 102     estimator.fit(input_fn=input_fn_train)
 103     estimator.predict(x=x)
 104   ```
 105 
 106   Args:
 107     model_fn: Model function with the signature:
 108       `(features, labels, mode) -> (predictions, loss, train_op)`.
 109       Expects the returned predictions to be probabilities in [0.0, 1.0].
 110     thresholds: List of floating point thresholds to use for accuracy,
 111       precision, and recall metrics. If `None`, defaults to `[0.5]`.
 112     model_dir: Directory to save model parameters, graphs, etc. This can also
 113       be used to load checkpoints from the directory into a estimator to
 114       continue training a previously saved model.
 115     config: A RunConfig configuration object.
 116     feature_engineering_fn: Feature engineering function. Takes features and
 117                       labels which are the output of `input_fn` and
 118                       returns features and labels which will be fed
 119                       into the model.
 120 
 121   Returns:
 122     An `Estimator` instance.
 123   """
 124   return estimator.Estimator(
 125       model_fn=_get_model_fn_with_logistic_metrics(model_fn),
 126       model_dir=model_dir,
 127       config=config,
 128       params={'thresholds': thresholds},
 129       feature_engineering_fn=feature_engineering_fn)

 132 def _make_logistic_eval_metric_ops(labels, predictions, thresholds):
 133   """Returns a dictionary of evaluation metric ops for logistic regression.
 134 
 135   Args:
 136     labels: The labels `Tensor`, or a dict with only one `Tensor` keyed by name.
 137     predictions: The predictions `Tensor`.
 138     thresholds: List of floating point thresholds to use for accuracy,
 139       precision, and recall metrics.
 140 
 141   Returns:
 142     A dict of metric results keyed by name.
 143   """
 144   # If labels is a dict with a single key, unpack into a single tensor.
 145   labels_tensor = labels
 146   if isinstance(labels, dict) and len(labels) == 1:
 147     labels_tensor = labels.values()[0]
 148 
 149   metrics = {}
 150   metrics[metric_key.MetricKey.PREDICTION_MEAN] = metrics_lib.streaming_mean(
 151       predictions)
 152   metrics[metric_key.MetricKey.LABEL_MEAN] = metrics_lib.streaming_mean(
 153       labels_tensor)
 154   # Also include the streaming mean of the label as an accuracy baseline, as
 155   # a reminder to users.
 156   metrics[metric_key.MetricKey.ACCURACY_BASELINE] = metrics_lib.streaming_mean(
 157       labels_tensor)
 158 
 159   metrics[metric_key.MetricKey.AUC] = metrics_lib.streaming_auc(
 160       labels=labels_tensor, predictions=predictions)
 161 
 162   for threshold in thresholds:
 163     predictions_at_threshold = math_ops.to_float(
 164         math_ops.greater_equal(predictions, threshold),
 165         name='predictions_at_threshold_%f' % threshold)
 166     metrics[metric_key.MetricKey.ACCURACY_MEAN % threshold] = (
 167         metrics_lib.streaming_accuracy(labels=labels_tensor,
 168                                        predictions=predictions_at_threshold))
 169     # Precision for positive examples.
 170     metrics[metric_key.MetricKey.PRECISION_MEAN % threshold] = (
 171         metrics_lib.streaming_precision(labels=labels_tensor,
 172                                         predictions=predictions_at_threshold))
 173     # Recall for positive examples.
 174     metrics[metric_key.MetricKey.RECALL_MEAN % threshold] = (
 175         metrics_lib.streaming_recall(labels=labels_tensor,
 176                                      predictions=predictions_at_threshold))
 177 
 178   return metrics
