 186 def _model_fn_args(fn):
 187   """Get argument names for function-like object.
 188 
 189   Args:
 190     fn: Function, or function-like object (e.g., result of `functools.partial`).
 191 
 192   Returns:
 193     `tuple` of string argument names.
 194 
 195   Raises:
 196     ValueError: if partial function has positionally bound arguments
 197   """
 198   _, fn = tf_decorator.unwrap(fn)
 199   if hasattr(fn, 'func') and hasattr(fn, 'keywords') and hasattr(fn, 'args'):
 200     # Handle functools.partial and similar objects.
 201     return tuple([
 202         arg for arg in tf_inspect.getargspec(fn.func).args[len(fn.args):]
 203         if arg not in set(fn.keywords.keys())
 204     ])
 205   # Handle function.
 206   return tuple(tf_inspect.getargspec(fn).args)

 241 def _make_metrics_ops(metrics, features, labels, predictions):
 242   """Add metrics based on `features`, `labels`, and `predictions`.
 243 
 244   `metrics` contains a specification for how to run metrics. It is a dict
 245   mapping friendly names to either `MetricSpec` objects, or directly to a metric
 246   function (assuming that `predictions` and `labels` are single tensors), or to
 247   `(pred_name, metric)` `tuple`, which passes `predictions[pred_name]` and
 248   `labels` to `metric` (assuming `labels` is a single tensor).
 249 
 250   Users are encouraged to use `MetricSpec` objects, which are more flexible and
 251   cleaner. They also lead to clearer errors.
 252 
 253   Args:
 254     metrics: A dict mapping names to metrics specification, for example
 255       `MetricSpec` objects.
 256     features: A dict of tensors returned from an input_fn as features/inputs.
 257     labels: A single tensor or a dict of tensors returned from an input_fn as
 258       labels.
 259     predictions: A single tensor or a dict of tensors output from a model as
 260       predictions.
 261 
 262   Returns:
 263     A dict mapping the friendly given in `metrics` to the result of calling the
 264     given metric function.
 265 
 266   Raises:
 267     ValueError: If metrics specifications do not work with the type of
 268       `features`, `labels`, or `predictions` provided. Mostly, a dict is given
 269       but no pred_name specified.
 270   """
 271   metrics = metrics or {}
 272 
 273   # If labels is a dict with a single key, unpack into a single tensor.
 274   labels_tensor_or_dict = labels
 275   if isinstance(labels, dict) and len(labels) == 1:
 276     labels_tensor_or_dict = labels[list(labels.keys())[0]]
 277 
 278   result = {}
 279   # Iterate in lexicographic order, so the graph is identical among runs.
 280   for name, metric in sorted(six.iteritems(metrics)):
 281     if isinstance(metric, metric_spec.MetricSpec):
 282       result[name] = metric.create_metric_ops(features, labels, predictions)
 283       continue
 284 
 285     # TODO(b/31229024): Remove the rest of this loop
 286     logging.warning('Please specify metrics using MetricSpec. Using bare '
 287                     'functions or (key, fn) tuples is deprecated and support '
 288                     'for it will be removed on Oct 1, 2016.')
 289 
 290     if isinstance(name, tuple):
 291       # Multi-head metrics.
 292       if len(name) != 2:
 293         raise ValueError('Invalid metric for {}. It returned a tuple with '
 294                          'len {}, expected 2.'.format(name, len(name)))
 295       if not isinstance(predictions, dict):
 296         raise ValueError('Metrics passed provide (name, prediction), '
 297                          'but predictions are not dict. '
 298                          'Metrics: %s, Predictions: %s.' % (metrics,
 299                                                             predictions))
 300       # Here are two options: labels are single Tensor or a dict.
 301       if isinstance(labels, dict) and name[1] in labels:
 302         # If labels are dict and the prediction name is in it, apply metric.
 303         result[name[0]] = metric(predictions[name[1]], labels[name[1]])
 304       else:
 305         # Otherwise pass the labels to the metric.
 306         result[name[0]] = metric(predictions[name[1]], labels_tensor_or_dict)
 307     else:
 308       # Single head metrics.
 309       if isinstance(predictions, dict):
 310         raise ValueError('Metrics passed provide only name, no prediction, '
 311                          'but predictions are dict. '
 312                          'Metrics: %s, Labels: %s.' % (metrics,
 313                                                        labels_tensor_or_dict))
 314       result[name] = metric(predictions, labels_tensor_or_dict)
 315   return result

 318 def _dict_to_str(dictionary):
 319   """Get a `str` representation of a `dict`.
 320 
 321   Args:
 322     dictionary: The `dict` to be represented as `str`.
 323 
 324   Returns:
 325     A `str` representing the `dictionary`.
 326   """
 327   results = []
 328   for k, v in sorted(dictionary.items()):
 329     if isinstance(v, float) or isinstance(v, np.float32) or isinstance(
 330         v, int) or isinstance(v, np.int64) or isinstance(v, np.int32):
 331       results.append('%s = %s' % (k, v))
 332     else:
 333       results.append('Type of %s = %s' % (k, type(v)))
 334 
 335   return ', '.join(results)

 338 def _write_dict_to_summary(output_dir, dictionary, current_global_step):
 339   """Writes a `dict` into summary file in given output directory.
 340 
 341   Args:
 342     output_dir: `str`, directory to write the summary file in.
 343     dictionary: the `dict` to be written to summary file.
 344     current_global_step: `int`, the current global step.
 345   """
 346   logging.info('Saving dict for global step %d: %s', current_global_step,
 347                _dict_to_str(dictionary))
 348   summary_writer = core_summary.FileWriterCache.get(output_dir)
 349   summary_proto = summary_pb2.Summary()
 350   for key in dictionary:
 351     if dictionary[key] is None:
 352       continue
 353     if key == 'global_step':
 354       continue
 355     if (isinstance(dictionary[key], np.float32) or
 356         isinstance(dictionary[key], float)):
 357       summary_proto.value.add(tag=key, simple_value=float(dictionary[key]))
 358     elif (isinstance(dictionary[key], np.int64) or
 359           isinstance(dictionary[key], np.int32) or
 360           isinstance(dictionary[key], int)):
 361       summary_proto.value.add(tag=key, simple_value=int(dictionary[key]))
 362     elif isinstance(dictionary[key], six.string_types):
 363       try:
 364         summ = summary_pb2.Summary.FromString(dictionary[key])
 365         for i, _ in enumerate(summ.value):
 366           summ.value[i].tag = key
 367         summary_proto.value.extend(summ.value)
 368       except message.DecodeError:
 369         logging.warn('Skipping summary for %s, cannot parse string to Summary.',
 370                      key)
 371         continue
 372     elif isinstance(dictionary[key], np.ndarray):
 373       value = summary_proto.value.add()
 374       value.tag = key
 375       value.node_name = key
 376       tensor_proto = tensor_util.make_tensor_proto(dictionary[key])
 377       value.tensor.CopyFrom(tensor_proto)
 378       logging.info(
 379           'Summary for np.ndarray is not visible in Tensorboard by default. '
 380           'Consider using a Tensorboard plugin for visualization (see '
 381           'https://github.com/tensorflow/tensorboard-plugin-example/blob/master/README.md'
 382           ' for more information).')
 383     else:
 384       logging.warn(
 385           'Skipping summary for %s, must be a float, np.float32, np.int64, '
 386           'np.int32 or int or np.ndarray or a serialized string of Summary.',
 387           key)
 388   summary_writer.add_summary(summary_proto, current_global_step)
 389   summary_writer.flush()

 879   def _evaluate_model(self,
 880                       input_fn,
 881                       steps,
 882                       feed_fn=None,
 883                       metrics=None,
 884                       name='',
 885                       checkpoint_path=None,
 886                       hooks=None,
 887                       log_progress=True):
 888     # TODO(wicke): Remove this once Model and associated code are gone.
 889     if (hasattr(self._config, 'execution_mode') and
 890         self._config.execution_mode not in ('all', 'evaluate', 'eval_evalset')):
 891       return None, None
 892 
 893     # Check that model has been trained (if nothing has been set explicitly).
 894     if not checkpoint_path:
 895       latest_path = checkpoint_management.latest_checkpoint(self._model_dir)
 896       if not latest_path:
 897         raise NotFittedError(
 898             "Couldn't find trained model at %s." % self._model_dir)
 899       checkpoint_path = latest_path
 900 
 901     # Setup output directory.
 902     eval_dir = os.path.join(self._model_dir, 'eval'
 903                             if not name else 'eval_' + name)
 904 
 905     with ops.Graph().as_default() as g:
 906       random_seed.set_random_seed(self._config.tf_random_seed)
 907       global_step = training_util.create_global_step(g)
 908       features, labels = input_fn()
 909       self._check_inputs(features, labels)
 910 
 911       model_fn_results = self._get_eval_ops(features, labels, metrics)
 912       eval_dict = model_fn_results.eval_metric_ops
 913 
 914       update_op, eval_dict = self._extract_metric_update_ops(eval_dict)
 915 
 916       # We need to copy the hook array as we modify it, thus [:].
 917       hooks = hooks[:] if hooks else []
 918       if feed_fn:
 919         hooks.append(basic_session_run_hooks.FeedFnHook(feed_fn))
 920       if steps == 0:
 921         logging.warning('evaluation steps are 0. If `input_fn` does not raise '
 922                         '`OutOfRangeError`, the evaluation will never stop. '
 923                         'Use steps=None if intended.')
 924       if steps:
 925         hooks.append(
 926             evaluation.StopAfterNEvalsHook(steps, log_progress=log_progress))
 927 
 928       global_step_key = 'global_step'
 929       while global_step_key in eval_dict:
 930         global_step_key = '_' + global_step_key
 931       eval_dict[global_step_key] = global_step
 932 
 933       eval_results = evaluation.evaluate_once(
 934           checkpoint_path=checkpoint_path,
 935           master=self._config.evaluation_master,
 936           scaffold=model_fn_results.scaffold,
 937           eval_ops=update_op,
 938           final_ops=eval_dict,
 939           hooks=hooks,
 940           config=self._session_config)
 941       current_global_step = eval_results[global_step_key]
 942 
 943       _write_dict_to_summary(eval_dir, eval_results, current_global_step)
 944 
 945     return eval_results, current_global_step

 953   def _infer_model(self,
 954                    input_fn,
 955                    feed_fn=None,
 956                    outputs=None,
 957                    as_iterable=True,
 958                    iterate_batches=False):
 959     # Check that model has been trained.
 960     checkpoint_path = checkpoint_management.latest_checkpoint(self._model_dir)
 961     if not checkpoint_path:
 962       raise NotFittedError(
 963           "Couldn't find trained model at %s." % self._model_dir)
 964 
 965     with ops.Graph().as_default() as g:
 966       random_seed.set_random_seed(self._config.tf_random_seed)
 967       training_util.create_global_step(g)
 968       features = self._get_features_from_input_fn(input_fn)
 969       infer_ops = self._get_predict_ops(features)
 970       predictions = self._filter_predictions(infer_ops.predictions, outputs)
 971       mon_sess = monitored_session.MonitoredSession(
 972           session_creator=monitored_session.ChiefSessionCreator(
 973               checkpoint_filename_with_path=checkpoint_path,
 974               scaffold=infer_ops.scaffold,
 975               config=self._session_config))
 976       if not as_iterable:
 977         with mon_sess:
 978           if not mon_sess.should_stop():
 979             return mon_sess.run(predictions, feed_fn() if feed_fn else None)
 980       else:
 981         return self._predict_generator(mon_sess, predictions, feed_fn,
 982                                        iterate_batches)
 983 
 984   def _predict_generator(self, mon_sess, predictions, feed_fn, iterate_batches):
 985     with mon_sess:
 986       while not mon_sess.should_stop():
 987         preds = mon_sess.run(predictions, feed_fn() if feed_fn else None)
 988         if iterate_batches:
 989           yield preds
 990         elif not isinstance(predictions, dict):
 991           for pred in preds:
 992             yield pred
 993         else:
 994           first_tensor = list(preds.values())[0]
 995           if isinstance(first_tensor, sparse_tensor.SparseTensorValue):
 996             batch_length = first_tensor.dense_shape[0]
 997           else:
 998             batch_length = first_tensor.shape[0]
 999           for i in range(batch_length):
1000             yield {key: value[i] for key, value in six.iteritems(preds)}
1001         if self._is_input_constant(feed_fn, mon_sess.graph):
1002           return

1004   def _is_input_constant(self, feed_fn, graph):
1005     # If there are no queue_runners, the input `predictions` is a
1006     # constant, and we should stop after the first epoch.  If,
1007     # instead, there are queue_runners, eventually they should throw
1008     # an `OutOfRangeError`.
1009     if graph.get_collection(ops.GraphKeys.QUEUE_RUNNERS):
1010       return False
1011     # data_feeder uses feed_fn to generate `OutOfRangeError`.
1012     if feed_fn is not None:
1013       return False
1014     return True
1015 
1016   def _filter_predictions(self, predictions, outputs):
1017     if not outputs:
1018       return predictions
1019     if not isinstance(predictions, dict):
1020       raise ValueError(
1021           'outputs argument is not valid in case of non-dict predictions.')
1022     existing_keys = predictions.keys()
1023     predictions = {
1024         key: value
1025         for key, value in six.iteritems(predictions)
1026         if key in outputs
1027     }
1028     if not predictions:
1029       raise ValueError('Expected to run at least one output from %s, '
1030                        'provided %s.' % (existing_keys, outputs))
1031     return predictions

1033   def _train_model(self, input_fn, hooks):
1034     all_hooks = []
1035     self._graph = ops.Graph()
1036     with self._graph.as_default() as g, g.device(self._device_fn):
1037       random_seed.set_random_seed(self._config.tf_random_seed)
1038       global_step = training_util.create_global_step(g)
1039       features, labels = input_fn()
1040       self._check_inputs(features, labels)
1041       training_util._get_or_create_global_step_read()  # pylint: disable=protected-access
1042       model_fn_ops = self._get_train_ops(features, labels)
1043       ops.add_to_collection(ops.GraphKeys.LOSSES, model_fn_ops.loss)
1044       all_hooks.extend(hooks)
1045       all_hooks.extend([
1046           basic_session_run_hooks.NanTensorHook(model_fn_ops.loss),
1047           basic_session_run_hooks.LoggingTensorHook(
1048               {
1049                   'loss': model_fn_ops.loss,
1050                   'step': global_step
1051               },
1052               every_n_iter=100)
1053       ])
1054 
1055       scaffold = model_fn_ops.scaffold or monitored_session.Scaffold()
1056       if not (scaffold.saver or ops.get_collection(ops.GraphKeys.SAVERS)):
1057         ops.add_to_collection(
1058             ops.GraphKeys.SAVERS,
1059             saver.Saver(
1060                 sharded=True,
1061                 max_to_keep=self._config.keep_checkpoint_max,
1062                 keep_checkpoint_every_n_hours=(
1063                     self._config.keep_checkpoint_every_n_hours),
1064                 defer_build=True,
1065                 save_relative_paths=True))
1066 
1067       chief_hooks = []
1068       if (self._config.save_checkpoints_secs or
1069           self._config.save_checkpoints_steps):
1070         saver_hook_exists = any([
1071             isinstance(h, basic_session_run_hooks.CheckpointSaverHook)
1072             for h in (all_hooks + model_fn_ops.training_hooks + chief_hooks +
1073                       model_fn_ops.training_chief_hooks)
1074         ])
1075         if not saver_hook_exists:
1076           chief_hooks = [
1077               basic_session_run_hooks.CheckpointSaverHook(
1078                   self._model_dir,
1079                   save_secs=self._config.save_checkpoints_secs,
1080                   save_steps=self._config.save_checkpoints_steps,
1081                   scaffold=scaffold)
1082           ]
1083       with monitored_session.MonitoredTrainingSession(
1084           master=self._config.master,
1085           is_chief=self._config.is_chief,
1086           checkpoint_dir=self._model_dir,
1087           scaffold=scaffold,
1088           hooks=all_hooks + model_fn_ops.training_hooks,
1089           chief_only_hooks=chief_hooks + model_fn_ops.training_chief_hooks,
1090           save_checkpoint_secs=0,  # Saving is handled by a hook.
1091           save_summaries_steps=self._config.save_summary_steps,
1092           config=self._session_config) as mon_sess:
1093         loss = None
1094         while not mon_sess.should_stop():
1095           _, loss = mon_sess.run([model_fn_ops.train_op, model_fn_ops.loss])
1096       return loss
