  36   @abc.abstractmethod
  37   def fit(self,
  38           x=None,
  39           y=None,
  40           input_fn=None,
  41           steps=None,
  42           batch_size=None,
  43           monitors=None,
  44           max_steps=None):
  45     """Trains a model given training data `x` predictions and `y` labels.
  46 
  47     Args:
  48       x: Matrix of shape [n_samples, n_features...] or the dictionary of
  49         Matrices.
  50          Can be iterator that returns arrays of features or dictionary of arrays
  51            of features.
  52          The training input samples for fitting the model. If set, `input_fn`
  53            must be `None`.
  54       y: Vector or matrix [n_samples] or [n_samples, n_outputs] or the
  55         dictionary of same.
  56          Can be iterator that returns array of labels or dictionary of array of
  57            labels.
  58          The training label values (class labels in classification, real numbers
  59            in regression).
  60          If set, `input_fn` must be `None`. Note: For classification, label
  61            values must
  62          be integers representing the class index (i.e. values from 0 to
  63          n_classes-1).
  64       input_fn: Input function returning a tuple of:
  65           features - `Tensor` or dictionary of string feature name to `Tensor`.
  66           labels - `Tensor` or dictionary of `Tensor` with labels.
  67         If input_fn is set, `x`, `y`, and `batch_size` must be `None`.
  68       steps: Number of steps for which to train model. If `None`, train forever.
  69         'steps' works incrementally. If you call two times fit(steps=10) then
  70         training occurs in total 20 steps. If you don't want to have incremental
  71         behavior please set `max_steps` instead. If set, `max_steps` must be
  72         `None`.
  73       batch_size: minibatch size to use on the input, defaults to first
  74         dimension of `x`. Must be `None` if `input_fn` is provided.
  75       monitors: List of `BaseMonitor` subclass instances. Used for callbacks
  76         inside the training loop.
  77       max_steps: Number of total steps for which to train model. If `None`,
  78         train forever. If set, `steps` must be `None`.
  79 
  80         Two calls to `fit(steps=100)` means 200 training
  81         iterations. On the other hand, two calls to `fit(max_steps=100)` means
  82         that the second call will not do any iteration since first call did
  83         all 100 steps.
  84 
  85     Returns:
  86       `self`, for chaining.
  87     """
