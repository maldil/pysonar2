  41 # List of all available datasets.
  42 # Note, currently they may return different types.
  43 DATASETS = {
  44     # Returns base.Dataset.
  45     'iris': base.load_iris,

  47     # Returns base.Datasets (train/validation/test sets).

  60 @deprecated(None, 'Please use tf.data.')
  61 def load_dataset(name, size='small', test_with_fake_data=False):
  62   """Loads dataset by name.
  63 
  64   Args:
  65     name: Name of the dataset to load.
  66     size: Size of the dataset to load.
  67     test_with_fake_data: If true, load with fake dataset.
  68 
  69   Returns:
  70     Features and labels for given dataset. Can be numpy or iterator.
  71 
  72   Raises:
  73     ValueError: if `name` is not found.
  74   """
  75   if name not in DATASETS:
  76     raise ValueError('Name of dataset is not found: %s' % name)
  77   if name == 'dbpedia':
  78     return DATASETS[name](size, test_with_fake_data)
  79   else:
  80     return DATASETS[name]()

  83 @deprecated(None, 'Please use tf.data.')
  84 def make_dataset(name, n_samples=100, noise=None, seed=42, *args, **kwargs):
  85   """Creates binary synthetic datasets.
  86 
  87   Args:
  88     name: str, name of the dataset to generate
  89     n_samples: int, number of datapoints to generate
  90     noise: float or None, standard deviation of the Gaussian noise added
  91     seed: int or None, seed for noise
  92 
  93   Returns:
  94     Shuffled features and labels for given synthetic dataset of type
  95     `base.Dataset`
  96 
  97   Raises:
  98     ValueError: Raised if `name` not found
  99 
 100   Note:
 101     - This is a generic synthetic data generator - individual generators might
 102     have more parameters!
 103       See documentation for individual parameters
 104     - Note that the `noise` parameter uses `numpy.random.normal` and depends on
 105     `numpy`'s seed
 106 
 107   TODO:
 108     - Support multiclass datasets
 109     - Need shuffling routine. Currently synthetic datasets are reshuffled to
 110     avoid train/test correlation,
 111       but that hurts reprodusability
 112   """
 113   # seed = kwargs.pop('seed', None)
 114   if name not in SYNTHETIC:
 115     raise ValueError('Synthetic dataset not found or not implemeted: %s' % name)
 116   else:
 117     return SYNTHETIC[name](
 118         n_samples=n_samples, noise=noise, seed=seed, *args, **kwargs)
