  32 @deprecated(None, 'Consider using synthetic datasets from scikits.learn.')
  33 def circles(n_samples=100,
  34             noise=None,
  35             seed=None,
  36             factor=0.8,
  37             n_classes=2,
  38             *args,
  39             **kwargs):
  40   """Create circles separated by some value
  41 
  42   Args:
  43     n_samples: int, number of datapoints to generate
  44     noise: float or None, standard deviation of the Gaussian noise added
  45     seed: int or None, seed for the noise
  46     factor: float, size factor of the inner circles with respect to the outer
  47       ones
  48     n_classes: int, number of classes to generate
  49 
  50   Returns:
  51     Shuffled features and labels for 'circles' synthetic dataset of type
  52     `base.Dataset`
  53 
  54   Note:
  55     The multi-class support might not work as expected if `noise` is enabled
  56 
  57   TODO:
  58     - Generation of unbalanced data
  59 
  60   Credit goes to (under BSD 3 clause):
  61     B. Thirion,
  62     G. Varoquaux,
  63     A. Gramfort,
  64     V. Michel,
  65     O. Grisel,
  66     G. Louppe,
  67     J. Nothman
  68   """
  69   if seed is not None:
  70     np.random.seed(seed)
  71   # Algo: 1) Generate initial circle, 2) For ever class generate a smaller radius circle
  72   linspace = np.linspace(0, 2 * np.pi, n_samples // n_classes)
  73   circ_x = np.empty(0, dtype=np.int32)
  74   circ_y = np.empty(0, dtype=np.int32)
  75   base_cos = np.cos(linspace)
  76   base_sin = np.sin(linspace)
  77 
  78   y = np.empty(0, dtype=np.int32)
  79   for label in range(n_classes):
  80     circ_x = np.append(circ_x, base_cos)
  81     circ_y = np.append(circ_y, base_sin)
  82     base_cos *= factor
  83     base_sin *= factor
  84     y = np.append(y, label * np.ones(n_samples // n_classes, dtype=np.int32))
  85 
  86   # Add more points if n_samples is not divisible by n_classes (unbalanced!)
  87   extras = n_samples % n_classes
  88   circ_x = np.append(circ_x, np.cos(np.random.rand(extras) * 2 * np.pi))
  89   circ_y = np.append(circ_y, np.sin(np.random.rand(extras) * 2 * np.pi))
  90   y = np.append(y, np.zeros(extras, dtype=np.int32))
  91 
  92   # Reshape the features/labels
  93   X = np.vstack((circ_x, circ_y)).T
  94   y = np.hstack(y)
  95 
  96   # Shuffle the data
  97   indices = np.random.permutation(range(n_samples))
  98   if noise is not None:
  99     X += np.random.normal(scale=noise, size=X.shape)
 100   return Dataset(data=X[indices], target=y[indices])

 103 @deprecated(None, 'Consider using synthetic datasets from scikits.learn.')
 104 def spirals(n_samples=100,
 105             noise=None,
 106             seed=None,
 107             mode='archimedes',
 108             n_loops=2,
 109             *args,
 110             **kwargs):
 111   """Create spirals
 112 
 113   Currently only binary classification is supported for spiral generation
 114 
 115   Args:
 116     n_samples: int, number of datapoints to generate
 117     noise: float or None, standard deviation of the Gaussian noise added
 118     seed: int or None, seed for the noise
 119     n_loops: int, number of spiral loops, doesn't play well with 'bernoulli'
 120     mode: str, how the spiral should be generated. Current implementations:
 121       'archimedes': a spiral with equal distances between branches
 122       'bernoulli': logarithmic spiral with branch distances increasing
 123       'fermat': a spiral with branch distances decreasing (sqrt)
 124 
 125   Returns:
 126     Shuffled features and labels for 'spirals' synthetic dataset of type
 127     `base.Dataset`
 128 
 129   Raises:
 130     ValueError: If the generation `mode` is not valid
 131 
 132   TODO:
 133     - Generation of unbalanced data
 134   """
 135   n_classes = 2  # I am not sure how to make it multiclass
 136 
 137   _modes = {
 138       'archimedes': _archimedes_spiral,
 139       'bernoulli': _bernoulli_spiral,
 140       'fermat': _fermat_spiral
 141   }
 142 
 143   if mode is None or mode not in _modes:
 144     raise ValueError('Cannot generate spiral with mode %s' % mode)
 145 
 146   if seed is not None:
 147     np.random.seed(seed)
 148   linspace = np.linspace(0, 2 * n_loops * np.pi, n_samples // n_classes)
 149   spir_x = np.empty(0, dtype=np.int32)
 150   spir_y = np.empty(0, dtype=np.int32)
 151 
 152   y = np.empty(0, dtype=np.int32)
 153   for label in range(n_classes):
 154     base_cos, base_sin = _modes[mode](linspace, label * np.pi, *args, **kwargs)
 155     spir_x = np.append(spir_x, base_cos)
 156     spir_y = np.append(spir_y, base_sin)
 157     y = np.append(y, label * np.ones(n_samples // n_classes, dtype=np.int32))
 158 
 159   # Add more points if n_samples is not divisible by n_classes (unbalanced!)
 160   extras = n_samples % n_classes
 161   if extras > 0:
 162     x_extra, y_extra = _modes[mode](np.random.rand(extras) * 2 * np.pi, *args,
 163                                     **kwargs)
 164     spir_x = np.append(spir_x, x_extra)
 165     spir_y = np.append(spir_y, y_extra)
 166     y = np.append(y, np.zeros(extras, dtype=np.int32))
 167 
 168   # Reshape the features/labels
 169   X = np.vstack((spir_x, spir_y)).T
 170   y = np.hstack(y)
 171 
 172   # Shuffle the data
 173   indices = np.random.permutation(range(n_samples))
 174   if noise is not None:
 175     X += np.random.normal(scale=noise, size=X.shape)
 176   return Dataset(data=X[indices], target=y[indices])

 179 def _archimedes_spiral(theta, theta_offset=0., *args, **kwargs):
 180   """Return Archimedes spiral
 181 
 182   Args:
 183     theta: array-like, angles from polar coordinates to be converted
 184     theta_offset: float, angle offset in radians (2*pi = 0)
 185   """
 186   x, y = theta * np.cos(theta + theta_offset), theta * np.sin(
 187       theta + theta_offset)
 188   x_norm = np.max(np.abs(x))
 189   y_norm = np.max(np.abs(y))
 190   x, y = x / x_norm, y / y_norm
 191   return x, y

 194 def _bernoulli_spiral(theta, theta_offset=0., *args, **kwargs):
 195   """Return Equiangular (Bernoulli's) spiral
 196 
 197   Args:
 198     theta: array-like, angles from polar coordinates to be converted
 199     theta_offset: float, angle offset in radians (2*pi = 0)
 200 
 201   Kwargs:
 202     exp_scale: growth rate of the exponential
 203   """
 204   exp_scale = kwargs.pop('exp_scale', 0.1)
 205 
 206   x, y = np.exp(exp_scale * theta) * np.cos(theta + theta_offset), np.exp(
 207       exp_scale * theta) * np.sin(theta + theta_offset)
 208   x_norm = np.max(np.abs(x))
 209   y_norm = np.max(np.abs(y))
 210   x, y = x / x_norm, y / y_norm
 211   return x, y

 214 def _fermat_spiral(theta, theta_offset=0., *args, **kwargs):
 215   """Return Parabolic (Fermat's) spiral
 216 
 217   Args:
 218     theta: array-like, angles from polar coordinates to be converted
 219     theta_offset: float, angle offset in radians (2*pi = 0)
 220   """
 221   x, y = np.sqrt(theta) * np.cos(theta + theta_offset), np.sqrt(theta) * np.sin(
 222       theta + theta_offset)
 223   x_norm = np.max(np.abs(x))
 224   y_norm = np.max(np.abs(y))
 225   x, y = x / x_norm, y / y_norm
 226   return x, y
