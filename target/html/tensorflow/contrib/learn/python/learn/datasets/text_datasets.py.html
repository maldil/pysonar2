  23 from __future__ import absolute_import
  24 from __future__ import division

  27 import os
  28 import tarfile

  30 import numpy as np
  31 
  32 from tensorflow.contrib.learn.python.learn.datasets import base
  33 from tensorflow.python.platform import gfile
  34 from tensorflow.python.util.deprecation import deprecated
  35 
  36 DBPEDIA_URL = 'https://github.com/le-scientifique/torchDatasets/raw/master/dbpedia_csv.tar.gz'
  37 
  38 
  39 @deprecated(None, 'See contrib/learn/README.md')
  40 def maybe_download_dbpedia(data_dir):
  41   """Download if DBpedia data is not present."""
  42   train_path = os.path.join(data_dir, 'dbpedia_csv/train.csv')
  43   test_path = os.path.join(data_dir, 'dbpedia_csv/test.csv')
  44   if not (gfile.Exists(train_path) and gfile.Exists(test_path)):
  45     archive_path = base.maybe_download(
  46         'dbpedia_csv.tar.gz', data_dir, DBPEDIA_URL)
  47     tfile = tarfile.open(archive_path, 'r:*')
  48     tfile.extractall(data_dir)
  49 
  50 
  51 @deprecated(None, 'See contrib/learn/README.md')
  52 def load_dbpedia(size='small', test_with_fake_data=False):
  53   """Get DBpedia datasets from CSV files."""
  54   if not test_with_fake_data:
  55     data_dir = os.path.join(os.getenv('TF_EXP_BASE_DIR', ''), 'dbpedia_data')
  56     maybe_download_dbpedia(data_dir)
  57 
  58     train_path = os.path.join(data_dir, 'dbpedia_csv', 'train.csv')
  59     test_path = os.path.join(data_dir, 'dbpedia_csv', 'test.csv')
  60 
  61     if size == 'small':
  62       # Reduce the size of original data by a factor of 1000.
  63       base.shrink_csv(train_path, 1000)
  64       base.shrink_csv(test_path, 1000)
  65       train_path = train_path.replace('train.csv', 'train_small.csv')
  66       test_path = test_path.replace('test.csv', 'test_small.csv')
  67   else:
  68     module_path = os.path.dirname(__file__)
  69     train_path = os.path.join(module_path, 'data', 'text_train.csv')
  70     test_path = os.path.join(module_path, 'data', 'text_test.csv')
  71 
  72   train = base.load_csv_without_header(
  73       train_path, target_dtype=np.int32, features_dtype=np.str, target_column=0)
  74   test = base.load_csv_without_header(
  75       test_path, target_dtype=np.int32, features_dtype=np.str, target_column=0)
  76 
  77   return base.Datasets(train=train, validation=None, test=test)
