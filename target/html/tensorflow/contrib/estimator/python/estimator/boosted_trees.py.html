  26 def _validate_input_fn_and_repeat_dataset(train_input_fn):
  27   """Validates whether the input_fn is valid, and repeat() if tf.Dataset."""
  28   def _input_fn():
  29     result_input_fn = train_input_fn()
  30     if isinstance(result_input_fn, dataset_ops.Dataset):
  31       return result_input_fn.repeat()
  32     return result_input_fn
  33 
  34   return _input_fn

  37 def _is_classification_head(head):
  38   """Infers if the head is a classification head."""
  39   # Check using all classification heads defined in canned/head.py. However, it
  40   # is not a complete list - it does not check for other classification heads
  41   # not defined in the head library.
  42   # pylint: disable=protected-access
  43   return isinstance(head,
  44                     (head_lib._BinaryLogisticHeadWithSigmoidCrossEntropyLoss,
  45                      head_lib._MultiClassHeadWithSoftmaxCrossEntropyLoss))

  49 class _BoostedTreesEstimator(canned_boosted_trees._BoostedTreesBase):  # pylint: disable=protected-access
  50   """An Estimator for Tensorflow Boosted Trees models."""
  51 
  52   def __init__(self,
  53                feature_columns,
  54                n_batches_per_layer,
  55                head,
  56                model_dir=None,
  57                weight_column=None,
  58                n_trees=100,
  59                max_depth=6,
  60                learning_rate=0.1,
  61                l1_regularization=0.,
  62                l2_regularization=0.,
  63                tree_complexity=0.,
  64                min_node_weight=0.,
  65                config=None,
  66                center_bias=False,
  67                pruning_mode='none'):
  68     """Initializes a `BoostedTreesEstimator` instance.
  69 
  70     Args:
  71       feature_columns: An iterable containing all the feature columns used by
  72         the model. All items in the set should be instances of classes derived
  73         from `FeatureColumn`.
  74       n_batches_per_layer: the number of batches to collect statistics per
  75         layer.
  76       head: the `Head` instance defined for Estimator.
  77       model_dir: Directory to save model parameters, graph and etc. This can
  78         also be used to load checkpoints from the directory into a estimator
  79         to continue training a previously saved model.
  80       weight_column: A string or a `_NumericColumn` created by
  81         `tf.feature_column.numeric_column` defining feature column representing
  82         weights. It is used to downweight or boost examples during training. It
  83         will be multiplied by the loss of the example. If it is a string, it is
  84         used as a key to fetch weight tensor from the `features`. If it is a
  85         `_NumericColumn`, raw tensor is fetched by key `weight_column.key`,
  86         then weight_column.normalizer_fn is applied on it to get weight tensor.
  87       n_trees: number trees to be created.
  88       max_depth: maximum depth of the tree to grow.
  89       learning_rate: shrinkage parameter to be used when a tree added to the
  90         model.
  91       l1_regularization: regularization multiplier applied to the absolute
  92         weights of the tree leafs.
  93       l2_regularization: regularization multiplier applied to the square weights
  94         of the tree leafs.
  95       tree_complexity: regularization factor to penalize trees with more leaves.
  96       min_node_weight: minimum hessian a node must have for a split to be
  97         considered. The value will be compared with sum(leaf_hessian)/
  98         (batch_size * n_batches_per_layer).
  99       config: `RunConfig` object to configure the runtime settings.
 100       center_bias: Whether bias centering needs to occur. Bias centering refers
 101         to the first node in the very first tree returning the prediction that
 102         is aligned with the original labels distribution. For example, for
 103         regression problems, the first node will return the mean of the labels.
 104         For binary classification problems, it will return a logit for a prior
 105         probability of label 1.
 106       pruning_mode: one of 'none', 'pre', 'post' to indicate no pruning, pre-
 107         pruning (do not split a node if not enough gain is observed) and post
 108         pruning (build the tree up to a max depth and then prune branches with
 109         negative gain). For pre and post pruning, you MUST provide
 110         tree_complexity >0.
 111 
 112     Raises:
 113       ValueError: when wrong arguments are given or unsupported functionalities
 114          are requested.
 115     """
 116     # HParams for the model.
 117     # pylint: disable=protected-access
 118     tree_hparams = canned_boosted_trees._TreeHParams(
 119         n_trees, max_depth, learning_rate, l1_regularization, l2_regularization,
 120         tree_complexity, min_node_weight, center_bias, pruning_mode)
 121 
 122     def _model_fn(features, labels, mode, config):
 123       return canned_boosted_trees._bt_model_fn(
 124           features,
 125           labels,
 126           mode,
 127           head,
 128           feature_columns,
 129           tree_hparams,
 130           n_batches_per_layer,
 131           config=config)
 132 
 133     super(_BoostedTreesEstimator, self).__init__(
 134         model_fn=_model_fn,
 135         model_dir=model_dir,
 136         config=config,
 137         feature_columns=feature_columns,
 138         head=head,
 139         center_bias=center_bias,
 140         is_classification=_is_classification_head(head))
 141     # pylint: enable=protected-access

 144 def boosted_trees_classifier_train_in_memory(
 145     train_input_fn,
 146     feature_columns,
 147     model_dir=None,
 148     n_classes=canned_boosted_trees._HOLD_FOR_MULTI_CLASS_SUPPORT,
 149     weight_column=None,
 150     label_vocabulary=None,
 151     n_trees=100,
 152     max_depth=6,
 153     learning_rate=0.1,
 154     l1_regularization=0.,
 155     l2_regularization=0.,
 156     tree_complexity=0.,
 157     min_node_weight=0.,
 158     config=None,
 159     train_hooks=None,
 160     center_bias=False,
 161     pruning_mode='none'):
 162   """Trains a boosted tree classifier with in memory dataset.
 163 
 164   Example:
 165 
 166   ```python
 167   bucketized_feature_1 = bucketized_column(
 168     numeric_column('feature_1'), BUCKET_BOUNDARIES_1)
 169   bucketized_feature_2 = bucketized_column(
 170     numeric_column('feature_2'), BUCKET_BOUNDARIES_2)
 171 
 172   def train_input_fn():
 173     dataset = create-dataset-from-training-data
 174     # This is tf.data.Dataset of a tuple of feature dict and label.
 175     #   e.g. Dataset.zip((Dataset.from_tensors({'f1': f1_array, ...}),
 176     #                     Dataset.from_tensors(label_array)))
 177     # The returned Dataset shouldn't be batched.
 178     # If Dataset repeats, only the first repetition would be used for training.
 179     return dataset
 180 
 181   classifier = boosted_trees_classifier_train_in_memory(
 182       train_input_fn,
 183       feature_columns=[bucketized_feature_1, bucketized_feature_2],
 184       n_trees=100,
 185       ... <some other params>
 186   )
 187 
 188   def input_fn_eval():
 189     ...
 190     return dataset
 191 
 192   metrics = classifier.evaluate(input_fn=input_fn_eval, steps=10)
 193   ```
 194 
 195   Args:
 196     train_input_fn: the input function returns a dataset containing a single
 197       epoch of *unbatched* features and labels.
 198     feature_columns: An iterable containing all the feature columns used by
 199       the model. All items in the set should be instances of classes derived
 200       from `FeatureColumn`.
 201     model_dir: Directory to save model parameters, graph and etc. This can
 202       also be used to load checkpoints from the directory into a estimator
 203       to continue training a previously saved model.
 204     n_classes: number of label classes. Default is binary classification.
 205       Multiclass support is not yet implemented.
 206     weight_column: A string or a `_NumericColumn` created by
 207       `tf.feature_column.numeric_column` defining feature column representing
 208       weights. It is used to downweight or boost examples during training. It
 209       will be multiplied by the loss of the example. If it is a string, it is
 210       used as a key to fetch weight tensor from the `features`. If it is a
 211       `_NumericColumn`, raw tensor is fetched by key `weight_column.key`,
 212       then weight_column.normalizer_fn is applied on it to get weight tensor.
 213     label_vocabulary: A list of strings represents possible label values. If
 214       given, labels must be string type and have any value in
 215       `label_vocabulary`. If it is not given, that means labels are
 216       already encoded as integer or float within [0, 1] for `n_classes=2` and
 217       encoded as integer values in {0, 1,..., n_classes-1} for `n_classes`>2 .
 218       Also there will be errors if vocabulary is not provided and labels are
 219       string.
 220     n_trees: number trees to be created.
 221     max_depth: maximum depth of the tree to grow.
 222     learning_rate: shrinkage parameter to be used when a tree added to the
 223       model.
 224     l1_regularization: regularization multiplier applied to the absolute
 225       weights of the tree leafs.
 226     l2_regularization: regularization multiplier applied to the square weights
 227       of the tree leafs.
 228     tree_complexity: regularization factor to penalize trees with more leaves.
 229     min_node_weight: minimum hessian a node must have for a split to be
 230         considered. The value will be compared with sum(leaf_hessian)/
 231         (batch_size * n_batches_per_layer).
 232     config: `RunConfig` object to configure the runtime settings.
 233     train_hooks: a list of Hook instances to be passed to estimator.train()
 234     center_bias: Whether bias centering needs to occur. Bias centering refers
 235         to the first node in the very first tree returning the prediction that
 236         is aligned with the original labels distribution. For example, for
 237         regression problems, the first node will return the mean of the labels.
 238         For binary classification problems, it will return a logit for a prior
 239         probability of label 1.
 240     pruning_mode: one of 'none', 'pre', 'post' to indicate no pruning, pre-
 241         pruning (do not split a node if not enough gain is observed) and post
 242         pruning (build the tree up to a max depth and then prune branches with
 243         negative gain). For pre and post pruning, you MUST provide
 244         tree_complexity >0.
 245 
 246   Returns:
 247     a `BoostedTreesClassifier` instance created with the given arguments and
 248       trained with the data loaded up on memory from the input_fn.
 249 
 250   Raises:
 251     ValueError: when wrong arguments are given or unsupported functionalities
 252        are requested.
 253   """
 254   # pylint: disable=protected-access
 255   # TODO(nponomareva): Support multi-class cases.
 256   if n_classes == canned_boosted_trees._HOLD_FOR_MULTI_CLASS_SUPPORT:
 257     n_classes = 2
 258   head, closed_form = (
 259       canned_boosted_trees._create_classification_head_and_closed_form(
 260           n_classes, weight_column, label_vocabulary=label_vocabulary))
 261 
 262   # HParams for the model.
 263   tree_hparams = canned_boosted_trees._TreeHParams(
 264       n_trees, max_depth, learning_rate, l1_regularization, l2_regularization,
 265       tree_complexity, min_node_weight, center_bias, pruning_mode)
 266 
 267   def _model_fn(features, labels, mode, config):
 268     return canned_boosted_trees._bt_model_fn(
 269         features,
 270         labels,
 271         mode,
 272         head,
 273         feature_columns,
 274         tree_hparams,
 275         n_batches_per_layer=1,
 276         config=config,
 277         closed_form_grad_and_hess_fn=closed_form,
 278         train_in_memory=True)
 279 
 280   in_memory_classifier = estimator.Estimator(
 281       model_fn=_model_fn, model_dir=model_dir, config=config)
 282 
 283   in_memory_classifier.train(
 284       input_fn=_validate_input_fn_and_repeat_dataset(train_input_fn),
 285       hooks=train_hooks)
 286 
 287   return in_memory_classifier
 288   # pylint: enable=protected-access
