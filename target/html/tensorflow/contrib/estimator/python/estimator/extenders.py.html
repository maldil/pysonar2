  37 def add_metrics(estimator, metric_fn):
  38   """Creates a new `tf.estimator.Estimator` which has given metrics.
  39 
  40   Example:
  41 
  42   ```python
  43     def my_auc(labels, predictions):
  44       return {'auc': tf.metrics.auc(labels, predictions['logistic'])}
  45 
  46     estimator = tf.estimator.DNNClassifier(...)
  47     estimator = tf.contrib.estimator.add_metrics(estimator, my_auc)
  48     estimator.train(...)
  49     estimator.evaluate(...)
  50   ```
  51   Example usage of custom metric which uses features:
  52 
  53   ```python
  54     def my_auc(features, labels, predictions):
  55       return {'auc': tf.metrics.auc(
  56         labels, predictions['logistic'], weights=features['weight'])}
  57 
  58     estimator = tf.estimator.DNNClassifier(...)
  59     estimator = tf.contrib.estimator.add_metrics(estimator, my_auc)
  60     estimator.train(...)
  61     estimator.evaluate(...)
  62   ```
  63 
  64   Args:
  65     estimator: A `tf.estimator.Estimator` object.
  66     metric_fn: A function which should obey the following signature:
  67       - Args: can only have following four arguments in any order:
  68         * predictions: Predictions `Tensor` or dict of `Tensor` created by given
  69           `estimator`.
  70         * features: Input `dict` of `Tensor` objects created by `input_fn` which
  71           is given to `estimator.evaluate` as an argument.
  72         * labels:  Labels `Tensor` or dict of `Tensor` created by `input_fn`
  73           which is given to `estimator.evaluate` as an argument.
  74         * config: config attribute of the `estimator`.
  75        - Returns:
  76          Dict of metric results keyed by name. Final metrics are a union of this
  77          and `estimator's` existing metrics. If there is a name conflict between
  78          this and `estimator`s existing metrics, this will override the existing
  79          one. The values of the dict are the results of calling a metric
  80          function, namely a `(metric_tensor, update_op)` tuple.
  81 
  82   Returns:
  83       A new `tf.estimator.Estimator` which has a union of original metrics with
  84         given ones.
  85   """
  86   _verify_metric_fn_args(metric_fn)
  87 
  88   def new_model_fn(features, labels, mode, config):
  89     spec = estimator.model_fn(features, labels, mode, config)
  90     if mode != model_fn_lib.ModeKeys.EVAL:
  91       return spec
  92     new_metrics = _call_metric_fn(metric_fn, features, labels, spec.predictions,
  93                                   config)
  94     all_metrics = spec.eval_metric_ops or {}
  95     all_metrics.update(new_metrics)
  96     return spec._replace(eval_metric_ops=all_metrics)
  97 
  98   return estimator_lib.Estimator(
  99       model_fn=new_model_fn,
 100       model_dir=estimator.model_dir,
 101       config=estimator.config,
 102       # pylint: disable=protected-access
 103       warm_start_from=estimator._warm_start_settings)
 104       # pylint: enable=protected-access

 107 def clip_gradients_by_norm(optimizer, clip_norm):
 108   """Returns an optimizer which clips gradients before applying them.
 109 
 110   Example:
 111 
 112   ```python
 113   optimizer = tf.train.ProximalAdagradOptimizer(
 114       learning_rate=0.1,
 115       l1_regularization_strength=0.001)
 116   optimizer = tf.contrib.estimator.clip_gradients_by_norm(
 117       optimizer, clip_norm)
 118   estimator = tf.estimator.DNNClassifier(
 119       feature_columns=[...],
 120       hidden_units=[1024, 512, 256],
 121       optimizer=optimizer)
 122   ```
 123 
 124   Args:
 125     optimizer: An `tf.Optimizer` object to apply gradients.
 126     clip_norm: A 0-D (scalar) `Tensor` > 0. The clipping ratio.
 127 
 128   Returns:
 129     A `tf.Optimizer`.
 130   """
 131 
 132   def clip_grads(grads_and_vars):
 133     gradients, variables = zip(*grads_and_vars)
 134     gradients = clip_ops.clip_by_global_norm(gradients, clip_norm)[0]
 135     grads_and_vars = list(zip(gradients, variables))
 136     return grads_and_vars
 137 
 138   return _TransformGradients(
 139       optimizer=optimizer,
 140       transform_grads_fn=clip_grads,
 141       name='ClipByNorm' + optimizer.get_name())

 144 def forward_features(estimator, keys=None, sparse_default_values=None):
 145   """Forward features to predictions dictionary.
 146 
 147   In some cases, user wants to see some of the features in estimators prediction
 148   output. As an example, consider a batch prediction service: The service simply
 149   runs inference on the users graph and returns the results. Keys are essential
 150   because there is no order guarantee on the outputs so they need to be rejoined
 151   to the inputs via keys or transclusion of the inputs in the outputs.
 152   Example:
 153   ```python
 154     def input_fn():
 155       features, labels = ...
 156       features['unique_example_id'] = ...
 157       features, labels
 158     estimator = tf.estimator.LinearClassifier(...)
 159     estimator = tf.contrib.estimator.forward_features(
 160         estimator, 'unique_example_id')
 161     estimator.train(...)
 162     assert 'unique_example_id' in estimator.predict(...)
 163   ```
 164   Args:
 165     estimator: A `tf.estimator.Estimator` object.
 166     keys: A `string` or a `list` of `string`. If it is `None`, all of the
 167       `features` in `dict` is forwarded to the `predictions`. If it is a
 168       `string`, only given key is forwarded. If it is a `list` of strings, all
 169       the given `keys` are forwarded.
 170     sparse_default_values: A dict of `str` keys mapping the name of the sparse
 171       features to be converted to dense, to the default value to use. Only
 172       sparse features indicated in the dictionary are converted to dense and the
 173       provided default value is used.
 174 
 175   Returns:
 176       A new `tf.estimator.Estimator` which forwards features to predictions.
 177   Raises:
 178     ValueError:
 179       * if `keys` is already part of `predictions`. We don't allow
 180         override.
 181       * if 'keys' does not exist in `features`.
 182     TypeError: if `keys` type is not one of `string` or list/tuple of `string`.
 183   """
 184 
 185   def verify_key_types(keys):  # pylint: disable=missing-docstring
 186     if keys is None:
 187       return keys
 188     if isinstance(keys, six.string_types):
 189       return [keys]
 190     if not isinstance(keys, (list, tuple)):
 191       raise TypeError('keys should be either a string or a list of strings. '
 192                       'Given: {}'.format(type(keys)))
 193     for key in keys:
 194       if not isinstance(key, six.string_types):
 195         raise TypeError('All items in the given keys list should be a string. '
 196                         'There exist an item with type: {}'.format(type(key)))
 197     return keys
 198 
 199   def get_keys(features):
 200     if keys is None:
 201       return features.keys()
 202     return keys
 203 
 204   def verify_keys_and_predictions(features, predictions):
 205     if not isinstance(predictions, dict):
 206       raise ValueError(
 207           'Predictions should be a dict to be able to forward features. '
 208           'Given: {}'.format(type(predictions)))
 209     for key in get_keys(features):
 210       if key not in features:
 211         raise ValueError(
 212             'keys should be exist in features. Key "{}" is not in features '
 213             'dict. features dict has following keys: {}. Please check '
 214             'arguments of forward_features.'.format(key, features.keys()))
 215       if key in predictions:
 216         raise ValueError(
 217             'Cannot forward feature key ({}). Since it does exist in '
 218             'predictions. Existing prediction keys: {}. Please check arguments '
 219             'of forward_features.'.format(key, predictions.keys()))
 220 
 221   keys = verify_key_types(keys)
 222 
 223   def new_model_fn(features, labels, mode, config):  # pylint: disable=missing-docstring
 224     spec = estimator.model_fn(features, labels, mode, config)
 225     predictions = spec.predictions
 226     if predictions is None:
 227       return spec
 228     verify_keys_and_predictions(features, predictions)
 229     for key in get_keys(features):
 230       feature = sparse_tensor_lib.convert_to_tensor_or_sparse_tensor(
 231           features[key])
 232       if sparse_default_values and (key in sparse_default_values):
 233         if not isinstance(feature, sparse_tensor_lib.SparseTensor):
 234           raise ValueError(
 235               'Feature ({}) is expected to be a `SparseTensor`.'.format(key))
 236         feature = sparse_ops.sparse_tensor_to_dense(
 237             feature, default_value=sparse_default_values[key])
 238       if not isinstance(feature, ops.Tensor):
 239         raise ValueError(
 240             'Feature ({}) should be a Tensor. Please use `keys` '
 241             'argument of forward_features to filter unwanted features, or'
 242             'add key to argument `sparse_default_values`.'
 243             'Type of features[{}] is {}.'.format(key, key, type(feature)))
 244       predictions[key] = feature
 245     spec = spec._replace(predictions=predictions)
 246     if spec.export_outputs:
 247       for ekey in ['predict', 'serving_default']:
 248         if (ekey in spec.export_outputs and
 249             isinstance(spec.export_outputs[ekey],
 250                        PredictOutput)):
 251           export_outputs = spec.export_outputs[ekey].outputs
 252           for key in get_keys(features):
 253             export_outputs[key] = predictions[key]
 254 
 255     return spec
 256 
 257   return estimator_lib.Estimator(
 258       model_fn=new_model_fn,
 259       model_dir=estimator.model_dir,
 260       config=estimator.config)

 263 class _TransformGradients(optimizer_lib.Optimizer):
 264   """Add given gradient transformation to the optimizer."""
 265 
 266   def __init__(self, optimizer, transform_grads_fn, name=None):
 267     """Construct an `tf.Optimizer` wrapper to apply given transformations.
 268 
 269     Example:
 270 
 271     ```python
 272     optimizer = tf.train.ProximalAdagradOptimizer(
 273         learning_rate=0.1,
 274         l1_regularization_strength=0.001)
 275     def clip_grads(grads_and_vars):
 276       gradients, variables = zip(*grads_and_vars)
 277       gradients = tf.clip_by_global_norm(grads, my_norm)[0]
 278       grads_and_vars = list(zip(gradients, variables))
 279       return grads_and_vars
 280     optimizer = _TransformGradients(
 281         opt=optimizer, transform_grads_fn=clip_grads)
 282     estimator = tf.estimator.DNNClassifier(
 283         feature_columns=[...],
 284         hidden_units=[1024, 512, 256],
 285         optimizer=optimizer)
 286     ```
 287 
 288     Args:
 289       optimizer: An `tf.Optimizer` object to apply gradients.
 290       transform_grads_fn: A function which takes a single argument, a list of
 291         gradient to variable pairs (tuples), performs any requested gradient
 292         updates, such as gradient clipping or multipliers, and returns the
 293         updated list.
 294       name: A string which will be used for debugging purposes.
 295     """
 296     super(_TransformGradients, self).__init__(
 297         use_locking=False, name=name or optimizer.get_name())
 298     self._optimizer = optimizer
 299     self._transform_grads_fn = transform_grads_fn
 300 
 301   def compute_gradients(self, *args, **kwargs):
 302     """See `tf.Optimizer`."""
 303     return self._optimizer.compute_gradients(*args, **kwargs)
 304 
 305   def apply_gradients(self, grads_and_vars, global_step=None, name=None):
 306     """Apply gradients to variables.
 307 
 308     Calls `transform_grads_fn`, and then applies the real optimizer.
 309 
 310     Args:
 311       grads_and_vars: List of (gradient, variable) pairs as returned by
 312         compute_gradients().
 313       global_step: Optional Variable to increment by one after the
 314         variables have been updated.
 315       name: Optional name for the returned operation.  Default to the
 316         name passed to the Optimizer constructor.
 317 
 318     Returns:
 319       An `Operation` that applies the gradients. If `global_step` was not None,
 320       that operation also increments `global_step`.
 321 
 322     Raises:
 323       ValueError: If the grads_and_vars is malformed.
 324     """
 325     grads_and_vars = self._transform_grads_fn(grads_and_vars)
 326     return self._optimizer.apply_gradients(grads_and_vars, global_step, name)
 327 
 328   def get_slot(self, *args, **kwargs):
 329     """See `tf.Optimizer`."""
 330     return self._optimizer.get_slot(*args, **kwargs)
 331 
 332   def get_slot_names(self, *args, **kwargs):
 333     """See `tf.Optimizer`."""
 334     return self._optimizer.get_slot_names(*args, **kwargs)

 337 def _verify_metric_fn_args(metric_fn):
 338   args = set(function_utils.fn_args(metric_fn))
 339   invalid_args = list(args - _VALID_METRIC_FN_ARGS)
 340   if invalid_args:
 341     raise ValueError('metric_fn (%s) has following not expected args: %s' %
 342                      (metric_fn, invalid_args))

 345 def _call_metric_fn(metric_fn, features, labels, predictions, config):
 346   """Calls metric fn with proper arguments."""
 347   metric_fn_args = function_utils.fn_args(metric_fn)
 348   kwargs = {}
 349   if 'features' in metric_fn_args:
 350     kwargs['features'] = features
 351   if 'labels' in metric_fn_args:
 352     kwargs['labels'] = labels
 353   if 'predictions' in metric_fn_args:
 354     kwargs['predictions'] = predictions
 355   if 'config' in metric_fn_args:
 356     kwargs['config'] = config
 357   return metric_fn(**kwargs)
