  26 class DNNLinearCombinedEstimator(estimator.Estimator):
  27   """An estimator for TensorFlow Linear and DNN joined models with custom head.
  28 
  29   Note: This estimator is also known as wide-n-deep.
  30 
  31   Example:
  32 
  33   ```python
  34   numeric_feature = numeric_column(...)
  35   categorical_column_a = categorical_column_with_hash_bucket(...)
  36   categorical_column_b = categorical_column_with_hash_bucket(...)
  37 
  38   categorical_feature_a_x_categorical_feature_b = crossed_column(...)
  39   categorical_feature_a_emb = embedding_column(
  40       categorical_column=categorical_feature_a, ...)
  41   categorical_feature_b_emb = embedding_column(
  42       categorical_column=categorical_feature_b, ...)
  43 
  44   estimator = DNNLinearCombinedEstimator(
  45       head=tf.contrib.estimator.multi_label_head(n_classes=3),
  46       # wide settings
  47       linear_feature_columns=[categorical_feature_a_x_categorical_feature_b],
  48       linear_optimizer=tf.train.FtrlOptimizer(...),
  49       # deep settings
  50       dnn_feature_columns=[
  51           categorical_feature_a_emb, categorical_feature_b_emb,
  52           numeric_feature],
  53       dnn_hidden_units=[1000, 500, 100],
  54       dnn_optimizer=tf.train.ProximalAdagradOptimizer(...))
  55 
  56   # To apply L1 and L2 regularization, you can set dnn_optimizer to:
  57   tf.train.ProximalAdagradOptimizer(
  58       learning_rate=0.1,
  59       l1_regularization_strength=0.001,
  60       l2_regularization_strength=0.001)
  61   # To apply learning rate decay, you can set dnn_optimizer to a callable:
  62   lambda: tf.AdamOptimizer(
  63       learning_rate=tf.exponential_decay(
  64           learning_rate=0.1,
  65           global_step=tf.get_global_step(),
  66           decay_steps=10000,
  67           decay_rate=0.96)
  68   # It is the same for linear_optimizer.
  69 
  70   # Input builders
  71   def input_fn_train: # returns x, y
  72     pass
  73   estimator.train(input_fn=input_fn_train, steps=100)
  74 
  75   def input_fn_eval: # returns x, y
  76     pass
  77   metrics = estimator.evaluate(input_fn=input_fn_eval, steps=10)
  78   def input_fn_predict: # returns x, None
  79     pass
  80   predictions = estimator.predict(input_fn=input_fn_predict)
  81   ```
  82 
  83   Input of `train` and `evaluate` should have following features,
  84   otherwise there will be a `KeyError`:
  85 
  86   * for each `column` in `dnn_feature_columns` + `linear_feature_columns`:
  87     - if `column` is a `_CategoricalColumn`, a feature with `key=column.name`
  88       whose `value` is a `SparseTensor`.
  89     - if `column` is a `_WeightedCategoricalColumn`, two features: the first
  90       with `key` the id column name, the second with `key` the weight column
  91       name. Both features' `value` must be a `SparseTensor`.
  92     - if `column` is a `_DenseColumn`, a feature with `key=column.name`
  93       whose `value` is a `Tensor`.
  94 
  95   Loss is calculated by using mean squared error.
  96 
  97   @compatibility(eager)
  98   Estimators are not compatible with eager execution.
  99   @end_compatibility
 100   """
 101 
 102   def __init__(self,
 103                head,
 104                model_dir=None,
 105                linear_feature_columns=None,
 106                linear_optimizer='Ftrl',
 107                dnn_feature_columns=None,
 108                dnn_optimizer='Adagrad',
 109                dnn_hidden_units=None,
 110                dnn_activation_fn=nn.relu,
 111                dnn_dropout=None,
 112                input_layer_partitioner=None,
 113                config=None,
 114                linear_sparse_combiner='sum'):
 115     """Initializes a DNNLinearCombinedEstimator instance.
 116 
 117     Args:
 118       head: A `_Head` instance constructed with a method such as
 119         `tf.contrib.estimator.multi_label_head`.
 120       model_dir: Directory to save model parameters, graph and etc. This can
 121         also be used to load checkpoints from the directory into a estimator
 122         to continue training a previously saved model.
 123       linear_feature_columns: An iterable containing all the feature columns
 124         used by linear part of the model. All items in the set must be
 125         instances of classes derived from `FeatureColumn`.
 126       linear_optimizer: An instance of `tf.Optimizer` used to apply gradients to
 127         the linear part of the model. Can also be a string (one of 'Adagrad',
 128         'Adam', 'Ftrl', 'RMSProp', 'SGD'), or callable. Defaults to FTRL
 129         optimizer.
 130       dnn_feature_columns: An iterable containing all the feature columns used
 131         by deep part of the model. All items in the set must be instances of
 132         classes derived from `FeatureColumn`.
 133       dnn_optimizer: An instance of `tf.Optimizer` used to apply gradients to
 134         the deep part of the model. Can also be a string (one of 'Adagrad',
 135         'Adam', 'Ftrl', 'RMSProp', 'SGD'), or callable. Defaults to Adagrad
 136         optimizer.
 137       dnn_hidden_units: List of hidden units per layer. All layers are fully
 138         connected.
 139       dnn_activation_fn: Activation function applied to each layer. If None,
 140         will use `tf.nn.relu`.
 141       dnn_dropout: When not None, the probability we will drop out
 142         a given coordinate.
 143       input_layer_partitioner: Partitioner for input layer. Defaults to
 144         `min_max_variable_partitioner` with `min_slice_size` 64 << 20.
 145       config: RunConfig object to configure the runtime settings.
 146       linear_sparse_combiner: A string specifying how to reduce the linear model
 147         if a categorical column is multivalent.  One of "mean", "sqrtn", and
 148         "sum" -- these are effectively different ways to do example-level
 149         normalization, which can be useful for bag-of-words features.  For more
 150         details, see `tf.feature_column.linear_model`.
 151 
 152     Raises:
 153       ValueError: If both linear_feature_columns and dnn_features_columns are
 154         empty at the same time.
 155     """
 156     linear_feature_columns = linear_feature_columns or []
 157     dnn_feature_columns = dnn_feature_columns or []
 158     self._feature_columns = (
 159         list(linear_feature_columns) + list(dnn_feature_columns))
 160     if not self._feature_columns:
 161       raise ValueError('Either linear_feature_columns or dnn_feature_columns '
 162                        'must be defined.')
 163 
 164     def _model_fn(features, labels, mode, config):
 165       return dnn_linear_combined_lib._dnn_linear_combined_model_fn(  # pylint: disable=protected-access
 166           features=features,
 167           labels=labels,
 168           mode=mode,
 169           head=head,
 170           linear_feature_columns=linear_feature_columns,
 171           linear_optimizer=linear_optimizer,
 172           dnn_feature_columns=dnn_feature_columns,
 173           dnn_optimizer=dnn_optimizer,
 174           dnn_hidden_units=dnn_hidden_units,
 175           dnn_activation_fn=dnn_activation_fn,
 176           dnn_dropout=dnn_dropout,
 177           input_layer_partitioner=input_layer_partitioner,
 178           config=config,
 179           linear_sparse_combiner=linear_sparse_combiner)
 180 
 181     super(DNNLinearCombinedEstimator, self).__init__(
 182         model_fn=_model_fn, model_dir=model_dir, config=config)
