  40 def multi_head(heads, head_weights=None):
  41   """Creates a `_Head` for multi-objective learning.
  42 
  43   This class merges the output of multiple `_Head` objects.
  44   Specifically:
  45   * For training, sums losses of each head, calls `train_op_fn` with this
  46     final loss.
  47   * For eval, merges metrics by adding `head.name` suffix to the keys in eval
  48     metrics, such as `precision/head1`, `precision/head2`.
  49   * For prediction, merges predictions and updates keys in prediction dict to a
  50     2-tuple, `(head.name, prediction_key)`. Merges `export_outputs` such that
  51     by default the first head is served.
  52 
  53   Usage:
  54 
  55   ```python
  56   # In `input_fn` specify labels as a dict keyed by head name:
  57   def input_fn():
  58     features = ...
  59     labels1 = ...
  60     labels2 = ...
  61     return features, {'head1': labels1, 'head2': labels2}
  62 
  63   # In `model_fn`, specify logits as a dict keyed by head name:
  64   def model_fn(features, labels, mode):
  65     # Create simple heads and specify head name.
  66     head1 = multi_class_head(n_classes=3, name='head1')
  67     head2 = binary_classification_head(name='head2')
  68     # Create multi-head from two simple heads.
  69     head = multi_head([head1, head2])
  70     # Create logits for each head, and combine them into a dict.
  71     logits1, logits2 = logit_fn()
  72     logits = {'head1': logits1, 'head2': logits2}
  73     # Return the merged EstimatorSpec
  74     return head.create_estimator_spec(..., logits=logits, ...)
  75 
  76   # Create an estimator with this model_fn.
  77   estimator = tf.estimator.Estimator(model_fn=model_fn)
  78   estimator.train(input_fn=input_fn, steps=100)
  79   ```
  80 
  81   Also supports `logits` as a `Tensor` of shape
  82   `[D0, D1, ... DN, logits_dimension]`. It will split the `Tensor` along the
  83   last dimension and distribute it appropriately among the heads. E.g.:
  84 
  85   ```python
  86   def model_fn(features, labels, mode):
  87     # Create simple heads and specify head name.
  88     head1 = multi_class_head(n_classes=3, name='head1')
  89     head2 = binary_classification_head(name='head2')
  90     # Create multi-head from two simple heads.
  91     head = multi_head([head1, head2])
  92     # Create logits for the multihead.
  93     logits = logit_fn(logits_dimension=head.logits_dimension)
  94     # Return the merged EstimatorSpec
  95     return head.create_estimator_spec(..., logits=logits, ...)
  96   ```
  97 
  98   Args:
  99     heads: List or tuple of `_Head` instances. All heads must have `name`
 100       specified. The first head in the list is the default used at serving time.
 101     head_weights: Optional list of weights, same length as `heads`. Used when
 102       merging losses to calculate the weighted sum of losses from each head. If
 103       `None`, all losses are weighted equally.
 104 
 105   Returns:
 106     A instance of `_Head` that merges multiple heads.
 107 
 108   Raises:
 109     ValueError: If `heads` is empty.
 110     ValueError: If any of the `heads` does not have `name` specified.
 111     ValueError: If `heads` and `head_weights` have different size.
 112   """
 113   if head_weights:
 114     if len(head_weights) != len(heads):
 115       raise ValueError(
 116           'heads and head_weights must have the same size. '
 117           'Given len(heads): {}. Given len(head_weights): {}.'.format(
 118               len(heads), len(head_weights)))
 119   if not heads:
 120     raise ValueError('Must specify heads. Given: {}'.format(heads))
 121   for head in heads:
 122     if not head.name:
 123       raise ValueError(
 124           'All given heads must have name specified. '
 125           'Given: {}'.format(head))
 126 
 127   return _MultiHead(
 128       heads=tuple(heads),
 129       head_weights=tuple(head_weights) if head_weights else tuple())

 132 def _no_op_train_fn(loss):
 133   del loss
 134   return control_flow_ops.no_op()

 137 def _merge_losses(losses, head_weights=None):
 138   """Merges the given losses into one tensor."""
 139   losses = tuple(losses)
 140   with ops.name_scope(
 141       'merge_losses', values=losses + (head_weights or tuple())):
 142     if head_weights:
 143       weighted_losses = []
 144       for loss, weight in zip(losses, head_weights):
 145         weighted_losses.append(math_ops.multiply(loss, weight))
 146     else:
 147       weighted_losses = losses
 148     return math_ops.add_n(weighted_losses)

 151 def _default_export_output(export_outputs, head_name):
 152   """Extracts the default export output from the given export_outputs dict."""
 153   if len(export_outputs) == 1:
 154     return next(six.itervalues(export_outputs))
 155   for k, v in six.iteritems(export_outputs):
 156     if k == _DEFAULT_SERVING_KEY:
 157       return v
 158   raise ValueError(
 159       '{} did not specify default export_outputs. '
 160       'Given: {} '
 161       'Suggested fix: Use one of the heads in tf.contrib.estimator, or include '
 162       'key {} in export_outputs.'.format(
 163           head_name, export_outputs, _DEFAULT_SERVING_KEY))

 166 class _MultiHead(head_lib._Head):  # pylint:disable=protected-access
 167   """`_Head` for multi objective learning."""
 168 
 169   def __init__(self, heads, head_weights):
 170     self._logits_dimension = 0
 171     for head in heads:
 172       self._logits_dimension += head.logits_dimension
 173 
 174     self._heads = heads
 175     self._head_weights = head_weights
 176 
 177   @property
 178   def name(self):
 179     return '_'.join([h.name for h in self._heads])
 180 
 181   @property
 182   def logits_dimension(self):
 183     return self._logits_dimension
 184 
 185   def create_loss(self, features, mode, logits, labels):
 186     """See `Head`."""
 187     if isinstance(logits, dict):
 188       logits_dict = logits
 189     else:
 190       logits_dict = self._split_logits(logits)
 191     training_losses = []
 192     labels_by_head = {}
 193     unreduced_losses_by_head = {}
 194     example_weights_by_head = {}
 195     for i, head in enumerate(self._heads):
 196       (training_loss, unreduced_loss,
 197        weights, processed_labels) = head.create_loss(
 198            features, mode, logits_dict[head.name], labels[head.name])
 199       training_losses.append(training_loss)
 200       labels_by_head[head.name] = processed_labels
 201       if self._head_weights:
 202         head_weight = self._head_weights[i]
 203         unreduced_losses_by_head[head.name] = math_ops.multiply(
 204             unreduced_loss, head_weight)
 205         example_weights_by_head[head.name] = math_ops.multiply(
 206             weights, head_weight)
 207       else:
 208         unreduced_losses_by_head[head.name] = unreduced_loss
 209         example_weights_by_head[head.name] = weights
 210 
 211     training_losses = tuple(training_losses)
 212     with ops.name_scope(
 213         'merge_losses',
 214         values=training_losses + (self._head_weights or tuple())):
 215       if self._head_weights:
 216         head_weighted_training_losses = []
 217         for training_loss, head_weight in zip(
 218             training_losses, self._head_weights):
 219           head_weighted_training_losses.append(
 220               math_ops.multiply(training_loss, head_weight))
 221         merged_training_loss = math_ops.add_n(head_weighted_training_losses)
 222       else:
 223         merged_training_loss = math_ops.add_n(training_losses)
 224 
 225     return head_lib.LossSpec(
 226         training_loss=merged_training_loss,
 227         unreduced_loss=unreduced_losses_by_head,
 228         weights=example_weights_by_head,
 229         processed_labels=labels_by_head)
 230 
 231   # TODO(b/65403806): Support regularization_losses arg.
 232   def create_estimator_spec(
 233       self, features, mode, logits, labels=None, optimizer=None,
 234       train_op_fn=None):
 235     """See `_Head`."""
 236     if isinstance(logits, dict):
 237       logits_dict = logits
 238     else:
 239       logits_dict = self._split_logits(logits)
 240     if labels and not isinstance(labels, dict):
 241       raise ValueError('labels must be a dict. Given: {}'.format(labels))
 242 
 243     all_estimator_spec = []
 244     for head in self._heads:
 245       head_name = head.name
 246       all_estimator_spec.append(
 247           head.create_estimator_spec(
 248               features=features,
 249               mode=mode,
 250               logits=logits_dict[head_name],
 251               labels=labels[head_name] if labels else None,
 252               train_op_fn=_no_op_train_fn))
 253 
 254     if mode == model_fn.ModeKeys.TRAIN:
 255       spec = self._merge_train(
 256           all_estimator_spec=all_estimator_spec,
 257           optimizer=optimizer,
 258           train_op_fn=train_op_fn)
 259       with ops.name_scope(''):
 260         summary.scalar(metric_keys.MetricKeys.LOSS, spec.loss)
 261       return spec
 262     if mode == model_fn.ModeKeys.PREDICT:
 263       return self._merge_predict(all_estimator_spec)
 264     if mode == model_fn.ModeKeys.EVAL:
 265       return self._merge_eval(all_estimator_spec)
 266     raise ValueError('mode={} unrecognized'.format(mode))
 267 
 268   def _split_logits(self, logits):
 269     """Splits logits along the last dimension and returns a dict."""
 270     logits_dict = {}
 271     with ops.name_scope(None, 'split_logits', values=[logits]):
 272       logits = ops.convert_to_tensor(logits)
 273       batch_shape = array_ops.shape(logits)[:-1]
 274       zeros_like_batch_shape = array_ops.zeros_like(batch_shape)
 275       minus_ones_like_batch_shape = -1 * array_ops.ones_like(batch_shape)
 276       begin_idx = 0
 277       for head in self._heads:
 278         begin_tensor = array_ops.concat(
 279             [zeros_like_batch_shape, [begin_idx]], axis=0)
 280         size_tensor = array_ops.concat(
 281             [minus_ones_like_batch_shape, [head.logits_dimension]], axis=0)
 282         logits_dict[head.name] = array_ops.slice(
 283             logits, begin=begin_tensor, size=size_tensor)
 284         begin_idx += head.logits_dimension
 285     return logits_dict

 287   def _merge_train(self, all_estimator_spec, optimizer, train_op_fn):
 288     """Merges list of `EstimatorSpec` for training.
 289 
 290     Args:
 291       all_estimator_spec: list of `EstimatorSpec` for the individual heads.
 292       optimizer: `Optimizer` instance to create train op. See
 293         `create_estimator_spec` documentation for more details.
 294       train_op_fn: Function to create train op. Used if `optimizer` is `None`.
 295 
 296     Returns:
 297       `EstimatorSpec` that merges all heads for TRAIN.
 298 
 299     Raises:
 300       ValueError: If both `train_op_fn` and `optimizer` are `None` in TRAIN
 301         mode.
 302     """
 303     losses = []
 304     metrics = {}
 305     for spec in all_estimator_spec:
 306       losses.append(spec.loss)
 307       # Metric keys already contain head.name.
 308       metrics.update(spec.eval_metric_ops or {})
 309     loss = _merge_losses(losses, self._head_weights)
 310     if optimizer is not None:
 311       if train_op_fn is not None:
 312         raise ValueError('train_op_fn and optimizer cannot both be set.')
 313       train_op = optimizer.minimize(
 314           loss, global_step=training_util.get_global_step())
 315     elif train_op_fn is not None:
 316       train_op = train_op_fn(loss)
 317     else:
 318       raise ValueError('train_op_fn and optimizer cannot both be None.')
 319 
 320     return model_fn.EstimatorSpec(
 321         mode=model_fn.ModeKeys.TRAIN,
 322         loss=loss,
 323         train_op=train_op,
 324         eval_metric_ops=metrics)
 325 
 326   def _merge_predict(self, all_estimator_spec):
 327     """Merges list of `EstimatorSpec` for prediction.
 328 
 329     Args:
 330       all_estimator_spec: list of `EstimatorSpec` for the individual heads.
 331 
 332     Returns:
 333       `EstimatorSpec` that merges all heads for PREDICT.
 334     """
 335     predictions = {}
 336     export_outputs = {
 337         _DEFAULT_SERVING_KEY: _default_export_output(
 338             all_estimator_spec[0].export_outputs,
 339             self._heads[0].name),
 340     }
 341     merged_predict_outputs = {}
 342     for head, spec in zip(self._heads, all_estimator_spec):
 343       head_name = head.name
 344       for k, v in six.iteritems(spec.export_outputs):
 345         if k == _DEFAULT_SERVING_KEY:
 346           key = head_name
 347         else:
 348           key = '%s/%s' % (head_name, k)
 349         export_outputs[key] = v
 350         if (k == head_lib._PREDICT_SERVING_KEY and  # pylint:disable=protected-access
 351             isinstance(v, export_output_lib.PredictOutput)):
 352           for kp, vp in six.iteritems(v.outputs):
 353             key = '%s/%s' % (head_name, kp)
 354             merged_predict_outputs[key] = vp
 355       for k, v in six.iteritems(spec.predictions):
 356         predictions[(head_name, k)] = v
 357     export_outputs[head_lib._PREDICT_SERVING_KEY] = (  # pylint:disable=protected-access
 358         export_output_lib.PredictOutput(merged_predict_outputs))
 359 
 360     return model_fn.EstimatorSpec(
 361         mode=model_fn.ModeKeys.PREDICT,
 362         predictions=predictions,
 363         export_outputs=export_outputs)
 364 
 365   def _merge_eval(self, all_estimator_spec):
 366     """Merges list of `EstimatorSpec` for eval.
 367 
 368     Args:
 369       all_estimator_spec: list of `EstimatorSpec` for the individual heads.
 370 
 371     Returns:
 372       `EstimatorSpec` that merges all heads for EVAL.
 373     """
 374     predictions = {}
 375     metrics = {}
 376     losses = []
 377     with ops.name_scope('merge_eval'):
 378       for head, spec in zip(self._heads, all_estimator_spec):
 379         losses.append(spec.loss)
 380         head_name = head.name
 381         # Loss metric is not added by default.
 382         loss_name = head_lib._summary_key(  # pylint:disable=protected-access
 383             head_name, metric_keys.MetricKeys.LOSS)
 384         metrics[loss_name] = metrics_lib.mean(spec.loss, name=loss_name)
 385         # Metric keys already contain head.name.
 386         metrics.update(spec.eval_metric_ops or {})
 387         for k, v in six.iteritems(spec.predictions):
 388           predictions[(head_name, k)] = v
 389       loss = _merge_losses(losses, self._head_weights)
 390 
 391     return model_fn.EstimatorSpec(
 392         mode=model_fn.ModeKeys.EVAL,
 393         predictions=predictions,
 394         loss=loss,
 395         eval_metric_ops=metrics)
