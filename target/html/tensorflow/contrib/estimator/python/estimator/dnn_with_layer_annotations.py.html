  35 class LayerAnnotationsCollectionNames(object):
  36   """Names for the collections containing the annotations."""
  37 
  38   UNPROCESSED_FEATURES = 'layer_annotations/unprocessed_features'
  39   PROCESSED_FEATURES = 'layer_annotatons/processed_features'

  51 def serialize_feature_column(feature_column):
  52   if isinstance(feature_column, feature_column_lib._EmbeddingColumn):  # pylint: disable=protected-access
  53     # We can't pickle nested functions, and we don't need the value of
  54     # layer_creator in most cases anyway, so just discard its value.
  55     args = feature_column._asdict()
  56     args['layer_creator'] = None
  57     temp = type(feature_column)(**args)
  58     return pickle.dumps(temp)
  59   return pickle.dumps(feature_column)

  62 def _to_any_wrapped_tensor_info(tensor):
  63   """Converts a `Tensor` to a `TensorInfo` wrapped in a proto `Any`."""
  64   any_buf = Any()
  65   tensor_info = saved_model_utils.build_tensor_info(tensor)
  66   any_buf.Pack(tensor_info)
  67   return any_buf

  70 def make_input_layer_with_layer_annotations(original_input_layer):
  71   """Make an input_layer replacement function that adds layer annotations."""
  72 
  73   def input_layer_with_layer_annotations(features,
  74                                          feature_columns,
  75                                          weight_collections=None,
  76                                          trainable=True,
  77                                          cols_to_vars=None,
  78                                          scope=None,
  79                                          cols_to_output_tensors=None,
  80                                          from_template=False):
  81     """Returns a dense `Tensor` as input layer based on given `feature_columns`.
  82 
  83     Generally a single example in training data is described with
  84     FeatureColumns.
  85     At the first layer of the model, this column oriented data should be
  86     converted
  87     to a single `Tensor`.
  88 
  89     This is like tf.feature_column.input_layer, except with added
  90     Integrated-Gradient annotations.
  91 
  92     Args:
  93       features: A mapping from key to tensors. `_FeatureColumn`s look up via
  94         these keys. For example `numeric_column('price')` will look at 'price'
  95         key in this dict. Values can be a `SparseTensor` or a `Tensor` depends
  96         on corresponding `_FeatureColumn`.
  97       feature_columns: An iterable containing the FeatureColumns to use as
  98         inputs to your model. All items should be instances of classes derived
  99         from `_DenseColumn` such as `numeric_column`, `embedding_column`,
 100         `bucketized_column`, `indicator_column`. If you have categorical
 101         features, you can wrap them with an `embedding_column` or
 102         `indicator_column`.
 103       weight_collections: A list of collection names to which the Variable will
 104         be added. Note that variables will also be added to collections
 105         `tf.GraphKeys.GLOBAL_VARIABLES` and `ops.GraphKeys.MODEL_VARIABLES`.
 106       trainable: If `True` also add the variable to the graph collection
 107         `GraphKeys.TRAINABLE_VARIABLES` (see `tf.Variable`).
 108       cols_to_vars: If not `None`, must be a dictionary that will be filled with
 109         a mapping from `_FeatureColumn` to list of `Variable`s.  For example,
 110         after the call, we might have cols_to_vars = {_EmbeddingColumn(
 111         categorical_column=_HashedCategoricalColumn( key='sparse_feature',
 112         hash_bucket_size=5, dtype=tf.string), dimension=10): [<tf.Variable
 113         'some_variable:0' shape=(5, 10), <tf.Variable 'some_variable:1'
 114           shape=(5, 10)]} If a column creates no variables, its value will be an
 115           empty list.
 116       scope: A name or variable scope to use
 117       cols_to_output_tensors: If not `None`, must be a dictionary that will be
 118         filled with a mapping from '_FeatureColumn' to the associated output
 119         `Tensor`s.
 120       from_template: True if the method is being instantiated from a
 121         `make_template`.
 122 
 123     Returns:
 124       A `Tensor` which represents input layer of a model. Its shape
 125       is (batch_size, first_layer_dimension) and its dtype is `float32`.
 126       first_layer_dimension is determined based on given `feature_columns`.
 127 
 128     Raises:
 129       ValueError: features and feature_columns have different lengths.
 130     """
 131 
 132     local_cols_to_output_tensors = {}
 133     input_layer = original_input_layer(
 134         features=features,
 135         feature_columns=feature_columns,
 136         weight_collections=weight_collections,
 137         trainable=trainable,
 138         cols_to_vars=cols_to_vars,
 139         scope=scope,
 140         cols_to_output_tensors=local_cols_to_output_tensors,
 141         from_template=from_template)
 142 
 143     if cols_to_output_tensors is not None:
 144       cols_to_output_tensors = local_cols_to_output_tensors
 145 
 146     # Annotate features.
 147     # These are the parsed Tensors, before embedding.
 148 
 149     # Only annotate features used by FeatureColumns.
 150     # We figure which ones are used by FeatureColumns by creating a parsing
 151     # spec and looking at the keys.
 152     spec = feature_column_lib.make_parse_example_spec(feature_columns)
 153     for key in spec.keys():
 154       tensor = ops.convert_to_tensor(features[key])
 155       ops.add_to_collection(
 156           LayerAnnotationsCollectionNames.keys(
 157               LayerAnnotationsCollectionNames.UNPROCESSED_FEATURES), key)
 158       ops.add_to_collection(
 159           LayerAnnotationsCollectionNames.values(
 160               LayerAnnotationsCollectionNames.UNPROCESSED_FEATURES),
 161           _to_any_wrapped_tensor_info(tensor))
 162 
 163     # Annotate feature columns.
 164     for column in feature_columns:
 165       # TODO(cyfoo): Find a better way to serialize and deserialize
 166       # _FeatureColumn.
 167       ops.add_to_collection(LayerAnnotationsCollectionNames.FEATURE_COLUMNS,
 168                             serialize_feature_column(column))
 169 
 170     for column, tensor in local_cols_to_output_tensors.items():
 171       ops.add_to_collection(
 172           LayerAnnotationsCollectionNames.keys(
 173               LayerAnnotationsCollectionNames.PROCESSED_FEATURES), column.name)
 174       ops.add_to_collection(
 175           LayerAnnotationsCollectionNames.values(
 176               LayerAnnotationsCollectionNames.PROCESSED_FEATURES),
 177           _to_any_wrapped_tensor_info(tensor))
 178 
 179     return input_layer
 180 
 181   return input_layer_with_layer_annotations

 184 @contextlib.contextmanager
 185 def _monkey_patch(module, function, replacement):
 186   old_function = getattr(module, function)
 187   setattr(module, function, replacement)
 188   yield
 189   setattr(module, function, old_function)

 192 def DNNClassifierWithLayerAnnotations(  # pylint: disable=invalid-name
 193     hidden_units,
 194     feature_columns,
 195     model_dir=None,
 196     n_classes=2,
 197     weight_column=None,
 198     label_vocabulary=None,
 199     optimizer='Adagrad',
 200     activation_fn=nn.relu,
 201     dropout=None,
 202     input_layer_partitioner=None,
 203     config=None,
 204     warm_start_from=None,
 205     loss_reduction=losses.Reduction.SUM):
 206   """A classifier for TensorFlow DNN models with layer annotations.
 207 
 208   This classifier is fuctionally identical to estimator.DNNClassifier as far as
 209   training and evaluating models is concerned. The key difference is that this
 210   classifier adds additional layer annotations, which can be used for computing
 211   Integrated Gradients.
 212 
 213   Integrated Gradients is a method for attributing a classifier's predictions
 214   to its input features (https://arxiv.org/pdf/1703.01365.pdf). Given an input
 215   instance, the method assigns attribution scores to individual features in
 216   proportion to the feature's importance to the classifier's prediction.
 217 
 218   See estimator.DNNClassifer for example code for training and evaluating models
 219   using this classifier.
 220 
 221   This classifier is checkpoint-compatible with estimator.DNNClassifier and
 222   therefore the following should work seamlessly:
 223 
 224   # Instantiate ordinary estimator as usual.
 225   estimator = tf.estimator.DNNClassifier(
 226     config, feature_columns, hidden_units, ...)
 227 
 228   # Train estimator, export checkpoint.
 229   tf.estimator.train_and_evaluate(estimator, ...)
 230 
 231   # Instantiate estimator with annotations with the same configuration as the
 232   # ordinary estimator.
 233   estimator_with_annotations = (
 234     tf.contrib.estimator.DNNClassifierWithLayerAnnotations(
 235       config, feature_columns, hidden_units, ...))
 236 
 237   # Call export_savedmodel with the same arguments as the ordinary estimator,
 238   # using the checkpoint produced for the ordinary estimator.
 239   estimator_with_annotations.export_saved_model(
 240     export_dir_base, serving_input_receiver, ...
 241     checkpoint_path='/path/to/ordinary/estimator/checkpoint/model.ckpt-1234')
 242 
 243   Args:
 244     hidden_units: Iterable of number hidden units per layer. All layers are
 245       fully connected. Ex. `[64, 32]` means first layer has 64 nodes and second
 246       one has 32.
 247     feature_columns: An iterable containing all the feature columns used by the
 248       model. All items in the set should be instances of classes derived from
 249       `_FeatureColumn`.
 250     model_dir: Directory to save model parameters, graph and etc. This can also
 251       be used to load checkpoints from the directory into a estimator to
 252       continue training a previously saved model.
 253     n_classes: Number of label classes. Defaults to 2, namely binary
 254       classification. Must be > 1.
 255     weight_column: A string or a `_NumericColumn` created by
 256       `tf.feature_column.numeric_column` defining feature column representing
 257       weights. It is used to down weight or boost examples during training. It
 258       will be multiplied by the loss of the example. If it is a string, it is
 259       used as a key to fetch weight tensor from the `features`. If it is a
 260       `_NumericColumn`, raw tensor is fetched by key `weight_column.key`, then
 261       weight_column.normalizer_fn is applied on it to get weight tensor.
 262     label_vocabulary: A list of strings represents possible label values. If
 263       given, labels must be string type and have any value in
 264       `label_vocabulary`. If it is not given, that means labels are already
 265       encoded as integer or float within [0, 1] for `n_classes=2` and encoded as
 266       integer values in {0, 1,..., n_classes-1} for `n_classes`>2 . Also there
 267       will be errors if vocabulary is not provided and labels are string.
 268     optimizer: An instance of `tf.Optimizer` used to train the model. Defaults
 269       to Adagrad optimizer.
 270     activation_fn: Activation function applied to each layer. If `None`, will
 271       use `tf.nn.relu`.
 272     dropout: When not `None`, the probability we will drop out a given
 273       coordinate.
 274     input_layer_partitioner: Optional. Partitioner for input layer. Defaults to
 275       `min_max_variable_partitioner` with `min_slice_size` 64 << 20.
 276     config: `RunConfig` object to configure the runtime settings.
 277     warm_start_from: A string filepath to a checkpoint to warm-start from, or a
 278       `WarmStartSettings` object to fully configure warm-starting.  If the
 279       string filepath is provided instead of a `WarmStartSettings`, then all
 280       weights are warm-started, and it is assumed that vocabularies and Tensor
 281       names are unchanged.
 282     loss_reduction: One of `tf.losses.Reduction` except `NONE`. Describes how to
 283       reduce training loss over batch. Defaults to `SUM`.
 284 
 285   Returns:
 286     DNNClassifier with layer annotations.
 287   """
 288 
 289   original = dnn.DNNClassifier(
 290       hidden_units=hidden_units,
 291       feature_columns=feature_columns,
 292       model_dir=model_dir,
 293       n_classes=n_classes,
 294       weight_column=weight_column,
 295       label_vocabulary=label_vocabulary,
 296       optimizer=optimizer,
 297       activation_fn=activation_fn,
 298       dropout=dropout,
 299       input_layer_partitioner=input_layer_partitioner,
 300       config=config,
 301       warm_start_from=warm_start_from,
 302       loss_reduction=loss_reduction)
 303 
 304   def _model_fn(features, labels, mode, config):
 305     with _monkey_patch(
 306         feature_column_lib, '_internal_input_layer',
 307         make_input_layer_with_layer_annotations(
 308             feature_column_lib._internal_input_layer)):  # pylint: disable=protected-access
 309       return original.model_fn(features, labels, mode, config)
 310 
 311   return estimator.Estimator(
 312       model_fn=_model_fn,
 313       model_dir=model_dir,
 314       config=config,
 315       warm_start_from=warm_start_from)

 318 def DNNRegressorWithLayerAnnotations(  # pylint: disable=invalid-name
 319     hidden_units,
 320     feature_columns,
 321     model_dir=None,
 322     label_dimension=1,
 323     weight_column=None,
 324     optimizer='Adagrad',
 325     activation_fn=nn.relu,
 326     dropout=None,
 327     input_layer_partitioner=None,
 328     config=None,
 329     warm_start_from=None,
 330     loss_reduction=losses.Reduction.SUM,
 331 ):
 332   """A regressor for TensorFlow DNN models with layer annotations.
 333 
 334   This regressor is fuctionally identical to estimator.DNNRegressor as far as
 335   training and evaluating models is concerned. The key difference is that this
 336   classifier adds additional layer annotations, which can be used for computing
 337   Integrated Gradients.
 338 
 339   Integrated Gradients is a method for attributing a classifier's predictions
 340   to its input features (https://arxiv.org/pdf/1703.01365.pdf). Given an input
 341   instance, the method assigns attribution scores to individual features in
 342   proportion to the feature's importance to the classifier's prediction.

 355   tf.estimator.train_and_evaluate(estimator, ...)
 356 
 357   # Instantiate estimator with annotations with the same configuration as the
 358   # ordinary estimator.
 359   estimator_with_annotations = (
 360     tf.contrib.estimator.DNNRegressorWithLayerAnnotations(
 361       config, feature_columns, hidden_units, ...))
 362 
 363   # Call export_savedmodel with the same arguments as the ordinary estimator,
 364   # using the checkpoint produced for the ordinary estimator.

 366     export_dir_base, serving_input_receiver, ...

 370     hidden_units: Iterable of number hidden units per layer. All layers are
 371       fully connected. Ex. `[64, 32]` means first layer has 64 nodes and second
 372       one has 32.
 373     feature_columns: An iterable containing all the feature columns used by the
 374       model. All items in the set should be instances of classes derived from
 375       `_FeatureColumn`.

 377       be used to load checkpoints from the directory into a estimator to
 378       continue training a previously saved model.
 379     label_dimension: Number of regression targets per example. This is the size
 380       of the last dimension of the labels and logits `Tensor` objects
 381       (typically, these have shape `[batch_size, label_dimension]`).

 387       `_NumericColumn`, raw tensor is fetched by key `weight_column.key`, then
 388       weight_column.normalizer_fn is applied on it to get weight tensor.
 389     optimizer: An instance of `tf.Optimizer` used to train the model. Defaults
 390       to Adagrad optimizer.
 391     activation_fn: Activation function applied to each layer. If `None`, will
 392       use `tf.nn.relu`.
 393     dropout: When not `None`, the probability we will drop out a given
 394       coordinate.
 395     input_layer_partitioner: Optional. Partitioner for input layer. Defaults to
 396       `min_max_variable_partitioner` with `min_slice_size` 64 << 20.
 397     config: `RunConfig` object to configure the runtime settings.
 398     warm_start_from: A string filepath to a checkpoint to warm-start from, or a
 399       `WarmStartSettings` object to fully configure warm-starting.  If the
 400       string filepath is provided instead of a `WarmStartSettings`, then all
 401       weights are warm-started, and it is assumed that vocabularies and Tensor
 402       names are unchanged.
 403     loss_reduction: One of `tf.losses.Reduction` except `NONE`. Describes how to
 404       reduce training loss over batch. Defaults to `SUM`.
 405 
 406   Returns:
 407     DNNRegressor with layer annotations.
 408   """
 409 
 410   original = dnn.DNNRegressor(
 411       hidden_units=hidden_units,
 412       feature_columns=feature_columns,
 413       model_dir=model_dir,
 414       label_dimension=label_dimension,
 415       weight_column=weight_column,
 416       optimizer=optimizer,
 417       activation_fn=activation_fn,
 418       dropout=dropout,
 419       input_layer_partitioner=input_layer_partitioner,
 420       config=config,
 421       warm_start_from=warm_start_from,
 422       loss_reduction=loss_reduction,
 423   )
 424 
 425   def _model_fn(features, labels, mode, config):
 426     with _monkey_patch(
 427         feature_column_lib, '_internal_input_layer',
 428         make_input_layer_with_layer_annotations(
 429             feature_column_lib._internal_input_layer)):  # pylint: disable=protected-access

 432   return estimator.Estimator(
 433       model_fn=_model_fn,
 434       model_dir=model_dir,
 435       config=config,
 436       warm_start_from=warm_start_from)
