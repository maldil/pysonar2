  34 class InMemoryEvaluatorHook(training.SessionRunHook):
  35   """Hook to run evaluation in training without a checkpoint.
  36 
  37   Example:
  38 
  39   ```python
  40   def train_input_fn():
  41     ...
  42     return train_dataset
  43 
  44   def eval_input_fn():
  45     ...
  46     return eval_dataset
  47 
  48   estimator = tf.estimator.DNNClassifier(...)
  49 
  50   evaluator = tf.contrib.estimator.InMemoryEvaluatorHook(
  51       estimator, eval_input_fn)
  52   estimator.train(train_input_fn, hooks=[evaluator])
  53   ```
  54 
  55   Current limitations of this approach are:
  56 
  57   * It doesn't support multi-node distributed mode.
  58   * It doesn't support saveable objects other than variables (such as boosted
  59     tree support)
  60   * It doesn't support custom saver logic (such as ExponentialMovingAverage
  61     support)
  62 
  63   """
  64 
  65   def __init__(self,
  66                estimator,
  67                input_fn,
  68                steps=None,
  69                hooks=None,
  70                name=None,
  71                every_n_iter=100):
  72     """Initializes a `InMemoryEvaluatorHook`.
  73 
  74     Args:
  75       estimator: A `tf.estimator.Estimator` instance to call evaluate.
  76       input_fn:  Equivalent to the `input_fn` arg to `estimator.evaluate`. A
  77         function that constructs the input data for evaluation.
  78         See [Createing input functions](
  79         https://tensorflow.org/guide/premade_estimators#create_input_functions)
  80         for more information. The function should construct and return one of
  81         the following:
  82 
  83           * A 'tf.data.Dataset' object: Outputs of `Dataset` object must be a
  84             tuple (features, labels) with same constraints as below.
  85           * A tuple (features, labels): Where `features` is a `Tensor` or a
  86             dictionary of string feature name to `Tensor` and `labels` is a
  87             `Tensor` or a dictionary of string label name to `Tensor`. Both
  88             `features` and `labels` are consumed by `model_fn`. They should
  89             satisfy the expectation of `model_fn` from inputs.
  90 
  91       steps: Equivalent to the `steps` arg to `estimator.evaluate`.  Number of
  92         steps for which to evaluate model. If `None`, evaluates until `input_fn`
  93         raises an end-of-input exception.
  94       hooks: Equivalent to the `hooks` arg to `estimator.evaluate`. List of
  95         `SessionRunHook` subclass instances. Used for callbacks inside the
  96         evaluation call.
  97       name:  Equivalent to the `name` arg to `estimator.evaluate`. Name of the
  98         evaluation if user needs to run multiple evaluations on different data
  99         sets, such as on training data vs test data. Metrics for different
 100         evaluations are saved in separate folders, and appear separately in
 101         tensorboard.
 102       every_n_iter: `int`, runs the evaluator once every N training iteration.
 103 
 104     Raises:
 105       ValueError: if `every_n_iter` is non-positive or it's not a single machine
 106         training
 107     """
 108     if every_n_iter is None or every_n_iter <= 0:
 109       raise ValueError('invalid every_n_iter=%s.' % every_n_iter)
 110     if (estimator.config.num_ps_replicas > 0 or
 111         estimator.config.num_worker_replicas > 1):
 112       raise ValueError(
 113           'InMemoryEvaluator supports only single machine (aka Local) setting.')
 114     self._estimator = estimator
 115     self._input_fn = input_fn
 116     self._steps = steps
 117     self._name = name
 118     self._every_n_iter = every_n_iter
 119     self._eval_dir = os.path.join(self._estimator.model_dir, 'eval'
 120                                   if not name else 'eval_' + name)
 121 
 122     self._graph = None
 123     self._hooks = estimator_lib._check_hooks_type(hooks)
 124     self._hooks.extend(self._estimator._convert_eval_steps_to_hooks(steps))
 125     self._timer = training.SecondOrStepTimer(every_steps=every_n_iter)
 126 
 127   def begin(self):
 128     """Build eval graph and restoring op."""
 129     self._timer.reset()
 130     self._iter_count = 0
 131     self._graph = ops.Graph()
 132     with self._graph.as_default():
 133       (self._scaffold, self._update_op, self._eval_dict,
 134        self._all_hooks) = self._estimator._evaluate_build_graph(
 135            self._input_fn, self._hooks, checkpoint_path=None)
 136 
 137       if self._scaffold.saver is not None:
 138         raise ValueError('InMemoryEvaluator does not support custom saver')
 139       if self._scaffold.init_fn is not None:
 140         raise ValueError('InMemoryEvaluator does not support custom init_fn')
 141 
 142       self._var_name_to_eval_var = {
 143           v.name: v for v in ops.get_collection(ops.GraphKeys.GLOBAL_VARIABLES)
 144       }
 145       self._var_name_to_placeholder = {
 146           v.name: array_ops.placeholder(v.dtype)
 147           for v in ops.get_collection(ops.GraphKeys.GLOBAL_VARIABLES)
 148       }
 149 
 150   def after_create_session(self, session, coord):  # pylint: disable=unused-argument
 151     """Does first run which shows the eval metrics before training."""
 152     if ops.get_collection(ops.GraphKeys.SAVEABLE_OBJECTS):
 153       raise ValueError(
 154           'InMemoryEvaluator does not support saveables other than global '
 155           'variables.')
 156     self._var_name_to_train_var = {
 157         v.name: v for v in ops.get_collection(ops.GraphKeys.GLOBAL_VARIABLES)
 158     }
 159     var_names_to_transfer = set(self._var_name_to_placeholder.keys()) & set(
 160         self._var_name_to_train_var.keys())
 161     # Filter training var names that are not exist in evaluation
 162     self._var_name_to_train_var = {
 163         v_name: self._var_name_to_train_var[v_name]
 164         for v_name in var_names_to_transfer
 165     }
 166     # Filter eval var names that are not exist in training
 167     self._var_name_to_eval_var = {
 168         v_name: self._var_name_to_eval_var[v_name]
 169         for v_name in var_names_to_transfer
 170     }
 171 
 172     with self._graph.as_default():
 173       self._var_feed_op = control_flow_ops.group([
 174           state_ops.assign(self._var_name_to_eval_var[v_name],
 175                            self._var_name_to_placeholder[v_name])
 176           for v_name in var_names_to_transfer
 177       ])
 178 
 179     self._evaluate(session)
 180 
 181   def _evaluate(self, train_session):
 182     var_name_to_value = train_session.run(self._var_name_to_train_var)
 183     placeholder_to_value = {
 184         self._var_name_to_placeholder[v_name]: var_name_to_value[v_name]
 185         for v_name in var_name_to_value
 186     }
 187 
 188     def feed_variables(scaffold, session):
 189       del scaffold
 190       session.run(self._var_feed_op, feed_dict=placeholder_to_value)
 191 
 192     scaffold = training.Scaffold(
 193         init_fn=feed_variables, copy_from_scaffold=self._scaffold)
 194 
 195     with self._graph.as_default():
 196       self._estimator._evaluate_run(
 197           checkpoint_path=None,
 198           scaffold=scaffold,
 199           update_op=self._update_op,
 200           eval_dict=self._eval_dict,
 201           all_hooks=self._all_hooks,
 202           output_dir=self._eval_dir)
 203 
 204     self._timer.update_last_triggered_step(self._iter_count)
 205 
 206   def after_run(self, run_context, run_values):  # pylint: disable=unused-argument
 207     """Runs evaluator."""
 208     self._iter_count += 1
 209     if self._timer.should_trigger_for_step(self._iter_count):
 210       self._evaluate(run_context.session)
 211 
 212   def end(self, session):  # pylint: disable=unused-argument
 213     """Runs evaluator for final model."""
 214     self._evaluate(session)

 217 class _StopAtCheckpointStepHook(training.SessionRunHook):
 218   """Hook that requests stop at a specified step based on checkpoint.
 219 
 220   Note: We recommend using 'make_stop_at_checkpoint_step_hook` to get the proper
 221   hook.
 222   """
 223 
 224   def __init__(self, model_dir, last_step,
 225                wait_after_file_check_secs=30):
 226     """Initializes a `StopAtCheckpointStepHook`.
 227 
 228     This hook requests stop after a last step has been reached. It checks latest
 229     checkpoint to verify last step is written on disk or not.
 230 
 231     Args:
 232       model_dir: Directory to read global step from latest checkpoint.
 233       last_step: Step after which to stop.
 234       wait_after_file_check_secs: Reading same file by many workers may create
 235       I/O issues. To throttle that we will wait given secs after each read of
 236       the file.
 237 
 238     Raises:
 239       ValueError: If one of the arguments is invalid.
 240     """
 241     if last_step is None:
 242       raise ValueError('last_step must be specified.')
 243     if model_dir is None:
 244       raise ValueError('model_dir must be specified.')
 245 
 246     self._model_dir = model_dir
 247     self._last_step = last_step
 248     self._wait_after_file_check_secs = wait_after_file_check_secs
 249 
 250   def begin(self):
 251     self._global_step_tensor = training_util._get_or_create_global_step_read()  # pylint: disable=protected-access
 252     if self._global_step_tensor is None:
 253       raise RuntimeError(
 254           'Global step should be created to use StopAtCheckpointStepHook.')
 255 
 256   def before_run(self, run_context):  # pylint: disable=unused-argument
 257     return training.SessionRunArgs(self._global_step_tensor)
 258 
 259   def after_run(self, run_context, run_values):
 260     global_step = run_values.results + 1
 261     if global_step >= self._last_step:
 262       # Check latest global step in the checkpoint to ensure that the targeted
 263       # last step is written on disk.
 264 
 265       step = estimator_lib._load_global_step_from_checkpoint_dir(
 266           self._model_dir)
 267       if step >= self._last_step:
 268         run_context.request_stop()
 269       else:
 270         time.sleep(self._wait_after_file_check_secs)
 271 
 272 
 273 def make_stop_at_checkpoint_step_hook(estimator,
 274                                       last_step,
 275                                       wait_after_file_check_secs=30):
 276   """Creates a proper StopAtCheckpointStepHook based on chief status."""
 277 
 278   if estimator.config.is_chief:
 279     return training.StopAtStepHook(last_step=last_step)
 280   return _StopAtCheckpointStepHook(
 281       model_dir=estimator.model_dir,
 282       last_step=last_step,
 283       wait_after_file_check_secs=wait_after_file_check_secs)
