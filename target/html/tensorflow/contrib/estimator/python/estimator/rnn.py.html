  59 def _single_rnn_cell(num_units, cell_type):
  60   cell_type = _CELL_TYPES.get(cell_type, cell_type)
  61   if not cell_type or not issubclass(cell_type, rnn_cell.RNNCell):
  62     raise ValueError('Supported cell types are {}; got {}'.format(
  63         list(_CELL_TYPES.keys()), cell_type))
  64   return cell_type(num_units=num_units)

  67 def _make_rnn_cell_fn(num_units, cell_type='basic_rnn'):
  68   """Convenience function to create `rnn_cell_fn` for canned RNN Estimators.
  69 
  70   Args:
  71     num_units: Iterable of integer number of hidden units per RNN layer.
  72     cell_type: A subclass of `tf.nn.rnn_cell.RNNCell` or a string specifying
  73       the cell type. Supported strings are: `'basic_rnn'`, `'lstm'`, and
  74       `'gru'`.
  75 
  76   Returns:
  77     A function that takes a single argument, an instance of
  78     `tf.estimator.ModeKeys`, and returns an instance derived from
  79     `tf.nn.rnn_cell.RNNCell`.
  80 
  81   Raises:
  82     ValueError: If cell_type is not supported.
  83   """
  84   def rnn_cell_fn(mode):
  85     # Unused. Part of the rnn_cell_fn interface since user specified functions
  86     # may need different behavior across modes (e.g. dropout).
  87     del mode
  88     cells = [_single_rnn_cell(n, cell_type) for n in num_units]
  89     if len(cells) == 1:
  90       return cells[0]
  91     return rnn_cell.MultiRNNCell(cells)
  92   return rnn_cell_fn

  95 def _concatenate_context_input(sequence_input, context_input):
  96   """Replicates `context_input` across all timesteps of `sequence_input`.
  97 
  98   Expands dimension 1 of `context_input` then tiles it `sequence_length` times.
  99   This value is appended to `sequence_input` on dimension 2 and the result is
 100   returned.
 101 
 102   Args:
 103     sequence_input: A `Tensor` of dtype `float32` and shape `[batch_size,
 104       padded_length, d0]`.
 105     context_input: A `Tensor` of dtype `float32` and shape `[batch_size, d1]`.
 106 
 107   Returns:
 108     A `Tensor` of dtype `float32` and shape `[batch_size, padded_length,
 109     d0 + d1]`.
 110 
 111   Raises:
 112     ValueError: If `sequence_input` does not have rank 3 or `context_input` does
 113       not have rank 2.
 114   """
 115   seq_rank_check = check_ops.assert_rank(
 116       sequence_input,
 117       3,
 118       message='sequence_input must have rank 3',
 119       data=[array_ops.shape(sequence_input)])
 120   seq_type_check = check_ops.assert_type(
 121       sequence_input,
 122       dtypes.float32,
 123       message='sequence_input must have dtype float32; got {}.'.format(
 124           sequence_input.dtype))
 125   ctx_rank_check = check_ops.assert_rank(
 126       context_input,
 127       2,
 128       message='context_input must have rank 2',
 129       data=[array_ops.shape(context_input)])
 130   ctx_type_check = check_ops.assert_type(
 131       context_input,
 132       dtypes.float32,
 133       message='context_input must have dtype float32; got {}.'.format(
 134           context_input.dtype))
 135   with ops.control_dependencies(
 136       [seq_rank_check, seq_type_check, ctx_rank_check, ctx_type_check]):
 137     padded_length = array_ops.shape(sequence_input)[1]
 138     tiled_context_input = array_ops.tile(
 139         array_ops.expand_dims(context_input, 1),
 140         array_ops.concat([[1], [padded_length], [1]], 0))
 141   return array_ops.concat([sequence_input, tiled_context_input], 2)

 144 def _select_last_activations(activations, sequence_lengths):
 145   """Selects the nth set of activations for each n in `sequence_length`.
 146 
 147   Returns a `Tensor` of shape `[batch_size, k]`. If `sequence_length` is not
 148   `None`, then `output[i, :] = activations[i, sequence_length[i] - 1, :]`. If
 149   `sequence_length` is `None`, then `output[i, :] = activations[i, -1, :]`.
 150 
 151   Args:
 152     activations: A `Tensor` with shape `[batch_size, padded_length, k]`.
 153     sequence_lengths: A `Tensor` with shape `[batch_size]` or `None`.
 154   Returns:
 155     A `Tensor` of shape `[batch_size, k]`.
 156   """
 157   with ops.name_scope(
 158       'select_last_activations', values=[activations, sequence_lengths]):
 159     activations_shape = array_ops.shape(activations)
 160     batch_size = activations_shape[0]
 161     padded_length = activations_shape[1]
 162     output_units = activations_shape[2]
 163     if sequence_lengths is None:
 164       sequence_lengths = padded_length
 165     start_indices = math_ops.to_int64(
 166         math_ops.range(batch_size) * padded_length)
 167     last_indices = start_indices + sequence_lengths - 1
 168     reshaped_activations = array_ops.reshape(
 169         activations, [batch_size * padded_length, output_units])
 170 
 171     last_activations = array_ops.gather(reshaped_activations, last_indices)
 172     last_activations.set_shape([activations.shape[0], activations.shape[2]])
 173     return last_activations

 176 def _rnn_logit_fn_builder(output_units, rnn_cell_fn, sequence_feature_columns,
 177                           context_feature_columns, input_layer_partitioner):
 178   """Function builder for a rnn logit_fn.
 179 
 180   Args:
 181     output_units: An int indicating the dimension of the logit layer.
 182     rnn_cell_fn: A function with one argument, a `tf.estimator.ModeKeys`, and
 183       returns an object of type `tf.nn.rnn_cell.RNNCell`.
 184     sequence_feature_columns: An iterable containing the `FeatureColumn`s
 185       that represent sequential input.
 186     context_feature_columns: An iterable containing the `FeatureColumn`s
 187       that represent contextual input.
 188     input_layer_partitioner: Partitioner for input layer.
 189 
 190   Returns:
 191     A logit_fn (see below).
 192 
 193   Raises:
 194     ValueError: If output_units is not an int.
 195   """
 196   if not isinstance(output_units, int):
 197     raise ValueError('output_units must be an int.  Given type: {}'.format(
 198         type(output_units)))
 199 
 200   def rnn_logit_fn(features, mode):
 201     """Recurrent Neural Network logit_fn.
 202 
 203     Args:
 204       features: This is the first item returned from the `input_fn`
 205                 passed to `train`, `evaluate`, and `predict`. This should be a
 206                 single `Tensor` or `dict` of same.
 207       mode: Optional. Specifies if this training, evaluation or prediction. See
 208             `ModeKeys`.
 209 
 210     Returns:
 211       A `Tensor` representing the logits.
 212     """
 213     with variable_scope.variable_scope(
 214         'sequence_input_layer',
 215         values=tuple(six.itervalues(features)),
 216         partitioner=input_layer_partitioner):
 217       sequence_input, sequence_length = seq_fc.sequence_input_layer(
 218           features=features, feature_columns=sequence_feature_columns)
 219       summary.histogram('sequence_length', sequence_length)
 220 
 221       if context_feature_columns:
 222         context_input = feature_column_lib.input_layer(
 223             features=features,
 224             feature_columns=context_feature_columns)
 225         sequence_input = _concatenate_context_input(sequence_input,
 226                                                     context_input)
 227 
 228     cell = rnn_cell_fn(mode)
 229     # Ignore output state.
 230     rnn_outputs, _ = rnn.dynamic_rnn(
 231         cell=cell,
 232         inputs=sequence_input,
 233         sequence_length=sequence_length,
 234         dtype=dtypes.float32,
 235         time_major=False)
 236     last_activations = _select_last_activations(rnn_outputs, sequence_length)
 237 
 238     with variable_scope.variable_scope('logits', values=(rnn_outputs,)):
 239       logits = core_layers.dense(
 240           last_activations,
 241           units=output_units,
 242           activation=None,
 243           kernel_initializer=init_ops.glorot_uniform_initializer())
 244     return logits
 245 
 246   return rnn_logit_fn

 249 def _rnn_model_fn(features,
 250                   labels,
 251                   mode,
 252                   head,
 253                   rnn_cell_fn,
 254                   sequence_feature_columns,
 255                   context_feature_columns,
 256                   optimizer='Adagrad',
 257                   input_layer_partitioner=None,
 258                   config=None):
 259   """Recurrent Neural Net model_fn.
 260 
 261   Args:
 262     features: dict of `Tensor` and `SparseTensor` objects returned from
 263       `input_fn`.
 264     labels: `Tensor` of shape [batch_size, 1] or [batch_size] with labels.
 265     mode: Defines whether this is training, evaluation or prediction.
 266       See `ModeKeys`.
 267     head: A `head_lib._Head` instance.
 268     rnn_cell_fn: A function with one argument, a `tf.estimator.ModeKeys`, and
 269       returns an object of type `tf.nn.rnn_cell.RNNCell`.
 270     sequence_feature_columns: Iterable containing `FeatureColumn`s that
 271       represent sequential model inputs.
 272     context_feature_columns: Iterable containing `FeatureColumn`s that
 273       represent model inputs not associated with a specific timestep.
 274     optimizer: String, `tf.Optimizer` object, or callable that creates the
 275       optimizer to use for training. If not specified, will use the Adagrad
 276       optimizer with a default learning rate of 0.05 and gradient clip norm of
 277       5.0.
 278     input_layer_partitioner: Partitioner for input layer. Defaults
 279       to `min_max_variable_partitioner` with `min_slice_size` 64 << 20.
 280     config: `RunConfig` object to configure the runtime settings.
 281 
 282   Returns:
 283     An `EstimatorSpec` instance.
 284 
 285   Raises:
 286     ValueError: If mode or optimizer is invalid, or features has the wrong type.
 287   """
 288   if not isinstance(features, dict):
 289     raise ValueError('features should be a dictionary of `Tensor`s. '
 290                      'Given type: {}'.format(type(features)))
 291 
 292   # If user does not provide an optimizer instance, use the optimizer specified
 293   # by the string with default learning rate and gradient clipping.
 294   if not isinstance(optimizer, optimizer_lib.Optimizer):
 295     optimizer = optimizers.get_optimizer_instance(
 296         optimizer, learning_rate=_DEFAULT_LEARNING_RATE)
 297     optimizer = extenders.clip_gradients_by_norm(optimizer, _DEFAULT_CLIP_NORM)
 298 
 299   num_ps_replicas = config.num_ps_replicas if config else 0
 300   partitioner = partitioned_variables.min_max_variable_partitioner(
 301       max_partitions=num_ps_replicas)
 302   with variable_scope.variable_scope(
 303       'rnn',
 304       values=tuple(six.itervalues(features)),
 305       partitioner=partitioner):
 306     input_layer_partitioner = input_layer_partitioner or (
 307         partitioned_variables.min_max_variable_partitioner(
 308             max_partitions=num_ps_replicas,
 309             min_slice_size=64 << 20))
 310 
 311     logit_fn = _rnn_logit_fn_builder(
 312         output_units=head.logits_dimension,
 313         rnn_cell_fn=rnn_cell_fn,
 314         sequence_feature_columns=sequence_feature_columns,
 315         context_feature_columns=context_feature_columns,
 316         input_layer_partitioner=input_layer_partitioner)
 317     logits = logit_fn(features=features, mode=mode)
 318 
 319     def _train_op_fn(loss):
 320       """Returns the op to optimize the loss."""
 321       return optimizer.minimize(
 322           loss,
 323           global_step=training_util.get_global_step())
 324 
 325     return head.create_estimator_spec(
 326         features=features,
 327         mode=mode,
 328         labels=labels,
 329         train_op_fn=_train_op_fn,
 330         logits=logits)
 331

 333 def _assert_rnn_cell_fn(rnn_cell_fn, num_units, cell_type):
 334   """Assert arguments are valid and return rnn_cell_fn."""
 335   if rnn_cell_fn and (num_units or cell_type != USE_DEFAULT):
 336     raise ValueError(
 337         'num_units and cell_type must not be specified when using rnn_cell_fn'
 338     )
 339   if not rnn_cell_fn:
 340     if cell_type == USE_DEFAULT:
 341       cell_type = 'basic_rnn'
 342     rnn_cell_fn = _make_rnn_cell_fn(num_units, cell_type)
 343   return rnn_cell_fn

 346 class RNNClassifier(estimator.Estimator):
 347   """A classifier for TensorFlow RNN models.
 348 
 349   Trains a recurrent neural network model to classify instances into one of
 350   multiple classes.
 351 
 352   Example:
 353 
 354   ```python
 355   token_sequence = sequence_categorical_column_with_hash_bucket(...)
 356   token_emb = embedding_column(categorical_column=token_sequence, ...)
 357 
 358   estimator = RNNClassifier(
 359       sequence_feature_columns=[token_emb],
 360       num_units=[32, 16], cell_type='lstm')
 361 
 362   # Input builders
 363   def input_fn_train: # returns x, y
 364     pass
 365   estimator.train(input_fn=input_fn_train, steps=100)
 366 
 367   def input_fn_eval: # returns x, y
 368     pass
 369   metrics = estimator.evaluate(input_fn=input_fn_eval, steps=10)
 370   def input_fn_predict: # returns x, None
 371     pass
 372   predictions = estimator.predict(input_fn=input_fn_predict)
 373   ```
 374 
 375   Input of `train` and `evaluate` should have following features,
 376   otherwise there will be a `KeyError`:
 377 
 378   * if `weight_column` is not `None`, a feature with
 379     `key=weight_column` whose value is a `Tensor`.
 380   * for each `column` in `sequence_feature_columns`:
 381     - a feature with `key=column.name` whose `value` is a `SparseTensor`.
 382   * for each `column` in `context_feature_columns`:
 383     - if `column` is a `_CategoricalColumn`, a feature with `key=column.name`
 384       whose `value` is a `SparseTensor`.
 385     - if `column` is a `_WeightedCategoricalColumn`, two features: the first
 386       with `key` the id column name, the second with `key` the weight column
 387       name. Both features' `value` must be a `SparseTensor`.
 388     - if `column` is a `_DenseColumn`, a feature with `key=column.name`
 389       whose `value` is a `Tensor`.
 390 
 391   Loss is calculated by using softmax cross entropy.
 392 
 393   @compatibility(eager)
 394   Estimators are not compatible with eager execution.
 395   @end_compatibility
 396   """
 397 
 398   def __init__(self,
 399                sequence_feature_columns,
 400                context_feature_columns=None,
 401                num_units=None,
 402                cell_type=USE_DEFAULT,
 403                rnn_cell_fn=None,
 404                model_dir=None,
 405                n_classes=2,
 406                weight_column=None,
 407                label_vocabulary=None,
 408                optimizer='Adagrad',
 409                loss_reduction=losses.Reduction.SUM_OVER_BATCH_SIZE,
 410                input_layer_partitioner=None,
 411                config=None):
 412     """Initializes a `RNNClassifier` instance.
 413 
 414     Args:
 415       sequence_feature_columns: An iterable containing the `FeatureColumn`s
 416         that represent sequential input. All items in the set should either be
 417         sequence columns (e.g. `sequence_numeric_column`) or constructed from
 418         one (e.g. `embedding_column` with `sequence_categorical_column_*` as
 419         input).
 420       context_feature_columns: An iterable containing the `FeatureColumn`s
 421         for contextual input. The data represented by these columns will be
 422         replicated and given to the RNN at each timestep. These columns must be
 423         instances of classes derived from `_DenseColumn` such as
 424         `numeric_column`, not the sequential variants.
 425       num_units: Iterable of integer number of hidden units per RNN layer. If
 426         set, `cell_type` must also be specified and `rnn_cell_fn` must be
 427         `None`.
 428       cell_type: A subclass of `tf.nn.rnn_cell.RNNCell` or a string specifying
 429         the cell type. Supported strings are: `'basic_rnn'`, `'lstm'`, and
 430         `'gru'`. If set, `num_units` must also be specified and `rnn_cell_fn`
 431         must be `None`.
 432       rnn_cell_fn: A function with one argument, a `tf.estimator.ModeKeys`, and
 433         returns an object of type `tf.nn.rnn_cell.RNNCell` that will be used to
 434         construct the RNN. If set, `num_units` and `cell_type` cannot be set.
 435         This is for advanced users who need additional customization beyond
 436         `num_units` and `cell_type`. Note that `tf.nn.rnn_cell.MultiRNNCell` is
 437         needed for stacked RNNs.
 438       model_dir: Directory to save model parameters, graph and etc. This can
 439         also be used to load checkpoints from the directory into a estimator to
 440         continue training a previously saved model.
 441       n_classes: Number of label classes. Defaults to 2, namely binary
 442         classification. Must be > 1.
 443       weight_column: A string or a `_NumericColumn` created by
 444         `tf.feature_column.numeric_column` defining feature column representing
 445         weights. It is used to down weight or boost examples during training. It
 446         will be multiplied by the loss of the example. If it is a string, it is
 447         used as a key to fetch weight tensor from the `features`. If it is a
 448         `_NumericColumn`, raw tensor is fetched by key `weight_column.key`,
 449         then weight_column.normalizer_fn is applied on it to get weight tensor.
 450       label_vocabulary: A list of strings represents possible label values. If
 451         given, labels must be string type and have any value in
 452         `label_vocabulary`. If it is not given, that means labels are
 453         already encoded as integer or float within [0, 1] for `n_classes=2` and
 454         encoded as integer values in {0, 1,..., n_classes-1} for `n_classes`>2 .
 455         Also there will be errors if vocabulary is not provided and labels are
 456         string.
 457       optimizer: An instance of `tf.Optimizer` or string specifying optimizer
 458         type. Defaults to Adagrad optimizer.
 459       loss_reduction: One of `tf.losses.Reduction` except `NONE`. Describes how
 460         to reduce training loss over batch. Defaults to `SUM_OVER_BATCH_SIZE`.
 461       input_layer_partitioner: Optional. Partitioner for input layer. Defaults
 462         to `min_max_variable_partitioner` with `min_slice_size` 64 << 20.
 463       config: `RunConfig` object to configure the runtime settings.
 464 
 465     Raises:
 466       ValueError: If `num_units`, `cell_type`, and `rnn_cell_fn` are not
 467         compatible.
 468     """
 469     rnn_cell_fn = _assert_rnn_cell_fn(rnn_cell_fn, num_units, cell_type)
 470 
 471     if n_classes == 2:
 472       head = head_lib._binary_logistic_head_with_sigmoid_cross_entropy_loss(  # pylint: disable=protected-access
 473           weight_column=weight_column,
 474           label_vocabulary=label_vocabulary,
 475           loss_reduction=loss_reduction)
 476     else:
 477       head = head_lib._multi_class_head_with_softmax_cross_entropy_loss(  # pylint: disable=protected-access
 478           n_classes,
 479           weight_column=weight_column,
 480           label_vocabulary=label_vocabulary,
 481           loss_reduction=loss_reduction)
 482 
 483     def _model_fn(features, labels, mode, config):
 484       return _rnn_model_fn(
 485           features=features,
 486           labels=labels,
 487           mode=mode,
 488           head=head,
 489           rnn_cell_fn=rnn_cell_fn,
 490           sequence_feature_columns=tuple(sequence_feature_columns or []),
 491           context_feature_columns=tuple(context_feature_columns or []),
 492           optimizer=optimizer,
 493           input_layer_partitioner=input_layer_partitioner,
 494           config=config)
 495     super(RNNClassifier, self).__init__(
 496         model_fn=_model_fn, model_dir=model_dir, config=config)

 499 class RNNEstimator(estimator.Estimator):
 500   """An Estimator for TensorFlow RNN models with user-specified head.
 501 
 502   Example:
 503 
 504   ```python
 505   token_sequence = sequence_categorical_column_with_hash_bucket(...)
 506   token_emb = embedding_column(categorical_column=token_sequence, ...)
 507 
 508   estimator = RNNEstimator(
 509       head=tf.contrib.estimator.regression_head(),
 510       sequence_feature_columns=[token_emb],
 511       num_units=[32, 16], cell_type='lstm')
 512 
 513   # Or with custom RNN cell:
 514   def rnn_cell_fn(mode):
 515     cells = [ tf.contrib.rnn.LSTMCell(size) for size in [32, 16] ]
 516     if mode == tf.estimator.ModeKeys.TRAIN:
 517       cells = [ tf.contrib.rnn.DropoutWrapper(cell, input_keep_prob=0.5)
 518                     for cell in cells ]
 519     return tf.contrib.rnn.MultiRNNCell(cells)
 520 
 521   estimator = RNNEstimator(
 522       head=tf.contrib.estimator.regression_head(),
 523       sequence_feature_columns=[token_emb],
 524       rnn_cell_fn=rnn_cell_fn)
 525 
 526   # Input builders
 527   def input_fn_train: # returns x, y
 528     pass
 529   estimator.train(input_fn=input_fn_train, steps=100)
 530 
 531   def input_fn_eval: # returns x, y
 532     pass
 533   metrics = estimator.evaluate(input_fn=input_fn_eval, steps=10)
 534   def input_fn_predict: # returns x, None
 535     pass
 536   predictions = estimator.predict(input_fn=input_fn_predict)
 537   ```
 538 
 539   Input of `train` and `evaluate` should have following features,
 540   otherwise there will be a `KeyError`:
 541 
 542   * if the head's `weight_column` is not `None`, a feature with
 543     `key=weight_column` whose value is a `Tensor`.
 544   * for each `column` in `sequence_feature_columns`:
 545     - a feature with `key=column.name` whose `value` is a `SparseTensor`.
 546   * for each `column` in `context_feature_columns`:
 547     - if `column` is a `_CategoricalColumn`, a feature with `key=column.name`
 548       whose `value` is a `SparseTensor`.
 549     - if `column` is a `_WeightedCategoricalColumn`, two features: the first
 550       with `key` the id column name, the second with `key` the weight column
 551       name. Both features' `value` must be a `SparseTensor`.
 552     - if `column` is a `_DenseColumn`, a feature with `key=column.name`
 553       whose `value` is a `Tensor`.
 554 
 555   Loss and predicted output are determined by the specified head.
 556 
 557   @compatibility(eager)
 558   Estimators are not compatible with eager execution.
 559   @end_compatibility
 560   """
 561

 563                head,
