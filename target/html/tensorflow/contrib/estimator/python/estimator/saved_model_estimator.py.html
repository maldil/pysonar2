  39 class SavedModelEstimator(estimator_lib.Estimator):
  40   """Create an Estimator from a SavedModel.
  41 
  42   Only SavedModels exported with
  43   `tf.contrib.estimator.export_all_saved_models()` or
  44   `tf.estimator.Estimator.export_savedmodel()` are supported for this class.
  45 
  46   Example with `tf.estimator.DNNClassifier`:
  47 
  48   **Step 1: Create and train DNNClassifier.**
  49 
  50   ```python
  51   feature1 = tf.feature_column.embedding_column(
  52       tf.feature_column.categorical_column_with_vocabulary_list(
  53           key='feature1', vocabulary_list=('green', 'yellow')), dimension=1)
  54   feature2 = tf.feature_column.numeric_column(key='feature2', default_value=0.0)
  55 
  56   classifier = tf.estimator.DNNClassifier(
  57       hidden_units=[4,2], feature_columns=[feature1, feature2])
  58 
  59   def input_fn():
  60     features = {'feature1': tf.constant(['green', 'green', 'yellow']),
  61                 'feature2': tf.constant([3.5, 4.2, 6.1])}
  62     label = tf.constant([1., 0., 0.])
  63     return tf.data.Dataset.from_tensors((features, label)).repeat()
  64 
  65   classifier.train(input_fn=input_fn, steps=10)
  66   ```
  67 
  68   **Step 2: Export classifier.**
  69   First, build functions that specify the expected inputs.
  70 
  71   ```python
  72   # During train and evaluation, both the features and labels should be defined.
  73   supervised_input_receiver_fn = (
  74       tf.contrib.estimator.build_raw_supervised_input_receiver_fn(
  75           {'feature1': tf.placeholder(dtype=tf.string, shape=[None]),
  76            'feature2': tf.placeholder(dtype=tf.float32, shape=[None])},
  77           tf.placeholder(dtype=tf.float32, shape=[None])))
  78 
  79   # During predict mode, expect to receive a `tf.Example` proto, so a parsing
  80   # function is used.
  81   serving_input_receiver_fn = (
  82       tf.estimator.export.build_parsing_serving_input_receiver_fn(
  83           tf.feature_column.make_parse_example_spec([feature1, feature2])))
  84   ```
  85 
  86   Next, export the model as a SavedModel. A timestamped directory will be
  87   created (for example `/tmp/export_all/1234567890`).
  88 
  89   ```python
  90   # Option 1: Save all modes (train, eval, predict)
  91   export_dir = tf.contrib.estimator.export_all_saved_models(
  92       classifier, '/tmp/export_all',
  93       {tf.estimator.ModeKeys.TRAIN: supervised_input_receiver_fn,
  94        tf.estimator.ModeKeys.EVAL: supervised_input_receiver_fn,
  95        tf.estimator.ModeKeys.PREDICT: serving_input_receiver_fn})
  96 
  97   # Option 2: Only export predict mode
  98   export_dir = classifier.export_savedmodel(
  99       '/tmp/export_predict', serving_input_receiver_fn)
 100   ```
 101 
 102   **Step 3: Create a SavedModelEstimator from the exported SavedModel.**
 103 
 104   ```python
 105   est = tf.contrib.estimator.SavedModelEstimator(export_dir)
 106 
 107   # If all modes were exported, you can immediately evaluate and predict, or
 108   # continue training. Otherwise only predict is available.
 109   eval_results = est.evaluate(input_fn=input_fn, steps=1)
 110   print(eval_results)
 111 
 112   est.train(input_fn=input_fn, steps=20)
 113 
 114   def predict_input_fn():
 115     example = tf.train.Example()
 116     example.features.feature['feature1'].bytes_list.value.extend(['yellow'])
 117     example.features.feature['feature2'].float_list.value.extend([1.])
 118     return {'inputs':tf.constant([example.SerializeToString()])}
 119 
 120   predictions = est.predict(predict_input_fn)
 121   print(next(predictions))
 122   ```
 123   """
 124 
 125   def __init__(self, saved_model_dir, model_dir=None):
 126     """Initialize a SavedModelEstimator.
 127 
 128     The SavedModelEstimator loads its model function and variable values from
 129     the graphs defined in the SavedModel. There is no option to pass in
 130     `RunConfig` or `params` arguments, because the model function graph is
 131     defined statically in the SavedModel.
 132 
 133     Args:
 134       saved_model_dir: Directory containing SavedModel protobuf and subfolders.
 135       model_dir: Directory to save new checkpoints during training.
 136 
 137     Raises:
 138       NotImplementedError: If a DistributionStrategy is defined in the config.
 139         Unless the SavedModelEstimator is subclassed, this shouldn't happen.
 140     """
 141     checkpoint = estimator_lib._get_saved_model_ckpt(saved_model_dir)  # pylint: disable=protected-access
 142     vars_to_warm_start = [name for name, _ in
 143                           checkpoint_utils.list_variables(checkpoint)]
 144     warm_start_settings = estimator_lib.WarmStartSettings(
 145         ckpt_to_initialize_from=checkpoint,
 146         vars_to_warm_start=vars_to_warm_start)
 147 
 148     super(SavedModelEstimator, self).__init__(
 149         model_fn=self._model_fn_from_saved_model, model_dir=model_dir,
 150         warm_start_from=warm_start_settings)
 151     if self._train_distribution or self._eval_distribution:
 152       raise NotImplementedError(
 153           'SavedModelEstimator currently does not support '
 154           'DistributionStrategy.')
 155     self.saved_model_dir = saved_model_dir
 156     self.saved_model_loader = loader_impl.SavedModelLoader(saved_model_dir)
 157     self._available_modes = self._extract_available_modes()
 158 
 159   def _extract_available_modes(self):
 160     """Return list of modes found in SavedModel."""
 161     available_modes = []
 162     logging.info('Checking available modes for SavedModelEstimator.')
 163     for mode in [model_fn_lib.ModeKeys.TRAIN, model_fn_lib.ModeKeys.EVAL,
 164                  model_fn_lib.ModeKeys.PREDICT]:
 165       try:
 166         self._get_meta_graph_def_for_mode(mode)
 167       except RuntimeError:
 168         logging.warning('%s mode not found in SavedModel.' % mode)
 169         continue
 170 
 171       if self._get_signature_def_for_mode(mode) is not None:
 172         available_modes.append(mode)
 173 
 174     logging.info('Available modes for Estimator: %s' % available_modes)
 175     return available_modes
 176 
 177   def _validate_mode(self, mode):
 178     """Make sure that mode can be run using the SavedModel."""
 179     if mode not in self._available_modes:
 180       raise RuntimeError('%s mode is not available in the SavedModel. Use '
 181                          'saved_model_cli to check that the Metagraph for this '
 182                          'mode has been exported.' % mode)
 183 
 184   def _get_meta_graph_def_for_mode(self, mode):
 185     tags = model_fn_lib.EXPORT_TAG_MAP[mode]
 186     return self.saved_model_loader.get_meta_graph_def_from_tags(tags)
 187 
 188   def _get_signature_def_for_mode(self, mode):
 189     meta_graph_def = self._get_meta_graph_def_for_mode(mode)
 190     sig_def_key = (signature_constants.DEFAULT_SERVING_SIGNATURE_DEF_KEY
 191                    if mode == model_fn_lib.ModeKeys.PREDICT else mode)
 192     if sig_def_key not in meta_graph_def.signature_def:
 193       logging.warning('Metagraph for mode %s was found, but SignatureDef with'
 194                       ' key \"%s\" is missing.' % (mode, sig_def_key))
 195       return None
 196     return meta_graph_def.signature_def[sig_def_key]
 197 
 198   def _create_and_assert_global_step(self, graph):
 199     # Do nothing here. The global step variable will be created/loaded from the
 200     # SavedModel. If a global step variable were created here, the result
 201     # will be two duplicate global step variables, causing issues during
 202     # the warm-start phase.
 203     # Due to the global variable being created in the model function, this may
 204     # cause issues when running DistributionStrategy. Thus, DistributionStrategy
 205     # is not yet supported with SavedModelEstimator.
 206     return None
 207 
 208   def _model_fn_from_saved_model(self, features, labels, mode):
 209     """Load a SavedModel graph and return an EstimatorSpec."""
 210     # TODO(kathywu): Model function loads placeholders from the graph. Calling
 211     # export_all_saved_models creates another placeholder for the inputs, on top
 212     # of the original placeholders. There should be a way to avoid this.
 213     self._validate_mode(mode)
 214 
 215     g = ops.get_default_graph()
 216     if  training_util.get_global_step(g) is not None:
 217       raise RuntimeError(
 218           'Graph must not contain a global step tensor before the SavedModel is'
 219           ' loaded. Please make sure that the input function does not create a '
 220           'global step.')
 221 
 222     # Extract SignatureDef for information about the input and output tensors.
 223     signature_def = self._get_signature_def_for_mode(mode)
 224 
 225     # Generate input map for replacing the inputs in the SavedModel graph with
 226     # the provided features and labels.
 227     input_map = _generate_input_map(signature_def, features, labels)
 228 
 229     # Create a list of the names of output tensors. When the graph is loaded,
 230     # names of the output tensors may be remapped. This ensures that the correct
 231     # tensors are returned in the EstimatorSpec.
 232     output_tensor_names = [
 233         value.name for value in six.itervalues(signature_def.outputs)]
 234 
 235     # Load the graph. `output_tensors` contains output `Tensors` in the same
 236     # same order as the `output_tensor_names` list.
 237     tags = model_fn_lib.EXPORT_TAG_MAP[mode]
 238     _, output_tensors = self.saved_model_loader.load_graph(
 239         g, tags, input_map=input_map, return_elements=output_tensor_names)
 240 
 241     # Create a scaffold from the MetaGraphDef that contains ops to initialize
 242     # the graph. This should mirror the steps from _add_meta_graph_for_mode(),
 243     # which creates a MetaGraphDef from the EstimatorSpec's scaffold.
 244     scaffold = monitored_session.Scaffold(
 245         local_init_op=loader_impl._get_main_op_tensor(  # pylint: disable=protected-access
 246             self._get_meta_graph_def_for_mode(mode)))
 247 
 248     # Ensure that a global step tensor has been created.
 249     global_step_tensor = training_util.get_global_step(g)
 250     training_util.assert_global_step(global_step_tensor)
 251 
 252     # Extract values to return in the EstimatorSpec.
 253     output_map = dict(zip(output_tensor_names, output_tensors))
 254     outputs = {key: output_map[value.name]
 255                for key, value in six.iteritems(signature_def.outputs)}
 256 
 257     loss, predictions, metrics = _validate_and_extract_outputs(
 258         mode, outputs, signature_def.method_name)
 259 
 260     train_op = ops.get_collection(constants.TRAIN_OP_KEY)
 261     if len(train_op) > 1:
 262       raise RuntimeError('Multiple ops found in the train_op collection.')
 263     train_op = None if not train_op else train_op[0]
 264 
 265     _clear_saved_model_collections()
 266     return model_fn_lib.EstimatorSpec(
 267         scaffold=scaffold,
 268         mode=mode,
 269         loss=loss,
 270         train_op=train_op,
 271         predictions=predictions,
 272         eval_metric_ops=metrics)

 275 def _clear_saved_model_collections():
 276   """Clear collections that are expected empty when exporting a SavedModel.
 277 
 278   The SavedModel builder uses these collections to track ops necessary to
 279   restore the graph state. These collections are expected to be empty before
 280   MetaGraphs are added to the builder.
 281   """
 282   del ops.get_collection_ref(constants.ASSETS_KEY)[:]
 283   del ops.get_collection_ref(constants.LEGACY_INIT_OP_KEY)[:]
 284   del ops.get_collection_ref(constants.MAIN_OP_KEY)[:]
 285   del ops.get_collection_ref(constants.TRAIN_OP_KEY)[:]

 288 def _generate_input_map(signature_def, features, labels):
 289   """Return dict mapping an input tensor name to a feature or label tensor.
 290 
 291   Args:
 292     signature_def: SignatureDef loaded from SavedModel
 293     features: A `Tensor`, `SparseTensor`, or dict of string to `Tensor` or
 294       `SparseTensor`, specifying the features to be passed to the model.
 295     labels: A `Tensor`, `SparseTensor`, or dict of string to `Tensor` or
 296       `SparseTensor`, specifying the labels to be passed to the model. May be
 297       `None`.
 298 
 299   Returns:
 300     dict mapping string names of inputs to features or labels tensors
 301 
 302   Raises:
 303     ValueError: if SignatureDef inputs are not completely mapped by the input
 304       features and labels.
 305   """
 306   # pylint: disable=protected-access
 307   if not isinstance(features, dict):
 308     features = {export_lib._SINGLE_FEATURE_DEFAULT_NAME: features}
 309   if labels is not None and not isinstance(labels, dict):
 310     labels = {export_lib._SINGLE_LABEL_DEFAULT_NAME: labels}
 311   # pylint: enable=protected-access
 312 
 313   inputs = signature_def.inputs
 314   input_map = {}
 315   for key, tensor_info in six.iteritems(inputs):
 316     input_name = tensor_info.name
 317     if ':' in input_name:
 318       input_name = input_name[:input_name.find(':')]
 319 
 320     # When tensors are used as control inputs for operations, their names are
 321     # prepended with a '^' character in the GraphDef. To handle possible control
 322     # flow edge cases, control input names must be included in the input map.
 323     control_dependency_name = '^' + input_name
 324 
 325     if key in features:
 326       _check_same_dtype_and_shape(features[key], tensor_info, key)
 327       input_map[input_name] = input_map[control_dependency_name] = features[key]
 328     elif labels is not None and key in labels:
 329       _check_same_dtype_and_shape(labels[key], tensor_info, key)
 330       input_map[input_name] = input_map[control_dependency_name] = labels[key]
 331     else:
 332       raise ValueError(
 333           'Key \"%s\" not found in features or labels passed in to the model '
 334           'function. All required keys: %s' % (key, inputs.keys()))
 335 
 336   return input_map

 339 def _check_same_dtype_and_shape(tensor, tensor_info, name):
 340   """Validate that tensor has the same properties as the TensorInfo proto.
 341 
 342   Args:
 343     tensor: a `Tensor` object.
 344     tensor_info: a `TensorInfo` proto.
 345     name: Name of the input (to identify Tensor if an error is raised).
 346 
 347   Raises:
 348     ValueError: If the tensor shape or dtype don't match the TensorInfo
 349   """
 350   dtype_error = (tensor.dtype != dtypes.DType(tensor_info.dtype))
 351   shape_error = not tensor.shape.is_compatible_with(tensor_info.tensor_shape)
 352 
 353   if dtype_error or shape_error:
 354     msg = 'Tensor shape and/or dtype validation failed for input %s:' % name
 355     if dtype_error:
 356       msg += ('\n\tExpected dtype: %s, Got: %s'
 357               % (dtypes.DType(tensor_info.dtype), tensor.dtype))
 358     if shape_error:
 359       msg += ('\n\tExpected shape: %s, Got: %s'
 360               % (tensor_shape.TensorShape(tensor_info.tensor_shape),
 361                  tensor.shape))
 362 
 363     raise ValueError(msg)

 366 def _extract_eval_metrics(output_dict):
 367   """Return a eval metric dict extracted from the output_dict.
 368 
 369   Eval metrics consist of a value tensor and an update op. Both must be in the
 370   passed-in tensor dictionary for an eval metric to be added to the returned
 371   dictionary.
 372 
 373   Args:
 374     output_dict: a dict that maps strings to tensors.
 375 
 376   Returns:
 377     dict mapping strings to (value, update_op) tuples.
 378   """
 379   # pylint: disable=protected-access
 380   metric_ops = {}
 381   separator_char = export_output._SupervisedOutput._SEPARATOR_CHAR
 382 
 383   for key, tensor in six.iteritems(output_dict):
 384     split_key = key.split(separator_char)
 385 
 386     # The metric name may contain the separator character, so recreate its name.
 387     metric_name = separator_char.join(split_key[:-1])
 388 
 389     if split_key[0] == export_output._SupervisedOutput.METRICS_NAME:
 390       # If the key ends with the value suffix, and there is a corresponding
 391       # key ending with the update_op suffix, then add tensors to metrics dict.
 392       if split_key[-1] == export_output._SupervisedOutput.METRIC_VALUE_SUFFIX:
 393         update_op = ''.join(
 394             [metric_name, separator_char,
 395              export_output._SupervisedOutput.METRIC_UPDATE_SUFFIX])
 396         if update_op in output_dict:
 397           update_op_tensor = output_dict[update_op]
 398           metric_ops[metric_name] = (tensor, update_op_tensor)
 399 
 400   # pylint: enable=protected-access
 401   return metric_ops

 404 def _validate_and_extract_outputs(mode, output_dict, method_name):
 405   """Extract values from SignatureDef output dictionary.
 406 
 407   Args:
 408     mode: One of the modes enumerated in `tf.estimator.ModeKeys`.
 409     output_dict: dict of string SignatureDef keys to `Tensor`.
 410     method_name: Method name of the SignatureDef as a string.
 411 
 412   Returns:
 413     Tuple of (
 414       loss: `Tensor` object,
 415       predictions: dictionary mapping string keys to `Tensor` objects,
 416       metrics: dictionary mapping string keys to a tuple of two `Tensor` objects
 417     )
 418 
 419   Raises:
 420     RuntimeError: raised if SignatureDef has an invalid method name for the mode
 421   """
 422   # pylint: disable=protected-access
 423   loss, predictions, metrics = None, None, None
 424 
 425   if mode == model_fn_lib.ModeKeys.PREDICT:
 426     predictions = output_dict
 427   else:
 428     # Validate that the SignatureDef's method name matches the expected name for
 429     # the given mode.
 430     expected_method_name = signature_constants.SUPERVISED_TRAIN_METHOD_NAME
 431     if mode == model_fn_lib.ModeKeys.EVAL:
 432       expected_method_name = signature_constants.SUPERVISED_EVAL_METHOD_NAME
 433     if method_name != expected_method_name:
 434       raise RuntimeError(
 435           'Invalid SignatureDef method name for mode %s.\n\tExpected: %s\n\t'
 436           'Got: %s\nPlease ensure that the SavedModel was exported with '
 437           '`tf.contrib.estimator.export_all_saved_models()`.' %
 438           (mode, expected_method_name, method_name))
 439 
 440     # Extract loss, metrics and predictions from the output dict.
 441     loss = output_dict[export_output._SupervisedOutput.LOSS_NAME]
 442     metrics = _extract_eval_metrics(output_dict)
 443     predictions = {
 444         key: value for key, value in six.iteritems(output_dict)
 445         if key.split(export_output._SupervisedOutput._SEPARATOR_CHAR)[0] == (
 446             export_output._SupervisedOutput.PREDICTIONS_NAME)}
 447 
 448   # pylint: enable=protected-access
 449   return loss, predictions, metrics
