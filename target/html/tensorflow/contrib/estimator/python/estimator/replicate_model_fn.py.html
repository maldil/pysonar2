  53 @deprecation.deprecated(
  54     '2018-05-31',
  55     'Please use `tf.contrib.distribute.MirroredStrategy` instead.')
  56 def replicate_model_fn(model_fn,
  57                        loss_reduction=losses.Reduction.SUM_BY_NONZERO_WEIGHTS,
  58                        devices=None):
  59   """Replicate `Estimator.model_fn` over GPUs.
  60 
  61   The given `model_fn` specifies a single forward pass of a model.  To replicate
  62   such a model over GPUs, each GPU gets its own instance of the forward pass
  63   (a.k.a. a tower).  The input features and labels get sharded into the chunks
  64   that correspond to the number of GPUs.  Each tower computes a loss based
  65   on its input.  For each such loss, gradients are computed.  After that, the
  66   available losses are aggregated to form aggregated loss.  Available
  67   gradients are summed.  Then, they update weights using the specified
  68   optimizer.
  69 
  70   If `devices` are `None`, then all available GPUs are going to be used for
  71   replication.  If no GPUs are available, then the model is going to be
  72   placed on the CPU.
  73 
  74   Two modes of local replication over available GPUs are supported:
  75     1)  If exactly 1 GPU is detected, then variables and operations are placed
  76         onto the GPU.
  77     2)  If more than 1 GPU is detected, then variables are going to be placed on
  78         the CPU.  Replicas of operations are placed on each individual GPU.
  79 
  80   Here is an example of how one might use their `model_fn` to run over GPUs:
  81     ```python
  82        ...
  83        def model_fn(...):  # See `model_fn` in `Estimator`.
  84          loss = ...
  85          optimizer = tf.train.GradientDescentOptimizer(learning_rate=0.001)
  86          optimizer = tf.contrib.estimator.TowerOptimizer(optimizer)
  87          if mode == tf.estimator.ModeKeys.TRAIN:
  88            #  See the section below on `EstimatorSpec.train_op`.
  89            return EstimatorSpec(mode=mode, loss=loss,
  90                                 train_op=optimizer.minimize(loss))
  91 
  92          #  No change for `ModeKeys.EVAL` or `ModeKeys.PREDICT`.
  93          return EstimatorSpec(...)
  94        ...
  95        classifier = tf.estimator.Estimator(
  96          model_fn=tf.contrib.estimator.replicate_model_fn(model_fn))
  97     ```
  98 
  99   Please see `DNNClassifierIntegrationTest` for an example with a canned
 100   Estimator.
 101 
 102   On `EstimatorSpec.train_op`:
 103   `model_fn` returns `EstimatorSpec.train_op` for
 104   `tf.estimator.GraphKeys.TRAIN`. It is typically derived using an optimizer.
 105   Towers are expected to populate it in the same way.  Gradients from all towers
 106   are reduced and applied in the last tower.  To achieve that in the case of
 107   multiple towers, `TowerOptimizer` needs to be used.  See `TowerOptimizer`.
 108 
 109   On sharding input features and labels:
 110   Input features and labels are split for consumption by each tower. They are
 111   split across the dimension 0.  Features and labels need to be batch major.
 112 
 113   On reduction algorithms:
 114   Certain algorithms were chosen for aggregating results of computations on
 115   multiple towers:
 116     - Losses from all towers are reduced according to `loss_reduction`.
 117     - Gradients from all towers are reduced according to `loss_reduction`
 118       for each trainable variable.
 119     - `eval_metrics_ops` are reduced per metric using `reduce_mean`.
 120     - `EstimatorSpec.predictions` and `EstimatorSpec.export_outputs` are
 121       reduced using concatenation.
 122     - For all other fields of `EstimatorSpec` the values of the first tower
 123       are taken.
 124 
 125   On distribution of variables:
 126   Variables are not duplicated between towers.  Instead, they are placed on a
 127   single device as defined above and shared across towers.
 128 
 129   On overhead:
 130   If only one device is specified, then aggregation of loss and gradients
 131   doesn't happen. Replication consists of placing `model_fn` onto the
 132   specified device.
 133 
 134   On current limitations:
 135     - `predictions` are not supported for `ModeKeys.EVAL`.  They are required
 136        for `tf.contrib.estimator.add_metrics`.
 137 
 138   Args:
 139     model_fn: `model_fn` as defined in `Estimator`.  See the section above about
 140       the train_op argument of `EstimatorSpec`.
 141     loss_reduction: controls whether losses are summed or averaged.
 142     devices: Optional list of devices to replicate the model across.  This
 143       argument can be used to replicate only on the subset of available GPUs.
 144       If `None`, then all available GPUs are going to be used for replication.
 145       If no GPUs are available, then the model is going to be placed on the CPU.
 146 
 147   Raises:
 148     ValueError: if there is no `loss_reduction` or if TowerOptimizer is
 149       mis-used.
 150 
 151   Returns:
 152     A replicated version of the supplied `model_fn`. Returned function that
 153       conforms to the requirements of `Estimator`'s `model_fn` and can be used
 154       instead of the supplied `model_fn`.
 155   """
 156   return _replicate_model_fn_with_mode(
 157       model_fn,
 158       loss_reduction,
 159       devices,
 160       # TODO(isaprykin): Query the system configuration to choose modes other
 161       # than `SHARED_LOCAL_PARAMETER_SERVER`, even though it is often
 162       # appropriate.
 163       mode=_VariableDistributionMode.SHARED_LOCAL_PARAMETER_SERVER)
 164

 166 class _VariableDistributionMode(object):
 167   """Modes for variable distribution used for forcing a particular one.
 168 
 169   Forcing a mode is meant for performance experimentation purposes rather than
 170   for general use cases.
 171   """
 172 
 173   SHARED_LOCAL_PARAMETER_SERVER = 1
 174   """Variables are placed on a single device and shared across all devices.
 175 
 176   Two ways to achieve this distribution over available GPUs are supported:
 177     1)  If exactly 1 GPU is detected, then variables and operations are placed
 178         onto GPU.
 179     2)  If more than 1 GPU is detected, then variables are going to be placed on
 180         the CPU.  Replicas of operations are placed on each individual GPU.
 181   """
 182 
 183   SHARED_ROUND_ROBIN = 2
 184   """Variables are placed on all devices in a round-robin fashion.
 185 
 186   Every subsequent variable is placed on the next device.  There is only one
 187   copy of each variable that is shared across all devices.
 188   """
 189 
 190 
 191 def _replicate_model_fn_with_mode(
 192     model_fn,
 193     loss_reduction,
 194     devices=None,
 195     mode=_VariableDistributionMode.SHARED_LOCAL_PARAMETER_SERVER):
 196   """A version of `replicate_model_fn` that allows to specify a `mode`."""
 197   if loss_reduction == losses.Reduction.NONE:
 198     raise ValueError('Tower losses need to be reduced in some way, yet {} '
 199                      'reduction is specified.'.format(loss_reduction))
 200   if not devices:
 201     devices = _get_local_devices('GPU') or _get_local_devices('CPU')
 202 
 203   is_a_single_gpu_case = len(devices) == 1 and 'GPU' in devices[0].upper()
 204   consolidation_device = devices[0] if is_a_single_gpu_case else '/CPU:0'
 205 
 206   ps_devices = [consolidation_device]
 207   if mode == _VariableDistributionMode.SHARED_ROUND_ROBIN:
 208     ps_devices = devices
 209 
 210   tf_logging.info('Replicating the `model_fn` across {}.  Variables are going '
 211                   'to be placed on {}.  Consolidation device is going to be {}.'
 212                   .format(devices, ps_devices, consolidation_device))
 213 
 214   def single_device_model_fn(features, labels, mode, params=None, config=None):
 215     """`model_fn` on a single device without reduction overhead."""
 216     return _get_loss_towers(
 217         model_fn=model_fn,
 218         mode=mode,
 219         features=[features],
 220         labels=[labels],
 221         params=params,
 222         loss_reduction=loss_reduction,
 223         config=config,
 224         devices=devices,
 225         local_ps_devices=ps_devices)[0]  # One device, so one spec is out.
 226 
 227   def replicated_model_fn(features, labels, mode, params=None, config=None):
 228     """Replicated version of `model_fn` to be used instead."""
 229     feature_shards, label_shards = _split_batch(
 230         features, labels, len(devices), device=consolidation_device)
 231     tower_specs = _get_loss_towers(
 232         model_fn=model_fn,
 233         mode=mode,
 234         features=feature_shards,
 235         labels=label_shards,
 236         params=params,
 237         loss_reduction=loss_reduction,
 238         config=config,
 239         devices=devices,
 240         local_ps_devices=ps_devices)
 241 
 242     if mode == model_fn_lib.ModeKeys.TRAIN:
 243       train_op = _minimize_towers(tower_specs)
 244       return _train_spec(
 245           tower_specs, train_op, aggregation_device=consolidation_device)
 246     elif mode == model_fn_lib.ModeKeys.EVAL:
 247       return _eval_spec(tower_specs, aggregation_device=consolidation_device)
 248     elif mode == model_fn_lib.ModeKeys.PREDICT:
 249       return _predict_spec(tower_specs, aggregation_device=consolidation_device)
 250 
 251   if len(devices) == 1:
 252     return single_device_model_fn
 253   else:
 254     return replicated_model_fn

 257 class TowerOptimizer(optimizer_lib.Optimizer):
 258   """Gathers gradients from all towers and reduces them in the last one."""
 259 
 260   COLLECTION_FOR_GRAPH_STATES = 'replicate_model_fn_graph_states'
 261 
 262   @deprecation.deprecated(
 263       '2018-05-31',
 264       'Please use `tf.contrib.distribute.MirroredStrategy` instead.')
 265   def __init__(self, optimizer_or_optimizer_fn):
 266     """Wrap an existing optimizer for gathering gradients across towers.
 267 
 268     Each invocation of model_fn has to call the same optimizers in the same
 269     order.
 270 
 271     Multiple optimizers that use the same or different losses are supported.
 272 
 273     If TowerOptimizer is used but `replicate_model_fn` isn't, then no
 274     aggregation will happen.  All calls will simply be forwarded to the
 275     underlying optimizer. The behavior is similar if there is only one tower.
 276 
 277     If TowerOptimizer is used together with SyncReplicasOptimizer that wraps
 278     the user's optimizer, then it's the SyncReplicasOptimizer that needs to be
 279     wrapped with TowerOptimizer.
 280 
 281     Args:
 282       optimizer_or_optimizer_fn: an instance of optimizer to wrap.  That
 283         instance is going to be used for optimizer-specific logic.  This can
 284         also be a no-argument function that returns such an optimizer instance.
 285     """
 286     self._optimizer_or_optimizer_fn = optimizer_or_optimizer_fn
 287 
 288   @staticmethod
 289   def has_been_used():
 290     return TowerOptimizer._graph_state().has_tower_optimizer_been_used
 291 
 292   def get_slot(self, *args, **kwargs):
 293     return self._get_optimizer().get_slot(*args, **kwargs)
 294 
 295   def get_slot_names(self, *args, **kwargs):
 296     return self._get_optimizer().get_slot_names(*args, **kwargs)
 297 
 298   def get_name(self, *args, **kwargs):
 299     return self._get_optimizer().get_name(*args, **kwargs)
 300 
 301   def variables(self, *args, **kwargs):
 302     return self._get_optimizer().variables(*args, **kwargs)
 303 
 304   def compute_gradients(self, loss, *args, **kwargs):
 305     """Compute gradients, but first, if needed, scale the loss."""
 306     loss = _scale_loss(loss,
 307                        self._graph_state().loss_reduction,
 308                        self._graph_state().number_of_towers)
 309     return self._get_optimizer().compute_gradients(loss, *args, **kwargs)
 310 
 311   def apply_gradients(self, grads_and_vars, global_step=None, **kwargs):
 312     """Collect gradients updates to apply them with the last tower."""
 313     if self._graph_state().number_of_towers == 1:
 314       # Avoid the overhead of reduction if there's only one tower.
 315       #
 316       # There assumed to be only one tower if aggregation-related methods were
 317       # not called by `_get_loss_towers`, for example if the model_fn uses
 318       # TowerEstimator, but `replicate_model_fn` isn't used.
 319       return self._get_optimizer().apply_gradients(grads_and_vars, global_step,
 320                                                    **kwargs)
 321 
 322     self._graph_state().collect_gradients(grads_and_vars)
 323 
 324     if not self._graph_state().is_the_last_tower:
 325       with ops_lib.control_dependencies(_extract_tensors(grads_and_vars)):
 326         return self._construct_no_op_train_op()
 327     else:
 328       # Gradients need to be gathered and applied in the scope of the first
 329       # tower, so that the tensors are accessible via names without prefixes.
 330       var_scope, name_scope = self._graph_state().scopes_of_the_first_tower
 331       with variable_scope.variable_scope(var_scope):
 332         with ops_lib.name_scope(name_scope):
 333           return self._apply_gathered_gradients(global_step, **kwargs)
 334 
 335   def _apply_gathered_gradients(self, global_step, **kwargs):
 336     graph_state = self._graph_state()
 337     optimizer = self._get_optimizer()
 338 
 339     grad_lists = {}
 340     for grad, var in graph_state.get_latest_gradients_from_all_towers():
 341       if grad is not None:
 342         grad_lists.setdefault(var, []).append(grad)
 343 
 344     aggregated_grads = []
 345     with ops_lib.name_scope('gradient_aggregating'):
 346       for var, grads in six.iteritems(grad_lists):
 347         grad = _compute_sum_on_device(grads, var.device)
 348         aggregated_grads.append((grad, var))
 349     return optimizer.apply_gradients(
 350         aggregated_grads, global_step=global_step, **kwargs)
 351 
 352   def _get_optimizer(self):
 353     if callable(self._optimizer_or_optimizer_fn):
 354       # If optimizer is given as a function then we need to wait till we are
 355       # under the right graph context before constructing it.  That's why the
 356       # optimizer is constructed in _get_optimizer() rather than __init__().
 357       self._optimizer_or_optimizer_fn = self._optimizer_or_optimizer_fn()
 358     self._graph_state().has_tower_optimizer_been_used = True
 359     return self._optimizer_or_optimizer_fn
 360 
 361   def _construct_no_op_train_op(self):
 362     return control_flow_ops.no_op(name='train_op_placeholder')
 363 
 364   @staticmethod
 365   def _graph_state():
 366     graph_states = ops_lib.get_default_graph().get_collection_ref(
 367         TowerOptimizer.COLLECTION_FOR_GRAPH_STATES)
 368     if not graph_states:
 369       graph_states.append(TowerOptimizer._PerGraphState())
 370     return graph_states[-1]
 371 
 372   @staticmethod
 373   def _did_towers_have_same_optimizer_calls():
 374     graph_state = TowerOptimizer._graph_state()
 375     return graph_state.did_towers_have_same_optimizer_calls()
 376 
 377   @staticmethod
 378   def _clear_graph_state():
 379     # Clearing the Graph collection will prevent _PerGraphState from being
 380     # serialized.
 381     ops_lib.get_default_graph().clear_collection(
 382         TowerOptimizer.COLLECTION_FOR_GRAPH_STATES)
 383 
 384   class _PerGraphState(object):
 385     """Gradient reduction related state of a Tensorflow graph."""
 386 
 387     def __init__(self):
 388       self._collected_grads_and_vars = defaultdict(list)
 389       self._current_tower_index = 0
 390       self._number_of_towers = 1
 391       self._loss_reduction = None
 392       # Scopes of the first tower that don't have a prefix:
 393       self._variable_scope = None
 394       self._name_scope = None
 395       # If needed, alert that TowerOptimizer needs to be used with model_fn.
 396       self._has_tower_optimizer_been_used = False
 397 
 398     def collect_gradients(self, grads_and_vars):
 399       self._collected_grads_and_vars[self._current_tower_index].append(
 400           grads_and_vars)
 401 
 402     def get_latest_gradients_from_all_towers(self):
 403       """Get gradients across towers for the last called optimizer."""
 404       grads_and_vars = []
 405       index_of_last_gradients = len(
 406           self._collected_grads_and_vars[self._current_tower_index]) - 1
 407       for tower_id in range(self._current_tower_index + 1):
 408         grads_and_vars.extend(
 409             self._collected_grads_and_vars[tower_id][index_of_last_gradients])
 410       return grads_and_vars
 411 
 412     def set_reduction_across_towers(self, loss_reduction, number_of_towers):
 413       self._loss_reduction = loss_reduction
 414       self._number_of_towers = number_of_towers
 415 
 416     @contextmanager
 417     def tower(self, tower_id, var_scope, name_scope):
 418       if tower_id == 0:
 419         self._variable_scope = var_scope
 420         self._name_scope = name_scope
 421       self._current_tower_index = tower_id
 422       yield
 423 
 424     @property
 425     def scopes_of_the_first_tower(self):
 426       return self._variable_scope, self._name_scope
 427 
 428     @property
 429     def is_the_last_tower(self):
 430       return self._current_tower_index == (self._number_of_towers - 1)
 431 
 432     @property
 433     def number_of_towers(self):
 434       return self._number_of_towers
 435 
 436     @property
 437     def loss_reduction(self):
 438       return self._loss_reduction
 439 
 440     @property
 441     def has_tower_optimizer_been_used(self):
 442       return self._has_tower_optimizer_been_used
 443 
 444     @has_tower_optimizer_been_used.setter
 445     def has_tower_optimizer_been_used(self, value):
 446       self._has_tower_optimizer_been_used = value
 447 
 448     def did_towers_have_same_optimizer_calls(self):
 449       total_number_of_grads = sum([
 450           len(grads)
 451           for _, grads in six.iteritems(self._collected_grads_and_vars)
 452       ])
 453       return total_number_of_grads % self._number_of_towers == 0

 456 def _get_local_devices(device_type):
 457   local_device_protos = device_lib.list_local_devices()
 458   return [
 459       device.name
 460       for device in local_device_protos
 461       if device.device_type == device_type
 462   ]

 465 def _split_batch(features, labels, number_of_shards, device):
 466   """Split input features and labels into batches."""
 467 
 468   def ensure_divisible_by_shards(sequence):
 469     batch_size = ops_lib.convert_to_tensor(sequence).get_shape()[0]
 470     if batch_size % number_of_shards != 0:
 471       raise ValueError(
 472           'Batch size {} needs to be divisible by the number of GPUs, which '
 473           'is {}.'.format(batch_size, number_of_shards))
 474 
 475   def split_dictionary(dictionary):
 476     """Split a dictionary into shards."""
 477     shards = [{} for _ in range(number_of_shards)]
 478     for name, tensor in six.iteritems(dictionary):
 479       if isinstance(tensor, sparse_tensor.SparseTensor):
 480         for i, shard in enumerate(
 481             sparse_ops.sparse_split(
 482                 sp_input=tensor, num_split=number_of_shards, axis=0)):
 483           shards[i][name] = shard
 484       else:
 485         ensure_divisible_by_shards(tensor)
 486         for i, shard in enumerate(array_ops.split(tensor, number_of_shards)):
 487           shards[i][name] = shard
 488     return shards
 489 
 490   with ops_lib.name_scope('split_inputs'):
 491     with ops_lib.device(device):
 492       if isinstance(features, dict):
 493         feature_shards = split_dictionary(features)
 494       else:
 495         ensure_divisible_by_shards(features)
 496         feature_shards = array_ops.split(features, number_of_shards)
 497 
 498       if labels is None:
 499         label_shards = None
 500       elif isinstance(labels, dict):
 501         label_shards = split_dictionary(labels)
 502       else:
 503         ensure_divisible_by_shards(labels)
 504         label_shards = array_ops.split(labels, number_of_shards)
 505   return feature_shards, label_shards

 511 def _get_loss_towers(model_fn,
 512                      mode,
 513                      features,
 514                      labels,
 515                      params,
 516                      config,
 517                      devices,
 518                      local_ps_devices,
 519                      loss_reduction,
 520                      name_scope_pattern=_DEFAULT_NAME_SCOPE_PATTERN):
 521   """Replicate the loss computation across devices."""
 522   tower_specs = []
 523 
 524   model_fn_args = function_utils.fn_args(model_fn)
 525   optional_params = {}
 526   if 'params' in model_fn_args:
 527     optional_params['params'] = copy.deepcopy(params)
 528   if 'config' in model_fn_args:
 529     optional_params['config'] = copy.deepcopy(config)
 530 
 531   # pylint: disable=protected-access
 532   round_robin_strategy = device_setter_lib._RoundRobinStrategy(
 533       num_tasks=len(local_ps_devices))
 534   TowerOptimizer._graph_state().set_reduction_across_towers(
 535       loss_reduction, len(devices))
 536 
 537   for i, device in enumerate(devices):
 538     is_the_first_tower = (i == 0)
 539 
 540     device_setter = _local_device_setter(
 541         worker_device=device,
 542         ps_devices=local_ps_devices,
 543         ps_strategy=round_robin_strategy)
 544 
 545     # We would like to preserve the names of the variables and ops that the user
 546     # might be relying on. Names without a prefix are going to resolve to
 547     # variables and ops of the first tower.
 548     name_scope = name_scope_pattern
 549     if is_the_first_tower:
 550       name_scope = ''
 551 
 552     with variable_scope.variable_scope(
 553         '', reuse=not is_the_first_tower) as var_scope:
 554       with ops_lib.name_scope(name_scope.format(i)) as name_scope:
 555         with TowerOptimizer._graph_state().tower(
 556             tower_id=i, var_scope=var_scope, name_scope=name_scope):
 557           with ops_lib.device(device_setter):
 558             labels_shard = None
 559             if labels:
 560               labels_shard = labels[i]
 561 
 562             tower_spec = model_fn(
 563                 mode=mode,
 564                 features=features[i],
 565                 labels=labels_shard,
 566                 **optional_params)
 567 
 568             if (tower_spec.train_op is not None and len(devices) > 1 and
 569                 not TowerOptimizer.has_been_used()):
 570               raise ValueError('Please wrap optimizers with TowerOptimizer'
 571                                ' in order to use replicate_model_fn with'
 572                                ' multiple `devices`.')
 573 
 574             # Scaling the loss here doesn't actually affect gradients.  Another
 575             # instance of scaling happens inside the TowerOptimizer.
 576             tower_spec = _scale_tower_loss(
 577                 tower_spec, loss_reduction, number_of_towers=len(devices))
 578             tower_specs.append(tower_spec)
 579 
 580   if not TowerOptimizer._did_towers_have_same_optimizer_calls():
 581     raise ValueError('Each invocation of model_fn was supposed to make the same'
 582                      ' optimizer calls.')
 583   TowerOptimizer._clear_graph_state()
 584   # pylint: enable=protected-access
 585   return tower_specs

 588 def _local_device_setter(worker_device, ps_devices, ps_strategy):
 589   """A device setter that puts distributes Var/Ops to PS/workers."""
 590   ps_ops = ['Variable', 'VariableV2', 'VarHandleOp']
 591 
 592   def local_device_chooser(op):
 593     current_device = framework_device.DeviceSpec.from_string(op.device or '')
 594 
 595     node_def = op if isinstance(op, node_def_pb2.NodeDef) else op.node_def
 596     if node_def.op in ps_ops:
 597       ps_device_spec = framework_device.DeviceSpec.from_string(
 598           '{}'.format(ps_devices[ps_strategy(op)]))
 599 
 600       ps_device_spec.merge_from(current_device)
 601       return ps_device_spec.to_string()
 602     else:
 603       worker_device_spec = framework_device.DeviceSpec.from_string(
 604           worker_device or '')
 605       worker_device_spec.merge_from(current_device)
 606       return worker_device_spec.to_string()
 607 
 608   return local_device_chooser

 611 def _scale_tower_loss(tower_spec, loss_reduction, number_of_towers):
 612   """Produce an EstimatorSpec with appropriately scaled loss."""
 613   if tower_spec.loss is None:
 614     return tower_spec
 615 
 616   estimator_spec = _asdict(tower_spec)
 617   estimator_spec['loss'] = _scale_loss(tower_spec.loss, loss_reduction,
 618                                        number_of_towers)
 619   return model_fn_lib.EstimatorSpec(**estimator_spec)

 622 def _scale_loss(loss, loss_reduction, number_of_towers):
 623   """If needed, scale down the loss for averaging loss by summing."""
 624   if loss is None:
 625     return None
 626   if number_of_towers == 1:
 627     return loss
 628 
 629   if loss_reduction != losses.Reduction.SUM:
 630     return math_ops.div(loss, 1.0 * number_of_towers, name='averaged_loss')
 631   else:
 632     return loss

 640 def _compute_sum_on_device(values, device, name=None):
 641   with ops_lib.device(device):
 642     if isinstance(values[0], ops_lib.IndexedSlices):
 643       if name:
 644         raise ValueError('The name {} is not expected to be given to '
 645                          'IndexedSlices {}'.format(name, values))
 646 
 647       values_concat = array_ops.concat([v.values for v in values], axis=0)
 648       indices_concat = array_ops.concat([v.indices for v in values], axis=0)
 649       return ops_lib.IndexedSlices(values_concat, indices_concat,
 650                                    values[0].dense_shape)
 651     else:
 652       return math_ops.add_n(values, name=name)

 655 def _train_spec(tower_specs,
 656                 train_op,
 657                 aggregation_device,
 658                 aggregated_loss_name='loss'):
 659   """Populate replicated EstimatorSpec for `GraphKeys.TRAIN`."""
 660   # Spec of the last tower is used as the template for the final spec, because
 661   # some `EstimatorSpec.training_hooks` rely on calls made in model_fn.  For
 662   # example, `SyncReplicasOptimizerHook` validates the
 663   # `SyncReplicasOptimizer.apply_gradients` call. `TowerEstimator` makes that
 664   # call only in the last tower.
 665   estimator_spec = _asdict(tower_specs[-1])
 666   estimator_spec['mode'] = model_fn_lib.ModeKeys.TRAIN
 667   estimator_spec['train_op'] = train_op
 668   estimator_spec['loss'] = _compute_sum_on_device(
 669       [spec.loss for spec in tower_specs], aggregation_device,
 670       aggregated_loss_name)
 671   return model_fn_lib.EstimatorSpec(**estimator_spec)

 674 def _eval_spec(tower_specs, aggregation_device, aggregated_loss_name='loss'):
 675   """Populate replicated EstimatorSpec for `GraphKeys.EVAL`."""
 676   estimator_spec = _asdict(tower_specs[0])
 677   estimator_spec['mode'] = model_fn_lib.ModeKeys.EVAL
 678   estimator_spec['loss'] = _compute_sum_on_device(
 679       [spec.loss for spec in tower_specs], aggregation_device,
 680       aggregated_loss_name)
 681 
 682   update_ops = []
 683   for tower_spec in tower_specs:
 684     for name, (_, update_op) in six.iteritems(tower_spec.eval_metric_ops):
 685       update_ops.append(update_op)
 686 
 687   with ops_lib.control_dependencies(update_ops):
 688     reduced_update_op = _reduce_metric_variables(len(tower_specs))
 689 
 690   eval_metric_ops = {}
 691   for name, (metric_tensor, _) in six.iteritems(tower_specs[0].eval_metric_ops):
 692     eval_metric_ops[name] = (metric_tensor, reduced_update_op)
 693   estimator_spec['eval_metric_ops'] = eval_metric_ops
 694   return model_fn_lib.EstimatorSpec(**estimator_spec)

 697 def _reduce_metric_variables(number_of_towers):
 698   """Aggregate local variables used in metrics into the first tower."""
 699   if number_of_towers == 1:
 700     return control_flow_ops.no_op(name='no_eval_metric_reduction')
 701 
 702   metric_variables = ops_lib.get_collection(ops_lib.GraphKeys.METRIC_VARIABLES)
 703   variables_per_tower = len(metric_variables) // number_of_towers
 704 
 705   if len(metric_variables) % number_of_towers != 0:
 706     raise ValueError(
 707         'Different `EstimatorSpec.eval_metric_ops` across `model_fn()` calls.'
 708         ' Expected {} local variables, but got {} instead.'.format(
 709             variables_per_tower * number_of_towers, len(metric_variables)))
 710 
 711   # `metric_variables` has the size of `variables_per_tower` x
 712   #  number_of_towers.  Each tower is produced by calling the same model_fn.
 713   #  First `variables_per_tower` correspond to the first tower.  Each such
 714   #  variable has an replica at the `(variables_per_tower * i)` position, where
 715   #  `i` is `[1.. number_of_towers]`.  We are going to add values from replicas
 716   #  to each variable of the first tower.  We then zero out replica values, so
 717   #  that `_reduce_metric_variables` operation is idempotent.  If a metric
 718   #  is then computed based on local variables from the first tower, then the
 719   #  resulting metric is an estimate for all `number_of_towers` towers.
 720   ops = []
 721   for i in range(0, variables_per_tower):
 722     next_replica_id = i + variables_per_tower
 723     replicas = [
 724         metric_variables[replica_id]
 725         for replica_id in range(next_replica_id, len(metric_variables),
 726                                 variables_per_tower)
 727     ]  #  `replicas` doesn't contain the first-tower variable.
 728 
 729     reduce_op = state_ops.assign_add(metric_variables[i],
 730                                      math_ops.add_n(replicas))
 731 
 732     with ops_lib.control_dependencies([reduce_op]):
 733       for replica in replicas:
 734         zeros_for_replica = array_ops.zeros(
 735             array_ops.shape(replica), dtype=replica.dtype)
 736         zero_out_replica_op = state_ops.assign(replica, zeros_for_replica)
 737         ops.append(zero_out_replica_op)
 738 
 739   return control_flow_ops.group(*ops)

 742 def _predict_spec(tower_specs, aggregation_device):
 743   """Populate replicated EstimatorSpec for `GraphKeys.PREDICT`."""
 744   estimator_spec = _asdict(tower_specs[0])
 745   estimator_spec['mode'] = model_fn_lib.ModeKeys.PREDICT
 746 
 747   with ops_lib.device(aggregation_device):
 748     estimator_spec['predictions'] = _concat_tensor_dicts(
 749         *[tower_spec.predictions for tower_spec in tower_specs])
 750 
 751     export_outputs_dict = _dict_concat(
 752         *[tower_spec.export_outputs for tower_spec in tower_specs])
 753 
 754     export_outputs = {}
 755     for name, export_output_list in six.iteritems(export_outputs_dict):
 756       if isinstance(export_output_list[0], export_output_lib.PredictOutput):
 757         export_outputs[name] = export_output_lib.PredictOutput(
 758             outputs=_concat_tensor_dicts(*[
 759                 export_output.outputs for export_output in export_output_list
 760             ]))
 761       elif isinstance(export_output_list[0],
 762                       export_output_lib.RegressionOutput):
 763         export_outputs[name] = export_output_lib.RegressionOutput(
 764             value=array_ops.concat(
 765                 [export_output.value for export_output in export_output_list],
 766                 axis=0))
 767       elif isinstance(export_output_list[0],
 768                       export_output_lib.ClassificationOutput):
 769         scores = None
 770         if export_output_list[0].scores is not None:
 771           scores = array_ops.concat(
 772               [export_output.scores for export_output in export_output_list],
 773               axis=0)
 774 
 775         classes = None
 776         if export_output_list[0].classes is not None:
 777           classes = array_ops.stack(
 778               [export_output.classes for export_output in export_output_list],
 779               axis=0)
 780 
 781         export_outputs[name] = export_output_lib.ClassificationOutput(
 782             scores=scores, classes=classes)
 783 
 784   estimator_spec['export_outputs'] = export_outputs
 785   return model_fn_lib.EstimatorSpec(**estimator_spec)

 788 def _concat_tensor_dicts(*tensor_dicts):
 789   return {
 790       name: array_ops.concat(tensors, axis=0, name=name)
 791       for name, tensors in six.iteritems(_dict_concat(*tensor_dicts))
 792   }

 795 def _extract_tensors(tensors_and_vars):
 796   tensors = []
 797   for tensor_and_var in tensors_and_vars:
 798     tensor, _ = tensor_and_var
 799     if isinstance(tensor, ops_lib.IndexedSlices):
 800       tensors.append(tensor.values)
 801     elif tensor is not None:
 802       tensors.append(tensor)
 803   return tensors

 806 def _dict_concat(*dicts):
 807   list_dict = {}
 808   for d in dicts:
 809     if d is None:
 810       continue
 811 
 812     for k, v in six.iteritems(d):
 813       list_dict.setdefault(k, []).append(v)
 814   return list_dict

 817 def _asdict(namedtuple):
 818   """Returns a namedtuple as a dictionary.
 819 
 820   This is required because `_asdict()` in Python 3.x.x is broken in classes
 821   that inherit from `collections.namedtuple`. See
 822   https://bugs.python.org/issue24931 for more details.
 823 
 824   Args:
 825     namedtuple: An object that inherits from `collections.namedtuple`.
 826 
 827   Returns:
 828     A dictionary version of the tuple.
 829   """
 830   return {k: getattr(namedtuple, k) for k in namedtuple._fields}
