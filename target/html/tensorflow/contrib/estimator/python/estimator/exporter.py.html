  32 class StepsExporter(exporter.Exporter):
  33   """This class exports the model in user specified steps.
  34 
  35   This class exports the model at the steps given by the `steps_to_keep`
  36   argument. Each number in the list is treated as a lower bound for model
  37   exports, to handle the case when evaluation is performed at different steps.
  38 
  39   Consider this example:
  40 
  41   ```
  42   steps_to_keep = [1, 2, 3, 6, 7, 10, 12, 25]
  43   ```
  44 
  45   The model is evaluated at step increments of 5: `[5, 10, 15, 20, 25, 30]`.
  46   The `StepsExporter` will export the model when it has reached steps
  47   `[5, 10, 15, 25]`.
  48 
  49   This example illustrates the two cases when the model is exported:
  50 
  51   1. Model is evaluated on a step defined in the list `steps_to_keep`.
  52 
  53      In the example, the model is exported on step `10` and `25`.
  54 
  55   2. Model is evaluated on a step not defined in the list `steps_to_keep`, but
  56      is still exported because a step in `steps_to_keep` was missed.
  57 
  58      In the example, when the model reaches step `5`, the model is exported even
  59      though  `steps_to_keep` does not contain `5`. Step `5` is exported to make
  60      up for step `3`, which was missed. Steps `1` and `2` in `steps_to_keep` are
  61      skipped completely (e.g. say the model is evaluated at step `6`. It will
  62      **not** be exported to make up for step `2`).
  63 
  64   Using the `steps_to_keep` list as a lower bound allows users to define
  65   approximate step boundaries for exporting their models, and avoid frustrating
  66   off-by-one calculation errors.
  67 
  68   Sample Use Cases:
  69     There are specific points during the training when having a saved version of
  70     the model would be useful. One example is at the end of each training phase
  71     when the set of freezed weights is changed.
  72     Another good use case is saving the model at the end of each epoch for
  73     visualization or retraining.
  74   """
  75 
  76   def __init__(self,
  77                steps_to_keep,
  78                name='steps_exporter',
  79                serving_input_receiver_fn=None,
  80                event_file_pattern='eval/*.tfevents.*',
  81                assets_extra=None,
  82                as_text=False):
  83     """Create an `StepsExporter` to use with `tf.estimator.EvalSpec`.
  84 
  85     Example of creating a StepsExporter for training and evaluation:
  86 
  87     ```python
  88     categorical_feature_a = categorical_column_with_hash_bucket(...)
  89     categorical_feature_b = categorical_column_with_hash_bucket(...)
  90 
  91     categorical_feature_a_emb = embedding_column(
  92         categorical_column=categorical_feature_a, ...)
  93     categorical_feature_b_emb = embedding_column(
  94         categorical_column=categorical_feature_b, ...)
  95 
  96     estimator = tf.estimator.DNNClassifier(
  97         feature_columns=[categorical_feature_a_emb, categorical_feature_b_emb],
  98         hidden_units=[1024, 512, 256])
  99 
 100     # Input pipeline for train and evaluate.
 101     def train_input_fn: # returns x, y
 102       # please shuffle the data.
 103       pass
 104     def eval_input_fn_eval: # returns x, y
 105       pass
 106 
 107     exporter = tf.contrib.estimator.exporter.StepsExporter(
 108         name="steps_exporter",
 109         serving_input_receiver_fn=serving_input_receiver_fn,
 110         event_file_pattern='eval/*.tfevents.*'
 111         steps_to_keep=[...])
 112 
 113     train_spec = tf.estimator.TrainSpec(input_fn=train_input_fn, max_steps=1000)
 114 
 115     eval_spec = [tf.estimator.EvalSpec(
 116       input_fn=eval_input_fn,
 117       steps=1,
 118       exporters=exporter,
 119       start_delay_secs=0,
 120       throttle_secs=5)]
 121 
 122     tf.estimator.train_and_evaluate(estimator, train_spec, eval_spec)
 123 
 124     # Models will be exported to estimator.model_dir in timestamped directories,
 125     # which can be used for serving, analysis with TFMA, or directly loaded in.
 126     # For example:
 127     export_dir = os.path.join(estimator.model_dir,
 128                               <timestamped directory name>)
 129 
 130     with ops.Graph().as_default() as graph:
 131       with session.Session(graph=graph) as sess:
 132         tf.saved_model.loader.load(
 133             sess, [tf.saved_model.tag_constants.SERVING], export_dir)
 134 
 135     ```
 136 
 137     Args:
 138       steps_to_keep: Non-empty list of positive integers containing
 139         the step numbers at which the model should be exported. All the exports
 140         will be kept, so there is no garbage collection.
 141       name: Unique name of this `Exporter` that is going to be used in the
 142         export path.
 143       serving_input_receiver_fn: A function that takes no arguments and returns
 144         a `ServingInputReceiver`.
 145       event_file_pattern: Event file name pattern relative to model_dir. If
 146         None, however, the exporter would not be preemption-safe. To be
 147         preemption-safe, event_file_pattern should be specified.
 148       assets_extra: An optional dict specifying how to populate the assets.extra
 149         directory within the exported SavedModel.  Each key should give the
 150         destination path (including the filename) relative to the assets.extra
 151         directory.  The corresponding value gives the full path of the source
 152         file to be copied.  For example, the simple case of copying a single
 153         file without renaming it is specified as `{'my_asset_file.txt':
 154         '/path/to/my_asset_file.txt'}`.
 155       as_text: Whether to write the SavedModel proto in text format. Defaults to
 156         `False`.
 157 
 158     Raises:
 159       ValueError: If any arguments is invalid.
 160     """
 161     # pylint: disable=protected-access
 162     self._saved_model_exporter = exporter._SavedModelExporter(
 163         name, serving_input_receiver_fn, assets_extra, as_text)
 164     # pylint: enable=protected-access
 165 
 166     self._event_file_pattern = event_file_pattern
 167     self._model_dir = None
 168 
 169     self._input_steps_to_keep = steps_to_keep
 170     steps_to_keep = [step for step in steps_to_keep if isinstance(step, int)]
 171     steps_to_keep = [step for step in steps_to_keep if step > 0]
 172     if not steps_to_keep:
 173       raise ValueError(
 174           '`steps_to_keep` list must have at least one positive integer')
 175     elif self._input_steps_to_keep != steps_to_keep:
 176       tf_logging.warn('Changed `steps_to_keep`, by omitting non-integer or'
 177                       ' less than 1 elements, to [%s]',
 178                       ', '.join(str(step) for step in steps_to_keep))
 179     self._steps_to_keep = sorted(steps_to_keep)
 180     self._steps_kept = []
 181 
 182   @property
 183   def name(self):
 184     return self._saved_model_exporter.name
 185 
 186   def export(self, estimator, export_path, checkpoint_path, eval_result,
 187              is_the_final_export):
 188     """Exports the given Estimator to a specific format.
 189 
 190     Args:
 191       estimator: A `tf.estimator.Estimator` instance to export.
 192       export_path: A string containing a directory where to write the export.
 193       checkpoint_path: The checkpoint path to export.
 194       eval_result: The output of Estimator.evaluate on this checkpoint.
 195       is_the_final_export: This boolean is True when this is an export in the
 196         end of training. It is False for the intermediate exports during the
 197         training. When passing Exporter to tf.estimator.train_and_evaluate
 198         is_the_final_export is always False if TrainSpec.max_steps is None.
 199 
 200     Returns:
 201       The string path to the exported directory or None if export is skipped.
 202 
 203     Raises:
 204       ValueError: If `eval_result` is None or doesn't have
 205         `ops.GraphKeys.GLOBAL_STEP` as a key.
 206     """
 207     export_result = None
 208 
 209     if not eval_result or DEFAULT_GLOBAL_STEP_KEY not in eval_result:
 210       raise ValueError(
 211           '`eval_result` is empty, or does not have global step. This'
 212           ' should never happen as Estimator always sets the global step in '
 213           '`eval_result`. Please file a bug report. Got eval_result: %s'
 214           % str(eval_result))
 215 
 216     if self._model_dir != estimator.model_dir and self._event_file_pattern:
 217       tf_logging.info('Loads the steps that the model was already evaluated at,'
 218                       'from event files')
 219       self._model_dir = estimator.model_dir
 220       full_event_file_pattern = os.path.join(self._model_dir,
 221                                              self._event_file_pattern)
 222       self._steps_kept = self._get_kept_steps(full_event_file_pattern)
 223 
 224       if self._steps_kept:
 225         self._steps_kept = sorted(self._steps_kept)
 226         self._steps_to_keep = [step for step in self._steps_to_keep if
 227                                step > self._steps_kept[-1]]
 228     # It is assumed that the model is exported at any evaluated step 'n' if
 229     # there is any `steps_missed` lower than 'n'. As a result, all the steps in
 230     # `_steps_to_keep` lower than the last evaluated step will be removed.
 231     steps_missed = [step for step in self._steps_to_keep
 232                     if step <= eval_result[DEFAULT_GLOBAL_STEP_KEY]]
 233 
 234     if steps_missed:
 235       # update the `_steps_to_keep` list by omitting all steps smaller than the
 236       # current global step which are missed to be exported
 237       export_result = self._saved_model_exporter.export(estimator, export_path,
 238                                                         checkpoint_path,
 239                                                         eval_result,
 240                                                         is_the_final_export)
 241       self._steps_to_keep = [step for step in self._steps_to_keep if step
 242                              not in steps_missed]
 243       # contains all the steps in which export has happened.
 244       self._steps_kept.append(eval_result[DEFAULT_GLOBAL_STEP_KEY])
 245       # Show warning for all the missed steps except the last one
 246       if steps_missed[:-1]:
 247         tf_logging.warn('Missed steps [%s] for exporting, as no evaluation'
 248                         ' took place at them.', ', '.join(str(step) for step in
 249                                                           steps_missed[:-1]))
 250       # Log model export if the last missed step is the same as the current step
 251       if steps_missed[-1] == eval_result[DEFAULT_GLOBAL_STEP_KEY]:
 252         tf_logging.info('Performing model export at step %d.',
 253                         eval_result[DEFAULT_GLOBAL_STEP_KEY])
 254       # Show warning for exporting model at another step instead of the user
 255       #   specified one
 256       else:
 257         tf_logging.warn('Performing model export at step %d instead of %d, as'
 258                         ' no evaluation took place at step %d.',
 259                         eval_result[DEFAULT_GLOBAL_STEP_KEY], steps_missed[-1],
 260                         steps_missed[-1])
 261     return export_result
 262 
 263   def _get_kept_steps(self, event_files):
 264     """Get the steps that the model was evaluated at, from event files.
 265 
 266     Args:
 267       event_files: Absolute pattern of event files.
 268 
 269     Returns:
 270       steps_kept: A list of steps in which the model was evaluated.
 271     """
 272     if not event_files:
 273       return None
 274 
 275     steps_kept = []
 276     for event_file in gfile.Glob(os.path.join(event_files)):
 277       for event in summary_iterator.summary_iterator(event_file):
 278         if event.step not in steps_kept:
 279           steps_kept.append(event.step)
 280     return steps_kept
