  41 def make_early_stopping_hook(estimator,
  42                              should_stop_fn,
  43                              run_every_secs=60,
  44                              run_every_steps=None):
  45   """Creates early-stopping hook.
  46 
  47   Returns a `SessionRunHook` that stops training when `should_stop_fn` returns
  48   `True`.
  49 
  50   Usage example:
  51 
  52   ```python
  53   estimator = ...
  54   hook = early_stopping.make_early_stopping_hook(
  55       estimator, should_stop_fn=make_stop_fn(...))
  56   train_spec = tf.estimator.TrainSpec(..., hooks=[hook])
  57   tf.estimator.train_and_evaluate(estimator, train_spec, ...)
  58   ```
  59 
  60   Caveat: Current implementation supports early-stopping both training and
  61   evaluation in local mode. In distributed mode, training can be stopped but
  62   evaluation (where it's a separate job) will indefinitely wait for new model
  63   checkpoints to evaluate, so you will need other means to detect and stop it.
  64   Early-stopping evaluation in distributed mode requires changes in
  65   `train_and_evaluate` API and will be addressed in a future revision.
  66 
  67   Args:
  68     estimator: A `tf.estimator.Estimator` instance.
  69     should_stop_fn: `callable`, function that takes no arguments and returns a
  70       `bool`. If the function returns `True`, stopping will be initiated by the
  71       chief.
  72     run_every_secs: If specified, calls `should_stop_fn` at an interval of
  73       `run_every_secs` seconds. Defaults to 60 seconds. Either this or
  74       `run_every_steps` must be set.
  75     run_every_steps: If specified, calls `should_stop_fn` every
  76       `run_every_steps` steps. Either this or `run_every_secs` must be set.
  77 
  78   Returns:
  79     A `SessionRunHook` that periodically executes `should_stop_fn` and initiates
  80     early stopping if the function returns `True`.
  81 
  82   Raises:
  83     TypeError: If `estimator` is not of type `tf.estimator.Estimator`.
  84     ValueError: If both `run_every_secs` and `run_every_steps` are set.
  85   """
  86   if not isinstance(estimator, estimator_lib.Estimator):
  87     raise TypeError('`estimator` must have type `tf.estimator.Estimator`. '
  88                     'Got: {}'.format(type(estimator)))
  89 
  90   if run_every_secs is not None and run_every_steps is not None:
  91     raise ValueError('Only one of `run_every_secs` and `run_every_steps` must '
  92                      'be set.')
  93 
  94   if estimator.config.is_chief:
  95     return _StopOnPredicateHook(should_stop_fn, run_every_secs, run_every_steps)
  96   else:
  97     return _CheckForStoppingHook()

 326 def read_eval_metrics(eval_dir):
 327   """Helper to read eval metrics from eval summary files.
 328 
 329   Args:
 330     eval_dir: Directory containing summary files with eval metrics.
 331 
 332   Returns:
 333     A `dict` with global steps mapping to `dict` of metric names and values.
 334   """
 335   eval_metrics_dict = {}
 336   for event in _summaries(eval_dir):
 337     if not event.HasField('summary'):
 338       continue
 339     metrics = {}
 340     for value in event.summary.value:
 341       if value.HasField('simple_value'):
 342         metrics[value.tag] = value.simple_value
 343     if metrics:
 344       eval_metrics_dict[event.step] = metrics
 345   return collections.OrderedDict(
 346       sorted(eval_metrics_dict.items(), key=lambda t: t[0]))

 349 def _stop_if_threshold_crossed_hook(estimator, metric_name, threshold,
 350                                     higher_is_better, eval_dir, min_steps,
 351                                     run_every_secs, run_every_steps):
 352   """Creates early-stopping hook to stop training if threshold is crossed."""
 353 
 354   if eval_dir is None:
 355     eval_dir = estimator.eval_dir()
 356 
 357   is_lhs_better = operator.gt if higher_is_better else operator.lt
 358   greater_or_lesser = 'greater than' if higher_is_better else 'less than'
 359 
 360   def stop_if_threshold_crossed_fn():
 361     """Returns `True` if the given metric crosses specified threshold."""
 362 
 363     eval_results = read_eval_metrics(eval_dir)
 364 
 365     for step, metrics in eval_results.items():
 366       if step < min_steps:
 367         continue
 368       val = metrics[metric_name]
 369       if is_lhs_better(val, threshold):
 370         tf_logging.info(
 371             'At step %s, metric "%s" has value %s which is %s the configured '
 372             'threshold (%s) for early stopping.', step, metric_name, val,
 373             greater_or_lesser, threshold)
 374         return True
 375     return False
 376 
 377   return make_early_stopping_hook(
 378       estimator=estimator,
 379       should_stop_fn=stop_if_threshold_crossed_fn,
 380       run_every_secs=run_every_secs,
 381       run_every_steps=run_every_steps)

 384 def _stop_if_no_metric_improvement_hook(
 385     estimator, metric_name, max_steps_without_improvement, higher_is_better,
 386     eval_dir, min_steps, run_every_secs, run_every_steps):
 387   """Returns hook to stop training if given metric shows no improvement."""
 388 
 389   if eval_dir is None:
 390     eval_dir = estimator.eval_dir()
 391 
 392   is_lhs_better = operator.gt if higher_is_better else operator.lt
 393   increase_or_decrease = 'increase' if higher_is_better else 'decrease'
 394 
 395   def stop_if_no_metric_improvement_fn():
 396     """Returns `True` if metric does not improve within max steps."""
 397 
 398     eval_results = read_eval_metrics(eval_dir)
 399 
 400     best_val = None
 401     best_val_step = None
 402     for step, metrics in eval_results.items():
 403       if step < min_steps:
 404         continue
 405       val = metrics[metric_name]
 406       if best_val is None or is_lhs_better(val, best_val):
 407         best_val = val
 408         best_val_step = step
 409       if step - best_val_step >= max_steps_without_improvement:
 410         tf_logging.info(
 411             'No %s in metric "%s" for %s steps, which is greater than or equal '
 412             'to max steps (%s) configured for early stopping.',
 413             increase_or_decrease, metric_name, step - best_val_step,
 414             max_steps_without_improvement)
 415         return True
 416     return False
 417 
 418   return make_early_stopping_hook(
 419       estimator=estimator,
 420       should_stop_fn=stop_if_no_metric_improvement_fn,
 421       run_every_secs=run_every_secs,
 422       run_every_steps=run_every_steps)

 425 def _summaries(eval_dir):
 426   """Yields `tensorflow.Event` protos from event files in the eval dir.
 427 
 428   Args:
 429     eval_dir: Directory containing summary files with eval metrics.
 430 
 431   Yields:
 432     `tensorflow.Event` object read from the event files.
 433   """
 434   if gfile.Exists(eval_dir):
 435     for event_file in gfile.Glob(
 436         os.path.join(eval_dir, _EVENT_FILE_GLOB_PATTERN)):
 437       for event in summary_iterator.summary_iterator(event_file):
 438         yield event

 441 def _get_or_create_stop_var():
 442   with variable_scope.variable_scope(
 443       name_or_scope='signal_early_stopping',
 444       values=[],
 445       reuse=variable_scope.AUTO_REUSE):
 446     return variable_scope.get_variable(
 447         name='STOP',
 448         shape=[],
 449         dtype=dtypes.bool,
 450         initializer=init_ops.constant_initializer(False),
 451         collections=[ops.GraphKeys.GLOBAL_VARIABLES],
 452         trainable=False)

 455 class _StopOnPredicateHook(session_run_hook.SessionRunHook):
 456   """Hook that requests stop when `should_stop_fn` returns `True`."""
 457 
 458   def __init__(self, should_stop_fn, run_every_secs=60, run_every_steps=None):
 459     if not callable(should_stop_fn):
 460       raise TypeError('`should_stop_fn` must be callable.')
 461 
 462     self._should_stop_fn = should_stop_fn
 463     self._timer = basic_session_run_hooks.SecondOrStepTimer(
 464         every_secs=run_every_secs, every_steps=run_every_steps)
 465     self._global_step_tensor = None
 466     self._stop_var = None
 467     self._stop_op = None
 468 
 469   def begin(self):
 470     self._global_step_tensor = training_util.get_global_step()
 471     self._stop_var = _get_or_create_stop_var()
 472     self._stop_op = state_ops.assign(self._stop_var, True)
 473 
 474   def before_run(self, run_context):
 475     del run_context
 476     return session_run_hook.SessionRunArgs(self._global_step_tensor)
 477 
 478   def after_run(self, run_context, run_values):
 479     global_step = run_values.results
 480     if self._timer.should_trigger_for_step(global_step):
 481       self._timer.update_last_triggered_step(global_step)
 482       if self._should_stop_fn():
 483         tf_logging.info('Requesting early stopping at global step %d',
 484                         global_step)
 485         run_context.session.run(self._stop_op)
 486         run_context.request_stop()
 487

 489 class _CheckForStoppingHook(session_run_hook.SessionRunHook):
 490   """Hook that requests stop if stop is requested by `_StopOnPredicateHook`."""
 491 
 492   def __init__(self):
 493     self._stop_var = None
 494 
 495   def begin(self):
 496     self._stop_var = _get_or_create_stop_var()
 497 
 498   def before_run(self, run_context):
 499     del run_context
 500     return session_run_hook.SessionRunArgs(self._stop_var)
 501 
 502   def after_run(self, run_context, run_values):
 503     should_early_stop = run_values.results
 504     if should_early_stop:
 505       tf_logging.info('Early stopping requested, suspending run.')
 506       run_context.request_stop()
