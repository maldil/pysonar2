  46 def multi_class_head(n_classes,
  47                      weight_column=None,
  48                      label_vocabulary=None,
  49                      loss_reduction=losses.Reduction.SUM_OVER_BATCH_SIZE,
  50                      loss_fn=None,
  51                      name=None):
  52   """Creates a `_Head` for multi class classification.
  53 
  54   Uses `sparse_softmax_cross_entropy` loss.
  55 
  56   The head expects `logits` with shape `[D0, D1, ... DN, n_classes]`.
  57   In many applications, the shape is `[batch_size, n_classes]`.
  58 
  59   `labels` must be a dense `Tensor` with shape matching `logits`, namely
  60   `[D0, D1, ... DN, 1]`. If `label_vocabulary` given, `labels` must be a string
  61   `Tensor` with values from the vocabulary. If `label_vocabulary` is not given,
  62   `labels` must be an integer `Tensor` with values specifying the class index.
  63 
  64   If `weight_column` is specified, weights must be of shape
  65   `[D0, D1, ... DN]`, or `[D0, D1, ... DN, 1]`.
  66 
  67   The loss is the weighted sum over the input dimensions. Namely, if the input
  68   labels have shape `[batch_size, 1]`, the loss is the weighted sum over
  69   `batch_size`.
  70 
  71   Also supports custom `loss_fn`. `loss_fn` takes `(labels, logits)` or
  72   `(labels, logits, features)` as arguments and returns unreduced loss with
  73   shape `[D0, D1, ... DN, 1]`. `loss_fn` must support integer `labels` with
  74   shape `[D0, D1, ... DN, 1]`. Namely, the head applies `label_vocabulary` to
  75   the input labels before passing them to `loss_fn`.
  76 
  77   The head can be used with a canned estimator. Example:
  78 
  79   ```python
  80   my_head = tf.contrib.estimator.multi_class_head(n_classes=3)
  81   my_estimator = tf.contrib.estimator.DNNEstimator(
  82       head=my_head,
  83       hidden_units=...,
  84       feature_columns=...)
  85   ```
  86 
  87   It can also be used with a custom `model_fn`. Example:
  88 
  89   ```python
  90   def _my_model_fn(features, labels, mode):
  91     my_head = tf.contrib.estimator.multi_class_head(n_classes=3)
  92     logits = tf.keras.Model(...)(features)
  93 
  94     return my_head.create_estimator_spec(
  95         features=features,
  96         mode=mode,
  97         labels=labels,
  98         optimizer=tf.AdagradOptimizer(learning_rate=0.1),
  99         logits=logits)
 100 
 101   my_estimator = tf.estimator.Estimator(model_fn=_my_model_fn)
 102   ```
 103 
 104   Args:
 105     n_classes: Number of classes, must be greater than 2 (for 2 classes, use
 106       `binary_classification_head`).
 107     weight_column: A string or a `_NumericColumn` created by
 108       `tf.feature_column.numeric_column` defining feature column representing
 109       weights. It is used to down weight or boost examples during training. It
 110       will be multiplied by the loss of the example.
 111     label_vocabulary: A list or tuple of strings representing possible label
 112       values. If it is not given, that means labels are already encoded as an
 113       integer within [0, n_classes). If given, labels must be of string type and
 114       have any value in `label_vocabulary`. Note that errors will be raised if
 115       `label_vocabulary` is not provided but labels are strings.
 116     loss_reduction: One of `tf.losses.Reduction` except `NONE`. Describes how to
 117       reduce training loss over batch. Defaults to `SUM_OVER_BATCH_SIZE`, namely
 118       weighted sum of losses divided by batch size. See `tf.losses.Reduction`.
 119     loss_fn: Optional loss function.
 120     name: name of the head. If provided, summary and metrics keys will be
 121       suffixed by `"/" + name`. Also used as `name_scope` when creating ops.
 122 
 123   Returns:
 124     An instance of `_Head` for multi class classification.
 125 
 126   Raises:
 127     ValueError: if `n_classes`, `label_vocabulary` or `loss_reduction` is
 128       invalid.
 129   """
 130   return head_lib._multi_class_head_with_softmax_cross_entropy_loss(  # pylint:disable=protected-access
 131       n_classes=n_classes,
 132       weight_column=weight_column,
 133       label_vocabulary=label_vocabulary,
 134       loss_reduction=loss_reduction,
 135       loss_fn=loss_fn,
 136       name=name)

 139 def binary_classification_head(
 140     weight_column=None,
 141     thresholds=None,
 142     label_vocabulary=None,
 143     loss_reduction=losses.Reduction.SUM_OVER_BATCH_SIZE,
 144     loss_fn=None,
 145     name=None):
 146   """Creates a `_Head` for single label binary classification.
 147 
 148   This head uses `sigmoid_cross_entropy_with_logits` loss.
 149 
 150   The head expects `logits` with shape `[D0, D1, ... DN, 1]`.
 151   In many applications, the shape is `[batch_size, 1]`.
 152 
 153   `labels` must be a dense `Tensor` with shape matching `logits`, namely
 154   `[D0, D1, ... DN, 1]`. If `label_vocabulary` given, `labels` must be a string
 155   `Tensor` with values from the vocabulary. If `label_vocabulary` is not given,
 156   `labels` must be float `Tensor` with values in the interval `[0, 1]`.
 157 
 158   If `weight_column` is specified, weights must be of shape
 159   `[D0, D1, ... DN]`, or `[D0, D1, ... DN, 1]`.
 160 
 161   The loss is the weighted sum over the input dimensions. Namely, if the input
 162   labels have shape `[batch_size, 1]`, the loss is the weighted sum over
 163   `batch_size`.
 164 
 165   Also supports custom `loss_fn`. `loss_fn` takes `(labels, logits)` or
 166   `(labels, logits, features)` as arguments and returns unreduced loss with
 167   shape `[D0, D1, ... DN, 1]`. `loss_fn` must support float `labels` with
 168   shape `[D0, D1, ... DN, 1]`. Namely, the head applies `label_vocabulary` to
 169   the input labels before passing them to `loss_fn`.
 170 
 171   The head can be used with a canned estimator. Example:
 172 
 173   ```python
 174   my_head = tf.contrib.estimator.binary_classification_head()
 175   my_estimator = tf.contrib.estimator.DNNEstimator(
 176       head=my_head,
 177       hidden_units=...,
 178       feature_columns=...)
 179   ```
 180 
 181   It can also be used with a custom `model_fn`. Example:
 182 
 183   ```python
 184   def _my_model_fn(features, labels, mode):
 185     my_head = tf.contrib.estimator.binary_classification_head()
 186     logits = tf.keras.Model(...)(features)
 187 
 188     return my_head.create_estimator_spec(
 189         features=features,
 190         mode=mode,
 191         labels=labels,
 192         optimizer=tf.AdagradOptimizer(learning_rate=0.1),
 193         logits=logits)
 194 
 195   my_estimator = tf.estimator.Estimator(model_fn=_my_model_fn)
 196   ```
 197 
 198   Args:
 199     weight_column: A string or a `_NumericColumn` created by
 200       `tf.feature_column.numeric_column` defining feature column representing
 201       weights. It is used to down weight or boost examples during training. It
 202       will be multiplied by the loss of the example.
 203     thresholds: Iterable of floats in the range `(0, 1)`. For binary
 204       classification metrics such as precision and recall, an eval metric is
 205       generated for each threshold value. This threshold is applied to the
 206       logistic values to determine the binary classification (i.e., above the
 207       threshold is `true`, below is `false`.
 208     label_vocabulary: A list or tuple of strings representing possible label
 209       values. If it is not given, labels must be float with values within
 210       [0, 1]. If given, labels must be string type and have any value in
 211       `label_vocabulary`. Note that errors will be raised if `label_vocabulary`
 212       is not provided but labels are strings.
 213     loss_reduction: One of `tf.losses.Reduction` except `NONE`. Describes how to
 214       reduce training loss over batch. Defaults to `SUM_OVER_BATCH_SIZE`, namely
 215       weighted sum of losses divided by batch size. See `tf.losses.Reduction`.
 216     loss_fn: Optional loss function.
 217     name: name of the head. If provided, summary and metrics keys will be
 218       suffixed by `"/" + name`. Also used as `name_scope` when creating ops.
 219 
 220   Returns:
 221     An instance of `_Head` for binary classification.
 222 
 223   Raises:
 224     ValueError: If `thresholds` contains a value outside of `(0, 1)`.
 225     ValueError: If `loss_reduction` is invalid.
 226   """
 227   return head_lib._binary_logistic_head_with_sigmoid_cross_entropy_loss(  # pylint:disable=protected-access
 228       weight_column=weight_column,
 229       thresholds=thresholds,
 230       label_vocabulary=label_vocabulary,
 231       loss_reduction=loss_reduction,
 232       loss_fn=loss_fn,
 233       name=name)

 236 def regression_head(weight_column=None,
 237                     label_dimension=1,
 238                     loss_reduction=losses.Reduction.SUM_OVER_BATCH_SIZE,
 239                     loss_fn=None,
 240                     inverse_link_fn=None,
 241                     name=None):
 242   """Creates a `_Head` for regression using the `mean_squared_error` loss.
 243 
 244   The loss is the weighted sum over all input dimensions. Namely, if the input
 245   labels have shape `[batch_size, label_dimension]`, the loss is the weighted
 246   sum over both `batch_size` and `label_dimension`.
 247 
 248   The head expects `logits` with shape `[D0, D1, ... DN, label_dimension]`.
 249   In many applications, the shape is `[batch_size, label_dimension]`.
 250 
 251   The `labels` shape must match `logits`, namely
 252   `[D0, D1, ... DN, label_dimension]`. If `label_dimension=1`, shape
 253   `[D0, D1, ... DN]` is also supported.
 254 
 255   If `weight_column` is specified, weights must be of shape
 256   `[D0, D1, ... DN]`, `[D0, D1, ... DN, 1]` or
 257   `[D0, D1, ... DN, label_dimension]`.
 258 
 259   Supports custom `loss_fn`. `loss_fn` takes `(labels, logits)` or
 260   `(labels, logits, features)` as arguments and returns unreduced loss with
 261   shape `[D0, D1, ... DN, label_dimension]`.
 262 
 263   Also supports custom `inverse_link_fn`, also known as 'mean function'.
 264   `inverse_link_fn` is only used in `PREDICT` mode. It takes `logits` as
 265   argument and returns predicted values. This function is the inverse of the
 266   link function defined in
 267   https://en.wikipedia.org/wiki/Generalized_linear_model#Link_function
 268   Namely, for poisson regression, set `inverse_link_fn=tf.exp`.
 269 
 270   The head can be used with a canned estimator. Example:
 271 
 272   ```python
 273   my_head = tf.contrib.estimator.regression_head()
 274   my_estimator = tf.contrib.estimator.DNNEstimator(
 275       head=my_head,
 276       hidden_units=...,
 277       feature_columns=...)
 278   ```
 279 
 280   It can also be used with a custom `model_fn`. Example:
 281 
 282   ```python
 283   def _my_model_fn(features, labels, mode):
 284     my_head = tf.contrib.estimator.regression_head()
 285     logits = tf.keras.Model(...)(features)
 286 
 287     return my_head.create_estimator_spec(
 288         features=features,
 289         mode=mode,
 290         labels=labels,
 291         optimizer=tf.AdagradOptimizer(learning_rate=0.1),
 292         logits=logits)
 293 
 294   my_estimator = tf.estimator.Estimator(model_fn=_my_model_fn)
 295   ```
 296 
 297   Args:
 298     weight_column: A string or a `_NumericColumn` created by
 299       `tf.feature_column.numeric_column` defining feature column representing
 300       weights. It is used to down weight or boost examples during training. It
 301       will be multiplied by the loss of the example.
 302     label_dimension: Number of regression labels per example. This is the size
 303       of the last dimension of the labels `Tensor` (typically, this has shape
 304       `[batch_size, label_dimension]`).
 305     loss_reduction: One of `tf.losses.Reduction` except `NONE`. Describes how to
 306       reduce training loss over batch and label dimension. Defaults to
 307       `SUM_OVER_BATCH_SIZE`, namely weighted sum of losses divided by
 308       `batch size * label_dimension`. See `tf.losses.Reduction`.
 309     loss_fn: Optional loss function. Defaults to `mean_squared_error`.
 310     inverse_link_fn: Optional inverse link function, also known as 'mean
 311       function'. Defaults to identity.
 312     name: name of the head. If provided, summary and metrics keys will be
 313       suffixed by `"/" + name`. Also used as `name_scope` when creating ops.
 314 
 315   Returns:
 316     An instance of `_Head` for linear regression.
 317 
 318   Raises:
 319     ValueError: If `label_dimension` or `loss_reduction` is invalid.
 320   """
 321   return head_lib._regression_head(  # pylint:disable=protected-access
 322       weight_column=weight_column,
 323       label_dimension=label_dimension,
 324       loss_reduction=loss_reduction,
 325       loss_fn=loss_fn,
 326       inverse_link_fn=inverse_link_fn,
 327       name=name)

 330 def poisson_regression_head(
 331     weight_column=None,
 332     label_dimension=1,
 333     loss_reduction=losses.Reduction.SUM_OVER_BATCH_SIZE,
 334     compute_full_loss=True,
 335     name=None):
 336   """Creates a `_Head` for poisson regression using `tf.nn.log_poisson_loss`.
 337 
 338   The loss is the weighted sum over all input dimensions. Namely, if the input
 339   labels have shape `[batch_size, label_dimension]`, the loss is the weighted
 340   sum over both `batch_size` and `label_dimension`.
 341 
 342   The head expects `logits` with shape `[D0, D1, ... DN, label_dimension]`.
 343   In many applications, the shape is `[batch_size, label_dimension]`.
 344 
 345   The `labels` shape must match `logits`, namely
 346   `[D0, D1, ... DN, label_dimension]`. If `label_dimension=1`, shape
 347   `[D0, D1, ... DN]` is also supported.
 348 
 349   If `weight_column` is specified, weights must be of shape
 350   `[D0, D1, ... DN]`, `[D0, D1, ... DN, 1]` or
 351   `[D0, D1, ... DN, label_dimension]`.
 352 
 353   This is implemented as a generalized linear model, see
 354   https://en.wikipedia.org/wiki/Generalized_linear_model.
 355 
 356   The head can be used with a canned estimator. Example:
 357 
 358   ```python
 359   my_head = tf.contrib.estimator.poisson_regression_head()
 360   my_estimator = tf.contrib.estimator.DNNEstimator(
 361       head=my_head,
 362       hidden_units=...,
 363       feature_columns=...)
 364   ```
 365 
 366   It can also be used with a custom `model_fn`. Example:
 367 
 368   ```python
 369   def _my_model_fn(features, labels, mode):
 370     my_head = tf.contrib.estimator.poisson_regression_head()
 371     logits = tf.keras.Model(...)(features)
 372 
 373     return my_head.create_estimator_spec(
 374         features=features,
 375         mode=mode,
 376         labels=labels,
 377         optimizer=tf.AdagradOptimizer(learning_rate=0.1),
 378         logits=logits)
 379 
 380   my_estimator = tf.estimator.Estimator(model_fn=_my_model_fn)
 381   ```
 382 
 383   Args:
 384     weight_column: A string or a `_NumericColumn` created by
 385       `tf.feature_column.numeric_column` defining feature column representing
 386       weights. It is used to down weight or boost examples during training. It
 387       will be multiplied by the loss of the example.
 388     label_dimension: Number of regression labels per example. This is the size
 389       of the last dimension of the labels `Tensor` (typically, this has shape
 390       `[batch_size, label_dimension]`).
 391     loss_reduction: One of `tf.losses.Reduction` except `NONE`. Describes how to
 392       reduce training loss over batch and label dimension. Defaults to
 393       `SUM_OVER_BATCH_SIZE`, namely weighted sum of losses divided by
 394       `batch size * label_dimension`. See `tf.losses.Reduction`.
 395     compute_full_loss: Whether to include the constant `log(z!)` term in
 396       computing the poisson loss. See `tf.nn.log_poisson_loss` for the full
 397       documentation.
 398     name: name of the head. If provided, summary and metrics keys will be
 399       suffixed by `"/" + name`. Also used as `name_scope` when creating ops.
 400 
 401   Returns:
 402     An instance of `_Head` for poisson regression.
 403 
 404   Raises:
 405     ValueError: If `label_dimension` or `loss_reduction` is invalid.
 406   """
 407   def _poisson_loss(labels, logits):
 408     return nn.log_poisson_loss(
 409         targets=labels, log_input=logits, compute_full_loss=compute_full_loss)
 410   return head_lib._regression_head(  # pylint:disable=protected-access
 411       weight_column=weight_column,
 412       label_dimension=label_dimension,
 413       loss_reduction=loss_reduction,
 414       loss_fn=_poisson_loss,
 415       inverse_link_fn=math_ops.exp,
 416       name=name)

 419 def logistic_regression_head(
 420     weight_column=None,
 421     loss_reduction=losses.Reduction.SUM_OVER_BATCH_SIZE,
 422     name=None):
 423   """Creates a `_Head` for logistic regression.
 424 
 425   Uses `sigmoid_cross_entropy_with_logits` loss, which is the same as
 426   `binary_classification_head`. The differences compared to
 427   `binary_classification_head` are:
 428 
 429   * Does not support `label_vocabulary`. Instead, labels must be float in the
 430     range [0, 1].
 431   * Does not calculate some metrics that do not make sense, such as AUC.
 432   * In `PREDICT` mode, only returns logits and predictions
 433     (`=tf.sigmoid(logits)`), whereas `binary_classification_head` also returns
 434     probabilities, classes, and class_ids.
 435   * Export output defaults to `RegressionOutput`, whereas
 436     `binary_classification_head` defaults to `PredictOutput`.
 437 
 438   The head expects `logits` with shape `[D0, D1, ... DN, 1]`.
 439   In many applications, the shape is `[batch_size, 1]`.
 440 
 441   The `labels` shape must match `logits`, namely
 442   `[D0, D1, ... DN]` or `[D0, D1, ... DN, 1]`.
 443 
 444   If `weight_column` is specified, weights must be of shape
 445   `[D0, D1, ... DN]` or `[D0, D1, ... DN, 1]`.
 446 
 447   This is implemented as a generalized linear model, see
 448   https://en.wikipedia.org/wiki/Generalized_linear_model.
 449 
 450   The head can be used with a canned estimator. Example:
 451 
 452   ```python
 453   my_head = tf.contrib.estimator.logistic_regression_head()
 454   my_estimator = tf.contrib.estimator.DNNEstimator(
 455       head=my_head,
 456       hidden_units=...,
 457       feature_columns=...)
 458   ```
 459 
 460   It can also be used with a custom `model_fn`. Example:
 461 
 462   ```python
 463   def _my_model_fn(features, labels, mode):
 464     my_head = tf.contrib.estimator.logistic_regression_head()
 465     logits = tf.keras.Model(...)(features)
 466 
 467     return my_head.create_estimator_spec(
 468         features=features,
 469         mode=mode,
 470         labels=labels,
 471         optimizer=tf.AdagradOptimizer(learning_rate=0.1),
 472         logits=logits)
 473 
 474   my_estimator = tf.estimator.Estimator(model_fn=_my_model_fn)
 475   ```
 476 
 477   Args:
 478     weight_column: A string or a `_NumericColumn` created by
 479       `tf.feature_column.numeric_column` defining feature column representing
 480       weights. It is used to down weight or boost examples during training. It
 481       will be multiplied by the loss of the example.
 482     loss_reduction: One of `tf.losses.Reduction` except `NONE`. Describes how to
 483       reduce training loss over batch and label dimension. Defaults to
 484       `SUM_OVER_BATCH_SIZE`, namely weighted sum of losses divided by
 485       `batch size * label_dimension`. See `tf.losses.Reduction`.
 486     name: name of the head. If provided, summary and metrics keys will be
 487       suffixed by `"/" + name`. Also used as `name_scope` when creating ops.
 488 
 489   Returns:
 490     An instance of `_Head` for logistic regression.
 491 
 492   Raises:
 493     ValueError: If `loss_reduction` is invalid.
 494   """
 495   def _logistic_loss(labels, logits):
 496     labels = head_lib._assert_range(  # pylint:disable=protected-access
 497         labels, n_classes=2, message='Labels must be in range [0, 1]')
 498     return nn.sigmoid_cross_entropy_with_logits(
 499         labels=labels, logits=logits)
 500   return head_lib._regression_head(  # pylint:disable=protected-access
 501       weight_column=weight_column,
 502       label_dimension=1,
 503       loss_reduction=loss_reduction,
 504       loss_fn=_logistic_loss,
 505       inverse_link_fn=math_ops.sigmoid,
 506       name=name)

 509 def multi_label_head(n_classes,
 510                      weight_column=None,
 511                      thresholds=None,
 512                      label_vocabulary=None,
 513                      loss_reduction=losses.Reduction.SUM_OVER_BATCH_SIZE,
 514                      loss_fn=None,
 515                      classes_for_class_based_metrics=None,
 516                      name=None):
 517   """Creates a `_Head` for multi-label classification.
 518 
 519   Multi-label classification handles the case where each example may have zero
 520   or more associated labels, from a discrete set. This is distinct from
 521   `multi_class_head` which has exactly one label per example.
 522 
 523   Uses `sigmoid_cross_entropy` loss average over classes and weighted sum over
 524   the batch. Namely, if the input logits have shape `[batch_size, n_classes]`,
 525   the loss is the average over `n_classes` and the weighted sum over
 526   `batch_size`.
 527 
 528   The head expects `logits` with shape `[D0, D1, ... DN, n_classes]`. In many
 529   applications, the shape is `[batch_size, n_classes]`.
 530 
 531   Labels can be:
 532 
 533   * A multi-hot tensor of shape `[D0, D1, ... DN, n_classes]`
 534   * An integer `SparseTensor` of class indices. The `dense_shape` must be
 535     `[D0, D1, ... DN, ?]` and the values within `[0, n_classes)`.
 536   * If `label_vocabulary` is given, a string `SparseTensor`. The `dense_shape`
 537     must be `[D0, D1, ... DN, ?]` and the values within `label_vocabulary` or a
 538     multi-hot tensor of shape `[D0, D1, ... DN, n_classes]`.
 539 
 540   If `weight_column` is specified, weights must be of shape
 541   `[D0, D1, ... DN]`, or `[D0, D1, ... DN, 1]`.
 542 
 543   Also supports custom `loss_fn`. `loss_fn` takes `(labels, logits)` or
 544   `(labels, logits, features)` as arguments and returns unreduced loss with
 545   shape `[D0, D1, ... DN, 1]`. `loss_fn` must support indicator `labels` with
 546   shape `[D0, D1, ... DN, n_classes]`. Namely, the head applies
 547   `label_vocabulary` to the input labels before passing them to `loss_fn`.
 548 
 549   The head can be used with a canned estimator. Example:
 550 
 551   ```python
 552   my_head = tf.contrib.estimator.multi_label_head(n_classes=3)
 553   my_estimator = tf.contrib.estimator.DNNEstimator(
 554       head=my_head,
 555       hidden_units=...,
 556       feature_columns=...)
 557   ```
 558 
 559   It can also be used with a custom `model_fn`. Example:
 560 
 561   ```python
 562   def _my_model_fn(features, labels, mode):
 563     my_head = tf.contrib.estimator.multi_label_head(n_classes=3)
 564     logits = tf.keras.Model(...)(features)
 565 
 566     return my_head.create_estimator_spec(
 567         features=features,
 568         mode=mode,
 569         labels=labels,
 570         optimizer=tf.AdagradOptimizer(learning_rate=0.1),
 571         logits=logits)
 572 
 573   my_estimator = tf.estimator.Estimator(model_fn=_my_model_fn)
 574   ```
 575 
 576   Args:
 577     n_classes: Number of classes, must be greater than 1 (for 1 class, use
 578       `binary_classification_head`).
 579     weight_column: A string or a `_NumericColumn` created by
 580       `tf.feature_column.numeric_column` defining feature column representing
 581       weights. It is used to down weight or boost examples during training. It
 582       will be multiplied by the loss of the example.  Per-class weighting is
 583       not supported.
 584     thresholds: Iterable of floats in the range `(0, 1)`. Accuracy, precision
 585       and recall metrics are evaluated for each threshold value. The threshold
 586       is applied to the predicted probabilities, i.e. above the threshold is
 587       `true`, below is `false`.
 588     label_vocabulary: A list of strings represents possible label values. If it
 589       is not given, that means labels are already encoded as integer within
 590       [0, n_classes) or multi-hot Tensor. If given, labels must be SparseTensor
 591       string type and have any value in `label_vocabulary`. Also there will be
 592       errors if vocabulary is not provided and labels are string.
 593     loss_reduction: One of `tf.losses.Reduction` except `NONE`. Describes how to
 594       reduce training loss over batch. Defaults to `SUM_OVER_BATCH_SIZE`, namely
 595       weighted sum of losses divided by batch size. See `tf.losses.Reduction`.
 596     loss_fn: Optional loss function.
 597     classes_for_class_based_metrics: List of integer class IDs or string class
 598       names for which per-class metrics are evaluated. If integers, all must be
 599       in the range `[0, n_classes - 1]`. If strings, all must be in
 600       `label_vocabulary`.
 601     name: name of the head. If provided, summary and metrics keys will be
 602       suffixed by `"/" + name`. Also used as `name_scope` when creating ops.
 603 
 604   Returns:
 605     An instance of `_Head` for multi-label classification.
 606 
 607   Raises:
 608     ValueError: if `n_classes`, `thresholds`, `loss_reduction`, `loss_fn` or
 609     `metric_class_ids` is invalid.
 610   """
 611   thresholds = tuple(thresholds) if thresholds else tuple()
 612   if n_classes is None or n_classes < 2:
 613     raise ValueError(
 614         'n_classes must be > 1 for multi-class classification. '
 615         'Given: {}'.format(n_classes))
 616   for threshold in thresholds:
 617     if (threshold <= 0.0) or (threshold >= 1.0):
 618       raise ValueError(
 619           'thresholds must be in (0, 1) range. Given: {}'.format(threshold))
 620   if label_vocabulary is not None:
 621     if not isinstance(label_vocabulary, (list, tuple)):
 622       raise ValueError(
 623           'label_vocabulary must be a list or tuple. '
 624           'Given type: {}'.format(type(label_vocabulary)))
 625     if len(label_vocabulary) != n_classes:
 626       raise ValueError(
 627           'Length of label_vocabulary must be n_classes ({}). '
 628           'Given: {}'.format(n_classes, len(label_vocabulary)))
 629   if loss_fn:
 630     head_lib._validate_loss_fn_args(loss_fn)  # pylint:disable=protected-access
 631   if (loss_reduction not in losses.Reduction.all() or
 632       loss_reduction == losses.Reduction.NONE):
 633     raise ValueError('Invalid loss_reduction: {}'.format(loss_reduction))
 634   classes_for_class_based_metrics = tuple(
 635       [] if classes_for_class_based_metrics is None
 636       else classes_for_class_based_metrics)
 637   if classes_for_class_based_metrics:
 638     if isinstance(classes_for_class_based_metrics[0], six.string_types):
 639       if not label_vocabulary:
 640         raise ValueError(
 641             'label_vocabulary must be provided when '
 642             'classes_for_class_based_metrics are sting.')
 643       class_ids = []
 644       for class_string in classes_for_class_based_metrics:
 645         class_ids.append(label_vocabulary.index(class_string))
 646       classes_for_class_based_metrics = tuple(class_ids)
 647     else:
 648       for class_id in classes_for_class_based_metrics:
 649         if (class_id < 0) or (class_id >= n_classes):
 650           raise ValueError(
 651               'All classes_for_class_based_metrics must be in range [0, {}]. '
 652               'Given: {}'.format(n_classes - 1, class_id))
 653   return _MultiLabelHead(
 654       n_classes=n_classes, weight_column=weight_column, thresholds=thresholds,
 655       label_vocabulary=label_vocabulary, loss_reduction=loss_reduction,
 656       loss_fn=loss_fn,
 657       classes_for_class_based_metrics=classes_for_class_based_metrics,
 658       name=name)

 690   def _process_labels(self, labels):
 691     if labels is None:
 692       raise ValueError(
 693           'You must provide a labels Tensor. Given: None. '
 694           'Suggested troubleshooting steps: Check that your data contain '
 695           'your label feature. Check that your input_fn properly parses and '
 696           'returns labels.')
 697     if isinstance(labels, sparse_tensor.SparseTensor):
 698       if labels.dtype == dtypes.string:
 699         label_ids_values = lookup_ops.index_table_from_tensor(
 700             vocabulary_list=tuple(self._label_vocabulary),
 701             name='class_id_lookup').lookup(labels.values)
 702         label_ids = sparse_tensor.SparseTensor(
 703             indices=labels.indices,
 704             values=label_ids_values,
 705             dense_shape=labels.dense_shape)
 706         return math_ops.to_int64(
 707             sparse_ops.sparse_to_indicator(label_ids, self._n_classes))
 708       else:
 709         err_msg = (
 710             r'labels must be an integer SparseTensor with values in '
 711             r'[0, {})'.format(self._n_classes))
 712         assert_int = check_ops.assert_integer(
 713             labels.values, message=err_msg)
 714         assert_less = check_ops.assert_less(
 715             labels.values,
 716             ops.convert_to_tensor(self._n_classes, dtype=labels.dtype),
 717             message=err_msg)
 718         assert_greater = check_ops.assert_non_negative(
 719             labels.values, message=err_msg)
 720         with ops.control_dependencies(
 721             [assert_int, assert_less, assert_greater]):
 722           return math_ops.to_int64(
 723               sparse_ops.sparse_to_indicator(labels, self._n_classes))
 724     err_msg = (
 725         r'labels must be an integer indicator Tensor with values in [0, 1]')
 726     return head_lib._assert_range(labels, 2, message=err_msg)  # pylint:disable=protected-access,

 728   def create_loss(self, features, mode, logits, labels):
 729     """See `Head`."""
 730     del mode  # Unused for this head.
 731     logits = ops.convert_to_tensor(logits)
 732     processed_labels = self._process_labels(labels)
 733     processed_labels = head_lib._check_dense_labels_match_logits_and_reshape(  # pylint:disable=protected-access
 734         labels=processed_labels, logits=logits,
 735         expected_labels_dimension=self.logits_dimension)
 736     if self._loss_fn:
 737       unweighted_loss = head_lib._call_loss_fn(  # pylint:disable=protected-access
 738           loss_fn=self._loss_fn, labels=processed_labels, logits=logits,
 739           features=features, expected_loss_dim=1)
 740     else:
 741       unweighted_loss = losses.sigmoid_cross_entropy(
 742           multi_class_labels=processed_labels, logits=logits,
 743           reduction=losses.Reduction.NONE)
 744       # Averages loss over classes.
 745       unweighted_loss = math_ops.reduce_mean(
 746           unweighted_loss, axis=-1, keepdims=True)
 747     weights = head_lib._get_weights_and_check_match_logits(  # pylint:disable=protected-access,
 748         features=features, weight_column=self._weight_column, logits=logits)
 749     training_loss = losses.compute_weighted_loss(
 750         unweighted_loss, weights=weights, reduction=self._loss_reduction)
 751     return head_lib.LossSpec(
 752         training_loss=training_loss,
 753         unreduced_loss=unweighted_loss,
 754         weights=weights,
 755         processed_labels=processed_labels)

 757   def _create_tpu_estimator_spec(
 758       self, features, mode, logits, labels=None, optimizer=None,
 759       train_op_fn=None, regularization_losses=None):
 760     """Returns an `model_fn._TPUEstimatorSpec`.
 761 
 762     Args:
 763       features: Input `dict` of `Tensor` or `SparseTensor` objects.
 764       mode: Estimator's `ModeKeys`.
 765       logits: logits `Tensor` with shape `[D0, D1, ... DN, n_classes]`.
 766         For many applications, the shape is `[batch_size, n_classes]`.
 767       labels: Labels with shape matching `logits`. Can be multi-hot `Tensor`
 768         with shape `[D0, D1, ... DN, n_classes]` or `SparseTensor` with
 769         `dense_shape` `[D0, D1, ... DN, ?]`. `labels` is required argument when
 770         `mode` equals `TRAIN` or `EVAL`.
 771       optimizer: `Optimizer` instance to optimize the loss in TRAIN mode.
 772         Namely, sets `train_op = optimizer.minimize(loss, global_step)`, which
 773         updates variables and increments `global_step`.
 774       train_op_fn: Function that takes a scalar loss `Tensor` and returns
 775         `train_op`. Used if `optimizer` is `None`.
 776       regularization_losses: A list of additional scalar losses to be added to
 777         the training loss, such as regularization losses. These losses are
 778         usually expressed as a batch average, so for best results users need to
 779         set `loss_reduction=SUM_OVER_BATCH_SIZE` or
 780         `loss_reduction=SUM_OVER_NONZERO_WEIGHTS` when creating the head to
 781         avoid scaling errors.
 782     Returns:
 783       `model_fn._TPUEstimatorSpec`.
 784     Raises:
 785       ValueError: If both `train_op_fn` and `optimizer` are `None` in TRAIN
 786         mode, or if both are set.
 787     """
 788     with ops.name_scope(self._name, 'head'):
 789       logits = head_lib._check_logits_final_dim(logits, self.logits_dimension)  # pylint:disable=protected-access
 790 
 791       # Predict.
 792       pred_keys = prediction_keys.PredictionKeys
 793       with ops.name_scope(None, 'predictions', (logits,)):
 794         probabilities = math_ops.sigmoid(logits, name=pred_keys.PROBABILITIES)
 795         predictions = {
 796             pred_keys.LOGITS: logits,
 797             pred_keys.PROBABILITIES: probabilities,
 798         }
 799       if mode == model_fn.ModeKeys.PREDICT:
 800         classifier_output = head_lib._classification_output(  # pylint:disable=protected-access
 801             scores=probabilities, n_classes=self._n_classes,
 802             label_vocabulary=self._label_vocabulary)
 803         return model_fn._TPUEstimatorSpec(  # pylint:disable=protected-access
 804             mode=model_fn.ModeKeys.PREDICT,
 805             predictions=predictions,
 806             export_outputs={
 807                 _DEFAULT_SERVING_KEY: classifier_output,
 808                 head_lib._CLASSIFY_SERVING_KEY: classifier_output,  # pylint:disable=protected-access
 809                 head_lib._PREDICT_SERVING_KEY: (  # pylint:disable=protected-access
 810                     export_output.PredictOutput(predictions))
 811             })
 812 
 813       (training_loss, unreduced_loss, weights,
 814        processed_labels) = self.create_loss(
 815            features=features, mode=mode, logits=logits, labels=labels)
 816       if regularization_losses:
 817         regularization_loss = math_ops.add_n(regularization_losses)
 818         regularized_training_loss = math_ops.add_n(
 819             [training_loss, regularization_loss])
 820       else:
 821         regularization_loss = None
 822         regularized_training_loss = training_loss
 823 
 824       # Eval.
 825       if mode == model_fn.ModeKeys.EVAL:
 826         return model_fn._TPUEstimatorSpec(  # pylint:disable=protected-access
 827             mode=model_fn.ModeKeys.EVAL,
 828             predictions=predictions,
 829             loss=regularized_training_loss,
 830             eval_metrics=head_lib._create_eval_metrics_tuple(  # pylint:disable=protected-access
 831                 self._eval_metric_ops, {
 832                     'labels': processed_labels,
 833                     'probabilities': probabilities,
 834                     'weights': weights,
 835                     'unreduced_loss': unreduced_loss,
 836                     'regularization_loss': regularization_loss,
 837                 }))
 838 
 839       # Train.
 840       if optimizer is not None:
 841         if train_op_fn is not None:
 842           raise ValueError('train_op_fn and optimizer cannot both be set.')
 843         train_op = optimizer.minimize(
 844             regularized_training_loss,
 845             global_step=training_util.get_global_step())
 846       elif train_op_fn is not None:
 847         train_op = train_op_fn(regularized_training_loss)
 848       else:
 849         raise ValueError('train_op_fn and optimizer cannot both be None.')
 850       train_op = head_lib._append_update_ops(train_op)  # pylint:disable=protected-access
 851       # Only summarize mean_loss for SUM reduction to preserve backwards
 852       # compatibility. Otherwise skip it to avoid unnecessary computation.
 853       if self._loss_reduction == losses.Reduction.SUM:
 854         example_weight_sum = math_ops.reduce_sum(
 855             weights * array_ops.ones_like(unreduced_loss))
 856         mean_loss = training_loss / example_weight_sum
 857       else:
 858         mean_loss = None
 859     with ops.name_scope(''):
 860       keys = metric_keys.MetricKeys
 861       summary.scalar(
 862           head_lib._summary_key(self._name, keys.LOSS),  # pylint:disable=protected-access
 863           regularized_training_loss)
 864       if mean_loss is not None:
 865         summary.scalar(
 866             head_lib._summary_key(self._name, keys.LOSS_MEAN),  # pylint:disable=protected-access
 867             mean_loss)
 868       if regularization_loss is not None:
 869         summary.scalar(
 870             head_lib._summary_key(self._name, keys.LOSS_REGULARIZATION),  # pylint:disable=protected-access
 871             regularization_loss)
 872     return model_fn._TPUEstimatorSpec(  # pylint:disable=protected-access
 873         mode=model_fn.ModeKeys.TRAIN,
 874         predictions=predictions,
 875         loss=regularized_training_loss,
 876         train_op=train_op)

 878   def _eval_metric_ops(
 879       self, labels, probabilities, weights, unreduced_loss,
 880       regularization_loss):
 881     """Returns a dict of metrics for eval_metric_ops."""
 882     with ops.name_scope(
 883         None, 'metrics',
 884         [labels, probabilities, weights, unreduced_loss, regularization_loss]):
 885       keys = metric_keys.MetricKeys
 886       metric_ops = {
 887           # Estimator already adds a metric for loss.
 888           head_lib._summary_key(self._name, keys.LOSS_MEAN):  # pylint:disable=protected-access
 889               metrics_lib.mean(
 890                   values=unreduced_loss,
 891                   weights=weights,
 892                   name=keys.LOSS_MEAN),
 893           head_lib._summary_key(self._name, keys.AUC):  # pylint:disable=protected-access
 894               metrics_lib.auc(labels=labels, predictions=probabilities,
 895                               weights=weights, name=keys.AUC),
 896           head_lib._summary_key(self._name, keys.AUC_PR):  # pylint:disable=protected-access
 897               metrics_lib.auc(labels=labels, predictions=probabilities,
 898                               weights=weights, curve='PR',
 899                               name=keys.AUC_PR),
 900       }
 901       if regularization_loss is not None:
 902         loss_regularization_key = head_lib._summary_key(  # pylint:disable=protected-access
 903             self._name, keys.LOSS_REGULARIZATION)
 904         metric_ops[loss_regularization_key] = (
 905             metrics_lib.mean(
 906                 values=regularization_loss,
 907                 name=keys.LOSS_REGULARIZATION))
 908       for threshold in self._thresholds:
 909         accuracy_key = keys.ACCURACY_AT_THRESHOLD % threshold
 910         metric_ops[head_lib._summary_key(self._name, accuracy_key)] = (  # pylint:disable=protected-access
 911             head_lib._accuracy_at_threshold(  # pylint:disable=protected-access
 912                 labels=labels,
 913                 predictions=probabilities,
 914                 weights=weights,
 915                 threshold=threshold,
 916                 name=accuracy_key))
 917         # Precision for positive examples.
 918         precision_key = keys.PRECISION_AT_THRESHOLD % threshold
 919         metric_ops[head_lib._summary_key(self._name, precision_key)] = (  # pylint:disable=protected-access
 920             head_lib._precision_at_threshold(  # pylint:disable=protected-access
 921                 labels=labels,
 922                 predictions=probabilities,
 923                 weights=weights,
 924                 threshold=threshold,
 925                 name=precision_key))
 926         # Recall for positive examples.
 927         recall_key = keys.RECALL_AT_THRESHOLD % threshold
 928         metric_ops[head_lib._summary_key(self._name, recall_key)] = (  # pylint:disable=protected-access
 929             head_lib._recall_at_threshold(  # pylint:disable=protected-access
 930                 labels=labels,
 931                 predictions=probabilities,
 932                 weights=weights,
 933                 threshold=threshold,
 934                 name=recall_key))
 935       for class_id in self._classes_for_class_based_metrics:
 936         batch_rank = array_ops.rank(probabilities) - 1
 937         begin = array_ops.concat(
 938             [array_ops.zeros([batch_rank], dtype=dtypes.int32), [class_id]],
 939             axis=0)
 940         size = array_ops.concat(
 941             [-1 * array_ops.ones([batch_rank], dtype=dtypes.int32), [1]],
 942             axis=0)
 943         class_probabilities = array_ops.slice(
 944             probabilities, begin=begin, size=size)
 945         class_labels = array_ops.slice(labels, begin=begin, size=size)
 946         if self._label_vocabulary is None:
 947           prob_key = keys.PROBABILITY_MEAN_AT_CLASS % class_id
 948         else:
 949           prob_key = (
 950               keys.PROBABILITY_MEAN_AT_NAME % self._label_vocabulary[class_id])
 951         metric_ops[head_lib._summary_key(self._name, prob_key)] = (  # pylint:disable=protected-access
 952             head_lib._predictions_mean(  # pylint:disable=protected-access
 953                 predictions=class_probabilities,
 954                 weights=weights,
 955                 name=prob_key))
 956         if self._label_vocabulary is None:
 957           auc_key = keys.AUC_AT_CLASS % class_id
 958         else:
 959           auc_key = keys.AUC_AT_NAME % self._label_vocabulary[class_id]
 960         metric_ops[head_lib._summary_key(self._name, auc_key)] = (  # pylint:disable=protected-access
 961             head_lib._auc(  # pylint:disable=protected-access
 962                 labels=class_labels,
 963                 predictions=class_probabilities,
 964                 weights=weights,
 965                 name=auc_key))
 966         if self._label_vocabulary is None:
 967           auc_pr_key = keys.AUC_PR_AT_CLASS % class_id
 968         else:
 969           auc_pr_key = keys.AUC_PR_AT_NAME % self._label_vocabulary[class_id]
 970         metric_ops[head_lib._summary_key(self._name, auc_pr_key)] = (  # pylint:disable=protected-access
 971             head_lib._auc(  # pylint:disable=protected-access
 972                 labels=class_labels,
 973                 predictions=class_probabilities,
 974                 weights=weights,
 975                 curve='PR',
 976                 name=auc_pr_key))
 977     return metric_ops
