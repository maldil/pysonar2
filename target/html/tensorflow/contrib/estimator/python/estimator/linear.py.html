  25 class LinearEstimator(estimator.Estimator):
  26   """An estimator for TensorFlow linear models with user-specified head.
  27 
  28   Example:
  29 
  30   ```python
  31   categorical_column_a = categorical_column_with_hash_bucket(...)
  32   categorical_column_b = categorical_column_with_hash_bucket(...)
  33 
  34   categorical_feature_a_x_categorical_feature_b = crossed_column(...)
  35 
  36   # Estimator using the default optimizer.
  37   estimator = LinearEstimator(
  38       head=tf.contrib.estimator.multi_label_head(n_classes=3),
  39       feature_columns=[categorical_column_a,
  40                        categorical_feature_a_x_categorical_feature_b])
  41 
  42   # Or estimator using an optimizer with a learning rate decay.
  43   estimator = LinearEstimator(
  44       head=tf.contrib.estimator.multi_label_head(n_classes=3),
  45       feature_columns=[categorical_column_a,
  46                        categorical_feature_a_x_categorical_feature_b],
  47       optimizer=lambda: tf.train.FtrlOptimizer(
  48           learning_rate=tf.exponential_decay(
  49               learning_rate=0.1,
  50               global_step=tf.get_global_step(),
  51               decay_steps=10000,
  52               decay_rate=0.96))
  53 
  54   # Or estimator using the FTRL optimizer with regularization.
  55   estimator = LinearEstimator(
  56       head=tf.contrib.estimator.multi_label_head(n_classes=3),
  57       feature_columns=[categorical_column_a,
  58                        categorical_feature_a_x_categorical_feature_b])
  59       optimizer=tf.train.FtrlOptimizer(
  60           learning_rate=0.1,
  61           l1_regularization_strength=0.001
  62       ))
  63 
  64   def input_fn_train: # returns x, y (where y represents label's class index).
  65     ...
  66   estimator.train(input_fn=input_fn_train, steps=100)
  67   def input_fn_eval: # returns x, y (where y represents label's class index).
  68     ...
  69   metrics = estimator.evaluate(input_fn=input_fn_eval, steps=10)
  70   def input_fn_predict: # returns x, None
  71     ...
  72   predictions = estimator.predict(input_fn=input_fn_predict)
  73   ```
  74 
  75   Input of `train` and `evaluate` should have following features,
  76   otherwise there will be a `KeyError`:
  77 
  78   * if `weight_column` is not `None`, a feature with
  79     `key=weight_column` whose value is a `Tensor`.
  80   * for each `column` in `feature_columns`:
  81     - if `column` is a `_CategoricalColumn`, a feature with `key=column.name`
  82       whose `value` is a `SparseTensor`.
  83     - if `column` is a `_WeightedCategoricalColumn`, two features: the first
  84       with `key` the id column name, the second with `key` the weight column
  85       name. Both features' `value` must be a `SparseTensor`.
  86     - if `column` is a `_DenseColumn`, a feature with `key=column.name`
  87       whose `value` is a `Tensor`.
  88 
  89   Loss and predicted output are determined by the specified head.
  90 
  91   @compatibility(eager)
  92   Estimators are not compatible with eager execution.
  93   @end_compatibility
  94   """
  95 
  96   def __init__(self,
  97                head,
  98                feature_columns,
  99                model_dir=None,
 100                optimizer='Ftrl',
 101                config=None,
 102                partitioner=None,
 103                sparse_combiner='sum'):
 104     """Initializes a `LinearEstimator` instance.
 105 
 106     Args:
 107       head: A `_Head` instance constructed with a method such as
 108         `tf.contrib.estimator.multi_label_head`.
 109       feature_columns: An iterable containing all the feature columns used by
 110         the model. All items in the set should be instances of classes derived
 111         from `FeatureColumn`.
 112       model_dir: Directory to save model parameters, graph and etc. This can
 113         also be used to load checkpoints from the directory into a estimator
 114         to continue training a previously saved model.
 115       optimizer: An instance of `tf.Optimizer` used to train the model. Can also
 116         be a string (one of 'Adagrad', 'Adam', 'Ftrl', 'RMSProp', 'SGD'), or
 117         callable. Defaults to FTRL optimizer.
 118       config: `RunConfig` object to configure the runtime settings.
 119       partitioner: Optional. Partitioner for input layer.
 120       sparse_combiner: A string specifying how to reduce if a categorical column
 121         is multivalent.  One of "mean", "sqrtn", and "sum" -- these are
 122         effectively different ways to do example-level normalization, which can
 123         be useful for bag-of-words features. for more details, see
 124         `tf.feature_column.linear_model`.
 125     """
 126     def _model_fn(features, labels, mode, config):
 127       return linear_lib._linear_model_fn(  # pylint: disable=protected-access
 128           features=features,
 129           labels=labels,
 130           mode=mode,
 131           head=head,
 132           feature_columns=tuple(feature_columns or []),
 133           optimizer=optimizer,
 134           partitioner=partitioner,
 135           config=config,
 136           sparse_combiner=sparse_combiner)
 137     super(LinearEstimator, self).__init__(
 138         model_fn=_model_fn, model_dir=model_dir, config=config)
