  72 def read_data(path, max_size=None):
  73     data_set = [[] for _ in _buckets]
  74     data = json.load(open(path, 'r'))
  75     counter = 0
  76     size_max = 0
  77     for pair in data:
  78         post = pair[0]
  79         responses = pair[1]
  80         source_ids = [int(x) for x in post[0]]
  81         for response in responses:
  82             if not max_size or counter < max_size:
  83                 counter += 1
  84                 if counter % 100000 == 0:
  85                     print("    reading data pair %d" % counter)
  86                     sys.stdout.flush()
  87                 target_ids = [int(x) for x in response[0]]
  88                 target_ids.append(data_utils.EOS_ID)
  89                 # size_max = len(source_ids) if len(source_ids) > size_max else size_max
  90                 # size_max = len(target_ids) if len(target_ids) > size_max else size_max
  91                 for bucket_id, (source_size, target_size) in enumerate(_buckets):
  92                     if len(source_ids) < source_size and len(target_ids) < target_size:
  93                         data_set[bucket_id].append([source_ids, target_ids, int(post[1]), int(response[1])])
  94                         break
  95     return data_set

  98 def refine_data(data):
  99     new_data = []
 100     for d in data:
 101         b = []
 102         for e in range(6):
 103             b.append([x for x in d if x[-1] == e])
 104         new_data.append(b)
 105     return new_data

 108 def create_model(session, forward_only, beam_search):
 109     dtype = tf.float16 if FLAGS.use_fp16 else tf.float32
 110     model = seq2seq_model.Seq2SeqModel(
 111         FLAGS.post_vocab_size,
 112         FLAGS.response_vocab_size,
 113         _buckets,
 114         FLAGS.size,
 115         FLAGS.num_layers,
 116         FLAGS.max_gradient_norm,
 117         FLAGS.batch_size,
 118         FLAGS.learning_rate,
 119         FLAGS.learning_rate_decay_factor,
 120         embedding_size=FLAGS.embedding_size,
 121         forward_only=forward_only,
 122         beam_search=beam_search,
 123         beam_size=FLAGS.beam_size,
 124         category=FLAGS.category,
 125         use_emb=FLAGS.use_emb,
 126         use_imemory=FLAGS.use_imemory,
 127         use_ememory=FLAGS.use_ememory,
 128         emotion_size=FLAGS.emotion_size,
 129         imemory_size=FLAGS.imemory_size,
 130         dtype=dtype)
 131     see_variable = True
 132     if see_variable == True:
 133         for i in tf.all_variables():
 134             print(i.name, i.get_shape())
 135     ckpt = tf.train.get_checkpoint_state(FLAGS.train_dir)
 136     pre_ckpt = tf.train.get_checkpoint_state(FLAGS.pretrain_dir)
 137     if ckpt:  # and tf.gfile.Exists(ckpt.model_checkpoint_path+".index"):
 138         if FLAGS.load_model == 0:
 139             print("Reading model parameters from %s" % ckpt.model_checkpoint_path)
 140             model.saver.restore(session, ckpt.model_checkpoint_path)
 141         else:
 142             path = ckpt.model_checkpoint_path[:ckpt.model_checkpoint_path.find('-') + 1] + str(FLAGS.load_model)
 143             print("Reading model parameters from %s" % path)
 144             model.saver.restore(session, path)
 145     else:
 146         if pre_ckpt:
 147             session.run(tf.initialize_variables(model.initial_var))
 148             if FLAGS.pretrain > -1:
 149                 path = pre_ckpt.model_checkpoint_path[:pre_ckpt.model_checkpoint_path.find('-') + 1] + str(
 150                     FLAGS.pretrain)
 151                 print("Reading pretrain model parameters from %s" % path)
 152                 model.pretrain_saver.restore(session, path)
 153             else:
 154                 print("Reading pretrain model parameters from %s" % pre_ckpt.model_checkpoint_path)
 155                 model.pretrain_saver.restore(session, pre_ckpt.model_checkpoint_path)
 156         else:
 157             print("Created model with fresh parameters.")
 158             session.run(tf.initialize_all_variables())
 159             vec_post, vec_response = data_utils.get_data(FLAGS.data_dir, FLAGS.post_vocab_size,
 160                                                          FLAGS.response_vocab_size)
 161             initvec_post = tf.constant(vec_post, dtype=dtype, name='init_wordvector_post')
 162             initvec_response = tf.constant(vec_response, dtype=dtype, name='init_wordvector_response')
 163             embedding_post = [x for x in tf.trainable_variables() if
 164                               x.name == 'embedding_attention_seq2seq/RNN/EmbeddingWrapper/embedding:0'][0]
 165             embedding_response = [x for x in tf.trainable_variables() if
 166                                   x.name == 'embedding_attention_seq2seq/embedding_attention_decoder/embedding:0'][0]
 167             session.run(embedding_post.assign(initvec_post))
 168             session.run(embedding_response.assign(initvec_response))
 169         if FLAGS.use_ememory:
 170             vec_ememory = data_utils.get_ememory(FLAGS.data_dir, FLAGS.response_vocab_size)
 171             initvec_ememory = tf.constant(vec_ememory, dtype=dtype, name='init_ememory')
 172             ememory = [x for x in tf.all_variables() if
 173                        x.name == 'embedding_attention_seq2seq/embedding_attention_decoder/external_memory:0'][0]
 174             session.run(ememory.assign(initvec_ememory))
 175     return model

 178 def train():
 179     print(FLAGS.__flags)
 180     # Prepare data.
 181     print("Preparing data in %s" % FLAGS.data_dir)
 182     train_path, dev_path, test_path, _, _ = data_utils.prepare_data(
 183         FLAGS.data_dir, FLAGS.post_vocab_size, FLAGS.response_vocab_size)
 184 
 185     with tf.Session(config=sess_config) as sess:
 186         # Create model.
 187         print("Creating %d layers of %d units." % (FLAGS.num_layers, FLAGS.size))
 188         model = create_model(sess, False, False)
 189 
 190         # Read data into buckets and compute their sizes.
 191         print("Reading development and training data (limit: %d)."
 192               % FLAGS.max_train_data_size)
 193         dev_set = read_data(dev_path)
 194         dev_set = refine_data(dev_set)
 195         train_set = read_data(train_path, FLAGS.max_train_data_size)
 196         train_bucket_sizes = [len(train_set[b]) for b in xrange(len(_buckets))]
 197         train_total_size = float(sum(train_bucket_sizes))
 198         print([len(x) for x in dev_set])
 199         print([len(x) for x in train_set])
 200         # A bucket scale is a list of increasing numbers from 0 to 1 that we'll use
 201         # to select a bucket. Length of [scale[i], scale[i+1]] is proportional to
 202         # the size if i-th training bucket, as used later.
 203         train_buckets_scale = [sum(train_bucket_sizes[:i + 1]) / train_total_size
 204                                for i in xrange(len(train_bucket_sizes))]
 205         print(train_buckets_scale)
 206 
 207         # This is the training loop.
 208         step_time, loss = 0.0, 0.0
 209         current_step = 0
 210         epoch_steps = 4400000 / FLAGS.batch_size
 211         previous_losses = []
 212         while True:
 213             # Choose a bucket according to data distribution. We pick a random number
 214             # in [0, 1] and use the corresponding interval in train_buckets_scale.
 215             random_number_01 = np.random.random_sample()
 216             bucket_id = min([i for i in xrange(len(train_buckets_scale))
 217                              if train_buckets_scale[i] > random_number_01])
 218 
 219             # Get a batch and make a step.
 220             start_time = time.time()
 221             encoder_inputs, decoder_inputs, target_weights, decoder_emotions = model.get_batch(
 222                 train_set, bucket_id)
 223             _, step_loss, _ = model.step(sess, encoder_inputs, decoder_inputs,
 224                                          target_weights, decoder_emotions, bucket_id, False, False)
 225             step_time += (time.time() - start_time) / FLAGS.steps_per_checkpoint
 226             loss += step_loss / FLAGS.steps_per_checkpoint
 227             current_step += 1
 228 
 229             # Once in a while, we save checkpoint, print statistics, and run evals.
 230             if current_step % FLAGS.steps_per_checkpoint == 0:
 231                 # Print statistics for the previous epoch.
 232                 perplexity = math.exp(float(loss)) if loss < 300 else float("inf")
 233                 print("global step %d (%.2f epoch) learning rate %.4f step-time %.2f perplexity "
 234                       "%.2f" % (model.global_step.eval(), model.global_step.eval() / float(epoch_steps),
 235                                 model.learning_rate.eval(), step_time, perplexity))
 236                 # Decrease learning rate if no improvement was seen over last 3 times.
 237                 if len(previous_losses) > 2 and loss > max(previous_losses[-3:]):
 238                     sess.run(model.learning_rate_decay_op)
 239                 previous_losses.append(loss)
 240                 # Save checkpoint and zero timer and loss.
 241                 if current_step % (FLAGS.steps_per_checkpoint * 10) == 0 or current_step % 34000 == 0:
 242                     checkpoint_path = os.path.join(FLAGS.train_dir, "translate.ckpt")
 243                     model.saver.save(sess, checkpoint_path, global_step=model.global_step)
 244                 step_time, loss = 0.0, 0
 245                 # dev set evaluation
 246                 total_loss = .0
 247                 total_len = .0
 248                 for bucket_id in xrange(len(_buckets)):
 249                     if len(dev_set[bucket_id]) == 0:
 250                         print("    eval: empty bucket %d" % (bucket_id))
 251                         continue
 252                     bucket_loss = .0
 253                     bucket_len = .0
 254                     for e in range(6):
 255                         len_data = len(dev_set[bucket_id][e])
 256                         for batch in xrange(0, len_data, FLAGS.batch_size):
 257                             step = min(FLAGS.batch_size, len_data - batch)
 258                             model.batch_size = step
 259                             encoder_inputs, decoder_inputs, target_weights, decoder_emotions = model.get_batch_data(
 260                                 dev_set[bucket_id][e][batch:batch + step], bucket_id)
 261                             _, eval_loss, _ = model.step(sess, encoder_inputs, decoder_inputs,
 262                                                          target_weights, decoder_emotions, bucket_id, True, False)
 263                             bucket_loss += eval_loss * step
 264                         bucket_len += len_data
 265                     total_loss += bucket_loss
 266                     total_len += bucket_len
 267                     bucket_loss = float(bucket_loss / bucket_len)
 268                     bucket_ppx = math.exp(bucket_loss) if bucket_loss < 300 else float(
 269                         "inf")
 270                     print("    dev_set eval: bucket %d perplexity %.2f" % (bucket_id, bucket_ppx))
 271                 total_loss = float(total_loss / total_len)
 272                 total_ppx = math.exp(total_loss) if total_loss < 300 else float(
 273                     "inf")
 274                 print("    dev_set eval: bucket avg perplexity %.2f" % (total_ppx))
 275                 sys.stdout.flush()
 276                 model.batch_size = FLAGS.batch_size

 279 def decode():
 280     try:
 281         from wordseg_python import Global
 282     except:
 283         Global = None
 284 
 285     def split(sent):
 286         sent = sent.decode('utf-8', 'ignore').encode('gbk', 'ignore')
 287         if Global == None:
 288             return sent.decode("gbk").split(' ')
 289         tuples = [(word.decode("gbk"), pos) for word, pos in Global.GetTokenPos(sent)]
 290         return [each[0] for each in tuples]
 291 
 292     with tf.Session(config=sess_config) as sess:
 293         with tf.device("/cpu:0"):
 294             # Create model and load parameters.
 295             model = create_model(sess, True, FLAGS.beam_search)
 296             model.batch_size = 1  # We decode one sentence at a time.
 297             beam_search = FLAGS.beam_search
 298             beam_size = FLAGS.beam_size
 299             num_output = 5
 300 
 301             # Load vocabularies.
 302             post_vocab_path = os.path.join(FLAGS.data_dir,
 303                                            config.get('data', 'post_vocab_file') % FLAGS.post_vocab_size)
 304             response_vocab_path = os.path.join(FLAGS.data_dir,
 305                                                config.get('data', 'response_vocab_file') % FLAGS.response_vocab_size)
 306             post_vocab, _ = data_utils.initialize_vocabulary(post_vocab_path)
 307             _, rev_response_vocab = data_utils.initialize_vocabulary(response_vocab_path)
 308 
 309             # Decode from standard input.
 310             sys.stdout.write("> ")
 311             sys.stdout.flush()
 312             sentence = sys.stdin.readline()
 313             while sentence:
 314                 sentence = " ".join(split(sentence))
 315                 # Get token-ids for the input sentence.
 316                 token_ids = data_utils.sentence_to_token_ids(sentence, post_vocab)
 317                 int2emotion = ['null', 'like', 'sad', 'disgust', 'angry', 'happy']
 318                 for decoder_emotion in range(1, 6):
 319                     bucket_id = min([b for b in xrange(len(_buckets))
 320                                      if _buckets[b][0] > len(token_ids)])
 321                     # Get a 1-element batch to feed the sentence to the model.
 322                     encoder_inputs, decoder_inputs, target_weights, decoder_emotions = model.get_batch_data(
 323                         [[token_ids, [], 0, decoder_emotion]], bucket_id)
 324                     # Get output logits for the sentence.
 325                     results, _, output_logits = model.step(sess, encoder_inputs, decoder_inputs,
 326                                                            target_weights, decoder_emotions, bucket_id, True,
 327                                                            beam_search)
 328                     if beam_search:
 329                         result = results[0]
 330                         symbol = results[1]
 331                         parent = results[2]
 332                         result = results[0]
 333                         symbol = results[1]
 334                         parent = results[2]
 335                         res = []
 336                         nounk = []
 337                         for i, (prb, _, prt) in enumerate(result):
 338                             if len(prb) == 0: continue
 339                             for j in xrange(len(prb)):
 340                                 p = prt[j]
 341                                 s = -1
 342                                 output = []
 343                                 for step in xrange(i - 1, -1, -1):
 344                                     s = symbol[step][p]
 345                                     p = parent[step][p]
 346                                     output.append(s)
 347                                 output.reverse()
 348                                 if data_utils.UNK_ID in output:
 349                                     res.append([prb[j][0], " ".join(
 350                                         [tf.compat.as_str(rev_response_vocab[int(x)]) for x in output])])
 351                                 else:
 352                                     nounk.append([prb[j][0], " ".join(
 353                                         [tf.compat.as_str(rev_response_vocab[int(x)]) for x in output])])
 354                         res.sort(key=lambda x: x[0], reverse=True)
 355                         nounk.sort(key=lambda x: x[0], reverse=True)
 356                         if len(nounk) < beam_size:
 357                             res = nounk + res[:(num_output - len(nounk))]
 358                         else:
 359                             res = nounk
 360                         for i in res[:num_output]:
 361                             print(int2emotion[decoder_emotion] + ': ' + i[1])
 362                     else:
 363 
 364                         # This is a greedy decoder - outputs are just argmaxes of output_logits.
 365                         outputs = [
 366                             int(np.argmax(np.split(logit, [2, FLAGS.response_vocab_size], axis=1)[1], axis=1) + 2) for
 367                             logit in output_logits]
 368                         # If there is an EOS symbol in outputs, cut them at that point.
 369                         if data_utils.EOS_ID in outputs:
 370                             outputs = outputs[:outputs.index(data_utils.EOS_ID)]
 371                         # Print out response sentence corresponding to outputs.
 372                         print(int2emotion[decoder_emotion] + ': ' + "".join(
 373                             [tf.compat.as_str(rev_response_vocab[output]) for output in outputs]))
 374                 print("> ", end="")
 375                 sys.stdout.flush()
 376                 sentence = sys.stdin.readline()
